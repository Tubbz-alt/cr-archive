 172 void SystemDictionary::compute_java_loaders(TRAPS) {
 173   JavaValue result(T_OBJECT);
 174   InstanceKlass* class_loader_klass = SystemDictionary::ClassLoader_klass();
 175   JavaCalls::call_static(&result,
 176                          class_loader_klass,
 177                          vmSymbols::getSystemClassLoader_name(),
 178                          vmSymbols::void_classloader_signature(),
 179                          CHECK);
 180 
 181   _java_system_loader = OopHandle::create((oop)result.get_jobject());
 182 
 183   JavaCalls::call_static(&result,
 184                          class_loader_klass,
 185                          vmSymbols::getPlatformClassLoader_name(),
 186                          vmSymbols::void_classloader_signature(),
 187                          CHECK);
 188 
 189   _java_platform_loader = OopHandle::create((oop)result.get_jobject());
 190 }

 242 Symbol* SystemDictionary::class_name_symbol(const char* name, Symbol* exception, TRAPS) {
 243   if (name == NULL) {
 244     THROW_MSG_0(exception, "No class name given");
 245   }
 246   if ((int)strlen(name) > Symbol::max_length()) {
 247     // It's impossible to create this class;  the name cannot fit
 248     // into the constant pool.
 249     Exceptions::fthrow(THREAD_AND_LOCATION, exception,
 250                        "Class name exceeds maximum length of %d: %s",
 251                        Symbol::max_length(),
 252                        name);
 253     return NULL;
 254   }
 255   // Callers should ensure that the name is never an illegal UTF8 string.
 256   assert(UTF8::is_legal_utf8((const unsigned char*)name, (int)strlen(name), false),
 257          "Class name is not a valid utf8 string.");
 258 
 259   // Make a new symbol for the class name.
 260   return SymbolTable::new_symbol(name);
 261 }

 265 Klass* SystemDictionary::resolve_or_fail(Symbol* class_name, Handle class_loader, Handle protection_domain, bool throw_error, TRAPS) {
 266   Klass* klass = resolve_or_null(class_name, class_loader, protection_domain, THREAD);
 267   if (HAS_PENDING_EXCEPTION || klass == NULL) {
 268     // can return a null klass
 269     klass = handle_resolution_exception(class_name, throw_error, klass, THREAD);
 270   }
 271   return klass;
 272 }

 274 Klass* SystemDictionary::handle_resolution_exception(Symbol* class_name,
 275                                                      bool throw_error,
 276                                                      Klass* klass, TRAPS) {
 277   if (HAS_PENDING_EXCEPTION) {
 278     // If we have a pending exception we forward it to the caller, unless throw_error is true,
 279     // in which case we have to check whether the pending exception is a ClassNotFoundException,
 280     // and if so convert it to a NoClassDefFoundError
 281     // And chain the original ClassNotFoundException
 282     if (throw_error && PENDING_EXCEPTION->is_a(SystemDictionary::ClassNotFoundException_klass())) {
 283       ResourceMark rm(THREAD);
 284       assert(klass == NULL, "Should not have result with exception pending");
 285       Handle e(THREAD, PENDING_EXCEPTION);
 286       CLEAR_PENDING_EXCEPTION;
 287       THROW_MSG_CAUSE_NULL(vmSymbols::java_lang_NoClassDefFoundError(), class_name->as_C_string(), e);
 288     } else {
 289       return NULL;
 290     }
 291   }
 292   // Class not found, throw appropriate error or exception depending on value of throw_error
 293   if (klass == NULL) {
 294     ResourceMark rm(THREAD);
 295     if (throw_error) {
 296       THROW_MSG_NULL(vmSymbols::java_lang_NoClassDefFoundError(), class_name->as_C_string());
 297     } else {
 298       THROW_MSG_NULL(vmSymbols::java_lang_ClassNotFoundException(), class_name->as_C_string());
 299     }
 300   }
 301   return klass;
 302 }

 322 // name may be in the form of "java/lang/Object" or "Ljava/lang/Object;"
 323 InstanceKlass* SystemDictionary::resolve_instance_class_or_null_helper(Symbol* class_name,
 324                                                                        Handle class_loader,
 325                                                                        Handle protection_domain,
 326                                                                        TRAPS) {
 327   assert(class_name != NULL && !Signature::is_array(class_name), "must be");
 328   if (Signature::has_envelope(class_name)) {
 329     ResourceMark rm(THREAD);
 330     // Ignore wrapping L and ;. (and Q and ; for value types);
 331     TempNewSymbol name = SymbolTable::new_symbol(class_name->as_C_string() + 1,
 332                                                  class_name->utf8_length() - 2);
 333     return resolve_instance_class_or_null(name, class_loader, protection_domain, THREAD);
 334   } else {
 335     return resolve_instance_class_or_null(class_name, class_loader, protection_domain, THREAD);
 336   }
 337 }

 343 // Forwards to resolve_instance_class_or_null
 344 
 345 Klass* SystemDictionary::resolve_array_class_or_null(Symbol* class_name,
 346                                                      Handle class_loader,
 347                                                      Handle protection_domain,
 348                                                      TRAPS) {
 349   assert(Signature::is_array(class_name), "must be array");
 350   ResourceMark rm(THREAD);
 351   SignatureStream ss(class_name, false);
 352   int ndims = ss.skip_array_prefix();  // skip all '['s
 353   Klass* k = NULL;
 354   BasicType t = ss.type();
 355   if (ss.has_envelope()) {
 356     Symbol* obj_class = ss.as_symbol();
 357     k = SystemDictionary::resolve_instance_class_or_null(obj_class,
 358                                                          class_loader,
 359                                                          protection_domain,
 360                                                          CHECK_NULL);
 361     if (k != NULL) {
 362       k = k->array_klass(ndims, CHECK_NULL);
 363     }
 364   } else {
 365     k = Universe::typeArrayKlassObj(t);
 366     k = TypeArrayKlass::cast(k)->array_klass(ndims, CHECK_NULL);
 367   }
 368   return k;
 369 }

 371 // Must be called for any super-class or super-interface resolution
 372 // during class definition to allow class circularity checking
 373 // super-interface callers:
 374 //    parse_interfaces - for defineClass & jvmtiRedefineClasses
 375 // super-class callers:
 376 //   ClassFileParser - for defineClass & jvmtiRedefineClasses
 377 //   load_shared_class - while loading a class from shared archive
 378 //   resolve_instance_class_or_null:
 379 //     via: handle_parallel_super_load
 380 //      when resolving a class that has an existing placeholder with
 381 //      a saved superclass [i.e. a defineClass is currently in progress]
 382 //      if another thread is trying to resolve the class, it must do
 383 //      super-class checks on its own thread to catch class circularity
 384 // This last call is critical in class circularity checking for cases
 385 // where classloading is delegated to different threads and the
 386 // classloader lock is released.
 387 // Take the case: Base->Super->Base
 388 //   1. If thread T1 tries to do a defineClass of class Base
 389 //    resolve_super_or_fail creates placeholder: T1, Base (super Super)
 390 //   2. resolve_instance_class_or_null does not find SD or placeholder for Super
 391 //    so it tries to load Super
 392 //   3. If we load the class internally, or user classloader uses same thread
 393 //      loadClassFromxxx or defineClass via parseClassFile Super ...
 394 //      3.1 resolve_super_or_fail creates placeholder: T1, Super (super Base)
 395 //      3.3 resolve_instance_class_or_null Base, finds placeholder for Base
 396 //      3.4 calls resolve_super_or_fail Base
 397 //      3.5 finds T1,Base -> throws class circularity
 398 //OR 4. If T2 tries to resolve Super via defineClass Super ...
 399 //      4.1 resolve_super_or_fail creates placeholder: T2, Super (super Base)
 400 //      4.2 resolve_instance_class_or_null Base, finds placeholder for Base (super Super)
 401 //      4.3 calls resolve_super_or_fail Super in parallel on own thread T2
 402 //      4.4 finds T2, Super -> throws class circularity
 403 // Must be called, even if superclass is null, since this is
 404 // where the placeholder entry is created which claims this
 405 // thread is loading this class/classloader.
 406 // Be careful when modifying this code: once you have run
 407 // placeholders()->find_and_add(PlaceholderTable::LOAD_SUPER),
 408 // you need to find_and_remove it before returning.
 409 // So be careful to not exit with a CHECK_ macro betweeen these calls.
 410 InstanceKlass* SystemDictionary::resolve_super_or_fail(Symbol* child_name,
 411                                                        Symbol* super_name,
 412                                                        Handle class_loader,
 413                                                        Handle protection_domain,
 414                                                        bool is_superclass,
 415                                                        TRAPS) {
 416   assert(!Signature::is_array(super_name), "invalid super class name");
 417 #if INCLUDE_CDS
 418   if (DumpSharedSpaces) {
 419     // Special processing for handling UNREGISTERED shared classes.
 420     InstanceKlass* k = SystemDictionaryShared::dump_time_resolve_super_or_fail(child_name,
 421         super_name, class_loader, protection_domain, is_superclass, CHECK_NULL);
 422     if (k) {
 423       return k;
 424     }
 425   }
 426 #endif // INCLUDE_CDS
 427 
 428   // Double-check, if child class is already loaded, just return super-class,interface
 429   // Don't add a placedholder if already loaded, i.e. already in appropriate class loader
 430   // dictionary.
 431   // Make sure there's a placeholder for the *child* before resolving.
 432   // Used as a claim that this thread is currently loading superclass/classloader
 433   // Used here for ClassCircularity checks and also for heap verification
 434   // (every InstanceKlass needs to be in its class loader dictionary or have a placeholder).
 435   // Must check ClassCircularity before checking if super class is already loaded.
 436   //
 437   // We might not already have a placeholder if this child_name was
 438   // first seen via resolve_from_stream (jni_DefineClass or JVM_DefineClass);
 439   // the name of the class might not be known until the stream is actually
 440   // parsed.
 441   // Bugs 4643874, 4715493
 442 
 443   ClassLoaderData* loader_data = class_loader_data(class_loader);
 444   Dictionary* dictionary = loader_data->dictionary();
 445   unsigned int d_hash = dictionary->compute_hash(child_name);
 446   unsigned int p_hash = placeholders()->compute_hash(child_name);
 447   int p_index = placeholders()->hash_to_index(p_hash);
 448   // can't throw error holding a lock
 449   bool child_already_loaded = false;
 450   bool throw_circularity_error = false;
 451   {
 452     MutexLocker mu(THREAD, SystemDictionary_lock);
 453     InstanceKlass* childk = find_class(d_hash, child_name, dictionary);
 454     InstanceKlass* quicksuperk;
 455     // to support // loading: if child done loading, just return superclass
 456     // if super_name, & class_loader don't match:
 457     // if initial define, SD update will give LinkageError
 458     // if redefine: compare_class_versions will give HIERARCHY_CHANGED
 459     // so we don't throw an exception here.
 460     // see: nsk redefclass014 & java.lang.instrument Instrument032
 461     if ((childk != NULL ) && (is_superclass) &&
 462         ((quicksuperk = childk->java_super()) != NULL) &&
 463          ((quicksuperk->name() == super_name) &&
 464             (quicksuperk->class_loader() == class_loader()))) {
 465            return quicksuperk;
 466     } else {
 467       PlaceholderEntry* probe = placeholders()->get_entry(p_index, p_hash, child_name, loader_data);
 468       if (probe && probe->check_seen_thread(THREAD, PlaceholderTable::LOAD_SUPER)) {
 469           throw_circularity_error = true;
 470       }
 471     }
 472     if (!throw_circularity_error) {
 473       // Be careful not to exit resolve_super
 474       PlaceholderEntry* newprobe = placeholders()->find_and_add(p_index, p_hash, child_name, loader_data, PlaceholderTable::LOAD_SUPER, super_name, THREAD);
 475     }
 476   }
 477   if (throw_circularity_error) {
 478       ResourceMark rm(THREAD);
 479       THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), child_name->as_C_string());
 480   }
 481 
 482 // java.lang.Object should have been found above
 483   assert(super_name != NULL, "null super class for resolving");
 484   // Resolve the super class or interface, check results on return
 485   InstanceKlass* superk =
 486     SystemDictionary::resolve_instance_class_or_null_helper(super_name,
 487                                                             class_loader,
 488                                                             protection_domain,
 489                                                             THREAD);
 490 
 491   // Clean up of placeholders moved so that each classloadAction registrar self-cleans up
 492   // It is no longer necessary to keep the placeholder table alive until update_dictionary
 493   // or error. GC used to walk the placeholder table as strong roots.
 494   // The instanceKlass is kept alive because the class loader is on the stack,
 495   // which keeps the loader_data alive, as well as all instanceKlasses in
 496   // the loader_data. parseClassFile adds the instanceKlass to loader_data.
 497   {
 498     MutexLocker mu(THREAD, SystemDictionary_lock);
 499     placeholders()->find_and_remove(p_index, p_hash, child_name, loader_data, PlaceholderTable::LOAD_SUPER, THREAD);
 500     SystemDictionary_lock->notify_all();
 501   }
 502   if (HAS_PENDING_EXCEPTION || superk == NULL) {
 503     // can null superk
 504     Klass* k = handle_resolution_exception(super_name, true, superk, THREAD);
 505     assert(k == NULL || k == superk, "must be");
 506     if (k == NULL) {
 507       superk = NULL;
 508     }
 509   }
 510 
 511   return superk;
 512 }

 514 Klass* SystemDictionary::resolve_flattenable_field_or_fail(AllFieldStream* fs,
 515                                                            Handle class_loader,
 516                                                            Handle protection_domain,
 517                                                            bool throw_error,
 518                                                            TRAPS) {
 519   Symbol* class_name = fs->signature()->fundamental_name(THREAD);
 520   class_loader = Handle(THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 521   ClassLoaderData* loader_data = class_loader_data(class_loader);
 522   unsigned int p_hash = placeholders()->compute_hash(class_name);
 523   int p_index = placeholders()->hash_to_index(p_hash);
 524   bool throw_circularity_error = false;
 525   PlaceholderEntry* oldprobe;
 526 
 527   {
 528     MutexLocker mu(THREAD, SystemDictionary_lock);
 529     oldprobe = placeholders()->get_entry(p_index, p_hash, class_name, loader_data);
 530     if (oldprobe != NULL &&
 531       oldprobe->check_seen_thread(THREAD, PlaceholderTable::FLATTENABLE_FIELD)) {
 532       throw_circularity_error = true;
 533 
 534     } else {
 535       placeholders()->find_and_add(p_index, p_hash, class_name, loader_data,
 536                                    PlaceholderTable::FLATTENABLE_FIELD, NULL, THREAD);
 537     }
 538   }
 539 
 540   Klass* klass = NULL;
 541   if (!throw_circularity_error) {
 542     klass = SystemDictionary::resolve_or_fail(class_name, class_loader,
 543                                                protection_domain, true, THREAD);
 544   } else {
 545     ResourceMark rm(THREAD);
 546     THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), class_name->as_C_string());
 547   }
 548 
 549   {
 550     MutexLocker mu(THREAD, SystemDictionary_lock);
 551     placeholders()->find_and_remove(p_index, p_hash, class_name, loader_data,
 552                                     PlaceholderTable::FLATTENABLE_FIELD, THREAD);
 553   }
 554 
 555   class_name->decrement_refcount();
 556   return klass;
 557 }

 559 void SystemDictionary::validate_protection_domain(InstanceKlass* klass,
 560                                                   Handle class_loader,
 561                                                   Handle protection_domain,
 562                                                   TRAPS) {
 563   // Now we have to call back to java to check if the initating class has access
 564   JavaValue result(T_VOID);
 565   LogTarget(Debug, protectiondomain) lt;
 566   if (lt.is_enabled()) {
 567     ResourceMark rm(THREAD);
 568     // Print out trace information
 569     LogStream ls(lt);
 570     ls.print_cr("Checking package access");
 571     if (class_loader() != NULL) {
 572       ls.print("class loader: ");

 626 // We only get here if this thread finds that another thread
 627 // has already claimed the placeholder token for the current operation,
 628 // but that other thread either never owned or gave up the
 629 // object lock
 630 // Waits on SystemDictionary_lock to indicate placeholder table updated
 631 // On return, caller must recheck placeholder table state
 632 //
 633 // We only get here if
 634 //  1) custom classLoader, i.e. not bootstrap classloader
 635 //  2) custom classLoader has broken the class loader objectLock
 636 //     so another thread got here in parallel
 637 //
 638 // lockObject must be held.
 639 // Complicated dance due to lock ordering:
 640 // Must first release the classloader object lock to
 641 // allow initial definer to complete the class definition
 642 // and to avoid deadlock
 643 // Reclaim classloader lock object with same original recursion count
 644 // Must release SystemDictionary_lock after notify, since
 645 // class loader lock must be claimed before SystemDictionary_lock
 646 // to prevent deadlocks
 647 //
 648 // The notify allows applications that did an untimed wait() on
 649 // the classloader object lock to not hang.
 650 void SystemDictionary::double_lock_wait(Handle lockObject, TRAPS) {
 651   assert_lock_strong(SystemDictionary_lock);
 652 
 653   bool calledholdinglock
 654       = ObjectSynchronizer::current_thread_holds_lock((JavaThread*)THREAD, lockObject);
 655   assert(calledholdinglock,"must hold lock for notify");
 656   assert((lockObject() != _system_loader_lock_obj.resolve() &&
 657          !is_parallelCapable(lockObject)), "unexpected double_lock_wait");
 658   ObjectSynchronizer::notifyall(lockObject, THREAD);
 659   intx recursions =  ObjectSynchronizer::complete_exit(lockObject, THREAD);
 660   SystemDictionary_lock->wait();
 661   SystemDictionary_lock->unlock();
 662   ObjectSynchronizer::reenter(lockObject, recursions, THREAD);
 663   SystemDictionary_lock->lock();
 664 }

 666 // If the class in is in the placeholder table, class loading is in progress
 667 // For cases where the application changes threads to load classes, it
 668 // is critical to ClassCircularity detection that we try loading
 669 // the superclass on the same thread internally, so we do parallel
 670 // super class loading here.
 671 // This also is critical in cases where the original thread gets stalled
 672 // even in non-circularity situations.
 673 // Note: must call resolve_super_or_fail even if null super -
 674 // to force placeholder entry creation for this class for circularity detection
 675 // Caller must check for pending exception
 676 // Returns non-null Klass* if other thread has completed load
 677 // and we are done,
 678 // If return null Klass* and no pending exception, the caller must load the class
 679 InstanceKlass* SystemDictionary::handle_parallel_super_load(
 680     Symbol* name, Symbol* superclassname, Handle class_loader,
 681     Handle protection_domain, Handle lockObject, TRAPS) {
 682 
 683   ClassLoaderData* loader_data = class_loader_data(class_loader);
 684   Dictionary* dictionary = loader_data->dictionary();
 685   unsigned int d_hash = dictionary->compute_hash(name);
 686   unsigned int p_hash = placeholders()->compute_hash(name);
 687   int p_index = placeholders()->hash_to_index(p_hash);
 688 
 689   // superk is not used, resolve_super called for circularity check only
 690   // This code is reached in two situations. One if this thread
 691   // is loading the same class twice (e.g. ClassCircularity, or
 692   // java.lang.instrument).
 693   // The second is if another thread started the resolve_super first
 694   // and has not yet finished.
 695   // In both cases the original caller will clean up the placeholder
 696   // entry on error.
 697   Klass* superk = SystemDictionary::resolve_super_or_fail(name,
 698                                                           superclassname,
 699                                                           class_loader,
 700                                                           protection_domain,
 701                                                           true,
 702                                                           CHECK_NULL);
 703 
 704   // parallelCapable class loaders do NOT wait for parallel superclass loads to complete
 705   // Serial class loaders and bootstrap classloader do wait for superclass loads
 706  if (!class_loader.is_null() && is_parallelCapable(class_loader)) {
 707     MutexLocker mu(THREAD, SystemDictionary_lock);
 708     // Check if classloading completed while we were loading superclass or waiting
 709     return find_class(d_hash, name, dictionary);
 710   }
 711 
 712   // must loop to both handle other placeholder updates
 713   // and spurious notifications
 714   bool super_load_in_progress = true;
 715   PlaceholderEntry* placeholder;
 716   while (super_load_in_progress) {
 717     MutexLocker mu(THREAD, SystemDictionary_lock);
 718     // Check if classloading completed while we were loading superclass or waiting
 719     InstanceKlass* check = find_class(d_hash, name, dictionary);
 720     if (check != NULL) {
 721       // Klass is already loaded, so just return it
 722       return check;
 723     } else {
 724       placeholder = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 725       if (placeholder && placeholder->super_load_in_progress() ){
 726         // We only get here if the application has released the
 727         // classloader lock when another thread was in the middle of loading a
 728         // superclass/superinterface for this class, and now
 729         // this thread is also trying to load this class.
 730         // To minimize surprises, the first thread that started to
 731         // load a class should be the one to complete the loading
 732         // with the classfile it initially expected.
 733         // This logic has the current thread wait once it has done
 734         // all the superclass/superinterface loading it can, until
 735         // the original thread completes the class loading or fails
 736         // If it completes we will use the resulting InstanceKlass
 737         // which we will find below in the systemDictionary.
 738         // We also get here for parallel bootstrap classloader
 739         if (class_loader.is_null()) {
 740           SystemDictionary_lock->wait();
 741         } else {
 742           double_lock_wait(lockObject, THREAD);
 743         }
 744       } else {
 745         // If not in SD and not in PH, other thread's load must have failed
 746         super_load_in_progress = false;
 747       }
 748     }
 749   }
 750   return NULL;
 751 }

 764 // Be careful when modifying this code: once you have run
 765 // placeholders()->find_and_add(PlaceholderTable::LOAD_INSTANCE),
 766 // you need to find_and_remove it before returning.
 767 // So be careful to not exit with a CHECK_ macro betweeen these calls.
 768 //
 769 // name must be in the form of "java/lang/Object" -- cannot be "Ljava/lang/Object;"
 770 InstanceKlass* SystemDictionary::resolve_instance_class_or_null(Symbol* name,
 771                                                                 Handle class_loader,
 772                                                                 Handle protection_domain,
 773                                                                 TRAPS) {
 774   assert(name != NULL && !Signature::is_array(name) &&
 775          !Signature::has_envelope(name), "invalid class name");
 776 
 777   EventClassLoad class_load_start_event;
 778 
 779   HandleMark hm(THREAD);
 780 
 781   // Fix for 4474172; see evaluation for more details
 782   class_loader = Handle(THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 783   ClassLoaderData* loader_data = register_loader(class_loader);
 784   Dictionary* dictionary = loader_data->dictionary();
 785   unsigned int d_hash = dictionary->compute_hash(name);
 786 
 787   // Do lookup to see if class already exist and the protection domain
 788   // has the right access
 789   // This call uses find which checks protection domain already matches
 790   // All subsequent calls use find_class, and set has_loaded_class so that
 791   // before we return a result we call out to java to check for valid protection domain
 792   // to allow returning the Klass* and add it to the pd_set if it is valid
 793   {
 794     InstanceKlass* probe = dictionary->find(d_hash, name, protection_domain);
 795     if (probe != NULL) return probe;
 796   }
 797 
 798   // Non-bootstrap class loaders will call out to class loader and
 799   // define via jvm/jni_DefineClass which will acquire the
 800   // class loader object lock to protect against multiple threads
 801   // defining the class in parallel by accident.
 802   // This lock must be acquired here so the waiter will find
 803   // any successful result in the SystemDictionary and not attempt
 804   // the define.
 805   // ParallelCapable Classloaders and the bootstrap classloader
 806   // do not acquire lock here.
 807   bool DoObjectLock = true;
 808   if (is_parallelCapable(class_loader)) {
 809     DoObjectLock = false;
 810   }
 811 
 812   unsigned int p_hash = placeholders()->compute_hash(name);
 813   int p_index = placeholders()->hash_to_index(p_hash);
 814 
 815   // Class is not in SystemDictionary so we have to do loading.
 816   // Make sure we are synchronized on the class loader before we proceed
 817   Handle lockObject = compute_loader_lock_object(class_loader, THREAD);
 818   check_loader_lock_contention(lockObject, THREAD);
 819   ObjectLocker ol(lockObject, THREAD, DoObjectLock);
 820 
 821   // Check again (after locking) if class already exist in SystemDictionary
 822   bool class_has_been_loaded   = false;
 823   bool super_load_in_progress  = false;
 824   bool havesupername = false;
 825   InstanceKlass* k = NULL;
 826   PlaceholderEntry* placeholder;
 827   Symbol* superclassname = NULL;
 828 
 829   assert(THREAD->can_call_java(),
 830          "can not load classes with compiler thread: class=%s, classloader=%s",
 831          name->as_C_string(),
 832          class_loader.is_null() ? "null" : class_loader->klass()->name()->as_C_string());
 833   {
 834     MutexLocker mu(THREAD, SystemDictionary_lock);
 835     InstanceKlass* check = find_class(d_hash, name, dictionary);
 836     if (check != NULL) {
 837       // InstanceKlass is already loaded, so just return it
 838       class_has_been_loaded = true;
 839       k = check;
 840     } else {
 841       placeholder = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 842       if (placeholder && placeholder->super_load_in_progress()) {
 843          super_load_in_progress = true;
 844          if (placeholder->havesupername() == true) {
 845            superclassname = placeholder->supername();
 846            havesupername = true;
 847          }
 848       }
 849     }
 850   }
 851 
 852   // If the class is in the placeholder table, class loading is in progress
 853   if (super_load_in_progress && havesupername==true) {
 854     k = handle_parallel_super_load(name,
 855                                    superclassname,
 856                                    class_loader,
 857                                    protection_domain,
 858                                    lockObject, THREAD);
 859     if (HAS_PENDING_EXCEPTION) {
 860       return NULL;
 861     }
 862     if (k != NULL) {
 863       class_has_been_loaded = true;
 864     }
 865   }
 866 
 867   bool throw_circularity_error = false;
 868   if (!class_has_been_loaded) {
 869     bool load_instance_added = false;
 870 
 871     // add placeholder entry to record loading instance class
 872     // Five cases:
 873     // All cases need to prevent modifying bootclasssearchpath
 874     // in parallel with a classload of same classname
 875     // Redefineclasses uses existence of the placeholder for the duration
 876     // of the class load to prevent concurrent redefinition of not completely
 877     // defined classes.
 878     // case 1. traditional classloaders that rely on the classloader object lock
 879     //   - no other need for LOAD_INSTANCE
 880     // case 2. traditional classloaders that break the classloader object lock
 881     //    as a deadlock workaround. Detection of this case requires that
 882     //    this check is done while holding the classloader object lock,
 883     //    and that lock is still held when calling classloader's loadClass.
 884     //    For these classloaders, we ensure that the first requestor
 885     //    completes the load and other requestors wait for completion.
 886     // case 3. Bootstrap classloader - don't own objectLocker
 887     //    This classloader supports parallelism at the classloader level,
 888     //    but only allows a single load of a class/classloader pair.
 889     //    No performance benefit and no deadlock issues.
 890     // case 4. parallelCapable user level classloaders - without objectLocker
 891     //    Allow parallel classloading of a class/classloader pair
 892 
 893     {
 894       MutexLocker mu(THREAD, SystemDictionary_lock);
 895       if (class_loader.is_null() || !is_parallelCapable(class_loader)) {
 896         PlaceholderEntry* oldprobe = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 897         if (oldprobe) {
 898           // only need check_seen_thread once, not on each loop
 899           // 6341374 java/lang/Instrument with -Xcomp
 900           if (oldprobe->check_seen_thread(THREAD, PlaceholderTable::LOAD_INSTANCE)) {
 901             throw_circularity_error = true;
 902           } else {
 903             // case 1: traditional: should never see load_in_progress.
 904             while (!class_has_been_loaded && oldprobe && oldprobe->instance_load_in_progress()) {
 905 
 906               // case 3: bootstrap classloader: prevent futile classloading,
 907               // wait on first requestor
 908               if (class_loader.is_null()) {
 909                 SystemDictionary_lock->wait();
 910               } else {
 911               // case 2: traditional with broken classloader lock. wait on first
 912               // requestor.
 913                 double_lock_wait(lockObject, THREAD);
 914               }
 915               // Check if classloading completed while we were waiting
 916               InstanceKlass* check = find_class(d_hash, name, dictionary);
 917               if (check != NULL) {
 918                 // Klass is already loaded, so just return it
 919                 k = check;
 920                 class_has_been_loaded = true;
 921               }
 922               // check if other thread failed to load and cleaned up
 923               oldprobe = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 924             }
 925           }
 926         }
 927       }
 928       // All cases: add LOAD_INSTANCE holding SystemDictionary_lock
 929       // case 4: parallelCapable: allow competing threads to try
 930       // LOAD_INSTANCE in parallel
 931 
 932       if (!throw_circularity_error && !class_has_been_loaded) {
 933         PlaceholderEntry* newprobe = placeholders()->find_and_add(p_index, p_hash, name, loader_data, PlaceholderTable::LOAD_INSTANCE, NULL, THREAD);
 934         load_instance_added = true;
 935         // For class loaders that do not acquire the classloader object lock,
 936         // if they did not catch another thread holding LOAD_INSTANCE,
 937         // need a check analogous to the acquire ObjectLocker/find_class
 938         // i.e. now that we hold the LOAD_INSTANCE token on loading this class/CL
 939         // one final check if the load has already completed
 940         // class loaders holding the ObjectLock shouldn't find the class here
 941         InstanceKlass* check = find_class(d_hash, name, dictionary);
 942         if (check != NULL) {
 943           // Klass is already loaded, so return it after checking/adding protection domain
 944           k = check;
 945           class_has_been_loaded = true;
 946         }
 947       }
 948     }
 949 
 950     // must throw error outside of owning lock
 951     if (throw_circularity_error) {
 952       assert(!HAS_PENDING_EXCEPTION && load_instance_added == false,"circularity error cleanup");
 953       ResourceMark rm(THREAD);
 954       THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), name->as_C_string());
 955     }
 956 
 957     if (!class_has_been_loaded) {
 958 
 959       // Do actual loading
 960       k = load_instance_class(name, class_loader, THREAD);
 961 
 962       // If everything was OK (no exceptions, no null return value), and
 963       // class_loader is NOT the defining loader, do a little more bookkeeping.
 964       if (!HAS_PENDING_EXCEPTION && k != NULL &&
 965         k->class_loader() != class_loader()) {
 966 
 967         check_constraints(d_hash, k, class_loader, false, THREAD);
 968 
 969         // Need to check for a PENDING_EXCEPTION again; check_constraints
 970         // can throw but we may have to remove entry from the placeholder table below.
 971         if (!HAS_PENDING_EXCEPTION) {
 972           // Record dependency for non-parent delegation.
 973           // This recording keeps the defining class loader of the klass (k) found
 974           // from being unloaded while the initiating class loader is loaded
 975           // even if the reference to the defining class loader is dropped
 976           // before references to the initiating class loader.
 977           loader_data->record_dependency(k);
 978 
 979           { // Grabbing the Compile_lock prevents systemDictionary updates
 980             // during compilations.
 981             MutexLocker mu(THREAD, Compile_lock);
 982             update_dictionary(d_hash, p_index, p_hash,
 983               k, class_loader, THREAD);
 984           }
 985 
 986           if (JvmtiExport::should_post_class_load()) {
 987             Thread *thread = THREAD;
 988             assert(thread->is_Java_thread(), "thread->is_Java_thread()");
 989             JvmtiExport::post_class_load((JavaThread *) thread, k);
 990           }
 991         }
 992       }
 993     } // load_instance_class
 994 
 995     if (load_instance_added == true) {
 996       // clean up placeholder entries for LOAD_INSTANCE success or error
 997       // This brackets the SystemDictionary updates for both defining
 998       // and initiating loaders
 999       MutexLocker mu(THREAD, SystemDictionary_lock);
1000       placeholders()->find_and_remove(p_index, p_hash, name, loader_data, PlaceholderTable::LOAD_INSTANCE, THREAD);
1001       SystemDictionary_lock->notify_all();
1002     }
1003   }
1004 
1005   if (HAS_PENDING_EXCEPTION || k == NULL) {
1006     return NULL;
1007   }
1008   if (class_load_start_event.should_commit()) {
1009     post_class_load_event(&class_load_start_event, k, loader_data);
1010   }
1011 #ifdef ASSERT
1012   {
1013     ClassLoaderData* loader_data = k->class_loader_data();
1014     MutexLocker mu(THREAD, SystemDictionary_lock);
1015     InstanceKlass* kk = find_class(name, loader_data);
1016     assert(kk == k, "should be present in dictionary");
1017   }
1018 #endif
1019 
1020   // return if the protection domain in NULL
1021   if (protection_domain() == NULL) return k;
1022 
1023   // Check the protection domain has the right access
1024   if (dictionary->is_valid_protection_domain(d_hash, name,
1025                                              protection_domain)) {
1026     return k;
1027   }
1028 
1029   // Verify protection domain. If it fails an exception is thrown
1030   validate_protection_domain(k, class_loader, protection_domain, CHECK_NULL);
1031 
1032   return k;
1033 }

1102 // does not publish the classes via the SystemDictionary.

1104 // and redefineclasses. RedefinedClasses do not add to the class hierarchy.

1116   //                               the same class loader as the unsafe_anonymous_host.

1118   //                                           whose loader is the Lookup class's loader.

1127   }

1129   assert(st != NULL, "invariant");
1130   assert(st->need_verify(), "invariant");
1131

1133   // Note that we do this even though this klass might
1134   // already be present in the SystemDictionary, otherwise we would not
1135   // throw potential ClassFormatErrors.
1136

1296                                                InstanceKlass* ik,
1297                                                PackageEntry* pkg_entry,
1298                                                Handle class_loader, TRAPS) {
1299   assert(!ModuleEntryTable::javabase_moduleEntry()->is_patched(),
1300          "Cannot use sharing if java.base is patched");

1308       return false;

1310       return true;

1332           return false;
1333         }
1334       }
1335     }
1336   }

1338   if (class_loader.is_null()) {
1339     assert(ent != NULL, "Shared class for NULL classloader must have valid SharedClassPathEntry");
1340     // The NULL classloader can load archived class originated from the
1341     // "modules" jimage and the -Xbootclasspath/a. For class from the
1342     // "modules" jimage, the PackageEntry/ModuleEntry must be defined
1343     // by the NULL classloader.
1344     if (mod_entry != NULL) {
1345       // PackageEntry/ModuleEntry is found in the classloader. Check if the
1346       // ModuleEntry's location agrees with the archived class' origination.
1347       if (ent->is_modules_image() && mod_entry->location()->starts_with("jrt:")) {

1349       }
1350     }
1351 
1352     // If the archived class is not from the "module" jimage, the class can be
1353     // loaded by the NULL classloader if
1354     //
1355     // 1. the class is from the unamed package
1356     // 2. or, the class is not from a module defined in the NULL classloader
1357     // 3. or, the class is from an unamed module
1358     if (!ent->is_modules_image() && ik->is_shared_boot_class()) {
1359       // the class is from the -Xbootclasspath/a
1360       if (pkg_name == NULL ||
1361           pkg_entry == NULL ||
1362           pkg_entry->in_unnamed_module()) {
1363         assert(mod_entry == NULL ||
1364                mod_entry == loader_data->unnamed_module(),
1365                "the unnamed module is not defined in the classloader");
1366         return true;
1367       }
1368     }
1369     return false;
1370   } else {
1371     bool res = SystemDictionaryShared::is_shared_class_visible_for_classloader(
1372               ik, class_loader, pkg_name, pkg_entry, mod_entry, CHECK_(false));
1373     return res;
1374   }

1579 InstanceKlass* SystemDictionary::load_instance_class(Symbol* class_name, Handle class_loader, TRAPS) {
1580 
1581   if (class_loader.is_null()) {
1582     ResourceMark rm(THREAD);
1583     PackageEntry* pkg_entry = NULL;
1584     bool search_only_bootloader_append = false;
1585     ClassLoaderData *loader_data = class_loader_data(class_loader);
1586 
1587     // Find the package in the boot loader's package entry table.
1588     TempNewSymbol pkg_name = ClassLoader::package_from_class_name(class_name);
1589     if (pkg_name != NULL) {
1590       pkg_entry = loader_data->packages()->lookup_only(pkg_name);
1591     }
1592 
1593     // Prior to attempting to load the class, enforce the boot loader's
1594     // visibility boundaries.
1595     if (!Universe::is_module_initialized()) {
1596       // During bootstrapping, prior to module initialization, any
1597       // class attempting to be loaded must be checked against the
1598       // java.base packages in the boot loader's PackageEntryTable.
1599       // No class outside of java.base is allowed to be loaded during
1600       // this bootstrapping window.
1601       if (pkg_entry == NULL || pkg_entry->in_unnamed_module()) {
1602         // Class is either in the unnamed package or in
1603         // a named package within the unnamed module.  Either
1604         // case is outside of java.base, do not attempt to
1605         // load the class post java.base definition.  If
1606         // java.base has not been defined, let the class load
1607         // and its package will be checked later by
1608         // ModuleEntryTable::verify_javabase_packages.
1609         if (ModuleEntryTable::javabase_defined()) {
1610           return NULL;
1611         }
1612       } else {
1613         // Check that the class' package is defined within java.base.
1614         ModuleEntry* mod_entry = pkg_entry->module();
1615         Symbol* mod_entry_name = mod_entry->name();
1616         if (mod_entry_name->fast_compare(vmSymbols::java_base()) != 0) {
1617           return NULL;
1618         }
1619       }
1620     } else {
1621       // After the module system has been initialized, check if the class'
1622       // package is in a module defined to the boot loader.
1623       if (pkg_name == NULL || pkg_entry == NULL || pkg_entry->in_unnamed_module()) {
1624         // Class is either in the unnamed package, in a named package
1625         // within a module not defined to the boot loader or in a
1626         // a named package within the unnamed module.  In all cases,
1627         // limit visibility to search for the class only in the boot
1628         // loader's append path.
1629         if (!ClassLoader::has_bootclasspath_append()) {
1630            // If there is no bootclasspath append entry, no need to continue
1631            // searching.
1632            return NULL;
1633         }
1634         search_only_bootloader_append = true;
1635       }
1636     }
1637 
1638     // Prior to bootstrapping's module initialization, never load a class outside
1639     // of the boot loader's module path
1640     assert(Universe::is_module_initialized() ||
1641            !search_only_bootloader_append,
1642            "Attempt to load a class outside of boot loader's module path");
1643 
1644     // Search for classes in the CDS archive.
1645     InstanceKlass* k = NULL;
1646     {
1647 #if INCLUDE_CDS
1648       PerfTraceTime vmtimer(ClassLoader::perf_shared_classload_time());
1649       k = load_shared_boot_class(class_name, pkg_entry, THREAD);
1650 #endif
1651     }
1652 
1653     if (k == NULL) {
1654       // Use VM class loader
1655       PerfTraceTime vmtimer(ClassLoader::perf_sys_classload_time());
1656       k = ClassLoader::load_class(class_name, search_only_bootloader_append, CHECK_NULL);
1657     }
1658 
1659     // find_or_define_instance_class may return a different InstanceKlass
1660     if (k != NULL) {
1661       InstanceKlass* defined_k =
1662         find_or_define_instance_class(class_name, class_loader, k, THREAD);
1663       if (!HAS_PENDING_EXCEPTION && defined_k != k) {
1664         // If a parallel capable class loader already defined this class, register 'k' for cleanup.
1665         assert(defined_k != NULL, "Should have a klass if there's no exception");
1666         loader_data->add_to_deallocate_list(k);
1667         k = defined_k;
1668       } else if (HAS_PENDING_EXCEPTION) {
1669         loader_data->add_to_deallocate_list(k);
1670         return NULL;
1671       }
1672     }
1673     return k;
1674   } else {
1675     // Use user specified class loader to load class. Call loadClass operation on class_loader.
1676     ResourceMark rm(THREAD);
1677 
1678     assert(THREAD->is_Java_thread(), "must be a JavaThread");
1679     JavaThread* jt = (JavaThread*) THREAD;
1680 
1681     PerfClassTraceTime vmtimer(ClassLoader::perf_app_classload_time(),
1682                                ClassLoader::perf_app_classload_selftime(),
1683                                ClassLoader::perf_app_classload_count(),
1684                                jt->get_thread_stat()->perf_recursion_counts_addr(),
1685                                jt->get_thread_stat()->perf_timers_addr(),
1686                                PerfClassTraceTime::CLASS_LOAD);
1687 
1688     Handle s = java_lang_String::create_from_symbol(class_name, CHECK_NULL);
1689     // Translate to external class name format, i.e., convert '/' chars to '.'
1690     Handle string = java_lang_String::externalize_classname(s, CHECK_NULL);
1691 
1692     JavaValue result(T_OBJECT);
1693 
1694     InstanceKlass* spec_klass = SystemDictionary::ClassLoader_klass();
1695 
1696     // Call public unsynchronized loadClass(String) directly for all class loaders.
1697     // For parallelCapable class loaders, JDK >=7, loadClass(String, boolean) will
1698     // acquire a class-name based lock rather than the class loader object lock.
1699     // JDK < 7 already acquire the class loader lock in loadClass(String, boolean).
1700     JavaCalls::call_virtual(&result,
1701                             class_loader,
1702                             spec_klass,
1703                             vmSymbols::loadClass_name(),
1704                             vmSymbols::string_class_signature(),
1705                             string,
1706                             CHECK_NULL);
1707 
1708     assert(result.get_type() == T_OBJECT, "just checking");
1709     oop obj = (oop) result.get_jobject();
1710 
1711     // Primitive classes return null since forName() can not be
1712     // used to obtain any of the Class objects representing primitives or void
1713     if ((obj != NULL) && !(java_lang_Class::is_primitive(obj))) {
1714       InstanceKlass* k = InstanceKlass::cast(java_lang_Class::as_Klass(obj));
1715       // For user defined Java class loaders, check that the name returned is
1716       // the same as that requested.  This check is done for the bootstrap
1717       // loader when parsing the class file.
1718       if (class_name == k->name()) {
1719         return k;
1720       }
1721     }
1722     // Class is not found or has the wrong name, return NULL
1723     return NULL;
1724   }

1736 void SystemDictionary::define_instance_class(InstanceKlass* k, TRAPS) {
1737 
1738   HandleMark hm(THREAD);
1739   ClassLoaderData* loader_data = k->class_loader_data();
1740   Handle class_loader_h(THREAD, loader_data->class_loader());
1741 
1742  // for bootstrap and other parallel classloaders don't acquire lock,
1743  // use placeholder token
1744  // If a parallelCapable class loader calls define_instance_class instead of
1745  // find_or_define_instance_class to get here, we have a timing
1746  // hole with systemDictionary updates and check_constraints
1747  if (!class_loader_h.is_null() && !is_parallelCapable(class_loader_h)) {
1748     assert(ObjectSynchronizer::current_thread_holds_lock((JavaThread*)THREAD,
1749          compute_loader_lock_object(class_loader_h, THREAD)),
1750          "define called without lock");
1751   }
1752 
1753   // Check class-loading constraints. Throw exception if violation is detected.
1754   // Grabs and releases SystemDictionary_lock
1755   // The check_constraints/find_class call and update_dictionary sequence
1756   // must be "atomic" for a specific class/classloader pair so we never
1757   // define two different instanceKlasses for that class/classloader pair.
1758   // Existing classloaders will call define_instance_class with the
1759   // classloader lock held
1760   // Parallel classloaders will call find_or_define_instance_class
1761   // which will require a token to perform the define class
1762   Symbol*  name_h = k->name();
1763   Dictionary* dictionary = loader_data->dictionary();
1764   unsigned int d_hash = dictionary->compute_hash(name_h);
1765   check_constraints(d_hash, k, class_loader_h, true, CHECK);
1766 
1767   // Register class just loaded with class loader (placed in ArrayList)
1768   // Note we do this before updating the dictionary, as this can
1769   // fail with an OutOfMemoryError (if it does, we will *not* put this
1770   // class in the dictionary and will not update the class hierarchy).
1771   // JVMTI FollowReferences needs to find the classes this way.
1772   if (k->class_loader() != NULL) {
1773     methodHandle m(THREAD, Universe::loader_addClass_method());
1774     JavaValue result(T_VOID);
1775     JavaCallArguments args(class_loader_h);
1776     args.push_oop(Handle(THREAD, k->java_mirror()));
1777     JavaCalls::call(&result, m, &args, CHECK);
1778   }
1779 
1780   // Add the new class. We need recompile lock during update of CHA.
1781   {
1782     unsigned int p_hash = placeholders()->compute_hash(name_h);
1783     int p_index = placeholders()->hash_to_index(p_hash);
1784 
1785     MutexLocker mu_r(THREAD, Compile_lock);
1786 
1787     // Add to class hierarchy, initialize vtables, and do possible
1788     // deoptimizations.
1789     add_to_hierarchy(k, CHECK); // No exception, but can block
1790 
1791     // Add to systemDictionary - so other classes can see it.
1792     // Grabs and releases SystemDictionary_lock
1793     update_dictionary(d_hash, p_index, p_hash,
1794                       k, class_loader_h, THREAD);
1795   }
1796   k->eager_initialize(THREAD);
1797 
1798   // notify jvmti
1799   if (JvmtiExport::should_post_class_load()) {
1800       assert(THREAD->is_Java_thread(), "thread->is_Java_thread()");
1801       JvmtiExport::post_class_load((JavaThread *) THREAD, k);
1802 
1803   }
1804   post_class_define_event(k, loader_data);
1805 }

1807 // Support parallel classloading
1808 // All parallel class loaders, including bootstrap classloader
1809 // lock a placeholder entry for this class/class_loader pair
1810 // to allow parallel defines of different classes for this class loader
1811 // With AllowParallelDefine flag==true, in case they do not synchronize around
1812 // FindLoadedClass/DefineClass, calls, we check for parallel
1813 // loading for them, wait if a defineClass is in progress
1814 // and return the initial requestor's results
1815 // This flag does not apply to the bootstrap classloader.
1816 // With AllowParallelDefine flag==false, call through to define_instance_class
1817 // which will throw LinkageError: duplicate class definition.
1818 // False is the requested default.
1819 // For better performance, the class loaders should synchronize
1820 // findClass(), i.e. FindLoadedClass/DefineClassIfAbsent or they
1821 // potentially waste time reading and parsing the bytestream.
1822 // Note: VM callers should ensure consistency of k/class_name,class_loader
1823 // Be careful when modifying this code: once you have run
1824 // placeholders()->find_and_add(PlaceholderTable::DEFINE_CLASS),
1825 // you need to find_and_remove it before returning.
1826 // So be careful to not exit with a CHECK_ macro betweeen these calls.
1827 InstanceKlass* SystemDictionary::find_or_define_instance_class(Symbol* class_name, Handle class_loader,
1828                                                                InstanceKlass* k, TRAPS) {
1829 
1830   Symbol*  name_h = k->name(); // passed in class_name may be null
1831   ClassLoaderData* loader_data = class_loader_data(class_loader);
1832   Dictionary* dictionary = loader_data->dictionary();
1833 
1834   unsigned int d_hash = dictionary->compute_hash(name_h);
1835 
1836   // Hold SD lock around find_class and placeholder creation for DEFINE_CLASS
1837   unsigned int p_hash = placeholders()->compute_hash(name_h);
1838   int p_index = placeholders()->hash_to_index(p_hash);
1839   PlaceholderEntry* probe;
1840 
1841   {
1842     MutexLocker mu(THREAD, SystemDictionary_lock);
1843     // First check if class already defined
1844     if (is_parallelDefine(class_loader)) {
1845       InstanceKlass* check = find_class(d_hash, name_h, dictionary);
1846       if (check != NULL) {
1847         return check;
1848       }
1849     }
1850 
1851     // Acquire define token for this class/classloader
1852     probe = placeholders()->find_and_add(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, NULL, THREAD);
1853     // Wait if another thread defining in parallel
1854     // All threads wait - even those that will throw duplicate class: otherwise
1855     // caller is surprised by LinkageError: duplicate, but findLoadedClass fails
1856     // if other thread has not finished updating dictionary
1857     while (probe->definer() != NULL) {
1858       SystemDictionary_lock->wait();
1859     }
1860     // Only special cases allow parallel defines and can use other thread's results
1861     // Other cases fall through, and may run into duplicate defines
1862     // caught by finding an entry in the SystemDictionary
1863     if (is_parallelDefine(class_loader) && (probe->instance_klass() != NULL)) {
1864         placeholders()->find_and_remove(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, THREAD);
1865         SystemDictionary_lock->notify_all();
1866 #ifdef ASSERT
1867         InstanceKlass* check = find_class(d_hash, name_h, dictionary);
1868         assert(check != NULL, "definer missed recording success");
1869 #endif
1870         return probe->instance_klass();
1871     } else {
1872       // This thread will define the class (even if earlier thread tried and had an error)
1873       probe->set_definer(THREAD);
1874     }
1875   }
1876 
1877   define_instance_class(k, THREAD);
1878 
1879   Handle linkage_exception = Handle(); // null handle
1880 
1881   // definer must notify any waiting threads
1882   {
1883     MutexLocker mu(THREAD, SystemDictionary_lock);
1884     PlaceholderEntry* probe = placeholders()->get_entry(p_index, p_hash, name_h, loader_data);
1885     assert(probe != NULL, "DEFINE_CLASS placeholder lost?");
1886     if (probe != NULL) {
1887       if (HAS_PENDING_EXCEPTION) {
1888         linkage_exception = Handle(THREAD,PENDING_EXCEPTION);
1889         CLEAR_PENDING_EXCEPTION;
1890       } else {
1891         probe->set_instance_klass(k);
1892       }
1893       probe->set_definer(NULL);
1894       placeholders()->find_and_remove(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, THREAD);
1895       SystemDictionary_lock->notify_all();
1896     }
1897   }
1898 
1899   // Can't throw exception while holding lock due to rank ordering
1900   if (linkage_exception() != NULL) {
1901     THROW_OOP_(linkage_exception(), NULL); // throws exception and returns
1902   }
1903 
1904   return k;
1905 }

2255 // found in the OOPSLA'98 paper "Dynamic Class Loading in the Java
2256 // Virtual Machine" by Sheng Liang and Gilad Bracha.  The basic idea is
2257 // that the dictionary needs to maintain a set of contraints that
2258 // must be satisfied by all classes in the dictionary.
2259 // if defining is true, then LinkageError if already in dictionary

2360 // loader constraints might know about a class that isn't fully loaded
2361 // yet and these will be ignored.
2362 Klass* SystemDictionary::find_constrained_instance_or_array_klass(
2363                     Symbol* class_name, Handle class_loader, TRAPS) {
2364 
2365   // First see if it has been loaded directly.
2366   // Force the protection domain to be null.  (This removes protection checks.)

2374   // a loader constraint that would require this loader to return the
2375   // klass that is already loaded.
2376   if (Signature::is_array(class_name)) {
2377     // For array classes, their Klass*s are not kept in the
2378     // constraint table. The element Klass*s are.

2392   } else {

2394     // Non-array classes are easy: simply check the constraint table.
2395     klass = constraints()->find_constrained_klass(class_name, class_loader);

2402                                              Klass* klass_being_linked,
2403                                              Handle class_loader1,
2404                                              Handle class_loader2,

2407   ClassLoaderData* loader_data2 = class_loader_data(class_loader2);
2408 
2409   Symbol* constraint_name = NULL;
2410 
2411   if (!Signature::is_array(class_name)) {
2412     constraint_name = class_name;
2413   } else {
2414     // For array classes, their Klass*s are not kept in the
2415     // constraint table. The element classes are.

2429   unsigned int d_hash1 = dictionary1->compute_hash(constraint_name);
2430

2432   unsigned int d_hash2 = dictionary2->compute_hash(constraint_name);
2433 
2434   {
2435     MutexLocker mu_s(THREAD, SystemDictionary_lock);
2436     InstanceKlass* klass1 = find_class(d_hash1, constraint_name, dictionary1);

2439                                            klass2, class_loader2);
2440 #if INCLUDE_CDS
2441     if (Arguments::is_dumping_archive() && klass_being_linked != NULL &&
2442         !klass_being_linked->is_shared()) {
2443          SystemDictionaryShared::record_linking_constraint(constraint_name,
2444                                      InstanceKlass::cast(klass_being_linked),
2445                                      class_loader1, class_loader2, THREAD);
2446     }
2447 #endif // INCLUDE_CDS
2448     if (Signature::is_array(class_name)) {
2449       constraint_name->decrement_refcount();
2450     }

2526 // Signature constraints ensure that callers and callees agree about
2527 // the meaning of type names in their signatures.  This routine is the
2528 // intake for constraints.  It collects them from several places:
2529 //
2530 //  * LinkResolver::resolve_method (if check_access is true) requires
2531 //    that the resolving class (the caller) and the defining class of
2532 //    the resolved method (the callee) agree on each type in the
2533 //    method's signature.
2534 //
2535 //  * LinkResolver::resolve_interface_method performs exactly the same
2536 //    checks.
2537 //
2538 //  * LinkResolver::resolve_field requires that the constant pool
2539 //    attempting to link to a field agree with the field's defining
2540 //    class about the type of the field signature.
2541 //
2542 //  * klassVtable::initialize_vtable requires that, when a class
2543 //    overrides a vtable entry allocated by a superclass, that the
2544 //    overriding method (i.e., the callee) agree with the superclass
2545 //    on each type in the method's signature.
2546 //
2547 //  * klassItable::initialize_itable requires that, when a class fills
2548 //    in its itables, for each non-abstract method installed in an
2549 //    itable, the method (i.e., the callee) agree with the interface
2550 //    on each type in the method's signature.
2551 //
2552 // All those methods have a boolean (check_access, checkconstraints)
2553 // which turns off the checks.  This is used from specialized contexts
2554 // such as bootstrapping, dumping, and debugging.
2555 //
2556 // No direct constraint is placed between the class and its
2557 // supertypes.  Constraints are only placed along linked relations
2558 // between callers and callees.  When a method overrides or implements
2559 // an abstract method in a supertype (superclass or interface), the
2560 // constraints are placed as if the supertype were the caller to the
2561 // overriding method.  (This works well, since callers to the
2562 // supertype have already established agreement between themselves and
2563 // the supertype.)  As a result of all this, a class can disagree with
2564 // its supertype about the meaning of a type name, as long as that
2565 // class neither calls a relevant method of the supertype, nor is
2566 // called (perhaps via an override) from the supertype.
2567 //
2568 //
2569 // SystemDictionary::check_signature_loaders(sig, klass_being_linked, l1, l2)
2570 //
2571 // Make sure all class components (including arrays) in the given
2572 // signature will be resolved to the same class in both loaders.
2573 // Returns the name of the type that failed a loader constraint check, or
2574 // NULL if no constraint failed.  No exception except OOME is thrown.
2575 // Arrays are not added to the loader constraint table, their elements are.
2576 Symbol* SystemDictionary::check_signature_loaders(Symbol* signature,
2577                                                Klass* klass_being_linked,
2578                                                Handle loader1, Handle loader2,
2579                                                bool is_method, TRAPS)  {
2580   // Nothing to do if loaders are the same.
2581   if (loader1() == loader2()) {
2582     return NULL;
2583   }
2584 
2585   for (SignatureStream ss(signature, is_method); !ss.is_done(); ss.next()) {
2586     if (ss.is_reference()) {
2587       Symbol* sig = ss.as_symbol();
2588       // Note: In the future, if template-like types can take
2589       // arguments, we will want to recognize them and dig out class
2590       // names hiding inside the argument lists.
2591       if (!add_loader_constraint(sig, klass_being_linked, loader1, loader2, THREAD)) {
2592         return sig;
2593       }
2594     }
2595   }
2596   return NULL;
2597 }

2782   vmIntrinsics::ID null_iid = vmIntrinsics::_none;  // distinct from all method handle invoker intrinsics

2784   int          index = invoke_method_table()->hash_to_index(hash);
2785   SymbolPropertyEntry* spe = invoke_method_table()->find_entry(index, hash, signature, null_iid);
2786   if (spe != NULL && spe->method_type() != NULL) {
2787     assert(java_lang_invoke_MethodType::is_instance(spe->method_type()), "");
2788     return Handle(THREAD, spe->method_type());
2789   } else if (!THREAD->can_call_java()) {
2790     warning("SystemDictionary::find_method_handle_type called from compiler thread");  // FIXME
2791     return Handle();  // do not attempt from within compiler, unless it was cached
2792   }
2793 
2794   Handle class_loader, protection_domain;
2795   if (accessing_klass != NULL) {
2796     class_loader      = Handle(THREAD, accessing_klass->class_loader());
2797     protection_domain = Handle(THREAD, accessing_klass->protection_domain());
2798   }
2799   bool can_be_cached = true;
2800   int npts = ArgumentCount(signature).size();
2801   objArrayHandle pts = oopFactory::new_objArray_handle(SystemDictionary::Class_klass(), npts, CHECK_(empty));
2802   int arg = 0;
2803   Handle rt; // the return type from the signature
2804   ResourceMark rm(THREAD);
2805   for (SignatureStream ss(signature); !ss.is_done(); ss.next()) {
2806     oop mirror = NULL;
2807     if (can_be_cached) {
2808       // Use neutral class loader to lookup candidate classes to be placed in the cache.
2809       mirror = ss.as_java_mirror(Handle(), Handle(),
2810                                  SignatureStream::ReturnNull, CHECK_(empty));
2811       if (mirror == NULL || (ss.is_reference() && !is_always_visible_class(mirror))) {
2812         // Fall back to accessing_klass context.
2813         can_be_cached = false;
2814       }
2815     }
2816     if (!can_be_cached) {
2817       // Resolve, throwing a real error if it doesn't work.
2818       mirror = ss.as_java_mirror(class_loader, protection_domain,
2819                                  SignatureStream::NCDFError, CHECK_(empty));
2820     }
2821     assert(mirror != NULL, "%s", ss.as_symbol()->as_C_string());
2822     if (ss.at_return_type())
2823       rt = Handle(THREAD, mirror);
2824     else
2825       pts->obj_at_put(arg++, mirror);
2826 
2827     // Check accessibility.
2828     if (!java_lang_Class::is_primitive(mirror) && accessing_klass != NULL) {
2829       Klass* sel_klass = java_lang_Class::as_Klass(mirror);
2830       mirror = NULL;  // safety
2831       // Emulate ConstantPool::verify_constant_pool_resolve.
2832       LinkResolver::check_klass_accessibility(accessing_klass, sel_klass, CHECK_(empty));
2833     }
2834   }
2835   assert(arg == npts, "");
2836 
2837   // call java.lang.invoke.MethodHandleNatives::findMethodHandleType(Class rt, Class[] pts) -> MethodType
2838   JavaCallArguments args(Handle(THREAD, rt()));
2839   args.push_oop(pts);
2840   JavaValue result(T_OBJECT);
2841   JavaCalls::call_static(&result,
2842                          SystemDictionary::MethodHandleNatives_klass(),
2843                          vmSymbols::findMethodHandleType_name(),
2844                          vmSymbols::findMethodHandleType_signature(),
2845                          &args, CHECK_(empty));
2846   Handle method_type(THREAD, (oop) result.get_jobject());
2847 
2848   if (can_be_cached) {
2849     // We can cache this MethodType inside the JVM.
2850     MutexLocker ml(THREAD, SystemDictionary_lock);
2851     spe = invoke_method_table()->find_entry(index, hash, signature, null_iid);
2852     if (spe == NULL)
2853       spe = invoke_method_table()->add_entry(index, hash, signature, null_iid);
2854     if (spe->method_type() == NULL) {
2855       spe->set_method_type(method_type());
2856     }
2857   }
2858 
2859   // report back to the caller with the MethodType
2860   return method_type;
2861 }

2863 Handle SystemDictionary::find_field_handle_type(Symbol* signature,
2864                                                 Klass* accessing_klass,
2865                                                 TRAPS) {
2866   Handle empty;
2867   ResourceMark rm(THREAD);
2868   SignatureStream ss(signature, /*is_method=*/ false);
2869   if (!ss.is_done()) {
2870     Handle class_loader, protection_domain;
2871     if (accessing_klass != NULL) {
2872       class_loader      = Handle(THREAD, accessing_klass->class_loader());
2873       protection_domain = Handle(THREAD, accessing_klass->protection_domain());
2874     }
2875     oop mirror = ss.as_java_mirror(class_loader, protection_domain, SignatureStream::NCDFError, CHECK_(empty));
2876     ss.next();
2877     if (ss.is_done()) {
2878       return Handle(THREAD, mirror);
2879     }
2880   }
2881   return empty;
2882 }

2884 // Ask Java code to find or construct a method handle constant.
2885 Handle SystemDictionary::link_method_handle_constant(Klass* caller,
2886                                                      int ref_kind, //e.g., JVM_REF_invokeVirtual
2887                                                      Klass* callee,
2888                                                      Symbol* name,
2889                                                      Symbol* signature,
2890                                                      TRAPS) {
2891   Handle empty;
2892   if (caller == NULL) {
2893     THROW_MSG_(vmSymbols::java_lang_InternalError(), "bad MH constant", empty);
2894   }
2895   Handle name_str      = java_lang_String::create_from_symbol(name,      CHECK_(empty));
2896   Handle signature_str = java_lang_String::create_from_symbol(signature, CHECK_(empty));
2897 
2898   // Put symbolic info from the MH constant into freshly created MemberName and resolve it.
2899   Handle mname = MemberName_klass()->allocate_instance_handle(CHECK_(empty));
2900   java_lang_invoke_MemberName::set_clazz(mname(), callee->java_mirror());
2901   java_lang_invoke_MemberName::set_name (mname(), name_str());
2902   java_lang_invoke_MemberName::set_type (mname(), signature_str());
2903   java_lang_invoke_MemberName::set_flags(mname(), MethodHandles::ref_kind_to_flags(ref_kind));
2904 
2905   if (ref_kind == JVM_REF_invokeVirtual &&
2906       MethodHandles::is_signature_polymorphic_public_name(callee, name)) {
2907     // Skip resolution for public signature polymorphic methods such as
2908     // j.l.i.MethodHandle.invoke()/invokeExact() and those on VarHandle
2909     // They require appendix argument which MemberName resolution doesn't handle.
2910     // There's special logic on JDK side to handle them
2911     // (see MethodHandles.linkMethodHandleConstant() and MethodHandles.findVirtualForMH()).
2912   } else {
2913     MethodHandles::resolve_MemberName(mname, caller, /*speculative_resolve*/false, CHECK_(empty));
2914   }
2915 
2916   // After method/field resolution succeeded, it's safe to resolve MH signature as well.
2917   Handle type = MethodHandles::resolve_MemberName_type(mname, caller, CHECK_(empty));
2918 
2919   // call java.lang.invoke.MethodHandleNatives::linkMethodHandleConstant(Class caller, int refKind, Class callee, String name, Object type) -> MethodHandle
2920   JavaCallArguments args;
2921   args.push_oop(Handle(THREAD, caller->java_mirror()));  // the referring class
2922   args.push_int(ref_kind);
2923   args.push_oop(Handle(THREAD, callee->java_mirror()));  // the target class
2924   args.push_oop(name_str);
2925   args.push_oop(type);
2926   JavaValue result(T_OBJECT);
2927   JavaCalls::call_static(&result,
2928                          SystemDictionary::MethodHandleNatives_klass(),
2929                          vmSymbols::linkMethodHandleConstant_name(),
2930                          vmSymbols::linkMethodHandleConstant_signature(),
2931                          &args, CHECK_(empty));
2932   return Handle(THREAD, (oop) result.get_jobject());
2933 }

2935 // Ask Java to run a bootstrap method, in order to create a dynamic call site
2936 // while linking an invokedynamic op, or compute a constant for Dynamic_info CP entry
2937 // with linkage results being stored back into the bootstrap specifier.
2938 void SystemDictionary::invoke_bootstrap_method(BootstrapInfo& bootstrap_specifier, TRAPS) {
2939   // Resolve the bootstrap specifier, its name, type, and static arguments
2940   bootstrap_specifier.resolve_bsm(CHECK);
2941 
2942   // This should not happen.  JDK code should take care of that.
2943   if (bootstrap_specifier.caller() == NULL || bootstrap_specifier.type_arg().is_null()) {
2944     THROW_MSG(vmSymbols::java_lang_InternalError(), "Invalid bootstrap method invocation with no caller or type argument");
2945   }
2946 
2947   bool is_indy = bootstrap_specifier.is_method_call();
2948   objArrayHandle appendix_box;
2949   if (is_indy) {
2950     // Some method calls may require an appendix argument.  Arrange to receive it.
2951     appendix_box = oopFactory::new_objArray_handle(SystemDictionary::Object_klass(), 1, CHECK);
2952     assert(appendix_box->obj_at(0) == NULL, "");
2953   }
2954 
2955   // call condy: java.lang.invoke.MethodHandleNatives::linkDynamicConstant(caller, condy_index, bsm, type, info)
2956   //       indy: java.lang.invoke.MethodHandleNatives::linkCallSite(caller, indy_index, bsm, name, mtype, info, &appendix)
2957   JavaCallArguments args;
2958   args.push_oop(Handle(THREAD, bootstrap_specifier.caller_mirror()));
2959   args.push_int(bootstrap_specifier.bss_index());
2960   args.push_oop(bootstrap_specifier.bsm());
2961   args.push_oop(bootstrap_specifier.name_arg());
2962   args.push_oop(bootstrap_specifier.type_arg());
2963   args.push_oop(bootstrap_specifier.arg_values());
2964   if (is_indy) {
2965     args.push_oop(appendix_box);
2966   }
2967   JavaValue result(T_OBJECT);
2968   JavaCalls::call_static(&result,
2969                          SystemDictionary::MethodHandleNatives_klass(),
2970                          is_indy ? vmSymbols::linkCallSite_name() : vmSymbols::linkDynamicConstant_name(),
2971                          is_indy ? vmSymbols::linkCallSite_signature() : vmSymbols::linkDynamicConstant_signature(),
2972                          &args, CHECK);
2973 
2974   Handle value(THREAD, (oop) result.get_jobject());
2975   if (is_indy) {
2976     Handle appendix;
2977     Method* method = unpack_method_and_appendix(value,
2978                                                 bootstrap_specifier.caller(),
2979                                                 appendix_box,
2980                                                 &appendix, CHECK);
2981     methodHandle mh(THREAD, method);
2982     bootstrap_specifier.set_resolved_method(mh, appendix);
2983   } else {
2984     bootstrap_specifier.set_resolved_value(value);
2985   }
2986 
2987   // sanity check
2988   assert(bootstrap_specifier.is_resolved() ||
2989          (bootstrap_specifier.is_method_call() &&
2990           bootstrap_specifier.resolved_method().not_null()), "bootstrap method call failed");
2991 }
