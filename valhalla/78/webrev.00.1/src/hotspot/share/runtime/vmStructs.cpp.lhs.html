3101 static int recursiveFindType(VMTypeEntry* origtypes, const char* typeName, bool isRecurse) {
3102   {
3103     VMTypeEntry* types = origtypes;
3104     while (types->typeName != NULL) {
3105       if (strcmp(typeName, types->typeName) == 0) {
3106         // Found it
3107         return 1;
3108       }
3109       ++types;
3110     }
3111   }
3112   // Search for the base type by peeling off const and *
3113   size_t len = strlen(typeName);
3114   if (typeName[len-1] == '*') {
3115     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3116     strncpy(s, typeName, len - 1);
3117     s[len-1] = '\0';
3118     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3119     if (recursiveFindType(origtypes, s, true) == 1) {
3120       FREE_C_HEAP_ARRAY(char, s);
3121       return 1;
3122     }
3123     FREE_C_HEAP_ARRAY(char, s);
3124   }
3125   const char* start = NULL;
3126   if (strstr(typeName, "GrowableArray<") == typeName) {
3127     start = typeName + strlen("GrowableArray<");
3128   } else if (strstr(typeName, "Array<") == typeName) {
3129     start = typeName + strlen("Array<");
3130   }
3131   if (start != NULL) {
3132     const char * end = strrchr(typeName, '>');
3133     int len = end - start + 1;
3134     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3135     strncpy(s, start, len - 1);
3136     s[len-1] = '\0';
3137     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3138     if (recursiveFindType(origtypes, s, true) == 1) {
3139       FREE_C_HEAP_ARRAY(char, s);
3140       return 1;
3141     }
3142     FREE_C_HEAP_ARRAY(char, s);
3143   }
3144   if (strstr(typeName, "const ") == typeName) {
3145     const char * s = typeName + strlen("const ");
3146     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3147     if (recursiveFindType(origtypes, s, true) == 1) {
3148       return 1;
3149     }
3150   }
3151   if (strstr(typeName, " const") == typeName + len - 6) {
3152     char * s = os::strdup_check_oom(typeName);
3153     s[len - 6] = '\0';
3154     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3155     if (recursiveFindType(origtypes, s, true) == 1) {
3156       os::free(s);
3157       return 1;
3158     }
3159     os::free(s);
3160   }
3161   if (!isRecurse) {
3162     tty->print_cr("type \"%s\" not found", typeName);
3163   }
3164   return 0;
3165 }
