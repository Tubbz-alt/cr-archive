  91 static inline InstanceKlass* get_ik(jclass def) {

  96 // If any of the classes are being redefined, wait
  97 // Parallel constant pool merging leads to indeterminate constant pools.
  98 void VM_RedefineClasses::lock_classes() {
  99   MonitorLocker ml(RedefineClasses_lock);
 100   bool has_redefined;
 101   do {
 102     has_redefined = false;
 103     // Go through classes each time until none are being redefined.
 104     for (int i = 0; i < _class_count; i++) {
 105       if (get_ik(_class_defs[i].klass)->is_being_redefined()) {
 106         ml.wait();
 107         has_redefined = true;
 108         break;  // for loop
 109       }
 110     }
 111   } while (has_redefined);
 112   for (int i = 0; i < _class_count; i++) {
 113     get_ik(_class_defs[i].klass)->set_is_being_redefined(true);
 114   }
 115   ml.notify_all();
 116 }
 117 
 118 void VM_RedefineClasses::unlock_classes() {
 119   MonitorLocker ml(RedefineClasses_lock);
 120   for (int i = 0; i < _class_count; i++) {
 121     assert(get_ik(_class_defs[i].klass)->is_being_redefined(),
 122            "should be being redefined to get here");
 123     get_ik(_class_defs[i].klass)->set_is_being_redefined(false);
 124   }
 125   ml.notify_all();
 126 }

 128 bool VM_RedefineClasses::doit_prologue() {
 129   if (_class_count == 0) {
 130     _res = JVMTI_ERROR_NONE;
 131     return false;
 132   }
 133   if (_class_defs == NULL) {
 134     _res = JVMTI_ERROR_NULL_POINTER;
 135     return false;
 136   }
 137 
 138   for (int i = 0; i < _class_count; i++) {
 139     if (_class_defs[i].klass == NULL) {
 140       _res = JVMTI_ERROR_INVALID_CLASS;
 141       return false;
 142     }
 143     if (_class_defs[i].class_byte_count == 0) {
 144       _res = JVMTI_ERROR_INVALID_CLASS_FORMAT;
 145       return false;
 146     }
 147     if (_class_defs[i].class_bytes == NULL) {
 148       _res = JVMTI_ERROR_NULL_POINTER;
 149       return false;
 150     }
 151 
 152     oop mirror = JNIHandles::resolve_non_null(_class_defs[i].klass);
 153     // classes for primitives, arrays, hidden and vm unsafe anonymous classes
 154     // cannot be redefined.
 155     if (!is_modifiable_class(mirror)) {
 156       _res = JVMTI_ERROR_UNMODIFIABLE_CLASS;
 157       return false;
 158     }
 159   }
 160 
 161   // Start timer after all the sanity checks; not quite accurate, but
 162   // better than adding a bunch of stop() calls.
 163   if (log_is_enabled(Info, redefine, class, timer)) {
 164     _timer_vm_op_prologue.start();
 165   }
 166 
 167   lock_classes();
 168   // We first load new class versions in the prologue, because somewhere down the
 169   // call chain it is required that the current thread is a Java thread.
 170   _res = load_new_class_versions(Thread::current());
 171   if (_res != JVMTI_ERROR_NONE) {
 172     // free any successfully created classes, since none are redefined
 173     for (int i = 0; i < _class_count; i++) {
 174       if (_scratch_classes[i] != NULL) {
 175         ClassLoaderData* cld = _scratch_classes[i]->class_loader_data();
 176         // Free the memory for this class at class unloading time.  Not before
 177         // because CMS might think this is still live.
 178         InstanceKlass* ik = get_ik(_class_defs[i].klass);
 179         if (ik->get_cached_class_file() == _scratch_classes[i]->get_cached_class_file()) {
 180           // Don't double-free cached_class_file copied from the original class if error.
 181           _scratch_classes[i]->set_cached_class_file(NULL);
 182         }
 183         cld->add_to_deallocate_list(InstanceKlass::cast(_scratch_classes[i]));
 184       }
 185     }
 186     // Free os::malloc allocated memory in load_new_class_version.
 187     os::free(_scratch_classes);
 188     _timer_vm_op_prologue.stop();
 189     unlock_classes();
 190     return false;
 191   }
 192 
 193   _timer_vm_op_prologue.stop();
 194   return true;
 195 }

 197 void VM_RedefineClasses::doit() {
 198   Thread *thread = Thread::current();
 199 
 200 #if INCLUDE_CDS
 201   if (UseSharedSpaces) {
 202     // Sharing is enabled so we remap the shared readonly space to
 203     // shared readwrite, private just in case we need to redefine
 204     // a shared class. We do the remap during the doit() phase of
 205     // the safepoint to be safer.
 206     if (!MetaspaceShared::remap_shared_readonly_as_readwrite()) {
 207       log_info(redefine, class, load)("failed to remap shared readonly space to readwrite, private");
 208       _res = JVMTI_ERROR_INTERNAL;
 209       return;
 210     }
 211   }
 212 #endif
 213 
 214   // Mark methods seen on stack and everywhere else so old methods are not
 215   // cleaned up if they're on the stack.
 216   MetadataOnStackMark md_on_stack(/*walk_all_metadata*/true, /*redefinition_walk*/true);
 217   HandleMark hm(thread);   // make sure any handles created are deleted
 218                            // before the stack walk again.
 219 
 220   for (int i = 0; i < _class_count; i++) {
 221     redefine_single_class(_class_defs[i].klass, _scratch_classes[i], thread);
 222   }
 223 
 224   // Flush all compiled code that depends on the classes redefined.
 225   flush_dependent_code();
 226 
 227   // Adjust constantpool caches and vtables for all classes
 228   // that reference methods of the evolved classes.
 229   // Have to do this after all classes are redefined and all methods that
 230   // are redefined are marked as old.
 231   AdjustAndCleanMetadata adjust_and_clean_metadata(thread);
 232   ClassLoaderDataGraph::classes_do(&adjust_and_clean_metadata);
 233 
 234   // JSR-292 support
 235   if (_any_class_has_resolved_methods) {
 236     bool trace_name_printed = false;
 237     ResolvedMethodTable::adjust_method_entries(&trace_name_printed);
 238   }
 239 
 240   // Increment flag indicating that some invariants are no longer true.
 241   // See jvmtiExport.hpp for detailed explanation.
 242   JvmtiExport::increment_redefinition_count();
 243 
 244   // check_class() is optionally called for product bits, but is
 245   // always called for non-product bits.
 246 #ifdef PRODUCT
 247   if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
 248 #endif
 249     log_trace(redefine, class, obsolete, metadata)("calling check_class");
 250     CheckClass check_class(thread);
 251     ClassLoaderDataGraph::classes_do(&check_class);
 252 #ifdef PRODUCT
 253   }
 254 #endif
 255 
 256   // Clean up any metadata now unreferenced while MetadataOnStackMark is set.
 257   ClassLoaderDataGraph::clean_deallocate_lists(false);
 258 }

 260 void VM_RedefineClasses::doit_epilogue() {
 261   unlock_classes();
 262 
 263   // Free os::malloc allocated memory.
 264   os::free(_scratch_classes);
 265 
 266   // Reset the_class to null for error printing.
 267   _the_class = NULL;
 268 
 269   if (log_is_enabled(Info, redefine, class, timer)) {
 270     // Used to have separate timers for "doit" and "all", but the timer
 271     // overhead skewed the measurements.
 272     julong doit_time = _timer_rsc_phase1.milliseconds() +
 273                        _timer_rsc_phase2.milliseconds();
 274     julong all_time = _timer_vm_op_prologue.milliseconds() + doit_time;
 275 
 276     log_info(redefine, class, timer)
 277       ("vm_op: all=" JULONG_FORMAT "  prologue=" JULONG_FORMAT "  doit=" JULONG_FORMAT,
 278        all_time, (julong)_timer_vm_op_prologue.milliseconds(), doit_time);
 279     log_info(redefine, class, timer)
 280       ("redefine_single_class: phase1=" JULONG_FORMAT "  phase2=" JULONG_FORMAT,
 281        (julong)_timer_rsc_phase1.milliseconds(), (julong)_timer_rsc_phase2.milliseconds());
 282   }
 283 }

 285 bool VM_RedefineClasses::is_modifiable_class(oop klass_mirror) {
 286   // classes for primitives cannot be redefined
 287   if (java_lang_Class::is_primitive(klass_mirror)) {
 288     return false;
 289   }
 290   Klass* k = java_lang_Class::as_Klass(klass_mirror);
 291   // classes for arrays cannot be redefined
 292   if (k == NULL || !k->is_instance_klass()) {
 293     return false;
 294   }
 295 
 296   // Cannot redefine or retransform a hidden or an unsafe anonymous class.
 297   if (InstanceKlass::cast(k)->is_hidden() ||
 298       InstanceKlass::cast(k)->is_unsafe_anonymous()) {
 299     return false;
 300   }
 301   return true;
 302 }

 304 // Append the current entry at scratch_i in scratch_cp to *merge_cp_p
 305 // where the end of *merge_cp_p is specified by *merge_cp_length_p. For
 306 // direct CP entries, there is just the current entry to append. For
 307 // indirect and double-indirect CP entries, there are zero or more
 308 // referenced CP entries along with the current entry to append.
 309 // Indirect and double-indirect CP entries are handled by recursive
 310 // calls to append_entry() as needed. The referenced CP entries are
 311 // always appended to *merge_cp_p before the referee CP entry. These
 312 // referenced CP entries may already exist in *merge_cp_p in which case
 313 // there is nothing extra to append and only the current entry is
 314 // appended.
 315 void VM_RedefineClasses::append_entry(const constantPoolHandle& scratch_cp,
 316        int scratch_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p,
 317        TRAPS) {
 318 
 319   // append is different depending on entry tag type
 320   switch (scratch_cp->tag_at(scratch_i).value()) {
 321 
 322     // The old verifier is implemented outside the VM. It loads classes,
 323     // but does not resolve constant pool entries directly so we never
 324     // see Class entries here with the old verifier. Similarly the old
 325     // verifier does not like Class entries in the input constant pool.
 326     // The split-verifier is implemented in the VM so it can optionally
 327     // and directly resolve constant pool entries to load classes. The
 328     // split-verifier can accept either Class entries or UnresolvedClass
 329     // entries in the input constant pool. We revert the appended copy
 330     // back to UnresolvedClass so that either verifier will be happy
 331     // with the constant pool entry.
 332     //
 333     // this is an indirect CP entry so it needs special handling
 334     case JVM_CONSTANT_Class:
 335     case JVM_CONSTANT_UnresolvedClass:
 336     {
 337       int name_i = scratch_cp->klass_name_index_at(scratch_i);
 338       int new_name_i = find_or_append_indirect_entry(scratch_cp, name_i, merge_cp_p,
 339                                                      merge_cp_length_p, THREAD);
 340 
 341       if (new_name_i != name_i) {
 342         log_trace(redefine, class, constantpool)
 343           ("Class entry@%d name_index change: %d to %d",
 344            *merge_cp_length_p, name_i, new_name_i);
 345       }
 346 
 347       (*merge_cp_p)->temp_unresolved_klass_at_put(*merge_cp_length_p, new_name_i);
 348       if (scratch_i != *merge_cp_length_p) {
 349         // The new entry in *merge_cp_p is at a different index than
 350         // the new entry in scratch_cp so we need to map the index values.
 351         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 352       }
 353       (*merge_cp_length_p)++;
 354     } break;
 355 
 356     // these are direct CP entries so they can be directly appended,
 357     // but double and long take two constant pool entries
 358     case JVM_CONSTANT_Double:  // fall through
 359     case JVM_CONSTANT_Long:
 360     {
 361       ConstantPool::copy_entry_to(scratch_cp, scratch_i, *merge_cp_p, *merge_cp_length_p,
 362         THREAD);
 363 
 364       if (scratch_i != *merge_cp_length_p) {
 365         // The new entry in *merge_cp_p is at a different index than
 366         // the new entry in scratch_cp so we need to map the index values.
 367         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 368       }
 369       (*merge_cp_length_p) += 2;
 370     } break;
 371 
 372     // these are direct CP entries so they can be directly appended
 373     case JVM_CONSTANT_Float:   // fall through
 374     case JVM_CONSTANT_Integer: // fall through
 375     case JVM_CONSTANT_Utf8:    // fall through
 376 
 377     // This was an indirect CP entry, but it has been changed into
 378     // Symbol*s so this entry can be directly appended.
 379     case JVM_CONSTANT_String:      // fall through
 380     {
 381       ConstantPool::copy_entry_to(scratch_cp, scratch_i, *merge_cp_p, *merge_cp_length_p,
 382         THREAD);
 383 
 384       if (scratch_i != *merge_cp_length_p) {
 385         // The new entry in *merge_cp_p is at a different index than
 386         // the new entry in scratch_cp so we need to map the index values.
 387         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 388       }
 389       (*merge_cp_length_p)++;
 390     } break;
 391 
 392     // this is an indirect CP entry so it needs special handling
 393     case JVM_CONSTANT_NameAndType:
 394     {
 395       int name_ref_i = scratch_cp->name_ref_index_at(scratch_i);
 396       int new_name_ref_i = find_or_append_indirect_entry(scratch_cp, name_ref_i, merge_cp_p,
 397                                                          merge_cp_length_p, THREAD);
 398 
 399       int signature_ref_i = scratch_cp->signature_ref_index_at(scratch_i);
 400       int new_signature_ref_i = find_or_append_indirect_entry(scratch_cp, signature_ref_i,
 401                                                               merge_cp_p, merge_cp_length_p,
 402                                                               THREAD);
 403 
 404       // If the referenced entries already exist in *merge_cp_p, then
 405       // both new_name_ref_i and new_signature_ref_i will both be 0.
 406       // In that case, all we are appending is the current entry.
 407       if (new_name_ref_i != name_ref_i) {
 408         log_trace(redefine, class, constantpool)
 409           ("NameAndType entry@%d name_ref_index change: %d to %d",
 410            *merge_cp_length_p, name_ref_i, new_name_ref_i);
 411       }
 412       if (new_signature_ref_i != signature_ref_i) {
 413         log_trace(redefine, class, constantpool)
 414           ("NameAndType entry@%d signature_ref_index change: %d to %d",
 415            *merge_cp_length_p, signature_ref_i, new_signature_ref_i);
 416       }
 417 
 418       (*merge_cp_p)->name_and_type_at_put(*merge_cp_length_p,
 419         new_name_ref_i, new_signature_ref_i);
 420       if (scratch_i != *merge_cp_length_p) {
 421         // The new entry in *merge_cp_p is at a different index than
 422         // the new entry in scratch_cp so we need to map the index values.
 423         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 424       }
 425       (*merge_cp_length_p)++;
 426     } break;
 427 
 428     // this is a double-indirect CP entry so it needs special handling
 429     case JVM_CONSTANT_Fieldref:           // fall through
 430     case JVM_CONSTANT_InterfaceMethodref: // fall through
 431     case JVM_CONSTANT_Methodref:
 432     {
 433       int klass_ref_i = scratch_cp->uncached_klass_ref_index_at(scratch_i);
 434       int new_klass_ref_i = find_or_append_indirect_entry(scratch_cp, klass_ref_i,
 435                                                           merge_cp_p, merge_cp_length_p, THREAD);
 436 
 437       int name_and_type_ref_i = scratch_cp->uncached_name_and_type_ref_index_at(scratch_i);
 438       int new_name_and_type_ref_i = find_or_append_indirect_entry(scratch_cp, name_and_type_ref_i,
 439                                                           merge_cp_p, merge_cp_length_p, THREAD);
 440 
 441       const char *entry_name = NULL;
 442       switch (scratch_cp->tag_at(scratch_i).value()) {
 443       case JVM_CONSTANT_Fieldref:
 444         entry_name = "Fieldref";
 445         (*merge_cp_p)->field_at_put(*merge_cp_length_p, new_klass_ref_i,
 446           new_name_and_type_ref_i);
 447         break;
 448       case JVM_CONSTANT_InterfaceMethodref:
 449         entry_name = "IFMethodref";
 450         (*merge_cp_p)->interface_method_at_put(*merge_cp_length_p,
 451           new_klass_ref_i, new_name_and_type_ref_i);
 452         break;
 453       case JVM_CONSTANT_Methodref:
 454         entry_name = "Methodref";
 455         (*merge_cp_p)->method_at_put(*merge_cp_length_p, new_klass_ref_i,
 456           new_name_and_type_ref_i);
 457         break;
 458       default:
 459         guarantee(false, "bad switch");
 460         break;
 461       }
 462 
 463       if (klass_ref_i != new_klass_ref_i) {
 464         log_trace(redefine, class, constantpool)
 465           ("%s entry@%d class_index changed: %d to %d", entry_name, *merge_cp_length_p, klass_ref_i, new_klass_ref_i);
 466       }
 467       if (name_and_type_ref_i != new_name_and_type_ref_i) {
 468         log_trace(redefine, class, constantpool)
 469           ("%s entry@%d name_and_type_index changed: %d to %d",
 470            entry_name, *merge_cp_length_p, name_and_type_ref_i, new_name_and_type_ref_i);
 471       }
 472 
 473       if (scratch_i != *merge_cp_length_p) {
 474         // The new entry in *merge_cp_p is at a different index than
 475         // the new entry in scratch_cp so we need to map the index values.
 476         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 477       }
 478       (*merge_cp_length_p)++;
 479     } break;
 480 
 481     // this is an indirect CP entry so it needs special handling
 482     case JVM_CONSTANT_MethodType:
 483     {
 484       int ref_i = scratch_cp->method_type_index_at(scratch_i);
 485       int new_ref_i = find_or_append_indirect_entry(scratch_cp, ref_i, merge_cp_p,
 486                                                     merge_cp_length_p, THREAD);
 487       if (new_ref_i != ref_i) {
 488         log_trace(redefine, class, constantpool)
 489           ("MethodType entry@%d ref_index change: %d to %d", *merge_cp_length_p, ref_i, new_ref_i);
 490       }
 491       (*merge_cp_p)->method_type_index_at_put(*merge_cp_length_p, new_ref_i);
 492       if (scratch_i != *merge_cp_length_p) {
 493         // The new entry in *merge_cp_p is at a different index than
 494         // the new entry in scratch_cp so we need to map the index values.
 495         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 496       }
 497       (*merge_cp_length_p)++;
 498     } break;
 499 
 500     // this is an indirect CP entry so it needs special handling
 501     case JVM_CONSTANT_MethodHandle:
 502     {
 503       int ref_kind = scratch_cp->method_handle_ref_kind_at(scratch_i);
 504       int ref_i = scratch_cp->method_handle_index_at(scratch_i);
 505       int new_ref_i = find_or_append_indirect_entry(scratch_cp, ref_i, merge_cp_p,
 506                                                     merge_cp_length_p, THREAD);
 507       if (new_ref_i != ref_i) {
 508         log_trace(redefine, class, constantpool)
 509           ("MethodHandle entry@%d ref_index change: %d to %d", *merge_cp_length_p, ref_i, new_ref_i);
 510       }
 511       (*merge_cp_p)->method_handle_index_at_put(*merge_cp_length_p, ref_kind, new_ref_i);
 512       if (scratch_i != *merge_cp_length_p) {
 513         // The new entry in *merge_cp_p is at a different index than
 514         // the new entry in scratch_cp so we need to map the index values.
 515         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 516       }
 517       (*merge_cp_length_p)++;
 518     } break;
 519 
 520     // this is an indirect CP entry so it needs special handling
 521     case JVM_CONSTANT_Dynamic:  // fall through
 522     case JVM_CONSTANT_InvokeDynamic:
 523     {
 524       // Index of the bootstrap specifier in the operands array
 525       int old_bs_i = scratch_cp->bootstrap_methods_attribute_index(scratch_i);
 526       int new_bs_i = find_or_append_operand(scratch_cp, old_bs_i, merge_cp_p,
 527                                             merge_cp_length_p, THREAD);
 528       // The bootstrap method NameAndType_info index
 529       int old_ref_i = scratch_cp->bootstrap_name_and_type_ref_index_at(scratch_i);
 530       int new_ref_i = find_or_append_indirect_entry(scratch_cp, old_ref_i, merge_cp_p,
 531                                                     merge_cp_length_p, THREAD);
 532       if (new_bs_i != old_bs_i) {
 533         log_trace(redefine, class, constantpool)
 534           ("Dynamic entry@%d bootstrap_method_attr_index change: %d to %d",
 535            *merge_cp_length_p, old_bs_i, new_bs_i);
 536       }
 537       if (new_ref_i != old_ref_i) {
 538         log_trace(redefine, class, constantpool)
 539           ("Dynamic entry@%d name_and_type_index change: %d to %d", *merge_cp_length_p, old_ref_i, new_ref_i);
 540       }
 541 
 542       if (scratch_cp->tag_at(scratch_i).is_dynamic_constant())
 543         (*merge_cp_p)->dynamic_constant_at_put(*merge_cp_length_p, new_bs_i, new_ref_i);
 544       else
 545         (*merge_cp_p)->invoke_dynamic_at_put(*merge_cp_length_p, new_bs_i, new_ref_i);
 546       if (scratch_i != *merge_cp_length_p) {
 547         // The new entry in *merge_cp_p is at a different index than
 548         // the new entry in scratch_cp so we need to map the index values.
 549         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 550       }
 551       (*merge_cp_length_p)++;
 552     } break;
 553 
 554     // At this stage, Class or UnresolvedClass could be in scratch_cp, but not
 555     // ClassIndex
 556     case JVM_CONSTANT_ClassIndex: // fall through
 557 
 558     // Invalid is used as the tag for the second constant pool entry
 559     // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
 560     // not be seen by itself.
 561     case JVM_CONSTANT_Invalid: // fall through
 562 
 563     // At this stage, String could be here, but not StringIndex
 564     case JVM_CONSTANT_StringIndex: // fall through
 565 
 566     // At this stage JVM_CONSTANT_UnresolvedClassInError should not be
 567     // here
 568     case JVM_CONSTANT_UnresolvedClassInError: // fall through
 569 
 570     default:
 571     {
 572       // leave a breadcrumb
 573       jbyte bad_value = scratch_cp->tag_at(scratch_i).value();
 574       ShouldNotReachHere();
 575     } break;
 576   } // end switch tag value
 577 } // end append_entry()

 580 int VM_RedefineClasses::find_or_append_indirect_entry(const constantPoolHandle& scratch_cp,
 581       int ref_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 582 
 583   int new_ref_i = ref_i;
 584   bool match = (ref_i < *merge_cp_length_p) &&
 585                scratch_cp->compare_entry_to(ref_i, *merge_cp_p, ref_i, THREAD);
 586 
 587   if (!match) {
 588     // forward reference in *merge_cp_p or not a direct match
 589     int found_i = scratch_cp->find_matching_entry(ref_i, *merge_cp_p, THREAD);
 590     if (found_i != 0) {
 591       guarantee(found_i != ref_i, "compare_entry_to() and find_matching_entry() do not agree");
 592       // Found a matching entry somewhere else in *merge_cp_p so just need a mapping entry.
 593       new_ref_i = found_i;
 594       map_index(scratch_cp, ref_i, found_i);
 595     } else {
 596       // no match found so we have to append this entry to *merge_cp_p
 597       append_entry(scratch_cp, ref_i, merge_cp_p, merge_cp_length_p, THREAD);
 598       // The above call to append_entry() can only append one entry
 599       // so the post call query of *merge_cp_length_p is only for
 600       // the sake of consistency.
 601       new_ref_i = *merge_cp_length_p - 1;
 602     }
 603   }
 604 
 605   return new_ref_i;
 606 } // end find_or_append_indirect_entry()

 609 // Append a bootstrap specifier into the merge_cp operands that is semantically equal
 610 // to the scratch_cp operands bootstrap specifier passed by the old_bs_i index.
 611 // Recursively append new merge_cp entries referenced by the new bootstrap specifier.
 612 void VM_RedefineClasses::append_operand(const constantPoolHandle& scratch_cp, int old_bs_i,
 613        constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 614 
 615   int old_ref_i = scratch_cp->operand_bootstrap_method_ref_index_at(old_bs_i);
 616   int new_ref_i = find_or_append_indirect_entry(scratch_cp, old_ref_i, merge_cp_p,
 617                                                 merge_cp_length_p, THREAD);
 618   if (new_ref_i != old_ref_i) {
 619     log_trace(redefine, class, constantpool)
 620       ("operands entry@%d bootstrap method ref_index change: %d to %d", _operands_cur_length, old_ref_i, new_ref_i);
 621   }
 622 
 623   Array<u2>* merge_ops = (*merge_cp_p)->operands();
 624   int new_bs_i = _operands_cur_length;
 625   // We have _operands_cur_length == 0 when the merge_cp operands is empty yet.
 626   // However, the operand_offset_at(0) was set in the extend_operands() call.
 627   int new_base = (new_bs_i == 0) ? (*merge_cp_p)->operand_offset_at(0)
 628                                  : (*merge_cp_p)->operand_next_offset_at(new_bs_i - 1);
 629   int argc     = scratch_cp->operand_argument_count_at(old_bs_i);
 630 
 631   ConstantPool::operand_offset_at_put(merge_ops, _operands_cur_length, new_base);
 632   merge_ops->at_put(new_base++, new_ref_i);
 633   merge_ops->at_put(new_base++, argc);
 634 
 635   for (int i = 0; i < argc; i++) {
 636     int old_arg_ref_i = scratch_cp->operand_argument_index_at(old_bs_i, i);
 637     int new_arg_ref_i = find_or_append_indirect_entry(scratch_cp, old_arg_ref_i, merge_cp_p,
 638                                                       merge_cp_length_p, THREAD);
 639     merge_ops->at_put(new_base++, new_arg_ref_i);
 640     if (new_arg_ref_i != old_arg_ref_i) {
 641       log_trace(redefine, class, constantpool)
 642         ("operands entry@%d bootstrap method argument ref_index change: %d to %d",
 643          _operands_cur_length, old_arg_ref_i, new_arg_ref_i);
 644     }
 645   }
 646   if (old_bs_i != _operands_cur_length) {
 647     // The bootstrap specifier in *merge_cp_p is at a different index than
 648     // that in scratch_cp so we need to map the index values.
 649     map_operand_index(old_bs_i, new_bs_i);
 650   }
 651   _operands_cur_length++;
 652 } // end append_operand()

 655 int VM_RedefineClasses::find_or_append_operand(const constantPoolHandle& scratch_cp,
 656       int old_bs_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 657 
 658   int new_bs_i = old_bs_i; // bootstrap specifier index
 659   bool match = (old_bs_i < _operands_cur_length) &&
 660                scratch_cp->compare_operand_to(old_bs_i, *merge_cp_p, old_bs_i, THREAD);
 661 
 662   if (!match) {
 663     // forward reference in *merge_cp_p or not a direct match
 664     int found_i = scratch_cp->find_matching_operand(old_bs_i, *merge_cp_p,
 665                                                     _operands_cur_length, THREAD);
 666     if (found_i != -1) {
 667       guarantee(found_i != old_bs_i, "compare_operand_to() and find_matching_operand() disagree");
 668       // found a matching operand somewhere else in *merge_cp_p so just need a mapping
 669       new_bs_i = found_i;
 670       map_operand_index(old_bs_i, found_i);
 671     } else {
 672       // no match found so we have to append this bootstrap specifier to *merge_cp_p
 673       append_operand(scratch_cp, old_bs_i, merge_cp_p, merge_cp_length_p, THREAD);
 674       new_bs_i = _operands_cur_length - 1;
 675     }
 676   }
 677   return new_bs_i;
 678 } // end find_or_append_operand()

 713 // The caller must have an active ResourceMark.
 714 static jvmtiError check_attribute_arrays(const char* attr_name,
 715            InstanceKlass* the_class, InstanceKlass* scratch_class,
 716            Array<u2>* the_array, Array<u2>* scr_array) {
 717   bool the_array_exists = the_array != Universe::the_empty_short_array();
 718   bool scr_array_exists = scr_array != Universe::the_empty_short_array();
 719 
 720   int array_len = the_array->length();
 721   if (the_array_exists && scr_array_exists) {
 722     if (array_len != scr_array->length()) {
 723       log_trace(redefine, class)
 724         ("redefined class %s attribute change error: %s len=%d changed to len=%d",
 725          the_class->external_name(), attr_name, array_len, scr_array->length());
 726       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 727     }
 728 
 729     // The order of entries in the attribute array is not specified so we
 730     // have to explicitly check for the same contents. We do this by copying
 731     // the referenced symbols into their own arrays, sorting them and then
 732     // comparing each element pair.
 733 
 734     Symbol** the_syms = NEW_RESOURCE_ARRAY_RETURN_NULL(Symbol*, array_len);
 735     Symbol** scr_syms = NEW_RESOURCE_ARRAY_RETURN_NULL(Symbol*, array_len);
 736 
 737     if (the_syms == NULL || scr_syms == NULL) {
 738       return JVMTI_ERROR_OUT_OF_MEMORY;
 739     }
 740 
 741     for (int i = 0; i < array_len; i++) {
 742       int the_cp_index = the_array->at(i);
 743       int scr_cp_index = scr_array->at(i);
 744       the_syms[i] = the_class->constants()->klass_name_at(the_cp_index);
 745       scr_syms[i] = scratch_class->constants()->klass_name_at(scr_cp_index);
 746     }
 747 
 748     qsort(the_syms, array_len, sizeof(Symbol*), symcmp);
 749     qsort(scr_syms, array_len, sizeof(Symbol*), symcmp);
 750 
 751     for (int i = 0; i < array_len; i++) {
 752       if (the_syms[i] != scr_syms[i]) {
 753         log_trace(redefine, class)
 754           ("redefined class %s attribute change error: %s[%d]: %s changed to %s",
 755            the_class->external_name(), attr_name, i,
 756            the_syms[i]->as_C_string(), scr_syms[i]->as_C_string());
 757         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 758       }
 759     }
 760   } else if (the_array_exists ^ scr_array_exists) {
 761     const char* action_str = (the_array_exists) ? "removed" : "added";
 762     log_trace(redefine, class)
 763       ("redefined class %s attribute change error: %s attribute %s",
 764        the_class->external_name(), attr_name, action_str);
 765     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 766   }
 767   return JVMTI_ERROR_NONE;
 768 }

 804   // Get lists of record components.
 805   Array<RecordComponent*>* the_record = the_class->record_components();
 806   Array<RecordComponent*>* scr_record = scratch_class->record_components();
 807   bool the_record_exists = the_record != NULL;

 816          the_class->external_name(), the_num_components, scr_num_components);
 817       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 818     }
 819 
 820     // Compare each field in each record component.
 821     ConstantPool* the_cp =  the_class->constants();
 822     ConstantPool* scr_cp =  scratch_class->constants();
 823     for (int x = 0; x < the_num_components; x++) {
 824       RecordComponent* the_component = the_record->at(x);
 825       RecordComponent* scr_component = scr_record->at(x);
 826       const Symbol* const the_name = the_cp->symbol_at(the_component->name_index());
 827       const Symbol* const scr_name = scr_cp->symbol_at(scr_component->name_index());
 828       const Symbol* const the_descr = the_cp->symbol_at(the_component->descriptor_index());
 829       const Symbol* const scr_descr = scr_cp->symbol_at(scr_component->descriptor_index());
 830       if (the_name != scr_name || the_descr != scr_descr) {
 831         log_trace(redefine, class, record)
 832           ("redefined class %s attribute change error: Record name_index, descriptor_index, and/or attributes_count changed",
 833            the_class->external_name());
 834         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 835       }
 836 
 837       int the_gen_sig = the_component->generic_signature_index();
 838       int scr_gen_sig = scr_component->generic_signature_index();
 839       const Symbol* const the_gen_sig_sym = (the_gen_sig == 0 ? NULL :
 840         the_cp->symbol_at(the_component->generic_signature_index()));
 841       const Symbol* const scr_gen_sig_sym = (scr_gen_sig == 0 ? NULL :
 842         scr_cp->symbol_at(scr_component->generic_signature_index()));
 843       if (the_gen_sig_sym != scr_gen_sig_sym) {
 844         log_trace(redefine, class, record)
 845           ("redefined class %s attribute change error: Record generic_signature attribute changed",
 846            the_class->external_name());
 847         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 848       }
 849 
 850       // It's okay if a record component's annotations were changed.
 851     }
 852 
 853   } else if (the_record_exists ^ scr_record_exists) {
 854     const char* action_str = (the_record_exists) ? "removed" : "added";
 855     log_trace(redefine, class, record)
 856       ("redefined class %s attribute change error: Record attribute %s",
 857        the_class->external_name(), action_str);
 858     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 859   }
 860 
 861   return JVMTI_ERROR_NONE;
 862 }

 882 
 883 jvmtiError VM_RedefineClasses::compare_and_normalize_class_versions(
 884              InstanceKlass* the_class,
 885              InstanceKlass* scratch_class) {
 886   int i;
 887 
 888   // Check superclasses, or rather their names, since superclasses themselves can be
 889   // requested to replace.
 890   // Check for NULL superclass first since this might be java.lang.Object
 891   if (the_class->super() != scratch_class->super() &&
 892       (the_class->super() == NULL || scratch_class->super() == NULL ||
 893        the_class->super()->name() !=
 894        scratch_class->super()->name())) {
 895     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 896   }
 897 
 898   // Check if the number, names and order of directly implemented interfaces are the same.
 899   // I think in principle we should just check if the sets of names of directly implemented
 900   // interfaces are the same, i.e. the order of declaration (which, however, if changed in the
 901   // .java file, also changes in .class file) should not matter. However, comparing sets is
 902   // technically a bit more difficult, and, more importantly, I am not sure at present that the
 903   // order of interfaces does not matter on the implementation level, i.e. that the VM does not
 904   // rely on it somewhere.
 905   Array<InstanceKlass*>* k_interfaces = the_class->local_interfaces();
 906   Array<InstanceKlass*>* k_new_interfaces = scratch_class->local_interfaces();
 907   int n_intfs = k_interfaces->length();
 908   if (n_intfs != k_new_interfaces->length()) {
 909     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 910   }
 911   for (i = 0; i < n_intfs; i++) {
 912     if (k_interfaces->at(i)->name() !=
 913         k_new_interfaces->at(i)->name()) {
 914       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 915     }
 916   }
 917 
 918   // Check whether class is in the error init state.
 919   if (the_class->is_in_error_state()) {
 920     // TBD #5057930: special error code is needed in 1.6
 921     return JVMTI_ERROR_INVALID_CLASS;
 922   }
 923 
 924   // Check whether the nest-related attributes have been changed.
 925   jvmtiError err = check_nest_attributes(the_class, scratch_class);
 926   if (err != JVMTI_ERROR_NONE) {
 927     return err;
 928   }
 929 
 930   // Check whether the Record attribute has been changed.
 931   err = check_record_attribute(the_class, scratch_class);
 932   if (err != JVMTI_ERROR_NONE) {
 933     return err;
 934   }
 935 
 936   // Check whether the PermittedSubclasses attribute has been changed.
 937   err = check_permitted_subclasses_attribute(the_class, scratch_class);
 938   if (err != JVMTI_ERROR_NONE) {
 939     return err;
 940   }
 941 
 942   // Check whether class modifiers are the same.
 943   jushort old_flags = (jushort) the_class->access_flags().get_flags();
 944   jushort new_flags = (jushort) scratch_class->access_flags().get_flags();
 945   if (old_flags != new_flags) {
 946     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_MODIFIERS_CHANGED;
 947   }
 948 
 949   // Check if the number, names, types and order of fields declared in these classes
 950   // are the same.
 951   JavaFieldStream old_fs(the_class);
 952   JavaFieldStream new_fs(scratch_class);
 953   for (; !old_fs.done() && !new_fs.done(); old_fs.next(), new_fs.next()) {
 954     // access
 955     old_flags = old_fs.access_flags().as_short();
 956     new_flags = new_fs.access_flags().as_short();
 957     if ((old_flags ^ new_flags) & JVM_RECOGNIZED_FIELD_MODIFIERS) {
 958       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 959     }
 960     // offset
 961     if (old_fs.offset() != new_fs.offset()) {
 962       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 963     }
 964     // name and signature
 965     Symbol* name_sym1 = the_class->constants()->symbol_at(old_fs.name_index());
 966     Symbol* sig_sym1 = the_class->constants()->symbol_at(old_fs.signature_index());
 967     Symbol* name_sym2 = scratch_class->constants()->symbol_at(new_fs.name_index());
 968     Symbol* sig_sym2 = scratch_class->constants()->symbol_at(new_fs.signature_index());
 969     if (name_sym1 != name_sym2 || sig_sym1 != sig_sym2) {
 970       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 971     }
 972   }
 973 
 974   // If both streams aren't done then we have a differing number of
 975   // fields.
 976   if (!old_fs.done() || !new_fs.done()) {
 977     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 978   }
 979 
 980   // Do a parallel walk through the old and new methods. Detect
 981   // cases where they match (exist in both), have been added in
 982   // the new methods, or have been deleted (exist only in the
 983   // old methods).  The class file parser places methods in order
 984   // by method name, but does not order overloaded methods by
 985   // signature.  In order to determine what fate befell the methods,
 986   // this code places the overloaded new methods that have matching
 987   // old methods in the same order as the old methods and places
 988   // new overloaded methods at the end of overloaded methods of
 989   // that name. The code for this order normalization is adapted
 990   // from the algorithm used in InstanceKlass::find_method().
 991   // Since we are swapping out of order entries as we find them,
 992   // we only have to search forward through the overloaded methods.
 993   // Methods which are added and have the same name as an existing
 994   // method (but different signature) will be put at the end of
 995   // the methods with that name, and the name mismatch code will
 996   // handle them.
 997   Array<Method*>* k_old_methods(the_class->methods());
 998   Array<Method*>* k_new_methods(scratch_class->methods());
 999   int n_old_methods = k_old_methods->length();
1000   int n_new_methods = k_new_methods->length();
1001   Thread* thread = Thread::current();
1002 
1003   int ni = 0;
1004   int oi = 0;
1005   while (true) {
1006     Method* k_old_method;
1007     Method* k_new_method;
1008     enum { matched, added, deleted, undetermined } method_was = undetermined;
1009 
1010     if (oi >= n_old_methods) {
1011       if (ni >= n_new_methods) {
1012         break; // we've looked at everything, done
1013       }
1014       // New method at the end
1015       k_new_method = k_new_methods->at(ni);
1016       method_was = added;
1017     } else if (ni >= n_new_methods) {
1018       // Old method, at the end, is deleted
1019       k_old_method = k_old_methods->at(oi);
1020       method_was = deleted;
1021     } else {
1022       // There are more methods in both the old and new lists
1023       k_old_method = k_old_methods->at(oi);
1024       k_new_method = k_new_methods->at(ni);
1025       if (k_old_method->name() != k_new_method->name()) {
1026         // Methods are sorted by method name, so a mismatch means added
1027         // or deleted
1028         if (k_old_method->name()->fast_compare(k_new_method->name()) > 0) {
1029           method_was = added;
1030         } else {
1031           method_was = deleted;
1032         }
1033       } else if (k_old_method->signature() == k_new_method->signature()) {
1034         // Both the name and signature match
1035         method_was = matched;
1036       } else {
1037         // The name matches, but the signature doesn't, which means we have to
1038         // search forward through the new overloaded methods.
1039         int nj;  // outside the loop for post-loop check
1040         for (nj = ni + 1; nj < n_new_methods; nj++) {
1041           Method* m = k_new_methods->at(nj);
1042           if (k_old_method->name() != m->name()) {
1043             // reached another method name so no more overloaded methods
1044             method_was = deleted;
1045             break;
1046           }
1047           if (k_old_method->signature() == m->signature()) {
1048             // found a match so swap the methods
1049             k_new_methods->at_put(ni, m);
1050             k_new_methods->at_put(nj, k_new_method);
1051             k_new_method = m;
1052             method_was = matched;
1053             break;
1054           }
1055         }
1056 
1057         if (nj >= n_new_methods) {
1058           // reached the end without a match; so method was deleted
1059           method_was = deleted;
1060         }
1061       }
1062     }
1063 
1064     switch (method_was) {
1065     case matched:
1066       // methods match, be sure modifiers do too
1067       old_flags = (jushort) k_old_method->access_flags().get_flags();
1068       new_flags = (jushort) k_new_method->access_flags().get_flags();
1069       if ((old_flags ^ new_flags) & ~(JVM_ACC_NATIVE)) {
1070         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_MODIFIERS_CHANGED;
1071       }
1072       {
1073         u2 new_num = k_new_method->method_idnum();
1074         u2 old_num = k_old_method->method_idnum();
1075         if (new_num != old_num) {
1076           Method* idnum_owner = scratch_class->method_with_idnum(old_num);
1077           if (idnum_owner != NULL) {
1078             // There is already a method assigned this idnum -- switch them
1079             // Take current and original idnum from the new_method
1080             idnum_owner->set_method_idnum(new_num);
1081             idnum_owner->set_orig_method_idnum(k_new_method->orig_method_idnum());
1082           }
1083           // Take current and original idnum from the old_method
1084           k_new_method->set_method_idnum(old_num);
1085           k_new_method->set_orig_method_idnum(k_old_method->orig_method_idnum());
1086           if (thread->has_pending_exception()) {
1087             return JVMTI_ERROR_OUT_OF_MEMORY;
1088           }
1089         }
1090       }
1091       log_trace(redefine, class, normalize)
1092         ("Method matched: new: %s [%d] == old: %s [%d]",
1093          k_new_method->name_and_sig_as_C_string(), ni, k_old_method->name_and_sig_as_C_string(), oi);
1094       // advance to next pair of methods
1095       ++oi;
1096       ++ni;
1097       break;
1098     case added:
1099       // method added, see if it is OK
1100       if (!can_add_or_delete(k_new_method)) {
1101         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_ADDED;
1102       }
1103       {
1104         u2 num = the_class->next_method_idnum();
1105         if (num == ConstMethod::UNSET_IDNUM) {
1106           // cannot add any more methods
1107           return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_ADDED;
1108         }
1109         u2 new_num = k_new_method->method_idnum();
1110         Method* idnum_owner = scratch_class->method_with_idnum(num);
1111         if (idnum_owner != NULL) {
1112           // There is already a method assigned this idnum -- switch them
1113           // Take current and original idnum from the new_method
1114           idnum_owner->set_method_idnum(new_num);
1115           idnum_owner->set_orig_method_idnum(k_new_method->orig_method_idnum());
1116         }
1117         k_new_method->set_method_idnum(num);
1118         k_new_method->set_orig_method_idnum(num);
1119         if (thread->has_pending_exception()) {
1120           return JVMTI_ERROR_OUT_OF_MEMORY;
1121         }
1122       }
1123       log_trace(redefine, class, normalize)
1124         ("Method added: new: %s [%d]", k_new_method->name_and_sig_as_C_string(), ni);
1125       ++ni; // advance to next new method
1126       break;
1127     case deleted:
1128       // method deleted, see if it is OK
1129       if (!can_add_or_delete(k_old_method)) {
1130         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_DELETED;
1131       }
1132       log_trace(redefine, class, normalize)
1133         ("Method deleted: old: %s [%d]", k_old_method->name_and_sig_as_C_string(), oi);
1134       ++oi; // advance to next old method
1135       break;
1136     default:
1137       ShouldNotReachHere();
1138     }
1139   }
1140 
1141   return JVMTI_ERROR_NONE;
1142 }

1200

1455 // Merge old_cp and scratch_cp and return the results of the merge via
1456 // merge_cp_p. The number of entries in *merge_cp_p is returned via
1457 // merge_cp_length_p. The entries in old_cp occupy the same locations
1458 // in *merge_cp_p. Also creates a map of indices from entries in
1459 // scratch_cp to the corresponding entry in *merge_cp_p. Index map
1460 // entries are only created for entries in scratch_cp that occupy a
1461 // different location in *merged_cp_p.
1462 bool VM_RedefineClasses::merge_constant_pools(const constantPoolHandle& old_cp,
1463        const constantPoolHandle& scratch_cp, constantPoolHandle *merge_cp_p,
1464        int *merge_cp_length_p, TRAPS) {
1465 
1466   if (merge_cp_p == NULL) {
1467     assert(false, "caller must provide scratch constantPool");
1468     return false; // robustness
1469   }
1470   if (merge_cp_length_p == NULL) {
1471     assert(false, "caller must provide scratch CP length");
1472     return false; // robustness
1473   }
1474   // Worst case we need old_cp->length() + scratch_cp()->length(),
1475   // but the caller might be smart so make sure we have at least
1476   // the minimum.
1477   if ((*merge_cp_p)->length() < old_cp->length()) {
1478     assert(false, "merge area too small");
1479     return false; // robustness
1480   }
1481 
1482   log_info(redefine, class, constantpool)("old_cp_len=%d, scratch_cp_len=%d", old_cp->length(), scratch_cp->length());
1483 
1484   {
1485     // Pass 0:
1486     // The old_cp is copied to *merge_cp_p; this means that any code
1487     // using old_cp does not have to change. This work looks like a
1488     // perfect fit for ConstantPool*::copy_cp_to(), but we need to
1489     // handle one special case:
1490     // - revert JVM_CONSTANT_Class to JVM_CONSTANT_UnresolvedClass
1491     // This will make verification happy.
1492 
1493     int old_i;  // index into old_cp
1494 
1495     // index zero (0) is not used in constantPools
1496     for (old_i = 1; old_i < old_cp->length(); old_i++) {
1497       // leave debugging crumb
1498       jbyte old_tag = old_cp->tag_at(old_i).value();
1499       switch (old_tag) {
1500       case JVM_CONSTANT_Class:
1501       case JVM_CONSTANT_UnresolvedClass:
1502         // revert the copy to JVM_CONSTANT_UnresolvedClass
1503         // May be resolving while calling this so do the same for
1504         // JVM_CONSTANT_UnresolvedClass (klass_name_at() deals with transition)
1505         (*merge_cp_p)->temp_unresolved_klass_at_put(old_i,
1506           old_cp->klass_name_index_at(old_i));
1507         break;
1508 
1509       case JVM_CONSTANT_Double:
1510       case JVM_CONSTANT_Long:
1511         // just copy the entry to *merge_cp_p, but double and long take
1512         // two constant pool entries
1513         ConstantPool::copy_entry_to(old_cp, old_i, *merge_cp_p, old_i, CHECK_false);
1514         old_i++;
1515         break;
1516 
1517       default:
1518         // just copy the entry to *merge_cp_p
1519         ConstantPool::copy_entry_to(old_cp, old_i, *merge_cp_p, old_i, CHECK_false);
1520         break;
1521       }
1522     } // end for each old_cp entry
1523 
1524     ConstantPool::copy_operands(old_cp, *merge_cp_p, CHECK_false);
1525     (*merge_cp_p)->extend_operands(scratch_cp, CHECK_false);
1526 
1527     // We don't need to sanity check that *merge_cp_length_p is within
1528     // *merge_cp_p bounds since we have the minimum on-entry check above.
1529     (*merge_cp_length_p) = old_i;
1530   }
1531 
1532   // merge_cp_len should be the same as old_cp->length() at this point
1533   // so this trace message is really a "warm-and-breathing" message.
1534   log_debug(redefine, class, constantpool)("after pass 0: merge_cp_len=%d", *merge_cp_length_p);
1535 
1536   int scratch_i;  // index into scratch_cp
1537   {
1538     // Pass 1a:
1539     // Compare scratch_cp entries to the old_cp entries that we have
1540     // already copied to *merge_cp_p. In this pass, we are eliminating
1541     // exact duplicates (matching entry at same index) so we only
1542     // compare entries in the common indice range.
1543     int increment = 1;
1544     int pass1a_length = MIN2(old_cp->length(), scratch_cp->length());
1545     for (scratch_i = 1; scratch_i < pass1a_length; scratch_i += increment) {
1546       switch (scratch_cp->tag_at(scratch_i).value()) {
1547       case JVM_CONSTANT_Double:
1548       case JVM_CONSTANT_Long:
1549         // double and long take two constant pool entries
1550         increment = 2;
1551         break;
1552 
1553       default:
1554         increment = 1;
1555         break;
1556       }
1557 
1558       bool match = scratch_cp->compare_entry_to(scratch_i, *merge_cp_p,
1559         scratch_i, CHECK_false);
1560       if (match) {
1561         // found a match at the same index so nothing more to do
1562         continue;
1563       } else if (is_unresolved_class_mismatch(scratch_cp, scratch_i,
1564                                               *merge_cp_p, scratch_i)) {
1565         // The mismatch in compare_entry_to() above is because of a
1566         // resolved versus unresolved class entry at the same index
1567         // with the same string value. Since Pass 0 reverted any
1568         // class entries to unresolved class entries in *merge_cp_p,
1569         // we go with the unresolved class entry.
1570         continue;
1571       }
1572 
1573       int found_i = scratch_cp->find_matching_entry(scratch_i, *merge_cp_p,
1574         CHECK_false);
1575       if (found_i != 0) {
1576         guarantee(found_i != scratch_i,
1577           "compare_entry_to() and find_matching_entry() do not agree");
1578 
1579         // Found a matching entry somewhere else in *merge_cp_p so
1580         // just need a mapping entry.
1581         map_index(scratch_cp, scratch_i, found_i);
1582         continue;
1583       }
1584 
1585       // The find_matching_entry() call above could fail to find a match
1586       // due to a resolved versus unresolved class or string entry situation
1587       // like we solved above with the is_unresolved_*_mismatch() calls.
1588       // However, we would have to call is_unresolved_*_mismatch() over
1589       // all of *merge_cp_p (potentially) and that doesn't seem to be
1590       // worth the time.
1591 
1592       // No match found so we have to append this entry and any unique
1593       // referenced entries to *merge_cp_p.
1594       append_entry(scratch_cp, scratch_i, merge_cp_p, merge_cp_length_p,
1595         CHECK_false);
1596     }
1597   }
1598 
1599   log_debug(redefine, class, constantpool)
1600     ("after pass 1a: merge_cp_len=%d, scratch_i=%d, index_map_len=%d",
1601      *merge_cp_length_p, scratch_i, _index_map_count);
1602 
1603   if (scratch_i < scratch_cp->length()) {
1604     // Pass 1b:
1605     // old_cp is smaller than scratch_cp so there are entries in
1606     // scratch_cp that we have not yet processed. We take care of
1607     // those now.
1608     int increment = 1;
1609     for (; scratch_i < scratch_cp->length(); scratch_i += increment) {
1610       switch (scratch_cp->tag_at(scratch_i).value()) {
1611       case JVM_CONSTANT_Double:
1612       case JVM_CONSTANT_Long:
1613         // double and long take two constant pool entries
1614         increment = 2;
1615         break;
1616 
1617       default:
1618         increment = 1;
1619         break;
1620       }
1621 
1622       int found_i =
1623         scratch_cp->find_matching_entry(scratch_i, *merge_cp_p, CHECK_false);
1624       if (found_i != 0) {
1625         // Found a matching entry somewhere else in *merge_cp_p so
1626         // just need a mapping entry.
1627         map_index(scratch_cp, scratch_i, found_i);
1628         continue;
1629       }
1630 
1631       // No match found so we have to append this entry and any unique
1632       // referenced entries to *merge_cp_p.
1633       append_entry(scratch_cp, scratch_i, merge_cp_p, merge_cp_length_p,
1634         CHECK_false);
1635     }
1636 
1637     log_debug(redefine, class, constantpool)
1638       ("after pass 1b: merge_cp_len=%d, scratch_i=%d, index_map_len=%d",
1639        *merge_cp_length_p, scratch_i, _index_map_count);
1640   }
1641   finalize_operands_merge(*merge_cp_p, THREAD);
1642 
1643   return true;
1644 } // end merge_constant_pools()

1664 // Merge constant pools between the_class and scratch_class and
1665 // potentially rewrite bytecodes in scratch_class to use the merged
1666 // constant pool.
1667 jvmtiError VM_RedefineClasses::merge_cp_and_rewrite(
1668              InstanceKlass* the_class, InstanceKlass* scratch_class,
1669              TRAPS) {
1670   // worst case merged constant pool length is old and new combined
1671   int merge_cp_length = the_class->constants()->length()
1672         + scratch_class->constants()->length();
1673 
1674   // Constant pools are not easily reused so we allocate a new one
1675   // each time.
1676   // merge_cp is created unsafe for concurrent GC processing.  It
1677   // should be marked safe before discarding it. Even though
1678   // garbage,  if it crosses a card boundary, it may be scanned
1679   // in order to find the start of the first complete object on the card.
1680   ClassLoaderData* loader_data = the_class->class_loader_data();
1681   ConstantPool* merge_cp_oop =
1682     ConstantPool::allocate(loader_data,
1683                            merge_cp_length,
1684                            CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1685   MergeCPCleaner cp_cleaner(loader_data, merge_cp_oop);
1686 
1687   HandleMark hm(THREAD);  // make sure handles are cleared before
1688                           // MergeCPCleaner clears out merge_cp_oop
1689   constantPoolHandle merge_cp(THREAD, merge_cp_oop);
1690 
1691   // Get constants() from the old class because it could have been rewritten
1692   // while we were at a safepoint allocating a new constant pool.
1693   constantPoolHandle old_cp(THREAD, the_class->constants());
1694   constantPoolHandle scratch_cp(THREAD, scratch_class->constants());
1695 
1696   // If the length changed, the class was redefined out from under us. Return
1697   // an error.
1698   if (merge_cp_length != the_class->constants()->length()
1699          + scratch_class->constants()->length()) {
1700     return JVMTI_ERROR_INTERNAL;
1701   }
1702 
1703   // Update the version number of the constant pools (may keep scratch_cp)
1704   merge_cp->increment_and_save_version(old_cp->version());
1705   scratch_cp->increment_and_save_version(old_cp->version());
1706 
1707   ResourceMark rm(THREAD);
1708   _index_map_count = 0;
1709   _index_map_p = new intArray(scratch_cp->length(), scratch_cp->length(), -1);
1710 
1711   _operands_cur_length = ConstantPool::operand_array_length(old_cp->operands());
1712   _operands_index_map_count = 0;
1713   int operands_index_map_len = ConstantPool::operand_array_length(scratch_cp->operands());
1714   _operands_index_map_p = new intArray(operands_index_map_len, operands_index_map_len, -1);
1715 
1716   // reference to the cp holder is needed for copy_operands()
1717   merge_cp->set_pool_holder(scratch_class);
1718   bool result = merge_constant_pools(old_cp, scratch_cp, &merge_cp,
1719                   &merge_cp_length, THREAD);
1720   merge_cp->set_pool_holder(NULL);
1721 
1722   if (!result) {
1723     // The merge can fail due to memory allocation failure or due
1724     // to robustness checks.
1725     return JVMTI_ERROR_INTERNAL;
1726   }
1727 
1728   // Save fields from the old_cp.
1729   merge_cp->copy_fields(old_cp());
1730   scratch_cp->copy_fields(old_cp());
1731 
1732   log_info(redefine, class, constantpool)("merge_cp_len=%d, index_map_len=%d", merge_cp_length, _index_map_count);
1733 
1734   if (_index_map_count == 0) {
1735     // there is nothing to map between the new and merged constant pools
1736 
1737     if (old_cp->length() == scratch_cp->length()) {
1738       // The old and new constant pools are the same length and the
1739       // index map is empty. This means that the three constant pools
1740       // are equivalent (but not the same). Unfortunately, the new
1741       // constant pool has not gone through link resolution nor have
1742       // the new class bytecodes gone through constant pool cache
1743       // rewriting so we can't use the old constant pool with the new
1744       // class.
1745 
1746       // toss the merged constant pool at return
1747     } else if (old_cp->length() < scratch_cp->length()) {
1748       // The old constant pool has fewer entries than the new constant
1749       // pool and the index map is empty. This means the new constant
1750       // pool is a superset of the old constant pool. However, the old
1751       // class bytecodes have already gone through constant pool cache
1752       // rewriting so we can't use the new constant pool with the old
1753       // class.
1754 
1755       // toss the merged constant pool at return
1756     } else {
1757       // The old constant pool has more entries than the new constant
1758       // pool and the index map is empty. This means that both the old
1759       // and merged constant pools are supersets of the new constant
1760       // pool.
1761 
1762       // Replace the new constant pool with a shrunken copy of the
1763       // merged constant pool
1764       set_new_constant_pool(loader_data, scratch_class, merge_cp, merge_cp_length,
1765                             CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1766       // The new constant pool replaces scratch_cp so have cleaner clean it up.
1767       // It can't be cleaned up while there are handles to it.
1768       cp_cleaner.add_scratch_cp(scratch_cp());
1769     }
1770   } else {
1771     if (log_is_enabled(Trace, redefine, class, constantpool)) {
1772       // don't want to loop unless we are tracing
1773       int count = 0;
1774       for (int i = 1; i < _index_map_p->length(); i++) {
1775         int value = _index_map_p->at(i);
1776 
1777         if (value != -1) {
1778           log_trace(redefine, class, constantpool)("index_map[%d]: old=%d new=%d", count, i, value);
1779           count++;
1780         }
1781       }
1782     }
1783 
1784     // We have entries mapped between the new and merged constant pools
1785     // so we have to rewrite some constant pool references.
1786     if (!rewrite_cp_refs(scratch_class, THREAD)) {
1787       return JVMTI_ERROR_INTERNAL;
1788     }
1789 
1790     // Replace the new constant pool with a shrunken copy of the
1791     // merged constant pool so now the rewritten bytecodes have
1792     // valid references; the previous new constant pool will get
1793     // GCed.
1794     set_new_constant_pool(loader_data, scratch_class, merge_cp, merge_cp_length,
1795                           CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1796     // The new constant pool replaces scratch_cp so have cleaner clean it up.
1797     // It can't be cleaned up while there are handles to it.
1798     cp_cleaner.add_scratch_cp(scratch_cp());
1799   }
1800 
1801   return JVMTI_ERROR_NONE;
1802 } // end merge_cp_and_rewrite()

1978 // Rewrite constant pool references in the methods.
1979 bool VM_RedefineClasses::rewrite_cp_refs_in_methods(
1980        InstanceKlass* scratch_class, TRAPS) {
1981 
1982   Array<Method*>* methods = scratch_class->methods();
1983 
1984   if (methods == NULL || methods->length() == 0) {
1985     // no methods so nothing to do
1986     return true;
1987   }
1988 
1989   // rewrite constant pool references in the methods:
1990   for (int i = methods->length() - 1; i >= 0; i--) {
1991     methodHandle method(THREAD, methods->at(i));
1992     methodHandle new_method;
1993     rewrite_cp_refs_in_method(method, &new_method, THREAD);
1994     if (!new_method.is_null()) {
1995       // the method has been replaced so save the new method version
1996       // even in the case of an exception.  original method is on the
1997       // deallocation list.
1998       methods->at_put(i, new_method());
1999     }
2000     if (HAS_PENDING_EXCEPTION) {
2001       Symbol* ex_name = PENDING_EXCEPTION->klass()->name();
2002       log_info(redefine, class, load, exceptions)("rewrite_cp_refs_in_method exception: '%s'", ex_name->as_C_string());
2003       // Need to clear pending exception here as the super caller sets
2004       // the JVMTI_ERROR_INTERNAL if the returned value is false.
2005       CLEAR_PENDING_EXCEPTION;
2006       return false;
2007     }
2008   }
2009 
2010   return true;
2011 }

2014 // Rewrite constant pool references in the specific method. This code
2015 // was adapted from Rewriter::rewrite_method().
2016 void VM_RedefineClasses::rewrite_cp_refs_in_method(methodHandle method,
2017        methodHandle *new_method_p, TRAPS) {
2018 
2019   *new_method_p = methodHandle();  // default is no new method
2020 
2021   // We cache a pointer to the bytecodes here in code_base. If GC
2022   // moves the Method*, then the bytecodes will also move which
2023   // will likely cause a crash. We create a NoSafepointVerifier
2024   // object to detect whether we pass a possible safepoint in this
2025   // code block.
2026   NoSafepointVerifier nsv;
2027 
2028   // Bytecodes and their length
2029   address code_base = method->code_base();
2030   int code_length = method->code_size();
2031 
2032   int bc_length;
2033   for (int bci = 0; bci < code_length; bci += bc_length) {
2034     address bcp = code_base + bci;
2035     Bytecodes::Code c = (Bytecodes::Code)(*bcp);
2036 
2037     bc_length = Bytecodes::length_for(c);
2038     if (bc_length == 0) {
2039       // More complicated bytecodes report a length of zero so
2040       // we have to try again a slightly different way.
2041       bc_length = Bytecodes::length_at(method(), bcp);
2042     }
2043 
2044     assert(bc_length != 0, "impossible bytecode length");
2045 
2046     switch (c) {
2047       case Bytecodes::_ldc:
2048       {
2049         int cp_index = *(bcp + 1);
2050         int new_index = find_new_index(cp_index);
2051 
2052         if (StressLdcRewrite && new_index == 0) {
2053           // If we are stressing ldc -> ldc_w rewriting, then we
2054           // always need a new_index value.
2055           new_index = cp_index;
2056         }
2057         if (new_index != 0) {
2058           // the original index is mapped so we have more work to do
2059           if (!StressLdcRewrite && new_index <= max_jubyte) {
2060             // The new value can still use ldc instead of ldc_w
2061             // unless we are trying to stress ldc -> ldc_w rewriting
2062             log_trace(redefine, class, constantpool)
2063               ("%s@" INTPTR_FORMAT " old=%d, new=%d", Bytecodes::name(c), p2i(bcp), cp_index, new_index);
2064             *(bcp + 1) = new_index;
2065           } else {
2066             log_trace(redefine, class, constantpool)
2067               ("%s->ldc_w@" INTPTR_FORMAT " old=%d, new=%d", Bytecodes::name(c), p2i(bcp), cp_index, new_index);
2068             // the new value needs ldc_w instead of ldc
2069             u_char inst_buffer[4]; // max instruction size is 4 bytes
2070             bcp = (address)inst_buffer;
2071             // construct new instruction sequence
2072             *bcp = Bytecodes::_ldc_w;
2073             bcp++;
2074             // Rewriter::rewrite_method() does not rewrite ldc -> ldc_w.
2075             // See comment below for difference between put_Java_u2()
2076             // and put_native_u2().
2077             Bytes::put_Java_u2(bcp, new_index);
2078 
2079             Relocator rc(method, NULL /* no RelocatorListener needed */);
2080             methodHandle m;
2081             {
2082               PauseNoSafepointVerifier pnsv(&nsv);
2083 
2084               // ldc is 2 bytes and ldc_w is 3 bytes
2085               m = rc.insert_space_at(bci, 3, inst_buffer, CHECK);
2086             }
2087 
2088             // return the new method so that the caller can update
2089             // the containing class
2090             *new_method_p = method = m;
2091             // switch our bytecode processing loop from the old method
2092             // to the new method
2093             code_base = method->code_base();

2105       case Bytecodes::_checkcast      : // fall through
2106       case Bytecodes::_getfield       : // fall through
2107       case Bytecodes::_getstatic      : // fall through
2108       case Bytecodes::_instanceof     : // fall through
2109       case Bytecodes::_invokedynamic  : // fall through

2112       case Bytecodes::_invokestatic   : // fall through

3142 // Rewrite constant pool references in the method's stackmap table.
3143 // These "structures" are adapted from the StackMapTable_attribute that
3144 // is described in section 4.8.4 of the 6.0 version of the VM spec
3145 // (dated 2005.10.26):
3146 // file:///net/quincunx.sfbay/export/gbracha/ClassFile-Java6.pdf
3147 //
3148 // stack_map {
3149 //   u2 number_of_entries;
3150 //   stack_map_frame entries[number_of_entries];
3151 // }
3152 //
3153 void VM_RedefineClasses::rewrite_cp_refs_in_stack_map_table(
3154        const methodHandle& method, TRAPS) {
3155 
3156   if (!method->has_stackmap_table()) {
3157     return;
3158   }
3159 
3160   AnnotationArray* stackmap_data = method->stackmap_data();
3161   address stackmap_p = (address)stackmap_data->adr_at(0);
3162   address stackmap_end = stackmap_p + stackmap_data->length();
3163 
3164   assert(stackmap_p + 2 <= stackmap_end, "no room for number_of_entries");
3165   u2 number_of_entries = Bytes::get_Java_u2(stackmap_p);
3166   stackmap_p += 2;
3167 
3168   log_debug(redefine, class, stackmap)("number_of_entries=%u", number_of_entries);
3169 
3170   // walk through each stack_map_frame
3171   u2 calc_number_of_entries = 0;
3172   for (; calc_number_of_entries < number_of_entries; calc_number_of_entries++) {
3173     // The stack_map_frame structure is a u1 frame_type followed by
3174     // 0 or more bytes of data:
3175     //
3176     // union stack_map_frame {
3177     //   same_frame;
3178     //   same_locals_1_stack_item_frame;
3179     //   same_locals_1_stack_item_frame_extended;
3180     //   chop_frame;
3181     //   same_frame_extended;
3182     //   append_frame;
3183     //   full_frame;
3184     // }
3185 
3186     assert(stackmap_p + 1 <= stackmap_end, "no room for frame_type");
3187     u1 frame_type = *stackmap_p;
3188     stackmap_p++;
3189 
3190     // same_frame {
3191     //   u1 frame_type = SAME; /* 0-63 */
3192     // }
3193     if (frame_type <= 63) {
3194       // nothing more to do for same_frame
3195     }
3196 
3197     // same_locals_1_stack_item_frame {
3198     //   u1 frame_type = SAME_LOCALS_1_STACK_ITEM; /* 64-127 */
3199     //   verification_type_info stack[1];
3200     // }
3201     else if (frame_type >= 64 && frame_type <= 127) {
3202       rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3203         calc_number_of_entries, frame_type, THREAD);
3204     }
3205 
3206     // reserved for future use
3207     else if (frame_type >= 128 && frame_type <= 246) {
3208       // nothing more to do for reserved frame_types
3209     }
3210 
3211     // same_locals_1_stack_item_frame_extended {
3212     //   u1 frame_type = SAME_LOCALS_1_STACK_ITEM_EXTENDED; /* 247 */
3213     //   u2 offset_delta;
3214     //   verification_type_info stack[1];
3215     // }
3216     else if (frame_type == 247) {
3217       stackmap_p += 2;
3218       rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3219         calc_number_of_entries, frame_type, THREAD);
3220     }
3221 
3222     // chop_frame {
3223     //   u1 frame_type = CHOP; /* 248-250 */
3224     //   u2 offset_delta;
3225     // }
3226     else if (frame_type >= 248 && frame_type <= 250) {
3227       stackmap_p += 2;
3228     }
3229 
3230     // same_frame_extended {
3231     //   u1 frame_type = SAME_FRAME_EXTENDED; /* 251*/
3232     //   u2 offset_delta;
3233     // }
3234     else if (frame_type == 251) {
3235       stackmap_p += 2;
3236     }
3237 
3238     // append_frame {
3239     //   u1 frame_type = APPEND; /* 252-254 */
3240     //   u2 offset_delta;
3241     //   verification_type_info locals[frame_type - 251];
3242     // }
3243     else if (frame_type >= 252 && frame_type <= 254) {
3244       assert(stackmap_p + 2 <= stackmap_end,
3245         "no room for offset_delta");
3246       stackmap_p += 2;
3247       u1 len = frame_type - 251;
3248       for (u1 i = 0; i < len; i++) {
3249         rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3250           calc_number_of_entries, frame_type, THREAD);
3251       }
3252     }
3253 
3254     // full_frame {
3255     //   u1 frame_type = FULL_FRAME; /* 255 */
3256     //   u2 offset_delta;
3257     //   u2 number_of_locals;
3258     //   verification_type_info locals[number_of_locals];
3259     //   u2 number_of_stack_items;
3260     //   verification_type_info stack[number_of_stack_items];
3261     // }
3262     else if (frame_type == 255) {
3263       assert(stackmap_p + 2 + 2 <= stackmap_end,
3264         "no room for smallest full_frame");
3265       stackmap_p += 2;
3266 
3267       u2 number_of_locals = Bytes::get_Java_u2(stackmap_p);
3268       stackmap_p += 2;
3269 
3270       for (u2 locals_i = 0; locals_i < number_of_locals; locals_i++) {
3271         rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3272           calc_number_of_entries, frame_type, THREAD);
3273       }
3274 
3275       // Use the largest size for the number_of_stack_items, but only get
3276       // the right number of bytes.

3290 // Rewrite constant pool references in the verification type info
3291 // portion of the method's stackmap table. These "structures" are
3292 // adapted from the StackMapTable_attribute that is described in
3293 // section 4.8.4 of the 6.0 version of the VM spec (dated 2005.10.26):
3294 // file:///net/quincunx.sfbay/export/gbracha/ClassFile-Java6.pdf
3295 //
3296 // The verification_type_info structure is a u1 tag followed by 0 or
3297 // more bytes of data:
3298 //
3299 // union verification_type_info {
3300 //   Top_variable_info;
3301 //   Integer_variable_info;
3302 //   Float_variable_info;
3303 //   Long_variable_info;
3304 //   Double_variable_info;
3305 //   Null_variable_info;
3306 //   UninitializedThis_variable_info;
3307 //   Object_variable_info;
3308 //   Uninitialized_variable_info;
3309 // }
3310 //
3311 void VM_RedefineClasses::rewrite_cp_refs_in_verification_type_info(
3312        address& stackmap_p_ref, address stackmap_end, u2 frame_i,
3313        u1 frame_type, TRAPS) {
3314 
3315   assert(stackmap_p_ref + 1 <= stackmap_end, "no room for tag");
3316   u1 tag = *stackmap_p_ref;
3317   stackmap_p_ref++;
3318 
3319   switch (tag) {
3320   // Top_variable_info {
3321   //   u1 tag = ITEM_Top; /* 0 */
3322   // }
3323   // verificationType.hpp has zero as ITEM_Bogus instead of ITEM_Top
3324   case 0:  // fall through
3325 
3326   // Integer_variable_info {
3327   //   u1 tag = ITEM_Integer; /* 1 */
3328   // }
3329   case ITEM_Integer:  // fall through
3330 
3331   // Float_variable_info {
3332   //   u1 tag = ITEM_Float; /* 2 */
3333   // }
3334   case ITEM_Float:  // fall through
3335 
3336   // Double_variable_info {
3337   //   u1 tag = ITEM_Double; /* 3 */
3338   // }
3339   case ITEM_Double:  // fall through
3340 
3341   // Long_variable_info {
3342   //   u1 tag = ITEM_Long; /* 4 */
3343   // }
3344   case ITEM_Long:  // fall through
3345 
3346   // Null_variable_info {
3347   //   u1 tag = ITEM_Null; /* 5 */
3348   // }
3349   case ITEM_Null:  // fall through
3350 
3351   // UninitializedThis_variable_info {
3352   //   u1 tag = ITEM_UninitializedThis; /* 6 */
3353   // }
3354   case ITEM_UninitializedThis:
3355     // nothing more to do for the above tag types
3356     break;
3357 
3358   // Object_variable_info {
3359   //   u1 tag = ITEM_Object; /* 7 */
3360   //   u2 cpool_index;
3361   // }
3362   case ITEM_Object:
3363   {
3364     assert(stackmap_p_ref + 2 <= stackmap_end, "no room for cpool_index");
3365     u2 cpool_index = Bytes::get_Java_u2(stackmap_p_ref);
3366     u2 new_cp_index = find_new_index(cpool_index);
3367     if (new_cp_index != 0) {
3368       log_debug(redefine, class, stackmap)("mapped old cpool_index=%d", cpool_index);
3369       Bytes::put_Java_u2(stackmap_p_ref, new_cp_index);
3370       cpool_index = new_cp_index;
3371     }
3372     stackmap_p_ref += 2;
3373 
3374     log_debug(redefine, class, stackmap)
3375       ("frame_i=%u, frame_type=%u, cpool_index=%d", frame_i, frame_type, cpool_index);
3376   } break;
3377 
3378   // Uninitialized_variable_info {
3379   //   u1 tag = ITEM_Uninitialized; /* 8 */
3380   //   u2 offset;
3381   // }
3382   case ITEM_Uninitialized:
3383     assert(stackmap_p_ref + 2 <= stackmap_end, "no room for offset");
3384     stackmap_p_ref += 2;
3385     break;
3386 
3387   default:
3388     log_debug(redefine, class, stackmap)("frame_i=%u, frame_type=%u, bad tag=0x%x", frame_i, frame_type, tag);
3389     ShouldNotReachHere();
3390     break;
3391   } // end switch (tag)
3392 } // end rewrite_cp_refs_in_verification_type_info()

3597   // printed out.

3599

3601   // array class vtables also
3602   if (k->is_array_klass() && _has_redefined_Object) {
3603     k->vtable().adjust_method_entries(&trace_name_printed);
3604 
3605   } else if (k->is_instance_klass()) {
3606     HandleMark hm(_thread);

3608 
3609     // Clean MethodData of this class's methods so they don't refer to
3610     // old methods that are no longer running.

3613     for (int index = 0; index < num_methods; ++index) {

3621     // user-defined class loader. This means that if the bootstrap
3622     // class loader is the initiating class loader, then it will also

3641

3696 int VM_RedefineClasses::check_methods_and_mark_as_obsolete() {
3697   int emcp_method_count = 0;
3698   int obsolete_count = 0;
3699   int old_index = 0;
3700   for (int j = 0; j < _matching_methods_length; ++j, ++old_index) {
3701     Method* old_method = _matching_old_methods[j];
3702     Method* new_method = _matching_new_methods[j];
3703     Method* old_array_method;
3704 
3705     // Maintain an old_index into the _old_methods array by skipping
3706     // deleted methods
3707     while ((old_array_method = _old_methods->at(old_index)) != old_method) {
3708       ++old_index;
3709     }
3710 
3711     if (MethodComparator::methods_EMCP(old_method, new_method)) {
3712       // The EMCP definition from JSR-163 requires the bytecodes to be
3713       // the same with the exception of constant pool indices which may
3714       // differ. However, the constants referred to by those indices
3715       // must be the same.
3716       //
3717       // We use methods_EMCP() for comparison since constant pool
3718       // merging can remove duplicate constant pool entries that were
3719       // present in the old method and removed from the rewritten new
3720       // method. A faster binary comparison function would consider the
3721       // old and new methods to be different when they are actually
3722       // EMCP.
3723       //
3724       // The old and new methods are EMCP and you would think that we
3725       // could get rid of one of them here and now and save some space.
3726       // However, the concept of EMCP only considers the bytecodes and
3727       // the constant pool entries in the comparison. Other things,
3728       // e.g., the line number table (LNT) or the local variable table
3729       // (LVT) don't count in the comparison. So the new (and EMCP)
3730       // method can have a new LNT that we need so we can't just
3731       // overwrite the new method with the old method.
3732       //
3733       // When this routine is called, we have already attached the new
3734       // methods to the_class so the old methods are effectively
3735       // overwritten. However, if an old method is still executing,
3736       // then the old method cannot be collected until sometime after
3737       // the old method call has returned. So the overwriting of old
3738       // methods by new methods will save us space except for those
3739       // (hopefully few) old methods that are still executing.
3740       //
3741       // A method refers to a ConstMethod* and this presents another
3742       // possible avenue to space savings. The ConstMethod* in the
3743       // new method contains possibly new attributes (LNT, LVT, etc).
3744       // At first glance, it seems possible to save space by replacing
3745       // the ConstMethod* in the old method with the ConstMethod*
3746       // from the new method. The old and new methods would share the
3747       // same ConstMethod* and we would save the space occupied by
3748       // the old ConstMethod*. However, the ConstMethod* contains
3749       // a back reference to the containing method. Sharing the
3750       // ConstMethod* between two methods could lead to confusion in
3751       // the code that uses the back reference. This would lead to
3752       // brittle code that could be broken in non-obvious ways now or
3753       // in the future.
3754       //
3755       // Another possibility is to copy the ConstMethod* from the new
3756       // method to the old method and then overwrite the new method with
3757       // the old method. Since the ConstMethod* contains the bytecodes
3758       // for the method embedded in the oop, this option would change
3759       // the bytecodes out from under any threads executing the old
3760       // method and make the thread's bcp invalid. Since EMCP requires
3761       // that the bytecodes be the same modulo constant pool indices, it
3762       // is straight forward to compute the correct new bcp in the new
3763       // ConstMethod* from the old bcp in the old ConstMethod*. The
3764       // time consuming part would be searching all the frames in all
3765       // of the threads to find all of the calls to the old method.
3766       //
3767       // It looks like we will have to live with the limited savings
3768       // that we get from effectively overwriting the old methods
3769       // when the new methods are attached to the_class.
3770 
3771       // Count number of methods that are EMCP.  The method will be marked
3772       // old but not obsolete if it is EMCP.
3773       emcp_method_count++;
3774 
3775       // An EMCP method is _not_ obsolete. An obsolete method has a
3776       // different jmethodID than the current method. An EMCP method
3777       // has the same jmethodID as the current method. Having the
3778       // same jmethodID for all EMCP versions of a method allows for
3779       // a consistent view of the EMCP methods regardless of which
3780       // EMCP method you happen to have in hand. For example, a
3781       // breakpoint set in one EMCP method will work for all EMCP
3782       // versions of the method including the current one.
3783     } else {
3784       // mark obsolete methods as such
3785       old_method->set_is_obsolete();
3786       obsolete_count++;
3787 
3788       // obsolete methods need a unique idnum so they become new entries in
3789       // the jmethodID cache in InstanceKlass
3790       assert(old_method->method_idnum() == new_method->method_idnum(), "must match");
3791       u2 num = InstanceKlass::cast(_the_class)->next_method_idnum();
3792       if (num != ConstMethod::UNSET_IDNUM) {
3793         old_method->set_method_idnum(num);

3833 // This internal class transfers the native function registration from old methods
3834 // to new methods.  It is designed to handle both the simple case of unchanged
3835 // native methods and the complex cases of native method prefixes being added and/or
3836 // removed.
3837 // It expects only to be used during the VM_RedefineClasses op (a safepoint).
3838 //
3839 // This class is used after the new methods have been installed in "the_class".
3840 //
3841 // So, for example, the following must be handled.  Where 'm' is a method and
3842 // a number followed by an underscore is a prefix.
3843 //
3844 //                                      Old Name    New Name
3845 // Simple transfer to new method        m       ->  m
3846 // Add prefix                           m       ->  1_m
3847 // Remove prefix                        1_m     ->  m
3848 // Simultaneous add of prefixes         m       ->  3_2_1_m
3849 // Simultaneous removal of prefixes     3_2_1_m ->  m
3850 // Simultaneous add and remove          1_m     ->  2_m
3851 // Same, caused by prefix removal only  3_2_1_m ->  3_2_m
3852 //
3853 class TransferNativeFunctionRegistration {
3854  private:
3855   InstanceKlass* the_class;
3856   int prefix_count;
3857   char** prefixes;
3858 
3859   // Recursively search the binary tree of possibly prefixed method names.
3860   // Iteration could be used if all agents were well behaved. Full tree walk is
3861   // more resilent to agents not cleaning up intermediate methods.
3862   // Branch at each depth in the binary tree is:
3863   //    (1) without the prefix.
3864   //    (2) with the prefix.
3865   // where 'prefix' is the prefix at that 'depth' (first prefix, second prefix,...)
3866   Method* search_prefix_name_space(int depth, char* name_str, size_t name_len,
3867                                      Symbol* signature) {
3868     TempNewSymbol name_symbol = SymbolTable::probe(name_str, (int)name_len);
3869     if (name_symbol != NULL) {
3870       Method* method = the_class->lookup_method(name_symbol, signature);
3871       if (method != NULL) {
3872         // Even if prefixed, intermediate methods must exist.
3873         if (method->is_native()) {
3874           // Wahoo, we found a (possibly prefixed) version of the method, return it.
3875           return method;
3876         }
3877         if (depth < prefix_count) {
3878           // Try applying further prefixes (other than this one).
3879           method = search_prefix_name_space(depth+1, name_str, name_len, signature);
3880           if (method != NULL) {
3881             return method; // found
3882           }
3883 
3884           // Try adding this prefix to the method name and see if it matches
3885           // another method name.
3886           char* prefix = prefixes[depth];
3887           size_t prefix_len = strlen(prefix);
3888           size_t trial_len = name_len + prefix_len;
3889           char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
3890           strcpy(trial_name_str, prefix);
3891           strcat(trial_name_str, name_str);
3892           method = search_prefix_name_space(depth+1, trial_name_str, trial_len,
3893                                             signature);
3894           if (method != NULL) {
3895             // If found along this branch, it was prefixed, mark as such
3896             method->set_is_prefixed_native();
3897             return method; // found
3898           }
3899         }
3900       }
3901     }
3902     return NULL;  // This whole branch bore nothing
3903   }
3904 
3905   // Return the method name with old prefixes stripped away.
3906   char* method_name_without_prefixes(Method* method) {
3907     Symbol* name = method->name();
3908     char* name_str = name->as_utf8();
3909 
3910     // Old prefixing may be defunct, strip prefixes, if any.
3911     for (int i = prefix_count-1; i >= 0; i--) {
3912       char* prefix = prefixes[i];
3913       size_t prefix_len = strlen(prefix);
3914       if (strncmp(prefix, name_str, prefix_len) == 0) {
3915         name_str += prefix_len;
3916       }
3917     }
3918     return name_str;
3919   }
3920 
3921   // Strip any prefixes off the old native method, then try to find a
3922   // (possibly prefixed) new native that matches it.
3923   Method* strip_and_search_for_new_native(Method* method) {
3924     ResourceMark rm;
3925     char* name_str = method_name_without_prefixes(method);
3926     return search_prefix_name_space(0, name_str, strlen(name_str),
3927                                     method->signature());
3928   }

3930  public:
3931 
3932   // Construct a native method transfer processor for this class.
3933   TransferNativeFunctionRegistration(InstanceKlass* _the_class) {
3934     assert(SafepointSynchronize::is_at_safepoint(), "sanity check");
3935 
3936     the_class = _the_class;
3937     prefixes = JvmtiExport::get_all_native_method_prefixes(&prefix_count);
3938   }

3940   // Attempt to transfer any of the old or deleted methods that are native
3941   void transfer_registrations(Method** old_methods, int methods_length) {
3942     for (int j = 0; j < methods_length; j++) {
3943       Method* old_method = old_methods[j];
3944 
3945       if (old_method->is_native() && old_method->has_native_function()) {
3946         Method* new_method = strip_and_search_for_new_native(old_method);
3947         if (new_method != NULL) {
3948           // Actually set the native function in the new method.
3949           // Redefine does not send events (except CFLH), certainly not this

3951           new_method->set_native_function(old_method->native_function(),
3952                               !Method::native_bind_event_is_interesting);
3953         }
3954       }
3955     }
3956   }

4018 void VM_RedefineClasses::compute_added_deleted_matching_methods() {
4019   Method* old_method;
4020   Method* new_method;
4021 
4022   _matching_old_methods = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4023   _matching_new_methods = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4024   _added_methods        = NEW_RESOURCE_ARRAY(Method*, _new_methods->length());
4025   _deleted_methods      = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4026 
4027   _matching_methods_length = 0;
4028   _deleted_methods_length  = 0;
4029   _added_methods_length    = 0;
4030 
4031   int nj = 0;
4032   int oj = 0;
4033   while (true) {
4034     if (oj >= _old_methods->length()) {
4035       if (nj >= _new_methods->length()) {
4036         break; // we've looked at everything, done
4037       }
4038       // New method at the end
4039       new_method = _new_methods->at(nj);
4040       _added_methods[_added_methods_length++] = new_method;
4041       ++nj;
4042     } else if (nj >= _new_methods->length()) {
4043       // Old method, at the end, is deleted
4044       old_method = _old_methods->at(oj);
4045       _deleted_methods[_deleted_methods_length++] = old_method;
4046       ++oj;
4047     } else {
4048       old_method = _old_methods->at(oj);
4049       new_method = _new_methods->at(nj);
4050       if (old_method->name() == new_method->name()) {
4051         if (old_method->signature() == new_method->signature()) {
4052           _matching_old_methods[_matching_methods_length  ] = old_method;
4053           _matching_new_methods[_matching_methods_length++] = new_method;
4054           ++nj;
4055           ++oj;
4056         } else {
4057           // added overloaded have already been moved to the end,
4058           // so this is a deleted overloaded method
4059           _deleted_methods[_deleted_methods_length++] = old_method;
4060           ++oj;
4061         }
4062       } else { // names don't match
4063         if (old_method->name()->fast_compare(new_method->name()) > 0) {
4064           // new method
4065           _added_methods[_added_methods_length++] = new_method;
4066           ++nj;
4067         } else {
4068           // deleted method
4069           _deleted_methods[_deleted_methods_length++] = old_method;
4070           ++oj;
4071         }
4072       }
4073     }
4074   }
4075   assert(_matching_methods_length + _deleted_methods_length == _old_methods->length(), "sanity");
4076   assert(_matching_methods_length + _added_methods_length == _new_methods->length(), "sanity");
4077 }

4329     // so that obsolete methods are not pinned.
4330     the_class->oop_map_cache()->flush_obsolete_entries();
4331   }
4332 
4333   increment_class_counter((InstanceKlass *)the_class, THREAD);
4334 
4335   if (EventClassRedefinition::is_enabled()) {
4336     EventClassRedefinition event;
4337     event.set_classModificationCount(java_lang_Class::classRedefinedCount(the_class->java_mirror()));
4338     event.set_redefinedClass(the_class);
4339     event.set_redefinitionId(_id);
4340     event.commit();
4341   }
4342 
4343   {
4344     ResourceMark rm(THREAD);
4345     // increment the classRedefinedCount field in the_class and in any
4346     // direct and indirect subclasses of the_class
4347     log_info(redefine, class, load)
4348       ("redefined name=%s, count=%d (avail_mem=" UINT64_FORMAT "K)",
4349        the_class->external_name(), java_lang_Class::classRedefinedCount(the_class->java_mirror()), os::available_memory() >> 10);
4350     Events::log_redefinition(THREAD, "redefined class name=%s, count=%d",
4351                              the_class->external_name(),
4352                              java_lang_Class::classRedefinedCount(the_class->java_mirror()));
4353 
4354   }
4355   _timer_rsc_phase2.stop();
4356 
4357 } // end redefine_single_class()

4362 void VM_RedefineClasses::increment_class_counter(InstanceKlass *ik, TRAPS) {

4365   int new_count = java_lang_Class::classRedefinedCount(class_mirror) + 1;

4367 
4368   if (class_oop != _the_class) {

4370     log_debug(redefine, class, subclass)("updated count in subclass=%s to %d", ik->external_name(), new_count);
4371   }
4372 
4373   for (Klass *subk = ik->subklass(); subk != NULL;
4374        subk = subk->next_sibling()) {

4376       // Only update instanceKlasses
4377       InstanceKlass *subik = InstanceKlass::cast(subk);
4378       // recursively do subclasses of the current subclass
4379       increment_class_counter(subik, THREAD);
4380     }
4381   }

4384 void VM_RedefineClasses::CheckClass::do_klass(Klass* k) {
4385   bool no_old_methods = true;  // be optimistic
4386 
4387   // Both array and instance classes have vtables.
4388   // a vtable should never contain old or obsolete methods
4389   ResourceMark rm(_thread);
4390   if (k->vtable_length() > 0 &&
4391       !k->vtable().check_no_old_or_obsolete_entries()) {
4392     if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4393       log_trace(redefine, class, obsolete, metadata)
4394         ("klassVtable::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4395          k->signature_name());
4396       k->vtable().dump_vtable();
4397     }
4398     no_old_methods = false;
4399   }
4400 
4401   if (k->is_instance_klass()) {
4402     HandleMark hm(_thread);
4403     InstanceKlass *ik = InstanceKlass::cast(k);
4404 
4405     // an itable should never contain old or obsolete methods
4406     if (ik->itable_length() > 0 &&
4407         !ik->itable().check_no_old_or_obsolete_entries()) {
4408       if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4409         log_trace(redefine, class, obsolete, metadata)
4410           ("klassItable::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4411            ik->signature_name());
4412         ik->itable().dump_itable();
4413       }
4414       no_old_methods = false;
4415     }
4416 
4417     // the constant pool cache should never contain non-deleted old or obsolete methods
4418     if (ik->constants() != NULL &&
4419         ik->constants()->cache() != NULL &&
4420         !ik->constants()->cache()->check_no_old_or_obsolete_entries()) {
4421       if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4422         log_trace(redefine, class, obsolete, metadata)
4423           ("cp-cache::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4424            ik->signature_name());
4425         ik->constants()->cache()->dump_cache();
4426       }
4427       no_old_methods = false;
4428     }
4429   }
4430 
4431   // print and fail guarantee if old methods are found.
4432   if (!no_old_methods) {
4433     if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4434       dump_methods();
4435     } else {
4436       log_trace(redefine, class)("Use the '-Xlog:redefine+class*:' option "
4437         "to see more info about the following guarantee() failure.");
4438     }
4439     guarantee(false, "OLD and/or OBSOLETE method(s) found");
4440   }
