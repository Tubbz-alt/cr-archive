  95 // The CDS archive is divided into the following regions:
  96 //     mc  - misc code (the method entry trampolines, c++ vtables)
  97 //     rw  - read-write metadata
  98 //     ro  - read-only metadata and read-only tables
  99 //
 100 //     ca0 - closed archive heap space #0
 101 //     ca1 - closed archive heap space #1 (may be empty)
 102 //     oa0 - open archive heap space #0
 103 //     oa1 - open archive heap space #1 (may be empty)
 104 //
 105 // The mc, rw, and ro regions are linearly allocated, starting from
 106 // SharedBaseAddress, in the order of mc->rw->ro. The size of these 3 regions
 107 // are page-aligned, and there's no gap between any consecutive regions.
 108 //
 109 // These 3 regions are populated in the following steps:
 110 // [1] All classes are loaded in MetaspaceShared::preload_classes(). All metadata are
 111 //     temporarily allocated outside of the shared regions. Only the method entry
 112 //     trampolines are written into the mc region.
 113 // [2] C++ vtables are copied into the mc region.
 114 // [3] ArchiveCompactor copies RW metadata into the rw region.
 115 // [4] ArchiveCompactor copies RO metadata into the ro region.
 116 // [5] SymbolTable, StringTable, SystemDictionary, and a few other read-only data
 117 //     are copied into the ro region as read-only tables.
 118 //
 119 // The s0/s1 and oa0/oa1 regions are populated inside HeapShared::archive_java_heap_objects.
 120 // Their layout is independent of the other 4 regions.
 121 
 122 char* DumpRegion::expand_top_to(char* newtop) {
 123   assert(is_allocatable(), "must be initialized and not packed");
 124   assert(newtop >= _top, "must not grow backwards");
 125   if (newtop > _end) {
 126     MetaspaceShared::report_out_of_space(_name, newtop - _top);
 127     ShouldNotReachHere();
 128   }
 129 
 130   if (_rs == MetaspaceShared::shared_rs()) {
 131     uintx delta;
 132     if (DynamicDumpSharedSpaces) {
 133       delta = DynamicArchive::object_delta_uintx(newtop);
 134     } else {
 135       delta = MetaspaceShared::object_delta_uintx(newtop);
 136     }
 137     if (delta > MAX_SHARED_DELTA) {
 138       // This is just a sanity check and should not appear in any real world usage. This
 139       // happens only if you allocate more than 2GB of shared objects and would require
 140       // millions of shared classes.
 141       vm_exit_during_initialization("Out of memory in the CDS archive",
 142                                     "Please reduce the number of shared classes.");
 143     }
 144   }
 145 
 146   MetaspaceShared::commit_to(_rs, _vs, newtop);
 147   _top = newtop;
 148   return _top;
 149 }

 151 char* DumpRegion::allocate(size_t num_bytes, size_t alignment) {
 152   char* p = (char*)align_up(_top, alignment);
 153   char* newtop = p + align_up(num_bytes, alignment);
 154   expand_top_to(newtop);
 155   memset(p, 0, newtop - p);
 156   return p;
 157 }

 159 void DumpRegion::append_intptr_t(intptr_t n, bool need_to_mark) {
 160   assert(is_aligned(_top, sizeof(intptr_t)), "bad alignment");
 161   intptr_t *p = (intptr_t*)_top;
 162   char* newtop = _top + sizeof(intptr_t);
 163   expand_top_to(newtop);
 164   *p = n;
 165   if (need_to_mark) {
 166     ArchivePtrMarker::mark_pointer(p);
 167   }
 168 }

 184 void DumpRegion::init(ReservedSpace* rs, VirtualSpace* vs) {
 185   _rs = rs;
 186   _vs = vs;
 187   // Start with 0 committed bytes. The memory will be committed as needed by
 188   // MetaspaceShared::commit_to().
 189   if (!_vs->initialize(*_rs, 0)) {
 190     fatal("Unable to allocate memory for shared space");
 191   }
 192   _base = _top = _rs->base();
 193   _end = _rs->end();
 194 }

 196 void DumpRegion::pack(DumpRegion* next) {
 197   assert(!is_packed(), "sanity");
 198   _end = (char*)align_up(_top, MetaspaceShared::reserved_space_alignment());
 199   _is_packed = true;
 200   if (next != NULL) {
 201     next->_rs = _rs;
 202     next->_vs = _vs;
 203     next->_base = next->_top = this->_end;
 204     next->_end = _rs->end();
 205   }
 206 }

 254 static bool shared_base_too_high(char* shared_base, size_t cds_total) {
 255   if (SharedBaseAddress != 0 && shared_base < (char*)SharedBaseAddress) {
 256     // SharedBaseAddress is very high (e.g., 0xffffffffffffff00) so
 257     // align_up(SharedBaseAddress, MetaspaceShared::reserved_space_alignment()) has wrapped around.
 258     return true;
 259   }
 260   if (max_uintx - uintx(shared_base) < uintx(cds_total)) {
 261     // The end of the archive will wrap around
 262     return true;
 263   }
 264 
 265   return false;
 266 }

 268 static char* compute_shared_base(size_t cds_total) {
 269   char* shared_base = (char*)align_up((char*)SharedBaseAddress, MetaspaceShared::reserved_space_alignment());
 270   const char* err = NULL;
 271   if (shared_base_too_high(shared_base, cds_total)) {
 272     err = "too high";
 273   } else if (!shared_base_valid(shared_base)) {
 274     err = "invalid for this platform";
 275   }
 276   if (err) {
 277     log_warning(cds)("SharedBaseAddress (" INTPTR_FORMAT ") is %s. Reverted to " INTPTR_FORMAT,
 278                      p2i((void*)SharedBaseAddress), err,
 279                      p2i((void*)Arguments::default_SharedBaseAddress()));
 280     SharedBaseAddress = Arguments::default_SharedBaseAddress();
 281     shared_base = (char*)align_up((char*)SharedBaseAddress, MetaspaceShared::reserved_space_alignment());
 282   }
 283   assert(!shared_base_too_high(shared_base, cds_total) && shared_base_valid(shared_base), "Sanity");
 284   return shared_base;
 285 }

 287 void MetaspaceShared::initialize_dumptime_shared_and_meta_spaces() {
 288   assert(DumpSharedSpaces, "should be called for dump time only");
 289 
 290   const size_t reserve_alignment = MetaspaceShared::reserved_space_alignment();
 291 
 292 #ifdef _LP64
 293   // On 64-bit VM we reserve a 4G range and, if UseCompressedClassPointers=1,
 294   //  will use that to house both the archives and the ccs. See below for
 295   //  details.
 296   const uint64_t UnscaledClassSpaceMax = (uint64_t(max_juint) + 1);
 297   const size_t cds_total = align_down(UnscaledClassSpaceMax, reserve_alignment);
 298 #else
 299   // We don't support archives larger than 256MB on 32-bit due to limited
 300   //  virtual address space.
 301   size_t cds_total = align_down(256*M, reserve_alignment);
 302 #endif
 303 
 304   char* shared_base = compute_shared_base(cds_total);
 305   _requested_base_address = shared_base;
 306 
 307   // Whether to use SharedBaseAddress as attach address.
 308   bool use_requested_base = true;
 309 
 310   if (shared_base == NULL) {
 311     use_requested_base = false;
 312   }
 313 
 314   if (ArchiveRelocationMode == 1) {
 315     log_info(cds)("ArchiveRelocationMode == 1: always allocate class space at an alternative address");
 316     use_requested_base = false;
 317   }
 318 
 319   // First try to reserve the space at the specified SharedBaseAddress.
 320   assert(!_shared_rs.is_reserved(), "must be");
 321   if (use_requested_base) {
 322     _shared_rs = ReservedSpace(cds_total, reserve_alignment,
 323                                false /* large */, (char*)shared_base);
 324     if (_shared_rs.is_reserved()) {
 325       assert(_shared_rs.base() == shared_base, "should match");
 326     } else {
 327       log_info(cds)("dumptime space reservation: failed to map at "
 328                     "SharedBaseAddress " PTR_FORMAT, p2i(shared_base));
 329     }
 330   }
 331   if (!_shared_rs.is_reserved()) {
 332     // Get a reserved space anywhere if attaching at the SharedBaseAddress
 333     //  fails:
 334     if (UseCompressedClassPointers) {
 335       // If we need to reserve class space as well, let the platform handle
 336       //  the reservation.
 337       LP64_ONLY(_shared_rs =
 338                 Metaspace::reserve_address_space_for_compressed_classes(cds_total);)
 339       NOT_LP64(ShouldNotReachHere();)
 340     } else {
 341       // anywhere is fine.
 342       _shared_rs = ReservedSpace(cds_total, reserve_alignment,
 343                                  false /* large */, (char*)NULL);
 344     }
 345   }
 346 
 347   if (!_shared_rs.is_reserved()) {
 348     vm_exit_during_initialization("Unable to reserve memory for shared space",
 349                                   err_msg(SIZE_FORMAT " bytes.", cds_total));
 350   }
 351 
 352 #ifdef _LP64
 353 
 354   if (UseCompressedClassPointers) {
 355 
 356     assert(CompressedKlassPointers::is_valid_base((address)_shared_rs.base()), "Sanity");
 357 
 358     // On 64-bit VM, if UseCompressedClassPointers=1, the compressed class space
 359     //  must be allocated near the cds such as that the compressed Klass pointer
 360     //  encoding can be used to en/decode pointers from both cds and ccs. Since
 361     //  Metaspace cannot do this (it knows nothing about cds), we do it for
 362     //  Metaspace here and pass it the space to use for ccs.
 363     //
 364     // We do this by reserving space for the ccs behind the archives. Note
 365     //  however that ccs follows a different alignment
 366     //  (Metaspace::reserve_alignment), so there may be a gap between ccs and
 367     //  cds.
 368     // We use a similar layout at runtime, see reserve_address_space_for_archives().
 369     //
 370     //                              +-- SharedBaseAddress (default = 0x800000000)
 371     //                              v
 372     // +-..---------+---------+ ... +----+----+----+--------+-----------------+
 373     // |    Heap    | Archive |     | MC | RW | RO | [gap]  |    class space  |
 374     // +-..---------+---------+ ... +----+----+----+--------+-----------------+
 375     // |<--   MaxHeapSize  -->|     |<-- UnscaledClassSpaceMax = 4GB -->|
 376     //
 377     // Note: ccs must follow the archives, and the archives must start at the
 378     //  encoding base. However, the exact placement of ccs does not matter as
 379     //  long as it it resides in the encoding range of CompressedKlassPointers
 380     //  and comes after the archive.
 381     //
 382     // We do this by splitting up the allocated 4G into 3G of archive space,
 383     //  followed by 1G for the ccs:
 384     // + The upper 1 GB is used as the "temporary compressed class space"
 385     //   -- preload_classes() will store Klasses into this space.
 386     // + The lower 3 GB is used for the archive -- when preload_classes()
 387     //   is done, ArchiveCompactor will copy the class metadata into this
 388     //   space, first the RW parts, then the RO parts.
 389 
 390     // Starting address of ccs must be aligned to Metaspace::reserve_alignment()...
 391     size_t class_space_size = align_down(_shared_rs.size() / 4, Metaspace::reserve_alignment());
 392     address class_space_start = (address)align_down(_shared_rs.end() - class_space_size, Metaspace::reserve_alignment());
 393     size_t archive_size = class_space_start - (address)_shared_rs.base();
 394 
 395     ReservedSpace tmp_class_space = _shared_rs.last_part(archive_size);
 396     _shared_rs = _shared_rs.first_part(archive_size);
 397 
 398     // ... as does the size of ccs.
 399     tmp_class_space = tmp_class_space.first_part(class_space_size);
 400     CompressedClassSpaceSize = class_space_size;
 401 
 402     // Let Metaspace initialize ccs
 403     Metaspace::initialize_class_space(tmp_class_space);
 404 
 405     // and set up CompressedKlassPointers encoding.
 406     CompressedKlassPointers::initialize((address)_shared_rs.base(), cds_total);
 407 
 408     log_info(cds)("narrow_klass_base = " PTR_FORMAT ", narrow_klass_shift = %d",
 409                   p2i(CompressedKlassPointers::base()), CompressedKlassPointers::shift());
 410 
 411     log_info(cds)("Allocated temporary class space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 412                   CompressedClassSpaceSize, p2i(tmp_class_space.base()));
 413 
 414     assert(_shared_rs.end() == tmp_class_space.base() &&
 415            is_aligned(_shared_rs.base(), MetaspaceShared::reserved_space_alignment()) &&
 416            is_aligned(tmp_class_space.base(), Metaspace::reserve_alignment()) &&
 417            is_aligned(tmp_class_space.size(), Metaspace::reserve_alignment()), "Sanity");
 418   }
 419 
 420 #endif
 421 
 422   init_shared_dump_space(&_mc_region);
 423   SharedBaseAddress = (size_t)_shared_rs.base();
 424   log_info(cds)("Allocated shared space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 425                 _shared_rs.size(), p2i(_shared_rs.base()));
 426 
 427   // We don't want any valid object to be at the very bottom of the archive.
 428   // See ArchivePtrMarker::mark_pointer().
 429   MetaspaceShared::misc_code_space_alloc(16);
 430 
 431   size_t symbol_rs_size = LP64_ONLY(3 * G) NOT_LP64(128 * M);
 432   _symbol_rs = ReservedSpace(symbol_rs_size);
 433   if (!_symbol_rs.is_reserved()) {
 434     vm_exit_during_initialization("Unable to reserve memory for symbols",
 435                                   err_msg(SIZE_FORMAT " bytes.", symbol_rs_size));
 436   }
 437   _symbol_region.init(&_symbol_rs, &_symbol_vs);
 438 }

 460 static GrowableArray<Handle>* _extra_interned_strings = NULL;
 461 
 462 void MetaspaceShared::read_extra_data(const char* filename, TRAPS) {
 463   _extra_interned_strings = new (ResourceObj::C_HEAP, mtInternal)GrowableArray<Handle>(10000, true);
 464 
 465   HashtableTextDump reader(filename);
 466   reader.check_version("VERSION: 1.0");
 467 
 468   while (reader.remain() > 0) {
 469     int utf8_length;
 470     int prefix_type = reader.scan_prefix(&utf8_length);
 471     ResourceMark rm(THREAD);
 472     if (utf8_length == 0x7fffffff) {
 473       // buf_len will overflown 32-bit value.
 474       vm_exit_during_initialization(err_msg("string length too large: %d", utf8_length));
 475     }
 476     int buf_len = utf8_length+1;
 477     char* utf8_buffer = NEW_RESOURCE_ARRAY(char, buf_len);
 478     reader.get_utf8(utf8_buffer, utf8_length);
 479     utf8_buffer[utf8_length] = '\0';
 480 
 481     if (prefix_type == HashtableTextDump::SymbolPrefix) {
 482       SymbolTable::new_permanent_symbol(utf8_buffer);
 483     } else{
 484       assert(prefix_type == HashtableTextDump::StringPrefix, "Sanity");
 485       oop s = StringTable::intern(utf8_buffer, THREAD);
 486 
 487       if (HAS_PENDING_EXCEPTION) {
 488         log_warning(cds, heap)("[line %d] extra interned string allocation failed; size too large: %d",
 489                                reader.last_line_no(), utf8_length);
 490         CLEAR_PENDING_EXCEPTION;
 491       } else {
 492 #if INCLUDE_G1GC
 493         if (UseG1GC) {
 494           typeArrayOop body = java_lang_String::value(s);
 495           const HeapRegion* hr = G1CollectedHeap::heap()->heap_region_containing(body);
 496           if (hr->is_humongous()) {
 497             // Don't keep it alive, so it will be GC'ed before we dump the strings, in order
 498             // to maximize free heap space and minimize fragmentation.
 499             log_warning(cds, heap)("[line %d] extra interned string ignored; size too large: %d",
 500                                 reader.last_line_no(), utf8_length);
 501             continue;
 502           }
 503         }
 504 #endif
 505         // Interned strings are GC'ed if there are no references to it, so let's
 506         // add a reference to keep this string alive.
 507         assert(s != NULL, "must succeed");
 508         Handle h(THREAD, s);
 509         _extra_interned_strings->append(h);
 510       }
 511     }
 512   }
 513 }

 515 void MetaspaceShared::commit_to(ReservedSpace* rs, VirtualSpace* vs, char* newtop) {
 516   Arguments::assert_is_dumping_archive();
 517   char* base = rs->base();
 518   size_t need_committed_size = newtop - base;
 519   size_t has_committed_size = vs->committed_size();
 520   if (need_committed_size < has_committed_size) {
 521     return;
 522   }
 523 
 524   size_t min_bytes = need_committed_size - has_committed_size;
 525   size_t preferred_bytes = 1 * M;
 526   size_t uncommitted = vs->reserved_size() - has_committed_size;
 527 
 528   size_t commit =MAX2(min_bytes, preferred_bytes);
 529   commit = MIN2(commit, uncommitted);
 530   assert(commit <= uncommitted, "sanity");
 531 
 532   bool result = vs->expand_by(commit, false);
 533   if (rs == &_shared_rs) {
 534     ArchivePtrMarker::expand_ptr_end((address*)vs->high());
 535   }
 536 
 537   if (!result) {
 538     vm_exit_during_initialization(err_msg("Failed to expand shared space to " SIZE_FORMAT " bytes",
 539                                           need_committed_size));
 540   }
 541 
 542   assert(rs == &_shared_rs || rs == &_symbol_rs, "must be");
 543   const char* which = (rs == &_shared_rs) ? "shared" : "symbol";
 544   log_debug(cds)("Expanding %s spaces by " SIZE_FORMAT_W(7) " bytes [total " SIZE_FORMAT_W(9)  " bytes ending at %p]",
 545                  which, commit, vs->actual_committed_size(), vs->high());
 546 }

 704 static void rewrite_nofast_bytecode(const methodHandle& method) {
 705   BytecodeStream bcs(method);
 706   while (!bcs.is_last_bytecode()) {
 707     Bytecodes::Code opcode = bcs.next();
 708     switch (opcode) {
 709     case Bytecodes::_getfield:      *bcs.bcp() = Bytecodes::_nofast_getfield;      break;
 710     case Bytecodes::_putfield:      *bcs.bcp() = Bytecodes::_nofast_putfield;      break;
 711     case Bytecodes::_aload_0:       *bcs.bcp() = Bytecodes::_nofast_aload_0;       break;
 712     case Bytecodes::_iload: {
 713       if (!bcs.is_wide()) {
 714         *bcs.bcp() = Bytecodes::_nofast_iload;
 715       }
 716       break;
 717     }
 718     default: break;
 719     }
 720   }
 721 }

 748 // Objects of the Metadata types (such as Klass and ConstantPool) have C++ vtables.
 749 // (In GCC this is the field <Type>::_vptr, i.e., first word in the object.)
 750 //
 751 // Addresses of the vtables and the methods may be different across JVM runs,
 752 // if libjvm.so is dynamically loaded at a different base address.
 753 //
 754 // To ensure that the Metadata objects in the CDS archive always have the correct vtable:
 755 //
 756 // + at dump time:  we redirect the _vptr to point to our own vtables inside
 757 //                  the CDS image
 758 // + at run time:   we clone the actual contents of the vtables from libjvm.so
 759 //                  into our own tables.
 760 
 761 // Currently, the archive contain ONLY the following types of objects that have C++ vtables.
 762 #define CPP_VTABLE_PATCH_TYPES_DO(f) \
 763   f(ConstantPool) \
 764   f(InstanceClassLoaderKlass) \
 765   f(InstanceKlass) \
 766   f(InstanceMirrorKlass) \
 767   f(InstanceRefKlass) \
 768   f(Method) \
 769   f(ObjArrayKlass) \
 770   f(TypeArrayKlass) \
 771   f(ValueArrayKlass) \
 772   f(ValueKlass)
 773 
 774 class CppVtableInfo {
 775   intptr_t _vtable_size;
 776   intptr_t _cloned_vtable[1];
 777 public:
 778   static int num_slots(int vtable_size) {
 779     return 1 + vtable_size; // Need to add the space occupied by _vtable_size;
 780   }
 781   int vtable_size()           { return int(uintx(_vtable_size)); }
 782   void set_vtable_size(int n) { _vtable_size = intptr_t(n); }
 783   intptr_t* cloned_vtable()   { return &_cloned_vtable[0]; }
 784   void zero()                 { memset(_cloned_vtable, 0, sizeof(intptr_t) * vtable_size()); }
 785   // Returns the address of the next CppVtableInfo that can be placed immediately after this CppVtableInfo
 786   static size_t byte_size(int vtable_size) {
 787     CppVtableInfo i;
 788     return pointer_delta(&i._cloned_vtable[vtable_size], &i, sizeof(u1));
 789   }
 790 };

 833 template <class T>
 834 intptr_t* CppVtableCloner<T>::clone_vtable(const char* name, CppVtableInfo* info) {
 835   if (!DumpSharedSpaces) {
 836     assert(_info == 0, "_info is initialized only at dump time");
 837     _info = info; // Remember it -- it will be used by MetaspaceShared::is_valid_shared_method()
 838   }
 839   T tmp; // Allocate temporary dummy metadata object to get to the original vtable.
 840   int n = info->vtable_size();
 841   intptr_t* srcvtable = vtable_of(tmp);
 842   intptr_t* dstvtable = info->cloned_vtable();
 843 
 844   // We already checked (and, if necessary, adjusted n) when the vtables were allocated, so we are
 845   // safe to do memcpy.
 846   log_debug(cds, vtables)("Copying %3d vtable entries for %s", n, name);
 847   memcpy(dstvtable, srcvtable, sizeof(intptr_t) * n);
 848   return dstvtable + n;
 849 }

 851 // To determine the size of the vtable for each type, we use the following
 852 // trick by declaring 2 subclasses:
 853 //
 854 //   class CppVtableTesterA: public InstanceKlass {virtual int   last_virtual_method() {return 1;}    };
 855 //   class CppVtableTesterB: public InstanceKlass {virtual void* last_virtual_method() {return NULL}; };
 856 //
 857 // CppVtableTesterA and CppVtableTesterB's vtables have the following properties:
 858 // - Their size (N+1) is exactly one more than the size of InstanceKlass's vtable (N)
 859 // - The first N entries have are exactly the same as in InstanceKlass's vtable.
 860 // - Their last entry is different.
 861 //
 862 // So to determine the value of N, we just walk CppVtableTesterA and CppVtableTesterB's tables
 863 // and find the first entry that's different.
 864 //
 865 // This works on all C++ compilers supported by Oracle, but you may need to tweak it for more
 866 // esoteric compilers.
 867 
 868 template <class T> class CppVtableTesterB: public T {
 869 public:
 870   virtual int last_virtual_method() {return 1;}
 871 };
 872 
 873 template <class T> class CppVtableTesterA : public T {
 874 public:
 875   virtual void* last_virtual_method() {
 876     // Make this different than CppVtableTesterB::last_virtual_method so the C++
 877     // compiler/linker won't alias the two functions.
 878     return NULL;
 879   }
 880 };
 881 
 882 template <class T>
 883 int CppVtableCloner<T>::get_vtable_length(const char* name) {
 884   CppVtableTesterA<T> a;
 885   CppVtableTesterB<T> b;
 886 
 887   intptr_t* avtable = vtable_of(a);
 888   intptr_t* bvtable = vtable_of(b);
 889 
 890   // Start at slot 1, because slot 0 may be RTTI (on Solaris/Sparc)
 891   int vtable_len = 1;
 892   for (; ; vtable_len++) {
 893     if (avtable[vtable_len] != bvtable[vtable_len]) {
 894       break;
 895     }
 896   }
 897   log_debug(cds, vtables)("Found   %3d vtable entries for %s", vtable_len, name);
 898 
 899   return vtable_len;
 900 }

 942   switch (msotype) {
 943   case MetaspaceObj::SymbolType:
 944   case MetaspaceObj::TypeArrayU1Type:
 945   case MetaspaceObj::TypeArrayU2Type:
 946   case MetaspaceObj::TypeArrayU4Type:

 969           kind = InstanceMirrorKlass_Kind;

 971           kind = InstanceKlass_Kind;

 974         kind = TypeArrayKlass_Kind;
 975       } else {
 976         assert(k->is_objArray_klass(), "must be");
 977         kind = ObjArrayKlass_Kind;

 981

 983     {
 984       Method* m = (Method*)obj;
 985       assert(m->is_method(), "must be");
 986       kind = Method_Kind;

1516   static void iterate_roots(MetaspaceClosure* it) {
1517     // To ensure deterministic contents in the archive, we just need to ensure that
1518     // we iterate the MetsapceObjs in a deterministic order. It doesn't matter where
1519     // the MetsapceObjs are located originally, as they are copied sequentially into
1520     // the archive during the iteration.
1521     //
1522     // The only issue here is that the symbol table and the system directories may be
1523     // randomly ordered, so we copy the symbols and klasses into two arrays and sort
1524     // them deterministically.
1525     //
1526     // During -Xshare:dump, the order of Symbol creation is strictly determined by
1527     // the SharedClassListFile (class loading is done in a single thread and the JIT
1528     // is disabled). Also, Symbols are allocated in monotonically increasing addresses
1529     // (see Symbol::operator new(size_t, int)). So if we iterate the Symbols by
1530     // ascending address order, we ensure that all Symbols are copied into deterministic
1531     // locations in the archive.
1532     GrowableArray<Symbol*>* symbols = _ssc->get_sorted_symbols();

1534       it->push(symbols->adr_at(i));

1537       // Need to fix up the pointers
1538       for (int i = 0; i < _global_klass_objects->length(); i++) {
1539         // NOTE -- this requires that the vtable is NOT yet patched, or else we are hosed.
1540         it->push(_global_klass_objects->adr_at(i));

1544     SystemDictionaryShared::dumptime_classes_do(it);

1546     SymbolTable::metaspace_pointers_do(it);
1547     vmSymbols::metaspace_pointers_do(it);

1619 void VM_PopulateDumpSharedSpace::relocate_to_requested_base_address(CHeapBitMap* ptrmap) {
1620   intx addr_delta = MetaspaceShared::final_delta();
1621   if (addr_delta == 0) {
1622     ArchivePtrMarker::compact((address)SharedBaseAddress, (address)_ro_region.top());
1623   } else {
1624     // We are not able to reserve space at MetaspaceShared::requested_base_address() (due to ASLR).
1625     // This means that the current content of the archive is based on a random
1626     // address. Let's relocate all the pointers, so that it can be mapped to
1627     // MetaspaceShared::requested_base_address() without runtime relocation.
1628     //
1629     // Note: both the base and dynamic archive are written with
1630     // FileMapHeader::_requested_base_address == MetaspaceShared::requested_base_address()
1631 
1632     // Patch all pointers that are marked by ptrmap within this region,
1633     // where we have just dumped all the metaspace data.
1634     address patch_base = (address)SharedBaseAddress;
1635     address patch_end  = (address)_ro_region.top();
1636     size_t size = patch_end - patch_base;
1637 
1638     // the current value of the pointers to be patched must be within this
1639     // range (i.e., must point to valid metaspace objects)
1640     address valid_old_base = patch_base;
1641     address valid_old_end  = patch_end;
1642 
1643     // after patching, the pointers must point inside this range
1644     // (the requested location of the archive, as mapped at runtime).
1645     address valid_new_base = (address)MetaspaceShared::requested_base_address();
1646     address valid_new_end  = valid_new_base + size;
1647 
1648     log_debug(cds)("Relocating archive from [" INTPTR_FORMAT " - " INTPTR_FORMAT " ] to "
1649                    "[" INTPTR_FORMAT " - " INTPTR_FORMAT " ]", p2i(patch_base), p2i(patch_end),
1650                    p2i(valid_new_base), p2i(valid_new_end));
1651 
1652     SharedDataRelocator<true> patcher((address*)patch_base, (address*)patch_end, valid_old_base, valid_old_end,
1653                                       valid_new_base, valid_new_end, addr_delta, ptrmap);
1654     ptrmap->iterate(&patcher);
1655     ArchivePtrMarker::compact(patcher.max_non_null_offset());
1656   }
1657 }

1662 
1663   // We should no longer allocate anything from the metaspace, so that:
1664   //
1665   // (1) Metaspace::allocate might trigger GC if we have run out of
1666   //     committed metaspace, but we can't GC because we're running
1667   //     in the VM thread.

1678   // exist yet, since the constraints table is not shared.  This becomes
1679   // more important now that we don't re-initialize vtables/itables for
1680   // shared classes at runtime, where constraints were previously created.
1681   guarantee(SystemDictionary::constraints()->number_of_entries() == 0,
1682             "loader constraints are not saved");
1683   guarantee(SystemDictionary::placeholders()->number_of_entries() == 0,
1684           "placeholders are not saved");
1685 
1686   // At this point, many classes have been loaded.
1687   // Gather systemDictionary classes in a global array and do everything to
1688   // that so we don't have to walk the SystemDictionary again.
1689   SystemDictionaryShared::check_excluded_classes();
1690   _global_klass_objects = new GrowableArray<Klass*>(1000);
1691   CollectClassesClosure collect_classes;
1692   ClassLoaderDataGraph::loaded_classes_do(&collect_classes);
1693   _global_klass_objects->sort(global_klass_compare);
1694 
1695   print_class_stats();
1696 
1697   // Ensure the ConstMethods won't be modified at run-time
1698   log_info(cds)("Updating ConstMethods ... ");

1700   log_info(cds)("done. ");
1701 
1702   // Remove all references outside the metadata
1703   log_info(cds)("Removing unshareable information ... ");
1704   remove_unshareable_in_classes();
1705   log_info(cds)("done. ");
1706 
1707   MetaspaceShared::allocate_cloned_cpp_vtptrs();
1708   char* cloned_vtables = _mc_region.top();
1709   MetaspaceShared::allocate_cpp_vtable_clones();

1711   ArchiveCompactor::initialize();
1712   ArchiveCompactor::copy_and_compact();
1713 
1714   dump_symbols();

1717   _closed_archive_heap_regions = NULL;

1727   // We don't want to write these addresses into the archive. Same for i2i buffer.

1730          MetaspaceShared::i2i_entry_code_buffers_size());

1734   relocate_to_requested_base_address(&ptrmap);
1735

1737

1739   mapinfo->populate_header(os::vm_allocation_granularity());
1740   mapinfo->set_serialized_data(serialized_data);
1741   mapinfo->set_cloned_vtables(cloned_vtables);
1742   mapinfo->set_i2i_entry_code_buffers(MetaspaceShared::i2i_entry_code_buffers(),

1744   mapinfo->open_for_write();
1745   MetaspaceShared::write_core_archive_regions(mapinfo, _closed_archive_heap_oopmaps, _open_archive_heap_oopmaps);

1747                                         _closed_archive_heap_regions,
1748                                         _closed_archive_heap_oopmaps,
1749                                         MetaspaceShared::first_closed_archive_heap_region,
1750                                         MetaspaceShared::max_closed_archive_heap_region);

1752                                         _open_archive_heap_regions,
1753                                         _open_archive_heap_oopmaps,
1754                                         MetaspaceShared::first_open_archive_heap_region,
1755                                         MetaspaceShared::max_open_archive_heap_region);
1756

1778   // which will fail because InstanceKlasses::remove_unshareable_info()

1913 // file.
1914 void MetaspaceShared::preload_and_dump(TRAPS) {
1915   { TraceTime timer("Dump Shared Spaces", TRACETIME_LOG(Info, startuptime));
1916     ResourceMark rm(THREAD);
1917     char class_list_path_str[JVM_MAXPATHLEN];
1918     // Preload classes to be shared.
1919     const char* class_list_path;
1920     if (SharedClassListFile == NULL) {
1921       // Construct the path to the class list (in jre/lib)
1922       // Walk up two directories from the location of the VM and
1923       // optionally tack on "lib" (depending on platform)
1924       os::jvm_path(class_list_path_str, sizeof(class_list_path_str));
1925       for (int i = 0; i < 3; i++) {
1926         char *end = strrchr(class_list_path_str, *os::file_separator());
1927         if (end != NULL) *end = '\0';
1928       }
1929       int class_list_path_len = (int)strlen(class_list_path_str);
1930       if (class_list_path_len >= 3) {
1931         if (strcmp(class_list_path_str + class_list_path_len - 3, "lib") != 0) {
1932           if (class_list_path_len < JVM_MAXPATHLEN - 4) {
1933             jio_snprintf(class_list_path_str + class_list_path_len,
1934                          sizeof(class_list_path_str) - class_list_path_len,
1935                          "%slib", os::file_separator());
1936             class_list_path_len += 4;
1937           }
1938         }
1939       }
1940       if (class_list_path_len < JVM_MAXPATHLEN - 10) {
1941         jio_snprintf(class_list_path_str + class_list_path_len,
1942                      sizeof(class_list_path_str) - class_list_path_len,
1943                      "%sclasslist", os::file_separator());
1944       }
1945       class_list_path = class_list_path_str;
1946     } else {
1947       class_list_path = SharedClassListFile;
1948     }
1949 
1950     log_info(cds)("Loading classes to share ...");
1951     _has_error_classes = false;
1952     int class_count = preload_classes(class_list_path, THREAD);
1953     if (ExtraSharedClassListFile) {
1954       class_count += preload_classes(ExtraSharedClassListFile, THREAD);
1955     }
1956     log_info(cds)("Loading classes to share: done.");
1957 
1958     log_info(cds)("Shared spaces: preloaded %d classes", class_count);
1959 
1960     if (SharedArchiveConfigFile) {
1961       log_info(cds)("Reading extra data from %s ...", SharedArchiveConfigFile);
1962       read_extra_data(SharedArchiveConfigFile, THREAD);
1963     }
1964     log_info(cds)("Reading extra data: done.");
1965 
1966     HeapShared::init_subgraph_entry_fields(THREAD);
1967 
1968     // Rewrite and link classes
1969     log_info(cds)("Rewriting and linking classes ...");
1970 
1971     // Link any classes which got missed. This would happen if we have loaded classes that
1972     // were not explicitly specified in the classlist. E.g., if an interface implemented by class K
1973     // fails verification, all other interfaces that were not specified in the classlist but
1974     // are implemented by K are not verified.
1975     link_and_cleanup_shared_classes(CATCH);
1976     log_info(cds)("Rewriting and linking classes: done");
1977 
1978     if (HeapShared::is_heap_object_archiving_allowed()) {
1979       // Avoid fragmentation while archiving heap objects.
1980       Universe::heap()->soft_ref_policy()->set_should_clear_all_soft_refs(true);
1981       Universe::heap()->collect(GCCause::_archive_time_gc);
1982       Universe::heap()->soft_ref_policy()->set_should_clear_all_soft_refs(false);
1983     }
1984 
1985     VM_PopulateDumpSharedSpace op;
1986     VMThread::execute(&op);
1987   }

1991 int MetaspaceShared::preload_classes(const char* class_list_path, TRAPS) {
1992   ClassListParser parser(class_list_path);
1993   int class_count = 0;
1994 
1995   while (parser.parse_one_line()) {
1996     Klass* klass = parser.load_current_class(THREAD);
1997     if (HAS_PENDING_EXCEPTION) {
1998       if (klass == NULL &&
1999           (PENDING_EXCEPTION->klass()->name() == vmSymbols::java_lang_ClassNotFoundException())) {
2000         // print a warning only when the pending exception is class not found
2001         log_warning(cds)("Preload Warning: Cannot find %s", parser.current_class_name());
2002       }
2003       CLEAR_PENDING_EXCEPTION;
2004     }
2005     if (klass != NULL) {
2006       if (log_is_enabled(Trace, cds)) {
2007         ResourceMark rm(THREAD);
2008         log_trace(cds)("Shared spaces preloaded: %s", klass->external_name());
2009       }
2010 
2011       if (klass->is_instance_klass()) {
2012         InstanceKlass* ik = InstanceKlass::cast(klass);
2013 
2014         // Link the class to cause the bytecodes to be rewritten and the
2015         // cpcache to be created. The linking is done as soon as classes
2016         // are loaded in order that the related data structures (klass and
2017         // cpCache) are located together.
2018         try_link_class(ik, THREAD);
2019         guarantee(!HAS_PENDING_EXCEPTION, "exception in link_class");
2020       }
2021 
2022       class_count++;
2023     }
2024   }
2025 
2026   return class_count;
2027 }

2029 // Returns true if the class's status has changed
2030 bool MetaspaceShared::try_link_class(InstanceKlass* ik, TRAPS) {
2031   Arguments::assert_is_dumping_archive();
2032   if (ik->init_state() < InstanceKlass::linked &&
2033       !SystemDictionaryShared::has_class_failed_verification(ik)) {
2034     bool saved = BytecodeVerificationLocal;
2035     if (ik->is_shared_unregistered_class() && ik->class_loader() == NULL) {
2036       // The verification decision is based on BytecodeVerificationRemote
2037       // for non-system classes. Since we are using the NULL classloader
2038       // to load non-system classes for customized class loaders during dumping,
2039       // we need to temporarily change BytecodeVerificationLocal to be the same as
2040       // BytecodeVerificationRemote. Note this can cause the parent system
2041       // classes also being verified. The extra overhead is acceptable during
2042       // dumping.
2043       BytecodeVerificationLocal = BytecodeVerificationRemote;
2044     }
2045     ik->link_class(THREAD);
2046     if (HAS_PENDING_EXCEPTION) {
2047       ResourceMark rm(THREAD);
2048       log_warning(cds)("Preload Warning: Verification failed for %s",
2049                     ik->external_name());
2050       CLEAR_PENDING_EXCEPTION;
2051       SystemDictionaryShared::set_class_has_failed_verification(ik);
2052       _has_error_classes = true;
2053     }
2054     BytecodeVerificationLocal = saved;
2055     return true;
2056   } else {
2057     return false;
2058   }
2059 }

2409 // This will reserve two address spaces suitable to house Klass structures, one
2410 //  for the cds archives (static archive and optionally dynamic archive) and
2411 //  optionally one move for ccs.
2412 //
2413 // Since both spaces must fall within the compressed class pointer encoding
2414 //  range, they are allocated close to each other.
2415 //
2416 // Space for archives will be reserved first, followed by a potential gap,
2417 //  followed by the space for ccs:
2418 //
2419 // +-- Base address             A        B                     End
2420 // |                            |        |                      |
2421 // v                            v        v                      v
2422 // +-------------+--------------+        +----------------------+
2423 // | static arc  | [dyn. arch]  | [gap]  | compr. class space   |
2424 // +-------------+--------------+        +----------------------+
2425 //
2426 // (The gap may result from different alignment requirements between metaspace
2427 //  and CDS)
2428 //
2429 // If UseCompressedClassPointers is disabled, only one address space will be
2430 //  reserved:
2431 //
2432 // +-- Base address             End
2433 // |                            |
2434 // v                            v
2435 // +-------------+--------------+
2436 // | static arc  | [dyn. arch]  |
2437 // +-------------+--------------+
2438 //
2439 // Base address: If use_archive_base_addr address is true, the Base address is
2440 //  determined by the address stored in the static archive. If
2441 //  use_archive_base_addr address is false, this base address is determined
2442 //  by the platform.
2443 //
2444 // If UseCompressedClassPointers=1, the range encompassing both spaces will be
2445 //  suitable to en/decode narrow Klass pointers: the base will be valid for
2446 //  encoding, the range [Base, End) not surpass KlassEncodingMetaspaceMax.
2447 //
2448 // Return:
2449 //
2450 // - On success:
2451 //    - archive_space_rs will be reserved and large enough to host static and
2452 //      if needed dynamic archive: [Base, A).
2453 //      archive_space_rs.base and size will be aligned to CDS reserve
2454 //      granularity.
2455 //    - class_space_rs: If UseCompressedClassPointers=1, class_space_rs will
2456 //      be reserved. Its start address will be aligned to metaspace reserve
2457 //      alignment, which may differ from CDS alignment. It will follow the cds
2458 //      archive space, close enough such that narrow class pointer encoding
2459 //      covers both spaces.
2460 //      If UseCompressedClassPointers=0, class_space_rs remains unreserved.
2461 // - On error: NULL is returned and the spaces remain unreserved.
2462 char* MetaspaceShared::reserve_address_space_for_archives(FileMapInfo* static_mapinfo,
2463                                                           FileMapInfo* dynamic_mapinfo,
2464                                                           bool use_archive_base_addr,
2465                                                           ReservedSpace& archive_space_rs,
2466                                                           ReservedSpace& class_space_rs) {
2467 
2468   address const base_address = (address) (use_archive_base_addr ? static_mapinfo->requested_base_address() : NULL);
2469   const size_t archive_space_alignment = MetaspaceShared::reserved_space_alignment();
2470 
2471   // Size and requested location of the archive_space_rs (for both static and dynamic archives)
2472   assert(static_mapinfo->mapping_base_offset() == 0, "Must be");
2473   size_t archive_end_offset  = (dynamic_mapinfo == NULL) ? static_mapinfo->mapping_end_offset() : dynamic_mapinfo->mapping_end_offset();
2474   size_t archive_space_size = align_up(archive_end_offset, archive_space_alignment);
2475 
2476   // If a base address is given, it must have valid alignment and be suitable as encoding base.
2477   if (base_address != NULL) {
2478     assert(is_aligned(base_address, archive_space_alignment),
2479            "Archive base address invalid: " PTR_FORMAT ".", p2i(base_address));
2480     if (Metaspace::using_class_space()) {
2481       assert(CompressedKlassPointers::is_valid_base(base_address),
2482              "Archive base address invalid: " PTR_FORMAT ".", p2i(base_address));
2483     }
2484   }
2485 
2486   if (!Metaspace::using_class_space()) {
2487     // Get the simple case out of the way first:
2488     // no compressed class space, simple allocation.

2490                                      false /* bool large */, (char*)base_address);
2491     if (archive_space_rs.is_reserved()) {

2493              (address)archive_space_rs.base() == base_address, "Sanity");
2494       return archive_space_rs.base();

2505   const size_t class_space_alignment = Metaspace::reserve_alignment();

2729   if (UseSharedSpaces || DumpSharedSpaces) {

2739       base = (address)_shared_rs.base();
