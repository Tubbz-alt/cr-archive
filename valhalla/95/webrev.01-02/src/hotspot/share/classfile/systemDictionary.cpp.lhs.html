 172 void SystemDictionary::compute_java_loaders(TRAPS) {
 173   JavaValue result(T_OBJECT);
 174   InstanceKlass* class_loader_klass = SystemDictionary::ClassLoader_klass();
 175   JavaCalls::call_static(&result,
 176                          class_loader_klass,
 177                          vmSymbols::getSystemClassLoader_name(),
 178                          vmSymbols::void_classloader_signature(),
 179                          CHECK);
 180 
 181   _java_system_loader = OopHandle::create((oop)result.get_jobject());
 182 
 183   JavaCalls::call_static(&result,
 184                          class_loader_klass,
 185                          vmSymbols::getPlatformClassLoader_name(),
 186                          vmSymbols::void_classloader_signature(),
 187                          CHECK);
 188 
 189   _java_platform_loader = OopHandle::create((oop)result.get_jobject());
 190 }

 242 Symbol* SystemDictionary::class_name_symbol(const char* name, Symbol* exception, TRAPS) {
 243   if (name == NULL) {
 244     THROW_MSG_0(exception, "No class name given");
 245   }
 246   if ((int)strlen(name) > Symbol::max_length()) {
 247     // It's impossible to create this class;  the name cannot fit
 248     // into the constant pool.
 249     Exceptions::fthrow(THREAD_AND_LOCATION, exception,
 250                        "Class name exceeds maximum length of %d: %s",
 251                        Symbol::max_length(),
 252                        name);
 253     return NULL;
 254   }
 255   // Callers should ensure that the name is never an illegal UTF8 string.
 256   assert(UTF8::is_legal_utf8((const unsigned char*)name, (int)strlen(name), false),
 257          "Class name is not a valid utf8 string.");
 258 
 259   // Make a new symbol for the class name.
 260   return SymbolTable::new_symbol(name);
 261 }

 265 Klass* SystemDictionary::resolve_or_fail(Symbol* class_name, Handle class_loader, Handle protection_domain, bool throw_error, TRAPS) {
 266   Klass* klass = resolve_or_null(class_name, class_loader, protection_domain, THREAD);
 267   if (HAS_PENDING_EXCEPTION || klass == NULL) {
 268     // can return a null klass
 269     klass = handle_resolution_exception(class_name, throw_error, klass, THREAD);
 270   }
 271   return klass;
 272 }

 274 Klass* SystemDictionary::handle_resolution_exception(Symbol* class_name,
 275                                                      bool throw_error,
 276                                                      Klass* klass, TRAPS) {
 277   if (HAS_PENDING_EXCEPTION) {
 278     // If we have a pending exception we forward it to the caller, unless throw_error is true,
 279     // in which case we have to check whether the pending exception is a ClassNotFoundException,
 280     // and if so convert it to a NoClassDefFoundError
 281     // And chain the original ClassNotFoundException
 282     if (throw_error && PENDING_EXCEPTION->is_a(SystemDictionary::ClassNotFoundException_klass())) {
 283       ResourceMark rm(THREAD);
 284       assert(klass == NULL, "Should not have result with exception pending");
 285       Handle e(THREAD, PENDING_EXCEPTION);
 286       CLEAR_PENDING_EXCEPTION;
 287       THROW_MSG_CAUSE_NULL(vmSymbols::java_lang_NoClassDefFoundError(), class_name->as_C_string(), e);
 288     } else {
 289       return NULL;
 290     }
 291   }
 292   // Class not found, throw appropriate error or exception depending on value of throw_error
 293   if (klass == NULL) {
 294     ResourceMark rm(THREAD);
 295     if (throw_error) {
 296       THROW_MSG_NULL(vmSymbols::java_lang_NoClassDefFoundError(), class_name->as_C_string());
 297     } else {
 298       THROW_MSG_NULL(vmSymbols::java_lang_ClassNotFoundException(), class_name->as_C_string());
 299     }
 300   }
 301   return klass;
 302 }

 322 // name may be in the form of "java/lang/Object" or "Ljava/lang/Object;"
 323 InstanceKlass* SystemDictionary::resolve_instance_class_or_null_helper(Symbol* class_name,
 324                                                                        Handle class_loader,
 325                                                                        Handle protection_domain,
 326                                                                        TRAPS) {
 327   assert(class_name != NULL && !Signature::is_array(class_name), "must be");
 328   if (Signature::has_envelope(class_name)) {
 329     ResourceMark rm(THREAD);
 330     // Ignore wrapping L and ;. (and Q and ; for value types);
 331     TempNewSymbol name = SymbolTable::new_symbol(class_name->as_C_string() + 1,
 332                                                  class_name->utf8_length() - 2);
 333     return resolve_instance_class_or_null(name, class_loader, protection_domain, THREAD);
 334   } else {
 335     return resolve_instance_class_or_null(class_name, class_loader, protection_domain, THREAD);
 336   }
 337 }

 343 // Forwards to resolve_instance_class_or_null
 344 
 345 Klass* SystemDictionary::resolve_array_class_or_null(Symbol* class_name,
 346                                                      Handle class_loader,
 347                                                      Handle protection_domain,
 348                                                      TRAPS) {
 349   assert(Signature::is_array(class_name), "must be array");
 350   ResourceMark rm(THREAD);
 351   SignatureStream ss(class_name, false);
 352   int ndims = ss.skip_array_prefix();  // skip all '['s
 353   Klass* k = NULL;
 354   BasicType t = ss.type();
 355   if (ss.has_envelope()) {
 356     Symbol* obj_class = ss.as_symbol();
 357     k = SystemDictionary::resolve_instance_class_or_null(obj_class,
 358                                                          class_loader,
 359                                                          protection_domain,
 360                                                          CHECK_NULL);
 361     if (k != NULL) {
 362       k = k->array_klass(ndims, CHECK_NULL);
 363     }
 364   } else {
 365     k = Universe::typeArrayKlassObj(t);
 366     k = TypeArrayKlass::cast(k)->array_klass(ndims, CHECK_NULL);
 367   }
 368   return k;
 369 }

 371 // Must be called for any super-class or super-interface resolution
 372 // during class definition to allow class circularity checking
 373 // super-interface callers:
 374 //    parse_interfaces - for defineClass & jvmtiRedefineClasses
 375 // super-class callers:
 376 //   ClassFileParser - for defineClass & jvmtiRedefineClasses
 377 //   load_shared_class - while loading a class from shared archive
 378 //   resolve_instance_class_or_null:
 379 //     via: handle_parallel_super_load
 380 //      when resolving a class that has an existing placeholder with
 381 //      a saved superclass [i.e. a defineClass is currently in progress]
 382 //      if another thread is trying to resolve the class, it must do
 383 //      super-class checks on its own thread to catch class circularity
 384 // This last call is critical in class circularity checking for cases
 385 // where classloading is delegated to different threads and the
 386 // classloader lock is released.
 387 // Take the case: Base->Super->Base
 388 //   1. If thread T1 tries to do a defineClass of class Base
 389 //    resolve_super_or_fail creates placeholder: T1, Base (super Super)
 390 //   2. resolve_instance_class_or_null does not find SD or placeholder for Super
 391 //    so it tries to load Super
 392 //   3. If we load the class internally, or user classloader uses same thread
 393 //      loadClassFromxxx or defineClass via parseClassFile Super ...
 394 //      3.1 resolve_super_or_fail creates placeholder: T1, Super (super Base)
 395 //      3.3 resolve_instance_class_or_null Base, finds placeholder for Base
 396 //      3.4 calls resolve_super_or_fail Base
 397 //      3.5 finds T1,Base -> throws class circularity
 398 //OR 4. If T2 tries to resolve Super via defineClass Super ...
 399 //      4.1 resolve_super_or_fail creates placeholder: T2, Super (super Base)
 400 //      4.2 resolve_instance_class_or_null Base, finds placeholder for Base (super Super)
 401 //      4.3 calls resolve_super_or_fail Super in parallel on own thread T2
 402 //      4.4 finds T2, Super -> throws class circularity
 403 // Must be called, even if superclass is null, since this is
 404 // where the placeholder entry is created which claims this
 405 // thread is loading this class/classloader.
 406 // Be careful when modifying this code: once you have run
 407 // placeholders()->find_and_add(PlaceholderTable::LOAD_SUPER),
 408 // you need to find_and_remove it before returning.
 409 // So be careful to not exit with a CHECK_ macro betweeen these calls.
 410 InstanceKlass* SystemDictionary::resolve_super_or_fail(Symbol* child_name,
 411                                                        Symbol* super_name,
 412                                                        Handle class_loader,
 413                                                        Handle protection_domain,
 414                                                        bool is_superclass,
 415                                                        TRAPS) {
 416   assert(!Signature::is_array(super_name), "invalid super class name");
 417 #if INCLUDE_CDS
 418   if (DumpSharedSpaces) {
 419     // Special processing for handling UNREGISTERED shared classes.
 420     InstanceKlass* k = SystemDictionaryShared::dump_time_resolve_super_or_fail(child_name,
 421         super_name, class_loader, protection_domain, is_superclass, CHECK_NULL);
 422     if (k) {
 423       return k;
 424     }
 425   }
 426 #endif // INCLUDE_CDS
 427 
 428   // Double-check, if child class is already loaded, just return super-class,interface
 429   // Don't add a placedholder if already loaded, i.e. already in appropriate class loader
 430   // dictionary.
 431   // Make sure there's a placeholder for the *child* before resolving.
 432   // Used as a claim that this thread is currently loading superclass/classloader
 433   // Used here for ClassCircularity checks and also for heap verification
 434   // (every InstanceKlass needs to be in its class loader dictionary or have a placeholder).
 435   // Must check ClassCircularity before checking if super class is already loaded.
 436   //
 437   // We might not already have a placeholder if this child_name was
 438   // first seen via resolve_from_stream (jni_DefineClass or JVM_DefineClass);
 439   // the name of the class might not be known until the stream is actually
 440   // parsed.
 441   // Bugs 4643874, 4715493
 442 
 443   ClassLoaderData* loader_data = class_loader_data(class_loader);
 444   Dictionary* dictionary = loader_data->dictionary();
 445   unsigned int d_hash = dictionary->compute_hash(child_name);
 446   unsigned int p_hash = placeholders()->compute_hash(child_name);
 447   int p_index = placeholders()->hash_to_index(p_hash);
 448   // can't throw error holding a lock
 449   bool child_already_loaded = false;
 450   bool throw_circularity_error = false;
 451   {
 452     MutexLocker mu(THREAD, SystemDictionary_lock);
 453     InstanceKlass* childk = find_class(d_hash, child_name, dictionary);
 454     InstanceKlass* quicksuperk;
 455     // to support // loading: if child done loading, just return superclass
 456     // if super_name, & class_loader don't match:
 457     // if initial define, SD update will give LinkageError
 458     // if redefine: compare_class_versions will give HIERARCHY_CHANGED
 459     // so we don't throw an exception here.
 460     // see: nsk redefclass014 & java.lang.instrument Instrument032
 461     if ((childk != NULL ) && (is_superclass) &&
 462         ((quicksuperk = childk->java_super()) != NULL) &&
 463          ((quicksuperk->name() == super_name) &&
 464             (quicksuperk->class_loader() == class_loader()))) {
 465            return quicksuperk;
 466     } else {
 467       PlaceholderEntry* probe = placeholders()->get_entry(p_index, p_hash, child_name, loader_data);
 468       if (probe && probe->check_seen_thread(THREAD, PlaceholderTable::LOAD_SUPER)) {
 469           throw_circularity_error = true;
 470       }
 471     }
 472     if (!throw_circularity_error) {
 473       // Be careful not to exit resolve_super
 474       PlaceholderEntry* newprobe = placeholders()->find_and_add(p_index, p_hash, child_name, loader_data, PlaceholderTable::LOAD_SUPER, super_name, THREAD);
 475     }
 476   }
 477   if (throw_circularity_error) {
 478       ResourceMark rm(THREAD);
 479       THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), child_name->as_C_string());
 480   }
 481 
 482 // java.lang.Object should have been found above
 483   assert(super_name != NULL, "null super class for resolving");
 484   // Resolve the super class or interface, check results on return
 485   InstanceKlass* superk =
 486     SystemDictionary::resolve_instance_class_or_null_helper(super_name,
 487                                                             class_loader,
 488                                                             protection_domain,
 489                                                             THREAD);
 490 
 491   // Clean up of placeholders moved so that each classloadAction registrar self-cleans up
 492   // It is no longer necessary to keep the placeholder table alive until update_dictionary
 493   // or error. GC used to walk the placeholder table as strong roots.
 494   // The instanceKlass is kept alive because the class loader is on the stack,
 495   // which keeps the loader_data alive, as well as all instanceKlasses in
 496   // the loader_data. parseClassFile adds the instanceKlass to loader_data.
 497   {
 498     MutexLocker mu(THREAD, SystemDictionary_lock);
 499     placeholders()->find_and_remove(p_index, p_hash, child_name, loader_data, PlaceholderTable::LOAD_SUPER, THREAD);
 500     SystemDictionary_lock->notify_all();
 501   }
 502   if (HAS_PENDING_EXCEPTION || superk == NULL) {
 503     // can null superk
 504     Klass* k = handle_resolution_exception(super_name, true, superk, THREAD);
 505     assert(k == NULL || k == superk, "must be");
 506     if (k == NULL) {
 507       superk = NULL;
 508     }
 509   }
 510 
 511   return superk;
 512 }

 514 Klass* SystemDictionary::resolve_inline_type_field_or_fail(AllFieldStream* fs,
 515                                                            Handle class_loader,
 516                                                            Handle protection_domain,
 517                                                            bool throw_error,
 518                                                            TRAPS) {
 519   Symbol* class_name = fs->signature()->fundamental_name(THREAD);
 520   class_loader = Handle(THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 521   ClassLoaderData* loader_data = class_loader_data(class_loader);
 522   unsigned int p_hash = placeholders()->compute_hash(class_name);
 523   int p_index = placeholders()->hash_to_index(p_hash);
 524   bool throw_circularity_error = false;
 525   PlaceholderEntry* oldprobe;
 526 
 527   {
 528     MutexLocker mu(THREAD, SystemDictionary_lock);
 529     oldprobe = placeholders()->get_entry(p_index, p_hash, class_name, loader_data);
 530     if (oldprobe != NULL &&
 531       oldprobe->check_seen_thread(THREAD, PlaceholderTable::INLINE_TYPE_FIELD)) {
 532       throw_circularity_error = true;
 533 
 534     } else {
 535       placeholders()->find_and_add(p_index, p_hash, class_name, loader_data,
 536                                    PlaceholderTable::INLINE_TYPE_FIELD, NULL, THREAD);
 537     }
 538   }
 539 
 540   Klass* klass = NULL;
 541   if (!throw_circularity_error) {
 542     klass = SystemDictionary::resolve_or_fail(class_name, class_loader,
 543                                                protection_domain, true, THREAD);
 544   } else {
 545     ResourceMark rm(THREAD);
 546     THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), class_name->as_C_string());
 547   }
 548 
 549   {
 550     MutexLocker mu(THREAD, SystemDictionary_lock);
 551     placeholders()->find_and_remove(p_index, p_hash, class_name, loader_data,
 552                                     PlaceholderTable::INLINE_TYPE_FIELD, THREAD);
 553   }
 554 
 555   class_name->decrement_refcount();
 556   return klass;
 557 }

 559 void SystemDictionary::validate_protection_domain(InstanceKlass* klass,
 560                                                   Handle class_loader,
 561                                                   Handle protection_domain,
 562                                                   TRAPS) {
 563   // Now we have to call back to java to check if the initating class has access
 564   JavaValue result(T_VOID);
 565   LogTarget(Debug, protectiondomain) lt;
 566   if (lt.is_enabled()) {
 567     ResourceMark rm(THREAD);
 568     // Print out trace information
 569     LogStream ls(lt);
 570     ls.print_cr("Checking package access");
 571     if (class_loader() != NULL) {
 572       ls.print("class loader: ");
 573       class_loader()->print_value_on(&ls);

 626 // We only get here if this thread finds that another thread
 627 // has already claimed the placeholder token for the current operation,
 628 // but that other thread either never owned or gave up the
 629 // object lock
 630 // Waits on SystemDictionary_lock to indicate placeholder table updated
 631 // On return, caller must recheck placeholder table state
 632 //
 633 // We only get here if
 634 //  1) custom classLoader, i.e. not bootstrap classloader
 635 //  2) custom classLoader has broken the class loader objectLock
 636 //     so another thread got here in parallel
 637 //
 638 // lockObject must be held.
 639 // Complicated dance due to lock ordering:
 640 // Must first release the classloader object lock to
 641 // allow initial definer to complete the class definition
 642 // and to avoid deadlock
 643 // Reclaim classloader lock object with same original recursion count
 644 // Must release SystemDictionary_lock after notify, since
 645 // class loader lock must be claimed before SystemDictionary_lock
 646 // to prevent deadlocks
 647 //
 648 // The notify allows applications that did an untimed wait() on
 649 // the classloader object lock to not hang.
 650 void SystemDictionary::double_lock_wait(Handle lockObject, TRAPS) {
 651   assert_lock_strong(SystemDictionary_lock);
 652 
 653   bool calledholdinglock
 654       = ObjectSynchronizer::current_thread_holds_lock((JavaThread*)THREAD, lockObject);
 655   assert(calledholdinglock,"must hold lock for notify");
 656   assert((lockObject() != _system_loader_lock_obj.resolve() &&
 657          !is_parallelCapable(lockObject)), "unexpected double_lock_wait");
 658   ObjectSynchronizer::notifyall(lockObject, THREAD);
 659   intx recursions =  ObjectSynchronizer::complete_exit(lockObject, THREAD);
 660   SystemDictionary_lock->wait();
 661   SystemDictionary_lock->unlock();
 662   ObjectSynchronizer::reenter(lockObject, recursions, THREAD);
 663   SystemDictionary_lock->lock();
 664 }

 666 // If the class in is in the placeholder table, class loading is in progress
 667 // For cases where the application changes threads to load classes, it
 668 // is critical to ClassCircularity detection that we try loading
 669 // the superclass on the same thread internally, so we do parallel
 670 // super class loading here.
 671 // This also is critical in cases where the original thread gets stalled
 672 // even in non-circularity situations.
 673 // Note: must call resolve_super_or_fail even if null super -
 674 // to force placeholder entry creation for this class for circularity detection
 675 // Caller must check for pending exception
 676 // Returns non-null Klass* if other thread has completed load
 677 // and we are done,
 678 // If return null Klass* and no pending exception, the caller must load the class
 679 InstanceKlass* SystemDictionary::handle_parallel_super_load(
 680     Symbol* name, Symbol* superclassname, Handle class_loader,
 681     Handle protection_domain, Handle lockObject, TRAPS) {
 682 
 683   ClassLoaderData* loader_data = class_loader_data(class_loader);
 684   Dictionary* dictionary = loader_data->dictionary();
 685   unsigned int d_hash = dictionary->compute_hash(name);
 686   unsigned int p_hash = placeholders()->compute_hash(name);
 687   int p_index = placeholders()->hash_to_index(p_hash);
 688 
 689   // superk is not used, resolve_super called for circularity check only
 690   // This code is reached in two situations. One if this thread
 691   // is loading the same class twice (e.g. ClassCircularity, or
 692   // java.lang.instrument).
 693   // The second is if another thread started the resolve_super first
 694   // and has not yet finished.
 695   // In both cases the original caller will clean up the placeholder
 696   // entry on error.
 697   Klass* superk = SystemDictionary::resolve_super_or_fail(name,
 698                                                           superclassname,
 699                                                           class_loader,
 700                                                           protection_domain,
 701                                                           true,
 702                                                           CHECK_NULL);
 703 
 704   // parallelCapable class loaders do NOT wait for parallel superclass loads to complete
 705   // Serial class loaders and bootstrap classloader do wait for superclass loads
 706  if (!class_loader.is_null() && is_parallelCapable(class_loader)) {
 707     MutexLocker mu(THREAD, SystemDictionary_lock);
 708     // Check if classloading completed while we were loading superclass or waiting
 709     return find_class(d_hash, name, dictionary);
 710   }
 711 
 712   // must loop to both handle other placeholder updates
 713   // and spurious notifications
 714   bool super_load_in_progress = true;
 715   PlaceholderEntry* placeholder;
 716   while (super_load_in_progress) {
 717     MutexLocker mu(THREAD, SystemDictionary_lock);
 718     // Check if classloading completed while we were loading superclass or waiting
 719     InstanceKlass* check = find_class(d_hash, name, dictionary);
 720     if (check != NULL) {
 721       // Klass is already loaded, so just return it
 722       return check;
 723     } else {
 724       placeholder = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 725       if (placeholder && placeholder->super_load_in_progress() ){
 726         // We only get here if the application has released the
 727         // classloader lock when another thread was in the middle of loading a
 728         // superclass/superinterface for this class, and now
 729         // this thread is also trying to load this class.
 730         // To minimize surprises, the first thread that started to
 731         // load a class should be the one to complete the loading
 732         // with the classfile it initially expected.
 733         // This logic has the current thread wait once it has done
 734         // all the superclass/superinterface loading it can, until
 735         // the original thread completes the class loading or fails
 736         // If it completes we will use the resulting InstanceKlass
 737         // which we will find below in the systemDictionary.
 738         // We also get here for parallel bootstrap classloader
 739         if (class_loader.is_null()) {
 740           SystemDictionary_lock->wait();
 741         } else {
 742           double_lock_wait(lockObject, THREAD);
 743         }
 744       } else {
 745         // If not in SD and not in PH, other thread's load must have failed
 746         super_load_in_progress = false;
 747       }
 748     }
 749   }
 750   return NULL;
 751 }

 764 // Be careful when modifying this code: once you have run
 765 // placeholders()->find_and_add(PlaceholderTable::LOAD_INSTANCE),
 766 // you need to find_and_remove it before returning.
 767 // So be careful to not exit with a CHECK_ macro betweeen these calls.
 768 //
 769 // name must be in the form of "java/lang/Object" -- cannot be "Ljava/lang/Object;"
 770 InstanceKlass* SystemDictionary::resolve_instance_class_or_null(Symbol* name,
 771                                                                 Handle class_loader,
 772                                                                 Handle protection_domain,
 773                                                                 TRAPS) {
 774   assert(name != NULL && !Signature::is_array(name) &&
 775          !Signature::has_envelope(name), "invalid class name");
 776 
 777   EventClassLoad class_load_start_event;
 778 
 779   HandleMark hm(THREAD);
 780 
 781   // Fix for 4474172; see evaluation for more details
 782   class_loader = Handle(THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 783   ClassLoaderData* loader_data = register_loader(class_loader);
 784   Dictionary* dictionary = loader_data->dictionary();
 785   unsigned int d_hash = dictionary->compute_hash(name);
 786 
 787   // Do lookup to see if class already exist and the protection domain
 788   // has the right access
 789   // This call uses find which checks protection domain already matches
 790   // All subsequent calls use find_class, and set has_loaded_class so that
 791   // before we return a result we call out to java to check for valid protection domain
 792   // to allow returning the Klass* and add it to the pd_set if it is valid
 793   {
 794     InstanceKlass* probe = dictionary->find(d_hash, name, protection_domain);
 795     if (probe != NULL) return probe;
 796   }
 797 
 798   // Non-bootstrap class loaders will call out to class loader and
 799   // define via jvm/jni_DefineClass which will acquire the
 800   // class loader object lock to protect against multiple threads
 801   // defining the class in parallel by accident.
 802   // This lock must be acquired here so the waiter will find
 803   // any successful result in the SystemDictionary and not attempt
 804   // the define.
 805   // ParallelCapable Classloaders and the bootstrap classloader
 806   // do not acquire lock here.
 807   bool DoObjectLock = true;
 808   if (is_parallelCapable(class_loader)) {
 809     DoObjectLock = false;
 810   }
 811 
 812   unsigned int p_hash = placeholders()->compute_hash(name);
 813   int p_index = placeholders()->hash_to_index(p_hash);
 814 
 815   // Class is not in SystemDictionary so we have to do loading.
 816   // Make sure we are synchronized on the class loader before we proceed
 817   Handle lockObject = compute_loader_lock_object(class_loader, THREAD);
 818   check_loader_lock_contention(lockObject, THREAD);
 819   ObjectLocker ol(lockObject, THREAD, DoObjectLock);
 820 
 821   // Check again (after locking) if class already exist in SystemDictionary
 822   bool class_has_been_loaded   = false;
 823   bool super_load_in_progress  = false;
 824   bool havesupername = false;
 825   InstanceKlass* k = NULL;
 826   PlaceholderEntry* placeholder;
 827   Symbol* superclassname = NULL;
 828 
 829   assert(THREAD->can_call_java(),
 830          "can not load classes with compiler thread: class=%s, classloader=%s",
 831          name->as_C_string(),
 832          class_loader.is_null() ? "null" : class_loader->klass()->name()->as_C_string());
 833   {
 834     MutexLocker mu(THREAD, SystemDictionary_lock);
 835     InstanceKlass* check = find_class(d_hash, name, dictionary);
 836     if (check != NULL) {
 837       // InstanceKlass is already loaded, so just return it
 838       class_has_been_loaded = true;
 839       k = check;
 840     } else {
 841       placeholder = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 842       if (placeholder && placeholder->super_load_in_progress()) {
 843          super_load_in_progress = true;
 844          if (placeholder->havesupername() == true) {
 845            superclassname = placeholder->supername();
 846            havesupername = true;
 847          }
 848       }
 849     }
 850   }
 851 
 852   // If the class is in the placeholder table, class loading is in progress
 853   if (super_load_in_progress && havesupername==true) {
 854     k = handle_parallel_super_load(name,
 855                                    superclassname,
 856                                    class_loader,
 857                                    protection_domain,
 858                                    lockObject, THREAD);
 859     if (HAS_PENDING_EXCEPTION) {
 860       return NULL;
 861     }
 862     if (k != NULL) {
 863       class_has_been_loaded = true;
 864     }
 865   }
 866 
 867   bool throw_circularity_error = false;
 868   if (!class_has_been_loaded) {
 869     bool load_instance_added = false;
 870 
 871     // add placeholder entry to record loading instance class
 872     // Five cases:
 873     // All cases need to prevent modifying bootclasssearchpath
 874     // in parallel with a classload of same classname
 875     // Redefineclasses uses existence of the placeholder for the duration
 876     // of the class load to prevent concurrent redefinition of not completely
 877     // defined classes.
 878     // case 1. traditional classloaders that rely on the classloader object lock
 879     //   - no other need for LOAD_INSTANCE
 880     // case 2. traditional classloaders that break the classloader object lock
 881     //    as a deadlock workaround. Detection of this case requires that
 882     //    this check is done while holding the classloader object lock,
 883     //    and that lock is still held when calling classloader's loadClass.
 884     //    For these classloaders, we ensure that the first requestor
 885     //    completes the load and other requestors wait for completion.
 886     // case 3. Bootstrap classloader - don't own objectLocker
 887     //    This classloader supports parallelism at the classloader level,
 888     //    but only allows a single load of a class/classloader pair.
 889     //    No performance benefit and no deadlock issues.
 890     // case 4. parallelCapable user level classloaders - without objectLocker
 891     //    Allow parallel classloading of a class/classloader pair
 892 
 893     {
 894       MutexLocker mu(THREAD, SystemDictionary_lock);
 895       if (class_loader.is_null() || !is_parallelCapable(class_loader)) {
 896         PlaceholderEntry* oldprobe = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 897         if (oldprobe) {
 898           // only need check_seen_thread once, not on each loop
 899           // 6341374 java/lang/Instrument with -Xcomp
 900           if (oldprobe->check_seen_thread(THREAD, PlaceholderTable::LOAD_INSTANCE)) {
 901             throw_circularity_error = true;
 902           } else {
 903             // case 1: traditional: should never see load_in_progress.
 904             while (!class_has_been_loaded && oldprobe && oldprobe->instance_load_in_progress()) {
 905 
 906               // case 3: bootstrap classloader: prevent futile classloading,
 907               // wait on first requestor
 908               if (class_loader.is_null()) {
 909                 SystemDictionary_lock->wait();
 910               } else {
 911               // case 2: traditional with broken classloader lock. wait on first
 912               // requestor.
 913                 double_lock_wait(lockObject, THREAD);
 914               }
 915               // Check if classloading completed while we were waiting
 916               InstanceKlass* check = find_class(d_hash, name, dictionary);
 917               if (check != NULL) {
 918                 // Klass is already loaded, so just return it
 919                 k = check;
 920                 class_has_been_loaded = true;
 921               }
 922               // check if other thread failed to load and cleaned up
 923               oldprobe = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 924             }
 925           }
 926         }
 927       }
 928       // All cases: add LOAD_INSTANCE holding SystemDictionary_lock
 929       // case 4: parallelCapable: allow competing threads to try
 930       // LOAD_INSTANCE in parallel
 931 
 932       if (!throw_circularity_error && !class_has_been_loaded) {
 933         PlaceholderEntry* newprobe = placeholders()->find_and_add(p_index, p_hash, name, loader_data, PlaceholderTable::LOAD_INSTANCE, NULL, THREAD);
 934         load_instance_added = true;
 935         // For class loaders that do not acquire the classloader object lock,
 936         // if they did not catch another thread holding LOAD_INSTANCE,
 937         // need a check analogous to the acquire ObjectLocker/find_class
 938         // i.e. now that we hold the LOAD_INSTANCE token on loading this class/CL
 939         // one final check if the load has already completed
 940         // class loaders holding the ObjectLock shouldn't find the class here
 941         InstanceKlass* check = find_class(d_hash, name, dictionary);
 942         if (check != NULL) {
 943           // Klass is already loaded, so return it after checking/adding protection domain
 944           k = check;
 945           class_has_been_loaded = true;
 946         }
 947       }
 948     }
 949 
 950     // must throw error outside of owning lock
 951     if (throw_circularity_error) {
 952       assert(!HAS_PENDING_EXCEPTION && load_instance_added == false,"circularity error cleanup");
 953       ResourceMark rm(THREAD);
 954       THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), name->as_C_string());
 955     }
 956 
 957     if (!class_has_been_loaded) {
 958 
 959       // Do actual loading
 960       k = load_instance_class(name, class_loader, THREAD);
 961 
 962       // If everything was OK (no exceptions, no null return value), and
 963       // class_loader is NOT the defining loader, do a little more bookkeeping.
 964       if (!HAS_PENDING_EXCEPTION && k != NULL &&
 965         k->class_loader() != class_loader()) {
 966 
 967         check_constraints(d_hash, k, class_loader, false, THREAD);
 968 
 969         // Need to check for a PENDING_EXCEPTION again; check_constraints
 970         // can throw but we may have to remove entry from the placeholder table below.
 971         if (!HAS_PENDING_EXCEPTION) {
 972           // Record dependency for non-parent delegation.
 973           // This recording keeps the defining class loader of the klass (k) found
 974           // from being unloaded while the initiating class loader is loaded
 975           // even if the reference to the defining class loader is dropped
 976           // before references to the initiating class loader.
 977           loader_data->record_dependency(k);
 978 
 979           { // Grabbing the Compile_lock prevents systemDictionary updates
 980             // during compilations.
 981             MutexLocker mu(THREAD, Compile_lock);
 982             update_dictionary(d_hash, p_index, p_hash,
 983               k, class_loader, THREAD);
 984           }
 985 
 986           if (JvmtiExport::should_post_class_load()) {
 987             Thread *thread = THREAD;
 988             assert(thread->is_Java_thread(), "thread->is_Java_thread()");
 989             JvmtiExport::post_class_load((JavaThread *) thread, k);
 990           }
 991         }
 992       }
 993     } // load_instance_class
 994 
 995     if (load_instance_added == true) {
 996       // clean up placeholder entries for LOAD_INSTANCE success or error
 997       // This brackets the SystemDictionary updates for both defining
 998       // and initiating loaders
 999       MutexLocker mu(THREAD, SystemDictionary_lock);
1000       placeholders()->find_and_remove(p_index, p_hash, name, loader_data, PlaceholderTable::LOAD_INSTANCE, THREAD);
1001       SystemDictionary_lock->notify_all();
1002     }
1003   }
1004 
1005   if (HAS_PENDING_EXCEPTION || k == NULL) {
1006     return NULL;
1007   }
1008   if (class_load_start_event.should_commit()) {
1009     post_class_load_event(&class_load_start_event, k, loader_data);
1010   }
1011 #ifdef ASSERT
1012   {
1013     ClassLoaderData* loader_data = k->class_loader_data();
1014     MutexLocker mu(THREAD, SystemDictionary_lock);
1015     InstanceKlass* kk = find_class(name, loader_data);
1016     assert(kk == k, "should be present in dictionary");
1017   }
1018 #endif
1019 
1020   // return if the protection domain in NULL
1021   if (protection_domain() == NULL) return k;
1022 
1023   // Check the protection domain has the right access
1024   if (dictionary->is_valid_protection_domain(d_hash, name,
1025                                              protection_domain)) {
1026     return k;
1027   }
1028 
1029   // Verify protection domain. If it fails an exception is thrown
1030   validate_protection_domain(k, class_loader, protection_domain, CHECK_NULL);
1031 
1032   return k;
1033 }

1130   assert(st->need_verify(), "invariant");

1133   // Note that we do this even though this klass might
1134   // already be present in the SystemDictionary, otherwise we would not
1135   // throw potential ClassFormatErrors.

1377                mod_entry == loader_data->unnamed_module(),
1378                "the unnamed module is not defined in the classloader");

1380       }
1381     }
1382     return false;
1383   } else {

1464                                                    Handle class_loader,
1465                                                    Handle protection_domain,
1466                                                    const ClassFileStream *cfs,
1467                                                    PackageEntry* pkg_entry,
1468                                                    TRAPS) {
1469   assert(ik != NULL, "sanity");
1470   assert(!ik->is_unshareable_info_restored(), "shared class can be loaded only once");
1471   Symbol* class_name = ik->name();
1472 
1473   bool visible = is_shared_class_visible(
1474                           class_name, ik, pkg_entry, class_loader, CHECK_NULL);
1475   if (!visible) {
1476     return NULL;
1477   }
1478 
1479   if (!check_shared_class_super_types(ik, class_loader, protection_domain, THREAD)) {
1480     return NULL;

1484   if (ik->has_inline_type_fields()) {
1485     for (AllFieldStream fs(ik->fields(), ik->constants()); !fs.done(); fs.next()) {
1486       if (Signature::basic_type(fs.signature()) == T_VALUETYPE) {
1487         if (!fs.access_flags().is_static()) {
1488           // Pre-load inline class
1489           Klass* real_k = SystemDictionary::resolve_inline_type_field_or_fail(&fs,
1490             class_loader, protection_domain, true, CHECK_NULL);
1491           Klass* k = ik->get_inline_type_field_klass_or_null(fs.index());

1494             return NULL;

1507   }
1508   if (new_ik != NULL) {
1509     // The class is changed by CFLH. Return the new class. The shared class is

1516   // must be reset.

1518   // Updating methods must be done under a lock so multiple
1519   // threads don't update these in parallel
1520   //
1521   // Shared classes are all currently loaded by either the bootstrap or
1522   // internal parallel class loaders, so this will never cause a deadlock
1523   // on a custom class loader lock.

1533     ik->restore_unshareable_info(loader_data, protection_domain, pkg_entry, CHECK_NULL);

1585 void SystemDictionary::quick_resolve(InstanceKlass* klass, ClassLoaderData* loader_data, Handle domain, TRAPS) {
1586   assert(!Universe::is_fully_initialized(), "We can make short cuts only during VM initialization");
1587   assert(klass->is_shared(), "Must be shared class");
1588   if (klass->class_loader_data() != NULL) {
1589     return;
1590   }
1591 
1592   // add super and interfaces first
1593   Klass* super = klass->super();
1594   if (super != NULL && super->class_loader_data() == NULL) {
1595     assert(super->is_instance_klass(), "Super should be instance klass");
1596     quick_resolve(InstanceKlass::cast(super), loader_data, domain, CHECK);
1597   }
1598 
1599   Array<InstanceKlass*>* ifs = klass->local_interfaces();
1600   for (int i = 0; i < ifs->length(); i++) {
1601     InstanceKlass* ik = ifs->at(i);
1602     if (ik->class_loader_data()  == NULL) {
1603       quick_resolve(ik, loader_data, domain, CHECK);
1604     }
1605   }
1606 
1607   if (klass->has_inline_type_fields()) {
1608     for (AllFieldStream fs(klass->fields(), klass->constants()); !fs.done(); fs.next()) {
1609       if (Signature::basic_type(fs.signature()) == T_VALUETYPE) {
1610         if (!fs.access_flags().is_static()) {
1611           Klass* real_k = SystemDictionary::resolve_inline_type_field_or_fail(&fs,
1612             Handle(THREAD, loader_data->class_loader()), domain, true, CHECK);
1613           Klass* k = klass->get_inline_type_field_klass_or_null(fs.index());
1614           assert(real_k == k, "oops, the app has substituted a different version of k!");
1615         } else {
1616           klass->reset_inline_type_field_klass(fs.index());
1617         }
1618       }
1619     }
1620   }
1621 
1622   klass->restore_unshareable_info(loader_data, domain, NULL, THREAD);
1623   load_shared_class_misc(klass, loader_data, CHECK);
1624   Dictionary* dictionary = loader_data->dictionary();
1625   unsigned int hash = dictionary->compute_hash(klass->name());
1626   dictionary->add_klass(hash, klass->name(), klass);
1627   add_to_hierarchy(klass, CHECK);
1628   assert(klass->is_loaded(), "Must be in at least loaded state");
1629 }

1632 InstanceKlass* SystemDictionary::load_instance_class(Symbol* class_name, Handle class_loader, TRAPS) {
1633 
1634   if (class_loader.is_null()) {
1635     ResourceMark rm(THREAD);
1636     PackageEntry* pkg_entry = NULL;
1637     bool search_only_bootloader_append = false;
1638     ClassLoaderData *loader_data = class_loader_data(class_loader);
1639 
1640     // Find the package in the boot loader's package entry table.
1641     TempNewSymbol pkg_name = ClassLoader::package_from_class_name(class_name);
1642     if (pkg_name != NULL) {
1643       pkg_entry = loader_data->packages()->lookup_only(pkg_name);
1644     }
1645 
1646     // Prior to attempting to load the class, enforce the boot loader's
1647     // visibility boundaries.
1648     if (!Universe::is_module_initialized()) {
1649       // During bootstrapping, prior to module initialization, any
1650       // class attempting to be loaded must be checked against the
1651       // java.base packages in the boot loader's PackageEntryTable.
1652       // No class outside of java.base is allowed to be loaded during
1653       // this bootstrapping window.
1654       if (pkg_entry == NULL || pkg_entry->in_unnamed_module()) {
1655         // Class is either in the unnamed package or in
1656         // a named package within the unnamed module.  Either
1657         // case is outside of java.base, do not attempt to
1658         // load the class post java.base definition.  If
1659         // java.base has not been defined, let the class load
1660         // and its package will be checked later by
1661         // ModuleEntryTable::verify_javabase_packages.
1662         if (ModuleEntryTable::javabase_defined()) {
1663           return NULL;
1664         }
1665       } else {
1666         // Check that the class' package is defined within java.base.
1667         ModuleEntry* mod_entry = pkg_entry->module();
1668         Symbol* mod_entry_name = mod_entry->name();
1669         if (mod_entry_name->fast_compare(vmSymbols::java_base()) != 0) {
1670           return NULL;
1671         }
1672       }
1673     } else {
1674       // After the module system has been initialized, check if the class'
1675       // package is in a module defined to the boot loader.
1676       if (pkg_name == NULL || pkg_entry == NULL || pkg_entry->in_unnamed_module()) {
1677         // Class is either in the unnamed package, in a named package
1678         // within a module not defined to the boot loader or in a
1679         // a named package within the unnamed module.  In all cases,
1680         // limit visibility to search for the class only in the boot
1681         // loader's append path.
1682         if (!ClassLoader::has_bootclasspath_append()) {
1683            // If there is no bootclasspath append entry, no need to continue
1684            // searching.
1685            return NULL;
1686         }
1687         search_only_bootloader_append = true;
1688       }
1689     }
1690 
1691     // Prior to bootstrapping's module initialization, never load a class outside
1692     // of the boot loader's module path
1693     assert(Universe::is_module_initialized() ||
1694            !search_only_bootloader_append,
1695            "Attempt to load a class outside of boot loader's module path");
1696 
1697     // Search for classes in the CDS archive.
1698     InstanceKlass* k = NULL;
1699 
1700 #if INCLUDE_CDS
1701     if (UseSharedSpaces)
1702     {
1703       PerfTraceTime vmtimer(ClassLoader::perf_shared_classload_time());
1704       k = load_shared_boot_class(class_name, pkg_entry, THREAD);
1705     }
1706 #endif
1707 
1708     if (k == NULL) {
1709       // Use VM class loader
1710       PerfTraceTime vmtimer(ClassLoader::perf_sys_classload_time());
1711       k = ClassLoader::load_class(class_name, search_only_bootloader_append, CHECK_NULL);
1712     }
1713 
1714     // find_or_define_instance_class may return a different InstanceKlass
1715     if (k != NULL) {
1716       InstanceKlass* defined_k =
1717         find_or_define_instance_class(class_name, class_loader, k, THREAD);
1718       if (!HAS_PENDING_EXCEPTION && defined_k != k) {
1719         // If a parallel capable class loader already defined this class, register 'k' for cleanup.
1720         assert(defined_k != NULL, "Should have a klass if there's no exception");
1721         loader_data->add_to_deallocate_list(k);
1722         k = defined_k;
1723       } else if (HAS_PENDING_EXCEPTION) {
1724         loader_data->add_to_deallocate_list(k);
1725         return NULL;
1726       }
1727     }
1728     return k;
1729   } else {
1730     // Use user specified class loader to load class. Call loadClass operation on class_loader.
1731     ResourceMark rm(THREAD);
1732 
1733     assert(THREAD->is_Java_thread(), "must be a JavaThread");
1734     JavaThread* jt = (JavaThread*) THREAD;
1735 
1736     PerfClassTraceTime vmtimer(ClassLoader::perf_app_classload_time(),
1737                                ClassLoader::perf_app_classload_selftime(),
1738                                ClassLoader::perf_app_classload_count(),
1739                                jt->get_thread_stat()->perf_recursion_counts_addr(),
1740                                jt->get_thread_stat()->perf_timers_addr(),
1741                                PerfClassTraceTime::CLASS_LOAD);
1742 
1743     Handle s = java_lang_String::create_from_symbol(class_name, CHECK_NULL);
1744     // Translate to external class name format, i.e., convert '/' chars to '.'
1745     Handle string = java_lang_String::externalize_classname(s, CHECK_NULL);
1746 
1747     JavaValue result(T_OBJECT);
1748 
1749     InstanceKlass* spec_klass = SystemDictionary::ClassLoader_klass();
1750 
1751     // Call public unsynchronized loadClass(String) directly for all class loaders.
1752     // For parallelCapable class loaders, JDK >=7, loadClass(String, boolean) will
1753     // acquire a class-name based lock rather than the class loader object lock.
1754     // JDK < 7 already acquire the class loader lock in loadClass(String, boolean).
1755     JavaCalls::call_virtual(&result,
1756                             class_loader,
1757                             spec_klass,
1758                             vmSymbols::loadClass_name(),
1759                             vmSymbols::string_class_signature(),
1760                             string,
1761                             CHECK_NULL);
1762 
1763     assert(result.get_type() == T_OBJECT, "just checking");
1764     oop obj = (oop) result.get_jobject();
1765 
1766     // Primitive classes return null since forName() can not be
1767     // used to obtain any of the Class objects representing primitives or void
1768     if ((obj != NULL) && !(java_lang_Class::is_primitive(obj))) {
1769       InstanceKlass* k = InstanceKlass::cast(java_lang_Class::as_Klass(obj));
1770       // For user defined Java class loaders, check that the name returned is
1771       // the same as that requested.  This check is done for the bootstrap
1772       // loader when parsing the class file.
1773       if (class_name == k->name()) {
1774         return k;
1775       }
1776     }
1777     // Class is not found or has the wrong name, return NULL
1778     return NULL;
1779   }

1791 void SystemDictionary::define_instance_class(InstanceKlass* k, TRAPS) {
1792 
1793   HandleMark hm(THREAD);
1794   ClassLoaderData* loader_data = k->class_loader_data();
1795   Handle class_loader_h(THREAD, loader_data->class_loader());
1796 
1797  // for bootstrap and other parallel classloaders don't acquire lock,
1798  // use placeholder token
1799  // If a parallelCapable class loader calls define_instance_class instead of
1800  // find_or_define_instance_class to get here, we have a timing
1801  // hole with systemDictionary updates and check_constraints
1802  if (!class_loader_h.is_null() && !is_parallelCapable(class_loader_h)) {
1803     assert(ObjectSynchronizer::current_thread_holds_lock((JavaThread*)THREAD,
1804          compute_loader_lock_object(class_loader_h, THREAD)),
1805          "define called without lock");
1806   }
1807 
1808   // Check class-loading constraints. Throw exception if violation is detected.
1809   // Grabs and releases SystemDictionary_lock
1810   // The check_constraints/find_class call and update_dictionary sequence
1811   // must be "atomic" for a specific class/classloader pair so we never
1812   // define two different instanceKlasses for that class/classloader pair.
1813   // Existing classloaders will call define_instance_class with the
1814   // classloader lock held
1815   // Parallel classloaders will call find_or_define_instance_class
1816   // which will require a token to perform the define class
1817   Symbol*  name_h = k->name();
1818   Dictionary* dictionary = loader_data->dictionary();
1819   unsigned int d_hash = dictionary->compute_hash(name_h);
1820   check_constraints(d_hash, k, class_loader_h, true, CHECK);
1821 
1822   // Register class just loaded with class loader (placed in ArrayList)
1823   // Note we do this before updating the dictionary, as this can
1824   // fail with an OutOfMemoryError (if it does, we will *not* put this
1825   // class in the dictionary and will not update the class hierarchy).
1826   // JVMTI FollowReferences needs to find the classes this way.
1827   if (k->class_loader() != NULL) {
1828     methodHandle m(THREAD, Universe::loader_addClass_method());
1829     JavaValue result(T_VOID);
1830     JavaCallArguments args(class_loader_h);
1831     args.push_oop(Handle(THREAD, k->java_mirror()));
1832     JavaCalls::call(&result, m, &args, CHECK);
1833   }
1834 
1835   // Add the new class. We need recompile lock during update of CHA.
1836   {
1837     unsigned int p_hash = placeholders()->compute_hash(name_h);
1838     int p_index = placeholders()->hash_to_index(p_hash);
1839 
1840     MutexLocker mu_r(THREAD, Compile_lock);
1841 
1842     // Add to class hierarchy, initialize vtables, and do possible
1843     // deoptimizations.
1844     add_to_hierarchy(k, CHECK); // No exception, but can block
1845 
1846     // Add to systemDictionary - so other classes can see it.
1847     // Grabs and releases SystemDictionary_lock
1848     update_dictionary(d_hash, p_index, p_hash,
1849                       k, class_loader_h, THREAD);
1850   }
1851   k->eager_initialize(THREAD);
1852 
1853   // notify jvmti
1854   if (JvmtiExport::should_post_class_load()) {
1855       assert(THREAD->is_Java_thread(), "thread->is_Java_thread()");
1856       JvmtiExport::post_class_load((JavaThread *) THREAD, k);
1857 
1858   }
1859   post_class_define_event(k, loader_data);
1860 }

1862 // Support parallel classloading
1863 // All parallel class loaders, including bootstrap classloader
1864 // lock a placeholder entry for this class/class_loader pair
1865 // to allow parallel defines of different classes for this class loader
1866 // With AllowParallelDefine flag==true, in case they do not synchronize around
1867 // FindLoadedClass/DefineClass, calls, we check for parallel
1868 // loading for them, wait if a defineClass is in progress
1869 // and return the initial requestor's results
1870 // This flag does not apply to the bootstrap classloader.
1871 // With AllowParallelDefine flag==false, call through to define_instance_class
1872 // which will throw LinkageError: duplicate class definition.
1873 // False is the requested default.
1874 // For better performance, the class loaders should synchronize
1875 // findClass(), i.e. FindLoadedClass/DefineClassIfAbsent or they
1876 // potentially waste time reading and parsing the bytestream.
1877 // Note: VM callers should ensure consistency of k/class_name,class_loader
1878 // Be careful when modifying this code: once you have run
1879 // placeholders()->find_and_add(PlaceholderTable::DEFINE_CLASS),
1880 // you need to find_and_remove it before returning.
1881 // So be careful to not exit with a CHECK_ macro betweeen these calls.
1882 InstanceKlass* SystemDictionary::find_or_define_instance_class(Symbol* class_name, Handle class_loader,
1883                                                                InstanceKlass* k, TRAPS) {
1884 
1885   Symbol*  name_h = k->name(); // passed in class_name may be null
1886   ClassLoaderData* loader_data = class_loader_data(class_loader);
1887   Dictionary* dictionary = loader_data->dictionary();
1888 
1889   unsigned int d_hash = dictionary->compute_hash(name_h);
1890 
1891   // Hold SD lock around find_class and placeholder creation for DEFINE_CLASS
1892   unsigned int p_hash = placeholders()->compute_hash(name_h);
1893   int p_index = placeholders()->hash_to_index(p_hash);
1894   PlaceholderEntry* probe;
1895 
1896   {
1897     MutexLocker mu(THREAD, SystemDictionary_lock);
1898     // First check if class already defined
1899     if (is_parallelDefine(class_loader)) {
1900       InstanceKlass* check = find_class(d_hash, name_h, dictionary);
1901       if (check != NULL) {
1902         return check;
1903       }
1904     }
1905 
1906     // Acquire define token for this class/classloader
1907     probe = placeholders()->find_and_add(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, NULL, THREAD);
1908     // Wait if another thread defining in parallel
1909     // All threads wait - even those that will throw duplicate class: otherwise
1910     // caller is surprised by LinkageError: duplicate, but findLoadedClass fails
1911     // if other thread has not finished updating dictionary
1912     while (probe->definer() != NULL) {
1913       SystemDictionary_lock->wait();
1914     }
1915     // Only special cases allow parallel defines and can use other thread's results
1916     // Other cases fall through, and may run into duplicate defines
1917     // caught by finding an entry in the SystemDictionary
1918     if (is_parallelDefine(class_loader) && (probe->instance_klass() != NULL)) {
1919         placeholders()->find_and_remove(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, THREAD);
1920         SystemDictionary_lock->notify_all();
1921 #ifdef ASSERT
1922         InstanceKlass* check = find_class(d_hash, name_h, dictionary);
1923         assert(check != NULL, "definer missed recording success");
1924 #endif
1925         return probe->instance_klass();
1926     } else {
1927       // This thread will define the class (even if earlier thread tried and had an error)
1928       probe->set_definer(THREAD);
1929     }
1930   }
1931 
1932   define_instance_class(k, THREAD);
1933 
1934   Handle linkage_exception = Handle(); // null handle
1935 
1936   // definer must notify any waiting threads
1937   {
1938     MutexLocker mu(THREAD, SystemDictionary_lock);
1939     PlaceholderEntry* probe = placeholders()->get_entry(p_index, p_hash, name_h, loader_data);
1940     assert(probe != NULL, "DEFINE_CLASS placeholder lost?");
1941     if (probe != NULL) {
1942       if (HAS_PENDING_EXCEPTION) {
1943         linkage_exception = Handle(THREAD,PENDING_EXCEPTION);
1944         CLEAR_PENDING_EXCEPTION;
1945       } else {
1946         probe->set_instance_klass(k);
1947       }
1948       probe->set_definer(NULL);
1949       placeholders()->find_and_remove(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, THREAD);
1950       SystemDictionary_lock->notify_all();
1951     }
1952   }
1953 
1954   // Can't throw exception while holding lock due to rank ordering
1955   if (linkage_exception() != NULL) {
1956     THROW_OOP_(linkage_exception(), NULL); // throws exception and returns
1957   }
1958 
1959   return k;
1960 }

2252   // the offsets to remove the referent and discovered fields from the oop maps,

2310 // found in the OOPSLA'98 paper "Dynamic Class Loading in the Java
2311 // Virtual Machine" by Sheng Liang and Gilad Bracha.  The basic idea is
2312 // that the dictionary needs to maintain a set of contraints that
2313 // must be satisfied by all classes in the dictionary.
2314 // if defining is true, then LinkageError if already in dictionary
2315 // if initiating loader, then ok if InstanceKlass matches existing entry

2317 void SystemDictionary::check_constraints(unsigned int d_hash,
2318                                          InstanceKlass* k,
2319                                          Handle class_loader,

2321                                          TRAPS) {

2324   bool throwException = false;

2334       // If different InstanceKlass - duplicate class definition,
2335       // else - ok, class loaded by a different thread in parallel.
2336       // We should only have found it if it was done loading and ok to use.
2337       // The dictionary only holds instance classes, placeholders

2414 // Try to find a class name using the loader constraints.  The
2415 // loader constraints might know about a class that isn't fully loaded
2416 // yet and these will be ignored.
2417 Klass* SystemDictionary::find_constrained_instance_or_array_klass(
2418                     Symbol* class_name, Handle class_loader, TRAPS) {
2419 
2420   // First see if it has been loaded directly.
2421   // Force the protection domain to be null.  (This removes protection checks.)
2422   Handle no_protection_domain;
2423   Klass* klass = find_instance_or_array_klass(class_name, class_loader,
2424                                               no_protection_domain, CHECK_NULL);
2425   if (klass != NULL)
2426     return klass;
2427 
2428   // Now look to see if it has been loaded elsewhere, and is subject to
2429   // a loader constraint that would require this loader to return the
2430   // klass that is already loaded.
2431   if (Signature::is_array(class_name)) {
2432     // For array classes, their Klass*s are not kept in the
2433     // constraint table. The element Klass*s are.
2434     SignatureStream ss(class_name, false);
2435     int ndims = ss.skip_array_prefix();  // skip all '['s
2436     BasicType t = ss.type();
2437     if (t != T_OBJECT && t != T_VALUETYPE) {
2438       klass = Universe::typeArrayKlassObj(t);
2439     } else {
2440       MutexLocker mu(THREAD, SystemDictionary_lock);
2441       klass = constraints()->find_constrained_klass(ss.as_symbol(), class_loader);
2442     }
2443     // If element class already loaded, allocate array klass
2444     if (klass != NULL) {
2445       klass = klass->array_klass_or_null(ndims);
2446     }
2447   } else {
2448     MutexLocker mu(THREAD, SystemDictionary_lock);
2449     // Non-array classes are easy: simply check the constraint table.
2450     klass = constraints()->find_constrained_klass(class_name, class_loader);
2451   }
2452 
2453   return klass;
2454 }

2456 bool SystemDictionary::add_loader_constraint(Symbol* class_name,
2457                                              Klass* klass_being_linked,
2458                                              Handle class_loader1,
2459                                              Handle class_loader2,
2460                                              Thread* THREAD) {
2461   ClassLoaderData* loader_data1 = class_loader_data(class_loader1);
2462   ClassLoaderData* loader_data2 = class_loader_data(class_loader2);
2463 
2464   Symbol* constraint_name = NULL;
2465 
2466   if (!Signature::is_array(class_name)) {
2467     constraint_name = class_name;
2468   } else {
2469     // For array classes, their Klass*s are not kept in the
2470     // constraint table. The element classes are.
2471     SignatureStream ss(class_name, false);
2472     ss.skip_array_prefix();  // skip all '['s
2473     if (!ss.has_envelope()) {
2474       return true;     // primitive types always pass
2475     }

2481   }
2482 
2483   Dictionary* dictionary1 = loader_data1->dictionary();
2484   unsigned int d_hash1 = dictionary1->compute_hash(constraint_name);
2485 
2486   Dictionary* dictionary2 = loader_data2->dictionary();
2487   unsigned int d_hash2 = dictionary2->compute_hash(constraint_name);
2488 
2489   {
2490     MutexLocker mu_s(THREAD, SystemDictionary_lock);
2491     InstanceKlass* klass1 = find_class(d_hash1, constraint_name, dictionary1);
2492     InstanceKlass* klass2 = find_class(d_hash2, constraint_name, dictionary2);
2493     bool result = constraints()->add_entry(constraint_name, klass1, class_loader1,
2494                                            klass2, class_loader2);
2495 #if INCLUDE_CDS
2496     if (Arguments::is_dumping_archive() && klass_being_linked != NULL &&
2497         !klass_being_linked->is_shared()) {
2498          SystemDictionaryShared::record_linking_constraint(constraint_name,
2499                                      InstanceKlass::cast(klass_being_linked),
2500                                      class_loader1, class_loader2, THREAD);
2501     }
2502 #endif // INCLUDE_CDS
2503     if (Signature::is_array(class_name)) {
2504       constraint_name->decrement_refcount();
2505     }
2506     return result;
2507   }

2581 // Signature constraints ensure that callers and callees agree about
2582 // the meaning of type names in their signatures.  This routine is the
2583 // intake for constraints.  It collects them from several places:
2584 //
2585 //  * LinkResolver::resolve_method (if check_access is true) requires
2586 //    that the resolving class (the caller) and the defining class of
2587 //    the resolved method (the callee) agree on each type in the
2588 //    method's signature.
2589 //
2590 //  * LinkResolver::resolve_interface_method performs exactly the same
2591 //    checks.
2592 //
2593 //  * LinkResolver::resolve_field requires that the constant pool
2594 //    attempting to link to a field agree with the field's defining
2595 //    class about the type of the field signature.
2596 //
2597 //  * klassVtable::initialize_vtable requires that, when a class
2598 //    overrides a vtable entry allocated by a superclass, that the
2599 //    overriding method (i.e., the callee) agree with the superclass
2600 //    on each type in the method's signature.
2601 //
2602 //  * klassItable::initialize_itable requires that, when a class fills
2603 //    in its itables, for each non-abstract method installed in an
2604 //    itable, the method (i.e., the callee) agree with the interface
2605 //    on each type in the method's signature.
2606 //
2607 // All those methods have a boolean (check_access, checkconstraints)
2608 // which turns off the checks.  This is used from specialized contexts
2609 // such as bootstrapping, dumping, and debugging.
2610 //
2611 // No direct constraint is placed between the class and its
2612 // supertypes.  Constraints are only placed along linked relations
2613 // between callers and callees.  When a method overrides or implements
2614 // an abstract method in a supertype (superclass or interface), the
2615 // constraints are placed as if the supertype were the caller to the
2616 // overriding method.  (This works well, since callers to the
2617 // supertype have already established agreement between themselves and
2618 // the supertype.)  As a result of all this, a class can disagree with
2619 // its supertype about the meaning of a type name, as long as that
2620 // class neither calls a relevant method of the supertype, nor is
2621 // called (perhaps via an override) from the supertype.
2622 //
2623 //
2624 // SystemDictionary::check_signature_loaders(sig, klass_being_linked, l1, l2)
2625 //
2626 // Make sure all class components (including arrays) in the given
2627 // signature will be resolved to the same class in both loaders.
2628 // Returns the name of the type that failed a loader constraint check, or
2629 // NULL if no constraint failed.  No exception except OOME is thrown.
2630 // Arrays are not added to the loader constraint table, their elements are.
2631 Symbol* SystemDictionary::check_signature_loaders(Symbol* signature,
2632                                                Klass* klass_being_linked,
2633                                                Handle loader1, Handle loader2,
2634                                                bool is_method, TRAPS)  {
2635   // Nothing to do if loaders are the same.
2636   if (loader1() == loader2()) {
2637     return NULL;
2638   }
2639 
2640   for (SignatureStream ss(signature, is_method); !ss.is_done(); ss.next()) {
2641     if (ss.is_reference()) {
2642       Symbol* sig = ss.as_symbol();
2643       // Note: In the future, if template-like types can take
2644       // arguments, we will want to recognize them and dig out class
2645       // names hiding inside the argument lists.
2646       if (!add_loader_constraint(sig, klass_being_linked, loader1, loader2, THREAD)) {
2647         return sig;
2648       }
2649     }
2650   }
2651   return NULL;
2652 }

2771 // Decide if we can globally cache a lookup of this class, to be returned to any client that asks.
2772 // We must ensure that all class loaders everywhere will reach this class, for any client.
2773 // This is a safe bet for public classes in java.lang, such as Object and String.
2774 // We also include public classes in java.lang.invoke, because they appear frequently in system-level method types.
2775 // Out of an abundance of caution, we do not include any other classes, not even for packages like java.util.
2776 static bool is_always_visible_class(oop mirror) {
2777   Klass* klass = java_lang_Class::as_Klass(mirror);
2778   if (klass->is_objArray_klass()) {
2779     klass = ObjArrayKlass::cast(klass)->bottom_klass(); // check element type
2780   }
2781   if (klass->is_typeArray_klass()) {
2782     return true; // primitive array
2783   }

2790 // Find or construct the Java mirror (java.lang.Class instance) for
2791 // the given field type signature, as interpreted relative to the
2792 // given class loader.  Handles primitives, void, references, arrays,
2793 // and all other reflectable types, except method types.
2794 // N.B.  Code in reflection should use this entry point.
2795 Handle SystemDictionary::find_java_mirror_for_type(Symbol* signature,
2796                                                    Klass* accessing_klass,
2797                                                    Handle class_loader,
2798                                                    Handle protection_domain,
2799                                                    SignatureStream::FailureMode failure_mode,
2800                                                    TRAPS) {
2801   assert(accessing_klass == NULL || (class_loader.is_null() && protection_domain.is_null()),
2802          "one or the other, or perhaps neither");
2803

2805   // and all valid field descriptors are supported.
2806   // Produce the same java.lang.Class that reflection reports.
2807   if (accessing_klass != NULL) {
2808     class_loader      = Handle(THREAD, accessing_klass->class_loader());
2809     protection_domain = Handle(THREAD, accessing_klass->protection_domain());

2812   oop mirror_oop = ss.as_java_mirror(failure_mode, CHECK_NH);

2814     return Handle();  // report failure this way
2815   }
2816   Handle mirror(THREAD, mirror_oop);
2817 
2818   if (accessing_klass != NULL) {
2819     // Check accessibility, emulating ConstantPool::verify_constant_pool_resolve.
2820     Klass* sel_klass = java_lang_Class::as_Klass(mirror());
2821     if (sel_klass != NULL) {
2822       LinkResolver::check_klass_accessibility(accessing_klass, sel_klass, CHECK_NH);
2823     }
2824   }
2825   return mirror;

2829 // Ask Java code to find or construct a java.lang.invoke.MethodType for the given
2830 // signature, as interpreted relative to the given class loader.
2831 // Because of class loader constraints, all method handle usage must be
2832 // consistent with this loader.
2833 Handle SystemDictionary::find_method_handle_type(Symbol* signature,
2834                                                  Klass* accessing_klass,
2835                                                  TRAPS) {
2836   Handle empty;
2837   vmIntrinsics::ID null_iid = vmIntrinsics::_none;  // distinct from all method handle invoker intrinsics
2838   unsigned int hash  = invoke_method_table()->compute_hash(signature, null_iid);
2839   int          index = invoke_method_table()->hash_to_index(hash);
2840   SymbolPropertyEntry* spe = invoke_method_table()->find_entry(index, hash, signature, null_iid);
2841   if (spe != NULL && spe->method_type() != NULL) {
2842     assert(java_lang_invoke_MethodType::is_instance(spe->method_type()), "");
2843     return Handle(THREAD, spe->method_type());
2844   } else if (!THREAD->can_call_java()) {
2845     warning("SystemDictionary::find_method_handle_type called from compiler thread");  // FIXME
2846     return Handle();  // do not attempt from within compiler, unless it was cached
2847   }
2848 
2849   Handle class_loader, protection_domain;
2850   if (accessing_klass != NULL) {
2851     class_loader      = Handle(THREAD, accessing_klass->class_loader());
2852     protection_domain = Handle(THREAD, accessing_klass->protection_domain());
2853   }
2854   bool can_be_cached = true;
2855   int npts = ArgumentCount(signature).size();
2856   objArrayHandle pts = oopFactory::new_objArray_handle(SystemDictionary::Class_klass(), npts, CHECK_(empty));
2857   int arg = 0;
2858   Handle rt; // the return type from the signature
2859   ResourceMark rm(THREAD);
2860   for (SignatureStream ss(signature); !ss.is_done(); ss.next()) {
2861     oop mirror = NULL;
2862     if (can_be_cached) {
2863       // Use neutral class loader to lookup candidate classes to be placed in the cache.
2864       mirror = ss.as_java_mirror(Handle(), Handle(),
2865                                  SignatureStream::ReturnNull, CHECK_(empty));
2866       if (mirror == NULL || (ss.is_reference() && !is_always_visible_class(mirror))) {
2867         // Fall back to accessing_klass context.
2868         can_be_cached = false;
2869       }
2870     }
2871     if (!can_be_cached) {
2872       // Resolve, throwing a real error if it doesn't work.
2873       mirror = ss.as_java_mirror(class_loader, protection_domain,
2874                                  SignatureStream::NCDFError, CHECK_(empty));
2875     }
2876     assert(mirror != NULL, "%s", ss.as_symbol()->as_C_string());
2877     if (ss.at_return_type())
2878       rt = Handle(THREAD, mirror);
2879     else
2880       pts->obj_at_put(arg++, mirror);
2881 
2882     // Check accessibility.
2883     if (!java_lang_Class::is_primitive(mirror) && accessing_klass != NULL) {
2884       Klass* sel_klass = java_lang_Class::as_Klass(mirror);
2885       mirror = NULL;  // safety
2886       // Emulate ConstantPool::verify_constant_pool_resolve.
2887       LinkResolver::check_klass_accessibility(accessing_klass, sel_klass, CHECK_(empty));
2888     }
2889   }
2890   assert(arg == npts, "");
2891 
2892   // call java.lang.invoke.MethodHandleNatives::findMethodHandleType(Class rt, Class[] pts) -> MethodType
2893   JavaCallArguments args(Handle(THREAD, rt()));
2894   args.push_oop(pts);
2895   JavaValue result(T_OBJECT);
2896   JavaCalls::call_static(&result,
2897                          SystemDictionary::MethodHandleNatives_klass(),
2898                          vmSymbols::findMethodHandleType_name(),
2899                          vmSymbols::findMethodHandleType_signature(),
2900                          &args, CHECK_(empty));
2901   Handle method_type(THREAD, (oop) result.get_jobject());
2902 
2903   if (can_be_cached) {
2904     // We can cache this MethodType inside the JVM.
2905     MutexLocker ml(THREAD, SystemDictionary_lock);
2906     spe = invoke_method_table()->find_entry(index, hash, signature, null_iid);
2907     if (spe == NULL)
2908       spe = invoke_method_table()->add_entry(index, hash, signature, null_iid);
2909     if (spe->method_type() == NULL) {
2910       spe->set_method_type(method_type());
2911     }
2912   }
2913 
2914   // report back to the caller with the MethodType
2915   return method_type;
2916 }

2939 // Ask Java code to find or construct a method handle constant.
2940 Handle SystemDictionary::link_method_handle_constant(Klass* caller,
2941                                                      int ref_kind, //e.g., JVM_REF_invokeVirtual
2942                                                      Klass* callee,
2943                                                      Symbol* name,
2944                                                      Symbol* signature,
2945                                                      TRAPS) {
2946   Handle empty;
2947   if (caller == NULL) {
2948     THROW_MSG_(vmSymbols::java_lang_InternalError(), "bad MH constant", empty);
2949   }
2950   Handle name_str      = java_lang_String::create_from_symbol(name,      CHECK_(empty));
2951   Handle signature_str = java_lang_String::create_from_symbol(signature, CHECK_(empty));
2952 
2953   // Put symbolic info from the MH constant into freshly created MemberName and resolve it.
2954   Handle mname = MemberName_klass()->allocate_instance_handle(CHECK_(empty));
2955   java_lang_invoke_MemberName::set_clazz(mname(), callee->java_mirror());
2956   java_lang_invoke_MemberName::set_name (mname(), name_str());
2957   java_lang_invoke_MemberName::set_type (mname(), signature_str());
2958   java_lang_invoke_MemberName::set_flags(mname(), MethodHandles::ref_kind_to_flags(ref_kind));
2959 
2960   if (ref_kind == JVM_REF_invokeVirtual &&
2961       MethodHandles::is_signature_polymorphic_public_name(callee, name)) {
2962     // Skip resolution for public signature polymorphic methods such as
2963     // j.l.i.MethodHandle.invoke()/invokeExact() and those on VarHandle
2964     // They require appendix argument which MemberName resolution doesn't handle.
2965     // There's special logic on JDK side to handle them
2966     // (see MethodHandles.linkMethodHandleConstant() and MethodHandles.findVirtualForMH()).
2967   } else {
2968     MethodHandles::resolve_MemberName(mname, caller, /*speculative_resolve*/false, CHECK_(empty));
2969   }
2970 
2971   // After method/field resolution succeeded, it's safe to resolve MH signature as well.
2972   Handle type = MethodHandles::resolve_MemberName_type(mname, caller, CHECK_(empty));
2973 
2974   // call java.lang.invoke.MethodHandleNatives::linkMethodHandleConstant(Class caller, int refKind, Class callee, String name, Object type) -> MethodHandle
2975   JavaCallArguments args;
2976   args.push_oop(Handle(THREAD, caller->java_mirror()));  // the referring class
2977   args.push_int(ref_kind);
2978   args.push_oop(Handle(THREAD, callee->java_mirror()));  // the target class
2979   args.push_oop(name_str);
2980   args.push_oop(type);
2981   JavaValue result(T_OBJECT);
2982   JavaCalls::call_static(&result,
2983                          SystemDictionary::MethodHandleNatives_klass(),
2984                          vmSymbols::linkMethodHandleConstant_name(),
2985                          vmSymbols::linkMethodHandleConstant_signature(),
2986                          &args, CHECK_(empty));
2987   return Handle(THREAD, (oop) result.get_jobject());
2988 }

2990 // Ask Java to run a bootstrap method, in order to create a dynamic call site
2991 // while linking an invokedynamic op, or compute a constant for Dynamic_info CP entry
2992 // with linkage results being stored back into the bootstrap specifier.
2993 void SystemDictionary::invoke_bootstrap_method(BootstrapInfo& bootstrap_specifier, TRAPS) {
2994   // Resolve the bootstrap specifier, its name, type, and static arguments
2995   bootstrap_specifier.resolve_bsm(CHECK);
2996 
2997   // This should not happen.  JDK code should take care of that.
2998   if (bootstrap_specifier.caller() == NULL || bootstrap_specifier.type_arg().is_null()) {
2999     THROW_MSG(vmSymbols::java_lang_InternalError(), "Invalid bootstrap method invocation with no caller or type argument");
3000   }
3001 
3002   bool is_indy = bootstrap_specifier.is_method_call();
3003   objArrayHandle appendix_box;
3004   if (is_indy) {
3005     // Some method calls may require an appendix argument.  Arrange to receive it.
3006     appendix_box = oopFactory::new_objArray_handle(SystemDictionary::Object_klass(), 1, CHECK);
3007     assert(appendix_box->obj_at(0) == NULL, "");
3008   }

3010   // call condy: java.lang.invoke.MethodHandleNatives::linkDynamicConstant(caller, condy_index, bsm, type, info)
3011   //       indy: java.lang.invoke.MethodHandleNatives::linkCallSite(caller, indy_index, bsm, name, mtype, info, &appendix)
3012   JavaCallArguments args;
3013   args.push_oop(Handle(THREAD, bootstrap_specifier.caller_mirror()));
3014   args.push_int(bootstrap_specifier.bss_index());

3017   args.push_oop(bootstrap_specifier.type_arg());

3020     args.push_oop(appendix_box);
3021   }
3022   JavaValue result(T_OBJECT);
3023   JavaCalls::call_static(&result,
3024                          SystemDictionary::MethodHandleNatives_klass(),
3025                          is_indy ? vmSymbols::linkCallSite_name() : vmSymbols::linkDynamicConstant_name(),
3026                          is_indy ? vmSymbols::linkCallSite_signature() : vmSymbols::linkDynamicConstant_signature(),
3027                          &args, CHECK);
3028 
3029   Handle value(THREAD, (oop) result.get_jobject());

3033                                                 bootstrap_specifier.caller(),
3034                                                 appendix_box,
3035                                                 &appendix, CHECK);

3040   }
