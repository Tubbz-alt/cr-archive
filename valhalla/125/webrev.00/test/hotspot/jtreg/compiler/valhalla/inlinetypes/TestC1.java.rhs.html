 34 /*
 35  * @test
 36  * @key randomness
 37  * @summary Various tests that are specific for C1.
 38  * @modules java.base/jdk.experimental.value
 39  * @library /testlibrary /test/lib /compiler/whitebox /
 40  * @requires os.simpleArch == "x64"
 41  * @compile -XDallowWithFieldOperator TestC1.java
 42  * @run driver ClassFileInstaller sun.hotspot.WhiteBox jdk.test.lib.Platform
 43  * @run main/othervm/timeout=300 -Xbootclasspath/a:. -XX:+IgnoreUnrecognizedVMOptions -XX:+UnlockDiagnosticVMOptions
 44  *                               -XX:+UnlockExperimentalVMOptions -XX:+WhiteBoxAPI
 45  *                               compiler.valhalla.inlinetypes.InlineTypeTest
 46  *                               compiler.valhalla.inlinetypes.TestC1
 47  */
 48 public class TestC1 extends InlineTypeTest {
 49     public static final int C1 = COMP_LEVEL_SIMPLE;
 50     public static final int C2 = COMP_LEVEL_FULL_OPTIMIZATION;
 51 
 52     public static void main(String[] args) throws Throwable {
 53         TestC1 test = new TestC1();
 54         test.run(args, MyValue1.class, MyValue2.class, MyValue2Inline.class, MyValue3.class, MyValue3Inline.class);
 55     }
 56 
 57     @Override
 58     public int getNumScenarios() {
 59         return 5;
 60     }
 61 
 62     @Override
 63     public String[] getVMParameters(int scenario) {
 64         switch (scenario) {
 65         case 0: return new String[] { // C1 only
 66                 "-XX:TieredStopAtLevel=1",
 67                 "-XX:+TieredCompilation",
 68             };
 69         case 1: return new String[] { // C2 only. (Make sure the tests are correctly written)
 70                 "-XX:TieredStopAtLevel=4",
 71                 "-XX:-TieredCompilation",
 72             };
 73         case 2: return new String[] { // interpreter only
 74                 "-Xint",
 75             };
 76         case 3: return new String[] {
 77                 // Xcomp Only C1.
 78                 "-XX:TieredStopAtLevel=1",
 79                 "-XX:+TieredCompilation",
 80                 "-Xcomp",
 81             };
 82         case 4: return new String[] {
 83                 // Xcomp Only C2.
 84                 "-XX:TieredStopAtLevel=4",
 85                 "-XX:-TieredCompilation",
 86                 "-Xcomp",
 87             };
 88         }
 89         return null;
 90     }
 91 
 92     // JDK-8229799
 93     @Test(compLevel=C1)
 94     public long test1(Object a, Object b, long n) {
 95         long r;
 96         n += (a == b) ? 0x5678123456781234L : 0x1234567812345678L;
 97         n -= 1;
 98         return n;
 99     }
100 
101     @DontCompile
102     public void test1_verifier(boolean warmup) {
103         MyValue1 v1 = MyValue1.createWithFieldsInline(rI, rL);
104         MyValue1 v2 = MyValue1.createWithFieldsInline(rI, rL+1);
105         long r1 = test1(v1, v1, 1);
106         long r2 = test1(v1, v2, 1);
107         Asserts.assertEQ(r1, 0x5678123456781234L);
108         Asserts.assertEQ(r2, 0x1234567812345678L);
109     }
110 
111     static inline class SimpleValue2 {
112         final int value;
113         SimpleValue2(int value) {
114             this.value = value;
115         }
116     }
117 
118     // JDK-8231961
119     // Test that the value numbering optimization does not remove
120     // the second load from the buffered array element.
121     @Test(compLevel=C1)
122     public int test2(SimpleValue2[] array) {
123         return array[0].value + array[0].value;
124     }
125 
126     @DontCompile
127     public void test2_verifier(boolean warmup) {
128         SimpleValue2[] array = new SimpleValue2[1];
129         array[0] = new SimpleValue2(rI);
130         int result = test2(array);
131         Asserts.assertEQ(result, 2*rI);
132     }
133 }
