 133 void BlockListBuilder::set_entries(int osr_bci) {
 134   // generate start blocks
 135   BlockBegin* std_entry = make_block_at(0, NULL);
 136   if (scope()->caller() == NULL) {
 137     std_entry->set(BlockBegin::std_entry_flag);
 138   }
 139   if (osr_bci != -1) {
 140     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 141     osr_entry->set(BlockBegin::osr_entry_flag);
 142   }
 143 
 144   // generate exception entry blocks
 145   XHandlers* list = xhandlers();
 146   const int n = list->length();
 147   for (int i = 0; i < n; i++) {
 148     XHandler* h = list->handler_at(i);
 149     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 150     entry->set(BlockBegin::exception_entry_flag);
 151     h->set_entry_block(entry);
 152   }
 153 }

 156 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 157   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 158 
 159   BlockBegin* block = _bci2block->at(cur_bci);
 160   if (block == NULL) {
 161     block = new BlockBegin(cur_bci);
 162     block->init_stores_to_locals(method()->max_locals());
 163     _bci2block->at_put(cur_bci, block);
 164     _blocks.append(block);
 165 
 166     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 167   }
 168 
 169   if (predecessor != NULL) {
 170     if (block->is_set(BlockBegin::exception_entry_flag)) {
 171       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 172     }
 173 
 174     predecessor->add_successor(block);
 175     block->increment_total_preds();
 176   }
 177 
 178   return block;
 179 }

 182 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 183   current->stores_to_locals().set_bit(local);

 185 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 186   store_one(current, local);
 187   store_one(current, local + 1);

 191 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 192   // Draws edges from a block to its exception handlers
 193   XHandlers* list = xhandlers();
 194   const int n = list->length();
 195 
 196   for (int i = 0; i < n; i++) {
 197     XHandler* h = list->handler_at(i);
 198 
 199     if (h->covers(cur_bci)) {
 200       BlockBegin* entry = h->entry_block();
 201       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 202       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 203 
 204       // add each exception handler only once
 205       if (!current->is_successor(entry)) {
 206         current->add_successor(entry);
 207         entry->increment_total_preds();
 208       }
 209 
 210       // stop when reaching catchall
 211       if (h->catch_type() == 0) break;
 212     }
 213   }
 214 }

 216 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 217   // start a new block after jsr-bytecode and link this block into cfg
 218   make_block_at(next_bci, current);
 219 
 220   // start a new block at the subroutine entry at mark it with special flag
 221   BlockBegin* sr_block = make_block_at(sr_bci, current);
 222   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 223     sr_block->set(BlockBegin::subroutine_entry_flag);
 224   }
 225 }

 228 void BlockListBuilder::set_leaders() {
 229   bool has_xhandlers = xhandlers()->has_handlers();
 230   BlockBegin* current = NULL;
 231 
 232   // The information which bci starts a new block simplifies the analysis
 233   // Without it, backward branches could jump to a bci where no block was created
 234   // during bytecode iteration. This would require the creation of a new block at the
 235   // branch target and a modification of the successor lists.
 236   const BitMap& bci_block_start = method()->bci_block_start();
 237 
 238   ciBytecodeStream s(method());
 239   while (s.next() != ciBytecodeStream::EOBC()) {
 240     int cur_bci = s.cur_bci();
 241 
 242     if (bci_block_start.at(cur_bci)) {
 243       current = make_block_at(cur_bci, current);
 244     }
 245     assert(current != NULL, "must have current block");
 246 
 247     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 248       handle_exceptions(current, cur_bci);
 249     }
 250 
 251     switch (s.cur_bc()) {
 252       // track stores to local variables for selective creation of phi functions
 253       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 254       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 255       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 256       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 257       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 258       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 259       case Bytecodes::_istore_0: store_one(current, 0); break;
 260       case Bytecodes::_istore_1: store_one(current, 1); break;
 261       case Bytecodes::_istore_2: store_one(current, 2); break;
 262       case Bytecodes::_istore_3: store_one(current, 3); break;
 263       case Bytecodes::_lstore_0: store_two(current, 0); break;
 264       case Bytecodes::_lstore_1: store_two(current, 1); break;
 265       case Bytecodes::_lstore_2: store_two(current, 2); break;
 266       case Bytecodes::_lstore_3: store_two(current, 3); break;
 267       case Bytecodes::_fstore_0: store_one(current, 0); break;
 268       case Bytecodes::_fstore_1: store_one(current, 1); break;
 269       case Bytecodes::_fstore_2: store_one(current, 2); break;
 270       case Bytecodes::_fstore_3: store_one(current, 3); break;
 271       case Bytecodes::_dstore_0: store_two(current, 0); break;
 272       case Bytecodes::_dstore_1: store_two(current, 1); break;
 273       case Bytecodes::_dstore_2: store_two(current, 2); break;
 274       case Bytecodes::_dstore_3: store_two(current, 3); break;
 275       case Bytecodes::_astore_0: store_one(current, 0); break;
 276       case Bytecodes::_astore_1: store_one(current, 1); break;
 277       case Bytecodes::_astore_2: store_one(current, 2); break;
 278       case Bytecodes::_astore_3: store_one(current, 3); break;
 279 
 280       // track bytecodes that affect the control flow
 281       case Bytecodes::_athrow:  // fall through
 282       case Bytecodes::_ret:     // fall through
 283       case Bytecodes::_ireturn: // fall through
 284       case Bytecodes::_lreturn: // fall through
 285       case Bytecodes::_freturn: // fall through
 286       case Bytecodes::_dreturn: // fall through
 287       case Bytecodes::_areturn: // fall through
 288       case Bytecodes::_return:
 289         current = NULL;
 290         break;
 291 
 292       case Bytecodes::_ifeq:      // fall through
 293       case Bytecodes::_ifne:      // fall through
 294       case Bytecodes::_iflt:      // fall through
 295       case Bytecodes::_ifge:      // fall through
 296       case Bytecodes::_ifgt:      // fall through
 297       case Bytecodes::_ifle:      // fall through
 298       case Bytecodes::_if_icmpeq: // fall through
 299       case Bytecodes::_if_icmpne: // fall through
 300       case Bytecodes::_if_icmplt: // fall through
 301       case Bytecodes::_if_icmpge: // fall through
 302       case Bytecodes::_if_icmpgt: // fall through
 303       case Bytecodes::_if_icmple: // fall through
 304       case Bytecodes::_if_acmpeq: // fall through
 305       case Bytecodes::_if_acmpne: // fall through
 306       case Bytecodes::_ifnull:    // fall through
 307       case Bytecodes::_ifnonnull:
 308         make_block_at(s.next_bci(), current);
 309         make_block_at(s.get_dest(), current);
 310         current = NULL;
 311         break;
 312 
 313       case Bytecodes::_goto:
 314         make_block_at(s.get_dest(), current);
 315         current = NULL;
 316         break;
 317 
 318       case Bytecodes::_goto_w:
 319         make_block_at(s.get_far_dest(), current);
 320         current = NULL;
 321         break;
 322 
 323       case Bytecodes::_jsr:
 324         handle_jsr(current, s.get_dest(), s.next_bci());
 325         current = NULL;
 326         break;
 327 
 328       case Bytecodes::_jsr_w:
 329         handle_jsr(current, s.get_far_dest(), s.next_bci());
 330         current = NULL;
 331         break;
 332 
 333       case Bytecodes::_tableswitch: {
 334         // set block for each case
 335         Bytecode_tableswitch sw(&s);
 336         int l = sw.length();
 337         for (int i = 0; i < l; i++) {
 338           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 339         }
 340         make_block_at(cur_bci + sw.default_offset(), current);
 341         current = NULL;
 342         break;
 343       }
 344 
 345       case Bytecodes::_lookupswitch: {
 346         // set block for each case
 347         Bytecode_lookupswitch sw(&s);
 348         int l = sw.number_of_pairs();
 349         for (int i = 0; i < l; i++) {
 350           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 351         }
 352         make_block_at(cur_bci + sw.default_offset(), current);
 353         current = NULL;
 354         break;
 355       }
 356 
 357       default:
 358         break;
 359     }
 360   }
 361 }

 364 void BlockListBuilder::mark_loops() {
 365   ResourceMark rm;
 366 
 367   _active.initialize(BlockBegin::number_of_blocks());
 368   _visited.initialize(BlockBegin::number_of_blocks());
 369   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 370   _next_loop_index = 0;
 371   _next_block_number = _blocks.length();
 372 
 373   // recursively iterate the control flow graph
 374   mark_loops(_bci2block->at(0), false);
 375   assert(_next_block_number >= 0, "invalid block numbers");
 376 
 377   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 378   _active.resize(0);
 379   _visited.resize(0);
 380 }

 382 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 383   if (block->is_set(BlockBegin::exception_entry_flag)) {
 384     // exception edges may look like loops but don't mark them as such
 385     // since it screws up block ordering.
 386     return;
 387   }
 388   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 389     block->set(BlockBegin::parser_loop_header_flag);
 390 
 391     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 392     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 393     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 394     if (_next_loop_index < 31) _next_loop_index++;
 395   } else {
 396     // block already marked as loop header
 397     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 398   }
 399 }

 401 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 402   int block_id = block->block_id();
 403 
 404   if (_visited.at(block_id)) {
 405     if (_active.at(block_id)) {
 406       // reached block via backward branch
 407       make_loop_header(block);
 408     }
 409     // return cached loop information for this block
 410     return _loop_map.at(block_id);
 411   }
 412 
 413   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 414     in_subroutine = true;
 415   }
 416 
 417   // set active and visited bits before successors are processed
 418   _visited.set_bit(block_id);
 419   _active.set_bit(block_id);
 420 
 421   intptr_t loop_state = 0;
 422   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 423     // recursively process all successors
 424     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 425   }
 426 
 427   // clear active-bit after all successors are processed
 428   _active.clear_bit(block_id);
 429 
 430   // reverse-post-order numbering of all blocks
 431   block->set_depth_first_number(_next_block_number);
 432   _next_block_number--;
 433 
 434   if (loop_state != 0 || in_subroutine ) {
 435     // block is contained at least in one loop, so phi functions are necessary
 436     // phi functions are also necessary for all locals stored in a subroutine
 437     scope()->requires_phi_function().set_union(block->stores_to_locals());
 438   }
 439 
 440   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 441     int header_loop_state = _loop_map.at(block_id);
 442     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 443 
 444     // If the highest bit is set (i.e. when integer value is negative), the method
 445     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 446     if (header_loop_state >= 0) {
 447       clear_bits(loop_state, header_loop_state);
 448     }
 449   }
 450 
 451   // cache and return loop information for this block
 452   _loop_map.at_put(block_id, loop_state);
 453   return loop_state;
 454 }

 463 void BlockListBuilder::print() {
 464   tty->print("----- initial block list of BlockListBuilder for method ");
 465   method()->print_short_name();
 466   tty->cr();
 467 
 468   // better readability if blocks are sorted in processing order
 469   _blocks.sort(compare_depth_first);
 470 
 471   for (int i = 0; i < _blocks.length(); i++) {
 472     BlockBegin* cur = _blocks.at(i);
 473     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 474 
 475     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 476     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 477     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 478     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 479     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 480 
 481     if (cur->number_of_sux() > 0) {
 482       tty->print("    sux: ");
 483       for (int j = 0; j < cur->number_of_sux(); j++) {
 484         BlockBegin* sux = cur->sux_at(j);
 485         tty->print("B%d ", sux->block_id());
 486       }
 487     }
 488     tty->cr();
 489   }
 490 }

 495 // A simple growable array of Values indexed by ciFields
 496 class FieldBuffer: public CompilationResourceObj {
 497  private:
 498   GrowableArray<Value> _values;
 499 
 500  public:
 501   FieldBuffer() {}
 502 
 503   void kill() {
 504     _values.trunc_to(0);
 505   }
 506 
 507   Value at(ciField* field) {
 508     assert(field->holder()->is_loaded(), "must be a loaded field");
 509     int offset = field->offset();
 510     if (offset < _values.length()) {
 511       return _values.at(offset);
 512     } else {
 513       return NULL;
 514     }
 515   }
 516 
 517   void at_put(ciField* field, Value value) {
 518     assert(field->holder()->is_loaded(), "must be a loaded field");
 519     int offset = field->offset();
 520     _values.at_put_grow(offset, value, NULL);
 521   }
 522 
 523 };

 526 // MemoryBuffer is fairly simple model of the current state of memory.
 527 // It partitions memory into several pieces.  The first piece is
 528 // generic memory where little is known about the owner of the memory.
 529 // This is conceptually represented by the tuple <O, F, V> which says
 530 // that the field F of object O has value V.  This is flattened so
 531 // that F is represented by the offset of the field and the parallel
 532 // arrays _objects and _values are used for O and V.  Loads of O.F can
 533 // simply use V.  Newly allocated objects are kept in a separate list
 534 // along with a parallel array for each object which represents the
 535 // current value of its fields.  Stores of the default value to fields
 536 // which have never been stored to before are eliminated since they
 537 // are redundant.  Once newly allocated objects are stored into
 538 // another object or they are passed out of the current compile they
 539 // are treated like generic memory.
 540 
 541 class MemoryBuffer: public CompilationResourceObj {
 542  private:
 543   FieldBuffer                 _values;
 544   GrowableArray<Value>        _objects;
 545   GrowableArray<Value>        _newobjects;
 546   GrowableArray<FieldBuffer*> _fields;
 547 
 548  public:
 549   MemoryBuffer() {}
 550 
 551   StoreField* store(StoreField* st) {
 552     if (!EliminateFieldAccess) {
 553       return st;
 554     }
 555 
 556     Value object = st->obj();
 557     Value value = st->value();
 558     ciField* field = st->field();
 559     if (field->holder()->is_loaded()) {
 560       int offset = field->offset();
 561       int index = _newobjects.find(object);
 562       if (index != -1) {
 563         // newly allocated object with no other stores performed on this field
 564         FieldBuffer* buf = _fields.at(index);
 565         if (buf->at(field) == NULL && is_default_value(value)) {
 566 #ifndef PRODUCT
 567           if (PrintIRDuringConstruction && Verbose) {
 568             tty->print_cr("Eliminated store for object %d:", index);
 569             st->print_line();
 570           }
 571 #endif
 572           return NULL;
 573         } else {
 574           buf->at_put(field, value);
 575         }
 576       } else {
 577         _objects.at_put_grow(offset, object, NULL);
 578         _values.at_put(field, value);
 579       }
 580 
 581       store_value(value);
 582     } else {
 583       // if we held onto field names we could alias based on names but
 584       // we don't know what's being stored to so kill it all.
 585       kill();
 586     }
 587     return st;
 588   }
 589 
 590 
 591   // return true if this value correspond to the default value of a field.
 592   bool is_default_value(Value value) {
 593     Constant* con = value->as_Constant();
 594     if (con) {
 595       switch (con->type()->tag()) {
 596         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 597         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 598         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 599         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 600         case objectTag: return con->type() == objectNull;
 601         default:  ShouldNotReachHere();
 602       }
 603     }
 604     return false;
 605   }
 606 
 607 
 608   // return either the actual value of a load or the load itself
 609   Value load(LoadField* load) {
 610     if (!EliminateFieldAccess) {
 611       return load;
 612     }
 613 
 614     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 615 #ifdef IA32
 616       if (UseSSE < 2) {
 617         // can't skip load since value might get rounded as a side effect
 618         return load;
 619       }
 620 #else
 621       Unimplemented();
 622 #endif // IA32
 623     }
 624 
 625     ciField* field = load->field();
 626     Value object   = load->obj();
 627     if (field->holder()->is_loaded() && !field->is_volatile()) {
 628       int offset = field->offset();
 629       Value result = NULL;
 630       int index = _newobjects.find(object);
 631       if (index != -1) {
 632         result = _fields.at(index)->at(field);
 633       } else if (_objects.at_grow(offset, NULL) == object) {
 634         result = _values.at(field);
 635       }
 636       if (result != NULL) {
 637 #ifndef PRODUCT
 638         if (PrintIRDuringConstruction && Verbose) {
 639           tty->print_cr("Eliminated load: ");
 640           load->print_line();
 641         }
 642 #endif
 643         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 644         return result;
 645       }
 646     }
 647     return load;
 648   }
 649 
 650   // Record this newly allocated object
 651   void new_instance(NewInstance* object) {
 652     int index = _newobjects.length();
 653     _newobjects.append(object);
 654     if (_fields.at_grow(index, NULL) == NULL) {
 655       _fields.at_put(index, new FieldBuffer());
 656     } else {
 657       _fields.at(index)->kill();
 658     }
 659   }
 660 
 661   // Record this newly allocated object
 662   void new_instance(NewInlineTypeInstance* object) {
 663     int index = _newobjects.length();
 664     _newobjects.append(object);
 665     if (_fields.at_grow(index, NULL) == NULL) {
 666       _fields.at_put(index, new FieldBuffer());
 667     } else {
 668       _fields.at(index)->kill();
 669     }
 670   }
 671 
 672   void store_value(Value value) {
 673     int index = _newobjects.find(value);
 674     if (index != -1) {
 675       // stored a newly allocated object into another object.
 676       // Assume we've lost track of it as separate slice of memory.
 677       // We could do better by keeping track of whether individual
 678       // fields could alias each other.
 679       _newobjects.remove_at(index);
 680       // pull out the field info and store it at the end up the list
 681       // of field info list to be reused later.
 682       _fields.append(_fields.at(index));
 683       _fields.remove_at(index);
 684     }
 685   }
 686 
 687   void kill() {
 688     _newobjects.trunc_to(0);
 689     _objects.trunc_to(0);
 690     _values.kill();
 691   }
 692 };

 733 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 734   if (parsing_jsr()) {
 735     // It is necessary to clone all blocks associated with a
 736     // subroutine, including those for exception handlers in the scope
 737     // of the method containing the jsr (because those exception
 738     // handlers may contain ret instructions in some cases).
 739     BlockBegin* block = bci2block()->at(bci);
 740     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 741       BlockBegin* new_block = new BlockBegin(block->bci());
 742       if (PrintInitialBlockList) {
 743         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 744                       block->block_id(), block->bci(), new_block->block_id());
 745       }
 746       // copy data from cloned blocked
 747       new_block->set_depth_first_number(block->depth_first_number());
 748       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 749       // Preserve certain flags for assertion checking
 750       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 751       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 752 
 753       // copy was_visited_flag to allow early detection of bailouts
 754       // if a block that is used in a jsr has already been visited before,
 755       // it is shared between the normal control flow and a subroutine
 756       // BlockBegin::try_merge returns false when the flag is set, this leads
 757       // to a compilation bailout
 758       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 759 
 760       bci2block()->at_put(bci, new_block);
 761       block = new_block;
 762     }
 763     return block;
 764   } else {
 765     return bci2block()->at(bci);
 766   }
 767 }

 824 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 825   assert(worklist->top() == top, "");
 826   // sort block descending into work list
 827   const int dfn = top->depth_first_number();
 828   assert(dfn != -1, "unknown depth first number");
 829   int i = worklist->length()-2;
 830   while (i >= 0) {
 831     BlockBegin* b = worklist->at(i);
 832     if (b->depth_first_number() < dfn) {
 833       worklist->at_put(i+1, b);
 834     } else {
 835       break;
 836     }
 837     i --;
 838   }
 839   if (i >= -1) worklist->at_put(i + 1, top);
 840 }

 897 void GraphBuilder::load_constant() {
 898   ciConstant con = stream()->get_constant();
 899   if (con.basic_type() == T_ILLEGAL) {
 900     // FIXME: an unresolved Dynamic constant can get here,
 901     // and that should not terminate the whole compilation.
 902     BAILOUT("could not resolve a constant");
 903   } else {
 904     ValueType* t = illegalType;
 905     ValueStack* patch_state = NULL;
 906     switch (con.basic_type()) {
 907       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 908       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 909       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 910       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 911       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 912       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 913       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 914       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 915       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 916       case T_OBJECT :
 917        {
 918         ciObject* obj = con.as_object();
 919         if (!obj->is_loaded()
 920             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 921           // A Class, MethodType, MethodHandle, or String.
 922           // Unloaded condy nodes show up as T_ILLEGAL, above.
 923           patch_state = copy_state_before();
 924           t = new ObjectConstant(obj);
 925         } else {
 926           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 927           // result, which might turn out to be an array.
 928           if (obj->is_null_object())
 929             t = objectNull;
 930           else if (obj->is_array())
 931             t = new ArrayConstant(obj->as_array());
 932           else
 933             t = new InstanceConstant(obj->as_instance());
 934         }
 935         break;
 936        }
 937       default       : ShouldNotReachHere();
 938     }
 939     Value x;
 940     if (patch_state != NULL) {
 941       x = new Constant(t, patch_state);
 942     } else {
 943       x = new Constant(t);
 944     }
 945     push(t, append(x));
 946   }
 947 }

 963 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 964   if (parsing_jsr()) {
 965     // We need to do additional tracking of the location of the return
 966     // address for jsrs since we don't handle arbitrary jsr/ret
 967     // constructs. Here we are figuring out in which circumstances we
 968     // need to bail out.
 969     if (x->type()->is_address()) {
 970       scope_data()->set_jsr_return_address_local(index);
 971 
 972       // Also check parent jsrs (if any) at this time to see whether
 973       // they are using this local. We don't handle skipping over a
 974       // ret.
 975       for (ScopeData* cur_scope_data = scope_data()->parent();
 976            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 977            cur_scope_data = cur_scope_data->parent()) {
 978         if (cur_scope_data->jsr_return_address_local() == index) {
 979           BAILOUT("subroutine overwrites return address from previous subroutine");
 980         }
 981       }
 982     } else if (index == scope_data()->jsr_return_address_local()) {
 983       scope_data()->set_jsr_return_address_local(-1);
 984     }
 985   }
 986 
 987   x->set_local_index(index);
 988   state->store_local(index, round_fp(x));
 989 }

 992 void GraphBuilder::load_indexed(BasicType type) {
 993   // In case of in block code motion in range check elimination
 994   ValueStack* state_before = NULL;
 995   int array_idx = state()->stack_size() - 2;
 996   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 997     // Save the entire state and re-execute on deopt when accessing flattened arrays
 998     state_before = copy_state_before();
 999     state_before->set_should_reexecute(true);
1000   } else {
1001     state_before = copy_state_indexed_access();
1002   }
1003   compilation()->set_has_access_indexed(true);
1004   Value index = ipop();
1005   Value array = apop();
1006   Value length = NULL;
1007   if (CSEArrayLength ||
1008       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1009       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1010     length = append(new ArrayLength(array, state_before));
1011   }
1012 
1013   LoadIndexed* load_indexed = NULL;
1014   Instruction* result = NULL;
1015   if (array->is_loaded_flattened_array()) {
1016     ciType* array_type = array->declared_type();
1017     ciInlineKlass* elem_klass = array_type->as_flat_array_klass()->element_klass()->as_inline_klass();
1018     NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(elem_klass, state_before, false);
1019     _memory->new_instance(new_instance);
1020     apush(append_split(new_instance));
1021     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1022     load_indexed->set_vt(new_instance);
1023   } else {
1024     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1025   }
1026   if (profile_array_accesses() && is_reference_type(type)) {
1027     compilation()->set_would_profile(true);
1028     load_indexed->set_should_profile(true);
1029     load_indexed->set_profiled_method(method());
1030     load_indexed->set_profiled_bci(bci());
1031   }
1032   result = append(load_indexed);
1033   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1034   if (!array->is_loaded_flattened_array()) {
1035     push(as_ValueType(type), result);
1036   }
1037 }

1090   switch (code) {
1091     case Bytecodes::_pop:
1092       { state()->raw_pop();
1093       }
1094       break;
1095     case Bytecodes::_pop2:
1096       { state()->raw_pop();
1097         state()->raw_pop();
1098       }
1099       break;
1100     case Bytecodes::_dup:
1101       { Value w = state()->raw_pop();
1102         state()->raw_push(w);
1103         state()->raw_push(w);
1104       }
1105       break;

1107       { Value w1 = state()->raw_pop();
1108         Value w2 = state()->raw_pop();

1116         Value w2 = state()->raw_pop();
1117         Value w3 = state()->raw_pop();

1119         state()->raw_push(w3);
1120         state()->raw_push(w2);
1121         state()->raw_push(w1);
1122       }
1123       break;
1124     case Bytecodes::_dup2:

1126         Value w2 = state()->raw_pop();

1191 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1192   Value s = ipop();
1193   Value x = pop(type);
1194   // try to simplify
1195   // Note: This code should go into the canonicalizer as soon as it can
1196   //       can handle canonicalized forms that contain more than one node.
1197   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1198     // pattern: x >>> s
1199     IntConstant* s1 = s->type()->as_IntConstant();
1200     if (s1 != NULL) {
1201       // pattern: x >>> s1, with s1 constant
1202       ShiftOp* l = x->as_ShiftOp();
1203       if (l != NULL && l->op() == Bytecodes::_ishl) {
1204         // pattern: (a << b) >>> s1
1205         IntConstant* s0 = l->y()->type()->as_IntConstant();
1206         if (s0 != NULL) {
1207           // pattern: (a << s0) >>> s1
1208           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1209           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1210           if (s0c == s1c) {
1211             if (s0c == 0) {
1212               // pattern: (a << 0) >>> 0 => simplify to: a
1213               ipush(l->x());
1214             } else {
1215               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1216               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1217               const int m = (1 << (BitsPerInt - s0c)) - 1;
1218               Value s = append(new Constant(new IntConstant(m)));
1219               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1220             }
1221             return;
1222           }
1223         }
1224       }
1225     }
1226   }
1227   // could not simplify
1228   push(type, append(new ShiftOp(code, x, s)));
1229 }

1276 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1277   BlockBegin* tsux = block_at(stream()->get_dest());
1278   BlockBegin* fsux = block_at(stream()->next_bci());
1279   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1280 
1281   bool subst_check = false;
1282   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne) &&
1283       method() != ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature())) {
1284     // If current method is ValueBootstrapMethods::isSubstitutable(),
1285     // compile the acmp as a regular pointer comparison otherwise we
1286     // could call ValueBootstrapMethods::isSubstitutable() back
1287     ValueType* left_vt = x->type();
1288     ValueType* right_vt = y->type();
1289     if (left_vt->is_object()) {
1290       assert(right_vt->is_object(), "must be");
1291       ciKlass* left_klass = x->as_loaded_klass_or_null();
1292       ciKlass* right_klass = y->as_loaded_klass_or_null();
1293 
1294       if (left_klass == NULL || right_klass == NULL) {
1295         // The klass is still unloaded, or came from a Phi node. Go slow case;
1296         subst_check = true;
1297       } else if (left_klass->can_be_inline_klass() || right_klass->can_be_inline_klass()) {
1298         // Either operand may be a value object, but we're not sure. Go slow case;
1299         subst_check = true;
1300       } else {
1301         // No need to do substitutability check
1302       }
1303     }
1304   }
1305 
1306   // In case of loop invariant code motion or predicate insertion
1307   // before the body of a loop the state is needed
1308   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1309 
1310   assert(i->as_Goto() == NULL ||
1311          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1312          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1313          "safepoint state of Goto returned by canonicalizer incorrect");
1314 
1315   if (is_profiling()) {
1316     If* if_node = i->as_If();
1317     if (if_node != NULL) {
1318       // Note that we'd collect profile data in this method if we wanted it.
1319       compilation()->set_would_profile(true);
1320       // At level 2 we need the proper bci to count backedges
1321       if_node->set_profiled_bci(bci());
1322       if (profile_branches()) {
1323         // Successors can be rotated by the canonicalizer, check for this case.
1324         if_node->set_profiled_method(method());
1325         if_node->set_should_profile(true);
1326         if (if_node->tsux() == fsux) {
1327           if_node->set_swapped(true);
1328         }
1329       }
1330       return;
1331     }
1332 
1333     // Check if this If was reduced to Goto.
1334     Goto *goto_node = i->as_Goto();
1335     if (goto_node != NULL) {
1336       compilation()->set_would_profile(true);
1337       goto_node->set_profiled_bci(bci());
1338       if (profile_branches()) {
1339         goto_node->set_profiled_method(method());
1340         goto_node->set_should_profile(true);
1341         // Find out which successor is used.
1342         if (goto_node->default_sux() == tsux) {
1343           goto_node->set_direction(Goto::taken);
1344         } else if (goto_node->default_sux() == fsux) {
1345           goto_node->set_direction(Goto::not_taken);
1346         } else {
1347           ShouldNotReachHere();
1348         }
1349       }
1350       return;
1351     }
1352   }
1353 }

1412 void GraphBuilder::table_switch() {
1413   Bytecode_tableswitch sw(stream());
1414   const int l = sw.length();
1415   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1416     // total of 2 successors => use If instead of switch
1417     // Note: This code should go into the canonicalizer as soon as it can
1418     //       can handle canonicalized forms that contain more than one node.
1419     Value key = append(new Constant(new IntConstant(sw.low_key())));
1420     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1421     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1422     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1423     // In case of loop invariant code motion or predicate insertion
1424     // before the body of a loop the state is needed
1425     ValueStack* state_before = copy_state_if_bb(is_bb);
1426     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1427   } else {
1428     // collect successors
1429     BlockList* sux = new BlockList(l + 1, NULL);
1430     int i;
1431     bool has_bb = false;
1432     for (i = 0; i < l; i++) {
1433       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1434       if (sw.dest_offset_at(i) < 0) has_bb = true;
1435     }
1436     // add default successor
1437     if (sw.default_offset() < 0) has_bb = true;
1438     sux->at_put(i, block_at(bci() + sw.default_offset()));
1439     // In case of loop invariant code motion or predicate insertion
1440     // before the body of a loop the state is needed
1441     ValueStack* state_before = copy_state_if_bb(has_bb);
1442     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1443 #ifdef ASSERT
1444     if (res->as_Goto()) {
1445       for (i = 0; i < l; i++) {
1446         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1447           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1448         }
1449       }
1450     }
1451 #endif
1452   }
1453 }

1456 void GraphBuilder::lookup_switch() {
1457   Bytecode_lookupswitch sw(stream());
1458   const int l = sw.number_of_pairs();
1459   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1460     // total of 2 successors => use If instead of switch
1461     // Note: This code should go into the canonicalizer as soon as it can
1462     //       can handle canonicalized forms that contain more than one node.
1463     // simplify to If
1464     LookupswitchPair pair = sw.pair_at(0);
1465     Value key = append(new Constant(new IntConstant(pair.match())));
1466     BlockBegin* tsux = block_at(bci() + pair.offset());
1467     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1468     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1469     // In case of loop invariant code motion or predicate insertion
1470     // before the body of a loop the state is needed
1471     ValueStack* state_before = copy_state_if_bb(is_bb);;
1472     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1473   } else {
1474     // collect successors & keys
1475     BlockList* sux = new BlockList(l + 1, NULL);
1476     intArray* keys = new intArray(l, l, 0);
1477     int i;
1478     bool has_bb = false;
1479     for (i = 0; i < l; i++) {
1480       LookupswitchPair pair = sw.pair_at(i);
1481       if (pair.offset() < 0) has_bb = true;
1482       sux->at_put(i, block_at(bci() + pair.offset()));
1483       keys->at_put(i, pair.match());
1484     }
1485     // add default successor
1486     if (sw.default_offset() < 0) has_bb = true;
1487     sux->at_put(i, block_at(bci() + sw.default_offset()));
1488     // In case of loop invariant code motion or predicate insertion
1489     // before the body of a loop the state is needed
1490     ValueStack* state_before = copy_state_if_bb(has_bb);
1491     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1492 #ifdef ASSERT
1493     if (res->as_Goto()) {
1494       for (i = 0; i < l; i++) {
1495         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1496           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1497         }
1498       }
1499     }
1500 #endif
1501   }
1502 }

1504 void GraphBuilder::call_register_finalizer() {
1505   // If the receiver requires finalization then emit code to perform
1506   // the registration on return.
1507 
1508   // Gather some type information about the receiver
1509   Value receiver = state()->local_at(0);
1510   assert(receiver != NULL, "must have a receiver");
1511   ciType* declared_type = receiver->declared_type();
1512   ciType* exact_type = receiver->exact_type();
1513   if (exact_type == NULL &&
1514       receiver->as_Local() &&
1515       receiver->as_Local()->java_index() == 0) {
1516     ciInstanceKlass* ik = compilation()->method()->holder();
1517     if (ik->is_final()) {
1518       exact_type = ik;
1519     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1520       // test class is leaf class
1521       compilation()->dependency_recorder()->assert_leaf_type(ik);
1522       exact_type = ik;
1523     } else {
1524       declared_type = ik;
1525     }
1526   }
1527 
1528   // see if we know statically that registration isn't required
1529   bool needs_check = true;
1530   if (exact_type != NULL) {
1531     needs_check = exact_type->as_instance_klass()->has_finalizer();
1532   } else if (declared_type != NULL) {
1533     ciInstanceKlass* ik = declared_type->as_instance_klass();
1534     if (!Dependencies::has_finalizable_subclass(ik)) {
1535       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1536       needs_check = false;
1537     }
1538   }
1539 
1540   if (needs_check) {
1541     // Perform the registration of finalizable objects.
1542     ValueStack* state_before = copy_state_for_exception();
1543     load_local(objectType, 0);
1544     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1545                                state()->pop_arguments(1),
1546                                true, state_before, true));
1547   }
1548 }

1551 void GraphBuilder::method_return(Value x, bool ignore_return) {
1552   if (RegisterFinalizersAtInit &&
1553       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1554     call_register_finalizer();
1555   }
1556 
1557   // The conditions for a memory barrier are described in Parse::do_exits().
1558   bool need_mem_bar = false;
1559   if (method()->is_object_constructor() &&
1560        (scope()->wrote_final() ||
1561          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1562          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1563     need_mem_bar = true;
1564   }
1565 
1566   BasicType bt = method()->return_type()->basic_type();
1567   switch (bt) {
1568     case T_BYTE:
1569     {
1570       Value shift = append(new Constant(new IntConstant(24)));
1571       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1572       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1573       break;
1574     }
1575     case T_SHORT:
1576     {
1577       Value shift = append(new Constant(new IntConstant(16)));
1578       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1579       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1580       break;
1581     }
1582     case T_CHAR:
1583     {
1584       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1585       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1586       break;
1587     }
1588     case T_BOOLEAN:
1589     {
1590       Value mask = append(new Constant(new IntConstant(1)));
1591       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1592       break;
1593     }
1594     default:
1595       break;
1596   }
1597 
1598   // Check to see whether we are inlining. If so, Return
1599   // instructions become Gotos to the continuation point.
1600   if (continuation() != NULL) {
1601 
1602     int invoke_bci = state()->caller_state()->bci();
1603 
1604     if (x != NULL  && !ignore_return) {
1605       ciMethod* caller = state()->scope()->caller()->method();
1606       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1607       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1608         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1609         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1610             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1611           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1612         }
1613       }
1614     }
1615 
1616     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1617 
1618     if (compilation()->env()->dtrace_method_probes()) {
1619       // Report exit from inline methods
1620       Values* args = new Values(1);
1621       args->push(append(new Constant(new MethodConstant(method()))));
1622       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1623     }
1624 
1625     // If the inlined method is synchronized, the monitor must be
1626     // released before we jump to the continuation block.
1627     if (method()->is_synchronized()) {
1628       assert(state()->locks_size() == 1, "receiver must be locked here");
1629       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1630     }
1631 
1632     if (need_mem_bar) {
1633       append(new MemBar(lir_membar_storestore));
1634     }
1635 
1636     // State at end of inlined method is the state of the caller
1637     // without the method parameters on stack, including the
1638     // return value, if any, of the inlined method on operand stack.
1639     set_state(state()->caller_state()->copy_for_parsing());
1640     if (x != NULL) {
1641       if (!ignore_return) {
1642         state()->push(x->type(), x);
1643       }
1644       if (profile_return() && x->type()->is_object_kind()) {
1645         ciMethod* caller = state()->scope()->method();
1646         profile_return_type(x, method(), caller, invoke_bci);
1647       }
1648     }
1649     Goto* goto_callee = new Goto(continuation(), false);
1650 
1651     // See whether this is the first return; if so, store off some
1652     // of the state for later examination
1653     if (num_returns() == 0) {
1654       set_inline_cleanup_info();
1655     }
1656 
1657     // The current bci() is in the wrong scope, so use the bci() of
1658     // the continuation point.
1659     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1660     incr_num_returns();
1661     return;
1662   }
1663 
1664   state()->truncate_stack(0);
1665   if (method()->is_synchronized()) {
1666     // perform the unlocking before exiting the method
1667     Value receiver;
1668     if (!method()->is_static()) {
1669       receiver = _initial_state->local_at(0);
1670     } else {
1671       receiver = append(new Constant(new ClassConstant(method()->holder())));
1672     }
1673     append_split(new MonitorExit(receiver, state()->unlock()));
1674   }
1675 
1676   if (need_mem_bar) {
1677       append(new MemBar(lir_membar_storestore));
1678   }
1679 
1680   assert(!ignore_return, "Ignoring return value works only for inlining");
1681   append(new Return(x));
1682 }

1751   }
1752 
1753   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1754     scope()->set_wrote_final();
1755   }
1756 
1757   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1758     scope()->set_wrote_fields();
1759     if (field->is_volatile()) {
1760       scope()->set_wrote_volatile();
1761     }
1762   }
1763 
1764   const int offset = !needs_patching ? field->offset() : -1;
1765   switch (code) {
1766     case Bytecodes::_getstatic: {
1767       // check for compile-time constants, i.e., initialized static final fields
1768       Value constant = NULL;
1769       if (field->is_static_constant() && !PatchALot) {
1770         ciConstant field_value = field->constant_value();
1771         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1772                "stable static w/ default value shouldn't be a constant");
1773         constant = make_constant(field_value, field);
1774       }
1775       if (constant != NULL) {
1776         push(type, append(constant));
1777       } else {
1778         if (state_before == NULL) {
1779           state_before = copy_state_for_exception();
1780         }
1781         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1782                                         state_before, needs_patching);
1783         push(type, append(load_field));
1784       }
1785       break;
1786     }
1787     case Bytecodes::_putstatic: {
1788       Value val = pop(type);
1789       val->set_escaped();
1790       if (state_before == NULL) {
1791         state_before = copy_state_for_exception();
1792       }
1793       if (field_type == T_BOOLEAN) {
1794         Value mask = append(new Constant(new IntConstant(1)));
1795         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1796       }
1797       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1798       break;
1799     }
1800     case Bytecodes::_getfield: {
1801       // Check for compile-time constants, i.e., trusted final non-static fields.
1802       Value constant = NULL;
1803       if (state_before == NULL && field->is_flattened()) {
1804         // Save the entire state and re-execute on deopt when accessing flattened fields
1805         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1806         state_before = copy_state_before();
1807       }
1808       obj = apop();
1809       ObjectType* obj_type = obj->type()->as_ObjectType();
1810       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1811         ciObject* const_oop = obj_type->constant_value();
1812         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1813           ciConstant field_value = field->constant_value_of(const_oop);
1814           if (field_value.is_valid()) {
1815             if (field->signature()->is_Q_signature() && field_value.is_null_or_zero()) {
1816               // Non-flattened inline type field. Replace null by the default value.
1817               constant = new Constant(new InstanceConstant(field->type()->as_inline_klass()->default_instance()));
1818             } else {
1819               constant = make_constant(field_value, field);
1820             }
1821             // For CallSite objects add a dependency for invalidation of the optimization.
1822             if (field->is_call_site_target()) {
1823               ciCallSite* call_site = const_oop->as_call_site();
1824               if (!call_site->is_fully_initialized_constant_call_site()) {
1825                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1826                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1827               }
1828             }
1829           }
1830         }
1831       }
1832       if (constant != NULL) {
1833         push(type, append(constant));
1834       } else {
1835         if (state_before == NULL) {
1836           state_before = copy_state_for_exception();
1837         }
1838         if (!field->is_flattened()) {
1839           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1840           Value replacement = !needs_patching ? _memory->load(load) : load;
1841           if (replacement != load) {
1842             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1843             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1844             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1845             BasicType bt = field->type()->basic_type();
1846             switch (bt) {
1847             case T_BOOLEAN:
1848             case T_BYTE:
1849               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1850               break;
1851             case T_CHAR:
1852               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1853               break;
1854             case T_SHORT:
1855               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1856               break;
1857             default:
1858               break;
1859             }
1860             push(type, replacement);
1861           } else {
1862             push(type, append(load));
1863           }
1864         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1865           assert(field->type()->is_inlinetype(), "Sanity check");
1866           ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1867           int flattening_offset = field->offset() - inline_klass->first_field_offset();
1868           assert(field->type()->is_inlinetype(), "Sanity check");
1869           scope()->set_wrote_final();
1870           scope()->set_wrote_fields();
1871           NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(inline_klass, state_before, false);
1872           _memory->new_instance(new_instance);
1873           apush(append_split(new_instance));
1874           copy_inline_content(inline_klass, obj, field->offset(), new_instance, inline_klass->first_field_offset(),
1875                        state_before, needs_patching);
1876         }
1877       }
1878       break;
1879     }
1880     case Bytecodes::_withfield:
1881     case Bytecodes::_putfield: {
1882       Value val = pop(type);
1883       val->set_escaped();
1884       obj = apop();
1885       if (state_before == NULL) {
1886         state_before = copy_state_for_exception();
1887       }
1888       if (field->type()->basic_type() == T_BOOLEAN) {
1889         Value mask = append(new Constant(new IntConstant(1)));
1890         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1891       }
1892 
1893       if (!field->is_flattened()) {
1894         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1895         if (!needs_patching) store = _memory->store(store);
1896         if (store != NULL) {
1897           append(store);
1898         }
1899       } else {
1900         assert(field->type()->is_inlinetype(), "Sanity check");
1901         ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1902         int flattening_offset = field->offset() - inline_klass->first_field_offset();
1903         copy_inline_content(inline_klass, val, inline_klass->first_field_offset(), obj, field->offset(),
1904                    state_before, needs_patching);
1905       }
1906       break;
1907     }
1908     default:
1909       ShouldNotReachHere();
1910       break;
1911   }
1912 }

1914 // Baseline version of withfield, allocate every time
1915 void GraphBuilder::withfield(int field_index)
1916 {
1917   bool will_link;
1918   ciField* field_modify = stream()->get_field(will_link);
1919   ciInstanceKlass* holder = field_modify->holder();
1920   BasicType field_type = field_modify->type()->basic_type();
1921   ValueType* type = as_ValueType(field_type);
1922 
1923   // call will_link again to determine if the field is valid.
1924   const bool needs_patching = !holder->is_loaded() ||
1925                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1926                               PatchALot;
1927 
1928 
1929   scope()->set_wrote_final();
1930   scope()->set_wrote_fields();
1931 
1932   const int offset = !needs_patching ? field_modify->offset() : -1;
1933 
1934   if (!holder->is_loaded()
1935       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1936       ) {
1937     ValueStack* state_before = copy_state_before();
1938     Value val = pop(type);
1939     Value obj = apop();
1940     apush(append_split(new WithField(state_before)));
1941     return;
1942   }
1943   ValueStack* state_before = copy_state_before();
1944 
1945   Value val = pop(type);
1946   Value obj = apop();
1947 
1948   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1949     int astore_index;
1950     ciBytecodeStream s(method());
1951     s.force_bci(bci());
1952     s.next();
1953     switch (s.cur_bc()) {
1954     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1955     case Bytecodes::_astore_0:  astore_index = 0; break;
1956     case Bytecodes::_astore_1:  astore_index = 1; break;
1957     case Bytecodes::_astore_2:  astore_index = 2; break;
1958     case Bytecodes::_astore_3:  astore_index = 3; break;
1959     default: astore_index = -1;
1960     }
1961 
1962     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1963       // We have a sequence like this, where we load a value object from a local slot,
1964       // and overwrite the same local slot with a modified copy of the inline object.
1965       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1966       //      astore 9
1967       //      ...
1968       //      iload_0
1969       //      aload 9
1970       //      swap
1971       //      withfield #7 // Field x:I
1972       //      astore 9
1973       // If this object was created by defaultvalue, and has not escaped, and is not stored
1974       // in any other local slots, we can effectively treat the withfield/astore
1975       // sequence as a single putfield bytecode.
1976       push(objectType, obj);
1977       push(type, val);
1978       access_field(Bytecodes::_withfield);
1979       stream()->next(); // skip the next astore/astore_n bytecode.
1980       return;
1981     }
1982   }
1983 
1984   assert(holder->is_inlinetype(), "must be an inline klass");
1985   // Save the entire state and re-execute on deopt when executing withfield
1986   state_before->set_should_reexecute(true);
1987   NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(holder->as_inline_klass(), state_before, false);
1988   _memory->new_instance(new_instance);
1989   apush(append_split(new_instance));
1990 
1991   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1992     ciField* field = holder->nonstatic_field_at(i);
1993     int off = field->offset();
1994 
1995     if (field->offset() != offset) {
1996       if (field->is_flattened()) {
1997         assert(field->type()->is_inlinetype(), "Sanity check");
1998         assert(field->type()->is_inlinetype(), "Only inline types can be flattened");
1999         ciInlineKlass* vk = field->type()->as_inline_klass();
2000         copy_inline_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
2001       } else {
2002         // Only load those fields who are not modified
2003         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2004         Value replacement = append(load);
2005         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2006         append(store);
2007       }
2008     }
2009   }
2010 
2011   // Field to modify
2012   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2013     Value mask = append(new Constant(new IntConstant(1)));
2014     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2015   }
2016   if (field_modify->is_flattened()) {
2017     assert(field_modify->type()->is_inlinetype(), "Only inline types can be flattened");
2018     ciInlineKlass* vk = field_modify->type()->as_inline_klass();
2019     copy_inline_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2020   } else {
2021     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2022     append(store);
2023   }
2024 }

2032 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2033   int n = 0;
2034   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2035   start = has_receiver ? 1 : 0;
2036   if (profile_arguments()) {
2037     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2038     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2039       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2040     }
2041   }
2042   // If we are inlining then we need to collect arguments to profile parameters for the target
2043   if (profile_parameters() && target != NULL) {
2044     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2045       // The receiver is profiled on method entry so it's included in
2046       // the number of parameters but here we're only interested in
2047       // actual arguments.
2048       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2049     }
2050   }
2051   if (n > 0) {
2052     return new Values(n);
2053   }
2054   return NULL;
2055 }

2088   ciSignature* declared_signature = NULL;

2090   ciKlass*              holder = stream()->get_declared_method_holder();

2092   assert(declared_signature != NULL, "cannot be null");
2093   assert(will_link == target->is_loaded(), "");
2094 
2095   ciInstanceKlass* klass = target->holder();

2097 
2098   // check if CHA possible: if so, change the code to invoke_special

2100   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);

2106                 log->identify(target),
2107                 Bytecodes::name(code));

2112           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2113                                                  calling_klass;

2115       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);

2124   // convert them directly to an invokespecial or invokestatic.
2125   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2126     switch (bc_raw) {
2127     case Bytecodes::_invokevirtual:
2128       code = Bytecodes::_invokespecial;
2129       break;
2130     case Bytecodes::_invokehandle:
2131       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2132       break;
2133     default:
2134       break;
2135     }
2136   } else {

2138       assert(!will_link, "should come here only for unlinked call");
2139       code = Bytecodes::_invokespecial;

2166     ciInstanceKlass* receiver_klass = NULL;

2174           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {

2177       }

2181             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2182           receiver_klass = (ciInstanceKlass*) type;

2184             // Insert a dependency on this type since
2185             // find_monomorphic_target may assume it's already done.

2188           }

2203         receiver_klass->is_subtype_of(actual_recv) &&

2211       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2212     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {

2215       // may be able bind this invoke directly to the implementing
2216       // klass but we need both a dependence on the single interface
2217       // and on the method we bind to.  Additionally since all we know
2218       // about the receiver type is the it's supposed to implement the
2219       // interface we have to insert a check that it's the class we

2221       // they are roughly equivalent to Object.

2223       // equal to the number of implementors for target->holder() so
2224       // if number of implementors of target->holder() == 1 then
2225       // number of implementors for decl_interface is 0 or 1. If
2226       // it's 0 then no class implements decl_interface and there's
2227       // no point in inlining.

2230       if (singleton != NULL &&
2231           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2232         assert(singleton != declared_interface, "not a unique implementor");
2233         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2234         if (cha_monomorphic_target != NULL) {

2236             // If CHA is able to bind this invoke then update the class
2237             // to match that class, otherwise klass will refer to the
2238             // interface.

2243             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2244             c->set_incompatible_class_change_check();
2245             c->set_direct_compare(klass->is_final());
2246             // pass the result of the checkcast so that the compiler has
2247             // more accurate type info in the inlinee
2248             better_receiver = append_split(c);
2249           } else {
2250             cha_monomorphic_target = NULL; // subtype check against Object is useless
2251           }
2252         }
2253       }
2254     }
2255   }
2256 
2257   if (cha_monomorphic_target != NULL) {
2258     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2259     assert(!cha_monomorphic_target->is_abstract(), "");
2260     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2261       // If we inlined because CHA revealed only a single target method,
2262       // then we are dependent on that target method not getting overridden
2263       // by dynamic class loading.  Be sure to test the "static" receiver
2264       // dest_method here, as opposed to the actual receiver, which may
2265       // falsely lead us to believe that the receiver is final or private.
2266       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2267     }
2268     code = Bytecodes::_invokespecial;
2269   }
2270 
2271   // check if we could do inlining
2272   if (!PatchALot && Inline && target->is_loaded() &&
2273       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2274       && !patch_for_appendix) {
2275     // callee is known => check if we have static binding
2276     if (code == Bytecodes::_invokestatic  ||
2277         code == Bytecodes::_invokespecial ||
2278         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2279         code == Bytecodes::_invokedynamic) {
2280       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2281       // static binding => check if callee is ok
2282       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2283 
2284       CHECK_BAILOUT();
2285       clear_inline_bailout();
2286 
2287       if (success) {
2288         // Register dependence if JVMTI has either breakpoint
2289         // setting or hotswapping of methods capabilities since they may
2290         // cause deoptimization.
2291         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2292           dependency_recorder()->assert_evol_method(inline_target);
2293         }
2294         return;
2295       }
2296     } else {
2297       print_inlining(target, "no static binding", /*success*/ false);
2298     }
2299   } else {
2300     print_inlining(target, "not inlineable", /*success*/ false);
2301   }
2302 
2303   // If we attempted an inline which did not succeed because of a
2304   // bailout during construction of the callee graph, the entire
2305   // compilation has to be aborted. This is fairly rare and currently
2306   // seems to only occur for jasm-generated classes which contain
2307   // jsr/ret pairs which are not associated with finally clauses and
2308   // do not have exception handlers in the containing method, and are
2309   // therefore not caught early enough to abort the inlining without
2310   // corrupting the graph. (We currently bail out with a non-empty
2311   // stack at a ret in these situations.)
2312   CHECK_BAILOUT();
2313 
2314   // inlining not successful => standard invoke
2315   ValueType* result_type = as_ValueType(declared_signature->return_type());
2316   ValueStack* state_before = copy_state_exhandling();
2317 
2318   // The bytecode (code) might change in this method so we are checking this very late.
2319   const bool has_receiver =
2320     code == Bytecodes::_invokespecial   ||
2321     code == Bytecodes::_invokevirtual   ||
2322     code == Bytecodes::_invokeinterface;
2323   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2324   Value recv = has_receiver ? apop() : NULL;
2325   int vtable_index = Method::invalid_vtable_index;
2326 
2327   // A null check is required here (when there is a receiver) for any of the following cases
2328   // - invokespecial, always need a null check.
2329   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2330   //   and require null checking. If the target is loaded a null check is emitted here.
2331   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2332   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2333   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2334   //   potentially fail, and can't have the null check before the resolution.)
2335   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2336   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2337   //
2338   // Normal invokevirtual will perform the null check during lookup
2339 
2340   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2341       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2342 
2343   if (need_null_check) {
2344     if (recv != NULL) {
2345       null_check(recv);
2346     }
2347 
2348     if (is_profiling()) {
2349       // Note that we'd collect profile data in this method if we wanted it.
2350       compilation()->set_would_profile(true);
2351 
2352       if (profile_calls()) {
2353         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2354         ciKlass* target_klass = NULL;
2355         if (cha_monomorphic_target != NULL) {
2356           target_klass = cha_monomorphic_target->holder();
2357         } else if (exact_target != NULL) {
2358           target_klass = exact_target->holder();
2359         }
2360         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2361       }
2362     }
2363   }
2364 
2365   if (recv != NULL) {
2366     recv->set_escaped();
2367   }
2368   for (int i=0; i<args->length(); i++) {
2369     args->at(0)->set_escaped();
2370   }
2371 
2372   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2373                               declared_signature->return_type()->is_inlinetype());
2374   // push result
2375   append_split(result);
2376 
2377   if (result_type != voidType) {
2378     if (method()->is_strict()) {
2379       push(result_type, round_fp(result));
2380     } else {
2381       push(result_type, result);
2382     }
2383   }
2384   if (profile_return() && result_type->is_object_kind()) {
2385     profile_return_type(result, target);
2386   }
2387 }

2570 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2571   Canonicalizer canon(compilation(), instr, bci);
2572   Instruction* i1 = canon.canonical();
2573   if (i1->is_linked() || !i1->can_be_linked()) {
2574     // Canonicalizer returned an instruction which was already
2575     // appended so simply return it.
2576     return i1;
2577   }
2578 
2579   if (UseLocalValueNumbering) {
2580     // Lookup the instruction in the ValueMap and add it to the map if
2581     // it's not found.
2582     Instruction* i2 = vmap()->find_insert(i1);
2583     if (i2 != i1) {
2584       // found an entry in the value map, so just return it.
2585       assert(i2->is_linked(), "should already be linked");
2586       return i2;
2587     }
2588     ValueNumberingEffects vne(vmap());
2589     i1->visit(&vne);
2590   }
2591 
2592   // i1 was not eliminated => append it
2593   assert(i1->next() == NULL, "shouldn't already be linked");
2594   _last = _last->set_next(i1, canon.bci());
2595 
2596   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2597     // set the bailout state but complete normal processing.  We
2598     // might do a little more work before noticing the bailout so we
2599     // want processing to continue normally until it's noticed.
2600     bailout("Method and/or inlining is too large");
2601   }
2602 
2603 #ifndef PRODUCT
2604   if (PrintIRDuringConstruction) {
2605     InstructionPrinter ip;
2606     ip.print_line(i1);
2607     if (Verbose) {
2608       state()->print();
2609     }
2610   }
2611 #endif
2612 
2613   // save state after modification of operand stack for StateSplit instructions
2614   StateSplit* s = i1->as_StateSplit();
2615   if (s != NULL) {
2616     if (EliminateFieldAccess) {
2617       Intrinsic* intrinsic = s->as_Intrinsic();
2618       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2619         _memory->kill();
2620       }
2621     }
2622     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2623   }
2624 
2625   // set up exception handlers for this instruction if necessary
2626   if (i1->can_trap()) {
2627     i1->set_exception_handlers(handle_exception(i1));
2628     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2629   }
2630   return i1;
2631 }

2665 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2666   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2667     assert(instruction->exception_state() == NULL
2668            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2669            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2670            "exception_state should be of exception kind");
2671     return new XHandlers();
2672   }
2673 
2674   XHandlers*  exception_handlers = new XHandlers();
2675   ScopeData*  cur_scope_data = scope_data();
2676   ValueStack* cur_state = instruction->state_before();
2677   ValueStack* prev_state = NULL;
2678   int scope_count = 0;
2679 
2680   assert(cur_state != NULL, "state_before must be set");
2681   do {
2682     int cur_bci = cur_state->bci();
2683     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2684     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2685 
2686     // join with all potential exception handlers
2687     XHandlers* list = cur_scope_data->xhandlers();
2688     const int n = list->length();
2689     for (int i = 0; i < n; i++) {
2690       XHandler* h = list->handler_at(i);
2691       if (h->covers(cur_bci)) {
2692         // h is a potential exception handler => join it
2693         compilation()->set_has_exception_handlers(true);
2694 
2695         BlockBegin* entry = h->entry_block();
2696         if (entry == block()) {
2697           // It's acceptable for an exception handler to cover itself
2698           // but we don't handle that in the parser currently.  It's
2699           // very rare so we bailout instead of trying to handle it.
2700           BAILOUT_("exception handler covers itself", exception_handlers);
2701         }
2702         assert(entry->bci() == h->handler_bci(), "must match");
2703         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2704 
2705         // previously this was a BAILOUT, but this is not necessary
2706         // now because asynchronous exceptions are not handled this way.
2707         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2708 
2709         // xhandler start with an empty expression stack
2710         if (cur_state->stack_size() != 0) {
2711           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2712         }
2713         if (instruction->exception_state() == NULL) {
2714           instruction->set_exception_state(cur_state);
2715         }
2716 
2717         // Note: Usually this join must work. However, very
2718         // complicated jsr-ret structures where we don't ret from
2719         // the subroutine can cause the objects on the monitor
2720         // stacks to not match because blocks can be parsed twice.
2721         // The only test case we've seen so far which exhibits this
2722         // problem is caught by the infinite recursion test in
2723         // GraphBuilder::jsr() if the join doesn't work.
2724         if (!entry->try_merge(cur_state)) {
2725           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2726         }
2727 
2728         // add current state for correct handling of phi functions at begin of xhandler
2729         int phi_operand = entry->add_exception_state(cur_state);
2730 
2731         // add entry to the list of xhandlers of this block
2732         _block->add_exception_handler(entry);
2733 
2734         // add back-edge from xhandler entry to this block
2735         if (!entry->is_predecessor(_block)) {
2736           entry->add_predecessor(_block);
2737         }
2738 
2739         // clone XHandler because phi_operand and scope_count can not be shared
2740         XHandler* new_xhandler = new XHandler(h);
2741         new_xhandler->set_phi_operand(phi_operand);
2742         new_xhandler->set_scope_count(scope_count);
2743         exception_handlers->append(new_xhandler);
2744 
2745         // fill in exception handler subgraph lazily
2746         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2747         cur_scope_data->add_to_work_list(entry);
2748 
2749         // stop when reaching catchall
2750         if (h->catch_type() == 0) {
2751           return exception_handlers;
2752         }
2753       }
2754     }
2755 
2756     if (exception_handlers->length() == 0) {
2757       // This scope and all callees do not handle exceptions, so the local
2758       // variables of this scope are not needed. However, the scope itself is
2759       // required for a correct exception stack trace -> clear out the locals.
2760       if (_compilation->env()->should_retain_local_variables()) {
2761         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2762       } else {
2763         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2764       }
2765       if (prev_state != NULL) {
2766         prev_state->set_caller_state(cur_state);
2767       }
2768       if (instruction->exception_state() == NULL) {
2769         instruction->set_exception_state(cur_state);
2770       }
2771     }
2772 
2773     // Set up iteration for next time.
2774     // If parsing a jsr, do not grab exception handlers from the
2775     // parent scopes for this method (already got them, and they
2776     // needed to be cloned)
2777 
2778     while (cur_scope_data->parsing_jsr()) {
2779       cur_scope_data = cur_scope_data->parent();
2780     }
2781 
2782     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2783     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2784 
2785     prev_state = cur_state;
2786     cur_state = cur_state->caller_state();
2787     cur_scope_data = cur_scope_data->parent();
2788     scope_count++;
2789   } while (cur_scope_data != NULL);
2790 
2791   return exception_handlers;
2792 }

2813 Value PhiSimplifier::simplify(Value v) {
2814   Phi* phi = v->as_Phi();
2815 
2816   if (phi == NULL) {
2817     // no phi function
2818     return v;
2819   } else if (v->has_subst()) {
2820     // already substituted; subst can be phi itself -> simplify
2821     return simplify(v->subst());
2822   } else if (phi->is_set(Phi::cannot_simplify)) {
2823     // already tried to simplify phi before
2824     return phi;
2825   } else if (phi->is_set(Phi::visited)) {
2826     // break cycles in phi functions
2827     return phi;
2828   } else if (phi->type()->is_illegal()) {
2829     // illegal phi functions are ignored anyway
2830     return phi;
2831 
2832   } else {
2833     // mark phi function as processed to break cycles in phi functions
2834     phi->set(Phi::visited);
2835 
2836     // simplify x = [y, x] and x = [y, y] to y
2837     Value subst = NULL;
2838     int opd_count = phi->operand_count();
2839     for (int i = 0; i < opd_count; i++) {

3234             // we setup the OSR edge leading into this block so that
3235             // Phis get setup correctly.

3238             b->clear(BlockBegin::osr_entry_flag);
3239           }
3240           b->set(BlockBegin::was_visited_flag);
3241           connect_to_end(b);
3242         }
3243       }
3244     }
3245   } while (!bailed_out() && !scope_data()->is_work_list_empty());

3372 void GraphBuilder::setup_osr_entry_block() {
3373   assert(compilation()->is_osr_compile(), "only for osrs");
3374 
3375   int osr_bci = compilation()->osr_bci();
3376   ciBytecodeStream s(method());
3377   s.reset_to_bci(osr_bci);
3378   s.next();
3379   scope_data()->set_stream(&s);
3380 
3381   // create a new block to be the osr setup code
3382   _osr_entry = new BlockBegin(osr_bci);
3383   _osr_entry->set(BlockBegin::osr_entry_flag);
3384   _osr_entry->set_depth_first_number(0);
3385   BlockBegin* target = bci2block()->at(osr_bci);
3386   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3387   // the osr entry has no values for locals
3388   ValueStack* state = target->state()->copy();
3389   _osr_entry->set_state(state);
3390 
3391   kill_all();
3392   _block = _osr_entry;
3393   _state = _osr_entry->state()->copy();
3394   assert(_state->bci() == osr_bci, "mismatch");
3395   _last  = _osr_entry;
3396   Value e = append(new OsrEntry());
3397   e->set_needs_null_check(false);
3398 
3399   // OSR buffer is
3400   //
3401   // locals[nlocals-1..0]
3402   // monitors[number_of_locks-1..0]
3403   //
3404   // locals is a direct copy of the interpreter frame so in the osr buffer
3405   // so first slot in the local array is the last local from the interpreter
3406   // and last slot is local[0] (receiver) from the interpreter
3407   //
3408   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3409   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3410   // in the interpreter frame (the method lock if a sync method)
3411 
3412   // Initialize monitors in the compiled activation.
3413 
3414   int index;
3415   Value local;
3416 
3417   // find all the locals that the interpreter thinks contain live oops
3418   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3419 
3420   // compute the offset into the locals so that we can treat the buffer
3421   // as if the locals were still in the interpreter frame
3422   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3423   for_each_local_value(state, index, local) {
3424     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3425     Value get;
3426     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3427       // The interpreter thinks this local is dead but the compiler
3428       // doesn't so pretend that the interpreter passed in null.
3429       get = append(new Constant(objectNull));
3430     } else {
3431       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3432                                     append(new Constant(new IntConstant(offset))),
3433                                     0,
3434                                     true /*unaligned*/, true /*wide*/));
3435     }
3436     _state->store_local(index, get);
3437   }
3438 
3439   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3440 
3441   assert(state->caller_state() == NULL, "should be top scope");
3442   state->clear_locals();
3443   Goto* g = new Goto(target, false);
3444   append(g);
3445   _osr_entry->set_end(g);
3446   target->merge(_osr_entry->end()->state());
3447 
3448   scope_data()->set_stream(NULL);
3449 }

3452 ValueStack* GraphBuilder::state_at_entry() {
3453   ValueStack* state = new ValueStack(scope(), NULL);
3454 
3455   // Set up locals for receiver
3456   int idx = 0;
3457   if (!method()->is_static()) {
3458     // we should always see the receiver
3459     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3460              /*receiver*/ true, /*never_null*/ method()->holder()->is_flat_array_klass()));
3461     idx = 1;
3462   }
3463 
3464   // Set up locals for incoming arguments
3465   ciSignature* sig = method()->signature();
3466   for (int i = 0; i < sig->count(); i++) {
3467     ciType* type = sig->type_at(i);
3468     BasicType basic_type = type->basic_type();
3469     // don't allow T_ARRAY to propagate into locals types
3470     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3471     ValueType* vt = as_ValueType(basic_type);
3472     state->store_local(idx, new Local(type, vt, idx, false, type->is_inlinetype()));
3473     idx += type->size();
3474   }
3475 
3476   // lock synchronized method
3477   if (method()->is_synchronized()) {
3478     state->lock(NULL);
3479   }
3480 
3481   return state;

3485 GraphBuilder::GraphBuilder(Compilation* compilation, IRScope* scope)
3486   : _scope_data(NULL)
3487   , _compilation(compilation)
3488   , _memory(new MemoryBuffer())
3489   , _inline_bailout_msg(NULL)
3490   , _instruction_count(0)
3491   , _osr_entry(NULL)
3492 {
3493   int osr_bci = compilation->osr_bci();
3494 
3495   // determine entry points and bci2block mapping
3496   BlockListBuilder blm(compilation, scope, osr_bci);
3497   CHECK_BAILOUT();
3498 
3499   BlockList* bci2block = blm.bci2block();
3500   BlockBegin* start_block = bci2block->at(0);
3501 
3502   push_root_scope(scope, bci2block, start_block);
3503 
3504   // setup state for std entry
3505   _initial_state = state_at_entry();
3506   start_block->merge(_initial_state);
3507 
3508   // complete graph
3509   _vmap        = new ValueMap();
3510   switch (scope->method()->intrinsic_id()) {
3511   case vmIntrinsics::_dabs          : // fall through
3512   case vmIntrinsics::_dsqrt         : // fall through
3513   case vmIntrinsics::_dsin          : // fall through
3514   case vmIntrinsics::_dcos          : // fall through
3515   case vmIntrinsics::_dtan          : // fall through
3516   case vmIntrinsics::_dlog          : // fall through
3517   case vmIntrinsics::_dlog10        : // fall through
3518   case vmIntrinsics::_dexp          : // fall through
3519   case vmIntrinsics::_dpow          : // fall through
3520     {
3521       // Compiles where the root method is an intrinsic need a special
3522       // compilation environment because the bytecodes for the method
3523       // shouldn't be parsed during the compilation, only the special
3524       // Intrinsic node should be emitted.  If this isn't done the the
3525       // code for the inlined version will be different than the root
3526       // compiled version which could lead to monotonicity problems on
3527       // intel.
3528       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3529         BAILOUT("failed to inline intrinsic, method not annotated");
3530       }
3531 
3532       // Set up a stream so that appending instructions works properly.
3533       ciBytecodeStream s(scope->method());
3534       s.reset_to_bci(0);
3535       scope_data()->set_stream(&s);
3536       s.next();
3537 
3538       // setup the initial block state
3539       _block = start_block;
3540       _state = start_block->state()->copy_for_parsing();
3541       _last  = start_block;
3542       load_local(doubleType, 0);
3543       if (scope->method()->intrinsic_id() == vmIntrinsics::_dpow) {
3544         load_local(doubleType, 2);
3545       }
3546 
3547       // Emit the intrinsic node.
3548       bool result = try_inline_intrinsics(scope->method());
3549       if (!result) BAILOUT("failed to inline intrinsic");
3550       method_return(dpop());
3551 
3552       // connect the begin and end blocks and we're all done.
3553       BlockEnd* end = last()->as_BlockEnd();
3554       block()->set_end(end);
3555       break;
3556     }
3557 
3558   case vmIntrinsics::_Reference_get:
3559     {
3560       {
3561         // With java.lang.ref.reference.get() we must go through the
3562         // intrinsic - when G1 is enabled - even when get() is the root
3563         // method of the compile so that, if necessary, the value in
3564         // the referent field of the reference object gets recorded by
3565         // the pre-barrier code.
3566         // Specifically, if G1 is enabled, the value in the referent
3567         // field is recorded by the G1 SATB pre barrier. This will
3568         // result in the referent being marked live and the reference
3569         // object removed from the list of discovered references during
3570         // reference processing.
3571         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3572           BAILOUT("failed to inline intrinsic, method not annotated");
3573         }
3574 
3575         // Also we need intrinsic to prevent commoning reads from this field
3576         // across safepoint since GC can change its value.
3577 
3578         // Set up a stream so that appending instructions works properly.
3579         ciBytecodeStream s(scope->method());
3580         s.reset_to_bci(0);
3581         scope_data()->set_stream(&s);
3582         s.next();
3583 
3584         // setup the initial block state
3585         _block = start_block;
3586         _state = start_block->state()->copy_for_parsing();
3587         _last  = start_block;
3588         load_local(objectType, 0);
3589 
3590         // Emit the intrinsic node.
3591         bool result = try_inline_intrinsics(scope->method());
3592         if (!result) BAILOUT("failed to inline intrinsic");
3593         method_return(apop());
3594 
3595         // connect the begin and end blocks and we're all done.
3596         BlockEnd* end = last()->as_BlockEnd();
3597         block()->set_end(end);
3598         break;
3599       }
3600       // Otherwise, fall thru
3601     }
3602 
3603   default:
3604     scope_data()->add_to_work_list(start_block);
3605     iterate_all_blocks();
3606     break;
3607   }
3608   CHECK_BAILOUT();
3609 
3610   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3611 
3612   eliminate_redundant_phis(_start);
3613 
3614   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3615   // for osr compile, bailout if some requirements are not fulfilled
3616   if (osr_bci != -1) {
3617     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3618     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3619       BAILOUT("osr entry must have been visited for osr compile");
3620     }
3621 
3622     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3623     if (!osr_block->state()->stack_is_empty()) {
3624       BAILOUT("stack not empty at OSR entry point");
3625     }
3626   }
3627 #ifndef PRODUCT
3628   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3629 #endif
3630 }

3677 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3678   const char* msg = NULL;
3679 
3680   // clear out any existing inline bailout condition
3681   clear_inline_bailout();
3682 
3683   // exclude methods we don't want to inline
3684   msg = should_not_inline(callee);
3685   if (msg != NULL) {
3686     print_inlining(callee, msg, /*success*/ false);
3687     return false;
3688   }
3689 
3690   // method handle invokes
3691   if (callee->is_method_handle_intrinsic()) {
3692     if (try_method_handle_inline(callee, ignore_return)) {
3693       if (callee->has_reserved_stack_access()) {
3694         compilation()->set_has_reserved_stack_access(true);
3695       }
3696       return true;
3697     }
3698     return false;
3699   }
3700 
3701   // handle intrinsics
3702   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3703       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3704     if (try_inline_intrinsics(callee, ignore_return)) {
3705       print_inlining(callee, "intrinsic");
3706       if (callee->has_reserved_stack_access()) {
3707         compilation()->set_has_reserved_stack_access(true);
3708       }
3709       return true;
3710     }
3711     // try normal inlining
3712   }
3713 
3714   // certain methods cannot be parsed at all
3715   msg = check_can_parse(callee);
3716   if (msg != NULL) {
3717     print_inlining(callee, msg, /*success*/ false);
3718     return false;
3719   }
3720 
3721   // If bytecode not set use the current one.
3722   if (bc == Bytecodes::_illegal) {
3723     bc = code();
3724   }
3725   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3726     if (callee->has_reserved_stack_access()) {
3727       compilation()->set_has_reserved_stack_access(true);
3728     }
3729     return true;
3730   }
3731 
3732   // Entire compilation could fail during try_inline_full call.
3733   // In that case printing inlining decision info is useless.
3734   if (!bailed_out())
3735     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3736 
3737   return false;
3738 }

3885 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {

3887   // will be replaced with Gotos to this point.
3888   BlockBegin* cont = block_at(next_bci());
3889   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3890 
3891   // Note: can not assign state to continuation yet, as we have to
3892   // pick up the state from the Ret instructions.
3893 
3894   // Push callee scope
3895   push_scope_for_jsr(cont, jsr_dest_bci);
3896 
3897   // Temporarily set up bytecode stream so we can append instructions
3898   // (only using the bci of this stream)
3899   scope_data()->set_stream(scope_data()->parent()->stream());
3900 
3901   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3902   assert(jsr_start_block != NULL, "jsr start block must exist");
3903   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3904   Goto* goto_sub = new Goto(jsr_start_block, false);
3905   // Must copy state to avoid wrong sharing when parsing bytecodes
3906   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3907   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3908   append(goto_sub);
3909   _block->set_end(goto_sub);
3910   _last = _block = jsr_start_block;
3911 
3912   // Clear out bytecode stream
3913   scope_data()->set_stream(NULL);
3914 
3915   scope_data()->add_to_work_list(jsr_start_block);
3916 
3917   // Ready to resume parsing in subroutine
3918   iterate_all_blocks();
3919 
3920   // If we bailed out during parsing, return immediately (this is bad news)
3921   CHECK_BAILOUT_(false);
3922 
3923   // Detect whether the continuation can actually be reached. If not,
3924   // it has not had state set by the join() operations in
3925   // iterate_bytecodes_for_block()/ret() and we should not touch the
3926   // iteration state. The calling activation of
3927   // iterate_bytecodes_for_block will then complete normally.
3928   if (cont->state() != NULL) {

3930       // add continuation to work list instead of parsing it immediately

3940 
3941   // continuation is in work list, so end iteration of current block
3942   _skip_block = true;
3943   pop_scope_for_jsr();
3944 
3945   return true;
3946 }

4038     INLINE_BAILOUT("inlining prohibited by policy");

4041   if (callee->has_exception_handlers() &&
4042       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4043   if (callee->is_synchronized() &&
4044       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4045   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4046   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");

4052       method()->is_strict() != callee->is_strict()) {

4054     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4055     // When SSE2 is present, no special handling is needed.
4056     if (UseSSE < 2) {
4057       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4058     }

4076     assert(!callee->is_static(), "callee must not be static");

4100     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");

4103         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4104       INLINE_BAILOUT("callee uses too much stack");
4105     }

4136     //       an implicit null check since the callee is in a different scope

4288     _last  = inline_cleanup_return_prev();
4289     _state = inline_cleanup_state();
4290   } else if (continuation_preds == cont->number_of_preds()) {
4291     // Inlining caused that the instructions after the invoke in the
4292     // caller are not reachable any more. So skip filling this block
4293     // with instructions!
4294     assert(cont == continuation(), "");
4295     assert(_last && _last->as_BlockEnd(), "");
4296     _skip_block = true;
4297   } else {
4298     // Resume parsing in continuation block unless it was already parsed.
4299     // Note that if we don't change _last here, iteration in
4300     // iterate_bytecodes_for_block will stop when we return.
4301     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4302       // add continuation to work list instead of parsing it immediately
4303       assert(_last && _last->as_BlockEnd(), "");
4304       scope_data()->parent()->add_to_work_list(continuation());
4305       _skip_block = true;
4306     }
4307   }
4308 
4309   // Fill the exception handler for synchronized methods with instructions
4310   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4311     fill_sync_handler(lock, sync_handler);
4312   } else {
4313     pop_scope();
4314   }
4315 
4316   compilation()->notice_inlined_method(callee);
4317 
4318   return true;
4319 }

4322 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4323   ValueStack* state_before = copy_state_before();
4324   vmIntrinsics::ID iid = callee->intrinsic_id();
4325   switch (iid) {
4326   case vmIntrinsics::_invokeBasic:
4327     {
4328       // get MethodHandle receiver
4329       const int args_base = state()->stack_size() - callee->arg_size();
4330       ValueType* type = state()->stack_at(args_base)->type();
4331       if (type->is_constant()) {
4332         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4333         // We don't do CHA here so only inline static and statically bindable methods.
4334         if (target->is_static() || target->can_be_statically_bound()) {
4335           if (ciMethod::is_consistent_info(callee, target)) {
4336             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4337             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4338             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4339               return true;
4340             }
4341           } else {
4342             print_inlining(target, "signatures mismatch", /*success*/ false);
4343           }
4344         } else {
4345           print_inlining(target, "not static or statically bindable", /*success*/ false);
4346         }
4347       } else {
4348         print_inlining(callee, "receiver not constant", /*success*/ false);
4349       }
4350     }
4351     break;
4352 
4353   case vmIntrinsics::_linkToVirtual:
4354   case vmIntrinsics::_linkToStatic:
4355   case vmIntrinsics::_linkToSpecial:
4356   case vmIntrinsics::_linkToInterface:
4357     {
4358       // pop MemberName argument
4359       const int args_base = state()->stack_size() - callee->arg_size();
4360       ValueType* type = apop()->type();
4361       if (type->is_constant()) {
4362         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4363         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4364         // If the target is another method handle invoke, try to recursively get
4365         // a better target.
4366         if (target->is_method_handle_intrinsic()) {
4367           if (try_method_handle_inline(target, ignore_return)) {
4368             return true;
4369           }
4370         } else if (!ciMethod::is_consistent_info(callee, target)) {
4371           print_inlining(target, "signatures mismatch", /*success*/ false);
4372         } else {
4373           ciSignature* signature = target->signature();
4374           const int receiver_skip = target->is_static() ? 0 : 1;
4375           // Cast receiver to its type.
4376           if (!target->is_static()) {
4377             ciKlass* tk = signature->accessing_klass();
4378             Value obj = state()->stack_at(args_base);
4379             if (obj->exact_type() == NULL &&
4380                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4381               TypeCast* c = new TypeCast(tk, obj, state_before);
4382               append(c);
4383               state()->stack_at_put(args_base, c);
4384             }
4385           }
4386           // Cast reference arguments to its type.
4387           for (int i = 0, j = 0; i < signature->count(); i++) {
4388             ciType* t = signature->type_at(i);
4389             if (t->is_klass()) {
4390               ciKlass* tk = t->as_klass();
4391               Value obj = state()->stack_at(args_base + receiver_skip + j);
4392               if (obj->exact_type() == NULL &&
4393                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4394                 TypeCast* c = new TypeCast(t, obj, state_before);
4395                 append(c);
4396                 state()->stack_at_put(args_base + receiver_skip + j, c);
4397               }
4398             }
4399             j += t->size();  // long and double take two slots
4400           }
4401           // We don't do CHA here so only inline static and statically bindable methods.
4402           if (target->is_static() || target->can_be_statically_bound()) {
4403             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4404             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4405               return true;
4406             }
4407           } else {
4408             print_inlining(target, "not static or statically bindable", /*success*/ false);
4409           }
4410         }
4411       } else {
4412         print_inlining(callee, "MemberName not constant", /*success*/ false);
4413       }
4414     }
4415     break;
4416 
4417   default:
4418     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4419     break;
4420   }
4421   set_state(state_before->copy_for_parsing());
4422   return false;
4423 }
