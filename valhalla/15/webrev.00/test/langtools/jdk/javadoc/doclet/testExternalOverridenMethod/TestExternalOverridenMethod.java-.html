24 /*
25  * @test
26  * @bug 4857717 8025633 8026567 8164407 8182765 8205593
27  * @summary Test to make sure that externally overriden and implemented methods
28  * are documented properly.  The method should still include "implements" or
29  * "overrides" documentation even though the method is external.
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.* TestExternalOverridenMethod
33  * @run main TestExternalOverridenMethod
34  */
35 import javadoc.tester.JavadocTester;
36 
37 public class TestExternalOverridenMethod extends JavadocTester {
38 
39     static final String uri = "http://java.sun.com/j2se/1.4.1/docs/api";
40 
41     public static void main(String... args) throws Exception {
42         TestExternalOverridenMethod tester = new TestExternalOverridenMethod();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         javadoc("-d", "out",
49                 "-source","8",
50                 "-sourcepath", testSrc,
51                 "-linkoffline", uri, testSrc,
52                 "pkg");
53         checkExit(Exit.OK);
54 
55         checkOutput("pkg/XReader.html", true,
56                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
57                 + "<dd><code><a href=\"" + uri + "/java/io/FilterReader.html#read()\" "
58                 + "title=\"class or interface in java.io\" class=\"externalLink\">read</a></code>&nbsp;in class&nbsp;<code>"
59                 + "<a href=\"" + uri + "/java/io/FilterReader.html\" "
60                 + "title=\"class or interface in java.io\" class=\"externalLink\">FilterReader</a></code></dd>",
61                 "<dt><span class=\"overrideSpecifyLabel\">Specified by:</span></dt>\n"
62                 + "<dd><code><a href=\"" + uri + "/java/io/DataInput.html#readInt()\" "
63                 + "title=\"class or interface in java.io\" class=\"externalLink\">readInt</a></code>&nbsp;in interface&nbsp;<code>"
64                 + "<a href=\"" + uri + "/java/io/DataInput.html\" "
65                 + "title=\"class or interface in java.io\" class=\"externalLink\">DataInput</a></code></dd>"
66         );
67     }
68 }
