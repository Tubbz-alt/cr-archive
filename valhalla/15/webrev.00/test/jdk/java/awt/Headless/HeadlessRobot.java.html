24 import java.awt.AWTException;
25 import java.awt.GraphicsConfiguration;
26 import java.awt.GraphicsDevice;
27 import java.awt.Robot;
28 
29 /**
30  * @test
31  * @bug 8238741
32  * @summary Check that Robot constructors throw AWTException in a headless
33  *          environment
34  * @run main/othervm -Djava.awt.headless=true HeadlessRobot
35  */
36 public final class HeadlessRobot {
37 
38     public static void main(String[] args) {
39         try {
40             new Robot();
41             throw new RuntimeException("Expected AWTException did not occur");
42         } catch (AWTException ignored) {
43             // expected AWTException
44         }
45 
46         try {
47             new Robot(new GraphicsDevice() {
48                 @Override
49                 public int getType() {
50                     return TYPE_RASTER_SCREEN;
51                 }
52 
53                 @Override
54                 public String getIDstring() {
55                     return "Stub device";
56                 }
57 
58                 @Override
59                 public GraphicsConfiguration[] getConfigurations() {
60                     return null;
61                 }
62 
63                 @Override
64                 public GraphicsConfiguration getDefaultConfiguration() {
65                     return null;
66                 }
67             });
68             throw new RuntimeException("Expected AWTException did not occur");
69         } catch (AWTException ignored) {
70             // expected AWTException
71         }
72     }
73 }
