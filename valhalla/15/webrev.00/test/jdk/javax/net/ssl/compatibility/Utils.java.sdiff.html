137         int length = hex.length();
138         if (length % 2 != 0) {
139             throw new IllegalArgumentException("Hex format is wrong.");

142         byte[] bytes = new byte[length / 2];
143         for (int i = 0; i < length; i += 2) {
144             bytes[i / 2] = (byte) ((Character.digit(hex.charAt(i), 16) << 4)
