 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 8216329
 28  * @summary Verify that JCheckBoxMenuItem is resized properly in Synth L&F
 29  * @run main/manual TestJCheckBoxMenuItem
 30  */
 31 
 32 import java.awt.Color;
 33 import java.awt.Dimension;
 34 import java.awt.GridBagConstraints;
 35 import java.awt.GridBagLayout;
 36 import java.awt.Insets;
 37 import java.awt.event.ActionEvent;
 38 import java.awt.event.KeyEvent;
 39 import java.awt.event.WindowAdapter;
 40 import java.awt.event.WindowEvent;
 41 import java.io.ByteArrayInputStream;
 42 import java.util.concurrent.CountDownLatch;
 43 import java.util.concurrent.TimeUnit;
 44 
 45 import javax.swing.JButton;
 46 import javax.swing.JFrame;
 47 import javax.swing.JPanel;
 48 import javax.swing.JTextArea;
 49 import javax.swing.JMenuBar;
 50 import javax.swing.JMenu;
 51 import javax.swing.JCheckBoxMenuItem;
 52 import javax.swing.SwingUtilities;
 53 import javax.swing.UIManager;
 54 import javax.swing.plaf.synth.SynthLookAndFeel;
 55 import javax.swing.UnsupportedLookAndFeelException;
 56 
 57 public class TestJCheckBoxMenuItem {
 58 
 59     private static final JFrame instructionFrame = new JFrame();
 60     private static final JFrame testFrame = new JFrame();
 61 
 62     private static volatile boolean testResult = false;
 63     private static volatile CountDownLatch countDownLatch;
 64     private static final String INSTRUCTIONS = "INSTRUCTIONS:\n\n" +
 65             "Click on the menu \"Menu\" in Test frame. Verify that the \n" +
 66             "JCheckBoxMenuItem is resized properly and text \n" +
 67             "\"JCheckBoxMenuItem Sample Text\" is not truncated.\n" +
 68             "If yes, Press Pass, Otherwise, Press Fail.";
 69 
 70     private static final String synthStyleXML =
 71             "<synth>" +
 72                 "<style id=\"Default\">" +
 73                     "<font name=\"Dialog\" size=\"14\" />" +
 74                 "</style>" +
 75                 "<bind style=\"Default\" type=\"region\" key=\".*\" />" +
 76                 "<style id=\"Check Box Menu Item\">" +
 77                     "<imageIcon id=\"CheckBoxMenuItem_Check_Icon_Selected\" " +
 78                         "path=\"Check_Icon.png\" />" +
 79                     "<insets top=\"4\" left=\"9\" bottom=\"4\" right=\"9\" />"+
 80                     "<state>" +
 81                         "<property key=\"CheckBoxMenuItem.checkIcon\" " +
 82                             "value=\"CheckBoxMenuItem_Check_Icon_Selected\" />"+
 83                         "<imagePainter method=\"checkBoxMenuItemBackground\" "+
 84                             "path=\"MenuItem_Selected.png\" " +
 85                             "sourceInsets=\"2 2 2 2\" paintCenter=\"true\" " +
 86                             "stretch=\"true\" center=\"false\" />"+
 87                     "</state>" +
 88                 "</style>" +
 89                 "<bind style=\"Check Box Menu Item\" type=\"region\" " +
 90                     "key=\"CheckBoxMenuItem\" />" +
 91             "</synth>";
 92     public static void main(String[] args) throws Exception {
 93         countDownLatch = new CountDownLatch(1);
 94 
 95         SwingUtilities.invokeAndWait(TestJCheckBoxMenuItem::createInstructionUI);
 96 
 97         SynthLookAndFeel lookAndFeel = new SynthLookAndFeel();
 98         lookAndFeel.load(
 99                 new ByteArrayInputStream(synthStyleXML.getBytes("UTF8")),
100                 TestJCheckBoxMenuItem.class);
101         try {
102             UIManager.setLookAndFeel(lookAndFeel);
103         } catch (final UnsupportedLookAndFeelException ignored) {
104             System.out.println("Synth L&F could not be set, so this test can" +
105                     "not be run in this scenario ");
106             return;
107         }
108         SwingUtilities.invokeAndWait(TestJCheckBoxMenuItem::createTestUI);
109 
110         countDownLatch.await(15, TimeUnit.MINUTES);
111 
112         disposeUI();
113         if (!testResult) {
114             throw new RuntimeException("Test failed!");
115         }
116     }
117 
118     private static void createInstructionUI() {
119         GridBagLayout layout = new GridBagLayout();
120         JPanel mainControlPanel = new JPanel(layout);
121         JPanel resultButtonPanel = new JPanel(layout);
122 
123         GridBagConstraints gbc = new GridBagConstraints();
124 
125         gbc.gridx = 0;
126         gbc.gridy = 0;
127         gbc.insets = new Insets(5, 15, 5, 15);
128         gbc.fill = GridBagConstraints.HORIZONTAL;
129 
130         JTextArea instructionTextArea = new JTextArea();
131         instructionTextArea.setText(INSTRUCTIONS);
132         instructionTextArea.setEditable(false);
133         instructionTextArea.setBackground(Color.white);
134         mainControlPanel.add(instructionTextArea, gbc);
135 
136         JButton passButton = new JButton("Pass");
137         passButton.setActionCommand("Pass");
138         passButton.addActionListener((ActionEvent e) -> {
139             testResult = true;
140             countDownLatch.countDown();
141 
142         });
143 
144         JButton failButton = new JButton("Fail");
145         failButton.setActionCommand("Fail");
146         failButton.addActionListener(e -> {
147             countDownLatch.countDown();
148         });
149 
150         gbc.gridx = 0;
151         gbc.gridy = 0;
152 
153         resultButtonPanel.add(passButton, gbc);
154 
155         gbc.gridx = 1;
156         gbc.gridy = 0;
157         resultButtonPanel.add(failButton, gbc);
158 
159         gbc.gridx = 0;
160         gbc.gridy = 2;
161         mainControlPanel.add(resultButtonPanel, gbc);
162 
163         instructionFrame.pack();
164         instructionFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
165         instructionFrame.add(mainControlPanel);
166         instructionFrame.pack();
167 
168         instructionFrame.addWindowListener(new WindowAdapter() {
169             @Override
170             public void windowClosing(WindowEvent e) {
171                 disposeUI();
172                 countDownLatch.countDown();
173             }
174         });
175         instructionFrame.setVisible(true);
176     }
177 
178     private static void createTestUI() {
179         JMenuBar menuBar = new JMenuBar();
180         JMenu menu = new JMenu("Menu");
181 
182         JCheckBoxMenuItem checkBoxMenuItem = new JCheckBoxMenuItem();
183         checkBoxMenuItem.setSelected(false);
184         checkBoxMenuItem.setMnemonic(KeyEvent.VK_K);
185         checkBoxMenuItem.setText("JCheckBoxMenuItem Sample Text");
186         checkBoxMenuItem.setHorizontalTextPosition(JCheckBoxMenuItem.LEADING);
187         checkBoxMenuItem.setVerticalTextPosition(JCheckBoxMenuItem.CENTER);
188 
189         menu.add(checkBoxMenuItem);
190         menuBar.add(menu);
191 
192         testFrame.setJMenuBar(menuBar);
193         testFrame.setLocation(
194                 instructionFrame.getX() + instructionFrame.getWidth(),
195                 instructionFrame.getY());
196         testFrame.setPreferredSize(new Dimension(instructionFrame.getWidth(),
197                 instructionFrame.getHeight()));
198         testFrame.pack();
199         testFrame.setTitle("Test");
200         testFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
201         testFrame.setVisible(true);
202     }
203 
204     private static void disposeUI() {
205         instructionFrame.dispose();
206         testFrame.dispose();
207     }
208 }
