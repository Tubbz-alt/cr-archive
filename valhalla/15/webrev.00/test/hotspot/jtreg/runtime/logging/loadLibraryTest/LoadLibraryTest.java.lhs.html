24 /*
25  * @test
26  * @bug 8187305
27  * @summary Tests logging of shared library loads and unloads.
28  * @library /runtime/testlibrary /test/lib
29  * @compile LoadLibrary.java


30  * @run main LoadLibraryTest
31  */
32 






33 import jdk.test.lib.process.ProcessTools;
34 import jdk.test.lib.process.OutputAnalyzer;



35 
36 public class LoadLibraryTest {
37 







































































38     public static void main(String... args) throws Exception {
39         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
40             "-Xlog:library=info", "-Djava.library.path=" + System.getProperty("java.library.path"),
41             "LoadLibrary", System.getProperty("test.classes"));



42 
43         OutputAnalyzer output = new OutputAnalyzer(pb.start());
44         output.shouldContain("Loaded library");
45         output.shouldContain("Found Java_LoadLibraryClass_nTest in library");
46         output.shouldContain("Unloaded library with handle");
47         output.shouldHaveExitValue(0);








48     }
49 }
