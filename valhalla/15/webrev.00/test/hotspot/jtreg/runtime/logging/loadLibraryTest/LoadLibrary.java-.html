24 import java.io.File;
25 import java.io.FileInputStream;
26 import java.io.IOException;
27 import java.nio.ByteBuffer;
28 import java.nio.channels.FileChannel;
29 
30 public class LoadLibrary {
31 
32     public static String testClasses;
33 
34     public static void runTest() throws Exception {
35         // create a classloader and load a class that loads a library.
36         MyClassLoader myLoader = new MyClassLoader();
37         Class<?> c = Class.forName("LoadLibraryClass", true, myLoader);
38     }
39 
40     public static void main(String[] args) throws Exception {
41         testClasses = args[0];
42         runTest();
43         ClassUnloadCommon.triggerUnloading();
44     }
45 
46     public static class MyClassLoader extends ClassLoader {
47 
48         public static final String CLASS_NAME = "LoadLibraryClass";
49 
50         static ByteBuffer readClassFile(String name) {
51             File f = new File(testClasses, name);
52             try (FileInputStream fin = new FileInputStream(f);
53                  FileChannel fc = fin.getChannel())
54             {
55                 return fc.map(FileChannel.MapMode.READ_ONLY, 0, fc.size());
56             } catch (IOException e) {
57                 throw new RuntimeException("Can't open file: " + name +
58                                            ", exception: " + e.toString());
59             }
60         }
61 
62         protected Class<?> loadClass(String name, boolean resolve)
63             throws ClassNotFoundException {
64             Class<?> c;
65             if (!"LoadLibraryClass".equals(name)) {
66                 c = super.loadClass(name, resolve);
67             } else {
68                 // should not delegate to the system class loader
69                 c = findClass(name);
70                 if (resolve) {
71                     resolveClass(c);
72                 }
73             }
74             return c;
75         }
76 
77         protected Class<?> findClass(String name) throws ClassNotFoundException {
78             if (!"LoadLibraryClass".equals(name)) {
79                 throw new ClassNotFoundException("Unexpected class: " + name);
80             }
81             return defineClass(name, readClassFile(name + ".class"), null);
82         }
83     } // MyClassLoader
84 
85 }
86 
87 
88 class LoadLibraryClass {
89     static {
90         System.loadLibrary("LoadLibraryClass");
91         nTest();
92     }
93     native static void nTest();
94 }
