 24 /*
 25  * @test
 26  * @bug 8187305
 27  * @summary Tests logging of shared library loads and unloads.
 28  * @library /runtime/testlibrary /test/lib
 29  * @build sun.hotspot.WhiteBox
 30  * @run main ClassFileInstaller sun.hotspot.WhiteBox
 31  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
 32  * @run main LoadLibraryTest
 33  */
 34 
 35 import java.io.File;
 36 import java.io.FileInputStream;
 37 import java.io.IOException;
 38 import java.nio.ByteBuffer;
 39 import java.nio.channels.FileChannel;
 40 
 41 import jdk.test.lib.process.ProcessTools;
 42 import jdk.test.lib.process.OutputAnalyzer;
 43 import jtreg.SkippedException;
 44 
 45 import sun.hotspot.WhiteBox;
 46 
 47 public class LoadLibraryTest {
 48 
 49     public static final String CLASS_NAME = "LoadLibraryTest$LoadLibraryClass";
 50 
 51     private static class LoadLibrary {
 52 
 53         static String testClasses;
 54 
 55         public static void runTest() throws Exception {
 56             // create a classloader and load a class that loads a library.
 57             MyClassLoader myLoader = new MyClassLoader();
 58             Class<?> c = Class.forName(CLASS_NAME, true, myLoader);
 59         }
 60 
 61         public static void main(String[] args) throws Exception {
 62             testClasses = args[0];
 63             runTest();
 64             ClassUnloadCommon.triggerUnloading();
 65             WhiteBox wb = WhiteBox.getWhiteBox();
 66             if (!wb.isClassAlive(CLASS_NAME)) {
 67                 System.out.println("Class LoadLibraryClass was unloaded");
 68             }
 69         }
 70 
 71 
 72         public static class MyClassLoader extends ClassLoader {
 73 
 74             static ByteBuffer readClassFile(String name) {
 75                 File f = new File(testClasses, name);
 76                 try (FileInputStream fin = new FileInputStream(f);
 77                      FileChannel fc = fin.getChannel())
 78                 {
 79                     return fc.map(FileChannel.MapMode.READ_ONLY, 0, fc.size());
 80                 } catch (IOException e) {
 81                     throw new RuntimeException("Can't open file: " + name +
 82                                                ", exception: " + e.toString());
 83                 }
 84             }
 85 
 86             protected Class<?> loadClass(String name, boolean resolve)
 87                 throws ClassNotFoundException {
 88                 Class<?> c;
 89                 if (!CLASS_NAME.equals(name)) {
 90                     c = super.loadClass(name, resolve);
 91                 } else {
 92                     // should not delegate to the system class loader
 93                     c = findClass(name);
 94                     if (resolve) {
 95                         resolveClass(c);
 96                     }
 97                 }
 98                 return c;
 99             }
100 
101             protected Class<?> findClass(String name) throws ClassNotFoundException {
102                 if (!CLASS_NAME.equals(name)) {
103                     throw new ClassNotFoundException("Unexpected class: " + name);
104                 }
105                 return defineClass(name, readClassFile(name + ".class"), null);
106             }
107         } // MyClassLoader
108     }
109 
110 
111     static class LoadLibraryClass {
112         static {
113             System.loadLibrary("LoadLibraryClass");
114             nTest();
115         }
116         native static void nTest();
117     }
118 
119 
120     public static void main(String... args) throws Exception {
121         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
122             "-Xbootclasspath/a:.", "-XX:+UnlockDiagnosticVMOptions",
123             "-XX:+WhiteBoxAPI", "-Xmn8m", "-Xlog:library=info",
124             "-Djava.library.path=" + System.getProperty("java.library.path"),
125             "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
126             "LoadLibraryTest$LoadLibrary", System.getProperty("test.classes"));
127 
128         OutputAnalyzer output = new OutputAnalyzer(pb.start());



129         output.shouldHaveExitValue(0);
130         output.shouldContain("Loaded library");
131         output.shouldContain("Found Java_LoadLibraryTest_00024LoadLibraryClass_nTest__ in library");
132         if (output.getOutput().contains("Class LoadLibraryClass was unloaded")) {
133             output.shouldContain("Unloaded library with handle");
134         } else {
135             throw new SkippedException(
136                 "Skipping check for library unloading logging because no unloading occurred");
137         }
138     }
139 }
