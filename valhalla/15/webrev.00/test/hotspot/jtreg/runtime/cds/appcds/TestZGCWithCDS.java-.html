1 /*
  2  * @test 8232069 for ZGC
  3  * @requires vm.cds
  4  * @requires (vm.gc=="null")
  5  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
  6  * @compile test-classes/Hello.java
  7  * @build sun.hotspot.WhiteBox
  8  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
  9  * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. TestZGCWithCDS
 10  */
 11 
 12 import jdk.test.lib.Platform;
 13 import jdk.test.lib.process.OutputAnalyzer;
 14 import jtreg.SkippedException;
 15 
 16 import sun.hotspot.gc.GC;
 17 
 18 public class TestZGCWithCDS {
 19     public final static String HELLO = "Hello World";
 20     public final static String UNABLE_TO_USE_ARCHIVE = "Unable to use shared archive.";
 21     public final static String ERR_MSG = "The saved state of UseCompressedOops and UseCompressedClassPointers is different from runtime, CDS will be disabled.";
 22     public static void main(String... args) throws Exception {
 23          // The test is only for 64-bit
 24          if (!Platform.is64bit()) {
 25              throw new SkippedException("Platform is not 64 bit, skipped");
 26          }
 27 
 28          // Platform must support ZGC
 29          if (!GC.Z.isSupported()) {
 30              throw new SkippedException("Platform does not support ZGC, skipped");
 31          }
 32 
 33          String helloJar = JarBuilder.build("hello", "Hello");
 34          // 0. dump with ZGC
 35          System.out.println("0. Dump with ZGC");
 36          OutputAnalyzer out = TestCommon
 37                                   .dump(helloJar,
 38                                         new String[] {"Hello"},
 39                                         "-XX:+UnlockExperimentalVMOptions",
 40                                         "-XX:+UseZGC",
 41                                         "-Xlog:cds");
 42          out.shouldContain("Dumping shared data to file:");
 43          out.shouldHaveExitValue(0);
 44 
 45          // 1. Run with same args of dump
 46          System.out.println("1. Run with same args of dump");
 47          out = TestCommon
 48                    .exec(helloJar,
 49                          "-XX:+UnlockExperimentalVMOptions",
 50                          "-XX:+UseZGC",
 51                          "-Xlog:cds",
 52                          "Hello");
 53          out.shouldContain(HELLO);
 54          out.shouldHaveExitValue(0);
 55 
 56          // 2. Run with ZGC turned off
 57          System.out.println("2. Run with ZGC turned off");
 58          out = TestCommon
 59                    .exec(helloJar,
 60                          "-XX:+UnlockExperimentalVMOptions",
 61                          "-XX:-UseZGC",
 62                          "-Xlog:cds",
 63                          "Hello");
 64          out.shouldContain(UNABLE_TO_USE_ARCHIVE);
 65          out.shouldContain(ERR_MSG);
 66          out.shouldHaveExitValue(1);
 67 
 68          // 3. Run with -UseCompressedOops -UseCompressedClassPointers
 69          System.out.println("3. Run with -UseCompressedOops -UseCompressedClassPointers");
 70          out = TestCommon
 71                    .exec(helloJar,
 72                          "-XX:-UseCompressedOops",
 73                          "-XX:-UseCompressedClassPointers",
 74                          "-Xlog:cds",
 75                          "Hello");
 76          out.shouldContain(HELLO);
 77          out.shouldHaveExitValue(0);
 78 
 79          // 4. Run with +UseCompressedOops -UseCompressedClassPointers
 80          System.out.println("4. Run with +UseCompressedOops -UseCompressedClassPointers");
 81          out = TestCommon
 82                    .exec(helloJar,
 83                          "-XX:+UseCompressedOops",
 84                          "-XX:-UseCompressedClassPointers",
 85                          "-Xlog:cds",
 86                          "Hello");
 87          out.shouldContain(UNABLE_TO_USE_ARCHIVE);
 88          out.shouldContain(ERR_MSG);
 89          out.shouldHaveExitValue(1);
 90 
 91          // 5. Run with +UseCompressedOops +UseCompressedClassPointers
 92          System.out.println("5. Run with +UseCompressedOops +UseCompressedClassPointers");
 93          out = TestCommon
 94                    .exec(helloJar,
 95                          "-XX:+UseCompressedOops",
 96                          "-XX:+UseCompressedClassPointers",
 97                          "-Xlog:cds",
 98                          "Hello");
 99          out.shouldContain(UNABLE_TO_USE_ARCHIVE);
100          out.shouldContain(ERR_MSG);
101          out.shouldHaveExitValue(1);
102 
103          // 6. dump with -UseCompressedOops -UseCompressedClassPointers
104          System.out.println("6. Dump with -UseCompressedOops -UseCompressedClassPointers");
105          out = TestCommon
106                    .dump(helloJar,
107                          new String[] {"Hello"},
108                          "-XX:-UseCompressedOops",
109                          "-XX:-UseCompressedClassPointers",
110                          "-Xlog:cds");
111          out.shouldContain("Dumping shared data to file:");
112          out.shouldHaveExitValue(0);
113 
114          // 7. Run with ZGC
115          System.out.println("7. Run with ZGC");
116          out = TestCommon
117                    .exec(helloJar,
118                          "-XX:+UnlockExperimentalVMOptions",
119                          "-XX:+UseZGC",
120                          "-Xlog:cds",
121                          "Hello");
122          out.shouldContain(HELLO);
123          out.shouldHaveExitValue(0);
124     }
125 }
