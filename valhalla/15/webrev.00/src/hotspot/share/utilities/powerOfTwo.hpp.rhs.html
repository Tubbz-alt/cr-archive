 35 // Power of two convenience library.
 36 
 37 template <typename T>
 38 bool is_power_of_2(T x) {
 39   return (x > T(0)) && ((x & (x - 1)) == T(0));
 40 }
 41 
 42 // Log2 of a power of 2
 43 inline int exact_log2(intptr_t x) {
 44   assert(is_power_of_2((uintptr_t)x), "x must be a power of 2: " INTPTR_FORMAT, x);
 45 
 46   const int bits = sizeof x * BitsPerByte;
 47   return bits - count_leading_zeros(x) - 1;
 48 }
 49 
 50 // Log2 of a power of 2
 51 inline int upper_log2(intptr_t x) {
 52   int shift = log2_intptr(x);
 53   intptr_t y = 1ULL << shift;
 54   if (y < x) {
 55     shift++;
 56   }
 57   return shift;
 58 }
 59 
 60 // Log2 of a power of 2
 61 inline int exact_log2_long(jlong x) {
 62   assert(is_power_of_2((julong)x), "x must be a power of 2: " JLONG_FORMAT, x);
 63 
 64   const int bits = sizeof x * BitsPerByte;
 65   return bits - count_leading_zeros(x) - 1;
 66 }

 68 // Round down to the closest power of two greater to or equal to the given
 69 // value.
 70 
 71 // Signed version: 0 is an invalid input, negative values are invalid
 72 template <typename T>
 73 inline typename EnableIf<IsSigned<T>::value, T>::type round_down_power_of_2(T value) {
 74   STATIC_ASSERT(IsIntegral<T>::value);
 75   assert(value > 0, "Invalid value");
 76   uint32_t lz = count_leading_zeros(value);
 77   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 78   return T(1) << (sizeof(T) * BitsPerByte - 1 - lz);
 79 }
 80 
 81 // Unsigned version: 0 is an invalid input
 82 template <typename T>
 83 inline typename EnableIf<!IsSigned<T>::value, T>::type round_down_power_of_2(T value) {
 84   STATIC_ASSERT(IsIntegral<T>::value);
 85   assert(value != 0, "Invalid value");
 86   uint32_t lz = count_leading_zeros(value);
 87   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 88   return T(1) << (sizeof(T) * BitsPerByte - 1 - lz);
 89 }
 90 
 91 // Round up to the closest power of two greater to or equal to
 92 // the given value.
 93 
 94 // Signed version: 0 is an invalid input, negative values are invalid,
 95 // overflows with assert if value is larger than 2^30 or 2^62 for 32- and
 96 // 64-bit integers, respectively
 97 template <typename T>
 98 inline typename EnableIf<IsSigned<T>::value, T>::type round_up_power_of_2(T value) {
 99   STATIC_ASSERT(IsIntegral<T>::value);
100   STATIC_ASSERT(IsSigned<T>::value);
101   assert(value > 0, "Invalid value");
102   if (is_power_of_2(value)) {
103     return value;
104   }
105   uint32_t lz = count_leading_zeros(value);
106   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
107   assert(lz > 1, "Will overflow");
108   return T(1) << (sizeof(T) * BitsPerByte - lz);
109 }
110 
111 // Unsigned version: 0 is an invalid input, overflows with assert if value
112 // is larger than 2^31 or 2^63 for 32- and 64-bit integers, respectively
113 template <typename T>
114 inline typename EnableIf<!IsSigned<T>::value, T>::type round_up_power_of_2(T value) {
115   STATIC_ASSERT(IsIntegral<T>::value);
116   STATIC_ASSERT(!IsSigned<T>::value);
117   assert(value != 0, "Invalid value");
118   if (is_power_of_2(value)) {
119     return value;
120   }
121   uint32_t lz = count_leading_zeros(value);
122   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
123   assert(lz > 0, "Will overflow");
124   return T(1) << (sizeof(T) * BitsPerByte - lz);
125 }
126 
127 // Helper function to get the maximum positive value. Implemented here
128 // since using std::numeric_limits<T>::max() seems problematic on some
129 // platforms.
130 
131 template <typename T> T max_value() {
132   if (IsSigned<T>::value) {
133     // Highest positive power of two expressible in the type
134     uint64_t val = static_cast<T>(1) << (sizeof(T) * BitsPerByte - 2);
135     // Fill lower bits with ones
136     val |= val >> 1;
137     val |= val >> 2;
138     val |= val >> 4;
139     if (sizeof(T) >= 2)  val |= val >> 8;
140     if (sizeof(T) >= 4)  val |= val >> 16;
141     if (sizeof(T) == 8)  val |= val >> 32;
142     return (T)val;
143   } else {
144     return ~(static_cast<T>(0));
145   }
146 }
147 
148 // Calculate the next power of two greater than the given value.
149 
150 // Accepts 0 (returns 1), overflows with assert if value is larger than
151 // or equal to 2^31 (signed: 2^30) or 2^63 (signed: 2^62), for 32-
152 // and 64-bit integers, respectively
153 template <typename T>
154 inline T next_power_of_2(T value)  {
155   assert(value != max_value<T>(), "Overflow");
156   return round_up_power_of_2(value + 1);
157 }
