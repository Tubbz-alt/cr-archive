 35 // Power of two convenience library.
 36 
 37 template <typename T>
 38 bool is_power_of_2(T x) {
 39   return (x != T(0)) && ((x & (x - 1)) == T(0));
 40 }
 41 
 42 // Log2 of a power of 2
 43 inline int exact_log2(intptr_t x) {
 44   assert(is_power_of_2(x), "x must be a power of 2: " INTPTR_FORMAT, x);
 45   return log2_intptr(x);
 46 }
 47 
 48 // Log2 of a power of 2
 49 inline int upper_log2(intptr_t x) {
 50   int shift = log2_intptr(x);
 51   intptr_t y = 1ULL << shift;
 52   if (y < x) {
 53     shift++;
 54   }
 55   return shift;
 56 }
 57 
 58 // Log2 of a power of 2
 59 inline int exact_log2_long(jlong x) {
 60   assert(is_power_of_2(x), "x must be a power of 2: " JLONG_FORMAT, x);
 61   return log2_long(x);
 62 }

 64 // Round down to the closest power of two greater to or equal to the given
 65 // value.
 66 
 67 // Signed version: 0 is an invalid input, negative values are invalid
 68 template <typename T>
 69 inline typename EnableIf<IsSigned<T>::value, T>::type round_down_power_of_2(T value) {
 70   STATIC_ASSERT(IsIntegral<T>::value);
 71   assert(value > 0, "Invalid value");
 72   uint32_t lz = count_leading_zeros(value);
 73   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 74   return T(1) << (sizeof(T) * BitsPerByte - 1 - lz);
 75 }
 76 
 77 // Unsigned version: 0 is an invalid input
 78 template <typename T>
 79 inline typename EnableIf<!IsSigned<T>::value, T>::type round_down_power_of_2(T value) {
 80   STATIC_ASSERT(IsIntegral<T>::value);
 81   assert(value != 0, "Invalid value");
 82   uint32_t lz = count_leading_zeros(value);
 83   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 84   return T(1) << (sizeof(T) * BitsPerByte - 1 - lz);
 85 }

 87 // Round up to the closest power of two greater to or equal to
 88 // the given value.
 89 
 90 // Signed version: 0 is an invalid input, negative values are invalid,
 91 // overflows with assert if value is larger than 2^30 or 2^62 for 32- and
 92 // 64-bit integers, respectively
 93 template <typename T>
 94 inline typename EnableIf<IsSigned<T>::value, T>::type round_up_power_of_2(T value) {
 95   STATIC_ASSERT(IsIntegral<T>::value);
 96   STATIC_ASSERT(IsSigned<T>::value);
 97   assert(value > 0, "Invalid value");
 98   if (is_power_of_2(value)) {
 99     return value;
100   }
101   uint32_t lz = count_leading_zeros(value);
102   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
103   assert(lz > 1, "Will overflow");
104   return T(1) << (sizeof(T) * BitsPerByte - lz);
105 }
106 
107 // Unsigned version: 0 is an invalid input, overflows with assert if value
108 // is larger than 2^31 or 2^63 for 32- and 64-bit integers, respectively
109 template <typename T>
110 inline typename EnableIf<!IsSigned<T>::value, T>::type round_up_power_of_2(T value) {
111   STATIC_ASSERT(IsIntegral<T>::value);
112   STATIC_ASSERT(!IsSigned<T>::value);
113   assert(value != 0, "Invalid value");
114   if (is_power_of_2(value)) {
115     return value;
116   }
117   uint32_t lz = count_leading_zeros(value);
118   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
119   assert(lz > 0, "Will overflow");
120   return T(1) << (sizeof(T) * BitsPerByte - lz);
121 }

123 // Helper function to get the maximum positive value. Implemented here
124 // since using std::numeric_limits<T>::max() seems problematic on some
125 // platforms.
126 
127 template <typename T> T max_value() {
128   if (IsSigned<T>::value) {
129     // Highest positive power of two expressible in the type
130     uint64_t val = static_cast<T>(1) << (sizeof(T) * BitsPerByte - 2);
131     // Fill lower bits with ones
132     val |= val >> 1;
133     val |= val >> 2;
134     val |= val >> 4;
135     if (sizeof(T) >= 2)  val |= val >> 8;
136     if (sizeof(T) >= 4)  val |= val >> 16;
137     if (sizeof(T) == 8)  val |= val >> 32;
138     return (T)val;
139   } else {
140     return ~(static_cast<T>(0));
141   }
142 }
143 
144 // Calculate the next power of two greater than the given value.
145 
146 // Accepts 0 (returns 1), overflows with assert if value is larger than
147 // or equal to 2^31 (signed: 2^30) or 2^63 (signed: 2^62), for 32-
148 // and 64-bit integers, respectively
149 template <typename T>
150 inline T next_power_of_2(T value)  {
151   assert(value != max_value<T>(), "Overflow");
152   return round_up_power_of_2(value + 1);
153 }
