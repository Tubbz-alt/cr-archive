 63   _valid_omv = false;

 66 void OopMapStream::find_next() {
 67   if (_position++ < _size) {
 68     _omv.read_from(_stream);
 69     _valid_omv = true;
 70     return;
 71   }
 72   _valid_omv = false;
 73 }

 78 // frame_size units are stack-slots (4 bytes) NOT intptr_t; we can name odd
 79 // slots to hold 4-byte values like ints and floats in the LP64 build.
 80 OopMap::OopMap(int frame_size, int arg_count) {
 81   // OopMaps are usually quite so small, so pick a small initial size
 82   set_write_stream(new CompressedWriteStream(32));
 83   set_omv_count(0);
 84 
 85 #ifdef ASSERT
 86   _locs_length = VMRegImpl::stack2reg(0)->value() + frame_size + arg_count;
 87   _locs_used   = NEW_RESOURCE_ARRAY(OopMapValue::oop_types, _locs_length);
 88   for(int i = 0; i < _locs_length; i++) _locs_used[i] = OopMapValue::unused_value;
 89 #endif
 90 }

 93 OopMap::OopMap(OopMap::DeepCopyToken, OopMap* source) {
 94   // This constructor does a deep copy
 95   // of the source OopMap.
 96   set_write_stream(new CompressedWriteStream(source->omv_count() * 2));
 97   set_omv_count(0);
 98   set_offset(source->offset());
 99 
100 #ifdef ASSERT
101   _locs_length = source->_locs_length;
102   _locs_used = NEW_RESOURCE_ARRAY(OopMapValue::oop_types, _locs_length);
103   for(int i = 0; i < _locs_length; i++) _locs_used[i] = OopMapValue::unused_value;
104 #endif
105 
106   // We need to copy the entries too.
107   for (OopMapStream oms(source); !oms.is_done(); oms.next()) {
108     OopMapValue omv = oms.current();
109     omv.write_on(write_stream());
110     increment_count();
111   }
112 }

123 int OopMap::heap_size() const {
124   int size = sizeof(OopMap);
125   int align = sizeof(void *) - 1;
126   size += write_stream()->position();
127   // Align to a reasonable ending point
128   size = ((size+align) & ~align);
129   return size;

135

142   increment_count();

188 void OopMapSet::add_gc_map(int pc_offset, OopMap *map ) {
189   assert(om_size() != -1,"Cannot grow a fixed OopMapSet");
190 
191   if(om_count() >= om_size()) {
192     grow_om_data();
193   }
194   map->set_offset(pc_offset);
195 
196 #ifdef ASSERT
197   if(om_count() > 0) {
198     OopMap* last = at(om_count()-1);
199     if (last->offset() == map->offset() ) {
200       fatal("OopMap inserted twice");
201     }
202     if(last->offset() > map->offset()) {
203       tty->print_cr( "WARNING, maps not sorted: pc[%d]=%d, pc[%d]=%d",
204                       om_count(),last->offset(),om_count()+1,map->offset());
205     }
206   }
207 #endif // ASSERT
208 
209   set(om_count(),map);
210   increment_count();
211 }

214 int OopMapSet::heap_size() const {
215   // The space we use
216   int size = sizeof(OopMap);
217   int align = sizeof(void *) - 1;
218   size = ((size+align) & ~align);
219   size += om_count() * sizeof(OopMap*);
220 
221   // Now add in the space needed for the indivdiual OopMaps
222   for(int i=0; i < om_count(); i++) {
223     size += at(i)->heap_size();
224   }
225   // We don't need to align this, it will be naturally pointer aligned
226   return size;
227 }

236 OopMap* OopMapSet::find_map_at_offset(int pc_offset) const {
237   int i, len = om_count();
238   assert( len > 0, "must have pointer maps" );
239 
240   // Scan through oopmaps. Stop when current offset is either equal or greater
241   // than the one we are looking for.
242   for( i = 0; i < len; i++) {
243     if( at(i)->offset() >= pc_offset )
244       break;
245   }
246 
247   assert( i < len, "oopmap not found" );
248 
249   OopMap* m = at(i);
250   assert( m->offset() == pc_offset, "oopmap not found" );
251   return m;
252 }

298 void OopMapSet::all_do(const frame *fr, const RegisterMap *reg_map,
299                        OopClosure* oop_fn, void derived_oop_fn(oop*, oop*),
300                        OopClosure* value_fn) {
301   CodeBlob* cb = fr->cb();
302   assert(cb != NULL, "no codeblob");
303 
304   NOT_PRODUCT(if (TraceCodeBlobStacks) trace_codeblob_maps(fr, reg_map);)
305 
306   const ImmutableOopMapSet* maps = cb->oop_maps();
307   const ImmutableOopMap* map = cb->oop_map_for_return_address(fr->pc());
308   assert(map != NULL, "no ptr map found");
309 
310   // handle derived pointers first (otherwise base pointer may be
311   // changed before derived pointer offset has been collected)
312   {
313     for (OopMapStream oms(map); !oms.is_done(); oms.next()) {
314       OopMapValue omv = oms.current();
315       if (omv.type() != OopMapValue::derived_oop_value) {
316         continue;
317       }
318 
319 #ifndef TIERED
320       COMPILER1_PRESENT(ShouldNotReachHere();)
321 #if INCLUDE_JVMCI
322       if (UseJVMCICompiler) {
323         ShouldNotReachHere();
324       }
325 #endif
326 #endif // !TIERED
327       oop* loc = fr->oopmapreg_to_location(omv.reg(),reg_map);
328       guarantee(loc != NULL, "missing saved register");
329       oop *derived_loc = loc;
330       oop *base_loc    = fr->oopmapreg_to_location(omv.content_reg(), reg_map);
331       // Ignore NULL oops and decoded NULL narrow oops which
332       // equal to CompressedOops::base() when a narrow oop
333       // implicit null check is used in compiled code.
334       // The narrow_oop_base could be NULL or be the address
335       // of the page below heap depending on compressed oops mode.
336       if (base_loc != NULL && *base_loc != NULL && !CompressedOops::is_base(*base_loc)) {
337         derived_oop_fn(base_loc, derived_loc);
338       }
339     }
340   }
341 
342   {
343     // We want coop and oop oop_types
344     for (OopMapStream oms(map); !oms.is_done(); oms.next()) {
345       OopMapValue omv = oms.current();
346       oop* loc = fr->oopmapreg_to_location(omv.reg(),reg_map);
347       // It should be an error if no location can be found for a
348       // register mentioned as contained an oop of some kind.  Maybe
349       // this was allowed previously because value_value items might
350       // be missing?
351       guarantee(loc != NULL, "missing saved register");
352       if ( omv.type() == OopMapValue::oop_value ) {
353         oop val = *loc;
354         if (val == NULL || CompressedOops::is_base(val)) {
355           // Ignore NULL oops and decoded NULL narrow oops which
356           // equal to CompressedOops::base() when a narrow oop
357           // implicit null check is used in compiled code.
358           // The narrow_oop_base could be NULL or be the address
359           // of the page below heap depending on compressed oops mode.
360           continue;
361         }
362 #ifdef ASSERT
363         if ((((uintptr_t)loc & (sizeof(*loc)-1)) != 0) ||
364             !Universe::heap()->is_in_or_null(*loc)) {
365           tty->print_cr("# Found non oop pointer.  Dumping state at failure");
366           // try to dump out some helpful debugging information
367           trace_codeblob_maps(fr, reg_map);
368           omv.print();
369           tty->print_cr("register r");
370           omv.reg()->print();
371           tty->print_cr("loc = %p *loc = %p\n", loc, cast_from_oop<address>(*loc));
372           // do the real assert.
373           assert(Universe::heap()->is_in_or_null(*loc), "found non oop pointer");
374         }
375 #endif // ASSERT
376         oop_fn->do_oop(loc);
377       } else if ( omv.type() == OopMapValue::narrowoop_value ) {
378         narrowOop *nl = (narrowOop*)loc;
379 #ifndef VM_LITTLE_ENDIAN
380         VMReg vmReg = omv.reg();
381         // Don't do this on SPARC float registers as they can be individually addressed
382         if (!vmReg->is_stack() SPARC_ONLY(&& !vmReg->is_FloatRegister())) {
383           // compressed oops in registers only take up 4 bytes of an
384           // 8 byte register but they are in the wrong part of the
385           // word so adjust loc to point at the right place.
386           nl = (narrowOop*)((address)nl + 4);
387         }
388 #endif
389         oop_fn->do_oop(nl);
390       }
391     }
392   }
393 }

396 // Update callee-saved register info for the following frame
397 void OopMapSet::update_register_map(const frame *fr, RegisterMap *reg_map) {
398   ResourceMark rm;
399   CodeBlob* cb = fr->cb();
400   assert(cb != NULL, "no codeblob");
401 
402   // Any reg might be saved by a safepoint handler (see generate_handler_blob).
403   assert( reg_map->_update_for_id == NULL || fr->is_older(reg_map->_update_for_id),
404          "already updated this map; do not 'update' it twice!" );
405   debug_only(reg_map->_update_for_id = fr->id());
406 
407   // Check if caller must update oop argument
408   assert((reg_map->include_argument_oops() ||
409           !cb->caller_must_gc_arguments(reg_map->thread())),
410          "include_argument_oops should already be set");
411 
412   // Scan through oopmap and find location of all callee-saved registers
413   // (we do not do update in place, since info could be overwritten)
414 
415   address pc = fr->pc();
416   const ImmutableOopMap* map  = cb->oop_map_for_return_address(pc);
417   assert(map != NULL, "no ptr map found");
418   DEBUG_ONLY(int nof_callee = 0;)
419 
420   for (OopMapStream oms(map); !oms.is_done(); oms.next()) {
421     OopMapValue omv = oms.current();
422     if (omv.type() == OopMapValue::callee_saved_value) {
423       VMReg reg = omv.content_reg();
424       oop* loc = fr->oopmapreg_to_location(omv.reg(), reg_map);
425       reg_map->set_location(reg, (address) loc);
426       DEBUG_ONLY(nof_callee++;)
427     }
428   }
429 
430   // Check that runtime stubs save all callee-saved registers
431 #ifdef COMPILER2
432   assert(cb->is_compiled_by_c1() || cb->is_compiled_by_jvmci() || !cb->is_runtime_stub() ||
433          (nof_callee >= SAVED_ON_ENTRY_REG_COUNT || nof_callee >= C_SAVED_ON_ENTRY_REG_COUNT),
434          "must save all");
435 #endif // COMPILER2
436 }

535 bool OopMap::equals(const OopMap* other) const {
536   if (other->_omv_count != _omv_count) {
537     return false;
538   }
539   if (other->write_stream()->position() != write_stream()->position()) {
540     return false;
541   }
542   if (memcmp(other->write_stream()->buffer(), write_stream()->buffer(), write_stream()->position()) != 0) {
543     return false;
544   }
545   return true;
546 }

548 const ImmutableOopMap* ImmutableOopMapSet::find_map_at_offset(int pc_offset) const {
549   ImmutableOopMapPair* pairs = get_pairs();
550   ImmutableOopMapPair* last  = NULL;
551 
552   for (int i = 0; i < _count; ++i) {
553     if (pairs[i].pc_offset() >= pc_offset) {
554       last = &pairs[i];
555       break;
556     }
557   }
558 
559   // Heal Coverity issue: potential index out of bounds access.
560   guarantee(last != NULL, "last may not be null");
561   assert(last->pc_offset() == pc_offset, "oopmap not found");
562   return last->get_from(this);
563 }

593 int ImmutableOopMapBuilder::heap_size() {
594   int base = sizeof(ImmutableOopMapSet);
595   base = align_up(base, 8);
596 
597   // all of ours pc / offset pairs
598   int pairs = _set->size() * sizeof(ImmutableOopMapPair);
599   pairs = align_up(pairs, 8);
600 
601   for (int i = 0; i < _set->size(); ++i) {
602     int size = 0;
603     OopMap* map = _set->at(i);
604 
605     if (is_empty(map)) {
606       /* only keep a single empty map in the set */
607       if (has_empty()) {
608         _mapping[i].set(Mapping::OOPMAP_EMPTY, _empty_offset, 0, map, _empty);
609       } else {
610         _empty_offset = _offset;
611         _empty = map;
612         size = size_for(map);
613         _mapping[i].set(Mapping::OOPMAP_NEW, _offset, size, map);
614       }
615     } else if (is_last_duplicate(map)) {
616       /* if this entry is identical to the previous one, just point it there */
617       _mapping[i].set(Mapping::OOPMAP_DUPLICATE, _last_offset, 0, map, _last);
618     } else {
619       /* not empty, not an identical copy of the previous entry */
620       size = size_for(map);
621       _mapping[i].set(Mapping::OOPMAP_NEW, _offset, size, map);
622       _last_offset = _offset;
623       _last = map;
624     }
625 
626     assert(_mapping[i]._map == map, "check");
627     _offset += size;
628   }
629 
630   int total = base + pairs + _offset;
631   DEBUG_ONLY(total += 8);
632   _required = total;
633   return total;
634 }
635 
636 void ImmutableOopMapBuilder::fill_pair(ImmutableOopMapPair* pair, const OopMap* map, int offset, const ImmutableOopMapSet* set) {
637   assert(offset < set->nr_of_bytes(), "check");
638   new ((address) pair) ImmutableOopMapPair(map->offset(), offset);
639 }
640 
641 int ImmutableOopMapBuilder::fill_map(ImmutableOopMapPair* pair, const OopMap* map, int offset, const ImmutableOopMapSet* set) {
642   fill_pair(pair, map, offset, set);
643   address addr = (address) pair->get_from(_new_set); // location of the ImmutableOopMap
644 
645   new (addr) ImmutableOopMap(map);
646   return size_for(map);
647 }
648 
649 void ImmutableOopMapBuilder::fill(ImmutableOopMapSet* set, int sz) {
650   ImmutableOopMapPair* pairs = set->get_pairs();
651 
652   for (int i = 0; i < set->count(); ++i) {
653     const OopMap* map = _mapping[i]._map;
654     ImmutableOopMapPair* pair = NULL;
655     int size = 0;
656 
657     if (_mapping[i]._kind == Mapping::OOPMAP_NEW) {
658       size = fill_map(&pairs[i], map, _mapping[i]._offset, set);
659     } else if (_mapping[i]._kind == Mapping::OOPMAP_DUPLICATE || _mapping[i]._kind == Mapping::OOPMAP_EMPTY) {
660       fill_pair(&pairs[i], map, _mapping[i]._offset, set);
661     }
662 
663     const ImmutableOopMap* nv = set->find_map_at_offset(map->offset());
664     assert(memcmp(map->data(), nv->data_addr(), map->data_size()) == 0, "check identity");
665   }
666 }
667 
668 #ifdef ASSERT
669 void ImmutableOopMapBuilder::verify(address buffer, int size, const ImmutableOopMapSet* set) {
670   for (int i = 0; i < 8; ++i) {
671     assert(buffer[size - 8 + i] == (unsigned char) 0xff, "overwritten memory check");
672   }
673 
674   for (int i = 0; i < set->count(); ++i) {
675     const ImmutableOopMapPair* pair = set->pair_at(i);
676     assert(pair->oopmap_offset() < set->nr_of_bytes(), "check size");
677     const ImmutableOopMap* map = pair->get_from(set);
678     int nr_of_bytes = map->nr_of_bytes();
679     assert(pair->oopmap_offset() + nr_of_bytes <= set->nr_of_bytes(), "check size + size");
680   }
681 }
682 #endif
683 
684 ImmutableOopMapSet* ImmutableOopMapBuilder::generate_into(address buffer) {
685   DEBUG_ONLY(memset(&buffer[_required-8], 0xff, 8));
686 
687   _new_set = new (buffer) ImmutableOopMapSet(_set, _required);
688   fill(_new_set, _required);
689 
690   DEBUG_ONLY(verify(buffer, _required, _new_set));
691 
692   return _new_set;
693 }
694 
695 ImmutableOopMapSet* ImmutableOopMapBuilder::build() {
696   _required = heap_size();
697 
698   // We need to allocate a chunk big enough to hold the ImmutableOopMapSet and all of its ImmutableOopMaps
699   address buffer = NEW_C_HEAP_ARRAY(unsigned char, _required, mtCode);
700   return generate_into(buffer);
701 }
702 
703 ImmutableOopMapSet* ImmutableOopMapSet::build_from(const OopMapSet* oopmap_set) {
704   ResourceMark mark;
705   ImmutableOopMapBuilder builder(oopmap_set);
706   return builder.build();
707 }

714 class DerivedPointerTable::Entry : public CHeapObj<mtCompiler> {
715   oop* _location;   // Location of derived pointer, also pointing to base
716   intptr_t _offset; // Offset from base pointer
717   Entry* volatile _next;
718 
719   static Entry* volatile* next_ptr(Entry& entry) { return &entry._next; }
720 
721 public:
722   Entry(oop* location, intptr_t offset) :
723     _location(location), _offset(offset), _next(NULL) {}
724 
725   oop* location() const { return _location; }
726   intptr_t offset() const { return _offset; }
727   Entry* next() const { return _next; }
728 
729   typedef LockFreeStack<Entry, &next_ptr> List;
730   static List* _list;
731 };
732 
733 DerivedPointerTable::Entry::List* DerivedPointerTable::Entry::_list = NULL;
734 bool DerivedPointerTable::_active = false;
735 
736 bool DerivedPointerTable::is_empty() {
737   return Entry::_list == NULL || Entry::_list->empty();
738 }
739 
740 void DerivedPointerTable::clear() {
741   // The first time, we create the list.  Otherwise it should be
742   // empty.  If not, then we have probably forgotton to call
743   // update_pointers after last GC/Scavenge.
744   assert (!_active, "should not be active");
745   assert(is_empty(), "table not empty");
746   if (Entry::_list == NULL) {
747     void* mem = NEW_C_HEAP_OBJ(Entry::List, mtCompiler);
748     Entry::_list = ::new (mem) Entry::List();
749   }
750   _active = true;
751 }
752 
753 // Returns value of location as an int
754 inline intptr_t value_of_loc(oop *pointer) {
755   return cast_from_oop<intptr_t>((*pointer));
756 }
757 
758 void DerivedPointerTable::add(oop *derived_loc, oop *base_loc) {
759   assert(Universe::heap()->is_in_or_null(*base_loc), "not an oop");
760   assert(derived_loc != base_loc, "Base and derived in same location");
761   if (_active) {
762     assert(*derived_loc != (void*)base_loc, "location already added");
763     assert(Entry::_list != NULL, "list must exist");
764     intptr_t offset = value_of_loc(derived_loc) - value_of_loc(base_loc);
765     // This assert is invalid because derived pointers can be
766     // arbitrarily far away from their base.
767     // assert(offset >= -1000000, "wrong derived pointer info");
768 
769     if (TraceDerivedPointers) {
770       tty->print_cr(
771         "Add derived pointer@" INTPTR_FORMAT
772         " - Derived: " INTPTR_FORMAT
773         " Base: " INTPTR_FORMAT " (@" INTPTR_FORMAT ") (Offset: " INTX_FORMAT ")",
774         p2i(derived_loc), p2i(*derived_loc), p2i(*base_loc), p2i(base_loc), offset
775       );
776     }
777     // Set derived oop location to point to base.
778     *derived_loc = (oop)base_loc;
779     Entry* entry = new Entry(derived_loc, offset);
780     Entry::_list->push(*entry);
781   }
782 }
783 
784 void DerivedPointerTable::update_pointers() {
785   assert(Entry::_list != NULL, "list must exist");
786   Entry* entries = Entry::_list->pop_all();
787   while (entries != NULL) {
788     Entry* entry = entries;
789     entries = entry->next();
790     oop* derived_loc = entry->location();
791     intptr_t offset  = entry->offset();
792     // The derived oop was setup to point to location of base
793     oop base = **(oop**)derived_loc;
794     assert(Universe::heap()->is_in_or_null(base), "must be an oop");
795 
796     *derived_loc = (oop)(cast_from_oop<address>(base) + offset);
797     assert(value_of_loc(derived_loc) - value_of_loc(&base) == offset, "sanity check");
798 
799     if (TraceDerivedPointers) {
800       tty->print_cr("Updating derived pointer@" INTPTR_FORMAT
801                     " - Derived: " INTPTR_FORMAT "  Base: " INTPTR_FORMAT " (Offset: " INTX_FORMAT ")",
802           p2i(derived_loc), p2i(*derived_loc), p2i(base), offset);
803     }
804 
805     // Delete entry
806     delete entry;
807   }
808   assert(Entry::_list->empty(), "invariant");
809   _active = false;
810 }
