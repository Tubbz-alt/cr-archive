 36 // Periodic task is useful for doing asynchronous things that do not require (heap) locks,
 37 // or synchronization with other parts of collector. These could run even when ShenandoahConcurrentThread
 38 // is busy driving the GC cycle.
 39 class ShenandoahPeriodicTask : public PeriodicTask {
 40 private:
 41   ShenandoahControlThread* _thread;
 42 public:
 43   ShenandoahPeriodicTask(ShenandoahControlThread* thread) :
 44           PeriodicTask(100), _thread(thread) {}
 45   virtual void task();
 46 };
 47 
 48 // Periodic task to flush SATB buffers periodically.
 49 class ShenandoahPeriodicSATBFlushTask : public PeriodicTask {
 50 public:
 51   ShenandoahPeriodicSATBFlushTask() : PeriodicTask(ShenandoahSATBBufferFlushInterval) {}
 52   virtual void task();
 53 };
 54 
 55 class ShenandoahControlThread: public ConcurrentGCThread {
 56   friend class VMStructs;
 57 
 58 private:
 59   typedef enum {
 60     none,
 61     concurrent_traversal,
 62     concurrent_normal,
 63     stw_degenerated,
 64     stw_full
 65   } GCMode;
 66 
 67   // While we could have a single lock for these, it may risk unblocking
 68   // GC waiters when alloc failure GC cycle finishes. We want instead
 69   // to make complete explicit cycle for for demanding customers.
 70   Monitor _alloc_failure_waiters_lock;
 71   Monitor _gc_waiters_lock;
 72   ShenandoahPeriodicTask _periodic_task;
 73   ShenandoahPeriodicSATBFlushTask _periodic_satb_flush_task;
 74 
 75 public:
 76   void run_service();
 77   void stop_service();
 78 
 79 private:
 80   ShenandoahSharedFlag _gc_requested;
 81   ShenandoahSharedFlag _alloc_failure_gc;
 82   ShenandoahSharedFlag _graceful_shutdown;
 83   ShenandoahSharedFlag _heap_changed;
 84   ShenandoahSharedFlag _do_counters_update;
 85   ShenandoahSharedFlag _force_counters_update;
 86   GCCause::Cause       _requested_gc_cause;
 87   ShenandoahHeap::ShenandoahDegenPoint _degen_point;
 88 
 89   shenandoah_padding(0);
 90   volatile size_t _allocs_seen;
 91   shenandoah_padding(1);
 92 
 93   bool check_cancellation_or_degen(ShenandoahHeap::ShenandoahDegenPoint point);
 94   void service_concurrent_normal_cycle(GCCause::Cause cause);
 95   void service_stw_full_cycle(GCCause::Cause cause);
 96   void service_stw_degenerated_cycle(GCCause::Cause cause, ShenandoahHeap::ShenandoahDegenPoint point);
 97   void service_concurrent_traversal_cycle(GCCause::Cause cause);
 98   void service_uncommit(double shrink_before);
 99 
100   bool try_set_alloc_failure_gc();
101   void notify_alloc_failure_waiters();
102   bool is_alloc_failure_gc();
103 
104   void notify_gc_waiters();
105 
106   // Handle GC request.
107   // Blocks until GC is over.
108   void handle_requested_gc(GCCause::Cause cause);
109 
110   bool is_explicit_gc(GCCause::Cause cause) const;
111 public:
112   // Constructor
113   ShenandoahControlThread();
114   ~ShenandoahControlThread();
115 
116   // Handle allocation failure from normal allocation.
117   // Blocks until memory is available.
118   void handle_alloc_failure(size_t words);
119 
120   // Handle allocation failure from evacuation path.
121   // Optionally blocks while collector is handling the failure.
122   void handle_alloc_failure_evac(size_t words);
123 
124   void request_gc(GCCause::Cause cause);
125 
126   void handle_counters_update();
127   void handle_force_counters_update();
128   void set_forced_counters_update(bool value);
129 
130   void notify_heap_changed();
131 
132   void pacing_notify_alloc(size_t words);
133 
134   void start();
135   void prepare_for_graceful_shutdown();
136   bool in_graceful_shutdown();
137 
138   char* name() const { return (char*)"ShenandoahControlThread";}
139 
140   // Printing
141   void print_on(outputStream* st) const;
142   void print() const;
143 };

145 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHCONTROLTHREAD_HPP
