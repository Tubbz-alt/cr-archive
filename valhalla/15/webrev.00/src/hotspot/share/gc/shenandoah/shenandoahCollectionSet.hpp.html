 25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHCOLLECTIONSET_HPP
 26 #define SHARE_GC_SHENANDOAH_SHENANDOAHCOLLECTIONSET_HPP
 27 
 28 #include "memory/allocation.hpp"
 29 #include "gc/shenandoah/shenandoahHeap.hpp"
 30 #include "gc/shenandoah/shenandoahHeapRegion.hpp"
 31 #include "gc/shenandoah/shenandoahPadding.hpp"
 32 
 33 class ShenandoahCollectionSet : public CHeapObj<mtGC> {
 34   friend class ShenandoahHeap;
 35 private:
 36   size_t const          _map_size;
 37   size_t const          _region_size_bytes_shift;
 38   ReservedSpace         _map_space;
 39   char* const           _cset_map;
 40   // Bias cset map's base address for fast test if an oop is in cset
 41   char* const           _biased_cset_map;
 42 
 43   ShenandoahHeap* const _heap;
 44 
 45   size_t                _garbage;
 46   size_t                _live_data;
 47   size_t                _used;
 48   size_t                _region_count;
 49 
 50   shenandoah_padding(0);
 51   volatile jint         _current_index;
 52   shenandoah_padding(1);
 53 
 54 public:
 55   ShenandoahCollectionSet(ShenandoahHeap* heap, char* heap_base, size_t size);
 56 
 57   // Add region to collection set
 58   void add_region(ShenandoahHeapRegion* r);
 59   bool add_region_check_for_duplicates(ShenandoahHeapRegion* r);
 60 
 61   // Bring per-region statuses to consistency with this collection.
 62   // TODO: This is a transitional interface that bridges the gap between
 63   // region statuses and this collection. Should go away after we merge them.
 64   void update_region_status();
 65 
 66   // Remove region from collection set
 67   void remove_region(ShenandoahHeapRegion* r);
 68 
 69   // MT version
 70   ShenandoahHeapRegion* claim_next();
 71 
 72   // Single-thread version
 73   ShenandoahHeapRegion* next();
 74 
 75   size_t count()  const { return _region_count; }
 76   bool is_empty() const { return _region_count == 0; }
 77 
 78   void clear_current_index() {
 79     _current_index = 0;
 80   }
 81 
 82   inline bool is_in(ShenandoahHeapRegion* r) const;
 83   inline bool is_in(size_t region_number)    const;
 84   inline bool is_in(oop obj)                 const;
 85   inline bool is_in_loc(void* loc)           const;
 86 
 87   void print_on(outputStream* out) const;
 88 
 89   size_t used()      const { return _used; }
 90   size_t live_data() const { return _live_data; }
 91   size_t garbage()   const { return _garbage;   }
 92   void clear();
 93 
 94 private:
 95   char* map_address() const {
 96     return _cset_map;
 97   }
 98   char* biased_map_address() const {
 99     return _biased_cset_map;
100   }
101 };
102 
103 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHCOLLECTIONSET_HPP
