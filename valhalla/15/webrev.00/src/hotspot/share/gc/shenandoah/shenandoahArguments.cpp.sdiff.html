186   // TLAB sizing policy makes resizing decisions before each GC cycle. It averages
187   // historical data, assigning more recent data the weight according to TLABAllocationWeight.
188   // Current default is good for generational collectors that run frequent young GCs.
189   // With Shenandoah, GC cycles are much less frequent, so we need we need sizing policy
190   // to converge faster over smaller number of resizing decisions.
191   if (FLAG_IS_DEFAULT(TLABAllocationWeight)) {
192     FLAG_SET_DEFAULT(TLABAllocationWeight, 90);
193   }
194 
195   // Make sure safepoint deadlocks are failing predictably. This sets up VM to report
196   // fatal error after 10 seconds of wait for safepoint syncronization (not the VM
197   // operation itself). There is no good reason why Shenandoah would spend that
198   // much time synchronizing.
199 #ifdef ASSERT
200   FLAG_SET_DEFAULT(SafepointTimeout, true);
201   FLAG_SET_DEFAULT(SafepointTimeoutDelay, 10000);
202   FLAG_SET_DEFAULT(AbortVMOnSafepointTimeout, true);
203 #endif

209     align = MAX2(align, os::large_page_size());
210   }
211   return align;
