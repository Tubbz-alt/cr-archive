 33 const Type* SubTypeCheckNode::sub(const Type* sub_t, const Type* super_t) const {
 34   ciKlass* superk = super_t->is_klassptr()->klass();
 35   ciKlass* subk   = sub_t->isa_klassptr() ? sub_t->is_klassptr()->klass() : sub_t->is_oopptr()->klass();
 36 
 37   bool xsubk = sub_t->isa_klassptr() ? sub_t->is_klassptr()->klass_is_exact() : sub_t->is_oopptr()->klass_is_exact();
 38 









 39   // Similar to logic in CmpPNode::sub()





 40   if (superk && subk &&
 41       superk->is_loaded() && !superk->is_interface() &&
 42       subk->is_loaded() && !subk->is_interface() &&
 43       (!superk->is_obj_array_klass() ||
 44        !superk->as_obj_array_klass()->base_element_klass()->is_interface()) &&
 45       (!subk->is_obj_array_klass() ||
 46        !subk->as_obj_array_klass()->base_element_klass()->is_interface() ||
 47        xsubk)) {
 48     bool unrelated_classes = false;
 49     if (superk->equals(subk)) {
 50       // skip
 51     } else if (superk->is_subtype_of(subk)) {



 52       unrelated_classes = xsubk;
 53     } else if (subk->is_subtype_of(superk)) {
 54       // skip
 55     } else {


 56       unrelated_classes = true;
 57     }
 58     // Ignore exactness of constant supertype (the type of the corresponding object may be non-exact).
 59     const TypeKlassPtr* casted_sup = super_t->is_klassptr()->cast_to_exactness(false)->is_klassptr();
 60     if (sub_t->is_ptr()->flat_array() && (!casted_sup->can_be_value_type() || (superk->is_valuetype() && !superk->flatten_array()))) {
 61       // Subtype is flattened in arrays but supertype is not. Must be unrelated.
 62       unrelated_classes = true;
 63     }
 64     if (unrelated_classes) {
 65       TypePtr::PTR jp = sub_t->is_ptr()->join_ptr(super_t->is_ptr()->_ptr);
 66       if (jp != TypePtr::Null && jp != TypePtr::BotPTR) {
 67         return TypeInt::CC_GT;
 68       }
 69     }
 70   }
 71 
 72   if (super_t->singleton()) {
 73     if (subk != NULL) {
 74       switch (Compile::current()->static_subtype_check(superk, subk)) {
 75       case Compile::SSC_always_false:
 76         return TypeInt::CC_GT;
 77       case Compile::SSC_always_true:
 78         return TypeInt::CC_EQ;
 79       case Compile::SSC_easy_test:
 80       case Compile::SSC_full_test:
 81         break;
 82       default:
 83         ShouldNotReachHere();
 84       }
 85     }
 86   }
 87 
 88   return bottom_type();
 89 }

 91 Node *SubTypeCheckNode::Ideal(PhaseGVN *phase, bool can_reshape) {
 92   // Verify that optimizing the subtype check to a simple code pattern
 93   // when possible would not constant fold better
 94 #ifdef ASSERT
 95   Node* obj_or_subklass = in(ObjOrSubKlass);
 96   Node* superklass = in(SuperKlass);
 97 
 98   if (obj_or_subklass == NULL ||
 99       superklass == NULL) {
100     return NULL;
101   }
102 
103   const Type* sub_t = phase->type(obj_or_subklass);
104   const Type* super_t = phase->type(superklass);
105 
106   if (!super_t->isa_klassptr() ||
107       (!sub_t->isa_klassptr() && !sub_t->isa_oopptr())) {
108     return NULL;
109   }
110 


















111 



112   ciKlass* superk = super_t->is_klassptr()->klass();
113   ciKlass* subk   = sub_t->isa_klassptr() ? sub_t->is_klassptr()->klass() : sub_t->is_oopptr()->klass();
114 
115   if (super_t->singleton() && subk != NULL && phase->C->static_subtype_check(superk, subk) == Compile::SSC_easy_test) {
116     Node* subklass = NULL;
117     if (sub_t->isa_oopptr()) {
118       Node* adr = phase->transform(new AddPNode(obj_or_subklass, obj_or_subklass, phase->MakeConX(oopDesc::klass_offset_in_bytes())));
119       subklass = phase->transform(LoadKlassNode::make(*phase, NULL, phase->C->immutable_memory(), adr, TypeInstPtr::KLASS));
120     } else {
121       subklass = obj_or_subklass;
122     }
123     Node* res = new CmpPNode(subklass, superklass);
124     const Type* t = phase->type(phase->transform(res));
125     assert((Value(phase) == t) || (t != TypeInt::CC_GT && t != TypeInt::CC_EQ), "missing Value() optimization");
126     if (phase->is_IterGVN()) {
127       phase->is_IterGVN()->_worklist.push(res);
128     }
129     return NULL;
130   }
131 
132   if (super_t->singleton() && subk != NULL && phase->C->static_subtype_check(superk, subk) == Compile::SSC_full_test) {
133     Node* subklass = NULL;
134     if (sub_t->isa_oopptr()) {
135       Node* adr = phase->transform(new AddPNode(obj_or_subklass, obj_or_subklass, phase->MakeConX(oopDesc::klass_offset_in_bytes())));
136       subklass = phase->transform(LoadKlassNode::make(*phase, NULL, phase->C->immutable_memory(), adr, TypeInstPtr::KLASS));
137     } else {
138       subklass = obj_or_subklass;
139     }
140 
141     Node *p1 = phase->transform(new AddPNode(superklass, superklass, phase->MakeConX(in_bytes(Klass::super_check_offset_offset()))));
142     Node* m = phase->C->immutable_memory();
143     Node *chk_off = phase->transform(new LoadINode(NULL, m, p1, phase->type(p1)->is_ptr(), TypeInt::INT, MemNode::unordered));
144     int cacheoff_con = in_bytes(Klass::secondary_super_cache_offset());
145     bool might_be_cache = (phase->find_int_con(chk_off, cacheoff_con) == cacheoff_con);
146 
147     if (might_be_cache) {
148       return NULL;
149     }
150 
151     Node *chk_off_X = chk_off;
152 #ifdef _LP64
153     chk_off_X = phase->transform(new ConvI2LNode(chk_off_X));
154 #endif
155     Node *p2 = phase->transform(new AddPNode(subklass,subklass,chk_off_X));
156     Node *kmem = phase->C->immutable_memory();
157     Node *nkls = phase->transform(LoadKlassNode::make(*phase, NULL, kmem, p2, phase->type(p2)->is_ptr(), TypeKlassPtr::OBJECT_OR_NULL));
158 
159     Node* res = new CmpPNode(superklass, nkls);
160     const Type* t = phase->type(phase->transform(res));
161     assert((Value(phase) == t) || (t != TypeInt::CC_GT && t != TypeInt::CC_EQ), "missing Value() optimization");
162     if (phase->is_IterGVN()) {
163       phase->is_IterGVN()->_worklist.push(res);
164     }
165     return NULL;
166   }
167 #endif
168 
169   return NULL;
170 }
