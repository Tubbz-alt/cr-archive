  55 const Type::Offset Type::Offset::meet(const Type::Offset other) const {
  56   // Either is 'TOP' offset?  Return the other offset!
  57   int offset = other._offset;
  58   if (_offset == OffsetTop) return Offset(offset);
  59   if (offset == OffsetTop) return Offset(_offset);
  60   // If either is different, return 'BOTTOM' offset
  61   if (_offset != offset) return bottom;
  62   return Offset(_offset);
  63 }

  65 const Type::Offset Type::Offset::dual() const {
  66   if (_offset == OffsetTop) return bottom;// Map 'TOP' into 'BOTTOM'
  67   if (_offset == OffsetBot) return top;// Map 'BOTTOM' into 'TOP'
  68   return Offset(_offset);               // Map everything else into self
  69 }

  71 const Type::Offset Type::Offset::add(intptr_t offset) const {
  72   // Adding to 'TOP' offset?  Return 'TOP'!
  73   if (_offset == OffsetTop || offset == OffsetTop) return top;
  74   // Adding to 'BOTTOM' offset?  Return 'BOTTOM'!
  75   if (_offset == OffsetBot || offset == OffsetBot) return bottom;
  76   // Addition overflows or "accidentally" equals to OffsetTop? Return 'BOTTOM'!
  77   offset += (intptr_t)_offset;
  78   if (offset != (int)offset || offset == OffsetTop) return bottom;
  79 
  80   // assert( _offset >= 0 && _offset+offset >= 0, "" );
  81   // It is possible to construct a negative offset during PhaseCCP
  82 
  83   return Offset((int)offset);        // Sum valid offsets
  84 }

  87   if (_offset == 0) {
  88     return;
  89   } else if (_offset == OffsetTop) {

  91   }
  92   else if (_offset == OffsetBot) {
  93     st->print("+bot");
  94   } else if (_offset) {

 182 //------------------------------get_const_type---------------------------
 183 const Type* Type::get_const_type(ciType* type) {
 184   if (type == NULL) {
 185     return NULL;
 186   } else if (type->is_primitive_type()) {
 187     return get_const_basic_type(type->basic_type());
 188   } else {
 189     return TypeOopPtr::make_from_klass(type->as_klass());
 190   }

 193 //---------------------------array_element_basic_type---------------------------------
 194 // Mapping to the array element's basic type.
 195 BasicType Type::array_element_basic_type() const {
 196   BasicType bt = basic_type();
 197   if (bt == T_INT) {
 198     if (this == TypeInt::INT)   return T_INT;
 199     if (this == TypeInt::CHAR)  return T_CHAR;
 200     if (this == TypeInt::BYTE)  return T_BYTE;
 201     if (this == TypeInt::BOOL)  return T_BOOLEAN;
 202     if (this == TypeInt::SHORT) return T_SHORT;
 203     return T_VOID;
 204   }
 205   return bt;
 206 }

 208 // For two instance arrays of same dimension, return the base element types.
 209 // Otherwise or if the arrays have different dimensions, return NULL.
 210 void Type::get_arrays_base_elements(const Type *a1, const Type *a2,
 211                                     const TypeInstPtr **e1, const TypeInstPtr **e2) {
 212 
 213   if (e1) *e1 = NULL;
 214   if (e2) *e2 = NULL;
 215   const TypeAryPtr* a1tap = (a1 == NULL) ? NULL : a1->isa_aryptr();
 216   const TypeAryPtr* a2tap = (a2 == NULL) ? NULL : a2->isa_aryptr();
 217 
 218   if (a1tap != NULL && a2tap != NULL) {
 219     // Handle multidimensional arrays
 220     const TypePtr* a1tp = a1tap->elem()->make_ptr();
 221     const TypePtr* a2tp = a2tap->elem()->make_ptr();
 222     while (a1tp && a1tp->isa_aryptr() && a2tp && a2tp->isa_aryptr()) {
 223       a1tap = a1tp->is_aryptr();
 224       a2tap = a2tp->is_aryptr();
 225       a1tp = a1tap->elem()->make_ptr();
 226       a2tp = a2tap->elem()->make_ptr();
 227     }
 228     if (a1tp && a1tp->isa_instptr() && a2tp && a2tp->isa_instptr()) {
 229       if (e1) *e1 = a1tp->is_instptr();
 230       if (e2) *e2 = a2tp->is_instptr();
 231     }
 232   }
 233 }

 235 //---------------------------get_typeflow_type---------------------------------
 236 // Import a type produced by ciTypeFlow.
 237 const Type* Type::get_typeflow_type(ciType* type) {
 238   switch (type->basic_type()) {
 239 
 240   case ciTypeFlow::StateVector::T_BOTTOM:
 241     assert(type == ciTypeFlow::StateVector::bottom_type(), "");
 242     return Type::BOTTOM;
 243 
 244   case ciTypeFlow::StateVector::T_TOP:
 245     assert(type == ciTypeFlow::StateVector::top_type(), "");
 246     return Type::TOP;
 247 
 248   case ciTypeFlow::StateVector::T_NULL:
 249     assert(type == ciTypeFlow::StateVector::null_type(), "");
 250     return TypePtr::NULL_PTR;
 251 
 252   case ciTypeFlow::StateVector::T_LONG2:
 253     // The ciTypeFlow pass pushes a long, then the half.
 254     // We do the same.
 255     assert(type == ciTypeFlow::StateVector::long2_type(), "");
 256     return TypeInt::TOP;
 257 
 258   case ciTypeFlow::StateVector::T_DOUBLE2:
 259     // The ciTypeFlow pass pushes double, then the half.
 260     // Our convention is the same.
 261     assert(type == ciTypeFlow::StateVector::double2_type(), "");
 262     return Type::TOP;
 263 
 264   case T_ADDRESS:
 265     assert(type->is_return_address(), "");
 266     return TypeRawPtr::make((address)(intptr_t)type->as_return_address()->bci());
 267 
 268   case T_VALUETYPE: {
 269     bool is_never_null = type->is_never_null();
 270     ciValueKlass* vk = type->unwrap()->as_value_klass();
 271     if (vk->is_scalarizable() && is_never_null) {
 272       return TypeValueType::make(vk);
 273     } else {
 274       return TypeOopPtr::make_from_klass(vk)->join_speculative(is_never_null ? TypePtr::NOTNULL : TypePtr::BOTTOM);
 275     }
 276   }
 277 
 278   default:
 279     // make sure we did not mix up the cases:
 280     assert(type != ciTypeFlow::StateVector::bottom_type(), "");
 281     assert(type != ciTypeFlow::StateVector::top_type(), "");
 282     assert(type != ciTypeFlow::StateVector::null_type(), "");
 283     assert(type != ciTypeFlow::StateVector::long2_type(), "");
 284     assert(type != ciTypeFlow::StateVector::double2_type(), "");
 285     assert(!type->is_return_address(), "");
 286 
 287     return Type::get_const_type(type);
 288   }
 289 }

 292 //-----------------------make_from_constant------------------------------------
 293 const Type* Type::make_from_constant(ciConstant constant, bool require_constant,
 294                                      int stable_dimension, bool is_narrow_oop,
 295                                      bool is_autobox_cache) {
 296   switch (constant.basic_type()) {
 297     case T_BOOLEAN:  return TypeInt::make(constant.as_boolean());
 298     case T_CHAR:     return TypeInt::make(constant.as_char());
 299     case T_BYTE:     return TypeInt::make(constant.as_byte());
 300     case T_SHORT:    return TypeInt::make(constant.as_short());
 301     case T_INT:      return TypeInt::make(constant.as_int());
 302     case T_LONG:     return TypeLong::make(constant.as_long());
 303     case T_FLOAT:    return TypeF::make(constant.as_float());
 304     case T_DOUBLE:   return TypeD::make(constant.as_double());
 305     case T_ARRAY:
 306     case T_VALUETYPE:
 307     case T_OBJECT: {
 308         const Type* con_type = NULL;
 309         ciObject* oop_constant = constant.as_object();
 310         if (oop_constant->is_null_object()) {
 311           con_type = Type::get_zero_type(T_OBJECT);
 312         } else {
 313           guarantee(require_constant || oop_constant->should_be_constant(), "con_type must get computed");
 314           con_type = TypeOopPtr::make_from_constant(oop_constant, require_constant);
 315           if (Compile::current()->eliminate_boxing() && is_autobox_cache) {
 316             con_type = con_type->is_aryptr()->cast_to_autobox_cache(true);
 317           }
 318           if (stable_dimension > 0) {
 319             assert(FoldStableValues, "sanity");
 320             assert(!con_type->is_zero_type(), "default value for stable field");
 321             con_type = con_type->is_aryptr()->cast_to_stable(true, stable_dimension);
 322           }
 323         }
 324         if (is_narrow_oop) {
 325           con_type = con_type->make_narrowoop();
 326         }
 327         return con_type;
 328       }
 329     case T_ILLEGAL:
 330       // Invalid ciConstant returned due to OutOfMemoryError in the CI
 331       assert(Compile::current()->env()->failing(), "otherwise should not see this");
 332       return NULL;
 333     default:
 334       // Fall through to failure
 335       return NULL;
 336   }
 337 }

 339 static ciConstant check_mismatched_access(ciConstant con, BasicType loadbt, bool is_unsigned) {
 340   BasicType conbt = con.basic_type();
 341   switch (conbt) {
 342     case T_BOOLEAN: conbt = T_BYTE;   break;
 343     case T_ARRAY:   conbt = T_OBJECT; break;
 344     case T_VALUETYPE: conbt = T_OBJECT; break;
 345     default:                          break;
 346   }
 347   switch (loadbt) {
 348     case T_BOOLEAN:   loadbt = T_BYTE;   break;
 349     case T_NARROWOOP: loadbt = T_OBJECT; break;
 350     case T_ARRAY:     loadbt = T_OBJECT; break;
 351     case T_VALUETYPE: loadbt = T_OBJECT; break;
 352     case T_ADDRESS:   loadbt = T_OBJECT; break;
 353     default:                             break;
 354   }
 355   if (conbt == loadbt) {
 356     if (is_unsigned && conbt == T_BYTE) {
 357       // LoadB (T_BYTE) with a small mask (<=8-bit) is converted to LoadUB (T_BYTE).
 358       return ciConstant(T_INT, con.as_int() & 0xFF);
 359     } else {
 360       return con;
 361     }
 362   }
 363   if (conbt == T_SHORT && loadbt == T_CHAR) {
 364     // LoadS (T_SHORT) with a small mask (<=16-bit) is converted to LoadUS (T_CHAR).
 365     return ciConstant(T_INT, con.as_int() & 0xFFFF);
 366   }
 367   return ciConstant(); // T_ILLEGAL
 368 }

 370 // Try to constant-fold a stable array element.
 371 const Type* Type::make_constant_from_array_element(ciArray* array, int off, int stable_dimension,
 372                                                    BasicType loadbt, bool is_unsigned_load) {
 373   // Decode the results of GraphKit::array_element_address.
 374   ciConstant element_value = array->element_value_by_offset(off);
 375   if (element_value.basic_type() == T_ILLEGAL) {
 376     return NULL; // wrong offset
 377   }
 378   ciConstant con = check_mismatched_access(element_value, loadbt, is_unsigned_load);
 379 
 380   assert(con.basic_type() != T_ILLEGAL, "elembt=%s; loadbt=%s; unsigned=%d",
 381          type2name(element_value.basic_type()), type2name(loadbt), is_unsigned_load);
 382 
 383   if (con.is_valid() &&          // not a mismatched access
 384       !con.is_null_or_zero()) {  // not a default value
 385     bool is_narrow_oop = (loadbt == T_NARROWOOP);
 386     return Type::make_from_constant(con, /*require_constant=*/true, stable_dimension, is_narrow_oop, /*is_autobox_cache=*/false);
 387   }
 388   return NULL;
 389 }

 391 const Type* Type::make_constant_from_field(ciInstance* holder, int off, bool is_unsigned_load, BasicType loadbt) {
 392   ciField* field;
 393   ciType* type = holder->java_mirror_type();
 394   if (type != NULL && type->is_instance_klass() && off >= InstanceMirrorKlass::offset_of_static_fields()) {
 395     // Static field
 396     field = type->as_instance_klass()->get_field_by_offset(off, /*is_static=*/true);
 397   } else {
 398     // Instance field
 399     field = holder->klass()->as_instance_klass()->get_field_by_offset(off, /*is_static=*/false);
 400   }
 401   if (field == NULL) {
 402     return NULL; // Wrong offset
 403   }
 404   return Type::make_constant_from_field(field, holder, loadbt, is_unsigned_load);
 405 }

 407 const Type* Type::make_constant_from_field(ciField* field, ciInstance* holder,
 408                                            BasicType loadbt, bool is_unsigned_load) {
 409   if (!field->is_constant()) {
 410     return NULL; // Non-constant field
 411   }
 412   ciConstant field_value;
 413   if (field->is_static()) {
 414     // final static field
 415     field_value = field->constant_value();
 416   } else if (holder != NULL) {
 417     // final or stable non-static field
 418     // Treat final non-static fields of trusted classes (classes in
 419     // java.lang.invoke and sun.invoke packages and subpackages) as
 420     // compile time constants.
 421     field_value = field->constant_value_of(holder);
 422   }
 423   if (!field_value.is_valid()) {
 424     return NULL; // Not a constant
 425   }
 426 
 427   ciConstant con = check_mismatched_access(field_value, loadbt, is_unsigned_load);
 428 
 429   assert(con.is_valid(), "elembt=%s; loadbt=%s; unsigned=%d",
 430          type2name(field_value.basic_type()), type2name(loadbt), is_unsigned_load);
 431 
 432   bool is_stable_array = FoldStableValues && field->is_stable() && field->type()->is_array_klass();
 433   int stable_dimension = (is_stable_array ? field->type()->as_array_klass()->dimension() : 0);
 434   bool is_narrow_oop = (loadbt == T_NARROWOOP);
 435 
 436   const Type* con_type = make_from_constant(con, /*require_constant=*/ true,
 437                                             stable_dimension, is_narrow_oop,
 438                                             field->is_autobox_cache());
 439   if (con_type != NULL && field->is_call_site_target()) {
 440     ciCallSite* call_site = holder->as_call_site();
 441     if (!call_site->is_fully_initialized_constant_call_site()) {
 442       ciMethodHandle* target = con.as_object()->as_method_handle();
 443       Compile::current()->dependencies()->assert_call_site_target_value(call_site, target);
 444     }
 445   }
 446   return con_type;
 447 }

 480 //--------------------------Initialize_shared----------------------------------
 481 void Type::Initialize_shared(Compile* current) {
 482   // This method does not need to be locked because the first system
 483   // compilations (stub compilations) occur serially.  If they are
 484   // changed to proceed in parallel, then this section will need
 485   // locking.
 486 
 487   Arena* save = current->type_arena();
 488   Arena* shared_type_arena = new (mtCompiler)Arena(mtCompiler);
 489

 772 //------------------------------hashcons---------------------------------------
 773 // Do the hash-cons trick.  If the Type already exists in the type table,
 774 // delete the current Type and return the existing Type.  Otherwise stick the
 775 // current Type in the Type table.
 776 const Type *Type::hashcons(void) {
 777   debug_only(base());           // Check the assertion in Type::base().
 778   // Look up the Type in the Type dictionary
 779   Dict *tdic = type_dict();
 780   Type* old = (Type*)(tdic->Insert(this, this, false));
 781   if( old ) {                   // Pre-existing Type?
 782     if( old != this )           // Yes, this guy is not the pre-existing?
 783       delete this;              // Yes, Nuke this guy
 784     assert( old->_dual, "" );
 785     return old;                 // Return pre-existing
 786   }
 787 
 788   // Every type has a dual (to make my lattice symmetric).
 789   // Since we just discovered a new Type, compute its dual right now.
 790   assert( !_dual, "" );         // No dual yet
 791   _dual = xdual();              // Compute the dual
 792   if (cmp(this, _dual) == 0) {  // Handle self-symmetric
 793     if (_dual != this) {
 794       delete _dual;
 795       _dual = this;
 796     }
 797     return this;
 798   }
 799   assert( !_dual->_dual, "" );  // No reverse dual yet
 800   assert( !(*tdic)[_dual], "" ); // Dual not in type system either
 801   // New Type, insert into Type table
 802   tdic->Insert((void*)_dual,(void*)_dual);
 803   ((Type*)_dual)->_dual = this; // Finish up being symmetric
 804 #ifdef ASSERT
 805   Type *dual_dual = (Type*)_dual->xdual();
 806   assert( eq(dual_dual), "xdual(xdual()) should be identity" );
 807   delete dual_dual;
 808 #endif
 809   return this;                  // Return new Type
 810 }

 913 // commutative and the lattice is symmetric.
 914 const Type *Type::meet_helper(const Type *t, bool include_speculative) const {
 915   if (isa_narrowoop() && t->isa_narrowoop()) {
 916     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 917     return result->make_narrowoop();
 918   }
 919   if (isa_narrowklass() && t->isa_narrowklass()) {
 920     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 921     return result->make_narrowklass();
 922   }
 923 
 924   const Type *this_t = maybe_remove_speculative(include_speculative);
 925   t = t->maybe_remove_speculative(include_speculative);
 926 
 927   const Type *mt = this_t->xmeet(t);
 928 #ifdef ASSERT
 929   if (isa_narrowoop() || t->isa_narrowoop()) return mt;
 930   if (isa_narrowklass() || t->isa_narrowklass()) return mt;
 931   Compile* C = Compile::current();
 932   if (!C->_type_verify_symmetry) {
 933     return mt;
 934   }
 935   this_t->check_symmetrical(t, mt);
 936   // In the case of an array, computing the meet above, caused the
 937   // computation of the meet of the elements which at verification
 938   // time caused the computation of the meet of the dual of the
 939   // elements. Computing the meet of the dual of the arrays here
 940   // causes the meet of the dual of the elements to be computed which
 941   // would cause the meet of the dual of the dual of the elements,
 942   // that is the meet of the elements already computed above to be
 943   // computed. Avoid redundant computations by requesting no
 944   // verification.
 945   C->_type_verify_symmetry = false;
 946   const Type *mt_dual = this_t->_dual->xmeet(t->_dual);
 947   this_t->_dual->check_symmetrical(t->_dual, mt_dual);
 948   assert(!C->_type_verify_symmetry, "shouldn't have changed");
 949   C->_type_verify_symmetry = true;
 950 #endif
 951   return mt;
 952 }

 954 //------------------------------xmeet------------------------------------------
 955 // Compute the MEET of two types.  It returns a new Type object.
 956 const Type *Type::xmeet( const Type *t ) const {
 957   // Perform a fast test for common case; meeting the same types together.
 958   if( this == t ) return this;  // Meeting same type-rep?
 959 
 960   // Meeting TOP with anything?
 961   if( _base == Top ) return t;
 962 
 963   // Meeting BOTTOM with anything?
 964   if( _base == Bottom ) return BOTTOM;
 965 
 966   // Current "this->_base" is one of: Bad, Multi, Control, Top,
 967   // Abio, Abstore, Floatxxx, Doublexxx, Bottom, lastype.
 968   switch (t->base()) {  // Switch on original type
 969 
 970   // Cut in half the number of cases I must handle.  Only need cases for when
 971   // the given enum "t->type" is less than or equal to the local enum "type".
 972   case FloatCon:
 973   case DoubleCon:
 974   case Int:
 975   case Long:
 976     return t->xmeet(this);
 977 
 978   case OopPtr:
 979     return t->xmeet(this);
 980 
 981   case InstPtr:
 982     return t->xmeet(this);
 983 
 984   case MetadataPtr:
 985   case KlassPtr:
 986     return t->xmeet(this);
 987 
 988   case AryPtr:
 989     return t->xmeet(this);
 990 
 991   case NarrowOop:
 992     return t->xmeet(this);
 993 
 994   case NarrowKlass:
 995     return t->xmeet(this);
 996 
 997   case ValueType:
 998     return t->xmeet(this);
 999 
1000   case Bad:                     // Type check
1001   default:                      // Bogus type not in lattice
1002     typerr(t);
1003     return Type::BOTTOM;
1004 
1005   case Bottom:                  // Ye Olde Default
1006     return t;
1007 
1008   case FloatTop:
1009     if( _base == FloatTop ) return this;
1010   case FloatBot:                // Float
1011     if( _base == FloatBot || _base == FloatTop ) return FLOAT;
1012     if( _base == DoubleTop || _base == DoubleBot ) return Type::BOTTOM;
1013     typerr(t);
1014     return Type::BOTTOM;
1015 
1016   case DoubleTop:
1017     if( _base == DoubleTop ) return this;
1018   case DoubleBot:               // Double
1019     if( _base == DoubleBot || _base == DoubleTop ) return DOUBLE;
1020     if( _base == FloatTop || _base == FloatBot ) return Type::BOTTOM;
1021     typerr(t);
1022     return Type::BOTTOM;
1023 
1024   // These next few cases must match exactly or it is a compile-time error.
1025   case Control:                 // Control of code
1026   case Abio:                    // State of world outside of program
1027   case Memory:
1028     if( _base == t->_base )  return this;
1029     typerr(t);
1030     return Type::BOTTOM;
1031 
1032   case Top:                     // Top of the lattice
1033     return this;
1034   }
1035 
1036   // The type is unchanged
1037   return this;
1038 }

1502 //------------------------------widen------------------------------------------
1503 // Only happens for optimistic top-down optimizations.
1504 const Type *TypeInt::widen( const Type *old, const Type* limit ) const {
1505   // Coming from TOP or such; no widening
1506   if( old->base() != Int ) return this;
1507   const TypeInt *ot = old->is_int();
1508 
1509   // If new guy is equal to old guy, no widening
1510   if( _lo == ot->_lo && _hi == ot->_hi )
1511     return old;
1512 
1513   // If new guy contains old, then we widened
1514   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1515     // New contains old
1516     // If new guy is already wider than old, no widening
1517     if( _widen > ot->_widen ) return this;
1518     // If old guy was a constant, do not bother
1519     if (ot->_lo == ot->_hi)  return this;
1520     // Now widen new guy.
1521     // Check for widening too far
1522     if (_widen == WidenMax) {
1523       int max = max_jint;
1524       int min = min_jint;
1525       if (limit->isa_int()) {
1526         max = limit->is_int()->_hi;
1527         min = limit->is_int()->_lo;
1528       }
1529       if (min < _lo && _hi < max) {
1530         // If neither endpoint is extremal yet, push out the endpoint
1531         // which is closer to its respective limit.
1532         if (_lo >= 0 ||                 // easy common case
1533             (juint)(_lo - min) >= (juint)(max - _hi)) {
1534           // Try to widen to an unsigned range type of 31 bits:
1535           return make(_lo, max, WidenMax);
1536         } else {
1537           return make(min, _hi, WidenMax);
1538         }
1539       }
1540       return TypeInt::INT;
1541     }
1542     // Returned widened new guy
1543     return make(_lo,_hi,_widen+1);
1544   }
1545 
1546   // If old guy contains new, then we probably widened too far & dropped to
1547   // bottom.  Return the wider fellow.
1548   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1549     return old;
1550 
1551   //fatal("Integer value range is not subset");
1552   //return this;
1553   return TypeInt::INT;
1554 }

1556 //------------------------------narrow---------------------------------------
1557 // Only happens for pessimistic optimizations.
1558 const Type *TypeInt::narrow( const Type *old ) const {
1559   if (_lo >= _hi)  return this;   // already narrow enough
1560   if (old == NULL)  return this;
1561   const TypeInt* ot = old->isa_int();
1562   if (ot == NULL)  return this;
1563   jint olo = ot->_lo;
1564   jint ohi = ot->_hi;
1565 
1566   // If new guy is equal to old guy, no narrowing
1567   if (_lo == olo && _hi == ohi)  return old;
1568 
1569   // If old guy was maximum range, allow the narrowing
1570   if (olo == min_jint && ohi == max_jint)  return this;
1571 
1572   if (_lo < olo || _hi > ohi)
1573     return this;                // doesn't narrow; pretty wierd
1574 
1575   // The new type narrows the old type, so look for a "death march".
1576   // See comments on PhaseTransform::saturate.
1577   juint nrange = (juint)_hi - _lo;
1578   juint orange = (juint)ohi - olo;
1579   if (nrange < max_juint - 1 && nrange > (orange >> 1) + (SMALLINT*2)) {
1580     // Use the new type only if the range shrinks a lot.
1581     // We do not want the optimizer computing 2^31 point by point.
1582     return old;
1583   }
1584 
1585   return this;
1586 }

1713 //------------------------------meet-------------------------------------------
1714 // Compute the MEET of two types.  It returns a new Type representation object
1715 // with reference count equal to the number of Types pointing at it.
1716 // Caller should wrap a Types around it.
1717 const Type *TypeLong::xmeet( const Type *t ) const {
1718   // Perform a fast test for common case; meeting the same types together.
1719   if( this == t ) return this;  // Meeting same type?
1720 
1721   // Currently "this->_base" is a TypeLong
1722   switch (t->base()) {          // Switch on original type
1723   case AnyPtr:                  // Mixing with oops happens when javac
1724   case RawPtr:                  // reuses local variables
1725   case OopPtr:
1726   case InstPtr:
1727   case AryPtr:
1728   case MetadataPtr:
1729   case KlassPtr:
1730   case NarrowOop:
1731   case NarrowKlass:
1732   case Int:
1733   case FloatTop:
1734   case FloatCon:
1735   case FloatBot:
1736   case DoubleTop:
1737   case DoubleCon:
1738   case DoubleBot:
1739   case Bottom:                  // Ye Olde Default
1740     return Type::BOTTOM;
1741   default:                      // All else is a mistake
1742     typerr(t);
1743   case Top:                     // No change
1744     return this;
1745   case Long:                    // Long vs Long?
1746     break;
1747   }
1748 
1749   // Expand covered set
1750   const TypeLong *r = t->is_long(); // Turn into a TypeLong
1751   return make( MIN2(_lo,r->_lo), MAX2(_hi,r->_hi), MAX2(_widen,r->_widen) );

1761 //------------------------------widen------------------------------------------
1762 // Only happens for optimistic top-down optimizations.
1763 const Type *TypeLong::widen( const Type *old, const Type* limit ) const {
1764   // Coming from TOP or such; no widening
1765   if( old->base() != Long ) return this;
1766   const TypeLong *ot = old->is_long();
1767 
1768   // If new guy is equal to old guy, no widening
1769   if( _lo == ot->_lo && _hi == ot->_hi )
1770     return old;
1771 
1772   // If new guy contains old, then we widened
1773   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1774     // New contains old
1775     // If new guy is already wider than old, no widening
1776     if( _widen > ot->_widen ) return this;
1777     // If old guy was a constant, do not bother
1778     if (ot->_lo == ot->_hi)  return this;
1779     // Now widen new guy.
1780     // Check for widening too far
1781     if (_widen == WidenMax) {
1782       jlong max = max_jlong;
1783       jlong min = min_jlong;
1784       if (limit->isa_long()) {
1785         max = limit->is_long()->_hi;
1786         min = limit->is_long()->_lo;
1787       }
1788       if (min < _lo && _hi < max) {
1789         // If neither endpoint is extremal yet, push out the endpoint
1790         // which is closer to its respective limit.
1791         if (_lo >= 0 ||                 // easy common case
1792             ((julong)_lo - min) >= ((julong)max - _hi)) {
1793           // Try to widen to an unsigned range type of 32/63 bits:
1794           if (max >= max_juint && _hi < max_juint)
1795             return make(_lo, max_juint, WidenMax);
1796           else
1797             return make(_lo, max, WidenMax);
1798         } else {
1799           return make(min, _hi, WidenMax);
1800         }
1801       }
1802       return TypeLong::LONG;
1803     }
1804     // Returned widened new guy
1805     return make(_lo,_hi,_widen+1);
1806   }
1807 
1808   // If old guy contains new, then we probably widened too far & dropped to
1809   // bottom.  Return the wider fellow.
1810   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1811     return old;
1812 
1813   //  fatal("Long value range is not subset");
1814   // return this;
1815   return TypeLong::LONG;
1816 }

2374 bool TypeAry::ary_must_be_exact() const {
2375   if (!UseExactTypes)       return false;

2385     toop = _elem->isa_oopptr();

2388   ciKlass* tklass = toop->klass();
2389   if (tklass == NULL)       return false;  // unloaded class
2390   if (!tklass->is_loaded()) return false;  // unloaded class
2391   const TypeInstPtr* tinst;
2392   if (_elem->isa_narrowoop())
2393     tinst = _elem->make_ptr()->isa_instptr();
2394   else
2395     tinst = _elem->isa_instptr();

2397     // [V? has a subtype: [V. So even though V is final, [V? is not exact.
2398     if (tklass->as_instance_klass()->is_final()) {
2399       if (tinst->is_valuetypeptr() && (tinst->ptr() == TypePtr::BotPTR || tinst->ptr() == TypePtr::TopPTR)) {

2402       return true;

2857  *

2861   bool this_has_spec = (_speculative != NULL);

3137 // Compute the MEET of two types.  It returns a new Type object.
3138 const Type *TypeRawPtr::xmeet( const Type *t ) const {
3139   // Perform a fast test for common case; meeting the same types together.
3140   if( this == t ) return this;  // Meeting same type-rep?
3141 
3142   // Current "this->_base" is RawPtr
3143   switch( t->base() ) {         // switch on original type
3144   case Bottom:                  // Ye Olde Default
3145     return t;

3389   case NarrowOop:
3390   case NarrowKlass:
3391   case Bottom:                  // Ye Olde Default
3392     return Type::BOTTOM;
3393   case Top:
3394     return this;
3395 
3396   default:                      // All else is a mistake
3397     typerr(t);
3398 
3399   case RawPtr:
3400   case MetadataPtr:
3401   case KlassPtr:
3402     return TypePtr::BOTTOM;     // Oop meet raw is not well defined
3403 
3404   case AnyPtr: {
3405     // Found an AnyPtr type vs self-OopPtr type
3406     const TypePtr *tp = t->is_ptr();
3407     Offset offset = meet_offset(tp->offset());
3408     PTR ptr = meet_ptr(tp->ptr());
3409     const TypePtr* speculative = xmeet_speculative(tp);
3410     int depth = meet_inline_depth(tp->inline_depth());
3411     switch (tp->ptr()) {
3412     case Null:
3413       if (ptr == Null)  return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
3414       // else fall through:
3415     case TopPTR:
3416     case AnyNull: {
3417       int instance_id = meet_instance_id(InstanceTop);
3418       return make(ptr, offset, instance_id, speculative, depth);
3419     }
3420     case BotPTR:
3421     case NotNull:
3422       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
3423     default: typerr(t);
3424     }
3425   }
3426 
3427   case OopPtr: {                 // Meeting to other OopPtrs
3428     const TypeOopPtr *tp = t->is_oopptr();
3429     int instance_id = meet_instance_id(tp->instance_id());
3430     const TypePtr* speculative = xmeet_speculative(tp);
3431     int depth = meet_inline_depth(tp->inline_depth());
3432     return make(meet_ptr(tp->ptr()), meet_offset(tp->offset()), instance_id, speculative, depth);
3433   }
3434 
3435   case InstPtr:                  // For these, flip the call around to cut down
3436   case AryPtr:
3437     return t->xmeet(this);      // Call in reverse direction
3438 
3439   } // End of switch
3440   return this;                  // Return the double constant
3441 }

3452 //--------------------------make_from_klass_common-----------------------------
3453 // Computes the element-type given a klass.
3454 const TypeOopPtr* TypeOopPtr::make_from_klass_common(ciKlass *klass, bool klass_change, bool try_for_exact) {
3455   if (klass->is_instance_klass() || klass->is_valuetype()) {
3456     Compile* C = Compile::current();
3457     Dependencies* deps = C->dependencies();
3458     assert((deps != NULL) == (C->method() != NULL && C->method()->code_size() > 0), "sanity");
3459     // Element is an instance
3460     bool klass_is_exact = false;
3461     if (klass->is_loaded()) {
3462       // Try to set klass_is_exact.
3463       ciInstanceKlass* ik = klass->as_instance_klass();
3464       klass_is_exact = ik->is_final();
3465       if (!klass_is_exact && klass_change
3466           && deps != NULL && UseUniqueSubclasses) {
3467         ciInstanceKlass* sub = ik->unique_concrete_subklass();
3468         if (sub != NULL) {
3469           deps->assert_abstract_with_unique_concrete_subtype(ik, sub);
3470           klass = ik = sub;
3471           klass_is_exact = sub->is_final();
3472         }
3473       }
3474       if (!klass_is_exact && try_for_exact
3475           && deps != NULL && UseExactTypes) {
3476         if (!ik->is_interface() && !ik->has_subklass()) {
3477           // Add a dependence; if concrete subclass added we need to recompile
3478           deps->assert_leaf_type(ik);
3479           klass_is_exact = true;
3480         }
3481       }
3482     }
3483     return TypeInstPtr::make(TypePtr::BotPTR, klass, klass_is_exact, NULL, Offset(0), klass->flatten_array());
3484   } else if (klass->is_obj_array_klass()) {
3485     // Element is an object or value array. Recursively call ourself.
3486     const TypeOopPtr* etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), false, try_for_exact);
3487     bool null_free = klass->is_loaded() && klass->as_array_klass()->storage_properties().is_null_free();
3488     if (null_free) {
3489       assert(etype->is_valuetypeptr(), "must be a valuetypeptr");
3490       etype = etype->join_speculative(TypePtr::NOTNULL)->is_oopptr();
3491     }
3492     // [V? has a subtype: [V. So even though V is final, [V? is not exact.
3493     bool xk = etype->klass_is_exact() && (!etype->is_valuetypeptr() || null_free);
3494     bool not_null_free = !etype->can_be_value_type() || xk;
3495     bool not_flat = !ValueArrayFlatten || not_null_free || (etype->is_valuetypeptr() && !etype->value_klass()->flatten_array());
3496     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS, false, not_flat, not_null_free);
3497     // We used to pass NotNull in here, asserting that the sub-arrays
3498     // are all not-null.  This is not true in generally, as code can
3499     // slam NULLs down in the subarrays.
3500     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, xk, Offset(0));
3501     return arr;
3502   } else if (klass->is_type_array_klass()) {
3503     // Element is an typeArray
3504     const Type* etype = get_const_basic_type(klass->as_type_array_klass()->element_type());
3505     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS,
3506                                         /* stable= */ false, /* not_flat= */ true, /* not_null_free= */ true);
3507     // We used to pass NotNull in here, asserting that the array pointer
3508     // is not-null. That was not true in general.
3509     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3510     return arr;
3511   } else if (klass->is_value_array_klass()) {
3512     ciValueKlass* vk = klass->as_array_klass()->element_klass()->as_value_klass();
3513     const TypeAry* arr0 = TypeAry::make(TypeValueType::make(vk), TypeInt::POS);
3514     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3515     return arr;
3516   } else {
3517     ShouldNotReachHere();
3518     return NULL;
3519   }
3520 }

3605 //-----------------------------filter------------------------------------------
3606 // Do not allow interface-vs.-noninterface joins to collapse to top.
3607 const Type *TypeOopPtr::filter_helper(const Type *kills, bool include_speculative) const {
3608 
3609   const Type* ft = join_helper(kills, include_speculative);
3610   const TypeInstPtr* ftip = ft->isa_instptr();
3611   const TypeInstPtr* ktip = kills->isa_instptr();
3612 
3613   if (ft->empty()) {
3614     // Check for evil case of 'this' being a class and 'kills' expecting an
3615     // interface.  This can happen because the bytecodes do not contain
3616     // enough type info to distinguish a Java-level interface variable
3617     // from a Java-level object variable.  If we meet 2 classes which
3618     // both implement interface I, but their meet is at 'j/l/O' which
3619     // doesn't implement I, we have no way to tell if the result should
3620     // be 'I' or 'j/l/O'.  Thus we'll pick 'j/l/O'.  If this then flows
3621     // into a Phi which "knows" it's an Interface type we'll have to
3622     // uplift the type.
3623     if (!empty()) {
3624       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3625         return kills;           // Uplift to interface
3626       }
3627       // Also check for evil cases of 'this' being a class array
3628       // and 'kills' expecting an array of interfaces.
3629       Type::get_arrays_base_elements(ft, kills, NULL, &ktip);
3630       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3631         return kills;           // Uplift to array of interface
3632       }
3633     }
3634 
3635     return Type::TOP;           // Canonical empty value
3636   }
3637 
3638   // If we have an interface-typed Phi or cast and we narrow to a class type,
3639   // the join should report back the class.  However, if we have a J/L/Object
3640   // class-typed Phi and an interface flows in, it's possible that the meet &
3641   // join report an interface back out.  This isn't possible but happens
3642   // because the type system doesn't interact well with interfaces.
3643   if (ftip != NULL && ktip != NULL &&
3644       ftip->is_loaded() &&  ftip->klass()->is_interface() &&
3645       ktip->is_loaded() && !ktip->klass()->is_interface()) {
3646     assert(!ftip->klass_is_exact(), "interface could not be exact");
3647     return ktip->cast_to_ptr_type(ftip->ptr());
3648   }
3649 
3650   return ft;
3651 }

3888 // Compute the MEET of two InstPtrs when at least one is unloaded.
3889 // Assume classes are different since called after check for same name/class-loader
3890 const TypeInstPtr *TypeInstPtr::xmeet_unloaded(const TypeInstPtr *tinst) const {
3891     Offset off = meet_offset(tinst->offset());
3892     PTR ptr = meet_ptr(tinst->ptr());
3893     int instance_id = meet_instance_id(tinst->instance_id());
3894     const TypePtr* speculative = xmeet_speculative(tinst);
3895     int depth = meet_inline_depth(tinst->inline_depth());
3896 
3897     const TypeInstPtr *loaded    = is_loaded() ? this  : tinst;
3898     const TypeInstPtr *unloaded  = is_loaded() ? tinst : this;
3899     if( loaded->klass()->equals(ciEnv::current()->Object_klass()) ) {
3900       //
3901       // Meet unloaded class with java/lang/Object
3902       //
3903       // Meet
3904       //          |                     Unloaded Class
3905       //  Object  |   TOP    |   AnyNull | Constant |   NotNull |  BOTTOM   |
3906       //  ===================================================================
3907       //   TOP    | ..........................Unloaded......................|
3908       //  AnyNull |  U-AN    |................Unloaded......................|
3909       // Constant | ... O-NN .................................. |   O-BOT   |
3910       //  NotNull | ... O-NN .................................. |   O-BOT   |
3911       //  BOTTOM  | ........................Object-BOTTOM ..................|
3912       //
3913       assert(loaded->ptr() != TypePtr::Null, "insanity check");
3914       //
3915       if(      loaded->ptr() == TypePtr::TopPTR ) { return unloaded; }
3916       else if (loaded->ptr() == TypePtr::AnyNull) { return TypeInstPtr::make(ptr, unloaded->klass(), false, NULL, off, false, instance_id, speculative, depth); }
3917       else if (loaded->ptr() == TypePtr::BotPTR ) { return TypeInstPtr::BOTTOM; }
3918       else if (loaded->ptr() == TypePtr::Constant || loaded->ptr() == TypePtr::NotNull) {
3919         if (unloaded->ptr() == TypePtr::BotPTR  ) { return TypeInstPtr::BOTTOM;  }
3920         else                                      { return TypeInstPtr::NOTNULL; }
3921       }
3922       else if( unloaded->ptr() == TypePtr::TopPTR )  { return unloaded; }
3923 
3924       return unloaded->cast_to_ptr_type(TypePtr::AnyNull)->is_instptr();
3925     }
3926 
3927     // Both are unloaded, not the same class, not Object
3928     // Or meet unloaded with a different loaded class, not java/lang/Object
3929     if( ptr != TypePtr::BotPTR ) {
3930       return TypeInstPtr::NOTNULL;
3931     }
3932     return TypeInstPtr::BOTTOM;
3933 }

3936 //------------------------------meet-------------------------------------------
3937 // Compute the MEET of two types.  It returns a new Type object.
3938 const Type *TypeInstPtr::xmeet_helper(const Type *t) const {
3939   // Perform a fast test for common case; meeting the same types together.
3940   if( this == t ) return this;  // Meeting same type-rep?
3941 
3942   // Current "this->_base" is Pointer
3943   switch (t->base()) {          // switch on original type
3944 
3945   case Int:                     // Mixing ints & oops happens when javac
3946   case Long:                    // reuses local variables
3947   case FloatTop:
3948   case FloatCon:
3949   case FloatBot:
3950   case DoubleTop:
3951   case DoubleCon:
3952   case DoubleBot:
3953   case NarrowOop:
3954   case NarrowKlass:
3955   case Bottom:                  // Ye Olde Default
3956     return Type::BOTTOM;
3957   case Top:
3958     return this;
3959 
3960   default:                      // All else is a mistake
3961     typerr(t);
3962 
3963   case MetadataPtr:
3964   case KlassPtr:
3965   case RawPtr: return TypePtr::BOTTOM;
3966 
3967   case AryPtr: {                // All arrays inherit from Object class
3968     const TypeAryPtr *tp = t->is_aryptr();
3969     Offset offset = meet_offset(tp->offset());
3970     PTR ptr = meet_ptr(tp->ptr());
3971     int instance_id = meet_instance_id(tp->instance_id());
3972     const TypePtr* speculative = xmeet_speculative(tp);
3973     int depth = meet_inline_depth(tp->inline_depth());
3974     switch (ptr) {
3975     case TopPTR:
3976     case AnyNull:                // Fall 'down' to dual of object klass
3977       // For instances when a subclass meets a superclass we fall
3978       // below the centerline when the superclass is exact. We need to
3979       // do the same here.
3980       if (klass()->equals(ciEnv::current()->Object_klass()) && !klass_is_exact()) {
3981         return TypeAryPtr::make(ptr, tp->ary(), tp->klass(), tp->klass_is_exact(), offset, tp->field_offset(), instance_id, speculative, depth);
3982       } else {
3983         // cannot subclass, so the meet has to fall badly below the centerline
3984         ptr = NotNull;
3985         instance_id = InstanceBot;
3986         return TypeInstPtr::make( ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
3987       }
3988     case Constant:
3989     case NotNull:
3990     case BotPTR:                // Fall down to object klass
3991       // LCA is object_klass, but if we subclass from the top we can do better
3992       if( above_centerline(_ptr) ) { // if( _ptr == TopPTR || _ptr == AnyNull )
3993         // If 'this' (InstPtr) is above the centerline and it is Object class
3994         // then we can subclass in the Java class hierarchy.
3995         // For instances when a subclass meets a superclass we fall
3996         // below the centerline when the superclass is exact. We need
3997         // to do the same here.
3998         if (klass()->equals(ciEnv::current()->Object_klass()) && !klass_is_exact()) {
3999           // that is, tp's array type is a subtype of my klass
4000           return TypeAryPtr::make(ptr, (ptr == Constant ? tp->const_oop() : NULL),
4001                                   tp->ary(), tp->klass(), tp->klass_is_exact(), offset, tp->field_offset(), instance_id, speculative, depth);
4002         }
4003       }
4004       // The other case cannot happen, since I cannot be a subtype of an array.
4005       // The meet falls down to Object class below centerline.
4006       if( ptr == Constant )
4007          ptr = NotNull;
4008       instance_id = InstanceBot;
4009       return make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4010     default: typerr(t);
4011     }
4012   }
4013 
4014   case OopPtr: {                // Meeting to OopPtrs
4015     // Found a OopPtr type vs self-InstPtr type
4016     const TypeOopPtr *tp = t->is_oopptr();
4017     Offset offset = meet_offset(tp->offset());
4018     PTR ptr = meet_ptr(tp->ptr());
4019     switch (tp->ptr()) {
4020     case TopPTR:
4021     case AnyNull: {
4022       int instance_id = meet_instance_id(InstanceTop);
4023       const TypePtr* speculative = xmeet_speculative(tp);
4024       int depth = meet_inline_depth(tp->inline_depth());
4025       return make(ptr, klass(), klass_is_exact(),
4026                   (ptr == Constant ? const_oop() : NULL), offset, flat_array(), instance_id, speculative, depth);
4027     }
4028     case NotNull:
4029     case BotPTR: {
4030       int instance_id = meet_instance_id(tp->instance_id());
4031       const TypePtr* speculative = xmeet_speculative(tp);
4032       int depth = meet_inline_depth(tp->inline_depth());
4033       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4034     }
4035     default: typerr(t);
4036     }
4037   }
4038 
4039   case AnyPtr: {                // Meeting to AnyPtrs
4040     // Found an AnyPtr type vs self-InstPtr type
4041     const TypePtr *tp = t->is_ptr();
4042     Offset offset = meet_offset(tp->offset());
4043     PTR ptr = meet_ptr(tp->ptr());
4044     int instance_id = meet_instance_id(InstanceTop);
4045     const TypePtr* speculative = xmeet_speculative(tp);
4046     int depth = meet_inline_depth(tp->inline_depth());
4047     switch (tp->ptr()) {
4048     case Null:
4049       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4050       // else fall through to AnyNull
4051     case TopPTR:
4052     case AnyNull: {
4053       return make(ptr, klass(), klass_is_exact(),
4054                   (ptr == Constant ? const_oop() : NULL), offset, flat_array(), instance_id, speculative, depth);
4055     }
4056     case NotNull:
4057     case BotPTR:
4058       return TypePtr::make(AnyPtr, ptr, offset, speculative,depth);
4059     default: typerr(t);
4060     }
4061   }
4062 
4063   /*
4064                  A-top         }
4065                /   |   \       }  Tops
4066            B-top A-any C-top   }
4067               | /  |  \ |      }  Any-nulls
4068            B-any   |   C-any   }
4069               |    |    |
4070            B-con A-con C-con   } constants; not comparable across classes
4071               |    |    |
4072            B-not   |   C-not   }
4073               | \  |  / |      }  not-nulls
4074            B-bot A-not C-bot   }
4075                \   |   /       }  Bottoms
4076                  A-bot         }
4077   */
4078 
4079   case InstPtr: {                // Meeting 2 Oops?
4080     // Found an InstPtr sub-type vs self-InstPtr type
4081     const TypeInstPtr *tinst = t->is_instptr();
4082     Offset off = meet_offset( tinst->offset() );
4083     PTR ptr = meet_ptr( tinst->ptr() );
4084     int instance_id = meet_instance_id(tinst->instance_id());
4085     const TypePtr* speculative = xmeet_speculative(tinst);
4086     int depth = meet_inline_depth(tinst->inline_depth());
4087 
4088     // Check for easy case; klasses are equal (and perhaps not loaded!)
4089     // If we have constants, then we created oops so classes are loaded
4090     // and we can handle the constants further down.  This case handles
4091     // both-not-loaded or both-loaded classes
4092     if (ptr != Constant && klass()->equals(tinst->klass()) && klass_is_exact() == tinst->klass_is_exact() &&
4093         flat_array() == tinst->flat_array()) {
4094       return make(ptr, klass(), klass_is_exact(), NULL, off, flat_array(), instance_id, speculative, depth);
4095     }
4096 
4097     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
4098     ciKlass* tinst_klass = tinst->klass();
4099     ciKlass* this_klass  = this->klass();
4100     bool tinst_xk = tinst->klass_is_exact();
4101     bool this_xk  = this->klass_is_exact();
4102     bool tinst_flat_array = tinst->flat_array();
4103     bool this_flat_array  = this->flat_array();
4104     if (!tinst_klass->is_loaded() || !this_klass->is_loaded() ) {
4105       // One of these classes has not been loaded
4106       const TypeInstPtr *unloaded_meet = xmeet_unloaded(tinst);
4107 #ifndef PRODUCT
4108       if( PrintOpto && Verbose ) {
4109         tty->print("meet of unloaded classes resulted in: "); unloaded_meet->dump(); tty->cr();
4110         tty->print("  this == "); this->dump(); tty->cr();
4111         tty->print(" tinst == "); tinst->dump(); tty->cr();
4112       }
4113 #endif
4114       return unloaded_meet;
4115     }
4116 
4117     // Handle mixing oops and interfaces first.
4118     if( this_klass->is_interface() && !(tinst_klass->is_interface() ||
4119                                         tinst_klass == ciEnv::current()->Object_klass())) {
4120       ciKlass *tmp = tinst_klass; // Swap interface around
4121       tinst_klass = this_klass;
4122       this_klass = tmp;
4123       bool tmp2 = tinst_xk;
4124       tinst_xk = this_xk;
4125       this_xk = tmp2;
4126       tmp2 = tinst_flat_array;
4127       tinst_flat_array = this_flat_array;
4128       this_flat_array = tmp2;
4129     }
4130     if (tinst_klass->is_interface() &&
4131         !(this_klass->is_interface() ||
4132           // Treat java/lang/Object as an honorary interface,
4133           // because we need a bottom for the interface hierarchy.
4134           this_klass == ciEnv::current()->Object_klass())) {
4135       // Oop meets interface!
4136 
4137       // See if the oop subtypes (implements) interface.
4138       ciKlass *k;
4139       bool xk;
4140       bool flat_array;
4141       if( this_klass->is_subtype_of( tinst_klass ) ) {
4142         // Oop indeed subtypes.  Now keep oop or interface depending
4143         // on whether we are both above the centerline or either is
4144         // below the centerline.  If we are on the centerline
4145         // (e.g., Constant vs. AnyNull interface), use the constant.
4146         k  = below_centerline(ptr) ? tinst_klass : this_klass;
4147         // If we are keeping this_klass, keep its exactness too.
4148         xk = below_centerline(ptr) ? tinst_xk    : this_xk;
4149         flat_array = below_centerline(ptr) ? tinst_flat_array    : this_flat_array;
4150       } else {                  // Does not implement, fall to Object
4151         // Oop does not implement interface, so mixing falls to Object
4152         // just like the verifier does (if both are above the
4153         // centerline fall to interface)
4154         k = above_centerline(ptr) ? tinst_klass : ciEnv::current()->Object_klass();
4155         xk = above_centerline(ptr) ? tinst_xk : false;
4156         flat_array = above_centerline(ptr) ? tinst_flat_array : false;
4157         // Watch out for Constant vs. AnyNull interface.
4158         if (ptr == Constant)  ptr = NotNull;   // forget it was a constant
4159         instance_id = InstanceBot;
4160       }
4161       ciObject* o = NULL;  // the Constant value, if any
4162       if (ptr == Constant) {
4163         // Find out which constant.
4164         o = (this_klass == klass()) ? const_oop() : tinst->const_oop();
4165       }
4166       return make(ptr, k, xk, o, off, flat_array, instance_id, speculative, depth);
4167     }
4168 
4169     // Either oop vs oop or interface vs interface or interface vs Object
4170 
4171     // !!! Here's how the symmetry requirement breaks down into invariants:
4172     // If we split one up & one down AND they subtype, take the down man.
4173     // If we split one up & one down AND they do NOT subtype, "fall hard".
4174     // If both are up and they subtype, take the subtype class.
4175     // If both are up and they do NOT subtype, "fall hard".
4176     // If both are down and they subtype, take the supertype class.
4177     // If both are down and they do NOT subtype, "fall hard".
4178     // Constants treated as down.
4179 
4180     // Now, reorder the above list; observe that both-down+subtype is also
4181     // "fall hard"; "fall hard" becomes the default case:
4182     // If we split one up & one down AND they subtype, take the down man.
4183     // If both are up and they subtype, take the subtype class.
4184 
4185     // If both are down and they subtype, "fall hard".
4186     // If both are down and they do NOT subtype, "fall hard".
4187     // If both are up and they do NOT subtype, "fall hard".
4188     // If we split one up & one down AND they do NOT subtype, "fall hard".
4189 
4190     // If a proper subtype is exact, and we return it, we return it exactly.
4191     // If a proper supertype is exact, there can be no subtyping relationship!
4192     // If both types are equal to the subtype, exactness is and-ed below the
4193     // centerline and or-ed above it.  (N.B. Constants are always exact.)
4194 
4195     // Check for subtyping:
4196     ciKlass *subtype = NULL;
4197     bool subtype_exact = false;
4198     bool flat_array = false;
4199     if( tinst_klass->equals(this_klass) ) {
4200       subtype = this_klass;
4201       subtype_exact = below_centerline(ptr) ? (this_xk && tinst_xk) : (this_xk || tinst_xk);
4202       flat_array = below_centerline(ptr) ? (this_flat_array && tinst_flat_array) : (this_flat_array || tinst_flat_array);
4203     } else if( !tinst_xk && this_klass->is_subtype_of( tinst_klass ) ) {
4204       subtype = this_klass;     // Pick subtyping class
4205       subtype_exact = this_xk;
4206       flat_array = this_flat_array;
4207     } else if( !this_xk && tinst_klass->is_subtype_of( this_klass ) ) {
4208       subtype = tinst_klass;    // Pick subtyping class
4209       subtype_exact = tinst_xk;
4210       flat_array = tinst_flat_array;
4211     }
4212 
4213     if( subtype ) {
4214       if( above_centerline(ptr) ) { // both are up?
4215         this_klass = tinst_klass = subtype;
4216         this_xk = tinst_xk = subtype_exact;
4217         this_flat_array = tinst_flat_array = flat_array;
4218       } else if( above_centerline(this ->_ptr) && !above_centerline(tinst->_ptr) ) {
4219         this_klass = tinst_klass; // tinst is down; keep down man
4220         this_xk = tinst_xk;
4221         this_flat_array = tinst_flat_array;
4222       } else if( above_centerline(tinst->_ptr) && !above_centerline(this ->_ptr) ) {
4223         tinst_klass = this_klass; // this is down; keep down man
4224         tinst_xk = this_xk;
4225         tinst_flat_array = this_flat_array;
4226       } else {
4227         this_xk = subtype_exact;  // either they are equal, or we'll do an LCA
4228         this_flat_array = flat_array;
4229       }
4230     }
4231 
4232     // Check for classes now being equal
4233     if (tinst_klass->equals(this_klass)) {
4234       // If the klasses are equal, the constants may still differ.  Fall to
4235       // NotNull if they do (neither constant is NULL; that is a special case
4236       // handled elsewhere).
4237       ciObject* o = NULL;             // Assume not constant when done
4238       ciObject* this_oop  = const_oop();
4239       ciObject* tinst_oop = tinst->const_oop();
4240       if( ptr == Constant ) {
4241         if (this_oop != NULL && tinst_oop != NULL &&
4242             this_oop->equals(tinst_oop) )
4243           o = this_oop;
4244         else if (above_centerline(this ->_ptr))
4245           o = tinst_oop;
4246         else if (above_centerline(tinst ->_ptr))
4247           o = this_oop;
4248         else
4249           ptr = NotNull;
4250       }
4251       return make(ptr, this_klass, this_xk, o, off, this_flat_array, instance_id, speculative, depth);
4252     } // Else classes are not equal
4253 
4254     // Since klasses are different, we require a LCA in the Java
4255     // class hierarchy - which means we have to fall to at least NotNull.
4256     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
4257       ptr = NotNull;
4258 
4259     instance_id = InstanceBot;
4260 
4261     // Now we find the LCA of Java classes
4262     ciKlass* k = this_klass->least_common_ancestor(tinst_klass);
4263     return make(ptr, k, false, NULL, off, false, instance_id, speculative, depth);
4264   } // End of case InstPtr
4265 
4266   case ValueType: {
4267     const TypeValueType* tv = t->is_valuetype();
4268     if (above_centerline(ptr())) {
4269       if (tv->value_klass()->is_subtype_of(_klass)) {
4270         return t;
4271       } else {
4272         return TypeInstPtr::make(NotNull, _klass);
4273       }
4274     } else {
4275       PTR ptr = this->_ptr;
4276       if (ptr == Constant) {
4277         ptr = NotNull;
4278       }
4279       if (tv->value_klass()->is_subtype_of(_klass)) {
4280         return TypeInstPtr::make(ptr, _klass);
4281       } else {
4282         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4283       }
4284     }
4285   }
4286 
4287   } // End of switch
4288   return this;                  // Return the double constant
4289 }

4603 //------------------------------meet-------------------------------------------
4604 // Compute the MEET of two types.  It returns a new Type object.
4605 const Type *TypeAryPtr::xmeet_helper(const Type *t) const {
4606   // Perform a fast test for common case; meeting the same types together.
4607   if( this == t ) return this;  // Meeting same type-rep?
4608   // Current "this->_base" is Pointer
4609   switch (t->base()) {          // switch on original type
4610 
4611   // Mixing ints & oops happens when javac reuses local variables
4612   case Int:
4613   case Long:
4614   case FloatTop:
4615   case FloatCon:
4616   case FloatBot:
4617   case DoubleTop:
4618   case DoubleCon:
4619   case DoubleBot:
4620   case NarrowOop:
4621   case NarrowKlass:
4622   case Bottom:                  // Ye Olde Default
4623     return Type::BOTTOM;
4624   case Top:
4625     return this;
4626 
4627   default:                      // All else is a mistake
4628     typerr(t);
4629 
4630   case OopPtr: {                // Meeting to OopPtrs
4631     // Found a OopPtr type vs self-AryPtr type
4632     const TypeOopPtr *tp = t->is_oopptr();
4633     Offset offset = meet_offset(tp->offset());
4634     PTR ptr = meet_ptr(tp->ptr());
4635     int depth = meet_inline_depth(tp->inline_depth());
4636     const TypePtr* speculative = xmeet_speculative(tp);
4637     switch (tp->ptr()) {
4638     case TopPTR:
4639     case AnyNull: {
4640       int instance_id = meet_instance_id(InstanceTop);
4641       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4642                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4643     }
4644     case BotPTR:
4645     case NotNull: {
4646       int instance_id = meet_instance_id(tp->instance_id());
4647       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4648     }
4649     default: ShouldNotReachHere();
4650     }
4651   }
4652 
4653   case AnyPtr: {                // Meeting two AnyPtrs
4654     // Found an AnyPtr type vs self-AryPtr type
4655     const TypePtr *tp = t->is_ptr();
4656     Offset offset = meet_offset(tp->offset());
4657     PTR ptr = meet_ptr(tp->ptr());
4658     const TypePtr* speculative = xmeet_speculative(tp);
4659     int depth = meet_inline_depth(tp->inline_depth());
4660     switch (tp->ptr()) {
4661     case TopPTR:
4662       return this;
4663     case BotPTR:
4664     case NotNull:
4665       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4666     case Null:
4667       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4668       // else fall through to AnyNull
4669     case AnyNull: {
4670       int instance_id = meet_instance_id(InstanceTop);
4671       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4672                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4673     }
4674     default: ShouldNotReachHere();
4675     }
4676   }
4677 
4678   case MetadataPtr:
4679   case KlassPtr:
4680   case RawPtr: return TypePtr::BOTTOM;
4681 
4682   case AryPtr: {                // Meeting 2 references?
4683     const TypeAryPtr *tap = t->is_aryptr();
4684     Offset off = meet_offset(tap->offset());
4685     Offset field_off = meet_field_offset(tap->field_offset());
4686     const TypeAry *tary = _ary->meet_speculative(tap->_ary)->is_ary();
4687     PTR ptr = meet_ptr(tap->ptr());
4688     int instance_id = meet_instance_id(tap->instance_id());
4689     const TypePtr* speculative = xmeet_speculative(tap);
4690     int depth = meet_inline_depth(tap->inline_depth());
4691     ciKlass* lazy_klass = NULL;
4692     if (tary->_elem->isa_int()) {
4693       // Integral array element types have irrelevant lattice relations.
4694       // It is the klass that determines array layout, not the element type.
4695       if (_klass == NULL)
4696         lazy_klass = tap->_klass;
4697       else if (tap->_klass == NULL || tap->_klass == _klass) {
4698         lazy_klass = _klass;
4699       } else {
4700         // Something like byte[int+] meets char[int+].
4701         // This must fall to bottom, not (int[-128..65535])[int+].
4702         instance_id = InstanceBot;
4703         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4704       }
4705     } else if (klass() != NULL && tap->klass() != NULL &&
4706                klass()->as_array_klass()->storage_properties().value() != tap->klass()->as_array_klass()->storage_properties().value()) {
4707       // Meeting value type arrays with conflicting storage properties
4708       if (tary->_elem->isa_valuetype()) {
4709         // Result is flattened
4710         off = Offset(elem()->isa_valuetype() ? offset() : tap->offset());
4711         field_off = elem()->isa_valuetype() ? field_offset() : tap->field_offset();
4712       } else if (tary->_elem->make_oopptr() != NULL && tary->_elem->make_oopptr()->isa_instptr() && below_centerline(ptr)) {
4713         // Result is non-flattened
4714         off = Offset(flattened_offset()).meet(Offset(tap->flattened_offset()));
4715         field_off = Offset::bottom;
4716       }
4717     } else // Non integral arrays.
4718       // Must fall to bottom if exact klasses in upper lattice
4719       // are not equal or super klass is exact.
4720       if ((above_centerline(ptr) || ptr == Constant) && klass() != tap->klass() &&
4721           // meet with top[] and bottom[] are processed further down:
4722           tap->_klass != NULL && this->_klass != NULL &&
4723           // both are exact and not equal:
4724           ((tap->_klass_is_exact && this->_klass_is_exact) ||
4725            // 'tap' is exact and super or unrelated:
4726            (tap->_klass_is_exact && !tap->klass()->is_subtype_of(klass())) ||
4727            // 'this' is exact and super or unrelated:
4728            (this->_klass_is_exact && !klass()->is_subtype_of(tap->klass())))) {
4729       if (above_centerline(ptr) || (tary->_elem->make_ptr() && above_centerline(tary->_elem->make_ptr()->_ptr))) {
4730         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4731       }
4732       return make(NotNull, NULL, tary, lazy_klass, false, off, field_off, InstanceBot, speculative, depth);
4733     }
4734 
4735     bool xk = false;
4736     switch (tap->ptr()) {
4737     case AnyNull:
4738     case TopPTR:
4739       // Compute new klass on demand, do not use tap->_klass
4740       if (below_centerline(this->_ptr)) {
4741         xk = this->_klass_is_exact;
4742       } else {
4743         xk = (tap->_klass_is_exact || this->_klass_is_exact);
4744       }
4745       return make(ptr, const_oop(), tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4746     case Constant: {
4747       ciObject* o = const_oop();
4748       if( _ptr == Constant ) {
4749         if( tap->const_oop() != NULL && !o->equals(tap->const_oop()) ) {
4750           xk = (klass() == tap->klass());
4751           ptr = NotNull;
4752           o = NULL;
4753           instance_id = InstanceBot;
4754         } else {
4755           xk = true;
4756         }
4757       } else if(above_centerline(_ptr)) {
4758         o = tap->const_oop();
4759         xk = true;
4760       } else {
4761         // Only precise for identical arrays
4762         xk = this->_klass_is_exact && (klass() == tap->klass());
4763       }
4764       return TypeAryPtr::make(ptr, o, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4765     }
4766     case NotNull:
4767     case BotPTR:
4768       // Compute new klass on demand, do not use tap->_klass
4769       if (above_centerline(this->_ptr))
4770             xk = tap->_klass_is_exact;
4771       else  xk = (tap->_klass_is_exact & this->_klass_is_exact) &&
4772               (klass() == tap->klass()); // Only precise for identical arrays
4773       return TypeAryPtr::make(ptr, NULL, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4774     default: ShouldNotReachHere();
4775     }
4776   }
4777 
4778   // All arrays inherit from Object class
4779   case InstPtr: {
4780     const TypeInstPtr *tp = t->is_instptr();
4781     Offset offset = meet_offset(tp->offset());
4782     PTR ptr = meet_ptr(tp->ptr());
4783     int instance_id = meet_instance_id(tp->instance_id());
4784     const TypePtr* speculative = xmeet_speculative(tp);
4785     int depth = meet_inline_depth(tp->inline_depth());
4786     switch (ptr) {
4787     case TopPTR:
4788     case AnyNull:                // Fall 'down' to dual of object klass
4789       // For instances when a subclass meets a superclass we fall
4790       // below the centerline when the superclass is exact. We need to
4791       // do the same here.
4792       if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact()) {
4793         return TypeAryPtr::make(ptr, _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4794       } else {
4795         // cannot subclass, so the meet has to fall badly below the centerline
4796         ptr = NotNull;
4797         instance_id = InstanceBot;
4798         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4799       }
4800     case Constant:
4801     case NotNull:
4802     case BotPTR:                // Fall down to object klass
4803       // LCA is object_klass, but if we subclass from the top we can do better
4804       if (above_centerline(tp->ptr())) {
4805         // If 'tp'  is above the centerline and it is Object class
4806         // then we can subclass in the Java class hierarchy.
4807         // For instances when a subclass meets a superclass we fall
4808         // below the centerline when the superclass is exact. We need
4809         // to do the same here.
4810         if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact()) {
4811           // that is, my array type is a subtype of 'tp' klass
4812           return make(ptr, (ptr == Constant ? const_oop() : NULL),
4813                       _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4814         }
4815       }
4816       // The other case cannot happen, since t cannot be a subtype of an array.
4817       // The meet falls down to Object class below centerline.
4818       if( ptr == Constant )
4819          ptr = NotNull;
4820       instance_id = InstanceBot;
4821       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4822     default: typerr(t);
4823     }
4824   }
4825 
4826   case ValueType: {
4827     // All value types inherit from Object
4828     PTR ptr = this->_ptr;
4829     if (ptr == Constant) {
4830       ptr = NotNull;
4831     }
4832     return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4833   }
4834 
4835   }
4836   return this;                  // Lint noise
4837 }

4957 const TypePtr* TypeAryPtr::add_field_offset_and_offset(intptr_t offset) const {
4958   int adj = 0;
4959   if (offset != Type::OffsetBot && offset != Type::OffsetTop) {
4960     const Type* elemtype = elem();
4961     if (elemtype->isa_valuetype()) {
4962       if (_offset.get() != OffsetBot && _offset.get() != OffsetTop) {
4963         adj = _offset.get();
4964         offset += _offset.get();
4965       }
4966       uint header = arrayOopDesc::base_offset_in_bytes(T_OBJECT);
4967       if (_field_offset.get() != OffsetBot && _field_offset.get() != OffsetTop) {
4968         offset += _field_offset.get();
4969         if (_offset.get() == OffsetBot || _offset.get() == OffsetTop) {
4970           offset += header;
4971         }
4972       }
4973       if (offset >= (intptr_t)header || offset < 0) {
4974         // Try to get the field of the value type array element we are pointing to
4975         ciKlass* arytype_klass = klass();
4976         ciValueArrayKlass* vak = arytype_klass->as_value_array_klass();
4977         ciValueKlass* vk = vak->element_klass()->as_value_klass();
4978         int shift = vak->log2_element_size();
4979         int mask = (1 << shift) - 1;
4980         intptr_t field_offset = ((offset - header) & mask);
4981         ciField* field = vk->get_field_by_offset(field_offset + vk->first_field_offset(), false);
4982         if (field == NULL) {
4983           // This may happen with nested AddP(base, AddP(base, base, offset), longcon(16))
4984           return add_offset(offset);
4985         } else {
4986           return with_field_offset(field_offset)->add_offset(offset - field_offset - adj);
4987         }
4988       }
4989     }
4990   }
4991   return add_offset(offset - adj);
4992 }

5233   if( this == t ) return this;  // Meeting same type-rep?
5234 
5235   // Current "this->_base" is OopPtr

5237

5421 //----------------------compute_klass------------------------------------------
5422 // Compute the defining klass for this class
5423 ciKlass* TypeAryPtr::compute_klass(DEBUG_ONLY(bool verify)) const {
5424   // Compute _klass based on element type.
5425   ciKlass* k_ary = NULL;
5426   const TypeAryPtr *tary;
5427   const Type* el = elem();
5428   if (el->isa_narrowoop()) {
5429     el = el->make_ptr();
5430   }
5431 
5432   // Get element klass
5433   if (el->isa_instptr()) {
5434     // Compute object array klass from element klass
5435     bool null_free = el->is_valuetypeptr() && el->isa_instptr()->ptr() != TypePtr::TopPTR && !el->isa_instptr()->maybe_null();

5437   } else if (el->isa_valuetype()) {
5438     if (el->value_klass() != NULL) {
5439       k_ary = ciArrayKlass::make(el->value_klass(), /* null_free */ true);
5440     }
5441   } else if ((tary = el->isa_aryptr()) != NULL) {
5442     // Compute array klass from element klass
5443     ciKlass* k_elem = tary->klass();
5444     // If element type is something like bottom[], k_elem will be null.
5445     if (k_elem != NULL)
5446       k_ary = ciObjArrayKlass::make(k_elem);
5447   } else if ((el->base() == Type::Top) ||
5448              (el->base() == Type::Bottom)) {
5449     // element type of Bottom occurs from meet of basic type
5450     // and object; Top occurs when doing join on Bottom.
5451     // Leave k_ary at NULL.
5452   } else {
5453     // Cannot compute array klass directly from basic type,
5454     // since subtypes of TypeInt all have basic type T_INT.
5455 #ifdef ASSERT

5458       BasicType bt = T_ILLEGAL;

5460         bt = T_BYTE;
5461       } else if (el == TypeInt::SHORT) {
5462         bt = T_SHORT;

5490     // The _klass field acts as a cache of the underlying
5491     // ciKlass for this array type.  In order to set the field,
5492     // we need to cast away const-ness.
5493     //
5494     // IMPORTANT NOTE: we *never* set the _klass field for the
5495     // type TypeAryPtr::OOPS.  This Type is shared between all
5496     // active compilations.  However, the ciKlass which represents
5497     // this Type is *not* shared between compilations, so caching
5498     // this value would result in fetching a dangling pointer.
5499     //
5500     // Recomputing the underlying ciKlass for each request is
5501     // a bit less efficient than caching, but calls to
5502     // TypeAryPtr::OOPS->klass() are not common enough to matter.
5503     ((TypeAryPtr*)this)->_klass = k_ary;
5504     if (UseCompressedOops && k_ary != NULL && k_ary->is_obj_array_klass() &&
5505         offset() != 0 && offset() != arrayOopDesc::length_offset_in_bytes()) {
5506       ((TypeAryPtr*)this)->_is_ptr_to_narrowoop = true;
5507     }
5508   }
5509   return k_ary;
5510 }

5553 //------------------------------xmeet------------------------------------------
5554 // Compute the MEET of two types, return a new Type object.
5555 const Type    *TypeKlassPtr::xmeet( const Type *t ) const {
5556   // Perform a fast test for common case; meeting the same types together.
5557   if( this == t ) return this;  // Meeting same type-rep?
5558 
5559   // Current "this->_base" is Pointer
5560   switch (t->base()) {          // switch on original type
5561 
5562   case Int:                     // Mixing ints & oops happens when javac
5563   case Long:                    // reuses local variables
5564   case FloatTop:
5565   case FloatCon:
5566   case FloatBot:
5567   case DoubleTop:
5568   case DoubleCon:
5569   case DoubleBot:
5570   case NarrowOop:
5571   case NarrowKlass:
5572   case Bottom:                  // Ye Olde Default
5573     return Type::BOTTOM;
5574   case Top:
5575     return this;
5576 
5577   default:                      // All else is a mistake
5578     typerr(t);
5579 
5580   case AnyPtr: {                // Meeting to AnyPtrs
5581     // Found an AnyPtr type vs self-KlassPtr type
5582     const TypePtr *tp = t->is_ptr();
5583     Offset offset = meet_offset(tp->offset());
5584     PTR ptr = meet_ptr(tp->ptr());
5585     switch (tp->ptr()) {
5586     case TopPTR:
5587       return this;
5588     case Null:
5589       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5590     case AnyNull:
5591       return make(ptr, klass(), offset, flat_array());
5592     case BotPTR:
5593     case NotNull:
5594       return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5595     default: typerr(t);
5596     }
5597   }
5598 
5599   case RawPtr:
5600   case MetadataPtr:
5601   case OopPtr:
5602   case AryPtr:                  // Meet with AryPtr
5603   case InstPtr:                 // Meet with InstPtr
5604     return TypePtr::BOTTOM;
5605 
5606   //
5607   //             A-top         }
5608   //           /   |   \       }  Tops
5609   //       B-top A-any C-top   }
5610   //          | /  |  \ |      }  Any-nulls
5611   //       B-any   |   C-any   }
5612   //          |    |    |
5613   //       B-con A-con C-con   } constants; not comparable across classes
5614   //          |    |    |
5615   //       B-not   |   C-not   }
5616   //          | \  |  / |      }  not-nulls
5617   //       B-bot A-not C-bot   }
5618   //           \   |   /       }  Bottoms
5619   //             A-bot         }
5620   //
5621 
5622   case KlassPtr: {  // Meet two KlassPtr types
5623     const TypeKlassPtr *tkls = t->is_klassptr();
5624     Offset  off  = meet_offset(tkls->offset());
5625     PTR  ptr     = meet_ptr(tkls->ptr());
5626 
5627     if (klass() == NULL || tkls->klass() == NULL) {
5628       ciKlass* k = NULL;
5629       if (ptr == Constant) {
5630         k = (klass() == NULL) ? tkls->klass() : klass();
5631       }
5632       return make(ptr, k, off, false);
5633     }
5634 
5635     // Check for easy case; klasses are equal (and perhaps not loaded!)
5636     // If we have constants, then we created oops so classes are loaded
5637     // and we can handle the constants further down.  This case handles
5638     // not-loaded classes
5639     if (ptr != Constant && tkls->klass()->equals(klass()) && flat_array() == tkls->flat_array()) {
5640       return make(ptr, klass(), off, flat_array());
5641     }
5642 
5643     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
5644     ciKlass* tkls_klass = tkls->klass();
5645     ciKlass* this_klass = this->klass();
5646     assert( tkls_klass->is_loaded(), "This class should have been loaded.");
5647     assert( this_klass->is_loaded(), "This class should have been loaded.");
5648     bool tkls_flat_array = tkls->flat_array();
5649     bool this_flat_array  = this->flat_array();
5650     bool flat_array = below_centerline(ptr) ? (this_flat_array && tkls_flat_array) : (this_flat_array || tkls_flat_array);
5651 
5652     // If 'this' type is above the centerline and is a superclass of the
5653     // other, we can treat 'this' as having the same type as the other.
5654     if ((above_centerline(this->ptr())) &&
5655         tkls_klass->is_subtype_of(this_klass)) {
5656       this_klass = tkls_klass;
5657     }
5658     // If 'tinst' type is above the centerline and is a superclass of the
5659     // other, we can treat 'tinst' as having the same type as the other.
5660     if ((above_centerline(tkls->ptr())) &&
5661         this_klass->is_subtype_of(tkls_klass)) {
5662       tkls_klass = this_klass;
5663     }
5664 
5665     // Check for classes now being equal
5666     if (tkls_klass->equals(this_klass)) {
5667       // If the klasses are equal, the constants may still differ.  Fall to
5668       // NotNull if they do (neither constant is NULL; that is a special case
5669       // handled elsewhere).
5670       if( ptr == Constant ) {
5671         if (this->_ptr == Constant && tkls->_ptr == Constant &&
5672             this->klass()->equals(tkls->klass()));
5673         else if (above_centerline(this->ptr()));
5674         else if (above_centerline(tkls->ptr()));
5675         else
5676           ptr = NotNull;
5677       }
5678       return make(ptr, this_klass, off, flat_array);
5679     } // Else classes are not equal
5680 
5681     // Since klasses are different, we require the LCA in the Java
5682     // class hierarchy - which means we have to fall to at least NotNull.
5683     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
5684       ptr = NotNull;
5685     // Now we find the LCA of Java classes
5686     ciKlass* k = this_klass->least_common_ancestor(tkls_klass);
5687     return   make(ptr, k, off, k->is_valuetype() && k->flatten_array());
5688   } // End of case KlassPtr
5689 
5690   } // End of switch
5691   return this;                  // Return the double constant
5692 }

5702   assert( _ptr == Null || _ptr == Constant, "" );
5703   assert(offset() >= 0, "");

5774 //------------------------------make-------------------------------------------
5775 const TypeFunc* TypeFunc::make(ciMethod* method, bool is_osr_compilation) {
5776   Compile* C = Compile::current();
5777   const TypeFunc* tf = NULL;
5778   if (!is_osr_compilation) {
5779     tf = C->last_tf(method); // check cache
5780     if (tf != NULL)  return tf;  // The hit rate here is almost 50%.
5781   }
5782   // Value types are not passed/returned by reference, instead each field of
5783   // the value type is passed/returned as an argument. We maintain two views of
5784   // the argument/return list here: one based on the signature (with a value
5785   // type argument/return as a single slot), one based on the actual calling
5786   // convention (with a value type argument/return as a list of its fields).
5787   bool has_scalar_args = method->has_scalarized_args() && !is_osr_compilation;
5788   const TypeTuple* domain_sig = is_osr_compilation ? osr_domain() : TypeTuple::make_domain(method, false);
5789   const TypeTuple* domain_cc = has_scalar_args ? TypeTuple::make_domain(method, true) : domain_sig;
5790   ciSignature* sig = method->signature();
5791   bool has_scalar_ret = sig->returns_never_null() && sig->return_type()->as_value_klass()->can_be_returned_as_fields();
5792   const TypeTuple* range_sig = TypeTuple::make_range(sig, false);
5793   const TypeTuple* range_cc = has_scalar_ret ? TypeTuple::make_range(sig, true) : range_sig;
5794   tf = TypeFunc::make(domain_sig, domain_cc, range_sig, range_cc);
5795   if (!is_osr_compilation) {
5796     C->set_last_tf(method, tf);  // fill cache
5797   }
5798   return tf;
5799 }
