26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import static javax.lang.model.SourceVersion.*;
31 
32 /**
33  * A simple visitor for annotation values with default behavior
34  * appropriate for the {@link SourceVersion#RELEASE_7 RELEASE_7}
35  * source version.  Visit methods call {@link #defaultAction
36  * defaultAction} passing their arguments to {@code defaultAction}'s
37  * corresponding parameters.
38  *
39  * @apiNote
40  * Methods in this class may be overridden subject to their general
41  * contract.


















42  *
43  * @param <R> the return type of this visitor's methods
44  * @param <P> the type of the additional parameter to this visitor's methods.
45  *
46  * @see <a href="SimpleAnnotationValueVisitor6.html#note_for_subclasses">
47  * <strong>Compatibility note for subclasses</strong></a>
48  * @see SimpleAnnotationValueVisitor6
49  * @see SimpleAnnotationValueVisitor8
50  * @see SimpleAnnotationValueVisitor9
51  * @see SimpleAnnotationValueVisitor14
52  * @since 1.7
53  */
54 @SupportedSourceVersion(RELEASE_7)
55 public class SimpleAnnotationValueVisitor7<R, P> extends SimpleAnnotationValueVisitor6<R, P> {
56     /**
57      * Constructor for concrete subclasses; uses {@code null} for the
58      * default value.
59      *
60      * @deprecated Release 7 is obsolete; update to a visitor for a newer
61      * release level.
62      */
63     @Deprecated(since="12")
64     protected SimpleAnnotationValueVisitor7() {
65         super(null); // Superclass constructor deprecated too
66     }
67 
68     /**
69      * Constructor for concrete subclasses; uses the argument for the
70      * default value.
71      *
72      * @param defaultValue the value to assign to {@link #DEFAULT_VALUE}
73      *
74      * @deprecated Release 7 is obsolete; update to a visitor for a newer
75      * release level.
76      */
77     @Deprecated(since="12")
78     protected SimpleAnnotationValueVisitor7(R defaultValue) {
79         super(defaultValue); // Superclass constructor deprecated too
80     }
81 }
