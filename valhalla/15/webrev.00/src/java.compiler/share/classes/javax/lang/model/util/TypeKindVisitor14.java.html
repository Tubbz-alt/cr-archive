26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import javax.lang.model.type.*;
31 import static javax.lang.model.SourceVersion.*;
32 
33 /**
34  * A visitor of types based on their {@linkplain TypeKind kind} with
35  * default behavior appropriate for source version {@link
36  * SourceVersion#RELEASE_14 RELEASE_14}.
37  *
38  * For {@linkplain
39  * TypeMirror types} <code><i>Xyz</i></code> that may have more than one
40  * kind, the <code>visit<i>Xyz</i></code> methods in this class delegate
41  * to the <code>visit<i>Xyz</i>As<i>Kind</i></code> method corresponding to the
42  * first argument's kind.  The <code>visit<i>Xyz</i>As<i>Kind</i></code> methods
43  * call {@link #defaultAction defaultAction}, passing their arguments
44  * to {@code defaultAction}'s corresponding parameters.
45  *
46  * @apiNote
47  * Methods in this class may be overridden subject to their general
48  * contract.
49  *
50  * @param <R> the return type of this visitor's methods.  Use {@link
51  *            Void} for visitors that do not need to return results.
52  * @param <P> the type of the additional parameter to this visitor's
53  *            methods.  Use {@code Void} for visitors that do not need an
54  *            additional parameter.
55  *
56  * @see <a href="TypeKindVisitor6.html#note_for_subclasses">
57  * <strong>Compatibility note for subclasses</strong></a>
58  * @see TypeKindVisitor6
59  * @see TypeKindVisitor7
60  * @see TypeKindVisitor8
61  * @see TypeKindVisitor9
62  * @since 14
63  */
64 @SupportedSourceVersion(RELEASE_15)
65 public class TypeKindVisitor14<R, P> extends TypeKindVisitor9<R, P> {
66     /**
67      * Constructor for concrete subclasses to call; uses {@code null}
68      * for the default value.
69      */
70     protected TypeKindVisitor14() {
71         super(null);
72     }
73 
74     /**
75      * Constructor for concrete subclasses to call; uses the argument
76      * for the default value.
77      *
78      * @param defaultValue the value to assign to {@link #DEFAULT_VALUE}
79      */
80     protected TypeKindVisitor14(R defaultValue) {
81         super(defaultValue);
82     }
83 }
