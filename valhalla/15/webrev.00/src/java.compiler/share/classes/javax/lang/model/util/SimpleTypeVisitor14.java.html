26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import static javax.lang.model.SourceVersion.*;
31 
32 /**
33  * A simple visitor of types with default behavior appropriate for
34  * source version {@link SourceVersion#RELEASE_14 RELEASE_14}.
35  *
36  * Visit methods corresponding to {@code RELEASE_14} and earlier
37  * language constructs call {@link #defaultAction defaultAction},
38  * passing their arguments to {@code defaultAction}'s corresponding
39  * parameters.
40  *
41  * @apiNote
42  * Methods in this class may be overridden subject to their general
43  * contract.
44  *
45  * @param <R> the return type of this visitor's methods.  Use {@link
46  *            Void} for visitors that do not need to return results.
47  * @param <P> the type of the additional parameter to this visitor's
48  *            methods.  Use {@code Void} for visitors that do not need an
49  *            additional parameter.
50  *
51  * @see <a href="SimpleTypeVisitor6.html#note_for_subclasses">
52  * <strong>Compatibility note for subclasses</strong></a>
53  * @see SimpleTypeVisitor6
54  * @see SimpleTypeVisitor7
55  * @see SimpleTypeVisitor8
56  * @see SimpleTypeVisitor9
57  * @since 14
58  */
59 @SupportedSourceVersion(RELEASE_15)
60 public class SimpleTypeVisitor14<R, P> extends SimpleTypeVisitor9<R, P> {
61     /**
62      * Constructor for concrete subclasses; uses {@code null} for the
63      * default value.
64      */
65     protected SimpleTypeVisitor14(){
66         super(null);
67     }
68 
69     /**
70      * Constructor for concrete subclasses; uses the argument for the
71      * default value.
72      *
73      * @param defaultValue the value to assign to {@link #DEFAULT_VALUE}
74      */
75     protected SimpleTypeVisitor14(R defaultValue){
76         super(defaultValue);
77     }
78 }
