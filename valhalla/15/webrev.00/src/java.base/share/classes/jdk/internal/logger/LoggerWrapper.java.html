27 package jdk.internal.logger;
28 
29 import java.util.Objects;
30 import java.lang.System.Logger;
31 import sun.util.logging.PlatformLogger;
32 
33 /**
34  * An implementation of {@link Logger} that redirects all calls to a wrapped
35  * instance of Logger.
36  *
37  * @param <L> Type of the wrapped Logger: {@code Logger} or an
38  *            extension of that interface.
39  */
40 public class LoggerWrapper<L extends Logger> extends AbstractLoggerWrapper<L> {
41 
42     final L wrapped;
43     final PlatformLogger.Bridge platformProxy;
44 
45     public LoggerWrapper(L wrapped) {
46         this(Objects.requireNonNull(wrapped), (Void)null);
47     }
48 
49     LoggerWrapper(L wrapped, Void unused) {
50         this.wrapped = wrapped;
51         this.platformProxy = (wrapped instanceof PlatformLogger.Bridge) ?
52             (PlatformLogger.Bridge) wrapped : null;
53     }
54 
55     @Override
56     public final L wrapped() {
57         return wrapped;
58     }
59 
60     @Override
61     public final PlatformLogger.Bridge platformProxy() {
62         return platformProxy;
63     }
64 
65 }
