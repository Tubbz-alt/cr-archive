34 // VirtualCallData
35 //
36 // A VirtualCallData is used to access profiling information about a
37 // call.  For now, it has nothing more than a ReceiverTypeData.
38 public class VirtualCallData<K,M> extends ReceiverTypeData<K,M> {
39   public VirtualCallData(MethodDataInterface<K,M> methodData, DataLayout layout) {
40     super(methodData, layout);
41     //assert(layout.tag() == DataLayout.virtualCallDataTag, "wrong type");
42   }
43 
44   static int staticCellCount() {
45     // At this point we could add more profile state, e.g., for arguments.
46     // But for now it's the same size as the base record type.
47     int cellCount = ReceiverTypeData.staticCellCount();
48     if (INCLUDE_JVMCI == 1) {
49       cellCount += MethodData.MethodProfileWidth * receiverTypeRowCellCount;
50     }
51     return cellCount;
52   }
53 
54   public int cellCount() {
55     return staticCellCount();
56   }
57 
58   // Direct accessors
59   static int virtualCallDataSize() {
60     return cellOffset(staticCellCount());
61   }
62 
63   public void printDataOn(PrintStream st) {
64     printShared(st, "VirtualCallData");
65     printReceiverDataOn(st);
66   }
67 }
