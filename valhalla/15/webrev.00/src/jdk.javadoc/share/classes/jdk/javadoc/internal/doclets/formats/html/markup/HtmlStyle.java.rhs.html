366     private final String cssName;
367 
368     HtmlStyle() {
369         cssName = Pattern.compile("\\p{Upper}")
370                 .matcher(toString())
371                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
372     }
373 
374     HtmlStyle(String cssName) {
375         this.cssName = cssName;
376     }
377 
378     /**
379      * Returns the CSS class name associated with this style.
380      * @return the CSS class name
381      */
382     public String cssName() {
383         return cssName;
384     }
