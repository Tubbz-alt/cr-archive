 26 package javax.swing.colorchooser;
 27 
 28 import java.awt.Color;
 29 import java.awt.ContainerOrderFocusTraversalPolicy;
 30 import java.awt.GridBagConstraints;
 31 import java.awt.GridBagLayout;
 32 import java.awt.Insets;
 33 import java.awt.event.ActionEvent;
 34 import java.awt.event.ActionListener;
 35 import javax.swing.ButtonGroup;
 36 import javax.swing.JLabel;
 37 import javax.swing.JPanel;
 38 import javax.swing.JRadioButton;
 39 import javax.swing.border.EmptyBorder;
 40 import javax.swing.JSpinner.DefaultEditor;
 41 
 42 @SuppressWarnings("serial") // Superclass is not serializable across versions
 43 final class ColorPanel extends JPanel implements ActionListener {
 44 
 45     private final SlidingSpinner[] spinners = new SlidingSpinner[5];
 46     private final float[] values = new float[this.spinners.length];
 47 
 48     private final ColorModel model;
 49     private Color color;
 50     private int x = 1;
 51     private int y = 2;
 52     private int z;
 53 
 54     ColorPanel(ColorModel model) {
 55         super(new GridBagLayout());
 56 
 57         GridBagConstraints gbc = new GridBagConstraints();
 58         gbc.fill = GridBagConstraints.HORIZONTAL;
 59 
 60         gbc.gridx = 1;
 61         ButtonGroup group = new ButtonGroup();
 62         EmptyBorder border = null;
 63         for (int i = 0; i < this.spinners.length; i++) {
 64             if (i < 3) {
 65                 JRadioButton button = new JRadioButton();
 66                 if (i == 0) {
 67                     Insets insets = button.getInsets();
 68                     insets.left = button.getPreferredSize().width;
 69                     border = new EmptyBorder(insets);
 70                     button.setSelected(true);
 71                     gbc.insets.top = 5;
 72                 }
 73                 add(button, gbc);
 74                 group.add(button);
 75                 button.setActionCommand(Integer.toString(i));
 76                 button.addActionListener(this);
 77                 this.spinners[i] = new SlidingSpinner(this, button);
 78             }
 79             else {
 80                 JLabel label = new JLabel();
 81                 add(label, gbc);
 82                 label.setBorder(border);
 83                 label.setFocusable(false);
 84                 this.spinners[i] = new SlidingSpinner(this, label);
 85             }
 86         }
 87         gbc.gridx = 2;
 88         gbc.weightx = 1.0;
 89         gbc.insets.top = 0;
 90         gbc.insets.left = 5;
 91         for (SlidingSpinner spinner : this.spinners) {
 92             add(spinner.getSlider(), gbc);
 93             gbc.insets.top = 5;
 94         }
 95         gbc.gridx = 3;
 96         gbc.weightx = 0.0;
 97         gbc.insets.top = 0;
 98         for (SlidingSpinner spinner : this.spinners) {
 99             add(spinner.getSpinner(), gbc);
100             gbc.insets.top = 5;
101         }


102         setFocusable(false);
103 
104         this.model = model;
105     }
106 
107     public void actionPerformed(ActionEvent event) {
108         try {
109             this.z = Integer.parseInt(event.getActionCommand());
110             this.y = (this.z != 2) ? 2 : 1;
111             this.x = (this.z != 0) ? 0 : 1;
112             getParent().repaint();
113         }
114         catch (NumberFormatException exception) {
115         }
116     }
117 
118     void buildPanel() {
119         int count = this.model.getCount();
120         this.spinners[4].setVisible(count > 4);
121         for (int i = 0; i < count; i++) {
122             String text = this.model.getLabel(this, i);
123             Object object = this.spinners[i].getLabel();
124             if (object instanceof JRadioButton) {
125                 JRadioButton button = (JRadioButton) object;
126                 button.setText(text);
127                 button.getAccessibleContext().setAccessibleDescription(text);
128             }
129             else if (object instanceof JLabel) {
130                 JLabel label = (JLabel) object;
131                 label.setText(text);
132             }
133             this.spinners[i].setRange(this.model.getMinimum(i), this.model.getMaximum(i));
134             this.spinners[i].setValue(this.values[i]);
135             this.spinners[i].getSlider().getAccessibleContext().setAccessibleName(text);
136             this.spinners[i].getSpinner().getAccessibleContext().setAccessibleName(text);
137             DefaultEditor editor = (DefaultEditor) this.spinners[i].getSpinner().getEditor();
138             editor.getTextField().getAccessibleContext().setAccessibleName(text);
139             this.spinners[i].getSlider().getAccessibleContext().setAccessibleDescription(text);
140             this.spinners[i].getSpinner().getAccessibleContext().setAccessibleDescription(text);
141             editor.getTextField().getAccessibleContext().setAccessibleDescription(text);
142         }
143     }
144 
145     void colorChanged() {
146         this.color = new Color(getColor(0), isColorTransparencySelectionEnabled());
147         Object parent = getParent();
148         if (parent instanceof ColorChooserPanel) {
149             ColorChooserPanel chooser = (ColorChooserPanel) parent;
150             chooser.setSelectedColor(this.color);
151             chooser.repaint();
152         }
153     }
154 
155     float getValueX() {
156         return this.spinners[this.x].getValue();
157     }
158 
159     float getValueY() {
160         return 1.0f - this.spinners[this.y].getValue();
161     }
162 
163     float getValueZ() {
164         return 1.0f - this.spinners[this.z].getValue();
165     }
166 
167     void setValue(float z) {
168         this.spinners[this.z].setValue(1.0f - z);
169         colorChanged();
170     }
171 
172     void setValue(float x, float y) {
173         this.spinners[this.x].setValue(x);
174         this.spinners[this.y].setValue(1.0f - y);
175         colorChanged();
176     }
177 
178     int getColor(float z) {
179         setDefaultValue(this.x);
180         setDefaultValue(this.y);
181         this.values[this.z] = 1.0f - z;
182         return getColor(3);
183     }
184 
185     int getColor(float x, float y) {
186         this.values[this.x] = x;
187         this.values[this.y] = 1.0f - y;
188         setValue(this.z);
189         return getColor(3);
190     }
191 
192     void setColor(Color color) {
193         if (!color.equals(this.color)) {
194             this.color = color;
195             this.model.setColor(color.getRGB(), this.values);
196             for (int i = 0; i < this.model.getCount(); i++) {
197                 this.spinners[i].setValue(this.values[i]);
198             }
199         }
200     }
201 
202     private int getColor(int index) {
203         while (index < this.model.getCount()) {
204             setValue(index++);
205         }
206         return this.model.getColor(this.values);
207     }
208 
209     void setColorTransparencySelectionEnabled(boolean b) {
210         if (spinners[model.getCount() - 1].isVisible() != b) {
211             spinners[model.getCount() - 1].setVisible(b);
212             colorChanged();
213         }
214     }
215 
216     boolean isColorTransparencySelectionEnabled() {
217         return spinners[model.getCount() - 1].isVisible();
218     }
219 
220     private void setValue(int index) {
221         this.values[index] = this.spinners[index].getValue();
222     }
223 
224     private void setDefaultValue(int index) {
225         float value = this.model.getDefault(index);
226         this.values[index] = (value < 0.0f)
227                 ? this.spinners[index].getValue()
228                 : value;
229     }
230 }
