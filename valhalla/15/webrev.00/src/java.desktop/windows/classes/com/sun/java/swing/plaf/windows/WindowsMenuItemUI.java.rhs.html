 55 public class WindowsMenuItemUI extends BasicMenuItemUI {
 56     /**
 57      * The instance of {@code PropertyChangeListener}.
 58      */
 59     private PropertyChangeListener changeListener;
 60 
 61     final WindowsMenuItemUIAccessor accessor =
 62         new  WindowsMenuItemUIAccessor() {
 63 
 64             public JMenuItem getMenuItem() {
 65                 return menuItem;
 66             }
 67 
 68             public State getState(JMenuItem menuItem) {
 69                 return WindowsMenuItemUI.getState(this, menuItem);
 70             }
 71 
 72             public Part getPart(JMenuItem menuItem) {
 73                 return WindowsMenuItemUI.getPart(this, menuItem);
 74             }
 75     };
 76     public static ComponentUI createUI(JComponent c) {
 77         return new WindowsMenuItemUI();
 78     }

 80     private void updateCheckIcon() {
 81         String prefix = getPropertyPrefix();
 82 
 83         if (checkIcon == null ||
 84                 checkIcon instanceof UIResource) {
 85             checkIcon = UIManager.getIcon(prefix + ".checkIcon");
 86             //In case of column layout, .checkIconFactory is defined for this UI,
 87             //the icon is compatible with it and useCheckAndArrow() is true,
 88             //then the icon is handled by the checkIcon.
 89             boolean isColumnLayout = MenuItemLayoutHelper.isColumnLayout(
 90                     menuItem.getComponentOrientation().isLeftToRight(), menuItem);
 91             if (isColumnLayout) {
 92                 MenuItemCheckIconFactory iconFactory =
 93                         (MenuItemCheckIconFactory) UIManager.get(prefix
 94                                 + ".checkIconFactory");
 95                 if (iconFactory != null
 96                         && MenuItemLayoutHelper.useCheckAndArrow(menuItem)
 97                         && iconFactory.isCompatible(checkIcon, prefix)) {
 98                     checkIcon = iconFactory.getIcon(menuItem);
 99                 }
100             }
101         }
102     }
103 
104     /**
105      * {@inheritDoc}
106      */
107     @Override
108     protected void installListeners() {
109         super.installListeners();
110         changeListener = new PropertyChangeListener() {
111             @Override
112             public void propertyChange(PropertyChangeEvent e) {
113                 String name = e.getPropertyName();
114                 if (name == "horizontalTextPosition") {
115                     updateCheckIcon();
116                 }
117             }
118         };
119         menuItem.addPropertyChangeListener(changeListener);
120     }
121 
122     /**
123      * {@inheritDoc}
124      */
125     @Override
126     protected void uninstallListeners() {
127         super.uninstallListeners();
128         if (changeListener != null) {
129             menuItem.removePropertyChangeListener(changeListener);
130         }
131         changeListener = null;
132     }

134     /**
135      * Method which renders the text of the current menu item.
136      *
137      * @param g Graphics context
138      * @param menuItem Current menu item to render
139      * @param textRect Bounding rectangle to render the text.
140      * @param text String to render
141      */
142     protected void paintText(Graphics g, JMenuItem menuItem,
143                              Rectangle textRect, String text) {
144         if (WindowsMenuItemUI.isVistaPainting()) {
145             WindowsMenuItemUI.paintText(accessor, g, menuItem, textRect, text);
146             return;
147         }
148         ButtonModel model = menuItem.getModel();
149         Color oldColor = g.getColor();
150 
151         if(model.isEnabled() &&
152             (model.isArmed() || (menuItem instanceof JMenu &&
153              model.isSelected()))) {
154             g.setColor(selectionForeground); // Uses protected field.
155         }
156 
157         WindowsGraphicsUtils.paintText(g, menuItem, textRect, text, 0);
158 
159         g.setColor(oldColor);
160     }
161 
162     @Override
163     protected void paintBackground(Graphics g, JMenuItem menuItem,
164             Color bgColor) {
165         if (WindowsMenuItemUI.isVistaPainting()) {
166             WindowsMenuItemUI.paintBackground(accessor, g, menuItem, bgColor);
167             return;
168         }
169         super.paintBackground(g, menuItem, bgColor);
170     }
171 
172     static void paintBackground(WindowsMenuItemUIAccessor menuItemUI,
173             Graphics g, JMenuItem menuItem, Color bgColor) {
174         XPStyle xp = XPStyle.getXP();
175         assert isVistaPainting(xp);
176         if (isVistaPainting(xp)) {
177             int menuWidth = menuItem.getWidth();
178             int menuHeight = menuItem.getHeight();
179             if (menuItem.isOpaque()) {
180                 Color oldColor = g.getColor();
181                 g.setColor(menuItem.getBackground());
182                 g.fillRect(0,0, menuWidth, menuHeight);
183                 g.setColor(oldColor);
184             }
185             Part part = menuItemUI.getPart(menuItem);
186             Skin skin = xp.getSkin(menuItem, part);
187             skin.paintSkin(g, 0 , 0,
188                 menuWidth,
189                 menuHeight,
190                 menuItemUI.getState(menuItem));
191         }
192     }
193 
194     static void paintText(WindowsMenuItemUIAccessor menuItemUI, Graphics g,
195                                 JMenuItem menuItem, Rectangle textRect,
196                                 String text) {
197         assert isVistaPainting();
198         if (isVistaPainting()) {
199             State state = menuItemUI.getState(menuItem);
200 
201             /* part of it copied from WindowsGraphicsUtils.java */
202             FontMetrics fm = SwingUtilities2.getFontMetrics(menuItem, g);
203             int mnemIndex = menuItem.getDisplayedMnemonicIndex();
204             // W2K Feature: Check to see if the Underscore should be rendered.
205             if (WindowsLookAndFeel.isMnemonicHidden() == true) {
206                 mnemIndex = -1;
207             }
208             WindowsGraphicsUtils.paintXPText(menuItem,
209                 menuItemUI.getPart(menuItem), state,
210                 g, textRect.x,
211                 textRect.y + fm.getAscent(),
212                 text, mnemIndex);
213         }
214     }
215 
216     static State getState(WindowsMenuItemUIAccessor menuItemUI, JMenuItem menuItem) {
217         State state;
218         ButtonModel model = menuItem.getModel();
219         if (model.isArmed()) {
220             state = (model.isEnabled()) ? State.HOT : State.DISABLEDHOT;
221         } else {
222             state = (model.isEnabled()) ? State.NORMAL : State.DISABLED;
223         }
224         return state;
225     }
226 
227     static Part getPart(WindowsMenuItemUIAccessor menuItemUI, JMenuItem menuItem) {
228         return Part.MP_POPUPITEM;
229     }
230 
231     /*
232      * TODO idk can we use XPStyle.isVista?
233      * is it possible that in some theme some Vista parts are not defined while
234      * others are?
235      */
236     static boolean isVistaPainting(final XPStyle xp) {
237         return xp != null && xp.isSkinDefined(null, Part.MP_POPUPITEM);
238     }
239 
240     static boolean isVistaPainting() {
241         return isVistaPainting(XPStyle.getXP());
242     }
