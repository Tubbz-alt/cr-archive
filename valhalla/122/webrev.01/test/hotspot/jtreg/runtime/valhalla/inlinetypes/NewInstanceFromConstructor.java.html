25 package runtime.valhalla.inlinetypes;
26 
27 public class NewInstanceFromConstructor {
28 
29     int value;
30     static int consCalls = 0;
31 
32     public NewInstanceFromConstructor() {
33         this(0);
34     }
35 
36     public NewInstanceFromConstructor(int v) {
37         value = v;
38         consCalls++;
39     }
40 
41     public NewInstanceFromConstructor(IntValue v) {
42         this(v.getInt());
43     }
44 
45     public NewInstanceFromConstructor(IntValue v1,
46                                       IntValue v2) {
47         this(v1.getInt() + v2.getInt());
48     }
49 
50     public NewInstanceFromConstructor(IntValue v1,
51                                       String s) {
52         this(v1);
53         throw new RuntimeException(s);
54     }
55 
56     public int getValue() { return value; }
57 
58     public static int getConsCalls() { return consCalls; }
59 }
