24 package runtime.valhalla.inlinetypes;
25 
26 public final inline class Person {
27 
28     final int    id;
29     final String firstName;
30     final String lastName;
31 
32     private Person(int id, String firstName, String lastName) {
33         this.id = id;
34         this.firstName = firstName;
35         this.lastName = lastName;
36     }
37 
38     public int getId() { return id; }
39     public String getFirstName() { return firstName; }
40     public String getLastName() { return lastName; }
41 
42     public String toString() {
43         return getFirstName() + " " + getLastName() + " (id=" + getId() + ")";
44     }
45 
46     static Person create(int id, String firstName, String lastName) {
47         return new Person(id, firstName, lastName);
48     }
49 }
