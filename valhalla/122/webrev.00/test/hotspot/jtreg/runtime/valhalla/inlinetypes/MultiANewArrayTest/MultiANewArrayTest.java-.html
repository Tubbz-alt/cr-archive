24 /*
25  * @test
26  * @summary test that mismatches in bottom class of multi-dimensional
27             arrays are correctly detected
28  * @library /testlibrary /test/lib
29  * @compile MultiANewArrayTypeCheck.jcod MultiANewArrayTest.java Element0.java Element1.java
30  * @run main/othervm MultiANewArrayTest
31  */
32 
33 import jdk.test.lib.Asserts;
34 
35 public class MultiANewArrayTest {
36 
37     public static void main(String[] args) {
38         Error ex = null;
39         try {
40             MultiANewArrayTypeCheck.createArray0();
41         } catch(Error e) {
42             ex = e;
43         }
44         Asserts.assertNotNull(ex, "An ICCE should have been thrown");
45         Asserts.assertEquals(ex.getClass(), IncompatibleClassChangeError.class, "Error is not an ICCE");
46         ex = null;
47         try {
48             MultiANewArrayTypeCheck.createArray1();
49         } catch(Error e) {
50             ex = e;
51         }
52         Asserts.assertNotNull(ex, "An ICCE should have been thrown");
53         Asserts.assertEquals(ex.getClass(), IncompatibleClassChangeError.class, "Error is not an ICCE");
54         ex = null;
55     }
56 }
