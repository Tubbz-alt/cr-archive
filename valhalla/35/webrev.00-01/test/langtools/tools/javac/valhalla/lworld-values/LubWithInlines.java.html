26 /*
27  * @test
28  * @bug 8244458 8244414
29  * @summary Check that javac does not crash while computing LUB involving values.
30  * @run main LubWithInlines
31  */
32 
33 public class LubWithInlines {
34     interface I {}
35     static class Node implements I {
36     }
37     static I foo(Node e) {
38         var ret = (e == null) ? new XNodeWrapper() : e;
39         return ret;
40     }
41     static inline class XNodeWrapper implements I {
42         int i = 42;
43     }
44     public static void main(String [] args) {
45         I i = foo(null);
46         if (!i.toString().equals("[LubWithInlines$XNodeWrapper i=42]"))
47             throw new AssertionError("Unexpected: " + i);
48     }
49 }
