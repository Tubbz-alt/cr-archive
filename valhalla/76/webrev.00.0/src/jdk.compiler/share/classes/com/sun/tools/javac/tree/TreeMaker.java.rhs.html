  54     /** The context key for the tree factory. */
  55     protected static final Context.Key<TreeMaker> treeMakerKey = new Context.Key<>();
  56 
  57     /** Get the TreeMaker instance. */
  58     public static TreeMaker instance(Context context) {
  59         TreeMaker instance = context.get(treeMakerKey);
  60         if (instance == null)
  61             instance = new TreeMaker(context);
  62         return instance;
  63     }
  64 
  65     /** The position at which subsequent trees will be created.
  66      */
  67     public int pos = Position.NOPOS;
  68 
  69     /** The toplevel tree to which created trees belong.
  70      */
  71     public JCCompilationUnit toplevel;
  72 
  73     /** The current name table. */
  74     Names names;
  75 
  76     Types types;
  77 
  78     /** The current symbol table. */
  79     Symtab syms;
  80 
  81     /** Create a tree maker with null toplevel and NOPOS as initial position.
  82      */
  83     protected TreeMaker(Context context) {
  84         context.put(treeMakerKey, this);
  85         this.pos = Position.NOPOS;
  86         this.toplevel = null;
  87         this.names = Names.instance(context);
  88         this.syms = Symtab.instance(context);
  89         this.types = Types.instance(context);
  90     }
  91 
  92     /** Create a tree maker with a given toplevel and FIRSTPOS as initial position.
  93      */
  94     protected TreeMaker(JCCompilationUnit toplevel, Names names, Types types, Symtab syms) {
  95         this.pos = Position.FIRSTPOS;
  96         this.toplevel = toplevel;
  97         this.names = names;
  98         this.types = types;
  99         this.syms = syms;
 100     }

 122     /**
 123      * Create given tree node at current position.
 124      * @param defs a list of PackageDef, ClassDef, Import, and Skip
 125      */
 126     public JCCompilationUnit TopLevel(List<JCTree> defs) {
 127         for (JCTree node : defs)
 128             Assert.check(node instanceof JCClassDecl
 129                 || node instanceof JCPackageDecl
 130                 || node instanceof JCImport
 131                 || node instanceof JCModuleDecl
 132                 || node instanceof JCSkip
 133                 || node instanceof JCErroneous
 134                 || (node instanceof JCExpressionStatement
 135                     && ((JCExpressionStatement)node).expr instanceof JCErroneous),
 136                     () -> node.getClass().getSimpleName());
 137         JCCompilationUnit tree = new JCCompilationUnit(defs);
 138         tree.pos = pos;
 139         return tree;
 140     }

 785     /** Create a tree representing given type.
 786      */
 787     public JCExpression Type(Type t) {
 788         if (t == null) return null;
 789         JCExpression tp;
 790         switch (t.getTag()) {
 791         case BYTE: case CHAR: case SHORT: case INT: case LONG: case FLOAT:
 792         case DOUBLE: case BOOLEAN: case VOID:
 793             tp = TypeIdent(t.getTag());
 794             break;
 795         case TYPEVAR:
 796             tp = Ident(t.tsym);
 797             break;
 798         case WILDCARD: {
 799             WildcardType a = ((WildcardType) t);
 800             tp = Wildcard(TypeBoundKind(a.kind), a.kind == BoundKind.UNBOUND ? null : Type(a.type));
 801             break;
 802         }
 803         case CLASS:
 804             switch (t.getKind()) {
 805             case UNION: {
 806                 UnionClassType tu = (UnionClassType)t;
 807                 ListBuffer<JCExpression> la = new ListBuffer<>();
 808                 for (Type ta : tu.getAlternativeTypes()) {
 809                     la.add(Type(ta));
 810                 }
 811                 tp = TypeUnion(la.toList());
 812                 break;
 813             }
 814             case INTERSECTION: {
 815                 IntersectionClassType it = (IntersectionClassType)t;
 816                 ListBuffer<JCExpression> la = new ListBuffer<>();
 817                 for (Type ta : it.getExplicitComponents()) {
 818                     la.add(Type(ta));
 819                 }
 820                 tp = TypeIntersection(la.toList());
 821                 break;
 822             }
 823             default: {
 824                 if (t.isReferenceProjection()) {
 825                     // For parameterized types, we want V.ref<A1 ... An> not V<A1 ... An>.ref
 826                     JCExpression vp = Type(t.valueProjection());
 827                     if (vp.hasTag(Tag.TYPEAPPLY)) {
 828                         // vp now is V<A1 ... An>, build V.ref<A1 ... An>
 829                         JCFieldAccess f = (JCFieldAccess) Select(((JCTypeApply) vp).clazz, t.tsym);
 830                         f.name = names.ref;
 831                         tp = TypeApply(f, ((JCTypeApply) vp).arguments);
 832                     } else {
 833                         JCFieldAccess f = (JCFieldAccess) Select(vp, t.tsym);
 834                         f.name = names.ref;
 835                         tp = f;
 836                     }
 837                 } else {
 838                     Type outer = t.getEnclosingType();
 839                     JCExpression clazz = outer.hasTag(CLASS) && t.tsym.owner.kind == TYP
 840                             ? Select(Type(outer), t.tsym)
 841                             : QualIdent(t.tsym);
 842                     tp = t.getTypeArguments().isEmpty()
 843                             ? clazz
 844                             : TypeApply(clazz, Types(t.getTypeArguments()));
 845                 }
 846                 break;
 847             }
 848             }
 849             break;
 850         case ARRAY:
 851             tp = TypeArray(Type(types.elemtype(t)));
 852             break;
 853         case ERROR:
 854             tp = TypeIdent(ERROR);
 855             break;
 856         default:
 857             throw new AssertionError("unexpected type: " + t);
 858         }
 859         return tp.setType(t);
 860     }

 896     public JCLiteral Literal(Object value) {
 897         JCLiteral result = null;
 898         if (value instanceof String) {
 899             result = Literal(CLASS, value).
 900                 setType(syms.stringType.constType(value));
 901         } else if (value instanceof Integer) {
 902             result = Literal(INT, value).
 903                 setType(syms.intType.constType(value));
 904         } else if (value instanceof Long) {
 905             result = Literal(LONG, value).
 906                 setType(syms.longType.constType(value));
 907         } else if (value instanceof Byte) {
 908             result = Literal(BYTE, value).
 909                 setType(syms.byteType.constType(value));
 910         } else if (value instanceof Character) {
 911             int v = (int) (((Character) value).toString().charAt(0));
 912             result = Literal(CHAR, v).
 913                 setType(syms.charType.constType(v));
 914         } else if (value instanceof Double) {
 915             result = Literal(DOUBLE, value).
 916                 setType(syms.doubleType.constType(value));
 917         } else if (value instanceof Float) {
 918             result = Literal(FLOAT, value).
 919                 setType(syms.floatType.constType(value));
 920         } else if (value instanceof Short) {
 921             result = Literal(SHORT, value).
 922                 setType(syms.shortType.constType(value));
 923         } else if (value instanceof Boolean) {
 924             int v = ((Boolean) value) ? 1 : 0;
 925             result = Literal(BOOLEAN, v).
 926                 setType(syms.booleanType.constType(v));
 927         } else {
 928             throw new AssertionError(value);
 929         }
 930         return result;
 931     }

 933     class AnnotationBuilder implements Attribute.Visitor {
 934         JCExpression result = null;
 935         public void visitConstant(Attribute.Constant v) {
 936             result = Literal(v.type.getTag(), v.value);
 937         }
 938         public void visitClass(Attribute.Class clazz) {
 939             result = ClassLiteral(clazz.classType).setType(syms.classType);
 940         }
 941         public void visitEnum(Attribute.Enum e) {
 942             result = QualIdent(e.value);
 943         }
 944         public void visitError(Attribute.Error e) {
 945             if (e instanceof UnresolvedClass) {
 946                 result = ClassLiteral(((UnresolvedClass) e).classType).setType(syms.classType);
 947             } else {
 948                 result = Erroneous();
 949             }
 950         }
 951         public void visitCompound(Attribute.Compound compound) {
 952             if (compound instanceof Attribute.TypeCompound) {
 953                 result = visitTypeCompoundInternal((Attribute.TypeCompound) compound);
 954             } else {
 955                 result = visitCompoundInternal(compound);
 956             }
 957         }
 958         public JCAnnotation visitCompoundInternal(Attribute.Compound compound) {
 959             ListBuffer<JCExpression> args = new ListBuffer<>();
 960             for (List<Pair<Symbol.MethodSymbol,Attribute>> values = compound.values; values.nonEmpty(); values=values.tail) {
 961                 Pair<MethodSymbol,Attribute> pair = values.head;
 962                 JCExpression valueTree = translate(pair.snd);
 963                 args.append(Assign(Ident(pair.fst), valueTree).setType(valueTree.type));
 964             }
 965             return Annotation(Type(compound.type), args.toList());
 966         }
 967         public JCAnnotation visitTypeCompoundInternal(Attribute.TypeCompound compound) {
 968             ListBuffer<JCExpression> args = new ListBuffer<>();
 969             for (List<Pair<Symbol.MethodSymbol,Attribute>> values = compound.values; values.nonEmpty(); values=values.tail) {
 970                 Pair<MethodSymbol,Attribute> pair = values.head;
 971                 JCExpression valueTree = translate(pair.snd);
 972                 args.append(Assign(Ident(pair.fst), valueTree).setType(valueTree.type));
 973             }
 974             return TypeAnnotation(Type(compound.type), args.toList());
 975         }
 976         public void visitArray(Attribute.Array array) {
 977             ListBuffer<JCExpression> elems = new ListBuffer<>();
 978             for (int i = 0; i < array.values.length; i++)
 979                 elems.append(translate(array.values[i]));
 980             result = NewArray(null, List.nil(), elems.toList()).setType(array.type);
 981         }
 982         JCExpression translate(Attribute a) {
 983             a.accept(this);
 984             return result;
 985         }
 986         JCAnnotation translate(Attribute.Compound a) {
 987             return visitCompoundInternal(a);
 988         }
 989         JCAnnotation translate(Attribute.TypeCompound a) {
 990             return visitTypeCompoundInternal(a);
 991         }
 992     }

1064                 params.append(Param(paramName(i++), l.head, owner));

1100     /** Can given symbol be referred to in unqualified form?
1101      */
1102     boolean isUnqualifiable(Symbol sym) {
1103         if (sym.name == names.empty ||
1104             sym.owner == null ||
1105             sym.owner == syms.rootPackage ||
1106             sym.owner.kind == MTH || sym.owner.kind == VAR) {
1107             return true;
1108         } else if (sym.kind == TYP && toplevel != null) {
1109             Iterator<Symbol> it = toplevel.namedImportScope.getSymbolsByName(sym.name).iterator();
1110             if (it.hasNext()) {
1111                 Symbol s = it.next();
1112                 return
1113                   s == sym &&
1114                   !it.hasNext();
1115             }
1116             it = toplevel.packge.members().getSymbolsByName(sym.name).iterator();
1117             if (it.hasNext()) {
1118                 Symbol s = it.next();
1119                 return
1120                   s == sym &&
1121                   !it.hasNext();
1122             }
1123             it = toplevel.starImportScope.getSymbolsByName(sym.name).iterator();
1124             if (it.hasNext()) {
1125                 Symbol s = it.next();
1126                 return
1127                   s == sym &&
1128                   !it.hasNext();
1129             }
1130         }
1131         return false;
1132     }
