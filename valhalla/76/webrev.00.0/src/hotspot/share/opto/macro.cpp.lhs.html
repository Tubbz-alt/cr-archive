  64 //
  65 // Replace any references to "oldref" in inputs to "use" with "newref".
  66 // Returns the number of replacements made.
  67 //
  68 int PhaseMacroExpand::replace_input(Node *use, Node *oldref, Node *newref) {
  69   int nreplacements = 0;
  70   uint req = use->req();
  71   for (uint j = 0; j < use->len(); j++) {
  72     Node *uin = use->in(j);
  73     if (uin == oldref) {
  74       if (j < req)
  75         use->set_req(j, newref);
  76       else
  77         use->set_prec(j, newref);
  78       nreplacements++;
  79     } else if (j >= req && uin == NULL) {
  80       break;
  81     }
  82   }
  83   return nreplacements;
  84 }

  86 Node* PhaseMacroExpand::opt_bits_test(Node* ctrl, Node* region, int edge, Node* word, int mask, int bits, bool return_fast_path) {
  87   Node* cmp;
  88   if (mask != 0) {
  89     Node* and_node = transform_later(new AndXNode(word, MakeConX(mask)));
  90     cmp = transform_later(new CmpXNode(and_node, MakeConX(bits)));
  91   } else {
  92     cmp = word;
  93   }
  94   Node* bol = transform_later(new BoolNode(cmp, BoolTest::ne));
  95   IfNode* iff = new IfNode( ctrl, bol, PROB_MIN, COUNT_UNKNOWN );
  96   transform_later(iff);
  97 
  98   // Fast path taken.
  99   Node *fast_taken = transform_later(new IfFalseNode(iff));
 100 
 101   // Fast path not-taken, i.e. slow path
 102   Node *slow_taken = transform_later(new IfTrueNode(iff));
 103 
 104   if (return_fast_path) {
 105     region->init_req(edge, slow_taken); // Capture slow-control
 106     return fast_taken;
 107   } else {
 108     region->init_req(edge, fast_taken); // Capture fast-control
 109     return slow_taken;
 110   }
 111 }

 123 //------------------------------make_slow_call---------------------------------
 124 CallNode* PhaseMacroExpand::make_slow_call(CallNode *oldcall, const TypeFunc* slow_call_type,
 125                                            address slow_call, const char* leaf_name, Node* slow_path,
 126                                            Node* parm0, Node* parm1, Node* parm2) {
 127 
 128   // Slow-path call
 129  CallNode *call = leaf_name
 130    ? (CallNode*)new CallLeafNode      ( slow_call_type, slow_call, leaf_name, TypeRawPtr::BOTTOM )
 131    : (CallNode*)new CallStaticJavaNode( slow_call_type, slow_call, OptoRuntime::stub_name(slow_call), oldcall->jvms()->bci(), TypeRawPtr::BOTTOM );
 132 
 133   // Slow path call has no side-effects, uses few values
 134   copy_predefined_input_for_runtime_call(slow_path, oldcall, call );
 135   if (parm0 != NULL)  call->init_req(TypeFunc::Parms+0, parm0);
 136   if (parm1 != NULL)  call->init_req(TypeFunc::Parms+1, parm1);
 137   if (parm2 != NULL)  call->init_req(TypeFunc::Parms+2, parm2);
 138   call->copy_call_debug_info(&_igvn, oldcall);
 139   call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
 140   _igvn.replace_node(oldcall, call);
 141   transform_later(call);
 142 
 143   return call;
 144 }

 146 void PhaseMacroExpand::extract_call_projections(CallNode *call) {
 147   _fallthroughproj = NULL;
 148   _fallthroughcatchproj = NULL;
 149   _ioproj_fallthrough = NULL;
 150   _ioproj_catchall = NULL;
 151   _catchallcatchproj = NULL;
 152   _memproj_fallthrough = NULL;
 153   _memproj_catchall = NULL;
 154   _resproj = NULL;
 155   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
 156     ProjNode *pn = call->fast_out(i)->as_Proj();
 157     switch (pn->_con) {
 158       case TypeFunc::Control:
 159       {
 160         // For Control (fallthrough) and I_O (catch_all_index) we have CatchProj -> Catch -> Proj
 161         _fallthroughproj = pn;
 162         DUIterator_Fast jmax, j = pn->fast_outs(jmax);
 163         const Node *cn = pn->fast_out(j);
 164         if (cn->is_Catch()) {
 165           ProjNode *cpn = NULL;
 166           for (DUIterator_Fast kmax, k = cn->fast_outs(kmax); k < kmax; k++) {
 167             cpn = cn->fast_out(k)->as_Proj();
 168             assert(cpn->is_CatchProj(), "must be a CatchProjNode");
 169             if (cpn->_con == CatchProjNode::fall_through_index)
 170               _fallthroughcatchproj = cpn;
 171             else {
 172               assert(cpn->_con == CatchProjNode::catch_all_index, "must be correct index.");
 173               _catchallcatchproj = cpn;
 174             }
 175           }
 176         }
 177         break;
 178       }
 179       case TypeFunc::I_O:
 180         if (pn->_is_io_use)
 181           _ioproj_catchall = pn;
 182         else
 183           _ioproj_fallthrough = pn;
 184         break;
 185       case TypeFunc::Memory:
 186         if (pn->_is_io_use)
 187           _memproj_catchall = pn;
 188         else
 189           _memproj_fallthrough = pn;
 190         break;
 191       case TypeFunc::Parms:
 192         _resproj = pn;
 193         break;
 194       default:
 195         assert(false, "unexpected projection from allocation node.");
 196     }
 197   }
 198

 206 // Search for a memory operation for the specified memory slice.
 207 static Node *scan_mem_chain(Node *mem, int alias_idx, int offset, Node *start_mem, Node *alloc, PhaseGVN *phase) {
 208   Node *orig_mem = mem;
 209   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 210   const TypeOopPtr *tinst = phase->C->get_adr_type(alias_idx)->isa_oopptr();
 211   while (true) {
 212     if (mem == alloc_mem || mem == start_mem ) {
 213       return mem;  // hit one of our sentinels
 214     } else if (mem->is_MergeMem()) {
 215       mem = mem->as_MergeMem()->memory_at(alias_idx);
 216     } else if (mem->is_Proj() && mem->as_Proj()->_con == TypeFunc::Memory) {
 217       Node *in = mem->in(0);
 218       // we can safely skip over safepoints, calls, locks and membars because we
 219       // already know that the object is safe to eliminate.
 220       if (in->is_Initialize() && in->as_Initialize()->allocation() == alloc) {
 221         return in;
 222       } else if (in->is_Call()) {
 223         CallNode *call = in->as_Call();
 224         if (call->may_modify(tinst, phase)) {
 225           assert(call->is_ArrayCopy(), "ArrayCopy is the only call node that doesn't make allocation escape");
 226           if (call->as_ArrayCopy()->modifies(offset, offset, phase, false)) {
 227             return in;
 228           }
 229         }
 230         mem = in->in(TypeFunc::Memory);
 231       } else if (in->is_MemBar()) {
 232         ArrayCopyNode* ac = NULL;
 233         if (ArrayCopyNode::may_modify(tinst, in->as_MemBar(), phase, ac)) {
 234           assert(ac != NULL && ac->is_clonebasic(), "Only basic clone is a non escaping clone");
 235           return ac;
 236         }
 237         mem = in->in(TypeFunc::Memory);
 238       } else {
 239         assert(false, "unexpected projection");
 240       }
 241     } else if (mem->is_Store()) {
 242       const TypePtr* atype = mem->as_Store()->adr_type();
 243       int adr_idx = phase->C->get_alias_index(atype);
 244       if (adr_idx == alias_idx) {
 245         assert(atype->isa_oopptr(), "address type must be oopptr");
 246         int adr_offset = atype->flattened_offset();
 247         uint adr_iid = atype->is_oopptr()->instance_id();
 248         // Array elements references have the same alias_idx
 249         // but different offset and different instance_id.
 250         if (adr_offset == offset && adr_iid == alloc->_idx)
 251           return mem;
 252       } else {
 253         assert(adr_idx == Compile::AliasIdxRaw, "address must match or be raw");
 254       }
 255       mem = mem->in(MemNode::Memory);
 256     } else if (mem->is_ClearArray()) {
 257       if (!ClearArrayNode::step_through(&mem, alloc->_idx, phase)) {
 258         // Can not bypass initialization of the instance
 259         // we are looking.
 260         debug_only(intptr_t offset;)
 261         assert(alloc == AllocateNode::Ideal_allocation(mem->in(3), phase, offset), "sanity");
 262         InitializeNode* init = alloc->as_Allocate()->initialization();
 263         // We are looking for stored value, return Initialize node
 264         // or memory edge from Allocate node.
 265         if (init != NULL)
 266           return init;
 267         else
 268           return alloc->in(TypeFunc::Memory); // It will produce zero value (see callers).
 269       }
 270       // Otherwise skip it (the call updated 'mem' value).
 271     } else if (mem->Opcode() == Op_SCMemProj) {
 272       mem = mem->in(0);
 273       Node* adr = NULL;
 274       if (mem->is_LoadStore()) {
 275         adr = mem->in(MemNode::Address);
 276       } else {
 277         assert(mem->Opcode() == Op_EncodeISOArray ||
 278                mem->Opcode() == Op_StrCompressedCopy, "sanity");
 279         adr = mem->in(3); // Destination array
 280       }
 281       const TypePtr* atype = adr->bottom_type()->is_ptr();
 282       int adr_idx = phase->C->get_alias_index(atype);
 283       if (adr_idx == alias_idx) {
 284         DEBUG_ONLY(mem->dump();)
 285         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 286         return NULL;
 287       }
 288       mem = mem->in(MemNode::Memory);
 289     } else if (mem->Opcode() == Op_StrInflatedCopy) {
 290       Node* adr = mem->in(3); // Destination array
 291       const TypePtr* atype = adr->bottom_type()->is_ptr();
 292       int adr_idx = phase->C->get_alias_index(atype);
 293       if (adr_idx == alias_idx) {
 294         DEBUG_ONLY(mem->dump();)
 295         assert(false, "Object is not scalar replaceable if a StrInflatedCopy node accesses its field");
 296         return NULL;
 297       }
 298       mem = mem->in(MemNode::Memory);
 299     } else {
 300       return mem;
 301     }
 302     assert(mem != orig_mem, "dead memory loop");
 303   }
 304 }

 306 // Generate loads from source of the arraycopy for fields of
 307 // destination needed at a deoptimization point
 308 Node* PhaseMacroExpand::make_arraycopy_load(ArrayCopyNode* ac, intptr_t offset, Node* ctl, Node* mem, BasicType ft, const Type *ftype, AllocateNode *alloc) {
 309   BasicType bt = ft;
 310   const Type *type = ftype;
 311   if (ft == T_NARROWOOP) {
 312     bt = T_OBJECT;
 313     type = ftype->make_oopptr();
 314   }
 315   Node* res = NULL;
 316   if (ac->is_clonebasic()) {
 317     assert(ac->in(ArrayCopyNode::Src) != ac->in(ArrayCopyNode::Dest), "clone source equals destination");
 318     Node* base = ac->in(ArrayCopyNode::Src);
 319     Node* adr = _igvn.transform(new AddPNode(base, base, MakeConX(offset)));
 320     const TypePtr* adr_type = _igvn.type(base)->is_ptr()->add_offset(offset);
 321     res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 322   } else {
 323     if (ac->modifies(offset, offset, &_igvn, true)) {
 324       assert(ac->in(ArrayCopyNode::Dest) == alloc->result_cast(), "arraycopy destination should be allocation's result");
 325       uint shift = exact_log2(type2aelembytes(bt));
 326       Node* src_pos = ac->in(ArrayCopyNode::SrcPos);
 327       Node* dest_pos = ac->in(ArrayCopyNode::DestPos);
 328       const TypeInt* src_pos_t = _igvn.type(src_pos)->is_int();
 329       const TypeInt* dest_pos_t = _igvn.type(dest_pos)->is_int();
 330 
 331       Node* adr = NULL;
 332       Node* base = ac->in(ArrayCopyNode::Src);
 333       const TypePtr* adr_type = _igvn.type(base)->is_ptr();
 334       assert(adr_type->isa_aryptr(), "only arrays here");
 335       if (src_pos_t->is_con() && dest_pos_t->is_con()) {
 336         intptr_t off = ((src_pos_t->get_con() - dest_pos_t->get_con()) << shift) + offset;
 337         adr = _igvn.transform(new AddPNode(base, base, MakeConX(off)));
 338         adr_type = _igvn.type(adr)->is_ptr();
 339         assert(adr_type == _igvn.type(base)->is_aryptr()->add_field_offset_and_offset(off), "incorrect address type");
 340         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 341           // Don't emit a new load from src if src == dst but try to get the value from memory instead
 342           return value_from_mem(ac->in(TypeFunc::Memory), ctl, ft, ftype, adr_type->isa_oopptr(), alloc);
 343         }
 344       } else {
 345         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 346           // Non constant offset in the array: we can't statically
 347           // determine the value
 348           return NULL;
 349         }
 350         Node* diff = _igvn.transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 351 #ifdef _LP64
 352         diff = _igvn.transform(new ConvI2LNode(diff));
 353 #endif
 354         diff = _igvn.transform(new LShiftXNode(diff, intcon(shift)));
 355 
 356         Node* off = _igvn.transform(new AddXNode(MakeConX(offset), diff));
 357         adr = _igvn.transform(new AddPNode(base, base, off));
 358         // In the case of a flattened value type array, each field has its
 359         // own slice so we need to extract the field being accessed from
 360         // the address computation
 361         adr_type = adr_type->is_aryptr()->add_field_offset_and_offset(offset)->add_offset(Type::OffsetBot);
 362         adr = _igvn.transform(new CastPPNode(adr, adr_type));
 363       }
 364       res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 365     }
 366   }
 367   if (res != NULL) {
 368     res = _igvn.transform(res);
 369     if (ftype->isa_narrowoop()) {
 370       // PhaseMacroExpand::scalar_replacement adds DecodeN nodes
 371       assert(res->isa_DecodeN(), "should be narrow oop");
 372       res = _igvn.transform(new EncodePNode(res, ftype));
 373     }
 374     return res;
 375   }
 376   return NULL;
 377 }

 379 //
 380 // Given a Memory Phi, compute a value Phi containing the values from stores
 381 // on the input paths.
 382 // Note: this function is recursive, its depth is limited by the "level" argument
 383 // Returns the computed Phi, or NULL if it cannot compute it.
 384 Node *PhaseMacroExpand::value_from_mem_phi(Node *mem, BasicType ft, const Type *phi_type, const TypeOopPtr *adr_t, AllocateNode *alloc, Node_Stack *value_phis, int level) {
 385   assert(mem->is_Phi(), "sanity");
 386   int alias_idx = C->get_alias_index(adr_t);
 387   int offset = adr_t->flattened_offset();
 388   int instance_id = adr_t->instance_id();
 389 
 390   // Check if an appropriate value phi already exists.
 391   Node* region = mem->in(0);
 392   for (DUIterator_Fast kmax, k = region->fast_outs(kmax); k < kmax; k++) {
 393     Node* phi = region->fast_out(k);
 394     if (phi->is_Phi() && phi != mem &&
 395         phi->as_Phi()->is_same_inst_field(phi_type, (int)mem->_idx, instance_id, alias_idx, offset)) {
 396       return phi;
 397     }
 398   }
 399   // Check if an appropriate new value phi already exists.
 400   Node* new_phi = value_phis->find(mem->_idx);
 401   if (new_phi != NULL)
 402     return new_phi;
 403 
 404   if (level <= 0) {
 405     return NULL; // Give up: phi tree too deep
 406   }
 407   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 408   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 409 
 410   uint length = mem->req();
 411   GrowableArray <Node *> values(length, length, NULL, false);
 412 
 413   // create a new Phi for the value
 414   PhiNode *phi = new PhiNode(mem->in(0), phi_type, NULL, mem->_idx, instance_id, alias_idx, offset);
 415   transform_later(phi);
 416   value_phis->push(phi, mem->_idx);
 417 
 418   for (uint j = 1; j < length; j++) {
 419     Node *in = mem->in(j);
 420     if (in == NULL || in->is_top()) {
 421       values.at_put(j, in);
 422     } else  {
 423       Node *val = scan_mem_chain(in, alias_idx, offset, start_mem, alloc, &_igvn);
 424       if (val == start_mem || val == alloc_mem) {
 425         // hit a sentinel, return appropriate 0 value
 426         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 427         if (default_value != NULL) {
 428           values.at_put(j, default_value);
 429         } else {
 430           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 431           values.at_put(j, _igvn.zerocon(ft));
 432         }
 433         continue;
 434       }
 435       if (val->is_Initialize()) {
 436         val = val->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 437       }
 438       if (val == NULL) {
 439         return NULL;  // can't find a value on this path
 440       }
 441       if (val == mem) {
 442         values.at_put(j, mem);
 443       } else if (val->is_Store()) {
 444         Node* n = val->in(MemNode::ValueIn);
 445         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 446         n = bs->step_over_gc_barrier(n);
 447         values.at_put(j, n);
 448       } else if(val->is_Proj() && val->in(0) == alloc) {
 449         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 450         if (default_value != NULL) {
 451           values.at_put(j, default_value);
 452         } else {
 453           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 454           values.at_put(j, _igvn.zerocon(ft));
 455         }
 456       } else if (val->is_Phi()) {
 457         val = value_from_mem_phi(val, ft, phi_type, adr_t, alloc, value_phis, level-1);
 458         if (val == NULL) {
 459           return NULL;
 460         }
 461         values.at_put(j, val);
 462       } else if (val->Opcode() == Op_SCMemProj) {
 463         assert(val->in(0)->is_LoadStore() ||
 464                val->in(0)->Opcode() == Op_EncodeISOArray ||
 465                val->in(0)->Opcode() == Op_StrCompressedCopy, "sanity");
 466         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 467         return NULL;
 468       } else if (val->is_ArrayCopy()) {
 469         Node* res = make_arraycopy_load(val->as_ArrayCopy(), offset, val->in(0), val->in(TypeFunc::Memory), ft, phi_type, alloc);
 470         if (res == NULL) {
 471           return NULL;
 472         }
 473         values.at_put(j, res);
 474       } else {
 475 #ifdef ASSERT
 476         val->dump();
 477         assert(false, "unknown node on this path");
 478 #endif
 479         return NULL;  // unknown node on this path
 480       }
 481     }
 482   }
 483   // Set Phi's inputs
 484   for (uint j = 1; j < length; j++) {
 485     if (values.at(j) == mem) {
 486       phi->init_req(j, phi);
 487     } else {
 488       phi->init_req(j, values.at(j));
 489     }
 490   }
 491   return phi;
 492 }

 494 // Search the last value stored into the object's field.
 495 Node *PhaseMacroExpand::value_from_mem(Node *sfpt_mem, Node *sfpt_ctl, BasicType ft, const Type *ftype, const TypeOopPtr *adr_t, AllocateNode *alloc) {
 496   assert(adr_t->is_known_instance_field(), "instance required");
 497   int instance_id = adr_t->instance_id();
 498   assert((uint)instance_id == alloc->_idx, "wrong allocation");
 499 
 500   int alias_idx = C->get_alias_index(adr_t);
 501   int offset = adr_t->flattened_offset();
 502   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 503   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 504   Arena *a = Thread::current()->resource_area();
 505   VectorSet visited(a);
 506 
 507   bool done = sfpt_mem == alloc_mem;
 508   Node *mem = sfpt_mem;
 509   while (!done) {
 510     if (visited.test_set(mem->_idx)) {
 511       return NULL;  // found a loop, give up
 512     }
 513     mem = scan_mem_chain(mem, alias_idx, offset, start_mem, alloc, &_igvn);
 514     if (mem == start_mem || mem == alloc_mem) {
 515       done = true;  // hit a sentinel, return appropriate 0 value
 516     } else if (mem->is_Initialize()) {
 517       mem = mem->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 518       if (mem == NULL) {
 519         done = true; // Something went wrong.
 520       } else if (mem->is_Store()) {
 521         const TypePtr* atype = mem->as_Store()->adr_type();
 522         assert(C->get_alias_index(atype) == Compile::AliasIdxRaw, "store is correct memory slice");
 523         done = true;
 524       }
 525     } else if (mem->is_Store()) {
 526       const TypeOopPtr* atype = mem->as_Store()->adr_type()->isa_oopptr();
 527       assert(atype != NULL, "address type must be oopptr");
 528       assert(C->get_alias_index(atype) == alias_idx &&
 529              atype->is_known_instance_field() && atype->flattened_offset() == offset &&
 530              atype->instance_id() == instance_id, "store is correct memory slice");
 531       done = true;
 532     } else if (mem->is_Phi()) {
 533       // try to find a phi's unique input
 534       Node *unique_input = NULL;
 535       Node *top = C->top();
 536       for (uint i = 1; i < mem->req(); i++) {
 537         Node *n = scan_mem_chain(mem->in(i), alias_idx, offset, start_mem, alloc, &_igvn);
 538         if (n == NULL || n == top || n == mem) {
 539           continue;
 540         } else if (unique_input == NULL) {
 541           unique_input = n;
 542         } else if (unique_input != n) {
 543           unique_input = top;
 544           break;
 545         }
 546       }
 547       if (unique_input != NULL && unique_input != top) {
 548         mem = unique_input;
 549       } else {
 550         done = true;
 551       }
 552     } else if (mem->is_ArrayCopy()) {
 553       done = true;
 554     } else {
 555       assert(false, "unexpected node");
 556     }
 557   }
 558   if (mem != NULL) {
 559     if (mem == start_mem || mem == alloc_mem) {
 560       // hit a sentinel, return appropriate 0 value
 561       Node* default_value = alloc->in(AllocateNode::DefaultValue);
 562       if (default_value != NULL) {
 563         return default_value;
 564       }
 565       assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 566       return _igvn.zerocon(ft);
 567     } else if (mem->is_Store()) {
 568       Node* n = mem->in(MemNode::ValueIn);
 569       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 570       n = bs->step_over_gc_barrier(n);
 571       return n;
 572     } else if (mem->is_Phi()) {
 573       // attempt to produce a Phi reflecting the values on the input paths of the Phi
 574       Node_Stack value_phis(a, 8);
 575       Node * phi = value_from_mem_phi(mem, ft, ftype, adr_t, alloc, &value_phis, ValueSearchLimit);
 576       if (phi != NULL) {
 577         return phi;
 578       } else {
 579         // Kill all new Phis
 580         while(value_phis.is_nonempty()) {
 581           Node* n = value_phis.node();
 582           _igvn.replace_node(n, C->top());
 583           value_phis.pop();
 584         }
 585       }
 586     } else if (mem->is_ArrayCopy()) {
 587       Node* ctl = mem->in(0);
 588       Node* m = mem->in(TypeFunc::Memory);
 589       if (sfpt_ctl->is_Proj() && sfpt_ctl->as_Proj()->is_uncommon_trap_proj(Deoptimization::Reason_none)) {
 590         // pin the loads in the uncommon trap path
 591         ctl = sfpt_ctl;
 592         m = sfpt_mem;
 593       }
 594       return make_arraycopy_load(mem->as_ArrayCopy(), offset, ctl, m, ft, ftype, alloc);
 595     }
 596   }
 597   // Something went wrong.
 598   return NULL;
 599 }

 601 // Search the last value stored into the value type's fields.
 602 Node* PhaseMacroExpand::value_type_from_mem(Node* mem, Node* ctl, ciValueKlass* vk, const TypeAryPtr* adr_type, int offset, AllocateNode* alloc) {
 603   // Subtract the offset of the first field to account for the missing oop header
 604   offset -= vk->first_field_offset();
 605   // Create a new ValueTypeNode and retrieve the field values from memory
 606   ValueTypeNode* vt = ValueTypeNode::make_uninitialized(_igvn, vk)->as_ValueType();
 607   for (int i = 0; i < vk->nof_declared_nonstatic_fields(); ++i) {
 608     ciType* field_type = vt->field_type(i);
 609     int field_offset = offset + vt->field_offset(i);
 610     // Each value type field has its own memory slice
 611     adr_type = adr_type->with_field_offset(field_offset);
 612     Node* value = NULL;
 613     if (vt->field_is_flattened(i)) {
 614       value = value_type_from_mem(mem, ctl, field_type->as_value_klass(), adr_type, field_offset, alloc);
 615     } else {
 616       const Type* ft = Type::get_const_type(field_type);
 617       BasicType bt = field_type->basic_type();
 618       if (UseCompressedOops && !is_java_primitive(bt)) {
 619         ft = ft->make_narrowoop();
 620         bt = T_NARROWOOP;
 621       }
 622       value = value_from_mem(mem, ctl, bt, ft, adr_type, alloc);
 623       if (value != NULL && ft->isa_narrowoop()) {
 624         assert(UseCompressedOops, "unexpected narrow oop");
 625         value = transform_later(new DecodeNNode(value, value->get_ptr_type()));
 626       }
 627     }
 628     if (value != NULL) {
 629       vt->set_field_value(i, value);
 630     } else {
 631       // We might have reached the TrackedInitializationLimit
 632       return NULL;
 633     }
 634   }
 635   return transform_later(vt);
 636 }

 638 // Check the possibility of scalar replacement.
 639 bool PhaseMacroExpand::can_eliminate_allocation(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 640   //  Scan the uses of the allocation to check for anything that would
 641   //  prevent us from eliminating it.
 642   NOT_PRODUCT( const char* fail_eliminate = NULL; )
 643   DEBUG_ONLY( Node* disq_node = NULL; )
 644   bool  can_eliminate = true;
 645 
 646   Node* res = alloc->result_cast();
 647   const TypeOopPtr* res_type = NULL;
 648   if (res == NULL) {
 649     // All users were eliminated.
 650   } else if (!res->is_CheckCastPP()) {
 651     NOT_PRODUCT(fail_eliminate = "Allocation does not have unique CheckCastPP";)
 652     can_eliminate = false;
 653   } else {
 654     res_type = _igvn.type(res)->isa_oopptr();
 655     if (res_type == NULL) {
 656       NOT_PRODUCT(fail_eliminate = "Neither instance or array allocation";)
 657       can_eliminate = false;
 658     } else if (res_type->isa_aryptr()) {
 659       int length = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 660       if (length < 0) {
 661         NOT_PRODUCT(fail_eliminate = "Array's size is not constant";)
 662         can_eliminate = false;
 663       }
 664     }
 665   }
 666 
 667   if (can_eliminate && res != NULL) {
 668     for (DUIterator_Fast jmax, j = res->fast_outs(jmax);
 669                                j < jmax && can_eliminate; j++) {
 670       Node* use = res->fast_out(j);
 671 
 672       if (use->is_AddP()) {
 673         const TypePtr* addp_type = _igvn.type(use)->is_ptr();
 674         int offset = addp_type->offset();
 675 
 676         if (offset == Type::OffsetTop || offset == Type::OffsetBot) {
 677           NOT_PRODUCT(fail_eliminate = "Undefined field referrence";)
 678           can_eliminate = false;
 679           break;
 680         }
 681         for (DUIterator_Fast kmax, k = use->fast_outs(kmax);
 682                                    k < kmax && can_eliminate; k++) {
 683           Node* n = use->fast_out(k);
 684           if (!n->is_Store() && n->Opcode() != Op_CastP2X
 685               SHENANDOAHGC_ONLY(&& (!UseShenandoahGC || !ShenandoahBarrierSetC2::is_shenandoah_wb_pre_call(n))) ) {
 686             DEBUG_ONLY(disq_node = n;)
 687             if (n->is_Load() || n->is_LoadStore()) {
 688               NOT_PRODUCT(fail_eliminate = "Field load";)
 689             } else {
 690               NOT_PRODUCT(fail_eliminate = "Not store field reference";)
 691             }
 692             can_eliminate = false;
 693           }
 694         }
 695       } else if (use->is_ArrayCopy() &&
 696                  (use->as_ArrayCopy()->is_clonebasic() ||
 697                   use->as_ArrayCopy()->is_arraycopy_validated() ||
 698                   use->as_ArrayCopy()->is_copyof_validated() ||
 699                   use->as_ArrayCopy()->is_copyofrange_validated()) &&
 700                  use->in(ArrayCopyNode::Dest) == res) {
 701         // ok to eliminate
 702       } else if (use->is_SafePoint()) {
 703         SafePointNode* sfpt = use->as_SafePoint();
 704         if (sfpt->is_Call() && sfpt->as_Call()->has_non_debug_use(res)) {
 705           // Object is passed as argument.
 706           DEBUG_ONLY(disq_node = use;)
 707           NOT_PRODUCT(fail_eliminate = "Object is passed as argument";)
 708           can_eliminate = false;
 709         }
 710         Node* sfptMem = sfpt->memory();
 711         if (sfptMem == NULL || sfptMem->is_top()) {
 712           DEBUG_ONLY(disq_node = use;)
 713           NOT_PRODUCT(fail_eliminate = "NULL or TOP memory";)
 714           can_eliminate = false;
 715         } else {
 716           safepoints.append_if_missing(sfpt);
 717         }
 718       } else if (use->is_ValueType() && use->isa_ValueType()->get_oop() == res) {
 719         // ok to eliminate
 720       } else if (use->Opcode() == Op_StoreX && use->in(MemNode::Address) == res) {
 721         // store to mark work
 722       } else if (use->Opcode() != Op_CastP2X) { // CastP2X is used by card mark
 723         if (use->is_Phi()) {
 724           if (use->outcnt() == 1 && use->unique_out()->Opcode() == Op_Return) {
 725             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 726           } else {
 727             NOT_PRODUCT(fail_eliminate = "Object is referenced by Phi";)
 728           }
 729           DEBUG_ONLY(disq_node = use;)
 730         } else {
 731           if (use->Opcode() == Op_Return) {
 732             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 733           } else {
 734             NOT_PRODUCT(fail_eliminate = "Object is referenced by node";)
 735           }
 736           DEBUG_ONLY(disq_node = use;)
 737         }
 738         can_eliminate = false;
 739       } else {
 740         assert(use->Opcode() == Op_CastP2X, "should be");
 741         assert(!use->has_out_with(Op_OrL), "should have been removed because oop is never null");
 742       }
 743     }
 744   }
 745 
 746 #ifndef PRODUCT
 747   if (PrintEliminateAllocations) {
 748     if (can_eliminate) {
 749       tty->print("Scalar ");
 750       if (res == NULL)
 751         alloc->dump();
 752       else
 753         res->dump();
 754     } else if (alloc->_is_scalar_replaceable) {
 755       tty->print("NotScalar (%s)", fail_eliminate);
 756       if (res == NULL)
 757         alloc->dump();
 758       else
 759         res->dump();
 760 #ifdef ASSERT
 761       if (disq_node != NULL) {
 762           tty->print("  >>>> ");
 763           disq_node->dump();
 764       }
 765 #endif /*ASSERT*/
 766     }
 767   }
 768 #endif
 769   return can_eliminate;
 770 }

 772 // Do scalar replacement.
 773 bool PhaseMacroExpand::scalar_replacement(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 774   GrowableArray <SafePointNode *> safepoints_done;
 775 
 776   ciKlass* klass = NULL;
 777   ciInstanceKlass* iklass = NULL;
 778   int nfields = 0;
 779   int array_base = 0;
 780   int element_size = 0;
 781   BasicType basic_elem_type = T_ILLEGAL;
 782   ciType* elem_type = NULL;
 783 
 784   Node* res = alloc->result_cast();
 785   assert(res == NULL || res->is_CheckCastPP(), "unexpected AllocateNode result");
 786   const TypeOopPtr* res_type = NULL;
 787   if (res != NULL) { // Could be NULL when there are no users
 788     res_type = _igvn.type(res)->isa_oopptr();
 789   }
 790 
 791   if (res != NULL) {
 792     klass = res_type->klass();
 793     if (res_type->isa_instptr()) {
 794       // find the fields of the class which will be needed for safepoint debug information
 795       assert(klass->is_instance_klass(), "must be an instance klass.");
 796       iklass = klass->as_instance_klass();
 797       nfields = iklass->nof_nonstatic_fields();
 798     } else {
 799       // find the array's elements which will be needed for safepoint debug information
 800       nfields = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 801       assert(klass->is_array_klass() && nfields >= 0, "must be an array klass.");
 802       elem_type = klass->as_array_klass()->element_type();
 803       basic_elem_type = elem_type->basic_type();
 804       if (elem_type->is_valuetype() && !klass->is_value_array_klass()) {
 805         assert(basic_elem_type == T_VALUETYPE, "unexpected element basic type");
 806         basic_elem_type = T_OBJECT;
 807       }
 808       array_base = arrayOopDesc::base_offset_in_bytes(basic_elem_type);
 809       element_size = type2aelembytes(basic_elem_type);
 810       if (klass->is_value_array_klass()) {
 811         // Flattened value type array
 812         element_size = klass->as_value_array_klass()->element_byte_size();
 813       }
 814     }
 815   }
 816   //
 817   // Process the safepoint uses
 818   //
 819   Unique_Node_List value_worklist;
 820   while (safepoints.length() > 0) {
 821     SafePointNode* sfpt = safepoints.pop();
 822     Node* mem = sfpt->memory();
 823     Node* ctl = sfpt->control();
 824     assert(sfpt->jvms() != NULL, "missed JVMS");
 825     // Fields of scalar objs are referenced only at the end
 826     // of regular debuginfo at the last (youngest) JVMS.
 827     // Record relative start index.
 828     uint first_ind = (sfpt->req() - sfpt->jvms()->scloff());
 829     SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(res_type,
 830 #ifdef ASSERT
 831                                                  alloc,
 832 #endif
 833                                                  first_ind, nfields);
 834     sobj->init_req(0, C->root());
 835     transform_later(sobj);
 836 
 837     // Scan object's fields adding an input to the safepoint for each field.
 838     for (int j = 0; j < nfields; j++) {
 839       intptr_t offset;
 840       ciField* field = NULL;
 841       if (iklass != NULL) {
 842         field = iklass->nonstatic_field_at(j);
 843         offset = field->offset();
 844         elem_type = field->type();
 845         basic_elem_type = field->layout_type();
 846         assert(!field->is_flattened(), "flattened value type fields should not have safepoint uses");
 847       } else {
 848         offset = array_base + j * (intptr_t)element_size;
 849       }
 850 
 851       const Type *field_type;
 852       // The next code is taken from Parse::do_get_xxx().
 853       if (is_reference_type(basic_elem_type)) {
 854         if (!elem_type->is_loaded()) {
 855           field_type = TypeInstPtr::BOTTOM;
 856         } else if (field != NULL && field->is_static_constant()) {
 857           // This can happen if the constant oop is non-perm.
 858           ciObject* con = field->constant_value().as_object();
 859           // Do not "join" in the previous type; it doesn't add value,
 860           // and may yield a vacuous result if the field is of interface type.
 861           field_type = TypeOopPtr::make_from_constant(con)->isa_oopptr();
 862           assert(field_type != NULL, "field singleton type must be consistent");
 863         } else {
 864           field_type = TypeOopPtr::make_from_klass(elem_type->as_klass());
 865         }
 866         if (UseCompressedOops) {
 867           field_type = field_type->make_narrowoop();
 868           basic_elem_type = T_NARROWOOP;
 869         }
 870       } else {
 871         field_type = Type::get_const_basic_type(basic_elem_type);
 872       }
 873 
 874       Node* field_val = NULL;
 875       const TypeOopPtr* field_addr_type = res_type->add_offset(offset)->isa_oopptr();
 876       if (klass->is_value_array_klass()) {
 877         ciValueKlass* vk = elem_type->as_value_klass();
 878         assert(vk->flatten_array(), "must be flattened");
 879         field_val = value_type_from_mem(mem, ctl, vk, field_addr_type->isa_aryptr(), 0, alloc);
 880       } else {
 881         field_val = value_from_mem(mem, ctl, basic_elem_type, field_type, field_addr_type, alloc);
 882       }
 883       if (field_val == NULL) {
 884         // We weren't able to find a value for this field,
 885         // give up on eliminating this allocation.
 886 
 887         // Remove any extra entries we added to the safepoint.
 888         uint last = sfpt->req() - 1;
 889         for (int k = 0;  k < j; k++) {
 890           sfpt->del_req(last--);
 891         }
 892         _igvn._worklist.push(sfpt);
 893         // rollback processed safepoints
 894         while (safepoints_done.length() > 0) {
 895           SafePointNode* sfpt_done = safepoints_done.pop();
 896           // remove any extra entries we added to the safepoint
 897           last = sfpt_done->req() - 1;
 898           for (int k = 0;  k < nfields; k++) {
 899             sfpt_done->del_req(last--);
 900           }
 901           JVMState *jvms = sfpt_done->jvms();
 902           jvms->set_endoff(sfpt_done->req());
 903           // Now make a pass over the debug information replacing any references
 904           // to SafePointScalarObjectNode with the allocated object.
 905           int start = jvms->debug_start();
 906           int end   = jvms->debug_end();
 907           for (int i = start; i < end; i++) {
 908             if (sfpt_done->in(i)->is_SafePointScalarObject()) {
 909               SafePointScalarObjectNode* scobj = sfpt_done->in(i)->as_SafePointScalarObject();
 910               if (scobj->first_index(jvms) == sfpt_done->req() &&
 911                   scobj->n_fields() == (uint)nfields) {
 912                 assert(scobj->alloc() == alloc, "sanity");
 913                 sfpt_done->set_req(i, res);
 914               }
 915             }
 916           }
 917           _igvn._worklist.push(sfpt_done);
 918         }
 919 #ifndef PRODUCT
 920         if (PrintEliminateAllocations) {
 921           if (field != NULL) {
 922             tty->print("=== At SafePoint node %d can't find value of Field: ",
 923                        sfpt->_idx);
 924             field->print();
 925             int field_idx = C->get_alias_index(field_addr_type);
 926             tty->print(" (alias_idx=%d)", field_idx);
 927           } else { // Array's element
 928             tty->print("=== At SafePoint node %d can't find value of array element [%d]",
 929                        sfpt->_idx, j);
 930           }
 931           tty->print(", which prevents elimination of: ");
 932           if (res == NULL)
 933             alloc->dump();
 934           else
 935             res->dump();
 936         }
 937 #endif
 938         return false;
 939       }
 940       if (field_val->is_ValueType()) {
 941         // Keep track of value types to scalarize them later
 942         value_worklist.push(field_val);
 943       } else if (UseCompressedOops && field_type->isa_narrowoop()) {
 944         // Enable "DecodeN(EncodeP(Allocate)) --> Allocate" transformation
 945         // to be able scalar replace the allocation.
 946         if (field_val->is_EncodeP()) {
 947           field_val = field_val->in(1);
 948         } else {
 949           field_val = transform_later(new DecodeNNode(field_val, field_val->get_ptr_type()));
 950         }
 951       }
 952       sfpt->add_req(field_val);
 953     }
 954     JVMState *jvms = sfpt->jvms();
 955     jvms->set_endoff(sfpt->req());
 956     // Now make a pass over the debug information replacing any references
 957     // to the allocated object with "sobj"
 958     int start = jvms->debug_start();
 959     int end   = jvms->debug_end();
 960     sfpt->replace_edges_in_range(res, sobj, start, end);
 961     _igvn._worklist.push(sfpt);
 962     safepoints_done.append_if_missing(sfpt); // keep it for rollback
 963   }
 964   // Scalarize value types that were added to the safepoint
 965   for (uint i = 0; i < value_worklist.size(); ++i) {
 966     Node* vt = value_worklist.at(i);
 967     vt->as_ValueType()->make_scalar_in_safepoints(&_igvn);
 968   }
 969   return true;
 970 }

 972 static void disconnect_projections(MultiNode* n, PhaseIterGVN& igvn) {
 973   Node* ctl_proj = n->proj_out_or_null(TypeFunc::Control);
 974   Node* mem_proj = n->proj_out_or_null(TypeFunc::Memory);
 975   if (ctl_proj != NULL) {
 976     igvn.replace_node(ctl_proj, n->in(0));
 977   }
 978   if (mem_proj != NULL) {
 979     igvn.replace_node(mem_proj, n->in(TypeFunc::Memory));
 980   }

 983 // Process users of eliminated allocation.
 984 void PhaseMacroExpand::process_users_of_allocation(CallNode *alloc) {
 985   Node* res = alloc->result_cast();
 986   if (res != NULL) {
 987     for (DUIterator_Last jmin, j = res->last_outs(jmin); j >= jmin; ) {
 988       Node *use = res->last_out(j);
 989       uint oc1 = res->outcnt();
 990 
 991       if (use->is_AddP()) {
 992         for (DUIterator_Last kmin, k = use->last_outs(kmin); k >= kmin; ) {
 993           Node *n = use->last_out(k);
 994           uint oc2 = use->outcnt();
 995           if (n->is_Store()) {
 996 #ifdef ASSERT
 997             // Verify that there is no dependent MemBarVolatile nodes,
 998             // they should be removed during IGVN, see MemBarNode::Ideal().
 999             for (DUIterator_Fast pmax, p = n->fast_outs(pmax);
1000                                        p < pmax; p++) {
1001               Node* mb = n->fast_out(p);
1002               assert(mb->is_Initialize() || !mb->is_MemBar() ||
1003                      mb->req() <= MemBarNode::Precedent ||
1004                      mb->in(MemBarNode::Precedent) != n,
1005                      "MemBarVolatile should be eliminated for non-escaping object");
1006             }
1007 #endif
1008             _igvn.replace_node(n, n->in(MemNode::Memory));
1009           } else {
1010             eliminate_gc_barrier(n);
1011           }
1012           k -= (oc2 - use->outcnt());
1013         }
1014         _igvn.remove_dead_node(use);
1015       } else if (use->is_ArrayCopy()) {
1016         // Disconnect ArrayCopy node
1017         ArrayCopyNode* ac = use->as_ArrayCopy();
1018         if (ac->is_clonebasic()) {
1019           Node* membar_after = ac->proj_out(TypeFunc::Control)->unique_ctrl_out();
1020           disconnect_projections(ac, _igvn);
1021           assert(alloc->in(0)->is_Proj() && alloc->in(0)->in(0)->Opcode() == Op_MemBarCPUOrder, "mem barrier expected before allocation");
1022           Node* membar_before = alloc->in(0)->in(0);
1023           disconnect_projections(membar_before->as_MemBar(), _igvn);
1024           if (membar_after->is_MemBar()) {
1025             disconnect_projections(membar_after->as_MemBar(), _igvn);
1026           }
1027         } else {
1028           assert(ac->is_arraycopy_validated() ||
1029                  ac->is_copyof_validated() ||
1030                  ac->is_copyofrange_validated(), "unsupported");
1031           CallProjections* callprojs = ac->extract_projections(true);
1032 
1033           _igvn.replace_node(callprojs->fallthrough_ioproj, ac->in(TypeFunc::I_O));
1034           _igvn.replace_node(callprojs->fallthrough_memproj, ac->in(TypeFunc::Memory));
1035           _igvn.replace_node(callprojs->fallthrough_catchproj, ac->in(TypeFunc::Control));
1036 
1037           // Set control to top. IGVN will remove the remaining projections
1038           ac->set_req(0, top());
1039           ac->replace_edge(res, top());
1040 
1041           // Disconnect src right away: it can help find new
1042           // opportunities for allocation elimination
1043           Node* src = ac->in(ArrayCopyNode::Src);
1044           ac->replace_edge(src, top());
1045           // src can be top at this point if src and dest of the
1046           // arraycopy were the same
1047           if (src->outcnt() == 0 && !src->is_top()) {
1048             _igvn.remove_dead_node(src);
1049           }
1050         }
1051         _igvn._worklist.push(ac);
1052       } else if (use->is_ValueType()) {
1053         assert(use->isa_ValueType()->get_oop() == res, "unexpected value type use");
1054          _igvn.rehash_node_delayed(use);
1055         use->isa_ValueType()->set_oop(_igvn.zerocon(T_VALUETYPE));
1056       } else if (use->is_Store()) {
1057         _igvn.replace_node(use, use->in(MemNode::Memory));
1058       } else {
1059         eliminate_gc_barrier(use);
1060       }
1061       j -= (oc1 - res->outcnt());
1062     }
1063     assert(res->outcnt() == 0, "all uses of allocated objects must be deleted");
1064     _igvn.remove_dead_node(res);
1065   }
1066 
1067   //
1068   // Process other users of allocation's projections
1069   //
1070   if (_resproj != NULL && _resproj->outcnt() != 0) {
1071     // First disconnect stores captured by Initialize node.
1072     // If Initialize node is eliminated first in the following code,
1073     // it will kill such stores and DUIterator_Last will assert.
1074     for (DUIterator_Fast jmax, j = _resproj->fast_outs(jmax);  j < jmax; j++) {
1075       Node *use = _resproj->fast_out(j);
1076       if (use->is_AddP()) {
1077         // raw memory addresses used only by the initialization
1078         _igvn.replace_node(use, C->top());
1079         --j; --jmax;
1080       }
1081     }
1082     for (DUIterator_Last jmin, j = _resproj->last_outs(jmin); j >= jmin; ) {
1083       Node *use = _resproj->last_out(j);
1084       uint oc1 = _resproj->outcnt();
1085       if (use->is_Initialize()) {
1086         // Eliminate Initialize node.
1087         InitializeNode *init = use->as_Initialize();
1088         assert(init->outcnt() <= 2, "only a control and memory projection expected");
1089         Node *ctrl_proj = init->proj_out_or_null(TypeFunc::Control);
1090         if (ctrl_proj != NULL) {
1091           _igvn.replace_node(ctrl_proj, init->in(TypeFunc::Control));
1092 #ifdef ASSERT
1093           Node* tmp = init->in(TypeFunc::Control);
1094           assert(tmp == _fallthroughcatchproj, "allocation control projection");
1095 #endif
1096         }
1097         Node *mem_proj = init->proj_out_or_null(TypeFunc::Memory);
1098         if (mem_proj != NULL) {
1099           Node *mem = init->in(TypeFunc::Memory);
1100 #ifdef ASSERT
1101           if (mem->is_MergeMem()) {
1102             assert(mem->in(TypeFunc::Memory) == _memproj_fallthrough, "allocation memory projection");
1103           } else {
1104             assert(mem == _memproj_fallthrough, "allocation memory projection");
1105           }
1106 #endif
1107           _igvn.replace_node(mem_proj, mem);
1108         }
1109       } else  {
1110         assert(false, "only Initialize or AddP expected");
1111       }
1112       j -= (oc1 - _resproj->outcnt());
1113     }
1114   }
1115   if (_fallthroughcatchproj != NULL) {
1116     _igvn.replace_node(_fallthroughcatchproj, alloc->in(TypeFunc::Control));
1117   }
1118   if (_memproj_fallthrough != NULL) {
1119     _igvn.replace_node(_memproj_fallthrough, alloc->in(TypeFunc::Memory));
1120   }
1121   if (_memproj_catchall != NULL) {
1122     _igvn.replace_node(_memproj_catchall, C->top());
1123   }
1124   if (_ioproj_fallthrough != NULL) {
1125     _igvn.replace_node(_ioproj_fallthrough, alloc->in(TypeFunc::I_O));
1126   }
1127   if (_ioproj_catchall != NULL) {
1128     _igvn.replace_node(_ioproj_catchall, C->top());
1129   }
1130   if (_catchallcatchproj != NULL) {
1131     _igvn.replace_node(_catchallcatchproj, C->top());
1132   }
1133 }

1135 bool PhaseMacroExpand::eliminate_allocate_node(AllocateNode *alloc) {
1136   // Don't do scalar replacement if the frame can be popped by JVMTI:
1137   // if reallocation fails during deoptimization we'll pop all
1138   // interpreter frames for this compiled frame and that won't play
1139   // nice with JVMTI popframe.
1140   if (!EliminateAllocations || JvmtiExport::can_pop_frame() || !alloc->_is_non_escaping) {
1141     return false;
1142   }
1143   Node* klass = alloc->in(AllocateNode::KlassNode);
1144   const TypeKlassPtr* tklass = _igvn.type(klass)->is_klassptr();
1145   Node* res = alloc->result_cast();
1146   // Eliminate boxing allocations which are not used
1147   // regardless scalar replacable status.
1148   bool boxing_alloc = C->eliminate_boxing() &&
1149                       tklass->klass()->is_instance_klass()  &&
1150                       tklass->klass()->as_instance_klass()->is_box_klass();
1151   if (!alloc->_is_scalar_replaceable && (!boxing_alloc || (res != NULL))) {
1152     return false;
1153   }
1154 
1155   extract_call_projections(alloc);
1156 
1157   GrowableArray <SafePointNode *> safepoints;
1158   if (!can_eliminate_allocation(alloc, safepoints)) {
1159     return false;
1160   }
1161 
1162   if (!alloc->_is_scalar_replaceable) {
1163     assert(res == NULL, "sanity");
1164     // We can only eliminate allocation if all debug info references
1165     // are already replaced with SafePointScalarObject because
1166     // we can't search for a fields value without instance_id.
1167     if (safepoints.length() > 0) {
1168       return false;
1169     }
1170   }
1171 
1172   if (!scalar_replacement(alloc, safepoints)) {
1173     return false;
1174   }
1175 
1176   CompileLog* log = C->log();
1177   if (log != NULL) {
1178     log->head("eliminate_allocation type='%d'",
1179               log->identify(tklass->klass()));
1180     JVMState* p = alloc->jvms();
1181     while (p != NULL) {
1182       log->elem("jvms bci='%d' method='%d'", p->bci(), log->identify(p->method()));
1183       p = p->caller();
1184     }
1185     log->tail("eliminate_allocation");
1186   }
1187 
1188   process_users_of_allocation(alloc);
1189 
1190 #ifndef PRODUCT
1191   if (PrintEliminateAllocations) {
1192     if (alloc->is_AllocateArray())
1193       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1194     else
1195       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1196   }
1197 #endif
1198 
1199   return true;
1200 }

1276 //=============================================================================
1277 //
1278 //                              A L L O C A T I O N
1279 //
1280 // Allocation attempts to be fast in the case of frequent small objects.
1281 // It breaks down like this:
1282 //
1283 // 1) Size in doublewords is computed.  This is a constant for objects and
1284 // variable for most arrays.  Doubleword units are used to avoid size
1285 // overflow of huge doubleword arrays.  We need doublewords in the end for
1286 // rounding.
1287 //
1288 // 2) Size is checked for being 'too large'.  Too-large allocations will go
1289 // the slow path into the VM.  The slow path can throw any required
1290 // exceptions, and does all the special checks for very large arrays.  The
1291 // size test can constant-fold away for objects.  For objects with
1292 // finalizers it constant-folds the otherway: you always go slow with
1293 // finalizers.
1294 //
1295 // 3) If NOT using TLABs, this is the contended loop-back point.
1296 // Load-Locked the heap top.  If using TLABs normal-load the heap top.
1297 //
1298 // 4) Check that heap top + size*8 < max.  If we fail go the slow ` route.
1299 // NOTE: "top+size*8" cannot wrap the 4Gig line!  Here's why: for largish
1300 // "size*8" we always enter the VM, where "largish" is a constant picked small
1301 // enough that there's always space between the eden max and 4Gig (old space is
1302 // there so it's quite large) and large enough that the cost of entering the VM
1303 // is dwarfed by the cost to initialize the space.
1304 //
1305 // 5) If NOT using TLABs, Store-Conditional the adjusted heap top back
1306 // down.  If contended, repeat at step 3.  If using TLABs normal-store
1307 // adjusted heap top back down; there is no contention.
1308 //
1309 // 6) If !ZeroTLAB then Bulk-clear the object/array.  Fill in klass & mark
1310 // fields.
1311 //
1312 // 7) Merge with the slow-path; cast the raw memory pointer to the correct
1313 // oop flavor.
1314 //
1315 //=============================================================================
1316 // FastAllocateSizeLimit value is in DOUBLEWORDS.
1317 // Allocations bigger than this always go the slow route.
1318 // This value must be small enough that allocation attempts that need to
1319 // trigger exceptions go the slow route.  Also, it must be small enough so
1320 // that heap_top + size_in_bytes does not wrap around the 4Gig limit.
1321 //=============================================================================j//
1322 // %%% Here is an old comment from parseHelper.cpp; is it outdated?
1323 // The allocator will coalesce int->oop copies away.  See comment in
1324 // coalesce.cpp about how this works.  It depends critically on the exact
1325 // code shape produced here, so if you are changing this code shape
1326 // make sure the GC info for the heap-top is correct in and around the
1327 // slow-path call.
1328 //
1329 
1330 void PhaseMacroExpand::expand_allocate_common(
1331             AllocateNode* alloc, // allocation node to be expanded
1332             Node* length,  // array length for an array allocation
1333             const TypeFunc* slow_call_type, // Type of slow call
1334             address slow_call_address  // Address of slow call
1335     )
1336 {
1337   Node* ctrl = alloc->in(TypeFunc::Control);
1338   Node* mem  = alloc->in(TypeFunc::Memory);
1339   Node* i_o  = alloc->in(TypeFunc::I_O);
1340   Node* size_in_bytes     = alloc->in(AllocateNode::AllocSize);
1341   Node* klass_node        = alloc->in(AllocateNode::KlassNode);
1342   Node* initial_slow_test = alloc->in(AllocateNode::InitialTest);
1343   assert(ctrl != NULL, "must have control");
1344 
1345   // We need a Region and corresponding Phi's to merge the slow-path and fast-path results.
1346   // they will not be used if "always_slow" is set
1347   enum { slow_result_path = 1, fast_result_path = 2 };
1348   Node *result_region = NULL;
1349   Node *result_phi_rawmem = NULL;
1350   Node *result_phi_rawoop = NULL;
1351   Node *result_phi_i_o = NULL;
1352 
1353   // The initial slow comparison is a size check, the comparison
1354   // we want to do is a BoolTest::gt
1355   bool expand_fast_path = true;
1356   int tv = _igvn.find_int_con(initial_slow_test, -1);
1357   if (tv >= 0) {
1358     // InitialTest has constant result
1359     //   0 - can fit in TLAB
1360     //   1 - always too big or negative
1361     assert(tv <= 1, "0 or 1 if a constant");
1362     expand_fast_path = (tv == 0);
1363     initial_slow_test = NULL;
1364   } else {
1365     initial_slow_test = BoolNode::make_predicate(initial_slow_test, &_igvn);
1366   }
1367 
1368   if (C->env()->dtrace_alloc_probes() ||
1369       (!UseTLAB && !Universe::heap()->supports_inline_contig_alloc())) {
1370     // Force slow-path allocation
1371     expand_fast_path = false;
1372     initial_slow_test = NULL;
1373   }
1374 
1375   bool allocation_has_use = (alloc->result_cast() != NULL);
1376   if (!allocation_has_use) {
1377     InitializeNode* init = alloc->initialization();
1378     if (init != NULL) {
1379       init->remove(&_igvn);
1380     }
1381     if (expand_fast_path && (initial_slow_test == NULL)) {
1382       // Remove allocation node and return.
1383       // Size is a non-negative constant -> no initial check needed -> directly to fast path.
1384       // Also, no usages -> empty fast path -> no fall out to slow path -> nothing left.
1385 #ifndef PRODUCT
1386       if (PrintEliminateAllocations) {
1387         tty->print("NotUsed ");
1388         Node* res = alloc->proj_out_or_null(TypeFunc::Parms);
1389         if (res != NULL) {
1390           res->dump();
1391         } else {
1392           alloc->dump();
1393         }
1394       }
1395 #endif
1396       yank_alloc_node(alloc);
1397       return;
1398     }
1399   }
1400 
1401   enum { too_big_or_final_path = 1, need_gc_path = 2 };
1402   Node *slow_region = NULL;
1403   Node *toobig_false = ctrl;
1404 
1405   // generate the initial test if necessary
1406   if (initial_slow_test != NULL ) {
1407     assert (expand_fast_path, "Only need test if there is a fast path");
1408     slow_region = new RegionNode(3);
1409 
1410     // Now make the initial failure test.  Usually a too-big test but
1411     // might be a TRUE for finalizers or a fancy class check for
1412     // newInstance0.
1413     IfNode* toobig_iff = new IfNode(ctrl, initial_slow_test, PROB_MIN, COUNT_UNKNOWN);
1414     transform_later(toobig_iff);
1415     // Plug the failing-too-big test into the slow-path region
1416     Node* toobig_true = new IfTrueNode(toobig_iff);
1417     transform_later(toobig_true);
1418     slow_region    ->init_req( too_big_or_final_path, toobig_true );
1419     toobig_false = new IfFalseNode(toobig_iff);
1420     transform_later(toobig_false);
1421   } else {
1422     // No initial test, just fall into next case
1423     assert(allocation_has_use || !expand_fast_path, "Should already have been handled");
1424     toobig_false = ctrl;
1425     debug_only(slow_region = NodeSentinel);
1426   }
1427 
1428   // If we are here there are several possibilities
1429   // - expand_fast_path is false - then only a slow path is expanded. That's it.
1430   // no_initial_check means a constant allocation.
1431   // - If check always evaluates to false -> expand_fast_path is false (see above)
1432   // - If check always evaluates to true -> directly into fast path (but may bailout to slowpath)
1433   // if !allocation_has_use the fast path is empty
1434   // if !allocation_has_use && no_initial_check
1435   // - Then there are no fastpath that can fall out to slowpath -> no allocation code at all.
1436   //   removed by yank_alloc_node above.
1437 
1438   Node *slow_mem = mem;  // save the current memory state for slow path
1439   // generate the fast allocation code unless we know that the initial test will always go slow
1440   if (expand_fast_path) {
1441     // Fast path modifies only raw memory.
1442     if (mem->is_MergeMem()) {
1443       mem = mem->as_MergeMem()->memory_at(Compile::AliasIdxRaw);
1444     }
1445 
1446     // allocate the Region and Phi nodes for the result
1447     result_region = new RegionNode(3);
1448     result_phi_rawmem = new PhiNode(result_region, Type::MEMORY, TypeRawPtr::BOTTOM);
1449     result_phi_i_o    = new PhiNode(result_region, Type::ABIO); // I/O is used for Prefetch
1450 
1451     // Grab regular I/O before optional prefetch may change it.
1452     // Slow-path does no I/O so just set it to the original I/O.
1453     result_phi_i_o->init_req(slow_result_path, i_o);
1454 
1455     // Name successful fast-path variables
1456     Node* fast_oop_ctrl;
1457     Node* fast_oop_rawmem;
1458 
1459     if (allocation_has_use) {
1460       Node* needgc_ctrl = NULL;
1461       result_phi_rawoop = new PhiNode(result_region, TypeRawPtr::BOTTOM);
1462 
1463       intx prefetch_lines = length != NULL ? AllocatePrefetchLines : AllocateInstancePrefetchLines;
1464       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1465       Node* fast_oop = bs->obj_allocate(this, ctrl, mem, toobig_false, size_in_bytes, i_o, needgc_ctrl,
1466                                         fast_oop_ctrl, fast_oop_rawmem,
1467                                         prefetch_lines);
1468 
1469       if (initial_slow_test != NULL) {
1470         // This completes all paths into the slow merge point
1471         slow_region->init_req(need_gc_path, needgc_ctrl);
1472         transform_later(slow_region);
1473       } else {
1474         // No initial slow path needed!
1475         // Just fall from the need-GC path straight into the VM call.
1476         slow_region = needgc_ctrl;
1477       }
1478 
1479       InitializeNode* init = alloc->initialization();
1480       fast_oop_rawmem = initialize_object(alloc,
1481                                           fast_oop_ctrl, fast_oop_rawmem, fast_oop,
1482                                           klass_node, length, size_in_bytes);
1483       expand_initialize_membar(alloc, init, fast_oop_ctrl, fast_oop_rawmem);
1484       expand_dtrace_alloc_probe(alloc, fast_oop, fast_oop_ctrl, fast_oop_rawmem);
1485 
1486       result_phi_rawoop->init_req(fast_result_path, fast_oop);
1487     } else {
1488       assert (initial_slow_test != NULL, "sanity");
1489       fast_oop_ctrl   = toobig_false;
1490       fast_oop_rawmem = mem;
1491       transform_later(slow_region);
1492     }
1493 
1494     // Plug in the successful fast-path into the result merge point
1495     result_region    ->init_req(fast_result_path, fast_oop_ctrl);
1496     result_phi_i_o   ->init_req(fast_result_path, i_o);
1497     result_phi_rawmem->init_req(fast_result_path, fast_oop_rawmem);
1498   } else {
1499     slow_region = ctrl;
1500     result_phi_i_o = i_o; // Rename it to use in the following code.
1501   }
1502 
1503   // Generate slow-path call
1504   CallNode *call = new CallStaticJavaNode(slow_call_type, slow_call_address,
1505                                OptoRuntime::stub_name(slow_call_address),
1506                                alloc->jvms()->bci(),
1507                                TypePtr::BOTTOM);
1508   call->init_req(TypeFunc::Control,   slow_region);
1509   call->init_req(TypeFunc::I_O,       top());    // does no i/o
1510   call->init_req(TypeFunc::Memory,    slow_mem); // may gc ptrs
1511   call->init_req(TypeFunc::ReturnAdr, alloc->in(TypeFunc::ReturnAdr));
1512   call->init_req(TypeFunc::FramePtr,  alloc->in(TypeFunc::FramePtr));
1513 
1514   call->init_req(TypeFunc::Parms+0, klass_node);
1515   if (length != NULL) {
1516     call->init_req(TypeFunc::Parms+1, length);
1517   } else {
1518     // Let the runtime know if this is a larval allocation
1519     call->init_req(TypeFunc::Parms+1, _igvn.intcon(alloc->_larval));
1520   }
1521 
1522   // Copy debug information and adjust JVMState information, then replace
1523   // allocate node with the call
1524   call->copy_call_debug_info(&_igvn, alloc);
1525   if (expand_fast_path) {
1526     call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
1527   } else {
1528     // Hook i_o projection to avoid its elimination during allocation
1529     // replacement (when only a slow call is generated).
1530     call->set_req(TypeFunc::I_O, result_phi_i_o);
1531   }
1532   _igvn.replace_node(alloc, call);
1533   transform_later(call);
1534 
1535   // Identify the output projections from the allocate node and
1536   // adjust any references to them.
1537   // The control and io projections look like:
1538   //
1539   //        v---Proj(ctrl) <-----+   v---CatchProj(ctrl)
1540   //  Allocate                   Catch
1541   //        ^---Proj(io) <-------+   ^---CatchProj(io)
1542   //
1543   //  We are interested in the CatchProj nodes.
1544   //
1545   extract_call_projections(call);
1546 
1547   // An allocate node has separate memory projections for the uses on
1548   // the control and i_o paths. Replace the control memory projection with
1549   // result_phi_rawmem (unless we are only generating a slow call when
1550   // both memory projections are combined)
1551   if (expand_fast_path && _memproj_fallthrough != NULL) {
1552     _igvn.replace_in_uses(_memproj_fallthrough, result_phi_rawmem);
1553   }
1554   // Now change uses of _memproj_catchall to use _memproj_fallthrough and delete
1555   // _memproj_catchall so we end up with a call that has only 1 memory projection.
1556   if (_memproj_catchall != NULL) {
1557     if (_memproj_fallthrough == NULL) {
1558       _memproj_fallthrough = new ProjNode(call, TypeFunc::Memory);
1559       transform_later(_memproj_fallthrough);
1560     }
1561     _igvn.replace_in_uses(_memproj_catchall, _memproj_fallthrough);
1562     _igvn.remove_dead_node(_memproj_catchall);
1563   }
1564 
1565   // An allocate node has separate i_o projections for the uses on the control
1566   // and i_o paths. Always replace the control i_o projection with result i_o
1567   // otherwise incoming i_o become dead when only a slow call is generated
1568   // (it is different from memory projections where both projections are
1569   // combined in such case).
1570   if (_ioproj_fallthrough != NULL) {
1571     _igvn.replace_in_uses(_ioproj_fallthrough, result_phi_i_o);
1572   }
1573   // Now change uses of _ioproj_catchall to use _ioproj_fallthrough and delete
1574   // _ioproj_catchall so we end up with a call that has only 1 i_o projection.
1575   if (_ioproj_catchall != NULL) {
1576     if (_ioproj_fallthrough == NULL) {
1577       _ioproj_fallthrough = new ProjNode(call, TypeFunc::I_O);
1578       transform_later(_ioproj_fallthrough);
1579     }
1580     _igvn.replace_in_uses(_ioproj_catchall, _ioproj_fallthrough);
1581     _igvn.remove_dead_node(_ioproj_catchall);
1582   }
1583 
1584   // if we generated only a slow call, we are done
1585   if (!expand_fast_path) {
1586     // Now we can unhook i_o.
1587     if (result_phi_i_o->outcnt() > 1) {
1588       call->set_req(TypeFunc::I_O, top());
1589     } else {
1590       assert(result_phi_i_o->unique_ctrl_out() == call, "sanity");
1591       // Case of new array with negative size known during compilation.
1592       // AllocateArrayNode::Ideal() optimization disconnect unreachable
1593       // following code since call to runtime will throw exception.
1594       // As result there will be no users of i_o after the call.
1595       // Leave i_o attached to this call to avoid problems in preceding graph.
1596     }
1597     return;
1598   }
1599 
1600   if (_fallthroughcatchproj != NULL) {
1601     ctrl = _fallthroughcatchproj->clone();
1602     transform_later(ctrl);
1603     _igvn.replace_node(_fallthroughcatchproj, result_region);
1604   } else {
1605     ctrl = top();
1606   }
1607   Node *slow_result;
1608   if (_resproj == NULL) {
1609     // no uses of the allocation result
1610     slow_result = top();
1611   } else {
1612     slow_result = _resproj->clone();
1613     transform_later(slow_result);
1614     _igvn.replace_node(_resproj, result_phi_rawoop);
1615   }
1616 
1617   // Plug slow-path into result merge point
1618   result_region->init_req( slow_result_path, ctrl);
1619   transform_later(result_region);
1620   if (allocation_has_use) {
1621     result_phi_rawoop->init_req(slow_result_path, slow_result);
1622     transform_later(result_phi_rawoop);
1623   }
1624   result_phi_rawmem->init_req(slow_result_path, _memproj_fallthrough);
1625   transform_later(result_phi_rawmem);
1626   transform_later(result_phi_i_o);
1627   // This completes all paths into the result merge point
1628 }

1630 // Remove alloc node that has no uses.
1631 void PhaseMacroExpand::yank_alloc_node(AllocateNode* alloc) {
1632   Node* ctrl = alloc->in(TypeFunc::Control);
1633   Node* mem  = alloc->in(TypeFunc::Memory);
1634   Node* i_o  = alloc->in(TypeFunc::I_O);
1635 
1636   extract_call_projections(alloc);
1637   if (_resproj != NULL) {
1638     for (DUIterator_Fast imax, i = _resproj->fast_outs(imax); i < imax; i++) {
1639       Node* use = _resproj->fast_out(i);
1640       use->isa_MemBar()->remove(&_igvn);
1641       --imax;
1642       --i; // back up iterator
1643     }
1644     assert(_resproj->outcnt() == 0, "all uses must be deleted");
1645     _igvn.remove_dead_node(_resproj);
1646   }
1647   if (_fallthroughcatchproj != NULL) {
1648     _igvn.replace_in_uses(_fallthroughcatchproj, ctrl);
1649     _igvn.remove_dead_node(_fallthroughcatchproj);
1650   }
1651   if (_catchallcatchproj != NULL) {
1652     _igvn.rehash_node_delayed(_catchallcatchproj);
1653     _catchallcatchproj->set_req(0, top());
1654   }
1655   if (_fallthroughproj != NULL) {
1656     Node* catchnode = _fallthroughproj->unique_ctrl_out();
1657     _igvn.remove_dead_node(catchnode);
1658     _igvn.remove_dead_node(_fallthroughproj);
1659   }
1660   if (_memproj_fallthrough != NULL) {
1661     _igvn.replace_in_uses(_memproj_fallthrough, mem);
1662     _igvn.remove_dead_node(_memproj_fallthrough);
1663   }
1664   if (_ioproj_fallthrough != NULL) {
1665     _igvn.replace_in_uses(_ioproj_fallthrough, i_o);
1666     _igvn.remove_dead_node(_ioproj_fallthrough);
1667   }
1668   if (_memproj_catchall != NULL) {
1669     _igvn.rehash_node_delayed(_memproj_catchall);
1670     _memproj_catchall->set_req(0, top());
1671   }
1672   if (_ioproj_catchall != NULL) {
1673     _igvn.rehash_node_delayed(_ioproj_catchall);
1674     _ioproj_catchall->set_req(0, top());
1675   }
1676 #ifndef PRODUCT
1677   if (PrintEliminateAllocations) {
1678     if (alloc->is_AllocateArray()) {
1679       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1680     } else {
1681       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1682     }
1683   }
1684 #endif
1685   _igvn.remove_dead_node(alloc);
1686 }

1688 void PhaseMacroExpand::expand_initialize_membar(AllocateNode* alloc, InitializeNode* init,
1689                                                 Node*& fast_oop_ctrl, Node*& fast_oop_rawmem) {
1690   // If initialization is performed by an array copy, any required
1691   // MemBarStoreStore was already added. If the object does not
1692   // escape no need for a MemBarStoreStore. If the object does not
1693   // escape in its initializer and memory barrier (MemBarStoreStore or
1694   // stronger) is already added at exit of initializer, also no need
1695   // for a MemBarStoreStore. Otherwise we need a MemBarStoreStore
1696   // so that stores that initialize this object can't be reordered
1697   // with a subsequent store that makes this object accessible by
1698   // other threads.
1699   // Other threads include java threads and JVM internal threads
1700   // (for example concurrent GC threads). Current concurrent GC
1701   // implementation: G1 will not scan newly created object,
1702   // so it's safe to skip storestore barrier when allocation does
1703   // not escape.
1704   if (!alloc->does_not_escape_thread() &&
1705     !alloc->is_allocation_MemBar_redundant() &&
1706     (init == NULL || !init->is_complete_with_arraycopy())) {
1707     if (init == NULL || init->req() < InitializeNode::RawStores) {
1708       // No InitializeNode or no stores captured by zeroing
1709       // elimination. Simply add the MemBarStoreStore after object
1710       // initialization.
1711       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1712       transform_later(mb);
1713 
1714       mb->init_req(TypeFunc::Memory, fast_oop_rawmem);
1715       mb->init_req(TypeFunc::Control, fast_oop_ctrl);
1716       fast_oop_ctrl = new ProjNode(mb, TypeFunc::Control);
1717       transform_later(fast_oop_ctrl);
1718       fast_oop_rawmem = new ProjNode(mb, TypeFunc::Memory);
1719       transform_later(fast_oop_rawmem);
1720     } else {
1721       // Add the MemBarStoreStore after the InitializeNode so that
1722       // all stores performing the initialization that were moved
1723       // before the InitializeNode happen before the storestore
1724       // barrier.
1725 
1726       Node* init_ctrl = init->proj_out_or_null(TypeFunc::Control);
1727       Node* init_mem = init->proj_out_or_null(TypeFunc::Memory);
1728 
1729       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1730       transform_later(mb);
1731 
1732       Node* ctrl = new ProjNode(init, TypeFunc::Control);
1733       transform_later(ctrl);
1734       Node* mem = new ProjNode(init, TypeFunc::Memory);
1735       transform_later(mem);
1736 
1737       // The MemBarStoreStore depends on control and memory coming
1738       // from the InitializeNode
1739       mb->init_req(TypeFunc::Memory, mem);
1740       mb->init_req(TypeFunc::Control, ctrl);
1741 
1742       ctrl = new ProjNode(mb, TypeFunc::Control);
1743       transform_later(ctrl);
1744       mem = new ProjNode(mb, TypeFunc::Memory);
1745       transform_later(mem);
1746 
1747       // All nodes that depended on the InitializeNode for control
1748       // and memory must now depend on the MemBarNode that itself
1749       // depends on the InitializeNode
1750       if (init_ctrl != NULL) {
1751         _igvn.replace_node(init_ctrl, ctrl);
1752       }
1753       if (init_mem != NULL) {
1754         _igvn.replace_node(init_mem, mem);
1755       }
1756     }
1757   }
1758 }

1789 // Helper for PhaseMacroExpand::expand_allocate_common.
1790 // Initializes the newly-allocated storage.
1791 Node* PhaseMacroExpand::initialize_object(AllocateNode* alloc,
1792                                           Node* control, Node* rawmem, Node* object,
1793                                           Node* klass_node, Node* length,
1794                                           Node* size_in_bytes) {
1795   InitializeNode* init = alloc->initialization();
1796   // Store the klass & mark bits
1797   Node* mark_node = alloc->make_ideal_mark(&_igvn, control, rawmem);
1798   if (!mark_node->is_Con()) {
1799     transform_later(mark_node);
1800   }
1801   rawmem = make_store(control, rawmem, object, oopDesc::mark_offset_in_bytes(), mark_node, TypeX_X->basic_type());
1802 
1803   rawmem = make_store(control, rawmem, object, oopDesc::klass_offset_in_bytes(), klass_node, T_METADATA);
1804   int header_size = alloc->minimum_header_size();  // conservatively small
1805 
1806   // Array length
1807   if (length != NULL) {         // Arrays need length field
1808     rawmem = make_store(control, rawmem, object, arrayOopDesc::length_offset_in_bytes(), length, T_INT);
1809     // conservatively small header size:
1810     header_size = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1811     ciKlass* k = _igvn.type(klass_node)->is_klassptr()->klass();
1812     if (k->is_array_klass())    // we know the exact header size in most cases:
1813       header_size = Klass::layout_helper_header_size(k->layout_helper());
1814   }
1815 
1816   // Clear the object body, if necessary.
1817   if (init == NULL) {
1818     // The init has somehow disappeared; be cautious and clear everything.
1819     //
1820     // This can happen if a node is allocated but an uncommon trap occurs
1821     // immediately.  In this case, the Initialize gets associated with the
1822     // trap, and may be placed in a different (outer) loop, if the Allocate
1823     // is in a loop.  If (this is rare) the inner loop gets unrolled, then
1824     // there can be two Allocates to one Initialize.  The answer in all these
1825     // edge cases is safety first.  It is always safe to clear immediately
1826     // within an Allocate, and then (maybe or maybe not) clear some more later.
1827     if (!(UseTLAB && ZeroTLAB)) {
1828       rawmem = ClearArrayNode::clear_memory(control, rawmem, object,
1829                                             alloc->in(AllocateNode::DefaultValue),
1830                                             alloc->in(AllocateNode::RawDefaultValue),
1831                                             header_size, size_in_bytes,
1832                                             &_igvn);
1833     }
1834   } else {
1835     if (!init->is_complete()) {
1836       // Try to win by zeroing only what the init does not store.
1837       // We can also try to do some peephole optimizations,
1838       // such as combining some adjacent subword stores.
1839       rawmem = init->complete_stores(control, rawmem, object,
1840                                      header_size, size_in_bytes, &_igvn);
1841     }
1842     // We have no more use for this link, since the AllocateNode goes away:
1843     init->set_req(InitializeNode::RawAddress, top());
1844     // (If we keep the link, it just confuses the register allocator,
1845     // who thinks he sees a real use of the address by the membar.)
1846   }
1847 
1848   return rawmem;
1849 }

1851 // Generate prefetch instructions for next allocations.
1852 Node* PhaseMacroExpand::prefetch_allocation(Node* i_o, Node*& needgc_false,
1853                                         Node*& contended_phi_rawmem,
1854                                         Node* old_eden_top, Node* new_eden_top,
1855                                         intx lines) {
1856    enum { fall_in_path = 1, pf_path = 2 };
1857    if( UseTLAB && AllocatePrefetchStyle == 2 ) {
1858       // Generate prefetch allocation with watermark check.
1859       // As an allocation hits the watermark, we will prefetch starting
1860       // at a "distance" away from watermark.
1861 
1862       Node *pf_region = new RegionNode(3);
1863       Node *pf_phi_rawmem = new PhiNode( pf_region, Type::MEMORY,
1864                                                 TypeRawPtr::BOTTOM );
1865       // I/O is used for Prefetch
1866       Node *pf_phi_abio = new PhiNode( pf_region, Type::ABIO );
1867 
1868       Node *thread = new ThreadLocalNode();
1869       transform_later(thread);
1870 
1871       Node *eden_pf_adr = new AddPNode( top()/*not oop*/, thread,
1872                    _igvn.MakeConX(in_bytes(JavaThread::tlab_pf_top_offset())) );
1873       transform_later(eden_pf_adr);
1874 
1875       Node *old_pf_wm = new LoadPNode(needgc_false,
1876                                    contended_phi_rawmem, eden_pf_adr,
1877                                    TypeRawPtr::BOTTOM, TypeRawPtr::BOTTOM,
1878                                    MemNode::unordered);
1879       transform_later(old_pf_wm);
1880 
1881       // check against new_eden_top
1882       Node *need_pf_cmp = new CmpPNode( new_eden_top, old_pf_wm );
1883       transform_later(need_pf_cmp);
1884       Node *need_pf_bol = new BoolNode( need_pf_cmp, BoolTest::ge );
1885       transform_later(need_pf_bol);
1886       IfNode *need_pf_iff = new IfNode( needgc_false, need_pf_bol,
1887                                        PROB_UNLIKELY_MAG(4), COUNT_UNKNOWN );
1888       transform_later(need_pf_iff);
1889 
1890       // true node, add prefetchdistance
1891       Node *need_pf_true = new IfTrueNode( need_pf_iff );
1892       transform_later(need_pf_true);
1893 
1894       Node *need_pf_false = new IfFalseNode( need_pf_iff );
1895       transform_later(need_pf_false);
1896 
1897       Node *new_pf_wmt = new AddPNode( top(), old_pf_wm,
1898                                     _igvn.MakeConX(AllocatePrefetchDistance) );
1899       transform_later(new_pf_wmt );
1900       new_pf_wmt->set_req(0, need_pf_true);
1901 
1902       Node *store_new_wmt = new StorePNode(need_pf_true,
1903                                        contended_phi_rawmem, eden_pf_adr,
1904                                        TypeRawPtr::BOTTOM, new_pf_wmt,
1905                                        MemNode::unordered);
1906       transform_later(store_new_wmt);
1907 
1908       // adding prefetches
1909       pf_phi_abio->init_req( fall_in_path, i_o );
1910 
1911       Node *prefetch_adr;
1912       Node *prefetch;
1913       uint step_size = AllocatePrefetchStepSize;
1914       uint distance = 0;
1915 
1916       for ( intx i = 0; i < lines; i++ ) {
1917         prefetch_adr = new AddPNode( old_pf_wm, new_pf_wmt,
1918                                             _igvn.MakeConX(distance) );
1919         transform_later(prefetch_adr);
1920         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
1921         transform_later(prefetch);
1922         distance += step_size;
1923         i_o = prefetch;
1924       }
1925       pf_phi_abio->set_req( pf_path, i_o );
1926 
1927       pf_region->init_req( fall_in_path, need_pf_false );
1928       pf_region->init_req( pf_path, need_pf_true );
1929 
1930       pf_phi_rawmem->init_req( fall_in_path, contended_phi_rawmem );
1931       pf_phi_rawmem->init_req( pf_path, store_new_wmt );
1932 
1933       transform_later(pf_region);
1934       transform_later(pf_phi_rawmem);
1935       transform_later(pf_phi_abio);
1936 
1937       needgc_false = pf_region;
1938       contended_phi_rawmem = pf_phi_rawmem;
1939       i_o = pf_phi_abio;
1940    } else if( UseTLAB && AllocatePrefetchStyle == 3 ) {
1941       // Insert a prefetch instruction for each allocation.
1942       // This code is used to generate 1 prefetch instruction per cache line.
1943 
1944       // Generate several prefetch instructions.
1945       uint step_size = AllocatePrefetchStepSize;
1946       uint distance = AllocatePrefetchDistance;
1947 
1948       // Next cache address.
1949       Node *cache_adr = new AddPNode(old_eden_top, old_eden_top,
1950                                      _igvn.MakeConX(step_size + distance));
1951       transform_later(cache_adr);
1952       cache_adr = new CastP2XNode(needgc_false, cache_adr);
1953       transform_later(cache_adr);
1954       // Address is aligned to execute prefetch to the beginning of cache line size
1955       // (it is important when BIS instruction is used on SPARC as prefetch).
1956       Node* mask = _igvn.MakeConX(~(intptr_t)(step_size-1));
1957       cache_adr = new AndXNode(cache_adr, mask);
1958       transform_later(cache_adr);
1959       cache_adr = new CastX2PNode(cache_adr);
1960       transform_later(cache_adr);
1961 
1962       // Prefetch
1963       Node *prefetch = new PrefetchAllocationNode( contended_phi_rawmem, cache_adr );
1964       prefetch->set_req(0, needgc_false);
1965       transform_later(prefetch);
1966       contended_phi_rawmem = prefetch;
1967       Node *prefetch_adr;
1968       distance = step_size;
1969       for ( intx i = 1; i < lines; i++ ) {
1970         prefetch_adr = new AddPNode( cache_adr, cache_adr,
1971                                             _igvn.MakeConX(distance) );
1972         transform_later(prefetch_adr);
1973         prefetch = new PrefetchAllocationNode( contended_phi_rawmem, prefetch_adr );
1974         transform_later(prefetch);
1975         distance += step_size;
1976         contended_phi_rawmem = prefetch;
1977       }
1978    } else if( AllocatePrefetchStyle > 0 ) {
1979       // Insert a prefetch for each allocation only on the fast-path
1980       Node *prefetch_adr;
1981       Node *prefetch;
1982       // Generate several prefetch instructions.
1983       uint step_size = AllocatePrefetchStepSize;
1984       uint distance = AllocatePrefetchDistance;
1985       for ( intx i = 0; i < lines; i++ ) {
1986         prefetch_adr = new AddPNode( old_eden_top, new_eden_top,
1987                                             _igvn.MakeConX(distance) );
1988         transform_later(prefetch_adr);
1989         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
1990         // Do not let it float too high, since if eden_top == eden_end,
1991         // both might be null.
1992         if( i == 0 ) { // Set control for first prefetch, next follows it
1993           prefetch->init_req(0, needgc_false);
1994         }
1995         transform_later(prefetch);
1996         distance += step_size;
1997         i_o = prefetch;
1998       }
1999    }
2000    return i_o;
2001 }

2029 //-------------------mark_eliminated_box----------------------------------
2030 //
2031 // During EA obj may point to several objects but after few ideal graph
2032 // transformations (CCP) it may point to only one non escaping object
2033 // (but still using phi), corresponding locks and unlocks will be marked
2034 // for elimination. Later obj could be replaced with a new node (new phi)
2035 // and which does not have escape information. And later after some graph
2036 // reshape other locks and unlocks (which were not marked for elimination
2037 // before) are connected to this new obj (phi) but they still will not be
2038 // marked for elimination since new obj has no escape information.
2039 // Mark all associated (same box and obj) lock and unlock nodes for
2040 // elimination if some of them marked already.
2041 void PhaseMacroExpand::mark_eliminated_box(Node* oldbox, Node* obj) {
2042   if (oldbox->as_BoxLock()->is_eliminated())
2043     return; // This BoxLock node was processed already.
2044 
2045   // New implementation (EliminateNestedLocks) has separate BoxLock
2046   // node for each locked region so mark all associated locks/unlocks as
2047   // eliminated even if different objects are referenced in one locked region
2048   // (for example, OSR compilation of nested loop inside locked scope).
2049   if (EliminateNestedLocks ||
2050       oldbox->as_BoxLock()->is_simple_lock_region(NULL, obj)) {
2051     // Box is used only in one lock region. Mark this box as eliminated.
2052     _igvn.hash_delete(oldbox);
2053     oldbox->as_BoxLock()->set_eliminated(); // This changes box's hash value
2054      _igvn.hash_insert(oldbox);
2055 
2056     for (uint i = 0; i < oldbox->outcnt(); i++) {
2057       Node* u = oldbox->raw_out(i);
2058       if (u->is_AbstractLock() && !u->as_AbstractLock()->is_non_esc_obj()) {
2059         AbstractLockNode* alock = u->as_AbstractLock();
2060         // Check lock's box since box could be referenced by Lock's debug info.
2061         if (alock->box_node() == oldbox) {
2062           // Mark eliminated all related locks and unlocks.
2063 #ifdef ASSERT
2064           alock->log_lock_optimization(C, "eliminate_lock_set_non_esc4");
2065 #endif
2066           alock->set_non_esc_obj();
2067         }
2068       }
2069     }
2070     return;
2071   }
2072 
2073   // Create new "eliminated" BoxLock node and use it in monitor debug info
2074   // instead of oldbox for the same object.
2075   BoxLockNode* newbox = oldbox->clone()->as_BoxLock();
2076 
2077   // Note: BoxLock node is marked eliminated only here and it is used
2078   // to indicate that all associated lock and unlock nodes are marked
2079   // for elimination.
2080   newbox->set_eliminated();
2081   transform_later(newbox);
2082 
2083   // Replace old box node with new box for all users of the same object.
2084   for (uint i = 0; i < oldbox->outcnt();) {
2085     bool next_edge = true;
2086 
2087     Node* u = oldbox->raw_out(i);
2088     if (u->is_AbstractLock()) {
2089       AbstractLockNode* alock = u->as_AbstractLock();
2090       if (alock->box_node() == oldbox && alock->obj_node()->eqv_uncast(obj)) {
2091         // Replace Box and mark eliminated all related locks and unlocks.
2092 #ifdef ASSERT
2093         alock->log_lock_optimization(C, "eliminate_lock_set_non_esc5");
2094 #endif
2095         alock->set_non_esc_obj();
2096         _igvn.rehash_node_delayed(alock);
2097         alock->set_box_node(newbox);
2098         next_edge = false;
2099       }
2100     }
2101     if (u->is_FastLock() && u->as_FastLock()->obj_node()->eqv_uncast(obj)) {
2102       FastLockNode* flock = u->as_FastLock();
2103       assert(flock->box_node() == oldbox, "sanity");
2104       _igvn.rehash_node_delayed(flock);
2105       flock->set_box_node(newbox);
2106       next_edge = false;
2107     }
2108 
2109     // Replace old box in monitor debug info.
2110     if (u->is_SafePoint() && u->as_SafePoint()->jvms()) {
2111       SafePointNode* sfn = u->as_SafePoint();
2112       JVMState* youngest_jvms = sfn->jvms();
2113       int max_depth = youngest_jvms->depth();
2114       for (int depth = 1; depth <= max_depth; depth++) {
2115         JVMState* jvms = youngest_jvms->of_depth(depth);
2116         int num_mon  = jvms->nof_monitors();
2117         // Loop over monitors
2118         for (int idx = 0; idx < num_mon; idx++) {
2119           Node* obj_node = sfn->monitor_obj(jvms, idx);
2120           Node* box_node = sfn->monitor_box(jvms, idx);
2121           if (box_node == oldbox && obj_node->eqv_uncast(obj)) {
2122             int j = jvms->monitor_box_offset(idx);
2123             _igvn.replace_input_of(u, j, newbox);
2124             next_edge = false;
2125           }
2126         }
2127       }
2128     }
2129     if (next_edge) i++;
2130   }
2131 }

2287 //------------------------------expand_lock_node----------------------
2288 void PhaseMacroExpand::expand_lock_node(LockNode *lock) {
2289 
2290   Node* ctrl = lock->in(TypeFunc::Control);
2291   Node* mem = lock->in(TypeFunc::Memory);
2292   Node* obj = lock->obj_node();
2293   Node* box = lock->box_node();
2294   Node* flock = lock->fastlock_node();
2295 
2296   assert(!box->as_BoxLock()->is_eliminated(), "sanity");
2297 
2298   // Make the merge point
2299   Node *region;
2300   Node *mem_phi;
2301   Node *slow_path;
2302 
2303   if (UseOptoBiasInlining) {
2304     /*
2305      *  See the full description in MacroAssembler::biased_locking_enter().
2306      *
2307      *  if( (mark_word & biased_lock_mask) == biased_lock_pattern ) {
2308      *    // The object is biased.
2309      *    proto_node = klass->prototype_header;
2310      *    o_node = thread | proto_node;
2311      *    x_node = o_node ^ mark_word;
2312      *    if( (x_node & ~age_mask) == 0 ) { // Biased to the current thread ?
2313      *      // Done.
2314      *    } else {
2315      *      if( (x_node & biased_lock_mask) != 0 ) {
2316      *        // The klass's prototype header is no longer biased.
2317      *        cas(&mark_word, mark_word, proto_node)
2318      *        goto cas_lock;
2319      *      } else {
2320      *        // The klass's prototype header is still biased.
2321      *        if( (x_node & epoch_mask) != 0 ) { // Expired epoch?
2322      *          old = mark_word;
2323      *          new = o_node;
2324      *        } else {
2325      *          // Different thread or anonymous biased.
2326      *          old = mark_word & (epoch_mask | age_mask | biased_lock_mask);
2327      *          new = thread | old;
2328      *        }
2329      *        // Try to rebias.
2330      *        if( cas(&mark_word, old, new) == 0 ) {
2331      *          // Done.
2332      *        } else {
2333      *          goto slow_path; // Failed.
2334      *        }
2335      *      }
2336      *    }
2337      *  } else {
2338      *    // The object is not biased.
2339      *    cas_lock:
2340      *    if( FastLock(obj) == 0 ) {
2341      *      // Done.
2342      *    } else {
2343      *      slow_path:
2344      *      OptoRuntime::complete_monitor_locking_Java(obj);
2345      *    }
2346      *  }
2347      */
2348 
2349     region  = new RegionNode(5);
2350     // create a Phi for the memory state
2351     mem_phi = new PhiNode( region, Type::MEMORY, TypeRawPtr::BOTTOM);
2352 
2353     Node* fast_lock_region  = new RegionNode(3);
2354     Node* fast_lock_mem_phi = new PhiNode( fast_lock_region, Type::MEMORY, TypeRawPtr::BOTTOM);
2355 
2356     // First, check mark word for the biased lock pattern.
2357     Node* mark_node = make_load(ctrl, mem, obj, oopDesc::mark_offset_in_bytes(), TypeX_X, TypeX_X->basic_type());
2358 
2359     // Get fast path - mark word has the biased lock pattern.
2360     ctrl = opt_bits_test(ctrl, fast_lock_region, 1, mark_node,
2361                          markWord::biased_lock_mask_in_place,
2362                          markWord::biased_lock_pattern, true);
2363     // fast_lock_region->in(1) is set to slow path.
2364     fast_lock_mem_phi->init_req(1, mem);
2365 
2366     // Now check that the lock is biased to the current thread and has
2367     // the same epoch and bias as Klass::_prototype_header.
2368 
2369     // Special-case a fresh allocation to avoid building nodes:
2370     Node* klass_node = AllocateNode::Ideal_klass(obj, &_igvn);
2371     if (klass_node == NULL) {
2372       Node* k_adr = basic_plus_adr(obj, oopDesc::klass_offset_in_bytes());
2373       klass_node = transform_later(LoadKlassNode::make(_igvn, NULL, mem, k_adr, _igvn.type(k_adr)->is_ptr()));
2374 #ifdef _LP64
2375       if (UseCompressedClassPointers && klass_node->is_DecodeNKlass()) {
2376         assert(klass_node->in(1)->Opcode() == Op_LoadNKlass, "sanity");
2377         klass_node->in(1)->init_req(0, ctrl);
2378       } else
2379 #endif
2380       klass_node->init_req(0, ctrl);
2381     }
2382     Node *proto_node = make_load(ctrl, mem, klass_node, in_bytes(Klass::prototype_header_offset()), TypeX_X, TypeX_X->basic_type());
2383 
2384     Node* thread = transform_later(new ThreadLocalNode());
2385     Node* cast_thread = transform_later(new CastP2XNode(ctrl, thread));
2386     Node* o_node = transform_later(new OrXNode(cast_thread, proto_node));
2387     Node* x_node = transform_later(new XorXNode(o_node, mark_node));
2388 
2389     // Get slow path - mark word does NOT match the value.
2390     STATIC_ASSERT(markWord::age_mask_in_place <= INT_MAX);
2391     Node* not_biased_ctrl =  opt_bits_test(ctrl, region, 3, x_node,
2392                                       (~(int)markWord::age_mask_in_place), 0);
2393     // region->in(3) is set to fast path - the object is biased to the current thread.
2394     mem_phi->init_req(3, mem);
2395 
2396 
2397     // Mark word does NOT match the value (thread | Klass::_prototype_header).
2398 
2399 
2400     // First, check biased pattern.
2401     // Get fast path - _prototype_header has the same biased lock pattern.
2402     ctrl =  opt_bits_test(not_biased_ctrl, fast_lock_region, 2, x_node,
2403                           markWord::biased_lock_mask_in_place, 0, true);
2404 
2405     not_biased_ctrl = fast_lock_region->in(2); // Slow path
2406     // fast_lock_region->in(2) - the prototype header is no longer biased
2407     // and we have to revoke the bias on this object.
2408     // We are going to try to reset the mark of this object to the prototype
2409     // value and fall through to the CAS-based locking scheme.
2410     Node* adr = basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
2411     Node* cas = new StoreXConditionalNode(not_biased_ctrl, mem, adr,
2412                                           proto_node, mark_node);
2413     transform_later(cas);
2414     Node* proj = transform_later(new SCMemProjNode(cas));
2415     fast_lock_mem_phi->init_req(2, proj);
2416 
2417 
2418     // Second, check epoch bits.
2419     Node* rebiased_region  = new RegionNode(3);
2420     Node* old_phi = new PhiNode( rebiased_region, TypeX_X);
2421     Node* new_phi = new PhiNode( rebiased_region, TypeX_X);
2422 
2423     // Get slow path - mark word does NOT match epoch bits.
2424     Node* epoch_ctrl =  opt_bits_test(ctrl, rebiased_region, 1, x_node,
2425                                       markWord::epoch_mask_in_place, 0);
2426     // The epoch of the current bias is not valid, attempt to rebias the object
2427     // toward the current thread.
2428     rebiased_region->init_req(2, epoch_ctrl);
2429     old_phi->init_req(2, mark_node);
2430     new_phi->init_req(2, o_node);
2431 
2432     // rebiased_region->in(1) is set to fast path.
2433     // The epoch of the current bias is still valid but we know
2434     // nothing about the owner; it might be set or it might be clear.
2435     Node* cmask   = MakeConX(markWord::biased_lock_mask_in_place |
2436                              markWord::age_mask_in_place |
2437                              markWord::epoch_mask_in_place);
2438     Node* old = transform_later(new AndXNode(mark_node, cmask));
2439     cast_thread = transform_later(new CastP2XNode(ctrl, thread));
2440     Node* new_mark = transform_later(new OrXNode(cast_thread, old));
2441     old_phi->init_req(1, old);
2442     new_phi->init_req(1, new_mark);
2443 
2444     transform_later(rebiased_region);
2445     transform_later(old_phi);
2446     transform_later(new_phi);
2447 
2448     // Try to acquire the bias of the object using an atomic operation.
2449     // If this fails we will go in to the runtime to revoke the object's bias.
2450     cas = new StoreXConditionalNode(rebiased_region, mem, adr, new_phi, old_phi);
2451     transform_later(cas);
2452     proj = transform_later(new SCMemProjNode(cas));
2453 
2454     // Get slow path - Failed to CAS.
2455     not_biased_ctrl = opt_bits_test(rebiased_region, region, 4, cas, 0, 0);
2456     mem_phi->init_req(4, proj);
2457     // region->in(4) is set to fast path - the object is rebiased to the current thread.
2458 
2459     // Failed to CAS.
2460     slow_path  = new RegionNode(3);
2461     Node *slow_mem = new PhiNode( slow_path, Type::MEMORY, TypeRawPtr::BOTTOM);
2462 
2463     slow_path->init_req(1, not_biased_ctrl); // Capture slow-control
2464     slow_mem->init_req(1, proj);
2465 
2466     // Call CAS-based locking scheme (FastLock node).
2467 
2468     transform_later(fast_lock_region);
2469     transform_later(fast_lock_mem_phi);
2470 
2471     // Get slow path - FastLock failed to lock the object.
2472     ctrl = opt_bits_test(fast_lock_region, region, 2, flock, 0, 0);
2473     mem_phi->init_req(2, fast_lock_mem_phi);
2474     // region->in(2) is set to fast path - the object is locked to the current thread.
2475 
2476     slow_path->init_req(2, ctrl); // Capture slow-control
2477     slow_mem->init_req(2, fast_lock_mem_phi);
2478 
2479     transform_later(slow_path);
2480     transform_later(slow_mem);
2481     // Reset lock's memory edge.
2482     lock->set_req(TypeFunc::Memory, slow_mem);
2483 
2484   } else {
2485     region  = new RegionNode(3);
2486     // create a Phi for the memory state
2487     mem_phi = new PhiNode( region, Type::MEMORY, TypeRawPtr::BOTTOM);
2488 
2489     // Optimize test; set region slot 2
2490     slow_path = opt_bits_test(ctrl, region, 2, flock, 0, 0);
2491     mem_phi->init_req(2, mem);
2492   }
2493 
2494   const TypeOopPtr* objptr = _igvn.type(obj)->make_oopptr();
2495   if (objptr->can_be_value_type()) {
2496     // Deoptimize and re-execute if a value
2497     assert(EnableValhalla, "should only be used if value types are enabled");
2498     Node* mark = make_load(slow_path, mem, obj, oopDesc::mark_offset_in_bytes(), TypeX_X, TypeX_X->basic_type());
2499     Node* value_mask = _igvn.MakeConX(markWord::always_locked_pattern);
2500     Node* is_value = _igvn.transform(new AndXNode(mark, value_mask));
2501     Node* cmp = _igvn.transform(new CmpXNode(is_value, value_mask));
2502     Node* bol = _igvn.transform(new BoolNode(cmp, BoolTest::eq));
2503     Node* unc_ctrl = generate_slow_guard(&slow_path, bol, NULL);
2504 
2505     int trap_request = Deoptimization::make_trap_request(Deoptimization::Reason_class_check, Deoptimization::Action_none);
2506     address call_addr = SharedRuntime::uncommon_trap_blob()->entry_point();
2507     const TypePtr* no_memory_effects = NULL;
2508     JVMState* jvms = lock->jvms();
2509     CallNode* unc = new CallStaticJavaNode(OptoRuntime::uncommon_trap_Type(), call_addr, "uncommon_trap",
2510                                            jvms->bci(), no_memory_effects);
2511 
2512     unc->init_req(TypeFunc::Control, unc_ctrl);
2513     unc->init_req(TypeFunc::I_O, lock->i_o());
2514     unc->init_req(TypeFunc::Memory, mem); // may gc ptrs
2515     unc->init_req(TypeFunc::FramePtr,  lock->in(TypeFunc::FramePtr));
2516     unc->init_req(TypeFunc::ReturnAdr, lock->in(TypeFunc::ReturnAdr));
2517     unc->init_req(TypeFunc::Parms+0, _igvn.intcon(trap_request));
2518     unc->set_cnt(PROB_UNLIKELY_MAG(4));
2519     unc->copy_call_debug_info(&_igvn, lock);
2520 
2521     assert(unc->peek_monitor_box() == box, "wrong monitor");
2522     assert(unc->peek_monitor_obj() == obj, "wrong monitor");
2523 
2524     // pop monitor and push obj back on stack: we trap before the monitorenter
2525     unc->pop_monitor();
2526     unc->grow_stack(unc->jvms(), 1);
2527     unc->set_stack(unc->jvms(), unc->jvms()->stk_size()-1, obj);
2528 
2529     _igvn.register_new_node_with_optimizer(unc);
2530 
2531     Node* ctrl = _igvn.transform(new ProjNode(unc, TypeFunc::Control));
2532     Node* halt = _igvn.transform(new HaltNode(ctrl, lock->in(TypeFunc::FramePtr), "monitor enter on value-type"));
2533     C->root()->add_req(halt);
2534   }
2535 
2536   // Make slow path call
2537   CallNode *call = make_slow_call((CallNode *) lock, OptoRuntime::complete_monitor_enter_Type(),
2538                                   OptoRuntime::complete_monitor_locking_Java(), NULL, slow_path,
2539                                   obj, box, NULL);
2540 
2541   extract_call_projections(call);
2542 
2543   // Slow path can only throw asynchronous exceptions, which are always
2544   // de-opted.  So the compiler thinks the slow-call can never throw an
2545   // exception.  If it DOES throw an exception we would need the debug
2546   // info removed first (since if it throws there is no monitor).
2547   assert ( _ioproj_fallthrough == NULL && _ioproj_catchall == NULL &&
2548            _memproj_catchall == NULL && _catchallcatchproj == NULL, "Unexpected projection from Lock");
2549 
2550   // Capture slow path
2551   // disconnect fall-through projection from call and create a new one
2552   // hook up users of fall-through projection to region
2553   Node *slow_ctrl = _fallthroughproj->clone();
2554   transform_later(slow_ctrl);
2555   _igvn.hash_delete(_fallthroughproj);
2556   _fallthroughproj->disconnect_inputs(NULL, C);
2557   region->init_req(1, slow_ctrl);
2558   // region inputs are now complete
2559   transform_later(region);
2560   _igvn.replace_node(_fallthroughproj, region);
2561 
2562   Node *memproj = transform_later(new ProjNode(call, TypeFunc::Memory));
2563   mem_phi->init_req(1, memproj );
2564   transform_later(mem_phi);
2565   _igvn.replace_node(_memproj_fallthrough, mem_phi);
2566 }

2879 //---------------------------eliminate_macro_nodes----------------------
2880 // Eliminate scalar replaced allocations and associated locks.
2881 void PhaseMacroExpand::eliminate_macro_nodes() {
2882   if (C->macro_count() == 0)
2883     return;
2884 
2885   // First, attempt to eliminate locks
2886   int cnt = C->macro_count();
2887   for (int i=0; i < cnt; i++) {
2888     Node *n = C->macro_node(i);
2889     if (n->is_AbstractLock()) { // Lock and Unlock nodes
2890       // Before elimination mark all associated (same box and obj)
2891       // lock and unlock nodes.
2892       mark_eliminated_locking_nodes(n->as_AbstractLock());
2893     }
2894   }
2895   bool progress = true;
2896   while (progress) {
2897     progress = false;
2898     for (int i = C->macro_count(); i > 0; i--) {
2899       Node * n = C->macro_node(i-1);
2900       bool success = false;
2901       debug_only(int old_macro_count = C->macro_count(););
2902       if (n->is_AbstractLock()) {
2903         success = eliminate_locking_node(n->as_AbstractLock());
2904       }
2905       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2906       progress = progress || success;
2907     }
2908   }
2909   // Next, attempt to eliminate allocations
2910   _has_locks = false;
2911   progress = true;
2912   while (progress) {
2913     progress = false;
2914     for (int i = C->macro_count(); i > 0; i--) {
2915       Node * n = C->macro_node(i-1);
2916       bool success = false;
2917       debug_only(int old_macro_count = C->macro_count(););
2918       switch (n->class_id()) {
2919       case Node::Class_Allocate:
2920       case Node::Class_AllocateArray:
2921         success = eliminate_allocate_node(n->as_Allocate());
2922         break;
2923       case Node::Class_CallStaticJava: {
2924         CallStaticJavaNode* call = n->as_CallStaticJava();
2925         if (!call->method()->is_method_handle_intrinsic()) {
2926           success = eliminate_boxing_node(n->as_CallStaticJava());
2927         }
2928         break;
2929       }
2930       case Node::Class_Lock:
2931       case Node::Class_Unlock:
2932         assert(!n->as_AbstractLock()->is_eliminated(), "sanity");
2933         _has_locks = true;
2934         break;
2935       case Node::Class_ArrayCopy:
2936         break;
2937       case Node::Class_OuterStripMinedLoop:
2938         break;
2939       case Node::Class_SubTypeCheck:
2940         break;
2941       default:
2942         assert(n->Opcode() == Op_LoopLimit ||
2943                n->Opcode() == Op_Opaque1   ||
2944                n->Opcode() == Op_Opaque2   ||
2945                n->Opcode() == Op_Opaque3   ||
2946                BarrierSet::barrier_set()->barrier_set_c2()->is_gc_barrier_node(n),
2947                "unknown node type in macro list");
2948       }
2949       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2950       progress = progress || success;
2951     }
2952   }
2953 }

2955 //------------------------------expand_macro_nodes----------------------
2956 //  Returns true if a failure occurred.
2957 bool PhaseMacroExpand::expand_macro_nodes() {
2958   // Last attempt to eliminate macro nodes.
2959   eliminate_macro_nodes();
2960 
2961   // Eliminate Opaque and LoopLimit nodes. Do it after all loop optimizations.
2962   bool progress = true;
2963   while (progress) {
2964     progress = false;
2965     for (int i = C->macro_count(); i > 0; i--) {
2966       Node* n = C->macro_node(i-1);
2967       bool success = false;
2968       debug_only(int old_macro_count = C->macro_count(););
2969       if (n->Opcode() == Op_LoopLimit) {
2970         // Remove it from macro list and put on IGVN worklist to optimize.
2971         C->remove_macro_node(n);
2972         _igvn._worklist.push(n);
2973         success = true;
2974       } else if (n->Opcode() == Op_CallStaticJava) {
2975         CallStaticJavaNode* call = n->as_CallStaticJava();
2976         if (!call->method()->is_method_handle_intrinsic()) {
2977           // Remove it from macro list and put on IGVN worklist to optimize.
2978           C->remove_macro_node(n);
2979           _igvn._worklist.push(n);
2980           success = true;
2981         }
2982       } else if (n->Opcode() == Op_Opaque1 || n->Opcode() == Op_Opaque2) {
2983         _igvn.replace_node(n, n->in(1));
2984         success = true;
2985 #if INCLUDE_RTM_OPT
2986       } else if ((n->Opcode() == Op_Opaque3) && ((Opaque3Node*)n)->rtm_opt()) {
2987         assert(C->profile_rtm(), "should be used only in rtm deoptimization code");
2988         assert((n->outcnt() == 1) && n->unique_out()->is_Cmp(), "");
2989         Node* cmp = n->unique_out();
2990 #ifdef ASSERT
2991         // Validate graph.
2992         assert((cmp->outcnt() == 1) && cmp->unique_out()->is_Bool(), "");
2993         BoolNode* bol = cmp->unique_out()->as_Bool();
2994         assert((bol->outcnt() == 1) && bol->unique_out()->is_If() &&
2995                (bol->_test._test == BoolTest::ne), "");
2996         IfNode* ifn = bol->unique_out()->as_If();
2997         assert((ifn->outcnt() == 2) &&
2998                ifn->proj_out(1)->is_uncommon_trap_proj(Deoptimization::Reason_rtm_state_change) != NULL, "");
2999 #endif
3000         Node* repl = n->in(1);
3001         if (!_has_locks) {
3002           // Remove RTM state check if there are no locks in the code.
3003           // Replace input to compare the same value.
3004           repl = (cmp->in(1) == n) ? cmp->in(2) : cmp->in(1);
3005         }
3006         _igvn.replace_node(n, repl);
3007         success = true;
3008 #endif
3009       } else if (n->Opcode() == Op_OuterStripMinedLoop) {
3010         n->as_OuterStripMinedLoop()->adjust_strip_mined_loop(&_igvn);
3011         C->remove_macro_node(n);
3012         success = true;
3013       }
3014       assert(!success || (C->macro_count() == (old_macro_count - 1)), "elimination must have deleted one node from macro list");
3015       progress = progress || success;
3016     }
3017   }
3018 
3019   // Clean up the graph so we're less likely to hit the maximum node
3020   // limit
3021   _igvn.set_delay_transform(false);
3022   _igvn.optimize();
3023   if (C->failing())  return true;
3024   _igvn.set_delay_transform(true);
3025 
3026 
3027   // Because we run IGVN after each expansion, some macro nodes may go
3028   // dead and be removed from the list as we iterate over it. Move
3029   // Allocate nodes (processed in a second pass) at the beginning of
3030   // the list and then iterate from the last element of the list until
3031   // an Allocate node is seen. This is robust to random deletion in
3032   // the list due to nodes going dead.
3033   C->sort_macro_nodes();
3034 
3035   // expand arraycopy "macro" nodes first
3036   // For ReduceBulkZeroing, we must first process all arraycopy nodes
3037   // before the allocate nodes are expanded.
3038   while (C->macro_count() > 0) {
3039     int macro_count = C->macro_count();
3040     Node * n = C->macro_node(macro_count-1);
3041     assert(n->is_macro(), "only macro nodes expected here");
3042     if (_igvn.type(n) == Type::TOP || (n->in(0) != NULL && n->in(0)->is_top())) {
3043       // node is unreachable, so don't try to expand it
3044       C->remove_macro_node(n);
3045       continue;
3046     }
3047     if (n->is_Allocate()) {
3048       break;
3049     }
3050     // Make sure expansion will not cause node limit to be exceeded.
3051     // Worst case is a macro node gets expanded into about 200 nodes.
3052     // Allow 50% more for optimization.
3053     if (C->check_node_count(300, "out of nodes before macro expansion")) {
3054       return true;
3055     }
3056 
3057     debug_only(int old_macro_count = C->macro_count(););
3058     switch (n->class_id()) {
3059     case Node::Class_Lock:

3065       assert(C->macro_count() == (old_macro_count - 1), "expansion must have deleted one node from macro list");

3074       break;
3075     case Node::Class_CallStaticJava:
3076       expand_mh_intrinsic_return(n->as_CallStaticJava());

3078       assert(C->macro_count() == (old_macro_count - 1), "expansion must have deleted one node from macro list");
3079       break;
3080     default:
3081       assert(false, "unknown node type in macro list");
3082     }
3083     assert(C->macro_count() < macro_count, "must have deleted a node from macro list");
3084     if (C->failing())  return true;
3085 
3086     // Clean up the graph so we're less likely to hit the maximum node
3087     // limit
3088     _igvn.set_delay_transform(false);
3089     _igvn.optimize();
3090     if (C->failing())  return true;
3091     _igvn.set_delay_transform(true);
3092   }
3093 
3094   // All nodes except Allocate nodes are expanded now. There could be
3095   // new optimization opportunities (such as folding newly created
3096   // load from a just allocated object). Run IGVN.
3097 
3098   // expand "macro" nodes
3099   // nodes are removed from the macro list as they are processed
3100   while (C->macro_count() > 0) {
3101     int macro_count = C->macro_count();
3102     Node * n = C->macro_node(macro_count-1);
3103     assert(n->is_macro(), "only macro nodes expected here");
3104     if (_igvn.type(n) == Type::TOP || (n->in(0) != NULL && n->in(0)->is_top())) {
3105       // node is unreachable, so don't try to expand it
3106       C->remove_macro_node(n);
3107       continue;
3108     }
3109     // Make sure expansion will not cause node limit to be exceeded.
3110     // Worst case is a macro node gets expanded into about 200 nodes.
3111     // Allow 50% more for optimization.
3112     if (C->check_node_count(300, "out of nodes before macro expansion")) {
3113       return true;
3114     }
3115     switch (n->class_id()) {
3116     case Node::Class_Allocate:
3117       expand_allocate(n->as_Allocate());
3118       break;
3119     case Node::Class_AllocateArray:
3120       expand_allocate_array(n->as_AllocateArray());
3121       break;
3122     default:
3123       assert(false, "unknown node type in macro list");
3124     }
3125     assert(C->macro_count() < macro_count, "must have deleted a node from macro list");
3126     if (C->failing())  return true;
3127 
3128     // Clean up the graph so we're less likely to hit the maximum node
3129     // limit
3130     _igvn.set_delay_transform(false);
3131     _igvn.optimize();

3133     _igvn.set_delay_transform(true);
3134   }
