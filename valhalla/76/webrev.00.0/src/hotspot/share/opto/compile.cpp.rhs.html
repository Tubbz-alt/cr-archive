  97 // Return the index at which m must be inserted (or already exists).
  98 // The sort order is by the address of the ciMethod, with is_virtual as minor key.
  99 class IntrinsicDescPair {
 100  private:
 101   ciMethod* _m;
 102   bool _is_virtual;
 103  public:
 104   IntrinsicDescPair(ciMethod* m, bool is_virtual) : _m(m), _is_virtual(is_virtual) {}
 105   static int compare(IntrinsicDescPair* const& key, CallGenerator* const& elt) {
 106     ciMethod* m= elt->method();
 107     ciMethod* key_m = key->_m;
 108     if (key_m < m)      return -1;
 109     else if (key_m > m) return 1;
 110     else {
 111       bool is_virtual = elt->is_virtual();
 112       bool key_virtual = key->_is_virtual;
 113       if (key_virtual < is_virtual)      return -1;
 114       else if (key_virtual > is_virtual) return 1;
 115       else                               return 0;
 116     }
 117   }
 118 };
 119 int Compile::intrinsic_insertion_index(ciMethod* m, bool is_virtual, bool& found) {
 120 #ifdef ASSERT
 121   for (int i = 1; i < _intrinsics->length(); i++) {
 122     CallGenerator* cg1 = _intrinsics->at(i-1);
 123     CallGenerator* cg2 = _intrinsics->at(i);
 124     assert(cg1->method() != cg2->method()
 125            ? cg1->method()     < cg2->method()
 126            : cg1->is_virtual() < cg2->is_virtual(),
 127            "compiler intrinsics list must stay sorted");
 128   }
 129 #endif
 130   IntrinsicDescPair pair(m, is_virtual);
 131   return _intrinsics->find_sorted<IntrinsicDescPair*, IntrinsicDescPair::compare>(&pair, found);
 132 }

 134 void Compile::register_intrinsic(CallGenerator* cg) {
 135   if (_intrinsics == NULL) {
 136     _intrinsics = new (comp_arena())GrowableArray<CallGenerator*>(comp_arena(), 60, 0, NULL);
 137   }
 138   int len = _intrinsics->length();
 139   bool found = false;
 140   int index = intrinsic_insertion_index(cg->method(), cg->is_virtual(), found);
 141   assert(!found, "registering twice");
 142   _intrinsics->insert_before(index, cg);
 143   assert(find_intrinsic(cg->method(), cg->is_virtual()) == cg, "registration worked");
 144 }

 146 CallGenerator* Compile::find_intrinsic(ciMethod* m, bool is_virtual) {
 147   assert(m->is_loaded(), "don't try this on unloaded methods");
 148   if (_intrinsics != NULL) {
 149     bool found = false;
 150     int index = intrinsic_insertion_index(m, is_virtual, found);
 151      if (found) {
 152       return _intrinsics->at(index);
 153     }
 154   }
 155   // Lazily create intrinsics for intrinsic IDs well-known in the runtime.
 156   if (m->intrinsic_id() != vmIntrinsics::_none &&
 157       m->intrinsic_id() <= vmIntrinsics::LAST_COMPILER_INLINE) {
 158     CallGenerator* cg = make_vm_intrinsic(m, is_virtual);
 159     if (cg != NULL) {
 160       // Save it for next time:
 161       register_intrinsic(cg);
 162       return cg;
 163     } else {
 164       gather_intrinsic_statistics(m->intrinsic_id(), is_virtual, _intrinsic_disabled);
 165     }
 166   }
 167   return NULL;
 168 }

 177 juint  Compile::_intrinsic_hist_count[vmIntrinsics::ID_LIMIT] = {0};
 178 jubyte Compile::_intrinsic_hist_flags[vmIntrinsics::ID_LIMIT] = {0};
 179 
 180 bool Compile::gather_intrinsic_statistics(vmIntrinsics::ID id, bool is_virtual, int flags) {
 181   assert(id > vmIntrinsics::_none && id < vmIntrinsics::ID_LIMIT, "oob");
 182   int oflags = _intrinsic_hist_flags[id];
 183   assert(flags != 0, "what happened?");
 184   if (is_virtual) {
 185     flags |= _intrinsic_virtual;
 186   }
 187   bool changed = (flags != oflags);
 188   if ((flags & _intrinsic_worked) != 0) {
 189     juint count = (_intrinsic_hist_count[id] += 1);
 190     if (count == 1) {
 191       changed = true;           // first time
 192     }
 193     // increment the overall count also:
 194     _intrinsic_hist_count[vmIntrinsics::_none] += 1;
 195   }
 196   if (changed) {
 197     if (((oflags ^ flags) & _intrinsic_virtual) != 0) {
 198       // Something changed about the intrinsic's virtuality.
 199       if ((flags & _intrinsic_virtual) != 0) {
 200         // This is the first use of this intrinsic as a virtual call.
 201         if (oflags != 0) {
 202           // We already saw it as a non-virtual, so note both cases.
 203           flags |= _intrinsic_both;
 204         }
 205       } else if ((oflags & _intrinsic_both) == 0) {
 206         // This is the first use of this intrinsic as a non-virtual
 207         flags |= _intrinsic_both;
 208       }
 209     }
 210     _intrinsic_hist_flags[id] = (jubyte) (oflags | flags);
 211   }
 212   // update the overall flags also:
 213   _intrinsic_hist_flags[vmIntrinsics::_none] |= (jubyte) flags;
 214   return changed;
 215 }

 217 static char* format_flags(int flags, char* buf) {
 218   buf[0] = 0;
 219   if ((flags & Compile::_intrinsic_worked) != 0)    strcat(buf, ",worked");
 220   if ((flags & Compile::_intrinsic_failed) != 0)    strcat(buf, ",failed");
 221   if ((flags & Compile::_intrinsic_disabled) != 0)  strcat(buf, ",disabled");
 222   if ((flags & Compile::_intrinsic_virtual) != 0)   strcat(buf, ",virtual");
 223   if ((flags & Compile::_intrinsic_both) != 0)      strcat(buf, ",nonvirtual");
 224   if (buf[0] == 0)  strcat(buf, ",");
 225   assert(buf[0] == ',', "must be");
 226   return &buf[1];
 227 }

 268 void Compile::gvn_replace_by(Node* n, Node* nn) {
 269   for (DUIterator_Last imin, i = n->last_outs(imin); i >= imin; ) {
 270     Node* use = n->last_out(i);
 271     bool is_in_table = initial_gvn()->hash_delete(use);
 272     uint uses_found = 0;
 273     for (uint j = 0; j < use->len(); j++) {
 274       if (use->in(j) == n) {
 275         if (j < use->req())
 276           use->set_req(j, nn);
 277         else
 278           use->set_prec(j, nn);
 279         uses_found++;
 280       }
 281     }
 282     if (is_in_table) {
 283       // reinsert into table
 284       initial_gvn()->hash_find_insert(use);
 285     }
 286     record_for_igvn(use);
 287     i -= uses_found;    // we deleted 1 or more copies of this edge
 288   }
 289 }

 292 static inline bool not_a_node(const Node* n) {
 293   if (n == NULL)                   return true;
 294   if (((intptr_t)n & 1) != 0)      return true;  // uninitialized, etc.
 295   if (*(address*)n == badAddress)  return true;  // kill by Node::destruct
 296   return false;
 297 }

 299 // Identify all nodes that are reachable from below, useful.
 300 // Use breadth-first pass that records state in a Unique_Node_List,
 301 // recursive traversal is slower.
 302 void Compile::identify_useful_nodes(Unique_Node_List &useful) {
 303   int estimated_worklist_size = live_nodes();
 304   useful.map( estimated_worklist_size, NULL );  // preallocate space
 305 
 306   // Initialize worklist
 307   if (root() != NULL)     { useful.push(root()); }
 308   // If 'top' is cached, declare it useful to preserve cached node
 309   if( cached_top_node() ) { useful.push(cached_top_node()); }
 310 
 311   // Push all useful nodes onto the list, breadthfirst
 312   for( uint next = 0; next < useful.size(); ++next ) {
 313     assert( next < unique(), "Unique useful nodes < total nodes");
 314     Node *n  = useful.at(next);
 315     uint max = n->len();
 316     for( uint i = 0; i < max; ++i ) {
 317       Node *m = n->in(i);
 318       if (not_a_node(m))  continue;
 319       useful.push(m);
 320     }
 321   }
 322 }

 324 // Update dead_node_list with any missing dead nodes using useful
 325 // list. Consider all non-useful nodes to be useless i.e., dead nodes.
 326 void Compile::update_dead_node_list(Unique_Node_List &useful) {
 327   uint max_idx = unique();
 328   VectorSet& useful_node_set = useful.member_set();
 329 
 330   for (uint node_idx = 0; node_idx < max_idx; node_idx++) {
 331     // If node with index node_idx is not in useful set,
 332     // mark it as dead in dead node list.
 333     if (!useful_node_set.test(node_idx)) {
 334       record_dead_node(node_idx);
 335     }
 336   }
 337 }

 339 void Compile::remove_useless_late_inlines(GrowableArray<CallGenerator*>* inlines, Unique_Node_List &useful) {
 340   int shift = 0;
 341   for (int i = 0; i < inlines->length(); i++) {
 342     CallGenerator* cg = inlines->at(i);
 343     CallNode* call = cg->call_node();
 344     if (shift > 0) {
 345       inlines->at_put(i-shift, cg);
 346     }
 347     if (!useful.member(call)) {
 348       shift++;
 349     }
 350   }
 351   inlines->trunc_to(inlines->length()-shift);
 352 }

 354 // Disconnect all useless nodes by disconnecting those at the boundary.
 355 void Compile::remove_useless_nodes(Unique_Node_List &useful) {
 356   uint next = 0;
 357   while (next < useful.size()) {
 358     Node *n = useful.at(next++);
 359     if (n->is_SafePoint()) {
 360       // We're done with a parsing phase. Replaced nodes are not valid
 361       // beyond that point.
 362       n->as_SafePoint()->delete_replaced_nodes();
 363     }
 364     // Use raw traversal of out edges since this code removes out edges
 365     int max = n->outcnt();
 366     for (int j = 0; j < max; ++j) {
 367       Node* child = n->raw_out(j);
 368       if (! useful.member(child)) {
 369         assert(!child->is_top() || child != top(),
 370                "If top is cached in Compile object it is in useful list");
 371         // Only need to remove this out-edge to the useless node
 372         n->raw_del_out(j);
 373         --j;
 374         --max;
 375       }
 376     }
 377     if (n->outcnt() == 1 && n->has_special_unique_user()) {
 378       record_for_igvn(n->unique_out());
 379     }
 380   }
 381   // Remove useless macro and predicate opaq nodes
 382   for (int i = C->macro_count()-1; i >= 0; i--) {
 383     Node* n = C->macro_node(i);
 384     if (!useful.member(n)) {
 385       remove_macro_node(n);
 386     }
 387   }
 388   // Remove useless CastII nodes with range check dependency
 389   for (int i = range_check_cast_count() - 1; i >= 0; i--) {
 390     Node* cast = range_check_cast_node(i);
 391     if (!useful.member(cast)) {
 392       remove_range_check_cast(cast);
 393     }
 394   }
 395   // Remove useless expensive nodes
 396   for (int i = C->expensive_count()-1; i >= 0; i--) {
 397     Node* n = C->expensive_node(i);
 398     if (!useful.member(n)) {
 399       remove_expensive_node(n);
 400     }
 401   }
 402   // Remove useless Opaque4 nodes
 403   for (int i = opaque4_count() - 1; i >= 0; i--) {
 404     Node* opaq = opaque4_node(i);
 405     if (!useful.member(opaq)) {
 406       remove_opaque4_node(opaq);
 407     }
 408   }
 409   // Remove useless value type nodes
 410   if (_value_type_nodes != NULL) {
 411     _value_type_nodes->remove_useless_nodes(useful.member_set());
 412   }
 413   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 414   bs->eliminate_useless_gc_barriers(useful, this);
 415   // clean up the late inline lists
 416   remove_useless_late_inlines(&_string_late_inlines, useful);
 417   remove_useless_late_inlines(&_boxing_late_inlines, useful);
 418   remove_useless_late_inlines(&_late_inlines, useful);
 419   debug_only(verify_graph_edges(true/*check for no_dead_code*/);)
 420 }

 586   // We can always print a disassembly, either abstract (hex dump) or
 587   // with the help of a suitable hsdis library. Thus, we should not
 588   // couple print_assembly and print_opto_assembly controls.
 589   // But: always print opto and regular assembly on compile command 'print'.
 590   bool print_assembly = directive->PrintAssemblyOption;
 591   set_print_assembly(print_opto_assembly || print_assembly);
 592 #else
 593   set_print_assembly(false); // must initialize.
 594 #endif
 595 
 596 #ifndef PRODUCT
 597   set_parsed_irreducible_loop(false);
 598 
 599   if (directive->ReplayInlineOption) {
 600     _replay_inline_data = ciReplay::load_inline_data(method(), entry_bci(), ci_env->comp_level());
 601   }
 602 #endif
 603   set_print_inlining(directive->PrintInliningOption || PrintOptoInlining);
 604   set_print_intrinsics(directive->PrintIntrinsicsOption);
 605   set_has_irreducible_loop(true); // conservative until build_loop_tree() reset it
 606 
 607   if (ProfileTraps RTM_OPT_ONLY( || UseRTMLocking )) {
 608     // Make sure the method being compiled gets its own MDO,
 609     // so we can at least track the decompile_count().
 610     // Need MDO to record RTM code generation state.
 611     method()->ensure_method_data();
 612   }
 613 
 614   Init(::AliasLevel);
 615 
 616 
 617   print_compile_messages();
 618 
 619   _ilt = InlineTree::build_inline_tree_root();
 620 
 621   // Even if NO memory addresses are used, MergeMem nodes must have at least 1 slice
 622   assert(num_alias_types() >= AliasIdxRaw, "");
 623 
 624 #define MINIMUM_NODE_HASH  1023
 625   // Node list that Iterative GVN will start with
 626   Unique_Node_List for_igvn(comp_arena());
 627   set_for_igvn(&for_igvn);
 628 
 629   // GVN that will be run immediately on new nodes
 630   uint estimated_size = method()->code_size()*4+64;
 631   estimated_size = (estimated_size < MINIMUM_NODE_HASH ? MINIMUM_NODE_HASH : estimated_size);
 632   PhaseGVN gvn(node_arena(), estimated_size);
 633   set_initial_gvn(&gvn);
 634 
 635   print_inlining_init();
 636   { // Scope for timing the parser
 637     TracePhase tp("parse", &timers[_t_parser]);
 638 
 639     // Put top into the hash table ASAP.
 640     initial_gvn()->transform_no_reclaim(top());
 641 
 642     // Set up tf(), start(), and find a CallGenerator.
 643     CallGenerator* cg = NULL;
 644     if (is_osr_compilation()) {
 645       init_tf(TypeFunc::make(method(), /* is_osr_compilation = */ true));
 646       StartNode* s = new StartOSRNode(root(), tf()->domain_sig());
 647       initial_gvn()->set_type_bottom(s);
 648       init_start(s);
 649       cg = CallGenerator::for_osr(method(), entry_bci());
 650     } else {
 651       // Normal case.
 652       init_tf(TypeFunc::make(method()));
 653       StartNode* s = new StartNode(root(), tf()->domain_cc());
 654       initial_gvn()->set_type_bottom(s);
 655       init_start(s);
 656       if (method()->intrinsic_id() == vmIntrinsics::_Reference_get) {
 657         // With java.lang.ref.reference.get() we must go through the
 658         // intrinsic - even when get() is the root
 659         // method of the compile - so that, if necessary, the value in
 660         // the referent field of the reference object gets recorded by
 661         // the pre-barrier code.
 662         cg = find_intrinsic(method(), false);
 663       }
 664       if (cg == NULL) {
 665         float past_uses = method()->interpreter_invocation_count();
 666         float expected_uses = past_uses;
 667         cg = CallGenerator::for_inline(method(), expected_uses);
 668       }
 669     }
 670     if (failing())  return;
 671     if (cg == NULL) {
 672       record_method_not_compilable("cannot parse method");
 673       return;
 674     }
 675     JVMState* jvms = build_start_state(start(), tf());
 676     if ((jvms = cg->generate(jvms)) == NULL) {
 677       if (!failure_reason_is(C2Compiler::retry_class_loading_during_parsing())) {
 678         record_method_not_compilable("method parse failed");
 679       }
 680       return;
 681     }
 682     GraphKit kit(jvms);
 683 
 684     if (!kit.stopped()) {
 685       // Accept return values, and transfer control we know not where.
 686       // This is done by a special, unique ReturnNode bound to root.
 687       return_values(kit.jvms());
 688     }
 689 
 690     if (kit.has_exceptions()) {
 691       // Any exceptions that escape from this call must be rethrown
 692       // to whatever caller is dynamically above us on the stack.
 693       // This is done by a special, unique RethrowNode bound to root.
 694       rethrow_exceptions(kit.transfer_exceptions_into_jvms());
 695     }
 696 
 697     assert(IncrementalInline || (_late_inlines.length() == 0 && !has_mh_late_inlines()), "incremental inlining is off");
 698 
 699     if (_late_inlines.length() == 0 && !has_mh_late_inlines() && !failing() && has_stringbuilder()) {
 700       inline_string_calls(true);
 701     }
 702 
 703     if (failing())  return;
 704 
 705     print_method(PHASE_BEFORE_REMOVEUSELESS, 3);
 706 
 707     // Remove clutter produced by parsing.
 708     if (!failing()) {
 709       ResourceMark rm;
 710       PhaseRemoveUseless pru(initial_gvn(), &for_igvn);
 711     }
 712   }
 713 
 714   // Note:  Large methods are capped off in do_one_bytecode().
 715   if (failing())  return;
 716 
 717   // After parsing, node notes are no longer automagic.
 718   // They must be propagated by register_new_node_with_optimizer(),
 719   // clone(), or the like.
 720   set_default_node_notes(NULL);
 721 
 722   for (;;) {
 723     int successes = Inline_Warm();
 724     if (failing())  return;
 725     if (successes == 0)  break;
 726   }
 727 
 728   // Drain the list.
 729   Finish_Warm();
 730 #ifndef PRODUCT
 731   if (_printer && _printer->should_print(1)) {
 732     _printer->print_inlining();
 733   }
 734 #endif
 735 
 736   if (failing())  return;
 737   NOT_PRODUCT( verify_graph_edges(); )
 738 
 739   // Now optimize
 740   Optimize();
 741   if (failing())  return;
 742   NOT_PRODUCT( verify_graph_edges(); )
 743 
 744 #ifndef PRODUCT
 745   if (print_ideal()) {
 746     ttyLocker ttyl;  // keep the following output all in one block
 747     // This output goes directly to the tty, not the compiler log.
 748     // To enable tools to match it up with the compilation activity,
 749     // be sure to tag this tty output with the compile ID.
 750     if (xtty != NULL) {
 751       xtty->head("ideal compile_id='%d'%s", compile_id(),
 752                  is_osr_compilation()    ? " compile_kind='osr'" :
 753                  "");
 754     }
 755     root()->dump(9999);
 756     if (xtty != NULL) {
 757       xtty->tail("ideal");
 758     }
 759   }
 760 #endif
 761 
 762 #ifdef ASSERT
 763   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 764   bs->verify_gc_barriers(this, BarrierSetC2::BeforeCodeGen);
 765 #endif
 766 
 767   // Dump compilation data to replay it.
 768   if (directive->DumpReplayOption) {
 769     env()->dump_replay_data(_compile_id);
 770   }
 771   if (directive->DumpInlineOption && (ilt() != NULL)) {
 772     env()->dump_inline_data(_compile_id);
 773   }
 774 
 775   // Now that we know the size of all the monitors we can add a fixed slot
 776   // for the original deopt pc.
 777   int next_slot = fixed_slots() + (sizeof(address) / VMRegImpl::stack_slot_size);
 778   if (needs_stack_repair()) {
 779     // One extra slot for the special stack increment value
 780     next_slot += 2;
 781   }
 782   set_fixed_slots(next_slot);
 783 
 784   // Compute when to use implicit null checks. Used by matching trap based
 785   // nodes and NullCheck optimization.
 786   set_allowed_deopt_reasons();
 787 
 788   // Now generate code
 789   Code_Gen();
 790 }

1038 /**
1039  * Return the 'StartNode'. We must not have a pending failure, since the ideal graph
1040  * can be in an inconsistent state, i.e., we can get segmentation faults when traversing
1041  * the ideal graph.
1042  */
1043 StartNode* Compile::start() const {
1044   assert (!failing(), "Must not have pending failure. Reason is: %s", failure_reason());
1045   for (DUIterator_Fast imax, i = root()->fast_outs(imax); i < imax; i++) {
1046     Node* start = root()->fast_out(i);
1047     if (start->is_Start()) {
1048       return start->as_Start();
1049     }
1050   }
1051   fatal("Did not find Start node!");
1052   return NULL;
1053 }

1070   return NULL;

1094 void Compile::print_missing_nodes() {
1095 
1096   // Return if CompileLog is NULL and PrintIdealNodeCount is false.
1097   if ((_log == NULL) && (! PrintIdealNodeCount)) {
1098     return;
1099   }
1100 
1101   // This is an expensive function. It is executed only when the user
1102   // specifies VerifyIdealNodeCount option or otherwise knows the
1103   // additional work that needs to be done to identify reachable nodes
1104   // by walking the flow graph and find the missing ones using
1105   // _dead_node_list.
1106 
1107   Unique_Node_List useful(comp_arena());
1108   // Get useful node list by walking the graph.
1109   identify_useful_nodes(useful);
1110 
1111   uint l_nodes = C->live_nodes();
1112   uint l_nodes_by_walk = useful.size();
1113 
1114   if (l_nodes != l_nodes_by_walk) {
1115     if (_log != NULL) {
1116       _log->begin_head("mismatched_nodes count='%d'", abs((int) (l_nodes - l_nodes_by_walk)));
1117       _log->stamp();
1118       _log->end_head();
1119     }
1120     VectorSet& useful_member_set = useful.member_set();
1121     int last_idx = l_nodes_by_walk;
1122     for (int i = 0; i < last_idx; i++) {
1123       if (useful_member_set.test(i)) {
1124         if (_dead_node_list.test(i)) {
1125           if (_log != NULL) {
1126             _log->elem("mismatched_node_info node_idx='%d' type='both live and dead'", i);
1127           }
1128           if (PrintIdealNodeCount) {
1129             // Print the log message to tty
1130               tty->print_cr("mismatched_node idx='%d' both live and dead'", i);
1131               useful.at(i)->dump();
1132           }
1133         }
1134       }
1135       else if (! _dead_node_list.test(i)) {
1136         if (_log != NULL) {
1137           _log->elem("mismatched_node_info node_idx='%d' type='neither live nor dead'", i);
1138         }
1139         if (PrintIdealNodeCount) {
1140           // Print the log message to tty
1141           tty->print_cr("mismatched_node idx='%d' type='neither live nor dead'", i);
1142         }

1192 bool Compile::copy_node_notes_to(Node* dest, Node* source) {

1204 
1205   if (node_note_array() == NULL)

1240 //------------------------------flatten_alias_type-----------------------------
1241 const TypePtr *Compile::flatten_alias_type( const TypePtr *tj ) const {
1242   int offset = tj->offset();
1243   TypePtr::PTR ptr = tj->ptr();
1244 
1245   // Known instance (scalarizable allocation) alias only with itself.
1246   bool is_known_inst = tj->isa_oopptr() != NULL &&
1247                        tj->is_oopptr()->is_known_instance();
1248 
1249   // Process weird unsafe references.
1250   if (offset == Type::OffsetBot && (tj->isa_instptr() /*|| tj->isa_klassptr()*/)) {
1251     bool default_value_load = EnableValhalla && tj->is_instptr()->klass() == ciEnv::current()->Class_klass();
1252     assert(InlineUnsafeOps || default_value_load, "indeterminate pointers come only from unsafe ops");
1253     assert(!is_known_inst, "scalarizable allocation should not have unsafe references");
1254     tj = TypeOopPtr::BOTTOM;
1255     ptr = tj->ptr();
1256     offset = tj->offset();
1257   }
1258 
1259   // Array pointers need some flattening
1260   const TypeAryPtr *ta = tj->isa_aryptr();
1261   if (ta && ta->is_stable()) {
1262     // Erase stability property for alias analysis.
1263     tj = ta = ta->cast_to_stable(false);
1264   }
1265   if (ta && ta->is_not_flat()) {
1266     // Erase not flat property for alias analysis.
1267     tj = ta = ta->cast_to_not_flat(false);
1268   }
1269   if (ta && ta->is_not_null_free()) {
1270     // Erase not null free property for alias analysis.
1271     tj = ta = ta->cast_to_not_null_free(false);
1272   }
1273 
1274   if( ta && is_known_inst ) {
1275     if ( offset != Type::OffsetBot &&
1276          offset > arrayOopDesc::length_offset_in_bytes() ) {
1277       offset = Type::OffsetBot; // Flatten constant access into array body only
1278       tj = ta = TypeAryPtr::make(ptr, ta->ary(), ta->klass(), true, Type::Offset(offset), ta->field_offset(), ta->instance_id());
1279     }
1280   } else if( ta && _AliasLevel >= 2 ) {
1281     // For arrays indexed by constant indices, we flatten the alias
1282     // space to include all of the array body.  Only the header, klass
1283     // and array length can be accessed un-aliased.
1284     // For flattened value type array, each field has its own slice so
1285     // we must include the field offset.
1286     if( offset != Type::OffsetBot ) {
1287       if( ta->const_oop() ) { // MethodData* or Method*
1288         offset = Type::OffsetBot;   // Flatten constant access into array body
1289         tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1290       } else if( offset == arrayOopDesc::length_offset_in_bytes() ) {
1291         // range is OK as-is.
1292         tj = ta = TypeAryPtr::RANGE;
1293       } else if( offset == oopDesc::klass_offset_in_bytes() ) {
1294         tj = TypeInstPtr::KLASS; // all klass loads look alike
1295         ta = TypeAryPtr::RANGE; // generic ignored junk
1296         ptr = TypePtr::BotPTR;
1297       } else if( offset == oopDesc::mark_offset_in_bytes() ) {
1298         tj = TypeInstPtr::MARK;
1299         ta = TypeAryPtr::RANGE; // generic ignored junk
1300         ptr = TypePtr::BotPTR;
1301       } else {                  // Random constant offset into array body
1302         offset = Type::OffsetBot;   // Flatten constant access into array body
1303         tj = ta = TypeAryPtr::make(ptr,ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1304       }
1305     }
1306     // Arrays of fixed size alias with arrays of unknown size.
1307     if (ta->size() != TypeInt::POS) {
1308       const TypeAry *tary = TypeAry::make(ta->elem(), TypeInt::POS);
1309       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,ta->klass(),false,Type::Offset(offset), ta->field_offset());
1310     }
1311     // Arrays of known objects become arrays of unknown objects.
1312     if (ta->elem()->isa_narrowoop() && ta->elem() != TypeNarrowOop::BOTTOM) {
1313       const TypeAry *tary = TypeAry::make(TypeNarrowOop::BOTTOM, ta->size());
1314       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), ta->field_offset());
1315     }
1316     if (ta->elem()->isa_oopptr() && ta->elem() != TypeInstPtr::BOTTOM) {
1317       const TypeAry *tary = TypeAry::make(TypeInstPtr::BOTTOM, ta->size());
1318       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), ta->field_offset());
1319     }
1320     // Initially all flattened array accesses share a single slice
1321     if (ta->elem()->isa_valuetype() && ta->elem() != TypeValueType::BOTTOM && _flattened_accesses_share_alias) {
1322       const TypeAry *tary = TypeAry::make(TypeValueType::BOTTOM, ta->size());
1323       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), Type::Offset(Type::OffsetBot));
1324     }
1325     // Arrays of bytes and of booleans both use 'bastore' and 'baload' so
1326     // cannot be distinguished by bytecode alone.
1327     if (ta->elem() == TypeInt::BOOL) {
1328       const TypeAry *tary = TypeAry::make(TypeInt::BYTE, ta->size());
1329       ciKlass* aklass = ciTypeArrayKlass::make(T_BYTE);
1330       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,aklass,false,Type::Offset(offset), ta->field_offset());
1331     }
1332     // During the 2nd round of IterGVN, NotNull castings are removed.
1333     // Make sure the Bottom and NotNull variants alias the same.
1334     // Also, make sure exact and non-exact variants alias the same.
1335     if (ptr == TypePtr::NotNull || ta->klass_is_exact() || ta->speculative() != NULL) {
1336       tj = ta = TypeAryPtr::make(TypePtr::BotPTR,ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1337     }
1338   }
1339 
1340   // Oop pointers need some flattening
1341   const TypeInstPtr *to = tj->isa_instptr();
1342   if( to && _AliasLevel >= 2 && to != TypeOopPtr::BOTTOM ) {
1343     ciInstanceKlass *k = to->klass()->as_instance_klass();
1344     if( ptr == TypePtr::Constant ) {
1345       if (to->klass() != ciEnv::current()->Class_klass() ||
1346           offset < k->size_helper() * wordSize) {
1347         // No constant oop pointers (such as Strings); they alias with
1348         // unknown strings.
1349         assert(!is_known_inst, "not scalarizable allocation");
1350         tj = to = TypeInstPtr::make(TypePtr::BotPTR,to->klass(),false,0,Type::Offset(offset), to->klass()->flatten_array());
1351       }
1352     } else if( is_known_inst ) {
1353       tj = to; // Keep NotNull and klass_is_exact for instance type
1354     } else if( ptr == TypePtr::NotNull || to->klass_is_exact() ) {
1355       // During the 2nd round of IterGVN, NotNull castings are removed.
1356       // Make sure the Bottom and NotNull variants alias the same.
1357       // Also, make sure exact and non-exact variants alias the same.
1358       tj = to = TypeInstPtr::make(TypePtr::BotPTR,to->klass(),false,0,Type::Offset(offset), to->klass()->flatten_array());
1359     }
1360     if (to->speculative() != NULL) {
1361       tj = to = TypeInstPtr::make(to->ptr(),to->klass(),to->klass_is_exact(),to->const_oop(),Type::Offset(to->offset()), to->klass()->flatten_array(), to->instance_id());
1362     }
1363     // Canonicalize the holder of this field
1364     if (offset >= 0 && offset < instanceOopDesc::base_offset_in_bytes()) {
1365       // First handle header references such as a LoadKlassNode, even if the
1366       // object's klass is unloaded at compile time (4965979).
1367       if (!is_known_inst) { // Do it only for non-instance types
1368         tj = to = TypeInstPtr::make(TypePtr::BotPTR, env()->Object_klass(), false, NULL, Type::Offset(offset), false);
1369       }
1370     } else if (offset < 0 || offset >= k->size_helper() * wordSize) {
1371       // Static fields are in the space above the normal instance
1372       // fields in the java.lang.Class instance.
1373       if (to->klass() != ciEnv::current()->Class_klass()) {
1374         to = NULL;
1375         tj = TypeOopPtr::BOTTOM;
1376         offset = tj->offset();
1377       }
1378     } else {
1379       ciInstanceKlass *canonical_holder = k->get_canonical_holder(offset);
1380       if (!k->equals(canonical_holder) || tj->offset() != offset) {
1381         if( is_known_inst ) {
1382           tj = to = TypeInstPtr::make(to->ptr(), canonical_holder, true, NULL, Type::Offset(offset), canonical_holder->flatten_array(), to->instance_id());
1383         } else {
1384           tj = to = TypeInstPtr::make(to->ptr(), canonical_holder, false, NULL, Type::Offset(offset), canonical_holder->flatten_array());
1385         }
1386       }
1387     }
1388   }
1389 
1390   // Klass pointers to object array klasses need some flattening
1391   const TypeKlassPtr *tk = tj->isa_klassptr();
1392   if( tk ) {
1393     // If we are referencing a field within a Klass, we need
1394     // to assume the worst case of an Object.  Both exact and
1395     // inexact types must flatten to the same alias class so
1396     // use NotNull as the PTR.
1397     if ( offset == Type::OffsetBot || (offset >= 0 && (size_t)offset < sizeof(Klass)) ) {
1398 
1399       tj = tk = TypeKlassPtr::make(TypePtr::NotNull,
1400                                    TypeKlassPtr::OBJECT->klass(),
1401                                    Type::Offset(offset),
1402                                    false);
1403     }
1404 
1405     ciKlass* klass = tk->klass();
1406     if (klass != NULL && klass->is_obj_array_klass()) {
1407       ciKlass* k = TypeAryPtr::OOPS->klass();
1408       if( !k || !k->is_loaded() )                  // Only fails for some -Xcomp runs
1409         k = TypeInstPtr::BOTTOM->klass();
1410       tj = tk = TypeKlassPtr::make(TypePtr::NotNull, k, Type::Offset(offset), false);
1411     }
1412 
1413     // Check for precise loads from the primary supertype array and force them
1414     // to the supertype cache alias index.  Check for generic array loads from
1415     // the primary supertype array and also force them to the supertype cache
1416     // alias index.  Since the same load can reach both, we need to merge
1417     // these 2 disparate memories into the same alias class.  Since the
1418     // primary supertype array is read-only, there's no chance of confusion
1419     // where we bypass an array load and an array store.
1420     int primary_supers_offset = in_bytes(Klass::primary_supers_offset());
1421     if (offset == Type::OffsetBot ||
1422         (offset >= primary_supers_offset &&
1423          offset < (int)(primary_supers_offset + Klass::primary_super_limit() * wordSize)) ||
1424         offset == (int)in_bytes(Klass::secondary_super_cache_offset())) {
1425       offset = in_bytes(Klass::secondary_super_cache_offset());
1426       tj = tk = TypeKlassPtr::make(TypePtr::NotNull, tk->klass(), Type::Offset(offset), tk->flat_array());
1427     }
1428   }
1429 
1430   // Flatten all Raw pointers together.
1431   if (tj->base() == Type::RawPtr)
1432     tj = TypeRawPtr::BOTTOM;
1433 
1434   if (tj->base() == Type::AnyPtr)
1435     tj = TypePtr::BOTTOM;      // An error, which the caller must check for.
1436 
1437   // Flatten all to bottom for now
1438   switch( _AliasLevel ) {
1439   case 0:
1440     tj = TypePtr::BOTTOM;
1441     break;
1442   case 1:                       // Flatten to: oop, static, field or array
1443     switch (tj->base()) {
1444     //case Type::AryPtr: tj = TypeAryPtr::RANGE;    break;
1445     case Type::RawPtr:   tj = TypeRawPtr::BOTTOM;   break;
1446     case Type::AryPtr:   // do not distinguish arrays at all
1447     case Type::InstPtr:  tj = TypeInstPtr::BOTTOM;  break;
1448     case Type::KlassPtr: tj = TypeKlassPtr::OBJECT; break;
1449     case Type::AnyPtr:   tj = TypePtr::BOTTOM;      break;  // caller checks it
1450     default: ShouldNotReachHere();
1451     }
1452     break;
1453   case 2:                       // No collapsing at level 2; keep all splits
1454   case 3:                       // No collapsing at level 3; keep all splits
1455     break;
1456   default:
1457     Unimplemented();
1458   }
1459 
1460   offset = tj->offset();
1461   assert( offset != Type::OffsetTop, "Offset has fallen from constant" );
1462 
1463   assert( (offset != Type::OffsetBot && tj->base() != Type::AryPtr) ||
1464           (offset == Type::OffsetBot && tj->base() == Type::AryPtr) ||
1465           (offset == Type::OffsetBot && tj == TypeOopPtr::BOTTOM) ||
1466           (offset == Type::OffsetBot && tj == TypePtr::BOTTOM) ||
1467           (offset == oopDesc::mark_offset_in_bytes() && tj->base() == Type::AryPtr) ||
1468           (offset == oopDesc::klass_offset_in_bytes() && tj->base() == Type::AryPtr) ||
1469           (offset == arrayOopDesc::length_offset_in_bytes() && tj->base() == Type::AryPtr),
1470           "For oops, klasses, raw offset must be constant; for arrays the offset is never known" );
1471   assert( tj->ptr() != TypePtr::TopPTR &&
1472           tj->ptr() != TypePtr::AnyNull &&
1473           tj->ptr() != TypePtr::Null, "No imprecise addresses" );
1474 //    assert( tj->ptr() != TypePtr::Constant ||
1475 //            tj->base() == Type::RawPtr ||
1476 //            tj->base() == Type::KlassPtr, "No constant oop addresses" );
1477 
1478   return tj;
1479 }

1564 //--------------------------------find_alias_type------------------------------
1565 Compile::AliasType* Compile::find_alias_type(const TypePtr* adr_type, bool no_create, ciField* original_field, bool uncached) {
1566   if (_AliasLevel == 0)
1567     return alias_type(AliasIdxBot);
1568 
1569   AliasCacheEntry* ace = NULL;
1570   if (!uncached) {
1571     ace = probe_alias_cache(adr_type);
1572     if (ace->_adr_type == adr_type) {
1573       return alias_type(ace->_index);
1574     }
1575   }
1576 
1577   // Handle special cases.
1578   if (adr_type == NULL)             return alias_type(AliasIdxTop);
1579   if (adr_type == TypePtr::BOTTOM)  return alias_type(AliasIdxBot);
1580 
1581   // Do it the slow way.
1582   const TypePtr* flat = flatten_alias_type(adr_type);
1583 
1584 #ifdef ASSERT
1585   {
1586     ResourceMark rm;
1587     assert(flat == flatten_alias_type(flat), "not idempotent: adr_type = %s; flat = %s => %s",
1588            Type::str(adr_type), Type::str(flat), Type::str(flatten_alias_type(flat)));
1589     assert(flat != TypePtr::BOTTOM, "cannot alias-analyze an untyped ptr: adr_type = %s",
1590            Type::str(adr_type));
1591     if (flat->isa_oopptr() && !flat->isa_klassptr()) {
1592       const TypeOopPtr* foop = flat->is_oopptr();
1593       // Scalarizable allocations have exact klass always.
1594       bool exact = !foop->klass_is_exact() || foop->is_known_instance();
1595       const TypePtr* xoop = foop->cast_to_exactness(exact)->is_ptr();
1596       assert(foop == flatten_alias_type(xoop), "exactness must not affect alias type: foop = %s; xoop = %s",
1597              Type::str(foop), Type::str(xoop));
1598     }
1599   }
1600 #endif
1601 
1602   int idx = AliasIdxTop;
1603   for (int i = 0; i < num_alias_types(); i++) {
1604     if (alias_type(i)->adr_type() == flat) {
1605       idx = i;
1606       break;
1607     }
1608   }
1609 
1610   if (idx == AliasIdxTop) {
1611     if (no_create)  return NULL;
1612     // Grow the array if necessary.
1613     if (_num_alias_types == _max_alias_types)  grow_alias_types();
1614     // Add a new alias type.
1615     idx = _num_alias_types++;
1616     _alias_types[idx]->Init(idx, flat);
1617     if (flat == TypeInstPtr::KLASS)  alias_type(idx)->set_rewritable(false);
1618     if (flat == TypeAryPtr::RANGE)   alias_type(idx)->set_rewritable(false);
1619     if (flat->isa_instptr()) {
1620       if (flat->offset() == java_lang_Class::klass_offset()
1621           && flat->is_instptr()->klass() == env()->Class_klass())
1622         alias_type(idx)->set_rewritable(false);
1623     }
1624     ciField* field = NULL;
1625     if (flat->isa_aryptr()) {
1626 #ifdef ASSERT
1627       const int header_size_min  = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1628       // (T_BYTE has the weakest alignment and size restrictions...)
1629       assert(flat->offset() < header_size_min, "array body reference must be OffsetBot");
1630 #endif
1631       const Type* elemtype = flat->is_aryptr()->elem();
1632       if (flat->offset() == TypePtr::OffsetBot) {
1633         alias_type(idx)->set_element(elemtype);
1634       }
1635       int field_offset = flat->is_aryptr()->field_offset().get();
1636       if (elemtype->isa_valuetype() &&
1637           elemtype->value_klass() != NULL &&
1638           field_offset != Type::OffsetBot) {
1639         ciValueKlass* vk = elemtype->value_klass();
1640         field_offset += vk->first_field_offset();
1641         field = vk->get_field_by_offset(field_offset, false);
1642       }
1643     }
1644     if (flat->isa_klassptr()) {
1645       if (flat->offset() == in_bytes(Klass::super_check_offset_offset()))
1646         alias_type(idx)->set_rewritable(false);
1647       if (flat->offset() == in_bytes(Klass::modifier_flags_offset()))
1648         alias_type(idx)->set_rewritable(false);
1649       if (flat->offset() == in_bytes(Klass::access_flags_offset()))
1650         alias_type(idx)->set_rewritable(false);
1651       if (flat->offset() == in_bytes(Klass::java_mirror_offset()))
1652         alias_type(idx)->set_rewritable(false);
1653       if (flat->offset() == in_bytes(Klass::layout_helper_offset()))
1654         alias_type(idx)->set_rewritable(false);
1655       if (flat->offset() == in_bytes(Klass::secondary_super_cache_offset()))
1656         alias_type(idx)->set_rewritable(false);
1657     }
1658     // %%% (We would like to finalize JavaThread::threadObj_offset(),
1659     // but the base pointer type is not distinctive enough to identify
1660     // references into JavaThread.)
1661 
1662     // Check for final fields.
1663     const TypeInstPtr* tinst = flat->isa_instptr();
1664     if (tinst && tinst->offset() >= instanceOopDesc::base_offset_in_bytes()) {
1665       if (tinst->const_oop() != NULL &&
1666           tinst->klass() == ciEnv::current()->Class_klass() &&
1667           tinst->offset() >= (tinst->klass()->as_instance_klass()->size_helper() * wordSize)) {
1668         // static field
1669         ciInstanceKlass* k = tinst->const_oop()->as_instance()->java_lang_Class_klass()->as_instance_klass();
1670         field = k->get_field_by_offset(tinst->offset(), true);
1671       } else if (tinst->klass()->is_valuetype()) {
1672         // Value type field
1673         ciValueKlass* vk = tinst->value_klass();
1674         field = vk->get_field_by_offset(tinst->offset(), false);
1675       } else {
1676         ciInstanceKlass* k = tinst->klass()->as_instance_klass();
1677         field = k->get_field_by_offset(tinst->offset(), false);
1678       }
1679     }
1680     assert(field == NULL ||
1681            original_field == NULL ||
1682            (field->holder() == original_field->holder() &&
1683             field->offset() == original_field->offset() &&
1684             field->is_static() == original_field->is_static()), "wrong field?");
1685     // Set field() and is_rewritable() attributes.
1686     if (field != NULL) {
1687       alias_type(idx)->set_field(field);
1688       if (flat->isa_aryptr()) {
1689         // Fields of flattened inline type arrays are rewritable although they are declared final
1690         assert(flat->is_aryptr()->elem()->isa_valuetype(), "must be a flattened value array");
1691         alias_type(idx)->set_rewritable(true);
1692       }
1693     }
1694   }
1695 
1696   // Fill the cache for next time.
1697   if (!uncached) {
1698     ace->_adr_type = adr_type;
1699     ace->_index    = idx;
1700     assert(alias_type(adr_type) == alias_type(idx),  "type must be installed");
1701 
1702     // Might as well try to fill the cache for the flattened version, too.
1703     AliasCacheEntry* face = probe_alias_cache(flat);
1704     if (face->_adr_type == NULL) {
1705       face->_adr_type = flat;
1706       face->_index    = idx;
1707       assert(alias_type(flat) == alias_type(idx), "flat type must work too");
1708     }
1709   }
1710 
1711   return alias_type(idx);
1712 }

1741 //-----------------------------must_alias--------------------------------------
1742 // True if all values of the given address type are in the given alias category.
1743 bool Compile::must_alias(const TypePtr* adr_type, int alias_idx) {
1744   if (alias_idx == AliasIdxBot)         return true;  // the universal category
1745   if (adr_type == NULL)                 return true;  // NULL serves as TypePtr::TOP
1746   if (alias_idx == AliasIdxTop)         return false; // the empty category
1747   if (adr_type->base() == Type::AnyPtr) return false; // TypePtr::BOTTOM or its twins
1748 
1749   // the only remaining possible overlap is identity
1750   int adr_idx = get_alias_index(adr_type);
1751   assert(adr_idx != AliasIdxBot && adr_idx != AliasIdxTop, "");
1752   assert(adr_idx == alias_idx ||
1753          (alias_type(alias_idx)->adr_type() != TypeOopPtr::BOTTOM
1754           && adr_type                       != TypeOopPtr::BOTTOM),
1755          "should not be testing for overlap with an unsafe pointer");
1756   return adr_idx == alias_idx;
1757 }

1759 //------------------------------can_alias--------------------------------------
1760 // True if any values of the given address type are in the given alias category.
1761 bool Compile::can_alias(const TypePtr* adr_type, int alias_idx) {
1762   if (alias_idx == AliasIdxTop)         return false; // the empty category
1763   if (adr_type == NULL)                 return false; // NULL serves as TypePtr::TOP
1764   // Known instance doesn't alias with bottom memory
1765   if (alias_idx == AliasIdxBot)         return !adr_type->is_known_instance();                   // the universal category
1766   if (adr_type->base() == Type::AnyPtr) return !C->get_adr_type(alias_idx)->is_known_instance(); // TypePtr::BOTTOM or its twins
1767 
1768   // the only remaining possible overlap is identity
1769   int adr_idx = get_alias_index(adr_type);
1770   assert(adr_idx != AliasIdxBot && adr_idx != AliasIdxTop, "");
1771   return adr_idx == alias_idx;
1772 }

1783 //----------------------------Inline_Warm--------------------------------------
1784 int Compile::Inline_Warm() {
1785   // If there is room, try to inline some more warm call sites.
1786   // %%% Do a graph index compaction pass when we think we're out of space?
1787   if (!InlineWarmCalls)  return 0;
1788 
1789   int calls_made_hot = 0;
1790   int room_to_grow   = NodeCountInliningCutoff - unique();
1791   int amount_to_grow = MIN2(room_to_grow, (int)NodeCountInliningStep);
1792   int amount_grown   = 0;
1793   WarmCallInfo* call;
1794   while (amount_to_grow > 0 && (call = pop_warm_call()) != NULL) {
1795     int est_size = (int)call->size();
1796     if (est_size > (room_to_grow - amount_grown)) {
1797       // This one won't fit anyway.  Get rid of it.
1798       call->make_cold();
1799       continue;
1800     }
1801     call->make_hot();
1802     calls_made_hot++;
1803     amount_grown   += est_size;
1804     amount_to_grow -= est_size;
1805   }
1806 
1807   if (calls_made_hot > 0)  set_major_progress();
1808   return calls_made_hot;
1809 }

1812 //----------------------------Finish_Warm--------------------------------------
1813 void Compile::Finish_Warm() {
1814   if (!InlineWarmCalls)  return;
1815   if (failing())  return;
1816   if (warm_calls() == NULL)  return;
1817 
1818   // Clean up loose ends, if we are out of space for inlining.
1819   WarmCallInfo* call;
1820   while ((call = pop_warm_call()) != NULL) {
1821     call->make_cold();
1822   }
1823 }

1944 void Compile::adjust_flattened_array_access_aliases(PhaseIterGVN& igvn) {
1945   if (!_has_flattened_accesses) {
1946     return;
1947   }
1948   // Initially, all flattened array accesses share the same slice to
1949   // keep dependencies with Object[] array accesses (that could be
1950   // to a flattened array) correct. We're done with parsing so we
1951   // now know all flattened array accesses in this compile
1952   // unit. Let's move flattened array accesses to their own slice,
1953   // one per element field. This should help memory access
1954   // optimizations.
1955   ResourceMark rm;
1956   Unique_Node_List wq;
1957   wq.push(root());
1958 
1959   Node_List mergememnodes;
1960   Node_List memnodes;
1961 
1962   // Alias index currently shared by all flattened memory accesses
1963   int index = get_alias_index(TypeAryPtr::VALUES);
1964 
1965   // Find MergeMem nodes and flattened array accesses
1966   for (uint i = 0; i < wq.size(); i++) {
1967     Node* n = wq.at(i);
1968     if (n->is_Mem()) {
1969       const TypePtr* adr_type = NULL;
1970       if (n->Opcode() == Op_StoreCM) {
1971         adr_type = get_adr_type(get_alias_index(n->in(MemNode::OopStore)->adr_type()));
1972       } else {
1973         adr_type = get_adr_type(get_alias_index(n->adr_type()));
1974       }
1975       if (adr_type == TypeAryPtr::VALUES) {
1976         memnodes.push(n);
1977       }
1978     } else if (n->is_MergeMem()) {
1979       MergeMemNode* mm = n->as_MergeMem();
1980       if (mm->memory_at(index) != mm->base_memory()) {
1981         mergememnodes.push(n);
1982       }
1983     }
1984     for (uint j = 0; j < n->req(); j++) {
1985       Node* m = n->in(j);
1986       if (m != NULL) {
1987         wq.push(m);
1988       }
1989     }
1990   }
1991 
1992   if (memnodes.size() > 0) {
1993     _flattened_accesses_share_alias = false;
1994 
1995     // We are going to change the slice for the flattened array
1996     // accesses so we need to clear the cache entries that refer to
1997     // them.
1998     for (uint i = 0; i < AliasCacheSize; i++) {
1999       AliasCacheEntry* ace = &_alias_cache[i];
2000       if (ace->_adr_type != NULL &&
2001           ace->_adr_type->isa_aryptr() &&
2002           ace->_adr_type->is_aryptr()->elem()->isa_valuetype()) {
2003         ace->_adr_type = NULL;
2004         ace->_index = (i != 0) ? 0 : AliasIdxTop; // Make sure the NULL adr_type resolves to AliasIdxTop
2005       }
2006     }
2007 
2008     // Find what aliases we are going to add
2009     int start_alias = num_alias_types()-1;
2010     int stop_alias = 0;
2011 
2012     for (uint i = 0; i < memnodes.size(); i++) {
2013       Node* m = memnodes.at(i);
2014       const TypePtr* adr_type = NULL;
2015       if (m->Opcode() == Op_StoreCM) {
2016         adr_type = m->in(MemNode::OopStore)->adr_type();
2017         Node* clone = new StoreCMNode(m->in(MemNode::Control), m->in(MemNode::Memory), m->in(MemNode::Address),
2018                                       m->adr_type(), m->in(MemNode::ValueIn), m->in(MemNode::OopStore),
2019                                       get_alias_index(adr_type));
2020         igvn.register_new_node_with_optimizer(clone);
2021         igvn.replace_node(m, clone);
2022       } else {
2023         adr_type = m->adr_type();
2024 #ifdef ASSERT
2025         m->as_Mem()->set_adr_type(adr_type);
2026 #endif
2027       }
2028       int idx = get_alias_index(adr_type);
2029       start_alias = MIN2(start_alias, idx);
2030       stop_alias = MAX2(stop_alias, idx);
2031     }
2032 
2033     assert(stop_alias >= start_alias, "should have expanded aliases");
2034 
2035     Node_Stack stack(0);
2036 #ifdef ASSERT
2037     VectorSet seen(Thread::current()->resource_area());
2038 #endif
2039     // Now let's fix the memory graph so each flattened array access
2040     // is moved to the right slice. Start from the MergeMem nodes.
2041     uint last = unique();
2042     for (uint i = 0; i < mergememnodes.size(); i++) {
2043       MergeMemNode* current = mergememnodes.at(i)->as_MergeMem();
2044       Node* n = current->memory_at(index);
2045       MergeMemNode* mm = NULL;
2046       do {
2047         // Follow memory edges through memory accesses, phis and
2048         // narrow membars and push nodes on the stack. Once we hit
2049         // bottom memory, we pop element off the stack one at a
2050         // time, in reverse order, and move them to the right slice
2051         // by changing their memory edges.
2052         if ((n->is_Phi() && n->adr_type() != TypePtr::BOTTOM) || n->is_Mem() || n->adr_type() == TypeAryPtr::VALUES) {
2053           assert(!seen.test_set(n->_idx), "");
2054           // Uses (a load for instance) will need to be moved to the
2055           // right slice as well and will get a new memory state
2056           // that we don't know yet. The use could also be the
2057           // backedge of a loop. We put a place holder node between
2058           // the memory node and its uses. We replace that place
2059           // holder with the correct memory state once we know it,
2060           // i.e. when nodes are popped off the stack. Using the
2061           // place holder make the logic work in the presence of
2062           // loops.
2063           if (n->outcnt() > 1) {
2064             Node* place_holder = NULL;
2065             assert(!n->has_out_with(Op_Node), "");
2066             for (DUIterator k = n->outs(); n->has_out(k); k++) {
2067               Node* u = n->out(k);
2068               if (u != current && u->_idx < last) {
2069                 bool success = false;
2070                 for (uint l = 0; l < u->req(); l++) {
2071                   if (!stack.is_empty() && u == stack.node() && l == stack.index()) {
2072                     continue;
2073                   }
2074                   Node* in = u->in(l);
2075                   if (in == n) {
2076                     if (place_holder == NULL) {
2077                       place_holder = new Node(1);
2078                       place_holder->init_req(0, n);
2079                     }
2080                     igvn.replace_input_of(u, l, place_holder);
2081                     success = true;
2082                   }
2083                 }
2084                 if (success) {
2085                   --k;
2086                 }
2087               }
2088             }
2089           }
2090           if (n->is_Phi()) {
2091             stack.push(n, 1);
2092             n = n->in(1);
2093           } else if (n->is_Mem()) {
2094             stack.push(n, n->req());
2095             n = n->in(MemNode::Memory);
2096           } else {
2097             assert(n->is_Proj() && n->in(0)->Opcode() == Op_MemBarCPUOrder, "");
2098             stack.push(n, n->req());
2099             n = n->in(0)->in(TypeFunc::Memory);
2100           }
2101         } else {
2102           assert(n->adr_type() == TypePtr::BOTTOM || (n->Opcode() == Op_Node && n->_idx >= last) || (n->is_Proj() && n->in(0)->is_Initialize()), "");
2103           // Build a new MergeMem node to carry the new memory state
2104           // as we build it. IGVN should fold extraneous MergeMem
2105           // nodes.
2106           mm = MergeMemNode::make(n);
2107           igvn.register_new_node_with_optimizer(mm);
2108           while (stack.size() > 0) {
2109             Node* m = stack.node();
2110             uint idx = stack.index();
2111             if (m->is_Mem()) {
2112               // Move memory node to its new slice
2113               const TypePtr* adr_type = m->adr_type();
2114               int alias = get_alias_index(adr_type);
2115               Node* prev = mm->memory_at(alias);
2116               igvn.replace_input_of(m, MemNode::Memory, prev);
2117               mm->set_memory_at(alias, m);
2118             } else if (m->is_Phi()) {
2119               // We need as many new phis as there are new aliases
2120               igvn.replace_input_of(m, idx, mm);
2121               if (idx == m->req()-1) {
2122                 Node* r = m->in(0);
2123                 for (uint j = (uint)start_alias; j <= (uint)stop_alias; j++) {
2124                   const Type* adr_type = get_adr_type(j);
2125                   if (!adr_type->isa_aryptr() || !adr_type->is_aryptr()->elem()->isa_valuetype()) {
2126                     continue;
2127                   }
2128                   Node* phi = new PhiNode(r, Type::MEMORY, get_adr_type(j));
2129                   igvn.register_new_node_with_optimizer(phi);
2130                   for (uint k = 1; k < m->req(); k++) {
2131                     phi->init_req(k, m->in(k)->as_MergeMem()->memory_at(j));
2132                   }
2133                   mm->set_memory_at(j, phi);
2134                 }
2135                 Node* base_phi = new PhiNode(r, Type::MEMORY, TypePtr::BOTTOM);
2136                 igvn.register_new_node_with_optimizer(base_phi);
2137                 for (uint k = 1; k < m->req(); k++) {
2138                   base_phi->init_req(k, m->in(k)->as_MergeMem()->base_memory());
2139                 }
2140                 mm->set_base_memory(base_phi);
2141               }
2142             } else {
2143               // This is a MemBarCPUOrder node from
2144               // Parse::array_load()/Parse::array_store(), in the
2145               // branch that handles flattened arrays hidden under
2146               // an Object[] array. We also need one new membar per
2147               // new alias to keep the unknown access that the
2148               // membars protect properly ordered with accesses to
2149               // known flattened array.
2150               assert(m->is_Proj(), "projection expected");
2151               Node* ctrl = m->in(0)->in(TypeFunc::Control);
2152               igvn.replace_input_of(m->in(0), TypeFunc::Control, top());
2153               for (uint j = (uint)start_alias; j <= (uint)stop_alias; j++) {

2702   uint cnt = 0;

2714     } else {
2715       uint last_req = n->req();
2716       if (is_vector_ternary_bitwise_op(n)) {

2722           inputs.push(def);
2723           cnt++;
2724         }
2725       }

2728   } else { // not a bitwise operations
2729     if (!inputs.member(n)) {
2730       inputs.push(n);

2732     }

2769 //
2770 //  A macro logic node represents a truth table. It has 4 inputs,
2771 //  First three inputs corresponds to 3 columns of a truth table
2772 //  and fourth input captures the logic function.
2773 //
2774 //  eg.  fn = (in1 AND in2) OR in3;
2775 //
2776 //      MacroNode(in1,in2,in3,fn)
2777 //
2778 //  -----------------
2779 //  in1 in2 in3  fn
2780 //  -----------------
2781 //  0    0   0    0
2782 //  0    0   1    1
2783 //  0    1   0    0
2784 //  0    1   1    1
2785 //  1    0   0    0
2786 //  1    0   1    1
2787 //  1    1   0    1
2788 //  1    1   1    1
2789 //

2791 uint Compile::eval_macro_logic_op(uint func, uint in1 , uint in2, uint in3) {
2792   int res = 0;
2793   for (int i = 0; i < 8; i++) {
2794     int bit1 = extract_bit(in1, i);
2795     int bit2 = extract_bit(in2, i);
2796     int bit3 = extract_bit(in3, i);
2797 
2798     int func_bit_pos = (bit1 << 2 | bit2 << 1 | bit3);
2799     int func_bit = extract_bit(func, func_bit_pos);
2800 
2801     res |= func_bit << i;
2802   }
2803   return res;
2804 }

2828 uint Compile::compute_truth_table(Unique_Node_List& partition, Unique_Node_List& inputs) {
2829   assert(inputs.size() <= 3, "sanity");
2830   ResourceMark rm;
2831   uint res = 0;
2832   ResourceHashtable<Node*,uint> eval_map;
2833 
2834   // Populate precomputed functions for inputs.
2835   // Each input corresponds to one column of 3 input truth-table.
2836   uint input_funcs[] = { 0xAA,   // (_, _, a) -> a
2837                          0xCC,   // (_, b, _) -> b
2838                          0xF0 }; // (c, _, _) -> c
2839   for (uint i = 0; i < inputs.size(); i++) {
2840     eval_map.put(inputs.at(i), input_funcs[i]);
2841   }
2842 
2843   for (uint i = 0; i < partition.size(); i++) {
2844     Node* n = partition.at(i);
2845 
2846     uint func1 = 0, func2 = 0, func3 = 0;
2847     eval_operands(n, func1, func2, func3, eval_map);
2848 
2849     switch (n->Opcode()) {
2850       case Op_OrV:
2851         assert(func3 == 0, "not binary");
2852         res = func1 | func2;
2853         break;
2854       case Op_AndV:
2855         assert(func3 == 0, "not binary");
2856         res = func1 & func2;
2857         break;
2858       case Op_XorV:
2859         if (VectorNode::is_vector_bitwise_not_pattern(n)) {
2860           assert(func2 == 0 && func3 == 0, "not unary");
2861           res = (~func1) & 0xFF;
2862         } else {
2863           assert(func3 == 0, "not binary");
2864           res = func1 ^ func2;
2865         }
2866         break;
2867       case Op_MacroLogicV:
2868         // Ordering of inputs may change during evaluation of sub-tree
2869         // containing MacroLogic node as a child node, thus a re-evaluation
2870         // makes sure that function is evaluated in context of current
2871         // inputs.
2872         res = eval_macro_logic_op(n->in(4)->get_int(), func1, func2, func3);
2873         break;
2874 
2875       default: assert(false, "not supported: %s", n->Name());
2876     }
2877     assert(res <= 0xFF, "invalid");
2878     eval_map.put(n, res);
2879   }
2880   return res;
2881 }

2883 bool Compile::compute_logic_cone(Node* n, Unique_Node_List& partition, Unique_Node_List& inputs) {
2884   assert(partition.size() == 0, "not empty");
2885   assert(inputs.size() == 0, "not empty");
2886   if (is_vector_ternary_bitwise_op(n)) {
2887     return false;
2888   }
2889 
2890   bool is_unary_op = is_vector_unary_bitwise_op(n);
2891   if (is_unary_op) {
2892     assert(collect_unique_inputs(n, partition, inputs) == 1, "not unary");
2893     return false; // too few inputs
2894   }
2895 
2896   assert(is_vector_binary_bitwise_op(n), "not binary");
2897   Node* in1 = n->in(1);
2898   Node* in2 = n->in(2);
2899 
2900   int in1_unique_inputs_cnt = collect_unique_inputs(in1, partition, inputs);
2901   int in2_unique_inputs_cnt = collect_unique_inputs(in2, partition, inputs);
2902   partition.push(n);
2903 
2904   // Too many inputs?
2905   if (inputs.size() > 3) {
2906     partition.clear();
2907     inputs.clear();
2908     { // Recompute in2 inputs
2909       Unique_Node_List not_used;
2910       in2_unique_inputs_cnt = collect_unique_inputs(in2, not_used, not_used);
2911     }
2912     // Pick the node with minimum number of inputs.
2913     if (in1_unique_inputs_cnt >= 3 && in2_unique_inputs_cnt >= 3) {
2914       return false; // still too many inputs
2915     }
2916     // Recompute partition & inputs.
2917     Node* child       = (in1_unique_inputs_cnt < in2_unique_inputs_cnt ? in1 : in2);
2918     collect_unique_inputs(child, partition, inputs);
2919 
2920     Node* other_input = (in1_unique_inputs_cnt < in2_unique_inputs_cnt ? in2 : in1);
2921     inputs.push(other_input);
2922 
2923     partition.push(n);
2924   }
2925 
2926   return (partition.size() == 2 || partition.size() == 3) &&
2927          (inputs.size()    == 2 || inputs.size()    == 3);
2928 }

3125 // Eliminate trivially redundant StoreCMs and accumulate their
3126 // precedence edges.
3127 void Compile::eliminate_redundant_card_marks(Node* n) {
3128   assert(n->Opcode() == Op_StoreCM, "expected StoreCM");
3129   if (n->in(MemNode::Address)->outcnt() > 1) {
3130     // There are multiple users of the same address so it might be
3131     // possible to eliminate some of the StoreCMs
3132     Node* mem = n->in(MemNode::Memory);
3133     Node* adr = n->in(MemNode::Address);
3134     Node* val = n->in(MemNode::ValueIn);
3135     Node* prev = n;
3136     bool done = false;
3137     // Walk the chain of StoreCMs eliminating ones that match.  As
3138     // long as it's a chain of single users then the optimization is
3139     // safe.  Eliminating partially redundant StoreCMs would require
3140     // cloning copies down the other paths.
3141     while (mem->Opcode() == Op_StoreCM && mem->outcnt() == 1 && !done) {
3142       if (adr == mem->in(MemNode::Address) &&
3143           val == mem->in(MemNode::ValueIn)) {
3144         // redundant StoreCM
3145         if (mem->req() > MemNode::OopStore) {
3146           // Hasn't been processed by this code yet.
3147           n->add_prec(mem->in(MemNode::OopStore));
3148         } else {
3149           // Already converted to precedence edge
3150           for (uint i = mem->req(); i < mem->len(); i++) {
3151             // Accumulate any precedence edges
3152             if (mem->in(i) != NULL) {
3153               n->add_prec(mem->in(i));
3154             }
3155           }
3156           // Everything above this point has been processed.
3157           done = true;
3158         }
3159         // Eliminate the previous StoreCM
3160         prev->set_req(MemNode::Memory, mem->in(MemNode::Memory));
3161         assert(mem->outcnt() == 0, "should be dead");
3162         mem->disconnect_inputs(NULL, this);
3163       } else {
3164         prev = mem;
3165       }
3166       mem = prev->in(MemNode::Memory);
3167     }
3168   }

4007 //------------------------------final_graph_reshaping--------------------------
4008 // Final Graph Reshaping.
4009 //
4010 // (1) Clone simple inputs to uncommon calls, so they can be scheduled late
4011 //     and not commoned up and forced early.  Must come after regular
4012 //     optimizations to avoid GVN undoing the cloning.  Clone constant
4013 //     inputs to Loop Phis; these will be split by the allocator anyways.
4014 //     Remove Opaque nodes.
4015 // (2) Move last-uses by commutative operations to the left input to encourage
4016 //     Intel update-in-place two-address operations and better register usage
4017 //     on RISCs.  Must come after regular optimizations to avoid GVN Ideal
4018 //     calls canonicalizing them back.
4019 // (3) Count the number of double-precision FP ops, single-precision FP ops
4020 //     and call sites.  On Intel, we can get correct rounding either by
4021 //     forcing singles to memory (requires extra stores and loads after each
4022 //     FP bytecode) or we can set a rounding mode bit (requires setting and
4023 //     clearing the mode bit around call sites).  The mode bit is only used
4024 //     if the relative frequency of single FP ops to calls is low enough.
4025 //     This is a key transform for SPEC mpeg_audio.
4026 // (4) Detect infinite loops; blobs of code reachable from above but not
4027 //     below.  Several of the Code_Gen algorithms fail on such code shapes,
4028 //     so we simply bail out.  Happens a lot in ZKM.jar, but also happens
4029 //     from time to time in other codes (such as -Xcomp finalizer loops, etc).
4030 //     Detection is by looking for IfNodes where only 1 projection is
4031 //     reachable from below or CatchNodes missing some targets.
4032 // (5) Assert for insane oop offsets in debug mode.
4033 
4034 bool Compile::final_graph_reshaping() {
4035   // an infinite loop may have been eliminated by the optimizer,
4036   // in which case the graph will be empty.
4037   if (root()->req() == 1) {
4038     record_method_not_compilable("trivial infinite loop");
4039     return true;
4040   }
4041 
4042   // Expensive nodes have their control input set to prevent the GVN
4043   // from freely commoning them. There's no GVN beyond this point so
4044   // no need to keep the control input. We want the expensive nodes to
4045   // be freely moved to the least frequent code path by gcm.
4046   assert(OptimizeExpensiveOps || expensive_count() == 0, "optimization off but list non empty?");
4047   for (int i = 0; i < expensive_count(); i++) {
4048     _expensive_nodes->at(i)->set_req(0, NULL);
4049   }
4050 
4051   Final_Reshape_Counts frc;
4052 
4053   // Visit everybody reachable!
4054   // Allocate stack of size C->live_nodes()/2 to avoid frequent realloc
4055   Node_Stack nstack(live_nodes() >> 1);
4056   final_graph_reshaping_walk(nstack, root(), frc);
4057 
4058   // Check for unreachable (from below) code (i.e., infinite loops).
4059   for( uint i = 0; i < frc._tests.size(); i++ ) {
4060     MultiBranchNode *n = frc._tests[i]->as_MultiBranch();
4061     // Get number of CFG targets.
4062     // Note that PCTables include exception targets after calls.
4063     uint required_outcnt = n->required_outcnt();
4064     if (n->outcnt() != required_outcnt) {
4065       // Check for a few special cases.  Rethrow Nodes never take the
4066       // 'fall-thru' path, so expected kids is 1 less.
4067       if (n->is_PCTable() && n->in(0) && n->in(0)->in(0)) {
4068         if (n->in(0)->in(0)->is_Call()) {
4069           CallNode *call = n->in(0)->in(0)->as_Call();
4070           if (call->entry_point() == OptoRuntime::rethrow_stub()) {
4071             required_outcnt--;      // Rethrow always has 1 less kid
4072           } else if (call->req() > TypeFunc::Parms &&
4073                      call->is_CallDynamicJava()) {
4074             // Check for null receiver. In such case, the optimizer has
4075             // detected that the virtual call will always result in a null
4076             // pointer exception. The fall-through projection of this CatchNode
4077             // will not be populated.
4078             Node *arg0 = call->in(TypeFunc::Parms);
4079             if (arg0->is_Type() &&
4080                 arg0->as_Type()->type()->higher_equal(TypePtr::NULL_PTR)) {
4081               required_outcnt--;
4082             }
4083           } else if (call->entry_point() == OptoRuntime::new_array_Java() &&
4084                      call->req() > TypeFunc::Parms+1 &&
4085                      call->is_CallStaticJava()) {
4086             // Check for negative array length. In such case, the optimizer has
4087             // detected that the allocation attempt will always result in an
4088             // exception. There is no fall-through projection of this CatchNode .
4089             Node *arg1 = call->in(TypeFunc::Parms+1);
4090             if (arg1->is_Type() &&
4091                 arg1->as_Type()->type()->join(TypeInt::POS)->empty()) {
4092               required_outcnt--;
4093             }
4094           }
4095         }
4096       }
4097       // Recheck with a better notion of 'required_outcnt'
4098       if (n->outcnt() != required_outcnt) {
4099         record_method_not_compilable("malformed control flow");
4100         return true;            // Not all targets reachable!
4101       }
4102     }
4103     // Check that I actually visited all kids.  Unreached kids
4104     // must be infinite loops.
4105     for (DUIterator_Fast jmax, j = n->fast_outs(jmax); j < jmax; j++)
4106       if (!frc._visited.test(n->fast_out(j)->_idx)) {
4107         record_method_not_compilable("infinite loop");
4108         return true;            // Found unvisited kid; must be unreach
4109       }
4110 
4111     // Here so verification code in final_graph_reshaping_walk()
4112     // always see an OuterStripMinedLoopEnd
4113     if (n->is_OuterStripMinedLoopEnd()) {
4114       IfNode* init_iff = n->as_If();
4115       Node* iff = new IfNode(init_iff->in(0), init_iff->in(1), init_iff->_prob, init_iff->_fcnt);
4116       n->subsume_by(iff, this);
4117     }
4118   }
4119 
4120 #ifdef IA32
4121   // If original bytecodes contained a mixture of floats and doubles
4122   // check if the optimizer has made it homogenous, item (3).
4123   if (UseSSE == 0 &&
4124       frc.get_float_count() > 32 &&
4125       frc.get_double_count() == 0 &&
4126       (10 * frc.get_call_count() < frc.get_float_count()) ) {
4127     set_24_bit_selection_and_mode(false, true);
4128   }
4129 #endif // IA32
4130 
4131   set_java_calls(frc.get_java_call_count());
4132   set_inner_loops(frc.get_inner_loop_count());
4133 
4134   // No infinite loops, no reason to bail out.
4135   return false;
4136 }

4200   uint bc_cutoff = (uint) PerBytecodeRecompilationCutoff / 8;

4208       && md->trap_recompiled_at(bci, m)

4212     // assume the worst.

4214       log()->elem("observe trap='%s recompiled' count='%d' recompiles2='%d'",

4223     if (log())

4226                   md->trap_count(reason), trap_count(reason),
4227                   md->decompile_count(), decompile_count());

4390 //----------------------------static_subtype_check-----------------------------
4391 // Shortcut important common cases when superklass is exact:
4392 // (0) superklass is java.lang.Object (can occur in reflective code)
4393 // (1) subklass is already limited to a subtype of superklass => always ok
4394 // (2) subklass does not overlap with superklass => always fail
4395 // (3) superklass has NO subtypes and we can check with a simple compare.
4396 int Compile::static_subtype_check(ciKlass* superk, ciKlass* subk) {
4397   if (StressReflectiveCode || superk == NULL || subk == NULL) {
4398     return SSC_full_test;       // Let caller generate the general case.
4399   }
4400 
4401   if (superk == env()->Object_klass()) {
4402     return SSC_always_true;     // (0) this test cannot fail
4403   }
4404 
4405   ciType* superelem = superk;
4406   if (superelem->is_array_klass()) {
4407     ciArrayKlass* ak = superelem->as_array_klass();
4408     superelem = superelem->as_array_klass()->base_element_type();
4409   }
4410 
4411   if (!subk->is_interface()) {  // cannot trust static interface types yet
4412     if (subk->is_subtype_of(superk)) {
4413       return SSC_always_true;   // (1) false path dead; no dynamic test needed
4414     }
4415     if (!(superelem->is_klass() && superelem->as_klass()->is_interface()) &&
4416         !superk->is_subtype_of(subk)) {
4417       return SSC_always_false;
4418     }
4419   }
4420 
4421   // If casting to an instance klass, it must have no subtypes
4422   if (superk->is_interface()) {
4423     // Cannot trust interfaces yet.
4424     // %%% S.B. superk->nof_implementors() == 1
4425   } else if (superelem->is_instance_klass()) {
4426     ciInstanceKlass* ik = superelem->as_instance_klass();
4427     if (!ik->has_subklass() && !ik->is_interface()) {
4428       if (!ik->is_final()) {
4429         // Add a dependency if there is a chance of a later subclass.
4430         dependencies()->assert_leaf_type(ik);
4431       }
4432       return SSC_easy_test;     // (3) caller can do a simple ptr comparison
4433     }

4749 void Compile::cleanup_expensive_nodes(PhaseIterGVN &igvn) {
4750   if (_expensive_nodes->length() == 0) {
4751     return;
4752   }
4753 
4754   assert(OptimizeExpensiveOps, "optimization off?");
4755 
4756   // Sort to bring similar nodes next to each other and clear the
4757   // control input of nodes for which there's only a single copy.
4758   sort_expensive_nodes();
4759 
4760   int j = 0;
4761   int identical = 0;
4762   int i = 0;
4763   bool modified = false;
4764   for (; i < _expensive_nodes->length()-1; i++) {
4765     assert(j <= i, "can't write beyond current index");
4766     if (_expensive_nodes->at(i)->Opcode() == _expensive_nodes->at(i+1)->Opcode()) {
4767       identical++;
4768       _expensive_nodes->at_put(j++, _expensive_nodes->at(i));
4769       continue;
4770     }
4771     if (identical > 0) {
4772       _expensive_nodes->at_put(j++, _expensive_nodes->at(i));
4773       identical = 0;

4775       Node* n = _expensive_nodes->at(i);
4776       igvn.replace_input_of(n, 0, NULL);
4777       igvn.hash_insert(n);
4778       modified = true;
4779     }
4780   }
4781   if (identical > 0) {
4782     _expensive_nodes->at_put(j++, _expensive_nodes->at(i));
4783   } else if (_expensive_nodes->length() >= 1) {
4784     Node* n = _expensive_nodes->at(i);
4785     igvn.replace_input_of(n, 0, NULL);
4786     igvn.hash_insert(n);

4788   }

4891 // Auxiliary method to support randomized stressing/fuzzing.
4892 //
4893 // This method can be called the arbitrary number of times, with current count
4894 // as the argument. The logic allows selecting a single candidate from the
4895 // running list of candidates as follows:
4896 //    int count = 0;
4897 //    Cand* selected = null;
4898 //    while(cand = cand->next()) {
4899 //      if (randomized_select(++count)) {
4900 //        selected = cand;
4901 //      }
4902 //    }
4903 //
4904 // Including count equalizes the chances any candidate is "selected".
4905 // This is useful when we don't have the complete list of candidates to choose
4906 // from uniformly. In this case, we need to adjust the randomicity of the
4907 // selection, or else we will end up biasing the selection towards the latter
4908 // candidates.
4909 //
4910 // Quick back-envelope calculation shows that for the list of n candidates
4911 // the equal probability for the candidate to persist as "best" can be
4912 // achieved by replacing it with "next" k-th candidate with the probability
4913 // of 1/k. It can be easily shown that by the end of the run, the
4914 // probability for any candidate is converged to 1/n, thus giving the
4915 // uniform distribution among all the candidates.
4916 //
4917 // We don't care about the domain size as long as (RANDOMIZED_DOMAIN / count) is large.
4918 #define RANDOMIZED_DOMAIN_POW 29
4919 #define RANDOMIZED_DOMAIN (1 << RANDOMIZED_DOMAIN_POW)
4920 #define RANDOMIZED_DOMAIN_MASK ((1 << (RANDOMIZED_DOMAIN_POW + 1)) - 1)
4921 bool Compile::randomized_select(int count) {
4922   assert(count > 0, "only positive");
4923   return (os::random() & RANDOMIZED_DOMAIN_MASK) < (RANDOMIZED_DOMAIN / count);
4924 }
