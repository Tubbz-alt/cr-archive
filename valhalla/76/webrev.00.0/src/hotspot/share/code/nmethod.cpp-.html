 195 struct pc_nmethod_stats_struct {
 196   int pc_desc_resets;   // number of resets (= number of caches)
 197   int pc_desc_queries;  // queries to nmethod::find_pc_desc
 198   int pc_desc_approx;   // number of those which have approximate true
 199   int pc_desc_repeats;  // number of _pc_descs[0] hits
 200   int pc_desc_hits;     // number of LRU cache hits
 201   int pc_desc_tests;    // total number of PcDesc examinations
 202   int pc_desc_searches; // total number of quasi-binary search steps
 203   int pc_desc_adds;     // number of LUR cache insertions
 204 
 205   void print_pc_stats() {
 206     tty->print_cr("PcDesc Statistics:  %d queries, %.2f comparisons per query",
 207                   pc_desc_queries,
 208                   (double)(pc_desc_tests + pc_desc_searches)
 209                   / pc_desc_queries);
 210     tty->print_cr("  caches=%d queries=%d/%d, hits=%d+%d, tests=%d+%d, adds=%d",
 211                   pc_desc_resets,
 212                   pc_desc_queries, pc_desc_approx,
 213                   pc_desc_repeats, pc_desc_hits,
 214                   pc_desc_tests, pc_desc_searches, pc_desc_adds);
 215   }
 216 };

 271 address ExceptionCache::match(Handle exception, address pc) {
 272   assert(pc != NULL,"Must be non null");
 273   assert(exception.not_null(),"Must be non null");
 274   if (exception->klass() == exception_type()) {
 275     return (test_address(pc));
 276   }
 277 
 278   return NULL;
 279 }

 282 bool ExceptionCache::match_exception_with_space(Handle exception) {
 283   assert(exception.not_null(),"Must be non null");
 284   if (exception->klass() == exception_type() && count() < cache_size) {
 285     return true;
 286   }
 287   return false;
 288 }

 291 address ExceptionCache::test_address(address addr) {
 292   int limit = count();
 293   for (int i = 0; i < limit; i++) {
 294     if (pc_at(i) == addr) {
 295       return handler_at(i);
 296     }
 297   }
 298   return NULL;
 299 }

 302 bool ExceptionCache::add_address_and_handler(address addr, address handler) {
 303   if (test_address(addr) == handler) return true;
 304 
 305   int index = count();
 306   if (index < cache_size) {
 307     set_pc_at(index, addr);
 308     set_handler_at(index, handler);
 309     increment_count();
 310     return true;
 311   }
 312   return false;
 313 }

 326 // Helper used by both find_pc_desc methods.
 327 static inline bool match_desc(PcDesc* pc, int pc_offset, bool approximate) {
 328   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_tests);
 329   if (!approximate)
 330     return pc->pc_offset() == pc_offset;
 331   else
 332     return (pc-1)->pc_offset() < pc_offset && pc_offset <= pc->pc_offset();
 333 }
 334 
 335 void PcDescCache::reset_to(PcDesc* initial_pc_desc) {
 336   if (initial_pc_desc == NULL) {
 337     _pc_descs[0] = NULL; // native method; no PcDescs at all
 338     return;
 339   }
 340   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_resets);
 341   // reset the cache by filling it with benign (non-null) values
 342   assert(initial_pc_desc->pc_offset() < 0, "must be sentinel");
 343   for (int i = 0; i < cache_size; i++)
 344     _pc_descs[i] = initial_pc_desc;
 345 }
 346 
 347 PcDesc* PcDescCache::find_pc_desc(int pc_offset, bool approximate) {
 348   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_queries);
 349   NOT_PRODUCT(if (approximate) ++pc_nmethod_stats.pc_desc_approx);
 350 
 351   // Note: one might think that caching the most recently
 352   // read value separately would be a win, but one would be
 353   // wrong.  When many threads are updating it, the cache
 354   // line it's in would bounce between caches, negating
 355   // any benefit.
 356 
 357   // In order to prevent race conditions do not load cache elements
 358   // repeatedly, but use a local copy:
 359   PcDesc* res;
 360 
 361   // Step one:  Check the most recently added value.
 362   res = _pc_descs[0];
 363   if (res == NULL) return NULL;  // native method; no PcDescs at all
 364   if (match_desc(res, pc_offset, approximate)) {
 365     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_repeats);
 366     return res;
 367   }
 368 
 369   // Step two:  Check the rest of the LRU cache.
 370   for (int i = 1; i < cache_size; ++i) {
 371     res = _pc_descs[i];
 372     if (res->pc_offset() < 0) break;  // optimization: skip empty cache
 373     if (match_desc(res, pc_offset, approximate)) {
 374       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_hits);
 375       return res;
 376     }
 377   }
 378 
 379   // Report failure.
 380   return NULL;
 381 }
 382 
 383 void PcDescCache::add_pc_desc(PcDesc* pc_desc) {
 384   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_adds);
 385   // Update the LRU cache by shifting pc_desc forward.
 386   for (int i = 0; i < cache_size; i++)  {
 387     PcDesc* next = _pc_descs[i];
 388     _pc_descs[i] = pc_desc;
 389     pc_desc = next;
 390   }
 391 }
 392 
 393 // adjust pcs_size so that it is a multiple of both oopSize and
 394 // sizeof(PcDesc) (assumes that if sizeof(PcDesc) is not a multiple
 395 // of oopSize, then 2*sizeof(PcDesc) is)
 396 static int adjust_pcs_size(int pcs_size) {
 397   int nsize = align_up(pcs_size,   oopSize);
 398   if ((nsize % sizeof(PcDesc)) != 0) {
 399     nsize = pcs_size + sizeof(PcDesc);
 400   }
 401   assert((nsize % oopSize) == 0, "correct alignment");
 402   return nsize;
 403 }

 448 nmethod* nmethod::new_native_nmethod(const methodHandle& method,
 449   int compile_id,
 450   CodeBuffer *code_buffer,
 451   int vep_offset,
 452   int frame_complete,
 453   int frame_size,
 454   ByteSize basic_lock_owner_sp_offset,
 455   ByteSize basic_lock_sp_offset,
 456   OopMapSet* oop_maps) {
 457   code_buffer->finalize_oop_references(method);
 458   // create nmethod
 459   nmethod* nm = NULL;
 460   {
 461     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 462     int native_nmethod_size = CodeBlob::allocation_size(code_buffer, sizeof(nmethod));
 463 
 464     CodeOffsets offsets;
 465     offsets.set_value(CodeOffsets::Verified_Entry, vep_offset);
 466     offsets.set_value(CodeOffsets::Frame_Complete, frame_complete);
 467     nm = new (native_nmethod_size, CompLevel_none)
 468     nmethod(method(), compiler_none, native_nmethod_size,
 469             compile_id, &offsets,
 470             code_buffer, frame_size,
 471             basic_lock_owner_sp_offset,
 472             basic_lock_sp_offset,
 473             oop_maps);
 474     NOT_PRODUCT(if (nm != NULL)  native_nmethod_stats.note_native_nmethod(nm));
 475   }
 476 
 477   if (nm != NULL) {
 478     // verify nmethod
 479     debug_only(nm->verify();) // might block
 480 
 481     nm->log_new_nmethod();
 482   }
 483   return nm;
 484 }

 486 nmethod* nmethod::new_nmethod(const methodHandle& method,
 487   int compile_id,
 488   int entry_bci,
 489   CodeOffsets* offsets,
 490   int orig_pc_offset,
 491   DebugInformationRecorder* debug_info,
 492   Dependencies* dependencies,
 493   CodeBuffer* code_buffer, int frame_size,
 494   OopMapSet* oop_maps,
 495   ExceptionHandlerTable* handler_table,
 496   ImplicitExceptionTable* nul_chk_table,
 497   AbstractCompiler* compiler,
 498   int comp_level
 499 #if INCLUDE_JVMCI
 500   , char* speculations,
 501   int speculations_len,
 502   int nmethod_mirror_index,
 503   const char* nmethod_mirror_name,
 504   FailedSpeculation** failed_speculations
 505 #endif
 506 )
 507 {
 508   assert(debug_info->oop_recorder() == code_buffer->oop_recorder(), "shared OR");
 509   code_buffer->finalize_oop_references(method);
 510   // create nmethod
 511   nmethod* nm = NULL;
 512   { MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 513 #if INCLUDE_JVMCI
 514     int jvmci_data_size = !compiler->is_jvmci() ? 0 : JVMCINMethodData::compute_size(nmethod_mirror_name);
 515 #endif
 516     int nmethod_size =
 517       CodeBlob::allocation_size(code_buffer, sizeof(nmethod))
 518       + adjust_pcs_size(debug_info->pcs_size())
 519       + align_up((int)dependencies->size_in_bytes(), oopSize)
 520       + align_up(handler_table->size_in_bytes()    , oopSize)
 521       + align_up(nul_chk_table->size_in_bytes()    , oopSize)
 522 #if INCLUDE_JVMCI
 523       + align_up(speculations_len                  , oopSize)
 524       + align_up(jvmci_data_size                   , oopSize)
 525 #endif
 526       + align_up(debug_info->data_size()           , oopSize);
 527 
 528     nm = new (nmethod_size, comp_level)
 529     nmethod(method(), compiler->type(), nmethod_size, compile_id, entry_bci, offsets,
 530             orig_pc_offset, debug_info, dependencies, code_buffer, frame_size,
 531             oop_maps,
 532             handler_table,
 533             nul_chk_table,
 534             compiler,
 535             comp_level
 536 #if INCLUDE_JVMCI
 537             , speculations,
 538             speculations_len,
 539             jvmci_data_size
 540 #endif
 541             );
 542 
 543     if (nm != NULL) {
 544 #if INCLUDE_JVMCI
 545       if (compiler->is_jvmci()) {
 546         // Initialize the JVMCINMethodData object inlined into nm
 547         nm->jvmci_nmethod_data()->initialize(nmethod_mirror_index, nmethod_mirror_name, failed_speculations);
 548       }
 549 #endif
 550       // To make dependency checking during class loading fast, record
 551       // the nmethod dependencies in the classes it is dependent on.
 552       // This allows the dependency checking code to simply walk the
 553       // class hierarchy above the loaded class, checking only nmethods
 554       // which are dependent on those classes.  The slow way is to
 555       // check every nmethod for dependencies which makes it linear in
 556       // the number of methods compiled.  For applications with a lot
 557       // classes the slow way is too slow.
 558       for (Dependencies::DepStream deps(nm); deps.next(); ) {
 559         if (deps.type() == Dependencies::call_site_target_value) {
 560           // CallSite dependencies are managed on per-CallSite instance basis.
 561           oop call_site = deps.argument_oop(0);
 562           MethodHandles::add_dependent_nmethod(call_site, nm);
 563         } else {
 564           Klass* klass = deps.context_type();
 565           if (klass == NULL) {
 566             continue;  // ignore things like evol_method
 567           }
 568           // record this nmethod as dependent on this klass
 569           InstanceKlass::cast(klass)->add_dependent_nmethod(nm);
 570         }
 571       }
 572       NOT_PRODUCT(if (nm != NULL)  note_java_nmethod(nm));
 573     }
 574   }
 575   // Do verification and logging outside CodeCache_lock.
 576   if (nm != NULL) {
 577     // Safepoints in nmethod::verify aren't allowed because nm hasn't been installed yet.
 578     DEBUG_ONLY(nm->verify();)
 579     nm->log_new_nmethod();
 580   }
 581   return nm;
 582 }

1040 // Have to have the same name because it's called by a template
1041 void nmethod::copy_values(GrowableArray<jobject>* array) {
1042   int length = array->length();
1043   assert((address)(oops_begin() + length) <= (address)oops_end(), "oops big enough");
1044   oop* dest = oops_begin();
1045   for (int index = 0 ; index < length; index++) {
1046     initialize_immediate_oop(&dest[index], array->at(index));

1050   // in the code because the assembler uses jobjects as placeholders.
1051   // The code and relocations have already been initialized by the
1052   // CodeBlob constructor, so it is valid even at this early point to
1053   // iterate over relocations and patch the code.

1066 void nmethod::fix_oop_relocations(address begin, address end, bool initialize_immediates) {
1067   // re-patch all oop-bearing instructions, just in case some oops moved
1068   RelocIterator iter(this, begin, end);
1069   while (iter.next()) {
1070     if (iter.type() == relocInfo::oop_type) {
1071       oop_Relocation* reloc = iter.oop_reloc();
1072       if (initialize_immediates && reloc->oop_is_immediate()) {
1073         oop* dest = reloc->oop_addr();
1074         initialize_immediate_oop(dest, cast_from_oop<jobject>(*dest));
1075       }
1076       // Refresh the oop-related bits of this instruction.
1077       reloc->fix_oop_relocation();
1078     } else if (iter.type() == relocInfo::metadata_type) {
1079       metadata_Relocation* reloc = iter.metadata_reloc();
1080       reloc->fix_metadata_relocation();
1081     }
1082   }
1083 }

1087   assert(CompiledICLocker::is_safe(this), "mt unsafe call");
1088 
1089   ResourceMark rm;

1095         CompiledIC *ic = CompiledIC_at(&iter);
1096         // Ok, to lookup references to zombies here
1097         CodeBlob *cb = CodeCache::find_blob_unsafe(ic->ic_destination());
1098         assert(cb != NULL, "destination not in CodeBlob?");
1099         nmethod* nm = cb->as_nmethod_or_null();
1100         if( nm != NULL ) {
1101           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1102           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1103             assert(ic->is_clean(), "IC should be clean");
1104           }
1105         }
1106         break;
1107       }
1108       case relocInfo::static_call_type: {
1109         CompiledStaticCall *csc = compiledStaticCall_at(iter.reloc());
1110         CodeBlob *cb = CodeCache::find_blob_unsafe(csc->destination());
1111         assert(cb != NULL, "destination not in CodeBlob?");
1112         nmethod* nm = cb->as_nmethod_or_null();
1113         if( nm != NULL ) {
1114           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1115           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1116             assert(csc->is_clean(), "IC should be clean");
1117           }
1118         }
1119         break;
1120       }
1121       default:
1122         break;
1123     }
1124   }
1125 }

1135 // Tell if a non-entrant method can be converted to a zombie (i.e.,
1136 // there are no activations on the stack, not in use by the VM,
1137 // and not in use by the ServiceThread)
1138 bool nmethod::can_convert_to_zombie() {
1139   // Note that this is called when the sweeper has observed the nmethod to be
1140   // not_entrant. However, with concurrent code cache unloading, the state
1141   // might have moved on to unloaded if it is_unloading(), due to racing
1142   // concurrent GC threads.
1143   assert(is_not_entrant() || is_unloading(), "must be a non-entrant method");
1144 
1145   // Since the nmethod sweeper only does partial sweep the sweeper's traversal
1146   // count can be greater than the stack traversal count before it hits the
1147   // nmethod for the second time.
1148   // If an is_unloading() nmethod is still not_entrant, then it is not safe to
1149   // convert it to zombie due to GC unloading interactions. However, if it
1150   // has become unloaded, then it is okay to convert such nmethods to zombie.
1151   return stack_traversal_mark() + 1 < NMethodSweeper::traversal_count() &&
1152          !is_locked_by_vm() && (!is_unloading() || is_unloaded());
1153 }

1166 bool nmethod::try_transition(int new_state_int) {
1167   signed char new_state = new_state_int;
1168 #ifdef DEBUG
1169   if (new_state != unloaded) {
1170     assert_lock_strong(CompiledMethod_lock);
1171   }
1172 #endif
1173   for (;;) {
1174     signed char old_state = Atomic::load(&_state);
1175     if (old_state >= new_state) {
1176       // Ensure monotonicity of transitions.
1177       return false;
1178     }
1179     if (Atomic::cmpxchg(&_state, old_state, new_state) == old_state) {
1180       return true;
1181     }
1182   }

1185 void nmethod::make_unloaded() {
1186   post_compiled_method_unload();
1187 
1188   // This nmethod is being unloaded, make sure that dependencies
1189   // recorded in instanceKlasses get flushed.
1190   // Since this work is being done during a GC, defer deleting dependencies from the
1191   // InstanceKlass.
1192   assert(Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread(),
1193          "should only be called during gc");
1194   flush_dependencies(/*delete_immediately*/false);
1195 
1196   // Break cycle between nmethod & method
1197   LogTarget(Trace, class, unload, nmethod) lt;
1198   if (lt.is_enabled()) {
1199     LogStream ls(lt);
1200     ls.print("making nmethod " INTPTR_FORMAT
1201              " unloadable, Method*(" INTPTR_FORMAT
1202              ") ",
1203              p2i(this), p2i(_method));
1204      ls.cr();
1205   }
1206   // Unlink the osr method, so we do not look this up again
1207   if (is_osr_method()) {
1208     // Invalidate the osr nmethod only once. Note that with concurrent
1209     // code cache unloading, OSR nmethods are invalidated before they
1210     // are made unloaded. Therefore, this becomes a no-op then.
1211     if (is_in_use()) {
1212       invalidate_osr_method();
1213     }
1214 #ifdef ASSERT
1215     if (method() != NULL) {
1216       // Make sure osr nmethod is invalidated, i.e. not on the list
1217       bool found = method()->method_holder()->remove_osr_nmethod(this);
1218       assert(!found, "osr nmethod should have been invalidated");
1219     }
1220 #endif
1221   }
1222 
1223   // If _method is already NULL the Method* is about to be unloaded,
1224   // so we don't have to break the cycle. Note that it is possible to
1225   // have the Method* live here, in case we unload the nmethod because
1226   // it is pointing to some oop (other than the Method*) being unloaded.
1227   if (_method != NULL) {
1228     _method->unlink_code(this);
1229   }
1230 
1231   // Make the class unloaded - i.e., change state and notify sweeper
1232   assert(SafepointSynchronize::is_at_safepoint() || Thread::current()->is_ConcurrentGC_thread(),
1233          "must be at safepoint");
1234 
1235   {
1236     // Clear ICStubs and release any CompiledICHolders.
1237     CompiledICLocker ml(this);
1238     clear_ic_callsites();
1239   }
1240 
1241   // Unregister must be done before the state change
1242   {
1243     MutexLocker ml(SafepointSynchronize::is_at_safepoint() ? NULL : CodeCache_lock,
1244                      Mutex::_no_safepoint_check_flag);
1245     Universe::heap()->unregister_nmethod(this);
1246   }
1247 
1248   // Clear the method of this dead nmethod
1249   set_method(NULL);
1250 
1251   // Log the unloading.
1252   log_state_change();
1253 
1254   // The Method* is gone at this point
1255   assert(_method == NULL, "Tautology");
1256 
1257   set_osr_link(NULL);
1258   NMethodSweeper::report_state_change(this);
1259 
1260   bool transition_success = try_transition(unloaded);
1261 
1262   // It is an important invariant that there exists no race between
1263   // the sweeper and GC thread competing for making the same nmethod
1264   // zombie and unloaded respectively. This is ensured by
1265   // can_convert_to_zombie() returning false for any is_unloading()
1266   // nmethod, informing the sweeper not to step on any GC toes.
1267   assert(transition_success, "Invalid nmethod transition to unloaded");
1268 
1269 #if INCLUDE_JVMCI
1270   // Clear the link between this nmethod and a HotSpotNmethod mirror
1271   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1272   if (nmethod_data != NULL) {
1273     nmethod_data->invalidate_nmethod_mirror(this);
1274     nmethod_data->clear_nmethod_mirror(this);
1275   }
1276 #endif
1277 }

1318 /**
1319  * Common functionality for both make_not_entrant and make_zombie
1320  */
1321 bool nmethod::make_not_entrant_or_zombie(int state) {
1322   assert(state == zombie || state == not_entrant, "must be zombie or not_entrant");
1323 
1324   if (Atomic::load(&_state) >= state) {
1325     // Avoid taking the lock if already in required state.
1326     // This is safe from races because the state is an end-state,
1327     // which the nmethod cannot back out of once entered.
1328     // No need for fencing either.
1329     return false;
1330   }
1331 
1332   // Make sure the nmethod is not flushed.
1333   nmethodLocker nml(this);
1334   // This can be called while the system is already at a safepoint which is ok
1335   NoSafepointVerifier nsv;
1336 
1337   // during patching, depending on the nmethod state we must notify the GC that
1338   // code has been unloaded, unregistering it. We cannot do this right while
1339   // holding the CompiledMethod_lock because we need to use the CodeCache_lock. This
1340   // would be prone to deadlocks.
1341   // This flag is used to remember whether we need to later lock and unregister.
1342   bool nmethod_needs_unregister = false;
1343 
1344   {
1345     // Enter critical section.  Does not block for safepoint.
1346     MutexLocker ml(CompiledMethod_lock->owned_by_self() ? NULL : CompiledMethod_lock, Mutex::_no_safepoint_check_flag);
1347 
1348     // This logic is equivalent to the logic below for patching the
1349     // verified entry point of regular methods. We check that the
1350     // nmethod is in use to ensure that it is invalidated only once.
1351     if (is_osr_method() && is_in_use()) {
1352       // this effectively makes the osr nmethod not entrant
1353       invalidate_osr_method();
1354     }
1355 
1356     if (Atomic::load(&_state) >= state) {
1357       // another thread already performed this transition so nothing
1358       // to do, but return false to indicate this.
1359       return false;
1360     }
1361 
1362     // The caller can be calling the method statically or through an inline
1363     // cache call.
1364     if (!is_osr_method() && !is_not_entrant()) {
1365       NativeJump::patch_verified_entry(entry_point(), verified_entry_point(),
1366                   SharedRuntime::get_handle_wrong_method_stub());
1367     }
1368 
1369     if (is_in_use() && update_recompile_counts()) {
1370       // It's a true state change, so mark the method as decompiled.
1371       // Do it only for transition from alive.
1372       inc_decompile_count();
1373     }
1374 
1375     // If the state is becoming a zombie, signal to unregister the nmethod with
1376     // the heap.
1377     // This nmethod may have already been unloaded during a full GC.
1378     if ((state == zombie) && !is_unloaded()) {
1379       nmethod_needs_unregister = true;
1380     }
1381 
1382     // Must happen before state change. Otherwise we have a race condition in
1383     // nmethod::can_convert_to_zombie(). I.e., a method can immediately
1384     // transition its state from 'not_entrant' to 'zombie' without having to wait
1385     // for stack scanning.
1386     if (state == not_entrant) {
1387       mark_as_seen_on_stack();
1388       OrderAccess::storestore(); // _stack_traversal_mark and _state
1389     }
1390 
1391     // Change state
1392     if (!try_transition(state)) {
1393       // If the transition fails, it is due to another thread making the nmethod more
1394       // dead. In particular, one thread might be making the nmethod unloaded concurrently.
1395       // If so, having patched in the jump in the verified entry unnecessarily is fine.
1396       // The nmethod is no longer possible to call by Java threads.
1397       // Incrementing the decompile count is also fine as the caller of make_not_entrant()
1398       // had a valid reason to deoptimize the nmethod.
1399       // Marking the nmethod as seen on stack also has no effect, as the nmethod is now
1400       // !is_alive(), and the seen on stack value is only used to convert not_entrant
1401       // nmethods to zombie in can_convert_to_zombie().
1402       return false;
1403     }
1404 
1405     // Log the transition once
1406     log_state_change();
1407 
1408     // Remove nmethod from method.
1409     unlink_from_method();
1410 
1411   } // leave critical region under CompiledMethod_lock
1412 
1413 #if INCLUDE_JVMCI
1414   // Invalidate can't occur while holding the Patching lock
1415   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1416   if (nmethod_data != NULL) {
1417     nmethod_data->invalidate_nmethod_mirror(this);
1418   }
1419 #endif
1420 
1421 #ifdef ASSERT
1422   if (is_osr_method() && method() != NULL) {
1423     // Make sure osr nmethod is invalidated, i.e. not on the list
1424     bool found = method()->method_holder()->remove_osr_nmethod(this);
1425     assert(!found, "osr nmethod should have been invalidated");
1426   }
1427 #endif
1428 
1429   // When the nmethod becomes zombie it is no longer alive so the
1430   // dependencies must be flushed.  nmethods in the not_entrant
1431   // state will be flushed later when the transition to zombie
1432   // happens or they get unloaded.
1433   if (state == zombie) {
1434     {
1435       // Flushing dependencies must be done before any possible
1436       // safepoint can sneak in, otherwise the oops used by the
1437       // dependency logic could have become stale.
1438       MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1439       if (nmethod_needs_unregister) {
1440         Universe::heap()->unregister_nmethod(this);
1441       }
1442       flush_dependencies(/*delete_immediately*/true);
1443     }
1444 
1445 #if INCLUDE_JVMCI
1446     // Now that the nmethod has been unregistered, it's
1447     // safe to clear the HotSpotNmethod mirror oop.
1448     if (nmethod_data != NULL) {
1449       nmethod_data->clear_nmethod_mirror(this);
1450     }
1451 #endif
1452 
1453     // Clear ICStubs to prevent back patching stubs of zombie or flushed
1454     // nmethods during the next safepoint (see ICStub::finalize), as well
1455     // as to free up CompiledICHolder resources.
1456     {
1457       CompiledICLocker ml(this);
1458       clear_ic_callsites();
1459     }
1460 
1461     // zombie only - if a JVMTI agent has enabled the CompiledMethodUnload
1462     // event and it hasn't already been reported for this nmethod then
1463     // report it now. The event may have been reported earlier if the GC
1464     // marked it for unloading). JvmtiDeferredEventQueue support means
1465     // we no longer go to a safepoint here.
1466     post_compiled_method_unload();
1467 
1468 #ifdef ASSERT
1469     // It's no longer safe to access the oops section since zombie
1470     // nmethods aren't scanned for GC.
1471     _oops_are_stale = true;
1472 #endif
1473      // the Method may be reclaimed by class unloading now that the
1474      // nmethod is in zombie state
1475     set_method(NULL);
1476   } else {
1477     assert(state == not_entrant, "other cases may need to be handled differently");
1478   }
1479 
1480   if (TraceCreateZombies && state == zombie) {
1481     ResourceMark m;
1482     tty->print_cr("nmethod <" INTPTR_FORMAT "> %s code made %s", p2i(this), this->method() ? this->method()->name_and_sig_as_C_string() : "null", (state == not_entrant) ? "not entrant" : "zombie");
1483   }
1484 
1485   NMethodSweeper::report_state_change(this);
1486   return true;
1487 }

1489 void nmethod::flush() {
1490   MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1491   // Note that there are no valid oops in the nmethod anymore.
1492   assert(!is_osr_method() || is_unloaded() || is_zombie(),
1493          "osr nmethod must be unloaded or zombie before flushing");
1494   assert(is_zombie() || is_osr_method(), "must be a zombie method");
1495   assert (!is_locked_by_vm(), "locked methods shouldn't be flushed");
1496   assert_locked_or_safepoint(CodeCache_lock);
1497 
1498   // completely deallocate this method
1499   Events::log(JavaThread::current(), "flushing nmethod " INTPTR_FORMAT, p2i(this));
1500   if (PrintMethodFlushing) {
1501     tty->print_cr("*flushing %s nmethod %3d/" INTPTR_FORMAT ". Live blobs:" UINT32_FORMAT
1502                   "/Free CodeCache:" SIZE_FORMAT "Kb",
1503                   is_osr_method() ? "osr" : "",_compile_id, p2i(this), CodeCache::blob_count(),
1504                   CodeCache::unallocated_capacity(CodeCache::get_code_blob_type(this))/1024);
1505   }
1506 
1507   // We need to deallocate any ExceptionCache data.
1508   // Note that we do not need to grab the nmethod lock for this, it
1509   // better be thread safe if we're disposing of it!
1510   ExceptionCache* ec = exception_cache();
1511   set_exception_cache(NULL);
1512   while(ec != NULL) {
1513     ExceptionCache* next = ec->next();
1514     delete ec;
1515     ec = next;
1516   }
1517 
1518   Universe::heap()->flush_nmethod(this);
1519   CodeCache::unregister_old_nmethod(this);
1520 
1521   CodeBlob::flush();
1522   CodeCache::free(this);
1523 }

1539 //
1540 // Notify all classes this nmethod is dependent on that it is no
1541 // longer dependent. This should only be called in two situations.
1542 // First, when a nmethod transitions to a zombie all dependents need
1543 // to be clear.  Since zombification happens at a safepoint there's no
1544 // synchronization issues.  The second place is a little more tricky.
1545 // During phase 1 of mark sweep class unloading may happen and as a
1546 // result some nmethods may get unloaded.  In this case the flushing
1547 // of dependencies must happen during phase 1 since after GC any
1548 // dependencies in the unloaded nmethod won't be updated, so
1549 // traversing the dependency information in unsafe.  In that case this
1550 // function is called with a boolean argument and this function only
1551 // notifies instanceKlasses that are reachable
1552 
1553 void nmethod::flush_dependencies(bool delete_immediately) {
1554   DEBUG_ONLY(bool called_by_gc = Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread();)
1555   assert(called_by_gc != delete_immediately,
1556   "delete_immediately is false if and only if we are called during GC");
1557   if (!has_flushed_dependencies()) {
1558     set_has_flushed_dependencies();
1559     for (Dependencies::DepStream deps(this); deps.next(); ) {
1560       if (deps.type() == Dependencies::call_site_target_value) {
1561         // CallSite dependencies are managed on per-CallSite instance basis.
1562         oop call_site = deps.argument_oop(0);
1563         if (delete_immediately) {
1564           assert_locked_or_safepoint(CodeCache_lock);
1565           MethodHandles::remove_dependent_nmethod(call_site, this);
1566         } else {
1567           MethodHandles::clean_dependency_context(call_site);
1568         }
1569       } else {
1570         Klass* klass = deps.context_type();
1571         if (klass == NULL) {
1572           continue;  // ignore things like evol_method
1573         }
1574         // During GC delete_immediately is false, and liveness
1575         // of dependee determines class that needs to be updated.
1576         if (delete_immediately) {
1577           assert_locked_or_safepoint(CodeCache_lock);
1578           InstanceKlass::cast(klass)->remove_dependent_nmethod(this);
1579         } else if (klass->is_loader_alive()) {
1580           // The GC may clean dependency contexts concurrently and in parallel.
1581           InstanceKlass::cast(klass)->clean_dependency_context();
1582         }
1583       }
1584     }
1585   }
1586 }

1589 // post_compiled_method_load_event
1590 // new method for install_code() path
1591 // Transfer information from compilation to jvmti
1592 void nmethod::post_compiled_method_load_event(JvmtiThreadState* state) {
1593 
1594   // Don't post this nmethod load event if it is already dying
1595   // because the sweeper might already be deleting this nmethod.
1596   if (is_not_entrant() && can_convert_to_zombie()) {
1597     return;
1598   }
1599 
1600   // This is a bad time for a safepoint.  We don't want
1601   // this nmethod to get unloaded while we're queueing the event.
1602   NoSafepointVerifier nsv;
1603 
1604   Method* m = method();
1605   HOTSPOT_COMPILED_METHOD_LOAD(
1606       (char *) m->klass_name()->bytes(),
1607       m->klass_name()->utf8_length(),
1608       (char *) m->name()->bytes(),
1609       m->name()->utf8_length(),
1610       (char *) m->signature()->bytes(),
1611       m->signature()->utf8_length(),
1612       insts_begin(), insts_size());
1613 
1614 
1615   if (JvmtiExport::should_post_compiled_method_load()) {
1616     // Only post unload events if load events are found.
1617     set_load_reported();
1618     // If a JavaThread hasn't been passed in, let the Service thread
1619     // (which is a real Java thread) post the event
1620     JvmtiDeferredEvent event = JvmtiDeferredEvent::compiled_method_load_event(this);
1621     if (state == NULL) {
1622       // Execute any barrier code for this nmethod as if it's called, since
1623       // keeping it alive looks like stack walking.
1624       run_nmethod_entry_barrier();
1625       ServiceThread::enqueue_deferred_event(&event);
1626     } else {
1627       // This enters the nmethod barrier outside in the caller.
1628       state->enqueue_event(&event);
1629     }
1630   }
1631 }

1633 void nmethod::post_compiled_method_unload() {
1634   if (unload_reported()) {
1635     // During unloading we transition to unloaded and then to zombie
1636     // and the unloading is reported during the first transition.
1637     return;
1638   }
1639 
1640   assert(_method != NULL && !is_unloaded(), "just checking");
1641   DTRACE_METHOD_UNLOAD_PROBE(method());
1642 
1643   // If a JVMTI agent has enabled the CompiledMethodUnload event then
1644   // post the event. Sometime later this nmethod will be made a zombie
1645   // by the sweeper but the Method* will not be valid at that point.
1646   // The jmethodID is a weak reference to the Method* so if
1647   // it's being unloaded there's no way to look it up since the weak
1648   // ref will have been cleared.
1649 
1650   // Don't bother posting the unload if the load event wasn't posted.
1651   if (load_reported() && JvmtiExport::should_post_compiled_method_unload()) {
1652     assert(!unload_reported(), "already unloaded");
1653     JvmtiDeferredEvent event =
1654       JvmtiDeferredEvent::compiled_method_unload_event(
1655           method()->jmethod_id(), insts_begin());
1656     ServiceThread::enqueue_deferred_event(&event);
1657   }
1658 
1659   // The JVMTI CompiledMethodUnload event can be enabled or disabled at
1660   // any time. As the nmethod is being unloaded now we mark it has
1661   // having the unload event reported - this will ensure that we don't

1714 // The _is_unloading_state encodes a tuple comprising the unloading cycle
1715 // and the result of IsUnloadingBehaviour::is_unloading() fpr that cycle.
1716 // This is the bit layout of the _is_unloading_state byte: 00000CCU
1717 // CC refers to the cycle, which has 2 bits, and U refers to the result of
1718 // IsUnloadingBehaviour::is_unloading() for that unloading cycle.
1719 
1720 class IsUnloadingState: public AllStatic {
1721   static const uint8_t _is_unloading_mask = 1;
1722   static const uint8_t _is_unloading_shift = 0;
1723   static const uint8_t _unloading_cycle_mask = 6;
1724   static const uint8_t _unloading_cycle_shift = 1;
1725 
1726   static uint8_t set_is_unloading(uint8_t state, bool value) {
1727     state &= ~_is_unloading_mask;
1728     if (value) {
1729       state |= 1 << _is_unloading_shift;
1730     }
1731     assert(is_unloading(state) == value, "unexpected unloading cycle overflow");
1732     return state;
1733   }
1734 
1735   static uint8_t set_unloading_cycle(uint8_t state, uint8_t value) {
1736     state &= ~_unloading_cycle_mask;
1737     state |= value << _unloading_cycle_shift;
1738     assert(unloading_cycle(state) == value, "unexpected unloading cycle overflow");
1739     return state;
1740   }
1741 
1742 public:
1743   static bool is_unloading(uint8_t state) { return (state & _is_unloading_mask) >> _is_unloading_shift == 1; }
1744   static uint8_t unloading_cycle(uint8_t state) { return (state & _unloading_cycle_mask) >> _unloading_cycle_shift; }
1745 
1746   static uint8_t create(bool is_unloading, uint8_t unloading_cycle) {
1747     uint8_t state = 0;
1748     state = set_is_unloading(state, is_unloading);
1749     state = set_unloading_cycle(state, unloading_cycle);
1750     return state;
1751   }
1752 };
1753 
1754 bool nmethod::is_unloading() {
1755   uint8_t state = RawAccess<MO_RELAXED>::load(&_is_unloading_state);
1756   bool state_is_unloading = IsUnloadingState::is_unloading(state);
1757   uint8_t state_unloading_cycle = IsUnloadingState::unloading_cycle(state);
1758   if (state_is_unloading) {
1759     return true;
1760   }
1761   uint8_t current_cycle = CodeCache::unloading_cycle();
1762   if (state_unloading_cycle == current_cycle) {
1763     return false;
1764   }
1765 
1766   // The IsUnloadingBehaviour is responsible for checking if there are any dead
1767   // oops in the CompiledMethod, by calling oops_do on it.
1768   state_unloading_cycle = current_cycle;
1769 
1770   if (is_zombie()) {
1771     // Zombies without calculated unloading epoch are never unloading due to GC.
1772 
1773     // There are no races where a previously observed is_unloading() nmethod
1774     // suddenly becomes not is_unloading() due to here being observed as zombie.
1775 
1776     // With STW unloading, all is_alive() && is_unloading() nmethods are unlinked
1777     // and unloaded in the safepoint. That makes races where an nmethod is first
1778     // observed as is_alive() && is_unloading() and subsequently observed as
1779     // is_zombie() impossible.
1780 
1781     // With concurrent unloading, all references to is_unloading() nmethods are
1782     // first unlinked (e.g. IC caches and dependency contexts). Then a global
1783     // handshake operation is performed with all JavaThreads before finally
1784     // unloading the nmethods. The sweeper never converts is_alive() && is_unloading()
1785     // nmethods to zombies; it waits for them to become is_unloaded(). So before
1786     // the global handshake, it is impossible for is_unloading() nmethods to
1787     // racingly become is_zombie(). And is_unloading() is calculated for all is_alive()
1788     // nmethods before taking that global handshake, meaning that it will never
1789     // be recalculated after the handshake.
1790 
1791     // After that global handshake, is_unloading() nmethods are only observable
1792     // to the iterators, and they will never trigger recomputation of the cached
1793     // is_unloading_state, and hence may not suffer from such races.
1794 
1795     state_is_unloading = false;
1796   } else {
1797     state_is_unloading = IsUnloadingBehaviour::current()->is_unloading(this);
1798   }
1799 
1800   state = IsUnloadingState::create(state_is_unloading, state_unloading_cycle);
1801 
1802   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);
1803 
1804   return state_is_unloading;
1805 }
1806 
1807 void nmethod::clear_unloading_state() {
1808   uint8_t state = IsUnloadingState::create(false, CodeCache::unloading_cycle());
1809   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);
1810 }

1963 void nmethod::oops_do_process_weak(OopsDoProcessor* p) {
1964   if (!oops_do_try_claim_weak_request()) {
1965     // Failed to claim for weak processing.
1966     oops_do_log_change("oops_do, mark weak request fail");
1967     return;
1968   }
1969 
1970   p->do_regular_processing(this);
1971 
1972   nmethod* old_head = oops_do_try_add_to_list_as_weak_done();
1973   if (old_head == NULL) {
1974     return;
1975   }
1976   oops_do_log_change("oops_do, mark weak done fail");
1977   // Adding to global list failed, another thread added a strong request.
1978   assert(extract_state(_oops_do_mark_link) == claim_strong_request_tag,
1979          "must be but is %u", extract_state(_oops_do_mark_link));
1980 
1981   oops_do_log_change("oops_do, mark weak request -> mark strong done");
1982 
1983   oops_do_set_strong_done(old_head);
1984   // Do missing strong processing.
1985   p->do_remaining_strong_processing(this);
1986 }

1988 void nmethod::oops_do_process_strong(OopsDoProcessor* p) {
1989   oops_do_mark_link* next_raw = oops_do_try_claim_strong_done();
1990   if (next_raw == NULL) {
1991     p->do_regular_processing(this);
1992     oops_do_add_to_list_as_strong_done();
1993     return;
1994   }
1995   // Claim failed. Figure out why and handle it.
1996   if (oops_do_has_weak_request(next_raw)) {
1997     oops_do_mark_link* old = next_raw;
1998     // Claim failed because being weak processed (state == "weak request").
1999     // Try to request deferred strong processing.
2000     next_raw = oops_do_try_add_strong_request(old);
2001     if (next_raw == old) {
2002       // Successfully requested deferred strong processing.
2003       return;
2004     }
2005     // Failed because of a concurrent transition. No longer in "weak request" state.
2006   }
2007   if (oops_do_has_any_strong_state(next_raw)) {
2008     // Already claimed for strong processing or requested for such.
2009     return;
2010   }
2011   if (oops_do_try_claim_weak_done_as_strong_done(next_raw)) {
2012     // Successfully claimed "weak done" as "strong done". Do the missing marking.
2013     p->do_remaining_strong_processing(this);
2014     return;
2015   }
2016   // Claim failed, some other thread got it.
2017 }

2126 // Finds a PcDesc with real-pc equal to "pc"
2127 PcDesc* PcDescContainer::find_pc_desc_internal(address pc, bool approximate, const PcDescSearch& search) {
2128   address base_address = search.code_begin();
2129   if ((pc < base_address) ||
2130       (pc - base_address) >= (ptrdiff_t) PcDesc::upper_offset_limit) {
2131     return NULL;  // PC is wildly out of range
2132   }
2133   int pc_offset = (int) (pc - base_address);
2134 
2135   // Check the PcDesc cache if it contains the desired PcDesc
2136   // (This as an almost 100% hit rate.)
2137   PcDesc* res = _pc_desc_cache.find_pc_desc(pc_offset, approximate);
2138   if (res != NULL) {
2139     assert(res == linear_search(search, pc_offset, approximate), "cache ok");
2140     return res;
2141   }
2142 
2143   // Fallback algorithm: quasi-linear search for the PcDesc
2144   // Find the last pc_offset less than the given offset.
2145   // The successor must be the required match, if there is a match at all.
2146   // (Use a fixed radix to avoid expensive affine pointer arithmetic.)
2147   PcDesc* lower = search.scopes_pcs_begin();
2148   PcDesc* upper = search.scopes_pcs_end();
2149   upper -= 1; // exclude final sentinel
2150   if (lower >= upper)  return NULL;  // native method; no PcDescs at all
2151 
2152 #define assert_LU_OK \
2153   /* invariant on lower..upper during the following search: */ \
2154   assert(lower->pc_offset() <  pc_offset, "sanity"); \
2155   assert(upper->pc_offset() >= pc_offset, "sanity")
2156   assert_LU_OK;
2157 
2158   // Use the last successful return as a split point.
2159   PcDesc* mid = _pc_desc_cache.last_pc_desc();
2160   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2161   if (mid->pc_offset() < pc_offset) {
2162     lower = mid;
2163   } else {
2164     upper = mid;
2165   }
2166 
2167   // Take giant steps at first (4096, then 256, then 16, then 1)
2168   const int LOG2_RADIX = 4 /*smaller steps in debug mode:*/ debug_only(-1);
2169   const int RADIX = (1 << LOG2_RADIX);
2170   for (int step = (1 << (LOG2_RADIX*3)); step > 1; step >>= LOG2_RADIX) {
2171     while ((mid = lower + step) < upper) {
2172       assert_LU_OK;
2173       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2174       if (mid->pc_offset() < pc_offset) {
2175         lower = mid;
2176       } else {
2177         upper = mid;
2178         break;
2179       }
2180     }
2181     assert_LU_OK;
2182   }
2183 
2184   // Sneak up on the value with a linear search of length ~16.
2185   while (true) {
2186     assert_LU_OK;
2187     mid = lower + 1;
2188     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2189     if (mid->pc_offset() < pc_offset) {
2190       lower = mid;
2191     } else {
