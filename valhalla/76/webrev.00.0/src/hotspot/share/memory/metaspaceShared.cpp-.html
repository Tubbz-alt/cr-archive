  94 // The CDS archive is divided into the following regions:
  95 //     mc  - misc code (the method entry trampolines, c++ vtables)
  96 //     rw  - read-write metadata
  97 //     ro  - read-only metadata and read-only tables
  98 //
  99 //     ca0 - closed archive heap space #0
 100 //     ca1 - closed archive heap space #1 (may be empty)
 101 //     oa0 - open archive heap space #0
 102 //     oa1 - open archive heap space #1 (may be empty)
 103 //
 104 // The mc, rw, and ro regions are linearly allocated, starting from
 105 // SharedBaseAddress, in the order of mc->rw->ro. The size of these 3 regions
 106 // are page-aligned, and there's no gap between any consecutive regions.
 107 //
 108 // These 3 regions are populated in the following steps:
 109 // [1] All classes are loaded in MetaspaceShared::preload_classes(). All metadata are
 110 //     temporarily allocated outside of the shared regions. Only the method entry
 111 //     trampolines are written into the mc region.
 112 // [2] C++ vtables are copied into the mc region.
 113 // [3] ArchiveCompactor copies RW metadata into the rw region.
 114 // [4] ArchiveCompactor copies RO metadata into the ro region.
 115 // [5] SymbolTable, StringTable, SystemDictionary, and a few other read-only data
 116 //     are copied into the ro region as read-only tables.
 117 //
 118 // The s0/s1 and oa0/oa1 regions are populated inside HeapShared::archive_java_heap_objects.
 119 // Their layout is independent of the other 4 regions.
 120 
 121 char* DumpRegion::expand_top_to(char* newtop) {
 122   assert(is_allocatable(), "must be initialized and not packed");
 123   assert(newtop >= _top, "must not grow backwards");
 124   if (newtop > _end) {
 125     MetaspaceShared::report_out_of_space(_name, newtop - _top);
 126     ShouldNotReachHere();
 127   }
 128 
 129   if (_rs == MetaspaceShared::shared_rs()) {
 130     uintx delta;
 131     if (DynamicDumpSharedSpaces) {
 132       delta = DynamicArchive::object_delta_uintx(newtop);
 133     } else {
 134       delta = MetaspaceShared::object_delta_uintx(newtop);
 135     }
 136     if (delta > MAX_SHARED_DELTA) {
 137       // This is just a sanity check and should not appear in any real world usage. This
 138       // happens only if you allocate more than 2GB of shared objects and would require
 139       // millions of shared classes.
 140       vm_exit_during_initialization("Out of memory in the CDS archive",
 141                                     "Please reduce the number of shared classes.");
 142     }
 143   }
 144 
 145   MetaspaceShared::commit_to(_rs, _vs, newtop);
 146   _top = newtop;
 147   return _top;
 148 }

 150 char* DumpRegion::allocate(size_t num_bytes, size_t alignment) {
 151   char* p = (char*)align_up(_top, alignment);
 152   char* newtop = p + align_up(num_bytes, alignment);
 153   expand_top_to(newtop);
 154   memset(p, 0, newtop - p);
 155   return p;
 156 }

 158 void DumpRegion::append_intptr_t(intptr_t n, bool need_to_mark) {
 159   assert(is_aligned(_top, sizeof(intptr_t)), "bad alignment");
 160   intptr_t *p = (intptr_t*)_top;
 161   char* newtop = _top + sizeof(intptr_t);
 162   expand_top_to(newtop);
 163   *p = n;
 164   if (need_to_mark) {
 165     ArchivePtrMarker::mark_pointer(p);
 166   }
 167 }

 183 void DumpRegion::init(ReservedSpace* rs, VirtualSpace* vs) {
 184   _rs = rs;
 185   _vs = vs;
 186   // Start with 0 committed bytes. The memory will be committed as needed by
 187   // MetaspaceShared::commit_to().
 188   if (!_vs->initialize(*_rs, 0)) {
 189     fatal("Unable to allocate memory for shared space");
 190   }
 191   _base = _top = _rs->base();
 192   _end = _rs->end();
 193 }

 195 void DumpRegion::pack(DumpRegion* next) {
 196   assert(!is_packed(), "sanity");
 197   _end = (char*)align_up(_top, MetaspaceShared::reserved_space_alignment());
 198   _is_packed = true;
 199   if (next != NULL) {
 200     next->_rs = _rs;
 201     next->_vs = _vs;
 202     next->_base = next->_top = this->_end;
 203     next->_end = _rs->end();
 204   }
 205 }

 260 void MetaspaceShared::initialize_dumptime_shared_and_meta_spaces() {
 261   assert(DumpSharedSpaces, "should be called for dump time only");
 262 
 263 #ifdef _LP64
 264   check_SharedBaseAddress();
 265 #endif
 266 
 267   const size_t reserve_alignment = MetaspaceShared::reserved_space_alignment();
 268   char* shared_base = (char*)align_up((char*)SharedBaseAddress, reserve_alignment);
 269 
 270 #ifdef _LP64
 271   assert(CompressedKlassPointers::is_valid_base((address)shared_base), "Sanity");
 272   // On 64-bit VM we reserve a 4G range and, if UseCompressedClassPointers=1,
 273   //  will use that to house both the archives and the ccs. See below for
 274   //  details.
 275   const uint64_t UnscaledClassSpaceMax = (uint64_t(max_juint) + 1);
 276   const size_t cds_total = align_down(UnscaledClassSpaceMax, reserve_alignment);
 277 #else
 278   // We don't support archives larger than 256MB on 32-bit due to limited
 279   //  virtual address space.
 280   size_t cds_total = align_down(256*M, reserve_alignment);
 281 #endif
 282 
 283   // Whether to use SharedBaseAddress as attach address.
 284   bool use_requested_base = true;
 285 
 286   if (shared_base == NULL) {
 287     use_requested_base = false;
 288   }
 289 
 290   if (ArchiveRelocationMode == 1) {
 291     log_info(cds)("ArchiveRelocationMode == 1: always allocate class space at an alternative address");
 292     use_requested_base = false;
 293   }
 294 
 295   // First try to reserve the space at the specified SharedBaseAddress.
 296   assert(!_shared_rs.is_reserved(), "must be");
 297   if (use_requested_base) {
 298     _shared_rs = ReservedSpace(cds_total, reserve_alignment,
 299                                false /* large */, (char*)shared_base);
 300     if (_shared_rs.is_reserved()) {
 301       assert(_shared_rs.base() == shared_base, "should match");
 302     } else {
 303       log_info(cds)("dumptime space reservation: failed to map at "
 304                     "SharedBaseAddress " PTR_FORMAT, p2i(shared_base));
 305     }
 306   }
 307   if (!_shared_rs.is_reserved()) {
 308     // Get a reserved space anywhere if attaching at the SharedBaseAddress
 309     //  fails:
 310     if (UseCompressedClassPointers) {
 311       // If we need to reserve class space as well, let the platform handle
 312       //  the reservation.
 313       LP64_ONLY(_shared_rs =
 314                 Metaspace::reserve_address_space_for_compressed_classes(cds_total);)
 315       NOT_LP64(ShouldNotReachHere();)
 316     } else {
 317       // anywhere is fine.
 318       _shared_rs = ReservedSpace(cds_total, reserve_alignment,
 319                                  false /* large */, (char*)NULL);
 320     }
 321   }
 322 
 323   if (!_shared_rs.is_reserved()) {
 324     vm_exit_during_initialization("Unable to reserve memory for shared space",
 325                                   err_msg(SIZE_FORMAT " bytes.", cds_total));
 326   }
 327 
 328 #ifdef _LP64
 329 
 330   if (UseCompressedClassPointers) {
 331 
 332     assert(CompressedKlassPointers::is_valid_base((address)_shared_rs.base()), "Sanity");
 333 
 334     // On 64-bit VM, if UseCompressedClassPointers=1, the compressed class space
 335     //  must be allocated near the cds such as that the compressed Klass pointer
 336     //  encoding can be used to en/decode pointers from both cds and ccs. Since
 337     //  Metaspace cannot do this (it knows nothing about cds), we do it for
 338     //  Metaspace here and pass it the space to use for ccs.
 339     //
 340     // We do this by reserving space for the ccs behind the archives. Note
 341     //  however that ccs follows a different alignment
 342     //  (Metaspace::reserve_alignment), so there may be a gap between ccs and
 343     //  cds.
 344     // We use a similar layout at runtime, see reserve_address_space_for_archives().
 345     //
 346     //                              +-- SharedBaseAddress (default = 0x800000000)
 347     //                              v
 348     // +-..---------+---------+ ... +----+----+----+--------+-----------------+
 349     // |    Heap    | Archive |     | MC | RW | RO | [gap]  |    class space  |
 350     // +-..---------+---------+ ... +----+----+----+--------+-----------------+
 351     // |<--   MaxHeapSize  -->|     |<-- UnscaledClassSpaceMax = 4GB -->|
 352     //
 353     // Note: ccs must follow the archives, and the archives must start at the
 354     //  encoding base. However, the exact placement of ccs does not matter as
 355     //  long as it it resides in the encoding range of CompressedKlassPointers
 356     //  and comes after the archive.
 357     //
 358     // We do this by splitting up the allocated 4G into 3G of archive space,
 359     //  followed by 1G for the ccs:
 360     // + The upper 1 GB is used as the "temporary compressed class space"
 361     //   -- preload_classes() will store Klasses into this space.
 362     // + The lower 3 GB is used for the archive -- when preload_classes()
 363     //   is done, ArchiveCompactor will copy the class metadata into this
 364     //   space, first the RW parts, then the RO parts.
 365 
 366     // Starting address of ccs must be aligned to Metaspace::reserve_alignment()...
 367     size_t class_space_size = align_down(_shared_rs.size() / 4, Metaspace::reserve_alignment());
 368     address class_space_start = (address)align_down(_shared_rs.end() - class_space_size, Metaspace::reserve_alignment());
 369     size_t archive_size = class_space_start - (address)_shared_rs.base();
 370 
 371     ReservedSpace tmp_class_space = _shared_rs.last_part(archive_size);
 372     _shared_rs = _shared_rs.first_part(archive_size);
 373 
 374     // ... as does the size of ccs.
 375     tmp_class_space = tmp_class_space.first_part(class_space_size);
 376     CompressedClassSpaceSize = class_space_size;
 377 
 378     // Let Metaspace initialize ccs
 379     Metaspace::initialize_class_space(tmp_class_space);
 380 
 381     // and set up CompressedKlassPointers encoding.
 382     CompressedKlassPointers::initialize((address)_shared_rs.base(), cds_total);
 383 
 384     log_info(cds)("narrow_klass_base = " PTR_FORMAT ", narrow_klass_shift = %d",
 385                   p2i(CompressedKlassPointers::base()), CompressedKlassPointers::shift());
 386 
 387     log_info(cds)("Allocated temporary class space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 388                   CompressedClassSpaceSize, p2i(tmp_class_space.base()));
 389 
 390     assert(_shared_rs.end() == tmp_class_space.base() &&
 391            is_aligned(_shared_rs.base(), MetaspaceShared::reserved_space_alignment()) &&
 392            is_aligned(tmp_class_space.base(), Metaspace::reserve_alignment()) &&
 393            is_aligned(tmp_class_space.size(), Metaspace::reserve_alignment()), "Sanity");
 394   }
 395 
 396 #endif
 397 
 398   init_shared_dump_space(&_mc_region);
 399   SharedBaseAddress = (size_t)_shared_rs.base();
 400   log_info(cds)("Allocated shared space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 401                 _shared_rs.size(), p2i(_shared_rs.base()));
 402 
 403   size_t symbol_rs_size = LP64_ONLY(3 * G) NOT_LP64(128 * M);
 404   _symbol_rs = ReservedSpace(symbol_rs_size);
 405   if (!_symbol_rs.is_reserved()) {
 406     vm_exit_during_initialization("Unable to reserve memory for symbols",
 407                                   err_msg(SIZE_FORMAT " bytes.", symbol_rs_size));
 408   }
 409   _symbol_region.init(&_symbol_rs, &_symbol_vs);
 410 }

 432 static GrowableArray<Handle>* _extra_interned_strings = NULL;
 433 
 434 void MetaspaceShared::read_extra_data(const char* filename, TRAPS) {
 435   _extra_interned_strings = new (ResourceObj::C_HEAP, mtInternal)GrowableArray<Handle>(10000, true);
 436 
 437   HashtableTextDump reader(filename);
 438   reader.check_version("VERSION: 1.0");
 439 
 440   while (reader.remain() > 0) {
 441     int utf8_length;
 442     int prefix_type = reader.scan_prefix(&utf8_length);
 443     ResourceMark rm(THREAD);
 444     if (utf8_length == 0x7fffffff) {
 445       // buf_len will overflown 32-bit value.
 446       vm_exit_during_initialization(err_msg("string length too large: %d", utf8_length));
 447     }
 448     int buf_len = utf8_length+1;
 449     char* utf8_buffer = NEW_RESOURCE_ARRAY(char, buf_len);
 450     reader.get_utf8(utf8_buffer, utf8_length);
 451     utf8_buffer[utf8_length] = '\0';
 452 
 453     if (prefix_type == HashtableTextDump::SymbolPrefix) {
 454       SymbolTable::new_permanent_symbol(utf8_buffer);
 455     } else{
 456       assert(prefix_type == HashtableTextDump::StringPrefix, "Sanity");
 457       oop s = StringTable::intern(utf8_buffer, THREAD);
 458 
 459       if (HAS_PENDING_EXCEPTION) {
 460         log_warning(cds, heap)("[line %d] extra interned string allocation failed; size too large: %d",
 461                                reader.last_line_no(), utf8_length);
 462         CLEAR_PENDING_EXCEPTION;
 463       } else {
 464 #if INCLUDE_G1GC
 465         if (UseG1GC) {
 466           typeArrayOop body = java_lang_String::value(s);
 467           const HeapRegion* hr = G1CollectedHeap::heap()->heap_region_containing(body);
 468           if (hr->is_humongous()) {
 469             // Don't keep it alive, so it will be GC'ed before we dump the strings, in order
 470             // to maximize free heap space and minimize fragmentation.
 471             log_warning(cds, heap)("[line %d] extra interned string ignored; size too large: %d",
 472                                 reader.last_line_no(), utf8_length);
 473             continue;
 474           }
 475         }
 476 #endif
 477         // Interned strings are GC'ed if there are no references to it, so let's
 478         // add a reference to keep this string alive.
 479         assert(s != NULL, "must succeed");
 480         Handle h(THREAD, s);
 481         _extra_interned_strings->append(h);
 482       }
 483     }
 484   }
 485 }

 487 void MetaspaceShared::commit_to(ReservedSpace* rs, VirtualSpace* vs, char* newtop) {
 488   Arguments::assert_is_dumping_archive();
 489   char* base = rs->base();
 490   size_t need_committed_size = newtop - base;
 491   size_t has_committed_size = vs->committed_size();
 492   if (need_committed_size < has_committed_size) {
 493     return;
 494   }
 495 
 496   size_t min_bytes = need_committed_size - has_committed_size;
 497   size_t preferred_bytes = 1 * M;
 498   size_t uncommitted = vs->reserved_size() - has_committed_size;
 499 
 500   size_t commit =MAX2(min_bytes, preferred_bytes);
 501   commit = MIN2(commit, uncommitted);
 502   assert(commit <= uncommitted, "sanity");
 503 
 504   bool result = vs->expand_by(commit, false);
 505   if (rs == &_shared_rs) {
 506     ArchivePtrMarker::expand_ptr_end((address*)vs->high());
 507   }
 508 
 509   if (!result) {
 510     vm_exit_during_initialization(err_msg("Failed to expand shared space to " SIZE_FORMAT " bytes",
 511                                           need_committed_size));
 512   }
 513 
 514   assert(rs == &_shared_rs || rs == &_symbol_rs, "must be");
 515   const char* which = (rs == &_shared_rs) ? "shared" : "symbol";
 516   log_debug(cds)("Expanding %s spaces by " SIZE_FORMAT_W(7) " bytes [total " SIZE_FORMAT_W(9)  " bytes ending at %p]",
 517                  which, commit, vs->actual_committed_size(), vs->high());
 518 }

 524 // Read/write a data stream for restoring/preserving metadata pointers and
 525 // miscellaneous data from/to the shared archive file.
 526 
 527 void MetaspaceShared::serialize(SerializeClosure* soc) {
 528   int tag = 0;
 529   soc->do_tag(--tag);
 530 
 531   // Verify the sizes of various metadata in the system.
 532   soc->do_tag(sizeof(Method));
 533   soc->do_tag(sizeof(ConstMethod));
 534   soc->do_tag(arrayOopDesc::base_offset_in_bytes(T_BYTE));
 535   soc->do_tag(sizeof(ConstantPool));
 536   soc->do_tag(sizeof(ConstantPoolCache));
 537   soc->do_tag(objArrayOopDesc::base_offset_in_bytes());
 538   soc->do_tag(typeArrayOopDesc::base_offset_in_bytes(T_BYTE));
 539   soc->do_tag(sizeof(Symbol));
 540 
 541   // Dump/restore miscellaneous metadata.
 542   JavaClasses::serialize_offsets(soc);
 543   Universe::serialize(soc);
 544   soc->do_tag(--tag);
 545 
 546   // Dump/restore references to commonly used names and signatures.
 547   vmSymbols::serialize(soc);
 548   soc->do_tag(--tag);
 549 
 550   // Dump/restore the symbol/string/subgraph_info tables

 683     case Bytecodes::_aload_0:       *bcs.bcp() = Bytecodes::_nofast_aload_0;       break;

 685       if (!bcs.is_wide()) {
 686         *bcs.bcp() = Bytecodes::_nofast_iload;
 687       }
 688       break;

 691     }

 720 // Objects of the Metadata types (such as Klass and ConstantPool) have C++ vtables.
 721 // (In GCC this is the field <Type>::_vptr, i.e., first word in the object.)
 722 //
 723 // Addresses of the vtables and the methods may be different across JVM runs,
 724 // if libjvm.so is dynamically loaded at a different base address.
 725 //
 726 // To ensure that the Metadata objects in the CDS archive always have the correct vtable:
 727 //
 728 // + at dump time:  we redirect the _vptr to point to our own vtables inside
 729 //                  the CDS image
 730 // + at run time:   we clone the actual contents of the vtables from libjvm.so
 731 //                  into our own tables.
 732 
 733 // Currently, the archive contain ONLY the following types of objects that have C++ vtables.
 734 #define CPP_VTABLE_PATCH_TYPES_DO(f) \
 735   f(ConstantPool) \
 736   f(InstanceClassLoaderKlass) \
 737   f(InstanceKlass) \
 738   f(InstanceMirrorKlass) \
 739   f(InstanceRefKlass) \
 740   f(Method) \
 741   f(ObjArrayKlass) \
 742   f(TypeArrayKlass) \
 743   f(ValueArrayKlass) \
 744   f(ValueKlass)

 746 class CppVtableInfo {
 747   intptr_t _vtable_size;
 748   intptr_t _cloned_vtable[1];
 749 public:
 750   static int num_slots(int vtable_size) {
 751     return 1 + vtable_size; // Need to add the space occupied by _vtable_size;
 752   }
 753   int vtable_size()           { return int(uintx(_vtable_size)); }
 754   void set_vtable_size(int n) { _vtable_size = intptr_t(n); }
 755   intptr_t* cloned_vtable()   { return &_cloned_vtable[0]; }
 756   void zero()                 { memset(_cloned_vtable, 0, sizeof(intptr_t) * vtable_size()); }
 757   // Returns the address of the next CppVtableInfo that can be placed immediately after this CppVtableInfo
 758   static size_t byte_size(int vtable_size) {
 759     CppVtableInfo i;
 760     return pointer_delta(&i._cloned_vtable[vtable_size], &i, sizeof(u1));
 761   }
 762 };

 805 template <class T>
 806 intptr_t* CppVtableCloner<T>::clone_vtable(const char* name, CppVtableInfo* info) {
 807   if (!DumpSharedSpaces) {
 808     assert(_info == 0, "_info is initialized only at dump time");
 809     _info = info; // Remember it -- it will be used by MetaspaceShared::is_valid_shared_method()
 810   }
 811   T tmp; // Allocate temporary dummy metadata object to get to the original vtable.
 812   int n = info->vtable_size();
 813   intptr_t* srcvtable = vtable_of(tmp);
 814   intptr_t* dstvtable = info->cloned_vtable();
 815 
 816   // We already checked (and, if necessary, adjusted n) when the vtables were allocated, so we are
 817   // safe to do memcpy.
 818   log_debug(cds, vtables)("Copying %3d vtable entries for %s", n, name);
 819   memcpy(dstvtable, srcvtable, sizeof(intptr_t) * n);
 820   return dstvtable + n;

 823 // To determine the size of the vtable for each type, we use the following
 824 // trick by declaring 2 subclasses:
 825 //
 826 //   class CppVtableTesterA: public InstanceKlass {virtual int   last_virtual_method() {return 1;}    };
 827 //   class CppVtableTesterB: public InstanceKlass {virtual void* last_virtual_method() {return NULL}; };
 828 //
 829 // CppVtableTesterA and CppVtableTesterB's vtables have the following properties:
 830 // - Their size (N+1) is exactly one more than the size of InstanceKlass's vtable (N)
 831 // - The first N entries have are exactly the same as in InstanceKlass's vtable.
 832 // - Their last entry is different.
 833 //
 834 // So to determine the value of N, we just walk CppVtableTesterA and CppVtableTesterB's tables
 835 // and find the first entry that's different.
 836 //
 837 // This works on all C++ compilers supported by Oracle, but you may need to tweak it for more
 838 // esoteric compilers.
 839 
 840 template <class T> class CppVtableTesterB: public T {
 841 public:
 842   virtual int last_virtual_method() {return 1;}
 843 };
 844 
 845 template <class T> class CppVtableTesterA : public T {
 846 public:
 847   virtual void* last_virtual_method() {
 848     // Make this different than CppVtableTesterB::last_virtual_method so the C++
 849     // compiler/linker won't alias the two functions.
 850     return NULL;
 851   }
 852 };
 853 
 854 template <class T>
 855 int CppVtableCloner<T>::get_vtable_length(const char* name) {
 856   CppVtableTesterA<T> a;
 857   CppVtableTesterB<T> b;
 858 
 859   intptr_t* avtable = vtable_of(a);
 860   intptr_t* bvtable = vtable_of(b);
 861 
 862   // Start at slot 1, because slot 0 may be RTTI (on Solaris/Sparc)
 863   int vtable_len = 1;
 864   for (; ; vtable_len++) {
 865     if (avtable[vtable_len] != bvtable[vtable_len]) {
 866       break;
 867     }
 868   }
 869   log_debug(cds, vtables)("Found   %3d vtable entries for %s", vtable_len, name);
 870 
 871   return vtable_len;
 872 }

1488   static void iterate_roots(MetaspaceClosure* it) {
1489     // To ensure deterministic contents in the archive, we just need to ensure that
1490     // we iterate the MetsapceObjs in a deterministic order. It doesn't matter where
1491     // the MetsapceObjs are located originally, as they are copied sequentially into
1492     // the archive during the iteration.
1493     //
1494     // The only issue here is that the symbol table and the system directories may be
1495     // randomly ordered, so we copy the symbols and klasses into two arrays and sort
1496     // them deterministically.
1497     //
1498     // During -Xshare:dump, the order of Symbol creation is strictly determined by
1499     // the SharedClassListFile (class loading is done in a single thread and the JIT
1500     // is disabled). Also, Symbols are allocated in monotonically increasing addresses
1501     // (see Symbol::operator new(size_t, int)). So if we iterate the Symbols by
1502     // ascending address order, we ensure that all Symbols are copied into deterministic
1503     // locations in the archive.
1504     GrowableArray<Symbol*>* symbols = _ssc->get_sorted_symbols();

1506       it->push(symbols->adr_at(i));

1509       // Need to fix up the pointers
1510       for (int i = 0; i < _global_klass_objects->length(); i++) {
1511         // NOTE -- this requires that the vtable is NOT yet patched, or else we are hosed.
1512         it->push(_global_klass_objects->adr_at(i));
1513       }

1516     SystemDictionaryShared::dumptime_classes_do(it);

1591 void VM_PopulateDumpSharedSpace::relocate_to_default_base_address(CHeapBitMap* ptrmap) {
1592   intx addr_delta = MetaspaceShared::final_delta();
1593   if (addr_delta == 0) {
1594     ArchivePtrMarker::compact((address)SharedBaseAddress, (address)_ro_region.top());
1595   } else {
1596     // We are not able to reserve space at Arguments::default_SharedBaseAddress() (due to ASLR).
1597     // This means that the current content of the archive is based on a random
1598     // address. Let's relocate all the pointers, so that it can be mapped to
1599     // Arguments::default_SharedBaseAddress() without runtime relocation.
1600     //
1601     // Note: both the base and dynamic archive are written with
1602     // FileMapHeader::_shared_base_address == Arguments::default_SharedBaseAddress()
1603 
1604     // Patch all pointers that are marked by ptrmap within this region,
1605     // where we have just dumped all the metaspace data.
1606     address patch_base = (address)SharedBaseAddress;
1607     address patch_end  = (address)_ro_region.top();
1608     size_t size = patch_end - patch_base;
1609 
1610     // the current value of the pointers to be patched must be within this
1611     // range (i.e., must point to valid metaspace objects)
1612     address valid_old_base = patch_base;
1613     address valid_old_end  = patch_end;
1614 
1615     // after patching, the pointers must point inside this range
1616     // (the requested location of the archive, as mapped at runtime).
1617     address valid_new_base = (address)Arguments::default_SharedBaseAddress();
1618     address valid_new_end  = valid_new_base + size;
1619 
1620     log_debug(cds)("Relocating archive from [" INTPTR_FORMAT " - " INTPTR_FORMAT " ] to "
1621                    "[" INTPTR_FORMAT " - " INTPTR_FORMAT " ]", p2i(patch_base), p2i(patch_end),
1622                    p2i(valid_new_base), p2i(valid_new_end));
1623 
1624     SharedDataRelocator<true> patcher((address*)patch_base, (address*)patch_end, valid_old_base, valid_old_end,
1625                                       valid_new_base, valid_new_end, addr_delta, ptrmap);
1626     ptrmap->iterate(&patcher);
1627     ArchivePtrMarker::compact(patcher.max_non_null_offset());
1628   }
1629 }

1631 void VM_PopulateDumpSharedSpace::doit() {
1632   CHeapBitMap ptrmap;
1633   MetaspaceShared::initialize_ptr_marker(&ptrmap);
1634 
1635   // We should no longer allocate anything from the metaspace, so that:
1636   //
1637   // (1) Metaspace::allocate might trigger GC if we have run out of
1638   //     committed metaspace, but we can't GC because we're running
1639   //     in the VM thread.
1640   // (2) ArchiveCompactor needs to work with a stable set of MetaspaceObjs.
1641   Metaspace::freeze();
1642   DEBUG_ONLY(SystemDictionaryShared::NoClassLoadingMark nclm);
1643 
1644   Thread* THREAD = VMThread::vm_thread();
1645 
1646   FileMapInfo::check_nonempty_dir_in_shared_path_table();
1647 
1648   NOT_PRODUCT(SystemDictionary::verify();)
1649   // The following guarantee is meant to ensure that no loader constraints
1650   // exist yet, since the constraints table is not shared.  This becomes
1651   // more important now that we don't re-initialize vtables/itables for
1652   // shared classes at runtime, where constraints were previously created.
1653   guarantee(SystemDictionary::constraints()->number_of_entries() == 0,
1654             "loader constraints are not saved");
1655   guarantee(SystemDictionary::placeholders()->number_of_entries() == 0,
1656           "placeholders are not saved");
1657 
1658   // At this point, many classes have been loaded.
1659   // Gather systemDictionary classes in a global array and do everything to
1660   // that so we don't have to walk the SystemDictionary again.
1661   SystemDictionaryShared::check_excluded_classes();

1963 int MetaspaceShared::preload_classes(const char* class_list_path, TRAPS) {
1964   ClassListParser parser(class_list_path);
1965   int class_count = 0;
1966 
1967   while (parser.parse_one_line()) {
1968     Klass* klass = parser.load_current_class(THREAD);
1969     if (HAS_PENDING_EXCEPTION) {
1970       if (klass == NULL &&
1971           (PENDING_EXCEPTION->klass()->name() == vmSymbols::java_lang_ClassNotFoundException())) {
1972         // print a warning only when the pending exception is class not found
1973         log_warning(cds)("Preload Warning: Cannot find %s", parser.current_class_name());
1974       }
1975       CLEAR_PENDING_EXCEPTION;
1976     }
1977     if (klass != NULL) {
1978       if (log_is_enabled(Trace, cds)) {
1979         ResourceMark rm(THREAD);
1980         log_trace(cds)("Shared spaces preloaded: %s", klass->external_name());
1981       }
1982 
1983       if (klass->is_instance_klass()) {
1984         InstanceKlass* ik = InstanceKlass::cast(klass);
1985 
1986         // Link the class to cause the bytecodes to be rewritten and the
1987         // cpcache to be created. The linking is done as soon as classes
1988         // are loaded in order that the related data structures (klass and
1989         // cpCache) are located together.
1990         try_link_class(ik, THREAD);
1991         guarantee(!HAS_PENDING_EXCEPTION, "exception in link_class");
1992       }
1993 
1994       class_count++;
1995     }
1996   }
1997 
1998   return class_count;
1999 }

2001 // Returns true if the class's status has changed
2002 bool MetaspaceShared::try_link_class(InstanceKlass* ik, TRAPS) {
2003   Arguments::assert_is_dumping_archive();
2004   if (ik->init_state() < InstanceKlass::linked &&
2005       !SystemDictionaryShared::has_class_failed_verification(ik)) {
2006     bool saved = BytecodeVerificationLocal;
2007     if (ik->is_shared_unregistered_class() && ik->class_loader() == NULL) {
2008       // The verification decision is based on BytecodeVerificationRemote
2009       // for non-system classes. Since we are using the NULL classloader
2010       // to load non-system classes for customized class loaders during dumping,
2011       // we need to temporarily change BytecodeVerificationLocal to be the same as
2012       // BytecodeVerificationRemote. Note this can cause the parent system
2013       // classes also being verified. The extra overhead is acceptable during
2014       // dumping.
2015       BytecodeVerificationLocal = BytecodeVerificationRemote;
2016     }
2017     ik->link_class(THREAD);
2018     if (HAS_PENDING_EXCEPTION) {
2019       ResourceMark rm(THREAD);
2020       log_warning(cds)("Preload Warning: Verification failed for %s",
2021                     ik->external_name());
2022       CLEAR_PENDING_EXCEPTION;
2023       SystemDictionaryShared::set_class_has_failed_verification(ik);
2024       _has_error_classes = true;
2025     }
2026     BytecodeVerificationLocal = saved;
2027     return true;
2028   } else {
2029     return false;
2030   }
2031 }

2375 // This will reserve two address spaces suitable to house Klass structures, one
2376 //  for the cds archives (static archive and optionally dynamic archive) and
2377 //  optionally one move for ccs.
2378 //
2379 // Since both spaces must fall within the compressed class pointer encoding
2380 //  range, they are allocated close to each other.
2381 //
2382 // Space for archives will be reserved first, followed by a potential gap,
2383 //  followed by the space for ccs:
2384 //
2385 // +-- Base address             A        B                     End
2386 // |                            |        |                      |
2387 // v                            v        v                      v
2388 // +-------------+--------------+        +----------------------+
2389 // | static arc  | [dyn. arch]  | [gap]  | compr. class space   |
2390 // +-------------+--------------+        +----------------------+
2391 //
2392 // (The gap may result from different alignment requirements between metaspace
2393 //  and CDS)
2394 //
2395 // If UseCompressedClassPointers is disabled, only one address space will be
2396 //  reserved:
2397 //
2398 // +-- Base address             End
2399 // |                            |
2400 // v                            v
2401 // +-------------+--------------+
2402 // | static arc  | [dyn. arch]  |
2403 // +-------------+--------------+
2404 //
2405 // Base address: If use_archive_base_addr address is true, the Base address is
2406 //  determined by the address stored in the static archive. If
2407 //  use_archive_base_addr address is false, this base address is determined
2408 //  by the platform.
2409 //
2410 // If UseCompressedClassPointers=1, the range encompassing both spaces will be
2411 //  suitable to en/decode narrow Klass pointers: the base will be valid for
2412 //  encoding, the range [Base, End) not surpass KlassEncodingMetaspaceMax.
2413 //
2414 // Return:
2415 //
2416 // - On success:
2417 //    - archive_space_rs will be reserved and large enough to host static and
2418 //      if needed dynamic archive: [Base, A).
2419 //      archive_space_rs.base and size will be aligned to CDS reserve
2420 //      granularity.
2421 //    - class_space_rs: If UseCompressedClassPointers=1, class_space_rs will
2422 //      be reserved. Its start address will be aligned to metaspace reserve
2423 //      alignment, which may differ from CDS alignment. It will follow the cds
2424 //      archive space, close enough such that narrow class pointer encoding
2425 //      covers both spaces.
2426 //      If UseCompressedClassPointers=0, class_space_rs remains unreserved.
2427 // - On error: NULL is returned and the spaces remain unreserved.
2428 char* MetaspaceShared::reserve_address_space_for_archives(FileMapInfo* static_mapinfo,
2429                                                           FileMapInfo* dynamic_mapinfo,
2430                                                           bool use_archive_base_addr,
2431                                                           ReservedSpace& archive_space_rs,
2432                                                           ReservedSpace& class_space_rs) {
2433 
2434   address const base_address = (address) (use_archive_base_addr ? static_mapinfo->requested_base_address() : NULL);
2435   const size_t archive_space_alignment = MetaspaceShared::reserved_space_alignment();
2436 
2437   // Size and requested location of the archive_space_rs (for both static and dynamic archives)
2438   assert(static_mapinfo->mapping_base_offset() == 0, "Must be");
2439   size_t archive_end_offset  = (dynamic_mapinfo == NULL) ? static_mapinfo->mapping_end_offset() : dynamic_mapinfo->mapping_end_offset();
2440   size_t archive_space_size = align_up(archive_end_offset, archive_space_alignment);
2441 
2442   // If a base address is given, it must have valid alignment and be suitable as encoding base.
2443   if (base_address != NULL) {
2444     assert(is_aligned(base_address, archive_space_alignment),
2445            "Archive base address invalid: " PTR_FORMAT ".", p2i(base_address));
2446     if (Metaspace::using_class_space()) {
2447       assert(CompressedKlassPointers::is_valid_base(base_address),
2448              "Archive base address invalid: " PTR_FORMAT ".", p2i(base_address));
2449     }
2450   }
2451 
2452   if (!Metaspace::using_class_space()) {
2453     // Get the simple case out of the way first:
2454     // no compressed class space, simple allocation.
2455     archive_space_rs = ReservedSpace(archive_space_size, archive_space_alignment,
2456                                      false /* bool large */, (char*)base_address);
2457     if (archive_space_rs.is_reserved()) {
2458       assert(base_address == NULL ||
2459              (address)archive_space_rs.base() == base_address, "Sanity");
2460       return archive_space_rs.base();
2461     }
2462     return NULL;
2463   }
2464 
2465 #ifdef _LP64
2466 
2467   // Complex case: two spaces adjacent to each other, both to be addressable
2468   //  with narrow class pointers.
2469   // We reserve the whole range spanning both spaces, then split that range up.
2470 
2471   const size_t class_space_alignment = Metaspace::reserve_alignment();
2472 
2473   // To simplify matters, lets assume that metaspace alignment will always be
2474   //  equal or a multiple of archive alignment.
2475   assert(is_power_of_2(class_space_alignment) &&
2476                        is_power_of_2(archive_space_alignment) &&
2477                        class_space_alignment >= archive_space_alignment,
2478                        "Sanity");
2479 
2480   const size_t class_space_size = CompressedClassSpaceSize;
2481   assert(CompressedClassSpaceSize > 0 &&
2482          is_aligned(CompressedClassSpaceSize, class_space_alignment),
2483          "CompressedClassSpaceSize malformed: "
2484          SIZE_FORMAT, CompressedClassSpaceSize);
2485 
2486   const size_t ccs_begin_offset = align_up(archive_space_size,
2487                                            class_space_alignment);
2488   const size_t gap_size = ccs_begin_offset - archive_space_size;
2489 
2490   const size_t total_range_size =
2491       align_up(archive_space_size + gap_size + class_space_size,
2492                os::vm_allocation_granularity());
2493 
2494   ReservedSpace total_rs;
2495   if (base_address != NULL) {
2496     // Reserve at the given archive base address, or not at all.
2497     total_rs = ReservedSpace(total_range_size, archive_space_alignment,
2498                              false /* bool large */, (char*) base_address);
2499   } else {
2500     // Reserve at any address, but leave it up to the platform to choose a good one.
2501     total_rs = Metaspace::reserve_address_space_for_compressed_classes(total_range_size);
2502   }
2503 
2504   if (!total_rs.is_reserved()) {
2505     return NULL;
2506   }
2507 
2508   // Paranoid checks:
2509   assert(base_address == NULL || (address)total_rs.base() == base_address,
2510          "Sanity (" PTR_FORMAT " vs " PTR_FORMAT ")", p2i(base_address), p2i(total_rs.base()));
2511   assert(is_aligned(total_rs.base(), archive_space_alignment), "Sanity");
2512   assert(total_rs.size() == total_range_size, "Sanity");
2513   assert(CompressedKlassPointers::is_valid_base((address)total_rs.base()), "Sanity");
2514 
2515   // Now split up the space into ccs and cds archive. For simplicity, just leave
2516   //  the gap reserved at the end of the archive space.
2517   archive_space_rs = total_rs.first_part(ccs_begin_offset,
2518                                          (size_t)os::vm_allocation_granularity(),
2519                                          /*split=*/true);
2520   class_space_rs = total_rs.last_part(ccs_begin_offset);
2521 
2522   assert(is_aligned(archive_space_rs.base(), archive_space_alignment), "Sanity");
2523   assert(is_aligned(archive_space_rs.size(), archive_space_alignment), "Sanity");
2524   assert(is_aligned(class_space_rs.base(), class_space_alignment), "Sanity");
2525   assert(is_aligned(class_space_rs.size(), class_space_alignment), "Sanity");
2526 
2527   return archive_space_rs.base();
2528 
2529 #else
2530   ShouldNotReachHere();
2531   return NULL;
2532 #endif
2533
