 329     /** Report duplicate declaration error.
 330      */
 331     void duplicateError(DiagnosticPosition pos, Symbol sym) {
 332         if (!sym.type.isErroneous()) {
 333             Symbol location = sym.location();
 334             if (location.kind == MTH &&
 335                     ((MethodSymbol)location).isStaticOrInstanceInit()) {
 336                 log.error(pos,
 337                           Errors.AlreadyDefinedInClinit(kindName(sym),
 338                                                         sym,
 339                                                         kindName(sym.location()),
 340                                                         kindName(sym.location().enclClass()),
 341                                                         sym.location().enclClass()));
 342             } else {
 343                 /* dont error if this is a duplicated parameter of a generated canonical constructor
 344                  * as we should have issued an error for the duplicated fields
 345                  */
 346                 if (location.kind != MTH ||
 347                         ((sym.owner.flags_field & GENERATEDCONSTR) == 0) ||
 348                         ((sym.owner.flags_field & RECORD) == 0)) {
 349                     log.error(pos,
 350                             Errors.AlreadyDefined(kindName(sym),
 351                                     sym,
 352                                     kindName(sym.location()),
 353                                     sym.location()));
 354                 }
 355             }
 356         }
 357     }

 371     /** Check that variable does not hide variable with same name in
 372      *  immediately enclosing local scope.
 373      *  @param pos           Position for error reporting.
 374      *  @param v             The symbol.
 375      *  @param s             The scope.
 376      */
 377     void checkTransparentVar(DiagnosticPosition pos, VarSymbol v, Scope s) {
 378         for (Symbol sym : s.getSymbolsByName(v.name)) {
 379             if (sym.owner != v.owner) break;
 380             if (sym.kind == VAR &&
 381                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 382                 v.name != names.error) {
 383                 duplicateError(pos, sym);
 384                 return;
 385             }
 386         }
 387     }
 388 
 389     /** Check that a class or interface does not hide a class or
 390      *  interface with same name in immediately enclosing local scope.
 391      *  @param pos           Position for error reporting.
 392      *  @param c             The symbol.
 393      *  @param s             The scope.
 394      */
 395     void checkTransparentClass(DiagnosticPosition pos, ClassSymbol c, Scope s) {
 396         for (Symbol sym : s.getSymbolsByName(c.name)) {
 397             if (sym.owner != c.owner) break;
 398             if (sym.kind == TYP && !sym.type.hasTag(TYPEVAR) &&
 399                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 400                 c.name != names.error) {
 401                 duplicateError(pos, sym);
 402                 return;
 403             }
 404         }
 405     }

 407     /** Check that class does not have the same name as one of
 408      *  its enclosing classes, or as a class defined in its enclosing scope.
 409      *  return true if class is unique in its enclosing scope.
 410      *  @param pos           Position for error reporting.
 411      *  @param name          The class name.
 412      *  @param s             The enclosing scope.
 413      */
 414     boolean checkUniqueClassName(DiagnosticPosition pos, Name name, Scope s) {
 415         for (Symbol sym : s.getSymbolsByName(name, NON_RECURSIVE)) {
 416             if (sym.kind == TYP && sym.name != names.error) {
 417                 duplicateError(pos, sym);
 418                 return false;
 419             }
 420         }
 421         for (Symbol sym = s.owner; sym != null; sym = sym.owner) {
 422             if (sym.kind == TYP && sym.name == name && sym.name != names.error) {
 423                 duplicateError(pos, sym);
 424                 return true;
 425             }
 426         }
 427         return true;
 428     }

 437     /** Return name of local class.
 438      *  This is of the form   {@code <enclClass> $ n <classname> }
 439      *  where
 440      *    enclClass is the flat name of the enclosing class,
 441      *    classname is the simple name of the local class
 442      */
 443     public Name localClassName(ClassSymbol c) {
 444         Name enclFlatname = c.owner.enclClass().flatname;
 445         String enclFlatnameStr = enclFlatname.toString();
 446         Pair<Name, Name> key = new Pair<>(enclFlatname, c.name);
 447         Integer index = localClassNameIndexes.get(key);
 448         for (int i = (index == null) ? 1 : index; ; i++) {
 449             Name flatname = names.fromString(enclFlatnameStr
 450                     + syntheticNameChar + i + c.name);
 451             if (getCompiled(c.packge().modle, flatname) == null) {
 452                 localClassNameIndexes.put(key, i + 1);
 453                 return flatname;
 454             }
 455         }
 456     }

 458     public void clearLocalClassNameIndexes(ClassSymbol c) {
 459         if (c.owner != null && c.owner.kind != NIL) {
 460             localClassNameIndexes.remove(new Pair<>(
 461                     c.owner.enclClass().flatname, c.name));
 462         }
 463     }

 594     Type checkType(final DiagnosticPosition pos, final Type found, final Type req, final CheckContext checkContext) {
 595         final InferenceContext inferenceContext = checkContext.inferenceContext();
 596         if (inferenceContext.free(req) || inferenceContext.free(found)) {
 597             inferenceContext.addFreeTypeListener(List.of(req, found),
 598                     solvedContext -> checkType(pos, solvedContext.asInstType(found), solvedContext.asInstType(req), checkContext));
 599         }
 600         if (req.hasTag(ERROR))
 601             return req;
 602         if (req.hasTag(NONE))
 603             return found;
 604         if (checkContext.compatible(found, req, checkContext.checkWarner(pos, found, req))) {
 605             return found;
 606         } else {
 607             if (found.isNumeric() && req.isNumeric()) {
 608                 checkContext.report(pos, diags.fragment(Fragments.PossibleLossOfPrecision(found, req)));
 609                 return types.createErrorType(found);
 610             }
 611             checkContext.report(pos, diags.fragment(Fragments.InconvertibleTypes(found, req)));
 612             return types.createErrorType(found);
 613         }
 614     }

 634     /** Check for redundant casts (i.e. where source type is a subtype of target type)
 635      * The problem should only be reported for non-292 cast
 636      */
 637     public void checkRedundantCast(Env<AttrContext> env, final JCTypeCast tree) {
 638         if (!tree.type.isErroneous()
 639                 && types.isSameType(tree.expr.type, tree.clazz.type)
 640                 && !(ignoreAnnotatedCasts && TreeInfo.containsTypeAnnotation(tree.clazz))
 641                 && !is292targetTypeCast(tree)) {

 981     Type checkMethod(final Type mtype,
 982             final Symbol sym,
 983             final Env<AttrContext> env,
 984             final List<JCExpression> argtrees,
 985             final List<Type> argtypes,
 986             final boolean useVarargs,
 987             InferenceContext inferenceContext) {
 988         // System.out.println("call   : " + env.tree);
 989         // System.out.println("method : " + owntype);
 990         // System.out.println("actuals: " + argtypes);
 991         if (inferenceContext.free(mtype)) {
 992             inferenceContext.addFreeTypeListener(List.of(mtype),
 993                     solvedContext -> checkMethod(solvedContext.asInstType(mtype), sym, env, argtrees, argtypes, useVarargs, solvedContext));
 994             return mtype;
 995         }
 996         Type owntype = mtype;
 997         List<Type> formals = owntype.getParameterTypes();
 998         List<Type> nonInferred = sym.type.getParameterTypes();
 999         if (nonInferred.length() != formals.length()) nonInferred = formals;
1000         Type last = useVarargs ? formals.last() : null;
1001         if (sym.name == names.init && sym.owner == syms.enumSym) {
1002             formals = formals.tail.tail;
1003             nonInferred = nonInferred.tail.tail;
1004         }
1005         if ((sym.flags() & ANONCONSTR_BASED) != 0) {
1006             formals = formals.tail;
1007             nonInferred = nonInferred.tail;
1008         }
1009         List<JCExpression> args = argtrees;
1010         if (args != null) {
1011             //this is null when type-checking a method reference
1012             while (formals.head != last) {
1013                 JCTree arg = args.head;
1014                 Warner warn = convertWarner(arg.pos(), arg.type, nonInferred.head);
1015                 assertConvertible(arg, arg.type, formals.head, warn);
1016                 args = args.tail;
1017                 formals = formals.tail;
1018                 nonInferred = nonInferred.tail;
1019             }
1020             if (useVarargs) {
1021                 Type varArg = types.elemtype(last);
1022                 while (args.tail != null) {
1023                     JCTree arg = args.head;
1024                     Warner warn = convertWarner(arg.pos(), arg.type, varArg);
1025                     assertConvertible(arg, arg.type, varArg, warn);
1026                     args = args.tail;
1027                 }
1028             } else if ((sym.flags() & (VARARGS | SIGNATURE_POLYMORPHIC)) == VARARGS) {
1029                 // non-varargs call to varargs method
1030                 Type varParam = owntype.getParameterTypes().last();
1031                 Type lastArg = argtypes.last();
1032                 if (types.isSubtypeUnchecked(lastArg, types.elemtype(varParam)) &&
1033                     !types.isSameType(types.erasure(varParam), types.erasure(lastArg)))
1034                     log.warning(argtrees.last().pos(),
1035                                 Warnings.InexactNonVarargsCall(types.elemtype(varParam),varParam));
1036             }
1037         }
1038         if (useVarargs) {
1039             Type argtype = owntype.getParameterTypes().last();
1040             if (!types.isReifiable(argtype) &&
1041                 (sym.baseSymbol().attribute(syms.trustMeType.tsym) == null ||
1042                  !isTrustMeAllowedOnMethod(sym))) {
1043                 warnUnchecked(env.tree.pos(), Warnings.UncheckedGenericArrayCreation(argtype));
1044             }
1045             TreeInfo.setVarargsElement(env.tree, types.elemtype(argtype));
1046          }
1047          return owntype;
1048     }

1071         private Type firstIncompatibleTypeArg(Type type) {
1072             List<Type> formals = type.tsym.type.allparams();
1073             List<Type> actuals = type.allparams();
1074             List<Type> args = type.getTypeArguments();
1075             List<Type> forms = type.tsym.type.getTypeArguments();
1076             ListBuffer<Type> bounds_buf = new ListBuffer<>();
1077 
1078             // For matching pairs of actual argument types `a' and
1079             // formal type parameters with declared bound `b' ...
1080             while (args.nonEmpty() && forms.nonEmpty()) {
1081                 // exact type arguments needs to know their
1082                 // bounds (for upper and lower bound
1083                 // calculations).  So we create new bounds where
1084                 // type-parameters are replaced with actuals argument types.
1085                 bounds_buf.append(types.subst(forms.head.getUpperBound(), formals, actuals));
1086                 args = args.tail;
1087                 forms = forms.tail;
1088             }
1089 
1090             args = type.getTypeArguments();
1091             List<Type> tvars_cap = types.substBounds(formals,
1092                                       formals,
1093                                       types.capture(type).allparams());
1094             while (args.nonEmpty() && tvars_cap.nonEmpty()) {
1095                 // Let the actual arguments know their bound
1096                 args.head.withTypeVar((TypeVar)tvars_cap.head);
1097                 args = args.tail;
1098                 tvars_cap = tvars_cap.tail;
1099             }
1100 
1101             args = type.getTypeArguments();
1102             List<Type> bounds = bounds_buf.toList();
1103 
1104             while (args.nonEmpty() && bounds.nonEmpty()) {
1105                 Type actual = args.head;
1106                 if (!isTypeArgErroneous(actual) &&
1107                         !bounds.head.isErroneous() &&
1108                         !checkExtends(actual, bounds.head)) {
1109                     return args.head;
1110                 }
1111                 args = args.tail;
1112                 bounds = bounds.tail;
1113             }
1114 
1115             args = type.getTypeArguments();
1116             bounds = bounds_buf.toList();
1117 
1118             for (Type arg : types.capture(type).getTypeArguments()) {
1119                 if (arg.hasTag(TYPEVAR) &&
1120                         arg.getUpperBound().isErroneous() &&
1121                         !bounds.head.isErroneous() &&
1122                         !isTypeArgErroneous(args.head)) {
1123                     return args.head;
1124                 }
1125                 bounds = bounds.tail;
1126                 args = args.tail;
1127             }
1128 
1129             return null;
1130         }

1155     /** Check that given modifiers are legal for given symbol and
1156      *  return modifiers together with any implicit modifiers for that symbol.
1157      *  Warning: we can't use flags() here since this method
1158      *  is called during class enter, when flags() would cause a premature
1159      *  completion.
1160      *  @param pos           Position to be used for error reporting.
1161      *  @param flags         The set of modifiers given in a definition.
1162      *  @param sym           The defined symbol.
1163      */
1164     long checkFlags(DiagnosticPosition pos, long flags, Symbol sym, JCTree tree) {
1165         long mask;
1166         long implicit = 0;
1167 
1168         switch (sym.kind) {
1169         case VAR:
1170             if (TreeInfo.isReceiverParam(tree))
1171                 mask = ReceiverParamFlags;
1172             else if (sym.owner.kind != TYP)
1173                 mask = LocalVarFlags;
1174             else if ((sym.owner.flags_field & INTERFACE) != 0)
1175                 mask = implicit = InterfaceVarFlags;
1176             else
1177                 mask = VarFlags;
1178             break;
1179         case MTH:
1180             if (sym.name == names.init) {
1181                 if ((sym.owner.flags_field & ENUM) != 0) {
1182                     // enum constructors cannot be declared public or
1183                     // protected and must be implicitly or explicitly
1184                     // private
1185                     implicit = PRIVATE;
1186                     mask = PRIVATE;
1187                 } else
1188                     mask = ConstructorFlags;
1189             }  else if ((sym.owner.flags_field & INTERFACE) != 0) {
1190                 if ((sym.owner.flags_field & ANNOTATION) != 0) {
1191                     mask = AnnotationTypeElementMask;
1192                     implicit = PUBLIC | ABSTRACT;
1193                 } else if ((flags & (DEFAULT | STATIC | PRIVATE)) != 0) {
1194                     mask = InterfaceMethodMask;
1195                     implicit = (flags & PRIVATE) != 0 ? 0 : PUBLIC;
1196                     if ((flags & DEFAULT) != 0) {
1197                         implicit |= ABSTRACT;
1198                     }
1199                 } else {
1200                     mask = implicit = InterfaceMethodFlags;
1201                 }
1202             } else if ((sym.owner.flags_field & RECORD) != 0) {
1203                 mask = RecordMethodFlags;
1204             } else {
1205                 mask = MethodFlags;
1206             }
1207             // Imply STRICTFP if owner has STRICTFP set.
1208             if (((flags|implicit) & Flags.ABSTRACT) == 0 ||
1209                 ((flags) & Flags.DEFAULT) != 0)
1210                 implicit |= sym.owner.flags_field & STRICTFP;
1211             break;
1212         case TYP:
1213             if (sym.isLocal()) {
1214                 boolean implicitlyStatic = !sym.isAnonymous() &&
1215                         ((flags & RECORD) != 0 || (flags & ENUM) != 0 || (flags & INTERFACE) != 0);
1216                 boolean staticOrImplicitlyStatic = (flags & STATIC) != 0 || implicitlyStatic;
1217                 mask = staticOrImplicitlyStatic && allowRecords ? StaticLocalFlags : LocalClassFlags;
1218                 implicit = implicitlyStatic ? STATIC : implicit;
1219                 if (staticOrImplicitlyStatic) {
1220                     if (sym.owner.kind == TYP) {
1221                         log.error(pos, Errors.StaticDeclarationNotAllowedInInnerClasses);
1222                     }
1223                 }
1224             } else if (sym.owner.kind == TYP) {
1225                 mask = (flags & RECORD) != 0 ? MemberRecordFlags : MemberClassFlags;
1226                 if (sym.owner.owner.kind == PCK ||
1227                     (sym.owner.flags_field & STATIC) != 0)
1228                     mask |= STATIC;
1229                 else if ((flags & ENUM) != 0 || (flags & RECORD) != 0) {
1230                     log.error(pos, Errors.StaticDeclarationNotAllowedInInnerClasses);
1231                 }
1232                 // Nested interfaces and enums are always STATIC (Spec ???)
1233                 if ((flags & (INTERFACE | ENUM | RECORD)) != 0 ) implicit = STATIC;
1234             } else {
1235                 mask = ClassFlags;
1236             }
1237             // Interfaces are always ABSTRACT
1238             if ((flags & INTERFACE) != 0) implicit |= ABSTRACT;
1239 
1240             if ((flags & ENUM) != 0) {
1241                 // enums can't be declared abstract or final
1242                 mask &= ~(ABSTRACT | FINAL);
1243                 implicit |= implicitEnumFinalFlag(tree);
1244             }
1245             if ((flags & RECORD) != 0) {
1246                 // records can't be declared abstract
1247                 mask &= ~ABSTRACT;
1248                 implicit |= FINAL;
1249             }
1250             // Imply STRICTFP if owner has STRICTFP set.
1251             implicit |= sym.owner.flags_field & STRICTFP;
1252             break;
1253         default:
1254             throw new AssertionError();
1255         }
1256         long illegal = flags & ExtendedStandardFlags & ~mask;
1257         if (illegal != 0) {
1258             if ((illegal & INTERFACE) != 0) {
1259                 log.error(pos, ((flags & ANNOTATION) != 0) ? Errors.AnnotationDeclNotAllowedHere : Errors.IntfNotAllowedHere);
1260                 mask |= INTERFACE;
1261             }
1262             else {
1263                 log.error(pos,
1264                         Errors.ModNotAllowedHere(asFlagSet(illegal)));
1265             }
1266         }
1267         else if ((sym.kind == TYP ||
1268                   // ISSUE: Disallowing abstract&private is no longer appropriate
1269                   // in the presence of inner classes. Should it be deleted here?
1270                   checkDisjoint(pos, flags,
1271                                 ABSTRACT,
1272                                 PRIVATE | STATIC | DEFAULT))
1273                  &&
1274                  checkDisjoint(pos, flags,
1275                                 STATIC | PRIVATE,
1276                                 DEFAULT)
1277                  &&
1278                  checkDisjoint(pos, flags,
1279                                ABSTRACT | INTERFACE,
1280                                FINAL | NATIVE | SYNCHRONIZED)
1281                  &&
1282                  checkDisjoint(pos, flags,
1283                                PUBLIC,
1284                                PRIVATE | PROTECTED)
1285                  &&
1286                  checkDisjoint(pos, flags,
1287                                PRIVATE,
1288                                PUBLIC | PROTECTED)
1289                  &&
1290                  checkDisjoint(pos, flags,
1291                                FINAL,
1292                                VOLATILE)
1293                  &&
1294                  (sym.kind == TYP ||
1295                   checkDisjoint(pos, flags,
1296                                 ABSTRACT | NATIVE,
1297                                 STRICTFP))) {
1298             // skip
1299         }
1300         return flags & (mask | ~ExtendedStandardFlags) | implicit;
1301     }

1720     /** Check that this method conforms with overridden method 'other'.
1721      *  where `origin' is the class where checking started.
1722      *  Complications:
1723      *  (1) Do not check overriding of synthetic methods
1724      *      (reason: they might be final).
1725      *      todo: check whether this is still necessary.
1726      *  (2) Admit the case where an interface proxy throws fewer exceptions
1727      *      than the method it implements. Augment the proxy methods with the
1728      *      undeclared exceptions in this case.
1729      *  (3) When generics are enabled, admit the case where an interface proxy
1730      *      has a result type
1731      *      extended by the result type of the method it implements.
1732      *      Change the proxies result type to the smaller type in this case.
1733      *
1734      *  @param tree         The tree from which positions
1735      *                      are extracted for errors.
1736      *  @param m            The overriding method.
1737      *  @param other        The overridden method.
1738      *  @param origin       The class of which the overriding method
1739      *                      is a member.
1740      */
1741     void checkOverride(JCTree tree,
1742                        MethodSymbol m,
1743                        MethodSymbol other,
1744                        ClassSymbol origin) {
1745         // Don't check overriding of synthetic methods or by bridge methods.
1746         if ((m.flags() & (SYNTHETIC|BRIDGE)) != 0 || (other.flags() & SYNTHETIC) != 0) {
1747             return;
1748         }
1749 
1750         // Error if static method overrides instance method (JLS 8.4.6.2).
1751         if ((m.flags() & STATIC) != 0 &&
1752                    (other.flags() & STATIC) == 0) {
1753             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1754                       Errors.OverrideStatic(cannotOverride(m, other)));
1755             m.flags_field |= BAD_OVERRIDE;
1756             return;
1757         }
1758 
1759         // Error if instance method overrides static or final
1760         // method (JLS 8.4.6.1).
1761         if ((other.flags() & FINAL) != 0 ||
1762                  (m.flags() & STATIC) == 0 &&
1763                  (other.flags() & STATIC) != 0) {
1764             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1765                       Errors.OverrideMeth(cannotOverride(m, other),
1766                                           asFlagSet(other.flags() & (FINAL | STATIC))));
1767             m.flags_field |= BAD_OVERRIDE;
1768             return;
1769         }
1770 
1771         if ((m.owner.flags() & ANNOTATION) != 0) {
1772             // handled in validateAnnotationMethod
1773             return;
1774         }
1775 
1776         // Error if overriding method has weaker access (JLS 8.4.6.3).
1777         if (protection(m.flags()) > protection(other.flags())) {
1778             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1779                       (other.flags() & AccessFlags) == 0 ?
1780                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1781                                                           "package") :
1782                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1783                                                           asFlagSet(other.flags() & AccessFlags)));
1784             m.flags_field |= BAD_OVERRIDE;
1785             return;
1786         }
1787 
1788         Type mt = types.memberType(origin.type, m);
1789         Type ot = types.memberType(origin.type, other);
1790         // Error if overriding result type is different
1791         // (or, in the case of generics mode, not a subtype) of
1792         // overridden result type. We have to rename any type parameters
1793         // before comparing types.
1794         List<Type> mtvars = mt.getTypeArguments();
1795         List<Type> otvars = ot.getTypeArguments();
1796         Type mtres = mt.getReturnType();
1797         Type otres = types.subst(ot.getReturnType(), otvars, mtvars);
1798 
1799         overrideWarner.clear();
1800         boolean resultTypesOK =
1801             types.returnTypeSubstitutable(mt, ot, otres, overrideWarner);
1802         if (!resultTypesOK) {
1803             if ((m.flags() & STATIC) != 0 && (other.flags() & STATIC) != 0) {
1804                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1805                           Errors.OverrideIncompatibleRet(Fragments.CantHide(m, m.location(), other,
1806                                         other.location()), mtres, otres));
1807                 m.flags_field |= BAD_OVERRIDE;
1808             } else {
1809                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1810                           Errors.OverrideIncompatibleRet(cannotOverride(m, other), mtres, otres));
1811                 m.flags_field |= BAD_OVERRIDE;
1812             }
1813             return;
1814         } else if (overrideWarner.hasNonSilentLint(LintCategory.UNCHECKED)) {
1815             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
1816                     Warnings.OverrideUncheckedRet(uncheckedOverrides(m, other), mtres, otres));
1817         }
1818 
1819         // Error if overriding method throws an exception not reported
1820         // by overridden method.
1821         List<Type> otthrown = types.subst(ot.getThrownTypes(), otvars, mtvars);
1822         List<Type> unhandledErased = unhandled(mt.getThrownTypes(), types.erasure(otthrown));
1823         List<Type> unhandledUnerased = unhandled(mt.getThrownTypes(), otthrown);
1824         if (unhandledErased.nonEmpty()) {
1825             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1826                       Errors.OverrideMethDoesntThrow(cannotOverride(m, other), unhandledUnerased.head));
1827             m.flags_field |= BAD_OVERRIDE;
1828             return;
1829         }
1830         else if (unhandledUnerased.nonEmpty()) {
1831             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
1832                           Warnings.OverrideUncheckedThrown(cannotOverride(m, other), unhandledUnerased.head));
1833             return;
1834         }
1835 
1836         // Optional warning if varargs don't agree

2652     /**
2653       * Report warnings for potentially ambiguous method declarations. Two declarations
2654       * are potentially ambiguous if they feature two unrelated functional interface
2655       * in same argument position (in which case, a call site passing an implicit
2656       * lambda would be ambiguous).
2657       */
2658     void checkPotentiallyAmbiguousOverloads(DiagnosticPosition pos, Type site,
2659             MethodSymbol msym1, MethodSymbol msym2) {
2660         if (msym1 != msym2 &&
2661                 Feature.DEFAULT_METHODS.allowedInSource(source) &&
2662                 lint.isEnabled(LintCategory.OVERLOADS) &&
2663                 (msym1.flags() & POTENTIALLY_AMBIGUOUS) == 0 &&
2664                 (msym2.flags() & POTENTIALLY_AMBIGUOUS) == 0) {
2665             Type mt1 = types.memberType(site, msym1);
2666             Type mt2 = types.memberType(site, msym2);
2667             //if both generic methods, adjust type variables
2668             if (mt1.hasTag(FORALL) && mt2.hasTag(FORALL) &&
2669                     types.hasSameBounds((ForAll)mt1, (ForAll)mt2)) {
2670                 mt2 = types.subst(mt2, ((ForAll)mt2).tvars, ((ForAll)mt1).tvars);
2671             }
2672             //expand varargs methods if needed
2673             int maxLength = Math.max(mt1.getParameterTypes().length(), mt2.getParameterTypes().length());
2674             List<Type> args1 = rs.adjustArgs(mt1.getParameterTypes(), msym1, maxLength, true);
2675             List<Type> args2 = rs.adjustArgs(mt2.getParameterTypes(), msym2, maxLength, true);
2676             //if arities don't match, exit
2677             if (args1.length() != args2.length()) return;
2678             boolean potentiallyAmbiguous = false;
2679             while (args1.nonEmpty() && args2.nonEmpty()) {
2680                 Type s = args1.head;
2681                 Type t = args2.head;
2682                 if (!types.isSubtype(t, s) && !types.isSubtype(s, t)) {
2683                     if (types.isFunctionalInterface(s) && types.isFunctionalInterface(t) &&
2684                             types.findDescriptorType(s).getParameterTypes().length() > 0 &&
2685                             types.findDescriptorType(s).getParameterTypes().length() ==
2686                             types.findDescriptorType(t).getParameterTypes().length()) {
2687                         potentiallyAmbiguous = true;
2688                     } else {
2689                         break;
2690                     }
2691                 }
2692                 args1 = args1.tail;
2693                 args2 = args2.tail;
2694             }
2695             if (potentiallyAmbiguous) {
2696                 //we found two incompatible functional interfaces with same arity
2697                 //this means a call site passing an implicit lambda would be ambiguous
2698                 msym1.flags_field |= POTENTIALLY_AMBIGUOUS;
2699                 msym2.flags_field |= POTENTIALLY_AMBIGUOUS;
2700                 log.warning(LintCategory.OVERLOADS, pos,
2701                             Warnings.PotentiallyAmbiguousOverload(msym1, msym1.location(),
2702                                                                   msym2, msym2.location()));
2703                 return;
2704             }
2705         }

2894     /** Check an annotation of a symbol.
2895      */
2896     private void validateAnnotation(JCAnnotation a, JCTree declarationTree, Symbol s) {
2897         validateAnnotationTree(a);
2898         boolean isRecordMember = (s.flags_field & RECORD) != 0 || s.enclClass() != null && s.enclClass().isRecord();
2899 
2900         boolean isRecordField = isRecordMember &&
2901                 (s.flags_field & (Flags.PRIVATE | Flags.FINAL | Flags.GENERATED_MEMBER | Flags.RECORD)) != 0 &&
2902                 declarationTree.hasTag(VARDEF) &&
2903                 s.owner.kind == TYP;
2904 
2905         if (isRecordField) {
2906             // first we need to check if the annotation is applicable to records
2907             Name[] targets = getTargetNames(a);
2908             boolean appliesToRecords = false;
2909             for (Name target : targets) {
2910                 appliesToRecords =
2911                                 target == names.FIELD ||
2912                                 target == names.PARAMETER ||
2913                                 target == names.METHOD ||
2914                                 target == names.TYPE_USE ||
2915                                 target == names.RECORD_COMPONENT;
2916                 if (appliesToRecords) {
2917                     break;
2918                 }
2919             }
2920             if (!appliesToRecords) {
2921                 log.error(a.pos(), Errors.AnnotationTypeNotApplicable);
2922             } else {
2923                 /* lets now find the annotations in the field that are targeted to record components and append them to
2924                  * the corresponding record component
2925                  */
2926                 ClassSymbol recordClass = (ClassSymbol) s.owner;
2927                 RecordComponent rc = recordClass.getRecordComponent((VarSymbol)s);
2928                 SymbolMetadata metadata = rc.getMetadata();
2929                 if (metadata == null || metadata.isEmpty()) {
2930                     /* if not is empty then we have already been here, which is the case if multiple annotations are applied
2931                      * to the record component declaration
2932                      */
2933                     rc.appendAttributes(s.getRawAttributes().stream().filter(anno ->
2934                             Arrays.stream(getTargetNames(anno.type.tsym)).anyMatch(name -> name == names.RECORD_COMPONENT)
2935                     ).collect(List.collector()));
2936                     rc.setTypeAttributes(s.getRawTypeAttributes());
2937                     // to get all the type annotations applied to the type
2938                     rc.type = s.type;
2939                 }
2940             }
2941         }
2942 
2943         /* the section below is tricky. Annotations applied to record components are propagated to the corresponding
2944          * record member so if an annotation has target: FIELD, it is propagated to the corresponding FIELD, if it has
2945          * target METHOD, it is propagated to the accessor and so on. But at the moment when method members are generated
2946          * there is no enough information to propagate only the right annotations. So all the annotations are propagated
2947          * to all the possible locations.
2948          *
2949          * At this point we need to remove all the annotations that are not in place before going on with the annotation
2950          * party. On top of the above there is the issue that there is no AST representing record components, just symbols
2951          * so the corresponding field has been holding all the annotations and it's metadata has been modified as if it
2952          * was both a field and a record component.
2953          *
2954          * So there are two places where we need to trim annotations from: the metadata of the symbol and / or the modifiers
2955          * in the AST. Whatever is in the metadata will be written to the class file, whatever is in the modifiers could
2956          * be see by annotation processors.
2957          *
