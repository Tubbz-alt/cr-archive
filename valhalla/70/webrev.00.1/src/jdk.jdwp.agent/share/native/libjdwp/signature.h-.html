26 #ifndef JDWP_SIGNATURE_H
27 #define JDWP_SIGNATURE_H
28 
29 #define JVM_TYPE_ASSERT(typeKey)\
30 JDI_ASSERT_MSG(JDWP_Tag_OBJECT == typeKey || \
31                JDWP_Tag_ARRAY == typeKey || \
32                JDWP_Tag_BOOLEAN == typeKey || \
33                JDWP_Tag_BYTE == typeKey || \
34                JDWP_Tag_CHAR == typeKey || \
35                JDWP_Tag_DOUBLE == typeKey || \
36                JDWP_Tag_FLOAT == typeKey || \
37                JDWP_Tag_INT == typeKey || \
38                JDWP_Tag_LONG == typeKey || \
39                JDWP_Tag_SHORT == typeKey || \
40                JDWP_Tag_VOID == typeKey, \
41                "Tag is not a JVM basic type")
42 
43 static inline jbyte jdwpTag(const char *signature) {
44      JVM_TYPE_ASSERT(signature[0]);
45      return signature[0];
46 }
47 
48 static inline jboolean isReferenceTag(jbyte typeKey) {
49     JVM_TYPE_ASSERT(typeKey);
50     return (typeKey == JDWP_TAG(OBJECT)) || (typeKey == JDWP_TAG(ARRAY));
51 }
52 
53 static inline jboolean isArrayTag(jbyte typeKey) {
54     JVM_TYPE_ASSERT(typeKey);
55     return (typeKey == JDWP_TAG(ARRAY));
56 }
