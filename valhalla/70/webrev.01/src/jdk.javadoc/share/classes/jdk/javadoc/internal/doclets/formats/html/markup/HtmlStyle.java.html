439     private final String cssName;
440 
441     HtmlStyle() {
442         cssName = Pattern.compile("\\p{Upper}")
443                 .matcher(toString())
444                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
445     }
446 
447     HtmlStyle(String cssName) {
448         this.cssName = cssName;
449     }
450 
451     /**
452      * Returns the CSS class name associated with this style.
453      * @return the CSS class name
454      */
455     public String cssName() {
456         return cssName;
457     }
