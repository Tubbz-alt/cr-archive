 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import jdk.javadoc.internal.doclets.toolkit.util.DocFileIOException;
 29 
 30 /**
 31  * The interface for writing module summary output.
 32  *
 33  *  <p><b>This is NOT part of any supported API.
 34  *  If you write code that depends on this, you do so at your own risk.
 35  *  This code and its internal interfaces are subject to change or
 36  *  deletion without notice.</b>
 37  */
 38 
 39 public interface ModuleSummaryWriter {
 40 
 41     /**
 42      * Get the header for the summary.
 43      *
 44      * @param heading module name.
 45      * @return the header to be added to the content tree
 46      */
 47     Content getModuleHeader(String heading);
 48 
 49     /**
 50      * Get the header for the module content.
 51      *
 52      * @return a content tree for the module content header
 53      */
 54     Content getContentHeader();
 55 
 56     /**
 57      * Get the header for the summary header.
 58      *
 59      * @return a content tree with the summary header
 60      */
 61     Content getSummariesList();
 62 
 63     /**
 64      * Get the header for the summary tree.
 65      *
 66      * @param summaryContentTree the content tree.
 67      * @return a content tree with the summary tree
 68      */
 69     Content getSummaryTree(Content summaryContentTree);
 70 
 71     /**
 72      * Adds the module description.
 73      *
 74      * @param moduleContentTree the content tree to which the module description
 75      *                           will be added
 76      */
 77     void addModuleDescription(Content moduleContentTree);
 78 
 79     /**
 80      * Adds the tag information from the "module-info.java" file to the documentation
 81      * tree.
 82      *
 83      * @param moduleContentTree the content tree to which the module tags will
 84      *                           be added
 85      */
 86     void addModuleTags(Content moduleContentTree);
 87 
 88     /**
 89      * Adds the summary of modules to the list of summaries.
 90      *
 91      * @param summariesList the list of summaries
 92      */
 93     void addModulesSummary(Content summariesList);
 94 
 95     /**
 96      * Adds the summary of packages to the list of summaries.
 97      *
 98      * @param summariesList the list of summaries
 99      */
100     void addPackagesSummary(Content summariesList);
101 
102     /**
103      * Adds the summary of services to the list of summaries.
104      *
105      * @param summariesList the list of summaries
106      */
107     void addServicesSummary(Content summariesList);
108 
109     /**
110      * Adds the module content tree to the documentation tree.
111      *
112      * @param moduleContentTree the content tree that will be added
113      */
114     void addModuleContent(Content moduleContentTree);
115 
116     /**
117      * Adds the footer to the documentation tree.
118      */
119     void addModuleFooter();
120 
121     /**
122      * Print the module summary document.
123      *
124      * @param contentTree the content tree that will be printed
125      * @throws DocFileIOException if there is a problem while writing the document
126      */
127     void printDocument(Content contentTree) throws DocFileIOException;
128 }
