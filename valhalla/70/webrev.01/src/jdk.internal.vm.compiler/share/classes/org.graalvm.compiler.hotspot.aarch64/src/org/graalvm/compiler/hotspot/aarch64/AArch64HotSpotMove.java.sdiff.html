137         @Override
138         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
139             Register resultRegister = asRegister(result);
140             Register ptr = asRegister(input);
141             Register base = (isRegister(baseRegister) ? asRegister(baseRegister) : zr);

142             // result = (ptr - base) >> shift
143             if (!encoding.hasBase()) {
144                 if (encoding.hasShift()) {
145                     masm.lshr(64, resultRegister, ptr, encoding.getShift());
146                 } else {
147                     masm.movx(resultRegister, ptr);
148                 }
149             } else if (nonNull) {
150                 masm.sub(64, resultRegister, ptr, base);
151                 if (encoding.hasShift()) {
152                     masm.lshr(64, resultRegister, resultRegister, encoding.getShift());
153                 }
154             } else {
155                 // if ptr is null it still has to be null after compression
156                 masm.cmp(64, ptr, 0);
157                 masm.cmov(64, resultRegister, ptr, base, AArch64Assembler.ConditionFlag.NE);
158                 masm.sub(64, resultRegister, resultRegister, base);
159                 if (encoding.hasShift()) {
160                     masm.lshr(64, resultRegister, resultRegister, encoding.getShift());
161                 }
162             }
163         }
