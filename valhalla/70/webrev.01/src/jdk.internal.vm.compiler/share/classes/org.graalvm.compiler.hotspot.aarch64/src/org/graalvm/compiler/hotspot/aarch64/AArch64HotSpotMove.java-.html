 55     public static class LoadHotSpotObjectConstantInline extends AArch64LIRInstruction implements LoadConstantOp {
 56         public static final LIRInstructionClass<LoadHotSpotObjectConstantInline> TYPE = LIRInstructionClass.create(LoadHotSpotObjectConstantInline.class);
 57 
 58         private HotSpotConstant constant;
 59         @Def({REG, STACK}) AllocatableValue result;
 60 
 61         public LoadHotSpotObjectConstantInline(HotSpotConstant constant, AllocatableValue result) {
 62             super(TYPE);
 63             this.constant = constant;
 64             this.result = result;
 65         }
 66 
 67         @Override
 68         protected void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
 69             crb.recordInlineDataInCode(constant);
 70             if (constant.isCompressed()) {
 71                 // masm.forceMov(asRegister(result), 0);
 72                 masm.movNarrowAddress(asRegister(result), 0);
 73             } else {
 74                 masm.movNativeAddress(asRegister(result), 0);
 75             }
 76         }
 77 
 78         @Override
 79         public AllocatableValue getResult() {
 80             return result;
 81         }
 82 
 83         @Override
 84         public Constant getConstant() {
 85             return constant;
 86         }
 87     }

 89     public static final class BaseMove extends AArch64LIRInstruction {
 90         public static final LIRInstructionClass<BaseMove> TYPE = LIRInstructionClass.create(BaseMove.class);
 91 
 92         @LIRInstruction.Def({REG, HINT}) protected AllocatableValue result;
 93         private final GraalHotSpotVMConfig config;
 94 
 95         public BaseMove(AllocatableValue result, GraalHotSpotVMConfig config) {
 96             super(TYPE);
 97             this.result = result;
 98             this.config = config;
 99         }
100 
101         @Override
102         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
103             try (AArch64MacroAssembler.ScratchRegister sc = masm.getScratchRegister()) {
104                 Register scratch = sc.getRegister();
105                 masm.adrp(scratch);
106                 masm.add(64, scratch, scratch, 1);
107                 masm.ldr(64, asRegister(result), AArch64Address.createBaseRegisterOnlyAddress(scratch));
108                 masm.nop();
109                 crb.recordMark(config.MARKID_NARROW_KLASS_BASE_ADDRESS);
110             }
111         }

115     /**
116      * Compresses a 8-byte pointer as a 4-byte int.
117      */
118     public static class CompressPointer extends AArch64LIRInstruction {
119         public static final LIRInstructionClass<CompressPointer> TYPE = LIRInstructionClass.create(CompressPointer.class);
120 
121         private final CompressEncoding encoding;
122         private final boolean nonNull;
123 
124         @Def({REG, HINT}) protected AllocatableValue result;
125         @Use({REG}) protected AllocatableValue input;
126         @Alive({REG, ILLEGAL}) protected AllocatableValue baseRegister;
127 
128         public CompressPointer(AllocatableValue result, AllocatableValue input, AllocatableValue baseRegister, CompressEncoding encoding, boolean nonNull) {
129             super(TYPE);
130             this.result = result;
131             this.input = input;
132             this.baseRegister = baseRegister;
133             this.encoding = encoding;
134             this.nonNull = nonNull;
135         }
136 
137         @Override
138         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
139             Register resultRegister = asRegister(result);
140             Register ptr = asRegister(input);
141             Register base = (isRegister(baseRegister) ? asRegister(baseRegister) : zr);
142             // result = (ptr - base) >> shift
143             if (!encoding.hasBase()) {
144                 if (encoding.hasShift()) {
145                     masm.lshr(64, resultRegister, ptr, encoding.getShift());
146                 } else {
147                     masm.movx(resultRegister, ptr);
148                 }
149             } else if (nonNull) {
150                 masm.sub(64, resultRegister, ptr, base);
151                 if (encoding.hasShift()) {
152                     masm.lshr(64, resultRegister, resultRegister, encoding.getShift());
153                 }
154             } else {
155                 // if ptr is null it still has to be null after compression
156                 masm.cmp(64, ptr, 0);
157                 masm.cmov(64, resultRegister, ptr, base, AArch64Assembler.ConditionFlag.NE);
158                 masm.sub(64, resultRegister, resultRegister, base);
159                 if (encoding.hasShift()) {
160                     masm.lshr(64, resultRegister, resultRegister, encoding.getShift());
161                 }
162             }
163         }
164     }

166     /**
167      * Decompresses a 4-byte offset into an actual pointer.
168      */
169     public static class UncompressPointer extends AArch64LIRInstruction {
170         public static final LIRInstructionClass<UncompressPointer> TYPE = LIRInstructionClass.create(UncompressPointer.class);
171 
172         private final CompressEncoding encoding;
173         private final boolean nonNull;
174 
175         @Def({REG}) protected AllocatableValue result;
176         @Use({REG}) protected AllocatableValue input;
177         @Alive({REG, ILLEGAL}) protected AllocatableValue baseRegister;
178 
179         public UncompressPointer(AllocatableValue result, AllocatableValue input, AllocatableValue baseRegister, CompressEncoding encoding, boolean nonNull) {
180             super(TYPE);
181             this.result = result;
182             this.input = input;
183             this.baseRegister = baseRegister;
184             this.encoding = encoding;
185             this.nonNull = nonNull;
186         }
187 
188         @Override
189         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
190             Register inputRegister = asRegister(input);
191             Register resultRegister = asRegister(result);
192             Register base = encoding.hasBase() ? asRegister(baseRegister) : null;
193             emitUncompressCode(masm, inputRegister, resultRegister, base, encoding.getShift(), nonNull);
194         }
195 
196         public static void emitUncompressCode(AArch64MacroAssembler masm, Register inputRegister, Register resReg, Register baseReg, int shift, boolean nonNull) {
197             // result = ptr << shift
198             if (baseReg == null) {
199                 if (shift != 0) {
200                     masm.shl(64, resReg, inputRegister, shift);
201                 } else if (!resReg.equals(inputRegister)) {
202                     masm.movx(resReg, inputRegister);
203                 }
204                 return;
205             }
206 
207             // result = base + (ptr << shift)
208             if (nonNull) {
209                 masm.add(64, resReg, baseReg, inputRegister, AArch64Assembler.ShiftType.LSL, shift);
210             } else {
211                 // if ptr is null it has to be null after decompression
212                 Label done = new Label();
213                 if (!resReg.equals(inputRegister)) {
214                     masm.mov(32, resReg, inputRegister);
215                 }
216                 masm.cbz(32, resReg, done);
217                 masm.add(64, resReg, baseReg, resReg, AArch64Assembler.ShiftType.LSL, shift);
218                 masm.bind(done);
219             }
220         }
221     }

237     public static void decodeKlassPointer(CompilationResultBuilder crb, AArch64MacroAssembler masm, Register result, Register ptr, CompressEncoding encoding, GraalHotSpotVMConfig config) {
238         try (AArch64MacroAssembler.ScratchRegister sc = masm.getScratchRegister()) {
239             Register scratch = sc.getRegister();
240             boolean pic = GeneratePIC.getValue(crb.getOptions());
241             if (pic || encoding.hasBase() || encoding.getShift() != 0) {
242                 if (pic) {
243                     masm.addressOf(scratch);
244                     masm.ldr(64, scratch, AArch64Address.createBaseRegisterOnlyAddress(scratch));
245                     masm.add(64, result, scratch, ptr, AArch64Assembler.ExtendType.UXTX, encoding.getShift());
246                     crb.recordMark(config.MARKID_NARROW_KLASS_BASE_ADDRESS);
247                 } else {
248                     masm.mov(scratch, encoding.getBase());
249                     masm.add(64, result, scratch, ptr, AArch64Assembler.ExtendType.UXTX, encoding.getShift());
250                 }
251             }
252         }
