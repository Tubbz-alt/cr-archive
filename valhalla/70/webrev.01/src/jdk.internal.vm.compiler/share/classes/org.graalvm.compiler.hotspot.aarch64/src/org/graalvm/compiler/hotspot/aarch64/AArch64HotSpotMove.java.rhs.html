 55     public static class LoadHotSpotObjectConstantInline extends AArch64LIRInstruction implements LoadConstantOp {
 56         public static final LIRInstructionClass<LoadHotSpotObjectConstantInline> TYPE = LIRInstructionClass.create(LoadHotSpotObjectConstantInline.class);
 57 
 58         private HotSpotConstant constant;
 59         @Def({REG, STACK}) AllocatableValue result;
 60 
 61         public LoadHotSpotObjectConstantInline(HotSpotConstant constant, AllocatableValue result) {
 62             super(TYPE);
 63             this.constant = constant;
 64             this.result = result;
 65         }
 66 
 67         @Override
 68         protected void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
 69             crb.recordInlineDataInCode(constant);
 70             if (constant.isCompressed()) {
 71                 // masm.forceMov(asRegister(result), 0);
 72                 masm.movNarrowAddress(asRegister(result), 0);
 73             } else {
 74                 masm.movNativeAddress(asRegister(result), 0);
 75             }
 76         }
 77 
 78         @Override
 79         public AllocatableValue getResult() {
 80             return result;
 81         }
 82 
 83         @Override
 84         public Constant getConstant() {
 85             return constant;
 86         }
 87     }

 89     public static final class BaseMove extends AArch64LIRInstruction {
 90         public static final LIRInstructionClass<BaseMove> TYPE = LIRInstructionClass.create(BaseMove.class);
 91 
 92         @LIRInstruction.Def({REG, HINT}) protected AllocatableValue result;
 93         private final GraalHotSpotVMConfig config;
 94 
 95         public BaseMove(AllocatableValue result, GraalHotSpotVMConfig config) {
 96             super(TYPE);
 97             this.result = result;
 98             this.config = config;
 99         }
100 
101         @Override
102         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
103             try (AArch64MacroAssembler.ScratchRegister sc = masm.getScratchRegister()) {
104                 Register scratch = sc.getRegister();
105                 masm.adrp(scratch);
106                 masm.add(64, scratch, scratch, 1);
107                 masm.ldr(64, asRegister(result), AArch64Address.createBaseRegisterOnlyAddress(scratch));
108                 masm.nop();
109                 crb.recordMark(config.MARKID_NARROW_KLASS_BASE_ADDRESS);
110             }
111         }

115     /**
116      * Compresses a 8-byte pointer as a 4-byte int.
117      */
118     public static class CompressPointer extends AArch64LIRInstruction {
119         public static final LIRInstructionClass<CompressPointer> TYPE = LIRInstructionClass.create(CompressPointer.class);
120 
121         private final CompressEncoding encoding;
122         private final boolean nonNull;
123 
124         @Def({REG, HINT}) protected AllocatableValue result;
125         @Use({REG}) protected AllocatableValue input;
126         @Alive({REG, ILLEGAL}) protected AllocatableValue baseRegister;
127 
128         public CompressPointer(AllocatableValue result, AllocatableValue input, AllocatableValue baseRegister, CompressEncoding encoding, boolean nonNull) {
129             super(TYPE);
130             this.result = result;
131             this.input = input;
132             this.baseRegister = baseRegister;
133             this.encoding = encoding;
134             this.nonNull = nonNull;
135         }
136 
137         @Override
138         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
139             Register resultRegister = asRegister(result);
140             Register ptr = asRegister(input);
141             Register base = (isRegister(baseRegister) ? asRegister(baseRegister) : zr);
142             boolean pic = GeneratePIC.getValue(crb.getOptions());
143             // result = (ptr - base) >> shift
144             if (!pic && !encoding.hasBase()) {
145                 if (encoding.hasShift()) {
146                     masm.lshr(64, resultRegister, ptr, encoding.getShift());
147                 } else {
148                     masm.movx(resultRegister, ptr);
149                 }
150             } else if (nonNull) {
151                 masm.sub(64, resultRegister, ptr, base);
152                 if (encoding.hasShift()) {
153                     masm.lshr(64, resultRegister, resultRegister, encoding.getShift());
154                 }
155             } else {
156                 // if ptr is null it still has to be null after compression
157                 masm.cmp(64, ptr, 0);
158                 masm.cmov(64, resultRegister, ptr, base, AArch64Assembler.ConditionFlag.NE);
159                 masm.sub(64, resultRegister, resultRegister, base);
160                 if (encoding.hasShift()) {
161                     masm.lshr(64, resultRegister, resultRegister, encoding.getShift());
162                 }
163             }
164         }
165     }

167     /**
168      * Decompresses a 4-byte offset into an actual pointer.
169      */
170     public static class UncompressPointer extends AArch64LIRInstruction {
171         public static final LIRInstructionClass<UncompressPointer> TYPE = LIRInstructionClass.create(UncompressPointer.class);
172 
173         private final CompressEncoding encoding;
174         private final boolean nonNull;
175 
176         @Def({REG}) protected AllocatableValue result;
177         @Use({REG}) protected AllocatableValue input;
178         @Alive({REG, ILLEGAL}) protected AllocatableValue baseRegister;
179 
180         public UncompressPointer(AllocatableValue result, AllocatableValue input, AllocatableValue baseRegister, CompressEncoding encoding, boolean nonNull) {
181             super(TYPE);
182             this.result = result;
183             this.input = input;
184             this.baseRegister = baseRegister;
185             this.encoding = encoding;
186             this.nonNull = nonNull;
187         }
188 
189         @Override
190         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
191             Register inputRegister = asRegister(input);
192             Register resultRegister = asRegister(result);
193             boolean pic = GeneratePIC.getValue(crb.getOptions());
194             Register base = pic || encoding.hasBase() ? asRegister(baseRegister) : null;
195             emitUncompressCode(masm, inputRegister, resultRegister, base, encoding.getShift(), nonNull);
196         }
197 
198         public static void emitUncompressCode(AArch64MacroAssembler masm, Register inputRegister, Register resReg, Register baseReg, int shift, boolean nonNull) {
199             // result = ptr << shift
200             if (baseReg == null) {
201                 if (shift != 0) {
202                     masm.shl(64, resReg, inputRegister, shift);
203                 } else if (!resReg.equals(inputRegister)) {
204                     masm.movx(resReg, inputRegister);
205                 }
206                 return;
207             }
208 
209             // result = base + (ptr << shift)
210             if (nonNull) {
211                 masm.add(64, resReg, baseReg, inputRegister, AArch64Assembler.ShiftType.LSL, shift);
212             } else {
213                 // if ptr is null it has to be null after decompression
214                 Label done = new Label();
215                 if (!resReg.equals(inputRegister)) {
216                     masm.mov(32, resReg, inputRegister);
217                 }
218                 masm.cbz(32, resReg, done);
219                 masm.add(64, resReg, baseReg, resReg, AArch64Assembler.ShiftType.LSL, shift);
220                 masm.bind(done);
221             }
222         }
223     }

239     public static void decodeKlassPointer(CompilationResultBuilder crb, AArch64MacroAssembler masm, Register result, Register ptr, CompressEncoding encoding, GraalHotSpotVMConfig config) {
240         try (AArch64MacroAssembler.ScratchRegister sc = masm.getScratchRegister()) {
241             Register scratch = sc.getRegister();
242             boolean pic = GeneratePIC.getValue(crb.getOptions());
243             if (pic || encoding.hasBase() || encoding.getShift() != 0) {
244                 if (pic) {
245                     masm.addressOf(scratch);
246                     masm.ldr(64, scratch, AArch64Address.createBaseRegisterOnlyAddress(scratch));
247                     masm.add(64, result, scratch, ptr, AArch64Assembler.ExtendType.UXTX, encoding.getShift());
248                     crb.recordMark(config.MARKID_NARROW_KLASS_BASE_ADDRESS);
249                 } else {
250                     masm.mov(scratch, encoding.getBase());
251                     masm.add(64, result, scratch, ptr, AArch64Assembler.ExtendType.UXTX, encoding.getShift());
252                 }
253             }
254         }
