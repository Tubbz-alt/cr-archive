26 package sun.security.util;
27 
28 import java.io.IOException;
29 import java.math.BigInteger;
30 
31 import java.security.spec.*;
32 
33 
34 /**
35  * Contains Elliptic Curve parameters.
36  *
37  * @since   1.6
38  * @author  Andreas Sterbenz
39  */
40 public final class NamedCurve extends ECParameterSpec {
41 
42     // friendly name for toString() output
43     private final String name;
44 
45     // well known OID
46     private final String oid;
47 
48     // encoded form (as NamedCurve identified via OID)
49     private final byte[] encoded;
50 
51     NamedCurve(String name, String oid, EllipticCurve curve,
52             ECPoint g, BigInteger n, int h) {
53         super(curve, g, n, h);
54         this.name = name;
55         this.oid = oid;
56 
57         DerOutputStream out = new DerOutputStream();
58 
59         try {
60             out.putOID(new ObjectIdentifier(oid));
61         } catch (IOException e) {
62             throw new RuntimeException("Internal error", e);
63         }
64 
65         encoded = out.toByteArray();
66     }
67 
68     public String getName() {
69         return name;
70     }
71 
72     public byte[] getEncoded() {
73         return encoded.clone();
74     }
75 
76     public String getObjectId() {
77         return oid;
78     }
79 
80     public String toString() {
81         return name + " (" + oid + ")";
82     }
83 }
