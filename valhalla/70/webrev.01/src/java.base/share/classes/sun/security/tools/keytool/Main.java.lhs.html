 417     private void run(String[] args, PrintStream out) throws Exception {
 418         try {
 419             args = parseArgs(args);
 420             if (command != null) {
 421                 doCommands(out);
 422             }
 423         } catch (Exception e) {
 424             System.out.println(rb.getString("keytool.error.") + e);
 425             if (verbose) {
 426                 e.printStackTrace(System.out);
 427             }
 428             if (!debug) {
 429                 System.exit(1);
 430             } else {
 431                 throw e;
 432             }
 433         } finally {
 434             printWeakWarnings(false);
 435             for (char[] pass : passwords) {
 436                 if (pass != null) {
 437                     Arrays.fill(pass, ' ');
 438                     pass = null;
 439                 }
 440             }
 441 
 442             if (ksStream != null) {
 443                 ksStream.close();
 444             }
 445         }
 446     }

 448     /**
 449      * Parse command line arguments.
 450      */
 451     String[] parseArgs(String[] args) throws Exception {
 452 
 453         int i=0;
 454         boolean help = args.length == 0;
 455 
 456         String confFile = null;
 457 
 458         // Records all commands and options set. Used to check dups.
 459         Set<String> optionsSet = new HashSet<>();
 460 
 461         for (i=0; i < args.length; i++) {
 462             String flags = args[i];
 463             if (flags.startsWith("-")) {
 464                 String lowerFlags = flags.toLowerCase(Locale.ROOT);
 465                 if (optionsSet.contains(lowerFlags)) {
 466                     switch (lowerFlags) {
 467                         case "-ext":
 468                         case "-id":
 469                         case "-provider":
 470                         case "-addprovider":
 471                         case "-providerclass":
 472                         case "-providerarg":
 473                             // These options are allowed multiple times
 474                             break;
 475                         default:
 476                             weakWarnings.add(String.format(
 477                                     rb.getString("option.1.set.twice"),
 478                                     lowerFlags));
 479                     }
 480                 } else {
 481                     optionsSet.add(lowerFlags);
 482                 }
 483                 if (collator.compare(flags, "-conf") == 0) {
 484                     if (i == args.length - 1) {
 485                         errorNeedArgument(flags);
 486                     }
 487                     confFile = args[++i];
 488                 } else {
 489                     Command c = Command.getCommand(flags);
 490                     if (c != null) {
 491                         if (command == null) {
 492                             command = c;
 493                         } else {
 494                             throw new Exception(String.format(
 495                                     rb.getString("multiple.commands.1.2"),
 496                                     command.name, c.name));
 497                         }
 498                     }
 499                 }
 500             }
 501         }
 502 
 503         if (confFile != null && command != null) {
 504             args = KeyStoreUtil.expandArgs("keytool", confFile,
 505                     command.toString(),
 506                     command.getAltName(), args);
 507         }
 508 
 509         debug = Arrays.stream(args).anyMatch(
 510                 x -> collator.compare(x, "-debug") == 0);
 511 
 512         if (debug) {
 513             // No need to localize debug output
 514             System.out.println("Command line args: " +
 515                     Arrays.toString(args));
 516         }
 517 
 518         for (i=0; (i < args.length) && args[i].startsWith("-"); i++) {
 519 
 520             String flags = args[i];
 521 
 522             // Check if the last option needs an arg
 523             if (i == args.length - 1) {
 524                 for (Option option: Option.values()) {
 525                     // Only options with an arg need to be checked
 526                     if (collator.compare(flags, option.toString()) == 0) {
 527                         if (option.arg != null) errorNeedArgument(flags);
 528                         break;
 529                     }
 530                 }
 531             }
 532 
 533             /*
 534              * Check modifiers
 535              */
 536             String modifier = null;
 537             int pos = flags.indexOf(':');
 538             if (pos > 0) {
 539                 modifier = flags.substring(pos+1);
 540                 flags = flags.substring(0, pos);
 541             }
 542 
 543             /*
 544              * command modes
 545              */
 546             Command c = Command.getCommand(flags);
 547 
 548             if (c != null) {
 549                 command = c;
 550             } else if (collator.compare(flags, "--help") == 0 ||
 551                        collator.compare(flags, "-h") == 0 ||
 552                        collator.compare(flags, "-?") == 0 ||
 553                        // -help: legacy.
 554                        collator.compare(flags, "-help") == 0) {
 555                 help = true;
 556             } else if (collator.compare(flags, "-conf") == 0) {
 557                 i++;
 558             } else if (collator.compare(flags, "-nowarn") == 0) {
 559                 nowarn = true;
 560             } else if (collator.compare(flags, "-keystore") == 0) {
 561                 ksfname = args[++i];
 562                 if (new File(ksfname).getCanonicalPath().equals(
 563                         new File(KeyStoreUtil.getCacerts()).getCanonicalPath())) {
 564                     System.err.println(rb.getString("warning.cacerts.option"));
 565                 }
 566             } else if (collator.compare(flags, "-destkeystore") == 0) {
 567                 ksfname = args[++i];
 568             } else if (collator.compare(flags, "-cacerts") == 0) {
 569                 cacerts = true;
 570             } else if (collator.compare(flags, "-storepass") == 0 ||
 571                     collator.compare(flags, "-deststorepass") == 0) {
 572                 storePass = getPass(modifier, args[++i]);
 573                 passwords.add(storePass);
 574             } else if (collator.compare(flags, "-storetype") == 0 ||
 575                     collator.compare(flags, "-deststoretype") == 0) {
 576                 storetype = KeyStoreUtil.niceStoreTypeName(args[++i]);
 577             } else if (collator.compare(flags, "-srcstorepass") == 0) {
 578                 srcstorePass = getPass(modifier, args[++i]);
 579                 passwords.add(srcstorePass);
 580             } else if (collator.compare(flags, "-srcstoretype") == 0) {
 581                 srcstoretype = KeyStoreUtil.niceStoreTypeName(args[++i]);
 582             } else if (collator.compare(flags, "-srckeypass") == 0) {
 583                 srckeyPass = getPass(modifier, args[++i]);
 584                 passwords.add(srckeyPass);
 585             } else if (collator.compare(flags, "-srcprovidername") == 0) {
 586                 srcProviderName = args[++i];
 587             } else if (collator.compare(flags, "-providername") == 0 ||
 588                     collator.compare(flags, "-destprovidername") == 0) {
 589                 providerName = args[++i];

 792                         ("if.source.keystore.is.not.password.protected.then.srcstorepass.and.srckeypass.must.not.be.specified"));
 793             }
 794         }

 797             throw new Exception
 798                 (rb.getString("Validity.must.be.greater.than.zero"));

 802         if (providers != null) {
 803             for (Pair<String, String> provider : providers) {
 804                 try {
 805                     KeyStoreUtil.loadProviderByName(
 806                             provider.fst, provider.snd);
 807                     if (debug) {
 808                         System.out.println("loadProviderByName: " + provider.fst);
 809                     }

 811                     throw new Exception(String.format(rb.getString(
 812                             "provider.name.not.found"), provider.fst));
 813                 }
 814             }
 815         }

 817             ClassLoader cl = null;
 818             if (pathlist != null) {
 819                 String path = null;

 821                         path, System.getProperty("java.class.path"));

 823                         path, System.getProperty("env.class.path"));

 830             }
 831             for (Pair<String, String> provider : providerClasses) {
 832                 try {
 833                     KeyStoreUtil.loadProviderByClass(
 834                             provider.fst, provider.snd, cl);

 836                         System.out.println("loadProviderByClass: " + provider.fst);

 840                             "provclass.not.a.provider"), provider.fst));

 843                             "provider.class.not.found"), provider.fst), e.getCause());

 856             throw new Exception(rb.getString
 857                 ("Key.password.must.be.at.least.6.characters"));
 858         }

 860             throw new Exception(rb.getString
 861                 ("New.password.must.be.at.least.6.characters"));
 862         }
 863         if (destKeyPass != null && destKeyPass.length < 6) {
 864             throw new Exception(rb.getString
 865                 ("New.password.must.be.at.least.6.characters"));
 866         }
 867

 869         if (ksfname == null) {
 870             ksfname = System.getProperty("user.home") + File.separator
 871                     + ".keystore";
 872         }
 873 
 874         KeyStore srcKeyStore = null;
 875         if (command == IMPORTKEYSTORE) {
 876             inplaceImport = inplaceImportCheck();
 877             if (inplaceImport) {
 878                 // We load srckeystore first so we have srcstorePass that
 879                 // can be assigned to storePass
 880                 srcKeyStore = loadSourceKeyStore();

 882                     storePass = srcstorePass;
 883                 }
 884             }
 885         }

 887         // Check if keystore exists.
 888         // If no keystore has been specified at the command line, try to use
 889         // the default, which is located in $HOME/.keystore.
 890         // No need to check if isKeyStoreRelated(command) is false.
 891 
 892         // DO NOT open the existing keystore if this is an in-place import.
 893         // The keystore should be created as brand new.
 894         if (isKeyStoreRelated(command) && !nullStream && !inplaceImport) {
 895             try {
 896                 ksfile = new File(ksfname);
 897                 // Check if keystore file is empty
 898                 if (ksfile.exists() && ksfile.length() == 0) {
 899                     throw new Exception(rb.getString
 900                             ("Keystore.file.exists.but.is.empty.") + ksfname);
 901                 }
 902                 ksStream = new FileInputStream(ksfile);
 903             } catch (FileNotFoundException e) {
 904                 // These commands do not need the keystore to be existing.
 905                 // Either it will create a new one or the keystore is
 906                 // optional (i.e. PRINTCRL).
 907                 if (command != GENKEYPAIR &&
 908                         command != GENSECKEY &&
 909                         command != IDENTITYDB &&
 910                         command != IMPORTCERT &&
 911                         command != IMPORTPASS &&
 912                         command != IMPORTKEYSTORE &&
 913                         command != PRINTCRL) {
 914                     throw new Exception(rb.getString
 915                             ("Keystore.file.does.not.exist.") + ksfname);

 917             }
 918         }

 921                 && dest == null) {
 922             dest = getAlias("destination");
 923             if ("".equals(dest)) {
 924                 throw new Exception(rb.getString
 925                         ("Must.specify.destination.alias"));
 926             }
 927         }
 928 
 929         if (command == DELETE && alias == null) {
 930             alias = getAlias(null);
 931             if ("".equals(alias)) {
 932                 throw new Exception(rb.getString("Must.specify.alias"));
 933             }
 934         }
 935 
 936         // Create new keystore
 937         // Probe for keystore type when filename is available
 938         if (ksfile != null && ksStream != null && providerName == null &&
 939                 storetype == null && !inplaceImport) {
 940             keyStore = KeyStore.getInstance(ksfile, storePass);
 941             storetype = keyStore.getType();
 942             if (storetype.equalsIgnoreCase("pkcs12")) {
 943                 isPasswordlessKeyStore = PKCS12KeyStore.isPasswordless(ksfile);
 944             }
 945         } else {
 946             if (storetype == null) {
 947                 storetype = KeyStore.getDefaultType();
 948             }
 949             if (providerName == null) {
 950                 keyStore = KeyStore.getInstance(storetype);
 951             } else {
 952                 keyStore = KeyStore.getInstance(storetype, providerName);
 953             }
 954             // When creating a new pkcs12 file, Do not prompt for storepass
 955             // if certProtectionAlgorithm and macAlgorithm are both NONE.
 956             if (storetype.equalsIgnoreCase("pkcs12")) {
 957                 isPasswordlessKeyStore =
 958                         "NONE".equals(SecurityProperties.privilegedGetOverridable(
 959                                 "keystore.pkcs12.certProtectionAlgorithm"))
 960                         && "NONE".equals(SecurityProperties.privilegedGetOverridable(
 961                                 "keystore.pkcs12.macAlgorithm"));
 962             }
 963 
 964             /*
 965              * Load the keystore data.
 966              *
 967              * At this point, it's OK if no keystore password has been provided.
 968              * We want to make sure that we can load the keystore data, i.e.,
 969              * the keystore data has the right format. If we cannot load the
 970              * keystore, why bother asking the user for his or her password?
 971              * Only if we were able to load the keystore, and no keystore
 972              * password has been provided, will we prompt the user for the
 973              * keystore password to verify the keystore integrity.
 974              * This means that the keystore is loaded twice: first load operation
 975              * checks the keystore format, second load operation verifies the
 976              * keystore integrity.
 977              *
 978              * If the keystore password has already been provided (at the
 979              * command line), however, the keystore is loaded only once, and the
 980              * keystore format and integrity are checked "at the same time".
 981              *
 982              * Null stream keystores are loaded later.
 983              */
 984             if (!nullStream) {
 985                 if (inplaceImport) {
 986                     keyStore.load(null, storePass);
 987                 } else {
 988                     keyStore.load(ksStream, storePass);
 989                 }
 990                 if (ksStream != null) {
 991                     ksStream.close();
 992                 }
 993             }
 994         }
 995 
 996         if (P12KEYSTORE.equalsIgnoreCase(storetype) && command == KEYPASSWD) {
 997             throw new UnsupportedOperationException(rb.getString
 998                     (".keypasswd.commands.not.supported.if.storetype.is.PKCS12"));
 999         }
1000 
1001         // All commands that create or modify the keystore require a keystore
1002         // password.
1003 
1004         if (nullStream && storePass != null) {
1005             keyStore.load(null, storePass);
1006         } else if (!nullStream && storePass != null) {
1007             // If we are creating a new non nullStream-based keystore,
1008             // insist that the password be at least 6 characters
1009             if (ksStream == null && storePass.length < 6) {
1010                 throw new Exception(rb.getString
1011                         ("Keystore.password.must.be.at.least.6.characters"));
1012             }
1013         } else if (storePass == null) {
1014             if (!protectedPath && !KeyStoreUtil.isWindowsKeyStore(storetype)
1015                     && isKeyStoreRelated(command)
1016                     && !isPasswordlessKeyStore) {
1017                 if (command == CERTREQ ||
1018                         command == DELETE ||
1019                         command == GENKEYPAIR ||
1020                         command == GENSECKEY ||
1021                         command == IMPORTCERT ||
1022                         command == IMPORTPASS ||
1023                         command == IMPORTKEYSTORE ||
1024                         command == KEYCLONE ||
1025                         command == CHANGEALIAS ||
1026                         command == SELFCERT ||
1027                         command == STOREPASSWD ||
1028                         command == KEYPASSWD ||
1029                         command == IDENTITYDB) {

1031                     do {
1032                         if (command == IMPORTKEYSTORE) {
1033                             System.err.print
1034                                     (rb.getString("Enter.destination.keystore.password."));
1035                         } else {
1036                             System.err.print
1037                                     (rb.getString("Enter.keystore.password."));
1038                         }
1039                         System.err.flush();
1040                         storePass = Password.readPassword(System.in);
1041                         passwords.add(storePass);
1042 
1043                         // If we are creating a new non nullStream-based keystore,
1044                         // insist that the password be at least 6 characters
1045                         if (!nullStream && (storePass == null || storePass.length < 6)) {
1046                             System.err.println(rb.getString
1047                                     ("Keystore.password.is.too.short.must.be.at.least.6.characters"));
1048                             storePass = null;
1049                         }
1050 
1051                         // If the keystore file does not exist and needs to be
1052                         // created, the storepass should be prompted twice.
1053                         if (storePass != null && !nullStream && ksStream == null) {
1054                             System.err.print(rb.getString("Re.enter.new.password."));
1055                             char[] storePassAgain = Password.readPassword(System.in);
1056                             passwords.add(storePassAgain);
1057                             if (!Arrays.equals(storePass, storePassAgain)) {
1058                                 System.err.println
1059                                         (rb.getString("They.don.t.match.Try.again"));
1060                                 storePass = null;
1061                             }
1062                         }
1063 
1064                         count++;
1065                     } while ((storePass == null) && count < 3);
1066 
1067 
1068                     if (storePass == null) {
1069                         System.err.println
1070                                 (rb.getString("Too.many.failures.try.later"));
1071                         return;
1072                     }
1073                 } else {
1074                     // here we have EXPORTCERT and LIST (info valid until STOREPASSWD)
1075                     if (command != PRINTCRL) {
1076                         System.err.print(rb.getString("Enter.keystore.password."));
1077                         System.err.flush();
1078                         storePass = Password.readPassword(System.in);
1079                         passwords.add(storePass);
1080                     }
1081                 }
1082             }

1084             // Now load a nullStream-based keystore,
1085             // or verify the integrity of an input stream-based keystore
1086             if (nullStream) {
1087                 keyStore.load(null, storePass);
1088             } else if (ksStream != null) {
1089                 ksStream = new FileInputStream(ksfile);
1090                 keyStore.load(ksStream, storePass);
1091                 ksStream.close();
1092             }
1093         }
1094 
1095         if (storePass != null && P12KEYSTORE.equalsIgnoreCase(storetype)) {
1096             MessageFormat form = new MessageFormat(rb.getString(
1097                 "Warning.Different.store.and.key.passwords.not.supported.for.PKCS12.KeyStores.Ignoring.user.specified.command.value."));
1098             if (keyPass != null && !Arrays.equals(storePass, keyPass)) {
1099                 Object[] source = {"-keypass"};
1100                 System.err.println(form.format(source));
1101                 keyPass = storePass;
1102             }
1103             if (destKeyPass != null && !Arrays.equals(storePass, destKeyPass)) {
1104                 Object[] source = {"-destkeypass"};
1105                 System.err.println(form.format(source));
1106                 destKeyPass = storePass;
1107             }
1108         }
1109 
1110         // -trustcacerts can only be specified on -importcert.
1111         // Reset it so that warnings on CA cert will remain for
1112         // -printcert, etc.
1113         if (command != IMPORTCERT) {
1114             trustcacerts = false;
1115         }

1117         if (trustcacerts) {
1118             caks = KeyStoreUtil.getCacertsKeyStore();
1119         }
1120 
1121         // Perform the specified command
1122         if (command == CERTREQ) {
1123             if (filename != null) {
1124                 try (PrintStream ps = new PrintStream(new FileOutputStream
1125                                                       (filename))) {
1126                     doCertReq(alias, sigAlgName, ps);
1127                 }

1130             }

1132                 MessageFormat form = new MessageFormat(rb.getString
1133                         ("Certification.request.stored.in.file.filename."));

1136                 System.err.println(rb.getString("Submit.this.to.your.CA"));
1137             }

1146                 }
1147             } else {

1149             }

1151                 MessageFormat form = new MessageFormat(rb.getString

1158                 throw new Exception(rb.getString(

1160             }
1161             doGenKeyPair(alias, dname, keyAlgName, keysize, groupName, sigAlgName);

1163         } else if (command == GENSECKEY) {
1164             if (keyAlgName == null) {
1165                 throw new Exception(rb.getString(

1167             }
1168             doGenSecretKey(alias, keyAlgName, keysize);
1169             kssave = true;
1170         } else if (command == IMPORTPASS) {

1172                 keyAlgName = "PBE";
1173             }
1174             // password is stored as a secret key
1175             doGenSecretKey(alias, keyAlgName, keysize);
1176             kssave = true;
1177         } else if (command == IDENTITYDB) {
1178             if (filename != null) {
1179                 try (InputStream inStream = new FileInputStream(filename)) {
1180                     doImportIdentityDatabase(inStream);
1181                 }
1182             } else {
1183                 doImportIdentityDatabase(System.in);
1184             }
1185         } else if (command == IMPORTCERT) {
1186             InputStream inStream = System.in;
1187             if (filename != null) {

2466         URI uri = null;

2468             in = System.in;
2469         } else {

2471                 uri = new URI(src);
2472                 if (uri.getScheme().equals("ldap")) {
2473                     // No input stream for LDAP
2474                 } else {

2476                 }
2477             } catch (Exception e) {

2479                     in = new FileInputStream(src);
2480                 } catch (Exception e2) {

2485                     }
