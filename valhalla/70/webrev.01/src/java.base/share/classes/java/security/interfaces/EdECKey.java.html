25 package java.security.interfaces;
26 
27 import java.security.spec.NamedParameterSpec;
28 
29 /**
30  * An interface for an elliptic curve public/private key as defined by
31  * <a href="https://tools.ietf.org/html/rfc8032">RFC 8032: Edwards-Curve
32  * Digital Signature Algorithm (EdDSA)</a>. These keys are distinct from the
33  * keys represented by {@code ECKey}, and they are intended for use with
34  * algorithms based on RFC 8032 such as the EdDSA {@code Signature} algorithm.
35  * This interface allows access to the algorithm parameters associated with
36  * the key.
37  *
38  * @since 15
39  */
40 public interface EdECKey {
41     /**
42      * Returns the algorithm parameters associated with the key.
43      *
44      * @return the associated algorithm parameters.
45      */
46     NamedParameterSpec getParams();
47 }
