25 package java.security.interfaces;
26 
27 import java.security.PublicKey;
28 import java.security.spec.EdECPoint;
29 
30 /**
31  * An interface for an elliptic curve public key as defined by
32  * <a href="https://tools.ietf.org/html/rfc8032">RFC 8032: Edwards-Curve
33  * Digital Signature Algorithm (EdDSA)</a>. These keys are distinct from the
34  * keys represented by {@code ECPublicKey}, and they are intended for use with
35  * algorithms based on RFC 8032 such as the EdDSA {@code Signature} algorithm.
36  * <p>
37  * An Edwards-Curve public key is a point on the curve, which is represented using an
38  * EdECPoint.
39  *
40  * @since 15
41  */
42 public interface EdECPublicKey extends EdECKey, PublicKey {
43 
44     /**
45      * Get the point representing the public key.
46      *
47      * @return the {@code EdECPoint} representing the public key.
48      */
49     EdECPoint getPoint();
50 }
