 42 /**
 43  * Service-provider class for selectors and selectable channels.
 44  *
 45  * <p> A selector provider is a concrete subclass of this class that has a
 46  * zero-argument constructor and implements the abstract methods specified
 47  * below.  A given invocation of the Java virtual machine maintains a single
 48  * system-wide default provider instance, which is returned by the {@link
 49  * #provider() provider} method.  The first invocation of that method will locate
 50  * the default provider as specified below.
 51  *
 52  * <p> The system-wide default provider is used by the static {@code open}
 53  * methods of the {@link java.nio.channels.DatagramChannel#open
 54  * DatagramChannel}, {@link java.nio.channels.Pipe#open Pipe}, {@link
 55  * java.nio.channels.Selector#open Selector}, {@link
 56  * java.nio.channels.ServerSocketChannel#open ServerSocketChannel}, and {@link
 57  * java.nio.channels.SocketChannel#open SocketChannel} classes.  It is also
 58  * used by the {@link java.lang.System#inheritedChannel System.inheritedChannel()}
 59  * method. A program may make use of a provider other than the default provider
 60  * by instantiating that provider and then directly invoking the {@code open}
 61  * methods defined in this class.
 62  *
 63  * <p> All of the methods in this class are safe for use by multiple concurrent
 64  * threads.  </p>
 65  *
 66  *
 67  * @author Mark Reinhold
 68  * @author JSR-51 Expert Group
 69  * @since 1.4
 70  */
 71 
 72 public abstract class SelectorProvider {
 73 
 74     private static Void checkPermission() {
 75         SecurityManager sm = System.getSecurityManager();
 76         if (sm != null)
 77             sm.checkPermission(new RuntimePermission("selectorProvider"));
 78         return null;
 79     }
 80     private SelectorProvider(Void ignore) { }
 81 
 82     /**
 83      * Initializes a new instance of this class.
 84      *
 85      * @throws  SecurityException
 86      *          If a security manager has been installed and it denies
 87      *          {@link RuntimePermission}{@code ("selectorProvider")}
 88      */
 89     protected SelectorProvider() {
 90         this(checkPermission());
 91     }
 92 
 93     private static class Holder {
 94         static final SelectorProvider INSTANCE = provider();
 95 
 96         static SelectorProvider provider() {
 97             PrivilegedAction<SelectorProvider> pa = () -> {
 98                 SelectorProvider sp;
 99                 if ((sp = loadProviderFromProperty()) != null)
100                     return sp;
101                 if ((sp = loadProviderAsService()) != null)
102                     return sp;
103                 return sun.nio.ch.DefaultSelectorProvider.get();
104             };
105             return AccessController.doPrivileged(pa);
106         }
107 
108         private static SelectorProvider loadProviderFromProperty() {
109             String cn = System.getProperty("java.nio.channels.spi.SelectorProvider");
110             if (cn == null)
111                 return null;
112             try {
113                 Class<?> clazz = Class.forName(cn, true, ClassLoader.getSystemClassLoader());
114                 return (SelectorProvider) clazz.getConstructor().newInstance();
115             } catch (ClassNotFoundException |
116                     NoSuchMethodException |
117                     IllegalAccessException |
118                     InvocationTargetException |
119                     InstantiationException |
120                     SecurityException x) {
121                 throw new ServiceConfigurationError(null, x);
122             }
123         }
124 
125         private static SelectorProvider loadProviderAsService() {
126             ServiceLoader<SelectorProvider> sl =
127                 ServiceLoader.load(SelectorProvider.class,
128                                    ClassLoader.getSystemClassLoader());
129             Iterator<SelectorProvider> i = sl.iterator();
130             for (;;) {
131                 try {
132                     return i.hasNext() ? i.next() : null;
133                 } catch (ServiceConfigurationError sce) {
134                     if (sce.getCause() instanceof SecurityException) {
135                         // Ignore the security exception, try the next provider
136                         continue;
137                     }
138                     throw sce;
139                 }
140             }
141         }
142     }
143 
144     /**
145      * Returns the system-wide default selector provider for this invocation of
146      * the Java virtual machine.
147      *
148      * <p> The first invocation of this method locates the default provider
149      * object as follows: </p>
150      *
151      * <ol>
152      *
153      *   <li><p> If the system property
154      *   {@systemProperty java.nio.channels.spi.SelectorProvider} is defined
155      *   then it is taken to be the fully-qualified name of a concrete provider
156      *   class. The class is loaded and instantiated; if this process fails then
157      *   an unspecified error is thrown.  </p></li>
158      *
159      *   <li><p> If a provider class has been installed in a jar file that is
160      *   visible to the system class loader, and that jar file contains a
161      *   provider-configuration file named
162      *   {@code java.nio.channels.spi.SelectorProvider} in the resource
163      *   directory {@code META-INF/services}, then the first class name
164      *   specified in that file is taken.  The class is loaded and
165      *   instantiated; if this process fails then an unspecified error is
166      *   thrown.  </p></li>
167      *
168      *   <li><p> Finally, if no provider has been specified by any of the above
169      *   means then the system-default provider class is instantiated and the
170      *   result is returned.  </p></li>
171      *
172      * </ol>
173      *
174      * <p> Subsequent invocations of this method return the provider that was
175      * returned by the first invocation.  </p>
176      *
177      * @return  The system-wide default selector provider
178      */
179     public static SelectorProvider provider() {
180         return Holder.INSTANCE;
181     }

256     /**
257      * Returns the channel inherited from the entity that created this
258      * Java virtual machine.
259      *
260      * <p> On many operating systems a process, such as a Java virtual
261      * machine, can be started in a manner that allows the process to
262      * inherit a channel from the entity that created the process. The
263      * manner in which this is done is system dependent, as are the
264      * possible entities to which the channel may be connected. For example,
265      * on UNIX systems, the Internet services daemon (<i>inetd</i>) is used to
266      * start programs to service requests when a request arrives on an
267      * associated network port. In this example, the process that is started,
268      * inherits a channel representing a network socket.
269      *
270      * <p> In cases where the inherited channel represents a network socket
271      * then the {@link java.nio.channels.Channel Channel} type returned
272      * by this method is determined as follows:
273      *
274      * <ul>
275      *
276      *  <li><p> If the inherited channel represents a stream-oriented connected
277      *  socket then a {@link java.nio.channels.SocketChannel SocketChannel} is
278      *  returned. The socket channel is, at least initially, in blocking
279      *  mode, bound to a socket address, and connected to a peer.
280      *  </p></li>
281      *
282      *  <li><p> If the inherited channel represents a stream-oriented listening
283      *  socket then a {@link java.nio.channels.ServerSocketChannel
284      *  ServerSocketChannel} is returned. The server-socket channel is, at
285      *  least initially, in blocking mode, and bound to a socket address.
286      *  </p></li>
287      *
288      *  <li><p> If the inherited channel is a datagram-oriented socket
289      *  then a {@link java.nio.channels.DatagramChannel DatagramChannel} is
290      *  returned. The datagram channel is, at least initially, in blocking
291      *  mode, and bound to a socket address.
292      *  </p></li>
293      *
294      * </ul>
295      *
296      * <p> In addition to the network-oriented channels described, this method
297      * may return other kinds of channels in the future.
298      *
299      * <p> The first invocation of this method creates the channel that is
300      * returned. Subsequent invocations of this method return the same
301      * channel. </p>
302      *
303      * @return  The inherited channel, if any, otherwise {@code null}.
304      *
305      * @throws  IOException
306      *          If an I/O error occurs
307      *
308      * @throws  SecurityException
309      *          If a security manager has been installed and it denies
310      *          {@link RuntimePermission}{@code ("inheritedChannel")}
311      *
312      * @since 1.5
313      */
314     public Channel inheritedChannel() throws IOException {
315         return null;
316     }
