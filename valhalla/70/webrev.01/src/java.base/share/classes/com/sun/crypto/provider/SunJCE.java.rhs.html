 44 /**
 45  * Defines the "SunJCE" provider.
 46  *
 47  * Supported algorithms and their names:
 48  *
 49  * - RSA encryption (PKCS#1 v1.5 and raw)
 50  *
 51  * - DES
 52  *
 53  * - DES-EDE
 54  *
 55  * - AES
 56  *
 57  * - Blowfish
 58  *
 59  * - RC2
 60  *
 61  * - ARCFOUR (RC4 compatible)
 62  *
 63  * - ChaCha20 (Stream cipher only and in AEAD mode with Poly1305)
 64  *
 65  * - Cipher modes ECB, CBC, CFB, OFB, PCBC, CTR, and CTS for all block ciphers
 66  *   and mode GCM for AES cipher
 67  *
 68  * - Cipher padding ISO10126Padding for non-PKCS#5 block ciphers and
 69  *   NoPadding and PKCS5Padding for all block ciphers
 70  *
 71  * - Password-based Encryption (PBE)
 72  *
 73  * - Diffie-Hellman Key Agreement
 74  *
 75  * - HMAC-MD5, HMAC-SHA1, HMAC with SHA2 family and SHA3 family of digests
 76  *
 77  */
 78

107     private void ps(String type, String algo, String cn, List<String> als,
108             HashMap<String, String> attrs) {
109         putService(new Provider.Service(this, type, algo, cn, als,

115             HashMap<String, String> attrs) {
116         putService(new Provider.Service(this, type, algo, cn, getAliases(algo),
117                    attrs));
118     }

120     public SunJCE() {
121         /* We are the "SunJCE" provider */
122         super("SunJCE", PROVIDER_VER, info);
123 
124         // if there is no security manager installed, put directly into
125         // the provider
126         if (System.getSecurityManager() == null) {
127             putEntries();
128         } else {
129             AccessController.doPrivileged(new PrivilegedAction<Void>() {
130                 @Override
131                 public Void run() {
132                     putEntries();
133                     return null;
134                 }
135             });
136         }
137         if (instance == null) {
138             instance = this;
139         }
140     }
