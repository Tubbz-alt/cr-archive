28 import sun.security.util.math.IntegerFieldModuloP;

30 /**
31  * A base interface for points on an elliptic curve over a finite field.
32  * Implementations may use different representations for points, and this
33  * interface creates a common API for manipulating points. This API has no
34  * methods for point arithmetic, which depends on group structure and curve
35  * parameters in addition to point representation.
36  */
37 public interface Point {
38 
39     IntegerFieldModuloP getField();
40     AffinePoint asAffine();

41 
42     ImmutablePoint fixed();
43     MutablePoint mutable();
44 
45 }
