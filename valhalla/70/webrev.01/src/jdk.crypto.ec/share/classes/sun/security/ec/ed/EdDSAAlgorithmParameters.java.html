 26 package sun.security.ec.ed;
 27 
 28 import java.io.IOException;
 29 import java.security.AlgorithmParametersSpi;
 30 import java.security.spec.AlgorithmParameterSpec;
 31 import java.security.spec.ECParameterSpec;
 32 import java.security.spec.EdDSAParameterSpec;
 33 import java.security.spec.InvalidParameterSpecException;
 34 
 35 /**
 36  * This AlgorithmParametersSpi only supports NamedParameterSpec.
 37  * EdDSAParameterSpec is not support because there is not ASN.1 format
 38  */
 39 
 40 public class EdDSAAlgorithmParameters extends AlgorithmParametersSpi {
 41 
 42     EdDSAParameterSpec edspec;
 43 
 44     // If no curve is provide, wait engineInit() to provide one.
 45     public EdDSAAlgorithmParameters() {
 46     }
 47 
 48     /**
 49      * NamedParameterSpec can only be used if curve was not specified
 50      * as part of getInstance(EdDSA).  If the curve was used, engineInit will
 51      * throws an exception for being already initialized.
 52      * EdDSAParameterSpec is not support because there is not ASN.1 format
 53      *
 54      * @param paramSpec NamedParameterSpec curve.
 55      *
 56      * @throws InvalidParameterSpecException
 57      */
 58     @Override
 59     protected void engineInit(AlgorithmParameterSpec paramSpec)
 60             throws InvalidParameterSpecException {
 61         if (paramSpec instanceof EdDSAParameterSpec) {
 62             edspec = (EdDSAParameterSpec)paramSpec;
 63             return;
 64         }
 65         throw new InvalidParameterSpecException(
 66                 "Unknown AlgorithmParameterSpec");
 67     }
 68 
 69     @Override
 70     protected void engineInit(byte[] params) throws IOException {
 71         throw new IOException(
 72                 "EdDSA does not support parameters as a byte array.");
 73     }
 74 
 75     @Override
 76     protected void engineInit(byte[] params, String format) throws IOException {
 77         engineInit(params);
 78     }
 79 
 80     @Override
 81     protected <T extends AlgorithmParameterSpec> T engineGetParameterSpec(
 82             Class<T> paramSpec) throws InvalidParameterSpecException {
 83 
 84         if (paramSpec.isAssignableFrom(ECParameterSpec.class)) {
 85             return paramSpec.cast(edspec);
 86         }
 87         throw new InvalidParameterSpecException(
 88                 "Only EDDSAParameterSpec supported.");
 89     }
 90 
 91     @Override
 92     protected byte[] engineGetEncoded() throws IOException {
 93         throw new IOException(
 94                 "EdDSA does not support parameters as a byte array.");
 95     }
 96 
 97     @Override
 98     protected byte[] engineGetEncoded(String format) throws IOException {
 99         throw new IOException(
100                 "EdDSA does not support parameters as a byte array.");
101     }
102 
103     @Override
104     protected String engineToString() {
105         return edspec.toString();
106     }
107 }
