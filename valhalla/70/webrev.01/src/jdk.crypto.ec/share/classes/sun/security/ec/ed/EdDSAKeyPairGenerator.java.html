 26 package sun.security.ec.ed;

 28 //import java.security.*;
 29 import java.security.InvalidAlgorithmParameterException;
 30 import java.security.InvalidKeyException;
 31 import java.security.InvalidParameterException;
 32 import java.security.KeyPair;
 33 import java.security.KeyPairGeneratorSpi;
 34 import java.security.NoSuchAlgorithmException;
 35 import java.security.ProviderException;
 36 import java.security.SecureRandom;
 37 import java.security.spec.AlgorithmParameterSpec;
 38 import java.security.spec.EdECPoint;
 39 import java.security.spec.NamedParameterSpec;
 40 
 41 import sun.security.jca.JCAUtil;
 42 import sun.security.util.SecurityProviderConstants;
 43 
 44 /**
 45  * Key pair generator for the EdDSA signature algorithm.
 46  */
 47 public class EdDSAKeyPairGenerator extends KeyPairGeneratorSpi {
 48 
 49     private SecureRandom random = null;
 50     private EdDSAOperations ops = null;
 51     private EdDSAParameters lockedParams = null;
 52 
 53     public EdDSAKeyPairGenerator() {
 54         initialize(SecurityProviderConstants.DEF_ED_KEY_SIZE, null);
 55     }
 56 
 57     private EdDSAKeyPairGenerator(NamedParameterSpec paramSpec) {
 58         tryInitialize(paramSpec);
 59         lockedParams = ops.getParameters();
 60     }
 61 
 62     private void tryInitialize(NamedParameterSpec paramSpec) {
 63         try {
 64             initialize(paramSpec, null);
 65         } catch (InvalidAlgorithmParameterException ex) {
 66             String name = paramSpec.getName();
 67             throw new ProviderException(name + " not supported");
 68         }
 69     }
 70 
 71     @Override
 72     public void initialize(int keySize, SecureRandom random) {
 73 
 74         EdDSAParameters params = EdDSAParameters.getBySize(
 75             InvalidParameterException::new, keySize);
 76 
 77         initializeImpl(params, random);
 78     }
 79 
 80     @Override
 81     public void initialize(AlgorithmParameterSpec params, SecureRandom random)
 82             throws InvalidAlgorithmParameterException {
 83 
 84         EdDSAParameters edParams = EdDSAParameters.get(
 85             InvalidAlgorithmParameterException::new, params);
 86 
 87         try {
 88             initializeImpl(edParams, random);
 89         } catch (InvalidParameterException e) {
 90             throw new InvalidAlgorithmParameterException(e);
 91         }
 92     }
 93 
 94     private void initializeImpl(EdDSAParameters params, SecureRandom random) {
 95 
 96         if (lockedParams != null && lockedParams != params) {
 97             throw new InvalidParameterException("Parameters must be " +
 98                 lockedParams.getName());
 99         }
100 
101         try {
102             this.ops = new EdDSAOperations(params);
103         } catch (NoSuchAlgorithmException ex) {
104             throw new ProviderException(ex);
105         }
106         this.random = random == null ? JCAUtil.getSecureRandom() : random;
107     }
108 
109 
110     @Override
111     public KeyPair generateKeyPair() {
112 
113         byte[] privateKey = ops.generatePrivate(random);
114         EdECPoint publicKey = ops.computePublic(privateKey);
115 
116         try {
117             return new KeyPair(
118                 new EdDSAPublicKeyImpl(ops.getParameters(), publicKey),
119                 new EdDSAPrivateKeyImpl(ops.getParameters(), privateKey)
120             );
121         } catch (InvalidKeyException ex) {
122             throw new ProviderException(ex);
123         }
124     }
125 
126     public static class Ed25519 extends EdDSAKeyPairGenerator {
127 
128         public Ed25519() {
129             super(NamedParameterSpec.ED25519);
130         }
131     }
132 
133     public static class Ed448 extends EdDSAKeyPairGenerator {
134 
135         public Ed448() {
136             super(NamedParameterSpec.ED448);
137         }
138     }
139 }
