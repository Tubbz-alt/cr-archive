 131 void BlockListBuilder::set_entries(int osr_bci) {
 132   // generate start blocks
 133   BlockBegin* std_entry = make_block_at(0, NULL);
 134   if (scope()->caller() == NULL) {
 135     std_entry->set(BlockBegin::std_entry_flag);
 136   }
 137   if (osr_bci != -1) {
 138     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 139     osr_entry->set(BlockBegin::osr_entry_flag);
 140   }
 141 
 142   // generate exception entry blocks
 143   XHandlers* list = xhandlers();
 144   const int n = list->length();
 145   for (int i = 0; i < n; i++) {
 146     XHandler* h = list->handler_at(i);
 147     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 148     entry->set(BlockBegin::exception_entry_flag);
 149     h->set_entry_block(entry);
 150   }
 151 }

 154 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 155   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 156 
 157   BlockBegin* block = _bci2block->at(cur_bci);
 158   if (block == NULL) {
 159     block = new BlockBegin(cur_bci);
 160     block->init_stores_to_locals(method()->max_locals());
 161     _bci2block->at_put(cur_bci, block);
 162     _blocks.append(block);
 163 
 164     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 165   }
 166 
 167   if (predecessor != NULL) {
 168     if (block->is_set(BlockBegin::exception_entry_flag)) {
 169       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 170     }
 171 
 172     predecessor->add_successor(block);
 173     block->increment_total_preds();
 174   }
 175 
 176   return block;
 177 }

 180 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 181   current->stores_to_locals().set_bit(local);

 183 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 184   store_one(current, local);
 185   store_one(current, local + 1);

 189 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 190   // Draws edges from a block to its exception handlers
 191   XHandlers* list = xhandlers();
 192   const int n = list->length();
 193 
 194   for (int i = 0; i < n; i++) {
 195     XHandler* h = list->handler_at(i);
 196 
 197     if (h->covers(cur_bci)) {
 198       BlockBegin* entry = h->entry_block();
 199       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 200       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 201 
 202       // add each exception handler only once
 203       if (!current->is_successor(entry)) {
 204         current->add_successor(entry);
 205         entry->increment_total_preds();
 206       }
 207 
 208       // stop when reaching catchall
 209       if (h->catch_type() == 0) break;
 210     }
 211   }
 212 }

 214 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 215   // start a new block after jsr-bytecode and link this block into cfg
 216   make_block_at(next_bci, current);
 217 
 218   // start a new block at the subroutine entry at mark it with special flag
 219   BlockBegin* sr_block = make_block_at(sr_bci, current);
 220   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 221     sr_block->set(BlockBegin::subroutine_entry_flag);
 222   }
 223 }

 226 void BlockListBuilder::set_leaders() {
 227   bool has_xhandlers = xhandlers()->has_handlers();
 228   BlockBegin* current = NULL;
 229 
 230   // The information which bci starts a new block simplifies the analysis
 231   // Without it, backward branches could jump to a bci where no block was created
 232   // during bytecode iteration. This would require the creation of a new block at the
 233   // branch target and a modification of the successor lists.
 234   const BitMap& bci_block_start = method()->bci_block_start();
 235 
 236   ciBytecodeStream s(method());
 237   while (s.next() != ciBytecodeStream::EOBC()) {
 238     int cur_bci = s.cur_bci();
 239 
 240     if (bci_block_start.at(cur_bci)) {
 241       current = make_block_at(cur_bci, current);
 242     }
 243     assert(current != NULL, "must have current block");
 244 
 245     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 246       handle_exceptions(current, cur_bci);
 247     }
 248 
 249     switch (s.cur_bc()) {
 250       // track stores to local variables for selective creation of phi functions
 251       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 252       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 253       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 254       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 255       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 256       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 257       case Bytecodes::_istore_0: store_one(current, 0); break;
 258       case Bytecodes::_istore_1: store_one(current, 1); break;
 259       case Bytecodes::_istore_2: store_one(current, 2); break;
 260       case Bytecodes::_istore_3: store_one(current, 3); break;
 261       case Bytecodes::_lstore_0: store_two(current, 0); break;
 262       case Bytecodes::_lstore_1: store_two(current, 1); break;
 263       case Bytecodes::_lstore_2: store_two(current, 2); break;
 264       case Bytecodes::_lstore_3: store_two(current, 3); break;
 265       case Bytecodes::_fstore_0: store_one(current, 0); break;
 266       case Bytecodes::_fstore_1: store_one(current, 1); break;
 267       case Bytecodes::_fstore_2: store_one(current, 2); break;
 268       case Bytecodes::_fstore_3: store_one(current, 3); break;
 269       case Bytecodes::_dstore_0: store_two(current, 0); break;
 270       case Bytecodes::_dstore_1: store_two(current, 1); break;
 271       case Bytecodes::_dstore_2: store_two(current, 2); break;
 272       case Bytecodes::_dstore_3: store_two(current, 3); break;
 273       case Bytecodes::_astore_0: store_one(current, 0); break;
 274       case Bytecodes::_astore_1: store_one(current, 1); break;
 275       case Bytecodes::_astore_2: store_one(current, 2); break;
 276       case Bytecodes::_astore_3: store_one(current, 3); break;
 277 
 278       // track bytecodes that affect the control flow
 279       case Bytecodes::_athrow:  // fall through
 280       case Bytecodes::_ret:     // fall through
 281       case Bytecodes::_ireturn: // fall through
 282       case Bytecodes::_lreturn: // fall through
 283       case Bytecodes::_freturn: // fall through
 284       case Bytecodes::_dreturn: // fall through
 285       case Bytecodes::_areturn: // fall through
 286       case Bytecodes::_return:
 287         current = NULL;
 288         break;
 289 
 290       case Bytecodes::_ifeq:      // fall through
 291       case Bytecodes::_ifne:      // fall through
 292       case Bytecodes::_iflt:      // fall through
 293       case Bytecodes::_ifge:      // fall through
 294       case Bytecodes::_ifgt:      // fall through
 295       case Bytecodes::_ifle:      // fall through
 296       case Bytecodes::_if_icmpeq: // fall through
 297       case Bytecodes::_if_icmpne: // fall through
 298       case Bytecodes::_if_icmplt: // fall through
 299       case Bytecodes::_if_icmpge: // fall through
 300       case Bytecodes::_if_icmpgt: // fall through
 301       case Bytecodes::_if_icmple: // fall through
 302       case Bytecodes::_if_acmpeq: // fall through
 303       case Bytecodes::_if_acmpne: // fall through
 304       case Bytecodes::_ifnull:    // fall through
 305       case Bytecodes::_ifnonnull:
 306         make_block_at(s.next_bci(), current);
 307         make_block_at(s.get_dest(), current);
 308         current = NULL;
 309         break;
 310 
 311       case Bytecodes::_goto:
 312         make_block_at(s.get_dest(), current);
 313         current = NULL;
 314         break;
 315 
 316       case Bytecodes::_goto_w:
 317         make_block_at(s.get_far_dest(), current);
 318         current = NULL;
 319         break;
 320 
 321       case Bytecodes::_jsr:
 322         handle_jsr(current, s.get_dest(), s.next_bci());
 323         current = NULL;
 324         break;
 325 
 326       case Bytecodes::_jsr_w:
 327         handle_jsr(current, s.get_far_dest(), s.next_bci());
 328         current = NULL;
 329         break;
 330 
 331       case Bytecodes::_tableswitch: {
 332         // set block for each case
 333         Bytecode_tableswitch sw(&s);
 334         int l = sw.length();
 335         for (int i = 0; i < l; i++) {
 336           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 337         }
 338         make_block_at(cur_bci + sw.default_offset(), current);
 339         current = NULL;
 340         break;
 341       }
 342 
 343       case Bytecodes::_lookupswitch: {
 344         // set block for each case
 345         Bytecode_lookupswitch sw(&s);
 346         int l = sw.number_of_pairs();
 347         for (int i = 0; i < l; i++) {
 348           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 349         }
 350         make_block_at(cur_bci + sw.default_offset(), current);
 351         current = NULL;
 352         break;
 353       }
 354 
 355       default:
 356         break;
 357     }
 358   }
 359 }

 362 void BlockListBuilder::mark_loops() {
 363   ResourceMark rm;
 364 
 365   _active.initialize(BlockBegin::number_of_blocks());
 366   _visited.initialize(BlockBegin::number_of_blocks());
 367   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 368   _next_loop_index = 0;
 369   _next_block_number = _blocks.length();
 370 
 371   // recursively iterate the control flow graph
 372   mark_loops(_bci2block->at(0), false);
 373   assert(_next_block_number >= 0, "invalid block numbers");
 374 
 375   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 376   _active.resize(0);
 377   _visited.resize(0);
 378 }

 380 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 381   if (block->is_set(BlockBegin::exception_entry_flag)) {
 382     // exception edges may look like loops but don't mark them as such
 383     // since it screws up block ordering.
 384     return;
 385   }
 386   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 387     block->set(BlockBegin::parser_loop_header_flag);
 388 
 389     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 390     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 391     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 392     if (_next_loop_index < 31) _next_loop_index++;
 393   } else {
 394     // block already marked as loop header
 395     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 396   }
 397 }

 399 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 400   int block_id = block->block_id();
 401 
 402   if (_visited.at(block_id)) {
 403     if (_active.at(block_id)) {
 404       // reached block via backward branch
 405       make_loop_header(block);
 406     }
 407     // return cached loop information for this block
 408     return _loop_map.at(block_id);
 409   }
 410 
 411   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 412     in_subroutine = true;
 413   }
 414 
 415   // set active and visited bits before successors are processed
 416   _visited.set_bit(block_id);
 417   _active.set_bit(block_id);
 418 
 419   intptr_t loop_state = 0;
 420   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 421     // recursively process all successors
 422     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 423   }
 424 
 425   // clear active-bit after all successors are processed
 426   _active.clear_bit(block_id);
 427 
 428   // reverse-post-order numbering of all blocks
 429   block->set_depth_first_number(_next_block_number);
 430   _next_block_number--;
 431 
 432   if (loop_state != 0 || in_subroutine ) {
 433     // block is contained at least in one loop, so phi functions are necessary
 434     // phi functions are also necessary for all locals stored in a subroutine
 435     scope()->requires_phi_function().set_union(block->stores_to_locals());
 436   }
 437 
 438   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 439     int header_loop_state = _loop_map.at(block_id);
 440     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 441 
 442     // If the highest bit is set (i.e. when integer value is negative), the method
 443     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 444     if (header_loop_state >= 0) {
 445       clear_bits(loop_state, header_loop_state);
 446     }
 447   }
 448 
 449   // cache and return loop information for this block
 450   _loop_map.at_put(block_id, loop_state);
 451   return loop_state;
 452 }

 461 void BlockListBuilder::print() {
 462   tty->print("----- initial block list of BlockListBuilder for method ");
 463   method()->print_short_name();
 464   tty->cr();
 465 
 466   // better readability if blocks are sorted in processing order
 467   _blocks.sort(compare_depth_first);
 468 
 469   for (int i = 0; i < _blocks.length(); i++) {
 470     BlockBegin* cur = _blocks.at(i);
 471     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 472 
 473     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 474     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 475     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 476     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 477     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 478 
 479     if (cur->number_of_sux() > 0) {
 480       tty->print("    sux: ");
 481       for (int j = 0; j < cur->number_of_sux(); j++) {
 482         BlockBegin* sux = cur->sux_at(j);
 483         tty->print("B%d ", sux->block_id());
 484       }
 485     }
 486     tty->cr();
 487   }
 488 }

 493 // A simple growable array of Values indexed by ciFields
 494 class FieldBuffer: public CompilationResourceObj {
 495  private:
 496   GrowableArray<Value> _values;
 497 
 498  public:
 499   FieldBuffer() {}
 500 
 501   void kill() {
 502     _values.trunc_to(0);
 503   }
 504 
 505   Value at(ciField* field) {
 506     assert(field->holder()->is_loaded(), "must be a loaded field");
 507     int offset = field->offset();
 508     if (offset < _values.length()) {
 509       return _values.at(offset);
 510     } else {
 511       return NULL;
 512     }
 513   }
 514 
 515   void at_put(ciField* field, Value value) {
 516     assert(field->holder()->is_loaded(), "must be a loaded field");
 517     int offset = field->offset();
 518     _values.at_put_grow(offset, value, NULL);
 519   }
 520 
 521 };

 524 // MemoryBuffer is fairly simple model of the current state of memory.
 525 // It partitions memory into several pieces.  The first piece is
 526 // generic memory where little is known about the owner of the memory.
 527 // This is conceptually represented by the tuple <O, F, V> which says
 528 // that the field F of object O has value V.  This is flattened so
 529 // that F is represented by the offset of the field and the parallel
 530 // arrays _objects and _values are used for O and V.  Loads of O.F can
 531 // simply use V.  Newly allocated objects are kept in a separate list
 532 // along with a parallel array for each object which represents the
 533 // current value of its fields.  Stores of the default value to fields
 534 // which have never been stored to before are eliminated since they
 535 // are redundant.  Once newly allocated objects are stored into
 536 // another object or they are passed out of the current compile they
 537 // are treated like generic memory.
 538 
 539 class MemoryBuffer: public CompilationResourceObj {
 540  private:
 541   FieldBuffer                 _values;
 542   GrowableArray<Value>        _objects;
 543   GrowableArray<Value>        _newobjects;
 544   GrowableArray<FieldBuffer*> _fields;
 545 
 546  public:
 547   MemoryBuffer() {}
 548 
 549   StoreField* store(StoreField* st) {
 550     if (!EliminateFieldAccess) {
 551       return st;
 552     }
 553 
 554     Value object = st->obj();
 555     Value value = st->value();
 556     ciField* field = st->field();
 557     if (field->holder()->is_loaded()) {
 558       int offset = field->offset();
 559       int index = _newobjects.find(object);
 560       if (index != -1) {
 561         // newly allocated object with no other stores performed on this field
 562         FieldBuffer* buf = _fields.at(index);
 563         if (buf->at(field) == NULL && is_default_value(value)) {
 564 #ifndef PRODUCT
 565           if (PrintIRDuringConstruction && Verbose) {
 566             tty->print_cr("Eliminated store for object %d:", index);
 567             st->print_line();
 568           }
 569 #endif
 570           return NULL;
 571         } else {
 572           buf->at_put(field, value);
 573         }
 574       } else {
 575         _objects.at_put_grow(offset, object, NULL);
 576         _values.at_put(field, value);
 577       }
 578 
 579       store_value(value);
 580     } else {
 581       // if we held onto field names we could alias based on names but
 582       // we don't know what's being stored to so kill it all.
 583       kill();
 584     }
 585     return st;
 586   }
 587 
 588 
 589   // return true if this value correspond to the default value of a field.
 590   bool is_default_value(Value value) {
 591     Constant* con = value->as_Constant();
 592     if (con) {
 593       switch (con->type()->tag()) {
 594         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 595         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 596         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 597         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 598         case objectTag: return con->type() == objectNull;
 599         default:  ShouldNotReachHere();
 600       }
 601     }
 602     return false;
 603   }
 604 
 605 
 606   // return either the actual value of a load or the load itself
 607   Value load(LoadField* load) {
 608     if (!EliminateFieldAccess) {
 609       return load;
 610     }
 611 
 612     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 613 #ifdef IA32
 614       if (UseSSE < 2) {
 615         // can't skip load since value might get rounded as a side effect
 616         return load;
 617       }
 618 #else
 619       Unimplemented();
 620 #endif // IA32
 621     }
 622 
 623     ciField* field = load->field();
 624     Value object   = load->obj();
 625     if (field->holder()->is_loaded() && !field->is_volatile()) {
 626       int offset = field->offset();
 627       Value result = NULL;
 628       int index = _newobjects.find(object);
 629       if (index != -1) {
 630         result = _fields.at(index)->at(field);
 631       } else if (_objects.at_grow(offset, NULL) == object) {
 632         result = _values.at(field);
 633       }
 634       if (result != NULL) {
 635 #ifndef PRODUCT
 636         if (PrintIRDuringConstruction && Verbose) {
 637           tty->print_cr("Eliminated load: ");
 638           load->print_line();
 639         }
 640 #endif
 641         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 642         return result;
 643       }
 644     }
 645     return load;
 646   }
 647 
 648   // Record this newly allocated object
 649   void new_instance(NewInstance* object) {
 650     int index = _newobjects.length();
 651     _newobjects.append(object);
 652     if (_fields.at_grow(index, NULL) == NULL) {
 653       _fields.at_put(index, new FieldBuffer());
 654     } else {
 655       _fields.at(index)->kill();
 656     }
 657   }
 658 
 659   // Record this newly allocated object
 660   void new_instance(NewValueTypeInstance* object) {
 661     int index = _newobjects.length();
 662     _newobjects.append(object);
 663     if (_fields.at_grow(index, NULL) == NULL) {
 664       _fields.at_put(index, new FieldBuffer());
 665     } else {
 666       _fields.at(index)->kill();
 667     }
 668   }
 669 
 670   void store_value(Value value) {
 671     int index = _newobjects.find(value);
 672     if (index != -1) {
 673       // stored a newly allocated object into another object.
 674       // Assume we've lost track of it as separate slice of memory.
 675       // We could do better by keeping track of whether individual
 676       // fields could alias each other.
 677       _newobjects.remove_at(index);
 678       // pull out the field info and store it at the end up the list
 679       // of field info list to be reused later.
 680       _fields.append(_fields.at(index));
 681       _fields.remove_at(index);
 682     }
 683   }
 684 
 685   void kill() {
 686     _newobjects.trunc_to(0);
 687     _objects.trunc_to(0);
 688     _values.kill();
 689   }
 690 };

 731 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 732   if (parsing_jsr()) {
 733     // It is necessary to clone all blocks associated with a
 734     // subroutine, including those for exception handlers in the scope
 735     // of the method containing the jsr (because those exception
 736     // handlers may contain ret instructions in some cases).
 737     BlockBegin* block = bci2block()->at(bci);
 738     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 739       BlockBegin* new_block = new BlockBegin(block->bci());
 740       if (PrintInitialBlockList) {
 741         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 742                       block->block_id(), block->bci(), new_block->block_id());
 743       }
 744       // copy data from cloned blocked
 745       new_block->set_depth_first_number(block->depth_first_number());
 746       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 747       // Preserve certain flags for assertion checking
 748       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 749       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 750 
 751       // copy was_visited_flag to allow early detection of bailouts
 752       // if a block that is used in a jsr has already been visited before,
 753       // it is shared between the normal control flow and a subroutine
 754       // BlockBegin::try_merge returns false when the flag is set, this leads
 755       // to a compilation bailout
 756       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 757 
 758       bci2block()->at_put(bci, new_block);
 759       block = new_block;
 760     }
 761     return block;
 762   } else {
 763     return bci2block()->at(bci);
 764   }

 822 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 823   assert(worklist->top() == top, "");
 824   // sort block descending into work list
 825   const int dfn = top->depth_first_number();
 826   assert(dfn != -1, "unknown depth first number");
 827   int i = worklist->length()-2;
 828   while (i >= 0) {
 829     BlockBegin* b = worklist->at(i);
 830     if (b->depth_first_number() < dfn) {
 831       worklist->at_put(i+1, b);
 832     } else {
 833       break;
 834     }
 835     i --;
 836   }
 837   if (i >= -1) worklist->at_put(i + 1, top);
 838 }

 895 void GraphBuilder::load_constant() {
 896   ciConstant con = stream()->get_constant();
 897   if (con.basic_type() == T_ILLEGAL) {
 898     // FIXME: an unresolved Dynamic constant can get here,
 899     // and that should not terminate the whole compilation.
 900     BAILOUT("could not resolve a constant");
 901   } else {
 902     ValueType* t = illegalType;
 903     ValueStack* patch_state = NULL;
 904     switch (con.basic_type()) {
 905       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 906       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 907       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 908       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 909       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 910       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 911       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 912       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 913       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 914       case T_OBJECT :
 915        {
 916         ciObject* obj = con.as_object();
 917         if (!obj->is_loaded()
 918             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 919           // A Class, MethodType, MethodHandle, or String.
 920           // Unloaded condy nodes show up as T_ILLEGAL, above.
 921           patch_state = copy_state_before();
 922           t = new ObjectConstant(obj);
 923         } else {
 924           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 925           // result, which might turn out to be an array.
 926           if (obj->is_null_object())
 927             t = objectNull;
 928           else if (obj->is_array())
 929             t = new ArrayConstant(obj->as_array());
 930           else
 931             t = new InstanceConstant(obj->as_instance());
 932         }
 933         break;
 934        }
 935       default       : ShouldNotReachHere();
 936     }
 937     Value x;
 938     if (patch_state != NULL) {
 939       x = new Constant(t, patch_state);
 940     } else {
 941       x = new Constant(t);
 942     }
 943     push(t, append(x));
 944   }
 945 }

 961 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 962   if (parsing_jsr()) {
 963     // We need to do additional tracking of the location of the return
 964     // address for jsrs since we don't handle arbitrary jsr/ret
 965     // constructs. Here we are figuring out in which circumstances we
 966     // need to bail out.
 967     if (x->type()->is_address()) {

 969 
 970       // Also check parent jsrs (if any) at this time to see whether
 971       // they are using this local. We don't handle skipping over a

 973       for (ScopeData* cur_scope_data = scope_data()->parent();
 974            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 975            cur_scope_data = cur_scope_data->parent()) {
 976         if (cur_scope_data->jsr_return_address_local() == index) {
 977           BAILOUT("subroutine overwrites return address from previous subroutine");
 978         }
 979       }
 980     } else if (index == scope_data()->jsr_return_address_local()) {
 981       scope_data()->set_jsr_return_address_local(-1);
 982     }
 983   }
 984 
 985   x->set_local_index(index);
 986   state->store_local(index, round_fp(x));

 990 void GraphBuilder::load_indexed(BasicType type) {
 991   // In case of in block code motion in range check elimination
 992   ValueStack* state_before = NULL;
 993   int array_idx = state()->stack_size() - 2;
 994   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 995     // Save the entire state and re-execute on deopt when accessing flattened arrays
 996     state_before = copy_state_before();
 997     state_before->set_should_reexecute(true);
 998   } else {
 999     state_before = copy_state_indexed_access();
1000   }
1001   compilation()->set_has_access_indexed(true);
1002   Value index = ipop();
1003   Value array = apop();
1004   Value length = NULL;
1005   if (CSEArrayLength ||
1006       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1007       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1008     length = append(new ArrayLength(array, state_before));
1009   }
1010 
1011   LoadIndexed* load_indexed = NULL;
1012   Instruction* result = NULL;
1013   if (array->is_loaded_flattened_array()) {
1014     ciType* array_type = array->declared_type();
1015     ciValueKlass* elem_klass = array_type->as_value_array_klass()->element_klass()->as_value_klass();
1016     NewValueTypeInstance* new_instance = new NewValueTypeInstance(elem_klass, state_before, false);
1017     _memory->new_instance(new_instance);
1018     apush(append_split(new_instance));
1019     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1020     load_indexed->set_vt(new_instance);
1021   } else {
1022     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1023   }
1024   if (profile_array_accesses() && is_reference_type(type)) {
1025     compilation()->set_would_profile(true);
1026     load_indexed->set_should_profile(true);
1027     load_indexed->set_profiled_method(method());
1028     load_indexed->set_profiled_bci(bci());
1029   }
1030   result = append(load_indexed);
1031   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1032   if (!array->is_loaded_flattened_array()) {
1033     push(as_ValueType(type), result);
1034   }
1035 }

1089     case Bytecodes::_pop:
1090       { state()->raw_pop();
1091       }
1092       break;

1096       }
1097       break;

1099       { Value w = state()->raw_pop();

1121       break;

1189 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1190   Value s = ipop();
1191   Value x = pop(type);
1192   // try to simplify
1193   // Note: This code should go into the canonicalizer as soon as it can
1194   //       can handle canonicalized forms that contain more than one node.
1195   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1196     // pattern: x >>> s
1197     IntConstant* s1 = s->type()->as_IntConstant();
1198     if (s1 != NULL) {
1199       // pattern: x >>> s1, with s1 constant
1200       ShiftOp* l = x->as_ShiftOp();
1201       if (l != NULL && l->op() == Bytecodes::_ishl) {
1202         // pattern: (a << b) >>> s1
1203         IntConstant* s0 = l->y()->type()->as_IntConstant();
1204         if (s0 != NULL) {
1205           // pattern: (a << s0) >>> s1
1206           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1207           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1208           if (s0c == s1c) {
1209             if (s0c == 0) {
1210               // pattern: (a << 0) >>> 0 => simplify to: a
1211               ipush(l->x());
1212             } else {
1213               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1214               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1215               const int m = (1 << (BitsPerInt - s0c)) - 1;
1216               Value s = append(new Constant(new IntConstant(m)));
1217               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1218             }
1219             return;
1220           }
1221         }
1222       }
1223     }
1224   }
1225   // could not simplify
1226   push(type, append(new ShiftOp(code, x, s)));
1227 }

1274 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1275   BlockBegin* tsux = block_at(stream()->get_dest());
1276   BlockBegin* fsux = block_at(stream()->next_bci());
1277   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1278 
1279   bool subst_check = false;
1280   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne) &&
1281       method() != ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature())) {
1282     // If current method is ValueBootstrapMethods::isSubstitutable(),
1283     // compile the acmp as a regular pointer comparison otherwise we
1284     // could call ValueBootstrapMethods::isSubstitutable() back
1285     ValueType* left_vt = x->type();
1286     ValueType* right_vt = y->type();
1287     if (left_vt->is_object()) {
1288       assert(right_vt->is_object(), "must be");
1289       ciKlass* left_klass = x->as_loaded_klass_or_null();
1290       ciKlass* right_klass = y->as_loaded_klass_or_null();
1291 
1292       if (left_klass == NULL || right_klass == NULL) {
1293         // The klass is still unloaded, or came from a Phi node. Go slow case;
1294         subst_check = true;
1295       } else if (left_klass->can_be_value_klass() || right_klass->can_be_value_klass()) {
1296         // Either operand may be a value object, but we're not sure. Go slow case;
1297         subst_check = true;
1298       } else {
1299         // No need to do substitutability check
1300       }
1301     }
1302   }
1303 
1304   // In case of loop invariant code motion or predicate insertion
1305   // before the body of a loop the state is needed
1306   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1307 
1308   assert(i->as_Goto() == NULL ||
1309          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1310          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1311          "safepoint state of Goto returned by canonicalizer incorrect");
1312 
1313   if (is_profiling()) {
1314     If* if_node = i->as_If();
1315     if (if_node != NULL) {
1316       // Note that we'd collect profile data in this method if we wanted it.
1317       compilation()->set_would_profile(true);
1318       // At level 2 we need the proper bci to count backedges
1319       if_node->set_profiled_bci(bci());
1320       if (profile_branches()) {
1321         // Successors can be rotated by the canonicalizer, check for this case.
1322         if_node->set_profiled_method(method());
1323         if_node->set_should_profile(true);
1324         if (if_node->tsux() == fsux) {
1325           if_node->set_swapped(true);
1326         }
1327       }
1328       return;
1329     }
1330 
1331     // Check if this If was reduced to Goto.
1332     Goto *goto_node = i->as_Goto();
1333     if (goto_node != NULL) {
1334       compilation()->set_would_profile(true);
1335       goto_node->set_profiled_bci(bci());
1336       if (profile_branches()) {
1337         goto_node->set_profiled_method(method());
1338         goto_node->set_should_profile(true);
1339         // Find out which successor is used.
1340         if (goto_node->default_sux() == tsux) {
1341           goto_node->set_direction(Goto::taken);
1342         } else if (goto_node->default_sux() == fsux) {
1343           goto_node->set_direction(Goto::not_taken);
1344         } else {
1345           ShouldNotReachHere();
1346         }
1347       }
1348       return;
1349     }
1350   }
1351 }

1410 void GraphBuilder::table_switch() {
1411   Bytecode_tableswitch sw(stream());
1412   const int l = sw.length();
1413   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1414     // total of 2 successors => use If instead of switch
1415     // Note: This code should go into the canonicalizer as soon as it can
1416     //       can handle canonicalized forms that contain more than one node.
1417     Value key = append(new Constant(new IntConstant(sw.low_key())));
1418     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1419     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1420     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1421     // In case of loop invariant code motion or predicate insertion
1422     // before the body of a loop the state is needed
1423     ValueStack* state_before = copy_state_if_bb(is_bb);
1424     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1425   } else {
1426     // collect successors
1427     BlockList* sux = new BlockList(l + 1, NULL);
1428     int i;
1429     bool has_bb = false;
1430     for (i = 0; i < l; i++) {
1431       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1432       if (sw.dest_offset_at(i) < 0) has_bb = true;
1433     }
1434     // add default successor
1435     if (sw.default_offset() < 0) has_bb = true;
1436     sux->at_put(i, block_at(bci() + sw.default_offset()));
1437     // In case of loop invariant code motion or predicate insertion
1438     // before the body of a loop the state is needed
1439     ValueStack* state_before = copy_state_if_bb(has_bb);
1440     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1441 #ifdef ASSERT
1442     if (res->as_Goto()) {
1443       for (i = 0; i < l; i++) {
1444         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1445           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1446         }
1447       }
1448     }
1449 #endif
1450   }
1451 }

1454 void GraphBuilder::lookup_switch() {
1455   Bytecode_lookupswitch sw(stream());
1456   const int l = sw.number_of_pairs();
1457   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1458     // total of 2 successors => use If instead of switch
1459     // Note: This code should go into the canonicalizer as soon as it can
1460     //       can handle canonicalized forms that contain more than one node.
1461     // simplify to If
1462     LookupswitchPair pair = sw.pair_at(0);
1463     Value key = append(new Constant(new IntConstant(pair.match())));
1464     BlockBegin* tsux = block_at(bci() + pair.offset());
1465     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1466     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1467     // In case of loop invariant code motion or predicate insertion
1468     // before the body of a loop the state is needed

1470     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1471   } else {
1472     // collect successors & keys
1473     BlockList* sux = new BlockList(l + 1, NULL);
1474     intArray* keys = new intArray(l, l, 0);
1475     int i;
1476     bool has_bb = false;
1477     for (i = 0; i < l; i++) {
1478       LookupswitchPair pair = sw.pair_at(i);
1479       if (pair.offset() < 0) has_bb = true;
1480       sux->at_put(i, block_at(bci() + pair.offset()));
1481       keys->at_put(i, pair.match());
1482     }
1483     // add default successor
1484     if (sw.default_offset() < 0) has_bb = true;
1485     sux->at_put(i, block_at(bci() + sw.default_offset()));
1486     // In case of loop invariant code motion or predicate insertion
1487     // before the body of a loop the state is needed
1488     ValueStack* state_before = copy_state_if_bb(has_bb);
1489     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1490 #ifdef ASSERT
1491     if (res->as_Goto()) {
1492       for (i = 0; i < l; i++) {
1493         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1494           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1495         }
1496       }
1497     }
1498 #endif
1499   }

1502 void GraphBuilder::call_register_finalizer() {
1503   // If the receiver requires finalization then emit code to perform
1504   // the registration on return.
1505 
1506   // Gather some type information about the receiver
1507   Value receiver = state()->local_at(0);
1508   assert(receiver != NULL, "must have a receiver");
1509   ciType* declared_type = receiver->declared_type();
1510   ciType* exact_type = receiver->exact_type();
1511   if (exact_type == NULL &&
1512       receiver->as_Local() &&
1513       receiver->as_Local()->java_index() == 0) {
1514     ciInstanceKlass* ik = compilation()->method()->holder();
1515     if (ik->is_final()) {
1516       exact_type = ik;
1517     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1518       // test class is leaf class
1519       compilation()->dependency_recorder()->assert_leaf_type(ik);
1520       exact_type = ik;
1521     } else {
1522       declared_type = ik;
1523     }
1524   }
1525 
1526   // see if we know statically that registration isn't required
1527   bool needs_check = true;
1528   if (exact_type != NULL) {
1529     needs_check = exact_type->as_instance_klass()->has_finalizer();
1530   } else if (declared_type != NULL) {
1531     ciInstanceKlass* ik = declared_type->as_instance_klass();
1532     if (!Dependencies::has_finalizable_subclass(ik)) {
1533       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1534       needs_check = false;
1535     }
1536   }
1537 
1538   if (needs_check) {
1539     // Perform the registration of finalizable objects.
1540     ValueStack* state_before = copy_state_for_exception();
1541     load_local(objectType, 0);
1542     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1543                                state()->pop_arguments(1),
1544                                true, state_before, true));
1545   }
1546 }

1549 void GraphBuilder::method_return(Value x, bool ignore_return) {
1550   if (RegisterFinalizersAtInit &&
1551       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1552     call_register_finalizer();
1553   }
1554 
1555   // The conditions for a memory barrier are described in Parse::do_exits().
1556   bool need_mem_bar = false;
1557   if (method()->is_object_constructor() &&
1558        (scope()->wrote_final() ||
1559          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1560          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1561     need_mem_bar = true;
1562   }
1563 
1564   BasicType bt = method()->return_type()->basic_type();
1565   switch (bt) {
1566     case T_BYTE:
1567     {
1568       Value shift = append(new Constant(new IntConstant(24)));
1569       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1570       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1571       break;
1572     }
1573     case T_SHORT:
1574     {
1575       Value shift = append(new Constant(new IntConstant(16)));
1576       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1577       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1578       break;
1579     }
1580     case T_CHAR:
1581     {
1582       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1583       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1584       break;
1585     }
1586     case T_BOOLEAN:
1587     {
1588       Value mask = append(new Constant(new IntConstant(1)));
1589       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1590       break;
1591     }
1592     default:
1593       break;
1594   }
1595 
1596   // Check to see whether we are inlining. If so, Return
1597   // instructions become Gotos to the continuation point.
1598   if (continuation() != NULL) {
1599 
1600     int invoke_bci = state()->caller_state()->bci();
1601 
1602     if (x != NULL  && !ignore_return) {
1603       ciMethod* caller = state()->scope()->caller()->method();
1604       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1605       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1606         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1607         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1608             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1609           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1610         }
1611       }
1612     }
1613 
1614     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1615 
1616     if (compilation()->env()->dtrace_method_probes()) {
1617       // Report exit from inline methods
1618       Values* args = new Values(1);
1619       args->push(append(new Constant(new MethodConstant(method()))));
1620       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1621     }
1622 
1623     // If the inlined method is synchronized, the monitor must be
1624     // released before we jump to the continuation block.
1625     if (method()->is_synchronized()) {
1626       assert(state()->locks_size() == 1, "receiver must be locked here");
1627       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1628     }
1629 
1630     if (need_mem_bar) {
1631       append(new MemBar(lir_membar_storestore));
1632     }
1633 
1634     // State at end of inlined method is the state of the caller
1635     // without the method parameters on stack, including the
1636     // return value, if any, of the inlined method on operand stack.
1637     set_state(state()->caller_state()->copy_for_parsing());
1638     if (x != NULL) {
1639       if (!ignore_return) {
1640         state()->push(x->type(), x);
1641       }
1642       if (profile_return() && x->type()->is_object_kind()) {
1643         ciMethod* caller = state()->scope()->method();
1644         profile_return_type(x, method(), caller, invoke_bci);
1645       }
1646     }
1647     Goto* goto_callee = new Goto(continuation(), false);
1648 
1649     // See whether this is the first return; if so, store off some
1650     // of the state for later examination
1651     if (num_returns() == 0) {
1652       set_inline_cleanup_info();
1653     }
1654 
1655     // The current bci() is in the wrong scope, so use the bci() of
1656     // the continuation point.
1657     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1658     incr_num_returns();
1659     return;
1660   }
1661 
1662   state()->truncate_stack(0);
1663   if (method()->is_synchronized()) {
1664     // perform the unlocking before exiting the method
1665     Value receiver;
1666     if (!method()->is_static()) {
1667       receiver = _initial_state->local_at(0);
1668     } else {
1669       receiver = append(new Constant(new ClassConstant(method()->holder())));
1670     }
1671     append_split(new MonitorExit(receiver, state()->unlock()));
1672   }
1673 
1674   if (need_mem_bar) {
1675       append(new MemBar(lir_membar_storestore));
1676   }
1677 
1678   assert(!ignore_return, "Ignoring return value works only for inlining");
1679   append(new Return(x));
1680 }

1722 void GraphBuilder::access_field(Bytecodes::Code code) {
1723   bool will_link;
1724   ciField* field = stream()->get_field(will_link);
1725   ciInstanceKlass* holder = field->holder();
1726   BasicType field_type = field->type()->basic_type();
1727   ValueType* type = as_ValueType(field_type);
1728 
1729   // call will_link again to determine if the field is valid.
1730   const bool needs_patching = !holder->is_loaded() ||
1731                               !field->will_link(method(), code) ||
1732                               PatchALot;
1733 
1734   ValueStack* state_before = NULL;
1735   if (!holder->is_initialized() || needs_patching) {
1736     // save state before instruction for debug info when
1737     // deoptimization happens during patching
1738     state_before = copy_state_before();
1739   }
1740 
1741   Value obj = NULL;
1742   if (code == Bytecodes::_getstatic || code == Bytecodes::_putstatic) {
1743     if (state_before != NULL) {
1744       // build a patching constant
1745       obj = new Constant(new InstanceConstant(holder->java_mirror()), state_before);
1746     } else {
1747       obj = new Constant(new InstanceConstant(holder->java_mirror()));
1748     }
1749   }
1750 
1751   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1752     scope()->set_wrote_final();
1753   }
1754 
1755   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1756     scope()->set_wrote_fields();
1757     if (field->is_volatile()) {
1758       scope()->set_wrote_volatile();
1759     }
1760   }
1761 
1762   const int offset = !needs_patching ? field->offset() : -1;
1763   switch (code) {
1764     case Bytecodes::_getstatic: {
1765       // check for compile-time constants, i.e., initialized static final fields
1766       Value constant = NULL;
1767       if (field->is_static_constant() && !PatchALot) {
1768         ciConstant field_value = field->constant_value();
1769         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1770                "stable static w/ default value shouldn't be a constant");
1771         constant = make_constant(field_value, field);
1772       }
1773       if (constant != NULL) {
1774         push(type, append(constant));
1775       } else {
1776         if (state_before == NULL) {
1777           state_before = copy_state_for_exception();
1778         }
1779         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1780                                         state_before, needs_patching);
1781         if (field->is_flattenable()) {
1782           load_field->set_never_null(true);
1783         }
1784         push(type, append(load_field));
1785       }
1786       break;
1787     }
1788     case Bytecodes::_putstatic: {
1789       Value val = pop(type);
1790       val->set_escaped();
1791       if (state_before == NULL) {
1792         state_before = copy_state_for_exception();
1793       }
1794       if (field->type()->basic_type() == T_BOOLEAN) {
1795         Value mask = append(new Constant(new IntConstant(1)));
1796         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1797       }
1798       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1799       break;
1800     }
1801     case Bytecodes::_getfield: {
1802       // Check for compile-time constants, i.e., trusted final non-static fields.
1803       Value constant = NULL;
1804       if (state_before == NULL && field->is_flattened()) {
1805         // Save the entire state and re-execute on deopt when accessing flattened fields
1806         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1807         state_before = copy_state_before();
1808       }
1809       obj = apop();
1810       ObjectType* obj_type = obj->type()->as_ObjectType();
1811       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1812         ciObject* const_oop = obj_type->constant_value();
1813         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1814           ciConstant field_value = field->constant_value_of(const_oop);
1815           if (field_value.is_valid()) {
1816             constant = make_constant(field_value, field);
1817             // For CallSite objects add a dependency for invalidation of the optimization.
1818             if (field->is_call_site_target()) {
1819               ciCallSite* call_site = const_oop->as_call_site();
1820               if (!call_site->is_fully_initialized_constant_call_site()) {
1821                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1822                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1823               }
1824             }
1825           }
1826         }
1827       }
1828       if (constant != NULL) {
1829         push(type, append(constant));
1830       } else {
1831         if (state_before == NULL) {
1832           state_before = copy_state_for_exception();
1833         }
1834         if (!field->is_flattened()) {
1835           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1836           Value replacement = !needs_patching ? _memory->load(load) : load;
1837           if (replacement != load) {
1838             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1839             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1840             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1841             BasicType bt = field->type()->basic_type();
1842             switch (bt) {
1843             case T_BOOLEAN:
1844             case T_BYTE:
1845               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1846               break;
1847             case T_CHAR:
1848               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1849               break;
1850             case T_SHORT:
1851               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1852               break;
1853             default:
1854               break;
1855             }
1856             push(type, replacement);
1857           } else {
1858             push(type, append(load));
1859           }
1860         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1861           assert(field->type()->is_valuetype(), "Sanity check");
1862           ciValueKlass* value_klass = field->type()->as_value_klass();
1863           int flattening_offset = field->offset() - value_klass->first_field_offset();
1864           assert(field->type()->is_valuetype(), "Sanity check");
1865           scope()->set_wrote_final();
1866           scope()->set_wrote_fields();
1867           NewValueTypeInstance* new_instance = new NewValueTypeInstance(value_klass, state_before, false);
1868           _memory->new_instance(new_instance);
1869           apush(append_split(new_instance));
1870           copy_value_content(value_klass, obj, field->offset(), new_instance, value_klass->first_field_offset(),
1871                        state_before, needs_patching);
1872         }
1873       }
1874       break;
1875     }
1876     case Bytecodes::_withfield:
1877     case Bytecodes::_putfield: {
1878       Value val = pop(type);
1879       val->set_escaped();
1880       obj = apop();
1881       if (state_before == NULL) {
1882         state_before = copy_state_for_exception();
1883       }
1884       if (field->type()->basic_type() == T_BOOLEAN) {
1885         Value mask = append(new Constant(new IntConstant(1)));
1886         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1887       }
1888 
1889       if (!field->is_flattened()) {
1890         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1891         if (!needs_patching) store = _memory->store(store);
1892         if (store != NULL) {
1893           append(store);
1894         }
1895       } else {
1896         assert(field->type()->is_valuetype(), "Sanity check");
1897         ciValueKlass* value_klass = field->type()->as_value_klass();
1898         int flattening_offset = field->offset() - value_klass->first_field_offset();
1899         copy_value_content(value_klass, val, value_klass->first_field_offset(), obj, field->offset(),
1900                    state_before, needs_patching);
1901       }
1902       break;
1903     }
1904     default:
1905       ShouldNotReachHere();
1906       break;
1907   }
1908 }

1910 // Baseline version of withfield, allocate every time
1911 void GraphBuilder::withfield(int field_index)
1912 {
1913   bool will_link;
1914   ciField* field_modify = stream()->get_field(will_link);
1915   ciInstanceKlass* holder = field_modify->holder();
1916   BasicType field_type = field_modify->type()->basic_type();
1917   ValueType* type = as_ValueType(field_type);
1918 
1919   // call will_link again to determine if the field is valid.
1920   const bool needs_patching = !holder->is_loaded() ||
1921                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1922                               PatchALot;
1923 
1924 
1925   scope()->set_wrote_final();
1926   scope()->set_wrote_fields();
1927 
1928   const int offset = !needs_patching ? field_modify->offset() : -1;
1929 
1930   if (!holder->is_loaded()
1931       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1932       ) {
1933     ValueStack* state_before = copy_state_before();
1934     Value val = pop(type);
1935     Value obj = apop();
1936     apush(append_split(new WithField(state_before)));
1937     return;
1938   }
1939   ValueStack* state_before = copy_state_before();
1940 
1941   Value val = pop(type);
1942   Value obj = apop();
1943 
1944   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1945     int astore_index;
1946     ciBytecodeStream s(method());
1947     s.force_bci(bci());
1948     s.next();
1949     switch (s.cur_bc()) {
1950     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1951     case Bytecodes::_astore_0:  astore_index = 0; break;
1952     case Bytecodes::_astore_1:  astore_index = 1; break;
1953     case Bytecodes::_astore_2:  astore_index = 2; break;
1954     case Bytecodes::_astore_3:  astore_index = 3; break;
1955     default: astore_index = -1;
1956     }
1957 
1958     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1959       // We have a sequence like this, where we load a value object from a local slot,
1960       // and overwrite the same local slot with a modified copy of the value object.
1961       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1962       //      astore 9
1963       //      ...
1964       //      iload_0
1965       //      aload 9
1966       //      swap
1967       //      withfield #7 // Field x:I
1968       //      astore 9
1969       // If this object was created by defaultvalue, and has not escaped, and is not stored
1970       // in any other local slots, we can effectively treat the withfield/astore
1971       // sequence as a single putfield bytecode.
1972       push(objectType, obj);
1973       push(type, val);
1974       access_field(Bytecodes::_withfield);
1975       stream()->next(); // skip the next astore/astore_n bytecode.
1976       return;
1977     }
1978   }
1979 
1980   assert(holder->is_valuetype(), "must be a value klass");
1981   // Save the entire state and re-execute on deopt when executing withfield
1982   state_before->set_should_reexecute(true);
1983   NewValueTypeInstance* new_instance = new NewValueTypeInstance(holder->as_value_klass(), state_before, false);
1984   _memory->new_instance(new_instance);
1985   apush(append_split(new_instance));
1986 
1987   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1988     ciField* field = holder->nonstatic_field_at(i);
1989     int off = field->offset();
1990 
1991     if (field->offset() != offset) {
1992       if (field->is_flattened()) {
1993         assert(field->type()->is_valuetype(), "Sanity check");
1994         assert(field->type()->is_valuetype(), "Only value types can be flattened");
1995         ciValueKlass* vk = field->type()->as_value_klass();
1996         copy_value_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
1997       } else {
1998         // Only load those fields who are not modified
1999         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2000         Value replacement = append(load);
2001         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2002         append(store);
2003       }
2004     }
2005   }
2006 
2007   // Field to modify
2008   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2009     Value mask = append(new Constant(new IntConstant(1)));
2010     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2011   }
2012   if (field_modify->is_flattened()) {
2013     assert(field_modify->type()->is_valuetype(), "Only value types can be flattened");
2014     ciValueKlass* vk = field_modify->type()->as_value_klass();
2015     copy_value_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2016   } else {
2017     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2018     append(store);
2019   }
2020 }

2027 // How many arguments do we want to profile?
2028 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2029   int n = 0;
2030   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2031   start = has_receiver ? 1 : 0;
2032   if (profile_arguments()) {
2033     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2034     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2035       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2036     }
2037   }
2038   // If we are inlining then we need to collect arguments to profile parameters for the target
2039   if (profile_parameters() && target != NULL) {
2040     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2041       // The receiver is profiled on method entry so it's included in
2042       // the number of parameters but here we're only interested in
2043       // actual arguments.
2044       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2045     }
2046   }
2047   if (n > 0) {
2048     return new Values(n);
2049   }
2050   return NULL;
2051 }

2082 void GraphBuilder::invoke(Bytecodes::Code code) {
2083   bool will_link;
2084   ciSignature* declared_signature = NULL;
2085   ciMethod*             target = stream()->get_method(will_link, &declared_signature);
2086   ciKlass*              holder = stream()->get_declared_method_holder();
2087   const Bytecodes::Code bc_raw = stream()->cur_bc_raw();
2088   assert(declared_signature != NULL, "cannot be null");
2089   assert(will_link == target->is_loaded(), "");
2090 
2091   ciInstanceKlass* klass = target->holder();
2092   assert(!target->is_loaded() || klass->is_loaded(), "loaded target must imply loaded klass");
2093 
2094   // check if CHA possible: if so, change the code to invoke_special
2095   ciInstanceKlass* calling_klass = method()->holder();
2096   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);
2097   ciInstanceKlass* actual_recv = callee_holder;
2098 
2099   CompileLog* log = compilation()->log();
2100   if (log != NULL)
2101       log->elem("call method='%d' instr='%s'",
2102                 log->identify(target),
2103                 Bytecodes::name(code));
2104 
2105   // invoke-special-super
2106   if (bc_raw == Bytecodes::_invokespecial && !target->is_object_constructor()) {
2107     ciInstanceKlass* sender_klass =
2108           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2109                                                  calling_klass;
2110     if (sender_klass->is_interface()) {
2111       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2112       Value receiver = state()->stack_at(index);
2113       CheckCast* c = new CheckCast(sender_klass, receiver, copy_state_before());
2114       c->set_invokespecial_receiver_check();
2115       state()->stack_at_put(index, append_split(c));
2116     }
2117   }
2118 
2119   // Some methods are obviously bindable without any type checks so
2120   // convert them directly to an invokespecial or invokestatic.
2121   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2122     switch (bc_raw) {
2123     case Bytecodes::_invokevirtual:
2124       code = Bytecodes::_invokespecial;
2125       break;
2126     case Bytecodes::_invokehandle:
2127       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2128       break;
2129     default:
2130       break;
2131     }
2132   } else {
2133     if (bc_raw == Bytecodes::_invokehandle) {
2134       assert(!will_link, "should come here only for unlinked call");
2135       code = Bytecodes::_invokespecial;
2136     }
2137   }
2138 
2139   // Push appendix argument (MethodType, CallSite, etc.), if one.
2140   bool patch_for_appendix = false;
2141   int patching_appendix_arg = 0;
2142   if (Bytecodes::has_optional_appendix(bc_raw) && (!will_link || PatchALot)) {
2143     Value arg = append(new Constant(new ObjectConstant(compilation()->env()->unloaded_ciinstance()), copy_state_before()));
2144     apush(arg);
2145     patch_for_appendix = true;
2146     patching_appendix_arg = (will_link && stream()->has_appendix()) ? 0 : 1;
2147   } else if (stream()->has_appendix()) {
2148     ciObject* appendix = stream()->get_appendix();
2149     Value arg = append(new Constant(new ObjectConstant(appendix)));
2150     apush(arg);
2151   }
2152 
2153   ciMethod* cha_monomorphic_target = NULL;
2154   ciMethod* exact_target = NULL;
2155   Value better_receiver = NULL;
2156   if (UseCHA && DeoptC1 && target->is_loaded() &&
2157       !(// %%% FIXME: Are both of these relevant?
2158         target->is_method_handle_intrinsic() ||
2159         target->is_compiled_lambda_form()) &&
2160       !patch_for_appendix) {
2161     Value receiver = NULL;
2162     ciInstanceKlass* receiver_klass = NULL;
2163     bool type_is_exact = false;
2164     // try to find a precise receiver type
2165     if (will_link && !target->is_static()) {
2166       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2167       receiver = state()->stack_at(index);
2168       ciType* type = receiver->exact_type();
2169       if (type != NULL && type->is_loaded() &&
2170           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2171         receiver_klass = (ciInstanceKlass*) type;
2172         type_is_exact = true;
2173       }
2174       if (type == NULL) {
2175         type = receiver->declared_type();
2176         if (type != NULL && type->is_loaded() &&
2177             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2178           receiver_klass = (ciInstanceKlass*) type;
2179           if (receiver_klass->is_leaf_type() && !receiver_klass->is_final()) {
2180             // Insert a dependency on this type since
2181             // find_monomorphic_target may assume it's already done.
2182             dependency_recorder()->assert_leaf_type(receiver_klass);
2183             type_is_exact = true;
2184           }
2185         }
2186       }
2187     }
2188     if (receiver_klass != NULL && type_is_exact &&
2189         receiver_klass->is_loaded() && code != Bytecodes::_invokespecial) {
2190       // If we have the exact receiver type we can bind directly to
2191       // the method to call.
2192       exact_target = target->resolve_invoke(calling_klass, receiver_klass);
2193       if (exact_target != NULL) {
2194         target = exact_target;
2195         code = Bytecodes::_invokespecial;
2196       }
2197     }
2198     if (receiver_klass != NULL &&
2199         receiver_klass->is_subtype_of(actual_recv) &&
2200         actual_recv->is_initialized()) {
2201       actual_recv = receiver_klass;
2202     }
2203 
2204     if ((code == Bytecodes::_invokevirtual && callee_holder->is_initialized()) ||
2205         (code == Bytecodes::_invokeinterface && callee_holder->is_initialized() && !actual_recv->is_interface())) {
2206       // Use CHA on the receiver to select a more precise method.
2207       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2208     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {
2209       assert(callee_holder->is_interface(), "invokeinterface to non interface?");
2210       // If there is only one implementor of this interface then we
2211       // may be able bind this invoke directly to the implementing
2212       // klass but we need both a dependence on the single interface
2213       // and on the method we bind to.  Additionally since all we know
2214       // about the receiver type is the it's supposed to implement the
2215       // interface we have to insert a check that it's the class we
2216       // expect.  Interface types are not checked by the verifier so
2217       // they are roughly equivalent to Object.
2218       // The number of implementors for declared_interface is less or
2219       // equal to the number of implementors for target->holder() so
2220       // if number of implementors of target->holder() == 1 then
2221       // number of implementors for decl_interface is 0 or 1. If
2222       // it's 0 then no class implements decl_interface and there's
2223       // no point in inlining.
2224       ciInstanceKlass* declared_interface = callee_holder;
2225       ciInstanceKlass* singleton = declared_interface->unique_implementor();
2226       if (singleton != NULL &&
2227           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2228         assert(singleton != declared_interface, "not a unique implementor");
2229         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2230         if (cha_monomorphic_target != NULL) {
2231           if (cha_monomorphic_target->holder() != compilation()->env()->Object_klass()) {
2232             // If CHA is able to bind this invoke then update the class
2233             // to match that class, otherwise klass will refer to the
2234             // interface.
2235             klass = cha_monomorphic_target->holder();
2236             actual_recv = declared_interface;
2237 
2238             // insert a check it's really the expected class.
2239             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2240             c->set_incompatible_class_change_check();
2241             c->set_direct_compare(klass->is_final());
2242             // pass the result of the checkcast so that the compiler has
2243             // more accurate type info in the inlinee
2244             better_receiver = append_split(c);
2245           } else {
2246             cha_monomorphic_target = NULL; // subtype check against Object is useless
2247           }
2248         }
2249       }
2250     }
2251   }
2252 
2253   if (cha_monomorphic_target != NULL) {
2254     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2255     assert(!cha_monomorphic_target->is_abstract(), "");
2256     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2257       // If we inlined because CHA revealed only a single target method,
2258       // then we are dependent on that target method not getting overridden
2259       // by dynamic class loading.  Be sure to test the "static" receiver
2260       // dest_method here, as opposed to the actual receiver, which may
2261       // falsely lead us to believe that the receiver is final or private.
2262       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2263     }
2264     code = Bytecodes::_invokespecial;
2265   }
2266 
2267   // check if we could do inlining
2268   if (!PatchALot && Inline && target->is_loaded() &&
2269       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2270       && !patch_for_appendix) {
2271     // callee is known => check if we have static binding
2272     if (code == Bytecodes::_invokestatic  ||
2273         code == Bytecodes::_invokespecial ||
2274         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2275         code == Bytecodes::_invokedynamic) {
2276       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2277       // static binding => check if callee is ok
2278       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2279 
2280       CHECK_BAILOUT();
2281       clear_inline_bailout();
2282 
2283       if (success) {
2284         // Register dependence if JVMTI has either breakpoint
2285         // setting or hotswapping of methods capabilities since they may
2286         // cause deoptimization.
2287         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2288           dependency_recorder()->assert_evol_method(inline_target);
2289         }
2290         return;
2291       }
2292     } else {
2293       print_inlining(target, "no static binding", /*success*/ false);
2294     }
2295   } else {
2296     print_inlining(target, "not inlineable", /*success*/ false);
2297   }
2298 
2299   // If we attempted an inline which did not succeed because of a
2300   // bailout during construction of the callee graph, the entire
2301   // compilation has to be aborted. This is fairly rare and currently
2302   // seems to only occur for jasm-generated classes which contain
2303   // jsr/ret pairs which are not associated with finally clauses and
2304   // do not have exception handlers in the containing method, and are
2305   // therefore not caught early enough to abort the inlining without
2306   // corrupting the graph. (We currently bail out with a non-empty
2307   // stack at a ret in these situations.)
2308   CHECK_BAILOUT();
2309 
2310   // inlining not successful => standard invoke
2311   ValueType* result_type = as_ValueType(declared_signature->return_type());
2312   ValueStack* state_before = copy_state_exhandling();
2313 
2314   // The bytecode (code) might change in this method so we are checking this very late.
2315   const bool has_receiver =
2316     code == Bytecodes::_invokespecial   ||
2317     code == Bytecodes::_invokevirtual   ||
2318     code == Bytecodes::_invokeinterface;
2319   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2320   Value recv = has_receiver ? apop() : NULL;
2321   int vtable_index = Method::invalid_vtable_index;
2322 
2323 #ifdef SPARC
2324   // Currently only supported on Sparc.
2325   // The UseInlineCaches only controls dispatch to invokevirtuals for
2326   // loaded classes which we weren't able to statically bind.
2327   if (!UseInlineCaches && target->is_loaded() && code == Bytecodes::_invokevirtual
2328       && !target->can_be_statically_bound()) {
2329     // Find a vtable index if one is available
2330     // For arrays, callee_holder is Object. Resolving the call with
2331     // Object would allow an illegal call to finalize() on an
2332     // array. We use holder instead: illegal calls to finalize() won't
2333     // be compiled as vtable calls (IC call resolution will catch the
2334     // illegal call) and the few legal calls on array types won't be
2335     // either.
2336     vtable_index = target->resolve_vtable_index(calling_klass, holder);
2337   }
2338 #endif
2339 
2340   // A null check is required here (when there is a receiver) for any of the following cases
2341   // - invokespecial, always need a null check.
2342   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2343   //   and require null checking. If the target is loaded a null check is emitted here.
2344   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2345   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2346   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2347   //   potentially fail, and can't have the null check before the resolution.)
2348   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2349   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2350   //
2351   // Normal invokevirtual will perform the null check during lookup
2352 
2353   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2354       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2355 
2356   if (need_null_check) {
2357     if (recv != NULL) {
2358       null_check(recv);
2359     }
2360 
2361     if (is_profiling()) {
2362       // Note that we'd collect profile data in this method if we wanted it.
2363       compilation()->set_would_profile(true);
2364 
2365       if (profile_calls()) {
2366         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2367         ciKlass* target_klass = NULL;
2368         if (cha_monomorphic_target != NULL) {
2369           target_klass = cha_monomorphic_target->holder();
2370         } else if (exact_target != NULL) {
2371           target_klass = exact_target->holder();
2372         }
2373         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2374       }
2375     }
2376   }
2377 
2378   if (recv != NULL) {
2379     recv->set_escaped();
2380   }
2381   for (int i=0; i<args->length(); i++) {
2382     args->at(0)->set_escaped();
2383   }
2384 
2385   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2386                               declared_signature->returns_never_null());
2387   // push result
2388   append_split(result);
2389 
2390   if (result_type != voidType) {
2391     if (method()->is_strict()) {
2392       push(result_type, round_fp(result));
2393     } else {
2394       push(result_type, result);
2395     }
2396   }
2397   if (profile_return() && result_type->is_object_kind()) {
2398     profile_return_type(result, target);
2399   }
2400 }

2583 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2584   Canonicalizer canon(compilation(), instr, bci);
2585   Instruction* i1 = canon.canonical();
2586   if (i1->is_linked() || !i1->can_be_linked()) {
2587     // Canonicalizer returned an instruction which was already
2588     // appended so simply return it.
2589     return i1;
2590   }
2591 
2592   if (UseLocalValueNumbering) {
2593     // Lookup the instruction in the ValueMap and add it to the map if
2594     // it's not found.
2595     Instruction* i2 = vmap()->find_insert(i1);
2596     if (i2 != i1) {
2597       // found an entry in the value map, so just return it.
2598       assert(i2->is_linked(), "should already be linked");
2599       return i2;
2600     }
2601     ValueNumberingEffects vne(vmap());
2602     i1->visit(&vne);
2603   }
2604 
2605   // i1 was not eliminated => append it
2606   assert(i1->next() == NULL, "shouldn't already be linked");
2607   _last = _last->set_next(i1, canon.bci());
2608 
2609   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2610     // set the bailout state but complete normal processing.  We
2611     // might do a little more work before noticing the bailout so we
2612     // want processing to continue normally until it's noticed.
2613     bailout("Method and/or inlining is too large");
2614   }
2615 
2616 #ifndef PRODUCT
2617   if (PrintIRDuringConstruction) {
2618     InstructionPrinter ip;
2619     ip.print_line(i1);
2620     if (Verbose) {
2621       state()->print();
2622     }
2623   }
2624 #endif
2625 
2626   // save state after modification of operand stack for StateSplit instructions
2627   StateSplit* s = i1->as_StateSplit();
2628   if (s != NULL) {
2629     if (EliminateFieldAccess) {
2630       Intrinsic* intrinsic = s->as_Intrinsic();
2631       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2632         _memory->kill();
2633       }
2634     }
2635     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2636   }
2637 
2638   // set up exception handlers for this instruction if necessary
2639   if (i1->can_trap()) {
2640     i1->set_exception_handlers(handle_exception(i1));
2641     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2642   }
2643   return i1;
2644 }

2678 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2679   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2680     assert(instruction->exception_state() == NULL
2681            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2682            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2683            "exception_state should be of exception kind");
2684     return new XHandlers();
2685   }
2686 
2687   XHandlers*  exception_handlers = new XHandlers();
2688   ScopeData*  cur_scope_data = scope_data();
2689   ValueStack* cur_state = instruction->state_before();
2690   ValueStack* prev_state = NULL;
2691   int scope_count = 0;
2692 
2693   assert(cur_state != NULL, "state_before must be set");
2694   do {
2695     int cur_bci = cur_state->bci();
2696     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2697     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2698 
2699     // join with all potential exception handlers
2700     XHandlers* list = cur_scope_data->xhandlers();
2701     const int n = list->length();
2702     for (int i = 0; i < n; i++) {
2703       XHandler* h = list->handler_at(i);
2704       if (h->covers(cur_bci)) {
2705         // h is a potential exception handler => join it
2706         compilation()->set_has_exception_handlers(true);
2707 
2708         BlockBegin* entry = h->entry_block();
2709         if (entry == block()) {
2710           // It's acceptable for an exception handler to cover itself
2711           // but we don't handle that in the parser currently.  It's
2712           // very rare so we bailout instead of trying to handle it.
2713           BAILOUT_("exception handler covers itself", exception_handlers);
2714         }
2715         assert(entry->bci() == h->handler_bci(), "must match");
2716         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2717 
2718         // previously this was a BAILOUT, but this is not necessary
2719         // now because asynchronous exceptions are not handled this way.
2720         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2721 
2722         // xhandler start with an empty expression stack
2723         if (cur_state->stack_size() != 0) {
2724           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2725         }
2726         if (instruction->exception_state() == NULL) {
2727           instruction->set_exception_state(cur_state);
2728         }
2729 
2730         // Note: Usually this join must work. However, very
2731         // complicated jsr-ret structures where we don't ret from
2732         // the subroutine can cause the objects on the monitor
2733         // stacks to not match because blocks can be parsed twice.
2734         // The only test case we've seen so far which exhibits this
2735         // problem is caught by the infinite recursion test in
2736         // GraphBuilder::jsr() if the join doesn't work.
2737         if (!entry->try_merge(cur_state)) {
2738           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2739         }
2740 
2741         // add current state for correct handling of phi functions at begin of xhandler
2742         int phi_operand = entry->add_exception_state(cur_state);
2743 
2744         // add entry to the list of xhandlers of this block
2745         _block->add_exception_handler(entry);
2746 
2747         // add back-edge from xhandler entry to this block
2748         if (!entry->is_predecessor(_block)) {
2749           entry->add_predecessor(_block);
2750         }
2751 
2752         // clone XHandler because phi_operand and scope_count can not be shared
2753         XHandler* new_xhandler = new XHandler(h);
2754         new_xhandler->set_phi_operand(phi_operand);
2755         new_xhandler->set_scope_count(scope_count);
2756         exception_handlers->append(new_xhandler);
2757 
2758         // fill in exception handler subgraph lazily
2759         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2760         cur_scope_data->add_to_work_list(entry);
2761 
2762         // stop when reaching catchall
2763         if (h->catch_type() == 0) {
2764           return exception_handlers;
2765         }
2766       }
2767     }
2768 
2769     if (exception_handlers->length() == 0) {
2770       // This scope and all callees do not handle exceptions, so the local
2771       // variables of this scope are not needed. However, the scope itself is
2772       // required for a correct exception stack trace -> clear out the locals.
2773       if (_compilation->env()->should_retain_local_variables()) {
2774         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2775       } else {
2776         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2777       }
2778       if (prev_state != NULL) {
2779         prev_state->set_caller_state(cur_state);
2780       }
2781       if (instruction->exception_state() == NULL) {
2782         instruction->set_exception_state(cur_state);
2783       }
2784     }
2785 
2786     // Set up iteration for next time.
2787     // If parsing a jsr, do not grab exception handlers from the
2788     // parent scopes for this method (already got them, and they
2789     // needed to be cloned)
2790 
2791     while (cur_scope_data->parsing_jsr()) {
2792       cur_scope_data = cur_scope_data->parent();
2793     }
2794 
2795     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2796     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2797 
2798     prev_state = cur_state;
2799     cur_state = cur_state->caller_state();
2800     cur_scope_data = cur_scope_data->parent();
2801     scope_count++;
2802   } while (cur_scope_data != NULL);
2803 
2804   return exception_handlers;
2805 }

2836     // already tried to simplify phi before

2838   } else if (phi->is_set(Phi::visited)) {
2839     // break cycles in phi functions
2840     return phi;
2841   } else if (phi->type()->is_illegal()) {
2842     // illegal phi functions are ignored anyway
2843     return phi;
2844 
2845   } else {
2846     // mark phi function as processed to break cycles in phi functions
2847     phi->set(Phi::visited);
2848 
2849     // simplify x = [y, x] and x = [y, y] to y
2850     Value subst = NULL;
2851     int opd_count = phi->operand_count();
2852     for (int i = 0; i < opd_count; i++) {
2853       Value opd = phi->operand_at(i);
2854       assert(opd != NULL, "Operand must exist!");
2855 
2856       if (opd->type()->is_illegal()) {
2857         // if one operand is illegal, the entire phi function is illegal
2858         phi->make_illegal();
2859         phi->clear(Phi::visited);
2860         return phi;
2861       }
2862 
2863       Value new_opd = simplify(opd);
2864       assert(new_opd != NULL, "Simplified operand must exist!");
2865 
2866       if (new_opd != phi && new_opd != subst) {
2867         if (subst == NULL) {
2868           subst = new_opd;
2869         } else {
2870           // no simplification possible
2871           phi->set(Phi::cannot_simplify);
2872           phi->clear(Phi::visited);
2873           return phi;
2874         }
2875       }
2876     }
2877 
2878     // sucessfully simplified phi function
2879     assert(subst != NULL, "illegal phi function");
2880     _has_substitutions = true;
2881     phi->clear(Phi::visited);
2882     phi->set_subst(subst);
2883 
2884 #ifndef PRODUCT
2885     if (PrintPhiFunctions) {
2886       tty->print_cr("simplified phi function %c%d to %c%d (Block B%d)", phi->type()->tchar(), phi->id(), subst->type()->tchar(), subst->id(), phi->block()->block_id());
2887     }
2888 #endif
2889 
2890     return subst;
2891   }
2892 }

3236 void GraphBuilder::iterate_all_blocks(bool start_in_current_block_for_inlining) {
3237   do {
3238     if (start_in_current_block_for_inlining && !bailed_out()) {
3239       iterate_bytecodes_for_block(0);
3240       start_in_current_block_for_inlining = false;
3241     } else {
3242       BlockBegin* b;
3243       while ((b = scope_data()->remove_from_work_list()) != NULL) {
3244         if (!b->is_set(BlockBegin::was_visited_flag)) {
3245           if (b->is_set(BlockBegin::osr_entry_flag)) {
3246             // we're about to parse the osr entry block, so make sure
3247             // we setup the OSR edge leading into this block so that
3248             // Phis get setup correctly.
3249             setup_osr_entry_block();
3250             // this is no longer the osr entry block, so clear it.
3251             b->clear(BlockBegin::osr_entry_flag);
3252           }
3253           b->set(BlockBegin::was_visited_flag);
3254           connect_to_end(b);
3255         }
3256       }
3257     }
3258   } while (!bailed_out() && !scope_data()->is_work_list_empty());
3259 }

3346   BlockBegin* start = new BlockBegin(0);

3366       new_header_block,
3367       osr_entry

3385 void GraphBuilder::setup_osr_entry_block() {
3386   assert(compilation()->is_osr_compile(), "only for osrs");
3387 
3388   int osr_bci = compilation()->osr_bci();
3389   ciBytecodeStream s(method());
3390   s.reset_to_bci(osr_bci);
3391   s.next();
3392   scope_data()->set_stream(&s);
3393 
3394   // create a new block to be the osr setup code
3395   _osr_entry = new BlockBegin(osr_bci);
3396   _osr_entry->set(BlockBegin::osr_entry_flag);
3397   _osr_entry->set_depth_first_number(0);
3398   BlockBegin* target = bci2block()->at(osr_bci);
3399   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3400   // the osr entry has no values for locals
3401   ValueStack* state = target->state()->copy();
3402   _osr_entry->set_state(state);
3403 
3404   kill_all();
3405   _block = _osr_entry;
3406   _state = _osr_entry->state()->copy();
3407   assert(_state->bci() == osr_bci, "mismatch");
3408   _last  = _osr_entry;
3409   Value e = append(new OsrEntry());
3410   e->set_needs_null_check(false);
3411 
3412   // OSR buffer is
3413   //
3414   // locals[nlocals-1..0]
3415   // monitors[number_of_locks-1..0]
3416   //
3417   // locals is a direct copy of the interpreter frame so in the osr buffer
3418   // so first slot in the local array is the last local from the interpreter
3419   // and last slot is local[0] (receiver) from the interpreter
3420   //
3421   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3422   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3423   // in the interpreter frame (the method lock if a sync method)
3424 
3425   // Initialize monitors in the compiled activation.
3426 
3427   int index;
3428   Value local;
3429 
3430   // find all the locals that the interpreter thinks contain live oops
3431   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3432 
3433   // compute the offset into the locals so that we can treat the buffer
3434   // as if the locals were still in the interpreter frame
3435   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3436   for_each_local_value(state, index, local) {
3437     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3438     Value get;
3439     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3440       // The interpreter thinks this local is dead but the compiler
3441       // doesn't so pretend that the interpreter passed in null.
3442       get = append(new Constant(objectNull));
3443     } else {
3444       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3445                                     append(new Constant(new IntConstant(offset))),
3446                                     0,
3447                                     true /*unaligned*/, true /*wide*/));
3448     }
3449     _state->store_local(index, get);
3450   }
3451 
3452   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3453 
3454   assert(state->caller_state() == NULL, "should be top scope");
3455   state->clear_locals();
3456   Goto* g = new Goto(target, false);
3457   append(g);
3458   _osr_entry->set_end(g);
3459   target->merge(_osr_entry->end()->state());
3460 
3461   scope_data()->set_stream(NULL);
3462 }

3465 ValueStack* GraphBuilder::state_at_entry() {
3466   ValueStack* state = new ValueStack(scope(), NULL);
3467 
3468   // Set up locals for receiver
3469   int idx = 0;
3470   if (!method()->is_static()) {
3471     // we should always see the receiver
3472     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3473              /*receiver*/ true, /*never_null*/ method()->holder()->is_value_array_klass()));
3474     idx = 1;
3475   }
3476 
3477   // Set up locals for incoming arguments
3478   ciSignature* sig = method()->signature();
3479   for (int i = 0; i < sig->count(); i++) {
3480     ciType* type = sig->type_at(i);
3481     BasicType basic_type = type->basic_type();
3482     // don't allow T_ARRAY to propagate into locals types
3483     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3484     ValueType* vt = as_ValueType(basic_type);
3485     state->store_local(idx, new Local(type, vt, idx, false, sig->is_never_null_at(i)));
3486     idx += type->size();
3487   }
3488 
3489   // lock synchronized method
3490   if (method()->is_synchronized()) {
3491     state->lock(NULL);
3492   }
3493 
3494   return state;
3495 }

3498 GraphBuilder::GraphBuilder(Compilation* compilation, IRScope* scope)
3499   : _scope_data(NULL)
3500   , _compilation(compilation)
3501   , _memory(new MemoryBuffer())
3502   , _inline_bailout_msg(NULL)
3503   , _instruction_count(0)
3504   , _osr_entry(NULL)
3505 {
3506   int osr_bci = compilation->osr_bci();
3507 
3508   // determine entry points and bci2block mapping
3509   BlockListBuilder blm(compilation, scope, osr_bci);
3510   CHECK_BAILOUT();
3511 
3512   BlockList* bci2block = blm.bci2block();
3513   BlockBegin* start_block = bci2block->at(0);
3514 
3515   push_root_scope(scope, bci2block, start_block);
3516 
3517   // setup state for std entry
3518   _initial_state = state_at_entry();
3519   start_block->merge(_initial_state);
3520 
3521   // complete graph
3522   _vmap        = new ValueMap();
3523   switch (scope->method()->intrinsic_id()) {
3524   case vmIntrinsics::_dabs          : // fall through
3525   case vmIntrinsics::_dsqrt         : // fall through
3526   case vmIntrinsics::_dsin          : // fall through
3527   case vmIntrinsics::_dcos          : // fall through
3528   case vmIntrinsics::_dtan          : // fall through
3529   case vmIntrinsics::_dlog          : // fall through
3530   case vmIntrinsics::_dlog10        : // fall through
3531   case vmIntrinsics::_dexp          : // fall through
3532   case vmIntrinsics::_dpow          : // fall through
3533     {
3534       // Compiles where the root method is an intrinsic need a special
3535       // compilation environment because the bytecodes for the method
3536       // shouldn't be parsed during the compilation, only the special
3537       // Intrinsic node should be emitted.  If this isn't done the the
3538       // code for the inlined version will be different than the root
3539       // compiled version which could lead to monotonicity problems on
3540       // intel.
3541       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3542         BAILOUT("failed to inline intrinsic, method not annotated");
3543       }
3544 
3545       // Set up a stream so that appending instructions works properly.
3546       ciBytecodeStream s(scope->method());
3547       s.reset_to_bci(0);
3548       scope_data()->set_stream(&s);
3549       s.next();
3550 
3551       // setup the initial block state
3552       _block = start_block;
3553       _state = start_block->state()->copy_for_parsing();
3554       _last  = start_block;
3555       load_local(doubleType, 0);
3556       if (scope->method()->intrinsic_id() == vmIntrinsics::_dpow) {
3557         load_local(doubleType, 2);
3558       }
3559 
3560       // Emit the intrinsic node.
3561       bool result = try_inline_intrinsics(scope->method());
3562       if (!result) BAILOUT("failed to inline intrinsic");
3563       method_return(dpop());
3564 
3565       // connect the begin and end blocks and we're all done.
3566       BlockEnd* end = last()->as_BlockEnd();
3567       block()->set_end(end);
3568       break;
3569     }
3570 
3571   case vmIntrinsics::_Reference_get:
3572     {
3573       {
3574         // With java.lang.ref.reference.get() we must go through the
3575         // intrinsic - when G1 is enabled - even when get() is the root
3576         // method of the compile so that, if necessary, the value in
3577         // the referent field of the reference object gets recorded by
3578         // the pre-barrier code.
3579         // Specifically, if G1 is enabled, the value in the referent
3580         // field is recorded by the G1 SATB pre barrier. This will
3581         // result in the referent being marked live and the reference
3582         // object removed from the list of discovered references during
3583         // reference processing.
3584         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3585           BAILOUT("failed to inline intrinsic, method not annotated");
3586         }
3587 
3588         // Also we need intrinsic to prevent commoning reads from this field
3589         // across safepoint since GC can change its value.
3590 
3591         // Set up a stream so that appending instructions works properly.
3592         ciBytecodeStream s(scope->method());
3593         s.reset_to_bci(0);
3594         scope_data()->set_stream(&s);
3595         s.next();
3596 
3597         // setup the initial block state
3598         _block = start_block;
3599         _state = start_block->state()->copy_for_parsing();
3600         _last  = start_block;
3601         load_local(objectType, 0);
3602 
3603         // Emit the intrinsic node.
3604         bool result = try_inline_intrinsics(scope->method());
3605         if (!result) BAILOUT("failed to inline intrinsic");
3606         method_return(apop());
3607 
3608         // connect the begin and end blocks and we're all done.
3609         BlockEnd* end = last()->as_BlockEnd();
3610         block()->set_end(end);
3611         break;
3612       }
3613       // Otherwise, fall thru
3614     }
3615 
3616   default:
3617     scope_data()->add_to_work_list(start_block);
3618     iterate_all_blocks();
3619     break;
3620   }
3621   CHECK_BAILOUT();
3622 
3623   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3624 
3625   eliminate_redundant_phis(_start);
3626 
3627   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3628   // for osr compile, bailout if some requirements are not fulfilled
3629   if (osr_bci != -1) {
3630     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3631     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3632       BAILOUT("osr entry must have been visited for osr compile");
3633     }
3634 
3635     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3636     if (!osr_block->state()->stack_is_empty()) {
3637       BAILOUT("stack not empty at OSR entry point");
3638     }
3639   }
3640 #ifndef PRODUCT
3641   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3642 #endif
3643 }

3690 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3691   const char* msg = NULL;
3692 
3693   // clear out any existing inline bailout condition
3694   clear_inline_bailout();
3695 
3696   // exclude methods we don't want to inline
3697   msg = should_not_inline(callee);
3698   if (msg != NULL) {
3699     print_inlining(callee, msg, /*success*/ false);
3700     return false;
3701   }
3702 
3703   // method handle invokes
3704   if (callee->is_method_handle_intrinsic()) {
3705     if (try_method_handle_inline(callee, ignore_return)) {
3706       if (callee->has_reserved_stack_access()) {
3707         compilation()->set_has_reserved_stack_access(true);
3708       }
3709       return true;
3710     }
3711     return false;
3712   }
3713 
3714   // handle intrinsics
3715   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3716       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3717     if (try_inline_intrinsics(callee, ignore_return)) {
3718       print_inlining(callee, "intrinsic");
3719       if (callee->has_reserved_stack_access()) {
3720         compilation()->set_has_reserved_stack_access(true);
3721       }
3722       return true;
3723     }
3724     // try normal inlining
3725   }
3726 
3727   // certain methods cannot be parsed at all
3728   msg = check_can_parse(callee);
3729   if (msg != NULL) {
3730     print_inlining(callee, msg, /*success*/ false);
3731     return false;
3732   }
3733 
3734   // If bytecode not set use the current one.
3735   if (bc == Bytecodes::_illegal) {
3736     bc = code();
3737   }
3738   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3739     if (callee->has_reserved_stack_access()) {
3740       compilation()->set_has_reserved_stack_access(true);
3741     }
3742     return true;
3743   }
3744 
3745   // Entire compilation could fail during try_inline_full call.
3746   // In that case printing inlining decision info is useless.
3747   if (!bailed_out())
3748     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3749 
3750   return false;
3751 }

3898 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3899   // Introduce a new callee continuation point - all Ret instructions
3900   // will be replaced with Gotos to this point.
3901   BlockBegin* cont = block_at(next_bci());
3902   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3903 
3904   // Note: can not assign state to continuation yet, as we have to
3905   // pick up the state from the Ret instructions.
3906 
3907   // Push callee scope
3908   push_scope_for_jsr(cont, jsr_dest_bci);
3909 
3910   // Temporarily set up bytecode stream so we can append instructions
3911   // (only using the bci of this stream)
3912   scope_data()->set_stream(scope_data()->parent()->stream());
3913 
3914   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3915   assert(jsr_start_block != NULL, "jsr start block must exist");
3916   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3917   Goto* goto_sub = new Goto(jsr_start_block, false);
3918   // Must copy state to avoid wrong sharing when parsing bytecodes
3919   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3920   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3921   append(goto_sub);
3922   _block->set_end(goto_sub);
3923   _last = _block = jsr_start_block;
3924 
3925   // Clear out bytecode stream
3926   scope_data()->set_stream(NULL);
3927 
3928   scope_data()->add_to_work_list(jsr_start_block);
3929 
3930   // Ready to resume parsing in subroutine
3931   iterate_all_blocks();
3932 
3933   // If we bailed out during parsing, return immediately (this is bad news)
3934   CHECK_BAILOUT_(false);
3935 
3936   // Detect whether the continuation can actually be reached. If not,
3937   // it has not had state set by the join() operations in
3938   // iterate_bytecodes_for_block()/ret() and we should not touch the
3939   // iteration state. The calling activation of
3940   // iterate_bytecodes_for_block will then complete normally.
3941   if (cont->state() != NULL) {
3942     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3943       // add continuation to work list instead of parsing it immediately
3944       scope_data()->parent()->add_to_work_list(cont);
3945     }
3946   }
3947 
3948   assert(jsr_continuation() == cont, "continuation must not have changed");
3949   assert(!jsr_continuation()->is_set(BlockBegin::was_visited_flag) ||
3950          jsr_continuation()->is_set(BlockBegin::parser_loop_header_flag),
3951          "continuation can only be visited in case of backward branches");
3952   assert(_last && _last->as_BlockEnd(), "block must have end");
3953 
3954   // continuation is in work list, so end iteration of current block
3955   _skip_block = true;
3956   pop_scope_for_jsr();
3957 
3958   return true;
3959 }

4050   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4051     INLINE_BAILOUT("inlining prohibited by policy");
4052   }
4053   // first perform tests of things it's not possible to inline
4054   if (callee->has_exception_handlers() &&
4055       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4056   if (callee->is_synchronized() &&
4057       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4058   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4059   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4060 
4061   // Proper inlining of methods with jsrs requires a little more work.
4062   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4063 
4064   if (strict_fp_requires_explicit_rounding &&
4065       method()->is_strict() != callee->is_strict()) {
4066 #ifdef IA32
4067     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4068     // When SSE2 is present, no special handling is needed.
4069     if (UseSSE < 2) {
4070       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4071     }
4072 #else
4073     Unimplemented();
4074 #endif // IA32
4075   }
4076 
4077   if (is_profiling() && !callee->ensure_method_data()) {
4078     INLINE_BAILOUT("mdo allocation failed");
4079   }
4080 
4081   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4082   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4083 
4084   const int args_base = state()->stack_size() - callee->arg_size();
4085   assert(args_base >= 0, "stack underflow during inlining");
4086 
4087   Value recv = NULL;
4088   if (has_receiver) {
4089     assert(!callee->is_static(), "callee must not be static");
4090     assert(callee->arg_size() > 0, "must have at least a receiver");
4091 
4092     recv = state()->stack_at(args_base);
4093     if (recv->is_null_obj()) {
4094       INLINE_BAILOUT("receiver is always null");
4095     }
4096   }
4097 
4098   // now perform tests that are based on flag settings
4099   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4100   if (callee->force_inline() || inlinee_by_directive) {
4101     if (inline_level() > MaxForceInlineLevel                    ) INLINE_BAILOUT("MaxForceInlineLevel");
4102     if (recursive_inline_level(callee) > MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4103 
4104     const char* msg = "";
4105     if (callee->force_inline())  msg = "force inline by annotation";
4106     if (inlinee_by_directive)    msg = "force inline by CompileCommand";
4107     print_inlining(callee, msg);
4108   } else {
4109     // use heuristic controls on inlining
4110     if (inline_level() > MaxInlineLevel                         ) INLINE_BAILOUT("inlining too deep");
4111     if (recursive_inline_level(callee) > MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4112     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4113 
4114     // don't inline throwable methods unless the inlining tree is rooted in a throwable class
4115     if (callee->name() == ciSymbol::object_initializer_name() &&
4116         callee->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4117       // Throwable constructor call
4118       IRScope* top = scope();
4119       while (top->caller() != NULL) {
4120         top = top->caller();
4121       }
4122       if (!top->method()->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4123         INLINE_BAILOUT("don't inline Throwable constructors");
4124       }
4125     }
4126 
4127     if (compilation()->env()->num_inlined_bytecodes() > DesiredMethodLimit) {
4128       INLINE_BAILOUT("total inlining greater than DesiredMethodLimit");
4129     }

4135   // GraphBuilder level, do not cause bailout just of the inlining but

4140   // Insert null check if necessary
4141   if (has_receiver) {
4142     // note: null check must happen even if first instruction of callee does
4143     //       an implicit null check since the callee is in a different scope

4222   BlockBegin* sync_handler = NULL;
4223 
4224   // Inline the locking of the receiver if the callee is synchronized

4226     lock = callee->is_static() ? append(new Constant(new InstanceConstant(callee->holder()->java_mirror())))

4275   // practice, we have only traversed the continuation if we are
4276   // inlining into a subroutine

4283   // first is an optimization: during parsing of the callee we
4284   // generated at least one Goto to the continuation block. If we

4292   if (num_returns() == 1
4293       && block() == orig_block
4294       && block() == inline_cleanup_block()) {
4295     _last  = inline_cleanup_return_prev();
4296     _state = inline_cleanup_state();
4297   } else if (continuation_preds == cont->number_of_preds()) {
4298     // Inlining caused that the instructions after the invoke in the
4299     // caller are not reachable any more. So skip filling this block
4300     // with instructions!
4301     assert(cont == continuation(), "");
4302     assert(_last && _last->as_BlockEnd(), "");
4303     _skip_block = true;
4304   } else {
4305     // Resume parsing in continuation block unless it was already parsed.
4306     // Note that if we don't change _last here, iteration in
4307     // iterate_bytecodes_for_block will stop when we return.
4308     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4309       // add continuation to work list instead of parsing it immediately
4310       assert(_last && _last->as_BlockEnd(), "");
4311       scope_data()->parent()->add_to_work_list(continuation());
4312       _skip_block = true;
4313     }
4314   }
4315 
4316   // Fill the exception handler for synchronized methods with instructions
4317   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4318     fill_sync_handler(lock, sync_handler);
4319   } else {
4320     pop_scope();
4321   }
4322 
4323   compilation()->notice_inlined_method(callee);
4324 
4325   return true;
4326 }

4329 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4330   ValueStack* state_before = copy_state_before();
4331   vmIntrinsics::ID iid = callee->intrinsic_id();
4332   switch (iid) {
4333   case vmIntrinsics::_invokeBasic:
4334     {
4335       // get MethodHandle receiver
4336       const int args_base = state()->stack_size() - callee->arg_size();
4337       ValueType* type = state()->stack_at(args_base)->type();
4338       if (type->is_constant()) {
4339         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4340         // We don't do CHA here so only inline static and statically bindable methods.
4341         if (target->is_static() || target->can_be_statically_bound()) {
4342           if (ciMethod::is_consistent_info(callee, target)) {
4343             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4344             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4345             if (try_inline(target, /*holder_known*/ true, ignore_return, bc)) {
4346               return true;
4347             }
4348           } else {
4349             print_inlining(target, "signatures mismatch", /*success*/ false);
4350           }
4351         } else {
4352           print_inlining(target, "not static or statically bindable", /*success*/ false);
4353         }
4354       } else {
4355         print_inlining(callee, "receiver not constant", /*success*/ false);
4356       }
4357     }
4358     break;
4359 
4360   case vmIntrinsics::_linkToVirtual:
4361   case vmIntrinsics::_linkToStatic:
4362   case vmIntrinsics::_linkToSpecial:
4363   case vmIntrinsics::_linkToInterface:
4364     {
4365       // pop MemberName argument
4366       const int args_base = state()->stack_size() - callee->arg_size();
4367       ValueType* type = apop()->type();
4368       if (type->is_constant()) {
4369         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4370         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4371         // If the target is another method handle invoke, try to recursively get
4372         // a better target.
4373         if (target->is_method_handle_intrinsic()) {
4374           if (try_method_handle_inline(target, ignore_return)) {
4375             return true;
4376           }
4377         } else if (!ciMethod::is_consistent_info(callee, target)) {
4378           print_inlining(target, "signatures mismatch", /*success*/ false);
4379         } else {
4380           ciSignature* signature = target->signature();
4381           const int receiver_skip = target->is_static() ? 0 : 1;
4382           // Cast receiver to its type.
4383           if (!target->is_static()) {
4384             ciKlass* tk = signature->accessing_klass();
4385             Value obj = state()->stack_at(args_base);
4386             if (obj->exact_type() == NULL &&
4387                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4388               TypeCast* c = new TypeCast(tk, obj, state_before);
4389               append(c);
4390               state()->stack_at_put(args_base, c);
4391             }
4392           }
4393           // Cast reference arguments to its type.
4394           for (int i = 0, j = 0; i < signature->count(); i++) {
4395             ciType* t = signature->type_at(i);
4396             if (t->is_klass()) {
4397               ciKlass* tk = t->as_klass();
4398               Value obj = state()->stack_at(args_base + receiver_skip + j);
4399               if (obj->exact_type() == NULL &&
4400                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4401                 TypeCast* c = new TypeCast(t, obj, state_before);
4402                 append(c);
4403                 state()->stack_at_put(args_base + receiver_skip + j, c);
4404               }
4405             }
4406             j += t->size();  // long and double take two slots
4407           }
4408           // We don't do CHA here so only inline static and statically bindable methods.
4409           if (target->is_static() || target->can_be_statically_bound()) {
4410             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4411             if (try_inline(target, /*holder_known*/ true, ignore_return, bc)) {
4412               return true;
4413             }
4414           } else {
4415             print_inlining(target, "not static or statically bindable", /*success*/ false);
4416           }
4417         }
4418       } else {
4419         print_inlining(callee, "MemberName not constant", /*success*/ false);
4420       }
4421     }
4422     break;
4423 
4424   default:
4425     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4426     break;
4427   }
4428   set_state(state_before->copy_for_parsing());
4429   return false;
4430 }
