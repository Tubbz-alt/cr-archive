 49 void Jfr::on_create_vm_1() {
 50   if (!JfrRecorder::on_create_vm_1()) {
 51     vm_exit_during_initialization("Failure when starting JFR on_create_vm_1");
 52   }
 53 }

 67 void Jfr::on_unloading_classes() {
 68   if (JfrRecorder::is_created()) {
 69     JfrCheckpointManager::write_type_set_for_unloaded_classes();
 70   }
 71 }
 72 
 73 void Jfr::on_thread_start(Thread* t) {
 74   JfrThreadLocal::on_start(t);
 75 }
 76 
 77 void Jfr::on_thread_exit(Thread* t) {
 78   JfrThreadLocal::on_exit(t);
 79 }
 80 
 81 void Jfr::exclude_thread(Thread* t) {
 82   JfrThreadLocal::exclude(t);
 83 }
 84 
 85 void Jfr::include_thread(Thread* t) {
 86   JfrThreadLocal::include(t);
 87 }
 88 
 89 bool Jfr::is_excluded(Thread* t) {
 90   return t != NULL && t->jfr_thread_local()->is_excluded();
 91 }
 92 
 93 void Jfr::on_java_thread_dismantle(JavaThread* jt) {
 94   if (JfrRecorder::is_recording()) {
 95     JfrCheckpointManager::write_thread_checkpoint(jt);
 96   }
 97 }
 98 
 99 void Jfr::on_vm_shutdown(bool exception_handler) {
100   if (JfrRecorder::is_recording()) {
101     JfrEmergencyDump::on_vm_shutdown(exception_handler);
102   }
103 }
104 
105 void Jfr::on_vm_error_report(outputStream* st) {
106   if (JfrRecorder::is_recording()) {
107     JfrRepository::on_vm_error_report(st);
108   }
109 }
110 
111 void Jfr::weak_oops_do(BoolObjectClosure* is_alive, OopClosure* f) {
112   if (LeakProfiler::is_running()) {
113     LeakProfiler::weak_oops_do(is_alive, f);
114   }
115 }
