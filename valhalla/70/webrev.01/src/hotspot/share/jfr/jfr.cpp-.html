 48 void Jfr::on_create_vm_1() {
 49   if (!JfrRecorder::on_create_vm_1()) {
 50     vm_exit_during_initialization("Failure when starting JFR on_create_vm_1");
 51   }
 52 }

 66 void Jfr::on_unloading_classes() {
 67   if (JfrRecorder::is_created()) {
 68     JfrCheckpointManager::write_type_set_for_unloaded_classes();
 69   }
 70 }
 71 
 72 void Jfr::on_thread_start(Thread* t) {
 73   JfrThreadLocal::on_start(t);
 74 }
 75 
 76 void Jfr::on_thread_exit(Thread* t) {
 77   JfrThreadLocal::on_exit(t);
 78 }
 79 
 80 void Jfr::exclude_thread(Thread* t) {
 81   JfrThreadLocal::exclude(t);
 82 }
 83 
 84 void Jfr::include_thread(Thread* t) {
 85   JfrThreadLocal::include(t);
 86 }
 87 
 88 bool Jfr::is_excluded(Thread* t) {
 89   return t != NULL && t->jfr_thread_local()->is_excluded();
 90 }
 91 
 92 void Jfr::on_java_thread_dismantle(JavaThread* jt) {
 93   if (JfrRecorder::is_recording()) {
 94     JfrCheckpointManager::write_thread_checkpoint(jt);
 95   }
 96 }
 97 
 98 void Jfr::on_vm_shutdown(bool exception_handler) {
 99   if (JfrRecorder::is_recording()) {
100     JfrEmergencyDump::on_vm_shutdown(exception_handler);
101   }
102 }
103 
104 void Jfr::weak_oops_do(BoolObjectClosure* is_alive, OopClosure* f) {
105   if (LeakProfiler::is_running()) {
106     LeakProfiler::weak_oops_do(is_alive, f);
107   }
108 }
