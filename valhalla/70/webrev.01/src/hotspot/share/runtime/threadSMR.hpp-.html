 37 // Thread Safe Memory Reclamation (Thread-SMR) support.
 38 //
 39 // ThreadsListHandles are used to safely perform operations on one or more
 40 // threads without the risk of the thread or threads exiting during the
 41 // operation. It is no longer necessary to hold the Threads_lock to safely
 42 // perform an operation on a target thread.
 43 //
 44 // There are several different ways to refer to java.lang.Thread objects
 45 // so we have a few ways to get a protected JavaThread *:
 46 //
 47 // JNI jobject example:
 48 //   jobject jthread = ...;
 49 //   :
 50 //   ThreadsListHandle tlh;
 51 //   JavaThread* jt = NULL;
 52 //   bool is_alive = tlh.cv_internal_thread_to_JavaThread(jthread, &jt, NULL);
 53 //   if (is_alive) {
 54 //     :  // do stuff with 'jt'...
 55 //   }
 56 //
 57 // JVM/TI jthread example:
 58 //   jthread thread = ...;
 59 //   :
 60 //   JavaThread* jt = NULL;
 61 //   ThreadsListHandle tlh;
 62 //   jvmtiError err = JvmtiExport::cv_external_thread_to_JavaThread(tlh.list(), thread, &jt, NULL);
 63 //   if (err != JVMTI_ERROR_NONE) {
 64 //     return err;
 65 //   }
 66 //   :  // do stuff with 'jt'...
 67 //
 68 // JVM/TI oop example (this one should be very rare):
 69 //   oop thread_obj = ...;
 70 //   :
 71 //   JavaThread *jt = NULL;
 72 //   ThreadsListHandle tlh;
 73 //   jvmtiError err = JvmtiExport::cv_oop_to_JavaThread(tlh.list(), thread_obj, &jt);
 74 //   if (err != JVMTI_ERROR_NONE) {
 75 //     return err;
 76 //   }
 77 //   :  // do stuff with 'jt'...
 78 //
 79 // A JavaThread * that is included in the ThreadsList that is held by
 80 // a ThreadsListHandle is protected as long as the ThreadsListHandle
 81 // remains in scope. The target JavaThread * may have logically exited,
 82 // but that target JavaThread * will not be deleted until it is no
 83 // longer protected by a ThreadsListHandle.
 84

 86 // SMR Support for the Threads class.
 87 //
 88 class ThreadsSMRSupport : AllStatic {
 89   friend class VMStructs;
 90   friend class SafeThreadsListPtr;  // for _nested_thread_list_max, delete_notify(), release_stable_list_wake_up() access
 91 
 92   // The coordination between ThreadsSMRSupport::release_stable_list() and
 93   // ThreadsSMRSupport::smr_delete() uses the delete_lock in order to
 94   // reduce the traffic on the Threads_lock.
 95   static Monitor* delete_lock() { return ThreadsSMRDelete_lock; }
 96 
 97   // The '_cnt', '_max' and '_times" fields are enabled via
 98   // -XX:+EnableThreadSMRStatistics (see thread.cpp for a
 99   // description about each field):
100   static uint                  _delete_lock_wait_cnt;
101   static uint                  _delete_lock_wait_max;
102   // The delete_notify flag is used for proper double-check
103   // locking in order to reduce the traffic on the system wide
104   // Thread-SMR delete_lock.
105   static volatile uint         _delete_notify;
106   static volatile uint         _deleted_thread_cnt;
107   static volatile uint         _deleted_thread_time_max;
108   static volatile uint         _deleted_thread_times;
109   static ThreadsList           _bootstrap_list;
110   static ThreadsList* volatile _java_thread_list;
111   static uint64_t              _java_thread_list_alloc_cnt;
112   static uint64_t              _java_thread_list_free_cnt;
113   static uint                  _java_thread_list_max;
114   static uint                  _nested_thread_list_max;
115   static volatile uint         _tlh_cnt;
116   static volatile uint         _tlh_time_max;
117   static volatile uint         _tlh_times;
118   static ThreadsList*          _to_delete_list;
119   static uint                  _to_delete_list_cnt;
120   static uint                  _to_delete_list_max;
121 
122   static ThreadsList *acquire_stable_list_fast_path(Thread *self);
123   static ThreadsList *acquire_stable_list_nested_path(Thread *self);
124   static void add_deleted_thread_times(uint add_value);
125   static void add_tlh_times(uint add_value);
126   static void clear_delete_notify();
127   static bool delete_notify();
128   static void free_list(ThreadsList* threads);
129   static void inc_deleted_thread_cnt();
130   static void inc_java_thread_list_alloc_cnt();
131   static void inc_tlh_cnt();
132   static void release_stable_list_wake_up(bool is_nested);
133   static void set_delete_notify();
134   static void threads_do(ThreadClosure *tc);
135   static void threads_do(ThreadClosure *tc, ThreadsList *list);
136   static void update_deleted_thread_time_max(uint new_value);
137   static void update_java_thread_list_max(uint new_value);
138   static void update_tlh_time_max(uint new_value);
139   static void verify_hazard_ptr_scanned(Thread *self, ThreadsList *threads);
140   static ThreadsList* xchg_java_thread_list(ThreadsList* new_list);
141 
142  public:
143   static void add_thread(JavaThread *thread);
144   static ThreadsList* get_java_thread_list();
145   static bool is_a_protected_JavaThread(JavaThread *thread);
146   static bool is_a_protected_JavaThread_with_lock(JavaThread *thread);
147   static void wait_until_not_protected(JavaThread *thread);
148   static bool is_bootstrap_list(ThreadsList* list);
149   static void remove_thread(JavaThread *thread);
150   static void smr_delete(JavaThread *thread);
151   static void update_tlh_stats(uint millis);
152 
153   // Logging and printing support:
154   static void log_statistics();
155   static void print_info_elements_on(outputStream* st, ThreadsList* t_list);
156   static void print_info_on(outputStream* st);
157   static void print_info_on(const Thread* thread, outputStream* st);
158 };

160 // A fast list of JavaThreads.
161 //
162 class ThreadsList : public CHeapObj<mtThread> {
163   friend class VMStructs;
164   friend class SafeThreadsListPtr;  // for {dec,inc}_nested_handle_cnt() access
165   friend class ThreadsSMRSupport;  // for _nested_handle_cnt, {add,remove}_thread(), {,set_}next_list() access
166 
167   const uint _length;
168   ThreadsList* _next_list;
169   JavaThread *const *const _threads;
170   volatile intx _nested_handle_cnt;
171 
172   template <class T>
173   void threads_do_dispatch(T *cl, JavaThread *const thread) const;
174 
175   ThreadsList *next_list() const        { return _next_list; }
176   void set_next_list(ThreadsList *list) { _next_list = list; }
177 
178   void inc_nested_handle_cnt();
179   void dec_nested_handle_cnt();
180 
181   static ThreadsList* add_thread(ThreadsList* list, JavaThread* java_thread);
182   static ThreadsList* remove_thread(ThreadsList* list, JavaThread* java_thread);
183 
184 public:
185   ThreadsList(int entries);
186   ~ThreadsList();
187 
188   template <class T>
189   void threads_do(T *cl) const;
190 
191   uint length() const                       { return _length; }
192 
193   JavaThread *const thread_at(uint i) const { return _threads[i]; }
194 
195   JavaThread *const *threads() const        { return _threads; }
196 
197   // Returns -1 if target is not found.
198   int find_index_of_JavaThread(JavaThread* target);
199   JavaThread* find_JavaThread_from_java_tid(jlong java_tid) const;
200   bool includes(const JavaThread * const p) const;
201 };

203 // An abstract safe ptr to a ThreadsList comprising either a stable hazard ptr
204 // for leaves, or a retained reference count for nested uses. The user of this
205 // API does not need to know which mechanism is providing the safety.
206 class SafeThreadsListPtr {
207   friend class ThreadsListSetter;
208 
209   SafeThreadsListPtr* _previous;
210   Thread*                 _thread;
211   ThreadsList*            _list;
212   bool                    _has_ref_count;
213   bool                    _needs_release;
214 
215   void acquire_stable_list();
216   void acquire_stable_list_fast_path();
217   void acquire_stable_list_nested_path();
218 
219   void release_stable_list();
220 
221   void verify_hazard_ptr_scanned();
222 
223 public:
224   // Constructor that attaches the list onto a thread.
225   SafeThreadsListPtr(Thread *thread, bool acquire) :
226     _previous(NULL),
227     _thread(thread),
228     _list(NULL),
229     _has_ref_count(false),
230     _needs_release(false)
231   {
232     if (acquire) {
233       acquire_stable_list();
234     }
235   }
236 
237   // Constructor that transfers ownership of the pointer.
238   SafeThreadsListPtr(SafeThreadsListPtr& other) :
239     _previous(other._previous),
240     _thread(other._thread),
241     _list(other._list),
242     _has_ref_count(other._has_ref_count),
243     _needs_release(other._needs_release)
244   {
245     other._needs_release = false;
246   }
247 
248   ~SafeThreadsListPtr() {
249     if (_needs_release) {
250       release_stable_list();
251     }
252   }
253 
254   ThreadsList* list() const { return _list; }
255   SafeThreadsListPtr* previous() const { return _previous; }
256   void print_on(outputStream* st);
257 };

259 // A helper to optionally set the hazard ptr in ourself. This helper can
260 // be used by ourself or by another thread. If the hazard ptr is set(),
261 // then the destructor will release it.
262 //
263 class ThreadsListSetter : public StackObj {
264 private:
265   SafeThreadsListPtr _list_ptr;
266 
267 public:
268   ThreadsListSetter() : _list_ptr(Thread::current(), /* acquire */ false) {}
269   ThreadsList* list() { return _list_ptr.list(); }
270   void set() { _list_ptr.acquire_stable_list(); }
271   bool is_set() { return _list_ptr._needs_release; }
272 };

274 // This stack allocated ThreadsListHandle keeps all JavaThreads in the
275 // ThreadsList from being deleted until it is safe.
276 //
277 class ThreadsListHandle : public StackObj {
278   SafeThreadsListPtr _list_ptr;
279   elapsedTimer _timer;  // Enabled via -XX:+EnableThreadSMRStatistics.
280 
281 public:
282   ThreadsListHandle(Thread *self = Thread::current());
283   ~ThreadsListHandle();
284 
285   ThreadsList *list() const {
286     return _list_ptr.list();
287   }
288 
289   template <class T>
290   void threads_do(T *cl) const {
291     return list()->threads_do(cl);
292   }
293 
294   bool cv_internal_thread_to_JavaThread(jobject jthread, JavaThread ** jt_pp, oop * thread_oop_p);
295 
296   bool includes(JavaThread* p) {
297     return list()->includes(p);
298   }
299 
300   uint length() const {
301     return list()->length();
302   }
303 
304   JavaThread *thread_at(uint i) const {
305     return list()->thread_at(i);
306   }
307 };

309 // This stack allocated JavaThreadIterator is used to walk the
310 // specified ThreadsList using the following style:
311 //
312 //   JavaThreadIterator jti(t_list);
313 //   for (JavaThread *jt = jti.first(); jt != NULL; jt = jti.next()) {
314 //     ...
315 //   }
316 //
317 class JavaThreadIterator : public StackObj {
318   ThreadsList * _list;
319   uint _index;
320 
321 public:
322   JavaThreadIterator(ThreadsList *list) : _list(list), _index(0) {
323     assert(list != NULL, "ThreadsList must not be NULL.");
324   }
325 
326   JavaThread *first() {
327     _index = 0;
328     return _list->thread_at(_index);
329   }
330 
331   uint length() const {
332     return _list->length();
333   }
334 
335   ThreadsList *list() const {
336     return _list;
337   }
338 
339   JavaThread *next() {
340     if (++_index >= length()) {
341       return NULL;
342     }
343     return _list->thread_at(_index);
344   }
345 };

347 // This stack allocated ThreadsListHandle and JavaThreadIterator combo
348 // is used to walk the ThreadsList in the included ThreadsListHandle
349 // using the following style:
350 //
351 //   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *jt = jtiwh.next(); ) {
352 //     ...
353 //   }
354 //
355 class JavaThreadIteratorWithHandle : public StackObj {
356   ThreadsListHandle _tlh;
357   uint _index;
358 
359 public:
360   JavaThreadIteratorWithHandle() : _index(0) {}
361 
362   uint length() const {
363     return _tlh.length();
364   }
365 
366   ThreadsList *list() const {
367     return _tlh.list();
368   }
369 
370   JavaThread *next() {
371     if (_index >= length()) {
372       return NULL;
373     }
374     return _tlh.list()->thread_at(_index++);
375   }
376 
377   void rewind() {
378     _index = 0;
379   }
380 };
