260 template <class T>
261 inline bool G1CMTask::deal_with_reference(T* p) {
262   increment_refs_reached();
263   oop const obj = RawAccess<MO_VOLATILE>::oop_load(p);

265     return false;
266   }
267   return make_reference_grey(obj);
