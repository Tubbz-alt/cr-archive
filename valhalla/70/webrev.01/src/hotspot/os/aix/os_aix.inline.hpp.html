 40 inline bool os::uses_stack_guard_pages() {
 41   return true;
 42 }
 43 
 44 // Whether or not calling code should/can commit/uncommit stack pages
 45 // before guarding them. Answer for AIX is definitly no, because memory
 46 // is automatically committed on touch.
 47 inline bool os::must_commit_stack_guard_pages() {
 48   assert(uses_stack_guard_pages(), "sanity check");
 49   return false;
 50 }

 72 // macros for restartable system calls
 73 
 74 #define RESTARTABLE(_cmd, _result) do { \
 75     _result = _cmd; \
 76   } while(((int)_result == OS_ERR) && (errno == EINTR))
 77 
 78 #define RESTARTABLE_RETURN_INT(_cmd) do { \
 79   int _result; \
 80   RESTARTABLE(_cmd, _result); \
 81   return _result; \
 82 } while(false)

 84 // We don't have NUMA support on Aix, but we need this for compilation.
 85 inline bool os::numa_has_static_binding()   { ShouldNotReachHere(); return true; }
 86 inline bool os::numa_has_group_homing()     { ShouldNotReachHere(); return false;  }

 88 inline size_t os::write(int fd, const void *buf, unsigned int nBytes) {
 89   size_t res;
 90   RESTARTABLE((size_t) ::write(fd, buf, (size_t) nBytes), res);
 91   return res;
 92 }
