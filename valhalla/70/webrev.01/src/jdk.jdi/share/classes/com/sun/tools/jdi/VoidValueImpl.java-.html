26 package com.sun.tools.jdi;
27 
28 import com.sun.jdi.InvalidTypeException;
29 import com.sun.jdi.Type;
30 import com.sun.jdi.VirtualMachine;
31 import com.sun.jdi.VoidValue;
32 
33 public class VoidValueImpl extends ValueImpl implements VoidValue {
34 
35     VoidValueImpl(VirtualMachine aVm) {
36         super(aVm);
37     }
38 
39     public boolean equals(Object obj) {
40         return (obj != null) && (obj instanceof VoidValue) && super.equals(obj);
41     }
42 
43     public int hashCode() {
44         /*
45          * TO DO: Better hash code
46          */
47         return 47245;
48     }
49 
50     public Type type() {
51         return vm.theVoidType();
52     }
53 
54     ValueImpl prepareForAssignmentTo(ValueContainer destination)
55         throws InvalidTypeException
56     {
57         if ("void".equals(destination.typeName())) {
58             return this;
59         }
60         throw new InvalidTypeException();
61     }
62 
63     public String toString() {
64         return "<void value>";
65     }
66 
67     byte typeValueKey() {
68         return JDWP.Tag.VOID;
69     }
70 }
