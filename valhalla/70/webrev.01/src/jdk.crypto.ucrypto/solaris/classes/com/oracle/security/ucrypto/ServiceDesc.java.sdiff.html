30 /**
31  * Class for encapsulating the type, algorithm, and class name of
32  * a Provider.Service object.
33  */
34 final class ServiceDesc {
35 
36     private final String type;
37     private final String algo;
38     private final String cn;
39     private final List<String> aliases;
40 
41     ServiceDesc(String type, String algo, String cn) {
42         this(type, algo, cn, null);
43     }
44 
45     ServiceDesc(String type, String algo, String cn, String[] aliases) {
46         this.type = type;
47         this.algo = algo;
48         this.cn = cn;
49         if (aliases != null) {
50             this.aliases = Arrays.asList(aliases);
51         } else {
52             this.aliases = null;
53         }
54     }
55     String getType() {
56         return type;
57     }
58     String getAlgorithm() {
59         return algo;
60     }
61     String getClassName() {
62         return cn;
63     }
64     List<String> getAliases() {
65         return aliases;
66     }
67     public String toString() {
68         return type + "." + algo + ": " + cn + ", aliases =" + aliases;
69     }
70 }
