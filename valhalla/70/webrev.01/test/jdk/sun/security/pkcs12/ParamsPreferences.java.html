179     private static ObjectIdentifier oid(KnownOIDs o) {
180         return ObjectIdentifier.of(o);
181     }
182 
183     /**
184      * Run once.
185      *
186      * @param args an array containing system properties and values, "-",
187      *             security properties and values, "-", expected certPbeAlg,
188      *             certPbeIC, keyPbeAlg, keyPbeIc, macAlg, macIC.
189      */
190     static void test(int n, Object... args) throws Exception {
191         boolean isSysProp = true;
192         String cmd = "-keystore ks" + n + " -genkeypair -keyalg EC "
193                 + "-alias a -dname CN=A -storepass changeit "
194                 + "-J-Djava.security.properties=" + n + ".conf";
195         List<String> jsConf = new ArrayList<>();
196         for (int i = 0; i < args.length; i++) {
197             if (isSysProp) {
198                 if (args[i].equals("-")) {
199                     isSysProp = false;
200                 } else {
201                     cmd += " -J-D" + args[i] + "=" + args[++i];
202                 }
203             } else {
204                 if (args[i] == "-") {
205                     Files.write(Path.of(n + ".conf"), jsConf);
206                     System.out.println("--------- test starts ----------");
207                     System.out.println(jsConf);
208                     SecurityTools.keytool(cmd).shouldHaveExitValue(0);
209 
210                     byte[] data = Files.readAllBytes(Path.of("ks" + n));
211 
212                     // cert pbe alg + ic
213                     if (args[i+1] == null) {
214                         checkAlg(data, "110c10", DATA_OID);
215                     } else {
216                         checkAlg(data, "110c10", ENCRYPTED_DATA_OID);
217                         checkAlg(data, "110c110110", (ObjectIdentifier)args[i+1]);
218                         checkInt(data, "110c1101111", (int)args[i+2]);
219                     }
220 
221                     // key pbe alg + ic
222                     checkAlg(data, "110c010c01000", (ObjectIdentifier)args[i+3]);
223                     checkInt(data, "110c010c010011", (int)args[i+4]);
224 
225                     // mac alg + ic
226                     if (args[i+5] == null) {
227                         shouldNotExist(data, "2");
228                     } else {
229                         checkAlg(data, "2000", (ObjectIdentifier)args[i+5]);
230                         checkInt(data, "22", (int)args[i+6]);
231                     }
232                 } else {
233                     jsConf.add(args[i] + "=" + args[++i]);
234                 }
235             }
236         }
237     }
