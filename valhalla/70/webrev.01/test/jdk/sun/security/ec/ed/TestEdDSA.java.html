324     private static void runSignTest(String algorithm,
325             AlgorithmParameterSpec params, String privateKey, String publicKey,
326             String message, String signature) throws Exception {
327 
328         byte[] privKeyBytes = Convert.hexStringToByteArray(privateKey);
329         EdECPoint pubKeyPoint = Convert.hexStringToEdPoint(publicKey);
330         byte[] msgBytes = Convert.hexStringToByteArray(message);
331         byte[] computedSig;
332 
333         NamedParameterSpec namedSpec = new NamedParameterSpec(algorithm);
334         EdECPrivateKeySpec privKeySpec =
335             new EdECPrivateKeySpec(namedSpec, privKeyBytes);
336         KeyFactory kf = KeyFactory.getInstance(algorithm);
337         PrivateKey privKey = kf.generatePrivate(privKeySpec);
338         Signature sig = Signature.getInstance(algorithm);
339         if (params != null) {
340             sig.setParameter(params);
341         }
342         sig.initSign(privKey);
343         sig.update(msgBytes);
344         computedSig = sig.sign();
345         if (!Arrays.equals(computedSig,
346             Convert.hexStringToByteArray(signature))) {
347             throw new RuntimeException("Incorrect signature");
348         }
349 
350         // test verification
351         sig = Signature.getInstance(algorithm);
352         if (params != null) {
353             sig.setParameter(params);
354         }
355         EdECPublicKeySpec pubKeySpec =
356             new EdECPublicKeySpec(namedSpec, pubKeyPoint);
357         PublicKey pubKey = kf.generatePublic(pubKeySpec);
358         sig.initVerify(pubKey);
359         sig.update(msgBytes);
360         if (!sig.verify(computedSig)) {
361             throw new RuntimeException("Signature did not verify");
362         }
363     }

382     private static void runBasicTest(String name, Object param)
383         throws Exception {
384 
385         KeyPairGenerator kpg = KeyPairGenerator.getInstance(name);
386         if (param instanceof Integer) {
387             kpg.initialize((Integer) param, random);
388         } else if (param instanceof String) {
389             kpg.initialize(new NamedParameterSpec((String) param), random);
390         }
391         KeyPair kp = kpg.generateKeyPair();
392 
393         Signature sig = Signature.getInstance(name);
394         sig.initSign(kp.getPrivate());
395         byte[] testMessage = new byte[1024];
396         random.nextBytes(testMessage);
397         sig.update(testMessage);
398         byte[] msgSig = sig.sign();
399 
400         // sign again, should return false
401         byte[] x = sig.sign();
402         if (Arrays.compare(msgSig, x) == 0) {
403             throw new RuntimeException("Second sign identical, didn't reset");
404         }
405 
406         // verify the signature
407         sig.initVerify(kp.getPublic());
408         sig.update(testMessage);
409         if (!sig.verify(msgSig)) {
410             throw new RuntimeException("Valid signature did not verify");
411         }
412 
413         // verify again, should return false
414         if (sig.verify(msgSig)) {
415             throw new RuntimeException("Second verify succeeded, didn't reset");
416         }
417 
418         // try verifying an incorrect signature
419         testMessage[0] ^= (byte) 0x01;
420         sig.update(testMessage);
421         if (sig.verify(msgSig)) {
422             throw new RuntimeException("Invalid signature verified");
423         }
424 
425         KeyFactory kf = KeyFactory.getInstance(name);
426         // Test with X509 and PKCS8 key specs
427         X509EncodedKeySpec pubSpec =
428             kf.getKeySpec(kp.getPublic(), X509EncodedKeySpec.class);
429         PKCS8EncodedKeySpec priSpec =
430             kf.getKeySpec(kp.getPrivate(), PKCS8EncodedKeySpec.class);
431 
432         PublicKey pubKey = kf.generatePublic(pubSpec);
433         PrivateKey priKey = kf.generatePrivate(priSpec);
434 
435         sig.initSign(priKey);
436         sig.update(testMessage);
437         msgSig = sig.sign();
438         sig.initVerify(pubKey);
439         sig.update(testMessage);
440         if (!sig.verify(msgSig)) {
441             throw new RuntimeException("Valid signature did not verify");
442         }
443 
444         // test with EdEC key specs
445         EdECPublicKeySpec edPublic =
446             kf.getKeySpec(kp.getPublic(), EdECPublicKeySpec.class);
447         EdECPrivateKeySpec edPrivate =
448             kf.getKeySpec(kp.getPrivate(), EdECPrivateKeySpec.class);
449         PublicKey pubKey2 = kf.generatePublic(edPublic);
450         PrivateKey priKey2 = kf.generatePrivate(edPrivate);
451         sig.initSign(priKey2);
452         sig.update(testMessage);
453         msgSig = sig.sign();
454         sig.initVerify(pubKey2);
455         sig.update(testMessage);
456         if (!sig.verify(msgSig)) {
457             throw new RuntimeException("Valid signature did not verify");
458         }
459     }

472     private static void runCurveMixTest(String providerName, String name,
473                                         Object param) throws Exception {
474 
475         KeyPairGenerator kpg = KeyPairGenerator.getInstance(name,
476             providerName);
477 
478         try {
479             if (param instanceof Integer) {
480                 kpg.initialize((Integer) param);
481             } else if (param instanceof String) {
482                 kpg.initialize(new NamedParameterSpec((String) param));
483             }
484             throw new RuntimeException(name + " KeyPairGenerator accepted "
485                     + param.toString() + " parameters");
486         } catch (InvalidParameterException ex) {
487             if (param instanceof String) {
488                 throw new RuntimeException(
489                         "InvalidAlgorithmParameterException expected", ex);
490             }
491             // expected
492 
493         } catch (InvalidAlgorithmParameterException ex) {
494             if (param instanceof Integer) {
495                 throw new RuntimeException("InvalidParameterException expected",
496                         ex);
497             }
498             // expected
499         }
500 
501         // the rest of the test uses the parameter as an algorithm name to
502         // produce keys
503         if (param instanceof Integer) {
504             return;
505         }
506         String otherName = (String) param;
507         KeyPairGenerator otherKpg = KeyPairGenerator.getInstance(otherName,
508             providerName);
509         KeyPair otherKp = otherKpg.generateKeyPair();
510 
511         // ensure the KeyFactory rejects incorrect keys
512         KeyFactory kf = KeyFactory.getInstance(name, providerName);
513         try {
514             kf.getKeySpec(otherKp.getPublic(), EdECPublicKeySpec.class);
515             throw new RuntimeException(name + " KeyFactory accepted "
516                 + param.toString() + " key");
517         } catch (InvalidKeySpecException ex) {
518             // expected
519         }
520         try {
521             kf.getKeySpec(otherKp.getPrivate(), EdECPrivateKeySpec.class);
522             throw new RuntimeException(name + " KeyFactory accepted "
523                 + param.toString() + " key");
524         } catch (InvalidKeySpecException ex) {
525             // expected
526         }
527 
528         try {
529             kf.translateKey(otherKp.getPublic());
530             throw new RuntimeException(name + " KeyFactory accepted "
531                 + param.toString() + " key");
532         } catch (InvalidKeyException ex) {
533             // expected
534         }
535         try {
536             kf.translateKey(otherKp.getPrivate());
537             throw new RuntimeException(name + " KeyFactory accepted "
538                 + param.toString() + " key");
539         } catch (InvalidKeyException ex) {
540             // expected
541         }
542 
543         KeyFactory otherKf = KeyFactory.getInstance(otherName, providerName);
544         EdECPublicKeySpec otherPubSpec = otherKf.getKeySpec(otherKp.getPublic(),
545             EdECPublicKeySpec.class);
546         try {
547             kf.generatePublic(otherPubSpec);
548             throw new RuntimeException(name + " KeyFactory accepted "
549                 + param.toString() + " key");
550         } catch (InvalidKeySpecException ex) {
551             // expected
552         }
553         EdECPrivateKeySpec otherPriSpec =
554             otherKf.getKeySpec(otherKp.getPrivate(), EdECPrivateKeySpec.class);
555         try {
556             kf.generatePrivate(otherPriSpec);
557             throw new RuntimeException(name + " KeyFactory accepted "
558                 + param.toString() + " key");
559         } catch (InvalidKeySpecException ex) {
560             // expected
561         }
562 
563         // ensure the Signature rejects incorrect keys
564         Signature sig = Signature.getInstance(name, providerName);
565         try {
566             sig.initSign(otherKp.getPrivate());
567             throw new RuntimeException(name + " Signature accepted "
568                 + param.toString() + " signing key");
569         } catch (InvalidKeyException ex) {
570             // expected
571         }
572 
573         try {
574             sig.initVerify(otherKp.getPublic());
575             throw new RuntimeException(name + " Signature accepted "
576                 + param.toString() + " verification key");
577         } catch (InvalidKeyException ex) {
578             // expected
579         }
