 88     private void runDiffieHellmanTest(String opName, String a_str,
 89         String b_str, String result_str) {
 90 
 91         NamedParameterSpec paramSpec = new NamedParameterSpec(opName);
 92         XECParameters settings =
 93             XECParameters.get(RuntimeException::new, paramSpec);
 94         XECOperations ops = new XECOperations(settings);
 95 
 96         byte[] basePoint = Convert.byteToByteArray(settings.getBasePoint(),
 97             settings.getBytes());
 98         byte[] a = Convert.hexStringToByteArray(a_str);
 99         byte[] b = Convert.hexStringToByteArray(b_str);
100         byte[] expectedResult = Convert.hexStringToByteArray(result_str);
101 
102         byte[] a_copy = Arrays.copyOf(a, a.length);
103         byte[] b_copy = Arrays.copyOf(b, b.length);
104         byte[] basePoint_copy = Arrays.copyOf(basePoint, basePoint.length);
105 
106         byte[] resultA = ops.encodedPointMultiply(b,
107             ops.encodedPointMultiply(a, basePoint));
108         byte[] resultB = ops.encodedPointMultiply(a_copy,
109             ops.encodedPointMultiply(b_copy, basePoint_copy));
110         if (!Arrays.equals(resultA, expectedResult)) {
111             throw new RuntimeException("fail");
112         }
113         if (!Arrays.equals(resultB, expectedResult)) {
114             throw new RuntimeException("fail");
115         }
116     }
117 
118     private void runTest(String opName, String k_in_str,
119         String u_in_str, String u_out_str) {
120 
121         byte[] k_in = Convert.hexStringToByteArray(k_in_str);
122         byte[] u_in = Convert.hexStringToByteArray(u_in_str);
123         byte[] u_out_expected = Convert.hexStringToByteArray(u_out_str);
124 
125         NamedParameterSpec paramSpec = new NamedParameterSpec(opName);
126         XECParameters settings =
127             XECParameters.get(RuntimeException::new, paramSpec);
128         XECOperations ops = new XECOperations(settings);
129         byte[] u_out = ops.encodedPointMultiply(k_in, u_in);
130 
131         if (!Arrays.equals(u_out, u_out_expected)) {
132             throw new RuntimeException("fail");
133         }
134     }
