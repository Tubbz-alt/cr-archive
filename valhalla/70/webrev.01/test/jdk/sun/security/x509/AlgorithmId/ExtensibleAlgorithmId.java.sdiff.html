24 /*
25  * @test
26  * @bug 4162868 8130181 8242151
27  * @modules java.base/sun.security.x509
28  * @modules java.base/sun.security.util
29  * @run main/othervm ExtensibleAlgorithmId
30  * @summary Algorithm Name-to-OID mapping needs to be made extensible.
31  */
32 
33 // Run in othervm, coz AlgorithmId.oidTable is only initialized once
34 
35 import java.security.*;
36 import sun.security.x509.AlgorithmId;
37 
38 public class ExtensibleAlgorithmId {
39 
40     public static void main(String[] args) throws Exception {
41         TestProvider p = new TestProvider();
42         Security.addProvider(p);
43         AlgorithmId algid = AlgorithmId.getAlgorithmId(TestProvider.ALG_NAME);
44         String oid = algid.getOID().toString();
45         if (!oid.equals(TestProvider.ALG_OID)) {
46             throw new Exception("Provider alias oid not used, found " + oid);
47         }
48         String name = algid.getName();
49         if (!name.equalsIgnoreCase(TestProvider.ALG_NAME)) {
50             throw new Exception("provider alias name not used, found " + name);
51         }
52         String alias = "Alg.Alias.Signature.OID." + oid;
53         String stdAlgName = p.getProperty(alias);
54         if (stdAlgName == null ||
55                 !stdAlgName.equalsIgnoreCase(TestProvider.ALG_NAME)) {
56             throw new Exception("Wrong OID");
57         }
58     }

59 
60     static class TestProvider extends Provider {
61 
62         static String ALG_OID = "1.2.3.4.5.6.7.8.9.0";
63         static String ALG_NAME = "XYZ";
64 
65         public TestProvider() {
66         super("Dummy", "1.0", "XYZ algorithm");
67 
68             AccessController.doPrivileged(new PrivilegedAction() {
69                 public Object run() {
70 
71                     put("Signature." + ALG_NAME, "test.xyz");
72                     // preferred OID
73                     put("Alg.Alias.Signature.OID." + ALG_OID,
74                         ALG_NAME);
75                     put("Alg.Alias.Signature.9.8.7.6.5.4.3.2.1.0",
76                         ALG_NAME);
77                     return null;
78                 }
79             });
80         }
81     }
82 }
