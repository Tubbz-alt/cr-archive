24 import javax.xml.parsers.DocumentBuilder;
25 import javax.xml.parsers.DocumentBuilderFactory;
26 import org.w3c.dom.DOMImplementation;
27 import org.w3c.dom.Document;
28 import org.w3c.dom.ls.DOMImplementationLS;
29 import org.w3c.dom.ls.LSSerializer;
30 
31 class AbstractMethodErrorTest {
32 
33     public static void main(String[] args) throws Exception {
34         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
35         DocumentBuilder builder = factory.newDocumentBuilder();
36         Document document = builder.newDocument();
37 
38         DOMImplementation impl = document.getImplementation();
39         DOMImplementationLS implLS = (DOMImplementationLS) impl.getFeature("LS", "3.0");
40         LSSerializer dsi = implLS.createLSSerializer();
41 
42         /* We should have here incorrect document without getXmlVersion() method:
43          * Such Document is generated by replacing the JDK bootclasses with it's
44          * own Node,Document and DocumentImpl classes (see run.sh). According to
45          * XERCESJ-1007 the AbstractMethodError should be thrown in such case.
46          */
47         String result = dsi.writeToString(document);
48         System.out.println("Result:" + result);
49     }
50 
51 }
