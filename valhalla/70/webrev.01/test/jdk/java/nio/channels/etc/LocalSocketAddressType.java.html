 33 import jdk.test.lib.NetworkConfiguration;
 34 import jdk.test.lib.net.IPSupport;
 35 import org.testng.annotations.BeforeTest;
 36 import org.testng.annotations.DataProvider;
 37 import org.testng.annotations.Test;
 38 
 39 import java.net.*;
 40 import java.nio.channels.DatagramChannel;
 41 import java.nio.channels.ServerSocketChannel;
 42 import java.nio.channels.SocketChannel;
 43 import java.util.Iterator;
 44 import java.util.stream.Collectors;
 45 import java.util.stream.Stream;
 46 
 47 import static java.lang.Boolean.parseBoolean;
 48 import static java.lang.System.getProperty;
 49 import static java.lang.System.out;
 50 import static jdk.test.lib.Asserts.assertEquals;
 51 import static jdk.test.lib.Asserts.assertTrue;
 52 import static jdk.test.lib.net.IPSupport.*;
 53 
 54 public class LocalSocketAddressType {
 55 
 56     @BeforeTest()
 57     public void setup() {
 58         IPSupport.printPlatformSupport(out);
 59         throwSkippedExceptionIfNonOperational();
 60     }
 61 
 62     @DataProvider(name = "addresses")
 63     public static Iterator<Object[]> addresses() throws Exception {
 64         NetworkConfiguration nc = NetworkConfiguration.probe();
 65         return Stream.concat(nc.ip4Addresses(), nc.ip6Addresses())
 66                 .map(ia -> new Object[] { new InetSocketAddress(ia, 0) })
 67                 .iterator();
 68     }
 69 
 70     @Test(dataProvider = "addresses")
 71     public static void testSocketChannel(InetSocketAddress addr) throws Exception {
 72         try (var c = SocketChannel.open()) {
 73             Class<? extends InetAddress> cls = addr.getAddress().getClass();
 74             InetAddress ia = ((InetSocketAddress)c.bind(addr).getLocalAddress()).getAddress();
 75             assertEquals(ia.getClass(), cls);
 76             ia = c.socket().getLocalAddress();
 77             assertEquals(ia.getClass(), cls);
 78         }
 79     }
 80 
 81     @Test(dataProvider = "addresses")
 82     public static void testServerSocketChannel(InetSocketAddress addr) throws Exception {
 83         try (var c = ServerSocketChannel.open()) {
 84             Class<? extends InetAddress> cls = addr.getAddress().getClass();
 85             InetAddress ia = ((InetSocketAddress)c.bind(addr).getLocalAddress()).getAddress();
 86             assertEquals(ia.getClass(), cls);
 87             ia = c.socket().getInetAddress();
 88             assertEquals(ia.getClass(), cls);
 89         }
 90     }
 91 
 92     @Test(dataProvider = "addresses")
 93     public static void testDatagramChannel(InetSocketAddress addr) throws Exception {
 94         try (var c = DatagramChannel.open()) {
 95             Class<? extends InetAddress> cls = addr.getAddress().getClass();
 96             InetAddress ia = ((InetSocketAddress)c.bind(addr).getLocalAddress()).getAddress();
 97             assertEquals(ia.getClass(), cls);
 98             ia = c.socket().getLocalAddress();
 99             assertEquals(ia.getClass(), cls);
100         }
101     }
102 }
