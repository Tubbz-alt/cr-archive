29 import java.net.*;
30 
31 public class SetTTLAndGetTTL {
32 
33     public static void main(String args[]) throws Exception {
34         MulticastSocket soc = null;
35 
36         try {
37             soc = new MulticastSocket();
38         } catch(Exception e) {
39             throw new Exception("Unexpected Exception");
40         }
41 
42         soc.setTTL((byte)200);
43         byte ttlValue = soc.getTTL();
44         if (ttlValue != (byte)200)
45             throw new Exception("setTTL/getTTL is broken");
46     }
47 }
