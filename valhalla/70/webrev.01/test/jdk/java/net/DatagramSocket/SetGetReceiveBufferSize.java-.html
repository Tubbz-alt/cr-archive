24 /*
25  * @test
26  * @bug 4173717 8243488
27  * @summary Check that setReceiveBufferSize and getReceiveBufferSize work as expected
28  * @run testng SetGetReceiveBufferSize
29  * @run testng/othervm -Djava.net.preferIPv4Stack=true SetGetReceiveBufferSize
30  */
31 
32 import org.testng.annotations.DataProvider;
33 import org.testng.annotations.Test;
34 
35 import java.io.IOException;
36 import java.net.DatagramSocket;
37 import java.net.MulticastSocket;
38 import java.net.SocketException;
39 import java.nio.channels.DatagramChannel;
40 
41 import static org.testng.Assert.assertEquals;
42 import static org.testng.Assert.expectThrows;
43 
44 public class SetGetReceiveBufferSize {
45     static final Class<SocketException> SE = SocketException.class;
46     static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
47 
48     @FunctionalInterface
49     public interface DatagramSocketSupplier {
50         DatagramSocket open() throws IOException;
51     }
52     static DatagramSocketSupplier supplier(DatagramSocketSupplier supplier) { return supplier; }
53 
54     @DataProvider
55     public Object[][] invariants() {
56         return new Object[][]{
57                 {"DatagramSocket", supplier(() -> new DatagramSocket())},
58                 {"MulticastSocket", supplier(() -> new MulticastSocket())},
59                 {"DatagramSocketAdaptor", supplier(() -> DatagramChannel.open().socket())},
60         };
61     }
62 
63     @Test(dataProvider = "invariants")
64     public void testSetInvalidBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
65         var invalidArgs = new int[]{ -1, 0 };
66 
67         try (var socket = supplier.open()) {
68             for (int i : invalidArgs) {
69                 Exception ex = expectThrows(IAE, () -> socket.setReceiveBufferSize(i));
70                 System.out.println(name + " got expected exception: " + ex);
71             }
72         }
73     }
74 
75     @Test(dataProvider = "invariants")
76     public void testSetAndGetBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
77         var validArgs = new int[]{ 1234, 2345, 3456 };
78 
79         try (var socket = supplier.open()) {
80             for (int i : validArgs) {
81                 socket.setReceiveBufferSize(i);
82                 assertEquals(socket.getReceiveBufferSize(), i, name);
83             }
84         }
85     }
86 
87     @Test(dataProvider = "invariants")
88     public void testSetGetAfterClose(String name, DatagramSocketSupplier supplier) throws IOException {
89         var socket = supplier.open();
90         socket.close();
91 
92         Exception setException = expectThrows(SE, () -> socket.setReceiveBufferSize(2345));
93         System.out.println(name + " got expected exception: " + setException);
94 
95         Exception getException = expectThrows(SE, () -> socket.getReceiveBufferSize());
96         System.out.println(name + " got expected exception: " + getException);
97     }
98 }
