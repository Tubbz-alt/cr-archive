 24 /*
 25  * @test
 26  * @bug 6505016
 27  * @summary Socket spec should clarify what getInetAddress/getPort/etc return after the Socket is closed
 28  * @run main TestAfterClose
 29  * @run main/othervm -Djdk.net.usePlainDatagramSocketImpl TestAfterClose
 30  */
 31 
 32 import java.net.*;
 33 import java.io.*;
 34 
 35 public class TestAfterClose
 36 {
 37     static int failCount;
 38 
 39     public static void main(String[] args) {
 40         try {
 41             DatagramSocket socket = new DatagramSocket();
 42             socket.connect(InetAddress.getLoopbackAddress(), 5001);
 43             test(socket, true);
 44 
 45             socket = new DatagramSocket();
 46             test(socket, false);
 47 
 48         } catch (IOException ioe) {
 49             ioe.printStackTrace();
 50         }
 51 
 52         if (failCount > 0)
 53             throw new RuntimeException("Failed: failcount = " + failCount);
 54 
 55     }
 56 
 57     static void test(DatagramSocket socket, boolean connected) throws IOException {
 58         //Before Close
 59         int socketPort = socket.getPort();
 60         InetAddress socketInetAddress = socket.getInetAddress();
 61         SocketAddress socketRemoteSocketAddress = socket.getRemoteSocketAddress();
 62 
 63         //After Close
 64         socket.close();
 65 
 66         if (connected ? !(socket.getPort() == socketPort)
 67                       : !(socket.getPort() == -1)) {
 68             System.out.println("Socket.getPort failed");
 69             failCount++;
 70         }
 71 
 72         if (connected ? !socket.getInetAddress().equals(socketInetAddress)
 73                       : !(socket.getInetAddress() == null)) {
 74             System.out.println("Socket.getInetAddress failed");
 75             failCount++;
 76         }
 77 
 78         if (connected ? !socket.getRemoteSocketAddress().equals(socketRemoteSocketAddress)
 79                       : !(socket.getRemoteSocketAddress() == null)) {
 80             System.out.println("Socket.getRemoteSocketAddresss failed");
 81             failCount++;
 82         }
 83 
 84         if (socket.getLocalPort() != -1) {
 85             System.out.println("Socket.getLocalPort failed");
 86             failCount++;
 87         }
 88 
 89         if (socket.getLocalAddress() != null) {
 90             System.out.println("Socket.getLocalAddress failed");
 91             failCount++;
 92         }
 93 
 94         if (socket.getLocalSocketAddress() != null) {
 95             System.out.println("Socket.getLocalSocketAddress failed");
 96             failCount++;
 97         }
 98 
 99         if (connected && !socket.isConnected()) {
100             System.out.println("Socket.isConnected failed");
101             failCount++;
102         }
103 
104         if (!socket.isBound()) {
105             System.out.println("Socket.isBound failed");
106             failCount++;
107         }
108 
109     }
110 }
