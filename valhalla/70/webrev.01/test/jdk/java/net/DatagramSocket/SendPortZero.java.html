 46 /*
 47  * @test
 48  * @bug 8236105 8240533
 49  * @summary Check that DatagramSocket throws expected
 50  *          Exception when sending a DatagramPacket with port 0
 51  * @run testng SendPortZero
 52  * @run testng/othervm -Djdk.net.usePlainDatagramSocketImpl SendPortZero

 55 public class SendPortZero {
 56     private InetAddress loopbackAddr, wildcardAddr;
 57     private DatagramSocket datagramSocket, datagramSocketAdaptor;
 58     private DatagramPacket loopbackZeroPkt, wildcardZeroPkt, wildcardValidPkt;
 59 
 60     private static final Class<SocketException> SE = SocketException.class;
 61     private static final Class<AccessControlException> ACE =
 62             AccessControlException.class;
 63 
 64     @BeforeTest
 65     public void setUp() throws IOException {
 66         datagramSocket = new DatagramSocket();
 67         datagramSocketAdaptor = DatagramChannel.open().socket();
 68 
 69         byte[] buf = "test".getBytes();
 70 
 71         // Addresses
 72         loopbackAddr = InetAddress.getLoopbackAddress();
 73         //wildcardAddr = new InetSocketAddress(0).getAddress();
 74 
 75         // Packets
 76         // loopback w/port 0
 77         loopbackZeroPkt = new DatagramPacket(buf, 0, buf.length);
 78         loopbackZeroPkt.setAddress(loopbackAddr);
 79         loopbackZeroPkt.setPort(0);
 80 
 81         /*
 82         //Commented until JDK-8236852 is fixed
 83 
 84         // wildcard w/port 0
 85         wildcardZeroPkt = new DatagramPacket(buf, 0, buf.length);
 86         wildcardZeroPkt.setAddress(wildcardAddr);
 87         wildcardZeroPkt.setPort(0);
 88 
 89         //Commented until JDK-8236807 is fixed
 90 
 91         // wildcard addr w/valid port
 92         wildcardValidPkt = new DatagramPacket(buf, 0, buf.length);
 93         var addr = socket.getAddress();
 94         wildcardValidPkt.setAddress(addr);
 95         wildcardValidPkt.setPort(socket.getLocalPort());
 96       */
 97     }
 98 
 99     @DataProvider(name = "data")
100     public Object[][] variants() {
101         return new Object[][]{
102                 { datagramSocket,        loopbackZeroPkt },
103                 { datagramSocketAdaptor, loopbackZeroPkt },
104         };
105     }
106 
107     @Test(dataProvider = "data")
108     public void testSend(DatagramSocket ds, DatagramPacket pkt) {
109         assertThrows(SE, () -> ds.send(pkt));
110     }
111 
112     // Check that 0 port check doesn't override security manager check
113     @Test(dataProvider = "data")
114     public void testSendWithSecurityManager(DatagramSocket ds,
115                                             DatagramPacket pkt) {
116         Policy defaultPolicy = Policy.getPolicy();
117         try {
118             Policy.setPolicy(new NoSendPolicy());
119             System.setSecurityManager(new SecurityManager());
120 
121             assertThrows(ACE, () -> ds.send(pkt));
122         } finally {
123             System.setSecurityManager(null);
124             Policy.setPolicy(defaultPolicy);
125         }
126     }
127 
128     static class NoSendPolicy extends Policy {
129         final PermissionCollection perms = new Permissions();
130         { perms.add(
131                 new SocketPermission("*:0", "connect")); }
132 
133         public boolean implies(ProtectionDomain domain, Permission perm) {
134             return !perms.implies(perm);
135         }
136     }
137 
138     @AfterTest
139     public void tearDown() {
140         datagramSocket.close();
141         datagramSocketAdaptor.close();
142     }
143 }
