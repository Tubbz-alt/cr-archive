 24 /*
 25  * @test
 26  * @bug 8243488
 27  * @library /test/lib
 28  * @build jdk.test.lib.Platform
 29  * @summary Check that setSendBufferSize and getSendBufferSize work as expected
 30  * @run testng SetGetSendBufferSize
 31  * @run testng/othervm -Djava.net.preferIPv4Stack=true SetGetSendBufferSize
 32  * @run testng/othervm -Djdk.net.usePlainDatagramSocketImpl SetGetSendBufferSize
 33  */
 34 
 35 import org.testng.annotations.DataProvider;
 36 import org.testng.annotations.Test;
 37 
 38 import java.io.IOException;
 39 import java.net.DatagramSocket;
 40 import java.net.MulticastSocket;
 41 import java.net.SocketException;
 42 import java.nio.channels.DatagramChannel;
 43 
 44 import jdk.test.lib.Platform;
 45 
 46 import static org.testng.Assert.assertEquals;
 47 import static org.testng.Assert.assertTrue;
 48 import static org.testng.Assert.expectThrows;
 49 
 50 public class SetGetSendBufferSize {
 51     static final Class<SocketException> SE = SocketException.class;
 52     static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
 53 
 54     public interface DatagramSocketSupplier {
 55         DatagramSocket open() throws IOException;
 56     }
 57     static DatagramSocketSupplier supplier(DatagramSocketSupplier supplier) { return supplier; }
 58 
 59     @DataProvider
 60     public Object[][] invariants() {
 61         return new Object[][]{
 62                 {"DatagramSocket", supplier(() -> new DatagramSocket())},
 63                 {"MulticastSocket", supplier(() -> new MulticastSocket())},
 64                 {"DatagramSocketAdaptor", supplier(() -> DatagramChannel.open().socket())},
 65         };
 66     }
 67 
 68     @Test(dataProvider = "invariants")
 69     public void testSetInvalidBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
 70         var invalidArgs = new int[]{ -1, 0 };
 71 
 72         try (var socket = supplier.open()) {
 73             for (int i : invalidArgs) {
 74                 Exception ex = expectThrows(IAE, () -> socket.setSendBufferSize(i));
 75                 System.out.println(name + " got expected exception: " + ex);
 76             }
 77         }
 78     }
 79 
 80     @Test(dataProvider = "invariants")
 81     public void testSetAndGetBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
 82         var validArgs = new int[]{ 2345, 3456 };
 83 
 84         try (var socket = supplier.open()) {
 85             for (int i : validArgs) {
 86                 socket.setSendBufferSize(i);
 87                 assertEquals(socket.getSendBufferSize(), i, name);
 88             }
 89         }
 90     }
 91 
 92     @Test(dataProvider = "invariants")
 93     public void testInitialSendBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
 94         if(Platform.isOSX()) {
 95             try (var socket = supplier.open()){
 96                 assertTrue(socket.getSendBufferSize() >= 65507, name);
 97             }
 98         }
 99     }
100 
101     @Test(dataProvider = "invariants")
102     public void testSetGetAfterClose(String name, DatagramSocketSupplier supplier) throws IOException {
103         var socket = supplier.open();
104         socket.close();
105 
106         Exception setException = expectThrows(SE, () -> socket.setSendBufferSize(2345));
107         System.out.println(name + " got expected exception: " + setException);
108 
109         Exception getException = expectThrows(SE, () -> socket.getSendBufferSize());
110         System.out.println(name + " got expected exception: " + getException);
111     }
112 }
