 24 /*
 25  * @test
 26  * @bug 6505016
 27  * @summary Socket spec should clarify what getInetAddress/getPort/etc return after the Socket is closed
 28  */
 29 
 30 import java.net.*;
 31 import java.io.*;
 32 
 33 public class TestAfterClose
 34 {
 35     static int failCount;
 36 
 37     public static void main(String[] args) {
 38         try {
 39             DatagramSocket socket = new DatagramSocket();
 40             socket.connect(InetAddress.getLoopbackAddress(), 5001);
 41             test(socket, true);
 42 
 43             socket = new DatagramSocket();
 44             test(socket, false);
 45 
 46         } catch (IOException ioe) {
 47             ioe.printStackTrace();
 48         }
 49 
 50         if (failCount > 0)
 51             throw new RuntimeException("Failed: failcount = " + failCount);
 52 
 53     }
 54 
 55     static void test(DatagramSocket socket, boolean connected) throws IOException {
 56         //Before Close
 57         int socketPort = socket.getPort();
 58         InetAddress socketInetAddress = socket.getInetAddress();
 59         SocketAddress socketRemoteSocketAddress = socket.getRemoteSocketAddress();
 60 
 61         //After Close
 62         socket.close();
 63 
 64         if (connected ? !(socket.getPort() == socketPort)
 65                       : !(socket.getPort() == -1)) {
 66             System.out.println("Socket.getPort failed");
 67             failCount++;
 68         }
 69 
 70         if (connected ? !socket.getInetAddress().equals(socketInetAddress)
 71                       : !(socket.getInetAddress() == null)) {
 72             System.out.println("Socket.getInetAddress failed");
 73             failCount++;
 74         }
 75 
 76         if (connected ? !socket.getRemoteSocketAddress().equals(socketRemoteSocketAddress)
 77                       : !(socket.getRemoteSocketAddress() == null)) {
 78             System.out.println("Socket.getRemoteSocketAddresss failed");
 79             failCount++;
 80         }
 81 
 82         if (socket.getLocalPort() != -1) {
 83             System.out.println("Socket.getLocalPort failed");
 84             failCount++;
 85         }
 86 
 87         if (socket.getLocalAddress() != null) {
 88             System.out.println("Socket.getLocalAddress failed");
 89             failCount++;
 90         }
 91 
 92         if (socket.getLocalSocketAddress() != null) {
 93             System.out.println("Socket.getLocalSocketAddress failed");
 94             failCount++;
 95         }
 96 
 97         if (connected && !socket.isConnected()) {
 98             System.out.println("Socket.isConnected failed");
 99             failCount++;
100         }
101 
102         if (!socket.isBound()) {
103             System.out.println("Socket.isBound failed");
104             failCount++;
105         }
106 
107     }
108 }
