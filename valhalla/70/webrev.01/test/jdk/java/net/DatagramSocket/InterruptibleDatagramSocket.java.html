 24 import java.net.DatagramPacket;
 25 import java.net.DatagramSocket;
 26 import java.net.MulticastSocket;
 27 import java.net.SocketException;
 28 import java.net.SocketTimeoutException;
 29 import java.nio.channels.ClosedByInterruptException;
 30 import java.nio.channels.DatagramChannel;
 31 import java.util.concurrent.CountDownLatch;
 32 
 33 import static java.lang.Thread.sleep;
 34 
 35 /*
 36  * @test
 37  * @summary Check interrupt mechanism for DatagramSocket,
 38  *      MulticastSocket, and DatagramSocketAdaptor
 39  * @run main InterruptibleDatagramSocket
 40  * @run main/othervm -Djdk.net.usePlainDatagramSocketImpl InterruptibleDatagramSocket
 41  */
 42 
 43 public class InterruptibleDatagramSocket {
 44     private static void test0(DatagramSocket s) throws Exception {
 45         CountDownLatch latch = new CountDownLatch(1);
 46         Thread testThread = Thread.currentThread();
 47 
 48         Thread coordinator = new Thread(() -> {
 49             try {
 50                 latch.await();
 51                 sleep(500);
 52                 testThread.interrupt();
 53             } catch (InterruptedException e) {
 54             }
 55         });
 56         byte[] data = {0, 1, 2};
 57         DatagramPacket p = new DatagramPacket(data, data.length,
 58                 s.getLocalAddress(), s.getLocalPort());
 59         s.setSoTimeout(2000);
 60         coordinator.start();
 61         latch.countDown();
 62         try {
 63             s.receive(p);
 64         } finally {
 65             try {
 66                 coordinator.join();
 67             } catch (InterruptedException e) {
 68             }
 69         }
 70     }
 71 
 72     static void test(DatagramSocket s, boolean interruptible) throws Exception {
 73         try {
 74             test0(s);
 75             throw new RuntimeException("Receive shouldn't have succeeded");
 76         } catch (SocketTimeoutException e) {
 77             if (interruptible)
 78                 throw e;
 79             System.out.println("Got expected SocketTimeoutException: " + e);
 80         } catch (SocketException e) {
 81             if ((e.getCause() instanceof ClosedByInterruptException) && interruptible) {
 82                 System.out.println("Got expected ClosedByInterruptException: " + e);
 83             } else {
 84                 throw e;
 85             }
 86         } catch (ClosedByInterruptException e) {
 87             if (!interruptible)
 88                 throw e;
 89             System.out.println("Got expected ClosedByInterruptException: " + e);
 90         }
 91         if (s.isClosed() && !interruptible)
 92             throw new RuntimeException("DatagramSocket should not be closed");
 93         else if (!s.isClosed() && interruptible)
 94             throw new RuntimeException("DatagramSocket should be closed");
 95     }
 96 
 97     public static void main(String[] args) throws Exception {
 98         try (DatagramSocket s = new DatagramSocket()) {
 99             test(s, false);
100         }
101         try (DatagramSocket s = new MulticastSocket()) {
102             test(s, false);
103         }
104         try (DatagramSocket s = DatagramChannel.open().socket()) {
105             test(s, true);
106         }
107     }
108 }
