 41 import static org.testng.Assert.expectThrows;

 43 /*
 44  * @test
 45  * @bug 8236105
 46  * @summary Check that DatagramSocket, MulticastSocket,
 47  *          DatagramSocketAdaptor and DatagramChannel all
 48  *          throw expected Execption when passed a DatagramPacket
 49  *          with invalid details
 50  * @run testng SendCheck
 51  * @run testng/othervm -Djdk.net.usePlainDatagramSocketImpl SendCheck
 52  */
 53 
 54 public class SendCheck {
 55     private InetAddress loopbackAddr, wildcardAddr;
 56     static final Class<IOException> IOE = IOException.class;
 57     static final Class<SocketException> SE = SocketException.class;
 58 
 59     static final byte[] buf = {0, 1, 2};
 60     static DatagramSocket socket;
 61 
 62     @BeforeTest
 63     public void setUp() {
 64         try {
 65             socket = new DatagramSocket();
 66         } catch (Exception e) {
 67             throw new ExceptionInInitializerError(e);
 68         }
 69     }
 70 
 71     @AfterTest
 72     public void closeDown() {
 73         socket.close();
 74     }
 75 
 76     static final class Packet {
 77         private Packet(String description, DatagramPacket packet) {
 78             this.description = description;
 79             this.packet = packet;
 80         }
 81 
 82         final String description;
 83         final DatagramPacket packet;
 84 
 85         public String toString() {
 86             return description;
 87         }
 88 
 89         public static Packet of(DatagramPacket packet) {
 90             InetAddress address = packet.getAddress();
 91             int port = packet.getPort();
 92             String description;
 93             if (address == null) {
 94                 description = "<null>:" + port;
 95             } else if (port < 0) {
 96                 description = packet.getAddress().toString() + ":" + port;
 97             } else {
 98                 description = packet.getSocketAddress().toString();
 99             }
100             return new Packet(description, packet);
101         }
102     }
103 
104     @DataProvider(name = "packets")
105     Object[][] providerIO() throws IOException {
106         loopbackAddr = InetAddress.getLoopbackAddress();
107         wildcardAddr = new InetSocketAddress(0).getAddress();
108 
109         // loopback addr with no port set
110         var pkt1 = new DatagramPacket(buf, 0, buf.length);
111         pkt1.setAddress(loopbackAddr);
112 
113         // wildcard addr with no port set
114         var pkt2 = new DatagramPacket(buf, 0, buf.length);
115         pkt2.setAddress(wildcardAddr);
116 
117         /*
118         Commented until JDK-8236807 is fixed
119 
120         // wildcard addr w/valid port
121         var pkt3 = new DatagramPacket(buf, 0, buf.length);
122         pkt3.setAddress(wildcardAddr);
123         pkt3.setPort(socket.getLocalPort());
124         */
125 
126         List<Packet> Packets = List.of(Packet.of(pkt1), Packet.of(pkt2));
127 
128         List<Sender> senders = List.of(
129                 Sender.of(new DatagramSocket(null)),
130                 Sender.of(new MulticastSocket(null), (byte) 0),
131                 Sender.of(DatagramChannel.open()),
132                 Sender.of(DatagramChannel.open().socket()),
133                 Sender.of((MulticastSocket)
134                         DatagramChannel.open().socket(), (byte) 0)
135         );
136 
137         List<Object[]> testcases = new ArrayList<>();
138         for (var packet : Packets) {
139             addTestCaseFor(testcases, senders, packet);
140         }
141 
142         return testcases.toArray(new Object[0][0]);
143     }
144 
145     static void addTestCaseFor(List<Object[]> testcases,
146                                List<Sender> senders, Packet p) {
147         for (var s : senders) {
148             Object[] testcase = new Object[]{s, p, s.expectedException()};
149             testcases.add(testcase);
150         }
151     }
152 
153     @Test(dataProvider = "packets")
154     public static void sendCheck(Sender<IOException> sender,
155                                         Packet packet,
156                                         Class<? extends Throwable> exception) {
157         DatagramPacket pkt = packet.packet;
158         if (exception != null) {
159             Throwable t = expectThrows(exception, () -> sender.send(pkt));
160             System.out.printf("%s got expected exception %s%n",
161                     packet.toString(), t);
162         } else {
163             try {
164                 sender.send(pkt);
165             } catch (IOException e) {
166                 throw new AssertionError("Unexpected exception for "
167                         + sender + " / " + packet, e);
168             }
169         }
170     }
171 
172     interface Sender<E extends Exception> extends AutoCloseable {
173         void send(DatagramPacket p) throws E;
174 
175         void close() throws E;
176 
177         Class<? extends E> expectedException();
178 
179         static Sender<IOException> of(DatagramSocket socket) {
180             return new SenderImpl<>(socket, socket::send, socket::close, SE);
181         }
182 
183         static Sender<IOException> of(MulticastSocket socket, byte ttl) {
184             SenderImpl.Send<IOException> send =
185                     (pkt) -> socket.send(pkt, ttl);
186             return new SenderImpl<>(socket, send, socket::close, SE);
187         }
188 
189         static Sender<IOException> of(DatagramChannel socket) {
190             SenderImpl.Send<IOException> send =
191                     (pkt) -> socket.send(ByteBuffer.wrap(pkt.getData()),
192                             pkt.getSocketAddress());
193             return new SenderImpl<>(socket, send, socket::close, SE);
194         }
195     }
196 
197     static final class SenderImpl<E extends Exception> implements Sender<E> {
198         @FunctionalInterface
199         interface Send<E extends Exception> {
200             void send(DatagramPacket p) throws E;
201         }
202 
203         @FunctionalInterface
204         interface Closer<E extends Exception> {
205             void close() throws E;
206         }
207 
208         private final Send<E> send;
209         private final Closer<E> closer;
210         private final Object socket;
211         private final Class<? extends E> expectedException;
212 
213         public SenderImpl(Object socket, Send<E> send, Closer<E> closer,
214                           Class<? extends E> expectedException) {
215             this.socket = socket;
216             this.send = send;
217             this.closer = closer;
218             this.expectedException = expectedException;
219         }
220 
221         @Override
222         public void send(DatagramPacket p) throws E {
223             send.send(p);
224         }
225 
226         @Override
227         public void close() throws E {
228             closer.close();
229         }
230 
231         @Override
232         public Class<? extends E> expectedException() {
233             return expectedException;
234         }
235 
236         @Override
237         public String toString() {
238             return socket.getClass().getSimpleName();
239         }
240     }
