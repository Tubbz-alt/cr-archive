24 /*
25  * @test
26  * @bug 4173717 8243488
27  * @summary Check that setReceiveBufferSize and getReceiveBufferSize work as expected
28  * @run testng SetGetReceiveBufferSize
29  * @run testng/othervm -Djava.net.preferIPv4Stack=true SetGetReceiveBufferSize
30  * @run testng/othervm -Djdk.net.usePlainDatagramSocketImpl SetGetReceiveBufferSize
31  */
32 
33 import org.testng.annotations.DataProvider;
34 import org.testng.annotations.Test;
35 
36 import java.io.IOException;
37 import java.net.DatagramSocket;
38 import java.net.MulticastSocket;
39 import java.net.SocketException;
40 import java.nio.channels.DatagramChannel;
41 
42 import static org.testng.Assert.assertEquals;
43 import static org.testng.Assert.expectThrows;
44 
45 public class SetGetReceiveBufferSize {
46     static final Class<SocketException> SE = SocketException.class;
47     static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
48 
49     @FunctionalInterface
50     public interface DatagramSocketSupplier {
51         DatagramSocket open() throws IOException;
52     }
53     static DatagramSocketSupplier supplier(DatagramSocketSupplier supplier) { return supplier; }
54 
55     @DataProvider
56     public Object[][] invariants() {
57         return new Object[][]{
58                 {"DatagramSocket", supplier(() -> new DatagramSocket())},
59                 {"MulticastSocket", supplier(() -> new MulticastSocket())},
60                 {"DatagramSocketAdaptor", supplier(() -> DatagramChannel.open().socket())},
61         };
62     }
63 
64     @Test(dataProvider = "invariants")
65     public void testSetInvalidBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
66         var invalidArgs = new int[]{ -1, 0 };
67 
68         try (var socket = supplier.open()) {
69             for (int i : invalidArgs) {
70                 Exception ex = expectThrows(IAE, () -> socket.setReceiveBufferSize(i));
71                 System.out.println(name + " got expected exception: " + ex);
72             }
73         }
74     }
75 
76     @Test(dataProvider = "invariants")
77     public void testSetAndGetBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
78         var validArgs = new int[]{ 1234, 2345, 3456 };
79 
80         try (var socket = supplier.open()) {
81             for (int i : validArgs) {
82                 socket.setReceiveBufferSize(i);
83                 assertEquals(socket.getReceiveBufferSize(), i, name);
84             }
85         }
86     }
87 
88     @Test(dataProvider = "invariants")
89     public void testSetGetAfterClose(String name, DatagramSocketSupplier supplier) throws IOException {
90         var socket = supplier.open();
91         socket.close();
92 
93         Exception setException = expectThrows(SE, () -> socket.setReceiveBufferSize(2345));
94         System.out.println(name + " got expected exception: " + setException);
95 
96         Exception getException = expectThrows(SE, () -> socket.getReceiveBufferSize());
97         System.out.println(name + " got expected exception: " + getException);
98     }
99 }
