 24 /*
 25  * @test
 26  * @bug 8233141
 27  * @summary DatagramSocket.send should throw IllegalArgumentException
 28  *          when the packet address is not correctly set.
 29  * @run main AddressNotSet
 30  * @run main/othervm -Djdk.net.usePlainDatagramSocketImpl AddressNotSet
 31  */
 32 
 33 import java.net.DatagramPacket;
 34 import java.net.DatagramSocket;
 35 import java.net.InetAddress;
 36 import java.net.MulticastSocket;
 37 import java.net.SocketAddress;
 38 import java.nio.channels.DatagramChannel;
 39 
 40 import static java.lang.System.out;
 41 
 42 public class AddressNotSet {
 43 
 44     final InetAddress loopbackAddress = InetAddress.getLoopbackAddress();
 45     final DatagramSocket serversock;
 46     int i;
 47     AddressNotSet() throws Exception {
 48         serversock = new DatagramSocket(0, loopbackAddress);
 49     }
 50 
 51     public static void main (String args[]) throws Exception {
 52         new AddressNotSet().run();
 53     }
 54 
 55     public void run() throws Exception {
 56         try (var ss = serversock) {
 57             try (DatagramSocket sock = new DatagramSocket()) {
 58                 test(sock);
 59             }
 60             try (DatagramSocket sock = new MulticastSocket()) {
 61                 test(sock);
 62             }
 63             try (DatagramSocket sock = DatagramChannel.open().socket()) {
 64                 test(sock);
 65             }
 66         }
 67     }
 68 
 69     private void test(DatagramSocket sock) throws Exception {
 70         out.println("Testing with " + sock.getClass());
 71         InetAddress addr = loopbackAddress;
 72         byte[] buf;
 73         DatagramPacket p;
 74         int port = serversock.getLocalPort();
 75         SocketAddress connectedAddress = serversock.getLocalSocketAddress();
 76 
 77         out.println("Checking send to non-connected address ...");
 78         try {
 79             out.println("Checking send with no packet address");
 80             buf = ("Hello, server"+(++i)).getBytes();
 81             p = new DatagramPacket(buf, buf.length);
 82             sock.send(p);
 83             throw new AssertionError("Expected IllegalArgumentException not received");
 84         } catch (IllegalArgumentException x) {
 85             out.println("Got expected exception: " + x);
 86         }
 87 
 88         out.println("Checking send to valid address");
 89         buf = ("Hello, server"+(++i)).getBytes();
 90         p = new DatagramPacket(buf, buf.length, addr, port);
 91         sock.send(p);
 92         serversock.receive(p);
 93 
 94         out.println("Connecting to server address: " + connectedAddress);
 95         sock.connect(connectedAddress);
 96 
 97         try {
 98             out.println("Checking send with different address than connected");
 99             buf = ("Hello, server"+(++i)).getBytes();
100             p = new DatagramPacket(buf, buf.length, addr, port+1);
101             sock.send(p);
102             throw new AssertionError("Expected IllegalArgumentException not received");
103         } catch (IllegalArgumentException x) {
104             out.println("Got expected exception: " + x);
105         }
106 
107         out.println("Checking send to valid address");
108         buf = ("Hello, server"+(++i)).getBytes();
109         p = new DatagramPacket(buf, buf.length, addr, port);
110         sock.send(p);
111         serversock.receive(p);
112 
113         if (sock instanceof MulticastSocket) {
114             sock.disconnect();
115             testTTL((MulticastSocket)sock);
116         }
117     }
118 
119     private void testTTL(MulticastSocket sock) throws Exception {
120         out.println("Testing deprecated send TTL with " + sock.getClass());
121         final byte ttl = 100;
122         InetAddress addr = loopbackAddress;
123         byte[] buf;
124         DatagramPacket p;
125         int port = serversock.getLocalPort();
126 
127         out.println("Checking send to non-connected address ...");
128         try {
129             out.println("Checking send with no packet address");
130             buf = ("Hello, server"+(++i)).getBytes();
131             p = new DatagramPacket(buf, buf.length);
132             sock.send(p,ttl);
133             throw new AssertionError("Expected IllegalArgumentException not received");
134         } catch (IllegalArgumentException x) {
135             out.println("Got expected exception: " + x);
136         }
137 
138         out.println("Connecting to connected address: " + sock);
139         sock.connect(addr, port);
140 
141         try {
142             out.println("Checking send with different address than connected");
143             buf = ("Hello, server"+(++i)).getBytes();
144             p = new DatagramPacket(buf, buf.length, addr, port+1);
145             sock.send(p, ttl);
146             throw new AssertionError("Expected IllegalArgumentException not received");
147         } catch (IllegalArgumentException x) {
148             out.println("Got expected exception: " + x);
149         }
150     }
151 }
