 41 import static org.testng.Assert.expectThrows;

 43 /*
 44  * @test
 45  * @bug 8236105
 46  * @summary Check that DatagramSocket, MulticastSocket,
 47  *          DatagramSocketAdaptor and DatagramChannel all
 48  *          throw expected Execption when passed a DatagramPacket
 49  *          with invalid details
 50  * @run testng/othervm SendCheck

 51  */
 52 
 53 public class SendCheck {
 54     private InetAddress loopbackAddr, wildcardAddr;
 55     static final Class<IOException> IOE = IOException.class;
 56     static final Class<SocketException> SE = SocketException.class;
 57 
 58     static final byte[] buf = {0, 1, 2};
 59     static DatagramSocket socket;
 60 
 61     @BeforeTest
 62     public void setUp() {
 63         try {
 64             socket = new DatagramSocket();
 65         } catch (Exception e) {
 66             throw new ExceptionInInitializerError(e);
 67         }
 68     }
 69 
 70     @AfterTest
 71     public void closeDown() {
 72         socket.close();
 73     }
 74 
 75     static final class Packet {
 76         private Packet(String description, DatagramPacket packet) {
 77             this.description = description;
 78             this.packet = packet;
 79         }
 80 
 81         final String description;
 82         final DatagramPacket packet;
 83 
 84         public String toString() {
 85             return description;
 86         }
 87 
 88         public static Packet of(DatagramPacket packet) {
 89             InetAddress address = packet.getAddress();
 90             int port = packet.getPort();
 91             String description;
 92             if (address == null) {
 93                 description = "<null>:" + port;
 94             } else if (port < 0) {
 95                 description = packet.getAddress().toString() + ":" + port;
 96             } else {
 97                 description = packet.getSocketAddress().toString();
 98             }
 99             return new Packet(description, packet);
100         }
101     }
102 
103     @DataProvider(name = "packets")
104     Object[][] providerIO() throws IOException {
105         loopbackAddr = InetAddress.getLoopbackAddress();
106         wildcardAddr = new InetSocketAddress(0).getAddress();
107 
108         // loopback addr with no port set
109         var pkt1 = new DatagramPacket(buf, 0, buf.length);
110         pkt1.setAddress(loopbackAddr);
111 
112         // wildcard addr with no port set
113         var pkt2 = new DatagramPacket(buf, 0, buf.length);
114         pkt2.setAddress(wildcardAddr);
115 
116         /*
117         Commented until JDK-8236807 is fixed
118 
119         // wildcard addr w/valid port
120         var pkt3 = new DatagramPacket(buf, 0, buf.length);
121         pkt3.setAddress(wildcardAddr);
122         pkt3.setPort(socket.getLocalPort());
123         */
124 
125         List<Packet> Packets = List.of(Packet.of(pkt1), Packet.of(pkt2));
126 
127         List<Sender> senders = List.of(
128                 Sender.of(new DatagramSocket(null)),
129                 Sender.of(new MulticastSocket(null), (byte) 0),
130                 Sender.of(DatagramChannel.open()),
131                 Sender.of(DatagramChannel.open().socket()),
132                 Sender.of((MulticastSocket)
133                         DatagramChannel.open().socket(), (byte) 0)
134         );
135 
136         List<Object[]> testcases = new ArrayList<>();
137         for (var packet : Packets) {
138             addTestCaseFor(testcases, senders, packet);
139         }
140 
141         return testcases.toArray(new Object[0][0]);
142     }
143 
144     static void addTestCaseFor(List<Object[]> testcases,
145                                List<Sender> senders, Packet p) {
146         for (var s : senders) {
147             Object[] testcase = new Object[]{s, p, s.expectedException()};
148             testcases.add(testcase);
149         }
150     }
151 
152     @Test(dataProvider = "packets")
153     public static void sendCheck(Sender<IOException> sender,
154                                         Packet packet,
155                                         Class<? extends Throwable> exception) {
156         DatagramPacket pkt = packet.packet;
157         if (exception != null) {
158             Throwable t = expectThrows(exception, () -> sender.send(pkt));
159             System.out.printf("%s got expected exception %s%n",
160                     packet.toString(), t);
161         } else {
162             try {
163                 sender.send(pkt);
164             } catch (IOException e) {
165                 throw new AssertionError("Unexpected exception for "
166                         + sender + " / " + packet, e);
167             }
168         }
169     }
170 
171     interface Sender<E extends Exception> extends AutoCloseable {
172         void send(DatagramPacket p) throws E;
173 
174         void close() throws E;
175 
176         Class<? extends E> expectedException();
177 
178         static Sender<IOException> of(DatagramSocket socket) {
179             return new SenderImpl<>(socket, socket::send, socket::close, SE);
180         }
181 
182         static Sender<IOException> of(MulticastSocket socket, byte ttl) {
183             SenderImpl.Send<IOException> send =
184                     (pkt) -> socket.send(pkt, ttl);
185             return new SenderImpl<>(socket, send, socket::close, SE);
186         }
187 
188         static Sender<IOException> of(DatagramChannel socket) {
189             SenderImpl.Send<IOException> send =
190                     (pkt) -> socket.send(ByteBuffer.wrap(pkt.getData()),
191                             pkt.getSocketAddress());
192             return new SenderImpl<>(socket, send, socket::close, SE);
193         }
194     }
195 
196     static final class SenderImpl<E extends Exception> implements Sender<E> {
197         @FunctionalInterface
198         interface Send<E extends Exception> {
199             void send(DatagramPacket p) throws E;
200         }
201 
202         @FunctionalInterface
203         interface Closer<E extends Exception> {
204             void close() throws E;
205         }
206 
207         private final Send<E> send;
208         private final Closer<E> closer;
209         private final Object socket;
210         private final Class<? extends E> expectedException;
211 
212         public SenderImpl(Object socket, Send<E> send, Closer<E> closer,
213                           Class<? extends E> expectedException) {
214             this.socket = socket;
215             this.send = send;
216             this.closer = closer;
217             this.expectedException = expectedException;
218         }
219 
220         @Override
221         public void send(DatagramPacket p) throws E {
222             send.send(p);
223         }
224 
225         @Override
226         public void close() throws E {
227             closer.close();
228         }
229 
230         @Override
231         public Class<? extends E> expectedException() {
232             return expectedException;
233         }
234 
235         @Override
236         public String toString() {
237             return socket.getClass().getSimpleName();
238         }
239     }
