 24 /*
 25  * @test
 26  * @bug 8243488
 27  * @library /test/lib
 28  * @build jdk.test.lib.Platform
 29  * @summary Check that setSendBufferSize and getSendBufferSize work as expected
 30  * @run testng SetGetSendBufferSize
 31  * @run testng/othervm -Djava.net.preferIPv4Stack=true SetGetSendBufferSize

 32  */
 33 
 34 import org.testng.annotations.DataProvider;
 35 import org.testng.annotations.Test;
 36 
 37 import java.io.IOException;
 38 import java.net.DatagramSocket;
 39 import java.net.MulticastSocket;
 40 import java.net.SocketException;
 41 import java.nio.channels.DatagramChannel;
 42 
 43 import jdk.test.lib.Platform;
 44 
 45 import static org.testng.Assert.assertEquals;
 46 import static org.testng.Assert.assertTrue;
 47 import static org.testng.Assert.expectThrows;
 48 
 49 public class SetGetSendBufferSize {
 50     static final Class<SocketException> SE = SocketException.class;
 51     static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
 52 
 53     public interface DatagramSocketSupplier {
 54         DatagramSocket open() throws IOException;
 55     }
 56     static DatagramSocketSupplier supplier(DatagramSocketSupplier supplier) { return supplier; }
 57 
 58     @DataProvider
 59     public Object[][] invariants() {
 60         return new Object[][]{
 61                 {"DatagramSocket", supplier(() -> new DatagramSocket())},
 62                 {"MulticastSocket", supplier(() -> new MulticastSocket())},
 63                 {"DatagramSocketAdaptor", supplier(() -> DatagramChannel.open().socket())},
 64         };
 65     }
 66 
 67     @Test(dataProvider = "invariants")
 68     public void testSetInvalidBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
 69         var invalidArgs = new int[]{ -1, 0 };
 70 
 71         try (var socket = supplier.open()) {
 72             for (int i : invalidArgs) {
 73                 Exception ex = expectThrows(IAE, () -> socket.setSendBufferSize(i));
 74                 System.out.println(name + " got expected exception: " + ex);
 75             }
 76         }
 77     }
 78 
 79     @Test(dataProvider = "invariants")
 80     public void testSetAndGetBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
 81         var validArgs = new int[]{ 2345, 3456 };
 82 
 83         try (var socket = supplier.open()) {
 84             for (int i : validArgs) {
 85                 socket.setSendBufferSize(i);
 86                 assertEquals(socket.getSendBufferSize(), i, name);
 87             }
 88         }
 89     }
 90 
 91     @Test(dataProvider = "invariants")
 92     public void testInitialSendBufferSize(String name, DatagramSocketSupplier supplier) throws IOException {
 93         if(Platform.isOSX()) {
 94             try (var socket = supplier.open()){
 95                 assertTrue(socket.getSendBufferSize() >= 65507, name);
 96             }
 97         }
 98     }
 99 
100     @Test(dataProvider = "invariants")
101     public void testSetGetAfterClose(String name, DatagramSocketSupplier supplier) throws IOException {
102         var socket = supplier.open();
103         socket.close();
104 
105         Exception setException = expectThrows(SE, () -> socket.setSendBufferSize(2345));
106         System.out.println(name + " got expected exception: " + setException);
107 
108         Exception getException = expectThrows(SE, () -> socket.getSendBufferSize());
109         System.out.println(name + " got expected exception: " + getException);
110     }
111 }
