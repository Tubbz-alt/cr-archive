 27  * @summary Basic test for WebSocketHandshakeException
 28  * @library /test/lib
 29  * @build jdk.test.lib.net.SimpleSSLContext
 30  * @modules java.net.http
 31  *          jdk.httpserver
 32  * @run testng/othervm -Djdk.internal.httpclient.debug=true WSHandshakeExceptionTest

 34 
 35 import com.sun.net.httpserver.HttpServer;
 36 import com.sun.net.httpserver.HttpsConfigurator;
 37 import com.sun.net.httpserver.HttpsServer;
 38 
 39 import java.net.InetAddress;
 40 import java.net.http.HttpClient;
 41 import java.net.http.WebSocket;
 42 import java.net.http.WebSocketHandshakeException;
 43 import jdk.test.lib.net.SimpleSSLContext;
 44 import org.testng.annotations.AfterTest;
 45 import org.testng.annotations.BeforeTest;
 46 import org.testng.annotations.DataProvider;
 47 import org.testng.annotations.Test;
 48 import javax.net.ssl.SSLContext;
 49 import java.net.InetSocketAddress;
 50 import java.net.URI;
 51 import java.util.concurrent.CompletionException;
 52 import java.util.concurrent.ExecutionException;
 53 import java.util.concurrent.ExecutorService;
 54 import java.util.concurrent.Executors;
 55 
 56 import static java.net.http.HttpClient.Builder.NO_PROXY;
 57 import static org.testng.Assert.assertEquals;
 58 import static org.testng.Assert.assertNotNull;
 59 import static org.testng.Assert.assertTrue;
 60 import static org.testng.Assert.fail;
 61 import static java.lang.System.out;
 62 
 63 public class WSHandshakeExceptionTest {
 64 
 65     SSLContext sslContext;
 66     HttpServer httpTestServer;         // HTTP/1.1    [ 2 servers ]
 67     HttpsServer httpsTestServer;       // HTTPS/1.1
 68     String httpURI;
 69     String httpsURI;
 70 
 71     static final int ITERATION_COUNT = 10;
 72     // a shared executor helps reduce the amount of threads created by the test
 73     static final ExecutorService executor = Executors.newCachedThreadPool();
 74 
 75     @DataProvider(name = "variants")
 76     public Object[][] variants() {
 77         return new Object[][]{
 78                 { httpURI,    false },
 79                 { httpsURI,   false },
 80                 { httpURI,    true },
 81                 { httpsURI,   true },
 82         };
 83     }
 84 
 85     HttpClient newHttpClient() {
 86         return HttpClient.newBuilder()
 87                          .proxy(NO_PROXY)
 88                          .executor(executor)
 89                          .sslContext(sslContext)
 90                          .build();
 91     }
 92 
 93     @Test(dataProvider = "variants")
 94     public void test(String uri, boolean sameClient) {
 95         HttpClient client = null;
 96         for (int i = 0; i < ITERATION_COUNT; i++) {
 97             System.out.printf("iteration %s%n", i);
 98             if (!sameClient || client == null)
 99                 client = newHttpClient();
100 
101             try {
102                 client.newWebSocketBuilder()
103                       .buildAsync(URI.create(uri), new WebSocket.Listener() { })
104                       .join();
105                 fail("Expected to throw");
106             } catch (CompletionException ce) {
107                 Throwable t = getCompletionCause(ce);
108                 if (!(t instanceof WebSocketHandshakeException)) {
109                     throw new AssertionError("Unexpected exception", t);
110                 }
111                 WebSocketHandshakeException wse = (WebSocketHandshakeException) t;
112                 assertNotNull(wse.getResponse());
113                 out.println("Status code is " + wse.getResponse().statusCode());
114                 out.println("Response is " + wse.getResponse().body());
115                 assertTrue(((String)wse.getResponse().body()).contains("404"));
116                 assertEquals(wse.getResponse().statusCode(), 404);
117             }
118         }
119     }
120 
121     @BeforeTest
122     public void setup() throws Exception {
123         sslContext = new SimpleSSLContext().get();
124         if (sslContext == null)
125             throw new AssertionError("Unexpected null sslContext");
126 
127         // HTTP/1.1
128         InetSocketAddress sa = new InetSocketAddress(InetAddress.getLoopbackAddress(), 0);
129         httpTestServer = HttpServer.create(sa, 0);
130         httpURI = "ws://localhost:" + httpTestServer.getAddress().getPort() + "/";
131 
132         httpsTestServer = HttpsServer.create(sa, 0);
133         httpsTestServer.setHttpsConfigurator(new HttpsConfigurator(sslContext));
134         httpsURI = "wss://localhost:" + httpsTestServer.getAddress().getPort() + "/";
135 
136         httpTestServer.start();
137         httpsTestServer.start();
138     }
139 
140     @AfterTest
141     public void teardown() {
142         httpTestServer.stop(0);
143         httpsTestServer.stop(0);
144         executor.shutdownNow();
145     }
146 
147     private static Throwable getCompletionCause(Throwable x) {
148         if (!(x instanceof CompletionException)
149                 && !(x instanceof ExecutionException)) return x;
150         final Throwable cause = x.getCause();
151         if (cause == null) {
152             throw new InternalError("Unexpected null cause", x);
153         }
154         return cause;
155     }
156 }
