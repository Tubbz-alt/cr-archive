 70 public class WSHandshakeExceptionTest {
 71 
 72     SSLContext sslContext;
 73     HttpServer httpTestServer;         // HTTP/1.1    [ 2 servers ]
 74     HttpsServer httpsTestServer;       // HTTPS/1.1
 75     String httpURI;
 76     String httpsURI;
 77     String httpNonUtf8URI;
 78     String httpsNonUtf8URI;
 79 
 80     static final int ITERATION_COUNT = 10;
 81     // a shared executor helps reduce the amount of threads created by the test
 82     static final ExecutorService executor = Executors.newCachedThreadPool();
 83 
 84     @DataProvider(name = "variants")
 85     public Object[][] variants() {
 86         return new Object[][]{
 87                 { httpURI,           false },
 88                 { httpsURI,          false },
 89                 { httpURI,           true  },
 90                 { httpsURI,          true  },
 91 
 92                 { httpNonUtf8URI,    true  },
 93                 { httpsNonUtf8URI,   true  },
 94                 { httpNonUtf8URI,    false },
 95                 { httpsNonUtf8URI,   false }
 96         };
 97     }
 98 
 99     HttpClient newHttpClient() {
100         return HttpClient.newBuilder()
101                          .proxy(NO_PROXY)
102                          .executor(executor)
103                          .sslContext(sslContext)
104                          .build();
105     }
106 
107     @Test(dataProvider = "variants")
108     public void test(String uri, boolean sameClient) {
109         HttpClient client = null;
110         for (int i = 0; i < ITERATION_COUNT; i++) {
111             System.out.printf("iteration %s%n", i);
112             if (!sameClient || client == null)
113                 client = newHttpClient();
114 
115             try {
116                 client.newWebSocketBuilder()
117                       .buildAsync(URI.create(uri), new WebSocket.Listener() { })
118                       .join();
119                 fail("Expected to throw");
120             } catch (CompletionException ce) {
121                 Throwable t = getCompletionCause(ce);
122                 if (!(t instanceof WebSocketHandshakeException)) {
123                     throw new AssertionError("Unexpected exception", t);
124                 }
125                 WebSocketHandshakeException wse = (WebSocketHandshakeException) t;
126                 assertNotNull(wse.getResponse());
127                 assertNotNull(wse.getResponse().body());
128                 assertEquals(wse.getResponse().body().getClass(), String.class);
129                 String body = (String)wse.getResponse().body();
130                 out.println("Status code is " + wse.getResponse().statusCode());
131                 out.println("Response is " + body);
132                 if(uri.contains("/nonutf8body")) {
133                     // the invalid sequence 0xFF should have been replaced
134                     // by the replacement character (U+FFFD)
135                     assertTrue(body.equals("\ufffd"));
136                 }
137                 else {
138                     // default HttpServer 404 body expected
139                     assertTrue(body.contains("404"));
140                 }
141                 assertEquals(wse.getResponse().statusCode(), 404);
142             }
143         }
144     }
145 
146     @BeforeTest
147     public void setup() throws Exception {
148         sslContext = new SimpleSSLContext().get();
149         if (sslContext == null)
150             throw new AssertionError("Unexpected null sslContext");
151 
152         // HTTP/1.1
153         InetSocketAddress sa = new InetSocketAddress(InetAddress.getLoopbackAddress(), 0);
154         httpTestServer = HttpServer.create(sa, 0);
155         httpURI = "ws://localhost:" + httpTestServer.getAddress().getPort() + "/";
156         httpNonUtf8URI = "ws://localhost:" + httpTestServer.getAddress().getPort() + "/nonutf8body";
157         httpTestServer.createContext("/nonutf8body", new BodyHandler());
158 
159         httpsTestServer = HttpsServer.create(sa, 0);
160         httpsTestServer.setHttpsConfigurator(new HttpsConfigurator(sslContext));
161         httpsURI = "wss://localhost:" + httpsTestServer.getAddress().getPort() + "/";
162         httpsNonUtf8URI = "wss://localhost:" + httpsTestServer.getAddress().getPort() + "/nonutf8body";
163         httpsTestServer.createContext("/nonutf8body", new BodyHandler());
164 
165         httpTestServer.start();
166         httpsTestServer.start();
167     }
168 
169     @AfterTest
170     public void teardown() {
171         httpTestServer.stop(0);
172         httpsTestServer.stop(0);
173         executor.shutdownNow();
174     }
175 
176     private static Throwable getCompletionCause(Throwable x) {
177         if (!(x instanceof CompletionException)
178                 && !(x instanceof ExecutionException)) return x;
179         final Throwable cause = x.getCause();
180         if (cause == null) {
181             throw new InternalError("Unexpected null cause", x);
182         }
183         return cause;
184     }
185 
186     static class BodyHandler implements HttpHandler {
187 
188         @Override
189         public void handle(HttpExchange e) throws IOException {
190             try(InputStream is = e.getRequestBody();
191                 OutputStream os = e.getResponseBody()) {
192                 byte[] invalidUtf8 = {(byte)0xFF}; //Invalid utf-8 byte
193                 e.sendResponseHeaders(404, invalidUtf8.length);
194                 os.write(invalidUtf8);
195             }
196         }
197     }
