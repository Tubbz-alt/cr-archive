 59     private String getKeyPairGeneratorName() {
 60         int withIndex = algorithm.lastIndexOf("with");
 61         if (withIndex < 0) {
 62             return algorithm;
 63         }
 64         String tail = algorithm.substring(withIndex + 4);
 65         return "ECDSA".equals(tail) ? "EC" : tail;
 66     }
