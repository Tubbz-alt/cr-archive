 23 package org.openjdk.bench.javax.crypto.full;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.Param;
 27 import org.openjdk.jmh.annotations.Setup;
 28 
 29 import java.security.InvalidKeyException;
 30 import java.security.KeyPair;
 31 import java.security.KeyPairGenerator;
 32 import java.security.NoSuchAlgorithmException;
 33 import java.security.PrivateKey;
 34 import java.security.PublicKey;
 35 import java.security.Signature;
 36 import java.security.SignatureException;
 37 import java.util.Random;
 38 
 39 public class SignatureBench extends CryptoBase {
 40 
 41     public static final int SET_SIZE = 128;
 42 
 43     @Param({"SHA256withDSA"})
 44     private String algorithm;
 45 
 46     @Param({"1024", "16384"})
 47     int dataSize;
 48 
 49     @Param({"1024"})
 50     private int keyLength;
 51 
 52     private PrivateKey privateKey;
 53     private PublicKey publicKey;
 54     private byte[][] data;
 55     private byte[][] signedData;
 56     int index;
 57 
 58 
 59     private String getKeyPairGeneratorName() {
 60         String tail = algorithm.substring(algorithm.lastIndexOf("with") + 4);
 61         return "ECDSA".equals(tail) ? "EC" : tail;
 62     }
 63 
 64     @Setup()
 65     public void setup() throws NoSuchAlgorithmException, InvalidKeyException, SignatureException {
 66         setupProvider();
 67         KeyPairGenerator kpg = KeyPairGenerator.getInstance(getKeyPairGeneratorName());
 68         kpg.initialize(keyLength);
 69         KeyPair keys = kpg.generateKeyPair();
 70         this.privateKey = keys.getPrivate();
 71         this.publicKey = keys.getPublic();
 72         data = fillRandom(new byte[SET_SIZE][dataSize]);
 73         signedData = new byte[data.length][];
 74         for (int i = 0; i < data.length; i++) {
 75             signedData[i] = sign(data[i]);
 76         }
 77 
 78     }
 79 
 80     public byte[] sign(byte[] data) throws NoSuchAlgorithmException, InvalidKeyException, SignatureException {
 81         Signature signature = (prov == null) ? Signature.getInstance(algorithm) : Signature.getInstance(algorithm, prov);
 82         signature.initSign(privateKey);
 83         signature.update(data);
 84         return signature.sign();
 85     }
 86 
 87     @Benchmark
 88     public byte[] sign() throws NoSuchAlgorithmException, InvalidKeyException, SignatureException {
 89         byte[] d = data[index];
 90         index = (index + 1) % SET_SIZE;
 91         return sign(d);
 92     }
 93 
 94     @Benchmark
 95     public boolean verify() throws NoSuchAlgorithmException, InvalidKeyException, SignatureException {
 96         Signature signature = (prov == null) ? Signature.getInstance(algorithm) : Signature.getInstance(algorithm, prov);
 97         signature.initVerify(publicKey);
 98         byte[] d = data[index];
 99         byte[] s = signedData[index];
100         index = (index + 1) % SET_SIZE;
101         signature.update(d);
102         return signature.verify(s);
103     }
104 
105     public static class RSA extends SignatureBench {
106 
107         @Param({"MD5withRSA", "SHA256withRSA"})
108         private String algorithm;
109 
110         @Param({"1024", "2048", "3072"})
111         private int keyLength;
112 
113     }
114 
115     public static class ECDSA extends SignatureBench {
116 
117         @Param({"SHA256withECDSA"})
118         private String algorithm;
119 
120         @Param({"160", "224", "256"})
121         private int keyLength;
122 
123     }
124 
125 }
