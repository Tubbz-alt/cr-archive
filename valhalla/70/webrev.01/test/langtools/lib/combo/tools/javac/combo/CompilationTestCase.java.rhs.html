 25 package tools.javac.combo;
 26 
 27 import java.io.File;
 28 import java.io.IOException;
 29 
 30 import java.util.Arrays;
 31 import java.util.function.Consumer;
 32 import java.util.stream.IntStream;
 33 
 34 import javax.tools.Diagnostic;
 35 
 36 import org.testng.ITestResult;
 37 import org.testng.annotations.AfterMethod;
 38 import org.testng.annotations.Test;
 39 
 40 import static java.util.stream.Collectors.toList;
 41 
 42 /**
 43  * Base class for negative and positive compilation tests.
 44  */
 45 @Test
 46 public class CompilationTestCase extends JavacTemplateTestBase {
 47     private String[] compileOptions = new String[] { };
 48     private String defaultFileName = "Source.java";
 49     private String programShell = "#";
 50 
 51     @AfterMethod
 52     public void dumpTemplateIfError(ITestResult result) {
 53         // Make sure offending template ends up in log file on failure
 54         if (!result.isSuccess()) {
 55             System.err.printf("Diagnostics: %s%nTemplate: %s%n", diags.errorKeys(),
 56                               sourceFiles.stream().map(p -> p.snd).collect(toList()));
 57         }
 58     }
 59 
 60     protected void setProgramShell(String shell) {
 61         programShell = shell;
 62     }
 63 
 64     protected void setCompileOptions(String... options) {
 65         compileOptions = options.clone();
 66     }
 67 
 68     protected void appendCompileOptions(String... additionalOptions) {
 69         String[] moreOptions = additionalOptions.clone();
 70         String[] newCompileOptions = Arrays.copyOf(compileOptions, compileOptions.length + additionalOptions.length);
 71         IntStream.range(0, additionalOptions.length).forEach(i -> {
 72             newCompileOptions[newCompileOptions.length - additionalOptions.length + i] = additionalOptions[i];
 73         });
 74         compileOptions = newCompileOptions;
 75     }
 76 
 77     protected void removeLastCompileOptions(int i) {
 78         if (i < 0) {
 79             throw new AssertionError("unexpected negative value " + i);
 80         }
 81         if (i >= compileOptions.length) {
 82             compileOptions = new String[] {};
 83         } else {
 84             compileOptions = Arrays.copyOf(compileOptions, compileOptions.length - i);
 85         }
 86     }
 87 
 88     protected void setDefaultFilename(String name) {
 89         defaultFileName = name;
 90     }
 91 
 92     protected String expandMarkers(String... constructs) {
 93         String s = programShell;
 94         for (String c : constructs)
 95             s = s.replaceFirst("#", c);
 96         return s;
 97     }
 98 
 99     private File assertCompile(String program, Runnable postTest, boolean generate) {
100         reset();
101         addCompileOptions(compileOptions);
102         addSourceFile(defaultFileName, program);
103         File dir = null;
104         try {
105             dir = compile(generate);
106         }
107         catch (IOException e) {
108             throw new RuntimeException(e);
109         }
110         postTest.run();
111         return dir;
112     }
113 
114     protected void assertOK(String... constructs) {
115         assertCompile(expandMarkers(constructs), this::assertCompileSucceeded, false);
116     }
117 
118     protected File assertOK(boolean generate, String... constructs) {
119         return assertCompile(expandMarkers(constructs), this::assertCompileSucceeded, generate);
120     }
121 
122     protected void assertOKWithWarning(String warning, String... constructs) {
123         assertCompile(expandMarkers(constructs), () -> assertCompileSucceededWithWarning(warning), false);
124     }
125 
126     protected void assertFail(String expectedDiag, String... constructs) {
127         assertCompile(expandMarkers(constructs), () -> assertCompileFailed(expectedDiag), false);
128     }
129 
130     protected void assertFail(String expectedDiag, Consumer<Diagnostic<?>> diagConsumer, String... constructs) {
131         assertCompile(expandMarkers(constructs), () -> assertCompileFailed(expectedDiag, diagConsumer), false);
132     }
133 }
