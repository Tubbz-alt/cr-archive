 26 /**
 27  * RecordMemberTests
 28  *
 29  * @test

 31  * @run testng/othervm --enable-preview RecordMemberTests

 34 import java.lang.reflect.Constructor;
 35 import java.lang.reflect.Field;
 36 import java.lang.reflect.Method;
 37 import java.lang.reflect.Modifier;
 38 import java.lang.reflect.Parameter;
 39 import java.util.List;
 40 import java.util.function.Supplier;
 41 
 42 import org.testng.annotations.*;
 43 import static org.testng.Assert.*;
 44 
 45 @Test
 46 public class RecordMemberTests {
 47     public record R1(int i, int j) {}
 48 
 49     public record R2(int i, int j) {
 50         public R2 {}
 51     }
 52 
 53     public record R3(int i, int j) {
 54         public R3(int i, int j) {
 55             this.i = i;
 56             this.j = j;
 57         }
 58     }
 59 
 60     public record R4(int i, int j) {
 61         public R4(int i, int j) { this.i = this.j = 0; }
 62     }
 63 
 64     R1 r1 = new R1(1, 2);
 65     R2 r2 = new R2(1, 2);
 66     R3 r3 = new R3(1, 2);
 67     R4 r4 = new R4(1, 2);
 68 
 69     public void testConstruction() {
 70         for (int i : new int[] { r1.i, r2.i, r3.i,
 71                                  r1.i(), r2.i(), r3.i() })
 72             assertEquals(i, 1);
 73 
 74         for (int j : new int[] { r1.j, r2.j, r3.j,
 75                                  r1.j(), r2.j(), r3.j() })
 76             assertEquals(j, 2);
 77 
 78         assertEquals(r4.i, 0);
 79         assertEquals(r4.j, 0);
 80     }
 81 
 82     public void testConstructorParameterNames() throws ReflectiveOperationException {
 83         for (Class cl : List.of(R1.class, R2.class, R3.class, R4.class)) {
 84             Constructor c = cl.getConstructor(int.class, int.class);
 85             assertNotNull(c);
 86             Parameter[] parameters = c.getParameters();
 87             assertEquals(parameters.length, 2);
 88             assertEquals(parameters[0].getName(), "i");
 89             assertEquals(parameters[1].getName(), "j");
 90         }
 91     }
 92 
 93     public void testSuperclass() {
 94         assertEquals(R1.class.getSuperclass(), Record.class);
 95         // class is final
 96         assertTrue((R1.class.getModifiers() & Modifier.FINAL) != 0);
 97     }
 98 
 99     public void testMandatedMembersPresent() throws ReflectiveOperationException {
100         // fields are present, of the right type, final and private
101         assertEquals(R1.class.getDeclaredFields().length, 2);
102         for (String s : List.of("i", "j")) {
103             Field iField = R1.class.getDeclaredField(s);
104             assertEquals(iField.getType(), int.class);
105             assertEquals((iField.getModifiers() & Modifier.STATIC), 0);
106             assertTrue((iField.getModifiers() & Modifier.PRIVATE) != 0);
107             assertTrue((iField.getModifiers() & Modifier.FINAL) != 0);
108         }
109 
110         // methods are present, of the right descriptor, and public/instance/concrete
111         for (String s : List.of("i", "j")) {
112             Method iMethod = R1.class.getDeclaredMethod(s);
113             assertEquals(iMethod.getReturnType(), int.class);
114             assertEquals(iMethod.getParameterCount(), 0);
115             assertEquals((iMethod.getModifiers() & (Modifier.PRIVATE | Modifier.PROTECTED | Modifier.STATIC | Modifier.ABSTRACT)), 0);
116         }
117 
118         Constructor c = R1.class.getConstructor(int.class, int.class);
119         R1 r1 = (R1) c.newInstance(1, 2);
120         assertEquals(r1.i(), 1);
121         assertEquals(r1.j(), 2);
122     }
123 
124     record OrdinaryMembers(int x) {
125         public static String ss;
126         public static String ssf () { return ss; }
127         public String sf () { return "instance"; }
128     }
129 
130     public void testOrdinaryMembers() {
131         OrdinaryMembers.ss = "foo";
132         assertEquals(OrdinaryMembers.ssf(), "foo");
133         OrdinaryMembers o = new OrdinaryMembers(3);
134         assertEquals(o.sf(), "instance");
135     }
136 
137     class LocalRecordHelper {
138         Class<?> m(int x) {
139             record R (int x) { }
140             assertEquals(new R(x).x(), x);
141             return R.class;
142         }
143     }
144 
145     public void testLocalRecordsStatic() {
146         Class<?> c = new LocalRecordHelper().m(3);
147         String message = c.toGenericString();
148         assertTrue(c.isRecord(), message);
149         assertTrue((c.getModifiers() & Modifier.STATIC) != 0, message);
150         assertTrue((c.getModifiers() & Modifier.FINAL)  != 0, message);
151     }
152 
153     static class NestedRecordHelper {
154         record R1(int x) { }
155 
156         static class Nested {
157             record R2(int x) { }
158         }
159 
160         Class<?> m() {
161             record R4(int x) { }
162             return R4.class;
163         }
164 
165         Class<?> m2() {
166             Supplier<Class<?>> s = () -> {
167                 record R5(int x) { }
168                 return R5.class;
169             };
170             return s.get();
171         }
172 
173         static Class<?> m3() {
174             record R6(int x) { }
175             return R6.class;
176         }
177 
178         static Class<?> m4() {
179             Supplier<Class<?>> s = () -> {
180                 record R7(int x) { }
181                 return R7.class;
182             };
183             return s.get();
184         }
185     }
186 
187     public void testNestedRecordsStatic() {
188         NestedRecordHelper n = new NestedRecordHelper();
189         for (Class<?> c : List.of(NestedRecordHelper.R1.class,
190                                   NestedRecordHelper.Nested.R2.class,
191                                   n.m(),
192                                   n.m2(),
193                                   NestedRecordHelper.m3(),
194                                   NestedRecordHelper.m4())) {
195             String message = c.toGenericString();
196             assertTrue(c.isRecord(), message);
197             assertTrue((c.getModifiers() & Modifier.STATIC) != 0, message);
198             assertTrue((c.getModifiers() & Modifier.FINAL) != 0, message);
199         }
200     }
