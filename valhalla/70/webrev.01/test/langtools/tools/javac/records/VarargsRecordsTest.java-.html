 26 import java.lang.reflect.Constructor;
 27 import java.lang.reflect.Field;
 28 import java.lang.reflect.Method;
 29 import java.lang.reflect.Modifier;
 30 import java.lang.reflect.Parameter;
 31 
 32 import org.testng.annotations.Test;
 33 import static org.testng.Assert.*;
 34 
 35 /**
 36  * VarargsRecordsTest
 37  *
 38  * @test
 39  * @compile --enable-preview -source ${jdk.version} VarargsRecordsTest.java
 40  * @run testng/othervm --enable-preview VarargsRecordsTest
 41  */
 42 @Test
 43 public class VarargsRecordsTest {
 44     record RI(int... xs) { }
 45     record RII(int x, int... xs) { }
 46     record RX(int[] xs) { }
 47 
 48     RI r1 = new RI();
 49     RI r2 = new RI(1);
 50     RI r3 = new RI(1, 2);
 51     RII r4 = new RII(1);
 52     RII r5 = new RII(1, 2);
 53     RII r6 = new RII(1, 2, 3);
 54 
 55     public void assertVarargsInstances() {
 56         assertEquals(r1.xs.length, 0);
 57         assertEquals(r2.xs.length, 1);
 58         assertEquals(r3.xs.length, 2);
 59         assertEquals(r4.xs.length, 0);
 60         assertEquals(r5.xs.length, 1);
 61         assertEquals(r6.xs.length, 2);
 62 
 63         assertEquals(r2.xs[0], 1);
 64         assertEquals(r3.xs[0], 1);
 65         assertEquals(r3.xs[1], 2);
 66 
 67         assertEquals(r5.xs[0], 2);
 68         assertEquals(r6.xs[0], 2);
 69         assertEquals(r6.xs[1], 3);
 70     }
 71 
 72     public void testMembers() throws ReflectiveOperationException {
 73         Constructor c = RI.class.getConstructor(int[].class);
 74         assertNotNull(c);
 75         assertTrue(c.isVarArgs());
 76         Parameter[] parameters = c.getParameters();
 77         assertEquals(parameters.length, 1);
 78         assertEquals(parameters[0].getName(), "xs");
 79 
 80         RI ri = (RI) c.newInstance(new int[]{1, 2});
 81         assertEquals(ri.xs()[0], 1);
 82         assertEquals(ri.xs()[1], 2);
 83 
 84         Field xsField = RI.class.getDeclaredField("xs");
 85         assertEquals(xsField.getType(), int[].class);
 86         assertEquals((xsField.getModifiers() & Modifier.STATIC), 0);
 87         assertTrue((xsField.getModifiers() & Modifier.PRIVATE) != 0);
 88         assertTrue((xsField.getModifiers() & Modifier.FINAL) != 0);
 89         assertEquals(((int[]) xsField.get(ri))[0], 1);
 90 
 91         Method xsMethod = RI.class.getDeclaredMethod("xs");
 92         assertEquals(xsMethod.getReturnType(), int[].class);
 93         assertEquals(xsMethod.getParameterCount(), 0);
 94         assertEquals((xsMethod.getModifiers() & (Modifier.PRIVATE | Modifier.PROTECTED | Modifier.STATIC | Modifier.ABSTRACT)), 0);
 95         assertEquals(((int[]) xsMethod.invoke(ri))[0], 1);
 96     }
 97 
 98     public void testNotVarargs() throws ReflectiveOperationException {
 99         Constructor c = RX.class.getConstructor(int[].class);
100         assertFalse(c.isVarArgs());
101     }
102 }
