 24 /**
 25  * @test
 26  * @bug 8207954
 27  * @summary Verify that CreateSymbols can handle classfiles from the current release.
 28  * @library /tools/lib /tools/javac/lib
 29  * @modules jdk.compiler/com.sun.tools.javac.api:+open
 30  *          jdk.compiler/com.sun.tools.javac.main
 31  *          jdk.compiler/com.sun.tools.javac.jvm:+open
 32  *          jdk.compiler/com.sun.tools.javac.util:+open
 33  *          jdk.jdeps/com.sun.tools.classfile:+open
 34  * @build toolbox.ToolBox toolbox.JavacTask toolbox.Task
 35  * @run main CanHandleClassFilesTest
 36  */
 37 
 38 import java.io.Writer;
 39 import java.nio.file.Files;
 40 import java.nio.file.Path;
 41 import java.nio.file.Paths;
 42 import java.util.stream.Stream;
 43 import javax.lang.model.SourceVersion;
 44 
 45 import javax.tools.StandardLocation;
 46 
 47 import toolbox.JavacTask;
 48 import toolbox.ToolBox;
 49 
 50 public class CanHandleClassFilesTest {
 51 
 52     public static void main(String... args) throws Exception {
 53         new CanHandleClassFilesTest().run();
 54     }
 55 
 56     void run() throws Exception {
 57         var testSrc = Paths.get(System.getProperty("test.src", "."));
 58         var targetDir = Paths.get(".");
 59         Path createSymbols = null;
 60         Path includeList = null;
 61         for (Path d = testSrc; d != null; d = d.getParent()) {
 62             if (Files.exists(d.resolve("TEST.ROOT"))) {
 63                 d = d.getParent().getParent();
 64                 Path test = d.resolve("make/langtools/src/classes/build/tools/symbolgenerator/CreateSymbols.java");
 65                 if (Files.exists(test)) {
 66                     createSymbols = test;
 67                     includeList = d.resolve("make/data/symbols/include.list");
 68                     break;
 69                 }
 70             }
 71         }
 72         if (createSymbols == null || includeList == null || !Files.isReadable(includeList)) {
 73             System.err.println("Warning: sources not found, test skipped.");
 74             return ;
 75         }
 76         try (ToolBox.MemoryFileManager mfm = new ToolBox.MemoryFileManager()) {
 77             ToolBox tb = new ToolBox();
 78             new JavacTask(tb)
 79               .options("--add-exports", "jdk.jdeps/com.sun.tools.classfile=ALL-UNNAMED",
 80                        "--add-exports", "jdk.compiler/com.sun.tools.javac.api=ALL-UNNAMED",
 81                        "--add-exports", "jdk.compiler/com.sun.tools.javac.jvm=ALL-UNNAMED",
 82                        "--add-exports", "jdk.compiler/com.sun.tools.javac.util=ALL-UNNAMED",
 83                        "--add-modules", "jdk.jdeps")
 84               .files(createSymbols)
 85               .fileManager(mfm)
 86               .run()
 87               .writeAll();
 88 
 89             ClassLoader cl = new ClassLoader() {
 90                 @Override
 91                 protected Class<?> findClass(String name) throws ClassNotFoundException {
 92                     byte[] data = mfm.getFileBytes(StandardLocation.CLASS_OUTPUT, name);
 93                     if (data != null) {
 94                         return defineClass(name, data, 0, data.length);
 95                     } else {
 96                         throw new ClassNotFoundException(name);
 97                     }
 98                 }
 99             };
100 
101             // open the non-exported packages needed by CreateSymbols to its module
102             Module targetModule = cl.getUnnamedModule();
103             Stream.of("jdk.compiler/com.sun.tools.javac.api",
104                       "jdk.compiler/com.sun.tools.javac.jvm",
105                       "jdk.compiler/com.sun.tools.javac.util",
106                       "jdk.jdeps/com.sun.tools.classfile")
107                     .forEach(p -> open(p, targetModule));
108 
109             var createSymbolsClass = Class.forName("build.tools.symbolgenerator.CreateSymbols", false, cl);
110             var main = createSymbolsClass.getMethod("main", String[].class);
111             var symbols = targetDir.resolve("symbols");
112             var systemModules = targetDir.resolve("system-modules");
113 
114             try (Writer w = Files.newBufferedWriter(symbols)) {}
115             try (Writer w = Files.newBufferedWriter(systemModules)) {}
116 
117             main.invoke(null,
118                         (Object) new String[] {"build-description-incremental",
119                                                symbols.toAbsolutePath().toString(),
120                                                includeList.toAbsolutePath().toString()});
121 
122             main.invoke(null,
123                         (Object) new String[] {"build-ctsym",
124                                                "does-not-exist",
125                                                symbols.toAbsolutePath().toString(),
126                                                targetDir.resolve("ct.sym").toAbsolutePath().toString(),
127                                                Long.toString(System.currentTimeMillis() / 1000),
128                                                "" + SourceVersion.latest().ordinal(),
129                                                systemModules.toAbsolutePath().toString()});
130         }
131     }
132 
133     void open(String moduleAndPackage, Module target) {
134         String[] s = moduleAndPackage.split("/");
135         var moduleName = s[0];
136         var packageName = s[1];
137         ModuleLayer.boot().findModule(moduleName).orElseThrow().addOpens(packageName, target);
138     }
139 
140 }
