34  * @run driver compiler.aot.AotCompiler -libname libTestHeapBase.so

36  *     -compile compiler.aot.TestHeapBase.test()V

38  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseAOT
39  *     -XX:+UseCompressedOops -XX:HeapBaseMinAddress=32g
40  *     -XX:AOTLibrary=./libTestHeapBase.so -Xbootclasspath/a:.
41  *     -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xlog:aot+class+load=trace
42  *     compiler.aot.TestHeapBase
43  * @summary check for crash when jaotc is run with zero-based compressed oops then
44  *          generated code is loaded in vm with non-zero-based compressed oops.
45  */
46 
47 package compiler.aot;
48 
49 import java.lang.reflect.Method;
50 import jdk.test.lib.Asserts;
51 import jdk.test.lib.Utils;
52 import sun.hotspot.WhiteBox;
53 import compiler.whitebox.CompilerWhiteBoxTest;
54 
55 public final class TestHeapBase {
56     private static final String TEST_METHOD = "test";
57     private static final WhiteBox WB = WhiteBox.getWhiteBox();
58     private final Method testMethod;
59 
60     private TestHeapBase() {
61         try {
62             testMethod = getClass().getDeclaredMethod(TEST_METHOD);
63         } catch (NoSuchMethodException e) {
64             throw new Error("TEST BUG: no test method found", e);
65         }
66     }
67 
68     public static void main(String args[]) {
69         new TestHeapBase().test();
70     }
71 
72     private void test() {
73         System.out.println("Hello, World!");
74 
75         Asserts.assertTrue(WB.isMethodCompiled(testMethod),
76                 "Method expected to be compiled");
77         Asserts.assertEQ(WB.getMethodCompilationLevel(testMethod),
78                 CompilerWhiteBoxTest.COMP_LEVEL_AOT,
79                 "Expected method to be AOT compiled");
80     }
81 }
