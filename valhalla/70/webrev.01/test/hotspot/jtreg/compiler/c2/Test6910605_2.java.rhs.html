24 /**
25  * @test
26  * @bug 6910605
27  * @summary C2: NullPointerException/ClassCaseException is thrown when C2 with DeoptimizeALot is used
28  *
29  * @run main/othervm -Xmx128m -XX:+IgnoreUnrecognizedVMOptions -XX:+DeoptimizeALot
30  *      -XX:+DoEscapeAnalysis -Xbatch -XX:+IgnoreUnrecognizedVMOptions -XX:InlineSmallCode=2000
31  *      compiler.c2.Test6910605_2
32  */
33 
34 package compiler.c2;
35 
36 /*
37  * Added InlineSmallCode=2000 to guarantee inlining of StringBuilder::append() to allow scalar replace StringBuilder object.
38  *
39  * original test: gc/gctests/StringGC
40  */
41 
42 public class Test6910605_2 {
43     private final String toAdd = "0123456789abcdef";
44     private int maxLength;
45     private static final int numberOfThreads = 8;
46 
47     private class StringAdder extends Thread {
48         private String s;
49 
50         public void test() {
51             s = s + toAdd;
52         }
53 
54         public void run() {
55             do {
56                 test();
57             } while (s.length() < maxLength);
58         }
59     }
60 
61     public void test() throws InterruptedException {
62         maxLength = toAdd.length() * 15000 / numberOfThreads;
63         StringAdder[] sa = new StringAdder[numberOfThreads];
64         for (int i = 0; i < numberOfThreads; i++) {
65             sa[i] = new StringAdder();
66             sa[i].start();
67         }
68         for (int i = 0; i < numberOfThreads; i++) {
69             sa[i].join();
70         }
71     }
72 
73     public static void main(String[] args) throws InterruptedException {
74         Test6910605_2 t = new Test6910605_2();
75         t.test();
76     }
77 }
