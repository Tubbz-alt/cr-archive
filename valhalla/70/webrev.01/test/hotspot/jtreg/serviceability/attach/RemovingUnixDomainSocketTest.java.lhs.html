24 /*
25  * @test
26  * @bug 8225193
27  * @requires os.family != "windows"
28  * @library /test/lib
29  * @run main RemovingUnixDomainSocketTest
30  */
31 

32 import java.io.IOException;
33 import java.nio.file.Path;

34 
35 import jdk.test.lib.Utils;
36 import jdk.test.lib.apps.LingeredApp;
37 import jdk.test.lib.JDKToolLauncher;
38 import jdk.test.lib.process.OutputAnalyzer;

39 
40 public class RemovingUnixDomainSocketTest {
41 



42     private static void runJCmd(long pid) throws InterruptedException, IOException {
43         JDKToolLauncher jcmd = JDKToolLauncher.createUsingTestJDK("jcmd");
44         jcmd.addVMArgs(Utils.getTestJavaOpts());
45         jcmd.addToolArg(Long.toString(pid));
46         jcmd.addToolArg("VM.version");
47 
48         ProcessBuilder pb = new ProcessBuilder(jcmd.getCommand());
49         Process jcmdProc = pb.start();
50 
51         OutputAnalyzer out = new OutputAnalyzer(jcmdProc);
52 
53         jcmdProc.waitFor();




54 
55         System.out.println(out.getStdout());
56         System.err.println(out.getStderr());

57 
58         out.stderrShouldBeEmptyIgnoreVMWarnings();

59     }
60 
61     public static void main(String... args) throws Exception {
62         LingeredApp app = null;
63         try {
64             app = LingeredApp.startApp();
65 
66             // Access to Attach Listener
67             runJCmd(app.getPid());
68 
69             // Remove unix domain socket file
70             var sockFile = Path.of(System.getProperty("java.io.tmpdir"),
71                                    ".java_pid" + app.getPid())
72                                .toFile();
73             System.out.println("Remove " + sockFile.toString());
74             sockFile.delete();
75 
76             // Access to Attach Listener again
77             runJCmd(app.getPid());
78         } finally {
79             LingeredApp.stopApp(app);
80         }
81     }
82 



83 }
