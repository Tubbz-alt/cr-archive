 24 /**
 25  * @test
 26  * @bug 8241071
 27  * @summary The same JDK build should always generate the same archive file (no randomness).
 28  * @requires vm.cds
 29  * @library /test/lib
 30  * @run driver DeterministicDump
 31  */
 32 
 33 import jdk.test.lib.cds.CDSTestUtils;
 34 import jdk.test.lib.Platform;
 35 import jdk.test.lib.process.ProcessTools;
 36 import jdk.test.lib.process.OutputAnalyzer;
 37 import java.io.FileInputStream;
 38 import java.io.IOException;
 39 
 40 public class DeterministicDump {
 41     public static void main(String[] args) throws Exception {
 42         for (int c = 0; c < 2; c++) { //  oop/klass compression
 43             String sign = (c == 0) ?  "+" : "-";
 44             String coop = "-XX:" + sign + "UseCompressedOops";
 45             String ckls = "-XX:" + sign + "UseCompressedClassPointers";
 46 
 47             if (!Platform.is64bit()) {
 48                 coop = "-showversion"; // no-op
 49                 ckls = "-showversion"; // no-op
 50             }
 51 
 52             for (int gc = 0; gc < 2; gc++) { // should we trigger GC during dump
 53                 for (int i = 0; i < 2; i++) {
 54                     String metaspaceSize = "-showversion"; // no-op
 55                     if (gc == 1 && i == 1) {
 56                         // This will cause GC to happen after we've allocated 1MB of metaspace objects
 57                         // while processing the built-in SharedClassListFile.
 58                         metaspaceSize = "-XX:MetaspaceSize=1M";
 59                     }
 60                     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
 61                                       coop, ckls, metaspaceSize,
 62                                       "-XX:SharedArchiveFile=SharedArchiveFile" + i + ".jsa",
 63                                       "-Xshare:dump", "-Xlog:cds=debug");
 64                     OutputAnalyzer out = CDSTestUtils.executeAndLog(pb, "SharedArchiveFile" + i);
 65                     CDSTestUtils.checkDump(out);
 66                 }
 67                 compare("SharedArchiveFile0.jsa", "SharedArchiveFile1.jsa");
 68             }
 69         }
 70     }
 71 
 72     static void compare(String file0, String file1) throws Exception {
 73         byte[] buff0 = new byte[4096];
 74         byte[] buff1 = new byte[4096];
 75         try (FileInputStream in0 = new FileInputStream(file0);
 76              FileInputStream in1 = new FileInputStream(file1)) {
 77             int total = 0;
 78             while (true) {
 79                 int n0 = read(in0, buff0);
 80                 int n1 = read(in1, buff1);
 81                 if (n0 != n1) {
 82                     throw new RuntimeException("File contents (file sizes?) are different after " + total + " bytes; n0 = "
 83                                                + n0 + ", n1 = " + n1);
 84                 }
 85                 if (n0 == 0) {
 86                     System.out.println("File contents are the same: " + total + " bytes");
 87                     break;
 88                 }
 89                 for (int i = 0; i < n0; i++) {
 90                     byte b0 = buff0[i];
 91                     byte b1 = buff1[i];
 92                     if (b0 != b1) {
 93                         throw new RuntimeException("File content different at byte #" + (total + i) + ", b0 = " + b0 + ", b1 = " + b1);
 94                     }
 95                 }
 96                 total += n0;
 97             }
 98         }
 99     }
100 
101     static int read(FileInputStream in, byte[] buff) throws IOException {
102         int total = 0;
103         while (total < buff.length) {
104             int n = in.read(buff, total, buff.length - total);
105             if (n <= 0) {
106                 return total;
107             }
108             total += n;
109         }
110 
111         return total;
112     }
113 }
