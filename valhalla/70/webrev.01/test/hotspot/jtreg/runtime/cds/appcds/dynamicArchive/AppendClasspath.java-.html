25 /*
26  * @test
27  * @summary At run time, it is OK to append new elements to the classpath that was used at dump time.
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  * @compile ../test-classes/Hello.java
31  * @compile ../test-classes/HelloMore.java
32  * @run driver AppendClasspath
33  */
34 
35 import java.io.File;
36 
37 public class AppendClasspath extends DynamicArchiveTestBase {
38 
39     public static void main(String[] args) throws Exception {
40         runTest(AppendClasspath::testDefaultBase);
41     }
42 
43     static void testDefaultBase() throws Exception {
44         String topArchiveName = getNewArchiveName("top");
45         doTest(topArchiveName);
46     }
47 
48     private static void doTest(String topArchiveName) throws Exception {
49         String appJar = JarBuilder.getOrCreateHelloJar();
50         String appJar2 = JarBuilder.build("AppendClasspath_HelloMore", "HelloMore");
51 
52         // Dump an archive with a specified JAR file in -classpath
53         dump(topArchiveName,
54              "-Xlog:cds",
55              "-Xlog:cds+dynamic=debug",
56              "-cp", appJar, "Hello")
57             .assertNormalExit(output -> {
58                     output.shouldContain("Buffer-space to target-space delta")
59                            .shouldContain("Written dynamic archive 0x");
60                 });
61 
62         // runtime with classpath containing the one used in dump time,
63         // i.e. the dump time classpath is a prefix of the runtime classpath.
64         run(topArchiveName,
65             "-Xlog:class+load",
66             "-Xlog:cds+dynamic=debug,cds=debug",
67             "-cp", appJar + File.pathSeparator + appJar2,
68             "HelloMore")
69             .assertNormalExit(output -> {
70                     output.shouldContain("Hello source: shared objects file")
71                           .shouldContain("Hello World ... More")
72                           .shouldHaveExitValue(0);
73                 });
74 
75         // reverse the order of the 2 jar files so that the dump time classpath
76         // is no longer a prefix of the runtime classpath. The Hello class
77         // should be loaded from the jar file.
78         run(topArchiveName,
79             "-Xlog:class+load",
80             "-Xlog:cds+dynamic=debug,cds=debug",
81             "-cp", appJar2 + File.pathSeparator + appJar,
82             "HelloMore")
83             .assertAbnormalExit(output -> {
84                     output.shouldContain("shared class paths mismatch")
85                           .shouldHaveExitValue(1);
86                 });
87 
88     }
89 }
