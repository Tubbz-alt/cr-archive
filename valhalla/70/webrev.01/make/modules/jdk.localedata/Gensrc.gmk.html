36 CLDR_DATA_DIR := $(TOPDIR)/make/data/cldr/common
37 GENSRC_DIR := $(SUPPORT_OUTPUTDIR)/gensrc/jdk.localedata
38 CLDR_GEN_DONE := $(GENSRC_DIR)/_cldr-gensrc.marker
39 
40 $(CLDR_GEN_DONE): $(wildcard $(CLDR_DATA_DIR)/dtd/*.dtd) \
41     $(wildcard $(CLDR_DATA_DIR)/main/*.xml) \
42     $(wildcard $(CLDR_DATA_DIR)/supplemental/*.xml) \
43     $(BUILD_TOOLS_JDK)
44 	$(call MakeDir, $(GENSRC_DIR))
45 	$(call LogInfo, Processing CLDR data for jdk.localedata)
46 	$(call ExecuteWithLog, $@, \
47 	    $(TOOL_CLDRCONVERTER) -base $(CLDR_DATA_DIR) \
48 	    -baselocales "en-US" \
49 	    -o $(GENSRC_DIR))
50 	$(TOUCH) $@
51 
52 TARGETS += $(CLDR_GEN_DONE)
