26 GENSRC_DIR := $(SUPPORT_OUTPUTDIR)/gensrc/$(MODULE)
27 SRC_DIR := $(TOPDIR)/src/$(MODULE)/share/classes
28 
29 ################################################################################
30 
31 PROC_SRC_SUBDIRS := \
32     org.graalvm.compiler.hotspot.management \
33     #
34 
35 PROC_SRC_DIRS := $(patsubst %, $(SRC_DIR)/%/src, $(PROC_SRC_SUBDIRS))
36 
37 PROC_SRCS := $(filter %.java, $(call FindFiles, $(PROC_SRC_DIRS)))
38 
39 ALL_SRC_DIRS := $(SRC_DIR) $(wildcard $(SRC_DIR)/*/src)
40 SOURCEPATH := $(call PathList, $(ALL_SRC_DIRS))
41 
42 PROCESSOR_JARS := \
43     $(BUILDTOOLS_OUTPUTDIR)/jdk.vm.compiler.serviceprovider.processor.jar \
44     #
45 PROCESSOR_PATH := $(call PathList, $(PROCESSOR_JARS))
46 
47 $(GENSRC_DIR)/_gensrc_proc_done: $(PROC_SRCS) $(PROCESSOR_JARS)
48 	$(call MakeDir, $(@D))
49 	$(eval $(call ListPathsSafely,PROC_SRCS,$(@D)/_gensrc_proc_files))
50 	$(JAVA) $(NEW_JAVAC) \
51 	    -XDignore.symbol.file \
52 	    --upgrade-module-path $(JDK_OUTPUTDIR)/modules --system none \
53 	    -sourcepath $(SOURCEPATH) \
54 	    -implicit:none \
55 	    -proc:only \
56 	    -processorpath $(PROCESSOR_PATH) \
57 	    -d $(GENSRC_DIR) \
58 	    -s $(GENSRC_DIR) \
59 	    @$(@D)/_gensrc_proc_files
60 	$(TOUCH) $@
61 
62 TARGETS += $(GENSRC_DIR)/_gensrc_proc_done
63 
64 ################################################################################
65 
66 $(GENSRC_DIR)/module-info.java.extra: $(GENSRC_DIR)/_gensrc_proc_done
67 	$(ECHO) "" > $@;
68 	($(CD) $(GENSRC_DIR)/META-INF/providers && \
69 	    p=""; \
70 	    impl=""; \
71 	    for i in $$($(NAWK) '$$0=FILENAME" "$$0' * | $(SORT) -k 2 | $(SED) 's/ .*//'); do \
72 	      c=$$($(CAT) $$i | $(TR) -d '\n\r'); \
73 	      if test x$$p != x$$c; then \
74                 if test x$$p != x; then \
75 	          $(ECHO) "    ;" >> $@; \
76 	        fi; \
77 	        $(ECHO) "provides $$c with" >> $@; \
78                 p=$$c; \
79 	        impl=""; \
80 	      fi; \
81               if test x$$impl != x; then \
82 	        $(ECHO) "  , $$i" >> $@; \
83               else \
84 	        $(ECHO) "    $$i" >> $@; \
85               fi; \
86               impl=$$i; \
87 	    done); \
88 	$(ECHO) "    ;" >> $@;
89 
90 TARGETS += $(GENSRC_DIR)/module-info.java.extra
