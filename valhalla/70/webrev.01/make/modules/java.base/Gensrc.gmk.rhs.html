 45 CLDR_DATA_DIR := $(TOPDIR)/make/data/cldr/common
 46 GENSRC_DIR := $(SUPPORT_OUTPUTDIR)/gensrc/java.base
 47 CLDR_GEN_DONE := $(GENSRC_DIR)/_cldr-gensrc.marker
 48 TZ_DATA_DIR := $(TOPDIR)/make/data/tzdata
 49 ZONENAME_TEMPLATE := $(TOPDIR)/src/java.base/share/classes/java/time/format/ZoneName.java.template
 50 
 51 $(CLDR_GEN_DONE): $(wildcard $(CLDR_DATA_DIR)/dtd/*.dtd) \
 52     $(wildcard $(CLDR_DATA_DIR)/main/en*.xml) \
 53     $(wildcard $(CLDR_DATA_DIR)/supplemental/*.xml) \
 54     $(wildcard $(TZ_DATA_DIR)/*) \
 55     $(ZONENAME_TEMPLATE) \
 56     $(BUILD_TOOLS_JDK)
 57 	$(call MakeDir, $(GENSRC_DIR))
 58 	$(call LogInfo, Processing CLDR data for java.base)
 59 	$(call ExecuteWithLog, $@, \
 60 	    $(TOOL_CLDRCONVERTER) -base $(CLDR_DATA_DIR) \
 61 	    -baselocales "en-US" \
 62 	    -o $(GENSRC_DIR) \
 63 	    -basemodule \
 64 	    -zntempfile $(ZONENAME_TEMPLATE) \
 65 	    -tzdatadir $(TZ_DATA_DIR))
 66 	$(TOUCH) $@
 67 
 68 TARGETS += $(CLDR_GEN_DONE)
