28 ################################################################################
29 # Translate the Java debugger wire protocol (jdwp.spec) file into a JDWP.java file
30 # and a JDWPCommands.h C-header file.
31 
32 JDWP_SPEC_FILE := $(TOPDIR)/make/data/jdwp/jdwp.spec
33 HEADER_FILE := $(SUPPORT_OUTPUTDIR)/headers/jdk.jdwp.agent/JDWPCommands.h
34 JAVA_FILE := $(SUPPORT_OUTPUTDIR)/gensrc/jdk.jdi/com/sun/tools/jdi/JDWP.java
35 HTML_FILE := $(SUPPORT_OUTPUTDIR)/gensrc/jdk.jdi/jdwp-protocol.html
36 
37 # Both the header and java file are created using the same recipe. By declaring
38 # this rule and adding header file to dependencies for java file, both are
39 # rebuilt if either is missing
40 $(HEADER_FILE): $(JDWP_SPEC_FILE) $(BUILD_TOOLS_JDK)
41 
42 # Touch the target of this rule at the end to avoid triggering false rebuilds
43 $(JAVA_FILE): $(JDWP_SPEC_FILE) $(BUILD_TOOLS_JDK) $(HEADER_FILE)
44 	$(call LogInfo, Creating JDWP.java and JDWPCommands.h from jdwp.spec)
45 	$(call MakeDir, $(@D) $(SUPPORT_OUTPUTDIR)/headers/jdk.jdwp.agent)
46 	$(RM) $@ $(SUPPORT_OUTPUTDIR)/headers/jdk.jdwp.agent/JDWPCommands.h
47 	$(TOOL_JDWPGEN) $< -jdi $@ -include \
48 	    $(SUPPORT_OUTPUTDIR)/headers/jdk.jdwp.agent/JDWPCommands.h
49 	$(TOUCH) $@
50 
51 $(HTML_FILE): $(JDWP_SPEC_FILE) \
52     $(BUILD_TOOLS_JDK)
53 	$(call LogInfo, Creating $(@F) from jdwp.spec)
54 	$(call MakeDir, $(@D))
55 	$(RM) $@
56 	$(TOOL_JDWPGEN) $< -doc $@
57 
58 TARGETS += $(JAVA_FILE) $(HEADER_FILE) $(HTML_FILE)
