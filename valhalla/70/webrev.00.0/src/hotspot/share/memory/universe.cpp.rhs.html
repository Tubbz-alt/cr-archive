 277 void Universe::check_alignment(uintx size, uintx alignment, const char* name) {
 278   if (size < alignment || size % alignment != 0) {
 279     vm_exit_during_initialization(
 280       err_msg("Size of %s (" UINTX_FORMAT " bytes) must be aligned to " UINTX_FORMAT " bytes", name, size, alignment));

 284 void initialize_basic_type_klass(Klass* k, TRAPS) {
 285   Klass* ok = SystemDictionary::Object_klass();
 286 #if INCLUDE_CDS
 287   if (UseSharedSpaces) {
 288     ClassLoaderData* loader_data = ClassLoaderData::the_null_class_loader_data();
 289     assert(k->super() == ok, "u3");
 290     if (k->is_instance_klass()) {
 291       InstanceKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), NULL, CHECK);
 292     } else {
 293       ArrayKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), CHECK);
 294     }
 295   } else
 296 #endif
 297   {
 298     k->initialize_supers(ok, NULL, CHECK);
 299   }
 300   k->append_to_sibling_list();
 301 }

 303 void Universe::genesis(TRAPS) {
 304   ResourceMark rm(THREAD);
 305 
 306   { FlagSetting fs(_bootstrapping, true);
 307 
 308     { MutexLocker mc(THREAD, Compile_lock);
 309 
 310       java_lang_Class::allocate_fixup_lists();
 311 
 312       // determine base vtable size; without that we cannot create the array klasses
 313       compute_base_vtable_size();
 314 
 315       if (!UseSharedSpaces) {
 316         for (int i = T_BOOLEAN; i < T_LONG+1; i++) {
 317           _typeArrayKlassObjs[i] = TypeArrayKlass::create_klass((BasicType)i, CHECK);
 318         }
 319 
 320         ClassLoaderData* null_cld = ClassLoaderData::the_null_class_loader_data();
 321 
 322         _the_array_interfaces_array     = MetadataFactory::new_array<Klass*>(null_cld, 3, NULL, CHECK);
 323         _the_empty_int_array            = MetadataFactory::new_array<int>(null_cld, 0, CHECK);
 324         _the_empty_short_array          = MetadataFactory::new_array<u2>(null_cld, 0, CHECK);
 325         _the_empty_method_array         = MetadataFactory::new_array<Method*>(null_cld, 0, CHECK);
 326         _the_empty_klass_array          = MetadataFactory::new_array<Klass*>(null_cld, 0, CHECK);
 327         _the_empty_instance_klass_array = MetadataFactory::new_array<InstanceKlass*>(null_cld, 0, CHECK);
 328       }
 329     }
 330 
 331     vmSymbols::initialize(CHECK);
 332 
 333     SystemDictionary::initialize(CHECK);
 334 
 335     Klass* ok = SystemDictionary::Object_klass();
 336 
 337     _the_null_string            = StringTable::intern("null", CHECK);
 338     _the_min_jint_string       = StringTable::intern("-2147483648", CHECK);
 339 
 340 #if INCLUDE_CDS
 341     if (UseSharedSpaces) {
 342       // Verify shared interfaces array.
 343       assert(_the_array_interfaces_array->at(0) ==
 344              SystemDictionary::Cloneable_klass(), "u3");
 345       assert(_the_array_interfaces_array->at(1) ==
 346              SystemDictionary::Serializable_klass(), "u3");
 347       assert(_the_array_interfaces_array->at(2) ==
 348                    SystemDictionary::IdentityObject_klass(), "u3");
 349 
 350       assert(_the_single_IdentityObject_klass_array->at(0) ==
 351           SystemDictionary::IdentityObject_klass(), "u3");
 352     } else
 353 #endif
 354     {
 355       // Set up shared interfaces array.  (Do this before supers are set up.)
 356       _the_array_interfaces_array->at_put(0, SystemDictionary::Cloneable_klass());
 357       _the_array_interfaces_array->at_put(1, SystemDictionary::Serializable_klass());
 358       _the_array_interfaces_array->at_put(2, SystemDictionary::IdentityObject_klass());
 359     }
 360 
 361     initialize_basic_type_klass(boolArrayKlassObj(), CHECK);
 362     initialize_basic_type_klass(charArrayKlassObj(), CHECK);
 363     initialize_basic_type_klass(floatArrayKlassObj(), CHECK);
 364     initialize_basic_type_klass(doubleArrayKlassObj(), CHECK);
 365     initialize_basic_type_klass(byteArrayKlassObj(), CHECK);
 366     initialize_basic_type_klass(shortArrayKlassObj(), CHECK);
 367     initialize_basic_type_klass(intArrayKlassObj(), CHECK);
 368     initialize_basic_type_klass(longArrayKlassObj(), CHECK);
 369   } // end of core bootstrapping
 370 
 371   {
 372     Handle tns = java_lang_String::create_from_str("<null_sentinel>", CHECK);
 373     _the_null_sentinel = tns();
 374   }
 375 
 376   // Maybe this could be lifted up now that object array can be initialized
 377   // during the bootstrapping.
 378 
 379   // OLD
 380   // Initialize _objectArrayKlass after core bootstraping to make
 381   // sure the super class is set up properly for _objectArrayKlass.
 382   // ---
 383   // NEW
 384   // Since some of the old system object arrays have been converted to
 385   // ordinary object arrays, _objectArrayKlass will be loaded when
 386   // SystemDictionary::initialize(CHECK); is run. See the extra check
 387   // for Object_klass_loaded in objArrayKlassKlass::allocate_objArray_klass_impl.
 388   _objectArrayKlassObj = InstanceKlass::
 389     cast(SystemDictionary::Object_klass())->array_klass(1, CHECK);
 390   // OLD
 391   // Add the class to the class hierarchy manually to make sure that
 392   // its vtable is initialized after core bootstrapping is completed.
 393   // ---
 394   // New
 395   // Have already been initialized.
 396   _objectArrayKlassObj->append_to_sibling_list();
 397 
 398   #ifdef ASSERT
 399   if (FullGCALot) {
 400     // Allocate an array of dummy objects.
 401     // We'd like these to be at the bottom of the old generation,
 402     // so that when we free one and then collect,
 403     // (almost) the whole heap moves
 404     // and we find out if we actually update all the oops correctly.
 405     // But we can't allocate directly in the old generation,
 406     // so we allocate wherever, and hope that the first collection
 407     // moves these objects to the bottom of the old generation.
 408     // We can allocate directly in the permanent generation, so we do.
 409     int size = FullGCALotDummies * 2;
 410 
 411     objArrayOop    naked_array = oopFactory::new_objArray(SystemDictionary::Object_klass(), size, CHECK);
 412     objArrayHandle dummy_array(THREAD, naked_array);
 413     int i = 0;
 414     while (i < size) {
 415         // Allocate dummy in old generation
 416       oop dummy = SystemDictionary::Object_klass()->allocate_instance(CHECK);
 417       dummy_array->obj_at_put(i++, dummy);
 418     }
 419     {
 420       // Only modify the global variable inside the mutex.
 421       // If we had a race to here, the other dummy_array instances
 422       // and their elements just get dropped on the floor, which is fine.
 423       MutexLocker ml(THREAD, FullGCALot_lock);
 424       if (_fullgc_alot_dummy_array == NULL) {
 425         _fullgc_alot_dummy_array = dummy_array();
 426       }
 427     }
 428     assert(i == _fullgc_alot_dummy_array->length(), "just checking");
 429   }
 430   #endif
 431 }

 488 void Universe::fixup_mirrors(TRAPS) {
 489   // Bootstrap problem: all classes gets a mirror (java.lang.Class instance) assigned eagerly,
 490   // but we cannot do that for classes created before java.lang.Class is loaded. Here we simply
 491   // walk over permanent objects created so far (mostly classes) and fixup their mirrors. Note
 492   // that the number of objects allocated at this point is very small.
 493   assert(SystemDictionary::Class_klass_loaded(), "java.lang.Class should be loaded");
 494   HandleMark hm(THREAD);
 495 
 496   if (!UseSharedSpaces) {
 497     // Cache the start of the static fields
 498     InstanceMirrorKlass::init_offset_of_static_fields();
 499   }
 500 
 501   GrowableArray <Klass*>* list = java_lang_Class::fixup_mirror_list();
 502   int list_length = list->length();
 503   for (int i = 0; i < list_length; i++) {
 504     Klass* k = list->at(i);
 505     assert(k->is_klass(), "List should only hold classes");
 506     EXCEPTION_MARK;
 507     java_lang_Class::fixup_mirror(k, CATCH);
 508   }
 509   delete java_lang_Class::fixup_mirror_list();
 510   java_lang_Class::set_fixup_mirror_list(NULL);
 511 }

 582 bool Universe::should_fill_in_stack_trace(Handle throwable) {
 583   // never attempt to fill in the stack trace of preallocated errors that do not have
 584   // backtrace. These errors are kept alive forever and may be "re-used" when all
 585   // preallocated errors with backtrace have been consumed. Also need to avoid
 586   // a potential loop which could happen if an out of memory occurs when attempting
 587   // to allocate the backtrace.
 588   return ((throwable() != Universe::_out_of_memory_error_java_heap) &&
 589           (throwable() != Universe::_out_of_memory_error_metaspace)  &&
 590           (throwable() != Universe::_out_of_memory_error_class_metaspace)  &&
 591           (throwable() != Universe::_out_of_memory_error_array_size) &&
 592           (throwable() != Universe::_out_of_memory_error_gc_overhead_limit) &&
 593           (throwable() != Universe::_out_of_memory_error_realloc_objects) &&
 594           (throwable() != Universe::_out_of_memory_error_retry));
 595 }

 597 
 598 oop Universe::gen_out_of_memory_error(oop default_err) {
 599   // generate an out of memory error:
 600   // - if there is a preallocated error and stack traces are available
 601   //   (j.l.Throwable is initialized), then return the preallocated
 602   //   error with a filled in stack trace, and with the message
 603   //   provided by the default error.
 604   // - otherwise, return the default error, without a stack trace.
 605   int next;
 606   if ((_preallocated_out_of_memory_error_avail_count > 0) &&
 607       SystemDictionary::Throwable_klass()->is_initialized()) {
 608     next = (int)Atomic::add(&_preallocated_out_of_memory_error_avail_count, -1);
 609     assert(next < (int)PreallocatedOutOfMemoryErrorCount, "avail count is corrupt");
 610   } else {
 611     next = -1;
 612   }
 613   if (next < 0) {
 614     // all preallocated errors have been used.
 615     // return default
 616     return default_err;
 617   } else {
 618     Thread* THREAD = Thread::current();
 619     Handle default_err_h(THREAD, default_err);
 620     // get the error object at the slot and set set it to NULL so that the
 621     // array isn't keeping it alive anymore.
 622     Handle exc(THREAD, preallocated_out_of_memory_errors()->obj_at(next));
 623     assert(exc() != NULL, "slot has been used already");
 624     preallocated_out_of_memory_errors()->obj_at_put(next, NULL);
 625 
 626     // use the message from the default error
 627     oop msg = java_lang_Throwable::message(default_err_h());
 628     assert(msg != NULL, "no message");
 629     java_lang_Throwable::set_message(exc(), msg);
 630 
 631     // populate the stack trace and return it.
 632     java_lang_Throwable::fill_in_stack_trace_of_preallocated_backtrace(exc);
 633     return exc();
 634   }
 635 }

 637 intptr_t Universe::_non_oop_bits = 0;
 638 
 639 void* Universe::non_oop_word() {
 640   // Neither the high bits nor the low bits of this value is allowed
 641   // to look like (respectively) the high or low bits of a real oop.
 642   //
 643   // High and low are CPU-specific notions, but low always includes
 644   // the low-order bit.  Since oops are always aligned at least mod 4,
 645   // setting the low-order bit will ensure that the low half of the
 646   // word will never look like that of a real oop.
 647   //
 648   // Using the OS-supplied non-memory-address word (usually 0 or -1)
 649   // will take care of the high bits, however many there are.
 650 
 651   if (_non_oop_bits == 0) {
 652     _non_oop_bits = (intptr_t)os::non_memory_address_word() | 1;
 653   }
 654 
 655   return (void*)_non_oop_bits;
 656 }

 760 ReservedHeapSpace Universe::reserve_heap(size_t heap_size, size_t alignment) {
 761 
 762   assert(alignment <= Arguments::conservative_max_heap_alignment(),
 763          "actual alignment " SIZE_FORMAT " must be within maximum heap alignment " SIZE_FORMAT,
 764          alignment, Arguments::conservative_max_heap_alignment());
 765 
 766   size_t total_reserved = align_up(heap_size, alignment);
 767   assert(!UseCompressedOops || (total_reserved <= (OopEncodingHeapMax - os::vm_page_size())),
 768       "heap size is too big for compressed oops");
 769 
 770   bool use_large_pages = UseLargePages && is_aligned(alignment, os::large_page_size());
 771   assert(!UseLargePages
 772       || UseParallelGC
 773       || use_large_pages, "Wrong alignment to use large pages");
 774 
 775   // Now create the space.
 776   ReservedHeapSpace total_rs(total_reserved, alignment, use_large_pages, AllocateHeapAt);
 777 
 778   if (total_rs.is_reserved()) {
 779     assert((total_reserved == total_rs.size()) && ((uintptr_t)total_rs.base() % alignment == 0),
 780            "must be exactly of required size and alignment");
 781     // We are good.
 782 
 783     if (AllocateHeapAt != NULL) {
 784       log_info(gc,heap)("Successfully allocated Java heap at location %s", AllocateHeapAt);
 785     }
 786 
 787     if (UseCompressedOops) {
 788       CompressedOops::initialize(total_rs);
 789     }
 790 
 791     Universe::calculate_verify_data((HeapWord*)total_rs.base(), (HeapWord*)total_rs.end());
 792 
 793     return total_rs;
 794   }
 795 
 796   vm_exit_during_initialization(
 797     err_msg("Could not reserve enough space for " SIZE_FORMAT "KB object heap",
 798             total_reserved/K));
 799 
 800   // satisfy compiler
 801   ShouldNotReachHere();
 802   return ReservedHeapSpace(0, 0, false);
 803 }

 896   Klass* k = SystemDictionary::OutOfMemoryError_klass();

 899   Universe::_out_of_memory_error_metaspace = ik->allocate_instance(CHECK_false);
 900   Universe::_out_of_memory_error_class_metaspace = ik->allocate_instance(CHECK_false);
 901   Universe::_out_of_memory_error_array_size = ik->allocate_instance(CHECK_false);
 902   Universe::_out_of_memory_error_gc_overhead_limit =
 903     ik->allocate_instance(CHECK_false);
 904   Universe::_out_of_memory_error_realloc_objects = ik->allocate_instance(CHECK_false);
 905   Universe::_out_of_memory_error_retry = ik->allocate_instance(CHECK_false);
 906 
 907   // Setup preallocated cause message for delayed StackOverflowError
 908   if (StackReservedPages > 0) {
 909     Universe::_delayed_stack_overflow_error_message =
 910       java_lang_String::create_oop_from_str("Delayed StackOverflowError due to ReservedStackAccess annotated method", CHECK_false);
 911   }
 912 
 913   // Setup preallocated NullPointerException
 914   // (this is currently used for a cheap & dirty solution in compiler exception handling)
 915   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_NullPointerException(), true, CHECK_false);
 916   Universe::_null_ptr_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 917   // Setup preallocated ArithmeticException
 918   // (this is currently used for a cheap & dirty solution in compiler exception handling)
 919   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_ArithmeticException(), true, CHECK_false);
 920   Universe::_arithmetic_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 921   // Virtual Machine Error for when we get into a situation we can't resolve
 922   k = SystemDictionary::VirtualMachineError_klass();
 923   bool linked = InstanceKlass::cast(k)->link_class_or_fail(CHECK_false);
 924   if (!linked) {
 925      tty->print_cr("Unable to link/verify VirtualMachineError class");
 926      return false; // initialization failed
 927   }
 928   Universe::_virtual_machine_error_instance =
 929     InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 930 
 931   Universe::_vm_exception = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 932 
 933   Handle msg = java_lang_String::create_from_str("Java heap space", CHECK_false);
 934   java_lang_Throwable::set_message(Universe::_out_of_memory_error_java_heap, msg());
 935 
 936   msg = java_lang_String::create_from_str("Metaspace", CHECK_false);
 937   java_lang_Throwable::set_message(Universe::_out_of_memory_error_metaspace, msg());
 938   msg = java_lang_String::create_from_str("Compressed class space", CHECK_false);
 939   java_lang_Throwable::set_message(Universe::_out_of_memory_error_class_metaspace, msg());
 940 
 941   msg = java_lang_String::create_from_str("Requested array size exceeds VM limit", CHECK_false);
 942   java_lang_Throwable::set_message(Universe::_out_of_memory_error_array_size, msg());
 943 
 944   msg = java_lang_String::create_from_str("GC overhead limit exceeded", CHECK_false);
 945   java_lang_Throwable::set_message(Universe::_out_of_memory_error_gc_overhead_limit, msg());
 946 
 947   msg = java_lang_String::create_from_str("Java heap space: failed reallocation of scalar replaced objects", CHECK_false);
 948   java_lang_Throwable::set_message(Universe::_out_of_memory_error_realloc_objects, msg());
 949 
 950   msg = java_lang_String::create_from_str("Java heap space: failed retryable allocation", CHECK_false);
 951   java_lang_Throwable::set_message(Universe::_out_of_memory_error_retry, msg());
 952 
 953   msg = java_lang_String::create_from_str("/ by zero", CHECK_false);
 954   java_lang_Throwable::set_message(Universe::_arithmetic_exception_instance, msg());
 955 
 956   // Setup the array of errors that have preallocated backtrace
 957   k = Universe::_out_of_memory_error_java_heap->klass();
 958   assert(k->name() == vmSymbols::java_lang_OutOfMemoryError(), "should be out of memory error");
 959   ik = InstanceKlass::cast(k);
 960 
 961   int len = (StackTraceInThrowable) ? (int)PreallocatedOutOfMemoryErrorCount : 0;
 962   Universe::_preallocated_out_of_memory_error_array = oopFactory::new_objArray(ik, len, CHECK_false);
 963   for (int i=0; i<len; i++) {
 964     oop err = ik->allocate_instance(CHECK_false);
 965     Handle err_h = Handle(THREAD, err);
 966     java_lang_Throwable::allocate_backtrace(err_h, CHECK_false);
 967     Universe::preallocated_out_of_memory_errors()->obj_at_put(i, err_h());
 968   }
 969   Universe::_preallocated_out_of_memory_error_avail_count = (jint)len;
 970 
 971   Universe::initialize_known_methods(CHECK_false);
 972 
 973   // This needs to be done before the first scavenge/gc, since
 974   // it's an input to soft ref clearing policy.
 975   {
 976     MutexLocker x(THREAD, Heap_lock);
 977     Universe::update_heap_info_at_gc();
 978   }
 979 
 980   // ("weak") refs processing infrastructure initialization
 981   Universe::heap()->post_initialize();
 982 
 983   MemoryService::add_metaspace_memory_pools();
 984 
 985   MemoryService::set_universe_heap(Universe::heap());
 986 #if INCLUDE_CDS
 987   MetaspaceShared::post_initialize(CHECK_false);
 988 #endif
 989   return true;
 990 }

1031 void Universe::initialize_verify_flags() {
1032   verify_flags = 0;
1033   const char delimiter[] = " ,";
1034 
1035   size_t length = strlen(VerifySubSet);
1036   char* subset_list = NEW_C_HEAP_ARRAY(char, length + 1, mtInternal);
1037   strncpy(subset_list, VerifySubSet, length + 1);
1038   char* save_ptr;
1039 
1040   char* token = strtok_r(subset_list, delimiter, &save_ptr);
1041   while (token != NULL) {
1042     if (strcmp(token, "threads") == 0) {
1043       verify_flags |= Verify_Threads;
1044     } else if (strcmp(token, "heap") == 0) {
1045       verify_flags |= Verify_Heap;
1046     } else if (strcmp(token, "symbol_table") == 0) {
1047       verify_flags |= Verify_SymbolTable;
1048     } else if (strcmp(token, "string_table") == 0) {
1049       verify_flags |= Verify_StringTable;
1050     } else if (strcmp(token, "codecache") == 0) {
1051       verify_flags |= Verify_CodeCache;
1052     } else if (strcmp(token, "dictionary") == 0) {
1053       verify_flags |= Verify_SystemDictionary;
1054     } else if (strcmp(token, "classloader_data_graph") == 0) {
1055       verify_flags |= Verify_ClassLoaderDataGraph;
1056     } else if (strcmp(token, "metaspace") == 0) {
1057       verify_flags |= Verify_MetaspaceUtils;
1058     } else if (strcmp(token, "jni_handles") == 0) {
1059       verify_flags |= Verify_JNIHandles;
1060     } else if (strcmp(token, "codecache_oops") == 0) {
1061       verify_flags |= Verify_CodeCacheOops;
1062     } else if (strcmp(token, "resolved_method_table") == 0) {
1063       verify_flags |= Verify_ResolvedMethodTable;
1064     } else {
1065       vm_exit_during_initialization(err_msg("VerifySubSet: \'%s\' memory sub-system is unknown, please correct it", token));
1066     }
1067     token = strtok_r(NULL, delimiter, &save_ptr);
1068   }
1069   FREE_C_HEAP_ARRAY(char, subset_list);
1070 }

1079 void Universe::verify(VerifyOption option, const char* prefix) {
1080   // The use of _verify_in_progress is a temporary work around for
1081   // 6320749.  Don't bother with a creating a class to set and clear
1082   // it since it is only used in this method and the control flow is
1083   // straight forward.
1084   _verify_in_progress = true;
1085 
1086   COMPILER2_PRESENT(
1087     assert(!DerivedPointerTable::is_active(),
1088          "DPT should not be active during verification "
1089          "(of thread stacks below)");
1090   )
1091 
1092   ResourceMark rm;
1093   HandleMark hm;  // Handles created during verification can be zapped
1094   _verify_count++;
1095 
1096   FormatBuffer<> title("Verifying %s", prefix);
1097   GCTraceTime(Info, gc, verify) tm(title.buffer());
1098   if (should_verify_subset(Verify_Threads)) {
1099     log_debug(gc, verify)("Threads");
1100     Threads::verify();
1101   }
1102   if (should_verify_subset(Verify_Heap)) {
1103     log_debug(gc, verify)("Heap");
1104     heap()->verify(option);
1105   }
1106   if (should_verify_subset(Verify_SymbolTable)) {
1107     log_debug(gc, verify)("SymbolTable");
1108     SymbolTable::verify();
1109   }
1110   if (should_verify_subset(Verify_StringTable)) {
1111     log_debug(gc, verify)("StringTable");
1112     StringTable::verify();
1113   }
1114   if (should_verify_subset(Verify_CodeCache)) {
1115   {
1116     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1117     log_debug(gc, verify)("CodeCache");
1118     CodeCache::verify();
1119   }
1120   }
1121   if (should_verify_subset(Verify_SystemDictionary)) {
1122     log_debug(gc, verify)("SystemDictionary");
1123     SystemDictionary::verify();
1124   }
1125 #ifndef PRODUCT
1126   if (should_verify_subset(Verify_ClassLoaderDataGraph)) {
1127     log_debug(gc, verify)("ClassLoaderDataGraph");
1128     ClassLoaderDataGraph::verify();
1129   }
1130 #endif
1131   if (should_verify_subset(Verify_MetaspaceUtils)) {
1132     log_debug(gc, verify)("MetaspaceUtils");
1133     MetaspaceUtils::verify_free_chunks();
1134   }
1135   if (should_verify_subset(Verify_JNIHandles)) {
1136     log_debug(gc, verify)("JNIHandles");
1137     JNIHandles::verify();
1138   }
1139   if (should_verify_subset(Verify_CodeCacheOops)) {
1140     log_debug(gc, verify)("CodeCache Oops");
1141     CodeCache::verify_oops();
1142   }
1143   if (should_verify_subset(Verify_ResolvedMethodTable)) {
1144     log_debug(gc, verify)("ResolvedMethodTable Oops");
1145     ResolvedMethodTable::verify();
1146   }
1147 
1148   _verify_in_progress = false;
1149 }

1153 void Universe::calculate_verify_data(HeapWord* low_boundary, HeapWord* high_boundary) {
1154   assert(low_boundary < high_boundary, "bad interval");
1155 
1156   // decide which low-order bits we require to be clear:
1157   size_t alignSize = MinObjAlignmentInBytes;
1158   size_t min_object_size = CollectedHeap::min_fill_size();
1159 
1160   // make an inclusive limit:
1161   uintptr_t max = (uintptr_t)high_boundary - min_object_size*wordSize;
1162   uintptr_t min = (uintptr_t)low_boundary;
1163   assert(min < max, "bad interval");
1164   uintptr_t diff = max ^ min;
1165 
1166   // throw away enough low-order bits to make the diff vanish
1167   uintptr_t mask = (uintptr_t)(-1);
1168   while ((mask & diff) != 0)
1169     mask <<= 1;
1170   uintptr_t bits = (min & mask);
1171   assert(bits == (max & mask), "correct mask");
1172   // check an intermediate value between min and max, just to make sure:
1173   assert(bits == ((min + (max-min)/2) & mask), "correct mask");
1174 
1175   // require address alignment, too:
1176   mask |= (alignSize - 1);
1177 
1178   if (!(_verify_oop_mask == 0 && _verify_oop_bits == (uintptr_t)-1)) {
1179     assert(_verify_oop_mask == mask && _verify_oop_bits == bits, "mask stability");
1180   }
1181   _verify_oop_mask = mask;
1182   _verify_oop_bits = bits;
1183 }
