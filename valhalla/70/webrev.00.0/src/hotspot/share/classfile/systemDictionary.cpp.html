 168 void SystemDictionary::compute_java_loaders(TRAPS) {
 169   JavaValue result(T_OBJECT);
 170   InstanceKlass* class_loader_klass = SystemDictionary::ClassLoader_klass();
 171   JavaCalls::call_static(&result,
 172                          class_loader_klass,
 173                          vmSymbols::getSystemClassLoader_name(),
 174                          vmSymbols::void_classloader_signature(),
 175                          CHECK);
 176 
 177   _java_system_loader = (oop)result.get_jobject();
 178 
 179   JavaCalls::call_static(&result,
 180                          class_loader_klass,
 181                          vmSymbols::getPlatformClassLoader_name(),
 182                          vmSymbols::void_classloader_signature(),
 183                          CHECK);
 184 
 185   _java_platform_loader = (oop)result.get_jobject();
 186 }

 240 Klass* SystemDictionary::resolve_or_fail(Symbol* class_name, Handle class_loader, Handle protection_domain, bool throw_error, TRAPS) {
 241   Klass* klass = resolve_or_null(class_name, class_loader, protection_domain, THREAD);
 242   if (HAS_PENDING_EXCEPTION || klass == NULL) {
 243     // can return a null klass
 244     klass = handle_resolution_exception(class_name, throw_error, klass, THREAD);
 245   }
 246   return klass;
 247 }

 249 Klass* SystemDictionary::handle_resolution_exception(Symbol* class_name,
 250                                                      bool throw_error,
 251                                                      Klass* klass, TRAPS) {
 252   if (HAS_PENDING_EXCEPTION) {
 253     // If we have a pending exception we forward it to the caller, unless throw_error is true,
 254     // in which case we have to check whether the pending exception is a ClassNotFoundException,
 255     // and if so convert it to a NoClassDefFoundError
 256     // And chain the original ClassNotFoundException
 257     if (throw_error && PENDING_EXCEPTION->is_a(SystemDictionary::ClassNotFoundException_klass())) {
 258       ResourceMark rm(THREAD);
 259       assert(klass == NULL, "Should not have result with exception pending");
 260       Handle e(THREAD, PENDING_EXCEPTION);
 261       CLEAR_PENDING_EXCEPTION;
 262       THROW_MSG_CAUSE_NULL(vmSymbols::java_lang_NoClassDefFoundError(), class_name->as_C_string(), e);
 263     } else {
 264       return NULL;
 265     }
 266   }
 267   // Class not found, throw appropriate error or exception depending on value of throw_error
 268   if (klass == NULL) {
 269     ResourceMark rm(THREAD);
 270     if (throw_error) {
 271       THROW_MSG_NULL(vmSymbols::java_lang_NoClassDefFoundError(), class_name->as_C_string());
 272     } else {
 273       THROW_MSG_NULL(vmSymbols::java_lang_ClassNotFoundException(), class_name->as_C_string());
 274     }
 275   }
 276   return klass;
 277 }

 297 // name may be in the form of "java/lang/Object" or "Ljava/lang/Object;"
 298 InstanceKlass* SystemDictionary::resolve_instance_class_or_null_helper(Symbol* class_name,
 299                                                                        Handle class_loader,
 300                                                                        Handle protection_domain,
 301                                                                        TRAPS) {
 302   assert(class_name != NULL && !Signature::is_array(class_name), "must be");
 303   if (Signature::has_envelope(class_name)) {
 304     ResourceMark rm(THREAD);
 305     // Ignore wrapping L and ;. (and Q and ; for value types);
 306     TempNewSymbol name = SymbolTable::new_symbol(class_name->as_C_string() + 1,
 307                                                  class_name->utf8_length() - 2);
 308     return resolve_instance_class_or_null(name, class_loader, protection_domain, THREAD);
 309   } else {
 310     return resolve_instance_class_or_null(class_name, class_loader, protection_domain, THREAD);
 311   }
 312 }

 318 // Forwards to resolve_instance_class_or_null
 319 
 320 Klass* SystemDictionary::resolve_array_class_or_null(Symbol* class_name,
 321                                                      Handle class_loader,
 322                                                      Handle protection_domain,
 323                                                      TRAPS) {
 324   assert(Signature::is_array(class_name), "must be array");
 325   ResourceMark rm(THREAD);
 326   SignatureStream ss(class_name, false);
 327   int ndims = ss.skip_array_prefix();  // skip all '['s
 328   Klass* k = NULL;
 329   BasicType t = ss.type();
 330   if (ss.has_envelope()) {
 331     Symbol* obj_class = ss.as_symbol();
 332     k = SystemDictionary::resolve_instance_class_or_null(obj_class,
 333                                                          class_loader,
 334                                                          protection_domain,
 335                                                          CHECK_NULL);
 336     if (k != NULL) {
 337       k = k->array_klass(ndims, CHECK_NULL);
 338     }
 339   } else {
 340     k = Universe::typeArrayKlassObj(t);
 341     k = TypeArrayKlass::cast(k)->array_klass(ndims, CHECK_NULL);
 342   }
 343   return k;
 344 }

 346 // Must be called for any super-class or super-interface resolution
 347 // during class definition to allow class circularity checking
 348 // super-interface callers:
 349 //    parse_interfaces - for defineClass & jvmtiRedefineClasses
 350 // super-class callers:
 351 //   ClassFileParser - for defineClass & jvmtiRedefineClasses
 352 //   load_shared_class - while loading a class from shared archive
 353 //   resolve_instance_class_or_null:
 354 //     via: handle_parallel_super_load
 355 //      when resolving a class that has an existing placeholder with
 356 //      a saved superclass [i.e. a defineClass is currently in progress]
 357 //      if another thread is trying to resolve the class, it must do
 358 //      super-class checks on its own thread to catch class circularity
 359 // This last call is critical in class circularity checking for cases
 360 // where classloading is delegated to different threads and the
 361 // classloader lock is released.
 362 // Take the case: Base->Super->Base
 363 //   1. If thread T1 tries to do a defineClass of class Base
 364 //    resolve_super_or_fail creates placeholder: T1, Base (super Super)
 365 //   2. resolve_instance_class_or_null does not find SD or placeholder for Super
 366 //    so it tries to load Super
 367 //   3. If we load the class internally, or user classloader uses same thread
 368 //      loadClassFromxxx or defineClass via parseClassFile Super ...
 369 //      3.1 resolve_super_or_fail creates placeholder: T1, Super (super Base)
 370 //      3.3 resolve_instance_class_or_null Base, finds placeholder for Base
 371 //      3.4 calls resolve_super_or_fail Base
 372 //      3.5 finds T1,Base -> throws class circularity
 373 //OR 4. If T2 tries to resolve Super via defineClass Super ...
 374 //      4.1 resolve_super_or_fail creates placeholder: T2, Super (super Base)
 375 //      4.2 resolve_instance_class_or_null Base, finds placeholder for Base (super Super)
 376 //      4.3 calls resolve_super_or_fail Super in parallel on own thread T2
 377 //      4.4 finds T2, Super -> throws class circularity
 378 // Must be called, even if superclass is null, since this is
 379 // where the placeholder entry is created which claims this
 380 // thread is loading this class/classloader.
 381 // Be careful when modifying this code: once you have run
 382 // placeholders()->find_and_add(PlaceholderTable::LOAD_SUPER),
 383 // you need to find_and_remove it before returning.
 384 // So be careful to not exit with a CHECK_ macro betweeen these calls.
 385 InstanceKlass* SystemDictionary::resolve_super_or_fail(Symbol* child_name,
 386                                                        Symbol* super_name,
 387                                                        Handle class_loader,
 388                                                        Handle protection_domain,
 389                                                        bool is_superclass,
 390                                                        TRAPS) {
 391   assert(!Signature::is_array(super_name), "invalid super class name");
 392 #if INCLUDE_CDS
 393   if (DumpSharedSpaces) {
 394     // Special processing for handling UNREGISTERED shared classes.
 395     InstanceKlass* k = SystemDictionaryShared::dump_time_resolve_super_or_fail(child_name,
 396         super_name, class_loader, protection_domain, is_superclass, CHECK_NULL);
 397     if (k) {
 398       return k;
 399     }
 400   }
 401 #endif // INCLUDE_CDS
 402 
 403   // Double-check, if child class is already loaded, just return super-class,interface
 404   // Don't add a placedholder if already loaded, i.e. already in appropriate class loader
 405   // dictionary.
 406   // Make sure there's a placeholder for the *child* before resolving.
 407   // Used as a claim that this thread is currently loading superclass/classloader
 408   // Used here for ClassCircularity checks and also for heap verification
 409   // (every InstanceKlass needs to be in its class loader dictionary or have a placeholder).
 410   // Must check ClassCircularity before checking if super class is already loaded.
 411   //
 412   // We might not already have a placeholder if this child_name was
 413   // first seen via resolve_from_stream (jni_DefineClass or JVM_DefineClass);
 414   // the name of the class might not be known until the stream is actually
 415   // parsed.
 416   // Bugs 4643874, 4715493
 417 
 418   ClassLoaderData* loader_data = class_loader_data(class_loader);
 419   Dictionary* dictionary = loader_data->dictionary();
 420   unsigned int d_hash = dictionary->compute_hash(child_name);
 421   unsigned int p_hash = placeholders()->compute_hash(child_name);
 422   int p_index = placeholders()->hash_to_index(p_hash);
 423   // can't throw error holding a lock
 424   bool child_already_loaded = false;
 425   bool throw_circularity_error = false;
 426   {
 427     MutexLocker mu(THREAD, SystemDictionary_lock);
 428     InstanceKlass* childk = find_class(d_hash, child_name, dictionary);
 429     InstanceKlass* quicksuperk;
 430     // to support // loading: if child done loading, just return superclass
 431     // if super_name, & class_loader don't match:
 432     // if initial define, SD update will give LinkageError
 433     // if redefine: compare_class_versions will give HIERARCHY_CHANGED
 434     // so we don't throw an exception here.
 435     // see: nsk redefclass014 & java.lang.instrument Instrument032
 436     if ((childk != NULL ) && (is_superclass) &&
 437         ((quicksuperk = childk->java_super()) != NULL) &&
 438          ((quicksuperk->name() == super_name) &&
 439             (quicksuperk->class_loader() == class_loader()))) {
 440            return quicksuperk;
 441     } else {
 442       PlaceholderEntry* probe = placeholders()->get_entry(p_index, p_hash, child_name, loader_data);
 443       if (probe && probe->check_seen_thread(THREAD, PlaceholderTable::LOAD_SUPER)) {
 444           throw_circularity_error = true;
 445       }
 446     }
 447     if (!throw_circularity_error) {
 448       // Be careful not to exit resolve_super
 449       PlaceholderEntry* newprobe = placeholders()->find_and_add(p_index, p_hash, child_name, loader_data, PlaceholderTable::LOAD_SUPER, super_name, THREAD);
 450     }
 451   }
 452   if (throw_circularity_error) {
 453       ResourceMark rm(THREAD);
 454       THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), child_name->as_C_string());
 455   }
 456 
 457 // java.lang.Object should have been found above
 458   assert(super_name != NULL, "null super class for resolving");
 459   // Resolve the super class or interface, check results on return
 460   InstanceKlass* superk =
 461     SystemDictionary::resolve_instance_class_or_null_helper(super_name,
 462                                                             class_loader,
 463                                                             protection_domain,
 464                                                             THREAD);
 465 
 466   // Clean up of placeholders moved so that each classloadAction registrar self-cleans up
 467   // It is no longer necessary to keep the placeholder table alive until update_dictionary
 468   // or error. GC used to walk the placeholder table as strong roots.
 469   // The instanceKlass is kept alive because the class loader is on the stack,
 470   // which keeps the loader_data alive, as well as all instanceKlasses in
 471   // the loader_data. parseClassFile adds the instanceKlass to loader_data.
 472   {
 473     MutexLocker mu(THREAD, SystemDictionary_lock);
 474     placeholders()->find_and_remove(p_index, p_hash, child_name, loader_data, PlaceholderTable::LOAD_SUPER, THREAD);
 475     SystemDictionary_lock->notify_all();
 476   }
 477   if (HAS_PENDING_EXCEPTION || superk == NULL) {
 478     // can null superk
 479     Klass* k = handle_resolution_exception(super_name, true, superk, THREAD);
 480     assert(k == NULL || k == superk, "must be");
 481     if (k == NULL) {
 482       superk = NULL;
 483     }
 484   }
 485 
 486   return superk;
 487 }

 489 Klass* SystemDictionary::resolve_flattenable_field_or_fail(AllFieldStream* fs,
 490                                                            Handle class_loader,
 491                                                            Handle protection_domain,
 492                                                            bool throw_error,
 493                                                            TRAPS) {
 494   Symbol* class_name = fs->signature()->fundamental_name(THREAD);
 495   class_loader = Handle(THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 496   ClassLoaderData* loader_data = class_loader_data(class_loader);
 497   unsigned int p_hash = placeholders()->compute_hash(class_name);
 498   int p_index = placeholders()->hash_to_index(p_hash);
 499   bool throw_circularity_error = false;
 500   PlaceholderEntry* oldprobe;
 501 
 502   {
 503     MutexLocker mu(THREAD, SystemDictionary_lock);
 504     oldprobe = placeholders()->get_entry(p_index, p_hash, class_name, loader_data);
 505     if (oldprobe != NULL &&
 506       oldprobe->check_seen_thread(THREAD, PlaceholderTable::FLATTENABLE_FIELD)) {
 507       throw_circularity_error = true;
 508 
 509     } else {
 510       placeholders()->find_and_add(p_index, p_hash, class_name, loader_data,
 511                                    PlaceholderTable::FLATTENABLE_FIELD, NULL, THREAD);
 512     }
 513   }
 514 
 515   Klass* klass = NULL;
 516   if (!throw_circularity_error) {
 517     klass = SystemDictionary::resolve_or_fail(class_name, class_loader,
 518                                                protection_domain, true, THREAD);
 519   } else {
 520     ResourceMark rm(THREAD);
 521     THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), class_name->as_C_string());
 522   }
 523 
 524   {
 525     MutexLocker mu(THREAD, SystemDictionary_lock);
 526     placeholders()->find_and_remove(p_index, p_hash, class_name, loader_data,
 527                                     PlaceholderTable::FLATTENABLE_FIELD, THREAD);
 528   }
 529 
 530   class_name->decrement_refcount();
 531   return klass;
 532 }

 534 void SystemDictionary::validate_protection_domain(InstanceKlass* klass,
 535                                                   Handle class_loader,
 536                                                   Handle protection_domain,
 537                                                   TRAPS) {
 538   // Now we have to call back to java to check if the initating class has access
 539   JavaValue result(T_VOID);
 540   LogTarget(Debug, protectiondomain) lt;
 541   if (lt.is_enabled()) {
 542     ResourceMark rm(THREAD);
 543     // Print out trace information
 544     LogStream ls(lt);
 545     ls.print_cr("Checking package access");
 546     if (class_loader() != NULL) {
 547       ls.print("class loader: ");
 548       class_loader()->print_value_on(&ls);
 549     } else {
 550       ls.print_cr("class loader: NULL");
 551     }
 552     if (protection_domain() != NULL) {
 553       ls.print(" protection domain: ");
 554       protection_domain()->print_value_on(&ls);
 555     } else {
 556       ls.print_cr(" protection domain: NULL");
 557     }
 558     ls.print(" loading: "); klass->print_value_on(&ls);
 559     ls.cr();
 560   }
 561 
 562   // This handle and the class_loader handle passed in keeps this class from
 563   // being unloaded through several GC points.
 564   // The class_loader handle passed in is the initiating loader.
 565   Handle mirror(THREAD, klass->java_mirror());
 566 
 567   InstanceKlass* system_loader = SystemDictionary::ClassLoader_klass();
 568   JavaCalls::call_special(&result,
 569                          class_loader,
 570                          system_loader,
 571                          vmSymbols::checkPackageAccess_name(),
 572                          vmSymbols::class_protectiondomain_signature(),
 573                          mirror,
 574                          protection_domain,
 575                          THREAD);
 576

 601 // We only get here if this thread finds that another thread
 602 // has already claimed the placeholder token for the current operation,
 603 // but that other thread either never owned or gave up the
 604 // object lock
 605 // Waits on SystemDictionary_lock to indicate placeholder table updated
 606 // On return, caller must recheck placeholder table state
 607 //
 608 // We only get here if
 609 //  1) custom classLoader, i.e. not bootstrap classloader
 610 //  2) custom classLoader has broken the class loader objectLock
 611 //     so another thread got here in parallel
 612 //
 613 // lockObject must be held.
 614 // Complicated dance due to lock ordering:
 615 // Must first release the classloader object lock to
 616 // allow initial definer to complete the class definition
 617 // and to avoid deadlock
 618 // Reclaim classloader lock object with same original recursion count
 619 // Must release SystemDictionary_lock after notify, since
 620 // class loader lock must be claimed before SystemDictionary_lock
 621 // to prevent deadlocks
 622 //
 623 // The notify allows applications that did an untimed wait() on
 624 // the classloader object lock to not hang.
 625 void SystemDictionary::double_lock_wait(Handle lockObject, TRAPS) {
 626   assert_lock_strong(SystemDictionary_lock);
 627 
 628   bool calledholdinglock
 629       = ObjectSynchronizer::current_thread_holds_lock((JavaThread*)THREAD, lockObject);
 630   assert(calledholdinglock,"must hold lock for notify");
 631   assert((lockObject() != _system_loader_lock_obj && !is_parallelCapable(lockObject)), "unexpected double_lock_wait");
 632   ObjectSynchronizer::notifyall(lockObject, THREAD);
 633   intx recursions =  ObjectSynchronizer::complete_exit(lockObject, THREAD);
 634   SystemDictionary_lock->wait();
 635   SystemDictionary_lock->unlock();
 636   ObjectSynchronizer::reenter(lockObject, recursions, THREAD);
 637   SystemDictionary_lock->lock();
 638 }

 640 // If the class in is in the placeholder table, class loading is in progress
 641 // For cases where the application changes threads to load classes, it
 642 // is critical to ClassCircularity detection that we try loading
 643 // the superclass on the same thread internally, so we do parallel
 644 // super class loading here.
 645 // This also is critical in cases where the original thread gets stalled
 646 // even in non-circularity situations.
 647 // Note: must call resolve_super_or_fail even if null super -
 648 // to force placeholder entry creation for this class for circularity detection
 649 // Caller must check for pending exception
 650 // Returns non-null Klass* if other thread has completed load
 651 // and we are done,
 652 // If return null Klass* and no pending exception, the caller must load the class
 653 InstanceKlass* SystemDictionary::handle_parallel_super_load(
 654     Symbol* name, Symbol* superclassname, Handle class_loader,
 655     Handle protection_domain, Handle lockObject, TRAPS) {
 656 
 657   ClassLoaderData* loader_data = class_loader_data(class_loader);
 658   Dictionary* dictionary = loader_data->dictionary();
 659   unsigned int d_hash = dictionary->compute_hash(name);
 660   unsigned int p_hash = placeholders()->compute_hash(name);
 661   int p_index = placeholders()->hash_to_index(p_hash);
 662 
 663   // superk is not used, resolve_super called for circularity check only
 664   // This code is reached in two situations. One if this thread
 665   // is loading the same class twice (e.g. ClassCircularity, or
 666   // java.lang.instrument).
 667   // The second is if another thread started the resolve_super first
 668   // and has not yet finished.
 669   // In both cases the original caller will clean up the placeholder
 670   // entry on error.
 671   Klass* superk = SystemDictionary::resolve_super_or_fail(name,
 672                                                           superclassname,
 673                                                           class_loader,
 674                                                           protection_domain,
 675                                                           true,
 676                                                           CHECK_NULL);
 677 
 678   // parallelCapable class loaders do NOT wait for parallel superclass loads to complete
 679   // Serial class loaders and bootstrap classloader do wait for superclass loads
 680  if (!class_loader.is_null() && is_parallelCapable(class_loader)) {
 681     MutexLocker mu(THREAD, SystemDictionary_lock);
 682     // Check if classloading completed while we were loading superclass or waiting
 683     return find_class(d_hash, name, dictionary);
 684   }
 685 
 686   // must loop to both handle other placeholder updates
 687   // and spurious notifications
 688   bool super_load_in_progress = true;
 689   PlaceholderEntry* placeholder;
 690   while (super_load_in_progress) {
 691     MutexLocker mu(THREAD, SystemDictionary_lock);
 692     // Check if classloading completed while we were loading superclass or waiting
 693     InstanceKlass* check = find_class(d_hash, name, dictionary);
 694     if (check != NULL) {
 695       // Klass is already loaded, so just return it
 696       return check;
 697     } else {
 698       placeholder = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 699       if (placeholder && placeholder->super_load_in_progress() ){
 700         // We only get here if the application has released the
 701         // classloader lock when another thread was in the middle of loading a
 702         // superclass/superinterface for this class, and now
 703         // this thread is also trying to load this class.
 704         // To minimize surprises, the first thread that started to
 705         // load a class should be the one to complete the loading
 706         // with the classfile it initially expected.
 707         // This logic has the current thread wait once it has done
 708         // all the superclass/superinterface loading it can, until
 709         // the original thread completes the class loading or fails
 710         // If it completes we will use the resulting InstanceKlass
 711         // which we will find below in the systemDictionary.
 712         // We also get here for parallel bootstrap classloader
 713         if (class_loader.is_null()) {
 714           SystemDictionary_lock->wait();
 715         } else {
 716           double_lock_wait(lockObject, THREAD);
 717         }
 718       } else {
 719         // If not in SD and not in PH, other thread's load must have failed
 720         super_load_in_progress = false;
 721       }
 722     }
 723   }
 724   return NULL;
 725 }

 738 // Be careful when modifying this code: once you have run
 739 // placeholders()->find_and_add(PlaceholderTable::LOAD_INSTANCE),
 740 // you need to find_and_remove it before returning.
 741 // So be careful to not exit with a CHECK_ macro betweeen these calls.
 742 //
 743 // name must be in the form of "java/lang/Object" -- cannot be "Ljava/lang/Object;"
 744 InstanceKlass* SystemDictionary::resolve_instance_class_or_null(Symbol* name,
 745                                                                 Handle class_loader,
 746                                                                 Handle protection_domain,
 747                                                                 TRAPS) {
 748   assert(name != NULL && !Signature::is_array(name) &&
 749          !Signature::has_envelope(name), "invalid class name");
 750 
 751   EventClassLoad class_load_start_event;
 752 
 753   HandleMark hm(THREAD);
 754 
 755   // Fix for 4474172; see evaluation for more details
 756   class_loader = Handle(THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 757   ClassLoaderData* loader_data = register_loader(class_loader);
 758   Dictionary* dictionary = loader_data->dictionary();
 759   unsigned int d_hash = dictionary->compute_hash(name);
 760 
 761   // Do lookup to see if class already exist and the protection domain
 762   // has the right access
 763   // This call uses find which checks protection domain already matches
 764   // All subsequent calls use find_class, and set has_loaded_class so that
 765   // before we return a result we call out to java to check for valid protection domain
 766   // to allow returning the Klass* and add it to the pd_set if it is valid
 767   {
 768     InstanceKlass* probe = dictionary->find(d_hash, name, protection_domain);
 769     if (probe != NULL) return probe;
 770   }
 771 
 772   // Non-bootstrap class loaders will call out to class loader and
 773   // define via jvm/jni_DefineClass which will acquire the
 774   // class loader object lock to protect against multiple threads
 775   // defining the class in parallel by accident.
 776   // This lock must be acquired here so the waiter will find
 777   // any successful result in the SystemDictionary and not attempt
 778   // the define.
 779   // ParallelCapable Classloaders and the bootstrap classloader
 780   // do not acquire lock here.
 781   bool DoObjectLock = true;
 782   if (is_parallelCapable(class_loader)) {
 783     DoObjectLock = false;
 784   }
 785 
 786   unsigned int p_hash = placeholders()->compute_hash(name);
 787   int p_index = placeholders()->hash_to_index(p_hash);
 788 
 789   // Class is not in SystemDictionary so we have to do loading.
 790   // Make sure we are synchronized on the class loader before we proceed
 791   Handle lockObject = compute_loader_lock_object(class_loader, THREAD);
 792   check_loader_lock_contention(lockObject, THREAD);
 793   ObjectLocker ol(lockObject, THREAD, DoObjectLock);
 794 
 795   // Check again (after locking) if class already exist in SystemDictionary
 796   bool class_has_been_loaded   = false;
 797   bool super_load_in_progress  = false;
 798   bool havesupername = false;
 799   InstanceKlass* k = NULL;
 800   PlaceholderEntry* placeholder;
 801   Symbol* superclassname = NULL;
 802 
 803   assert(THREAD->can_call_java(),
 804          "can not load classes with compiler thread: class=%s, classloader=%s",
 805          name->as_C_string(),
 806          class_loader.is_null() ? "null" : class_loader->klass()->name()->as_C_string());
 807   {
 808     MutexLocker mu(THREAD, SystemDictionary_lock);
 809     InstanceKlass* check = find_class(d_hash, name, dictionary);
 810     if (check != NULL) {
 811       // InstanceKlass is already loaded, so just return it
 812       class_has_been_loaded = true;
 813       k = check;
 814     } else {
 815       placeholder = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 816       if (placeholder && placeholder->super_load_in_progress()) {
 817          super_load_in_progress = true;
 818          if (placeholder->havesupername() == true) {
 819            superclassname = placeholder->supername();
 820            havesupername = true;
 821          }
 822       }
 823     }
 824   }
 825 
 826   // If the class is in the placeholder table, class loading is in progress
 827   if (super_load_in_progress && havesupername==true) {
 828     k = handle_parallel_super_load(name,
 829                                    superclassname,
 830                                    class_loader,
 831                                    protection_domain,
 832                                    lockObject, THREAD);
 833     if (HAS_PENDING_EXCEPTION) {
 834       return NULL;
 835     }
 836     if (k != NULL) {
 837       class_has_been_loaded = true;
 838     }
 839   }
 840 
 841   bool throw_circularity_error = false;
 842   if (!class_has_been_loaded) {
 843     bool load_instance_added = false;
 844 
 845     // add placeholder entry to record loading instance class
 846     // Five cases:
 847     // All cases need to prevent modifying bootclasssearchpath
 848     // in parallel with a classload of same classname
 849     // Redefineclasses uses existence of the placeholder for the duration
 850     // of the class load to prevent concurrent redefinition of not completely
 851     // defined classes.
 852     // case 1. traditional classloaders that rely on the classloader object lock
 853     //   - no other need for LOAD_INSTANCE
 854     // case 2. traditional classloaders that break the classloader object lock
 855     //    as a deadlock workaround. Detection of this case requires that
 856     //    this check is done while holding the classloader object lock,
 857     //    and that lock is still held when calling classloader's loadClass.
 858     //    For these classloaders, we ensure that the first requestor
 859     //    completes the load and other requestors wait for completion.
 860     // case 3. Bootstrap classloader - don't own objectLocker
 861     //    This classloader supports parallelism at the classloader level,
 862     //    but only allows a single load of a class/classloader pair.
 863     //    No performance benefit and no deadlock issues.
 864     // case 4. parallelCapable user level classloaders - without objectLocker
 865     //    Allow parallel classloading of a class/classloader pair
 866 
 867     {
 868       MutexLocker mu(THREAD, SystemDictionary_lock);
 869       if (class_loader.is_null() || !is_parallelCapable(class_loader)) {
 870         PlaceholderEntry* oldprobe = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 871         if (oldprobe) {
 872           // only need check_seen_thread once, not on each loop
 873           // 6341374 java/lang/Instrument with -Xcomp
 874           if (oldprobe->check_seen_thread(THREAD, PlaceholderTable::LOAD_INSTANCE)) {
 875             throw_circularity_error = true;
 876           } else {
 877             // case 1: traditional: should never see load_in_progress.
 878             while (!class_has_been_loaded && oldprobe && oldprobe->instance_load_in_progress()) {
 879 
 880               // case 3: bootstrap classloader: prevent futile classloading,
 881               // wait on first requestor
 882               if (class_loader.is_null()) {
 883                 SystemDictionary_lock->wait();
 884               } else {
 885               // case 2: traditional with broken classloader lock. wait on first
 886               // requestor.
 887                 double_lock_wait(lockObject, THREAD);
 888               }
 889               // Check if classloading completed while we were waiting
 890               InstanceKlass* check = find_class(d_hash, name, dictionary);
 891               if (check != NULL) {
 892                 // Klass is already loaded, so just return it
 893                 k = check;
 894                 class_has_been_loaded = true;
 895               }
 896               // check if other thread failed to load and cleaned up
 897               oldprobe = placeholders()->get_entry(p_index, p_hash, name, loader_data);
 898             }
 899           }
 900         }
 901       }
 902       // All cases: add LOAD_INSTANCE holding SystemDictionary_lock
 903       // case 4: parallelCapable: allow competing threads to try
 904       // LOAD_INSTANCE in parallel
 905 
 906       if (!throw_circularity_error && !class_has_been_loaded) {
 907         PlaceholderEntry* newprobe = placeholders()->find_and_add(p_index, p_hash, name, loader_data, PlaceholderTable::LOAD_INSTANCE, NULL, THREAD);
 908         load_instance_added = true;
 909         // For class loaders that do not acquire the classloader object lock,
 910         // if they did not catch another thread holding LOAD_INSTANCE,
 911         // need a check analogous to the acquire ObjectLocker/find_class
 912         // i.e. now that we hold the LOAD_INSTANCE token on loading this class/CL
 913         // one final check if the load has already completed
 914         // class loaders holding the ObjectLock shouldn't find the class here
 915         InstanceKlass* check = find_class(d_hash, name, dictionary);
 916         if (check != NULL) {
 917           // Klass is already loaded, so return it after checking/adding protection domain
 918           k = check;
 919           class_has_been_loaded = true;
 920         }
 921       }
 922     }
 923 
 924     // must throw error outside of owning lock
 925     if (throw_circularity_error) {
 926       assert(!HAS_PENDING_EXCEPTION && load_instance_added == false,"circularity error cleanup");
 927       ResourceMark rm(THREAD);
 928       THROW_MSG_NULL(vmSymbols::java_lang_ClassCircularityError(), name->as_C_string());
 929     }
 930 
 931     if (!class_has_been_loaded) {
 932 
 933       // Do actual loading
 934       k = load_instance_class(name, class_loader, THREAD);
 935 
 936       // If everything was OK (no exceptions, no null return value), and
 937       // class_loader is NOT the defining loader, do a little more bookkeeping.
 938       if (!HAS_PENDING_EXCEPTION && k != NULL &&
 939         k->class_loader() != class_loader()) {
 940 
 941         check_constraints(d_hash, k, class_loader, false, THREAD);
 942 
 943         // Need to check for a PENDING_EXCEPTION again; check_constraints
 944         // can throw but we may have to remove entry from the placeholder table below.
 945         if (!HAS_PENDING_EXCEPTION) {
 946           // Record dependency for non-parent delegation.
 947           // This recording keeps the defining class loader of the klass (k) found
 948           // from being unloaded while the initiating class loader is loaded
 949           // even if the reference to the defining class loader is dropped
 950           // before references to the initiating class loader.
 951           loader_data->record_dependency(k);
 952 
 953           { // Grabbing the Compile_lock prevents systemDictionary updates
 954             // during compilations.
 955             MutexLocker mu(THREAD, Compile_lock);
 956             update_dictionary(d_hash, p_index, p_hash,
 957               k, class_loader, THREAD);
 958           }
 959 
 960           if (JvmtiExport::should_post_class_load()) {
 961             Thread *thread = THREAD;
 962             assert(thread->is_Java_thread(), "thread->is_Java_thread()");
 963             JvmtiExport::post_class_load((JavaThread *) thread, k);
 964           }
 965         }
 966       }
 967     } // load_instance_class
 968 
 969     if (load_instance_added == true) {
 970       // clean up placeholder entries for LOAD_INSTANCE success or error
 971       // This brackets the SystemDictionary updates for both defining
 972       // and initiating loaders
 973       MutexLocker mu(THREAD, SystemDictionary_lock);
 974       placeholders()->find_and_remove(p_index, p_hash, name, loader_data, PlaceholderTable::LOAD_INSTANCE, THREAD);
 975       SystemDictionary_lock->notify_all();
 976     }
 977   }
 978 
 979   if (HAS_PENDING_EXCEPTION || k == NULL) {
 980     return NULL;
 981   }
 982   if (class_load_start_event.should_commit()) {
 983     post_class_load_event(&class_load_start_event, k, loader_data);
 984   }
 985 #ifdef ASSERT
 986   {
 987     ClassLoaderData* loader_data = k->class_loader_data();
 988     MutexLocker mu(THREAD, SystemDictionary_lock);
 989     InstanceKlass* kk = find_class(name, loader_data);
 990     assert(kk == k, "should be present in dictionary");
 991   }
 992 #endif
 993 
 994   // return if the protection domain in NULL
 995   if (protection_domain() == NULL) return k;
 996 
 997   // Check the protection domain has the right access
 998   if (dictionary->is_valid_protection_domain(d_hash, name,
 999                                              protection_domain)) {
1000     return k;
1001   }
1002 
1003   // Verify protection domain. If it fails an exception is thrown
1004   validate_protection_domain(k, class_loader, protection_domain, CHECK_NULL);
1005 
1006   return k;
1007 }

1092   //                                           whose loader is the Lookup class's loader.

1099   } else {

1101   }
1102 
1103   assert(st != NULL, "invariant");
1104   assert(st->need_verify(), "invariant");

1107   // Note that we do this even though this klass might
1108   // already be present in the SystemDictionary, otherwise we would not
1109   // throw potential ClassFormatErrors.
1110

1112                                                       class_name,
1113                                                       loader_data,
1114                                                       cl_info,
1115                                                       CHECK_NULL);

1117   if ((cl_info.is_hidden() || is_unsafe_anon_class) && k != NULL) {
1118     // Hidden classes that are not strong and unsafe anonymous classes must update
1119     // ClassLoaderData holder so that they can be unloaded when the mirror is no
1120     // longer referenced.
1121     if (!cl_info.is_strong_hidden() || is_unsafe_anon_class) {
1122       k->class_loader_data()->initialize_holder(Handle(THREAD, k->java_mirror()));
1123     }
1124

1126       MutexLocker mu_r(THREAD, Compile_lock);
1127 
1128       // Add to class hierarchy, initialize vtables, and do possible
1129       // deoptimizations.
1130       add_to_hierarchy(k, CHECK_NULL); // No exception, but can block
1131       // But, do not add to dictionary.
1132     }
1133 
1134     // Rewrite and patch constant pool here.
1135     k->link_class(CHECK_NULL);
1136     if (cl_info.cp_patches() != NULL) {
1137       k->constants()->patch_resolved_references(cl_info.cp_patches());
1138     }

1141     if (is_unsafe_anon_class) {

1143     }

1517 InstanceKlass* SystemDictionary::load_instance_class(Symbol* class_name, Handle class_loader, TRAPS) {
1518 
1519   if (class_loader.is_null()) {
1520     ResourceMark rm(THREAD);
1521     PackageEntry* pkg_entry = NULL;
1522     bool search_only_bootloader_append = false;
1523     ClassLoaderData *loader_data = class_loader_data(class_loader);
1524 
1525     // Find the package in the boot loader's package entry table.
1526     TempNewSymbol pkg_name = ClassLoader::package_from_class_name(class_name);
1527     if (pkg_name != NULL) {
1528       pkg_entry = loader_data->packages()->lookup_only(pkg_name);
1529     }
1530 
1531     // Prior to attempting to load the class, enforce the boot loader's
1532     // visibility boundaries.
1533     if (!Universe::is_module_initialized()) {
1534       // During bootstrapping, prior to module initialization, any
1535       // class attempting to be loaded must be checked against the
1536       // java.base packages in the boot loader's PackageEntryTable.
1537       // No class outside of java.base is allowed to be loaded during
1538       // this bootstrapping window.
1539       if (pkg_entry == NULL || pkg_entry->in_unnamed_module()) {
1540         // Class is either in the unnamed package or in
1541         // a named package within the unnamed module.  Either
1542         // case is outside of java.base, do not attempt to
1543         // load the class post java.base definition.  If
1544         // java.base has not been defined, let the class load
1545         // and its package will be checked later by
1546         // ModuleEntryTable::verify_javabase_packages.
1547         if (ModuleEntryTable::javabase_defined()) {
1548           return NULL;
1549         }
1550       } else {
1551         // Check that the class' package is defined within java.base.
1552         ModuleEntry* mod_entry = pkg_entry->module();
1553         Symbol* mod_entry_name = mod_entry->name();
1554         if (mod_entry_name->fast_compare(vmSymbols::java_base()) != 0) {
1555           return NULL;
1556         }
1557       }
1558     } else {
1559       // After the module system has been initialized, check if the class'
1560       // package is in a module defined to the boot loader.
1561       if (pkg_name == NULL || pkg_entry == NULL || pkg_entry->in_unnamed_module()) {
1562         // Class is either in the unnamed package, in a named package
1563         // within a module not defined to the boot loader or in a
1564         // a named package within the unnamed module.  In all cases,
1565         // limit visibility to search for the class only in the boot
1566         // loader's append path.
1567         if (!ClassLoader::has_bootclasspath_append()) {
1568            // If there is no bootclasspath append entry, no need to continue
1569            // searching.
1570            return NULL;
1571         }
1572         search_only_bootloader_append = true;
1573       }
1574     }
1575 
1576     // Prior to bootstrapping's module initialization, never load a class outside
1577     // of the boot loader's module path
1578     assert(Universe::is_module_initialized() ||
1579            !search_only_bootloader_append,
1580            "Attempt to load a class outside of boot loader's module path");
1581 
1582     // Search for classes in the CDS archive.
1583     InstanceKlass* k = NULL;
1584     {
1585 #if INCLUDE_CDS
1586       PerfTraceTime vmtimer(ClassLoader::perf_shared_classload_time());
1587       k = load_shared_boot_class(class_name, pkg_entry, THREAD);
1588 #endif
1589     }
1590 
1591     if (k == NULL) {
1592       // Use VM class loader
1593       PerfTraceTime vmtimer(ClassLoader::perf_sys_classload_time());
1594       k = ClassLoader::load_class(class_name, search_only_bootloader_append, CHECK_NULL);
1595     }
1596 
1597     // find_or_define_instance_class may return a different InstanceKlass
1598     if (k != NULL) {
1599       InstanceKlass* defined_k =
1600         find_or_define_instance_class(class_name, class_loader, k, THREAD);
1601       if (!HAS_PENDING_EXCEPTION && defined_k != k) {
1602         // If a parallel capable class loader already defined this class, register 'k' for cleanup.
1603         assert(defined_k != NULL, "Should have a klass if there's no exception");
1604         loader_data->add_to_deallocate_list(k);
1605         k = defined_k;
1606       } else if (HAS_PENDING_EXCEPTION) {
1607         loader_data->add_to_deallocate_list(k);
1608         return NULL;
1609       }
1610     }
1611     return k;
1612   } else {
1613     // Use user specified class loader to load class. Call loadClass operation on class_loader.
1614     ResourceMark rm(THREAD);
1615 
1616     assert(THREAD->is_Java_thread(), "must be a JavaThread");
1617     JavaThread* jt = (JavaThread*) THREAD;
1618 
1619     PerfClassTraceTime vmtimer(ClassLoader::perf_app_classload_time(),
1620                                ClassLoader::perf_app_classload_selftime(),
1621                                ClassLoader::perf_app_classload_count(),
1622                                jt->get_thread_stat()->perf_recursion_counts_addr(),
1623                                jt->get_thread_stat()->perf_timers_addr(),
1624                                PerfClassTraceTime::CLASS_LOAD);
1625 
1626     Handle s = java_lang_String::create_from_symbol(class_name, CHECK_NULL);
1627     // Translate to external class name format, i.e., convert '/' chars to '.'
1628     Handle string = java_lang_String::externalize_classname(s, CHECK_NULL);
1629 
1630     JavaValue result(T_OBJECT);
1631 
1632     InstanceKlass* spec_klass = SystemDictionary::ClassLoader_klass();
1633 
1634     // Call public unsynchronized loadClass(String) directly for all class loaders.
1635     // For parallelCapable class loaders, JDK >=7, loadClass(String, boolean) will
1636     // acquire a class-name based lock rather than the class loader object lock.
1637     // JDK < 7 already acquire the class loader lock in loadClass(String, boolean).
1638     JavaCalls::call_virtual(&result,
1639                             class_loader,
1640                             spec_klass,
1641                             vmSymbols::loadClass_name(),
1642                             vmSymbols::string_class_signature(),
1643                             string,
1644                             CHECK_NULL);
1645 
1646     assert(result.get_type() == T_OBJECT, "just checking");
1647     oop obj = (oop) result.get_jobject();
1648 
1649     // Primitive classes return null since forName() can not be
1650     // used to obtain any of the Class objects representing primitives or void
1651     if ((obj != NULL) && !(java_lang_Class::is_primitive(obj))) {
1652       InstanceKlass* k = InstanceKlass::cast(java_lang_Class::as_Klass(obj));
1653       // For user defined Java class loaders, check that the name returned is
1654       // the same as that requested.  This check is done for the bootstrap
1655       // loader when parsing the class file.
1656       if (class_name == k->name()) {
1657         return k;
1658       }
1659     }
1660     // Class is not found or has the wrong name, return NULL
1661     return NULL;
1662   }

1674 void SystemDictionary::define_instance_class(InstanceKlass* k, TRAPS) {
1675 
1676   HandleMark hm(THREAD);
1677   ClassLoaderData* loader_data = k->class_loader_data();
1678   Handle class_loader_h(THREAD, loader_data->class_loader());
1679 
1680  // for bootstrap and other parallel classloaders don't acquire lock,
1681  // use placeholder token
1682  // If a parallelCapable class loader calls define_instance_class instead of
1683  // find_or_define_instance_class to get here, we have a timing
1684  // hole with systemDictionary updates and check_constraints
1685  if (!class_loader_h.is_null() && !is_parallelCapable(class_loader_h)) {
1686     assert(ObjectSynchronizer::current_thread_holds_lock((JavaThread*)THREAD,
1687          compute_loader_lock_object(class_loader_h, THREAD)),
1688          "define called without lock");
1689   }
1690 
1691   // Check class-loading constraints. Throw exception if violation is detected.
1692   // Grabs and releases SystemDictionary_lock
1693   // The check_constraints/find_class call and update_dictionary sequence
1694   // must be "atomic" for a specific class/classloader pair so we never
1695   // define two different instanceKlasses for that class/classloader pair.
1696   // Existing classloaders will call define_instance_class with the
1697   // classloader lock held
1698   // Parallel classloaders will call find_or_define_instance_class
1699   // which will require a token to perform the define class
1700   Symbol*  name_h = k->name();
1701   Dictionary* dictionary = loader_data->dictionary();
1702   unsigned int d_hash = dictionary->compute_hash(name_h);
1703   check_constraints(d_hash, k, class_loader_h, true, CHECK);
1704 
1705   // Register class just loaded with class loader (placed in ArrayList)
1706   // Note we do this before updating the dictionary, as this can
1707   // fail with an OutOfMemoryError (if it does, we will *not* put this
1708   // class in the dictionary and will not update the class hierarchy).
1709   // JVMTI FollowReferences needs to find the classes this way.
1710   if (k->class_loader() != NULL) {
1711     methodHandle m(THREAD, Universe::loader_addClass_method());
1712     JavaValue result(T_VOID);
1713     JavaCallArguments args(class_loader_h);
1714     args.push_oop(Handle(THREAD, k->java_mirror()));
1715     JavaCalls::call(&result, m, &args, CHECK);
1716   }
1717 
1718   // Add the new class. We need recompile lock during update of CHA.
1719   {
1720     unsigned int p_hash = placeholders()->compute_hash(name_h);
1721     int p_index = placeholders()->hash_to_index(p_hash);
1722 
1723     MutexLocker mu_r(THREAD, Compile_lock);
1724 
1725     // Add to class hierarchy, initialize vtables, and do possible
1726     // deoptimizations.
1727     add_to_hierarchy(k, CHECK); // No exception, but can block
1728 
1729     // Add to systemDictionary - so other classes can see it.
1730     // Grabs and releases SystemDictionary_lock
1731     update_dictionary(d_hash, p_index, p_hash,
1732                       k, class_loader_h, THREAD);
1733   }

1735 
1736   // notify jvmti
1737   if (JvmtiExport::should_post_class_load()) {
1738       assert(THREAD->is_Java_thread(), "thread->is_Java_thread()");
1739       JvmtiExport::post_class_load((JavaThread *) THREAD, k);
1740 
1741   }
1742   post_class_define_event(k, loader_data);

1745 // Support parallel classloading
1746 // All parallel class loaders, including bootstrap classloader
1747 // lock a placeholder entry for this class/class_loader pair
1748 // to allow parallel defines of different classes for this class loader
1749 // With AllowParallelDefine flag==true, in case they do not synchronize around
1750 // FindLoadedClass/DefineClass, calls, we check for parallel
1751 // loading for them, wait if a defineClass is in progress
1752 // and return the initial requestor's results
1753 // This flag does not apply to the bootstrap classloader.
1754 // With AllowParallelDefine flag==false, call through to define_instance_class
1755 // which will throw LinkageError: duplicate class definition.
1756 // False is the requested default.
1757 // For better performance, the class loaders should synchronize
1758 // findClass(), i.e. FindLoadedClass/DefineClassIfAbsent or they
1759 // potentially waste time reading and parsing the bytestream.
1760 // Note: VM callers should ensure consistency of k/class_name,class_loader
1761 // Be careful when modifying this code: once you have run
1762 // placeholders()->find_and_add(PlaceholderTable::DEFINE_CLASS),
1763 // you need to find_and_remove it before returning.
1764 // So be careful to not exit with a CHECK_ macro betweeen these calls.
1765 InstanceKlass* SystemDictionary::find_or_define_instance_class(Symbol* class_name, Handle class_loader,
1766                                                                InstanceKlass* k, TRAPS) {
1767 
1768   Symbol*  name_h = k->name(); // passed in class_name may be null
1769   ClassLoaderData* loader_data = class_loader_data(class_loader);
1770   Dictionary* dictionary = loader_data->dictionary();
1771 
1772   unsigned int d_hash = dictionary->compute_hash(name_h);
1773 
1774   // Hold SD lock around find_class and placeholder creation for DEFINE_CLASS
1775   unsigned int p_hash = placeholders()->compute_hash(name_h);
1776   int p_index = placeholders()->hash_to_index(p_hash);
1777   PlaceholderEntry* probe;
1778 
1779   {
1780     MutexLocker mu(THREAD, SystemDictionary_lock);
1781     // First check if class already defined
1782     if (is_parallelDefine(class_loader)) {
1783       InstanceKlass* check = find_class(d_hash, name_h, dictionary);
1784       if (check != NULL) {
1785         return check;
1786       }
1787     }
1788 
1789     // Acquire define token for this class/classloader
1790     probe = placeholders()->find_and_add(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, NULL, THREAD);
1791     // Wait if another thread defining in parallel
1792     // All threads wait - even those that will throw duplicate class: otherwise
1793     // caller is surprised by LinkageError: duplicate, but findLoadedClass fails
1794     // if other thread has not finished updating dictionary
1795     while (probe->definer() != NULL) {
1796       SystemDictionary_lock->wait();
1797     }
1798     // Only special cases allow parallel defines and can use other thread's results
1799     // Other cases fall through, and may run into duplicate defines
1800     // caught by finding an entry in the SystemDictionary
1801     if (is_parallelDefine(class_loader) && (probe->instance_klass() != NULL)) {
1802         placeholders()->find_and_remove(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, THREAD);
1803         SystemDictionary_lock->notify_all();
1804 #ifdef ASSERT
1805         InstanceKlass* check = find_class(d_hash, name_h, dictionary);
1806         assert(check != NULL, "definer missed recording success");
1807 #endif
1808         return probe->instance_klass();
1809     } else {
1810       // This thread will define the class (even if earlier thread tried and had an error)
1811       probe->set_definer(THREAD);
1812     }
1813   }
1814 
1815   define_instance_class(k, THREAD);
1816 
1817   Handle linkage_exception = Handle(); // null handle
1818 
1819   // definer must notify any waiting threads
1820   {
1821     MutexLocker mu(THREAD, SystemDictionary_lock);
1822     PlaceholderEntry* probe = placeholders()->get_entry(p_index, p_hash, name_h, loader_data);
1823     assert(probe != NULL, "DEFINE_CLASS placeholder lost?");
1824     if (probe != NULL) {
1825       if (HAS_PENDING_EXCEPTION) {
1826         linkage_exception = Handle(THREAD,PENDING_EXCEPTION);
1827         CLEAR_PENDING_EXCEPTION;
1828       } else {
1829         probe->set_instance_klass(k);
1830       }
1831       probe->set_definer(NULL);
1832       placeholders()->find_and_remove(p_index, p_hash, name_h, loader_data, PlaceholderTable::DEFINE_CLASS, THREAD);
1833       SystemDictionary_lock->notify_all();
1834     }
1835   }
1836 
1837   // Can't throw exception while holding lock due to rank ordering
1838   if (linkage_exception() != NULL) {
1839     THROW_OOP_(linkage_exception(), NULL); // throws exception and returns
1840   }
1841 
1842   return k;
1843 }

1948 // Assumes classes in the SystemDictionary are only unloaded at a safepoint
1949 // Note: anonymous classes are not in the SD.
1950 bool SystemDictionary::do_unloading(GCTimer* gc_timer) {
1951 
1952   bool unloading_occurred;
1953   bool is_concurrent = !SafepointSynchronize::is_at_safepoint();
1954   {
1955     GCTraceTime(Debug, gc, phases) t("ClassLoaderData", gc_timer);
1956     assert_locked_or_safepoint(ClassLoaderDataGraph_lock);  // caller locks.
1957     // First, mark for unload all ClassLoaderData referencing a dead class loader.
1958     unloading_occurred = ClassLoaderDataGraph::do_unloading();

1960       MutexLocker ml2(is_concurrent ? Module_lock : NULL);
1961       JFR_ONLY(Jfr::on_unloading_classes();)
1962 
1963       MutexLocker ml1(is_concurrent ? SystemDictionary_lock : NULL);
1964       ClassLoaderDataGraph::clean_module_and_package_info();
1965       constraints()->purge_loader_constraints();
1966       resolution_errors()->purge_resolution_errors();

1968   }
1969 
1970   GCTraceTime(Debug, gc, phases) t("Trigger cleanups", gc_timer);

1972   if (unloading_occurred) {

1974 
1975     // Oops referenced by the protection domain cache table may get unreachable independently
1976     // of the class loader (eg. cached protection domain oops). So we need to
1977     // explicitly unlink them here.
1978     // All protection domain oops are linked to the caller class, so if nothing
1979     // unloads, this is not needed.

2204 // Constraints on class loaders. The details of the algorithm can be
2205 // found in the OOPSLA'98 paper "Dynamic Class Loading in the Java
2206 // Virtual Machine" by Sheng Liang and Gilad Bracha.  The basic idea is
2207 // that the dictionary needs to maintain a set of contraints that
2208 // must be satisfied by all classes in the dictionary.
2209 // if defining is true, then LinkageError if already in dictionary
2210 // if initiating loader, then ok if InstanceKlass matches existing entry
2211 
2212 void SystemDictionary::check_constraints(unsigned int d_hash,
2213                                          InstanceKlass* k,
2214                                          Handle class_loader,
2215                                          bool defining,
2216                                          TRAPS) {
2217   ResourceMark rm(THREAD);
2218   stringStream ss;
2219   bool throwException = false;
2220 
2221   {
2222     Symbol *name = k->name();
2223     ClassLoaderData *loader_data = class_loader_data(class_loader);
2224 
2225     MutexLocker mu(THREAD, SystemDictionary_lock);
2226 
2227     InstanceKlass* check = find_class(d_hash, name, loader_data->dictionary());
2228     if (check != NULL) {
2229       // If different InstanceKlass - duplicate class definition,
2230       // else - ok, class loaded by a different thread in parallel.
2231       // We should only have found it if it was done loading and ok to use.
2232       // The dictionary only holds instance classes, placeholders
2233       // also hold array classes.
2234 
2235       assert(check->is_instance_klass(), "noninstance in systemdictionary");
2236       if ((defining == true) || (k != check)) {
2237         throwException = true;
2238         ss.print("loader %s", loader_data->loader_name_and_id());
2239         ss.print(" attempted duplicate %s definition for %s. (%s)",
2240                  k->external_kind(), k->external_name(), k->class_in_module_of_loader(false, true));
2241       } else {
2242         return;
2243       }
2244     }
2245 
2246 #ifdef ASSERT
2247     Symbol* ph_check = find_placeholder(name, loader_data);
2248     assert(ph_check == NULL || ph_check == name, "invalid symbol");
2249 #endif

2476 // Signature constraints ensure that callers and callees agree about
2477 // the meaning of type names in their signatures.  This routine is the
2478 // intake for constraints.  It collects them from several places:
2479 //
2480 //  * LinkResolver::resolve_method (if check_access is true) requires
2481 //    that the resolving class (the caller) and the defining class of
2482 //    the resolved method (the callee) agree on each type in the
2483 //    method's signature.
2484 //
2485 //  * LinkResolver::resolve_interface_method performs exactly the same
2486 //    checks.
2487 //
2488 //  * LinkResolver::resolve_field requires that the constant pool
2489 //    attempting to link to a field agree with the field's defining
2490 //    class about the type of the field signature.
2491 //
2492 //  * klassVtable::initialize_vtable requires that, when a class
2493 //    overrides a vtable entry allocated by a superclass, that the
2494 //    overriding method (i.e., the callee) agree with the superclass
2495 //    on each type in the method's signature.
2496 //
2497 //  * klassItable::initialize_itable requires that, when a class fills
2498 //    in its itables, for each non-abstract method installed in an
2499 //    itable, the method (i.e., the callee) agree with the interface
2500 //    on each type in the method's signature.
2501 //
2502 // All those methods have a boolean (check_access, checkconstraints)
2503 // which turns off the checks.  This is used from specialized contexts
2504 // such as bootstrapping, dumping, and debugging.
2505 //
2506 // No direct constraint is placed between the class and its
2507 // supertypes.  Constraints are only placed along linked relations
2508 // between callers and callees.  When a method overrides or implements
2509 // an abstract method in a supertype (superclass or interface), the
2510 // constraints are placed as if the supertype were the caller to the
2511 // overriding method.  (This works well, since callers to the
2512 // supertype have already established agreement between themselves and
2513 // the supertype.)  As a result of all this, a class can disagree with
2514 // its supertype about the meaning of a type name, as long as that
2515 // class neither calls a relevant method of the supertype, nor is
2516 // called (perhaps via an override) from the supertype.
2517 //
2518 //
2519 // SystemDictionary::check_signature_loaders(sig, klass_being_linked, l1, l2)
2520 //
2521 // Make sure all class components (including arrays) in the given
2522 // signature will be resolved to the same class in both loaders.
2523 // Returns the name of the type that failed a loader constraint check, or
2524 // NULL if no constraint failed.  No exception except OOME is thrown.
2525 // Arrays are not added to the loader constraint table, their elements are.
2526 Symbol* SystemDictionary::check_signature_loaders(Symbol* signature,
2527                                                Klass* klass_being_linked,
2528                                                Handle loader1, Handle loader2,
2529                                                bool is_method, TRAPS)  {
2530   // Nothing to do if loaders are the same.
2531   if (loader1() == loader2()) {
2532     return NULL;
2533   }
2534 
2535   for (SignatureStream ss(signature, is_method); !ss.is_done(); ss.next()) {
2536     if (ss.is_reference()) {
2537       Symbol* sig = ss.as_symbol();
2538       // Note: In the future, if template-like types can take
2539       // arguments, we will want to recognize them and dig out class
2540       // names hiding inside the argument lists.
2541       if (!add_loader_constraint(sig, klass_being_linked, loader1, loader2, THREAD)) {
2542         return sig;
2543       }
2544     }
2545   }
2546   return NULL;
2547 }

2780       mirror = NULL;  // safety
2781       // Emulate ConstantPool::verify_constant_pool_resolve.
2782       LinkResolver::check_klass_accessibility(accessing_klass, sel_klass, CHECK_(empty));
2783     }
2784   }
2785   assert(arg == npts, "");
2786 
2787   // call java.lang.invoke.MethodHandleNatives::findMethodHandleType(Class rt, Class[] pts) -> MethodType
2788   JavaCallArguments args(Handle(THREAD, rt()));
2789   args.push_oop(pts);
2790   JavaValue result(T_OBJECT);
2791   JavaCalls::call_static(&result,
2792                          SystemDictionary::MethodHandleNatives_klass(),
2793                          vmSymbols::findMethodHandleType_name(),
2794                          vmSymbols::findMethodHandleType_signature(),
2795                          &args, CHECK_(empty));
2796   Handle method_type(THREAD, (oop) result.get_jobject());
2797 
2798   if (can_be_cached) {
2799     // We can cache this MethodType inside the JVM.
2800     MutexLocker ml(THREAD, SystemDictionary_lock);
2801     spe = invoke_method_table()->find_entry(index, hash, signature, null_iid);
2802     if (spe == NULL)
2803       spe = invoke_method_table()->add_entry(index, hash, signature, null_iid);
2804     if (spe->method_type() == NULL) {
2805       spe->set_method_type(method_type());
2806     }
2807   }
2808 
2809   // report back to the caller with the MethodType
2810   return method_type;
2811 }

2813 Handle SystemDictionary::find_field_handle_type(Symbol* signature,
2814                                                 Klass* accessing_klass,
2815                                                 TRAPS) {
2816   Handle empty;
2817   ResourceMark rm(THREAD);
2818   SignatureStream ss(signature, /*is_method=*/ false);
2819   if (!ss.is_done()) {
2820     Handle class_loader, protection_domain;
2821     if (accessing_klass != NULL) {
2822       class_loader      = Handle(THREAD, accessing_klass->class_loader());

2826     ss.next();

2834 // Ask Java code to find or construct a method handle constant.
2835 Handle SystemDictionary::link_method_handle_constant(Klass* caller,
2836                                                      int ref_kind, //e.g., JVM_REF_invokeVirtual
2837                                                      Klass* callee,
2838                                                      Symbol* name,
2839                                                      Symbol* signature,
2840                                                      TRAPS) {
2841   Handle empty;
2842   if (caller == NULL) {
2843     THROW_MSG_(vmSymbols::java_lang_InternalError(), "bad MH constant", empty);
2844   }
2845   Handle name_str      = java_lang_String::create_from_symbol(name,      CHECK_(empty));
2846   Handle signature_str = java_lang_String::create_from_symbol(signature, CHECK_(empty));
2847 
2848   // Put symbolic info from the MH constant into freshly created MemberName and resolve it.
2849   Handle mname = MemberName_klass()->allocate_instance_handle(CHECK_(empty));
2850   java_lang_invoke_MemberName::set_clazz(mname(), callee->java_mirror());
2851   java_lang_invoke_MemberName::set_name (mname(), name_str());
2852   java_lang_invoke_MemberName::set_type (mname(), signature_str());
2853   java_lang_invoke_MemberName::set_flags(mname(), MethodHandles::ref_kind_to_flags(ref_kind));
2854 
2855   if (ref_kind == JVM_REF_invokeVirtual &&
2856       MethodHandles::is_signature_polymorphic_public_name(callee, name)) {
2857     // Skip resolution for public signature polymorphic methods such as
2858     // j.l.i.MethodHandle.invoke()/invokeExact() and those on VarHandle
2859     // They require appendix argument which MemberName resolution doesn't handle.
2860     // There's special logic on JDK side to handle them
2861     // (see MethodHandles.linkMethodHandleConstant() and MethodHandles.findVirtualForMH()).
2862   } else {
2863     MethodHandles::resolve_MemberName(mname, caller, /*speculative_resolve*/false, CHECK_(empty));
2864   }
2865 
2866   // After method/field resolution succeeded, it's safe to resolve MH signature as well.
2867   Handle type = MethodHandles::resolve_MemberName_type(mname, caller, CHECK_(empty));
2868 
2869   // call java.lang.invoke.MethodHandleNatives::linkMethodHandleConstant(Class caller, int refKind, Class callee, String name, Object type) -> MethodHandle
2870   JavaCallArguments args;
2871   args.push_oop(Handle(THREAD, caller->java_mirror()));  // the referring class
2872   args.push_int(ref_kind);
2873   args.push_oop(Handle(THREAD, callee->java_mirror()));  // the target class
2874   args.push_oop(name_str);
2875   args.push_oop(type);
2876   JavaValue result(T_OBJECT);
2877   JavaCalls::call_static(&result,
2878                          SystemDictionary::MethodHandleNatives_klass(),
2879                          vmSymbols::linkMethodHandleConstant_name(),
2880                          vmSymbols::linkMethodHandleConstant_signature(),
2881                          &args, CHECK_(empty));
2882   return Handle(THREAD, (oop) result.get_jobject());
2883 }

2885 // Ask Java to run a bootstrap method, in order to create a dynamic call site
2886 // while linking an invokedynamic op, or compute a constant for Dynamic_info CP entry
2887 // with linkage results being stored back into the bootstrap specifier.
2888 void SystemDictionary::invoke_bootstrap_method(BootstrapInfo& bootstrap_specifier, TRAPS) {
2889   // Resolve the bootstrap specifier, its name, type, and static arguments
2890   bootstrap_specifier.resolve_bsm(CHECK);
2891 
2892   // This should not happen.  JDK code should take care of that.
2893   if (bootstrap_specifier.caller() == NULL || bootstrap_specifier.type_arg().is_null()) {
2894     THROW_MSG(vmSymbols::java_lang_InternalError(), "Invalid bootstrap method invocation with no caller or type argument");
2895   }
2896 
2897   bool is_indy = bootstrap_specifier.is_method_call();
2898   objArrayHandle appendix_box;
2899   if (is_indy) {
2900     // Some method calls may require an appendix argument.  Arrange to receive it.
2901     appendix_box = oopFactory::new_objArray_handle(SystemDictionary::Object_klass(), 1, CHECK);
2902     assert(appendix_box->obj_at(0) == NULL, "");
2903   }
2904 
2905   // call condy: java.lang.invoke.MethodHandleNatives::linkDynamicConstant(caller, condy_index, bsm, type, info)
2906   //       indy: java.lang.invoke.MethodHandleNatives::linkCallSite(caller, indy_index, bsm, name, mtype, info, &appendix)
2907   JavaCallArguments args;
2908   args.push_oop(Handle(THREAD, bootstrap_specifier.caller_mirror()));
2909   args.push_int(bootstrap_specifier.bss_index());
2910   args.push_oop(bootstrap_specifier.bsm());
2911   args.push_oop(bootstrap_specifier.name_arg());
2912   args.push_oop(bootstrap_specifier.type_arg());
2913   args.push_oop(bootstrap_specifier.arg_values());
2914   if (is_indy) {
2915     args.push_oop(appendix_box);
2916   }
2917   JavaValue result(T_OBJECT);
2918   JavaCalls::call_static(&result,
2919                          SystemDictionary::MethodHandleNatives_klass(),
2920                          is_indy ? vmSymbols::linkCallSite_name() : vmSymbols::linkDynamicConstant_name(),
2921                          is_indy ? vmSymbols::linkCallSite_signature() : vmSymbols::linkDynamicConstant_signature(),
2922                          &args, CHECK);
2923 
2924   Handle value(THREAD, (oop) result.get_jobject());
2925   if (is_indy) {
2926     Handle appendix;
2927     Method* method = unpack_method_and_appendix(value,
2928                                                 bootstrap_specifier.caller(),
2929                                                 appendix_box,
2930                                                 &appendix, CHECK);
2931     methodHandle mh(THREAD, method);
2932     bootstrap_specifier.set_resolved_method(mh, appendix);
2933   } else {
2934     bootstrap_specifier.set_resolved_value(value);
2935   }
2936 
2937   // sanity check
2938   assert(bootstrap_specifier.is_resolved() ||
2939          (bootstrap_specifier.is_method_call() &&
2940           bootstrap_specifier.resolved_method().not_null()), "bootstrap method call failed");
2941 }
