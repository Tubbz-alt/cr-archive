26 #ifndef JDWP_SIGNATURE_H
27 #define JDWP_SIGNATURE_H
28 
29 #define JVM_TYPE_ASSERT(typeKey)\
30 JDI_ASSERT_MSG(JDWP_Tag_OBJECT == typeKey || \
31                JDWP_Tag_ARRAY == typeKey || \
32                JDWP_Tag_INLINE_OBJECT == typeKey || \
33                JDWP_Tag_BOOLEAN == typeKey || \
34                JDWP_Tag_BYTE == typeKey || \
35                JDWP_Tag_CHAR == typeKey || \
36                JDWP_Tag_DOUBLE == typeKey || \
37                JDWP_Tag_FLOAT == typeKey || \
38                JDWP_Tag_INT == typeKey || \
39                JDWP_Tag_LONG == typeKey || \
40                JDWP_Tag_SHORT == typeKey || \
41                JDWP_Tag_VOID == typeKey, \
42                "Tag is not a JVM basic type")
43 
44 static inline jbyte jdwpTag(const char *signature) {
45      JVM_TYPE_ASSERT(signature[0]);
46      return signature[0];
47 }
48 
49 static inline jboolean isReferenceTag(jbyte typeKey) {
50     JVM_TYPE_ASSERT(typeKey);
51     return (typeKey == JDWP_TAG(OBJECT)) || (typeKey == JDWP_TAG(ARRAY)) || (typeKey == JDWP_TAG(INLINE_OBJECT));
52 }
53 
54 static inline jboolean isArrayTag(jbyte typeKey) {
55     JVM_TYPE_ASSERT(typeKey);
56     return (typeKey == JDWP_TAG(ARRAY));
57 }
