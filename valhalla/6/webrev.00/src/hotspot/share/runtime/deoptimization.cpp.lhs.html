  85 Deoptimization::UnrollBlock::UnrollBlock(int  size_of_deoptimized_frame,
  86                                          int  caller_adjustment,
  87                                          int  caller_actual_parameters,
  88                                          int  number_of_frames,
  89                                          intptr_t* frame_sizes,
  90                                          address* frame_pcs,
  91                                          BasicType return_type,
  92                                          int exec_mode) {
  93   _size_of_deoptimized_frame = size_of_deoptimized_frame;
  94   _caller_adjustment         = caller_adjustment;
  95   _caller_actual_parameters  = caller_actual_parameters;
  96   _number_of_frames          = number_of_frames;
  97   _frame_sizes               = frame_sizes;
  98   _frame_pcs                 = frame_pcs;
  99   _register_block            = NEW_C_HEAP_ARRAY(intptr_t, RegisterMap::reg_count * 2, mtCompiler);
 100   _return_type               = return_type;
 101   _initial_info              = 0;
 102   // PD (x86 only)
 103   _counter_temp              = 0;
 104   _unpack_kind               = exec_mode;
 105   _sender_sp_temp            = 0;
 106 
 107   _total_frame_sizes         = size_of_frames();
 108   assert(exec_mode >= 0 && exec_mode < Unpack_LIMIT, "Unexpected exec_mode");
 109 }

 126 int Deoptimization::UnrollBlock::size_of_frames() const {
 127   // Acount first for the adjustment of the initial frame
 128   int result = _caller_adjustment;
 129   for (int index = 0; index < number_of_frames(); index++) {
 130     result += frame_sizes()[index];
 131   }
 132   return result;
 133 }

 148 // In order to make fetch_unroll_info work properly with escape
 149 // analysis, The method was changed from JRT_LEAF to JRT_BLOCK_ENTRY and
 150 // ResetNoHandleMark and HandleMark were removed from it. The actual reallocation
 151 // of previously eliminated objects occurs in realloc_objects, which is
 152 // called from the method fetch_unroll_info_helper below.
 153 JRT_BLOCK_ENTRY(Deoptimization::UnrollBlock*, Deoptimization::fetch_unroll_info(JavaThread* thread, int exec_mode))
 154   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 155   // but makes the entry a little slower. There is however a little dance we have to
 156   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 157 
 158   // fetch_unroll_info() is called at the beginning of the deoptimization
 159   // handler. Note this fact before we start generating temporary frames
 160   // that can confuse an asynchronous stack walker. This counter is
 161   // decremented at the end of unpack_frames().
 162   if (TraceDeoptimization) {
 163     tty->print_cr("Deoptimizing thread " INTPTR_FORMAT, p2i(thread));
 164   }
 165   thread->inc_in_deopt_handler();
 166 
 167   return fetch_unroll_info_helper(thread, exec_mode);
 168 JRT_END

 170 #if COMPILER2_OR_JVMCI
 171 static bool eliminate_allocations(JavaThread* thread, int exec_mode, CompiledMethod* compiled_method,
 172                                   frame& deoptee, RegisterMap& map, GrowableArray<compiledVFrame*>* chunk) {
 173   bool realloc_failures = false;
 174   assert (chunk->at(0)->scope() != NULL,"expect only compiled java frames");
 175 
 176   GrowableArray<ScopeValue*>* objects = chunk->at(0)->scope()->objects();
 177 
 178   // The flag return_oop() indicates call sites which return oop
 179   // in compiled code. Such sites include java method calls,
 180   // runtime calls (for example, used to allocate new objects/arrays
 181   // on slow code path) and any other calls generated in compiled code.
 182   // It is not guaranteed that we can get such information here only
 183   // by analyzing bytecode in deoptimized frames. This is why this flag
 184   // is set during method compilation (see Compile::Process_OopMap_Node()).
 185   // If the previous frame was popped or if we are dispatching an exception,
 186   // we don't have an oop result.
 187   ScopeDesc* scope = chunk->at(0)->scope();
 188   bool save_oop_result = scope->return_oop() && !thread->popframe_forcing_deopt_reexecution() && (exec_mode == Deoptimization::Unpack_deopt);
 189   // In case of the return of multiple values, we must take care
 190   // of all oop return values.
 191   GrowableArray<Handle> return_oops;
 192   ValueKlass* vk = NULL;
 193   if (save_oop_result && scope->return_vt()) {
 194     vk = ValueKlass::returned_value_klass(map);
 195     if (vk != NULL) {
 196       vk->save_oop_fields(map, return_oops);
 197       save_oop_result = false;
 198     }
 199   }
 200   if (save_oop_result) {
 201     // Reallocation may trigger GC. If deoptimization happened on return from
 202     // call which returns oop we need to save it since it is not in oopmap.
 203     oop result = deoptee.saved_oop_result(&map);
 204     assert(oopDesc::is_oop_or_null(result), "must be oop");
 205     return_oops.push(Handle(thread, result));
 206     assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
 207     if (TraceDeoptimization) {
 208       ttyLocker ttyl;
 209       tty->print_cr("SAVED OOP RESULT " INTPTR_FORMAT " in thread " INTPTR_FORMAT, p2i(result), p2i(thread));
 210     }
 211   }
 212   if (objects != NULL || vk != NULL) {
 213     bool skip_internal = (compiled_method != NULL) && !compiled_method->is_compiled_by_jvmci();
 214     JRT_BLOCK
 215       if (vk != NULL) {
 216         realloc_failures = Deoptimization::realloc_value_type_result(vk, map, return_oops, THREAD);
 217       }
 218       if (objects != NULL) {
 219         realloc_failures = realloc_failures || Deoptimization::realloc_objects(thread, &deoptee, &map, objects, THREAD);
 220         Deoptimization::reassign_fields(&deoptee, &map, objects, realloc_failures, skip_internal, THREAD);
 221       }
 222     JRT_END
 223 #ifndef PRODUCT
 224     if (TraceDeoptimization) {
 225       ttyLocker ttyl;
 226       tty->print_cr("REALLOC OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 227       Deoptimization::print_objects(objects, realloc_failures);
 228     }
 229 #endif
 230   }
 231   if (save_oop_result || vk != NULL) {
 232     // Restore result.
 233     assert(return_oops.length() == 1, "no value type");
 234     deoptee.set_saved_oop_result(&map, return_oops.pop()());
 235   }
 236   return realloc_failures;
 237 }

 239 static void eliminate_locks(JavaThread* thread, GrowableArray<compiledVFrame*>* chunk, bool realloc_failures) {
 240 #ifndef PRODUCT
 241   bool first = true;
 242 #endif
 243   for (int i = 0; i < chunk->length(); i++) {
 244     compiledVFrame* cvf = chunk->at(i);
 245     assert (cvf->scope() != NULL,"expect only compiled java frames");
 246     GrowableArray<MonitorInfo*>* monitors = cvf->monitors();
 247     if (monitors->is_nonempty()) {
 248       Deoptimization::relock_objects(monitors, thread, realloc_failures);
 249 #ifndef PRODUCT
 250       if (PrintDeoptimizationDetails) {
 251         ttyLocker ttyl;
 252         for (int j = 0; j < monitors->length(); j++) {
 253           MonitorInfo* mi = monitors->at(j);
 254           if (mi->eliminated()) {
 255             if (first) {
 256               first = false;
 257               tty->print_cr("RELOCK OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 258             }
 259             if (mi->owner_is_scalar_replaced()) {
 260               Klass* k = java_lang_Class::as_Klass(mi->owner_klass());
 261               tty->print_cr("     failed reallocation for klass %s", k->external_name());
 262             } else {
 263               tty->print_cr("     object <" INTPTR_FORMAT "> locked", p2i(mi->owner()));
 264             }
 265           }
 266         }
 267       }
 268 #endif // !PRODUCT
 269     }
 270   }
 271 }

 274 // This is factored, since it is both called from a JRT_LEAF (deoptimization) and a JRT_ENTRY (uncommon_trap)
 275 Deoptimization::UnrollBlock* Deoptimization::fetch_unroll_info_helper(JavaThread* thread, int exec_mode) {
 276 
 277   // Note: there is a safepoint safety issue here. No matter whether we enter
 278   // via vanilla deopt or uncommon trap we MUST NOT stop at a safepoint once
 279   // the vframeArray is created.
 280   //
 281 
 282   // Allocate our special deoptimization ResourceMark
 283   DeoptResourceMark* dmark = new DeoptResourceMark(thread);
 284   assert(thread->deopt_mark() == NULL, "Pending deopt!");
 285   thread->set_deopt_mark(dmark);
 286 
 287   frame stub_frame = thread->last_frame(); // Makes stack walkable as side effect
 288   RegisterMap map(thread, true);
 289   RegisterMap dummy_map(thread, false);
 290   // Now get the deoptee with a valid map
 291   frame deoptee = stub_frame.sender(&map);
 292   // Set the deoptee nmethod
 293   assert(thread->deopt_compiled_method() == NULL, "Pending deopt!");
 294   CompiledMethod* cm = deoptee.cb()->as_compiled_method_or_null();
 295   thread->set_deopt_compiled_method(cm);
 296 
 297   if (VerifyStack) {
 298     thread->validate_frame_layout();
 299   }
 300 
 301   // Create a growable array of VFrames where each VFrame represents an inlined
 302   // Java frame.  This storage is allocated with the usual system arena.
 303   assert(deoptee.is_compiled_frame(), "Wrong frame type");
 304   GrowableArray<compiledVFrame*>* chunk = new GrowableArray<compiledVFrame*>(10);
 305   vframe* vf = vframe::new_vframe(&deoptee, &map, thread);
 306   while (!vf->is_top()) {
 307     assert(vf->is_compiled_frame(), "Wrong frame type");
 308     chunk->push(compiledVFrame::cast(vf));
 309     vf = vf->sender();
 310   }
 311   assert(vf->is_compiled_frame(), "Wrong frame type");
 312   chunk->push(compiledVFrame::cast(vf));
 313 
 314   bool realloc_failures = false;
 315 
 316 #if COMPILER2_OR_JVMCI
 317 #if INCLUDE_JVMCI
 318   bool jvmci_enabled = true;
 319 #else
 320   bool jvmci_enabled = false;
 321 #endif
 322 
 323   // Reallocate the non-escaping objects and restore their fields. Then
 324   // relock objects if synchronization on them was eliminated.
 325   if (jvmci_enabled COMPILER2_PRESENT( || (DoEscapeAnalysis && EliminateAllocations) )) {
 326     realloc_failures = eliminate_allocations(thread, exec_mode, cm, deoptee, map, chunk);
 327   }
 328 #endif // COMPILER2_OR_JVMCI
 329 
 330   // Revoke biases, done with in java state.
 331   // No safepoints allowed after this
 332   revoke_from_deopt_handler(thread, deoptee, &map);
 333 
 334   // Ensure that no safepoint is taken after pointers have been stored
 335   // in fields of rematerialized objects.  If a safepoint occurs from here on
 336   // out the java state residing in the vframeArray will be missed.
 337   // Locks may be rebaised in a safepoint.
 338   NoSafepointVerifier no_safepoint;
 339 
 340 #if COMPILER2_OR_JVMCI
 341   if (jvmci_enabled COMPILER2_PRESENT( || ((DoEscapeAnalysis || EliminateNestedLocks) && EliminateLocks) )) {
 342     eliminate_locks(thread, chunk, realloc_failures);
 343   }
 344 #endif // COMPILER2_OR_JVMCI
 345 
 346   ScopeDesc* trap_scope = chunk->at(0)->scope();
 347   Handle exceptionObject;
 348   if (trap_scope->rethrow_exception()) {
 349     if (PrintDeoptimizationDetails) {
 350       tty->print_cr("Exception to be rethrown in the interpreter for method %s::%s at bci %d", trap_scope->method()->method_holder()->name()->as_C_string(), trap_scope->method()->name()->as_C_string(), trap_scope->bci());
 351     }
 352     GrowableArray<ScopeValue*>* expressions = trap_scope->expressions();
 353     guarantee(expressions != NULL && expressions->length() > 0, "must have exception to throw");
 354     ScopeValue* topOfStack = expressions->top();
 355     exceptionObject = StackValue::create_stack_value(&deoptee, &map, topOfStack)->get_obj();
 356     guarantee(exceptionObject() != NULL, "exception oop can not be null");
 357   }
 358 
 359   vframeArray* array = create_vframeArray(thread, deoptee, &map, chunk, realloc_failures);
 360 #if COMPILER2_OR_JVMCI
 361   if (realloc_failures) {
 362     pop_frames_failed_reallocs(thread, array);
 363   }
 364 #endif
 365 
 366   assert(thread->vframe_array_head() == NULL, "Pending deopt!");
 367   thread->set_vframe_array_head(array);
 368 
 369   // Now that the vframeArray has been created if we have any deferred local writes
 370   // added by jvmti then we can free up that structure as the data is now in the
 371   // vframeArray
 372 
 373   if (thread->deferred_locals() != NULL) {
 374     GrowableArray<jvmtiDeferredLocalVariableSet*>* list = thread->deferred_locals();
 375     int i = 0;
 376     do {
 377       // Because of inlining we could have multiple vframes for a single frame
 378       // and several of the vframes could have deferred writes. Find them all.
 379       if (list->at(i)->id() == array->original().id()) {
 380         jvmtiDeferredLocalVariableSet* dlv = list->at(i);
 381         list->remove_at(i);
 382         // individual jvmtiDeferredLocalVariableSet are CHeapObj's
 383         delete dlv;
 384       } else {
 385         i++;
 386       }
 387     } while ( i < list->length() );
 388     if (list->length() == 0) {
 389       thread->set_deferred_locals(NULL);
 390       // free the list and elements back to C heap.
 391       delete list;
 392     }
 393 
 394   }
 395 
 396   // Compute the caller frame based on the sender sp of stub_frame and stored frame sizes info.
 397   CodeBlob* cb = stub_frame.cb();
 398   // Verify we have the right vframeArray
 399   assert(cb->frame_size() >= 0, "Unexpected frame size");
 400   intptr_t* unpack_sp = stub_frame.sp() + cb->frame_size();
 401 
 402   // If the deopt call site is a MethodHandle invoke call site we have
 403   // to adjust the unpack_sp.
 404   nmethod* deoptee_nm = deoptee.cb()->as_nmethod_or_null();
 405   if (deoptee_nm != NULL && deoptee_nm->is_method_handle_return(deoptee.pc()))
 406     unpack_sp = deoptee.unextended_sp();
 407 
 408 #ifdef ASSERT
 409   assert(cb->is_deoptimization_stub() ||
 410          cb->is_uncommon_trap_stub() ||
 411          strcmp("Stub<DeoptimizationStub.deoptimizationHandler>", cb->name()) == 0 ||
 412          strcmp("Stub<UncommonTrapStub.uncommonTrapHandler>", cb->name()) == 0,
 413          "unexpected code blob: %s", cb->name());
 414 #endif
 415 
 416   // This is a guarantee instead of an assert because if vframe doesn't match
 417   // we will unpack the wrong deoptimized frame and wind up in strange places
 418   // where it will be very difficult to figure out what went wrong. Better
 419   // to die an early death here than some very obscure death later when the
 420   // trail is cold.
 421   // Note: on ia64 this guarantee can be fooled by frames with no memory stack
 422   // in that it will fail to detect a problem when there is one. This needs
 423   // more work in tiger timeframe.
 424   guarantee(array->unextended_sp() == unpack_sp, "vframe_array_head must contain the vframeArray to unpack");
 425 
 426   int number_of_frames = array->frames();
 427 
 428   // Compute the vframes' sizes.  Note that frame_sizes[] entries are ordered from outermost to innermost
 429   // virtual activation, which is the reverse of the elements in the vframes array.
 430   intptr_t* frame_sizes = NEW_C_HEAP_ARRAY(intptr_t, number_of_frames, mtCompiler);
 431   // +1 because we always have an interpreter return address for the final slot.
 432   address* frame_pcs = NEW_C_HEAP_ARRAY(address, number_of_frames + 1, mtCompiler);
 433   int popframe_extra_args = 0;
 434   // Create an interpreter return address for the stub to use as its return
 435   // address so the skeletal frames are perfectly walkable
 436   frame_pcs[number_of_frames] = Interpreter::deopt_entry(vtos, 0);
 437 
 438   // PopFrame requires that the preserved incoming arguments from the recently-popped topmost
 439   // activation be put back on the expression stack of the caller for reexecution
 440   if (JvmtiExport::can_pop_frame() && thread->popframe_forcing_deopt_reexecution()) {
 441     popframe_extra_args = in_words(thread->popframe_preserved_args_size_in_words());
 442   }
 443 
 444   // Find the current pc for sender of the deoptee. Since the sender may have been deoptimized
 445   // itself since the deoptee vframeArray was created we must get a fresh value of the pc rather
 446   // than simply use array->sender.pc(). This requires us to walk the current set of frames
 447   //
 448   frame deopt_sender = stub_frame.sender(&dummy_map); // First is the deoptee frame
 449   deopt_sender = deopt_sender.sender(&dummy_map);     // Now deoptee caller
 450 
 451   // It's possible that the number of parameters at the call site is
 452   // different than number of arguments in the callee when method
 453   // handles are used.  If the caller is interpreted get the real
 454   // value so that the proper amount of space can be added to it's
 455   // frame.
 456   bool caller_was_method_handle = false;
 457   if (deopt_sender.is_interpreted_frame()) {
 458     methodHandle method(thread, deopt_sender.interpreter_frame_method());
 459     Bytecode_invoke cur = Bytecode_invoke_check(method, deopt_sender.interpreter_frame_bci());
 460     if (cur.is_invokedynamic() || cur.is_invokehandle()) {
 461       // Method handle invokes may involve fairly arbitrary chains of
 462       // calls so it's impossible to know how much actual space the
 463       // caller has for locals.
 464       caller_was_method_handle = true;
 465     }
 466   }
 467 
 468   //
 469   // frame_sizes/frame_pcs[0] oldest frame (int or c2i)
 470   // frame_sizes/frame_pcs[1] next oldest frame (int)
 471   // frame_sizes/frame_pcs[n] youngest frame (int)
 472   //
 473   // Now a pc in frame_pcs is actually the return address to the frame's caller (a frame
 474   // owns the space for the return address to it's caller).  Confusing ain't it.
 475   //
 476   // The vframe array can address vframes with indices running from
 477   // 0.._frames-1. Index  0 is the youngest frame and _frame - 1 is the oldest (root) frame.
 478   // When we create the skeletal frames we need the oldest frame to be in the zero slot
 479   // in the frame_sizes/frame_pcs so the assembly code can do a trivial walk.
 480   // so things look a little strange in this loop.
 481   //
 482   int callee_parameters = 0;
 483   int callee_locals = 0;
 484   for (int index = 0; index < array->frames(); index++ ) {
 485     // frame[number_of_frames - 1 ] = on_stack_size(youngest)
 486     // frame[number_of_frames - 2 ] = on_stack_size(sender(youngest))
 487     // frame[number_of_frames - 3 ] = on_stack_size(sender(sender(youngest)))
 488     frame_sizes[number_of_frames - 1 - index] = BytesPerWord * array->element(index)->on_stack_size(callee_parameters,
 489                                                                                                     callee_locals,
 490                                                                                                     index == 0,
 491                                                                                                     popframe_extra_args);
 492     // This pc doesn't have to be perfect just good enough to identify the frame
 493     // as interpreted so the skeleton frame will be walkable
 494     // The correct pc will be set when the skeleton frame is completely filled out
 495     // The final pc we store in the loop is wrong and will be overwritten below
 496     frame_pcs[number_of_frames - 1 - index ] = Interpreter::deopt_entry(vtos, 0) - frame::pc_return_offset;
 497 
 498     callee_parameters = array->element(index)->method()->size_of_parameters();
 499     callee_locals = array->element(index)->method()->max_locals();
 500     popframe_extra_args = 0;
 501   }
 502 
 503   // Compute whether the root vframe returns a float or double value.
 504   BasicType return_type;
 505   {
 506     methodHandle method(thread, array->element(0)->method());
 507     Bytecode_invoke invoke = Bytecode_invoke_check(method, array->element(0)->bci());
 508     return_type = invoke.is_valid() ? invoke.result_type() : T_ILLEGAL;
 509   }
 510 
 511   // Compute information for handling adapters and adjusting the frame size of the caller.
 512   int caller_adjustment = 0;
 513 
 514   // Compute the amount the oldest interpreter frame will have to adjust
 515   // its caller's stack by. If the caller is a compiled frame then
 516   // we pretend that the callee has no parameters so that the
 517   // extension counts for the full amount of locals and not just
 518   // locals-parms. This is because without a c2i adapter the parm
 519   // area as created by the compiled frame will not be usable by
 520   // the interpreter. (Depending on the calling convention there
 521   // may not even be enough space).
 522 
 523   // QQQ I'd rather see this pushed down into last_frame_adjust
 524   // and have it take the sender (aka caller).
 525 
 526   if (deopt_sender.is_compiled_frame() || caller_was_method_handle) {
 527     caller_adjustment = last_frame_adjust(0, callee_locals);
 528   } else if (callee_locals > callee_parameters) {
 529     // The caller frame may need extending to accommodate
 530     // non-parameter locals of the first unpacked interpreted frame.
 531     // Compute that adjustment.
 532     caller_adjustment = last_frame_adjust(callee_parameters, callee_locals);
 533   }
 534 
 535   // If the sender is deoptimized we must retrieve the address of the handler
 536   // since the frame will "magically" show the original pc before the deopt
 537   // and we'd undo the deopt.
 538 
 539   frame_pcs[0] = deopt_sender.raw_pc();
 540 
 541   assert(CodeCache::find_blob_unsafe(frame_pcs[0]) != NULL, "bad pc");
 542 
 543 #if INCLUDE_JVMCI
 544   if (exceptionObject() != NULL) {
 545     thread->set_exception_oop(exceptionObject());
 546     exec_mode = Unpack_exception;
 547   }
 548 #endif
 549 
 550   if (thread->frames_to_pop_failed_realloc() > 0 && exec_mode != Unpack_uncommon_trap) {
 551     assert(thread->has_pending_exception(), "should have thrown OOME");
 552     thread->set_exception_oop(thread->pending_exception());
 553     thread->clear_pending_exception();
 554     exec_mode = Unpack_exception;
 555   }
 556 
 557 #if INCLUDE_JVMCI
 558   if (thread->frames_to_pop_failed_realloc() > 0) {
 559     thread->set_pending_monitorenter(false);
 560   }
 561 #endif
 562 
 563   UnrollBlock* info = new UnrollBlock(array->frame_size() * BytesPerWord,
 564                                       caller_adjustment * BytesPerWord,
 565                                       caller_was_method_handle ? 0 : callee_parameters,
 566                                       number_of_frames,
 567                                       frame_sizes,
 568                                       frame_pcs,
 569                                       return_type,
 570                                       exec_mode);
 571   // On some platforms, we need a way to pass some platform dependent
 572   // information to the unpacking code so the skeletal frames come out
 573   // correct (initial fp value, unextended sp, ...)
 574   info->set_initial_info((intptr_t) array->sender().initial_deoptimization_info());
 575 
 576   if (array->frames() > 1) {
 577     if (VerifyStack && TraceDeoptimization) {
 578       ttyLocker ttyl;
 579       tty->print_cr("Deoptimizing method containing inlining");
 580     }
 581   }
 582 
 583   array->set_unroll_block(info);
 584   return info;
 585 }

 587 // Called to cleanup deoptimization data structures in normal case
 588 // after unpacking to stack and when stack overflow error occurs
 589 void Deoptimization::cleanup_deopt_info(JavaThread *thread,
 590                                         vframeArray *array) {
 591 
 592   // Get array if coming from exception
 593   if (array == NULL) {
 594     array = thread->vframe_array_head();
 595   }
 596   thread->set_vframe_array_head(NULL);
 597 
 598   // Free the previous UnrollBlock
 599   vframeArray* old_array = thread->vframe_array_last();
 600   thread->set_vframe_array_last(array);
 601 
 602   if (old_array != NULL) {
 603     UnrollBlock* old_info = old_array->unroll_block();
 604     old_array->set_unroll_block(NULL);
 605     delete old_info;
 606     delete old_array;
 607   }
 608 
 609   // Deallocate any resource creating in this routine and any ResourceObjs allocated
 610   // inside the vframeArray (StackValueCollections)
 611 
 612   delete thread->deopt_mark();
 613   thread->set_deopt_mark(NULL);
 614   thread->set_deopt_compiled_method(NULL);
 615 
 616 
 617   if (JvmtiExport::can_pop_frame()) {
 618 #ifndef CC_INTERP
 619     // Regardless of whether we entered this routine with the pending
 620     // popframe condition bit set, we should always clear it now
 621     thread->clear_popframe_condition();
 622 #else
 623     // C++ interpreter will clear has_pending_popframe when it enters
 624     // with method_resume. For deopt_resume2 we clear it now.
 625     if (thread->popframe_forcing_deopt_reexecution())
 626         thread->clear_popframe_condition();
 627 #endif /* CC_INTERP */
 628   }
 629 
 630   // unpack_frames() is called at the end of the deoptimization handler
 631   // and (in C2) at the end of the uncommon trap handler. Note this fact
 632   // so that an asynchronous stack walker can work again. This counter is
 633   // incremented at the beginning of fetch_unroll_info() and (in C2) at
 634   // the beginning of uncommon_trap().
 635   thread->dec_in_deopt_handler();
 636 }

 638 // Moved from cpu directories because none of the cpus has callee save values.
 639 // If a cpu implements callee save values, move this to deoptimization_<cpu>.cpp.
 640 void Deoptimization::unwind_callee_save_values(frame* f, vframeArray* vframe_array) {
 641 
 642   // This code is sort of the equivalent of C2IAdapter::setup_stack_frame back in
 643   // the days we had adapter frames. When we deoptimize a situation where a
 644   // compiled caller calls a compiled caller will have registers it expects
 645   // to survive the call to the callee. If we deoptimize the callee the only
 646   // way we can restore these registers is to have the oldest interpreter
 647   // frame that we create restore these values. That is what this routine
 648   // will accomplish.
 649 
 650   // At the moment we have modified c2 to not have any callee save registers
 651   // so this problem does not exist and this routine is just a place holder.
 652 
 653   assert(f->is_interpreted_frame(), "must be interpreted");
 654 }
 655 
 656 // Return BasicType of value being returned
 657 JRT_LEAF(BasicType, Deoptimization::unpack_frames(JavaThread* thread, int exec_mode))
 658 
 659   // We are already active in the special DeoptResourceMark any ResourceObj's we
 660   // allocate will be freed at the end of the routine.
 661 
 662   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 663   // but makes the entry a little slower. There is however a little dance we have to
 664   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 665   ResetNoHandleMark rnhm; // No-op in release/product versions
 666   HandleMark hm;
 667 
 668   frame stub_frame = thread->last_frame();
 669 
 670   // Since the frame to unpack is the top frame of this thread, the vframe_array_head
 671   // must point to the vframeArray for the unpack frame.
 672   vframeArray* array = thread->vframe_array_head();
 673 
 674 #ifndef PRODUCT
 675   if (TraceDeoptimization) {
 676     ttyLocker ttyl;
 677     tty->print_cr("DEOPT UNPACKING thread " INTPTR_FORMAT " vframeArray " INTPTR_FORMAT " mode %d",
 678                   p2i(thread), p2i(array), exec_mode);
 679   }
 680 #endif
 681   Events::log_deopt_message(thread, "DEOPT UNPACKING pc=" INTPTR_FORMAT " sp=" INTPTR_FORMAT " mode %d",
 682               p2i(stub_frame.pc()), p2i(stub_frame.sp()), exec_mode);
 683 
 684   UnrollBlock* info = array->unroll_block();
 685 
 686   // Unpack the interpreter frames and any adapter frame (c2 only) we might create.
 687   array->unpack_to_stack(stub_frame, exec_mode, info->caller_actual_parameters());
 688 
 689   BasicType bt = info->return_type();
 690 
 691   // If we have an exception pending, claim that the return type is an oop
 692   // so the deopt_blob does not overwrite the exception_oop.
 693 
 694   if (exec_mode == Unpack_exception)
 695     bt = T_OBJECT;
 696 
 697   // Cleanup thread deopt data
 698   cleanup_deopt_info(thread, array);
 699 
 700 #ifndef PRODUCT
 701   if (VerifyStack) {
 702     ResourceMark res_mark;
 703     // Clear pending exception to not break verification code (restored afterwards)
 704     PRESERVE_EXCEPTION_MARK;
 705 
 706     thread->validate_frame_layout();
 707 
 708     // Verify that the just-unpacked frames match the interpreter's
 709     // notions of expression stack and locals
 710     vframeArray* cur_array = thread->vframe_array_last();
 711     RegisterMap rm(thread, false);
 712     rm.set_include_argument_oops(false);
 713     bool is_top_frame = true;
 714     int callee_size_of_parameters = 0;
 715     int callee_max_locals = 0;
 716     for (int i = 0; i < cur_array->frames(); i++) {
 717       vframeArrayElement* el = cur_array->element(i);
 718       frame* iframe = el->iframe();
 719       guarantee(iframe->is_interpreted_frame(), "Wrong frame type");
 720 
 721       // Get the oop map for this bci
 722       InterpreterOopMap mask;
 723       int cur_invoke_parameter_size = 0;
 724       bool try_next_mask = false;
 725       int next_mask_expression_stack_size = -1;
 726       int top_frame_expression_stack_adjustment = 0;
 727       methodHandle mh(thread, iframe->interpreter_frame_method());
 728       OopMapCache::compute_one_oop_map(mh, iframe->interpreter_frame_bci(), &mask);
 729       BytecodeStream str(mh, iframe->interpreter_frame_bci());
 730       int max_bci = mh->code_size();
 731       // Get to the next bytecode if possible
 732       assert(str.bci() < max_bci, "bci in interpreter frame out of bounds");
 733       // Check to see if we can grab the number of outgoing arguments
 734       // at an uncommon trap for an invoke (where the compiler
 735       // generates debug info before the invoke has executed)
 736       Bytecodes::Code cur_code = str.next();
 737       if (Bytecodes::is_invoke(cur_code)) {
 738         Bytecode_invoke invoke(mh, iframe->interpreter_frame_bci());
 739         cur_invoke_parameter_size = invoke.size_of_parameters();
 740         if (i != 0 && !invoke.is_invokedynamic() && MethodHandles::has_member_arg(invoke.klass(), invoke.name())) {
 741           callee_size_of_parameters++;
 742         }
 743       }
 744       if (str.bci() < max_bci) {
 745         Bytecodes::Code next_code = str.next();
 746         if (next_code >= 0) {
 747           // The interpreter oop map generator reports results before
 748           // the current bytecode has executed except in the case of
 749           // calls. It seems to be hard to tell whether the compiler
 750           // has emitted debug information matching the "state before"
 751           // a given bytecode or the state after, so we try both
 752           if (!Bytecodes::is_invoke(cur_code) && cur_code != Bytecodes::_athrow) {
 753             // Get expression stack size for the next bytecode
 754             InterpreterOopMap next_mask;
 755             OopMapCache::compute_one_oop_map(mh, str.bci(), &next_mask);
 756             next_mask_expression_stack_size = next_mask.expression_stack_size();
 757             if (Bytecodes::is_invoke(next_code)) {
 758               Bytecode_invoke invoke(mh, str.bci());
 759               next_mask_expression_stack_size += invoke.size_of_parameters();
 760             }
 761             // Need to subtract off the size of the result type of
 762             // the bytecode because this is not described in the
 763             // debug info but returned to the interpreter in the TOS
 764             // caching register
 765             BasicType bytecode_result_type = Bytecodes::result_type(cur_code);
 766             if (bytecode_result_type != T_ILLEGAL) {
 767               top_frame_expression_stack_adjustment = type2size[bytecode_result_type];
 768             }
 769             assert(top_frame_expression_stack_adjustment >= 0, "stack adjustment must be positive");
 770             try_next_mask = true;
 771           }
 772         }
 773       }
 774 
 775       // Verify stack depth and oops in frame
 776       // This assertion may be dependent on the platform we're running on and may need modification (tested on x86 and sparc)
 777       if (!(
 778             /* SPARC */
 779             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_size_of_parameters) ||
 780             /* x86 */
 781             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_max_locals) ||
 782             (try_next_mask &&
 783              (iframe->interpreter_frame_expression_stack_size() == (next_mask_expression_stack_size -
 784                                                                     top_frame_expression_stack_adjustment))) ||
 785             (is_top_frame && (exec_mode == Unpack_exception) && iframe->interpreter_frame_expression_stack_size() == 0) ||
 786             (is_top_frame && (exec_mode == Unpack_uncommon_trap || exec_mode == Unpack_reexecute || el->should_reexecute()) &&
 787              (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + cur_invoke_parameter_size))
 788             )) {
 789         {
 790           ttyLocker ttyl;
 791 
 792           // Print out some information that will help us debug the problem
 793           tty->print_cr("Wrong number of expression stack elements during deoptimization");
 794           tty->print_cr("  Error occurred while verifying frame %d (0..%d, 0 is topmost)", i, cur_array->frames() - 1);
 795           tty->print_cr("  Fabricated interpreter frame had %d expression stack elements",
 796                         iframe->interpreter_frame_expression_stack_size());
 797           tty->print_cr("  Interpreter oop map had %d expression stack elements", mask.expression_stack_size());
 798           tty->print_cr("  try_next_mask = %d", try_next_mask);
 799           tty->print_cr("  next_mask_expression_stack_size = %d", next_mask_expression_stack_size);
 800           tty->print_cr("  callee_size_of_parameters = %d", callee_size_of_parameters);
 801           tty->print_cr("  callee_max_locals = %d", callee_max_locals);
 802           tty->print_cr("  top_frame_expression_stack_adjustment = %d", top_frame_expression_stack_adjustment);
 803           tty->print_cr("  exec_mode = %d", exec_mode);
 804           tty->print_cr("  cur_invoke_parameter_size = %d", cur_invoke_parameter_size);
 805           tty->print_cr("  Thread = " INTPTR_FORMAT ", thread ID = %d", p2i(thread), thread->osthread()->thread_id());
 806           tty->print_cr("  Interpreted frames:");
 807           for (int k = 0; k < cur_array->frames(); k++) {
 808             vframeArrayElement* el = cur_array->element(k);
 809             tty->print_cr("    %s (bci %d)", el->method()->name_and_sig_as_C_string(), el->bci());
 810           }
 811           cur_array->print_on_2(tty);
 812         } // release tty lock before calling guarantee
 813         guarantee(false, "wrong number of expression stack elements during deopt");
 814       }
 815       VerifyOopClosure verify;
 816       iframe->oops_interpreted_do(&verify, &rm, false);
 817       callee_size_of_parameters = mh->size_of_parameters();
 818       callee_max_locals = mh->max_locals();
 819       is_top_frame = false;
 820     }
 821   }
 822 #endif /* !PRODUCT */
 823 
 824 
 825   return bt;
 826 JRT_END

 910     }

 915     if (sizeof(PrimitiveType) > sizeof(jint)) {
 916       jlong value = (jlong)raw_value;
 917       return lookup(value);
 918     }
 919     PrimitiveType value = (PrimitiveType)*((jint*)&raw_value);
 920     return lookup(value);
 921   }

 995 bool Deoptimization::realloc_objects(JavaThread* thread, frame* fr, RegisterMap* reg_map, GrowableArray<ScopeValue*>* objects, TRAPS) {
 996   Handle pending_exception(THREAD, thread->pending_exception());
 997   const char* exception_file = thread->exception_file();
 998   int exception_line = thread->exception_line();
 999   thread->clear_pending_exception();

1002

1004     assert(objects->at(i)->is_object(), "invalid debug information");
1005     ObjectValue* sv = (ObjectValue*) objects->at(i);
1006 
1007     Klass* k = java_lang_Class::as_Klass(sv->klass()->as_ConstantOopReadValue()->value()());
1008     oop obj = NULL;
1009 
1010     if (k->is_instance_klass()) {
1011 #if INCLUDE_JVMCI || INCLUDE_AOT
1012       CompiledMethod* cm = fr->cb()->as_compiled_method_or_null();
1013       if (cm->is_compiled_by_jvmci() && sv->is_auto_box()) {
1014         AutoBoxObjectValue* abv = (AutoBoxObjectValue*) sv;
1015         obj = get_cached_box(abv, fr, reg_map, THREAD);
1016         if (obj != NULL) {
1017           // Set the flag to indicate the box came from a cache, so that we can skip the field reassignment for it.
1018           abv->set_cached(true);
1019         }
1020       }
1021 #endif // INCLUDE_JVMCI || INCLUDE_AOT
1022       InstanceKlass* ik = InstanceKlass::cast(k);
1023       if (obj == NULL) {
1024         obj = ik->allocate_instance(THREAD);
1025       }
1026     } else if (k->is_valueArray_klass()) {
1027       ValueArrayKlass* ak = ValueArrayKlass::cast(k);
1028       // Value type array must be zeroed because not all memory is reassigned
1029       obj = ak->allocate(sv->field_size(), THREAD);
1030     } else if (k->is_typeArray_klass()) {
1031       TypeArrayKlass* ak = TypeArrayKlass::cast(k);
1032       assert(sv->field_size() % type2size[ak->element_type()] == 0, "non-integral array length");
1033       int len = sv->field_size() / type2size[ak->element_type()];

1037       obj = ak->allocate(sv->field_size(), THREAD);
1038     }

1041       failures = true;
1042     }

1047     sv->set_value(obj);
1048   }

1051     THROW_OOP_(Universe::out_of_memory_error_realloc_objects(), failures);
1052   } else if (pending_exception.not_null()) {
1053     thread->set_pending_exception(pending_exception(), exception_file, exception_line);
1054   }
1055 
1056   return failures;

1197 // Restore fields of an eliminated instance object using the same field order
1198 // returned by HotSpotResolvedObjectTypeImpl.getInstanceFields(true)
1199 static int reassign_fields_by_klass(InstanceKlass* klass, frame* fr, RegisterMap* reg_map, ObjectValue* sv, int svIndex, oop obj, bool skip_internal, int base_offset, TRAPS) {
1200 
1201   GrowableArray<ReassignedField>* fields = new GrowableArray<ReassignedField>();
1202   InstanceKlass* ik = klass;
1203   while (ik != NULL) {
1204     for (AllFieldStream fs(ik); !fs.done(); fs.next()) {
1205       if (!fs.access_flags().is_static() && (!skip_internal || !fs.access_flags().is_internal())) {
1206         ReassignedField field;
1207         field._offset = fs.offset();
1208         field._type = Signature::basic_type(fs.signature());
1209         if (field._type == T_VALUETYPE) {
1210           field._type = T_OBJECT;
1211         }
1212         if (fs.is_flattened()) {
1213           // Resolve klass of flattened value type field
1214           Klass* vk = klass->get_value_field_klass(fs.index());
1215           field._klass = ValueKlass::cast(vk);
1216           field._type = T_VALUETYPE;
1217         }
1218         fields->append(field);
1219       }
1220     }
1221     ik = ik->superklass();
1222   }
1223   fields->sort(compare);
1224   for (int i = 0; i < fields->length(); i++) {
1225     intptr_t val;
1226     ScopeValue* scope_field = sv->field_at(svIndex);
1227     StackValue* value = StackValue::create_stack_value(fr, reg_map, scope_field);
1228     int offset = base_offset + fields->at(i)._offset;
1229     BasicType type = fields->at(i)._type;
1230     switch (type) {
1231       case T_OBJECT:
1232       case T_ARRAY:
1233         assert(value->type() == T_OBJECT, "Agreement.");
1234         obj->obj_field_put(offset, value->get_obj()());
1235         break;
1236 
1237       case T_VALUETYPE: {
1238         // Recursively re-assign flattened value type fields
1239         InstanceKlass* vk = fields->at(i)._klass;
1240         assert(vk != NULL, "must be resolved");
1241         offset -= ValueKlass::cast(vk)->first_field_offset(); // Adjust offset to omit oop header
1242         svIndex = reassign_fields_by_klass(vk, fr, reg_map, sv, svIndex, obj, skip_internal, offset, CHECK_0);
1243         continue; // Continue because we don't need to increment svIndex
1244       }
1245 
1246       // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1247       case T_INT: case T_FLOAT: { // 4 bytes.
1248         assert(value->type() == T_INT, "Agreement.");
1249         bool big_value = false;
1250         if (i+1 < fields->length() && fields->at(i+1)._type == T_INT) {
1251           if (scope_field->is_location()) {
1252             Location::Type type = ((LocationValue*) scope_field)->location().type();
1253             if (type == Location::dbl || type == Location::lng) {
1254               big_value = true;
1255             }
1256           }
1257           if (scope_field->is_constant_int()) {
1258             ScopeValue* next_scope_field = sv->field_at(svIndex + 1);
1259             if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1260               big_value = true;
1261             }
1262           }
1263         }
1264 
1265         if (big_value) {
1266           i++;
1267           assert(i < fields->length(), "second T_INT field needed");
1268           assert(fields->at(i)._type == T_INT, "T_INT field needed");
1269         } else {
1270           val = value->get_int();
1271           obj->int_field_put(offset, (jint)*((jint*)&val));
1272           break;
1273         }
1274       }
1275         /* no break */
1276 
1277       case T_LONG: case T_DOUBLE: {
1278         assert(value->type() == T_INT, "Agreement.");
1279         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++svIndex));
1280 #ifdef _LP64
1281         jlong res = (jlong)low->get_int();
1282 #else
1283 #ifdef SPARC
1284         // For SPARC we have to swap high and low words.
1285         jlong res = jlong_from((jint)low->get_int(), (jint)value->get_int());
1286 #else
1287         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1288 #endif //SPARC
1289 #endif
1290         obj->long_field_put(offset, res);
1291         break;
1292       }
1293 
1294       case T_SHORT:
1295         assert(value->type() == T_INT, "Agreement.");
1296         val = value->get_int();
1297         obj->short_field_put(offset, (jshort)*((jint*)&val));
1298         break;
1299 
1300       case T_CHAR:
1301         assert(value->type() == T_INT, "Agreement.");
1302         val = value->get_int();
1303         obj->char_field_put(offset, (jchar)*((jint*)&val));
1304         break;
1305 
1306       case T_BYTE:
1307         assert(value->type() == T_INT, "Agreement.");
1308         val = value->get_int();
1309         obj->byte_field_put(offset, (jbyte)*((jint*)&val));
1310         break;
1311 
1312       case T_BOOLEAN:
1313         assert(value->type() == T_INT, "Agreement.");
1314         val = value->get_int();
1315         obj->bool_field_put(offset, (jboolean)*((jint*)&val));
1316         break;
1317 
1318       default:
1319         ShouldNotReachHere();
1320     }
1321     svIndex++;
1322   }
1323   return svIndex;
1324 }

1375 // relock objects for which synchronization was eliminated
1376 void Deoptimization::relock_objects(GrowableArray<MonitorInfo*>* monitors, JavaThread* thread, bool realloc_failures) {
1377   for (int i = 0; i < monitors->length(); i++) {
1378     MonitorInfo* mon_info = monitors->at(i);
1379     if (mon_info->eliminated()) {
1380       assert(!mon_info->owner_is_scalar_replaced() || realloc_failures, "reallocation was missed");
1381       if (!mon_info->owner_is_scalar_replaced()) {
1382         Handle obj(thread, mon_info->owner());
1383         markWord mark = obj->mark();
1384         if (UseBiasedLocking && mark.has_bias_pattern()) {
1385           // New allocated objects may have the mark set to anonymously biased.
1386           // Also the deoptimized method may called methods with synchronization
1387           // where the thread-local object is bias locked to the current thread.
1388           assert(mark.is_biased_anonymously() ||
1389                  mark.biased_locker() == thread, "should be locked to current thread");
1390           // Reset mark word to unbiased prototype.
1391           markWord unbiased_prototype = markWord::prototype().set_age(mark.age());
1392           obj->set_mark(unbiased_prototype);
1393         }
1394         BasicLock* lock = mon_info->lock();
1395         ObjectSynchronizer::enter(obj, lock, thread);
1396         assert(mon_info->owner()->is_locked(), "object must be locked now");
1397       }
1398     }
1399   }
1400 }

1528 static void get_monitors_from_stack(GrowableArray<Handle>* objects_to_revoke, JavaThread* thread, frame fr, RegisterMap* map) {
1529   // Unfortunately we don't have a RegisterMap available in most of
1530   // the places we want to call this routine so we need to walk the
1531   // stack again to update the register map.
1532   if (map == NULL || !map->update_map()) {
1533     StackFrameStream sfs(thread, true);
1534     bool found = false;
1535     while (!found && !sfs.is_done()) {
1536       frame* cur = sfs.current();
1537       sfs.next();
1538       found = cur->id() == fr.id();
1539     }
1540     assert(found, "frame to be deoptimized not found on target thread's stack");
1541     map = sfs.register_map();
1542   }

1545   compiledVFrame* cvf = compiledVFrame::cast(vf);
1546   // Revoke monitors' biases in all scopes
1547   while (!cvf->is_top()) {
1548     collect_monitors(cvf, objects_to_revoke);
1549     cvf = compiledVFrame::cast(cvf->sender());
1550   }
1551   collect_monitors(cvf, objects_to_revoke);

1898 #if INCLUDE_JVMCI

1911           unresolved = true;
1912           if (xtty != NULL)
1913             xtty->print(" unresolved='1'");

1915           class_name = constants->symbol_at(unloaded_class_index);
1916         }
1917         if (xtty != NULL)
1918           xtty->name(class_name);
1919       }

1922         // This is the deopt count for the current reason, any previous
1923         // reasons or recompiles seen at this point.
1924         int dcnt = trap_mdo->trap_count(reason);
1925         if (dcnt != 0)
1926           xtty->print(" count='%d'", dcnt);

1928         int dos = (pdata == NULL)? 0: pdata->trap_state();

1933             if (recnt2 != 0)
1934               xtty->print(" recompiles2='%d'", recnt2);
1935           }
1936         }
1937       }
1938       if (xtty != NULL) {
1939         xtty->stamp();
1940         xtty->end_head();
1941       }
1942       if (TraceDeoptimization) {  // make noise on the tty
1943         tty->print("Uncommon trap occurred in");
1944         nm->method()->print_short_name(tty);
1945         tty->print(" compiler=%s compile_id=%d", nm->compiler_name(), nm->compile_id());
1946 #if INCLUDE_JVMCI
1947         if (nm->is_nmethod()) {
1948           const char* installed_code_name = nm->as_nmethod()->jvmci_name();

1956                    os::current_thread_id(),

1958                    trap_action_name(action),

1962 #endif

1966           class_name->print_symbol_on(tty);
1967         }

1969       }
1970       if (xtty != NULL) {
1971         // Log the precise location of the trap.
1972         for (ScopeDesc* sd = trap_scope; ; sd = sd->sender()) {
1973           xtty->begin_elem("jvms bci='%d'", sd->bci());
1974           xtty->method(sd->method());
1975           xtty->end_elem();
1976           if (sd->is_top())  break;
1977         }
1978         xtty->tail("uncommon_trap");
1979       }
1980     }
1981     // (End diagnostic printout.)
1982 
1983     // Load class if necessary
1984     if (unloaded_class_index >= 0) {
1985       constantPoolHandle constants(THREAD, trap_method->constants());
1986       load_class_by_index(constants, unloaded_class_index);
1987     }
1988 
1989     // Flush the nmethod if necessary and desirable.
1990     //
1991     // We need to avoid situations where we are re-flushing the nmethod
1992     // because of a hot deoptimization site.  Repeated flushes at the same
1993     // point need to be detected by the compiler and avoided.  If the compiler
1994     // cannot avoid them (or has a bug and "refuses" to avoid them), this
1995     // module must take measures to avoid an infinite cycle of recompilation
1996     // and deoptimization.  There are several such measures:
1997     //
1998     //   1. If a recompilation is ordered a second time at some site X
1999     //   and for the same reason R, the action is adjusted to 'reinterpret',
2000     //   to give the interpreter time to exercise the method more thoroughly.
2001     //   If this happens, the method's overflow_recompile_count is incremented.
2002     //
2003     //   2. If the compiler fails to reduce the deoptimization rate, then
2004     //   the method's overflow_recompile_count will begin to exceed the set
2005     //   limit PerBytecodeRecompilationCutoff.  If this happens, the action
2006     //   is adjusted to 'make_not_compilable', and the method is abandoned
2007     //   to the interpreter.  This is a performance hit for hot methods,
2008     //   but is better than a disastrous infinite cycle of recompilations.
2009     //   (Actually, only the method containing the site X is abandoned.)
2010     //
2011     //   3. In parallel with the previous measures, if the total number of
2012     //   recompilations of a method exceeds the much larger set limit
2013     //   PerMethodRecompilationCutoff, the method is abandoned.
2014     //   This should only happen if the method is very large and has
2015     //   many "lukewarm" deoptimizations.  The code which enforces this
2016     //   limit is elsewhere (class nmethod, class Method).
2017     //
2018     // Note that the per-BCI 'is_recompiled' bit gives the compiler one chance
2019     // to recompile at each bytecode independently of the per-BCI cutoff.
2020     //
2021     // The decision to update code is up to the compiler, and is encoded
2022     // in the Action_xxx code.  If the compiler requests Action_none
2023     // no trap state is changed, no compiled code is changed, and the
2024     // computation suffers along in the interpreter.
2025     //
2026     // The other action codes specify various tactics for decompilation
2027     // and recompilation.  Action_maybe_recompile is the loosest, and
2028     // allows the compiled code to stay around until enough traps are seen,
2029     // and until the compiler gets around to recompiling the trapping method.
2030     //
2031     // The other actions cause immediate removal of the present code.
2032 
2033     // Traps caused by injected profile shouldn't pollute trap counts.
2034     bool injected_profile_trap = trap_method->has_injected_profile() &&
2035                                  (reason == Reason_intrinsic || reason == Reason_unreached);
2036 
2037     bool update_trap_state = (reason != Reason_tenured) && !injected_profile_trap;
2038     bool make_not_entrant = false;
2039     bool make_not_compilable = false;
2040     bool reprofile = false;
2041     switch (action) {
2042     case Action_none:
2043       // Keep the old code.
2044       update_trap_state = false;
2045       break;
2046     case Action_maybe_recompile:
2047       // Do not need to invalidate the present code, but we can
2048       // initiate another
2049       // Start compiler without (necessarily) invalidating the nmethod.
2050       // The system will tolerate the old code, but new code should be
2051       // generated when possible.
2052       break;
2053     case Action_reinterpret:
2054       // Go back into the interpreter for a while, and then consider
2055       // recompiling form scratch.
2056       make_not_entrant = true;
2057       // Reset invocation counter for outer most method.
2058       // This will allow the interpreter to exercise the bytecodes
2059       // for a while before recompiling.
2060       // By contrast, Action_make_not_entrant is immediate.
2061       //
2062       // Note that the compiler will track null_check, null_assert,
2063       // range_check, and class_check events and log them as if they
2064       // had been traps taken from compiled code.  This will update
2065       // the MDO trap history so that the next compilation will
2066       // properly detect hot trap sites.
2067       reprofile = true;
2068       break;
2069     case Action_make_not_entrant:
2070       // Request immediate recompilation, and get rid of the old code.
2071       // Make them not entrant, so next time they are called they get
2072       // recompiled.  Unloaded classes are loaded now so recompile before next
2073       // time they are called.  Same for uninitialized.  The interpreter will
2074       // link the missing class, if any.
2075       make_not_entrant = true;
2076       break;
2077     case Action_make_not_compilable:
2078       // Give up on compiling this method at all.
2079       make_not_entrant = true;
2080       make_not_compilable = true;
2081       break;
2082     default:
2083       ShouldNotReachHere();
2084     }
2085 
2086     // Setting +ProfileTraps fixes the following, on all platforms:
2087     // 4852688: ProfileInterpreter is off by default for ia64.  The result is
2088     // infinite heroic-opt-uncommon-trap/deopt/recompile cycles, since the
2089     // recompile relies on a MethodData* to record heroic opt failures.
2090 
2091     // Whether the interpreter is producing MDO data or not, we also need
2092     // to use the MDO to detect hot deoptimization points and control
2093     // aggressive optimization.
2094     bool inc_recompile_count = false;
2095     ProfileData* pdata = NULL;
2096     if (ProfileTraps && !is_client_compilation_mode_vm() && update_trap_state && trap_mdo != NULL) {
2097       assert(trap_mdo == get_method_data(thread, profiled_method, false), "sanity");
2098       uint this_trap_count = 0;
2099       bool maybe_prior_trap = false;
2100       bool maybe_prior_recompile = false;
2101       pdata = query_update_method_data(trap_mdo, trap_bci, reason, true,
2102 #if INCLUDE_JVMCI
2103                                    nm->is_compiled_by_jvmci() && nm->is_osr_method(),
2104 #endif
2105                                    nm->method(),
2106                                    //outputs:
2107                                    this_trap_count,
2108                                    maybe_prior_trap,
2109                                    maybe_prior_recompile);
2110       // Because the interpreter also counts null, div0, range, and class
2111       // checks, these traps from compiled code are double-counted.
2112       // This is harmless; it just means that the PerXTrapLimit values
2113       // are in effect a little smaller than they look.
2114 
2115       DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2116       if (per_bc_reason != Reason_none) {
2117         // Now take action based on the partially known per-BCI history.
2118         if (maybe_prior_trap
2119             && this_trap_count >= (uint)PerBytecodeTrapLimit) {
2120           // If there are too many traps at this BCI, force a recompile.
2121           // This will allow the compiler to see the limit overflow, and
2122           // take corrective action, if possible.  The compiler generally
2123           // does not use the exact PerBytecodeTrapLimit value, but instead
2124           // changes its tactics if it sees any traps at all.  This provides
2125           // a little hysteresis, delaying a recompile until a trap happens
2126           // several times.
2127           //
2128           // Actually, since there is only one bit of counter per BCI,
2129           // the possible per-BCI counts are {0,1,(per-method count)}.
2130           // This produces accurate results if in fact there is only
2131           // one hot trap site, but begins to get fuzzy if there are
2132           // many sites.  For example, if there are ten sites each
2133           // trapping two or more times, they each get the blame for
2134           // all of their traps.
2135           make_not_entrant = true;
2136         }
2137 
2138         // Detect repeated recompilation at the same BCI, and enforce a limit.
2139         if (make_not_entrant && maybe_prior_recompile) {
2140           // More than one recompile at this point.
2141           inc_recompile_count = maybe_prior_trap;
2142         }
2143       } else {
2144         // For reasons which are not recorded per-bytecode, we simply
2145         // force recompiles unconditionally.
2146         // (Note that PerMethodRecompilationCutoff is enforced elsewhere.)
2147         make_not_entrant = true;
2148       }
2149 
2150       // Go back to the compiler if there are too many traps in this method.
2151       if (this_trap_count >= per_method_trap_limit(reason)) {
2152         // If there are too many traps in this method, force a recompile.
2153         // This will allow the compiler to see the limit overflow, and
2154         // take corrective action, if possible.
2155         // (This condition is an unlikely backstop only, because the
2156         // PerBytecodeTrapLimit is more likely to take effect first,
2157         // if it is applicable.)
2158         make_not_entrant = true;
2159       }
2160 
2161       // Here's more hysteresis:  If there has been a recompile at
2162       // this trap point already, run the method in the interpreter
2163       // for a while to exercise it more thoroughly.
2164       if (make_not_entrant && maybe_prior_recompile && maybe_prior_trap) {
2165         reprofile = true;
2166       }
2167     }
2168 
2169     // Take requested actions on the method:
2170 
2171     // Recompile
2172     if (make_not_entrant) {
2173       if (!nm->make_not_entrant()) {
2174         return; // the call did not change nmethod's state
2175       }
2176 
2177       if (pdata != NULL) {
2178         // Record the recompilation event, if any.
2179         int tstate0 = pdata->trap_state();
2180         int tstate1 = trap_state_set_recompiled(tstate0, true);
2181         if (tstate1 != tstate0)
2182           pdata->set_trap_state(tstate1);
2183       }
2184 
2185 #if INCLUDE_RTM_OPT
2186       // Restart collecting RTM locking abort statistic if the method
2187       // is recompiled for a reason other than RTM state change.
2188       // Assume that in new recompiled code the statistic could be different,
2189       // for example, due to different inlining.
2190       if ((reason != Reason_rtm_state_change) && (trap_mdo != NULL) &&
2191           UseRTMDeopt && (nm->as_nmethod()->rtm_state() != ProfileRTM)) {
2192         trap_mdo->atomic_set_rtm_state(ProfileRTM);
2193       }
2194 #endif
2195       // For code aging we count traps separately here, using make_not_entrant()
2196       // as a guard against simultaneous deopts in multiple threads.
2197       if (reason == Reason_tenured && trap_mdo != NULL) {
2198         trap_mdo->inc_tenure_traps();
2199       }
2200     }
2201 
2202     if (inc_recompile_count) {
2203       trap_mdo->inc_overflow_recompile_count();
2204       if ((uint)trap_mdo->overflow_recompile_count() >
2205           (uint)PerBytecodeRecompilationCutoff) {
2206         // Give up on the method containing the bad BCI.
2207         if (trap_method() == nm->method()) {
2208           make_not_compilable = true;
2209         } else {
2210           trap_method->set_not_compilable("overflow_recompile_count > PerBytecodeRecompilationCutoff", CompLevel_full_optimization);
2211           // But give grace to the enclosing nm->method().
2212         }
2213       }
2214     }
2215 
2216     // Reprofile
2217     if (reprofile) {
2218       CompilationPolicy::policy()->reprofile(trap_scope, nm->is_osr_method());
2219     }
2220 
2221     // Give up compiling
2222     if (make_not_compilable && !nm->method()->is_not_compilable(CompLevel_full_optimization)) {
2223       assert(make_not_entrant, "consistent");
2224       nm->method()->set_not_compilable("give up compiling", CompLevel_full_optimization);
2225     }
2226 
2227   } // Free marked resources
2228 
2229 }

2235                                          Deoptimization::DeoptReason reason,
2236                                          bool update_total_trap_count,
2237 #if INCLUDE_JVMCI
2238                                          bool is_osr,
2239 #endif
2240                                          Method* compiled_method,
2241                                          //outputs:
2242                                          uint& ret_this_trap_count,
2243                                          bool& ret_maybe_prior_trap,
2244                                          bool& ret_maybe_prior_recompile) {
2245   bool maybe_prior_trap = false;
2246   bool maybe_prior_recompile = false;
2247   uint this_trap_count = 0;
2248   if (update_total_trap_count) {
2249     uint idx = reason;
2250 #if INCLUDE_JVMCI
2251     if (is_osr) {

2253     }
2254 #endif
2255     uint prior_trap_count = trap_mdo->trap_count(idx);
2256     this_trap_count  = trap_mdo->inc_trap_count(idx);
2257 
2258     // If the runtime cannot find a place to store trap history,
2259     // it is estimated based on the general condition of the method.
2260     // If the method has ever been recompiled, or has ever incurred
2261     // a trap with the present reason , then this BCI is assumed
2262     // (pessimistically) to be the culprit.
2263     maybe_prior_trap      = (prior_trap_count != 0);
2264     maybe_prior_recompile = (trap_mdo->decompile_count() != 0);
2265   }
2266   ProfileData* pdata = NULL;
2267 
2268 
2269   // For reasons which are recorded per bytecode, we check per-BCI data.
2270   DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2271   assert(per_bc_reason != Reason_none || update_total_trap_count, "must be");
2272   if (per_bc_reason != Reason_none) {
2273     // Find the profile data for this BCI.  If there isn't one,
2274     // try to allocate one from the MDO's set of spares.
2275     // This will let us detect a repeated trap at this point.
2276     pdata = trap_mdo->allocate_bci_to_data(trap_bci, reason_is_speculate(reason) ? compiled_method : NULL);
2277 
2278     if (pdata != NULL) {
2279       if (reason_is_speculate(reason) && !pdata->is_SpeculativeTrapData()) {
2280         if (LogCompilation && xtty != NULL) {
2281           ttyLocker ttyl;
2282           // no more room for speculative traps in this MDO
2283           xtty->elem("speculative_traps_oom");
2284         }
2285       }
2286       // Query the trap state of this profile datum.
2287       int tstate0 = pdata->trap_state();
2288       if (!trap_state_has_reason(tstate0, per_bc_reason))
2289         maybe_prior_trap = false;
2290       if (!trap_state_is_recompiled(tstate0))
2291         maybe_prior_recompile = false;
2292 
2293       // Update the trap state of this profile datum.
2294       int tstate1 = tstate0;
2295       // Record the reason.
2296       tstate1 = trap_state_add_reason(tstate1, per_bc_reason);
2297       // Store the updated state on the MDO, for next time.
2298       if (tstate1 != tstate0)
2299         pdata->set_trap_state(tstate1);
2300     } else {
2301       if (LogCompilation && xtty != NULL) {
2302         ttyLocker ttyl;
2303         // Missing MDP?  Leave a small complaint in the log.
2304         xtty->elem("missing_mdp bci='%d'", trap_bci);
2305       }
2306     }
2307   }
2308 
2309   // Return results:
2310   ret_this_trap_count = this_trap_count;
2311   ret_maybe_prior_trap = maybe_prior_trap;
2312   ret_maybe_prior_recompile = maybe_prior_recompile;
2313   return pdata;
2314 }

2350 // Local derived constants.
2351 // Further breakdown of DataLayout::trap_state, as promised by DataLayout.
2352 const int DS_REASON_MASK   = ((uint)DataLayout::trap_mask) >> 1;
2353 const int DS_RECOMPILE_BIT = DataLayout::trap_mask - DS_REASON_MASK;
2354 
2355 //---------------------------trap_state_reason---------------------------------
2356 Deoptimization::DeoptReason
2357 Deoptimization::trap_state_reason(int trap_state) {
2358   // This assert provides the link between the width of DataLayout::trap_bits
2359   // and the encoding of "recorded" reasons.  It ensures there are enough
2360   // bits to store all needed reasons in the per-BCI MDO profile.
2361   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2362   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2363   trap_state -= recompile_bit;
2364   if (trap_state == DS_REASON_MASK) {
2365     return Reason_many;
2366   } else {
2367     assert((int)Reason_none == 0, "state=0 => Reason_none");
2368     return (DeoptReason)trap_state;
2369   }
2370 }
2371 //-------------------------trap_state_has_reason-------------------------------
2372 int Deoptimization::trap_state_has_reason(int trap_state, int reason) {
2373   assert(reason_is_recorded_per_bytecode((DeoptReason)reason), "valid reason");
2374   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2375   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2376   trap_state -= recompile_bit;
2377   if (trap_state == DS_REASON_MASK) {
2378     return -1;  // true, unspecifically (bottom of state lattice)
2379   } else if (trap_state == reason) {
2380     return 1;   // true, definitely
2381   } else if (trap_state == 0) {
2382     return 0;   // false, definitely (top of state lattice)
2383   } else {
2384     return 0;   // false, definitely
2385   }
2386 }
2387 //-------------------------trap_state_add_reason-------------------------------
2388 int Deoptimization::trap_state_add_reason(int trap_state, int reason) {
2389   assert(reason_is_recorded_per_bytecode((DeoptReason)reason) || reason == Reason_many, "valid reason");
2390   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2391   trap_state -= recompile_bit;
2392   if (trap_state == DS_REASON_MASK) {
2393     return trap_state + recompile_bit;     // already at state lattice bottom
2394   } else if (trap_state == reason) {
2395     return trap_state + recompile_bit;     // the condition is already true
2396   } else if (trap_state == 0) {
2397     return reason + recompile_bit;          // no condition has yet been true
2398   } else {
2399     return DS_REASON_MASK + recompile_bit;  // fall to state lattice bottom
2400   }
2401 }
2402 //-----------------------trap_state_is_recompiled------------------------------
2403 bool Deoptimization::trap_state_is_recompiled(int trap_state) {
2404   return (trap_state & DS_RECOMPILE_BIT) != 0;
2405 }
2406 //-----------------------trap_state_set_recompiled-----------------------------
2407 int Deoptimization::trap_state_set_recompiled(int trap_state, bool z) {
2408   if (z)  return trap_state |  DS_RECOMPILE_BIT;
2409   else    return trap_state & ~DS_RECOMPILE_BIT;
2410 }

2547 void Deoptimization::gather_statistics(DeoptReason reason, DeoptAction action,
2548                                        Bytecodes::Code bc) {
2549   assert(reason >= 0 && reason < Reason_LIMIT, "oob");
2550   assert(action >= 0 && action < Action_LIMIT, "oob");
2551   _deoptimization_hist[Reason_none][0][0] += 1;  // total
2552   _deoptimization_hist[reason][0][0]      += 1;  // per-reason total
2553   juint* cases = _deoptimization_hist[reason][1+action];
2554   juint* bc_counter_addr = NULL;
2555   juint  bc_counter      = 0;
2556   // Look for an unused counter, or an exact match to this BC.
2557   if (bc != Bytecodes::_illegal) {
2558     for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2559       juint* counter_addr = &cases[bc_case];
2560       juint  counter = *counter_addr;
2561       if ((counter == 0 && bc_counter_addr == NULL)
2562           || (Bytecodes::Code)(counter & LSB_MASK) == bc) {
2563         // this counter is either free or is already devoted to this BC
2564         bc_counter_addr = counter_addr;
2565         bc_counter = counter | bc;
2566       }
2567     }
2568   }
2569   if (bc_counter_addr == NULL) {
2570     // Overflow, or no given bytecode.
2571     bc_counter_addr = &cases[BC_CASE_LIMIT-1];
2572     bc_counter = (*bc_counter_addr & ~LSB_MASK);  // clear LSB
2573   }
2574   *bc_counter_addr = bc_counter + (1 << LSB_BITS);
2575 }
