  57 RegisterMap::RegisterMap(JavaThread *thread, bool update_map) {
  58   _thread         = thread;
  59   _update_map     = update_map;
  60   clear();
  61   debug_only(_update_for_id = NULL;)
  62 #ifndef PRODUCT
  63   for (int i = 0; i < reg_count ; i++ ) _location[i] = NULL;
  64 #endif /* PRODUCT */
  65 }
  66 
  67 RegisterMap::RegisterMap(const RegisterMap* map) {
  68   assert(map != this, "bad initialization parameter");
  69   assert(map != NULL, "RegisterMap must be present");
  70   _thread                = map->thread();
  71   _update_map            = map->update_map();
  72   _include_argument_oops = map->include_argument_oops();
  73   debug_only(_update_for_id = map->_update_for_id;)
  74   pd_initialize_from(map);
  75   if (update_map()) {
  76     for(int i = 0; i < location_valid_size; i++) {
  77       LocationValidType bits = !update_map() ? 0 : map->_location_valid[i];
  78       _location_valid[i] = bits;
  79       // for whichever bits are set, pull in the corresponding map->_location
  80       int j = i*location_valid_type_size;
  81       while (bits != 0) {
  82         if ((bits & 1) != 0) {
  83           assert(0 <= j && j < reg_count, "range check");
  84           _location[j] = map->_location[j];
  85         }
  86         bits >>= 1;
  87         j += 1;
  88       }
  89     }
  90   }
  91 }
  92 
  93 void RegisterMap::clear() {
  94   set_include_argument_oops(true);
  95   if (_update_map) {
  96     for(int i = 0; i < location_valid_size; i++) {
  97       _location_valid[i] = 0;
  98     }
  99     pd_clear();
 100   } else {
 101     pd_initialize();
 102   }
 103 }

 113     if (src != NULL) {
 114 
 115       r->print_on(st);
 116       st->print(" [" INTPTR_FORMAT "] = ", p2i(src));

 118         st->print_cr("<misaligned>");
 119       } else {
 120         st->print_cr(INTPTR_FORMAT, *src);
 121       }
 122     }
 123   }

 131 // This returns the pc that if you were in the debugger you'd see. Not
 132 // the idealized value in the frame object. This undoes the magic conversion
 133 // that happens for deoptimized frames. In addition it makes the value the
 134 // hardware would want to see in the native frame. The only user (at this point)
 135 // is deoptimization. It likely no one else should ever use it.
 136 
 137 address frame::raw_pc() const {
 138   if (is_deoptimized_frame()) {
 139     CompiledMethod* cm = cb()->as_compiled_method_or_null();
 140     if (cm->is_method_handle_return(pc()))
 141       return cm->deopt_mh_handler_begin() - pc_return_offset;
 142     else
 143       return cm->deopt_handler_begin() - pc_return_offset;
 144   } else {
 145     return (pc() - pc_return_offset);
 146   }
 147 }

 149 // Change the pc in a frame object. This does not change the actual pc in
 150 // actual frame. To do that use patch_pc.
 151 //
 152 void frame::set_pc(address   newpc ) {
 153 #ifdef ASSERT
 154   if (_cb != NULL && _cb->is_nmethod()) {
 155     assert(!((nmethod*)_cb)->is_deopt_pc(_pc), "invariant violation");
 156   }
 157 #endif // ASSERT
 158 
 159   // Unsafe to use the is_deoptimzed tester after changing pc
 160   _deopt_state = unknown;
 161   _pc = newpc;
 162   _cb = CodeCache::find_blob_unsafe(_pc);
 163 
 164 }

 220 JavaCallWrapper* frame::entry_frame_call_wrapper_if_safe(JavaThread* thread) const {
 221   JavaCallWrapper** jcw = entry_frame_call_wrapper_addr();
 222   address addr = (address) jcw;
 223 
 224   // addr must be within the usable part of the stack
 225   if (thread->is_in_usable_stack(addr)) {
 226     return *jcw;
 227   }
 228 
 229   return NULL;
 230 }

 232 bool frame::is_entry_frame_valid(JavaThread* thread) const {
 233   // Validate the JavaCallWrapper an entry frame must have
 234   address jcw = (address)entry_frame_call_wrapper();
 235   if (!thread->is_in_stack_range_excl(jcw, (address)fp())) {
 236     return false;
 237   }
 238 
 239   // Validate sp saved in the java frame anchor
 240   JavaFrameAnchor* jfa = entry_frame_call_wrapper()->anchor();
 241   return (jfa->last_Java_sp() > sp());
 242 }

 244 bool frame::should_be_deoptimized() const {
 245   if (_deopt_state == is_deoptimized ||
 246       !is_compiled_frame() ) return false;
 247   assert(_cb != NULL && _cb->is_compiled(), "must be an nmethod");
 248   CompiledMethod* nm = (CompiledMethod *)_cb;
 249   if (TraceDependencies) {
 250     tty->print("checking (%s) ", nm->is_marked_for_deoptimization() ? "true" : "false");
 251     nm->print_value_on(tty);
 252     tty->cr();
 253   }
 254 
 255   if( !nm->is_marked_for_deoptimization() )
 256     return false;
 257 
 258   // If at the return point, then the frame has already been popped, and
 259   // only the return needs to be executed. Don't deoptimize here.
 260   return !nm->is_at_poll_return(pc());
 261 }
 262 
 263 bool frame::can_be_deoptimized() const {
 264   if (!is_compiled_frame()) return false;
 265   CompiledMethod* nm = (CompiledMethod*)_cb;
 266 
 267   if( !nm->can_be_deoptimized() )
 268     return false;
 269 
 270   return !nm->is_at_poll_return(pc());
 271 }

 273 void frame::deoptimize(JavaThread* thread) {
 274   assert(thread->frame_anchor()->has_last_Java_frame() &&
 275          thread->frame_anchor()->walkable(), "must be");
 276   // Schedule deoptimization of an nmethod activation with this frame.
 277   assert(_cb != NULL && _cb->is_compiled(), "must be");
 278 
 279   // If the call site is a MethodHandle call site use the MH deopt
 280   // handler.
 281   CompiledMethod* cm = (CompiledMethod*) _cb;
 282   address deopt = cm->is_method_handle_return(pc()) ?
 283                         cm->deopt_mh_handler_begin() :
 284                         cm->deopt_handler_begin();
 285 
 286   // Save the original pc before we patch in the new one
 287   cm->set_original_pc(this, pc());



















 288   patch_pc(thread, deopt);
 289 
 290 #ifdef ASSERT
 291   {
 292     RegisterMap map(thread, false);
 293     frame check = thread->last_frame();
 294     while (id() != check.id()) {
 295       check = check.sender(&map);
 296     }
 297     assert(check.is_deoptimized_frame(), "missed deopt");
 298   }
 299 #endif // ASSERT
 300 }

 302 frame frame::java_sender() const {
 303   RegisterMap map(JavaThread::current(), false);
 304   frame s;
 305   for (s = sender(&map); !(s.is_java_frame() || s.is_first_frame()); s = s.sender(&map)) ;
 306   guarantee(s.is_java_frame(), "tried to get caller of first java frame");
 307   return s;
 308 }

 310 frame frame::real_sender(RegisterMap* map) const {
 311   frame result = sender(map);
 312   while (result.is_runtime_frame() ||
 313          result.is_ignored_frame()) {
 314     result = result.sender(map);
 315   }
 316   return result;
 317 }

 399 intptr_t* frame::interpreter_frame_expression_stack_at(jint offset) const {
 400   const int i = offset * interpreter_frame_expression_stack_direction();
 401   const int n = i * Interpreter::stackElementWords;
 402   return &(interpreter_frame_expression_stack()[n]);
 403 }
 404 
 405 jint frame::interpreter_frame_expression_stack_size() const {
 406   // Number of elements on the interpreter expression stack
 407   // Callers should span by stackElementWords
 408   int element_size = Interpreter::stackElementWords;
 409   size_t stack_size = 0;
 410   if (frame::interpreter_frame_expression_stack_direction() < 0) {
 411     stack_size = (interpreter_frame_expression_stack() -
 412                   interpreter_frame_tos_address() + 1)/element_size;
 413   } else {
 414     stack_size = (interpreter_frame_tos_address() -
 415                   interpreter_frame_expression_stack() + 1)/element_size;
 416   }
 417   assert( stack_size <= (size_t)max_jint, "stack size too big");
 418   return ((jint)stack_size);
 419 }

 435 void frame::print_value_on(outputStream* st, JavaThread *thread) const {
 436   NOT_PRODUCT(address begin = pc()-40;)
 437   NOT_PRODUCT(address end   = NULL;)
 438 
 439   st->print("%s frame (sp=" INTPTR_FORMAT " unextended sp=" INTPTR_FORMAT, print_name(), p2i(sp()), p2i(unextended_sp()));
 440   if (sp() != NULL)
 441     st->print(", fp=" INTPTR_FORMAT ", real_fp=" INTPTR_FORMAT ", pc=" INTPTR_FORMAT,
 442               p2i(fp()), p2i(real_fp()), p2i(pc()));
 443 
 444   if (StubRoutines::contains(pc())) {
 445     st->print_cr(")");
 446     st->print("(");
 447     StubCodeDesc* desc = StubCodeDesc::desc_for(pc());
 448     st->print("~Stub::%s", desc->name());
 449     NOT_PRODUCT(begin = desc->begin(); end = desc->end();)
 450   } else if (Interpreter::contains(pc())) {
 451     st->print_cr(")");
 452     st->print("(");
 453     InterpreterCodelet* desc = Interpreter::codelet_containing(pc());
 454     if (desc != NULL) {
 455       st->print("~");
 456       desc->print_on(st);
 457       NOT_PRODUCT(begin = desc->code_begin(); end = desc->code_end();)
 458     } else {
 459       st->print("~interpreter");
 460     }
 461   }
 462   st->print_cr(")");
 463 
 464   if (_cb != NULL) {
 465     st->print("     ");
 466     _cb->print_value_on(st);
 467     st->cr();
 468 #ifndef PRODUCT
 469     if (end == NULL) {
 470       begin = _cb->code_begin();
 471       end   = _cb->code_end();
 472     }
 473 #endif
 474   }
 475   NOT_PRODUCT(if (WizardMode && Verbose) Disassembler::decode(begin, end);)
 476 }

 487 void frame::interpreter_frame_print_on(outputStream* st) const {
 488 #ifndef PRODUCT
 489   assert(is_interpreted_frame(), "Not an interpreted frame");
 490   jint i;
 491   for (i = 0; i < interpreter_frame_method()->max_locals(); i++ ) {
 492     intptr_t x = *interpreter_frame_local_at(i);
 493     st->print(" - local  [" INTPTR_FORMAT "]", x);
 494     st->fill_to(23);
 495     st->print_cr("; #%d", i);
 496   }
 497   for (i = interpreter_frame_expression_stack_size() - 1; i >= 0; --i ) {
 498     intptr_t x = *interpreter_frame_expression_stack_at(i);
 499     st->print(" - stack  [" INTPTR_FORMAT "]", x);
 500     st->fill_to(23);
 501     st->print_cr("; #%d", i);
 502   }
 503   // locks for synchronization
 504   for (BasicObjectLock* current = interpreter_frame_monitor_end();
 505        current < interpreter_frame_monitor_begin();
 506        current = next_monitor_in_interpreter_frame(current)) {
 507     st->print(" - obj    [");
 508     current->obj()->print_value_on(st);
 509     st->print_cr("]");
 510     st->print(" - lock   [");
 511     current->lock()->print_on(st);
 512     st->print_cr("]");
 513   }
 514   // monitor
 515   st->print_cr(" - monitor[" INTPTR_FORMAT "]", p2i(interpreter_frame_monitor_begin()));
 516   // bcp
 517   st->print(" - bcp    [" INTPTR_FORMAT "]", p2i(interpreter_frame_bcp()));
 518   st->fill_to(23);
 519   st->print_cr("; @%d", interpreter_frame_bci());
 520   // locals
 521   st->print_cr(" - locals [" INTPTR_FORMAT "]", p2i(interpreter_frame_local_at(0)));
 522   // method
 523   st->print(" - method [" INTPTR_FORMAT "]", p2i(interpreter_frame_method()));
 524   st->fill_to(23);
 525   st->print("; ");
 526   interpreter_frame_method()->print_name(st);
 527   st->cr();
 528 #endif
 529 }

 531 // Print whether the frame is in the VM or OS indicating a HotSpot problem.
 532 // Otherwise, it's likely a bug in the native library that the Java code calls,
 533 // hopefully indicating where to submit bugs.
 534 void frame::print_C_frame(outputStream* st, char* buf, int buflen, address pc) {
 535   // C/C++ frame
 536   bool in_vm = os::address_is_in_vm(pc);
 537   st->print(in_vm ? "V" : "C");
 538 
 539   int offset;
 540   bool found;
 541 
 542   // libname
 543   found = os::dll_address_to_library_name(pc, buf, buflen, &offset);
 544   if (found) {
 545     // skip directory names
 546     const char *p1, *p2;
 547     p1 = buf;
 548     int len = (int)strlen(os::file_separator());
 549     while ((p2 = strstr(p1, os::file_separator())) != NULL) p1 = p2 + len;
 550     st->print("  [%s+0x%x]", p1, offset);
 551   } else {
 552     st->print("  " PTR_FORMAT, p2i(pc));
 553   }
 554 
 555   found = os::dll_address_to_function_name(pc, buf, buflen, &offset);
 556   if (found) {
 557     st->print("  %s+0x%x", buf, offset);
 558   }
 559 }

 561 // frame::print_on_error() is called by fatal error handler. Notice that we may
 562 // crash inside this function if stack frame is corrupted. The fatal error
 563 // handler can catch and handle the crash. Here we assume the frame is valid.
 564 //
 565 // First letter indicates type of the frame:
 566 //    J: Java frame (compiled)
 567 //    A: Java frame (aot compiled)
 568 //    j: Java frame (interpreted)
 569 //    V: VM frame (C/C++)
 570 //    v: Other frames running VM generated code (e.g. stubs, adapters, etc.)
 571 //    C: C/C++ frame
 572 //
 573 // We don't need detailed frame type as that in frame::print_name(). "C"
 574 // suggests the problem is in user lib; everything else is likely a VM bug.
 575 
 576 void frame::print_on_error(outputStream* st, char* buf, int buflen, bool verbose) const {
 577   if (_cb != NULL) {
 578     if (Interpreter::contains(pc())) {
 579       Method* m = this->interpreter_frame_method();
 580       if (m != NULL) {
 581         m->name_and_sig_as_C_string(buf, buflen);
 582         st->print("j  %s", buf);
 583         st->print("+%d", this->interpreter_frame_bci());
 584         ModuleEntry* module = m->method_holder()->module();
 585         if (module->is_named()) {
 586           module->name()->as_C_string(buf, buflen);
 587           st->print(" %s", buf);
 588           if (module->version() != NULL) {
 589             module->version()->as_C_string(buf, buflen);
 590             st->print("@%s", buf);
 591           }
 592         }
 593       } else {
 594         st->print("j  " PTR_FORMAT, p2i(pc()));
 595       }
 596     } else if (StubRoutines::contains(pc())) {
 597       StubCodeDesc* desc = StubCodeDesc::desc_for(pc());
 598       if (desc != NULL) {
 599         st->print("v  ~StubRoutines::%s", desc->name());
 600       } else {
 601         st->print("v  ~StubRoutines::" PTR_FORMAT, p2i(pc()));
 602       }
 603     } else if (_cb->is_buffer_blob()) {
 604       st->print("v  ~BufferBlob::%s", ((BufferBlob *)_cb)->name());
 605     } else if (_cb->is_compiled()) {
 606       CompiledMethod* cm = (CompiledMethod*)_cb;
 607       Method* m = cm->method();
 608       if (m != NULL) {
 609         if (cm->is_aot()) {
 610           st->print("A %d ", cm->compile_id());
 611         } else if (cm->is_nmethod()) {
 612           nmethod* nm = cm->as_nmethod();
 613           st->print("J %d%s", nm->compile_id(), (nm->is_osr_method() ? "%" : ""));
 614           st->print(" %s", nm->compiler_name());
 615         }
 616         m->name_and_sig_as_C_string(buf, buflen);
 617         st->print(" %s", buf);
 618         ModuleEntry* module = m->method_holder()->module();
 619         if (module->is_named()) {

 638       } else {

 640       }
 641     } else if (_cb->is_runtime_stub()) {
 642       st->print("v  ~RuntimeStub::%s", ((RuntimeStub *)_cb)->name());
 643     } else if (_cb->is_deoptimization_stub()) {
 644       st->print("v  ~DeoptimizationBlob");
 645     } else if (_cb->is_exception_stub()) {
 646       st->print("v  ~ExceptionBlob");

 651     } else if (_cb->is_vtable_blob()) {

 666 /*
 667   The interpreter_frame_expression_stack_at method in the case of SPARC needs the
 668   max_stack value of the method in order to compute the expression stack address.
 669   It uses the Method* in order to get the max_stack value but during GC this
 670   Method* value saved on the frame is changed by reverse_and_push and hence cannot
 671   be used. So we save the max_stack value in the FrameClosure object and pass it
 672   down to the interpreter_frame_expression_stack_at method
 673 */
 674 class InterpreterFrameClosure : public OffsetClosure {
 675  private:
 676   frame* _fr;
 677   OopClosure* _f;
 678   int    _max_locals;
 679   int    _max_stack;
 680 
 681  public:
 682   InterpreterFrameClosure(frame* fr, int max_locals, int max_stack,
 683                           OopClosure* f, BufferedValueClosure* bvt_f) {
 684     _fr         = fr;
 685     _max_locals = max_locals;
 686     _max_stack  = max_stack;
 687     _f          = f;
 688   }
 689 
 690   void offset_do(int offset) {
 691     oop* addr;
 692     if (offset < _max_locals) {
 693       addr = (oop*) _fr->interpreter_frame_local_at(offset);
 694       assert((intptr_t*)addr >= _fr->sp(), "must be inside the frame");
 695       if (_f != NULL) {
 696         _f->do_oop(addr);
 697       }
 698     } else {
 699       addr = (oop*) _fr->interpreter_frame_expression_stack_at((offset - _max_locals));
 700       // In case of exceptions, the expression stack is invalid and the esp will be reset to express
 701       // this condition. Therefore, we call f only if addr is 'inside' the stack (i.e., addr >= esp for Intel).
 702       bool in_stack;
 703       if (frame::interpreter_frame_expression_stack_direction() > 0) {
 704         in_stack = (intptr_t*)addr <= _fr->interpreter_frame_tos_address();
 705       } else {
 706         in_stack = (intptr_t*)addr >= _fr->interpreter_frame_tos_address();
 707       }
 708       if (in_stack) {
 709         if (_f != NULL) {
 710           _f->do_oop(addr);
 711         }
 712       }
 713     }
 714   }
 715 
 716   int max_locals()  { return _max_locals; }
 717   frame* fr()       { return _fr; }
 718 };

 819 void frame::oops_interpreted_do(OopClosure* f, const RegisterMap* map, bool query_oop_map_cache) {
 820   assert(is_interpreted_frame(), "Not an interpreted frame");
 821   assert(map != NULL, "map must be set");
 822   Thread *thread = Thread::current();
 823   methodHandle m (thread, interpreter_frame_method());
 824   jint      bci = interpreter_frame_bci();
 825 
 826   assert(!Universe::heap()->is_in(m()),
 827           "must be valid oop");
 828   assert(m->is_method(), "checking frame value");
 829   assert((m->is_native() && bci == 0)  ||
 830          (!m->is_native() && bci >= 0 && bci < m->code_size()),
 831          "invalid bci value");
 832 
 833   // Handle the monitor elements in the activation
 834   for (
 835     BasicObjectLock* current = interpreter_frame_monitor_end();
 836     current < interpreter_frame_monitor_begin();
 837     current = next_monitor_in_interpreter_frame(current)
 838   ) {
 839 #ifdef ASSERT
 840     interpreter_frame_verify_monitor(current);
 841 #endif
 842     current->oops_do(f);
 843   }
 844 
 845   if (m->is_native()) {
 846     f->do_oop(interpreter_frame_temp_oop_addr());
 847   }
 848 
 849   // The method pointer in the frame might be the only path to the method's
 850   // klass, and the klass needs to be kept alive while executing. The GCs
 851   // don't trace through method pointers, so the mirror of the method's klass
 852   // is installed as a GC root.
 853   f->do_oop(interpreter_frame_mirror_addr());
 854 
 855   int max_locals = m->is_native() ? m->size_of_parameters() : m->max_locals();
 856 
 857   Symbol* signature = NULL;
 858   bool has_receiver = false;
 859 
 860   // Process a callee's arguments if we are at a call site
 861   // (i.e., if we are at an invoke bytecode)
 862   // This is used sometimes for calling into the VM, not for another
 863   // interpreted or compiled frame.
 864   if (!m->is_native()) {
 865     Bytecode_invoke call = Bytecode_invoke_check(m, bci);
 866     if (call.is_valid()) {
 867       signature = call.signature();
 868       has_receiver = call.has_receiver();
 869       if (map->include_argument_oops() &&
 870           interpreter_frame_expression_stack_size() > 0) {
 871         ResourceMark rm(thread);  // is this right ???
 872         // we are at a call site & the expression stack is not empty
 873         // => process callee's arguments
 874         //
 875         // Note: The expression stack can be empty if an exception
 876         //       occurred during method resolution/execution. In all
 877         //       cases we empty the expression stack completely be-
 878         //       fore handling the exception (the exception handling
 879         //       code in the interpreter calls a blocking runtime
 880         //       routine which can cause this code to be executed).
 881         //       (was bug gri 7/27/98)
 882         oops_interpreted_arguments_do(signature, has_receiver, f);
 883       }
 884     }
 885   }
 886 
 887   InterpreterFrameClosure blk(this, max_locals, m->max_stack(), f, NULL);
 888 
 889   // process locals & expression stack
 890   InterpreterOopMap mask;
 891   if (query_oop_map_cache) {
 892     m->mask_for(bci, &mask);
 893   } else {
 894     OopMapCache::compute_one_oop_map(m, bci, &mask);
 895   }
 896   mask.iterate_oop(&blk);
 897 }

 942 class CompiledArgumentOopFinder: public SignatureIterator {
 943  protected:
 944   OopClosure*     _f;
 945   int             _offset;        // the current offset, incremented with each argument
 946   bool            _has_receiver;  // true if the callee has a receiver
 947   bool            _has_appendix;  // true if the call has an appendix
 948   frame           _fr;
 949   RegisterMap*    _reg_map;
 950   int             _arg_size;
 951   VMRegPair*      _regs;        // VMReg list of arguments
 952 
 953   friend class SignatureIterator;  // so do_parameters_on can call do_type
 954   void do_type(BasicType type) {
 955     if (is_reference_type(type))  handle_oop_offset();
 956     _offset += parameter_type_word_count(type);
 957   }
 958 
 959   virtual void handle_oop_offset() {
 960     // Extract low order register number from register array.
 961     // In LP64-land, the high-order bits are valid but unhelpful.
 962     assert(_offset < _arg_size, "out of bounds");
 963     VMReg reg = _regs[_offset].first();
 964     oop *loc = _fr.oopmapreg_to_location(reg, _reg_map);
 965     _f->do_oop(loc);
 966   }
 967 
 968  public:
 969   CompiledArgumentOopFinder(Symbol* signature, bool has_receiver, bool has_appendix, OopClosure* f, frame fr, const RegisterMap* reg_map)
 970     : SignatureIterator(signature) {
 971 
 972     // initialize CompiledArgumentOopFinder
 973     _f         = f;
 974     _offset    = 0;
 975     _has_receiver = has_receiver;
 976     _has_appendix = has_appendix;
 977     _fr        = fr;
 978     _reg_map   = (RegisterMap*)reg_map;
 979     _regs = SharedRuntime::find_callee_arguments(signature, has_receiver, has_appendix, &_arg_size);
 980   }
 981 
 982   void oops_do() {
 983     if (_has_receiver) {
 984       handle_oop_offset();
 985       _offset++;
 986     }
 987     do_parameters_on(this);
 988     if (_has_appendix) {
 989       handle_oop_offset();
 990       _offset++;
 991     }
 992   }
 993 };

1003 // Get receiver out of callers frame, i.e. find parameter 0 in callers
1004 // frame.  Consult ADLC for where parameter 0 is to be found.  Then
1005 // check local reg_map for it being a callee-save register or argument
1006 // register, both of which are saved in the local frame.  If not found
1007 // there, it must be an in-stack argument of the caller.
1008 // Note: caller.sp() points to callee-arguments
1009 oop frame::retrieve_receiver(RegisterMap* reg_map) {
1010   frame caller = *this;
1011 
1012   // First consult the ADLC on where it puts parameter 0 for this signature.
1013   VMReg reg = SharedRuntime::name_for_receiver();
1014   oop* oop_adr = caller.oopmapreg_to_location(reg, reg_map);
1015   if (oop_adr == NULL) {
1016     guarantee(oop_adr != NULL, "bad register save location");
1017     return NULL;
1018   }
1019   oop r = *oop_adr;
1020   assert(Universe::heap()->is_in_or_null(r), "bad receiver: " INTPTR_FORMAT " (" INTX_FORMAT ")", p2i(r), p2i(r));
1021   return r;
1022 }

1160   }

1174     values.describe(-1, info_address,
1175                     err_msg("- %d locals %d max stack", m->max_locals(), m->max_stack()), 1);
1176     if (m->max_locals() > 0) {
1177       intptr_t* l0 = interpreter_frame_local_at(0);

1183         values.describe(frame_no, l0, err_msg("local %d", l));
1184       }

1186 
1187     // Compute the actual expression stack size
1188     InterpreterOopMap mask;
1189     OopMapCache::compute_one_oop_map(methodHandle(Thread::current(), m), bci, &mask);
1190     intptr_t* tos = NULL;
1191     // Report each stack element and mark as owned by this frame
1192     for (int e = 0; e < mask.expression_stack_size(); e++) {
1193       tos = MAX2(tos, interpreter_frame_expression_stack_at(e));
1194       values.describe(frame_no, interpreter_frame_expression_stack_at(e),
1195                       err_msg("stack %d", e));
1196     }
1197     if (tos != NULL) {
1198       values.describe(-1, tos, err_msg("expression stack for #%d", frame_no), 1);
1199     }
1200     if (interpreter_frame_monitor_begin() != interpreter_frame_monitor_end()) {
1201       values.describe(frame_no, (intptr_t*)interpreter_frame_monitor_begin(), "monitors begin");
1202       values.describe(frame_no, (intptr_t*)interpreter_frame_monitor_end(), "monitors end");
1203     }
1204   } else if (is_entry_frame()) {
1205     // For now just label the frame
1206     values.describe(-1, info_address, err_msg("#%d entry frame", frame_no), 2);
1207   } else if (is_compiled_frame()) {
1208     // For now just label the frame
1209     CompiledMethod* cm = (CompiledMethod*)cb();
1210     values.describe(-1, info_address,
1211                     FormatBuffer<1024>("#%d nmethod " INTPTR_FORMAT " for method %s%s%s", frame_no,
1212                                        p2i(cm),
1213                                        (cm->is_aot() ? "A ": "J "),
1214                                        cm->method()->name_and_sig_as_C_string(),
1215                                        (_deopt_state == is_deoptimized) ?
1216                                        " (deoptimized)" :
1217                                        ((_deopt_state == unknown) ? " (state unknown)" : "")),
1218                     2);
1219   } else if (is_native_frame()) {
1220     // For now just label the frame
1221     nmethod* nm = cb()->as_nmethod_or_null();
1222     values.describe(-1, info_address,
1223                     FormatBuffer<1024>("#%d nmethod " INTPTR_FORMAT " for native method %s", frame_no,
1224                                        p2i(nm), nm->method()->name_and_sig_as_C_string()), 2);
1225   } else {
1226     // provide default info if not handled before
1227     char *info = (char *) "special frame";
1228     if ((_cb != NULL) &&
1229         (_cb->name() != NULL)) {
1230       info = (char *)_cb->name();
1231     }
1232     values.describe(-1, info_address, err_msg("#%d <%s>", frame_no, info), 2);
1233   }
1234 
1235   // platform dependent additional data
1236   describe_pd(values, frame_no);
1237 }

1266 void FrameValues::validate() {
1267   _values.sort(compare);
1268   bool error = false;
1269   FrameValue prev;
1270   prev.owner = -1;
1271   for (int i = _values.length() - 1; i >= 0; i--) {
1272     FrameValue fv = _values.at(i);
1273     if (fv.owner == -1) continue;
1274     if (prev.owner == -1) {
1275       prev = fv;
1276       continue;
1277     }
1278     if (prev.location == fv.location) {
1279       if (fv.owner != prev.owner) {
1280         tty->print_cr("overlapping storage");
1281         tty->print_cr(" " INTPTR_FORMAT ": " INTPTR_FORMAT " %s", p2i(prev.location), *prev.location, prev.description);
1282         tty->print_cr(" " INTPTR_FORMAT ": " INTPTR_FORMAT " %s", p2i(fv.location), *fv.location, fv.description);
1283         error = true;
1284       }
1285     } else {
1286       prev = fv;
1287     }
1288   }
1289   assert(!error, "invalid layout");
1290 }

1293 void FrameValues::print(JavaThread* thread) {
1294   _values.sort(compare);
1295 
1296   // Sometimes values like the fp can be invalid values if the
1297   // register map wasn't updated during the walk.  Trim out values
1298   // that aren't actually in the stack of the thread.
1299   int min_index = 0;
1300   int max_index = _values.length() - 1;
1301   intptr_t* v0 = _values.at(min_index).location;
1302   intptr_t* v1 = _values.at(max_index).location;
1303 
1304   if (thread == Thread::current()) {
1305     while (!thread->is_in_live_stack((address)v0)) {
1306       v0 = _values.at(++min_index).location;
1307     }
1308     while (!thread->is_in_live_stack((address)v1)) {
1309       v1 = _values.at(--max_index).location;
1310     }
1311   } else {
1312     while (!thread->is_in_full_stack((address)v0)) {
1313       v0 = _values.at(++min_index).location;
1314     }
1315     while (!thread->is_in_full_stack((address)v1)) {
1316       v1 = _values.at(--max_index).location;
1317     }
1318   }
1319   intptr_t* min = MIN2(v0, v1);
1320   intptr_t* max = MAX2(v0, v1);
1321   intptr_t* cur = max;
1322   intptr_t* last = NULL;
1323   for (int i = max_index; i >= min_index; i--) {
1324     FrameValue fv = _values.at(i);
1325     while (cur > fv.location) {
1326       tty->print_cr(" " INTPTR_FORMAT ": " INTPTR_FORMAT, p2i(cur), *cur);
1327       cur--;
1328     }
1329     if (last == fv.location) {
1330       const char* spacer = "          " LP64_ONLY("        ");
1331       tty->print_cr(" %s  %s %s", spacer, spacer, fv.description);
1332     } else {
1333       tty->print_cr(" " INTPTR_FORMAT ": " INTPTR_FORMAT " %s", p2i(fv.location), *fv.location, fv.description);
1334       last = fv.location;
1335       cur--;
1336     }
1337   }
1338 }
