  45 bool ShenandoahBarrierC2Support::expand(Compile* C, PhaseIterGVN& igvn) {
  46   ShenandoahBarrierSetC2State* state = ShenandoahBarrierSetC2::bsc2()->state();
  47   if ((state->enqueue_barriers_count() +
  48        state->load_reference_barriers_count()) > 0) {
  49     bool attempt_more_loopopts = ShenandoahLoopOptsAfterExpansion;
  50     C->clear_major_progress();
  51     PhaseIdealLoop ideal_loop(igvn, LoopOptsShenandoahExpand);
  52     if (C->failing()) return false;
  53     PhaseIdealLoop::verify(igvn);
  54     DEBUG_ONLY(verify_raw_mem(C->root());)
  55     if (attempt_more_loopopts) {
  56       C->set_major_progress();
  57       if (!C->optimize_loops(igvn, LoopOptsShenandoahPostExpand)) {
  58         return false;
  59       }
  60       C->clear_major_progress();
  61       if (C->range_check_cast_count() > 0) {
  62         // No more loop optimizations. Remove all range check dependent CastIINodes.
  63         C->remove_range_check_casts(igvn);
  64         igvn.optimize();
  65       }
  66     }
  67   }
  68   return true;
  69 }

  71 bool ShenandoahBarrierC2Support::is_gc_state_test(Node* iff, int mask) {
  72   if (!UseShenandoahGC) {
  73     return false;
  74   }
  75   assert(iff->is_If(), "bad input");
  76   if (iff->Opcode() != Op_If) {
  77     return false;
  78   }
  79   Node* bol = iff->in(1);
  80   if (!bol->is_Bool() || bol->as_Bool()->_test._test != BoolTest::ne) {
  81     return false;
  82   }
  83   Node* cmp = bol->in(1);
  84   if (cmp->Opcode() != Op_CmpI) {
  85     return false;
  86   }
  87   Node* in1 = cmp->in(1);
  88   Node* in2 = cmp->in(2);
  89   if (in2->find_int_con(-1) != 0) {
  90     return false;
  91   }
  92   if (in1->Opcode() != Op_AndI) {
  93     return false;
  94   }
  95   in2 = in1->in(2);
  96   if (in2->find_int_con(-1) != mask) {
  97     return false;
  98   }
  99   in1 = in1->in(1);
 100 
 101   return is_gc_state_load(in1);
 102 }

 108 bool ShenandoahBarrierC2Support::is_gc_state_load(Node *n) {
 109   if (!UseShenandoahGC) {
 110     return false;
 111   }
 112   if (n->Opcode() != Op_LoadB && n->Opcode() != Op_LoadUB) {
 113     return false;
 114   }
 115   Node* addp = n->in(MemNode::Address);
 116   if (!addp->is_AddP()) {
 117     return false;
 118   }
 119   Node* base = addp->in(AddPNode::Address);
 120   Node* off = addp->in(AddPNode::Offset);
 121   if (base->Opcode() != Op_ThreadLocal) {
 122     return false;
 123   }
 124   if (off->find_intptr_t_con(-1) != in_bytes(ShenandoahThreadLocalData::gc_state_offset())) {
 125     return false;
 126   }
 127   return true;
 128 }

 130 bool ShenandoahBarrierC2Support::has_safepoint_between(Node* start, Node* stop, PhaseIdealLoop *phase) {
 131   assert(phase->is_dominator(stop, start), "bad inputs");
 132   ResourceMark rm;
 133   Unique_Node_List wq;
 134   wq.push(start);
 135   for (uint next = 0; next < wq.size(); next++) {
 136     Node *m = wq.at(next);
 137     if (m == stop) {
 138       continue;
 139     }
 140     if (m->is_SafePoint() && !m->is_CallLeaf()) {
 141       return true;
 142     }
 143     if (m->is_Region()) {
 144       for (uint i = 1; i < m->req(); i++) {
 145         wq.push(m->in(i));
 146       }
 147     } else {
 148       wq.push(m->in(0));
 149     }
 150   }
 151   return false;
 152 }

 155 bool ShenandoahBarrierC2Support::verify_helper(Node* in, Node_Stack& phis, VectorSet& visited, verify_type t, bool trace, Unique_Node_List& barriers_used) {
 156   assert(phis.size() == 0, "");
 157 
 158   while (true) {
 159     if (in->bottom_type() == TypePtr::NULL_PTR) {
 160       if (trace) {tty->print_cr("NULL");}
 161     } else if (!in->bottom_type()->make_ptr()->make_oopptr()) {
 162       if (trace) {tty->print_cr("Non oop");}
 163     } else {
 164       if (in->is_ConstraintCast()) {
 165         in = in->in(1);
 166         continue;
 167       } else if (in->is_AddP()) {
 168         assert(!in->in(AddPNode::Address)->is_top(), "no raw memory access");
 169         in = in->in(AddPNode::Address);
 170         continue;
 171       } else if (in->is_Con()) {
 172         if (trace) {
 173           tty->print("Found constant");
 174           in->dump();
 175         }
 176       } else if (in->Opcode() == Op_Parm) {
 177         if (trace) {
 178           tty->print("Found argument");
 179         }
 180       } else if (in->Opcode() == Op_CreateEx) {
 181         if (trace) {
 182           tty->print("Found create-exception");
 183         }
 184       } else if (in->Opcode() == Op_LoadP && in->adr_type() == TypeRawPtr::BOTTOM) {
 185         if (trace) {
 186           tty->print("Found raw LoadP (OSR argument?)");
 187         }
 188       } else if (in->Opcode() == Op_ShenandoahLoadReferenceBarrier) {
 189         if (t == ShenandoahOopStore) {
 190           uint i = 0;
 191           for (; i < phis.size(); i++) {
 192             Node* n = phis.node_at(i);
 193             if (n->Opcode() == Op_ShenandoahEnqueueBarrier) {
 194               break;
 195             }
 196           }
 197           if (i == phis.size()) {
 198             return false;
 199           }
 200         }
 201         barriers_used.push(in);
 202         if (trace) {tty->print("Found barrier"); in->dump();}
 203       } else if (in->Opcode() == Op_ShenandoahEnqueueBarrier) {
 204         if (t != ShenandoahOopStore) {
 205           in = in->in(1);
 206           continue;
 207         }
 208         if (trace) {tty->print("Found enqueue barrier"); in->dump();}
 209         phis.push(in, in->req());
 210         in = in->in(1);
 211         continue;
 212       } else if (in->is_Proj() && in->in(0)->is_Allocate()) {
 213         if (trace) {
 214           tty->print("Found alloc");
 215           in->in(0)->dump();
 216         }
 217       } else if (in->is_Proj() && (in->in(0)->Opcode() == Op_CallStaticJava || in->in(0)->Opcode() == Op_CallDynamicJava)) {
 218         if (trace) {
 219           tty->print("Found Java call");
 220         }
 221       } else if (in->is_Phi()) {
 222         if (!visited.test_set(in->_idx)) {
 223           if (trace) {tty->print("Pushed phi:"); in->dump();}
 224           phis.push(in, 2);
 225           in = in->in(1);
 226           continue;
 227         }
 228         if (trace) {tty->print("Already seen phi:"); in->dump();}
 229       } else if (in->Opcode() == Op_CMoveP || in->Opcode() == Op_CMoveN) {
 230         if (!visited.test_set(in->_idx)) {
 231           if (trace) {tty->print("Pushed cmovep:"); in->dump();}
 232           phis.push(in, CMoveNode::IfTrue);
 233           in = in->in(CMoveNode::IfFalse);
 234           continue;
 235         }
 236         if (trace) {tty->print("Already seen cmovep:"); in->dump();}
 237       } else if (in->Opcode() == Op_EncodeP || in->Opcode() == Op_DecodeN) {
 238         in = in->in(1);
 239         continue;
 240       } else {
 241         return false;
 242       }
 243     }
 244     bool cont = false;
 245     while (phis.is_nonempty()) {
 246       uint idx = phis.index();
 247       Node* phi = phis.node();
 248       if (idx >= phi->req()) {
 249         if (trace) {tty->print("Popped phi:"); phi->dump();}
 250         phis.pop();
 251         continue;
 252       }
 253       if (trace) {tty->print("Next entry(%d) for phi:", idx); phi->dump();}
 254       in = phi->in(idx);
 255       phis.set_index(idx+1);
 256       cont = true;
 257       break;
 258     }
 259     if (!cont) {
 260       break;
 261     }
 262   }
 263   return true;
 264 }

 266 void ShenandoahBarrierC2Support::report_verify_failure(const char* msg, Node* n1, Node* n2) {
 267   if (n1 != NULL) {
 268     n1->dump(+10);
 269   }
 270   if (n2 != NULL) {
 271     n2->dump(+10);
 272   }
 273   fatal("%s", msg);
 274 }

 276 void ShenandoahBarrierC2Support::verify(RootNode* root) {
 277   ResourceMark rm;
 278   Unique_Node_List wq;
 279   GrowableArray<Node*> barriers;
 280   Unique_Node_List barriers_used;
 281   Node_Stack phis(0);
 282   VectorSet visited(Thread::current()->resource_area());
 283   const bool trace = false;
 284   const bool verify_no_useless_barrier = false;
 285 
 286   wq.push(root);
 287   for (uint next = 0; next < wq.size(); next++) {
 288     Node *n = wq.at(next);
 289     if (n->is_Load()) {
 290       const bool trace = false;
 291       if (trace) {tty->print("Verifying"); n->dump();}
 292       if (n->Opcode() == Op_LoadRange || n->Opcode() == Op_LoadKlass || n->Opcode() == Op_LoadNKlass) {
 293         if (trace) {tty->print_cr("Load range/klass");}
 294       } else {
 295         const TypePtr* adr_type = n->as_Load()->adr_type();
 296 
 297         if (adr_type->isa_oopptr() && adr_type->is_oopptr()->offset() == oopDesc::mark_offset_in_bytes()) {
 298           if (trace) {tty->print_cr("Mark load");}
 299         } else if (adr_type->isa_instptr() &&
 300                    adr_type->is_instptr()->klass()->is_subtype_of(Compile::current()->env()->Reference_klass()) &&
 301                    adr_type->is_instptr()->offset() == java_lang_ref_Reference::referent_offset) {
 302           if (trace) {tty->print_cr("Reference.get()");}
 303         } else if (!verify_helper(n->in(MemNode::Address), phis, visited, ShenandoahLoad, trace, barriers_used)) {
 304           report_verify_failure("Shenandoah verification: Load should have barriers", n);
 305         }
 306       }
 307     } else if (n->is_Store()) {
 308       const bool trace = false;
 309 
 310       if (trace) {tty->print("Verifying"); n->dump();}
 311       if (n->in(MemNode::ValueIn)->bottom_type()->make_oopptr()) {
 312         Node* adr = n->in(MemNode::Address);
 313         bool verify = true;
 314 
 315         if (adr->is_AddP() && adr->in(AddPNode::Base)->is_top()) {
 316           adr = adr->in(AddPNode::Address);
 317           if (adr->is_AddP()) {
 318             assert(adr->in(AddPNode::Base)->is_top(), "");
 319             adr = adr->in(AddPNode::Address);
 320             if (adr->Opcode() == Op_LoadP &&
 321                 adr->in(MemNode::Address)->in(AddPNode::Base)->is_top() &&
 322                 adr->in(MemNode::Address)->in(AddPNode::Address)->Opcode() == Op_ThreadLocal &&
 323                 adr->in(MemNode::Address)->in(AddPNode::Offset)->find_intptr_t_con(-1) == in_bytes(ShenandoahThreadLocalData::satb_mark_queue_buffer_offset())) {
 324               if (trace) {tty->print_cr("SATB prebarrier");}
 325               verify = false;
 326             }
 327           }
 328         }
 329 
 330         if (verify && !verify_helper(n->in(MemNode::ValueIn), phis, visited, ShenandoahStoreValEnqueueBarrier ? ShenandoahOopStore : ShenandoahValue, trace, barriers_used)) {
 331           report_verify_failure("Shenandoah verification: Store should have barriers", n);
 332         }
 333       }
 334       if (!verify_helper(n->in(MemNode::Address), phis, visited, ShenandoahStore, trace, barriers_used)) {
 335         report_verify_failure("Shenandoah verification: Store (address) should have barriers", n);
 336       }
 337     } else if (n->Opcode() == Op_CmpP) {
 338       const bool trace = false;
 339 
 340       Node* in1 = n->in(1);
 341       Node* in2 = n->in(2);
 342       if (in1->bottom_type()->isa_oopptr()) {
 343         if (trace) {tty->print("Verifying"); n->dump();}
 344 
 345         bool mark_inputs = false;
 346         if (in1->bottom_type() == TypePtr::NULL_PTR || in2->bottom_type() == TypePtr::NULL_PTR ||
 347             (in1->is_Con() || in2->is_Con())) {
 348           if (trace) {tty->print_cr("Comparison against a constant");}
 349           mark_inputs = true;
 350         } else if ((in1->is_CheckCastPP() && in1->in(1)->is_Proj() && in1->in(1)->in(0)->is_Allocate()) ||
 351                    (in2->is_CheckCastPP() && in2->in(1)->is_Proj() && in2->in(1)->in(0)->is_Allocate())) {
 352           if (trace) {tty->print_cr("Comparison with newly alloc'ed object");}
 353           mark_inputs = true;
 354         } else {
 355           assert(in2->bottom_type()->isa_oopptr(), "");
 356 
 357           if (!verify_helper(in1, phis, visited, ShenandoahStore, trace, barriers_used) ||
 358               !verify_helper(in2, phis, visited, ShenandoahStore, trace, barriers_used)) {
 359             report_verify_failure("Shenandoah verification: Cmp should have barriers", n);
 360           }
 361         }
 362         if (verify_no_useless_barrier &&
 363             mark_inputs &&
 364             (!verify_helper(in1, phis, visited, ShenandoahValue, trace, barriers_used) ||
 365              !verify_helper(in2, phis, visited, ShenandoahValue, trace, barriers_used))) {
 366           phis.clear();
 367           visited.reset();
 368         }
 369       }
 370     } else if (n->is_LoadStore()) {

 476       } else if (strlen(call->_name) > 5 &&
 477                  !strcmp(call->_name + strlen(call->_name) - 5, "_fill")) {
 478         if (!verify_helper(n->in(TypeFunc::Parms), phis, visited, ShenandoahStore, trace, barriers_used)) {
 479           report_verify_failure("Shenandoah verification: _fill should have barriers", n);
 480         }
 481       } else if (!strcmp(call->_name, "shenandoah_wb_pre")) {
 482         // skip
 483       } else {
 484         const int calls_len = sizeof(calls) / sizeof(calls[0]);
 485         int i = 0;
 486         for (; i < calls_len; i++) {
 487           if (!strcmp(calls[i].name, call->_name)) {
 488             break;
 489           }
 490         }
 491         if (i != calls_len) {
 492           const uint args_len = sizeof(calls[0].args) / sizeof(calls[0].args[0]);
 493           for (uint j = 0; j < args_len; j++) {
 494             int pos = calls[i].args[j].pos;
 495             if (pos == -1) {
 496               break;
 497             }
 498             if (!verify_helper(call->in(pos), phis, visited, calls[i].args[j].t, trace, barriers_used)) {
 499               report_verify_failure("Shenandoah verification: intrinsic calls should have barriers", n);
 500             }
 501           }
 502           for (uint j = TypeFunc::Parms; j < call->req(); j++) {
 503             if (call->in(j)->bottom_type()->make_ptr() &&
 504                 call->in(j)->bottom_type()->make_ptr()->isa_oopptr()) {
 505               uint k = 0;
 506               for (; k < args_len && calls[i].args[k].pos != (int)j; k++);
 507               if (k == args_len) {
 508                 fatal("arg %d for call %s not covered", j, call->_name);
 509               }
 510             }
 511           }
 512         } else {
 513           for (uint j = TypeFunc::Parms; j < call->req(); j++) {
 514             if (call->in(j)->bottom_type()->make_ptr() &&
 515                 call->in(j)->bottom_type()->make_ptr()->isa_oopptr()) {
 516               fatal("%s not covered", call->_name);
 517             }
 518           }
 519         }
 520       }
 521     } else if (n->Opcode() == Op_ShenandoahEnqueueBarrier || n->Opcode() == Op_ShenandoahLoadReferenceBarrier) {
 522       // skip
 523     } else if (n->is_AddP()
 524                || n->is_Phi()
 525                || n->is_ConstraintCast()
 526                || n->Opcode() == Op_Return
 527                || n->Opcode() == Op_CMoveP
 528                || n->Opcode() == Op_CMoveN
 529                || n->Opcode() == Op_Rethrow
 530                || n->is_MemBar()
 531                || n->Opcode() == Op_Conv2B
 532                || n->Opcode() == Op_SafePoint
 533                || n->is_CallJava()
 534                || n->Opcode() == Op_Unlock
 535                || n->Opcode() == Op_EncodeP
 536                || n->Opcode() == Op_DecodeN) {
 537       // nothing to do
 538     } else {
 539       static struct {
 540         int opcode;
 541         struct {
 542           int pos;
 543           verify_type t;
 544         } inputs[2];
 545       } others[] = {
 546         Op_FastLock,
 547         { { 1, ShenandoahLoad },                  { -1, ShenandoahNone} },
 548         Op_Lock,
 549         { { TypeFunc::Parms, ShenandoahLoad },    { -1, ShenandoahNone} },
 550         Op_ArrayCopy,
 551         { { ArrayCopyNode::Src, ShenandoahLoad }, { ArrayCopyNode::Dest, ShenandoahStore } },
 552         Op_StrCompressedCopy,
 553         { { 2, ShenandoahLoad },                  { 3, ShenandoahStore } },
 554         Op_StrInflatedCopy,
 555         { { 2, ShenandoahLoad },                  { 3, ShenandoahStore } },
 556         Op_AryEq,
 557         { { 2, ShenandoahLoad },                  { 3, ShenandoahLoad } },
 558         Op_StrIndexOf,
 559         { { 2, ShenandoahLoad },                  { 4, ShenandoahLoad } },
 560         Op_StrComp,
 561         { { 2, ShenandoahLoad },                  { 4, ShenandoahLoad } },
 562         Op_StrEquals,
 563         { { 2, ShenandoahLoad },                  { 3, ShenandoahLoad } },
 564         Op_EncodeISOArray,
 565         { { 2, ShenandoahLoad },                  { 3, ShenandoahStore } },
 566         Op_HasNegatives,
 567         { { 2, ShenandoahLoad },                  { -1, ShenandoahNone} },
 568         Op_CastP2X,
 569         { { 1, ShenandoahLoad },                  { -1, ShenandoahNone} },
 570         Op_StrIndexOfChar,
 571         { { 2, ShenandoahLoad },                  { -1, ShenandoahNone } },
 572       };
 573 
 574       const int others_len = sizeof(others) / sizeof(others[0]);
 575       int i = 0;
 576       for (; i < others_len; i++) {
 577         if (others[i].opcode == n->Opcode()) {
 578           break;
 579         }
 580       }
 581       uint stop = n->is_Call() ? n->as_Call()->tf()->domain_sig()->cnt() : n->req();
 582       if (i != others_len) {
 583         const uint inputs_len = sizeof(others[0].inputs) / sizeof(others[0].inputs[0]);
 584         for (uint j = 0; j < inputs_len; j++) {
 585           int pos = others[i].inputs[j].pos;
 586           if (pos == -1) {
 587             break;
 588           }
 589           if (!verify_helper(n->in(pos), phis, visited, others[i].inputs[j].t, trace, barriers_used)) {
 590             report_verify_failure("Shenandoah verification: intrinsic calls should have barriers", n);
 591           }
 592         }
 593         for (uint j = 1; j < stop; j++) {
 594           if (n->in(j) != NULL && n->in(j)->bottom_type()->make_ptr() &&
 595               n->in(j)->bottom_type()->make_ptr()->make_oopptr()) {
 596             uint k = 0;
 597             for (; k < inputs_len && others[i].inputs[k].pos != (int)j; k++);
 598             if (k == inputs_len) {
 599               fatal("arg %d for node %s not covered", j, n->Name());
 600             }
 601           }
 602         }
 603       } else {
 604         for (uint j = 1; j < stop; j++) {
 605           if (n->in(j) != NULL && n->in(j)->bottom_type()->make_ptr() &&
 606               n->in(j)->bottom_type()->make_ptr()->make_oopptr()) {
 607             fatal("%s not covered", n->Name());
 608           }
 609         }
 610       }
 611     }
 612 
 613     if (n->is_SafePoint()) {
 614       SafePointNode* sfpt = n->as_SafePoint();
 615       if (verify_no_useless_barrier && sfpt->jvms() != NULL) {
 616         for (uint i = sfpt->jvms()->scloff(); i < sfpt->jvms()->endoff(); i++) {
 617           if (!verify_helper(sfpt->in(i), phis, visited, ShenandoahLoad, trace, barriers_used)) {
 618             phis.clear();
 619             visited.reset();
 620           }
 621         }
 622       }
 623     }
 624   }
 625 
 626   if (verify_no_useless_barrier) {
 627     for (int i = 0; i < barriers.length(); i++) {
 628       Node* n = barriers.at(i);
 629       if (!barriers_used.member(n)) {
 630         tty->print("XXX useless barrier"); n->dump(-2);
 631         ShouldNotReachHere();
 632       }
 633     }
 634   }
 635 }

 638 bool ShenandoahBarrierC2Support::is_dominator_same_ctrl(Node* c, Node* d, Node* n, PhaseIdealLoop* phase) {
 639   // That both nodes have the same control is not sufficient to prove
 640   // domination, verify that there's no path from d to n
 641   ResourceMark rm;
 642   Unique_Node_List wq;
 643   wq.push(d);
 644   for (uint next = 0; next < wq.size(); next++) {
 645     Node *m = wq.at(next);
 646     if (m == n) {
 647       return false;
 648     }
 649     if (m->is_Phi() && m->in(0)->is_Loop()) {
 650       assert(phase->ctrl_or_self(m->in(LoopNode::EntryControl)) != c, "following loop entry should lead to new control");
 651     } else {
 652       if (m->is_Store() || m->is_LoadStore()) {
 653         // Take anti-dependencies into account
 654         Node* mem = m->in(MemNode::Memory);
 655         for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
 656           Node* u = mem->fast_out(i);
 657           if (u->is_Load() && phase->C->can_alias(m->adr_type(), phase->C->get_alias_index(u->adr_type())) &&
 658               phase->ctrl_or_self(u) == c) {
 659             wq.push(u);
 660           }
 661         }
 662       }
 663       for (uint i = 0; i < m->req(); i++) {
 664         if (m->in(i) != NULL && phase->ctrl_or_self(m->in(i)) == c) {
 665           wq.push(m->in(i));
 666         }
 667       }
 668     }
 669   }
 670   return true;
 671 }

 680 Node* next_mem(Node* mem, int alias) {
 681   Node* res = NULL;
 682   if (mem->is_Proj()) {
 683     res = mem->in(0);
 684   } else if (mem->is_SafePoint() || mem->is_MemBar()) {
 685     res = mem->in(TypeFunc::Memory);
 686   } else if (mem->is_Phi()) {
 687     res = mem->in(1);
 688   } else if (mem->is_MergeMem()) {
 689     res = mem->as_MergeMem()->memory_at(alias);
 690   } else if (mem->is_Store() || mem->is_LoadStore() || mem->is_ClearArray()) {
 691     assert(alias = Compile::AliasIdxRaw, "following raw memory can't lead to a barrier");
 692     res = mem->in(MemNode::Memory);
 693   } else {
 694 #ifdef ASSERT
 695     mem->dump();
 696 #endif
 697     ShouldNotReachHere();
 698   }
 699   return res;

 702 Node* ShenandoahBarrierC2Support::no_branches(Node* c, Node* dom, bool allow_one_proj, PhaseIdealLoop* phase) {
 703   Node* iffproj = NULL;
 704   while (c != dom) {
 705     Node* next = phase->idom(c);
 706     assert(next->unique_ctrl_out() == c || c->is_Proj() || c->is_Region(), "multiple control flow out but no proj or region?");
 707     if (c->is_Region()) {
 708       ResourceMark rm;
 709       Unique_Node_List wq;
 710       wq.push(c);
 711       for (uint i = 0; i < wq.size(); i++) {
 712         Node *n = wq.at(i);
 713         if (n == next) {
 714           continue;
 715         }
 716         if (n->is_Region()) {
 717           for (uint j = 1; j < n->req(); j++) {
 718             wq.push(n->in(j));
 719           }
 720         } else {
 721           wq.push(n->in(0));
 722         }
 723       }
 724       for (uint i = 0; i < wq.size(); i++) {
 725         Node *n = wq.at(i);
 726         assert(n->is_CFG(), "");
 727         if (n->is_Multi()) {
 728           for (DUIterator_Fast jmax, j = n->fast_outs(jmax); j < jmax; j++) {
 729             Node* u = n->fast_out(j);
 730             if (u->is_CFG()) {
 731               if (!wq.member(u) && !u->as_Proj()->is_uncommon_trap_proj(Deoptimization::Reason_none)) {
 732                 return NodeSentinel;
 733               }
 734             }
 735           }
 736         }
 737       }
 738     } else  if (c->is_Proj()) {
 739       if (c->is_IfProj()) {
 740         if (c->as_Proj()->is_uncommon_trap_if_pattern(Deoptimization::Reason_none) != NULL) {
 741           // continue;
 742         } else {
 743           if (!allow_one_proj) {
 744             return NodeSentinel;
 745           }
 746           if (iffproj == NULL) {
 747             iffproj = c;
 748           } else {
 749             return NodeSentinel;
 750           }
 751         }
 752       } else if (c->Opcode() == Op_JumpProj) {
 753         return NodeSentinel; // unsupported
 754       } else if (c->Opcode() == Op_CatchProj) {
 755         return NodeSentinel; // unsupported
 756       } else if (c->Opcode() == Op_CProj && next->Opcode() == Op_NeverBranch) {
 757         return NodeSentinel; // unsupported
 758       } else {
 759         assert(next->unique_ctrl_out() == c, "unsupported branch pattern");
 760       }
 761     }
 762     c = next;
 763   }
 764   return iffproj;
 765 }

 767 Node* ShenandoahBarrierC2Support::dom_mem(Node* mem, Node* ctrl, int alias, Node*& mem_ctrl, PhaseIdealLoop* phase) {
 768   ResourceMark rm;
 769   VectorSet wq(Thread::current()->resource_area());
 770   wq.set(mem->_idx);
 771   mem_ctrl = phase->ctrl_or_self(mem);
 772   while (!phase->is_dominator(mem_ctrl, ctrl) || mem_ctrl == ctrl) {
 773     mem = next_mem(mem, alias);
 774     if (wq.test_set(mem->_idx)) {
 775       return NULL;
 776     }
 777     mem_ctrl = phase->ctrl_or_self(mem);
 778   }
 779   if (mem->is_MergeMem()) {
 780     mem = mem->as_MergeMem()->memory_at(alias);
 781     mem_ctrl = phase->ctrl_or_self(mem);
 782   }
 783   return mem;
 784 }

 786 Node* ShenandoahBarrierC2Support::find_bottom_mem(Node* ctrl, PhaseIdealLoop* phase) {
 787   Node* mem = NULL;
 788   Node* c = ctrl;
 789   do {
 790     if (c->is_Region()) {
 791       for (DUIterator_Fast imax, i = c->fast_outs(imax); i < imax && mem == NULL; i++) {
 792         Node* u = c->fast_out(i);
 793         if (u->is_Phi() && u->bottom_type() == Type::MEMORY) {
 794           if (u->adr_type() == TypePtr::BOTTOM) {
 795             mem = u;
 796           }
 797         }
 798       }
 799     } else {
 800       if (c->is_Call() && c->as_Call()->adr_type() != NULL) {
 801         CallProjections* projs = c->as_Call()->extract_projections(true, false);
 802         if (projs->fallthrough_memproj != NULL) {
 803           if (projs->fallthrough_memproj->adr_type() == TypePtr::BOTTOM) {
 804             if (projs->catchall_memproj == NULL) {
 805               mem = projs->fallthrough_memproj;
 806             } else {
 807               if (phase->is_dominator(projs->fallthrough_catchproj, ctrl)) {
 808                 mem = projs->fallthrough_memproj;
 809               } else {
 810                 assert(phase->is_dominator(projs->catchall_catchproj, ctrl), "one proj must dominate barrier");
 811                 mem = projs->catchall_memproj;
 812               }
 813             }
 814           }
 815         } else {
 816           Node* proj = c->as_Call()->proj_out(TypeFunc::Memory);
 817           if (proj != NULL &&
 818               proj->adr_type() == TypePtr::BOTTOM) {
 819             mem = proj;
 820           }
 821         }
 822       } else {
 823         for (DUIterator_Fast imax, i = c->fast_outs(imax); i < imax; i++) {
 824           Node* u = c->fast_out(i);
 825           if (u->is_Proj() &&
 826               u->bottom_type() == Type::MEMORY &&
 827               u->adr_type() == TypePtr::BOTTOM) {
 828               assert(c->is_SafePoint() || c->is_MemBar() || c->is_Start(), "");
 829               assert(mem == NULL, "only one proj");
 830               mem = u;
 831           }
 832         }
 833         assert(!c->is_Call() || c->as_Call()->adr_type() != NULL || mem == NULL, "no mem projection expected");
 834       }
 835     }
 836     c = phase->idom(c);
 837   } while (mem == NULL);
 838   return mem;
 839 }

1000 void ShenandoahBarrierC2Support::fix_ctrl(Node* barrier, Node* region, const MemoryGraphFixer& fixer, Unique_Node_List& uses, Unique_Node_List& uses_to_ignore, uint last, PhaseIdealLoop* phase) {
1001   Node* ctrl = phase->get_ctrl(barrier);
1002   Node* init_raw_mem = fixer.find_mem(ctrl, barrier);
1003 
1004   // Update the control of all nodes that should be after the
1005   // barrier control flow
1006   uses.clear();
1007   // Every node that is control dependent on the barrier's input
1008   // control will be after the expanded barrier. The raw memory (if
1009   // its memory is control dependent on the barrier's input control)
1010   // must stay above the barrier.
1011   uses_to_ignore.clear();
1012   if (phase->has_ctrl(init_raw_mem) && phase->get_ctrl(init_raw_mem) == ctrl && !init_raw_mem->is_Phi()) {
1013     uses_to_ignore.push(init_raw_mem);
1014   }
1015   for (uint next = 0; next < uses_to_ignore.size(); next++) {
1016     Node *n = uses_to_ignore.at(next);
1017     for (uint i = 0; i < n->req(); i++) {
1018       Node* in = n->in(i);
1019       if (in != NULL && phase->has_ctrl(in) && phase->get_ctrl(in) == ctrl) {
1020         uses_to_ignore.push(in);
1021       }
1022     }
1023   }
1024   for (DUIterator_Fast imax, i = ctrl->fast_outs(imax); i < imax; i++) {
1025     Node* u = ctrl->fast_out(i);
1026     if (u->_idx < last &&
1027         u != barrier &&
1028         !uses_to_ignore.member(u) &&
1029         (u->in(0) != ctrl || (!u->is_Region() && !u->is_Phi())) &&
1030         (ctrl->Opcode() != Op_CatchProj || u->Opcode() != Op_CreateEx)) {
1031       Node* old_c = phase->ctrl_or_self(u);
1032       Node* c = old_c;
1033       if (c != ctrl ||
1034           is_dominator_same_ctrl(old_c, barrier, u, phase) ||
1035           ShenandoahBarrierSetC2::is_shenandoah_state_load(u)) {
1036         phase->igvn().rehash_node_delayed(u);
1037         int nb = u->replace_edge(ctrl, region);
1038         if (u->is_CFG()) {
1039           if (phase->idom(u) == ctrl) {
1040             phase->set_idom(u, region, phase->dom_depth(region));
1041           }
1042         } else if (phase->get_ctrl(u) == ctrl) {
1043           assert(u != init_raw_mem, "should leave input raw mem above the barrier");
1044           uses.push(u);
1045         }
1046         assert(nb == 1, "more than 1 ctrl input?");
1047         --i, imax -= nb;
1048       }
1049     }
1050   }
1051 }

1053 static Node* create_phis_on_call_return(Node* ctrl, Node* c, Node* n, Node* n_clone, const CallProjections* projs, PhaseIdealLoop* phase) {
1054   Node* region = NULL;
1055   while (c != ctrl) {
1056     if (c->is_Region()) {
1057       region = c;
1058     }
1059     c = phase->idom(c);
1060   }
1061   assert(region != NULL, "");
1062   Node* phi = new PhiNode(region, n->bottom_type());
1063   for (uint j = 1; j < region->req(); j++) {
1064     Node* in = region->in(j);
1065     if (phase->is_dominator(projs->fallthrough_catchproj, in)) {
1066       phi->init_req(j, n);
1067     } else if (phase->is_dominator(projs->catchall_catchproj, in)) {
1068       phi->init_req(j, n_clone);
1069     } else {
1070       phi->init_req(j, create_phis_on_call_return(ctrl, in, n, n_clone, projs, phase));
1071     }
1072   }
1073   phase->register_new_node(phi, region);
1074   return phi;

1583 Node* ShenandoahBarrierC2Support::get_load_addr(PhaseIdealLoop* phase, VectorSet& visited, Node* in) {
1584   if (visited.test_set(in->_idx)) {
1585     return NULL;
1586   }
1587   switch (in->Opcode()) {
1588     case Op_Proj:
1589       return get_load_addr(phase, visited, in->in(0));
1590     case Op_CastPP:
1591     case Op_CheckCastPP:
1592     case Op_DecodeN:
1593     case Op_EncodeP:
1594       return get_load_addr(phase, visited, in->in(1));
1595     case Op_LoadN:
1596     case Op_LoadP:
1597       return in->in(MemNode::Address);
1598     case Op_CompareAndExchangeN:
1599     case Op_CompareAndExchangeP:
1600     case Op_GetAndSetN:
1601     case Op_GetAndSetP:
1602     case Op_ShenandoahCompareAndExchangeP:
1603     case Op_ShenandoahCompareAndExchangeN:
1604       // Those instructions would just have stored a different
1605       // value into the field. No use to attempt to fix it at this point.
1606       return phase->igvn().zerocon(T_OBJECT);
1607     case Op_CMoveP:
1608     case Op_CMoveN: {
1609       Node* t = get_load_addr(phase, visited, in->in(CMoveNode::IfTrue));
1610       Node* f = get_load_addr(phase, visited, in->in(CMoveNode::IfFalse));
1611       // Handle unambiguous cases: single address reported on both branches.
1612       if (t != NULL && f == NULL) return t;
1613       if (t == NULL && f != NULL) return f;
1614       if (t != NULL && t == f)    return t;
1615       // Ambiguity.
1616       return phase->igvn().zerocon(T_OBJECT);
1617     }
1618     case Op_Phi: {
1619       Node* addr = NULL;
1620       for (uint i = 1; i < in->req(); i++) {
1621         Node* addr1 = get_load_addr(phase, visited, in->in(i));
1622         if (addr == NULL) {
1623           addr = addr1;
1624         }
1625         if (addr != addr1) {
1626           return phase->igvn().zerocon(T_OBJECT);
1627         }
1628       }
1629       return addr;
1630     }
1631     case Op_ShenandoahLoadReferenceBarrier:
1632       return get_load_addr(phase, visited, in->in(ShenandoahLoadReferenceBarrierNode::ValueIn));
1633     case Op_ShenandoahEnqueueBarrier:
1634       return get_load_addr(phase, visited, in->in(1));
1635     case Op_CallDynamicJava:
1636     case Op_CallLeaf:
1637     case Op_CallStaticJava:
1638     case Op_ConN:
1639     case Op_ConP:
1640     case Op_Parm:
1641     case Op_CreateEx:
1642       return phase->igvn().zerocon(T_OBJECT);
1643     default:
1644 #ifdef ASSERT
1645       fatal("Unknown node in get_load_addr: %s", NodeClassNames[in->Opcode()]);
1646 #endif
1647       return phase->igvn().zerocon(T_OBJECT);
1648   }
1649 
1650 }

1684 bool ShenandoahBarrierC2Support::identical_backtoback_ifs(Node* n, PhaseIdealLoop* phase) {
1685   if (!n->is_If() || n->is_CountedLoopEnd()) {
1686     return false;
1687   }
1688   Node* region = n->in(0);
1689 
1690   if (!region->is_Region()) {
1691     return false;
1692   }
1693   Node* dom = phase->idom(region);
1694   if (!dom->is_If()) {
1695     return false;
1696   }
1697 
1698   if (!is_heap_stable_test(n) || !is_heap_stable_test(dom)) {
1699     return false;
1700   }
1701 
1702   IfNode* dom_if = dom->as_If();
1703   Node* proj_true = dom_if->proj_out(1);
1704   Node* proj_false = dom_if->proj_out(0);
1705 
1706   for (uint i = 1; i < region->req(); i++) {
1707     if (phase->is_dominator(proj_true, region->in(i))) {
1708       continue;
1709     }
1710     if (phase->is_dominator(proj_false, region->in(i))) {
1711       continue;
1712     }
1713     return false;
1714   }
1715 
1716   return true;
1717 }

1719 void ShenandoahBarrierC2Support::merge_back_to_back_tests(Node* n, PhaseIdealLoop* phase) {
1720   assert(is_heap_stable_test(n), "no other tests");
1721   if (identical_backtoback_ifs(n, phase)) {
1722     Node* n_ctrl = n->in(0);
1723     if (phase->can_split_if(n_ctrl)) {
1724       IfNode* dom_if = phase->idom(n_ctrl)->as_If();
1725       if (is_heap_stable_test(n)) {
1726         Node* gc_state_load = n->in(1)->in(1)->in(1)->in(1);
1727         assert(is_gc_state_load(gc_state_load), "broken");
1728         Node* dom_gc_state_load = dom_if->in(1)->in(1)->in(1)->in(1);
1729         assert(is_gc_state_load(dom_gc_state_load), "broken");
1730         if (gc_state_load != dom_gc_state_load) {
1731           phase->igvn().replace_node(gc_state_load, dom_gc_state_load);
1732         }
1733       }
1734       PhiNode* bolphi = PhiNode::make_blank(n_ctrl, n->in(1));
1735       Node* proj_true = dom_if->proj_out(1);
1736       Node* proj_false = dom_if->proj_out(0);
1737       Node* con_true = phase->igvn().makecon(TypeInt::ONE);
1738       Node* con_false = phase->igvn().makecon(TypeInt::ZERO);
1739 
1740       for (uint i = 1; i < n_ctrl->req(); i++) {
1741         if (phase->is_dominator(proj_true, n_ctrl->in(i))) {
1742           bolphi->init_req(i, con_true);
1743         } else {
1744           assert(phase->is_dominator(proj_false, n_ctrl->in(i)), "bad if");
1745           bolphi->init_req(i, con_false);
1746         }
1747       }
1748       phase->register_new_node(bolphi, n_ctrl);
1749       phase->igvn().replace_input_of(n, 1, bolphi);
1750       phase->do_split_if(n);
1751     }
1752   }

1755 IfNode* ShenandoahBarrierC2Support::find_unswitching_candidate(const IdealLoopTree* loop, PhaseIdealLoop* phase) {
1756   // Find first invariant test that doesn't exit the loop
1757   LoopNode *head = loop->_head->as_Loop();
1758   IfNode* unswitch_iff = NULL;
1759   Node* n = head->in(LoopNode::LoopBackControl);
1760   int loop_has_sfpts = -1;
1761   while (n != head) {
1762     Node* n_dom = phase->idom(n);
1763     if (n->is_Region()) {
1764       if (n_dom->is_If()) {
1765         IfNode* iff = n_dom->as_If();
1766         if (iff->in(1)->is_Bool()) {
1767           BoolNode* bol = iff->in(1)->as_Bool();
1768           if (bol->in(1)->is_Cmp()) {
1769             // If condition is invariant and not a loop exit,
1770             // then found reason to unswitch.
1771             if (is_heap_stable_test(iff) &&
1772                 (loop_has_sfpts == -1 || loop_has_sfpts == 0)) {
1773               assert(!loop->is_loop_exit(iff), "both branches should be in the loop");
1774               if (loop_has_sfpts == -1) {
1775                 for(uint i = 0; i < loop->_body.size(); i++) {
1776                   Node *m = loop->_body[i];
1777                   if (m->is_SafePoint() && !m->is_CallLeaf()) {
1778                     loop_has_sfpts = 1;
1779                     break;
1780                   }
1781                 }
1782                 if (loop_has_sfpts == -1) {
1783                   loop_has_sfpts = 0;
1784                 }
1785               }
1786               if (!loop_has_sfpts) {
1787                 unswitch_iff = iff;
1788               }
1789             }
1790           }
1791         }
1792       }
1793     }
1794     n = n_dom;
1795   }
1796   return unswitch_iff;
1797 }

1800 void ShenandoahBarrierC2Support::optimize_after_expansion(VectorSet &visited, Node_Stack &stack, Node_List &old_new, PhaseIdealLoop* phase) {
1801   Node_List heap_stable_tests;
1802   stack.push(phase->C->start(), 0);
1803   do {
1804     Node* n = stack.node();
1805     uint i = stack.index();
1806 
1807     if (i < n->outcnt()) {
1808       Node* u = n->raw_out(i);
1809       stack.set_index(i+1);
1810       if (!visited.test_set(u->_idx)) {
1811         stack.push(u, 0);
1812       }
1813     } else {
1814       stack.pop();
1815       if (n->is_If() && is_heap_stable_test(n)) {
1816         heap_stable_tests.push(n);
1817       }
1818     }
1819   } while (stack.size() > 0);
1820 
1821   for (uint i = 0; i < heap_stable_tests.size(); i++) {
1822     Node* n = heap_stable_tests.at(i);
1823     assert(is_heap_stable_test(n), "only evacuation test");
1824     merge_back_to_back_tests(n, phase);
1825   }
1826 
1827   if (!phase->C->major_progress()) {
1828     VectorSet seen(Thread::current()->resource_area());
1829     for (uint i = 0; i < heap_stable_tests.size(); i++) {
1830       Node* n = heap_stable_tests.at(i);
1831       IdealLoopTree* loop = phase->get_loop(n);
1832       if (loop != phase->ltree_root() &&
1833           loop->_child == NULL &&
1834           !loop->_irreducible) {
1835         Node* head = loop->_head;
1836         if (head->is_Loop() &&
1837             (!head->is_CountedLoop() || head->as_CountedLoop()->is_main_loop() || head->as_CountedLoop()->is_normal_loop()) &&
1838             !seen.test_set(head->_idx)) {
1839           IfNode* iff = find_unswitching_candidate(loop, phase);
1840           if (iff != NULL) {
1841             Node* bol = iff->in(1);
1842             if (head->as_Loop()->is_strip_mined()) {
1843               head->as_Loop()->verify_strip_mined(0);
1844             }
1845             move_gc_state_test_out_of_loop(iff, phase);
1846 
1847             AutoNodeBudget node_budget(phase);
1848 
1849             if (loop->policy_unswitching(phase)) {
1850               if (head->as_Loop()->is_strip_mined()) {
1851                 OuterStripMinedLoopNode* outer = head->as_CountedLoop()->outer_loop();
1852                 hide_strip_mined_loop(outer, head->as_CountedLoop(), phase);
1853               }
1854               phase->do_unswitching(loop, old_new);
1855             } else {
1856               // Not proceeding with unswitching. Move load back in
1857               // the loop.
1858               phase->igvn().replace_input_of(iff, 1, bol);
1859             }
1860           }
1861         }
1862       }
1863     }
1864   }
1865 }

2525                   DEBUG_ONLY(if (trace) { tty->print("ZZZ NOT setting mem"); m->dump(); });
2526                   for (;;) {
2527                     assert(m->is_Mem() || m->is_LoadStore() || m->is_Proj(), "");
2528                     Node* next = NULL;
2529                     if (m->is_Proj()) {
2530                       next = m->in(0);
2531                     } else {
2532                       assert(m->is_Mem() || m->is_LoadStore(), "");
2533                       assert(_alias == Compile::AliasIdxRaw, "");
2534                       next = m->in(MemNode::Memory);
2535                     }
2536                     if (_phase->get_ctrl(next) != u) {
2537                       break;
2538                     }
2539                     if (next->is_MergeMem()) {
2540                       assert(_phase->get_ctrl(next->as_MergeMem()->memory_at(_alias)) != u, "");
2541                       break;
2542                     }
2543                     if (next->is_Phi()) {
2544                       assert(next->adr_type() == TypePtr::BOTTOM && next->in(0) == u, "");
2545                       break;
2546                     }
2547                     m = next;
2548                   }
2549 
2550                   DEBUG_ONLY(if (trace) { tty->print("ZZZ setting to phi"); m->dump(); });
2551                   assert(m->is_Mem() || m->is_LoadStore(), "");
2552                   uint input = (uint)MemNode::Memory;
2553                   _phase->igvn().replace_input_of(m, input, phi);
2554                   push = false;
2555                 }
2556               } else {
2557                 DEBUG_ONLY(if (trace) { tty->print("ZZZ skipping region"); u->dump(); });
2558               }
2559               if (push) {
2560                 uses.push(u);
2561               }
2562             }
2563           } else if (!mem_is_valid(m, u) &&
2564                      !(u->Opcode() == Op_CProj && u->in(0)->Opcode() == Op_NeverBranch && u->as_Proj()->_con == 1)) {
2565             uses.push(u);
2566           }
2567         }
2568       }
2569     }
2570     for (int i = 0; i < phis.length(); i++) {
2571       Node* n = phis.at(i);
2572       Node* r = n->in(0);
2573       DEBUG_ONLY(if (trace) { tty->print("ZZZ fixing new phi"); n->dump(); });
2574       for (uint j = 1; j < n->req(); j++) {
2575         Node* m = find_mem(r->in(j), NULL);
2576         _phase->igvn().replace_input_of(n, j, m);
2577         DEBUG_ONLY(if (trace) { tty->print("ZZZ fixing new phi: %d", j); m->dump(); });
2578       }
2579     }
2580   }
2581   uint last = _phase->C->unique();
2582   MergeMemNode* mm = NULL;
2583   int alias = _alias;
2584   DEBUG_ONLY(if (trace) { tty->print("ZZZ raw mem is"); mem->dump(); });
2585   // Process loads first to not miss an anti-dependency: if the memory
2586   // edge of a store is updated before a load is processed then an
2587   // anti-dependency may be missed.
2588   for (DUIterator i = mem->outs(); mem->has_out(i); i++) {
2589     Node* u = mem->out(i);
2590     if (u->_idx < last && u->is_Load() && _phase->C->get_alias_index(u->adr_type()) == alias) {
2591       Node* m = find_mem(_phase->get_ctrl(u), u);
2592       if (m != mem) {
2593         DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of use"); u->dump(); });
2594         _phase->igvn().replace_input_of(u, MemNode::Memory, m);
2595         --i;
2596       }
2597     }
2598   }
2599   for (DUIterator i = mem->outs(); mem->has_out(i); i++) {
2600     Node* u = mem->out(i);
2601     if (u->_idx < last) {
2602       if (u->is_Mem()) {
2603         if (_phase->C->get_alias_index(u->adr_type()) == alias) {
2604           Node* m = find_mem(_phase->get_ctrl(u), u);
2605           if (m != mem) {
2606             DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of use"); u->dump(); });
2607             _phase->igvn().replace_input_of(u, MemNode::Memory, m);
2608             --i;
2609           }
2610         }
2611       } else if (u->is_MergeMem()) {
2612         MergeMemNode* u_mm = u->as_MergeMem();
2613         if (u_mm->memory_at(alias) == mem) {
2614           MergeMemNode* newmm = NULL;
2615           for (DUIterator_Fast jmax, j = u->fast_outs(jmax); j < jmax; j++) {
2616             Node* uu = u->fast_out(j);
2617             assert(!uu->is_MergeMem(), "chain of MergeMems?");
2618             if (uu->is_Phi()) {
2619               assert(uu->adr_type() == TypePtr::BOTTOM, "");
2620               Node* region = uu->in(0);
2621               int nb = 0;
2622               for (uint k = 1; k < uu->req(); k++) {
2623                 if (uu->in(k) == u) {
2624                   Node* m = find_mem(region->in(k), NULL);
2625                   if (m != mem) {
2626                     DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of phi %d", k); uu->dump(); });
2627                     newmm = clone_merge_mem(u, mem, m, _phase->ctrl_or_self(m), i);
2628                     if (newmm != u) {
2629                       _phase->igvn().replace_input_of(uu, k, newmm);
2630                       nb++;
2631                       --jmax;
2632                     }
2633                   }
2634                 }
2635               }
2636               if (nb > 0) {
2637                 --j;
2638               }
2639             } else {
2640               Node* m = find_mem(_phase->ctrl_or_self(uu), uu);
2641               if (m != mem) {
2642                 DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of use"); uu->dump(); });
2643                 newmm = clone_merge_mem(u, mem, m, _phase->ctrl_or_self(m), i);
2644                 if (newmm != u) {
2645                   _phase->igvn().replace_input_of(uu, uu->find_edge(u), newmm);
2646                   --j, --jmax;
2647                 }
2648               }
2649             }
2650           }
2651         }
2652       } else if (u->is_Phi()) {
2653         assert(u->bottom_type() == Type::MEMORY, "what else?");
2654         if (_phase->C->get_alias_index(u->adr_type()) == alias || u->adr_type() == TypePtr::BOTTOM) {
2655           Node* region = u->in(0);
2656           bool replaced = false;
2657           for (uint j = 1; j < u->req(); j++) {
2658             if (u->in(j) == mem) {
2659               Node* m = find_mem(region->in(j), NULL);
2660               Node* nnew = m;
2661               if (m != mem) {
2662                 if (u->adr_type() == TypePtr::BOTTOM) {
2663                   mm = allocate_merge_mem(mem, m, _phase->ctrl_or_self(m));
2664                   nnew = mm;
2665                 }
2666                 DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of phi %d", j); u->dump(); });
2667                 _phase->igvn().replace_input_of(u, j, nnew);
2668                 replaced = true;
2669               }
2670             }
2671           }
2672           if (replaced) {
2673             --i;
2674           }
2675         }
2676       } else if ((u->adr_type() == TypePtr::BOTTOM && u->Opcode() != Op_StrInflatedCopy) ||
2677                  u->adr_type() == NULL) {
2678         assert(u->adr_type() != NULL ||
2679                u->Opcode() == Op_Rethrow ||
2680                u->Opcode() == Op_Return ||
2681                u->Opcode() == Op_SafePoint ||
2682                (u->is_CallStaticJava() && u->as_CallStaticJava()->uncommon_trap_request() != 0) ||
2683                (u->is_CallStaticJava() && u->as_CallStaticJava()->_entry_point == OptoRuntime::rethrow_stub()) ||
2684                u->Opcode() == Op_CallLeaf, "");
2685         Node* m = find_mem(_phase->ctrl_or_self(u), u);
2686         if (m != mem) {
2687           mm = allocate_merge_mem(mem, m, _phase->get_ctrl(m));
2688           _phase->igvn().replace_input_of(u, u->find_edge(mem), mm);
2689           --i;
2690         }
2691       } else if (_phase->C->get_alias_index(u->adr_type()) == alias) {
2692         Node* m = find_mem(_phase->ctrl_or_self(u), u);
2693         if (m != mem) {
2694           DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of use"); u->dump(); });
2695           _phase->igvn().replace_input_of(u, u->find_edge(mem), m);
2696           --i;
2697         }
2698       } else if (u->adr_type() != TypePtr::BOTTOM &&
2699                  _memory_nodes[_phase->ctrl_or_self(u)->_idx] == u) {
2700         Node* m = find_mem(_phase->ctrl_or_self(u), u);
2701         assert(m != mem, "");
2702         // u is on the wrong slice...
2703         assert(u->is_ClearArray(), "");
2704         DEBUG_ONLY(if (trace) { tty->print("ZZZ setting memory of use"); u->dump(); });
2705         _phase->igvn().replace_input_of(u, u->find_edge(mem), m);
2706         --i;
2707       }
2708     }
2709   }
2710 #ifdef ASSERT
2711   assert(new_mem->outcnt() > 0, "");
2712   for (int i = 0; i < phis.length(); i++) {
2713     Node* n = phis.at(i);
2714     assert(n->outcnt() > 0, "new phi must have uses now");
2715   }
2716 #endif
2717 }

2726 MergeMemNode* MemoryGraphFixer::clone_merge_mem(Node* u, Node* mem, Node* rep_proj, Node* rep_ctrl, DUIterator& i) const {
2727   MergeMemNode* newmm = NULL;
2728   MergeMemNode* u_mm = u->as_MergeMem();
2729   Node* c = _phase->get_ctrl(u);
2730   if (_phase->is_dominator(c, rep_ctrl)) {
2731     c = rep_ctrl;
2732   } else {
2733     assert(_phase->is_dominator(rep_ctrl, c), "one must dominate the other");
2734   }
2735   if (u->outcnt() == 1) {
2736     if (u->req() > (uint)_alias && u->in(_alias) == mem) {
2737       _phase->igvn().replace_input_of(u, _alias, rep_proj);
2738       --i;
2739     } else {
2740       _phase->igvn().rehash_node_delayed(u);
2741       u_mm->set_memory_at(_alias, rep_proj);
2742     }
2743     newmm = u_mm;
2744     _phase->set_ctrl_and_loop(u, c);
2745   } else {
2746     // can't simply clone u and then change one of its input because
2747     // it adds and then removes an edge which messes with the
2748     // DUIterator
2749     newmm = MergeMemNode::make(u_mm->base_memory());
2750     for (uint j = 0; j < u->req(); j++) {
2751       if (j < newmm->req()) {
2752         if (j == (uint)_alias) {
2753           newmm->set_req(j, rep_proj);
2754         } else if (newmm->in(j) != u->in(j)) {
2755           newmm->set_req(j, u->in(j));
2756         }
2757       } else if (j == (uint)_alias) {
2758         newmm->add_req(rep_proj);
2759       } else {
2760         newmm->add_req(u->in(j));
2761       }
2762     }
2763     if ((uint)_alias >= u->req()) {
2764       newmm->set_memory_at(_alias, rep_proj);
2765     }
2766     _phase->register_new_node(newmm, c);
2767   }
2768   return newmm;
2769 }

2786   uint last = _phase-> C->unique();

2792       if (u->is_MergeMem()) {
2793         MergeMemNode* u_mm = u->as_MergeMem();
2794         if (u_mm->memory_at(_alias) == mem) {
2795           MergeMemNode* newmm = NULL;

2801                 Node* region = uu->in(0);
2802                 int nb = 0;

2822                   newmm = clone_merge_mem(u, mem, rep_proj, rep_ctrl, i);

2835         if (should_process_phi(u)) {
2836           bool replaced = false;
2837           for (uint j = 1; j < u->req(); j++) {
2838             if (u->in(j) == mem && _phase->is_dominator(rep_ctrl, region->in(j))) {
2839               Node* nnew = rep_proj;
2840               if (u->adr_type() == TypePtr::BOTTOM) {
2841                 if (mm == NULL) {
2842                   mm = allocate_merge_mem(mem, rep_proj, rep_ctrl);

2852           }
2853 
2854         }
2855       } else if ((u->adr_type() == TypePtr::BOTTOM && u->Opcode() != Op_StrInflatedCopy) ||
2856                  u->adr_type() == NULL) {
2857         assert(u->adr_type() != NULL ||

2875           _phase->igvn().replace_input_of(u, u->find_edge(mem), rep_proj);
2876           --i;
2877         }
2878       }
2879     }
2880   }

2942 bool ShenandoahLoadReferenceBarrierNode::needs_barrier_impl(PhaseGVN* phase, Node* n, Unique_Node_List &visited) {
2943   if (n == NULL) return false;
2944   if (visited.member(n)) {
2945     return false; // Been there.
2946   }
2947   visited.push(n);
2948 
2949   if (n->is_Allocate()) {
2950     // tty->print_cr("optimize barrier on alloc");
2951     return false;
2952   }
2953   if (n->is_Call()) {
2954     // tty->print_cr("optimize barrier on call");
2955     return false;
2956   }
2957 
2958   const Type* type = phase->type(n);
2959   if (type == Type::TOP) {
2960     return false;
2961   }
2962   if (type->make_ptr()->higher_equal(TypePtr::NULL_PTR)) {
2963     // tty->print_cr("optimize barrier on null");
2964     return false;
2965   }
2966   if (type->make_oopptr() && type->make_oopptr()->const_oop() != NULL) {
2967     // tty->print_cr("optimize barrier on constant");
2968     return false;
2969   }
2970 
2971   switch (n->Opcode()) {
2972     case Op_AddP:
2973       return true; // TODO: Can refine?
2974     case Op_LoadP:
2975     case Op_ShenandoahCompareAndExchangeN:
2976     case Op_ShenandoahCompareAndExchangeP:
2977     case Op_CompareAndExchangeN:
2978     case Op_CompareAndExchangeP:
2979     case Op_GetAndSetN:
2980     case Op_GetAndSetP:
2981       return true;
2982     case Op_Phi: {
2983       for (uint i = 1; i < n->req(); i++) {
2984         if (needs_barrier_impl(phase, n->in(i), visited)) return true;
2985       }
2986       return false;
2987     }
2988     case Op_CheckCastPP:
2989     case Op_CastPP:
2990       return needs_barrier_impl(phase, n->in(1), visited);
2991     case Op_Proj:
2992       return needs_barrier_impl(phase, n->in(0), visited);
2993     case Op_ShenandoahLoadReferenceBarrier:
2994       // tty->print_cr("optimize barrier on barrier");
2995       return false;
2996     case Op_Parm:
2997       // tty->print_cr("optimize barrier on input arg");
2998       return false;
2999     case Op_DecodeN:
3000     case Op_EncodeP:
3001       return needs_barrier_impl(phase, n->in(1), visited);
3002     case Op_LoadN:
3003       return true;
3004     case Op_CMoveN:
3005     case Op_CMoveP:
3006       return needs_barrier_impl(phase, n->in(2), visited) ||
3007              needs_barrier_impl(phase, n->in(3), visited);
3008     case Op_ShenandoahEnqueueBarrier:
3009       return needs_barrier_impl(phase, n->in(1), visited);
3010     case Op_CreateEx:
3011       return false;
3012     default:
3013       break;
3014   }
3015 #ifdef ASSERT
3016   tty->print("need barrier on?: ");
3017   tty->print_cr("ins:");
3018   n->dump(2);
3019   tty->print_cr("outs:");
3020   n->dump(-2);
3021   ShouldNotReachHere();
3022 #endif
3023   return true;
3024 }

3026 bool ShenandoahLoadReferenceBarrierNode::is_redundant() {
3027   Unique_Node_List visited;
3028   Node_Stack stack(0);
3029   stack.push(this, 0);
3030 
3031   // Check if the barrier is actually useful: go over nodes looking for useful uses
3032   // (e.g. memory accesses). Stop once we detected a required use. Otherwise, walk
3033   // until we ran out of nodes, and then declare the barrier redundant.
3034   while (stack.size() > 0) {
3035     Node* n = stack.node();
3036     if (visited.member(n)) {
3037       stack.pop();
3038       continue;
3039     }
3040     visited.push(n);
3041     bool visit_users = false;
3042     switch (n->Opcode()) {
3043       case Op_CallStaticJava:
3044       case Op_CallDynamicJava:
3045       case Op_CallLeaf:
3046       case Op_CallLeafNoFP:
3047       case Op_CompareAndSwapL:
3048       case Op_CompareAndSwapI:
3049       case Op_CompareAndSwapB:
3050       case Op_CompareAndSwapS:
3051       case Op_CompareAndSwapN:
3052       case Op_CompareAndSwapP:
3053       case Op_CompareAndExchangeL:
3054       case Op_CompareAndExchangeI:
3055       case Op_CompareAndExchangeB:
3056       case Op_CompareAndExchangeS:
3057       case Op_CompareAndExchangeN:
3058       case Op_CompareAndExchangeP:
3059       case Op_WeakCompareAndSwapL:
3060       case Op_WeakCompareAndSwapI:
3061       case Op_WeakCompareAndSwapB:
3062       case Op_WeakCompareAndSwapS:
