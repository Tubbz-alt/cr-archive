 739 address DynamicArchiveBuilder::reserve_space_and_init_buffer_to_target_delta() {
 740   size_t total = estimate_archive_size();
 741   ReservedSpace rs = MetaspaceShared::reserve_shared_space(total);
 742   if (!rs.is_reserved()) {
 743     log_error(cds, dynamic)("Failed to reserve %d bytes of output buffer.", (int)total);
 744     vm_direct_exit(0);
 745   }
 746 
 747   address buffer_base = (address)rs.base();
 748   log_info(cds, dynamic)("Reserved output buffer space at    : " PTR_FORMAT " [%d bytes]",
 749                          p2i(buffer_base), (int)total);
 750   MetaspaceShared::set_shared_rs(rs);
 751 
 752   // At run time, we will mmap the dynamic archive at target_space_bottom.
 753   // However, at dump time, we may not be able to write into the target_space,
 754   // as it's occupied by dynamically loaded Klasses. So we allocate a buffer
 755   // at an arbitrary location chosen by the OS. We will write all the dynamically
 756   // archived classes into this buffer. At the final stage of dumping, we relocate
 757   // all pointers that are inside the buffer_space to point to their (runtime)
 758   // target location inside thetarget_space.
 759   address target_space_bottom =
 760     (address)align_up(MetaspaceShared::shared_metaspace_top(), reserve_alignment());
 761   _buffer_to_target_delta = intx(target_space_bottom) - intx(buffer_base);
