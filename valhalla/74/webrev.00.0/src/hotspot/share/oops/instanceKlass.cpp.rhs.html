 140 static inline bool is_class_loader(const Symbol* class_name,
 141                                    const ClassFileParser& parser) {
 142   assert(class_name != NULL, "invariant");
 143 
 144   if (class_name == vmSymbols::java_lang_ClassLoader()) {
 145     return true;
 146   }
 147 
 148   if (SystemDictionary::ClassLoader_klass_loaded()) {
 149     const Klass* const super_klass = parser.super_klass();
 150     if (super_klass != NULL) {
 151       if (super_klass->is_subtype_of(SystemDictionary::ClassLoader_klass())) {
 152         return true;
 153       }
 154     }
 155   }
 156   return false;
 157 }

 159 // private: called to verify that k is a static member of this nest.
 160 // We know that k is an instance class in the same package and hence the
 161 // same classloader.
 162 bool InstanceKlass::has_nest_member(InstanceKlass* k, TRAPS) const {
 163   assert(!is_hidden(), "unexpected hidden class");
 164   if (_nest_members == NULL || _nest_members == Universe::the_empty_short_array()) {
 165     if (log_is_enabled(Trace, class, nestmates)) {
 166       ResourceMark rm(THREAD);
 167       log_trace(class, nestmates)("Checked nest membership of %s in non-nest-host class %s",
 168                                   k->external_name(), this->external_name());
 169     }
 170     return false;
 171   }
 172 
 173   if (log_is_enabled(Trace, class, nestmates)) {
 174     ResourceMark rm(THREAD);
 175     log_trace(class, nestmates)("Checking nest membership of %s in %s",
 176                                 k->external_name(), this->external_name());
 177   }
 178 
 179   // Check for a resolved cp entry , else fall back to a name check.
 180   // We don't want to resolve any class other than the one being checked.
 181   for (int i = 0; i < _nest_members->length(); i++) {
 182     int cp_index = _nest_members->at(i);
 183     if (_constants->tag_at(cp_index).is_klass()) {
 184       Klass* k2 = _constants->klass_at(cp_index, THREAD);
 185       assert(!HAS_PENDING_EXCEPTION || PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass()),
 186              "Exceptions should not be possible here");
 187       if (k2 == k) {
 188         log_trace(class, nestmates)("- class is listed at nest_members[%d] => cp[%d]", i, cp_index);
 189         return true;
 190       }
 191     }
 192     else {
 193       Symbol* name = _constants->klass_name_at(cp_index);
 194       if (name == k->name()) {
 195         log_trace(class, nestmates)("- Found it at nest_members[%d] => cp[%d]", i, cp_index);
 196 
 197         // Names match so check actual klass. This may trigger class loading if
 198         // it doesn't match though that should be impossible as it means one classloader
 199         // has defined two different classes with the same name! A compiler thread won't be
 200         // able to perform that loading but we can't exclude the compiler threads from
 201         // executing this logic. But it should actually be impossible to trigger loading here.
 202         Klass* k2 = _constants->klass_at(cp_index, THREAD);
 203         assert(!HAS_PENDING_EXCEPTION || PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass()),
 204                "Exceptions should not be possible here");
 205         if (k2 == k) {
 206           log_trace(class, nestmates)("- class is listed as a nest member");
 207           return true;
 208         }
 209         else {
 210           // same name but different klass!
 211           log_trace(class, nestmates)(" - klass comparison failed!");
 212           // can't have two names the same, so we're done
 213           return false;
 214         }
 215       }
 216     }
 217   }
 218   log_trace(class, nestmates)("- class is NOT a nest member!");
 219   return false;
 220 }

 222 // Return nest-host class, resolving, validating and saving it if needed.
 223 // In cases where this is called from a thread that cannot do classloading
 224 // (such as a native JIT thread) then we simply return NULL, which in turn
 225 // causes the access check to return false. Such code will retry the access
 226 // from a more suitable environment later. Otherwise the _nest_host is always
 227 // set once this method returns.
 228 // Any errors from nest-host resolution must be preserved so they can be queried
 229 // from higher-level access checking code, and reported as part of access checking
 230 // exceptions.
 231 // VirtualMachineErrors are propagated with a NULL return.
 232 // Under any conditions where the _nest_host can be set to non-NULL the resulting
 233 // value of it and, if applicable, the nest host resolution/validation error,
 234 // are idempotent.
 235 InstanceKlass* InstanceKlass::nest_host(TRAPS) {
 236   InstanceKlass* nest_host_k = _nest_host;
 237   if (nest_host_k != NULL) {
 238     return nest_host_k;
 239   }
 240 
 241   ResourceMark rm(THREAD);
 242 
 243   // need to resolve and save our nest-host class.
 244   if (_nest_host_index != 0) { // we have a real nest_host
 245     // Before trying to resolve check if we're in a suitable context
 246     if (!THREAD->can_call_java() && !_constants->tag_at(_nest_host_index).is_klass()) {
 247       log_trace(class, nestmates)("Rejected resolution of nest-host of %s in unsuitable thread",
 248                                   this->external_name());
 249       return NULL; // sentinel to say "try again from a different context"
 250     }
 251 
 252     log_trace(class, nestmates)("Resolving nest-host of %s using cp entry for %s",
 253                                 this->external_name(),
 254                                 _constants->klass_name_at(_nest_host_index)->as_C_string());
 255 
 256     Klass* k = _constants->klass_at(_nest_host_index, THREAD);
 257     if (HAS_PENDING_EXCEPTION) {
 258       if (PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass())) {
 259         return NULL; // propagate VMEs
 260       }
 261       stringStream ss;
 262       char* target_host_class = _constants->klass_name_at(_nest_host_index)->as_C_string();
 263       ss.print("Nest host resolution of %s with host %s failed: ",
 264                this->external_name(), target_host_class);
 265       java_lang_Throwable::print(PENDING_EXCEPTION, &ss);
 266       const char* msg = ss.as_string(true /* on C-heap */);
 267       constantPoolHandle cph(THREAD, constants());
 268       SystemDictionary::add_nest_host_error(cph, _nest_host_index, msg);
 269       CLEAR_PENDING_EXCEPTION;
 270 
 271       log_trace(class, nestmates)("%s", msg);
 272     } else {
 273       // A valid nest-host is an instance class in the current package that lists this
 274       // class as a nest member. If any of these conditions are not met the class is
 275       // its own nest-host.
 276       const char* error = NULL;
 277 
 278       // JVMS 5.4.4 indicates package check comes first
 279       if (is_same_class_package(k)) {
 280         // Now check actual membership. We can't be a member if our "host" is
 281         // not an instance class.
 282         if (k->is_instance_klass()) {
 283           nest_host_k = InstanceKlass::cast(k);
 284           bool is_member = nest_host_k->has_nest_member(this, THREAD);
 285           // exception is rare, perhaps impossible
 286           if (!HAS_PENDING_EXCEPTION) {
 287             if (is_member) {
 288               _nest_host = nest_host_k; // save resolved nest-host value
 289 
 290               log_trace(class, nestmates)("Resolved nest-host of %s to %s",
 291                                           this->external_name(), k->external_name());
 292               return nest_host_k;
 293             } else {
 294               error = "current type is not listed as a nest member";
 295             }
 296           } else {
 297             if (PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass())) {
 298               return NULL; // propagate VMEs
 299             }
 300             stringStream ss;
 301             ss.print("exception on member check: ");
 302             java_lang_Throwable::print(PENDING_EXCEPTION, &ss);
 303             error = ss.as_string();
 304           }
 305         } else {
 306           error = "host is not an instance class";
 307         }
 308       } else {
 309         error = "types are in different packages";
 310       }
 311 
 312       // something went wrong, so record what and log it
 313       {
 314         stringStream ss;
 315         ss.print("Type %s (loader: %s) is not a nest member of type %s (loader: %s): %s",
 316                  this->external_name(),
 317                  this->class_loader_data()->loader_name_and_id(),
 318                  k->external_name(),
 319                  k->class_loader_data()->loader_name_and_id(),
 320                  error);
 321         const char* msg = ss.as_string(true /* on C-heap */);
 322         constantPoolHandle cph(THREAD, constants());
 323         SystemDictionary::add_nest_host_error(cph, _nest_host_index, msg);
 324         log_trace(class, nestmates)("%s", msg);
 325       }
 326     }
 327   } else {
 328     log_trace(class, nestmates)("Type %s is not part of a nest: setting nest-host to self",
 329                                 this->external_name());
 330   }
 331 
 332   // Either not in an explicit nest, or else an error occurred, so
 333   // the nest-host is set to `this`. Any thread that sees this assignment
 334   // will also see any setting of nest_host_error(), if applicable.
 335   return (_nest_host = this);
 336 }

 338 // Dynamic nest member support: set this class's nest host to the given class.
 339 // This occurs as part of the class definition, as soon as the instanceKlass
 340 // has been created and doesn't require further resolution. The code:
 341 //    lookup().defineHiddenClass(bytes_for_X, NESTMATE);
 342 // results in:
 343 //    class_of_X.set_nest_host(lookup().lookupClass().getNestHost())
 344 // If it has an explicit _nest_host_index or _nest_members, these will be ignored.
 345 // We also know the "host" is a valid nest-host in the same package so we can
 346 // assert some of those facts.
 347 void InstanceKlass::set_nest_host(InstanceKlass* host, TRAPS) {
 348   assert(is_hidden(), "must be a hidden class");
 349   assert(host != NULL, "NULL nest host specified");
 350   assert(_nest_host == NULL, "current class has resolved nest-host");
 351   assert(nest_host_error(THREAD) == NULL, "unexpected nest host resolution error exists: %s",
 352          nest_host_error(THREAD));
 353   assert((host->_nest_host == NULL && host->_nest_host_index == 0) ||
 354          (host->_nest_host == host), "proposed host is not a valid nest-host");
 355   // Can't assert this as package is not set yet:
 356   // assert(is_same_class_package(host), "proposed host is in wrong package");
 357 
 358   if (log_is_enabled(Trace, class, nestmates)) {
 359     ResourceMark rm(THREAD);
 360     const char* msg = "";
 361     // a hidden class does not expect a statically defined nest-host
 362     if (_nest_host_index > 0) {
 363       msg = "(the NestHost attribute in the current class is ignored)";
 364     } else if (_nest_members != NULL && _nest_members != Universe::the_empty_short_array()) {
 365       msg = "(the NestMembers attribute in the current class is ignored)";
 366     }
 367     log_trace(class, nestmates)("Injected type %s into the nest of %s %s",
 368                                 this->external_name(),
 369                                 host->external_name(),
 370                                 msg);
 371   }
 372   // set dynamic nest host
 373   _nest_host = host;
 374   // Record dependency to keep nest host from being unloaded before this class.
 375   ClassLoaderData* this_key = class_loader_data();
 376   this_key->record_dependency(host);
 377 }

 379 // check if 'this' and k are nestmates (same nest_host), or k is our nest_host,
 380 // or we are k's nest_host - all of which is covered by comparing the two
 381 // resolved_nest_hosts.
 382 // Any exceptions (i.e. VMEs) are propagated.
 383 bool InstanceKlass::has_nestmate_access_to(InstanceKlass* k, TRAPS) {
 384 
 385   assert(this != k, "this should be handled by higher-level code");
 386 
 387   // Per JVMS 5.4.4 we first resolve and validate the current class, then
 388   // the target class k.
 389 
 390   InstanceKlass* cur_host = nest_host(CHECK_false);
 391   if (cur_host == NULL) {
 392     return false;
 393   }
 394 
 395   Klass* k_nest_host = k->nest_host(CHECK_false);
 396   if (k_nest_host == NULL) {
 397     return false;
 398   }
 399 
 400   bool access = (cur_host == k_nest_host);
 401 
 402   ResourceMark rm(THREAD);
 403   log_trace(class, nestmates)("Class %s does %shave nestmate access to %s",
 404                               this->external_name(),
 405                               access ? "" : "NOT ",
 406                               k->external_name());
 407   return access;
 408 }

 410 const char* InstanceKlass::nest_host_error(TRAPS) {
 411   if (_nest_host_index == 0) {
 412     return NULL;
 413   } else {
 414     constantPoolHandle cph(THREAD, constants());
 415     return SystemDictionary::find_nest_host_error(cph, (int)_nest_host_index);
 416   }

 419 InstanceKlass* InstanceKlass::allocate_instance_klass(const ClassFileParser& parser, TRAPS) {
 420   bool is_hidden_or_anonymous = parser.is_hidden() || parser.is_unsafe_anonymous();
 421   const int size = InstanceKlass::size(parser.vtable_size(),
 422                                        parser.itable_size(),
 423                                        nonstatic_oop_map_size(parser.total_oop_map_count()),
 424                                        parser.is_interface(),
 425                                        parser.is_unsafe_anonymous(),
 426                                        should_store_fingerprint(is_hidden_or_anonymous),
 427                                        parser.has_flattenable_fields() ? parser.java_fields_count() : 0,
 428                                        parser.is_inline_type());
 429 
 430   const Symbol* const class_name = parser.class_name();
 431   assert(class_name != NULL, "invariant");
 432   ClassLoaderData* loader_data = parser.loader_data();
 433   assert(loader_data != NULL, "invariant");
 434 
 435   InstanceKlass* ik;
 436 
 437   // Allocation
 438   if (REF_NONE == parser.reference_type()) {
 439     if (class_name == vmSymbols::java_lang_Class()) {
 440       // mirror
 441       ik = new (loader_data, size, THREAD) InstanceMirrorKlass(parser);
 442     } else if (is_class_loader(class_name, parser)) {
 443       // class loader
 444       ik = new (loader_data, size, THREAD) InstanceClassLoaderKlass(parser);
 445     } else if (parser.is_inline_type()) {
 446       // inline type
 447       ik = new (loader_data, size, THREAD) ValueKlass(parser);
 448     } else {
 449       // normal
 450       ik = new (loader_data, size, THREAD) InstanceKlass(parser, InstanceKlass::_kind_other);
 451     }
 452   } else {
 453     // reference
 454     ik = new (loader_data, size, THREAD) InstanceRefKlass(parser);
 455   }
 456 
 457   // Check for pending exception before adding to the loader data and incrementing
 458   // class count.  Can get OOM here.
 459   if (HAS_PENDING_EXCEPTION) {
 460     return NULL;
 461   }
 462 
 463 #ifdef ASSERT
 464   assert(ik->size() == size, "");
 465   ik->bounds_check((address) ik->start_of_vtable(), false, size);
 466   ik->bounds_check((address) ik->start_of_itable(), false, size);
 467   ik->bounds_check((address) ik->end_of_itable(), true, size);
 468   ik->bounds_check((address) ik->end_of_nonstatic_oop_maps(), true, size);
 469 #endif //ASSERT
 470   return ik;
 471 }

 475   const char* bad = NULL;

 488     return true;
 489   }
 490   tty->print_cr("%s object bounds: " INTPTR_FORMAT " [" INTPTR_FORMAT ".." INTPTR_FORMAT "]",
 491       bad, (intptr_t)addr, (intptr_t)this, (intptr_t)end);
 492   Verbose = WizardMode = true; this->print(); //@@
 493   return false;
 494 }

 497 // copy method ordering from resource area to Metaspace
 498 void InstanceKlass::copy_method_ordering(const intArray* m, TRAPS) {
 499   if (m != NULL) {
 500     // allocate a new array and copy contents (memcpy?)
 501     _method_ordering = MetadataFactory::new_array<int>(class_loader_data(), m->length(), CHECK);
 502     for (int i = 0; i < m->length(); i++) {
 503       _method_ordering->at_put(i, m->at(i));
 504     }
 505   } else {
 506     _method_ordering = Universe::the_empty_int_array();
 507   }

 518 InstanceKlass::InstanceKlass(const ClassFileParser& parser, unsigned kind, KlassID id) :
 519   Klass(id),
 520   _nest_members(NULL),
 521   _nest_host(NULL),
 522   _record_components(NULL),
 523   _static_field_size(parser.static_field_size()),
 524   _nonstatic_oop_map_size(nonstatic_oop_map_size(parser.total_oop_map_count())),
 525   _itable_len(parser.itable_size()),
 526   _nest_host_index(0),
 527   _init_state(allocated),
 528   _reference_type(parser.reference_type()),
 529   _init_thread(NULL),
 530   _value_field_klasses(NULL),
 531   _adr_valueklass_fixed_block(NULL)
 532 {
 533   set_vtable_length(parser.vtable_size());
 534   set_kind(kind);
 535   set_access_flags(parser.access_flags());
 536   if (parser.is_hidden()) set_is_hidden();
 537   set_is_unsafe_anonymous(parser.is_unsafe_anonymous());
 538   set_layout_helper(Klass::instance_layout_helper(parser.layout_size(),
 539                                                     false));
 540     if (parser.has_flattenable_fields()) {
 541       set_has_inline_fields();
 542     }
 543     _java_fields_count = parser.java_fields_count();
 544 
 545     assert(NULL == _methods, "underlying memory not zeroed?");
 546     assert(is_instance_klass(), "is layout incorrect?");
 547     assert(size_helper() == parser.layout_size(), "incorrect size_helper?");
 548 
 549   // Set biased locking bit for all instances of this class; it will be
 550   // cleared if revocation occurs too often for this type
 551   if (UseBiasedLocking && BiasedLocking::enabled()) {
 552     set_prototype_header(markWord::biased_locking_prototype());
 553   }
 554   if (has_inline_fields()) {
 555     _value_field_klasses = (const Klass**) adr_value_fields_klasses();
 556   }

 611 // This function deallocates the metadata and C heap pointers that the
 612 // InstanceKlass points to.
 613 void InstanceKlass::deallocate_contents(ClassLoaderData* loader_data) {
 614 
 615   // Orphan the mirror first, CMS thinks it's still live.
 616   if (java_mirror() != NULL) {
 617     java_lang_Class::set_klass(java_mirror(), NULL);
 618   }
 619 
 620   // Also remove mirror from handles
 621   loader_data->remove_handle(_java_mirror);
 622 
 623   // Need to take this class off the class loader data list.
 624   loader_data->remove_class(this);
 625 
 626   // The array_klass for this class is created later, after error handling.
 627   // For class redefinition, we keep the original class so this scratch class
 628   // doesn't have an array class.  Either way, assert that there is nothing
 629   // to deallocate.
 630   assert(array_klasses() == NULL, "array classes shouldn't be created for this class yet");
 631 
 632   // Release C heap allocated data that this points to, which includes
 633   // reference counting symbol names.
 634   release_C_heap_structures_internal();
 635 
 636   deallocate_methods(loader_data, methods());
 637   set_methods(NULL);
 638 
 639   deallocate_record_components(loader_data, record_components());
 640   set_record_components(NULL);
 641 
 642   if (method_ordering() != NULL &&
 643       method_ordering() != Universe::the_empty_int_array() &&
 644       !method_ordering()->is_shared()) {
 645     MetadataFactory::free_array<int>(loader_data, method_ordering());
 646   }
 647   set_method_ordering(NULL);
 648 
 649   // default methods can be empty
 650   if (default_methods() != NULL &&
 651       default_methods() != Universe::the_empty_method_array() &&
 652       !default_methods()->is_shared()) {
 653     MetadataFactory::free_array<Method*>(loader_data, default_methods());
 654   }
 655   // Do NOT deallocate the default methods, they are owned by superinterfaces.
 656   set_default_methods(NULL);
 657 
 658   // default methods vtable indices can be empty
 659   if (default_vtable_indices() != NULL &&
 660       !default_vtable_indices()->is_shared()) {
 661     MetadataFactory::free_array<int>(loader_data, default_vtable_indices());
 662   }
 663   set_default_vtable_indices(NULL);
 664 
 665 
 666   // This array is in Klass, but remove it with the InstanceKlass since
 667   // this place would be the only caller and it can share memory with transitive
 668   // interfaces.
 669   if (secondary_supers() != NULL &&
 670       secondary_supers() != Universe::the_empty_klass_array() &&
 671       // see comments in compute_secondary_supers about the following cast
 672       (address)(secondary_supers()) != (address)(transitive_interfaces()) &&
 673       !secondary_supers()->is_shared()) {
 674     MetadataFactory::free_array<Klass*>(loader_data, secondary_supers());
 675   }
 676   set_secondary_supers(NULL);
 677 
 678   deallocate_interfaces(loader_data, super(), local_interfaces(), transitive_interfaces());
 679   set_transitive_interfaces(NULL);
 680   set_local_interfaces(NULL);
 681 
 682   if (fields() != NULL && !fields()->is_shared()) {
 683     MetadataFactory::free_array<jushort>(loader_data, fields());
 684   }
 685   set_fields(NULL, 0);
 686 
 687   // If a method from a redefined class is using this constant pool, don't
 688   // delete it, yet.  The new class's previous version will point to this.
 689   if (constants() != NULL) {
 690     assert (!constants()->on_stack(), "shouldn't be called if anything is onstack");
 691     if (!constants()->is_shared()) {
 692       MetadataFactory::free_metadata(loader_data, constants());
 693     }
 694     // Delete any cached resolution errors for the constant pool
 695     SystemDictionary::delete_resolution_error(constants());
 696 
 697     set_constants(NULL);
 698   }
 699 
 700   if (inner_classes() != NULL &&
 701       inner_classes() != Universe::the_empty_short_array() &&
 702       !inner_classes()->is_shared()) {
 703     MetadataFactory::free_array<jushort>(loader_data, inner_classes());
 704   }
 705   set_inner_classes(NULL);
 706 
 707   if (nest_members() != NULL &&
 708       nest_members() != Universe::the_empty_short_array() &&
 709       !nest_members()->is_shared()) {
 710     MetadataFactory::free_array<jushort>(loader_data, nest_members());
 711   }
 712   set_nest_members(NULL);
 713 
 714   // We should deallocate the Annotations instance if it's not in shared spaces.
 715   if (annotations() != NULL && !annotations()->is_shared()) {
 716     MetadataFactory::free_metadata(loader_data, annotations());
 717   }
 718   set_annotations(NULL);
 719 
 720   if (Arguments::is_dumping_archive()) {
 721     SystemDictionaryShared::remove_dumptime_info(this);
 722   }
 723 }

 733 void InstanceKlass::eager_initialize(Thread *thread) {
 734   if (!EagerInitialization) return;
 735 
 736   if (this->is_not_initialized()) {
 737     // abort if the the class has a class initializer
 738     if (this->class_initializer() != NULL) return;
 739 
 740     // abort if it is java.lang.Object (initialization is handled in genesis)
 741     Klass* super_klass = super();

 743 
 744     // abort if the super class should be initialized
 745     if (!InstanceKlass::cast(super_klass)->is_initialized()) return;
 746 
 747     // call body to expose the this pointer
 748     eager_initialize_impl();
 749   }

 777 // threads to get this lock will see a null lock and will not lock.

 787 void InstanceKlass::eager_initialize_impl() {
 788   EXCEPTION_MARK;
 789   HandleMark hm(THREAD);
 790   Handle h_init_lock(THREAD, init_lock());
 791   ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
 792 
 793   // abort if someone beat us to the initialization
 794   if (!is_not_initialized()) return;  // note: not equivalent to is_initialized()
 795 
 796   ClassState old_state = init_state();
 797   link_class_impl(THREAD);
 798   if (HAS_PENDING_EXCEPTION) {
 799     CLEAR_PENDING_EXCEPTION;
 800     // Abort if linking the class throws an exception.
 801 
 802     // Use a test to avoid redundantly resetting the state if there's
 803     // no change.  Set_init_state() asserts that state changes make
 804     // progress, whereas here we might just be spinning in place.
 805     if (old_state != _init_state)
 806       set_init_state(old_state);
 807   } else {
 808     // linking successfull, mark class as initialized
 809     set_init_state(fully_initialized);
 810     fence_and_clear_init_lock();
 811     // trace
 812     if (log_is_enabled(Info, class, init)) {
 813       ResourceMark rm(THREAD);
 814       log_info(class, init)("[Initialized %s without side effects]", external_name());
 815     }
 816   }
 817 }

 857 bool InstanceKlass::link_class_impl(TRAPS) {
 858   if (DumpSharedSpaces && SystemDictionaryShared::has_class_failed_verification(this)) {
 859     // This is for CDS dumping phase only -- we use the in_error_state to indicate that
 860     // the class has failed verification. Throwing the NoClassDefFoundError here is just
 861     // a convenient way to stop repeat attempts to verify the same (bad) class.
 862     //
 863     // Note that the NoClassDefFoundError is not part of the JLS, and should not be thrown
 864     // if we are executing Java code. This is not a problem for CDS dumping phase since
 865     // it doesn't execute any Java code.
 866     ResourceMark rm(THREAD);
 867     Exceptions::fthrow(THREAD_AND_LOCATION,
 868                        vmSymbols::java_lang_NoClassDefFoundError(),
 869                        "Class %s, or one of its supertypes, failed class initialization",
 870                        external_name());
 871     return false;
 872   }
 873   // return if already verified
 874   if (is_linked()) {
 875     return true;
 876   }
 877 
 878   // Timing
 879   // timer handles recursion
 880   assert(THREAD->is_Java_thread(), "non-JavaThread in link_class_impl");
 881   JavaThread* jt = (JavaThread*)THREAD;
 882 
 883   // link super class before linking this class
 884   Klass* super_klass = super();
 885   if (super_klass != NULL) {
 886     if (super_klass->is_interface()) {  // check if super class is an interface
 887       ResourceMark rm(THREAD);
 888       Exceptions::fthrow(
 889         THREAD_AND_LOCATION,
 890         vmSymbols::java_lang_IncompatibleClassChangeError(),
 891         "class %s has interface %s as super class",
 892         external_name(),
 893         super_klass->external_name()
 894       );
 895       return false;
 896     }
 897 
 898     InstanceKlass* ik_super = InstanceKlass::cast(super_klass);
 899     ik_super->link_class_impl(CHECK_false);
 900   }
 901 
 902   // link all interfaces implemented by this class before linking this class
 903   Array<InstanceKlass*>* interfaces = local_interfaces();
 904   int num_interfaces = interfaces->length();
 905   for (int index = 0; index < num_interfaces; index++) {
 906     InstanceKlass* interk = interfaces->at(index);
 907     interk->link_class_impl(CHECK_false);
 908   }
 909 
 910 
 911   // If a class declares a method that uses an inline class as an argument
 912   // type or return inline type, this inline class must be loaded during the
 913   // linking of this class because size and properties of the inline class
 914   // must be known in order to be able to perform inline type optimizations.
 915   // The implementation below is an approximation of this rule, the code
 916   // iterates over all methods of the current class (including overridden
 917   // methods), not only the methods declared by this class. This
 918   // approximation makes the code simpler, and doesn't change the semantic
 919   // because classes declaring methods overridden by the current class are
 920   // linked (and have performed their own pre-loading) before the linking
 921   // of the current class.
 922 
 923 
 924   // Note:
 925   // Inline class types used for flattenable fields are loaded during
 926   // the loading phase (see ClassFileParser::post_process_parsed_stream()).
 927   // Inline class types used as element types for array creation
 928   // are not pre-loaded. Their loading is triggered by either anewarray
 929   // or multianewarray bytecodes.
 930 
 931   // Could it be possible to do the following processing only if the
 932   // class uses inline types?
 933   {
 934     ResourceMark rm(THREAD);
 935     for (int i = 0; i < methods()->length(); i++) {
 936       Method* m = methods()->at(i);
 937       for (SignatureStream ss(m->signature()); !ss.is_done(); ss.next()) {
 938         if (ss.is_reference()) {
 939           if (ss.is_array()) {
 940             ss.skip_array_prefix();
 941           }
 942           if (ss.type() == T_VALUETYPE) {
 943             Symbol* symb = ss.as_symbol();
 944 
 945             oop loader = class_loader();
 946             oop protection_domain = this->protection_domain();
 947             Klass* klass = SystemDictionary::resolve_or_fail(symb,
 948                                                              Handle(THREAD, loader), Handle(THREAD, protection_domain), true,
 949                                                              CHECK_false);
 950             if (klass == NULL) {
 951               THROW_(vmSymbols::java_lang_LinkageError(), false);
 952             }
 953             if (!klass->is_value()) {
 954               Exceptions::fthrow(
 955                 THREAD_AND_LOCATION,
 956                 vmSymbols::java_lang_IncompatibleClassChangeError(),
 957                 "class %s is not an inline type",
 958                 klass->external_name());
 959             }
 960           }
 961         }
 962       }
 963     }
 964   }
 965 
 966   // in case the class is linked in the process of linking its superclasses
 967   if (is_linked()) {
 968     return true;
 969   }
 970 
 971   // trace only the link time for this klass that includes
 972   // the verification time
 973   PerfClassTraceTime vmtimer(ClassLoader::perf_class_link_time(),
 974                              ClassLoader::perf_class_link_selftime(),
 975                              ClassLoader::perf_classes_linked(),
 976                              jt->get_thread_stat()->perf_recursion_counts_addr(),
 977                              jt->get_thread_stat()->perf_timers_addr(),
 978                              PerfClassTraceTime::CLASS_LINK);
 979 
 980   // verification & rewriting
 981   {
 982     HandleMark hm(THREAD);
 983     Handle h_init_lock(THREAD, init_lock());
 984     ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
 985     // rewritten will have been set if loader constraint error found
 986     // on an earlier link attempt
 987     // don't verify or rewrite if already rewritten
 988     //
 989 
 990     if (!is_linked()) {
 991       if (!is_rewritten()) {
 992         {
 993           bool verify_ok = verify_code(THREAD);
 994           if (!verify_ok) {
 995             return false;
 996           }
 997         }
 998 
 999         // Just in case a side-effect of verify linked this class already
1000         // (which can sometimes happen since the verifier loads classes
1001         // using custom class loaders, which are free to initialize things)
1002         if (is_linked()) {
1003           return true;
1004         }
1005 
1006         // also sets rewritten
1007         rewrite_class(CHECK_false);
1008       } else if (is_shared()) {
1009         SystemDictionaryShared::check_verification_constraints(this, CHECK_false);
1010       }
1011 
1012       // relocate jsrs and link methods after they are all rewritten
1013       link_methods(CHECK_false);
1014 
1015       // Initialize the vtable and interface table after
1016       // methods have been rewritten since rewrite may
1017       // fabricate new Method*s.
1018       // also does loader constraint checking
1019       //
1020       // initialize_vtable and initialize_itable need to be rerun
1021       // for a shared class if
1022       // 1) the class is loaded by custom class loader or
1023       // 2) the class is loaded by built-in class loader but failed to add archived loader constraints
1024       bool need_init_table = true;
1025       if (is_shared() && SystemDictionaryShared::check_linking_constraints(this, THREAD)) {
1026         need_init_table = false;
1027       }
1028       if (need_init_table) {
1029         vtable().initialize_vtable(true, CHECK_false);
1030         itable().initialize_itable(true, CHECK_false);
1031       }
1032 #ifdef ASSERT
1033       vtable().verify(tty, true);
1034       // In case itable verification is ever added.
1035       // itable().verify(tty, true);
1036 #endif
1037 
1038       set_init_state(linked);
1039       if (JvmtiExport::should_post_class_prepare()) {
1040         Thread *thread = THREAD;
1041         assert(thread->is_Java_thread(), "thread->is_Java_thread()");
1042         JvmtiExport::post_class_prepare((JavaThread *) thread, this);
1043       }
1044     }
1045   }
1046   return true;
1047 }

1075 // Eagerly initialize superinterfaces that declare default methods (concrete instance: any access)
1076 void InstanceKlass::initialize_super_interfaces(TRAPS) {
1077   assert (has_nonstatic_concrete_methods(), "caller should have checked this");
1078   for (int i = 0; i < local_interfaces()->length(); ++i) {
1079     InstanceKlass* ik = local_interfaces()->at(i);
1080 
1081     // Initialization is depth first search ie. we start with top of the inheritance tree
1082     // has_nonstatic_concrete_methods drives searching superinterfaces since it
1083     // means has_nonstatic_concrete_methods in its superinterface hierarchy
1084     if (ik->has_nonstatic_concrete_methods()) {
1085       ik->initialize_super_interfaces(CHECK);
1086     }
1087 
1088     // Only initialize() interfaces that "declare" concrete methods.
1089     if (ik->should_be_initialized() && ik->declares_nonstatic_concrete_methods()) {
1090       ik->initialize(CHECK);
1091     }
1092   }

1095 void InstanceKlass::initialize_impl(TRAPS) {
1096   HandleMark hm(THREAD);
1097 
1098   // Make sure klass is linked (verified) before initialization
1099   // A class could already be verified, since it has been reflected upon.
1100   link_class(CHECK);
1101 
1102   DTRACE_CLASSINIT_PROBE(required, -1);
1103 
1104   bool wait = false;
1105 
1106   assert(THREAD->is_Java_thread(), "non-JavaThread in initialize_impl");
1107   JavaThread* jt = (JavaThread*)THREAD;
1108 
1109   // refer to the JVM book page 47 for description of steps
1110   // Step 1
1111   {
1112     Handle h_init_lock(THREAD, init_lock());
1113     ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
1114 
1115     // Step 2
1116     // If we were to use wait() instead of waitInterruptibly() then
1117     // we might end up throwing IE from link/symbol resolution sites
1118     // that aren't expected to throw.  This would wreak havoc.  See 6320309.
1119     while (is_being_initialized() && !is_reentrant_initialization(jt)) {
1120       wait = true;
1121       jt->set_class_to_be_initialized(this);
1122       ol.wait_uninterruptibly(jt);
1123       jt->set_class_to_be_initialized(NULL);
1124     }
1125 
1126     // Step 3
1127     if (is_being_initialized() && is_reentrant_initialization(jt)) {
1128       DTRACE_CLASSINIT_PROBE_WAIT(recursive, -1, wait);
1129       return;
1130     }
1131 
1132     // Step 4
1133     if (is_initialized()) {
1134       DTRACE_CLASSINIT_PROBE_WAIT(concurrent, -1, wait);
1135       return;
1136     }
1137 
1138     // Step 5
1139     if (is_in_error_state()) {
1140       DTRACE_CLASSINIT_PROBE_WAIT(erroneous, -1, wait);
1141       ResourceMark rm(THREAD);
1142       const char* desc = "Could not initialize class ";
1143       const char* className = external_name();
1144       size_t msglen = strlen(desc) + strlen(className) + 1;
1145       char* message = NEW_RESOURCE_ARRAY(char, msglen);
1146       if (NULL == message) {
1147         // Out of memory: can't create detailed error message
1148           THROW_MSG(vmSymbols::java_lang_NoClassDefFoundError(), className);
1149       } else {
1150         jio_snprintf(message, msglen, "%s%s", desc, className);
1151           THROW_MSG(vmSymbols::java_lang_NoClassDefFoundError(), message);
1152       }
1153     }
1154 
1155     // Step 6
1156     set_init_state(being_initialized);
1157     set_init_thread(jt);
1158   }
1159 
1160   // Step 7
1161   // Next, if C is a class rather than an interface, initialize it's super class and super
1162   // interfaces.
1163   if (!is_interface()) {
1164     Klass* super_klass = super();
1165     if (super_klass != NULL && super_klass->should_be_initialized()) {
1166       super_klass->initialize(THREAD);
1167     }
1168     // If C implements any interface that declares a non-static, concrete method,
1169     // the initialization of C triggers initialization of its super interfaces.
1170     // Only need to recurse if has_nonstatic_concrete_methods which includes declaring and
1171     // having a superinterface that declares, non-static, concrete methods
1172     if (!HAS_PENDING_EXCEPTION && has_nonstatic_concrete_methods()) {
1173       initialize_super_interfaces(THREAD);
1174     }
1175 
1176     // If any exceptions, complete abruptly, throwing the same exception as above.
1177     if (HAS_PENDING_EXCEPTION) {
1178       Handle e(THREAD, PENDING_EXCEPTION);
1179       CLEAR_PENDING_EXCEPTION;
1180       {
1181         EXCEPTION_MARK;
1182         // Locks object, set state, and notify all waiting threads
1183         set_initialization_state_and_notify(initialization_error, THREAD);
1184         CLEAR_PENDING_EXCEPTION;
1185       }
1186       DTRACE_CLASSINIT_PROBE_WAIT(super__failed, -1, wait);
1187       THROW_OOP(e());
1188     }
1189   }
1190 
1191   // Step 8
1192   // Initialize classes of flattenable fields
1193   {
1194     for (AllFieldStream fs(this); !fs.done(); fs.next()) {
1195       if (fs.is_flattenable()) {
1196         Klass* klass = this->get_value_field_klass_or_null(fs.index());
1197         if (klass == NULL) {
1198           assert(fs.access_flags().is_static() && fs.access_flags().is_flattenable(),
1199               "Otherwise should have been pre-loaded");
1200           klass = SystemDictionary::resolve_or_fail(field_signature(fs.index())->fundamental_name(THREAD),
1201               Handle(THREAD, class_loader()),
1202               Handle(THREAD, protection_domain()),
1203               true, CHECK);
1204           if (klass == NULL) {
1205             THROW(vmSymbols::java_lang_NoClassDefFoundError());
1206           }
1207           if (!klass->is_value()) {
1208             THROW(vmSymbols::java_lang_IncompatibleClassChangeError());
1209           }
1210           this->set_value_field_klass(fs.index(), klass);
1211         }
1212         InstanceKlass::cast(klass)->initialize(CHECK);
1213         if (fs.access_flags().is_static()) {
1214           if (java_mirror()->obj_field(fs.offset()) == NULL) {
1215             java_mirror()->obj_field_put(fs.offset(), ValueKlass::cast(klass)->default_value());
1216           }
1217         }
1218       }
1219     }
1220   }
1221 
1222 
1223   // Look for aot compiled methods for this klass, including class initializer.
1224   AOTLoader::load_for_klass(this, THREAD);
1225 
1226   // Step 9
1227   {
1228     DTRACE_CLASSINIT_PROBE_WAIT(clinit, -1, wait);
1229     // Timer includes any side effects of class initialization (resolution,
1230     // etc), but not recursive entry into call_class_initializer().
1231     PerfClassTraceTime timer(ClassLoader::perf_class_init_time(),
1232                              ClassLoader::perf_class_init_selftime(),
1233                              ClassLoader::perf_classes_inited(),
1234                              jt->get_thread_stat()->perf_recursion_counts_addr(),
1235                              jt->get_thread_stat()->perf_timers_addr(),
1236                              PerfClassTraceTime::CLASS_CLINIT);
1237     call_class_initializer(THREAD);
1238   }
1239 
1240   // Step 10
1241   if (!HAS_PENDING_EXCEPTION) {
1242     set_initialization_state_and_notify(fully_initialized, CHECK);
1243     {
1244       debug_only(vtable().verify(tty, true);)
1245     }
1246   }
1247   else {
1248     // Step 11 and 12
1249     Handle e(THREAD, PENDING_EXCEPTION);
1250     CLEAR_PENDING_EXCEPTION;
1251     // JVMTI has already reported the pending exception
1252     // JVMTI internal flag reset is needed in order to report ExceptionInInitializerError
1253     JvmtiExport::clear_detected_exception(jt);
1254     {
1255       EXCEPTION_MARK;
1256       set_initialization_state_and_notify(initialization_error, THREAD);
1257       CLEAR_PENDING_EXCEPTION;   // ignore any exception thrown, class initialization error is thrown below
1258       // JVMTI has already reported the pending exception
1259       // JVMTI internal flag reset is needed in order to report ExceptionInInitializerError
1260       JvmtiExport::clear_detected_exception(jt);
1261     }
1262     DTRACE_CLASSINIT_PROBE_WAIT(error, -1, wait);
1263     if (e->is_a(SystemDictionary::Error_klass())) {
1264       THROW_OOP(e());
1265     } else {
1266       JavaCallArguments args(e);
1267       THROW_ARG(vmSymbols::java_lang_ExceptionInInitializerError(),
1268                 vmSymbols::throwable_void_signature(),
1269                 &args);
1270     }
1271   }
1272   DTRACE_CLASSINIT_PROBE_WAIT(end, -1, wait);
1273 }

1328 // The embedded _implementor field can only record one implementor.
1329 // When there are more than one implementors, the _implementor field
1330 // is set to the interface Klass* itself. Following are the possible
1331 // values for the _implementor field:
1332 //   NULL                  - no implementor
1333 //   implementor Klass*    - one implementor
1334 //   self                  - more than one implementor
1335 //
1336 // The _implementor field only exists for interfaces.
1337 void InstanceKlass::add_implementor(Klass* k) {
1338   if (Universe::is_fully_initialized()) {
1339     assert_lock_strong(Compile_lock);
1340   }
1341   assert(is_interface(), "not interface");
1342   // Filter out my subinterfaces.
1343   // (Note: Interfaces are never on the subklass list.)
1344   if (InstanceKlass::cast(k)->is_interface()) return;
1345 
1346   // Filter out subclasses whose supers already implement me.
1347   // (Note: CHA must walk subclasses of direct implementors
1348   // in order to locate indirect implementors.)
1349   Klass* sk = k->super();
1350   if (sk != NULL && InstanceKlass::cast(sk)->implements_interface(this))
1351     // We only need to check one immediate superclass, since the
1352     // implements_interface query looks at transitive_interfaces.
1353     // Any supers of the super have the same (or fewer) transitive_interfaces.
1354     return;
1355 
1356   Klass* ik = implementor();
1357   if (ik == NULL) {
1358     set_implementor(k);
1359   } else if (ik != this && ik != k) {
1360     // There is already an implementor. Use itself as an indicator of
1361     // more than one implementors.
1362     set_implementor(this);
1363   }
1364 
1365   // The implementor also implements the transitive_interfaces
1366   for (int index = 0; index < local_interfaces()->length(); index++) {
1367     InstanceKlass::cast(local_interfaces()->at(index))->add_implementor(k);
1368   }
1369 }

1397   // The secondaries are the implemented interfaces.

1420   }

1618 Klass* InstanceKlass::find_field(Symbol* name, Symbol* sig, fieldDescriptor* fd) const {
1619   // search order according to newest JVM spec (5.4.3.2, p.167).
1620   // 1) search for field in current klass
1621   if (find_local_field(name, sig, fd)) {
1622     return const_cast<InstanceKlass*>(this);
1623   }
1624   // 2) search for field recursively in direct superinterfaces
1625   { Klass* intf = find_interface_field(name, sig, fd);
1626     if (intf != NULL) return intf;
1627   }
1628   // 3) apply field lookup recursively if superclass exists
1629   { Klass* supr = super();
1630     if (supr != NULL) return InstanceKlass::cast(supr)->find_field(name, sig, fd);
1631   }
1632   // 4) otherwise field lookup fails
1633   return NULL;
1634 }

1734   fieldDescriptor fd;
1735   int length = java_fields_count();
1736   // In DebugInfo nonstatic fields are sorted by offset.
1737   int* fields_sorted = NEW_C_HEAP_ARRAY(int, 2*(length+1), mtClass);

1739   for (int i = 0; i < length; i += 1) {
1740     fd.reinitialize(this, i);
1741     if (!fd.is_static()) {
1742       fields_sorted[j + 0] = fd.offset();
1743       fields_sorted[j + 1] = i;
1744       j += 2;
1745     }
1746   }
1747   if (j > 0) {
1748     length = j;
1749     // _sort_Fn is defined in growableArray.hpp.
1750     qsort(fields_sorted, length/2, 2*sizeof(int), (_sort_Fn)compare_fields_by_offset);
1751     for (int i = 0; i < length; i += 2) {
1752       fd.reinitialize(this, fields_sorted[i + 1]);
1753       assert(!fd.is_static() && fd.offset() == fields_sorted[i], "only nonstatic fields");
1754       cl->do_field(&fd);
1755     }
1756   }
1757   FREE_C_HEAP_ARRAY(int, fields_sorted);

1803 inline int InstanceKlass::quick_search(const Array<Method*>* methods, const Symbol* name) {
1804   if (_disable_method_binary_search) {
1805     assert(DynamicDumpSharedSpaces, "must be");
1806     // At the final stage of dynamic dumping, the methods array may not be sorted
1807     // by ascending addresses of their names, so we can't use binary search anymore.
1808     // However, methods with the same name are still laid out consecutively inside the
1809     // methods array, so let's look for the first one that matches.
1810     return linear_search(methods, name);
1811   }
1812 
1813   int len = methods->length();
1814   int l = 0;
1815   int h = len - 1;
1816 
1817   // methods are sorted by ascending addresses of their names, so do binary search
1818   while (l <= h) {
1819     int mid = (l + h) >> 1;
1820     Method* m = methods->at(mid);
1821     assert(m->is_method(), "must be method");
1822     int res = m->name()->fast_compare(name);
1823     if (res == 0) {
1824       return mid;
1825     } else if (res < 0) {
1826       l = mid + 1;
1827     } else {
1828       h = mid - 1;
1829     }
1830   }
1831   return -1;
1832 }

1936 // true if method matches signature and conforms to skipping_X conditions.
1937 static bool method_matches(const Method* m,
1938                            const Symbol* signature,
1939                            bool skipping_overpass,
1940                            bool skipping_static,
1941                            bool skipping_private) {
1942   return ((m->signature() == signature) &&

1948 // Used directly for default_methods to find the index into the
1949 // default_vtable_indices, and indirectly by find_method
1950 // find_method_index looks in the local methods array to return the index
1951 // of the matching name/signature. If, overpass methods are being ignored,
1952 // the search continues to find a potential non-overpass match.  This capability
1953 // is important during method resolution to prefer a static method, for example,
1954 // over an overpass method.
1955 // There is the possibility in any _method's array to have the same name/signature
1956 // for a static method, an overpass method and a local instance method
1957 // To correctly catch a given method, the search criteria may need
1958 // to explicitly skip the other two. For local instance methods, it
1959 // is often necessary to skip private methods
1960 int InstanceKlass::find_method_index(const Array<Method*>* methods,
1961                                      const Symbol* name,
1962                                      const Symbol* signature,
1963                                      OverpassLookupMode overpass_mode,
1964                                      StaticLookupMode static_mode,
1965                                      PrivateLookupMode private_mode) {
1966   const bool skipping_overpass = (overpass_mode == skip_overpass);
1967   const bool skipping_static = (static_mode == skip_static);
1968   const bool skipping_private = (private_mode == skip_private);
1969   const int hit = quick_search(methods, name);
1970   if (hit != -1) {
1971     const Method* const m = methods->at(hit);
1972 
1973     // Do linear search to find matching signature.  First, quick check
1974     // for common case, ignoring overpasses if requested.
1975     if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1976       return hit;
1977     }
1978 
1979     // search downwards through overloaded methods
1980     int i;
1981     for (i = hit - 1; i >= 0; --i) {
1982         const Method* const m = methods->at(i);
1983         assert(m->is_method(), "must be method");
1984         if (m->name() != name) {
1985           break;
1986         }
1987         if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1988           return i;
1989         }
1990     }
1991     // search upwards
1992     for (i = hit + 1; i < methods->length(); ++i) {
1993         const Method* const m = methods->at(i);
1994         assert(m->is_method(), "must be method");
1995         if (m->name() != name) {
1996           break;
1997         }
1998         if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1999           return i;
2000         }
2001     }
2002     // not found
2003 #ifdef ASSERT
2004     const int index = (skipping_overpass || skipping_static || skipping_private) ? -1 :
2005       linear_search(methods, name, signature);
2006     assert(-1 == index, "binary search should have found entry %d", index);
2007 #endif
2008   }
2009   return -1;
2010 }

2159 // Lookup or create a jmethodID.
2160 // This code is called by the VMThread and JavaThreads so the
2161 // locking has to be done very carefully to avoid deadlocks
2162 // and/or other cache consistency problems.
2163 //
2164 jmethodID InstanceKlass::get_jmethod_id(const methodHandle& method_h) {
2165   size_t idnum = (size_t)method_h->method_idnum();
2166   jmethodID* jmeths = methods_jmethod_ids_acquire();
2167   size_t length = 0;
2168   jmethodID id = NULL;
2169 
2170   // We use a double-check locking idiom here because this cache is
2171   // performance sensitive. In the normal system, this cache only
2172   // transitions from NULL to non-NULL which is safe because we use
2173   // release_set_methods_jmethod_ids() to advertise the new cache.
2174   // A partially constructed cache should never be seen by a racing
2175   // thread. We also use release_store() to save a new jmethodID
2176   // in the cache so a partially constructed jmethodID should never be
2177   // seen either. Cache reads of existing jmethodIDs proceed without a
2178   // lock, but cache writes of a new jmethodID requires uniqueness and
2179   // creation of the cache itself requires no leaks so a lock is
2180   // generally acquired in those two cases.
2181   //
2182   // If the RedefineClasses() API has been used, then this cache can
2183   // grow and we'll have transitions from non-NULL to bigger non-NULL.
2184   // Cache creation requires no leaks and we require safety between all
2185   // cache accesses and freeing of the old cache so a lock is generally
2186   // acquired when the RedefineClasses() API has been used.
2187 
2188   if (jmeths != NULL) {
2189     // the cache already exists
2190     if (!idnum_can_increment()) {
2191       // the cache can't grow so we can just get the current values
2192       get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2193     } else {
2194       // cache can grow so we have to be more careful
2195       if (Threads::number_of_threads() == 0 ||
2196           SafepointSynchronize::is_at_safepoint()) {
2197         // we're single threaded or at a safepoint - no locking needed
2198         get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2199       } else {
2200         MutexLocker ml(JmethodIdCreation_lock, Mutex::_no_safepoint_check_flag);
2201         get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2202       }
2203     }
2204   }
2205   // implied else:
2206   // we need to allocate a cache so default length and id values are good
2207 
2208   if (jmeths == NULL ||   // no cache yet
2209       length <= idnum ||  // cache is too short
2210       id == NULL) {       // cache doesn't contain entry
2211 
2212     // This function can be called by the VMThread so we have to do all
2213     // things that might block on a safepoint before grabbing the lock.
2214     // Otherwise, we can deadlock with the VMThread or have a cache
2215     // consistency issue. These vars keep track of what we might have
2216     // to free after the lock is dropped.
2217     jmethodID  to_dealloc_id     = NULL;
2218     jmethodID* to_dealloc_jmeths = NULL;
2219 
2220     // may not allocate new_jmeths or use it if we allocate it
2221     jmethodID* new_jmeths = NULL;
2222     if (length <= idnum) {
2223       // allocate a new cache that might be used
2224       size_t size = MAX2(idnum+1, (size_t)idnum_allocated_count());
2225       new_jmeths = NEW_C_HEAP_ARRAY(jmethodID, size+1, mtClass);
2226       memset(new_jmeths, 0, (size+1)*sizeof(jmethodID));
2227       // cache size is stored in element[0], other elements offset by one
2228       new_jmeths[0] = (jmethodID)size;
2229     }
2230 
2231     // allocate a new jmethodID that might be used
2232     jmethodID new_id = NULL;
2233     if (method_h->is_old() && !method_h->is_obsolete()) {
2234       // The method passed in is old (but not obsolete), we need to use the current version
2235       Method* current_method = method_with_idnum((int)idnum);
2236       assert(current_method != NULL, "old and but not obsolete, so should exist");
2237       new_id = Method::make_jmethod_id(class_loader_data(), current_method);
2238     } else {
2239       // It is the current version of the method or an obsolete method,
2240       // use the version passed in
2241       new_id = Method::make_jmethod_id(class_loader_data(), method_h());
2242     }
2243 
2244     if (Threads::number_of_threads() == 0 ||
2245         SafepointSynchronize::is_at_safepoint()) {
2246       // we're single threaded or at a safepoint - no locking needed
2247       id = get_jmethod_id_fetch_or_update(idnum, new_id, new_jmeths,
2248                                           &to_dealloc_id, &to_dealloc_jmeths);
2249     } else {
2250       MutexLocker ml(JmethodIdCreation_lock, Mutex::_no_safepoint_check_flag);
2251       id = get_jmethod_id_fetch_or_update(idnum, new_id, new_jmeths,
2252                                           &to_dealloc_id, &to_dealloc_jmeths);
2253     }
2254 
2255     // The lock has been dropped so we can free resources.
2256     // Free up either the old cache or the new cache if we allocated one.
2257     if (to_dealloc_jmeths != NULL) {
2258       FreeHeap(to_dealloc_jmeths);
2259     }
2260     // free up the new ID since it wasn't needed
2261     if (to_dealloc_id != NULL) {
2262       Method::destroy_jmethod_id(class_loader_data(), to_dealloc_id);
2263     }
2264   }
2265   return id;
2266 }

2307   size_t     length = 0;

2309   if (jmeths == NULL ||                         // no cache yet
2310       (length = (size_t)jmeths[0]) <= idnum) {  // cache is too short
2311     if (jmeths != NULL) {
2312       // copy any existing entries from the old cache
2313       for (size_t index = 0; index < length; index++) {
2314         new_jmeths[index+1] = jmeths[index+1];
2315       }
2316       *to_dealloc_jmeths_p = jmeths;  // save old cache for later delete
2317     }
2318     release_set_methods_jmethod_ids(jmeths = new_jmeths);
2319   } else {
2320     // fetch jmethodID (if any) from the existing cache
2321     id = jmeths[idnum+1];
2322     *to_dealloc_jmeths_p = new_jmeths;  // save new cache for later delete
2323   }
2324   if (id == NULL) {
2325     // No matching jmethodID in the existing cache or we have a new
2326     // cache or we just grew the cache. This cache write is done here
2327     // by the first thread to win the foot race because a jmethodID
2328     // needs to be unique once it is generally available.
2329     id = new_id;
2330 
2331     // The jmethodID cache can be read while unlocked so we have to
2332     // make sure the new jmethodID is complete before installing it
2333     // in the cache.
2334     Atomic::release_store(&jmeths[idnum+1], id);
2335   } else {
2336     *to_dealloc_id_p = new_id; // save new id for later delete
2337   }
2338   return id;
2339 }

3777   // Avoid redundant verifies, this really should be in product.
3778   if (_verify_count == Universe::verify_count()) return;
3779   _verify_count = Universe::verify_count();

3782   // Verify Klass

3790   if (is_linked()) {
3791     // $$$ This used to be done only for m/s collections.  Doing it

3793     vtable().verify(st);

3797   if (subklass() != NULL) {

3802   Klass* super = this->super();

3980 // Purge previous versions before adding new previous versions of the class and
3981 // during class unloading.
3982 void InstanceKlass::purge_previous_version_list() {
3983   assert(SafepointSynchronize::is_at_safepoint(), "only called at safepoint");
3984   assert(has_been_redefined(), "Should only be called for main class");
3985 
3986   // Quick exit.
3987   if (previous_versions() == NULL) {
3988     return;
3989   }
3990 
3991   // This klass has previous versions so see what we can cleanup
3992   // while it is safe to do so.
3993 
3994   int deleted_count = 0;    // leave debugging breadcrumbs
3995   int live_count = 0;
3996   ClassLoaderData* loader_data = class_loader_data();
3997   assert(loader_data != NULL, "should never be null");
3998 
3999   ResourceMark rm;
4000   log_trace(redefine, class, iklass, purge)("%s: previous versions", external_name());
4001 
4002   // previous versions are linked together through the InstanceKlass
4003   InstanceKlass* pv_node = previous_versions();
4004   InstanceKlass* last = this;
4005   int version = 0;
4006 
4007   // check the previous versions list
4008   for (; pv_node != NULL; ) {
4009 
4010     ConstantPool* pvcp = pv_node->constants();
4011     assert(pvcp != NULL, "cp ref was unexpectedly cleared");
4012 
4013     if (!pvcp->on_stack()) {
4014       // If the constant pool isn't on stack, none of the methods
4015       // are executing.  Unlink this previous_version.
4016       // The previous version InstanceKlass is on the ClassLoaderData deallocate list
4017       // so will be deallocated during the next phase of class unloading.
4018       log_trace(redefine, class, iklass, purge)
4019         ("previous version " INTPTR_FORMAT " is dead.", p2i(pv_node));
4020       // For debugging purposes.
4021       pv_node->set_is_scratch_class();
4022       // Unlink from previous version list.
4023       assert(pv_node->class_loader_data() == loader_data, "wrong loader_data");
4024       InstanceKlass* next = pv_node->previous_versions();
4025       pv_node->link_previous_versions(NULL);   // point next to NULL
4026       last->link_previous_versions(next);
4027       // Add to the deallocate list after unlinking
4028       loader_data->add_to_deallocate_list(pv_node);
4029       pv_node = next;
4030       deleted_count++;
4031       version++;
4032       continue;
4033     } else {
4034       log_trace(redefine, class, iklass, purge)("previous version " INTPTR_FORMAT " is alive", p2i(pv_node));
4035       assert(pvcp->pool_holder() != NULL, "Constant pool with no holder");
4036       guarantee (!loader_data->is_unloading(), "unloaded classes can't be on the stack");
4037       live_count++;
4038       // found a previous version for next time we do class unloading
4039       _has_previous_versions = true;
4040     }
4041 
4042     // At least one method is live in this previous version.
4043     // Reset dead EMCP methods not to get breakpoints.
4044     // All methods are deallocated when all of the methods for this class are no
4045     // longer running.
4046     Array<Method*>* method_refs = pv_node->methods();
4047     if (method_refs != NULL) {
4048       log_trace(redefine, class, iklass, purge)("previous methods length=%d", method_refs->length());
4049       for (int j = 0; j < method_refs->length(); j++) {
4050         Method* method = method_refs->at(j);
4051 
4052         if (!method->on_stack()) {
4053           // no breakpoints for non-running methods
4054           if (method->is_running_emcp()) {
4055             method->set_running_emcp(false);
4056           }
4057         } else {
4058           assert (method->is_obsolete() || method->is_running_emcp(),
4059                   "emcp method cannot run after emcp bit is cleared");
4060           log_trace(redefine, class, iklass, purge)
4061             ("purge: %s(%s): prev method @%d in version @%d is alive",
4062              method->name()->as_C_string(), method->signature()->as_C_string(), j, version);
4063         }
4064       }
4065     }
4066     // next previous version
4067     last = pv_node;
4068     pv_node = pv_node->previous_versions();
4069     version++;
4070   }
4071   log_trace(redefine, class, iklass, purge)
4072     ("previous version stats: live=%d, deleted=%d", live_count, deleted_count);
4073 }

4075 void InstanceKlass::mark_newly_obsolete_methods(Array<Method*>* old_methods,
4076                                                 int emcp_method_count) {
4077   int obsolete_method_count = old_methods->length() - emcp_method_count;
4078 
4079   if (emcp_method_count != 0 && obsolete_method_count != 0 &&
4080       _previous_versions != NULL) {
4081     // We have a mix of obsolete and EMCP methods so we have to
4082     // clear out any matching EMCP method entries the hard way.
4083     int local_count = 0;
4084     for (int i = 0; i < old_methods->length(); i++) {
4085       Method* old_method = old_methods->at(i);
4086       if (old_method->is_obsolete()) {
4087         // only obsolete methods are interesting
4088         Symbol* m_name = old_method->name();
4089         Symbol* m_signature = old_method->signature();
4090 
4091         // previous versions are linked together through the InstanceKlass
4092         int j = 0;
4093         for (InstanceKlass* prev_version = _previous_versions;
4094              prev_version != NULL;
4095              prev_version = prev_version->previous_versions(), j++) {
4096 
4097           Array<Method*>* method_refs = prev_version->methods();
4098           for (int k = 0; k < method_refs->length(); k++) {
4099             Method* method = method_refs->at(k);
4100 
4101             if (!method->is_obsolete() &&
4102                 method->name() == m_name &&
4103                 method->signature() == m_signature) {
4104               // The current RedefineClasses() call has made all EMCP
4105               // versions of this method obsolete so mark it as obsolete
4106               log_trace(redefine, class, iklass, add)
4107                 ("%s(%s): flush obsolete method @%d in version @%d",
4108                  m_name->as_C_string(), m_signature->as_C_string(), k, j);
4109 
4110               method->set_is_obsolete();
4111               break;
4112             }
4113           }
4114 
4115           // The previous loop may not find a matching EMCP method, but
4116           // that doesn't mean that we can optimize and not go any
4117           // further back in the PreviousVersion generations. The EMCP
4118           // method for this generation could have already been made obsolete,
4119           // but there still may be an older EMCP method that has not
4120           // been made obsolete.
4121         }
4122 
4123         if (++local_count >= obsolete_method_count) {
4124           // no more obsolete methods so bail out now
4125           break;
4126         }
4127       }
4128     }
4129   }
4130 }

4132 // Save the scratch_class as the previous version if any of the methods are running.
4133 // The previous_versions are used to set breakpoints in EMCP methods and they are
4134 // also used to clean MethodData links to redefined methods that are no longer running.
4135 void InstanceKlass::add_previous_version(InstanceKlass* scratch_class,
4136                                          int emcp_method_count) {
4137   assert(Thread::current()->is_VM_thread(),
4138          "only VMThread can add previous versions");
4139 
4140   ResourceMark rm;
4141   log_trace(redefine, class, iklass, add)
4142     ("adding previous version ref for %s, EMCP_cnt=%d", scratch_class->external_name(), emcp_method_count);
4143 
4144   // Clean out old previous versions for this class
4145   purge_previous_version_list();
4146 
4147   // Mark newly obsolete methods in remaining previous versions.  An EMCP method from
4148   // a previous redefinition may be made obsolete by this redefinition.
4149   Array<Method*>* old_methods = scratch_class->methods();
4150   mark_newly_obsolete_methods(old_methods, emcp_method_count);
4151 
4152   // If the constant pool for this previous version of the class
4153   // is not marked as being on the stack, then none of the methods
4154   // in this previous version of the class are on the stack so
4155   // we don't need to add this as a previous version.
4156   ConstantPool* cp_ref = scratch_class->constants();
4157   if (!cp_ref->on_stack()) {
4158     log_trace(redefine, class, iklass, add)("scratch class not added; no methods are running");
4159     // For debugging purposes.
4160     scratch_class->set_is_scratch_class();
4161     scratch_class->class_loader_data()->add_to_deallocate_list(scratch_class);
4162     return;
4163   }
4164 
4165   if (emcp_method_count != 0) {
4166     // At least one method is still running, check for EMCP methods
4167     for (int i = 0; i < old_methods->length(); i++) {
4168       Method* old_method = old_methods->at(i);
4169       if (!old_method->is_obsolete() && old_method->on_stack()) {
4170         // if EMCP method (not obsolete) is on the stack, mark as EMCP so that
4171         // we can add breakpoints for it.
4172 
4173         // We set the method->on_stack bit during safepoints for class redefinition
4174         // and use this bit to set the is_running_emcp bit.
4175         // After the safepoint, the on_stack bit is cleared and the running emcp
4176         // method may exit.   If so, we would set a breakpoint in a method that
4177         // is never reached, but this won't be noticeable to the programmer.
4178         old_method->set_running_emcp(true);
4179         log_trace(redefine, class, iklass, add)
4180           ("EMCP method %s is on_stack " INTPTR_FORMAT, old_method->name_and_sig_as_C_string(), p2i(old_method));
4181       } else if (!old_method->is_obsolete()) {
4182         log_trace(redefine, class, iklass, add)
4183           ("EMCP method %s is NOT on_stack " INTPTR_FORMAT, old_method->name_and_sig_as_C_string(), p2i(old_method));
4184       }
4185     }
4186   }
4187 
4188   // Add previous version if any methods are still running.
4189   // Set has_previous_version flag for processing during class unloading.
4190   _has_previous_versions = true;
4191   log_trace(redefine, class, iklass, add) ("scratch class added; one of its methods is on_stack.");
4192   assert(scratch_class->previous_versions() == NULL, "shouldn't have a previous version");
4193   scratch_class->link_previous_versions(previous_versions());
4194   link_previous_versions(scratch_class);
4195 } // end add_previous_version()
