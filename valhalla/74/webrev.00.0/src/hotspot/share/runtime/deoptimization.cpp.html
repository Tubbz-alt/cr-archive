  86 Deoptimization::UnrollBlock::UnrollBlock(int  size_of_deoptimized_frame,
  87                                          int  caller_adjustment,
  88                                          int  caller_actual_parameters,
  89                                          int  number_of_frames,
  90                                          intptr_t* frame_sizes,
  91                                          address* frame_pcs,
  92                                          BasicType return_type,
  93                                          int exec_mode) {
  94   _size_of_deoptimized_frame = size_of_deoptimized_frame;
  95   _caller_adjustment         = caller_adjustment;
  96   _caller_actual_parameters  = caller_actual_parameters;
  97   _number_of_frames          = number_of_frames;
  98   _frame_sizes               = frame_sizes;
  99   _frame_pcs                 = frame_pcs;
 100   _register_block            = NEW_C_HEAP_ARRAY(intptr_t, RegisterMap::reg_count * 2, mtCompiler);
 101   _return_type               = return_type;
 102   _initial_info              = 0;
 103   // PD (x86 only)
 104   _counter_temp              = 0;
 105   _unpack_kind               = exec_mode;
 106   _sender_sp_temp            = 0;
 107 
 108   _total_frame_sizes         = size_of_frames();
 109   assert(exec_mode >= 0 && exec_mode < Unpack_LIMIT, "Unexpected exec_mode");
 110 }

 127 int Deoptimization::UnrollBlock::size_of_frames() const {
 128   // Acount first for the adjustment of the initial frame
 129   int result = _caller_adjustment;
 130   for (int index = 0; index < number_of_frames(); index++) {
 131     result += frame_sizes()[index];
 132   }
 133   return result;
 134 }

 149 // In order to make fetch_unroll_info work properly with escape
 150 // analysis, The method was changed from JRT_LEAF to JRT_BLOCK_ENTRY and
 151 // ResetNoHandleMark and HandleMark were removed from it. The actual reallocation
 152 // of previously eliminated objects occurs in realloc_objects, which is
 153 // called from the method fetch_unroll_info_helper below.
 154 JRT_BLOCK_ENTRY(Deoptimization::UnrollBlock*, Deoptimization::fetch_unroll_info(JavaThread* thread, int exec_mode))
 155   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 156   // but makes the entry a little slower. There is however a little dance we have to
 157   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 158 
 159   // fetch_unroll_info() is called at the beginning of the deoptimization
 160   // handler. Note this fact before we start generating temporary frames
 161   // that can confuse an asynchronous stack walker. This counter is
 162   // decremented at the end of unpack_frames().
 163   if (TraceDeoptimization) {
 164     tty->print_cr("Deoptimizing thread " INTPTR_FORMAT, p2i(thread));
 165   }
 166   thread->inc_in_deopt_handler();
 167 
 168   return fetch_unroll_info_helper(thread, exec_mode);
 169 JRT_END

 171 #if COMPILER2_OR_JVMCI
 172 static bool eliminate_allocations(JavaThread* thread, int exec_mode, CompiledMethod* compiled_method,
 173                                   frame& deoptee, RegisterMap& map, GrowableArray<compiledVFrame*>* chunk) {
 174   bool realloc_failures = false;
 175   assert (chunk->at(0)->scope() != NULL,"expect only compiled java frames");
 176 
 177   GrowableArray<ScopeValue*>* objects = chunk->at(0)->scope()->objects();
 178 
 179   // The flag return_oop() indicates call sites which return oop
 180   // in compiled code. Such sites include java method calls,
 181   // runtime calls (for example, used to allocate new objects/arrays
 182   // on slow code path) and any other calls generated in compiled code.
 183   // It is not guaranteed that we can get such information here only
 184   // by analyzing bytecode in deoptimized frames. This is why this flag
 185   // is set during method compilation (see Compile::Process_OopMap_Node()).
 186   // If the previous frame was popped or if we are dispatching an exception,
 187   // we don't have an oop result.
 188   ScopeDesc* scope = chunk->at(0)->scope();
 189   bool save_oop_result = scope->return_oop() && !thread->popframe_forcing_deopt_reexecution() && (exec_mode == Deoptimization::Unpack_deopt);
 190   // In case of the return of multiple values, we must take care
 191   // of all oop return values.
 192   GrowableArray<Handle> return_oops;
 193   ValueKlass* vk = NULL;
 194   if (save_oop_result && scope->return_vt()) {
 195     vk = ValueKlass::returned_value_klass(map);
 196     if (vk != NULL) {
 197       vk->save_oop_fields(map, return_oops);
 198       save_oop_result = false;
 199     }
 200   }
 201   if (save_oop_result) {
 202     // Reallocation may trigger GC. If deoptimization happened on return from
 203     // call which returns oop we need to save it since it is not in oopmap.
 204     oop result = deoptee.saved_oop_result(&map);
 205     assert(oopDesc::is_oop_or_null(result), "must be oop");
 206     return_oops.push(Handle(thread, result));
 207     assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
 208     if (TraceDeoptimization) {
 209       ttyLocker ttyl;
 210       tty->print_cr("SAVED OOP RESULT " INTPTR_FORMAT " in thread " INTPTR_FORMAT, p2i(result), p2i(thread));
 211     }
 212   }
 213   if (objects != NULL || vk != NULL) {
 214     bool skip_internal = (compiled_method != NULL) && !compiled_method->is_compiled_by_jvmci();
 215     JRT_BLOCK
 216       if (vk != NULL) {
 217         realloc_failures = Deoptimization::realloc_value_type_result(vk, map, return_oops, THREAD);
 218       }
 219       if (objects != NULL) {
 220         realloc_failures = realloc_failures || Deoptimization::realloc_objects(thread, &deoptee, &map, objects, THREAD);
 221         Deoptimization::reassign_fields(&deoptee, &map, objects, realloc_failures, skip_internal, THREAD);
 222       }
 223     JRT_END
 224 #ifndef PRODUCT
 225     if (TraceDeoptimization) {
 226       ttyLocker ttyl;
 227       tty->print_cr("REALLOC OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 228       if (objects != NULL) {
 229         Deoptimization::print_objects(objects, realloc_failures);
 230       } else {
 231         Handle obj = realloc_failures ? Handle() : return_oops.first();
 232         Deoptimization::print_object(vk, obj, realloc_failures);
 233       }
 234     }
 235 #endif
 236   }
 237   if (save_oop_result || vk != NULL) {
 238     // Restore result.
 239     assert(return_oops.length() == 1, "no value type");
 240     deoptee.set_saved_oop_result(&map, return_oops.pop()());
 241   }
 242   return realloc_failures;
 243 }

 245 static void eliminate_locks(JavaThread* thread, GrowableArray<compiledVFrame*>* chunk, bool realloc_failures) {
 246 #ifndef PRODUCT
 247   bool first = true;
 248 #endif
 249   for (int i = 0; i < chunk->length(); i++) {
 250     compiledVFrame* cvf = chunk->at(i);
 251     assert (cvf->scope() != NULL,"expect only compiled java frames");
 252     GrowableArray<MonitorInfo*>* monitors = cvf->monitors();
 253     if (monitors->is_nonempty()) {
 254       Deoptimization::relock_objects(monitors, thread, realloc_failures);
 255 #ifndef PRODUCT
 256       if (PrintDeoptimizationDetails) {
 257         ttyLocker ttyl;
 258         for (int j = 0; j < monitors->length(); j++) {
 259           MonitorInfo* mi = monitors->at(j);
 260           if (mi->eliminated()) {
 261             if (first) {
 262               first = false;
 263               tty->print_cr("RELOCK OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 264             }
 265             if (mi->owner_is_scalar_replaced()) {
 266               Klass* k = java_lang_Class::as_Klass(mi->owner_klass());
 267               tty->print_cr("     failed reallocation for klass %s", k->external_name());
 268             } else {
 269               tty->print_cr("     object <" INTPTR_FORMAT "> locked", p2i(mi->owner()));
 270             }
 271           }
 272         }
 273       }
 274 #endif // !PRODUCT
 275     }
 276   }
 277 }

 280 // This is factored, since it is both called from a JRT_LEAF (deoptimization) and a JRT_ENTRY (uncommon_trap)
 281 Deoptimization::UnrollBlock* Deoptimization::fetch_unroll_info_helper(JavaThread* thread, int exec_mode) {
 282 
 283   // Note: there is a safepoint safety issue here. No matter whether we enter
 284   // via vanilla deopt or uncommon trap we MUST NOT stop at a safepoint once
 285   // the vframeArray is created.
 286   //
 287 
 288   // Allocate our special deoptimization ResourceMark
 289   DeoptResourceMark* dmark = new DeoptResourceMark(thread);
 290   assert(thread->deopt_mark() == NULL, "Pending deopt!");
 291   thread->set_deopt_mark(dmark);
 292 
 293   frame stub_frame = thread->last_frame(); // Makes stack walkable as side effect
 294   RegisterMap map(thread, true);
 295   RegisterMap dummy_map(thread, false);
 296   // Now get the deoptee with a valid map
 297   frame deoptee = stub_frame.sender(&map);
 298   // Set the deoptee nmethod
 299   assert(thread->deopt_compiled_method() == NULL, "Pending deopt!");
 300   CompiledMethod* cm = deoptee.cb()->as_compiled_method_or_null();
 301   thread->set_deopt_compiled_method(cm);
 302 
 303   if (VerifyStack) {
 304     thread->validate_frame_layout();
 305   }
 306 
 307   // Create a growable array of VFrames where each VFrame represents an inlined
 308   // Java frame.  This storage is allocated with the usual system arena.
 309   assert(deoptee.is_compiled_frame(), "Wrong frame type");
 310   GrowableArray<compiledVFrame*>* chunk = new GrowableArray<compiledVFrame*>(10);
 311   vframe* vf = vframe::new_vframe(&deoptee, &map, thread);
 312   while (!vf->is_top()) {
 313     assert(vf->is_compiled_frame(), "Wrong frame type");
 314     chunk->push(compiledVFrame::cast(vf));
 315     vf = vf->sender();
 316   }
 317   assert(vf->is_compiled_frame(), "Wrong frame type");
 318   chunk->push(compiledVFrame::cast(vf));
 319 
 320   bool realloc_failures = false;
 321 
 322 #if COMPILER2_OR_JVMCI
 323 #if INCLUDE_JVMCI
 324   bool jvmci_enabled = true;
 325 #else
 326   bool jvmci_enabled = false;
 327 #endif
 328 
 329   // Reallocate the non-escaping objects and restore their fields. Then
 330   // relock objects if synchronization on them was eliminated.
 331   if (jvmci_enabled COMPILER2_PRESENT( || (DoEscapeAnalysis && EliminateAllocations) )) {
 332     realloc_failures = eliminate_allocations(thread, exec_mode, cm, deoptee, map, chunk);
 333   }
 334 #endif // COMPILER2_OR_JVMCI
 335 
 336   // Revoke biases, done with in java state.
 337   // No safepoints allowed after this
 338   revoke_from_deopt_handler(thread, deoptee, &map);
 339 
 340   // Ensure that no safepoint is taken after pointers have been stored
 341   // in fields of rematerialized objects.  If a safepoint occurs from here on
 342   // out the java state residing in the vframeArray will be missed.
 343   // Locks may be rebaised in a safepoint.
 344   NoSafepointVerifier no_safepoint;
 345 
 346 #if COMPILER2_OR_JVMCI
 347   if (jvmci_enabled COMPILER2_PRESENT( || ((DoEscapeAnalysis || EliminateNestedLocks) && EliminateLocks) )) {
 348     eliminate_locks(thread, chunk, realloc_failures);
 349   }
 350 #endif // COMPILER2_OR_JVMCI
 351 
 352   ScopeDesc* trap_scope = chunk->at(0)->scope();
 353   Handle exceptionObject;
 354   if (trap_scope->rethrow_exception()) {
 355     if (PrintDeoptimizationDetails) {
 356       tty->print_cr("Exception to be rethrown in the interpreter for method %s::%s at bci %d", trap_scope->method()->method_holder()->name()->as_C_string(), trap_scope->method()->name()->as_C_string(), trap_scope->bci());
 357     }
 358     GrowableArray<ScopeValue*>* expressions = trap_scope->expressions();
 359     guarantee(expressions != NULL && expressions->length() > 0, "must have exception to throw");
 360     ScopeValue* topOfStack = expressions->top();
 361     exceptionObject = StackValue::create_stack_value(&deoptee, &map, topOfStack)->get_obj();
 362     guarantee(exceptionObject() != NULL, "exception oop can not be null");
 363   }
 364 
 365   vframeArray* array = create_vframeArray(thread, deoptee, &map, chunk, realloc_failures);
 366 #if COMPILER2_OR_JVMCI
 367   if (realloc_failures) {
 368     pop_frames_failed_reallocs(thread, array);
 369   }
 370 #endif
 371 
 372   assert(thread->vframe_array_head() == NULL, "Pending deopt!");
 373   thread->set_vframe_array_head(array);
 374 
 375   // Now that the vframeArray has been created if we have any deferred local writes
 376   // added by jvmti then we can free up that structure as the data is now in the
 377   // vframeArray
 378 
 379   if (thread->deferred_locals() != NULL) {
 380     GrowableArray<jvmtiDeferredLocalVariableSet*>* list = thread->deferred_locals();
 381     int i = 0;
 382     do {
 383       // Because of inlining we could have multiple vframes for a single frame
 384       // and several of the vframes could have deferred writes. Find them all.
 385       if (list->at(i)->id() == array->original().id()) {
 386         jvmtiDeferredLocalVariableSet* dlv = list->at(i);
 387         list->remove_at(i);
 388         // individual jvmtiDeferredLocalVariableSet are CHeapObj's
 389         delete dlv;
 390       } else {
 391         i++;
 392       }
 393     } while ( i < list->length() );
 394     if (list->length() == 0) {
 395       thread->set_deferred_locals(NULL);
 396       // free the list and elements back to C heap.
 397       delete list;
 398     }
 399 
 400   }
 401 
 402   // Compute the caller frame based on the sender sp of stub_frame and stored frame sizes info.
 403   CodeBlob* cb = stub_frame.cb();
 404   // Verify we have the right vframeArray
 405   assert(cb->frame_size() >= 0, "Unexpected frame size");
 406   intptr_t* unpack_sp = stub_frame.sp() + cb->frame_size();
 407 
 408   // If the deopt call site is a MethodHandle invoke call site we have
 409   // to adjust the unpack_sp.
 410   nmethod* deoptee_nm = deoptee.cb()->as_nmethod_or_null();
 411   if (deoptee_nm != NULL && deoptee_nm->is_method_handle_return(deoptee.pc()))
 412     unpack_sp = deoptee.unextended_sp();
 413 
 414 #ifdef ASSERT
 415   assert(cb->is_deoptimization_stub() ||
 416          cb->is_uncommon_trap_stub() ||
 417          strcmp("Stub<DeoptimizationStub.deoptimizationHandler>", cb->name()) == 0 ||
 418          strcmp("Stub<UncommonTrapStub.uncommonTrapHandler>", cb->name()) == 0,
 419          "unexpected code blob: %s", cb->name());
 420 #endif
 421 
 422   // This is a guarantee instead of an assert because if vframe doesn't match
 423   // we will unpack the wrong deoptimized frame and wind up in strange places
 424   // where it will be very difficult to figure out what went wrong. Better
 425   // to die an early death here than some very obscure death later when the
 426   // trail is cold.
 427   // Note: on ia64 this guarantee can be fooled by frames with no memory stack
 428   // in that it will fail to detect a problem when there is one. This needs
 429   // more work in tiger timeframe.
 430   guarantee(array->unextended_sp() == unpack_sp, "vframe_array_head must contain the vframeArray to unpack");
 431 
 432   int number_of_frames = array->frames();
 433 
 434   // Compute the vframes' sizes.  Note that frame_sizes[] entries are ordered from outermost to innermost
 435   // virtual activation, which is the reverse of the elements in the vframes array.
 436   intptr_t* frame_sizes = NEW_C_HEAP_ARRAY(intptr_t, number_of_frames, mtCompiler);
 437   // +1 because we always have an interpreter return address for the final slot.
 438   address* frame_pcs = NEW_C_HEAP_ARRAY(address, number_of_frames + 1, mtCompiler);
 439   int popframe_extra_args = 0;
 440   // Create an interpreter return address for the stub to use as its return
 441   // address so the skeletal frames are perfectly walkable
 442   frame_pcs[number_of_frames] = Interpreter::deopt_entry(vtos, 0);
 443 
 444   // PopFrame requires that the preserved incoming arguments from the recently-popped topmost
 445   // activation be put back on the expression stack of the caller for reexecution
 446   if (JvmtiExport::can_pop_frame() && thread->popframe_forcing_deopt_reexecution()) {
 447     popframe_extra_args = in_words(thread->popframe_preserved_args_size_in_words());
 448   }
 449 
 450   // Find the current pc for sender of the deoptee. Since the sender may have been deoptimized
 451   // itself since the deoptee vframeArray was created we must get a fresh value of the pc rather
 452   // than simply use array->sender.pc(). This requires us to walk the current set of frames
 453   //
 454   frame deopt_sender = stub_frame.sender(&dummy_map); // First is the deoptee frame
 455   deopt_sender = deopt_sender.sender(&dummy_map);     // Now deoptee caller
 456 
 457   // It's possible that the number of parameters at the call site is
 458   // different than number of arguments in the callee when method
 459   // handles are used.  If the caller is interpreted get the real
 460   // value so that the proper amount of space can be added to it's
 461   // frame.
 462   bool caller_was_method_handle = false;
 463   if (deopt_sender.is_interpreted_frame()) {
 464     methodHandle method(thread, deopt_sender.interpreter_frame_method());
 465     Bytecode_invoke cur = Bytecode_invoke_check(method, deopt_sender.interpreter_frame_bci());
 466     if (cur.is_invokedynamic() || cur.is_invokehandle()) {
 467       // Method handle invokes may involve fairly arbitrary chains of
 468       // calls so it's impossible to know how much actual space the
 469       // caller has for locals.
 470       caller_was_method_handle = true;
 471     }
 472   }
 473 
 474   //
 475   // frame_sizes/frame_pcs[0] oldest frame (int or c2i)
 476   // frame_sizes/frame_pcs[1] next oldest frame (int)
 477   // frame_sizes/frame_pcs[n] youngest frame (int)
 478   //
 479   // Now a pc in frame_pcs is actually the return address to the frame's caller (a frame
 480   // owns the space for the return address to it's caller).  Confusing ain't it.
 481   //
 482   // The vframe array can address vframes with indices running from
 483   // 0.._frames-1. Index  0 is the youngest frame and _frame - 1 is the oldest (root) frame.
 484   // When we create the skeletal frames we need the oldest frame to be in the zero slot
 485   // in the frame_sizes/frame_pcs so the assembly code can do a trivial walk.
 486   // so things look a little strange in this loop.
 487   //
 488   int callee_parameters = 0;
 489   int callee_locals = 0;
 490   for (int index = 0; index < array->frames(); index++ ) {
 491     // frame[number_of_frames - 1 ] = on_stack_size(youngest)
 492     // frame[number_of_frames - 2 ] = on_stack_size(sender(youngest))
 493     // frame[number_of_frames - 3 ] = on_stack_size(sender(sender(youngest)))
 494     frame_sizes[number_of_frames - 1 - index] = BytesPerWord * array->element(index)->on_stack_size(callee_parameters,
 495                                                                                                     callee_locals,
 496                                                                                                     index == 0,
 497                                                                                                     popframe_extra_args);
 498     // This pc doesn't have to be perfect just good enough to identify the frame
 499     // as interpreted so the skeleton frame will be walkable
 500     // The correct pc will be set when the skeleton frame is completely filled out
 501     // The final pc we store in the loop is wrong and will be overwritten below
 502     frame_pcs[number_of_frames - 1 - index ] = Interpreter::deopt_entry(vtos, 0) - frame::pc_return_offset;
 503 
 504     callee_parameters = array->element(index)->method()->size_of_parameters();
 505     callee_locals = array->element(index)->method()->max_locals();
 506     popframe_extra_args = 0;
 507   }
 508 
 509   // Compute whether the root vframe returns a float or double value.
 510   BasicType return_type;
 511   {
 512     methodHandle method(thread, array->element(0)->method());
 513     Bytecode_invoke invoke = Bytecode_invoke_check(method, array->element(0)->bci());
 514     return_type = invoke.is_valid() ? invoke.result_type() : T_ILLEGAL;
 515   }
 516 
 517   // Compute information for handling adapters and adjusting the frame size of the caller.
 518   int caller_adjustment = 0;
 519 
 520   // Compute the amount the oldest interpreter frame will have to adjust
 521   // its caller's stack by. If the caller is a compiled frame then
 522   // we pretend that the callee has no parameters so that the
 523   // extension counts for the full amount of locals and not just
 524   // locals-parms. This is because without a c2i adapter the parm
 525   // area as created by the compiled frame will not be usable by
 526   // the interpreter. (Depending on the calling convention there
 527   // may not even be enough space).
 528 
 529   // QQQ I'd rather see this pushed down into last_frame_adjust
 530   // and have it take the sender (aka caller).
 531 
 532   if (deopt_sender.is_compiled_frame() || caller_was_method_handle) {
 533     caller_adjustment = last_frame_adjust(0, callee_locals);
 534   } else if (callee_locals > callee_parameters) {
 535     // The caller frame may need extending to accommodate
 536     // non-parameter locals of the first unpacked interpreted frame.
 537     // Compute that adjustment.
 538     caller_adjustment = last_frame_adjust(callee_parameters, callee_locals);
 539   }
 540 
 541   // If the sender is deoptimized we must retrieve the address of the handler
 542   // since the frame will "magically" show the original pc before the deopt
 543   // and we'd undo the deopt.
 544 
 545   frame_pcs[0] = deopt_sender.raw_pc();
 546 
 547   assert(CodeCache::find_blob_unsafe(frame_pcs[0]) != NULL, "bad pc");
 548 
 549 #if INCLUDE_JVMCI
 550   if (exceptionObject() != NULL) {
 551     thread->set_exception_oop(exceptionObject());
 552     exec_mode = Unpack_exception;
 553   }
 554 #endif
 555 
 556   if (thread->frames_to_pop_failed_realloc() > 0 && exec_mode != Unpack_uncommon_trap) {
 557     assert(thread->has_pending_exception(), "should have thrown OOME");
 558     thread->set_exception_oop(thread->pending_exception());
 559     thread->clear_pending_exception();
 560     exec_mode = Unpack_exception;
 561   }
 562 
 563 #if INCLUDE_JVMCI
 564   if (thread->frames_to_pop_failed_realloc() > 0) {
 565     thread->set_pending_monitorenter(false);
 566   }
 567 #endif
 568 
 569   UnrollBlock* info = new UnrollBlock(array->frame_size() * BytesPerWord,
 570                                       caller_adjustment * BytesPerWord,
 571                                       caller_was_method_handle ? 0 : callee_parameters,
 572                                       number_of_frames,
 573                                       frame_sizes,
 574                                       frame_pcs,
 575                                       return_type,
 576                                       exec_mode);
 577   // On some platforms, we need a way to pass some platform dependent
 578   // information to the unpacking code so the skeletal frames come out
 579   // correct (initial fp value, unextended sp, ...)
 580   info->set_initial_info((intptr_t) array->sender().initial_deoptimization_info());
 581 
 582   if (array->frames() > 1) {
 583     if (VerifyStack && TraceDeoptimization) {
 584       ttyLocker ttyl;
 585       tty->print_cr("Deoptimizing method containing inlining");
 586     }
 587   }
 588 
 589   array->set_unroll_block(info);
 590   return info;
 591 }

 594 // after unpacking to stack and when stack overflow error occurs
 595 void Deoptimization::cleanup_deopt_info(JavaThread *thread,
 596                                         vframeArray *array) {
 597 
 598   // Get array if coming from exception
 599   if (array == NULL) {
 600     array = thread->vframe_array_head();
 601   }
 602   thread->set_vframe_array_head(NULL);
 603 
 604   // Free the previous UnrollBlock
 605   vframeArray* old_array = thread->vframe_array_last();
 606   thread->set_vframe_array_last(array);
 607 
 608   if (old_array != NULL) {
 609     UnrollBlock* old_info = old_array->unroll_block();
 610     old_array->set_unroll_block(NULL);
 611     delete old_info;
 612     delete old_array;
 613   }
 614 
 615   // Deallocate any resource creating in this routine and any ResourceObjs allocated
 616   // inside the vframeArray (StackValueCollections)
 617

 619   thread->set_deopt_mark(NULL);
 620   thread->set_deopt_compiled_method(NULL);
 621 
 622 
 623   if (JvmtiExport::can_pop_frame()) {
 624 #ifndef CC_INTERP
 625     // Regardless of whether we entered this routine with the pending
 626     // popframe condition bit set, we should always clear it now
 627     thread->clear_popframe_condition();

 638   // so that an asynchronous stack walker can work again. This counter is
 639   // incremented at the beginning of fetch_unroll_info() and (in C2) at
 640   // the beginning of uncommon_trap().
 641   thread->dec_in_deopt_handler();

 644 // Moved from cpu directories because none of the cpus has callee save values.
 645 // If a cpu implements callee save values, move this to deoptimization_<cpu>.cpp.
 646 void Deoptimization::unwind_callee_save_values(frame* f, vframeArray* vframe_array) {
 647 
 648   // This code is sort of the equivalent of C2IAdapter::setup_stack_frame back in
 649   // the days we had adapter frames. When we deoptimize a situation where a
 650   // compiled caller calls a compiled caller will have registers it expects
 651   // to survive the call to the callee. If we deoptimize the callee the only
 652   // way we can restore these registers is to have the oldest interpreter
 653   // frame that we create restore these values. That is what this routine
 654   // will accomplish.
 655 
 656   // At the moment we have modified c2 to not have any callee save registers
 657   // so this problem does not exist and this routine is just a place holder.
 658 
 659   assert(f->is_interpreted_frame(), "must be interpreted");
 660 }
 661 
 662 // Return BasicType of value being returned
 663 JRT_LEAF(BasicType, Deoptimization::unpack_frames(JavaThread* thread, int exec_mode))
 664 
 665   // We are already active in the special DeoptResourceMark any ResourceObj's we
 666   // allocate will be freed at the end of the routine.
 667 
 668   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 669   // but makes the entry a little slower. There is however a little dance we have to
 670   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 671   ResetNoHandleMark rnhm; // No-op in release/product versions
 672   HandleMark hm;
 673 
 674   frame stub_frame = thread->last_frame();
 675 
 676   // Since the frame to unpack is the top frame of this thread, the vframe_array_head
 677   // must point to the vframeArray for the unpack frame.
 678   vframeArray* array = thread->vframe_array_head();
 679 
 680 #ifndef PRODUCT
 681   if (TraceDeoptimization) {
 682     ttyLocker ttyl;
 683     tty->print_cr("DEOPT UNPACKING thread " INTPTR_FORMAT " vframeArray " INTPTR_FORMAT " mode %d",
 684                   p2i(thread), p2i(array), exec_mode);
 685   }
 686 #endif
 687   Events::log_deopt_message(thread, "DEOPT UNPACKING pc=" INTPTR_FORMAT " sp=" INTPTR_FORMAT " mode %d",
 688               p2i(stub_frame.pc()), p2i(stub_frame.sp()), exec_mode);
 689 
 690   UnrollBlock* info = array->unroll_block();
 691 
 692   // Unpack the interpreter frames and any adapter frame (c2 only) we might create.
 693   array->unpack_to_stack(stub_frame, exec_mode, info->caller_actual_parameters());
 694 
 695   BasicType bt = info->return_type();
 696 
 697   // If we have an exception pending, claim that the return type is an oop
 698   // so the deopt_blob does not overwrite the exception_oop.
 699 
 700   if (exec_mode == Unpack_exception)
 701     bt = T_OBJECT;
 702 
 703   // Cleanup thread deopt data
 704   cleanup_deopt_info(thread, array);
 705 
 706 #ifndef PRODUCT
 707   if (VerifyStack) {
 708     ResourceMark res_mark;
 709     // Clear pending exception to not break verification code (restored afterwards)
 710     PRESERVE_EXCEPTION_MARK;
 711 
 712     thread->validate_frame_layout();
 713 
 714     // Verify that the just-unpacked frames match the interpreter's
 715     // notions of expression stack and locals
 716     vframeArray* cur_array = thread->vframe_array_last();
 717     RegisterMap rm(thread, false);
 718     rm.set_include_argument_oops(false);
 719     bool is_top_frame = true;
 720     int callee_size_of_parameters = 0;
 721     int callee_max_locals = 0;
 722     for (int i = 0; i < cur_array->frames(); i++) {
 723       vframeArrayElement* el = cur_array->element(i);
 724       frame* iframe = el->iframe();
 725       guarantee(iframe->is_interpreted_frame(), "Wrong frame type");
 726 
 727       // Get the oop map for this bci
 728       InterpreterOopMap mask;
 729       int cur_invoke_parameter_size = 0;
 730       bool try_next_mask = false;
 731       int next_mask_expression_stack_size = -1;
 732       int top_frame_expression_stack_adjustment = 0;
 733       methodHandle mh(thread, iframe->interpreter_frame_method());
 734       OopMapCache::compute_one_oop_map(mh, iframe->interpreter_frame_bci(), &mask);
 735       BytecodeStream str(mh, iframe->interpreter_frame_bci());
 736       int max_bci = mh->code_size();
 737       // Get to the next bytecode if possible
 738       assert(str.bci() < max_bci, "bci in interpreter frame out of bounds");
 739       // Check to see if we can grab the number of outgoing arguments
 740       // at an uncommon trap for an invoke (where the compiler
 741       // generates debug info before the invoke has executed)
 742       Bytecodes::Code cur_code = str.next();
 743       if (Bytecodes::is_invoke(cur_code)) {
 744         Bytecode_invoke invoke(mh, iframe->interpreter_frame_bci());
 745         cur_invoke_parameter_size = invoke.size_of_parameters();
 746         if (i != 0 && !invoke.is_invokedynamic() && MethodHandles::has_member_arg(invoke.klass(), invoke.name())) {
 747           callee_size_of_parameters++;
 748         }
 749       }
 750       if (str.bci() < max_bci) {
 751         Bytecodes::Code next_code = str.next();
 752         if (next_code >= 0) {
 753           // The interpreter oop map generator reports results before
 754           // the current bytecode has executed except in the case of
 755           // calls. It seems to be hard to tell whether the compiler
 756           // has emitted debug information matching the "state before"
 757           // a given bytecode or the state after, so we try both
 758           if (!Bytecodes::is_invoke(cur_code) && cur_code != Bytecodes::_athrow) {
 759             // Get expression stack size for the next bytecode
 760             InterpreterOopMap next_mask;
 761             OopMapCache::compute_one_oop_map(mh, str.bci(), &next_mask);
 762             next_mask_expression_stack_size = next_mask.expression_stack_size();
 763             if (Bytecodes::is_invoke(next_code)) {
 764               Bytecode_invoke invoke(mh, str.bci());
 765               next_mask_expression_stack_size += invoke.size_of_parameters();
 766             }
 767             // Need to subtract off the size of the result type of
 768             // the bytecode because this is not described in the
 769             // debug info but returned to the interpreter in the TOS
 770             // caching register
 771             BasicType bytecode_result_type = Bytecodes::result_type(cur_code);
 772             if (bytecode_result_type != T_ILLEGAL) {
 773               top_frame_expression_stack_adjustment = type2size[bytecode_result_type];
 774             }
 775             assert(top_frame_expression_stack_adjustment >= 0, "stack adjustment must be positive");
 776             try_next_mask = true;
 777           }
 778         }
 779       }
 780 
 781       // Verify stack depth and oops in frame
 782       // This assertion may be dependent on the platform we're running on and may need modification (tested on x86 and sparc)
 783       if (!(
 784             /* SPARC */
 785             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_size_of_parameters) ||
 786             /* x86 */
 787             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_max_locals) ||
 788             (try_next_mask &&
 789              (iframe->interpreter_frame_expression_stack_size() == (next_mask_expression_stack_size -
 790                                                                     top_frame_expression_stack_adjustment))) ||
 791             (is_top_frame && (exec_mode == Unpack_exception) && iframe->interpreter_frame_expression_stack_size() == 0) ||
 792             (is_top_frame && (exec_mode == Unpack_uncommon_trap || exec_mode == Unpack_reexecute || el->should_reexecute()) &&
 793              (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + cur_invoke_parameter_size))
 794             )) {
 795         {
 796           ttyLocker ttyl;
 797 
 798           // Print out some information that will help us debug the problem
 799           tty->print_cr("Wrong number of expression stack elements during deoptimization");
 800           tty->print_cr("  Error occurred while verifying frame %d (0..%d, 0 is topmost)", i, cur_array->frames() - 1);
 801           tty->print_cr("  Fabricated interpreter frame had %d expression stack elements",
 802                         iframe->interpreter_frame_expression_stack_size());
 803           tty->print_cr("  Interpreter oop map had %d expression stack elements", mask.expression_stack_size());
 804           tty->print_cr("  try_next_mask = %d", try_next_mask);
 805           tty->print_cr("  next_mask_expression_stack_size = %d", next_mask_expression_stack_size);
 806           tty->print_cr("  callee_size_of_parameters = %d", callee_size_of_parameters);
 807           tty->print_cr("  callee_max_locals = %d", callee_max_locals);
 808           tty->print_cr("  top_frame_expression_stack_adjustment = %d", top_frame_expression_stack_adjustment);
 809           tty->print_cr("  exec_mode = %d", exec_mode);
 810           tty->print_cr("  cur_invoke_parameter_size = %d", cur_invoke_parameter_size);
 811           tty->print_cr("  Thread = " INTPTR_FORMAT ", thread ID = %d", p2i(thread), thread->osthread()->thread_id());
 812           tty->print_cr("  Interpreted frames:");
 813           for (int k = 0; k < cur_array->frames(); k++) {
 814             vframeArrayElement* el = cur_array->element(k);
 815             tty->print_cr("    %s (bci %d)", el->method()->name_and_sig_as_C_string(), el->bci());
 816           }
 817           cur_array->print_on_2(tty);
 818         } // release tty lock before calling guarantee
 819         guarantee(false, "wrong number of expression stack elements during deopt");
 820       }
 821       VerifyOopClosure verify;
 822       iframe->oops_interpreted_do(&verify, &rm, false);
 823       callee_size_of_parameters = mh->size_of_parameters();
 824       callee_max_locals = mh->max_locals();
 825       is_top_frame = false;
 826     }
 827   }
 828 #endif /* !PRODUCT */
 829 
 830 
 831   return bt;
 832 JRT_END

 923       return lookup(value);
 924     }
 925     PrimitiveType value = (PrimitiveType)*((jint*)&raw_value);
 926     return lookup(value);
 927   }

1001 bool Deoptimization::realloc_objects(JavaThread* thread, frame* fr, RegisterMap* reg_map, GrowableArray<ScopeValue*>* objects, TRAPS) {
1002   Handle pending_exception(THREAD, thread->pending_exception());
1003   const char* exception_file = thread->exception_file();
1004   int exception_line = thread->exception_line();
1005   thread->clear_pending_exception();
1006

1008

1010     assert(objects->at(i)->is_object(), "invalid debug information");
1011     ObjectValue* sv = (ObjectValue*) objects->at(i);
1012 
1013     Klass* k = java_lang_Class::as_Klass(sv->klass()->as_ConstantOopReadValue()->value()());
1014     oop obj = NULL;
1015 
1016     if (k->is_instance_klass()) {
1017 #if INCLUDE_JVMCI || INCLUDE_AOT
1018       CompiledMethod* cm = fr->cb()->as_compiled_method_or_null();
1019       if (cm->is_compiled_by_jvmci() && sv->is_auto_box()) {
1020         AutoBoxObjectValue* abv = (AutoBoxObjectValue*) sv;
1021         obj = get_cached_box(abv, fr, reg_map, THREAD);
1022         if (obj != NULL) {
1023           // Set the flag to indicate the box came from a cache, so that we can skip the field reassignment for it.
1024           abv->set_cached(true);
1025         }
1026       }
1027 #endif // INCLUDE_JVMCI || INCLUDE_AOT
1028       InstanceKlass* ik = InstanceKlass::cast(k);
1029       if (obj == NULL) {
1030         obj = ik->allocate_instance(THREAD);
1031       }
1032     } else if (k->is_valueArray_klass()) {
1033       ValueArrayKlass* ak = ValueArrayKlass::cast(k);
1034       // Value type array must be zeroed because not all memory is reassigned
1035       obj = ak->allocate(sv->field_size(), THREAD);
1036     } else if (k->is_typeArray_klass()) {

1039       int len = sv->field_size() / type2size[ak->element_type()];

1059     thread->set_pending_exception(pending_exception(), exception_file, exception_line);
1060   }

1081 /**
1082  * For primitive types whose kind gets "erased" at runtime (shorts become stack ints),
1083  * we need to somehow be able to recover the actual kind to be able to write the correct
1084  * amount of bytes.
1085  * For that purpose, this method assumes that, for an entry spanning n bytes at index i,
1086  * the entries at index n + 1 to n + i are 'markers'.
1087  * For example, if we were writing a short at index 4 of a byte array of size 8, the
1088  * expected form of the array would be:
1089  *
1090  * {b0, b1, b2, b3, INT, marker, b6, b7}
1091  *
1092  * Thus, in order to get back the size of the entry, we simply need to count the number
1093  * of marked entries
1094  *
1095  * @param virtualArray the virtualized byte array
1096  * @param i index of the virtual entry we are recovering
1097  * @return The number of bytes the entry spans
1098  */
1099 static int count_number_of_bytes_for_entry(ObjectValue *virtualArray, int i) {
1100   int index = i;
1101   while (++index < virtualArray->field_size() &&
1102            virtualArray->field_at(index)->is_marker()) {}
1103   return index - i;
1104 }

1106 /**
1107  * If there was a guarantee for byte array to always start aligned to a long, we could
1108  * do a simple check on the parity of the index. Unfortunately, that is not always the
1109  * case. Thus, we check alignment of the actual address we are writing to.
1110  * In the unlikely case index 0 is 5-aligned for example, it would then be possible to
1111  * write a long to index 3.
1112  */
1113 static jbyte* check_alignment_get_addr(typeArrayOop obj, int index, int expected_alignment) {
1114     jbyte* res = obj->byte_at_addr(index);
1115     assert((((intptr_t) res) % expected_alignment) == 0, "Non-aligned write");
1116     return res;
1117 }

1120   switch (byte_count) {
1121     case 1:
1122       obj->byte_at_put(index, (jbyte) *((jint *) &val));
1123       break;
1124     case 2:

1126       break;
1127     case 4:
1128       *((jint *) check_alignment_get_addr(obj, index, 4)) = (jint) *((jint *) &val);
1129       break;
1130     case 8:

1140 // restore elements of an eliminated type array
1141 void Deoptimization::reassign_type_array_elements(frame* fr, RegisterMap* reg_map, ObjectValue* sv, typeArrayOop obj, BasicType type) {
1142   int index = 0;
1143   intptr_t val;
1144 
1145   for (int i = 0; i < sv->field_size(); i++) {
1146     StackValue* value = StackValue::create_stack_value(fr, reg_map, sv->field_at(i));
1147     switch(type) {
1148     case T_LONG: case T_DOUBLE: {
1149       assert(value->type() == T_INT, "Agreement.");
1150       StackValue* low =
1151         StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1152 #ifdef _LP64
1153       jlong res = (jlong)low->get_int();
1154 #else
1155       jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1156 #endif
1157       obj->long_at_put(index, res);
1158       break;
1159     }
1160 
1161     // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1162     case T_INT: case T_FLOAT: { // 4 bytes.
1163       assert(value->type() == T_INT, "Agreement.");
1164       bool big_value = false;
1165       if (i + 1 < sv->field_size() && type == T_INT) {
1166         if (sv->field_at(i)->is_location()) {
1167           Location::Type type = ((LocationValue*) sv->field_at(i))->location().type();
1168           if (type == Location::dbl || type == Location::lng) {
1169             big_value = true;
1170           }
1171         } else if (sv->field_at(i)->is_constant_int()) {
1172           ScopeValue* next_scope_field = sv->field_at(i + 1);
1173           if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1174             big_value = true;
1175           }
1176         }
1177       }
1178 
1179       if (big_value) {
1180         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1181   #ifdef _LP64
1182         jlong res = (jlong)low->get_int();
1183   #else
1184         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1185   #endif
1186         obj->int_at_put(index, (jint)*((jint*)&res));
1187         obj->int_at_put(++index, (jint)*(((jint*)&res) + 1));
1188       } else {
1189         val = value->get_int();
1190         obj->int_at_put(index, (jint)*((jint*)&val));
1191       }
1192       break;
1193     }
1194 
1195     case T_SHORT:
1196       assert(value->type() == T_INT, "Agreement.");
1197       val = value->get_int();
1198       obj->short_at_put(index, (jshort)*((jint*)&val));
1199       break;
1200 
1201     case T_CHAR:
1202       assert(value->type() == T_INT, "Agreement.");
1203       val = value->get_int();
1204       obj->char_at_put(index, (jchar)*((jint*)&val));
1205       break;
1206 
1207     case T_BYTE: {
1208       assert(value->type() == T_INT, "Agreement.");
1209       // The value we get is erased as a regular int. We will need to find its actual byte count 'by hand'.
1210       val = value->get_int();
1211 #if INCLUDE_JVMCI
1212       int byte_count = count_number_of_bytes_for_entry(sv, i);
1213       byte_array_put(obj, val, index, byte_count);
1214       // According to byte_count contract, the values from i + 1 to i + byte_count are illegal values. Skip.
1215       i += byte_count - 1; // Balance the loop counter.
1216       index += byte_count;

1218       continue;
1219 #else
1220       obj->byte_at_put(index, (jbyte)*((jint*)&val));
1221       break;
1222 #endif // INCLUDE_JVMCI
1223     }
1224 
1225     case T_BOOLEAN: {
1226       assert(value->type() == T_INT, "Agreement.");
1227       val = value->get_int();
1228       obj->bool_at_put(index, (jboolean)*((jint*)&val));
1229       break;
1230     }
1231 
1232       default:
1233         ShouldNotReachHere();
1234     }
1235     index++;
1236   }
1237 }

1265 // Restore fields of an eliminated instance object using the same field order
1266 // returned by HotSpotResolvedObjectTypeImpl.getInstanceFields(true)
1267 static int reassign_fields_by_klass(InstanceKlass* klass, frame* fr, RegisterMap* reg_map, ObjectValue* sv, int svIndex, oop obj, bool skip_internal, int base_offset, TRAPS) {
1268 
1269   GrowableArray<ReassignedField>* fields = new GrowableArray<ReassignedField>();
1270   InstanceKlass* ik = klass;
1271   while (ik != NULL) {
1272     for (AllFieldStream fs(ik); !fs.done(); fs.next()) {
1273       if (!fs.access_flags().is_static() && (!skip_internal || !fs.access_flags().is_internal())) {
1274         ReassignedField field;
1275         field._offset = fs.offset();
1276         field._type = Signature::basic_type(fs.signature());
1277         if (field._type == T_VALUETYPE) {
1278           field._type = T_OBJECT;
1279         }
1280         if (fs.is_flattened()) {
1281           // Resolve klass of flattened value type field
1282           Klass* vk = klass->get_value_field_klass(fs.index());
1283           field._klass = ValueKlass::cast(vk);
1284           field._type = T_VALUETYPE;
1285         }
1286         fields->append(field);
1287       }
1288     }
1289     ik = ik->superklass();
1290   }
1291   fields->sort(compare);
1292   for (int i = 0; i < fields->length(); i++) {
1293     intptr_t val;
1294     ScopeValue* scope_field = sv->field_at(svIndex);
1295     StackValue* value = StackValue::create_stack_value(fr, reg_map, scope_field);
1296     int offset = base_offset + fields->at(i)._offset;
1297     BasicType type = fields->at(i)._type;
1298     switch (type) {
1299       case T_OBJECT:
1300       case T_ARRAY:
1301         assert(value->type() == T_OBJECT, "Agreement.");
1302         obj->obj_field_put(offset, value->get_obj()());
1303         break;
1304 
1305       case T_VALUETYPE: {
1306         // Recursively re-assign flattened value type fields
1307         InstanceKlass* vk = fields->at(i)._klass;
1308         assert(vk != NULL, "must be resolved");
1309         offset -= ValueKlass::cast(vk)->first_field_offset(); // Adjust offset to omit oop header
1310         svIndex = reassign_fields_by_klass(vk, fr, reg_map, sv, svIndex, obj, skip_internal, offset, CHECK_0);
1311         continue; // Continue because we don't need to increment svIndex
1312       }
1313 
1314       // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1315       case T_INT: case T_FLOAT: { // 4 bytes.
1316         assert(value->type() == T_INT, "Agreement.");
1317         bool big_value = false;
1318         if (i+1 < fields->length() && fields->at(i+1)._type == T_INT) {
1319           if (scope_field->is_location()) {
1320             Location::Type type = ((LocationValue*) scope_field)->location().type();
1321             if (type == Location::dbl || type == Location::lng) {
1322               big_value = true;
1323             }
1324           }
1325           if (scope_field->is_constant_int()) {
1326             ScopeValue* next_scope_field = sv->field_at(svIndex + 1);
1327             if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1328               big_value = true;
1329             }
1330           }
1331         }
1332 
1333         if (big_value) {
1334           i++;
1335           assert(i < fields->length(), "second T_INT field needed");
1336           assert(fields->at(i)._type == T_INT, "T_INT field needed");
1337         } else {
1338           val = value->get_int();
1339           obj->int_field_put(offset, (jint)*((jint*)&val));
1340           break;
1341         }
1342       }
1343         /* no break */
1344 
1345       case T_LONG: case T_DOUBLE: {
1346         assert(value->type() == T_INT, "Agreement.");
1347         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++svIndex));
1348 #ifdef _LP64
1349         jlong res = (jlong)low->get_int();
1350 #else
1351         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1352 #endif
1353         obj->long_field_put(offset, res);
1354         break;
1355       }
1356 
1357       case T_SHORT:
1358         assert(value->type() == T_INT, "Agreement.");
1359         val = value->get_int();
1360         obj->short_field_put(offset, (jshort)*((jint*)&val));
1361         break;
1362 
1363       case T_CHAR:
1364         assert(value->type() == T_INT, "Agreement.");
1365         val = value->get_int();
1366         obj->char_field_put(offset, (jchar)*((jint*)&val));
1367         break;
1368 
1369       case T_BYTE:
1370         assert(value->type() == T_INT, "Agreement.");
1371         val = value->get_int();
1372         obj->byte_field_put(offset, (jbyte)*((jint*)&val));
1373         break;
1374 
1375       case T_BOOLEAN:
1376         assert(value->type() == T_INT, "Agreement.");
1377         val = value->get_int();
1378         obj->bool_field_put(offset, (jboolean)*((jint*)&val));
1379         break;
1380 
1381       default:
1382         ShouldNotReachHere();
1383     }
1384     svIndex++;
1385   }
1386   return svIndex;
1387 }

1439 void Deoptimization::relock_objects(GrowableArray<MonitorInfo*>* monitors, JavaThread* thread, bool realloc_failures) {
1440   for (int i = 0; i < monitors->length(); i++) {
1441     MonitorInfo* mon_info = monitors->at(i);

1443       assert(!mon_info->owner_is_scalar_replaced() || realloc_failures, "reallocation was missed");

1445         Handle obj(thread, mon_info->owner());
1446         markWord mark = obj->mark();
1447         if (UseBiasedLocking && mark.has_bias_pattern()) {
1448           // New allocated objects may have the mark set to anonymously biased.
1449           // Also the deoptimized method may called methods with synchronization
1450           // where the thread-local object is bias locked to the current thread.
1451           assert(mark.is_biased_anonymously() ||
1452                  mark.biased_locker() == thread, "should be locked to current thread");
1453           // Reset mark word to unbiased prototype.
1454           markWord unbiased_prototype = markWord::prototype().set_age(mark.age());
1455           obj->set_mark(unbiased_prototype);
1456         }
1457         BasicLock* lock = mon_info->lock();
1458         ObjectSynchronizer::enter(obj, lock, thread);
1459         assert(mon_info->owner()->is_locked(), "object must be locked now");
1460       }
1461     }
1462   }

1592 static void get_monitors_from_stack(GrowableArray<Handle>* objects_to_revoke, JavaThread* thread, frame fr, RegisterMap* map) {
1593   // Unfortunately we don't have a RegisterMap available in most of
1594   // the places we want to call this routine so we need to walk the
1595   // stack again to update the register map.
1596   if (map == NULL || !map->update_map()) {
1597     StackFrameStream sfs(thread, true);
1598     bool found = false;
1599     while (!found && !sfs.is_done()) {
1600       frame* cur = sfs.current();
1601       sfs.next();
1602       found = cur->id() == fr.id();
1603     }
1604     assert(found, "frame to be deoptimized not found on target thread's stack");
1605     map = sfs.register_map();
1606   }
1607 
1608   vframe* vf = vframe::new_vframe(&fr, map, thread);
1609   compiledVFrame* cvf = compiledVFrame::cast(vf);
1610   // Revoke monitors' biases in all scopes
1611   while (!cvf->is_top()) {
1612     collect_monitors(cvf, objects_to_revoke);
1613     cvf = compiledVFrame::cast(cvf->sender());
1614   }
1615   collect_monitors(cvf, objects_to_revoke);
1616 }

1619   if (!UseBiasedLocking) {

1853   HandleMark hm;
1854 
1855   // uncommon_trap() is called at the beginning of the uncommon trap
1856   // handler. Note this fact before we start generating temporary frames
1857   // that can confuse an asynchronous stack walker. This counter is
1858   // decremented at the end of unpack_frames().
1859   thread->inc_in_deopt_handler();

1861   // We need to update the map if we have biased locking.
1862 #if INCLUDE_JVMCI

1864   RegisterMap reg_map(thread, true);
1865 #else
1866   RegisterMap reg_map(thread, UseBiasedLocking);
1867 #endif

1869   frame fr = stub_frame.sender(&reg_map);
1870   // Make sure the calling nmethod is not getting deoptimized and removed
1871   // before we are done with it.
1872   nmethodLocker nl(fr.pc());
1873 
1874   // Log a message
1875   Events::log_deopt_message(thread, "Uncommon trap: trap_request=" PTR32_FORMAT " fr.pc=" INTPTR_FORMAT " relative=" INTPTR_FORMAT,
1876               trap_request, p2i(fr.pc()), fr.pc() - fr.cb()->code_begin());
1877 
1878   {
1879     ResourceMark rm;
1880 
1881     DeoptReason reason = trap_request_reason(trap_request);
1882     DeoptAction action = trap_request_action(trap_request);
1883 #if INCLUDE_JVMCI
1884     int debug_id = trap_request_debug_id(trap_request);
1885 #endif
1886     jint unloaded_class_index = trap_request_index(trap_request); // CP idx or -1
1887 
1888     vframe*  vf  = vframe::new_vframe(&fr, &reg_map, thread);
1889     compiledVFrame* cvf = compiledVFrame::cast(vf);
1890 
1891     CompiledMethod* nm = cvf->code();
1892 
1893     ScopeDesc*      trap_scope  = cvf->scope();
1894 
1895     if (TraceDeoptimization) {
1896       ttyLocker ttyl;
1897       tty->print_cr("  bci=%d pc=" INTPTR_FORMAT ", relative_pc=" INTPTR_FORMAT ", method=%s" JVMCI_ONLY(", debug_id=%d"), trap_scope->bci(), p2i(fr.pc()), fr.pc() - nm->code_begin(), trap_scope->method()->name_and_sig_as_C_string()
1898 #if INCLUDE_JVMCI
1899           , debug_id
1900 #endif
1901           );
1902     }
1903 
1904     methodHandle    trap_method(THREAD, trap_scope->method());
1905     int             trap_bci    = trap_scope->bci();
1906 #if INCLUDE_JVMCI
1907     jlong           speculation = thread->pending_failed_speculation();
1908     if (nm->is_compiled_by_jvmci() && nm->is_nmethod()) { // Exclude AOTed methods
1909       nm->as_nmethod()->update_speculation(thread);
1910     } else {
1911       assert(speculation == 0, "There should not be a speculation for methods compiled by non-JVMCI compilers");
1912     }
1913 
1914     if (trap_bci == SynchronizationEntryBCI) {
1915       trap_bci = 0;
1916       thread->set_pending_monitorenter(true);
1917     }
1918 
1919     if (reason == Deoptimization::Reason_transfer_to_interpreter) {
1920       thread->set_pending_transfer_to_interpreter(true);
1921     }
1922 #endif
1923 
1924     Bytecodes::Code trap_bc     = trap_method->java_code_at(trap_bci);
1925     // Record this event in the histogram.
1926     gather_statistics(reason, action, trap_bc);
1927 
1928     // Ensure that we can record deopt. history:
1929     // Need MDO to record RTM code generation state.
1930     bool create_if_missing = ProfileTraps || UseCodeAging RTM_OPT_ONLY( || UseRTMLocking );
1931 
1932     methodHandle profiled_method;
1933 #if INCLUDE_JVMCI
1934     if (nm->is_compiled_by_jvmci()) {
1935       profiled_method = methodHandle(THREAD, nm->method());
1936     } else {
1937       profiled_method = trap_method;
1938     }
1939 #else
1940     profiled_method = trap_method;
1941 #endif
1942 
1943     MethodData* trap_mdo =
1944       get_method_data(thread, profiled_method, create_if_missing);
1945 
1946     JFR_ONLY(post_deoptimization_event(nm, trap_method(), trap_bci, trap_bc, reason, action);)
1947 
1948     // Log a message
1949     Events::log_deopt_message(thread, "Uncommon trap: reason=%s action=%s pc=" INTPTR_FORMAT " method=%s @ %d %s",
1950                               trap_reason_name(reason), trap_action_name(action), p2i(fr.pc()),
1951                               trap_method->name_and_sig_as_C_string(), trap_bci, nm->compiler_name());
1952 
1953     // Print a bunch of diagnostics, if requested.
1954     if (TraceDeoptimization || LogCompilation) {
1955       ResourceMark rm;
1956       ttyLocker ttyl;
1957       char buf[100];
1958       if (xtty != NULL) {
1959         xtty->begin_head("uncommon_trap thread='" UINTX_FORMAT "' %s",
1960                          os::current_thread_id(),
1961                          format_trap_request(buf, sizeof(buf), trap_request));
1962 #if INCLUDE_JVMCI
1963         if (speculation != 0) {
1964           xtty->print(" speculation='" JLONG_FORMAT "'", speculation);
1965         }
1966 #endif
1967         nm->log_identity(xtty);
1968       }
1969       Symbol* class_name = NULL;
1970       bool unresolved = false;
1971       if (unloaded_class_index >= 0) {
1972         constantPoolHandle constants (THREAD, trap_method->constants());
1973         if (constants->tag_at(unloaded_class_index).is_unresolved_klass()) {
1974           class_name = constants->klass_name_at(unloaded_class_index);
1975           unresolved = true;
1976           if (xtty != NULL)
1977             xtty->print(" unresolved='1'");
1978         } else if (constants->tag_at(unloaded_class_index).is_symbol()) {
1979           class_name = constants->symbol_at(unloaded_class_index);
1980         }
1981         if (xtty != NULL)
1982           xtty->name(class_name);
1983       }
1984       if (xtty != NULL && trap_mdo != NULL && (int)reason < (int)MethodData::_trap_hist_limit) {
1985         // Dump the relevant MDO state.
1986         // This is the deopt count for the current reason, any previous
1987         // reasons or recompiles seen at this point.
1988         int dcnt = trap_mdo->trap_count(reason);
1989         if (dcnt != 0)
1990           xtty->print(" count='%d'", dcnt);
1991         ProfileData* pdata = trap_mdo->bci_to_data(trap_bci);
1992         int dos = (pdata == NULL)? 0: pdata->trap_state();
1993         if (dos != 0) {
1994           xtty->print(" state='%s'", format_trap_state(buf, sizeof(buf), dos));
1995           if (trap_state_is_recompiled(dos)) {
1996             int recnt2 = trap_mdo->overflow_recompile_count();
1997             if (recnt2 != 0)
1998               xtty->print(" recompiles2='%d'", recnt2);
1999           }
2000         }
2001       }
2002       if (xtty != NULL) {
2003         xtty->stamp();
2004         xtty->end_head();
2005       }
2006       if (TraceDeoptimization) {  // make noise on the tty
2007         tty->print("Uncommon trap occurred in");
2008         nm->method()->print_short_name(tty);
2009         tty->print(" compiler=%s compile_id=%d", nm->compiler_name(), nm->compile_id());
2010 #if INCLUDE_JVMCI
2011         if (nm->is_nmethod()) {
2012           const char* installed_code_name = nm->as_nmethod()->jvmci_name();
2013           if (installed_code_name != NULL) {
2014             tty->print(" (JVMCI: installed code name=%s) ", installed_code_name);
2015           }
2016         }
2017 #endif
2018         tty->print(" (@" INTPTR_FORMAT ") thread=" UINTX_FORMAT " reason=%s action=%s unloaded_class_index=%d" JVMCI_ONLY(" debug_id=%d"),
2019                    p2i(fr.pc()),
2020                    os::current_thread_id(),
2021                    trap_reason_name(reason),
2022                    trap_action_name(action),
2023                    unloaded_class_index
2024 #if INCLUDE_JVMCI
2025                    , debug_id
2026 #endif
2027                    );
2028         if (class_name != NULL) {
2029           tty->print(unresolved ? " unresolved class: " : " symbol: ");
2030           class_name->print_symbol_on(tty);
2031         }
2032         tty->cr();
2033       }
2034       if (xtty != NULL) {
2035         // Log the precise location of the trap.
2036         for (ScopeDesc* sd = trap_scope; ; sd = sd->sender()) {
2037           xtty->begin_elem("jvms bci='%d'", sd->bci());
2038           xtty->method(sd->method());
2039           xtty->end_elem();
2040           if (sd->is_top())  break;
2041         }
2042         xtty->tail("uncommon_trap");
2043       }
2044     }
2045     // (End diagnostic printout.)
2046 
2047     // Load class if necessary
2048     if (unloaded_class_index >= 0) {
2049       constantPoolHandle constants(THREAD, trap_method->constants());
2050       load_class_by_index(constants, unloaded_class_index);
2051     }
2052 
2053     // Flush the nmethod if necessary and desirable.
2054     //
2055     // We need to avoid situations where we are re-flushing the nmethod
2056     // because of a hot deoptimization site.  Repeated flushes at the same
2057     // point need to be detected by the compiler and avoided.  If the compiler
2058     // cannot avoid them (or has a bug and "refuses" to avoid them), this
2059     // module must take measures to avoid an infinite cycle of recompilation
2060     // and deoptimization.  There are several such measures:
2061     //
2062     //   1. If a recompilation is ordered a second time at some site X
2063     //   and for the same reason R, the action is adjusted to 'reinterpret',
2064     //   to give the interpreter time to exercise the method more thoroughly.
2065     //   If this happens, the method's overflow_recompile_count is incremented.
2066     //
2067     //   2. If the compiler fails to reduce the deoptimization rate, then
2068     //   the method's overflow_recompile_count will begin to exceed the set
2069     //   limit PerBytecodeRecompilationCutoff.  If this happens, the action
2070     //   is adjusted to 'make_not_compilable', and the method is abandoned
2071     //   to the interpreter.  This is a performance hit for hot methods,
2072     //   but is better than a disastrous infinite cycle of recompilations.
2073     //   (Actually, only the method containing the site X is abandoned.)
2074     //
2075     //   3. In parallel with the previous measures, if the total number of
2076     //   recompilations of a method exceeds the much larger set limit
2077     //   PerMethodRecompilationCutoff, the method is abandoned.
2078     //   This should only happen if the method is very large and has
2079     //   many "lukewarm" deoptimizations.  The code which enforces this
2080     //   limit is elsewhere (class nmethod, class Method).
2081     //
2082     // Note that the per-BCI 'is_recompiled' bit gives the compiler one chance
2083     // to recompile at each bytecode independently of the per-BCI cutoff.
2084     //
2085     // The decision to update code is up to the compiler, and is encoded
2086     // in the Action_xxx code.  If the compiler requests Action_none
2087     // no trap state is changed, no compiled code is changed, and the
2088     // computation suffers along in the interpreter.
2089     //
2090     // The other action codes specify various tactics for decompilation
2091     // and recompilation.  Action_maybe_recompile is the loosest, and
2092     // allows the compiled code to stay around until enough traps are seen,
2093     // and until the compiler gets around to recompiling the trapping method.
2094     //
2095     // The other actions cause immediate removal of the present code.
2096 
2097     // Traps caused by injected profile shouldn't pollute trap counts.
2098     bool injected_profile_trap = trap_method->has_injected_profile() &&
2099                                  (reason == Reason_intrinsic || reason == Reason_unreached);
2100 
2101     bool update_trap_state = (reason != Reason_tenured) && !injected_profile_trap;

2103     bool make_not_compilable = false;
2104     bool reprofile = false;

2106     case Action_none:
2107       // Keep the old code.
2108       update_trap_state = false;
2109       break;
2110     case Action_maybe_recompile:
2111       // Do not need to invalidate the present code, but we can
2112       // initiate another
2113       // Start compiler without (necessarily) invalidating the nmethod.
2114       // The system will tolerate the old code, but new code should be
2115       // generated when possible.
2116       break;
2117     case Action_reinterpret:
2118       // Go back into the interpreter for a while, and then consider
2119       // recompiling form scratch.
2120       make_not_entrant = true;
2121       // Reset invocation counter for outer most method.
2122       // This will allow the interpreter to exercise the bytecodes
2123       // for a while before recompiling.
2124       // By contrast, Action_make_not_entrant is immediate.
2125       //
2126       // Note that the compiler will track null_check, null_assert,
2127       // range_check, and class_check events and log them as if they
2128       // had been traps taken from compiled code.  This will update
2129       // the MDO trap history so that the next compilation will
2130       // properly detect hot trap sites.
2131       reprofile = true;
2132       break;
2133     case Action_make_not_entrant:
2134       // Request immediate recompilation, and get rid of the old code.
2135       // Make them not entrant, so next time they are called they get
2136       // recompiled.  Unloaded classes are loaded now so recompile before next
2137       // time they are called.  Same for uninitialized.  The interpreter will
2138       // link the missing class, if any.
2139       make_not_entrant = true;
2140       break;
2141     case Action_make_not_compilable:
2142       // Give up on compiling this method at all.
2143       make_not_entrant = true;
2144       make_not_compilable = true;
2145       break;
2146     default:
2147       ShouldNotReachHere();
2148     }

2151     // 4852688: ProfileInterpreter is off by default for ia64.  The result is
2152     // infinite heroic-opt-uncommon-trap/deopt/recompile cycles, since the
2153     // recompile relies on a MethodData* to record heroic opt failures.
2154 
2155     // Whether the interpreter is producing MDO data or not, we also need
2156     // to use the MDO to detect hot deoptimization points and control
2157     // aggressive optimization.
2158     bool inc_recompile_count = false;
2159     ProfileData* pdata = NULL;
2160     if (ProfileTraps && !is_client_compilation_mode_vm() && update_trap_state && trap_mdo != NULL) {
2161       assert(trap_mdo == get_method_data(thread, profiled_method, false), "sanity");
2162       uint this_trap_count = 0;
2163       bool maybe_prior_trap = false;
2164       bool maybe_prior_recompile = false;

2166 #if INCLUDE_JVMCI
2167                                    nm->is_compiled_by_jvmci() && nm->is_osr_method(),
2168 #endif
2169                                    nm->method(),
2170                                    //outputs:

2173                                    maybe_prior_recompile);

2175       // checks, these traps from compiled code are double-counted.
2176       // This is harmless; it just means that the PerXTrapLimit values
2177       // are in effect a little smaller than they look.
2178

2180       if (per_bc_reason != Reason_none) {
2181         // Now take action based on the partially known per-BCI history.
2182         if (maybe_prior_trap
2183             && this_trap_count >= (uint)PerBytecodeTrapLimit) {
2184           // If there are too many traps at this BCI, force a recompile.
2185           // This will allow the compiler to see the limit overflow, and
2186           // take corrective action, if possible.  The compiler generally
2187           // does not use the exact PerBytecodeTrapLimit value, but instead
2188           // changes its tactics if it sees any traps at all.  This provides
2189           // a little hysteresis, delaying a recompile until a trap happens
2190           // several times.
2191           //
2192           // Actually, since there is only one bit of counter per BCI,
2193           // the possible per-BCI counts are {0,1,(per-method count)}.
2194           // This produces accurate results if in fact there is only
2195           // one hot trap site, but begins to get fuzzy if there are
2196           // many sites.  For example, if there are ten sites each
2197           // trapping two or more times, they each get the blame for
2198           // all of their traps.
2199           make_not_entrant = true;
2200         }
2201 
2202         // Detect repeated recompilation at the same BCI, and enforce a limit.
2203         if (make_not_entrant && maybe_prior_recompile) {
2204           // More than one recompile at this point.
2205           inc_recompile_count = maybe_prior_trap;
2206         }
2207       } else {
2208         // For reasons which are not recorded per-bytecode, we simply
2209         // force recompiles unconditionally.
2210         // (Note that PerMethodRecompilationCutoff is enforced elsewhere.)
2211         make_not_entrant = true;
2212       }
2213 
2214       // Go back to the compiler if there are too many traps in this method.
2215       if (this_trap_count >= per_method_trap_limit(reason)) {
2216         // If there are too many traps in this method, force a recompile.
2217         // This will allow the compiler to see the limit overflow, and
2218         // take corrective action, if possible.
2219         // (This condition is an unlikely backstop only, because the
2220         // PerBytecodeTrapLimit is more likely to take effect first,
2221         // if it is applicable.)
2222         make_not_entrant = true;
2223       }
2224 
2225       // Here's more hysteresis:  If there has been a recompile at
2226       // this trap point already, run the method in the interpreter
2227       // for a while to exercise it more thoroughly.
2228       if (make_not_entrant && maybe_prior_recompile && maybe_prior_trap) {
2229         reprofile = true;
2230       }
2231     }
2232 
2233     // Take requested actions on the method:
2234 
2235     // Recompile
2236     if (make_not_entrant) {
2237       if (!nm->make_not_entrant()) {
2238         return; // the call did not change nmethod's state
2239       }

2243         int tstate0 = pdata->trap_state();

2245         if (tstate1 != tstate0)

2247       }
2248 
2249 #if INCLUDE_RTM_OPT
2250       // Restart collecting RTM locking abort statistic if the method
2251       // is recompiled for a reason other than RTM state change.
2252       // Assume that in new recompiled code the statistic could be different,
2253       // for example, due to different inlining.

2258 #endif
2259       // For code aging we count traps separately here, using make_not_entrant()
2260       // as a guard against simultaneous deopts in multiple threads.
2261       if (reason == Reason_tenured && trap_mdo != NULL) {
2262         trap_mdo->inc_tenure_traps();
2263       }
2264     }
2265 
2266     if (inc_recompile_count) {

2269           (uint)PerBytecodeRecompilationCutoff) {
2270         // Give up on the method containing the bad BCI.
2271         if (trap_method() == nm->method()) {
2272           make_not_compilable = true;
2273         } else {
2274           trap_method->set_not_compilable("overflow_recompile_count > PerBytecodeRecompilationCutoff", CompLevel_full_optimization);
2275           // But give grace to the enclosing nm->method().
2276         }
2277       }
2278     }
2279 
2280     // Reprofile
2281     if (reprofile) {
2282       CompilationPolicy::policy()->reprofile(trap_scope, nm->is_osr_method());
2283     }
2284 
2285     // Give up compiling
2286     if (make_not_compilable && !nm->method()->is_not_compilable(CompLevel_full_optimization)) {
2287       assert(make_not_entrant, "consistent");
2288       nm->method()->set_not_compilable("give up compiling", CompLevel_full_optimization);
2289     }
2290 
2291   } // Free marked resources
2292

2417 const int DS_RECOMPILE_BIT = DataLayout::trap_mask - DS_REASON_MASK;

2419 //---------------------------trap_state_reason---------------------------------
2420 Deoptimization::DeoptReason
2421 Deoptimization::trap_state_reason(int trap_state) {
2422   // This assert provides the link between the width of DataLayout::trap_bits
2423   // and the encoding of "recorded" reasons.  It ensures there are enough
2424   // bits to store all needed reasons in the per-BCI MDO profile.
2425   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2426   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2427   trap_state -= recompile_bit;
2428   if (trap_state == DS_REASON_MASK) {
2429     return Reason_many;
2430   } else {
2431     assert((int)Reason_none == 0, "state=0 => Reason_none");
2432     return (DeoptReason)trap_state;
2433   }
2434 }
2435 //-------------------------trap_state_has_reason-------------------------------
2436 int Deoptimization::trap_state_has_reason(int trap_state, int reason) {
2437   assert(reason_is_recorded_per_bytecode((DeoptReason)reason), "valid reason");
2438   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2439   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2440   trap_state -= recompile_bit;
2441   if (trap_state == DS_REASON_MASK) {
2442     return -1;  // true, unspecifically (bottom of state lattice)
2443   } else if (trap_state == reason) {
2444     return 1;   // true, definitely
2445   } else if (trap_state == 0) {
2446     return 0;   // false, definitely (top of state lattice)
2447   } else {
2448     return 0;   // false, definitely
2449   }
2450 }
2451 //-------------------------trap_state_add_reason-------------------------------
2452 int Deoptimization::trap_state_add_reason(int trap_state, int reason) {
2453   assert(reason_is_recorded_per_bytecode((DeoptReason)reason) || reason == Reason_many, "valid reason");
2454   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2455   trap_state -= recompile_bit;
2456   if (trap_state == DS_REASON_MASK) {
2457     return trap_state + recompile_bit;     // already at state lattice bottom
2458   } else if (trap_state == reason) {
2459     return trap_state + recompile_bit;     // the condition is already true
2460   } else if (trap_state == 0) {
2461     return reason + recompile_bit;          // no condition has yet been true
2462   } else {
2463     return DS_REASON_MASK + recompile_bit;  // fall to state lattice bottom
2464   }
2465 }
2466 //-----------------------trap_state_is_recompiled------------------------------
2467 bool Deoptimization::trap_state_is_recompiled(int trap_state) {
2468   return (trap_state & DS_RECOMPILE_BIT) != 0;
2469 }

2472   if (z)  return trap_state |  DS_RECOMPILE_BIT;

2611 void Deoptimization::gather_statistics(DeoptReason reason, DeoptAction action,
2612                                        Bytecodes::Code bc) {
2613   assert(reason >= 0 && reason < Reason_LIMIT, "oob");
2614   assert(action >= 0 && action < Action_LIMIT, "oob");
2615   _deoptimization_hist[Reason_none][0][0] += 1;  // total
2616   _deoptimization_hist[reason][0][0]      += 1;  // per-reason total
2617   juint* cases = _deoptimization_hist[reason][1+action];
2618   juint* bc_counter_addr = NULL;
2619   juint  bc_counter      = 0;
2620   // Look for an unused counter, or an exact match to this BC.
2621   if (bc != Bytecodes::_illegal) {
2622     for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2623       juint* counter_addr = &cases[bc_case];
2624       juint  counter = *counter_addr;
2625       if ((counter == 0 && bc_counter_addr == NULL)
2626           || (Bytecodes::Code)(counter & LSB_MASK) == bc) {
2627         // this counter is either free or is already devoted to this BC
2628         bc_counter_addr = counter_addr;
2629         bc_counter = counter | bc;
2630       }
2631     }
2632   }
2633   if (bc_counter_addr == NULL) {
2634     // Overflow, or no given bytecode.
2635     bc_counter_addr = &cases[BC_CASE_LIMIT-1];
2636     bc_counter = (*bc_counter_addr & ~LSB_MASK);  // clear LSB
2637   }
2638   *bc_counter_addr = bc_counter + (1 << LSB_BITS);

2645 void Deoptimization::print_statistics() {
2646   juint total = total_deoptimization_count();
2647   juint account = total;
2648   if (total != 0) {
2649     ttyLocker ttyl;
2650     if (xtty != NULL)  xtty->head("statistics type='deoptimization'");
2651     tty->print_cr("Deoptimization traps recorded:");
2652     #define PRINT_STAT_LINE(name, r) \
2653       tty->print_cr("  %4d (%4.1f%%) %s", (int)(r), ((r) * 100.0) / total, name);
2654     PRINT_STAT_LINE("total", total);
2655     // For each non-zero entry in the histogram, print the reason,
2656     // the action, and (if specifically known) the type of bytecode.
2657     for (int reason = 0; reason < Reason_LIMIT; reason++) {
2658       for (int action = 0; action < Action_LIMIT; action++) {
2659         juint* cases = _deoptimization_hist[reason][1+action];
2660         for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2661           juint counter = cases[bc_case];
2662           if (counter != 0) {
2663             char name[1*K];
2664             Bytecodes::Code bc = (Bytecodes::Code)(counter & LSB_MASK);
2665             if (bc_case == BC_CASE_LIMIT && (int)bc == 0)
2666               bc = Bytecodes::_illegal;
2667             sprintf(name, "%s/%s/%s",
2668                     trap_reason_name(reason),
2669                     trap_action_name(action),
2670                     Bytecodes::is_defined(bc)? Bytecodes::name(bc): "other");
2671             juint r = counter >> LSB_BITS;
2672             tty->print_cr("  %40s: " UINT32_FORMAT " (%.1f%%)", name, r, (r * 100.0) / total);
2673             account -= r;
2674           }
2675         }
2676       }
2677     }
2678     if (account != 0) {
2679       PRINT_STAT_LINE("unaccounted", account);
2680     }
2681     #undef PRINT_STAT_LINE
2682     if (xtty != NULL)  xtty->tail("statistics");
2683   }
2684 }
