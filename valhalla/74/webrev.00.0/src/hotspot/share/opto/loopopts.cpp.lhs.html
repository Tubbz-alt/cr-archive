  48 //------------------------------split_thru_phi---------------------------------
  49 // Split Node 'n' through merge point if there is enough win.
  50 Node *PhaseIdealLoop::split_thru_phi( Node *n, Node *region, int policy ) {
  51   if (n->Opcode() == Op_ConvI2L && n->bottom_type() != TypeLong::LONG) {
  52     // ConvI2L may have type information on it which is unsafe to push up
  53     // so disable this for now
  54     return NULL;
  55   }
  56 
  57   // Splitting range check CastIIs through a loop induction Phi can
  58   // cause new Phis to be created that are left unrelated to the loop
  59   // induction Phi and prevent optimizations (vectorization)
  60   if (n->Opcode() == Op_CastII && n->as_CastII()->has_range_check() &&
  61       region->is_CountedLoop() && n->in(1) == region->as_CountedLoop()->phi()) {
  62     return NULL;
  63   }
  64 
  65   // Value types should not be split through Phis because they cannot be merged
  66   // through Phi nodes but each value input needs to be merged individually.
  67   if (n->is_ValueType()) {
  68     return NULL;
  69   }
  70 
  71   int wins = 0;
  72   assert(!n->is_CFG(), "");
  73   assert(region->is_Region(), "");
  74 
  75   const Type* type = n->bottom_type();
  76   const TypeOopPtr *t_oop = _igvn.type(n)->isa_oopptr();
  77   Node *phi;
  78   if (t_oop != NULL && t_oop->is_known_instance_field()) {
  79     int iid    = t_oop->instance_id();
  80     int index  = C->get_alias_index(t_oop);
  81     int offset = t_oop->offset();
  82     phi = new PhiNode(region, type, NULL, iid, index, offset);
  83   } else {
  84     phi = PhiNode::make_blank(region, n);
  85   }
  86   uint old_unique = C->unique();
  87   for (uint i = 1; i < region->req(); i++) {
  88     Node *x;
  89     Node* the_clone = NULL;
  90     if (region->in(i) == C->top()) {
  91       x = C->top();             // Dead path?  Use a dead data op
  92     } else {
  93       x = n->clone();           // Else clone up the data op
  94       the_clone = x;            // Remember for possible deletion.
  95       // Alter data node to use pre-phi inputs
  96       if (n->in(0) == region)
  97         x->set_req( 0, region->in(i) );
  98       for (uint j = 1; j < n->req(); j++) {
  99         Node *in = n->in(j);
 100         if (in->is_Phi() && in->in(0) == region)
 101           x->set_req( j, in->in(i) ); // Use pre-Phi input for the clone
 102       }
 103     }
 104     // Check for a 'win' on some paths
 105     const Type *t = x->Value(&_igvn);
 106 
 107     bool singleton = t->singleton();
 108 
 109     // A TOP singleton indicates that there are no possible values incoming
 110     // along a particular edge. In most cases, this is OK, and the Phi will
 111     // be eliminated later in an Ideal call. However, we can't allow this to
 112     // happen if the singleton occurs on loop entry, as the elimination of
 113     // the PhiNode may cause the resulting node to migrate back to a previous
 114     // loop iteration.
 115     if (singleton && t == Type::TOP) {
 116       // Is_Loop() == false does not confirm the absence of a loop (e.g., an
 117       // irreducible loop may not be indicated by an affirmative is_Loop());
 118       // therefore, the only top we can split thru a phi is on a backedge of
 119       // a loop.
 120       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
 121     }
 122 
 123     if (singleton) {
 124       wins++;
 125       x = ((PhaseGVN&)_igvn).makecon(t);
 126     } else {
 127       // We now call Identity to try to simplify the cloned node.
 128       // Note that some Identity methods call phase->type(this).
 129       // Make sure that the type array is big enough for
 130       // our new node, even though we may throw the node away.
 131       // (Note: This tweaking with igvn only works because x is a new node.)
 132       _igvn.set_type(x, t);
 133       // If x is a TypeNode, capture any more-precise type permanently into Node
 134       // otherwise it will be not updated during igvn->transform since
 135       // igvn->type(x) is set to x->Value() already.
 136       x->raise_bottom_type(t);
 137       Node* y = x->Identity(&_igvn);
 138       if (y != x) {
 139         wins++;
 140         x = y;
 141       } else {
 142         y = _igvn.hash_find(x);
 143         if (y) {
 144           wins++;
 145           x = y;
 146         } else {
 147           // Else x is a new node we are keeping
 148           // We do not need register_new_node_with_optimizer
 149           // because set_type has already been called.
 150           _igvn._worklist.push(x);
 151         }
 152       }
 153     }
 154     if (x != the_clone && the_clone != NULL)
 155       _igvn.remove_dead_node(the_clone);
 156     phi->set_req( i, x );
 157   }
 158   // Too few wins?
 159   if (wins <= policy) {
 160     _igvn.remove_dead_node(phi);
 161     return NULL;
 162   }
 163 
 164   // Record Phi
 165   register_new_node( phi, region );
 166 
 167   for (uint i2 = 1; i2 < phi->req(); i2++) {
 168     Node *x = phi->in(i2);
 169     // If we commoned up the cloned 'x' with another existing Node,
 170     // the existing Node picks up a new use.  We need to make the
 171     // existing Node occur higher up so it dominates its uses.
 172     Node *old_ctrl;
 173     IdealLoopTree *old_loop;
 174 
 175     if (x->is_Con()) {
 176       // Constant's control is always root.
 177       set_ctrl(x, C->root());
 178       continue;
 179     }
 180     // The occasional new node
 181     if (x->_idx >= old_unique) {     // Found a new, unplaced node?
 182       old_ctrl = NULL;
 183       old_loop = NULL;               // Not in any prior loop
 184     } else {
 185       old_ctrl = get_ctrl(x);
 186       old_loop = get_loop(old_ctrl); // Get prior loop
 187     }
 188     // New late point must dominate new use
 189     Node *new_ctrl = dom_lca(old_ctrl, region->in(i2));
 190     if (new_ctrl == old_ctrl) // Nothing is changed
 191       continue;
 192 
 193     IdealLoopTree *new_loop = get_loop(new_ctrl);
 194 
 195     // Don't move x into a loop if its uses are
 196     // outside of loop. Otherwise x will be cloned
 197     // for each use outside of this loop.
 198     IdealLoopTree *use_loop = get_loop(region);
 199     if (!new_loop->is_member(use_loop) &&
 200         (old_loop == NULL || !new_loop->is_member(old_loop))) {
 201       // Take early control, later control will be recalculated
 202       // during next iteration of loop optimizations.
 203       new_ctrl = get_early_ctrl(x);
 204       new_loop = get_loop(new_ctrl);
 205     }
 206     // Set new location
 207     set_ctrl(x, new_ctrl);
 208     // If changing loop bodies, see if we need to collect into new body
 209     if (old_loop != new_loop) {
 210       if (old_loop && !old_loop->_child)
 211         old_loop->_body.yank(x);
 212       if (!new_loop->_child)
 213         new_loop->_body.push(x);  // Collect body info
 214     }
 215   }
 216 
 217   return phi;
 218 }

 220 //------------------------------dominated_by------------------------------------
 221 // Replace the dominated test with an obvious true or false.  Place it on the
 222 // IGVN worklist for later cleanup.  Move control-dependent data Nodes on the
 223 // live path up to the dominating control.
 224 void PhaseIdealLoop::dominated_by( Node *prevdom, Node *iff, bool flip, bool exclude_loop_predicate ) {
 225   if (VerifyLoopOptimizations && PrintOpto) { tty->print_cr("dominating test"); }
 226 
 227   // prevdom is the dominating projection of the dominating test.
 228   assert( iff->is_If(), "" );
 229   assert(iff->Opcode() == Op_If || iff->Opcode() == Op_CountedLoopEnd || iff->Opcode() == Op_RangeCheck, "Check this code when new subtype is added");
 230   int pop = prevdom->Opcode();
 231   assert( pop == Op_IfFalse || pop == Op_IfTrue, "" );
 232   if (flip) {
 233     if (pop == Op_IfTrue)
 234       pop = Op_IfFalse;
 235     else
 236       pop = Op_IfTrue;
 237   }
 238   // 'con' is set to true or false to kill the dominated test.
 239   Node *con = _igvn.makecon(pop == Op_IfTrue ? TypeInt::ONE : TypeInt::ZERO);
 240   set_ctrl(con, C->root()); // Constant gets a new use
 241   // Hack the dominated test
 242   _igvn.replace_input_of(iff, 1, con);
 243 
 244   // If I dont have a reachable TRUE and FALSE path following the IfNode then
 245   // I can assume this path reaches an infinite loop.  In this case it's not
 246   // important to optimize the data Nodes - either the whole compilation will
 247   // be tossed or this path (and all data Nodes) will go dead.
 248   if (iff->outcnt() != 2) return;
 249 
 250   // Make control-dependent data Nodes on the live path (path that will remain
 251   // once the dominated IF is removed) become control-dependent on the
 252   // dominating projection.
 253   Node* dp = iff->as_If()->proj_out_or_null(pop == Op_IfTrue);
 254 
 255   // Loop predicates may have depending checks which should not
 256   // be skipped. For example, range check predicate has two checks
 257   // for lower and upper bounds.
 258   if (dp == NULL)
 259     return;
 260 
 261   ProjNode* dp_proj  = dp->as_Proj();
 262   ProjNode* unc_proj = iff->as_If()->proj_out(1 - dp_proj->_con)->as_Proj();
 263   if (exclude_loop_predicate &&
 264       (unc_proj->is_uncommon_trap_proj(Deoptimization::Reason_predicate) != NULL ||
 265        unc_proj->is_uncommon_trap_proj(Deoptimization::Reason_profile_predicate) != NULL ||
 266        unc_proj->is_uncommon_trap_proj(Deoptimization::Reason_range_check) != NULL)) {
 267     // If this is a range check (IfNode::is_range_check), do not
 268     // reorder because Compile::allow_range_check_smearing might have
 269     // changed the check.
 270     return; // Let IGVN transformation change control dependence.
 271   }
 272 
 273   IdealLoopTree *old_loop = get_loop(dp);
 274 
 275   for (DUIterator_Fast imax, i = dp->fast_outs(imax); i < imax; i++) {
 276     Node* cd = dp->fast_out(i); // Control-dependent node
 277     if (cd->depends_only_on_test()) {
 278       assert(cd->in(0) == dp, "");
 279       _igvn.replace_input_of(cd, 0, prevdom);
 280       set_early_ctrl(cd);
 281       IdealLoopTree *new_loop = get_loop(get_ctrl(cd));
 282       if (old_loop != new_loop) {
 283         if (!old_loop->_child) old_loop->_body.yank(cd);
 284         if (!new_loop->_child) new_loop->_body.push(cd);
 285       }
 286       --i;
 287       --imax;
 288     }
 289   }
 290 }

 292 //------------------------------has_local_phi_input----------------------------
 293 // Return TRUE if 'n' has Phi inputs from its local block and no other
 294 // block-local inputs (all non-local-phi inputs come from earlier blocks)
 295 Node *PhaseIdealLoop::has_local_phi_input( Node *n ) {
 296   Node *n_ctrl = get_ctrl(n);
 297   // See if some inputs come from a Phi in this block, or from before
 298   // this block.
 299   uint i;
 300   for( i = 1; i < n->req(); i++ ) {
 301     Node *phi = n->in(i);
 302     if( phi->is_Phi() && phi->in(0) == n_ctrl )
 303       break;
 304   }
 305   if( i >= n->req() )
 306     return NULL;                // No Phi inputs; nowhere to clone thru
 307 
 308   // Check for inputs created between 'n' and the Phi input.  These
 309   // must split as well; they have already been given the chance
 310   // (courtesy of a post-order visit) and since they did not we must
 311   // recover the 'cost' of splitting them by being very profitable
 312   // when splitting 'n'.  Since this is unlikely we simply give up.
 313   for( i = 1; i < n->req(); i++ ) {
 314     Node *m = n->in(i);
 315     if( get_ctrl(m) == n_ctrl && !m->is_Phi() ) {
 316       // We allow the special case of AddP's with no local inputs.
 317       // This allows us to split-up address expressions.
 318       if (m->is_AddP() &&
 319           get_ctrl(m->in(2)) != n_ctrl &&
 320           get_ctrl(m->in(3)) != n_ctrl) {
 321         // Move the AddP up to dominating point
 322         Node* c = find_non_split_ctrl(idom(n_ctrl));
 323         if (c->is_OuterStripMinedLoop()) {
 324           c->as_Loop()->verify_strip_mined(1);
 325           c = c->in(LoopNode::EntryControl);
 326         }
 327         set_ctrl_and_loop(m, c);
 328         continue;
 329       }
 330       return NULL;
 331     }
 332     assert(n->is_Phi() || m->is_Phi() || is_dominator(get_ctrl(m), n_ctrl), "m has strange control");
 333   }
 334 
 335   return n_ctrl;
 336 }

 338 //------------------------------remix_address_expressions----------------------
 339 // Rework addressing expressions to get the most loop-invariant stuff
 340 // moved out.  We'd like to do all associative operators, but it's especially
 341 // important (common) to do address expressions.
 342 Node *PhaseIdealLoop::remix_address_expressions( Node *n ) {
 343   if (!has_ctrl(n))  return NULL;
 344   Node *n_ctrl = get_ctrl(n);
 345   IdealLoopTree *n_loop = get_loop(n_ctrl);
 346 
 347   // See if 'n' mixes loop-varying and loop-invariant inputs and
 348   // itself is loop-varying.
 349 
 350   // Only interested in binary ops (and AddP)
 351   if( n->req() < 3 || n->req() > 4 ) return NULL;
 352 
 353   Node *n1_ctrl = get_ctrl(n->in(                    1));
 354   Node *n2_ctrl = get_ctrl(n->in(                    2));
 355   Node *n3_ctrl = get_ctrl(n->in(n->req() == 3 ? 2 : 3));
 356   IdealLoopTree *n1_loop = get_loop( n1_ctrl );
 357   IdealLoopTree *n2_loop = get_loop( n2_ctrl );
 358   IdealLoopTree *n3_loop = get_loop( n3_ctrl );
 359 
 360   // Does one of my inputs spin in a tighter loop than self?
 361   if( (n_loop->is_member( n1_loop ) && n_loop != n1_loop) ||
 362       (n_loop->is_member( n2_loop ) && n_loop != n2_loop) ||
 363       (n_loop->is_member( n3_loop ) && n_loop != n3_loop) )
 364     return NULL;                // Leave well enough alone
 365 
 366   // Is at least one of my inputs loop-invariant?
 367   if( n1_loop == n_loop &&
 368       n2_loop == n_loop &&
 369       n3_loop == n_loop )
 370     return NULL;                // No loop-invariant inputs
 371 
 372 
 373   int n_op = n->Opcode();
 374 
 375   // Replace expressions like ((V+I) << 2) with (V<<2 + I<<2).
 376   if( n_op == Op_LShiftI ) {
 377     // Scale is loop invariant
 378     Node *scale = n->in(2);
 379     Node *scale_ctrl = get_ctrl(scale);
 380     IdealLoopTree *scale_loop = get_loop(scale_ctrl );
 381     if( n_loop == scale_loop || !scale_loop->is_member( n_loop ) )
 382       return NULL;
 383     const TypeInt *scale_t = scale->bottom_type()->isa_int();
 384     if( scale_t && scale_t->is_con() && scale_t->get_con() >= 16 )
 385       return NULL;              // Dont bother with byte/short masking
 386     // Add must vary with loop (else shift would be loop-invariant)
 387     Node *add = n->in(1);
 388     Node *add_ctrl = get_ctrl(add);
 389     IdealLoopTree *add_loop = get_loop(add_ctrl);
 390     //assert( n_loop == add_loop, "" );
 391     if( n_loop != add_loop ) return NULL;  // happens w/ evil ZKM loops
 392 
 393     // Convert I-V into I+ (0-V); same for V-I
 394     if( add->Opcode() == Op_SubI &&
 395         _igvn.type( add->in(1) ) != TypeInt::ZERO ) {
 396       Node *zero = _igvn.intcon(0);
 397       set_ctrl(zero, C->root());
 398       Node *neg = new SubINode( _igvn.intcon(0), add->in(2) );
 399       register_new_node( neg, get_ctrl(add->in(2) ) );
 400       add = new AddINode( add->in(1), neg );
 401       register_new_node( add, add_ctrl );
 402     }
 403     if( add->Opcode() != Op_AddI ) return NULL;
 404     // See if one add input is loop invariant
 405     Node *add_var = add->in(1);
 406     Node *add_var_ctrl = get_ctrl(add_var);
 407     IdealLoopTree *add_var_loop = get_loop(add_var_ctrl );
 408     Node *add_invar = add->in(2);
 409     Node *add_invar_ctrl = get_ctrl(add_invar);
 410     IdealLoopTree *add_invar_loop = get_loop(add_invar_ctrl );
 411     if( add_var_loop == n_loop ) {
 412     } else if( add_invar_loop == n_loop ) {
 413       // Swap to find the invariant part
 414       add_invar = add_var;
 415       add_invar_ctrl = add_var_ctrl;
 416       add_invar_loop = add_var_loop;
 417       add_var = add->in(2);
 418       Node *add_var_ctrl = get_ctrl(add_var);
 419       IdealLoopTree *add_var_loop = get_loop(add_var_ctrl );
 420     } else                      // Else neither input is loop invariant
 421       return NULL;
 422     if( n_loop == add_invar_loop || !add_invar_loop->is_member( n_loop ) )
 423       return NULL;              // No invariant part of the add?
 424 
 425     // Yes!  Reshape address expression!
 426     Node *inv_scale = new LShiftINode( add_invar, scale );
 427     Node *inv_scale_ctrl =
 428       dom_depth(add_invar_ctrl) > dom_depth(scale_ctrl) ?
 429       add_invar_ctrl : scale_ctrl;
 430     register_new_node( inv_scale, inv_scale_ctrl );
 431     Node *var_scale = new LShiftINode( add_var, scale );
 432     register_new_node( var_scale, n_ctrl );
 433     Node *var_add = new AddINode( var_scale, inv_scale );
 434     register_new_node( var_add, n_ctrl );
 435     _igvn.replace_node( n, var_add );
 436     return var_add;
 437   }
 438 
 439   // Replace (I+V) with (V+I)
 440   if( n_op == Op_AddI ||
 441       n_op == Op_AddL ||
 442       n_op == Op_AddF ||
 443       n_op == Op_AddD ||
 444       n_op == Op_MulI ||
 445       n_op == Op_MulL ||
 446       n_op == Op_MulF ||
 447       n_op == Op_MulD ) {
 448     if( n2_loop == n_loop ) {
 449       assert( n1_loop != n_loop, "" );
 450       n->swap_edges(1, 2);
 451     }
 452   }
 453 
 454   // Replace ((I1 +p V) +p I2) with ((I1 +p I2) +p V),
 455   // but not if I2 is a constant.
 456   if( n_op == Op_AddP ) {
 457     if( n2_loop == n_loop && n3_loop != n_loop ) {
 458       if( n->in(2)->Opcode() == Op_AddP && !n->in(3)->is_Con() ) {
 459         Node *n22_ctrl = get_ctrl(n->in(2)->in(2));
 460         Node *n23_ctrl = get_ctrl(n->in(2)->in(3));
 461         IdealLoopTree *n22loop = get_loop( n22_ctrl );
 462         IdealLoopTree *n23_loop = get_loop( n23_ctrl );
 463         if( n22loop != n_loop && n22loop->is_member(n_loop) &&
 464             n23_loop == n_loop ) {
 465           Node *add1 = new AddPNode( n->in(1), n->in(2)->in(2), n->in(3) );
 466           // Stuff new AddP in the loop preheader
 467           register_new_node( add1, n_loop->_head->in(LoopNode::EntryControl) );
 468           Node *add2 = new AddPNode( n->in(1), add1, n->in(2)->in(3) );
 469           register_new_node( add2, n_ctrl );
 470           _igvn.replace_node( n, add2 );
 471           return add2;
 472         }
 473       }
 474     }
 475 
 476     // Replace (I1 +p (I2 + V)) with ((I1 +p I2) +p V)
 477     if (n2_loop != n_loop && n3_loop == n_loop) {
 478       if (n->in(3)->Opcode() == Op_AddX) {
 479         Node *V = n->in(3)->in(1);
 480         Node *I = n->in(3)->in(2);
 481         if (is_member(n_loop,get_ctrl(V))) {
 482         } else {
 483           Node *tmp = V; V = I; I = tmp;
 484         }
 485         if (!is_member(n_loop,get_ctrl(I))) {
 486           Node *add1 = new AddPNode(n->in(1), n->in(2), I);
 487           // Stuff new AddP in the loop preheader
 488           register_new_node(add1, n_loop->_head->in(LoopNode::EntryControl));
 489           Node *add2 = new AddPNode(n->in(1), add1, V);
 490           register_new_node(add2, n_ctrl);
 491           _igvn.replace_node(n, add2);
 492           return add2;
 493         }
 494       }
 495     }
 496   }
 497 
 498   return NULL;
 499 }

 501 // Optimize ((in1[2*i] * in2[2*i]) + (in1[2*i+1] * in2[2*i+1]))
 502 Node *PhaseIdealLoop::convert_add_to_muladd(Node* n) {
 503   assert(n->Opcode() == Op_AddI, "sanity");
 504   Node * nn = NULL;
 505   Node * in1 = n->in(1);
 506   Node * in2 = n->in(2);
 507   if (in1->Opcode() == Op_MulI && in2->Opcode() == Op_MulI) {
 508     IdealLoopTree* loop_n = get_loop(get_ctrl(n));
 509     if (loop_n->is_counted() &&
 510         loop_n->_head->as_Loop()->is_valid_counted_loop() &&
 511         Matcher::match_rule_supported(Op_MulAddVS2VI) &&
 512         Matcher::match_rule_supported(Op_MulAddS2I)) {
 513       Node* mul_in1 = in1->in(1);
 514       Node* mul_in2 = in1->in(2);
 515       Node* mul_in3 = in2->in(1);
 516       Node* mul_in4 = in2->in(2);
 517       if (mul_in1->Opcode() == Op_LoadS &&
 518           mul_in2->Opcode() == Op_LoadS &&
 519           mul_in3->Opcode() == Op_LoadS &&
 520           mul_in4->Opcode() == Op_LoadS) {
 521         IdealLoopTree* loop1 = get_loop(get_ctrl(mul_in1));
 522         IdealLoopTree* loop2 = get_loop(get_ctrl(mul_in2));
 523         IdealLoopTree* loop3 = get_loop(get_ctrl(mul_in3));
 524         IdealLoopTree* loop4 = get_loop(get_ctrl(mul_in4));
 525         IdealLoopTree* loop5 = get_loop(get_ctrl(in1));
 526         IdealLoopTree* loop6 = get_loop(get_ctrl(in2));
 527         // All nodes should be in the same counted loop.
 528         if (loop_n == loop1 && loop_n == loop2 && loop_n == loop3 &&
 529             loop_n == loop4 && loop_n == loop5 && loop_n == loop6) {
 530           Node* adr1 = mul_in1->in(MemNode::Address);
 531           Node* adr2 = mul_in2->in(MemNode::Address);
 532           Node* adr3 = mul_in3->in(MemNode::Address);
 533           Node* adr4 = mul_in4->in(MemNode::Address);
 534           if (adr1->is_AddP() && adr2->is_AddP() && adr3->is_AddP() && adr4->is_AddP()) {
 535             if ((adr1->in(AddPNode::Base) == adr3->in(AddPNode::Base)) &&
 536                 (adr2->in(AddPNode::Base) == adr4->in(AddPNode::Base))) {
 537               nn = new MulAddS2INode(mul_in1, mul_in2, mul_in3, mul_in4);
 538               register_new_node(nn, get_ctrl(n));
 539               _igvn.replace_node(n, nn);
 540               return nn;
 541             } else if ((adr1->in(AddPNode::Base) == adr4->in(AddPNode::Base)) &&
 542                        (adr2->in(AddPNode::Base) == adr3->in(AddPNode::Base))) {
 543               nn = new MulAddS2INode(mul_in1, mul_in2, mul_in4, mul_in3);
 544               register_new_node(nn, get_ctrl(n));
 545               _igvn.replace_node(n, nn);
 546               return nn;
 547             }
 548           }
 549         }
 550       }
 551     }
 552   }
 553   return nn;

 556 //------------------------------conditional_move-------------------------------
 557 // Attempt to replace a Phi with a conditional move.  We have some pretty
 558 // strict profitability requirements.  All Phis at the merge point must
 559 // be converted, so we can remove the control flow.  We need to limit the
 560 // number of c-moves to a small handful.  All code that was in the side-arms
 561 // of the CFG diamond is now speculatively executed.  This code has to be
 562 // "cheap enough".  We are pretty much limited to CFG diamonds that merge
 563 // 1 or 2 items with a total of 1 or 2 ops executed speculatively.
 564 Node *PhaseIdealLoop::conditional_move( Node *region ) {
 565 
 566   assert(region->is_Region(), "sanity check");
 567   if (region->req() != 3) return NULL;
 568 
 569   // Check for CFG diamond
 570   Node *lp = region->in(1);
 571   Node *rp = region->in(2);
 572   if (!lp || !rp) return NULL;
 573   Node *lp_c = lp->in(0);
 574   if (lp_c == NULL || lp_c != rp->in(0) || !lp_c->is_If()) return NULL;
 575   IfNode *iff = lp_c->as_If();
 576 
 577   // Check for ops pinned in an arm of the diamond.
 578   // Can't remove the control flow in this case
 579   if (lp->outcnt() > 1) return NULL;
 580   if (rp->outcnt() > 1) return NULL;
 581 
 582   IdealLoopTree* r_loop = get_loop(region);
 583   assert(r_loop == get_loop(iff), "sanity");
 584   // Always convert to CMOVE if all results are used only outside this loop.
 585   bool used_inside_loop = (r_loop == _ltree_root);
 586 
 587   // Check profitability
 588   int cost = 0;
 589   int phis = 0;
 590   for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
 591     Node *out = region->fast_out(i);
 592     if (!out->is_Phi()) continue; // Ignore other control edges, etc
 593     phis++;
 594     PhiNode* phi = out->as_Phi();
 595     BasicType bt = phi->type()->basic_type();
 596     switch (bt) {
 597     case T_DOUBLE:
 598     case T_FLOAT:
 599       if (C->use_cmove()) {
 600         continue; //TODO: maybe we want to add some cost
 601       }
 602       cost += Matcher::float_cmove_cost(); // Could be very expensive
 603       break;
 604     case T_LONG: {
 605       cost += Matcher::long_cmove_cost(); // May encodes as 2 CMOV's
 606     }
 607     case T_INT:                 // These all CMOV fine
 608     case T_ADDRESS: {           // (RawPtr)
 609       cost++;
 610       break;
 611     }
 612     case T_NARROWOOP: // Fall through
 613     case T_OBJECT: {            // Base oops are OK, but not derived oops
 614       const TypeOopPtr *tp = phi->type()->make_ptr()->isa_oopptr();
 615       // Derived pointers are Bad (tm): what's the Base (for GC purposes) of a
 616       // CMOVE'd derived pointer?  It's a CMOVE'd derived base.  Thus
 617       // CMOVE'ing a derived pointer requires we also CMOVE the base.  If we
 618       // have a Phi for the base here that we convert to a CMOVE all is well
 619       // and good.  But if the base is dead, we'll not make a CMOVE.  Later
 620       // the allocator will have to produce a base by creating a CMOVE of the
 621       // relevant bases.  This puts the allocator in the business of
 622       // manufacturing expensive instructions, generally a bad plan.
 623       // Just Say No to Conditionally-Moved Derived Pointers.
 624       if (tp && tp->offset() != 0)
 625         return NULL;
 626       cost++;
 627       break;
 628     }
 629     default:
 630       return NULL;              // In particular, can't do memory or I/O
 631     }
 632     // Add in cost any speculative ops
 633     for (uint j = 1; j < region->req(); j++) {
 634       Node *proj = region->in(j);
 635       Node *inp = phi->in(j);
 636       if (get_ctrl(inp) == proj) { // Found local op
 637         cost++;
 638         // Check for a chain of dependent ops; these will all become
 639         // speculative in a CMOV.
 640         for (uint k = 1; k < inp->req(); k++)
 641           if (get_ctrl(inp->in(k)) == proj)
 642             cost += ConditionalMoveLimit; // Too much speculative goo
 643       }
 644     }
 645     // See if the Phi is used by a Cmp or Narrow oop Decode/Encode.
 646     // This will likely Split-If, a higher-payoff operation.
 647     for (DUIterator_Fast kmax, k = phi->fast_outs(kmax); k < kmax; k++) {
 648       Node* use = phi->fast_out(k);
 649       if (use->is_Cmp() || use->is_DecodeNarrowPtr() || use->is_EncodeNarrowPtr())
 650         cost += ConditionalMoveLimit;
 651       // Is there a use inside the loop?
 652       // Note: check only basic types since CMoveP is pinned.
 653       if (!used_inside_loop && is_java_primitive(bt)) {
 654         IdealLoopTree* u_loop = get_loop(has_ctrl(use) ? get_ctrl(use) : use);
 655         if (r_loop == u_loop || r_loop->is_member(u_loop)) {
 656           used_inside_loop = true;
 657         }
 658       }
 659     }
 660   }//for
 661   Node* bol = iff->in(1);
 662   if (bol->Opcode() == Op_Opaque4) {
 663     return NULL; // Ignore loop predicate checks (the Opaque4 ensures they will go away)
 664   }
 665   assert(bol->Opcode() == Op_Bool, "Unexpected node");
 666   int cmp_op = bol->in(1)->Opcode();
 667   if (cmp_op == Op_SubTypeCheck) { // SubTypeCheck expansion expects an IfNode
 668     return NULL;
 669   }
 670   // It is expensive to generate flags from a float compare.
 671   // Avoid duplicated float compare.
 672   if (phis > 1 && (cmp_op == Op_CmpF || cmp_op == Op_CmpD)) return NULL;
 673 
 674   float infrequent_prob = PROB_UNLIKELY_MAG(3);
 675   // Ignore cost and blocks frequency if CMOVE can be moved outside the loop.
 676   if (used_inside_loop) {
 677     if (cost >= ConditionalMoveLimit) return NULL; // Too much goo
 678 
 679     // BlockLayoutByFrequency optimization moves infrequent branch
 680     // from hot path. No point in CMOV'ing in such case (110 is used
 681     // instead of 100 to take into account not exactness of float value).
 682     if (BlockLayoutByFrequency) {
 683       infrequent_prob = MAX2(infrequent_prob, (float)BlockLayoutMinDiamondPercentage/110.0f);
 684     }
 685   }
 686   // Check for highly predictable branch.  No point in CMOV'ing if
 687   // we are going to predict accurately all the time.
 688   if (C->use_cmove() && (cmp_op == Op_CmpF || cmp_op == Op_CmpD)) {
 689     //keep going
 690   } else if (iff->_prob < infrequent_prob ||
 691       iff->_prob > (1.0f - infrequent_prob))
 692     return NULL;
 693 
 694   // --------------
 695   // Now replace all Phis with CMOV's
 696   Node *cmov_ctrl = iff->in(0);
 697   uint flip = (lp->Opcode() == Op_IfTrue);
 698   Node_List wq;
 699   while (1) {
 700     PhiNode* phi = NULL;
 701     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
 702       Node *out = region->fast_out(i);
 703       if (out->is_Phi()) {
 704         phi = out->as_Phi();
 705         break;
 706       }
 707     }
 708     if (phi == NULL)  break;
 709     if (PrintOpto && VerifyLoopOptimizations) { tty->print_cr("CMOV"); }
 710     // Move speculative ops
 711     wq.push(phi);
 712     while (wq.size() > 0) {
 713       Node *n = wq.pop();
 714       for (uint j = 1; j < n->req(); j++) {
 715         Node* m = n->in(j);
 716         if (m != NULL && !is_dominator(get_ctrl(m), cmov_ctrl)) {
 717 #ifndef PRODUCT
 718           if (PrintOpto && VerifyLoopOptimizations) {
 719             tty->print("  speculate: ");
 720             m->dump();
 721           }
 722 #endif
 723           set_ctrl(m, cmov_ctrl);
 724           wq.push(m);
 725         }
 726       }
 727     }
 728     Node *cmov = CMoveNode::make(cmov_ctrl, iff->in(1), phi->in(1+flip), phi->in(2-flip), _igvn.type(phi));
 729     register_new_node( cmov, cmov_ctrl );
 730     _igvn.replace_node( phi, cmov );
 731 #ifndef PRODUCT
 732     if (TraceLoopOpts) {
 733       tty->print("CMOV  ");
 734       r_loop->dump_head();
 735       if (Verbose) {
 736         bol->in(1)->dump(1);
 737         cmov->dump(1);
 738       }
 739     }
 740     if (VerifyLoopOptimizations) verify();
 741 #endif
 742   }
 743 
 744   // The useless CFG diamond will fold up later; see the optimization in
 745   // RegionNode::Ideal.
 746   _igvn._worklist.push(region);
 747 
 748   return iff->in(1);
 749 }

 757         enqueue_cfg_uses(u, wq);
 758       } else {
 759         wq.push(u);
 760       }
 761     }

 765 // Try moving a store out of a loop, right before the loop
 766 Node* PhaseIdealLoop::try_move_store_before_loop(Node* n, Node *n_ctrl) {
 767   // Store has to be first in the loop body
 768   IdealLoopTree *n_loop = get_loop(n_ctrl);
 769   if (n->is_Store() && n_loop != _ltree_root &&
 770       n_loop->is_loop() && n_loop->_head->is_Loop() &&
 771       n->in(0) != NULL) {
 772     Node* address = n->in(MemNode::Address);
 773     Node* value = n->in(MemNode::ValueIn);
 774     Node* mem = n->in(MemNode::Memory);
 775     IdealLoopTree* address_loop = get_loop(get_ctrl(address));
 776     IdealLoopTree* value_loop = get_loop(get_ctrl(value));
 777 
 778     // - address and value must be loop invariant
 779     // - memory must be a memory Phi for the loop
 780     // - Store must be the only store on this memory slice in the
 781     // loop: if there's another store following this one then value
 782     // written at iteration i by the second store could be overwritten
 783     // at iteration i+n by the first store: it's not safe to move the
 784     // first store out of the loop
 785     // - nothing must observe the memory Phi: it guarantees no read
 786     // before the store, we are also guaranteed the store post
 787     // dominates the loop head (ignoring a possible early
 788     // exit). Otherwise there would be extra Phi involved between the
 789     // loop's Phi and the store.
 790     // - there must be no early exit from the loop before the Store
 791     // (such an exit most of the time would be an extra use of the
 792     // memory Phi but sometimes is a bottom memory Phi that takes the
 793     // store as input).
 794 
 795     if (!n_loop->is_member(address_loop) &&
 796         !n_loop->is_member(value_loop) &&
 797         mem->is_Phi() && mem->in(0) == n_loop->_head &&
 798         mem->outcnt() == 1 &&
 799         mem->in(LoopNode::LoopBackControl) == n) {
 800 
 801       assert(n_loop->_tail != NULL, "need a tail");
 802       assert(is_dominator(n_ctrl, n_loop->_tail), "store control must not be in a branch in the loop");
 803 
 804       // Verify that there's no early exit of the loop before the store.
 805       bool ctrl_ok = false;
 806       {
 807         // Follow control from loop head until n, we exit the loop or
 808         // we reach the tail
 809         ResourceMark rm;
 810         Unique_Node_List wq;
 811         wq.push(n_loop->_head);
 812 
 813         for (uint next = 0; next < wq.size(); ++next) {
 814           Node *m = wq.at(next);
 815           if (m == n->in(0)) {
 816             ctrl_ok = true;
 817             continue;
 818           }
 819           assert(!has_ctrl(m), "should be CFG");
 820           if (!n_loop->is_member(get_loop(m)) || m == n_loop->_tail) {
 821             ctrl_ok = false;
 822             break;
 823           }
 824           enqueue_cfg_uses(m, wq);
 825           if (wq.size() > 10) {
 826             ctrl_ok = false;
 827             break;
 828           }
 829         }
 830       }
 831       if (ctrl_ok) {
 832         // move the Store
 833         _igvn.replace_input_of(mem, LoopNode::LoopBackControl, mem);
 834         _igvn.replace_input_of(n, 0, n_loop->_head->as_Loop()->skip_strip_mined()->in(LoopNode::EntryControl));
 835         _igvn.replace_input_of(n, MemNode::Memory, mem->in(LoopNode::EntryControl));
 836         // Disconnect the phi now. An empty phi can confuse other
 837         // optimizations in this pass of loop opts.
 838         _igvn.replace_node(mem, mem->in(LoopNode::EntryControl));
 839         n_loop->_body.yank(mem);
 840 
 841         set_ctrl_and_loop(n, n->in(0));
 842 
 843         return n;
 844       }
 845     }
 846   }
 847   return NULL;
 848 }

 850 // Try moving a store out of a loop, right after the loop
 851 void PhaseIdealLoop::try_move_store_after_loop(Node* n) {
 852   if (n->is_Store() && n->in(0) != NULL) {
 853     Node *n_ctrl = get_ctrl(n);
 854     IdealLoopTree *n_loop = get_loop(n_ctrl);
 855     // Store must be in a loop
 856     if (n_loop != _ltree_root && !n_loop->_irreducible) {
 857       Node* address = n->in(MemNode::Address);
 858       Node* value = n->in(MemNode::ValueIn);
 859       IdealLoopTree* address_loop = get_loop(get_ctrl(address));
 860       // address must be loop invariant
 861       if (!n_loop->is_member(address_loop)) {
 862         // Store must be last on this memory slice in the loop and
 863         // nothing in the loop must observe it
 864         Node* phi = NULL;
 865         for (DUIterator_Fast imax, i = n->fast_outs(imax); i < imax; i++) {
 866           Node* u = n->fast_out(i);
 867           if (has_ctrl(u)) { // control use?
 868             IdealLoopTree *u_loop = get_loop(get_ctrl(u));
 869             if (!n_loop->is_member(u_loop)) {
 870               continue;
 871             }
 872             if (u->is_Phi() && u->in(0) == n_loop->_head) {
 873               assert(_igvn.type(u) == Type::MEMORY, "bad phi");
 874               // multiple phis on the same slice are possible
 875               if (phi != NULL) {
 876                 return;
 877               }
 878               phi = u;
 879               continue;
 880             }
 881           }
 882           return;
 883         }
 884         if (phi != NULL) {
 885           // Nothing in the loop before the store (next iteration)
 886           // must observe the stored value
 887           bool mem_ok = true;
 888           {
 889             ResourceMark rm;
 890             Unique_Node_List wq;
 891             wq.push(phi);
 892             for (uint next = 0; next < wq.size() && mem_ok; ++next) {
 893               Node *m = wq.at(next);
 894               for (DUIterator_Fast imax, i = m->fast_outs(imax); i < imax && mem_ok; i++) {
 895                 Node* u = m->fast_out(i);
 896                 if (u->is_Store() || u->is_Phi()) {
 897                   if (u != n) {
 898                     wq.push(u);
 899                     mem_ok = (wq.size() <= 10);
 900                   }
 901                 } else {
 902                   mem_ok = false;
 903                   break;
 904                 }
 905               }
 906             }
 907           }
 908           if (mem_ok) {
 909             // Move the store out of the loop if the LCA of all
 910             // users (except for the phi) is outside the loop.
 911             Node* hook = new Node(1);
 912             hook->init_req(0, n_ctrl); // Add an input to prevent hook from being dead
 913             _igvn.rehash_node_delayed(phi);
 914             int count = phi->replace_edge(n, hook);
 915             assert(count > 0, "inconsistent phi");
 916 
 917             // Compute latest point this store can go
 918             Node* lca = get_late_ctrl(n, get_ctrl(n));
 919             if (n_loop->is_member(get_loop(lca))) {
 920               // LCA is in the loop - bail out
 921               _igvn.replace_node(hook, n);
 922               return;
 923             }
 924 #ifdef ASSERT
 925             if (n_loop->_head->is_Loop() && n_loop->_head->as_Loop()->is_strip_mined()) {
 926               assert(n_loop->_head->Opcode() == Op_CountedLoop, "outer loop is a strip mined");
 927               n_loop->_head->as_Loop()->verify_strip_mined(1);
 928               Node* outer = n_loop->_head->as_CountedLoop()->outer_loop();
 929               IdealLoopTree* outer_loop = get_loop(outer);
 930               assert(n_loop->_parent == outer_loop, "broken loop tree");
 931               assert(get_loop(lca) == outer_loop, "safepoint in outer loop consume all memory state");
 932             }
 933 #endif
 934 
 935             // Move store out of the loop
 936             _igvn.replace_node(hook, n->in(MemNode::Memory));
 937             _igvn.replace_input_of(n, 0, lca);
 938             set_ctrl_and_loop(n, lca);
 939 
 940             // Disconnect the phi now. An empty phi can confuse other
 941             // optimizations in this pass of loop opts..
 942             if (phi->in(LoopNode::LoopBackControl) == phi) {
 943               _igvn.replace_node(phi, phi->in(LoopNode::EntryControl));
 944               n_loop->_body.yank(phi);
 945             }
 946           }
 947         }
 948       }
 949     }
 950   }
 951 }

 953 //------------------------------split_if_with_blocks_pre-----------------------
 954 // Do the real work in a non-recursive function.  Data nodes want to be
 955 // cloned in the pre-order so they can feed each other nicely.
 956 Node *PhaseIdealLoop::split_if_with_blocks_pre( Node *n ) {
 957   // Cloning these guys is unlikely to win
 958   int n_op = n->Opcode();
 959   if( n_op == Op_MergeMem ) return n;
 960   if( n->is_Proj() ) return n;
 961   // Do not clone-up CmpFXXX variations, as these are always
 962   // followed by a CmpI
 963   if( n->is_Cmp() ) return n;
 964   // Attempt to use a conditional move instead of a phi/branch
 965   if( ConditionalMoveLimit > 0 && n_op == Op_Region ) {
 966     Node *cmov = conditional_move( n );
 967     if( cmov ) return cmov;
 968   }
 969   if( n->is_CFG() || n->is_LoadStore() )
 970     return n;
 971   if( n_op == Op_Opaque1 ||     // Opaque nodes cannot be mod'd
 972       n_op == Op_Opaque2 ) {
 973     if( !C->major_progress() )   // If chance of no more loop opts...
 974       _igvn._worklist.push(n);  // maybe we'll remove them
 975     return n;
 976   }
 977 
 978   if( n->is_Con() ) return n;   // No cloning for Con nodes
 979 
 980   Node *n_ctrl = get_ctrl(n);
 981   if( !n_ctrl ) return n;       // Dead node
 982 
 983   Node* res = try_move_store_before_loop(n, n_ctrl);
 984   if (res != NULL) {
 985     return n;
 986   }
 987 
 988   // Attempt to remix address expressions for loop invariants
 989   Node *m = remix_address_expressions( n );
 990   if( m ) return m;
 991 
 992   if (n_op == Op_AddI) {
 993     Node *nn = convert_add_to_muladd( n );
 994     if ( nn ) return nn;
 995   }
 996 
 997   if (n->is_ConstraintCast()) {
 998     Node* dom_cast = n->as_ConstraintCast()->dominating_cast(&_igvn, this);
 999     // ConstraintCastNode::dominating_cast() uses node control input to determine domination.
1000     // Node control inputs don't necessarily agree with loop control info (due to
1001     // transformations happened in between), thus additional dominance check is needed
1002     // to keep loop info valid.
1003     if (dom_cast != NULL && is_dominator(get_ctrl(dom_cast), get_ctrl(n))) {
1004       _igvn.replace_node(n, dom_cast);
1005       return dom_cast;
1006     }
1007   }
1008 
1009   // Determine if the Node has inputs from some local Phi.
1010   // Returns the block to clone thru.
1011   Node *n_blk = has_local_phi_input( n );
1012   if( !n_blk ) return n;
1013 
1014   // Do not clone the trip counter through on a CountedLoop
1015   // (messes up the canonical shape).
1016   if( n_blk->is_CountedLoop() && n->Opcode() == Op_AddI ) return n;
1017 
1018   // Check for having no control input; not pinned.  Allow
1019   // dominating control.
1020   if (n->in(0)) {
1021     Node *dom = idom(n_blk);
1022     if (dom_lca(n->in(0), dom) != n->in(0)) {
1023       return n;
1024     }
1025   }
1026   // Policy: when is it profitable.  You must get more wins than
1027   // policy before it is considered profitable.  Policy is usually 0,
1028   // so 1 win is considered profitable.  Big merges will require big
1029   // cloning, so get a larger policy.
1030   int policy = n_blk->req() >> 2;
1031 
1032   // If the loop is a candidate for range check elimination,
1033   // delay splitting through it's phi until a later loop optimization
1034   if (n_blk->is_CountedLoop()) {
1035     IdealLoopTree *lp = get_loop(n_blk);
1036     if (lp && lp->_rce_candidate) {
1037       return n;
1038     }
1039   }
1040 
1041   if (must_throttle_split_if()) return n;
1042 
1043   // Split 'n' through the merge point if it is profitable
1044   Node *phi = split_thru_phi( n, n_blk, policy );
1045   if (!phi) return n;
1046 
1047   // Found a Phi to split thru!
1048   // Replace 'n' with the new phi
1049   _igvn.replace_node( n, phi );
1050   // Moved a load around the loop, 'en-registering' something.
1051   if (n_blk->is_Loop() && n->is_Load() &&
1052       !phi->in(LoopNode::LoopBackControl)->is_Load())
1053     C->set_major_progress();
1054 
1055   return phi;
1056 }

1058 static bool merge_point_too_heavy(Compile* C, Node* region) {
1059   // Bail out if the region and its phis have too many users.
1060   int weight = 0;
1061   for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1062     weight += region->fast_out(i)->outcnt();
1063   }
1064   int nodes_left = C->max_node_limit() - C->live_nodes();
1065   if (weight * 8 > nodes_left) {
1066     if (PrintOpto) {
1067       tty->print_cr("*** Split-if bails out:  %d nodes, region weight %d", C->unique(), weight);
1068     }
1069     return true;
1070   } else {
1071     return false;
1072   }
1073 }

1077   // having a PhiNode input. This sidesteps the dangerous case where the split

1081   // A better fix for this problem can be found in the BugTraq entry, but

1084   for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {

1092           return false;
1093         }
1094       }
1095     }

1097 #endif

1150 bool PhaseIdealLoop::can_split_if(Node* n_ctrl) {
1151   if (must_throttle_split_if()) {
1152     return false;
1153   }
1154 
1155   // Do not do 'split-if' if irreducible loops are present.
1156   if (_has_irreducible_loops) {
1157     return false;
1158   }
1159 
1160   if (merge_point_too_heavy(C, n_ctrl)) {
1161     return false;
1162   }
1163 
1164   // Do not do 'split-if' if some paths are dead.  First do dead code
1165   // elimination and then see if its still profitable.
1166   for (uint i = 1; i < n_ctrl->req(); i++) {
1167     if (n_ctrl->in(i) == C->top()) {
1168       return false;
1169     }
1170   }
1171 
1172   // If trying to do a 'Split-If' at the loop head, it is only
1173   // profitable if the cmp folds up on BOTH paths.  Otherwise we
1174   // risk peeling a loop forever.
1175 
1176   // CNC - Disabled for now.  Requires careful handling of loop
1177   // body selection for the cloned code.  Also, make sure we check
1178   // for any input path not being in the same loop as n_ctrl.  For
1179   // irreducible loops we cannot check for 'n_ctrl->is_Loop()'
1180   // because the alternative loop entry points won't be converted
1181   // into LoopNodes.
1182   IdealLoopTree *n_loop = get_loop(n_ctrl);
1183   for (uint j = 1; j < n_ctrl->req(); j++) {
1184     if (get_loop(n_ctrl->in(j)) != n_loop) {
1185       return false;
1186     }
1187   }
1188 
1189   // Check for safety of the merge point.
1190   if (!merge_point_safe(n_ctrl)) {
1191     return false;
1192   }
1193 
1194   return true;
1195 }

1213 bool PhaseIdealLoop::flatten_array_element_type_check(Node *n) {
1214   // If the CmpP is a subtype check for a value that has just been
1215   // loaded from an array, the subtype checks guarantees the value
1216   // can't be stored in a flattened array and the load of the value
1217   // happens with a flattened array check then: push the type check
1218   // through the phi of the flattened array check. This needs special
1219   // logic because the subtype check's input is not a phi but a
1220   // LoadKlass that must first be cloned through the phi.
1221   if (n->Opcode() != Op_CmpP) {
1222     return false;
1223   }
1224 
1225   Node* klassptr = n->in(1);
1226   Node* klasscon = n->in(2);
1227 
1228   if (klassptr->is_DecodeNarrowPtr()) {
1229     klassptr = klassptr->in(1);
1230   }
1231 
1232   if (klassptr->Opcode() != Op_LoadKlass && klassptr->Opcode() != Op_LoadNKlass) {
1233     return false;
1234   }
1235 
1236   if (!klasscon->is_Con()) {
1237     return false;
1238   }
1239 
1240   Node* addr = klassptr->in(MemNode::Address);
1241 
1242   if (!addr->is_AddP()) {
1243     return false;
1244   }
1245 
1246   intptr_t offset;
1247   Node* obj = AddPNode::Ideal_base_and_offset(addr, &_igvn, offset);
1248 
1249   if (obj == NULL) {
1250     return false;
1251   }
1252 
1253   assert(obj != NULL && addr->in(AddPNode::Base) == addr->in(AddPNode::Address), "malformed AddP?");
1254   if (obj->Opcode() == Op_CastPP) {
1255     obj = obj->in(1);
1256   }
1257 
1258   if (!obj->is_Phi()) {
1259     return false;
1260   }
1261 
1262   Node* region = obj->in(0);
1263 
1264   Node* phi = PhiNode::make_blank(region, n->in(1));
1265   for (uint i = 1; i < region->req(); i++) {
1266     Node* in = obj->in(i);
1267     Node* ctrl = get_ctrl(in);
1268     if (addr->in(AddPNode::Base) != obj) {
1269       Node* cast = addr->in(AddPNode::Base);
1270       assert(cast->Opcode() == Op_CastPP && cast->in(0) != NULL, "inconsistent subgraph");
1271       Node* cast_clone = cast->clone();
1272       cast_clone->set_req(0, region->in(i));
1273       cast_clone->set_req(1, in);
1274       register_new_node(cast_clone, region->in(i));
1275       _igvn.set_type(cast_clone, cast_clone->Value(&_igvn));
1276       in = cast_clone;
1277     }
1278     Node* addr_clone = addr->clone();
1279     addr_clone->set_req(AddPNode::Base, in);
1280     addr_clone->set_req(AddPNode::Address, in);
1281     register_new_node(addr_clone, ctrl);
1282     _igvn.set_type(addr_clone, addr_clone->Value(&_igvn));
1283     Node* klassptr_clone = klassptr->clone();
1284     klassptr_clone->set_req(2, addr_clone);
1285     register_new_node(klassptr_clone, ctrl);
1286     _igvn.set_type(klassptr_clone, klassptr_clone->Value(&_igvn));
1287     if (klassptr != n->in(1)) {
1288       Node* decode = n->in(1);
1289       assert(decode->is_DecodeNarrowPtr(), "inconcistent subgraph");
1290       Node* decode_clone = decode->clone();
1291       decode_clone->set_req(1, klassptr_clone);
1292       register_new_node(decode_clone, ctrl);
1293       _igvn.set_type(decode_clone, decode_clone->Value(&_igvn));
1294       klassptr_clone = decode_clone;
1295     }
1296     phi->set_req(i, klassptr_clone);
1297   }
1298   register_new_node(phi, region);
1299   Node* orig = n->in(1);
1300   _igvn.replace_input_of(n, 1, phi);
1301   split_if_with_blocks_post(n);
1302   if (n->outcnt() != 0) {
1303     _igvn.replace_input_of(n, 1, orig);
1304     _igvn.remove_dead_node(phi);
1305   }
1306   return true;
1307 }

1309 //------------------------------split_if_with_blocks_post----------------------
1310 // Do the real work in a non-recursive function.  CFG hackery wants to be
1311 // in the post-order, so it can dirty the I-DOM info and not use the dirtied
1312 // info.
1313 void PhaseIdealLoop::split_if_with_blocks_post(Node *n) {
1314 
1315   if (flatten_array_element_type_check(n)) {
1316     return;
1317   }
1318 
1319   // Cloning Cmp through Phi's involves the split-if transform.
1320   // FastLock is not used by an If
1321   if (n->is_Cmp() && !n->is_FastLock()) {
1322     Node *n_ctrl = get_ctrl(n);
1323     // Determine if the Node has inputs from some local Phi.
1324     // Returns the block to clone thru.
1325     Node *n_blk = has_local_phi_input(n);
1326     if (n_blk != n_ctrl) {
1327       return;
1328     }
1329 
1330     if (!can_split_if(n_ctrl)) {
1331       return;
1332     }
1333 
1334     if (n->outcnt() != 1) {
1335       return; // Multiple bool's from 1 compare?
1336     }
1337     Node *bol = n->unique_out();
1338     assert(bol->is_Bool(), "expect a bool here");
1339     if (bol->outcnt() != 1) {
1340       return;// Multiple branches from 1 compare?
1341     }
1342     Node *iff = bol->unique_out();
1343 
1344     // Check some safety conditions
1345     if (iff->is_If()) {        // Classic split-if?
1346       if (iff->in(0) != n_ctrl) {
1347         return; // Compare must be in same blk as if
1348       }
1349     } else if (iff->is_CMove()) { // Trying to split-up a CMOVE
1350       // Can't split CMove with different control edge.
1351       if (iff->in(0) != NULL && iff->in(0) != n_ctrl ) {
1352         return;
1353       }
1354       if (get_ctrl(iff->in(2)) == n_ctrl ||
1355           get_ctrl(iff->in(3)) == n_ctrl) {
1356         return;                 // Inputs not yet split-up
1357       }
1358       if (get_loop(n_ctrl) != get_loop(get_ctrl(iff))) {
1359         return;                 // Loop-invar test gates loop-varying CMOVE
1360       }
1361     } else {
1362       return;  // some other kind of node, such as an Allocate
1363     }
1364 
1365     // When is split-if profitable?  Every 'win' on means some control flow
1366     // goes dead, so it's almost always a win.
1367     int policy = 0;
1368     // Split compare 'n' through the merge point if it is profitable
1369     Node *phi = split_thru_phi( n, n_ctrl, policy);
1370     if (!phi) {
1371       return;
1372     }
1373 
1374     // Found a Phi to split thru!
1375     // Replace 'n' with the new phi
1376     _igvn.replace_node(n, phi);
1377 
1378     // Now split the bool up thru the phi
1379     Node *bolphi = split_thru_phi(bol, n_ctrl, -1);
1380     guarantee(bolphi != NULL, "null boolean phi node");
1381 
1382     _igvn.replace_node(bol, bolphi);
1383     assert(iff->in(1) == bolphi, "");
1384 
1385     if (bolphi->Value(&_igvn)->singleton()) {
1386       return;
1387     }
1388 
1389     // Conditional-move?  Must split up now
1390     if (!iff->is_If()) {
1391       Node *cmovphi = split_thru_phi(iff, n_ctrl, -1);
1392       _igvn.replace_node(iff, cmovphi);
1393       return;
1394     }
1395 
1396     // Now split the IF
1397     do_split_if(iff);
1398     return;
1399   }
1400 
1401   // Two identical ifs back to back can be merged
1402   if (identical_backtoback_ifs(n) && can_split_if(n->in(0))) {
1403     Node *n_ctrl = n->in(0);
1404     PhiNode* bolphi = PhiNode::make_blank(n_ctrl, n->in(1));
1405     IfNode* dom_if = idom(n_ctrl)->as_If();
1406     Node* proj_true = dom_if->proj_out(1);
1407     Node* proj_false = dom_if->proj_out(0);
1408     Node* con_true = _igvn.makecon(TypeInt::ONE);
1409     Node* con_false = _igvn.makecon(TypeInt::ZERO);
1410 
1411     for (uint i = 1; i < n_ctrl->req(); i++) {
1412       if (is_dominator(proj_true, n_ctrl->in(i))) {
1413         bolphi->init_req(i, con_true);
1414       } else {
1415         assert(is_dominator(proj_false, n_ctrl->in(i)), "bad if");
1416         bolphi->init_req(i, con_false);
1417       }
1418     }
1419     register_new_node(bolphi, n_ctrl);
1420     _igvn.replace_input_of(n, 1, bolphi);
1421 
1422     // Now split the IF
1423     do_split_if(n);
1424     return;
1425   }
1426 
1427   // Check for an IF ready to split; one that has its
1428   // condition codes input coming from a Phi at the block start.
1429   int n_op = n->Opcode();
1430 
1431   // Check for an IF being dominated by another IF same test
1432   if (n_op == Op_If ||
1433       n_op == Op_RangeCheck) {
1434     Node *bol = n->in(1);
1435     uint max = bol->outcnt();
1436     // Check for same test used more than once?
1437     if (max > 1 && bol->is_Bool()) {
1438       // Search up IDOMs to see if this IF is dominated.
1439       Node *cutoff = get_ctrl(bol);
1440 
1441       // Now search up IDOMs till cutoff, looking for a dominating test
1442       Node *prevdom = n;
1443       Node *dom = idom(prevdom);
1444       while (dom != cutoff) {
1445         if (dom->req() > 1 && dom->in(1) == bol && prevdom->in(0) == dom) {
1446           // It's invalid to move control dependent data nodes in the inner
1447           // strip-mined loop, because:
1448           //  1) break validation of LoopNode::verify_strip_mined()
1449           //  2) move code with side-effect in strip-mined loop
1450           // Move to the exit of outer strip-mined loop in that case.
1451           Node* out_le = is_inner_of_stripmined_loop(dom);
1452           if (out_le != NULL) {
1453             prevdom = out_le;
1454           }
1455           // Replace the dominated test with an obvious true or false.
1456           // Place it on the IGVN worklist for later cleanup.
1457           C->set_major_progress();
1458           dominated_by(prevdom, n, false, true);
1459 #ifndef PRODUCT
1460           if( VerifyLoopOptimizations ) verify();
1461 #endif
1462           return;
1463         }
1464         prevdom = dom;
1465         dom = idom(prevdom);
1466       }
1467     }
1468   }
1469 
1470   // See if a shared loop-varying computation has no loop-varying uses.
1471   // Happens if something is only used for JVM state in uncommon trap exits,
1472   // like various versions of induction variable+offset.  Clone the
1473   // computation per usage to allow it to sink out of the loop.
1474   if (has_ctrl(n) && !n->in(0)) {// n not dead and has no control edge (can float about)
1475     Node *n_ctrl = get_ctrl(n);
1476     IdealLoopTree *n_loop = get_loop(n_ctrl);
1477     if( n_loop != _ltree_root ) {
1478       DUIterator_Fast imax, i = n->fast_outs(imax);
1479       for (; i < imax; i++) {
1480         Node* u = n->fast_out(i);
1481         if( !has_ctrl(u) )     break; // Found control user
1482         IdealLoopTree *u_loop = get_loop(get_ctrl(u));
1483         if( u_loop == n_loop ) break; // Found loop-varying use
1484         if( n_loop->is_member( u_loop ) ) break; // Found use in inner loop
1485         if( u->Opcode() == Op_Opaque1 ) break; // Found loop limit, bugfix for 4677003
1486       }
1487       bool did_break = (i < imax);  // Did we break out of the previous loop?
1488       if (!did_break && n->outcnt() > 1) { // All uses in outer loops!
1489         Node *late_load_ctrl = NULL;
1490         if (n->is_Load()) {
1491           // If n is a load, get and save the result from get_late_ctrl(),
1492           // to be later used in calculating the control for n's clones.
1493           clear_dom_lca_tags();
1494           late_load_ctrl = get_late_ctrl(n, n_ctrl);
1495         }
1496         // If n is a load, and the late control is the same as the current
1497         // control, then the cloning of n is a pointless exercise, because
1498         // GVN will ensure that we end up where we started.
1499         if (!n->is_Load() || late_load_ctrl != n_ctrl) {
1500           for (DUIterator_Last jmin, j = n->last_outs(jmin); j >= jmin; ) {
1501             Node *u = n->last_out(j); // Clone private computation per use
1502             _igvn.rehash_node_delayed(u);
1503             Node *x = n->clone(); // Clone computation
1504             Node *x_ctrl = NULL;
1505             if( u->is_Phi() ) {
1506               // Replace all uses of normal nodes.  Replace Phi uses
1507               // individually, so the separate Nodes can sink down
1508               // different paths.
1509               uint k = 1;
1510               while( u->in(k) != n ) k++;
1511               u->set_req( k, x );
1512               // x goes next to Phi input path
1513               x_ctrl = u->in(0)->in(k);
1514               --j;
1515             } else {              // Normal use
1516               // Replace all uses
1517               for( uint k = 0; k < u->req(); k++ ) {
1518                 if( u->in(k) == n ) {
1519                   u->set_req( k, x );
1520                   --j;
1521                 }
1522               }
1523               x_ctrl = get_ctrl(u);
1524             }
1525 
1526             // Find control for 'x' next to use but not inside inner loops.
1527             // For inner loop uses get the preheader area.
1528             x_ctrl = place_near_use(x_ctrl);
1529 
1530             if (n->is_Load()) {
1531               // For loads, add a control edge to a CFG node outside of the loop
1532               // to force them to not combine and return back inside the loop
1533               // during GVN optimization (4641526).
1534               //
1535               // Because we are setting the actual control input, factor in
1536               // the result from get_late_ctrl() so we respect any
1537               // anti-dependences. (6233005).
1538               x_ctrl = dom_lca(late_load_ctrl, x_ctrl);
1539 
1540               // Don't allow the control input to be a CFG splitting node.
1541               // Such nodes should only have ProjNodes as outs, e.g. IfNode
1542               // should only have IfTrueNode and IfFalseNode (4985384).
1543               x_ctrl = find_non_split_ctrl(x_ctrl);
1544 
1545               IdealLoopTree* x_loop = get_loop(x_ctrl);
1546               Node* x_head = x_loop->_head;
1547               if (x_head->is_Loop() && (x_head->is_OuterStripMinedLoop() || x_head->as_Loop()->is_strip_mined()) && is_dominator(n_ctrl, x_head)) {
1548                 // Anti dependence analysis is sometimes too
1549                 // conservative: a store in the outer strip mined loop
1550                 // can prevent a load from floating out of the outer
1551                 // strip mined loop but the load may not be referenced
1552                 // from the safepoint: loop strip mining verification
1553                 // code reports a problem in that case. Make sure the
1554                 // load is not moved in the outer strip mined loop in
1555                 // that case.
1556                 x_ctrl = x_head->as_Loop()->skip_strip_mined()->in(LoopNode::EntryControl);
1557               }
1558               assert(dom_depth(n_ctrl) <= dom_depth(x_ctrl), "n is later than its clone");
1559 
1560               x->set_req(0, x_ctrl);
1561             }
1562             register_new_node(x, x_ctrl);
1563 
1564             // Some institutional knowledge is needed here: 'x' is
1565             // yanked because if the optimizer runs GVN on it all the
1566             // cloned x's will common up and undo this optimization and
1567             // be forced back in the loop.
1568             // I tried setting control edges on the x's to force them to
1569             // not combine, but the matching gets worried when it tries
1570             // to fold a StoreP and an AddP together (as part of an
1571             // address expression) and the AddP and StoreP have
1572             // different controls.
1573             if (!x->is_Load() && !x->is_DecodeNarrowPtr()) _igvn._worklist.yank(x);
1574           }
1575           _igvn.remove_dead_node(n);
1576         }
1577       }
1578     }
1579   }
1580 
1581   try_move_store_after_loop(n);
1582 
1583   // Remove multiple allocations of the same value type
1584   if (n->is_ValueType()) {
1585     n->as_ValueType()->remove_redundant_allocations(&_igvn, this);
1586     return; // n is now dead
1587   }
1588 
1589   // Check for Opaque2's who's loop has disappeared - who's input is in the
1590   // same loop nest as their output.  Remove 'em, they are no longer useful.
1591   if( n_op == Op_Opaque2 &&
1592       n->in(1) != NULL &&
1593       get_loop(get_ctrl(n)) == get_loop(get_ctrl(n->in(1))) ) {
1594     _igvn.replace_node( n, n->in(1) );
1595   }
1596 }

1598 //------------------------------split_if_with_blocks---------------------------
1599 // Check for aggressive application of 'split-if' optimization,
1600 // using basic block level info.
1601 void PhaseIdealLoop::split_if_with_blocks(VectorSet &visited, Node_Stack &nstack) {
1602   Node* root = C->root();
1603   visited.set(root->_idx); // first, mark root as visited
1604   // Do pre-visit work for root
1605   Node* n   = split_if_with_blocks_pre(root);
1606   uint  cnt = n->outcnt();
1607   uint  i   = 0;
1608 
1609   while (true) {
1610     // Visit all children
1611     if (i < cnt) {
1612       Node* use = n->raw_out(i);
1613       ++i;
1614       if (use->outcnt() != 0 && !visited.test_set(use->_idx)) {
1615         // Now do pre-visit work for this use
1616         use = split_if_with_blocks_pre(use);
1617         nstack.push(n, i); // Save parent and next use's index.
1618         n   = use;         // Process all children of current use.
1619         cnt = use->outcnt();
1620         i   = 0;
1621       }
1622     }
1623     else {
1624       // All of n's children have been processed, complete post-processing.
1625       if (cnt != 0 && !n->is_Con()) {
1626         assert(has_node(n), "no dead nodes");
1627         split_if_with_blocks_post(n);
1628       }
1629       if (must_throttle_split_if()) {
1630         nstack.clear();
1631       }
1632       if (nstack.is_empty()) {
1633         // Finished all nodes on stack.
1634         break;
1635       }
1636       // Get saved parent node and next use's index. Visit the rest of uses.
1637       n   = nstack.node();
1638       cnt = n->outcnt();
1639       i   = nstack.index();
1640       nstack.pop();
1641     }
1642   }
1643 }

1652 // Passed in a Phi merging (recursively) some nearly equivalent Bool/Cmps.
1653 // "Nearly" because all Nodes have been cloned from the original in the loop,
1654 // but the fall-in edges to the Cmp are different.  Clone bool/Cmp pairs
1655 // through the Phi recursively, and return a Bool.
1656 Node* PhaseIdealLoop::clone_iff(PhiNode *phi, IdealLoopTree *loop) {
1657 
1658   // Convert this Phi into a Phi merging Bools
1659   uint i;
1660   for (i = 1; i < phi->req(); i++) {
1661     Node *b = phi->in(i);
1662     if (b->is_Phi()) {
1663       _igvn.replace_input_of(phi, i, clone_iff(b->as_Phi(), loop));
1664     } else {
1665       assert(b->is_Bool() || b->Opcode() == Op_Opaque4, "");
1666     }
1667   }
1668 
1669   Node* n = phi->in(1);
1670   Node* sample_opaque = NULL;
1671   Node *sample_bool = NULL;
1672   if (n->Opcode() == Op_Opaque4) {
1673     sample_opaque = n;
1674     sample_bool = n->in(1);
1675     assert(sample_bool->is_Bool(), "wrong type");
1676   } else {
1677     sample_bool = n;
1678   }
1679   Node *sample_cmp = sample_bool->in(1);
1680 
1681   // Make Phis to merge the Cmp's inputs.
1682   PhiNode *phi1 = new PhiNode(phi->in(0), Type::TOP);
1683   PhiNode *phi2 = new PhiNode(phi->in(0), Type::TOP);
1684   for (i = 1; i < phi->req(); i++) {
1685     Node *n1 = sample_opaque == NULL ? phi->in(i)->in(1)->in(1) : phi->in(i)->in(1)->in(1)->in(1);
1686     Node *n2 = sample_opaque == NULL ? phi->in(i)->in(1)->in(2) : phi->in(i)->in(1)->in(1)->in(2);
1687     phi1->set_req(i, n1);
1688     phi2->set_req(i, n2);
1689     phi1->set_type(phi1->type()->meet_speculative(n1->bottom_type()));
1690     phi2->set_type(phi2->type()->meet_speculative(n2->bottom_type()));
1691   }
1692   // See if these Phis have been made before.
1693   // Register with optimizer
1694   Node *hit1 = _igvn.hash_find_insert(phi1);
1695   if (hit1) {                   // Hit, toss just made Phi
1696     _igvn.remove_dead_node(phi1); // Remove new phi
1697     assert(hit1->is_Phi(), "" );
1698     phi1 = (PhiNode*)hit1;      // Use existing phi
1699   } else {                      // Miss
1700     _igvn.register_new_node_with_optimizer(phi1);
1701   }
1702   Node *hit2 = _igvn.hash_find_insert(phi2);
1703   if (hit2) {                   // Hit, toss just made Phi
1704     _igvn.remove_dead_node(phi2); // Remove new phi
1705     assert(hit2->is_Phi(), "" );
1706     phi2 = (PhiNode*)hit2;      // Use existing phi
1707   } else {                      // Miss
1708     _igvn.register_new_node_with_optimizer(phi2);
1709   }
1710   // Register Phis with loop/block info
1711   set_ctrl(phi1, phi->in(0));
1712   set_ctrl(phi2, phi->in(0));
1713   // Make a new Cmp
1714   Node *cmp = sample_cmp->clone();

1716   cmp->set_req(2, phi2);
1717   _igvn.register_new_node_with_optimizer(cmp);
1718   set_ctrl(cmp, phi->in(0));
1719 
1720   // Make a new Bool
1721   Node *b = sample_bool->clone();

1723   _igvn.register_new_node_with_optimizer(b);
1724   set_ctrl(b, phi->in(0));
1725 
1726   if (sample_opaque != NULL) {
1727     Node* opaque = sample_opaque->clone();
1728     opaque->set_req(1, b);
1729     _igvn.register_new_node_with_optimizer(opaque);
1730     set_ctrl(opaque, phi->in(0));
1731     return opaque;
1732   }
1733 
1734   assert(b->is_Bool(), "");

1736 }

1818 void PhaseIdealLoop::clone_loop_handle_data_uses(Node* old, Node_List &old_new,
1819                                                  IdealLoopTree* loop, IdealLoopTree* outer_loop,
1820                                                  Node_List*& split_if_set, Node_List*& split_bool_set,
1821                                                  Node_List*& split_cex_set, Node_List& worklist,
1822                                                  uint new_counter, CloneLoopMode mode) {
1823   Node* nnn = old_new[old->_idx];
1824   // Copy uses to a worklist, so I can munge the def-use info
1825   // with impunity.
1826   for (DUIterator_Fast jmax, j = old->fast_outs(jmax); j < jmax; j++)
1827     worklist.push(old->fast_out(j));
1828 
1829   while( worklist.size() ) {
1830     Node *use = worklist.pop();
1831     if (!has_node(use))  continue; // Ignore dead nodes
1832     if (use->in(0) == C->top())  continue;
1833     IdealLoopTree *use_loop = get_loop( has_ctrl(use) ? get_ctrl(use) : use );
1834     // Check for data-use outside of loop - at least one of OLD or USE
1835     // must not be a CFG node.
1836 #ifdef ASSERT
1837     if (loop->_head->as_Loop()->is_strip_mined() && outer_loop->is_member(use_loop) && !loop->is_member(use_loop) && old_new[use->_idx] == NULL) {
1838       Node* sfpt = loop->_head->as_CountedLoop()->outer_safepoint();
1839       assert(mode != IgnoreStripMined, "incorrect cloning mode");
1840       assert((mode == ControlAroundStripMined && use == sfpt) || !use->is_reachable_from_root(), "missed a node");
1841     }
1842 #endif
1843     if (!loop->is_member(use_loop) && !outer_loop->is_member(use_loop) && (!old->is_CFG() || !use->is_CFG())) {
1844 
1845       // If the Data use is an IF, that means we have an IF outside of the
1846       // loop that is switching on a condition that is set inside of the
1847       // loop.  Happens if people set a loop-exit flag; then test the flag
1848       // in the loop to break the loop, then test is again outside of the
1849       // loop to determine which way the loop exited.
1850       // Loop predicate If node connects to Bool node through Opaque1 node.
1851       if (use->is_If() || use->is_CMove() || C->is_predicate_opaq(use) || use->Opcode() == Op_Opaque4) {
1852         // Since this code is highly unlikely, we lazily build the worklist
1853         // of such Nodes to go split.
1854         if (!split_if_set) {
1855           ResourceArea *area = Thread::current()->resource_area();
1856           split_if_set = new Node_List(area);
1857         }
1858         split_if_set->push(use);
1859       }
1860       if (use->is_Bool()) {
1861         if (!split_bool_set) {
1862           ResourceArea *area = Thread::current()->resource_area();
1863           split_bool_set = new Node_List(area);
1864         }
1865         split_bool_set->push(use);
1866       }
1867       if (use->Opcode() == Op_CreateEx) {
1868         if (!split_cex_set) {
1869           ResourceArea *area = Thread::current()->resource_area();
1870           split_cex_set = new Node_List(area);
1871         }
1872         split_cex_set->push(use);
1873       }
1874 
1875 
1876       // Get "block" use is in
1877       uint idx = 0;
1878       while( use->in(idx) != old ) idx++;
1879       Node *prev = use->is_CFG() ? use : get_ctrl(use);
1880       assert(!loop->is_member(get_loop(prev)) && !outer_loop->is_member(get_loop(prev)), "" );
1881       Node *cfg = prev->_idx >= new_counter
1882         ? prev->in(2)
1883         : idom(prev);
1884       if( use->is_Phi() )     // Phi use is in prior block
1885         cfg = prev->in(idx);  // NOT in block of Phi itself
1886       if (cfg->is_top()) {    // Use is dead?
1887         _igvn.replace_input_of(use, idx, C->top());
1888         continue;
1889       }
1890 
1891       // If use is referenced through control edge... (idx == 0)
1892       if (mode == IgnoreStripMined && idx == 0) {
1893         LoopNode *head = loop->_head->as_Loop();
1894         if (head->is_strip_mined() && is_dominator(head->outer_loop_exit(), prev)) {
1895           // That node is outside the inner loop, leave it outside the
1896           // outer loop as well to not confuse verification code.
1897           assert(!loop->_parent->is_member(use_loop), "should be out of the outer loop");
1898           _igvn.replace_input_of(use, 0, head->outer_loop_exit());
1899           continue;
1900         }
1901       }
1902 
1903       while(!outer_loop->is_member(get_loop(cfg))) {
1904         prev = cfg;
1905         cfg = cfg->_idx >= new_counter ? cfg->in(2) : idom(cfg);
1906       }
1907       // If the use occurs after merging several exits from the loop, then
1908       // old value must have dominated all those exits.  Since the same old
1909       // value was used on all those exits we did not need a Phi at this
1910       // merge point.  NOW we do need a Phi here.  Each loop exit value
1911       // is now merged with the peeled body exit; each exit gets its own
1912       // private Phi and those Phis need to be merged here.
1913       Node *phi;
1914       if( prev->is_Region() ) {
1915         if( idx == 0 ) {      // Updating control edge?
1916           phi = prev;         // Just use existing control
1917         } else {              // Else need a new Phi
1918           phi = PhiNode::make( prev, old );
1919           // Now recursively fix up the new uses of old!
1920           for( uint i = 1; i < prev->req(); i++ ) {
1921             worklist.push(phi); // Onto worklist once for each 'old' input
1922           }
1923         }
1924       } else {
1925         // Get new RegionNode merging old and new loop exits
1926         prev = old_new[prev->_idx];
1927         assert( prev, "just made this in step 7" );
1928         if( idx == 0) {      // Updating control edge?
1929           phi = prev;         // Just use existing control
1930         } else {              // Else need a new Phi
1931           // Make a new Phi merging data values properly
1932           phi = PhiNode::make( prev, old );
1933           phi->set_req( 1, nnn );
1934         }
1935       }
1936       // If inserting a new Phi, check for prior hits
1937       if( idx != 0 ) {
1938         Node *hit = _igvn.hash_find_insert(phi);
1939         if( hit == NULL ) {
1940           _igvn.register_new_node_with_optimizer(phi); // Register new phi
1941         } else {                                      // or
1942           // Remove the new phi from the graph and use the hit
1943           _igvn.remove_dead_node(phi);
1944           phi = hit;                                  // Use existing phi
1945         }
1946         set_ctrl(phi, prev);
1947       }
1948       // Make 'use' use the Phi instead of the old loop body exit value
1949       _igvn.replace_input_of(use, idx, phi);
1950       if( use->_idx >= new_counter ) { // If updating new phis
1951         // Not needed for correctness, but prevents a weak assert
1952         // in AddPNode from tripping (when we end up with different
1953         // base & derived Phis that will become the same after
1954         // IGVN does CSE).
1955         Node *hit = _igvn.hash_find_insert(use);
1956         if( hit )             // Go ahead and re-hash for hits.
1957           _igvn.replace_node( use, hit );
1958       }
1959 
1960       // If 'use' was in the loop-exit block, it now needs to be sunk
1961       // below the post-loop merge point.
1962       sink_use( use, prev );
1963     }
1964   }
1965 }

1985                                       IdealLoopTree* outer_loop, int dd, Node_List &old_new,

2110 //------------------------------clone_loop-------------------------------------
2111 //
2112 //                   C L O N E   A   L O O P   B O D Y
2113 //
2114 // This is the basic building block of the loop optimizations.  It clones an
2115 // entire loop body.  It makes an old_new loop body mapping; with this mapping
2116 // you can find the new-loop equivalent to an old-loop node.  All new-loop
2117 // nodes are exactly equal to their old-loop counterparts, all edges are the
2118 // same.  All exits from the old-loop now have a RegionNode that merges the
2119 // equivalent new-loop path.  This is true even for the normal "loop-exit"
2120 // condition.  All uses of loop-invariant old-loop values now come from (one
2121 // or more) Phis that merge their new-loop equivalents.
2122 //

2361 //---------------------- stride_of_possible_iv -------------------------------------
2362 // Looks for an iff/bool/comp with one operand of the compare
2363 // being a cycle involving an add and a phi,
2364 // with an optional truncation (left-shift followed by a right-shift)
2365 // of the add. Returns zero if not an iv.
2366 int PhaseIdealLoop::stride_of_possible_iv(Node* iff) {
2367   Node* trunc1 = NULL;
2368   Node* trunc2 = NULL;
2369   const TypeInt* ttype = NULL;
2370   if (!iff->is_If() || iff->in(1) == NULL || !iff->in(1)->is_Bool()) {
2371     return 0;
2372   }
2373   BoolNode* bl = iff->in(1)->as_Bool();
2374   Node* cmp = bl->in(1);
2375   if (!cmp || (cmp->Opcode() != Op_CmpI && cmp->Opcode() != Op_CmpU)) {
2376     return 0;
2377   }
2378   // Must have an invariant operand
2379   if (is_member(get_loop(iff), get_ctrl(cmp->in(2)))) {
2380     return 0;
2381   }
2382   Node* add2 = NULL;
2383   Node* cmp1 = cmp->in(1);
2384   if (cmp1->is_Phi()) {
2385     // (If (Bool (CmpX phi:(Phi ...(Optional-trunc(AddI phi add2))) )))
2386     Node* phi = cmp1;
2387     for (uint i = 1; i < phi->req(); i++) {
2388       Node* in = phi->in(i);
2389       Node* add = CountedLoopNode::match_incr_with_optional_truncation(in,
2390                                 &trunc1, &trunc2, &ttype);
2391       if (add && add->in(1) == phi) {
2392         add2 = add->in(2);
2393         break;
2394       }
2395     }
2396   } else {
2397     // (If (Bool (CmpX addtrunc:(Optional-trunc((AddI (Phi ...addtrunc...) add2)) )))
2398     Node* addtrunc = cmp1;
2399     Node* add = CountedLoopNode::match_incr_with_optional_truncation(addtrunc,
2400                                 &trunc1, &trunc2, &ttype);
2401     if (add && add->in(1)->is_Phi()) {
2402       Node* phi = add->in(1);
2403       for (uint i = 1; i < phi->req(); i++) {
2404         if (phi->in(i) == addtrunc) {
2405           add2 = add->in(2);
2406           break;
2407         }
2408       }
2409     }
2410   }
2411   if (add2 != NULL) {
2412     const TypeInt* add2t = _igvn.type(add2)->is_int();
2413     if (add2t->is_con()) {
2414       return add2t->get_con();
2415     }
2416   }
2417   return 0;
2418 }

2586 // Clone a signed compare loop exit from an unsigned compare and

2588 // All new nodes inserted in the dominator tree between the original
2589 // if and it's projections.  The original if test is replaced with
2590 // a constant to force the stay-in-loop path.

2592 // This is done to make sure that the original if and it's projections
2593 // still dominate the same set of control nodes, that the ctrl() relation
2594 // from data nodes to them is preserved, and that their loop nesting is
2595 // preserved.
2596 //
2597 // before

2600 //        v        v

2602 //
2603 // after
2604 //          if(stay-in-loop-const)  original if
2605 //         /       |
2606 //        /        v
2607 //       /  if(i <  limit)    new signed test
2608 //      /  /       |
2609 //     /  /        v
2610 //    /  /  if(i <u limit)    new cloned unsigned test
2611 //   /  /   /      |
2612 //   v  v  v       |
2613 //    region       |
2614 //        |        |
2615 //      dum-if     |
2616 //     /  |        |
2617 // ether  |        |
2618 //        v        v
2619 //   exit-proj   stay-in-loop-proj
2620 //
2621 IfNode* PhaseIdealLoop::insert_cmpi_loop_exit(IfNode* if_cmpu, IdealLoopTree *loop) {

2623   const bool Unsigned = false;

2627   CmpNode* cmpu = bol->in(1)->as_Cmp();
2628   if (cmpu->Opcode() != Op_CmpU) return NULL;
2629   int stride = stride_of_possible_iv(if_cmpu);
2630   if (stride == 0) return NULL;
2631 
2632   Node* lp_proj = stay_in_loop(if_cmpu, loop);
2633   guarantee(lp_proj != NULL, "null loop node");
2634 
2635   ProjNode* lp_continue = lp_proj->as_Proj();

2640     limit = cmpu->in(2);
2641   } else {

2646   RegionNode* reg = insert_region_before_proj(lp_exit);

2677 //------------------------------ scheduled_nodelist -------------------------------------
2678 // Create a post order schedule of nodes that are in the
2679 // "member" set.  The list is returned in "sched".
2680 // The first node in "sched" is the loop head, followed by
2681 // nodes which have no inputs in the "member" set, and then
2682 // followed by the nodes that have an immediate input dependence
2683 // on a node in "sched".
2684 void PhaseIdealLoop::scheduled_nodelist( IdealLoopTree *loop, VectorSet& member, Node_List &sched ) {
2685 
2686   assert(member.test(loop->_head->_idx), "loop head must be in member set");
2687   Arena *a = Thread::current()->resource_area();
2688   VectorSet visited(a);
2689   Node_Stack nstack(a, loop->_body.size());
2690 
2691   Node* n  = loop->_head;  // top of stack is cached in "n"
2692   uint idx = 0;
2693   visited.set(n->_idx);
2694 
2695   // Initially push all with no inputs from within member set
2696   for(uint i = 0; i < loop->_body.size(); i++ ) {
2697     Node *elt = loop->_body.at(i);
2698     if (member.test(elt->_idx)) {
2699       bool found = false;
2700       for (uint j = 0; j < elt->req(); j++) {
2701         Node* def = elt->in(j);
2702         if (def && member.test(def->_idx) && def != elt) {
2703           found = true;
2704           break;
2705         }
2706       }
2707       if (!found && elt != loop->_head) {
2708         nstack.push(n, idx);
2709         n = elt;
2710         assert(!visited.test(n->_idx), "not seen yet");
2711         visited.set(n->_idx);
2712       }
2713     }
2714   }
2715 
2716   // traverse out's that are in the member set
2717   while (true) {
2718     if (idx < n->outcnt()) {
2719       Node* use = n->raw_out(idx);
2720       idx++;
2721       if (!visited.test_set(use->_idx)) {
2722         if (member.test(use->_idx)) {
2723           nstack.push(n, idx);
2724           n = use;
2725           idx = 0;
2726         }
2727       }
2728     } else {
2729       // All outputs processed
2730       sched.push(n);
2731       if (nstack.is_empty()) break;
2732       n   = nstack.node();
2733       idx = nstack.index();
2734       nstack.pop();
2735     }
2736   }
2737 }

2767 //------------------------------ clone_for_use_outside_loop -------------------------------------
2768 // clone "n" for uses that are outside of loop
2769 int PhaseIdealLoop::clone_for_use_outside_loop( IdealLoopTree *loop, Node* n, Node_List& worklist ) {
2770   int cloned = 0;
2771   assert(worklist.size() == 0, "should be empty");
2772   for (DUIterator_Fast jmax, j = n->fast_outs(jmax); j < jmax; j++) {
2773     Node* use = n->fast_out(j);
2774     if( !loop->is_member(get_loop(has_ctrl(use) ? get_ctrl(use) : use)) ) {
2775       worklist.push(use);
2776     }
2777   }
2778   while( worklist.size() ) {
2779     Node *use = worklist.pop();
2780     if (!has_node(use) || use->in(0) == C->top()) continue;
2781     uint j;
2782     for (j = 0; j < use->req(); j++) {
2783       if (use->in(j) == n) break;

2785     assert(j < use->req(), "must be there");
2786 
2787     // clone "n" and insert it between the inputs of "n" and the use outside the loop
2788     Node* n_clone = n->clone();

2796       use_c = use->in(0)->in(j);

2807   }

2815 // marks the method as not compilable.  For example, if a "BoolNode"
2816 // is in a different basic block than the "IfNode" that uses it, then

2851         }
2852       }
2853     }
2854   }

2979 //         section that can be executed after the cut into

3166 
3167   assert(!loop->_head->is_CountedLoop(), "Non-counted loop only");
3168   if (!loop->_head->is_Loop()) {
3169     return false;
3170   }
3171   LoopNode *head = loop->_head->as_Loop();
3172 
3173   if (head->is_partial_peel_loop() || head->partial_peel_has_failed()) {
3174     return false;
3175   }
3176 
3177   // Check for complex exit control
3178   for (uint ii = 0; ii < loop->_body.size(); ii++) {
3179     Node *n = loop->_body.at(ii);
3180     int opc = n->Opcode();
3181     if (n->is_Call()        ||
3182         opc == Op_Catch     ||
3183         opc == Op_CatchProj ||
3184         opc == Op_Jump      ||
3185         opc == Op_JumpProj) {
3186 #ifndef PRODUCT
3187       if (TracePartialPeeling) {
3188         tty->print_cr("\nExit control too complex: lp: %d", head->_idx);
3189       }
3190 #endif
3191       return false;
3192     }
3193   }
3194 
3195   int dd = dom_depth(head);
3196 
3197   // Step 1: find cut point
3198 
3199   // Walk up dominators to loop head looking for first loop exit
3200   // which is executed on every path thru loop.
3201   IfNode *peel_if = NULL;
3202   IfNode *peel_if_cmpu = NULL;
3203 
3204   Node *iff = loop->tail();
3205   while (iff != head) {
3206     if (iff->is_If()) {
3207       Node *ctrl = get_ctrl(iff->in(1));
3208       if (ctrl->is_top()) return false; // Dead test on live IF.
3209       // If loop-varying exit-test, check for induction variable
3210       if (loop->is_member(get_loop(ctrl)) &&
3211           loop->is_loop_exit(iff) &&
3212           is_possible_iv_test(iff)) {
3213         Node* cmp = iff->in(1)->in(1);
3214         if (cmp->Opcode() == Op_CmpI) {
3215           peel_if = iff->as_If();
3216         } else {
3217           assert(cmp->Opcode() == Op_CmpU, "must be CmpI or CmpU");
3218           peel_if_cmpu = iff->as_If();
3219         }
3220       }
3221     }
3222     iff = idom(iff);
3223   }
3224 
3225   // Prefer signed compare over unsigned compare.
3226   IfNode* new_peel_if = NULL;
3227   if (peel_if == NULL) {
3228     if (!PartialPeelAtUnsignedTests || peel_if_cmpu == NULL) {
3229       return false;   // No peel point found
3230     }
3231     new_peel_if = insert_cmpi_loop_exit(peel_if_cmpu, loop);
3232     if (new_peel_if == NULL) {
3233       return false;   // No peel point found
3234     }
3235     peel_if = new_peel_if;
3236   }
3237   Node* last_peel        = stay_in_loop(peel_if, loop);
3238   Node* first_not_peeled = stay_in_loop(last_peel, loop);
3239   if (first_not_peeled == NULL || first_not_peeled == head) {
3240     return false;
3241   }
3242 
3243 #ifndef PRODUCT

3556 //------------------------------reorg_offsets----------------------------------
3557 // Reorganize offset computations to lower register pressure.  Mostly
3558 // prevent loop-fallout uses of the pre-incremented trip counter (which are
3559 // then alive with the post-incremented trip counter forcing an extra
3560 // register move)
3561 void PhaseIdealLoop::reorg_offsets(IdealLoopTree *loop) {
3562   // Perform it only for canonical counted loops.
3563   // Loop's shape could be messed up by iteration_split_impl.
3564   if (!loop->_head->is_CountedLoop())
3565     return;
3566   if (!loop->_head->as_Loop()->is_valid_counted_loop())
3567     return;
3568 
3569   CountedLoopNode *cl = loop->_head->as_CountedLoop();
3570   CountedLoopEndNode *cle = cl->loopexit();
3571   Node *exit = cle->proj_out(false);
3572   Node *phi = cl->phi();
3573 
3574   // Check for the special case when using the pre-incremented trip-counter on
3575   // the fall-out  path (forces the pre-incremented  and post-incremented trip
3576   // counter to be live  at the same time).  Fix this by  adjusting to use the
3577   // post-increment trip counter.
3578 
3579   bool progress = true;
3580   while (progress) {
3581     progress = false;
3582     for (DUIterator_Fast imax, i = phi->fast_outs(imax); i < imax; i++) {
3583       Node* use = phi->fast_out(i);   // User of trip-counter
3584       if (!has_ctrl(use))  continue;
3585       Node *u_ctrl = get_ctrl(use);
3586       if (use->is_Phi()) {
3587         u_ctrl = NULL;
3588         for (uint j = 1; j < use->req(); j++)
3589           if (use->in(j) == phi)
3590             u_ctrl = dom_lca(u_ctrl, use->in(0)->in(j));
3591       }
3592       IdealLoopTree *u_loop = get_loop(u_ctrl);
3593       // Look for loop-invariant use
3594       if (u_loop == loop) continue;
3595       if (loop->is_member(u_loop)) continue;
3596       // Check that use is live out the bottom.  Assuming the trip-counter
3597       // update is right at the bottom, uses of of the loop middle are ok.
3598       if (dom_lca(exit, u_ctrl) != exit) continue;
3599       // Hit!  Refactor use to use the post-incremented tripcounter.
3600       // Compute a post-increment tripcounter.
3601       Node* c = exit;
3602       if (cl->is_strip_mined()) {
3603         IdealLoopTree* outer_loop = get_loop(cl->outer_loop());
3604         if (!outer_loop->is_member(u_loop)) {
3605           c = cl->outer_loop_exit();
3606         }
3607       }
3608       Node *opaq = new Opaque2Node(C, cle->incr());
3609       register_new_node(opaq, c);
3610       Node *neg_stride = _igvn.intcon(-cle->stride_con());
3611       set_ctrl(neg_stride, C->root());
3612       Node *post = new AddINode(opaq, neg_stride);
3613       register_new_node(post, c);
3614       _igvn.rehash_node_delayed(use);
3615       for (uint j = 1; j < use->req(); j++) {
3616         if (use->in(j) == phi)
3617           use->set_req(j, post);
3618       }
3619       // Since DU info changed, rerun loop
3620       progress = true;
3621       break;
3622     }
3623   }
