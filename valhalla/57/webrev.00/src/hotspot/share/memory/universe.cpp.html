 276 void Universe::check_alignment(uintx size, uintx alignment, const char* name) {

 283 void initialize_basic_type_klass(Klass* k, TRAPS) {
 284   Klass* ok = SystemDictionary::Object_klass();
 285 #if INCLUDE_CDS
 286   if (UseSharedSpaces) {
 287     ClassLoaderData* loader_data = ClassLoaderData::the_null_class_loader_data();
 288     assert(k->super() == ok, "u3");
 289     if (k->is_instance_klass()) {
 290       InstanceKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), NULL, CHECK);
 291     } else {
 292       ArrayKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), CHECK);
 293     }
 294   } else
 295 #endif
 296   {
 297     k->initialize_supers(ok, NULL, CHECK);
 298   }
 299   k->append_to_sibling_list();
 300 }

 302 void Universe::genesis(TRAPS) {
 303   ResourceMark rm(THREAD);
 304 
 305   { FlagSetting fs(_bootstrapping, true);
 306 
 307     { MutexLocker mc(THREAD, Compile_lock);
 308 
 309       java_lang_Class::allocate_fixup_lists();
 310 
 311       // determine base vtable size; without that we cannot create the array klasses
 312       compute_base_vtable_size();
 313 
 314       if (!UseSharedSpaces) {
 315         for (int i = T_BOOLEAN; i < T_LONG+1; i++) {
 316           _typeArrayKlassObjs[i] = TypeArrayKlass::create_klass((BasicType)i, CHECK);
 317         }
 318 
 319         ClassLoaderData* null_cld = ClassLoaderData::the_null_class_loader_data();
 320 
 321         _the_array_interfaces_array     = MetadataFactory::new_array<Klass*>(null_cld, 3, NULL, CHECK);
 322         _the_empty_int_array            = MetadataFactory::new_array<int>(null_cld, 0, CHECK);
 323         _the_empty_short_array          = MetadataFactory::new_array<u2>(null_cld, 0, CHECK);
 324         _the_empty_method_array         = MetadataFactory::new_array<Method*>(null_cld, 0, CHECK);
 325         _the_empty_klass_array          = MetadataFactory::new_array<Klass*>(null_cld, 0, CHECK);
 326         _the_empty_instance_klass_array = MetadataFactory::new_array<InstanceKlass*>(null_cld, 0, CHECK);
 327       }
 328     }
 329 
 330     vmSymbols::initialize(CHECK);
 331 
 332     SystemDictionary::initialize(CHECK);
 333 
 334     Klass* ok = SystemDictionary::Object_klass();
 335 
 336     _the_null_string            = StringTable::intern("null", CHECK);
 337     _the_min_jint_string       = StringTable::intern("-2147483648", CHECK);
 338 
 339 #if INCLUDE_CDS
 340     if (UseSharedSpaces) {
 341       // Verify shared interfaces array.
 342       assert(_the_array_interfaces_array->at(0) ==
 343              SystemDictionary::Cloneable_klass(), "u3");
 344       assert(_the_array_interfaces_array->at(1) ==
 345              SystemDictionary::Serializable_klass(), "u3");
 346       assert(_the_array_interfaces_array->at(2) ==
 347                    SystemDictionary::IdentityObject_klass(), "u3");
 348 
 349       assert(_the_single_IdentityObject_klass_array->at(0) ==
 350           SystemDictionary::IdentityObject_klass(), "u3");
 351     } else
 352 #endif
 353     {
 354       // Set up shared interfaces array.  (Do this before supers are set up.)
 355       _the_array_interfaces_array->at_put(0, SystemDictionary::Cloneable_klass());
 356       _the_array_interfaces_array->at_put(1, SystemDictionary::Serializable_klass());
 357       _the_array_interfaces_array->at_put(2, SystemDictionary::IdentityObject_klass());
 358     }
 359 
 360     initialize_basic_type_klass(boolArrayKlassObj(), CHECK);
 361     initialize_basic_type_klass(charArrayKlassObj(), CHECK);
 362     initialize_basic_type_klass(floatArrayKlassObj(), CHECK);
 363     initialize_basic_type_klass(doubleArrayKlassObj(), CHECK);
 364     initialize_basic_type_klass(byteArrayKlassObj(), CHECK);
 365     initialize_basic_type_klass(shortArrayKlassObj(), CHECK);
 366     initialize_basic_type_klass(intArrayKlassObj(), CHECK);
 367     initialize_basic_type_klass(longArrayKlassObj(), CHECK);
 368   } // end of core bootstrapping
 369 
 370   {
 371     Handle tns = java_lang_String::create_from_str("<null_sentinel>", CHECK);
 372     _the_null_sentinel = tns();
 373   }
 374 
 375   // Maybe this could be lifted up now that object array can be initialized
 376   // during the bootstrapping.
 377 
 378   // OLD
 379   // Initialize _objectArrayKlass after core bootstraping to make
 380   // sure the super class is set up properly for _objectArrayKlass.
 381   // ---
 382   // NEW
 383   // Since some of the old system object arrays have been converted to
 384   // ordinary object arrays, _objectArrayKlass will be loaded when
 385   // SystemDictionary::initialize(CHECK); is run. See the extra check
 386   // for Object_klass_loaded in objArrayKlassKlass::allocate_objArray_klass_impl.
 387   _objectArrayKlassObj = InstanceKlass::
 388     cast(SystemDictionary::Object_klass())->array_klass(1, CHECK);
 389   // OLD
 390   // Add the class to the class hierarchy manually to make sure that
 391   // its vtable is initialized after core bootstrapping is completed.
 392   // ---
 393   // New
 394   // Have already been initialized.
 395   _objectArrayKlassObj->append_to_sibling_list();
 396 
 397   #ifdef ASSERT
 398   if (FullGCALot) {
 399     // Allocate an array of dummy objects.
 400     // We'd like these to be at the bottom of the old generation,
 401     // so that when we free one and then collect,
 402     // (almost) the whole heap moves
 403     // and we find out if we actually update all the oops correctly.
 404     // But we can't allocate directly in the old generation,
 405     // so we allocate wherever, and hope that the first collection
 406     // moves these objects to the bottom of the old generation.
 407     // We can allocate directly in the permanent generation, so we do.
 408     int size = FullGCALotDummies * 2;
 409 
 410     objArrayOop    naked_array = oopFactory::new_objArray(SystemDictionary::Object_klass(), size, CHECK);
 411     objArrayHandle dummy_array(THREAD, naked_array);
 412     int i = 0;
 413     while (i < size) {
 414         // Allocate dummy in old generation
 415       oop dummy = SystemDictionary::Object_klass()->allocate_instance(CHECK);
 416       dummy_array->obj_at_put(i++, dummy);
 417     }
 418     {
 419       // Only modify the global variable inside the mutex.
 420       // If we had a race to here, the other dummy_array instances
 421       // and their elements just get dropped on the floor, which is fine.
 422       MutexLocker ml(THREAD, FullGCALot_lock);
 423       if (_fullgc_alot_dummy_array == NULL) {
 424         _fullgc_alot_dummy_array = dummy_array();
 425       }
 426     }
 427     assert(i == _fullgc_alot_dummy_array->length(), "just checking");
 428   }
 429   #endif
 430 }

 487 void Universe::fixup_mirrors(TRAPS) {
 488   // Bootstrap problem: all classes gets a mirror (java.lang.Class instance) assigned eagerly,
 489   // but we cannot do that for classes created before java.lang.Class is loaded. Here we simply
 490   // walk over permanent objects created so far (mostly classes) and fixup their mirrors. Note
 491   // that the number of objects allocated at this point is very small.
 492   assert(SystemDictionary::Class_klass_loaded(), "java.lang.Class should be loaded");
 493   HandleMark hm(THREAD);
 494 
 495   if (!UseSharedSpaces) {
 496     // Cache the start of the static fields
 497     InstanceMirrorKlass::init_offset_of_static_fields();
 498   }
 499 
 500   GrowableArray <Klass*>* list = java_lang_Class::fixup_mirror_list();
 501   int list_length = list->length();
 502   for (int i = 0; i < list_length; i++) {
 503     Klass* k = list->at(i);
 504     assert(k->is_klass(), "List should only hold classes");
 505     EXCEPTION_MARK;
 506     java_lang_Class::fixup_mirror(k, CATCH);
 507   }
 508   delete java_lang_Class::fixup_mirror_list();
 509   java_lang_Class::set_fixup_mirror_list(NULL);
 510 }

 581 bool Universe::should_fill_in_stack_trace(Handle throwable) {
 582   // never attempt to fill in the stack trace of preallocated errors that do not have
 583   // backtrace. These errors are kept alive forever and may be "re-used" when all
 584   // preallocated errors with backtrace have been consumed. Also need to avoid
 585   // a potential loop which could happen if an out of memory occurs when attempting
 586   // to allocate the backtrace.
 587   return ((throwable() != Universe::_out_of_memory_error_java_heap) &&
 588           (throwable() != Universe::_out_of_memory_error_metaspace)  &&
 589           (throwable() != Universe::_out_of_memory_error_class_metaspace)  &&
 590           (throwable() != Universe::_out_of_memory_error_array_size) &&
 591           (throwable() != Universe::_out_of_memory_error_gc_overhead_limit) &&
 592           (throwable() != Universe::_out_of_memory_error_realloc_objects) &&
 593           (throwable() != Universe::_out_of_memory_error_retry));
 594 }

 596 
 597 oop Universe::gen_out_of_memory_error(oop default_err) {
 598   // generate an out of memory error:
 599   // - if there is a preallocated error and stack traces are available
 600   //   (j.l.Throwable is initialized), then return the preallocated
 601   //   error with a filled in stack trace, and with the message
 602   //   provided by the default error.
 603   // - otherwise, return the default error, without a stack trace.
 604   int next;
 605   if ((_preallocated_out_of_memory_error_avail_count > 0) &&
 606       SystemDictionary::Throwable_klass()->is_initialized()) {
 607     next = (int)Atomic::add(&_preallocated_out_of_memory_error_avail_count, -1);
 608     assert(next < (int)PreallocatedOutOfMemoryErrorCount, "avail count is corrupt");
 609   } else {
 610     next = -1;
 611   }
 612   if (next < 0) {
 613     // all preallocated errors have been used.
 614     // return default
 615     return default_err;
 616   } else {
 617     Thread* THREAD = Thread::current();
 618     Handle default_err_h(THREAD, default_err);
 619     // get the error object at the slot and set set it to NULL so that the
 620     // array isn't keeping it alive anymore.
 621     Handle exc(THREAD, preallocated_out_of_memory_errors()->obj_at(next));
 622     assert(exc() != NULL, "slot has been used already");
 623     preallocated_out_of_memory_errors()->obj_at_put(next, NULL);
 624 
 625     // use the message from the default error
 626     oop msg = java_lang_Throwable::message(default_err_h());
 627     assert(msg != NULL, "no message");
 628     java_lang_Throwable::set_message(exc(), msg);
 629 
 630     // populate the stack trace and return it.
 631     java_lang_Throwable::fill_in_stack_trace_of_preallocated_backtrace(exc);
 632     return exc();
 633   }
 634 }

 636 intptr_t Universe::_non_oop_bits = 0;
 637 
 638 void* Universe::non_oop_word() {
 639   // Neither the high bits nor the low bits of this value is allowed
 640   // to look like (respectively) the high or low bits of a real oop.
 641   //
 642   // High and low are CPU-specific notions, but low always includes
 643   // the low-order bit.  Since oops are always aligned at least mod 4,
 644   // setting the low-order bit will ensure that the low half of the
 645   // word will never look like that of a real oop.
 646   //
 647   // Using the OS-supplied non-memory-address word (usually 0 or -1)
 648   // will take care of the high bits, however many there are.
 649 
 650   if (_non_oop_bits == 0) {
 651     _non_oop_bits = (intptr_t)os::non_memory_address_word() | 1;
 652   }
 653 
 654   return (void*)_non_oop_bits;
 655 }

 761 ReservedHeapSpace Universe::reserve_heap(size_t heap_size, size_t alignment) {
 762 
 763   assert(alignment <= Arguments::conservative_max_heap_alignment(),
 764          "actual alignment " SIZE_FORMAT " must be within maximum heap alignment " SIZE_FORMAT,
 765          alignment, Arguments::conservative_max_heap_alignment());
 766 
 767   size_t total_reserved = align_up(heap_size, alignment);
 768   assert(!UseCompressedOops || (total_reserved <= (OopEncodingHeapMax - os::vm_page_size())),
 769       "heap size is too big for compressed oops");
 770 
 771   bool use_large_pages = UseLargePages && is_aligned(alignment, os::large_page_size());
 772   assert(!UseLargePages
 773       || UseParallelGC
 774       || use_large_pages, "Wrong alignment to use large pages");
 775 
 776   // Now create the space.
 777   ReservedHeapSpace total_rs(total_reserved, alignment, use_large_pages, AllocateHeapAt);
 778 
 779   if (total_rs.is_reserved()) {
 780     assert((total_reserved == total_rs.size()) && ((uintptr_t)total_rs.base() % alignment == 0),
 781            "must be exactly of required size and alignment");
 782     // We are good.
 783 
 784     if (AllocateHeapAt != NULL) {
 785       log_info(gc,heap)("Successfully allocated Java heap at location %s", AllocateHeapAt);
 786     }
 787 
 788     if (UseCompressedOops) {
 789       CompressedOops::initialize(total_rs);
 790     }
 791 
 792     Universe::calculate_verify_data((HeapWord*)total_rs.base(), (HeapWord*)total_rs.end());
 793 
 794     return total_rs;
 795   }
 796 
 797   vm_exit_during_initialization(
 798     err_msg("Could not reserve enough space for " SIZE_FORMAT "KB object heap",
 799             total_reserved/K));
 800 
 801   // satisfy compiler
 802   ShouldNotReachHere();
 803   return ReservedHeapSpace(0, 0, false);
 804 }

 882 bool universe_post_init() {
 883   assert(!is_init_completed(), "Error: initialization not yet completed!");
 884   Universe::_fully_initialized = true;
 885   EXCEPTION_MARK;
 886   if (!UseSharedSpaces) {
 887     ResourceMark rm;
 888     Universe::reinitialize_vtables(CHECK_false);
 889     Universe::reinitialize_itables(CHECK_false);
 890   }
 891 
 892   HandleMark hm(THREAD);
 893   // Setup preallocated empty java.lang.Class array
 894   Universe::_the_empty_class_klass_array = oopFactory::new_objArray(SystemDictionary::Class_klass(), 0, CHECK_false);
 895 
 896   // Setup preallocated OutOfMemoryError errors
 897   Klass* k = SystemDictionary::OutOfMemoryError_klass();
 898   InstanceKlass* ik = InstanceKlass::cast(k);
 899   Universe::_out_of_memory_error_java_heap = ik->allocate_instance(CHECK_false);
 900   Universe::_out_of_memory_error_metaspace = ik->allocate_instance(CHECK_false);
 901   Universe::_out_of_memory_error_class_metaspace = ik->allocate_instance(CHECK_false);
 902   Universe::_out_of_memory_error_array_size = ik->allocate_instance(CHECK_false);
 903   Universe::_out_of_memory_error_gc_overhead_limit =
 904     ik->allocate_instance(CHECK_false);
 905   Universe::_out_of_memory_error_realloc_objects = ik->allocate_instance(CHECK_false);
 906   Universe::_out_of_memory_error_retry = ik->allocate_instance(CHECK_false);
 907 
 908   // Setup preallocated cause message for delayed StackOverflowError
 909   if (StackReservedPages > 0) {
 910     Universe::_delayed_stack_overflow_error_message =
 911       java_lang_String::create_oop_from_str("Delayed StackOverflowError due to ReservedStackAccess annotated method", CHECK_false);
 912   }
 913 
 914   // Setup preallocated NullPointerException
 915   // (this is currently used for a cheap & dirty solution in compiler exception handling)
 916   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_NullPointerException(), true, CHECK_false);
 917   Universe::_null_ptr_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 918   // Setup preallocated ArithmeticException
 919   // (this is currently used for a cheap & dirty solution in compiler exception handling)
 920   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_ArithmeticException(), true, CHECK_false);
 921   Universe::_arithmetic_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 922   // Virtual Machine Error for when we get into a situation we can't resolve
 923   k = SystemDictionary::VirtualMachineError_klass();
 924   bool linked = InstanceKlass::cast(k)->link_class_or_fail(CHECK_false);
 925   if (!linked) {
 926      tty->print_cr("Unable to link/verify VirtualMachineError class");
 927      return false; // initialization failed
 928   }
 929   Universe::_virtual_machine_error_instance =
 930     InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 931 
 932   Universe::_vm_exception = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 933 
 934   Handle msg = java_lang_String::create_from_str("Java heap space", CHECK_false);
 935   java_lang_Throwable::set_message(Universe::_out_of_memory_error_java_heap, msg());
 936 
 937   msg = java_lang_String::create_from_str("Metaspace", CHECK_false);
 938   java_lang_Throwable::set_message(Universe::_out_of_memory_error_metaspace, msg());
 939   msg = java_lang_String::create_from_str("Compressed class space", CHECK_false);
 940   java_lang_Throwable::set_message(Universe::_out_of_memory_error_class_metaspace, msg());
 941 
 942   msg = java_lang_String::create_from_str("Requested array size exceeds VM limit", CHECK_false);
 943   java_lang_Throwable::set_message(Universe::_out_of_memory_error_array_size, msg());
 944 
 945   msg = java_lang_String::create_from_str("GC overhead limit exceeded", CHECK_false);
 946   java_lang_Throwable::set_message(Universe::_out_of_memory_error_gc_overhead_limit, msg());
 947 
 948   msg = java_lang_String::create_from_str("Java heap space: failed reallocation of scalar replaced objects", CHECK_false);
 949   java_lang_Throwable::set_message(Universe::_out_of_memory_error_realloc_objects, msg());
 950 
 951   msg = java_lang_String::create_from_str("Java heap space: failed retryable allocation", CHECK_false);
 952   java_lang_Throwable::set_message(Universe::_out_of_memory_error_retry, msg());
 953 
 954   msg = java_lang_String::create_from_str("/ by zero", CHECK_false);
 955   java_lang_Throwable::set_message(Universe::_arithmetic_exception_instance, msg());
 956 
 957   // Setup the array of errors that have preallocated backtrace
 958   k = Universe::_out_of_memory_error_java_heap->klass();
 959   assert(k->name() == vmSymbols::java_lang_OutOfMemoryError(), "should be out of memory error");
 960   ik = InstanceKlass::cast(k);
 961 
 962   int len = (StackTraceInThrowable) ? (int)PreallocatedOutOfMemoryErrorCount : 0;
 963   Universe::_preallocated_out_of_memory_error_array = oopFactory::new_objArray(ik, len, CHECK_false);
 964   for (int i=0; i<len; i++) {
 965     oop err = ik->allocate_instance(CHECK_false);
 966     Handle err_h = Handle(THREAD, err);
 967     java_lang_Throwable::allocate_backtrace(err_h, CHECK_false);
 968     Universe::preallocated_out_of_memory_errors()->obj_at_put(i, err_h());
 969   }
 970   Universe::_preallocated_out_of_memory_error_avail_count = (jint)len;
 971 
 972   Universe::initialize_known_methods(CHECK_false);
 973 
 974   // This needs to be done before the first scavenge/gc, since
 975   // it's an input to soft ref clearing policy.
 976   {
 977     MutexLocker x(THREAD, Heap_lock);
 978     Universe::update_heap_info_at_gc();
 979   }
 980 
 981   // ("weak") refs processing infrastructure initialization
 982   Universe::heap()->post_initialize();
 983 
 984   MemoryService::add_metaspace_memory_pools();
 985 
 986   MemoryService::set_universe_heap(Universe::heap());
 987 #if INCLUDE_CDS
 988   MetaspaceShared::post_initialize(CHECK_false);
 989 #endif
 990   return true;
 991 }

1032 void Universe::initialize_verify_flags() {
1033   verify_flags = 0;
1034   const char delimiter[] = " ,";
1035 
1036   size_t length = strlen(VerifySubSet);
1037   char* subset_list = NEW_C_HEAP_ARRAY(char, length + 1, mtInternal);
1038   strncpy(subset_list, VerifySubSet, length + 1);
1039   char* save_ptr;
1040 
1041   char* token = strtok_r(subset_list, delimiter, &save_ptr);
1042   while (token != NULL) {
1043     if (strcmp(token, "threads") == 0) {
1044       verify_flags |= Verify_Threads;
1045     } else if (strcmp(token, "heap") == 0) {
1046       verify_flags |= Verify_Heap;
1047     } else if (strcmp(token, "symbol_table") == 0) {
1048       verify_flags |= Verify_SymbolTable;
1049     } else if (strcmp(token, "string_table") == 0) {
1050       verify_flags |= Verify_StringTable;
1051     } else if (strcmp(token, "codecache") == 0) {
1052       verify_flags |= Verify_CodeCache;
1053     } else if (strcmp(token, "dictionary") == 0) {
1054       verify_flags |= Verify_SystemDictionary;
1055     } else if (strcmp(token, "classloader_data_graph") == 0) {
1056       verify_flags |= Verify_ClassLoaderDataGraph;
1057     } else if (strcmp(token, "metaspace") == 0) {
1058       verify_flags |= Verify_MetaspaceUtils;
1059     } else if (strcmp(token, "jni_handles") == 0) {
1060       verify_flags |= Verify_JNIHandles;
1061     } else if (strcmp(token, "codecache_oops") == 0) {
1062       verify_flags |= Verify_CodeCacheOops;
1063     } else if (strcmp(token, "resolved_method_table") == 0) {
1064       verify_flags |= Verify_ResolvedMethodTable;
1065     } else {
1066       vm_exit_during_initialization(err_msg("VerifySubSet: \'%s\' memory sub-system is unknown, please correct it", token));
1067     }
1068     token = strtok_r(NULL, delimiter, &save_ptr);
1069   }
1070   FREE_C_HEAP_ARRAY(char, subset_list);

1080 void Universe::verify(VerifyOption option, const char* prefix) {
1081   // The use of _verify_in_progress is a temporary work around for
1082   // 6320749.  Don't bother with a creating a class to set and clear
1083   // it since it is only used in this method and the control flow is
1084   // straight forward.
1085   _verify_in_progress = true;
1086 
1087   COMPILER2_PRESENT(
1088     assert(!DerivedPointerTable::is_active(),
1089          "DPT should not be active during verification "
1090          "(of thread stacks below)");
1091   )
1092 
1093   ResourceMark rm;
1094   HandleMark hm;  // Handles created during verification can be zapped
1095   _verify_count++;
1096 
1097   FormatBuffer<> title("Verifying %s", prefix);
1098   GCTraceTime(Info, gc, verify) tm(title.buffer());
1099   if (should_verify_subset(Verify_Threads)) {
1100     log_debug(gc, verify)("Threads");
1101     Threads::verify();
1102   }
1103   if (should_verify_subset(Verify_Heap)) {
1104     log_debug(gc, verify)("Heap");
1105     heap()->verify(option);
1106   }
1107   if (should_verify_subset(Verify_SymbolTable)) {
1108     log_debug(gc, verify)("SymbolTable");
1109     SymbolTable::verify();
1110   }
1111   if (should_verify_subset(Verify_StringTable)) {
1112     log_debug(gc, verify)("StringTable");
1113     StringTable::verify();
1114   }
1115   if (should_verify_subset(Verify_CodeCache)) {
1116   {
1117     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1118     log_debug(gc, verify)("CodeCache");
1119     CodeCache::verify();
1120   }
1121   }
1122   if (should_verify_subset(Verify_SystemDictionary)) {
1123     log_debug(gc, verify)("SystemDictionary");
1124     SystemDictionary::verify();
1125   }
1126 #ifndef PRODUCT
1127   if (should_verify_subset(Verify_ClassLoaderDataGraph)) {
1128     log_debug(gc, verify)("ClassLoaderDataGraph");
1129     ClassLoaderDataGraph::verify();
1130   }
1131 #endif
1132   if (should_verify_subset(Verify_MetaspaceUtils)) {
1133     log_debug(gc, verify)("MetaspaceUtils");
1134     MetaspaceUtils::verify_free_chunks();
1135   }
1136   if (should_verify_subset(Verify_JNIHandles)) {
1137     log_debug(gc, verify)("JNIHandles");
1138     JNIHandles::verify();
1139   }
1140   if (should_verify_subset(Verify_CodeCacheOops)) {
1141     log_debug(gc, verify)("CodeCache Oops");
1142     CodeCache::verify_oops();
1143   }
1144   if (should_verify_subset(Verify_ResolvedMethodTable)) {
1145     log_debug(gc, verify)("ResolvedMethodTable Oops");
1146     ResolvedMethodTable::verify();
1147   }
1148 
1149   _verify_in_progress = false;
1150 }

1153 #ifndef PRODUCT
1154 void Universe::calculate_verify_data(HeapWord* low_boundary, HeapWord* high_boundary) {
1155   assert(low_boundary < high_boundary, "bad interval");
1156 
1157   // decide which low-order bits we require to be clear:
1158   size_t alignSize = MinObjAlignmentInBytes;
1159   size_t min_object_size = CollectedHeap::min_fill_size();
1160 
1161   // make an inclusive limit:
1162   uintptr_t max = (uintptr_t)high_boundary - min_object_size*wordSize;
1163   uintptr_t min = (uintptr_t)low_boundary;
1164   assert(min < max, "bad interval");
1165   uintptr_t diff = max ^ min;
1166 
1167   // throw away enough low-order bits to make the diff vanish
1168   uintptr_t mask = (uintptr_t)(-1);
1169   while ((mask & diff) != 0)
1170     mask <<= 1;
1171   uintptr_t bits = (min & mask);
1172   assert(bits == (max & mask), "correct mask");
1173   // check an intermediate value between min and max, just to make sure:
1174   assert(bits == ((min + (max-min)/2) & mask), "correct mask");
1175 
1176   // require address alignment, too:
1177   mask |= (alignSize - 1);
1178 
1179   if (!(_verify_oop_mask == 0 && _verify_oop_bits == (uintptr_t)-1)) {
1180     assert(_verify_oop_mask == mask && _verify_oop_bits == bits, "mask stability");
1181   }
1182   _verify_oop_mask = mask;
1183   _verify_oop_bits = bits;
1184 }
