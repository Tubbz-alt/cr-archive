24 /**
25  * @test
26  * @bug 8244982
27  * @summary Javac has trouble compiling method references
28  * @run main StreamsTest
29  */
30 
31 import java.util.Arrays;
32 
33 public class StreamsTest {
34 
35     public static inline class X {
36 
37         String data;
38 
39         X(String data) {
40             this.data = data;
41         }
42 
43         String data() { return data; }
44 
45         static String accumulate = "";
46 
47         static void accumulate(String s) {
48             accumulate += s;
49         }
50 
51         static String streamedData() {
52 
53             X [] xs = new X[] {
54                                  new X("Streams "),
55                                  new X("test "),
56                                  new X("passed OK!")
57                       };
58 
59             Arrays.stream(xs)
60                         .map(X.ref::data)
61                         .filter(p -> p != null)
62                         .forEach(X::accumulate);
63 
64             return accumulate;
65         }
66     }
67 
68     public static void main(String [] args) {
69         if (!X.streamedData().equals("Streams test passed OK!"))
70             throw new AssertionError("Unexpected data in stream");
71     }
72 }
