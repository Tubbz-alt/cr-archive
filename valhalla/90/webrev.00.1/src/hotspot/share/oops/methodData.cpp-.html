  50 // Some types of data layouts need a length field.
  51 bool DataLayout::needs_array_len(u1 tag) {
  52   return (tag == multi_branch_data_tag) || (tag == arg_info_data_tag) || (tag == parameters_type_data_tag);
  53 }

  55 // Perform generic initialization of the data.  More specific
  56 // initialization occurs in overrides of ProfileData::post_initialize.
  57 void DataLayout::initialize(u1 tag, u2 bci, int cell_count) {
  58   _header._bits = (intptr_t)0;
  59   _header._struct._tag = tag;
  60   _header._struct._bci = bci;
  61   for (int i = 0; i < cell_count; i++) {
  62     set_cell_at(i, (intptr_t)0);
  63   }
  64   if (needs_array_len(tag)) {
  65     set_cell_at(ArrayData::array_len_off_set, cell_count - 1); // -1 for header.
  66   }
  67   if (tag == call_type_data_tag) {
  68     CallTypeData::initialize(this, cell_count);
  69   } else if (tag == virtual_call_type_data_tag) {
  70     VirtualCallTypeData::initialize(this, cell_count);
  71   }
  72 }

  91 char* ProfileData::print_data_on_helper(const MethodData* md) const {
  92   DataLayout* dp  = md->extra_data_base();
  93   DataLayout* end = md->args_data_limit();
  94   stringStream ss;
  95   for (;; dp = MethodData::next_extra(dp)) {
  96     assert(dp < end, "moved past end of extra data");
  97     switch(dp->tag()) {
  98     case DataLayout::speculative_trap_data_tag:
  99       if (dp->bci() == bci()) {
 100         SpeculativeTrapData* data = new SpeculativeTrapData(dp);
 101         int trap = data->trap_state();
 102         char buf[100];
 103         ss.print("trap/");
 104         data->method()->print_short_name(&ss);
 105         ss.print("(%s) ", Deoptimization::format_trap_state(buf, sizeof(buf), trap));
 106       }
 107       break;
 108     case DataLayout::bit_data_tag:
 109       break;
 110     case DataLayout::no_tag:
 111     case DataLayout::arg_info_data_tag:
 112       return ss.as_string();
 113       break;
 114     default:
 115       fatal("unexpected tag %d", dp->tag());
 116     }
 117   }
 118   return NULL;
 119 }

 125 void ProfileData::print_shared(outputStream* st, const char* name, const char* extra) const {
 126   st->print("bci: %d", bci());
 127   st->fill_to(tab_width_one);
 128   st->print("%s", name);
 129   tab(st);
 130   int trap = trap_state();
 131   if (trap != 0) {
 132     char buf[100];
 133     st->print("trap(%s) ", Deoptimization::format_trap_state(buf, sizeof(buf), trap));
 134   }
 135   if (extra != NULL) {
 136     st->print("%s", extra);
 137   }
 138   int flags = data()->flags();
 139   if (flags != 0) {
 140     st->print("flags(%d) ", flags);
 141   }
 142 }

 170 // ==================================================================
 171 // JumpData
 172 //
 173 // A JumpData is used to access profiling information for a direct
 174 // branch.  It is a counter, used for counting the number of branches,
 175 // plus a data displacement, used for realigning the data pointer to
 176 // the corresponding target bci.
 177 
 178 void JumpData::post_initialize(BytecodeStream* stream, MethodData* mdo) {
 179   assert(stream->bci() == bci(), "wrong pos");
 180   int target;
 181   Bytecodes::Code c = stream->code();
 182   if (c == Bytecodes::_goto_w || c == Bytecodes::_jsr_w) {
 183     target = stream->dest_w();
 184   } else {
 185     target = stream->dest();
 186   }
 187   int my_di = mdo->dp_to_di(dp());
 188   int target_di = mdo->bci_to_di(target);
 189   int offset = target_di - my_di;
 190   set_displacement(offset);
 191 }

 194   print_shared(st, "JumpData", extra);
 195   st->print_cr("taken(%u) displacement(%d)", taken(), displacement());
 196 }

 198 int TypeStackSlotEntries::compute_cell_count(Symbol* signature, bool include_receiver, int max) {
 199   // Parameter profiling include the receiver
 200   int args_count = include_receiver ? 1 : 0;
 201   ResourceMark rm;
 202   ReferenceArgumentCount rac(signature);
 203   args_count += rac.count();
 204   args_count = MIN2(args_count, max);
 205   return args_count * per_arg_cell_count;
 206 }

 208 int TypeEntriesAtCall::compute_cell_count(BytecodeStream* stream) {
 209   assert(Bytecodes::is_invoke(stream->code()), "should be invoke");
 210   assert(TypeStackSlotEntries::per_arg_count() > ReturnTypeEntry::static_cell_count(), "code to test for arguments/results broken");
 211   const methodHandle m = stream->method();
 212   int bci = stream->bci();
 213   Bytecode_invoke inv(m, bci);
 214   int args_cell = 0;
 215   if (MethodData::profile_arguments_for_invoke(m, bci)) {
 216     args_cell = TypeStackSlotEntries::compute_cell_count(inv.signature(), false, TypeProfileArgsLimit);
 217   }
 218   int ret_cell = 0;
 219   if (MethodData::profile_return_for_invoke(m, bci) && is_reference_type(inv.result_type())) {
 220     ret_cell = ReturnTypeEntry::static_cell_count();
 221   }
 222   int header_cell = 0;
 223   if (args_cell + ret_cell > 0) {
 224     header_cell = header_cell_count();
 225   }
 226 
 227   return header_cell + args_cell + ret_cell;
 228 }

 230 class ArgumentOffsetComputer : public SignatureIterator {
 231 private:
 232   int _max;
 233   int _offset;
 234   GrowableArray<int> _offsets;
 235 
 236   friend class SignatureIterator;  // so do_parameters_on can call do_type
 237   void do_type(BasicType type) {
 238     if (is_reference_type(type) && _offsets.length() < _max) {
 239       _offsets.push(_offset);
 240     }
 241     _offset += parameter_type_word_count(type);
 242   }
 243 
 244  public:
 245   ArgumentOffsetComputer(Symbol* signature, int max)
 246     : SignatureIterator(signature),
 247       _max(max), _offset(0),
 248       _offsets(max) {
 249     do_parameters_on(this);  // non-virtual template execution
 250   }
 251 
 252   int off_at(int i) const { return _offsets.at(i); }
 253 };

 255 void TypeStackSlotEntries::post_initialize(Symbol* signature, bool has_receiver, bool include_receiver) {
 256   ResourceMark rm;
 257   int start = 0;
 258   // Parameter profiling include the receiver
 259   if (include_receiver && has_receiver) {
 260     set_stack_slot(0, 0);
 261     set_type(0, type_none());
 262     start += 1;
 263   }
 264   ArgumentOffsetComputer aos(signature, _number_of_entries-start);
 265   for (int i = start; i < _number_of_entries; i++) {
 266     set_stack_slot(i, aos.off_at(i-start) + (has_receiver ? 1 : 0));
 267     set_type(i, type_none());
 268   }
 269 }

 271 void CallTypeData::post_initialize(BytecodeStream* stream, MethodData* mdo) {
 272   assert(Bytecodes::is_invoke(stream->code()), "should be invoke");
 273   Bytecode_invoke inv(stream->method(), stream->bci());
 274 
 275   if (has_arguments()) {
 276 #ifdef ASSERT
 277     ResourceMark rm;
 278     ReferenceArgumentCount rac(inv.signature());
 279     int count = MIN2(rac.count(), (int)TypeProfileArgsLimit);
 280     assert(count > 0, "room for args type but none found?");
 281     check_number_of_arguments(count);
 282 #endif
 283     _args.post_initialize(inv.signature(), inv.has_receiver(), false);
 284   }
 285 
 286   if (has_return()) {
 287     assert(is_reference_type(inv.result_type()), "room for a ret type but doesn't return obj?");
 288     _ret.post_initialize();
 289   }
 290 }

 400 // that the check is reached, and a series of (Klass*, count) pairs

 403 void ReceiverTypeData::clean_weak_klass_links(bool always_clean) {
 404     for (uint row = 0; row < row_limit(); row++) {
 405     Klass* p = receiver(row);
 406     if (p != NULL && (always_clean || !p->is_loader_alive())) {
 407       clear_row(row);
 408     }
 409   }
 410 }

 412 void ReceiverTypeData::print_receiver_data_on(outputStream* st) const {
 413   uint row;
 414   int entries = 0;
 415   for (row = 0; row < row_limit(); row++) {
 416     if (receiver(row) != NULL)  entries++;
 417   }
 418 #if INCLUDE_JVMCI
 419   st->print_cr("count(%u) nonprofiled_count(%u) entries(%u)", count(), nonprofiled_count(), entries);
 420 #else
 421   st->print_cr("count(%u) entries(%u)", count(), entries);
 422 #endif
 423   int total = count();
 424   for (row = 0; row < row_limit(); row++) {
 425     if (receiver(row) != NULL) {
 426       total += receiver_count(row);
 427     }
 428   }
 429   for (row = 0; row < row_limit(); row++) {
 430     if (receiver(row) != NULL) {
 431       tab(st);
 432       receiver(row)->print_value_on(st);
 433       st->print_cr("(%u %4.2f)", receiver_count(row), (float) receiver_count(row) / (float) total);
 434     }
 435   }
 436 }

 439   print_receiver_data_on(st);

 447 // ==================================================================
 448 // RetData
 449 //
 450 // A RetData is used to access profiling information for a ret bytecode.
 451 // It is composed of a count of the number of times that the ret has
 452 // been executed, followed by a series of triples of the form
 453 // (bci, count, di) which count the number of times that some bci was the
 454 // target of the ret and cache a corresponding displacement.
 455 
 456 void RetData::post_initialize(BytecodeStream* stream, MethodData* mdo) {
 457   for (uint row = 0; row < row_limit(); row++) {
 458     set_bci_displacement(row, -1);
 459     set_bci(row, no_bci);
 460   }
 461   // release so other threads see a consistent state.  bci is used as
 462   // a valid flag for bci_displacement.
 463   OrderAccess::release();
 464 }
 465 
 466 // This routine needs to atomically update the RetData structure, so the
 467 // caller needs to hold the RetData_lock before it gets here.  Since taking
 468 // the lock can block (and allow GC) and since RetData is a ProfileData is a
 469 // wrapper around a derived oop, taking the lock in _this_ method will
 470 // basically cause the 'this' pointer's _data field to contain junk after the
 471 // lock.  We require the caller to take the lock before making the ProfileData
 472 // structure.  Currently the only caller is InterpreterRuntime::update_mdp_for_ret
 473 address RetData::fixup_ret(int return_bci, MethodData* h_mdo) {
 474   // First find the mdp which corresponds to the return bci.
 475   address mdp = h_mdo->bci_to_dp(return_bci);
 476 
 477   // Now check to see if any of the cache slots are open.
 478   for (uint row = 0; row < row_limit(); row++) {
 479     if (bci(row) == no_bci) {
 480       set_bci_displacement(row, mdp - dp());
 481       set_bci_count(row, DataLayout::counter_increment);
 482       // Barrier to ensure displacement is written before the bci; allows
 483       // the interpreter to read displacement without fear of race condition.
 484       release_set_bci(row, return_bci);
 485       break;
 486     }
 487   }
 488   return mdp;
 489 }
 490 
 491 void RetData::print_data_on(outputStream* st, const char* extra) const {
 492   print_shared(st, "RetData", extra);
 493   uint row;
 494   int entries = 0;
 495   for (row = 0; row < row_limit(); row++) {
 496     if (bci(row) != no_bci)  entries++;
 497   }
 498   st->print_cr("count(%u) entries(%u)", count(), entries);
 499   for (row = 0; row < row_limit(); row++) {
 500     if (bci(row) != no_bci) {
 501       tab(st);
 502       st->print_cr("bci(%d: count(%u) displacement(%d))",
 503                    bci(row), bci_count(row), bci_displacement(row));
 504     }
 505   }
 506 }

 532 // ==================================================================
 533 // MultiBranchData
 534 //
 535 // A MultiBranchData is used to access profiling information for
 536 // a multi-way branch (*switch bytecodes).  It consists of a series
 537 // of (count, displacement) pairs, which count the number of times each
 538 // case was taken and specify the data displacment for each branch target.
 539 
 540 int MultiBranchData::compute_cell_count(BytecodeStream* stream) {
 541   int cell_count = 0;
 542   if (stream->code() == Bytecodes::_tableswitch) {
 543     Bytecode_tableswitch sw(stream->method()(), stream->bcp());
 544     cell_count = 1 + per_case_cell_count * (1 + sw.length()); // 1 for default
 545   } else {
 546     Bytecode_lookupswitch sw(stream->method()(), stream->bcp());
 547     cell_count = 1 + per_case_cell_count * (sw.number_of_pairs() + 1); // 1 for default
 548   }
 549   return cell_count;
 550 }
 551 
 552 void MultiBranchData::post_initialize(BytecodeStream* stream,
 553                                       MethodData* mdo) {
 554   assert(stream->bci() == bci(), "wrong pos");
 555   int target;
 556   int my_di;
 557   int target_di;
 558   int offset;
 559   if (stream->code() == Bytecodes::_tableswitch) {
 560     Bytecode_tableswitch sw(stream->method()(), stream->bcp());
 561     int len = sw.length();
 562     assert(array_len() == per_case_cell_count * (len + 1), "wrong len");
 563     for (int count = 0; count < len; count++) {
 564       target = sw.dest_offset_at(count) + bci();
 565       my_di = mdo->dp_to_di(dp());
 566       target_di = mdo->bci_to_di(target);
 567       offset = target_di - my_di;
 568       set_displacement_at(count, offset);
 569     }
 570     target = sw.default_offset() + bci();
 571     my_di = mdo->dp_to_di(dp());
 572     target_di = mdo->bci_to_di(target);
 573     offset = target_di - my_di;
 574     set_default_displacement(offset);
 575 
 576   } else {
 577     Bytecode_lookupswitch sw(stream->method()(), stream->bcp());
 578     int npairs = sw.number_of_pairs();
 579     assert(array_len() == per_case_cell_count * (npairs + 1), "wrong len");
 580     for (int count = 0; count < npairs; count++) {
 581       LookupswitchPair pair = sw.pair_at(count);
 582       target = pair.offset() + bci();
 583       my_di = mdo->dp_to_di(dp());
 584       target_di = mdo->bci_to_di(target);
 585       offset = target_di - my_di;
 586       set_displacement_at(count, offset);
 587     }
 588     target = sw.default_offset() + bci();
 589     my_di = mdo->dp_to_di(dp());
 590     target_di = mdo->bci_to_di(target);
 591     offset = target_di - my_di;
 592     set_default_displacement(offset);
 593   }
 594 }

 617 int ParametersTypeData::compute_cell_count(Method* m) {
 618   if (!MethodData::profile_parameters_for_method(methodHandle(Thread::current(), m))) {
 619     return 0;
 620   }
 621   int max = TypeProfileParmsLimit == -1 ? INT_MAX : TypeProfileParmsLimit;
 622   int obj_args = TypeStackSlotEntries::compute_cell_count(m->signature(), !m->is_static(), max);

 662 int MethodData::bytecode_cell_count(Bytecodes::Code code) {
 663   if (is_client_compilation_mode_vm()) {
 664     return no_profile_data;
 665   }
 666   switch (code) {
 667   case Bytecodes::_checkcast:
 668   case Bytecodes::_instanceof:
 669   case Bytecodes::_aastore:
 670     if (TypeProfileCasts) {
 671       return ReceiverTypeData::static_cell_count();
 672     } else {
 673       return BitData::static_cell_count();
 674     }
 675   case Bytecodes::_invokespecial:
 676   case Bytecodes::_invokestatic:
 677     if (MethodData::profile_arguments() || MethodData::profile_return()) {
 678       return variable_cell_count;
 679     } else {
 680       return CounterData::static_cell_count();
 681     }
 682   case Bytecodes::_goto:
 683   case Bytecodes::_goto_w:
 684   case Bytecodes::_jsr:
 685   case Bytecodes::_jsr_w:
 686     return JumpData::static_cell_count();
 687   case Bytecodes::_invokevirtual:
 688   case Bytecodes::_invokeinterface:
 689     if (MethodData::profile_arguments() || MethodData::profile_return()) {
 690       return variable_cell_count;
 691     } else {
 692       return VirtualCallData::static_cell_count();
 693     }
 694   case Bytecodes::_invokedynamic:
 695     if (MethodData::profile_arguments() || MethodData::profile_return()) {
 696       return variable_cell_count;
 697     } else {
 698       return CounterData::static_cell_count();
 699     }
 700   case Bytecodes::_ret:
 701     return RetData::static_cell_count();
 702   case Bytecodes::_ifeq:
 703   case Bytecodes::_ifne:
 704   case Bytecodes::_iflt:
 705   case Bytecodes::_ifge:
 706   case Bytecodes::_ifgt:
 707   case Bytecodes::_ifle:
 708   case Bytecodes::_if_icmpeq:
 709   case Bytecodes::_if_icmpne:
 710   case Bytecodes::_if_icmplt:
 711   case Bytecodes::_if_icmpge:
 712   case Bytecodes::_if_icmpgt:
 713   case Bytecodes::_if_icmple:
 714   case Bytecodes::_if_acmpeq:
 715   case Bytecodes::_if_acmpne:
 716   case Bytecodes::_ifnull:
 717   case Bytecodes::_ifnonnull:
 718     return BranchData::static_cell_count();
 719   case Bytecodes::_lookupswitch:
 720   case Bytecodes::_tableswitch:
 721     return variable_cell_count;
 722   default:
 723     return no_profile_data;
 724   }
 725 }

 727 // Compute the size of the profiling information corresponding to
 728 // the current bytecode.
 729 int MethodData::compute_data_size(BytecodeStream* stream) {
 730   int cell_count = bytecode_cell_count(stream->code());
 731   if (cell_count == no_profile_data) {
 732     return 0;
 733   }
 734   if (cell_count == variable_cell_count) {
 735     switch (stream->code()) {
 736     case Bytecodes::_lookupswitch:
 737     case Bytecodes::_tableswitch:
 738       cell_count = MultiBranchData::compute_cell_count(stream);
 739       break;
 740     case Bytecodes::_invokespecial:
 741     case Bytecodes::_invokestatic:
 742     case Bytecodes::_invokedynamic:
 743       assert(MethodData::profile_arguments() || MethodData::profile_return(), "should be collecting args profile");
 744       if (profile_arguments_for_invoke(stream->method(), stream->bci()) ||
 745           profile_return_for_invoke(stream->method(), stream->bci())) {
 746         cell_count = CallTypeData::compute_cell_count(stream);
 747       } else {
 748         cell_count = CounterData::static_cell_count();
 749       }
 750       break;
 751     case Bytecodes::_invokevirtual:
 752     case Bytecodes::_invokeinterface: {
 753       assert(MethodData::profile_arguments() || MethodData::profile_return(), "should be collecting args profile");
 754       if (profile_arguments_for_invoke(stream->method(), stream->bci()) ||
 755           profile_return_for_invoke(stream->method(), stream->bci())) {
 756         cell_count = VirtualCallTypeData::compute_cell_count(stream);
 757       } else {
 758         cell_count = VirtualCallData::static_cell_count();
 759       }
 760       break;
 761     }
 762     default:
 763       fatal("unexpected bytecode for var length profile data");
 764     }
 765   }
 766   // Note:  cell_count might be zero, meaning that there is just
 767   //        a DataLayout header, with no extra cells.
 768   assert(cell_count >= 0, "sanity");
 769   return DataLayout::compute_size_in_bytes(cell_count);
 770 }

 806 // A heuristic check to detect nmethods that outlive a failed speculations list.
 807 static void guarantee_failed_speculations_alive(nmethod* nm, FailedSpeculation** failed_speculations_address) {
 808   jlong head = (jlong)(address) *failed_speculations_address;
 809   if ((head & 0x1) == 0x1) {
 810     stringStream st;
 811     if (nm != NULL) {
 812       st.print("%d", nm->compile_id());

 814       st.print_raw("{");

 819         if (jvmci_name != NULL) {
 820           st.print_raw(jvmci_name);
 821         }
 822       }
 823       st.print_raw("}");
 824     } else {
 825       st.print("<unknown>");
 826     }
 827     fatal("Adding to failed speculations list that appears to have been freed. Source: %s", st.as_string());

 831 bool FailedSpeculation::add_failed_speculation(nmethod* nm, FailedSpeculation** failed_speculations_address, address speculation, int speculation_len) {
 832   assert(failed_speculations_address != NULL, "must be");
 833   size_t fs_size = sizeof(FailedSpeculation) + speculation_len;
 834   FailedSpeculation* fs = new (fs_size) FailedSpeculation(speculation, speculation_len);
 835   if (fs == NULL) {
 836     // no memory -> ignore failed speculation
 837     return false;
 838   }
 839 
 840   guarantee(is_aligned(fs, sizeof(FailedSpeculation*)), "FailedSpeculation objects must be pointer aligned");
 841   guarantee_failed_speculations_alive(nm, failed_speculations_address);
 842 
 843   FailedSpeculation** cursor = failed_speculations_address;
 844   do {
 845     if (*cursor == NULL) {
 846       FailedSpeculation* old_fs = Atomic::cmpxchg(cursor, (FailedSpeculation*) NULL, fs);
 847       if (old_fs == NULL) {
 848         // Successfully appended fs to end of the list
 849         return true;
 850       }
 851       cursor = old_fs->next_adr();
 852     } else {
 853       cursor = (*cursor)->next_adr();
 854     }
 855   } while (true);
 856 }

 858 void FailedSpeculation::free_failed_speculations(FailedSpeculation** failed_speculations_address) {
 859   assert(failed_speculations_address != NULL, "must be");
 860   FailedSpeculation* fs = *failed_speculations_address;
 861   while (fs != NULL) {
 862     FailedSpeculation* next = fs->next();
 863     delete fs;
 864     fs = next;
 865   }
 866 
 867   // Write an unaligned value to failed_speculations_address to denote
 868   // that it is no longer a valid pointer. This is allows for the check
 869   // in add_failed_speculation against adding to a freed failed
 870   // speculations list.
 871   long* head = (long*) failed_speculations_address;
 872   (*head) = (*head) | 0x1;
 873 }

 925 // Compute the size of the MethodData* necessary to store
 926 // profiling information about a given method.  Size is in bytes.
 927 int MethodData::compute_allocation_size_in_bytes(const methodHandle& method) {
 928   int data_size = 0;
 929   BytecodeStream stream(method);
 930   Bytecodes::Code c;
 931   int empty_bc_count = 0;  // number of bytecodes lacking data
 932   bool needs_speculative_traps = false;
 933   while ((c = stream.next()) >= 0) {
 934     int size_in_bytes = compute_data_size(&stream);
 935     data_size += size_in_bytes;
 936     if (size_in_bytes == 0 JVMCI_ONLY(&& Bytecodes::can_trap(c)))  empty_bc_count += 1;
 937     needs_speculative_traps = needs_speculative_traps || is_speculative_trap_bytecode(c);
 938   }
 939   int object_size = in_bytes(data_offset()) + data_size;
 940 
 941   // Add some extra DataLayout cells (at least one) to track stray traps.
 942   int extra_data_count = compute_extra_data_count(data_size, empty_bc_count, needs_speculative_traps);
 943   object_size += extra_data_count * DataLayout::compute_size_in_bytes(0);
 944 
 945   // Add a cell to record information about modified arguments.
 946   int arg_size = method->size_of_parameters();
 947   object_size += DataLayout::compute_size_in_bytes(arg_size+1);
 948 
 949   // Reserve room for an area of the MDO dedicated to profiling of
 950   // parameters
 951   int args_cell = ParametersTypeData::compute_cell_count(method());
 952   if (args_cell > 0) {
 953     object_size += DataLayout::compute_size_in_bytes(args_cell);
 954   }
 955   return object_size;
 956 }

 966 // Initialize an individual data segment.  Returns the size of
 967 // the segment in bytes.
 968 int MethodData::initialize_data(BytecodeStream* stream,
 969                                        int data_index) {
 970   if (is_client_compilation_mode_vm()) {
 971     return 0;
 972   }
 973   int cell_count = -1;
 974   int tag = DataLayout::no_tag;
 975   DataLayout* data_layout = data_layout_at(data_index);
 976   Bytecodes::Code c = stream->code();
 977   switch (c) {
 978   case Bytecodes::_checkcast:
 979   case Bytecodes::_instanceof:
 980   case Bytecodes::_aastore:
 981     if (TypeProfileCasts) {
 982       cell_count = ReceiverTypeData::static_cell_count();
 983       tag = DataLayout::receiver_type_data_tag;
 984     } else {
 985       cell_count = BitData::static_cell_count();
 986       tag = DataLayout::bit_data_tag;
 987     }
 988     break;
 989   case Bytecodes::_invokespecial:
 990   case Bytecodes::_invokestatic: {
 991     int counter_data_cell_count = CounterData::static_cell_count();
 992     if (profile_arguments_for_invoke(stream->method(), stream->bci()) ||
 993         profile_return_for_invoke(stream->method(), stream->bci())) {
 994       cell_count = CallTypeData::compute_cell_count(stream);
 995     } else {
 996       cell_count = counter_data_cell_count;
 997     }
 998     if (cell_count > counter_data_cell_count) {
 999       tag = DataLayout::call_type_data_tag;
1000     } else {
1001       tag = DataLayout::counter_data_tag;
1002     }
1003     break;
1004   }
1005   case Bytecodes::_goto:
1006   case Bytecodes::_goto_w:
1007   case Bytecodes::_jsr:
1008   case Bytecodes::_jsr_w:
1009     cell_count = JumpData::static_cell_count();
1010     tag = DataLayout::jump_data_tag;
1011     break;
1012   case Bytecodes::_invokevirtual:
1013   case Bytecodes::_invokeinterface: {
1014     int virtual_call_data_cell_count = VirtualCallData::static_cell_count();
1015     if (profile_arguments_for_invoke(stream->method(), stream->bci()) ||
1016         profile_return_for_invoke(stream->method(), stream->bci())) {
1017       cell_count = VirtualCallTypeData::compute_cell_count(stream);
1018     } else {
1019       cell_count = virtual_call_data_cell_count;
1020     }
1021     if (cell_count > virtual_call_data_cell_count) {
1022       tag = DataLayout::virtual_call_type_data_tag;
1023     } else {
1024       tag = DataLayout::virtual_call_data_tag;
1025     }
1026     break;
1027   }
1028   case Bytecodes::_invokedynamic: {
1029     // %%% should make a type profile for any invokedynamic that takes a ref argument
1030     int counter_data_cell_count = CounterData::static_cell_count();
1031     if (profile_arguments_for_invoke(stream->method(), stream->bci()) ||
1032         profile_return_for_invoke(stream->method(), stream->bci())) {
1033       cell_count = CallTypeData::compute_cell_count(stream);
1034     } else {
1035       cell_count = counter_data_cell_count;
1036     }
1037     if (cell_count > counter_data_cell_count) {
1038       tag = DataLayout::call_type_data_tag;
1039     } else {
1040       tag = DataLayout::counter_data_tag;
1041     }
1042     break;
1043   }
1044   case Bytecodes::_ret:
1045     cell_count = RetData::static_cell_count();
1046     tag = DataLayout::ret_data_tag;
1047     break;
1048   case Bytecodes::_ifeq:
1049   case Bytecodes::_ifne:
1050   case Bytecodes::_iflt:
1051   case Bytecodes::_ifge:
1052   case Bytecodes::_ifgt:
1053   case Bytecodes::_ifle:
1054   case Bytecodes::_if_icmpeq:
1055   case Bytecodes::_if_icmpne:
1056   case Bytecodes::_if_icmplt:
1057   case Bytecodes::_if_icmpge:
1058   case Bytecodes::_if_icmpgt:
1059   case Bytecodes::_if_icmple:
1060   case Bytecodes::_if_acmpeq:
1061   case Bytecodes::_if_acmpne:

1063   case Bytecodes::_ifnonnull:

1065     tag = DataLayout::branch_data_tag;
1066     break;
1067   case Bytecodes::_lookupswitch:
1068   case Bytecodes::_tableswitch:
1069     cell_count = MultiBranchData::compute_cell_count(stream);
1070     tag = DataLayout::multi_branch_data_tag;
1071     break;
1072   default:
1073     break;
1074   }
1075   assert(tag == DataLayout::multi_branch_data_tag ||
1076          ((MethodData::profile_arguments() || MethodData::profile_return()) &&
1077           (tag == DataLayout::call_type_data_tag ||
1078            tag == DataLayout::counter_data_tag ||
1079            tag == DataLayout::virtual_call_type_data_tag ||
1080            tag == DataLayout::virtual_call_data_tag)) ||
1081          cell_count == bytecode_cell_count(c), "cell counts must agree");
1082   if (cell_count >= 0) {
1083     assert(tag != DataLayout::no_tag, "bad tag");
1084     assert(bytecode_has_profile(c), "agree w/ BHP");
1085     data_layout->initialize(tag, stream->bci(), cell_count);
1086     return DataLayout::compute_size_in_bytes(cell_count);
1087   } else {
1088     assert(!bytecode_has_profile(c), "agree w/ !BHP");
1089     return 0;
1090   }
1091 }

1193 
1194   // Add some extra DataLayout cells (at least one) to track stray traps.
1195   int extra_data_count = compute_extra_data_count(data_size, empty_bc_count, needs_speculative_traps);
1196   int extra_size = extra_data_count * DataLayout::compute_size_in_bytes(0);
1197 
1198   // Let's zero the space for the extra data
1199   Copy::zero_to_bytes(((address)_data) + data_size, extra_size);
1200 
1201   // Add a cell to record information about modified arguments.
1202   // Set up _args_modified array after traps cells so that
1203   // the code for traps cells works.
1204   DataLayout *dp = data_layout_at(data_size + extra_size);
1205 
1206   int arg_size = method()->size_of_parameters();
1207   dp->initialize(DataLayout::arg_info_data_tag, 0, arg_size+1);
1208 
1209   int arg_data_size = DataLayout::compute_size_in_bytes(arg_size+1);
1210   object_size += extra_size + arg_data_size;
1211 
1212   int parms_cell = ParametersTypeData::compute_cell_count(method());
1213   // If we are profiling parameters, we reserver an area near the end
1214   // of the MDO after the slots for bytecodes (because there's no bci
1215   // for method entry so they don't fit with the framework for the
1216   // profiling of bytecodes). We store the offset within the MDO of
1217   // this area (or -1 if no parameter is profiled)
1218   if (parms_cell > 0) {
1219     object_size += DataLayout::compute_size_in_bytes(parms_cell);
1220     _parameters_type_data_di = data_size + extra_size + arg_data_size;
1221     DataLayout *dp = data_layout_at(data_size + extra_size + arg_data_size);
1222     dp->initialize(DataLayout::parameters_type_data_tag, 0, parms_cell);
1223   } else {
1224     _parameters_type_data_di = no_parameters;
1225   }
1226 
1227   // Set an initial hint. Don't use set_hint_di() because
1228   // first_di() may be out of bounds if data_size is 0.
1229   // In that situation, _hint_di is never used, but at
1230   // least well-defined.
1231   _hint_di = first_di();
1232 
1233   post_initialize(&stream);
1234 
1235   assert(object_size == compute_allocation_size_in_bytes(methodHandle(thread, _method)), "MethodData: computed size != initialized size");
1236   set_size(object_size);
1237 }

1287 // Get a measure of how much mileage the method has on it.
1288 int MethodData::mileage_of(Method* method) {
1289   int mileage = 0;
1290   if (TieredCompilation) {
1291     mileage = MAX2(method->invocation_count(), method->backedge_count());
1292   } else {
1293     int iic = method->interpreter_invocation_count();
1294     if (mileage < iic)  mileage = iic;
1295     MethodCounters* mcs = method->method_counters();
1296     if (mcs != NULL) {
1297       InvocationCounter* ic = mcs->invocation_counter();
1298       InvocationCounter* bc = mcs->backedge_counter();
1299       int icval = ic->count();
1300       if (ic->carry()) icval += CompileThreshold;
1301       if (mileage < icval)  mileage = icval;
1302       int bcval = bc->count();
1303       if (bc->carry()) bcval += CompileThreshold;
1304       if (mileage < bcval)  mileage = bcval;
1305     }
1306   }
1307   return mileage;
1308 }

1344 DataLayout* MethodData::next_extra(DataLayout* dp) {
1345   int nb_cells = 0;
1346   switch(dp->tag()) {
1347   case DataLayout::bit_data_tag:
1348   case DataLayout::no_tag:
1349     nb_cells = BitData::static_cell_count();
1350     break;
1351   case DataLayout::speculative_trap_data_tag:

1353     break;
1354   default:

1356   }
1357   return (DataLayout*)((address)dp + DataLayout::compute_size_in_bytes(nb_cells));

1360 ProfileData* MethodData::bci_to_extra_data_helper(int bci, Method* m, DataLayout*& dp, bool concurrent) {
1361   DataLayout* end = args_data_limit();
1362 
1363   for (;; dp = next_extra(dp)) {
1364     assert(dp < end, "moved past end of extra data");
1365     // No need for "Atomic::load_acquire" ops,
1366     // since the data structure is monotonic.
1367     switch(dp->tag()) {
1368     case DataLayout::no_tag:
1369       return NULL;
1370     case DataLayout::arg_info_data_tag:
1371       dp = end;
1372       return NULL; // ArgInfoData is at the end of extra data section.
1373     case DataLayout::bit_data_tag:
1374       if (m == NULL && dp->bci() == bci) {
1375         return new BitData(dp);
1376       }
1377       break;
1378     case DataLayout::speculative_trap_data_tag:
1379       if (m != NULL) {
1380         SpeculativeTrapData* data = new SpeculativeTrapData(dp);
1381         // data->method() may be null in case of a concurrent
1382         // allocation. Maybe it's for the same method. Try to use that
1383         // entry in that case.
1384         if (dp->bci() == bci) {
1385           if (data->method() == NULL) {
1386             assert(concurrent, "impossible because no concurrent allocation");
1387             return NULL;
1388           } else if (data->method() == m) {
1389             return data;
1390           }
1391         }
1392       }
1393       break;
1394     default:
1395       fatal("unexpected tag %d", dp->tag());
1396     }
1397   }
1398   return NULL;
1399 }

1402 // Translate a bci to its corresponding extra data, or NULL.
1403 ProfileData* MethodData::bci_to_extra_data(int bci, Method* m, bool create_if_missing) {
1404   // This code assumes an entry for a SpeculativeTrapData is 2 cells
1405   assert(2*DataLayout::compute_size_in_bytes(BitData::static_cell_count()) ==
1406          DataLayout::compute_size_in_bytes(SpeculativeTrapData::static_cell_count()),
1407          "code needs to be adjusted");
1408 
1409   // Do not create one of these if method has been redefined.
1410   if (m != NULL && m->is_old()) {
1411     return NULL;
1412   }
1413 
1414   DataLayout* dp  = extra_data_base();
1415   DataLayout* end = args_data_limit();
1416 
1417   // Allocation in the extra data space has to be atomic because not
1418   // all entries have the same size and non atomic concurrent
1419   // allocation would result in a corrupted extra data space.
1420   ProfileData* result = bci_to_extra_data_helper(bci, m, dp, true);
1421   if (result != NULL) {
1422     return result;
1423   }
1424 
1425   if (create_if_missing && dp < end) {
1426     MutexLocker ml(&_extra_data_lock);
1427     // Check again now that we have the lock. Another thread may
1428     // have added extra data entries.
1429     ProfileData* result = bci_to_extra_data_helper(bci, m, dp, false);
1430     if (result != NULL || dp >= end) {
1431       return result;
1432     }
1433 
1434     assert(dp->tag() == DataLayout::no_tag || (dp->tag() == DataLayout::speculative_trap_data_tag && m != NULL), "should be free");
1435     assert(next_extra(dp)->tag() == DataLayout::no_tag || next_extra(dp)->tag() == DataLayout::arg_info_data_tag, "should be free or arg info");
1436     u1 tag = m == NULL ? DataLayout::bit_data_tag : DataLayout::speculative_trap_data_tag;
1437     // SpeculativeTrapData is 2 slots. Make sure we have room.
1438     if (m != NULL && next_extra(dp)->tag() != DataLayout::no_tag) {
1439       return NULL;
1440     }
1441     DataLayout temp;
1442     temp.initialize(tag, bci, 0);
1443 
1444     dp->set_header(temp.header());
1445     assert(dp->tag() == tag, "sane");
1446     assert(dp->bci() == bci, "no concurrent allocation");
1447     if (tag == DataLayout::bit_data_tag) {
1448       return new BitData(dp);
1449     } else {
1450       SpeculativeTrapData* data = new SpeculativeTrapData(dp);
1451       data->set_method(m);
1452       return data;
1453     }
1454   }
1455   return NULL;
1456 }

1658 void MethodData::clean_extra_data_helper(DataLayout* dp, int shift, bool reset) {
1659   if (shift == 0) {
1660     return;
1661   }
1662   if (!reset) {
1663     // Move all cells of trap entry at dp left by "shift" cells
1664     intptr_t* start = (intptr_t*)dp;
1665     intptr_t* end = (intptr_t*)next_extra(dp);
1666     for (intptr_t* ptr = start; ptr < end; ptr++) {
1667       *(ptr-shift) = *ptr;
1668     }
1669   } else {
1670     // Reset "shift" cells stopping at dp
1671     intptr_t* start = ((intptr_t*)dp) - shift;
1672     intptr_t* end = (intptr_t*)dp;
1673     for (intptr_t* ptr = start; ptr < end; ptr++) {
1674       *ptr = 0;
1675     }
1676   }
1677 }

1697 // Remove SpeculativeTrapData entries that reference an unloaded or
1698 // redefined method
1699 void MethodData::clean_extra_data(CleanExtraDataClosure* cl) {
1700   DataLayout* dp  = extra_data_base();
1701   DataLayout* end = args_data_limit();
1702 
1703   int shift = 0;
1704   for (; dp < end; dp = next_extra(dp)) {
1705     switch(dp->tag()) {
1706     case DataLayout::speculative_trap_data_tag: {
1707       SpeculativeTrapData* data = new SpeculativeTrapData(dp);
1708       Method* m = data->method();
1709       assert(m != NULL, "should have a method");
1710       if (!cl->is_live(m)) {
1711         // "shift" accumulates the number of cells for dead
1712         // SpeculativeTrapData entries that have been seen so
1713         // far. Following entries must be shifted left by that many
1714         // cells to remove the dead SpeculativeTrapData entries.
1715         shift += (int)((intptr_t*)next_extra(dp) - (intptr_t*)dp);
1716       } else {
1717         // Shift this entry left if it follows dead
1718         // SpeculativeTrapData entries
1719         clean_extra_data_helper(dp, shift);
1720       }
1721       break;
1722     }
1723     case DataLayout::bit_data_tag:
1724       // Shift this entry left if it follows dead SpeculativeTrapData
1725       // entries
1726       clean_extra_data_helper(dp, shift);
1727       continue;
1728     case DataLayout::no_tag:
1729     case DataLayout::arg_info_data_tag:
1730       // We are at end of the live trap entries. The previous "shift"
1731       // cells contain entries that are either dead or were shifted
1732       // left. They need to be reset to no_tag
1733       clean_extra_data_helper(dp, shift, true);
1734       return;
1735     default:
1736       fatal("unexpected tag %d", dp->tag());
1737     }
1738   }
1739 }
