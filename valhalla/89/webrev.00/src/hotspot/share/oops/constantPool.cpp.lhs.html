 112 void ConstantPool::deallocate_contents(ClassLoaderData* loader_data) {
 113   if (cache() != NULL) {
 114     MetadataFactory::free_metadata(loader_data, cache());
 115     set_cache(NULL);
 116   }
 117 
 118   MetadataFactory::free_array<Klass*>(loader_data, resolved_klasses());
 119   set_resolved_klasses(NULL);
 120 
 121   MetadataFactory::free_array<jushort>(loader_data, operands());
 122   set_operands(NULL);
 123 
 124   release_C_heap_structures();
 125 
 126   // free tag array
 127   MetadataFactory::free_array<u1>(loader_data, tags());
 128   set_tags(NULL);
 129 }

 136 void ConstantPool::metaspace_pointers_do(MetaspaceClosure* it) {
 137   log_trace(cds)("Iter(ConstantPool): %p", this);
 138 
 139   it->push(&_tags, MetaspaceClosure::_writable);
 140   it->push(&_cache);
 141   it->push(&_pool_holder);
 142   it->push(&_operands);
 143   it->push(&_resolved_klasses, MetaspaceClosure::_writable);
 144 
 145   for (int i = 0; i < length(); i++) {
 146     // The only MSO's embedded in the CP entries are Symbols:
 147     //   JVM_CONSTANT_String (normal and pseudo)
 148     //   JVM_CONSTANT_Utf8
 149     constantTag ctag = tag_at(i);
 150     if (ctag.is_string() || ctag.is_utf8()) {
 151       it->push(symbol_at_addr(i));
 152     }
 153   }
 154 }

 170 // Create resolved_references array and mapping array for original cp indexes
 171 // The ldc bytecode was rewritten to have the resolved reference array index so need a way
 172 // to map it back for resolving and some unlikely miscellaneous uses.
 173 // The objects created by invokedynamic are appended to this list.
 174 void ConstantPool::initialize_resolved_references(ClassLoaderData* loader_data,
 175                                                   const intStack& reference_map,
 176                                                   int constant_pool_map_length,
 177                                                   TRAPS) {
 178   // Initialized the resolved object cache.
 179   int map_length = reference_map.length();
 180   if (map_length > 0) {
 181     // Only need mapping back to constant pool entries.  The map isn't used for
 182     // invokedynamic resolved_reference entries.  For invokedynamic entries,
 183     // the constant pool cache index has the mapping back to both the constant
 184     // pool and to the resolved reference index.
 185     if (constant_pool_map_length > 0) {
 186       Array<u2>* om = MetadataFactory::new_array<u2>(loader_data, constant_pool_map_length, CHECK);
 187 
 188       for (int i = 0; i < constant_pool_map_length; i++) {
 189         int x = reference_map.at(i);
 190         assert(x == (int)(jushort) x, "klass index is too big");
 191         om->at_put(i, (jushort)x);
 192       }
 193       set_reference_map(om);
 194     }
 195 
 196     // Create Java array for holding resolved strings, methodHandles,
 197     // methodTypes, invokedynamic and invokehandle appendix objects, etc.
 198     objArrayOop stom = oopFactory::new_objArray(SystemDictionary::Object_klass(), map_length, CHECK);
 199     Handle refs_handle (THREAD, (oop)stom);  // must handleize.
 200     set_resolved_references(loader_data->add_handle(refs_handle));
 201   }
 202 }

 204 void ConstantPool::allocate_resolved_klasses(ClassLoaderData* loader_data, int num_klasses, TRAPS) {
 205   // A ConstantPool can't possibly have 0xffff valid class entries,
 206   // because entry #0 must be CONSTANT_Invalid, and each class entry must refer to a UTF8
 207   // entry for the class's name. So at most we will have 0xfffe class entries.
 208   // This allows us to use 0xffff (ConstantPool::_temp_resolved_klass_index) to indicate
 209   // UnresolvedKlass entries that are temporarily created during class redefinition.
 210   assert(num_klasses < CPKlassSlot::_temp_resolved_klass_index, "sanity");
 211   assert(resolved_klasses() == NULL, "sanity");
 212   Array<Klass*>* rk = MetadataFactory::new_array<Klass*>(loader_data, num_klasses, CHECK);
 213   set_resolved_klasses(rk);
 214 }

 216 void ConstantPool::initialize_unresolved_klasses(ClassLoaderData* loader_data, TRAPS) {
 217   int len = length();
 218   int num_klasses = 0;
 219   for (int i = 1; i <len; i++) {
 220     switch (tag_at(i).value()) {
 221     case JVM_CONSTANT_ClassIndex:
 222       {
 223         const int class_index = klass_index_at(i);
 224         unresolved_klass_at_put(i, class_index, num_klasses++);
 225       }
 226       break;
 227 #ifndef PRODUCT
 228     case JVM_CONSTANT_Class:
 229     case JVM_CONSTANT_UnresolvedClass:
 230     case JVM_CONSTANT_UnresolvedClassInError:
 231       // All of these should have been reverted back to Unresolved before calling
 232       // this function.
 233       ShouldNotReachHere();
 234 #endif
 235     }
 236   }
 237   allocate_resolved_klasses(loader_data, num_klasses, THREAD);
 238 }

 240 // Unsafe anonymous class support:
 241 void ConstantPool::klass_at_put(int class_index, int name_index, int resolved_klass_index, Klass* k, Symbol* name) {
 242   assert(is_within_bounds(class_index), "index out of bounds");
 243   assert(is_within_bounds(name_index), "index out of bounds");
 244   assert((resolved_klass_index & 0xffff0000) == 0, "must be");
 245   *int_at_addr(class_index) =
 246     build_int_from_shorts((jushort)resolved_klass_index, (jushort)name_index);
 247 
 248   symbol_at_put(name_index, name);
 249   name->increment_refcount();
 250   Klass** adr = resolved_klasses()->adr_at(resolved_klass_index);
 251   Atomic::release_store(adr, k);
 252 
 253   // The interpreter assumes when the tag is stored, the klass is resolved
 254   // and the Klass* non-NULL, so we need hardware store ordering here.
 255   jbyte qdesc_bit = (name->is_Q_signature()) ? (jbyte) JVM_CONSTANT_QDescBit : 0;
 256   if (k != NULL) {
 257     release_tag_at_put(class_index, JVM_CONSTANT_Class | qdesc_bit);
 258   } else {
 259     release_tag_at_put(class_index, JVM_CONSTANT_UnresolvedClass | qdesc_bit);
 260   }
 261 }

 263 // Unsafe anonymous class support:
 264 void ConstantPool::klass_at_put(int class_index, Klass* k) {
 265   assert(k != NULL, "must be valid klass");
 266   CPKlassSlot kslot = klass_slot_at(class_index);
 267   int resolved_klass_index = kslot.resolved_klass_index();
 268   Klass** adr = resolved_klasses()->adr_at(resolved_klass_index);
 269   Atomic::release_store(adr, k);
 270 
 271   // The interpreter assumes when the tag is stored, the klass is resolved
 272   // and the Klass* non-NULL, so we need hardware store ordering here.
 273   assert(!k->name()->is_Q_signature(), "Q-type without JVM_CONSTANT_QDescBit");
 274   release_tag_at_put(class_index, JVM_CONSTANT_Class);
 275 }

 277 #if INCLUDE_CDS_JAVA_HEAP
 278 // Archive the resolved references
 279 void ConstantPool::archive_resolved_references(Thread* THREAD) {
 280   if (_cache == NULL) {
 281     return; // nothing to do
 282   }
 283 
 284   InstanceKlass *ik = pool_holder();
 285   if (!(ik->is_shared_boot_class() || ik->is_shared_platform_class() ||
 286         ik->is_shared_app_class())) {
 287     // Archiving resolved references for classes from non-builtin loaders
 288     // is not yet supported.
 289     set_resolved_references(OopHandle());
 290     return;
 291   }
 292 
 293   objArrayOop rr = resolved_references();
 294   Array<u2>* ref_map = reference_map();
 295   if (rr != NULL) {
 296     int ref_map_len = ref_map == NULL ? 0 : ref_map->length();
 297     int rr_len = rr->length();
 298     for (int i = 0; i < rr_len; i++) {
 299       oop p = rr->obj_at(i);
 300       rr->obj_at_put(i, NULL);
 301       if (p != NULL && i < ref_map_len) {
 302         int index = object_to_cp_index(i);
 303         if (tag_at(index).is_string()) {
 304           oop op = StringTable::create_archived_string(p, THREAD);
 305           // If the String object is not archived (possibly too large),
 306           // NULL is returned. Also set it in the array, so we won't
 307           // have a 'bad' reference in the archived resolved_reference
 308           // array.
 309           rr->obj_at_put(i, op);
 310         }
 311       }
 312     }
 313 
 314     oop archived = HeapShared::archive_heap_object(rr, THREAD);
 315     // If the resolved references array is not archived (too large),
 316     // the 'archived' object is NULL. No need to explicitly check
 317     // the return value of archive_heap_object here. At runtime, the
 318     // resolved references will be created using the normal process
 319     // when there is no archived value.
 320     _cache->set_archived_references(archived);
 321     set_resolved_references(OopHandle());
 322   }
 323 }

 325 void ConstantPool::resolve_class_constants(TRAPS) {
 326   assert(DumpSharedSpaces, "used during dump time only");
 327   // The _cache may be NULL if the _pool_holder klass fails verification
 328   // at dump time due to missing dependencies.
 329   if (cache() == NULL || reference_map() == NULL) {
 330     return; // nothing to do
 331   }
 332 
 333   constantPoolHandle cp(THREAD, this);
 334   for (int index = 1; index < length(); index++) { // Index 0 is unused
 335     if (tag_at(index).is_string() && !cp->is_pseudo_string_at(index)) {
 336       int cache_index = cp->cp_to_object_index(index);
 337       string_at_impl(cp, index, cache_index, CHECK);
 338     }
 339   }
 340 }

 343 // CDS support. Create a new resolved_references array.
 344 void ConstantPool::restore_unshareable_info(TRAPS) {
 345   assert(is_constantPool(), "ensure C++ vtable is restored");
 346   assert(on_stack(), "should always be set for shared constant pools");
 347   assert(is_shared(), "should always be set for shared constant pools");
 348   assert(_cache != NULL, "constant pool _cache should not be NULL");
 349 
 350   // Only create the new resolved references array if it hasn't been attempted before
 351   if (resolved_references() != NULL) return;
 352 
 353   // restore the C++ vtable from the shared archive
 354   restore_vtable();
 355 
 356   if (SystemDictionary::Object_klass_loaded()) {
 357     ClassLoaderData* loader_data = pool_holder()->class_loader_data();
 358 #if INCLUDE_CDS_JAVA_HEAP
 359     if (HeapShared::open_archive_heap_region_mapped() &&
 360         _cache->archived_references() != NULL) {
 361       oop archived = _cache->archived_references();
 362       // Create handle for the archived resolved reference array object
 363       Handle refs_handle(THREAD, archived);
 364       set_resolved_references(loader_data->add_handle(refs_handle));
 365     } else
 366 #endif
 367     {
 368       // No mapped archived resolved reference array
 369       // Recreate the object array and add to ClassLoaderData.
 370       int map_length = resolved_reference_length();
 371       if (map_length > 0) {
 372         objArrayOop stom = oopFactory::new_objArray(SystemDictionary::Object_klass(), map_length, CHECK);
 373         Handle refs_handle(THREAD, (oop)stom);  // must handleize.
 374         set_resolved_references(loader_data->add_handle(refs_handle));
 375       }
 376     }
 377   }
 378 }

 380 void ConstantPool::remove_unshareable_info() {
 381   // Resolved references are not in the shared archive.
 382   // Save the length for restoration.  It is not necessarily the same length
 383   // as reference_map.length() if invokedynamic is saved. It is needed when
 384   // re-creating the resolved reference array if archived heap data cannot be map
 385   // at runtime.
 386   set_resolved_reference_length(
 387     resolved_references() != NULL ? resolved_references()->length() : 0);
 388 
 389   // If archiving heap objects is not allowed, clear the resolved references.
 390   // Otherwise, it is cleared after the resolved references array is cached
 391   // (see archive_resolved_references()).
 392   // If DynamicDumpSharedSpaces is enabled, clear the resolved references also
 393   // as java objects are not archived in the top layer.
 394   if (!HeapShared::is_heap_object_archiving_allowed() || DynamicDumpSharedSpaces) {
 395     set_resolved_references(OopHandle());
 396   }
 397 
 398   // Shared ConstantPools are in the RO region, so the _flags cannot be modified.
 399   // The _on_stack flag is used to prevent ConstantPools from deallocation during
 400   // class redefinition. Since shared ConstantPools cannot be deallocated anyway,
 401   // we always set _on_stack to true to avoid having to change _flags during runtime.
 402   _flags |= (_on_stack | _is_shared);
 403   int num_klasses = 0;
 404   for (int index = 1; index < length(); index++) { // Index 0 is unused
 405     if (!DynamicDumpSharedSpaces) {
 406       assert(!tag_at(index).is_unresolved_klass_in_error(), "This must not happen during static dump time");
 407     } else {
 408       if (tag_at(index).is_unresolved_klass_in_error() ||
 409           tag_at(index).is_method_handle_in_error()    ||
 410           tag_at(index).is_method_type_in_error()      ||
 411           tag_at(index).is_dynamic_constant_in_error()) {
 412         tag_at_put(index, JVM_CONSTANT_UnresolvedClass);
 413       }
 414     }
 415     if (tag_at(index).is_klass()) {
 416       // This class was resolved as a side effect of executing Java code
 417       // during dump time. We need to restore it back to an UnresolvedClass,
 418       // so that the proper class loading and initialization can happen
 419       // at runtime.
 420       bool clear_it = true;
 421       if (pool_holder()->is_hidden() && index == pool_holder()->this_class_index()) {
 422         // All references to a hidden class's own field/methods are through this
 423         // index. We cannot clear it. See comments in ClassFileParser::fill_instance_klass.
 424         clear_it = false;
 425       }
 426       if (clear_it) {
 427         CPKlassSlot kslot = klass_slot_at(index);
 428         int resolved_klass_index = kslot.resolved_klass_index();
 429         int name_index = kslot.name_index();
 430         assert(tag_at(name_index).is_symbol(), "sanity");
 431         resolved_klasses()->at_put(resolved_klass_index, NULL);
 432         tag_at_put(index, JVM_CONSTANT_UnresolvedClass);
 433         assert(klass_name_at(index) == symbol_at(name_index), "sanity");
 434       }
 435     }
 436   }
 437   if (cache() != NULL) {
 438     cache()->remove_unshareable_info();
 439   }
 440 }

 453 void ConstantPool::trace_class_resolution(const constantPoolHandle& this_cp, Klass* k) {
 454   ResourceMark rm;
 455   int line_number = -1;
 456   const char * source_file = NULL;
 457   if (JavaThread::current()->has_last_Java_frame()) {
 458     // try to identify the method which called this function.
 459     vframeStream vfst(JavaThread::current());
 460     if (!vfst.at_end()) {
 461       line_number = vfst.method()->line_number_from_bci(vfst.bci());
 462       Symbol* s = vfst.method()->method_holder()->source_file_name();
 463       if (s != NULL) {
 464         source_file = s->as_C_string();
 465       }
 466     }
 467   }
 468   if (k != this_cp->pool_holder()) {
 469     // only print something if the classes are different
 470     if (source_file != NULL) {
 471       log_debug(class, resolve)("%s %s %s:%d",
 472                  this_cp->pool_holder()->external_name(),
 473                  k->external_name(), source_file, line_number);
 474     } else {
 475       log_debug(class, resolve)("%s %s",
 476                  this_cp->pool_holder()->external_name(),
 477                  k->external_name());
 478     }
 479   }
 480 }

 488 Klass* ConstantPool::klass_at_impl(const constantPoolHandle& this_cp, int which,
 489                                    bool save_resolution_error, TRAPS) {
 490   assert(THREAD->is_Java_thread(), "must be a Java thread");
 491   JavaThread* javaThread = (JavaThread*)THREAD;
 492 
 493   // A resolved constantPool entry will contain a Klass*, otherwise a Symbol*.
 494   // It is not safe to rely on the tag bit's here, since we don't have a lock, and
 495   // the entry and tag is not updated atomicly.
 496   CPKlassSlot kslot = this_cp->klass_slot_at(which);
 497   int resolved_klass_index = kslot.resolved_klass_index();
 498   int name_index = kslot.name_index();
 499   assert(this_cp->tag_at(name_index).is_symbol(), "sanity");
 500 
 501   Klass* klass = this_cp->resolved_klasses()->at(resolved_klass_index);
 502   if (klass != NULL) {
 503     return klass;
 504   }
 505 
 506   // This tag doesn't change back to unresolved class unless at a safepoint.
 507   if (this_cp->tag_at(which).is_unresolved_klass_in_error()) {
 508     // The original attempt to resolve this constant pool entry failed so find the
 509     // class of the original error and throw another error of the same class
 510     // (JVMS 5.4.3).
 511     // If there is a detail message, pass that detail message to the error.
 512     // The JVMS does not strictly require us to duplicate the same detail message,
 513     // or any internal exception fields such as cause or stacktrace.  But since the
 514     // detail message is often a class name or other literal string, we will repeat it
 515     // if we can find it in the symbol table.
 516     throw_resolution_error(this_cp, which, CHECK_NULL);
 517     ShouldNotReachHere();
 518   }
 519 
 520   Handle mirror_handle;
 521   Symbol* name = this_cp->symbol_at(name_index);
 522   bool inline_type_signature = false;
 523   if (name->is_Q_signature()) {
 524     name = name->fundamental_name(THREAD);
 525     inline_type_signature = true;
 526   }
 527   Handle loader (THREAD, this_cp->pool_holder()->class_loader());
 528   Handle protection_domain (THREAD, this_cp->pool_holder()->protection_domain());
 529 
 530   Klass* k;
 531   {
 532     // Turn off the single stepping while doing class resolution
 533     JvmtiHideSingleStepping jhss(javaThread);
 534     k = SystemDictionary::resolve_or_fail(name, loader, protection_domain, true, THREAD);
 535   } //  JvmtiHideSingleStepping jhss(javaThread);
 536   if (inline_type_signature) {
 537     name->decrement_refcount();
 538   }
 539 
 540   if (!HAS_PENDING_EXCEPTION) {
 541     // preserve the resolved klass from unloading
 542     mirror_handle = Handle(THREAD, k->java_mirror());
 543     // Do access check for klasses
 544     verify_constant_pool_resolve(this_cp, k, THREAD);
 545   }
 546 
 547   if (!HAS_PENDING_EXCEPTION && inline_type_signature) {
 548     check_is_inline_type(k, THREAD);
 549   }
 550 
 551   if (!HAS_PENDING_EXCEPTION) {
 552     Klass* bottom_klass = NULL;
 553     if (k->is_objArray_klass()) {
 554       bottom_klass = ObjArrayKlass::cast(k)->bottom_klass();
 555       assert(bottom_klass != NULL, "Should be set");
 556       assert(bottom_klass->is_instance_klass() || bottom_klass->is_typeArray_klass(), "Sanity check");
 557     } else if (k->is_valueArray_klass()) {
 558       bottom_klass = ValueArrayKlass::cast(k)->element_klass();
 559       assert(bottom_klass != NULL, "Should be set");
 560     }
 561   }
 562 
 563   // Failed to resolve class. We must record the errors so that subsequent attempts
 564   // to resolve this constant pool entry fail with the same error (JVMS 5.4.3).
 565   if (HAS_PENDING_EXCEPTION) {
 566     if (save_resolution_error) {
 567       save_and_throw_exception(this_cp, which, constantTag(JVM_CONSTANT_UnresolvedClass), CHECK_NULL);
 568       // If CHECK_NULL above doesn't return the exception, that means that
 569       // some other thread has beaten us and has resolved the class.
 570       // To preserve old behavior, we return the resolved class.
 571       klass = this_cp->resolved_klasses()->at(resolved_klass_index);
 572       assert(klass != NULL, "must be resolved if exception was cleared");
 573       return klass;
 574     } else {
 575       return NULL;  // return the pending exception
 576     }
 577   }
 578 
 579   // logging for class+resolve.
 580   if (log_is_enabled(Debug, class, resolve)){
 581     trace_class_resolution(this_cp, k);
 582   }
 583   Klass** adr = this_cp->resolved_klasses()->adr_at(resolved_klass_index);
 584   Atomic::release_store(adr, k);
 585   // The interpreter assumes when the tag is stored, the klass is resolved
 586   // and the Klass* stored in _resolved_klasses is non-NULL, so we need
 587   // hardware store ordering here.
 588   jbyte tag = JVM_CONSTANT_Class;
 589   if (this_cp->tag_at(which).is_Qdescriptor_klass()) {
 590     tag |= JVM_CONSTANT_QDescBit;
 591   }
 592   this_cp->release_tag_at_put(which, tag);
 593   return k;
 594 }

 597 // Does not update ConstantPool* - to avoid any exception throwing. Used
 598 // by compiler and exception handling.  Also used to avoid classloads for
 599 // instanceof operations. Returns NULL if the class has not been loaded or
 600 // if the verification of constant pool failed
 601 Klass* ConstantPool::klass_at_if_loaded(const constantPoolHandle& this_cp, int which) {
 602   CPKlassSlot kslot = this_cp->klass_slot_at(which);
 603   int resolved_klass_index = kslot.resolved_klass_index();
 604   int name_index = kslot.name_index();
 605   assert(this_cp->tag_at(name_index).is_symbol(), "sanity");
 606 
 607   Klass* k = this_cp->resolved_klasses()->at(resolved_klass_index);
 608   if (k != NULL) {
 609     return k;
 610   } else {
 611     Thread *thread = Thread::current();
 612     Symbol* name = this_cp->symbol_at(name_index);
 613     oop loader = this_cp->pool_holder()->class_loader();
 614     oop protection_domain = this_cp->pool_holder()->protection_domain();
 615     Handle h_prot (thread, protection_domain);
 616     Handle h_loader (thread, loader);
 617     Klass* k = SystemDictionary::find(name, h_loader, h_prot, thread);
 618 
 619     // Avoid constant pool verification at a safepoint, which takes the Module_lock.
 620     if (k != NULL && !SafepointSynchronize::is_at_safepoint()) {
 621       // Make sure that resolving is legal
 622       EXCEPTION_MARK;
 623       // return NULL if verification fails
 624       verify_constant_pool_resolve(this_cp, k, THREAD);
 625       if (HAS_PENDING_EXCEPTION) {
 626         CLEAR_PENDING_EXCEPTION;
 627         return NULL;
 628       }
 629       return k;
 630     } else {
 631       return k;
 632     }
 633   }
 634 }

 683 int ConstantPool::impl_name_and_type_ref_index_at(int which, bool uncached) {
 684   int i = which;
 685   if (!uncached && cache() != NULL) {
 686     if (ConstantPool::is_invokedynamic_index(which)) {
 687       // Invokedynamic index is index into the constant pool cache
 688       int pool_index = invokedynamic_bootstrap_ref_index_at(which);
 689       pool_index = bootstrap_name_and_type_ref_index_at(pool_index);
 690       assert(tag_at(pool_index).is_name_and_type(), "");
 691       return pool_index;
 692     }
 693     // change byte-ordering and go via cache
 694     i = remap_instruction_operand_from_cache(which);
 695   } else {
 696     if (tag_at(which).has_bootstrap()) {
 697       int pool_index = bootstrap_name_and_type_ref_index_at(which);
 698       assert(tag_at(pool_index).is_name_and_type(), "");
 699       return pool_index;
 700     }
 701   }
 702   assert(tag_at(i).is_field_or_method(), "Corrupted constant pool");
 703   assert(!tag_at(i).has_bootstrap(), "Must be handled above");
 704   jint ref_index = *int_at_addr(i);
 705   return extract_high_short_from_int(ref_index);
 706 }

 802 Symbol* ConstantPool::exception_message(const constantPoolHandle& this_cp, int which, constantTag tag, oop pending_exception) {
 803   // Dig out the detailed message to reuse if possible
 804   Symbol* message = java_lang_Throwable::detail_message(pending_exception);
 805   if (message != NULL) {
 806     return message;
 807   }
 808 
 809   // Return specific message for the tag
 810   switch (tag.value()) {
 811   case JVM_CONSTANT_UnresolvedClass:
 812     // return the class name in the error message
 813     message = this_cp->klass_name_at(which);

 815   case JVM_CONSTANT_MethodHandle:
 816     // return the method handle name in the error message

 819   case JVM_CONSTANT_MethodType:
 820     // return the method type signature in the error message
 821     message = this_cp->method_type_signature_at(which);

 825     message = this_cp->uncached_name_ref_at(which);
 826     break;

 829   }

 847 // If resolution for Class, Dynamic constant, MethodHandle or MethodType fails, save the
 848 // exception in the resolution error table, so that the same exception is thrown again.
 849 void ConstantPool::save_and_throw_exception(const constantPoolHandle& this_cp, int which,
 850                                             constantTag tag, TRAPS) {
 851   Symbol* error = PENDING_EXCEPTION->klass()->name();
 852 
 853   int error_tag = tag.error_value();
 854 
 855   if (!PENDING_EXCEPTION->
 856     is_a(SystemDictionary::LinkageError_klass())) {
 857     // Just throw the exception and don't prevent these classes from
 858     // being loaded due to virtual machine errors like StackOverflow
 859     // and OutOfMemoryError, etc, or if the thread was hit by stop()
 860     // Needs clarification to section 5.4.3 of the VM spec (see 6308271)
 861   } else if (this_cp->tag_at(which).value() != error_tag) {
 862     Symbol* message = exception_message(this_cp, which, tag, PENDING_EXCEPTION);
 863     SystemDictionary::add_resolution_error(this_cp, which, error, message);
 864     // CAS in the tag.  If a thread beat us to registering this error that's fine.
 865     // If another thread resolved the reference, this is a race condition. This
 866     // thread may have had a security manager or something temporary.
 867     // This doesn't deterministically get an error.   So why do we save this?
 868     // We save this because jvmti can add classes to the bootclass path after
 869     // this error, so it needs to get the same error if the error is first.
 870     jbyte old_tag = Atomic::cmpxchg((jbyte*)this_cp->tag_addr_at(which),
 871                                     (jbyte)tag.value(),
 872                                     (jbyte)error_tag);
 873     if (old_tag != error_tag && old_tag != tag.value()) {
 874       // MethodHandles and MethodType doesn't change to resolved version.
 875       assert(this_cp->tag_at(which).is_klass(), "Wrong tag value");
 876       // Forget the exception and use the resolved class.
 877       CLEAR_PENDING_EXCEPTION;
 878     }
 879   } else {
 880     // some other thread put this in error state
 881     throw_resolution_error(this_cp, which, CHECK);
 882   }
 883 }

 907 // Called to resolve constants in the constant pool and return an oop.
 908 // Some constant pool entries cache their resolved oop. This is also
 909 // called to create oops from constants to use in arguments for invokedynamic
 910 oop ConstantPool::resolve_constant_at_impl(const constantPoolHandle& this_cp,
 911                                            int index, int cache_index,
 912                                            bool* status_return, TRAPS) {
 913   oop result_oop = NULL;
 914   Handle throw_exception;
 915 
 916   if (cache_index == _possible_index_sentinel) {
 917     // It is possible that this constant is one which is cached in the objects.
 918     // We'll do a linear search.  This should be OK because this usage is rare.
 919     // FIXME: If bootstrap specifiers stress this code, consider putting in
 920     // a reverse index.  Binary search over a short array should do it.
 921     assert(index > 0, "valid index");
 922     cache_index = this_cp->cp_to_object_index(index);
 923   }
 924   assert(cache_index == _no_index_sentinel || cache_index >= 0, "");
 925   assert(index == _no_index_sentinel || index >= 0, "");
 926 
 927   if (cache_index >= 0) {
 928     result_oop = this_cp->resolved_references()->obj_at(cache_index);
 929     if (result_oop != NULL) {
 930       if (result_oop == Universe::the_null_sentinel()) {
 931         DEBUG_ONLY(int temp_index = (index >= 0 ? index : this_cp->object_to_cp_index(cache_index)));
 932         assert(this_cp->tag_at(temp_index).is_dynamic_constant(), "only condy uses the null sentinel");
 933         result_oop = NULL;
 934       }
 935       if (status_return != NULL)  (*status_return) = true;
 936       return result_oop;
 937       // That was easy...
 938     }
 939     index = this_cp->object_to_cp_index(cache_index);
 940   }
 941 
 942   jvalue prim_value;  // temp used only in a few cases below
 943 
 944   constantTag tag = this_cp->tag_at(index);
 945 
 946   if (status_return != NULL) {
 947     // don't trigger resolution if the constant might need it
 948     switch (tag.value()) {
 949     case JVM_CONSTANT_Class:
 950     {
 951       CPKlassSlot kslot = this_cp->klass_slot_at(index);
 952       int resolved_klass_index = kslot.resolved_klass_index();
 953       if (this_cp->resolved_klasses()->at(resolved_klass_index) == NULL) {
 954         (*status_return) = false;
 955         return NULL;
 956       }
 957       // the klass is waiting in the CP; go get it
 958       break;
 959     }
 960     case JVM_CONSTANT_String:
 961     case JVM_CONSTANT_Integer:
 962     case JVM_CONSTANT_Float:
 963     case JVM_CONSTANT_Long:
 964     case JVM_CONSTANT_Double:
 965       // these guys trigger OOM at worst
 966       break;
 967     default:
 968       (*status_return) = false;
 969       return NULL;
 970     }
 971     // from now on there is either success or an OOME
 972     (*status_return) = true;
 973   }
 974 
 975   switch (tag.value()) {
 976 
 977   case JVM_CONSTANT_UnresolvedClass:
 978   case JVM_CONSTANT_UnresolvedClassInError:
 979   case JVM_CONSTANT_Class:
 980     {
 981       assert(cache_index == _no_index_sentinel, "should not have been set");
 982       Klass* resolved = klass_at_impl(this_cp, index, true, CHECK_NULL);
 983       // ldc wants the java mirror.
 984       result_oop = resolved->java_mirror();
 985       break;
 986     }
 987 
 988   case JVM_CONSTANT_Dynamic:
 989     {
 990       // Resolve the Dynamically-Computed constant to invoke the BSM in order to obtain the resulting oop.
 991       BootstrapInfo bootstrap_specifier(this_cp, index);
 992 
 993       // The initial step in resolving an unresolved symbolic reference to a
 994       // dynamically-computed constant is to resolve the symbolic reference to a
 995       // method handle which will be the bootstrap method for the dynamically-computed
 996       // constant. If resolution of the java.lang.invoke.MethodHandle for the bootstrap
 997       // method fails, then a MethodHandleInError is stored at the corresponding
 998       // bootstrap method's CP index for the CONSTANT_MethodHandle_info. No need to
 999       // set a DynamicConstantInError here since any subsequent use of this
1000       // bootstrap method will encounter the resolution of MethodHandleInError.
1001       // Both the first, (resolution of the BSM and its static arguments), and the second tasks,
1002       // (invocation of the BSM), of JVMS Section 5.4.3.6 occur within invoke_bootstrap_method()
1003       // for the bootstrap_specifier created above.
1004       SystemDictionary::invoke_bootstrap_method(bootstrap_specifier, THREAD);
1005       Exceptions::wrap_dynamic_exception(/* is_indy */ false, THREAD);
1006       if (HAS_PENDING_EXCEPTION) {
1007         // Resolution failure of the dynamically-computed constant, save_and_throw_exception
1008         // will check for a LinkageError and store a DynamicConstantInError.
1009         save_and_throw_exception(this_cp, index, tag, CHECK_NULL);
1010       }
1011       result_oop = bootstrap_specifier.resolved_value()();
1012       BasicType type = Signature::basic_type(bootstrap_specifier.signature());
1013       if (!is_reference_type(type)) {
1014         // Make sure the primitive value is properly boxed.
1015         // This is a JDK responsibility.
1016         const char* fail = NULL;
1017         if (result_oop == NULL) {
1018           fail = "null result instead of box";
1019         } else if (!is_java_primitive(type)) {
1020           // FIXME: support value types via unboxing
1021           fail = "can only handle references and primitives";
1022         } else if (!java_lang_boxing_object::is_instance(result_oop, type)) {
1023           fail = "primitive is not properly boxed";
1024         }
1025         if (fail != NULL) {
1026           // Since this exception is not a LinkageError, throw exception
1027           // but do not save a DynamicInError resolution result.
1028           // See section 5.4.3 of the VM spec.
1029           THROW_MSG_NULL(vmSymbols::java_lang_InternalError(), fail);
1030         }
1031       }
1032 
1033       LogTarget(Debug, methodhandles, condy) lt_condy;
1034       if (lt_condy.is_enabled()) {
1035         LogStream ls(lt_condy);
1036         bootstrap_specifier.print_msg_on(&ls, "resolve_constant_at_impl");
1037       }
1038       break;
1039     }
1040 
1041   case JVM_CONSTANT_String:
1042     assert(cache_index != _no_index_sentinel, "should have been set");
1043     if (this_cp->is_pseudo_string_at(index)) {
1044       result_oop = this_cp->pseudo_string_at(index, cache_index);
1045       break;
1046     }
1047     result_oop = string_at_impl(this_cp, index, cache_index, CHECK_NULL);
1048     break;
1049 
1050   case JVM_CONSTANT_DynamicInError:
1051   case JVM_CONSTANT_MethodHandleInError:
1052   case JVM_CONSTANT_MethodTypeInError:
1053     {
1054       throw_resolution_error(this_cp, index, CHECK_NULL);
1055       break;
1056     }
1057 
1058   case JVM_CONSTANT_MethodHandle:
1059     {
1060       int ref_kind                 = this_cp->method_handle_ref_kind_at(index);
1061       int callee_index             = this_cp->method_handle_klass_index_at(index);
1062       Symbol*  name =      this_cp->method_handle_name_ref_at(index);
1063       Symbol*  signature = this_cp->method_handle_signature_ref_at(index);
1064       constantTag m_tag  = this_cp->tag_at(this_cp->method_handle_index_at(index));
1065       { ResourceMark rm(THREAD);
1066         log_debug(class, resolve)("resolve JVM_CONSTANT_MethodHandle:%d [%d/%d/%d] %s.%s",
1067                               ref_kind, index, this_cp->method_handle_index_at(index),
1068                               callee_index, name->as_C_string(), signature->as_C_string());
1069       }
1070 
1071       Klass* callee = klass_at_impl(this_cp, callee_index, true, CHECK_NULL);
1072 
1073       // Check constant pool method consistency
1074       if ((callee->is_interface() && m_tag.is_method()) ||
1075           ((!callee->is_interface() && m_tag.is_interface_method()))) {
1076         ResourceMark rm(THREAD);
1077         stringStream ss;
1078         ss.print("Inconsistent constant pool data in classfile for class %s. "
1079                  "Method '", callee->name()->as_C_string());
1080         signature->print_as_signature_external_return_type(&ss);
1081         ss.print(" %s(", name->as_C_string());
1082         signature->print_as_signature_external_parameters(&ss);
1083         ss.print(")' at index %d is %s and should be %s",
1084                  index,
1085                  callee->is_interface() ? "CONSTANT_MethodRef" : "CONSTANT_InterfaceMethodRef",
1086                  callee->is_interface() ? "CONSTANT_InterfaceMethodRef" : "CONSTANT_MethodRef");
1087         THROW_MSG_NULL(vmSymbols::java_lang_IncompatibleClassChangeError(), ss.as_string());
1088       }
1089 
1090       Klass* klass = this_cp->pool_holder();
1091       Handle value = SystemDictionary::link_method_handle_constant(klass, ref_kind,
1092                                                                    callee, name, signature,
1093                                                                    THREAD);
1094       result_oop = value();
1095       if (HAS_PENDING_EXCEPTION) {
1096         save_and_throw_exception(this_cp, index, tag, CHECK_NULL);
1097       }
1098       break;
1099     }
1100 
1101   case JVM_CONSTANT_MethodType:
1102     {
1103       Symbol*  signature = this_cp->method_type_signature_at(index);
1104       { ResourceMark rm(THREAD);
1105         log_debug(class, resolve)("resolve JVM_CONSTANT_MethodType [%d/%d] %s",
1106                               index, this_cp->method_type_index_at(index),
1107                               signature->as_C_string());
1108       }
1109       Klass* klass = this_cp->pool_holder();
1110       Handle value = SystemDictionary::find_method_handle_type(signature, klass, THREAD);
1111       result_oop = value();
1112       if (HAS_PENDING_EXCEPTION) {
1113         save_and_throw_exception(this_cp, index, tag, CHECK_NULL);
1114       }
1115       break;
1116     }
1117 
1118   case JVM_CONSTANT_Integer:
1119     assert(cache_index == _no_index_sentinel, "should not have been set");
1120     prim_value.i = this_cp->int_at(index);
1121     result_oop = java_lang_boxing_object::create(T_INT, &prim_value, CHECK_NULL);
1122     break;
1123 
1124   case JVM_CONSTANT_Float:
1125     assert(cache_index == _no_index_sentinel, "should not have been set");
1126     prim_value.f = this_cp->float_at(index);
1127     result_oop = java_lang_boxing_object::create(T_FLOAT, &prim_value, CHECK_NULL);
1128     break;
1129 
1130   case JVM_CONSTANT_Long:
1131     assert(cache_index == _no_index_sentinel, "should not have been set");
1132     prim_value.j = this_cp->long_at(index);
1133     result_oop = java_lang_boxing_object::create(T_LONG, &prim_value, CHECK_NULL);
1134     break;
1135 
1136   case JVM_CONSTANT_Double:
1137     assert(cache_index == _no_index_sentinel, "should not have been set");
1138     prim_value.d = this_cp->double_at(index);
1139     result_oop = java_lang_boxing_object::create(T_DOUBLE, &prim_value, CHECK_NULL);
1140     break;
1141 
1142   default:
1143     DEBUG_ONLY( tty->print_cr("*** %p: tag at CP[%d/%d] = %d",
1144                               this_cp(), index, cache_index, tag.value()));
1145     assert(false, "unexpected constant tag");
1146     break;
1147   }
1148 
1149   if (cache_index >= 0) {
1150     // Benign race condition:  resolved_references may already be filled in.
1151     // The important thing here is that all threads pick up the same result.
1152     // It doesn't matter which racing thread wins, as long as only one
1153     // result is used by all threads, and all future queries.
1154     oop new_result = (result_oop == NULL ? Universe::the_null_sentinel() : result_oop);
1155     oop old_result = this_cp->resolved_references()
1156       ->atomic_compare_exchange_oop(cache_index, new_result, NULL);
1157     if (old_result == NULL) {
1158       return result_oop;  // was installed
1159     } else {
1160       // Return the winning thread's result.  This can be different than
1161       // the result here for MethodHandles.
1162       if (old_result == Universe::the_null_sentinel())
1163         old_result = NULL;
1164       return old_result;
1165     }
1166   } else {
1167     assert(result_oop != Universe::the_null_sentinel(), "");
1168     return result_oop;
1169   }
1170 }

1179 void ConstantPool::copy_bootstrap_arguments_at_impl(const constantPoolHandle& this_cp, int index,
1180                                                     int start_arg, int end_arg,
1181                                                     objArrayHandle info, int pos,
1182                                                     bool must_resolve, Handle if_not_available,
1183                                                     TRAPS) {
1184   int argc;
1185   int limit = pos + end_arg - start_arg;
1186   // checks: index in range [0..this_cp->length),
1187   // tag at index, start..end in range [0..argc],
1188   // info array non-null, pos..limit in [0..info.length]
1189   if ((0 >= index    || index >= this_cp->length())  ||
1190       !(this_cp->tag_at(index).is_invoke_dynamic()    ||
1191         this_cp->tag_at(index).is_dynamic_constant()) ||
1192       (0 > start_arg || start_arg > end_arg) ||
1193       (end_arg > (argc = this_cp->bootstrap_argument_count_at(index))) ||
1194       (0 > pos       || pos > limit)         ||
1195       (info.is_null() || limit > info->length())) {
1196     // An index or something else went wrong; throw an error.
1197     // Since this is an internal API, we don't expect this,
1198     // so we don't bother to craft a nice message.
1199     THROW_MSG(vmSymbols::java_lang_LinkageError(), "bad BSM argument access");
1200   }
1201   // now we can loop safely
1202   int info_i = pos;
1203   for (int i = start_arg; i < end_arg; i++) {
1204     int arg_index = this_cp->bootstrap_argument_index_at(index, i);
1205     oop arg_oop;
1206     if (must_resolve) {
1207       arg_oop = this_cp->resolve_possibly_cached_constant_at(arg_index, CHECK);
1208     } else {
1209       bool found_it = false;
1210       arg_oop = this_cp->find_cached_constant_at(arg_index, found_it, CHECK);
1211       if (!found_it)  arg_oop = if_not_available();
1212     }
1213     info->obj_at_put(info_i++, arg_oop);
1214   }

1251 // Compare this constant pool's entry at index1 to the constant pool
1252 // cp2's entry at index2.
1253 bool ConstantPool::compare_entry_to(int index1, const constantPoolHandle& cp2,
1254        int index2, TRAPS) {
1255 
1256   // The error tags are equivalent to non-error tags when comparing
1257   jbyte t1 = tag_at(index1).non_error_value();
1258   jbyte t2 = cp2->tag_at(index2).non_error_value();
1259 
1260   if (t1 != t2) {
1261     // Not the same entry type so there is nothing else to check. Note
1262     // that this style of checking will consider resolved/unresolved
1263     // class pairs as different.
1264     // From the ConstantPool* API point of view, this is correct
1265     // behavior. See VM_RedefineClasses::merge_constant_pools() to see how this
1266     // plays out in the context of ConstantPool* merging.
1267     return false;
1268   }
1269 
1270   switch (t1) {
1271   case JVM_CONSTANT_Class:
1272   {
1273     Klass* k1 = klass_at(index1, CHECK_false);
1274     Klass* k2 = cp2->klass_at(index2, CHECK_false);
1275     if (k1 == k2) {
1276       return true;
1277     }
1278   } break;
1279 
1280   case JVM_CONSTANT_ClassIndex:
1281   {
1282     int recur1 = klass_index_at(index1);
1283     int recur2 = cp2->klass_index_at(index2);
1284     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1285     if (match) {
1286       return true;
1287     }
1288   } break;
1289 
1290   case JVM_CONSTANT_Double:
1291   {
1292     jdouble d1 = double_at(index1);
1293     jdouble d2 = cp2->double_at(index2);
1294     if (d1 == d2) {
1295       return true;
1296     }
1297   } break;
1298 
1299   case JVM_CONSTANT_Fieldref:
1300   case JVM_CONSTANT_InterfaceMethodref:
1301   case JVM_CONSTANT_Methodref:
1302   {
1303     int recur1 = uncached_klass_ref_index_at(index1);
1304     int recur2 = cp2->uncached_klass_ref_index_at(index2);
1305     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1306     if (match) {
1307       recur1 = uncached_name_and_type_ref_index_at(index1);
1308       recur2 = cp2->uncached_name_and_type_ref_index_at(index2);
1309       match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1310       if (match) {
1311         return true;
1312       }
1313     }
1314   } break;
1315 
1316   case JVM_CONSTANT_Float:
1317   {
1318     jfloat f1 = float_at(index1);
1319     jfloat f2 = cp2->float_at(index2);
1320     if (f1 == f2) {
1321       return true;
1322     }
1323   } break;
1324 
1325   case JVM_CONSTANT_Integer:
1326   {
1327     jint i1 = int_at(index1);
1328     jint i2 = cp2->int_at(index2);
1329     if (i1 == i2) {
1330       return true;
1331     }
1332   } break;
1333 
1334   case JVM_CONSTANT_Long:
1335   {
1336     jlong l1 = long_at(index1);
1337     jlong l2 = cp2->long_at(index2);
1338     if (l1 == l2) {
1339       return true;
1340     }
1341   } break;
1342 
1343   case JVM_CONSTANT_NameAndType:
1344   {
1345     int recur1 = name_ref_index_at(index1);
1346     int recur2 = cp2->name_ref_index_at(index2);
1347     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1348     if (match) {
1349       recur1 = signature_ref_index_at(index1);
1350       recur2 = cp2->signature_ref_index_at(index2);
1351       match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1352       if (match) {
1353         return true;
1354       }
1355     }
1356   } break;
1357 
1358   case JVM_CONSTANT_StringIndex:
1359   {
1360     int recur1 = string_index_at(index1);
1361     int recur2 = cp2->string_index_at(index2);
1362     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1363     if (match) {
1364       return true;
1365     }
1366   } break;
1367 
1368   case JVM_CONSTANT_UnresolvedClass:
1369   {
1370     Symbol* k1 = klass_name_at(index1);
1371     Symbol* k2 = cp2->klass_name_at(index2);
1372     if (k1 == k2) {
1373       return true;
1374     }
1375   } break;
1376 
1377   case JVM_CONSTANT_MethodType:
1378   {
1379     int k1 = method_type_index_at(index1);
1380     int k2 = cp2->method_type_index_at(index2);
1381     bool match = compare_entry_to(k1, cp2, k2, CHECK_false);
1382     if (match) {
1383       return true;
1384     }
1385   } break;
1386 
1387   case JVM_CONSTANT_MethodHandle:
1388   {
1389     int k1 = method_handle_ref_kind_at(index1);
1390     int k2 = cp2->method_handle_ref_kind_at(index2);
1391     if (k1 == k2) {
1392       int i1 = method_handle_index_at(index1);
1393       int i2 = cp2->method_handle_index_at(index2);
1394       bool match = compare_entry_to(i1, cp2, i2, CHECK_false);
1395       if (match) {
1396         return true;
1397       }
1398     }
1399   } break;
1400 
1401   case JVM_CONSTANT_Dynamic:
1402   {
1403     int k1 = bootstrap_name_and_type_ref_index_at(index1);
1404     int k2 = cp2->bootstrap_name_and_type_ref_index_at(index2);
1405     int i1 = bootstrap_methods_attribute_index(index1);
1406     int i2 = cp2->bootstrap_methods_attribute_index(index2);
1407     // separate statements and variables because CHECK_false is used
1408     bool match_entry = compare_entry_to(k1, cp2, k2, CHECK_false);
1409     bool match_operand = compare_operand_to(i1, cp2, i2, CHECK_false);
1410     return (match_entry && match_operand);
1411   } break;
1412 
1413   case JVM_CONSTANT_InvokeDynamic:
1414   {
1415     int k1 = bootstrap_name_and_type_ref_index_at(index1);
1416     int k2 = cp2->bootstrap_name_and_type_ref_index_at(index2);
1417     int i1 = bootstrap_methods_attribute_index(index1);
1418     int i2 = cp2->bootstrap_methods_attribute_index(index2);
1419     // separate statements and variables because CHECK_false is used
1420     bool match_entry = compare_entry_to(k1, cp2, k2, CHECK_false);
1421     bool match_operand = compare_operand_to(i1, cp2, i2, CHECK_false);
1422     return (match_entry && match_operand);
1423   } break;
1424 
1425   case JVM_CONSTANT_String:
1426   {
1427     Symbol* s1 = unresolved_string_at(index1);
1428     Symbol* s2 = cp2->unresolved_string_at(index2);
1429     if (s1 == s2) {
1430       return true;
1431     }
1432   } break;
1433 
1434   case JVM_CONSTANT_Utf8:
1435   {
1436     Symbol* s1 = symbol_at(index1);
1437     Symbol* s2 = cp2->symbol_at(index2);
1438     if (s1 == s2) {
1439       return true;
1440     }
1441   } break;
1442 
1443   // Invalid is used as the tag for the second constant pool entry
1444   // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
1445   // not be seen by itself.
1446   case JVM_CONSTANT_Invalid: // fall through
1447 
1448   default:
1449     ShouldNotReachHere();
1450     break;
1451   }
1452 
1453   return false;
1454 } // end compare_entry_to()

1457 // Resize the operands array with delta_len and delta_size.
1458 // Used in RedefineClasses for CP merge.
1459 void ConstantPool::resize_operands(int delta_len, int delta_size, TRAPS) {
1460   int old_len  = operand_array_length(operands());
1461   int new_len  = old_len + delta_len;
1462   int min_len  = (delta_len > 0) ? old_len : new_len;
1463 
1464   int old_size = operands()->length();
1465   int new_size = old_size + delta_size;
1466   int min_size = (delta_size > 0) ? old_size : new_size;
1467 
1468   ClassLoaderData* loader_data = pool_holder()->class_loader_data();
1469   Array<u2>* new_ops = MetadataFactory::new_array<u2>(loader_data, new_size, CHECK);
1470 
1471   // Set index in the resized array for existing elements only
1472   for (int idx = 0; idx < min_len; idx++) {
1473     int offset = operand_offset_at(idx);                       // offset in original array
1474     operand_offset_at_put(new_ops, idx, offset + 2*delta_len); // offset in resized array
1475   }
1476   // Copy the bootstrap specifiers only
1477   Copy::conjoint_memory_atomic(operands()->adr_at(2*old_len),
1478                                new_ops->adr_at(2*new_len),
1479                                (min_size - 2*min_len) * sizeof(u2));
1480   // Explicitly deallocate old operands array.
1481   // Note, it is not needed for 7u backport.
1482   if ( operands() != NULL) { // the safety check
1483     MetadataFactory::free_array<u2>(loader_data, operands());
1484   }
1485   set_operands(new_ops);
1486 } // end resize_operands()

1531 void ConstantPool::copy_operands(const constantPoolHandle& from_cp,
1532                                  const constantPoolHandle& to_cp,
1533                                  TRAPS) {
1534 
1535   int from_oplen = operand_array_length(from_cp->operands());
1536   int old_oplen  = operand_array_length(to_cp->operands());
1537   if (from_oplen != 0) {
1538     ClassLoaderData* loader_data = to_cp->pool_holder()->class_loader_data();
1539     // append my operands to the target's operands array
1540     if (old_oplen == 0) {
1541       // Can't just reuse from_cp's operand list because of deallocation issues
1542       int len = from_cp->operands()->length();
1543       Array<u2>* new_ops = MetadataFactory::new_array<u2>(loader_data, len, CHECK);
1544       Copy::conjoint_memory_atomic(
1545           from_cp->operands()->adr_at(0), new_ops->adr_at(0), len * sizeof(u2));
1546       to_cp->set_operands(new_ops);
1547     } else {
1548       int old_len  = to_cp->operands()->length();
1549       int from_len = from_cp->operands()->length();
1550       int old_off  = old_oplen * sizeof(u2);
1551       int from_off = from_oplen * sizeof(u2);
1552       // Use the metaspace for the destination constant pool
1553       Array<u2>* new_operands = MetadataFactory::new_array<u2>(loader_data, old_len + from_len, CHECK);
1554       int fillp = 0, len = 0;
1555       // first part of dest
1556       Copy::conjoint_memory_atomic(to_cp->operands()->adr_at(0),
1557                                    new_operands->adr_at(fillp),
1558                                    (len = old_off) * sizeof(u2));
1559       fillp += len;
1560       // first part of src
1561       Copy::conjoint_memory_atomic(from_cp->operands()->adr_at(0),
1562                                    new_operands->adr_at(fillp),
1563                                    (len = from_off) * sizeof(u2));
1564       fillp += len;
1565       // second part of dest
1566       Copy::conjoint_memory_atomic(to_cp->operands()->adr_at(old_off),
1567                                    new_operands->adr_at(fillp),
1568                                    (len = old_len - old_off) * sizeof(u2));
1569       fillp += len;
1570       // second part of src
1571       Copy::conjoint_memory_atomic(from_cp->operands()->adr_at(from_off),
1572                                    new_operands->adr_at(fillp),
1573                                    (len = from_len - from_off) * sizeof(u2));
1574       fillp += len;
1575       assert(fillp == new_operands->length(), "");
1576 
1577       // Adjust indexes in the first part of the copied operands array.
1578       for (int j = 0; j < from_oplen; j++) {
1579         int offset = operand_offset_at(new_operands, old_oplen + j);
1580         assert(offset == operand_offset_at(from_cp->operands(), j), "correct copy");
1581         offset += old_len;  // every new tuple is preceded by old_len extra u2's
1582         operand_offset_at_put(new_operands, old_oplen + j, offset);
1583       }
1584 
1585       // replace target operands array with combined array
1586       to_cp->set_operands(new_operands);
1587     }
1588   }
1589 } // end copy_operands()

1624 // Copy this constant pool's entry at from_i to the constant pool
1625 // to_cp's entry at to_i.
1626 void ConstantPool::copy_entry_to(const constantPoolHandle& from_cp, int from_i,
1627                                         const constantPoolHandle& to_cp, int to_i,
1628                                         TRAPS) {
1629 
1630   int tag = from_cp->tag_at(from_i).value();
1631   switch (tag) {
1632   case JVM_CONSTANT_ClassIndex:
1633   {
1634     jint ki = from_cp->klass_index_at(from_i);
1635     to_cp->klass_index_at_put(to_i, ki);
1636   } break;
1637 
1638   case JVM_CONSTANT_Double:
1639   {
1640     jdouble d = from_cp->double_at(from_i);
1641     to_cp->double_at_put(to_i, d);
1642     // double takes two constant pool entries so init second entry's tag
1643     to_cp->tag_at_put(to_i + 1, JVM_CONSTANT_Invalid);
1644   } break;
1645 
1646   case JVM_CONSTANT_Fieldref:
1647   {
1648     int class_index = from_cp->uncached_klass_ref_index_at(from_i);
1649     int name_and_type_index = from_cp->uncached_name_and_type_ref_index_at(from_i);
1650     to_cp->field_at_put(to_i, class_index, name_and_type_index);
1651   } break;
1652 
1653   case JVM_CONSTANT_Float:
1654   {
1655     jfloat f = from_cp->float_at(from_i);
1656     to_cp->float_at_put(to_i, f);
1657   } break;
1658 
1659   case JVM_CONSTANT_Integer:
1660   {
1661     jint i = from_cp->int_at(from_i);
1662     to_cp->int_at_put(to_i, i);
1663   } break;
1664 
1665   case JVM_CONSTANT_InterfaceMethodref:
1666   {
1667     int class_index = from_cp->uncached_klass_ref_index_at(from_i);
1668     int name_and_type_index = from_cp->uncached_name_and_type_ref_index_at(from_i);
1669     to_cp->interface_method_at_put(to_i, class_index, name_and_type_index);
1670   } break;
1671 
1672   case JVM_CONSTANT_Long:
1673   {
1674     jlong l = from_cp->long_at(from_i);
1675     to_cp->long_at_put(to_i, l);
1676     // long takes two constant pool entries so init second entry's tag
1677     to_cp->tag_at_put(to_i + 1, JVM_CONSTANT_Invalid);
1678   } break;
1679 
1680   case JVM_CONSTANT_Methodref:
1681   {
1682     int class_index = from_cp->uncached_klass_ref_index_at(from_i);
1683     int name_and_type_index = from_cp->uncached_name_and_type_ref_index_at(from_i);
1684     to_cp->method_at_put(to_i, class_index, name_and_type_index);
1685   } break;
1686 
1687   case JVM_CONSTANT_NameAndType:
1688   {
1689     int name_ref_index = from_cp->name_ref_index_at(from_i);
1690     int signature_ref_index = from_cp->signature_ref_index_at(from_i);
1691     to_cp->name_and_type_at_put(to_i, name_ref_index, signature_ref_index);
1692   } break;
1693 
1694   case JVM_CONSTANT_StringIndex:
1695   {
1696     jint si = from_cp->string_index_at(from_i);
1697     to_cp->string_index_at_put(to_i, si);
1698   } break;
1699 
1700   case JVM_CONSTANT_Class:
1701   case JVM_CONSTANT_UnresolvedClass:
1702   case JVM_CONSTANT_UnresolvedClassInError:
1703   {
1704     // Revert to JVM_CONSTANT_ClassIndex
1705     int name_index = from_cp->klass_slot_at(from_i).name_index();
1706     assert(from_cp->tag_at(name_index).is_symbol(), "sanity");
1707     to_cp->klass_index_at_put(to_i, name_index);
1708   } break;
1709 
1710   case JVM_CONSTANT_String:
1711   {
1712     Symbol* s = from_cp->unresolved_string_at(from_i);
1713     to_cp->unresolved_string_at_put(to_i, s);
1714   } break;
1715 
1716   case JVM_CONSTANT_Utf8:
1717   {
1718     Symbol* s = from_cp->symbol_at(from_i);
1719     // Need to increase refcount, the old one will be thrown away and deferenced
1720     s->increment_refcount();
1721     to_cp->symbol_at_put(to_i, s);
1722   } break;
1723 
1724   case JVM_CONSTANT_MethodType:
1725   case JVM_CONSTANT_MethodTypeInError:
1726   {
1727     jint k = from_cp->method_type_index_at(from_i);
1728     to_cp->method_type_index_at_put(to_i, k);
1729   } break;
1730 
1731   case JVM_CONSTANT_MethodHandle:
1732   case JVM_CONSTANT_MethodHandleInError:
1733   {
1734     int k1 = from_cp->method_handle_ref_kind_at(from_i);
1735     int k2 = from_cp->method_handle_index_at(from_i);
1736     to_cp->method_handle_index_at_put(to_i, k1, k2);
1737   } break;
1738 
1739   case JVM_CONSTANT_Dynamic:
1740   case JVM_CONSTANT_DynamicInError:
1741   {
1742     int k1 = from_cp->bootstrap_methods_attribute_index(from_i);
1743     int k2 = from_cp->bootstrap_name_and_type_ref_index_at(from_i);
1744     k1 += operand_array_length(to_cp->operands());  // to_cp might already have operands
1745     to_cp->dynamic_constant_at_put(to_i, k1, k2);
1746   } break;
1747 
1748   case JVM_CONSTANT_InvokeDynamic:
1749   {
1750     int k1 = from_cp->bootstrap_methods_attribute_index(from_i);
1751     int k2 = from_cp->bootstrap_name_and_type_ref_index_at(from_i);
1752     k1 += operand_array_length(to_cp->operands());  // to_cp might already have operands
1753     to_cp->invoke_dynamic_at_put(to_i, k1, k2);
1754   } break;
1755 
1756   // Invalid is used as the tag for the second constant pool entry
1757   // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
1758   // not be seen by itself.
1759   case JVM_CONSTANT_Invalid: // fall through
1760 
1761   default:
1762   {
1763     ShouldNotReachHere();
1764   } break;
1765   }
1766 } // end copy_entry_to()

1769 // constant pool's entry at pattern_i. Returns the index of a
1770 // matching entry or zero (0) if there is no matching entry.

1773

1786 // Compare this constant pool's bootstrap specifier at idx1 to the constant pool
1787 // cp2's bootstrap specifier at idx2.
1788 bool ConstantPool::compare_operand_to(int idx1, const constantPoolHandle& cp2, int idx2, TRAPS) {
1789   int k1 = operand_bootstrap_method_ref_index_at(idx1);
1790   int k2 = cp2->operand_bootstrap_method_ref_index_at(idx2);
1791   bool match = compare_entry_to(k1, cp2, k2, CHECK_false);
1792 
1793   if (!match) {
1794     return false;
1795   }
1796   int argc = operand_argument_count_at(idx1);
1797   if (argc == cp2->operand_argument_count_at(idx2)) {
1798     for (int j = 0; j < argc; j++) {
1799       k1 = operand_argument_index_at(idx1, j);
1800       k2 = cp2->operand_argument_index_at(idx2, j);
1801       match = compare_entry_to(k1, cp2, k2, CHECK_false);
1802       if (!match) {
1803         return false;
1804       }
1805     }
1806     return true;           // got through loop; all elements equal
1807   }
1808   return false;
1809 } // end compare_operand_to()

2071 // Copy cpool bytes.
2072 // Returns:
2073 //    0, in case of OutOfMemoryError
2074 //   -1, in case of internal error
2075 //  > 0, count of the raw cpool bytes that have been copied
2076 int ConstantPool::copy_cpool_bytes(int cpool_size,
2077                                           SymbolHashMap* tbl,
2078                                           unsigned char *bytes) {
2079   u2   idx1, idx2;
2080   jint size  = 0;
2081   jint cnt   = length();
2082   unsigned char *start_bytes = bytes;
2083 
2084   for (jint idx = 1; idx < cnt; idx++) {
2085     u1   tag      = tag_at(idx).value();
2086     jint ent_size = cpool_entry_size(idx);
2087 
2088     assert(size + ent_size <= cpool_size, "Size mismatch");
2089 
2090     *bytes = tag;
2091     DBG(printf("#%03hd tag=%03hd, ", (short)idx, (short)tag));
2092     switch(tag) {
2093       case JVM_CONSTANT_Invalid: {
2094         DBG(printf("JVM_CONSTANT_Invalid"));
2095         break;
2096       }
2097       case JVM_CONSTANT_Unicode: {
2098         assert(false, "Wrong constant pool tag: JVM_CONSTANT_Unicode");
2099         DBG(printf("JVM_CONSTANT_Unicode"));
2100         break;
2101       }
2102       case JVM_CONSTANT_Utf8: {
2103         Symbol* sym = symbol_at(idx);
2104         char*     str = sym->as_utf8();
2105         // Warning! It's crashing on x86 with len = sym->utf8_length()
2106         int       len = (int) strlen(str);
2107         Bytes::put_Java_u2((address) (bytes+1), (u2) len);
2108         for (int i = 0; i < len; i++) {
2109             bytes[3+i] = (u1) str[i];
2110         }
2111         DBG(printf("JVM_CONSTANT_Utf8: %s ", str));
2112         break;
2113       }
2114       case JVM_CONSTANT_Integer: {
2115         jint val = int_at(idx);
2116         Bytes::put_Java_u4((address) (bytes+1), *(u4*)&val);
2117         break;
2118       }
2119       case JVM_CONSTANT_Float: {
2120         jfloat val = float_at(idx);
2121         Bytes::put_Java_u4((address) (bytes+1), *(u4*)&val);
2122         break;
2123       }
2124       case JVM_CONSTANT_Long: {
2125         jlong val = long_at(idx);

2131         jdouble val = double_at(idx);
2132         Bytes::put_Java_u8((address) (bytes+1), *(u8*)&val);

2137       case JVM_CONSTANT_UnresolvedClass:
2138       case JVM_CONSTANT_UnresolvedClassInError: {
2139         assert(!tag_at(idx).is_Qdescriptor_klass(), "Failed to encode QDesc");

2151         idx1 = tbl->symbol_to_value(sym);
2152         assert(idx1 != 0, "Have not found a hashtable entry");

2154         DBG(printf("JVM_CONSTANT_String: idx=#%03hd, %s", idx1, sym->as_utf8()));
2155         break;
2156       }
2157       case JVM_CONSTANT_Fieldref:
2158       case JVM_CONSTANT_Methodref:
2159       case JVM_CONSTANT_InterfaceMethodref: {
2160         idx1 = uncached_klass_ref_index_at(idx);
2161         idx2 = uncached_name_and_type_ref_index_at(idx);

2166       }

2168         idx1 = name_ref_index_at(idx);
2169         idx2 = signature_ref_index_at(idx);
2170         Bytes::put_Java_u2((address) (bytes+1), idx1);
2171         Bytes::put_Java_u2((address) (bytes+3), idx2);
2172         DBG(printf("JVM_CONSTANT_NameAndType: %hd %hd", idx1, idx2));
2173         break;
2174       }
2175       case JVM_CONSTANT_ClassIndex: {
2176         *bytes = JVM_CONSTANT_Class;
2177         idx1 = klass_index_at(idx);
2178         Bytes::put_Java_u2((address) (bytes+1), idx1);
2179         DBG(printf("JVM_CONSTANT_ClassIndex: %hd", idx1));
2180         break;
2181       }
2182       case JVM_CONSTANT_StringIndex: {
2183         *bytes = JVM_CONSTANT_String;

2191         *bytes = JVM_CONSTANT_MethodHandle;
2192         int kind = method_handle_ref_kind_at(idx);

2196         DBG(printf("JVM_CONSTANT_MethodHandle: %d %hd", kind, idx1));

2199       case JVM_CONSTANT_MethodType:
2200       case JVM_CONSTANT_MethodTypeInError: {
2201         *bytes = JVM_CONSTANT_MethodType;
2202         idx1 = method_type_index_at(idx);

2208       case JVM_CONSTANT_DynamicInError: {
2209         *bytes = tag;

2211         idx2 = extract_high_short_from_int(*int_at_addr(idx));
2212         assert(idx2 == bootstrap_name_and_type_ref_index_at(idx), "correct half of u4");

2259 // JSR 292 support for patching constant pool oops after the class is linked and
2260 // the oop array for resolved references are created.
2261 // We can't do this during classfile parsing, which is how the other indexes are
2262 // patched.  The other patches are applied early for some error checking
2263 // so only defer the pseudo_strings.
2264 void ConstantPool::patch_resolved_references(GrowableArray<Handle>* cp_patches) {
2265   for (int index = 1; index < cp_patches->length(); index++) { // Index 0 is unused
2266     Handle patch = cp_patches->at(index);
2267     if (patch.not_null()) {
2268       assert (tag_at(index).is_string(), "should only be string left");
2269       // Patching a string means pre-resolving it.
2270       // The spelling in the constant pool is ignored.
2271       // The constant reference may be any object whatever.
2272       // If it is not a real interned string, the constant is referred
2273       // to as a "pseudo-string", and must be presented to the CP
2274       // explicitly, because it may require scavenging.
2275       int obj_index = cp_to_object_index(index);
2276       pseudo_string_at_put(index, obj_index, patch());
2277      DEBUG_ONLY(cp_patches->at_put(index, Handle());)
2278     }
2279   }
2280 #ifdef ASSERT
2281   // Ensure that all the patches have been used.
2282   for (int index = 0; index < cp_patches->length(); index++) {
2283     assert(cp_patches->at(index).is_null(),
2284            "Unused constant pool patch at %d in class file %s",

2488 void SymbolHashMap::add_entry(Symbol* sym, u2 value) {
2489   char *str = sym->as_utf8();
2490   unsigned int hash = compute_hash(str, sym->utf8_length());
2491   unsigned int index = hash % table_size();
2492 
2493   // check if already in map
2494   // we prefer the first entry since it is more likely to be what was used in
2495   // the class file
2496   for (SymbolHashMapEntry *en = bucket(index); en != NULL; en = en->next()) {
2497     assert(en->symbol() != NULL, "SymbolHashMapEntry symbol is NULL");
2498     if (en->hash() == hash && en->symbol() == sym) {
2499         return;  // already there
2500     }
2501   }
2502 
2503   SymbolHashMapEntry* entry = new SymbolHashMapEntry(hash, sym, value);
2504   entry->set_next(bucket(index));
2505   _buckets[index].set_entry(entry);
2506   assert(entry->symbol() != NULL, "SymbolHashMapEntry symbol is NULL");
2507 }
