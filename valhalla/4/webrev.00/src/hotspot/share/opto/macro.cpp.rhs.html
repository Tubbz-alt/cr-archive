  64 //
  65 // Replace any references to "oldref" in inputs to "use" with "newref".
  66 // Returns the number of replacements made.
  67 //
  68 int PhaseMacroExpand::replace_input(Node *use, Node *oldref, Node *newref) {
  69   int nreplacements = 0;
  70   uint req = use->req();
  71   for (uint j = 0; j < use->len(); j++) {
  72     Node *uin = use->in(j);
  73     if (uin == oldref) {
  74       if (j < req)
  75         use->set_req(j, newref);
  76       else
  77         use->set_prec(j, newref);
  78       nreplacements++;
  79     } else if (j >= req && uin == NULL) {
  80       break;
  81     }
  82   }
  83   return nreplacements;
  84 }

  86 Node* PhaseMacroExpand::opt_bits_test(Node* ctrl, Node* region, int edge, Node* word, int mask, int bits, bool return_fast_path) {
  87   Node* cmp;
  88   if (mask != 0) {
  89     Node* and_node = transform_later(new AndXNode(word, MakeConX(mask)));
  90     cmp = transform_later(new CmpXNode(and_node, MakeConX(bits)));
  91   } else {
  92     cmp = word;
  93   }
  94   Node* bol = transform_later(new BoolNode(cmp, BoolTest::ne));
  95   IfNode* iff = new IfNode( ctrl, bol, PROB_MIN, COUNT_UNKNOWN );
  96   transform_later(iff);
  97 
  98   // Fast path taken.
  99   Node *fast_taken = transform_later(new IfFalseNode(iff));
 100 
 101   // Fast path not-taken, i.e. slow path
 102   Node *slow_taken = transform_later(new IfTrueNode(iff));
 103 
 104   if (return_fast_path) {
 105     region->init_req(edge, slow_taken); // Capture slow-control
 106     return fast_taken;
 107   } else {
 108     region->init_req(edge, fast_taken); // Capture fast-control
 109     return slow_taken;
 110   }
 111 }

 123 //------------------------------make_slow_call---------------------------------
 124 CallNode* PhaseMacroExpand::make_slow_call(CallNode *oldcall, const TypeFunc* slow_call_type,
 125                                            address slow_call, const char* leaf_name, Node* slow_path,
 126                                            Node* parm0, Node* parm1, Node* parm2) {
 127 
 128   // Slow-path call
 129  CallNode *call = leaf_name
 130    ? (CallNode*)new CallLeafNode      ( slow_call_type, slow_call, leaf_name, TypeRawPtr::BOTTOM )
 131    : (CallNode*)new CallStaticJavaNode( slow_call_type, slow_call, OptoRuntime::stub_name(slow_call), oldcall->jvms()->bci(), TypeRawPtr::BOTTOM );
 132 
 133   // Slow path call has no side-effects, uses few values
 134   copy_predefined_input_for_runtime_call(slow_path, oldcall, call );
 135   if (parm0 != NULL)  call->init_req(TypeFunc::Parms+0, parm0);
 136   if (parm1 != NULL)  call->init_req(TypeFunc::Parms+1, parm1);
 137   if (parm2 != NULL)  call->init_req(TypeFunc::Parms+2, parm2);
 138   call->copy_call_debug_info(&_igvn, oldcall);
 139   call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
 140   _igvn.replace_node(oldcall, call);
 141   transform_later(call);
 142 
 143   return call;
 144 }

 146 void PhaseMacroExpand::extract_call_projections(CallNode *call) {
 147   _fallthroughproj = NULL;
 148   _fallthroughcatchproj = NULL;
 149   _ioproj_fallthrough = NULL;
 150   _ioproj_catchall = NULL;
 151   _catchallcatchproj = NULL;
 152   _memproj_fallthrough = NULL;
 153   _memproj_catchall = NULL;
 154   _resproj = NULL;
 155   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
 156     ProjNode *pn = call->fast_out(i)->as_Proj();
 157     switch (pn->_con) {
 158       case TypeFunc::Control:
 159       {
 160         // For Control (fallthrough) and I_O (catch_all_index) we have CatchProj -> Catch -> Proj
 161         _fallthroughproj = pn;
 162         DUIterator_Fast jmax, j = pn->fast_outs(jmax);
 163         const Node *cn = pn->fast_out(j);
 164         if (cn->is_Catch()) {
 165           ProjNode *cpn = NULL;
 166           for (DUIterator_Fast kmax, k = cn->fast_outs(kmax); k < kmax; k++) {
 167             cpn = cn->fast_out(k)->as_Proj();
 168             assert(cpn->is_CatchProj(), "must be a CatchProjNode");
 169             if (cpn->_con == CatchProjNode::fall_through_index)
 170               _fallthroughcatchproj = cpn;
 171             else {
 172               assert(cpn->_con == CatchProjNode::catch_all_index, "must be correct index.");
 173               _catchallcatchproj = cpn;
 174             }
 175           }
 176         }
 177         break;
 178       }
 179       case TypeFunc::I_O:
 180         if (pn->_is_io_use)
 181           _ioproj_catchall = pn;
 182         else
 183           _ioproj_fallthrough = pn;
 184         break;
 185       case TypeFunc::Memory:
 186         if (pn->_is_io_use)
 187           _memproj_catchall = pn;
 188         else
 189           _memproj_fallthrough = pn;
 190         break;
 191       case TypeFunc::Parms:
 192         _resproj = pn;
 193         break;
 194       default:
 195         assert(false, "unexpected projection from allocation node.");
 196     }
 197   }
 198 
 199 }

 206 // Search for a memory operation for the specified memory slice.
 207 static Node *scan_mem_chain(Node *mem, int alias_idx, int offset, Node *start_mem, Node *alloc, PhaseGVN *phase) {
 208   Node *orig_mem = mem;
 209   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 210   const TypeOopPtr *tinst = phase->C->get_adr_type(alias_idx)->isa_oopptr();
 211   while (true) {
 212     if (mem == alloc_mem || mem == start_mem ) {
 213       return mem;  // hit one of our sentinels
 214     } else if (mem->is_MergeMem()) {
 215       mem = mem->as_MergeMem()->memory_at(alias_idx);
 216     } else if (mem->is_Proj() && mem->as_Proj()->_con == TypeFunc::Memory) {
 217       Node *in = mem->in(0);
 218       // we can safely skip over safepoints, calls, locks and membars because we
 219       // already know that the object is safe to eliminate.
 220       if (in->is_Initialize() && in->as_Initialize()->allocation() == alloc) {
 221         return in;
 222       } else if (in->is_Call()) {
 223         CallNode *call = in->as_Call();
 224         if (call->may_modify(tinst, phase)) {
 225           assert(call->is_ArrayCopy(), "ArrayCopy is the only call node that doesn't make allocation escape");
 226           if (call->as_ArrayCopy()->modifies(offset, offset, phase, false)) {
 227             return in;
 228           }
 229         }
 230         mem = in->in(TypeFunc::Memory);
 231       } else if (in->is_MemBar()) {
 232         ArrayCopyNode* ac = NULL;
 233         if (ArrayCopyNode::may_modify(tinst, in->as_MemBar(), phase, ac)) {
 234           assert(ac != NULL && ac->is_clonebasic(), "Only basic clone is a non escaping clone");
 235           return ac;
 236         }
 237         mem = in->in(TypeFunc::Memory);
 238       } else {
 239         assert(false, "unexpected projection");
 240       }
 241     } else if (mem->is_Store()) {
 242       const TypePtr* atype = mem->as_Store()->adr_type();
 243       int adr_idx = phase->C->get_alias_index(atype);
 244       if (adr_idx == alias_idx) {
 245         assert(atype->isa_oopptr(), "address type must be oopptr");
 246         int adr_offset = atype->flattened_offset();
 247         uint adr_iid = atype->is_oopptr()->instance_id();
 248         // Array elements references have the same alias_idx
 249         // but different offset and different instance_id.
 250         if (adr_offset == offset && adr_iid == alloc->_idx)
 251           return mem;
 252       } else {
 253         assert(adr_idx == Compile::AliasIdxRaw, "address must match or be raw");
 254       }
 255       mem = mem->in(MemNode::Memory);
 256     } else if (mem->is_ClearArray()) {
 257       if (!ClearArrayNode::step_through(&mem, alloc->_idx, phase)) {
 258         // Can not bypass initialization of the instance
 259         // we are looking.
 260         debug_only(intptr_t offset;)
 261         assert(alloc == AllocateNode::Ideal_allocation(mem->in(3), phase, offset), "sanity");
 262         InitializeNode* init = alloc->as_Allocate()->initialization();
 263         // We are looking for stored value, return Initialize node
 264         // or memory edge from Allocate node.
 265         if (init != NULL)
 266           return init;
 267         else
 268           return alloc->in(TypeFunc::Memory); // It will produce zero value (see callers).
 269       }
 270       // Otherwise skip it (the call updated 'mem' value).
 271     } else if (mem->Opcode() == Op_SCMemProj) {
 272       mem = mem->in(0);
 273       Node* adr = NULL;
 274       if (mem->is_LoadStore()) {
 275         adr = mem->in(MemNode::Address);
 276       } else {
 277         assert(mem->Opcode() == Op_EncodeISOArray ||
 278                mem->Opcode() == Op_StrCompressedCopy, "sanity");
 279         adr = mem->in(3); // Destination array
 280       }
 281       const TypePtr* atype = adr->bottom_type()->is_ptr();
 282       int adr_idx = phase->C->get_alias_index(atype);
 283       if (adr_idx == alias_idx) {
 284         DEBUG_ONLY(mem->dump();)
 285         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 286         return NULL;
 287       }
 288       mem = mem->in(MemNode::Memory);
 289     } else if (mem->Opcode() == Op_StrInflatedCopy) {
 290       Node* adr = mem->in(3); // Destination array
 291       const TypePtr* atype = adr->bottom_type()->is_ptr();
 292       int adr_idx = phase->C->get_alias_index(atype);
 293       if (adr_idx == alias_idx) {
 294         DEBUG_ONLY(mem->dump();)
 295         assert(false, "Object is not scalar replaceable if a StrInflatedCopy node accesses its field");
 296         return NULL;
 297       }
 298       mem = mem->in(MemNode::Memory);
 299     } else {
 300       return mem;
 301     }
 302     assert(mem != orig_mem, "dead memory loop");
 303   }
 304 }

 306 // Generate loads from source of the arraycopy for fields of
 307 // destination needed at a deoptimization point
 308 Node* PhaseMacroExpand::make_arraycopy_load(ArrayCopyNode* ac, intptr_t offset, Node* ctl, Node* mem, BasicType ft, const Type *ftype, AllocateNode *alloc) {
 309   BasicType bt = ft;
 310   const Type *type = ftype;
 311   if (ft == T_NARROWOOP) {
 312     bt = T_OBJECT;
 313     type = ftype->make_oopptr();
 314   }
 315   Node* res = NULL;
 316   if (ac->is_clonebasic()) {
 317     assert(ac->in(ArrayCopyNode::Src) != ac->in(ArrayCopyNode::Dest), "clone source equals destination");
 318     Node* base = ac->in(ArrayCopyNode::Src)->in(AddPNode::Base);
 319     Node* adr = _igvn.transform(new AddPNode(base, base, MakeConX(offset)));
 320     const TypePtr* adr_type = _igvn.type(base)->is_ptr()->add_offset(offset);
 321     res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 322   } else {
 323     if (ac->modifies(offset, offset, &_igvn, true)) {
 324       assert(ac->in(ArrayCopyNode::Dest) == alloc->result_cast(), "arraycopy destination should be allocation's result");
 325       uint shift = exact_log2(type2aelembytes(bt));
 326       Node* src_pos = ac->in(ArrayCopyNode::SrcPos);
 327       Node* dest_pos = ac->in(ArrayCopyNode::DestPos);
 328       const TypeInt* src_pos_t = _igvn.type(src_pos)->is_int();
 329       const TypeInt* dest_pos_t = _igvn.type(dest_pos)->is_int();
 330 
 331       Node* adr = NULL;
 332       Node* base = ac->in(ArrayCopyNode::Src);
 333       const TypePtr* adr_type = _igvn.type(base)->is_ptr();
 334       assert(adr_type->isa_aryptr(), "only arrays here");
 335       if (src_pos_t->is_con() && dest_pos_t->is_con()) {
 336         intptr_t off = ((src_pos_t->get_con() - dest_pos_t->get_con()) << shift) + offset;
 337         adr = _igvn.transform(new AddPNode(base, base, MakeConX(off)));
 338         adr_type = _igvn.type(adr)->is_ptr();
 339         assert(adr_type == _igvn.type(base)->is_aryptr()->add_field_offset_and_offset(off), "incorrect address type");
 340         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 341           // Don't emit a new load from src if src == dst but try to get the value from memory instead
 342           return value_from_mem(ac->in(TypeFunc::Memory), ctl, ft, ftype, adr_type->isa_oopptr(), alloc);
 343         }
 344       } else {
 345         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 346           // Non constant offset in the array: we can't statically
 347           // determine the value
 348           return NULL;
 349         }
 350         Node* diff = _igvn.transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 351 #ifdef _LP64
 352         diff = _igvn.transform(new ConvI2LNode(diff));
 353 #endif
 354         diff = _igvn.transform(new LShiftXNode(diff, intcon(shift)));
 355 
 356         Node* off = _igvn.transform(new AddXNode(MakeConX(offset), diff));
 357         adr = _igvn.transform(new AddPNode(base, base, off));
 358         // In the case of a flattened value type array, each field has its
 359         // own slice so we need to extract the field being accessed from
 360         // the address computation
 361         adr_type = adr_type->is_aryptr()->add_field_offset_and_offset(offset)->add_offset(Type::OffsetBot);
 362         adr = _igvn.transform(new CastPPNode(adr, adr_type));
 363       }
 364       res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 365     }
 366   }
 367   if (res != NULL) {
 368     res = _igvn.transform(res);
 369     if (ftype->isa_narrowoop()) {
 370       // PhaseMacroExpand::scalar_replacement adds DecodeN nodes
 371       assert(res->isa_DecodeN(), "should be narrow oop");
 372       res = _igvn.transform(new EncodePNode(res, ftype));
 373     }
 374     return res;
 375   }
 376   return NULL;
 377 }

 379 //
 380 // Given a Memory Phi, compute a value Phi containing the values from stores
 381 // on the input paths.
 382 // Note: this function is recursive, its depth is limited by the "level" argument
 383 // Returns the computed Phi, or NULL if it cannot compute it.
 384 Node *PhaseMacroExpand::value_from_mem_phi(Node *mem, BasicType ft, const Type *phi_type, const TypeOopPtr *adr_t, AllocateNode *alloc, Node_Stack *value_phis, int level) {
 385   assert(mem->is_Phi(), "sanity");
 386   int alias_idx = C->get_alias_index(adr_t);
 387   int offset = adr_t->flattened_offset();
 388   int instance_id = adr_t->instance_id();
 389 
 390   // Check if an appropriate value phi already exists.
 391   Node* region = mem->in(0);
 392   for (DUIterator_Fast kmax, k = region->fast_outs(kmax); k < kmax; k++) {
 393     Node* phi = region->fast_out(k);
 394     if (phi->is_Phi() && phi != mem &&
 395         phi->as_Phi()->is_same_inst_field(phi_type, (int)mem->_idx, instance_id, alias_idx, offset)) {
 396       return phi;
 397     }
 398   }
 399   // Check if an appropriate new value phi already exists.
 400   Node* new_phi = value_phis->find(mem->_idx);
 401   if (new_phi != NULL)
 402     return new_phi;
 403 
 404   if (level <= 0) {
 405     return NULL; // Give up: phi tree too deep
 406   }
 407   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 408   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 409 
 410   uint length = mem->req();
 411   GrowableArray <Node *> values(length, length, NULL, false);
 412 
 413   // create a new Phi for the value
 414   PhiNode *phi = new PhiNode(mem->in(0), phi_type, NULL, mem->_idx, instance_id, alias_idx, offset);
 415   transform_later(phi);
 416   value_phis->push(phi, mem->_idx);
 417 
 418   for (uint j = 1; j < length; j++) {
 419     Node *in = mem->in(j);
 420     if (in == NULL || in->is_top()) {
 421       values.at_put(j, in);
 422     } else  {
 423       Node *val = scan_mem_chain(in, alias_idx, offset, start_mem, alloc, &_igvn);
 424       if (val == start_mem || val == alloc_mem) {
 425         // hit a sentinel, return appropriate 0 value
 426         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 427         if (default_value != NULL) {
 428           values.at_put(j, default_value);
 429         } else {
 430           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 431           values.at_put(j, _igvn.zerocon(ft));
 432         }
 433         continue;
 434       }
 435       if (val->is_Initialize()) {
 436         val = val->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 437       }
 438       if (val == NULL) {
 439         return NULL;  // can't find a value on this path
 440       }
 441       if (val == mem) {
 442         values.at_put(j, mem);
 443       } else if (val->is_Store()) {
 444         Node* n = val->in(MemNode::ValueIn);
 445         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 446         n = bs->step_over_gc_barrier(n);
 447         values.at_put(j, n);
 448       } else if(val->is_Proj() && val->in(0) == alloc) {
 449         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 450         if (default_value != NULL) {
 451           values.at_put(j, default_value);
 452         } else {
 453           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 454           values.at_put(j, _igvn.zerocon(ft));
 455         }
 456       } else if (val->is_Phi()) {
 457         val = value_from_mem_phi(val, ft, phi_type, adr_t, alloc, value_phis, level-1);
 458         if (val == NULL) {
 459           return NULL;
 460         }
 461         values.at_put(j, val);
 462       } else if (val->Opcode() == Op_SCMemProj) {
 463         assert(val->in(0)->is_LoadStore() ||
 464                val->in(0)->Opcode() == Op_EncodeISOArray ||
 465                val->in(0)->Opcode() == Op_StrCompressedCopy, "sanity");
 466         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 467         return NULL;
 468       } else if (val->is_ArrayCopy()) {
 469         Node* res = make_arraycopy_load(val->as_ArrayCopy(), offset, val->in(0), val->in(TypeFunc::Memory), ft, phi_type, alloc);
 470         if (res == NULL) {
 471           return NULL;
 472         }
 473         values.at_put(j, res);
 474       } else {
 475 #ifdef ASSERT
 476         val->dump();
 477         assert(false, "unknown node on this path");
 478 #endif
 479         return NULL;  // unknown node on this path
 480       }
 481     }
 482   }
 483   // Set Phi's inputs
 484   for (uint j = 1; j < length; j++) {
 485     if (values.at(j) == mem) {
 486       phi->init_req(j, phi);
 487     } else {
 488       phi->init_req(j, values.at(j));
 489     }
 490   }
 491   return phi;
 492 }

 494 // Search the last value stored into the object's field.
 495 Node *PhaseMacroExpand::value_from_mem(Node *sfpt_mem, Node *sfpt_ctl, BasicType ft, const Type *ftype, const TypeOopPtr *adr_t, AllocateNode *alloc) {
 496   assert(adr_t->is_known_instance_field(), "instance required");
 497   int instance_id = adr_t->instance_id();
 498   assert((uint)instance_id == alloc->_idx, "wrong allocation");
 499 
 500   int alias_idx = C->get_alias_index(adr_t);
 501   int offset = adr_t->flattened_offset();
 502   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 503   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 504   Arena *a = Thread::current()->resource_area();
 505   VectorSet visited(a);
 506 
 507   bool done = sfpt_mem == alloc_mem;
 508   Node *mem = sfpt_mem;
 509   while (!done) {
 510     if (visited.test_set(mem->_idx)) {
 511       return NULL;  // found a loop, give up
 512     }
 513     mem = scan_mem_chain(mem, alias_idx, offset, start_mem, alloc, &_igvn);
 514     if (mem == start_mem || mem == alloc_mem) {
 515       done = true;  // hit a sentinel, return appropriate 0 value
 516     } else if (mem->is_Initialize()) {
 517       mem = mem->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 518       if (mem == NULL) {
 519         done = true; // Something went wrong.
 520       } else if (mem->is_Store()) {
 521         const TypePtr* atype = mem->as_Store()->adr_type();
 522         assert(C->get_alias_index(atype) == Compile::AliasIdxRaw, "store is correct memory slice");
 523         done = true;
 524       }
 525     } else if (mem->is_Store()) {
 526       const TypeOopPtr* atype = mem->as_Store()->adr_type()->isa_oopptr();
 527       assert(atype != NULL, "address type must be oopptr");
 528       assert(C->get_alias_index(atype) == alias_idx &&
 529              atype->is_known_instance_field() && atype->flattened_offset() == offset &&
 530              atype->instance_id() == instance_id, "store is correct memory slice");
 531       done = true;
 532     } else if (mem->is_Phi()) {
 533       // try to find a phi's unique input
 534       Node *unique_input = NULL;
 535       Node *top = C->top();
 536       for (uint i = 1; i < mem->req(); i++) {
 537         Node *n = scan_mem_chain(mem->in(i), alias_idx, offset, start_mem, alloc, &_igvn);
 538         if (n == NULL || n == top || n == mem) {
 539           continue;
 540         } else if (unique_input == NULL) {
 541           unique_input = n;
 542         } else if (unique_input != n) {
 543           unique_input = top;
 544           break;
 545         }
 546       }
 547       if (unique_input != NULL && unique_input != top) {
 548         mem = unique_input;
 549       } else {
 550         done = true;
 551       }
 552     } else if (mem->is_ArrayCopy()) {
 553       done = true;
 554     } else {
 555       assert(false, "unexpected node");
 556     }
 557   }
 558   if (mem != NULL) {
 559     if (mem == start_mem || mem == alloc_mem) {
 560       // hit a sentinel, return appropriate 0 value
 561       Node* default_value = alloc->in(AllocateNode::DefaultValue);
 562       if (default_value != NULL) {
 563         return default_value;
 564       }
 565       assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 566       return _igvn.zerocon(ft);
 567     } else if (mem->is_Store()) {
 568       Node* n = mem->in(MemNode::ValueIn);
 569       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 570       n = bs->step_over_gc_barrier(n);
 571       return n;
 572     } else if (mem->is_Phi()) {
 573       // attempt to produce a Phi reflecting the values on the input paths of the Phi
 574       Node_Stack value_phis(a, 8);
 575       Node * phi = value_from_mem_phi(mem, ft, ftype, adr_t, alloc, &value_phis, ValueSearchLimit);
 576       if (phi != NULL) {
 577         return phi;
 578       } else {
 579         // Kill all new Phis
 580         while(value_phis.is_nonempty()) {
 581           Node* n = value_phis.node();
 582           _igvn.replace_node(n, C->top());
 583           value_phis.pop();
 584         }
 585       }
 586     } else if (mem->is_ArrayCopy()) {
 587       Node* ctl = mem->in(0);
 588       Node* m = mem->in(TypeFunc::Memory);
 589       if (sfpt_ctl->is_Proj() && sfpt_ctl->as_Proj()->is_uncommon_trap_proj(Deoptimization::Reason_none)) {
 590         // pin the loads in the uncommon trap path
 591         ctl = sfpt_ctl;
 592         m = sfpt_mem;
 593       }
 594       return make_arraycopy_load(mem->as_ArrayCopy(), offset, ctl, m, ft, ftype, alloc);
 595     }
 596   }
 597   // Something went wrong.
 598   return NULL;
 599 }

 601 // Search the last value stored into the value type's fields.
 602 Node* PhaseMacroExpand::value_type_from_mem(Node* mem, Node* ctl, ciValueKlass* vk, const TypeAryPtr* adr_type, int offset, AllocateNode* alloc) {
 603   // Subtract the offset of the first field to account for the missing oop header
 604   offset -= vk->first_field_offset();
 605   // Create a new ValueTypeNode and retrieve the field values from memory
 606   ValueTypeNode* vt = ValueTypeNode::make_uninitialized(_igvn, vk)->as_ValueType();
 607   for (int i = 0; i < vk->nof_declared_nonstatic_fields(); ++i) {
 608     ciType* field_type = vt->field_type(i);
 609     int field_offset = offset + vt->field_offset(i);
 610     // Each value type field has its own memory slice
 611     adr_type = adr_type->with_field_offset(field_offset);
 612     Node* value = NULL;
 613     if (vt->field_is_flattened(i)) {
 614       value = value_type_from_mem(mem, ctl, field_type->as_value_klass(), adr_type, field_offset, alloc);
 615     } else {
 616       const Type* ft = Type::get_const_type(field_type);
 617       BasicType bt = field_type->basic_type();
 618       if (UseCompressedOops && !is_java_primitive(bt)) {
 619         ft = ft->make_narrowoop();
 620         bt = T_NARROWOOP;
 621       }
 622       value = value_from_mem(mem, ctl, bt, ft, adr_type, alloc);
 623       if (value != NULL && ft->isa_narrowoop()) {
 624         assert(UseCompressedOops, "unexpected narrow oop");
 625         value = transform_later(new DecodeNNode(value, value->get_ptr_type()));
 626       }
 627     }
 628     if (value != NULL) {
 629       vt->set_field_value(i, value);
 630     } else {
 631       // We might have reached the TrackedInitializationLimit
 632       return NULL;
 633     }
 634   }
 635   return transform_later(vt);
 636 }

 638 // Check the possibility of scalar replacement.
 639 bool PhaseMacroExpand::can_eliminate_allocation(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 640   //  Scan the uses of the allocation to check for anything that would
 641   //  prevent us from eliminating it.
 642   NOT_PRODUCT( const char* fail_eliminate = NULL; )
 643   DEBUG_ONLY( Node* disq_node = NULL; )
 644   bool  can_eliminate = true;
 645 
 646   Node* res = alloc->result_cast();
 647   const TypeOopPtr* res_type = NULL;
 648   if (res == NULL) {
 649     // All users were eliminated.
 650   } else if (!res->is_CheckCastPP()) {
 651     NOT_PRODUCT(fail_eliminate = "Allocation does not have unique CheckCastPP";)
 652     can_eliminate = false;
 653   } else {
 654     res_type = _igvn.type(res)->isa_oopptr();
 655     if (res_type == NULL) {
 656       NOT_PRODUCT(fail_eliminate = "Neither instance or array allocation";)
 657       can_eliminate = false;
 658     } else if (res_type->isa_aryptr()) {
 659       int length = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 660       if (length < 0) {
 661         NOT_PRODUCT(fail_eliminate = "Array's size is not constant";)
 662         can_eliminate = false;
 663       }
 664     }
 665   }
 666 
 667   if (can_eliminate && res != NULL) {
 668     for (DUIterator_Fast jmax, j = res->fast_outs(jmax);
 669                                j < jmax && can_eliminate; j++) {
 670       Node* use = res->fast_out(j);
 671 
 672       if (use->is_AddP()) {
 673         const TypePtr* addp_type = _igvn.type(use)->is_ptr();
 674         int offset = addp_type->offset();
 675 
 676         if (offset == Type::OffsetTop || offset == Type::OffsetBot) {
 677           NOT_PRODUCT(fail_eliminate = "Undefined field referrence";)
 678           can_eliminate = false;
 679           break;
 680         }
 681         for (DUIterator_Fast kmax, k = use->fast_outs(kmax);
 682                                    k < kmax && can_eliminate; k++) {
 683           Node* n = use->fast_out(k);
 684           if (!n->is_Store() && n->Opcode() != Op_CastP2X &&
 685               SHENANDOAHGC_ONLY((!UseShenandoahGC || !ShenandoahBarrierSetC2::is_shenandoah_wb_pre_call(n)) &&)
 686               !(n->is_ArrayCopy() &&
 687                 n->as_ArrayCopy()->is_clonebasic() &&
 688                 n->in(ArrayCopyNode::Dest) == use)) {
 689             DEBUG_ONLY(disq_node = n;)
 690             if (n->is_Load() || n->is_LoadStore()) {
 691               NOT_PRODUCT(fail_eliminate = "Field load";)
 692             } else {
 693               NOT_PRODUCT(fail_eliminate = "Not store field reference";)
 694             }
 695             can_eliminate = false;
 696           }
 697         }
 698       } else if (use->is_ArrayCopy() &&
 699                  (use->as_ArrayCopy()->is_arraycopy_validated() ||
 700                   use->as_ArrayCopy()->is_copyof_validated() ||
 701                   use->as_ArrayCopy()->is_copyofrange_validated()) &&
 702                  use->in(ArrayCopyNode::Dest) == res) {
 703         // ok to eliminate
 704       } else if (use->is_SafePoint()) {
 705         SafePointNode* sfpt = use->as_SafePoint();
 706         if (sfpt->is_Call() && sfpt->as_Call()->has_non_debug_use(res)) {
 707           // Object is passed as argument.
 708           DEBUG_ONLY(disq_node = use;)
 709           NOT_PRODUCT(fail_eliminate = "Object is passed as argument";)
 710           can_eliminate = false;
 711         }
 712         Node* sfptMem = sfpt->memory();
 713         if (sfptMem == NULL || sfptMem->is_top()) {
 714           DEBUG_ONLY(disq_node = use;)
 715           NOT_PRODUCT(fail_eliminate = "NULL or TOP memory";)
 716           can_eliminate = false;
 717         } else {
 718           safepoints.append_if_missing(sfpt);
 719         }
 720       } else if (use->is_ValueType() && use->isa_ValueType()->get_oop() == res) {
 721         // ok to eliminate
 722       } else if (use->Opcode() == Op_StoreX && use->in(MemNode::Address) == res) {
 723         // store to mark work
 724       } else if (use->Opcode() != Op_CastP2X) { // CastP2X is used by card mark
 725         if (use->is_Phi()) {
 726           if (use->outcnt() == 1 && use->unique_out()->Opcode() == Op_Return) {
 727             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 728           } else {
 729             NOT_PRODUCT(fail_eliminate = "Object is referenced by Phi";)
 730           }
 731           DEBUG_ONLY(disq_node = use;)
 732         } else {
 733           if (use->Opcode() == Op_Return) {
 734             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 735           } else {
 736             NOT_PRODUCT(fail_eliminate = "Object is referenced by node";)
 737           }
 738           DEBUG_ONLY(disq_node = use;)
 739         }
 740         can_eliminate = false;
 741       } else {
 742         assert(use->Opcode() == Op_CastP2X, "should be");
 743         assert(!use->has_out_with(Op_OrL), "should have been removed because oop is never null");
 744       }
 745     }
 746   }
 747 
 748 #ifndef PRODUCT
 749   if (PrintEliminateAllocations) {
 750     if (can_eliminate) {
 751       tty->print("Scalar ");
 752       if (res == NULL)
 753         alloc->dump();
 754       else
 755         res->dump();
 756     } else if (alloc->_is_scalar_replaceable) {
 757       tty->print("NotScalar (%s)", fail_eliminate);
 758       if (res == NULL)
 759         alloc->dump();
 760       else
 761         res->dump();
 762 #ifdef ASSERT
 763       if (disq_node != NULL) {
 764           tty->print("  >>>> ");
 765           disq_node->dump();
 766       }
 767 #endif /*ASSERT*/
 768     }
 769   }
 770 #endif
 771   return can_eliminate;
 772 }

 774 // Do scalar replacement.
 775 bool PhaseMacroExpand::scalar_replacement(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 776   GrowableArray <SafePointNode *> safepoints_done;
 777 
 778   ciKlass* klass = NULL;
 779   ciInstanceKlass* iklass = NULL;
 780   int nfields = 0;
 781   int array_base = 0;
 782   int element_size = 0;
 783   BasicType basic_elem_type = T_ILLEGAL;
 784   ciType* elem_type = NULL;
 785 
 786   Node* res = alloc->result_cast();
 787   assert(res == NULL || res->is_CheckCastPP(), "unexpected AllocateNode result");
 788   const TypeOopPtr* res_type = NULL;
 789   if (res != NULL) { // Could be NULL when there are no users
 790     res_type = _igvn.type(res)->isa_oopptr();
 791   }
 792 
 793   if (res != NULL) {
 794     klass = res_type->klass();
 795     if (res_type->isa_instptr()) {
 796       // find the fields of the class which will be needed for safepoint debug information
 797       assert(klass->is_instance_klass(), "must be an instance klass.");
 798       iklass = klass->as_instance_klass();
 799       nfields = iklass->nof_nonstatic_fields();
 800     } else {
 801       // find the array's elements which will be needed for safepoint debug information
 802       nfields = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 803       assert(klass->is_array_klass() && nfields >= 0, "must be an array klass.");
 804       elem_type = klass->as_array_klass()->element_type();
 805       basic_elem_type = elem_type->basic_type();
 806       if (elem_type->is_valuetype() && !klass->is_value_array_klass()) {
 807         assert(basic_elem_type == T_VALUETYPE, "unexpected element basic type");
 808         basic_elem_type = T_OBJECT;
 809       }
 810       array_base = arrayOopDesc::base_offset_in_bytes(basic_elem_type);
 811       element_size = type2aelembytes(basic_elem_type);
 812       if (klass->is_value_array_klass()) {
 813         // Flattened value type array
 814         element_size = klass->as_value_array_klass()->element_byte_size();
 815       }
 816     }
 817   }
 818   //
 819   // Process the safepoint uses
 820   //
 821   Unique_Node_List value_worklist;
 822   while (safepoints.length() > 0) {
 823     SafePointNode* sfpt = safepoints.pop();
 824     Node* mem = sfpt->memory();
 825     Node* ctl = sfpt->control();
 826     assert(sfpt->jvms() != NULL, "missed JVMS");
 827     // Fields of scalar objs are referenced only at the end
 828     // of regular debuginfo at the last (youngest) JVMS.
 829     // Record relative start index.
 830     uint first_ind = (sfpt->req() - sfpt->jvms()->scloff());
 831     SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(res_type,
 832 #ifdef ASSERT
 833                                                  alloc,
 834 #endif
 835                                                  first_ind, nfields);
 836     sobj->init_req(0, C->root());
 837     transform_later(sobj);
 838 
 839     // Scan object's fields adding an input to the safepoint for each field.
 840     for (int j = 0; j < nfields; j++) {
 841       intptr_t offset;
 842       ciField* field = NULL;
 843       if (iklass != NULL) {
 844         field = iklass->nonstatic_field_at(j);
 845         offset = field->offset();
 846         elem_type = field->type();
 847         basic_elem_type = field->layout_type();
 848         assert(!field->is_flattened(), "flattened value type fields should not have safepoint uses");
 849       } else {
 850         offset = array_base + j * (intptr_t)element_size;
 851       }
 852 
 853       const Type *field_type;
 854       // The next code is taken from Parse::do_get_xxx().
 855       if (is_reference_type(basic_elem_type)) {
 856         if (!elem_type->is_loaded()) {
 857           field_type = TypeInstPtr::BOTTOM;
 858         } else if (field != NULL && field->is_static_constant()) {
 859           // This can happen if the constant oop is non-perm.
 860           ciObject* con = field->constant_value().as_object();
 861           // Do not "join" in the previous type; it doesn't add value,
 862           // and may yield a vacuous result if the field is of interface type.
 863           field_type = TypeOopPtr::make_from_constant(con)->isa_oopptr();
 864           assert(field_type != NULL, "field singleton type must be consistent");
 865         } else {
 866           field_type = TypeOopPtr::make_from_klass(elem_type->as_klass());
 867         }
 868         if (UseCompressedOops) {
 869           field_type = field_type->make_narrowoop();
 870           basic_elem_type = T_NARROWOOP;
 871         }
 872       } else {
 873         field_type = Type::get_const_basic_type(basic_elem_type);
 874       }
 875 
 876       Node* field_val = NULL;
 877       const TypeOopPtr* field_addr_type = res_type->add_offset(offset)->isa_oopptr();
 878       if (klass->is_value_array_klass()) {
 879         ciValueKlass* vk = elem_type->as_value_klass();
 880         assert(vk->flatten_array(), "must be flattened");
 881         field_val = value_type_from_mem(mem, ctl, vk, field_addr_type->isa_aryptr(), 0, alloc);
 882       } else {
 883         field_val = value_from_mem(mem, ctl, basic_elem_type, field_type, field_addr_type, alloc);
 884       }
 885       if (field_val == NULL) {
 886         // We weren't able to find a value for this field,
 887         // give up on eliminating this allocation.
 888 
 889         // Remove any extra entries we added to the safepoint.
 890         uint last = sfpt->req() - 1;
 891         for (int k = 0;  k < j; k++) {
 892           sfpt->del_req(last--);
 893         }
 894         _igvn._worklist.push(sfpt);
 895         // rollback processed safepoints
 896         while (safepoints_done.length() > 0) {
 897           SafePointNode* sfpt_done = safepoints_done.pop();
 898           // remove any extra entries we added to the safepoint
 899           last = sfpt_done->req() - 1;
 900           for (int k = 0;  k < nfields; k++) {
 901             sfpt_done->del_req(last--);
 902           }
 903           JVMState *jvms = sfpt_done->jvms();
 904           jvms->set_endoff(sfpt_done->req());
 905           // Now make a pass over the debug information replacing any references
 906           // to SafePointScalarObjectNode with the allocated object.
 907           int start = jvms->debug_start();
 908           int end   = jvms->debug_end();
 909           for (int i = start; i < end; i++) {
 910             if (sfpt_done->in(i)->is_SafePointScalarObject()) {
 911               SafePointScalarObjectNode* scobj = sfpt_done->in(i)->as_SafePointScalarObject();
 912               if (scobj->first_index(jvms) == sfpt_done->req() &&
 913                   scobj->n_fields() == (uint)nfields) {
 914                 assert(scobj->alloc() == alloc, "sanity");
 915                 sfpt_done->set_req(i, res);
 916               }
 917             }
 918           }
 919           _igvn._worklist.push(sfpt_done);
 920         }
 921 #ifndef PRODUCT
 922         if (PrintEliminateAllocations) {
 923           if (field != NULL) {
 924             tty->print("=== At SafePoint node %d can't find value of Field: ",
 925                        sfpt->_idx);
 926             field->print();
 927             int field_idx = C->get_alias_index(field_addr_type);
 928             tty->print(" (alias_idx=%d)", field_idx);
 929           } else { // Array's element
 930             tty->print("=== At SafePoint node %d can't find value of array element [%d]",
 931                        sfpt->_idx, j);
 932           }
 933           tty->print(", which prevents elimination of: ");
 934           if (res == NULL)
 935             alloc->dump();
 936           else
 937             res->dump();
 938         }
 939 #endif
 940         return false;
 941       }
 942       if (field_val->is_ValueType()) {
 943         // Keep track of value types to scalarize them later
 944         value_worklist.push(field_val);
 945       } else if (UseCompressedOops && field_type->isa_narrowoop()) {
 946         // Enable "DecodeN(EncodeP(Allocate)) --> Allocate" transformation
 947         // to be able scalar replace the allocation.
 948         if (field_val->is_EncodeP()) {
 949           field_val = field_val->in(1);
 950         } else {
 951           field_val = transform_later(new DecodeNNode(field_val, field_val->get_ptr_type()));
 952         }
 953       }
 954       sfpt->add_req(field_val);
 955     }
 956     JVMState *jvms = sfpt->jvms();
 957     jvms->set_endoff(sfpt->req());
 958     // Now make a pass over the debug information replacing any references
 959     // to the allocated object with "sobj"
 960     int start = jvms->debug_start();
 961     int end   = jvms->debug_end();
 962     sfpt->replace_edges_in_range(res, sobj, start, end);
 963     _igvn._worklist.push(sfpt);
 964     safepoints_done.append_if_missing(sfpt); // keep it for rollback
 965   }
 966   // Scalarize value types that were added to the safepoint
 967   for (uint i = 0; i < value_worklist.size(); ++i) {
 968     Node* vt = value_worklist.at(i);
 969     vt->as_ValueType()->make_scalar_in_safepoints(&_igvn);
 970   }
 971   return true;
 972 }

 974 static void disconnect_projections(MultiNode* n, PhaseIterGVN& igvn) {
 975   Node* ctl_proj = n->proj_out_or_null(TypeFunc::Control);
 976   Node* mem_proj = n->proj_out_or_null(TypeFunc::Memory);
 977   if (ctl_proj != NULL) {
 978     igvn.replace_node(ctl_proj, n->in(0));
 979   }
 980   if (mem_proj != NULL) {
 981     igvn.replace_node(mem_proj, n->in(TypeFunc::Memory));
 982   }

 985 // Process users of eliminated allocation.
 986 void PhaseMacroExpand::process_users_of_allocation(CallNode *alloc) {
 987   Node* res = alloc->result_cast();
 988   if (res != NULL) {
 989     for (DUIterator_Last jmin, j = res->last_outs(jmin); j >= jmin; ) {
 990       Node *use = res->last_out(j);
 991       uint oc1 = res->outcnt();
 992 
 993       if (use->is_AddP()) {
 994         for (DUIterator_Last kmin, k = use->last_outs(kmin); k >= kmin; ) {
 995           Node *n = use->last_out(k);
 996           uint oc2 = use->outcnt();
 997           if (n->is_Store()) {
 998 #ifdef ASSERT
 999             // Verify that there is no dependent MemBarVolatile nodes,
1000             // they should be removed during IGVN, see MemBarNode::Ideal().
1001             for (DUIterator_Fast pmax, p = n->fast_outs(pmax);
1002                                        p < pmax; p++) {
1003               Node* mb = n->fast_out(p);
1004               assert(mb->is_Initialize() || !mb->is_MemBar() ||
1005                      mb->req() <= MemBarNode::Precedent ||
1006                      mb->in(MemBarNode::Precedent) != n,
1007                      "MemBarVolatile should be eliminated for non-escaping object");
1008             }
1009 #endif
1010             _igvn.replace_node(n, n->in(MemNode::Memory));
1011           } else if (n->is_ArrayCopy()) {
1012             // Disconnect ArrayCopy node
1013             ArrayCopyNode* ac = n->as_ArrayCopy();
1014             assert(ac->is_clonebasic(), "unexpected array copy kind");
1015             Node* membar_after = ac->proj_out(TypeFunc::Control)->unique_ctrl_out();
1016             disconnect_projections(ac, _igvn);
1017             assert(alloc->in(0)->is_Proj() && alloc->in(0)->in(0)->Opcode() == Op_MemBarCPUOrder, "mem barrier expected before allocation");
1018             Node* membar_before = alloc->in(0)->in(0);
1019             disconnect_projections(membar_before->as_MemBar(), _igvn);
1020             if (membar_after->is_MemBar()) {
1021               disconnect_projections(membar_after->as_MemBar(), _igvn);
1022             }
1023           } else {
1024             eliminate_gc_barrier(n);
1025           }
1026           k -= (oc2 - use->outcnt());
1027         }
1028         _igvn.remove_dead_node(use);
1029       } else if (use->is_ArrayCopy()) {
1030         // Disconnect ArrayCopy node
1031         ArrayCopyNode* ac = use->as_ArrayCopy();
1032         assert(ac->is_arraycopy_validated() ||
1033                ac->is_copyof_validated() ||
1034                ac->is_copyofrange_validated(), "unsupported");
1035         CallProjections* callprojs = ac->extract_projections(true);
1036 
1037         _igvn.replace_node(callprojs->fallthrough_ioproj, ac->in(TypeFunc::I_O));
1038         _igvn.replace_node(callprojs->fallthrough_memproj, ac->in(TypeFunc::Memory));
1039         _igvn.replace_node(callprojs->fallthrough_catchproj, ac->in(TypeFunc::Control));
1040 
1041         // Set control to top. IGVN will remove the remaining projections
1042         ac->set_req(0, top());
1043         ac->replace_edge(res, top());
1044 
1045         // Disconnect src right away: it can help find new
1046         // opportunities for allocation elimination
1047         Node* src = ac->in(ArrayCopyNode::Src);
1048         ac->replace_edge(src, top());
1049         // src can be top at this point if src and dest of the
1050         // arraycopy were the same
1051         if (src->outcnt() == 0 && !src->is_top()) {
1052           _igvn.remove_dead_node(src);
1053         }
1054 
1055         _igvn._worklist.push(ac);
1056       } else if (use->is_ValueType()) {
1057         assert(use->isa_ValueType()->get_oop() == res, "unexpected value type use");
1058          _igvn.rehash_node_delayed(use);
1059         use->isa_ValueType()->set_oop(_igvn.zerocon(T_VALUETYPE));
1060       } else if (use->is_Store()) {
1061         _igvn.replace_node(use, use->in(MemNode::Memory));
1062       } else {
1063         eliminate_gc_barrier(use);
1064       }
1065       j -= (oc1 - res->outcnt());
1066     }
1067     assert(res->outcnt() == 0, "all uses of allocated objects must be deleted");
1068     _igvn.remove_dead_node(res);
1069   }
1070 
1071   //
1072   // Process other users of allocation's projections
1073   //
1074   if (_resproj != NULL && _resproj->outcnt() != 0) {
1075     // First disconnect stores captured by Initialize node.
1076     // If Initialize node is eliminated first in the following code,
1077     // it will kill such stores and DUIterator_Last will assert.
1078     for (DUIterator_Fast jmax, j = _resproj->fast_outs(jmax);  j < jmax; j++) {
1079       Node *use = _resproj->fast_out(j);
1080       if (use->is_AddP()) {
1081         // raw memory addresses used only by the initialization
1082         _igvn.replace_node(use, C->top());
1083         --j; --jmax;
1084       }
1085     }
1086     for (DUIterator_Last jmin, j = _resproj->last_outs(jmin); j >= jmin; ) {
1087       Node *use = _resproj->last_out(j);
1088       uint oc1 = _resproj->outcnt();
1089       if (use->is_Initialize()) {
1090         // Eliminate Initialize node.
1091         InitializeNode *init = use->as_Initialize();
1092         assert(init->outcnt() <= 2, "only a control and memory projection expected");
1093         Node *ctrl_proj = init->proj_out_or_null(TypeFunc::Control);
1094         if (ctrl_proj != NULL) {
1095           _igvn.replace_node(ctrl_proj, init->in(TypeFunc::Control));
1096 #ifdef ASSERT
1097           Node* tmp = init->in(TypeFunc::Control);
1098           assert(tmp == _fallthroughcatchproj, "allocation control projection");
1099 #endif
1100         }
1101         Node *mem_proj = init->proj_out_or_null(TypeFunc::Memory);
1102         if (mem_proj != NULL) {
1103           Node *mem = init->in(TypeFunc::Memory);
1104 #ifdef ASSERT
1105           if (mem->is_MergeMem()) {
1106             assert(mem->in(TypeFunc::Memory) == _memproj_fallthrough, "allocation memory projection");
1107           } else {
1108             assert(mem == _memproj_fallthrough, "allocation memory projection");
1109           }
1110 #endif
1111           _igvn.replace_node(mem_proj, mem);
1112         }
1113       } else  {
1114         assert(false, "only Initialize or AddP expected");
1115       }
1116       j -= (oc1 - _resproj->outcnt());
1117     }
1118   }
1119   if (_fallthroughcatchproj != NULL) {
1120     _igvn.replace_node(_fallthroughcatchproj, alloc->in(TypeFunc::Control));
1121   }
1122   if (_memproj_fallthrough != NULL) {
1123     _igvn.replace_node(_memproj_fallthrough, alloc->in(TypeFunc::Memory));
1124   }
1125   if (_memproj_catchall != NULL) {
1126     _igvn.replace_node(_memproj_catchall, C->top());
1127   }
1128   if (_ioproj_fallthrough != NULL) {
1129     _igvn.replace_node(_ioproj_fallthrough, alloc->in(TypeFunc::I_O));
1130   }
1131   if (_ioproj_catchall != NULL) {
1132     _igvn.replace_node(_ioproj_catchall, C->top());
1133   }
1134   if (_catchallcatchproj != NULL) {
1135     _igvn.replace_node(_catchallcatchproj, C->top());
1136   }
1137 }

1139 bool PhaseMacroExpand::eliminate_allocate_node(AllocateNode *alloc) {
1140   // Don't do scalar replacement if the frame can be popped by JVMTI:
1141   // if reallocation fails during deoptimization we'll pop all
1142   // interpreter frames for this compiled frame and that won't play
1143   // nice with JVMTI popframe.
1144   if (!EliminateAllocations || JvmtiExport::can_pop_frame() || !alloc->_is_non_escaping) {
1145     return false;
1146   }
1147   Node* klass = alloc->in(AllocateNode::KlassNode);
1148   const TypeKlassPtr* tklass = _igvn.type(klass)->is_klassptr();
1149   Node* res = alloc->result_cast();
1150   // Eliminate boxing allocations which are not used
1151   // regardless scalar replacable status.
1152   bool boxing_alloc = C->eliminate_boxing() &&
1153                       tklass->klass()->is_instance_klass()  &&
1154                       tklass->klass()->as_instance_klass()->is_box_klass();
1155   if (!alloc->_is_scalar_replaceable && (!boxing_alloc || (res != NULL))) {
1156     return false;
1157   }
1158 
1159   extract_call_projections(alloc);
1160 
1161   GrowableArray <SafePointNode *> safepoints;
1162   if (!can_eliminate_allocation(alloc, safepoints)) {
1163     return false;
1164   }
1165 
1166   if (!alloc->_is_scalar_replaceable) {
1167     assert(res == NULL, "sanity");
1168     // We can only eliminate allocation if all debug info references
1169     // are already replaced with SafePointScalarObject because
1170     // we can't search for a fields value without instance_id.
1171     if (safepoints.length() > 0) {
1172       return false;
1173     }
1174   }
1175 
1176   if (!scalar_replacement(alloc, safepoints)) {
1177     return false;
1178   }
1179 
1180   CompileLog* log = C->log();
1181   if (log != NULL) {
1182     log->head("eliminate_allocation type='%d'",
1183               log->identify(tklass->klass()));
1184     JVMState* p = alloc->jvms();
1185     while (p != NULL) {
1186       log->elem("jvms bci='%d' method='%d'", p->bci(), log->identify(p->method()));
1187       p = p->caller();
1188     }
1189     log->tail("eliminate_allocation");
1190   }
1191 
1192   process_users_of_allocation(alloc);
1193 
1194 #ifndef PRODUCT
1195   if (PrintEliminateAllocations) {
1196     if (alloc->is_AllocateArray())
1197       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1198     else
1199       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1200   }
1201 #endif
1202 
1203   return true;
1204 }

1280 //=============================================================================
1281 //
1282 //                              A L L O C A T I O N
1283 //
1284 // Allocation attempts to be fast in the case of frequent small objects.
1285 // It breaks down like this:
1286 //
1287 // 1) Size in doublewords is computed.  This is a constant for objects and
1288 // variable for most arrays.  Doubleword units are used to avoid size
1289 // overflow of huge doubleword arrays.  We need doublewords in the end for
1290 // rounding.
1291 //
1292 // 2) Size is checked for being 'too large'.  Too-large allocations will go
1293 // the slow path into the VM.  The slow path can throw any required
1294 // exceptions, and does all the special checks for very large arrays.  The
1295 // size test can constant-fold away for objects.  For objects with
1296 // finalizers it constant-folds the otherway: you always go slow with
1297 // finalizers.
1298 //
1299 // 3) If NOT using TLABs, this is the contended loop-back point.
1300 // Load-Locked the heap top.  If using TLABs normal-load the heap top.
1301 //
1302 // 4) Check that heap top + size*8 < max.  If we fail go the slow ` route.
1303 // NOTE: "top+size*8" cannot wrap the 4Gig line!  Here's why: for largish
1304 // "size*8" we always enter the VM, where "largish" is a constant picked small
1305 // enough that there's always space between the eden max and 4Gig (old space is
1306 // there so it's quite large) and large enough that the cost of entering the VM
1307 // is dwarfed by the cost to initialize the space.
1308 //
1309 // 5) If NOT using TLABs, Store-Conditional the adjusted heap top back
1310 // down.  If contended, repeat at step 3.  If using TLABs normal-store
1311 // adjusted heap top back down; there is no contention.
1312 //
1313 // 6) If !ZeroTLAB then Bulk-clear the object/array.  Fill in klass & mark
1314 // fields.
1315 //
1316 // 7) Merge with the slow-path; cast the raw memory pointer to the correct
1317 // oop flavor.
1318 //
1319 //=============================================================================
1320 // FastAllocateSizeLimit value is in DOUBLEWORDS.
1321 // Allocations bigger than this always go the slow route.
1322 // This value must be small enough that allocation attempts that need to
1323 // trigger exceptions go the slow route.  Also, it must be small enough so
1324 // that heap_top + size_in_bytes does not wrap around the 4Gig limit.
1325 //=============================================================================j//
1326 // %%% Here is an old comment from parseHelper.cpp; is it outdated?
1327 // The allocator will coalesce int->oop copies away.  See comment in
1328 // coalesce.cpp about how this works.  It depends critically on the exact
1329 // code shape produced here, so if you are changing this code shape
1330 // make sure the GC info for the heap-top is correct in and around the
1331 // slow-path call.
1332 //
1333 
1334 void PhaseMacroExpand::expand_allocate_common(
1335             AllocateNode* alloc, // allocation node to be expanded
1336             Node* length,  // array length for an array allocation
1337             const TypeFunc* slow_call_type, // Type of slow call
1338             address slow_call_address  // Address of slow call
1339     )
1340 {
1341   Node* ctrl = alloc->in(TypeFunc::Control);
1342   Node* mem  = alloc->in(TypeFunc::Memory);
1343   Node* i_o  = alloc->in(TypeFunc::I_O);
1344   Node* size_in_bytes     = alloc->in(AllocateNode::AllocSize);
1345   Node* klass_node        = alloc->in(AllocateNode::KlassNode);
1346   Node* initial_slow_test = alloc->in(AllocateNode::InitialTest);
1347   assert(ctrl != NULL, "must have control");
1348 
1349   // We need a Region and corresponding Phi's to merge the slow-path and fast-path results.
1350   // they will not be used if "always_slow" is set
1351   enum { slow_result_path = 1, fast_result_path = 2 };
1352   Node *result_region = NULL;
1353   Node *result_phi_rawmem = NULL;
1354   Node *result_phi_rawoop = NULL;
1355   Node *result_phi_i_o = NULL;
1356 
1357   // The initial slow comparison is a size check, the comparison
1358   // we want to do is a BoolTest::gt
1359   bool expand_fast_path = true;
1360   int tv = _igvn.find_int_con(initial_slow_test, -1);
1361   if (tv >= 0) {
1362     // InitialTest has constant result
1363     //   0 - can fit in TLAB
1364     //   1 - always too big or negative
1365     assert(tv <= 1, "0 or 1 if a constant");
1366     expand_fast_path = (tv == 0);
1367     initial_slow_test = NULL;
1368   } else {
1369     initial_slow_test = BoolNode::make_predicate(initial_slow_test, &_igvn);
1370   }
1371 
1372   if (C->env()->dtrace_alloc_probes() ||
1373       (!UseTLAB && !Universe::heap()->supports_inline_contig_alloc())) {
1374     // Force slow-path allocation
1375     expand_fast_path = false;
1376     initial_slow_test = NULL;
1377   }
1378 
1379   bool allocation_has_use = (alloc->result_cast() != NULL);
1380   if (!allocation_has_use) {
1381     InitializeNode* init = alloc->initialization();
1382     if (init != NULL) {
1383       init->remove(&_igvn);
1384     }
1385     if (expand_fast_path && (initial_slow_test == NULL)) {
1386       // Remove allocation node and return.
1387       // Size is a non-negative constant -> no initial check needed -> directly to fast path.
1388       // Also, no usages -> empty fast path -> no fall out to slow path -> nothing left.
1389 #ifndef PRODUCT
1390       if (PrintEliminateAllocations) {
1391         tty->print("NotUsed ");
1392         Node* res = alloc->proj_out_or_null(TypeFunc::Parms);
1393         if (res != NULL) {
1394           res->dump();
1395         } else {
1396           alloc->dump();
1397         }
1398       }
1399 #endif
1400       yank_alloc_node(alloc);
1401       return;
1402     }
1403   }
1404 
1405   enum { too_big_or_final_path = 1, need_gc_path = 2 };
1406   Node *slow_region = NULL;
1407   Node *toobig_false = ctrl;
1408 
1409   // generate the initial test if necessary
1410   if (initial_slow_test != NULL ) {
1411     assert (expand_fast_path, "Only need test if there is a fast path");
1412     slow_region = new RegionNode(3);
1413 
1414     // Now make the initial failure test.  Usually a too-big test but
1415     // might be a TRUE for finalizers or a fancy class check for
1416     // newInstance0.
1417     IfNode* toobig_iff = new IfNode(ctrl, initial_slow_test, PROB_MIN, COUNT_UNKNOWN);
1418     transform_later(toobig_iff);
1419     // Plug the failing-too-big test into the slow-path region
1420     Node* toobig_true = new IfTrueNode(toobig_iff);
1421     transform_later(toobig_true);
1422     slow_region    ->init_req( too_big_or_final_path, toobig_true );
1423     toobig_false = new IfFalseNode(toobig_iff);
1424     transform_later(toobig_false);
1425   } else {
1426     // No initial test, just fall into next case
1427     assert(allocation_has_use || !expand_fast_path, "Should already have been handled");
1428     toobig_false = ctrl;
1429     debug_only(slow_region = NodeSentinel);
1430   }
1431 
1432   // If we are here there are several possibilities
1433   // - expand_fast_path is false - then only a slow path is expanded. That's it.
1434   // no_initial_check means a constant allocation.
1435   // - If check always evaluates to false -> expand_fast_path is false (see above)
1436   // - If check always evaluates to true -> directly into fast path (but may bailout to slowpath)
1437   // if !allocation_has_use the fast path is empty
1438   // if !allocation_has_use && no_initial_check
1439   // - Then there are no fastpath that can fall out to slowpath -> no allocation code at all.
1440   //   removed by yank_alloc_node above.
1441 
1442   Node *slow_mem = mem;  // save the current memory state for slow path
1443   // generate the fast allocation code unless we know that the initial test will always go slow
1444   if (expand_fast_path) {
1445     // Fast path modifies only raw memory.
1446     if (mem->is_MergeMem()) {
1447       mem = mem->as_MergeMem()->memory_at(Compile::AliasIdxRaw);
1448     }
1449 
1450     // allocate the Region and Phi nodes for the result
1451     result_region = new RegionNode(3);
1452     result_phi_rawmem = new PhiNode(result_region, Type::MEMORY, TypeRawPtr::BOTTOM);
1453     result_phi_i_o    = new PhiNode(result_region, Type::ABIO); // I/O is used for Prefetch
1454 
1455     // Grab regular I/O before optional prefetch may change it.
1456     // Slow-path does no I/O so just set it to the original I/O.
1457     result_phi_i_o->init_req(slow_result_path, i_o);
1458 
1459     // Name successful fast-path variables
1460     Node* fast_oop_ctrl;
1461     Node* fast_oop_rawmem;
1462 
1463     if (allocation_has_use) {
1464       Node* needgc_ctrl = NULL;
1465       result_phi_rawoop = new PhiNode(result_region, TypeRawPtr::BOTTOM);
1466 
1467       intx prefetch_lines = length != NULL ? AllocatePrefetchLines : AllocateInstancePrefetchLines;
1468       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1469       Node* fast_oop = bs->obj_allocate(this, ctrl, mem, toobig_false, size_in_bytes, i_o, needgc_ctrl,
1470                                         fast_oop_ctrl, fast_oop_rawmem,
1471                                         prefetch_lines);
1472 
1473       if (initial_slow_test != NULL) {
1474         // This completes all paths into the slow merge point
1475         slow_region->init_req(need_gc_path, needgc_ctrl);
1476         transform_later(slow_region);
1477       } else {
1478         // No initial slow path needed!
1479         // Just fall from the need-GC path straight into the VM call.
1480         slow_region = needgc_ctrl;
1481       }
1482 
1483       InitializeNode* init = alloc->initialization();
1484       fast_oop_rawmem = initialize_object(alloc,
1485                                           fast_oop_ctrl, fast_oop_rawmem, fast_oop,
1486                                           klass_node, length, size_in_bytes);
1487       expand_initialize_membar(alloc, init, fast_oop_ctrl, fast_oop_rawmem);
1488       expand_dtrace_alloc_probe(alloc, fast_oop, fast_oop_ctrl, fast_oop_rawmem);
1489 
1490       result_phi_rawoop->init_req(fast_result_path, fast_oop);
1491     } else {
1492       assert (initial_slow_test != NULL, "sanity");
1493       fast_oop_ctrl   = toobig_false;
1494       fast_oop_rawmem = mem;
1495       transform_later(slow_region);
1496     }
1497 
1498     // Plug in the successful fast-path into the result merge point
1499     result_region    ->init_req(fast_result_path, fast_oop_ctrl);
1500     result_phi_i_o   ->init_req(fast_result_path, i_o);
1501     result_phi_rawmem->init_req(fast_result_path, fast_oop_rawmem);
1502   } else {
1503     slow_region = ctrl;
1504     result_phi_i_o = i_o; // Rename it to use in the following code.
1505   }
1506 
1507   // Generate slow-path call
1508   CallNode *call = new CallStaticJavaNode(slow_call_type, slow_call_address,
1509                                OptoRuntime::stub_name(slow_call_address),
1510                                alloc->jvms()->bci(),
1511                                TypePtr::BOTTOM);
1512   call->init_req(TypeFunc::Control,   slow_region);
1513   call->init_req(TypeFunc::I_O,       top());    // does no i/o
1514   call->init_req(TypeFunc::Memory,    slow_mem); // may gc ptrs
1515   call->init_req(TypeFunc::ReturnAdr, alloc->in(TypeFunc::ReturnAdr));
1516   call->init_req(TypeFunc::FramePtr,  alloc->in(TypeFunc::FramePtr));
1517 
1518   call->init_req(TypeFunc::Parms+0, klass_node);
1519   if (length != NULL) {
1520     call->init_req(TypeFunc::Parms+1, length);
1521   } else {
1522     // Let the runtime know if this is a larval allocation
1523     call->init_req(TypeFunc::Parms+1, _igvn.intcon(alloc->_larval));
1524   }
1525 
1526   // Copy debug information and adjust JVMState information, then replace
1527   // allocate node with the call
1528   call->copy_call_debug_info(&_igvn, alloc);
1529   if (expand_fast_path) {
1530     call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
1531   } else {
1532     // Hook i_o projection to avoid its elimination during allocation
1533     // replacement (when only a slow call is generated).
1534     call->set_req(TypeFunc::I_O, result_phi_i_o);
1535   }
1536   _igvn.replace_node(alloc, call);
1537   transform_later(call);
1538 
1539   // Identify the output projections from the allocate node and
1540   // adjust any references to them.
1541   // The control and io projections look like:
1542   //
1543   //        v---Proj(ctrl) <-----+   v---CatchProj(ctrl)
1544   //  Allocate                   Catch
1545   //        ^---Proj(io) <-------+   ^---CatchProj(io)
1546   //
1547   //  We are interested in the CatchProj nodes.
1548   //
1549   extract_call_projections(call);
1550 
1551   // An allocate node has separate memory projections for the uses on
1552   // the control and i_o paths. Replace the control memory projection with
1553   // result_phi_rawmem (unless we are only generating a slow call when
1554   // both memory projections are combined)
1555   if (expand_fast_path && _memproj_fallthrough != NULL) {
1556     _igvn.replace_in_uses(_memproj_fallthrough, result_phi_rawmem);
1557   }
1558   // Now change uses of _memproj_catchall to use _memproj_fallthrough and delete
1559   // _memproj_catchall so we end up with a call that has only 1 memory projection.
1560   if (_memproj_catchall != NULL) {
1561     if (_memproj_fallthrough == NULL) {
1562       _memproj_fallthrough = new ProjNode(call, TypeFunc::Memory);
1563       transform_later(_memproj_fallthrough);
1564     }
1565     _igvn.replace_in_uses(_memproj_catchall, _memproj_fallthrough);
1566     _igvn.remove_dead_node(_memproj_catchall);
1567   }
1568 
1569   // An allocate node has separate i_o projections for the uses on the control
1570   // and i_o paths. Always replace the control i_o projection with result i_o
1571   // otherwise incoming i_o become dead when only a slow call is generated
1572   // (it is different from memory projections where both projections are
1573   // combined in such case).
1574   if (_ioproj_fallthrough != NULL) {
1575     _igvn.replace_in_uses(_ioproj_fallthrough, result_phi_i_o);
1576   }
1577   // Now change uses of _ioproj_catchall to use _ioproj_fallthrough and delete
1578   // _ioproj_catchall so we end up with a call that has only 1 i_o projection.
1579   if (_ioproj_catchall != NULL) {
1580     if (_ioproj_fallthrough == NULL) {
1581       _ioproj_fallthrough = new ProjNode(call, TypeFunc::I_O);
1582       transform_later(_ioproj_fallthrough);
1583     }
1584     _igvn.replace_in_uses(_ioproj_catchall, _ioproj_fallthrough);
1585     _igvn.remove_dead_node(_ioproj_catchall);
1586   }
1587 
1588   // if we generated only a slow call, we are done
1589   if (!expand_fast_path) {
1590     // Now we can unhook i_o.
1591     if (result_phi_i_o->outcnt() > 1) {
1592       call->set_req(TypeFunc::I_O, top());
1593     } else {
1594       assert(result_phi_i_o->unique_ctrl_out() == call, "sanity");
1595       // Case of new array with negative size known during compilation.
1596       // AllocateArrayNode::Ideal() optimization disconnect unreachable
1597       // following code since call to runtime will throw exception.
1598       // As result there will be no users of i_o after the call.
1599       // Leave i_o attached to this call to avoid problems in preceding graph.
1600     }
1601     return;
1602   }
1603 
1604   if (_fallthroughcatchproj != NULL) {
1605     ctrl = _fallthroughcatchproj->clone();
1606     transform_later(ctrl);
1607     _igvn.replace_node(_fallthroughcatchproj, result_region);
1608   } else {
1609     ctrl = top();
1610   }
1611   Node *slow_result;
1612   if (_resproj == NULL) {
1613     // no uses of the allocation result
1614     slow_result = top();
1615   } else {
1616     slow_result = _resproj->clone();
1617     transform_later(slow_result);
1618     _igvn.replace_node(_resproj, result_phi_rawoop);
1619   }
1620 
1621   // Plug slow-path into result merge point
1622   result_region->init_req( slow_result_path, ctrl);
1623   transform_later(result_region);
1624   if (allocation_has_use) {
1625     result_phi_rawoop->init_req(slow_result_path, slow_result);
1626     transform_later(result_phi_rawoop);
1627   }
1628   result_phi_rawmem->init_req(slow_result_path, _memproj_fallthrough);
1629   transform_later(result_phi_rawmem);
1630   transform_later(result_phi_i_o);
1631   // This completes all paths into the result merge point
1632 }

1634 // Remove alloc node that has no uses.
1635 void PhaseMacroExpand::yank_alloc_node(AllocateNode* alloc) {
1636   Node* ctrl = alloc->in(TypeFunc::Control);
1637   Node* mem  = alloc->in(TypeFunc::Memory);
1638   Node* i_o  = alloc->in(TypeFunc::I_O);
1639 
1640   extract_call_projections(alloc);
1641   if (_resproj != NULL) {
1642     for (DUIterator_Fast imax, i = _resproj->fast_outs(imax); i < imax; i++) {
1643       Node* use = _resproj->fast_out(i);
1644       use->isa_MemBar()->remove(&_igvn);
1645       --imax;
1646       --i; // back up iterator
1647     }
1648     assert(_resproj->outcnt() == 0, "all uses must be deleted");
1649     _igvn.remove_dead_node(_resproj);
1650   }
1651   if (_fallthroughcatchproj != NULL) {
1652     _igvn.replace_in_uses(_fallthroughcatchproj, ctrl);
1653     _igvn.remove_dead_node(_fallthroughcatchproj);
1654   }
1655   if (_catchallcatchproj != NULL) {
1656     _igvn.rehash_node_delayed(_catchallcatchproj);
1657     _catchallcatchproj->set_req(0, top());
1658   }
1659   if (_fallthroughproj != NULL) {
1660     Node* catchnode = _fallthroughproj->unique_ctrl_out();
1661     _igvn.remove_dead_node(catchnode);
1662     _igvn.remove_dead_node(_fallthroughproj);
1663   }
1664   if (_memproj_fallthrough != NULL) {
1665     _igvn.replace_in_uses(_memproj_fallthrough, mem);
1666     _igvn.remove_dead_node(_memproj_fallthrough);
1667   }
1668   if (_ioproj_fallthrough != NULL) {
1669     _igvn.replace_in_uses(_ioproj_fallthrough, i_o);
1670     _igvn.remove_dead_node(_ioproj_fallthrough);
1671   }
1672   if (_memproj_catchall != NULL) {
1673     _igvn.rehash_node_delayed(_memproj_catchall);
1674     _memproj_catchall->set_req(0, top());
1675   }
1676   if (_ioproj_catchall != NULL) {
1677     _igvn.rehash_node_delayed(_ioproj_catchall);
1678     _ioproj_catchall->set_req(0, top());
1679   }
1680 #ifndef PRODUCT
1681   if (PrintEliminateAllocations) {
1682     if (alloc->is_AllocateArray()) {}
1683       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1684     } else {
1685       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1686     }
1687 #endif
1688   _igvn.remove_dead_node(alloc);
1689 }

1691 void PhaseMacroExpand::expand_initialize_membar(AllocateNode* alloc, InitializeNode* init,
1692                                                 Node*& fast_oop_ctrl, Node*& fast_oop_rawmem) {
1693   // If initialization is performed by an array copy, any required
1694   // MemBarStoreStore was already added. If the object does not
1695   // escape no need for a MemBarStoreStore. If the object does not
1696   // escape in its initializer and memory barrier (MemBarStoreStore or
1697   // stronger) is already added at exit of initializer, also no need
1698   // for a MemBarStoreStore. Otherwise we need a MemBarStoreStore
1699   // so that stores that initialize this object can't be reordered
1700   // with a subsequent store that makes this object accessible by
1701   // other threads.
1702   // Other threads include java threads and JVM internal threads
1703   // (for example concurrent GC threads). Current concurrent GC
1704   // implementation: G1 will not scan newly created object,
1705   // so it's safe to skip storestore barrier when allocation does
1706   // not escape.
1707   if (!alloc->does_not_escape_thread() &&
1708     !alloc->is_allocation_MemBar_redundant() &&
1709     (init == NULL || !init->is_complete_with_arraycopy())) {
1710     if (init == NULL || init->req() < InitializeNode::RawStores) {
1711       // No InitializeNode or no stores captured by zeroing
1712       // elimination. Simply add the MemBarStoreStore after object
1713       // initialization.
1714       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1715       transform_later(mb);
1716 
1717       mb->init_req(TypeFunc::Memory, fast_oop_rawmem);
1718       mb->init_req(TypeFunc::Control, fast_oop_ctrl);
1719       fast_oop_ctrl = new ProjNode(mb, TypeFunc::Control);
1720       transform_later(fast_oop_ctrl);
1721       fast_oop_rawmem = new ProjNode(mb, TypeFunc::Memory);
1722       transform_later(fast_oop_rawmem);
1723     } else {
1724       // Add the MemBarStoreStore after the InitializeNode so that
1725       // all stores performing the initialization that were moved
1726       // before the InitializeNode happen before the storestore
1727       // barrier.
1728 
1729       Node* init_ctrl = init->proj_out_or_null(TypeFunc::Control);
1730       Node* init_mem = init->proj_out_or_null(TypeFunc::Memory);
1731 
1732       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1733       transform_later(mb);
1734 
1735       Node* ctrl = new ProjNode(init, TypeFunc::Control);
1736       transform_later(ctrl);
1737       Node* mem = new ProjNode(init, TypeFunc::Memory);
1738       transform_later(mem);
1739 
1740       // The MemBarStoreStore depends on control and memory coming
1741       // from the InitializeNode
1742       mb->init_req(TypeFunc::Memory, mem);
1743       mb->init_req(TypeFunc::Control, ctrl);
1744 
1745       ctrl = new ProjNode(mb, TypeFunc::Control);
1746       transform_later(ctrl);
1747       mem = new ProjNode(mb, TypeFunc::Memory);
1748       transform_later(mem);
1749 
1750       // All nodes that depended on the InitializeNode for control
1751       // and memory must now depend on the MemBarNode that itself
1752       // depends on the InitializeNode
1753       if (init_ctrl != NULL) {
1754         _igvn.replace_node(init_ctrl, ctrl);
1755       }
1756       if (init_mem != NULL) {
1757         _igvn.replace_node(init_mem, mem);
1758       }
1759     }
1760   }
1761 }

1792 // Helper for PhaseMacroExpand::expand_allocate_common.
1793 // Initializes the newly-allocated storage.
1794 Node* PhaseMacroExpand::initialize_object(AllocateNode* alloc,
1795                                           Node* control, Node* rawmem, Node* object,
1796                                           Node* klass_node, Node* length,
1797                                           Node* size_in_bytes) {
1798   InitializeNode* init = alloc->initialization();
1799   // Store the klass & mark bits
1800   Node* mark_node = alloc->make_ideal_mark(&_igvn, control, rawmem);
1801   if (!mark_node->is_Con()) {
1802     transform_later(mark_node);
1803   }
1804   rawmem = make_store(control, rawmem, object, oopDesc::mark_offset_in_bytes(), mark_node, TypeX_X->basic_type());
1805 
1806   BasicType bt = T_METADATA;
1807   Node* metadata = klass_node;
1808   Node* properties = alloc->in(AllocateNode::StorageProperties);
1809   if (properties != NULL) {
1810     // Encode array storage properties into klass pointer
1811     assert(EnableValhalla, "array storage properties not supported");
1812     if (UseCompressedClassPointers) {
1813       // Compress the klass pointer before inserting the storage properties value
1814       metadata = transform_later(new EncodePKlassNode(metadata, metadata->bottom_type()->make_narrowklass()));
1815       metadata = transform_later(new CastN2INode(metadata));
1816       metadata = transform_later(new OrINode(metadata, transform_later(new ConvL2INode(properties))));
1817       bt = T_INT;
1818     } else {
1819       metadata = transform_later(new CastP2XNode(NULL, metadata));
1820       metadata = transform_later(new OrXNode(metadata, properties));
1821       bt = T_LONG;
1822     }
1823   }
1824   rawmem = make_store(control, rawmem, object, oopDesc::klass_offset_in_bytes(), metadata, bt);
1825 
1826   int header_size = alloc->minimum_header_size();  // conservatively small
1827 
1828   // Array length
1829   if (length != NULL) {         // Arrays need length field
1830     rawmem = make_store(control, rawmem, object, arrayOopDesc::length_offset_in_bytes(), length, T_INT);
1831     // conservatively small header size:
1832     header_size = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1833     ciKlass* k = _igvn.type(klass_node)->is_klassptr()->klass();
1834     if (k->is_array_klass())    // we know the exact header size in most cases:
1835       header_size = Klass::layout_helper_header_size(k->layout_helper());
1836   }
1837 
1838   // Clear the object body, if necessary.
1839   if (init == NULL) {
1840     // The init has somehow disappeared; be cautious and clear everything.
1841     //
1842     // This can happen if a node is allocated but an uncommon trap occurs
1843     // immediately.  In this case, the Initialize gets associated with the
1844     // trap, and may be placed in a different (outer) loop, if the Allocate
1845     // is in a loop.  If (this is rare) the inner loop gets unrolled, then
1846     // there can be two Allocates to one Initialize.  The answer in all these
1847     // edge cases is safety first.  It is always safe to clear immediately
1848     // within an Allocate, and then (maybe or maybe not) clear some more later.
1849     if (!(UseTLAB && ZeroTLAB)) {
1850       rawmem = ClearArrayNode::clear_memory(control, rawmem, object,
1851                                             alloc->in(AllocateNode::DefaultValue),
1852                                             alloc->in(AllocateNode::RawDefaultValue),
1853                                             header_size, size_in_bytes,
1854                                             &_igvn);
1855     }
1856   } else {
1857     if (!init->is_complete()) {
1858       // Try to win by zeroing only what the init does not store.
1859       // We can also try to do some peephole optimizations,
1860       // such as combining some adjacent subword stores.
1861       rawmem = init->complete_stores(control, rawmem, object,
1862                                      header_size, size_in_bytes, &_igvn);
1863     }
1864     // We have no more use for this link, since the AllocateNode goes away:
1865     init->set_req(InitializeNode::RawAddress, top());
1866     // (If we keep the link, it just confuses the register allocator,
1867     // who thinks he sees a real use of the address by the membar.)
1868   }
1869 
1870   return rawmem;
1871 }

1873 // Generate prefetch instructions for next allocations.
1874 Node* PhaseMacroExpand::prefetch_allocation(Node* i_o, Node*& needgc_false,
1875                                         Node*& contended_phi_rawmem,
1876                                         Node* old_eden_top, Node* new_eden_top,
1877                                         intx lines) {
1878    enum { fall_in_path = 1, pf_path = 2 };
1879    if( UseTLAB && AllocatePrefetchStyle == 2 ) {
1880       // Generate prefetch allocation with watermark check.
1881       // As an allocation hits the watermark, we will prefetch starting
1882       // at a "distance" away from watermark.
1883 
1884       Node *pf_region = new RegionNode(3);
1885       Node *pf_phi_rawmem = new PhiNode( pf_region, Type::MEMORY,
1886                                                 TypeRawPtr::BOTTOM );
1887       // I/O is used for Prefetch
1888       Node *pf_phi_abio = new PhiNode( pf_region, Type::ABIO );
1889 
1890       Node *thread = new ThreadLocalNode();
1891       transform_later(thread);
1892 
1893       Node *eden_pf_adr = new AddPNode( top()/*not oop*/, thread,
1894                    _igvn.MakeConX(in_bytes(JavaThread::tlab_pf_top_offset())) );
1895       transform_later(eden_pf_adr);
1896 
1897       Node *old_pf_wm = new LoadPNode(needgc_false,
1898                                    contended_phi_rawmem, eden_pf_adr,
1899                                    TypeRawPtr::BOTTOM, TypeRawPtr::BOTTOM,
1900                                    MemNode::unordered);
1901       transform_later(old_pf_wm);
1902 
1903       // check against new_eden_top
1904       Node *need_pf_cmp = new CmpPNode( new_eden_top, old_pf_wm );
1905       transform_later(need_pf_cmp);
1906       Node *need_pf_bol = new BoolNode( need_pf_cmp, BoolTest::ge );
1907       transform_later(need_pf_bol);
1908       IfNode *need_pf_iff = new IfNode( needgc_false, need_pf_bol,
1909                                        PROB_UNLIKELY_MAG(4), COUNT_UNKNOWN );
1910       transform_later(need_pf_iff);
1911 
1912       // true node, add prefetchdistance
1913       Node *need_pf_true = new IfTrueNode( need_pf_iff );
1914       transform_later(need_pf_true);
1915 
1916       Node *need_pf_false = new IfFalseNode( need_pf_iff );
1917       transform_later(need_pf_false);
1918 
1919       Node *new_pf_wmt = new AddPNode( top(), old_pf_wm,
1920                                     _igvn.MakeConX(AllocatePrefetchDistance) );
1921       transform_later(new_pf_wmt );
1922       new_pf_wmt->set_req(0, need_pf_true);
1923 
1924       Node *store_new_wmt = new StorePNode(need_pf_true,
1925                                        contended_phi_rawmem, eden_pf_adr,
1926                                        TypeRawPtr::BOTTOM, new_pf_wmt,
1927                                        MemNode::unordered);
1928       transform_later(store_new_wmt);
1929 
1930       // adding prefetches
1931       pf_phi_abio->init_req( fall_in_path, i_o );
1932 
1933       Node *prefetch_adr;
1934       Node *prefetch;
1935       uint step_size = AllocatePrefetchStepSize;
1936       uint distance = 0;
1937 
1938       for ( intx i = 0; i < lines; i++ ) {
1939         prefetch_adr = new AddPNode( old_pf_wm, new_pf_wmt,
1940                                             _igvn.MakeConX(distance) );
1941         transform_later(prefetch_adr);
1942         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
1943         transform_later(prefetch);
1944         distance += step_size;
1945         i_o = prefetch;
1946       }
1947       pf_phi_abio->set_req( pf_path, i_o );
1948 
1949       pf_region->init_req( fall_in_path, need_pf_false );
1950       pf_region->init_req( pf_path, need_pf_true );
1951 
1952       pf_phi_rawmem->init_req( fall_in_path, contended_phi_rawmem );
1953       pf_phi_rawmem->init_req( pf_path, store_new_wmt );
1954 
1955       transform_later(pf_region);
1956       transform_later(pf_phi_rawmem);
1957       transform_later(pf_phi_abio);
1958 
1959       needgc_false = pf_region;
1960       contended_phi_rawmem = pf_phi_rawmem;
1961       i_o = pf_phi_abio;
1962    } else if( UseTLAB && AllocatePrefetchStyle == 3 ) {
1963       // Insert a prefetch instruction for each allocation.
1964       // This code is used to generate 1 prefetch instruction per cache line.
1965 
1966       // Generate several prefetch instructions.
1967       uint step_size = AllocatePrefetchStepSize;
1968       uint distance = AllocatePrefetchDistance;
1969 
1970       // Next cache address.
1971       Node *cache_adr = new AddPNode(old_eden_top, old_eden_top,
1972                                      _igvn.MakeConX(step_size + distance));
1973       transform_later(cache_adr);
1974       cache_adr = new CastP2XNode(needgc_false, cache_adr);
1975       transform_later(cache_adr);
1976       // Address is aligned to execute prefetch to the beginning of cache line size
1977       // (it is important when BIS instruction is used on SPARC as prefetch).
1978       Node* mask = _igvn.MakeConX(~(intptr_t)(step_size-1));
1979       cache_adr = new AndXNode(cache_adr, mask);
1980       transform_later(cache_adr);
1981       cache_adr = new CastX2PNode(cache_adr);
1982       transform_later(cache_adr);
1983 
1984       // Prefetch
1985       Node *prefetch = new PrefetchAllocationNode( contended_phi_rawmem, cache_adr );
1986       prefetch->set_req(0, needgc_false);
1987       transform_later(prefetch);
1988       contended_phi_rawmem = prefetch;
1989       Node *prefetch_adr;
1990       distance = step_size;
1991       for ( intx i = 1; i < lines; i++ ) {
1992         prefetch_adr = new AddPNode( cache_adr, cache_adr,
1993                                             _igvn.MakeConX(distance) );
1994         transform_later(prefetch_adr);
1995         prefetch = new PrefetchAllocationNode( contended_phi_rawmem, prefetch_adr );
1996         transform_later(prefetch);
1997         distance += step_size;
1998         contended_phi_rawmem = prefetch;
1999       }
2000    } else if( AllocatePrefetchStyle > 0 ) {
2001       // Insert a prefetch for each allocation only on the fast-path
2002       Node *prefetch_adr;
2003       Node *prefetch;
2004       // Generate several prefetch instructions.
2005       uint step_size = AllocatePrefetchStepSize;
2006       uint distance = AllocatePrefetchDistance;
2007       for ( intx i = 0; i < lines; i++ ) {
2008         prefetch_adr = new AddPNode( old_eden_top, new_eden_top,
2009                                             _igvn.MakeConX(distance) );
2010         transform_later(prefetch_adr);
2011         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
2012         // Do not let it float too high, since if eden_top == eden_end,
2013         // both might be null.
2014         if( i == 0 ) { // Set control for first prefetch, next follows it
2015           prefetch->init_req(0, needgc_false);
2016         }
2017         transform_later(prefetch);
2018         distance += step_size;
2019         i_o = prefetch;
2020       }
2021    }
2022    return i_o;
2023 }

2051 //-------------------mark_eliminated_box----------------------------------
2052 //
2053 // During EA obj may point to several objects but after few ideal graph
2054 // transformations (CCP) it may point to only one non escaping object
2055 // (but still using phi), corresponding locks and unlocks will be marked
2056 // for elimination. Later obj could be replaced with a new node (new phi)
2057 // and which does not have escape information. And later after some graph
2058 // reshape other locks and unlocks (which were not marked for elimination
2059 // before) are connected to this new obj (phi) but they still will not be
2060 // marked for elimination since new obj has no escape information.
2061 // Mark all associated (same box and obj) lock and unlock nodes for
2062 // elimination if some of them marked already.
2063 void PhaseMacroExpand::mark_eliminated_box(Node* oldbox, Node* obj) {
2064   if (oldbox->as_BoxLock()->is_eliminated())
2065     return; // This BoxLock node was processed already.
2066 
2067   // New implementation (EliminateNestedLocks) has separate BoxLock
2068   // node for each locked region so mark all associated locks/unlocks as
2069   // eliminated even if different objects are referenced in one locked region
2070   // (for example, OSR compilation of nested loop inside locked scope).
2071   if (EliminateNestedLocks ||
2072       oldbox->as_BoxLock()->is_simple_lock_region(NULL, obj)) {
2073     // Box is used only in one lock region. Mark this box as eliminated.
2074     _igvn.hash_delete(oldbox);
2075     oldbox->as_BoxLock()->set_eliminated(); // This changes box's hash value
2076      _igvn.hash_insert(oldbox);
2077 
2078     for (uint i = 0; i < oldbox->outcnt(); i++) {
2079       Node* u = oldbox->raw_out(i);
2080       if (u->is_AbstractLock() && !u->as_AbstractLock()->is_non_esc_obj()) {

2355      *          goto slow_path; // Failed.

2901 //---------------------------eliminate_macro_nodes----------------------
2902 // Eliminate scalar replaced allocations and associated locks.
2903 void PhaseMacroExpand::eliminate_macro_nodes() {
2904   if (C->macro_count() == 0)
2905     return;
2906 
2907   // First, attempt to eliminate locks
2908   int cnt = C->macro_count();
2909   for (int i=0; i < cnt; i++) {
2910     Node *n = C->macro_node(i);
2911     if (n->is_AbstractLock()) { // Lock and Unlock nodes
2912       // Before elimination mark all associated (same box and obj)
2913       // lock and unlock nodes.
2914       mark_eliminated_locking_nodes(n->as_AbstractLock());
2915     }
2916   }
2917   bool progress = true;
2918   while (progress) {
2919     progress = false;
2920     for (int i = C->macro_count(); i > 0; i--) {
2921       Node * n = C->macro_node(i-1);
2922       bool success = false;
2923       debug_only(int old_macro_count = C->macro_count(););
2924       if (n->is_AbstractLock()) {
2925         success = eliminate_locking_node(n->as_AbstractLock());
2926       }
2927       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2928       progress = progress || success;
2929     }
2930   }
2931   // Next, attempt to eliminate allocations
2932   _has_locks = false;
2933   progress = true;
2934   while (progress) {
2935     progress = false;
2936     for (int i = C->macro_count(); i > 0; i--) {
2937       Node * n = C->macro_node(i-1);
2938       bool success = false;
2939       debug_only(int old_macro_count = C->macro_count(););
2940       switch (n->class_id()) {
2941       case Node::Class_Allocate:
2942       case Node::Class_AllocateArray:
2943         success = eliminate_allocate_node(n->as_Allocate());
2944         break;
2945       case Node::Class_CallStaticJava: {
2946         CallStaticJavaNode* call = n->as_CallStaticJava();
2947         if (!call->method()->is_method_handle_intrinsic()) {
2948           success = eliminate_boxing_node(n->as_CallStaticJava());
2949         }
2950         break;
2951       }
2952       case Node::Class_Lock:
2953       case Node::Class_Unlock:
2954         assert(!n->as_AbstractLock()->is_eliminated(), "sanity");
2955         _has_locks = true;
2956         break;
2957       case Node::Class_ArrayCopy:
2958         break;
2959       case Node::Class_OuterStripMinedLoop:
2960         break;
2961       case Node::Class_SubTypeCheck:
2962         break;
2963       default:
2964         assert(n->Opcode() == Op_LoopLimit ||
2965                n->Opcode() == Op_Opaque1   ||
2966                n->Opcode() == Op_Opaque2   ||
2967                n->Opcode() == Op_Opaque3   ||
2968                BarrierSet::barrier_set()->barrier_set_c2()->is_gc_barrier_node(n),
2969                "unknown node type in macro list");
2970       }
2971       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2972       progress = progress || success;
2973     }
2974   }
2975 }

2977 //------------------------------expand_macro_nodes----------------------
2978 //  Returns true if a failure occurred.
2979 bool PhaseMacroExpand::expand_macro_nodes() {
2980   // Last attempt to eliminate macro nodes.
2981   eliminate_macro_nodes();
2982 
2983   // Make sure expansion will not cause node limit to be exceeded.
2984   // Worst case is a macro node gets expanded into about 200 nodes.
2985   // Allow 50% more for optimization.
2986   if (C->check_node_count(C->macro_count() * 300, "out of nodes before macro expansion" ) )
2987     return true;
2988 
2989   // Eliminate Opaque and LoopLimit nodes. Do it after all loop optimizations.
2990   bool progress = true;
2991   while (progress) {
2992     progress = false;
2993     for (int i = C->macro_count(); i > 0; i--) {
2994       Node* n = C->macro_node(i-1);
2995       bool success = false;
2996       debug_only(int old_macro_count = C->macro_count(););
2997       if (n->Opcode() == Op_LoopLimit) {
2998         // Remove it from macro list and put on IGVN worklist to optimize.
2999         C->remove_macro_node(n);
3000         _igvn._worklist.push(n);
3001         success = true;
3002       } else if (n->Opcode() == Op_CallStaticJava) {
3003         CallStaticJavaNode* call = n->as_CallStaticJava();
3004         if (!call->method()->is_method_handle_intrinsic()) {
3005           // Remove it from macro list and put on IGVN worklist to optimize.
3006           C->remove_macro_node(n);
3007           _igvn._worklist.push(n);
3008           success = true;
3009         }
3010       } else if (n->Opcode() == Op_Opaque1 || n->Opcode() == Op_Opaque2) {
3011         _igvn.replace_node(n, n->in(1));
3012         success = true;
3013 #if INCLUDE_RTM_OPT
3014       } else if ((n->Opcode() == Op_Opaque3) && ((Opaque3Node*)n)->rtm_opt()) {
3015         assert(C->profile_rtm(), "should be used only in rtm deoptimization code");
3016         assert((n->outcnt() == 1) && n->unique_out()->is_Cmp(), "");
3017         Node* cmp = n->unique_out();
3018 #ifdef ASSERT
3019         // Validate graph.
3020         assert((cmp->outcnt() == 1) && cmp->unique_out()->is_Bool(), "");
3021         BoolNode* bol = cmp->unique_out()->as_Bool();
3022         assert((bol->outcnt() == 1) && bol->unique_out()->is_If() &&
3023                (bol->_test._test == BoolTest::ne), "");
3024         IfNode* ifn = bol->unique_out()->as_If();
3025         assert((ifn->outcnt() == 2) &&
3026                ifn->proj_out(1)->is_uncommon_trap_proj(Deoptimization::Reason_rtm_state_change) != NULL, "");
3027 #endif
3028         Node* repl = n->in(1);
3029         if (!_has_locks) {
3030           // Remove RTM state check if there are no locks in the code.
3031           // Replace input to compare the same value.
3032           repl = (cmp->in(1) == n) ? cmp->in(2) : cmp->in(1);
3033         }
3034         _igvn.replace_node(n, repl);
3035         success = true;
3036 #endif
3037       } else if (n->Opcode() == Op_OuterStripMinedLoop) {
3038         n->as_OuterStripMinedLoop()->adjust_strip_mined_loop(&_igvn);
3039         C->remove_macro_node(n);
3040         success = true;
3041       }

3048   // For ReduceBulkZeroing, we must first process all arraycopy nodes

3052     assert(n->is_macro(), "only macro nodes expected here");
3053     if (_igvn.type(n) == Type::TOP || (n->in(0) != NULL && n->in(0)->is_top())) {
3054       // node is unreachable, so don't try to expand it
3055       C->remove_macro_node(n);
3056       continue;

3081   // new optimization opportunities (such as folding newly created
3082   // load from a just allocated object). Run IGVN.
3083   _igvn.set_delay_transform(false);
3084   _igvn.optimize();

3086 
3087   _igvn.set_delay_transform(true);
3088

3090   // nodes are removed from the macro list as they are processed
3091   while (C->macro_count() > 0) {

3096       // node is unreachable, so don't try to expand it
3097       C->remove_macro_node(n);
3098       continue;
3099     }
3100     switch (n->class_id()) {

3114     assert(C->macro_count() < macro_count, "must have deleted a node from macro list");
3115     if (C->failing())  return true;

3119   _igvn.optimize();
