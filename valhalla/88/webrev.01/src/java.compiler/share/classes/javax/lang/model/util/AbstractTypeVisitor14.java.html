28 import javax.annotation.processing.SupportedSourceVersion;

30 import static javax.lang.model.SourceVersion.*;

32 /**
33  * A skeletal visitor of types with default behavior appropriate for the
34  * {@link SourceVersion#RELEASE_14 RELEASE_14} source version.
35  *
36  * @param <R> the return type of this visitor's methods.  Use {@link
37  *            Void} for visitors that do not need to return results.
38  * @param <P> the type of the additional parameter to this visitor's
39  *            methods.  Use {@code Void} for visitors that do not need an
40  *            additional parameter.
41  *
42  * @see <a href="AbstractTypeVisitor6.html#note_for_subclasses">
43  * <strong>Compatibility note for subclasses</strong></a>
44  * @see AbstractTypeVisitor6
45  * @see AbstractTypeVisitor7
46  * @see AbstractTypeVisitor8
47  * @see AbstractTypeVisitor9
48  * @since 14
49  */
50 @SupportedSourceVersion(RELEASE_16)
51 public abstract class AbstractTypeVisitor14<R, P> extends AbstractTypeVisitor9<R, P> {
52     /**
53      * Constructor for concrete subclasses to call.
54      */
55     protected AbstractTypeVisitor14() {
56         super();
57     }
58 }
