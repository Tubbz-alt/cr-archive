26 package javax.net.ssl;
27 
28 import java.security.*;
29 
30 /**
31  * This class defines the <i>Service Provider Interface</i> (<b>SPI</b>)
32  * for the <code>TrustManagerFactory</code> class.
33  *
34  * <p> All the abstract methods in this class must be implemented by each
35  * cryptographic service provider who wishes to supply the implementation
36  * of a particular trust manager factory.
37  *
38  * @since 1.4
39  * @see TrustManagerFactory
40  * @see TrustManager
41  */
42 public abstract class TrustManagerFactorySpi {





43     /**
44      * Initializes this factory with a source of certificate
45      * authorities and related trust material.
46      *
47      * @param ks the key store or null
48      * @throws KeyStoreException if this operation fails
49      * @see TrustManagerFactory#init(KeyStore)
50      */
51     protected abstract void engineInit(KeyStore ks) throws KeyStoreException;
52 
53     /**
54      * Initializes this factory with a source of provider-specific
55      * key material.
56      * <P>
57      * In some cases, initialization parameters other than a keystore
58      * may be needed by a provider.  Users of that
59      * particular provider are expected to pass an implementation of
60      * the appropriate <CODE>ManagerFactoryParameters</CODE> as
61      * defined by the provider.  The provider can then call the
62      * specified methods in the <CODE>ManagerFactoryParameters</CODE>
63      * implementation to obtain the needed information.
64      *
65      * @param spec an implementation of a provider-specific parameter
66      *          specification
67      * @throws InvalidAlgorithmParameterException if there is problem
68      *          with the parameters
69      * @see TrustManagerFactory#init(ManagerFactoryParameters spec)
70      */
71     protected abstract void engineInit(ManagerFactoryParameters spec)
72         throws InvalidAlgorithmParameterException;
73 
74     /**
75      * Returns one trust manager for each type of trust material.
76      *
77      * @throws IllegalStateException if the factory is not initialized.
78      *
79      * @return the trust managers
80      */
81     protected abstract TrustManager[] engineGetTrustManagers();
82 }
