26 package javax.net.ssl;
27 
28 import java.security.*;
29 
30 /**
31  * This class defines the <i>Service Provider Interface</i> (<b>SPI</b>)
32  * for the <code>KeyManagerFactory</code> class.
33  *
34  * <p> All the abstract methods in this class must be implemented by each
35  * cryptographic service provider who wishes to supply the implementation
36  * of a particular key manager factory.
37  *
38  * @since 1.4
39  * @see KeyManagerFactory
40  * @see KeyManager
41  */
42 public abstract class KeyManagerFactorySpi {
43     /**
44      * Constructor for subclasses to call.
45      */
46     public KeyManagerFactorySpi() {}
47 
48     /**
49      * Initializes this factory with a source of key material.
50      *
51      * @param ks the key store or null
52      * @param password the password for recovering keys
53      * @throws KeyStoreException if this operation fails
54      * @throws NoSuchAlgorithmException if the specified algorithm is not
55      *          available from the specified provider.
56      * @throws UnrecoverableKeyException if the key cannot be recovered
57      * @see KeyManagerFactory#init(KeyStore, char[])
58      */
59     protected abstract void engineInit(KeyStore ks, char[] password) throws
60         KeyStoreException, NoSuchAlgorithmException, UnrecoverableKeyException;
61 
62     /**
63      * Initializes this factory with a source of key material.
64      * <P>
65      * In some cases, initialization parameters other than a keystore
66      * and password may be needed by a provider.  Users of that
67      * particular provider are expected to pass an implementation of
68      * the appropriate <CODE>ManagerFactoryParameters</CODE> as
69      * defined by the provider.  The provider can then call the
70      * specified methods in the ManagerFactoryParameters
71      * implementation to obtain the needed information.
72      *
73      * @param spec an implementation of a provider-specific parameter
74      *          specification
75      * @throws InvalidAlgorithmParameterException if there is problem
76      *          with the parameters
77      * @see KeyManagerFactory#init(ManagerFactoryParameters spec)
78      */
79     protected abstract void engineInit(ManagerFactoryParameters spec)
80         throws InvalidAlgorithmParameterException;
81 
82     /**
83      * Returns one key manager for each type of key material.
84      *
85      * @return the key managers
86      * @throws IllegalStateException
87      *         if the KeyManagerFactorySpi is not initialized
88      */
89     protected abstract KeyManager[] engineGetKeyManagers();
90 }
