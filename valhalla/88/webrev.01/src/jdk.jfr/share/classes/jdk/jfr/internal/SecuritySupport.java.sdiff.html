433     static void ensureClassIsInitialized(Class<?> clazz) {
434         try {
435             MethodHandles.Lookup lookup;
436             if (System.getSecurityManager() == null) {
437                 lookup = MethodHandles.privateLookupIn(clazz, LOOKUP);
438             } else {
439                 lookup = AccessController.doPrivileged(new PrivilegedExceptionAction<>() {
440                     @Override
441                     public MethodHandles.Lookup run() throws IllegalAccessException {
442                         return MethodHandles.privateLookupIn(clazz, LOOKUP);
443                     }
444                 }, null, new ReflectPermission("suppressAccessChecks"));
445             }
446             lookup.ensureInitialized(clazz);
447         } catch (IllegalAccessException e) {
448             throw new InternalError(e);
449         } catch (PrivilegedActionException e) {
450             throw new InternalError(e.getCause());
451         }
452     }
