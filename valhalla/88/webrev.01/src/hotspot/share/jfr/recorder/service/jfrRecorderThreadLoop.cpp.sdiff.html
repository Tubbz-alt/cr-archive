 34 //
 35 // Entry point for "JFR Recorder Thread" message loop.
 36 // The recorder thread executes service requests collected from the message system.
 37 //
 38 void recorderthread_entry(JavaThread* thread, Thread* unused) {
 39   assert(thread != NULL, "invariant");
 40   #define START (msgs & (MSGBIT(MSG_START)))
 41   #define SHUTDOWN (msgs & MSGBIT(MSG_SHUTDOWN))
 42   #define ROTATE (msgs & (MSGBIT(MSG_ROTATE)|MSGBIT(MSG_STOP)))
 43   #define FLUSHPOINT (msgs & (MSGBIT(MSG_FLUSHPOINT)))
 44   #define PROCESS_FULL_BUFFERS (msgs & (MSGBIT(MSG_ROTATE)|MSGBIT(MSG_STOP)|MSGBIT(MSG_FULLBUFFER)))
 45 
 46   JfrPostBox& post_box = JfrRecorderThread::post_box();
 47   log_debug(jfr, system)("Recorder thread STARTED");
 48 
 49   {
 50     bool done = false;
 51     int msgs = 0;
 52     JfrRecorderService service;
 53     MutexLocker msg_lock(JfrMsg_lock);
 54 
 55     // JFR MESSAGE LOOP PROCESSING - BEGIN
 56     while (!done) {
 57       if (post_box.is_empty()) {
 58         JfrMsg_lock->wait();
 59       }
 60       msgs = post_box.collect();
 61       JfrMsg_lock->unlock();
 62       if (PROCESS_FULL_BUFFERS) {
 63         service.process_full_buffers();
 64       }
 65       // Check amount of data written to chunk already
 66       // if it warrants asking for a new chunk
 67       service.evaluate_chunk_size_for_rotation();
 68       if (START) {
 69         service.start();
 70       } else if (ROTATE) {
 71         service.rotate(msgs);
 72       } else if (FLUSHPOINT) {
 73         service.flushpoint();






 74       }
 75       JfrMsg_lock->lock();
 76       post_box.notify_waiters();
 77       if (SHUTDOWN) {
 78         log_debug(jfr, system)("Request to STOP recorder");
 79         done = true;
 80       }
 81     } // JFR MESSAGE LOOP PROCESSING - END
 82 
 83   } // JfrMsg_lock scope
 84 
 85   assert(!JfrMsg_lock->owned_by_self(), "invariant");
 86   post_box.notify_collection_stop();
 87   JfrRecorder::on_recorder_thread_exit();
 88 
 89   #undef START
 90   #undef SHUTDOWN
 91   #undef ROTATE
 92   #undef FLUSHPOINT
 93   #undef PROCESS_FULL_BUFFERS

 38 //
 39 // Entry point for "JFR Recorder Thread" message loop.
 40 // The recorder thread executes service requests collected from the message system.
 41 //
 42 void recorderthread_entry(JavaThread* thread, Thread* unused) {
 43   assert(thread != NULL, "invariant");
 44   #define START (msgs & (MSGBIT(MSG_START)))
 45   #define SHUTDOWN (msgs & MSGBIT(MSG_SHUTDOWN))
 46   #define ROTATE (msgs & (MSGBIT(MSG_ROTATE)|MSGBIT(MSG_STOP)))
 47   #define FLUSHPOINT (msgs & (MSGBIT(MSG_FLUSHPOINT)))
 48   #define PROCESS_FULL_BUFFERS (msgs & (MSGBIT(MSG_ROTATE)|MSGBIT(MSG_STOP)|MSGBIT(MSG_FULLBUFFER)))
 49 
 50   JfrPostBox& post_box = JfrRecorderThread::post_box();
 51   log_debug(jfr, system)("Recorder thread STARTED");
 52 
 53   {
 54     bool done = false;
 55     int msgs = 0;
 56     JfrRecorderService service;
 57     MutexLocker msg_lock(JfrMsg_lock);
 58 
 59     // JFR MESSAGE LOOP PROCESSING - BEGIN
 60     while (!done) {
 61       if (post_box.is_empty()) {
 62         JfrMsg_lock->wait();
 63       }
 64       msgs = post_box.collect();
 65       JfrMsg_lock->unlock();
 66       {
 67         // Run as _thread_in_native as much a possible
 68         // to minimize impact on safepoint synchronizations.
 69         NoHandleMark nhm;
 70         ThreadToNativeFromVM transition(thread);
 71         if (PROCESS_FULL_BUFFERS) {
 72           service.process_full_buffers();
 73         }
 74         // Check amount of data written to chunk already
 75         // if it warrants asking for a new chunk.
 76         service.evaluate_chunk_size_for_rotation();
 77         if (START) {
 78           service.start();
 79         } else if (ROTATE) {
 80           service.rotate(msgs);
 81         } else if (FLUSHPOINT) {
 82           service.flushpoint();
 83         }
 84       }
 85       JfrMsg_lock->lock();
 86       post_box.notify_waiters();
 87       if (SHUTDOWN) {
 88         log_debug(jfr, system)("Request to STOP recorder");
 89         done = true;
 90       }
 91     } // JFR MESSAGE LOOP PROCESSING - END
 92 
 93   } // JfrMsg_lock scope
 94 
 95   assert(!JfrMsg_lock->owned_by_self(), "invariant");
 96   post_box.notify_collection_stop();
 97   JfrRecorder::on_recorder_thread_exit();
 98 
 99   #undef START
100   #undef SHUTDOWN
101   #undef ROTATE
102   #undef FLUSHPOINT
103   #undef PROCESS_FULL_BUFFERS
