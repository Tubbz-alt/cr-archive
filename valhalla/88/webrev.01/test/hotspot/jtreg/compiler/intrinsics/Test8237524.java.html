25 /*
26  * @test
27  * @bug 8237524
28  * @summary String.compareTo() may return incorrect result in this case
29  *
30  * @modules java.base/java.lang:open
31  *
32  * @run main/othervm compiler.intrinsics.Test8237524
33  */
34 
35 package compiler.intrinsics;
36 
37 import java.lang.reflect.Constructor;
38 
39 public class Test8237524 {
40 
41   private static int stringCompareTo(String s1, String s2) {
42     return s1.compareTo(s2);
43   }
44 
45   public static void main(String[] args) throws Exception {
46     Constructor<String> c = String.class.getDeclaredConstructor(byte[].class, byte.class);
47     c.setAccessible(true);
48 
49     byte[] bytes = new byte[] {
50       'Y', 'm', '_', 'l', 'V', 'n', 'W', 'S', 'w', 'm', 'W', 'S'
51     };
52 
53     String s1 = c.newInstance(bytes, (byte) 0); // a Latin string
54     String s2 = c.newInstance(bytes, (byte) 1); // a Chinese string
55 
56     for (int i = 0; i < 50000; i++) {
57       if (stringCompareTo(s1, s2) >= 0) {
58         System.out.println("FAIL. s1 should be less than s2 according to Java API Spec");
59         System.exit(1);
60       }
61     }
62   }
63 }
