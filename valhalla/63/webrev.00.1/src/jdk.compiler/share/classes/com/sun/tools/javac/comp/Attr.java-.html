 219     /** Check kind and type of given tree against protokind and prototype.
 220      *  If check succeeds, store type in tree and return it.
 221      *  If check fails, store errType in tree and return it.
 222      *  No checks are performed if the prototype is a method type.
 223      *  It is not necessary in this case since we know that kind and type
 224      *  are correct.
 225      *
 226      *  @param tree     The tree whose kind and type is checked
 227      *  @param found    The computed type of the tree
 228      *  @param ownkind  The computed kind of the tree
 229      *  @param resultInfo  The expected result of the tree
 230      */
 231     Type check(final JCTree tree,
 232                final Type found,
 233                final KindSelector ownkind,
 234                final ResultInfo resultInfo) {
 235         InferenceContext inferenceContext = resultInfo.checkContext.inferenceContext();
 236         Type owntype;
 237         boolean shouldCheck = !found.hasTag(ERROR) &&
 238                 !resultInfo.pt.hasTag(METHOD) &&
 239                 !resultInfo.pt.hasTag(FORALL);
 240         if (shouldCheck && !ownkind.subset(resultInfo.pkind)) {
 241             log.error(tree.pos(),
 242                       Errors.UnexpectedType(resultInfo.pkind.kindNames(),
 243                                             ownkind.kindNames()));
 244             owntype = types.createErrorType(found);
 245         } else if (allowPoly && inferenceContext.free(found)) {
 246             //delay the check if there are inference variables in the found type
 247             //this means we are dealing with a partially inferred poly expression
 248             owntype = shouldCheck ? resultInfo.pt : found;
 249             if (resultInfo.checkMode.installPostInferenceHook()) {
 250                 inferenceContext.addFreeTypeListener(List.of(found),
 251                         instantiatedContext -> {
 252                             ResultInfo pendingResult =
 253                                     resultInfo.dup(inferenceContext.asInstType(resultInfo.pt));
 254                             check(tree, inferenceContext.asInstType(found), ownkind, pendingResult);
 255                         });
 256             }
 257         } else {
 258             owntype = shouldCheck ?
 259             resultInfo.check(tree, found) :
 260             found;
 261         }
 262         if (resultInfo.checkMode.updateTreeType()) {
 263             tree.type = owntype;
 264         }
 265         return owntype;
 266     }

 268     /** Is given blank final variable assignable, i.e. in a scope where it
 269      *  may be assigned to even though it is final?
 270      *  @param v      The blank final variable.
 271      *  @param env    The current environment.
 272      */
 273     boolean isAssignableAsBlankFinal(VarSymbol v, Env<AttrContext> env) {
 274         Symbol owner = env.info.scope.owner;
 275            // owner refers to the innermost variable, method or
 276            // initializer block declaration at this point.
 277         return
 278             v.owner == owner
 279             ||
 280             ((owner.name == names.init ||    // i.e. we are in a constructor
 281               owner.kind == VAR ||           // i.e. we are in a variable initializer
 282               (owner.flags() & BLOCK) != 0)  // i.e. we are in an initializer block
 283              &&
 284              v.owner == owner.owner
 285              &&
 286              ((v.flags() & STATIC) != 0) == Resolve.isStatic(env));
 287     }

 289     /** Check that variable can be assigned to.
 290      *  @param pos    The current source code position.
 291      *  @param v      The assigned variable
 292      *  @param base   If the variable is referred to in a Select, the part
 293      *                to the left of the `.', null otherwise.
 294      *  @param env    The current environment.
 295      */
 296     void checkAssignable(DiagnosticPosition pos, VarSymbol v, JCTree base, Env<AttrContext> env) {
 297         if (v.name == names._this) {
 298             log.error(pos, Errors.CantAssignValToThis);
 299         } else if ((v.flags() & FINAL) != 0 &&
 300             ((v.flags() & HASINIT) != 0
 301              ||
 302              !((base == null ||
 303                TreeInfo.isThisQualifier(base)) &&
 304                isAssignableAsBlankFinal(v, env)))) {
 305             if (v.isResourceVariable()) { //TWR resource
 306                 log.error(pos, Errors.TryResourceMayNotBeAssigned(v));
 307             } else if ((v.flags() & MATCH_BINDING) != 0) {
 308                 log.error(pos, Errors.PatternBindingMayNotBeAssigned(v));
 309             } else {
 310                 log.error(pos, Errors.CantAssignValToFinalVar(v));
 311             }
 312         }
 313     }

 364         private TreeVisitor<Symbol,Env<AttrContext>> identAttributer = new IdentAttributer();
 365         private class IdentAttributer extends SimpleTreeVisitor<Symbol,Env<AttrContext>> {
 366             @Override @DefinedBy(Api.COMPILER_TREE)
 367             public Symbol visitMemberSelect(MemberSelectTree node, Env<AttrContext> env) {
 368                 Symbol site = visit(node.getExpression(), env);
 369                 if (site.kind == ERR || site.kind == ABSENT_TYP || site.kind == HIDDEN)
 370                     return site;
 371                 Name name = (Name)node.getIdentifier();
 372                 if (site.kind == PCK) {
 373                     env.toplevel.packge = (PackageSymbol)site;
 374                     return rs.findIdentInPackage(null, env, (TypeSymbol)site, name,
 375                             KindSelector.TYP_PCK);
 376                 } else {
 377                     env.enclClass.sym = (ClassSymbol)site;
 378                     return rs.findMemberType(env, site.asType(), name, (TypeSymbol)site);
 379                 }
 380             }
 381 
 382             @Override @DefinedBy(Api.COMPILER_TREE)
 383             public Symbol visitIdentifier(IdentifierTree node, Env<AttrContext> env) {
 384                 return rs.findIdent(null, env, (Name)node.getName(), KindSelector.TYP_PCK);

 415     private Env<AttrContext> attribToTree(JCTree root, Env<AttrContext> env, JCTree tree, ResultInfo resultInfo) {
 416         breakTree = tree;
 417         JavaFileObject prev = log.useSource(env.toplevel.sourcefile);
 418         try {
 419             deferredAttr.attribSpeculative(root, env, resultInfo,
 420                     null, DeferredAttr.AttributionMode.ANALYZER,
 421                     argumentAttr.withLocalCacheContext());
 422         } catch (BreakAttr b) {
 423             return b.env;
 424         } catch (AssertionError ae) {
 425             if (ae.getCause() instanceof BreakAttr) {
 426                 return ((BreakAttr)(ae.getCause())).env;
 427             } else {
 428                 throw ae;
 429             }
 430         } finally {
 431             breakTree = null;
 432             log.useSource(prev);
 433         }
 434         return env;
 435     }

 647     /** Visitor method: attribute a tree, catching any completion failure
 648      *  exceptions. Return the tree's type.
 649      *
 650      *  @param tree    The tree to be visited.
 651      *  @param env     The environment visitor argument.
 652      *  @param resultInfo   The result info visitor argument.
 653      */
 654     Type attribTree(JCTree tree, Env<AttrContext> env, ResultInfo resultInfo) {
 655         Env<AttrContext> prevEnv = this.env;
 656         ResultInfo prevResult = this.resultInfo;
 657         try {
 658             this.env = env;
 659             this.resultInfo = resultInfo;
 660             if (resultInfo.needsArgumentAttr(tree)) {
 661                 result = argumentAttr.attribArg(tree, env);
 662             } else {
 663                 tree.accept(this);
 664             }
 665             matchBindings = matchBindingsComputer.finishBindings(tree,
 666                                                                  matchBindings);
 667             if (tree == breakTree &&
 668                     resultInfo.checkContext.deferredAttrContext().mode == AttrMode.CHECK) {
 669                 breakTreeFound(copyEnv(env));
 670             }
 671             return result;
 672         } catch (CompletionFailure ex) {
 673             tree.type = syms.errType;
 674             return chk.completionError(tree.pos(), ex);
 675         } finally {
 676             this.env = prevEnv;
 677             this.resultInfo = prevResult;
 678         }
 679     }

 814             for (JCTypeParameter tvar : typarams) {
 815                 chk.checkNonCyclic(tvar.pos(), (TypeVar)tvar.type);
 816             }
 817         }

 863     /** Attribute type reference in an `extends' or `implements' clause.
 864      *  Supertypes of anonymous inner classes are usually already attributed.
 865      *
 866      *  @param tree              The tree making up the type reference.
 867      *  @param env               The environment current at the reference.
 868      *  @param classExpected     true if only a class is expected here.
 869      *  @param interfaceExpected true if only an interface is expected here.
 870      */
 871     Type attribBase(JCTree tree,
 872                     Env<AttrContext> env,
 873                     boolean classExpected,
 874                     boolean interfaceExpected,
 875                     boolean checkExtensible) {
 876         Type t = tree.type != null ?
 877             tree.type :
 878             attribType(tree, env);
 879         return checkBase(t, tree, env, classExpected, interfaceExpected, checkExtensible);
 880     }
 881     Type checkBase(Type t,
 882                    JCTree tree,
 883                    Env<AttrContext> env,
 884                    boolean classExpected,
 885                    boolean interfaceExpected,
 886                    boolean checkExtensible) {
 887         final DiagnosticPosition pos = tree.hasTag(TYPEAPPLY) ?
 888                 (((JCTypeApply) tree).clazz).pos() : tree.pos();
 889         if (t.tsym.isAnonymous()) {
 890             log.error(pos, Errors.CantInheritFromAnon);
 891             return types.createErrorType(t);
 892         }
 893         if (t.isErroneous())
 894             return t;
 895         if (t.hasTag(TYPEVAR) && !classExpected && !interfaceExpected) {
 896             // check that type variable is already visible
 897             if (t.getUpperBound() == null) {
 898                 log.error(pos, Errors.IllegalForwardRef);
 899                 return types.createErrorType(t);
 900             }
 901         } else {
 902             t = chk.checkClassType(pos, t, checkExtensible);
 903         }
 904         if (interfaceExpected && (t.tsym.flags() & INTERFACE) == 0) {
 905             log.error(pos, Errors.IntfExpectedHere);
 906             // return errType is necessary since otherwise there might
 907             // be undetected cycles which cause attribution to loop
 908             return types.createErrorType(t);
 909         } else if (checkExtensible &&
 910                    classExpected &&
 911                    (t.tsym.flags() & INTERFACE) != 0) {
 912             log.error(pos, Errors.NoIntfExpectedHere);
 913             return types.createErrorType(t);
 914         }
 915         if (checkExtensible &&
 916             ((t.tsym.flags() & FINAL) != 0)) {
 917             log.error(pos,
 918                       Errors.CantInheritFromFinal(t.tsym));
 919         }
 920         chk.checkNonCyclic(pos, t);
 921         return t;
 922     }

 931     public void visitClassDef(JCClassDecl tree) {
 932         Optional<ArgumentAttr.LocalCacheContext> localCacheContext =
 933                 Optional.ofNullable(env.info.attributionMode.isSpeculative ?
 934                         argumentAttr.withLocalCacheContext() : null);
 935         try {
 936             // Local and anonymous classes have not been entered yet, so we need to
 937             // do it now.
 938             if (env.info.scope.owner.kind.matches(KindSelector.VAL_MTH)) {
 939                 enter.classEnter(tree, env);
 940             } else {
 941                 // If this class declaration is part of a class level annotation,
 942                 // as in @MyAnno(new Object() {}) class MyClass {}, enter it in
 943                 // order to simplify later steps and allow for sensible error
 944                 // messages.
 945                 if (env.tree.hasTag(NEWCLASS) && TreeInfo.isInAnnotation(env, tree))
 946                     enter.classEnter(tree, env);
 947             }
 948 
 949             ClassSymbol c = tree.sym;
 950             if (c == null) {
 951                 // exit in case something drastic went wrong during enter.
 952                 result = null;
 953             } else {
 954                 // make sure class has been completed:
 955                 c.complete();
 956 
 957                 // If this class appears as an anonymous class
 958                 // in a superclass constructor call
 959                 // disable implicit outer instance from being passed.
 960                 // (This would be an illegal access to "this before super").
 961                 if (env.info.isSelfCall &&
 962                         env.tree.hasTag(NEWCLASS)) {
 963                     c.flags_field |= NOOUTERTHIS;
 964                 }
 965                 attribClass(tree.pos(), c);
 966                 result = tree.type = c.type;
 967             }
 968         } finally {
 969             localCacheContext.ifPresent(LocalCacheContext::leave);
 970         }
 971     }

 973     public void visitMethodDef(JCMethodDecl tree) {
 974         MethodSymbol m = tree.sym;
 975         boolean isDefaultMethod = (m.flags() & DEFAULT) != 0;
 976 
 977         Lint lint = env.info.lint.augment(m);
 978         Lint prevLint = chk.setLint(lint);
 979         MethodSymbol prevMethod = chk.setMethod(m);
 980         try {
 981             deferredLintHandler.flush(tree.pos());
 982             chk.checkDeprecatedAnnotation(tree.pos(), m);
 983 
 984 
 985             // Create a new environment with local scope
 986             // for attributing the method.
 987             Env<AttrContext> localEnv = memberEnter.methodEnv(tree, env);
 988             localEnv.info.lint = lint;
 989 
 990             attribStats(tree.typarams, localEnv);
 991 
 992             // If we override any other methods, check that we do so properly.
 993             // JLS ???
 994             if (m.isStatic()) {
 995                 chk.checkHideClashes(tree.pos(), env.enclClass.type, m);
 996             } else {
 997                 chk.checkOverrideClashes(tree.pos(), env.enclClass.type, m);
 998             }
 999             chk.checkOverride(env, tree, m);
1000 
1001             if (isDefaultMethod && types.overridesObjectMethod(m.enclClass(), m)) {
1002                 log.error(tree, Errors.DefaultOverridesObjectMember(m.name, Kinds.kindName(m.location()), m.location()));
1003             }
1004 
1005             // Enter all type parameters into the local method scope.
1006             for (List<JCTypeParameter> l = tree.typarams; l.nonEmpty(); l = l.tail)
1007                 localEnv.info.scope.enterIfAbsent(l.head.type.tsym);
1008 
1009             ClassSymbol owner = env.enclClass.sym;
1010             if ((owner.flags() & ANNOTATION) != 0 &&
1011                     (tree.params.nonEmpty() ||
1012                     tree.recvparam != null))
1013                 log.error(tree.params.nonEmpty() ?
1014                         tree.params.head.pos() :
1015                         tree.recvparam.pos(),
1016                         Errors.IntfAnnotationMembersCantHaveParams);
1017 
1018             // Attribute all value parameters.
1019             for (List<JCVariableDecl> l = tree.params; l.nonEmpty(); l = l.tail) {
1020                 attribStat(l.head, localEnv);
1021             }
1022 
1023             chk.checkVarargsMethodDecl(localEnv, tree);
1024 
1025             // Check that type parameters are well-formed.
1026             chk.validate(tree.typarams, localEnv);
1027 
1028             // Check that result type is well-formed.
1029             if (tree.restype != null && !tree.restype.type.hasTag(VOID))
1030                 chk.validate(tree.restype, localEnv);
1031 
1032             // Check that receiver type is well-formed.
1033             if (tree.recvparam != null) {
1034                 // Use a new environment to check the receiver parameter.
1035                 // Otherwise I get "might not have been initialized" errors.
1036                 // Is there a better way?
1037                 Env<AttrContext> newEnv = memberEnter.methodEnv(tree, env);
1038                 attribType(tree.recvparam, newEnv);
1039                 chk.validate(tree.recvparam, newEnv);
1040             }
1041 
1042             if (env.enclClass.sym.isRecord() && tree.sym.owner.kind == TYP) {
1043                 // lets find if this method is an accessor
1044                 Optional<? extends RecordComponent> recordComponent = env.enclClass.sym.getRecordComponents().stream()
1045                         .filter(rc -> rc.accessor == tree.sym && (rc.accessor.flags_field & GENERATED_MEMBER) == 0).findFirst();
1046                 if (recordComponent.isPresent()) {
1047                     // the method is a user defined accessor lets check that everything is fine
1048                     if (!tree.sym.isPublic()) {
1049                         log.error(tree, Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.MethodMustBePublic));
1050                     }
1051                     if (!types.isSameType(tree.sym.type.getReturnType(), recordComponent.get().type)) {
1052                         log.error(tree, Errors.InvalidAccessorMethodInRecord(env.enclClass.sym,
1053                                 Fragments.AccessorReturnTypeDoesntMatch(tree.sym, recordComponent.get())));
1054                     }
1055                     if (tree.sym.type.asMethodType().thrown != null && !tree.sym.type.asMethodType().thrown.isEmpty()) {
1056                         log.error(tree,
1057                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodCantThrowException));
1058                     }
1059                     if (!tree.typarams.isEmpty()) {
1060                         log.error(tree,
1061                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodMustNotBeGeneric));
1062                     }
1063                     if (tree.sym.isStatic()) {
1064                         log.error(tree,
1065                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodMustNotBeStatic));
1066                     }
1067                 }
1068 
1069                 if (tree.name == names.init) {
1070                     // if this a constructor other than the canonical one
1071                     if ((tree.sym.flags_field & RECORD) == 0) {
1072                         JCMethodInvocation app = TreeInfo.firstConstructorCall(tree);
1073                         if (app == null ||
1074                                 TreeInfo.name(app.meth) != names._this ||
1075                                 !checkFirstConstructorStat(app, tree, false)) {
1076                             log.error(tree, Errors.FirstStatementMustBeCallToAnotherConstructor);
1077                         }
1078                     } else {
1079                         // but if it is the canonical:
1080 
1081                         // if user generated, then it shouldn't explicitly invoke any other constructor
1082                         if ((tree.sym.flags_field & GENERATEDCONSTR) == 0) {
1083                             JCMethodInvocation app = TreeInfo.firstConstructorCall(tree);
1084                             if (app != null &&
1085                                     (TreeInfo.name(app.meth) == names._this ||
1086                                             TreeInfo.name(app.meth) == names._super) &&
1087                                     checkFirstConstructorStat(app, tree, false)) {
1088                                 log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1089                                         Fragments.Canonical, tree.sym.name,
1090                                         Fragments.CanonicalMustNotContainExplicitConstructorInvocation));
1091                             }
1092                         }
1093 
1094                         // also we want to check that no type variables have been defined
1095                         if (!tree.typarams.isEmpty()) {
1096                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1097                                     Fragments.Canonical, tree.sym.name, Fragments.CanonicalMustNotDeclareTypeVariables));
1098                         }
1099 
1100                         /* and now we need to check that the constructor's arguments are exactly the same as those of the
1101                          * record components
1102                          */
1103                         List<Type> recordComponentTypes = TreeInfo.recordFields(env.enclClass).map(vd -> vd.sym.type);
1104                         for (JCVariableDecl param: tree.params) {
1105                             if (!types.isSameType(param.type, recordComponentTypes.head)) {
1106                                 log.error(param, Errors.InvalidCanonicalConstructorInRecord(
1107                                         Fragments.Canonical, tree.sym.name, Fragments.TypeMustBeIdenticalToCorrespondingRecordComponentType));
1108                             }
1109                             recordComponentTypes = recordComponentTypes.tail;
1110                         }
1111                     }
1112                 }
1113             }
1114 
1115             // annotation method checks
1116             if ((owner.flags() & ANNOTATION) != 0) {
1117                 // annotation method cannot have throws clause
1118                 if (tree.thrown.nonEmpty()) {
1119                     log.error(tree.thrown.head.pos(),
1120                               Errors.ThrowsNotAllowedInIntfAnnotation);
1121                 }
1122                 // annotation method cannot declare type-parameters
1123                 if (tree.typarams.nonEmpty()) {
1124                     log.error(tree.typarams.head.pos(),
1125                               Errors.IntfAnnotationMembersCantHaveTypeParams);
1126                 }
1127                 // validate annotation method's return type (could be an annotation type)
1128                 chk.validateAnnotationType(tree.restype);
1129                 // ensure that annotation method does not clash with members of Object/Annotation
1130                 chk.validateAnnotationMethod(tree.pos(), m);
1131             }
1132 
1133             for (List<JCExpression> l = tree.thrown; l.nonEmpty(); l = l.tail)
1134                 chk.checkType(l.head.pos(), l.head.type, syms.throwableType);
1135 
1136             if (tree.body == null) {
1137                 // Empty bodies are only allowed for
1138                 // abstract, native, or interface methods, or for methods
1139                 // in a retrofit signature class.
1140                 if (tree.defaultValue != null) {
1141                     if ((owner.flags() & ANNOTATION) == 0)
1142                         log.error(tree.pos(),
1143                                   Errors.DefaultAllowedInIntfAnnotationMember);
1144                 }
1145                 if (isDefaultMethod || (tree.sym.flags() & (ABSTRACT | NATIVE)) == 0)
1146                     log.error(tree.pos(), Errors.MissingMethBodyOrDeclAbstract);
1147             } else if ((tree.sym.flags() & (ABSTRACT|DEFAULT|PRIVATE)) == ABSTRACT) {
1148                 if ((owner.flags() & INTERFACE) != 0) {
1149                     log.error(tree.body.pos(), Errors.IntfMethCantHaveBody);
1150                 } else {
1151                     log.error(tree.pos(), Errors.AbstractMethCantHaveBody);
1152                 }
1153             } else if ((tree.mods.flags & NATIVE) != 0) {
1154                 log.error(tree.pos(), Errors.NativeMethCantHaveBody);
1155             } else {
1156                 // Add an implicit super() call unless an explicit call to
1157                 // super(...) or this(...) is given
1158                 // or we are compiling class java.lang.Object.
1159                 if (tree.name == names.init && owner.type != syms.objectType) {
1160                     JCBlock body = tree.body;
1161                     if (body.stats.isEmpty() ||
1162                             TreeInfo.getConstructorInvocationName(body.stats, names) == names.empty) {
1163                         JCStatement supCall = make.at(body.pos).Exec(make.Apply(List.nil(),
1164                                 make.Ident(names._super), make.Idents(List.nil())));
1165                         body.stats = body.stats.prepend(supCall);
1166                     } else if ((env.enclClass.sym.flags() & ENUM) != 0 &&
1167                             (tree.mods.flags & GENERATEDCONSTR) == 0 &&
1168                             TreeInfo.isSuperCall(body.stats.head)) {
1169                         // enum constructors are not allowed to call super
1170                         // directly, so make sure there aren't any super calls
1171                         // in enum constructors, except in the compiler
1172                         // generated one.
1173                         log.error(tree.body.stats.head.pos(),
1174                                   Errors.CallToSuperNotAllowedInEnumCtor(env.enclClass.sym));
1175                     }
1176                     if (env.enclClass.sym.isRecord() && (tree.sym.flags_field & RECORD) != 0) { // we are seeing the canonical constructor
1177                         List<Name> recordComponentNames = TreeInfo.recordFields(env.enclClass).map(vd -> vd.sym.name);
1178                         List<Name> initParamNames = tree.sym.params.map(p -> p.name);
1179                         if (!initParamNames.equals(recordComponentNames)) {
1180                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1181                                     Fragments.Canonical, env.enclClass.sym.name, Fragments.CanonicalWithNameMismatch));
1182                         }
1183                         if (!tree.sym.isPublic()) {
1184                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1185                                     TreeInfo.isCompactConstructor(tree) ? Fragments.Compact : Fragments.Canonical,
1186                                     env.enclClass.sym.name, Fragments.CanonicalConstructorMustBePublic));
1187                         }
1188                         if (tree.sym.type.asMethodType().thrown != null && !tree.sym.type.asMethodType().thrown.isEmpty()) {
1189                             log.error(tree,
1190                                     Errors.InvalidCanonicalConstructorInRecord(
1191                                             TreeInfo.isCompactConstructor(tree) ? Fragments.Compact : Fragments.Canonical,
1192                                             env.enclClass.sym.name,
1193                                             Fragments.ThrowsClauseNotAllowedForCanonicalConstructor(
1194                                                     TreeInfo.isCompactConstructor(tree) ? Fragments.Compact : Fragments.Canonical)));
1195                         }
1196                     }
1197                 }
1198 
1199                 // Attribute all type annotations in the body
1200                 annotate.queueScanTreeAndTypeAnnotate(tree.body, localEnv, m, null);
1201                 annotate.flush();
1202 
1203                 // Attribute method body.
1204                 attribStat(tree.body, localEnv);
1205             }
1206 
1207             localEnv.info.scope.leave();
1208             result = tree.type = m.type;
1209         } finally {
1210             chk.setLint(prevLint);
1211             chk.setMethod(prevMethod);
1212         }

1215     public void visitVarDef(JCVariableDecl tree) {
1216         // Local variables have not been entered yet, so we need to do it now:
1217         if (env.info.scope.owner.kind == MTH || env.info.scope.owner.kind == VAR) {
1218             if (tree.sym != null) {
1219                 // parameters have already been entered
1220                 env.info.scope.enter(tree.sym);
1221             } else {
1222                 if (tree.isImplicitlyTyped() && (tree.getModifiers().flags & PARAMETER) == 0) {
1223                     if (tree.init == null) {
1224                         //cannot use 'var' without initializer
1225                         log.error(tree, Errors.CantInferLocalVarType(tree.name, Fragments.LocalMissingInit));
1226                         tree.vartype = make.Erroneous();
1227                     } else {
1228                         Fragment msg = canInferLocalVarType(tree);
1229                         if (msg != null) {
1230                             //cannot use 'var' with initializer which require an explicit target
1231                             //(e.g. lambda, method reference, array initializer).
1232                             log.error(tree, Errors.CantInferLocalVarType(tree.name, msg));
1233                             tree.vartype = make.Erroneous();
1234                         }
1235                     }
1236                 }
1237                 try {
1238                     annotate.blockAnnotations();
1239                     memberEnter.memberEnter(tree, env);
1240                 } finally {
1241                     annotate.unblockAnnotations();
1242                 }
1243             }
1244         } else {
1245             if (tree.init != null) {
1246                 // Field initializer expression need to be entered.
1247                 annotate.queueScanTreeAndTypeAnnotate(tree.init, env, tree.sym, tree.pos());
1248                 annotate.flush();
1249             }
1250         }
1251 
1252         VarSymbol v = tree.sym;
1253         Lint lint = env.info.lint.augment(v);
1254         Lint prevLint = chk.setLint(lint);
1255 
1256         // Check that the variable's declared type is well-formed.
1257         boolean isImplicitLambdaParameter = env.tree.hasTag(LAMBDA) &&
1258                 ((JCLambda)env.tree).paramKind == JCLambda.ParameterKind.IMPLICIT &&
1259                 (tree.sym.flags() & PARAMETER) != 0;
1260         chk.validate(tree.vartype, env, !isImplicitLambdaParameter && !tree.isImplicitlyTyped());
1261 
1262         try {
1263             v.getConstValue(); // ensure compile-time constant initializer is evaluated
1264             deferredLintHandler.flush(tree.pos());
1265             chk.checkDeprecatedAnnotation(tree.pos(), v);
1266 
1267             if (tree.init != null) {
1268                 if ((v.flags_field & FINAL) == 0 ||
1269                     !memberEnter.needsLazyConstValue(tree.init)) {
1270                     // Not a compile-time constant
1271                     // Attribute initializer in a new environment
1272                     // with the declared variable as owner.
1273                     // Check that initializer conforms to variable's declared type.
1274                     Env<AttrContext> initEnv = memberEnter.initEnv(tree, env);
1275                     initEnv.info.lint = lint;
1276                     // In order to catch self-references, we set the variable's
1277                     // declaration position to maximal possible value, effectively
1278                     // marking the variable as undefined.
1279                     initEnv.info.enclVar = v;
1280                     attribExpr(tree.init, initEnv, v.type);
1281                     if (tree.isImplicitlyTyped()) {
1282                         //fixup local variable type
1283                         v.type = chk.checkLocalVarType(tree, tree.init.type.baseType(), tree.name);
1284                     }
1285                 }
1286                 if (tree.isImplicitlyTyped()) {
1287                     setSyntheticVariableType(tree, v.type);
1288                 }
1289             }
1290             result = tree.type = v.type;
1291             if (env.enclClass.sym.isRecord() && tree.sym.owner.kind == TYP && !v.isStatic()) {
1292                 if (isNonArgsMethodInObject(v.name)) {
1293                     log.error(tree, Errors.IllegalRecordComponentName(v));
1294                 }
1295             }
1296         }
1297         finally {
1298             chk.setLint(prevLint);
1299         }

1613     private void handleSwitch(JCTree switchTree,
1614                               JCExpression selector,
1615                               List<JCCase> cases,
1616                               BiConsumer<JCCase, Env<AttrContext>> attribCase) {
1617         Type seltype = attribExpr(selector, env);
1618 
1619         Env<AttrContext> switchEnv =
1620             env.dup(switchTree, env.info.dup(env.info.scope.dup()));
1621 
1622         try {
1623             boolean enumSwitch = (seltype.tsym.flags() & Flags.ENUM) != 0;
1624             boolean stringSwitch = types.isSameType(seltype, syms.stringType);
1625             if (!enumSwitch && !stringSwitch)
1626                 seltype = chk.checkType(selector.pos(), seltype, syms.intType);
1627 
1628             // Attribute all cases and
1629             // check that there are no duplicate case labels or default clauses.
1630             Set<Object> labels = new HashSet<>(); // The set of case labels.
1631             boolean hasDefault = false;      // Is there a default label?
1632             CaseTree.CaseKind caseKind = null;
1633             boolean wasError = false;
1634             for (List<JCCase> l = cases; l.nonEmpty(); l = l.tail) {
1635                 JCCase c = l.head;
1636                 if (caseKind == null) {
1637                     caseKind = c.caseKind;
1638                 } else if (caseKind != c.caseKind && !wasError) {
1639                     log.error(c.pos(),
1640                               Errors.SwitchMixingCaseTypes);
1641                     wasError = true;
1642                 }
1643                 if (c.getExpressions().nonEmpty()) {
1644                     for (JCExpression pat : c.getExpressions()) {
1645                         if (TreeInfo.isNull(pat)) {
1646                             log.error(pat.pos(),
1647                                       Errors.SwitchNullNotAllowed);
1648                         } else if (enumSwitch) {
1649                             Symbol sym = enumConstant(pat, seltype);
1650                             if (sym == null) {
1651                                 log.error(pat.pos(), Errors.EnumLabelMustBeUnqualifiedEnum);
1652                             } else if (!labels.add(sym)) {
1653                                 log.error(c.pos(), Errors.DuplicateCaseLabel);
1654                             }
1655                         } else {
1656                             Type pattype = attribExpr(pat, switchEnv, seltype);
1657                             if (!pattype.hasTag(ERROR)) {
1658                                 if (pattype.constValue() == null) {
1659                                     log.error(pat.pos(),
1660                                               (stringSwitch ? Errors.StringConstReq : Errors.ConstExprReq));
1661                                 } else if (!labels.add(pattype.constValue())) {
1662                                     log.error(c.pos(), Errors.DuplicateCaseLabel);
1663                                 }
1664                             }
1665                         }
1666                     }
1667                 } else if (hasDefault) {
1668                     log.error(c.pos(), Errors.DuplicateDefaultLabel);
1669                 } else {
1670                     hasDefault = true;
1671                 }
1672                 Env<AttrContext> caseEnv =
1673                     switchEnv.dup(c, env.info.dup(switchEnv.info.scope.dup()));
1674                 try {
1675                     attribCase.accept(c, caseEnv);
1676                 } finally {
1677                     caseEnv.info.scope.leave();
1678                 }
1679                 addVars(c.stats, switchEnv.info.scope);
1680             }
1681         } finally {
1682             switchEnv.info.scope.leave();
1683         }
1684     }

1787         }
1788         finally {
1789             localEnv.info.scope.leave();

1938         /** Compute the type of a conditional expression, after
1939          *  checking that it exists.  See JLS 15.25. Does not take into
1940          *  account the special case where condition and both arms
1941          *  are constants.
1942          *
1943          *  @param pos      The source position to be used for error
1944          *                  diagnostics.
1945          *  @param thentype The type of the expression's then-part.
1946          *  @param elsetype The type of the expression's else-part.
1947          */
1948         Type condType(List<DiagnosticPosition> positions, List<Type> condTypes) {
1949             if (condTypes.isEmpty()) {
1950                 return syms.objectType; //TODO: how to handle?
1951             }
1952             Type first = condTypes.head;
1953             // If same type, that is the result
1954             if (condTypes.tail.stream().allMatch(t -> types.isSameType(first, t)))
1955                 return first.baseType();
1956 
1957             List<Type> unboxedTypes = condTypes.stream()
1958                                                .map(t -> t.isPrimitive() ? t : types.unboxedType(t))
1959                                                .collect(List.collector());
1960 
1961             // Otherwise, if both arms can be converted to a numeric
1962             // type, return the least numeric type that fits both arms
1963             // (i.e. return larger of the two, or return int if one
1964             // arm is short, the other is char).
1965             if (unboxedTypes.stream().allMatch(t -> t.isPrimitive())) {
1966                 // If one arm has an integer subrange type (i.e., byte,
1967                 // short, or char), and the other is an integer constant
1968                 // that fits into the subrange, return the subrange type.
1969                 for (Type type : unboxedTypes) {
1970                     if (!type.getTag().isStrictSubRangeOf(INT)) {
1971                         continue;
1972                     }
1973                     if (unboxedTypes.stream().filter(t -> t != type).allMatch(t -> t.hasTag(INT) && types.isAssignable(t, type)))
1974                         return type.baseType();
1975                 }
1976 
1977                 for (TypeTag tag : primitiveTags) {
1978                     Type candidate = syms.typeOfTag[tag.ordinal()];
1979                     if (unboxedTypes.stream().allMatch(t -> types.isSubtype(t, candidate))) {
1980                         return candidate;
1981                     }
1982                 }
1983             }
1984 
1985             // Those were all the cases that could result in a primitive
1986             condTypes = condTypes.stream()
1987                                  .map(t -> t.isPrimitive() ? types.boxedClass(t).type : t)
1988                                  .collect(List.collector());
1989 
1990             for (Type type : condTypes) {
1991                 if (condTypes.stream().filter(t -> t != type).allMatch(t -> types.isAssignable(t, type)))
1992                     return type.baseType();
1993             }
1994 
1995             Iterator<DiagnosticPosition> posIt = positions.iterator();
1996 
1997             condTypes = condTypes.stream()
1998                                  .map(t -> chk.checkNonVoid(posIt.next(), t))
1999                                  .collect(List.collector());
2000 
2001             // both are known to be reference types.  The result is
2002             // lub(thentype,elsetype). This cannot fail, as it will
2003             // always be possible to infer "Object" if nothing better.
2004             return types.lub(condTypes.stream().map(t -> t.baseType()).collect(List.collector()));
2005         }

2151          *                 Tree.BREAK or Tree.CONTINUE.
2152          *  @param label   The label of the jump statement, or null if no
2153          *                 label is given.
2154          *  @param env     The environment current at the jump statement.
2155          */
2156         private Pair<JCTree, JCDiagnostic.Error> findJumpTargetNoError(JCTree.Tag tag,
2157                                                                        Name label,
2158                                                                        Env<AttrContext> env) {
2159             // Search environments outwards from the point of jump.
2160             Env<AttrContext> env1 = env;
2161             JCDiagnostic.Error pendingError = null;
2162             LOOP:
2163             while (env1 != null) {
2164                 switch (env1.tree.getTag()) {
2165                     case LABELLED:
2166                         JCLabeledStatement labelled = (JCLabeledStatement)env1.tree;
2167                         if (label == labelled.label) {
2168                             // If jump is a continue, check that target is a loop.
2169                             if (tag == CONTINUE) {
2170                                 if (!labelled.body.hasTag(DOLOOP) &&
2171                                         !labelled.body.hasTag(WHILELOOP) &&
2172                                         !labelled.body.hasTag(FORLOOP) &&
2173                                         !labelled.body.hasTag(FOREACHLOOP)) {
2174                                     pendingError = Errors.NotLoopLabel(label);
2175                                 }
2176                                 // Found labelled statement target, now go inwards
2177                                 // to next non-labelled tree.
2178                                 return Pair.of(TreeInfo.referencedStatement(labelled), pendingError);
2179                             } else {
2180                                 return Pair.of(labelled, pendingError);
2181                             }
2182                         }
2183                         break;
2184                     case DOLOOP:
2185                     case WHILELOOP:
2186                     case FORLOOP:
2187                     case FOREACHLOOP:
2188                         if (label == null) return Pair.of(env1.tree, pendingError);
2189                         break;
2190                     case SWITCH:
2191                         if (label == null && tag == BREAK) return Pair.of(env1.tree, null);
2192                         break;
2193                     case SWITCH_EXPRESSION:
2194                         if (tag == YIELD) {
2195                             return Pair.of(env1.tree, null);
2196                         } else if (tag == BREAK) {
2197                             pendingError = Errors.BreakOutsideSwitchExpression;
2198                         } else {
2199                             pendingError = Errors.ContinueOutsideSwitchExpression;
2200                         }
2201                         break;
2202                     case LAMBDA:
2203                     case METHODDEF:
2204                     case CLASSDEF:
2205                         break LOOP;
2206                     default:
2207                 }
2208                 env1 = env1.next;
2209             }
2210             if (label != null)
2211                 return Pair.of(null, Errors.UndefLabel(label));
2212             else if (pendingError != null)
2213                 return Pair.of(null, pendingError);
2214             else if (tag == CONTINUE)
2215                 return Pair.of(null, Errors.ContOutsideLoop);
2216             else
2217                 return Pair.of(null, Errors.BreakOutsideSwitchLoop);

2689             // In this case, the class instance creation
2690             // expression
2691             //
2692             //    E.new <typeargs1>C<typargs2>(args) { ... }
2693             //
2694             // is represented internally as
2695             //
2696             //    E . new <typeargs1>C<typargs2>(args) ( class <empty-name> { ... } )  .
2697             //
2698             // This expression is then *transformed* as follows:
2699             //
2700             // (1) add an extends or implements clause
2701             // (2) add a constructor.
2702             //
2703             // For instance, if C is a class, and ET is the type of E,
2704             // the expression
2705             //
2706             //    E.new <typeargs1>C<typargs2>(args) { ... }
2707             //
2708             // is translated to (where X is a fresh name and typarams is the
2709             // parameter list of the super constructor):
2710             //
2711             //   new <typeargs1>X(<*nullchk*>E, args) where
2712             //     X extends C<typargs2> {
2713             //       <typarams> X(ET e, args) {
2714             //         e.<typeargs1>super(args)
2715             //       }
2716             //       ...
2717             //     }
2718             InferenceContext inferenceContext = resultInfo.checkContext.inferenceContext();
2719             final boolean isDiamond = TreeInfo.isDiamond(tree);
2720             if (isDiamond
2721                     && ((tree.constructorType != null && inferenceContext.free(tree.constructorType))
2722                     || (tree.clazz.type != null && inferenceContext.free(tree.clazz.type)))) {
2723                 final ResultInfo resultInfoForClassDefinition = this.resultInfo;
2724                 inferenceContext.addFreeTypeListener(List.of(tree.constructorType, tree.clazz.type),
2725                         instantiatedContext -> {
2726                             tree.constructorType = instantiatedContext.asInstType(tree.constructorType);
2727                             tree.clazz.type = clazz.type = instantiatedContext.asInstType(clazz.type);
2728                             ResultInfo prevResult = this.resultInfo;
2729                             try {
2730                                 this.resultInfo = resultInfoForClassDefinition;
2731                                 visitAnonymousClassDefinition(tree, clazz, clazz.type, cdef,

2733                             } finally {
2734                                 this.resultInfo = prevResult;

2739                     List<Type> invalidDiamondArgs = chk.checkDiamondDenotable((ClassType)clazztype);
2740                     if (!clazztype.isErroneous() && invalidDiamondArgs.nonEmpty()) {

2745                                         fragment,
2746                                         invalidDiamondArgs.size() > 1 ?
2747                                                 DiamondInvalidArgs(invalidDiamondArgs, fragment) :
2748                                                 DiamondInvalidArg(invalidDiamondArgs, fragment)));

2751                     for (Type t : clazztype.getTypeArguments()) {

2757                 // false for isInterface call even when the original type is an interface.

2759                         clazztype.isErroneous() && !clazztype.getOriginalType().hasTag(NONE) &&
2760                         clazztype.getOriginalType().tsym.isInterface();

2763                     cdef.implementing = List.of(clazz);
2764                 } else {
2765                     cdef.extending = clazz;
2766                 }
2767

2769                     isSerializable(clazztype)) {
2770                     localEnv.info.isSerializable = true;
2771                 }
2772 
2773                 attribStat(cdef, localEnv);
2774 
2775                 List<Type> finalargtypes;
2776                 // If an outer instance is given,
2777                 // prefix it to the constructor arguments
2778                 // and delete it from the new expression

3611     boolean checkExConstraints(
3612             List<Type> thrownByFuncExpr,
3613             List<Type> thrownAtFuncType,
3614             InferenceContext inferenceContext) {
3615         /** 18.2.5: Otherwise, let E1, ..., En be the types in the function type's throws clause that
3616          *  are not proper types
3617          */
3618         List<Type> nonProperList = thrownAtFuncType.stream()
3619                 .filter(e -> inferenceContext.free(e)).collect(List.collector());
3620         List<Type> properList = thrownAtFuncType.diff(nonProperList);
3621 
3622         /** Let X1,...,Xm be the checked exception types that the lambda body can throw or
3623          *  in the throws clause of the invocation type of the method reference's compile-time
3624          *  declaration
3625          */
3626         List<Type> checkedList = thrownByFuncExpr.stream()
3627                 .filter(e -> chk.isChecked(e)).collect(List.collector());
3628 
3629         /** If n = 0 (the function type's throws clause consists only of proper types), then
3630          *  if there exists some i (1 <= i <= m) such that Xi is not a subtype of any proper type
3631          *  in the throws clause, the constraint reduces to false; otherwise, the constraint
3632          *  reduces to true
3633          */
3634         ListBuffer<Type> uncaughtByProperTypes = new ListBuffer<>();
3635         for (Type checked : checkedList) {
3636             boolean isSubtype = false;
3637             for (Type proper : properList) {
3638                 if (types.isSubtype(checked, proper)) {
3639                     isSubtype = true;
3640                     break;
3641                 }
3642             }
3643             if (!isSubtype) {
3644                 uncaughtByProperTypes.add(checked);
3645             }
3646         }
3647 
3648         if (nonProperList.isEmpty() && !uncaughtByProperTypes.isEmpty()) {
3649             return false;
3650         }
3651 
3652         /** If n > 0, the constraint reduces to a set of subtyping constraints:
3653          *  for all i (1 <= i <= m), if Xi is not a subtype of any proper type in the
3654          *  throws clause, then the constraints include, for all j (1 <= j <= n), <Xi <: Ej>
3655          */
3656         List<Type> nonProperAsUndet = inferenceContext.asUndetVars(nonProperList);
3657         uncaughtByProperTypes.forEach(checkedEx -> {
3658             nonProperAsUndet.forEach(nonProper -> {
3659                 types.isSubtype(checkedEx, nonProper);
3660             });
3661         });
3662 
3663         /** In addition, for all j (1 <= j <= n), the constraint reduces to the bound throws Ej
3664          */
3665         nonProperAsUndet.stream()
3666                 .filter(t -> t.hasTag(UNDETVAR))
3667                 .forEach(t -> ((UndetVar)t).setThrow());
3668         return true;
3669     }

4519                 // the reference is OK.
4520                 if (!Resolve.isInitializer(env))

4796     Type checkIntersection(JCTree tree, List<JCExpression> bounds) {
4797         Set<Type> boundSet = new HashSet<>();
4798         if (bounds.nonEmpty()) {
4799             // accept class or interface or typevar as first bound.
4800             bounds.head.type = checkBase(bounds.head.type, bounds.head, env, false, false, false);
4801             boundSet.add(types.erasure(bounds.head.type));
4802             if (bounds.head.type.isErroneous()) {
4803                 return bounds.head.type;
4804             }
4805             else if (bounds.head.type.hasTag(TYPEVAR)) {
4806                 // if first bound was a typevar, do not accept further bounds.
4807                 if (bounds.tail.nonEmpty()) {
4808                     log.error(bounds.tail.head.pos(),
4809                               Errors.TypeVarMayNotBeFollowedByOtherBounds);
4810                     return bounds.head.type;
4811                 }
4812             } else {
4813                 // if first bound was a class or interface, accept only interfaces
4814                 // as further bounds.
4815                 for (JCExpression bound : bounds.tail) {
4816                     bound.type = checkBase(bound.type, bound, env, false, true, false);
4817                     if (bound.type.isErroneous()) {
4818                         bounds = List.of(bound);
4819                     }
4820                     else if (bound.type.hasTag(CLASS)) {
4821                         chk.checkNotRepeated(bound.pos(), types.erasure(bound.type), boundSet);
4822                     }
4823                 }
4824             }
4825         }
4826 
4827         if (bounds.length() == 0) {
4828             return syms.objectType;
4829         } else if (bounds.length() == 1) {
4830             return bounds.head.type;
4831         } else {
4832             Type owntype = types.makeIntersectionType(TreeInfo.types(bounds));
4833             // ... the variable's bound is a class type flagged COMPOUND
4834             // (see comment for TypeVar.bound).
4835             // In this case, generate a class tree that represents the
4836             // bound class, ...
4837             JCExpression extending;
4838             List<JCExpression> implementing;
4839             if (!bounds.head.type.isInterface()) {
4840                 extending = bounds.head;
4841                 implementing = bounds.tail;
4842             } else {
4843                 extending = null;
4844                 implementing = bounds;
4845             }
4846             JCClassDecl cd = make.at(tree).ClassDef(
4847                 make.Modifiers(PUBLIC | ABSTRACT),
4848                 names.empty, List.nil(),
4849                 extending, implementing, List.nil());
4850 
4851             ClassSymbol c = (ClassSymbol)owntype.tsym;
4852             Assert.check((c.flags() & COMPOUND) != 0);
4853             cd.sym = c;
4854             c.sourcefile = env.toplevel.sourcefile;
4855 
4856             // ... and attribute the bound class
4857             c.flags_field |= UNATTRIBUTED;
4858             Env<AttrContext> cenv = enter.classEnv(cd, env);
4859             typeEnvs.put(c, cenv);
4860             attribClass(c);
4861             return owntype;
4862         }

4979     /** Attribute class definition associated with given class symbol.
4980      *  @param c   The class symbol whose definition will be attributed.
4981      */
4982     void attribClass(ClassSymbol c) throws CompletionFailure {
4983         if (c.type.hasTag(ERROR)) return;
4984 
4985         // Check for cycles in the inheritance graph, which can arise from
4986         // ill-formed class files.
4987         chk.checkNonCyclic(null, c.type);
4988 
4989         Type st = types.supertype(c.type);
4990         if ((c.flags_field & Flags.COMPOUND) == 0) {
4991             // First, attribute superclass.
4992             if (st.hasTag(CLASS))
4993                 attribClass((ClassSymbol)st.tsym);
4994 
4995             // Next attribute owner, if it is a class.
4996             if (c.owner.kind == TYP && c.owner.type.hasTag(CLASS))
4997                 attribClass((ClassSymbol)c.owner);
4998         }
4999 
5000         // The previous operations might have attributed the current class
5001         // if there was a cycle. So we test first whether the class is still
5002         // UNATTRIBUTED.
5003         if ((c.flags_field & UNATTRIBUTED) != 0) {
5004             c.flags_field &= ~UNATTRIBUTED;
5005 
5006             // Get environment current at the point of class definition.
5007             Env<AttrContext> env = typeEnvs.get(c);
5008 
5009             // The info.lint field in the envs stored in typeEnvs is deliberately uninitialized,
5010             // because the annotations were not available at the time the env was created. Therefore,
5011             // we look up the environment chain for the first enclosing environment for which the
5012             // lint value is set. Typically, this is the parent env, but might be further if there
5013             // are any envs created as a result of TypeParameter nodes.
5014             Env<AttrContext> lintEnv = env;
5015             while (lintEnv.info.lint == null)
5016                 lintEnv = lintEnv.next;
5017 
5018             // Having found the enclosing lint value, we can initialize the lint value for this class
5019             env.info.lint = lintEnv.info.lint.augment(c);
5020 
5021             Lint prevLint = chk.setLint(env.info.lint);
5022             JavaFileObject prev = log.useSource(c.sourcefile);
5023             ResultInfo prevReturnRes = env.info.returnResult;
5024 
5025             try {
5026                 deferredLintHandler.flush(env.tree);
5027                 env.info.returnResult = null;
5028                 // java.lang.Enum may not be subclassed by a non-enum
5029                 if (st.tsym == syms.enumSym &&
5030                     ((c.flags_field & (Flags.ENUM|Flags.COMPOUND)) == 0))
5031                     log.error(env.tree.pos(), Errors.EnumNoSubclassing);
5032 
5033                 // Enums may not be extended by source-level classes
5034                 if (st.tsym != null &&
5035                     ((st.tsym.flags_field & Flags.ENUM) != 0) &&
5036                     ((c.flags_field & (Flags.ENUM | Flags.COMPOUND)) == 0)) {
5037                     log.error(env.tree.pos(), Errors.EnumTypesNotExtensible);

5078     /** Finish the attribution of a class. */
5079     private void attribClassBody(Env<AttrContext> env, ClassSymbol c) {
5080         JCClassDecl tree = (JCClassDecl)env.tree;
5081         Assert.check(c == tree.sym);
5082 
5083         // Validate type parameters, supertype and interfaces.
5084         attribStats(tree.typarams, env);
5085         if (!c.isAnonymous()) {
5086             //already checked if anonymous
5087             chk.validate(tree.typarams, env);
5088             chk.validate(tree.extending, env);
5089             chk.validate(tree.implementing, env);
5090         }
5091 
5092         c.markAbstractIfNeeded(types);
5093 
5094         // If this is a non-abstract class, check that it has no abstract
5095         // methods or unimplemented methods of an implemented interface.
5096         if ((c.flags() & (ABSTRACT | INTERFACE)) == 0) {
5097             chk.checkAllDefined(tree.pos(), c);
5098         }
5099 
5100         if ((c.flags() & ANNOTATION) != 0) {
5101             if (tree.implementing.nonEmpty())
5102                 log.error(tree.implementing.head.pos(),
5103                           Errors.CantExtendIntfAnnotation);
5104             if (tree.typarams.nonEmpty()) {
5105                 log.error(tree.typarams.head.pos(),
5106                           Errors.IntfAnnotationCantHaveTypeParams(c));
5107             }
5108 
5109             // If this annotation type has a @Repeatable, validate
5110             Attribute.Compound repeatable = c.getAnnotationTypeMetadata().getRepeatable();
5111             // If this annotation type has a @Repeatable, validate
5112             if (repeatable != null) {
5113                 // get diagnostic position for error reporting
5114                 DiagnosticPosition cbPos = getDiagnosticPosition(tree, repeatable.type);
5115                 Assert.checkNonNull(cbPos);
5116 
5117                 chk.validateRepeatable(c, repeatable, cbPos);
5118             }
5119         } else {
5120             // Check that all extended classes and interfaces
5121             // are compatible (i.e. no two define methods with same arguments
5122             // yet different return types).  (JLS 8.4.6.3)
5123             chk.checkCompatibleSupertypes(tree.pos(), c.type);
5124             if (allowDefaultMethods) {
5125                 chk.checkDefaultMethodClashes(tree.pos(), c.type);
5126             }
5127         }
5128 
5129         // Check that class does not import the same parameterized interface
5130         // with two different argument lists.
5131         chk.checkClassBounds(tree.pos(), c.type);
5132 
5133         tree.type = c.type;
5134 
5135         for (List<JCTypeParameter> l = tree.typarams;
5136              l.nonEmpty(); l = l.tail) {
5137              Assert.checkNonNull(env.info.scope.findFirst(l.head.name));
5138         }
5139 
5140         // Check that a generic class doesn't extend Throwable
5141         if (!c.type.allparams().isEmpty() && types.isSubtype(c.type, syms.throwableType))
5142             log.error(tree.extending.pos(), Errors.GenericThrowable);
5143 
5144         // Check that all methods which implement some
5145         // method conform to the method they implement.
5146         chk.checkImplementations(tree);
5147 
5148         //check that a resource implementing AutoCloseable cannot throw InterruptedException
5149         checkAutoCloseable(tree.pos(), env, c.type);
5150 
5151         for (List<JCTree> l = tree.defs; l.nonEmpty(); l = l.tail) {
5152             // Attribute declaration
5153             attribStat(l.head, env);
5154             // Check that declarations in inner classes are not static (JLS 8.1.2)
5155             // Make an exception for static constants.
5156             if (c.owner.kind != PCK &&
5157                 ((c.flags() & STATIC) == 0 || c.name == names.empty) &&
5158                 (TreeInfo.flags(l.head) & (STATIC | INTERFACE)) != 0) {
5159                 Symbol sym = null;
5160                 if (l.head.hasTag(VARDEF)) sym = ((JCVariableDecl) l.head).sym;
5161                 if (sym == null ||
5162                     sym.kind != VAR ||
5163                     ((VarSymbol) sym).getConstValue() == null)
5164                     log.error(l.head.pos(), Errors.IclsCantHaveStaticDecl(c));
5165             }
5166         }
5167 
5168         // Check for cycles among non-initial constructors.
5169         chk.checkCyclicConstructors(tree);
5170 
5171         // Check for cycles among annotation elements.
5172         chk.checkNonCyclicElements(tree);
5173 
5174         // Check for proper use of serialVersionUID
5175         if (env.info.lint.isEnabled(LintCategory.SERIAL)
5176                 && isSerializable(c.type)
5177                 && (c.flags() & (Flags.ENUM | Flags.INTERFACE)) == 0
5178                 && !c.isAnonymous()) {
5179             checkSerialVersionUID(tree, c);
5180         }
5181         if (allowTypeAnnos) {
5182             // Correctly organize the positions of the type annotations
5183             typeAnnotations.organizeTypeAnnotationsBodies(tree);

5372         /* I would want to model this after
5373          * com.sun.tools.javac.comp.Check.Validator.visitSelectInternal(JCFieldAccess)
5374          * and override visitSelect and visitTypeApply.
5375          * However, we only set the annotated type in the top-level type
5376          * of the symbol.
5377          * Therefore, we need to override each individual location where a type
5378          * can occur.
5379          */
5380         private void validateAnnotatedType(final JCTree errtree, final Type type) {
5381             //System.err.println("Attr.validateAnnotatedType: " + errtree + " type: " + type);
5382 
5383             if (type.isPrimitiveOrVoid()) {
5384                 return;
5385             }
5386 
5387             JCTree enclTr = errtree;
5388             Type enclTy = type;
5389 
5390             boolean repeat = true;
5391             while (repeat) {
5392                 if (enclTr.hasTag(TYPEAPPLY)) {
5393                     List<Type> tyargs = enclTy.getTypeArguments();
5394                     List<JCExpression> trargs = ((JCTypeApply)enclTr).getTypeArguments();
5395                     if (trargs.length() > 0) {
5396                         // Nothing to do for diamonds
5397                         if (tyargs.length() == trargs.length()) {
5398                             for (int i = 0; i < tyargs.length(); ++i) {
5399                                 validateAnnotatedType(trargs.get(i), tyargs.get(i));
5400                             }
5401                         }
5402                         // If the lengths don't match, it's either a diamond
5403                         // or some nested type that redundantly provides
5404                         // type arguments in the tree.
5405                     }
5406 
5407                     // Look at the clazz part of a generic type
5408                     enclTr = ((JCTree.JCTypeApply)enclTr).clazz;
5409                 }
5410 
5411                 if (enclTr.hasTag(SELECT)) {
5412                     enclTr = ((JCTree.JCFieldAccess)enclTr).getExpression();
5413                     if (enclTy != null &&
5414                             !enclTy.hasTag(NONE)) {
5415                         enclTy = enclTy.getEnclosingType();
5416                     }
5417                 } else if (enclTr.hasTag(ANNOTATED_TYPE)) {
5418                     JCAnnotatedType at = (JCTree.JCAnnotatedType) enclTr;
5419                     if (enclTy == null || enclTy.hasTag(NONE)) {
5420                         if (at.getAnnotations().size() == 1) {
5421                             log.error(at.underlyingType.pos(), Errors.CantTypeAnnotateScoping1(at.getAnnotations().head.attribute));
5422                         } else {
5423                             ListBuffer<Attribute.Compound> comps = new ListBuffer<>();
5424                             for (JCAnnotation an : at.getAnnotations()) {
5425                                 comps.add(an.attribute);
5426                             }
5427                             log.error(at.underlyingType.pos(), Errors.CantTypeAnnotateScoping(comps.toList()));
5428                         }
5429                         repeat = false;
5430                     }
5431                     enclTr = at.underlyingType;
5432                     // enclTy doesn't need to be changed
5433                 } else if (enclTr.hasTag(IDENT)) {
5434                     repeat = false;
5435                 } else if (enclTr.hasTag(JCTree.Tag.WILDCARD)) {
5436                     JCWildcard wc = (JCWildcard) enclTr;
5437                     if (wc.getKind() == JCTree.Kind.EXTENDS_WILDCARD ||
5438                             wc.getKind() == JCTree.Kind.SUPER_WILDCARD) {
5439                         validateAnnotatedType(wc.getBound(), wc.getBound().type);
5440                     } else {
5441                         // Nothing to do for UNBOUND
5442                     }
5443                     repeat = false;
5444                 } else if (enclTr.hasTag(TYPEARRAY)) {
5445                     JCArrayTypeTree art = (JCArrayTypeTree) enclTr;
5446                     validateAnnotatedType(art.getType(), art.elemtype.type);
5447                     repeat = false;
5448                 } else if (enclTr.hasTag(TYPEUNION)) {
5449                     JCTypeUnion ut = (JCTypeUnion) enclTr;
5450                     for (JCTree t : ut.getTypeAlternatives()) {
5451                         validateAnnotatedType(t, t.type);
5452                     }
5453                     repeat = false;
5454                 } else if (enclTr.hasTag(TYPEINTERSECTION)) {
5455                     JCTypeIntersection it = (JCTypeIntersection) enclTr;
5456                     for (JCTree t : it.getBounds()) {
5457                         validateAnnotatedType(t, t.type);
5458                     }
5459                     repeat = false;
5460                 } else if (enclTr.getKind() == JCTree.Kind.PRIMITIVE_TYPE ||
5461                            enclTr.getKind() == JCTree.Kind.ERRONEOUS) {
5462                     repeat = false;
5463                 } else {
5464                     Assert.error("Unexpected tree: " + enclTr + " with kind: " + enclTr.getKind() +
5465                             " within: "+ errtree + " with kind: " + errtree.getKind());
5466                 }
5467             }
5468         }
