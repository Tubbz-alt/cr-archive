  87     private static Class<? extends Annotation> findPolymorphicSignatureClass() {
  88         Class<? extends Annotation> signaturePolyAnnotation = null;
  89         try {
  90             for (Class<?> clazz : TestResolvedJavaType.class.getClassLoader().loadClass("java.lang.invoke.MethodHandle").getDeclaredClasses()) {
  91                 if (clazz.getName().endsWith("PolymorphicSignature") && Annotation.class.isAssignableFrom(clazz)) {
  92                     signaturePolyAnnotation = (Class<? extends Annotation>) clazz;
  93                     break;
  94                 }
  95             }
  96         } catch (Throwable e) {
  97             throw new AssertionError("Could not find annotation PolymorphicSignature in java.lang.invoke.MethodHandle", e);
  98         }
  99         assertNotNull(signaturePolyAnnotation);
 100         return signaturePolyAnnotation;
 101     }

 271     public Class<?> getSupertype(Class<?> c) {
 272         assert !c.isPrimitive();
 273         if (c.isArray()) {
 274             Class<?> componentType = c.getComponentType();
 275             if (componentType.isPrimitive() || componentType == Object.class) {
 276                 return Object.class;
 277             }
 278             return getArrayClass(getSupertype(componentType));
 279         }
 280         if (c.isInterface()) {
 281             return Object.class;
 282         }
 283         return c.getSuperclass();
 284     }

 286     public Class<?> findLeastCommonAncestor(Class<?> c1Initial, Class<?> c2Initial) {
 287         if (c1Initial.isPrimitive() || c2Initial.isPrimitive()) {
 288             return null;
 289         } else {
 290             Class<?> c1 = c1Initial;
 291             Class<?> c2 = c2Initial;
 292             while (true) {
 293                 if (c1.isAssignableFrom(c2)) {
 294                     return c1;
 295                 }
 296                 if (c2.isAssignableFrom(c1)) {
 297                     return c2;
 298                 }
 299                 c1 = getSupertype(c1);
 300                 c2 = getSupertype(c2);
 301             }
 302         }
 303     }

 560     /**
 561      * See <a href="http://docs.oracle.com/javase/specs/jvms/se7/html/jvms-5.html#jvms-5.4.5">Method
 562      * overriding</a>.
 563      */
 564     static boolean isOverriderOf(Method impl, Method m) {
 565         if (!isPrivate(m.getModifiers()) && !isFinal(m.getModifiers())) {
 566             if (m.getName().equals(impl.getName())) {
 567                 if (m.getReturnType() == impl.getReturnType()) {
 568                     if (Arrays.equals(m.getParameterTypes(), impl.getParameterTypes())) {
 569                         if (isPublic(m.getModifiers()) || isProtected(m.getModifiers())) {
 570                             // m is public or protected
 571                             return isPublic(impl.getModifiers()) || isProtected(impl.getModifiers());
 572                         } else {
 573                             // m is package-private
 574                             return impl.getDeclaringClass().getPackage() == m.getDeclaringClass().getPackage();
 575                         }
 576                     }
 577                 }
 578             }
 579         }
 580         return false;
 581     }

 583     static final Map<Class<?>, VTable> vtables = new HashMap<>();
 584 
 585     static class VTable {
 586 
 587         final Map<NameAndSignature, Method> methods = new HashMap<>();
 588     }
 589 
 590     static synchronized VTable getVTable(Class<?> c) {
 591         VTable vtable = vtables.get(c);
 592         if (vtable == null) {
 593             vtable = new VTable();
 594             if (c != Object.class) {
 595                 VTable superVtable = getVTable(c.getSuperclass());
 596                 vtable.methods.putAll(superVtable.methods);
 597             }
 598             for (Method m : c.getDeclaredMethods()) {
 599                 if (!isStatic(m.getModifiers()) && !isPrivate(m.getModifiers())) {
 600                     if (isAbstract(m.getModifiers())) {
 601                         // A subclass makes a concrete method in a superclass abstract
 602                         vtable.methods.remove(new NameAndSignature(m));
 603                     } else {
 604                         vtable.methods.put(new NameAndSignature(m), m);
 605                     }
 606                 }
 607             }
 608             vtables.put(c, vtable);
 609         }
 610         return vtable;
 611     }

 613     static Set<Method> findDeclarations(Method impl, Class<?> c) {
 614         Set<Method> declarations = new HashSet<>();
 615         NameAndSignature implSig = new NameAndSignature(impl);
 616         if (c != null) {
 617             for (Method m : c.getDeclaredMethods()) {
 618                 if (new NameAndSignature(m).equals(implSig)) {
 619                     declarations.add(m);
 620                     break;
 621                 }
 622             }
 623             if (!c.isInterface()) {
 624                 declarations.addAll(findDeclarations(impl, c.getSuperclass()));
 625             }
 626             for (Class<?> i : c.getInterfaces()) {
 627                 declarations.addAll(findDeclarations(impl, i));
 628             }
 629         }
 630         return declarations;
 631     }

 633     @Test
 634     public void resolveMethodTest() {
 635         ResolvedJavaType context = metaAccess.lookupJavaType(TestResolvedJavaType.class);
 636         for (Class<?> c : classes) {
 637             ResolvedJavaType type = metaAccess.lookupJavaType(c);
 638             if (c.isInterface()) {
 639                 for (Method m : c.getDeclaredMethods()) {
 640                     ResolvedJavaMethod resolved = metaAccess.lookupJavaMethod(m);
 641                     ResolvedJavaMethod impl = type.resolveMethod(resolved, context);
 642                     assertEquals(m.toString(), null, impl);
 643                 }
 644             } else if (c.isPrimitive()) {
 645                 assertEquals("No methods expected", c.getDeclaredMethods().length, 0);
 646             } else {
 647                 VTable vtable = getVTable(c);
 648                 for (Method impl : vtable.methods.values()) {
 649                     Set<Method> decls = findDeclarations(impl, c);
 650                     for (Method decl : decls) {

 715     public static Set<Field> getInstanceFields(Class<?> c, boolean includeSuperclasses) {
 716         if (c.isArray() || c.isPrimitive() || c.isInterface()) {
 717             return Collections.emptySet();
 718         }
 719         Set<Field> result = new HashSet<>();
 720         for (Field f : c.getDeclaredFields()) {
 721             if (!Modifier.isStatic(f.getModifiers())) {
 722                 result.add(f);
 723             }
 724         }
 725         if (includeSuperclasses && c != Object.class) {
 726             result.addAll(getInstanceFields(c.getSuperclass(), true));
 727         }
 728         return result;
 729     }

 764     private static boolean isHiddenFromReflection(ResolvedJavaField f) {
 765         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Throwable.class)) && f.getName().equals("backtrace")) {
 766             return true;
 767         }
 768         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(ConstantPool.class)) && f.getName().equals("constantPoolOop")) {
 769             return true;
 770         }
 771         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Class.class)) && f.getName().equals("classLoader")) {
 772             return true;
 773         }
 774         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Lookup.class))) {
 775             return f.getName().equals("allowedModes") || f.getName().equals("lookupClass");
 776         }
 777         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(ClassLoader.class)) ||
 778             f.getDeclaringClass().equals(metaAccess.lookupJavaType(AccessibleObject.class)) ||
 779             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Constructor.class)) ||
 780             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Field.class)) ||
 781             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Method.class)) ||
 782             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Module.class))) {
 783             return true;
 784         }
 785         return false;
 786     }
