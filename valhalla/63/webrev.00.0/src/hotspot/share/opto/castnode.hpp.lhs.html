 32 //------------------------------ConstraintCastNode-----------------------------
 33 // cast to a different range
 34 class ConstraintCastNode: public TypeNode {
 35   protected:
 36   // Can this node be removed post CCP or does it carry a required dependency?
 37   const bool _carry_dependency;
 38   virtual bool cmp( const Node &n ) const;
 39   virtual uint size_of() const;
 40 
 41   public:
 42   ConstraintCastNode(Node *n, const Type *t, bool carry_dependency)
 43     : TypeNode(t,2), _carry_dependency(carry_dependency) {
 44     init_class_id(Class_ConstraintCast);
 45     init_req(1, n);
 46   }
 47   virtual Node* Identity(PhaseGVN* phase);
 48   virtual const Type* Value(PhaseGVN* phase) const;
 49   virtual Node *Ideal(PhaseGVN *phase, bool can_reshape);
 50   virtual int Opcode() const;
 51   virtual uint ideal_reg() const = 0;
 52   virtual bool depends_only_on_test() const { return !_carry_dependency; }
 53   bool carry_dependency() const { return _carry_dependency; }
 54   TypeNode* dominating_cast(PhaseGVN* gvn, PhaseTransform* pt) const;
 55   static Node* make_cast(int opcode,  Node* c, Node *n, const Type *t, bool carry_dependency);
 56 
 57 #ifndef PRODUCT
 58   virtual void dump_spec(outputStream *st) const;
 59 #endif
 60 };
 61 
 62 //------------------------------CastIINode-------------------------------------
 63 // cast integer to integer (different range)
 64 class CastIINode: public ConstraintCastNode {
 65   protected:
 66   // Is this node dependent on a range check?
 67   const bool _range_check_dependency;
 68   virtual bool cmp(const Node &n) const;
 69   virtual uint size_of() const;
 70 
 71   public:
 72   CastIINode(Node* n, const Type* t, bool carry_dependency = false, bool range_check_dependency = false)
 73     : ConstraintCastNode(n, t, carry_dependency), _range_check_dependency(range_check_dependency) {
 74     init_class_id(Class_CastII);
 75   }
 76   virtual int Opcode() const;
 77   virtual uint ideal_reg() const { return Op_RegI; }
 78   virtual const Type* Value(PhaseGVN* phase) const;
 79   virtual Node *Ideal(PhaseGVN *phase, bool can_reshape);
 80   const bool has_range_check() {
 81 #ifdef _LP64
 82     return _range_check_dependency;
 83 #else
 84     assert(!_range_check_dependency, "Should not have range check dependency");
 85     return false;
 86 #endif
 87   }
 88 
 89 #ifndef PRODUCT
 90   virtual void dump_spec(outputStream* st) const;
 91 #endif
 92 };

 94 //------------------------------CastLLNode-------------------------------------
 95 // cast long to long (different range)
 96 class CastLLNode: public ConstraintCastNode {
 97   public:
 98   CastLLNode(Node* n, const Type* t, bool carry_dependency = false)
 99     : ConstraintCastNode(n, t, carry_dependency) {
100     init_class_id(Class_CastLL);
101   }
102   virtual int Opcode() const;
103   virtual uint ideal_reg() const { return Op_RegL; }
104   virtual Node* Ideal(PhaseGVN* phase, bool can_reshape);
105 };

107 //------------------------------CastPPNode-------------------------------------
108 // cast pointer to pointer (different type)
109 class CastPPNode: public ConstraintCastNode {
110   public:
111   CastPPNode (Node *n, const Type *t, bool carry_dependency = false)
112     : ConstraintCastNode(n, t, carry_dependency) {
113   }
114   virtual int Opcode() const;
115   virtual uint ideal_reg() const { return Op_RegP; }
116 };
117 
118 //------------------------------CheckCastPPNode--------------------------------
119 // for _checkcast, cast pointer to pointer (different type), without JOIN,
120 class CheckCastPPNode: public ConstraintCastNode {
121   public:
122   CheckCastPPNode(Node *c, Node *n, const Type *t, bool carry_dependency = false)
123     : ConstraintCastNode(n, t, carry_dependency) {
124     init_class_id(Class_CheckCastPP);
125     init_req(0, c);
126   }
127 
128   virtual Node* Identity(PhaseGVN* phase);
129   virtual const Type* Value(PhaseGVN* phase) const;
130   virtual int   Opcode() const;
131   virtual uint  ideal_reg() const { return Op_RegP; }
132   bool depends_only_on_test() const { return !type()->isa_rawptr() && ConstraintCastNode::depends_only_on_test(); }
133  };

136 //------------------------------CastX2PNode-------------------------------------
137 // convert a machine-pointer-sized integer to a raw pointer
138 class CastX2PNode : public Node {
139   public:
140   CastX2PNode( Node *n ) : Node(NULL, n) {}
141   virtual int Opcode() const;
142   virtual const Type* Value(PhaseGVN* phase) const;
143   virtual Node *Ideal(PhaseGVN *phase, bool can_reshape);
144   virtual Node* Identity(PhaseGVN* phase);
145   virtual uint ideal_reg() const { return Op_RegP; }
146   virtual const Type *bottom_type() const { return TypeRawPtr::BOTTOM; }
147 };
148 
149 //------------------------------CastP2XNode-------------------------------------
150 // Used in both 32-bit and 64-bit land.
151 // Used for card-marks and unsafe pointer math.
152 class CastP2XNode : public Node {
153   public:
154   CastP2XNode( Node *ctrl, Node *n ) : Node(ctrl, n) {}
155   virtual int Opcode() const;
156   virtual const Type* Value(PhaseGVN* phase) const;
157   virtual Node *Ideal(PhaseGVN *phase, bool can_reshape);
158   virtual Node* Identity(PhaseGVN* phase);
159   virtual uint ideal_reg() const { return Op_RegX; }
160   virtual const Type *bottom_type() const { return TypeX_X; }
161   // Return false to keep node from moving away from an associated card mark.
162   virtual bool depends_only_on_test() const { return false; }
163 };
164 
165 // convert between a 32 bit integer and a compressed pointer
166 class CastI2NNode : public TypeNode {
167   public:
168   CastI2NNode(Node *n, const Type *t) : TypeNode(t, 2) {
169     init_req(1, n);
170   }
171   virtual int Opcode() const;
172   virtual uint ideal_reg() const { return Op_RegN; }
173 };

175 class CastN2INode : public Node {
