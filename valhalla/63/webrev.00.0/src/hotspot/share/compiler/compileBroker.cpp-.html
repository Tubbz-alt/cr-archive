 246 CompileTaskWrapper::CompileTaskWrapper(CompileTask* task) {
 247   CompilerThread* thread = CompilerThread::current();
 248   thread->set_task(task);
 249 #if INCLUDE_JVMCI
 250   if (task->is_blocking() && CompileBroker::compiler(task->comp_level())->is_jvmci()) {
 251     task->set_jvmci_compiler_thread(thread);
 252   }
 253 #endif
 254   CompileLog*     log  = thread->log();
 255   if (log != NULL && !task->is_unloaded())  task->log_task_start(log);
 256 }
 257 
 258 CompileTaskWrapper::~CompileTaskWrapper() {
 259   CompilerThread* thread = CompilerThread::current();
 260   CompileTask* task = thread->task();
 261   CompileLog*  log  = thread->log();
 262   if (log != NULL && !task->is_unloaded())  task->log_task_done(log);
 263   thread->set_task(NULL);
 264   task->set_code_handle(NULL);
 265   thread->set_env(NULL);
 266   if (task->is_blocking()) {
 267     bool free_task = false;
 268     {
 269       MutexLocker notifier(thread, task->lock());
 270       task->mark_complete();
 271 #if INCLUDE_JVMCI
 272       if (CompileBroker::compiler(task->comp_level())->is_jvmci()) {
 273         if (!task->has_waiter()) {
 274           // The waiting thread timed out and thus did not free the task.
 275           free_task = true;
 276         }
 277         task->set_jvmci_compiler_thread(NULL);
 278       }
 279 #endif
 280       if (!free_task) {
 281         // Notify the waiting thread that the compilation has completed
 282         // so that it can free the task.
 283         task->lock()->notify_all();
 284       }
 285     }
 286     if (free_task) {
 287       // The task can only be freed once the task lock is released.
 288       CompileTask::free(task);
 289     }
 290   } else {
 291     task->mark_complete();
 292 
 293     // By convention, the compiling thread is responsible for
 294     // recycling a non-blocking CompileTask.
 295     CompileTask::free(task);
 296   }
 297 }

 299 /**
 300  * Check if a CompilerThread can be removed and update count if requested.
 301  */
 302 bool CompileBroker::can_remove(CompilerThread *ct, bool do_it) {
 303   assert(UseDynamicNumberOfCompilerThreads, "or shouldn't be here");
 304   if (!ReduceNumberOfCompilerThreads) return false;
 305 
 306   AbstractCompiler *compiler = ct->compiler();
 307   int compiler_count = compiler->num_compiler_threads();
 308   bool c1 = compiler->is_c1();
 309 
 310   // Keep at least 1 compiler thread of each type.
 311   if (compiler_count < 2) return false;
 312 
 313   // Keep thread alive for at least some time.
 314   if (ct->idle_time_millis() < (c1 ? 500 : 100)) return false;
 315 
 316 #if INCLUDE_JVMCI
 317   if (compiler->is_jvmci()) {
 318     // Handles for JVMCI thread objects may get released concurrently.
 319     if (do_it) {
 320       assert(CompileThread_lock->owner() == ct, "must be holding lock");
 321     } else {
 322       // Skip check if it's the last thread and let caller check again.
 323       return true;
 324     }
 325   }
 326 #endif
 327 
 328   // We only allow the last compiler thread of each type to get removed.
 329   jobject last_compiler = c1 ? compiler1_object(compiler_count - 1)
 330                              : compiler2_object(compiler_count - 1);
 331   if (ct->threadObj() == JNIHandles::resolve_non_null(last_compiler)) {
 332     if (do_it) {
 333       assert_locked_or_safepoint(CompileThread_lock); // Update must be consistent.
 334       compiler->set_num_compiler_threads(compiler_count - 1);
 335 #if INCLUDE_JVMCI
 336       if (compiler->is_jvmci()) {
 337         // Old j.l.Thread object can die when no longer referenced elsewhere.
 338         JNIHandles::destroy_global(compiler2_object(compiler_count - 1));
 339         _compiler2_objects[compiler_count - 1] = NULL;
 340       }
 341 #endif
 342     }
 343     return true;
 344   }
 345   return false;
 346 }

 348 /**
 349  * Add a CompileTask to a CompileQueue.
 350  */
 351 void CompileQueue::add(CompileTask* task) {
 352   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 353 
 354   task->set_next(NULL);
 355   task->set_prev(NULL);
 356 
 357   if (_last == NULL) {
 358     // The compile queue is empty.
 359     assert(_first == NULL, "queue is empty");
 360     _first = task;
 361     _last = task;
 362   } else {
 363     // Append the task to the queue.
 364     assert(_last->next() == NULL, "not last");
 365     _last->set_next(task);
 366     task->set_prev(_last);
 367     _last = task;
 368   }
 369   ++_size;
 370 
 371   // Mark the method as being in the compile queue.
 372   task->method()->set_queued_for_compilation();
 373 
 374   if (CIPrintCompileQueue) {
 375     print_tty();
 376   }
 377 
 378   if (LogCompilation && xtty != NULL) {
 379     task->log_task_queued();
 380   }
 381 
 382   // Notify CompilerThreads that a task is available.
 383   MethodCompileQueue_lock->notify_all();
 384 }

 386 /**
 387  * Empties compilation queue by putting all compilation tasks onto
 388  * a freelist. Furthermore, the method wakes up all threads that are
 389  * waiting on a compilation task to finish. This can happen if background
 390  * compilation is disabled.
 391  */
 392 void CompileQueue::free_all() {
 393   MutexLocker mu(MethodCompileQueue_lock);
 394   CompileTask* next = _first;
 395 
 396   // Iterate over all tasks in the compile queue
 397   while (next != NULL) {
 398     CompileTask* current = next;
 399     next = current->next();
 400     {
 401       // Wake up thread that blocks on the compile task.
 402       MutexLocker ct_lock(current->lock());
 403       current->lock()->notify();
 404     }
 405     // Put the task back on the freelist.
 406     CompileTask::free(current);
 407   }
 408   _first = NULL;
 409 
 410   // Wake up all threads that block on the queue.
 411   MethodCompileQueue_lock->notify_all();
 412 }

 414 /**
 415  * Get the next CompileTask from a CompileQueue
 416  */
 417 CompileTask* CompileQueue::get() {
 418   // save methods from RedefineClasses across safepoint
 419   // across MethodCompileQueue_lock below.
 420   methodHandle save_method;
 421   methodHandle save_hot_method;
 422 
 423   MonitorLocker locker(MethodCompileQueue_lock);
 424   // If _first is NULL we have no more compile jobs. There are two reasons for
 425   // having no compile jobs: First, we compiled everything we wanted. Second,
 426   // we ran out of code cache so compilation has been disabled. In the latter
 427   // case we perform code cache sweeps to free memory such that we can re-enable
 428   // compilation.
 429   while (_first == NULL) {
 430     // Exit loop if compilation is disabled forever
 431     if (CompileBroker::is_compilation_disabled_forever()) {
 432       return NULL;
 433     }
 434 
 435     // If there are no compilation tasks and we can compile new jobs
 436     // (i.e., there is enough free space in the code cache) there is
 437     // no need to invoke the sweeper. As a result, the hotness of methods
 438     // remains unchanged. This behavior is desired, since we want to keep
 439     // the stable state, i.e., we do not want to evict methods from the
 440     // code cache if it is unnecessary.
 441     // We need a timed wait here, since compiler threads can exit if compilation
 442     // is disabled forever. We use 5 seconds wait time; the exiting of compiler threads
 443     // is not critical and we do not want idle compiler threads to wake up too often.
 444     locker.wait(5*1000);
 445 
 446     if (UseDynamicNumberOfCompilerThreads && _first == NULL) {
 447       // Still nothing to compile. Give caller a chance to stop this thread.
 448       if (CompileBroker::can_remove(CompilerThread::current(), false)) return NULL;
 449     }
 450   }
 451 
 452   if (CompileBroker::is_compilation_disabled_forever()) {
 453     return NULL;
 454   }
 455 
 456   CompileTask* task;
 457   {
 458     NoSafepointVerifier nsv;
 459     task = CompilationPolicy::policy()->select_task(this);
 460     if (task != NULL) {
 461       task = task->select_for_compilation();
 462     }
 463   }
 464 
 465   if (task != NULL) {
 466     // Save method pointers across unlock safepoint.  The task is removed from
 467     // the compilation queue, which is walked during RedefineClasses.
 468     Thread* thread = Thread::current();
 469     save_method = methodHandle(thread, task->method());
 470     save_hot_method = methodHandle(thread, task->hot_method());
 471 
 472     remove(task);
 473   }
 474   purge_stale_tasks(); // may temporarily release MCQ lock
 475   return task;
 476 }

 478 // Clean & deallocate stale compile tasks.
 479 // Temporarily releases MethodCompileQueue lock.
 480 void CompileQueue::purge_stale_tasks() {
 481   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 482   if (_first_stale != NULL) {
 483     // Stale tasks are purged when MCQ lock is released,
 484     // but _first_stale updates are protected by MCQ lock.
 485     // Once task processing starts and MCQ lock is released,
 486     // other compiler threads can reuse _first_stale.
 487     CompileTask* head = _first_stale;
 488     _first_stale = NULL;
 489     {
 490       MutexUnlocker ul(MethodCompileQueue_lock);
 491       for (CompileTask* task = head; task != NULL; ) {
 492         CompileTask* next_task = task->next();
 493         CompileTaskWrapper ctw(task); // Frees the task
 494         task->set_failure_reason("stale task");
 495         task = next_task;
 496       }
 497     }
 498   }
 499 }

 501 void CompileQueue::remove(CompileTask* task) {
 502   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 503   if (task->prev() != NULL) {
 504     task->prev()->set_next(task->next());
 505   } else {
 506     // max is the first element
 507     assert(task == _first, "Sanity");
 508     _first = task->next();
 509   }
 510 
 511   if (task->next() != NULL) {
 512     task->next()->set_prev(task->prev());
 513   } else {
 514     // max is the last element
 515     assert(task == _last, "Sanity");
 516     _last = task->prev();
 517   }
 518   --_size;
 519 }

 521 void CompileQueue::remove_and_mark_stale(CompileTask* task) {
 522   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 523   remove(task);
 524 
 525   // Enqueue the task for reclamation (should be done outside MCQ lock)
 526   task->set_next(_first_stale);
 527   task->set_prev(NULL);
 528   _first_stale = task;
 529 }

 567   CompileTask* task = _first;

 804 JavaThread* CompileBroker::make_thread(jobject thread_handle, CompileQueue* queue, AbstractCompiler* comp, Thread* THREAD) {
 805   JavaThread* new_thread = NULL;
 806   {
 807     MutexLocker mu(THREAD, Threads_lock);
 808     if (comp != NULL) {
 809       if (!InjectCompilerCreationFailure || comp->num_compiler_threads() == 0) {
 810         CompilerCounters* counters = new CompilerCounters();
 811         new_thread = new CompilerThread(queue, counters);
 812       }
 813     } else {
 814       new_thread = new CodeCacheSweeperThread();
 815     }
 816     // At this point the new CompilerThread data-races with this startup
 817     // thread (which I believe is the primoridal thread and NOT the VM
 818     // thread).  This means Java bytecodes being executed at startup can
 819     // queue compile jobs which will run at whatever default priority the
 820     // newly created CompilerThread runs at.
 821 
 822 
 823     // At this point it may be possible that no osthread was created for the
 824     // JavaThread due to lack of memory. We would have to throw an exception
 825     // in that case. However, since this must work and we do not allow
 826     // exceptions anyway, check and abort if this fails. But first release the
 827     // lock.
 828 
 829     if (new_thread != NULL && new_thread->osthread() != NULL) {
 830 
 831       java_lang_Thread::set_thread(JNIHandles::resolve_non_null(thread_handle), new_thread);
 832 
 833       // Note that this only sets the JavaThread _priority field, which by
 834       // definition is limited to Java priorities and not OS priorities.
 835       // The os-priority is set in the CompilerThread startup code itself
 836 
 837       java_lang_Thread::set_priority(JNIHandles::resolve_non_null(thread_handle), NearMaxPriority);
 838 
 839       // Note that we cannot call os::set_priority because it expects Java
 840       // priorities and we are *explicitly* using OS priorities so that it's
 841       // possible to set the compiler thread priority higher than any Java
 842       // thread.
 843 
 844       int native_prio = CompilerThreadPriority;
 845       if (native_prio == -1) {
 846         if (UseCriticalCompilerThreadPriority) {
 847           native_prio = os::java_to_os_priority[CriticalPriority];
 848         } else {
 849           native_prio = os::java_to_os_priority[NearMaxPriority];
 850         }
 851       }
 852       os::set_native_priority(new_thread, native_prio);
 853 
 854       java_lang_Thread::set_daemon(JNIHandles::resolve_non_null(thread_handle));
 855 
 856       new_thread->set_threadObj(JNIHandles::resolve_non_null(thread_handle));
 857       if (comp != NULL) {
 858         new_thread->as_CompilerThread()->set_compiler(comp);
 859       }
 860       Threads::add(new_thread);
 861       Thread::start(new_thread);
 862     }
 863   }
 864 
 865   // First release lock before aborting VM.
 866   if (new_thread == NULL || new_thread->osthread() == NULL) {
 867     if (UseDynamicNumberOfCompilerThreads && comp != NULL && comp->num_compiler_threads() > 0) {
 868       if (new_thread != NULL) {
 869         new_thread->smr_delete();
 870       }
 871       return NULL;
 872     }
 873     vm_exit_during_initialization("java.lang.OutOfMemoryError",
 874                                   os::native_thread_creation_failed_msg());
 875   }
 876 
 877   // Let go of Threads_lock before yielding
 878   os::naked_yield(); // make sure that the compiler thread is started early (especially helpful on SOLARIS)
 879 
 880   return new_thread;
 881 }

 884 void CompileBroker::init_compiler_sweeper_threads() {
 885   // Ensure any exceptions lead to vm_exit_during_initialization.
 886   EXCEPTION_MARK;
 887 #if !defined(ZERO)
 888   assert(_c2_count > 0 || _c1_count > 0, "No compilers?");
 889 #endif // !ZERO
 890   // Initialize the compilation queue
 891   if (_c2_count > 0) {
 892     const char* name = JVMCI_ONLY(UseJVMCICompiler ? "JVMCI compile queue" :) "C2 compile queue";
 893     _c2_compile_queue  = new CompileQueue(name);
 894     _compiler2_objects = NEW_C_HEAP_ARRAY(jobject, _c2_count, mtCompiler);
 895     _compiler2_logs = NEW_C_HEAP_ARRAY(CompileLog*, _c2_count, mtCompiler);
 896   }
 897   if (_c1_count > 0) {
 898     _c1_compile_queue  = new CompileQueue("C1 compile queue");
 899     _compiler1_objects = NEW_C_HEAP_ARRAY(jobject, _c1_count, mtCompiler);
 900     _compiler1_logs = NEW_C_HEAP_ARRAY(CompileLog*, _c1_count, mtCompiler);
 901   }
 902 
 903   char name_buffer[256];
 904 
 905   for (int i = 0; i < _c2_count; i++) {
 906     jobject thread_handle = NULL;
 907     // Create all j.l.Thread objects for C1 and C2 threads here, but only one
 908     // for JVMCI compiler which can create further ones on demand.
 909     JVMCI_ONLY(if (!UseJVMCICompiler || !UseDynamicNumberOfCompilerThreads || i == 0) {)
 910     // Create a name for our thread.
 911     sprintf(name_buffer, "%s CompilerThread%d", _compilers[1]->name(), i);
 912     Handle thread_oop = create_thread_oop(name_buffer, CHECK);
 913     thread_handle = JNIHandles::make_global(thread_oop);
 914     JVMCI_ONLY(})
 915     _compiler2_objects[i] = thread_handle;
 916     _compiler2_logs[i] = NULL;
 917 
 918     if (!UseDynamicNumberOfCompilerThreads || i == 0) {
 919       JavaThread *ct = make_thread(thread_handle, _c2_compile_queue, _compilers[1], THREAD);
 920       assert(ct != NULL, "should have been handled for initial thread");
 921       _compilers[1]->set_num_compiler_threads(i + 1);
 922       if (TraceCompilerThreads) {
 923         ResourceMark rm;
 924         MutexLocker mu(Threads_lock);
 925         tty->print_cr("Added initial compiler thread %s", ct->get_thread_name());
 926       }
 927     }
 928   }
 929 
 930   for (int i = 0; i < _c1_count; i++) {
 931     // Create a name for our thread.
 932     sprintf(name_buffer, "C1 CompilerThread%d", i);
 933     Handle thread_oop = create_thread_oop(name_buffer, CHECK);
 934     jobject thread_handle = JNIHandles::make_global(thread_oop);
 935     _compiler1_objects[i] = thread_handle;
 936     _compiler1_logs[i] = NULL;
 937 
 938     if (!UseDynamicNumberOfCompilerThreads || i == 0) {
 939       JavaThread *ct = make_thread(thread_handle, _c1_compile_queue, _compilers[0], THREAD);
 940       assert(ct != NULL, "should have been handled for initial thread");
 941       _compilers[0]->set_num_compiler_threads(i + 1);
 942       if (TraceCompilerThreads) {
 943         ResourceMark rm;
 944         MutexLocker mu(Threads_lock);
 945         tty->print_cr("Added initial compiler thread %s", ct->get_thread_name());
 946       }
 947     }
 948   }
 949 
 950   if (UsePerfData) {
 951     PerfDataManager::create_constant(SUN_CI, "threads", PerfData::U_Bytes, _c1_count + _c2_count, CHECK);
 952   }
 953 
 954   if (MethodFlushing) {
 955     // Initialize the sweeper thread
 956     Handle thread_oop = create_thread_oop("Sweeper thread", CHECK);
 957     jobject thread_handle = JNIHandles::make_local(THREAD, thread_oop());
 958     make_thread(thread_handle, NULL, NULL, THREAD);
 959   }
 960 }

 962 void CompileBroker::possibly_add_compiler_threads(Thread* THREAD) {
 963 
 964   julong available_memory = os::available_memory();
 965   // If SegmentedCodeCache is off, both values refer to the single heap (with type CodeBlobType::All).
 966   size_t available_cc_np  = CodeCache::unallocated_capacity(CodeBlobType::MethodNonProfiled),
 967          available_cc_p   = CodeCache::unallocated_capacity(CodeBlobType::MethodProfiled);
 968 
 969   // Only do attempt to start additional threads if the lock is free.
 970   if (!CompileThread_lock->try_lock()) return;
 971 
 972   if (_c2_compile_queue != NULL) {
 973     int old_c2_count = _compilers[1]->num_compiler_threads();
 974     int new_c2_count = MIN4(_c2_count,
 975         _c2_compile_queue->size() / 2,
 976         (int)(available_memory / (200*M)),
 977         (int)(available_cc_np / (128*K)));
 978 
 979     for (int i = old_c2_count; i < new_c2_count; i++) {
 980 #if INCLUDE_JVMCI
 981       if (UseJVMCICompiler) {
 982         // Native compiler threads as used in C1/C2 can reuse the j.l.Thread
 983         // objects as their existence is completely hidden from the rest of
 984         // the VM (and those compiler threads can't call Java code to do the
 985         // creation anyway). For JVMCI we have to create new j.l.Thread objects
 986         // as they are visible and we can see unexpected thread lifecycle
 987         // transitions if we bind them to new JavaThreads.
 988         if (!THREAD->can_call_java()) break;
 989         char name_buffer[256];
 990         sprintf(name_buffer, "%s CompilerThread%d", _compilers[1]->name(), i);
 991         Handle thread_oop;
 992         {
 993           // We have to give up the lock temporarily for the Java calls.
 994           MutexUnlocker mu(CompileThread_lock);
 995           thread_oop = create_thread_oop(name_buffer, THREAD);
 996         }
 997         if (HAS_PENDING_EXCEPTION) {
 998           if (TraceCompilerThreads) {
 999             ResourceMark rm;
1000             tty->print_cr("JVMCI compiler thread creation failed:");
1001             PENDING_EXCEPTION->print();
1002           }
1003           CLEAR_PENDING_EXCEPTION;
1004           break;
1005         }
1006         // Check if another thread has beaten us during the Java calls.
1007         if (_compilers[1]->num_compiler_threads() != i) break;
1008         jobject thread_handle = JNIHandles::make_global(thread_oop);
1009         assert(compiler2_object(i) == NULL, "Old one must be released!");
1010         _compiler2_objects[i] = thread_handle;
1011       }
1012 #endif
1013       JavaThread *ct = make_thread(compiler2_object(i), _c2_compile_queue, _compilers[1], THREAD);
1014       if (ct == NULL) break;
1015       _compilers[1]->set_num_compiler_threads(i + 1);
1016       if (TraceCompilerThreads) {
1017         ResourceMark rm;
1018         MutexLocker mu(Threads_lock);
1019         tty->print_cr("Added compiler thread %s (available memory: %dMB, available non-profiled code cache: %dMB)",
1020                       ct->get_thread_name(), (int)(available_memory/M), (int)(available_cc_np/M));
1021       }
1022     }
1023   }
1024 
1025   if (_c1_compile_queue != NULL) {
1026     int old_c1_count = _compilers[0]->num_compiler_threads();
1027     int new_c1_count = MIN4(_c1_count,
1028         _c1_compile_queue->size() / 4,
1029         (int)(available_memory / (100*M)),
1030         (int)(available_cc_p / (128*K)));
1031 
1032     for (int i = old_c1_count; i < new_c1_count; i++) {
1033       JavaThread *ct = make_thread(compiler1_object(i), _c1_compile_queue, _compilers[0], THREAD);
1034       if (ct == NULL) break;
1035       _compilers[0]->set_num_compiler_threads(i + 1);
1036       if (TraceCompilerThreads) {
1037         ResourceMark rm;
1038         MutexLocker mu(Threads_lock);
1039         tty->print_cr("Added compiler thread %s (available memory: %dMB, available profiled code cache: %dMB)",
1040                       ct->get_thread_name(), (int)(available_memory/M), (int)(available_cc_p/M));
1041       }
1042     }
1043   }
1044 
1045   CompileThread_lock->unlock();
1046 }

1065 // ------------------------------------------------------------------
1066 // CompileBroker::compile_method
1067 //
1068 // Request compilation of a method.
1069 void CompileBroker::compile_method_base(const methodHandle& method,
1070                                         int osr_bci,
1071                                         int comp_level,
1072                                         const methodHandle& hot_method,
1073                                         int hot_count,
1074                                         CompileTask::CompileReason compile_reason,
1075                                         bool blocking,
1076                                         Thread* thread) {
1077   guarantee(!method->is_abstract(), "cannot compile abstract methods");
1078   assert(method->method_holder()->is_instance_klass(),
1079          "sanity check");
1080   assert(!method->method_holder()->is_not_initialized(),
1081          "method holder must be initialized");
1082   assert(!method->is_method_handle_intrinsic(), "do not enqueue these guys");
1083 
1084   if (CIPrintRequests) {
1085     tty->print("request: ");
1086     method->print_short_name(tty);
1087     if (osr_bci != InvocationEntryBci) {
1088       tty->print(" osr_bci: %d", osr_bci);
1089     }
1090     tty->print(" level: %d comment: %s count: %d", comp_level, CompileTask::reason_name(compile_reason), hot_count);
1091     if (!hot_method.is_null()) {
1092       tty->print(" hot: ");
1093       if (hot_method() != method()) {
1094           hot_method->print_short_name(tty);
1095       } else {
1096         tty->print("yes");
1097       }
1098     }
1099     tty->cr();
1100   }
1101 
1102   // A request has been made for compilation.  Before we do any
1103   // real work, check to see if the method has been compiled
1104   // in the meantime with a definitive result.
1105   if (compilation_is_complete(method, osr_bci, comp_level)) {
1106     return;
1107   }
1108 
1109 #ifndef PRODUCT
1110   if (osr_bci != -1 && !FLAG_IS_DEFAULT(OSROnlyBCI)) {
1111     if ((OSROnlyBCI > 0) ? (OSROnlyBCI != osr_bci) : (-OSROnlyBCI == osr_bci)) {
1112       // Positive OSROnlyBCI means only compile that bci.  Negative means don't compile that BCI.
1113       return;
1114     }
1115   }
1116 #endif
1117 
1118   // If this method is already in the compile queue, then
1119   // we do not block the current thread.
1120   if (compilation_is_in_queue(method)) {
1121     // We may want to decay our counter a bit here to prevent
1122     // multiple denied requests for compilation.  This is an
1123     // open compilation policy issue. Note: The other possibility,
1124     // in the case that this is a blocking compile request, is to have
1125     // all subsequent blocking requesters wait for completion of
1126     // ongoing compiles. Note that in this case we'll need a protocol
1127     // for freeing the associated compile tasks. [Or we could have
1128     // a single static monitor on which all these waiters sleep.]
1129     return;
1130   }
1131 
1132   if (TieredCompilation) {
1133     // Tiered policy requires MethodCounters to exist before adding a method to
1134     // the queue. Create if we don't have them yet.
1135     method->get_method_counters(thread);
1136   }
1137 
1138   // Outputs from the following MutexLocker block:
1139   CompileTask* task     = NULL;
1140   CompileQueue* queue  = compile_queue(comp_level);
1141 
1142   // Acquire our lock.
1143   {
1144     MutexLocker locker(thread, MethodCompileQueue_lock);
1145 
1146     // Make sure the method has not slipped into the queues since
1147     // last we checked; note that those checks were "fast bail-outs".
1148     // Here we need to be more careful, see 14012000 below.
1149     if (compilation_is_in_queue(method)) {
1150       return;
1151     }
1152 
1153     // We need to check again to see if the compilation has
1154     // completed.  A previous compilation may have registered
1155     // some result.
1156     if (compilation_is_complete(method, osr_bci, comp_level)) {
1157       return;
1158     }
1159 
1160     // We now know that this compilation is not pending, complete,
1161     // or prohibited.  Assign a compile_id to this compilation
1162     // and check to see if it is in our [Start..Stop) range.
1163     int compile_id = assign_compile_id(method, osr_bci);
1164     if (compile_id == 0) {
1165       // The compilation falls outside the allowed range.
1166       return;
1167     }
1168 
1169 #if INCLUDE_JVMCI
1170     if (UseJVMCICompiler && blocking) {
1171       // Don't allow blocking compiles for requests triggered by JVMCI.
1172       if (thread->is_Compiler_thread()) {
1173         blocking = false;
1174       }
1175 
1176       if (!UseJVMCINativeLibrary) {
1177         // Don't allow blocking compiles if inside a class initializer or while performing class loading
1178         vframeStream vfst((JavaThread*) thread);
1179         for (; !vfst.at_end(); vfst.next()) {
1180         if (vfst.method()->is_class_initializer() ||
1181               (vfst.method()->method_holder()->is_subclass_of(SystemDictionary::ClassLoader_klass()) &&
1182                   vfst.method()->name() == vmSymbols::loadClass_name())) {
1183             blocking = false;
1184             break;
1185           }
1186         }
1187       }
1188 
1189       // Don't allow blocking compilation requests to JVMCI
1190       // if JVMCI itself is not yet initialized
1191       if (!JVMCI::is_compiler_initialized() && compiler(comp_level)->is_jvmci()) {
1192         blocking = false;
1193       }
1194 
1195       // Don't allow blocking compilation requests if we are in JVMCIRuntime::shutdown
1196       // to avoid deadlock between compiler thread(s) and threads run at shutdown
1197       // such as the DestroyJavaVM thread.
1198       if (JVMCI::shutdown_called()) {
1199         blocking = false;
1200       }
1201     }
1202 #endif // INCLUDE_JVMCI
1203 
1204     // We will enter the compilation in the queue.
1205     // 14012000: Note that this sets the queued_for_compile bits in
1206     // the target method. We can now reason that a method cannot be
1207     // queued for compilation more than once, as follows:
1208     // Before a thread queues a task for compilation, it first acquires
1209     // the compile queue lock, then checks if the method's queued bits
1210     // are set or it has already been compiled. Thus there can not be two
1211     // instances of a compilation task for the same method on the
1212     // compilation queue. Consider now the case where the compilation
1213     // thread has already removed a task for that method from the queue
1214     // and is in the midst of compiling it. In this case, the
1215     // queued_for_compile bits must be set in the method (and these
1216     // will be visible to the current thread, since the bits were set
1217     // under protection of the compile queue lock, which we hold now.
1218     // When the compilation completes, the compiler thread first sets
1219     // the compilation result and then clears the queued_for_compile
1220     // bits. Neither of these actions are protected by a barrier (or done
1221     // under the protection of a lock), so the only guarantee we have
1222     // (on machines with TSO (Total Store Order)) is that these values
1223     // will update in that order. As a result, the only combinations of
1224     // these bits that the current thread will see are, in temporal order:
1225     // <RESULT, QUEUE> :
1226     //     <0, 1> : in compile queue, but not yet compiled
1227     //     <1, 1> : compiled but queue bit not cleared
1228     //     <1, 0> : compiled and queue bit cleared
1229     // Because we first check the queue bits then check the result bits,
1230     // we are assured that we cannot introduce a duplicate task.
1231     // Note that if we did the tests in the reverse order (i.e. check
1232     // result then check queued bit), we could get the result bit before
1233     // the compilation completed, and the queue bit after the compilation
1234     // completed, and end up introducing a "duplicate" (redundant) task.
1235     // In that case, the compiler thread should first check if a method
1236     // has already been compiled before trying to compile it.
1237     // NOTE: in the event that there are multiple compiler threads and
1238     // there is de-optimization/recompilation, things will get hairy,
1239     // and in that case it's best to protect both the testing (here) of
1240     // these bits, and their updating (here and elsewhere) under a
1241     // common lock.
1242     task = create_compile_task(queue,
1243                                compile_id, method,
1244                                osr_bci, comp_level,
1245                                hot_method, hot_count, compile_reason,
1246                                blocking);
1247   }
1248 
1249   if (blocking) {
1250     wait_for_completion(task);
1251   }
1252 }

1409     }

1586 /**
1587  * Waits for a JVMCI compiler to complete a given task. This thread
1588  * waits until either the task completes or it sees no JVMCI compilation
1589  * progress for N consecutive milliseconds where N is
1590  * JVMCI_COMPILATION_PROGRESS_WAIT_TIMESLICE *
1591  * JVMCI_COMPILATION_PROGRESS_WAIT_ATTEMPTS.
1592  *
1593  * @return true if this thread needs to free/recycle the task
1594  */
1595 bool CompileBroker::wait_for_jvmci_completion(JVMCICompiler* jvmci, CompileTask* task, JavaThread* thread) {
1596   MonitorLocker ml(thread, task->lock());
1597   int progress_wait_attempts = 0;
1598   int methods_compiled = jvmci->methods_compiled();
1599   while (!task->is_complete() && !is_compilation_disabled_forever() &&
1600          ml.wait(JVMCI_COMPILATION_PROGRESS_WAIT_TIMESLICE)) {
1601     CompilerThread* jvmci_compiler_thread = task->jvmci_compiler_thread();
1602 
1603     bool progress;
1604     if (jvmci_compiler_thread != NULL) {
1605       // If the JVMCI compiler thread is not blocked or suspended, we deem it to be making progress.
1606       progress = jvmci_compiler_thread->thread_state() != _thread_blocked &&
1607         !jvmci_compiler_thread->is_external_suspend();
1608     } else {
1609       // Still waiting on JVMCI compiler queue. This thread may be holding a lock
1610       // that all JVMCI compiler threads are blocked on. We use the counter for
1611       // successful JVMCI compilations to determine whether JVMCI compilation
1612       // is still making progress through the JVMCI compiler queue.
1613       progress = jvmci->methods_compiled() != methods_compiled;
1614     }
1615 
1616     if (!progress) {
1617       if (++progress_wait_attempts == JVMCI_COMPILATION_PROGRESS_WAIT_ATTEMPTS) {
1618         if (PrintCompilation) {
1619           task->print(tty, "wait for blocking compilation timed out");
1620         }
1621         break;
1622       }
1623     } else {
1624       progress_wait_attempts = 0;
1625       if (jvmci_compiler_thread == NULL) {
1626         methods_compiled = jvmci->methods_compiled();
1627       }
1628     }
1629   }
1630   task->clear_waiter();
1631   return task->is_complete();
1632 }

1635 /**
1636  *  Wait for the compilation task to complete.
1637  */
1638 void CompileBroker::wait_for_completion(CompileTask* task) {
1639   if (CIPrintCompileQueue) {
1640     ttyLocker ttyl;
1641     tty->print_cr("BLOCKING FOR COMPILE");
1642   }
1643 
1644   assert(task->is_blocking(), "can only wait on blocking task");
1645 
1646   JavaThread* thread = JavaThread::current();
1647 
1648   methodHandle method(thread, task->method());
1649   bool free_task;
1650 #if INCLUDE_JVMCI
1651   AbstractCompiler* comp = compiler(task->comp_level());
1652   if (comp->is_jvmci()) {
1653     free_task = wait_for_jvmci_completion((JVMCICompiler*) comp, task, thread);
1654   } else
1655 #endif
1656   {
1657     MonitorLocker ml(thread, task->lock());

1660       ml.wait();
1661     }

1672     assert(task->is_complete(), "Compilation should have completed");

1674

1676     // blocking CompileTask. Since there is only one waiter ever
1677     // waiting on a CompileTask, we know that no one else will
1678     // be using this CompileTask; we can free it.
1679     CompileTask::free(task);
1680   }

1684  * Initialize compiler thread(s) + compiler object(s). The postcondition
1685  * of this function is that the compiler runtimes are initialized and that
1686  * compiler threads can start compiling.
1687  */
1688 bool CompileBroker::init_compiler_runtime() {
1689   CompilerThread* thread = CompilerThread::current();
1690   AbstractCompiler* comp = thread->compiler();
1691   // Final sanity check - the compiler object must exist
1692   guarantee(comp != NULL, "Compiler object must exist");

1695     // Must switch to native to allocate ci_env
1696     ThreadToNativeFromVM ttn(thread);

1704     ThreadInVMfromNative tv(thread);
1705     ResetNoHandleMark rnhm;

1714     // particular compiler runtime will ever start to compile methods.
1715     shutdown_compiler_runtime(comp, thread);
1716     return false;
1717   }
1718

1720   if (comp->is_c1() && (thread->get_buffer_blob() == NULL)) {
1721     warning("Initialization of %s thread failed (no space to run compilers)", thread->name());
1722     return false;
1723   }

1728 /**
1729  * If C1 and/or C2 initialization failed, we shut down all compilation.
1730  * We do this to keep things simple. This can be changed if it ever turns
1731  * out to be a problem.
1732  */
1733 void CompileBroker::shutdown_compiler_runtime(AbstractCompiler* comp, CompilerThread* thread) {
1734   // Free buffer blob, if allocated
1735   if (thread->get_buffer_blob() != NULL) {
1736     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1737     CodeCache::free(thread->get_buffer_blob());
1738   }
1739 
1740   if (comp->should_perform_shutdown()) {
1741     // There are two reasons for shutting down the compiler
1742     // 1) compiler runtime initialization failed
1743     // 2) The code cache is full and the following flag is set: -XX:-UseCodeCacheFlushing
1744     warning("%s initialization failed. Shutting down all compilers", comp->name());
1745 
1746     // Only one thread per compiler runtime object enters here
1747     // Set state to shut down
1748     comp->set_shut_down();
1749

1751     if (_c1_compile_queue != NULL) {

1769 /**
1770  * Helper function to create new or reuse old CompileLog.
1771  */
1772 CompileLog* CompileBroker::get_log(CompilerThread* ct) {
1773   if (!LogCompilation) return NULL;
1774 
1775   AbstractCompiler *compiler = ct->compiler();
1776   bool c1 = compiler->is_c1();
1777   jobject* compiler_objects = c1 ? _compiler1_objects : _compiler2_objects;
1778   assert(compiler_objects != NULL, "must be initialized at this point");
1779   CompileLog** logs = c1 ? _compiler1_logs : _compiler2_logs;
1780   assert(logs != NULL, "must be initialized at this point");

1784   oop compiler_obj = ct->threadObj();

1786   bool found = false;
1787   for (; compiler_number < count; compiler_number++) {
1788     if (JNIHandles::resolve_non_null(compiler_objects[compiler_number]) == compiler_obj) {
1789       found = true;
1790       break;
1791     }
1792   }
1793   assert(found, "Compiler must exist at this point");
1794 
1795   // Determine pointer for this thread's log.
1796   CompileLog** log_ptr = &logs[compiler_number];
1797 
1798   // Return old one if it exists.
1799   CompileLog* log = *log_ptr;
1800   if (log != NULL) {
1801     ct->init_log(log);
1802     return log;
1803   }
1804 
1805   // Create a new one and remember it.
1806   init_compiler_thread_log();
1807   log = ct->log();
1808   *log_ptr = log;
1809   return log;
1810 }

1812 // ------------------------------------------------------------------
1813 // CompileBroker::compiler_thread_loop
1814 //
1815 // The main loop run by a CompilerThread.
1816 void CompileBroker::compiler_thread_loop() {
1817   CompilerThread* thread = CompilerThread::current();
1818   CompileQueue* queue = thread->queue();
1819   // For the thread that initializes the ciObjectFactory
1820   // this resource mark holds all the shared objects
1821   ResourceMark rm;
1822 
1823   // First thread to get here will initialize the compiler interface
1824 
1825   {
1826     ASSERT_IN_VM;
1827     MutexLocker only_one (thread, CompileThread_lock);
1828     if (!ciObjectFactory::is_initialized()) {
1829       ciObjectFactory::initialize();
1830     }
1831   }
1832 
1833   // Open a log.
1834   CompileLog* log = get_log(thread);
1835   if (log != NULL) {
1836     log->begin_elem("start_compile_thread name='%s' thread='" UINTX_FORMAT "' process='%d'",
1837                     thread->name(),
1838                     os::current_thread_id(),
1839                     os::current_process_id());
1840     log->stamp();
1841     log->end_elem();
1842   }
1843 
1844   // If compiler thread/runtime initialization fails, exit the compiler thread
1845   if (!init_compiler_runtime()) {
1846     return;
1847   }
1848 
1849   thread->start_idle_timer();
1850 
1851   // Poll for new compilation tasks as long as the JVM runs. Compilation
1852   // should only be disabled if something went wrong while initializing the
1853   // compiler runtimes. This, in turn, should not happen. The only known case
1854   // when compiler runtime initialization fails is if there is not enough free
1855   // space in the code cache to generate the necessary stubs, etc.
1856   while (!is_compilation_disabled_forever()) {
1857     // We need this HandleMark to avoid leaking VM handles.
1858     HandleMark hm(thread);
1859 
1860     CompileTask* task = queue->get();
1861     if (task == NULL) {
1862       if (UseDynamicNumberOfCompilerThreads) {
1863         // Access compiler_count under lock to enforce consistency.
1864         MutexLocker only_one(CompileThread_lock);
1865         if (can_remove(thread, true)) {
1866           if (TraceCompilerThreads) {
1867             tty->print_cr("Removing compiler thread %s after " JLONG_FORMAT " ms idle time",
1868                           thread->name(), thread->idle_time_millis());
1869           }
1870           // Free buffer blob, if allocated
1871           if (thread->get_buffer_blob() != NULL) {
1872             MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1873             CodeCache::free(thread->get_buffer_blob());
1874           }
1875           return; // Stop this thread.
1876         }
1877       }
1878     } else {
1879       // Assign the task to the current thread.  Mark this compilation
1880       // thread as active for the profiler.
1881       // CompileTaskWrapper also keeps the Method* from being deallocated if redefinition
1882       // occurs after fetching the compile task off the queue.
1883       CompileTaskWrapper ctw(task);
1884       nmethodLocker result_handle;  // (handle for the nmethod produced by this task)
1885       task->set_code_handle(&result_handle);
1886       methodHandle method(thread, task->method());
1887 
1888       // Never compile a method if breakpoints are present in it
1889       if (method()->number_of_breakpoints() == 0) {
1890         // Compile the method.
1891         if ((UseCompiler || AlwaysCompileLoopMethods) && CompileBroker::should_compile_new_jobs()) {
1892           invoke_compiler_on_method(task);
1893           thread->start_idle_timer();
1894         } else {
1895           // After compilation is disabled, remove remaining methods from queue
1896           method->clear_queued_for_compilation();
1897           task->set_failure_reason("compilation is disabled");
1898         }
1899       }
1900 
1901       if (UseDynamicNumberOfCompilerThreads) {
1902         possibly_add_compiler_threads(thread);
1903         assert(!thread->has_pending_exception(), "should have been handled");
1904       }
1905     }
1906   }
1907 
1908   // Shut down compiler runtime
1909   shutdown_compiler_runtime(thread->compiler(), thread);
1910 }

2364   // check if we need to truncate the string
2365   if (s1len + s2len + 2 > maxLen) {
2366 
2367     // the strategy is to lop off the leading characters of the
2368     // class name and the trailing characters of the method name.
2369 
2370     if (s2len + 2 > maxLen) {
2371       // lop of the entire class name string, let snprintf handle
2372       // truncation of the method name.
2373       class_name += s1len; // null string
2374     }
2375     else {
2376       // lop off the extra characters from the front of the class name
2377       class_name += ((s1len + s2len + 2) - maxLen);
2378     }
2379   }
2380 
2381   jio_snprintf(current_method, maxLen, "%s %s", class_name, method_name);
2382 
2383   int last_compile_type = normal_compile;
2384   if (CICountOSR && is_osr) {
2385     last_compile_type = osr_compile;
2386   }
2387 
2388   CompilerCounters* counters = thread->counters();
2389   counters->set_current_method(current_method);
2390   counters->set_compile_type((jlong) last_compile_type);
2391 }

2431   bool success = task->is_success();

2433   uint compile_id = task->compile_id();

2437 
2438   assert(code == NULL || code->is_locked_by_vm(), "will survive the MutexLocker");
2439   MutexLocker locker(CompileStatistics_lock);
2440 
2441   // _perf variables are production performance counters which are
2442   // updated regardless of the setting of the CITime and CITimeEach flags
2443   //
2444 
2445   // account all time, including bailouts and failures in this counter;
2446   // C1 and C2 counters are counting both successful and unsuccessful compiles
2447   _t_total_compilation.add(time);
2448 
2449   if (!success) {
2450     _total_bailout_count++;
2451     if (UsePerfData) {
2452       _perf_last_failed_method->set_value(counters->current_method());
2453       _perf_last_failed_type->set_value(counters->compile_type());
2454       _perf_total_bailout_count->inc();
2455     }
2456     _t_bailedout_compilation.add(time);
2457   } else if (code == NULL) {
2458     if (UsePerfData) {
2459       _perf_last_invalidated_method->set_value(counters->current_method());
2460       _perf_last_invalidated_type->set_value(counters->compile_type());
2461       _perf_total_invalidated_count->inc();
2462     }
2463     _total_invalidated_count++;
2464     _t_invalidated_compilation.add(time);
2465   } else {
2466     // Compilation succeeded
2467 
2468     // update compilation ticks - used by the implementation of
2469     // java.lang.management.CompilationMBean
2470     _perf_total_compilation->inc(time.ticks());
2471     _peak_compilation_time = time.milliseconds() > _peak_compilation_time ? time.milliseconds() : _peak_compilation_time;
2472 
2473     if (CITime) {
2474       int bytes_compiled = method->code_size() + task->num_inlined_bytecodes();
2475       if (is_osr) {
2476         _t_osr_compilation.add(time);
2477         _sum_osr_bytes_compiled += bytes_compiled;
2478       } else {
2479         _t_standard_compilation.add(time);
2480         _sum_standard_bytes_compiled += method->code_size() + task->num_inlined_bytecodes();
2481       }
2482 
2483 #if INCLUDE_JVMCI
2484       AbstractCompiler* comp = compiler(task->comp_level());
2485       if (comp) {
2486         CompilerStatistics* stats = comp->stats();
2487         if (stats) {
2488           if (is_osr) {
2489             stats->_osr.update(time, bytes_compiled);
2490           } else {
2491             stats->_standard.update(time, bytes_compiled);
2492           }
2493           stats->_nmethods_size += code->total_size();
2494           stats->_nmethods_code_size += code->insts_size();
2495         } else { // if (!stats)
2496           assert(false, "Compiler statistics object must exist");
2497         }
2498       } else { // if (!comp)
2499         assert(false, "Compiler object must exist");
2500       }
2501 #endif // INCLUDE_JVMCI
2502     }
2503 
2504     if (UsePerfData) {
2505       // save the name of the last method compiled
2506       _perf_last_method->set_value(counters->current_method());
2507       _perf_last_compile_type->set_value(counters->compile_type());
2508       _perf_last_compile_size->set_value(method->code_size() +
2509                                          task->num_inlined_bytecodes());
2510       if (is_osr) {
2511         _perf_osr_compilation->inc(time.ticks());
2512         _perf_sum_osr_bytes_compiled->inc(method->code_size() + task->num_inlined_bytecodes());
2513       } else {
2514         _perf_standard_compilation->inc(time.ticks());
2515         _perf_sum_standard_bytes_compiled->inc(method->code_size() + task->num_inlined_bytecodes());
2516       }
2517     }
2518 
2519     if (CITimeEach) {
2520       float bytes_per_sec = 1.0 * (method->code_size() + task->num_inlined_bytecodes()) / time.seconds();
2521       tty->print_cr("%3d   seconds: %f bytes/sec : %f (bytes %d + %d inlined)",
2522                     compile_id, time.seconds(), bytes_per_sec, method->code_size(), task->num_inlined_bytecodes());
2523     }
2524 
2525     // Collect counts of successful compilations
2526     _sum_nmethod_size      += code->total_size();
2527     _sum_nmethod_code_size += code->insts_size();
2528     _total_compile_count++;
2529 
2530     if (UsePerfData) {
2531       _perf_sum_nmethod_size->inc(     code->total_size());
2532       _perf_sum_nmethod_code_size->inc(code->insts_size());
2533       _perf_total_compile_count->inc();
2534     }
2535 
2536     if (is_osr) {
2537       if (UsePerfData) _perf_total_osr_compile_count->inc();
2538       _total_osr_compile_count++;
2539     } else {
2540       if (UsePerfData) _perf_total_standard_compile_count->inc();
2541       _total_standard_compile_count++;
2542     }
2543   }
2544   // set the current method for the thread to null
2545   if (UsePerfData) counters->set_current_method("");
2546 }

2723 // Note: tty_lock must not be held upon entry to this function.
2724 //       Print functions called from herein do "micro-locking" on tty_lock.
2725 //       That's a tradeoff which keeps together important blocks of output.
2726 //       At the same time, continuous tty_lock hold time is kept in check,
2727 //       preventing concurrently printing threads from stalling a long time.
2728 void CompileBroker::print_heapinfo(outputStream* out, const char* function, size_t granularity) {
2729   TimeStamp ts_total;
2730   TimeStamp ts_global;
2731   TimeStamp ts;
2732 
2733   bool allFun = !strcmp(function, "all");
2734   bool aggregate = !strcmp(function, "aggregate") || !strcmp(function, "analyze") || allFun;
2735   bool usedSpace = !strcmp(function, "UsedSpace") || allFun;
2736   bool freeSpace = !strcmp(function, "FreeSpace") || allFun;
2737   bool methodCount = !strcmp(function, "MethodCount") || allFun;
2738   bool methodSpace = !strcmp(function, "MethodSpace") || allFun;
2739   bool methodAge = !strcmp(function, "MethodAge") || allFun;
2740   bool methodNames = !strcmp(function, "MethodNames") || allFun;
2741   bool discard = !strcmp(function, "discard") || allFun;
2742 
2743   if (out == NULL) {
2744     out = tty;
2745   }
2746 
2747   if (!(aggregate || usedSpace || freeSpace || methodCount || methodSpace || methodAge || methodNames || discard)) {
2748     out->print_cr("\n__ CodeHeapStateAnalytics: Function %s is not supported", function);
2749     out->cr();
2750     return;
2751   }
2752 
2753   ts_total.update(); // record starting point
2754 
2755   if (aggregate) {
2756     print_info(out);
2757   }
2758 
2759   // We hold the CodeHeapStateAnalytics_lock all the time, from here until we leave this function.
2760   // That prevents another thread from destroying our view on the CodeHeap.
2761   // When we request individual parts of the analysis via the jcmd interface, it is possible
2762   // that in between another thread (another jcmd user or the vm running into CodeCache OOM)
2763   // updated the aggregated data. That's a tolerable tradeoff because we can't hold a lock
2764   // across user interaction.
2765   // Acquire this lock before acquiring the CodeCache_lock.
2766   // CodeHeapStateAnalytics_lock could be held by a concurrent thread for a long time,
2767   // leading to an unnecessarily long hold time of the CodeCache_lock.
2768   ts.update(); // record starting point
2769   MutexLocker mu1(CodeHeapStateAnalytics_lock, Mutex::_no_safepoint_check_flag);
2770   out->print_cr("\n__ CodeHeapStateAnalytics lock wait took %10.3f seconds _________\n", ts.seconds());
2771 
2772   // If we serve an "allFun" call, it is beneficial to hold the CodeCache_lock
2773   // for the entire duration of aggregation and printing. That makes sure
2774   // we see a consistent picture and do not run into issues caused by
2775   // the CodeHeap being altered concurrently.
2776   Mutex* global_lock   = allFun ? CodeCache_lock : NULL;
2777   Mutex* function_lock = allFun ? NULL : CodeCache_lock;
2778   ts_global.update(); // record starting point
2779   MutexLocker mu2(global_lock, Mutex::_no_safepoint_check_flag);
2780   if (global_lock != NULL) {
2781     out->print_cr("\n__ CodeCache (global) lock wait took %10.3f seconds _________\n", ts_global.seconds());
2782     ts_global.update(); // record starting point
2783   }
2784 
2785   if (aggregate) {
2786     ts.update(); // record starting point
2787     MutexLocker mu3(function_lock, Mutex::_no_safepoint_check_flag);
2788     if (function_lock != NULL) {
2789       out->print_cr("\n__ CodeCache (function) lock wait took %10.3f seconds _________\n", ts.seconds());
2790     }
2791 
2792     ts.update(); // record starting point
2793     CodeCache::aggregate(out, granularity);
2794     if (function_lock != NULL) {
2795       out->print_cr("\n__ CodeCache (function) lock hold took %10.3f seconds _________\n", ts.seconds());
2796     }
2797   }
2798 
2799   if (usedSpace) CodeCache::print_usedSpace(out);
2800   if (freeSpace) CodeCache::print_freeSpace(out);
2801   if (methodCount) CodeCache::print_count(out);
2802   if (methodSpace) CodeCache::print_space(out);
2803   if (methodAge) CodeCache::print_age(out);
2804   if (methodNames) {
2805     // print_names() has shown to be sensitive to concurrent CodeHeap modifications.
2806     // Therefore, request  the CodeCache_lock before calling...
2807     MutexLocker mu3(function_lock, Mutex::_no_safepoint_check_flag);
2808     CodeCache::print_names(out);
2809   }
2810   if (discard) CodeCache::discard(out);
2811 
2812   if (global_lock != NULL) {
2813     out->print_cr("\n__ CodeCache (global) lock hold took %10.3f seconds _________\n", ts_global.seconds());
2814   }
2815   out->print_cr("\n__ CodeHeapStateAnalytics total duration %10.3f seconds _________\n", ts_total.seconds());
