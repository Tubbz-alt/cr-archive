 213     lm.print("\n");
 214     log(thread, "%s", (const char*)lm);

 246 CompileTaskWrapper::CompileTaskWrapper(CompileTask* task) {
 247   CompilerThread* thread = CompilerThread::current();
 248   thread->set_task(task);
 249 #if INCLUDE_JVMCI
 250   if (task->is_blocking() && CompileBroker::compiler(task->comp_level())->is_jvmci()) {
 251     task->set_jvmci_compiler_thread(thread);
 252   }
 253 #endif
 254   CompileLog*     log  = thread->log();
 255   if (log != NULL && !task->is_unloaded())  task->log_task_start(log);
 256 }
 257 
 258 CompileTaskWrapper::~CompileTaskWrapper() {
 259   CompilerThread* thread = CompilerThread::current();
 260   CompileTask* task = thread->task();
 261   CompileLog*  log  = thread->log();
 262   if (log != NULL && !task->is_unloaded())  task->log_task_done(log);
 263   thread->set_task(NULL);
 264   task->set_code_handle(NULL);
 265   thread->set_env(NULL);
 266   if (task->is_blocking()) {
 267     bool free_task = false;
 268     {
 269       MutexLocker notifier(thread, task->lock());
 270       task->mark_complete();
 271 #if INCLUDE_JVMCI
 272       if (CompileBroker::compiler(task->comp_level())->is_jvmci()) {
 273         if (!task->has_waiter()) {
 274           // The waiting thread timed out and thus did not free the task.
 275           free_task = true;
 276         }
 277         task->set_jvmci_compiler_thread(NULL);
 278       }
 279 #endif
 280       if (!free_task) {
 281         // Notify the waiting thread that the compilation has completed
 282         // so that it can free the task.
 283         task->lock()->notify_all();
 284       }
 285     }
 286     if (free_task) {
 287       // The task can only be freed once the task lock is released.
 288       CompileTask::free(task);
 289     }
 290   } else {
 291     task->mark_complete();
 292 
 293     // By convention, the compiling thread is responsible for
 294     // recycling a non-blocking CompileTask.
 295     CompileTask::free(task);
 296   }
 297 }

 299 /**
 300  * Check if a CompilerThread can be removed and update count if requested.
 301  */
 302 bool CompileBroker::can_remove(CompilerThread *ct, bool do_it) {
 303   assert(UseDynamicNumberOfCompilerThreads, "or shouldn't be here");
 304   if (!ReduceNumberOfCompilerThreads) return false;
 305 
 306   AbstractCompiler *compiler = ct->compiler();
 307   int compiler_count = compiler->num_compiler_threads();
 308   bool c1 = compiler->is_c1();
 309 
 310   // Keep at least 1 compiler thread of each type.
 311   if (compiler_count < 2) return false;
 312 
 313   // Keep thread alive for at least some time.
 314   if (ct->idle_time_millis() < (c1 ? 500 : 100)) return false;
 315 
 316 #if INCLUDE_JVMCI
 317   if (compiler->is_jvmci()) {
 318     // Handles for JVMCI thread objects may get released concurrently.
 319     if (do_it) {
 320       assert(CompileThread_lock->owner() == ct, "must be holding lock");
 321     } else {
 322       // Skip check if it's the last thread and let caller check again.
 323       return true;
 324     }
 325   }
 326 #endif
 327 
 328   // We only allow the last compiler thread of each type to get removed.
 329   jobject last_compiler = c1 ? compiler1_object(compiler_count - 1)
 330                              : compiler2_object(compiler_count - 1);
 331   if (ct->threadObj() == JNIHandles::resolve_non_null(last_compiler)) {
 332     if (do_it) {
 333       assert_locked_or_safepoint(CompileThread_lock); // Update must be consistent.
 334       compiler->set_num_compiler_threads(compiler_count - 1);
 335 #if INCLUDE_JVMCI
 336       if (compiler->is_jvmci()) {
 337         // Old j.l.Thread object can die when no longer referenced elsewhere.
 338         JNIHandles::destroy_global(compiler2_object(compiler_count - 1));
 339         _compiler2_objects[compiler_count - 1] = NULL;
 340       }
 341 #endif
 342     }
 343     return true;
 344   }
 345   return false;
 346 }

 348 /**
 349  * Add a CompileTask to a CompileQueue.
 350  */
 351 void CompileQueue::add(CompileTask* task) {
 352   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 353 
 354   task->set_next(NULL);
 355   task->set_prev(NULL);
 356 
 357   if (_last == NULL) {
 358     // The compile queue is empty.
 359     assert(_first == NULL, "queue is empty");
 360     _first = task;
 361     _last = task;
 362   } else {
 363     // Append the task to the queue.
 364     assert(_last->next() == NULL, "not last");
 365     _last->set_next(task);
 366     task->set_prev(_last);
 367     _last = task;
 368   }
 369   ++_size;
 370 
 371   // Mark the method as being in the compile queue.
 372   task->method()->set_queued_for_compilation();
 373 
 374   if (CIPrintCompileQueue) {
 375     print_tty();
 376   }
 377 
 378   if (LogCompilation && xtty != NULL) {
 379     task->log_task_queued();
 380   }
 381 
 382   // Notify CompilerThreads that a task is available.
 383   MethodCompileQueue_lock->notify_all();
 384 }

 386 /**
 387  * Empties compilation queue by putting all compilation tasks onto
 388  * a freelist. Furthermore, the method wakes up all threads that are
 389  * waiting on a compilation task to finish. This can happen if background
 390  * compilation is disabled.
 391  */
 392 void CompileQueue::free_all() {
 393   MutexLocker mu(MethodCompileQueue_lock);
 394   CompileTask* next = _first;
 395 
 396   // Iterate over all tasks in the compile queue
 397   while (next != NULL) {
 398     CompileTask* current = next;
 399     next = current->next();
 400     {
 401       // Wake up thread that blocks on the compile task.
 402       MutexLocker ct_lock(current->lock());
 403       current->lock()->notify();
 404     }
 405     // Put the task back on the freelist.
 406     CompileTask::free(current);
 407   }
 408   _first = NULL;
 409 
 410   // Wake up all threads that block on the queue.
 411   MethodCompileQueue_lock->notify_all();
 412 }

 414 /**
 415  * Get the next CompileTask from a CompileQueue
 416  */
 417 CompileTask* CompileQueue::get() {
 418   // save methods from RedefineClasses across safepoint
 419   // across MethodCompileQueue_lock below.
 420   methodHandle save_method;
 421   methodHandle save_hot_method;
 422 
 423   MonitorLocker locker(MethodCompileQueue_lock);
 424   // If _first is NULL we have no more compile jobs. There are two reasons for
 425   // having no compile jobs: First, we compiled everything we wanted. Second,
 426   // we ran out of code cache so compilation has been disabled. In the latter
 427   // case we perform code cache sweeps to free memory such that we can re-enable
 428   // compilation.
 429   while (_first == NULL) {
 430     // Exit loop if compilation is disabled forever
 431     if (CompileBroker::is_compilation_disabled_forever()) {
 432       return NULL;
 433     }
 434 
 435     // If there are no compilation tasks and we can compile new jobs
 436     // (i.e., there is enough free space in the code cache) there is
 437     // no need to invoke the sweeper. As a result, the hotness of methods
 438     // remains unchanged. This behavior is desired, since we want to keep
 439     // the stable state, i.e., we do not want to evict methods from the
 440     // code cache if it is unnecessary.
 441     // We need a timed wait here, since compiler threads can exit if compilation
 442     // is disabled forever. We use 5 seconds wait time; the exiting of compiler threads
 443     // is not critical and we do not want idle compiler threads to wake up too often.
 444     locker.wait(5*1000);
 445 
 446     if (UseDynamicNumberOfCompilerThreads && _first == NULL) {
 447       // Still nothing to compile. Give caller a chance to stop this thread.
 448       if (CompileBroker::can_remove(CompilerThread::current(), false)) return NULL;
 449     }
 450   }
 451 
 452   if (CompileBroker::is_compilation_disabled_forever()) {
 453     return NULL;
 454   }
 455 
 456   CompileTask* task;
 457   {
 458     NoSafepointVerifier nsv;
 459     task = CompilationPolicy::policy()->select_task(this);
 460     if (task != NULL) {
 461       task = task->select_for_compilation();
 462     }
 463   }
 464 
 465   if (task != NULL) {
 466     // Save method pointers across unlock safepoint.  The task is removed from
 467     // the compilation queue, which is walked during RedefineClasses.
 468     Thread* thread = Thread::current();
 469     save_method = methodHandle(thread, task->method());
 470     save_hot_method = methodHandle(thread, task->hot_method());
 471 
 472     remove(task);
 473   }
 474   purge_stale_tasks(); // may temporarily release MCQ lock
 475   return task;
 476 }

 478 // Clean & deallocate stale compile tasks.
 479 // Temporarily releases MethodCompileQueue lock.
 480 void CompileQueue::purge_stale_tasks() {
 481   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 482   if (_first_stale != NULL) {
 483     // Stale tasks are purged when MCQ lock is released,
 484     // but _first_stale updates are protected by MCQ lock.
 485     // Once task processing starts and MCQ lock is released,
 486     // other compiler threads can reuse _first_stale.
 487     CompileTask* head = _first_stale;
 488     _first_stale = NULL;
 489     {
 490       MutexUnlocker ul(MethodCompileQueue_lock);
 491       for (CompileTask* task = head; task != NULL; ) {
 492         CompileTask* next_task = task->next();
 493         CompileTaskWrapper ctw(task); // Frees the task
 494         task->set_failure_reason("stale task");
 495         task = next_task;
 496       }
 497     }
 498   }
 499 }

 501 void CompileQueue::remove(CompileTask* task) {
 502   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 503   if (task->prev() != NULL) {
 504     task->prev()->set_next(task->next());
 505   } else {
 506     // max is the first element
 507     assert(task == _first, "Sanity");
 508     _first = task->next();
 509   }
 510 
 511   if (task->next() != NULL) {
 512     task->next()->set_prev(task->prev());
 513   } else {
 514     // max is the last element
 515     assert(task == _last, "Sanity");
 516     _last = task->prev();
 517   }
 518   --_size;
 519 }
 520 
 521 void CompileQueue::remove_and_mark_stale(CompileTask* task) {
 522   assert(MethodCompileQueue_lock->owned_by_self(), "must own lock");
 523   remove(task);
 524 
 525   // Enqueue the task for reclamation (should be done outside MCQ lock)
 526   task->set_next(_first_stale);
 527   task->set_prev(NULL);
 528   _first_stale = task;
 529 }

 531 // methods in the compile queue need to be marked as used on the stack

 564 void CompileQueue::print(outputStream* st) {
 565   assert_locked_or_safepoint(MethodCompileQueue_lock);
 566   st->print_cr("%s:", name());
 567   CompileTask* task = _first;
 568   if (task == NULL) {
 569     st->print_cr("Empty");
 570   } else {
 571     while (task != NULL) {
 572       task->print(st, NULL, true, true);
 573       task = task->next();
 574     }
 575   }
 576   st->cr();
 577 }

 806 JavaThread* CompileBroker::make_thread(jobject thread_handle, CompileQueue* queue, AbstractCompiler* comp, Thread* THREAD) {
 807   JavaThread* new_thread = NULL;
 808   {
 809     MutexLocker mu(THREAD, Threads_lock);
 810     if (comp != NULL) {
 811       if (!InjectCompilerCreationFailure || comp->num_compiler_threads() == 0) {
 812         CompilerCounters* counters = new CompilerCounters();
 813         new_thread = new CompilerThread(queue, counters);
 814       }
 815     } else {
 816       new_thread = new CodeCacheSweeperThread();
 817     }
 818     // At this point the new CompilerThread data-races with this startup
 819     // thread (which I believe is the primoridal thread and NOT the VM
 820     // thread).  This means Java bytecodes being executed at startup can
 821     // queue compile jobs which will run at whatever default priority the
 822     // newly created CompilerThread runs at.
 823 
 824 
 825     // At this point it may be possible that no osthread was created for the
 826     // JavaThread due to lack of memory. We would have to throw an exception
 827     // in that case. However, since this must work and we do not allow
 828     // exceptions anyway, check and abort if this fails. But first release the
 829     // lock.
 830 
 831     if (new_thread != NULL && new_thread->osthread() != NULL) {
 832 
 833       java_lang_Thread::set_thread(JNIHandles::resolve_non_null(thread_handle), new_thread);
 834 
 835       // Note that this only sets the JavaThread _priority field, which by
 836       // definition is limited to Java priorities and not OS priorities.
 837       // The os-priority is set in the CompilerThread startup code itself
 838 
 839       java_lang_Thread::set_priority(JNIHandles::resolve_non_null(thread_handle), NearMaxPriority);
 840 
 841       // Note that we cannot call os::set_priority because it expects Java
 842       // priorities and we are *explicitly* using OS priorities so that it's
 843       // possible to set the compiler thread priority higher than any Java
 844       // thread.
 845 
 846       int native_prio = CompilerThreadPriority;
 847       if (native_prio == -1) {
 848         if (UseCriticalCompilerThreadPriority) {
 849           native_prio = os::java_to_os_priority[CriticalPriority];
 850         } else {
 851           native_prio = os::java_to_os_priority[NearMaxPriority];
 852         }
 853       }
 854       os::set_native_priority(new_thread, native_prio);
 855 
 856       java_lang_Thread::set_daemon(JNIHandles::resolve_non_null(thread_handle));
 857 
 858       new_thread->set_threadObj(JNIHandles::resolve_non_null(thread_handle));
 859       if (comp != NULL) {
 860         new_thread->as_CompilerThread()->set_compiler(comp);
 861       }
 862       Threads::add(new_thread);
 863       Thread::start(new_thread);
 864     }
 865   }
 866 
 867   // First release lock before aborting VM.
 868   if (new_thread == NULL || new_thread->osthread() == NULL) {
 869     if (UseDynamicNumberOfCompilerThreads && comp != NULL && comp->num_compiler_threads() > 0) {
 870       if (new_thread != NULL) {
 871         new_thread->smr_delete();
 872       }
 873       return NULL;
 874     }
 875     vm_exit_during_initialization("java.lang.OutOfMemoryError",
 876                                   os::native_thread_creation_failed_msg());
 877   }
 878 
 879   // Let go of Threads_lock before yielding
 880   os::naked_yield(); // make sure that the compiler thread is started early (especially helpful on SOLARIS)
 881 
 882   return new_thread;
 883 }

 886 void CompileBroker::init_compiler_sweeper_threads() {
 887   // Ensure any exceptions lead to vm_exit_during_initialization.
 888   EXCEPTION_MARK;
 889 #if !defined(ZERO)
 890   assert(_c2_count > 0 || _c1_count > 0, "No compilers?");
 891 #endif // !ZERO
 892   // Initialize the compilation queue
 893   if (_c2_count > 0) {
 894     const char* name = JVMCI_ONLY(UseJVMCICompiler ? "JVMCI compile queue" :) "C2 compile queue";
 895     _c2_compile_queue  = new CompileQueue(name);
 896     _compiler2_objects = NEW_C_HEAP_ARRAY(jobject, _c2_count, mtCompiler);
 897     _compiler2_logs = NEW_C_HEAP_ARRAY(CompileLog*, _c2_count, mtCompiler);
 898   }
 899   if (_c1_count > 0) {
 900     _c1_compile_queue  = new CompileQueue("C1 compile queue");
 901     _compiler1_objects = NEW_C_HEAP_ARRAY(jobject, _c1_count, mtCompiler);
 902     _compiler1_logs = NEW_C_HEAP_ARRAY(CompileLog*, _c1_count, mtCompiler);
 903   }
 904 
 905   char name_buffer[256];
 906 
 907   for (int i = 0; i < _c2_count; i++) {
 908     jobject thread_handle = NULL;
 909     // Create all j.l.Thread objects for C1 and C2 threads here, but only one
 910     // for JVMCI compiler which can create further ones on demand.
 911     JVMCI_ONLY(if (!UseJVMCICompiler || !UseDynamicNumberOfCompilerThreads || i == 0) {)
 912     // Create a name for our thread.
 913     sprintf(name_buffer, "%s CompilerThread%d", _compilers[1]->name(), i);
 914     Handle thread_oop = create_thread_oop(name_buffer, CHECK);
 915     thread_handle = JNIHandles::make_global(thread_oop);
 916     JVMCI_ONLY(})
 917     _compiler2_objects[i] = thread_handle;
 918     _compiler2_logs[i] = NULL;
 919 
 920     if (!UseDynamicNumberOfCompilerThreads || i == 0) {
 921       JavaThread *ct = make_thread(thread_handle, _c2_compile_queue, _compilers[1], THREAD);
 922       assert(ct != NULL, "should have been handled for initial thread");
 923       _compilers[1]->set_num_compiler_threads(i + 1);
 924       if (TraceCompilerThreads) {
 925         ResourceMark rm;
 926         MutexLocker mu(Threads_lock);
 927         tty->print_cr("Added initial compiler thread %s", ct->get_thread_name());
 928       }
 929     }
 930   }
 931 
 932   for (int i = 0; i < _c1_count; i++) {
 933     // Create a name for our thread.
 934     sprintf(name_buffer, "C1 CompilerThread%d", i);
 935     Handle thread_oop = create_thread_oop(name_buffer, CHECK);
 936     jobject thread_handle = JNIHandles::make_global(thread_oop);
 937     _compiler1_objects[i] = thread_handle;
 938     _compiler1_logs[i] = NULL;
 939 
 940     if (!UseDynamicNumberOfCompilerThreads || i == 0) {
 941       JavaThread *ct = make_thread(thread_handle, _c1_compile_queue, _compilers[0], THREAD);
 942       assert(ct != NULL, "should have been handled for initial thread");
 943       _compilers[0]->set_num_compiler_threads(i + 1);
 944       if (TraceCompilerThreads) {
 945         ResourceMark rm;
 946         MutexLocker mu(Threads_lock);
 947         tty->print_cr("Added initial compiler thread %s", ct->get_thread_name());
 948       }
 949     }
 950   }
 951 
 952   if (UsePerfData) {
 953     PerfDataManager::create_constant(SUN_CI, "threads", PerfData::U_Bytes, _c1_count + _c2_count, CHECK);
 954   }
 955 
 956   if (MethodFlushing) {
 957     // Initialize the sweeper thread
 958     Handle thread_oop = create_thread_oop("Sweeper thread", CHECK);
 959     jobject thread_handle = JNIHandles::make_local(THREAD, thread_oop());
 960     make_thread(thread_handle, NULL, NULL, THREAD);
 961   }
 962 }

 964 void CompileBroker::possibly_add_compiler_threads(Thread* THREAD) {
 965 
 966   julong available_memory = os::available_memory();
 967   // If SegmentedCodeCache is off, both values refer to the single heap (with type CodeBlobType::All).
 968   size_t available_cc_np  = CodeCache::unallocated_capacity(CodeBlobType::MethodNonProfiled),
 969          available_cc_p   = CodeCache::unallocated_capacity(CodeBlobType::MethodProfiled);
 970 
 971   // Only do attempt to start additional threads if the lock is free.
 972   if (!CompileThread_lock->try_lock()) return;
 973 
 974   if (_c2_compile_queue != NULL) {
 975     int old_c2_count = _compilers[1]->num_compiler_threads();
 976     int new_c2_count = MIN4(_c2_count,
 977         _c2_compile_queue->size() / 2,
 978         (int)(available_memory / (200*M)),
 979         (int)(available_cc_np / (128*K)));
 980 
 981     for (int i = old_c2_count; i < new_c2_count; i++) {
 982 #if INCLUDE_JVMCI
 983       if (UseJVMCICompiler) {
 984         // Native compiler threads as used in C1/C2 can reuse the j.l.Thread
 985         // objects as their existence is completely hidden from the rest of
 986         // the VM (and those compiler threads can't call Java code to do the
 987         // creation anyway). For JVMCI we have to create new j.l.Thread objects
 988         // as they are visible and we can see unexpected thread lifecycle
 989         // transitions if we bind them to new JavaThreads.
 990         if (!THREAD->can_call_java()) break;
 991         char name_buffer[256];
 992         sprintf(name_buffer, "%s CompilerThread%d", _compilers[1]->name(), i);
 993         Handle thread_oop;
 994         {
 995           // We have to give up the lock temporarily for the Java calls.
 996           MutexUnlocker mu(CompileThread_lock);
 997           thread_oop = create_thread_oop(name_buffer, THREAD);
 998         }
 999         if (HAS_PENDING_EXCEPTION) {
1000           if (TraceCompilerThreads) {
1001             ResourceMark rm;
1002             tty->print_cr("JVMCI compiler thread creation failed:");
1003             PENDING_EXCEPTION->print();
1004           }
1005           CLEAR_PENDING_EXCEPTION;
1006           break;
1007         }
1008         // Check if another thread has beaten us during the Java calls.
1009         if (_compilers[1]->num_compiler_threads() != i) break;
1010         jobject thread_handle = JNIHandles::make_global(thread_oop);
1011         assert(compiler2_object(i) == NULL, "Old one must be released!");
1012         _compiler2_objects[i] = thread_handle;
1013       }
1014 #endif
1015       JavaThread *ct = make_thread(compiler2_object(i), _c2_compile_queue, _compilers[1], THREAD);
1016       if (ct == NULL) break;
1017       _compilers[1]->set_num_compiler_threads(i + 1);
1018       if (TraceCompilerThreads) {
1019         ResourceMark rm;
1020         MutexLocker mu(Threads_lock);
1021         tty->print_cr("Added compiler thread %s (available memory: %dMB, available non-profiled code cache: %dMB)",
1022                       ct->get_thread_name(), (int)(available_memory/M), (int)(available_cc_np/M));
1023       }
1024     }
1025   }
1026 
1027   if (_c1_compile_queue != NULL) {
1028     int old_c1_count = _compilers[0]->num_compiler_threads();
1029     int new_c1_count = MIN4(_c1_count,
1030         _c1_compile_queue->size() / 4,
1031         (int)(available_memory / (100*M)),
1032         (int)(available_cc_p / (128*K)));
1033 
1034     for (int i = old_c1_count; i < new_c1_count; i++) {
1035       JavaThread *ct = make_thread(compiler1_object(i), _c1_compile_queue, _compilers[0], THREAD);
1036       if (ct == NULL) break;
1037       _compilers[0]->set_num_compiler_threads(i + 1);
1038       if (TraceCompilerThreads) {
1039         ResourceMark rm;
1040         MutexLocker mu(Threads_lock);
1041         tty->print_cr("Added compiler thread %s (available memory: %dMB, available profiled code cache: %dMB)",
1042                       ct->get_thread_name(), (int)(available_memory/M), (int)(available_cc_p/M));
1043       }
1044     }
1045   }
1046 
1047   CompileThread_lock->unlock();
1048 }

1067 // ------------------------------------------------------------------
1068 // CompileBroker::compile_method
1069 //
1070 // Request compilation of a method.
1071 void CompileBroker::compile_method_base(const methodHandle& method,
1072                                         int osr_bci,
1073                                         int comp_level,
1074                                         const methodHandle& hot_method,
1075                                         int hot_count,
1076                                         CompileTask::CompileReason compile_reason,
1077                                         bool blocking,
1078                                         Thread* thread) {
1079   guarantee(!method->is_abstract(), "cannot compile abstract methods");
1080   assert(method->method_holder()->is_instance_klass(),
1081          "sanity check");
1082   assert(!method->method_holder()->is_not_initialized(),
1083          "method holder must be initialized");
1084   assert(!method->is_method_handle_intrinsic(), "do not enqueue these guys");
1085 
1086   if (CIPrintRequests) {
1087     tty->print("request: ");
1088     method->print_short_name(tty);
1089     if (osr_bci != InvocationEntryBci) {
1090       tty->print(" osr_bci: %d", osr_bci);
1091     }
1092     tty->print(" level: %d comment: %s count: %d", comp_level, CompileTask::reason_name(compile_reason), hot_count);
1093     if (!hot_method.is_null()) {
1094       tty->print(" hot: ");
1095       if (hot_method() != method()) {
1096           hot_method->print_short_name(tty);
1097       } else {
1098         tty->print("yes");
1099       }
1100     }
1101     tty->cr();
1102   }
1103 
1104   // A request has been made for compilation.  Before we do any
1105   // real work, check to see if the method has been compiled
1106   // in the meantime with a definitive result.
1107   if (compilation_is_complete(method, osr_bci, comp_level)) {
1108     return;
1109   }
1110 
1111 #ifndef PRODUCT
1112   if (osr_bci != -1 && !FLAG_IS_DEFAULT(OSROnlyBCI)) {
1113     if ((OSROnlyBCI > 0) ? (OSROnlyBCI != osr_bci) : (-OSROnlyBCI == osr_bci)) {
1114       // Positive OSROnlyBCI means only compile that bci.  Negative means don't compile that BCI.
1115       return;
1116     }
1117   }
1118 #endif
1119 
1120   // If this method is already in the compile queue, then
1121   // we do not block the current thread.
1122   if (compilation_is_in_queue(method)) {
1123     // We may want to decay our counter a bit here to prevent
1124     // multiple denied requests for compilation.  This is an
1125     // open compilation policy issue. Note: The other possibility,
1126     // in the case that this is a blocking compile request, is to have
1127     // all subsequent blocking requesters wait for completion of
1128     // ongoing compiles. Note that in this case we'll need a protocol
1129     // for freeing the associated compile tasks. [Or we could have
1130     // a single static monitor on which all these waiters sleep.]
1131     return;
1132   }
1133 
1134   if (TieredCompilation) {
1135     // Tiered policy requires MethodCounters to exist before adding a method to
1136     // the queue. Create if we don't have them yet.
1137     method->get_method_counters(thread);
1138   }
1139 
1140   // Outputs from the following MutexLocker block:
1141   CompileTask* task     = NULL;
1142   CompileQueue* queue  = compile_queue(comp_level);
1143 
1144   // Acquire our lock.
1145   {
1146     MutexLocker locker(thread, MethodCompileQueue_lock);
1147 
1148     // Make sure the method has not slipped into the queues since
1149     // last we checked; note that those checks were "fast bail-outs".
1150     // Here we need to be more careful, see 14012000 below.
1151     if (compilation_is_in_queue(method)) {
1152       return;
1153     }
1154 
1155     // We need to check again to see if the compilation has
1156     // completed.  A previous compilation may have registered
1157     // some result.
1158     if (compilation_is_complete(method, osr_bci, comp_level)) {
1159       return;
1160     }
1161 
1162     // We now know that this compilation is not pending, complete,
1163     // or prohibited.  Assign a compile_id to this compilation
1164     // and check to see if it is in our [Start..Stop) range.
1165     int compile_id = assign_compile_id(method, osr_bci);
1166     if (compile_id == 0) {
1167       // The compilation falls outside the allowed range.
1168       return;
1169     }
1170 
1171 #if INCLUDE_JVMCI
1172     if (UseJVMCICompiler && blocking) {
1173       // Don't allow blocking compiles for requests triggered by JVMCI.
1174       if (thread->is_Compiler_thread()) {
1175         blocking = false;
1176       }
1177 
1178       if (!UseJVMCINativeLibrary) {
1179         // Don't allow blocking compiles if inside a class initializer or while performing class loading
1180         vframeStream vfst((JavaThread*) thread);
1181         for (; !vfst.at_end(); vfst.next()) {
1182         if (vfst.method()->is_class_initializer() ||
1183               (vfst.method()->method_holder()->is_subclass_of(SystemDictionary::ClassLoader_klass()) &&
1184                   vfst.method()->name() == vmSymbols::loadClass_name())) {
1185             blocking = false;
1186             break;
1187           }
1188         }
1189       }
1190 
1191       // Don't allow blocking compilation requests to JVMCI
1192       // if JVMCI itself is not yet initialized
1193       if (!JVMCI::is_compiler_initialized() && compiler(comp_level)->is_jvmci()) {
1194         blocking = false;
1195       }
1196 
1197       // Don't allow blocking compilation requests if we are in JVMCIRuntime::shutdown
1198       // to avoid deadlock between compiler thread(s) and threads run at shutdown
1199       // such as the DestroyJavaVM thread.
1200       if (JVMCI::shutdown_called()) {
1201         blocking = false;
1202       }
1203     }
1204 #endif // INCLUDE_JVMCI
1205 
1206     // We will enter the compilation in the queue.
1207     // 14012000: Note that this sets the queued_for_compile bits in
1208     // the target method. We can now reason that a method cannot be
1209     // queued for compilation more than once, as follows:
1210     // Before a thread queues a task for compilation, it first acquires
1211     // the compile queue lock, then checks if the method's queued bits
1212     // are set or it has already been compiled. Thus there can not be two
1213     // instances of a compilation task for the same method on the
1214     // compilation queue. Consider now the case where the compilation
1215     // thread has already removed a task for that method from the queue
1216     // and is in the midst of compiling it. In this case, the
1217     // queued_for_compile bits must be set in the method (and these
1218     // will be visible to the current thread, since the bits were set
1219     // under protection of the compile queue lock, which we hold now.
1220     // When the compilation completes, the compiler thread first sets
1221     // the compilation result and then clears the queued_for_compile
1222     // bits. Neither of these actions are protected by a barrier (or done
1223     // under the protection of a lock), so the only guarantee we have
1224     // (on machines with TSO (Total Store Order)) is that these values
1225     // will update in that order. As a result, the only combinations of
1226     // these bits that the current thread will see are, in temporal order:
1227     // <RESULT, QUEUE> :
1228     //     <0, 1> : in compile queue, but not yet compiled
1229     //     <1, 1> : compiled but queue bit not cleared
1230     //     <1, 0> : compiled and queue bit cleared
1231     // Because we first check the queue bits then check the result bits,
1232     // we are assured that we cannot introduce a duplicate task.
1233     // Note that if we did the tests in the reverse order (i.e. check
1234     // result then check queued bit), we could get the result bit before
1235     // the compilation completed, and the queue bit after the compilation
1236     // completed, and end up introducing a "duplicate" (redundant) task.
1237     // In that case, the compiler thread should first check if a method
1238     // has already been compiled before trying to compile it.
1239     // NOTE: in the event that there are multiple compiler threads and
1240     // there is de-optimization/recompilation, things will get hairy,
1241     // and in that case it's best to protect both the testing (here) of
1242     // these bits, and their updating (here and elsewhere) under a
1243     // common lock.
1244     task = create_compile_task(queue,
1245                                compile_id, method,
1246                                osr_bci, comp_level,
1247                                hot_method, hot_count, compile_reason,
1248                                blocking);
1249   }
1250 
1251   if (blocking) {
1252     wait_for_completion(task);
1253   }
1254 }

1404     }

1411     }

1422     } else {

1425   }

1580 // The number of milliseconds to wait before checking if
1581 // JVMCI compilation has made progress.

1588 /**
1589  * Waits for a JVMCI compiler to complete a given task. This thread
1590  * waits until either the task completes or it sees no JVMCI compilation
1591  * progress for N consecutive milliseconds where N is
1592  * JVMCI_COMPILATION_PROGRESS_WAIT_TIMESLICE *
1593  * JVMCI_COMPILATION_PROGRESS_WAIT_ATTEMPTS.
1594  *
1595  * @return true if this thread needs to free/recycle the task
1596  */
1597 bool CompileBroker::wait_for_jvmci_completion(JVMCICompiler* jvmci, CompileTask* task, JavaThread* thread) {
1598   MonitorLocker ml(thread, task->lock());
1599   int progress_wait_attempts = 0;
1600   int methods_compiled = jvmci->methods_compiled();
1601   while (!task->is_complete() && !is_compilation_disabled_forever() &&
1602          ml.wait(JVMCI_COMPILATION_PROGRESS_WAIT_TIMESLICE)) {
1603     CompilerThread* jvmci_compiler_thread = task->jvmci_compiler_thread();
1604 
1605     bool progress;
1606     if (jvmci_compiler_thread != NULL) {
1607       // If the JVMCI compiler thread is not blocked or suspended, we deem it to be making progress.
1608       progress = jvmci_compiler_thread->thread_state() != _thread_blocked &&
1609         !jvmci_compiler_thread->is_external_suspend();
1610     } else {
1611       // Still waiting on JVMCI compiler queue. This thread may be holding a lock
1612       // that all JVMCI compiler threads are blocked on. We use the counter for
1613       // successful JVMCI compilations to determine whether JVMCI compilation
1614       // is still making progress through the JVMCI compiler queue.
1615       progress = jvmci->methods_compiled() != methods_compiled;
1616     }
1617 
1618     if (!progress) {
1619       if (++progress_wait_attempts == JVMCI_COMPILATION_PROGRESS_WAIT_ATTEMPTS) {
1620         if (PrintCompilation) {
1621           task->print(tty, "wait for blocking compilation timed out");
1622         }
1623         break;
1624       }
1625     } else {
1626       progress_wait_attempts = 0;
1627       if (jvmci_compiler_thread == NULL) {
1628         methods_compiled = jvmci->methods_compiled();
1629       }
1630     }
1631   }
1632   task->clear_waiter();
1633   return task->is_complete();
1634 }

1637 /**
1638  *  Wait for the compilation task to complete.
1639  */
1640 void CompileBroker::wait_for_completion(CompileTask* task) {
1641   if (CIPrintCompileQueue) {
1642     ttyLocker ttyl;
1643     tty->print_cr("BLOCKING FOR COMPILE");
1644   }
1645 
1646   assert(task->is_blocking(), "can only wait on blocking task");
1647 
1648   JavaThread* thread = JavaThread::current();
1649 
1650   methodHandle method(thread, task->method());
1651   bool free_task;
1652 #if INCLUDE_JVMCI
1653   AbstractCompiler* comp = compiler(task->comp_level());
1654   if (comp->is_jvmci()) {
1655     free_task = wait_for_jvmci_completion((JVMCICompiler*) comp, task, thread);
1656   } else
1657 #endif
1658   {
1659     MonitorLocker ml(thread, task->lock());
1660     free_task = true;
1661     while (!task->is_complete() && !is_compilation_disabled_forever()) {
1662       ml.wait();
1663     }
1664   }
1665 
1666   if (free_task) {
1667     if (is_compilation_disabled_forever()) {
1668       CompileTask::free(task);
1669       return;
1670     }
1671 
1672     // It is harmless to check this status without the lock, because
1673     // completion is a stable property (until the task object is recycled).
1674     assert(task->is_complete(), "Compilation should have completed");
1675     assert(task->code_handle() == NULL, "must be reset");
1676 
1677     // By convention, the waiter is responsible for recycling a
1678     // blocking CompileTask. Since there is only one waiter ever
1679     // waiting on a CompileTask, we know that no one else will
1680     // be using this CompileTask; we can free it.
1681     CompileTask::free(task);
1682   }

1686  * Initialize compiler thread(s) + compiler object(s). The postcondition
1687  * of this function is that the compiler runtimes are initialized and that
1688  * compiler threads can start compiling.
1689  */
1690 bool CompileBroker::init_compiler_runtime() {
1691   CompilerThread* thread = CompilerThread::current();
1692   AbstractCompiler* comp = thread->compiler();
1693   // Final sanity check - the compiler object must exist
1694   guarantee(comp != NULL, "Compiler object must exist");

1697     // Must switch to native to allocate ci_env
1698     ThreadToNativeFromVM ttn(thread);

1701     ci_env.cache_jvmti_state();
1702     // Cache DTrace flags

1706     ThreadInVMfromNative tv(thread);
1707     ResetNoHandleMark rnhm;

1710     comp->initialize();
1711   }

1716     // particular compiler runtime will ever start to compile methods.
1717     shutdown_compiler_runtime(comp, thread);
1718     return false;
1719   }
1720

1722   if (comp->is_c1() && (thread->get_buffer_blob() == NULL)) {
1723     warning("Initialization of %s thread failed (no space to run compilers)", thread->name());
1724     return false;
1725   }
1726 
1727   return true;

1730 /**
1731  * If C1 and/or C2 initialization failed, we shut down all compilation.
1732  * We do this to keep things simple. This can be changed if it ever turns
1733  * out to be a problem.
1734  */
1735 void CompileBroker::shutdown_compiler_runtime(AbstractCompiler* comp, CompilerThread* thread) {
1736   // Free buffer blob, if allocated
1737   if (thread->get_buffer_blob() != NULL) {
1738     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1739     CodeCache::free(thread->get_buffer_blob());
1740   }
1741 
1742   if (comp->should_perform_shutdown()) {
1743     // There are two reasons for shutting down the compiler
1744     // 1) compiler runtime initialization failed
1745     // 2) The code cache is full and the following flag is set: -XX:-UseCodeCacheFlushing
1746     warning("%s initialization failed. Shutting down all compilers", comp->name());
1747 
1748     // Only one thread per compiler runtime object enters here
1749     // Set state to shut down

1771 /**
1772  * Helper function to create new or reuse old CompileLog.
1773  */
1774 CompileLog* CompileBroker::get_log(CompilerThread* ct) {
1775   if (!LogCompilation) return NULL;
1776 
1777   AbstractCompiler *compiler = ct->compiler();
1778   bool c1 = compiler->is_c1();
1779   jobject* compiler_objects = c1 ? _compiler1_objects : _compiler2_objects;

1781   CompileLog** logs = c1 ? _compiler1_logs : _compiler2_logs;
1782   assert(logs != NULL, "must be initialized at this point");

1785   // Find Compiler number by its threadObj.
1786   oop compiler_obj = ct->threadObj();

1788   bool found = false;
1789   for (; compiler_number < count; compiler_number++) {
1790     if (JNIHandles::resolve_non_null(compiler_objects[compiler_number]) == compiler_obj) {
1791       found = true;
1792       break;
1793     }
1794   }
1795   assert(found, "Compiler must exist at this point");
1796 
1797   // Determine pointer for this thread's log.
1798   CompileLog** log_ptr = &logs[compiler_number];
1799 
1800   // Return old one if it exists.
1801   CompileLog* log = *log_ptr;
1802   if (log != NULL) {
1803     ct->init_log(log);
1804     return log;
1805   }
1806 
1807   // Create a new one and remember it.
1808   init_compiler_thread_log();

1810   *log_ptr = log;
1811   return log;
1812 }

1814 // ------------------------------------------------------------------
1815 // CompileBroker::compiler_thread_loop
1816 //
1817 // The main loop run by a CompilerThread.
1818 void CompileBroker::compiler_thread_loop() {
1819   CompilerThread* thread = CompilerThread::current();
1820   CompileQueue* queue = thread->queue();
1821   // For the thread that initializes the ciObjectFactory
1822   // this resource mark holds all the shared objects
1823   ResourceMark rm;
1824 
1825   // First thread to get here will initialize the compiler interface
1826 
1827   {
1828     ASSERT_IN_VM;
1829     MutexLocker only_one (thread, CompileThread_lock);
1830     if (!ciObjectFactory::is_initialized()) {
1831       ciObjectFactory::initialize();
1832     }
1833   }
1834 
1835   // Open a log.
1836   CompileLog* log = get_log(thread);
1837   if (log != NULL) {
1838     log->begin_elem("start_compile_thread name='%s' thread='" UINTX_FORMAT "' process='%d'",
1839                     thread->name(),
1840                     os::current_thread_id(),
1841                     os::current_process_id());
1842     log->stamp();
1843     log->end_elem();
1844   }
1845 
1846   // If compiler thread/runtime initialization fails, exit the compiler thread
1847   if (!init_compiler_runtime()) {
1848     return;
1849   }
1850 
1851   thread->start_idle_timer();
1852 
1853   // Poll for new compilation tasks as long as the JVM runs. Compilation
1854   // should only be disabled if something went wrong while initializing the
1855   // compiler runtimes. This, in turn, should not happen. The only known case
1856   // when compiler runtime initialization fails is if there is not enough free
1857   // space in the code cache to generate the necessary stubs, etc.
1858   while (!is_compilation_disabled_forever()) {
1859     // We need this HandleMark to avoid leaking VM handles.
1860     HandleMark hm(thread);
1861 
1862     CompileTask* task = queue->get();
1863     if (task == NULL) {
1864       if (UseDynamicNumberOfCompilerThreads) {
1865         // Access compiler_count under lock to enforce consistency.
1866         MutexLocker only_one(CompileThread_lock);
1867         if (can_remove(thread, true)) {
1868           if (TraceCompilerThreads) {
1869             tty->print_cr("Removing compiler thread %s after " JLONG_FORMAT " ms idle time",
1870                           thread->name(), thread->idle_time_millis());
1871           }
1872           // Free buffer blob, if allocated
1873           if (thread->get_buffer_blob() != NULL) {
1874             MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1875             CodeCache::free(thread->get_buffer_blob());
1876           }
1877           return; // Stop this thread.
1878         }
1879       }
1880     } else {
1881       // Assign the task to the current thread.  Mark this compilation
1882       // thread as active for the profiler.
1883       // CompileTaskWrapper also keeps the Method* from being deallocated if redefinition
1884       // occurs after fetching the compile task off the queue.
1885       CompileTaskWrapper ctw(task);
1886       nmethodLocker result_handle;  // (handle for the nmethod produced by this task)
1887       task->set_code_handle(&result_handle);
1888       methodHandle method(thread, task->method());
1889 
1890       // Never compile a method if breakpoints are present in it
1891       if (method()->number_of_breakpoints() == 0) {
1892         // Compile the method.
1893         if ((UseCompiler || AlwaysCompileLoopMethods) && CompileBroker::should_compile_new_jobs()) {
1894           invoke_compiler_on_method(task);
1895           thread->start_idle_timer();
1896         } else {
1897           // After compilation is disabled, remove remaining methods from queue
1898           method->clear_queued_for_compilation();
1899           task->set_failure_reason("compilation is disabled");
1900         }
1901       }
1902 
1903       if (UseDynamicNumberOfCompilerThreads) {
1904         possibly_add_compiler_threads(thread);
1905         assert(!thread->has_pending_exception(), "should have been handled");
1906       }
1907     }
1908   }
1909 
1910   // Shut down compiler runtime
1911   shutdown_compiler_runtime(thread->compiler(), thread);
1912 }

2364   size_t s2len = strlen(method_name);
2365 
2366   // check if we need to truncate the string
2367   if (s1len + s2len + 2 > maxLen) {
2368 
2369     // the strategy is to lop off the leading characters of the
2370     // class name and the trailing characters of the method name.
2371 
2372     if (s2len + 2 > maxLen) {
2373       // lop of the entire class name string, let snprintf handle
2374       // truncation of the method name.
2375       class_name += s1len; // null string
2376     }
2377     else {
2378       // lop off the extra characters from the front of the class name
2379       class_name += ((s1len + s2len + 2) - maxLen);
2380     }
2381   }
2382 
2383   jio_snprintf(current_method, maxLen, "%s %s", class_name, method_name);
2384 
2385   int last_compile_type = normal_compile;
2386   if (CICountOSR && is_osr) {
2387     last_compile_type = osr_compile;
2388   }
2389 
2390   CompilerCounters* counters = thread->counters();
2391   counters->set_current_method(current_method);
2392   counters->set_compile_type((jlong) last_compile_type);
2393 }

2432 void CompileBroker::collect_statistics(CompilerThread* thread, elapsedTimer time, CompileTask* task) {
2433   bool success = task->is_success();

2435   uint compile_id = task->compile_id();
2436   bool is_osr = (task->osr_bci() != standard_entry_bci);
2437   nmethod* code = task->code();
2438   CompilerCounters* counters = thread->counters();
2439 
2440   assert(code == NULL || code->is_locked_by_vm(), "will survive the MutexLocker");
2441   MutexLocker locker(CompileStatistics_lock);
2442 
2443   // _perf variables are production performance counters which are
2444   // updated regardless of the setting of the CITime and CITimeEach flags
2445   //
2446 
2447   // account all time, including bailouts and failures in this counter;
2448   // C1 and C2 counters are counting both successful and unsuccessful compiles
2449   _t_total_compilation.add(time);
2450 
2451   if (!success) {
2452     _total_bailout_count++;
2453     if (UsePerfData) {
2454       _perf_last_failed_method->set_value(counters->current_method());
2455       _perf_last_failed_type->set_value(counters->compile_type());
2456       _perf_total_bailout_count->inc();
2457     }
2458     _t_bailedout_compilation.add(time);
2459   } else if (code == NULL) {
2460     if (UsePerfData) {
2461       _perf_last_invalidated_method->set_value(counters->current_method());
2462       _perf_last_invalidated_type->set_value(counters->compile_type());
2463       _perf_total_invalidated_count->inc();
2464     }
2465     _total_invalidated_count++;
2466     _t_invalidated_compilation.add(time);
2467   } else {
2468     // Compilation succeeded
2469 
2470     // update compilation ticks - used by the implementation of
2471     // java.lang.management.CompilationMBean
2472     _perf_total_compilation->inc(time.ticks());
2473     _peak_compilation_time = time.milliseconds() > _peak_compilation_time ? time.milliseconds() : _peak_compilation_time;
2474 
2475     if (CITime) {
2476       int bytes_compiled = method->code_size() + task->num_inlined_bytecodes();
2477       if (is_osr) {
2478         _t_osr_compilation.add(time);
2479         _sum_osr_bytes_compiled += bytes_compiled;
2480       } else {
2481         _t_standard_compilation.add(time);
2482         _sum_standard_bytes_compiled += method->code_size() + task->num_inlined_bytecodes();
2483       }
2484 
2485 #if INCLUDE_JVMCI
2486       AbstractCompiler* comp = compiler(task->comp_level());
2487       if (comp) {
2488         CompilerStatistics* stats = comp->stats();
2489         if (stats) {
2490           if (is_osr) {
2491             stats->_osr.update(time, bytes_compiled);
2492           } else {
2493             stats->_standard.update(time, bytes_compiled);
2494           }
2495           stats->_nmethods_size += code->total_size();
2496           stats->_nmethods_code_size += code->insts_size();
2497         } else { // if (!stats)
2498           assert(false, "Compiler statistics object must exist");
2499         }
2500       } else { // if (!comp)
2501         assert(false, "Compiler object must exist");
2502       }
2503 #endif // INCLUDE_JVMCI
2504     }
2505 
2506     if (UsePerfData) {
2507       // save the name of the last method compiled
2508       _perf_last_method->set_value(counters->current_method());
2509       _perf_last_compile_type->set_value(counters->compile_type());
2510       _perf_last_compile_size->set_value(method->code_size() +
2511                                          task->num_inlined_bytecodes());
2512       if (is_osr) {
2513         _perf_osr_compilation->inc(time.ticks());
2514         _perf_sum_osr_bytes_compiled->inc(method->code_size() + task->num_inlined_bytecodes());
2515       } else {
2516         _perf_standard_compilation->inc(time.ticks());
2517         _perf_sum_standard_bytes_compiled->inc(method->code_size() + task->num_inlined_bytecodes());
2518       }
2519     }
2520 
2521     if (CITimeEach) {
2522       float bytes_per_sec = 1.0 * (method->code_size() + task->num_inlined_bytecodes()) / time.seconds();
2523       tty->print_cr("%3d   seconds: %f bytes/sec : %f (bytes %d + %d inlined)",
2524                     compile_id, time.seconds(), bytes_per_sec, method->code_size(), task->num_inlined_bytecodes());
2525     }
2526 
2527     // Collect counts of successful compilations
2528     _sum_nmethod_size      += code->total_size();
2529     _sum_nmethod_code_size += code->insts_size();
2530     _total_compile_count++;
2531 
2532     if (UsePerfData) {
2533       _perf_sum_nmethod_size->inc(     code->total_size());
2534       _perf_sum_nmethod_code_size->inc(code->insts_size());
2535       _perf_total_compile_count->inc();
2536     }
2537 
2538     if (is_osr) {
2539       if (UsePerfData) _perf_total_osr_compile_count->inc();
2540       _total_osr_compile_count++;
2541     } else {
2542       if (UsePerfData) _perf_total_standard_compile_count->inc();
2543       _total_standard_compile_count++;
2544     }
2545   }
2546   // set the current method for the thread to null
2547   if (UsePerfData) counters->set_current_method("");
2548 }

2725 // Note: tty_lock must not be held upon entry to this function.
2726 //       Print functions called from herein do "micro-locking" on tty_lock.
2727 //       That's a tradeoff which keeps together important blocks of output.
2728 //       At the same time, continuous tty_lock hold time is kept in check,
2729 //       preventing concurrently printing threads from stalling a long time.
2730 void CompileBroker::print_heapinfo(outputStream* out, const char* function, size_t granularity) {
2731   TimeStamp ts_total;
2732   TimeStamp ts_global;
2733   TimeStamp ts;
2734 
2735   bool allFun = !strcmp(function, "all");
2736   bool aggregate = !strcmp(function, "aggregate") || !strcmp(function, "analyze") || allFun;
2737   bool usedSpace = !strcmp(function, "UsedSpace") || allFun;
2738   bool freeSpace = !strcmp(function, "FreeSpace") || allFun;
2739   bool methodCount = !strcmp(function, "MethodCount") || allFun;
2740   bool methodSpace = !strcmp(function, "MethodSpace") || allFun;
2741   bool methodAge = !strcmp(function, "MethodAge") || allFun;
2742   bool methodNames = !strcmp(function, "MethodNames") || allFun;
2743   bool discard = !strcmp(function, "discard") || allFun;
2744 
2745   if (out == NULL) {
2746     out = tty;
2747   }
2748 
2749   if (!(aggregate || usedSpace || freeSpace || methodCount || methodSpace || methodAge || methodNames || discard)) {
2750     out->print_cr("\n__ CodeHeapStateAnalytics: Function %s is not supported", function);
2751     out->cr();
2752     return;
2753   }
2754 
2755   ts_total.update(); // record starting point
2756 
2757   if (aggregate) {
2758     print_info(out);
2759   }
2760 
2761   // We hold the CodeHeapStateAnalytics_lock all the time, from here until we leave this function.
2762   // That prevents another thread from destroying our view on the CodeHeap.
2763   // When we request individual parts of the analysis via the jcmd interface, it is possible
2764   // that in between another thread (another jcmd user or the vm running into CodeCache OOM)
2765   // updated the aggregated data. That's a tolerable tradeoff because we can't hold a lock
2766   // across user interaction.
2767   // Acquire this lock before acquiring the CodeCache_lock.
2768   // CodeHeapStateAnalytics_lock could be held by a concurrent thread for a long time,
2769   // leading to an unnecessarily long hold time of the CodeCache_lock.
2770   ts.update(); // record starting point
2771   MutexLocker mu1(CodeHeapStateAnalytics_lock, Mutex::_no_safepoint_check_flag);
2772   out->print_cr("\n__ CodeHeapStateAnalytics lock wait took %10.3f seconds _________\n", ts.seconds());
2773 
2774   // If we serve an "allFun" call, it is beneficial to hold the CodeCache_lock
2775   // for the entire duration of aggregation and printing. That makes sure
2776   // we see a consistent picture and do not run into issues caused by
2777   // the CodeHeap being altered concurrently.
2778   Mutex* global_lock   = allFun ? CodeCache_lock : NULL;
2779   Mutex* function_lock = allFun ? NULL : CodeCache_lock;
2780   ts_global.update(); // record starting point
2781   MutexLocker mu2(global_lock, Mutex::_no_safepoint_check_flag);
2782   if (global_lock != NULL) {
2783     out->print_cr("\n__ CodeCache (global) lock wait took %10.3f seconds _________\n", ts_global.seconds());
2784     ts_global.update(); // record starting point
2785   }
2786 
2787   if (aggregate) {
2788     ts.update(); // record starting point
2789     MutexLocker mu3(function_lock, Mutex::_no_safepoint_check_flag);
2790     if (function_lock != NULL) {
2791       out->print_cr("\n__ CodeCache (function) lock wait took %10.3f seconds _________\n", ts.seconds());
2792     }
2793 
2794     ts.update(); // record starting point
2795     CodeCache::aggregate(out, granularity);
2796     if (function_lock != NULL) {
2797       out->print_cr("\n__ CodeCache (function) lock hold took %10.3f seconds _________\n", ts.seconds());
2798     }
2799   }
2800 
2801   if (usedSpace) CodeCache::print_usedSpace(out);
2802   if (freeSpace) CodeCache::print_freeSpace(out);
2803   if (methodCount) CodeCache::print_count(out);
2804   if (methodSpace) CodeCache::print_space(out);
2805   if (methodAge) CodeCache::print_age(out);
2806   if (methodNames) {
2807     // print_names() has shown to be sensitive to concurrent CodeHeap modifications.
2808     // Therefore, request  the CodeCache_lock before calling...
2809     MutexLocker mu3(function_lock, Mutex::_no_safepoint_check_flag);
2810     CodeCache::print_names(out);
2811   }

2813 
2814   if (global_lock != NULL) {
2815     out->print_cr("\n__ CodeCache (global) lock hold took %10.3f seconds _________\n", ts_global.seconds());
2816   }
2817   out->print_cr("\n__ CodeHeapStateAnalytics total duration %10.3f seconds _________\n", ts_total.seconds());
