 51 void ConstantPoolCacheEntry::initialize_entry(int index) {
 52   assert(0 < index && index < 0x10000, "sanity check");
 53   _indices = index;
 54   _f1 = NULL;
 55   _f2 = _flags = 0;
 56   assert(constant_pool_index() == index, "");
 57 }
 58 
 59 void ConstantPoolCacheEntry::verify_just_initialized(bool f2_used) {
 60   assert((_indices & (~cp_index_mask)) == 0, "sanity");
 61   assert(_f1 == NULL, "sanity");
 62   assert(_flags == 0, "sanity");
 63   if (!f2_used) {
 64     assert(_f2 == 0, "sanity");
 65   }
 66 }
 67 
 68 void ConstantPoolCacheEntry::reinitialize(bool f2_used) {
 69   _indices &= cp_index_mask;
 70   _f1 = NULL;
 71   _flags = 0;
 72   if (!f2_used) {
 73     _f2 = 0;
 74   }
 75 }

 77 int ConstantPoolCacheEntry::make_flags(TosState state,
 78                                        int option_bits,
 79                                        int field_index_or_method_params) {
 80   assert(state < number_of_states, "Invalid state in make_flags");
 81   int f = ((int)state << tos_state_shift) | option_bits | field_index_or_method_params;
 82   // Preserve existing flag bit values
 83   // The low bits are a field offset, or else the method parameter size.
 84 #ifdef ASSERT
 85   TosState old_state = flag_state();
 86   assert(old_state == (TosState)0 || old_state == state,
 87          "inconsistent cpCache flags state");
 88 #endif
 89   return (_flags | f) ;
 90 }

 92 void ConstantPoolCacheEntry::set_bytecode_1(Bytecodes::Code code) {
 93 #ifdef ASSERT
 94   // Read once.
 95   volatile Bytecodes::Code c = bytecode_1();
 96   assert(c == 0 || c == code || code == 0, "update must be consistent");
 97 #endif
 98   // Need to flush pending stores here before bytecode is written.
 99   Atomic::release_store(&_indices, _indices | ((u_char)code << bytecode_1_shift));
100 }
101 
102 void ConstantPoolCacheEntry::set_bytecode_2(Bytecodes::Code code) {
103 #ifdef ASSERT
104   // Read once.
105   volatile Bytecodes::Code c = bytecode_2();
106   assert(c == 0 || c == code || code == 0, "update must be consistent");
107 #endif
108   // Need to flush pending stores here before bytecode is written.
109   Atomic::release_store(&_indices, _indices | ((u_char)code << bytecode_2_shift));
110 }

112 // Sets f1, ordering with previous writes.
113 void ConstantPoolCacheEntry::release_set_f1(Metadata* f1) {
114   assert(f1 != NULL, "");
115   Atomic::release_store(&_f1, f1);
116 }
117 
118 void ConstantPoolCacheEntry::set_indy_resolution_failed() {
119   Atomic::release_store(&_flags, _flags | (1 << indy_resolution_failed_shift));
120 }

122 // Note that concurrent update of both bytecodes can leave one of them
123 // reset to zero.  This is harmless; the interpreter will simply re-resolve
124 // the damaged entry.  More seriously, the memory synchronization is needed
125 // to flush other fields (f1, f2) completely to memory before the bytecodes
126 // are updated, lest other processors see a non-zero bytecode but zero f1/f2.
127 void ConstantPoolCacheEntry::set_field(Bytecodes::Code get_code,
128                                        Bytecodes::Code put_code,
129                                        Klass* field_holder,
130                                        int field_index,
131                                        int field_offset,
132                                        TosState field_type,
133                                        bool is_final,
134                                        bool is_volatile,
135                                        bool is_flattened,
136                                        bool is_flattenable,
137                                        Klass* root_klass) {
138   set_f1(field_holder);
139   set_f2(field_offset);
140   assert((field_index & field_index_mask) == field_index,
141          "field index does not fit in low flag bits");
142   assert(!is_flattened || is_flattenable, "Sanity check");
143   set_field_flags(field_type,
144                   ((is_volatile ? 1 : 0) << is_volatile_shift) |
145                   ((is_final    ? 1 : 0) << is_final_shift) |
146                   ((is_flattened  ? 1 : 0) << is_flattened_field_shift) |
147                   ((is_flattenable ? 1 : 0) << is_flattenable_field_shift),
148                   field_index);
149   set_bytecode_1(get_code);
150   set_bytecode_2(put_code);
151   NOT_PRODUCT(verify(tty));
152 }

154 void ConstantPoolCacheEntry::set_parameter_size(int value) {
155   // This routine is called only in corner cases where the CPCE is not yet initialized.
156   // See AbstractInterpreter::deopt_continue_after_entry.
157   assert(_flags == 0 || parameter_size() == 0 || parameter_size() == value,
158          "size must not change: parameter_size=%d, value=%d", parameter_size(), value);
159   // Setting the parameter size by itself is only safe if the
160   // current value of _flags is 0, otherwise another thread may have
161   // updated it and we don't want to overwrite that value.  Don't
162   // bother trying to update it once it's nonzero but always make
163   // sure that the final parameter size agrees with what was passed.
164   if (_flags == 0) {
165     intx newflags = (value & parameter_size_mask);
166     Atomic::cmpxchg(&_flags, (intx)0, newflags);
167   }
168   guarantee(parameter_size() == value,
169             "size must not change: parameter_size=%d, value=%d", parameter_size(), value);
170 }

172 void ConstantPoolCacheEntry::set_direct_or_vtable_call(Bytecodes::Code invoke_code,
173                                                        const methodHandle& method,
174                                                        int vtable_index,
175                                                        bool sender_is_interface) {
176   bool is_vtable_call = (vtable_index >= 0);  // FIXME: split this method on this boolean
177   assert(method->interpreter_entry() != NULL, "should have been set at this point");
178   assert(!method->is_obsolete(),  "attempt to write obsolete method to cpCache");
179 
180   int byte_no = -1;
181   bool change_to_virtual = false;
182   InstanceKlass* holder = NULL;  // have to declare this outside the switch
183   switch (invoke_code) {
184     case Bytecodes::_invokeinterface:
185       holder = method->method_holder();
186       // check for private interface method invocations
187       if (vtable_index == Method::nonvirtual_vtable_index && holder->is_interface() ) {
188         assert(method->is_private(), "unexpected non-private method");
189         assert(method->can_be_statically_bound(), "unexpected non-statically-bound method");
190         // set_f2_as_vfinal_method checks if is_vfinal flag is true.
191         set_method_flags(as_TosState(method->result_type()),
192                          (                             1      << is_vfinal_shift) |
193                          ((method->is_final_method() ? 1 : 0) << is_final_shift),
194                          method()->size_of_parameters());
195         set_f2_as_vfinal_method(method());
196         byte_no = 2;
197         set_f1(holder); // interface klass*
198         break;
199       }
200       else {
201         // We get here from InterpreterRuntime::resolve_invoke when an invokeinterface
202         // instruction links to a non-interface method (in Object). This can happen when
203         // an interface redeclares an Object method (like CharSequence declaring toString())
204         // or when invokeinterface is used explicitly.
205         // In that case, the method has no itable index and must be invoked as a virtual.
206         // Set a flag to keep track of this corner case.
207         assert(holder->is_interface() || holder == SystemDictionary::Object_klass(), "unexpected holder class");
208         assert(method->is_public(), "Calling non-public method in Object with invokeinterface");
209         change_to_virtual = true;
210 
211         // ...and fall through as if we were handling invokevirtual:
212       }
213     case Bytecodes::_invokevirtual:
214       {
215         if (!is_vtable_call) {
216           assert(method->can_be_statically_bound(), "");
217           // set_f2_as_vfinal_method checks if is_vfinal flag is true.
218           set_method_flags(as_TosState(method->result_type()),
219                            (                             1      << is_vfinal_shift) |
220                            ((method->is_final_method() ? 1 : 0) << is_final_shift)  |
221                            ((change_to_virtual         ? 1 : 0) << is_forced_virtual_shift),
222                            method()->size_of_parameters());
223           set_f2_as_vfinal_method(method());
224         } else {
225           assert(!method->can_be_statically_bound(), "");
226           assert(vtable_index >= 0, "valid index");
227           assert(!method->is_final_method(), "sanity");
228           set_method_flags(as_TosState(method->result_type()),
229                            ((change_to_virtual ? 1 : 0) << is_forced_virtual_shift),
230                            method()->size_of_parameters());
231           set_f2(vtable_index);
232         }
233         byte_no = 2;
234         break;
235       }
236 
237     case Bytecodes::_invokespecial:
238     case Bytecodes::_invokestatic:
239       assert(!is_vtable_call, "");
240       // Note:  Read and preserve the value of the is_vfinal flag on any
241       // invokevirtual bytecode shared with this constant pool cache entry.
242       // It is cheap and safe to consult is_vfinal() at all times.
243       // Once is_vfinal is set, it must stay that way, lest we get a dangling oop.
244       set_method_flags(as_TosState(method->result_type()),
245                        ((is_vfinal()               ? 1 : 0) << is_vfinal_shift) |
246                        ((method->is_final_method() ? 1 : 0) << is_final_shift),
247                        method()->size_of_parameters());
248       set_f1(method());
249       byte_no = 1;
250       break;
251     default:
252       ShouldNotReachHere();
253       break;
254   }
255 
256   // Note:  byte_no also appears in TemplateTable::resolve.
257   if (byte_no == 1) {
258     assert(invoke_code != Bytecodes::_invokevirtual &&
259            invoke_code != Bytecodes::_invokeinterface, "");
260     bool do_resolve = true;
261     // Don't mark invokespecial to method as resolved if sender is an interface.  The receiver
262     // has to be checked that it is a subclass of the current class every time this bytecode
263     // is executed.
264     if (invoke_code == Bytecodes::_invokespecial && sender_is_interface &&
265         method->name() != vmSymbols::object_initializer_name()) {
266       do_resolve = false;
267     }
268     if (invoke_code == Bytecodes::_invokestatic) {
269       assert(method->method_holder()->is_initialized() ||
270              method->method_holder()->is_reentrant_initialization(Thread::current()),
271              "invalid class initialization state for invoke_static");
272 
273       if (!VM_Version::supports_fast_class_init_checks() && method->needs_clinit_barrier()) {
274         // Don't mark invokestatic to method as resolved if the holder class has not yet completed
275         // initialization. An invokestatic must only proceed if the class is initialized, but if
276         // we resolve it before then that class initialization check is skipped.
277         //
278         // When fast class initialization checks are supported (VM_Version::supports_fast_class_init_checks() == true),
279         // template interpreter supports fast class initialization check for
280         // invokestatic which doesn't require call site re-resolution to
281         // enforce class initialization barrier.
282         do_resolve = false;
283       }
284     }
285     if (do_resolve) {
286       set_bytecode_1(invoke_code);
287     }
288   } else if (byte_no == 2)  {
289     if (change_to_virtual) {
290       assert(invoke_code == Bytecodes::_invokeinterface, "");
291       // NOTE: THIS IS A HACK - BE VERY CAREFUL!!!
292       //
293       // Workaround for the case where we encounter an invokeinterface, but we
294       // should really have an _invokevirtual since the resolved method is a
295       // virtual method in java.lang.Object. This is a corner case in the spec
296       // but is presumably legal. javac does not generate this code.
297       //
298       // We do not set bytecode_1() to _invokeinterface, because that is the
299       // bytecode # used by the interpreter to see if it is resolved.  In this
300       // case, the method gets reresolved with caller for each interface call
301       // because the actual selected method may not be public.
302       //
303       // We set bytecode_2() to _invokevirtual.
304       // See also interpreterRuntime.cpp. (8/25/2000)
305       invoke_code = Bytecodes::_invokevirtual;
306     } else {
307       assert(invoke_code == Bytecodes::_invokevirtual ||
308              (invoke_code == Bytecodes::_invokeinterface &&
309               ((method->is_private() ||
310                 (method->is_final() && method->method_holder() == SystemDictionary::Object_klass())))),
311              "unexpected invocation mode");
312       if (invoke_code == Bytecodes::_invokeinterface &&
313           (method->is_private() || method->is_final())) {
314         // We set bytecode_1() to _invokeinterface, because that is the
315         // bytecode # used by the interpreter to see if it is resolved.
316         // We set bytecode_2() to _invokevirtual.
317         set_bytecode_1(invoke_code);
318       }
319     }
320     // set up for invokevirtual, even if linking for invokeinterface also:
321     set_bytecode_2(invoke_code);
322   } else {
323     ShouldNotReachHere();
324   }
325   NOT_PRODUCT(verify(tty));
326 }

342 void ConstantPoolCacheEntry::set_itable_call(Bytecodes::Code invoke_code,
343                                              Klass* referenced_klass,
344                                              const methodHandle& method, int index) {
345   assert(method->method_holder()->verify_itable_index(index), "");
346   assert(invoke_code == Bytecodes::_invokeinterface, "");
347   InstanceKlass* interf = method->method_holder();
348   assert(interf->is_interface(), "must be an interface");
349   assert(!method->is_final_method(), "interfaces do not have final methods; cannot link to one here");
350   set_f1(referenced_klass);
351   set_f2((intx)method());
352   set_method_flags(as_TosState(method->result_type()),
353                    0,  // no option bits
354                    method()->size_of_parameters());
355   set_bytecode_1(Bytecodes::_invokeinterface);
356 }

367 void ConstantPoolCacheEntry::set_method_handle_common(const constantPoolHandle& cpool,
368                                                       Bytecodes::Code invoke_code,
369                                                       const CallInfo &call_info) {
370   // NOTE: This CPCE can be the subject of data races.
371   // There are three words to update: flags, refs[f2], f1 (in that order).
372   // Writers must store all other values before f1.
373   // Readers must test f1 first for non-null before reading other fields.
374   // Competing writers must acquire exclusive access via a lock.
375   // A losing writer waits on the lock until the winner writes f1 and leaves
376   // the lock, so that when the losing writer returns, he can use the linked
377   // cache entry.
378 
379   objArrayHandle resolved_references(Thread::current(), cpool->resolved_references());
380   // Use the resolved_references() lock for this cpCache entry.
381   // resolved_references are created for all classes with Invokedynamic, MethodHandle
382   // or MethodType constant pool cache entries.
383   assert(resolved_references() != NULL,
384          "a resolved_references array should have been created for this class");
385   ObjectLocker ol(resolved_references, Thread::current());
386   if (!is_f1_null()) {
387     return;
388   }
389 
390   if (indy_resolution_failed()) {
391     // Before we got here, another thread got a LinkageError exception during
392     // resolution.  Ignore our success and throw their exception.
393     ConstantPoolCache* cpCache = cpool->cache();
394     int index = -1;
395     for (int i = 0; i < cpCache->length(); i++) {
396       if (cpCache->entry_at(i) == this) {
397         index = i;
398         break;
399       }
400     }
401     guarantee(index >= 0, "Didn't find cpCache entry!");
402     int encoded_index = ResolutionErrorTable::encode_cpcache_index(
403                           ConstantPool::encode_invokedynamic_index(index));
404     Thread* THREAD = Thread::current();
405     ConstantPool::throw_resolution_error(cpool, encoded_index, THREAD);
406     return;
407   }
408 
409   Method* adapter            = call_info.resolved_method();
410   const Handle appendix      = call_info.resolved_appendix();
411   const bool has_appendix    = appendix.not_null();
412 
413   // Write the flags.
414   // MHs and indy are always sig-poly and have a local signature.
415   set_method_flags(as_TosState(adapter->result_type()),
416                    ((has_appendix    ? 1 : 0) << has_appendix_shift        ) |
417                    (                   1      << has_local_signature_shift ) |
418                    (                   1      << is_final_shift            ),
419                    adapter->size_of_parameters());
420 
421   if (TraceInvokeDynamic) {
422     ttyLocker ttyl;
423     tty->print_cr("set_method_handle bc=%d appendix=" PTR_FORMAT "%s method=" PTR_FORMAT " (local signature) ",
424                   invoke_code,
425                   p2i(appendix()),
426                   (has_appendix ? "" : " (unused)"),
427                   p2i(adapter));
428     adapter->print();
429     if (has_appendix)  appendix()->print();
430   }
431 
432   // Method handle invokes and invokedynamic sites use both cp cache words.
433   // refs[f2], if not null, contains a value passed as a trailing argument to the adapter.
434   // In the general case, this could be the call site's MethodType,
435   // for use with java.lang.Invokers.checkExactType, or else a CallSite object.
436   // f1 contains the adapter method which manages the actual call.
437   // In the general case, this is a compiled LambdaForm.
438   // (The Java code is free to optimize these calls by binding other
439   // sorts of methods and appendices to call sites.)
440   // JVM-level linking is via f1, as if for invokespecial, and signatures are erased.
441   // The appendix argument (if any) is added to the signature, and is counted in the parameter_size bits.
442   // Even with the appendix, the method will never take more than 255 parameter slots.
443   //
444   // This means that given a call site like (List)mh.invoke("foo"),
445   // the f1 method has signature '(Ljl/Object;Ljl/invoke/MethodType;)Ljl/Object;',
446   // not '(Ljava/lang/String;)Ljava/util/List;'.
447   // The fact that String and List are involved is encoded in the MethodType in refs[f2].
448   // This allows us to create fewer Methods, while keeping type safety.
449   //
450 
451   // Store appendix, if any.
452   if (has_appendix) {
453     const int appendix_index = f2_as_index();
454     assert(appendix_index >= 0 && appendix_index < resolved_references->length(), "oob");
455     assert(resolved_references->obj_at(appendix_index) == NULL, "init just once");
456     resolved_references->obj_at_put(appendix_index, appendix());
457   }
458 
459   release_set_f1(adapter);  // This must be the last one to set (see NOTE above)!
460 
461   // The interpreter assembly code does not check byte_2,
462   // but it is used by is_resolved, method_if_resolved, etc.
463   set_bytecode_1(invoke_code);
464   NOT_PRODUCT(verify(tty));
465   if (TraceInvokeDynamic) {
466     ttyLocker ttyl;
467     this->print(tty, 0);
468   }
469 
470   assert(has_appendix == this->has_appendix(), "proper storage of appendix flag");
471   assert(this->has_local_signature(), "proper storage of signature flag");
472 }

474 bool ConstantPoolCacheEntry::save_and_throw_indy_exc(
475   const constantPoolHandle& cpool, int cpool_index, int index, constantTag tag, TRAPS) {
476 
477   assert(HAS_PENDING_EXCEPTION, "No exception got thrown!");
478   assert(PENDING_EXCEPTION->is_a(SystemDictionary::LinkageError_klass()),
479          "No LinkageError exception");
480 
481   // Use the resolved_references() lock for this cpCache entry.
482   // resolved_references are created for all classes with Invokedynamic, MethodHandle
483   // or MethodType constant pool cache entries.
484   objArrayHandle resolved_references(Thread::current(), cpool->resolved_references());
485   assert(resolved_references() != NULL,
486          "a resolved_references array should have been created for this class");
487   ObjectLocker ol(resolved_references, THREAD);
488 
489   // if f1 is not null or the indy_resolution_failed flag is set then another
490   // thread either succeeded in resolving the method or got a LinkageError
491   // exception, before this thread was able to record its failure.  So, clear
492   // this thread's exception and return false so caller can use the earlier
493   // thread's result.
494   if (!is_f1_null() || indy_resolution_failed()) {
495     CLEAR_PENDING_EXCEPTION;
496     return false;
497   }
498 
499   Symbol* error = PENDING_EXCEPTION->klass()->name();
500   Symbol* message = java_lang_Throwable::detail_message(PENDING_EXCEPTION);
501 
502   SystemDictionary::add_resolution_error(cpool, index, error, message);
503   set_indy_resolution_failed();
504   return true;
505 }

507 Method* ConstantPoolCacheEntry::method_if_resolved(const constantPoolHandle& cpool) {
508   // Decode the action of set_method and set_interface_call
509   Bytecodes::Code invoke_code = bytecode_1();
510   if (invoke_code != (Bytecodes::Code)0) {
511     Metadata* f1 = f1_ord();
512     if (f1 != NULL) {
513       switch (invoke_code) {
514       case Bytecodes::_invokeinterface:
515         assert(f1->is_klass(), "");
516         return f2_as_interface_method();
517       case Bytecodes::_invokestatic:
518       case Bytecodes::_invokespecial:
519         assert(!has_appendix(), "");
520       case Bytecodes::_invokehandle:
521       case Bytecodes::_invokedynamic:
522         assert(f1->is_method(), "");
523         return (Method*)f1;
524       default:
525         break;
526       }
527     }
528   }
529   invoke_code = bytecode_2();
530   if (invoke_code != (Bytecodes::Code)0) {
531     switch (invoke_code) {
532     case Bytecodes::_invokevirtual:
533       if (is_vfinal()) {
534         // invokevirtual
535         Method* m = f2_as_vfinal_method();
536         assert(m->is_method(), "");
537         return m;
538       } else {
539         int holder_index = cpool->uncached_klass_ref_index_at(constant_pool_index());
540         if (cpool->tag_at(holder_index).is_klass()) {
541           Klass* klass = cpool->resolved_klass_at(holder_index);
542           return klass->method_at_vtable(f2_as_index());
543         }
544       }
545       break;
546     default:
547       break;
548     }
549   }
550   return NULL;
551 }

553 
554 oop ConstantPoolCacheEntry::appendix_if_resolved(const constantPoolHandle& cpool) {
555   if (!has_appendix())
556     return NULL;
557   const int ref_index = f2_as_index();
558   objArrayOop resolved_references = cpool->resolved_references();
559   return resolved_references->obj_at(ref_index);

577 // RedefineClasses() API support:
578 // If this ConstantPoolCacheEntry refers to old_method then update it
579 // to refer to new_method.
580 void ConstantPoolCacheEntry::adjust_method_entry(Method* old_method,
581        Method* new_method, bool * trace_name_printed) {
582 
583   if (is_vfinal()) {
584     // virtual and final so _f2 contains method ptr instead of vtable index
585     if (f2_as_vfinal_method() == old_method) {
586       // match old_method so need an update
587       // NOTE: can't use set_f2_as_vfinal_method as it asserts on different values
588       _f2 = (intptr_t)new_method;
589       log_adjust("vfinal", old_method, new_method, trace_name_printed);
590     }
591     return;
592   }
593 
594   assert (_f1 != NULL, "should not call with uninteresting entry");
595 
596   if (!(_f1->is_method())) {
597     // _f1 is a Klass* for an interface, _f2 is the method
598     if (f2_as_interface_method() == old_method) {
599       _f2 = (intptr_t)new_method;
600       log_adjust("interface", old_method, new_method, trace_name_printed);
601     }
602   } else if (_f1 == old_method) {
603     _f1 = new_method;
604     log_adjust("special, static or dynamic", old_method, new_method, trace_name_printed);
605   }
606 }

608 // a constant pool cache entry should never contain old or obsolete methods
609 bool ConstantPoolCacheEntry::check_no_old_or_obsolete_entries() {
610   Method* m = get_interesting_method_entry();
611   // return false if m refers to a non-deleted old or obsolete method
612   if (m != NULL) {
613     assert(m->is_valid() && m->is_method(), "m is a valid method");
614     return !m->is_old() && !m->is_obsolete(); // old is always set for old and obsolete
615   } else {
616     return true;
617   }
618 }
619 
620 Method* ConstantPoolCacheEntry::get_interesting_method_entry() {
621   if (!is_method_entry()) {
622     // not a method entry so not interesting by default
623     return NULL;
624   }
625   Method* m = NULL;
626   if (is_vfinal()) {
627     // virtual and final so _f2 contains method ptr instead of vtable index
628     m = f2_as_vfinal_method();
629   } else if (is_f1_null()) {
630     // NULL _f1 means this is a virtual entry so also not interesting
631     return NULL;
632   } else {
633     if (!(_f1->is_method())) {
634       // _f1 is a Klass* for an interface
635       m = f2_as_interface_method();
636     } else {
637       m = f1_as_method();
638     }
639   }
640   assert(m != NULL && m->is_method(), "sanity check");
641   if (m == NULL || !m->is_method()) {
642     return NULL;
643   }
644   return m;
645 }

667 ConstantPoolCache* ConstantPoolCache::allocate(ClassLoaderData* loader_data,
668                                      const intStack& index_map,
669                                      const intStack& invokedynamic_index_map,
670                                      const intStack& invokedynamic_map, TRAPS) {

673   int size = ConstantPoolCache::size(length);

676     ConstantPoolCache(length, index_map, invokedynamic_index_map, invokedynamic_map);

679 void ConstantPoolCache::initialize(const intArray& inverse_index_map,
680                                    const intArray& invokedynamic_inverse_index_map,
681                                    const intArray& invokedynamic_references_map) {
682   for (int i = 0; i < inverse_index_map.length(); i++) {
683     ConstantPoolCacheEntry* e = entry_at(i);
684     int original_index = inverse_index_map.at(i);
685     e->initialize_entry(original_index);
686     assert(entry_at(i) == e, "sanity");
687   }
688 
689   // Append invokedynamic entries at the end
690   int invokedynamic_offset = inverse_index_map.length();
691   for (int i = 0; i < invokedynamic_inverse_index_map.length(); i++) {
692     int offset = i + invokedynamic_offset;
693     ConstantPoolCacheEntry* e = entry_at(offset);
694     int original_index = invokedynamic_inverse_index_map.at(i);
695     e->initialize_entry(original_index);
696     assert(entry_at(offset) == e, "sanity");
697   }
698 
699   for (int ref = 0; ref < invokedynamic_references_map.length(); ref++) {
700     const int cpci = invokedynamic_references_map.at(ref);
701     if (cpci >= 0) {
702       entry_at(cpci)->initialize_resolved_reference_index(ref);
703     }
704   }
705 }

714 
715 void ConstantPoolCache::walk_entries_for_initialization(bool check_only) {
716   Arguments::assert_is_dumping_archive();
717   // When dumping the archive, we want to clean up the ConstantPoolCache
718   // to remove any effect of linking due to the execution of Java code --
719   // each ConstantPoolCacheEntry will have the same contents as if
720   // ConstantPoolCache::initialize has just returned:
721   //
722   // - We keep the ConstantPoolCache::constant_pool_index() bits for all entries.
723   // - We keep the "f2" field for entries used by invokedynamic and invokehandle
724   // - All other bits in the entries are cleared to zero.
725   ResourceMark rm;
726 
727   InstanceKlass* ik = constant_pool()->pool_holder();
728   bool* f2_used = NEW_RESOURCE_ARRAY(bool, length());
729   memset(f2_used, 0, sizeof(bool) * length());
730 
731   Thread* THREAD = Thread::current();
732 
733   // Find all the slots that we need to preserve f2
734   for (int i = 0; i < ik->methods()->length(); i++) {
735     Method* m = ik->methods()->at(i);
736     RawBytecodeStream bcs(methodHandle(THREAD, m));
737     while (!bcs.is_last_bytecode()) {
738       Bytecodes::Code opcode = bcs.raw_next();
739       switch (opcode) {
740       case Bytecodes::_invokedynamic: {
741           int index = Bytes::get_native_u4(bcs.bcp() + 1);
742           int cp_cache_index = constant_pool()->invokedynamic_cp_cache_index(index);
743           f2_used[cp_cache_index] = 1;
744         }
745         break;
746       case Bytecodes::_invokehandle: {
747           int cp_cache_index = Bytes::get_native_u2(bcs.bcp() + 1);
748           f2_used[cp_cache_index] = 1;
749         }
750         break;
751       default:
752         break;
753       }
754     }
755   }
756 
757   if (check_only) {
758     DEBUG_ONLY(
759       for (int i=0; i<length(); i++) {
760         entry_at(i)->verify_just_initialized(f2_used[i]);
761       })
762   } else {
763     for (int i=0; i<length(); i++) {
764       entry_at(i)->reinitialize(f2_used[i]);
765     }
766   }
767 }

791 #if INCLUDE_JVMTI
792 // RedefineClasses() API support:
793 // If any entry of this ConstantPoolCache points to any of
794 // old_methods, replace it with the corresponding new_method.
795 void ConstantPoolCache::adjust_method_entries(bool * trace_name_printed) {
796   for (int i = 0; i < length(); i++) {
797     ConstantPoolCacheEntry* entry = entry_at(i);
798     Method* old_method = entry->get_interesting_method_entry();
799     if (old_method == NULL || !old_method->is_old()) {
800       continue; // skip uninteresting entries
801     }
802     if (old_method->is_deleted()) {
803       // clean up entries with deleted methods
804       entry->initialize_entry(entry->constant_pool_index());
805       continue;
806     }
807     Method* new_method = old_method->get_new_method();
808     entry_at(i)->adjust_method_entry(old_method, new_method, trace_name_printed);
809   }
810 }

812 // the constant pool cache should never contain old or obsolete methods
813 bool ConstantPoolCache::check_no_old_or_obsolete_entries() {
814   for (int i = 1; i < length(); i++) {
815     if (entry_at(i)->get_interesting_method_entry() != NULL &&
816         !entry_at(i)->check_no_old_or_obsolete_entries()) {
817       return false;
818     }
819   }
820   return true;
821 }
