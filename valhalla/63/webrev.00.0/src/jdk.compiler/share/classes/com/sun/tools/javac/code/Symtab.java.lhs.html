 75 /** A class that defines all predefined constants and operators
 76  *  as well as special classes such as java.lang.Object, which need
 77  *  to be known to the compiler. All symbols are held in instance
 78  *  fields. This makes it possible to work in multiple concurrent
 79  *  projects, which might use different class files for library classes.
 80  *
 81  *  <p><b>This is NOT part of any supported API.
 82  *  If you write code that depends on this, you do so at your own risk.
 83  *  This code and its internal interfaces are subject to change or
 84  *  deletion without notice.</b>
 85  */

 90     /** Get the symbol table instance. */
 91     public static Symtab instance(Context context) {
 92         Symtab instance = context.get(symtabKey);
 93         if (instance == null)
 94             instance = new Symtab(context);
 95         return instance;
 96     }

265     public VarSymbol getClassField(Type type, Types types) {
266         return classFields.computeIfAbsent(
267             new UniqueType(type, types), k -> {
268                 Type arg = null;
269                 if (type.getTag() == ARRAY || type.getTag() == CLASS) {
270                     /* Temporary treatment for inline class: Given an inline class V that implements
271                        I1, I2, ... In, V.class is typed to be Class<? extends Object & I1 & I2 .. & In>
272                     */
273                     if (type.isValue()) {
274                         List<Type> bounds = List.of(objectType).appendList(((ClassSymbol) type.tsym).getInterfaces());
275                         arg = new WildcardType(types.makeIntersectionType(bounds), BoundKind.EXTENDS, boundClass);
276                     } else {
277                         arg = types.erasure(type);
278                     }
279                 }
280                 else if (type.isPrimitiveOrVoid())
281                     arg = types.boxedClass(type).type;
282                 else
283                     throw new AssertionError(type);
284 
285                 Type t = new ClassType(
286                     classType.getEnclosingType(), List.of(arg), classType.tsym);
287                 return new VarSymbol(
288                     STATIC | PUBLIC | FINAL, names._class, t, type.tsym);
289             });
290     }

320     public void synthesizeEmptyInterfaceIfMissing(final Type type) {
321         final Completer completer = type.tsym.completer;
322         type.tsym.completer = new Completer() {
323             @Override
324             public void complete(Symbol sym) throws CompletionFailure {
325                 try {
326                     completer.complete(sym);
327                 } catch (CompletionFailure e) {
328                     sym.flags_field |= (PUBLIC | INTERFACE);
329                     ((ClassType) sym.type).supertype_field = objectType;
330                 }
331             }
332 
333             @Override
334             public boolean isTerminal() {
335                 return completer.isTerminal();
336             }
337         };
338     }

340     public void synthesizeBoxTypeIfMissing(final Type type) {
341         ClassSymbol sym = enterClass(java_base, boxedName[type.getTag().ordinal()]);
342         final Completer completer = sym.completer;
343         sym.completer = new Completer() {
344             @Override
345             public void complete(Symbol sym) throws CompletionFailure {
346                 try {
347                     completer.complete(sym);
348                 } catch (CompletionFailure e) {
349                     sym.flags_field |= PUBLIC;
350                     ((ClassType) sym.type).supertype_field = objectType;
351                     MethodSymbol boxMethod =
352                         new MethodSymbol(PUBLIC | STATIC, names.valueOf,
353                                          new MethodType(List.of(type), sym.type,
354                                 List.nil(), methodClass),
355                             sym);
356                     sym.members().enter(boxMethod);
357                     MethodSymbol unboxMethod =
358                         new MethodSymbol(PUBLIC,
359                             type.tsym.name.append(names.Value), // x.intValue()
360                             new MethodType(List.nil(), type,
361                                 List.nil(), methodClass),
362                             sym);
363                     sym.members().enter(unboxMethod);
364                 }
365             }
366 
367             @Override
368             public boolean isTerminal() {
369                 return completer.isTerminal();
370             }
371         };
372     }

374     // Enter a synthetic class that is used to mark classes in ct.sym.
375     // This class does not have a class file.
376     private Type enterSyntheticAnnotation(String name) {
377         // for now, leave the module null, to prevent problems from synthesizing the
378         // existence of a class in any specific module, including noModule
379         ClassType type = (ClassType)enterClass(java_base, names.fromString(name)).type;
380         ClassSymbol sym = (ClassSymbol)type.tsym;
381         sym.completer = Completer.NULL_COMPLETER;
382         sym.flags_field = PUBLIC|ACYCLIC|ANNOTATION|INTERFACE;
383         sym.erasure_field = type;
384         sym.members_field = WriteableScope.create(sym);
385         type.typarams_field = List.nil();
386         type.allparams_field = List.nil();
387         type.supertype_field = annotationType;
388         type.interfaces_field = List.nil();
389         return type;
390     }

616         // Enter a synthetic class that is used to provide profile info for
617         // classes in ct.sym.  This class does not have a class file.
618         profileType = enterSyntheticAnnotation("jdk.Profile+Annotation");
619         MethodSymbol m = new MethodSymbol(PUBLIC | ABSTRACT, names.value, intType, profileType.tsym);
620         profileType.tsym.members().enter(m);
621 
622         // Enter a class for arrays.
623         // The class implements java.lang.Cloneable and java.io.Serializable.
624         // It has a final length field and a clone method.
625         ClassType arrayClassType = (ClassType)arrayClass.type;
626         arrayClassType.supertype_field = objectType;
627         arrayClassType.interfaces_field =
628                 List.of(cloneableType, serializableType, identityObjectType);
629 
630         arrayClass.members_field = WriteableScope.create(arrayClass);
631         lengthVar = new VarSymbol(
632             PUBLIC | FINAL,
633             names.length,
634             intType,
635             arrayClass);
636         arrayClass.members().enter(lengthVar);
637         arrayCloneMethod = new MethodSymbol(
638             PUBLIC,
639             names.clone,
640             new MethodType(List.nil(), objectType,
641                            List.nil(), methodClass),
642             arrayClass);
643         arrayClass.members().enter(arrayCloneMethod);
644 
645         if (java_base != noModule)
646             java_base.completer = moduleCompleter::complete; //bootstrap issues
647 
648     }

659      *  and owner and enter in `classes' unless already there.
660      */
661     public ClassSymbol enterClass(ModuleSymbol msym, Name name, TypeSymbol owner) {
662         Assert.checkNonNull(msym);
663         Name flatname = TypeSymbol.formFlatName(name, owner);
664         ClassSymbol c = getClass(msym, flatname);
665         if (c == null) {
666             c = defineClass(name, owner);
667             doEnterClass(msym, c);
668         } else if ((c.name != name || c.owner != owner) && owner.kind == TYP && c.owner.kind == PCK) {
669             // reassign fields of classes that might have been loaded with
670             // their flat names.
671             c.owner.members().remove(c);
672             c.name = name;
673             c.owner = owner;
674             c.fullname = ClassSymbol.formFullName(name, owner);
675         }
676         return c;
677     }

684     public PackageSymbol lookupPackage(ModuleSymbol msym, Name flatName) {
685         Assert.checkNonNull(msym);
686 
687         if (flatName.isEmpty()) {
688             //unnamed packages only from the current module - visiblePackages contains *root* package, not unnamed package!
689             return msym.unnamedPackage;
690         }
691 
692         if (msym == noModule) {
693             return enterPackage(msym, flatName);
694         }
695 
696         msym.complete();
697 
698         PackageSymbol pack;
699 
700         pack = msym.visiblePackages.get(flatName);
701 
702         if (pack != null)
703             return pack;
704 
705         pack = getPackage(msym, flatName);
706 
707         if (pack != null && pack.exists())
708             return pack;
709 
710         boolean dependsOnUnnamed = msym.requires != null &&
711                                    msym.requires.stream()
712                                                 .map(rd -> rd.module)
713                                                 .anyMatch(mod -> mod == unnamedModule);
714 
715         if (dependsOnUnnamed) {
716             //msyms depends on the unnamed module, for which we generally don't know
717             //the list of packages it "exports" ahead of time. So try to lookup the package in the
718             //current module, and in the unnamed module and see if it exists in one of them
719             PackageSymbol unnamedPack = getPackage(unnamedModule, flatName);
720 
721             if (unnamedPack != null && unnamedPack.exists()) {
722                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
723                 return unnamedPack;
724             }
725 
726             pack = enterPackage(msym, flatName);
727             pack.complete();
728             if (pack.exists())
729                 return pack;
730 
731             unnamedPack = enterPackage(unnamedModule, flatName);
732             unnamedPack.complete();
733             if (unnamedPack.exists()) {
734                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
735                 return unnamedPack;
736             }
737 
738             return pack;
739         }
740 
741         return enterPackage(msym, flatName);
742     }

784     public PackageSymbol enterPackage(ModuleSymbol currModule, Name fullname) {
785         Assert.checkNonNull(currModule);
786         PackageSymbol p = getPackage(currModule, fullname);
787         if (p == null) {
788             Assert.check(!fullname.isEmpty(), () -> "rootPackage missing!; currModule: " + currModule);
789             p = new PackageSymbol(
790                     Convert.shortName(fullname),
791                     enterPackage(currModule, Convert.packagePart(fullname)));
792             p.completer = initialCompleter;
793             p.modle = currModule;
794             doEnterPackage(currModule, p);
795         }
796         return p;
797     }
