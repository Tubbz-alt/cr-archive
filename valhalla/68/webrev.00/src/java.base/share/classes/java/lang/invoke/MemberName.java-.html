  46 /**
  47  * A {@code MemberName} is a compact symbolic datum which fully characterizes
  48  * a method or field reference.
  49  * A member name refers to a field, method, constructor, or member type.
  50  * Every member name has a simple name (a string) and a type (either a Class or MethodType).
  51  * A member name may also have a non-null declaring class, or it may be simply
  52  * a naked name/type pair.
  53  * A member name may also have non-zero modifier flags.
  54  * Finally, a member name may be either resolved or unresolved.
  55  * If it is resolved, the existence of the named member has been determined by the JVM.
  56  * <p>
  57  * Whether resolved or not, a member name provides no access rights or
  58  * invocation capability to its possessor.  It is merely a compact
  59  * representation of all symbolic information necessary to link to
  60  * and properly use the named member.
  61  * <p>
  62  * When resolved, a member name's internal implementation may include references to JVM metadata.
  63  * This representation is stateless and only descriptive.
  64  * It provides no private information and no capability to use the member.
  65  * <p>
  66  * By contrast, a {@linkplain java.lang.reflect.Method} contains fuller information
  67  * about the internals of a method (except its bytecodes) and also
  68  * allows invocation.  A MemberName is much lighter than a Method,
  69  * since it contains about 7 fields to the 16 of Method (plus its sub-arrays),
  70  * and those seven fields omit much of the information in Method.
  71  * @author jrose
  72  */
  73 /*non-public*/
  74 final class ResolvedMethodName {
  75     //@Injected JVM_Method* vmtarget;
  76     //@Injected Class<?>    vmholder;
  77 };

 126     /** Return the declared type of this member, which
 127      *  must be a method or constructor.
 128      */
 129     public MethodType getMethodType() {
 130         if (type == null) {
 131             expandFromVM();
 132             if (type == null) {
 133                 return null;
 134             }
 135         }
 136         if (!isInvocable()) {
 137             throw newIllegalArgumentException("not invocable, no method type");
 138         }
 139 
 140         {
 141             // Get a snapshot of type which doesn't get changed by racing threads.
 142             final Object type = this.type;
 143             if (type instanceof MethodType) {
 144                 return (MethodType) type;
 145             }
 146         }
 147 
 148         // type is not a MethodType yet.  Convert it thread-safely.
 149         synchronized (this) {
 150             if (type instanceof String) {
 151                 String sig = (String) type;
 152                 MethodType res = MethodType.fromDescriptor(sig, getClassLoader());
 153                 type = res;
 154             } else if (type instanceof Object[]) {
 155                 Object[] typeInfo = (Object[]) type;
 156                 Class<?>[] ptypes = (Class<?>[]) typeInfo[1];
 157                 Class<?> rtype = (Class<?>) typeInfo[0];
 158                 MethodType res = MethodType.makeImpl(rtype, ptypes, true);
 159                 type = res;
 160             }
 161             // Make sure type is a MethodType for racing threads.
 162             assert type instanceof MethodType : "bad method type " + type;
 163         }
 164         return (MethodType) type;
 165     }

 213     /** Return the declared type of this member, which
 214      *  must be a field or type.
 215      *  If it is a type member, that type itself is returned.
 216      */
 217     public Class<?> getFieldType() {
 218         if (type == null) {
 219             expandFromVM();
 220             if (type == null) {
 221                 return null;
 222             }
 223         }
 224         if (isInvocable()) {
 225             throw newIllegalArgumentException("not a field or nested class, no simple type");
 226         }
 227 
 228         {
 229             // Get a snapshot of type which doesn't get changed by racing threads.
 230             final Object type = this.type;
 231             if (type instanceof Class<?>) {
 232                 return (Class<?>) type;
 233             }
 234         }
 235 
 236         // type is not a Class yet.  Convert it thread-safely.
 237         synchronized (this) {
 238             if (type instanceof String) {
 239                 String sig = (String) type;
 240                 MethodType mtype = MethodType.fromDescriptor("()"+sig, getClassLoader());
 241                 Class<?> res = mtype.returnType();
 242                 type = res;
 243             }
 244             // Make sure type is a Class for racing threads.
 245             assert type instanceof Class<?> : "bad field type " + type;
 246         }
 247         return (Class<?>) type;
 248     }

 283     private boolean referenceKindIsConsistent() {
 284         byte refKind = getReferenceKind();
 285         if (refKind == REF_NONE)  return isType();
 286         if (isField()) {
 287             assert(staticIsConsistent());
 288             assert(MethodHandleNatives.refKindIsField(refKind));
 289         } else if (isObjectConstructor()) {
 290             assert(refKind == REF_newInvokeSpecial || refKind == REF_invokeSpecial);
 291         } else if (isMethod()) {
 292             assert(staticIsConsistent());
 293             assert(MethodHandleNatives.refKindIsMethod(refKind));
 294             if (clazz.isInterface())
 295                 assert(refKind == REF_invokeInterface ||
 296                        refKind == REF_invokeStatic    ||
 297                        refKind == REF_invokeSpecial   ||
 298                        refKind == REF_invokeVirtual && isObjectPublicMethod());
 299         } else {
 300             assert(false);
 301         }
 302         return true;
 303     }
 304     private boolean isObjectPublicMethod() {
 305         if (clazz == Object.class)  return true;
 306         MethodType mtype = getMethodType();
 307         if (name.equals("toString") && mtype.returnType() == String.class && mtype.parameterCount() == 0)
 308             return true;
 309         if (name.equals("hashCode") && mtype.returnType() == int.class && mtype.parameterCount() == 0)
 310             return true;
 311         if (name.equals("equals") && mtype.returnType() == boolean.class && mtype.parameterCount() == 1 && mtype.parameterType(0) == Object.class)
 312             return true;
 313         return false;
 314     }

 316     /*non-public*/
 317     boolean referenceKindIsConsistentWith(int originalRefKind) {
 318         int refKind = getReferenceKind();
 319         if (refKind == originalRefKind)  return true;
 320         switch (originalRefKind) {
 321         case REF_invokeInterface:
 322             // Looking up an interface method, can get (e.g.) Object.hashCode
 323             assert(refKind == REF_invokeVirtual ||
 324                    refKind == REF_invokeSpecial) : this;
 325             return true;
 326         case REF_invokeVirtual:
 327         case REF_newInvokeSpecial:
 328             // Looked up a virtual, can get (e.g.) final String.hashCode.
 329             assert(refKind == REF_invokeSpecial) : this;
 330             return true;
 331         }
 332         assert(false) : this+" != "+MethodHandleNatives.refKindName((byte)originalRefKind);
 333         return true;
 334     }
 335     private boolean staticIsConsistent() {
 336         byte refKind = getReferenceKind();
 337         return MethodHandleNatives.refKindIsStatic(refKind) == isStatic() || getModifiers() == 0;
 338     }
 339     private boolean vminfoIsConsistent() {
 340         byte refKind = getReferenceKind();
 341         assert(isResolved());  // else don't call
 342         Object vminfo = MethodHandleNatives.getMemberVMInfo(this);
 343         assert(vminfo instanceof Object[]);
 344         long vmindex = (Long) ((Object[])vminfo)[0];
 345         Object vmtarget = ((Object[])vminfo)[1];
 346         if (MethodHandleNatives.refKindIsField(refKind)) {
 347             assert(vmindex >= 0) : vmindex + ":" + this;
 348             assert(vmtarget instanceof Class);
 349         } else {
 350             if (MethodHandleNatives.refKindDoesDispatch(refKind))
 351                 assert(vmindex >= 0) : vmindex + ":" + this;
 352             else
 353                 assert(vmindex < 0) : vmindex;
 354             assert(vmtarget instanceof MemberName) : vmtarget + " in " + this;
 355         }
 356         return true;
 357     }

 359     private MemberName changeReferenceKind(byte refKind, byte oldKind) {
 360         assert(getReferenceKind() == oldKind);
 361         assert(MethodHandleNatives.refKindIsValid(refKind));
 362         flags += (((int)refKind - oldKind) << MN_REFERENCE_KIND_SHIFT);
 363         return this;
 364     }

 376     /** Utility method to query if this member is a method handle invocation (invoke or invokeExact).
 377      */
 378     public boolean isMethodHandleInvoke() {
 379         final int bits = MH_INVOKE_MODS &~ Modifier.PUBLIC;
 380         final int negs = Modifier.STATIC;
 381         if (testFlags(bits | negs, bits) &&
 382             clazz == MethodHandle.class) {
 383             return isMethodHandleInvokeName(name);
 384         }
 385         return false;
 386     }

 390         case "invokeExact":
 391             return true;
 392         default:
 393             return false;
 394         }

 594     // Capturing information from the Core Reflection API:
 595     private static int flagsMods(int flags, int mods, byte refKind) {
 596         assert((flags & RECOGNIZED_MODIFIERS) == 0);
 597         assert((mods & ~RECOGNIZED_MODIFIERS) == 0);
 598         assert((refKind & ~MN_REFERENCE_KIND_MASK) == 0);
 599         return flags | mods | (refKind << MN_REFERENCE_KIND_SHIFT);
 600     }
 601     /** Create a name for the given reflected method.  The resulting name will be in a resolved state. */
 602     public MemberName(Method m) {
 603         this(m, false);
 604     }
 605     @SuppressWarnings("LeakingThisInConstructor")
 606     public MemberName(Method m, boolean wantSpecial) {
 607         Objects.requireNonNull(m);
 608         // fill in vmtarget, vmindex while we have m in hand:
 609         MethodHandleNatives.init(this, m);
 610         if (clazz == null) {  // MHN.init failed
 611             if (m.getDeclaringClass() == MethodHandle.class &&
 612                 isMethodHandleInvokeName(m.getName())) {
 613                 // The JVM did not reify this signature-polymorphic instance.
 614                 // Need a special case here.
 615                 // See comments on MethodHandleNatives.linkMethod.
 616                 MethodType type = MethodType.methodType(m.getReturnType(), m.getParameterTypes());
 617                 int flags = flagsMods(IS_METHOD, m.getModifiers(), REF_invokeVirtual);
 618                 init(MethodHandle.class, m.getName(), type, flags);
 619                 if (isMethodHandleInvoke())
 620                     return;
 621             }
 622             if (m.getDeclaringClass() == VarHandle.class &&
 623                 isVarHandleMethodInvokeName(m.getName())) {
 624                 // The JVM did not reify this signature-polymorphic instance.
 625                 // Need a special case here.
 626                 // See comments on MethodHandleNatives.linkMethod.
 627                 MethodType type = MethodType.methodType(m.getReturnType(), m.getParameterTypes());
 628                 int flags = flagsMods(IS_METHOD, m.getModifiers(), REF_invokeVirtual);
 629                 init(VarHandle.class, m.getName(), type, flags);
 630                 if (isVarHandleMethodInvoke())
 631                     return;
 632             }
 633             throw new LinkageError(m.toString());
 634         }
 635         assert(isResolved() && this.clazz != null);
 636         this.name = m.getName();
 637         if (this.type == null)
 638             this.type = new Object[] { m.getReturnType(), m.getParameterTypes() };
 639         if (wantSpecial) {

 643                 changeReferenceKind(REF_invokeSpecial, REF_invokeVirtual);
 644             else if (getReferenceKind() == REF_invokeInterface)
 645                 // invokeSpecial on a default method
 646                 changeReferenceKind(REF_invokeSpecial, REF_invokeInterface);
 647         }

 668     /** If this MN is a REF_invokeSpecial, return a clone with the "normal" kind
 669      *  REF_invokeVirtual; also switch either to REF_invokeInterface if clazz.isInterface.
 670      *  The end result is to get a fully virtualized version of the MN.
 671      *  (Note that resolving in the JVM will sometimes devirtualize, changing
 672      *  REF_invokeVirtual of a final to REF_invokeSpecial, and REF_invokeInterface
 673      *  in some corner cases to either of the previous two; this transform
 674      *  undoes that change under the assumption that it occurred.)
 675      */
 676     public MemberName asNormalOriginal() {
 677         byte normalVirtual = clazz.isInterface() ? REF_invokeInterface : REF_invokeVirtual;
 678         byte refKind = getReferenceKind();
 679         byte newRefKind = refKind;
 680         MemberName result = this;
 681         switch (refKind) {
 682         case REF_invokeInterface:
 683         case REF_invokeVirtual:
 684         case REF_invokeSpecial:
 685             newRefKind = normalVirtual;
 686             break;
 687         }
 688         if (newRefKind == refKind)
 689             return this;
 690         result = clone().changeReferenceKind(newRefKind, refKind);
 691         assert(this.referenceKindIsConsistentWith(result.getReferenceKind()));
 692         return result;
 693     }

 922     /** Produce a string form of this member name.
 923      *  For types, it is simply the type's own string (as reported by {@code toString}).
 924      *  For fields, it is {@code "DeclaringClass.name/type"}.
 925      *  For methods and constructors, it is {@code "DeclaringClass.name(ptype...)rtype"}.
 926      *  If the declaring class is null, the prefix {@code "DeclaringClass."} is omitted.
 927      *  If the member is unresolved, a prefix {@code "*."} is prepended.
 928      */
 929     @SuppressWarnings("LocalVariableHidesMemberVariable")
 930     @Override
 931     public String toString() {
 932         if (isType())
 933             return type.toString();  // class java.lang.String
 934         // else it is a field, method, or constructor
 935         StringBuilder buf = new StringBuilder();
 936         if (getDeclaringClass() != null) {
 937             buf.append(getName(clazz));
 938             buf.append('.');
 939         }
 940         String name = this.name; // avoid expanding from VM
 941         buf.append(name == null ? "*" : name);
 942         Object type = this.type; // avoid expanding from VM
 943         if (!isInvocable()) {
 944             buf.append('/');
 945             buf.append(type == null ? "*" : getName(type));
 946         } else {
 947             buf.append(type == null ? "(*)*" : getName(type));
 948         }
 949         byte refKind = getReferenceKind();
 950         if (refKind != REF_NONE) {
 951             buf.append('/');
 952             buf.append(MethodHandleNatives.refKindName(refKind));
 953         }
 954         //buf.append("#").append(System.identityHashCode(this));
 955         return buf.toString();
 956     }

 963     public IllegalAccessException makeAccessException(String message, Object from) {
 964         message = message + ": "+ toString();
 965         if (from != null)  {
 966             if (from == MethodHandles.publicLookup()) {
 967                 message += ", from public Lookup";
 968             } else {
 969                 Module m;
 970                 Class<?> plc;
 971                 if (from instanceof MethodHandles.Lookup) {
 972                     MethodHandles.Lookup lookup = (MethodHandles.Lookup)from;
 973                     from = lookup.lookupClass();
 974                     m = lookup.lookupClass().getModule();
 975                     plc = lookup.previousLookupClass();
 976                 } else {
 977                     m = ((Class<?>)from).getModule();
 978                     plc = null;
 979                 }
 980                 message += ", from " + from + " (" + m + ")";
 981                 if (plc != null) {
 982                     message += ", previous lookup " +
 983                         plc.getName() + " (" + plc.getModule() + ")";
 984                 }
 985             }
 986         }
 987         return new IllegalAccessException(message);
 988     }
 989     private String message() {
 990         if (isResolved())
 991             return "no access";
 992         else if (isObjectConstructor())
 993             return "no such constructor";
 994         else if (isMethod())
 995             return "no such method";
 996         else
 997             return "no such field";
 998     }
 999     public ReflectiveOperationException makeAccessException() {
1000         String message = message() + ": "+ toString();
1001         ReflectiveOperationException ex;
1002         if (isResolved() || !(resolution instanceof NoSuchMethodError ||
1003                               resolution instanceof NoSuchFieldError))
1004             ex = new IllegalAccessException(message);
1005         else if (isObjectConstructor())
1006             ex = new NoSuchMethodException(message);
1007         else if (isMethod())
1008             ex = new NoSuchMethodException(message);
1009         else
1010             ex = new NoSuchFieldException(message);
1011         if (resolution instanceof Throwable)
1012             ex.initCause((Throwable) resolution);
1013         return ex;
1014     }

1021     /** A factory type for resolving member names with the help of the VM.
1022      *  TBD: Define access-safe public constructors for this factory.

1024     /*non-public*/
1025     static class Factory {
1026         private Factory() { } // singleton pattern
1027         static Factory INSTANCE = new Factory();

1029         private static int ALLOWED_FLAGS = ALL_KINDS;
1030 
1031         /// Queries
1032         List<MemberName> getMembers(Class<?> defc,
1033                 String matchName, Object matchType,
1034                 int matchFlags, Class<?> lookupClass) {
1035             matchFlags &= ALLOWED_FLAGS;
1036             String matchSig = null;
1037             if (matchType != null) {
1038                 matchSig = BytecodeDescriptor.unparse(matchType);
1039                 if (matchSig.startsWith("("))
1040                     matchFlags &= ~(ALL_KINDS & ~IS_INVOCABLE);
1041                 else
1042                     matchFlags &= ~(ALL_KINDS & ~IS_FIELD);
1043             }
1044             final int BUF_MAX = 0x2000;
1045             int len1 = matchName == null ? 10 : matchType == null ? 4 : 1;
1046             MemberName[] buf = newMemberBuffer(len1);
1047             int totalCount = 0;
1048             ArrayList<MemberName[]> bufs = null;
1049             int bufCount = 0;
1050             for (;;) {
1051                 bufCount = MethodHandleNatives.getMembers(defc,
1052                         matchName, matchSig, matchFlags,
1053                         lookupClass,
1054                         totalCount, buf);
1055                 if (bufCount <= buf.length) {
1056                     if (bufCount < 0)  bufCount = 0;
1057                     totalCount += bufCount;
1058                     break;
1059                 }
1060                 // JVM returned to us with an intentional overflow!
1061                 totalCount += buf.length;
1062                 int excess = bufCount - buf.length;
1063                 if (bufs == null)  bufs = new ArrayList<>(1);
1064                 bufs.add(buf);
1065                 int len2 = buf.length;
1066                 len2 = Math.max(len2, excess);
1067                 len2 = Math.max(len2, totalCount / 4);
1068                 buf = newMemberBuffer(Math.min(BUF_MAX, len2));
1069             }
1070             ArrayList<MemberName> result = new ArrayList<>(totalCount);
1071             if (bufs != null) {
1072                 for (MemberName[] buf0 : bufs) {
1073                     Collections.addAll(result, buf0);
1074                 }
1075             }
1076             for (int i = 0; i < bufCount; i++) {
1077                 result.add(buf[i]);
1078             }
1079             // Signature matching is not the same as type matching, since
1080             // one signature might correspond to several types.
1081             // So if matchType is a Class or MethodType, refilter the results.
1082             if (matchType != null && matchType != matchSig) {
1083                 for (Iterator<MemberName> it = result.iterator(); it.hasNext();) {
1084                     MemberName m = it.next();
1085                     if (!matchType.equals(m.getType()))
1086                         it.remove();
1087                 }
1088             }
1089             return result;
1090         }
1091         /** Produce a resolved version of the given member.
1092          *  Super types are searched (for inherited members) if {@code searchSupers} is true.
1093          *  Access checking is performed on behalf of the given {@code lookupClass}.
1094          *  If lookup fails or access is not permitted, null is returned.
1095          *  Otherwise a fresh copy of the given member is returned, with modifier bits filled in.
1096          */
1097         private MemberName resolve(byte refKind, MemberName ref, Class<?> lookupClass,
1098                                    boolean speculativeResolve) {
1099             MemberName m = ref.clone();  // JVM will side-effect the ref
1100             assert(refKind == m.getReferenceKind());
1101             try {
1102                 // There are 4 entities in play here:
1103                 //   * LC: lookupClass
1104                 //   * REFC: symbolic reference class (MN.clazz before resolution);
1105                 //   * DEFC: resolved method holder (MN.clazz after resolution);
1106                 //   * PTYPES: parameter types (MN.type)
1107                 //
1108                 // What we care about when resolving a MemberName is consistency between DEFC and PTYPES.
1109                 // We do type alias (TA) checks on DEFC to ensure that. DEFC is not known until the JVM
1110                 // finishes the resolution, so do TA checks right after MHN.resolve() is over.
1111                 //
1112                 // All parameters passed by a caller are checked against MH type (PTYPES) on every invocation,
1113                 // so it is safe to call a MH from any context.
1114                 //
1115                 // REFC view on PTYPES doesn't matter, since it is used only as a starting point for resolution and doesn't
1116                 // participate in method selection.
1117                 m = MethodHandleNatives.resolve(m, lookupClass, speculativeResolve);
1118                 if (m == null && speculativeResolve) {
1119                     return null;
1120                 }
1121                 m.checkForTypeAlias(m.getDeclaringClass());
1122                 m.resolution = null;
1123             } catch (ClassNotFoundException | LinkageError ex) {
1124                 // JVM reports that the "bytecode behavior" would get an error
1125                 assert(!m.isResolved());
1126                 m.resolution = ex;
1127                 return m;
1128             }
1129             assert(m.referenceKindIsConsistent());
1130             m.initResolved(true);
1131             assert(m.vminfoIsConsistent());
1132             return m;
1133         }
1134         /** Produce a resolved version of the given member.
1135          *  Super types are searched (for inherited members) if {@code searchSupers} is true.
1136          *  Access checking is performed on behalf of the given {@code lookupClass}.
1137          *  If lookup fails or access is not permitted, a {@linkplain ReflectiveOperationException} is thrown.
1138          *  Otherwise a fresh copy of the given member is returned, with modifier bits filled in.
