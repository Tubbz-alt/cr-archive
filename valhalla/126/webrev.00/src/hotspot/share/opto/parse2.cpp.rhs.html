  55 Node* Parse::record_profile_for_speculation_at_array_load(Node* ld) {
  56   // Feed unused profile data to type speculation
  57   if (UseTypeSpeculation && UseArrayLoadStoreProfile) {
  58     ciKlass* array_type = NULL;
  59     ciKlass* element_type = NULL;
  60     ProfilePtrKind element_ptr = ProfileMaybeNull;
  61     bool flat_array = true;
  62     bool null_free_array = true;
  63     method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
  64     if (element_type != NULL || element_ptr != ProfileMaybeNull) {
  65       ld = record_profile_for_speculation(ld, element_type, element_ptr);
  66     }
  67   }
  68   return ld;
  69 }

  72 //---------------------------------array_load----------------------------------
  73 void Parse::array_load(BasicType bt) {
  74   const Type* elemtype = Type::TOP;
  75   Node* adr = array_addressing(bt, 0, elemtype);
  76   if (stopped())  return;     // guaranteed null or range check
  77 
  78   Node* idx = pop();
  79   Node* ary = pop();
  80 
  81   // Handle inline type arrays
  82   const TypeOopPtr* elemptr = elemtype->make_oopptr();
  83   const TypeAryPtr* ary_t = _gvn.type(ary)->is_aryptr();
  84   if (elemtype->isa_inlinetype() != NULL) {
  85     C->set_flattened_accesses();
  86     // Load from flattened inline type array
  87     Node* vt = InlineTypeNode::make_from_flattened(this, elemtype->inline_klass(), ary, adr);
  88     push(vt);
  89     return;
  90   } else if (elemptr != NULL && elemptr->is_inlinetypeptr() && !elemptr->maybe_null()) {
  91     // Load from non-flattened inline type array (elements can never be null)
  92     bt = T_INLINE_TYPE;
  93   } else if (!ary_t->is_not_flat()) {
  94     // Cannot statically determine if array is flattened, emit runtime check
  95     assert(UseFlatArray && is_reference_type(bt) && elemptr->can_be_inline_type() && !ary_t->klass_is_exact() && !ary_t->is_not_null_free() &&
  96            (!elemptr->is_inlinetypeptr() || elemptr->inline_klass()->flatten_array()), "array can't be flattened");
  97     IdealKit ideal(this);
  98     IdealVariable res(ideal);
  99     ideal.declarations_done();
 100     ideal.if_then(is_non_flattened_array(ary)); {
 101       // non-flattened
 102       assert(ideal.ctrl()->in(0)->as_If()->is_non_flattened_array_check(&_gvn), "Should be found");
 103       sync_kit(ideal);
 104       const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 105       Node* ld = access_load_at(ary, adr, adr_type, elemptr, bt,
 106                                 IN_HEAP | IS_ARRAY | C2_CONTROL_DEPENDENT_LOAD);
 107       ideal.sync_kit(this);
 108       ideal.set(res, ld);
 109     } ideal.else_(); {
 110       // flattened
 111       sync_kit(ideal);
 112       if (elemptr->is_inlinetypeptr()) {
 113         // Element type is known, cast and load from flattened representation
 114         ciInlineKlass* vk = elemptr->inline_klass();
 115         assert(vk->flatten_array() && elemptr->maybe_null(), "never/always flat - should be optimized");
 116         ciArrayKlass* array_klass = ciArrayKlass::make(vk);
 117         const TypeAryPtr* arytype = TypeOopPtr::make_from_klass(array_klass)->isa_aryptr();
 118         Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, arytype));
 119         Node* casted_adr = array_element_address(cast, idx, T_INLINE_TYPE, ary_t->size(), control());
 120         // Re-execute flattened array load if buffering triggers deoptimization
 121         PreserveReexecuteState preexecs(this);
 122         jvms()->set_should_reexecute(true);
 123         inc_sp(2);
 124         Node* vt = InlineTypeNode::make_from_flattened(this, vk, cast, casted_adr)->buffer(this, false);
 125         ideal.set(res, vt);
 126         ideal.sync_kit(this);
 127       } else {
 128         // Element type is unknown, emit runtime call
 129         Node* kls = load_object_klass(ary);
 130         Node* k_adr = basic_plus_adr(kls, in_bytes(ArrayKlass::element_klass_offset()));
 131         Node* elem_klass = _gvn.transform(LoadKlassNode::make(_gvn, NULL, immutable_memory(), k_adr, TypeInstPtr::KLASS));
 132         Node* obj_size  = NULL;
 133         kill_dead_locals();
 134         // Re-execute flattened array load if buffering triggers deoptimization
 135         PreserveReexecuteState preexecs(this);
 136         jvms()->set_bci(_bci);
 137         jvms()->set_should_reexecute(true);
 138         inc_sp(2);
 139         Node* alloc_obj = new_instance(elem_klass, NULL, &obj_size, /*deoptimize_on_exception=*/true);
 140 
 141         AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_obj, &_gvn);
 142         assert(alloc->maybe_set_complete(&_gvn), "");
 143         alloc->initialization()->set_complete_with_arraycopy();
 144 
 145         // This membar keeps this access to an unknown flattened array
 146         // correctly ordered with other unknown and known flattened
 147         // array accesses.
 148         insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::INLINES));
 149 
 150         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 151         // Unknown inline type might contain reference fields
 152         if (false && !bs->array_copy_requires_gc_barriers(false, T_OBJECT, false, BarrierSetC2::Parsing)) {
 153           // FIXME 8230656 also merge changes from 8238759 in
 154           int base_off = sizeof(instanceOopDesc);
 155           Node* dst_base = basic_plus_adr(alloc_obj, base_off);
 156           Node* countx = obj_size;
 157           countx = _gvn.transform(new SubXNode(countx, MakeConX(base_off)));
 158           countx = _gvn.transform(new URShiftXNode(countx, intcon(LogBytesPerLong)));
 159 
 160           assert(Klass::_lh_log2_element_size_shift == 0, "use shift in place");
 161           Node* lhp = basic_plus_adr(kls, in_bytes(Klass::layout_helper_offset()));
 162           Node* elem_shift = make_load(NULL, lhp, TypeInt::INT, T_INT, MemNode::unordered);
 163           uint header = arrayOopDesc::base_offset_in_bytes(T_INLINE_TYPE);
 164           Node* base  = basic_plus_adr(ary, header);
 165           idx = Compile::conv_I2X_index(&_gvn, idx, TypeInt::POS, control());
 166           Node* scale = _gvn.transform(new LShiftXNode(idx, elem_shift));
 167           Node* adr = basic_plus_adr(ary, base, scale);
 168 
 169           access_clone(adr, dst_base, countx, false);
 170         } else {
 171           ideal.sync_kit(this);
 172           ideal.make_leaf_call(OptoRuntime::load_unknown_inline_type(),
 173                                CAST_FROM_FN_PTR(address, OptoRuntime::load_unknown_inline),
 174                                "load_unknown_inline",
 175                                ary, idx, alloc_obj);
 176           sync_kit(ideal);
 177         }
 178 
 179         // This makes sure no other thread sees a partially initialized buffered inline type
 180         insert_mem_bar_volatile(Op_MemBarStoreStore, Compile::AliasIdxRaw, alloc->proj_out_or_null(AllocateNode::RawAddress));
 181 
 182         // Same as MemBarCPUOrder above: keep this unknown flattened
 183         // array access correctly ordered with other flattened array
 184         // access
 185         insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::INLINES));
 186 
 187         // Prevent any use of the newly allocated inline type before it is fully initialized
 188         alloc_obj = new CastPPNode(alloc_obj, _gvn.type(alloc_obj), true);
 189         alloc_obj->set_req(0, control());
 190         alloc_obj = _gvn.transform(alloc_obj);
 191 
 192         const Type* unknown_value = elemptr->is_instptr()->cast_to_flat_array();
 193         alloc_obj = _gvn.transform(new CheckCastPPNode(control(), alloc_obj, unknown_value));
 194 
 195         ideal.sync_kit(this);
 196         ideal.set(res, alloc_obj);
 197       }
 198     } ideal.end_if();
 199     sync_kit(ideal);
 200     Node* ld = _gvn.transform(ideal.value(res));
 201     ld = record_profile_for_speculation_at_array_load(ld);
 202     push_node(bt, ld);
 203     return;
 204   }
 205 
 206   if (elemtype == TypeInt::BOOL) {
 207     bt = T_BOOLEAN;
 208   }
 209   const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 210   Node* ld = access_load_at(ary, adr, adr_type, elemtype, bt,
 211                             IN_HEAP | IS_ARRAY | C2_CONTROL_DEPENDENT_LOAD);
 212   if (bt == T_INLINE_TYPE) {
 213     // Loading a non-flattened inline type from an array
 214     assert(!gvn().type(ld)->maybe_null(), "inline type array elements should never be null");
 215     if (elemptr->inline_klass()->is_scalarizable()) {
 216       ld = InlineTypeNode::make_from_oop(this, ld, elemptr->inline_klass());
 217     }
 218   }
 219   if (!ld->is_InlineType()) {
 220     ld = record_profile_for_speculation_at_array_load(ld);
 221   }
 222 
 223   push_node(bt, ld);
 224 }

 227 //--------------------------------array_store----------------------------------
 228 void Parse::array_store(BasicType bt) {
 229   const Type* elemtype = Type::TOP;
 230   Node* adr = array_addressing(bt, type2size[bt], elemtype);
 231   if (stopped())  return;     // guaranteed null or range check
 232   Node* cast_val = NULL;
 233   if (bt == T_OBJECT) {
 234     cast_val = array_store_check();
 235     if (stopped()) return;
 236   }
 237   Node* val = pop_node(bt); // Value to store
 238   Node* idx = pop();        // Index in the array
 239   Node* ary = pop();        // The array itself
 240 
 241   const TypeAryPtr* ary_t = _gvn.type(ary)->is_aryptr();
 242   const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 243 
 244   if (elemtype == TypeInt::BOOL) {
 245     bt = T_BOOLEAN;
 246   } else if (bt == T_OBJECT) {
 247     elemtype = elemtype->make_oopptr();
 248     const Type* tval = _gvn.type(cast_val);
 249     // We may have lost type information for 'val' here due to the casts
 250     // emitted by the array_store_check code (see JDK-6312651)
 251     // TODO Remove this code once JDK-6312651 is in.
 252     const Type* tval_init = _gvn.type(val);
 253     bool not_inline = !tval->isa_inlinetype() && (tval == TypePtr::NULL_PTR || !tval_init->is_oopptr()->can_be_inline_type() || !tval->is_oopptr()->can_be_inline_type());
 254     bool not_flattened = !UseFlatArray || not_inline || ((tval_init->is_inlinetypeptr() || tval_init->isa_inlinetype()) && !tval_init->inline_klass()->flatten_array());
 255 
 256     if (!ary_t->is_not_null_free() && not_inline && (!tval->maybe_null() || !tval_init->maybe_null())) {
 257       // Storing a non-inline type, mark array as not null-free (-> not flat).
 258       // This is only legal for non-null stores because the array_store_check always passes for null.
 259       // Null stores are handled in GraphKit::gen_inline_array_null_guard().
 260       ary_t = ary_t->cast_to_not_null_free();
 261       Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, ary_t));
 262       replace_in_map(ary, cast);
 263       ary = cast;
 264     } else if (!ary_t->is_not_flat() && not_flattened) {
 265       // Storing a non-flattened value, mark array as not flat.
 266       ary_t = ary_t->cast_to_not_flat();
 267       Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, ary_t));
 268       replace_in_map(ary, cast);
 269       ary = cast;
 270     }
 271 
 272     if (ary_t->elem()->isa_inlinetype() != NULL) {
 273       // Store to flattened inline type array
 274       C->set_flattened_accesses();
 275       if (!cast_val->is_InlineType()) {
 276         inc_sp(3);
 277         cast_val = null_check(cast_val);
 278         if (stopped()) return;
 279         dec_sp(3);
 280         cast_val = InlineTypeNode::make_from_oop(this, cast_val, ary_t->elem()->inline_klass());
 281       }
 282       // Re-execute flattened array store if buffering triggers deoptimization
 283       PreserveReexecuteState preexecs(this);
 284       inc_sp(3);
 285       jvms()->set_should_reexecute(true);
 286       cast_val->as_InlineType()->store_flattened(this, ary, adr, NULL, 0, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 287       return;
 288     } else if (elemtype->is_inlinetypeptr() && !elemtype->maybe_null()) {
 289       // Store to non-flattened inline type array (elements can never be null)
 290       if (!cast_val->is_InlineType() && tval->maybe_null()) {
 291         inc_sp(3);
 292         cast_val = null_check(cast_val);
 293         if (stopped()) return;
 294         dec_sp(3);
 295       }
 296     } else if (!ary_t->is_not_flat()) {
 297       // Array might be flattened, emit runtime checks
 298       assert(UseFlatArray && !not_flattened && elemtype->is_oopptr()->can_be_inline_type() &&
 299              !ary_t->klass_is_exact() && !ary_t->is_not_null_free(), "array can't be flattened");
 300       IdealKit ideal(this);
 301       ideal.if_then(is_non_flattened_array(ary)); {
 302         // non-flattened
 303         assert(ideal.ctrl()->in(0)->as_If()->is_non_flattened_array_check(&_gvn), "Should be found");
 304         sync_kit(ideal);
 305         gen_inline_array_null_guard(ary, cast_val, 3);
 306         inc_sp(3);
 307         access_store_at(ary, adr, adr_type, cast_val, elemtype, bt, MO_UNORDERED | IN_HEAP | IS_ARRAY, false);
 308         dec_sp(3);
 309         ideal.sync_kit(this);
 310       } ideal.else_(); {
 311         Node* val = cast_val;
 312         // flattened
 313         if (!val->is_InlineType() && tval->maybe_null()) {
 314           // Add null check
 315           sync_kit(ideal);
 316           Node* null_ctl = top();
 317           val = null_check_oop(val, &null_ctl);
 318           if (null_ctl != top()) {
 319             PreserveJVMState pjvms(this);
 320             inc_sp(3);
 321             set_control(null_ctl);
 322             uncommon_trap(Deoptimization::Reason_null_check, Deoptimization::Action_none);
 323             dec_sp(3);
 324           }
 325           ideal.sync_kit(this);
 326         }
 327         // Try to determine the inline klass
 328         ciInlineKlass* vk = NULL;
 329         if (tval->isa_inlinetype() || tval->is_inlinetypeptr()) {
 330           vk = tval->inline_klass();
 331         } else if (tval_init->isa_inlinetype() || tval_init->is_inlinetypeptr()) {
 332           vk = tval_init->inline_klass();
 333         } else if (elemtype->is_inlinetypeptr()) {
 334           vk = elemtype->inline_klass();
 335         }
 336         Node* casted_ary = ary;
 337         if (vk != NULL && !stopped()) {
 338           // Element type is known, cast and store to flattened representation
 339           sync_kit(ideal);
 340           assert(vk->flatten_array() && elemtype->maybe_null(), "never/always flat - should be optimized");
 341           ciArrayKlass* array_klass = ciArrayKlass::make(vk);
 342           const TypeAryPtr* arytype = TypeOopPtr::make_from_klass(array_klass)->isa_aryptr();
 343           casted_ary = _gvn.transform(new CheckCastPPNode(control(), casted_ary, arytype));
 344           Node* casted_adr = array_element_address(casted_ary, idx, T_OBJECT, arytype->size(), control());
 345           if (!val->is_InlineType()) {
 346             assert(!gvn().type(val)->maybe_null(), "inline type array elements should never be null");
 347             val = InlineTypeNode::make_from_oop(this, val, vk);
 348           }
 349           // Re-execute flattened array store if buffering triggers deoptimization
 350           PreserveReexecuteState preexecs(this);
 351           inc_sp(3);
 352           jvms()->set_should_reexecute(true);
 353           val->as_InlineType()->store_flattened(this, casted_ary, casted_adr, NULL, 0, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 354           ideal.sync_kit(this);
 355         } else if (!ideal.ctrl()->is_top()) {
 356           // Element type is unknown, emit runtime call
 357           sync_kit(ideal);
 358 
 359           // This membar keeps this access to an unknown flattened
 360           // array correctly ordered with other unknown and known
 361           // flattened array accesses.
 362           insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::INLINES));
 363           ideal.sync_kit(this);
 364 
 365           ideal.make_leaf_call(OptoRuntime::store_unknown_inline_type(),
 366                                CAST_FROM_FN_PTR(address, OptoRuntime::store_unknown_inline),
 367                                "store_unknown_inline",
 368                                val, casted_ary, idx);
 369 
 370           sync_kit(ideal);
 371           // Same as MemBarCPUOrder above: keep this unknown
 372           // flattened array access correctly ordered with other
 373           // flattened array accesses.
 374           insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::INLINES));
 375           ideal.sync_kit(this);
 376         }
 377       }
 378       ideal.end_if();
 379       sync_kit(ideal);
 380       return;
 381     } else if (!ary_t->is_not_null_free()) {
 382       // Array is not flattened but may be null free
 383       assert(elemtype->is_oopptr()->can_be_inline_type() && !ary_t->klass_is_exact(), "array can't be null free");
 384       ary = gen_inline_array_null_guard(ary, cast_val, 3, true);
 385     }
 386   }
 387   inc_sp(3);
 388   access_store_at(ary, adr, adr_type, val, elemtype, bt, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 389   dec_sp(3);
 390 }

 393 //------------------------------array_addressing-------------------------------
 394 // Pull array and index from the stack.  Compute pointer-to-element.
 395 Node* Parse::array_addressing(BasicType type, int vals, const Type*& elemtype) {
 396   Node *idx   = peek(0+vals);   // Get from stack without popping
 397   Node *ary   = peek(1+vals);   // in case of exception
 398 
 399   // Null check the array base, with correct stack contents
 400   ary = null_check(ary, T_ARRAY);
 401   // Compile-time detect of null-exception?
 402   if (stopped())  return top();
 403 
 404   const TypeAryPtr* arytype  = _gvn.type(ary)->is_aryptr();
 405   const TypeInt*    sizetype = arytype->size();
 406   elemtype = arytype->elem();
 407 
 408   if (UseUniqueSubclasses) {
 409     const Type* el = elemtype->make_ptr();
 410     if (el && el->isa_instptr()) {
 411       const TypeInstPtr* toop = el->is_instptr();
 412       if (toop->klass()->as_instance_klass()->unique_concrete_subklass()) {
 413         // If we load from "AbstractClass[]" we must see "ConcreteSubClass".
 414         const Type* subklass = Type::get_const_type(toop->klass());
 415         elemtype = subklass->join_speculative(el);
 416       }
 417     }
 418   }
 419 
 420   // Check for big class initializers with all constant offsets
 421   // feeding into a known-size array.
 422   const TypeInt* idxtype = _gvn.type(idx)->is_int();
 423   // See if the highest idx value is less than the lowest array bound,
 424   // and if the idx value cannot be negative:
 425   bool need_range_check = true;
 426   if (idxtype->_hi < sizetype->_lo && idxtype->_lo >= 0) {
 427     need_range_check = false;
 428     if (C->log() != NULL)   C->log()->elem("observe that='!need_range_check'");
 429   }
 430 
 431   ciKlass * arytype_klass = arytype->klass();
 432   if ((arytype_klass != NULL) && (!arytype_klass->is_loaded())) {
 433     // Only fails for some -Xcomp runs
 434     // The class is unloaded.  We have to run this bytecode in the interpreter.
 435     uncommon_trap(Deoptimization::Reason_unloaded,
 436                   Deoptimization::Action_reinterpret,
 437                   arytype->klass(), "!loaded array");
 438     return top();
 439   }
 440 
 441   // Do the range check
 442   if (GenerateRangeChecks && need_range_check) {
 443     Node* tst;
 444     if (sizetype->_hi <= 0) {
 445       // The greatest array bound is negative, so we can conclude that we're
 446       // compiling unreachable code, but the unsigned compare trick used below
 447       // only works with non-negative lengths.  Instead, hack "tst" to be zero so
 448       // the uncommon_trap path will always be taken.
 449       tst = _gvn.intcon(0);
 450     } else {
 451       // Range is constant in array-oop, so we can use the original state of mem
 452       Node* len = load_array_length(ary);
 453 
 454       // Test length vs index (standard trick using unsigned compare)
 455       Node* chk = _gvn.transform( new CmpUNode(idx, len) );
 456       BoolTest::mask btest = BoolTest::lt;
 457       tst = _gvn.transform( new BoolNode(chk, btest) );
 458     }
 459     RangeCheckNode* rc = new RangeCheckNode(control(), tst, PROB_MAX, COUNT_UNKNOWN);
 460     _gvn.set_type(rc, rc->Value(&_gvn));
 461     if (!tst->is_Con()) {
 462       record_for_igvn(rc);
 463     }
 464     set_control(_gvn.transform(new IfTrueNode(rc)));
 465     // Branch to failure if out of bounds
 466     {
 467       PreserveJVMState pjvms(this);
 468       set_control(_gvn.transform(new IfFalseNode(rc)));
 469       if (C->allow_range_check_smearing()) {
 470         // Do not use builtin_throw, since range checks are sometimes
 471         // made more stringent by an optimistic transformation.
 472         // This creates "tentative" range checks at this point,
 473         // which are not guaranteed to throw exceptions.
 474         // See IfNode::Ideal, is_range_check, adjust_check.
 475         uncommon_trap(Deoptimization::Reason_range_check,
 476                       Deoptimization::Action_make_not_entrant,
 477                       NULL, "range_check");
 478       } else {
 479         // If we have already recompiled with the range-check-widening
 480         // heroic optimization turned off, then we must really be throwing
 481         // range check exceptions.
 482         builtin_throw(Deoptimization::Reason_range_check, idx);
 483       }
 484     }
 485   }
 486   // Check for always knowing you are throwing a range-check exception
 487   if (stopped())  return top();
 488 
 489   // This could be an access to an inline type array. We can't tell if it's
 490   // flat or not. Speculating it's not leads to a much simpler graph
 491   // shape. Check profiling.
 492   // For aastore, by the time we're here, the array store check should
 493   // have already taken advantage of profiling to cast the array to an
 494   // exact type reported by profiling
 495   const TypeOopPtr* elemptr = elemtype->make_oopptr();
 496   if (elemtype->isa_inlinetype() == NULL &&
 497       (elemptr == NULL || !elemptr->is_inlinetypeptr() || elemptr->maybe_null()) &&
 498       !arytype->is_not_flat()) {
 499     assert(is_reference_type(type), "Only references");
 500     // First check the speculative type
 501     Deoptimization::DeoptReason reason = Deoptimization::Reason_speculate_class_check;
 502     ciKlass* array_type = arytype->speculative_type();
 503     if (too_many_traps_or_recompiles(reason) || array_type == NULL) {
 504       // No speculative type, check profile data at this bci
 505       array_type = NULL;
 506       reason = Deoptimization::Reason_class_check;
 507       if (UseArrayLoadStoreProfile && !too_many_traps_or_recompiles(reason)) {
 508         ciKlass* element_type = NULL;
 509         ProfilePtrKind element_ptr = ProfileMaybeNull;
 510         bool flat_array = true;
 511         bool null_free_array = true;
 512         method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 513       }
 514     }
 515     if (array_type != NULL) {
 516       // Speculate that this array has the exact type reported by profile data
 517       Node* better_ary = NULL;
 518       Node* slow_ctl = type_check_receiver(ary, array_type, 1.0, &better_ary);
 519       { PreserveJVMState pjvms(this);
 520         set_control(slow_ctl);
 521         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 522       }
 523       replace_in_map(ary, better_ary);
 524       ary = better_ary;
 525       arytype  = _gvn.type(ary)->is_aryptr();
 526       elemtype = arytype->elem();
 527     }
 528   } else if (UseTypeSpeculation && UseArrayLoadStoreProfile) {
 529     // No need to speculate: feed profile data at this bci for the
 530     // array to type speculation
 531     ciKlass* array_type = NULL;
 532     ciKlass* element_type = NULL;
 533     ProfilePtrKind element_ptr = ProfileMaybeNull;
 534     bool flat_array = true;
 535     bool null_free_array = true;
 536     method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 537     if (array_type != NULL) {
 538       record_profile_for_speculation(ary, array_type, ProfileMaybeNull);
 539     }
 540   }
 541 
 542   // We have no exact array type from profile data. Check profile data
 543   // for a non null free or non flat array. Non null free implies non
 544   // flat so check this one first. Speculating on a non null free
 545   // array doesn't help aaload but could be profitable for a
 546   // subsequent aastore.
 547   elemptr = elemtype->make_oopptr();
 548   if (!arytype->is_not_null_free() &&
 549       elemtype->isa_inlinetype() == NULL &&
 550       (elemptr == NULL || !elemptr->is_inlinetypeptr()) &&
 551       UseArrayLoadStoreProfile) {
 552     assert(is_reference_type(type), "");
 553     bool null_free_array = true;
 554     Deoptimization::DeoptReason reason = Deoptimization::Reason_none;
 555     if (arytype->speculative() != NULL &&
 556         arytype->speculative()->is_aryptr()->is_not_null_free() &&
 557         !too_many_traps_or_recompiles(Deoptimization::Reason_speculate_class_check)) {
 558       null_free_array = false;
 559       reason = Deoptimization::Reason_speculate_class_check;
 560     } else if (!too_many_traps_or_recompiles(Deoptimization::Reason_class_check)) {
 561       ciKlass* array_type = NULL;
 562       ciKlass* element_type = NULL;
 563       ProfilePtrKind element_ptr = ProfileMaybeNull;
 564       bool flat_array = true;
 565       method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 566       reason = Deoptimization::Reason_class_check;
 567     }
 568     if (!null_free_array) {
 569       { // Deoptimize if null-free array
 570         BuildCutout unless(this, is_nullable_array(ary), PROB_MAX);
 571         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 572       }
 573       Node* better_ary = _gvn.transform(new CheckCastPPNode(control(), ary, arytype->cast_to_not_null_free()));
 574       replace_in_map(ary, better_ary);
 575       ary = better_ary;
 576       arytype = _gvn.type(ary)->is_aryptr();
 577     }
 578   }
 579 
 580   if (!arytype->is_not_flat() && elemtype->isa_inlinetype() == NULL) {
 581     assert(is_reference_type(type), "");
 582     bool flat_array = true;
 583     Deoptimization::DeoptReason reason = Deoptimization::Reason_none;
 584     if (arytype->speculative() != NULL &&
 585         arytype->speculative()->is_aryptr()->is_not_flat() &&
 586         !too_many_traps_or_recompiles(Deoptimization::Reason_speculate_class_check)) {
 587       flat_array = false;
 588       reason = Deoptimization::Reason_speculate_class_check;
 589     } else if (UseArrayLoadStoreProfile && !too_many_traps_or_recompiles(reason)) {
 590       ciKlass* array_type = NULL;
 591       ciKlass* element_type = NULL;
 592       ProfilePtrKind element_ptr = ProfileMaybeNull;
 593       bool null_free_array = true;
 594       method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 595       reason = Deoptimization::Reason_class_check;
 596     }
 597     if (!flat_array) {
 598       { // Deoptimize if flat array
 599         BuildCutout unless(this, is_non_flattened_array(ary), PROB_MAX);
 600         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 601       }
 602       Node* better_ary = _gvn.transform(new CheckCastPPNode(control(), ary, arytype->cast_to_not_flat()));
 603       replace_in_map(ary, better_ary);
 604       ary = better_ary;
 605       arytype = _gvn.type(ary)->is_aryptr();
 606     }
 607   }
 608 
 609   // Make array address computation control dependent to prevent it
 610   // from floating above the range check during loop optimizations.
 611   Node* ptr = array_element_address(ary, idx, type, sizetype, control());
 612   assert(ptr != top(), "top should go hand-in-hand with stopped");
 613 
 614   return ptr;
 615 }

 718 class SwitchRange : public StackObj {
 719   // a range of integers coupled with a bci destination
 720   jint _lo;                     // inclusive lower limit
 721   jint _hi;                     // inclusive upper limit
 722   int _dest;
 723   int _table_index;             // index into method data table
 724   float _cnt;                   // how many times this range was hit according to profiling
 725 
 726 public:
 727   jint lo() const              { return _lo;   }
 728   jint hi() const              { return _hi;   }
 729   int  dest() const            { return _dest; }
 730   int  table_index() const     { return _table_index; }
 731   bool is_singleton() const    { return _lo == _hi; }
 732   float cnt() const            { return _cnt; }
 733 
 734   void setRange(jint lo, jint hi, int dest, int table_index, float cnt) {
 735     assert(lo <= hi, "must be a non-empty range");
 736     _lo = lo, _hi = hi; _dest = dest; _table_index = table_index; _cnt = cnt;
 737     assert(_cnt >= 0, "");
 738   }
 739   bool adjoinRange(jint lo, jint hi, int dest, int table_index, float cnt, bool trim_ranges) {
 740     assert(lo <= hi, "must be a non-empty range");
 741     if (lo == _hi+1 && table_index == _table_index) {
 742       // see merge_ranges() comment below
 743       if (trim_ranges) {
 744         if (cnt == 0) {
 745           if (_cnt != 0) {
 746             return false;
 747           }
 748           if (dest != _dest) {
 749             _dest = never_reached;
 750           }
 751         } else {
 752           if (_cnt == 0) {
 753             return false;
 754           }
 755           if (dest != _dest) {
 756             return false;
 757           }
 758         }
 759       } else {
 760         if (dest != _dest) {
 761           return false;
 762         }
 763       }
 764       _hi = hi;
 765       _cnt += cnt;
 766       return true;
 767     }
 768     return false;
 769   }
 770 
 771   void set (jint value, int dest, int table_index, float cnt) {
 772     setRange(value, value, dest, table_index, cnt);
 773   }
 774   bool adjoin(jint value, int dest, int table_index, float cnt, bool trim_ranges) {
 775     return adjoinRange(value, value, dest, table_index, cnt, trim_ranges);
 776   }
 777   bool adjoin(SwitchRange& other) {
 778     return adjoinRange(other._lo, other._hi, other._dest, other._table_index, other._cnt, false);
 779   }

 784     else if (lo() == min_jint)
 785       tty->print(" {..%d}=>%d (cnt=%f)", hi(), dest(), cnt());
 786     else if (hi() == max_jint)
 787       tty->print(" {%d..}=>%d (cnt=%f)", lo(), dest(), cnt());
 788     else
 789       tty->print(" {%d..%d}=>%d (cnt=%f)", lo(), hi(), dest(), cnt());
 790   }

 793 // We try to minimize the number of ranges and the size of the taken
 794 // ones using profiling data. When ranges are created,
 795 // SwitchRange::adjoinRange() only allows 2 adjoining ranges to merge
 796 // if both were never hit or both were hit to build longer unreached
 797 // ranges. Here, we now merge adjoining ranges with the same
 798 // destination and finally set destination of unreached ranges to the
 799 // special value never_reached because it can help minimize the number
 800 // of tests that are necessary.
 801 //
 802 // For instance:
 803 // [0, 1] to target1 sometimes taken
 804 // [1, 2] to target1 never taken
 805 // [2, 3] to target2 never taken
 806 // would lead to:
 807 // [0, 1] to target1 sometimes taken
 808 // [1, 3] never taken
 809 //
 810 // (first 2 ranges to target1 are not merged)
 811 static void merge_ranges(SwitchRange* ranges, int& rp) {
 812   if (rp == 0) {
 813     return;
 814   }
 815   int shift = 0;
 816   for (int j = 0; j < rp; j++) {
 817     SwitchRange& r1 = ranges[j-shift];
 818     SwitchRange& r2 = ranges[j+1];
 819     if (r1.adjoin(r2)) {
 820       shift++;
 821     } else if (shift > 0) {
 822       ranges[j+1-shift] = r2;
 823     }
 824   }
 825   rp -= shift;
 826   for (int j = 0; j <= rp; j++) {
 827     SwitchRange& r = ranges[j];
 828     if (r.cnt() == 0 && r.dest() != never_reached) {
 829       r.setRange(r.lo(), r.hi(), never_reached, r.table_index(), r.cnt());
 830     }
 831   }
 832 }

 834 //-------------------------------do_tableswitch--------------------------------
 835 void Parse::do_tableswitch() {
 836   Node* lookup = pop();
 837   // Get information about tableswitch
 838   int default_dest = iter().get_dest_table(0);
 839   int lo_index     = iter().get_int_table(1);
 840   int hi_index     = iter().get_int_table(2);
 841   int len          = hi_index - lo_index + 1;
 842 
 843   if (len < 1) {
 844     // If this is a backward branch, add safepoint
 845     maybe_add_safepoint(default_dest);
 846     merge(default_dest);
 847     return;
 848   }
 849 
 850   ciMethodData* methodData = method()->method_data();
 851   ciMultiBranchData* profile = NULL;
 852   if (methodData->is_mature() && UseSwitchProfiling) {
 853     ciProfileData* data = methodData->bci_to_data(bci());
 854     if (data != NULL && data->is_MultiBranchData()) {
 855       profile = (ciMultiBranchData*)data;
 856     }
 857   }
 858   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
 859 
 860   // generate decision tree, using trichotomy when possible
 861   int rnum = len+2;
 862   bool makes_backward_branch = false;
 863   SwitchRange* ranges = NEW_RESOURCE_ARRAY(SwitchRange, rnum);
 864   int rp = -1;
 865   if (lo_index != min_jint) {
 866     uint cnt = 1;
 867     if (profile != NULL) {
 868       cnt = profile->default_count() / (hi_index != max_jint ? 2 : 1);
 869     }
 870     ranges[++rp].setRange(min_jint, lo_index-1, default_dest, NullTableIndex, cnt);
 871   }
 872   for (int j = 0; j < len; j++) {
 873     jint match_int = lo_index+j;
 874     int  dest      = iter().get_dest_table(j+3);
 875     makes_backward_branch |= (dest <= bci());
 876     int  table_index = method_data_update() ? j : NullTableIndex;
 877     uint cnt = 1;
 878     if (profile != NULL) {
 879       cnt = profile->count_at(j);
 880     }
 881     if (rp < 0 || !ranges[rp].adjoin(match_int, dest, table_index, cnt, trim_ranges)) {
 882       ranges[++rp].set(match_int, dest, table_index, cnt);
 883     }
 884   }
 885   jint highest = lo_index+(len-1);
 886   assert(ranges[rp].hi() == highest, "");
 887   if (highest != max_jint) {
 888     uint cnt = 1;
 889     if (profile != NULL) {
 890       cnt = profile->default_count() / (lo_index != min_jint ? 2 : 1);
 891     }
 892     if (!ranges[rp].adjoinRange(highest+1, max_jint, default_dest, NullTableIndex, cnt, trim_ranges)) {
 893       ranges[++rp].setRange(highest+1, max_jint, default_dest, NullTableIndex, cnt);
 894     }
 895   }
 896   assert(rp < len+2, "not too many ranges");
 897 
 898   if (trim_ranges) {
 899     merge_ranges(ranges, rp);
 900   }
 901 
 902   // Safepoint in case if backward branch observed
 903   if( makes_backward_branch && UseLoopSafepoints )
 904     add_safepoint();
 905 
 906   jump_switch_ranges(lookup, &ranges[0], &ranges[rp]);
 907 }

 910 //------------------------------do_lookupswitch--------------------------------
 911 void Parse::do_lookupswitch() {
 912   Node *lookup = pop();         // lookup value
 913   // Get information about lookupswitch
 914   int default_dest = iter().get_dest_table(0);
 915   int len          = iter().get_int_table(1);
 916 
 917   if (len < 1) {    // If this is a backward branch, add safepoint
 918     maybe_add_safepoint(default_dest);
 919     merge(default_dest);
 920     return;
 921   }
 922 
 923   ciMethodData* methodData = method()->method_data();
 924   ciMultiBranchData* profile = NULL;
 925   if (methodData->is_mature() && UseSwitchProfiling) {
 926     ciProfileData* data = methodData->bci_to_data(bci());
 927     if (data != NULL && data->is_MultiBranchData()) {
 928       profile = (ciMultiBranchData*)data;
 929     }
 930   }
 931   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
 932 
 933   // generate decision tree, using trichotomy when possible
 934   jint* table = NEW_RESOURCE_ARRAY(jint, len*3);
 935   {
 936     for (int j = 0; j < len; j++) {
 937       table[3*j+0] = iter().get_int_table(2+2*j);
 938       table[3*j+1] = iter().get_dest_table(2+2*j+1);
 939       table[3*j+2] = profile == NULL ? 1 : profile->count_at(j);
 940     }
 941     qsort(table, len, 3*sizeof(table[0]), jint_cmp);
 942   }
 943 
 944   float defaults = 0;
 945   jint prev = min_jint;
 946   for (int j = 0; j < len; j++) {
 947     jint match_int = table[3*j+0];
 948     if (match_int != prev) {
 949       defaults += (float)match_int - prev;
 950     }
 951     prev = match_int+1;
 952   }
 953   if (prev-1 != max_jint) {
 954     defaults += (float)max_jint - prev + 1;
 955   }
 956   float default_cnt = 1;
 957   if (profile != NULL) {
 958     default_cnt = profile->default_count()/defaults;
 959   }
 960 
 961   int rnum = len*2+1;
 962   bool makes_backward_branch = false;
 963   SwitchRange* ranges = NEW_RESOURCE_ARRAY(SwitchRange, rnum);
 964   int rp = -1;
 965   for (int j = 0; j < len; j++) {
 966     jint match_int   = table[3*j+0];
 967     int  dest        = table[3*j+1];
 968     int  cnt         = table[3*j+2];
 969     int  next_lo     = rp < 0 ? min_jint : ranges[rp].hi()+1;
 970     int  table_index = method_data_update() ? j : NullTableIndex;
 971     makes_backward_branch |= (dest <= bci());
 972     float c = default_cnt * ((float)match_int - next_lo);
 973     if (match_int != next_lo && (rp < 0 || !ranges[rp].adjoinRange(next_lo, match_int-1, default_dest, NullTableIndex, c, trim_ranges))) {
 974       assert(default_dest != never_reached, "sentinel value for dead destinations");
 975       ranges[++rp].setRange(next_lo, match_int-1, default_dest, NullTableIndex, c);
 976     }
 977     if (rp < 0 || !ranges[rp].adjoin(match_int, dest, table_index, cnt, trim_ranges)) {
 978       assert(dest != never_reached, "sentinel value for dead destinations");
 979       ranges[++rp].set(match_int, dest, table_index, cnt);
 980     }
 981   }
 982   jint highest = table[3*(len-1)];
 983   assert(ranges[rp].hi() == highest, "");
 984   if (highest != max_jint &&
 985       !ranges[rp].adjoinRange(highest+1, max_jint, default_dest, NullTableIndex, default_cnt * ((float)max_jint - highest), trim_ranges)) {
 986     ranges[++rp].setRange(highest+1, max_jint, default_dest, NullTableIndex, default_cnt * ((float)max_jint - highest));
 987   }
 988   assert(rp < rnum, "not too many ranges");
 989 
 990   if (trim_ranges) {
 991     merge_ranges(ranges, rp);
 992   }
 993 
 994   // Safepoint in case backward branch observed
 995   if (makes_backward_branch && UseLoopSafepoints)
 996     add_safepoint();
 997 
 998   jump_switch_ranges(lookup, &ranges[0], &ranges[rp]);
 999 }

1050 static float compute_tree_cost(SwitchRange *lo, SwitchRange *hi, float total_cnt) {
1051   GrowableArray<SwitchRanges> tree;
1052   SwitchRanges root(lo, hi);
1053   tree.push(root);
1054 
1055   float cost = 0;
1056   do {
1057     SwitchRanges& r = *tree.adr_at(tree.length()-1);
1058     if (r._hi != r._lo) {
1059       if (r._mid == NULL) {
1060         float r_cnt = sum_of_cnts(r._lo, r._hi);
1061 
1062         if (r_cnt == 0) {
1063           tree.pop();
1064           cost = 0;
1065           continue;
1066         }
1067 
1068         SwitchRange* mid = NULL;
1069         mid = r._lo;
1070         for (float cnt = 0; ; ) {
1071           assert(mid <= r._hi, "out of bounds");
1072           cnt += mid->cnt();
1073           if (cnt > r_cnt / 2) {
1074             break;
1075           }
1076           mid++;
1077         }
1078         assert(mid <= r._hi, "out of bounds");
1079         r._mid = mid;
1080         r._cost = r_cnt / total_cnt;
1081       }
1082       r._cost += cost;
1083       if (r._state < SwitchRanges::LeftDone && r._mid > r._lo) {
1084         cost = 0;
1085         r._state = SwitchRanges::LeftDone;
1086         tree.push(SwitchRanges(r._lo, r._mid-1));
1087       } else if (r._state < SwitchRanges::RightDone) {
1088         cost = 0;
1089         r._state = SwitchRanges::RightDone;
1090         tree.push(SwitchRanges(r._mid == r._lo ? r._mid+1 : r._mid, r._hi));
1091       } else {
1092         tree.pop();
1093         cost = r._cost;
1094       }
1095     } else {
1096       tree.pop();
1097       cost = r._cost;
1098     }
1099   } while (tree.length() > 0);
1100 
1101 
1102   return cost;
1103 }

1105 // It sometimes pays off to test most common ranges before the binary search
1106 void Parse::linear_search_switch_ranges(Node* key_val, SwitchRange*& lo, SwitchRange*& hi) {
1107   uint nr = hi - lo + 1;
1108   float total_cnt = sum_of_cnts(lo, hi);
1109 
1110   float min = compute_tree_cost(lo, hi, total_cnt);
1111   float extra = 1;
1112   float sub = 0;
1113 
1114   SwitchRange* array1 = lo;
1115   SwitchRange* array2 = NEW_RESOURCE_ARRAY(SwitchRange, nr);
1116 
1117   SwitchRange* ranges = NULL;
1118 
1119   while (nr >= 2) {
1120     assert(lo == array1 || lo == array2, "one the 2 already allocated arrays");
1121     ranges = (lo == array1) ? array2 : array1;
1122 
1123     // Find highest frequency range
1124     SwitchRange* candidate = lo;
1125     for (SwitchRange* sr = lo+1; sr <= hi; sr++) {
1126       if (sr->cnt() > candidate->cnt()) {
1127         candidate = sr;
1128       }
1129     }
1130     SwitchRange most_freq = *candidate;
1131     if (most_freq.cnt() == 0) {
1132       break;
1133     }
1134 
1135     // Copy remaining ranges into another array
1136     int shift = 0;
1137     for (uint i = 0; i < nr; i++) {
1138       SwitchRange* sr = &lo[i];
1139       if (sr != candidate) {
1140         ranges[i-shift] = *sr;
1141       } else {
1142         shift++;
1143         if (i > 0 && i < nr-1) {
1144           SwitchRange prev = lo[i-1];
1145           prev.setRange(prev.lo(), sr->hi(), prev.dest(), prev.table_index(), prev.cnt());
1146           if (prev.adjoin(lo[i+1])) {
1147             shift++;
1148             i++;
1149           }
1150           ranges[i-shift] = prev;
1151         }
1152       }
1153     }
1154     nr -= shift;
1155 
1156     // Evaluate cost of testing the most common range and performing a
1157     // binary search on the other ranges
1158     float cost = extra + compute_tree_cost(&ranges[0], &ranges[nr-1], total_cnt);
1159     if (cost >= min) {
1160       break;
1161     }
1162     // swap arrays
1163     lo = &ranges[0];
1164     hi = &ranges[nr-1];
1165 
1166     // It pays off: emit the test for the most common range
1167     assert(most_freq.cnt() > 0, "must be taken");
1168     Node* val = _gvn.transform(new SubINode(key_val, _gvn.intcon(most_freq.lo())));
1169     Node* cmp = _gvn.transform(new CmpUNode(val, _gvn.intcon(most_freq.hi() - most_freq.lo())));
1170     Node* tst = _gvn.transform(new BoolNode(cmp, BoolTest::le));
1171     IfNode* iff = create_and_map_if(control(), tst, if_prob(most_freq.cnt(), total_cnt), if_cnt(most_freq.cnt()));
1172     jump_if_true_fork(iff, most_freq.dest(), most_freq.table_index(), false);
1173 
1174     sub += most_freq.cnt() / total_cnt;
1175     extra += 1 - sub;
1176     min = cost;
1177   }
1178 }

1180 //----------------------------create_jump_tables-------------------------------
1181 bool Parse::create_jump_tables(Node* key_val, SwitchRange* lo, SwitchRange* hi) {
1182   // Are jumptables enabled
1183   if (!UseJumpTables)  return false;
1184 
1185   // Are jumptables supported
1186   if (!Matcher::has_match_rule(Op_Jump))  return false;
1187 
1188   // Don't make jump table if profiling
1189   if (method_data_update())  return false;
1190 
1191   bool trim_ranges = !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
1192 
1193   // Decide if a guard is needed to lop off big ranges at either (or
1194   // both) end(s) of the input set. We'll call this the default target
1195   // even though we can't be sure that it is the true "default".
1196 
1197   bool needs_guard = false;
1198   int default_dest;
1199   int64_t total_outlier_size = 0;
1200   int64_t hi_size = ((int64_t)hi->hi()) - ((int64_t)hi->lo()) + 1;
1201   int64_t lo_size = ((int64_t)lo->hi()) - ((int64_t)lo->lo()) + 1;
1202 
1203   if (lo->dest() == hi->dest()) {
1204     total_outlier_size = hi_size + lo_size;
1205     default_dest = lo->dest();
1206   } else if (lo_size > hi_size) {
1207     total_outlier_size = lo_size;
1208     default_dest = lo->dest();
1209   } else {
1210     total_outlier_size = hi_size;
1211     default_dest = hi->dest();
1212   }
1213 
1214   float total = sum_of_cnts(lo, hi);
1215   float cost = compute_tree_cost(lo, hi, total);
1216 
1217   // If a guard test will eliminate very sparse end ranges, then
1218   // it is worth the cost of an extra jump.
1219   float trimmed_cnt = 0;
1220   if (total_outlier_size > (MaxJumpTableSparseness * 4)) {
1221     needs_guard = true;
1222     if (default_dest == lo->dest()) {
1223       trimmed_cnt += lo->cnt();
1224       lo++;
1225     }
1226     if (default_dest == hi->dest()) {
1227       trimmed_cnt += hi->cnt();
1228       hi--;
1229     }
1230   }
1231 
1232   // Find the total number of cases and ranges
1233   int64_t num_cases = ((int64_t)hi->hi()) - ((int64_t)lo->lo()) + 1;
1234   int num_range = hi - lo + 1;
1235 
1236   // Don't create table if: too large, too small, or too sparse.
1237   if (num_cases > MaxJumpTableSize)
1238     return false;
1239   if (UseSwitchProfiling) {
1240     // MinJumpTableSize is set so with a well balanced binary tree,
1241     // when the number of ranges is MinJumpTableSize, it's cheaper to
1242     // go through a JumpNode that a tree of IfNodes. Average cost of a
1243     // tree of IfNodes with MinJumpTableSize is
1244     // log2f(MinJumpTableSize) comparisons. So if the cost computed
1245     // from profile data is less than log2f(MinJumpTableSize) then
1246     // going with the binary search is cheaper.
1247     if (cost < log2f(MinJumpTableSize)) {
1248       return false;
1249     }
1250   } else {
1251     if (num_cases < MinJumpTableSize)
1252       return false;
1253   }
1254   if (num_cases > (MaxJumpTableSparseness * num_range))
1255     return false;
1256 
1257   // Normalize table lookups to zero
1258   int lowval = lo->lo();
1259   key_val = _gvn.transform( new SubINode(key_val, _gvn.intcon(lowval)) );
1260 
1261   // Generate a guard to protect against input keyvals that aren't
1262   // in the switch domain.
1263   if (needs_guard) {
1264     Node*   size = _gvn.intcon(num_cases);
1265     Node*   cmp = _gvn.transform(new CmpUNode(key_val, size));
1266     Node*   tst = _gvn.transform(new BoolNode(cmp, BoolTest::ge));
1267     IfNode* iff = create_and_map_if(control(), tst, if_prob(trimmed_cnt, total), if_cnt(trimmed_cnt));
1268     jump_if_true_fork(iff, default_dest, NullTableIndex, trim_ranges && trimmed_cnt == 0);
1269 
1270     total -= trimmed_cnt;
1271   }
1272 
1273   // Create an ideal node JumpTable that has projections
1274   // of all possible ranges for a switch statement
1275   // The key_val input must be converted to a pointer offset and scaled.
1276   // Compare Parse::array_addressing above.
1277 
1278   // Clean the 32-bit int into a real 64-bit offset.
1279   // Otherwise, the jint value 0 might turn into an offset of 0x0800000000.
1280   const TypeInt* ikeytype = TypeInt::make(0, num_cases, Type::WidenMin);
1281   // Make I2L conversion control dependent to prevent it from
1282   // floating above the range check during loop optimizations.
1283   key_val = C->conv_I2X_index(&_gvn, key_val, ikeytype, control());
1284 
1285   // Shift the value by wordsize so we have an index into the table, rather
1286   // than a switch value
1287   Node *shiftWord = _gvn.MakeConX(wordSize);
1288   key_val = _gvn.transform( new MulXNode( key_val, shiftWord));
1289 
1290   // Create the JumpNode
1291   Arena* arena = C->comp_arena();
1292   float* probs = (float*)arena->Amalloc(sizeof(float)*num_cases);
1293   int i = 0;
1294   if (total == 0) {
1295     for (SwitchRange* r = lo; r <= hi; r++) {
1296       for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1297         probs[i] = 1.0F / num_cases;
1298       }
1299     }
1300   } else {
1301     for (SwitchRange* r = lo; r <= hi; r++) {
1302       float prob = r->cnt()/total;
1303       for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1304         probs[i] = prob / (r->hi() - r->lo() + 1);
1305       }
1306     }
1307   }
1308 
1309   ciMethodData* methodData = method()->method_data();
1310   ciMultiBranchData* profile = NULL;
1311   if (methodData->is_mature()) {
1312     ciProfileData* data = methodData->bci_to_data(bci());
1313     if (data != NULL && data->is_MultiBranchData()) {
1314       profile = (ciMultiBranchData*)data;
1315     }
1316   }
1317 
1318   Node* jtn = _gvn.transform(new JumpNode(control(), key_val, num_cases, probs, profile == NULL ? COUNT_UNKNOWN : total));
1319 
1320   // These are the switch destinations hanging off the jumpnode
1321   i = 0;
1322   for (SwitchRange* r = lo; r <= hi; r++) {
1323     for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1324       Node* input = _gvn.transform(new JumpProjNode(jtn, i, r->dest(), (int)(j - lowval)));
1325       {
1326         PreserveJVMState pjvms(this);
1327         set_control(input);
1328         jump_if_always_fork(r->dest(), r->table_index(), trim_ranges && r->cnt() == 0);
1329       }
1330     }
1331   }
1332   assert(i == num_cases, "miscount of cases");
1333   stop_and_kill_map();  // no more uses for this JVMS
1334   return true;
1335 }

1337 //----------------------------jump_switch_ranges-------------------------------
1338 void Parse::jump_switch_ranges(Node* key_val, SwitchRange *lo, SwitchRange *hi, int switch_depth) {
1339   Block* switch_block = block();
1340   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
1341 
1342   if (switch_depth == 0) {
1343     // Do special processing for the top-level call.
1344     assert(lo->lo() == min_jint, "initial range must exhaust Type::INT");
1345     assert(hi->hi() == max_jint, "initial range must exhaust Type::INT");
1346 
1347     // Decrement pred-numbers for the unique set of nodes.
1348 #ifdef ASSERT
1349     if (!trim_ranges) {
1350       // Ensure that the block's successors are a (duplicate-free) set.
1351       int successors_counted = 0;  // block occurrences in [hi..lo]
1352       int unique_successors = switch_block->num_successors();
1353       for (int i = 0; i < unique_successors; i++) {
1354         Block* target = switch_block->successor_at(i);
1355 
1356         // Check that the set of successors is the same in both places.
1357         int successors_found = 0;
1358         for (SwitchRange* p = lo; p <= hi; p++) {
1359           if (p->dest() == target->start())  successors_found++;
1360         }
1361         assert(successors_found > 0, "successor must be known");
1362         successors_counted += successors_found;
1363       }
1364       assert(successors_counted == (hi-lo)+1, "no unexpected successors");
1365     }
1366 #endif
1367 
1368     // Maybe prune the inputs, based on the type of key_val.
1369     jint min_val = min_jint;
1370     jint max_val = max_jint;
1371     const TypeInt* ti = key_val->bottom_type()->isa_int();
1372     if (ti != NULL) {
1373       min_val = ti->_lo;
1374       max_val = ti->_hi;
1375       assert(min_val <= max_val, "invalid int type");
1376     }
1377     while (lo->hi() < min_val) {
1378       lo++;
1379     }
1380     if (lo->lo() < min_val)  {
1381       lo->setRange(min_val, lo->hi(), lo->dest(), lo->table_index(), lo->cnt());
1382     }
1383     while (hi->lo() > max_val) {
1384       hi--;
1385     }
1386     if (hi->hi() > max_val) {
1387       hi->setRange(hi->lo(), max_val, hi->dest(), hi->table_index(), hi->cnt());
1388     }
1389 
1390     linear_search_switch_ranges(key_val, lo, hi);
1391   }
1392 
1393 #ifndef PRODUCT
1394   if (switch_depth == 0) {
1395     _max_switch_depth = 0;
1396     _est_switch_depth = log2_intptr((hi-lo+1)-1)+1;
1397   }
1398 #endif
1399 
1400   assert(lo <= hi, "must be a non-empty set of ranges");
1401   if (lo == hi) {
1402     jump_if_always_fork(lo->dest(), lo->table_index(), trim_ranges && lo->cnt() == 0);
1403   } else {
1404     assert(lo->hi() == (lo+1)->lo()-1, "contiguous ranges");
1405     assert(hi->lo() == (hi-1)->hi()+1, "contiguous ranges");
1406 
1407     if (create_jump_tables(key_val, lo, hi)) return;
1408 
1409     SwitchRange* mid = NULL;
1410     float total_cnt = sum_of_cnts(lo, hi);
1411 
1412     int nr = hi - lo + 1;
1413     if (UseSwitchProfiling) {
1414       // Don't keep the binary search tree balanced: pick up mid point
1415       // that split frequencies in half.
1416       float cnt = 0;
1417       for (SwitchRange* sr = lo; sr <= hi; sr++) {
1418         cnt += sr->cnt();
1419         if (cnt >= total_cnt / 2) {
1420           mid = sr;
1421           break;
1422         }
1423       }
1424     } else {
1425       mid = lo + nr/2;
1426 
1427       // if there is an easy choice, pivot at a singleton:
1428       if (nr > 3 && !mid->is_singleton() && (mid-1)->is_singleton())  mid--;
1429 
1430       assert(lo < mid && mid <= hi, "good pivot choice");
1431       assert(nr != 2 || mid == hi,   "should pick higher of 2");
1432       assert(nr != 3 || mid == hi-1, "should pick middle of 3");
1433     }
1434 
1435 
1436     Node *test_val = _gvn.intcon(mid == lo ? mid->hi() : mid->lo());
1437 
1438     if (mid->is_singleton()) {
1439       IfNode *iff_ne = jump_if_fork_int(key_val, test_val, BoolTest::ne, 1-if_prob(mid->cnt(), total_cnt), if_cnt(mid->cnt()));
1440       jump_if_false_fork(iff_ne, mid->dest(), mid->table_index(), trim_ranges && mid->cnt() == 0);
1441 
1442       // Special Case:  If there are exactly three ranges, and the high
1443       // and low range each go to the same place, omit the "gt" test,
1444       // since it will not discriminate anything.
1445       bool eq_test_only = (hi == lo+2 && hi->dest() == lo->dest() && mid == hi-1) || mid == lo;
1446 
1447       // if there is a higher range, test for it and process it:
1448       if (mid < hi && !eq_test_only) {
1449         // two comparisons of same values--should enable 1 test for 2 branches
1450         // Use BoolTest::lt instead of BoolTest::gt
1451         float cnt = sum_of_cnts(lo, mid-1);
1452         IfNode *iff_lt  = jump_if_fork_int(key_val, test_val, BoolTest::lt, if_prob(cnt, total_cnt), if_cnt(cnt));
1453         Node   *iftrue  = _gvn.transform( new IfTrueNode(iff_lt) );
1454         Node   *iffalse = _gvn.transform( new IfFalseNode(iff_lt) );
1455         { PreserveJVMState pjvms(this);
1456           set_control(iffalse);
1457           jump_switch_ranges(key_val, mid+1, hi, switch_depth+1);
1458         }
1459         set_control(iftrue);
1460       }
1461 
1462     } else {
1463       // mid is a range, not a singleton, so treat mid..hi as a unit
1464       float cnt = sum_of_cnts(mid == lo ? mid+1 : mid, hi);
1465       IfNode *iff_ge = jump_if_fork_int(key_val, test_val, mid == lo ? BoolTest::gt : BoolTest::ge, if_prob(cnt, total_cnt), if_cnt(cnt));
1466 
1467       // if there is a higher range, test for it and process it:
1468       if (mid == hi) {
1469         jump_if_true_fork(iff_ge, mid->dest(), mid->table_index(), trim_ranges && cnt == 0);
1470       } else {
1471         Node *iftrue  = _gvn.transform( new IfTrueNode(iff_ge) );
1472         Node *iffalse = _gvn.transform( new IfFalseNode(iff_ge) );
1473         { PreserveJVMState pjvms(this);
1474           set_control(iftrue);
1475           jump_switch_ranges(key_val, mid == lo ? mid+1 : mid, hi, switch_depth+1);
1476         }
1477         set_control(iffalse);
1478       }
1479     }
1480 
1481     // in any case, process the lower range
1482     if (mid == lo) {
1483       if (mid->is_singleton()) {
1484         jump_switch_ranges(key_val, lo+1, hi, switch_depth+1);
1485       } else {
1486         jump_if_always_fork(lo->dest(), lo->table_index(), trim_ranges && lo->cnt() == 0);
1487       }
1488     } else {
1489       jump_switch_ranges(key_val, lo, mid-1, switch_depth+1);
1490     }
1491   }
1492 
1493   // Decrease pred_count for each successor after all is done.
1494   if (switch_depth == 0) {
1495     int unique_successors = switch_block->num_successors();
1496     for (int i = 0; i < unique_successors; i++) {
1497       Block* target = switch_block->successor_at(i);
1498       // Throw away the pre-allocated path for each unique successor.
1499       target->next_path_num();
1500     }
1501   }
1502 
1503 #ifndef PRODUCT
1504   _max_switch_depth = MAX2(switch_depth, _max_switch_depth);
1505   if (TraceOptoParse && Verbose && WizardMode && switch_depth == 0) {
1506     SwitchRange* r;
1507     int nsing = 0;
1508     for( r = lo; r <= hi; r++ ) {
1509       if( r->is_singleton() )  nsing++;
1510     }
1511     tty->print(">>> ");
1512     _method->print_short_name();
1513     tty->print_cr(" switch decision tree");
1514     tty->print_cr("    %d ranges (%d singletons), max_depth=%d, est_depth=%d",
1515                   (int) (hi-lo+1), nsing, _max_switch_depth, _est_switch_depth);
1516     if (_max_switch_depth > _est_switch_depth) {
1517       tty->print_cr("******** BAD SWITCH DEPTH ********");
1518     }
1519     tty->print("   ");
1520     for( r = lo; r <= hi; r++ ) {
1521       r->print();
1522     }
1523     tty->cr();
1524   }
1525 #endif
1526 }

1569 void Parse::do_irem() {
1570   // Must keep both values on the expression-stack during null-check
1571   zero_check_int(peek());
1572   // Compile-time detect of null-exception?
1573   if (stopped())  return;
1574 
1575   Node* b = pop();
1576   Node* a = pop();
1577 
1578   const Type *t = _gvn.type(b);
1579   if (t != Type::TOP) {
1580     const TypeInt *ti = t->is_int();
1581     if (ti->is_con()) {
1582       int divisor = ti->get_con();
1583       // check for positive power of 2
1584       if (divisor > 0 &&
1585           (divisor & ~(divisor-1)) == divisor) {
1586         // yes !
1587         Node *mask = _gvn.intcon((divisor - 1));
1588         // Sigh, must handle negative dividends
1589         Node *zero = _gvn.intcon(0);
1590         IfNode *ifff = jump_if_fork_int(a, zero, BoolTest::lt, PROB_FAIR, COUNT_UNKNOWN);
1591         Node *iff = _gvn.transform( new IfFalseNode(ifff) );
1592         Node *ift = _gvn.transform( new IfTrueNode (ifff) );
1593         Node *reg = jump_if_join(ift, iff);
1594         Node *phi = PhiNode::make(reg, NULL, TypeInt::INT);
1595         // Negative path; negate/and/negate
1596         Node *neg = _gvn.transform( new SubINode(zero, a) );
1597         Node *andn= _gvn.transform( new AndINode(neg, mask) );
1598         Node *negn= _gvn.transform( new SubINode(zero, andn) );
1599         phi->init_req(1, negn);
1600         // Fast positive case
1601         Node *andx = _gvn.transform( new AndINode(a, mask) );
1602         phi->init_req(2, andx);
1603         // Push the merge
1604         push( _gvn.transform(phi) );
1605         return;
1606       }
1607     }
1608   }
1609   // Default case
1610   push( _gvn.transform( new ModINode(control(),a,b) ) );

1650 static bool has_injected_profile(BoolTest::mask btest, Node* test, int& taken, int& not_taken) {
1651   if (btest != BoolTest::eq && btest != BoolTest::ne) {
1652     // Only ::eq and ::ne are supported for profile injection.
1653     return false;
1654   }
1655   if (test->is_Cmp() &&
1656       test->in(1)->Opcode() == Op_ProfileBoolean) {
1657     ProfileBooleanNode* profile = (ProfileBooleanNode*)test->in(1);
1658     int false_cnt = profile->false_count();
1659     int  true_cnt = profile->true_count();
1660 
1661     // Counts matching depends on the actual test operation (::eq or ::ne).
1662     // No need to scale the counts because profile injection was designed
1663     // to feed exact counts into VM.
1664     taken     = (btest == BoolTest::eq) ? false_cnt :  true_cnt;
1665     not_taken = (btest == BoolTest::eq) ?  true_cnt : false_cnt;
1666 
1667     profile->consume();
1668     return true;
1669   }
1670   return false;
1671 }
1672 //--------------------------dynamic_branch_prediction--------------------------
1673 // Try to gather dynamic branch prediction behavior.  Return a probability
1674 // of the branch being taken and set the "cnt" field.  Returns a -1.0
1675 // if we need to use static prediction for some reason.
1676 float Parse::dynamic_branch_prediction(float &cnt, BoolTest::mask btest, Node* test) {
1677   ResourceMark rm;
1678 
1679   cnt  = COUNT_UNKNOWN;
1680 
1681   int     taken = 0;
1682   int not_taken = 0;
1683 
1684   bool use_mdo = !has_injected_profile(btest, test, taken, not_taken);
1685 
1686   if (use_mdo) {
1687     // Use MethodData information if it is available
1688     // FIXME: free the ProfileData structure
1689     ciMethodData* methodData = method()->method_data();
1690     if (!methodData->is_mature())  return PROB_UNKNOWN;
1691     ciProfileData* data = methodData->bci_to_data(bci());
1692     if (data == NULL) {
1693       return PROB_UNKNOWN;
1694     }
1695     if (!data->is_JumpData())  return PROB_UNKNOWN;
1696 
1697     // get taken and not taken values
1698     taken = data->as_JumpData()->taken();
1699     not_taken = 0;
1700     if (data->is_BranchData()) {
1701       not_taken = data->as_BranchData()->not_taken();
1702     }
1703 
1704     // scale the counts to be commensurate with invocation counts:
1705     taken = method()->scale_count(taken);
1706     not_taken = method()->scale_count(not_taken);
1707   }
1708 
1709   // Give up if too few (or too many, in which case the sum will overflow) counts to be meaningful.
1710   // We also check that individual counters are positive first, otherwise the sum can become positive.
1711   if (taken < 0 || not_taken < 0 || taken + not_taken < 40) {
1712     if (C->log() != NULL) {
1713       C->log()->elem("branch target_bci='%d' taken='%d' not_taken='%d'", iter().get_dest(), taken, not_taken);
1714     }
1715     return PROB_UNKNOWN;
1716   }
1717 
1718   // Compute frequency that we arrive here
1719   float sum = taken + not_taken;
1720   // Adjust, if this block is a cloned private block but the
1721   // Jump counts are shared.  Taken the private counts for
1722   // just this path instead of the shared counts.
1723   if( block()->count() > 0 )
1724     sum = block()->count();
1725   cnt = sum / FreqCountInvocations;
1726 
1727   // Pin probability to sane limits
1728   float prob;
1729   if( !taken )
1730     prob = (0+PROB_MIN) / 2;
1731   else if( !not_taken )
1732     prob = (1+PROB_MAX) / 2;
1733   else {                         // Compute probability of true path
1734     prob = (float)taken / (float)(taken + not_taken);
1735     if (prob > PROB_MAX)  prob = PROB_MAX;
1736     if (prob < PROB_MIN)   prob = PROB_MIN;
1737   }
1738 
1739   assert((cnt > 0.0f) && (prob > 0.0f),
1740          "Bad frequency assignment in if");
1741 
1742   if (C->log() != NULL) {
1743     const char* prob_str = NULL;
1744     if (prob >= PROB_MAX)  prob_str = (prob == PROB_MAX) ? "max" : "always";
1745     if (prob <= PROB_MIN)  prob_str = (prob == PROB_MIN) ? "min" : "never";
1746     char prob_str_buf[30];
1747     if (prob_str == NULL) {
1748       jio_snprintf(prob_str_buf, sizeof(prob_str_buf), "%20.2f", prob);
1749       prob_str = prob_str_buf;
1750     }
1751     C->log()->elem("branch target_bci='%d' taken='%d' not_taken='%d' cnt='%f' prob='%s'",
1752                    iter().get_dest(), taken, not_taken, cnt, prob_str);
1753   }
1754   return prob;
1755 }

1757 //-----------------------------branch_prediction-------------------------------
1758 float Parse::branch_prediction(float& cnt,
1759                                BoolTest::mask btest,
1760                                int target_bci,
1761                                Node* test) {
1762   float prob = dynamic_branch_prediction(cnt, btest, test);
1763   // If prob is unknown, switch to static prediction
1764   if (prob != PROB_UNKNOWN)  return prob;
1765 
1766   prob = PROB_FAIR;                   // Set default value
1767   if (btest == BoolTest::eq)          // Exactly equal test?
1768     prob = PROB_STATIC_INFREQUENT;    // Assume its relatively infrequent
1769   else if (btest == BoolTest::ne)
1770     prob = PROB_STATIC_FREQUENT;      // Assume its relatively frequent
1771 
1772   // If this is a conditional test guarding a backwards branch,
1773   // assume its a loop-back edge.  Make it a likely taken branch.
1774   if (target_bci < bci()) {
1775     if (is_osr_parse()) {    // Could be a hot OSR'd loop; force deopt
1776       // Since it's an OSR, we probably have profile data, but since
1777       // branch_prediction returned PROB_UNKNOWN, the counts are too small.
1778       // Let's make a special check here for completely zero counts.
1779       ciMethodData* methodData = method()->method_data();
1780       if (!methodData->is_empty()) {
1781         ciProfileData* data = methodData->bci_to_data(bci());
1782         // Only stop for truly zero counts, which mean an unknown part
1783         // of the OSR-ed method, and we want to deopt to gather more stats.
1784         // If you have ANY counts, then this loop is simply 'cold' relative
1785         // to the OSR loop.
1786         if (data == NULL ||
1787             (data->as_BranchData()->taken() +  data->as_BranchData()->not_taken() == 0)) {
1788           // This is the only way to return PROB_UNKNOWN:
1789           return PROB_UNKNOWN;
1790         }
1791       }
1792     }
1793     prob = PROB_STATIC_FREQUENT;     // Likely to take backwards branch
1794   }
1795 
1796   assert(prob != PROB_UNKNOWN, "must have some guess at this point");
1797   return prob;
1798 }

1800 // The magic constants are chosen so as to match the output of
1801 // branch_prediction() when the profile reports a zero taken count.
1802 // It is important to distinguish zero counts unambiguously, because
1803 // some branches (e.g., _213_javac.Assembler.eliminate) validly produce
1804 // very small but nonzero probabilities, which if confused with zero
1805 // counts would keep the program recompiling indefinitely.
1806 bool Parse::seems_never_taken(float prob) const {
1807   return prob < PROB_MIN;

1810 // True if the comparison seems to be the kind that will not change its
1811 // statistics from true to false.  See comments in adjust_map_after_if.
1812 // This question is only asked along paths which are already
1813 // classifed as untaken (by seems_never_taken), so really,
1814 // if a path is never taken, its controlling comparison is
1815 // already acting in a stable fashion.  If the comparison
1816 // seems stable, we will put an expensive uncommon trap
1817 // on the untaken path.

1819   if (C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if)) {
1820     return false;
1821   }
1822   return true;

1842 //----------------------------------do_ifnull----------------------------------
1843 void Parse::do_ifnull(BoolTest::mask btest, Node *c) {
1844   int target_bci = iter().get_dest();
1845 
1846   Block* branch_block = successor_for_bci(target_bci);
1847   Block* next_block   = successor_for_bci(iter().next_bci());
1848 
1849   float cnt;
1850   float prob = branch_prediction(cnt, btest, target_bci, c);
1851   if (prob == PROB_UNKNOWN) {
1852     // (An earlier version of do_ifnull omitted this trap for OSR methods.)
1853     if (PrintOpto && Verbose) {
1854       tty->print_cr("Never-taken edge stops compilation at bci %d", bci());
1855     }
1856     repush_if_args(); // to gather stats on loop
1857     // We need to mark this branch as taken so that if we recompile we will
1858     // see that it is possible. In the tiered system the interpreter doesn't
1859     // do profiling and by the time we get to the lower tier from the interpreter
1860     // the path may be cold again. Make sure it doesn't look untaken
1861     profile_taken_branch(target_bci, !ProfileInterpreter);
1862     uncommon_trap(Deoptimization::Reason_unreached,
1863                   Deoptimization::Action_reinterpret,
1864                   NULL, "cold");
1865     if (C->eliminate_boxing()) {
1866       // Mark the successor blocks as parsed
1867       branch_block->next_path_num();
1868       next_block->next_path_num();
1869     }
1870     return;
1871   }
1872 
1873   NOT_PRODUCT(explicit_null_checks_inserted++);
1874 
1875   // Generate real control flow
1876   Node   *tst = _gvn.transform( new BoolNode( c, btest ) );
1877 
1878   // Sanity check the probability value
1879   assert(prob > 0.0f,"Bad probability in Parser");
1880  // Need xform to put node in hash table
1881   IfNode *iff = create_and_xform_if( control(), tst, prob, cnt );
1882   assert(iff->_prob > 0.0f,"Optimizer made bad probability in parser");
1883   // True branch
1884   { PreserveJVMState pjvms(this);
1885     Node* iftrue  = _gvn.transform( new IfTrueNode (iff) );
1886     set_control(iftrue);
1887 
1888     if (stopped()) {            // Path is dead?
1889       NOT_PRODUCT(explicit_null_checks_elided++);
1890       if (C->eliminate_boxing()) {
1891         // Mark the successor block as parsed
1892         branch_block->next_path_num();
1893       }
1894     } else {                    // Path is live.
1895       // Update method data
1896       profile_taken_branch(target_bci);
1897       adjust_map_after_if(btest, c, prob, branch_block);
1898       if (!stopped()) {
1899         merge(target_bci);
1900       }
1901     }
1902   }
1903 
1904   // False branch
1905   Node* iffalse = _gvn.transform( new IfFalseNode(iff) );
1906   set_control(iffalse);
1907 
1908   if (stopped()) {              // Path is dead?
1909     NOT_PRODUCT(explicit_null_checks_elided++);
1910     if (C->eliminate_boxing()) {
1911       // Mark the successor block as parsed
1912       next_block->next_path_num();
1913     }
1914   } else  {                     // Path is live.
1915     // Update method data
1916     profile_not_taken_branch();
1917     adjust_map_after_if(BoolTest(btest).negate(), c, 1.0-prob, next_block);
1918   }
1919 }

1921 //------------------------------------do_if------------------------------------
1922 void Parse::do_if(BoolTest::mask btest, Node* c, bool new_path, Node** ctrl_taken) {
1923   int target_bci = iter().get_dest();
1924 
1925   Block* branch_block = successor_for_bci(target_bci);
1926   Block* next_block   = successor_for_bci(iter().next_bci());
1927 
1928   float cnt;
1929   float prob = branch_prediction(cnt, btest, target_bci, c);
1930   float untaken_prob = 1.0 - prob;
1931 
1932   if (prob == PROB_UNKNOWN) {
1933     if (PrintOpto && Verbose) {
1934       tty->print_cr("Never-taken edge stops compilation at bci %d", bci());
1935     }
1936     repush_if_args(); // to gather stats on loop
1937     // We need to mark this branch as taken so that if we recompile we will
1938     // see that it is possible. In the tiered system the interpreter doesn't
1939     // do profiling and by the time we get to the lower tier from the interpreter
1940     // the path may be cold again. Make sure it doesn't look untaken
1941     profile_taken_branch(target_bci, !ProfileInterpreter);
1942     uncommon_trap(Deoptimization::Reason_unreached,
1943                   Deoptimization::Action_reinterpret,
1944                   NULL, "cold");
1945     if (C->eliminate_boxing()) {
1946       // Mark the successor blocks as parsed
1947       branch_block->next_path_num();
1948       next_block->next_path_num();
1949     }
1950     return;
1951   }
1952 
1953   // Sanity check the probability value
1954   assert(0.0f < prob && prob < 1.0f,"Bad probability in Parser");
1955 
1956   bool taken_if_true = true;
1957   // Convert BoolTest to canonical form:
1958   if (!BoolTest(btest).is_canonical()) {
1959     btest         = BoolTest(btest).negate();
1960     taken_if_true = false;
1961     // prob is NOT updated here; it remains the probability of the taken
1962     // path (as opposed to the prob of the path guarded by an 'IfTrueNode').
1963   }
1964   assert(btest != BoolTest::eq, "!= is the only canonical exact test");
1965 
1966   Node* tst0 = new BoolNode(c, btest);
1967   Node* tst = _gvn.transform(tst0);
1968   BoolTest::mask taken_btest   = BoolTest::illegal;
1969   BoolTest::mask untaken_btest = BoolTest::illegal;
1970 
1971   if (tst->is_Bool()) {
1972     // Refresh c from the transformed bool node, since it may be
1973     // simpler than the original c.  Also re-canonicalize btest.
1974     // This wins when (Bool ne (Conv2B p) 0) => (Bool ne (CmpP p NULL)).
1975     // That can arise from statements like: if (x instanceof C) ...
1976     if (tst != tst0) {
1977       // Canonicalize one more time since transform can change it.
1978       btest = tst->as_Bool()->_test._test;
1979       if (!BoolTest(btest).is_canonical()) {
1980         // Reverse edges one more time...
1981         tst   = _gvn.transform( tst->as_Bool()->negate(&_gvn) );
1982         btest = tst->as_Bool()->_test._test;
1983         assert(BoolTest(btest).is_canonical(), "sanity");
1984         taken_if_true = !taken_if_true;
1985       }
1986       c = tst->in(1);
1987     }
1988     BoolTest::mask neg_btest = BoolTest(btest).negate();
1989     taken_btest   = taken_if_true ?     btest : neg_btest;
1990     untaken_btest = taken_if_true ? neg_btest :     btest;
1991   }
1992 
1993   // Generate real control flow
1994   float true_prob = (taken_if_true ? prob : untaken_prob);
1995   IfNode* iff = create_and_map_if(control(), tst, true_prob, cnt);
1996   assert(iff->_prob > 0.0f,"Optimizer made bad probability in parser");
1997   Node* taken_branch   = new IfTrueNode(iff);
1998   Node* untaken_branch = new IfFalseNode(iff);
1999   if (!taken_if_true) {  // Finish conversion to canonical form
2000     Node* tmp      = taken_branch;
2001     taken_branch   = untaken_branch;
2002     untaken_branch = tmp;
2003   }
2004 
2005   // Branch is taken:
2006   { PreserveJVMState pjvms(this);
2007     taken_branch = _gvn.transform(taken_branch);
2008     set_control(taken_branch);
2009 
2010     if (stopped()) {
2011       if (C->eliminate_boxing() && !new_path) {
2012         // Mark the successor block as parsed (if we haven't created a new path)
2013         branch_block->next_path_num();
2014       }
2015     } else {
2016       // Update method data
2017       profile_taken_branch(target_bci);
2018       adjust_map_after_if(taken_btest, c, prob, branch_block);
2019       if (!stopped()) {
2020         if (new_path) {
2021           // Merge by using a new path
2022           merge_new_path(target_bci);
2023         } else if (ctrl_taken != NULL) {
2024           // Don't merge but save taken branch to be wired by caller
2025           *ctrl_taken = control();
2026         } else {
2027           merge(target_bci);
2028         }
2029       }
2030     }
2031   }
2032 
2033   untaken_branch = _gvn.transform(untaken_branch);
2034   set_control(untaken_branch);
2035 
2036   // Branch not taken.
2037   if (stopped() && ctrl_taken == NULL) {
2038     if (C->eliminate_boxing()) {
2039       // Mark the successor block as parsed (if caller does not re-wire control flow)
2040       next_block->next_path_num();
2041     }
2042   } else {
2043     // Update method data
2044     profile_not_taken_branch();
2045     adjust_map_after_if(untaken_btest, c, untaken_prob, next_block);
2046   }
2047 }

2132   Node* not_value = _gvn.transform(new IfFalseNode(is_value_iff));
2133   ne_region->init_req(2, not_value);
2134   set_control(_gvn.transform(new IfTrueNode(is_value_iff)));
2135 
2136   // The first operand is an inline type, check if the second operand is non-null
2137   inc_sp(2);
2138   null_ctl = top();
2139   Node* not_null_b = null_check_oop(b, &null_ctl, !too_many_traps(Deoptimization::Reason_null_check), false, false);
2140   dec_sp(2);

2146       {
2147         PreserveJVMState pjvms(this);

2353 void Parse::sharpen_type_after_if(BoolTest::mask btest,
2354                                   Node* con, const Type* tcon,
2355                                   Node* val, const Type* tval) {
2356   // Look for opportunities to sharpen the type of a node
2357   // whose klass is compared with a constant klass.
2358   if (btest == BoolTest::eq && tcon->isa_klassptr()) {
2359     Node* obj = extract_obj_from_klass_load(&_gvn, val);
2360     const TypeOopPtr* con_type = tcon->isa_klassptr()->as_instance_type();
2361     if (obj != NULL && (con_type->isa_instptr() || con_type->isa_aryptr())) {
2362        // Found:
2363        //   Bool(CmpP(LoadKlass(obj._klass), ConP(Foo.klass)), [eq])
2364        // or the narrowOop equivalent.
2365        const Type* obj_type = _gvn.type(obj);
2366        const TypeOopPtr* tboth = obj_type->join_speculative(con_type)->isa_oopptr();
2367        if (tboth != NULL && tboth->klass_is_exact() && tboth != obj_type &&
2368            tboth->higher_equal(obj_type)) {
2369           // obj has to be of the exact type Foo if the CmpP succeeds.
2370           int obj_in_map = map()->find_edge(obj);
2371           JVMState* jvms = this->jvms();
2372           if (obj_in_map >= 0 &&
2373               (jvms->is_loc(obj_in_map) || jvms->is_stk(obj_in_map))) {
2374             TypeNode* ccast = new CheckCastPPNode(control(), obj, tboth);
2375             const Type* tcc = ccast->as_Type()->type();
2376             assert(tcc != obj_type && tcc->higher_equal(obj_type), "must improve");
2377             // Delay transform() call to allow recovery of pre-cast value
2378             // at the control merge.
2379             _gvn.set_type_bottom(ccast);
2380             record_for_igvn(ccast);
2381             // Here's the payoff.
2382             replace_in_map(obj, ccast);
2383           }
2384        }
2385     }
2386   }
2387 
2388   int val_in_map = map()->find_edge(val);
2389   if (val_in_map < 0)  return;          // replace_in_map would be useless
2390   {
2391     JVMState* jvms = this->jvms();
2392     if (!(jvms->is_loc(val_in_map) ||
2393           jvms->is_stk(val_in_map)))
2394       return;                           // again, it would be useless
2395   }
2396 
2397   // Check for a comparison to a constant, and "know" that the compared
2398   // value is constrained on this path.
2399   assert(tcon->singleton(), "");
2400   ConstraintCastNode* ccast = NULL;
2401   Node* cast = NULL;
2402 
2403   switch (btest) {
2404   case BoolTest::eq:                    // Constant test?
2405     {
2406       const Type* tboth = tcon->join_speculative(tval);
2407       if (tboth == tval)  break;        // Nothing to gain.
2408       if (tcon->isa_int()) {
2409         ccast = new CastIINode(val, tboth);
2410       } else if (tcon == TypePtr::NULL_PTR) {
2411         // Cast to null, but keep the pointer identity temporarily live.
2412         ccast = new CastPPNode(val, tboth);
2413       } else {
2414         const TypeF* tf = tcon->isa_float_constant();
2415         const TypeD* td = tcon->isa_double_constant();
2416         // Exclude tests vs float/double 0 as these could be
2417         // either +0 or -0.  Just because you are equal to +0
2418         // doesn't mean you ARE +0!
2419         // Note, following code also replaces Long and Oop values.
2420         if ((!tf || tf->_f != 0.0) &&
2421             (!td || td->_d != 0.0))
2422           cast = con;                   // Replace non-constant val by con.
2423       }
2424     }
2425     break;
2426 
2427   case BoolTest::ne:
2428     if (tcon == TypePtr::NULL_PTR) {
2429       cast = cast_not_null(val, false);
2430     }
2431     break;
2432 
2433   default:
2434     // (At this point we could record int range types with CastII.)
2435     break;
2436   }
2437 
2438   if (ccast != NULL) {
2439     const Type* tcc = ccast->as_Type()->type();
2440     assert(tcc != tval && tcc->higher_equal(tval), "must improve");
2441     // Delay transform() call to allow recovery of pre-cast value
2442     // at the control merge.
2443     ccast->set_req(0, control());
2444     _gvn.set_type_bottom(ccast);
2445     record_for_igvn(ccast);
2446     cast = ccast;
2447   }
2448 
2449   if (cast != NULL) {                   // Here's the payoff.
2450     replace_in_map(val, cast);
2451   }
2452 }
