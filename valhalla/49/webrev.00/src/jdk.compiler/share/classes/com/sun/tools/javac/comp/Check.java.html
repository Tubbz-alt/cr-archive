 233             }
 234         }

 330     /** Report duplicate declaration error.
 331      */
 332     void duplicateError(DiagnosticPosition pos, Symbol sym) {
 333         if (!sym.type.isErroneous()) {
 334             Symbol location = sym.location();
 335             if (location.kind == MTH &&
 336                     ((MethodSymbol)location).isStaticOrInstanceInit()) {
 337                 log.error(pos,
 338                           Errors.AlreadyDefinedInClinit(kindName(sym),
 339                                                         sym,
 340                                                         kindName(sym.location()),
 341                                                         kindName(sym.location().enclClass()),
 342                                                         sym.location().enclClass()));
 343             } else {
 344                 /* dont error if this is a duplicated parameter of a generated canonical constructor
 345                  * as we should have issued an error for the duplicated fields
 346                  */
 347                 if (location.kind != MTH ||
 348                         ((sym.owner.flags_field & GENERATEDCONSTR) == 0) ||
 349                         ((sym.owner.flags_field & RECORD) == 0)) {
 350                     log.error(pos,
 351                             Errors.AlreadyDefined(kindName(sym),
 352                                     sym,
 353                                     kindName(sym.location()),
 354                                     sym.location()));
 355                 }
 356             }
 357         }
 358     }

 372     /** Check that variable does not hide variable with same name in
 373      *  immediately enclosing local scope.
 374      *  @param pos           Position for error reporting.
 375      *  @param v             The symbol.
 376      *  @param s             The scope.
 377      */
 378     void checkTransparentVar(DiagnosticPosition pos, VarSymbol v, Scope s) {
 379         for (Symbol sym : s.getSymbolsByName(v.name)) {
 380             if (sym.owner != v.owner) break;
 381             if (sym.kind == VAR &&
 382                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 383                 v.name != names.error) {
 384                 duplicateError(pos, sym);
 385                 return;
 386             }
 387         }
 388     }
 389 
 390     /** Check that a class or interface does not hide a class or
 391      *  interface with same name in immediately enclosing local scope.
 392      *  @param pos           Position for error reporting.
 393      *  @param c             The symbol.
 394      *  @param s             The scope.
 395      */
 396     void checkTransparentClass(DiagnosticPosition pos, ClassSymbol c, Scope s) {
 397         for (Symbol sym : s.getSymbolsByName(c.name)) {
 398             if (sym.owner != c.owner) break;
 399             if (sym.kind == TYP && !sym.type.hasTag(TYPEVAR) &&
 400                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 401                 c.name != names.error) {
 402                 duplicateError(pos, sym);
 403                 return;
 404             }
 405         }
 406     }

 408     /** Check that class does not have the same name as one of
 409      *  its enclosing classes, or as a class defined in its enclosing scope.
 410      *  return true if class is unique in its enclosing scope.
 411      *  @param pos           Position for error reporting.
 412      *  @param name          The class name.
 413      *  @param s             The enclosing scope.
 414      */
 415     boolean checkUniqueClassName(DiagnosticPosition pos, Name name, Scope s) {
 416         for (Symbol sym : s.getSymbolsByName(name, NON_RECURSIVE)) {
 417             if (sym.kind == TYP && sym.name != names.error) {
 418                 duplicateError(pos, sym);
 419                 return false;
 420             }
 421         }
 422         for (Symbol sym = s.owner; sym != null; sym = sym.owner) {
 423             if (sym.kind == TYP && sym.name == name && sym.name != names.error) {
 424                 duplicateError(pos, sym);
 425                 return true;
 426             }
 427         }
 428         return true;
 429     }

 438     /** Return name of local class.
 439      *  This is of the form   {@code <enclClass> $ n <classname> }
 440      *  where
 441      *    enclClass is the flat name of the enclosing class,
 442      *    classname is the simple name of the local class
 443      */
 444     public Name localClassName(ClassSymbol c) {
 445         Name enclFlatname = c.owner.enclClass().flatname;
 446         String enclFlatnameStr = enclFlatname.toString();
 447         Pair<Name, Name> key = new Pair<>(enclFlatname, c.name);
 448         Integer index = localClassNameIndexes.get(key);
 449         for (int i = (index == null) ? 1 : index; ; i++) {
 450             Name flatname = names.fromString(enclFlatnameStr
 451                     + syntheticNameChar + i + c.name);
 452             if (getCompiled(c.packge().modle, flatname) == null) {
 453                 localClassNameIndexes.put(key, i + 1);
 454                 return flatname;
 455             }
 456         }
 457     }
 458 
 459     public void clearLocalClassNameIndexes(ClassSymbol c) {
 460         if (c.owner != null && c.owner.kind != NIL) {
 461             localClassNameIndexes.remove(new Pair<>(
 462                     c.owner.enclClass().flatname, c.name));
 463         }
 464     }

 595     Type checkType(final DiagnosticPosition pos, final Type found, final Type req, final CheckContext checkContext) {
 596         final InferenceContext inferenceContext = checkContext.inferenceContext();
 597         if (inferenceContext.free(req) || inferenceContext.free(found)) {
 598             inferenceContext.addFreeTypeListener(List.of(req, found),
 599                     solvedContext -> checkType(pos, solvedContext.asInstType(found), solvedContext.asInstType(req), checkContext));
 600         } else {
 601             if (found.hasTag(CLASS)) {
 602                 if (inferenceContext != infer.emptyContext)
 603                     checkParameterizationWithValues(pos, found);
 604             }
 605         }
 606         if (req.hasTag(ERROR))
 607             return req;
 608         if (req.hasTag(NONE))
 609             return found;
 610         if (checkContext.compatible(found, req, checkContext.checkWarner(pos, found, req))) {
 611             if (found.hasTag(BOT) && types.isValueBased(req)) {
 612                 log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 613             }
 614             return found;
 615         } else {
 616             if (found.isNumeric() && req.isNumeric()) {
 617                 checkContext.report(pos, diags.fragment(Fragments.PossibleLossOfPrecision(found, req)));
 618                 return types.createErrorType(found);
 619             }
 620             checkContext.report(pos, diags.fragment(Fragments.InconvertibleTypes(found, req)));
 621             return types.createErrorType(found);
 622         }
 623     }

 625     /** Check that a given type can be cast to a given target type.
 626      *  Return the result of the cast.

 628      *  @param found      The type that is being cast.
 629      *  @param req        The target type of the cast.

 634     Type checkCastable(DiagnosticPosition pos, Type found, Type req, CheckContext checkContext) {
 635         if (types.isCastable(found, req, castWarner(pos, found, req))) {
 636             if (types.isValueBased(req)) {
 637                 if (found.hasTag(BOT)) {
 638                     log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 639                 } else if (!types.isValueBased(found)) {
 640                     log.warning(pos, Warnings.PotentialNullPollution(found));
 641                 }

1136             final Symbol sym,

1144         // System.out.println("actuals: " + argtypes);

1147                     solvedContext -> checkMethod(solvedContext.asInstType(mtype), sym, env, argtrees, argtypes, useVarargs, solvedContext));
1148             return mtype;

1169                 assertConvertible(arg, arg.type, formals.head, warn);

1171                 formals = formals.tail;
1172                 nonInferred = nonInferred.tail;
1173             }

1175                 Type varArg = types.elemtype(last);
1176                 while (args.tail != null) {
1177                     JCTree arg = args.head;
1178                     Warner warn = convertWarner(arg.pos(), arg.type, varArg);
1179                     assertConvertible(arg, arg.type, varArg, warn);
1180                     args = args.tail;
1181                 }
1182             } else if ((sym.flags() & (VARARGS | SIGNATURE_POLYMORPHIC)) == VARARGS) {
1183                 // non-varargs call to varargs method
1184                 Type varParam = owntype.getParameterTypes().last();
1185                 Type lastArg = argtypes.last();
1186                 if (types.isSubtypeUnchecked(lastArg, types.elemtype(varParam)) &&

1189                                 Warnings.InexactNonVarargsCall(types.elemtype(varParam),varParam));
1190             }

1194             if (!types.isReifiable(argtype) &&
1195                 (sym.baseSymbol().attribute(syms.trustMeType.tsym) == null ||
1196                  !isTrustMeAllowedOnMethod(sym))) {
1197                 warnUnchecked(env.tree.pos(), Warnings.UncheckedGenericArrayCreation(argtype));
1198             }
1199             TreeInfo.setVarargsElement(env.tree, types.elemtype(argtype));
1200          }
1201          return owntype;

1225         private Type firstIncompatibleTypeArg(Type type) {
1226             List<Type> formals = type.tsym.type.allparams();
1227             List<Type> actuals = type.allparams();
1228             List<Type> args = type.getTypeArguments();
1229             List<Type> forms = type.tsym.type.getTypeArguments();
1230             ListBuffer<Type> bounds_buf = new ListBuffer<>();
1231 
1232             // For matching pairs of actual argument types `a' and
1233             // formal type parameters with declared bound `b' ...
1234             while (args.nonEmpty() && forms.nonEmpty()) {
1235                 // exact type arguments needs to know their
1236                 // bounds (for upper and lower bound
1237                 // calculations).  So we create new bounds where
1238                 // type-parameters are replaced with actuals argument types.
1239                 bounds_buf.append(types.subst(forms.head.getUpperBound(), formals, actuals));
1240                 args = args.tail;
1241                 forms = forms.tail;
1242             }
1243 
1244             args = type.getTypeArguments();
1245             List<Type> tvars_cap = types.substBounds(formals,
1246                                       formals,
1247                                       types.capture(type).allparams());
1248             while (args.nonEmpty() && tvars_cap.nonEmpty()) {
1249                 // Let the actual arguments know their bound
1250                 args.head.withTypeVar((TypeVar)tvars_cap.head);
1251                 args = args.tail;
1252                 tvars_cap = tvars_cap.tail;
1253             }
1254 
1255             args = type.getTypeArguments();
1256             List<Type> bounds = bounds_buf.toList();
1257 
1258             while (args.nonEmpty() && bounds.nonEmpty()) {
1259                 Type actual = args.head;
1260                 if (!isTypeArgErroneous(actual) &&
1261                         !bounds.head.isErroneous() &&
1262                         !checkExtends(actual, bounds.head)) {
1263                     return args.head;
1264                 }
1265                 args = args.tail;
1266                 bounds = bounds.tail;
1267             }
1268 
1269             args = type.getTypeArguments();
1270             bounds = bounds_buf.toList();
1271 
1272             for (Type arg : types.capture(type).getTypeArguments()) {
1273                 if (arg.hasTag(TYPEVAR) &&
1274                         arg.getUpperBound().isErroneous() &&
1275                         !bounds.head.isErroneous() &&
1276                         !isTypeArgErroneous(args.head)) {
1277                     return args.head;
1278                 }
1279                 bounds = bounds.tail;
1280                 args = args.tail;
1281             }
1282 
1283             return null;
1284         }

1309     /** Check that given modifiers are legal for given symbol and
1310      *  return modifiers together with any implicit modifiers for that symbol.
1311      *  Warning: we can't use flags() here since this method
1312      *  is called during class enter, when flags() would cause a premature
1313      *  completion.
1314      *  @param pos           Position to be used for error reporting.
1315      *  @param flags         The set of modifiers given in a definition.
1316      *  @param sym           The defined symbol.
1317      */
1318     long checkFlags(DiagnosticPosition pos, long flags, Symbol sym, JCTree tree) {
1319         long mask;
1320         long implicit = 0;
1321 
1322         switch (sym.kind) {
1323         case VAR:
1324             if (TreeInfo.isReceiverParam(tree))
1325                 mask = ReceiverParamFlags;
1326             else if (sym.owner.kind != TYP)
1327                 mask = LocalVarFlags;
1328             else if ((sym.owner.flags_field & INTERFACE) != 0)
1329                 mask = implicit = InterfaceVarFlags;
1330             else {
1331                 mask = VarFlags;
1332                 if (types.isValue(sym.owner.type) && (flags & STATIC) == 0) {
1333                     implicit |= FINAL;
1334                 }
1335             }
1336             break;
1337         case MTH:
1338             if (sym.name == names.init) {
1339                 if ((sym.owner.flags_field & ENUM) != 0) {
1340                     // enum constructors cannot be declared public or
1341                     // protected and must be implicitly or explicitly
1342                     // private
1343                     implicit = PRIVATE;
1344                     mask = PRIVATE;
1345                 } else
1346                     mask = ConstructorFlags;
1347             }  else if ((sym.owner.flags_field & INTERFACE) != 0) {
1348                 if ((sym.owner.flags_field & ANNOTATION) != 0) {
1349                     mask = AnnotationTypeElementMask;
1350                     implicit = PUBLIC | ABSTRACT;
1351                 } else if ((flags & (DEFAULT | STATIC | PRIVATE)) != 0) {
1352                     mask = InterfaceMethodMask;
1353                     implicit = (flags & PRIVATE) != 0 ? 0 : PUBLIC;
1354                     if ((flags & DEFAULT) != 0) {
1355                         implicit |= ABSTRACT;
1356                     }
1357                 } else {
1358                     mask = implicit = InterfaceMethodFlags;
1359                 }
1360             } else if ((sym.owner.flags_field & RECORD) != 0) {
1361                 mask = RecordMethodFlags;
1362             } else {
1363                 // instance methods of value types do not have a monitor associated with their `this'
1364                 mask = ((sym.owner.flags_field & VALUE) != 0 && (flags & Flags.STATIC) == 0) ?
1365                         MethodFlags & ~SYNCHRONIZED : MethodFlags;
1366             }
1367             // Imply STRICTFP if owner has STRICTFP set.
1368             if (((flags|implicit) & Flags.ABSTRACT) == 0 ||
1369                 ((flags) & Flags.DEFAULT) != 0)
1370                 implicit |= sym.owner.flags_field & STRICTFP;
1371             break;
1372         case TYP:
1373             if (sym.isLocal()) {
1374                 mask = (flags & RECORD) != 0 ? LocalRecordFlags : LocalClassFlags;
1375                 if ((flags & RECORD) != 0) {
1376                     implicit = STATIC;
1377                     if (sym.owner.kind == TYP) {
1378                         log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1379                     }
1380                 }
1381                 if ((sym.owner.flags_field & STATIC) == 0 &&
1382                     (flags & ENUM) != 0) {
1383                     log.error(pos, Errors.EnumsMustBeStatic);
1384                 }
1385             } else if (sym.owner.kind == TYP) {
1386                 mask = (flags & RECORD) != 0 ? MemberRecordFlags : MemberClassFlags;
1387                 if (sym.owner.owner.kind == PCK ||
1388                     (sym.owner.flags_field & STATIC) != 0)
1389                     mask |= STATIC;
1390                 else if ((flags & ENUM) != 0) {
1391                     log.error(pos, Errors.EnumsMustBeStatic);
1392                 } else if ((flags & RECORD) != 0) {
1393                     log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1394                 }
1395                 // Nested interfaces and enums are always STATIC (Spec ???)
1396                 if ((flags & (INTERFACE | ENUM | RECORD)) != 0 ) implicit = STATIC;
1397             } else {
1398                 mask = ClassFlags;
1399             }
1400             // Interfaces are always ABSTRACT
1401             if ((flags & INTERFACE) != 0) implicit |= ABSTRACT;
1402 
1403             if ((flags & ENUM) != 0) {
1404                 // enums can't be declared abstract or final or value type
1405                 mask &= ~(ABSTRACT | FINAL | VALUE);
1406                 implicit |= implicitEnumFinalFlag(tree);
1407             }
1408             if ((flags & RECORD) != 0) {
1409                 // records can't be declared abstract
1410                 mask &= ~ABSTRACT;
1411                 implicit |= FINAL;
1412             }
1413             // Imply STRICTFP if owner has STRICTFP set.
1414             implicit |= sym.owner.flags_field & STRICTFP;
1415             break;
1416         default:
1417             throw new AssertionError();
1418         }
1419         long illegal = flags & ExtendedStandardFlags & ~mask;
1420         if (illegal != 0) {
1421             if ((illegal & INTERFACE) != 0) {
1422                 log.error(pos, ((flags & ANNOTATION) != 0) ? Errors.AnnotationDeclNotAllowedHere : Errors.IntfNotAllowedHere);
1423                 mask |= INTERFACE;
1424             }
1425             else {
1426                 log.error(pos,
1427                           Errors.ModNotAllowedHere(asFlagSet(illegal)));
1428             }
1429         }
1430         else if ((sym.kind == TYP ||
1431                   // ISSUE: Disallowing abstract&private is no longer appropriate
1432                   // in the presence of inner classes. Should it be deleted here?
1433                   checkDisjoint(pos, flags,
1434                                 ABSTRACT,
1435                                 PRIVATE | STATIC | DEFAULT))
1436                  &&
1437                  checkDisjoint(pos, flags,
1438                                 STATIC | PRIVATE,
1439                                 DEFAULT)
1440                  &&
1441                  checkDisjoint(pos, flags,
1442                                ABSTRACT | INTERFACE,
1443                                FINAL | NATIVE | SYNCHRONIZED | VALUE)
1444                  &&
1445                  checkDisjoint(pos, flags,
1446                                PUBLIC,
1447                                PRIVATE | PROTECTED)
1448                  &&
1449                  checkDisjoint(pos, flags,
1450                                PRIVATE,
1451                                PUBLIC | PROTECTED)
1452                  &&
1453                  checkDisjoint(pos, (flags | implicit), // complain against volatile & implcitly final entities too.
1454                                FINAL,
1455                                VOLATILE)
1456                  &&
1457                  (sym.kind == TYP ||
1458                   checkDisjoint(pos, flags,
1459                                 ABSTRACT | NATIVE,
1460                                 STRICTFP))) {
1461             // skip
1462         }
1463         return flags & (mask | ~ExtendedStandardFlags) | implicit;
1464     }

1886     /** Check that this method conforms with overridden method 'other'.
1887      *  where `origin' is the class where checking started.
1888      *  Complications:
1889      *  (1) Do not check overriding of synthetic methods
1890      *      (reason: they might be final).
1891      *      todo: check whether this is still necessary.
1892      *  (2) Admit the case where an interface proxy throws fewer exceptions
1893      *      than the method it implements. Augment the proxy methods with the
1894      *      undeclared exceptions in this case.
1895      *  (3) When generics are enabled, admit the case where an interface proxy
1896      *      has a result type
1897      *      extended by the result type of the method it implements.
1898      *      Change the proxies result type to the smaller type in this case.
1899      *
1900      *  @param tree         The tree from which positions
1901      *                      are extracted for errors.
1902      *  @param m            The overriding method.
1903      *  @param other        The overridden method.
1904      *  @param origin       The class of which the overriding method
1905      *                      is a member.
1906      */
1907     void checkOverride(JCTree tree,
1908                        MethodSymbol m,
1909                        MethodSymbol other,
1910                        ClassSymbol origin) {
1911         // Don't check overriding of synthetic methods or by bridge methods.
1912         if ((m.flags() & (SYNTHETIC|BRIDGE)) != 0 || (other.flags() & SYNTHETIC) != 0) {
1913             return;
1914         }
1915 
1916         // Error if static method overrides instance method (JLS 8.4.6.2).
1917         if ((m.flags() & STATIC) != 0 &&
1918                    (other.flags() & STATIC) == 0) {
1919             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1920                       Errors.OverrideStatic(cannotOverride(m, other)));
1921             m.flags_field |= BAD_OVERRIDE;
1922             return;
1923         }
1924 
1925         // Error if instance method overrides static or final
1926         // method (JLS 8.4.6.1).
1927         if ((other.flags() & FINAL) != 0 ||
1928                  (m.flags() & STATIC) == 0 &&
1929                  (other.flags() & STATIC) != 0) {
1930             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1931                       Errors.OverrideMeth(cannotOverride(m, other),
1932                                           asFlagSet(other.flags() & (FINAL | STATIC))));
1933             m.flags_field |= BAD_OVERRIDE;
1934             return;
1935         }
1936 
1937         if ((m.owner.flags() & ANNOTATION) != 0) {
1938             // handled in validateAnnotationMethod
1939             return;
1940         }
1941 
1942         // Error if overriding method has weaker access (JLS 8.4.6.3).
1943         if (protection(m.flags()) > protection(other.flags())) {
1944             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1945                       (other.flags() & AccessFlags) == 0 ?
1946                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1947                                                           "package") :
1948                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1949                                                           asFlagSet(other.flags() & AccessFlags)));
1950             m.flags_field |= BAD_OVERRIDE;
1951             return;
1952         }
1953 
1954         if (origin.isValue() && other.owner == syms.objectType.tsym && m.type.getParameterTypes().size() == 0) {
1955             if (m.name == names.clone || m.name == names.finalize) {
1956                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1957                         Errors.InlineClassMayNotOverride(m.name));
1958                 m.flags_field |= BAD_OVERRIDE;
1959                 return;
1960             }
1961         }
1962 
1963         Type mt = types.memberType(origin.type, m);
1964         Type ot = types.memberType(origin.type, other);
1965         // Error if overriding result type is different
1966         // (or, in the case of generics mode, not a subtype) of
1967         // overridden result type. We have to rename any type parameters
1968         // before comparing types.
1969         List<Type> mtvars = mt.getTypeArguments();
1970         List<Type> otvars = ot.getTypeArguments();
1971         Type mtres = mt.getReturnType();
1972         Type otres = types.subst(ot.getReturnType(), otvars, mtvars);
1973 
1974         overrideWarner.clear();
1975         boolean resultTypesOK =
1976             types.returnTypeSubstitutable(mt, ot, otres, overrideWarner);
1977         if (!resultTypesOK) {
1978             if ((m.flags() & STATIC) != 0 && (other.flags() & STATIC) != 0) {
1979                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1980                           Errors.OverrideIncompatibleRet(Fragments.CantHide(m, m.location(), other,
1981                                         other.location()), mtres, otres));
1982                 m.flags_field |= BAD_OVERRIDE;
1983             } else {
1984                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1985                           Errors.OverrideIncompatibleRet(cannotOverride(m, other), mtres, otres));
1986                 m.flags_field |= BAD_OVERRIDE;
1987             }
1988             return;
1989         } else if (overrideWarner.hasNonSilentLint(LintCategory.UNCHECKED)) {
1990             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
1991                     Warnings.OverrideUncheckedRet(uncheckedOverrides(m, other), mtres, otres));
1992         }
1993 
1994         // Error if overriding method throws an exception not reported
1995         // by overridden method.
1996         List<Type> otthrown = types.subst(ot.getThrownTypes(), otvars, mtvars);
1997         List<Type> unhandledErased = unhandled(mt.getThrownTypes(), types.erasure(otthrown));
1998         List<Type> unhandledUnerased = unhandled(mt.getThrownTypes(), otthrown);
1999         if (unhandledErased.nonEmpty()) {
2000             log.error(TreeInfo.diagnosticPositionFor(m, tree),
2001                       Errors.OverrideMethDoesntThrow(cannotOverride(m, other), unhandledUnerased.head));
2002             m.flags_field |= BAD_OVERRIDE;
2003             return;
2004         }
2005         else if (unhandledUnerased.nonEmpty()) {
2006             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
2007                           Warnings.OverrideUncheckedThrown(cannotOverride(m, other), unhandledUnerased.head));
2008             return;
2009         }
2010 
2011         // Optional warning if varargs don't agree
2012         if ((((m.flags() ^ other.flags()) & Flags.VARARGS) != 0)
2013             && lint.isEnabled(LintCategory.OVERRIDES)) {
2014             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2015                         ((m.flags() & Flags.VARARGS) != 0)
2016                         ? Warnings.OverrideVarargsMissing(varargsOverrides(m, other))
2017                         : Warnings.OverrideVarargsExtra(varargsOverrides(m, other)));
2018         }
2019 
2020         // Warn if instance method overrides bridge method (compiler spec ??)
2021         if ((other.flags() & BRIDGE) != 0) {
2022             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2023                         Warnings.OverrideBridge(uncheckedOverrides(m, other)));
2024         }
2025 
2026         // Warn if a deprecated method overridden by a non-deprecated one.
2027         if (!isDeprecatedOverrideIgnorable(other, origin)) {
2028             Lint prevLint = setLint(lint.augment(m));
2029             try {
2030                 checkDeprecated(() -> TreeInfo.diagnosticPositionFor(m, tree), m, other);
2031             } finally {
2032                 setLint(prevLint);
2033             }
2034         }
2035     }

2865     /**
2866       * Report warnings for potentially ambiguous method declarations. Two declarations
2867       * are potentially ambiguous if they feature two unrelated functional interface
2868       * in same argument position (in which case, a call site passing an implicit
2869       * lambda would be ambiguous).
2870       */
2871     void checkPotentiallyAmbiguousOverloads(DiagnosticPosition pos, Type site,
2872             MethodSymbol msym1, MethodSymbol msym2) {
2873         if (msym1 != msym2 &&
2874                 Feature.DEFAULT_METHODS.allowedInSource(source) &&
2875                 lint.isEnabled(LintCategory.OVERLOADS) &&
2876                 (msym1.flags() & POTENTIALLY_AMBIGUOUS) == 0 &&
2877                 (msym2.flags() & POTENTIALLY_AMBIGUOUS) == 0) {
2878             Type mt1 = types.memberType(site, msym1);
2879             Type mt2 = types.memberType(site, msym2);
2880             //if both generic methods, adjust type variables
2881             if (mt1.hasTag(FORALL) && mt2.hasTag(FORALL) &&
2882                     types.hasSameBounds((ForAll)mt1, (ForAll)mt2)) {
2883                 mt2 = types.subst(mt2, ((ForAll)mt2).tvars, ((ForAll)mt1).tvars);
2884             }
2885             //expand varargs methods if needed
2886             int maxLength = Math.max(mt1.getParameterTypes().length(), mt2.getParameterTypes().length());
2887             List<Type> args1 = rs.adjustArgs(mt1.getParameterTypes(), msym1, maxLength, true);
2888             List<Type> args2 = rs.adjustArgs(mt2.getParameterTypes(), msym2, maxLength, true);
2889             //if arities don't match, exit
2890             if (args1.length() != args2.length()) return;
2891             boolean potentiallyAmbiguous = false;
2892             while (args1.nonEmpty() && args2.nonEmpty()) {
2893                 Type s = args1.head;
2894                 Type t = args2.head;
2895                 if (!types.isSubtype(t, s) && !types.isSubtype(s, t)) {
2896                     if (types.isFunctionalInterface(s) && types.isFunctionalInterface(t) &&
2897                             types.findDescriptorType(s).getParameterTypes().length() > 0 &&
2898                             types.findDescriptorType(s).getParameterTypes().length() ==
2899                             types.findDescriptorType(t).getParameterTypes().length()) {
2900                         potentiallyAmbiguous = true;
2901                     } else {
2902                         break;
2903                     }
2904                 }
2905                 args1 = args1.tail;
2906                 args2 = args2.tail;
2907             }
2908             if (potentiallyAmbiguous) {
2909                 //we found two incompatible functional interfaces with same arity
2910                 //this means a call site passing an implicit lambda would be ambiguous
2911                 msym1.flags_field |= POTENTIALLY_AMBIGUOUS;
2912                 msym2.flags_field |= POTENTIALLY_AMBIGUOUS;
2913                 log.warning(LintCategory.OVERLOADS, pos,
2914                             Warnings.PotentiallyAmbiguousOverload(msym1, msym1.location(),
2915                                                                   msym2, msym2.location()));
2916                 return;
2917             }
2918         }
