26 /*
27  * @test
28  * @bug 8222555 8222553
29  * @summary Prove that code suspected of not compiling actually compiles fine.
30  * @compile -XDallowWithFieldOperator CompilesJustFine.java
31  */
32 
33 class CompilesFine {
34 
35     static Point.ref nfspQm;
36 
37     public static void main(String[] args) {
38         nfspQm = null;
39     }
40 }
41 inline final class Point {
42     final int x;
43     final int y;
44 
45     Point() {
46         x = 0;
47         y = 0;
48     }
49 }
50 
51 class CompilesJustFine {
52 
53     static final inline class Value {
54         final PointBug2.ref nfpQm;
55 
56         private Value() {
57             nfpQm = PointBug2.createPoint(0, 0);
58         }
59     }
60 }
61 inline final class PointBug2 {
62     final int x;
63     final int y;
64 
65     PointBug2() {
66         x = 0;
67         y = 0;
68     }
69 
70     public static PointBug2 createPoint(int x, int y) {
71         PointBug2 p = PointBug2.default;
72         p = __WithField(p.x, x);
73         p = __WithField(p.y, y);
74         return p;
75     }
76 }
