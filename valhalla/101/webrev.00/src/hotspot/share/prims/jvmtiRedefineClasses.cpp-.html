  95 // If any of the classes are being redefined, wait
  96 // Parallel constant pool merging leads to indeterminate constant pools.
  97 void VM_RedefineClasses::lock_classes() {
  98   MonitorLocker ml(RedefineClasses_lock);
  99   bool has_redefined;
 100   do {
 101     has_redefined = false;
 102     // Go through classes each time until none are being redefined.
 103     for (int i = 0; i < _class_count; i++) {
 104       if (get_ik(_class_defs[i].klass)->is_being_redefined()) {
 105         ml.wait();
 106         has_redefined = true;
 107         break;  // for loop
 108       }
 109     }
 110   } while (has_redefined);
 111   for (int i = 0; i < _class_count; i++) {
 112     get_ik(_class_defs[i].klass)->set_is_being_redefined(true);
 113   }
 114   ml.notify_all();
 115 }
 116 
 117 void VM_RedefineClasses::unlock_classes() {
 118   MonitorLocker ml(RedefineClasses_lock);
 119   for (int i = 0; i < _class_count; i++) {
 120     assert(get_ik(_class_defs[i].klass)->is_being_redefined(),
 121            "should be being redefined to get here");
 122     get_ik(_class_defs[i].klass)->set_is_being_redefined(false);
 123   }
 124   ml.notify_all();
 125 }

 127 bool VM_RedefineClasses::doit_prologue() {
 128   if (_class_count == 0) {
 129     _res = JVMTI_ERROR_NONE;
 130     return false;
 131   }
 132   if (_class_defs == NULL) {
 133     _res = JVMTI_ERROR_NULL_POINTER;
 134     return false;
 135   }
 136 
 137   for (int i = 0; i < _class_count; i++) {
 138     if (_class_defs[i].klass == NULL) {
 139       _res = JVMTI_ERROR_INVALID_CLASS;
 140       return false;
 141     }
 142     if (_class_defs[i].class_byte_count == 0) {
 143       _res = JVMTI_ERROR_INVALID_CLASS_FORMAT;
 144       return false;
 145     }
 146     if (_class_defs[i].class_bytes == NULL) {
 147       _res = JVMTI_ERROR_NULL_POINTER;
 148       return false;
 149     }
 150 
 151     oop mirror = JNIHandles::resolve_non_null(_class_defs[i].klass);
 152     // classes for primitives, arrays, hidden and vm unsafe anonymous classes
 153     // cannot be redefined.
 154     if (!is_modifiable_class(mirror)) {
 155       _res = JVMTI_ERROR_UNMODIFIABLE_CLASS;
 156       return false;
 157     }
 158   }
 159 
 160   // Start timer after all the sanity checks; not quite accurate, but
 161   // better than adding a bunch of stop() calls.
 162   if (log_is_enabled(Info, redefine, class, timer)) {
 163     _timer_vm_op_prologue.start();
 164   }
 165 
 166   lock_classes();
 167   // We first load new class versions in the prologue, because somewhere down the
 168   // call chain it is required that the current thread is a Java thread.
 169   _res = load_new_class_versions(Thread::current());
 170   if (_res != JVMTI_ERROR_NONE) {
 171     // free any successfully created classes, since none are redefined
 172     for (int i = 0; i < _class_count; i++) {
 173       if (_scratch_classes[i] != NULL) {
 174         ClassLoaderData* cld = _scratch_classes[i]->class_loader_data();
 175         // Free the memory for this class at class unloading time.  Not before
 176         // because CMS might think this is still live.
 177         InstanceKlass* ik = get_ik(_class_defs[i].klass);
 178         if (ik->get_cached_class_file() == _scratch_classes[i]->get_cached_class_file()) {
 179           // Don't double-free cached_class_file copied from the original class if error.
 180           _scratch_classes[i]->set_cached_class_file(NULL);
 181         }
 182         cld->add_to_deallocate_list(InstanceKlass::cast(_scratch_classes[i]));
 183       }
 184     }
 185     // Free os::malloc allocated memory in load_new_class_version.
 186     os::free(_scratch_classes);
 187     _timer_vm_op_prologue.stop();
 188     unlock_classes();
 189     return false;
 190   }
 191 
 192   _timer_vm_op_prologue.stop();
 193   return true;
 194 }

 196 void VM_RedefineClasses::doit() {
 197   Thread *thread = Thread::current();
 198 
 199 #if INCLUDE_CDS
 200   if (UseSharedSpaces) {
 201     // Sharing is enabled so we remap the shared readonly space to
 202     // shared readwrite, private just in case we need to redefine
 203     // a shared class. We do the remap during the doit() phase of
 204     // the safepoint to be safer.
 205     if (!MetaspaceShared::remap_shared_readonly_as_readwrite()) {
 206       log_info(redefine, class, load)("failed to remap shared readonly space to readwrite, private");
 207       _res = JVMTI_ERROR_INTERNAL;
 208       return;
 209     }
 210   }
 211 #endif
 212 
 213   // Mark methods seen on stack and everywhere else so old methods are not
 214   // cleaned up if they're on the stack.
 215   MetadataOnStackMark md_on_stack(/*walk_all_metadata*/true, /*redefinition_walk*/true);
 216   HandleMark hm(thread);   // make sure any handles created are deleted
 217                            // before the stack walk again.
 218 
 219   for (int i = 0; i < _class_count; i++) {
 220     redefine_single_class(_class_defs[i].klass, _scratch_classes[i], thread);
 221   }
 222 
 223   // Flush all compiled code that depends on the classes redefined.
 224   flush_dependent_code();
 225 
 226   // Adjust constantpool caches and vtables for all classes
 227   // that reference methods of the evolved classes.
 228   // Have to do this after all classes are redefined and all methods that
 229   // are redefined are marked as old.
 230   AdjustAndCleanMetadata adjust_and_clean_metadata(thread);
 231   ClassLoaderDataGraph::classes_do(&adjust_and_clean_metadata);
 232 
 233   // JSR-292 support
 234   if (_any_class_has_resolved_methods) {
 235     bool trace_name_printed = false;
 236     ResolvedMethodTable::adjust_method_entries(&trace_name_printed);
 237   }
 238 
 239   // Increment flag indicating that some invariants are no longer true.
 240   // See jvmtiExport.hpp for detailed explanation.
 241   JvmtiExport::increment_redefinition_count();
 242 
 243   // check_class() is optionally called for product bits, but is
 244   // always called for non-product bits.
 245 #ifdef PRODUCT
 246   if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
 247 #endif
 248     log_trace(redefine, class, obsolete, metadata)("calling check_class");
 249     CheckClass check_class(thread);
 250     ClassLoaderDataGraph::classes_do(&check_class);
 251 #ifdef PRODUCT
 252   }
 253 #endif
 254 
 255   // Clean up any metadata now unreferenced while MetadataOnStackMark is set.
 256   ClassLoaderDataGraph::clean_deallocate_lists(false);
 257 }

 259 void VM_RedefineClasses::doit_epilogue() {
 260   unlock_classes();
 261 
 262   // Free os::malloc allocated memory.
 263   os::free(_scratch_classes);
 264 
 265   // Reset the_class to null for error printing.
 266   _the_class = NULL;
 267 
 268   if (log_is_enabled(Info, redefine, class, timer)) {
 269     // Used to have separate timers for "doit" and "all", but the timer
 270     // overhead skewed the measurements.
 271     julong doit_time = _timer_rsc_phase1.milliseconds() +
 272                        _timer_rsc_phase2.milliseconds();
 273     julong all_time = _timer_vm_op_prologue.milliseconds() + doit_time;
 274 
 275     log_info(redefine, class, timer)
 276       ("vm_op: all=" JULONG_FORMAT "  prologue=" JULONG_FORMAT "  doit=" JULONG_FORMAT,
 277        all_time, (julong)_timer_vm_op_prologue.milliseconds(), doit_time);
 278     log_info(redefine, class, timer)
 279       ("redefine_single_class: phase1=" JULONG_FORMAT "  phase2=" JULONG_FORMAT,
 280        (julong)_timer_rsc_phase1.milliseconds(), (julong)_timer_rsc_phase2.milliseconds());
 281   }
 282 }

 284 bool VM_RedefineClasses::is_modifiable_class(oop klass_mirror) {
 285   // classes for primitives cannot be redefined
 286   if (java_lang_Class::is_primitive(klass_mirror)) {
 287     return false;
 288   }
 289   Klass* k = java_lang_Class::as_Klass(klass_mirror);
 290   // classes for arrays cannot be redefined
 291   if (k == NULL || !k->is_instance_klass()) {
 292     return false;
 293   }
 294 
 295   // Cannot redefine or retransform a hidden or an unsafe anonymous class.
 296   if (InstanceKlass::cast(k)->is_hidden() ||
 297       InstanceKlass::cast(k)->is_unsafe_anonymous()) {
 298     return false;
 299   }
 300   return true;
 301 }

 303 // Append the current entry at scratch_i in scratch_cp to *merge_cp_p
 304 // where the end of *merge_cp_p is specified by *merge_cp_length_p. For
 305 // direct CP entries, there is just the current entry to append. For
 306 // indirect and double-indirect CP entries, there are zero or more
 307 // referenced CP entries along with the current entry to append.
 308 // Indirect and double-indirect CP entries are handled by recursive
 309 // calls to append_entry() as needed. The referenced CP entries are
 310 // always appended to *merge_cp_p before the referee CP entry. These
 311 // referenced CP entries may already exist in *merge_cp_p in which case
 312 // there is nothing extra to append and only the current entry is
 313 // appended.
 314 void VM_RedefineClasses::append_entry(const constantPoolHandle& scratch_cp,
 315        int scratch_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p,
 316        TRAPS) {
 317 
 318   // append is different depending on entry tag type
 319   switch (scratch_cp->tag_at(scratch_i).value()) {
 320 
 321     // The old verifier is implemented outside the VM. It loads classes,
 322     // but does not resolve constant pool entries directly so we never
 323     // see Class entries here with the old verifier. Similarly the old
 324     // verifier does not like Class entries in the input constant pool.
 325     // The split-verifier is implemented in the VM so it can optionally
 326     // and directly resolve constant pool entries to load classes. The
 327     // split-verifier can accept either Class entries or UnresolvedClass
 328     // entries in the input constant pool. We revert the appended copy
 329     // back to UnresolvedClass so that either verifier will be happy
 330     // with the constant pool entry.
 331     //
 332     // this is an indirect CP entry so it needs special handling
 333     case JVM_CONSTANT_Class:
 334     case JVM_CONSTANT_UnresolvedClass:
 335     {
 336       int name_i = scratch_cp->klass_name_index_at(scratch_i);
 337       int new_name_i = find_or_append_indirect_entry(scratch_cp, name_i, merge_cp_p,
 338                                                      merge_cp_length_p, THREAD);
 339 
 340       if (new_name_i != name_i) {
 341         log_trace(redefine, class, constantpool)
 342           ("Class entry@%d name_index change: %d to %d",
 343            *merge_cp_length_p, name_i, new_name_i);
 344       }
 345 
 346       (*merge_cp_p)->temp_unresolved_klass_at_put(*merge_cp_length_p, new_name_i);
 347       if (scratch_i != *merge_cp_length_p) {
 348         // The new entry in *merge_cp_p is at a different index than
 349         // the new entry in scratch_cp so we need to map the index values.
 350         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 351       }
 352       (*merge_cp_length_p)++;
 353     } break;
 354 
 355     // these are direct CP entries so they can be directly appended,
 356     // but double and long take two constant pool entries
 357     case JVM_CONSTANT_Double:  // fall through
 358     case JVM_CONSTANT_Long:
 359     {
 360       ConstantPool::copy_entry_to(scratch_cp, scratch_i, *merge_cp_p, *merge_cp_length_p,
 361         THREAD);
 362 
 363       if (scratch_i != *merge_cp_length_p) {
 364         // The new entry in *merge_cp_p is at a different index than
 365         // the new entry in scratch_cp so we need to map the index values.
 366         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 367       }
 368       (*merge_cp_length_p) += 2;
 369     } break;
 370 
 371     // these are direct CP entries so they can be directly appended
 372     case JVM_CONSTANT_Float:   // fall through
 373     case JVM_CONSTANT_Integer: // fall through
 374     case JVM_CONSTANT_Utf8:    // fall through
 375 
 376     // This was an indirect CP entry, but it has been changed into
 377     // Symbol*s so this entry can be directly appended.
 378     case JVM_CONSTANT_String:      // fall through
 379     {
 380       ConstantPool::copy_entry_to(scratch_cp, scratch_i, *merge_cp_p, *merge_cp_length_p,
 381         THREAD);
 382 
 383       if (scratch_i != *merge_cp_length_p) {
 384         // The new entry in *merge_cp_p is at a different index than
 385         // the new entry in scratch_cp so we need to map the index values.
 386         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 387       }
 388       (*merge_cp_length_p)++;
 389     } break;
 390 
 391     // this is an indirect CP entry so it needs special handling
 392     case JVM_CONSTANT_NameAndType:
 393     {
 394       int name_ref_i = scratch_cp->name_ref_index_at(scratch_i);
 395       int new_name_ref_i = find_or_append_indirect_entry(scratch_cp, name_ref_i, merge_cp_p,
 396                                                          merge_cp_length_p, THREAD);
 397 
 398       int signature_ref_i = scratch_cp->signature_ref_index_at(scratch_i);
 399       int new_signature_ref_i = find_or_append_indirect_entry(scratch_cp, signature_ref_i,
 400                                                               merge_cp_p, merge_cp_length_p,
 401                                                               THREAD);
 402 
 403       // If the referenced entries already exist in *merge_cp_p, then
 404       // both new_name_ref_i and new_signature_ref_i will both be 0.
 405       // In that case, all we are appending is the current entry.
 406       if (new_name_ref_i != name_ref_i) {
 407         log_trace(redefine, class, constantpool)
 408           ("NameAndType entry@%d name_ref_index change: %d to %d",
 409            *merge_cp_length_p, name_ref_i, new_name_ref_i);
 410       }
 411       if (new_signature_ref_i != signature_ref_i) {
 412         log_trace(redefine, class, constantpool)
 413           ("NameAndType entry@%d signature_ref_index change: %d to %d",
 414            *merge_cp_length_p, signature_ref_i, new_signature_ref_i);
 415       }
 416 
 417       (*merge_cp_p)->name_and_type_at_put(*merge_cp_length_p,
 418         new_name_ref_i, new_signature_ref_i);
 419       if (scratch_i != *merge_cp_length_p) {
 420         // The new entry in *merge_cp_p is at a different index than
 421         // the new entry in scratch_cp so we need to map the index values.
 422         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 423       }
 424       (*merge_cp_length_p)++;
 425     } break;
 426 
 427     // this is a double-indirect CP entry so it needs special handling
 428     case JVM_CONSTANT_Fieldref:           // fall through
 429     case JVM_CONSTANT_InterfaceMethodref: // fall through
 430     case JVM_CONSTANT_Methodref:
 431     {
 432       int klass_ref_i = scratch_cp->uncached_klass_ref_index_at(scratch_i);
 433       int new_klass_ref_i = find_or_append_indirect_entry(scratch_cp, klass_ref_i,
 434                                                           merge_cp_p, merge_cp_length_p, THREAD);
 435 
 436       int name_and_type_ref_i = scratch_cp->uncached_name_and_type_ref_index_at(scratch_i);
 437       int new_name_and_type_ref_i = find_or_append_indirect_entry(scratch_cp, name_and_type_ref_i,
 438                                                           merge_cp_p, merge_cp_length_p, THREAD);
 439 
 440       const char *entry_name = NULL;
 441       switch (scratch_cp->tag_at(scratch_i).value()) {
 442       case JVM_CONSTANT_Fieldref:
 443         entry_name = "Fieldref";
 444         (*merge_cp_p)->field_at_put(*merge_cp_length_p, new_klass_ref_i,
 445           new_name_and_type_ref_i);
 446         break;
 447       case JVM_CONSTANT_InterfaceMethodref:
 448         entry_name = "IFMethodref";
 449         (*merge_cp_p)->interface_method_at_put(*merge_cp_length_p,
 450           new_klass_ref_i, new_name_and_type_ref_i);
 451         break;
 452       case JVM_CONSTANT_Methodref:
 453         entry_name = "Methodref";
 454         (*merge_cp_p)->method_at_put(*merge_cp_length_p, new_klass_ref_i,
 455           new_name_and_type_ref_i);
 456         break;
 457       default:
 458         guarantee(false, "bad switch");
 459         break;
 460       }
 461 
 462       if (klass_ref_i != new_klass_ref_i) {
 463         log_trace(redefine, class, constantpool)
 464           ("%s entry@%d class_index changed: %d to %d", entry_name, *merge_cp_length_p, klass_ref_i, new_klass_ref_i);
 465       }
 466       if (name_and_type_ref_i != new_name_and_type_ref_i) {
 467         log_trace(redefine, class, constantpool)
 468           ("%s entry@%d name_and_type_index changed: %d to %d",
 469            entry_name, *merge_cp_length_p, name_and_type_ref_i, new_name_and_type_ref_i);
 470       }
 471 
 472       if (scratch_i != *merge_cp_length_p) {
 473         // The new entry in *merge_cp_p is at a different index than
 474         // the new entry in scratch_cp so we need to map the index values.
 475         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 476       }
 477       (*merge_cp_length_p)++;
 478     } break;
 479 
 480     // this is an indirect CP entry so it needs special handling
 481     case JVM_CONSTANT_MethodType:
 482     {
 483       int ref_i = scratch_cp->method_type_index_at(scratch_i);
 484       int new_ref_i = find_or_append_indirect_entry(scratch_cp, ref_i, merge_cp_p,
 485                                                     merge_cp_length_p, THREAD);
 486       if (new_ref_i != ref_i) {
 487         log_trace(redefine, class, constantpool)
 488           ("MethodType entry@%d ref_index change: %d to %d", *merge_cp_length_p, ref_i, new_ref_i);
 489       }
 490       (*merge_cp_p)->method_type_index_at_put(*merge_cp_length_p, new_ref_i);
 491       if (scratch_i != *merge_cp_length_p) {
 492         // The new entry in *merge_cp_p is at a different index than
 493         // the new entry in scratch_cp so we need to map the index values.
 494         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 495       }
 496       (*merge_cp_length_p)++;
 497     } break;
 498 
 499     // this is an indirect CP entry so it needs special handling
 500     case JVM_CONSTANT_MethodHandle:
 501     {
 502       int ref_kind = scratch_cp->method_handle_ref_kind_at(scratch_i);
 503       int ref_i = scratch_cp->method_handle_index_at(scratch_i);
 504       int new_ref_i = find_or_append_indirect_entry(scratch_cp, ref_i, merge_cp_p,
 505                                                     merge_cp_length_p, THREAD);
 506       if (new_ref_i != ref_i) {
 507         log_trace(redefine, class, constantpool)
 508           ("MethodHandle entry@%d ref_index change: %d to %d", *merge_cp_length_p, ref_i, new_ref_i);
 509       }
 510       (*merge_cp_p)->method_handle_index_at_put(*merge_cp_length_p, ref_kind, new_ref_i);
 511       if (scratch_i != *merge_cp_length_p) {
 512         // The new entry in *merge_cp_p is at a different index than
 513         // the new entry in scratch_cp so we need to map the index values.
 514         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 515       }
 516       (*merge_cp_length_p)++;
 517     } break;
 518 
 519     // this is an indirect CP entry so it needs special handling
 520     case JVM_CONSTANT_Dynamic:  // fall through
 521     case JVM_CONSTANT_InvokeDynamic:
 522     {
 523       // Index of the bootstrap specifier in the operands array
 524       int old_bs_i = scratch_cp->bootstrap_methods_attribute_index(scratch_i);
 525       int new_bs_i = find_or_append_operand(scratch_cp, old_bs_i, merge_cp_p,
 526                                             merge_cp_length_p, THREAD);
 527       // The bootstrap method NameAndType_info index
 528       int old_ref_i = scratch_cp->bootstrap_name_and_type_ref_index_at(scratch_i);
 529       int new_ref_i = find_or_append_indirect_entry(scratch_cp, old_ref_i, merge_cp_p,
 530                                                     merge_cp_length_p, THREAD);
 531       if (new_bs_i != old_bs_i) {
 532         log_trace(redefine, class, constantpool)
 533           ("Dynamic entry@%d bootstrap_method_attr_index change: %d to %d",
 534            *merge_cp_length_p, old_bs_i, new_bs_i);
 535       }
 536       if (new_ref_i != old_ref_i) {
 537         log_trace(redefine, class, constantpool)
 538           ("Dynamic entry@%d name_and_type_index change: %d to %d", *merge_cp_length_p, old_ref_i, new_ref_i);
 539       }
 540 
 541       if (scratch_cp->tag_at(scratch_i).is_dynamic_constant())
 542         (*merge_cp_p)->dynamic_constant_at_put(*merge_cp_length_p, new_bs_i, new_ref_i);
 543       else
 544         (*merge_cp_p)->invoke_dynamic_at_put(*merge_cp_length_p, new_bs_i, new_ref_i);
 545       if (scratch_i != *merge_cp_length_p) {
 546         // The new entry in *merge_cp_p is at a different index than
 547         // the new entry in scratch_cp so we need to map the index values.
 548         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 549       }
 550       (*merge_cp_length_p)++;
 551     } break;
 552 
 553     // At this stage, Class or UnresolvedClass could be in scratch_cp, but not
 554     // ClassIndex
 555     case JVM_CONSTANT_ClassIndex: // fall through
 556 
 557     // Invalid is used as the tag for the second constant pool entry
 558     // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
 559     // not be seen by itself.
 560     case JVM_CONSTANT_Invalid: // fall through
 561 
 562     // At this stage, String could be here, but not StringIndex
 563     case JVM_CONSTANT_StringIndex: // fall through
 564 
 565     // At this stage JVM_CONSTANT_UnresolvedClassInError should not be here
 566     case JVM_CONSTANT_UnresolvedClassInError: // fall through
 567 
 568     default:
 569     {
 570       // leave a breadcrumb
 571       jbyte bad_value = scratch_cp->tag_at(scratch_i).value();
 572       ShouldNotReachHere();
 573     } break;
 574   } // end switch tag value
 575 } // end append_entry()

 578 int VM_RedefineClasses::find_or_append_indirect_entry(const constantPoolHandle& scratch_cp,
 579       int ref_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 580 
 581   int new_ref_i = ref_i;
 582   bool match = (ref_i < *merge_cp_length_p) &&
 583                scratch_cp->compare_entry_to(ref_i, *merge_cp_p, ref_i, THREAD);
 584 
 585   if (!match) {
 586     // forward reference in *merge_cp_p or not a direct match
 587     int found_i = scratch_cp->find_matching_entry(ref_i, *merge_cp_p, THREAD);
 588     if (found_i != 0) {
 589       guarantee(found_i != ref_i, "compare_entry_to() and find_matching_entry() do not agree");
 590       // Found a matching entry somewhere else in *merge_cp_p so just need a mapping entry.
 591       new_ref_i = found_i;
 592       map_index(scratch_cp, ref_i, found_i);
 593     } else {
 594       // no match found so we have to append this entry to *merge_cp_p
 595       append_entry(scratch_cp, ref_i, merge_cp_p, merge_cp_length_p, THREAD);
 596       // The above call to append_entry() can only append one entry
 597       // so the post call query of *merge_cp_length_p is only for
 598       // the sake of consistency.
 599       new_ref_i = *merge_cp_length_p - 1;
 600     }
 601   }
 602 
 603   return new_ref_i;
 604 } // end find_or_append_indirect_entry()

 607 // Append a bootstrap specifier into the merge_cp operands that is semantically equal
 608 // to the scratch_cp operands bootstrap specifier passed by the old_bs_i index.
 609 // Recursively append new merge_cp entries referenced by the new bootstrap specifier.
 610 void VM_RedefineClasses::append_operand(const constantPoolHandle& scratch_cp, int old_bs_i,
 611        constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 612 
 613   int old_ref_i = scratch_cp->operand_bootstrap_method_ref_index_at(old_bs_i);
 614   int new_ref_i = find_or_append_indirect_entry(scratch_cp, old_ref_i, merge_cp_p,
 615                                                 merge_cp_length_p, THREAD);
 616   if (new_ref_i != old_ref_i) {
 617     log_trace(redefine, class, constantpool)
 618       ("operands entry@%d bootstrap method ref_index change: %d to %d", _operands_cur_length, old_ref_i, new_ref_i);
 619   }
 620 
 621   Array<u2>* merge_ops = (*merge_cp_p)->operands();
 622   int new_bs_i = _operands_cur_length;
 623   // We have _operands_cur_length == 0 when the merge_cp operands is empty yet.
 624   // However, the operand_offset_at(0) was set in the extend_operands() call.
 625   int new_base = (new_bs_i == 0) ? (*merge_cp_p)->operand_offset_at(0)
 626                                  : (*merge_cp_p)->operand_next_offset_at(new_bs_i - 1);
 627   int argc     = scratch_cp->operand_argument_count_at(old_bs_i);
 628 
 629   ConstantPool::operand_offset_at_put(merge_ops, _operands_cur_length, new_base);
 630   merge_ops->at_put(new_base++, new_ref_i);
 631   merge_ops->at_put(new_base++, argc);
 632 
 633   for (int i = 0; i < argc; i++) {
 634     int old_arg_ref_i = scratch_cp->operand_argument_index_at(old_bs_i, i);
 635     int new_arg_ref_i = find_or_append_indirect_entry(scratch_cp, old_arg_ref_i, merge_cp_p,
 636                                                       merge_cp_length_p, THREAD);
 637     merge_ops->at_put(new_base++, new_arg_ref_i);
 638     if (new_arg_ref_i != old_arg_ref_i) {
 639       log_trace(redefine, class, constantpool)
 640         ("operands entry@%d bootstrap method argument ref_index change: %d to %d",
 641          _operands_cur_length, old_arg_ref_i, new_arg_ref_i);
 642     }
 643   }
 644   if (old_bs_i != _operands_cur_length) {
 645     // The bootstrap specifier in *merge_cp_p is at a different index than
 646     // that in scratch_cp so we need to map the index values.
 647     map_operand_index(old_bs_i, new_bs_i);
 648   }
 649   _operands_cur_length++;
 650 } // end append_operand()

 653 int VM_RedefineClasses::find_or_append_operand(const constantPoolHandle& scratch_cp,
 654       int old_bs_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 655 
 656   int new_bs_i = old_bs_i; // bootstrap specifier index
 657   bool match = (old_bs_i < _operands_cur_length) &&
 658                scratch_cp->compare_operand_to(old_bs_i, *merge_cp_p, old_bs_i, THREAD);
 659 
 660   if (!match) {
 661     // forward reference in *merge_cp_p or not a direct match
 662     int found_i = scratch_cp->find_matching_operand(old_bs_i, *merge_cp_p,
 663                                                     _operands_cur_length, THREAD);
 664     if (found_i != -1) {
 665       guarantee(found_i != old_bs_i, "compare_operand_to() and find_matching_operand() disagree");
 666       // found a matching operand somewhere else in *merge_cp_p so just need a mapping
 667       new_bs_i = found_i;
 668       map_operand_index(old_bs_i, found_i);
 669     } else {
 670       // no match found so we have to append this bootstrap specifier to *merge_cp_p
 671       append_operand(scratch_cp, old_bs_i, merge_cp_p, merge_cp_length_p, THREAD);
 672       new_bs_i = _operands_cur_length - 1;
 673     }
 674   }
 675   return new_bs_i;
 676 } // end find_or_append_operand()

 711 // The caller must have an active ResourceMark.
 712 static jvmtiError check_attribute_arrays(const char* attr_name,
 713            InstanceKlass* the_class, InstanceKlass* scratch_class,
 714            Array<u2>* the_array, Array<u2>* scr_array) {
 715   bool the_array_exists = the_array != Universe::the_empty_short_array();
 716   bool scr_array_exists = scr_array != Universe::the_empty_short_array();
 717 
 718   int array_len = the_array->length();
 719   if (the_array_exists && scr_array_exists) {
 720     if (array_len != scr_array->length()) {
 721       log_trace(redefine, class)
 722         ("redefined class %s attribute change error: %s len=%d changed to len=%d",
 723          the_class->external_name(), attr_name, array_len, scr_array->length());
 724       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 725     }
 726 
 727     // The order of entries in the attribute array is not specified so we
 728     // have to explicitly check for the same contents. We do this by copying
 729     // the referenced symbols into their own arrays, sorting them and then
 730     // comparing each element pair.
 731 
 732     Symbol** the_syms = NEW_RESOURCE_ARRAY_RETURN_NULL(Symbol*, array_len);
 733     Symbol** scr_syms = NEW_RESOURCE_ARRAY_RETURN_NULL(Symbol*, array_len);
 734 
 735     if (the_syms == NULL || scr_syms == NULL) {
 736       return JVMTI_ERROR_OUT_OF_MEMORY;
 737     }
 738 
 739     for (int i = 0; i < array_len; i++) {
 740       int the_cp_index = the_array->at(i);
 741       int scr_cp_index = scr_array->at(i);
 742       the_syms[i] = the_class->constants()->klass_name_at(the_cp_index);
 743       scr_syms[i] = scratch_class->constants()->klass_name_at(scr_cp_index);
 744     }
 745 
 746     qsort(the_syms, array_len, sizeof(Symbol*), symcmp);
 747     qsort(scr_syms, array_len, sizeof(Symbol*), symcmp);
 748 
 749     for (int i = 0; i < array_len; i++) {
 750       if (the_syms[i] != scr_syms[i]) {
 751         log_trace(redefine, class)
 752           ("redefined class %s attribute change error: %s[%d]: %s changed to %s",
 753            the_class->external_name(), attr_name, i,
 754            the_syms[i]->as_C_string(), scr_syms[i]->as_C_string());
 755         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 756       }
 757     }
 758   } else if (the_array_exists ^ scr_array_exists) {
 759     const char* action_str = (the_array_exists) ? "removed" : "added";
 760     log_trace(redefine, class)
 761       ("redefined class %s attribute change error: %s attribute %s",
 762        the_class->external_name(), attr_name, action_str);
 763     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 764   }
 765   return JVMTI_ERROR_NONE;
 766 }

 802   // Get lists of record components.
 803   Array<RecordComponent*>* the_record = the_class->record_components();
 804   Array<RecordComponent*>* scr_record = scratch_class->record_components();
 805   bool the_record_exists = the_record != NULL;
 806   bool scr_record_exists = scr_record != NULL;
 807 
 808   if (the_record_exists && scr_record_exists) {
 809     int the_num_components = the_record->length();
 810     int scr_num_components = scr_record->length();

 813         ("redefined class %s attribute change error: Record num_components=%d changed to num_components=%d",
 814          the_class->external_name(), the_num_components, scr_num_components);
 815       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 816     }
 817 
 818     // Compare each field in each record component.
 819     ConstantPool* the_cp =  the_class->constants();
 820     ConstantPool* scr_cp =  scratch_class->constants();
 821     for (int x = 0; x < the_num_components; x++) {
 822       RecordComponent* the_component = the_record->at(x);
 823       RecordComponent* scr_component = scr_record->at(x);
 824       const Symbol* const the_name = the_cp->symbol_at(the_component->name_index());
 825       const Symbol* const scr_name = scr_cp->symbol_at(scr_component->name_index());
 826       const Symbol* const the_descr = the_cp->symbol_at(the_component->descriptor_index());
 827       const Symbol* const scr_descr = scr_cp->symbol_at(scr_component->descriptor_index());
 828       if (the_name != scr_name || the_descr != scr_descr) {
 829         log_trace(redefine, class, record)
 830           ("redefined class %s attribute change error: Record name_index, descriptor_index, and/or attributes_count changed",
 831            the_class->external_name());
 832         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 833       }
 834 
 835       int the_gen_sig = the_component->generic_signature_index();
 836       int scr_gen_sig = scr_component->generic_signature_index();
 837       const Symbol* const the_gen_sig_sym = (the_gen_sig == 0 ? NULL :
 838         the_cp->symbol_at(the_component->generic_signature_index()));
 839       const Symbol* const scr_gen_sig_sym = (scr_gen_sig == 0 ? NULL :
 840         scr_cp->symbol_at(scr_component->generic_signature_index()));
 841       if (the_gen_sig_sym != scr_gen_sig_sym) {
 842         log_trace(redefine, class, record)
 843           ("redefined class %s attribute change error: Record generic_signature attribute changed",
 844            the_class->external_name());
 845         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 846       }
 847 
 848       // It's okay if a record component's annotations were changed.
 849     }
 850 
 851   } else if (the_record_exists ^ scr_record_exists) {
 852     const char* action_str = (the_record_exists) ? "removed" : "added";
 853     log_trace(redefine, class, record)
 854       ("redefined class %s attribute change error: Record attribute %s",
 855        the_class->external_name(), action_str);
 856     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 857   }
 858 
 859   return JVMTI_ERROR_NONE;
 860 }

 881 jvmtiError VM_RedefineClasses::compare_and_normalize_class_versions(
 882              InstanceKlass* the_class,
 883              InstanceKlass* scratch_class) {
 884   int i;
 885 
 886   // Check superclasses, or rather their names, since superclasses themselves can be
 887   // requested to replace.
 888   // Check for NULL superclass first since this might be java.lang.Object
 889   if (the_class->super() != scratch_class->super() &&
 890       (the_class->super() == NULL || scratch_class->super() == NULL ||
 891        the_class->super()->name() !=
 892        scratch_class->super()->name())) {
 893     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 894   }
 895 
 896   // Check if the number, names and order of directly implemented interfaces are the same.
 897   // I think in principle we should just check if the sets of names of directly implemented
 898   // interfaces are the same, i.e. the order of declaration (which, however, if changed in the
 899   // .java file, also changes in .class file) should not matter. However, comparing sets is
 900   // technically a bit more difficult, and, more importantly, I am not sure at present that the
 901   // order of interfaces does not matter on the implementation level, i.e. that the VM does not
 902   // rely on it somewhere.
 903   Array<InstanceKlass*>* k_interfaces = the_class->local_interfaces();
 904   Array<InstanceKlass*>* k_new_interfaces = scratch_class->local_interfaces();
 905   int n_intfs = k_interfaces->length();
 906   if (n_intfs != k_new_interfaces->length()) {
 907     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 908   }
 909   for (i = 0; i < n_intfs; i++) {
 910     if (k_interfaces->at(i)->name() !=
 911         k_new_interfaces->at(i)->name()) {
 912       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 913     }
 914   }
 915 
 916   // Check whether class is in the error init state.
 917   if (the_class->is_in_error_state()) {
 918     // TBD #5057930: special error code is needed in 1.6
 919     return JVMTI_ERROR_INVALID_CLASS;
 920   }
 921 
 922   // Check whether the nest-related attributes have been changed.
 923   jvmtiError err = check_nest_attributes(the_class, scratch_class);
 924   if (err != JVMTI_ERROR_NONE) {
 925     return err;
 926   }
 927 
 928   // Check whether the Record attribute has been changed.
 929   err = check_record_attribute(the_class, scratch_class);
 930   if (err != JVMTI_ERROR_NONE) {
 931     return err;
 932   }
 933 
 934   // Check whether the PermittedSubclasses attribute has been changed.
 935   err = check_permitted_subclasses_attribute(the_class, scratch_class);
 936   if (err != JVMTI_ERROR_NONE) {
 937     return err;
 938   }
 939 
 940   // Check whether class modifiers are the same.
 941   jushort old_flags = (jushort) the_class->access_flags().get_flags();
 942   jushort new_flags = (jushort) scratch_class->access_flags().get_flags();
 943   if (old_flags != new_flags) {
 944     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_MODIFIERS_CHANGED;
 945   }
 946 
 947   // Check if the number, names, types and order of fields declared in these classes
 948   // are the same.
 949   JavaFieldStream old_fs(the_class);
 950   JavaFieldStream new_fs(scratch_class);
 951   for (; !old_fs.done() && !new_fs.done(); old_fs.next(), new_fs.next()) {
 952     // access
 953     old_flags = old_fs.access_flags().as_short();
 954     new_flags = new_fs.access_flags().as_short();
 955     if ((old_flags ^ new_flags) & JVM_RECOGNIZED_FIELD_MODIFIERS) {
 956       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 957     }
 958     // offset
 959     if (old_fs.offset() != new_fs.offset()) {
 960       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 961     }
 962     // name and signature
 963     Symbol* name_sym1 = the_class->constants()->symbol_at(old_fs.name_index());
 964     Symbol* sig_sym1 = the_class->constants()->symbol_at(old_fs.signature_index());
 965     Symbol* name_sym2 = scratch_class->constants()->symbol_at(new_fs.name_index());
 966     Symbol* sig_sym2 = scratch_class->constants()->symbol_at(new_fs.signature_index());
 967     if (name_sym1 != name_sym2 || sig_sym1 != sig_sym2) {
 968       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 969     }
 970   }
 971 
 972   // If both streams aren't done then we have a differing number of
 973   // fields.
 974   if (!old_fs.done() || !new_fs.done()) {
 975     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 976   }
 977 
 978   // Do a parallel walk through the old and new methods. Detect
 979   // cases where they match (exist in both), have been added in
 980   // the new methods, or have been deleted (exist only in the
 981   // old methods).  The class file parser places methods in order
 982   // by method name, but does not order overloaded methods by
 983   // signature.  In order to determine what fate befell the methods,
 984   // this code places the overloaded new methods that have matching
 985   // old methods in the same order as the old methods and places
 986   // new overloaded methods at the end of overloaded methods of
 987   // that name. The code for this order normalization is adapted
 988   // from the algorithm used in InstanceKlass::find_method().
 989   // Since we are swapping out of order entries as we find them,
 990   // we only have to search forward through the overloaded methods.
 991   // Methods which are added and have the same name as an existing
 992   // method (but different signature) will be put at the end of
 993   // the methods with that name, and the name mismatch code will
 994   // handle them.
 995   Array<Method*>* k_old_methods(the_class->methods());
 996   Array<Method*>* k_new_methods(scratch_class->methods());
 997   int n_old_methods = k_old_methods->length();
 998   int n_new_methods = k_new_methods->length();
 999   Thread* thread = Thread::current();
1000 
1001   int ni = 0;
1002   int oi = 0;
1003   while (true) {
1004     Method* k_old_method;
1005     Method* k_new_method;
1006     enum { matched, added, deleted, undetermined } method_was = undetermined;
1007 
1008     if (oi >= n_old_methods) {
1009       if (ni >= n_new_methods) {
1010         break; // we've looked at everything, done
1011       }
1012       // New method at the end
1013       k_new_method = k_new_methods->at(ni);
1014       method_was = added;
1015     } else if (ni >= n_new_methods) {
1016       // Old method, at the end, is deleted
1017       k_old_method = k_old_methods->at(oi);
1018       method_was = deleted;
1019     } else {
1020       // There are more methods in both the old and new lists
1021       k_old_method = k_old_methods->at(oi);
1022       k_new_method = k_new_methods->at(ni);
1023       if (k_old_method->name() != k_new_method->name()) {
1024         // Methods are sorted by method name, so a mismatch means added
1025         // or deleted
1026         if (k_old_method->name()->fast_compare(k_new_method->name()) > 0) {
1027           method_was = added;
1028         } else {
1029           method_was = deleted;
1030         }
1031       } else if (k_old_method->signature() == k_new_method->signature()) {
1032         // Both the name and signature match
1033         method_was = matched;
1034       } else {
1035         // The name matches, but the signature doesn't, which means we have to
1036         // search forward through the new overloaded methods.
1037         int nj;  // outside the loop for post-loop check
1038         for (nj = ni + 1; nj < n_new_methods; nj++) {
1039           Method* m = k_new_methods->at(nj);
1040           if (k_old_method->name() != m->name()) {
1041             // reached another method name so no more overloaded methods
1042             method_was = deleted;
1043             break;
1044           }
1045           if (k_old_method->signature() == m->signature()) {
1046             // found a match so swap the methods
1047             k_new_methods->at_put(ni, m);
1048             k_new_methods->at_put(nj, k_new_method);
1049             k_new_method = m;
1050             method_was = matched;
1051             break;
1052           }
1053         }
1054 
1055         if (nj >= n_new_methods) {
1056           // reached the end without a match; so method was deleted
1057           method_was = deleted;
1058         }
1059       }
1060     }
1061 
1062     switch (method_was) {
1063     case matched:
1064       // methods match, be sure modifiers do too
1065       old_flags = (jushort) k_old_method->access_flags().get_flags();
1066       new_flags = (jushort) k_new_method->access_flags().get_flags();
1067       if ((old_flags ^ new_flags) & ~(JVM_ACC_NATIVE)) {
1068         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_MODIFIERS_CHANGED;
1069       }
1070       {
1071         u2 new_num = k_new_method->method_idnum();
1072         u2 old_num = k_old_method->method_idnum();
1073         if (new_num != old_num) {
1074           Method* idnum_owner = scratch_class->method_with_idnum(old_num);
1075           if (idnum_owner != NULL) {
1076             // There is already a method assigned this idnum -- switch them
1077             // Take current and original idnum from the new_method
1078             idnum_owner->set_method_idnum(new_num);
1079             idnum_owner->set_orig_method_idnum(k_new_method->orig_method_idnum());
1080           }
1081           // Take current and original idnum from the old_method
1082           k_new_method->set_method_idnum(old_num);
1083           k_new_method->set_orig_method_idnum(k_old_method->orig_method_idnum());
1084           if (thread->has_pending_exception()) {
1085             return JVMTI_ERROR_OUT_OF_MEMORY;
1086           }
1087         }
1088       }
1089       log_trace(redefine, class, normalize)
1090         ("Method matched: new: %s [%d] == old: %s [%d]",
1091          k_new_method->name_and_sig_as_C_string(), ni, k_old_method->name_and_sig_as_C_string(), oi);
1092       // advance to next pair of methods
1093       ++oi;
1094       ++ni;
1095       break;
1096     case added:
1097       // method added, see if it is OK
1098       if (!can_add_or_delete(k_new_method)) {
1099         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_ADDED;
1100       }
1101       {
1102         u2 num = the_class->next_method_idnum();
1103         if (num == ConstMethod::UNSET_IDNUM) {
1104           // cannot add any more methods
1105           return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_ADDED;
1106         }
1107         u2 new_num = k_new_method->method_idnum();
1108         Method* idnum_owner = scratch_class->method_with_idnum(num);
1109         if (idnum_owner != NULL) {
1110           // There is already a method assigned this idnum -- switch them
1111           // Take current and original idnum from the new_method
1112           idnum_owner->set_method_idnum(new_num);
1113           idnum_owner->set_orig_method_idnum(k_new_method->orig_method_idnum());
1114         }
1115         k_new_method->set_method_idnum(num);
1116         k_new_method->set_orig_method_idnum(num);
1117         if (thread->has_pending_exception()) {
1118           return JVMTI_ERROR_OUT_OF_MEMORY;
1119         }
1120       }
1121       log_trace(redefine, class, normalize)
1122         ("Method added: new: %s [%d]", k_new_method->name_and_sig_as_C_string(), ni);
1123       ++ni; // advance to next new method
1124       break;
1125     case deleted:
1126       // method deleted, see if it is OK
1127       if (!can_add_or_delete(k_old_method)) {
1128         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_DELETED;
1129       }
1130       log_trace(redefine, class, normalize)
1131         ("Method deleted: old: %s [%d]", k_old_method->name_and_sig_as_C_string(), oi);
1132       ++oi; // advance to next old method
1133       break;
1134     default:
1135       ShouldNotReachHere();
1136     }
1137   }
1138 
1139   return JVMTI_ERROR_NONE;
1140 }

1463 // Merge old_cp and scratch_cp and return the results of the merge via
1464 // merge_cp_p. The number of entries in *merge_cp_p is returned via
1465 // merge_cp_length_p. The entries in old_cp occupy the same locations
1466 // in *merge_cp_p. Also creates a map of indices from entries in
1467 // scratch_cp to the corresponding entry in *merge_cp_p. Index map
1468 // entries are only created for entries in scratch_cp that occupy a
1469 // different location in *merged_cp_p.
1470 bool VM_RedefineClasses::merge_constant_pools(const constantPoolHandle& old_cp,
1471        const constantPoolHandle& scratch_cp, constantPoolHandle *merge_cp_p,
1472        int *merge_cp_length_p, TRAPS) {
1473 
1474   if (merge_cp_p == NULL) {
1475     assert(false, "caller must provide scratch constantPool");
1476     return false; // robustness
1477   }
1478   if (merge_cp_length_p == NULL) {
1479     assert(false, "caller must provide scratch CP length");
1480     return false; // robustness
1481   }
1482   // Worst case we need old_cp->length() + scratch_cp()->length(),
1483   // but the caller might be smart so make sure we have at least
1484   // the minimum.
1485   if ((*merge_cp_p)->length() < old_cp->length()) {
1486     assert(false, "merge area too small");
1487     return false; // robustness
1488   }
1489 
1490   log_info(redefine, class, constantpool)("old_cp_len=%d, scratch_cp_len=%d", old_cp->length(), scratch_cp->length());
1491 
1492   {
1493     // Pass 0:
1494     // The old_cp is copied to *merge_cp_p; this means that any code
1495     // using old_cp does not have to change. This work looks like a
1496     // perfect fit for ConstantPool*::copy_cp_to(), but we need to
1497     // handle one special case:
1498     // - revert JVM_CONSTANT_Class to JVM_CONSTANT_UnresolvedClass
1499     // This will make verification happy.
1500 
1501     int old_i;  // index into old_cp
1502 
1503     // index zero (0) is not used in constantPools
1504     for (old_i = 1; old_i < old_cp->length(); old_i++) {
1505       // leave debugging crumb
1506       jbyte old_tag = old_cp->tag_at(old_i).value();
1507       switch (old_tag) {
1508       case JVM_CONSTANT_Class:
1509       case JVM_CONSTANT_UnresolvedClass:
1510         // revert the copy to JVM_CONSTANT_UnresolvedClass
1511         // May be resolving while calling this so do the same for
1512         // JVM_CONSTANT_UnresolvedClass (klass_name_at() deals with transition)
1513         (*merge_cp_p)->temp_unresolved_klass_at_put(old_i,
1514           old_cp->klass_name_index_at(old_i));
1515         break;
1516 
1517       case JVM_CONSTANT_Double:
1518       case JVM_CONSTANT_Long:
1519         // just copy the entry to *merge_cp_p, but double and long take
1520         // two constant pool entries
1521         ConstantPool::copy_entry_to(old_cp, old_i, *merge_cp_p, old_i, CHECK_false);
1522         old_i++;
1523         break;
1524 
1525       default:
1526         // just copy the entry to *merge_cp_p
1527         ConstantPool::copy_entry_to(old_cp, old_i, *merge_cp_p, old_i, CHECK_false);
1528         break;
1529       }
1530     } // end for each old_cp entry
1531 
1532     ConstantPool::copy_operands(old_cp, *merge_cp_p, CHECK_false);
1533     (*merge_cp_p)->extend_operands(scratch_cp, CHECK_false);
1534 
1535     // We don't need to sanity check that *merge_cp_length_p is within
1536     // *merge_cp_p bounds since we have the minimum on-entry check above.
1537     (*merge_cp_length_p) = old_i;
1538   }
1539 
1540   // merge_cp_len should be the same as old_cp->length() at this point
1541   // so this trace message is really a "warm-and-breathing" message.
1542   log_debug(redefine, class, constantpool)("after pass 0: merge_cp_len=%d", *merge_cp_length_p);
1543 
1544   int scratch_i;  // index into scratch_cp
1545   {
1546     // Pass 1a:
1547     // Compare scratch_cp entries to the old_cp entries that we have
1548     // already copied to *merge_cp_p. In this pass, we are eliminating
1549     // exact duplicates (matching entry at same index) so we only
1550     // compare entries in the common indice range.
1551     int increment = 1;
1552     int pass1a_length = MIN2(old_cp->length(), scratch_cp->length());
1553     for (scratch_i = 1; scratch_i < pass1a_length; scratch_i += increment) {
1554       switch (scratch_cp->tag_at(scratch_i).value()) {
1555       case JVM_CONSTANT_Double:
1556       case JVM_CONSTANT_Long:
1557         // double and long take two constant pool entries
1558         increment = 2;
1559         break;
1560 
1561       default:
1562         increment = 1;
1563         break;
1564       }
1565 
1566       bool match = scratch_cp->compare_entry_to(scratch_i, *merge_cp_p,
1567         scratch_i, CHECK_false);
1568       if (match) {
1569         // found a match at the same index so nothing more to do
1570         continue;
1571       } else if (is_unresolved_class_mismatch(scratch_cp, scratch_i,
1572                                               *merge_cp_p, scratch_i)) {
1573         // The mismatch in compare_entry_to() above is because of a
1574         // resolved versus unresolved class entry at the same index
1575         // with the same string value. Since Pass 0 reverted any
1576         // class entries to unresolved class entries in *merge_cp_p,
1577         // we go with the unresolved class entry.
1578         continue;
1579       }
1580 
1581       int found_i = scratch_cp->find_matching_entry(scratch_i, *merge_cp_p,
1582         CHECK_false);
1583       if (found_i != 0) {
1584         guarantee(found_i != scratch_i,
1585           "compare_entry_to() and find_matching_entry() do not agree");
1586 
1587         // Found a matching entry somewhere else in *merge_cp_p so
1588         // just need a mapping entry.
1589         map_index(scratch_cp, scratch_i, found_i);
1590         continue;
1591       }
1592 
1593       // The find_matching_entry() call above could fail to find a match
1594       // due to a resolved versus unresolved class or string entry situation
1595       // like we solved above with the is_unresolved_*_mismatch() calls.
1596       // However, we would have to call is_unresolved_*_mismatch() over
1597       // all of *merge_cp_p (potentially) and that doesn't seem to be
1598       // worth the time.
1599 
1600       // No match found so we have to append this entry and any unique
1601       // referenced entries to *merge_cp_p.
1602       append_entry(scratch_cp, scratch_i, merge_cp_p, merge_cp_length_p,
1603         CHECK_false);
1604     }
1605   }
1606 
1607   log_debug(redefine, class, constantpool)
1608     ("after pass 1a: merge_cp_len=%d, scratch_i=%d, index_map_len=%d",
1609      *merge_cp_length_p, scratch_i, _index_map_count);
1610 
1611   if (scratch_i < scratch_cp->length()) {
1612     // Pass 1b:
1613     // old_cp is smaller than scratch_cp so there are entries in
1614     // scratch_cp that we have not yet processed. We take care of
1615     // those now.
1616     int increment = 1;
1617     for (; scratch_i < scratch_cp->length(); scratch_i += increment) {
1618       switch (scratch_cp->tag_at(scratch_i).value()) {
1619       case JVM_CONSTANT_Double:
1620       case JVM_CONSTANT_Long:
1621         // double and long take two constant pool entries
1622         increment = 2;
1623         break;
1624 
1625       default:
1626         increment = 1;
1627         break;
1628       }
1629 
1630       int found_i =
1631         scratch_cp->find_matching_entry(scratch_i, *merge_cp_p, CHECK_false);
1632       if (found_i != 0) {
1633         // Found a matching entry somewhere else in *merge_cp_p so
1634         // just need a mapping entry.
1635         map_index(scratch_cp, scratch_i, found_i);
1636         continue;
1637       }
1638 
1639       // No match found so we have to append this entry and any unique
1640       // referenced entries to *merge_cp_p.
1641       append_entry(scratch_cp, scratch_i, merge_cp_p, merge_cp_length_p,
1642         CHECK_false);
1643     }
1644 
1645     log_debug(redefine, class, constantpool)
1646       ("after pass 1b: merge_cp_len=%d, scratch_i=%d, index_map_len=%d",
1647        *merge_cp_length_p, scratch_i, _index_map_count);
1648   }
1649   finalize_operands_merge(*merge_cp_p, THREAD);
1650 
1651   return true;
1652 } // end merge_constant_pools()

1672 // Merge constant pools between the_class and scratch_class and
1673 // potentially rewrite bytecodes in scratch_class to use the merged
1674 // constant pool.
1675 jvmtiError VM_RedefineClasses::merge_cp_and_rewrite(
1676              InstanceKlass* the_class, InstanceKlass* scratch_class,
1677              TRAPS) {
1678   // worst case merged constant pool length is old and new combined
1679   int merge_cp_length = the_class->constants()->length()
1680         + scratch_class->constants()->length();
1681 
1682   // Constant pools are not easily reused so we allocate a new one
1683   // each time.
1684   // merge_cp is created unsafe for concurrent GC processing.  It
1685   // should be marked safe before discarding it. Even though
1686   // garbage,  if it crosses a card boundary, it may be scanned
1687   // in order to find the start of the first complete object on the card.
1688   ClassLoaderData* loader_data = the_class->class_loader_data();
1689   ConstantPool* merge_cp_oop =
1690     ConstantPool::allocate(loader_data,
1691                            merge_cp_length,
1692                            CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1693   MergeCPCleaner cp_cleaner(loader_data, merge_cp_oop);
1694 
1695   HandleMark hm(THREAD);  // make sure handles are cleared before
1696                           // MergeCPCleaner clears out merge_cp_oop
1697   constantPoolHandle merge_cp(THREAD, merge_cp_oop);
1698 
1699   // Get constants() from the old class because it could have been rewritten
1700   // while we were at a safepoint allocating a new constant pool.
1701   constantPoolHandle old_cp(THREAD, the_class->constants());
1702   constantPoolHandle scratch_cp(THREAD, scratch_class->constants());
1703 
1704   // If the length changed, the class was redefined out from under us. Return
1705   // an error.
1706   if (merge_cp_length != the_class->constants()->length()
1707          + scratch_class->constants()->length()) {
1708     return JVMTI_ERROR_INTERNAL;
1709   }
1710 
1711   // Update the version number of the constant pools (may keep scratch_cp)
1712   merge_cp->increment_and_save_version(old_cp->version());
1713   scratch_cp->increment_and_save_version(old_cp->version());
1714 
1715   ResourceMark rm(THREAD);
1716   _index_map_count = 0;
1717   _index_map_p = new intArray(scratch_cp->length(), scratch_cp->length(), -1);
1718 
1719   _operands_cur_length = ConstantPool::operand_array_length(old_cp->operands());
1720   _operands_index_map_count = 0;
1721   int operands_index_map_len = ConstantPool::operand_array_length(scratch_cp->operands());
1722   _operands_index_map_p = new intArray(operands_index_map_len, operands_index_map_len, -1);
1723 
1724   // reference to the cp holder is needed for copy_operands()
1725   merge_cp->set_pool_holder(scratch_class);
1726   bool result = merge_constant_pools(old_cp, scratch_cp, &merge_cp,
1727                   &merge_cp_length, THREAD);
1728   merge_cp->set_pool_holder(NULL);
1729 
1730   if (!result) {
1731     // The merge can fail due to memory allocation failure or due
1732     // to robustness checks.
1733     return JVMTI_ERROR_INTERNAL;
1734   }
1735 
1736   // Save fields from the old_cp.
1737   merge_cp->copy_fields(old_cp());
1738   scratch_cp->copy_fields(old_cp());
1739 
1740   log_info(redefine, class, constantpool)("merge_cp_len=%d, index_map_len=%d", merge_cp_length, _index_map_count);
1741 
1742   if (_index_map_count == 0) {
1743     // there is nothing to map between the new and merged constant pools
1744 
1745     if (old_cp->length() == scratch_cp->length()) {
1746       // The old and new constant pools are the same length and the
1747       // index map is empty. This means that the three constant pools
1748       // are equivalent (but not the same). Unfortunately, the new
1749       // constant pool has not gone through link resolution nor have
1750       // the new class bytecodes gone through constant pool cache
1751       // rewriting so we can't use the old constant pool with the new
1752       // class.
1753 
1754       // toss the merged constant pool at return
1755     } else if (old_cp->length() < scratch_cp->length()) {
1756       // The old constant pool has fewer entries than the new constant
1757       // pool and the index map is empty. This means the new constant
1758       // pool is a superset of the old constant pool. However, the old
1759       // class bytecodes have already gone through constant pool cache
1760       // rewriting so we can't use the new constant pool with the old
1761       // class.
1762 
1763       // toss the merged constant pool at return
1764     } else {
1765       // The old constant pool has more entries than the new constant
1766       // pool and the index map is empty. This means that both the old
1767       // and merged constant pools are supersets of the new constant
1768       // pool.
1769 
1770       // Replace the new constant pool with a shrunken copy of the
1771       // merged constant pool
1772       set_new_constant_pool(loader_data, scratch_class, merge_cp, merge_cp_length,
1773                             CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1774       // The new constant pool replaces scratch_cp so have cleaner clean it up.
1775       // It can't be cleaned up while there are handles to it.
1776       cp_cleaner.add_scratch_cp(scratch_cp());
1777     }
1778   } else {
1779     if (log_is_enabled(Trace, redefine, class, constantpool)) {
1780       // don't want to loop unless we are tracing
1781       int count = 0;
1782       for (int i = 1; i < _index_map_p->length(); i++) {
1783         int value = _index_map_p->at(i);
1784 
1785         if (value != -1) {
1786           log_trace(redefine, class, constantpool)("index_map[%d]: old=%d new=%d", count, i, value);
1787           count++;
1788         }
1789       }
1790     }
1791 
1792     // We have entries mapped between the new and merged constant pools
1793     // so we have to rewrite some constant pool references.
1794     if (!rewrite_cp_refs(scratch_class, THREAD)) {
1795       return JVMTI_ERROR_INTERNAL;
1796     }
1797 
1798     // Replace the new constant pool with a shrunken copy of the
1799     // merged constant pool so now the rewritten bytecodes have
1800     // valid references; the previous new constant pool will get
1801     // GCed.
1802     set_new_constant_pool(loader_data, scratch_class, merge_cp, merge_cp_length,
1803                           CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1804     // The new constant pool replaces scratch_cp so have cleaner clean it up.
1805     // It can't be cleaned up while there are handles to it.
1806     cp_cleaner.add_scratch_cp(scratch_cp());
1807   }
1808 
1809   return JVMTI_ERROR_NONE;
1810 } // end merge_cp_and_rewrite()

1818   if (!rewrite_cp_refs_in_nest_attributes(scratch_class)) {
1819     // propagate failure back to caller
1820     return false;
1821   }
1822 
1823   // rewrite constant pool references in the Record attribute:
1824   if (!rewrite_cp_refs_in_record_attribute(scratch_class, THREAD)) {

1830   if (!rewrite_cp_refs_in_permitted_subclasses_attribute(scratch_class)) {

1836   if (!rewrite_cp_refs_in_methods(scratch_class, THREAD)) {

1844     return false;
1845   }
1846

1848   if (!rewrite_cp_refs_in_fields_annotations(scratch_class, THREAD)) {
1849     // propagate failure back to caller
1850     return false;
1851   }
1852 
1853   // rewrite constant pool references in the methods_annotations:
1854   if (!rewrite_cp_refs_in_methods_annotations(scratch_class, THREAD)) {
1855     // propagate failure back to caller
1856     return false;
1857   }
1858 
1859   // rewrite constant pool references in the methods_parameter_annotations:
1860   if (!rewrite_cp_refs_in_methods_parameter_annotations(scratch_class,

1868          THREAD)) {
1869     // propagate failure back to caller
1870     return false;
1871   }
1872 
1873   // rewrite constant pool references in the class_type_annotations:
1874   if (!rewrite_cp_refs_in_class_type_annotations(scratch_class, THREAD)) {
1875     // propagate failure back to caller
1876     return false;

1879   // rewrite constant pool references in the fields_type_annotations:
1880   if (!rewrite_cp_refs_in_fields_type_annotations(scratch_class, THREAD)) {
1881     // propagate failure back to caller
1882     return false;
1883   }
1884

1886   if (!rewrite_cp_refs_in_methods_type_annotations(scratch_class, THREAD)) {
1887     // propagate failure back to caller
1888     return false;
1889   }
1890 
1891   // There can be type annotations in the Code part of a method_info attribute.
1892   // These annotations are not accessible, even by reflection.
1893   // Currently they are not even parsed by the ClassFileParser.
1894   // If runtime access is added they will also need to be rewritten.
1895 
1896   // rewrite source file name index:
1897   u2 source_file_name_idx = scratch_class->source_file_name_index();
1898   if (source_file_name_idx != 0) {
1899     u2 new_source_file_name_idx = find_new_index(source_file_name_idx);
1900     if (new_source_file_name_idx != 0) {
1901       scratch_class->set_source_file_name_index(new_source_file_name_idx);
1902     }
1903   }
1904

1906   u2 generic_signature_index = scratch_class->generic_signature_index();
1907   if (generic_signature_index != 0) {
1908     u2 new_generic_signature_index = find_new_index(generic_signature_index);
1909     if (new_generic_signature_index != 0) {
1910       scratch_class->set_generic_signature_index(new_generic_signature_index);
1911     }
1912   }
1913

1986 // Rewrite constant pool references in the methods.
1987 bool VM_RedefineClasses::rewrite_cp_refs_in_methods(
1988        InstanceKlass* scratch_class, TRAPS) {
1989 
1990   Array<Method*>* methods = scratch_class->methods();
1991 
1992   if (methods == NULL || methods->length() == 0) {
1993     // no methods so nothing to do
1994     return true;
1995   }
1996 
1997   // rewrite constant pool references in the methods:
1998   for (int i = methods->length() - 1; i >= 0; i--) {
1999     methodHandle method(THREAD, methods->at(i));
2000     methodHandle new_method;
2001     rewrite_cp_refs_in_method(method, &new_method, THREAD);
2002     if (!new_method.is_null()) {
2003       // the method has been replaced so save the new method version
2004       // even in the case of an exception.  original method is on the
2005       // deallocation list.
2006       methods->at_put(i, new_method());
2007     }
2008     if (HAS_PENDING_EXCEPTION) {
2009       Symbol* ex_name = PENDING_EXCEPTION->klass()->name();
2010       log_info(redefine, class, load, exceptions)("rewrite_cp_refs_in_method exception: '%s'", ex_name->as_C_string());
2011       // Need to clear pending exception here as the super caller sets
2012       // the JVMTI_ERROR_INTERNAL if the returned value is false.
2013       CLEAR_PENDING_EXCEPTION;
2014       return false;
2015     }
2016   }
2017 
2018   return true;
2019 }

2022 // Rewrite constant pool references in the specific method. This code
2023 // was adapted from Rewriter::rewrite_method().
2024 void VM_RedefineClasses::rewrite_cp_refs_in_method(methodHandle method,
2025        methodHandle *new_method_p, TRAPS) {
2026 
2027   *new_method_p = methodHandle();  // default is no new method
2028 
2029   // We cache a pointer to the bytecodes here in code_base. If GC
2030   // moves the Method*, then the bytecodes will also move which
2031   // will likely cause a crash. We create a NoSafepointVerifier
2032   // object to detect whether we pass a possible safepoint in this
2033   // code block.
2034   NoSafepointVerifier nsv;
2035 
2036   // Bytecodes and their length
2037   address code_base = method->code_base();
2038   int code_length = method->code_size();
2039 
2040   int bc_length;
2041   for (int bci = 0; bci < code_length; bci += bc_length) {
2042     address bcp = code_base + bci;
2043     Bytecodes::Code c = (Bytecodes::Code)(*bcp);
2044 
2045     bc_length = Bytecodes::length_for(c);
2046     if (bc_length == 0) {
2047       // More complicated bytecodes report a length of zero so
2048       // we have to try again a slightly different way.
2049       bc_length = Bytecodes::length_at(method(), bcp);
2050     }
2051 
2052     assert(bc_length != 0, "impossible bytecode length");
2053 
2054     switch (c) {
2055       case Bytecodes::_ldc:
2056       {
2057         int cp_index = *(bcp + 1);
2058         int new_index = find_new_index(cp_index);
2059 
2060         if (StressLdcRewrite && new_index == 0) {
2061           // If we are stressing ldc -> ldc_w rewriting, then we
2062           // always need a new_index value.
2063           new_index = cp_index;
2064         }
2065         if (new_index != 0) {
2066           // the original index is mapped so we have more work to do
2067           if (!StressLdcRewrite && new_index <= max_jubyte) {
2068             // The new value can still use ldc instead of ldc_w
2069             // unless we are trying to stress ldc -> ldc_w rewriting
2070             log_trace(redefine, class, constantpool)
2071               ("%s@" INTPTR_FORMAT " old=%d, new=%d", Bytecodes::name(c), p2i(bcp), cp_index, new_index);
2072             *(bcp + 1) = new_index;
2073           } else {
2074             log_trace(redefine, class, constantpool)
2075               ("%s->ldc_w@" INTPTR_FORMAT " old=%d, new=%d", Bytecodes::name(c), p2i(bcp), cp_index, new_index);
2076             // the new value needs ldc_w instead of ldc
2077             u_char inst_buffer[4]; // max instruction size is 4 bytes
2078             bcp = (address)inst_buffer;
2079             // construct new instruction sequence
2080             *bcp = Bytecodes::_ldc_w;
2081             bcp++;
2082             // Rewriter::rewrite_method() does not rewrite ldc -> ldc_w.
2083             // See comment below for difference between put_Java_u2()
2084             // and put_native_u2().
2085             Bytes::put_Java_u2(bcp, new_index);
2086 
2087             Relocator rc(method, NULL /* no RelocatorListener needed */);
2088             methodHandle m;
2089             {

3150 // Rewrite constant pool references in the method's stackmap table.
3151 // These "structures" are adapted from the StackMapTable_attribute that
3152 // is described in section 4.8.4 of the 6.0 version of the VM spec
3153 // (dated 2005.10.26):
3154 // file:///net/quincunx.sfbay/export/gbracha/ClassFile-Java6.pdf
3155 //
3156 // stack_map {
3157 //   u2 number_of_entries;
3158 //   stack_map_frame entries[number_of_entries];
3159 // }
3160 //
3161 void VM_RedefineClasses::rewrite_cp_refs_in_stack_map_table(
3162        const methodHandle& method, TRAPS) {
3163 
3164   if (!method->has_stackmap_table()) {
3165     return;
3166   }
3167 
3168   AnnotationArray* stackmap_data = method->stackmap_data();
3169   address stackmap_p = (address)stackmap_data->adr_at(0);
3170   address stackmap_end = stackmap_p + stackmap_data->length();
3171 
3172   assert(stackmap_p + 2 <= stackmap_end, "no room for number_of_entries");
3173   u2 number_of_entries = Bytes::get_Java_u2(stackmap_p);
3174   stackmap_p += 2;
3175 
3176   log_debug(redefine, class, stackmap)("number_of_entries=%u", number_of_entries);
3177 
3178   // walk through each stack_map_frame
3179   u2 calc_number_of_entries = 0;
3180   for (; calc_number_of_entries < number_of_entries; calc_number_of_entries++) {
3181     // The stack_map_frame structure is a u1 frame_type followed by
3182     // 0 or more bytes of data:
3183     //
3184     // union stack_map_frame {
3185     //   same_frame;
3186     //   same_locals_1_stack_item_frame;
3187     //   same_locals_1_stack_item_frame_extended;
3188     //   chop_frame;
3189     //   same_frame_extended;
3190     //   append_frame;
3191     //   full_frame;
3192     // }
3193 
3194     assert(stackmap_p + 1 <= stackmap_end, "no room for frame_type");
3195     u1 frame_type = *stackmap_p;
3196     stackmap_p++;
3197 
3198     // same_frame {
3199     //   u1 frame_type = SAME; /* 0-63 */
3200     // }
3201     if (frame_type <= 63) {
3202       // nothing more to do for same_frame
3203     }
3204 
3205     // same_locals_1_stack_item_frame {
3206     //   u1 frame_type = SAME_LOCALS_1_STACK_ITEM; /* 64-127 */
3207     //   verification_type_info stack[1];
3208     // }
3209     else if (frame_type >= 64 && frame_type <= 127) {
3210       rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3211         calc_number_of_entries, frame_type, THREAD);
3212     }
3213 
3214     // reserved for future use
3215     else if (frame_type >= 128 && frame_type <= 246) {
3216       // nothing more to do for reserved frame_types
3217     }
3218 
3219     // same_locals_1_stack_item_frame_extended {
3220     //   u1 frame_type = SAME_LOCALS_1_STACK_ITEM_EXTENDED; /* 247 */
3221     //   u2 offset_delta;
3222     //   verification_type_info stack[1];
3223     // }
3224     else if (frame_type == 247) {
3225       stackmap_p += 2;
3226       rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3227         calc_number_of_entries, frame_type, THREAD);
3228     }
3229 
3230     // chop_frame {
3231     //   u1 frame_type = CHOP; /* 248-250 */
3232     //   u2 offset_delta;
3233     // }
3234     else if (frame_type >= 248 && frame_type <= 250) {
3235       stackmap_p += 2;
3236     }
3237 
3238     // same_frame_extended {
3239     //   u1 frame_type = SAME_FRAME_EXTENDED; /* 251*/
3240     //   u2 offset_delta;
3241     // }
3242     else if (frame_type == 251) {
3243       stackmap_p += 2;
3244     }
3245 
3246     // append_frame {
3247     //   u1 frame_type = APPEND; /* 252-254 */
3248     //   u2 offset_delta;
3249     //   verification_type_info locals[frame_type - 251];
3250     // }
3251     else if (frame_type >= 252 && frame_type <= 254) {
3252       assert(stackmap_p + 2 <= stackmap_end,
3253         "no room for offset_delta");
3254       stackmap_p += 2;
3255       u1 len = frame_type - 251;
3256       for (u1 i = 0; i < len; i++) {
3257         rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3258           calc_number_of_entries, frame_type, THREAD);
3259       }
3260     }
3261 
3262     // full_frame {
3263     //   u1 frame_type = FULL_FRAME; /* 255 */
3264     //   u2 offset_delta;
3265     //   u2 number_of_locals;
3266     //   verification_type_info locals[number_of_locals];
3267     //   u2 number_of_stack_items;
3268     //   verification_type_info stack[number_of_stack_items];
3269     // }
3270     else if (frame_type == 255) {
3271       assert(stackmap_p + 2 + 2 <= stackmap_end,
3272         "no room for smallest full_frame");
3273       stackmap_p += 2;
3274 
3275       u2 number_of_locals = Bytes::get_Java_u2(stackmap_p);
3276       stackmap_p += 2;
3277 
3278       for (u2 locals_i = 0; locals_i < number_of_locals; locals_i++) {

3299 // portion of the method's stackmap table. These "structures" are
3300 // adapted from the StackMapTable_attribute that is described in
3301 // section 4.8.4 of the 6.0 version of the VM spec (dated 2005.10.26):

3304 // The verification_type_info structure is a u1 tag followed by 0 or
3305 // more bytes of data:
3306 //
3307 // union verification_type_info {
3308 //   Top_variable_info;
3309 //   Integer_variable_info;
3310 //   Float_variable_info;

3312 //   Double_variable_info;
3313 //   Null_variable_info;

3316 //   Uninitialized_variable_info;
3317 // }

3325   stackmap_p_ref++;

3328   // Top_variable_info {
3329   //   u1 tag = ITEM_Top; /* 0 */
3330   // }

3332   case 0:  // fall through
3333

3335   //   u1 tag = ITEM_Integer; /* 1 */
3336   // }
3337   case ITEM_Integer:  // fall through
3338 
3339   // Float_variable_info {
3340   //   u1 tag = ITEM_Float; /* 2 */
3341   // }
3342   case ITEM_Float:  // fall through
3343 
3344   // Double_variable_info {
3345   //   u1 tag = ITEM_Double; /* 3 */
3346   // }

3350   //   u1 tag = ITEM_Long; /* 4 */

3694 int VM_RedefineClasses::check_methods_and_mark_as_obsolete() {
3695   int emcp_method_count = 0;
3696   int obsolete_count = 0;
3697   int old_index = 0;
3698   for (int j = 0; j < _matching_methods_length; ++j, ++old_index) {
3699     Method* old_method = _matching_old_methods[j];
3700     Method* new_method = _matching_new_methods[j];
3701     Method* old_array_method;
3702 
3703     // Maintain an old_index into the _old_methods array by skipping
3704     // deleted methods
3705     while ((old_array_method = _old_methods->at(old_index)) != old_method) {
3706       ++old_index;
3707     }
3708 
3709     if (MethodComparator::methods_EMCP(old_method, new_method)) {
3710       // The EMCP definition from JSR-163 requires the bytecodes to be
3711       // the same with the exception of constant pool indices which may
3712       // differ. However, the constants referred to by those indices
3713       // must be the same.
3714       //
3715       // We use methods_EMCP() for comparison since constant pool
3716       // merging can remove duplicate constant pool entries that were
3717       // present in the old method and removed from the rewritten new
3718       // method. A faster binary comparison function would consider the
3719       // old and new methods to be different when they are actually
3720       // EMCP.
3721       //
3722       // The old and new methods are EMCP and you would think that we
3723       // could get rid of one of them here and now and save some space.
3724       // However, the concept of EMCP only considers the bytecodes and
3725       // the constant pool entries in the comparison. Other things,
3726       // e.g., the line number table (LNT) or the local variable table
3727       // (LVT) don't count in the comparison. So the new (and EMCP)
3728       // method can have a new LNT that we need so we can't just
3729       // overwrite the new method with the old method.
3730       //
3731       // When this routine is called, we have already attached the new
3732       // methods to the_class so the old methods are effectively
3733       // overwritten. However, if an old method is still executing,
3734       // then the old method cannot be collected until sometime after
3735       // the old method call has returned. So the overwriting of old
3736       // methods by new methods will save us space except for those
3737       // (hopefully few) old methods that are still executing.
3738       //
3739       // A method refers to a ConstMethod* and this presents another
3740       // possible avenue to space savings. The ConstMethod* in the
3741       // new method contains possibly new attributes (LNT, LVT, etc).
3742       // At first glance, it seems possible to save space by replacing
3743       // the ConstMethod* in the old method with the ConstMethod*
3744       // from the new method. The old and new methods would share the
3745       // same ConstMethod* and we would save the space occupied by
3746       // the old ConstMethod*. However, the ConstMethod* contains
3747       // a back reference to the containing method. Sharing the
3748       // ConstMethod* between two methods could lead to confusion in
3749       // the code that uses the back reference. This would lead to
3750       // brittle code that could be broken in non-obvious ways now or
3751       // in the future.
3752       //
3753       // Another possibility is to copy the ConstMethod* from the new
3754       // method to the old method and then overwrite the new method with
3755       // the old method. Since the ConstMethod* contains the bytecodes
3756       // for the method embedded in the oop, this option would change
3757       // the bytecodes out from under any threads executing the old
3758       // method and make the thread's bcp invalid. Since EMCP requires
3759       // that the bytecodes be the same modulo constant pool indices, it
3760       // is straight forward to compute the correct new bcp in the new
3761       // ConstMethod* from the old bcp in the old ConstMethod*. The
3762       // time consuming part would be searching all the frames in all
3763       // of the threads to find all of the calls to the old method.
3764       //
3765       // It looks like we will have to live with the limited savings
3766       // that we get from effectively overwriting the old methods
3767       // when the new methods are attached to the_class.
3768 
3769       // Count number of methods that are EMCP.  The method will be marked
3770       // old but not obsolete if it is EMCP.
3771       emcp_method_count++;
3772 
3773       // An EMCP method is _not_ obsolete. An obsolete method has a
3774       // different jmethodID than the current method. An EMCP method
3775       // has the same jmethodID as the current method. Having the
3776       // same jmethodID for all EMCP versions of a method allows for
3777       // a consistent view of the EMCP methods regardless of which
3778       // EMCP method you happen to have in hand. For example, a
3779       // breakpoint set in one EMCP method will work for all EMCP
3780       // versions of the method including the current one.
3781     } else {
3782       // mark obsolete methods as such
3783       old_method->set_is_obsolete();
3784       obsolete_count++;
3785 
3786       // obsolete methods need a unique idnum so they become new entries in
3787       // the jmethodID cache in InstanceKlass
3788       assert(old_method->method_idnum() == new_method->method_idnum(), "must match");

3831 // This internal class transfers the native function registration from old methods
3832 // to new methods.  It is designed to handle both the simple case of unchanged
3833 // native methods and the complex cases of native method prefixes being added and/or
3834 // removed.
3835 // It expects only to be used during the VM_RedefineClasses op (a safepoint).
3836 //
3837 // This class is used after the new methods have been installed in "the_class".
3838 //
3839 // So, for example, the following must be handled.  Where 'm' is a method and
3840 // a number followed by an underscore is a prefix.
3841 //
3842 //                                      Old Name    New Name
3843 // Simple transfer to new method        m       ->  m
3844 // Add prefix                           m       ->  1_m
3845 // Remove prefix                        1_m     ->  m
3846 // Simultaneous add of prefixes         m       ->  3_2_1_m
3847 // Simultaneous removal of prefixes     3_2_1_m ->  m
3848 // Simultaneous add and remove          1_m     ->  2_m
3849 // Same, caused by prefix removal only  3_2_1_m ->  3_2_m
3850 //
3851 class TransferNativeFunctionRegistration {
3852  private:
3853   InstanceKlass* the_class;
3854   int prefix_count;
3855   char** prefixes;
3856 
3857   // Recursively search the binary tree of possibly prefixed method names.
3858   // Iteration could be used if all agents were well behaved. Full tree walk is
3859   // more resilent to agents not cleaning up intermediate methods.
3860   // Branch at each depth in the binary tree is:
3861   //    (1) without the prefix.
3862   //    (2) with the prefix.
3863   // where 'prefix' is the prefix at that 'depth' (first prefix, second prefix,...)
3864   Method* search_prefix_name_space(int depth, char* name_str, size_t name_len,
3865                                      Symbol* signature) {
3866     TempNewSymbol name_symbol = SymbolTable::probe(name_str, (int)name_len);
3867     if (name_symbol != NULL) {
3868       Method* method = the_class->lookup_method(name_symbol, signature);
3869       if (method != NULL) {
3870         // Even if prefixed, intermediate methods must exist.
3871         if (method->is_native()) {
3872           // Wahoo, we found a (possibly prefixed) version of the method, return it.
3873           return method;
3874         }
3875         if (depth < prefix_count) {
3876           // Try applying further prefixes (other than this one).
3877           method = search_prefix_name_space(depth+1, name_str, name_len, signature);
3878           if (method != NULL) {
3879             return method; // found
3880           }
3881 
3882           // Try adding this prefix to the method name and see if it matches
3883           // another method name.
3884           char* prefix = prefixes[depth];
3885           size_t prefix_len = strlen(prefix);
3886           size_t trial_len = name_len + prefix_len;
3887           char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
3888           strcpy(trial_name_str, prefix);
3889           strcat(trial_name_str, name_str);
3890           method = search_prefix_name_space(depth+1, trial_name_str, trial_len,
3891                                             signature);
3892           if (method != NULL) {
3893             // If found along this branch, it was prefixed, mark as such
3894             method->set_is_prefixed_native();
3895             return method; // found
3896           }
3897         }
3898       }
3899     }
3900     return NULL;  // This whole branch bore nothing
3901   }
3902 
3903   // Return the method name with old prefixes stripped away.
3904   char* method_name_without_prefixes(Method* method) {
3905     Symbol* name = method->name();
3906     char* name_str = name->as_utf8();
3907 
3908     // Old prefixing may be defunct, strip prefixes, if any.
3909     for (int i = prefix_count-1; i >= 0; i--) {
3910       char* prefix = prefixes[i];
3911       size_t prefix_len = strlen(prefix);
3912       if (strncmp(prefix, name_str, prefix_len) == 0) {
3913         name_str += prefix_len;
3914       }
3915     }
3916     return name_str;
3917   }
3918 
3919   // Strip any prefixes off the old native method, then try to find a
3920   // (possibly prefixed) new native that matches it.
3921   Method* strip_and_search_for_new_native(Method* method) {
3922     ResourceMark rm;
3923     char* name_str = method_name_without_prefixes(method);
3924     return search_prefix_name_space(0, name_str, strlen(name_str),
3925                                     method->signature());
3926   }

3928  public:
3929 
3930   // Construct a native method transfer processor for this class.
3931   TransferNativeFunctionRegistration(InstanceKlass* _the_class) {
3932     assert(SafepointSynchronize::is_at_safepoint(), "sanity check");
3933 
3934     the_class = _the_class;
3935     prefixes = JvmtiExport::get_all_native_method_prefixes(&prefix_count);
3936   }

3938   // Attempt to transfer any of the old or deleted methods that are native
3939   void transfer_registrations(Method** old_methods, int methods_length) {
3940     for (int j = 0; j < methods_length; j++) {
3941       Method* old_method = old_methods[j];
3942 
3943       if (old_method->is_native() && old_method->has_native_function()) {
3944         Method* new_method = strip_and_search_for_new_native(old_method);
3945         if (new_method != NULL) {
3946           // Actually set the native function in the new method.
3947           // Redefine does not send events (except CFLH), certainly not this

3949           new_method->set_native_function(old_method->native_function(),
3950                               !Method::native_bind_event_is_interesting);
3951         }
3952       }
3953     }
3954   }

4016 void VM_RedefineClasses::compute_added_deleted_matching_methods() {
4017   Method* old_method;
4018   Method* new_method;
4019 
4020   _matching_old_methods = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4021   _matching_new_methods = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4022   _added_methods        = NEW_RESOURCE_ARRAY(Method*, _new_methods->length());
4023   _deleted_methods      = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4024 
4025   _matching_methods_length = 0;
4026   _deleted_methods_length  = 0;
4027   _added_methods_length    = 0;
4028 
4029   int nj = 0;
4030   int oj = 0;
4031   while (true) {
4032     if (oj >= _old_methods->length()) {
4033       if (nj >= _new_methods->length()) {
4034         break; // we've looked at everything, done
4035       }
4036       // New method at the end
4037       new_method = _new_methods->at(nj);
4038       _added_methods[_added_methods_length++] = new_method;
4039       ++nj;
4040     } else if (nj >= _new_methods->length()) {
4041       // Old method, at the end, is deleted
4042       old_method = _old_methods->at(oj);
4043       _deleted_methods[_deleted_methods_length++] = old_method;
4044       ++oj;
4045     } else {
4046       old_method = _old_methods->at(oj);
4047       new_method = _new_methods->at(nj);
4048       if (old_method->name() == new_method->name()) {
4049         if (old_method->signature() == new_method->signature()) {
4050           _matching_old_methods[_matching_methods_length  ] = old_method;
4051           _matching_new_methods[_matching_methods_length++] = new_method;
4052           ++nj;
4053           ++oj;
4054         } else {
4055           // added overloaded have already been moved to the end,
4056           // so this is a deleted overloaded method
4057           _deleted_methods[_deleted_methods_length++] = old_method;
4058           ++oj;
4059         }
4060       } else { // names don't match
4061         if (old_method->name()->fast_compare(new_method->name()) > 0) {
4062           // new method
4063           _added_methods[_added_methods_length++] = new_method;
4064           ++nj;
4065         } else {
4066           // deleted method
4067           _deleted_methods[_deleted_methods_length++] = old_method;
4068           ++oj;
4069         }
4070       }
4071     }
4072   }
4073   assert(_matching_methods_length + _deleted_methods_length == _old_methods->length(), "sanity");
4074   assert(_matching_methods_length + _added_methods_length == _new_methods->length(), "sanity");
4075 }

4322   }
4323 
4324   if (the_class->oop_map_cache() != NULL) {
4325     // Flush references to any obsolete methods from the oop map cache
4326     // so that obsolete methods are not pinned.
4327     the_class->oop_map_cache()->flush_obsolete_entries();
4328   }
4329 
4330   increment_class_counter((InstanceKlass *)the_class, THREAD);
4331 
4332   if (EventClassRedefinition::is_enabled()) {
4333     EventClassRedefinition event;
4334     event.set_classModificationCount(java_lang_Class::classRedefinedCount(the_class->java_mirror()));
4335     event.set_redefinedClass(the_class);
4336     event.set_redefinitionId(_id);
4337     event.commit();
4338   }
4339 
4340   {
4341     ResourceMark rm(THREAD);
4342     // increment the classRedefinedCount field in the_class and in any
4343     // direct and indirect subclasses of the_class
4344     log_info(redefine, class, load)
4345       ("redefined name=%s, count=%d (avail_mem=" UINT64_FORMAT "K)",
4346        the_class->external_name(), java_lang_Class::classRedefinedCount(the_class->java_mirror()), os::available_memory() >> 10);
4347     Events::log_redefinition(THREAD, "redefined class name=%s, count=%d",
4348                              the_class->external_name(),
4349                              java_lang_Class::classRedefinedCount(the_class->java_mirror()));
4350 
4351   }
4352   _timer_rsc_phase2.stop();
4353 
4354 } // end redefine_single_class()

4358 // and in all direct and indirect subclasses.
4359 void VM_RedefineClasses::increment_class_counter(InstanceKlass *ik, TRAPS) {
4360   oop class_mirror = ik->java_mirror();

4362   int new_count = java_lang_Class::classRedefinedCount(class_mirror) + 1;
4363   java_lang_Class::set_classRedefinedCount(class_mirror, new_count);
4364 
4365   if (class_oop != _the_class) {

4367     log_debug(redefine, class, subclass)("updated count in subclass=%s to %d", ik->external_name(), new_count);
4368   }
4369 
4370   for (Klass *subk = ik->subklass(); subk != NULL;
4371        subk = subk->next_sibling()) {

4374       InstanceKlass *subik = InstanceKlass::cast(subk);

4376       increment_class_counter(subik, THREAD);
4377     }
4378   }

4381 void VM_RedefineClasses::CheckClass::do_klass(Klass* k) {
4382   bool no_old_methods = true;  // be optimistic
4383 
4384   // Both array and instance classes have vtables.
4385   // a vtable should never contain old or obsolete methods
4386   ResourceMark rm(_thread);
4387   if (k->vtable_length() > 0 &&
4388       !k->vtable().check_no_old_or_obsolete_entries()) {
4389     if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4390       log_trace(redefine, class, obsolete, metadata)
4391         ("klassVtable::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4392          k->signature_name());
4393       k->vtable().dump_vtable();
4394     }
4395     no_old_methods = false;
4396   }
4397 
4398   if (k->is_instance_klass()) {
4399     HandleMark hm(_thread);
4400     InstanceKlass *ik = InstanceKlass::cast(k);
4401 
4402     // an itable should never contain old or obsolete methods
4403     if (ik->itable_length() > 0 &&
4404         !ik->itable().check_no_old_or_obsolete_entries()) {
4405       if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4406         log_trace(redefine, class, obsolete, metadata)
4407           ("klassItable::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4408            ik->signature_name());
4409         ik->itable().dump_itable();
4410       }
4411       no_old_methods = false;
4412     }
4413 
4414     // the constant pool cache should never contain non-deleted old or obsolete methods
4415     if (ik->constants() != NULL &&
4416         ik->constants()->cache() != NULL &&
4417         !ik->constants()->cache()->check_no_old_or_obsolete_entries()) {
4418       if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4419         log_trace(redefine, class, obsolete, metadata)
4420           ("cp-cache::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4421            ik->signature_name());
4422         ik->constants()->cache()->dump_cache();
4423       }
4424       no_old_methods = false;
4425     }
4426   }
4427 
4428   // print and fail guarantee if old methods are found.
4429   if (!no_old_methods) {
4430     if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4431       dump_methods();
4432     } else {
4433       log_trace(redefine, class)("Use the '-Xlog:redefine+class*:' option "
4434         "to see more info about the following guarantee() failure.");
4435     }
4436     guarantee(false, "OLD and/or OBSOLETE method(s) found");
4437   }
4438 }
