308      * Create a Type from an Java type signature.
309      * @exception CompilerError invalid type signature.
310      */
311     public static synchronized Type tType(String sig) {
312         Type t = typeHash.get(sig);
313         if (t != null) {
314             return t;
315         }
316 
317         switch (sig.charAt(0)) {
318           case SIGC_ARRAY:

320 
321           case SIGC_CLASS:
