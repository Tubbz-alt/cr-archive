26 package com.sun.security.auth.module;
27 
28 /**
29  * This class implementation retrieves and makes available Unix
30  * UID/GID/groups information for the current user.
31  */
32 public class UnixSystem {
33 
34     private native void getUnixInfo();
35 
36     // Warning: the following 4 fields are used by Unix.c
37     protected String username;
38     protected long uid;
39     protected long gid;
40     protected long[] groups;
41 
42     /**
43      * Instantiate a {@code UnixSystem} and load
44      * the native library to access the underlying system information.
45      */
46     public UnixSystem() {
47         System.loadLibrary("jaas");
48         getUnixInfo();
49     }
50 
51     /**
52      * Get the username for the current Unix user.
53      *
54      * @return the username for the current Unix user.
55      */
56     public String getUsername() {
57         return username;
58     }
59 
60     /**
61      * Get the UID for the current Unix user.
62      *
63      * @return the UID for the current Unix user.
64      */
65     public long getUid() {
66         return uid;
67     }
68 
69     /**
70      * Get the GID for the current Unix user.
71      *
72      * @return the GID for the current Unix user.
73      */
74     public long getGid() {
75         return gid;
76     }
77 
78     /**
79      * Get the supplementary groups for the current Unix user.
80      *
81      * @return the supplementary groups for the current Unix user.
82      */
83     public long[] getGroups() {
84         return groups == null ? null : groups.clone();
85     }
86 }
