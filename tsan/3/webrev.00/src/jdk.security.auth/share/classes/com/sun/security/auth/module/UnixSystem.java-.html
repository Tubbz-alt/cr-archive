26 package com.sun.security.auth.module;
27 
28 /**
29  * This class implementation retrieves and makes available Unix
30  * UID/GID/groups information for the current user.
31  */
32 public class UnixSystem {
33 
34     private native void getUnixInfo();
35 
36     protected String username;
37     protected long uid;
38     protected long gid;
39     protected long[] groups;
40 
41     /**
42      * Instantiate a {@code UnixSystem} and load
43      * the native library to access the underlying system information.
44      */
45     public UnixSystem() {
46         System.loadLibrary("jaas");
47         getUnixInfo();
48     }
49 
50     /**
51      * Get the username for the current Unix user.
52      *
53      * @return the username for the current Unix user.
54      */
55     public String getUsername() {
56         return username;
57     }
58 
59     /**
60      * Get the UID for the current Unix user.
61      *
62      * @return the UID for the current Unix user.
63      */
64     public long getUid() {
65         return uid;
66     }
67 
68     /**
69      * Get the GID for the current Unix user.
70      *
71      * @return the GID for the current Unix user.
72      */
73     public long getGid() {
74         return gid;
75     }
76 
77     /**
78      * Get the supplementary groups for the current Unix user.
79      *
80      * @return the supplementary groups for the current Unix user.
81      */
82     public long[] getGroups() {
83         return groups == null ? null : groups.clone();
84     }
85 }
