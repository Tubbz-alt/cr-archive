25 package sun.awt.windows;
26 
27 import java.awt.*;
28 import java.awt.peer.*;
29 
30 class WMenuPeer extends WMenuItemPeer implements MenuPeer {
31 
32     // MenuPeer implementation



33     @Override
34     public void addItem(MenuItem item) {

35     }
36     @Override
37     public native void delItem(int index);
38 
39     // Toolkit & peer internals
40 
41     WMenuPeer() {}   // used by subclasses.
42 
43     WMenuPeer(Menu target) {
44         this.target = target;
45         MenuContainer parent = target.getParent();
46 
47         if (parent instanceof MenuBar) {
48             WMenuBarPeer mbPeer = (WMenuBarPeer) WToolkit.targetToPeer(parent);
49             this.parent = mbPeer;
50             mbPeer.addChildPeer(this);
51             createMenu(mbPeer);
52         }
53         else if (parent instanceof Menu) {
54             this.parent = (WMenuPeer) WToolkit.targetToPeer(parent);
55             this.parent.addChildPeer(this);
56             createSubMenu(this.parent);
57         }
58         else {
59             throw new IllegalArgumentException("unknown menu container class");
60         }
61         // fix for 5088782: check if menu object is created successfully
62         checkMenuCreation();
63     }
64 
65     native void createMenu(WMenuBarPeer parent);
66     native void createSubMenu(WMenuPeer parent);
67 }
