26 package sun.awt;
27 
28 import java.awt.GraphicsEnvironment;
29 import java.awt.Toolkit;
30 
31 public class PlatformGraphicsInfo {
32 
33     public static GraphicsEnvironment createGE() {
34         return new Win32GraphicsEnvironment();
35     }
36 
37     public static Toolkit createToolkit() {
38         return new sun.awt.windows.WToolkit();
39     }
40 
41     public static boolean getDefaultHeadlessProperty() {
42         // On Windows, we assume we can always create headful apps.
43         // Here is where we can add code that would actually check.
44         return false;
45     }
46 
47     /*
48      * Called from java.awt.GraphicsEnvironment when
49      * getDefaultHeadlessProperty() has returned true, and
50      * the application has called an API that requires headful.
51      */
52     public static String getDefaultHeadlessMessage() {
53         return
54             "\nThe application does not have desktop access,\n" +
55             "but this program performed an operation which requires it.";
56     }
57 
58 }
