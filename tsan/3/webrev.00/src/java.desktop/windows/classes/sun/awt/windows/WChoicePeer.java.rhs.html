 26 package sun.awt.windows;
 27 
 28 import java.awt.Choice;
 29 import java.awt.Component;
 30 import java.awt.Dimension;
 31 import java.awt.FontMetrics;
 32 import java.awt.Window;
 33 import java.awt.event.ItemEvent;
 34 import java.awt.event.WindowAdapter;
 35 import java.awt.event.WindowEvent;
 36 import java.awt.event.WindowListener;
 37 import java.awt.peer.ChoicePeer;
 38 
 39 import sun.awt.AWTAccessor;
 40 import sun.awt.SunToolkit;
 41 
 42 final class WChoicePeer extends WComponentPeer implements ChoicePeer {
 43 
 44     // WComponentPeer overrides
 45 
 46     @Override
 47     public Dimension getMinimumSize() {
 48         FontMetrics fm = getFontMetrics(((Choice)target).getFont());
 49         Choice c = (Choice)target;
 50         int w = 0;
 51         for (int i = c.getItemCount() ; i-- > 0 ;) {
 52             w = Math.max(fm.stringWidth(c.getItem(i)), w);
 53         }
 54         return new Dimension(28 + w, Math.max(fm.getHeight() + 6, 15));
 55     }
 56     @Override
 57     public boolean isFocusable() {
 58         return true;
 59     }
 60 
 61     // ChoicePeer implementation
 62 
 63     @Override
 64     public native void select(int index);
 65 
 66     @Override
 67     public void add(String item, int index) {
 68         addItems(new String[] {item}, index);
 69     }
 70 
 71     @Override
 72     public boolean shouldClearRectBeforePaint() {
 73         return false;
 74     }
 75 
 76     @Override
 77     public native void removeAll();
 78     @Override
 79     public native void remove(int index);
 80 
 81     public native void addItems(String[] items, int index);
 82 
 83     @Override
 84     public synchronized native void reshape(int x, int y, int width, int height);
 85 
 86     private WindowListener windowListener;
 87 
 88     // Toolkit & peer internals
 89 
 90     WChoicePeer(Choice target) {
 91         super(target);
 92     }
 93 
 94     @Override
 95     native void create(WComponentPeer parent);
 96 
 97     @Override
 98     void initialize() {
 99         Choice opt = (Choice)target;
100         int itemCount = opt.getItemCount();
101         if (itemCount > 0) {
102             String[] items = new String[itemCount];
103             for (int i=0; i < itemCount; i++) {
104                 items[i] = opt.getItem(i);
105             }
106             addItems(items, 0);
107             if (opt.getSelectedIndex() >= 0) {
108                 select(opt.getSelectedIndex());
109             }
110         }
111 
112         Window parentWindow = SunToolkit.getContainingWindow((Component)target);
113         if (parentWindow != null) {
114             final WWindowPeer wpeer = AWTAccessor.getComponentAccessor()
115                                                  .getPeer(parentWindow);
116             if (wpeer != null) {
117                 windowListener = new WindowAdapter() {
118                         @Override
119                         public void windowIconified(WindowEvent e) {
120                             closeList();
121                         }
122                         @Override
123                         public void windowClosing(WindowEvent e) {
124                             closeList();
125                         }
126                     };
127                 wpeer.addWindowListener(windowListener);
128             }
129         }
130         super.initialize();
131     }
132 
133     @Override
134     protected void disposeImpl() {
135         // TODO: we should somehow reset the listener when the choice
136         // is moved to another toplevel without destroying its peer.
137         Window parentWindow = SunToolkit.getContainingWindow((Component)target);
138         if (parentWindow != null) {
139             final WWindowPeer wpeer = AWTAccessor.getComponentAccessor()
140                                                 .getPeer(parentWindow);
141             if (wpeer != null) {
142                 wpeer.removeWindowListener(windowListener);
143             }
144         }
145         super.disposeImpl();
146     }
147 
148     // native callbacks
149 
150     void handleAction(final int index) {
151         final Choice c = (Choice)target;
152         WToolkit.executeOnEventHandlerThread(c, new Runnable() {
153             @Override
154             public void run() {
155                 c.select(index);
156                 postEvent(new ItemEvent(c, ItemEvent.ITEM_STATE_CHANGED,
157                                 c.getItem(index), ItemEvent.SELECTED));
158             }
159         });
160     }
161 







162     native void closeList();
163 }
