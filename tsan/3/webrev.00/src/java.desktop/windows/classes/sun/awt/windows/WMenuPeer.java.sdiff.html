28 import java.awt.peer.*;
29 
30 class WMenuPeer extends WMenuItemPeer implements MenuPeer {
31 
32     // MenuPeer implementation
33 
34     @Override
35     public native void addSeparator();
36     @Override
37     public void addItem(MenuItem item) {
38         WMenuItemPeer itemPeer = (WMenuItemPeer) WToolkit.targetToPeer(item);
39     }
40     @Override
41     public native void delItem(int index);
42 
43     // Toolkit & peer internals
44 
45     WMenuPeer() {}   // used by subclasses.
46 
47     WMenuPeer(Menu target) {
48         this.target = target;
49         MenuContainer parent = target.getParent();
50 
51         if (parent instanceof MenuBar) {
52             WMenuBarPeer mbPeer = (WMenuBarPeer) WToolkit.targetToPeer(parent);
53             this.parent = mbPeer;
54             mbPeer.addChildPeer(this);
55             createMenu(mbPeer);
56         }
57         else if (parent instanceof Menu) {
58             this.parent = (WMenuPeer) WToolkit.targetToPeer(parent);
