26 package com.sun.media.sound;
27 
28 import javax.sound.midi.InvalidMidiDataException;
29 import javax.sound.midi.SysexMessage;
30 
31 /**
32  * optimized FastSysexMessage that doesn't copy the array upon instantiation.
33  *
34  * @author Florian Bomers
35  */
36 final class FastSysexMessage extends SysexMessage {
37 
38     FastSysexMessage(byte[] data) throws InvalidMidiDataException {
39         super(data);
40         if (data.length==0 || (((data[0] & 0xFF) != 0xF0) && ((data[0] & 0xFF) != 0xF7))) {
41             super.setMessage(data, data.length); // will throw Exception
42         }
43     }
44 
45     /**
46      * The returned array may be larger than this message is.
47      * Use getLength() to get the real length of the message.
48      */
49     byte[] getReadOnlyMessage() {
50         return data;
51     }
52 
53     // overwrite this method so that the original data array,
54     // which is shared among all transmitters, cannot be modified
55     @Override
56     public void setMessage(byte[] data, int length) throws InvalidMidiDataException {
57         if ((data.length == 0) || (((data[0] & 0xFF) != 0xF0) && ((data[0] & 0xFF) != 0xF7))) {
58             super.setMessage(data, data.length); // will throw Exception
59         }
60         this.length = length;
61         this.data = new byte[this.length];
62         System.arraycopy(data, 0, this.data, 0, length);
63     }
64 
65 } // class FastSysexMessage
