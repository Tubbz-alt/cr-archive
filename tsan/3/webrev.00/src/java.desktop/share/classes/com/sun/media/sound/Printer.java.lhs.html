 38     static final boolean debug = false;
 39     static final boolean trace = false;
 40     static final boolean verbose = false;
 41     static final boolean release = false;
 42 
 43     static final boolean SHOW_THREADID = false;
 44     static final boolean SHOW_TIMESTAMP = false;

 71     /**
 72      * Suppresses default constructor, ensuring non-instantiability.
 73      */
 74     private Printer() {
 75     }
 76 







 77     public static void err(String str) {
 78 
 79         if (err)
 80             println(str);
 81     }
 82 
 83     public static void debug(String str) {
 84 
 85         if (debug)
 86             println(str);
 87     }
 88 
 89     public static void trace(String str) {
 90 
 91         if (trace)
 92             println(str);
 93     }
 94 
 95     public static void verbose(String str) {
 96 
 97         if (verbose)
 98             println(str);

 99     }
100 
101     public static void release(String str) {
102 
103         if (release)
104             println(str);



105     }
106 
107     private static long startTime = 0;
108 
109     public static void println(String s) {
110         String prepend = "";
111         if (SHOW_THREADID) {
112             prepend = "thread "  + Thread.currentThread().getId() + " " + prepend;
113         }
114         if (SHOW_TIMESTAMP) {
115             if (startTime == 0) {
116                 startTime = System.nanoTime() / 1000000l;
117             }
118             prepend = prepend + ((System.nanoTime()/1000000l) - startTime) + "millis: ";
119         }
120         System.out.println(prepend + s);
121     }
122 
123     public static void println() {
124         System.out.println();
125     }
126
