 42 /** Managing security in the Java Sound implementation.

 55     static void checkRecordPermission() throws SecurityException {
 56         if(Printer.trace) Printer.trace("JSSecurityManager.checkRecordPermission()");
 57         SecurityManager sm = System.getSecurityManager();
 58         if (sm != null) {
 59             sm.checkPermission(new AudioPermission("record"));
 60         }
 61     }
