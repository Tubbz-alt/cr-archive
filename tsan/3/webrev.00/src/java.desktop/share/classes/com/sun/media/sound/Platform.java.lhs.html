 26 package com.sun.media.sound;
 27 
 28 import java.security.AccessController;
 29 import java.security.PrivilegedAction;
 30 import java.util.StringTokenizer;
 31 
 32 /**
 33  * Audio configuration class for exposing attributes specific to the platform or system.
 34  *
 35  * @author Kara Kytle
 36  * @author Florian Bomers
 37  */
 38 final class Platform {
 39 
 40     // native library we need to load
 41     private static final String libName = "jsound";
 42 
 43     private static boolean isNativeLibLoaded;
 44 
 45     // SYSTEM CHARACTERISTICS
 46     // vary according to hardware architecture
 47 
 48     // intel is little-endian.  sparc is big-endian.
 49     private static boolean bigEndian;
 50 
 51     static {
 52         if(Printer.trace)Printer.trace(">> Platform.java: static");
 53 
 54         loadLibraries();
 55     }
 56 
 57     /**
 58      * Private constructor.
 59      */
 60     private Platform() {
 61     }
 62 
 63     /**
 64      * Dummy method for forcing initialization.
 65      */
 66     static void initialize() {
 67         if(Printer.trace)Printer.trace("Platform: initialize()");
 68     }
 69 
 70     /**
 71      * Determine whether the system is big-endian.
 72      */
 73     static boolean isBigEndian() {
 74         return bigEndian;
 75     }
 76 
 77     /**
 78      * Load the native library or libraries.
 79      */
 80     private static void loadLibraries() {
 81         if(Printer.trace)Printer.trace(">>Platform.loadLibraries");
 82 
 83         // load the native library
 84         isNativeLibLoaded = true;
 85         try {
 86             AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
 87                 System.loadLibrary(libName);
 88                 return null;
 89             });
 90         } catch (Throwable t) {
 91             if (Printer.err) Printer.err("Couldn't load library "+libName+": "+t.toString());
 92             isNativeLibLoaded = false;
 93         }
 94         if (isNativeLibLoaded) {
 95             bigEndian = nIsBigEndian();
 96         }
 97     }
 98 
 99     static boolean isMidiIOEnabled() {
100         if (Printer.debug) Printer.debug("Platform: Checking for MidiIO; library is loaded=" + isNativeLibLoaded);
101         return isNativeLibLoaded;
102     }
103 
104     static boolean isPortsEnabled() {
105         if (Printer.debug) Printer.debug("Platform: Checking for Ports; library is loaded=" + isNativeLibLoaded);
106         return isNativeLibLoaded;
107     }
108 
109     static boolean isDirectAudioEnabled() {
110         if (Printer.debug) Printer.debug("Platform: Checking for DirectAudio; library is loaded=" + isNativeLibLoaded);
111         return isNativeLibLoaded;
112     }
113 
114     // the following native method is implemented in Platform.c
115     private static native boolean nIsBigEndian();
116 }
