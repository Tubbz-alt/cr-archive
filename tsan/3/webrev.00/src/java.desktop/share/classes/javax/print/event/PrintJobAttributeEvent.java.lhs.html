26 package javax.print.event;
27 
28 import javax.print.DocPrintJob;
29 import javax.print.attribute.AttributeSetUtilities;
30 import javax.print.attribute.PrintJobAttributeSet;
31 
32 /**
33  * Class {@code PrintJobAttributeEvent} encapsulates an event a
34  * {@code PrintService} reports to let the client know that one or more printing
35  * attributes for a {@code PrintJob} have changed.
36  */
37 public class PrintJobAttributeEvent extends PrintEvent {
38 
39     /**
40      * Use serialVersionUID from JDK 1.4 for interoperability.
41      */
42     private static final long serialVersionUID = -6534469883874742101L;
43 
44     /**
45      * The printing service attributes that changed.
46      */

47     private PrintJobAttributeSet attributes;
48 
49     /**
50      * Constructs a {@code PrintJobAttributeEvent} object.
51      *
52      * @param  source the print job generating this event
53      * @param  attributes the attribute changes being reported
54      * @throws IllegalArgumentException if {@code source} is {@code null}
55      */
56     public PrintJobAttributeEvent (DocPrintJob source,
57                                    PrintJobAttributeSet attributes)  {
58         super(source);
59 
60         this.attributes = AttributeSetUtilities.unmodifiableView(attributes);
61     }
62 
63     /**
64      * Determine the {@code PrintJob} to which this print job event pertains.
65      *
66      * @return {@code PrintJob} object
67      */
68     public DocPrintJob getPrintJob() {
69 
70         return (DocPrintJob) getSource();
71     }
72 
73     /**
74      * Determine the printing attributes that changed and their new values.
75      *
76      * @return attributes containing the new values for the print job attributes
77      *         that changed. The returned set may not be modifiable.
78      */
79     public PrintJobAttributeSet getAttributes() {
80 
81         return attributes;
82     }
83 }
