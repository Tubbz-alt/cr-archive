 34 <h1>BMP Metadata Format Specification</h1>
 35 
 36 The XML schema for the native image metadata format is as follows:
 37 
 38 <pre>
 39 &lt;?xml version="1.0" encoding="UTF-8"?&gt;
 40 
 41 &lt;!-- Schema for BMP native image metadata format. --&gt;
 42 
 43 &lt;xsd:schema xmlns:xsd="http://www.w3.org/2001/XMLSchema"
 44             xmlns="http://javax.imageio.plugins"
 45             targetNamespace="http://javax.imageio.plugins"&gt;
 46 
 47   &lt;!-- Coordinates of a point in the CIE XYZ color space. --&gt;
 48   &lt;xsd:complexType name="XYZPoint"&gt;
 49     &lt;xsd:sequence&gt;
 50       &lt;xsd:element name="X" type="xsd:double"/&gt;
 51       &lt;xsd:element name="Y" type="xsd:double"/&gt;
 52       &lt;xsd:element name="Z" type="xsd:double"/&gt;
 53     &lt;/xsd:sequence&gt;

144                   &lt;xsd:sequence&gt;
145                     &lt;xsd:element name="Red" type="xsd:unsignedByte"/&gt;
146                     &lt;xsd:element name="Green" type="xsd:unsignedByte"/&gt;
147                     &lt;xsd:element name="Blue" type="xsd:unsignedByte"/&gt;
148                     &lt;xsd:element name="Alpha" type="xsd:unsignedByte" minOccurs="0"/&gt;
149                   &lt;/xsd:sequence&gt;
150 	        &lt;/xsd:complexType&gt;
151               &lt;/xsd:element&gt;
152             &lt;/xsd:sequence&gt;
153 	  &lt;/xsd:complexType&gt;
154 	&lt;/xsd:element&gt;
155 
156       &lt;/xsd:sequence&gt;
157     &lt;/xsd:complexType&gt;
158   &lt;/xsd:element&gt; &lt;!-- bmp_image_1.0 --&gt;
159 
160 &lt;/xsd:schema&gt;
161 </pre>
162
