26 package javax.print.event;
27 
28 import javax.print.PrintService;
29 import javax.print.attribute.AttributeSetUtilities;
30 import javax.print.attribute.PrintServiceAttributeSet;
31 
32 /**
33  * Class {@code PrintServiceAttributeEvent} encapsulates an event a Print
34  * Service instance reports to let the client know of changes in the print
35  * service state.
36  */
37 public class PrintServiceAttributeEvent extends PrintEvent {
38 
39     /**
40      * Use serialVersionUID from JDK 1.4 for interoperability.
41      */
42     private static final long serialVersionUID = -7565987018140326600L;
43 
44     /**
45      * The printing service attributes that changed.
46      */
47     @SuppressWarnings("serial") // Not statically typed as Serializable
48     private PrintServiceAttributeSet attributes;
49 
50     /**
51      * Constructs a {@code PrintServiceAttributeEvent} object.
52      *
53      * @param  source the print job generating this event
54      * @param  attributes the attribute changes being reported
55      * @throws IllegalArgumentException if {@code source} is {@code null}
56      */
57     public PrintServiceAttributeEvent(PrintService source,
58                                       PrintServiceAttributeSet attributes) {
59 
60         super(source);
61         this.attributes = AttributeSetUtilities.unmodifiableView(attributes);
62     }
63 
64     /**
65      * Returns the print service.
66      *
67      * @return {@code PrintService} object
68      */
69     public PrintService getPrintService() {
70 
71         return (PrintService) getSource();
72     }
73 
74     /**
75      * Determine the printing service attributes that changed and their new
76      * values.
77      *
78      * @return attributes containing the new values for the service attributes
79      *         that changed. The returned set may be unmodifiable.
80      */
81     public PrintServiceAttributeSet getAttributes() {
82 
83         return attributes;
84     }
85 }
