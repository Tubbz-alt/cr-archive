26 package javax.print.event;
27 
28 import javax.print.DocPrintJob;
29 import javax.print.attribute.AttributeSetUtilities;
30 import javax.print.attribute.PrintJobAttributeSet;
31 
32 /**
33  * Class {@code PrintJobAttributeEvent} encapsulates an event a
34  * {@code PrintService} reports to let the client know that one or more printing
35  * attributes for a {@code PrintJob} have changed.
36  */
37 public class PrintJobAttributeEvent extends PrintEvent {
38 
39     /**
40      * Use serialVersionUID from JDK 1.4 for interoperability.
41      */
42     private static final long serialVersionUID = -6534469883874742101L;
43 
44     /**
45      * The printing service attributes that changed.
46      */
47     @SuppressWarnings("serial") // Not statically typed as Serializable
48     private PrintJobAttributeSet attributes;
49 
50     /**
51      * Constructs a {@code PrintJobAttributeEvent} object.
52      *
53      * @param  source the print job generating this event
54      * @param  attributes the attribute changes being reported
55      * @throws IllegalArgumentException if {@code source} is {@code null}
56      */
57     public PrintJobAttributeEvent (DocPrintJob source,
58                                    PrintJobAttributeSet attributes)  {
59         super(source);
60 
61         this.attributes = AttributeSetUtilities.unmodifiableView(attributes);
62     }
63 
64     /**
65      * Determine the {@code PrintJob} to which this print job event pertains.
66      *
67      * @return {@code PrintJob} object
68      */
69     public DocPrintJob getPrintJob() {
70 
71         return (DocPrintJob) getSource();
72     }
73 
74     /**
75      * Determine the printing attributes that changed and their new values.
76      *
77      * @return attributes containing the new values for the print job attributes
78      *         that changed. The returned set may not be modifiable.
79      */
80     public PrintJobAttributeSet getAttributes() {
81 
82         return attributes;
83     }
84 }
