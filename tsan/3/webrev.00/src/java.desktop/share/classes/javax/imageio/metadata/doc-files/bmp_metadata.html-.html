 34 <h1>BMP Metadata Format Specification</h1>
 35 
 36 The XML schema for the native image metadata format is as follows:
 37 
 38 <pre>
 39 &lt;?xml version="1.0" encoding="UTF-8"?&gt;
 40 
 41 &lt;!-- Schema for BMP native image metadata format. --&gt;
 42 
 43 &lt;xsd:schema xmlns:xsd="http://www.w3.org/2001/XMLSchema"
 44             xmlns="http://javax.imageio.plugins"
 45             targetNamespace="http://javax.imageio.plugins"&gt;
 46 
 47   &lt;!-- Coordinates of a point in the CIE XYZ color space. --&gt;
 48   &lt;xsd:complexType name="XYZPoint"&gt;
 49     &lt;xsd:sequence&gt;
 50       &lt;xsd:element name="X" type="xsd:double"/&gt;
 51       &lt;xsd:element name="Y" type="xsd:double"/&gt;
 52       &lt;xsd:element name="Z" type="xsd:double"/&gt;
 53     &lt;/xsd:sequence&gt;
 54   &lt;/xsd:complexType&gt;
 55 
 56   &lt;!-- BMP Schema 1.0 root element. --&gt;
 57   &lt;xsd:element name="javax_imageio_bmp_1.0"&gt;
 58     &lt;xsd:complexType&gt;
 59       &lt;xsd:sequence&gt;
 60 
 61         &lt;!-- BMP version string --&gt;
 62         &lt;xsd:element name="BMPVersion" type="xsd:string"/&gt;
 63 
 64         &lt;!-- Bitmap width --&gt;
 65         &lt;xsd:element name="Width" type="xsd:unsignedInt"/&gt;
 66 
 67         &lt;!-- Bitmap height --&gt;
 68         &lt;xsd:element name="Height" type="xsd:unsignedInt"/&gt;
 69 
 70         &lt;!-- Number of bits per pixel --&gt;
 71         &lt;xsd:element name="BitsPerPixel" type="xsd:unsignedShort"/&gt;
 72 
 73         &lt;!-- Compression type --&gt;
 74         &lt;xsd:element name="Compression" type="xsd:unsignedInt" minOccurs="0"/&gt;
 75 
 76         &lt;!-- Image size in bytes --&gt;
 77         &lt;xsd:element name="ImageSize" type="xsd:unsignedInt" minOccurs="0"/&gt;
 78 
 79 	&lt;!-- Resolution in pixels per unit distance --&gt;
 80         &lt;xsd:element name="PixelsPerMeter" minOccurs="0"&gt;
 81 	  &lt;xsd:complexType&gt;
 82             &lt;xsd:sequence&gt;
 83               &lt;xsd:element name="X" type="xsd:unsignedInt"/&gt;
 84               &lt;xsd:element name="Y" type="xsd:unsignedInt"/&gt;
 85             &lt;/xsd:sequence&gt;
 86 	  &lt;/xsd:complexType&gt;
 87         &lt;/xsd:element&gt; &lt;!-- PixelsPerMeter --&gt;
 88 
 89 
 90 	&lt;!-- Number of color indexes in the color table actually used --&gt;
 91         &lt;xsd:element name="ColorsUsed" type="xsd:unsignedInt" minOccurs="0"/&gt;
 92 
 93 	&lt;!-- Number of color indexes considered important for display --&gt;
 94         &lt;xsd:element name="ColorsImportant" type="xsd:unsignedInt"
 95 		     minOccurs="0"/&gt;
 96 
 97 	&lt;!-- Color masks; present for BI_BITFIELDS compression only --&gt;
 98         &lt;xsd:element name="Mask" minOccurs="0"&gt;
 99 	  &lt;xsd:complexType&gt;
100             &lt;xsd:sequence&gt;
101               &lt;xsd:element name="Red" type="xsd:unsignedInt"/&gt;
102               &lt;xsd:element name="Green" type="xsd:unsignedInt"/&gt;
103               &lt;xsd:element name="Blue" type="xsd:unsignedInt"/&gt;
104               &lt;xsd:element name="Alpha" type="xsd:unsignedInt" minOccurs="0"/&gt;
105             &lt;/xsd:sequence&gt;
106 	  &lt;/xsd:complexType&gt;
107 	&lt;/xsd:element&gt;
108 
109         &lt;!-- Color space  --&gt;
110         &lt;xsd:element name="ColorSpaceType" type="xsd:unsignedInt"
111 		     minOccurs="0"/&gt;
112 
113 	&lt;!-- CIE XYZ for the LCS_CALIBRATED_RGB color space --&gt;
114         &lt;xsd:element name="CIEXYZEndpoints" minOccurs="0"&gt;
115 	  &lt;xsd:complexType&gt;
116             &lt;xsd:sequence&gt;
117               &lt;xsd:element name="Red" type="XYZPoint"/&gt;
118               &lt;xsd:element name="Green" type="XYZPoint"/&gt;
119               &lt;xsd:element name="Blue" type="XYZPoint"/&gt;
120             &lt;/xsd:sequence&gt;
121 	  &lt;/xsd:complexType&gt;
122         &lt;/xsd:element&gt;
123 
124 	&lt;!-- Gamma values for the LCS_CALIBRATED_RGB color space --&gt;
125         &lt;xsd:element name="Gamma" minOccurs="0"&gt;
126 	  &lt;xsd:complexType&gt;
127             &lt;xsd:sequence&gt;
128               &lt;xsd:element name="Red" type="xsd:unsignedInt"/&gt;
129               &lt;xsd:element name="Green" type="xsd:unsignedInt"/&gt;
130               &lt;xsd:element name="Blue" type="xsd:unsignedInt"/&gt;
131             &lt;/xsd:sequence&gt;
132 	  &lt;/xsd:complexType&gt;
133 	&lt;/xsd:element&gt;
134 
135 	&lt;!-- Rendering intent --&gt;
136         &lt;xsd:element name="Intent" type="xsd:unsignedInt" minOccurs="0"/&gt;
137 
138 	&lt;!-- The image colormap --&gt;
139         &lt;xsd:element name="Palette" minOccurs="0"&gt;
140 	  &lt;xsd:complexType&gt;
141             &lt;xsd:sequence&gt;
142               &lt;xsd:element name="PaletteEntry"&gt;
143 	        &lt;xsd:complexType&gt;
144                   &lt;xsd:sequence&gt;
145                     &lt;xsd:element name="Red" type="xsd:unsignedByte"/&gt;
146                     &lt;xsd:element name="Green" type="xsd:unsignedByte"/&gt;
147                     &lt;xsd:element name="Blue" type="xsd:unsignedByte"/&gt;
148                     &lt;xsd:element name="Alpha" type="xsd:unsignedByte" minOccurs="0"/&gt;
149                   &lt;/xsd:sequence&gt;
150 	        &lt;/xsd:complexType&gt;
151               &lt;/xsd:element&gt;
152             &lt;/xsd:sequence&gt;
153 	  &lt;/xsd:complexType&gt;
154 	&lt;/xsd:element&gt;
155 
156       &lt;/xsd:sequence&gt;
157     &lt;/xsd:complexType&gt;
158   &lt;/xsd:element&gt; &lt;!-- bmp_image_1.0 --&gt;
159 
160 &lt;/xsd:schema&gt;
161 </pre>
