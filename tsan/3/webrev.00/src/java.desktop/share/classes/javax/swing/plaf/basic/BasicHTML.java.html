 38 /**
 39  * Support for providing html views for the swing components.
 40  * This translates a simple html string to a javax.swing.text.View
 41  * implementation that can render the html and provide the necessary
 42  * layout semantics.
 43  *
 44  * @author  Timothy Prinzing
 45  * @since 1.3
 46  */

 49     /**
 50      * Create an html renderer for the given component and
 51      * string of html.
 52      *
 53      * @param c a component
 54      * @param html an HTML string
 55      * @return an HTML renderer
 56      */
 57     public static View createHTMLView(JComponent c, String html) {
 58         BasicEditorKit kit = getFactory();
 59         Document doc = kit.createDefaultDocument(c.getFont(),
 60                                                  c.getForeground());
 61         Object base = c.getClientProperty(documentBaseKey);
 62         if (base instanceof URL) {
 63             ((HTMLDocument)doc).setBase((URL)base);
 64         }
 65         Reader r = new StringReader(html);
 66         try {
 67             kit.read(r, doc, 0);
 68         } catch (Throwable e) {
 69         }
 70         ViewFactory f = kit.getViewFactory();
 71         View hview = f.create(doc.getDefaultRootElement());
 72         View v = new Renderer(c, f, hview);
 73         return v;
 74     }

 76     /**
 77      * Returns the baseline for the html renderer.
 78      *
 79      * @param view the View to get the baseline for
 80      * @param w the width to get the baseline for
 81      * @param h the height to get the baseline for
 82      * @throws IllegalArgumentException if width or height is &lt; 0
 83      * @return baseline or a value &lt; 0 indicating there is no reasonable
 84      *                  baseline
 85      * @see java.awt.FontMetrics
 86      * @see javax.swing.JComponent#getBaseline(int,int)
 87      * @since 1.6
 88      */
 89     public static int getHTMLBaseline(View view, int w, int h) {
 90         if (w < 0 || h < 0) {
 91             throw new IllegalArgumentException(
 92                     "Width and height must be >= 0");
 93         }
 94         if (view instanceof Renderer) {
 95             return getBaseline(view.getView(0), w, h);
 96         }
 97         return -1;
 98     }

100     /**
101      * Gets the baseline for the specified component.  This digs out
102      * the View client property, and if non-null the baseline is calculated
103      * from it.  Otherwise the baseline is the value <code>y + ascent</code>.
104      */
105     static int getBaseline(JComponent c, int y, int ascent,
106                                   int w, int h) {
107         View view = (View)c.getClientProperty(BasicHTML.propertyKey);
108         if (view != null) {
109             int baseline = getHTMLBaseline(view, w, h);
110             if (baseline < 0) {
111                 return baseline;
112             }
113             return y + baseline;
114         }
115         return y + ascent;
116     }

118     /**
119      * Gets the baseline for the specified View.
120      */
121     static int getBaseline(View view, int w, int h) {
122         if (hasParagraph(view)) {
123             view.setSize(w, h);
124             return getBaseline(view, new Rectangle(0, 0, w, h));
125         }
126         return -1;
127     }
128 
129     private static int getBaseline(View view, Shape bounds) {
130         if (view.getViewCount() == 0) {
131             return -1;
132         }
133         AttributeSet attributes = view.getElement().getAttributes();
134         Object name = null;
135         if (attributes != null) {
136             name = attributes.getAttribute(StyleConstants.NameAttribute);
137         }
138         int index = 0;
139         if (name == HTML.Tag.HTML && view.getViewCount() > 1) {
140             // For html on widgets the header is not visible, skip it.
141             index++;
142         }
143         bounds = view.getChildAllocation(index, bounds);
144         if (bounds == null) {
145             return -1;
146         }
147         View child = view.getView(index);
148         if (view instanceof javax.swing.text.ParagraphView) {
149             Rectangle rect;
150             if (bounds instanceof Rectangle) {
151                 rect = (Rectangle)bounds;
152             }
153             else {
154                 rect = bounds.getBounds();
155             }
156             return rect.y + (int)(rect.height *
157                                   child.getAlignment(View.Y_AXIS));
158         }
159         return getBaseline(child, bounds);
160     }

162     private static boolean hasParagraph(View view) {
163         if (view instanceof javax.swing.text.ParagraphView) {
164             return true;
165         }
166         if (view.getViewCount() == 0) {
167             return false;
168         }
169         AttributeSet attributes = view.getElement().getAttributes();
170         Object name = null;
171         if (attributes != null) {
172             name = attributes.getAttribute(StyleConstants.NameAttribute);
173         }
174         int index = 0;
175         if (name == HTML.Tag.HTML && view.getViewCount() > 1) {
176             // For html on widgets the header is not visible, skip it.
177             index = 1;
178         }
179         return hasParagraph(view.getView(index));
180     }

182     /**
183      * Check the given string to see if it should trigger the
184      * html rendering logic in a non-text component that supports
185      * html rendering.
186      *
187      * @param s a text
188      * @return {@code true} if the given string should trigger the
189      *         html rendering logic in a non-text component
190      */
191     public static boolean isHTMLString(String s) {
192         if (s != null) {
193             if ((s.length() >= 6) && (s.charAt(0) == '<') && (s.charAt(5) == '>')) {
194                 String tag = s.substring(1,5);
195                 return tag.equalsIgnoreCase(propertyKey);
196             }
197         }
198         return false;
199     }

201     /**
202      * Stash the HTML render for the given text into the client
203      * properties of the given JComponent. If the given text is
204      * <em>NOT HTML</em> the property will be cleared of any
205      * renderer.
206      * <p>
207      * This method is useful for ComponentUI implementations
208      * that are static (i.e. shared) and get their state
209      * entirely from the JComponent.
210      *
211      * @param c a component
212      * @param text a text
213      */
214     public static void updateRenderer(JComponent c, String text) {
215         View value = null;
216         View oldValue = (View)c.getClientProperty(BasicHTML.propertyKey);
217         Boolean htmlDisabled = (Boolean) c.getClientProperty(htmlDisable);
218         if (htmlDisabled != Boolean.TRUE && BasicHTML.isHTMLString(text)) {
219             value = BasicHTML.createHTMLView(c, text);
220         }
221         if (value != oldValue && oldValue != null) {
222             for (int i = 0; i < oldValue.getViewCount(); i++) {
223                 oldValue.getView(i).setParent(null);
224             }
225         }
226         c.putClientProperty(BasicHTML.propertyKey, value);
227         String currentAccessibleNameProperty =
228             (String) c.getClientProperty(AccessibleContext.ACCESSIBLE_NAME_PROPERTY);
229         String previousParsedText = null;
230         if (currentAccessibleNameProperty != null && oldValue != null) {
231             try {
232                 previousParsedText =
233                     (oldValue.getDocument().getText(0, oldValue.getDocument().getLength())).strip();
234             } catch (BadLocationException e) {
235             }
236         }
237 
238         // AccessibleContext.ACCESSIBLE_NAME_PROPERTY should be set from here only if,
239         // 1. If AccessibleContext.ACCESSIBLE_NAME_PROPERTY was NOT set before
240         //        i.e. currentAccessibleNameProperty is null. and,
241         // 2. If AccessibleContext.ACCESSIBLE_NAME_PROPERTY was previously set from this method
242         //        using the value.getDocument().getText().
243         if (currentAccessibleNameProperty == null ||
244                 currentAccessibleNameProperty.equals(previousParsedText)) {
245             String parsedText = null;
246             if (value != null) {
247                 try {
248                     parsedText =
249                         (value.getDocument().getText(0, value.getDocument().getLength())).strip();
250                 } catch (BadLocationException e) {
251                 }
252             }
253             c.putClientProperty(AccessibleContext.ACCESSIBLE_NAME_PROPERTY, parsedText);
254         }
255     }

303     private static final String styleChanges =

307     /**
308      * The views produced for the ComponentUI implementations aren't
309      * going to be edited and don't need full html support.  This kit
310      * alters the HTMLEditorKit to try and trim things down a bit.
311      * It does the following:
312      * <ul>
313      * <li>It doesn't produce Views for things like comments,
314      * head, title, unknown tags, etc.
315      * <li>It installs a different set of css settings from the default
316      * provided by HTMLEditorKit.
317      * </ul>
318      */
319     @SuppressWarnings("serial") // JDK-implementation class
320     static class BasicEditorKit extends HTMLEditorKit {
321         /** Shared base style for all documents created by us use. */
322         private static StyleSheet defaultStyles;
323 
324         /**
325          * Overriden to return our own slimmed down style sheet.
326          */
327         public StyleSheet getStyleSheet() {
328             if (defaultStyles == null) {
329                 defaultStyles = new StyleSheet();
330                 StringReader r = new StringReader(styleChanges);
331                 try {
332                     defaultStyles.loadRules(r, null);
333                 } catch (Throwable e) {
334                     // don't want to die in static initialization...
335                     // just display things wrong.
336                 }
337                 r.close();
338                 defaultStyles.addStyleSheet(super.getStyleSheet());
339             }
340             return defaultStyles;
341         }

343         /**
344          * Sets the async policy to flush everything in one chunk, and
345          * to not display unknown tags.
346          */
347         public Document createDefaultDocument(Font defaultFont,
348                                               Color foreground) {
349             StyleSheet styles = getStyleSheet();
350             StyleSheet ss = new StyleSheet();
351             ss.addStyleSheet(styles);
352             BasicDocument doc = new BasicDocument(ss, defaultFont, foreground);
353             doc.setAsynchronousLoadPriority(Integer.MAX_VALUE);
354             doc.setPreservesUnknownTags(false);
355             return doc;
356         }

384     /**
385      * The subclass of HTMLDocument that is used as the model. getForeground
386      * is overridden to return the foreground property from the Component this
387      * was created for.
388      */
389     @SuppressWarnings("serial") // Superclass is not serializable across versions
390     static class BasicDocument extends HTMLDocument {
391         /** The host, that is where we are rendering. */
392         // private JComponent host;
393 
394         BasicDocument(StyleSheet s, Font defaultFont, Color foreground) {
395             super(s);
396             setPreservesUnknownTags(false);
397             setFontAndColor(defaultFont, foreground);
398         }
399 
400         /**
401          * Sets the default font and default color. These are set by
402          * adding a rule for the body that specifies the font and color.
403          * This allows the html to override these should it wish to have
404          * a custom font or color.
405          */
406         private void setFontAndColor(Font font, Color fg) {
407             getStyleSheet().addRule(sun.swing.SwingUtilities2.
408                                     displayPropertiesToCSS(font,fg));
409         }
410     }

413     /**
414      * Root text view that acts as an HTML renderer.
415      */
416     static class Renderer extends View {
417 
418         Renderer(JComponent c, ViewFactory f, View v) {
419             super(null);
420             host = c;
421             factory = f;
422             view = v;
423             view.setParent(this);
424             // initially layout to the preferred size
425             setSize(view.getPreferredSpan(X_AXIS), view.getPreferredSpan(Y_AXIS));
426         }
427 
428         /**
429          * Fetches the attributes to use when rendering.  At the root
430          * level there are no attributes.  If an attribute is resolved
431          * up the view hierarchy this is the end of the line.
432          */
433         public AttributeSet getAttributes() {
434             return null;
435         }
436 
437         /**
438          * Determines the preferred span for this view along an axis.
439          *
440          * @param axis may be either X_AXIS or Y_AXIS
441          * @return the span the view would like to be rendered into.
442          *         Typically the view is told to render into the span
443          *         that is returned, although there is no guarantee.
444          *         The parent may choose to resize or break the view.
445          */
446         public float getPreferredSpan(int axis) {
447             if (axis == X_AXIS) {
448                 // width currently laid out to
449                 return width;
450             }
451             return view.getPreferredSpan(axis);
452         }
453 
454         /**
455          * Determines the minimum span for this view along an axis.
456          *
457          * @param axis may be either X_AXIS or Y_AXIS
458          * @return the span the view would like to be rendered into.
459          *         Typically the view is told to render into the span
460          *         that is returned, although there is no guarantee.
461          *         The parent may choose to resize or break the view.
462          */
463         public float getMinimumSpan(int axis) {
464             return view.getMinimumSpan(axis);
465         }
466 
467         /**
468          * Determines the maximum span for this view along an axis.
469          *
470          * @param axis may be either X_AXIS or Y_AXIS
471          * @return the span the view would like to be rendered into.
472          *         Typically the view is told to render into the span
473          *         that is returned, although there is no guarantee.
474          *         The parent may choose to resize or break the view.
475          */
476         public float getMaximumSpan(int axis) {
477             return Integer.MAX_VALUE;
478         }
479 
480         /**
481          * Specifies that a preference has changed.
482          * Child views can call this on the parent to indicate that
483          * the preference has changed.  The root view routes this to
484          * invalidate on the hosting component.
485          * <p>
486          * This can be called on a different thread from the
487          * event dispatching thread and is basically unsafe to
488          * propagate into the component.  To make this safe,
489          * the operation is transferred over to the event dispatching
490          * thread for completion.  It is a design goal that all view
491          * methods be safe to call without concern for concurrency,
492          * and this behavior helps make that true.
493          *
494          * @param child the child view
495          * @param width true if the width preference has changed
496          * @param height true if the height preference has changed
497          */
498         public void preferenceChanged(View child, boolean width, boolean height) {
499             host.revalidate();
500             host.repaint();
501         }
502 
503         /**
504          * Determines the desired alignment for this view along an axis.
505          *
506          * @param axis may be either X_AXIS or Y_AXIS
507          * @return the desired alignment, where 0.0 indicates the origin
508          *     and 1.0 the full span away from the origin
509          */
510         public float getAlignment(int axis) {
511             return view.getAlignment(axis);
512         }
