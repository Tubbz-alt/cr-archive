26 package javax.print.event;
27 
28 import javax.print.PrintService;
29 import javax.print.attribute.AttributeSetUtilities;
30 import javax.print.attribute.PrintServiceAttributeSet;
31 
32 /**
33  * Class {@code PrintServiceAttributeEvent} encapsulates an event a Print
34  * Service instance reports to let the client know of changes in the print
35  * service state.
36  */
37 public class PrintServiceAttributeEvent extends PrintEvent {
38 
39     /**
40      * Use serialVersionUID from JDK 1.4 for interoperability.
41      */
42     private static final long serialVersionUID = -7565987018140326600L;
43 
44     /**
45      * The printing service attributes that changed.
46      */

47     private PrintServiceAttributeSet attributes;
48 
49     /**
50      * Constructs a {@code PrintServiceAttributeEvent} object.
51      *
52      * @param  source the print job generating this event
53      * @param  attributes the attribute changes being reported
54      * @throws IllegalArgumentException if {@code source} is {@code null}
55      */
56     public PrintServiceAttributeEvent(PrintService source,
57                                       PrintServiceAttributeSet attributes) {
58 
59         super(source);
60         this.attributes = AttributeSetUtilities.unmodifiableView(attributes);
61     }
62 
63     /**
64      * Returns the print service.
65      *
66      * @return {@code PrintService} object
67      */
68     public PrintService getPrintService() {
69 
70         return (PrintService) getSource();
71     }
72 
73     /**
74      * Determine the printing service attributes that changed and their new
75      * values.
76      *
77      * @return attributes containing the new values for the service attributes
78      *         that changed. The returned set may be unmodifiable.
79      */
80     public PrintServiceAttributeSet getAttributes() {
81 
82         return attributes;
83     }
84 }
