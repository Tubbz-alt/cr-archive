25 package java.awt.peer;
26 
27 import java.awt.Menu;
28 import java.awt.MenuItem;
29 
30 /**
31  * The peer interface for menus. This is used by {@link Menu}.
32  *
33  * The peer interfaces are intended only for use in porting
34  * the AWT. They are not intended for use by application
35  * developers, and developers should not implement peers
36  * nor invoke any of the peer methods directly on the peer
37  * instances.
38  */
39 public interface MenuPeer extends MenuItemPeer {
40 
41     /**
42      * Adds a separator (e.g. a horizontal line or similar) to the menu.
43      *
44      * @see Menu#addSeparator()
45      */
46     void addSeparator();
47 
48     /**
49      * Adds the specified menu item to the menu.
50      *
51      * @param item the menu item to add
52      *
53      * @see Menu#add(MenuItem)
54      */
55     void addItem(MenuItem item);
56 
57     /**
58      * Removes the menu item at the specified index.
59      *
60      * @param index the index of the item to remove
61      *
62      * @see Menu#remove(int)
63      */
64     void delItem(int index);
65 }
