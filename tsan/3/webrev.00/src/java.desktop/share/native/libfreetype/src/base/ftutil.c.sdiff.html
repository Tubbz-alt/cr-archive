 50   ft_mem_alloc( FT_Memory  memory,
 51                 FT_Long    size,
 52                 FT_Error  *p_error )
 53   {
 54     FT_Error    error;
 55     FT_Pointer  block = ft_mem_qalloc( memory, size, &error );
 56 
 57     if ( !error && size > 0 )
 58       FT_MEM_ZERO( block, size );
 59 
 60     *p_error = error;
 61     return block;
