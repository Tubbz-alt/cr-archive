 68 #define AF_LATIN_BLUE_ACTIVE      ( 1U << 0 ) /* zone height is <= 3/4px   */

 70 #define AF_LATIN_BLUE_SUB_TOP     ( 1U << 2 ) /* we have a subscript top   */

144 #define AF_LATIN_HINTS_VERT_SNAP    ( 1U << 1 ) /* stem height snapping */
