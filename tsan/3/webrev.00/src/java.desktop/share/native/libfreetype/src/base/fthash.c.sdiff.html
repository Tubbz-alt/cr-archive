 50   static FT_ULong
 51   hash_str_lookup( FT_Hashkey*  key )
 52   {
 53     const char*  kp  = key->str;
 54     FT_ULong     res = 0;
 55 
 56 
 57     /* Mocklisp hash function. */
 58     while ( *kp )
 59       res = ( res << 5 ) - res + (FT_ULong)*kp++;
