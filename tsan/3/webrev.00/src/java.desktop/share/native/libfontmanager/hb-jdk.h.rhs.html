37 typedef struct JDKFontInfo_Struct {
38     JNIEnv* env;
39     jobject font2D;
40     jobject fontStrike;
41     long nativeFont;

42     float matrix[4];
43     float ptSize;
44     float xPtSize;
45     float yPtSize;
46     float devScale; // How much applying the full glyph tx scales x distance.
