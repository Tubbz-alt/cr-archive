105 /**
106  * hb_direction_t:
107  * @HB_DIRECTION_INVALID: Initial, unset direction.
108  * @HB_DIRECTION_LTR: Text is set horizontally from left to right.
109  * @HB_DIRECTION_RTL: Text is set horizontally from right to left.
110  * @HB_DIRECTION_TTB: Text is set vertically from top to bottom.
111  * @HB_DIRECTION_BTT: Text is set vertically from bottom to top.
112  */
113 typedef enum {
114   HB_DIRECTION_INVALID = 0,
115   HB_DIRECTION_LTR = 4,
116   HB_DIRECTION_RTL,
117   HB_DIRECTION_TTB,
118   HB_DIRECTION_BTT
119 } hb_direction_t;
120 
121 /* len=-1 means str is NUL-terminated */
122 HB_EXTERN hb_direction_t
123 hb_direction_from_string (const char *str, int len);
124 
125 HB_EXTERN const char *
126 hb_direction_to_string (hb_direction_t direction);
127 
128 #define HB_DIRECTION_IS_VALID(dir)      ((((unsigned int) (dir)) & ~3U) == 4)
129 /* Direction must be valid for the following */
130 #define HB_DIRECTION_IS_HORIZONTAL(dir) ((((unsigned int) (dir)) & ~1U) == 4)
131 #define HB_DIRECTION_IS_VERTICAL(dir)   ((((unsigned int) (dir)) & ~1U) == 6)
132 #define HB_DIRECTION_IS_FORWARD(dir)    ((((unsigned int) (dir)) & ~2U) == 4)
133 #define HB_DIRECTION_IS_BACKWARD(dir)   ((((unsigned int) (dir)) & ~2U) == 5)
134 #define HB_DIRECTION_REVERSE(dir)       ((hb_direction_t) (((unsigned int) (dir)) ^ 1))
