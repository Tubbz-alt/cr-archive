 37 struct hb_ot_shape_plan_t
 38 {
 39   hb_segment_properties_t props;
 40   const struct hb_ot_complex_shaper_t *shaper;
 41   hb_ot_map_t map;
 42   const void *data;
 43   hb_mask_t rtlm_mask, frac_mask, numr_mask, dnom_mask;
 44   hb_mask_t kern_mask;
 45   unsigned int has_frac : 1;
 46   unsigned int has_kern : 1;
 47   unsigned int has_mark : 1;
 48 
 49   inline void collect_lookups (hb_tag_t table_tag, hb_set_t *lookups) const
 50   {
 51     unsigned int table_index;
 52     switch (table_tag) {
 53       case HB_OT_TAG_GSUB: table_index = 0; break;
 54       case HB_OT_TAG_GPOS: table_index = 1; break;
 55       default: return;
 56     }
 57     map.collect_lookups (table_index, lookups);
 58   }
 59   inline void substitute (hb_font_t *font, hb_buffer_t *buffer) const { map.substitute (this, font, buffer); }
 60   inline void position (hb_font_t *font, hb_buffer_t *buffer) const { map.position (this, font, buffer); }
 61 
 62   void init (void)
 63   {
 64     memset (this, 0, sizeof (*this));
 65     map.init ();
 66   }
 67   void fini (void) {
 68     map.fini ();
 69   }
 70 };
 71 
 72 struct hb_ot_shape_planner_t
 73 {
 74   /* In the order that they are filled in. */
 75   hb_face_t *face;
 76   hb_segment_properties_t props;
 77   const struct hb_ot_complex_shaper_t *shaper;
 78   hb_ot_map_builder_t map;
 79 
 80   hb_ot_shape_planner_t (const hb_shape_plan_t *master_plan) :
 81                          face (master_plan->face_unsafe),
 82                          props (master_plan->props),
 83                          shaper (nullptr),
 84                          map (face, &props) {}
 85 
 86   inline void compile (hb_ot_shape_plan_t &plan,
 87                        const int          *coords,
 88                        unsigned int        num_coords)
 89   {
 90     plan.props = props;
 91     plan.shaper = shaper;
 92     map.compile (plan.map, coords, num_coords);
 93 
 94     plan.rtlm_mask = plan.map.get_1_mask (HB_TAG ('r','t','l','m'));
 95     plan.frac_mask = plan.map.get_1_mask (HB_TAG ('f','r','a','c'));
 96     plan.numr_mask = plan.map.get_1_mask (HB_TAG ('n','u','m','r'));
 97     plan.dnom_mask = plan.map.get_1_mask (HB_TAG ('d','n','o','m'));
 98 
 99     plan.kern_mask = plan.map.get_mask (HB_DIRECTION_IS_HORIZONTAL (plan.props.direction) ?
100                                         HB_TAG ('k','e','r','n') : HB_TAG ('v','k','r','n'));
101 
102     plan.has_frac = plan.frac_mask || (plan.numr_mask && plan.dnom_mask);
103     plan.has_kern = !!plan.kern_mask;
104     plan.has_mark = !!plan.map.get_1_mask (HB_TAG ('m','a','r','k'));
105   }
106 
107   private:
108   HB_DISALLOW_COPY_AND_ASSIGN (hb_ot_shape_planner_t);
109 };
