27 #include <stdio.h>

28 #include <sys/ldr.h>
29 #include <errno.h>

31 #include "porting_aix.h"

33 static unsigned char dladdr_buffer[0x4000];
34 
35 static void fill_dll_info(void) {
36   int rc = loadquery(L_GETINFO,dladdr_buffer, sizeof(dladdr_buffer));
37   if (rc == -1) {
38     fprintf(stderr, "loadquery failed (%d %s)", errno, strerror(errno));
39     fflush(stderr);
40   }
41 }
42 
43 static int dladdr_dont_reload(void* addr, Dl_info* info) {
44   const struct ld_info* p = (struct ld_info*) dladdr_buffer;
45   info->dli_fbase = 0; info->dli_fname = 0;
46   info->dli_sname = 0; info->dli_saddr = 0;
47   for (;;) {
48     if (addr >= p->ldinfo_textorg &&
49         addr < (((char*)p->ldinfo_textorg) + p->ldinfo_textsize)) {
50       info->dli_fname = p->ldinfo_filename;
51       info->dli_fbase = p->ldinfo_textorg;
52       return 1; /* [sic] */
53     }
54     if (!p->ldinfo_next) {
55       break;
56     }
57     p = (struct ld_info*)(((char*)p) + p->ldinfo_next);
58   }
59   return 0; /* [sic] */
60 }
61 
62 #ifdef __cplusplus
63 extern "C"
64 #endif
65 int dladdr(void *addr, Dl_info *info) {
66   static int loaded = 0;
67   if (!loaded) {
68     fill_dll_info();
69     loaded = 1;
70   }
71   if (!addr) {
72     return 0;  /* [sic] */
73   }
74   /* Address could be AIX function descriptor? */
75   void* const addr0 = *( (void**) addr );
76   int rc = dladdr_dont_reload(addr, info);
77   if (rc == 0) {
78     rc = dladdr_dont_reload(addr0, info);
79     if (rc == 0) { /* [sic] */
80       fill_dll_info(); /* refill, maybe loadquery info is outdated */
81       rc = dladdr_dont_reload(addr, info);
82       if (rc == 0) {
83         rc = dladdr_dont_reload(addr0, info);
84       }
85     }
86   }
87   return rc;
88 }
