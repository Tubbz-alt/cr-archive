26 package sun.awt;
27 
28 import java.awt.GraphicsConfiguration;
29 import java.awt.Rectangle;
30 import java.awt.Transparency;
31 import java.awt.geom.AffineTransform;
32 import java.awt.image.ColorModel;
33 
34 import sun.java2d.SurfaceData;
35 import sun.java2d.opengl.CGLLayer;
36 import sun.lwawt.LWGraphicsConfig;
37 
38 public abstract class CGraphicsConfig extends GraphicsConfiguration
39         implements LWGraphicsConfig {
40 
41     private final CGraphicsDevice device;
42     private ColorModel colorModel;
43 
44     protected CGraphicsConfig(CGraphicsDevice device) {
45         this.device = device;
46     }
47 
48     @Override
49     public final Rectangle getBounds() {
50         return device.getBounds();
51     }
52 
53     @Override
54     public ColorModel getColorModel() {
55         if (colorModel == null) {
56             colorModel = getColorModel(Transparency.OPAQUE);
57         }
58         return colorModel;
59     }
60 
61     @Override
62     public AffineTransform getDefaultTransform() {
63         double scaleFactor = device.getScaleFactor();
64         return AffineTransform.getScaleInstance(scaleFactor, scaleFactor);
65     }
66 
67     @Override
68     public CGraphicsDevice getDevice() {
69         return device;
70     }
71 
72     @Override
73     public AffineTransform getNormalizingTransform() {
74         double xscale = device.getXResolution() / 72.0;
75         double yscale = device.getYResolution() / 72.0;
76         return new AffineTransform(xscale, 0.0, 0.0, yscale, 0.0, 0.0);
77     }
78 
79     /**
80      * Creates a new SurfaceData that will be associated with the given
81      * CGLLayer.
82      */
83     public abstract SurfaceData createSurfaceData(CGLLayer layer);
84 
85     @Override
86     public final boolean isTranslucencyCapable() {
87         //we know for sure we have capable config :)
88         return true;
89     }
90 }
