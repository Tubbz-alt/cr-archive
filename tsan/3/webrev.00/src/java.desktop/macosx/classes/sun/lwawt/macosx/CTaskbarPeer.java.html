26 package sun.lwawt.macosx;
27 
28 import java.awt.Image;
29 import java.awt.PopupMenu;
30 import java.awt.Taskbar.Feature;
31 import java.awt.peer.TaskbarPeer;
32 
33 import apple.laf.JRSUIUtils;
34 import com.apple.eawt.Application;
35 
36 final public class CTaskbarPeer implements TaskbarPeer {
37 
38     CTaskbarPeer() {}
39 
40     @Override
41     public boolean isSupported(Feature feature) {
42         switch(feature) {
43             case ICON_BADGE_TEXT:
44             case ICON_BADGE_NUMBER:
45                 return JRSUIUtils.TaskBar.isIconBadgeSupported();
46             case ICON_IMAGE:
47             case MENU:
48             case PROGRESS_VALUE:
49             case USER_ATTENTION:
50                 return true;
51             default:
52                 return false;
53         }
54     }
55 
56     @Override
57     public void setProgressValue(int value) {
58         Application.getApplication().setDockIconProgress(value);
59     }
60 
61     @Override
62     public void setIconBadge(String badge) {
63         Application.getApplication().setDockIconBadge(badge);
64     }
65 
66     @Override
67     public Image getIconImage() {
68         return Application.getApplication().getDockIconImage();
69     }
70 
71     @Override
72     public void setIconImage(Image image) {
73         Application.getApplication().setDockIconImage(image);
74     }
75 
76     @Override
77     public PopupMenu getMenu() {
78         return Application.getApplication().getDockMenu();
79     }
80 
81     @Override
82     public void setMenu(PopupMenu menu) {
83         Application.getApplication().setDockMenu(menu);
84     }
85 
86     @Override
87     public void requestUserAttention(boolean enabled, boolean critical) {
88         if (enabled) {
89             Application.getApplication().requestUserAttention(critical);
90         }
91     }
92 }
