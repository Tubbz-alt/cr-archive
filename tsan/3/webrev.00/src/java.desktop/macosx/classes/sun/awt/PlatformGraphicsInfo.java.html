26 package sun.awt;
27 
28 import java.awt.GraphicsEnvironment;
29 import java.awt.Toolkit;
30 import java.security.AccessController;
31 import java.security.PrivilegedAction;
32 
33 public class PlatformGraphicsInfo {
34 
35     static {
36         AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
37             System.loadLibrary("awt");
38             return null;
39         });
40     }
41 
42     public static GraphicsEnvironment createGE() {
43         return new CGraphicsEnvironment();
44     }
45 
46     public static Toolkit createToolkit() {
47         return new sun.lwawt.macosx.LWCToolkit();
48     }
49 
50     /**
51      * Returns true if the WindowServer is available, false otherwise.
52      *
53      * @return true if the WindowServer is available, false otherwise
54      */
55     public static native boolean isInAquaSession();
56 
57     public static boolean getDefaultHeadlessProperty() {
58          return !isInAquaSession();
59     }
60 
61     /*
62      * Called from java.awt.GraphicsEnvironment when
63      * getDefaultHeadlessProperty() has returned true, and
64      * the application has called an API that requires headful.
65      */
66     public static String getDefaultHeadlessMessage() {
67         return
68             "\nThe application is not running in a desktop session,\n" +
69             "but this program performed an operation which requires it.";
70     }
71 
72 }
