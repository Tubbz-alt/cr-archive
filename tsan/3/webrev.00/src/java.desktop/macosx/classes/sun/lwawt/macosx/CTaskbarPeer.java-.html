26 package sun.lwawt.macosx;
27 
28 import com.apple.eawt.Application;
29 import java.awt.Image;
30 import java.awt.PopupMenu;
31 import java.awt.Taskbar.Feature;
32 import java.awt.peer.TaskbarPeer;
33 
34 final public class CTaskbarPeer implements TaskbarPeer {
35 
36     CTaskbarPeer() {}
37 
38     @Override
39     public boolean isSupported(Feature feature) {
40         switch(feature) {
41             case ICON_BADGE_TEXT:
42             case ICON_BADGE_NUMBER:
43             case ICON_IMAGE:
44             case MENU:
45             case PROGRESS_VALUE:
46             case USER_ATTENTION:
47                 return true;
48             default:
49                 return false;
50         }
51     }
52 
53     @Override
54     public void setProgressValue(int value) {
55         Application.getApplication().setDockIconProgress(value);
56     }
57 
58     @Override
59     public void setIconBadge(String badge) {
60         Application.getApplication().setDockIconBadge(badge);
61     }
62 
63     @Override
64     public Image getIconImage() {
65         return Application.getApplication().getDockIconImage();
66     }
67 
68     @Override
69     public void setIconImage(Image image) {
70         Application.getApplication().setDockIconImage(image);
71     }
72 
73     @Override
74     public PopupMenu getMenu() {
75         return Application.getApplication().getDockMenu();
76     }
77 
78     @Override
79     public void setMenu(PopupMenu menu) {
80         Application.getApplication().setDockMenu(menu);
81     }
82 
83     @Override
84     public void requestUserAttention(boolean enabled, boolean critical) {
85         if (enabled) {
86             Application.getApplication().requestUserAttention(critical);
87         }
88     }
89 }
