 26 package sun.java2d.opengl;
 27 









 28 import java.awt.GraphicsConfiguration;
 29 import java.awt.Rectangle;
 30 import java.awt.Transparency;
 31 
 32 import sun.awt.CGraphicsConfig;
 33 import sun.java2d.NullSurfaceData;
 34 import sun.java2d.SurfaceData;
 35 import sun.lwawt.LWWindowPeer;
 36 import sun.lwawt.macosx.CFRetainedResource;
 37 
 38 public class CGLLayer extends CFRetainedResource {
 39 
 40     private native long nativeCreateLayer();
 41     private static native void nativeSetScale(long layerPtr, double scale);
 42     private static native void validate(long layerPtr, CGLSurfaceData cglsd);
 43     private static native void blitTexture(long layerPtr);
 44 
 45     private LWWindowPeer peer;
 46     private int scale = 1;
 47 
 48     private SurfaceData surfaceData; // represents intermediate buffer (texture)
 49 
 50     public CGLLayer(LWWindowPeer peer) {
 51         super(0, true);
 52 
 53         setPtr(nativeCreateLayer());
 54         this.peer = peer;
 55     }
 56 
 57     public long getPointer() {
 58         return ptr;
 59     }
 60 
 61     public Rectangle getBounds() {
 62         return peer.getBounds();
 63     }
 64 
 65     public GraphicsConfiguration getGraphicsConfiguration() {
 66         return peer.getGraphicsConfiguration();
 67     }
 68 
 69     public boolean isOpaque() {
 70         return !peer.isTranslucent();
 71     }
 72 
 73     public int getTransparency() {
 74         return isOpaque() ? Transparency.OPAQUE : Transparency.TRANSLUCENT;
 75     }
 76 
 77     public Object getDestination() {
 78         return peer.getTarget();
 79     }
 80 
 81     public SurfaceData replaceSurfaceData() {
 82         if (getBounds().isEmpty()) {
 83             surfaceData = NullSurfaceData.theInstance;
 84             return surfaceData;
 85         }
 86 
 87         // the layer redirects all painting to the buffer's graphics
 88         // and blits the buffer to the layer surface (in drawInCGLContext callback)
 89         CGraphicsConfig gc = (CGraphicsConfig)getGraphicsConfiguration();
 90         surfaceData = gc.createSurfaceData(this);
 91         setScale(gc.getDevice().getScaleFactor());
 92         // the layer holds a reference to the buffer, which in
 93         // turn has a reference back to this layer
 94         if (surfaceData instanceof CGLSurfaceData) {
 95             validate((CGLSurfaceData)surfaceData);
 96         }
 97 
 98         return surfaceData;
 99     }
100 
101     public SurfaceData getSurfaceData() {
102         return surfaceData;
103     }
104 
105     public void validate(final CGLSurfaceData cglsd) {
106         OGLRenderQueue rq = OGLRenderQueue.getInstance();
107         rq.lock();
108         try {
109             execute(ptr -> validate(ptr, cglsd));
110         } finally {
111             rq.unlock();
112         }
113     }
114 
115     @Override
116     public void dispose() {
117         // break the connection between the layer and the buffer
118         validate(null);
119         SurfaceData oldData = surfaceData;
120         surfaceData = NullSurfaceData.theInstance;;
121         if (oldData != null) {
122             oldData.flush();
123         }
124         super.dispose();
125     }
126 
127     private void setScale(final int _scale) {
128         if (scale != _scale) {
129             scale = _scale;
130             execute(ptr -> nativeSetScale(ptr, scale));
131         }
132     }
133 
134     // ----------------------------------------------------------------------
135     // NATIVE CALLBACKS
136     // ----------------------------------------------------------------------
137 
138     private void drawInCGLContext() {
139         // tell the flusher thread not to update the intermediate buffer
140         // until we are done blitting from it
141         OGLRenderQueue rq = OGLRenderQueue.getInstance();
142         rq.lock();
143         try {
144             execute(ptr -> blitTexture(ptr));
145         } finally {
146             rq.unlock();
147         }
148     }
149 }
