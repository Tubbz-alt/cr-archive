26 package sun.awt;
27 
28 import java.awt.GraphicsConfiguration;
29 import java.awt.Rectangle;
30 import java.awt.Transparency;
31 import java.awt.geom.AffineTransform;
32 import java.awt.image.ColorModel;
33 
34 import sun.java2d.SurfaceData;
35 import sun.java2d.opengl.CGLLayer;
36 import sun.lwawt.LWGraphicsConfig;
37 import sun.lwawt.macosx.CPlatformView;
38 
39 public abstract class CGraphicsConfig extends GraphicsConfiguration
40         implements LWGraphicsConfig {
41 
42     private final CGraphicsDevice device;
43     private ColorModel colorModel;
44 
45     protected CGraphicsConfig(CGraphicsDevice device) {
46         this.device = device;
47     }
48 
49     @Override
50     public final Rectangle getBounds() {
51         return device.getBounds();
52     }
53 
54     @Override
55     public ColorModel getColorModel() {
56         if (colorModel == null) {
57             colorModel = getColorModel(Transparency.OPAQUE);
58         }
59         return colorModel;
60     }
61 
62     @Override
63     public AffineTransform getDefaultTransform() {
64         double scaleFactor = device.getScaleFactor();
65         return AffineTransform.getScaleInstance(scaleFactor, scaleFactor);
66     }
67 
68     @Override
69     public CGraphicsDevice getDevice() {
70         return device;
71     }
72 
73     @Override
74     public AffineTransform getNormalizingTransform() {
75         double xscale = device.getXResolution() / 72.0;
76         double yscale = device.getYResolution() / 72.0;
77         return new AffineTransform(xscale, 0.0, 0.0, yscale, 0.0, 0.0);
78     }
79 
80     /**
81      * Creates a new SurfaceData that will be associated with the given
82      * LWWindowPeer.
83      */
84     public abstract SurfaceData createSurfaceData(CPlatformView pView);
85 
86     /**
87      * Creates a new SurfaceData that will be associated with the given
88      * CGLLayer.
89      */
90     public abstract SurfaceData createSurfaceData(CGLLayer layer);
91 
92     @Override
93     public final boolean isTranslucencyCapable() {
94         //we know for sure we have capable config :)
95         return true;
96     }
97 }
