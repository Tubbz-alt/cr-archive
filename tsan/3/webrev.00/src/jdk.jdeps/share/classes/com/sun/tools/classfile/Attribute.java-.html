 82         public Attribute createAttribute(ClassReader cr, int name_index, byte[] data)
 83                 throws IOException {
 84             if (standardAttributes == null) {
 85                 init();
 86             }
 87 
 88             ConstantPool cp = cr.getConstantPool();
 89             String reasonForDefaultAttr;
 90             try {
 91                 String name = cp.getUTF8Value(name_index);
 92                 Class<? extends Attribute> attrClass = standardAttributes.get(name);
 93                 if (attrClass != null) {
 94                     try {
 95                         Class<?>[] constrArgTypes = {ClassReader.class, int.class, int.class};
 96                         Constructor<? extends Attribute> constr = attrClass.getDeclaredConstructor(constrArgTypes);
 97                         return constr.newInstance(cr, name_index, data.length);
 98                     } catch (Throwable t) {
 99                         reasonForDefaultAttr = t.toString();
100                         // fall through and use DefaultAttribute
101                         // t.printStackTrace();
102                     }
103                 } else {
104                     reasonForDefaultAttr = "unknown attribute";
105                 }
106             } catch (ConstantPoolException e) {
107                 reasonForDefaultAttr = e.toString();
108                 // fall through and use DefaultAttribute
109             }
110             return new DefaultAttribute(cr, name_index, data, reasonForDefaultAttr);
111         }

155     public static Attribute read(ClassReader cr) throws IOException {
156         return cr.readAttribute();
157     }
158 
159     protected Attribute(int name_index, int length) {
160         attribute_name_index = name_index;
161         attribute_length = length;
162     }
163 
164     public String getName(ConstantPool constant_pool) throws ConstantPoolException {
165         return constant_pool.getUTF8Value(attribute_name_index);
166     }

168     public abstract <R,D> R accept(Attribute.Visitor<R,D> visitor, D data);
169 
170     public int byteLength() {
171         return 6 + attribute_length;
172     }
