26 package com.sun.tools.classfile;
27 
28 import java.io.IOException;
29 import com.sun.tools.classfile.Attribute.Visitor;
30 
31 /**
32  *  <p><b>This is NOT part of any supported API.
33  *  If you write code that depends on this, you do so at your own risk.
34  *  This code and its internal interfaces are subject to change or
35  *  deletion without notice.</b>
36  */
37 public class Record_attribute extends Attribute {
38     Record_attribute(ClassReader cr, int name_index, int length) throws IOException {
39         super(name_index, length);
40         component_count = cr.readUnsignedShort();
41         component_info_arr = new ComponentInfo[component_count];
42         for (int i = 0; i < component_count; i++) {
43             component_info_arr[i] = new ComponentInfo(cr);
44         }
45     }
46 
47     @Override
48     public <R, D> R accept(Visitor<R, D> visitor, D data) {
49         return visitor.visitRecord(this, data);
50     }
51 
52     public final int component_count;
53     public final ComponentInfo[] component_info_arr;
54 
55     public static class ComponentInfo {
56         ComponentInfo(ClassReader cr) throws IOException {
57             name_index = cr.readUnsignedShort();
58             descriptor = new Descriptor(cr);
59             attributes = new Attributes(cr);
60         }
61 
62         public String getName(ConstantPool constant_pool) throws ConstantPoolException {
63             return constant_pool.getUTF8Value(name_index);
64         }
65 
66         public final int name_index;
67         public final Descriptor descriptor;
68         public final Attributes attributes;
69     }
70 }
