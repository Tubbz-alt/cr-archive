25 #ifndef _PROC_SERVICE_H_
26 #define _PROC_SERVICE_H_
27 
28 #include <stdio.h>
29 #include <sys/procfs.h>
30 #include "jni.h"
31 #include "libproc.h"
32 
33 
34 // copied from Solaris "proc_service.h"
35 typedef enum {
36         PS_OK,          /* generic "call succeeded" */
37         PS_ERR,         /* generic error */
38         PS_BADPID,      /* bad process handle */
39         PS_BADLID,      /* bad lwp identifier */
40         PS_BADADDR,     /* bad address */
41         PS_NOSYM,       /* p_lookup() could not find given symbol */
42         PS_NOFREGS      /* FPU register set not available for given lwp */
43 } ps_err_e;
44 
45 #ifdef __cplusplus
46 extern "C" {
47 #endif
48 
49 // ps_getpid() is only defined on Linux to return a thread's process ID
50 JNIEXPORT pid_t JNICALL
51 ps_getpid(struct ps_prochandle *ph);
52 
53 // ps_pglobal_lookup() looks up the symbol sym_name in the symbol table
54 // of the load object object_name in the target process identified by ph.
55 // It returns the symbol's value as an address in the target process in
56 // *sym_addr.
57 
58 JNIEXPORT ps_err_e JNICALL
59 ps_pglobal_lookup(struct ps_prochandle *ph, const char *object_name,
60                     const char *sym_name, psaddr_t *sym_addr);
61 
62 // read "size" bytes of data from debuggee at address "addr"
63 JNIEXPORT ps_err_e JNICALL
64 ps_pdread(struct ps_prochandle *ph, psaddr_t  addr,
65                    void *buf, size_t size);
66 
67 // write "size" bytes of data to debuggee at address "addr"
68 JNIEXPORT ps_err_e JNICALL
69 ps_pdwrite(struct ps_prochandle *ph, psaddr_t addr,
70                     const void *buf, size_t size);
71 
72 JNIEXPORT ps_err_e JNICALL
73 ps_lsetfpregs(struct ps_prochandle *ph, lwpid_t lid, const prfpregset_t *fpregs);
74 
75 JNIEXPORT ps_err_e JNICALL
76 ps_lsetregs(struct ps_prochandle *ph, lwpid_t lid, const prgregset_t gregset);
77 
78 JNIEXPORT ps_err_e JNICALL
79 ps_lgetfpregs(struct ps_prochandle *ph, lwpid_t lid, prfpregset_t *fpregs);
80 
81 JNIEXPORT ps_err_e JNICALL
82 ps_lgetregs(struct ps_prochandle *ph, lwpid_t lid, prgregset_t gregset);
83 
84 #ifdef __cplusplus
85 }
86 #endif
87 
88 #endif /* _PROC_SERVICE_H_ */
