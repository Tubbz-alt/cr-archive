 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.util.*;
 28 
 29 import sun.jvm.hotspot.oops.*;
 30 import sun.jvm.hotspot.utilities.*;
 31 import sun.jvm.hotspot.debugger.*;
 32 import sun.jvm.hotspot.types.*;
 33 
 34 public class ObjectSynchronizer {
 35   static {
 36     VM.registerVMInitializedObserver(new Observer() {
 37         public void update(Observable o, Object data) {
 38           initialize(VM.getVM().getTypeDataBase());
 39         }
 40       });
 41   }
 42 
 43   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 44     Type type;
 45     try {
 46       type = db.lookupType("ObjectSynchronizer");
 47       gBlockList = type.getAddressField("gBlockList").getValue();
 48       blockSize = db.lookupIntConstant("ObjectSynchronizer::_BLOCKSIZE").intValue();
 49       defaultCacheLineSize = db.lookupIntConstant("DEFAULT_CACHE_LINE_SIZE").intValue();
 50     } catch (RuntimeException e) { }
 51     type = db.lookupType("ObjectMonitor");
 52     objectMonitorTypeSize = type.getSize();
 53     if ((objectMonitorTypeSize % defaultCacheLineSize) != 0) {
 54       // sizeof(ObjectMonitor) is not already a multiple of a cache line.
 55       // The ObjectMonitor allocation code in ObjectSynchronizer pads each
 56       // ObjectMonitor in a block to the next cache line boundary.
 57       int needLines = ((int)objectMonitorTypeSize / defaultCacheLineSize) + 1;
 58       objectMonitorTypeSize = needLines * defaultCacheLineSize;
 59     }
 60   }
 61 
 62   public long identityHashValueFor(Oop obj) {
 63     Mark mark = obj.getMark();
 64     if (mark.isUnlocked()) {
 65       // FIXME: can not generate marks in debugging system
 66       return mark.hash();
 67     } else if (mark.hasMonitor()) {
 68       ObjectMonitor monitor = mark.monitor();
 69       Mark temp = monitor.header();
 70       return temp.hash();
 71     } else {
 72       if (Assert.ASSERTS_ENABLED) {
 73         Assert.that(VM.getVM().isDebugging(), "Can not access displaced header otherwise");
 74       }
 75       if (mark.hasDisplacedMarkHelper()) {
 76         Mark temp = mark.displacedMarkHelper();
 77         return temp.hash();
 78       }
 79       // FIXME: can not do anything else here in debugging system
 80       return 0;
 81     }
 82   }
 83 
 84   public static Iterator objectMonitorIterator() {
 85     if (gBlockList != null) {
 86       return new ObjectMonitorIterator();
 87     } else {
 88       return null;
 89     }
 90   }
 91 
 92   private static class ObjectMonitorIterator implements Iterator {
 93 
 94     // JVMTI raw monitors are not pointed by gBlockList
 95     // and are not included by this Iterator. May add them later.
 96 
 97     ObjectMonitorIterator() {
 98       blockAddr = gBlockList;
 99       index = blockSize - 1;
100       block = new ObjectMonitor(blockAddr);
101     }
102 
103     public boolean hasNext() {
104       return (index > 0 || block.freeNext() != null);
105     }
106 
107     public Object next() {
108       Address addr;
109       if (index == 0) {
110         // advance to next block
111         blockAddr = block.freeNext();
112         if (blockAddr == null) {
113           throw new NoSuchElementException();
114         }
115         block = new ObjectMonitor(blockAddr);
116         index = blockSize - 1;
117       }
118       addr = blockAddr.addOffsetTo(index*objectMonitorTypeSize);
119       index --;
120       return new ObjectMonitor(addr);
121     }
122 
123     public void remove() {
124       throw new UnsupportedOperationException();
125     }
126 
127     private ObjectMonitor block;
128     private int index;
129     private Address blockAddr;
130   }
131 
132   private static Address gBlockList;
133   private static int blockSize;
134   private static int defaultCacheLineSize;
135   private static long objectMonitorTypeSize;
136 
137 }
