25 package sun.jvm.hotspot.gc.cms;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 
33 public class CMSCollector extends VMObject {
34   private static long markBitMapFieldOffset;
35 
36   public CMSCollector(Address addr) {
37     super(addr);
38   }
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static synchronized void initialize(TypeDataBase db) {
49     Type type = db.lookupType("CMSCollector");
50     markBitMapFieldOffset = type.getField("_markBitMap").getOffset();
51   }
52 
53   //Accessing mark bitmap
54   public CMSBitMap markBitMap() {
55    return (CMSBitMap) VMObjectFactory.newObject(
56                                 CMSBitMap.class,
57                                 addr.addOffsetTo(markBitMapFieldOffset));
58   }
59 
60   public long blockSizeUsingPrintezisBits(Address addr) {
61     CMSBitMap markBitMap = markBitMap();
62     long addressSize = VM.getVM().getAddressSize();
63     if ( markBitMap.isMarked(addr) &&  markBitMap.isMarked(addr.addOffsetTo(1*addressSize)) ) {
64       Address nextOneAddr = markBitMap.getNextMarkedWordAddress(addr.addOffsetTo(2*addressSize));
65       //return size in bytes
66       long size =  (nextOneAddr.addOffsetTo(1*addressSize)).minus(addr);
67       return size;
68     } else {
69       //missing Printezis marks
70       return -1;
71     }
72 
73   }
74 }
