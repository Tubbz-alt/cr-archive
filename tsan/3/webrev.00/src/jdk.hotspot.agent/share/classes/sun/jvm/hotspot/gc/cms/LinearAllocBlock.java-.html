26 package sun.jvm.hotspot.gc.cms;
27 
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.runtime.*;
32 
33 public class LinearAllocBlock extends VMObject {
34    static {
35       VM.registerVMInitializedObserver(new Observer() {
36          public void update(Observable o, Object data) {
37             initialize(VM.getVM().getTypeDataBase());
38          }
39       });
40    }
41 
42    private static synchronized void initialize(TypeDataBase db) {
43       Type type = db.lookupType("LinearAllocBlock");
44       word_sizeField= type.getCIntegerField("_word_size");
45    }
46 
47    // Fields
48    private static CIntegerField word_sizeField;
49 
50    // Accessors
51    public long word_size() {
52       return word_sizeField.getValue(addr);
53    }
54 
55    // Constructor
56    public LinearAllocBlock(Address addr) {
57       super(addr);
58    }
59 }
