 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.util.*;
 28 
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.oops.*;
 31 import sun.jvm.hotspot.types.*;
 32 
 33 public class ObjectMonitor extends VMObject {
 34   static {
 35     VM.registerVMInitializedObserver(new Observer() {
 36         public void update(Observable o, Object data) {
 37           initialize(VM.getVM().getTypeDataBase());
 38         }
 39       });
 40   }
 41 
 42   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 43     heap = VM.getVM().getObjectHeap();
 44     Type type  = db.lookupType("ObjectMonitor");
 45     sun.jvm.hotspot.types.Field f = type.getField("_header");
 46     headerFieldOffset = f.getOffset();
 47     f = type.getField("_object");
 48     objectFieldOffset = f.getOffset();
 49     f = type.getField("_owner");
 50     ownerFieldOffset = f.getOffset();
 51     f = type.getField("_next_om");
 52     nextOMFieldOffset = f.getOffset();
 53     contentionsField  = type.getJIntField("_contentions");
 54     waitersField = type.getJIntField("_waiters");
 55     recursionsField = type.getCIntegerField("_recursions");
 56   }
 57 
 58   public ObjectMonitor(Address addr) {
 59     super(addr);
 60   }
 61 
 62   public Mark header() {
 63     return new Mark(addr.addOffsetTo(headerFieldOffset));
 64   }
 65 
 66   // FIXME
 67   //  void      set_header(markWord hdr);
 68 
 69   // FIXME: must implement and delegate to platform-dependent implementation
 70   //  public boolean isBusy();
 71   public boolean isEntered(sun.jvm.hotspot.runtime.Thread current) {
 72     Address o = owner();
 73     if (current.threadObjectAddress().equals(o) ||
 74         current.isLockOwned(o)) {
 75       return true;
 76     }
 77     return false;
 78   }
 79 
 80   public Address owner() { return addr.getAddressAt(ownerFieldOffset); }
 81   // FIXME
 82   //  void      set_owner(void* owner);
 83 
 84   public int    waiters() { return waitersField.getValue(addr); }
 85 
 86   public Address nextOM() { return addr.getAddressAt(nextOMFieldOffset); }
 87   // FIXME
 88   //  void      set_queue(void* owner);
 89 




 90   public long recursions() { return recursionsField.getValue(addr); }
 91 
 92   public OopHandle object() {
 93     return addr.getOopHandleAt(objectFieldOffset);
 94   }
 95 

 96   public int contentions() {
 97       return contentionsField.getValue(addr);
 98   }
 99 
100   // FIXME
101   //  void*     object_addr();
102   //  void      set_object(void* obj);
103 
104   // The following four either aren't expressed as typed fields in
105   // vmStructs.cpp because they aren't strongly typed in the VM, or
106   // would confuse the SA's type system.
107   private static ObjectHeap    heap;
108   private static long          headerFieldOffset;
109   private static long          objectFieldOffset;
110   private static long          ownerFieldOffset;
111   private static long          nextOMFieldOffset;
112   private static JIntField     contentionsField;
113   private static JIntField     waitersField;
114   private static CIntegerField recursionsField;
115   // FIXME: expose platform-dependent stuff
116 }
