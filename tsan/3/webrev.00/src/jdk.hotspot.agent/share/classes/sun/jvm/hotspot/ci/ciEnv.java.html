 25 package sun.jvm.hotspot.ci;
 26 
 27 import java.io.PrintStream;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.oops.*;
 32 import sun.jvm.hotspot.opto.*;
 33 import sun.jvm.hotspot.compiler.CompileTask;
 34 import sun.jvm.hotspot.prims.JvmtiExport;
 35 import sun.jvm.hotspot.types.*;
 36 import sun.jvm.hotspot.utilities.GrowableArray;
 37 
 38 public class ciEnv extends VMObject {
 39   static {
 40     VM.registerVMInitializedObserver(new Observer() {
 41         public void update(Observable o, Object data) {
 42           initialize(VM.getVM().getTypeDataBase());
 43         }
 44       });
 45   }
 46 
 47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 48     Type type      = db.lookupType("ciEnv");
 49     dependenciesField = type.getAddressField("_dependencies");
 50     factoryField = type.getAddressField("_factory");
 51     compilerDataField = type.getAddressField("_compiler_data");
 52     taskField = type.getAddressField("_task");
 53   }
 54 
 55   private static AddressField dependenciesField;
 56   private static AddressField factoryField;
 57   private static AddressField compilerDataField;
 58   private static AddressField taskField;
 59 
 60   public ciEnv(Address addr) {
 61     super(addr);
 62   }
 63 
 64   public Compile compilerData() {
 65     Address addr = compilerDataField.getValue(this.getAddress());
 66     if (addr == null) {
 67       return null;
 68     }
 69     return new Compile(addr);
 70   }
 71 
 72   public ciObjectFactory factory() {
 73     return new ciObjectFactory(factoryField.getValue(this.getAddress()));
 74   }
 75 
 76   public CompileTask task() {
 77     return new CompileTask(taskField.getValue(this.getAddress()));
 78   }
 79 
 80   public void dumpReplayData(PrintStream out) {
 81     out.println("JvmtiExport can_access_local_variables " +
 82                 (JvmtiExport.canAccessLocalVariables() ? '1' : '0'));
 83     out.println("JvmtiExport can_hotswap_or_post_breakpoint " +
 84                 (JvmtiExport.canHotswapOrPostBreakpoint() ? '1' : '0'));
 85     out.println("JvmtiExport can_post_on_exceptions " +
 86                 (JvmtiExport.canPostOnExceptions() ? '1' : '0'));
 87 
 88     GrowableArray<ciMetadata> objects = factory().objects();
 89     out.println("# " + objects.length() + " ciObject found");
 90     for (int i = 0; i < objects.length(); i++) {
 91       ciMetadata o = objects.at(i);
 92       out.println("# ciMetadata" + i + " @ " + o);
 93       o.dumpReplayData(out);
 94     }
 95     CompileTask task = task();
 96     Method method = task.method();
 97     int entryBci = task.osrBci();
 98     int compLevel = task.compLevel();
 99     out.print("compile " + method.nameAsAscii() + " " +
100               entryBci + " " + compLevel);
101     Compile compiler = compilerData();
102     if (compiler != null) {
103       // Dump inlining data.
104       compiler.dumpInlineData(out);
105     }
106     out.println();
107   }
108 }
