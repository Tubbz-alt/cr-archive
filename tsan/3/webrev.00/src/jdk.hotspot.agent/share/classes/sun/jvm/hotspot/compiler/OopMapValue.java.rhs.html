 25 package sun.jvm.hotspot.compiler;
 26 
 27 import java.util.*;
 28 
 29 import sun.jvm.hotspot.code.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;
 32 import sun.jvm.hotspot.utilities.*;
 33 
 34 public class OopMapValue {
 35   private short value;
 36   private short contentReg;
 37 
 38   /** Read from target VM; located in compiler/oopMap.hpp */
 39   // How bits are organized
 40   static int TYPE_BITS;
 41   static int REGISTER_BITS;
 42   static int TYPE_SHIFT;
 43   static int REGISTER_SHIFT;
 44   static int TYPE_MASK;
 45   static int TYPE_MASK_IN_PLACE;
 46   static int REGISTER_MASK;
 47   static int REGISTER_MASK_IN_PLACE;
 48 
 49   // Types of OopValues

 50   static int OOP_VALUE;
 51   static int NARROWOOP_VALUE;
 52   static int CALLEE_SAVED_VALUE;
 53   static int DERIVED_OOP_VALUE;
 54 
 55   static {
 56     VM.registerVMInitializedObserver(new Observer() {
 57         public void update(Observable o, Object data) {
 58           initialize(VM.getVM().getTypeDataBase());
 59         }
 60       });
 61   }
 62 
 63   private static void initialize(TypeDataBase db) {
 64     TYPE_BITS              = db.lookupIntConstant("OopMapValue::type_bits").intValue();
 65     REGISTER_BITS          = db.lookupIntConstant("OopMapValue::register_bits").intValue();
 66     TYPE_SHIFT             = db.lookupIntConstant("OopMapValue::type_shift").intValue();
 67     REGISTER_SHIFT         = db.lookupIntConstant("OopMapValue::register_shift").intValue();
 68     TYPE_MASK              = db.lookupIntConstant("OopMapValue::type_mask").intValue();
 69     TYPE_MASK_IN_PLACE     = db.lookupIntConstant("OopMapValue::type_mask_in_place").intValue();
 70     REGISTER_MASK          = db.lookupIntConstant("OopMapValue::register_mask").intValue();
 71     REGISTER_MASK_IN_PLACE = db.lookupIntConstant("OopMapValue::register_mask_in_place").intValue();

 72     OOP_VALUE              = db.lookupIntConstant("OopMapValue::oop_value").intValue();
 73     NARROWOOP_VALUE        = db.lookupIntConstant("OopMapValue::narrowoop_value").intValue();
 74     CALLEE_SAVED_VALUE     = db.lookupIntConstant("OopMapValue::callee_saved_value").intValue();
 75     DERIVED_OOP_VALUE      = db.lookupIntConstant("OopMapValue::derived_oop_value").intValue();
 76   }
 77 
 78   public static abstract class OopTypes {

 79     public static final OopTypes OOP_VALUE          = new OopTypes() { int getValue() { return OopMapValue.OOP_VALUE;          }};
 80     public static final OopTypes NARROWOOP_VALUE    = new OopTypes() { int getValue() { return OopMapValue.NARROWOOP_VALUE;         }};
 81     public static final OopTypes CALLEE_SAVED_VALUE = new OopTypes() { int getValue() { return OopMapValue.CALLEE_SAVED_VALUE; }};
 82     public static final OopTypes DERIVED_OOP_VALUE  = new OopTypes() { int getValue() { return OopMapValue.DERIVED_OOP_VALUE;  }};
 83 
 84     abstract int getValue();
 85     protected OopTypes() {}
 86   }
 87 
 88   public OopMapValue()                                  { setValue((short) 0); setContentReg(new VMReg(0)); }
 89   public OopMapValue(VMReg reg, OopTypes t)             { setReg(reg); setType(t);                      }
 90   public OopMapValue(VMReg reg, OopTypes t, VMReg reg2) { setReg(reg); setType(t); setContentReg(reg2); }
 91   public OopMapValue(CompressedReadStream stream)       { readFrom(stream);                             }
 92 
 93   public void readFrom(CompressedReadStream stream) {
 94     setValue((short) stream.readInt());
 95     if (isCalleeSaved() || isDerivedOop()) {
 96       setContentReg(new VMReg(stream.readInt()));
 97     }
 98   }
 99 
100   // Querying
101   public boolean isOop()         { return (getValue() & TYPE_MASK_IN_PLACE) == OOP_VALUE;          }
102   public boolean isNarrowOop()   { return (getValue() & TYPE_MASK_IN_PLACE) == NARROWOOP_VALUE;    }
103   public boolean isCalleeSaved() { return (getValue() & TYPE_MASK_IN_PLACE) == CALLEE_SAVED_VALUE; }
104   public boolean isDerivedOop()  { return (getValue() & TYPE_MASK_IN_PLACE) == DERIVED_OOP_VALUE;  }
105 
106   public VMReg getReg() { return new VMReg((getValue() & REGISTER_MASK_IN_PLACE) >> REGISTER_SHIFT); }
107   public void  setReg(VMReg r) { setValue((short) (r.getValue() << REGISTER_SHIFT | (getValue() & TYPE_MASK_IN_PLACE))); }
108 
109   public OopTypes getType() {
110     int which = (getValue() & TYPE_MASK_IN_PLACE);
111          if (which == OOP_VALUE)    return OopTypes.OOP_VALUE;

112     else if (which == NARROWOOP_VALUE)   return OopTypes.NARROWOOP_VALUE;
113     else if (which == CALLEE_SAVED_VALUE) return OopTypes.CALLEE_SAVED_VALUE;
114     else if (which == DERIVED_OOP_VALUE)  return OopTypes.DERIVED_OOP_VALUE;
115     else throw new InternalError("unknown which " + which + " (TYPE_MASK_IN_PLACE = " + TYPE_MASK_IN_PLACE + ")");
116   }
117   public void setType(OopTypes t) { setValue((short) ((getValue() & REGISTER_MASK_IN_PLACE) | t.getValue())); }
118 
119   public VMReg getContentReg()        { return new VMReg(contentReg); }
120   public void  setContentReg(VMReg r) { contentReg = (short) r.getValue(); }
121 
122   /** Physical location queries */
123   public boolean isRegisterLoc()      { return (getReg().lessThan(VM.getVM().getVMRegImplInfo().getStack0())); }
124   public boolean isStackLoc()         { return (getReg().greaterThanOrEqual(VM.getVM().getVMRegImplInfo().getStack0())); }
125 
126   /** Returns offset from sp. */
127   public int getStackOffset() {
128     if (Assert.ASSERTS_ENABLED) {
129       Assert.that(isStackLoc(), "must be stack location");
130     }
131     return getReg().minus(VM.getVM().getVMRegImplInfo().getStack0());
132   }
133 
134   //--------------------------------------------------------------------------------
135   // Internals only below this point
136   //
137 
138   private void setValue(short value) {
139     this.value = value;
140   }
141 
142   private int getValue() {
143     return value;
144   }
145 }
