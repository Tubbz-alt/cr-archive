25 package sun.jvm.hotspot.gc.cms;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.gc.shared.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 
34 public class ConcurrentMarkSweepGeneration extends CardGeneration {
35   private static AddressField cmsSpaceField;
36 
37   public ConcurrentMarkSweepGeneration(Address addr) {
38     super(addr);
39   }
40 
41   static {
42     VM.registerVMInitializedObserver(new Observer() {
43         public void update(Observable o, Object data) {
44           initialize(VM.getVM().getTypeDataBase());
45         }
46       });
47   }
48 
49   private static synchronized void initialize(TypeDataBase db) {
50     Type type = db.lookupType("ConcurrentMarkSweepGeneration");
51     cmsSpaceField = type.getAddressField("_cmsSpace");
52   }
53 
54   // Accessing space
55   public CompactibleFreeListSpace cmsSpace() {
56     return (CompactibleFreeListSpace) VMObjectFactory.newObject(
57                                  CompactibleFreeListSpace.class,
58                                  cmsSpaceField.getValue(addr));
59   }
60 
61   public long capacity()                { return cmsSpace().capacity(); }
62   public long used()                    { return cmsSpace().used(); }
63   public long free()                    { return cmsSpace().free(); }
64   public long contiguousAvailable()     { throw new RuntimeException("not yet implemented"); }
65   public boolean contains(Address p)    { return cmsSpace().contains(p); }
66   public void spaceIterate(SpaceClosure blk, boolean usedOnly) {
67      blk.doSpace(cmsSpace());
68   }
69   public void liveRegionsIterate(LiveRegionsClosure closure) {
70       closure.doLiveRegions(cmsSpace());
71   }
72 
73   public Generation.Name kind() {
74     return Generation.Name.CONCURRENT_MARK_SWEEP;
75   }
76 
77   public String name() {
78     return "concurrent mark-sweep generation";
79   }
80 
81   public void printOn(PrintStream tty) {
82     tty.println(name());
83     cmsSpace().printOn(tty);
84   }
85 }
