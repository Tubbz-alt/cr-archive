 25 package sun.jvm.hotspot.tools;
 26 
 27 import java.io.PrintStream;
 28 
 29 import sun.jvm.hotspot.HotSpotAgent;
 30 import sun.jvm.hotspot.debugger.DebuggerException;
 31 import sun.jvm.hotspot.debugger.JVMDebugger;
 32 import sun.jvm.hotspot.runtime.VM;
 33 
 34 // generic command line or GUI tool.
 35 // override run & code main as shown below.
 36 
 37 public abstract class Tool implements Runnable {
 38    private HotSpotAgent agent;
 39    private JVMDebugger jvmDebugger;
 40    private int debugeeType;
 41 
 42    // debugeeType is one of constants below
 43    protected static final int DEBUGEE_PID    = 0;
 44    protected static final int DEBUGEE_CORE   = 1;
 45    protected static final int DEBUGEE_REMOTE = 2;
 46 
 47    public Tool() {
 48    }
 49 
 50    public Tool(JVMDebugger d) {
 51       jvmDebugger = d;
 52    }
 53 
 54    public String getName() {
 55       return getClass().getName();
 56    }
 57 
 58    protected boolean needsJavaPrefix() {
 59       return true;
 60    }
 61 
 62    protected void setAgent(HotSpotAgent a) {
 63       agent = a;
 64    }
 65 
 66    protected void setDebugeeType(int dt) {
 67       debugeeType = dt;
 68    }
 69 
 70    protected HotSpotAgent getAgent() {
 71       return agent;
 72    }
 73 
 74    protected int getDebugeeType() {
 75       return debugeeType;
 76    }
 77 
 78    protected void printUsage() {
 79       String name = null;
 80       if (needsJavaPrefix()) {
 81          name = "java " + getName();
 82       } else {
 83          name = getName();
 84       }
 85       System.out.println("Usage: " + name + " [option] <pid>");
 86       System.out.println("\t\t(to connect to a live java process)");
 87       System.out.println("   or " + name + " [option] <executable> <core>");
 88       System.out.println("\t\t(to connect to a core file)");
 89       System.out.println("   or " + name + " [option] [server_id@]<remote server IP or hostname>");
 90       System.out.println("\t\t(to connect to a remote debug server)");
 91       System.out.println();
 92       System.out.println("where option must be one of:");
 93       printFlagsUsage();
 94    }
 95 
 96    protected void printFlagsUsage() {
 97        System.out.println("    -h | -help\tto print this help message");
 98    }
 99 
100    protected void usage() {
101       printUsage();
102    }
103 
104    /*
105       Derived class main should be of the following form:
106 
107       public static void main(String[] args) {
108          <derived class> obj = new <derived class>;
109          obj.execute(args);
110       }
111 
112    */
113 
114    protected void execute(String[] args) {
115        int returnStatus = 1;
116 
117        try {
118            returnStatus = start(args);
119        } catch (Throwable t) {
120            t.printStackTrace(System.err);
121        } finally {
122            stop();
123        }
124 
125        // Exit with 0 or 1
126        System.exit(returnStatus);
127    }
128 
129    public void stop() {
130       if (agent != null) {
131          agent.detach();
132       }
133    }
134 
135    private int start(String[] args) {
136 
137       if ((args.length < 1) || (args.length > 2)) {
138          usage();
139          return 1;
140       }
141 
142       // Attempt to handle -h or -help or some invalid flag
143       if (args[0].startsWith("-h")) {
144           usage();
145           return 0;
146       } else if (args[0].startsWith("-")) {
147           usage();
148           return 1;
149       }
150 
151       PrintStream err = System.err;
152       PrintStream out = System.out;
153 
154       int pid = 0;
155       String coreFileName   = null;
156       String executableName = null;
157       String remoteServer   = null;
158 
159       switch (args.length) {
160         case 1:
161            try {
162               pid = Integer.parseInt(args[0]);
163               debugeeType = DEBUGEE_PID;
164            } catch (NumberFormatException e) {
165               // try remote server
166               remoteServer = args[0];
167               debugeeType  = DEBUGEE_REMOTE;
168            }
169            break;
170 
171         case 2:
172            executableName = args[0];
173            coreFileName   = args[1];
174            debugeeType    = DEBUGEE_CORE;
175            break;
176 
177         default:
178            usage();
179            return 1;
180       }
181 
182       agent = new HotSpotAgent();
183       try {
184         switch (debugeeType) {
185           case DEBUGEE_PID:
186              out.println("Attaching to process ID " + pid + ", please wait...");
187              agent.attach(pid);
188              break;
189 
190           case DEBUGEE_CORE:
191              out.println("Attaching to core " + coreFileName +
192                          " from executable " + executableName + ", please wait...");
193              agent.attach(executableName, coreFileName);
194              break;
195 
196           case DEBUGEE_REMOTE:
197              out.println("Attaching to remote server " + remoteServer + ", please wait...");
198              agent.attach(remoteServer);
199              break;
200         }
201       }
202       catch (DebuggerException e) {
203         switch (debugeeType) {
204           case DEBUGEE_PID:
205              err.print("Error attaching to process: ");
206              break;
207 
208           case DEBUGEE_CORE:
209              err.print("Error attaching to core file: ");
210              break;
211 
212           case DEBUGEE_REMOTE:
213              err.print("Error attaching to remote server: ");
214              break;
215         }
216         if (e.getMessage() != null) {
217           err.println(e.getMessage());
218           e.printStackTrace();
219         }
220         err.println();
221         return 1;
222       }
223 
224       out.println("Debugger attached successfully.");
225       startInternal();
226       return 0;
227    }
228 
229    // When using an existing JVMDebugger.
230    public void start() {
231 
232       if (jvmDebugger == null) {
233          throw new RuntimeException("Tool.start() called with no JVMDebugger set.");
234       }
235       agent = new HotSpotAgent();
236       agent.attach(jvmDebugger);
237       startInternal();
238    }
239 
240    // Remains of the start mechanism, common to both start methods.
241    private void startInternal() {
242 
243       PrintStream out = System.out;
244       VM vm = VM.getVM();
245       if (vm.isCore()) {
246         out.println("Core build detected.");
247       } else if (vm.isClientCompiler()) {
248         out.println("Client compiler detected.");
249       } else if (vm.isServerCompiler()) {
250         out.println("Server compiler detected.");
251       } else {
252         throw new RuntimeException("Fatal error: "
253             + "should have been able to detect core/C1/C2 build");
254       }
255 
256       String version = vm.getVMRelease();
257       if (version != null) {
258         out.print("JVM version is ");
259         out.println(version);
260       }
261 
262       run();
263    }
264 }
