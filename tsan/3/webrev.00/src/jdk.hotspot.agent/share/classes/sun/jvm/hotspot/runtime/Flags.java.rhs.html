29 public enum Flags {
30   // value origin
31   DEFAULT ("Default"),
32   COMMAND_LINE ("Command line"),
33   ENVIRON_VAR ("Environment variable"),
34   CONFIG_FILE ("Config file"),
35   MANAGEMENT ("Management"),
36   ERGONOMIC ("Ergonomic"),
37   ATTACH_ON_DEMAND ("Attach on demand"),
38   INTERNAL ("Internal"),
39   JIMAGE_RESOURCE ("JImage");
40 
41   private final String value;
42 
43   Flags(String val) {
44     this.value = val;
45   }
46   public String value() {
47     return value;
48   }
49 }
