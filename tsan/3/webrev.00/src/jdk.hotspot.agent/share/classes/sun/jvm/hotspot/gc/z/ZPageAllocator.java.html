25 package sun.jvm.hotspot.gc.z;
26 
27 import sun.jvm.hotspot.debugger.Address;
28 import sun.jvm.hotspot.runtime.VM;
29 import sun.jvm.hotspot.runtime.VMObject;
30 import sun.jvm.hotspot.types.CIntegerField;
31 import sun.jvm.hotspot.types.Type;
32 import sun.jvm.hotspot.types.TypeDataBase;
33 
34 // Mirror class for ZPageAllocator
35 
36 public class ZPageAllocator extends VMObject {
37 
38     private static CIntegerField maxCapacityField;
39     private static CIntegerField capacityField;
40     private static CIntegerField usedField;
41 
42     static {
43         VM.registerVMInitializedObserver((o, d) -> initialize(VM.getVM().getTypeDataBase()));
44     }
45 
46     static private synchronized void initialize(TypeDataBase db) {
47         Type type = db.lookupType("ZPageAllocator");
48 
49         maxCapacityField = type.getCIntegerField("_max_capacity");
50         capacityField = type.getCIntegerField("_capacity");
51         usedField = type.getCIntegerField("_used");
52     }
53 
54     public long maxCapacity() {
55         return maxCapacityField.getValue(addr);
56     }
57 
58     public long capacity() {
59         return capacityField.getValue(addr);
60     }
61 
62     public long used() {
63         return usedField.getValue(addr);
64     }
65 
66     public ZPageAllocator(Address addr) {
67         super(addr);
68     }
69 }
