 27 package com.sun.security.sasl.gsskerb;
 28 
 29 import java.util.Locale;
 30 import java.util.Map;
 31 import java.util.logging.Level;
 32 import javax.security.sasl.*;
 33 import com.sun.security.sasl.util.AbstractSaslImpl;
 34 import org.ietf.jgss.*;
 35 import com.sun.security.jgss.ExtendedGSSContext;
 36 import com.sun.security.jgss.InquireType;
 37 
 38 abstract class GssKrb5Base extends AbstractSaslImpl {
 39 
 40     private static final String KRB5_OID_STR = "1.2.840.113554.1.2.2";
 41     protected static Oid KRB5_OID;
 42     protected static final byte[] EMPTY = new byte[0];
 43 
 44     static {
 45         try {
 46             KRB5_OID = new Oid(KRB5_OID_STR);
 47         } catch (GSSException ignore) {}
 48     }
 49 
 50     protected GSSContext secCtx = null;
 51     protected static final int JGSS_QOP = 0;    // unrelated to SASL QOP mask
 52 
 53     protected GssKrb5Base(Map<String, ?> props, String className)
 54         throws SaslException {
 55         super(props, className);
 56     }
 57 
 58     /**
 59      * Retrieves this mechanism's name.
 60      *
 61      * @return  The string "GSSAPI".
 62      */
 63     public String getMechanismName() {
 64         return "GSSAPI";
 65     }
 66 
 67     @Override
 68     public Object getNegotiatedProperty(String propName) {
 69         if (!completed) {
 70             throw new IllegalStateException("Authentication incomplete");
 71         }
 72         String xprefix = "com.sun.security.jgss.inquiretype.";
 73         if (propName.startsWith(xprefix)) {
 74             String type = propName.substring(xprefix.length());
 75             if (logger.isLoggable(Level.FINEST)) {
 76                 logger.logp(Level.FINE, "GssKrb5Base",
 77                         "getNegotiatedProperty", propName);
 78             }
 79             for (InquireType t: InquireType.values()) {
 80                 if (t.name().toLowerCase(Locale.US).equals(type)) {
 81                     try {
 82                         return ((ExtendedGSSContext)secCtx).inquireSecContext(t);
 83                     } catch (GSSException e) {
 84                         if (logger.isLoggable(Level.FINEST)) {
 85                             logger.log(Level.WARNING, "inquireSecContext error", e);
 86                         }
 87                         return null;
 88                     }
 89                 }
 90             }
 91             // No such InquireType. Although not likely to be defined
 92             // as a property in a parent class, still try it.
 93         }
 94         return super.getNegotiatedProperty(propName);
 95     }
 96 
 97     public byte[] unwrap(byte[] incoming, int start, int len)
 98         throws SaslException {
 99         if (!completed) {
100             throw new IllegalStateException("GSSAPI authentication not completed");
101         }
102 
103         // integrity will be true if either privacy or integrity negotiated
104         if (!integrity) {
105             throw new IllegalStateException("No security layer negotiated");
106         }
107 
108         try {
109             MessageProp msgProp = new MessageProp(JGSS_QOP, false);
110             byte[] answer = secCtx.unwrap(incoming, start, len, msgProp);
111             if (privacy && !msgProp.getPrivacy()) {
112                 throw new SaslException("Privacy not protected");
113             }
114             checkMessageProp("", msgProp);
115             if (logger.isLoggable(Level.FINEST)) {
116                 traceOutput(myClassName, "KRB501:Unwrap", "incoming: ",
117                     incoming, start, len);
118                 traceOutput(myClassName, "KRB502:Unwrap", "unwrapped: ",
119                     answer, 0, answer.length);
120             }
121             return answer;
122         } catch (GSSException e) {
123             throw new SaslException("Problems unwrapping SASL buffer", e);
124         }
125     }
126 
127     public byte[] wrap(byte[] outgoing, int start, int len) throws SaslException {
128         if (!completed) {
129             throw new IllegalStateException("GSSAPI authentication not completed");
130         }
131 
132         // integrity will be true if either privacy or integrity negotiated
133         if (!integrity) {
134             throw new IllegalStateException("No security layer negotiated");
135         }
136 
137         // Generate GSS token
138         try {
139             MessageProp msgProp = new MessageProp(JGSS_QOP, privacy);
140             byte[] answer = secCtx.wrap(outgoing, start, len, msgProp);
141             if (logger.isLoggable(Level.FINEST)) {
142                 traceOutput(myClassName, "KRB503:Wrap", "outgoing: ",
143                     outgoing, start, len);
144                 traceOutput(myClassName, "KRB504:Wrap", "wrapped: ",
145                     answer, 0, answer.length);
146             }
147             return answer;
148 
149         } catch (GSSException e) {
150             throw new SaslException("Problem performing GSS wrap", e);
151         }
152     }
153 
154     public void dispose() throws SaslException {
155         if (secCtx != null) {
156             try {
157                 secCtx.dispose();
158             } catch (GSSException e) {
159                 throw new SaslException("Problem disposing GSS context", e);
160             }
161             secCtx = null;
162         }
163     }
164 
165     @SuppressWarnings("deprecation")
166     protected void finalize() throws Throwable {
167         dispose();
168     }
169 
170     void checkMessageProp(String label, MessageProp msgProp)
171             throws SaslException {
172         if (msgProp.isDuplicateToken()) {
173             throw new SaslException(label + "Duplicate token");
174         }
175         if (msgProp.isGapToken()) {
176             throw new SaslException(label + "Gap token");
177         }
178         if (msgProp.isOldToken()) {
179             throw new SaslException(label + "Old token");
180         }
181         if (msgProp.isUnseqToken()) {
182             throw new SaslException(label + "Token not in sequence");
183         }
184     }
185 }
