26 package sun.tools.common;
27 
28 import java.io.BufferedInputStream;
29 import java.io.IOException;
30 import java.io.InputStream;
31 import java.io.InputStreamReader;
32 import java.io.PrintStream;
33 import java.util.Arrays;
34 
35 /**
36  * A helper class which prints the content of input streams to print streams.
37  */
38 public class PrintStreamPrinter {
39 
40     /**
41      * Reads characters in UTF-8 format from the input stream and prints them
42      * with the given print stream. Closes the input stream before it returns.
43      *
44      * @return The number of printed characters.
45      */
46     public static long drainUTF8(InputStream is, PrintStream ps) throws IOException {
47         long result = 0;
48 
49         try (BufferedInputStream bis = new BufferedInputStream(is);
50              InputStreamReader isr = new InputStreamReader(bis, "UTF-8")) {
51             char c[] = new char[256];
52             int n;
53 
54             do {
55                 n = isr.read(c);
56 
57                 if (n > 0) {
58                     result += n;
59                     ps.print(n == c.length ? c : Arrays.copyOf(c, n));
60                 }
61             } while (n > 0);
62         }
63 
64         return result;
65     }
66 }
