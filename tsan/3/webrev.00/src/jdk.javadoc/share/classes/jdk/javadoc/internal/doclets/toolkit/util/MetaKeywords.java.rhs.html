 26 package jdk.javadoc.internal.doclets.toolkit.util;
 27 
 28 import java.util.*;
 29 
 30 import javax.lang.model.element.Element;
 31 import javax.lang.model.element.ModuleElement;
 32 import javax.lang.model.element.PackageElement;
 33 import javax.lang.model.element.TypeElement;
 34 
 35 import jdk.javadoc.internal.doclets.toolkit.BaseConfiguration;
 36 import jdk.javadoc.internal.doclets.toolkit.BaseOptions;
 37 import jdk.javadoc.internal.doclets.toolkit.Resources;
 38 
 39 /**
 40  * Provides methods for creating an array of class, method and
 41  * field names to be included as meta keywords in the HTML header
 42  * of class pages.  These keywords improve search results
 43  * on browsers that look for keywords.
 44  *
 45  *  <p><b>This is NOT part of any supported API.
 46  *  If you write code that depends on this, you do so at your own risk.
 47  *  This code and its internal interfaces are subject to change or
 48  *  deletion without notice.</b>


 49  */
 50 public class MetaKeywords {
 51 
 52     private final BaseOptions options;
 53     private final Resources resources;
 54     private final Utils utils;

 55 
 56     /**
 57      * Constructor
 58      */
 59     public MetaKeywords(BaseConfiguration configuration) {
 60         options = configuration.getOptions();
 61         resources = configuration.getDocResources();
 62         utils = configuration.utils;
 63     }
 64 
 65     /**
 66      * Returns an array of strings where each element
 67      * is a class, method or field name.  This array is
 68      * used to create one meta keyword tag for each element.
 69      * Method parameter lists are converted to "()" and
 70      * overloads are combined.
 71      *
 72      * Constructors are not included because they have the same
 73      * name as the class, which is already included.
 74      * Nested class members are not included because their
 75      * definitions are on separate pages.
 76      */
 77     public List<String> getMetaKeywords(TypeElement typeElement) {
 78         ArrayList<String> results = new ArrayList<>();
 79 
 80         // Add field and method keywords only if -keywords option is used
 81         if (options.keywords()) {
 82             results.addAll(getClassKeyword(typeElement));
 83             results.addAll(getMemberKeywords(utils.getFields(typeElement)));
 84             results.addAll(getMemberKeywords(utils.getMethods(typeElement)));
 85         }
 86         ((ArrayList)results).trimToSize();
 87         return results;
 88     }
 89 
 90     /**
 91      * Get the current class for a meta tag keyword, as the first
 92      * and only element of an array list.
 93      */
 94     protected List<String> getClassKeyword(TypeElement typeElement) {
 95         ArrayList<String> metakeywords = new ArrayList<>(1);
 96         String cltypelower = utils.isInterface(typeElement) ? "interface" : "class";
 97         metakeywords.add(utils.getFullyQualifiedName(typeElement) + " " + cltypelower);
 98         return metakeywords;
 99     }
100 
101     /**
102      * Get the package keywords.
103      */
104     public List<String> getMetaKeywords(PackageElement packageElement) {
105         List<String> result = new ArrayList<>(1);
106         if (options.keywords()) {
107             String pkgName = utils.getPackageName(packageElement);
108             result.add(pkgName + " " + "package");
109         }
110         return result;
111     }
112 
113     /**
114      * Get the module keywords.
115      *
116      * @param mdle the module being documented
117      */
118     public List<String> getMetaKeywordsForModule(ModuleElement mdle) {
119         if (options.keywords()) {
120             return Arrays.asList(mdle.getQualifiedName() + " " + "module");
121         } else {
122             return Collections.emptyList();
123         }
124     }
125 
126     /**
127      * Get the overview keywords.
128      */
129     public List<String> getOverviewMetaKeywords(String title, String docTitle) {
130          List<String> result = new ArrayList<>(1);
131         if (options.keywords()) {
132             String windowOverview = resources.getText(title);
133             if (docTitle.length() > 0) {
134                 result.add(windowOverview + ", " + docTitle);
135             } else {
136                 result.add(windowOverview);
137             }
138         }
139         return result;
140     }
141 
142     /**
143      * Get members for meta tag keywords as an array,
144      * where each member name is a string element of the array.
145      * The parameter lists are not included in the keywords;
146      * therefore all overloaded methods are combined.<br>
147      * Example: getValue(Object) is returned in array as getValue()
148      *
149      * @param members  array of members to be added to keywords
150      */
151     protected List<String> getMemberKeywords(List<? extends Element> members) {
152         ArrayList<String> results = new ArrayList<>();
153         for (Element member : members) {
154             String membername = utils.isMethod(member)
155                     ? utils.getSimpleName(member) + "()"
156                     : utils.getSimpleName(member);
157             if (!results.contains(membername)) {
158                 results.add(membername);
159             }
160         }
161         ((ArrayList)results).trimToSize();
162         return results;
163     }
164 }
