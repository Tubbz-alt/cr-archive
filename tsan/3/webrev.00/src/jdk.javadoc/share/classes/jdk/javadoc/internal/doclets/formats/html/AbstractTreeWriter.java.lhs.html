 26 package jdk.javadoc.internal.doclets.formats.html;
 27 
 28 import java.util.*;
 29 
 30 import javax.lang.model.element.TypeElement;
 31 
 32 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlStyle;
 33 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTag;
 34 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTree;
 35 import jdk.javadoc.internal.doclets.toolkit.Content;
 36 import jdk.javadoc.internal.doclets.toolkit.util.ClassTree;
 37 import jdk.javadoc.internal.doclets.toolkit.util.DocPath;

 40 /**
 41  * Abstract class to print the class hierarchy page for all the Classes. This
 42  * is sub-classed by {@link PackageTreeWriter} and {@link TreeWriter} to
 43  * generate the Package Tree and global Tree(for all the classes and packages)
 44  * pages.
 45  *
 46  *  <p><b>This is NOT part of any supported API.
 47  *  If you write code that depends on this, you do so at your own risk.
 48  *  This code and its internal interfaces are subject to change or
 49  *  deletion without notice.</b>
 50  *
 51  * @author Atul M Dambalkar
 52  */
 53 public abstract class AbstractTreeWriter extends HtmlDocletWriter {
 54 
 55     /**
 56      * The class and interface tree built by using {@link ClassTree}
 57      */
 58     protected final ClassTree classtree;
 59 
 60     /**
 61      * Constructor initializes classtree variable. This constructor will be used
 62      * while generating global tree file "overview-tree.html".
 63      *
 64      * @param configuration  The current configuration
 65      * @param filename   File to be generated.
 66      * @param classtree  Tree built by {@link ClassTree}.
 67      */
 68     protected AbstractTreeWriter(HtmlConfiguration configuration,
 69                                  DocPath filename, ClassTree classtree) {
 70         super(configuration, filename);
 71         this.classtree = classtree;
 72     }
 73 
 74     /**
 75      * Add each level of the class tree. For each sub-class or
 76      * sub-interface indents the next level information.
 77      * Recurses itself to add sub-classes info.
 78      *
 79      * @param parent the superclass or superinterface of the sset
 80      * @param collection  a collection of the sub-classes at this level
 81      * @param isEnum true if we are generating a tree for enums
 82      * @param contentTree the content tree to which the level information will be added
 83      */
 84     protected void addLevelInfo(TypeElement parent, Collection<TypeElement> collection,
 85             boolean isEnum, Content contentTree) {
 86         if (!collection.isEmpty()) {
 87             Content ul = new HtmlTree(HtmlTag.UL);
 88             for (TypeElement local : collection) {
 89                 HtmlTree li = new HtmlTree(HtmlTag.LI);
 90                 li.setStyle(HtmlStyle.circle);
 91                 addPartialInfo(local, li);
 92                 addExtendsImplements(parent, local, li);
 93                 addLevelInfo(local, classtree.directSubClasses(local, isEnum),
 94                              isEnum, li);   // Recurse
 95                 ul.add(li);
 96             }
 97             contentTree.add(ul);
 98         }
 99     }
100 
101     /**
102      * Add the heading for the tree depending upon tree type if it's a
103      * Class Tree or Interface tree.
104      *
105      * @param sset classes which are at the most base level, all the
106      * other classes in this run will derive from these classes
107      * @param heading heading for the tree
108      * @param div the content tree to which the tree will be added
109      */
110     protected void addTree(SortedSet<TypeElement> sset, String heading, HtmlTree div) {
111         addTree(sset, heading, div, false);
112     }
113 
114     protected void addTree(SortedSet<TypeElement> sset, String heading,
115                            HtmlTree div, boolean isEnums) {
116         if (!sset.isEmpty()) {
117             TypeElement firstTypeElement = sset.first();
118             Content headingContent = contents.getContent(heading);
119             Content sectionHeading = HtmlTree.HEADING(Headings.CONTENT_HEADING, true,
120                     headingContent);
121             HtmlTree htmlTree = HtmlTree.SECTION(sectionHeading);
122             addLevelInfo(!utils.isInterface(firstTypeElement) ? firstTypeElement : null,
123                     sset, isEnums, htmlTree);
124             div.add(htmlTree);
125         }
126     }
127 
128     /**
129      * Add information regarding the classes which this class extends or
130      * implements.
131      *
132      * @param parent the parent class of the class being documented
133      * @param typeElement the TypeElement under consideration
134      * @param contentTree the content tree to which the information will be added
135      */
136     protected void addExtendsImplements(TypeElement parent, TypeElement typeElement,
137             Content contentTree) {


138         SortedSet<TypeElement> interfaces = new TreeSet<>(utils.makeGeneralPurposeComparator());
139         typeElement.getInterfaces().stream().forEach((t) -> {
140             interfaces.add(utils.asTypeElement(t));
141         });
142         if (interfaces.size() > (utils.isInterface(typeElement) ? 1 : 0)) {
143             boolean isFirst = true;
144             for (TypeElement intf : interfaces) {
145                 if (parent != intf) {
146                     if (utils.isPublic(intf) || utils.isLinkable(intf)) {
147                         if (isFirst) {
148                             isFirst = false;
149                             if (utils.isInterface(typeElement)) {
150                                 contentTree.add(" (");
151                                 contentTree.add(contents.also);
152                                 contentTree.add(" extends ");
153                             } else {
154                                 contentTree.add(" (implements ");
155                             }
156                         } else {
157                             contentTree.add(", ");
158                         }
159                         addPreQualifiedClassLink(LinkInfoImpl.Kind.TREE, intf, contentTree);
160                     }
161                 }
162             }
163             if (!isFirst) {
164                 contentTree.add(")");
165             }
166         }
167     }
168 
169     /**
170      * Add information about the class kind, if it's a "class" or "interface".
171      *
172      * @param typeElement the class being documented
173      * @param contentTree the content tree to which the information will be added
174      */
175     protected void addPartialInfo(TypeElement typeElement, Content contentTree) {
176         addPreQualifiedStrongClassLink(LinkInfoImpl.Kind.TREE, typeElement, contentTree);
177     }
178 }
