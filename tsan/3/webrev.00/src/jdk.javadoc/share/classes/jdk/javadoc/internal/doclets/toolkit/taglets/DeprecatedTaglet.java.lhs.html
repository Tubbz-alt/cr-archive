26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;

32 import jdk.javadoc.internal.doclets.toolkit.Content;
33 
34 import static com.sun.source.doctree.DocTree.Kind.DEPRECATED;
35 
36 /**
37  * A taglet that represents the @deprecated tag.
38  *
39  *  <p><b>This is NOT part of any supported API.
40  *  If you write code that depends on this, you do so at your own risk.
41  *  This code and its internal interfaces are subject to change or
42  *  deletion without notice.</b>
43  *
44  * @author Jamie Ho
45  */
46 
47 public class DeprecatedTaglet extends BaseTaglet {
48 
49     public DeprecatedTaglet() {
50         super(DEPRECATED.tagName, false,
51                 EnumSet.of(Site.MODULE, Site.TYPE, Site.CONSTRUCTOR, Site.METHOD, Site.FIELD));
52     }
53 
54     @Override
55     public Content getTagletOutput(Element holder, TagletWriter writer) {
56         return writer.deprecatedTagOutput(holder);
57     }
58 }
