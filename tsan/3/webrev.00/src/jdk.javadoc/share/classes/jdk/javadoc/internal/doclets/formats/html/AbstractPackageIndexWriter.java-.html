 26 package jdk.javadoc.internal.doclets.formats.html;
 27 
 28 import java.util.*;
 29 
 30 import javax.lang.model.element.PackageElement;
 31 
 32 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlStyle;
 33 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTag;
 34 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTree;
 35 import jdk.javadoc.internal.doclets.formats.html.markup.Navigation;
 36 import jdk.javadoc.internal.doclets.formats.html.markup.Navigation.PageMode;
 37 import jdk.javadoc.internal.doclets.formats.html.markup.RawHtml;
 38 import jdk.javadoc.internal.doclets.toolkit.Content;
 39 import jdk.javadoc.internal.doclets.toolkit.util.DocFileIOException;
 40 import jdk.javadoc.internal.doclets.toolkit.util.DocPath;

 42 /**
 43  * Abstract class to generate the overview files in
 44  * Frame and Non-Frame format. This will be sub-classed by to
 45  * generate overview-frame.html as well as overview-summary.html.
 46  *
 47  *  <p><b>This is NOT part of any supported API.
 48  *  If you write code that depends on this, you do so at your own risk.
 49  *  This code and its internal interfaces are subject to change or
 50  *  deletion without notice.</b>
 51  *
 52  * @author Atul M Dambalkar
 53  * @author Bhavesh Patel (Modified)
 54  */
 55 public abstract class AbstractPackageIndexWriter extends HtmlDocletWriter {
 56 
 57     /**
 58      * A Set of Packages to be documented.
 59      */
 60     protected SortedSet<PackageElement> packages;
 61 
 62     protected Navigation navBar;
 63 
 64     /**
 65      * Constructor. Also initializes the packages variable.
 66      *
 67      * @param configuration  The current configuration
 68      * @param filename Name of the package index file to be generated.
 69      */
 70     public AbstractPackageIndexWriter(HtmlConfiguration configuration,
 71                                       DocPath filename) {
 72         super(configuration, filename);
 73         packages = configuration.packages;
 74         this.navBar = new Navigation(null, configuration, fixedNavDiv, PageMode.OVERVIEW, path);
 75     }
 76 
 77     /**
 78      * Adds the navigation bar header to the documentation tree.
 79      *
 80      * @param header the document tree to which the navigation bar header will be added
 81      */
 82     protected abstract void addNavigationBarHeader(Content header);
 83 
 84     /**
 85      * Adds the navigation bar footer to the documentation tree.
 86      *
 87      * @param body the document tree to which the navigation bar footer will be added
 88      */
 89     protected abstract void addNavigationBarFooter(Content body);
 90 
 91     /**
 92      * Adds the overview header to the documentation tree.
 93      *
 94      * @param footer the document tree to which the overview header will be added
 95      */
 96     protected abstract void addOverviewHeader(Content footer);
 97 
 98     /**
 99      * Adds the packages list to the documentation tree.
100      *
101      * @param main the document tree to which the packages list will be added
102      */
103     protected abstract void addPackagesList(Content main);
104 
105     /**
106      * Generate and prints the contents in the package index file. Call appropriate
107      * methods from the sub-class in order to generate Frame or Non
108      * Frame format.
109      *
110      * @param title the title of the window
111      * @param description the content for the description META tag
112      * @param includeScript boolean set true if windowtitle script is to be included
113      * @throws DocFileIOException if there is a problem building the package index file
114      */
115     protected void buildPackageIndexFile(String title, String description, boolean includeScript)
116             throws DocFileIOException {
117         String windowOverview = resources.getText(title);
118         Content body = getBody(includeScript, getWindowTitle(windowOverview));
119         Content header = HtmlTree.HEADER();
120         addNavigationBarHeader(header);
121         Content main = HtmlTree.MAIN();
122         addOverviewHeader(main);
123         addIndex(header, main);
124         addOverview(main);
125         Content footer = HtmlTree.FOOTER();
126         addNavigationBarFooter(footer);
127         body.add(header);
128         body.add(main);
129         body.add(footer);
130         printHtmlDocument(
131                 configuration.metakeywords.getOverviewMetaKeywords(title, configuration.doctitle),
132                 description, body);
133     }
134 
135     /**
136      * Default to no overview, override to add overview.
137      *
138      * @param main the document tree to which the overview will be added
139      */
140     protected void addOverview(Content main) { }
141 
142     /**
143      * Adds the frame or non-frame package index to the documentation tree.
144      *
145      * @param header the document tree to which the navigation links will be added
146      * @param main the document tree to which the packages list will be added
147      */
148     protected void addIndex(Content header, Content main) {
149         addIndexContents(header, main);
150     }
151 
152     /**
153      * Adds package index contents. Call appropriate methods from
154      * the sub-classes. Adds it to the body HtmlTree
155      *
156      * @param header the document tree to which navigation links will be added
157      * @param main the document tree to which the packages list will be added
158      */
159     protected void addIndexContents(Content header, Content main) {
160         if (!packages.isEmpty()) {
161             HtmlTree htmlTree = HtmlTree.NAV();
162             htmlTree.setStyle(HtmlStyle.indexNav);
163             HtmlTree ul = new HtmlTree(HtmlTag.UL);
164             addAllClassesLink(ul);
165             if (configuration.showModules  && configuration.modules.size() > 1) {
166                 addAllModulesLink(ul);
167             }
168             htmlTree.add(ul);
169             header.add(htmlTree);
170             addPackagesList(main);
171         }
172     }
173 
174     /**
175      * Adds the doctitle to the documentation tree, if it is specified on the command line.
176      *
177      * @param body the document tree to which the title will be added
178      */
179     protected void addConfigurationTitle(Content body) {
180         if (configuration.doctitle.length() > 0) {
181             Content title = new RawHtml(configuration.doctitle);
182             Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING,
183                     HtmlStyle.title, title);
184             Content div = HtmlTree.DIV(HtmlStyle.header, heading);
185             body.add(div);
186         }
187     }
188 
189     /**
190      * Do nothing. This will be overridden.
191      *
192      * @param div the document tree to which the all classes link will be added
193      */
194     protected void addAllClassesLink(Content div) {
195     }
196 
197     /**
198      * Do nothing. This will be overridden.
199      *
200      * @param div the document tree to which the all modules link will be added
201      */
202     protected void addAllModulesLink(Content div) {
203     }
204 }
