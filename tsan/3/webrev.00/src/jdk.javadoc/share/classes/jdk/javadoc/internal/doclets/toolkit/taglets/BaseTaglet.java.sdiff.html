 26 package jdk.javadoc.internal.doclets.toolkit.taglets;
 27 
 28 import java.util.Set;
 29 import javax.lang.model.element.Element;
 30 
 31 import com.sun.source.doctree.DocTree;


 32 import jdk.javadoc.internal.doclets.toolkit.Content;

 34 /**
 35  * A base class that implements the {@link Taglet} interface.
 36  *
 37  *  <p><b>This is NOT part of any supported API.
 38  *  If you write code that depends on this, you do so at your own risk.
 39  *  This code and its internal interfaces are subject to change or
 40  *  deletion without notice.</b>
 41  *
 42  * @author Jamie Ho
 43  */
 44 public class BaseTaglet implements Taglet {
 45     /**
 46      * The different kinds of place where any given tag may be used.
 47      */
 48     enum Site {
 49         OVERVIEW, MODULE, PACKAGE, TYPE, CONSTRUCTOR, METHOD, FIELD
 50     }
 51 

 52     protected final String name;
 53     private final boolean inline;
 54     private final Set<Site> sites;
 55 
 56     BaseTaglet(String name, boolean inline, Set<Site> sites) {








 57         this.name = name;
 58         this.inline = inline;
 59         this.sites = sites;
 60     }
 61 





 62     /**
 63      * Returns true if this {@code Taglet} can be used in constructor documentation.

 64      * @return true if this {@code Taglet} can be used in constructor documentation and false
 65      * otherwise.
 66      */

 67     public final boolean inConstructor() {
 68         return sites.contains(Site.CONSTRUCTOR);
 69     }
 70 
 71     /**
 72      * Returns true if this {@code Taglet} can be used in field documentation.

 73      * @return true if this {@code Taglet} can be used in field documentation and false
 74      * otherwise.
 75      */

 76     public final boolean inField() {
 77         return  sites.contains(Site.FIELD);
 78     }
 79 
 80     /**
 81      * Returns true if this {@code Taglet} can be used in method documentation.

 82      * @return true if this {@code Taglet} can be used in method documentation and false
 83      * otherwise.
 84      */

 85     public final boolean inMethod() {
 86         return  sites.contains(Site.METHOD);
 87     }
 88 
 89     /**
 90      * Returns true if this {@code Taglet} can be used in overview documentation.

 91      * @return true if this {@code Taglet} can be used in method documentation and false
 92      * otherwise.
 93      */

 94     public final boolean inOverview() {
 95         return  sites.contains(Site.OVERVIEW);
 96     }
 97 
 98     /**
 99      * Returns true if this {@code Taglet} can be used in module documentation.

100      * @return true if this {@code Taglet} can be used in module documentation and false
101      * otherwise.
102      */

103     public final boolean inModule() {
104         return  sites.contains(Site.MODULE);
105     }
106 
107     /**
108      * Returns true if this {@code Taglet} can be used in package documentation.

109      * @return true if this {@code Taglet} can be used in package documentation and false
110      * otherwise.
111      */

112     public final boolean inPackage() {
113         return  sites.contains(Site.PACKAGE);
114     }
115 
116     /**
117      * Returns true if this {@code Taglet} can be used in type documentation (classes or interfaces).

118      * @return true if this {@code Taglet} can be used in type documentation and false
119      * otherwise.
120      */

121     public final boolean inType() {
122         return  sites.contains(Site.TYPE);
123     }
124 
125     /**
126      * Returns true if this {@code Taglet} is an inline tag.

127      * @return true if this {@code Taglet} represents an inline tag and false otherwise.
128      */

129     public final boolean isInlineTag() {
130         return inline;
131     }
132 









133     /**
134      * Returns the name of this tag.

135      * @return the name of this tag.
136      */

137     public String getName() {
138         return name;
139     }
140 















141     /**
142      * {@inheritDoc}
143      * @throws UnsupportedTagletOperationException thrown when the method is
144      *         not supported by the taglet.
145      */

146     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
147         throw new UnsupportedTagletOperationException("Method not supported in taglet " + getName() + ".");
148     }
149 
150     /**
151      * {@inheritDoc}
152      * @throws UnsupportedTagletOperationException thrown when the method is not
153      *         supported by the taglet.
154      */

155     public Content getTagletOutput(Element holder, TagletWriter writer) {
156         throw new UnsupportedTagletOperationException("Method not supported in taglet " + getName() + ".");
157     }
158 }

 26 package jdk.javadoc.internal.doclets.toolkit.taglets;
 27 
 28 import java.util.Set;
 29 import javax.lang.model.element.Element;
 30 
 31 import com.sun.source.doctree.DocTree;
 32 import com.sun.source.doctree.UnknownBlockTagTree;
 33 import jdk.javadoc.doclet.Taglet.Location;
 34 import jdk.javadoc.internal.doclets.toolkit.Content;
 35 
 36 /**
 37  * A base class that implements the {@link Taglet} interface.
 38  *
 39  *  <p><b>This is NOT part of any supported API.
 40  *  If you write code that depends on this, you do so at your own risk.
 41  *  This code and its internal interfaces are subject to change or
 42  *  deletion without notice.</b>


 43  */
 44 public class BaseTaglet implements Taglet {






 45 
 46     protected final DocTree.Kind tagKind;
 47     protected final String name;
 48     private final boolean inline;
 49     private final Set<Location> sites;
 50 
 51     BaseTaglet(DocTree.Kind tagKind, boolean inline, Set<Location> sites) {
 52         this.tagKind = tagKind;
 53         this.name = tagKind.tagName;
 54         this.inline = inline;
 55         this.sites = sites;
 56     }
 57 
 58     BaseTaglet(String name, boolean inline, Set<Location> sites) {
 59         this.tagKind = inline ? DocTree.Kind.UNKNOWN_INLINE_TAG : DocTree.Kind.UNKNOWN_BLOCK_TAG;
 60         this.name = name;
 61         this.inline = inline;
 62         this.sites = sites;
 63     }
 64 
 65     @Override
 66     public Set<Location> getAllowedLocations() {
 67         return sites;
 68     }
 69 
 70     /**
 71      * Returns true if this {@code Taglet} can be used in constructor documentation.
 72      *
 73      * @return true if this {@code Taglet} can be used in constructor documentation and false
 74      * otherwise.
 75      */
 76     @Override
 77     public final boolean inConstructor() {
 78         return sites.contains(Location.CONSTRUCTOR);
 79     }
 80 
 81     /**
 82      * Returns true if this {@code Taglet} can be used in field documentation.
 83      *
 84      * @return true if this {@code Taglet} can be used in field documentation and false
 85      * otherwise.
 86      */
 87     @Override
 88     public final boolean inField() {
 89         return sites.contains(Location.FIELD);
 90     }
 91 
 92     /**
 93      * Returns true if this {@code Taglet} can be used in method documentation.
 94      *
 95      * @return true if this {@code Taglet} can be used in method documentation and false
 96      * otherwise.
 97      */
 98     @Override
 99     public final boolean inMethod() {
100         return sites.contains(Location.METHOD);
101     }
102 
103     /**
104      * Returns true if this {@code Taglet} can be used in overview documentation.
105      *
106      * @return true if this {@code Taglet} can be used in method documentation and false
107      * otherwise.
108      */
109     @Override
110     public final boolean inOverview() {
111         return sites.contains(Location.OVERVIEW);
112     }
113 
114     /**
115      * Returns true if this {@code Taglet} can be used in module documentation.
116      *
117      * @return true if this {@code Taglet} can be used in module documentation and false
118      * otherwise.
119      */
120     @Override
121     public final boolean inModule() {
122         return sites.contains(Location.MODULE);
123     }
124 
125     /**
126      * Returns true if this {@code Taglet} can be used in package documentation.
127      *
128      * @return true if this {@code Taglet} can be used in package documentation and false
129      * otherwise.
130      */
131     @Override
132     public final boolean inPackage() {
133         return sites.contains(Location.PACKAGE);
134     }
135 
136     /**
137      * Returns true if this {@code Taglet} can be used in type documentation (classes or interfaces).
138      *
139      * @return true if this {@code Taglet} can be used in type documentation and false
140      * otherwise.
141      */
142     @Override
143     public final boolean inType() {
144         return sites.contains(Location.TYPE);
145     }
146 
147     /**
148      * Returns true if this {@code Taglet} is an inline tag.
149      *
150      * @return true if this {@code Taglet} represents an inline tag and false otherwise.
151      */
152     @Override
153     public final boolean isInlineTag() {
154         return inline;
155     }
156 
157     /**
158      * Returns the kind of trees recognized by this taglet.
159      *
160      * @return the kind of trees recognized by this taglet
161      */
162     public DocTree.Kind getTagKind() {
163         return tagKind;
164     }
165 
166     /**
167      * Returns the name of this tag.
168      *
169      * @return the name of this tag.
170      */
171     @Override
172     public String getName() {
173         return name;
174     }
175 
176     /**
177      * Returns whether or not this taglet accepts a {@code DocTree} node.
178      * The taglet accepts a tree node if it has the same kind, and
179      * if the kind is {@code UNKNOWN_BLOCK_TAG} the same tag name.
180      *
181      * @param tree the tree node
182      * @return {@code true} if this taglet accepts this tree node.
183      */
184     public boolean accepts(DocTree tree) {
185         return (tree.getKind() == DocTree.Kind.UNKNOWN_BLOCK_TAG
186                     && tagKind == DocTree.Kind.UNKNOWN_BLOCK_TAG)
187                 ? ((UnknownBlockTagTree) tree).getTagName().equals(name)
188                 : tree.getKind() == tagKind;
189     }
190 
191     /**
192      * {@inheritDoc}
193      *
194      * @implSpec This implementation throws {@link UnsupportedTagletOperationException}.
195      */
196     @Override
197     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
198         throw new UnsupportedTagletOperationException("Method not supported in taglet " + getName() + ".");
199     }
200 
201     /**
202      * {@inheritDoc}
203      *
204      * @implSpec This implementation throws {@link UnsupportedTagletOperationException}
205      */
206     @Override
207     public Content getTagletOutput(Element holder, TagletWriter writer) {
208         throw new UnsupportedTagletOperationException("Method not supported in taglet " + getName() + ".");
209     }
210 }
