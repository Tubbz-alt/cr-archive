 44 /**
 45  * A builder for HTML tables, such as the summary tables for various
 46  * types of element.
 47  *
 48  * <p>The table should be used in three phases:
 49  * <ol>
 50  * <li>Configuration: the overall characteristics of the table should be specified
 51  * <li>Population: the content for the cells in each row should be added
 52  * <li>Generation: the HTML content and any associated JavaScript can be accessed
 53  * </ol>
 54  *
 55  * Many methods return the current object, to facilitate fluent builder-style usage.
 56  *
 57  *  <p><b>This is NOT part of any supported API.
 58  *  If you write code that depends on this, you do so at your own risk.
 59  *  This code and its internal interfaces are subject to change or
 60  *  deletion without notice.</b>
 61  */
 62 public class Table {
 63     private final HtmlStyle tableStyle;
 64     private Content caption;
 65     private Map<String, Predicate<Element>> tabMap;
 66     private String defaultTab;
 67     private Set<String> tabs;
 68     private HtmlStyle activeTabStyle = HtmlStyle.activeTableTab;
 69     private HtmlStyle tabStyle = HtmlStyle.tableTab;
 70     private HtmlStyle tabEnd = HtmlStyle.tabEnd;
 71     private IntFunction<String> tabScript;
 72     private Function<Integer, String> tabId = (i -> "t" + i);
 73     private TableHeader header;
 74     private List<HtmlStyle> columnStyles;
 75     private int rowScopeColumnIndex;
 76     private List<HtmlStyle> stripedStyles = Arrays.asList(HtmlStyle.altColor, HtmlStyle.rowColor);
 77     private final List<Content> bodyRows;
 78     private final List<Integer> bodyRowMasks;
 79     private String rowIdPrefix = "i";

 81     /**
 82      * Creates a builder for an HTML table.
 83      *
 84      * @param style     the style class for the {@code <table>} tag
 85      */
 86     public Table(HtmlStyle style) {
 87         this.tableStyle = style;
 88         bodyRows = new ArrayList<>();
 89         bodyRowMasks = new ArrayList<>();
 90     }

 92     /**
 93      * Sets the caption for the table.
 94      * This is ignored if the table is configured to provide tabs to select
 95      * different subsets of rows within the table.
 96      * The caption should be suitable for use as the content of a {@code <caption>}
 97      * element.
 98      *
 99      * <b>For compatibility, the code currently accepts a {@code <caption>} element
100      * as well. This should be removed when all clients rely on using the {@code <caption>}
101      * element being generated by this class.</b>
102      *
103      * @param captionContent the caption
104      * @return this object
105      */
106     public Table setCaption(Content captionContent) {
107         if (captionContent instanceof HtmlTree
108                 && ((HtmlTree) captionContent).htmlTag == HtmlTag.CAPTION) {
109             caption = captionContent;
110         } else {
111             caption = getCaption(captionContent);
112         }
113         return this;
114     }

116     /**
117      * Adds a tab to the table.
118      * Tabs provide a way to display subsets of rows, as determined by a
119      * predicate for the tab, and an element associated with each row.
120      * Tabs will appear left-to-right in the order they are added.
121      *
122      * @param name      the name of the tab
123      * @param predicate the predicate
124      * @return this object
125      */
126     public Table addTab(String name, Predicate<Element> predicate) {
127         if (tabMap == null) {
128             tabMap = new LinkedHashMap<>();     // preserves order that tabs are added
129             tabs = new HashSet<>();             // order not significant
130         }
131         tabMap.put(name, predicate);
132         return this;
133     }

306     /**
307      * Add a row of data to the table.
308      * Each item of content should be suitable for use as the content of a
309      * {@code <th>} or {@code <td>} cell.
310      *
311      * If tabs have been added to the table, the specified element will be used
312      * to determine whether the row should be displayed when any particular tab
313      * is selected, using the predicate specified when the tab was
314      * {@link #add(String,Predicate) added}.
315      *
316      * @param element the element
317      * @param contents the contents for the row
318      * @throws NullPointerException if tabs have previously been added to the table
319      *      and {@code element} is null
320      */
321     public void addRow(Element element, Content... contents) {
322         addRow(element, Arrays.asList(contents));
323     }

325     /**
326      * Add a row of data to the table.
327      * Each item of content should be suitable for use as the content of a
328      * {@code <th>} or {@code <td>} cell.
329      *
330      * If tabs have been added to the table, the specified element will be used
331      * to determine whether the row should be displayed when any particular tab
332      * is selected, using the predicate specified when the tab was
333      * {@link #add(String,Predicate) added}.
334      *
335      * @param element the element
336      * @param contents the contents for the row
337      * @throws NullPointerException if tabs have previously been added to the table
338      *      and {@code element} is null
339      */
340     public void addRow(Element element, List<Content> contents) {
341         if (tabMap != null && element == null) {
342             throw new NullPointerException();
343         }
344 
345         HtmlTree row = new HtmlTree(HtmlTag.TR);
346 
347         if (stripedStyles != null) {
348             int rowIndex = bodyRows.size();
349             row.put(HtmlAttr.CLASS, stripedStyles.get(rowIndex % 2).name());
350         }
351         int colIndex = 0;
352         for (Content c : contents) {
353             HtmlStyle cellStyle = (columnStyles == null || colIndex > columnStyles.size())
354                     ? null
355                     : columnStyles.get(colIndex);
356             HtmlTree cell = (colIndex == rowScopeColumnIndex)
357                     ? HtmlTree.TH(cellStyle, "row", c)
358                     : HtmlTree.TD(cellStyle, c);
359             row.add(cell);
360             colIndex++;
361         }
362         bodyRows.add(row);
363 
364         if (tabMap != null) {
365             int index = bodyRows.size() - 1;
366             row.put(HtmlAttr.ID, (rowIdPrefix + index));
367             int mask = 0;
368             int maskBit = 1;
369             for (Map.Entry<String, Predicate<Element>> e : tabMap.entrySet()) {
370                 String name = e.getKey();
371                 Predicate<Element> predicate = e.getValue();
372                 if (predicate.test(element)) {
373                     tabs.add(name);
374                     mask |= maskBit;
375                 }
376                 maskBit = (maskBit << 1);
377             }
378             bodyRowMasks.add(mask);
379         }
380     }

384      * The table is empty if it has no (body) rows.
385      *
386      * @return true if the table has no rows

392     /**
393      * Returns the HTML for the table.
394      *
395      * @return the HTML
396      */
397     public Content toContent() {
398         HtmlTree mainDiv = new HtmlTree(HtmlTag.DIV);
399         mainDiv.setStyle(tableStyle);



400         HtmlTree table = new HtmlTree(HtmlTag.TABLE);
401         if (tabMap == null || tabs.size() == 1) {
402             if (tabMap == null) {
403                 table.add(caption);
404             } else if (tabs.size() == 1) {
405                 String tabName = tabs.iterator().next();
406                 table.add(getCaption(new StringContent(tabName)));
407             }
408             table.add(getTableBody());
409             mainDiv.add(table);
410         } else {
411             HtmlTree tablist = new HtmlTree(HtmlTag.DIV)
412                     .put(HtmlAttr.ROLE, "tablist")
413                     .put(HtmlAttr.ARIA_ORIENTATION, "horizontal");
414 
415             int tabIndex = 0;
416             tablist.add(createTab(tabId.apply(tabIndex), activeTabStyle, true, defaultTab));
417             table.put(HtmlAttr.ARIA_LABELLEDBY, tabId.apply(tabIndex));
418             for (String tabName : tabMap.keySet()) {
419                 tabIndex++;
420                 if (tabs.contains(tabName)) {
421                     String script = tabScript.apply(1 << (tabIndex - 1));
422                     HtmlTree tab = createTab(tabId.apply(tabIndex), tabStyle, false, tabName);
423                     tab.put(HtmlAttr.ONCLICK, script);
424                     tablist.add(tab);
425                 }
426             }
427             HtmlTree tabpanel = new HtmlTree(HtmlTag.DIV)
428                     .put(HtmlAttr.ID, tableStyle + "_tabpanel")
429                     .put(HtmlAttr.ROLE, "tabpanel");
430             table.add(getTableBody());
431             tabpanel.add(table);
432             mainDiv.add(tablist);
433             mainDiv.add(tabpanel);
434         }
435         return mainDiv;
436     }

438     private HtmlTree createTab(String tabId, HtmlStyle style, boolean defaultTab, String tabName) {
439         HtmlTree tab = new HtmlTree(HtmlTag.BUTTON)
440                 .put(HtmlAttr.ROLE, "tab")
441                 .put(HtmlAttr.ARIA_SELECTED, defaultTab ? "true" : "false")
442                 .put(HtmlAttr.ARIA_CONTROLS, tableStyle + "_tabpanel")
443                 .put(HtmlAttr.TABINDEX, defaultTab ? "0" : "-1")
444                 .put(HtmlAttr.ONKEYDOWN, "switchTab(event)")
445                 .put(HtmlAttr.ID, tabId)
446                 .setStyle(style);
447         tab.add(tabName);
448         return tab;
449     }

451     private Content getTableBody() {
452         ContentBuilder tableContent = new ContentBuilder();
453         tableContent.add(header.toContent());


454         Content tbody = new HtmlTree(HtmlTag.TBODY);
455         bodyRows.forEach(row -> tbody.add(row));
456         tableContent.add(tbody);
457         return tableContent;
458     }

460     /**
461      * Returns whether or not the table needs JavaScript support.
462      * It requires such support if tabs have been added.
463      *
464      * @return true if JavaScript is required
465      */
466     public boolean needsScript() {
467         return (tabs != null) && (tabs.size() > 1);
468     }

470     /**
471      * Returns the script to be used in conjunction with the table.
472      *
473      * @return the script
474      */
475     public String getScript() {
476         if (tabMap == null)
477             throw new IllegalStateException();
478 
479         StringBuilder sb = new StringBuilder();
480 
481         // Add the variable defining the bitmask for each row
482         sb.append("var data").append(" = {");
483         int rowIndex = 0;
484         for (int mask : bodyRowMasks) {
485             if (rowIndex > 0) {
486                 sb.append(",");
487             }
488             sb.append("\"").append(rowIdPrefix).append(rowIndex).append("\":").append(mask);
489             rowIndex++;
490         }
491         sb.append("};\n");
492 
493         // Add the variable defining the tabs
494         sb.append("var tabs = {");
495         appendTabInfo(sb, 65535, tabId.apply(0), defaultTab);
496         int tabIndex = 1;
497         int maskBit = 1;
498         for (String tabName: tabMap.keySet()) {
499             if (tabs.contains(tabName)) {
500                 sb.append(",");
501                 appendTabInfo(sb, maskBit, tabId.apply(tabIndex), tabName);
502             }
503             tabIndex++;
504             maskBit = (maskBit << 1);
505         }
506         sb.append("};\n");
507 
508         // Add the variables defining the stylenames
509         appendStyleInfo(sb,
510                 stripedStyles.get(0), stripedStyles.get(1), tabStyle, activeTabStyle);
511         return sb.toString();
512     }
513 
514     private void appendTabInfo(StringBuilder sb, int value, String id, String name) {
515         sb.append(value)
516                 .append(":[")
517                 .append(Script.stringLiteral(id))
518                 .append(",")
519                 .append(Script.stringLiteral(name))
520                 .append("]");
521     }
522 
523     private void appendStyleInfo(StringBuilder sb, HtmlStyle... styles) {
524         for (HtmlStyle style : styles) {
525             sb.append("var ").append(style).append(" = \"").append(style).append("\";\n");
526         }
527 
528     }

530     private HtmlTree getCaption(Content title) {
531         return new HtmlTree(HtmlTag.CAPTION,
532                 HtmlTree.SPAN(title),
533                 HtmlTree.SPAN(tabEnd, Contents.SPACE));
534     }
