 26 package jdk.javadoc.internal.doclets.toolkit.builders;
 27 
 28 import javax.lang.model.element.ModuleElement;
 29 
 30 import jdk.javadoc.internal.doclets.toolkit.Content;
 31 import jdk.javadoc.internal.doclets.toolkit.DocletException;
 32 import jdk.javadoc.internal.doclets.toolkit.ModuleSummaryWriter;
 33 
 34 
 35 /**
 36  * Builds the summary for a given module.
 37  *
 38  *  <p><b>This is NOT part of any supported API.
 39  *  If you write code that depends on this, you do so at your own risk.
 40  *  This code and its internal interfaces are subject to change or
 41  *  deletion without notice.</b>
 42  *
 43  * @author Bhavesh Patel
 44  */
 45 public class ModuleSummaryBuilder extends AbstractBuilder {
 46 
 47     /**
 48      * The module being documented.
 49      */
 50     private final ModuleElement mdle;
 51 
 52     /**
 53      * The doclet specific writer that will output the result.
 54      */
 55     private final ModuleSummaryWriter moduleWriter;
 56 
 57     /**
 58      * The content that will be added to the module summary documentation tree.
 59      */
 60     private Content contentTree;
 61 
 62     /**
 63      * Construct a new ModuleSummaryBuilder.
 64      *
 65      * @param context  the build context.
 66      * @param mdle the module being documented.
 67      * @param moduleWriter the doclet specific writer that will output the
 68      *        result.
 69      */
 70     private ModuleSummaryBuilder(Context context,
 71             ModuleElement mdle, ModuleSummaryWriter moduleWriter) {
 72         super(context);
 73         this.mdle = mdle;
 74         this.moduleWriter = moduleWriter;
 75     }
 76 
 77     /**
 78      * Construct a new ModuleSummaryBuilder.
 79      *
 80      * @param context  the build context.
 81      * @param mdle the module being documented.
 82      * @param moduleWriter the doclet specific writer that will output the
 83      *        result.
 84      *
 85      * @return an instance of a ModuleSummaryBuilder.
 86      */
 87     public static ModuleSummaryBuilder getInstance(Context context,
 88             ModuleElement mdle, ModuleSummaryWriter moduleWriter) {
 89         return new ModuleSummaryBuilder(context, mdle, moduleWriter);
 90     }
 91 
 92     /**
 93      * Build the module summary.
 94      *
 95      * @throws DocletException if there is a problem while building the documentation
 96      */
 97     @Override
 98     public void build() throws DocletException {
 99         if (moduleWriter == null) {
100             //Doclet does not support this output.
101             return;
102         }
103         buildModuleDoc(contentTree);
104     }
105 
106     /**
107      * Build the module documentation.
108      *
109      * @param contentTree the content tree to which the documentation will be added
110      * @throws DocletException if there is a problem while building the documentation
111      */
112     protected void buildModuleDoc(Content contentTree) throws DocletException {
113         contentTree = moduleWriter.getModuleHeader(mdle.getQualifiedName().toString());
114 
115         buildContent(contentTree);
116 
117         moduleWriter.addModuleFooter(contentTree);
118         moduleWriter.printDocument(contentTree);
119         // uncomment to support doc-files in modules
120         // DocFilesHandler docFilesHandler = configuration.getWriterFactory().getDocFilesWriter(mdle);
121         // docFilesHandler.copyDocFiles();
122     }
123 
124     /**
125      * Build the content for the module doc.
126      *
127      * @param contentTree the content tree to which the module contents
128      *                    will be added
129      * @throws DocletException if there is a problem while building the documentation
130      */
131     protected void buildContent(Content contentTree) throws DocletException {
132         Content moduleContentTree = moduleWriter.getContentHeader();
133 
134         buildModuleDescription(moduleContentTree);
135         buildModuleTags(moduleContentTree);
136         buildSummary(moduleContentTree);
137 
138         moduleWriter.addModuleContent(contentTree, moduleContentTree);
139     }
140 
141     /**
142      * Build the module summary.
143      *
144      * @param moduleContentTree the module content tree to which the summaries will
145      *                           be added
146      * @throws DocletException if there is a problem while building the documentation
147      */
148     protected void buildSummary(Content moduleContentTree) throws DocletException {
149         Content summaryContentTree = moduleWriter.getSummaryHeader();
150 
151         buildPackagesSummary(summaryContentTree);
152         buildModulesSummary(summaryContentTree);
153         buildServicesSummary(summaryContentTree);
154 
155         moduleContentTree.add(moduleWriter.getSummaryTree(summaryContentTree));
156     }
157 
158     /**
159      * Build the modules summary.
160      *
161      * @param summaryContentTree the content tree to which the summaries will
162      *                           be added
163      */
164     protected void buildModulesSummary(Content summaryContentTree) {
165         moduleWriter.addModulesSummary(summaryContentTree);
166     }
167 
168     /**
169      * Build the package summary.
170      *
171      * @param summaryContentTree the content tree to which the summaries will be added
172      */
173     protected void buildPackagesSummary(Content summaryContentTree) {
174         moduleWriter.addPackagesSummary(summaryContentTree);
175     }
176 
177     /**
178      * Build the services summary.
179      *
180      * @param summaryContentTree the content tree to which the summaries will be added
181      */
182     protected void buildServicesSummary(Content summaryContentTree) {
183         moduleWriter.addServicesSummary(summaryContentTree);
184     }
185 
186     /**
187      * Build the description for the module.
188      *
189      * @param moduleContentTree the tree to which the module description will
190      *                           be added
191      */
192     protected void buildModuleDescription(Content moduleContentTree) {
193         if (!configuration.nocomment) {
194             moduleWriter.addModuleDescription(moduleContentTree);
195         }
196     }
197 
198     /**
199      * Build the tags of the summary.
200      *
201      * @param moduleContentTree the tree to which the module tags will be added
202      */
203     protected void buildModuleTags(Content moduleContentTree) {
204         if (!configuration.nocomment) {
205             moduleWriter.addModuleTags(moduleContentTree);
206         }
207     }
208 }
