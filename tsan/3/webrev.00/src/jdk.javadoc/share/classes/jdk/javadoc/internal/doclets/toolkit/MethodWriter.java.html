 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import javax.lang.model.element.ExecutableElement;
 29 import javax.lang.model.element.TypeElement;
 30 import javax.lang.model.type.TypeMirror;
 31 
 32 /**
 33  * The interface for writing method output.
 34  *
 35  *  <p><b>This is NOT part of any supported API.
 36  *  If you write code that depends on this, you do so at your own risk.
 37  *  This code and its internal interfaces are subject to change or
 38  *  deletion without notice.</b>
 39  */
 40 
 41 public interface MethodWriter {
 42 
 43     /**
 44      * Get the method details tree header.
 45      *
 46      * @param memberDetailsTree the content tree representing member details
 47      * @return content tree for the method details header
 48      */
 49     Content getMethodDetailsTreeHeader(Content memberDetailsTree);
 50 
 51     /**
 52      * Get the method documentation tree header.
 53      *
 54      * @param method the method being documented
 55      * @return content tree for the method documentation header
 56      */
 57     Content getMethodDocTreeHeader(ExecutableElement method);
 58 
 59     /**
 60      * Get the signature for the given method.
 61      *
 62      * @param method the method being documented
 63      * @return content tree for the method signature
 64      */
 65     Content getSignature(ExecutableElement method);
 66 
 67     /**
 68      * Add the deprecated output for the given method.
 69      *
 70      * @param method the method being documented
 71      * @param methodDocTree content tree to which the deprecated information will be added
 72      */
 73     void addDeprecated(ExecutableElement method, Content methodDocTree);
 74 
 75     /**
 76      * Add the comments for the given method.
 77      *
 78      * @param holder the holder type (not erasure) of the method
 79      * @param method the method being documented
 80      * @param methodDocTree the content tree to which the comments will be added
 81      */
 82     void addComments(TypeMirror holder, ExecutableElement method, Content methodDocTree);
 83 
 84     /**
 85      * Add the tags for the given method.
 86      *
 87      * @param method the method being documented
 88      * @param methodDocTree the content tree to which the tags will be added
 89      */
 90     void addTags(ExecutableElement method, Content methodDocTree);
 91 
 92     /**
 93      * Get the method details tree.
 94      *
 95      * @param methodDetailsTreeHeader the content tree representing method details header
 96      * @param methodDetailsTree the content tree representing method details
 97      * @return content tree for the method details
 98      */
 99     Content getMethodDetails(Content methodDetailsTreeHeader, Content methodDetailsTree);
100 
101     /**
102      * Get the method documentation.
103      *
104      * @param methodDocTree the content tree representing method documentation
105      * @return content tree for the method documentation
106      */
107     Content getMethodDoc(Content methodDocTree);
108 
109     /**
110      * Gets the member header tree.
111      *
112      * @return a content tree for the member header
113      */
114     Content getMemberTreeHeader();
115 }
