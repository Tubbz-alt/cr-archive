 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.util.SortedSet;
 29 
 30 import javax.lang.model.element.TypeElement;
 31 
 32 import jdk.javadoc.internal.doclets.toolkit.util.DocFileIOException;
 33 
 34 /**
 35  * The interface for writing package summary output.
 36  *
 37  *  <p><b>This is NOT part of any supported API.
 38  *  If you write code that depends on this, you do so at your own risk.
 39  *  This code and its internal interfaces are subject to change or
 40  *  deletion without notice.</b>



 41  */
 42 
 43 public interface PackageSummaryWriter {
 44 
 45     /**
 46      * Get the header for the summary.
 47      *
 48      * @param heading Package name.
 49      * @return the header to be added to the content tree
 50      */
 51     Content getPackageHeader(String heading);
 52 
 53     /**
 54      * Get the header for the package content.
 55      *
 56      * @return a content tree for the package content header
 57      */
 58     Content getContentHeader();
 59 
 60     /**
 61      * Get the header for the package summary.
 62      *
 63      * @return a content tree with the package summary header
 64      */
 65     Content getSummaryHeader();
 66 
 67     /**
 68      * Adds the table of interfaces to the documentation tree.
 69      *
 70      * @param interfaces the interfaces to document.
 71      * @param summaryContentTree the content tree to which the summaries will be added
 72      */
 73     void addInterfaceSummary(SortedSet<TypeElement> interfaces, Content summaryContentTree);

 74 
 75     /**
 76      * Adds the table of classes to the documentation tree.
 77      *
 78      * @param classes the classes to document.
 79      * @param summaryContentTree the content tree to which the summaries will be added
 80      */
 81     void addClassSummary(SortedSet<TypeElement> classes, Content summaryContentTree);

 82 
 83     /**
 84      * Adds the table of enums to the documentation tree.
 85      *
 86      * @param enums the enums to document.
 87      * @param summaryContentTree the content tree to which the summaries will be added
 88      */
 89     void addEnumSummary(SortedSet<TypeElement> enums, Content summaryContentTree);
 90 
 91     /**
 92      * Adds the table of records to the documentation tree.
 93      *
 94      * @param records the records to document.
 95      * @param summaryContentTree the content tree to which the summaries will be added
 96      */
 97     void addRecordSummary(SortedSet<TypeElement> records, Content summaryContentTree);
 98 
 99     /**
100      * Adds the table of exceptions to the documentation tree.
101      *
102      * @param exceptions the exceptions to document.
103      * @param summaryContentTree the content tree to which the summaries will be added
104      */
105     void addExceptionSummary(SortedSet<TypeElement> exceptions, Content summaryContentTree);

106 
107     /**
108      * Adds the table of errors to the documentation tree.
109      *
110      * @param errors the errors to document.
111      * @param summaryContentTree the content tree to which the summaries will be added
112      */
113     void addErrorSummary(SortedSet<TypeElement> errors, Content summaryContentTree);

114 
115     /**
116      * Adds the table of annotation types to the documentation tree.
117      *
118      * @param annoTypes the annotation types to document.
119      * @param summaryContentTree the content tree to which the summaries will be added
120      */
121     void addAnnotationTypeSummary(SortedSet<TypeElement> annoTypes, Content summaryContentTree);

122 
123     /**
124      * Adds the package description from the "packages.html" file to the documentation
125      * tree.
126      *
127      * @param packageContentTree the content tree to which the package description
128      *                           will be added
129      */
130     void addPackageDescription(Content packageContentTree);
131 
132     /**
133      * Adds the tag information from the "packages.html" file to the documentation
134      * tree.
135      *
136      * @param packageContentTree the content tree to which the package tags will
137      *                           be added
138      */
139     void addPackageTags(Content packageContentTree);
140 
141     /**
142      * Adds the tag information from the "packages.html" or "package-info.java" file to the
143      * documentation tree.
144      *

145      * @param packageContentTree the package content tree to be added
146      */
147     void addPackageContent(Content packageContentTree);
148 
149     /**
150      * Adds the footer to the documentation tree.


151      */
152     void addPackageFooter();
153 
154     /**
155      * Print the package summary document.
156      *
157      * @param contentTree the content tree that will be printed
158      * @throws DocFileIOException if there is a problem while writing the document
159      */
160     void printDocument(Content contentTree) throws DocFileIOException;
161 
162     /**
163      * Gets the package summary tree.
164      * @param summaryContentTree the content tree representing the package summary
165      * @return a content tree for the package summary
166      */
167     Content getPackageSummary(Content summaryContentTree);
168 }
