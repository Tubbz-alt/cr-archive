 26 package jdk.javadoc.internal.doclets.formats.html;
 27 
 28 /**
 29  * Index item for search.
 30  *
 31  *  <p><b>This is NOT part of any supported API.
 32  *  If you write code that depends on this, you do so at your own risk.
 33  *  This code and its internal interfaces are subject to change or
 34  *  deletion without notice.</b>
 35  */
 36 public class SearchIndexItem {
 37 
 38     enum Category {
 39         MODULES,
 40         PACKAGES,
 41         TYPES,
 42         MEMBERS,
 43         SEARCH_TAGS
 44     }
 45 
 46     private Category category;
 47     private String label = "";
 48     private String url = "";
 49     private String containingModule = "";
 50     private String containingPackage = "";
 51     private String containingClass = "";
 52     private String holder = "";
 53     private String description = "";
 54     private boolean systemProperty;
 55 
 56     public void setLabel(String l) {
 57         label = l;
 58     }
 59 
 60     public String getLabel() {
 61         return label;
 62     }
 63 
 64     public void setUrl(String u) {
 65         url = u;
 66     }
 67 
 68     public String getUrl() {
 69         return url;
 70     }
 71 
 72     public void setContainingModule(String m) {
 73         containingModule = m;
 74     }
 75 
 76     public void setContainingPackage(String p) {
 77         containingPackage = p;
 78     }
 79 
 80     public void setContainingClass(String c) {
 81         containingClass = c;
 82     }
 83 
 84     public void setCategory(Category c) {
 85         category = c;
 86     }
 87 
 88     public void setHolder(String h) {
 89         holder = h;
 90     }
 91 
 92     public String getHolder() {
 93         return holder;
 94     }
 95 
 96     public void setDescription(String d) {
 97         description = d;
 98     }
 99 
100     public String getDescription() {
101         return description;
102     }
103 
104     public void setSystemProperty(boolean value) {
105         systemProperty = value;
106     }
107 
108     public boolean isSystemProperty() {
109         return systemProperty;
110     }
111 
112     @Override
113     public String toString() {
114         // TODO: Additional processing is required, see JDK-8238495
115         StringBuilder item = new StringBuilder();
116         switch (category) {
117             case MODULES:
118                 item.append("{")
119                         .append("\"l\":\"").append(label).append("\"")
120                         .append("}");
121                 break;
122             case PACKAGES:
123                 item.append("{");
124                 if (!containingModule.isEmpty()) {
125                     item.append("\"m\":\"").append(containingModule).append("\",");
126                 }
127                 item.append("\"l\":\"").append(label).append("\"");
128                 if (!url.isEmpty()) {
129                     item.append(",\"u\":\"").append(url).append("\"");
130                 }
131                 item.append("}");
132                 break;
133             case TYPES:
134                 item.append("{");
135                 if (!containingPackage.isEmpty()) {
136                     item.append("\"p\":\"").append(containingPackage).append("\",");
137                 }
138                 item.append("\"l\":\"").append(label).append("\"");
139                 if (!url.isEmpty()) {
140                     item.append(",\"u\":\"").append(url).append("\"");
141                 }
142                 item.append("}");
143                 break;
144             case MEMBERS:
145                 item.append("{")
146                         .append("\"p\":\"").append(containingPackage).append("\",")
147                         .append("\"c\":\"").append(containingClass).append("\",")
148                         .append("\"l\":\"").append(label).append("\"");
149                 if (!url.isEmpty()) {
150                     item.append(",\"u\":\"").append(url).append("\"");
151                 }
152                 item.append("}");
153                 break;
154             case SEARCH_TAGS:
155                 item.append("{")
156                         .append("\"l\":\"").append(label).append("\",")
157                         .append("\"h\":\"").append(holder).append("\",");
158                 if (!description.isEmpty()) {
159                     item.append("\"d\":\"").append(description).append("\",");
160                 }
161                 item.append("\"u\":\"").append(url).append("\"")
162                         .append("}");
163                 break;
164             default:
165                 throw new AssertionError("Unexpected category: " + category);
166         }
167         return item.toString();
168     }
169 
170     /**
171      * Get the part of the label after the last dot, or whole label if no dots.
172      *
173      * @return the simple name
174      */
175     public String getSimpleName() {
176         return label.substring(label.lastIndexOf('.') + 1);
177     }
178 }
