26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import java.util.List;
30 
31 import javax.lang.model.element.Element;
32 
33 import com.sun.source.doctree.DocTree;

34 import jdk.javadoc.internal.doclets.toolkit.Content;
35 import jdk.javadoc.internal.doclets.toolkit.util.CommentHelper;
36 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder;
37 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder.Input;
38 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
39 
40 import static com.sun.source.doctree.DocTree.Kind.SEE;
41 
42 /**
43  * A taglet that represents the @see tag.
44  *
45  *  <p><b>This is NOT part of any supported API.
46  *  If you write code that depends on this, you do so at your own risk.
47  *  This code and its internal interfaces are subject to change or
48  *  deletion without notice.</b>
49  *
50  * @author Jamie Ho
51  */
52 public class SeeTaglet extends BaseTaglet implements InheritableTaglet {
53 
54     public SeeTaglet() {
55         super(SEE.tagName, false, EnumSet.allOf(Site.class));
56     }
57 
58     @Override
59     public void inherit(DocFinder.Input input, DocFinder.Output output) {
60         List<? extends DocTree> tags = input.utils.getSeeTrees(input.element);
61         if (!tags.isEmpty()) {
62             CommentHelper ch =  input.utils.getCommentHelper(input.element);
63             output.holder = input.element;
64             output.holderTag = tags.get(0);
65             output.inlineTags = input.isFirstSentence
66                     ? ch.getFirstSentenceTrees(input.utils.configuration, output.holderTag)
67                     : ch.getReference(output.holderTag);
68         }
69     }
70 
71     @Override
72     public Content getTagletOutput(Element holder, TagletWriter writer) {
73         Utils utils = writer.configuration().utils;
74         List<? extends DocTree> tags = utils.getSeeTrees(holder);
75         Element e = holder;
76         if (tags.isEmpty() && utils.isExecutableElement(holder)) {
77             Input input = new DocFinder.Input(utils, holder, this);
78             DocFinder.Output inheritedDoc = DocFinder.search(writer.configuration(), input);
79             if (inheritedDoc.holder != null) {
80                 tags = utils.getSeeTrees(inheritedDoc.holder);
81                 e = inheritedDoc.holder;
82             }
83         }
84         return writer.seeTagOutput(e, tags);
85     }
86 }
