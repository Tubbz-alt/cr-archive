 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.io.IOException;
 29 import java.io.StringWriter;
 30 import java.io.Writer;
 31 import java.util.Objects;
 32 
 33 /**
 34  * A class to create content for javadoc output pages.
 35  *
 36  *  <p><b>This is NOT part of any supported API.
 37  *  If you write code that depends on this, you do so at your own risk.
 38  *  This code and its internal interfaces are subject to change or
 39  *  deletion without notice.</b>
 40  *
 41  * @author Bhavesh Patel
 42  */
 43 public abstract class Content {
 44 
 45     /**
 46      * Returns a string representation of the content.
 47      *
 48      * @return string representation of the content
 49      */
 50     @Override
 51     public String toString() {
 52         StringWriter out = new StringWriter();
 53         try {
 54             write(out, true);
 55         } catch (IOException e) {
 56             // cannot happen from StringWriter
 57             throw new AssertionError(e);
 58         }
 59         return out.toString();
 60     }
 61 
 62     /**
 63      * Adds content to the existing content.
 64      *
 65      * @param content content that needs to be added
 66      */
 67     public abstract void add(Content content);
 68 
 69     /**
 70      * Adds a string content to the existing content.
 71      *
 72      * @param stringContent the string content to be added
 73      */
 74     public abstract void add(CharSequence stringContent);
 75 
 76     /**
 77      * Writes content to a writer.
 78      *
 79      * @param writer the writer
 80      * @param atNewline whether the writer has just written a newline
 81      * @return  whether the writer has just written a newline
 82      * @throws IOException if an error occurs while writing the output
 83      */
 84     public abstract boolean write(Writer writer, boolean atNewline) throws IOException ;
 85 
 86     /**
 87      * Returns true if the content is empty.
 88      *
 89      * @return true if no content to be displayed else return false
 90      */
 91     public abstract boolean isEmpty();
 92 
 93     /**
 94      * Returns true if the content is valid.
 95      *
 96      * @return true if the content is valid else return false
 97      */
 98     public boolean isValid() {
 99         return !isEmpty();
100     }
101 
102     /**
103      * Return the number of characters of plain text content in this object
104      * (optional operation.)
105      * @return the number of characters of plain text content in this
106      */
107     public int charCount() {
108         return 0;
109     }
110 
111     /**
112      * Checks for null values.
113      *
114      * @param <T> the type of the item being checked
115      * @param t reference type to check for null values
116      * @return the reference type if not null or else throws a null pointer exception
117      */
118     protected static <T> T nullCheck(T t) {
119         return Objects.requireNonNull(t);
120     }
121 }
