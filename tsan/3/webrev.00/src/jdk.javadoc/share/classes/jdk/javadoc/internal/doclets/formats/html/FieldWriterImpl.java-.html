 29 import jdk.javadoc.internal.doclets.formats.html.markup.TableHeader;

 31 import java.util.Arrays;
 32 import java.util.List;
 33 
 34 import javax.lang.model.element.Element;
 35 import javax.lang.model.element.TypeElement;
 36 import javax.lang.model.element.VariableElement;
 37 
 38 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlStyle;
 39 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTag;
 40 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTree;
 41 import jdk.javadoc.internal.doclets.formats.html.markup.StringContent;

 44 import jdk.javadoc.internal.doclets.toolkit.MemberSummaryWriter;

 46 /**
 47  * Writes field documentation in HTML format.
 48  *
 49  *  <p><b>This is NOT part of any supported API.
 50  *  If you write code that depends on this, you do so at your own risk.
 51  *  This code and its internal interfaces are subject to change or
 52  *  deletion without notice.</b>
 53  *
 54  * @author Robert Field
 55  * @author Atul M Dambalkar
 56  * @author Jamie Ho (rewrite)
 57  * @author Bhavesh Patel (Modified)
 58  */
 59 public class FieldWriterImpl extends AbstractMemberWriter
 60     implements FieldWriter, MemberSummaryWriter {
 61 
 62     public FieldWriterImpl(SubWriterHolderWriter writer, TypeElement typeElement) {
 63         super(writer, typeElement);
 64     }
 65 
 66     public FieldWriterImpl(SubWriterHolderWriter writer) {
 67         super(writer);
 68     }
 69 
 70     /**
 71      * {@inheritDoc}
 72      */
 73     @Override
 74     public Content getMemberSummaryHeader(TypeElement typeElement,
 75             Content memberSummaryTree) {
 76         memberSummaryTree.add(MarkerComments.START_OF_FIELD_SUMMARY);
 77         Content memberTree = writer.getMemberTreeHeader();
 78         writer.addSummaryHeader(this, typeElement, memberTree);
 79         return memberTree;
 80     }
 81 
 82     /**
 83      * {@inheritDoc}
 84      */
 85     @Override
 86     public void addMemberTree(Content memberSummaryTree, Content memberTree) {
 87         writer.addMemberTree(memberSummaryTree, memberTree);
 88     }
 89 
 90     /**
 91      * {@inheritDoc}
 92      */
 93     @Override
 94     public Content getFieldDetailsTreeHeader(TypeElement typeElement, Content memberDetailsTree) {
 95         memberDetailsTree.add(MarkerComments.START_OF_FIELD_DETAILS);
 96         Content fieldDetailsTree = writer.getMemberTreeHeader();
 97         fieldDetailsTree.add(links.createAnchor(
 98                 SectionName.FIELD_DETAIL));
 99         Content heading = HtmlTree.HEADING(Headings.TypeDeclaration.DETAILS_HEADING,
100                 contents.fieldDetailsLabel);
101         fieldDetailsTree.add(heading);
102         return fieldDetailsTree;
103     }
104 
105     /**
106      * {@inheritDoc}
107      */
108     @Override
109     public Content getFieldDocTreeHeader(VariableElement field, Content fieldDetailsTree) {
110         fieldDetailsTree.add(links.createAnchor(name(field)));
111         Content fieldTree = writer.getMemberTreeHeader();
112         Content heading = new HtmlTree(Headings.TypeDeclaration.MEMBER_HEADING);
113         heading.add(name(field));
114         fieldTree.add(heading);
115         return fieldTree;
116     }
117 
118     /**
119      * {@inheritDoc}
120      */
121     @Override
122     public Content getSignature(VariableElement field) {
123         Content pre = new HtmlTree(HtmlTag.PRE);
124         writer.addAnnotationInfo(field, pre);
125         addModifiers(field, pre);
126         Content fieldlink = writer.getLink(new LinkInfoImpl(
127                 configuration, LinkInfoImpl.Kind.MEMBER, field.asType()));
128         pre.add(fieldlink);
129         pre.add(" ");
130         if (configuration.linksource) {
131             Content fieldName = new StringContent(name(field));
132             writer.addSrcLink(field, fieldName, pre);
133         } else {
134             addName(name(field), pre);
135         }
136         return pre;
137     }
138 
139     /**
140      * {@inheritDoc}
141      */
142     @Override
143     public void addDeprecated(VariableElement field, Content fieldTree) {
144         addDeprecatedInfo(field, fieldTree);
145     }
146 
147     /**
148      * {@inheritDoc}
149      */
150     @Override
151     public void addComments(VariableElement field, Content fieldTree) {
152         if (!utils.getFullBody(field).isEmpty()) {
153             writer.addInlineComment(field, fieldTree);
154         }
155     }
156 
157     /**
158      * {@inheritDoc}
159      */
160     @Override
161     public void addTags(VariableElement field, Content fieldTree) {
162         writer.addTagsInfo(field, fieldTree);
163     }
164 
165     /**
166      * {@inheritDoc}
167      */
168     @Override
169     public Content getFieldDetails(Content fieldDetailsTree) {
170         return HtmlTree.SECTION(getMemberTree(fieldDetailsTree));
171     }
172 
173     /**
174      * {@inheritDoc}
175      */
176     @Override
177     public Content getFieldDoc(Content fieldTree,
178             boolean isLastContent) {
179         return getMemberTree(fieldTree, isLastContent);
180     }
181 
182     /**
183      * {@inheritDoc}
184      */
185     @Override
186     public void addSummaryLabel(Content memberTree) {
187         Content label = HtmlTree.HEADING(Headings.TypeDeclaration.SUMMARY_HEADING,
188                 contents.fieldSummaryLabel);
189         memberTree.add(label);
190     }
191 
192     /**
193      * {@inheritDoc}
194      */
195     @Override
196     public TableHeader getSummaryTableHeader(Element member) {
197         return new TableHeader(contents.modifierAndTypeLabel, contents.fieldLabel,
198                 contents.descriptionLabel);
199     }
200 
201     @Override
202     protected Table createSummaryTable() {
203         List<HtmlStyle> bodyRowStyles = Arrays.asList(HtmlStyle.colFirst, HtmlStyle.colSecond,
204                 HtmlStyle.colLast);
205 
206         return new Table(HtmlStyle.memberSummary)
207                 .setCaption(contents.fields)
208                 .setHeader(getSummaryTableHeader(typeElement))
209                 .setRowScopeColumn(1)
210                 .setColumnStyles(bodyRowStyles);
211     }
212 
213     /**
214      * {@inheritDoc}
215      */
216     @Override
217     public void addSummaryAnchor(TypeElement typeElement, Content memberTree) {
218         memberTree.add(links.createAnchor(
219                 SectionName.FIELD_SUMMARY));
220     }
221 
222     /**
223      * {@inheritDoc}
224      */
225     @Override
226     public void addInheritedSummaryAnchor(TypeElement typeElement, Content inheritedTree) {
227         inheritedTree.add(links.createAnchor(
228                 SectionName.FIELDS_INHERITANCE, configuration.getClassName(typeElement)));
229     }
230 
231     /**
232      * {@inheritDoc}
233      */
234     @Override
235     public void addInheritedSummaryLabel(TypeElement typeElement, Content inheritedTree) {
236         Content classLink = writer.getPreQualifiedClassLink(
237                 LinkInfoImpl.Kind.MEMBER, typeElement, false);
238         Content label;
239         if (configuration.summarizeOverriddenMethods) {
240             label = new StringContent(utils.isClass(typeElement)
241                     ? resources.getText("doclet.Fields_Declared_In_Class")
242                     : resources.getText("doclet.Fields_Declared_In_Interface"));
243         } else {
244             label = new StringContent(utils.isClass(typeElement)
245                     ? resources.getText("doclet.Fields_Inherited_From_Class")
246                     : resources.getText("doclet.Fields_Inherited_From_Interface"));
247         }
248         Content labelHeading = HtmlTree.HEADING(Headings.TypeDeclaration.INHERITED_SUMMARY_HEADING,
249                 label);
250         labelHeading.add(Contents.SPACE);
251         labelHeading.add(classLink);
252         inheritedTree.add(labelHeading);
253     }
254 
255     /**
256      * {@inheritDoc}
257      */
258     @Override
259     protected void addSummaryLink(LinkInfoImpl.Kind context, TypeElement typeElement, Element member,
260             Content tdSummary) {
261         Content memberLink = HtmlTree.SPAN(HtmlStyle.memberNameLink,
262                 writer.getDocLink(context, typeElement , member, name(member), false));
263         Content code = HtmlTree.CODE(memberLink);
264         tdSummary.add(code);
265     }
266 
267     /**
268      * {@inheritDoc}
269      */
270     @Override
271     protected void addInheritedSummaryLink(TypeElement typeElement, Element member, Content linksTree) {
272         linksTree.add(
273                 writer.getDocLink(LinkInfoImpl.Kind.MEMBER, typeElement, member,
274                 name(member), false));
275     }
276 
277     /**
278      * {@inheritDoc}
279      */
280     @Override
281     protected void addSummaryType(Element member, Content tdSummaryType) {
282         addModifierAndType(member, member.asType(), tdSummaryType);
283     }
284 
285     /**
286      * {@inheritDoc}
287      */
288     @Override
289     protected Content getDeprecatedLink(Element member) {
290         String name = utils.getFullyQualifiedName(member) + "." + member.getSimpleName();
291         return writer.getDocLink(LinkInfoImpl.Kind.MEMBER, member, name);
292     }
