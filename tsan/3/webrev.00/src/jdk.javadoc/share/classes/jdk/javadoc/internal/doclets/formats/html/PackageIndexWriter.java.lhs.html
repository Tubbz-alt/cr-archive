 44 /**
 45  * Generate the package index page "overview-summary.html" for the right-hand
 46  * frame. A click on the package name on this page will update the same frame
 47  * with the "package-summary.html" file for the clicked package.
 48  *
 49  *  <p><b>This is NOT part of any supported API.
 50  *  If you write code that depends on this, you do so at your own risk.
 51  *  This code and its internal interfaces are subject to change or
 52  *  deletion without notice.</b>
 53  *
 54  * @author Atul M Dambalkar
 55  * @author Bhavesh Patel (Modified)
 56  */
 57 public class PackageIndexWriter extends AbstractPackageIndexWriter {





 58 
 59     /**
 60      * Construct the PackageIndexWriter. Also constructs the grouping
 61      * information as provided on the command line by "-group" option. Stores
 62      * the order of groups specified by the user.
 63      *
 64      * @param configuration the configuration for this doclet
 65      * @param filename the path of the page to be generated
 66      * @see Group
 67      */
 68     public PackageIndexWriter(HtmlConfiguration configuration, DocPath filename) {
 69         super(configuration, filename);

 70     }
 71 
 72     /**
 73      * Generate the package index page for the right-hand frame.
 74      *
 75      * @param configuration the current configuration of the doclet.
 76      * @throws DocFileIOException if there is a problem generating the package index page
 77      */
 78     public static void generate(HtmlConfiguration configuration) throws DocFileIOException {
 79         DocPath filename = DocPaths.overviewSummary(configuration.frames);
 80         PackageIndexWriter packgen = new PackageIndexWriter(configuration, filename);
 81         packgen.buildPackageIndexFile("doclet.Window_Overview_Summary", "package index", true);
 82     }
 83 
 84     /**
 85      * Depending upon the grouping information and their titles, add
 86      * separate table indices for each package group.
 87      *
 88      * @param header the documentation tree to which the navigational links will be added
 89      * @param main the documentation tree to which the packages list will be added
 90      */
 91     @Override
 92     protected void addIndex(Content header, Content main) {
 93         addIndexContents(header, main);
 94     }
 95 
 96     /**
 97      * {@inheritDoc}
 98      */
 99     @Override
100     protected void addPackagesList(Content main) {
101         Map<String, SortedSet<PackageElement>> groupPackageMap
102                 = configuration.group.groupPackages(packages);
103 
104         if (!groupPackageMap.keySet().isEmpty()) {
105             Table table =  new Table(HtmlStyle.overviewSummary)
106                     .setHeader(getPackageTableHeader())
107                     .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast)
108                     .setDefaultTab(resources.getText("doclet.All_Packages"))
109                     .setTabScript(i -> "show(" + i + ");")
110                     .setTabId(i -> (i == 0) ? "t0" : ("t" + (1 << (i - 1))));
111 
112             // add the tabs in command-line order
113             for (String groupName : configuration.group.getGroupList()) {
114                 Set<PackageElement> groupPackages = groupPackageMap.get(groupName);
115                 if (groupPackages != null) {
116                     table.addTab(groupName, groupPackages::contains);
117                 }
118             }
119 
120             for (PackageElement pkg : configuration.packages) {
121                 if (!pkg.isUnnamed()) {
122                     if (!(configuration.nodeprecated && utils.isDeprecated(pkg))) {
123                         Content packageLinkContent = getPackageLink(pkg, getPackageName(pkg));
124                         Content summaryContent = new ContentBuilder();
125                         addSummaryComment(pkg, summaryContent);
126                         table.addRow(pkg, packageLinkContent, summaryContent);
127                     }
128                 }
129             }
130 
131             Content div = HtmlTree.DIV(HtmlStyle.contentContainer, table.toContent());
132             main.add(div);
133 
134             if (table.needsScript()) {
135                 getMainBodyScript().append(table.getScript());
136             }
137         }
138     }
139 
140     /**
141      * Adds the overview summary comment for this documentation. Add one line
142      * summary at the top of the page and generate a link to the description,
143      * which is added at the end of this page.
144      *
145      * @param main the documentation tree to which the overview header will be added
146      */
147     @Override
148     protected void addOverviewHeader(Content main) {
149         addConfigurationTitle(main);
150         if (!utils.getFullBody(configuration.overviewElement).isEmpty()) {
151             HtmlTree div = new HtmlTree(HtmlTag.DIV);
152             div.setStyle(HtmlStyle.contentContainer);
153             addOverviewComment(div);
154             main.add(div);
155         }
156     }
157 
158     /**
159      * Adds the overview comment as provided in the file specified by the
160      * "-overview" option on the command line.
161      *
162      * @param htmltree the documentation tree to which the overview comment will
163      *                 be added
164      */
165     protected void addOverviewComment(Content htmltree) {
166         if (!utils.getFullBody(configuration.overviewElement).isEmpty()) {
167             addInlineComment(configuration.overviewElement, htmltree);
168         }
169     }
170 
171     /**
172      * Adds the top text (from the -top option), the upper
173      * navigation bar, and then the title (from the"-title"
174      * option), at the top of page.
175      *
176      * @param header the documentation tree to which the navigation bar header will be added
177      */
178     @Override
179     protected void addNavigationBarHeader(Content header) {
180         addTop(header);
181         navBar.setUserHeader(getUserHeaderFooter(true));
182         header.add(navBar.getContent(true));
183     }
184 
185     /**
186      * Adds the lower navigation bar and the bottom text
187      * (from the -bottom option) at the bottom of page.
188      *
189      * @param footer the documentation tree to which the navigation bar footer will be added
190      */
191     @Override
192     protected void addNavigationBarFooter(Content footer) {
193         navBar.setUserFooter(getUserHeaderFooter(false));
194         footer.add(navBar.getContent(false));
195         addBottom(footer);
196     }
197 }
