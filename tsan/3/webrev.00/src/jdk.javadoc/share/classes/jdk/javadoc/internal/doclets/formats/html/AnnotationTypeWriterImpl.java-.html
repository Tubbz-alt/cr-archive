 47  * Generate the Class Information Page.
 48  *
 49  *  <p><b>This is NOT part of any supported API.
 50  *  If you write code that depends on this, you do so at your own risk.
 51  *  This code and its internal interfaces are subject to change or
 52  *  deletion without notice.</b>
 53  *
 54  * @see java.util.Collections
 55  * @see java.util.List
 56  * @see java.util.ArrayList
 57  * @see java.util.HashMap
 58  *
 59  * @author Atul M Dambalkar
 60  * @author Robert Field
 61  * @author Bhavesh Patel (Modified)
 62  */
 63 public class AnnotationTypeWriterImpl extends SubWriterHolderWriter
 64         implements AnnotationTypeWriter {
 65 
 66     protected TypeElement annotationType;
 67 
 68     private final Navigation navBar;
 69 
 70     /**
 71      * @param configuration the configuration
 72      * @param annotationType the annotation type being documented.
 73      */
 74     public AnnotationTypeWriterImpl(HtmlConfiguration configuration,
 75             TypeElement annotationType) {
 76         super(configuration, configuration.docPaths.forClass(annotationType));
 77         this.annotationType = annotationType;
 78         configuration.currentTypeElement = annotationType;
 79         this.navBar = new Navigation(annotationType, configuration, fixedNavDiv, PageMode.CLASS, path);
 80     }
 81 
 82     /**
 83      * {@inheritDoc}
 84      */
 85     @Override
 86     public Content getHeader(String header) {
 87         HtmlTree bodyTree = getBody(true, getWindowTitle(utils.getSimpleName(annotationType)));
 88         HtmlTree htmlTree = HtmlTree.HEADER();
 89         addTop(htmlTree);
 90         Content linkContent = getModuleLink(utils.elementUtils.getModuleOf(annotationType),
 91                 contents.moduleLabel);
 92         navBar.setNavLinkModule(linkContent);
 93         navBar.setMemberSummaryBuilder(configuration.getBuilderFactory().getMemberSummaryBuilder(this));
 94         navBar.setUserHeader(getUserHeaderFooter(true));
 95         htmlTree.add(navBar.getContent(true));
 96         bodyTree.add(htmlTree);
 97         bodyTree.add(MarkerComments.START_OF_CLASS_DATA);
 98         HtmlTree div = new HtmlTree(HtmlTag.DIV);
 99         div.setStyle(HtmlStyle.header);
100         if (configuration.showModules) {
101             ModuleElement mdle = configuration.docEnv.getElementUtils().getModuleOf(annotationType);
102             Content typeModuleLabel = HtmlTree.SPAN(HtmlStyle.moduleLabelInType, contents.moduleLabel);
103             Content moduleNameDiv = HtmlTree.DIV(HtmlStyle.subTitle, typeModuleLabel);
104             moduleNameDiv.add(Contents.SPACE);
105             moduleNameDiv.add(getModuleLink(mdle, new StringContent(mdle.getQualifiedName())));
106             div.add(moduleNameDiv);
107         }
108         PackageElement pkg = utils.containingPackage(annotationType);
109         if (!pkg.isUnnamed()) {
110             Content typePackageLabel = HtmlTree.SPAN(HtmlStyle.packageLabelInType, contents.packageLabel);
111             Content pkgNameDiv = HtmlTree.DIV(HtmlStyle.subTitle, typePackageLabel);
112             pkgNameDiv.add(Contents.SPACE);
113             Content pkgNameContent = getPackageLink(pkg, new StringContent(utils.getPackageName(pkg)));
114             pkgNameDiv.add(pkgNameContent);
115             div.add(pkgNameDiv);
116         }
117         LinkInfoImpl linkInfo = new LinkInfoImpl(configuration,
118                 LinkInfoImpl.Kind.CLASS_HEADER, annotationType);
119         Content headerContent = new StringContent(header);
120         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
121                 HtmlStyle.title, headerContent);
122         heading.add(getTypeParameterLinks(linkInfo));
123         div.add(heading);
124         mainTree.add(div);
125         return bodyTree;
126     }
127 
128     /**
129      * {@inheritDoc}
130      */
131     @Override
132     public Content getAnnotationContentHeader() {
133         return getContentHeader();
134     }
135 
136     /**
137      * {@inheritDoc}
138      */
139     @Override
140     public void addFooter(Content contentTree) {
141         contentTree.add(MarkerComments.END_OF_CLASS_DATA);
142         Content htmlTree = HtmlTree.FOOTER();
143         navBar.setUserFooter(getUserHeaderFooter(false));
144         htmlTree.add(navBar.getContent(false));
145         addBottom(htmlTree);
146         contentTree.add(htmlTree);
147     }
148 
149     /**
150      * {@inheritDoc}
151      */
152     @Override
153     public void printDocument(Content contentTree) throws DocFileIOException {
154         String description = getDescription("declaration", annotationType);
155         PackageElement pkg = utils.containingPackage(this.annotationType);
156         Content stylesheetContent = getLocalStylesheetContent(pkg);
157         printHtmlDocument(configuration.metakeywords.getMetaKeywords(annotationType),
158                 description, stylesheetContent, contentTree);
159     }
160 
161     /**
162      * {@inheritDoc}
163      */
164     @Override
165     public Content getAnnotationInfoTreeHeader() {
166         return getMemberTreeHeader();
167     }
168 
169     /**
170      * {@inheritDoc}
171      */
172     @Override
173     public Content getAnnotationInfo(Content annotationInfoTree) {
174         return getMemberTree(HtmlStyle.description, annotationInfoTree);
175     }
176 
177     /**
178      * {@inheritDoc}
179      */
180     @Override
181     public void addAnnotationTypeSignature(String modifiers, Content annotationInfoTree) {
182         Content hr = new HtmlTree(HtmlTag.HR);
183         annotationInfoTree.add(hr);
184         Content pre = new HtmlTree(HtmlTag.PRE);
185         addAnnotationInfo(annotationType, pre);
186         pre.add(modifiers);
187         LinkInfoImpl linkInfo = new LinkInfoImpl(configuration,
188                 LinkInfoImpl.Kind.CLASS_SIGNATURE, annotationType);
189         Content annotationName = new StringContent(utils.getSimpleName(annotationType));
190         Content parameterLinks = getTypeParameterLinks(linkInfo);
191         if (configuration.linksource) {
192             addSrcLink(annotationType, annotationName, pre);
193             pre.add(parameterLinks);
194         } else {
195             Content span = HtmlTree.SPAN(HtmlStyle.memberNameLabel, annotationName);
196             span.add(parameterLinks);
197             pre.add(span);
198         }
199         annotationInfoTree.add(pre);
200     }
201 
202     /**
203      * {@inheritDoc}
204      */
205     @Override
206     public void addAnnotationTypeDescription(Content annotationInfoTree) {
207         if (!configuration.nocomment) {
208             if (!utils.getFullBody(annotationType).isEmpty()) {
209                 addInlineComment(annotationType, annotationInfoTree);
210             }
211         }
212     }
213 
214     /**
215      * {@inheritDoc}
216      */
217     @Override
218     public void addAnnotationTypeTagInfo(Content annotationInfoTree) {
219         if (!configuration.nocomment) {
220             addTagsInfo(annotationType, annotationInfoTree);
221         }
222     }
223 
224     /**
225      * {@inheritDoc}
226      */
227     @Override
228     public void addAnnotationTypeDeprecationInfo(Content annotationInfoTree) {
229         List<? extends DocTree> deprs = utils.getBlockTags(annotationType, DocTree.Kind.DEPRECATED);
230         if (utils.isDeprecated(annotationType)) {
231             CommentHelper ch = utils.getCommentHelper(annotationType);
232             Content deprLabel = HtmlTree.SPAN(HtmlStyle.deprecatedLabel, getDeprecatedPhrase(annotationType));
233             Content div = HtmlTree.DIV(HtmlStyle.deprecationBlock, deprLabel);
234             if (!deprs.isEmpty()) {
235 
236                 List<? extends DocTree> commentTags = ch.getDescription(configuration, deprs.get(0));
237                 if (!commentTags.isEmpty()) {
238                     addInlineDeprecatedComment(annotationType, deprs.get(0), div);
239                 }
240             }
241             annotationInfoTree.add(div);
242         }
243     }
244 
245     /**
246      * {@inheritDoc}
247      */
248     @Override
249     public TypeElement getAnnotationTypeElement() {
250         return annotationType;
251     }
