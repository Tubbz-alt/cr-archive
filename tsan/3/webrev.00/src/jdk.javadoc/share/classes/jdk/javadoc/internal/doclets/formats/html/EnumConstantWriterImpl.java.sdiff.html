 44 /**
 45  * Writes enum constant documentation in HTML format.
 46  *
 47  *  <p><b>This is NOT part of any supported API.
 48  *  If you write code that depends on this, you do so at your own risk.
 49  *  This code and its internal interfaces are subject to change or
 50  *  deletion without notice.</b>
 51  *
 52  * @author Jamie Ho
 53  * @author Bhavesh Patel (Modified)
 54  */
 55 public class EnumConstantWriterImpl extends AbstractMemberWriter
 56     implements EnumConstantWriter, MemberSummaryWriter {
 57 
 58     public EnumConstantWriterImpl(SubWriterHolderWriter writer, TypeElement typeElement) {
 59         super(writer, typeElement);
 60     }
 61 
 62     public EnumConstantWriterImpl(SubWriterHolderWriter writer) {
 63         super(writer);
 64     }
 65 
 66     /**
 67      * {@inheritDoc}
 68      */
 69     @Override
 70     public Content getMemberSummaryHeader(TypeElement typeElement,
 71             Content memberSummaryTree) {
 72         memberSummaryTree.add(MarkerComments.START_OF_ENUM_CONSTANT_SUMMARY);
 73         Content memberTree = writer.getMemberTreeHeader();
 74         writer.addSummaryHeader(this, typeElement, memberTree);
 75         return memberTree;
 76     }
 77 
 78     /**
 79      * {@inheritDoc}
 80      */
 81     @Override
 82     public void addMemberTree(Content memberSummaryTree, Content memberTree) {
 83         writer.addMemberTree(memberSummaryTree, memberTree);

 84     }
 85 
 86     /**
 87      * {@inheritDoc}
 88      */
 89     @Override
 90     public Content getEnumConstantsDetailsTreeHeader(TypeElement typeElement,
 91             Content memberDetailsTree) {
 92         memberDetailsTree.add(MarkerComments.START_OF_ENUM_CONSTANT_DETAILS);
 93         Content enumConstantsDetailsTree = writer.getMemberTreeHeader();
 94         enumConstantsDetailsTree.add(links.createAnchor(
 95                 SectionName.ENUM_CONSTANT_DETAIL));
 96         Content heading = HtmlTree.HEADING(Headings.TypeDeclaration.DETAILS_HEADING,
 97                 contents.enumConstantDetailLabel);
 98         enumConstantsDetailsTree.add(heading);
 99         return enumConstantsDetailsTree;
100     }
101 
102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public Content getEnumConstantsTreeHeader(VariableElement enumConstant,
107             Content enumConstantsDetailsTree) {
108         enumConstantsDetailsTree.add(links.createAnchor(name(enumConstant)));
109         Content enumConstantsTree = writer.getMemberTreeHeader();
110         Content heading = new HtmlTree(Headings.TypeDeclaration.MEMBER_HEADING);
111         heading.add(name(enumConstant));
112         enumConstantsTree.add(heading);
113         return enumConstantsTree;

114     }
115 
116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     public Content getSignature(VariableElement enumConstant) {
121         Content pre = new HtmlTree(HtmlTag.PRE);
122         writer.addAnnotationInfo(enumConstant, pre);
123         addModifiers(enumConstant, pre);
124         Content enumConstantLink = writer.getLink(new LinkInfoImpl(
125                 configuration, LinkInfoImpl.Kind.MEMBER, enumConstant.asType()));
126         pre.add(enumConstantLink);
127         pre.add(" ");
128         if (configuration.linksource) {
129             Content enumConstantName = new StringContent(name(enumConstant));
130             writer.addSrcLink(enumConstant, enumConstantName, pre);
131         } else {
132             addName(name(enumConstant), pre);
133         }
134         return pre;
135     }
136 
137     /**
138      * {@inheritDoc}
139      */
140     @Override
141     public void addDeprecated(VariableElement enumConstant, Content enumConstantsTree) {
142         addDeprecatedInfo(enumConstant, enumConstantsTree);
143     }
144 
145     /**
146      * {@inheritDoc}
147      */
148     @Override
149     public void addComments(VariableElement enumConstant, Content enumConstantsTree) {
150         addComment(enumConstant, enumConstantsTree);
151     }
152 
153     /**
154      * {@inheritDoc}
155      */
156     @Override
157     public void addTags(VariableElement enumConstant, Content enumConstantsTree) {
158         writer.addTagsInfo(enumConstant, enumConstantsTree);
159     }
160 
161     /**
162      * {@inheritDoc}
163      */
164     @Override
165     public Content getEnumConstantsDetails(Content enumConstantsDetailsTree) {
166         return HtmlTree.SECTION(getMemberTree(enumConstantsDetailsTree));




167     }
168 
169     /**
170      * {@inheritDoc}
171      */
172     @Override
173     public Content getEnumConstants(Content enumConstantsTree,
174             boolean isLastContent) {
175         return getMemberTree(enumConstantsTree, isLastContent);
176     }
177 
178     /**
179      * {@inheritDoc}
180      */
181     @Override
182     public void addSummaryLabel(Content memberTree) {
183         Content label = HtmlTree.HEADING(Headings.TypeDeclaration.SUMMARY_HEADING,
184                 contents.enumConstantSummary);
185         memberTree.add(label);
186     }
187 
188     /**
189      * {@inheritDoc}
190      */
191     @Override
192     public TableHeader getSummaryTableHeader(Element member) {
193         return new TableHeader(contents.enumConstantLabel, contents.descriptionLabel);
194     }
195 
196     /**
197      * {@inheritDoc}
198      */
199     @Override
200     protected Table createSummaryTable() {
201         return new Table(HtmlStyle.memberSummary)
202                 .setCaption(contents.getContent("doclet.Enum_Constants"))
203                 .setHeader(getSummaryTableHeader(typeElement))
204                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);
205     }
206 
207     /**
208      * {@inheritDoc}
209      */
210     @Override
211     public void addSummaryAnchor(TypeElement typeElement, Content memberTree) {
212         memberTree.add(links.createAnchor(SectionName.ENUM_CONSTANT_SUMMARY));
213     }
214 
215     /**
216      * {@inheritDoc}
217      */
218     @Override
219     public void addInheritedSummaryAnchor(TypeElement typeElement, Content inheritedTree) {
220     }
221

224      */
225     @Override

227     }

229     /**
230      * {@inheritDoc}
231      */
232     @Override
233     protected void addSummaryLink(LinkInfoImpl.Kind context, TypeElement typeElement, Element member,
234             Content tdSummary) {
235         Content memberLink = HtmlTree.SPAN(HtmlStyle.memberNameLink,
236                 writer.getDocLink(context, member, name(member), false));
237         Content code = HtmlTree.CODE(memberLink);
238         tdSummary.add(code);
239     }
240 
241     /**
242      * {@inheritDoc}
243      */
244     @Override
245     protected void addInheritedSummaryLink(TypeElement typeElement, Element member, Content linksTree) {
246     }
247 
248     /**
249      * {@inheritDoc}
250      */
251     @Override
252     protected void addSummaryType(Element member, Content tdSummaryType) {
253         //Not applicable.
254     }
255 
256     /**
257      * {@inheritDoc}
258      */
259     @Override
260     protected Content getDeprecatedLink(Element member) {
261         String name = utils.getFullyQualifiedName(member) + "." + member.getSimpleName();
262         return writer.getDocLink(LinkInfoImpl.Kind.MEMBER, member, name);
263     }

 59         super(writer);

 61 



 62     @Override
 63     public Content getMemberSummaryHeader(TypeElement typeElement,
 64             Content memberSummaryTree) {
 65         memberSummaryTree.add(MarkerComments.START_OF_ENUM_CONSTANT_SUMMARY);
 66         Content memberTree = new ContentBuilder();
 67         writer.addSummaryHeader(this, memberTree);
 68         return memberTree;
 69     }
 70 



 71     @Override
 72     public void addMemberTree(Content memberSummaryTree, Content memberTree) {
 73         writer.addMemberTree(HtmlStyle.constantsSummary,
 74                 SectionName.ENUM_CONSTANT_SUMMARY, memberSummaryTree, memberTree);
 75     }
 76 



 77     @Override
 78     public Content getEnumConstantsDetailsTreeHeader(TypeElement typeElement,
 79             Content memberDetailsTree) {
 80         memberDetailsTree.add(MarkerComments.START_OF_ENUM_CONSTANT_DETAILS);
 81         Content enumConstantsDetailsTree = new ContentBuilder();


 82         Content heading = HtmlTree.HEADING(Headings.TypeDeclaration.DETAILS_HEADING,
 83                 contents.enumConstantDetailLabel);
 84         enumConstantsDetailsTree.add(heading);
 85         return enumConstantsDetailsTree;
 86     }
 87 



 88     @Override
 89     public Content getEnumConstantsTreeHeader(VariableElement enumConstant,
 90             Content enumConstantsDetailsTree) {
 91         Content enumConstantsTree = new ContentBuilder();
 92         HtmlTree heading = new HtmlTree(Headings.TypeDeclaration.MEMBER_HEADING,
 93                 new StringContent(name(enumConstant)));

 94         enumConstantsTree.add(heading);
 95         return HtmlTree.SECTION(HtmlStyle.detail, enumConstantsTree)
 96                 .setId(name(enumConstant));
 97     }
 98 



 99     @Override
100     public Content getSignature(VariableElement enumConstant) {
101         return new MemberSignature(enumConstant)
102                 .addType(enumConstant.asType())
103                 .toContent();











104     }
105 



106     @Override
107     public void addDeprecated(VariableElement enumConstant, Content enumConstantsTree) {
108         addDeprecatedInfo(enumConstant, enumConstantsTree);
109     }
110 



111     @Override
112     public void addComments(VariableElement enumConstant, Content enumConstantsTree) {
113         addComment(enumConstant, enumConstantsTree);
114     }
115 



116     @Override
117     public void addTags(VariableElement enumConstant, Content enumConstantsTree) {
118         writer.addTagsInfo(enumConstant, enumConstantsTree);
119     }
120 



121     @Override
122     public Content getEnumConstantsDetails(Content enumConstantsDetailsTreeHeader,
123             Content enumConstantsDetailsTree) {
124         Content enumConstantsDetails =
125                 new ContentBuilder(enumConstantsDetailsTreeHeader, enumConstantsDetailsTree);
126         return getMemberTree(HtmlTree.SECTION(HtmlStyle.constantDetails, enumConstantsDetails)
127                 .setId(SectionName.ENUM_CONSTANT_DETAIL.getName()));
128     }

133     }
