26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;
32 import jdk.javadoc.doclet.Taglet.Location;
33 import jdk.javadoc.internal.doclets.toolkit.Content;
34 


35 /**
36  * An inline Taglet used to index word or a phrase.
37  * The enclosed text is interpreted as not containing HTML markup or
38  * nested javadoc tags.


39  */
40 
41 public class IndexTaglet extends BaseTaglet {
42 
43     IndexTaglet() {
44         super(DocTree.Kind.INDEX, true, EnumSet.allOf(Location.class));
45     }
46 
47     @Override
48     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
49         return writer.indexTagOutput(element, tag);
50     }
51 }
