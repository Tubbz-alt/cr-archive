26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;

32 import jdk.javadoc.internal.doclets.toolkit.Content;
33 
34 import static com.sun.source.doctree.DocTree.Kind.DOC_ROOT;
35 
36 /**
37  * An inline Taglet representing {&#064;docRoot}.  This taglet is
38  * used to get the relative path to the document's root output
39  * directory.
40  *
41  *  <p><b>This is NOT part of any supported API.
42  *  If you write code that depends on this, you do so at your own risk.
43  *  This code and its internal interfaces are subject to change or
44  *  deletion without notice.</b>
45  *
46  * @author Jamie Ho
47  * @author Doug Kramer
48  */
49 
50 public class DocRootTaglet extends BaseTaglet {
51 
52     /**
53      * Construct a new DocRootTaglet.
54      */
55     public DocRootTaglet() {
56         super(DOC_ROOT.tagName, true, EnumSet.allOf(Site.class));
57     }
58 
59     @Override
60     public Content getTagletOutput(Element holder, DocTree tag, TagletWriter writer) {
61         return writer.getDocRootOutput();
62     }
63 }
