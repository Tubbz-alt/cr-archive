 26 package jdk.javadoc.internal.doclets.toolkit.util;
 27 
 28 import java.io.*;
 29 
 30 import javax.lang.model.element.ModuleElement;
 31 import javax.lang.model.element.PackageElement;
 32 
 33 import jdk.javadoc.doclet.DocletEnvironment;
 34 import jdk.javadoc.internal.doclets.toolkit.BaseConfiguration;

 35 
 36 
 37 /**
 38  * Write out the element index.
 39  *
 40  *  <p><b>This is NOT part of any supported API.
 41  *  If you write code that depends on this, you do so at your own risk.
 42  *  This code and its internal interfaces are subject to change or
 43  *  deletion without notice.</b>
 44  *
 45  * @author Atul M Dambalkar
 46  */
 47 public class ElementListWriter {
 48 
 49     private final BaseConfiguration configuration;

 50     private final Utils utils;
 51     private final DocFile file;
 52 
 53     /**
 54      * Constructor.
 55      *
 56      * @param configuration the current configuration of the doclet.
 57      */
 58     public ElementListWriter(BaseConfiguration configuration) {
 59         file = DocFile.createFileForOutput(configuration, DocPaths.ELEMENT_LIST);
 60         this.configuration = configuration;

 61         this.utils = configuration.utils;
 62     }
 63 
 64     /**
 65      * Generate the element index.
 66      *
 67      * @param configuration the current configuration of the doclet.
 68      * @throws DocFileIOException if there is a problem writing the output
 69      */
 70     public static void generate(BaseConfiguration configuration) throws DocFileIOException {
 71         ElementListWriter elemgen = new ElementListWriter(configuration);
 72         elemgen.generateElementListFile(configuration.docEnv);
 73     }
 74 
 75     protected void generateElementListFile(DocletEnvironment docEnv) throws DocFileIOException {
 76         try (BufferedWriter out = new BufferedWriter(file.openWriter())) {
 77             if (configuration.showModules) {
 78                 for (ModuleElement mdle : configuration.modulePackages.keySet()) {
 79                     if (!(configuration.nodeprecated && utils.isDeprecated(mdle))) {
 80                         out.write(DocletConstants.MODULE_PREFIX + mdle.toString());
 81                         out.newLine();
 82                         for (PackageElement pkg : configuration.modulePackages.get(mdle)) {
 83                             out.write(pkg.toString());
 84                             out.newLine();
 85                         }
 86                     }
 87                 }
 88             } else {
 89                 for (PackageElement pkg : configuration.packages) {
 90                     // if the -nodeprecated option is set and the package is marked as
 91                     // deprecated, do not include it in the packages list.
 92                     if (!(configuration.nodeprecated && utils.isDeprecated(pkg))) {
 93                         out.write(pkg.toString());
 94                         out.newLine();
 95                     }
 96                 }
 97             }
 98         } catch (IOException e) {
 99             throw new DocFileIOException(file, DocFileIOException.Mode.WRITE, e);
100         }
101     }
102 }
