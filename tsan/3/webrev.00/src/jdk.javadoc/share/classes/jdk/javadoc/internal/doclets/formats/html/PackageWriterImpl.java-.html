 59  *  deletion without notice.</b>

 61  * @author Atul M Dambalkar
 62  * @author Bhavesh Patel (Modified)
 63  */

 65     implements PackageSummaryWriter {
 66 
 67     /**
 68      * The package being documented.
 69      */
 70     protected PackageElement packageElement;
 71

 73      * The HTML tree for main tag.
 74      */
 75     protected HtmlTree mainTree = HtmlTree.MAIN();
 76 
 77     /**
 78      * The HTML tree for section tag.
 79      */
 80     protected HtmlTree sectionTree = HtmlTree.SECTION();
 81 
 82     private final Navigation navBar;
 83 
 84     /**
 85      * Constructor to construct PackageWriter object and to generate
 86      * "package-summary.html" file in the respective package directory.
 87      * For example for package "java.lang" this will generate file
 88      * "package-summary.html" file in the "java/lang" directory. It will also
 89      * create "java/lang" directory in the current or the destination directory
 90      * if it doesn't exist.
 91      *
 92      * @param configuration the configuration of the doclet.
 93      * @param packageElement    PackageElement under consideration.
 94      */
 95     public PackageWriterImpl(HtmlConfiguration configuration, PackageElement packageElement) {
 96         super(configuration,
 97                 configuration.docPaths.forPackage(packageElement)
 98                 .resolve(DocPaths.PACKAGE_SUMMARY));
 99         this.packageElement = packageElement;
100         this.navBar = new Navigation(packageElement, configuration, fixedNavDiv, PageMode.PACKAGE, path);
101     }
102 
103     /**
104      * {@inheritDoc}
105      */
106     @Override
107     public Content getPackageHeader(String heading) {
108         HtmlTree bodyTree = getBody(true, getWindowTitle(utils.getPackageName(packageElement)));
109         HtmlTree htmlTree = HtmlTree.HEADER();
110         addTop(htmlTree);
111         Content linkContent = getModuleLink(utils.elementUtils.getModuleOf(packageElement),
112                 contents.moduleLabel);
113         navBar.setNavLinkModule(linkContent);
114         navBar.setUserHeader(getUserHeaderFooter(true));
115         htmlTree.add(navBar.getContent(true));
116         bodyTree.add(htmlTree);
117         HtmlTree div = new HtmlTree(HtmlTag.DIV);
118         div.setStyle(HtmlStyle.header);
119         if (configuration.showModules) {
120             ModuleElement mdle = configuration.docEnv.getElementUtils().getModuleOf(packageElement);
121             Content classModuleLabel = HtmlTree.SPAN(HtmlStyle.moduleLabelInPackage, contents.moduleLabel);
122             Content moduleNameDiv = HtmlTree.DIV(HtmlStyle.subTitle, classModuleLabel);
123             moduleNameDiv.add(Contents.SPACE);
124             moduleNameDiv.add(getModuleLink(mdle,
125                     new StringContent(mdle.getQualifiedName().toString())));
126             div.add(moduleNameDiv);
127         }
128         Content annotationContent = new HtmlTree(HtmlTag.P);
129         addAnnotationInfo(packageElement, annotationContent);
130         div.add(annotationContent);
131         Content tHeading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
132                 HtmlStyle.title, contents.packageLabel);
133         tHeading.add(Contents.SPACE);
134         Content packageHead = new StringContent(heading);
135         tHeading.add(packageHead);
136         div.add(tHeading);
137         mainTree.add(div);
138         return bodyTree;
139     }

151     /**
152      * Add the package deprecation information to the documentation tree.
153      *
154      * @param div the content tree to which the deprecation information will be added
155      */
156     public void addDeprecationInfo(Content div) {
157         List<? extends DocTree> deprs = utils.getBlockTags(packageElement, DocTree.Kind.DEPRECATED);
158         if (utils.isDeprecated(packageElement)) {
159             CommentHelper ch = utils.getCommentHelper(packageElement);
160             HtmlTree deprDiv = new HtmlTree(HtmlTag.DIV);
161             deprDiv.setStyle(HtmlStyle.deprecationBlock);
162             Content deprPhrase = HtmlTree.SPAN(HtmlStyle.deprecatedLabel, getDeprecatedPhrase(packageElement));
163             deprDiv.add(deprPhrase);
164             if (!deprs.isEmpty()) {
165                 List<? extends DocTree> commentTags = ch.getDescription(configuration, deprs.get(0));
166                 if (!commentTags.isEmpty()) {
167                     addInlineDeprecatedComment(packageElement, deprs.get(0), deprDiv);
168                 }
169             }
170             div.add(deprDiv);
171         }
172     }

238     public void addClassesSummary(SortedSet<TypeElement> classes, String label,
239             TableHeader tableHeader, Content summaryContentTree) {
240         if(!classes.isEmpty()) {
241             Table table = new Table(HtmlStyle.typeSummary)
242                     .setCaption(getTableCaption(new StringContent(label)))
243                     .setHeader(tableHeader)
244                     .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);
245 
246             for (TypeElement klass : classes) {
247                 if (!utils.isCoreClass(klass) || !configuration.isGeneratedDoc(klass)) {
248                     continue;
249                 }
250                 Content classLink = getLink(new LinkInfoImpl(
251                         configuration, LinkInfoImpl.Kind.PACKAGE, klass));
252                 ContentBuilder description = new ContentBuilder();
253                 if (utils.isDeprecated(klass)) {
254                     description.add(getDeprecatedPhrase(klass));
255                     List<? extends DocTree> tags = utils.getDeprecatedTrees(klass);
256                     if (!tags.isEmpty()) {
257                         addSummaryDeprecatedComment(klass, tags.get(0), description);
258                     }
259                 } else {
260                     addSummaryComment(klass, description);
261                 }
262                 table.addRow(classLink, description);
263             }
264             Content li = HtmlTree.LI(HtmlStyle.blockList, table.toContent());
265             summaryContentTree.add(li);
266         }
267     }

269     /**
270      * {@inheritDoc}
271      */
272     @Override
273     public void addPackageDescription(Content packageContentTree) {
274         if (!utils.getBody(packageElement).isEmpty()) {
275             Content tree = sectionTree;
276             tree.add(links.createAnchor(SectionName.PACKAGE_DESCRIPTION));
277             addDeprecationInfo(tree);
278             addInlineComment(packageElement, tree);
279         }
280     }

289         packageContentTree.add(sectionTree);
290     }

301     /**
302      * {@inheritDoc}
303      */
304     @Override
305     public void addPackageFooter(Content contentTree) {
306         Content htmlTree = HtmlTree.FOOTER();
307         navBar.setUserFooter(getUserHeaderFooter(false));
308         htmlTree.add(navBar.getContent(false));
309         addBottom(htmlTree);
310         contentTree.add(htmlTree);
311     }

313     /**
314      * {@inheritDoc}
315      */
316     @Override
317     public void printDocument(Content contentTree) throws DocFileIOException {
318         String description = getDescription("declaration", packageElement);
319         Content stylesheetContent = getLocalStylesheetContent(packageElement);
320         printHtmlDocument(configuration.metakeywords.getMetaKeywords(packageElement),
321                 description, stylesheetContent, contentTree);
322     }
