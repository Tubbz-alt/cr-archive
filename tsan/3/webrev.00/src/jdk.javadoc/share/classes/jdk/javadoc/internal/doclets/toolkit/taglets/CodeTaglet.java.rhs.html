26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 
30 import javax.lang.model.element.Element;
31 
32 import com.sun.source.doctree.DocTree;
33 import jdk.javadoc.doclet.Taglet.Location;
34 import jdk.javadoc.internal.doclets.toolkit.Content;
35 


36 /**
37  * An inline Taglet used to denote literal code fragments.
38  * The enclosed text is interpreted as not containing HTML markup or
39  * nested javadoc tags, and is rendered in a font suitable for code.
40  *
41  * <p> The tag {@code {@code ...}} is equivalent to
42  * {@code <code>{@literal ...}</code>}.
43  * For example, the text:
44  * <blockquote>  The type {@code {@code List<P>}}  </blockquote>
45  * displays as:
46  * <blockquote>  The type {@code List<P>}  </blockquote>
47  *
48  *  <p><b>This is NOT part of any supported API.
49  *  If you write code that depends on this, you do so at your own risk.
50  *  This code and its internal interfaces are subject to change or
51  *  deletion without notice.</b>


52  */
53 
54 public class CodeTaglet extends BaseTaglet {
55 
56     CodeTaglet() {
57         super(DocTree.Kind.CODE, true, EnumSet.allOf(Location.class));
58     }
59 
60     @Override
61     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
62         return writer.codeTagOutput(element, tag);
63     }
64 }
