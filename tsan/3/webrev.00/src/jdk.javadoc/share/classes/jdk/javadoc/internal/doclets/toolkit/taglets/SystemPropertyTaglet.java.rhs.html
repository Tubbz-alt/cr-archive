28 import com.sun.source.doctree.DocTree;
29 import jdk.javadoc.doclet.Taglet.Location;
30 import jdk.javadoc.internal.doclets.toolkit.Content;
31 
32 import javax.lang.model.element.Element;
33 import java.util.EnumSet;
34 


35 /**
36  * A taglet that represents the {@code @systemProperty} tag.
37  *
38  *  <p><b>This is NOT part of any supported API.
39  *  If you write code that depends on this, you do so at your own risk.
40  *  This code and its internal interfaces are subject to change or
41  *  deletion without notice.</b>
42  */
43 
44 public class SystemPropertyTaglet extends BaseTaglet {
45 
46     SystemPropertyTaglet(){
47         super(DocTree.Kind.SYSTEM_PROPERTY, true, EnumSet.of(Location.CONSTRUCTOR, Location.METHOD, Location.FIELD,
48                 Location.PACKAGE, Location.MODULE, Location.TYPE));
49     }
50 
51     @Override
52     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
53         return writer.systemPropertyTagOutput(element, tag);
54     }
55 }
