26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;
32 import jdk.javadoc.internal.doclets.toolkit.Content;
33 
34 /**
35  * An abstract class that implements the {@link Taglet} interface and
36  * serves as a base for JavaFX property getter and setter taglets.
37  *
38  *  <p><b>This is NOT part of any supported API.
39  *  If you write code that depends on this, you do so at your own risk.
40  *  This code and its internal interfaces are subject to change or
41  *  deletion without notice.</b>
42  *
43  */
44 public abstract class BasePropertyTaglet extends BaseTaglet {
45 
46     public BasePropertyTaglet(String name) {
47         super(name, false, EnumSet.of(Site.METHOD));
48     }
49 
50     /**
51      * This method returns the text to be put in the resulting javadoc before
52      * the property name.
53      *
54      * @param tagletWriter the taglet writer for output
55      * @return the string to be put in the resulting javadoc.
56      */
57     abstract String getText(TagletWriter tagletWriter);
58 
59     /**
60      * Given the <code>Tag</code> representation of this custom
61      * tag, return its string representation, which is output
62      * to the generated page.
63      * @param element
64      * @param tag the <code>Tag</code> representation of this custom tag.
65      * @param tagletWriter the taglet writer for output.
66      * @return the TagletOutput representation of this <code>Tag</code>.
67      */
68     public Content getTagletOutput(Element element, DocTree tag, TagletWriter tagletWriter) {
69         return tagletWriter.propertyTagOutput(element, tag, getText(tagletWriter));
70     }
71 }
