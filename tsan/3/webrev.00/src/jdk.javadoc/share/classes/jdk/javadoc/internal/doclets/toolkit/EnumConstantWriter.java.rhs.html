 26 package jdk.javadoc.internal.doclets.toolkit;
 27 


 28 import javax.lang.model.element.TypeElement;
 29 import javax.lang.model.element.VariableElement;
 30 
 31 /**
 32  * The interface for writing enum constant output.
 33  *
 34  *  <p><b>This is NOT part of any supported API.
 35  *  If you write code that depends on this, you do so at your own risk.
 36  *  This code and its internal interfaces are subject to change or
 37  *  deletion without notice.</b>



 38  */
 39 
 40 public interface EnumConstantWriter {
 41 
 42     /**
 43      * Get the enum constants details tree header.
 44      *
 45      * @param typeElement the class being documented
 46      * @param memberDetailsTree the content tree representing member details
 47      * @return content tree for the enum constants details header
 48      */
 49     Content getEnumConstantsDetailsTreeHeader(TypeElement typeElement,
 50                                               Content memberDetailsTree);
 51 
 52     /**
 53      * Get the enum constants documentation tree header.
 54      *
 55      * @param enumConstant the enum constant being documented
 56      * @param enumConstantsDetailsTree the content tree representing enum constant details
 57      * @return content tree for the enum constant documentation header
 58      */
 59     Content getEnumConstantsTreeHeader(VariableElement enumConstant,
 60                                        Content enumConstantsDetailsTree);
 61 
 62     /**
 63      * Get the signature for the given enum constant.
 64      *
 65      * @param enumConstant the enum constant being documented
 66      * @return content tree for the enum constant signature
 67      */
 68     Content getSignature(VariableElement enumConstant);
 69 
 70     /**
 71      * Add the deprecated output for the given enum constant.
 72      *
 73      * @param enumConstant the enum constant being documented
 74      * @param enumConstantsTree content tree to which the deprecated information will be added
 75      */
 76     void addDeprecated(VariableElement enumConstant, Content enumConstantsTree);
 77 
 78     /**
 79      * Add the comments for the given enum constant.
 80      *
 81      * @param enumConstant the enum constant being documented
 82      * @param enumConstantsTree the content tree to which the comments will be added
 83      */
 84     void addComments(VariableElement enumConstant, Content enumConstantsTree);
 85 
 86     /**
 87      * Add the tags for the given enum constant.
 88      *
 89      * @param enumConstant the enum constant being documented
 90      * @param enumConstantsTree the content tree to which the tags will be added
 91      */
 92     void addTags(VariableElement enumConstant, Content enumConstantsTree);
 93 
 94     /**
 95      * Get the enum constants details tree.
 96      *
 97      * @param memberDetailsTree the content tree representing member details
 98      * @return content tree for the enum constant details
 99      */
100     Content getEnumConstantsDetails(Content memberDetailsTreeHeader, Content memberDetailsTree);
101 
102     /**
103      * Get the enum constants documentation.
104      *
105      * @param enumConstantsTree the content tree representing enum constants documentation

106      * @return content tree for the enum constants documentation
107      */
108     Content getEnumConstants(Content enumConstantsTree);
109 
110     /**
111      * Gets the member header tree.
112      *
113      * @return a content tree for the member header
114      */
115     Content getMemberTreeHeader();
116 }
