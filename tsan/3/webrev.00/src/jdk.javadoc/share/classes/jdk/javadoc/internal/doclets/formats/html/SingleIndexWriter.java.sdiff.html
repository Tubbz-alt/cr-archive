 41 /**
 42  * Generate only one index file for all the Member Names with Indexing in
 43  * Unicode Order. The name of the generated file is "index-all.html" and it is
 44  * generated in current or the destination directory.
 45  *
 46  *  <p><b>This is NOT part of any supported API.
 47  *  If you write code that depends on this, you do so at your own risk.
 48  *  This code and its internal interfaces are subject to change or
 49  *  deletion without notice.</b>
 50  *
 51  * @see java.lang.Character
 52  * @author Atul M Dambalkar
 53  * @author Bhavesh Patel (Modified)
 54  */
 55 public class SingleIndexWriter extends AbstractIndexWriter {
 56 
 57     private Set<Character> elements;
 58 
 59     /**
 60      * Construct the SingleIndexWriter with filename "index-all.html" and the
 61      * {@link IndexBuilder}
 62      *
 63      * @param configuration the configuration for this doclet
 64      * @param filename     Name of the index file to be generated.
 65      * @param indexbuilder Unicode based Index from {@link IndexBuilder}
 66      */
 67     public SingleIndexWriter(HtmlConfiguration configuration,
 68                              DocPath filename,
 69                              IndexBuilder indexbuilder) {
 70         super(configuration, filename, indexbuilder);
 71     }
 72 
 73     /**
 74      * Generate single index file, for all Unicode characters.
 75      *
 76      * @param configuration the configuration for this doclet
 77      * @param indexbuilder IndexBuilder built by {@link IndexBuilder}
 78      * @throws DocFileIOException if there is a problem generating the index
 79      */
 80     public static void generate(HtmlConfiguration configuration,
 81                                 IndexBuilder indexbuilder) throws DocFileIOException {
 82         DocPath filename = DocPaths.INDEX_ALL;
 83         SingleIndexWriter indexgen = new SingleIndexWriter(configuration,
 84                                          filename, indexbuilder);
 85         indexgen.generateIndexFile();
 86     }
 87 
 88     /**
 89      * Generate the contents of each index file, with Header, Footer,
 90      * Member Field, Method and Constructor Description.
 91      * @throws DocFileIOException if there is a problem generating the index
 92      */
 93     protected void generateIndexFile() throws DocFileIOException {
 94         String title = resources.getText("doclet.Window_Single_Index");
 95         HtmlTree body = getBody(true, getWindowTitle(title));
 96         HtmlTree header = HtmlTree.HEADER();
 97         addTop(header);
 98         navBar.setUserHeader(getUserHeaderFooter(true));
 99         header.add(navBar.getContent(true));
100         body.add(header);
101         HtmlTree main = HtmlTree.MAIN();
102         main.add(HtmlTree.DIV(HtmlStyle.header,
103                 HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING,
104                         contents.getContent("doclet.Index"))));
105         HtmlTree divTree = new HtmlTree(HtmlTag.DIV);
106         divTree.setStyle(HtmlStyle.contentContainer);
107         elements = new TreeSet<>(indexbuilder.getIndexMap().keySet());
108         elements.addAll(configuration.tagSearchIndexKeys);
109         addLinksForIndexes(divTree);
110         for (Character unicode : elements) {
111             if (configuration.tagSearchIndexMap.get(unicode) == null) {
112                 addContents(unicode, indexbuilder.getMemberList(unicode), divTree);
113             } else if (indexbuilder.getMemberList(unicode) == null) {
114                 addSearchContents(unicode, configuration.tagSearchIndexMap.get(unicode), divTree);
115             } else {
116                 addContents(unicode, indexbuilder.getMemberList(unicode),
117                         configuration.tagSearchIndexMap.get(unicode), divTree);
118             }
119         }
120         addLinksForIndexes(divTree);
121         main.add(divTree);
122         body.add(main);
123         HtmlTree footer = HtmlTree.FOOTER();
124         navBar.setUserFooter(getUserHeaderFooter(false));
125         footer.add(navBar.getContent(false));
126         addBottom(footer);
127         body.add(footer);







128         createSearchIndexFiles();
129         printHtmlDocument(null, "index", body);
130     }
131 
132     /**
133      * Add links for all the Index Files per unicode character.
134      *
135      * @param contentTree the content tree to which the links for indexes will be added
136      */
137     protected void addLinksForIndexes(Content contentTree) {
138         for (Object ch : elements) {
139             String unicode = ch.toString();
140             contentTree.add(
141                     links.createLink(getNameForIndex(unicode),
142                             new StringContent(unicode)));
143             contentTree.add(Contents.SPACE);
144         }
145         contentTree.add(new HtmlTree(HtmlTag.BR));
146         contentTree.add(links.createLink(DocPaths.ALLCLASSES_INDEX,
147                 contents.allClassesLabel));
148         if (!configuration.packages.isEmpty()) {
149             contentTree.add(Contents.SPACE);
150             contentTree.add(links.createLink(DocPaths.ALLPACKAGES_INDEX,
151                     contents.allPackagesLabel));





152     }
153 }

 45 /**
 46  * Generate only one index file for all the Member Names with Indexing in
 47  * Unicode Order. The name of the generated file is "index-all.html" and it is
 48  * generated in current or the destination directory.
 49  *
 50  *  <p><b>This is NOT part of any supported API.
 51  *  If you write code that depends on this, you do so at your own risk.
 52  *  This code and its internal interfaces are subject to change or
 53  *  deletion without notice.</b>
 54  *
 55  * @see java.lang.Character


 56  */
 57 public class SingleIndexWriter extends AbstractIndexWriter {
 58 
 59     private Set<Character> elements;
 60 
 61     /**
 62      * Construct the SingleIndexWriter with filename "index-all.html" and the
 63      * {@link IndexBuilder}
 64      *
 65      * @param configuration the configuration for this doclet
 66      * @param filename     Name of the index file to be generated.
 67      * @param indexBuilder Unicode based Index from {@link IndexBuilder}
 68      */
 69     public SingleIndexWriter(HtmlConfiguration configuration,
 70                              DocPath filename,
 71                              IndexBuilder indexBuilder) {
 72         super(configuration, filename, indexBuilder);
 73     }
 74 
 75     /**
 76      * Generate single index file, for all Unicode characters.
 77      *
 78      * @param configuration the configuration for this doclet
 79      * @param indexBuilder IndexBuilder built by {@link IndexBuilder}
 80      * @throws DocFileIOException if there is a problem generating the index
 81      */
 82     public static void generate(HtmlConfiguration configuration,
 83                                 IndexBuilder indexBuilder) throws DocFileIOException {
 84         DocPath filename = DocPaths.INDEX_ALL;
 85         SingleIndexWriter indexgen = new SingleIndexWriter(configuration,
 86                                          filename, indexBuilder);
 87         indexgen.generateIndexFile();
 88     }
 89 
 90     /**
 91      * Generate the contents of each index file, with Header, Footer,
 92      * Member Field, Method and Constructor Description.
 93      * @throws DocFileIOException if there is a problem generating the index
 94      */
 95     protected void generateIndexFile() throws DocFileIOException {
 96         String title = resources.getText("doclet.Window_Single_Index");
 97         HtmlTree body = getBody(getWindowTitle(title));
 98         Content headerContent = new ContentBuilder();
 99         addTop(headerContent);
100         navBar.setUserHeader(getUserHeaderFooter(true));
101         headerContent.add(navBar.getContent(true));





102         HtmlTree divTree = new HtmlTree(HtmlTag.DIV);
103         divTree.setStyle(HtmlStyle.contentContainer);
104         elements = new TreeSet<>(indexBuilder.asMap().keySet());
105         elements.addAll(configuration.tagSearchIndexKeys);
106         addLinksForIndexes(divTree);
107         for (Character unicode : elements) {
108             if (configuration.tagSearchIndexMap.get(unicode) == null) {
109                 addContents(unicode, indexBuilder.getMemberList(unicode), divTree);
110             } else if (indexBuilder.getMemberList(unicode) == null) {
111                 addSearchContents(unicode, configuration.tagSearchIndexMap.get(unicode), divTree);
112             } else {
113                 addContents(unicode, indexBuilder.getMemberList(unicode),
114                             configuration.tagSearchIndexMap.get(unicode), divTree);
115             }
116         }
117         addLinksForIndexes(divTree);


118         HtmlTree footer = HtmlTree.FOOTER();
119         navBar.setUserFooter(getUserHeaderFooter(false));
120         footer.add(navBar.getContent(false));
121         addBottom(footer);
122         body.add(new BodyContents()
123                 .setHeader(headerContent)
124                 .addMainContent(HtmlTree.DIV(HtmlStyle.header,
125                         HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING,
126                                 contents.getContent("doclet.Index"))))
127                 .addMainContent(divTree)
128                 .setFooter(footer)
129                 .toContent());
130         createSearchIndexFiles();
131         printHtmlDocument(null, "index", body);
132     }
133 
134     /**
135      * Add links for all the Index Files per unicode character.
136      *
137      * @param contentTree the content tree to which the links for indexes will be added
138      */
139     protected void addLinksForIndexes(Content contentTree) {
140         for (Object ch : elements) {
141             String unicode = ch.toString();
142             contentTree.add(
143                     links.createLink(getNameForIndex(unicode),
144                             new StringContent(unicode)));
145             contentTree.add(Entity.NO_BREAK_SPACE);
146         }
147         contentTree.add(new HtmlTree(HtmlTag.BR));
148         contentTree.add(links.createLink(DocPaths.ALLCLASSES_INDEX,
149                 contents.allClassesLabel));
150         if (!configuration.packages.isEmpty()) {
151             contentTree.add(getVerticalSeparator());
152             contentTree.add(links.createLink(DocPaths.ALLPACKAGES_INDEX,
153                     contents.allPackagesLabel));
154         }
155         if (!configuration.tagSearchIndex.isEmpty()) {
156             contentTree.add(getVerticalSeparator());
157             contentTree.add(links.createLink(DocPaths.SYSTEM_PROPERTIES, contents.systemPropertiesLabel));
158         }
159     }
