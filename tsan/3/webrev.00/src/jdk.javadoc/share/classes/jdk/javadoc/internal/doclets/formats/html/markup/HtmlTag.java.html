 26 package jdk.javadoc.internal.doclets.formats.html.markup;
 27 
 28 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
 29 
 30 /**
 31  * Enum representing HTML tags.
 32  *
 33  *  <p><b>This is NOT part of any supported API.
 34  *  If you write code that depends on this, you do so at your own risk.
 35  *  This code and its internal interfaces are subject to change or
 36  *  deletion without notice.</b>
 37  */
 38 public enum HtmlTag {
 39     A(BlockType.INLINE, EndTag.END),
 40     BUTTON(BlockType.INLINE, EndTag.END),
 41     BLOCKQUOTE,
 42     BODY(BlockType.OTHER, EndTag.END),
 43     BR(BlockType.INLINE, EndTag.NOEND),
 44     CAPTION,
 45     CODE(BlockType.INLINE, EndTag.END),
 46     DD,
 47     DIV,
 48     DL,
 49     DT,
 50     EM(BlockType.INLINE, EndTag.END),
 51     FOOTER,
 52     H1,
 53     H2,
 54     H3,
 55     H4,
 56     H5,
 57     H6,
 58     HEAD(BlockType.OTHER, EndTag.END),
 59     HEADER,
 60     HR(BlockType.BLOCK, EndTag.NOEND),
 61     HTML(BlockType.OTHER, EndTag.END),
 62     I(BlockType.INLINE, EndTag.END),
 63     IFRAME(BlockType.OTHER, EndTag.END),
 64     IMG(BlockType.INLINE, EndTag.NOEND),
 65     INPUT(BlockType.BLOCK, EndTag.NOEND),
 66     LABEL(BlockType.INLINE, EndTag.END),
 67     LI,
 68     LISTING,
 69     LINK(BlockType.OTHER, EndTag.NOEND),
 70     MAIN,
 71     MENU,
 72     META(BlockType.OTHER, EndTag.NOEND),
 73     NAV,
 74     NOSCRIPT(BlockType.OTHER, EndTag.END),
 75     OL,
 76     P,
 77     PRE,
 78     SCRIPT(BlockType.OTHER, EndTag.END),
 79     SECTION,
 80     SMALL(BlockType.INLINE, EndTag.END),
 81     SPAN(BlockType.INLINE, EndTag.END),
 82     STRONG(BlockType.INLINE, EndTag.END),
 83     SUB(BlockType.INLINE, EndTag.END),
 84     TABLE,
 85     TBODY,
 86     THEAD,
 87     TD,
 88     TH,
 89     TITLE(BlockType.OTHER, EndTag.END),
 90     TR,
 91     UL;
 92 
 93     public final BlockType blockType;
 94     public final EndTag endTag;
 95     public final String value;
 96 
 97     /**
 98      * Enum representing the type of HTML element.
 99      */
100     public enum BlockType {
101         BLOCK,
102         INLINE,
103         OTHER
104     }
105 
106     /**
107      * Enum representing HTML end tag requirement.
108      */
109     public enum EndTag {
110         END,
111         NOEND
112     }
113 
114     HtmlTag() {
115         this(BlockType.BLOCK, EndTag.END);
116     }
117 
118     HtmlTag(BlockType blockType, EndTag endTag) {
119         this.blockType = blockType;
120         this.endTag = endTag;
121         this.value = Utils.toLowerCase(name());
122     }
123 
124     /**
125      * Returns true if the end tag is required. This is specific to the standard
126      * doclet and does not exactly resemble the W3C specifications.
127      *
128      * @return true if end tag needs to be displayed else return false
129      */
130     public boolean endTagRequired() {
131         return (endTag == EndTag.END);
132     }
133 
134     public String toString() {
135         return value;
136     }
137 }
