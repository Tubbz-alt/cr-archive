 44 /**
 45  * Generate Class Hierarchy page for all the Classes in this run.  Use
 46  * ClassTree for building the Tree. The name of
 47  * the generated file is "overview-tree.html" and it is generated in the
 48  * current or the destination directory.
 49  *
 50  *  <p><b>This is NOT part of any supported API.
 51  *  If you write code that depends on this, you do so at your own risk.
 52  *  This code and its internal interfaces are subject to change or
 53  *  deletion without notice.</b>
 54  *
 55  * @author Atul M Dambalkar
 56  * @author Bhavesh Patel (Modified)
 57  */
 58 public class TreeWriter extends AbstractTreeWriter {
 59 
 60     /**
 61      * Packages in this run.
 62      */
 63     SortedSet<PackageElement> packages;
 64 
 65     /**
 66      * True if there are no packages specified on the command line,
 67      * False otherwise.
 68      */
 69     private final boolean classesOnly;
 70 
 71     private final Navigation navBar;
 72 
 73     /**
 74      * Constructor to construct TreeWriter object.
 75      *
 76      * @param configuration the current configuration of the doclet.
 77      * @param filename String filename
 78      * @param classtree the tree being built.
 79      */
 80     public TreeWriter(HtmlConfiguration configuration, DocPath filename, ClassTree classtree) {
 81         super(configuration, filename, classtree);
 82         packages = configuration.packages;
 83         classesOnly = packages.isEmpty();
 84         this.navBar = new Navigation(null, configuration, fixedNavDiv, PageMode.TREE, path);
 85     }
 86 
 87     /**
 88      * Create a TreeWriter object and use it to generate the
 89      * "overview-tree.html" file.
 90      *
 91      * @param configuration the configuration for this doclet
 92      * @param classtree the class tree being documented.
 93      * @throws  DocFileIOException if there is a problem generating the overview tree page
 94      */
 95     public static void generate(HtmlConfiguration configuration,
 96                                 ClassTree classtree) throws DocFileIOException {
 97         DocPath filename = DocPaths.OVERVIEW_TREE;
 98         TreeWriter treegen = new TreeWriter(configuration, filename, classtree);
 99         treegen.generateTreeFile();
100     }
101 
102     /**
103      * Generate the interface hierarchy and class hierarchy.
104      *
105      * @throws DocFileIOException if there is a problem generating the overview tree page
106      */
107     public void generateTreeFile() throws DocFileIOException {
108         HtmlTree body = getTreeHeader();
109         Content headContent = contents.hierarchyForAllPackages;
110         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, false,
111                 HtmlStyle.title, headContent);
112         Content div = HtmlTree.DIV(HtmlStyle.header, heading);
113         addPackageTreeLinks(div);
114         HtmlTree htmlTree = HtmlTree.MAIN();
115         htmlTree.add(div);
116         HtmlTree divTree = new HtmlTree(HtmlTag.DIV);
117         divTree.setStyle(HtmlStyle.contentContainer);
118         addTree(classtree.baseClasses(), "doclet.Class_Hierarchy", divTree);
119         addTree(classtree.baseInterfaces(), "doclet.Interface_Hierarchy", divTree);
120         addTree(classtree.baseAnnotationTypes(), "doclet.Annotation_Type_Hierarchy", divTree);
121         addTree(classtree.baseEnums(), "doclet.Enum_Hierarchy", divTree, true);
122         htmlTree.add(divTree);
123         body.add(htmlTree);
124         htmlTree = HtmlTree.FOOTER();
125         navBar.setUserFooter(getUserHeaderFooter(false));
126         htmlTree.add(navBar.getContent(false));
127         addBottom(htmlTree);
128         body.add(htmlTree);
129         printHtmlDocument(null, "class tree", body);
130     }
131 
132     /**
133      * Add the links to all the package tree files.
134      *
135      * @param contentTree the content tree to which the links will be added
136      */
137     protected void addPackageTreeLinks(Content contentTree) {
138         //Do nothing if only unnamed package is used
139         if (isUnnamedPackage()) {
140             return;
141         }
142         if (!classesOnly) {
143             Content span = HtmlTree.SPAN(HtmlStyle.packageHierarchyLabel,
144                     contents.packageHierarchies);
145             contentTree.add(span);
146             HtmlTree ul = new HtmlTree(HtmlTag.UL);
147             ul.setStyle(HtmlStyle.horizontal);
148             int i = 0;
149             for (PackageElement pkg : packages) {
150                 // If the package name length is 0 or if -nodeprecated option
151                 // is set and the package is marked as deprecated, do not include
152                 // the page in the list of package hierarchies.
153                 if (pkg.isUnnamed() ||
154                         (configuration.nodeprecated && utils.isDeprecated(pkg))) {
155                     i++;
156                     continue;
157                 }
158                 DocPath link = pathString(pkg, DocPaths.PACKAGE_TREE);
159                 Content li = HtmlTree.LI(links.createLink(link,
160                         new StringContent(utils.getPackageName(pkg))));
161                 if (i < packages.size() - 1) {
162                     li.add(", ");
163                 }
164                 ul.add(li);
165                 i++;
166             }
167             contentTree.add(ul);
168         }
169     }
170 
171     /**
172      * Get the tree header.
173      *
174      * @return a content tree for the tree header
175      */
176     protected HtmlTree getTreeHeader() {
177         String title = resources.getText("doclet.Window_Class_Hierarchy");
178         HtmlTree bodyTree = getBody(true, getWindowTitle(title));
179         HtmlTree htmlTree = HtmlTree.HEADER();
180         addTop(htmlTree);
181         navBar.setUserHeader(getUserHeaderFooter(true));
182         htmlTree.add(navBar.getContent(true));
183         bodyTree.add(htmlTree);
184         return bodyTree;
185     }
186 
187     private boolean isUnnamedPackage() {
188         return packages.size() == 1 && packages.first().isUnnamed();
189     }
