 26 package jdk.javadoc.internal.doclets.formats.html;
 27 
 28 
 29 import javax.lang.model.element.Element;
 30 import javax.lang.model.element.ModuleElement;
 31 import javax.lang.model.element.PackageElement;
 32 import javax.lang.model.element.TypeElement;
 33 
 34 import jdk.javadoc.internal.doclets.toolkit.AnnotationTypeFieldWriter;
 35 import jdk.javadoc.internal.doclets.toolkit.AnnotationTypeOptionalMemberWriter;
 36 import jdk.javadoc.internal.doclets.toolkit.AnnotationTypeRequiredMemberWriter;
 37 import jdk.javadoc.internal.doclets.toolkit.AnnotationTypeWriter;
 38 import jdk.javadoc.internal.doclets.toolkit.ClassWriter;
 39 import jdk.javadoc.internal.doclets.toolkit.ConstantsSummaryWriter;
 40 import jdk.javadoc.internal.doclets.toolkit.DocFilesHandler;
 41 import jdk.javadoc.internal.doclets.toolkit.MemberSummaryWriter;
 42 import jdk.javadoc.internal.doclets.toolkit.ModuleSummaryWriter;
 43 import jdk.javadoc.internal.doclets.toolkit.PackageSummaryWriter;
 44 import jdk.javadoc.internal.doclets.toolkit.SerializedFormWriter;
 45 import jdk.javadoc.internal.doclets.toolkit.WriterFactory;
 46 import jdk.javadoc.internal.doclets.toolkit.util.ClassTree;
 47 import jdk.javadoc.internal.doclets.toolkit.util.VisibleMemberTable;
 48 
 49 /**
 50  * The factory that returns HTML writers.
 51  *
 52  *  <p><b>This is NOT part of any supported API.
 53  *  If you write code that depends on this, you do so at your own risk.
 54  *  This code and its internal interfaces are subject to change or
 55  *  deletion without notice.</b>
 56  *
 57  * @author Jamie Ho
 58  */
 59 public class WriterFactoryImpl implements WriterFactory {
 60 
 61     private final HtmlConfiguration configuration;
 62     public WriterFactoryImpl(HtmlConfiguration configuration) {
 63         this.configuration = configuration;
 64     }
 65 
 66     /**
 67      * {@inheritDoc}
 68      */
 69     @Override
 70     public ConstantsSummaryWriter getConstantsSummaryWriter() {
 71         return new ConstantsSummaryWriterImpl(configuration);
 72     }
 73 
 74     /**
 75      * {@inheritDoc}
 76      */
 77     @Override
 78     public PackageSummaryWriter getPackageSummaryWriter(PackageElement packageElement) {
 79         return new PackageWriterImpl(configuration, packageElement);
 80     }
 81 
 82     /**
 83      * {@inheritDoc}
 84      */
 85     public ModuleSummaryWriter getModuleSummaryWriter(ModuleElement mdle) {
 86         return new ModuleWriterImpl(configuration, mdle);
 87     }
 88 
 89     /**
 90      * {@inheritDoc}
 91      */
 92     @Override
 93     public ClassWriter getClassWriter(TypeElement typeElement, ClassTree classTree) {
 94         return new ClassWriterImpl(configuration, typeElement, classTree);
 95     }
 96 
 97     /**
 98      * {@inheritDoc}
 99      */
100     @Override
101     public AnnotationTypeWriter getAnnotationTypeWriter(TypeElement annotationType) {
102         return new AnnotationTypeWriterImpl(configuration, annotationType);
103     }
104 
105     /**
106      * {@inheritDoc}
107      */
108     @Override
109     public AnnotationTypeFieldWriter getAnnotationTypeFieldWriter(
110             AnnotationTypeWriter annotationTypeWriter) {
111         TypeElement te = annotationTypeWriter.getAnnotationTypeElement();
112         return new AnnotationTypeFieldWriterImpl(
113             (SubWriterHolderWriter) annotationTypeWriter, te);
114     }
115 
116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     public AnnotationTypeOptionalMemberWriter getAnnotationTypeOptionalMemberWriter(
121         AnnotationTypeWriter annotationTypeWriter) {
122         TypeElement te = annotationTypeWriter.getAnnotationTypeElement();
123         return new AnnotationTypeOptionalMemberWriterImpl(
124             (SubWriterHolderWriter) annotationTypeWriter, te);
125     }
126 
127     /**
128      * {@inheritDoc}
129      */
130     @Override
131     public AnnotationTypeRequiredMemberWriter getAnnotationTypeRequiredMemberWriter(
132             AnnotationTypeWriter annotationTypeWriter) {
133         TypeElement te = annotationTypeWriter.getAnnotationTypeElement();
134         return new AnnotationTypeRequiredMemberWriterImpl(
135             (SubWriterHolderWriter) annotationTypeWriter, te);
136     }
137 
138     /**
139      * {@inheritDoc}
140      */
141     @Override
142     public EnumConstantWriterImpl getEnumConstantWriter(ClassWriter classWriter) {
143         return new EnumConstantWriterImpl((SubWriterHolderWriter) classWriter,
144                 classWriter.getTypeElement());
145     }
146 
147     /**
148      * {@inheritDoc}
149      */
150     @Override
151     public FieldWriterImpl getFieldWriter(ClassWriter classWriter) {
152         return new FieldWriterImpl((SubWriterHolderWriter) classWriter, classWriter.getTypeElement());
153     }
154 
155     /**
156      * {@inheritDoc}
157      */
158     @Override
159     public PropertyWriterImpl getPropertyWriter(ClassWriter classWriter) {
160         return new PropertyWriterImpl((SubWriterHolderWriter) classWriter,
161                 classWriter.getTypeElement());
162     }
163 
164     /**
165      * {@inheritDoc}
166      */
167     @Override
168     public MethodWriterImpl getMethodWriter(ClassWriter classWriter) {
169         return new MethodWriterImpl((SubWriterHolderWriter) classWriter, classWriter.getTypeElement());
170     }
171 
172     /**
173      * {@inheritDoc}
174      */
175     @Override
176     public ConstructorWriterImpl getConstructorWriter(ClassWriter classWriter) {
177         return new ConstructorWriterImpl((SubWriterHolderWriter) classWriter,
178                 classWriter.getTypeElement());
179     }
180 
181     /**
182      * {@inheritDoc}
183      */
184     @Override
185     public MemberSummaryWriter getMemberSummaryWriter(ClassWriter classWriter,
186             VisibleMemberTable.Kind memberType) {
187         switch (memberType) {
188             case CONSTRUCTORS:
189                 return getConstructorWriter(classWriter);
190             case ENUM_CONSTANTS:
191                 return getEnumConstantWriter(classWriter);
192             case FIELDS:
193                 return getFieldWriter(classWriter);
194             case PROPERTIES:
195                 return getPropertyWriter(classWriter);
196             case INNER_CLASSES:
197                 return new NestedClassWriterImpl((SubWriterHolderWriter)
198                     classWriter, classWriter.getTypeElement());
199             case METHODS:
200                 return getMethodWriter(classWriter);
201             default:
202                 return null;
203         }
204     }
205 
206     /**
207      * {@inheritDoc}
208      */
209     @Override
210     public MemberSummaryWriter getMemberSummaryWriter(AnnotationTypeWriter annotationTypeWriter,
211             VisibleMemberTable.Kind memberType) {
212         switch (memberType) {
213             case ANNOTATION_TYPE_FIELDS:
214                 return (AnnotationTypeFieldWriterImpl)
215                     getAnnotationTypeFieldWriter(annotationTypeWriter);
216             case ANNOTATION_TYPE_MEMBER_OPTIONAL:
217                 return (AnnotationTypeOptionalMemberWriterImpl)
218                     getAnnotationTypeOptionalMemberWriter(annotationTypeWriter);
219             case ANNOTATION_TYPE_MEMBER_REQUIRED:
220                 return (AnnotationTypeRequiredMemberWriterImpl)
221                     getAnnotationTypeRequiredMemberWriter(annotationTypeWriter);
222             default:
223                 return null;
224         }
225     }
226 
227     /**
228      * {@inheritDoc}
229      */
230     @Override
231     public SerializedFormWriter getSerializedFormWriter() {
232         return new SerializedFormWriterImpl(configuration);
233     }
234 
235     /**
236      * {@inheritDoc}
237      */
238     @Override
239     public DocFilesHandler getDocFilesHandler(Element element) {
240         return new DocFilesHandlerImpl(configuration, element);
241     }
242 }
