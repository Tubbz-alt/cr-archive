135     public Content deprecatedTagOutput(Element element) {
136         ContentBuilder result = new ContentBuilder();
137         CommentHelper ch = utils.getCommentHelper(element);
138         List<? extends DocTree> deprs = utils.getBlockTags(element, DocTree.Kind.DEPRECATED);
139         if (utils.isTypeElement(element)) {
140             if (utils.isDeprecated(element)) {
141                 result.add(HtmlTree.SPAN(HtmlStyle.deprecatedLabel,
142                         htmlWriter.getDeprecatedPhrase(element)));
143                 if (!deprs.isEmpty()) {
144                     List<? extends DocTree> commentTags = ch.getDescription(configuration, deprs.get(0));
145                     if (!commentTags.isEmpty()) {
146                         result.add(commentTagsToOutput(null, element, commentTags, false));
147                     }
148                 }
149             }
150         } else {
151             if (utils.isDeprecated(element)) {
152                 result.add(HtmlTree.SPAN(HtmlStyle.deprecatedLabel,
153                         htmlWriter.getDeprecatedPhrase(element)));
154                 if (!deprs.isEmpty()) {
155                     List<? extends DocTree> bodyTags = ch.getBody(configuration, deprs.get(0));
156                     Content body = commentTagsToOutput(null, element, bodyTags, false);
157                     if (!body.isEmpty())
158                         result.add(HtmlTree.DIV(HtmlStyle.deprecationComment, body));
159                 }
160             } else {
161                 Element ee = utils.getEnclosingTypeElement(element);
162                 if (utils.isDeprecated(ee)) {
163                     result.add(HtmlTree.SPAN(HtmlStyle.deprecatedLabel,
164                         htmlWriter.getDeprecatedPhrase(ee)));
165                 }
166             }
167         }
168         return result;
169     }

337     public Content throwsTagOutput(Element element, DocTree throwsTag) {
338         ContentBuilder body = new ContentBuilder();
339         CommentHelper ch = utils.getCommentHelper(element);
340         Element exception = ch.getException(configuration, throwsTag);
341         Content excName;
342         if (exception == null) {



343             excName = new RawHtml(ch.getExceptionName(throwsTag).toString());
344         } else if (exception.asType() == null) {
345             excName = new RawHtml(utils.getFullyQualifiedName(exception));
346         } else {
347             LinkInfoImpl link = new LinkInfoImpl(configuration, LinkInfoImpl.Kind.MEMBER,
348                                                  exception.asType());
349             link.excludeTypeBounds = true;
350             excName = htmlWriter.getLink(link);
351         }
352         body.add(HtmlTree.CODE(excName));
353         List<? extends DocTree> description = ch.getDescription(configuration, throwsTag);
354         Content desc = htmlWriter.commentTagsToContent(throwsTag, element, description, false, inSummary);
355         if (desc != null && !desc.isEmpty()) {
356             body.add(" - ");
357             body.add(desc);
358         }
359         HtmlTree result = HtmlTree.DD(body);
360         return result;
361     }

411     private Content createAnchorAndSearchIndex(Element element, String tagText, String desc){
412         String anchorName = htmlWriter.links.getName(tagText);





413         Content result = null;
414         if (isFirstSentence && inSummary) {
415             result = new StringContent(tagText);
416         } else {
417             result = HtmlTree.A_ID(HtmlStyle.searchTagResult, anchorName, new StringContent(tagText));
418             if (configuration.createindex && !tagText.isEmpty()) {






419                 SearchIndexItem si = new SearchIndexItem();

420                 si.setLabel(tagText);
421                 si.setDescription(desc);
422                 si.setUrl(htmlWriter.path.getPath() + "#" + anchorName);
423                 DocPaths docPaths = configuration.docPaths;
424                 new SimpleElementVisitor9<Void, Void>() {




























425                     @Override
426                     public Void visitVariable(VariableElement e, Void p) {
427                         TypeElement te = utils.getEnclosingTypeElement(e);
428                         si.setHolder(utils.getFullyQualifiedName(e) + "." + utils.getSimpleName(e));
429                         return null;
430                     }
431 
432                     @Override
433                     public Void visitUnknown(Element e, Void p) {
434                         if (e instanceof DocletElement) {
435                             DocletElement de = (DocletElement) e;
436                             switch (de.getSubKind()) {
437                                 case OVERVIEW:
438                                     si.setHolder(resources.getText("doclet.Overview"));
439                                     break;
440                                 case DOCFILE:
441                                     si.setHolder(de.getPackageElement().toString());
442                                     break;
443                                 default:
444                                     throw new IllegalStateException();
445                             }
446                             return null;
447                         } else {
448                             return super.visitUnknown(e, p);
449                         }
450                     }
451 
452                     @Override
453                     protected Void defaultAction(Element e, Void p) {
454                         si.setHolder(utils.getFullyQualifiedName(e));
455                         return null;
456                     }
457                 }.visit(element);
458                 si.setCategory(SearchIndexItem.Category.SEARCH_TAGS);
459                 configuration.tagSearchIndex.add(si);
460             }
461         }
462         return result;
463     }

318         ContentBuilder body = new ContentBuilder();
319         CommentHelper ch = utils.getCommentHelper(element);
320         Element exception = ch.getException(throwsTag);

325         } else if (exception == null) {
326             excName = new RawHtml(ch.getExceptionName(throwsTag).toString());
327         } else if (exception.asType() == null) {
328             excName = new RawHtml(utils.getFullyQualifiedName(exception));

330             LinkInfoImpl link = new LinkInfoImpl(configuration, LinkInfoImpl.Kind.MEMBER,
331                                                  exception.asType());

334         }
335         body.add(HtmlTree.CODE(excName));
336         List<? extends DocTree> description = ch.getDescription(throwsTag);
337         Content desc = htmlWriter.commentTagsToContent(throwsTag, element, description, false, inSummary);
338         if (desc != null && !desc.isEmpty()) {
339             body.add(" - ");
340             body.add(desc);
341         }
342         HtmlTree result = HtmlTree.DD(body);
343         return result;
344     }

390     @SuppressWarnings("preview")
391     private Content createAnchorAndSearchIndex(Element element, String tagText, String desc, boolean isSystemProperty) {
392         Content result = null;
393         if (isFirstSentence && inSummary) {
394             result = new StringContent(tagText);
395         } else {
396             String anchorName = htmlWriter.links.getName(tagText);
397             int count = htmlWriter.indexAnchorTable
398                     .compute(anchorName, (k, v) -> v == null ? 0 : v + 1);
399             if (count > 0) {
400                 anchorName += "-" + count;
401             }
402             result = HtmlTree.SPAN(anchorName, HtmlStyle.searchTagResult, new StringContent(tagText));
403             if (options.createIndex() && !tagText.isEmpty()) {
404                 SearchIndexItem si = new SearchIndexItem();
405                 si.setSystemProperty(isSystemProperty);
406                 si.setLabel(tagText);
407                 si.setDescription(desc);
408                 si.setUrl(htmlWriter.path.getPath() + "#" + anchorName);
409                 new SimpleElementVisitor14<Void, Void>() {
410 
411                     @Override
412                     public Void visitModule(ModuleElement e, Void p) {
413                         si.setHolder(resources.getText("doclet.module")
414                                              + " " + utils.getFullyQualifiedName(e));
415                         return null;
416                     }
417 
418                     @Override
419                     public Void visitPackage(PackageElement e, Void p) {
420                         si.setHolder(resources.getText("doclet.package")
421                                              + " " + utils.getFullyQualifiedName(e));
422                         return null;
423                     }
424 
425                     @Override
426                     public Void visitType(TypeElement e, Void p) {
427                         si.setHolder(utils.getTypeElementName(e, true)
428                                              + " " + utils.getFullyQualifiedName(e));
429                         return null;
430                     }
431 
432                     @Override
433                     public Void visitExecutable(ExecutableElement e, Void p) {
434                         si.setHolder(utils.getFullyQualifiedName(utils.getEnclosingTypeElement(e))
435                                              + "." + utils.getSimpleName(e) + utils.flatSignature(e));
436                         return null;
437                     }
438 
439                     @Override
440                     public Void visitVariable(VariableElement e, Void p) {
441                         TypeElement te = utils.getEnclosingTypeElement(e);
442                         si.setHolder(utils.getFullyQualifiedName(te) + "." + utils.getSimpleName(e));
443                         return null;
444                     }
445 
446                     @Override
447                     public Void visitUnknown(Element e, Void p) {
448                         if (e instanceof DocletElement) {
449                             DocletElement de = (DocletElement) e;
450                             switch (de.getSubKind()) {
451                                 case OVERVIEW:
452                                     si.setHolder(resources.getText("doclet.Overview"));
453                                     break;
454                                 case DOCFILE:
455                                     si.setHolder(getHolderName(de));
456                                     break;
457                                 default:
458                                     throw new IllegalStateException();
459                             }
460                             return null;
461                         } else {
462                             return super.visitUnknown(e, p);
463                         }
464                     }
465 
466                     @Override
467                     protected Void defaultAction(Element e, Void p) {
468                         si.setHolder(utils.getFullyQualifiedName(e));
469                         return null;
470                     }
471                 }.visit(element);
472                 si.setCategory(SearchIndexItem.Category.SEARCH_TAGS);
473                 configuration.tagSearchIndex.add(si);
474             }
475         }
476         return result;
477     }

479     private String getHolderName(DocletElement de) {
480         PackageElement pe = de.getPackageElement();
481         if (pe.isUnnamed()) {
482             // if package is unnamed use enclosing module only if it is named
483             Element ee = pe.getEnclosingElement();
484             if (ee instanceof ModuleElement && !((ModuleElement)ee).isUnnamed()) {
485                 return resources.getText("doclet.module") + " " + utils.getFullyQualifiedName(ee);
486             }
487             return pe.toString(); // "Unnamed package" or similar
488         }
489         return resources.getText("doclet.package") + " " + utils.getFullyQualifiedName(pe);
490     }
