 26 package jdk.javadoc.internal.doclets.formats.html.markup;
 27 
 28 import java.io.IOException;
 29 import java.io.StringWriter;
 30 import java.io.Writer;
 31 import java.util.*;
 32 
 33 import jdk.javadoc.internal.doclets.toolkit.Content;
 34 import jdk.javadoc.internal.doclets.toolkit.util.DocFile;
 35 import jdk.javadoc.internal.doclets.toolkit.util.DocFileIOException;
 36 import jdk.javadoc.internal.doclets.toolkit.util.DocletConstants;
 37 
 38 /**
 39  * Class for generating an HTML document for javadoc output.
 40  *
 41  *  <p><b>This is NOT part of any supported API.
 42  *  If you write code that depends on this, you do so at your own risk.
 43  *  This code and its internal interfaces are subject to change or
 44  *  deletion without notice.</b>


 45  */
 46 public class HtmlDocument {
 47     private final DocType docType = DocType.HTML5;
 48     private final List<Content> docContent;
 49 
 50     /**
 51      * Constructor to construct an HTML document.
 52      *
 53      * @param docComment comment for the document
 54      * @param htmlTree HTML tree of the document
 55      */
 56     public HtmlDocument(Content docComment, Content htmlTree) {
 57         docContent = Arrays.asList(docComment, htmlTree);
 58     }
 59 
 60     /**
 61      * Constructor to construct an HTML document.
 62      *
 63      * @param htmlTree HTML tree of the document
 64      */
 65     public HtmlDocument(Content htmlTree) {
 66         docContent = Collections.singletonList(htmlTree);
 67     }
 68 
 69     /**
 70      * Writes the content of this document to the specified file.
 71      *
 72      * @param docFile the file
 73      * @throws DocFileIOException if an {@code IOException} occurs while writing the file
 74      */
 75     public void write(DocFile docFile) throws DocFileIOException {
 76         try (Writer writer = docFile.openWriter()) {
 77             write(writer);
 78         } catch (IOException e) {
 79             throw new DocFileIOException(docFile, DocFileIOException.Mode.WRITE, e);
 80         }
 81     }
 82 
 83     @Override
 84     public String toString() {
 85         try (Writer writer = new StringWriter()) {
 86             write(writer);
 87             return writer.toString();
 88         } catch (IOException e) {
 89             throw new Error(e);
 90         }
 91     }
 92 
 93     private void write(Writer writer) throws IOException {
 94         writer.write(docType.text);
 95         writer.write(DocletConstants.NL);
 96         boolean atNewline = true;
 97         for (Content c : docContent) {
 98             atNewline = c.write(writer, atNewline);
 99         }
100     }
101 }
