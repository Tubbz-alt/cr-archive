100     protected Content codeTagOutput(Element element, DocTree tag) {
101         CommentHelper ch = utils.getCommentHelper(element);
102         StringContent content = new StringContent(utils.normalizeNewlines(ch.getText(tag)));
103         Content result = HtmlTree.CODE(content);
104         return result;
105     }

107     protected Content indexTagOutput(Element element, DocTree tag) {
108         CommentHelper ch = utils.getCommentHelper(element);
109         IndexTree itt = (IndexTree)tag;
110 
111         String tagText =  ch.getText(itt.getSearchTerm());
112         if (tagText.charAt(0) == '"' && tagText.charAt(tagText.length() - 1) == '"') {
113             tagText = tagText.substring(1, tagText.length() - 1);
114         }
115         String desc = ch.getText(itt.getDescription());
116 
117         return createAnchorAndSearchIndex(element, tagText,desc);
118     }

132     /**
133      * {@inheritDoc}
134      */
135     public Content deprecatedTagOutput(Element element) {
136         ContentBuilder result = new ContentBuilder();
137         CommentHelper ch = utils.getCommentHelper(element);
138         List<? extends DocTree> deprs = utils.getBlockTags(element, DocTree.Kind.DEPRECATED);
139         if (utils.isTypeElement(element)) {
140             if (utils.isDeprecated(element)) {
141                 result.add(HtmlTree.SPAN(HtmlStyle.deprecatedLabel,
142                         htmlWriter.getDeprecatedPhrase(element)));
143                 if (!deprs.isEmpty()) {
144                     List<? extends DocTree> commentTags = ch.getDescription(configuration, deprs.get(0));
145                     if (!commentTags.isEmpty()) {
146                         result.add(commentTagsToOutput(null, element, commentTags, false));
147                     }
148                 }
149             }
150         } else {
151             if (utils.isDeprecated(element)) {
152                 result.add(HtmlTree.SPAN(HtmlStyle.deprecatedLabel,
153                         htmlWriter.getDeprecatedPhrase(element)));
154                 if (!deprs.isEmpty()) {
155                     List<? extends DocTree> bodyTags = ch.getBody(configuration, deprs.get(0));
156                     Content body = commentTagsToOutput(null, element, bodyTags, false);
157                     if (!body.isEmpty())
158                         result.add(HtmlTree.DIV(HtmlStyle.deprecationComment, body));
159                 }
160             } else {
161                 Element ee = utils.getEnclosingTypeElement(element);
162                 if (utils.isDeprecated(ee)) {
163                     result.add(HtmlTree.SPAN(HtmlStyle.deprecatedLabel,
164                         htmlWriter.getDeprecatedPhrase(ee)));
165                 }
166             }
167         }
168         return result;
169     }

230     /**
231      * {@inheritDoc}
232      */
233     public Content seeTagOutput(Element holder, List<? extends DocTree> seeTags) {
234         ContentBuilder body = new ContentBuilder();
235         for (DocTree dt : seeTags) {
236             appendSeparatorIfNotEmpty(body);
237             body.add(htmlWriter.seeTagToContent(holder, dt));
238         }
239         if (utils.isVariableElement(holder) && ((VariableElement)holder).getConstantValue() != null &&
240                 htmlWriter instanceof ClassWriterImpl) {
241             //Automatically add link to constant values page for constant fields.
242             appendSeparatorIfNotEmpty(body);
243             DocPath constantsPath =
244                     htmlWriter.pathToRoot.resolve(DocPaths.CONSTANT_VALUES);
245             String whichConstant =
246                     ((ClassWriterImpl) htmlWriter).getTypeElement().getQualifiedName() + "." +
247                     utils.getSimpleName(holder);
248             DocLink link = constantsPath.fragment(whichConstant);
249             body.add(htmlWriter.links.createLink(link,
250                     new StringContent(resources.getText("doclet.Constants_Summary"))));
251         }
252         if (utils.isClass(holder) && utils.isSerializable((TypeElement)holder)) {
253             //Automatically add link to serialized form page for serializable classes.
254             if (SerializedFormBuilder.serialInclude(utils, holder) &&
255                       SerializedFormBuilder.serialInclude(utils, utils.containingPackage(holder))) {
256                 appendSeparatorIfNotEmpty(body);
257                 DocPath serialPath = htmlWriter.pathToRoot.resolve(DocPaths.SERIALIZED_FORM);
258                 DocLink link = serialPath.fragment(utils.getFullyQualifiedName(holder));
259                 body.add(htmlWriter.links.createLink(link,
260                         new StringContent(resources.getText("doclet.Serialized_Form"))));
261             }
262         }
263         if (body.isEmpty())
264             return body;
265 
266         ContentBuilder result = new ContentBuilder();
267         result.add(HtmlTree.DT(HtmlTree.SPAN(HtmlStyle.seeLabel,
268                 new StringContent(resources.getText("doclet.See_Also")))));
269         result.add(HtmlTree.DD(body));
270         return result;
271 
272     }
273 
274     private void appendSeparatorIfNotEmpty(ContentBuilder body) {
275         if (!body.isEmpty()) {
276             body.add(", ");
277             body.add(DocletConstants.NL);
278         }
279     }

281     /**
282      * {@inheritDoc}
283      */
284     public Content simpleTagOutput(Element element, List<? extends DocTree> simpleTags, String header) {
285         CommentHelper ch = utils.getCommentHelper(element);
286         ContentBuilder result = new ContentBuilder();
287         result.add(HtmlTree.DT(HtmlTree.SPAN(HtmlStyle.simpleTagLabel, new RawHtml(header))));
288         ContentBuilder body = new ContentBuilder();
289         boolean many = false;
290         for (DocTree simpleTag : simpleTags) {
291             if (many) {
292                 body.add(", ");
293             }
294             List<? extends DocTree> bodyTags = ch.getBody(configuration, simpleTag);
295             body.add(htmlWriter.commentTagsToContent(simpleTag, element, bodyTags, false, inSummary));
296             many = true;
297         }
298         result.add(HtmlTree.DD(body));
299         return result;
300     }

334     /**
335      * {@inheritDoc}
336      */
337     public Content throwsTagOutput(Element element, DocTree throwsTag) {
338         ContentBuilder body = new ContentBuilder();
339         CommentHelper ch = utils.getCommentHelper(element);
340         Element exception = ch.getException(configuration, throwsTag);
341         Content excName;
342         if (exception == null) {
343             excName = new RawHtml(ch.getExceptionName(throwsTag).toString());
344         } else if (exception.asType() == null) {
345             excName = new RawHtml(utils.getFullyQualifiedName(exception));
346         } else {
347             LinkInfoImpl link = new LinkInfoImpl(configuration, LinkInfoImpl.Kind.MEMBER,
348                                                  exception.asType());
349             link.excludeTypeBounds = true;
350             excName = htmlWriter.getLink(link);
351         }
352         body.add(HtmlTree.CODE(excName));
353         List<? extends DocTree> description = ch.getDescription(configuration, throwsTag);
354         Content desc = htmlWriter.commentTagsToContent(throwsTag, element, description, false, inSummary);
355         if (desc != null && !desc.isEmpty()) {
356             body.add(" - ");
357             body.add(desc);
358         }
359         HtmlTree result = HtmlTree.DD(body);
360         return result;
361     }

411     private Content createAnchorAndSearchIndex(Element element, String tagText, String desc){
412         String anchorName = htmlWriter.links.getName(tagText);
413         Content result = null;
414         if (isFirstSentence && inSummary) {
415             result = new StringContent(tagText);
416         } else {
417             result = HtmlTree.A_ID(HtmlStyle.searchTagResult, anchorName, new StringContent(tagText));
418             if (configuration.createindex && !tagText.isEmpty()) {
419                 SearchIndexItem si = new SearchIndexItem();
420                 si.setLabel(tagText);
421                 si.setDescription(desc);
422                 si.setUrl(htmlWriter.path.getPath() + "#" + anchorName);
423                 DocPaths docPaths = configuration.docPaths;
424                 new SimpleElementVisitor9<Void, Void>() {
425                     @Override
426                     public Void visitVariable(VariableElement e, Void p) {
427                         TypeElement te = utils.getEnclosingTypeElement(e);
428                         si.setHolder(utils.getFullyQualifiedName(e) + "." + utils.getSimpleName(e));
429                         return null;
430                     }
431 
432                     @Override
433                     public Void visitUnknown(Element e, Void p) {
434                         if (e instanceof DocletElement) {
435                             DocletElement de = (DocletElement) e;
436                             switch (de.getSubKind()) {
437                                 case OVERVIEW:
438                                     si.setHolder(resources.getText("doclet.Overview"));
439                                     break;
440                                 case DOCFILE:
441                                     si.setHolder(de.getPackageElement().toString());
442                                     break;
443                                 default:
444                                     throw new IllegalStateException();
445                             }
446                             return null;
447                         } else {
448                             return super.visitUnknown(e, p);
449                         }
450                     }
451 
452                     @Override
453                     protected Void defaultAction(Element e, Void p) {
454                         si.setHolder(utils.getFullyQualifiedName(e));
455                         return null;
456                     }
457                 }.visit(element);
458                 si.setCategory(SearchIndexItem.Category.SEARCH_TAGS);
459                 configuration.tagSearchIndex.add(si);
460             }
461         }
462         return result;
463     }
