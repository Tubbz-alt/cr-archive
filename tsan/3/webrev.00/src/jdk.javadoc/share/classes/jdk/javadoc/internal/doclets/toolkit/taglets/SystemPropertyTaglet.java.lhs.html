31 import javax.lang.model.element.Element;
32 import java.util.EnumSet;
33 
34 import static com.sun.source.doctree.DocTree.Kind.SYSTEM_PROPERTY;
35 
36 /**
37  * A taglet that represents the {@code @systemProperty} tag.
38  *
39  *  <p><b>This is NOT part of any supported API.
40  *  If you write code that depends on this, you do so at your own risk.
41  *  This code and its internal interfaces are subject to change or
42  *  deletion without notice.</b>
43  */
44 
45 public class SystemPropertyTaglet extends BaseTaglet {
46 
47     SystemPropertyTaglet(){
48         super(SYSTEM_PROPERTY.tagName, true, EnumSet.of(Site.CONSTRUCTOR, Site.METHOD, Site.FIELD,
49                 Site.PACKAGE, Site.MODULE, Site.TYPE));
50     }
51 
52     @Override
53     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
54         return writer.systemPropertyTagOutput(element, tag);
55     }
56 }
