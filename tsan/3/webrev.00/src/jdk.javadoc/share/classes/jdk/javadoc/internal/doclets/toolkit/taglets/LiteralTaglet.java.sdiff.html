26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;

32 import jdk.javadoc.internal.doclets.toolkit.Content;
33 
34 import static com.sun.source.doctree.DocTree.Kind.LITERAL;
35 /**
36  * An inline Taglet used to denote literal text.
37  * For example, the text:
38  * <blockquote>  {@code {@literal a<B>c}}  </blockquote>
39  * displays as:
40  * <blockquote>  {@literal a<B>c}  </blockquote>
41  *
42  *  <p><b>This is NOT part of any supported API.
43  *  If you write code that depends on this, you do so at your own risk.
44  *  This code and its internal interfaces are subject to change or
45  *  deletion without notice.</b>
46  *
47  * @author Scott Seligman
48  */
49 
50 public class LiteralTaglet extends BaseTaglet {
51 
52     LiteralTaglet() {
53         super(LITERAL.tagName, true, EnumSet.allOf(Site.class));
54     }
55 
56     @Override
57     public Content getTagletOutput(Element e, DocTree tag, TagletWriter writer) {
58         return writer.literalTagOutput(e, tag);
59     }
60 }
