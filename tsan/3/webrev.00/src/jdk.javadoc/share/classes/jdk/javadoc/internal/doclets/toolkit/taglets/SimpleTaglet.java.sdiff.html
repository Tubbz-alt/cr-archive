 52 public class SimpleTaglet extends BaseTaglet implements InheritableTaglet {
 53 
 54     /**
 55      * The header to output.
 56      */
 57     protected String header;
 58 
 59     /**
 60      * Whether or not the taglet should generate output.
 61      * Standard tags like at-author, at-since, at-version can be disabled
 62      * by command-line options; custom tags created with -tag can be
 63      * disabled with an X in the defining string.
 64      */
 65     protected final boolean enabled;
 66 
 67     /**
 68      * Construct a <code>SimpleTaglet</code>.
 69      * @param tagName the name of this tag
 70      * @param header the header to output.
 71      * @param locations the possible locations that this tag
 72      * can appear in.  The <code>String</code> can contain 'p'
 73      * for package, 't' for type, 'm' for method, 'c' for constructor
 74      * and 'f' for field.


 75      */
 76     public SimpleTaglet(String tagName, String header, String locations) {
 77         this(tagName, header, getSites(locations), isEnabled(locations));























 78     }
 79 
 80     /**
 81      * Construct a <code>SimpleTaglet</code>.
 82      * @param tagName the name of this tag
 83      * @param header the header to output.
 84      * @param sites the possible sites (locations) that this tag
 85      * can appear in.  The <code>String</code> can contain 'p'
 86      * for package, 't' for type, 'm' for method, 'c' for constructor
 87      * and 'f' for field.
 88      */
 89     public SimpleTaglet(String tagName, String header, Set<Site> sites) {
 90         this(tagName, header, sites, true);


 91     }
 92 
 93     /**
 94      * Construct a <code>SimpleTaglet</code>.
 95      * @param tagName the name of this tag
 96      * @param header the header to output.
 97      * @param sites the possible sites (locations) that this tag
 98      * can appear in.  The <code>String</code> can contain 'p'
 99      * for package, 't' for type, 'm' for method, 'c' for constructor
100      * and 'f' for field.
101      */
102     public SimpleTaglet(String tagName, String header, Set<Site> sites, boolean enabled) {
103         super(tagName, false, sites);
104         this.header = header;
105         this.enabled = enabled;
106     }
107 
108     private static Set<Site> getSites(String locations) {
109         Set<Site> set = EnumSet.noneOf(Site.class);
110         for (int i = 0; i < locations.length(); i++) {
111             switch (locations.charAt(i)) {
112                 case 'a':  case 'A':
113                     return EnumSet.allOf(Site.class);
114                 case 'c':  case 'C':
115                     set.add(Site.CONSTRUCTOR);
116                     break;
117                 case 'f':  case 'F':
118                     set.add(Site.FIELD);
119                     break;
120                 case 'm':  case 'M':
121                     set.add(Site.METHOD);
122                     break;
123                 case 'o':  case 'O':
124                     set.add(Site.OVERVIEW);
125                     break;
126                 case 'p':  case 'P':
127                     set.add(Site.PACKAGE);
128                     break;
129                 case 's':  case 'S':        // super-packages, anyone?
130                     set.add(Site.MODULE);
131                     break;
132                 case 't':  case 'T':
133                     set.add(Site.TYPE);
134                     break;
135                 case 'x':  case 'X':
136                     break;
137             }
138         }
139         return set;
140     }
141 
142     private static boolean isEnabled(String locations) {
143         return locations.matches("[^Xx]*");
144     }
145 
146     @Override
147     public void inherit(DocFinder.Input input, DocFinder.Output output) {
148         List<? extends DocTree> tags = input.utils.getBlockTags(input.element, name);
149         if (!tags.isEmpty()) {
150             output.holder = input.element;
151             output.holderTag = tags.get(0);
152             CommentHelper ch = input.utils.getCommentHelper(output.holder);
153             output.inlineTags = input.isFirstSentence
154                     ? ch.getFirstSentenceTrees(input.utils.configuration, output.holderTag)
155                     : ch.getTags(input.utils.configuration, output.holderTag);
156         }
157     }
158 
159     @Override
160     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
161         return header == null || tag == null ? null : writer.simpleTagOutput(element, tag, header);
162     }
163 
164     @Override
165     public Content getTagletOutput(Element holder, TagletWriter writer) {
166         Utils utils = writer.configuration().utils;
167         List<? extends DocTree> tags = utils.getBlockTags(holder, getName());
168         if (header == null || tags.isEmpty()) {
169             return null;
170         }
171         return writer.simpleTagOutput(holder, tags, header);
172     }
173 }
