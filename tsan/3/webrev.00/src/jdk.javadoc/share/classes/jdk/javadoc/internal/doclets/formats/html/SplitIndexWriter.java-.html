 46 /**
 47  * Generate Separate Index Files for all the member names with Indexing in
 48  * Unicode Order. This will create "index-files" directory in the current or
 49  * destination directory and will generate separate file for each unicode index.
 50  *
 51  *  <p><b>This is NOT part of any supported API.
 52  *  If you write code that depends on this, you do so at your own risk.
 53  *  This code and its internal interfaces are subject to change or
 54  *  deletion without notice.</b>
 55  *
 56  * @see java.lang.Character
 57  * @author Atul M Dambalkar
 58  * @author Bhavesh Patel (Modified)
 59  */
 60 public class SplitIndexWriter extends AbstractIndexWriter {
 61 
 62     private final List<Character> indexElements;
 63 
 64     /**
 65      * Construct the SplitIndexWriter. Uses path to this file and relative path
 66      * from this file.
 67      *
 68      * @param configuration the configuration for this doclet
 69      * @param path       Path to the file which is getting generated.
 70      * @param indexbuilder Unicode based Index from {@link IndexBuilder}
 71      * @param elements the collection of characters for which to generate index files
 72      */
 73     public SplitIndexWriter(HtmlConfiguration configuration,
 74                             DocPath path,
 75                             IndexBuilder indexbuilder,
 76                             Collection<Character> elements) {
 77         super(configuration, path, indexbuilder);
 78         this.indexElements = new ArrayList<>(elements);
 79     }
 80 
 81     /**
 82      * Generate separate index files, for each Unicode character, listing all
 83      * the members starting with the particular unicode character.
 84      *
 85      * @param configuration the configuration for this doclet
 86      * @param indexbuilder IndexBuilder built by {@link IndexBuilder}
 87      * @throws DocFileIOException if there is a problem generating the index files
 88      */
 89     public static void generate(HtmlConfiguration configuration,
 90                                 IndexBuilder indexbuilder) throws DocFileIOException {
 91         DocPath path = DocPaths.INDEX_FILES;
 92         Set<Character> keys = new TreeSet<>(indexbuilder.getIndexMap().keySet());
 93         keys.addAll(configuration.tagSearchIndexKeys);
 94         ListIterator<Character> li = new ArrayList<>(keys).listIterator();
 95         int prev;
 96         int next;
 97         while (li.hasNext()) {
 98             prev = (li.hasPrevious()) ? li.previousIndex() + 1 : -1;
 99             Object ch = li.next();
100             next = (li.hasNext()) ? li.nextIndex() + 1 : -1;
101             DocPath filename = DocPaths.indexN(li.nextIndex());
102             SplitIndexWriter indexgen = new SplitIndexWriter(configuration,
103                     path.resolve(filename),
104                     indexbuilder, keys);
105             indexgen.generateIndexFile((Character) ch);
106             if (!li.hasNext()) {
107                 indexgen.createSearchIndexFiles();
108             }
109         }
110     }
111 
112     /**
113      * Generate the contents of each index file, with Header, Footer,
114      * Member Field, Method and Constructor Description.
115      *
116      * @param unicode Unicode character referring to the character for the
117      * index.
118      * @throws DocFileIOException if there is a problem generating an index file
119      */
120     protected void generateIndexFile(Character unicode) throws DocFileIOException {
121         String title = resources.getText("doclet.Window_Split_Index",
122                 unicode.toString());
123         HtmlTree body = getBody(true, getWindowTitle(title));
124         HtmlTree header = HtmlTree.HEADER();
125         addTop(header);
126         navBar.setUserHeader(getUserHeaderFooter(true));
127         header.add(navBar.getContent(true));
128         body.add(header);
129         HtmlTree main = HtmlTree.MAIN();
130         main.add(HtmlTree.DIV(HtmlStyle.header,
131                 HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING,
132                         contents.getContent("doclet.Index"))));
133         HtmlTree divTree = new HtmlTree(HtmlTag.DIV);
134         divTree.setStyle(HtmlStyle.contentContainer);
135         addLinksForIndexes(divTree);
136         if (configuration.tagSearchIndexMap.get(unicode) == null) {
137             addContents(unicode, indexbuilder.getMemberList(unicode), divTree);
138         } else if (indexbuilder.getMemberList(unicode) == null) {
139             addSearchContents(unicode, configuration.tagSearchIndexMap.get(unicode), divTree);
140         } else {
141             addContents(unicode, indexbuilder.getMemberList(unicode),
142                     configuration.tagSearchIndexMap.get(unicode), divTree);
143         }
144         addLinksForIndexes(divTree);
145         main.add(divTree);
146         body.add(main);
147         HtmlTree footer = HtmlTree.FOOTER();
148         navBar.setUserFooter(getUserHeaderFooter(false));
149         footer.add(navBar.getContent(false));
150         addBottom(footer);
151         body.add(footer);
152         String description = "index: " + unicode;
153         printHtmlDocument(null, description, body);
154     }
155 
156     /**
157      * Add links for all the Index Files per unicode character.
158      *
159      * @param contentTree the content tree to which the links for indexes will be added
160      */
161     protected void addLinksForIndexes(Content contentTree) {
162         for (int i = 0; i < indexElements.size(); i++) {
163             int j = i + 1;
164             contentTree.add(links.createLink(DocPaths.indexN(j),
165                     new StringContent(indexElements.get(i).toString())));
166             contentTree.add(Contents.SPACE);
167         }
168         contentTree.add(new HtmlTree(HtmlTag.BR));
169         contentTree.add(links.createLink(pathToRoot.resolve(DocPaths.ALLCLASSES_INDEX),
170                 contents.allClassesLabel));
171         if (!configuration.packages.isEmpty()) {
172             contentTree.add(Contents.SPACE);
173             contentTree.add(links.createLink(pathToRoot.resolve(DocPaths.ALLPACKAGES_INDEX),
174                     contents.allPackagesLabel));
175     }
176 }
