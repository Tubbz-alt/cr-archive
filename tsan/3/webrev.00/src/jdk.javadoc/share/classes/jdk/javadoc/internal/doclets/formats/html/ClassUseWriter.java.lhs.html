107     /**
108      * Constructor.
109      *
110      * @param filename the file to be generated.
111      */
112     public ClassUseWriter(HtmlConfiguration configuration,
113                           ClassUseMapper mapper, DocPath filename,
114                           TypeElement typeElement) {
115         super(configuration, filename);
116         this.typeElement = typeElement;
117         if (mapper.classToPackageAnnotations.containsKey(typeElement)) {
118             pkgToPackageAnnotations = new TreeSet<>(utils.makeClassUseComparator());
119             pkgToPackageAnnotations.addAll(mapper.classToPackageAnnotations.get(typeElement));
120         }
121         configuration.currentTypeElement = typeElement;
122         this.pkgSet = new TreeSet<>(utils.makePackageComparator());
123         this.pkgToClassTypeParameter = pkgDivide(mapper.classToClassTypeParam);
124         this.pkgToClassAnnotations = pkgDivide(mapper.classToClassAnnotations);
125         this.pkgToMethodTypeParameter = pkgDivide(mapper.classToMethodTypeParam);
126         this.pkgToMethodArgTypeParameter = pkgDivide(mapper.classToMethodArgTypeParam);
127         this.pkgToFieldTypeParameter = pkgDivide(mapper.classToFieldTypeParam);
128         this.pkgToFieldAnnotations = pkgDivide(mapper.annotationToField);
129         this.pkgToMethodReturnTypeParameter = pkgDivide(mapper.classToMethodReturnTypeParam);
130         this.pkgToMethodAnnotations = pkgDivide(mapper.classToMethodAnnotations);
131         this.pkgToMethodParameterAnnotations = pkgDivide(mapper.classToMethodParamAnnotation);
132         this.pkgToSubclass = pkgDivide(mapper.classToSubclass);
133         this.pkgToSubinterface = pkgDivide(mapper.classToSubinterface);
134         this.pkgToImplementingClass = pkgDivide(mapper.classToImplementingClass);
135         this.pkgToField = pkgDivide(mapper.classToField);
136         this.pkgToMethodReturn = pkgDivide(mapper.classToMethodReturn);
137         this.pkgToMethodArgs = pkgDivide(mapper.classToMethodArgs);
138         this.pkgToMethodThrows = pkgDivide(mapper.classToMethodThrows);
139         this.pkgToConstructorAnnotations = pkgDivide(mapper.classToConstructorAnnotations);
140         this.pkgToConstructorParameterAnnotations = pkgDivide(mapper.classToConstructorParamAnnotation);
141         this.pkgToConstructorArgs = pkgDivide(mapper.classToConstructorArgs);
142         this.pkgToConstructorArgTypeParameter = pkgDivide(mapper.classToConstructorArgTypeParam);
143         this.pkgToConstructorThrows = pkgDivide(mapper.classToConstructorThrows);
144         //tmp test
145         if (pkgSet.size() > 0 &&
146             mapper.classToPackage.containsKey(this.typeElement) &&
147             !pkgSet.equals(mapper.classToPackage.get(this.typeElement))) {
148             configuration.reporter.print(Diagnostic.Kind.WARNING,
149                     "Internal error: package sets don't match: "
150                     + pkgSet + " with: " + mapper.classToPackage.get(this.typeElement));
151         }
152 
153         methodSubWriter = new MethodWriterImpl(this);
154         constrSubWriter = new ConstructorWriterImpl(this);
155         fieldSubWriter = new FieldWriterImpl(this);
156         classSubWriter = new NestedClassWriterImpl(this);
157         this.navBar = new Navigation(typeElement, configuration, fixedNavDiv, PageMode.USE, path);
158     }

160     /**
161      * Write out class use pages.
162      *
163      * @param configuration the configuration for this doclet
164      * @param classtree the class tree hierarchy
165      * @throws DocFileIOException if there is an error while generating the documentation
166      */
167     public static void generate(HtmlConfiguration configuration, ClassTree classtree) throws DocFileIOException  {
168         ClassUseMapper mapper = new ClassUseMapper(configuration, classtree);


169         for (TypeElement aClass : configuration.getIncludedTypeElements()) {
170             // If -nodeprecated option is set and the containing package is marked
171             // as deprecated, do not generate the class-use page. We will still generate
172             // the class-use page if the class is marked as deprecated but the containing
173             // package is not since it could still be linked from that package-use page.
174             if (!(configuration.nodeprecated &&
175                   configuration.utils.isDeprecated(configuration.utils.containingPackage(aClass))))
176                 ClassUseWriter.generate(configuration, mapper, aClass);
177         }
178         for (PackageElement pkg : configuration.packages) {
179             // If -nodeprecated option is set and the package is marked
180             // as deprecated, do not generate the package-use page.
181             if (!(configuration.nodeprecated && configuration.utils.isDeprecated(pkg)))
182                 PackageUseWriter.generate(configuration, mapper, pkg);
183         }
184     }

186     private Map<PackageElement, List<Element>> pkgDivide(Map<TypeElement, ? extends List<? extends Element>> classMap) {
187         Map<PackageElement, List<Element>> map = new HashMap<>();
188         List<? extends Element> elements = (List<? extends Element>) classMap.get(typeElement);
189         if (elements != null) {
190             Collections.sort(elements, utils.makeClassUseComparator());
191             for (Element e : elements) {
192                 PackageElement pkg = utils.containingPackage(e);
193                 pkgSet.add(pkg);
194                 List<Element> inPkg = map.get(pkg);
195                 if (inPkg == null) {
196                     inPkg = new ArrayList<>();
197                     map.put(pkg, inPkg);
198                 }
199                 inPkg.add(e);
200             }
201         }
202         return map;
203     }

205     /**
206      * Generate a class page.
207      *
208      * @throws DocFileIOException if there is a problem while generating the documentation
209      */
210     public static void generate(HtmlConfiguration configuration, ClassUseMapper mapper,
211                                 TypeElement typeElement) throws DocFileIOException {
212         ClassUseWriter clsgen;
213         DocPath path = configuration.docPaths.forPackage(typeElement)
214                               .resolve(DocPaths.CLASS_USE)
215                               .resolve(configuration.docPaths.forName( typeElement));
216         clsgen = new ClassUseWriter(configuration, mapper, path, typeElement);
217         clsgen.generateClassUseFile();
218     }

220     /**
221      * Generate the class use elements.
222      *
223      * @throws DocFileIOException if there is a problem while generating the documentation
224      */
225     protected void generateClassUseFile() throws DocFileIOException {
226         HtmlTree body = getClassUseHeader();
227         HtmlTree div = new HtmlTree(HtmlTag.DIV);
228         div.setStyle(HtmlStyle.classUseContainer);
229         if (pkgSet.size() > 0) {
230             addClassUse(div);
231         } else {
232             div.add(contents.getContent("doclet.ClassUse_No.usage.of.0",
233                     utils.getFullyQualifiedName(typeElement)));
234         }
235         mainTree.add(div);
236         body.add(mainTree);
237         HtmlTree footer = HtmlTree.FOOTER();
238         navBar.setUserFooter(getUserHeaderFooter(false));
239         footer.add(navBar.getContent(false));
240         addBottom(footer);
241         body.add(footer);

242         String description = getDescription("use", typeElement);
243         printHtmlDocument(null, description, body);
244     }

246     /**
247      * Add the class use documentation.
248      *
249      * @param contentTree the content tree to which the class use information will be added
250      */
251     protected void addClassUse(Content contentTree) {
252         HtmlTree ul = new HtmlTree(HtmlTag.UL);
253         ul.setStyle(HtmlStyle.blockList);
254         if (configuration.packages.size() > 1) {
255             addPackageList(ul);
256             addPackageAnnotationList(ul);
257         }
258         addClassList(ul);
259         contentTree.add(ul);
260     }

262     /**
263      * Add the packages elements that use the given class.
264      *
265      * @param contentTree the content tree to which the packages elements will be added
266      */
267     protected void addPackageList(Content contentTree) {
268         Content caption = getTableCaption(contents.getContent(
269                 "doclet.ClassUse_Packages.that.use.0",
270                 getLink(new LinkInfoImpl(configuration,
271                         LinkInfoImpl.Kind.CLASS_USE_HEADER, typeElement))));
272         Table table = new Table(HtmlStyle.useSummary)
273                 .setCaption(caption)
274                 .setHeader(getPackageTableHeader())
275                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);
276         for (PackageElement pkg : pkgSet) {
277             addPackageUse(pkg, table);
278         }
279         Content li = HtmlTree.LI(HtmlStyle.blockList, table.toContent());
280         contentTree.add(li);
281     }
282 
283     /**
284      * Add the package annotation elements.
285      *
286      * @param contentTree the content tree to which the package annotation elements will be added
287      */
288     protected void addPackageAnnotationList(Content contentTree) {
289         if (!utils.isAnnotationType(typeElement) ||
290                 pkgToPackageAnnotations == null ||
291                 pkgToPackageAnnotations.isEmpty()) {
292             return;
293         }
294         Content caption = getTableCaption(contents.getContent(
295                 "doclet.ClassUse_PackageAnnotation",
296                 getLink(new LinkInfoImpl(configuration,
297                         LinkInfoImpl.Kind.CLASS_USE_HEADER, typeElement))));
298 
299         Table table = new Table(HtmlStyle.useSummary)
300                 .setCaption(caption)
301                 .setHeader(getPackageTableHeader())
302                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);
303         for (PackageElement pkg : pkgToPackageAnnotations) {
304             Content summary = new ContentBuilder();
305             addSummaryComment(pkg, summary);
306             table.addRow(getPackageLink(pkg), summary);
307         }
308         Content li = HtmlTree.LI(HtmlStyle.blockList, table.toContent());
309         contentTree.add(li);
310     }

312     /**
313      * Add the class elements that use the given class.
314      *
315      * @param contentTree the content tree to which the class elements will be added
316      */
317     protected void addClassList(Content contentTree) {
318         HtmlTree ul = new HtmlTree(HtmlTag.UL);
319         ul.setStyle(HtmlStyle.blockList);
320         for (PackageElement pkg : pkgSet) {
321             Content markerAnchor = links.createAnchor(getPackageAnchorName(pkg));
322             HtmlTree htmlTree = HtmlTree.SECTION(markerAnchor);
323             Content link = contents.getContent("doclet.ClassUse_Uses.of.0.in.1",
324                     getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.CLASS_USE_HEADER,
325                             typeElement)),
326                     getPackageLink(pkg, utils.getPackageName(pkg)));
327             Content heading = HtmlTree.HEADING(Headings.TypeUse.SUMMARY_HEADING, link);
328             htmlTree.add(heading);
329             addClassUse(pkg, htmlTree);
330             ul.add(HtmlTree.LI(HtmlStyle.blockList, htmlTree));
331         }
332         Content li = HtmlTree.LI(HtmlStyle.blockList, ul);
333         contentTree.add(li);
334     }

425     /**
426      * Get the header for the class use Listing.
427      *
428      * @return a content tree representing the class use header
429      */
430     protected HtmlTree getClassUseHeader() {
431         String cltype = resources.getText(utils.isInterface(typeElement)
432                 ? "doclet.Interface"
433                 : "doclet.Class");
434         String clname = utils.getFullyQualifiedName(typeElement);
435         String title = resources.getText("doclet.Window_ClassUse_Header",
436                 cltype, clname);
437         HtmlTree bodyTree = getBody(true, getWindowTitle(title));
438         HtmlTree htmlTree = HtmlTree.HEADER();
439         addTop(htmlTree);
440         Content mdleLinkContent = getModuleLink(utils.elementUtils.getModuleOf(typeElement),
441                 contents.moduleLabel);
442         navBar.setNavLinkModule(mdleLinkContent);
443         Content classLinkContent = getLink(new LinkInfoImpl(
444                 configuration, LinkInfoImpl.Kind.CLASS_USE_HEADER, typeElement)
445                 .label(resources.getText("doclet.Class")));
446         navBar.setNavLinkClass(classLinkContent);
447         navBar.setUserHeader(getUserHeaderFooter(true));
448         htmlTree.add(navBar.getContent(true));
449         bodyTree.add(htmlTree);
450         ContentBuilder headContent = new ContentBuilder();
451         headContent.add(contents.getContent("doclet.ClassUse_Title", cltype));
452         headContent.add(new HtmlTree(HtmlTag.BR));
453         headContent.add(clname);
454         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING,
455                 true, HtmlStyle.title, headContent);
456         Content div = HtmlTree.DIV(HtmlStyle.header, heading);
457         mainTree.add(div);
458         return bodyTree;
459     }
