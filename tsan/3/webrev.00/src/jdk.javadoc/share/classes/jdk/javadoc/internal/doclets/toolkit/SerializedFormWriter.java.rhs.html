 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import javax.lang.model.element.ExecutableElement;
 29 import javax.lang.model.element.TypeElement;
 30 import javax.lang.model.element.VariableElement;
 31 import javax.lang.model.type.TypeMirror;
 32 
 33 import com.sun.source.doctree.DocTree;
 34 import jdk.javadoc.internal.doclets.toolkit.util.DocFileIOException;
 35 
 36 /**
 37  * The interface for writing serialized form output.
 38  *
 39  *  <p><b>This is NOT part of any supported API.
 40  *  If you write code that depends on this, you do so at your own risk.
 41  *  This code and its internal interfaces are subject to change or
 42  *  deletion without notice.</b>


 43  */
 44 
 45 public interface SerializedFormWriter {
 46 
 47     /**
 48      * Get the header.
 49      *
 50      * @param header the header to write.
 51      * @return the header content tree
 52      */
 53     Content getHeader(String header);
 54 
 55     /**
 56      * Get the serialized form summaries header.
 57      *
 58      * @return the serialized form summary header tree
 59      */
 60     Content getSerializedSummariesHeader();
 61 
 62     /**
 63      * Get the package serialized form header.
 64      *
 65      * @return the package serialized form header tree
 66      */
 67     Content getPackageSerializedHeader();
 68 
 69     /**
 70      * Add the serialized tree per package to the serialized summaries tree.
 71      *
 72      * @param serializedSummariesTree the serialized tree to which the package serialized tree will be added
 73      * @param packageSerializedTree the serialized tree per package that needs to be added
 74      */
 75     void addPackageSerializedTree(Content serializedSummariesTree, Content packageSerializedTree);
 76 
 77     /**
 78      * Get the given package header.
 79      *
 80      * @param packageName the package header to write
 81      * @return a content tree for the package header
 82      */
 83     Content getPackageHeader(String packageName);
 84 
 85     /**
 86      * Get the serialized class header.
 87      *
 88      * @return a content tree for the serialized class header
 89      */
 90     Content getClassSerializedHeader();
 91 
 92     /**
 93      * Get the heading for the serializable class.
 94      *
 95      * @param typeElement the class being processed
 96      * @return a content tree for the class heading
 97      */
 98     Content getClassHeader(TypeElement typeElement);
 99 
100     /**
101      * Get the serial UID info header.
102      *
103      * @return a content tree for the serial uid info header
104      */
105     Content getSerialUIDInfoHeader();
106 
107     /**
108      * Adds the serial UID info.
109      *
110      * @param header the header that will show up before the UID.
111      * @param serialUID the serial UID to print.
112      * @param serialUidTree the serial UID tree to which the content will be added.
113      */
114     void addSerialUIDInfo(String header, String serialUID, Content serialUidTree);

115 
116     /**
117      * Get the class serialize content header.
118      *
119      * @return a content tree for the class serialize content header
120      */
121     Content getClassContentHeader();
122 
123     /**
124      * Return an instance of a SerialFieldWriter for a class.
125      *
126      * @param typeElement the class
127      * @return an instance of a SerialFieldWriter.
128      */
129     SerialFieldWriter getSerialFieldWriter(TypeElement typeElement);
130 
131     /**
132      * Return an instance of a SerialMethodWriter for a class.
133      *
134      * @param typeElement the class
135      * @return an instance of a SerialMethodWriter.
136      */
137     SerialMethodWriter getSerialMethodWriter(TypeElement typeElement);
138 
139     /**
140      * Add the serialized content to the body content.
141      *
142      * @param serializedTreeContent content for serialized data

143      */
144     void addSerializedContent(Content serializedTreeContent);
145 
146     /**
147      * Add the footer.


148      */
149     void addFooter();
150 
151     /**
152      * Print the serialized form document.
153      *
154      * @param serializedTree the content tree that will be printed
155      * @throws DocFileIOException if there is a problem while writing the document
156      */
157     void printDocument(Content serializedTree) throws DocFileIOException;
158 
159     /**
160      * Gets the member tree.
161      *
162      * @param contentTree the tree used to generate the complete member tree
163      * @return a content tree for the member
164      */
165     Content getMemberTree(Content contentTree);
166 
167     /**
168      * Write the serialized form for a given field.
169      */
170     interface SerialFieldWriter {
171 
172         /**
173          * Get the serializable field header.
174          *
175          * @return serialized fields header content tree
176          */
177         Content getSerializableFieldsHeader();
178 
179         /**
180          * Get the field content header.
181          *
182          * @param isLastContent true if this is the last content to be documented
183          * @return fields header content tree
184          */
185         Content getFieldsContentHeader(boolean isLastContent);
186 
187         /**
188          * Get the fields content.
189          *
190          * @param heading the heading to write.
191          * @param contentTree content tree to which the heading will be added
192          * @return serializable fields content tree
193          */
194         Content getSerializableFields(String heading, Content contentTree);
195 
196         /**
197          * Adds the deprecated information for this member.
198          *
199          * @param field the field to document.
200          * @param contentTree content tree to which the deprecated information will be added
201          */
202         void addMemberDeprecatedInfo(VariableElement field, Content contentTree);
203 
204         /**
205          * Adds the description text for this member.
206          *
207          * @param field the field to document
208          * @param contentTree content tree to which the member description will be added
209          */
210         void addMemberDescription(VariableElement field, Content contentTree);
211 
212         /**
213          * Adds the description text for this member represented by the tag.
214          *
215          * @param field the field to document
216          * @param serialFieldTag the field to document (represented by tag)
217          * @param contentTree content tree to which the member description will be added
218          */
219         void addMemberDescription(VariableElement field, DocTree serialFieldTag, Content contentTree);
220 
221         /**
222          * Adds the tag information for this member.
223          *
224          * @param field the field to document
225          * @param contentTree content tree to which the member tags will be added
226          */
227         void addMemberTags(VariableElement field, Content contentTree);
228 
229         /**
230          * Adds the member header.
231          *
232          * @param fieldType the type of the field
233          * @param fieldTypeStr the type of the field in string format, used
234          *                     only if the type cannot be linked
235          * @param fieldDimensions the dimensions of the field
236          * @param fieldName the name of the field
237          * @param contentTree content tree to which the member header will be added
238          */
239         void addMemberHeader(TypeElement fieldType, String fieldTypeStr, String fieldDimensions,
240                              String fieldName, Content contentTree);
241 
242         /**
243          * Adds the member header.
244          *
245          * @param fieldType the type of the field
246          * @param fieldName the name of the field
247          * @param contentTree content tree to which the member header will be added
248          */
249         void addMemberHeader(TypeMirror fieldType, String fieldName, Content contentTree);
250 
251         /**
252          * Check to see if overview details should be printed. If
253          * nocomment option set or if there is no text to be printed
254          * for deprecation info, inline comment or tags,
255          * do not print overview details.
256          *
257          * @param field the field to check overview details for
258          * @return true if overview details need to be printed
259          */
260         boolean shouldPrintOverview(VariableElement field);
261     }
262 
263     /**
264      * Write the serialized form for a given field.
265      */
266     interface SerialMethodWriter {
267 
268         /**
269          * Get the serializable method header.
270          *
271          * @return serializable methods content tree
272          */
273         Content getSerializableMethodsHeader();
274 
275         /**
276          * Get the method content header.
277          *
278          * @param isLastContent true if this is the last content to be documented
279          * @return methods content tree
280          */
281         Content getMethodsContentHeader(boolean isLastContent);
282 
283         /**
284          * Write the given heading.
285          *
286          * @param heading the heading to write
287          * @param serializableMethodTree content tree which will be added
288          * @return serializable methods content tree
289          */
290         Content getSerializableMethods(String heading, Content serializableMethodTree);
291 
292         /**
293          * Write a warning that no serializable methods exist.
294          *
295          * @param msg the warning to print
296          * @return no customization message tree
297          */
298         Content getNoCustomizationMsg(String msg);
299 
300         /**
301          * Adds the header.
302          *
303          * @param member the member to write the header for
304          * @param methodsContentTree content tree to which the header will be added
305          */
306         void addMemberHeader(ExecutableElement member, Content methodsContentTree);
307 
308         /**
309          * Adds the deprecated information for this member.
310          *
311          * @param member the member to write the deprecated information for
312          * @param methodsContentTree content tree to which the deprecated
313          * information will be added
314          */
315         void addDeprecatedMemberInfo(ExecutableElement member, Content methodsContentTree);
316 
317         /**
318          * Adds the description for this member.
319          *
320          * @param member the member to write the information for
321          * @param methodsContentTree content tree to which the member
322          * information will be added
323          */
324         void addMemberDescription(ExecutableElement member, Content methodsContentTree);
325 
326         /**
327          * Adds the tag information for this member.
328          *
329          * @param member the member to write the tags information for
330          * @param methodsContentTree content tree to which the tags
331          * information will be added
332          */
333         void addMemberTags(ExecutableElement member, Content methodsContentTree);
334     }
335 }
