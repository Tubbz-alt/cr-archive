26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;
32 import jdk.javadoc.doclet.Taglet.Location;
33 import jdk.javadoc.internal.doclets.toolkit.Content;
34 


35 /**
36  * An inline Taglet representing {&#064;docRoot}.  This taglet is
37  * used to get the relative path to the document's root output
38  * directory.
39  *
40  *  <p><b>This is NOT part of any supported API.
41  *  If you write code that depends on this, you do so at your own risk.
42  *  This code and its internal interfaces are subject to change or
43  *  deletion without notice.</b>



44  */
45 
46 public class DocRootTaglet extends BaseTaglet {
47 
48     /**
49      * Construct a new DocRootTaglet.
50      */
51     public DocRootTaglet() {
52         super(DocTree.Kind.DOC_ROOT, true, EnumSet.allOf(Location.class));
53     }
54 
55     @Override
56     public Content getTagletOutput(Element holder, DocTree tag, TagletWriter writer) {
57         return writer.getDocRootOutput();
58     }
59 }
