26 package jdk.javadoc.internal.doclets.toolkit.taglets;

28 import java.util.ArrayList;

30 import java.util.List;

32 import javax.lang.model.element.Element;
33 import javax.lang.model.element.ExecutableElement;
34 import javax.lang.model.type.TypeMirror;

36 import com.sun.source.doctree.DocTree;
37 import jdk.javadoc.doclet.Taglet.Location;
38 import jdk.javadoc.internal.doclets.toolkit.Content;

45 /**
46  * A taglet that represents the @return tag.
47  *
48  *  <p><b>This is NOT part of any supported API.
49  *  If you write code that depends on this, you do so at your own risk.
50  *  This code and its internal interfaces are subject to change or
51  *  deletion without notice.</b>


52  */
53 public class ReturnTaglet extends BaseTaglet implements InheritableTaglet {
54 
55     public ReturnTaglet() {
56         super(DocTree.Kind.RETURN, false, EnumSet.of(Location.METHOD));
57     }
58 
59     @Override
60     public void inherit(DocFinder.Input input, DocFinder.Output output) {
61         List<? extends DocTree> tags = input.utils.getBlockTags(input.element, DocTree.Kind.RETURN);
62         CommentHelper ch = input.utils.getCommentHelper(input.element);
63         if (!tags.isEmpty()) {
64             output.holder = input.element;
65             output.holderTag = tags.get(0);
66             output.inlineTags = input.isFirstSentence
67                     ? ch.getFirstSentenceTrees(output.holderTag)
68                     : ch.getDescription(output.holderTag);
69         }
70     }
71 
72     @Override
73     public Content getTagletOutput(Element holder, TagletWriter writer) {
74         Messages messages = writer.configuration().getMessages();
75         Utils utils = writer.configuration().utils;
76         TypeMirror returnType = utils.getReturnType(writer.getCurrentPageElement(), (ExecutableElement)holder);
77         List<? extends DocTree> tags = utils.getBlockTags(holder, DocTree.Kind.RETURN);
78 
79         //Make sure we are not using @return tag on method with void return type.
80         if (returnType != null && utils.isVoid(returnType)) {
81             if (!tags.isEmpty()) {
82                 messages.warning(holder, "doclet.Return_tag_on_void_method");
83             }
84             return null;
85         }
86         if (!tags.isEmpty())
87             return writer.returnTagOutput(holder, tags.get(0));
88         //Inherit @return tag if necessary.
89         List<DocTree> ntags = new ArrayList<>();
90         Input input = new DocFinder.Input(utils, holder, this);
91         DocFinder.Output inheritedDoc = DocFinder.search(writer.configuration(), input);
92         if (inheritedDoc.holderTag != null) {
93             CommentHelper ch = utils.getCommentHelper(input.element);
94             ch.setOverrideElement(inheritedDoc.holder);
95             ntags.add(inheritedDoc.holderTag);
96         }
97         return !ntags.isEmpty() ? writer.returnTagOutput(holder, ntags.get(0)) : null;
98     }
99 }
