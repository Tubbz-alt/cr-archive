 45 /**
 46  * Writes annotation type required member documentation in HTML format.
 47  *
 48  *  <p><b>This is NOT part of any supported API.
 49  *  If you write code that depends on this, you do so at your own risk.
 50  *  This code and its internal interfaces are subject to change or
 51  *  deletion without notice.</b>
 52  *
 53  * @author Jamie Ho
 54  * @author Bhavesh Patel (Modified)
 55  */
 56 public class AnnotationTypeRequiredMemberWriterImpl extends AbstractMemberWriter
 57     implements AnnotationTypeRequiredMemberWriter, MemberSummaryWriter {
 58 
 59     /**
 60      * Construct a new AnnotationTypeRequiredMemberWriterImpl.
 61      *
 62      * @param writer         the writer that will write the output.
 63      * @param annotationType the AnnotationType that holds this member.
 64      */
 65     public AnnotationTypeRequiredMemberWriterImpl(SubWriterHolderWriter writer,
 66             TypeElement annotationType) {
 67         super(writer, annotationType);
 68     }
 69 
 70     /**
 71      * {@inheritDoc}
 72      */
 73     public Content getMemberSummaryHeader(TypeElement typeElement,
 74             Content memberSummaryTree) {
 75         memberSummaryTree.add(
 76                 MarkerComments.START_OF_ANNOTATION_TYPE_REQUIRED_MEMBER_SUMMARY);
 77         Content memberTree = writer.getMemberTreeHeader();
 78         writer.addSummaryHeader(this, typeElement, memberTree);
 79         return memberTree;
 80     }
 81 
 82     /**
 83      * {@inheritDoc}
 84      */
 85     public Content getMemberTreeHeader() {
 86         return writer.getMemberTreeHeader();
 87     }
 88 
 89     /**
 90      * {@inheritDoc}
 91      */
 92     public void addMemberTree(Content memberSummaryTree, Content memberTree) {
 93         writer.addMemberTree(memberSummaryTree, memberTree);

 94     }
 95 
 96     /**
 97      * {@inheritDoc}
 98      */
 99     public void addAnnotationDetailsMarker(Content memberDetails) {
100         memberDetails.add(MarkerComments.START_OF_ANNOTATION_TYPE_DETAILS);
101     }
102 
103     /**
104      * {@inheritDoc}
105      */
106     public void addAnnotationDetailsTreeHeader(TypeElement te,
107             Content memberDetailsTree) {
108         if (!writer.printedAnnotationHeading) {
109             memberDetailsTree.add(links.createAnchor(
110                     SectionName.ANNOTATION_TYPE_ELEMENT_DETAIL));
111             Content heading = HtmlTree.HEADING(Headings.TypeDeclaration.DETAILS_HEADING,
112                     contents.annotationTypeDetailsLabel);
113             memberDetailsTree.add(heading);
114             writer.printedAnnotationHeading = true;
115         }

116     }
117 
118     /**
119      * {@inheritDoc}
120      */
121     @Override
122     public Content getAnnotationDocTreeHeader(Element member, Content annotationDetailsTree) {
123         String simpleName = name(member);
124         annotationDetailsTree.add(links.createAnchor(
125                 simpleName + utils.signature((ExecutableElement) member)));
126         Content annotationDocTree = writer.getMemberTreeHeader();
127         Content heading = new HtmlTree(Headings.TypeDeclaration.MEMBER_HEADING);
128         heading.add(simpleName);
129         annotationDocTree.add(heading);
130         return annotationDocTree;

131     }
132 
133     /**
134      * {@inheritDoc}
135      */
136     public Content getSignature(Element member) {
137         Content pre = new HtmlTree(HtmlTag.PRE);
138         writer.addAnnotationInfo(member, pre);
139         addModifiers(member, pre);
140         Content link =
141                 writer.getLink(new LinkInfoImpl(configuration,
142                         LinkInfoImpl.Kind.MEMBER, getType(member)));
143         pre.add(link);
144         pre.add(Contents.SPACE);
145         if (configuration.linksource) {
146             Content memberName = new StringContent(name(member));
147             writer.addSrcLink(member, memberName, pre);
148         } else {
149             addName(name(member), pre);
150         }
151         return pre;
152     }
153 
154     /**
155      * {@inheritDoc}
156      */
157     public void addDeprecated(Element member, Content annotationDocTree) {
158         addDeprecatedInfo(member, annotationDocTree);
159     }
160 
161     /**
162      * {@inheritDoc}
163      */
164     public void addComments(Element member, Content annotationDocTree) {
165         addComment(member, annotationDocTree);
166     }
167 
168     /**
169      * {@inheritDoc}
170      */
171     public void addTags(Element member, Content annotationDocTree) {
172         writer.addTagsInfo(member, annotationDocTree);
173     }
174 
175     /**
176      * {@inheritDoc}
177      */
178     public Content getAnnotationDetails(Content annotationDetailsTree) {
179         return HtmlTree.SECTION(getMemberTree(annotationDetailsTree));
180     }
181 
182     /**
183      * {@inheritDoc}
184      */
185     public Content getAnnotationDoc(Content annotationDocTree,
186             boolean isLastContent) {
187         return getMemberTree(annotationDocTree, isLastContent);
188     }
189 
190     /**
191      * {@inheritDoc}
192      */
193     public void addSummaryLabel(Content memberTree) {
194         Content label = HtmlTree.HEADING(Headings.TypeDeclaration.SUMMARY_HEADING,
195                 contents.annotateTypeRequiredMemberSummaryLabel);
196         memberTree.add(label);
197     }
198 
199     /**
200      * Get the caption for the summary table.
201      * @return the caption
202      */
203     // Overridden by AnnotationTypeOptionalMemberWriterImpl
204     protected Content getCaption() {
205         return contents.getContent("doclet.Annotation_Type_Required_Members");
206     }
207 
208     /**
209      * {@inheritDoc}
210      */
211     @Override
212     public TableHeader getSummaryTableHeader(Element member) {
213         return new TableHeader(contents.modifierAndTypeLabel,
214                 contents.annotationTypeRequiredMemberLabel, contents.descriptionLabel);
215     }
216 
217     /**
218      * {@inheritDoc}
219      */
220     @Override
221     protected Table createSummaryTable() {
222         return new Table(HtmlStyle.memberSummary)
223                 .setCaption(getCaption())
224                 .setHeader(getSummaryTableHeader(typeElement))
225                 .setRowScopeColumn(1)
226                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colSecond, HtmlStyle.colLast);
227     }
228 
229     /**
230      * {@inheritDoc}
231      */
232     public void addSummaryAnchor(TypeElement typeElement, Content memberTree) {
233         memberTree.add(links.createAnchor(
234                 SectionName.ANNOTATION_TYPE_REQUIRED_ELEMENT_SUMMARY));
235     }
236 
237     /**
238      * {@inheritDoc}
239      */
240     public void addInheritedSummaryAnchor(TypeElement typeElement, Content inheritedTree) {
241     }
242 
243     /**
244      * {@inheritDoc}
245      */
246     public void addInheritedSummaryLabel(TypeElement typeElement, Content inheritedTree) {
247     }
248 
249     /**
250      * {@inheritDoc}
251      */
252     protected void addSummaryLink(LinkInfoImpl.Kind context, TypeElement typeElement, Element member,
253             Content tdSummary) {
254         Content memberLink = HtmlTree.SPAN(HtmlStyle.memberNameLink,
255                 writer.getDocLink(context, member, name(member), false));
256         Content code = HtmlTree.CODE(memberLink);
257         tdSummary.add(code);
258     }
259 
260     /**
261      * {@inheritDoc}
262      */
263     protected void addInheritedSummaryLink(TypeElement typeElement,
264             Element member, Content linksTree) {
265         //Not applicable.
266     }
267 
268     /**
269      * {@inheritDoc}
270      */
271     protected void addSummaryType(Element member, Content tdSummaryType) {
272         addModifierAndType(member, getType(member), tdSummaryType);
273     }
274 
275     /**
276      * {@inheritDoc}
277      */
278     protected Content getDeprecatedLink(Element member) {
279         String name = utils.getFullyQualifiedName(member) + "." + member.getSimpleName();
280         return writer.getDocLink(LinkInfoImpl.Kind.MEMBER, member, name);
281     }
282 
283     private TypeMirror getType(Element member) {
284         return utils.isExecutableElement(member)
285                 ? utils.getReturnType((ExecutableElement) member)
286                 : member.asType();
287     }
288 }
