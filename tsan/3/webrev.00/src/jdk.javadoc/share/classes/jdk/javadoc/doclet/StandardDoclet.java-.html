26 package jdk.javadoc.doclet;
27 
28 import java.util.Locale;
29 import java.util.Set;
30 
31 import javax.lang.model.SourceVersion;
32 
33 import jdk.javadoc.internal.doclets.formats.html.HtmlDoclet;
34 
35 /**
36  * This doclet generates HTML-formatted documentation for the specified modules,
37  * packages and types.
38  *
39  * @see <a href="{@docRoot}/../specs/doc-comment-spec.html">
40  *      Documentation Comment Specification for the Standard Doclet</a>
41  */
42 public class StandardDoclet implements Doclet {
43 
44     private final HtmlDoclet htmlDoclet;
45 
46     public StandardDoclet() {
47         htmlDoclet = new HtmlDoclet(this);
48     }
49 
50     @Override
51     public void init(Locale locale, Reporter reporter) {
52         htmlDoclet.init(locale, reporter);
53     }
54 
55     @Override
56     public String getName() {
57         return "Standard";
58     }
59 
60     @Override
61     public Set<Doclet.Option> getSupportedOptions() {
62         return htmlDoclet.getSupportedOptions();
63     }
64 
65     @Override
66     public SourceVersion getSupportedSourceVersion() {
67         return htmlDoclet.getSupportedSourceVersion();
68     }
69 
70     @Override
71     public boolean run(DocletEnvironment docEnv) {
72         return htmlDoclet.run(docEnv);
73     }
74 }
