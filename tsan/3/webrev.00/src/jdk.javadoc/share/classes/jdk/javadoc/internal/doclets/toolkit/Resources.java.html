 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.text.MessageFormat;
 29 import java.util.Locale;
 30 import java.util.MissingResourceException;
 31 import java.util.ResourceBundle;
 32 
 33 /**
 34  * Access to the localizable resources used by a doclet.
 35  * The resources are split across two resource bundles:
 36  * one that contains format-neutral strings common to
 37  * all supported formats, and one that contains strings
 38  * specific to the selected doclet, such as the standard
 39  * HTML doclet.
 40  */
 41 public class Resources {
 42     public final String annotationTypeSummary;
 43     public final String classSummary;
 44     public final String enumSummary;
 45     public final String errorSummary;
 46     public final String exceptionSummary;
 47     public final String interfaceSummary;
 48     public final String packageSummary;
 49     public final String recordSummary;
 50 
 51     protected ResourceBundle commonBundle;
 52     protected ResourceBundle docletBundle;
 53 
 54     /**
 55      * Creates a {@code Resources} object to provide access the resource
 56      * bundles used by a doclet.
 57      *
 58      * @param locale           the locale to be used when accessing the
 59      *                         resource bundles.
 60      * @param commonBundleName the name of the bundle containing the strings
 61      *                         common to all output formats
 62      * @param docletBundleName the name of the bundle containing the strings
 63      *                         specific to a particular format
 64      */
 65     public Resources(Locale locale, String commonBundleName, String docletBundleName) {
 66         this.commonBundle = ResourceBundle.getBundle(commonBundleName, locale);
 67         this.docletBundle = ResourceBundle.getBundle(docletBundleName, locale);
 68 
 69         this.annotationTypeSummary = getText("doclet.Annotation_Types_Summary");
 70         this.classSummary = getText("doclet.Class_Summary");
 71         this.enumSummary = getText("doclet.Enum_Summary");
 72         this.errorSummary = getText("doclet.Error_Summary");
 73         this.exceptionSummary = getText("doclet.Exception_Summary");
 74         this.interfaceSummary = getText("doclet.Interface_Summary");
 75         this.packageSummary = getText("doclet.Package_Summary");
 76         this.recordSummary = getText("doclet.Record_Summary");
 77     }
 78 
 79     /**
 80      * Returns the string for the given key from one of the doclet's
 81      * resource bundles.
 82      *
 83      * The more specific bundle is checked first;
 84      * if it is not there, the common bundle is then checked.
 85      *
 86      * @param key the key for the desired string
 87      * @return the string for the given key
 88      * @throws MissingResourceException if the key is not found in either
 89      *                                  bundle.
 90      */
 91     public String getText(String key) throws MissingResourceException {
 92         if (docletBundle.containsKey(key))
 93             return docletBundle.getString(key);
 94 
 95         return commonBundle.getString(key);
 96     }
 97     /**
 98      * Returns the string for the given key from one of the doclet's
 99      * resource bundles, substituting additional arguments into
100      * into the resulting string with {@link MessageFormat#format}.
101      *
102      * The more specific bundle is checked first;
103      * if it is not there, the common bundle is then checked.
104      *
105      * @param key the key for the desired string
106      * @param args values to be substituted into the resulting string
107      * @return the string for the given key
108      * @throws MissingResourceException if the key is not found in either
109      *  bundle.
110      */
111     public String getText(String key, Object... args) throws MissingResourceException {
112         return MessageFormat.format(getText(key), args);
113     }
114 }
