26 package jdk.javadoc.internal.tool;
27 
28 import java.util.EnumSet;
29 
30 import javax.tools.JavaFileObject;
31 
32 import com.sun.tools.javac.api.JavacTrees;
33 import com.sun.tools.javac.code.Symbol.PackageSymbol;
34 import com.sun.tools.javac.code.ClassFinder;
35 import com.sun.tools.javac.util.Context;
36 import com.sun.tools.javac.util.Context.Factory;
37 
38 /** Javadoc uses an extended class finder that records package.html entries
39  *
40  *  <p><b>This is NOT part of any supported API.
41  *  If you write code that depends on this, you do so at your own risk.
42  *  This code and its internal interfaces are subject to change or
43  *  deletion without notice.</b>
44  *
45  *  @author Neal Gafter
46  */
47 public class JavadocClassFinder extends ClassFinder {
48 
49     public static JavadocClassFinder instance(Context context) {
50         ClassFinder instance = context.get(classFinderKey);
51         if (instance == null)
52             instance = new JavadocClassFinder(context);
53         return (JavadocClassFinder)instance;
54     }
55 
56     public static void preRegister(Context context) {
57         context.put(classFinderKey, (Factory<ClassFinder>)JavadocClassFinder::new);
58     }
59 
60     private ToolEnvironment toolEnv;
61     private EnumSet<JavaFileObject.Kind> all = EnumSet.of(JavaFileObject.Kind.CLASS,
62                                                           JavaFileObject.Kind.SOURCE,
63                                                           JavaFileObject.Kind.HTML);
64     private EnumSet<JavaFileObject.Kind> noSource = EnumSet.of(JavaFileObject.Kind.CLASS,
65                                                                JavaFileObject.Kind.HTML);
66     private JavacTrees jctrees;
67 
68     private final JavacTrees trees;
69 
70     public JavadocClassFinder(Context context) {
71         super(context);
72         toolEnv = ToolEnvironment.instance(context);
73         preferSource = true;
74         trees = JavacTrees.instance(context);
75     }
76 
77     /**
78      * Override getPackageFileKinds to include search for package.html
79      */
80     @Override
81     protected EnumSet<JavaFileObject.Kind> getPackageFileKinds() {
82         return toolEnv.docClasses ? noSource : all;
83     }
84 
85     /**
86      * Override extraFileActions to check for package documentation
87      */
88     @Override
89     protected void extraFileActions(PackageSymbol pack, JavaFileObject fo) {
90         if (fo.isNameCompatible("package", JavaFileObject.Kind.HTML)) {
91             pack.sourcefile = fo;
92         }
93     }
94 }
