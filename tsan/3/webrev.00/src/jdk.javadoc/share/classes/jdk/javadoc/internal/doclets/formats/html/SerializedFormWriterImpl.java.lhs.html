 43 /**
 44  *  Generates the Serialized Form Information Page, <i>serialized-form.html</i>.
 45  *
 46  *  <p><b>This is NOT part of any supported API.
 47  *  If you write code that depends on this, you do so at your own risk.
 48  *  This code and its internal interfaces are subject to change or
 49  *  deletion without notice.</b>
 50  *
 51  * @author Atul M Dambalkar
 52  */
 53 public class SerializedFormWriterImpl extends SubWriterHolderWriter
 54     implements SerializedFormWriter {
 55 
 56     Set<TypeElement> visibleClasses;
 57 
 58     /**
 59      * HTML tree for main tag.
 60      */
 61     private HtmlTree mainTree = HtmlTree.MAIN();
 62 
 63     private final Navigation navBar;
 64 
 65     /**
 66      * @param configuration the configuration data for the doclet
 67      */
 68     public SerializedFormWriterImpl(HtmlConfiguration configuration) {
 69         super(configuration, DocPaths.SERIALIZED_FORM);
 70         visibleClasses = configuration.getIncludedTypeElements();
 71         this.navBar = new Navigation(null, configuration, fixedNavDiv, PageMode.SERIALIZEDFORM, path);
 72     }
 73 
 74     /**
 75      * Get the given header.
 76      *
 77      * @param header the header to write
 78      * @return the body content tree
 79      */

 80     public Content getHeader(String header) {
 81         HtmlTree bodyTree = getBody(true, getWindowTitle(header));
 82         HtmlTree htmlTree = HtmlTree.HEADER();
 83         addTop(htmlTree);
 84         navBar.setUserHeader(getUserHeaderFooter(true));
 85         htmlTree.add(navBar.getContent(true));
 86         bodyTree.add(htmlTree);
 87         Content h1Content = new StringContent(header);
 88         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
 89                 HtmlStyle.title, h1Content);
 90         Content div = HtmlTree.DIV(HtmlStyle.header, heading);
 91         mainTree.add(div);

 92         return bodyTree;
 93     }
 94 
 95     /**
 96      * Get the serialized form summaries header.
 97      *
 98      * @return the serialized form summary header tree
 99      */

100     public Content getSerializedSummariesHeader() {
101         HtmlTree ul = new HtmlTree(HtmlTag.UL);
102         ul.setStyle(HtmlStyle.blockList);
103         return ul;
104     }
105 
106     /**
107      * Get the package serialized form header.
108      *
109      * @return the package serialized form header tree
110      */

111     public Content getPackageSerializedHeader() {
112         return HtmlTree.SECTION();
113     }
114 
115     /**
116      * Get the given package header.
117      *
118      * @param packageName the package header to write
119      * @return a content tree for the package header
120      */

121     public Content getPackageHeader(String packageName) {
122         Content heading = HtmlTree.HEADING(Headings.SerializedForm.PACKAGE_HEADING, true,
123                 contents.packageLabel);
124         heading.add(Contents.SPACE);
125         heading.add(packageName);
126         return heading;
127     }
128 
129     /**
130      * Get the serialized class header.
131      *
132      * @return a content tree for the serialized class header
133      */

134     public Content getClassSerializedHeader() {
135         HtmlTree ul = new HtmlTree(HtmlTag.UL);
136         ul.setStyle(HtmlStyle.blockList);
137         return ul;
138     }
139 
140     /**
141      * Checks if a class is generated and is visible.
142      *
143      * @param typeElement the class being processed.
144      * @return true if the class, that is being processed, is generated and is visible.
145      */
146     public boolean isVisibleClass(TypeElement typeElement) {
147         return visibleClasses.contains(typeElement) && configuration.isGeneratedDoc(typeElement);
148     }
149 
150     /**
151      * Get the serializable class heading.
152      *
153      * @param typeElement the class being processed
154      * @return a content tree for the class header
155      */

156     public Content getClassHeader(TypeElement typeElement) {
157         Content classLink = (isVisibleClass(typeElement))
158                 ? getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.DEFAULT, typeElement)
159                         .label(configuration.getClassName(typeElement)))
160                 : new StringContent(utils.getFullyQualifiedName(typeElement));
161         Content li = HtmlTree.LI(HtmlStyle.blockList, links.createAnchor(
162                 utils.getFullyQualifiedName(typeElement)));
163         Content superClassLink = typeElement.getSuperclass() != null
164                 ? getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.SERIALIZED_FORM,
165                         typeElement.getSuperclass()))
166                 : null;
167 
168         //Print the heading.
169         Content className = superClassLink == null ?
170             contents.getContent(
171             "doclet.Class_0_implements_serializable", classLink) :
172             contents.getContent(
173             "doclet.Class_0_extends_implements_serializable", classLink,
174             superClassLink);
175         li.add(HtmlTree.HEADING(Headings.SerializedForm.CLASS_HEADING, className));
176         return li;
177     }
178 
179     /**
180      * Get the serial UID info header.
181      *
182      * @return a content tree for the serial uid info header
183      */

184     public Content getSerialUIDInfoHeader() {
185         HtmlTree dl = new HtmlTree(HtmlTag.DL);
186         dl.setStyle(HtmlStyle.nameValue);
187         return dl;
188     }
189 
190     /**
191      * Adds the serial UID info.
192      *
193      * @param header the header that will show up before the UID.
194      * @param serialUID the serial UID to print.
195      * @param serialUidTree the serial UID content tree to which the serial UID
196      *                      content will be added
197      */
198     public void addSerialUIDInfo(String header, String serialUID,
199             Content serialUidTree) {



200         Content headerContent = new StringContent(header);
201         serialUidTree.add(HtmlTree.DT(headerContent));
202         Content serialContent = new StringContent(serialUID);
203         serialUidTree.add(HtmlTree.DD(serialContent));
204     }
205 
206     /**
207      * Get the class serialize content header.
208      *
209      * @return a content tree for the class serialize content header
210      */

211     public Content getClassContentHeader() {
212         HtmlTree ul = new HtmlTree(HtmlTag.UL);
213         ul.setStyle(HtmlStyle.blockList);
214         return ul;
215     }
216 
217     /**
218      * Get the serialized content tree section.
219      *
220      * @param serializedTreeContent the serialized content tree to be added
221      * @return a div content tree
222      */
223     public Content getSerializedContent(Content serializedTreeContent) {

224         HtmlTree divContent = HtmlTree.DIV(HtmlStyle.serializedFormContainer,
225                 serializedTreeContent);
226         mainTree.add(divContent);
227         return mainTree;
228     }
229 
230     /**
231      * {@inheritDoc}
232      */
233     public void addPackageSerializedTree(Content serializedSummariesTree,
234             Content packageSerializedTree) {

235         serializedSummariesTree.add(HtmlTree.LI(HtmlStyle.blockList, packageSerializedTree));
236     }
237 
238     /**
239      * Add the footer.
240      *
241      * @param serializedTree the serialized tree to be added
242      */
243     public void addFooter(Content serializedTree) {

244         Content htmlTree = HtmlTree.FOOTER();
245         navBar.setUserFooter(getUserHeaderFooter(false));
246         htmlTree.add(navBar.getContent(false));
247         addBottom(htmlTree);
248         serializedTree.add(htmlTree);
249     }
250 
251     /**
252      * {@inheritDoc}
253      */
254     @Override
255     public void printDocument(Content serializedTree) throws DocFileIOException {

256         printHtmlDocument(null, "serialized forms", serializedTree);
257     }
