117     /**
118      * Start the generation of files. Call generate methods in the individual
119      * writers, which will in turn generate the documentation files. Call the
120      * TreeWriter generation first to ensure the Class Hierarchy is built
121      * first and then can be used in the later generation.
122      *
123      * For new format.
124      *
125      * @throws DocletException if there is a problem while writing the other files
126      */
127     @Override // defined by AbstractDoclet
128     protected void generateOtherFiles(DocletEnvironment docEnv, ClassTree classtree)
129             throws DocletException {
130         super.generateOtherFiles(docEnv, classtree);
131         HtmlOptions options = configuration.getOptions();
132         if (options.linkSource()) {
133             SourceToHTMLConverter.convertRoot(configuration,
134                 docEnv, DocPaths.SOURCE_OUTPUT);
135         }
136         // Modules with no documented classes may be specified on the
137         // command line to specify a service provider, allow these.
138         if (configuration.getSpecifiedModuleElements().isEmpty() &&
139                 configuration.topFile.isEmpty()) {
140             messages.error("doclet.No_Non_Deprecated_Classes_To_Document");
141             return;
142         }
143         boolean nodeprecated = options.noDeprecated();
144         performCopy(options.helpFile());
145         performCopy(options.stylesheetFile());
146         for (String stylesheet : options.additionalStylesheets()) {
147             performCopy(stylesheet);
148         }
149         // do early to reduce memory footprint
150         if (options.classUse()) {
151             ClassUseWriter.generate(configuration, classtree);
152         }

153 
154         if (options.createTree()) {
155             TreeWriter.generate(configuration, classtree);
156         }
157 
158         if (!(options.noDeprecatedList() || nodeprecated)) {
159             DeprecatedListWriter.generate(configuration);
160         }
161 
162         if (options.createOverview()) {







163             if (configuration.showModules) {
164                 ModuleIndexWriter.generate(configuration);
165             } else {
166                 PackageIndexWriter.generate(configuration);
167             }
168         }
169 
170         if (options.createIndex()) {
171             IndexBuilder indexBuilder = new IndexBuilder(configuration, nodeprecated);
172             configuration.buildSearchTagIndex();
173             if (options.splitIndex()) {
174                 SplitIndexWriter.generate(configuration, indexBuilder);
175             } else {
176                 SingleIndexWriter.generate(configuration, indexBuilder);
177             }
178             AllClassesIndexWriter.generate(configuration,
179                     new IndexBuilder(configuration, nodeprecated, true));
180             if (!configuration.packages.isEmpty()) {
181                 AllPackagesIndexWriter.generate(configuration);
182             }
183             SystemPropertiesWriter.generate(configuration);
184         }
185 
186         if (options.createOverview()) {
187             IndexRedirectWriter.generate(configuration, DocPaths.OVERVIEW_SUMMARY, DocPaths.INDEX);
188         } else {
189             IndexRedirectWriter.generate(configuration);


190         }
191 
192         if (options.helpFile().isEmpty() && !options.noHelp()) {
193             HelpWriter.generate(configuration);
194         }
195         // If a stylesheet file is not specified, copy the default stylesheet
196         // and replace newline with platform-specific newline.
197         DocFile f;
198         if (options.stylesheetFile().length() == 0) {
199             f = DocFile.createFileForOutput(configuration, DocPaths.STYLESHEET);
200             f.copyResource(DocPaths.RESOURCES.resolve(DocPaths.STYLESHEET), true, true);
201         }
202         f = DocFile.createFileForOutput(configuration, DocPaths.JAVASCRIPT);
203         f.copyResource(DocPaths.RESOURCES.resolve(DocPaths.JAVASCRIPT), true, true);
204         if (options.createIndex()) {
205             f = DocFile.createFileForOutput(configuration, DocPaths.SEARCH_JS);
206             f.copyResource(DOCLET_RESOURCES.resolve(DocPaths.SEARCH_JS), true, true);
207 
208             f = DocFile.createFileForOutput(configuration, DocPaths.RESOURCES.resolve(DocPaths.GLASS_IMG));
209             f.copyResource(DOCLET_RESOURCES.resolve(DocPaths.GLASS_IMG), true, false);
210 
211             f = DocFile.createFileForOutput(configuration, DocPaths.RESOURCES.resolve(DocPaths.X_IMG));
212             f.copyResource(DOCLET_RESOURCES.resolve(DocPaths.X_IMG), true, false);
213             copyJqueryFiles();
214         }
215     }

245     @Override // defined by AbstractDoclet
246     protected void generateClassFiles(SortedSet<TypeElement> typeElems, ClassTree classTree)
247             throws DocletException {
248         for (TypeElement te : typeElems) {
249             if (utils.hasHiddenTag(te) ||
250                     !(configuration.isGeneratedDoc(te) && utils.isIncluded(te))) {

251                 continue;
252             }
253             if (utils.isAnnotationType(te)) {
254                 AbstractBuilder annotationTypeBuilder =
255                     configuration.getBuilderFactory()
256                         .getAnnotationTypeBuilder(te);
257                 annotationTypeBuilder.build();
258             } else {
259                 AbstractBuilder classBuilder =
260                     configuration.getBuilderFactory().getClassBuilder(te, classTree);
261                 classBuilder.build();
262             }
263         }
264     }
265 



266     @Override // defined by AbstractDoclet
267     protected void generateModuleFiles() throws DocletException {
268         if (configuration.showModules) {



269             List<ModuleElement> mdles = new ArrayList<>(configuration.modulePackages.keySet());
270             for (ModuleElement mdle : mdles) {




271                 AbstractBuilder moduleSummaryBuilder =
272                         configuration.getBuilderFactory().getModuleSummaryBuilder(mdle);
273                 moduleSummaryBuilder.build();
274             }
275         }
276     }
277 



278     @Override // defined by AbstractDoclet
279     protected void generatePackageFiles(ClassTree classtree) throws DocletException {
280         HtmlOptions options = configuration.getOptions();
281         Set<PackageElement> packages = configuration.packages;



282         List<PackageElement> pList = new ArrayList<>(packages);
283         for (PackageElement pkg : pList) {
284             // if -nodeprecated option is set and the package is marked as
285             // deprecated, do not generate the package-summary.html, package-frame.html
286             // and package-tree.html pages for that package.
287             if (!(options.noDeprecated() && utils.isDeprecated(pkg))) {



288                 AbstractBuilder packageSummaryBuilder =
289                         configuration.getBuilderFactory().getPackageSummaryBuilder(pkg);
290                 packageSummaryBuilder.build();
291                 if (options.createTree()) {
292                     PackageTreeWriter.generate(configuration, pkg, options.noDeprecated());
293                 }
294             }
295         }
296     }

298     @Override // defined by Doclet
299     public Set<? extends Option> getSupportedOptions() {
300         return configuration.getOptions().getSupportedOptions();
301     }
302 
303     private void performCopy(String filename) throws DocFileIOException {
304         if (filename.isEmpty())
305             return;
306 
307         DocFile fromfile = DocFile.createFileForInput(configuration, filename);
308         DocPath path = DocPath.create(fromfile.getName());
309         DocFile toFile = DocFile.createFileForOutput(configuration, path);
310         if (toFile.isSameFile(fromfile))
311             return;
312 
313         messages.notice("doclet.Copying_File_0_To_File_1",
314                 fromfile.toString(), path.getPath());
315         toFile.copyFile(fromfile);
316     }
