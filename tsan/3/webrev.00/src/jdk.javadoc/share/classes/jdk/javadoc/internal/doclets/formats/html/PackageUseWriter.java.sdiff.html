 60 public class PackageUseWriter extends SubWriterHolderWriter {
 61 
 62     final PackageElement packageElement;
 63     final SortedMap<String, Set<TypeElement>> usingPackageToUsedClasses = new TreeMap<>();
 64     protected HtmlTree mainTree = HtmlTree.MAIN();
 65     final String packageUseTableSummary;
 66     private final Navigation navBar;
 67 
 68     /**
 69      * Constructor.
 70      *
 71      * @param configuration the configuration
 72      * @param mapper a mapper to provide details of where elements are used
 73      * @param filename the file to be generated
 74      * @param pkgElement the package element to be documented
 75      */
 76     public PackageUseWriter(HtmlConfiguration configuration,
 77                             ClassUseMapper mapper, DocPath filename,
 78                             PackageElement pkgElement) {
 79         super(configuration, configuration.docPaths.forPackage(pkgElement).resolve(filename));
 80         this.packageElement = pkgElement;
 81 
 82         // by examining all classes in this package, find what packages
 83         // use these classes - produce a map between using package and
 84         // used classes.
 85         for (TypeElement usedClass : utils.getEnclosedTypeElements(pkgElement)) {
 86             Set<TypeElement> usingClasses = mapper.classToClass.get(usedClass);
 87             if (usingClasses != null) {
 88                 for (TypeElement usingClass : usingClasses) {
 89                     PackageElement usingPackage = utils.containingPackage(usingClass);
 90                     Set<TypeElement> usedClasses = usingPackageToUsedClasses
 91                             .get(utils.getPackageName(usingPackage));
 92                     if (usedClasses == null) {
 93                         usedClasses = new TreeSet<>(utils.makeGeneralPurposeComparator());
 94                         usingPackageToUsedClasses.put(utils.getPackageName(usingPackage),
 95                                                       usedClasses);
 96                     }
 97                     usedClasses.add(usedClass);
 98                 }
 99             }
100         }
101 
102         packageUseTableSummary = resources.getText("doclet.Use_Table_Summary",
103                 resources.getText("doclet.packages"));
104         this.navBar = new Navigation(packageElement, configuration, fixedNavDiv, PageMode.USE, path);
105     }
106 
107     /**
108      * Generate a class page.
109      *
110      * @param configuration the current configuration of the doclet.
111      * @param mapper        the mapping of the class usage.
112      * @param pkgElement    the package being documented.
113      * @throws DocFileIOException if there is a problem generating the package use page
114      */
115     public static void generate(HtmlConfiguration configuration,
116                                 ClassUseMapper mapper, PackageElement pkgElement)
117             throws DocFileIOException {
118         DocPath filename = DocPaths.PACKAGE_USE;
119         PackageUseWriter pkgusegen = new PackageUseWriter(configuration, mapper, filename, pkgElement);
120         pkgusegen.generatePackageUseFile();
121     }
122 
123     /**
124      * Generate the package use list.
125      * @throws DocFileIOException if there is a problem generating the package use page
126      */
127     protected void generatePackageUseFile() throws DocFileIOException {
128         HtmlTree body = getPackageUseHeader();
129         HtmlTree div = new HtmlTree(HtmlTag.DIV);
130         div.setStyle(HtmlStyle.contentContainer);
131         if (usingPackageToUsedClasses.isEmpty()) {
132             div.add(contents.getContent("doclet.ClassUse_No.usage.of.0", utils.getPackageName(packageElement)));
133         } else {
134             addPackageUse(div);
135         }
136         mainTree.add(div);
137         body.add(mainTree);
138         HtmlTree footer = HtmlTree.FOOTER();
139         navBar.setUserFooter(getUserHeaderFooter(false));
140         footer.add(navBar.getContent(false));
141         addBottom(footer);
142         body.add(footer);

143         printHtmlDocument(null,
144                 getDescription("use", packageElement),
145                 body);
146     }
147 
148     /**
149      * Add the package use information.
150      *
151      * @param contentTree the content tree to which the package use information will be added
152      */
153     protected void addPackageUse(Content contentTree) {
154         HtmlTree ul = new HtmlTree(HtmlTag.UL);
155         ul.setStyle(HtmlStyle.blockList);
156         if (configuration.packages.size() > 1) {
157             addPackageList(ul);
158         }
159         addClassList(ul);
160         contentTree.add(ul);
161     }
162 
163     /**
164      * Add the list of packages that use the given package.
165      *
166      * @param contentTree the content tree to which the package list will be added
167      */
168     protected void addPackageList(Content contentTree) {
169         Content caption = contents.getContent(
170                 "doclet.ClassUse_Packages.that.use.0",
171                 getPackageLink(packageElement, utils.getPackageName(packageElement)));
172         Table table = new Table(HtmlStyle.useSummary)
173                 .setCaption(caption)
174                 .setHeader(getPackageTableHeader())
175                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);
176         for (String pkgname: usingPackageToUsedClasses.keySet()) {
177             PackageElement pkg = utils.elementUtils.getPackageElement(pkgname);
178             Content packageLink = links.createLink(getPackageAnchorName(pkg),
179                     new StringContent(utils.getPackageName(pkg)));
180             Content summary = new ContentBuilder();
181             if (pkg != null && !pkg.isUnnamed()) {
182                 addSummaryComment(pkg, summary);
183             } else {
184                 summary.add(Contents.SPACE);
185             }
186             table.addRow(packageLink, summary);
187         }
188         Content li = HtmlTree.LI(HtmlStyle.blockList, table.toContent());
189         contentTree.add(li);
190     }
191 
192     /**
193      * Add the list of classes that use the given package.
194      *
195      * @param contentTree the content tree to which the class list will be added
196      */
197     protected void addClassList(Content contentTree) {
198         TableHeader classTableHeader = new TableHeader(
199                 contents.classLabel, contents.descriptionLabel);


200         for (String packageName : usingPackageToUsedClasses.keySet()) {
201             PackageElement usingPackage = utils.elementUtils.getPackageElement(packageName);
202             HtmlTree li = new HtmlTree(HtmlTag.LI);
203             li.setStyle(HtmlStyle.blockList);
204             li.add(links.createAnchor(getPackageAnchorName(usingPackage)));
205             String tableSummary = resources.getText("doclet.Use_Table_Summary",
206                                                         resources.getText("doclet.classes"));
207             Content caption = contents.getContent(
208                     "doclet.ClassUse_Classes.in.0.used.by.1",
209                     getPackageLink(packageElement, utils.getPackageName(packageElement)),
210                     getPackageLink(usingPackage, utils.getPackageName(usingPackage)));
211             Table table = new Table(HtmlStyle.useSummary)
212                     .setCaption(caption)
213                     .setHeader(classTableHeader)
214                     .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);
215             for (TypeElement te : usingPackageToUsedClasses.get(packageName)) {
216                 DocPath dp = pathString(te,
217                         DocPaths.CLASS_USE.resolve(docPaths.forName(te)));
218                 Content stringContent = new StringContent(utils.getSimpleName(te));
219                 Content typeContent = links.createLink(dp.fragment(getPackageAnchorName(usingPackage)),
220                         stringContent);
221                 Content summary = new ContentBuilder();
222                 addIndexComment(te, summary);
223 
224                 table.addRow(typeContent, summary);
225             }
226             li.add(table.toContent());
227             contentTree.add(li);
228         }


229     }

231     /**
232      * Get the header for the package use listing.
233      *
234      * @return a content tree representing the package use header
235      */
236     private HtmlTree getPackageUseHeader() {
237         String packageText = resources.getText("doclet.Package");
238         String name = packageElement.isUnnamed() ? "" : utils.getPackageName(packageElement);
239         String title = resources.getText("doclet.Window_ClassUse_Header", packageText, name);
240         HtmlTree bodyTree = getBody(true, getWindowTitle(title));
241         HtmlTree htmlTree = HtmlTree.HEADER();
242         addTop(htmlTree);
243         Content linkContent = getModuleLink(utils.elementUtils.getModuleOf(packageElement),
244                 contents.moduleLabel);
245         navBar.setNavLinkModule(linkContent);
246         navBar.setUserHeader(getUserHeaderFooter(true));
247         htmlTree.add(navBar.getContent(true));
248         bodyTree.add(htmlTree);
249         ContentBuilder headContent = new ContentBuilder();
250         headContent.add(contents.getContent("doclet.ClassUse_Title", packageText));
251         headContent.add(new HtmlTree(HtmlTag.BR));
252         headContent.add(name);
253         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
254                 HtmlStyle.title, headContent);
255         Content div = HtmlTree.DIV(HtmlStyle.header, heading);
256         mainTree.add(div);

257         return bodyTree;
258     }

192      * @param contentTree the content tree to which the class list will be added

195         TableHeader classTableHeader = new TableHeader(
196                 contents.classLabel, contents.descriptionLabel);
197         HtmlTree ul = new HtmlTree(HtmlTag.UL);
198         ul.setStyle(HtmlStyle.blockList);

200             PackageElement usingPackage = utils.elementUtils.getPackageElement(packageName);
201             HtmlTree section = HtmlTree.SECTION(HtmlStyle.detail)
202                     .setId(getPackageAnchorName(usingPackage));

203             String tableSummary = resources.getText("doclet.Use_Table_Summary",
204                                                         resources.getText("doclet.classes"));
205             Content caption = contents.getContent(
206                     "doclet.ClassUse_Classes.in.0.used.by.1",

212                     .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast);

214                 DocPath dp = pathString(te,
215                         DocPaths.CLASS_USE.resolve(docPaths.forName(te)));

223             }
224             section.add(table.toContent());
225             ul.add(HtmlTree.LI(HtmlStyle.blockList, section));
226         }
227         Content li = HtmlTree.SECTION(HtmlStyle.packageUses, ul);
228         contentTree.add(li);
229     }
