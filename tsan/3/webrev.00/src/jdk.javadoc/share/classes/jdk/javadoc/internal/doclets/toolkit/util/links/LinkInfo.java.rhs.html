 26 package jdk.javadoc.internal.doclets.toolkit.util.links;
 27 
 28 import javax.lang.model.element.ExecutableElement;
 29 import javax.lang.model.element.TypeElement;
 30 import javax.lang.model.type.TypeMirror;
 31 
 32 import jdk.javadoc.internal.doclets.toolkit.BaseConfiguration;
 33 import jdk.javadoc.internal.doclets.toolkit.Content;
 34 
 35 /**
 36  *  Encapsulates information about a link.
 37  *
 38  *  <p><b>This is NOT part of any supported API.
 39  *  If you write code that depends on this, you do so at your own risk.
 40  *  This code and its internal interfaces are subject to change or
 41  *  deletion without notice.</b>


 42  */
 43 public abstract class LinkInfo {
 44 
 45     /**
 46      * The class we want to link to.  Null if we are not linking
 47      * to a class.
 48      */
 49     public TypeElement typeElement;
 50 
 51     /**
 52      * The executable element we want to link to.  Null if we are not linking
 53      * to an executable element.
 54      */
 55     public ExecutableElement executableElement;
 56 
 57     /**
 58      * The Type we want to link to.  Null if we are not linking to a type.
 59      */
 60     public TypeMirror type;
 61 
 62     /**
 63      * True if this is a link to a VarArg.
 64      */
 65     public boolean isVarArg = false;
 66 
 67     /**
 68      * Set this to true to indicate that you are linking to a type parameter.
 69      */
 70     public boolean isTypeBound = false;
 71 






 72     /**
 73      * The label for the link.
 74      */
 75     public Content label;
 76 
 77     /**
 78      * True if the link should be strong.
 79      */
 80     public boolean isStrong = false;
 81 
 82     /**
 83      * True if we should include the type in the link label.  False otherwise.
 84      */
 85     public boolean includeTypeInClassLinkLabel = true;
 86 
 87     /**
 88      * True if we should include the type as separate link.  False otherwise.
 89      */
 90     public boolean includeTypeAsSepLink = false;
 91 
 92     /**
 93      * True if we should exclude the type bounds for the type parameter.
 94      */
 95     public boolean excludeTypeBounds = false;
 96 
 97     /**
 98      * True if we should print the type parameters, but not link them.
 99      */
100     public boolean excludeTypeParameterLinks = false;
101 
102     /**
103      * True if we should print the type bounds, but not link them.
104      */
105     public boolean excludeTypeBoundsLinks = false;
106 
107     /**
108      * By default, the link can be to the page it's already on.  However,
109      * there are cases where we don't want this (e.g. heading of class page).
110      */
111     public boolean linkToSelf = true;
112 
113     /**
114      * Return an empty instance of a content object.
115      *
116      * @return an empty instance of a content object.
117      */
118     protected abstract Content newContent();
119 
120     /**
121      * Return true if this link is linkable and false if we can't link to the
122      * desired place.
123      *
124      * @return true if this link is linkable and false if we can't link to the
125      * desired place.
126      */
127     public abstract boolean isLinkable();
128 
129     /**
130      * Return the label for this class link.
131      *
132      * @param configuration the current configuration of the doclet.
133      * @return the label for this class link.
134      */
135     public Content getClassLinkLabel(BaseConfiguration configuration) {
136         if (label != null && !label.isEmpty()) {
137             return label;
138         } else if (isLinkable()) {
139             Content tlabel = newContent();
140             tlabel.add(configuration.utils.getSimpleName(typeElement));
141             return tlabel;
142         } else {
143             Content tlabel = newContent();
144             tlabel.add(configuration.getClassName(typeElement));
145             return tlabel;
146         }
147     }
148 
149     @Override
150     public String toString() {
151         return "LinkInfo{" + "typeElement=" + typeElement +
152                 ", executableElement=" + executableElement +
153                 ", type=" + type +
154                 ", isVarArg=" + isVarArg +
155                 ", isTypeBound=" + isTypeBound +

156                 ", label=" + label +
157                 ", isStrong=" + isStrong +
158                 ", includeTypeInClassLinkLabel=" + includeTypeInClassLinkLabel +
159                 ", includeTypeAsSepLink=" + includeTypeAsSepLink +
160                 ", excludeTypeBounds=" + excludeTypeBounds +
161                 ", excludeTypeParameterLinks=" + excludeTypeParameterLinks +
162                 ", excludeTypeBoundsLinks=" + excludeTypeBoundsLinks +
163                 ", linkToSelf=" + linkToSelf + '}';
164     }
165 }
