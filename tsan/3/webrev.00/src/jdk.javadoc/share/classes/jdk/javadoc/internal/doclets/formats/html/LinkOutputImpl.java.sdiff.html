41 
42     /**
43      * The output of the link.
44      */
45     public StringBuilder output;
46 
47     /**
48      * Construct a new LinkOutputImpl.
49      */
50     public LinkOutputImpl() {
51         output = new StringBuilder();
52     }
53 
54     /**
55      * {@inheritDoc}
56      */
57     public void append(Object o) {
58         output.append(o instanceof String ?
59             (String) o : ((LinkOutputImpl)o).toString());
60     }
61 
62     /**
63      * {@inheritDoc}
64      */
65     public void insert(int offset, Object o) {
66         output.insert(offset, o.toString());
67     }
68 
69     /**
70      * {@inheritDoc}
71      */
72     public String toString() {
73         return output.toString();
74     }

76 }
