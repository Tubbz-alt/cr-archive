 42 /**
 43  *  Javadoc's own memberEnter phase does a few things above and beyond that
 44  *  done by javac.
 45  *
 46  *  <p><b>This is NOT part of any supported API.
 47  *  If you write code that depends on this, you do so at your own risk.
 48  *  This code and its internal interfaces are subject to change or
 49  *  deletion without notice.</b>


 50  */
 51 public class JavadocMemberEnter extends MemberEnter {
 52     public static JavadocMemberEnter instance0(Context context) {
 53         MemberEnter instance = context.get(memberEnterKey);
 54         if (instance == null)
 55             instance = new JavadocMemberEnter(context);
 56         return (JavadocMemberEnter)instance;
 57     }
 58 
 59     public static void preRegister(Context context) {
 60         context.put(memberEnterKey, (Context.Factory<MemberEnter>)JavadocMemberEnter::new);
 61     }
 62 
 63     final ToolEnvironment toolEnv;
 64 
 65 
 66     protected JavadocMemberEnter(Context context) {
 67         super(context);
 68         toolEnv = ToolEnvironment.instance(context);
 69     }
 70 
 71     @Override
 72     public void visitMethodDef(JCMethodDecl tree) {
 73         super.visitMethodDef(tree);
 74         MethodSymbol meth = tree.sym;
 75         if (meth == null || meth.kind != MTH) return;
 76         TreePath treePath = toolEnv.getTreePath(env.toplevel, env.enclClass, tree);
 77         // do not add those methods that may be mandated by the spec,
 78         // or those that are synthesized, thus if it does not exist in
 79         // tree best to let other logic determine the TreePath.
 80         if (env.enclClass.defs.contains(tree)) {
 81             toolEnv.setElementToTreePath(meth, treePath);
 82         }
 83         // release resources
 84         // handle constructors for record types specially, because of downstream checks
 85         if ((env.enclClass.mods.flags & Flags.RECORD) != 0 && TreeInfo.isConstructor(tree)) {
 86             tree.body.stats = List.nil();
 87         } else {
 88         tree.body = null;
 89         }
 90     }
 91 
 92     @Override
 93     public void visitVarDef(JCVariableDecl tree) {
 94         if (tree.init != null) {
 95             boolean isFinal = (tree.mods.flags & FINAL) != 0
 96                     || (env.enclClass.mods.flags & INTERFACE) != 0;
 97             if (!isFinal || containsNonConstantExpression(tree.init)) {
 98                 // Avoid unnecessary analysis and release resources.
 99                 // In particular, remove non-constant expressions
100                 // which may trigger Attr.attribClass, since
101                 // method bodies are also removed, in visitMethodDef.
102                 tree.init = null;
103             }
104         }
105         super.visitVarDef(tree);
106         if (tree.sym != null && tree.sym.kind == VAR && !isParameter(tree.sym)) {
107             toolEnv.setElementToTreePath(tree.sym, toolEnv.getTreePath(env.toplevel, env.enclClass, tree));
108         }
109     }
110 
111     private static boolean isParameter(VarSymbol var) {
112         return (var.flags() & Flags.PARAMETER) != 0;
113     }
114 
115     /**
116      * Simple analysis of an expression tree to see if it contains tree nodes
117      * for any non-constant expression. This does not include checking references
118      * to other fields which may or may not be constant.
119      */
120     private static boolean containsNonConstantExpression(JCExpression tree) {
121         return new MaybeConstantExpressionScanner().containsNonConstantExpression(tree);
122     }
123 
124     /**
125      * See JLS 15.18, Constant Expression
126      */
127     private static class MaybeConstantExpressionScanner extends JCTree.Visitor {
128         boolean maybeConstantExpr = true;
129 
130         public boolean containsNonConstantExpression(JCExpression tree) {
131             scan(tree);
132             return !maybeConstantExpr;
133         }
134 
135         public void scan(JCTree tree) {
136             // short circuit scan when end result is definitely false
137             if (maybeConstantExpr && tree != null)
138                 tree.accept(this);
139         }
140 
141         @Override
142         /** default for any non-overridden visit method. */
143         public void visitTree(JCTree tree) {
144             maybeConstantExpr = false;
145         }
146 
147         @Override
148         public void visitBinary(JCBinary tree) {
149             switch (tree.getTag()) {
150                 case MUL: case DIV: case MOD:
151                 case PLUS: case MINUS:
152                 case SL: case SR: case USR:
153                 case LT: case LE: case GT: case GE:
154                 case EQ: case NE:
155                 case BITAND: case BITXOR: case BITOR:
156                 case AND: case OR:
157                     break;
158                 default:
159                     maybeConstantExpr = false;
160             }
161         }
162 
163         @Override
164         public void visitConditional(JCConditional tree) {
165             scan(tree.cond);
166             scan(tree.truepart);
167             scan(tree.falsepart);
168         }
169 
170         @Override
171         public void visitIdent(JCIdent tree) { }
172 
173         @Override
174         public void visitLiteral(JCLiteral tree) { }
175 
176         @Override
177         public void visitParens(JCParens tree) {
178             scan(tree.expr);
179         }
180 
181         @Override
182         public void visitSelect(JCTree.JCFieldAccess tree) {
183             scan(tree.selected);
184         }
185 
186         @Override
187         public void visitTypeCast(JCTypeCast tree) {
188             scan(tree.clazz);
189             scan(tree.expr);
190         }
191 
192         @Override
193         public void visitTypeIdent(JCPrimitiveTypeTree tree) { }
194 
195         @Override
196         public void visitUnary(JCUnary tree) {
197             switch (tree.getTag()) {
198                 case POS: case NEG: case COMPL: case NOT:
199                     break;
200                 default:
201                     maybeConstantExpr = false;
202             }
203         }
204     }
205 }
