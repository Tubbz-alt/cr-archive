 26 package jdk.javadoc.internal.doclets.formats.html;
 27 
 28 import jdk.javadoc.internal.doclets.formats.html.markup.Table;
 29 import jdk.javadoc.internal.doclets.formats.html.markup.TableHeader;
 30 
 31 import java.util.Arrays;
 32 import java.util.List;
 33 
 34 import javax.lang.model.element.Element;
 35 import javax.lang.model.element.TypeElement;
 36 
 37 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlStyle;
 38 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTree;
 39 import jdk.javadoc.internal.doclets.formats.html.markup.StringContent;
 40 import jdk.javadoc.internal.doclets.toolkit.Content;
 41 import jdk.javadoc.internal.doclets.toolkit.MemberSummaryWriter;
 42 
 43 /**
 44  * Writes nested class documentation in HTML format.
 45  *
 46  *  <p><b>This is NOT part of any supported API.
 47  *  If you write code that depends on this, you do so at your own risk.
 48  *  This code and its internal interfaces are subject to change or
 49  *  deletion without notice.</b>
 50  *
 51  * @author Robert Field
 52  * @author Atul M Dambalkar
 53  * @author Jamie Ho (rewrite)
 54  * @author Bhavesh Patel (Modified)
 55  */
 56 public class NestedClassWriterImpl extends AbstractMemberWriter
 57     implements MemberSummaryWriter {
 58 
 59     public NestedClassWriterImpl(SubWriterHolderWriter writer, TypeElement typeElement) {
 60         super(writer, typeElement);
 61     }
 62 
 63     public NestedClassWriterImpl(SubWriterHolderWriter writer) {
 64         super(writer);
 65     }
 66 
 67     /**
 68      * {@inheritDoc}
 69      */
 70     @Override
 71     public Content getMemberSummaryHeader(TypeElement typeElement,
 72             Content memberSummaryTree) {
 73         memberSummaryTree.add(MarkerComments.START_OF_NESTED_CLASS_SUMMARY);
 74         Content memberTree = writer.getMemberTreeHeader();
 75         writer.addSummaryHeader(this, typeElement, memberTree);
 76         return memberTree;
 77     }
 78 
 79     /**
 80      * {@inheritDoc}
 81      */
 82     @Override
 83     public void addMemberTree(Content memberSummaryTree, Content memberTree) {
 84         writer.addMemberTree(memberSummaryTree, memberTree);
 85     }
 86 
 87     /**
 88      * {@inheritDoc}
 89      */
 90     @Override
 91     public void addSummaryLabel(Content memberTree) {
 92         Content label = HtmlTree.HEADING(Headings.TypeDeclaration.SUMMARY_HEADING,
 93                 contents.nestedClassSummary);
 94         memberTree.add(label);
 95     }
 96 
 97     /**
 98      * {@inheritDoc}
 99      */
100     @Override
101     public TableHeader getSummaryTableHeader(Element member) {
102         Content label = utils.isInterface(member) ?
103                 contents.interfaceLabel : contents.classLabel;
104 
105         return new TableHeader(contents.modifierAndTypeLabel, label, contents.descriptionLabel);
106     }
107 
108     @Override
109     protected Table createSummaryTable() {
110         List<HtmlStyle> bodyRowStyles = Arrays.asList(HtmlStyle.colFirst, HtmlStyle.colSecond,
111                 HtmlStyle.colLast);
112 
113         return new Table(HtmlStyle.memberSummary)
114                 .setCaption(contents.getContent("doclet.Nested_Classes"))
115                 .setHeader(getSummaryTableHeader(typeElement))
116                 .setRowScopeColumn(1)
117                 .setColumnStyles(bodyRowStyles);
118     }
119 
120     /**
121      * {@inheritDoc}
122      */
123     @Override
124     public void addSummaryAnchor(TypeElement typeElement, Content memberTree) {
125         memberTree.add(links.createAnchor(
126                 SectionName.NESTED_CLASS_SUMMARY));
127     }
128 
129     /**
130      * {@inheritDoc}
131      */
132     @Override
133     public void addInheritedSummaryAnchor(TypeElement typeElement, Content inheritedTree) {
134         inheritedTree.add(links.createAnchor(
135                 SectionName.NESTED_CLASSES_INHERITANCE,
136                 utils.getFullyQualifiedName(typeElement)));
137     }
138 
139     /**
140      * {@inheritDoc}
141      */
142     @Override
143     public void addInheritedSummaryLabel(TypeElement typeElement, Content inheritedTree) {
144         Content classLink = writer.getPreQualifiedClassLink(
145                 LinkInfoImpl.Kind.MEMBER, typeElement, false);
146         Content label;
147         if (configuration.summarizeOverriddenMethods) {
148             label = new StringContent(utils.isInterface(typeElement)
149                     ? resources.getText("doclet.Nested_Classes_Interfaces_Declared_In_Interface")
150                     : resources.getText("doclet.Nested_Classes_Interfaces_Declared_In_Class"));
151         } else {
152             label = new StringContent(utils.isInterface(typeElement)
153                     ? resources.getText("doclet.Nested_Classes_Interfaces_Inherited_From_Interface")
154                     : resources.getText("doclet.Nested_Classes_Interfaces_Inherited_From_Class"));
155         }
156         Content labelHeading = HtmlTree.HEADING(Headings.TypeDeclaration.SUMMARY_HEADING, label);
157         labelHeading.add(Contents.SPACE);
158         labelHeading.add(classLink);
159         inheritedTree.add(labelHeading);
160     }
161 
162     /**
163      * {@inheritDoc}
164      */
165     @Override
166     protected void addSummaryLink(LinkInfoImpl.Kind context, TypeElement typeElement, Element member,
167             Content tdSummary) {
168         Content memberLink = HtmlTree.SPAN(HtmlStyle.memberNameLink,
169                 writer.getLink(new LinkInfoImpl(configuration, context, (TypeElement)member)));
170         Content code = HtmlTree.CODE(memberLink);
171         tdSummary.add(code);
172     }
173 
174     /**
175      * {@inheritDoc}
176      */
177     @Override
178     protected void addInheritedSummaryLink(TypeElement typeElement, Element member, Content linksTree) {
179         linksTree.add(
180                 writer.getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.MEMBER,
181                         (TypeElement)member)));
182     }
183 
184     /**
185      * {@inheritDoc}
186      */
187     @Override
188     protected void addSummaryType(Element member, Content tdSummaryType) {
189         addModifierAndType(member, null, tdSummaryType);
190     }
191 
192     /**
193      * {@inheritDoc}
194      */
195     @Override
196     protected Content getDeprecatedLink(Element member) {
197         return writer.getQualifiedClassLink(LinkInfoImpl.Kind.MEMBER, member);
198     }
199 }
