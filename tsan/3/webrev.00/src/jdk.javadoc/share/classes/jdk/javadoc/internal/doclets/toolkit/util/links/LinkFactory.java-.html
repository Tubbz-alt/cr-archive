 26 package jdk.javadoc.internal.doclets.toolkit.util.links;
 27 
 28 import java.util.List;
 29 
 30 import javax.lang.model.element.Element;
 31 import javax.lang.model.element.TypeElement;
 32 import javax.lang.model.element.TypeParameterElement;
 33 import javax.lang.model.type.ArrayType;
 34 import javax.lang.model.type.DeclaredType;
 35 import javax.lang.model.type.TypeMirror;
 36 import javax.lang.model.type.TypeVariable;
 37 import javax.lang.model.type.WildcardType;
 38 import javax.lang.model.util.SimpleTypeVisitor9;
 39 
 40 import jdk.javadoc.internal.doclets.toolkit.Content;
 41 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
 42 
 43 /**
 44  * A factory that constructs links from given link information.
 45  *
 46  *  <p><b>This is NOT part of any supported API.
 47  *  If you write code that depends on this, you do so at your own risk.
 48  *  This code and its internal interfaces are subject to change or
 49  *  deletion without notice.</b>
 50  *
 51  * @author Jamie Ho
 52  */
 53 public abstract class LinkFactory {
 54     protected final Utils utils;
 55 
 56     protected LinkFactory(Utils utils) {
 57         this.utils = utils;
 58     }
 59 
 60     /**
 61      * Return an empty instance of a content object.
 62      *
 63      * @return an empty instance of a content object.
 64      */
 65     protected abstract Content newContent();
 66 
 67     /**
 68      * Constructs a link from the given link information.
 69      *
 70      * @param linkInfo the information about the link.
 71      * @return the output of the link.
 72      */
 73     public Content getLink(LinkInfo linkInfo) {
 74         if (linkInfo.type != null) {
 75             SimpleTypeVisitor9<Content, LinkInfo> linkVisitor =
 76                     new SimpleTypeVisitor9<Content, LinkInfo>() {
 77 
 78                 TypeMirror componentType = utils.getComponentType(linkInfo.type);
 79                 Content link = newContent();
 80 
 81                 // handles primitives, no types and error types
 82                 @Override
 83                 protected Content defaultAction(TypeMirror type, LinkInfo linkInfo) {
 84                     link.add(utils.getTypeName(type, false));
 85                     return link;
 86                 }
 87 
 88                 int currentDepth = 0;
 89                 @Override
 90                 public Content visitArray(ArrayType type, LinkInfo linkInfo) {
 91                     // keep track of the dimension depth and replace the last dimension
 92                     // specifier with vararags, when the stack is fully unwound.
 93                     currentDepth++;
 94                     linkInfo.type = type.getComponentType();
 95                     visit(linkInfo.type, linkInfo);
 96                     currentDepth--;
 97                     if (utils.isAnnotated(type)) {
 98                         linkInfo.type = type;
 99                         link.add(" ");
100                         link.add(getTypeAnnotationLinks(linkInfo));
101                     }
102                     // use vararg if required
103                     if (linkInfo.isVarArg && currentDepth == 0) {
104                         link.add("...");
105                     } else {
106                         link.add("[]");
107                     }
108                     return link;
109                 }
110 
111                 @Override
112                 public Content visitWildcard(WildcardType type, LinkInfo linkInfo) {
113                     linkInfo.isTypeBound = true;
114                     link.add("?");
115                     TypeMirror extendsBound = type.getExtendsBound();
116                     if (extendsBound != null) {
117                         link.add(" extends ");
118                         setBoundsLinkInfo(linkInfo, extendsBound);
119                         link.add(getLink(linkInfo));
120                     }
121                     TypeMirror superBound = type.getSuperBound();
122                     if (superBound != null) {
123                         link.add(" super ");
124                         setBoundsLinkInfo(linkInfo, superBound);
125                         link.add(getLink(linkInfo));
126                     }
127                     return link;
128                 }
129 
130                 @Override
131                 public Content visitTypeVariable(TypeVariable type, LinkInfo linkInfo) {
132                     link.add(getTypeAnnotationLinks(linkInfo));
133                     linkInfo.isTypeBound = true;
134                     TypeVariable typevariable = (utils.isArrayType(type))
135                             ? (TypeVariable) componentType
136                             : type;
137                     Element owner = typevariable.asElement().getEnclosingElement();
138                     if ((!linkInfo.excludeTypeParameterLinks) && utils.isTypeElement(owner)) {
139                         linkInfo.typeElement = (TypeElement) owner;
140                         Content label = newContent();
141                         label.add(utils.getTypeName(type, false));
142                         linkInfo.label = label;
143                         link.add(getClassLink(linkInfo));
144                     } else {
145                         // No need to link method type parameters.
146                         link.add(utils.getTypeName(typevariable, false));
147                     }
148 
149                     if (!linkInfo.excludeTypeBounds) {
150                         linkInfo.excludeTypeBounds = true;
151                         TypeParameterElement tpe = ((TypeParameterElement) typevariable.asElement());
152                         boolean more = false;
153                         List<? extends TypeMirror> bounds = utils.getBounds(tpe);
154                         for (TypeMirror bound : bounds) {
155                             // we get everything as extends java.lang.Object we suppress
156                             // all of them except those that have multiple extends
157                             if (bounds.size() == 1 &&
158                                     bound.equals(utils.getObjectType()) &&
159                                     !utils.isAnnotated(bound)) {
160                                 continue;
161                             }
162                             link.add(more ? " & " : " extends ");
163                             setBoundsLinkInfo(linkInfo, bound);
164                             link.add(getLink(linkInfo));
165                             more = true;
166                         }
167                     }
168                     return link;
169                 }
170 
171                 @Override
172                 public Content visitDeclared(DeclaredType type, LinkInfo linkInfo) {
173                     if (linkInfo.isTypeBound && linkInfo.excludeTypeBoundsLinks) {
174                         // Since we are excluding type parameter links, we should not
175                         // be linking to the type bound.
176                         link.add(utils.getTypeName(type, false));
177                         link.add(getTypeParameterLinks(linkInfo));
178                         return link;
179                     } else {
180                         link = newContent();
181                         link.add(getTypeAnnotationLinks(linkInfo));
182                         linkInfo.typeElement = utils.asTypeElement(type);
183                         link.add(getClassLink(linkInfo));
184                         if (linkInfo.includeTypeAsSepLink) {
185                             link.add(getTypeParameterLinks(linkInfo, false));
186                         }
187                     }
188                     return link;
189                 }
190             };
191             return linkVisitor.visit(linkInfo.type, linkInfo);
192         } else if (linkInfo.typeElement != null) {
193             Content link = newContent();
194             link.add(getClassLink(linkInfo));
195             if (linkInfo.includeTypeAsSepLink) {
196                 link.add(getTypeParameterLinks(linkInfo, false));
197             }
198             return link;
199         } else {
200             return null;
201         }
202     }
203 
204     private void setBoundsLinkInfo(LinkInfo linkInfo, TypeMirror bound) {
205         linkInfo.typeElement = null;
206         linkInfo.label = null;
207         linkInfo.type = bound;
208     }
209 
210     /**
211      * Returns a link to the given class.
212      *
213      * @param linkInfo the information about the link to construct
214      *
215      * @return the link for the given class.
216      */
217     protected abstract Content getClassLink(LinkInfo linkInfo);
218 
219     /**
220      * Returns links to the type parameters.
221      *
222      * @param linkInfo     the information about the link to construct
223      * @param isClassLabel true if this is a class label, or false if it is
224      *                     the type parameters portion of the link
225      * @return the links to the type parameters
226      */
227     protected abstract Content getTypeParameterLinks(LinkInfo linkInfo, boolean isClassLabel);
228 
229     /**
230      * Returns links to the type parameters.
231      *
232      * @param linkInfo     the information about the link to construct
233      * @return the links to the type parameters.
234      */
235     public Content getTypeParameterLinks(LinkInfo linkInfo) {
236         return getTypeParameterLinks(linkInfo, true);
237     }
238 
239     public abstract Content getTypeAnnotationLinks(LinkInfo linkInfo);
240 }
