 26 package jdk.javadoc.internal.doclets.formats.html.markup;
 27 
 28 import java.io.IOException;
 29 import java.io.StringWriter;
 30 import java.io.Writer;
 31 import java.util.*;
 32 
 33 import jdk.javadoc.internal.doclets.toolkit.Content;
 34 import jdk.javadoc.internal.doclets.toolkit.util.DocFile;
 35 import jdk.javadoc.internal.doclets.toolkit.util.DocFileIOException;
 36 import jdk.javadoc.internal.doclets.toolkit.util.DocletConstants;
 37 
 38 /**
 39  * Class for generating an HTML document for javadoc output.
 40  *
 41  *  <p><b>This is NOT part of any supported API.
 42  *  If you write code that depends on this, you do so at your own risk.
 43  *  This code and its internal interfaces are subject to change or
 44  *  deletion without notice.</b>
 45  *
 46  * @author Bhavesh Patel
 47  */
 48 public class HtmlDocument {
 49     private final DocType docType = DocType.HTML5;
 50     private final List<Content> docContent;
 51 
 52     /**
 53      * Constructor to construct an HTML document.
 54      *
 55      * @param docComment comment for the document
 56      * @param htmlTree HTML tree of the document
 57      */
 58     public HtmlDocument(Content docComment, Content htmlTree) {
 59         docContent = Arrays.asList(docComment, htmlTree);
 60     }
 61 
 62     /**
 63      * Constructor to construct an HTML document.
 64      *
 65      * @param htmlTree HTML tree of the document
 66      */
 67     public HtmlDocument(Content htmlTree) {
 68         docContent = Collections.singletonList(htmlTree);
 69     }
 70 
 71     /**
 72      * Writes the content of this document to the specified file.
 73      *
 74      * @param docFile the file
 75      * @throws DocFileIOException if an {@code IOException} occurs while writing the file
 76      */
 77     public void write(DocFile docFile) throws DocFileIOException {
 78         try (Writer writer = docFile.openWriter()) {
 79             write(writer);
 80         } catch (IOException e) {
 81             throw new DocFileIOException(docFile, DocFileIOException.Mode.WRITE, e);
 82         }
 83     }
 84 
 85     @Override
 86     public String toString() {
 87         try (Writer writer = new StringWriter()) {
 88             write(writer);
 89             return writer.toString();
 90         } catch (IOException e) {
 91             throw new Error(e);
 92         }
 93     }
 94 
 95     private void write(Writer writer) throws IOException {
 96         writer.write(docType.text);
 97         writer.write(DocletConstants.NL);
 98         boolean atNewline = true;
 99         for (Content c : docContent) {
100             atNewline = c.write(writer, atNewline);
101         }
102     }
103 }
