 49 /**
 50  * Writes constructor documentation.
 51  *
 52  *  <p><b>This is NOT part of any supported API.
 53  *  If you write code that depends on this, you do so at your own risk.
 54  *  This code and its internal interfaces are subject to change or
 55  *  deletion without notice.</b>
 56  *
 57  * @author Robert Field
 58  * @author Atul M Dambalkar
 59  * @author Bhavesh Patel (Modified)
 60  */
 61 public class ConstructorWriterImpl extends AbstractExecutableMemberWriter
 62     implements ConstructorWriter, MemberSummaryWriter {
 63 
 64     private boolean foundNonPubConstructor = false;
 65 
 66     /**
 67      * Construct a new ConstructorWriterImpl.
 68      *
 69      * @param writer The writer for the class that the constructors belong to.
 70      * @param typeElement the class being documented.
 71      */
 72     public ConstructorWriterImpl(SubWriterHolderWriter writer, TypeElement typeElement) {
 73         super(writer, typeElement);
 74 
 75         VisibleMemberTable vmt = configuration.getVisibleMemberTable(typeElement);
 76         List<? extends Element> constructors = vmt.getVisibleMembers(CONSTRUCTORS);
 77 
 78         for (Element constructor : constructors) {
 79             if (utils.isProtected(constructor) || utils.isPrivate(constructor)) {
 80                 setFoundNonPubConstructor(true);
 81             }
 82         }
 83     }
 84 
 85     /**
 86      * Construct a new ConstructorWriterImpl.
 87      *
 88      * @param writer The writer for the class that the constructors belong to.
 89      */
 90     public ConstructorWriterImpl(SubWriterHolderWriter writer) {
 91         super(writer);
 92     }
 93 
 94     /**
 95      * {@inheritDoc}
 96      */
 97     @Override
 98     public Content getMemberSummaryHeader(TypeElement typeElement,
 99             Content memberSummaryTree) {
100         memberSummaryTree.add(MarkerComments.START_OF_CONSTRUCTOR_SUMMARY);
101         Content memberTree = writer.getMemberTreeHeader();
102         writer.addSummaryHeader(this, typeElement, memberTree);
103         return memberTree;
104     }
105 
106     /**
107      * {@inheritDoc}
108      */
109     @Override
110     public void addMemberTree(Content memberSummaryTree, Content memberTree) {
111         writer.addMemberTree(memberSummaryTree, memberTree);

112     }
113 
114     /**
115      * {@inheritDoc}
116      */
117     @Override
118     public Content getConstructorDetailsTreeHeader(TypeElement typeElement,
119             Content memberDetailsTree) {
120         memberDetailsTree.add(MarkerComments.START_OF_CONSTRUCTOR_DETAILS);
121         Content constructorDetailsTree = writer.getMemberTreeHeader();
122         constructorDetailsTree.add(links.createAnchor(
123                 SectionName.CONSTRUCTOR_DETAIL));
124         Content heading = HtmlTree.HEADING(Headings.TypeDeclaration.DETAILS_HEADING,
125                 contents.constructorDetailsLabel);
126         constructorDetailsTree.add(heading);
127         return constructorDetailsTree;
128     }
129 
130     /**
131      * {@inheritDoc}
132      */
133     @Override
134     public Content getConstructorDocTreeHeader(ExecutableElement constructor,
135             Content constructorDetailsTree) {
136         String erasureAnchor;



137         if ((erasureAnchor = getErasureAnchor(constructor)) != null) {
138             constructorDetailsTree.add(links.createAnchor((erasureAnchor)));
139         }
140         constructorDetailsTree.add(links.createAnchor(writer.getAnchor(constructor)));
141         Content constructorDocTree = writer.getMemberTreeHeader();
142         Content heading = new HtmlTree(Headings.TypeDeclaration.MEMBER_HEADING);
143         heading.add(name(constructor));
144         constructorDocTree.add(heading);
145         return constructorDocTree;

146     }
147 
148     /**
149      * {@inheritDoc}
150      */
151     @Override
152     public Content getSignature(ExecutableElement constructor) {
153         Content pre = new HtmlTree(HtmlTag.PRE);
154         writer.addAnnotationInfo(constructor, pre);
155         int annotationLength = pre.charCount();
156         addModifiers(constructor, pre);
157         if (configuration.linksource) {
158             Content constructorName = new StringContent(name(constructor));
159             writer.addSrcLink(constructor, constructorName, pre);
160         } else {
161             addName(name(constructor), pre);
162         }
163         int indent = pre.charCount() - annotationLength;
164         addParameters(constructor, pre, indent);
165         addExceptions(constructor, pre, indent);
166         return pre;
167     }
168 
169     /**
170      * {@inheritDoc}
171      */
172     @Override
173     public void addDeprecated(ExecutableElement constructor, Content constructorDocTree) {
174         addDeprecatedInfo(constructor, constructorDocTree);
175     }
176 
177     /**
178      * {@inheritDoc}
179      */
180     @Override
181     public void addComments(ExecutableElement constructor, Content constructorDocTree) {
182         addComment(constructor, constructorDocTree);
183     }
184 
185     /**
186      * {@inheritDoc}
187      */
188     @Override
189     public void addTags(ExecutableElement constructor, Content constructorDocTree) {
190         writer.addTagsInfo(constructor, constructorDocTree);
191     }
192

194      * {@inheritDoc}

196     @Override
197     public Content getConstructorDetails(Content constructorDetailsTree) {
198         return HtmlTree.SECTION(getMemberTree(constructorDetailsTree));


199     }
200

202      * {@inheritDoc}
203      */
204     @Override
205     public Content getConstructorDoc(Content constructorDocTree,
206             boolean isLastContent) {
207         return getMemberTree(constructorDocTree, isLastContent);
208     }

211      * Let the writer know whether a non public constructor was found.
212      *
213      * @param foundNonPubConstructor true if we found a non public constructor.
214      */
215     @Override
216     public void setFoundNonPubConstructor(boolean foundNonPubConstructor) {
217         this.foundNonPubConstructor = foundNonPubConstructor;
218     }
219 
220     /**
221      * {@inheritDoc}
222      */
223     @Override
224     public void addSummaryLabel(Content memberTree) {
225         Content label = HtmlTree.HEADING(Headings.TypeDeclaration.SUMMARY_HEADING,
226                 contents.constructorSummaryLabel);
227         memberTree.add(label);
228     }
229 
230     /**
231      * {@inheritDoc}
232      */
233     @Override
234     public TableHeader getSummaryTableHeader(Element member) {
235         if (foundNonPubConstructor) {
236             return new TableHeader(contents.modifierLabel, contents.constructorLabel,
237                     contents.descriptionLabel);
238         } else {
239             return new TableHeader(contents.constructorLabel, contents.descriptionLabel);
240         }
241     }
242 
243     @Override
244     protected Table createSummaryTable() {
245         List<HtmlStyle> bodyRowStyles;
246         int rowScopeColumn;
247 
248         if (foundNonPubConstructor) {
249             bodyRowStyles = Arrays.asList(HtmlStyle.colFirst, HtmlStyle.colConstructorName,
250                     HtmlStyle.colLast);
251             rowScopeColumn = 1;
252         } else {
253             bodyRowStyles = Arrays.asList(HtmlStyle.colConstructorName, HtmlStyle.colLast);
254             rowScopeColumn = 0;
255         }
256 
257         return new Table(HtmlStyle.memberSummary)
258                 .setCaption(contents.constructors)
259                 .setHeader(getSummaryTableHeader(typeElement))
260                 .setRowScopeColumn(rowScopeColumn)
261                 .setColumnStyles(bodyRowStyles);
262     }
263 
264     /**
265      * {@inheritDoc}
266      */
267     @Override
268     public void addSummaryAnchor(TypeElement typeElement, Content memberTree) {
269         memberTree.add(links.createAnchor(SectionName.CONSTRUCTOR_SUMMARY));
270     }
271 
272     /**
273      * {@inheritDoc}
274      */
275     @Override
276     public void addInheritedSummaryAnchor(TypeElement typeElement, Content inheritedTree) {
277     }
278 
279     /**
280      * {@inheritDoc}
281      */
282     @Override
283     public void addInheritedSummaryLabel(TypeElement typeElement, Content inheritedTree) {
284     }
285 
286     /**
287      * {@inheritDoc}
288      */
289     @Override
290     protected void addSummaryType(Element member, Content tdSummaryType) {
291         if (foundNonPubConstructor) {
292             Content code = new HtmlTree(HtmlTag.CODE);
293             if (utils.isProtected(member)) {
294                 code.add("protected ");
295             } else if (utils.isPrivate(member)) {
296                 code.add("private ");
297             } else if (utils.isPublic(member)) {
298                 code.add(Contents.SPACE);
299             } else {
300                 code.add(
301                         resources.getText("doclet.Package_private"));
302             }
303             tdSummaryType.add(code);
304         }
305     }

117     @Override
118     public Content getConstructorDocTreeHeader(ExecutableElement constructor) {

119         String erasureAnchor;
120         Content constructorDocTree = new ContentBuilder();
121         HtmlTree heading = new HtmlTree(Headings.TypeDeclaration.MEMBER_HEADING,
