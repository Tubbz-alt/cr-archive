 25 package jdk.javadoc.internal.doclets.toolkit;
 26 
 27 import javax.lang.model.element.Element;
 28 import javax.tools.Diagnostic;
 29 
 30 import com.sun.source.util.DocTreePath;
 31 import jdk.javadoc.doclet.Reporter;
 32 
 33 import static javax.tools.Diagnostic.Kind.*;
 34 
 35 /**
 36  * Provides standardized access to the diagnostic reporting facilities
 37  * for a doclet.
 38  *
 39  * Messages are specified by resource keys to be found in the doclet's
 40  * {@link Resources resources}.  Values can be substituted into the
 41  * strings obtained from the resource files.
 42  *
 43  * Messages are reported to the doclet's {@link Reporter reporter}.
 44  */
 45 public class Messages {
 46     private final BaseConfiguration configuration;
 47     private final Resources resources;
 48     private final Reporter reporter;
 49 
 50     /**
 51      * Creates a {@code Messages} object to provide standardized access to
 52      * the doclet's diagnostic reporting mechanisms.
 53      *
 54      * @param configuration the doclet's configuration, used to access the doclet's
 55      *                      reporter, and additional methods and state used to
 56      *                      filter out messages, if any, which should be suppressed.
 57      * @param resources     resources for console messages and exceptions
 58      */
 59     public Messages(BaseConfiguration configuration, Resources resources) {
 60         this.configuration = configuration;
 61         this.resources = resources;
 62         reporter = configuration.getReporter();
 63     }
 64 
 65     /**
 66      * Returns the resources being used when generating messages.
 67      *
 68      * @return the resources
 69      */
 70     public Resources getResources() {
 71         return resources;
 72     }
 73 
 74     // ***** Errors *****
 75 
 76     /**
 77      * Reports an error message to the doclet's reporter.
 78      *
 79      * @param key the name of a resource containing the message to be printed
 80      * @param args optional arguments to be replaced in the message.
 81      */
 82     public void error(String key, Object... args) {
 83         report(ERROR, resources.getText(key, args));
 84     }
 85 
 86     /**
 87      * Reports an error message to the doclet's reporter.
 88      *
 89      * @param path a path identifying the position to be included with
 90      *  the message
 91      * @param key the name of a resource containing the message to be printed
 92      * @param args optional arguments to be replaced in the message.
 93      */
 94     public void error(DocTreePath path, String key, Object... args) {
 95         report(ERROR, path, resources.getText(key, args));
 96     }
 97 
 98     // ***** Warnings *****
 99 
100     /**
101      * Reports a warning message to the doclet's reporter.
102      *
103      * @param key the name of a resource containing the message to be printed
104      * @param args optional arguments to be replaced in the message.
105      */
106     public void warning(String key, Object... args) {
107         report(WARNING, resources.getText(key, args));
108     }
109 
110     /**
111      * Reports a warning message to the doclet's reporter.
112      *
113      * @param path a path identifying the position to be included with
114      *  the message
115      * @param key the name of a resource containing the message to be printed
116      * @param args optional arguments to be replaced in the message.
117      */
118     public void warning(DocTreePath path, String key, Object... args) {
119         if (configuration.showMessage(path, key))
120             report(WARNING, path, resources.getText(key, args));
121     }
122 
123     /**
124      * Reports a warning message to the doclet's reporter.
125      *
126      * @param e an element identifying the declaration whose position should
127      *  to be included with the message
128      * @param key the name of a resource containing the message to be printed
129      * @param args optional arguments to be replaced in the message.
130      */
131     public void warning(Element e, String key, Object... args) {
132         if (configuration.showMessage(e, key)) {
133             report(WARNING, e, resources.getText(key, args));
134         }
135     }
136 
137     // ***** Notices *****
138 
139     /**
140      * Reports an informational notice to the doclet's reporter.
141      *
142      * @param key the name of a resource containing the message to be printed
143      * @param args optional arguments to be replaced in the message.
144      */
145     public void notice(String key, Object... args) {
146         if (!configuration.getOptions().quiet()) {
147             report(NOTE, resources.getText(key, args));
148         }
149     }
150 
151     // ***** Internal support *****
152 
153     private void report(Diagnostic.Kind k, String msg) {
154         reporter.print(k, msg);
155     }
156 
157     private void report(Diagnostic.Kind k, DocTreePath p, String msg) {
158         reporter.print(k, p, msg);
159     }
160 
161     private void report(Diagnostic.Kind k, Element e, String msg) {
162         reporter.print(k, e, msg);
163     }
164 }
