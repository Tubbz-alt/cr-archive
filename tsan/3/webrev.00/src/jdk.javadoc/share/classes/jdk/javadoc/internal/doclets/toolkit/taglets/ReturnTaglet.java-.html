 26 package jdk.javadoc.internal.doclets.toolkit.taglets;
 27 
 28 import java.util.ArrayList;
 29 import java.util.EnumSet;
 30 import java.util.List;
 31 
 32 import javax.lang.model.element.Element;
 33 import javax.lang.model.element.ExecutableElement;
 34 import javax.lang.model.type.TypeMirror;
 35 
 36 import com.sun.source.doctree.DocTree;
 37 import jdk.javadoc.internal.doclets.toolkit.Content;
 38 import jdk.javadoc.internal.doclets.toolkit.Messages;
 39 import jdk.javadoc.internal.doclets.toolkit.util.CommentHelper;
 40 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder;
 41 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder.Input;
 42 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
 43 
 44 import static com.sun.source.doctree.DocTree.Kind.RETURN;
 45 
 46 /**
 47  * A taglet that represents the @return tag.
 48  *
 49  *  <p><b>This is NOT part of any supported API.
 50  *  If you write code that depends on this, you do so at your own risk.
 51  *  This code and its internal interfaces are subject to change or
 52  *  deletion without notice.</b>
 53  *
 54  * @author Jamie Ho
 55  */
 56 public class ReturnTaglet extends BaseTaglet implements InheritableTaglet {
 57 
 58     public ReturnTaglet() {
 59         super(RETURN.tagName, false, EnumSet.of(Site.METHOD));
 60     }
 61 
 62     @Override
 63     public void inherit(DocFinder.Input input, DocFinder.Output output) {
 64         List<? extends DocTree> tags = input.utils.getBlockTags(input.element, DocTree.Kind.RETURN);
 65         CommentHelper ch = input.utils.getCommentHelper(input.element);
 66         if (!tags.isEmpty()) {
 67             output.holder = input.element;
 68             output.holderTag = tags.get(0);
 69             output.inlineTags = input.isFirstSentence
 70                     ? ch.getFirstSentenceTrees(input.utils.configuration, output.holderTag)
 71                     : ch.getDescription(input.utils.configuration, output.holderTag);
 72         }
 73     }
 74 
 75     @Override
 76     public Content getTagletOutput(Element holder, TagletWriter writer) {
 77         Messages messages = writer.configuration().getMessages();
 78         Utils utils = writer.configuration().utils;
 79         TypeMirror returnType = utils.getReturnType((ExecutableElement)holder);
 80         List<? extends DocTree> tags = utils.getBlockTags(holder, name);
 81 
 82         //Make sure we are not using @return tag on method with void return type.
 83         if (returnType != null && utils.isVoid(returnType)) {
 84             if (!tags.isEmpty()) {
 85                 messages.warning(holder, "doclet.Return_tag_on_void_method");
 86             }
 87             return null;
 88         }
 89         if (!tags.isEmpty())
 90             return writer.returnTagOutput(holder, tags.get(0));
 91         //Inherit @return tag if necessary.
 92         List<DocTree> ntags = new ArrayList<>();
 93         Input input = new DocFinder.Input(utils, holder, this);
 94         DocFinder.Output inheritedDoc = DocFinder.search(writer.configuration(), input);
 95         if (inheritedDoc.holderTag != null) {
 96             CommentHelper ch = utils.getCommentHelper(input.element);
 97             ch.setOverrideElement(inheritedDoc.holder);
 98             ntags.add(inheritedDoc.holderTag);
 99         }
100         return !ntags.isEmpty() ? writer.returnTagOutput(holder, ntags.get(0)) : null;
101     }
102 }
