 26 package jdk.javadoc.internal.doclets.toolkit.taglets;
 27 
 28 import java.util.EnumSet;
 29 import javax.lang.model.element.Element;
 30 import javax.lang.model.element.ExecutableElement;
 31 
 32 import com.sun.source.doctree.DocTree;

 33 import jdk.javadoc.internal.doclets.toolkit.BaseConfiguration;
 34 import jdk.javadoc.internal.doclets.toolkit.Content;
 35 import jdk.javadoc.internal.doclets.toolkit.Messages;
 36 import jdk.javadoc.internal.doclets.toolkit.util.CommentHelper;
 37 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder;
 38 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder.Input;
 39 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
 40 
 41 import static com.sun.source.doctree.DocTree.Kind.INHERIT_DOC;
 42 
 43 /**
 44  * An inline Taglet representing the {@code inheritDoc} tag. This tag should only
 45  * be used with a method.  It is used to inherit documentation from overriden
 46  * and implemented methods.
 47  *
 48  *  <p><b>This is NOT part of any supported API.
 49  *  If you write code that depends on this, you do so at your own risk.
 50  *  This code and its internal interfaces are subject to change or
 51  *  deletion without notice.</b>
 52  *
 53  * @author Jamie Ho
 54  */
 55 
 56 public class InheritDocTaglet extends BaseTaglet {
 57 
 58     /**
 59      * Construct a new InheritDocTaglet.
 60      */
 61     public InheritDocTaglet () {
 62         super(INHERIT_DOC.tagName, true, EnumSet.of(Site.TYPE, Site.METHOD));
 63     }
 64 
 65     /**
 66      * Given an element, a {@code DocTree} in the element's doc comment
 67      * replace all occurrences of @inheritDoc with documentation from its
 68      * superclass or superinterface.
 69      *
 70      * @param writer the writer that is writing the output.
 71      * @param e the {@link Element} that we are documenting.
 72      * @param holderTag the tag that holds the inheritDoc tag or null for type
 73      * (class) docs.
 74      * @param isFirstSentence true if we only want to inherit the first sentence.
 75      */
 76     private Content retrieveInheritedDocumentation(TagletWriter writer,
 77             Element e, DocTree holderTag, boolean isFirstSentence) {
 78         Content replacement = writer.getOutputInstance();
 79         BaseConfiguration configuration = writer.configuration();
 80         Messages messages = configuration.getMessages();
 81         Utils utils = configuration.utils;
 82         CommentHelper ch = utils.getCommentHelper(e);
 83         Taglet inheritableTaglet = holderTag == null
 84                 ? null
 85                 : configuration.tagletManager.getTaglet(ch.getTagName(holderTag));
 86         if (inheritableTaglet != null &&
 87             !(inheritableTaglet instanceof InheritableTaglet)) {
 88                 String message = utils.getSimpleName(e) +
 89                     ((utils.isExecutableElement(e))
 90                         ? utils.flatSignature((ExecutableElement)e)
 91                         : "");
 92                 //This tag does not support inheritance.
 93                 messages.warning(e, "doclet.noInheritedDoc", message);
 94         }
 95         Input input = new DocFinder.Input(utils, e,
 96                 (InheritableTaglet) inheritableTaglet, new DocFinder.DocTreeInfo(holderTag, e),
 97                 isFirstSentence, true);
 98         DocFinder.Output inheritedDoc = DocFinder.search(configuration, input);
 99         if (inheritedDoc.isValidInheritDocTag) {
100             if (!inheritedDoc.inlineTags.isEmpty()) {
101                 replacement = writer.commentTagsToOutput(inheritedDoc.holderTag,
102                     inheritedDoc.holder, inheritedDoc.inlineTags, isFirstSentence);
103                 ch.setOverrideElement(inheritedDoc.holder);
104             }
105 
106         } else {
107             String message = utils.getSimpleName(e) +
108                     ((utils.isExecutableElement(e))
109                         ? utils.flatSignature((ExecutableElement)e)
110                         : "");
111             messages.warning(e, "doclet.noInheritedDoc", message);
112         }
113         return replacement;
114     }
115 
116     @Override
117     public Content getTagletOutput(Element e, DocTree tag, TagletWriter tagletWriter) {
118         DocTree inheritTag = (tag.getKind() == INHERIT_DOC) ? null : tag;
119         return retrieveInheritedDocumentation(tagletWriter, e,
120                 inheritTag, tagletWriter.isFirstSentence);
121     }
122 }
