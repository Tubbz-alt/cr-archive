 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import javax.lang.model.element.Element;
 29 import javax.lang.model.element.ModuleElement;
 30 import javax.lang.model.element.PackageElement;
 31 import javax.lang.model.element.TypeElement;
 32 
 33 import jdk.javadoc.internal.doclets.toolkit.util.ClassTree;
 34 import jdk.javadoc.internal.doclets.toolkit.util.VisibleMemberTable;

 36 /**
 37  * The interface for a factory creates writers.
 38  *
 39  *  <p><b>This is NOT part of any supported API.
 40  *  If you write code that depends on this, you do so at your own risk.
 41  *  This code and its internal interfaces are subject to change or
 42  *  deletion without notice.</b>
 43  *
 44  * @author Jamie Ho
 45  */
 46 
 47 public interface WriterFactory {
 48 
 49     /**
 50      * Return the writer for the constant summary.
 51      *
 52      * @return the writer for the constant summary.  Return null if this
 53      * writer is not supported by the doclet.
 54      */
 55     public abstract ConstantsSummaryWriter getConstantsSummaryWriter();
 56 
 57     /**
 58      * Return the writer for the package summary.
 59      *
 60      * @param packageElement the package being documented.
 61      * @return the writer for the package summary.  Return null if this
 62      * writer is not supported by the doclet.
 63      */
 64     public abstract PackageSummaryWriter getPackageSummaryWriter(PackageElement packageElement);
 65 
 66     /**
 67      * Return the writer for the module summary.
 68      *
 69      * @param mdle the module being documented.
 70      * @return the writer for the module summary.  Return null if this
 71      * writer is not supported by the doclet.
 72      */
 73     public abstract ModuleSummaryWriter getModuleSummaryWriter(ModuleElement mdle);
 74 
 75     /**
 76      * Return the writer for a class.
 77      *
 78      * @param typeElement the class being documented.
 79      * @param classTree the class tree.
 80      * @return the writer for the class.  Return null if this
 81      * writer is not supported by the doclet.
 82      */
 83     public abstract ClassWriter getClassWriter(TypeElement typeElement, ClassTree classTree);
 84 
 85     /**
 86      * Return the writer for an annotation type.
 87      *
 88      * @param annotationType the type being documented.
 89      * @return the writer for the annotation type.  Return null if this
 90      * writer is not supported by the doclet.
 91      */
 92     public abstract AnnotationTypeWriter getAnnotationTypeWriter(TypeElement annotationType);
 93 
 94     /**
 95      * Return the method writer for a given class.
 96      *
 97      * @param classWriter the writer for the class being documented.
 98      * @return the method writer for the give class.  Return null if this
 99      * writer is not supported by the doclet.
100      */
101     public abstract MethodWriter getMethodWriter(ClassWriter classWriter);
102 
103     /**
104      * Return the annotation type field writer for a given annotation type.
105      *
106      * @param annotationTypeWriter the writer for the annotation type
107      *        being documented.
108      * @return the member writer for the given annotation type.  Return null if
109      *         this writer is not supported by the doclet.
110      */
111     public abstract AnnotationTypeFieldWriter
112             getAnnotationTypeFieldWriter(
113         AnnotationTypeWriter annotationTypeWriter);
114 
115     /**
116      * Return the annotation type optional member writer for a given annotation
117      * type.
118      *
119      * @param annotationTypeWriter the writer for the annotation type
120      *        being documented.
121      * @return the member writer for the given annotation type.  Return null if
122      *         this writer is not supported by the doclet.
123      */
124     public abstract AnnotationTypeOptionalMemberWriter
125             getAnnotationTypeOptionalMemberWriter(
126         AnnotationTypeWriter annotationTypeWriter);
127 
128     /**
129      * Return the annotation type required member writer for a given annotation type.
130      *
131      * @param annotationTypeWriter the writer for the annotation type
132      *        being documented.
133      * @return the member writer for the given annotation type.  Return null if
134      *         this writer is not supported by the doclet.
135      */
136     public abstract AnnotationTypeRequiredMemberWriter
137             getAnnotationTypeRequiredMemberWriter(
138         AnnotationTypeWriter annotationTypeWriter);
139 
140     /**
141      * Return the enum constant writer for a given class.
142      *
143      * @param classWriter the writer for the class being documented.
144      * @return the enum constant writer for the give class.  Return null if this
145      * writer is not supported by the doclet.
146      */
147     public abstract EnumConstantWriter getEnumConstantWriter(
148         ClassWriter classWriter);
149 
150     /**
151      * Return the field writer for a given class.
152      *
153      * @param classWriter the writer for the class being documented.
154      * @return the field writer for the give class.  Return null if this
155      * writer is not supported by the doclet.
156      */
157     public abstract FieldWriter getFieldWriter(ClassWriter classWriter);
158 
159     /**
160      * Return the property writer for a given class.
161      *
162      * @param classWriter the writer for the class being documented.
163      * @return the property writer for the give class.  Return null if this
164      * writer is not supported by the doclet.
165      */
166     public abstract PropertyWriter getPropertyWriter(ClassWriter classWriter);
167 
168     /**
169      * Return the constructor writer for a given class.
170      *
171      * @param classWriter the writer for the class being documented.
172      * @return the method writer for the give class.  Return null if this
173      * writer is not supported by the doclet.
174      */
175     public abstract ConstructorWriter getConstructorWriter(
176         ClassWriter classWriter);
177 
178     /**
179      * Return the specified member summary writer for a given class.
180      *
181      * @param classWriter the writer for the class being documented.
182      * @param memberType  the {@link VisibleMemberTable} member type indicating
183      *                    the type of member summary that should be returned.
184      * @return the summary writer for the give class.  Return null if this
185      * writer is not supported by the doclet.
186      *
187      * @see VisibleMemberTable
188      */
189     public abstract MemberSummaryWriter getMemberSummaryWriter(
190         ClassWriter classWriter, VisibleMemberTable.Kind memberType);
191 
192     /**
193      * Return the specified member summary writer for a given annotation type.
194      *
195      * @param annotationTypeWriter the writer for the annotation type being
196      *                             documented.
197      * @param memberType  the {@link VisibleMemberTable} member type indicating
198      *                    the type of member summary that should be returned.
199      * @return the summary writer for the give class.  Return null if this
200      * writer is not supported by the doclet.
201      *
202      * @see VisibleMemberTable
203      */
204     public abstract MemberSummaryWriter getMemberSummaryWriter(
205         AnnotationTypeWriter annotationTypeWriter, VisibleMemberTable.Kind memberType);
206 
207     /**
208      * Return the writer for the serialized form.
209      *
210      * @return the writer for the serialized form.
211      */
212     public SerializedFormWriter getSerializedFormWriter();
213 
214     /**
215      * Return the handler for doc files.
216      *
217      * @return the handler for the doc files.
218      */
219     DocFilesHandler getDocFilesHandler(Element pkg);
220 }
