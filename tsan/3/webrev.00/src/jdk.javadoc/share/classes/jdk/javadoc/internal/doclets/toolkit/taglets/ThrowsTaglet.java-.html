 38 import jdk.javadoc.internal.doclets.toolkit.util.CommentHelper;
 39 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder;

 43 import static com.sun.source.doctree.DocTree.Kind.THROWS;

 45 /**
 46  * A taglet that represents the @throws tag.
 47  *
 48  *  <p><b>This is NOT part of any supported API.
 49  *  If you write code that depends on this, you do so at your own risk.
 50  *  This code and its internal interfaces are subject to change or
 51  *  deletion without notice.</b>
 52  *
 53  * @author Jamie Ho
 54  */
 55 public class ThrowsTaglet extends BaseTaglet
 56     implements InheritableTaglet {
 57 
 58     public ThrowsTaglet() {
 59         super(THROWS.tagName, false, EnumSet.of(Site.CONSTRUCTOR, Site.METHOD));
 60     }
 61 
 62     @Override
 63     public void inherit(DocFinder.Input input, DocFinder.Output output) {
 64         Utils utils = input.utils;
 65         Element exception;
 66         CommentHelper ch = utils.getCommentHelper(input.element);
 67         if (input.tagId == null) {
 68             exception = ch.getException(utils.configuration, input.docTreeInfo.docTree);
 69             input.tagId = exception == null
 70                     ? ch.getExceptionName(input.docTreeInfo.docTree).getSignature()
 71                     : utils.getFullyQualifiedName(exception);
 72         } else {
 73             TypeElement element = input.utils.findClass(input.element, input.tagId);
 74             exception = (element == null) ? null : element;
 75         }
 76 
 77         for (DocTree dt : input.utils.getThrowsTrees(input.element)) {
 78             Element texception = ch.getException(utils.configuration, dt);
 79             if (texception != null && (input.tagId.equals(utils.getSimpleName(texception)) ||
 80                  (input.tagId.equals(utils.getFullyQualifiedName(texception))))) {
 81                 output.holder = input.element;
 82                 output.holderTag = dt;
 83                 output.inlineTags = ch.getBody(input.utils.configuration, output.holderTag);
 84                 output.tagList.add(dt);
 85             } else if (exception != null && texception != null &&
 86                     utils.isTypeElement(texception) && utils.isTypeElement(exception) &&
 87                     utils.isSubclassOf((TypeElement)texception, (TypeElement)exception)) {
 88                 output.tagList.add(dt);
 89             }
 90         }
 91     }
 92 
 93     /**
 94      * Add links for exceptions that are declared but not documented.
 95      */
 96     private Content linkToUndocumentedDeclaredExceptions(List<? extends TypeMirror> declaredExceptionTypes,
 97             Set<String> alreadyDocumented, TagletWriter writer) {
 98         Utils utils = writer.configuration().utils;
 99         Content result = writer.getOutputInstance();
100         //Add links to the exceptions declared but not documented.
101         for (TypeMirror declaredExceptionType : declaredExceptionTypes) {
102             TypeElement klass = utils.asTypeElement(declaredExceptionType);
103             if (klass != null &&
104                 !alreadyDocumented.contains(utils.getSimpleName(klass)) &&
105                 !alreadyDocumented.contains(utils.getFullyQualifiedName(klass))) {
106                 if (alreadyDocumented.isEmpty()) {
107                     result.add(writer.getThrowsHeader());
108                 }
109                 result.add(writer.throwsTagOutput(declaredExceptionType));
110                 alreadyDocumented.add(utils.getSimpleName(klass));
111             }
112         }
113         return result;
114     }
115 
116     /**
117      * Inherit throws documentation for exceptions that were declared but not
118      * documented.
119      */
120     private Content inheritThrowsDocumentation(Element holder,
121             List<? extends TypeMirror> declaredExceptionTypes, Set<String> alreadyDocumented,
122             TagletWriter writer) {
123         Utils utils = writer.configuration().utils;
124         Content result = writer.getOutputInstance();
125         if (utils.isExecutableElement(holder)) {
126             Map<List<? extends DocTree>, ExecutableElement> declaredExceptionTags = new LinkedHashMap<>();
127             for (TypeMirror declaredExceptionType : declaredExceptionTypes) {
128                 Input input = new DocFinder.Input(utils, holder, this,
129                         utils.getTypeName(declaredExceptionType, false));
130                 DocFinder.Output inheritedDoc = DocFinder.search(writer.configuration(), input);
131                 if (inheritedDoc.tagList.isEmpty()) {
132                     String typeName = utils.getTypeName(declaredExceptionType, true);
133                     input = new DocFinder.Input(utils, holder, this, typeName);
134                     inheritedDoc = DocFinder.search(writer.configuration(), input);
135                 }
136                 if (!inheritedDoc.tagList.isEmpty()) {
137                     if (inheritedDoc.holder == null) {
138                         inheritedDoc.holder = holder;
139                     }
140                     declaredExceptionTags.put(inheritedDoc.tagList, (ExecutableElement)inheritedDoc.holder);
141                 }
142             }
143             result.add(throwsTagsOutput(declaredExceptionTags, writer, alreadyDocumented, false));
144         }
145         return result;
146     }
147 
148     /**
149      * {@inheritDoc}
150      */
151     public Content getTagletOutput(Element holder, TagletWriter writer) {
152         Utils utils = writer.configuration().utils;
153         ExecutableElement execHolder = (ExecutableElement) holder;
154         Map<List<? extends DocTree>, ExecutableElement> tagsMap = new LinkedHashMap<>();
155         tagsMap.put(utils.getThrowsTrees(execHolder), execHolder);
156         Content result = writer.getOutputInstance();
157         HashSet<String> alreadyDocumented = new HashSet<>();
158         if (!tagsMap.isEmpty()) {
159             result.add(throwsTagsOutput(tagsMap, writer, alreadyDocumented, true));
160         }
161         result.add(inheritThrowsDocumentation(holder,
162             execHolder.getThrownTypes(), alreadyDocumented, writer));
163         result.add(linkToUndocumentedDeclaredExceptions(
164             execHolder.getThrownTypes(), alreadyDocumented, writer));
165         return result;
166     }
167 
168     /**
169      * Given an array of <code>Tag</code>s representing this custom
170      * tag, return its string representation.
171      * @param throwTags the array of <code>ThrowsTag</code>s to convert.
172      * @param writer the TagletWriter that will write this tag.
173      * @param alreadyDocumented the set of exceptions that have already
174      *        been documented.
175      * @param allowDups True if we allow duplicate throws tags to be documented.
176      * @return the Content representation of this <code>Tag</code>.
177      */
178     protected Content throwsTagsOutput(Map<List<? extends DocTree>, ExecutableElement> throwTags,
179         TagletWriter writer, Set<String> alreadyDocumented, boolean allowDups) {
180         Utils utils = writer.configuration().utils;
181         Content result = writer.getOutputInstance();
182         if (!throwTags.isEmpty()) {
183             for (Entry<List<? extends DocTree>, ExecutableElement> entry : throwTags.entrySet()) {
184                 CommentHelper ch = utils.getCommentHelper(entry.getValue());
185                 Element e = entry.getValue();
186                 for (DocTree dt : entry.getKey()) {
187                     Element te = ch.getException(utils.configuration, dt);
188                     String excName = ch.getExceptionName(dt).toString();
189                     if ((!allowDups) &&
190                         (alreadyDocumented.contains(excName) ||
191                         (te != null && alreadyDocumented.contains(utils.getFullyQualifiedName(te))))) {
192                         continue;
193                     }
194                     if (alreadyDocumented.isEmpty()) {
195                         result.add(writer.getThrowsHeader());
196                     }
197                     result.add(writer.throwsTagOutput(e, dt));
198                     alreadyDocumented.add(te != null
199                             ? utils.getFullyQualifiedName(te)
200                             : excName);
201                 }
202             }
203         }
204         return result;
205     }
