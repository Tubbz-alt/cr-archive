42 /**
43  * This is a pseudo element wrapper for doc-files html contents, essentially to
44  * associate the doc-file's html documentation's DocCommentTree to an element.
45  *
46  *  <p><b>This is NOT part of any supported API.
47  *  If you write code that depends on this, you do so at your own risk.
48  *  This code and its internal interfaces are subject to change or
49  *  deletion without notice.</b>
50  */
51 public class DocFileElement implements DocletElement {
52 
53     private final Element element;
54     private final FileObject fo;
55 
56     public DocFileElement(Element element, FileObject fo) {
57         this.element = element;
58         this.fo = fo;
59     }
60 
61     @Override
62     public PackageElement getPackageElement() {
63         switch(element.getKind()) {
64             case MODULE:
65                 // uncomment to support doc-files in modules
66                 // return configuration.workArounds.getUnnamedPackage();
67                 throw new UnsupportedOperationException("not implemented");

68             case PACKAGE:
69                 return (PackageElement)element;


70             default:
71                 throw new AssertionError("unknown kind: " + element.getKind());
72         }
73     }
74 





75     @Override
76     public FileObject getFileObject() {
77         return fo;
78     }
79 
80     @Override
81     public Kind getSubKind() {
82         return Kind.DOCFILE;
83     }
84 }

28 import javax.lang.model.element.Element;
29 import javax.lang.model.element.ModuleElement;


30 import javax.lang.model.element.PackageElement;

31 import javax.tools.FileObject;
32 
33 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
34 
35 /**
36  * This is a pseudo element wrapper for doc-files html contents, essentially to
37  * associate the doc-file's html documentation's DocCommentTree to an element.
38  *
39  *  <p><b>This is NOT part of any supported API.
40  *  If you write code that depends on this, you do so at your own risk.
41  *  This code and its internal interfaces are subject to change or
42  *  deletion without notice.</b>
43  */
44 public class DocFileElement implements DocletElement {
45 
46     private final PackageElement packageElement;
47     private final FileObject fo;
48 
49     public DocFileElement(Utils utils, Element element, FileObject fo) {

50         this.fo = fo;

51 


52         switch(element.getKind()) {
53             case MODULE:
54                 ModuleElement moduleElement = (ModuleElement) element;
55                 packageElement = utils.elementUtils.getPackageElement(moduleElement, "");
56                 break;
57 
58             case PACKAGE:
59                 packageElement = (PackageElement) element;
60                 break;
61 
62             default:
63                 throw new AssertionError("unknown kind: " + element.getKind());
64         }
65     }
66 
67     @Override
68     public PackageElement getPackageElement() {
69         return packageElement;
70     }
71 
72     @Override
73     public FileObject getFileObject() {
74         return fo;
75     }
76 
77     @Override
78     public Kind getSubKind() {
79         return Kind.DOCFILE;
80     }
81 }
