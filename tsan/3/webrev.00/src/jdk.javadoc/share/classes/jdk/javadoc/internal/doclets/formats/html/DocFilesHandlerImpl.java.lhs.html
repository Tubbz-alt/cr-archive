 58 public class DocFilesHandlerImpl implements DocFilesHandler {
 59 
 60     public final Element element;
 61     public final Location location;
 62     public final DocPath  source;
 63     public final HtmlConfiguration configuration;

 64     private Navigation navBar;
 65 
 66     /**
 67      * Constructor to construct the DocFilesWriter object.
 68      *
 69      * @param configuration the configuration of this doclet.
 70      * @param element the containing element of the doc-files.
 71      *
 72      */
 73     public DocFilesHandlerImpl(HtmlConfiguration configuration, Element element) {
 74         this.configuration = configuration;

 75         this.element = element;
 76 
 77         switch (element.getKind()) {
 78             case MODULE:
 79                 ModuleElement mdle = (ModuleElement)element;
 80                 location = configuration.utils.getLocationForModule(mdle);
 81                 source = DocPaths.DOC_FILES;
 82                 break;
 83             case PACKAGE:
 84                 PackageElement pkg = (PackageElement)element;
 85                 location = configuration.utils.getLocationForPackage(pkg);
 86                 // Note, given that we have a module-specific location,
 87                 // we want a module-relative path for the source, and not the
 88                 // standard path that may include the module directory
 89                 source = DocPath.create(pkg.getQualifiedName().toString().replace('.', '/'))
 90                         .resolve(DocPaths.DOC_FILES);
 91                 break;
 92             default:
 93                 throw new AssertionError("unsupported element " + element);
 94         }
 95     }
 96 
 97     /**
 98      * Copy doc-files directory and its contents from the source
 99      * elements directory to the generated documentation directory.
100      *
101      * @throws DocFileIOException if there is a problem while copying
102      *         the documentation files
103      */
104 
105     public void copyDocFiles()  throws DocFileIOException {
106         boolean first = true;
107         for (DocFile srcdir : DocFile.list(configuration, location, source)) {
108             if (!srcdir.isDirectory()) {
109                 continue;
110             }
111             DocPath path = null;
112             switch (this.element.getKind()) {
113                 case MODULE:
114                     path = DocPaths.forModule((ModuleElement)this.element);
115                     break;
116                 case PACKAGE:
117                     path = configuration.docPaths.forPackage((PackageElement)this.element);
118                     break;
119                 default:
120                     throw new AssertionError("unknown kind:" + this.element.getKind());
121             }
122             copyDirectory(srcdir, path.resolve(DocPaths.DOC_FILES), first);
123             first = false;
124         }
125     }

127     public List<DocPath> getStylesheets() throws DocFileIOException {
128         List<DocPath> stylesheets = new ArrayList<DocPath>();
129         for (DocFile srcdir : DocFile.list(configuration, location, source)) {
130             for (DocFile srcFile : srcdir.list()) {
131                 if (srcFile.getName().endsWith(".css"))
132                     stylesheets.add(DocPaths.DOC_FILES.resolve(srcFile.getName()));
133             }
134         }
135         return stylesheets;
136     }
137 
138     private void copyDirectory(DocFile srcdir, final DocPath dstDocPath,
139                                boolean first) throws DocFileIOException {
140         DocFile dstdir = DocFile.createFileForOutput(configuration, dstDocPath);
141         if (srcdir.isSameFile(dstdir)) {
142             return;
143         }
144         for (DocFile srcfile: srcdir.list()) {
145             DocFile destfile = dstdir.resolve(srcfile.getName());
146             if (srcfile.isFile()) {
147                 if (destfile.exists() && !first) {
148                     configuration.messages.warning("doclet.Copy_Overwrite_warning",
149                             srcfile.getPath(), dstdir.getPath());
150                 } else {
151                     if (Utils.toLowerCase(srcfile.getPath()).endsWith(".html")) {
152                         handleHtmlFile(srcfile, dstDocPath);
153                     } else {
154                         configuration.messages.notice("doclet.Copying_File_0_To_Dir_1",
155                                 srcfile.getPath(), dstdir.getPath());
156                         destfile.copyFile(srcfile);
157                     }
158                 }
159             } else if (srcfile.isDirectory()) {
160                 if (configuration.copydocfilesubdirs
161                         && !configuration.shouldExcludeDocFileDir(srcfile.getName())) {
162                     DocPath dirDocPath = dstDocPath.resolve(srcfile.getName());
163                     copyDirectory(srcfile, dirDocPath, first);
164                 }
165             }
166         }
167     }
168 
169     private void handleHtmlFile(DocFile srcfile, DocPath dstPath) throws DocFileIOException {
170         Utils utils = configuration.utils;
171         FileObject fileObject = srcfile.getFileObject();
172         DocFileElement dfElement = new DocFileElement(element, fileObject);
173 
174         DocPath dfilePath = dstPath.resolve(srcfile.getName());
175         HtmlDocletWriter docletWriter = new DocFileWriter(configuration, dfilePath, element);
176         configuration.messages.notice("doclet.Generating_0", docletWriter.filename.getPath());
177 
178         List<? extends DocTree> localTags = getLocalHeaderTags(utils.getPreamble(dfElement));
179         Content localTagsContent = docletWriter.commentTagsToContent(null, dfElement, localTags, false);
180 
181         String title = getWindowTitle(docletWriter, dfElement).trim();
182         HtmlTree htmlContent = docletWriter.getBody(true, title);
183         docletWriter.addTop(htmlContent);
184         PackageElement pkg = (PackageElement) element;
185         this.navBar = new Navigation(pkg, configuration, docletWriter.fixedNavDiv,
186                 PageMode.DOCFILE, docletWriter.path);
187         Content mdleLinkContent = docletWriter.getModuleLink(utils.elementUtils.getModuleOf(pkg),
188                 docletWriter.contents.moduleLabel);
189         navBar.setNavLinkModule(mdleLinkContent);
190         Content pkgLinkContent = docletWriter.getPackageLink(pkg, docletWriter.contents.packageLabel);
191         navBar.setNavLinkPackage(pkgLinkContent);
192         navBar.setUserHeader(docletWriter.getUserHeaderFooter(true));
193         Content header = HtmlTree.HEADER();
194         header.add(navBar.getContent(true));
195         htmlContent.add(header);
196 
197         List<? extends DocTree> fullBody = utils.getFullBody(dfElement);
198         Content bodyContent = docletWriter.commentTagsToContent(null, dfElement, fullBody, false);
199         docletWriter.addTagsInfo(dfElement, bodyContent);
200         Content main = HtmlTree.MAIN();
201         main.add(bodyContent);
202         htmlContent.add(main);
203 
204         navBar.setUserFooter(docletWriter.getUserHeaderFooter(false));
205         Content footer = HtmlTree.FOOTER();
206         footer.add(navBar.getContent(false));
207         docletWriter.addBottom(footer);
208         htmlContent.add(footer);
209         docletWriter.printHtmlDocument(Collections.emptyList(), null, localTagsContent, htmlContent);




210     }

213     private List<? extends DocTree> getLocalHeaderTags(List<? extends DocTree> dtrees) {
214         List<DocTree> localTags = new ArrayList<>();
215         DocTreeFactory docTreeFactory = configuration.docEnv.getDocTrees().getDocTreeFactory();
216         boolean inHead = false;
217         boolean inTitle = false;
218         loop:
219         for (DocTree dt : dtrees) {
220             switch (dt.getKind()) {
221                 case START_ELEMENT:
222                     StartElementTree startElem = (StartElementTree)dt;
223                     switch (HtmlTag.get(startElem.getName())) {
224                         case HEAD:
225                             inHead = true;
226                             break;
227                         case META:
228                             break;
229                         case TITLE:
230                             inTitle = true;
231                             break;
232                         default:
233                             if (inHead) {
234                                 localTags.add(startElem);
235                                 localTags.add(docTreeFactory.newTextTree(DocletConstants.NL));
236                             }
237                     }
238                     break;
239                 case END_ELEMENT:
240                     EndElementTree endElem = (EndElementTree)dt;
241                     switch (HtmlTag.get(endElem.getName())) {
242                         case HEAD:
243                             inHead = false;
244                             break loop;
245                         case TITLE:
246                             inTitle = false;
247                             break;
248                         default:
249                             if (inHead) {
250                                 localTags.add(endElem);
251                                 localTags.add(docTreeFactory.newTextTree(DocletConstants.NL));
252                             }
253                     }
254                     break;
255                 case ENTITY:
256                 case TEXT:
257                     if (inHead && !inTitle) {
258                         localTags.add(dt);
259                     }
260                     break;
261             }
262         }
263         return localTags;
264     }

266     private String getWindowTitle(HtmlDocletWriter docletWriter, Element element) {
267         List<? extends DocTree> preamble = configuration.utils.getPreamble(element);
268         StringBuilder sb = new StringBuilder();
269         boolean titleFound = false;
270         loop:
271         for (DocTree dt : preamble) {
272             switch (dt.getKind()) {
273                 case START_ELEMENT:
274                     StartElementTree nodeStart = (StartElementTree)dt;
275                     if (Utils.toLowerCase(nodeStart.getName().toString()).equals("title")) {
276                         titleFound = true;
277                     }
278                     break;
279 
280                 case END_ELEMENT:
281                     EndElementTree nodeEnd = (EndElementTree)dt;
282                     if (Utils.toLowerCase(nodeEnd.getName().toString()).equals("title")) {
283                         break loop;
284                     }
285                     break;
286 
287                 case TEXT:
288                     TextTree nodeText = (TextTree)dt;
289                     if (titleFound)
290                         sb.append(nodeText.getBody());
291                     break;
292 
293                 default:
294                     // do nothing
295             }
296         }
297         return docletWriter.getWindowTitle(sb.toString().trim());
298     }

300     private static class DocFileWriter extends HtmlDocletWriter {
301 
302         final PackageElement pkg;
303 
304         /**
305          * Constructor to construct the HtmlDocletWriter object.
306          *
307          * @param configuration the configuruation of this doclet.
308          * @param path          the file to be generated.
309          * @param e             the anchoring element.
310          */
311         public DocFileWriter(HtmlConfiguration configuration, DocPath path, Element e) {
312             super(configuration, path);
313             switch (e.getKind()) {
314                 case PACKAGE:
315                     pkg = (PackageElement)e;
316                     break;
317                 default:
318                     throw new AssertionError("unsupported element: " + e.getKind());
319             }
320         }
321     }
