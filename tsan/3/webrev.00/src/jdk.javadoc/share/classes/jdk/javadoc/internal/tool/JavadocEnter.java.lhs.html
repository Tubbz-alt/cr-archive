 26 package jdk.javadoc.internal.tool;
 27 
 28 import javax.tools.JavaFileObject;
 29 
 30 import com.sun.source.util.TreePath;
 31 import com.sun.tools.javac.code.Symbol.*;
 32 import com.sun.tools.javac.comp.Enter;
 33 import com.sun.tools.javac.tree.JCTree.*;
 34 import com.sun.tools.javac.util.Context;
 35 import com.sun.tools.javac.util.JCDiagnostic.DiagnosticPosition;
 36 import com.sun.tools.javac.util.List;
 37 
 38 import static com.sun.tools.javac.code.Kinds.Kind.*;
 39 import com.sun.tools.javac.main.JavaCompiler;
 40 
 41 /**
 42  *  Javadoc's own enter phase does a few things above and beyond that
 43  *  done by javac.
 44  *
 45  *  <p><b>This is NOT part of any supported API.
 46  *  If you write code that depends on this, you do so at your own risk.
 47  *  This code and its internal interfaces are subject to change or
 48  *  deletion without notice.</b>
 49  *
 50  *  @author Neal Gafter
 51  */
 52 public class JavadocEnter extends Enter {
 53     public static JavadocEnter instance(Context context) {
 54         Enter instance = context.get(enterKey);
 55         if (instance == null)
 56             instance = new JavadocEnter(context);
 57         return (JavadocEnter)instance;
 58     }
 59 
 60     public static void preRegister(Context context) {
 61         context.put(enterKey, (Context.Factory<Enter>)JavadocEnter::new);
 62     }
 63 
 64     protected JavadocEnter(Context context) {
 65         super(context);
 66         messager = Messager.instance0(context);
 67         toolEnv = ToolEnvironment.instance(context);
 68         compiler = JavaCompiler.instance(context);
 69     }
 70 
 71     final Messager messager;
 72     final ToolEnvironment toolEnv;
 73     final JavaCompiler compiler;
 74 
 75     @Override
 76     public void main(List<JCCompilationUnit> trees) {
 77         // cache the error count if we need to convert Enter errors as warnings.
 78         int nerrors = messager.nerrors;
 79         super.main(trees);
 80         compiler.enterDone();
 81         if (toolEnv.ignoreSourceErrors) {
 82             messager.nwarnings += (messager.nerrors - nerrors);
 83             messager.nerrors = nerrors;
 84         }
 85     }
 86 
 87     @Override
 88     public void visitTopLevel(JCCompilationUnit tree) {
 89         super.visitTopLevel(tree);
 90         if (tree.sourcefile.isNameCompatible("package-info", JavaFileObject.Kind.SOURCE)) {
 91             JCPackageDecl pd = tree.getPackage();
 92             TreePath tp = pd == null ? toolEnv.getTreePath(tree) : toolEnv.getTreePath(tree, pd);
 93             toolEnv.setElementToTreePath(tree.packge, tp);
 94         }
 95     }
 96 
 97     @Override
 98     public void visitClassDef(JCClassDecl tree) {
 99         super.visitClassDef(tree);
100         if (tree.sym == null) return;
101         if (tree.sym.kind == TYP || tree.sym.kind == ERR) {
102             ClassSymbol c = tree.sym;
103             toolEnv.setElementToTreePath(c, toolEnv.getTreePath(env.toplevel, tree));
104             c.classfile = env.toplevel.sourcefile;
105         }
106     }
107 
108     /** Don't complain about a duplicate class. */
109     @Override
110     protected void duplicateClass(DiagnosticPosition pos, ClassSymbol c) {}
111 
112 }
