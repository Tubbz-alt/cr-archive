 48 /**
 49  * Generate the file with list of all the classes in this run.
 50  */
 51 public class AllClassesIndexWriter extends HtmlDocletWriter {
 52 
 53     /**
 54      * Index of all the classes.
 55      */
 56     protected IndexBuilder indexbuilder;
 57 
 58     /**
 59      * The HTML tree for main tag.
 60      */
 61     protected HtmlTree mainTree = HtmlTree.MAIN();
 62 
 63     private final Navigation navBar;
 64 
 65     /**
 66      * Construct AllClassesFrameWriter object. Also initializes the indexbuilder variable in this
 67      * class.
 68      *
 69      * @param configuration The current configuration
 70      * @param filename Path to the file which is getting generated.
 71      * @param indexbuilder Unicode based Index from {@link IndexBuilder}
 72      */
 73     public AllClassesIndexWriter(HtmlConfiguration configuration,
 74             DocPath filename, IndexBuilder indexbuilder) {
 75         super(configuration, filename);
 76         this.indexbuilder = indexbuilder;
 77         this.navBar = new Navigation(null, configuration, fixedNavDiv, PageMode.ALLCLASSES, path);
 78     }
 79 
 80     /**
 81      * Create AllClassesIndexWriter object.
 82      *
 83      * @param configuration The current configuration
 84      * @param indexBuilder IndexBuilder object for all classes index.
 85      * @throws DocFileIOException
 86      */
 87     public static void generate(HtmlConfiguration configuration,
 88             IndexBuilder indexBuilder) throws DocFileIOException {
 89         generate(configuration, indexBuilder, DocPaths.ALLCLASSES_INDEX);
 90     }
 91 
 92     private static void generate(HtmlConfiguration configuration, IndexBuilder indexBuilder,
 93             DocPath fileName) throws DocFileIOException {
 94         AllClassesIndexWriter allClassGen = new AllClassesIndexWriter(configuration,
 95                 fileName, indexBuilder);
 96         allClassGen.buildAllClassesFile();
 97     }
 98 
 99     /**
100      * Print all the classes in the file.
101      */
102     protected void buildAllClassesFile() throws DocFileIOException {
103         String label = resources.getText("doclet.All_Classes");
104         HtmlTree bodyTree = getBody(true, getWindowTitle(label));
105         HtmlTree header = HtmlTree.HEADER();
106         addTop(header);

107         navBar.setUserHeader(getUserHeaderFooter(true));
108         header.add(navBar.getContent(true));
109         bodyTree.add(header);
110         Content allClassesContent = new ContentBuilder();
111         addContents(allClassesContent);
112         mainTree.add(allClassesContent);
113         bodyTree.add(mainTree);
114         Content footer = HtmlTree.FOOTER();
115         navBar.setUserFooter(getUserHeaderFooter(false));
116         footer.add(navBar.getContent(false));
117         addBottom(footer);
118         bodyTree.add(footer);





119         printHtmlDocument(null, "class index", bodyTree);
120     }
121 
122     /**
123      * Add all types to the content tree.
124      *
125      * @param content HtmlTree content to which the links will be added
126      */
127     protected void addContents(Content content) {
128         Table table = new Table(HtmlStyle.typeSummary)
129                 .setHeader(new TableHeader(contents.classLabel, contents.descriptionLabel))
130                 .setRowScopeColumn(1)
131                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast)
132                 .setDefaultTab(resources.getText("doclet.All_Classes"))
133                 .addTab(resources.interfaceSummary, utils::isInterface)
134                 .addTab(resources.classSummary, e -> utils.isOrdinaryClass((TypeElement)e))
135                 .addTab(resources.enumSummary, utils::isEnum)
136                 .addTab(resources.exceptionSummary, e -> utils.isException((TypeElement)e))
137                 .addTab(resources.errorSummary, e -> utils.isError((TypeElement)e))
138                 .addTab(resources.annotationTypeSummary, utils::isAnnotationType)
139                 .setTabScript(i -> "show(" + i + ");");
140         for (Character unicode : indexbuilder.index()) {
141             for (Element element : indexbuilder.getMemberList(unicode)) {
142                 TypeElement typeElement = (TypeElement) element;
143                 if (!utils.isCoreClass(typeElement)) {
144                     continue;
145                 }
146                 addTableRow(table, typeElement);
147             }
148         }
149         Content titleContent = contents.allClassesLabel;
150         Content pHeading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
151                 HtmlStyle.title, titleContent);
152         Content headerDiv = HtmlTree.DIV(HtmlStyle.header, pHeading);
153         content.add(headerDiv);
154         if (!table.isEmpty()) {
155             HtmlTree li = HtmlTree.LI(HtmlStyle.blockList, table.toContent());
156             HtmlTree ul = HtmlTree.UL(HtmlStyle.blockList, li);
157             HtmlTree div = new HtmlTree(HtmlTag.DIV);
158             div.setStyle(HtmlStyle.allClassesContainer);
159             div.add(ul);
160             content.add(div);
161             if (table.needsScript()) {
162                 getMainBodyScript().append(table.getScript());
163             }
164         }
165     }
166 
167     /**
168      * Add table row.
169      *
170      * @param table the table to which the row will be added
171      * @param klass the type to be added to the table
172      */
173     protected void addTableRow(Table table, TypeElement klass) {
174         List<Content> rowContents = new ArrayList<>();
175         Content classLink = getLink(new LinkInfoImpl(
176                 configuration, LinkInfoImpl.Kind.INDEX, klass));
177         ContentBuilder description = new ContentBuilder();
178         if (utils.isDeprecated(klass)) {
179             description.add(getDeprecatedPhrase(klass));
180             List<? extends DocTree> tags = utils.getDeprecatedTrees(klass);
181             if (!tags.isEmpty()) {
182                 addSummaryDeprecatedComment(klass, tags.get(0), description);
183             }
184         } else {
185             addSummaryComment(klass, description);
186         }
187         rowContents.add(classLink);
188         rowContents.add(description);
189         table.addRow(klass, rowContents);
190     }
191 }
