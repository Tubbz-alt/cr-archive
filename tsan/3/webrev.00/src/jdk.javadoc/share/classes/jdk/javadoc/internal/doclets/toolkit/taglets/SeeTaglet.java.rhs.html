26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import java.util.List;
30 
31 import javax.lang.model.element.Element;
32 
33 import com.sun.source.doctree.DocTree;
34 import jdk.javadoc.doclet.Taglet.Location;
35 import jdk.javadoc.internal.doclets.toolkit.Content;
36 import jdk.javadoc.internal.doclets.toolkit.util.CommentHelper;
37 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder;
38 import jdk.javadoc.internal.doclets.toolkit.util.DocFinder.Input;
39 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
40 


41 /**
42  * A taglet that represents the @see tag.
43  *
44  *  <p><b>This is NOT part of any supported API.
45  *  If you write code that depends on this, you do so at your own risk.
46  *  This code and its internal interfaces are subject to change or
47  *  deletion without notice.</b>


48  */
49 public class SeeTaglet extends BaseTaglet implements InheritableTaglet {
50 
51     public SeeTaglet() {
52         super(DocTree.Kind.SEE, false, EnumSet.allOf(Location.class));
53     }
54 
55     @Override
56     public void inherit(DocFinder.Input input, DocFinder.Output output) {
57         List<? extends DocTree> tags = input.utils.getSeeTrees(input.element);
58         if (!tags.isEmpty()) {
59             CommentHelper ch =  input.utils.getCommentHelper(input.element);
60             output.holder = input.element;
61             output.holderTag = tags.get(0);
62             output.inlineTags = input.isFirstSentence
63                     ? ch.getFirstSentenceTrees(output.holderTag)
64                     : ch.getReference(output.holderTag);
65         }
66     }
67 
68     @Override
69     public Content getTagletOutput(Element holder, TagletWriter writer) {
70         Utils utils = writer.configuration().utils;
71         List<? extends DocTree> tags = utils.getSeeTrees(holder);
72         Element e = holder;
73         if (tags.isEmpty() && utils.isExecutableElement(holder)) {
74             Input input = new DocFinder.Input(utils, holder, this);
75             DocFinder.Output inheritedDoc = DocFinder.search(writer.configuration(), input);
76             if (inheritedDoc.holder != null) {
77                 tags = utils.getSeeTrees(inheritedDoc.holder);
78                 e = inheritedDoc.holder;
79             }
80         }
81         return writer.seeTagOutput(e, tags);
82     }
83 }
