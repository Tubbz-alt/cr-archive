 26 package jdk.javadoc.internal.doclets.formats.html.markup;
 27 
 28 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
 29 
 30 /**
 31  * Enum representing HTML tags.
 32  *
 33  *  <p><b>This is NOT part of any supported API.
 34  *  If you write code that depends on this, you do so at your own risk.
 35  *  This code and its internal interfaces are subject to change or
 36  *  deletion without notice.</b>
 37  *
 38  * @author Bhavesh Patel
 39  */
 40 public enum HtmlTag {
 41     A(BlockType.INLINE, EndTag.END),
 42     BUTTON(BlockType.INLINE, EndTag.END),
 43     BLOCKQUOTE,
 44     BODY(BlockType.OTHER, EndTag.END),
 45     BR(BlockType.INLINE, EndTag.NOEND),
 46     CAPTION,
 47     CODE(BlockType.INLINE, EndTag.END),
 48     DD,
 49     DIV,
 50     DL,
 51     DT,
 52     EM(BlockType.INLINE, EndTag.END),
 53     FOOTER,
 54     H1,
 55     H2,
 56     H3,
 57     H4,
 58     H5,
 59     H6,
 60     HEAD(BlockType.OTHER, EndTag.END),
 61     HEADER,
 62     HR(BlockType.BLOCK, EndTag.NOEND),
 63     HTML(BlockType.OTHER, EndTag.END),
 64     I(BlockType.INLINE, EndTag.END),
 65     IFRAME(BlockType.OTHER, EndTag.END),
 66     IMG(BlockType.INLINE, EndTag.NOEND),
 67     INPUT(BlockType.BLOCK, EndTag.NOEND),
 68     LABEL(BlockType.INLINE, EndTag.END),
 69     LI,
 70     LISTING,
 71     LINK(BlockType.OTHER, EndTag.NOEND),
 72     MAIN,
 73     MENU,
 74     META(BlockType.OTHER, EndTag.NOEND),
 75     NAV,
 76     NOSCRIPT(BlockType.OTHER, EndTag.END),
 77     OL,
 78     P,
 79     PRE,
 80     SCRIPT(BlockType.OTHER, EndTag.END),
 81     SECTION,
 82     SMALL(BlockType.INLINE, EndTag.END),
 83     SPAN(BlockType.INLINE, EndTag.END),
 84     STRONG(BlockType.INLINE, EndTag.END),
 85     SUB(BlockType.INLINE, EndTag.END),
 86     TABLE,
 87     TBODY,

 88     TD,
 89     TH,
 90     TITLE(BlockType.OTHER, EndTag.END),
 91     TR,
 92     UL;
 93 
 94     public final BlockType blockType;
 95     public final EndTag endTag;
 96     public final String value;
 97 
 98     /**
 99      * Enum representing the type of HTML element.
100      */
101     public static enum BlockType {
102         BLOCK,
103         INLINE,
104         OTHER
105     }
106 
107     /**
108      * Enum representing HTML end tag requirement.
109      */
110     public static enum EndTag {
111         END,
112         NOEND
113     }
114 
115     HtmlTag() {
116         this(BlockType.BLOCK, EndTag.END);
117     }
118 
119     HtmlTag(BlockType blockType, EndTag endTag) {
120         this.blockType = blockType;
121         this.endTag = endTag;
122         this.value = Utils.toLowerCase(name());
123     }
124 
125     /**
126      * Returns true if the end tag is required. This is specific to the standard
127      * doclet and does not exactly resemble the W3C specifications.
128      *
129      * @return true if end tag needs to be displayed else return false
130      */
131     public boolean endTagRequired() {
132         return (endTag == EndTag.END);
133     }
134 
135     public String toString() {
136         return value;
137     }
138 }
