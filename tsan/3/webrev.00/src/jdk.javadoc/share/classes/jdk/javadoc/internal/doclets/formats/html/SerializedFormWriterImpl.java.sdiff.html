 53 public class SerializedFormWriterImpl extends SubWriterHolderWriter
 54     implements SerializedFormWriter {
 55 
 56     Set<TypeElement> visibleClasses;
 57 
 58     /**
 59      * HTML tree for main tag.
 60      */
 61     private HtmlTree mainTree = HtmlTree.MAIN();
 62 
 63     private final Navigation navBar;
 64 
 65     /**
 66      * @param configuration the configuration data for the doclet
 67      */
 68     public SerializedFormWriterImpl(HtmlConfiguration configuration) {
 69         super(configuration, DocPaths.SERIALIZED_FORM);
 70         visibleClasses = configuration.getIncludedTypeElements();
 71         this.navBar = new Navigation(null, configuration, fixedNavDiv, PageMode.SERIALIZEDFORM, path);
 72     }
 73 
 74     /**
 75      * Get the given header.
 76      *
 77      * @param header the header to write
 78      * @return the body content tree
 79      */

 80     public Content getHeader(String header) {
 81         HtmlTree bodyTree = getBody(true, getWindowTitle(header));
 82         HtmlTree htmlTree = HtmlTree.HEADER();
 83         addTop(htmlTree);
 84         navBar.setUserHeader(getUserHeaderFooter(true));
 85         htmlTree.add(navBar.getContent(true));
 86         bodyTree.add(htmlTree);
 87         Content h1Content = new StringContent(header);
 88         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
 89                 HtmlStyle.title, h1Content);
 90         Content div = HtmlTree.DIV(HtmlStyle.header, heading);
 91         mainTree.add(div);

 92         return bodyTree;
 93     }
 94

 96      * Get the serialized form summaries header.
 97      *
 98      * @return the serialized form summary header tree
 99      */

100     public Content getSerializedSummariesHeader() {
101         HtmlTree ul = new HtmlTree(HtmlTag.UL);
102         ul.setStyle(HtmlStyle.blockList);
103         return ul;
104     }

107      * Get the package serialized form header.

111     public Content getPackageSerializedHeader() {

113     }

115     /**
116      * Get the given package header.
117      *
118      * @param packageName the package header to write

120      */

121     public Content getPackageHeader(String packageName) {
122         Content heading = HtmlTree.HEADING(Headings.SerializedForm.PACKAGE_HEADING, true,

124         heading.add(Contents.SPACE);

130      * Get the serialized class header.

140     /**
141      * Checks if a class is generated and is visible.
142      *
143      * @param typeElement the class being processed.
144      * @return true if the class, that is being processed, is generated and is visible.
145      */
146     public boolean isVisibleClass(TypeElement typeElement) {
147         return visibleClasses.contains(typeElement) && configuration.isGeneratedDoc(typeElement);
148     }
149 
150     /**
151      * Get the serializable class heading.
152      *
153      * @param typeElement the class being processed
154      * @return a content tree for the class header
155      */

156     public Content getClassHeader(TypeElement typeElement) {
157         Content classLink = (isVisibleClass(typeElement))
158                 ? getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.DEFAULT, typeElement)
159                         .label(configuration.getClassName(typeElement)))
160                 : new StringContent(utils.getFullyQualifiedName(typeElement));
161         Content li = HtmlTree.LI(HtmlStyle.blockList, links.createAnchor(
162                 utils.getFullyQualifiedName(typeElement)));
163         Content superClassLink = typeElement.getSuperclass() != null
164                 ? getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.SERIALIZED_FORM,
165                         typeElement.getSuperclass()))
166                 : null;
167 
168         //Print the heading.
169         Content className = superClassLink == null ?
170             contents.getContent(
171             "doclet.Class_0_implements_serializable", classLink) :
172             contents.getContent(
173             "doclet.Class_0_extends_implements_serializable", classLink,
174             superClassLink);
175         li.add(HtmlTree.HEADING(Headings.SerializedForm.CLASS_HEADING, className));
176         return li;
177     }

 70      * Get the given header.
 71      *
 72      * @param header the header to write
 73      * @return the body content tree
 74      */
 75     @Override
 76     public Content getHeader(String header) {
 77         HtmlTree bodyTree = getBody(getWindowTitle(header));
 78         Content headerContent = new ContentBuilder();
 79         addTop(headerContent);
 80         navBar.setUserHeader(getUserHeaderFooter(true));
 81         headerContent.add(navBar.getContent(true));

 82         Content h1Content = new StringContent(header);
 83         Content heading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
 84                 HtmlStyle.title, h1Content);
 85         Content div = HtmlTree.DIV(HtmlStyle.header, heading);
 86         bodyContents.setHeader(headerContent)
 87                 .addMainContent(div);
 88         return bodyTree;
 89     }
