 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.util.*;
 29 
 30 import javax.lang.model.element.Element;
 31 import javax.lang.model.element.TypeElement;
 32 
 33 import com.sun.source.doctree.DocTree;
 34 
 35 /**
 36  * The interface for writing member summary output.
 37  *
 38  *  <p><b>This is NOT part of any supported API.
 39  *  If you write code that depends on this, you do so at your own risk.
 40  *  This code and its internal interfaces are subject to change or
 41  *  deletion without notice.</b>
 42  *
 43  * @author Jamie Ho
 44  * @author Bhavesh Patel (Modified)
 45  */
 46 
 47 public interface MemberSummaryWriter {
 48 
 49     /**
 50      * Get the member summary header for the given class.
 51      *
 52      * @param typeElement the class the summary belongs to
 53      * @param memberSummaryTree the content tree to which the member summary will be added
 54      * @return a content tree for the member summary header
 55      */
 56     public Content getMemberSummaryHeader(TypeElement typeElement,
 57             Content memberSummaryTree);
 58 
 59     /**
 60      * Get the summary table for the given class.
 61      *
 62      * @param typeElement the class the summary table belongs to
 63      * @return a content tree for the member summary table
 64      */
 65     public Content getSummaryTableTree(TypeElement typeElement);
 66 
 67     /**
 68      * Add the member summary for the given class and member.
 69      *
 70      * @param typeElement the class the summary belongs to
 71      * @param member the member that is documented
 72      * @param firstSentenceTags the tags for the sentence being documented
 73      */
 74     public void addMemberSummary(TypeElement typeElement, Element member,
 75             List<? extends DocTree> firstSentenceTags);
 76 
 77     /**
 78      * Get the inherited member summary header for the given class.
 79      *
 80      * @param typeElement the class the summary belongs to
 81      * @return a content tree containing the inherited summary header
 82      */
 83     public Content getInheritedSummaryHeader(TypeElement typeElement);
 84 
 85     /**
 86      * Add the inherited member summary for the given class and member.
 87      *
 88      * @param typeElement the class the inherited member belongs to
 89      * @param member the inherited member that is being documented
 90      * @param isFirst true if this is the first member in the list
 91      * @param isLast true if this is the last member in the list
 92      * @param linksTree the content tree to which the links will be added
 93      */
 94     public void addInheritedMemberSummary(TypeElement typeElement,
 95         Element member, boolean isFirst, boolean isLast,
 96         Content linksTree);
 97 
 98     /**
 99      * Get inherited summary links.
100      *
101      * @return a content tree containing the inherited summary links
102      */
103     public Content getInheritedSummaryLinksTree();
104 
105     /**
106      * Add the member tree to the member summary tree.
107      *
108      * @param memberSummaryTree the content tree representing the member summary
109      * @param memberTree the content tree representing the member
110      */
111     public void addMemberTree(Content memberSummaryTree, Content memberTree);
112 
113     /**
114      * Get the member tree.
115      *
116      * @param memberTree the content tree representing the member
117      * @return a content tree for the member
118      */
119     public Content getMemberTree(Content memberTree);
120 }
