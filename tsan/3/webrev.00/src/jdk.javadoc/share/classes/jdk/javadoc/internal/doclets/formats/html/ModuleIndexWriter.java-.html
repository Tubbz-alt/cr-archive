 46 /**
 47  * Generate the module index page "overview-summary.html" for the right-hand
 48  * frame.
 49  *
 50  *  <p><b>This is NOT part of any supported API.
 51  *  If you write code that depends on this, you do so at your own risk.
 52  *  This code and its internal interfaces are subject to change or
 53  *  deletion without notice.</b>
 54  *
 55  * @author Bhavesh Patel
 56  */
 57 public class ModuleIndexWriter extends AbstractModuleIndexWriter {
 58 
 59     /**
 60      * Construct the ModuleIndexWriter.
 61      * @param configuration the configuration object
 62      * @param filename the name of the generated file
 63      */
 64     public ModuleIndexWriter(HtmlConfiguration configuration, DocPath filename) {
 65         super(configuration, filename);
 66     }
 67 
 68     /**
 69      * Generate the module index page for the right-hand frame.
 70      *
 71      * @param configuration the current configuration of the doclet.
 72      * @throws DocFileIOException if there is a problem generating the module index page
 73      */
 74     public static void generate(HtmlConfiguration configuration) throws DocFileIOException {
 75         DocPath filename = DocPaths.overviewSummary(configuration.frames);
 76         ModuleIndexWriter mdlgen = new ModuleIndexWriter(configuration, filename);
 77         mdlgen.buildModuleIndexFile("doclet.Window_Overview_Summary", "module index", true);
 78     }
 79 
 80     /**
 81      * Add the module index.
 82      *
 83      * @param header the documentation tree to which the navigational links will be added
 84      * @param main the documentation tree to which the modules list will be added
 85      */
 86     @Override
 87     protected void addIndex(Content header, Content main) {
 88         addIndexContents(header, main);
 89     }
 90 
 91     /**
 92      * Adds module index contents.
 93      *
 94      * @param header the document tree to which the navigational links will be added
 95      * @param main the document tree to which the modules list will be added
 96      */
 97     protected void addIndexContents(Content header, Content main) {
 98         HtmlTree htmltree = HtmlTree.NAV();
 99         htmltree.setStyle(HtmlStyle.indexNav);
100         HtmlTree ul = new HtmlTree(HtmlTag.UL);
101         addAllClassesLink(ul);
102         if (configuration.showModules) {
103             addAllModulesLink(ul);
104         }
105         htmltree.add(ul);
106         header.add(htmltree);
107         addModulesList(main);
108     }
109 
110     /**
111      * Add the list of modules.
112      *
113      * @param main the content tree to which the module list will be added
114      */
115     @Override
116     protected void addModulesList(Content main) {
117         Map<String, SortedSet<ModuleElement>> groupModuleMap
118                 = configuration.group.groupModules(configuration.modules);
119 
120         if (!groupModuleMap.keySet().isEmpty()) {
121             String tableSummary = resources.getText("doclet.Member_Table_Summary",
122                     resources.getText("doclet.Module_Summary"), resources.getText("doclet.modules"));
123             TableHeader header = new TableHeader(contents.moduleLabel, contents.descriptionLabel);
124             Table table =  new Table(HtmlStyle.overviewSummary)
125                     .setHeader(header)
126                     .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colLast)
127                     .setDefaultTab(resources.getText("doclet.All_Modules"))
128                     .setTabScript(i -> "show(" + i + ");")
129                     .setTabId(i -> (i == 0) ? "t0" : ("t" + (1 << (i - 1))));
130 
131             // add the tabs in command-line order
132             for (String groupName : configuration.group.getGroupList()) {
133                 Set<ModuleElement> groupModules = groupModuleMap.get(groupName);
134                 if (groupModules != null) {
135                     table.addTab(groupName, groupModules::contains);
136                 }
137             }
138 
139             for (ModuleElement mdle : configuration.modules) {
140                 if (!mdle.isUnnamed()) {
141                     if (!(configuration.nodeprecated && utils.isDeprecated(mdle))) {
142                         Content moduleLinkContent = getModuleLink(mdle, new StringContent(mdle.getQualifiedName().toString()));
143                         Content summaryContent = new ContentBuilder();
144                         addSummaryComment(mdle, summaryContent);
145                         table.addRow(mdle, moduleLinkContent, summaryContent);
146                     }
147                 }
148             }
149 
150             Content div = HtmlTree.DIV(HtmlStyle.contentContainer, table.toContent());
151             main.add(div);
152 
153             if (table.needsScript()) {
154                 mainBodyScript.append(table.getScript());
155             }
156         }
157     }
158 
159     /**
160      * Adds the overview summary comment for this documentation. Add one line
161      * summary at the top of the page and generate a link to the description,
162      * which is added at the end of this page.
163      *
164      * @param main the documentation tree to which the overview header will be added
165      */
166     @Override
167     protected void addOverviewHeader(Content main) {
168         addConfigurationTitle(main);
169         if (!utils.getFullBody(configuration.overviewElement).isEmpty()) {
170             HtmlTree div = new HtmlTree(HtmlTag.DIV);
171             div.setStyle(HtmlStyle.contentContainer);
172             addOverviewComment(div);
173             main.add(div);
174         }
175     }
176 
177     /**
178      * Adds the overview comment as provided in the file specified by the
179      * "-overview" option on the command line.
180      *
181      * @param htmltree the documentation tree to which the overview comment will
182      *                 be added
183      */
184     protected void addOverviewComment(Content htmltree) {
185         if (!utils.getFullBody(configuration.overviewElement).isEmpty()) {
186             addInlineComment(configuration.overviewElement, htmltree);
187         }
188     }
189 
190     /**
191      * Adds the top text (from the -top option), the upper
192      * navigation bar, and then the title (from the"-title"
193      * option), at the top of page.
194      *
195      * @param header the documentation tree to which the navigation bar header will be added
196      */
197     @Override
198     protected void addNavigationBarHeader(Content header) {
199         addTop(header);
200         navBar.setUserHeader(getUserHeaderFooter(true));
201         header.add(navBar.getContent(true));
202     }
203 
204     /**
205      * Adds the lower navigation bar and the bottom text
206      * (from the -bottom option) at the bottom of page.
207      *
208      * @param footer the documentation tree to which the navigation bar footer will be added
209      */
210     @Override
211     protected void addNavigationBarFooter(Content footer) {
212         navBar.setUserFooter(getUserHeaderFooter(false));
213         footer.add(navBar.getContent(false));
214         addBottom(footer);
215     }

218     protected void addModulePackagesList(Map<ModuleElement, Set<PackageElement>> modules, String text,
219             String tableSummary, Content main, ModuleElement mdle) {
220     }
