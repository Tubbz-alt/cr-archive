26 package jdk.javadoc.internal.doclets.toolkit.taglets;
27 
28 import java.util.EnumSet;
29 import javax.lang.model.element.Element;
30 
31 import com.sun.source.doctree.DocTree;
32 import jdk.javadoc.internal.doclets.toolkit.Content;
33 
34 import static com.sun.source.doctree.DocTree.Kind.INDEX;
35 
36 /**
37  * An inline Taglet used to index word or a phrase.
38  * The enclosed text is interpreted as not containing HTML markup or
39  * nested javadoc tags.
40  *
41  * @author Bhavesh Patel
42  */
43 
44 public class IndexTaglet extends BaseTaglet {
45 
46     IndexTaglet() {
47         super(INDEX.tagName, true, EnumSet.allOf(Site.class));
48     }
49 
50     @Override
51     public Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) {
52         return writer.indexTagOutput(element, tag);
53     }
54 }
