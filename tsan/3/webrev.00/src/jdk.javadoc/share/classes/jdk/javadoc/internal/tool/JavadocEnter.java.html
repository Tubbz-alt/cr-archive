 26 package jdk.javadoc.internal.tool;
 27 
 28 import javax.tools.JavaFileObject;
 29 
 30 import com.sun.source.util.TreePath;
 31 import com.sun.tools.javac.code.Symbol.*;
 32 import com.sun.tools.javac.comp.Enter;
 33 import com.sun.tools.javac.tree.JCTree.*;
 34 import com.sun.tools.javac.util.Context;
 35 import com.sun.tools.javac.util.JCDiagnostic.DiagnosticPosition;
 36 import com.sun.tools.javac.util.List;
 37 
 38 import static com.sun.tools.javac.code.Kinds.Kind.*;
 39 import com.sun.tools.javac.main.JavaCompiler;
 40 
 41 /**
 42  *  Javadoc's own enter phase does a few things above and beyond that
 43  *  done by javac.
 44  *
 45  *  <p><b>This is NOT part of any supported API.
 46  *  If you write code that depends on this, you do so at your own risk.
 47  *  This code and its internal interfaces are subject to change or
 48  *  deletion without notice.</b>
 49  */
 50 public class JavadocEnter extends Enter {
 51     public static JavadocEnter instance(Context context) {
 52         Enter instance = context.get(enterKey);
 53         if (instance == null)
 54             instance = new JavadocEnter(context);
 55         return (JavadocEnter)instance;
 56     }
 57 
 58     public static void preRegister(Context context) {
 59         context.put(enterKey, (Context.Factory<Enter>)JavadocEnter::new);
 60     }
 61 
 62     protected JavadocEnter(Context context) {
 63         super(context);
 64         messager = Messager.instance0(context);
 65         toolEnv = ToolEnvironment.instance(context);
 66         compiler = JavaCompiler.instance(context);
 67     }
 68 
 69     final Messager messager;
 70     final ToolEnvironment toolEnv;
 71     final JavaCompiler compiler;
 72 
 73     @Override
 74     public void main(List<JCCompilationUnit> trees) {
 75         // cache the error count if we need to convert Enter errors as warnings.
 76         int nerrors = messager.nerrors;
 77         super.main(trees);
 78         compiler.enterDone();
 79         if (toolEnv.ignoreSourceErrors) {
 80             messager.nwarnings += (messager.nerrors - nerrors);
 81             messager.nerrors = nerrors;
 82         }
 83     }
 84 
 85     @Override
 86     public void visitTopLevel(JCCompilationUnit tree) {
 87         super.visitTopLevel(tree);
 88         if (tree.sourcefile.isNameCompatible("package-info", JavaFileObject.Kind.SOURCE)) {
 89             JCPackageDecl pd = tree.getPackage();
 90             TreePath tp = pd == null ? toolEnv.getTreePath(tree) : toolEnv.getTreePath(tree, pd);
 91             toolEnv.setElementToTreePath(tree.packge, tp);
 92         }
 93     }
 94 
 95     @Override
 96     public void visitClassDef(JCClassDecl tree) {
 97         super.visitClassDef(tree);
 98         if (tree.sym == null) return;
 99         if (tree.sym.kind == TYP || tree.sym.kind == ERR) {
100             ClassSymbol c = tree.sym;
101             toolEnv.setElementToTreePath(c, toolEnv.getTreePath(env.toplevel, tree));
102             c.classfile = env.toplevel.sourcefile;
103         }
104     }
105 
106     /** Don't complain about a duplicate class. */
107     @Override
108     protected void duplicateClass(DiagnosticPosition pos, ClassSymbol c) {}
109 
110 }
