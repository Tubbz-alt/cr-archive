 26 package jdk.javadoc.internal.doclets.toolkit.taglets;
 27 

 28 import javax.lang.model.element.Element;
 29 
 30 import com.sun.source.doctree.DocTree;

 31 import jdk.javadoc.internal.doclets.toolkit.Content;
 32 
 33 /**
 34  * This is the Taglet interface used internally within the doclet.
 35  *
 36  * @author Jamie Ho
 37  */
 38 
 39 public interface Taglet {







 40 
 41     /**
 42      * Return true if this <code>Taglet</code>
 43      * is used in field documentation.
 44      * @return true if this <code>Taglet</code>
 45      * is used in field documentation and false
 46      * otherwise.
 47      */
 48     public abstract boolean inField();
 49 
 50     /**
 51      * Return true if this <code>Taglet</code>
 52      * is used in constructor documentation.
 53      * @return true if this <code>Taglet</code>
 54      * is used in constructor documentation and false
 55      * otherwise.
 56      */
 57     public abstract boolean inConstructor();
 58 
 59     /**
 60      * Return true if this <code>Taglet</code>
 61      * is used in method documentation.
 62      * @return true if this <code>Taglet</code>
 63      * is used in method documentation and false
 64      * otherwise.
 65      */
 66     public abstract boolean inMethod();
 67 
 68     /**
 69      * Return true if this <code>Taglet</code>
 70      * is used in overview documentation.
 71      * @return true if this <code>Taglet</code>
 72      * is used in method documentation and false
 73      * otherwise.
 74      */
 75     public abstract boolean inOverview();
 76 
 77     /**
 78      * Return true if this <code>Taglet</code>
 79      * is used in module documentation.
 80      * @return true if this <code>Taglet</code>
 81      * is used in module documentation and false
 82      * otherwise.
 83      */
 84     public abstract boolean inModule();
 85 
 86     /**
 87      * Return true if this <code>Taglet</code>
 88      * is used in package documentation.
 89      * @return true if this <code>Taglet</code>
 90      * is used in package documentation and false
 91      * otherwise.
 92      */
 93     public abstract boolean inPackage();
 94 
 95     /**
 96      * Return true if this <code>Taglet</code>
 97      * is used in type documentation (classes or
 98      * interfaces).
 99      * @return true if this <code>Taglet</code>
100      * is used in type documentation and false
101      * otherwise.
102      */
103     public abstract boolean inType();
104 
105     /**
106      * Return true if this <code>Taglet</code>
107      * is an inline tag. Return false otherwise.
108      * @return true if this <code>Taglet</code>
109      * is an inline tag and false otherwise.
110      */
111     public abstract boolean isInlineTag();











112 
113     /**
114      * Return the name of this custom tag.
115      * @return the name of this custom tag.
116      */
117     public abstract String getName();
118 
119     /**
120      * Given the <code>Tag</code> representation of this custom
121      * tag, return its Content representation, which is output
122      * to the generated page.
123      * @param holder the element holding the tag
124      * @param tag the <code>Tag</code> representation of this custom tag.
125      * @param writer a {@link TagletWriter} Taglet writer.
126      * @throws UnsupportedOperationException thrown when the method is not supported by the taglet.
127      * @return the Content representation of this <code>Tag</code>.
128      */
129     public abstract Content getTagletOutput(Element holder, DocTree tag, TagletWriter writer) throws
130             UnsupportedOperationException;
131 
132     /**
133      * Given a <code>Doc</code> object, check if it holds any tags of
134      * this type.  If it does, return the string representing the output.
135      * If it does not, return null.
136      * @param holder a {@link Doc} object holding the custom tag.
137      * @param writer a {@link TagletWriter} Taglet writer.
138      * @throws UnsupportedTagletOperationException thrown when the method is not
139      *         supported by the taglet.
140      * @return the TagletOutput representation of this <code>Tag</code>.
141      */
142     public abstract Content getTagletOutput(Element holder, TagletWriter writer) throws
143             UnsupportedTagletOperationException;
144 
145     @Override
146     public abstract String toString();
147 
148     static class UnsupportedTagletOperationException extends UnsupportedOperationException {
149         private static final long serialVersionUID = -3530273193380250271L;
150         public UnsupportedTagletOperationException(String message) {
151             super(message);
152         }
153     };
154 }

 26 package jdk.javadoc.internal.doclets.toolkit.taglets;
 27 
 28 import java.util.Set;
 29 import javax.lang.model.element.Element;
 30 
 31 import com.sun.source.doctree.DocTree;
 32 import jdk.javadoc.doclet.Taglet.Location;
 33 import jdk.javadoc.internal.doclets.toolkit.Content;
 34 
 35 /**
 36  * This is the Taglet interface used internally within the doclet.


 37  */
 38 
 39 public interface Taglet {
 40     /**
 41      * Returns the set of allowed locations for block tags
 42      * handled by this taglet.
 43      *
 44      * @return the set of allowable locations
 45      */
 46     Set<Location> getAllowedLocations();
 47 
 48     /**
 49      * Return true if this <code>Taglet</code>
 50      * is used in field documentation.
 51      * @return true if this <code>Taglet</code>
 52      * is used in field documentation and false
 53      * otherwise.
 54      */
 55     boolean inField();
 56 
 57     /**
 58      * Return true if this <code>Taglet</code>
 59      * is used in constructor documentation.
 60      * @return true if this <code>Taglet</code>
 61      * is used in constructor documentation and false
 62      * otherwise.
 63      */
 64     boolean inConstructor();
 65 
 66     /**
 67      * Return true if this <code>Taglet</code>
 68      * is used in method documentation.
 69      * @return true if this <code>Taglet</code>
 70      * is used in method documentation and false
 71      * otherwise.
 72      */
 73     boolean inMethod();
 74 
 75     /**
 76      * Return true if this <code>Taglet</code>
 77      * is used in overview documentation.
 78      * @return true if this <code>Taglet</code>
 79      * is used in method documentation and false
 80      * otherwise.
 81      */
 82     boolean inOverview();
 83 
 84     /**
 85      * Return true if this <code>Taglet</code>
 86      * is used in module documentation.
 87      * @return true if this <code>Taglet</code>
 88      * is used in module documentation and false
 89      * otherwise.
 90      */
 91     boolean inModule();
 92 
 93     /**
 94      * Return true if this <code>Taglet</code>
 95      * is used in package documentation.
 96      * @return true if this <code>Taglet</code>
 97      * is used in package documentation and false
 98      * otherwise.
 99      */
100     boolean inPackage();
101 
102     /**
103      * Return true if this <code>Taglet</code>
104      * is used in type documentation (classes or
105      * interfaces).
106      * @return true if this <code>Taglet</code>
107      * is used in type documentation and false
108      * otherwise.
109      */
110     boolean inType();
111 
112     /**
113      * Return true if this <code>Taglet</code>
114      * is an inline tag. Return false otherwise.
115      * @return true if this <code>Taglet</code>
116      * is an inline tag and false otherwise.
117      */
118     boolean isInlineTag();
119 
120     /**
121      * Indicates whether this taglet supports block tags.
122      *
123      * @return true if this taglet supports block tags
124      * @implSpec This implementation returns the inverse
125      * result to {@code isInlineTag}.
126      */
127     default boolean isBlockTag() {
128         return !isInlineTag();
129     }
130 
131     /**
132      * Return the name of this custom tag.
133      * @return the name of this custom tag.
134      */
135     String getName();
136 
137     /**
138      * Returns the content to be included in the generated output for an
139      * instance of a tag handled by this taglet.
140      *
141      * @param element the element for the enclosing doc comment
142      * @param tag     the tag
143      * @param writer  the taglet-writer used in this doclet
144      * @return the output for this tag
145      * @throws UnsupportedTagletOperationException thrown when the method is not supported by the taglet
146      */
147     Content getTagletOutput(Element element, DocTree tag, TagletWriter writer) throws
148             UnsupportedTagletOperationException;
149 
150     /**
151      * Returns the content to be included in the generated output for all
152      * instances of tags handled by this taglet.
153      *
154      * @param element the element for the enclosing doc comment
155      * @param writer  the taglet-writer used in this doclet
156      * @return the output for this tag
157      * @throws UnsupportedTagletOperationException thrown when the method is not supported by the taglet

158      */
159     Content getTagletOutput(Element element, TagletWriter writer) throws
160             UnsupportedTagletOperationException;
161 
162     class UnsupportedTagletOperationException extends UnsupportedOperationException {



163         private static final long serialVersionUID = -3530273193380250271L;
164         public UnsupportedTagletOperationException(String message) {
165             super(message);
166         }
167     }
168 }
