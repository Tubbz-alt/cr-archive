 30 /**
 31  * Enum representing HTML tag attributes.

 33  *  <p><b>This is NOT part of any supported API.
 34  *  If you write code that depends on this, you do so at your own risk.
 35  *  This code and its internal interfaces are subject to change or
 36  *  deletion without notice.</b>
 37  *
 38  * @author Bhavesh Patel

 40 public enum HtmlAttr {
 41     ALT,
 42     ARIA_CONTROLS("aria-controls"),
 43     ARIA_LABELLEDBY("aria-labelledby"),
 44     ARIA_ORIENTATION("aria-orientation"),
 45     ARIA_SELECTED("aria-selected"),
 46     CLASS,
 47     CLEAR,
 48     COLS,
 49     CONTENT,
 50     DISABLED,
 51     FOR,
 52     HREF,
 53     HTTP_EQUIV("http-equiv"),
 54     ID,
 55     LANG,
 56     NAME,
 57     ONCLICK,
 58     ONKEYDOWN,
 59     ONLOAD,
 60     REL,
 61     ROLE,
 62     ROWS,
 63     SCOPE,
 64     SCROLLING,
 65     SRC,
 66     SUMMARY,
 67     TABINDEX,
 68     TARGET,
 69     TITLE,
 70     TYPE,
 71     VALUE,
 72     WIDTH;
 73 
 74     private final String value;
 75 
 76     public enum Role {
 77 
 78         BANNER,
 79         CONTENTINFO,
 80         MAIN,
 81         NAVIGATION,
 82         REGION;
 83 
 84         private final String role;
 85 
 86         Role() {
 87             role = Utils.toLowerCase(name());
 88         }
 89 
 90         public String toString() {
 91             return role;
 92         }
 93     }
 94 
 95     HtmlAttr() {
 96         this.value = Utils.toLowerCase(name());
 97     }
 98 
 99     HtmlAttr(String name) {
100         this.value = name;
101     }
102 
103     public String toString() {
104         return value;
105     }
106 }
