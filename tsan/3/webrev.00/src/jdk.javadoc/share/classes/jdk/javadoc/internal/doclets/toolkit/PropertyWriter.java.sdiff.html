 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.io.*;
 29 
 30 import javax.lang.model.element.ExecutableElement;
 31 import javax.lang.model.element.TypeElement;
 32 
 33 /**
 34  * The interface for writing property output.
 35  *
 36  *  <p><b>This is NOT part of any supported API.
 37  *  If you write code that depends on this, you do so at your own risk.
 38  *  This code and its internal interfaces are subject to change or
 39  *  deletion without notice.</b>
 40  *
 41  * @author Jamie Ho
 42  * @author Bhavesh Patel (Modified)
 43  */
 44 
 45 public interface PropertyWriter {
 46 
 47     /**
 48      * Get the property details tree header.
 49      *
 50      * @param typeElement the class being documented
 51      * @param memberDetailsTree the content tree representing member details
 52      * @return content tree for the property details header
 53      */
 54     public Content getPropertyDetailsTreeHeader(TypeElement typeElement,
 55             Content memberDetailsTree);
 56 
 57     /**
 58      * Get the property documentation tree header.
 59      *
 60      * @param property the property being documented
 61      * @param propertyDetailsTree the content tree representing property details
 62      * @return content tree for the property documentation header
 63      */
 64     public Content getPropertyDocTreeHeader(ExecutableElement property,
 65             Content propertyDetailsTree);
 66 
 67     /**
 68      * Get the signature for the given property.
 69      *
 70      * @param property the property being documented
 71      * @return content tree for the property signature
 72      */
 73     public Content getSignature(ExecutableElement property);
 74 
 75     /**
 76      * Add the deprecated output for the given property.
 77      *
 78      * @param property the property being documented
 79      * @param propertyDocTree content tree to which the deprecated information will be added
 80      */
 81     public void addDeprecated(ExecutableElement property, Content propertyDocTree);
 82 
 83     /**
 84      * Add the comments for the given property.
 85      *
 86      * @param property the property being documented
 87      * @param propertyDocTree the content tree to which the comments will be added
 88      */
 89     public void addComments(ExecutableElement property, Content propertyDocTree);
 90 
 91     /**
 92      * Add the tags for the given property.
 93      *
 94      * @param property the property being documented
 95      * @param propertyDocTree the content tree to which the tags will be added
 96      */
 97     public void addTags(ExecutableElement property, Content propertyDocTree);
 98 
 99     /**
100      * Get the property details tree.
101      *

102      * @param memberDetailsTree the content tree representing member details
103      * @return content tree for the property details
104      */
105     public Content getPropertyDetails(Content memberDetailsTree);
106 
107     /**
108      * Get the property documentation.
109      *
110      * @param propertyDocTree the content tree representing property documentation
111      * @param isLastContent true if the content to be added is the last content
112      * @return content tree for the property documentation
113      */
114     public Content getPropertyDoc(Content propertyDocTree, boolean isLastContent);







115 }
