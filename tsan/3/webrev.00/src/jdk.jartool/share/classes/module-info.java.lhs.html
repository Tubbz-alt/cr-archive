26 /**
27  * Defines tools for manipulating Java Archive (JAR) files,
28  * including the <em>{@index jar jar tool}</em> and
29  * <em>{@index jarsigner jarsigner tool}</em> tools.
30  * This module also defines APIs for signing JAR files.
31  *
32  * <p> This module provides the equivalent of command-line access to
33  * <em>jar</em> via the {@link java.util.spi.ToolProvider ToolProvider} SPI.
34  * Instances of the tool can be obtained by calling
35  * {@link java.util.spi.ToolProvider#findFirst ToolProvider.findFirst}
36  * or the {@linkplain java.util.ServiceLoader service loader} with the name
37  * {@code "jar"}.
38  *
39  * <dl style="font-family:'DejaVu Sans', Arial, Helvetica, sans serif">
40  * <dt class="simpleTagLabel">Tool Guides:
41  * <dd>{@extLink jar_tool_reference jar},
42  *     {@extLink jarsigner_tool_reference jarsigner}
43  * </dl>
44  *
45  * @moduleGraph
46  * @since 9
47  */
48 module jdk.jartool {
49     exports com.sun.jarsigner;
50     exports jdk.security.jarsigner;
51 
52     provides java.util.spi.ToolProvider with
53         sun.tools.jar.JarToolProvider;
54 }
