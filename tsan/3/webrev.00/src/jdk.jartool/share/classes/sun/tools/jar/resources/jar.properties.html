163 usage.compat=\
164 \Compatibility Interface:\
165 \n\
166 Usage: jar {ctxui}[vfmn0PMe] [jar-file] [manifest-file] [entry-point] [-C dir] files] ...\n\
167 Options:\n\
168 \ \   -c  create new archive\n\
169 \ \   -t  list table of contents for archive\n\
170 \ \   -x  extract named (or all) files from archive\n\
171 \ \   -u  update existing archive\n\
172 \ \   -v  generate verbose output on standard output\n\
173 \ \   -f  specify archive file name\n\
174 \ \   -m  include manifest information from specified manifest file\n\
175 \ \   -e  specify application entry point for stand-alone application \n\
176 \ \       bundled into an executable jar file\n\
177 \ \   -0  store only; use no ZIP compression\n\
178 \ \   -P  preserve leading '/' (absolute path) and ".." (parent directory) components from file names\n\
179 \ \   -M  do not create a manifest file for the entries\n\
180 \ \   -i  generate index information for the specified jar files\n\
181 \ \   -C  change to the specified directory and include the following file\n\
182 If any file is a directory then it is processed recursively.\n\
183 The manifest file name, the archive file name and the entry point name are\n\
184 specified in the same order as the 'm', 'f' and 'e' flags.\n\n\
185 Example 1: to archive two class files into an archive called classes.jar: \n\
186 \ \      jar cvf classes.jar Foo.class Bar.class \n\
187 Example 2: use an existing manifest file 'mymanifest' and archive all the\n\
188 \ \          files in the foo/ directory into 'classes.jar': \n\
189 \ \      jar cvfm classes.jar mymanifest -C foo/ .\n

196 main.help.preopt=\
197 Usage: jar [OPTION...] [ [--release VERSION] [-C dir] files] ...\n\
198 jar creates an archive for classes and resources, and can manipulate or\n\
199 restore individual classes or resources from an archive.\n\
200 \n\
201 \ Examples:\n\
202 \ # Create an archive called classes.jar with two class files:\n\
203 \ jar --create --file classes.jar Foo.class Bar.class\n\
204 \ # Create an archive using an existing manifest, with all the files in foo/:\n\
205 \ jar --create --file classes.jar --manifest mymanifest -C foo/ .\n\
206 \ # Create a modular jar archive, where the module descriptor is located in\n\
207 \ # classes/module-info.class:\n\
208 \ jar --create --file foo.jar --main-class com.foo.Main --module-version 1.0\n\
209 \     -C foo/ classes resources\n\
210 \ # Update an existing non-modular jar to a modular jar:\n\
211 \ jar --update --file foo.jar --main-class com.foo.Main --module-version 1.0\n\
212 \     -C foo/ module-info.class\n\
213 \ # Create a multi-release jar, placing some files in the META-INF/versions/9 directory:\n\
214 \ jar --create --file mr.jar -C foo classes --release 9 -C foo9 classes\n\
215 \n\
216 To shorten or simplify the jar command, you can specify arguments in a separate\n\
217 text file and pass it to the jar command with the at sign (@) as a prefix.\n\
218 \n\
219 \ Examples:\n\
220 \ # Read additional options and list of class files from the file classes.list\n\
221 \ jar --create --file my.jar @classes.list\
222 \n
