26 package sun.management.counter.perf;
27 
28 import sun.management.counter.*;
29 import java.nio.*;
30 
31 public class PerfByteArrayCounter extends AbstractCounter
32        implements ByteArrayCounter {
33 

34     ByteBuffer bb;
35 
36     PerfByteArrayCounter(String name, Units u, Variability v,
37                          int flags, int vectorLength,
38                          ByteBuffer bb) {
39 
40         super(name, u, v, flags, vectorLength);
41         this.bb = bb;
42     }
43 
44     public Object getValue() {
45         return byteArrayValue();
46     }
47 
48     /**
49      * Get a copy of the elements of the ByteArrayCounter.
50      */
51     public byte[] byteArrayValue() {
52 
53         bb.position(0);
54         byte[] b = new byte[bb.limit()];
55 
56         // copy the bytes
57         bb.get(b);
58 
59         return b;
60     }
61 
62     /**
63      * Get the value of an element of the ByteArrayCounter object.
64      */
65     public byte byteAt(int index) {
66         bb.position(index);
67         return bb.get();
68     }
69 
70     public String toString() {
71         String result = getName() + ": " + new String(byteArrayValue()) +
72                         " " + getUnits();
73         if (isInternal()) {
74             return result + " [INTERNAL]";
75         } else {
76             return result;
77         }
78     }
79 
80     /**
81      * Serialize as a snapshot object.
82      */
83     protected Object writeReplace() throws java.io.ObjectStreamException {
84        return new ByteArrayCounterSnapshot(getName(),
85                                            getUnits(),
86                                            getVariability(),
87                                            getFlags(),
88                                            getVectorLength(),
89                                            byteArrayValue());
90     }
91 
92     private static final long serialVersionUID = 2545474036937279921L;
93 }
