26 package sun.management.counter.perf;
27 
28 import sun.management.counter.*;
29 
30 import java.nio.LongBuffer;
31 import java.nio.ReadOnlyBufferException;
32 
33 public class PerfLongArrayCounter extends AbstractCounter
34        implements LongArrayCounter {
35 
36 
37     @SuppressWarnings("serial") // Value indirectly copied as a long[] in writeReplace
38     LongBuffer lb;
39 
40     PerfLongArrayCounter(String name, Units u, Variability v,
41                          int flags, int vectorLength,
42                          LongBuffer lb) {
43 
44         super(name, u, v, flags, vectorLength);
45         this.lb = lb;
46     }
47 
48     public Object getValue() {
49         return longArrayValue();
50     }
51 
52     /**
53      * Get a copy of the elements of the LongArrayCounter.
54      */
55     public long[] longArrayValue() {
56 
57         lb.position(0);
58         long[] l = new long[lb.limit()];
59 
60         // copy the bytes
61         lb.get(l);
62 
63         return l;
64     }
65 
66     /**
67      * Get the value of an element of the LongArrayCounter object.
68      */
69     public long longAt(int index) {
70         lb.position(index);
71         return lb.get();
72     }
73 
74     /**
75      * Serialize as a snapshot object.
76      */
77     protected Object writeReplace() throws java.io.ObjectStreamException {
78         return new LongArrayCounterSnapshot(getName(),
79                                             getUnits(),
80                                             getVariability(),
81                                             getFlags(),
82                                             getVectorLength(),
83                                             longArrayValue());
84     }
85 
86     private static final long serialVersionUID = -2733617913045487126L;
87 }
