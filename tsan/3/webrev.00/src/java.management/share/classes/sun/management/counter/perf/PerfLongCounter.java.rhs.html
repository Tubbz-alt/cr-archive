26 package sun.management.counter.perf;
27 
28 import sun.management.counter.*;
29 import java.nio.LongBuffer;
30 import java.nio.ReadOnlyBufferException;
31 
32 public class PerfLongCounter extends AbstractCounter
33        implements LongCounter {
34 
35     @SuppressWarnings("serial") // Value indirectly copied as a long[] in writeReplace
36     LongBuffer lb;
37 
38     // package private
39     PerfLongCounter(String name, Units u, Variability v, int flags,
40                     LongBuffer lb) {
41         super(name, u, v, flags);
42         this.lb = lb;
43     }
44 
45     public Object getValue() {
46         return Long.valueOf(lb.get(0));
47     }
48 
49     /**
50      * Get the value of this Long performance counter
51      */
52     public long longValue() {
53         return lb.get(0);
54     }
55 
56     /**
57      * Serialize as a snapshot object.
58      */
59     protected Object writeReplace() throws java.io.ObjectStreamException {
60         return new LongCounterSnapshot(getName(),
61                                        getUnits(),
62                                        getVariability(),
63                                        getFlags(),
64                                        longValue());
65     }
66 
67     private static final long serialVersionUID = 857711729279242948L;
68 }
