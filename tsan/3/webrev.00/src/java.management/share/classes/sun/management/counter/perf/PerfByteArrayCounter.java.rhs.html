26 package sun.management.counter.perf;
27 
28 import sun.management.counter.*;
29 import java.nio.*;
30 
31 public class PerfByteArrayCounter extends AbstractCounter
32        implements ByteArrayCounter {
33 
34     @SuppressWarnings("serial") // Value indirectly copied as a byte[] in writeReplace
35     ByteBuffer bb;
36 
37     PerfByteArrayCounter(String name, Units u, Variability v,
38                          int flags, int vectorLength,
39                          ByteBuffer bb) {
40 
41         super(name, u, v, flags, vectorLength);
42         this.bb = bb;
43     }
44 
45     public Object getValue() {
46         return byteArrayValue();
47     }
48 
49     /**
50      * Get a copy of the elements of the ByteArrayCounter.
51      */
52     public byte[] byteArrayValue() {
53 
54         bb.position(0);
55         byte[] b = new byte[bb.limit()];
56 
57         // copy the bytes
58         bb.get(b);
59 
60         return b;
61     }
62 
63     /**
64      * Get the value of an element of the ByteArrayCounter object.
65      */
66     public byte byteAt(int index) {
67         bb.position(index);
68         return bb.get();
69     }
70 
71     public String toString() {
72         String result = getName() + ": " + new String(byteArrayValue()) +
73                         " " + getUnits();
74         if (isInternal()) {
75             return result + " [INTERNAL]";
76         } else {
77             return result;
78         }
79     }
80 
81     /**
82      * Serialize as a snapshot object.
83      */
84     protected Object writeReplace() throws java.io.ObjectStreamException {
85        return new ByteArrayCounterSnapshot(getName(),
86                                            getUnits(),
87                                            getVariability(),
88                                            getFlags(),
89                                            getVectorLength(),
90                                            byteArrayValue());
91     }
92 
93     private static final long serialVersionUID = 2545474036937279921L;
94 }
