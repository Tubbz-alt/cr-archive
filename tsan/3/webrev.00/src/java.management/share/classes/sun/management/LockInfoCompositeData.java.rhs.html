 26 package sun.management;
 27 
 28 import java.lang.management.LockInfo;
 29 import java.util.Map;
 30 import javax.management.openmbean.CompositeType;
 31 import javax.management.openmbean.CompositeData;
 32 import javax.management.openmbean.CompositeDataSupport;
 33 import javax.management.openmbean.OpenDataException;
 34 
 35 /**
 36  * A CompositeData for LockInfo for the local management support.
 37  * This class avoids the performance penalty paid to the
 38  * construction of a CompositeData use in the local case.
 39  */
 40 public class LockInfoCompositeData extends LazyCompositeData {
 41     @SuppressWarnings("serial") // Not statically typed as Serializable
 42     private final LockInfo lock;
 43 
 44     private LockInfoCompositeData(LockInfo li) {
 45         this.lock = li;
 46     }
 47 
 48     public LockInfo getLockInfo() {
 49         return lock;
 50     }
 51 
 52     public static CompositeData toCompositeData(LockInfo li) {
 53         if (li == null) {
 54             return null;
 55         }
 56 
 57         LockInfoCompositeData licd = new LockInfoCompositeData(li);
 58         return licd.getCompositeData();
 59     }
 60 
 61     protected CompositeData getCompositeData() {
 62         Map<String,Object> items = Map.of(
 63             CLASS_NAME,         lock.getClassName(),
 64             IDENTITY_HASH_CODE, lock.getIdentityHashCode()
 65         );
 66 
 67         try {
 68             return new CompositeDataSupport(LOCK_INFO_COMPOSITE_TYPE, items);
 69         } catch (OpenDataException e) {
 70             // Should never reach here
 71             throw Util.newException(e);
 72         }
 73     }
 74 
 75     private static final CompositeType LOCK_INFO_COMPOSITE_TYPE;
 76     static {
 77         try {
 78             LOCK_INFO_COMPOSITE_TYPE = (CompositeType)
 79                 MappedMXBeanType.toOpenType(LockInfo.class);
 80         } catch (OpenDataException e) {
 81             // Should never reach here
 82             throw Util.newException(e);
 83         }
 84     }
 85 
 86     static CompositeType compositeType() {
 87         return LOCK_INFO_COMPOSITE_TYPE;
 88     }
 89 
 90     private static final String CLASS_NAME         = "className";
 91     private static final String IDENTITY_HASH_CODE = "identityHashCode";
 92 
 93     /*
 94      * Returns a LockInfo object mapped from the given CompositeData.
 95      */
 96     public static LockInfo toLockInfo(CompositeData cd) {
 97         if (cd == null) {
 98             throw new NullPointerException("Null CompositeData");
 99         }
100 
101         if (!isTypeMatched(LOCK_INFO_COMPOSITE_TYPE, cd.getCompositeType())) {
102             throw new IllegalArgumentException(
103                 "Unexpected composite type for LockInfo");
104         }
105 
106         String className = getString(cd, CLASS_NAME);
107         int identityHashCode = getInt(cd, IDENTITY_HASH_CODE);
108         return new LockInfo(className, identityHashCode);
109     }
110 
111     private static final long serialVersionUID = -6374759159749014052L;
112 }
