26 #include "util.h"
27 #include "ClassLoaderReferenceImpl.h"
28 #include "inStream.h"
29 #include "outStream.h"
30 
31 static jboolean
32 visibleClasses(PacketInputStream *in, PacketOutputStream *out)
33 {
34     JNIEnv *env = getEnv();
35     jobject loader;
36 
37     loader = inStream_readClassLoaderRef(env, in);
38     if (inStream_error(in)) {
39         return JNI_TRUE;
40     }
41 
42     WITH_LOCAL_REFS(env, 1) {
43 
44         jvmtiError error;
45         jint count;
46         jclass *classes;
47         int i;
48 
49         error = allClassLoaderClasses(loader, &classes, &count);
50         if (error != JVMTI_ERROR_NONE) {
51             outStream_setError(out, map2jdwpError(error));
52         } else {
53             (void)outStream_writeInt(out, count);
54             for (i = 0; i < count; i++) {
55                 jbyte tag;
56                 jclass clazz;
57 
58                 clazz = classes[i];
59                 tag = referenceTypeTag(clazz);
60 
61                 (void)outStream_writeByte(out, tag);
62                 (void)outStream_writeObjectRef(env, out, clazz);
63             }
64         }
65 
66         if ( classes != NULL )
67             jvmtiDeallocate(classes);
68 
69      } END_WITH_LOCAL_REFS(env);
70 
71     return JNI_TRUE;
72 }
73 
74 Command ClassLoaderReference_Commands[] = {
75     {visibleClasses, "VisibleClasses"}
76 };
