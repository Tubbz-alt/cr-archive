26 #include "util.h"
27 #include "StringReferenceImpl.h"
28 #include "inStream.h"
29 #include "outStream.h"
30 
31 static jboolean
32 value(PacketInputStream *in, PacketOutputStream *out)
33 {
34     JNIEnv *env;
35     jstring string;
36 
37     env = getEnv();
38 
39     string = inStream_readStringRef(env, in);
40     if (inStream_error(in)) {
41         return JNI_TRUE;
42     }
43 
44     WITH_LOCAL_REFS(env, 1) {
45 
46         char *utf;
47 
48         utf = (char *)JNI_FUNC_PTR(env,GetStringUTFChars)(env, string, NULL);
49         if (!(*env)->ExceptionCheck(env)) {
50             (void)outStream_writeString(out, utf);
51             JNI_FUNC_PTR(env,ReleaseStringUTFChars)(env, string, utf);
52         }
53 
54     } END_WITH_LOCAL_REFS(env);
55 
56     return JNI_TRUE;
57 }
58 
59 Command StringReference_Commands[] = {
60     {value, "Value"}
61 };
