26 #include "util.h"
27 #include "ClassObjectReferenceImpl.h"
28 #include "inStream.h"
29 #include "outStream.h"
30 
31 static jboolean
32 reflectedType(PacketInputStream *in, PacketOutputStream *out)
33 {
34     jbyte tag;
35     jobject object;
36     JNIEnv *env;
37 
38     env = getEnv();
39 
40     object = inStream_readObjectRef(env, in);
41     if (inStream_error(in)) {
42         return JNI_TRUE;
43     }
44 
45     /*
46      * In our implementation, the reference type id is the same as the
47      * class object id, so we bounce it right back.
48      *
49      */
50 
51     tag = referenceTypeTag(object);
52     (void)outStream_writeByte(out, tag);
53     (void)outStream_writeObjectRef(env, out, object);
54 
55     return JNI_TRUE;
56 }
57 
58 Command ClassObjectReference_Commands[] = {
59     {reflectedType, "ReflectedType"}
60 };
