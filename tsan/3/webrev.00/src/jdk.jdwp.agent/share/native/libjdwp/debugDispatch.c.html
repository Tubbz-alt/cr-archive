 47 static CommandSet **cmdSetsArray;
 48 
 49 void
 50 debugDispatch_initialize(void)
 51 {
 52     /*
 53      * Create the level-one (CommandSet) dispatch table.
 54      * Zero the table so that unknown CommandSets do not
 55      * cause random errors.
 56      */
 57     cmdSetsArray = jvmtiAllocate((JDWP_HIGHEST_COMMAND_SET+1) * sizeof(CommandSet *));
 58 
 59     if (cmdSetsArray == NULL) {
 60         EXIT_ERROR(AGENT_ERROR_OUT_OF_MEMORY,"command set array");
 61     }
 62 
 63     (void)memset(cmdSetsArray, 0, (JDWP_HIGHEST_COMMAND_SET+1) * sizeof(CommandSet *));
 64 
 65     /*
 66      * Create the level-two (Command) dispatch tables to the
 67      * corresponding slots in the CommandSet dispatch table..
 68      */
 69     cmdSetsArray[JDWP_COMMAND_SET(VirtualMachine)] = &VirtualMachine_CmdSet;
 70     cmdSetsArray[JDWP_COMMAND_SET(ReferenceType)] = &ReferenceType_CmdSet;
 71     cmdSetsArray[JDWP_COMMAND_SET(ClassType)] = &ClassType_CmdSet;
 72     cmdSetsArray[JDWP_COMMAND_SET(InterfaceType)] = &InterfaceType_CmdSet;
 73     cmdSetsArray[JDWP_COMMAND_SET(ArrayType)] = &ArrayType_CmdSet;
 74 
 75     cmdSetsArray[JDWP_COMMAND_SET(Field)] = &Field_CmdSet;
 76     cmdSetsArray[JDWP_COMMAND_SET(Method)] = &Method_CmdSet;
 77     cmdSetsArray[JDWP_COMMAND_SET(ObjectReference)] = &ObjectReference_CmdSet;
 78     cmdSetsArray[JDWP_COMMAND_SET(StringReference)] = &StringReference_CmdSet;
 79     cmdSetsArray[JDWP_COMMAND_SET(ThreadReference)] = &ThreadReference_CmdSet;
 80     cmdSetsArray[JDWP_COMMAND_SET(ThreadGroupReference)] = &ThreadGroupReference_CmdSet;
 81     cmdSetsArray[JDWP_COMMAND_SET(ClassLoaderReference)] = &ClassLoaderReference_CmdSet;
 82     cmdSetsArray[JDWP_COMMAND_SET(ArrayReference)] = &ArrayReference_CmdSet;
 83     cmdSetsArray[JDWP_COMMAND_SET(EventRequest)] = &EventRequest_CmdSet;
 84     cmdSetsArray[JDWP_COMMAND_SET(StackFrame)] = &StackFrame_CmdSet;
 85     cmdSetsArray[JDWP_COMMAND_SET(ClassObjectReference)] = &ClassObjectReference_CmdSet;
 86     cmdSetsArray[JDWP_COMMAND_SET(ModuleReference)] = &ModuleReference_CmdSet;
 87 }
 88 
 89 void
 90 debugDispatch_reset(void)
 91 {
 92 }
 93 
 94 CommandHandler
 95 debugDispatch_getHandler(int cmdSetNum, int cmdNum, const char **cmdSetName_p, const char **cmdName_p)
 96 {
 97     CommandSet *cmd_set;
 98     *cmdSetName_p = "<Invalid CommandSet>";
 99     *cmdName_p = "<Unkown Command>";
100 
101     if (cmdSetNum > JDWP_HIGHEST_COMMAND_SET) {
102         return NULL;
103     }
104 
105     cmd_set = cmdSetsArray[cmdSetNum];
106     if (cmd_set == NULL) {
107       return NULL;
108     }
109 
110     *cmdSetName_p = cmd_set->cmd_set_name;
111     if (cmdNum > cmd_set->num_cmds) {
112         *cmdName_p = "<Invalid Command>";
113         return NULL;
114     } else {
115         *cmdName_p = cmd_set->cmds[cmdNum - 1].cmd_name;
116         return cmd_set->cmds[cmdNum - 1].cmd_handler;
117     }
118 }
