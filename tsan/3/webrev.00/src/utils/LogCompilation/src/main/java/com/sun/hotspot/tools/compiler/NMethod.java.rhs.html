 25 package com.sun.hotspot.tools.compiler;
 26 
 27 import java.io.PrintStream;
 28 
 29 /**
 30  * A compilation log event that is signalled whenever a new nmethod (a native
 31  * method, a compilation result) is created.
 32  */
 33 public class NMethod extends BasicLogEvent {
 34 
 35     /**
 36      * The nmethod's starting address in memory.
 37      */
 38     private long address;
 39 
 40     /**
 41      * The nmethod's size in bytes.
 42      */
 43     private long size;
 44 
 45     /**
 46      * The nmethod's compilation level.
 47      */
 48     private long level;
 49 
 50     /**
 51      * The name of the compiler performing this compilation.
 52      */
 53     private String compiler;
 54 
 55     NMethod(double s, String i, long a, long sz) {
 56         super(s, i);
 57         address = a;
 58         size = sz;
 59     }
 60 
 61     public void print(PrintStream out, boolean printID) {
 62         // XXX Currently we do nothing
 63         // throw new InternalError();
 64     }
 65 
 66     public long getAddress() {
 67         return address;
 68     }
 69 
 70     public void setAddress(long address) {
 71         this.address = address;
 72     }
 73 
 74     public long getSize() {
 75         return size;
 76     }
 77 
 78     public void setSize(long size) {
 79         this.size = size;
 80     }
 81 
 82     /**
 83      * @return the level
 84      */
 85     public long getLevel() {
 86         return level;
 87     }
 88 
 89     /**
 90      * @param level the level to set
 91      */
 92     public void setLevel(long level) {
 93         assert this.level == 0 || this.level == level;
 94         this.level = level;
 95     }
 96 
 97     /**
 98      * @return the compiler
 99      */
100     public String getCompiler() {
101         return compiler;
102     }
103 
104     /**
105      * @param compiler the compiler to set
106      */
107     public void setCompiler(String compiler) {
108         this.compiler = compiler;
109     }
110 }
