25 package com.sun.hotspot.tools.compiler;
26 
27 import java.io.PrintStream;
28 
29 /**
30  * A compilation log event that is signalled whenever a new nmethod (a native
31  * method, a compilation result) is created.
32  */
33 public class NMethod extends BasicLogEvent {
34 
35     /**
36      * The nmethod's starting address in memory.
37      */
38     private long address;
39 
40     /**
41      * The nmethod's size in bytes.
42      */
43     private long size;
44 
45     /**
46      * The nmethod's compilation level.
47      */
48     private long level;
49 
50     NMethod(double s, String i, long a, long sz) {
51         super(s, i);
52         address = a;
53         size = sz;
54     }
55 
56     public void print(PrintStream out, boolean printID) {
57         // XXX Currently we do nothing
58         // throw new InternalError();
59     }
60 
61     public long getAddress() {
62         return address;
63     }
64 
65     public void setAddress(long address) {
66         this.address = address;
67     }
68 
69     public long getSize() {
70         return size;
71     }
72 
73     public void setSize(long size) {
74         this.size = size;
75     }
76 
77     /**
78      * @return the level
79      */
80     public long getLevel() {
81         return level;
82     }
83 
84     /**
85      * @param level the level to set
86      */
87     public void setLevel(long level) {
88         this.level = level;
89     }
90 }
