 30 /**
 31  * One particular compilation, represented in the compilation log file as a
 32  * {@code task} element.
 33  */
 34 public class Compilation implements LogEvent {
 35 
 36     /**
 37      * The compilation ID.
 38      */
 39     private int id;
 40 
 41     /**
 42      * Whether this is a compilation for on-stack replacement (OSR).
 43      */
 44     private boolean osr;
 45 
 46     /**
 47      * The method being compiled.
 48      */
 49     private Method method;
 50 
 51     /**
 52      * The {@linkplain CallSite scope} of this compilation. This is created as
 53      * an empty {@link CallSite} instance, to be filled with data (and
 54      * meaning) later on.
 55      */
 56     private CallSite call = new CallSite();
 57 
 58     /**
 59      * In case a {@code late_inline} event occurs during the compilation, this
 60      * field holds the information about it.
 61      */
 62     private CallSite lateInlineCall = new CallSite();
 63 
 64     /**
 65      * The bytecode instruction index for on-stack replacement compilations; -1
 66      * if this is not an OSR compilation.
 67      */
 68     private int bci;
 69 
 70     /**
 71      * The method under compilation's invocation count.
 72      */
 73     private String icount;
 74 
 75     /**
 76      * The method under compilation's backedge count.
 77      */
 78     private String bcount;
 79 
 80     /**
 81      * Additional information for special compilations (e.g., adapters).
 82      */
 83     private String special;
 84 
 85     /**
 86      * The name of the compiler performing this compilation.
 87      */
 88     private String compiler;
 89 
 90     /**
 91      * Start time stamp.
 92      */
 93     private double start;
 94 
 95     /**
 96      * End time stamp.
 97      */
 98     private double end;
 99 
100     /**
101      * Trip count of the register allocator.
102      */
103     private int attempts;
104 
105     /**
106      * The compilation result (a native method).
107      */
108     private NMethod nmethod;
109 
110     /**
111      * The phases through which this compilation goes.
112      */
113     private ArrayList<Phase> phases = new ArrayList<>(4);
114 
115     /**
116      * In case this compilation fails, the reason for that.
117      */
118     private String failureReason;
119 
120     Compilation(int id) {
121         this.id = id;
122     }
123 
124     void reset() {
125         call = new CallSite();
126         lateInlineCall = new CallSite();
127         phases = new ArrayList<>(4);
128     }
129 
130     /**
131      * Get a compilation phase by name, or {@code null}.
132      *
133      * @param s the name of the phase to retrieve in this compilation.
134      *
135      * @return a compilation phase, or {@code null} if no phase with the given
136      *         name is found.
137      */
138     Phase getPhase(String s) {
139         for (Phase p : getPhases()) {
140             if (p.getName().equals(s)) {
141                 return p;
142             }
143         }
144         return null;
145     }
146 
147     double getRegallocTime() {
148         return getPhase("regalloc").getElapsedTime();
149     }

163     @Override
164     public String toString() {
165         StringBuilder sb = new StringBuilder();
166         sb.append(getId());
167         sb.append(" ");
168         sb.append(getCompiler());
169         sb.append(" ");
170         sb.append(getMethod());
171         sb.append(" ");
172         sb.append(getIcount());
173         sb.append("+");
174         sb.append(getBcount());
175         sb.append("\n");
176         if (getCall() != null && getCall().getCalls() != null) {
177             for (CallSite site : getCall().getCalls()) {
178                 sb.append(site);
179                 sb.append("\n");
180             }
181         }
182         if (getLateInlineCall().getCalls() != null) {
183             sb.append("late inline:\n");
184             for (CallSite site : getLateInlineCall().getCalls()) {
185                 sb.append(site);
186                 sb.append("\n");
187             }
188         }
189         return sb.toString();
190     }

192     public void printShort(PrintStream stream) {
193         if (getMethod() == null) {
194             stream.println(getSpecial());
195         } else {
196             int bc = isOsr() ? getBCI() : -1;
197             stream.print(getId() + getMethod().decodeFlags(bc) + " " + compiler + " " + getMethod().format(bc));
198         }
199     }

202         print(stream, 0, printID, true, false);
203     }

206         print(stream, 0, printID, printInlining, false);
207     }

209     public void print(PrintStream stream, boolean printID, boolean printInlining, boolean printUncommonTraps) {
210         print(stream, 0, printID, printInlining, printUncommonTraps);
211     }
212 
213     public void print(PrintStream stream, int indent, boolean printID, boolean printInlining, boolean printUncommonTraps) {
214         if (getMethod() == null) {
215             stream.println(getSpecial());
216         } else {
217             if (printID) {
218                 stream.print(getId());
219                 // Print the comp level next to the id as with +PrintCompilation
220                 if (nmethod != null && nmethod.getLevel() != 0) {
221                     stream.print(" " + nmethod.getLevel());
222                 }
223             }
224             int bc = isOsr() ? getBCI() : -1;
225             stream.print(getMethod().decodeFlags(bc) + " " + compiler + " " + getMethod().format(bc));
226             stream.println();
227             if (getFailureReason() != null) {
228                 stream.println("COMPILE SKIPPED: " + getFailureReason() + " (not retryable)");
229             }
230             if (printInlining && call.getCalls() != null) {
231                 for (CallSite site : call.getCalls()) {
232                     site.print(stream, indent + 2, printInlining, printUncommonTraps);
233                 }
234             }
235             if (printUncommonTraps && call.getTraps() != null) {
236                 for (UncommonTrap site : call.getTraps()) {
237                     site.print(stream, indent + 2);
238                 }
239             }
240             if (printInlining && lateInlineCall.getCalls() != null) {
241                 stream.println("late inline:");
242                 for (CallSite site : lateInlineCall.getCalls()) {
243                     site.print(stream, indent + 2, printInlining, printUncommonTraps);
244                 }
245             }
246         }
247     }
