 25 package com.sun.hotspot.tools.compiler;
 26 
 27 import java.util.Arrays;
 28 
 29 import static com.sun.hotspot.tools.compiler.Constants.*;
 30 
 31 /**
 32  * Representation of a Java method in a compilation log.
 33  */
 34 public class Method {
 35 
 36     /**
 37      * The name of the class holding the method.
 38      */
 39     private String holder;
 40 
 41     /**
 42      * The method's name.
 43      */
 44     private String name;
 45 
 46     /**
 47      * The return type of the method, as a fully qualified (source-level) class
 48      * or primitive type name.
 49      */
 50     private String returnType;
 51 
 52     /**
 53      * The method's signature, in internal form.
 54      */
 55     private String signature;
 56 
 57     /**
 58      * The length of the method's byte code.
 59      */
 60     private String bytes;
 61 
 62     /**
 63      * The number of times this method was invoked in the interpreter.
 64      */
 65     private String iicount;
 66 
 67     /**
 68      * The method's flags, in the form of a {@code String} representing the
 69      * {@code int} encoding them.
 70      */
 71     private String flags;
 72 
 73     /**
 74      * The name of the compiler performing this compilation.
 75      */
 76     private String compiler;
 77 
 78     /**
 79      * The nmethod's compilation level.
 80      */
 81     private long level;
 82 
 83     /**
 84      * Decode the {@link flags} numerical string to a format for console
 85      * output. The result does not honour all possible flags but includes
 86      * information about OSR compilation.
 87      *
 88      * @param osr_bci the byte code index at which an OSR compilation takes
 89      * place, or -1 if the compilation is not an OSR one.
 90      */
 91     String decodeFlags(int osr_bci) {
 92         int f = Integer.parseInt(getFlags());
 93         char[] c = new char[4];
 94         Arrays.fill(c, ' ');
 95         if (osr_bci >= 0) {
 96             c[0] = '%';
 97         }
 98         if ((f & JVM_ACC_SYNCHRONIZED) != 0) {
 99             c[1] = 's';
100         }
101         return new String(c);
102     }
103 
104     /**
105      * Format this method for console output.
106      *
107      * @param osr_bci the byte code index at which OSR takes place, or -1 if no
108      * OSR compilation is going on.
109      */
110     String format(int osr_bci) {
111         if (osr_bci >= 0) {
112             return getHolder() + "::" + getName() + " @ " + osr_bci + " (" + getBytes() + " bytes)";
113         } else {
114             return getHolder() + "::" + getName() + " (" + getBytes() + " bytes)";
115         }
116     }
117 
118     @Override
119     public String toString() {
120         return getHolder() + "::" + getName() + " (" + getBytes() + " bytes)";
121     }
122 
123     public String getFullName() {
124         return getHolder().replace('/', '.') + "." + getName() + signature;
125     }
126 
127     public String getHolder() {
128         return holder;
129     }
130 
131     public void setHolder(String holder) {
132         this.holder = holder;
133     }
134 
135     public String getName() {
136         return name;
137     }
138 
139     public void setName(String name) {
140         this.name = name;
141     }
142 
143     public String getReturnType() {
144         return returnType;
145     }
146 
147     public void setReturnType(String returnType) {
148         this.returnType = returnType;
149     }
150 
151     public String getSignature() {
152         return signature;
153     }
154 
155     public void setSignature(String signature) {
156         this.signature = signature.replace('/', '.');
157     }
158 
159     public String getArguments() {
160         return signature.substring(0, signature.indexOf(')') + 1);
161     }
162 
163     public String getBytes() {
164         return bytes;
165     }
166 
167     public void setBytes(String bytes) {
168         this.bytes = bytes;
169     }
170 
171     public String getIICount() {
172         return iicount;
173     }
174 
175     public void setIICount(String iicount) {
176         this.iicount = iicount;
177     }
178 
179     public String getFlags() {
180         return flags;
181     }
182 
183     public void setFlags(String flags) {
184         this.flags = flags;
185     }
186 
187     @Override
188     public boolean equals(Object o) {
189         if (o instanceof Method) {
190             Method other = (Method) o;
191             return holder.equals(other.holder) && name.equals(other.name) && signature.equals(other.signature);
192         }
193         return false;
194     }
195 
196     public int hashCode() {
197         return holder.hashCode() ^ name.hashCode();
198     }
199 
200     /**
201      * @return the compiler
202      */
203     public String getCompiler() {
204         return compiler;
205     }
206 
207     /**
208      * @param compiler the compiler to set
209      */
210     public void setCompiler(String compiler) {
211         this.compiler = compiler;
212     }
213 
214     /**
215      * @return the level
216      */
217     public long getLevel() {
218         return level;
219     }
220 
221     /**
222      * @param level the level to set
223      */
224     public void setLevel(long level) {
225         assert this.level == 0 || this.level == level;
226         this.level = level;
227     }
228 }
