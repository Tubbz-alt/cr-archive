 45 /**
 46  * Filters for selecting just the elements of interest from a
 47  * collection of elements.  The returned sets and lists are new
 48  * collections and do use the argument as a backing store.  The
 49  * methods in this class do not make any attempts to guard against
 50  * concurrent modifications of the arguments.  The returned sets and
 51  * lists are mutable but unsafe for concurrent access.  A returned set
 52  * has the same iteration order as the argument set to a method.
 53  *
 54  * <p>If iterables and sets containing {@code null} are passed as
 55  * arguments to methods in this class, a {@code NullPointerException}
 56  * will be thrown.
 57  *
 58  * @author Joseph D. Darcy
 59  * @author Scott Seligman
 60  * @author Peter von der Ah&eacute;
 61  * @author Martin Buchholz
 62  * @since 1.6
 63  */
 64 public class ElementFilter {
 65     private ElementFilter() {} // Do not instantiate.
 66

 74         Collections.unmodifiableSet(EnumSet.of(ElementKind.METHOD));

260     // Assumes targetKinds and E are sensible.
261     private static <E extends Element> List<E> listFilter(Iterable<? extends Element> elements,
262                                                           Set<ElementKind> targetKinds,
263                                                           Class<E> clazz) {
264         List<E> list = new ArrayList<>();
265         for (Element e : elements) {
266             if (targetKinds.contains(e.getKind()))
267                 list.add(clazz.cast(e));
268         }
269         return list;
270     }
271 
272     // Assumes targetKinds and E are sensible.
273     private static <E extends Element> Set<E> setFilter(Set<? extends Element> elements,
274                                                         Set<ElementKind> targetKinds,
275                                                         Class<E> clazz) {
276         // Return set preserving iteration order of input set.
277         Set<E> set = new LinkedHashSet<>();
278         for (Element e : elements) {
279             if (targetKinds.contains(e.getKind()))
280                 set.add(clazz.cast(e));
281         }
282         return set;
283     }

344     // Assumes directiveKind and D are sensible.
345     private static <D extends Directive> List<D> listFilter(Iterable<? extends Directive> directives,
346                                                           DirectiveKind directiveKind,
347                                                           Class<D> clazz) {
348         List<D> list = new ArrayList<>();
349         for (Directive d : directives) {
350             if (d.getKind() == directiveKind)
351                 list.add(clazz.cast(d));
352         }
353         return list;
354     }
