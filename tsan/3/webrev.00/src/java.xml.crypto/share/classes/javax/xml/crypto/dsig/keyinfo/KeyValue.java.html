 28 package javax.xml.crypto.dsig.keyinfo;
 29 
 30 import java.security.KeyException;
 31 import java.security.PublicKey;
 32 import java.security.interfaces.DSAPublicKey;
 33 import java.security.interfaces.RSAPublicKey;
 34 import javax.xml.crypto.XMLStructure;
 35 
 36 /**
 37  * A representation of the XML <code>KeyValue</code> element as defined
 38  * in the <a href="http://www.w3.org/TR/xmldsig-core/">
 39  * W3C Recommendation for XML-Signature Syntax and Processing</a>. A
 40  * <code>KeyValue</code> object contains a single public key that may be
 41  * useful in validating the signature. The XML schema definition is defined as:
 42  *
 43  * <pre>
 44  *    &lt;element name="KeyValue" type="ds:KeyValueType"/&gt;
 45  *    &lt;complexType name="KeyValueType" mixed="true"&gt;
 46  *      &lt;choice&gt;
 47  *        &lt;element ref="ds:DSAKeyValue"/&gt;
 48  *        &lt;element ref="ds:RSAKeyValue"/&gt;
 49  *        &lt;!-- &lt;element ref="dsig11:ECKeyValue"/&gt; --&gt;
 50  *        &lt;!-- ECC keys (XMLDsig 1.1) will use the any element --&gt;
 51  *        &lt;any namespace="##other" processContents="lax"/&gt;
 52  *      &lt;/choice&gt;
 53  *    &lt;/complexType&gt;
 54  *
 55  *    &lt;element name="DSAKeyValue" type="ds:DSAKeyValueType"/&gt;
 56  *    &lt;complexType name="DSAKeyValueType"&gt;
 57  *      &lt;sequence&gt;
 58  *        &lt;sequence minOccurs="0"&gt;
 59  *          &lt;element name="P" type="ds:CryptoBinary"/&gt;
 60  *          &lt;element name="Q" type="ds:CryptoBinary"/&gt;
 61  *        &lt;/sequence&gt;
 62  *        &lt;element name="G" type="ds:CryptoBinary" minOccurs="0"/&gt;
 63  *        &lt;element name="Y" type="ds:CryptoBinary"/&gt;
 64  *        &lt;element name="J" type="ds:CryptoBinary" minOccurs="0"/&gt;
 65  *        &lt;sequence minOccurs="0"&gt;
 66  *          &lt;element name="Seed" type="ds:CryptoBinary"/&gt;
 67  *          &lt;element name="PgenCounter" type="ds:CryptoBinary"/&gt;
 68  *        &lt;/sequence&gt;
 69  *      &lt;/sequence&gt;
 70  *    &lt;/complexType&gt;
 71  *
 72  *    &lt;element name="RSAKeyValue" type="ds:RSAKeyValueType"/&gt;
 73  *    &lt;complexType name="RSAKeyValueType"&gt;
 74  *      &lt;sequence&gt;
 75  *        &lt;element name="Modulus" type="ds:CryptoBinary"/&gt;
 76  *        &lt;element name="Exponent" type="ds:CryptoBinary"/&gt;
 77  *      &lt;/sequence&gt;
 78  *    &lt;/complexType&gt;
 79  *
 80  *    &lt;complexType name="ECKeyValueType"&gt;
 81  *      &lt;sequence&gt;
 82  *        &lt;choice&gt;
 83  *          &lt;element name="ECParameters" type="dsig11:ECParametersType" /&gt;
 84  *          &lt;element name="NamedCurve" type="dsig11:NamedCurveType" /&gt;
 85  *        &lt;/choice&gt;
 86  *        &lt;element name="PublicKey" type="dsig11:ECPointType" /&gt;
 87  *      &lt;/sequence&gt;
 88  *      &lt;attribute name="Id" type="ID" use="optional" /&gt;
 89  *    &lt;/complexType&gt;
 90  *
 91  *    &lt;complexType name="NamedCurveType"&gt;
 92  *      &lt;attribute name="URI" type="anyURI" use="required" /&gt;
 93  *    &lt;/complexType&gt;
 94  *
 95  *    &lt;simpleType name="ECPointType"&gt;
 96  *      &lt;restriction base="ds:CryptoBinary" /&gt;
 97  *    &lt;/simpleType&gt;
 98  * </pre>
 99  * See section 4.5.2.3.1 of the W3C Recommendation for the definition
100  * of ECParametersType.
101  *
102  * <p>A <code>KeyValue</code> instance may be created by invoking the
103  * {@link KeyInfoFactory#newKeyValue newKeyValue} method of the
104  * {@link KeyInfoFactory} class, and passing it a {@link
105  * java.security.PublicKey} representing the value of the public key. Here is
106  * an example of creating a <code>KeyValue</code> from a {@link DSAPublicKey}
107  * of a {@link java.security.cert.Certificate} stored in a
108  * {@link java.security.KeyStore}:
109  * <pre>
110  * KeyStore keyStore = KeyStore.getInstance(KeyStore.getDefaultType());
111  * PublicKey dsaPublicKey = keyStore.getCertificate("myDSASigningCert").getPublicKey();
112  * KeyInfoFactory factory = KeyInfoFactory.getInstance("DOM");
113  * KeyValue keyValue = factory.newKeyValue(dsaPublicKey);
114  * </pre>
115  *
116  * This class returns the <code>DSAKeyValue</code> and
117  * <code>RSAKeyValue</code> elements as objects of type
118  * {@link DSAPublicKey} and {@link RSAPublicKey}, respectively. Note that not
119  * all of the fields in the schema are accessible as parameters of these
120  * types.
121  *
122  * @author Sean Mullan
123  * @author JSR 105 Expert Group
124  * @since 1.6
125  * @see KeyInfoFactory#newKeyValue(PublicKey)
126  */
127 public interface KeyValue extends XMLStructure {
128 
129     /**
130      * URI identifying the DSA KeyValue KeyInfo type:
131      * http://www.w3.org/2000/09/xmldsig#DSAKeyValue. This can be specified as
132      * the value of the <code>type</code> parameter of the
133      * {@link RetrievalMethod} class to describe a remote
134      * <code>DSAKeyValue</code> structure.
135      */
136     final static String DSA_TYPE =
137         "http://www.w3.org/2000/09/xmldsig#DSAKeyValue";
138 
139     /**
140      * URI identifying the RSA KeyValue KeyInfo type:
141      * http://www.w3.org/2000/09/xmldsig#RSAKeyValue. This can be specified as
142      * the value of the <code>type</code> parameter of the
143      * {@link RetrievalMethod} class to describe a remote
144      * <code>RSAKeyValue</code> structure.
145      */
146     final static String RSA_TYPE =
147         "http://www.w3.org/2000/09/xmldsig#RSAKeyValue";
148 
149     /**
150      * URI identifying the EC KeyValue KeyInfo type:
151      * http://www.w3.org/2009/xmldsig11#ECKeyValue. This can be specified as
152      * the value of the <code>type</code> parameter of the
153      * {@link RetrievalMethod} class to describe a remote
154      * <code>ECKeyValue</code> structure.
155      */
156     final static String EC_TYPE =
157         "http://www.w3.org/2009/xmldsig11#ECKeyValue";
158 
159     /**
160      * Returns the public key of this <code>KeyValue</code>.
161      *
162      * @return the public key of this <code>KeyValue</code>
163      * @throws KeyException if this <code>KeyValue</code> cannot be converted
164      *    to a <code>PublicKey</code>
165      */
166     PublicKey getPublicKey() throws KeyException;
167 }
