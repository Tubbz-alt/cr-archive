 44 /**
 45  * DOM-based implementation of PGPData.
 46  *
 47  */
 48 public final class DOMPGPData extends DOMStructure implements PGPData {
 49 
 50     private final byte[] keyId;
 51     private final byte[] keyPacket;
 52     private final List<XMLStructure> externalElements;
 53 
 54     /**
 55      * Creates a {@code DOMPGPData} containing the specified key packet.
 56      * and optional list of external elements.
 57      *
 58      * @param keyPacket a PGP Key Material Packet as defined in section 5.5 of
 59      *    <a href="http://www.ietf.org/rfc/rfc2440.txt">RFC 2440</a>. The
 60      *    array is cloned to prevent subsequent modification.
 61      * @param other a list of {@link XMLStructure}s representing elements from
 62      *    an external namespace. The list is defensively copied to prevent
 63      *    subsequent modification. May be {@code null} or empty.
 64      * @throws NullPointerException if {@code keyPacket} is
 65      *    {@code null}
 66      * @throws IllegalArgumentException if the key packet is not in the
 67      *    correct format
 68      * @throws ClassCastException if {@code other} contains any
 69      *    entries that are not of type {@link XMLStructure}
 70      */
 71     public DOMPGPData(byte[] keyPacket, List<? extends XMLStructure> other) {
 72         if (keyPacket == null) {
 73             throw new NullPointerException("keyPacket cannot be null");
 74         }
 75         if (other == null || other.isEmpty()) {
 76             this.externalElements = Collections.emptyList();
 77         } else {
 78             this.externalElements =
 79                 Collections.unmodifiableList(new ArrayList<>(other));
 80             for (int i = 0, size = this.externalElements.size(); i < size; i++) {
 81                 if (!(this.externalElements.get(i) instanceof XMLStructure)) {
 82                     throw new ClassCastException
 83                         ("other["+i+"] is not a valid PGPData type");
 84                 }
 85             }
 86         }
 87         this.keyPacket = keyPacket.clone();
 88         checkKeyPacket(keyPacket);
 89         this.keyId = null;
 90     }
 91 
 92     /**
 93      * Creates a {@code DOMPGPData} containing the specified key id and
 94      * optional key packet and list of external elements.
 95      *
 96      * @param keyId a PGP public key id as defined in section 11.2 of
 97      *    <a href="http://www.ietf.org/rfc/rfc2440.txt">RFC 2440</a>. The
 98      *    array is cloned to prevent subsequent modification.
 99      * @param keyPacket a PGP Key Material Packet as defined in section 5.5 of
100      *    <a href="http://www.ietf.org/rfc/rfc2440.txt">RFC 2440</a> (may
101      *    be {@code null}). The array is cloned to prevent subsequent
102      *    modification.
103      * @param other a list of {@link XMLStructure}s representing elements from
104      *    an external namespace. The list is defensively copied to prevent
105      *    subsequent modification. May be {@code null} or empty.
106      * @throws NullPointerException if {@code keyId} is {@code null}
107      * @throws IllegalArgumentException if the key id or packet is not in the
108      *    correct format
109      * @throws ClassCastException if {@code other} contains any
110      *    entries that are not of type {@link XMLStructure}
111      */
112     public DOMPGPData(byte[] keyId, byte[] keyPacket,
113                       List<? extends XMLStructure> other)
114     {
115         if (keyId == null) {
116             throw new NullPointerException("keyId cannot be null");
117         }
118         // key ids must be 8 bytes
119         if (keyId.length != 8) {
120             throw new IllegalArgumentException("keyId must be 8 bytes long");

140     }
141 
142     /**
143      * Creates a {@code DOMPGPData} from an element.
144      *
145      * @param pdElem a PGPData element
146      */
147     public DOMPGPData(Element pdElem) throws MarshalException {
148         // get all children nodes
149         byte[] pgpKeyId = null;
150         byte[] pgpKeyPacket = null;
151 
152         List<XMLStructure> other = new ArrayList<>();
153         Node firstChild = pdElem.getFirstChild();
154         while (firstChild != null) {
155             if (firstChild.getNodeType() == Node.ELEMENT_NODE) {
156                 Element childElem = (Element)firstChild;
157                 String localName = childElem.getLocalName();
158                 String namespace = childElem.getNamespaceURI();
159                 if ("PGPKeyID".equals(localName) && XMLSignature.XMLNS.equals(namespace)) {
160                     String content = XMLUtils.getFullTextChildrenFromElement(childElem);
161                     pgpKeyId = XMLUtils.decode(content);
162                 } else if ("PGPKeyPacket".equals(localName) && XMLSignature.XMLNS.equals(namespace)) {
163                     String content = XMLUtils.getFullTextChildrenFromElement(childElem);
164                     pgpKeyPacket = XMLUtils.decode(content);
165                 } else {
166                     other.add
167                     (new javax.xml.crypto.dom.DOMStructure(childElem));
168                 }
169             }
170             firstChild = firstChild.getNextSibling();
171         }
172         this.keyId = pgpKeyId;
173         this.keyPacket = pgpKeyPacket;
174         this.externalElements = Collections.unmodifiableList(other);
175     }
176 
177     public byte[] getKeyId() {
178         return keyId == null ? null : keyId.clone();
179     }
180 
181     public byte[] getKeyPacket() {
182         return keyPacket == null ? null : keyPacket.clone();
183     }
