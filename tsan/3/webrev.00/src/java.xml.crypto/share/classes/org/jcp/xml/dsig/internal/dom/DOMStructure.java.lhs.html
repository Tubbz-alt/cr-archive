29 package org.jcp.xml.dsig.internal.dom;
30 
31 import javax.xml.crypto.MarshalException;
32 import javax.xml.crypto.XMLStructure;
33 import javax.xml.crypto.dom.DOMCryptoContext;
34 import org.w3c.dom.Node;
35 
36 /**
37  * DOM-based abstract implementation of XMLStructure.
38  *
39  */
40 public abstract class DOMStructure implements XMLStructure {
41 
42     public final boolean isFeatureSupported(String feature) {
43         if (feature == null) {
44             throw new NullPointerException();
45         } else {
46             return false;
47         }
48     }
49 
50     public abstract void marshal(Node parent, String dsPrefix,
51         DOMCryptoContext context) throws MarshalException;
52 }
