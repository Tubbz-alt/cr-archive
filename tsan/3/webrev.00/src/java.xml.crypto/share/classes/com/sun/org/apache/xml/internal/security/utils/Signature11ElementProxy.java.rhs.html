23 package com.sun.org.apache.xml.internal.security.utils;
24 
25 import com.sun.org.apache.xml.internal.security.exceptions.XMLSecurityException;
26 import org.w3c.dom.Document;
27 import org.w3c.dom.Element;
28 
29 /**
30  * Class SignatureElementProxy
31  *
32  */
33 public abstract class Signature11ElementProxy extends ElementProxy {
34 
35     protected Signature11ElementProxy() {
36     }
37 
38     /**
39      * Constructor Signature11ElementProxy
40      *
41      * @param doc
42      */
43     public Signature11ElementProxy(Document doc) {
44         if (doc == null) {
45             throw new RuntimeException("Document is null");
46         }
47 
48         setDocument(doc);
49         setElement(XMLUtils.createElementInSignature11Space(doc, this.getBaseLocalName()));
50 
51         String prefix = ElementProxy.getDefaultPrefix(getBaseNamespace());
52         if (prefix == null || prefix.length() == 0) {
53             getElement().setAttributeNS(Constants.NamespaceSpecNS, "xmlns", getBaseNamespace());
54         } else {
55             getElement().setAttributeNS(Constants.NamespaceSpecNS, "xmlns:" + prefix, getBaseNamespace());
56         }
57     }
58 
59     /**
60      * Constructor Signature11ElementProxy
61      *
62      * @param element
63      * @param baseURI
64      * @throws XMLSecurityException
65      */
66     public Signature11ElementProxy(Element element, String baseURI) throws XMLSecurityException {
67         super(element, baseURI);
68 
69     }
70 
71     /** {@inheritDoc} */
72     public String getBaseNamespace() {
73         return Constants.SignatureSpec11NS;
74     }
75 }
