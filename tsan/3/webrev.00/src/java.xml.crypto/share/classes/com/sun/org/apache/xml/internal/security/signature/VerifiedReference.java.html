23 package com.sun.org.apache.xml.internal.security.signature;
24 
25 import java.util.Collections;
26 import java.util.List;
27 
28 /**
29  * Holds the result of a Reference validation.
30  */
31 public class VerifiedReference {
32 
33     private final boolean valid;
34     private final String uri;
35     private final List<VerifiedReference> manifestReferences;
36 
37     /**
38      * @param valid Whether this Reference was successfully validated or not
39      * @param uri The URI of this Reference
40      * @param manifestReferences If this reference is a reference to a Manifest, this holds the list
41      * of verified referenes associated with this Manifest
42      */
43     public VerifiedReference(boolean valid, String uri, List<VerifiedReference> manifestReferences) {
44         this.valid = valid;
45         this.uri = uri;
46         if (manifestReferences != null) {
47             this.manifestReferences = manifestReferences;
48         } else {
49             this.manifestReferences = Collections.emptyList();
50         }
51     }
52 
53     public boolean isValid() {
54         return valid;
55     }
56 
57     public String getUri() {
58         return uri;
59     }
60 
61     public List<VerifiedReference> getManifestReferences() {
62         return Collections.unmodifiableList(manifestReferences);
63     }
64 }
