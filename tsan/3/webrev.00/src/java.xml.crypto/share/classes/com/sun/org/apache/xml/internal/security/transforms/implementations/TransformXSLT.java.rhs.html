 50 /**
 51  * Class TransformXSLT
 52  *
 53  * Implements the {@code http://www.w3.org/TR/1999/REC-xslt-19991116}
 54  * transform.
 55  *
 56  */
 57 public class TransformXSLT extends TransformSpi {
 58 
 59     /** Field implementedTransformURI */
 60     public static final String implementedTransformURI =
 61         Transforms.TRANSFORM_XSLT;
 62 
 63     static final String XSLTSpecNS = "http://www.w3.org/1999/XSL/Transform";
 64     static final String defaultXSLTSpecNSprefix = "xslt";
 65     static final String XSLTSTYLESHEET = "stylesheet";
 66 
 67     private static final com.sun.org.slf4j.internal.Logger LOG =
 68         com.sun.org.slf4j.internal.LoggerFactory.getLogger(TransformXSLT.class);
 69 
 70     /**
 71      * Method engineGetURI
 72      *
 73      * {@inheritDoc}
 74      */
 75     protected String engineGetURI() {
 76         return implementedTransformURI;
 77     }
 78 
 79     protected XMLSignatureInput enginePerformTransform(
 80         XMLSignatureInput input, OutputStream baos, Transform transformObject
 81     ) throws IOException, TransformationException {
 82         try {
 83             Element transformElement = transformObject.getElement();
 84 
 85             Element xsltElement =
 86                 XMLUtils.selectNode(transformElement.getFirstChild(), XSLTSpecNS, "stylesheet", 0);
 87             if (xsltElement == null) {
 88                 xsltElement =
 89                     XMLUtils.selectNode(transformElement.getFirstChild(), XSLTSpecNS, "transform", 0);
 90             }
 91             if (xsltElement == null) {
 92                 Object exArgs[] = { "xslt:stylesheet", "Transform" };
 93 
 94                 throw new TransformationException("xml.WrongContent", exArgs);
 95             }
 96 
 97             TransformerFactory tFactory = TransformerFactory.newInstance();
 98             // Process XSLT stylesheets in a secure manner
 99             tFactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, Boolean.TRUE);
100 
101             /*
102              * This transform requires an octet stream as input. If the actual
103              * input is an XPath node-set, then the signature application should
104              * attempt to convert it to octets (apply Canonical XML]) as described
105              * in the Reference Processing Model (section 4.3.3.2).
106              */
107             Source stylesheet;
108 
109             /*
110              * This complicated transformation of the stylesheet itself is necessary
111              * because of the need to get the pure style sheet. If we simply say
112              * Source stylesheet = new DOMSource(this.xsltElement);
113              * whereby this.xsltElement is not the rootElement of the Document,
114              * this causes problems;
115              * so we convert the stylesheet to byte[] and use this as input stream
116              */
117             {
118                 try (ByteArrayOutputStream os = new ByteArrayOutputStream()) {
119                     Transformer transformer = tFactory.newTransformer();
120                     DOMSource source = new DOMSource(xsltElement);
121                     StreamResult result = new StreamResult(os);
122 
123                     transformer.transform(source, result);
124 
125                     stylesheet =
126                         new StreamSource(new ByteArrayInputStream(os.toByteArray()));
127                 }
128             }
129 
130             Transformer transformer = tFactory.newTransformer(stylesheet);
131 
132             // Force Xalan to use \n as line separator on all OSes. This
133             // avoids OS specific signature validation failures due to line
134             // separator differences in the transformed output. Unfortunately,
135             // this is not a standard JAXP property so will not work with non-Xalan
136             // implementations.
137             try {
138                 transformer.setOutputProperty("{http://xml.apache.org/xalan}line-separator", "\n");
139             } catch (Exception e) {
140                 LOG.warn("Unable to set Xalan line-separator property: " + e.getMessage());
141             }
142 
143             try (InputStream is = new ByteArrayInputStream(input.getBytes())) {
144                 Source xmlSource = new StreamSource(is);
145                 if (baos == null) {
146                     try (ByteArrayOutputStream baos1 = new ByteArrayOutputStream()) {
147                         StreamResult outputTarget = new StreamResult(baos1);
148                         transformer.transform(xmlSource, outputTarget);
149                         XMLSignatureInput output = new XMLSignatureInput(baos1.toByteArray());
150                         output.setSecureValidation(secureValidation);
151                         return output;
152                     }
153                 }
154                 StreamResult outputTarget = new StreamResult(baos);
155 
156                 transformer.transform(xmlSource, outputTarget);
157             }
158             XMLSignatureInput output = new XMLSignatureInput((byte[])null);
159             output.setSecureValidation(secureValidation);
160             output.setOutputStream(baos);
161             return output;
162         } catch (XMLSecurityException ex) {
163             throw new TransformationException(ex);
164         } catch (TransformerConfigurationException ex) {
165             throw new TransformationException(ex);
166         } catch (TransformerException ex) {
167             throw new TransformationException(ex);
168         }
169     }
170 }
