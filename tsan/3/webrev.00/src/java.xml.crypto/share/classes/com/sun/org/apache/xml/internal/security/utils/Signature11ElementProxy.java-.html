23 package com.sun.org.apache.xml.internal.security.utils;
24 
25 import com.sun.org.apache.xml.internal.security.exceptions.XMLSecurityException;
26 import org.w3c.dom.Document;
27 import org.w3c.dom.Element;
28 
29 /**
30  * Class SignatureElementProxy
31  *
32  */
33 public abstract class Signature11ElementProxy extends ElementProxy {
34 
35     protected Signature11ElementProxy() {
36     }
37 
38     /**
39      * Constructor Signature11ElementProxy
40      *
41      * @param doc
42      */
43     public Signature11ElementProxy(Document doc) {
44         if (doc == null) {
45             throw new RuntimeException("Document is null");
46         }
47 
48         setDocument(doc);
49         setElement(XMLUtils.createElementInSignature11Space(doc, this.getBaseLocalName()));
50     }
51 
52     /**
53      * Constructor Signature11ElementProxy
54      *
55      * @param element
56      * @param baseURI
57      * @throws XMLSecurityException
58      */
59     public Signature11ElementProxy(Element element, String baseURI) throws XMLSecurityException {
60         super(element, baseURI);
61 
62     }
63 
64     /** {@inheritDoc} */
65     public String getBaseNamespace() {
66         return Constants.SignatureSpec11NS;
67     }
68 }
