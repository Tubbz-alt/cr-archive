175     /*
176      * fields access methods
177      */
178     ///////////////////////////////////////////////////////
179     static final int CH(byte[] b, int n) {
180         return Byte.toUnsignedInt(b[n]);
181     }
182 
183     static final int SH(byte[] b, int n) {
184         return Byte.toUnsignedInt(b[n]) | (Byte.toUnsignedInt(b[n + 1]) << 8);
185     }
186 
187     static final long LG(byte[] b, int n) {
188         return ((SH(b, n)) | (SH(b, n + 2) << 16)) & 0xffffffffL;
189     }
190 
191     static final long LL(byte[] b, int n) {
192         return (LG(b, n)) | (LG(b, n + 4) << 32);
193     }
194 
195     static long getSig(byte[] b, int n) { return LG(b, n); }
196 
197     private static boolean pkSigAt(byte[] b, int n, int b1, int b2) {
198         return b[n] == 'P' & b[n + 1] == 'K' & b[n + 2] == b1 & b[n + 3] == b2;
199     }
200 
201     static boolean cenSigAt(byte[] b, int n) { return pkSigAt(b, n, 1, 2); }
202     static boolean locSigAt(byte[] b, int n) { return pkSigAt(b, n, 3, 4); }
203     static boolean endSigAt(byte[] b, int n) { return pkSigAt(b, n, 5, 6); }
204     static boolean extSigAt(byte[] b, int n) { return pkSigAt(b, n, 7, 8); }
205     static boolean end64SigAt(byte[] b, int n) { return pkSigAt(b, n, 6, 6); }
206     static boolean locator64SigAt(byte[] b, int n) { return pkSigAt(b, n, 6, 7); }
