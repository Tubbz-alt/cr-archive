165     /*
166      * fields access methods
167      */
168     ///////////////////////////////////////////////////////
169     static final int CH(byte[] b, int n) {
170         return Byte.toUnsignedInt(b[n]);
171     }
172 
173     static final int SH(byte[] b, int n) {
174         return Byte.toUnsignedInt(b[n]) | (Byte.toUnsignedInt(b[n + 1]) << 8);
175     }
176 
177     static final long LG(byte[] b, int n) {
178         return ((SH(b, n)) | (SH(b, n + 2) << 16)) & 0xffffffffL;
179     }
180 
181     static final long LL(byte[] b, int n) {
182         return (LG(b, n)) | (LG(b, n + 4) << 32);
183     }
184 
185     static long getSig(byte[] b, int n) { return LG(b, n); }
186 
187     private static boolean pkSigAt(byte[] b, int n, int b1, int b2) {
188         return b[n] == 'P' & b[n + 1] == 'K' & b[n + 2] == b1 & b[n + 3] == b2;
189     }
190 
191     static boolean cenSigAt(byte[] b, int n) { return pkSigAt(b, n, 1, 2); }
192     static boolean locSigAt(byte[] b, int n) { return pkSigAt(b, n, 3, 4); }
193     static boolean endSigAt(byte[] b, int n) { return pkSigAt(b, n, 5, 6); }
194     static boolean extSigAt(byte[] b, int n) { return pkSigAt(b, n, 7, 8); }
195     static boolean end64SigAt(byte[] b, int n) { return pkSigAt(b, n, 6, 6); }
196     static boolean locator64SigAt(byte[] b, int n) { return pkSigAt(b, n, 6, 7); }
