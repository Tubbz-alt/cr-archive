26 package sun.security.mscapi;
27 
28 import java.security.PrivateKey;
29 
30 /**
31  * The handle for a private key using the Microsoft Crypto API.
32  *
33  * @author Stanley Man-Kit Ho
34  * @since 1.6
35  */
36 class CPrivateKey extends CKey implements PrivateKey {
37 
38     private static final long serialVersionUID = 8113152807912338063L;
39 
40     private CPrivateKey(String alg, NativeHandles handles, int keyLength) {
41         super(alg, handles, keyLength);
42     }
43 
44     // Called by native code inside security.cpp
45     static CPrivateKey of(
46             String alg, long hCryptProv, long hCryptKey, int keyLength) {
47         return of(alg, new NativeHandles(hCryptProv, hCryptKey), keyLength);
48     }
49 
50     public static CPrivateKey of(String alg, NativeHandles handles, int keyLength) {
51         return new CPrivateKey(alg, handles, keyLength);
52     }
53 
54     // this key does not support encoding
55     public String getFormat()     {
56         return null;
57     }
58 
59     // this key does not support encoding
60     public byte[] getEncoded() {
61         return null;
62     }
63 
64     public String toString() {
65         if (handles.hCryptKey != 0) {
66             return algorithm + "PrivateKey [size=" + keyLength + " bits, type=" +
67                     getKeyType(handles.hCryptKey) + ", container=" +
68                     getContainerName(handles.hCryptProv) + "]";
69         } else {
70             return algorithm + "PrivateKey [size=" + keyLength + " bits, type=CNG]";
71         }
72     }
73 
74     // This class is not serializable
75     private void writeObject(java.io.ObjectOutputStream out)
76             throws java.io.IOException {
77         throw new java.io.NotSerializableException();
78     }
79 }
