26 package sun.security.mscapi;
27 
28 import java.security.PrivateKey;
29 
30 /**
31  * The handle for a private key using the Microsoft Crypto API.
32  *
33  * @author Stanley Man-Kit Ho
34  * @since 1.6
35  */
36 class CPrivateKey extends CKey implements PrivateKey {
37 
38     private static final long serialVersionUID = 8113152807912338063L;
39 
40     private CPrivateKey(String alg, long hCryptProv, long hCryptKey, int keyLength) {
41         super(alg, hCryptProv, hCryptKey, keyLength);
42     }
43 
44     public static CPrivateKey of(String alg, long hCryptProv, long hCryptKey, int keyLength) {
45         return new CPrivateKey(alg, hCryptProv, hCryptKey, keyLength);
46     }
47 
48     // this key does not support encoding
49     public String getFormat()     {
50         return null;
51     }
52 
53     // this key does not support encoding
54     public byte[] getEncoded() {
55         return null;
56     }
57 
58     public String toString() {
59         return algorithm + "PrivateKey [size=" + keyLength + " bits, type=" +
60             getKeyType(handles.hCryptKey) + ", container=" +
61             getContainerName(handles.hCryptProv) + "]";
62     }
63 
64     // This class is not serializable
65     private void writeObject(java.io.ObjectOutputStream out)
66             throws java.io.IOException {
67         throw new java.io.NotSerializableException();
68     }
69 }
