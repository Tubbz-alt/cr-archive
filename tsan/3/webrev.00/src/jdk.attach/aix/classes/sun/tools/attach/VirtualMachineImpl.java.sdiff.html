262     // On Aix a simple handshake is used to start the attach mechanism
263     // if not already started. The client creates a .attach_pid<pid> file in the
264     // target VM's working directory (or temp directory), and the SIGQUIT handler
265     // checks for the file.
266     private File createAttachFile(int pid) throws IOException {
267         String fn = ".attach_pid" + pid;
268         String path = "/proc/" + pid + "/cwd/" + fn;
269         File f = new File(path);
270         try {
271             f = f.getCanonicalFile();
272             f.createNewFile();
273         } catch (IOException x) {
274             f = new File(tmpdir, fn);
275             f.createNewFile();
276         }
277         return f;
278     }
