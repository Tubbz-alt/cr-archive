26 package jdk.incubator.jpackage.internal;
27 
28 import java.io.File;
29 import java.util.Collection;
30 import java.util.LinkedHashSet;
31 import java.util.Set;
32 
33 /**
34  * RelativeFileSet
35  *
36  * A class encapsulating a directory and a set of files within it.
37  */
38 class RelativeFileSet {
39 
40     private File basedir;
41     private Set<String> files = new LinkedHashSet<>();
42 
43     RelativeFileSet(File base, Collection<File> files) {
44         basedir = base;
45         String baseAbsolute = basedir.getAbsolutePath();
46         for (File f: files) {
47             String absolute = f.getAbsolutePath();
48             if (!absolute.startsWith(baseAbsolute)) {
49                 throw new RuntimeException("File " + f.getAbsolutePath() +
50                         " does not belong to " + baseAbsolute);
51             }
52             if (!absolute.equals(baseAbsolute)) {
53                     // possible in jpackage case
54                 this.files.add(absolute.substring(baseAbsolute.length()+1));
55             }
56         }
57     }
58 
59     RelativeFileSet(File base, Set<File> files) {
60         this(base, (Collection<File>) files);
61     }
62 
63     File getBaseDirectory() {
64         return basedir;
65     }
66 
67     Set<String> getIncludedFiles() {
68         return files;
69     }
70 
71     @Override
72     public String toString() {
73         if (files.size() ==  1) {
74             return "" + basedir + File.pathSeparator + files;
75         }
76         return "RelativeFileSet {basedir:" + basedir
77                 + ", files: {" + files + "}";
78     }
79 
80 }
