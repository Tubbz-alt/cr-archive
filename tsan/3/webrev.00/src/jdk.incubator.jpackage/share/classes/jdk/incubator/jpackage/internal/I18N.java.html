25 package jdk.incubator.jpackage.internal;
26 
27 import java.util.ResourceBundle;
28 
29 class I18N {
30 
31     static String getString(String key) {
32         if (PLATFORM.containsKey(key)) {
33             return PLATFORM.getString(key);
34         }
35         return SHARED.getString(key);
36     }
37 
38     private static final ResourceBundle SHARED = ResourceBundle.getBundle(
39             "jdk.incubator.jpackage.internal.resources.MainResources");
40 
41     private static final ResourceBundle PLATFORM;
42 
43     static {
44         if (Platform.isLinux()) {
45             PLATFORM = ResourceBundle.getBundle(
46                     "jdk.incubator.jpackage.internal.resources.LinuxResources");
47         } else if (Platform.isWindows()) {
48             PLATFORM = ResourceBundle.getBundle(
49                     "jdk.incubator.jpackage.internal.resources.WinResources");
50         } else if (Platform.isMac()) {
51             PLATFORM = ResourceBundle.getBundle(
52                     "jdk.incubator.jpackage.internal.resources.MacResources");
53         } else {
54             throw new IllegalStateException("Unknwon platform");
55         }
56     }
57 }
