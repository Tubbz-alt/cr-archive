26 package jdk.incubator.jpackage.internal;
27 
28 import java.text.MessageFormat;
29 import java.util.ResourceBundle;
30 
31 public class PackagerException extends Exception {
32     private static final long serialVersionUID = 1L;
33     private static final ResourceBundle bundle = ResourceBundle.getBundle(
34             "jdk.incubator.jpackage.internal.resources.MainResources");
35 
36     public PackagerException(Throwable cause) {
37         super(cause);
38     }
39 
40     public PackagerException(String key, Throwable cause) {
41         super(bundle.getString(key), cause);
42     }
43 
44     public PackagerException(String key) {
45         super(bundle.getString(key));
46     }
47 
48     public PackagerException(String key, String ... arguments) {
49         super(MessageFormat.format(
50                 bundle.getString(key), (Object[]) arguments));
51     }
52 
53     public PackagerException(
54             Throwable cause, String key, String ... arguments) {
55         super(MessageFormat.format(bundle.getString(key),
56                 (Object[]) arguments), cause);
57     }
58 
59 }
