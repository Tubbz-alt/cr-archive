26 package jdk.incubator.jpackage.internal;
27 
28 import java.io.PrintWriter;
29 import java.util.spi.ToolProvider;
30 
31 /**
32  * JPackageToolProvider
33  *
34  * This is the ToolProvider implementation exported
35  * to java.util.spi.ToolProvider and ultimately javax.tools.ToolProvider
36  */
37 public class JPackageToolProvider implements ToolProvider {
38 
39     public String name() {
40         return "jpackage";
41     }
42 
43     public synchronized int run(
44             PrintWriter out, PrintWriter err, String... args) {
45         try {
46             return new jdk.incubator.jpackage.main.Main().execute(out, err, args);
47         } catch (RuntimeException re) {
48             Log.error(re.getMessage());
49             Log.verbose(re);
50             return 1;
51         }
52     }
53 }
