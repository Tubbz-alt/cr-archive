 25 package jdk.incubator.jpackage.internal;
 26 
 27 import java.io.BufferedReader;
 28 import java.io.IOException;
 29 import java.io.InputStreamReader;
 30 import java.util.List;
 31 import java.util.function.Consumer;
 32 import java.util.function.Supplier;
 33 import java.util.stream.Collectors;
 34 import java.util.stream.Stream;
 35 
 36 final public class Executor {
 37 
 38     Executor() {
 39     }
 40 
 41     Executor setOutputConsumer(Consumer<Stream<String>> v) {
 42         outputConsumer = v;
 43         return this;
 44     }
 45 
 46     Executor saveOutput(boolean v) {
 47         saveOutput = v;
 48         return this;
 49     }
 50 
 51     Executor setWaitBeforeOutput(boolean v) {
 52         waitBeforeOutput = v;
 53         return this;
 54     }
 55 
 56     Executor setProcessBuilder(ProcessBuilder v) {
 57         pb = v;
 58         return this;
 59     }
 60 
 61     Executor setCommandLine(String... cmdline) {
 62         return setProcessBuilder(new ProcessBuilder(cmdline));
 63     }
 64 
 65     List<String> getOutput() {
 66         return output;
 67     }
 68 
 69     Executor executeExpectSuccess() throws IOException {
 70         int ret = execute();
 71         if (0 != ret) {
 72             throw new IOException(
 73                     String.format("Command %s exited with %d code",
 74                             createLogMessage(pb), ret));
 75         }
 76         return this;
 77     }
 78 
 79     int execute() throws IOException {
 80         output = null;
 81 
 82         boolean needProcessOutput = outputConsumer != null || Log.isVerbose() || saveOutput;
 83         if (needProcessOutput) {
 84             pb.redirectErrorStream(true);
 85         } else {
 86             // We are not going to read process output, so need to notify
 87             // ProcessBuilder about this. Otherwise some processes might just
 88             // hang up (`ldconfig -p`).
 89             pb.redirectError(ProcessBuilder.Redirect.DISCARD);
 90             pb.redirectOutput(ProcessBuilder.Redirect.DISCARD);
 91         }
 92 
 93         Log.verbose(String.format("Running %s", createLogMessage(pb)));
 94         Process p = pb.start();
 95 
 96         int code = 0;
 97         if (waitBeforeOutput) {
 98             try {
 99                 code = p.waitFor();
100             } catch (InterruptedException ex) {
101                 Log.verbose(ex);
102                 throw new RuntimeException(ex);
103             }
104         }
105 
106         if (needProcessOutput) {
107             try (var br = new BufferedReader(new InputStreamReader(
108                     p.getInputStream()))) {
109                 final List<String> savedOutput;
110                 // Need to save output if explicitely requested (saveOutput=true) or
111                 // if will be used used by multiple consumers
112                 if ((outputConsumer != null && Log.isVerbose()) || saveOutput) {
113                     savedOutput = br.lines().collect(Collectors.toList());
114                     if (saveOutput) {
115                         output = savedOutput;
116                     }
117                 } else {
118                     savedOutput = null;
119                 }
120 
121                 Supplier<Stream<String>> outputStream = () -> {
122                     if (savedOutput != null) {
123                         return savedOutput.stream();
124                     }
125                     return br.lines();
126                 };
127 
128                 if (Log.isVerbose()) {
129                     outputStream.get().forEach(Log::verbose);
130                 }
131 
132                 if (outputConsumer != null) {
133                     outputConsumer.accept(outputStream.get());
134                 }
135 
136                 if (savedOutput == null) {
137                     // For some processes on Linux if the output stream
138                     // of the process is opened but not consumed, the process
139                     // would exit with code 141.
140                     // It turned out that reading just a single line of process
141                     // output fixes the problem, but let's process
142                     // all of the output, just in case.
143                     br.lines().forEach(x -> {});
144                 }
145             }
146         }
147 
148         try {
149             if (!waitBeforeOutput) {
150                 code = p.waitFor();
151             }
152             return code;
153         } catch (InterruptedException ex) {
154             Log.verbose(ex);
155             throw new RuntimeException(ex);
156         }
157     }
158 
159     static Executor of(String... cmdline) {
160         return new Executor().setCommandLine(cmdline);
161     }
162 
163     static Executor of(ProcessBuilder pb) {
164         return new Executor().setProcessBuilder(pb);
165     }
166 
167     private static String createLogMessage(ProcessBuilder pb) {
168         StringBuilder sb = new StringBuilder();
169         sb.append(String.format("%s", pb.command()));
170         if (pb.directory() != null) {
171             sb.append(String.format("in %s", pb.directory().getAbsolutePath()));
172         }
173         return sb.toString();
174     }
175 
176     private ProcessBuilder pb;
177     private boolean saveOutput;
178     private boolean waitBeforeOutput;
179     private List<String> output;
180     private Consumer<Stream<String>> outputConsumer;
181 }
