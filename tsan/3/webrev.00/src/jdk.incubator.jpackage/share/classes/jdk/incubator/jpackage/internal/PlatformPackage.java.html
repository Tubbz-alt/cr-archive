25 package jdk.incubator.jpackage.internal;
26 
27 import java.nio.file.Path;
28 
29 /**
30  *
31  * Platform package of an application.
32  */
33 interface PlatformPackage {
34     /**
35      * Platform-specific package name.
36      */
37     String name();
38 
39     /**
40      * Root directory where sources for packaging tool should be stored
41      */
42     Path sourceRoot();
43 
44     /**
45      * Source application layout from which to build the package.
46      */
47     ApplicationLayout sourceApplicationLayout();
48 
49     /**
50      * Application layout of the installed package.
51      */
52     ApplicationLayout installedApplicationLayout();
53 }
