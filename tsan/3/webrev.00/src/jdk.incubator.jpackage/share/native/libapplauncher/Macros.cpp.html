37 void Macros::Initialize() {
38     Package& package = Package::GetInstance();
39     Macros& macros = Macros::GetInstance();
40 
41     // Public macros.
42     macros.AddMacro(_T("$ROOTDIR"), package.GetPackageRootDirectory());
43     macros.AddMacro(_T("$APPDIR"), package.GetPackageAppDirectory());
44     macros.AddMacro(_T("$BINDIR"), package.GetPackageLauncherDirectory());
45 }
46 
47 Macros& Macros::GetInstance() {
48     static Macros instance;
49     return instance;
50 }
51 
52 TString Macros::ExpandMacros(TString Value) {
53     TString result = Value;
54 
55     for (std::map<TString, TString>::iterator iterator = FData.begin();
56         iterator != FData.end();
57         iterator++) {
58 
59         TString name = iterator->first;
60 
61         if (Value.find(name) != TString::npos) {
62             TString lvalue = iterator->second;
63             result = Helpers::ReplaceString(Value, name, lvalue);
64             result = ExpandMacros(result);
65             break;
66         }
67     }
68 
69     return result;
70 }
71 
72 void Macros::AddMacro(TString Key, TString Value) {
73     FData.insert(std::map<TString, TString>::value_type(Key, Value));
74 }
