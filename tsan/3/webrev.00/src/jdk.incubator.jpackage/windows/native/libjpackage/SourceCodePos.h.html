31 //
32 // Position in source code.
33 //
34 
35 struct SourceCodePos
36 {
37     SourceCodePos(const char* fl, const char* fnc, int l):
38                                                 file(fl), func(fnc), lno(l)
39     {
40     }
41 
42     const char* file;
43     const char* func;
44     int lno;
45 };
46 
47 
48 // Initializes SourceCodePos instance with the
49 // information from the point of calling.
50 #define JP_SOURCE_CODE_POS SourceCodePos(__FILE__, __FUNCTION__, __LINE__)
