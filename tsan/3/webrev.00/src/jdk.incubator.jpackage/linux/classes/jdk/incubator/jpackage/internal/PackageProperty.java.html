26 package jdk.incubator.jpackage.internal;
27 
28 import java.text.MessageFormat;
29 
30 final class PackageProperty {
31     /**
32      * Constructor
33      *
34      * @param name property name
35      * @param expectedValue expected property value
36      * @param substString substitution string to be placed in resource file to
37      * be replaced with the expected property value by jpackage at package build
38      * time
39      * @param customResource name of custom resource from resource directory in
40      * which this package property can be set
41      */
42     PackageProperty(String name, String expectedValue, String substString,
43             String customResource) {
44         this.name = name;
45         this.expectedValue = expectedValue;
46         this.substString = substString;
47         this.customResource = customResource;
48     }
49 
50     ConfigException verifyValue(String actualValue) {
51         if (expectedValue.equals(actualValue)) {
52             return null;
53         }
54 
55         final String advice;
56         if (substString != null) {
57             advice = MessageFormat.format(I18N.getString(
58                     "error.unexpected-package-property.advice"), substString,
59                     actualValue, name, customResource);
60         } else {
61             advice = MessageFormat.format(I18N.getString(
62                     "error.unexpected-default-package-property.advice"), name,
63                     customResource);
64         }
65 
66         return new ConfigException(MessageFormat.format(I18N.getString(
67                 "error.unexpected-package-property"), name,
68                 expectedValue, actualValue, customResource, substString), advice);
69     }
70 
71     final String name;
72     private final String expectedValue;
73     private final String substString;
74     private final String customResource;
75 }
