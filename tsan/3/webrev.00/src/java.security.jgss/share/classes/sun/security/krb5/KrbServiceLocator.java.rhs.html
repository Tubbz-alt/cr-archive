 42 /**
 43  * This class discovers the location of Kerberos services by querying DNS,
 44  * as defined in RFC 4120.
 45  *
 46  * @author Seema Malkani
 47  * @since 1.7
 48  */
 49 
 50 class KrbServiceLocator {
 51 
 52     private static final String SRV_RR = "SRV";
 53     private static final String[] SRV_RR_ATTR = new String[] {SRV_RR};
 54 
 55     private static final String SRV_TXT = "TXT";
 56     private static final String[] SRV_TXT_ATTR = new String[] {SRV_TXT};
 57 
 58     private static final Random random = new Random();
 59 


 60     private KrbServiceLocator() {
 61     }
 62 
 63     /**
 64      * Locates the KERBEROS service for a given domain.
 65      * Queries DNS for a list of KERBEROS Service Text Records (TXT) for a
 66      * given domain name.
 67      * Information on the mapping of DNS hostnames and domain names
 68      * to Kerberos realms is stored using DNS TXT records
 69      *
 70      * @param realmName A string realm name.
 71      * @return An ordered list of hostports for the Kerberos service or null if
 72      *          the service has not been located.
 73      */
 74     static String[] getKerberosService(String realmName) {
 75 
 76         // search realm in SRV TXT records
 77         String dnsUrl = "dns:///_kerberos." + realmName;
 78         String[] records = null;
 79         try {
 80             // Create the DNS context using NamingManager rather than using
 81             // the initial context constructor. This avoids having the initial
 82             // context constructor call itself (when processing the URL
 83             // argument in the getAttributes call).
 84             Context ctx = NamingManager.getURLContext("dns", new Hashtable<>(0));
 85             if (!(ctx instanceof DirContext)) {
 86                 return null; // cannot create a DNS context
 87             }
 88             Attributes attrs = null;
 89             try {
 90                 // both connect and accept are needed since DNS is thru UDP
 91                 attrs = AccessController.doPrivileged(
 92                         (PrivilegedExceptionAction<Attributes>)
 93                                 () -> ((DirContext)ctx).getAttributes(
 94                                         dnsUrl, SRV_TXT_ATTR),
 95                         null,
 96                         new java.net.SocketPermission("*", "connect,accept"));
 97             } catch (PrivilegedActionException e) {
 98                 throw (NamingException)e.getCause();
 99             }
100             Attribute attr;
101 
102             if (attrs != null && ((attr = attrs.get(SRV_TXT)) != null)) {
103                 int numValues = attr.size();
104                 int numRecords = 0;
105                 String[] txtRecords = new String[numValues];
106 
107                 // gather the text records
108                 int i = 0;
109                 int j = 0;
110                 while (i < numValues) {
111                     try {
112                         txtRecords[j] = (String)attr.get(i);
113                         j++;
114                     } catch (Exception e) {
115                         // ignore bad value
116                     }
117                     i++;
118                 }
119                 numRecords = j;
120 
121                 // trim
122                 if (numRecords < numValues) {
123                     String[] trimmed = new String[numRecords];
124                     System.arraycopy(txtRecords, 0, trimmed, 0, numRecords);
125                     records = trimmed;
126                 } else {
127                     records = txtRecords;
128                 }
129             }
130         } catch (NamingException e) {
131             // ignore
132         }
133         return records;
134     }
135 
136     /**
137      * Locates the KERBEROS service for a given domain.
138      * Queries DNS for a list of KERBEROS Service Location Records (SRV) for a
139      * given domain name.
140      *
141      * @param realmName A string realm name.
142      * @param protocol the protocol string, can be "_udp" or "_tcp"
143      * @return An ordered list of hostports for the Kerberos service or null if
144      *          the service has not been located.
145      */
146     static String[] getKerberosService(String realmName, String protocol) {
147 
148         String dnsUrl = "dns:///_kerberos." + protocol + "." + realmName;
149         String[] hostports = null;
150 
151         try {
152             // Create the DNS context using NamingManager rather than using
153             // the initial context constructor. This avoids having the initial
154             // context constructor call itself (when processing the URL
155             // argument in the getAttributes call).
156             Context ctx = NamingManager.getURLContext("dns", new Hashtable<>(0));
157             if (!(ctx instanceof DirContext)) {
158                 return null; // cannot create a DNS context
159             }
160 
161             Attributes attrs = null;
162             try {
163                 // both connect and accept are needed since DNS is thru UDP
164                 attrs = AccessController.doPrivileged(
165                         (PrivilegedExceptionAction<Attributes>)
166                                 () -> ((DirContext)ctx).getAttributes(
167                                         dnsUrl, SRV_RR_ATTR),
168                         null,
169                         new java.net.SocketPermission("*", "connect,accept"));
170             } catch (PrivilegedActionException e) {
171                 throw (NamingException)e.getCause();
172             }
173 
174             Attribute attr;
175 
176             if (attrs != null && ((attr = attrs.get(SRV_RR)) != null)) {
177                 int numValues = attr.size();
178                 int numRecords = 0;
179                 SrvRecord[] srvRecords = new SrvRecord[numValues];
180 
181                 // create the service records
182                 int i = 0;
183                 int j = 0;
184                 while (i < numValues) {
185                     try {
186                         srvRecords[j] = new SrvRecord((String) attr.get(i));
187                         j++;
188                     } catch (Exception e) {
189                         // ignore bad value
190                     }
191                     i++;
192                 }
193                 numRecords = j;
194 
195                 // trim
196                 if (numRecords < numValues) {
197                     SrvRecord[] trimmed = new SrvRecord[numRecords];
198                     System.arraycopy(srvRecords, 0, trimmed, 0, numRecords);
199                     srvRecords = trimmed;
200                 }
201 
202                 // Sort the service records in ascending order of their
203                 // priority value. For records with equal priority, move
204                 // those with weight 0 to the top of the list.
205                 if (numRecords > 1) {
206                     Arrays.sort(srvRecords);
207                 }
208 
209                 // extract the host and port number from each service record
210                 hostports = extractHostports(srvRecords);
211             }
212         } catch (NamingException e) {
213             // e.printStackTrace();
214             // ignore
215         }
216         return hostports;
217     }
218 
219     /**
220      * Extract hosts and port numbers from a list of SRV records.
221      * An array of hostports is returned or null if none were found.
222      */
223     private static String[] extractHostports(SrvRecord[] srvRecords) {
224         String[] hostports = null;
225 
226         int head = 0;
227         int tail = 0;
228         int sublistLength = 0;
229         int k = 0;
230         for (int i = 0; i < srvRecords.length; i++) {
231             if (hostports == null) {
232                 hostports = new String[srvRecords.length];
233             }
234             // find the head and tail of the list of records having the same
235             // priority value.
236             head = i;
237             while (i < srvRecords.length - 1 &&
238                 srvRecords[i].priority == srvRecords[i + 1].priority) {
239                 i++;
240             }
241             tail = i;
242 
243             // select hostports from the sublist
244             sublistLength = (tail - head) + 1;
245             for (int j = 0; j < sublistLength; j++) {
246                 hostports[k++] = selectHostport(srvRecords, head, tail);
247             }
248         }
249         return hostports;
250     }
251 
252     /*
253      * Randomly select a service record in the range [head, tail] and return
254      * its hostport value. Follows the algorithm in RFC 2782.
255      */
256     private static String selectHostport(SrvRecord[] srvRecords, int head,
257             int tail) {
258         if (head == tail) {
259             return srvRecords[head].hostport;
260         }
261 
262         // compute the running sum for records between head and tail
263         int sum = 0;
264         for (int i = head; i <= tail; i++) {
265             if (srvRecords[i] != null) {
266                 sum += srvRecords[i].weight;
267                 srvRecords[i].sum = sum;
268             }
269         }
270         String hostport = null;
271 
272         // If all records have zero weight, select first available one;
273         // otherwise, randomly select a record according to its weight
274         int target = (sum == 0 ? 0 : random.nextInt(sum + 1));
275         for (int i = head; i <= tail; i++) {
276             if (srvRecords[i] != null && srvRecords[i].sum >= target) {
277                 hostport = srvRecords[i].hostport;
278                 srvRecords[i] = null; // make this record unavailable
279                 break;
280             }
281         }
282         return hostport;
283     }

285 /**
286  * This class holds a DNS service (SRV) record.
287  * See http://www.ietf.org/rfc/rfc2782.txt
288  */
289 
290 static class SrvRecord implements Comparable<SrvRecord> {
291 
292     int priority;
293     int weight;
294     int sum;
295     String hostport;
296 
297     /**
298      * Creates a service record object from a string record.
299      * DNS supplies the string record in the following format:
300      * <pre>
301      *          <Priority> " " <Weight> " " <Port> " " <Host>
302      * </pre>
303      */
304     SrvRecord(String srvRecord) throws Exception {
305         StringTokenizer tokenizer = new StringTokenizer(srvRecord, " ");
306         String port;
307 
308         if (tokenizer.countTokens() == 4) {
309             priority = Integer.parseInt(tokenizer.nextToken());
310             weight = Integer.parseInt(tokenizer.nextToken());
311             port = tokenizer.nextToken();
312             hostport = tokenizer.nextToken() + ":" + port;
313         } else {
314             throw new IllegalArgumentException();
315         }
316     }
317 
318     /*
319      * Sort records in ascending order of priority value. For records with
320      * equal priority move those with weight 0 to the top of the list.
321      */
322     public int compareTo(SrvRecord that) {
323         if (priority > that.priority) {
324             return 1; // this > that
325         } else if (priority < that.priority) {
326             return -1; // this < that
327         } else if (weight == 0 && that.weight != 0) {
328             return -1; // this < that
329         } else if (weight != 0 && that.weight == 0) {
330             return 1; // this > that
331         } else {
332             return 0; // this == that
333         }
334     }
335 }
