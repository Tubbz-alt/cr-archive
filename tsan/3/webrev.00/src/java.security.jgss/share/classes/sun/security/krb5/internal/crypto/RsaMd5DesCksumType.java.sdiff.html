 93         System.arraycopy(conf, 0, cksum, 0, confounderSize());
 94         System.arraycopy(mdc_cksum, 0, cksum, confounderSize(),
 95                          cksumSize() - confounderSize());

113         byte[] enc_cksum = new byte[cksum.length];
