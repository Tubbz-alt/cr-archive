 26 package sun.security.jgss.krb5;
 27 
 28 import org.ietf.jgss.*;

 29 import sun.security.jgss.spi.*;
 30 import java.util.Date;




 31 import sun.security.krb5.internal.Ticket;
 32 


 33 /**
 34  * Implements the krb5 proxy credential element used in constrained
 35  * delegation. It is used in both impersonation (where there is no Kerberos 5
 36  * communication between the middle server and the client) and normal
 37  * constrained delegation (where there is, but client has not called
 38  * requestCredDeleg(true)).
 39  * @since 1.8
 40  */
 41 
 42 public class Krb5ProxyCredential
 43     implements Krb5CredElement {
 44 
 45     public final Krb5InitCredential self;   // the middle server
 46     private final Krb5NameElement client;     // the client
 47 
 48     // The ticket with cname=client and sname=self. This can be a normal
 49     // service ticket or an S4U2self ticket.
 50     public final Ticket tkt;
 51 
 52     Krb5ProxyCredential(Krb5InitCredential self, Krb5NameElement client,
 53             Ticket tkt) {
 54         this.self = self;
 55         this.tkt = tkt;
 56         this.client = client;
 57     }
 58 
 59     // The client name behind the proxy
 60     @Override
 61     public final Krb5NameElement getName() throws GSSException {
 62         return client;
 63     }
 64 
 65     @Override
 66     public int getInitLifetime() throws GSSException {
 67         // endTime of tkt is not used by KDC, and it's also not
 68         // available in the case of kerberos constr deleg
 69         return self.getInitLifetime();
 70     }
 71 
 72     @Override
 73     public int getAcceptLifetime() throws GSSException {
 74         return 0;
 75     }
 76 
 77     @Override
 78     public boolean isInitiatorCredential() throws GSSException {
 79         return true;
 80     }
 81 
 82     @Override
 83     public boolean isAcceptorCredential() throws GSSException {
 84         return false;
 85     }
 86 
 87     @Override
 88     public final Oid getMechanism() {
 89         return Krb5MechFactory.GSS_KRB5_MECH_OID;
 90     }
 91 
 92     @Override
 93     public final java.security.Provider getProvider() {
 94         return Krb5MechFactory.PROVIDER;
 95     }
 96 
 97     @Override
 98     public void dispose() throws GSSException {
 99         try {
100             self.destroy();
101         } catch (javax.security.auth.DestroyFailedException e) {
102             GSSException gssException =
103                 new GSSException(GSSException.FAILURE, -1,
104                  "Could not destroy credentials - " + e.getMessage());
105             gssException.initCause(e);
106         }
107     }
108 
109     @Override
110     public GSSCredentialSpi impersonate(GSSNameSpi name) throws GSSException {
111         // Cannot impersonate multiple levels without the impersonatee's TGT.
112         throw new GSSException(GSSException.FAILURE, -1,
113                 "Only an initiate credentials can impersonate");
114     }




















115 }
