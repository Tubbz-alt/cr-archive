 26 package sun.security.krb5.internal.crypto;
 27 
 28 import sun.security.krb5.Checksum;
 29 import sun.security.krb5.KrbCryptoException;
 30 import sun.security.krb5.internal.*;


 31 import java.security.GeneralSecurityException;
 32 
 33 public class HmacSha1Des3KdCksumType extends CksumType {
 34 
 35     public HmacSha1Des3KdCksumType() {
 36     }
 37 
 38     public int confounderSize() {
 39         return 8;
 40     }
 41 
 42     public int cksumType() {
 43         return Checksum.CKSUMTYPE_HMAC_SHA1_DES3_KD;
 44     }
 45 
 46     public boolean isKeyed() {
 47         return true;
 48     }
 49 
 50     public int cksumSize() {
 51         return 20;  // bytes
 52     }
 53 
 54     public int keyType() {
 55         return Krb5.KEYTYPE_DES3;
 56     }
 57 
 58     public int keySize() {
 59         return 24;   // bytes
 60     }
 61 




 62     /**
 63      * Calculates keyed checksum.
 64      * @param data the data used to generate the checksum.
 65      * @param size length of the data.
 66      * @param key the key used to encrypt the checksum.
 67      * @return keyed checksum.
 68      */
 69     public byte[] calculateChecksum(byte[] data, int size, byte[] key,
 70         int usage) throws KrbCryptoException {
 71 
 72          try {
 73              return Des3.calculateChecksum(key, usage, data, 0, size);
 74          } catch (GeneralSecurityException e) {
 75              KrbCryptoException ke = new KrbCryptoException(e.getMessage());
 76              ke.initCause(e);
 77              throw ke;
 78          }
 79     }
 80 
 81     /**
 82      * Verifies keyed checksum.
 83      * @param data the data.
 84      * @param size the length of data.
 85      * @param key the key used to encrypt the checksum.
 86      * @param checksum the checksum.
 87      * @return true if verification is successful.
 88      */
 89     public boolean verifyChecksum(byte[] data, int size,
 90         byte[] key, byte[] checksum, int usage) throws KrbCryptoException {
 91 
 92          try {
 93              byte[] newCksum = Des3.calculateChecksum(key, usage,
 94                  data, 0, size);
 95 
 96              return isChecksumEqual(checksum, newCksum);
 97          } catch (GeneralSecurityException e) {
 98              KrbCryptoException ke = new KrbCryptoException(e.getMessage());
 99              ke.initCause(e);
100              throw ke;
101          }
102      }
103 }
