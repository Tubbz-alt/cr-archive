 32 package sun.security.krb5.internal.ktab;
 33 
 34 import sun.security.krb5.internal.*;
 35 import sun.security.krb5.internal.util.KrbDataOutputStream;
 36 import java.io.IOException;
 37 import java.io.FileOutputStream;
 38 import java.io.OutputStream;
 39 
 40 import static java.nio.charset.StandardCharsets.ISO_8859_1;
 41 
 42 /**
 43  * This class implements a buffered input stream. It is used for parsing key table
 44  * data to memory.
 45  *
 46  * @author Yanni Zhang
 47  *
 48  */
 49 public class KeyTabOutputStream extends KrbDataOutputStream implements KeyTabConstants {
 50     private KeyTabEntry entry;
 51     private int keyType;
 52     private byte[] keyValue;
 53     public int version;
 54 
 55     public KeyTabOutputStream(OutputStream os) {
 56         super(os);
 57     }
 58 
 59     public void writeVersion(int num) throws IOException {
 60         version = num;
 61         write16(num);           //we use the standard version.
 62     }
 63 
 64     public void writeEntry(KeyTabEntry entry) throws IOException {
 65         write32(entry.entryLength());
 66         String[] serviceNames =  entry.service.getNameStrings();
 67         int comp_num = serviceNames.length;
 68         if (version == KRB5_KT_VNO_1) {
 69             write16(comp_num + 1);
 70         }
 71         else write16(comp_num);
 72 
 73         byte[] realm = entry.service.getRealmString().getBytes(ISO_8859_1);
 74         write16(realm.length);
 75         write(realm);
 76 
 77         for (int i = 0; i < comp_num; i++) {
 78             byte[] serviceName = serviceNames[i].getBytes(ISO_8859_1);
 79             write16(serviceName.length);
 80             write(serviceName);
 81         }
 82 
 83         write32(entry.service.getNameType());
 84         //time is long, but we only use 4 bytes to store the data.
 85         write32((int)(entry.timestamp.getTime()/1000));
 86 
 87         // the key version might be a 32 bit extended number.
 88         write8(entry.keyVersion % 256 );
 89         write16(entry.keyType);
 90         write16(entry.keyblock.length);
 91         write(entry.keyblock);
 92 
 93         // if the key version isn't smaller than 256, it could be saved as
 94         // extension key version number in 4 bytes. The nonzero extension
 95         // key version number will be trusted. However, it isn't standardized
 96         // yet, we won't support it.
 97         // if (entry.keyVersion >= 256) {
 98         //    write32(entry.keyVersion);
 99         //}
100     }
101 }
