 26 package sun.security.krb5.internal.crypto;
 27 
 28 import sun.security.krb5.Checksum;
 29 import sun.security.krb5.KrbCryptoException;
 30 import sun.security.krb5.internal.*;
 31 import javax.crypto.spec.DESKeySpec;
 32 import java.security.InvalidKeyException;
 33 import java.security.GeneralSecurityException;
 34 
 35 public class HmacSha1Des3KdCksumType extends CksumType {
 36 
 37     public HmacSha1Des3KdCksumType() {
 38     }
 39 
 40     public int confounderSize() {
 41         return 8;
 42     }
 43 
 44     public int cksumType() {
 45         return Checksum.CKSUMTYPE_HMAC_SHA1_DES3_KD;
 46     }
 47 
 48     public boolean isSafe() {
 49         return true;
 50     }
 51 
 52     public int cksumSize() {
 53         return 20;  // bytes
 54     }
 55 
 56     public int keyType() {
 57         return Krb5.KEYTYPE_DES3;
 58     }
 59 
 60     public int keySize() {
 61         return 24;   // bytes
 62     }
 63 
 64     public byte[] calculateChecksum(byte[] data, int size) {
 65         return null;
 66     }
 67 
 68     /**
 69      * Calculates keyed checksum.
 70      * @param data the data used to generate the checksum.
 71      * @param size length of the data.
 72      * @param key the key used to encrypt the checksum.
 73      * @return keyed checksum.
 74      */
 75     public byte[] calculateKeyedChecksum(byte[] data, int size, byte[] key,
 76         int usage) throws KrbCryptoException {
 77 
 78          try {
 79              return Des3.calculateChecksum(key, usage, data, 0, size);
 80          } catch (GeneralSecurityException e) {
 81              KrbCryptoException ke = new KrbCryptoException(e.getMessage());
 82              ke.initCause(e);
 83              throw ke;
 84          }
 85     }
 86 
 87     /**
 88      * Verifies keyed checksum.
 89      * @param data the data.
 90      * @param size the length of data.
 91      * @param key the key used to encrypt the checksum.
 92      * @param checksum the checksum.
 93      * @return true if verification is successful.
 94      */
 95     public boolean verifyKeyedChecksum(byte[] data, int size,
 96         byte[] key, byte[] checksum, int usage) throws KrbCryptoException {
 97 
 98          try {
 99              byte[] newCksum = Des3.calculateChecksum(key, usage,
100                  data, 0, size);
101 
102              return isChecksumEqual(checksum, newCksum);
103          } catch (GeneralSecurityException e) {
104              KrbCryptoException ke = new KrbCryptoException(e.getMessage());
105              ke.initCause(e);
106              throw ke;
107          }
108      }
109 }
