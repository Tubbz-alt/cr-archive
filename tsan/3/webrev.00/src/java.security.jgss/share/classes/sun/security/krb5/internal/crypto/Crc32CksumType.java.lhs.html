33 import sun.security.krb5.*;
34 import sun.security.krb5.internal.*;
35 import java.util.zip.CRC32;
36 
37 public class Crc32CksumType extends CksumType {
38 
39     public Crc32CksumType() {
40     }
41 
42     public int confounderSize() {
43         return 0;
44     }
45 
46     public int cksumType() {
47         return Checksum.CKSUMTYPE_CRC32;
48     }
49 
50     public boolean isSafe() {
51         return false;
52     }
53 
54     public int cksumSize() {
55         return 4;
56     }
57 
58     public int keyType() {
59         return Krb5.KEYTYPE_NULL;
60     }
61 
62     public int keySize() {
63         return 0;
64     }
65 
66     public byte[] calculateChecksum(byte[] data, int size) {

67         return crc32.byte2crc32sum_bytes(data, size);
68     }
69 
70     public byte[] calculateKeyedChecksum(byte[] data, int size,
71                                          byte[] key, int usage) {
72                                              return null;
73                                          }
74 
75     public boolean verifyKeyedChecksum(byte[] data, int size,
76                                        byte[] key, byte[] checksum, int usage) {
77         return false;
78     }
79 
80     public static byte[] int2quad(long input) {
81         byte[] output = new byte[4];
82         for (int i = 0; i < 4; i++) {
83             output[i] = (byte)((input >>> (i * 8)) & 0xff);
84         }
85         return output;
86     }
87 
88     public static long bytes2long(byte[] input) {
89         long result = 0;
90 
91         result |= (((long)input[0]) & 0xffL) << 24;
92         result |= (((long)input[1]) & 0xffL) << 16;
93         result |= (((long)input[2]) & 0xffL) << 8;
94         result |= (((long)input[3]) & 0xffL);
95         return result;
96     }
97 }
