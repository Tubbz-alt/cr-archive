 30 package sun.security.krb5.internal.crypto;
 31 
 32 import sun.security.krb5.Checksum;
 33 import sun.security.krb5.Confounder;
 34 import sun.security.krb5.KrbCryptoException;
 35 import sun.security.krb5.internal.*;
 36 import javax.crypto.spec.DESKeySpec;
 37 import java.security.MessageDigest;
 38 import java.security.InvalidKeyException;
 39 
 40 public final class RsaMd5DesCksumType extends CksumType {
 41 
 42     public RsaMd5DesCksumType() {
 43     }
 44 
 45     public int confounderSize() {
 46         return 8;
 47     }
 48 
 49     public int cksumType() {
 50         return Checksum.CKSUMTYPE_RSA_MD5_DES;
 51     }
 52 
 53     public boolean isKeyed() {
 54         return true;
 55     }
 56 
 57     public int cksumSize() {
 58         return 24;
 59     }
 60 
 61     public int keyType() {
 62         return Krb5.KEYTYPE_DES;
 63     }
 64 
 65     public int keySize() {
 66         return 8;
 67     }
 68 
 69     /**
 70      * Calculates keyed checksum.
 71      * @param data the data used to generate the checksum.
 72      * @param size length of the data.
 73      * @param key the key used to encrypt the checksum.
 74      * @return keyed checksum.
 75      *
 76      * @modified by Yanni Zhang, 12/08/99.
 77      */
 78     public byte[] calculateChecksum(byte[] data, int size, byte[] key,
 79         int usage) throws KrbCryptoException {
 80         //prepend confounder
 81         byte[] new_data = new byte[size + confounderSize()];
 82         byte[] conf = Confounder.bytes(confounderSize());
 83         System.arraycopy(conf, 0, new_data, 0, confounderSize());
 84         System.arraycopy(data, 0, new_data, confounderSize(), size);
 85 
 86         //calculate md5 cksum
 87         byte[] mdc_cksum = calculateRawChecksum(new_data, new_data.length);
 88         byte[] cksum = new byte[cksumSize()];
 89         System.arraycopy(conf, 0, cksum, 0, confounderSize());
 90         System.arraycopy(mdc_cksum, 0, cksum, confounderSize(),
 91                          cksumSize() - confounderSize());
 92 
 93         //compute modified key
 94         byte[] new_key = new byte[keySize()];
 95         System.arraycopy(key, 0, new_key, 0, key.length);
 96         for (int i = 0; i < new_key.length; i++)
 97         new_key[i] = (byte)(new_key[i] ^ 0xf0);
 98         //check for weak keys
 99         try {
100             if (DESKeySpec.isWeak(new_key, 0)) {
101                 new_key[7] = (byte)(new_key[7] ^ 0xF0);
102             }
103         } catch (InvalidKeyException ex) {
104             // swallow, since it should never happen
105         }
106         byte[] ivec = new byte[new_key.length];
107 
108         //des-cbc encrypt
109         byte[] enc_cksum = new byte[cksum.length];
110         Des.cbc_encrypt(cksum, enc_cksum, new_key, ivec, true);
111         return enc_cksum;
112     }
113 
114     /**
115      * Verifies keyed checksum.
116      * @param data the data.
117      * @param size the length of data.
118      * @param key the key used to encrypt the checksum.
119      * @param checksum the checksum.
120      * @return true if verification is successful.
121      *
122      * @modified by Yanni Zhang, 12/08/99.
123      */
124     public boolean verifyChecksum(byte[] data, int size,
125         byte[] key, byte[] checksum, int usage) throws KrbCryptoException {
126         //decrypt checksum
127         byte[] cksum = decryptKeyedChecksum(checksum, key);
128 
129         //prepend confounder
130         byte[] new_data = new byte[size + confounderSize()];
131         System.arraycopy(cksum, 0, new_data, 0, confounderSize());
132         System.arraycopy(data, 0, new_data, confounderSize(), size);
133 
134         byte[] new_cksum = calculateRawChecksum(new_data, new_data.length);
135         //extract original cksum value
136         byte[] orig_cksum = new byte[cksumSize() - confounderSize()];
137         System.arraycopy(cksum,  confounderSize(), orig_cksum, 0,
138                          cksumSize() - confounderSize());
139 
140         return isChecksumEqual(orig_cksum, new_cksum);
141     }
142 
143     /**
144      * Decrypts keyed checksum.
145      * @param enc_cksum the buffer for encrypted checksum.
146      * @param key the key.
147      * @return the checksum.
148      *
149      * @modified by Yanni Zhang, 12/08/99.
150      */
151     private byte[] decryptKeyedChecksum(byte[] enc_cksum, byte[] key) throws KrbCryptoException {
152         //compute modified key
153         byte[] new_key = new byte[keySize()];
154         System.arraycopy(key, 0, new_key, 0, key.length);
155         for (int i = 0; i < new_key.length; i++)
156         new_key[i] = (byte)(new_key[i] ^ 0xf0);
157         //check for weak keys
158         try {
159             if (DESKeySpec.isWeak(new_key, 0)) {
160                 new_key[7] = (byte)(new_key[7] ^ 0xF0);
161             }
162         } catch (InvalidKeyException ex) {
163             // swallow, since it should never happen
164         }
165         byte[] ivec = new byte[new_key.length];
166 
167         byte[] cksum = new byte[enc_cksum.length];
168         Des.cbc_encrypt(enc_cksum, cksum, new_key, ivec, false);
169         return cksum;
170     }
171 
172     /**
173      * Calculates checksum using MD5.
174      * @param data the data used to generate the checksum.
175      * @param size length of the data.
176      * @return the checksum.
177      *
178      * @modified by Yanni Zhang, 12/08/99.
179      */
180     private byte[] calculateRawChecksum(byte[] data, int size) throws KrbCryptoException{
181         MessageDigest md5;
182         byte[] result = null;
183         try {
184             md5 = MessageDigest.getInstance("MD5");
185         } catch (Exception e) {
186             throw new KrbCryptoException("JCE provider may not be installed. " + e.getMessage());
187         }
188         try {
189             md5.update(data);
190             result = md5.digest();
191         } catch (Exception e) {
192             throw new KrbCryptoException(e.getMessage());
193         }
194         return result;
195     }
196 }
