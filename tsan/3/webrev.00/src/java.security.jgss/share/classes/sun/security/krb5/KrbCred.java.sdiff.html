 59     public KrbCred(Credentials tgt,
 60                    Credentials serviceTicket,
 61                    EncryptionKey key)
 62         throws KrbException, IOException {
 63 
 64         PrincipalName client = tgt.getClient();
 65         PrincipalName tgService = tgt.getServer();
 66         if (!serviceTicket.getClient().equals(client))
 67             throw new KrbException(Krb5.KRB_ERR_GENERIC,
 68                                 "Client principal does not match");
 69 
 70         // XXX Check Windows flag OK-TO-FORWARD-TO
 71 
 72         // Invoke TGS-REQ to get a forwarded TGT for the peer
 73 
 74         KDCOptions options = new KDCOptions();
 75         options.set(KDCOptions.FORWARDED, true);
 76         options.set(KDCOptions.FORWARDABLE, true);
 77 
 78         KrbTgsReq tgsReq = new KrbTgsReq(options, tgt, tgService,
 79                 null, null, null, null,
 80                 null,   // No easy way to get addresses right
 81                 null, null, null);
 82         credMessg = createMessage(tgsReq.sendAndGetCreds(), key);
 83 
 84         obuf = credMessg.asn1Encode();
 85     }
