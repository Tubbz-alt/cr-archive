31 package sun.security.krb5.internal.crypto;
32 
33 import sun.security.krb5.*;
34 import sun.security.krb5.internal.*;
35 
36 public class Crc32CksumType extends CksumType {
37 
38     public Crc32CksumType() {
39     }
40 
41     public int confounderSize() {
42         return 0;
43     }
44 
45     public int cksumType() {
46         return Checksum.CKSUMTYPE_CRC32;
47     }
48 
49     public boolean isKeyed() {
50         return false;
51     }
52 
53     public int cksumSize() {
54         return 4;
55     }
56 
57     public int keyType() {
58         return Krb5.KEYTYPE_NULL;
59     }
60 
61     public int keySize() {
62         return 0;
63     }
64 
65     public byte[] calculateChecksum(byte[] data, int size,
66             byte[] key, int usage) {
67         return crc32.byte2crc32sum_bytes(data, size);
68     }
69 
70     public boolean verifyChecksum(byte[] data, int size,
71             byte[] key, byte[] checksum, int usage) {
72         return CksumType.isChecksumEqual(checksum,
73                 crc32.byte2crc32sum_bytes(data));
74     }
75 
76     public static byte[] int2quad(long input) {
77         byte[] output = new byte[4];
78         for (int i = 0; i < 4; i++) {
79             output[i] = (byte)((input >>> (i * 8)) & 0xff);
80         }
81         return output;
82     }
83 
84     public static long bytes2long(byte[] input) {
85         long result = 0;
86 
87         result |= (((long)input[0]) & 0xffL) << 24;
88         result |= (((long)input[1]) & 0xffL) << 16;
89         result |= (((long)input[2]) & 0xffL) << 8;
90         result |= (((long)input[3]) & 0xffL);
91         return result;
92     }
93 }
