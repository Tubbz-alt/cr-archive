 30 package sun.security.krb5.internal.crypto;
 31 
 32 import sun.security.krb5.Checksum;
 33 import sun.security.krb5.KrbCryptoException;
 34 import sun.security.krb5.internal.*;
 35 import java.security.MessageDigest;


 36 
 37 public final class RsaMd5CksumType extends CksumType {
 38 
 39     public RsaMd5CksumType() {
 40     }
 41 
 42     public int confounderSize() {
 43         return 0;
 44     }
 45 
 46     public int cksumType() {
 47         return Checksum.CKSUMTYPE_RSA_MD5;
 48     }
 49 
 50     public boolean isKeyed() {
 51         return false;
 52     }
 53 
 54     public int cksumSize() {
 55         return 16;
 56     }
 57 
 58     public int keyType() {
 59         return Krb5.KEYTYPE_NULL;
 60     }
 61 
 62     public int keySize() {
 63         return 0;
 64     }
 65 
 66     /**
 67      * Calculates checksum using MD5.
 68      * @param data the data used to generate the checksum.
 69      * @param size length of the data.
 70      * @return the checksum.
 71      *
 72      * @modified by Yanni Zhang, 12/08/99.
 73      */
 74 
 75     public byte[] calculateChecksum(byte[] data, int size,
 76             byte[] key, int usage) throws KrbCryptoException{
 77         MessageDigest md5;
 78         byte[] result = null;
 79         try {
 80             md5 = MessageDigest.getInstance("MD5");
 81         } catch (Exception e) {
 82             throw new KrbCryptoException("JCE provider may not be installed. " + e.getMessage());
 83         }
 84         try {
 85             md5.update(data);
 86             result = md5.digest();
 87         } catch (Exception e) {
 88             throw new KrbCryptoException(e.getMessage());
 89         }
 90         return result;
 91     }
 92 
 93     @Override
 94     public boolean verifyChecksum(byte[] data, int size,
 95             byte[] key, byte[] checksum, int usage)
 96             throws KrbCryptoException {
 97         try {
 98             byte[] calculated = MessageDigest.getInstance("MD5").digest(data);
 99             return CksumType.isChecksumEqual(calculated, checksum);
100         } catch (Exception e) {
101             return false;
102         }
103     }

104 }
