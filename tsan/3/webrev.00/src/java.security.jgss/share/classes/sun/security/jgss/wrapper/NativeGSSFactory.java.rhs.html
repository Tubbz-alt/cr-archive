 26 package sun.security.jgss.wrapper;
 27 

 28 import java.security.Provider;
 29 import java.util.Vector;
 30 import org.ietf.jgss.*;
 31 import sun.security.jgss.GSSUtil;
 32 import sun.security.jgss.GSSCaller;
 33 import sun.security.jgss.GSSExceptionImpl;
 34 import sun.security.jgss.spi.*;
 35 
 36 import static java.nio.charset.StandardCharsets.UTF_8;
 37 
 38 /**
 39  * JGSS plugin for generic mechanisms provided through native GSS framework.
 40  *
 41  * @author Valerie Peng
 42  */
 43 
 44 public final class NativeGSSFactory implements MechanismFactory {
 45 
 46     GSSLibStub cStub = null;
 47     private final GSSCaller caller;
 48 
 49     private GSSCredElement getCredFromSubject(GSSNameElement name,
 50                                               boolean initiate)
 51         throws GSSException {
 52         Oid mech = cStub.getMech();
 53         Vector<GSSCredElement> creds = GSSUtil.searchSubject
 54             (name, mech, initiate, GSSCredElement.class);
 55 
 56         // If Subject is present but no native creds available
 57         if (creds != null && creds.isEmpty()) {
 58             if (GSSUtil.useSubjectCredsOnly(caller)) {
 59                 throw new GSSException(GSSException.NO_CRED);
 60             }
 61         }
 62 
 63         GSSCredElement result = ((creds == null || creds.isEmpty()) ?
 64                                  null : creds.firstElement());
 65         // Force permission check before returning the cred to caller
 66         if (result != null) {
 67             result.doServicePermCheck();
 68         }
 69         return result;
 70     }
 71 
 72     public NativeGSSFactory(GSSCaller caller) {
 73         this.caller = caller;
 74         // Have to call setMech(Oid) explicitly before calling other
 75         // methods. Otherwise, NPE may be thrown unexpectantly
 76     }
 77 
 78     public void setMech(Oid mech) throws GSSException {
 79         cStub = GSSLibStub.getInstance(mech);
 80     }
 81 
 82     public GSSNameSpi getNameElement(String nameStr, Oid nameType)
 83         throws GSSException {
 84         byte[] nameBytes =
 85                 (nameStr == null ? null : nameStr.getBytes(UTF_8));
 86         return new GSSNameElement(nameBytes, nameType, cStub);





 87     }
 88 
 89     public GSSNameSpi getNameElement(byte[] name, Oid nameType)
 90         throws GSSException {
 91         return new GSSNameElement(name, nameType, cStub);
 92     }
 93 
 94     public GSSCredentialSpi getCredentialElement(GSSNameSpi name,
 95                                                  int initLifetime,
 96                                                  int acceptLifetime,
 97                                                  int usage)
 98         throws GSSException {
 99         GSSNameElement nname = null;
100         if (name != null && !(name instanceof GSSNameElement)) {
101             nname = (GSSNameElement)
102                 getNameElement(name.toString(), name.getStringNameType());
103         } else nname = (GSSNameElement) name;
104 
105         if (usage == GSSCredential.INITIATE_AND_ACCEPT) {
106             // Force separate acqusition of cred element since
107             // MIT's impl does not correctly report NO_CRED error.
108             usage = GSSCredential.INITIATE_ONLY;
109         }
110 
111         GSSCredElement credElement =
112             getCredFromSubject(nname, (usage == GSSCredential.INITIATE_ONLY));
113 
114         if (credElement == null) {
115             // No cred in the Subject
116             if (usage == GSSCredential.INITIATE_ONLY) {
117                 credElement = new GSSCredElement(nname, initLifetime,
118                                                  usage, cStub);
119             } else if (usage == GSSCredential.ACCEPT_ONLY) {
120                 if (nname == null) {
121                     nname = GSSNameElement.DEF_ACCEPTOR;
122                 }
123                 credElement = new GSSCredElement(nname, acceptLifetime,
124                                                  usage, cStub);
125             } else {
126                 throw new GSSException(GSSException.FAILURE, -1,
127                                        "Unknown usage mode requested");
128             }
129         }
130         return credElement;
131     }
132 
133     public GSSContextSpi getMechanismContext(GSSNameSpi peer,
134                                              GSSCredentialSpi myCred,
135                                              int lifetime)
136         throws GSSException {
137         if (peer == null) {
138             throw new GSSException(GSSException.BAD_NAME);
139         } else if (!(peer instanceof GSSNameElement)) {
140             peer = (GSSNameElement)
141                 getNameElement(peer.toString(), peer.getStringNameType());
142         }
143         if (myCred == null) {
144             myCred = getCredFromSubject(null, true);
145         } else if (!(myCred instanceof GSSCredElement)) {
146             throw new GSSException(GSSException.NO_CRED);
147         }
148         return new NativeGSSContext((GSSNameElement) peer,
149                                      (GSSCredElement) myCred,
150                                      lifetime, cStub);
151     }
152 
153     public GSSContextSpi getMechanismContext(GSSCredentialSpi myCred)
154         throws GSSException {
155         if (myCred == null) {
156             myCred = getCredFromSubject(null, false);
157         } else if (!(myCred instanceof GSSCredElement)) {
158             throw new GSSException(GSSException.NO_CRED);
159         }
160         return new NativeGSSContext((GSSCredElement) myCred, cStub);
161     }
162 
163     public GSSContextSpi getMechanismContext(byte[] exportedContext)
164         throws GSSException {
165         return cStub.importContext(exportedContext);
166     }
167 
168     public final Oid getMechanismOid() {
169         return cStub.getMech();
170     }
171 
172     public Provider getProvider() {
173         return SunNativeProvider.INSTANCE;
174     }
175 
176     public Oid[] getNameTypes() throws GSSException {
177         return cStub.inquireNamesForMech();
178     }
179 }
