 75     public byte[] calculateChecksum(byte[] data, int size,
 76             byte[] key, int usage) throws KrbCryptoException{
 77         MessageDigest md5;
 78         byte[] result = null;
 79         try {
 80             md5 = MessageDigest.getInstance("MD5");
 81         } catch (Exception e) {
 82             throw new KrbCryptoException("JCE provider may not be installed. " + e.getMessage());
 83         }
 84         try {
 85             md5.update(data);
 86             result = md5.digest();
 87         } catch (Exception e) {

 89         }
 90         return result;
 91     }

 93     @Override
 94     public boolean verifyChecksum(byte[] data, int size,
 95             byte[] key, byte[] checksum, int usage)
 96             throws KrbCryptoException {
 97         try {
 98             byte[] calculated = MessageDigest.getInstance("MD5").digest(data);
 99             return CksumType.isChecksumEqual(calculated, checksum);
100         } catch (Exception e) {
101             return false;
102         }
