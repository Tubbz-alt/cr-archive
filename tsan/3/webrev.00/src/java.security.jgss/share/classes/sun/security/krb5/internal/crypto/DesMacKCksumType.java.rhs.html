30 package sun.security.krb5.internal.crypto;
31 
32 import sun.security.krb5.Checksum;
33 import sun.security.krb5.KrbCryptoException;
34 import sun.security.krb5.internal.*;
35 import javax.crypto.spec.DESKeySpec;
36 import java.security.InvalidKeyException;
37 
38 public class DesMacKCksumType extends CksumType {
39 
40     public DesMacKCksumType() {
41     }
42 
43     public int confounderSize() {
44         return 0;
45     }
46 
47     public int cksumType() {
48         return Checksum.CKSUMTYPE_DES_MAC_K;
49     }
50 
51     public boolean isKeyed() {
52         return true;
53     }
54 
55     public int cksumSize() {
56         return 16;
57     }
58 
59     public int keyType() {
60         return Krb5.KEYTYPE_DES;
61     }
62 
63     public int keySize() {
64         return 8;
65     }
66 




67     /**
68      * Calculates keyed checksum.
69      * @param data the data used to generate the checksum.
70      * @param size length of the data.
71      * @param key the key used to encrypt the checksum.
72      * @return keyed checksum.
73      *
74      * @modified by Yanni Zhang, 12/08/99.
75      */
76     public byte[] calculateChecksum(byte[] data, int size, byte[] key,
77         int usage) throws KrbCryptoException {
78         //check for weak keys
79         try {
80             if (DESKeySpec.isWeak(key, 0)) {
81                 key[7] = (byte)(key[7] ^ 0xF0);
82             }
83         } catch (InvalidKeyException ex) {
84             // swallow, since it should never happen
85         }
86         byte[] ivec = new byte[key.length];
87         System.arraycopy(key, 0, ivec, 0, key.length);
88         byte[] cksum = Des.des_cksum(ivec, data, key);
89         return cksum;
90     }
91 
92     public boolean verifyChecksum(byte[] data, int size,
93         byte[] key, byte[] checksum, int usage) throws KrbCryptoException {
94         byte[] new_cksum = calculateChecksum(data, data.length, key, usage);
95         return isChecksumEqual(checksum, new_cksum);
96     }
97 
98 }
