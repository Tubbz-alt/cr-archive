 31 package sun.security.krb5.internal;
 32 
 33 import sun.security.krb5.PrincipalName;
 34 import sun.security.krb5.EncryptedData;
 35 import sun.security.krb5.Asn1Exception;
 36 import sun.security.krb5.Realm;
 37 import sun.security.krb5.RealmException;
 38 import sun.security.util.*;
 39 import java.io.IOException;
 40 import java.math.BigInteger;
 41 
 42 /**
 43  * Implements the ASN.1 Ticket type.
 44  *
 45  * <pre>{@code
 46  * Ticket               ::= [APPLICATION 1] SEQUENCE {
 47  *      tkt-vno         [0] INTEGER (5),
 48  *      realm           [1] Realm,
 49  *      sname           [2] PrincipalName,
 50  *      enc-part        [3] EncryptedData -- EncTicketPart
 51  * }
 52  * }</pre>
 53  *
 54  * <p>
 55  * This definition reflects the Network Working Group RFC 4120
 56  * specification available at
 57  * <a href="http://www.ietf.org/rfc/rfc4120.txt">
 58  * http://www.ietf.org/rfc/rfc4120.txt</a>.
 59  */
 60 
 61 public class Ticket implements Cloneable {
 62     public int tkt_vno;
 63     public PrincipalName sname;
 64     public EncryptedData encPart;
 65 
 66     private Ticket() {
 67     }
 68 
 69     public Object clone() {
 70         Ticket new_ticket = new Ticket();
 71         new_ticket.sname = (PrincipalName)sname.clone();
 72         new_ticket.encPart = (EncryptedData)encPart.clone();
 73         new_ticket.tkt_vno = tkt_vno;
 74         return new_ticket;
 75     }
 76 
 77     public Ticket(
 78                   PrincipalName new_sname,
 79                   EncryptedData new_encPart
 80                       ) {
 81         tkt_vno = Krb5.TICKET_VNO;
 82         sname = new_sname;
 83         encPart = new_encPart;
 84     }
 85 
 86     // Warning: called by NativeCreds.c and nativeccache.c
 87     public Ticket(byte[] data) throws Asn1Exception,
 88     RealmException, KrbApErrException, IOException {
 89         init(new DerValue(data));
 90     }
 91 
 92     public Ticket(DerValue encoding) throws Asn1Exception,
 93     RealmException, KrbApErrException, IOException {
 94         init(encoding);
 95     }
 96 
 97     /**
 98      * Initializes a Ticket object.
 99      * @param encoding a single DER-encoded value.
100      * @exception Asn1Exception if an error occurs while decoding an ASN1 encoded data.
101      * @exception IOException if an I/O error occurs while reading encoded data.
102      * @exception KrbApErrException if the value read from the DER-encoded data stream does not match the pre-defined value.
103      * @exception RealmException if an error occurs while parsing a Realm object.
104      */
105 
106     private void init(DerValue encoding) throws Asn1Exception,
107     RealmException, KrbApErrException, IOException {
108         DerValue der;
109         DerValue subDer;
110         if (((encoding.getTag() & (byte)0x1F) != Krb5.KRB_TKT)
111             || (encoding.isApplication() != true)
112             || (encoding.isConstructed() != true))
113             throw new Asn1Exception(Krb5.ASN1_BAD_ID);
114         der = encoding.getData().getDerValue();
115         if (der.getTag() != DerValue.tag_Sequence)
116             throw new Asn1Exception(Krb5.ASN1_BAD_ID);
117         subDer = der.getData().getDerValue();
118         if ((subDer.getTag() & (byte)0x1F) != (byte)0x00)
119             throw new Asn1Exception(Krb5.ASN1_BAD_ID);
120         tkt_vno = subDer.getData().getBigInteger().intValue();
121         if (tkt_vno != Krb5.TICKET_VNO)
122             throw new KrbApErrException(Krb5.KRB_AP_ERR_BADVERSION);
123         Realm srealm = Realm.parse(der.getData(), (byte)0x01, false);
124         sname = PrincipalName.parse(der.getData(), (byte)0x02, false, srealm);
125         encPart = EncryptedData.parse(der.getData(), (byte)0x03, false);
126         if (der.getData().available() > 0)
127             throw new Asn1Exception(Krb5.ASN1_BAD_ID);
128     }
129 
130     /**
131      * Encodes a Ticket object.
132      * @return byte array of encoded ticket object.
133      * @exception Asn1Exception if an error occurs while decoding an ASN1 encoded data.
134      * @exception IOException if an I/O error occurs while reading encoded data.
135      */
136     public byte[] asn1Encode() throws Asn1Exception, IOException {
137         DerOutputStream bytes = new DerOutputStream();
138         DerOutputStream temp = new DerOutputStream();
139         DerValue[] der = new DerValue[4];
140         temp.putInteger(BigInteger.valueOf(tkt_vno));
141         bytes.write(DerValue.createTag(DerValue.TAG_CONTEXT, true, (byte)0x00), temp);
142         bytes.write(DerValue.createTag(DerValue.TAG_CONTEXT, true, (byte)0x01), sname.getRealm().asn1Encode());
143         bytes.write(DerValue.createTag(DerValue.TAG_CONTEXT, true, (byte)0x02), sname.asn1Encode());
144         bytes.write(DerValue.createTag(DerValue.TAG_CONTEXT, true, (byte)0x03), encPart.asn1Encode());
145         temp = new DerOutputStream();
146         temp.write(DerValue.tag_Sequence, bytes);
147         DerOutputStream ticket = new DerOutputStream();
148         ticket.write(DerValue.createTag(DerValue.TAG_APPLICATION, true, (byte)0x01), temp);
149         return ticket.toByteArray();
150     }
151 
152     /**
153      * Parse (unmarshal) a Ticket from a DER input stream.  This form
154      * parsing might be used when expanding a value which is part of
155      * a constructed sequence and uses explicitly tagged type.
156      *
157      * @exception Asn1Exception on error.
158      * @param data the Der input stream value, which contains one or more marshaled value.
159      * @param explicitTag tag number.
160      * @param optional indicate if this data field is optional
161      * @return an instance of Ticket.
162      */
163     public static Ticket parse(DerInputStream data, byte explicitTag, boolean optional) throws Asn1Exception, IOException, RealmException, KrbApErrException {
164         if ((optional) && (((byte)data.peekByte() & (byte)0x1F)!= explicitTag))
165             return null;
166         DerValue der = data.getDerValue();
167         if (explicitTag != (der.getTag() & (byte)0x1F))  {
168             throw new Asn1Exception(Krb5.ASN1_BAD_ID);
169         }
170         else {
171             DerValue subDer = der.getData().getDerValue();
172             return new Ticket(subDer);
173         }
174     }
175 
176 
177 }
