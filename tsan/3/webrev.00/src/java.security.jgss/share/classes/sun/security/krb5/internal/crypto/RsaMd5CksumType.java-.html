 50     }

 53         return false;
 54     }
 55 
 56     public int cksumSize() {
 57         return 16;
 58     }
 59 
 60     public int keyType() {
 61         return Krb5.KEYTYPE_NULL;
 62     }
 63 
 64     public int keySize() {
 65         return 0;
 66     }
 67 
 68     /**
 69      * Calculates checksum using MD5.
 70      * @param data the data used to generate the checksum.
 71      * @param size length of the data.
 72      * @return the checksum.
 73      *
 74      * @modified by Yanni Zhang, 12/08/99.
 75      */
 76 
 77     public byte[] calculateChecksum(byte[] data, int size) throws KrbCryptoException{
 78         MessageDigest md5;
 79         byte[] result = null;
 80         try {
 81             md5 = MessageDigest.getInstance("MD5");
 82         } catch (Exception e) {
 83             throw new KrbCryptoException("JCE provider may not be installed. " + e.getMessage());
 84         }
 85         try {
 86             md5.update(data);
 87             result = md5.digest();
 88         } catch (Exception e) {
 89             throw new KrbCryptoException(e.getMessage());
 90         }
 91         return result;
 92     }
 93

 95         byte[] key, int usage) throws KrbCryptoException {
 96                                              return null;
 97                                          }

 99     public boolean verifyKeyedChecksum(byte[] data, int size,
100         byte[] key, byte[] checksum, int usage) throws KrbCryptoException {
101         return false;
102     }
