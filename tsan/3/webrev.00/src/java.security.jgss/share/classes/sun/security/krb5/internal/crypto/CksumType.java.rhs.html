 32 package sun.security.krb5.internal.crypto;
 33 

 34 import sun.security.krb5.Checksum;


 35 import sun.security.krb5.KrbCryptoException;
 36 import sun.security.krb5.internal.*;
 37 
 38 public abstract class CksumType {
 39 
 40     private static boolean DEBUG = Krb5.DEBUG;
 41 
 42     public static CksumType getInstance(int cksumTypeConst)
 43         throws KdcErrException {
 44         CksumType cksumType = null;
 45         String cksumTypeName = null;
 46         switch (cksumTypeConst) {
 47         case Checksum.CKSUMTYPE_CRC32:
 48             cksumType = new Crc32CksumType();
 49             cksumTypeName = "sun.security.krb5.internal.crypto.Crc32CksumType";
 50             break;
 51         case Checksum.CKSUMTYPE_DES_MAC:
 52             cksumType = new DesMacCksumType();
 53             cksumTypeName = "sun.security.krb5.internal.crypto.DesMacCksumType";
 54             break;
 55         case Checksum.CKSUMTYPE_DES_MAC_K:
 56             cksumType = new DesMacKCksumType();
 57             cksumTypeName =
 58                 "sun.security.krb5.internal.crypto.DesMacKCksumType";
 59             break;
 60         case Checksum.CKSUMTYPE_RSA_MD5:
 61             cksumType = new RsaMd5CksumType();
 62             cksumTypeName = "sun.security.krb5.internal.crypto.RsaMd5CksumType";
 63             break;
 64         case Checksum.CKSUMTYPE_RSA_MD5_DES:
 65             cksumType = new RsaMd5DesCksumType();
 66             cksumTypeName =
 67                 "sun.security.krb5.internal.crypto.RsaMd5DesCksumType";
 68             break;
 69 
 70         case Checksum.CKSUMTYPE_HMAC_SHA1_DES3_KD:
 71             cksumType = new HmacSha1Des3KdCksumType();
 72             cksumTypeName =
 73                 "sun.security.krb5.internal.crypto.HmacSha1Des3KdCksumType";
 74             break;
 75 
 76         case Checksum.CKSUMTYPE_HMAC_SHA1_96_AES128:
 77             cksumType = new HmacSha1Aes128CksumType();
 78             cksumTypeName =
 79                 "sun.security.krb5.internal.crypto.HmacSha1Aes128CksumType";
 80             break;
 81 
 82         case Checksum.CKSUMTYPE_HMAC_SHA1_96_AES256:
 83             cksumType = new HmacSha1Aes256CksumType();
 84             cksumTypeName =
 85                 "sun.security.krb5.internal.crypto.HmacSha1Aes256CksumType";
 86             break;
 87 
 88         case Checksum.CKSUMTYPE_HMAC_SHA256_128_AES128:
 89             cksumType = new HmacSha2Aes128CksumType();
 90             cksumTypeName =
 91                     "sun.security.krb5.internal.crypto.HmacSha2Aes128CksumType";
 92             break;
 93 
 94         case Checksum.CKSUMTYPE_HMAC_SHA384_192_AES256:
 95             cksumType = new HmacSha2Aes256CksumType();
 96             cksumTypeName =
 97                     "sun.security.krb5.internal.crypto.HmacSha2Aes256CksumType";
 98             break;
 99 
100         case Checksum.CKSUMTYPE_HMAC_MD5_ARCFOUR:
101             cksumType = new HmacMd5ArcFourCksumType();
102             cksumTypeName =
103                 "sun.security.krb5.internal.crypto.HmacMd5ArcFourCksumType";
104             break;
105 
106             // currently we don't support MD4.
107         case Checksum.CKSUMTYPE_RSA_MD4_DES_K:
108             // cksumType = new RsaMd4DesKCksumType();
109             // cksumTypeName =
110             //          "sun.security.krb5.internal.crypto.RsaMd4DesKCksumType";
111         case Checksum.CKSUMTYPE_RSA_MD4:
112             // cksumType = new RsaMd4CksumType();
113             // linux box support rsamd4, how to solve conflict?
114             // cksumTypeName =
115             //          "sun.security.krb5.internal.crypto.RsaMd4CksumType";
116         case Checksum.CKSUMTYPE_RSA_MD4_DES:
117             // cksumType = new RsaMd4DesCksumType();
118             // cksumTypeName =
119             //          "sun.security.krb5.internal.crypto.RsaMd4DesCksumType";
120 
121         default:
122             throw new KdcErrException(Krb5.KDC_ERR_SUMTYPE_NOSUPP);
123         }
124         if (DEBUG) {
125             System.out.println(">>> CksumType: " + cksumTypeName);
126         }
127         return cksumType;
128     }
129 





















130     public abstract int confounderSize();
131 
132     public abstract int cksumType();
133 
134     public abstract boolean isKeyed();
135 
136     public abstract int cksumSize();
137 
138     public abstract int keyType();
139 
140     public abstract int keySize();
141 
142     // Note: key and usage will be ignored for an unkeyed checksum.
143     public abstract byte[] calculateChecksum(byte[] data, int size,


144         byte[] key, int usage) throws KrbCryptoException;
145 
146     // Note: key and usage will be ignored for an unkeyed checksum.
147     public abstract boolean verifyChecksum(byte[] data, int size,
148         byte[] key, byte[] checksum, int usage) throws KrbCryptoException;
149 
150     public static boolean isChecksumEqual(byte[] cksum1, byte[] cksum2) {
151         if (cksum1 == cksum2)
152             return true;
153         if ((cksum1 == null && cksum2 != null) ||
154             (cksum1 != null && cksum2 == null))
155             return false;
156         if (cksum1.length != cksum2.length)
157             return false;
158         for (int i = 0; i < cksum1.length; i++)
159             if (cksum1[i] != cksum2[i])
160                 return false;
161         return true;
162     }
163 
164 }
