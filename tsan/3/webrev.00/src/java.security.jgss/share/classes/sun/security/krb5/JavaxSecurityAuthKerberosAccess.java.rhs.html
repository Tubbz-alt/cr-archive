26 package sun.security.krb5;
27 
28 import javax.security.auth.kerberos.KerberosPrincipal;
29 import javax.security.auth.kerberos.KerberosTicket;
30 import javax.security.auth.kerberos.KeyTab;


31 
32 /**
33  * An unsafe tunnel to get non-public access to classes in the
34  * javax.security.auth.kerberos package.
35  */
36 public interface JavaxSecurityAuthKerberosAccess {
37     /**
38      * Returns a snapshot to the backing keytab
39      */
40     public sun.security.krb5.internal.ktab.KeyTab keyTabTakeSnapshot(
41             KeyTab ktab);
42 
43     public KerberosPrincipal kerberosTicketGetClientAlias(KerberosTicket t);
44 
45     public void kerberosTicketSetClientAlias(KerberosTicket t, KerberosPrincipal a);
46 
47     public KerberosPrincipal kerberosTicketGetServerAlias(KerberosTicket t);
48 
49     public void kerberosTicketSetServerAlias(KerberosTicket t, KerberosPrincipal a);
50     /**
51      * Returns the proxy for a KerberosTicket.
52      */
53     public KerberosTicket kerberosTicketGetProxy(KerberosTicket t);
54 
55     /**
56      * Sets the proxy for a KerberosTicket.
57      */
58     public void kerberosTicketSetProxy(KerberosTicket t, KerberosTicket p);
59 }
