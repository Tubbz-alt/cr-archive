26 package javax.security.auth.kerberos;
27 
28 import sun.security.krb5.JavaxSecurityAuthKerberosAccess;


29 
30 class JavaxSecurityAuthKerberosAccessImpl
31         implements JavaxSecurityAuthKerberosAccess {
32     public sun.security.krb5.internal.ktab.KeyTab keyTabTakeSnapshot(
33             KeyTab ktab) {
34         return ktab.takeSnapshot();
35     }
36 
37     public KerberosPrincipal kerberosTicketGetClientAlias(KerberosTicket t) {
38         return t.clientAlias;
39     }
40 
41     public void kerberosTicketSetClientAlias(KerberosTicket t, KerberosPrincipal a) {
42         t.clientAlias = a;
43     }
44 
45     public KerberosPrincipal kerberosTicketGetServerAlias(KerberosTicket t) {
46         return t.serverAlias;
47     }
48 
49     public void kerberosTicketSetServerAlias(KerberosTicket t, KerberosPrincipal a) {
50         t.serverAlias = a;
51     }
52 
53     public KerberosTicket kerberosTicketGetProxy(KerberosTicket t) {
54         return t.proxy;
55     }
56 
57     public void kerberosTicketSetProxy(KerberosTicket t, KerberosTicket p) {
58         t.proxy = p;
59     }
60 }
