 41 package j2dbench;
 42 
 43 import java.util.Hashtable;
 44 import java.util.Properties;
 45 import java.util.Enumeration;
 46 import java.util.Vector;
 47 import java.io.PrintWriter;
 48 
 49 public class ResultSet {
 50     static Hashtable preferprops;
 51     static Hashtable ignoreprops;
 52 
 53     // Preferred props - will be listed even if undefined
 54     static String[] preferredkeys = {
 55         "java.version",
 56         "java.runtime.version",
 57         "java.class.version",
 58         "java.vm.version",
 59         "java.vm.name",
 60         "java.vm.info",
 61         "java.home",
 62         "java.compiler",
 63         "os.arch",
 64         "os.name",
 65         "os.version",
 66         "user.name",
 67         "sun.cpu.endian",
 68         "sun.cpu.isalist",
 69     };
 70 
 71     // Ignored props - will not be copied to results file
 72     static String[] ignoredkeys = {
 73         "user.dir",
 74         "user.home",
 75         "user.timezone",
 76         "path.separator",
 77         "line.separator",
 78         "file.separator",
 79         "file.encoding",
 80         "java.class.path",
 81         "java.library.path",
 82         "java.io.tmpdir",
 83         "java.util.prefs.PreferencesFactory",
 84         "sun.java2d.fontpath",
 85         "sun.boot.library.path",
 86     };
 87 
 88     /*
 89      * Other props, as of 1.4.0, not classified as "preferred" or "ignored"
 90      * Allowed to propagate to the results file if defined.
 91      *
 92      * java.runtime.name
 93      * java.vendor
 94      * java.vendor.url
 95      * java.vendor.url.bug
 96      * java.specification.name
 97      * java.specification.vendor
 98      * java.specification.version
 99      * java.vm.specification.name
100      * java.vm.specification.vendor
101      * java.vm.specification.version
102      * java.vm.vendor
103      * java.awt.graphicsenv
104      * java.awt.printerjob
105      * user.language
106      * sun.os.patch.level
107      * sun.arch.data.model
108      * sun.io.unicode.encoding
109      */
110 
111     static String unknown = "unknown";
112 
113     static {
114         preferprops = new Hashtable();
115         for (int i = 0; i < preferredkeys.length; i++) {
116             preferprops.put(preferredkeys[i], unknown);
117         }
118         ignoreprops = new Hashtable();
119         for (int i = 0; i < ignoredkeys.length; i++) {
120             ignoreprops.put(ignoredkeys[i], unknown);
121         }
122     }
123 
124     Hashtable props;
125     Vector results;
126     long start;
127     long end;
128     String title;
129     String description;
130 
131     public ResultSet() {
132         Properties sysprops = System.getProperties();
133         props = (Hashtable) preferprops.clone();
134         Enumeration enum_ = sysprops.keys();
135         while (enum_.hasMoreElements()) {
136             Object key = enum_.nextElement();
137             if (!ignoreprops.containsKey(key)) {
138                 props.put(key, sysprops.get(key));
139             }
140         }
141         results = new Vector();
142         start = System.currentTimeMillis();
143     }
144 
145     public void setTitle(String title) {
146         this.title = title;
147     }
148 
149     public void setDescription(String desc) {
150         this.description = desc;
151     }
152 
153     public void record(Result result) {
154         results.addElement(result);
155     }
156 
157     public void summarize() {
158         end = System.currentTimeMillis();
159         for (int i = 0; i < results.size(); i++) {
160             ((Result) results.elementAt(i)).summarize();
161         }
162     }
163 
164     public void write(PrintWriter pw) {
165         pw.println("<result-set version=\"0.1\" name=\""+title+"\">");
166         pw.println("  <test-desc>"+description+"</test-desc>");
167         pw.println("  <test-date start=\""+start+"\" end=\""+end+"\"/>");
168         for (int i = 0; i < preferredkeys.length; i++) {
169             String key = preferredkeys[i];
170             pw.println("  <sys-prop key=\""+key+
171                        "\" value=\""+props.get(key)+"\"/>");
172         }
173         Enumeration enum_ = props.keys();
174         while (enum_.hasMoreElements()) {
175             Object key = enum_.nextElement();
176             if (!preferprops.containsKey(key)) {
177                 pw.println("  <sys-prop key=\""+key+
178                            "\" value=\""+props.get(key)+"\"/>");
179             }
180         }
181         for (int i = 0; i < results.size(); i++) {
182             ((Result) results.elementAt(i)).write(pw);
183         }
184         pw.println("</result-set>");
185     }
186 }
