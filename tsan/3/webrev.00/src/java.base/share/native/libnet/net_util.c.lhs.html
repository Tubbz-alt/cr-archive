 46 JNIEXPORT jint JNICALL
 47 DEF_JNI_OnLoad(JavaVM *vm, void *reserved)
 48 {
 49     JNIEnv *env;
 50     jclass iCls;
 51     jmethodID mid;
 52     jstring s;
 53     jint preferIPv4Stack;
 54     if ((*vm)->GetEnv(vm, (void**) &env, JNI_VERSION_1_2) != JNI_OK) {
 55         return JNI_EVERSION; /* JNI version not supported */
 56     }
 57 
 58     iCls = (*env)->FindClass(env, "java/lang/Boolean");
 59     CHECK_NULL_RETURN(iCls, JNI_VERSION_1_2);
 60     mid = (*env)->GetStaticMethodID(env, iCls, "getBoolean", "(Ljava/lang/String;)Z");
 61     CHECK_NULL_RETURN(mid, JNI_VERSION_1_2);
 62     s = (*env)->NewStringUTF(env, "java.net.preferIPv4Stack");
 63     CHECK_NULL_RETURN(s, JNI_VERSION_1_2);
 64     preferIPv4Stack = (*env)->CallStaticBooleanMethod(env, iCls, mid, s);
 65 
 66     /*
 67      * Since we have initialized and loaded the socket library we will
 68      * check now whether we have IPv6 on this platform and if the
 69      * supporting socket APIs are available
 70      */

 71     IPv6_available = IPv6_supported() & (!preferIPv4Stack);
 72 
 73     /* check if SO_REUSEPORT is supported on this platform */
 74     REUSEPORT_available = reuseport_supported();
 75     platformInit();
 76     parseExclusiveBindProperty(env);
 77 
 78     return JNI_VERSION_1_2;
 79 }

 83 JNIEXPORT void JNICALL initInetAddressIDs(JNIEnv *env) {
 84     if (!initialized) {
 85         Java_java_net_InetAddress_init(env, 0);
 86         JNU_CHECK_EXCEPTION(env);
 87         Java_java_net_Inet4Address_init(env, 0);
 88         JNU_CHECK_EXCEPTION(env);
 89         Java_java_net_Inet6Address_init(env, 0);
 90         JNU_CHECK_EXCEPTION(env);
 91         initialized = 1;
 92     }
 93 }

145 jboolean getInet6Address_ipaddress(JNIEnv *env, jobject iaObj, char *dest) {
146     jobject holder, addr;
147 
148     holder = (*env)->GetObjectField(env, iaObj, ia6_holder6ID);

156 jboolean setInet6Address_ipaddress(JNIEnv *env, jobject iaObj, char *address) {
157     jobject holder;
158     jbyteArray addr;
159 
160     holder = (*env)->GetObjectField(env, iaObj, ia6_holder6ID);
161     CHECK_NULL_RETURN(holder, JNI_FALSE);
162     addr =  (jbyteArray)(*env)->GetObjectField(env, holder, ia6_ipaddressID);
163     if (addr == NULL) {
164         addr = (*env)->NewByteArray(env, 16);
165         CHECK_NULL_RETURN(addr, JNI_FALSE);
166         (*env)->SetObjectField(env, holder, ia6_ipaddressID, addr);
167     }
168     (*env)->SetByteArrayRegion(env, addr, 0, 16, (jbyte *)address);
169     return JNI_TRUE;
170 }

209 JNIEXPORT jobject JNICALL
210 NET_SockaddrToInetAddress(JNIEnv *env, SOCKETADDRESS *sa, int *port) {
211     jobject iaObj;
212     if (sa->sa.sa_family == AF_INET6) {
213         jbyte *caddr = (jbyte *)&sa->sa6.sin6_addr;
214         if (NET_IsIPv4Mapped(caddr)) {
215             int address;
216             iaObj = (*env)->NewObject(env, ia4_class, ia4_ctrID);
217             CHECK_NULL_RETURN(iaObj, NULL);
218             address = NET_IPv4MappedToIPv4(caddr);
219             setInetAddress_addr(env, iaObj, address);
220             JNU_CHECK_EXCEPTION_RETURN(env, NULL);
221             setInetAddress_family(env, iaObj, java_net_InetAddress_IPv4);
222             JNU_CHECK_EXCEPTION_RETURN(env, NULL);
223         } else {
224             jboolean ret;
225             iaObj = (*env)->NewObject(env, ia6_class, ia6_ctrID);
226             CHECK_NULL_RETURN(iaObj, NULL);
227             ret = setInet6Address_ipaddress(env, iaObj, (char *)&sa->sa6.sin6_addr);
228             if (ret == JNI_FALSE)
229                 return NULL;
230             setInetAddress_family(env, iaObj, java_net_InetAddress_IPv6);
231             JNU_CHECK_EXCEPTION_RETURN(env, NULL);
232             setInet6Address_scopeid(env, iaObj, sa->sa6.sin6_scope_id);
233         }
234         *port = ntohs(sa->sa6.sin6_port);
235     } else {
236         iaObj = (*env)->NewObject(env, ia4_class, ia4_ctrID);
237         CHECK_NULL_RETURN(iaObj, NULL);
238         setInetAddress_family(env, iaObj, java_net_InetAddress_IPv4);
239         JNU_CHECK_EXCEPTION_RETURN(env, NULL);
240         setInetAddress_addr(env, iaObj, ntohl(sa->sa4.sin_addr.s_addr));
241         JNU_CHECK_EXCEPTION_RETURN(env, NULL);
242         *port = ntohs(sa->sa4.sin_port);
243     }
244     return iaObj;
245 }

247 JNIEXPORT jboolean JNICALL
248 NET_SockaddrEqualsInetAddress(JNIEnv *env, SOCKETADDRESS *sa, jobject iaObj)
249 {
250     jint family = getInetAddress_family(env, iaObj) ==
251         java_net_InetAddress_IPv4 ? AF_INET : AF_INET6;
252     JNU_CHECK_EXCEPTION_RETURN(env, JNI_FALSE);
253     if (sa->sa.sa_family == AF_INET6) {
254         jbyte *caddrNew = (jbyte *)&sa->sa6.sin6_addr;
255         if (NET_IsIPv4Mapped(caddrNew)) {
256             int addrNew, addrCur;
257             if (family == AF_INET6) {
258                 return JNI_FALSE;
259             }
260             addrNew = NET_IPv4MappedToIPv4(caddrNew);
261             addrCur = getInetAddress_addr(env, iaObj);
262             JNU_CHECK_EXCEPTION_RETURN(env, JNI_FALSE);
263             if (addrNew == addrCur) {
264                 return JNI_TRUE;
265             } else {
266                 return JNI_FALSE;
267             }
268         } else {
269             jbyte caddrCur[16];
270             if (family == AF_INET) {
271                 return JNI_FALSE;
272             }
273             getInet6Address_ipaddress(env, iaObj, (char *)caddrCur);
274             if (NET_IsEqual(caddrNew, caddrCur) &&
275                 sa->sa6.sin6_scope_id == getInet6Address_scopeid(env, iaObj))
276             {
277                 return JNI_TRUE;
278             } else {
279                 return JNI_FALSE;
280             }
281         }
282     } else {
283         int addrNew, addrCur;
284         if (family != AF_INET) {
285             return JNI_FALSE;
286         }
287         addrNew = ntohl(sa->sa4.sin_addr.s_addr);
288         addrCur = getInetAddress_addr(env, iaObj);
289         JNU_CHECK_EXCEPTION_RETURN(env, JNI_FALSE);
290         if (addrNew == addrCur) {
291             return JNI_TRUE;
292         } else {
293             return JNI_FALSE;
294         }
295     }
296 }

298 JNIEXPORT jint JNICALL
299 NET_GetPortFromSockaddr(SOCKETADDRESS *sa) {
300     if (sa->sa.sa_family == AF_INET6) {
301         return ntohs(sa->sa6.sin6_port);
302     } else {
303         return ntohs(sa->sa4.sin_port);
304     }
305 }

307 unsigned short
308 in_cksum(unsigned short *addr, int len) {
309     int nleft = len;
310     int sum = 0;
311     unsigned short *w = addr;
312     unsigned short answer = 0;
313     while(nleft > 1) {
314         sum += *w++;
315         nleft -= 2;
316     }
317 
318     if (nleft == 1) {
319         *(unsigned char *) (&answer) = *(unsigned char *)w;
320         sum += answer;
321     }
322 
323     sum = (sum >> 16) + (sum & 0xffff);
324     sum += (sum >> 16);
325     answer = ~sum;
326     return (answer);
327 }
