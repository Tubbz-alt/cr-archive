26 package jdk.internal.misc;
27 
28 import java.lang.invoke.MethodHandle;
29 import java.lang.invoke.MethodHandles;
30 import java.lang.invoke.MethodHandles.Lookup;
31 import java.lang.invoke.MethodType;
32 import java.nio.channels.FileChannel.MapMode;
33 import java.security.AccessController;
34 import java.security.PrivilegedExceptionAction;
35 
36 /**
37  * JDK-specific map modes implemented in java.base.
38  */
39 public class ExtendedMapMode {
40 
41     static final MethodHandle MAP_MODE_CONSTRUCTOR;
42     static {
43         try {
44             PrivilegedExceptionAction<Lookup> pae = () ->
45                 MethodHandles.privateLookupIn(MapMode.class, MethodHandles.lookup());
46             Lookup lookup = AccessController.doPrivileged(pae);
47             var methodType = MethodType.methodType(void.class, String.class);
48             MAP_MODE_CONSTRUCTOR = lookup.findConstructor(MapMode.class, methodType);
49         } catch (Exception e) {
50             throw new InternalError(e);
51         }
52     }
53 
54     public static final MapMode READ_ONLY_SYNC = newMapMode("READ_ONLY_SYNC");
55 
56     public static final MapMode READ_WRITE_SYNC = newMapMode("READ_WRITE_SYNC");
57 
58     private static MapMode newMapMode(String name) {
59         try {
60             return (MapMode) MAP_MODE_CONSTRUCTOR.invoke(name);
61         } catch (Throwable e) {
62             throw new InternalError(e);
63         }
64     }
65 
66     private ExtendedMapMode() { }
67 }
