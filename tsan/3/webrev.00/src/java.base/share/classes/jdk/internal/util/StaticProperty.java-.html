 26 package jdk.internal.util;
 27 
 28 /**
 29  * System Property access for internal use only.
 30  * Read-only access to System property values initialized during Phase 1
 31  * are cached.  Setting, clearing, or modifying the value using
 32  * {@link System#setProperty) or {@link System#getProperties()} is ignored.
 33  * <strong>{@link SecurityManager#checkPropertyAccess} is NOT checked
 34  * in these access methods. The caller of these methods should take care to ensure
 35  * that the returned property is not made accessible to untrusted code.</strong>
 36  */
 37 public final class StaticProperty {
 38 
 39     // The class static initialization is triggered to initialize these final
 40     // fields during init Phase 1 and before a security manager is set.
 41     private static final String JAVA_HOME = initProperty("java.home");
 42     private static final String USER_HOME = initProperty("user.home");
 43     private static final String USER_DIR  = initProperty("user.dir");
 44     private static final String USER_NAME = initProperty("user.name");
 45 
 46     private StaticProperty() {}
 47 
 48     private static String initProperty(String key) {
 49         String v = System.getProperty(key);
 50         if (v == null) {
 51             throw new InternalError("null property: " + key);
 52         }
 53         return v;
 54     }
 55 
 56     /**
 57      * Return the {@code java.home} system property.
 58      *
 59      * <strong>{@link SecurityManager#checkPropertyAccess} is NOT checked
 60      * in this method. The caller of this method should take care to ensure
 61      * that the returned property is not made accessible to untrusted code.</strong>
 62      *
 63      * @return the {@code java.home} system property
 64      */
 65     public static String javaHome() {
 66         return JAVA_HOME;
 67     }
 68 
 69     /**
 70      * Return the {@code user.home} system property.
 71      *
 72      * <strong>{@link SecurityManager#checkPropertyAccess} is NOT checked
 73      * in this method. The caller of this method should take care to ensure
 74      * that the returned property is not made accessible to untrusted code.</strong>
 75      *
 76      * @return the {@code user.home} system property
 77      */
 78     public static String userHome() {
 79         return USER_HOME;
 80     }
 81 
 82     /**
 83      * Return the {@code user.dir} system property.
 84      *
 85      * <strong>{@link SecurityManager#checkPropertyAccess} is NOT checked
 86      * in this method. The caller of this method should take care to ensure
 87      * that the returned property is not made accessible to untrusted code.</strong>
 88      *
 89      * @return the {@code user.dir} system property
 90      */
 91     public static String userDir() {
 92         return USER_DIR;
 93     }
 94 
 95     /**
 96      * Return the {@code user.name} system property.
 97      *
 98      * <strong>{@link SecurityManager#checkPropertyAccess} is NOT checked
 99      * in this method. The caller of this method should take care to ensure
100      * that the returned property is not made accessible to untrusted code.</strong>
101      *
102      * @return the {@code user.name} system property
103      */
104     public static String userName() {
105         return USER_NAME;
106     }
107 }
