27 package javax.security.cert;

29 /**
30  * Certificate Encoding Exception. This is thrown whenever an error
31  * occurs whilst attempting to encode a certificate.
32  *
33  * <p><em>Note: The classes in the package {@code javax.security.cert}
34  * exist for compatibility with earlier versions of the
35  * Java Secure Sockets Extension (JSSE). New applications should instead
36  * use the standard Java SE certificate classes located in
37  * {@code java.security.cert}.</em></p>
38  *
39  * @since 1.4
40  * @author Hemma Prafullchandra
41  * @deprecated Use the classes in {@code java.security.cert} instead.
42  */
43 @SuppressWarnings("removal")
44 @Deprecated(since="9", forRemoval=true)
45 public class CertificateEncodingException extends CertificateException {
46 
47     @java.io.Serial
48     private static final long serialVersionUID = -8187642723048403470L;
49     /**
50      * Constructs a CertificateEncodingException with no detail message. A
51      * detail message is a String that describes this particular
52      * exception.
53      */
54     public CertificateEncodingException() {
55         super();
56     }
57 
58     /**
59      * Constructs a CertificateEncodingException with the specified detail
60      * message. A detail message is a String that describes this
61      * particular exception.
62      *
63      * @param message the detail message.
64      */
65     public CertificateEncodingException(String message) {
66         super(message);
67     }
68 }
