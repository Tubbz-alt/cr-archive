26 package javax.crypto;
27 
28 import java.security.GeneralSecurityException;
29 
30 /**
31  * This exception is thrown when a particular padding mechanism is
32  * requested but is not available in the environment.
33  *
34  * @author Jan Luehe
35  *
36  * @since 1.4
37  */
38 
39 public class NoSuchPaddingException extends GeneralSecurityException {
40 
41     @java.io.Serial
42     private static final long serialVersionUID = -4572885201200175466L;
43 
44     /**
45      * Constructs a NoSuchPaddingException with no detail
46      * message. A detail message is a String that describes this
47      * particular exception.
48      */
49     public NoSuchPaddingException() {
50         super();
51     }
52 
53     /**
54      * Constructs a NoSuchPaddingException with the specified
55      * detail message.
56      *
57      * @param msg the detail message.
58      */
59     public NoSuchPaddingException(String msg) {
60         super(msg);
61     }
62 }
