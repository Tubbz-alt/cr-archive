 27 package javax.net.ssl;
 28 
 29 import java.util.Enumeration;
 30 
 31 
 32 /**
 33  * A <code>SSLSessionContext</code> represents a set of
 34  * <code>SSLSession</code>s associated with a single entity. For example,
 35  * it could be associated with a server or client who participates in many
 36  * sessions concurrently.
 37  * <p>
 38  * Not all environments will contain session contexts.  For example, stateless
 39  * session resumption.
 40  * <p>
 41  * Session contexts may not contain all sessions. For example, stateless
 42  * sessions are not stored in the session context.
 43  * <p>
 44  * There are <code>SSLSessionContext</code> parameters that affect how
 45  * sessions are stored:
 46  * <UL>
 47  *      <LI>Sessions can be set to expire after a specified
 48  *      time limit.
 49  *      <LI>The number of sessions that can be stored in context
 50  *      can be limited.
 51  * </UL>
 52  * A session can be retrieved based on its session id, and all session id's
 53  * in a <code>SSLSessionContext</code> can be listed.
 54  *
 55  * @see SSLSession
 56  *
 57  * @since 1.4
 58  * @author Nathan Abramson
 59  * @author David Brownell
 60  */
 61 public interface SSLSessionContext {
 62 
 63     /**
 64      * Returns the <code>SSLSession</code> bound to the specified session id.
 65      *
 66      * @param sessionId the Session identifier
 67      * @return the <code>SSLSession</code> or null if
 68      * the specified session id does not refer to a valid SSLSession.
 69      *
 70      * @throws NullPointerException if <code>sessionId</code> is null.
 71      */
 72     public SSLSession getSession(byte[] sessionId);
 73 
 74     /**
 75      * Returns an Enumeration of all known session id's grouped under this
 76      * <code>SSLSessionContext</code>.
 77      * <p>Session contexts may not contain all sessions. For example,
 78      * stateless sessions are not stored in the session context.
 79      * <p>
 80      *
 81      * @return an enumeration of all the Session id's
 82      */
 83     public Enumeration<byte[]> getIds();
 84 
 85     /**
 86      * Sets the timeout limit for <code>SSLSession</code> objects grouped
 87      * under this <code>SSLSessionContext</code>.
 88      * <p>
 89      * If the timeout limit is set to 't' seconds, a session exceeds the
 90      * timeout limit 't' seconds after its creation time.
 91      * When the timeout limit is exceeded for a session, the
 92      * <code>SSLSession</code> object is invalidated and future connections
 93      * cannot resume or rejoin the session.
 94      * A check for sessions exceeding the timeout is made immediately whenever
 95      * the timeout limit is changed for this <code>SSLSessionContext</code>.
 96      *
 97      * @apiNote Note that the JDK Implementation uses default values for both
 98      *          the session cache size and timeout.  See
 99      *          {@code getSessionCacheSize} and {@code getSessionTimeout} for
100      *          more information.  Applications should consider their
101      *          performance requirements and override the defaults if necessary.
102      *
103      * @param seconds the new session timeout limit in seconds; zero means
104      *        there is no limit.
105      *
106      * @throws IllegalArgumentException if the timeout specified is {@code < 0}.
107      *
108      * @see #getSessionTimeout
109      */
110     public void setSessionTimeout(int seconds)
111                  throws IllegalArgumentException;
112 
113     /**
114      * Returns the timeout limit of <code>SSLSession</code> objects grouped
115      * under this <code>SSLSessionContext</code>.
116      * <p>
117      * If the timeout limit is set to 't' seconds, a session exceeds the
118      * timeout limit 't' seconds after its creation time.
119      * When the timeout limit is exceeded for a session, the
120      * <code>SSLSession</code> object is invalidated and future connections
121      * cannot resume or rejoin the session.
122      * A check for sessions exceeding the timeout limit is made immediately
123      * whenever the timeout limit is changed for this
124      * <code>SSLSessionContext</code>.
125      *
126      * @implNote The JDK implementation returns the session timeout as set by
127      *           the {@code setSessionTimeout} method, or if not set, a default
128      *           value of 86400 seconds (24 hours).
129      *
130      * @return the session timeout limit in seconds; zero means there is no
131      *         limit.
132      *
133      * @see #setSessionTimeout
134      */
135     public int getSessionTimeout();
136 
137     /**
138      * Sets the size of the cache used for storing <code>SSLSession</code>
139      * objects grouped under this <code>SSLSessionContext</code>.
140      *
141      * @apiNote Note that the JDK Implementation uses default values for both
142      *          the session cache size and timeout.  See
143      *          {@code getSessionCacheSize} and {@code getSessionTimeout} for
144      *          more information.  Applications should consider their
145      *          performance requirements and override the defaults if necessary.
146      *
147      * @param size the new session cache size limit; zero means there is no
148      *        limit.
149      *
150      * @throws IllegalArgumentException if the specified size is {@code < 0}.
151      *
152      * @see #getSessionCacheSize
153      */
154     public void setSessionCacheSize(int size)
155                  throws IllegalArgumentException;
156 
157     /**
158      * Returns the size of the cache used for storing <code>SSLSession</code>
159      * objects grouped under this <code>SSLSessionContext</code>.
160      *
161      * @implNote The JDK implementation returns the cache size as set by
162      *           the {@code setSessionCacheSize} method, or if not set, the
163      *           value of the {@systemProperty javax.net.ssl.sessionCacheSize}
164      *           system property.  If neither is set, it returns a default
165      *           value of 20480.
166      *
167      * @return size of the session cache; zero means there is no size limit.
168      *
169      * @see #setSessionCacheSize
170      */
171     public int getSessionCacheSize();
172 }
