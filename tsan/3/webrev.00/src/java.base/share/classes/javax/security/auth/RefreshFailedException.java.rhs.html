26 package javax.security.auth;
27 
28 /**
29  * Signals that a {@code refresh} operation failed.
30  *
31  * <p> This exception is thrown by credentials implementing
32  * the {@code Refreshable} interface when the {@code refresh}
33  * method fails.
34  *
35  * @since 1.4
36  */
37 public class RefreshFailedException extends Exception {
38 
39     @java.io.Serial
40     private static final long serialVersionUID = 5058444488565265840L;
41 
42     /**
43      * Constructs a RefreshFailedException with no detail message. A detail
44      * message is a String that describes this particular exception.
45      */
46     public RefreshFailedException() {
47         super();
48     }
49 
50     /**
51      * Constructs a RefreshFailedException with the specified detail
52      * message.  A detail message is a String that describes this particular
53      * exception.
54      *
55      * @param msg the detail message.
56      */
57     public RefreshFailedException(String msg) {
58         super(msg);
59     }
60 }
