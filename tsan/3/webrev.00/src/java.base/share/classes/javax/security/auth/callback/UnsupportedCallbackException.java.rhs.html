26 package javax.security.auth.callback;
27 
28 /**
29  * Signals that a {@code CallbackHandler} does not
30  * recognize a particular {@code Callback}.
31  *
32  * @since 1.4
33  */
34 public class UnsupportedCallbackException extends Exception {
35 
36     @java.io.Serial
37     private static final long serialVersionUID = -6873556327655666839L;
38 
39     /**
40      * @serial
41      */
42     @SuppressWarnings("serial") // Not statically typed as Serializable
43     private Callback callback;
44 
45     /**
46      * Constructs an {@code UnsupportedCallbackException}
47      * with no detail message.
48      *
49      * @param callback the unrecognized {@code Callback}.
50      */
51     public UnsupportedCallbackException(Callback callback) {
52         super();
53         this.callback = callback;
54     }
55 
56     /**
57      * Constructs a UnsupportedCallbackException with the specified detail
58      * message.  A detail message is a String that describes this particular
59      * exception.
60      *
61      * @param callback the unrecognized {@code Callback}.
62      *
63      * @param msg the detail message.
64      */
65     public UnsupportedCallbackException(Callback callback, String msg) {
66         super(msg);
67         this.callback = callback;
68     }
69 
70     /**
71      * Get the unrecognized {@code Callback}.
72      *
73      * @return the unrecognized {@code Callback}.
74      */
75     public Callback getCallback() {
76         return callback;
77     }
78 }
