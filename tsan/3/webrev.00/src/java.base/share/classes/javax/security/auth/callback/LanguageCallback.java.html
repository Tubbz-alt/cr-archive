26 package javax.security.auth.callback;
27 
28 import java.util.Locale;
29 
30 /**
31  * <p> Underlying security services instantiate and pass a
32  * {@code LanguageCallback} to the {@code handle}
33  * method of a {@code CallbackHandler} to retrieve the {@code Locale}
34  * used for localizing text.
35  *
36  * @since 1.4
37  * @see javax.security.auth.callback.CallbackHandler
38  */
39 public class LanguageCallback implements Callback, java.io.Serializable {
40 
41     @java.io.Serial
42     private static final long serialVersionUID = 2019050433478903213L;
43 
44     /**
45      * @serial
46      * @since 1.4
47      */
48     private Locale locale;
49 
50     /**
51      * Construct a {@code LanguageCallback}.
52      */
53     public LanguageCallback() { }
54 
55     /**
56      * Set the retrieved {@code Locale}.
57      *
58      * @param locale the retrieved {@code Locale}.
59      *
60      * @see #getLocale
61      */
62     public void setLocale(Locale locale) {
63         this.locale = locale;
64     }
65 
66     /**
67      * Get the retrieved {@code Locale}.
68      *
69      * @return the retrieved {@code Locale}, or null
70      *          if no {@code Locale} could be retrieved.
71      *
72      * @see #setLocale
73      */
74     public Locale getLocale() {
75         return locale;
76     }
77 }
