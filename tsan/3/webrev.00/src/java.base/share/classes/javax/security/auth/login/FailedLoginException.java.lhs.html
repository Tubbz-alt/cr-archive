26 package javax.security.auth.login;
27 
28 /**
29  * Signals that user authentication failed.
30  *
31  * <p> This exception is thrown by LoginModules if authentication failed.
32  * For example, a {@code LoginModule} throws this exception if
33  * the user entered an incorrect password.
34  *
35  * @since 1.4
36  */
37 public class FailedLoginException extends LoginException {
38 

39     private static final long serialVersionUID = 802556922354616286L;
40 
41     /**
42      * Constructs a FailedLoginException with no detail message. A detail
43      * message is a String that describes this particular exception.
44      */
45     public FailedLoginException() {
46         super();
47     }
48 
49     /**
50      * Constructs a FailedLoginException with the specified detail
51      * message.  A detail message is a String that describes this particular
52      * exception.
53      *
54      * @param msg the detail message.
55      */
56     public FailedLoginException(String msg) {
57         super(msg);
58     }
59 }
