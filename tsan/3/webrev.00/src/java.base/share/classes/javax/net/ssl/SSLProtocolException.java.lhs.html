29 /**
30  * Reports an error in the operation of the SSL protocol.  Normally
31  * this indicates a flaw in one of the protocol implementations.
32  *
33  * @since 1.4
34  * @author David Brownell
35  */
36 public
37 class SSLProtocolException extends SSLException
38 {

39     private static final long serialVersionUID = 5445067063799134928L;
40 
41     /**
42      * Constructs an exception reporting an SSL protocol error
43      * detected by an SSL subsystem.
44      *
45      * @param reason describes the problem.
46      */
47     public SSLProtocolException(String reason)
48     {
49         super(reason);
50     }
51 }
