 26 package javax.security.auth.callback;
 27 
 28 /**
 29  * <p> Underlying security services instantiate and pass a
 30  * {@code NameCallback} to the {@code handle}
 31  * method of a {@code CallbackHandler} to retrieve name information.
 32  *
 33  * @since 1.4
 34  * @see javax.security.auth.callback.CallbackHandler
 35  */
 36 public class NameCallback implements Callback, java.io.Serializable {
 37 
 38     @java.io.Serial
 39     private static final long serialVersionUID = 3770938795909392253L;
 40 
 41     /**
 42      * @serial
 43      * @since 1.4
 44      */
 45     private String prompt;
 46     /**
 47      * @serial
 48      * @since 1.4
 49      */
 50     private String defaultName;
 51     /**
 52      * @serial
 53      * @since 1.4
 54      */
 55     private String inputName;
 56 
 57     /**
 58      * Construct a {@code NameCallback} with a prompt.
 59      *
 60      * @param prompt the prompt used to request the name.
 61      *
 62      * @exception IllegalArgumentException if {@code prompt} is null
 63      *                  or if {@code prompt} has a length of 0.
 64      */
 65     public NameCallback(String prompt) {
 66         if (prompt == null || prompt.isEmpty())
 67             throw new IllegalArgumentException();
 68         this.prompt = prompt;
 69     }
 70 
 71     /**
 72      * Construct a {@code NameCallback} with a prompt
 73      * and default name.
 74      *
 75      * @param prompt the prompt used to request the information.
 76      *
 77      * @param defaultName the name to be used as the default name displayed
 78      *                  with the prompt.
 79      *
 80      * @exception IllegalArgumentException if {@code prompt} is null,
 81      *                  if {@code prompt} has a length of 0,
 82      *                  if {@code defaultName} is null,
 83      *                  or if {@code defaultName} has a length of 0.
 84      */
 85     public NameCallback(String prompt, String defaultName) {
 86         if (prompt == null || prompt.isEmpty() ||
 87             defaultName == null || defaultName.isEmpty())
 88             throw new IllegalArgumentException();
 89 
 90         this.prompt = prompt;
 91         this.defaultName = defaultName;
 92     }
 93 
 94     /**
 95      * Get the prompt.
 96      *
 97      * @return the prompt.
 98      */
 99     public String getPrompt() {
100         return prompt;
101     }
102 
103     /**
104      * Get the default name.
105      *
106      * @return the default name, or null if this {@code NameCallback}
107      *          was not instantiated with a {@code defaultName}.
108      */
109     public String getDefaultName() {
110         return defaultName;
111     }
112 
113     /**
114      * Set the retrieved name.
115      *
116      * @param name the retrieved name (which may be null).
117      *
118      * @see #getName
119      */
120     public void setName(String name) {
121         this.inputName = name;
122     }
123 
124     /**
125      * Get the retrieved name.
126      *
127      * @return the retrieved name (which may be null)
128      *
129      * @see #setName
130      */
131     public String getName() {
132         return inputName;
133     }
134 }
