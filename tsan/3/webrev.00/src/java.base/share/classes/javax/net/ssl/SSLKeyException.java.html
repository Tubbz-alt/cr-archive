29 /**
30  * Reports a bad SSL key.  Normally, this indicates misconfiguration
31  * of the server or client SSL certificate and private key.
32  *
33  * @since 1.4
34  * @author David Brownell
35  */
36 public
37 class SSLKeyException extends SSLException
38 {
39     @java.io.Serial
40     private static final long serialVersionUID = -8071664081941937874L;
41 
42     /**
43      * Constructs an exception reporting a key management error
44      * found by an SSL subsystem.
45      *
46      * @param reason describes the problem.
47      */
48     public SSLKeyException(String reason)
49     {
50         super(reason);
51     }
52 }
