 26 package javax.crypto;
 27 
 28 import java.security.*;
 29 import java.util.Enumeration;
 30 import java.util.Vector;
 31 
 32 /**
 33  * The CryptoAllPermission is a permission that implies
 34  * any other crypto permissions.
 35  * <p>
 36  *
 37  * @see java.security.Permission
 38  * @see java.security.AllPermission
 39  *
 40  * @author Sharon Liu
 41  * @since 1.4
 42  */
 43 
 44 final class CryptoAllPermission extends CryptoPermission {
 45 
 46     @java.io.Serial
 47     private static final long serialVersionUID = -5066513634293192112L;
 48 
 49     // This class is similar to java.security.AllPermission.
 50     static final String ALG_NAME = "CryptoAllPermission";
 51     static final CryptoAllPermission INSTANCE =
 52         new CryptoAllPermission();
 53 
 54     private CryptoAllPermission() {
 55         super(ALG_NAME);
 56     }
 57 
 58     /**
 59      * Checks if the specified permission is implied by
 60      * this object.
 61      *
 62      * @param p the permission to check against.
 63      *
 64      * @return true if the specified permission is an
 65      * instance of CryptoPermission.
 66      */
 67     public boolean implies(Permission p) {
 68          return (p instanceof CryptoPermission);
 69     }
 70 
 71     /**
 72      * Checks two CryptoAllPermission objects for equality.
 73      * Two CryptoAllPermission objects are always equal.
 74      *
 75      * @param obj the object to test for equality with this object.
 76      *
 77      * @return true if <i>obj</i> is a CryptoAllPermission object.
 78      */
 79     public boolean equals(Object obj) {
 80         return (obj == INSTANCE);
 81     }
 82 
 83     /**
 84      *
 85      * Returns the hash code value for this object.
 86      *
 87      * @return a hash code value for this object.
 88      */
 89     public int hashCode() {
 90         return 1;
 91     }
 92 
 93     /**
 94      * Returns a new PermissionCollection object for storing
 95      * CryptoAllPermission objects.
 96      * <p>
 97      *
 98      * @return a new PermissionCollection object suitable for
 99      * storing CryptoAllPermissions.
100      */
101     public PermissionCollection newPermissionCollection() {
102         return new CryptoAllPermissionCollection();
103     }
104 }
105 
106 /**
107  * A CryptoAllPermissionCollection stores a collection
108  * of CryptoAllPermission permissions.
109  *
110  * @see java.security.Permission
111  * @see java.security.Permissions
112  * @see javax.crypto.CryptoPermission
113  *
114  * @author Sharon Liu
115  */
116 final class CryptoAllPermissionCollection extends PermissionCollection
117     implements java.io.Serializable
118 {
119 
120     @java.io.Serial
121     private static final long serialVersionUID = 7450076868380144072L;
122 
123     // true if a CryptoAllPermission has been added
124     private boolean all_allowed;
125 
126     /**
127      * Create an empty CryptoAllPermissions object.
128      */
129     CryptoAllPermissionCollection() {
130         all_allowed = false;
131     }
132 
133     /**
134      * Adds a permission to the CryptoAllPermissions.
135      *
136      * @param permission the Permission object to add.
137      *
138      * @exception SecurityException - if this CryptoAllPermissionCollection
139      * object has been marked readonly
140      */
141     public void add(Permission permission) {
142         if (isReadOnly())
143             throw new SecurityException("attempt to add a Permission to " +
144                                         "a readonly PermissionCollection");
145 
146         if (permission != CryptoAllPermission.INSTANCE)
147             return;
148 
149         all_allowed = true;
150     }
151 
152     /**
153      * Check and see if this set of permissions implies the permissions
154      * expressed in "permission".
155      *
156      * @param permission the Permission object to compare
157      *
158      * @return true if the given permission is implied by this
159      * CryptoAllPermissionCollection.
160      */
161     public boolean implies(Permission permission) {
162         if (!(permission instanceof CryptoPermission)) {
163             return false;
164         }
165         return all_allowed;
166     }
167 
168     /**
169      * Returns an enumeration of all the CryptoAllPermission
170      * objects in the  container.
171      *
172      * @return an enumeration of all the CryptoAllPermission objects.
173      */
174     public Enumeration<Permission> elements() {
175         Vector<Permission> v = new Vector<>(1);
176         if (all_allowed) v.add(CryptoAllPermission.INSTANCE);
177         return v.elements();
178     }
179 }
