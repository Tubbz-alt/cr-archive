29 /**
30  * Certificate Parsing Exception. This is thrown whenever
31  * invalid DER encoded certificate is parsed or unsupported DER features
32  * are found in the Certificate.
33  *
34  * <p><em>Note: The classes in the package {@code javax.security.cert}
35  * exist for compatibility with earlier versions of the
36  * Java Secure Sockets Extension (JSSE). New applications should instead
37  * use the standard Java SE certificate classes located in
38  * {@code java.security.cert}.</em></p>
39  *
40  * @since 1.4
41  * @author Hemma Prafullchandra
42  * @deprecated Use the classes in {@code java.security.cert} instead.
43  */
44 @SuppressWarnings("removal")
45 @Deprecated(since="9", forRemoval=true)
46 public class CertificateParsingException extends CertificateException {
47 
48     @java.io.Serial
49     private static final long serialVersionUID = -8449352422951136229L;
50 
51     /**
52      * Constructs a CertificateParsingException with no detail message. A
53      * detail message is a String that describes this particular
54      * exception.
55      */
56     public CertificateParsingException() {
57         super();
58     }
59 
60     /**
61      * Constructs a CertificateParsingException with the specified detail
62      * message. A detail message is a String that describes this
63      * particular exception.
64      *
65      * @param message the detail message.
66      */
67     public CertificateParsingException(String message) {
68         super(message);
69     }
70 }
