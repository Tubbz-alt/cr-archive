 26 package javax.security.auth.callback;
 27 
 28 /**
 29  * <p> Underlying security services instantiate and pass a
 30  * {@code PasswordCallback} to the {@code handle}
 31  * method of a {@code CallbackHandler} to retrieve password information.
 32  *
 33  * @since 1.4
 34  * @see javax.security.auth.callback.CallbackHandler
 35  */
 36 public class PasswordCallback implements Callback, java.io.Serializable {
 37 
 38     @java.io.Serial
 39     private static final long serialVersionUID = 2267422647454909926L;
 40 
 41     /**
 42      * @serial
 43      * @since 1.4
 44      */
 45     private String prompt;
 46     /**
 47      * @serial
 48      * @since 1.4
 49      */
 50     private boolean echoOn;
 51     /**
 52      * @serial
 53      * @since 1.4
 54      */
 55     private char[] inputPassword;
 56 
 57     /**
 58      * Construct a {@code PasswordCallback} with a prompt
 59      * and a boolean specifying whether the password should be displayed
 60      * as it is being typed.
 61      *
 62      * @param prompt the prompt used to request the password.
 63      *
 64      * @param echoOn true if the password should be displayed
 65      *                  as it is being typed.
 66      *
 67      * @exception IllegalArgumentException if {@code prompt} is null or
 68      *                  if {@code prompt} has a length of 0.
 69      */
 70     public PasswordCallback(String prompt, boolean echoOn) {
 71         if (prompt == null || prompt.isEmpty())
 72             throw new IllegalArgumentException();
 73 
 74         this.prompt = prompt;
 75         this.echoOn = echoOn;
 76     }
 77 
 78     /**
 79      * Get the prompt.
 80      *
 81      * @return the prompt.
 82      */
 83     public String getPrompt() {
 84         return prompt;
 85     }
 86 
 87     /**
 88      * Return whether the password
 89      * should be displayed as it is being typed.
 90      *
 91      * @return the whether the password
 92      *          should be displayed as it is being typed.
 93      */
 94     public boolean isEchoOn() {
 95         return echoOn;
 96     }
 97 
 98     /**
 99      * Set the retrieved password.
100      *
101      * <p> This method makes a copy of the input {@code password}
102      * before storing it.
103      *
104      * @param password the retrieved password, which may be null.
105      *
106      * @see #getPassword
107      */
108     public void setPassword(char[] password) {
109         this.inputPassword = (password == null ? null : password.clone());
110     }
111 
112     /**
113      * Get the retrieved password.
114      *
115      * <p> This method returns a copy of the retrieved password.
116      *
117      * @return the retrieved password, which may be null.
118      *
119      * @see #setPassword
120      */
121     public char[] getPassword() {
122         return (inputPassword == null ? null : inputPassword.clone());
123     }
124 
125     /**
126      * Clear the retrieved password.
127      */
128     public void clearPassword() {
129         if (inputPassword != null) {
130             for (int i = 0; i < inputPassword.length; i++)
131                 inputPassword[i] = ' ';
132         }
133     }
134 }
