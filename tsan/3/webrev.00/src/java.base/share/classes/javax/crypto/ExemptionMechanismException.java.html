26 package javax.crypto;
27 
28 import java.security.GeneralSecurityException;
29 
30 /**
31  * This is the generic ExemptionMechanism exception.
32  *
33  * @since 1.4
34  */
35 
36 public class ExemptionMechanismException extends GeneralSecurityException {
37 
38     @java.io.Serial
39     private static final long serialVersionUID = 1572699429277957109L;
40 
41     /**
42      * Constructs a ExemptionMechanismException with no detailed message.
43      * (A detailed message is a String that describes this particular
44      * exception.)
45      */
46     public ExemptionMechanismException() {
47         super();
48     }
49 
50     /**
51      * Constructs a ExemptionMechanismException with the specified
52      * detailed message. (A detailed message is a String that describes
53      * this particular exception.)
54      *
55      * @param msg the detailed message.
56      */
57    public ExemptionMechanismException(String msg) {
58        super(msg);
59     }
60 }
