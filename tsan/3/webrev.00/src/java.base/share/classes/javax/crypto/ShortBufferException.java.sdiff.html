33  *
34  * @author Jan Luehe
35  *
36  * @since 1.4
37  */
38 
39 public class ShortBufferException extends GeneralSecurityException {
40 

41     private static final long serialVersionUID = 8427718640832943747L;
42 
43     /**

47      */

51 
52     /**
53      * Constructs a ShortBufferException with the specified
54      * detail message.
