26 package sun.nio.ch;
27 
28 import java.io.IOException;
29 import java.net.ProtocolFamily;
30 import java.nio.channels.DatagramChannel;
31 import java.nio.channels.Pipe;
32 import java.nio.channels.ServerSocketChannel;
33 import java.nio.channels.SocketChannel;
34 import java.nio.channels.spi.AbstractSelector;
35 import java.nio.channels.spi.SelectorProvider;
36 
37 public abstract class SelectorProviderImpl
38     extends SelectorProvider
39 {
40     @Override
41     public DatagramChannel openDatagramChannel() throws IOException {
42         return new DatagramChannelImpl(this, /*interruptible*/true);
43     }
44 
45     /**
46      * SelectorProviderImpl specific method to create a DatagramChannel that
47      * is not interruptible.
48      */
49     public DatagramChannel openUninterruptibleDatagramChannel() throws IOException {
50         return new DatagramChannelImpl(this, /*interruptible*/false);
51     }
52 
53     @Override
54     public DatagramChannel openDatagramChannel(ProtocolFamily family) throws IOException {
55         return new DatagramChannelImpl(this, family, /*interruptible*/true);
56     }
57 
58     @Override
59     public Pipe openPipe() throws IOException {
60         return new PipeImpl(this);
61     }
62 
63     @Override
64     public abstract AbstractSelector openSelector() throws IOException;
65 
66     @Override
67     public ServerSocketChannel openServerSocketChannel() throws IOException {
68         return new ServerSocketChannelImpl(this);
69     }
70 
71     @Override
72     public SocketChannel openSocketChannel() throws IOException {
73         return new SocketChannelImpl(this);
74     }
75 }
