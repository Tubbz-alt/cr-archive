28  */

30 package sun.net.www.protocol.ftp;
31 
32 import java.io.IOException;
33 import java.net.URL;
34 import java.net.Proxy;
35 import java.util.Map;
36 import java.util.HashMap;
37 import java.util.Objects;
38 import sun.net.ftp.FtpClient;
39 import sun.net.www.protocol.http.HttpURLConnection;
40 
41 /** open an ftp connection given a URL */
42 public class Handler extends java.net.URLStreamHandler {
43 
44     protected int getDefaultPort() {
45         return 21;
46     }
47 
48     protected boolean equals(URL u1, URL u2) {
49         String userInfo1 = u1.getUserInfo();
50         String userInfo2 = u2.getUserInfo();
51         return super.equals(u1, u2) && Objects.equals(userInfo1, userInfo2);
52     }
53 
54     protected java.net.URLConnection openConnection(URL u)
55         throws IOException {
56         return openConnection(u, null);
57     }
58 
59     protected java.net.URLConnection openConnection(URL u, Proxy p)
60         throws IOException {
61         return new FtpURLConnection(u, p);
62     }
63 }
