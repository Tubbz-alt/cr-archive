26 package sun.text;
27 
28 import sun.text.normalizer.NormalizerBase;
29 
30 public class CollatorUtilities {
31 
32     public static int toLegacyMode(NormalizerBase.Mode mode) {
33         // find the index of the legacy mode in the table;
34         // if it's not there, default to Collator.NO_DECOMPOSITION (0)
35         int legacyMode = legacyModeMap.length;
36         while (legacyMode > 0) {
37             --legacyMode;
38             if (legacyModeMap[legacyMode] == mode) {
39                 break;
40             }
41         }
42         return legacyMode;
43     }
44 
45     public static NormalizerBase.Mode toNormalizerMode(int mode) {
46         NormalizerBase.Mode normalizerMode;
47 
48         try {
49             normalizerMode = legacyModeMap[mode];
50         }
51         catch(ArrayIndexOutOfBoundsException e) {
52             normalizerMode = NormalizerBase.NONE;
53         }
54         return normalizerMode;
55 
56     }
57 
58 
59     static NormalizerBase.Mode[] legacyModeMap = {
60         NormalizerBase.NONE,   // Collator.NO_DECOMPOSITION
61         NormalizerBase.NFD,    // Collator.CANONICAL_DECOMPOSITION
62         NormalizerBase.NFKD,   // Collator.FULL_DECOMPOSITION
63     };
64 
65 }
