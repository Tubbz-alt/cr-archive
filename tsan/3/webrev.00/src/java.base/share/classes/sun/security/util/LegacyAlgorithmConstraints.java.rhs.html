26 package sun.security.util;
27 
28 import java.security.AlgorithmParameters;
29 import java.security.CryptoPrimitive;
30 import java.security.Key;
31 import java.util.List;
32 import java.util.Set;

33 
34 /**
35  * Algorithm constraints for legacy algorithms.
36  */
37 public class LegacyAlgorithmConstraints extends AbstractAlgorithmConstraints {
38 
39     // the known security property, jdk.tls.legacyAlgorithms
40     public static final String PROPERTY_TLS_LEGACY_ALGS =
41             "jdk.tls.legacyAlgorithms";
42 
43     private final List<String> legacyAlgorithms;
44 
45     public LegacyAlgorithmConstraints(String propertyName,
46             AlgorithmDecomposer decomposer) {
47         super(decomposer);
48         legacyAlgorithms = getAlgorithms(propertyName);
49     }
50 
51     @Override
52     public final boolean permits(Set<CryptoPrimitive> primitives,
53             String algorithm, AlgorithmParameters parameters) {
54         if (primitives == null || primitives.isEmpty()) {
55             throw new IllegalArgumentException("The primitives cannot be null" +
56                     " or empty.");
57         }
58         return checkAlgorithm(legacyAlgorithms, algorithm, decomposer);
59     }
60 
61     @Override
62     public final boolean permits(Set<CryptoPrimitive> primitives, Key key) {
63          if (primitives == null || primitives.isEmpty()) {
64             throw new IllegalArgumentException("The primitives cannot be null" +
65             " or empty.");
66         }
67         return true;
68     }
69 
70     @Override
71     public final boolean permits(Set<CryptoPrimitive> primitives,
72             String algorithm, Key key, AlgorithmParameters parameters) {
73         if (primitives == null || primitives.isEmpty()) {
74             throw new IllegalArgumentException("The primitives cannot be null" +
75                     " or empty.");
76         }
77         return checkAlgorithm(legacyAlgorithms, algorithm, decomposer);
78     }
79 
80 }
