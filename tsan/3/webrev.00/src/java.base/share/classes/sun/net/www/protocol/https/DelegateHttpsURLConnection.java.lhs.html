26 package sun.net.www.protocol.https;
27 
28 import java.net.URL;
29 import java.net.Proxy;
30 import java.io.IOException;
31 
32 /**
33  * This class was introduced to provide an additional level of
34  * abstraction between javax.net.ssl.HttpURLConnection and
35  * com.sun.net.ssl.HttpURLConnection objects. <p>
36  *
37  * javax.net.ssl.HttpURLConnection is used in the new sun.net version
38  * of protocol implementation (this one)
39  * com.sun.net.ssl.HttpURLConnection is used in the com.sun version.
40  *
41  */
42 public class DelegateHttpsURLConnection extends AbstractDelegateHttpsURLConnection {
43 
44     // we need a reference to the HttpsURLConnection to get
45     // the properties set there
46     // we also need it to be public so that it can be referenced
47     // from sun.net.www.protocol.http.HttpURLConnection
48     // this is for ResponseCache.put(URI, URLConnection)
49     // second parameter needs to be cast to javax.net.ssl.HttpsURLConnection
50     // instead of AbstractDelegateHttpsURLConnection
51     public javax.net.ssl.HttpsURLConnection httpsURLConnection;
52 
53     DelegateHttpsURLConnection(URL url,
54             sun.net.www.protocol.http.Handler handler,
55             javax.net.ssl.HttpsURLConnection httpsURLConnection)
56             throws IOException {
57         this(url, null, handler, httpsURLConnection);
58     }
59 
60     DelegateHttpsURLConnection(URL url, Proxy p,
61             sun.net.www.protocol.http.Handler handler,
62             javax.net.ssl.HttpsURLConnection httpsURLConnection)
63             throws IOException {
64         super(url, p, handler);
65         this.httpsURLConnection = httpsURLConnection;
66     }
67 
68     protected javax.net.ssl.SSLSocketFactory getSSLSocketFactory() {
69         return httpsURLConnection.getSSLSocketFactory();
70     }
71 
72     protected javax.net.ssl.HostnameVerifier getHostnameVerifier() {
73         return httpsURLConnection.getHostnameVerifier();
74     }
75 
76     /*
77      * Called by layered delegator's finalize() method to handle closing
78      * the underlying object.
79      */
80     @SuppressWarnings("deprecation")
81     protected void dispose() throws Throwable {
82         super.finalize();
83     }
84 }
