 57 /**
 58  * Concrete implementation of the  {@link java.text.spi.NumberFormatProvider
 59  * NumberFormatProvider} class for the JRE LocaleProviderAdapter.
 60  *
 61  * @author Naoto Sato
 62  * @author Masayoshi Okutsu
 63  */
 64 public class NumberFormatProviderImpl extends NumberFormatProvider implements AvailableLanguageTags {
 65 
 66     // Constants used by factory methods to specify a style of format.
 67     private static final int NUMBERSTYLE = 0;
 68     private static final int CURRENCYSTYLE = 1;
 69     private static final int PERCENTSTYLE = 2;
 70     private static final int ACCOUNTINGSTYLE = 3;
 71     private static final int INTEGERSTYLE = 4;
 72 
 73     private final LocaleProviderAdapter.Type type;
 74     private final Set<String> langtags;
 75 
 76     private static Map<String, String> rulesMap =
 77             Arrays.stream(PluralRules.rulesArray).collect(Collectors.toMap(
 78                     sa -> sa[0],
 79                     sa -> sa[1])
 80             );
 81 
 82     public NumberFormatProviderImpl(LocaleProviderAdapter.Type type, Set<String> langtags) {
 83         this.type = type;
 84         this.langtags = langtags;
 85     }

 87     /**
 88      * Returns an array of all locales for which this locale service provider
 89      * can provide localized objects or names.
 90      *
 91      * @return An array of all locales for which this locale service provider
 92      * can provide localized objects or names.
 93      */
 94     @Override
 95     public Locale[] getAvailableLocales() {
 96         return LocaleProviderAdapter.forType(type).getAvailableLocales();
 97     }

 99     @Override
100     public boolean isSupportedLocale(Locale locale) {
101         return LocaleProviderAdapter.forType(type).isSupportedProviderLocale(locale, langtags);
102     }

104     /**
105      * Returns a new <code>NumberFormat</code> instance which formats
106      * monetary values for the specified locale.
107      *
108      * @param locale the desired locale.
109      * @exception NullPointerException if <code>locale</code> is null
110      * @exception IllegalArgumentException if <code>locale</code> isn't
111      *     one of the locales returned from
112      *     {@link java.util.spi.LocaleServiceProvider#getAvailableLocales()
113      *     getAvailableLocales()}.
114      * @return a currency formatter
115      * @see java.text.NumberFormat#getCurrencyInstance(java.util.Locale)
116      */
117     @Override
118     public NumberFormat getCurrencyInstance(Locale locale) {
119         return getInstance(locale, CURRENCYSTYLE);
120     }
121 
122     /**
123      * Returns a new <code>NumberFormat</code> instance which formats
124      * integer values for the specified locale.
125      * The returned number format is configured to
126      * round floating point numbers to the nearest integer using
127      * half-even rounding (see {@link java.math.RoundingMode#HALF_EVEN HALF_EVEN})
128      * for formatting, and to parse only the integer part of
129      * an input string (see {@link
130      * java.text.NumberFormat#isParseIntegerOnly isParseIntegerOnly}).
131      *
132      * @param locale the desired locale
133      * @exception NullPointerException if <code>locale</code> is null
134      * @exception IllegalArgumentException if <code>locale</code> isn't
135      *     one of the locales returned from
136      *     {@link java.util.spi.LocaleServiceProvider#getAvailableLocales()
137      *     getAvailableLocales()}.
138      * @return a number format for integer values
139      * @see java.text.NumberFormat#getIntegerInstance(java.util.Locale)
140      */
141     @Override
142     public NumberFormat getIntegerInstance(Locale locale) {
143         return getInstance(locale, INTEGERSTYLE);
144     }
145 
146     /**
147      * Returns a new general-purpose <code>NumberFormat</code> instance for
148      * the specified locale.
149      *
150      * @param locale the desired locale
151      * @exception NullPointerException if <code>locale</code> is null
152      * @exception IllegalArgumentException if <code>locale</code> isn't
153      *     one of the locales returned from
154      *     {@link java.util.spi.LocaleServiceProvider#getAvailableLocales()
155      *     getAvailableLocales()}.
156      * @return a general-purpose number formatter
157      * @see java.text.NumberFormat#getNumberInstance(java.util.Locale)
158      */
159     @Override
160     public NumberFormat getNumberInstance(Locale locale) {
161         return getInstance(locale, NUMBERSTYLE);
162     }
163 
164     /**
165      * Returns a new <code>NumberFormat</code> instance which formats
166      * percentage values for the specified locale.
167      *
168      * @param locale the desired locale
169      * @exception NullPointerException if <code>locale</code> is null
170      * @exception IllegalArgumentException if <code>locale</code> isn't
171      *     one of the locales returned from
172      *     {@link java.util.spi.LocaleServiceProvider#getAvailableLocales()
173      *     getAvailableLocales()}.
174      * @return a percent formatter
175      * @see java.text.NumberFormat#getPercentInstance(java.util.Locale)
176      */
177     @Override
178     public NumberFormat getPercentInstance(Locale locale) {
179         return getInstance(locale, PERCENTSTYLE);
180     }
181 
182     private NumberFormat getInstance(Locale locale,
183                                             int choice) {
184         if (locale == null) {
185             throw new NullPointerException();
186         }
187 
188         // Check for region override
189         Locale override = locale.getUnicodeLocaleType("nu") == null ?
190             CalendarDataUtility.findRegionOverride(locale) :
191             locale;
192 
193         LocaleProviderAdapter adapter = LocaleProviderAdapter.forType(type);
194         String[] numberPatterns = adapter.getLocaleResources(override).getNumberPatterns();
195         DecimalFormatSymbols symbols = DecimalFormatSymbols.getInstance(override);
196         int entry = (choice == INTEGERSTYLE) ? NUMBERSTYLE : choice;
197         if (choice == CURRENCYSTYLE &&
198             numberPatterns.length > ACCOUNTINGSTYLE &&
199             !numberPatterns[ACCOUNTINGSTYLE].isEmpty() &&
200             "account".equalsIgnoreCase(override.getUnicodeLocaleType("cf"))) {
201             entry = ACCOUNTINGSTYLE;
202         }
203         DecimalFormat format = new DecimalFormat(numberPatterns[entry], symbols);
204 
205         if (choice == INTEGERSTYLE) {
206             format.setMaximumFractionDigits(0);
207             format.setDecimalSeparatorAlwaysShown(false);
208             format.setParseIntegerOnly(true);
209         } else if (choice == CURRENCYSTYLE) {
210             adjustForCurrencyDefaultFractionDigits(format, symbols);
211         }
212 
213         return format;
214     }
215 
216     /**
217      * Adjusts the minimum and maximum fraction digits to values that
218      * are reasonable for the currency's default fraction digits.
219      */
220     private static void adjustForCurrencyDefaultFractionDigits(
221             DecimalFormat format, DecimalFormatSymbols symbols) {
222         Currency currency = symbols.getCurrency();
223         if (currency == null) {
224             try {
225                 currency = Currency.getInstance(symbols.getInternationalCurrencySymbol());
226             } catch (IllegalArgumentException e) {
227             }
228         }
229         if (currency != null) {
230             int digits = currency.getDefaultFractionDigits();
231             if (digits != -1) {
232                 int oldMinDigits = format.getMinimumFractionDigits();
233                 // Common patterns are "#.##", "#.00", "#".
234                 // Try to adjust all of them in a reasonable way.
235                 if (oldMinDigits == format.getMaximumFractionDigits()) {
236                     format.setMinimumFractionDigits(digits);
237                     format.setMaximumFractionDigits(digits);
238                 } else {
239                     format.setMinimumFractionDigits(Math.min(digits, oldMinDigits));
240                     format.setMaximumFractionDigits(digits);
241                 }
242             }
243         }
244     }

246     /**
247      * Returns a new {@code NumberFormat} instance which formats
248      * a number in its compact form for the specified
249      * {@code locale} and {@code formatStyle}.
250      *
251      * @param locale the desired locale
252      * @param formatStyle the style for formatting a number
253      * @throws NullPointerException if {@code locale} or {@code formatStyle}
254      *     is {@code null}
255      * @throws IllegalArgumentException if {@code locale} isn't
256      *     one of the locales returned from
257      *     {@link java.util.spi.LocaleServiceProvider#getAvailableLocales()
258      *     getAvailableLocales()}.
259      * @return a compact number formatter
260      *
261      * @see java.text.NumberFormat#getCompactNumberInstance(Locale,
262      *                      NumberFormat.Style)
263      * @since 12
264      */
265     @Override
266     public NumberFormat getCompactNumberInstance(Locale locale,
267             NumberFormat.Style formatStyle) {
268 
269         Objects.requireNonNull(locale);
270         Objects.requireNonNull(formatStyle);
271 
272         // Check for region override
273         Locale override = locale.getUnicodeLocaleType("nu") == null
274                 ? CalendarDataUtility.findRegionOverride(locale)
275                 : locale;
276 
277         LocaleProviderAdapter adapter = LocaleProviderAdapter.forType(type);
278         LocaleResources resource = adapter.getLocaleResources(override);
279 
280         String[] numberPatterns = resource.getNumberPatterns();
281         DecimalFormatSymbols symbols = DecimalFormatSymbols.getInstance(override);
282         String[] cnPatterns = resource.getCNPatterns(formatStyle);
283 
284         // plural rules
285         String pluralRules = rulesMap.getOrDefault(override.toString(),
286                 rulesMap.getOrDefault(override.getLanguage(), ""));
287 
288         CompactNumberFormat format = new CompactNumberFormat(numberPatterns[0],
289                 symbols, cnPatterns, pluralRules);
290         return format;
291     }
292 
293     @Override
294     public Set<String> getAvailableLanguageTags() {
295         return langtags;
296     }
