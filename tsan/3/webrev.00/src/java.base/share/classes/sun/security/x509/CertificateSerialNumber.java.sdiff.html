184     /**
185      * Generates a new random serial number.
186      */
187     public static CertificateSerialNumber newRandom64bit(Random rand) {
188         while (true) {
189             BigInteger b = new BigInteger(64, rand);
190             if (b.signum() != 0) {
191                 return new CertificateSerialNumber(b);
192             }
193         }
