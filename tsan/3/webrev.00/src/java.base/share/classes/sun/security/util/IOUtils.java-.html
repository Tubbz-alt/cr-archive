26 /**
27  * IOUtils: A collection of IO-related public static methods.
28  */
29 
30 package sun.security.util;
31 
32 import java.io.EOFException;
33 import java.io.IOException;
34 import java.io.InputStream;
35 import java.util.Arrays;
36 
37 public class IOUtils {
38 
39     /**
40      * Read up to {@code length} of bytes from {@code in}
41      * until EOF is detected.
42      * @param is input stream, must not be null
43      * @param length number of bytes to read
44      * @param readAll if true, an EOFException will be thrown if not enough
45      *        bytes are read.
46      * @return bytes read
47      * @throws IOException Any IO error or a premature EOF is detected
48      */
49     public static byte[] readFully(InputStream is, int length, boolean readAll)
50             throws IOException {
51         if (length < 0) {
52             throw new IOException("Invalid length");
53         }
54         byte[] output = {};
55         int pos = 0;
56         while (pos < length) {
57             int bytesToRead;
58             if (pos >= output.length) { // Only expand when there's no room
59                 bytesToRead = Math.min(length - pos, output.length + 1024);
60                 if (output.length < pos + bytesToRead) {
61                     output = Arrays.copyOf(output, pos + bytesToRead);
62                 }
63             } else {
64                 bytesToRead = output.length - pos;
65             }
66             int cc = is.read(output, pos, bytesToRead);
67             if (cc < 0) {
68                 if (readAll) {
69                     throw new EOFException("Detect premature EOF");
70                 } else {
71                     if (output.length != pos) {
72                         output = Arrays.copyOf(output, pos);
73                     }
74                     break;
75                 }
76             }
77             pos += cc;
78         }
79         return output;
80     }
81 
82     /**
83      * Read {@code length} of bytes from {@code in}. An exception is
84      * thrown if there are not enough bytes in the stream.
85      *
86      * @param is input stream, must not be null
87      * @param length number of bytes to read, must not be negative
88      * @return bytes read
89      * @throws IOException if any IO error or a premature EOF is detected, or
90      *      if {@code length} is negative since this length is usually also
91      *      read from {@code is}.
92      */
93     public static byte[] readNBytes(InputStream is, int length) throws IOException {
94         if (length < 0) {
95             throw new IOException("length cannot be negative: " + length);
96         }
97         return readFully(is, length, true);
98     }
99 }
