 34 /**
 35  * The JSSE provider.
 36  *
 37  * The RSA implementation has been removed from JSSE, but we still need to
 38  * register the same algorithms for compatibility. We just point to the RSA
 39  * implementation in the SunRsaSign provider. This works because all classes
 40  * are in the bootclasspath and therefore loaded by the same classloader.
 41  *
 42  * SunJSSE now supports an experimental FIPS compliant mode when used with an
 43  * appropriate FIPS certified crypto provider. In FIPS mode, we:
 44  *  . allow only TLS 1.0 or later
 45  *  . allow only FIPS approved ciphersuites
 46  *  . perform all crypto in the FIPS crypto provider
 47  *
 48  * It is currently not possible to use both FIPS compliant SunJSSE and
 49  * standard JSSE at the same time because of the various static data structures
 50  * we use.
 51  *
 52  * However, we do want to allow FIPS mode to be enabled at runtime and without
 53  * editing the java.security file. That means we need to allow
 54  * Security.removeProvider("SunJSSE") to work, which creates an instance of
 55  * this class in non-FIPS mode. That is why we delay the selection of the mode
 56  * as long as possible. This is until we open an SSL/TLS connection and the
 57  * data structures need to be initialized or until SunJSSE is initialized in
 58  * FIPS mode.
 59  *
 60  */
 61 public class SunJSSE extends java.security.Provider {
 62 

 63     private static final long serialVersionUID = 3231825739635378733L;
 64 
 65     private static String info = "Sun JSSE provider" +
 66         "(PKCS12, SunX509/PKIX key/trust factories, " +
 67         "SSLv3/TLSv1/TLSv1.1/TLSv1.2/TLSv1.3/DTLSv1.0/DTLSv1.2)";
 68 
 69     public SunJSSE() {
 70         super("SunJSSE", PROVIDER_VER, info);
 71         registerAlgorithms();
 72     }
 73 
 74     private void registerAlgorithms() {
 75         AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
 76             doRegister();
 77             return null;
 78         });
 79     }
 80 
 81     private void ps(String type, String algo, String cn,
 82             List<String> aliases, HashMap<String, String> attrs) {
 83         putService(new Provider.Service(this, type, algo, cn, aliases, attrs));
 84     }
 85 
 86     private void doRegister() {
 87         Iterator<Provider.Service> rsaIter =
 88             new SunRsaSignEntries(this).iterator();
 89         while (rsaIter.hasNext()) {
 90             putService(rsaIter.next());
 91         }
 92 
 93         ps("Signature", "MD5andSHA1withRSA",
 94             "sun.security.ssl.RSASignature", null, null);
 95 
 96         ps("KeyManagerFactory", "SunX509",
 97             "sun.security.ssl.KeyManagerFactoryImpl$SunX509", null, null);
 98         ps("KeyManagerFactory", "NewSunX509",
 99             "sun.security.ssl.KeyManagerFactoryImpl$X509",
100             createAliases("PKIX"), null);
101 
102         ps("TrustManagerFactory", "SunX509",
103             "sun.security.ssl.TrustManagerFactoryImpl$SimpleFactory",
104             null, null);
105         ps("TrustManagerFactory", "PKIX",
106             "sun.security.ssl.TrustManagerFactoryImpl$PKIXFactory",
107             createAliases("SunPKIX", "X509", "X.509"), null);
108 
109         ps("SSLContext", "TLSv1",
110             "sun.security.ssl.SSLContextImpl$TLS10Context",
111             createAliases("SSLv3"), null);
