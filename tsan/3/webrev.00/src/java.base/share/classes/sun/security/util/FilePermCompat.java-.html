26 package sun.security.util;
27 
28 import sun.security.action.GetPropertyAction;
29 
30 import java.io.FilePermission;
31 import java.security.Permission;
32 import jdk.internal.access.SharedSecrets;
33 
34 /**
35  * Take care of FilePermission compatibility after JDK-8164705.
36  */
37 public class FilePermCompat {
38     /**
39      * New behavior? Keep compatibility? Both default true.
40      */
41     public static final boolean nb;
42     public static final boolean compat;
43 
44     static {
45         String flag = GetPropertyAction.privilegedGetProperty(
46                 "jdk.io.permissionsUseCanonicalPath", "false");
47         switch (flag) {
48             case "true":
49                 nb = false;
50                 compat = false;
51                 break;
52             case "false":
53                 nb = true;
54                 compat = true;
55                 break;
56             default:
57                 throw new RuntimeException(
58                         "Invalid jdk.io.permissionsUseCanonicalPath: " + flag);
59         }
60     }
61 
62     public static Permission newPermPlusAltPath(Permission input) {
63         if (compat && input instanceof FilePermission) {
64             return SharedSecrets.getJavaIOFilePermissionAccess()
65                     .newPermPlusAltPath((FilePermission) input);
66         }
67         return input;
68     }
69 
70     public static Permission newPermUsingAltPath(Permission input) {
71         if (input instanceof FilePermission) {
72             return SharedSecrets.getJavaIOFilePermissionAccess()
73                     .newPermUsingAltPath((FilePermission) input);
74         }
75         return null;
76     }
77 }
