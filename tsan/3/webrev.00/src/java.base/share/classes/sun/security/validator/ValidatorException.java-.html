 26 package sun.security.validator;
 27 
 28 import java.security.cert.*;
 29 
 30 /**
 31  * ValidatorException thrown by the Validator. It has optional fields that
 32  * allow better error diagnostics.
 33  *
 34  * @author Andreas Sterbenz
 35  */
 36 public class ValidatorException extends CertificateException {
 37 
 38     private static final long serialVersionUID = -2836879718282292155L;
 39 
 40     public static final Object T_NO_TRUST_ANCHOR =
 41         "No trusted certificate found";
 42 
 43     public static final Object T_EE_EXTENSIONS =
 44         "End entity certificate extension check failed";
 45 
 46     public static final Object T_CA_EXTENSIONS =
 47         "CA certificate extension check failed";
 48 
 49     public static final Object T_CERT_EXPIRED =
 50         "Certificate expired";
 51 
 52     public static final Object T_SIGNATURE_ERROR =
 53         "Certificate signature validation failed";
 54 
 55     public static final Object T_NAME_CHAINING =
 56         "Certificate chaining error";
 57 
 58     public static final Object T_ALGORITHM_DISABLED =
 59         "Certificate signature algorithm disabled";
 60 
 61     public static final Object T_UNTRUSTED_CERT =
 62         "Untrusted certificate";
 63 
 64     private Object type;
 65     private X509Certificate cert;
 66 
 67     public ValidatorException(String msg) {
 68         super(msg);
 69     }
 70 
 71     public ValidatorException(String msg, Throwable cause) {
 72         super(msg);
 73         initCause(cause);
 74     }
 75 
 76     public ValidatorException(Object type) {
 77         this(type, null);
 78     }
 79 
 80     public ValidatorException(Object type, X509Certificate cert) {
 81         super((String)type);
 82         this.type = type;
 83         this.cert = cert;
 84     }
 85 
 86     public ValidatorException(Object type, X509Certificate cert,
 87             Throwable cause) {
 88         this(type, cert);
 89         initCause(cause);
 90     }
 91 
 92     public ValidatorException(String msg, Object type, X509Certificate cert) {
 93         super(msg);
 94         this.type = type;
 95         this.cert = cert;
 96     }
 97 
 98     public ValidatorException(String msg, Object type, X509Certificate cert,
 99             Throwable cause) {
100         this(msg, type, cert);
101         initCause(cause);
102     }
103 
104     /**
105      * Get the type of the failure (one of the T_XXX constants), if
106      * available. This may be helpful when designing a user interface.
107      */
108     public Object getErrorType() {
109         return type;
110     }
111 
112     /**
113      * Get the certificate causing the exception, if available.
114      */
115     public X509Certificate getErrorCertificate() {
116         return cert;
117     }
118 
119 }
