26 package sun.security.rsa;
27 
28 import java.util.*;
29 
30 import java.security.*;
31 
32 import static sun.security.util.SecurityConstants.PROVIDER_VER;
33 
34 /**
35  * Provider class for the RSA signature provider. Supports RSA keyfactory,
36  * keypair generation, and RSA signatures.
37  *
38  * @since   1.5
39  * @author  Andreas Sterbenz
40  */
41 public final class SunRsaSign extends Provider {
42 
43     @java.io.Serial
44     private static final long serialVersionUID = 866040293550393045L;
45 
46     public SunRsaSign() {
47         super("SunRsaSign", PROVIDER_VER, "Sun RSA signature provider");
48 
49         Provider p = this;
50         Iterator<Provider.Service> serviceIter = new SunRsaSignEntries(p).iterator();
51 
52         if (System.getSecurityManager() == null) {
53             putEntries(serviceIter);
54         } else {
55             AccessController.doPrivileged(new PrivilegedAction<Void>() {
56                 @Override
57                 public Void run() {
58                     putEntries(serviceIter);
59                     return null;
60                 }
61             });
62         }
63     }
64     void putEntries(Iterator<Provider.Service> i) {
65         while (i.hasNext()) {
66             putService(i.next());
67         }
68     }
69 }
