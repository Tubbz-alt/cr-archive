26 package sun.security.provider;
27 
28 import java.util.*;
29 import java.security.*;
30 
31 import static sun.security.util.SecurityConstants.PROVIDER_VER;
32 
33 
34 /**
35  * The SUN Security Provider.
36  *
37  */
38 public final class Sun extends Provider {
39 
40     @java.io.Serial
41     private static final long serialVersionUID = 6440182097568097204L;
42 
43     private static final String INFO = "SUN " +
44     "(DSA key/parameter generation; DSA signing; SHA-1, MD5 digests; " +
45     "SecureRandom; X.509 certificates; PKCS12, JKS & DKS keystores; " +
46     "PKIX CertPathValidator; " +
47     "PKIX CertPathBuilder; LDAP, Collection CertStores, JavaPolicy Policy; " +
48     "JavaLoginConfig Configuration)";
49 
50     public Sun() {
51         /* We are the SUN provider */
52         super("SUN", PROVIDER_VER, INFO);
53 
54         Provider p = this;
55         Iterator<Provider.Service> serviceIter = new SunEntries(p).iterator();
56 
57         // if there is no security manager installed, put directly into
58         // the provider
59         if (System.getSecurityManager() == null) {
60             putEntries(serviceIter);
61         } else {
62             AccessController.doPrivileged(new PrivilegedAction<Void>() {
63                 @Override
64                 public Void run() {
65                     putEntries(serviceIter);
66                     return null;
67                 }
68             });
69         }
70     }
71 
72     void putEntries(Iterator<Provider.Service> i) {
73         while (i.hasNext()) {
74             putService(i.next());
75         }
76     }
77 }
