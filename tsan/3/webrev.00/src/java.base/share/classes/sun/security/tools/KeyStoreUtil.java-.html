 66     private static final Collator collator = Collator.getInstance();
 67     static {
 68         // this is for case insensitive string comparisons
 69         collator.setStrength(Collator.PRIMARY);

 72     /**
 73      * Returns true if the certificate is self-signed, false otherwise.
 74      */
 75     public static boolean isSelfSigned(X509Certificate cert) {
 76         return signedBy(cert, cert);
 77     }
 78 
 79     public static boolean signedBy(X509Certificate end, X509Certificate ca) {
 80         if (!ca.getSubjectX500Principal().equals(end.getIssuerX500Principal())) {
 81             return false;
 82         }
 83         try {
 84             end.verify(ca.getPublicKey());
 85             return true;
 86         } catch (Exception e) {
 87             return false;
 88         }
 89     }

 91     /**
 92      * Returns true if KeyStore has a password. This is true except for
 93      * MSCAPI KeyStores
 94      */
 95     public static boolean isWindowsKeyStore(String storetype) {
 96         return storetype != null
 97                 && (storetype.equalsIgnoreCase("Windows-MY")
 98                     || storetype.equalsIgnoreCase("Windows-ROOT"));
 99     }
100 
101     /**
102      * Returns standard-looking names for storetype
103      */
104     public static String niceStoreTypeName(String storetype) {
105         if (storetype.equalsIgnoreCase("Windows-MY")) {
106             return "Windows-MY";
107         } else if(storetype.equalsIgnoreCase("Windows-ROOT")) {
108             return "Windows-ROOT";
109         } else {
110             return storetype.toUpperCase(Locale.ENGLISH);
111         }
112     }

114     /**
115      * Returns the file name of the keystore with the configured CA certificates.
116      */
117     public static String getCacerts() {
118         String sep = File.separator;
119         return System.getProperty("java.home") + sep
120                 + "lib" + sep + "security" + sep
121                 + "cacerts";
122     }
123 
124     /**
125      * Returns the keystore with the configured CA certificates.
126      */
127     public static KeyStore getCacertsKeyStore() throws Exception {
128         File file = new File(getCacerts());
129         if (!file.exists()) {
130             return null;
131         }
132         return KeyStore.getInstance(file, (char[])null);
133     }

135     public static char[] getPassWithModifier(String modifier, String arg,
136                                              java.util.ResourceBundle rb) {
137         if (modifier == null) {
138             return arg.toCharArray();
139         } else if (collator.compare(modifier, "env") == 0) {
140             String value = System.getenv(arg);
141             if (value == null) {
142                 System.err.println(rb.getString(
143                         "Cannot.find.environment.variable.") + arg);
144                 return null;
145             } else {
146                 return value.toCharArray();
147             }
148         } else if (collator.compare(modifier, "file") == 0) {
149             try {
150                 URL url = null;
151                 try {
152                     url = new URL(arg);
153                 } catch (java.net.MalformedURLException mue) {
154                     File f = new File(arg);
155                     if (f.exists()) {
156                         url = f.toURI().toURL();
157                     } else {
158                         System.err.println(rb.getString(
159                                 "Cannot.find.file.") + arg);
160                         return null;
161                     }
162                 }
163 
164                 try (BufferedReader br =
165                      new BufferedReader(new InputStreamReader(
166                          url.openStream()))) {
167                     String value = br.readLine();
168 
169                     if (value == null) {
170                         return new char[0];
171                     }
172 
173                     return value.toCharArray();
174                 }
175             } catch (IOException ioe) {
176                 System.err.println(ioe);
177                 return null;
178             }
179         } else {
180             System.err.println(rb.getString("Unknown.password.type.") +
181                     modifier);
182             return null;
183         }
184     }

186     /**
187      * Parses a option line likes
188      *    -genkaypair -dname "CN=Me"
189      * and add the results into a list
190      * @param list the list to fill into
191      * @param s the line
192      */
193     private static void parseArgsLine(List<String> list, String s)
194             throws IOException, PropertyExpander.ExpandException {
195         StreamTokenizer st = new StreamTokenizer(new StringReader(s));
196 
197         st.resetSyntax();
198         st.whitespaceChars(0x00, 0x20);
199         st.wordChars(0x21, 0xFF);
200         // Everything is a word char except for quotation and apostrophe
201         st.quoteChar('"');
202         st.quoteChar('\'');
203 
204         while (true) {
205             if (st.nextToken() == StreamTokenizer.TT_EOF) {
206                 break;
207             }
208             list.add(PropertyExpander.expand(st.sval));
209         }
210     }

212     /**
213      * Prepends matched options from a pre-configured options file.
214      *
215      * @param tool the name of the tool, can be "keytool" or "jarsigner"
216      * @param file the pre-configured options file
217      * @param c1 the name of the command, with the "-" prefix,
218      *        must not be null
219      * @param c2 the alternative command name, with the "-" prefix,
220      *        null if none. For example, "genkey" is alt name for
221      *        "genkeypair". A command can only have one alt name now.
222      * @param args existing arguments
223      * @return arguments combined
224      * @throws IOException if there is a file I/O or format error
225      * @throws PropertyExpander.ExpandException
226      *         if there is a property expansion error
227      */
228     public static String[] expandArgs(String tool, String file,
229                     String c1, String c2, String[] args)
230             throws IOException, PropertyExpander.ExpandException {
231 
232         List<String> result = new ArrayList<>();
233         Properties p = new Properties();
234         p.load(new FileInputStream(file));
235 
236         String s = p.getProperty(tool + ".all");
237         if (s != null) {
238             parseArgsLine(result, s);
239         }
240 
241         // Cannot provide both -genkey and -genkeypair
242         String s1 = p.getProperty(tool + "." + c1.substring(1));
243         String s2 = null;
244         if (c2 != null) {
245             s2 = p.getProperty(tool + "." + c2.substring(1));
246         }
247         if (s1 != null && s2 != null) {
248             throw new IOException("Cannot have both " + c1 + " and "
249                     + c2 + " as pre-configured options");
250         }
251         if (s1 == null) {
252             s1 = s2;
253         }
254         if (s1 != null) {
255             parseArgsLine(result, s1);
256         }
257 
258         if (result.isEmpty()) {
259             return args;
260         } else {
261             result.addAll(Arrays.asList(args));
262             return result.toArray(new String[result.size()]);
263         }
264     }

266     /**
267      * Loads a security provider as a service.
268      *
269      * @param provName the name
270      * @param arg optional arg
271      * @throws IllegalArgumentException if no provider matches the name
272      */
273     public static void loadProviderByName(String provName, String arg) {
274         Provider loaded = Security.getProvider(provName);
275         if (loaded != null) {
276             if (arg != null) {
277                 loaded = loaded.configure(arg);
278                 Security.addProvider(loaded);
279             }
280             return;
281         }
282         for (Provider p : ServiceLoader.load(Provider.class,
283                 ClassLoader.getSystemClassLoader())) {
284             if (p.getName().equals(provName)) {
285                 if (arg != null) {
286                     p = p.configure(arg);
287                 }
288                 Security.addProvider(p);
289                 return;
290             }
291         }
292         throw new IllegalArgumentException("No provider found");
293     }

295     /**
296      * Loads a security provider by a fully-qualified class name.
297      *
298      * @param provClass the class name
299      * @param arg optional arg
300      * @param cl optional class loader
301      * @throws IllegalArgumentException if no provider matches the class name
302      * @throws ClassCastException if the class has not extended Provider
303      */
304     public static void loadProviderByClass(
305             String provClass, String arg, ClassLoader cl) {
306 
307         // For compatibility, SunPKCS11 and OracleUcrypto can still be
308         // loadable with -providerClass.
309         if (provClass.equals("sun.security.pkcs11.SunPKCS11")) {
310             loadProviderByName("SunPKCS11", arg);
311             return;
312         } else if (provClass.equals("com.oracle.security.crypto.UcryptoProvider")) {
313             loadProviderByName("OracleUcrypto", arg);
314             return;
315         }
316 
317         Provider prov;
318         try {
319             Class<?> clazz = Class.forName(provClass, false, cl);
320             prov = (Provider) clazz.getConstructor().newInstance();
321         } catch (ReflectiveOperationException e) {
322             throw new IllegalArgumentException(e);
323         }
324         if (arg != null) {
325             prov = prov.configure(arg);
326         }
327         Security.addProvider(prov);
328     }
