 76     /**
 77      * The "ServerKeyExchange" handshake message consumer.
 78      */
 79     private static final
 80             class ServerKeyExchangeConsumer implements SSLConsumer {
 81         // Prevent instantiation of this class.
 82         private ServerKeyExchangeConsumer() {
 83             // blank
 84         }
 85 
 86         @Override
 87         public void consume(ConnectionContext context,
 88                 ByteBuffer message) throws IOException {
 89             // The consuming happens in client side only.
 90             ClientHandshakeContext chc = (ClientHandshakeContext)context;
 91 
 92             // clean up this consumer
 93             chc.handshakeConsumers.remove(SSLHandshake.SERVER_KEY_EXCHANGE.id);
 94 









 95             SSLKeyExchange ke = SSLKeyExchange.valueOf(
 96                     chc.negotiatedCipherSuite.keyExchange,
 97                     chc.negotiatedProtocol);
 98             if (ke != null) {
 99                 for (Map.Entry<Byte, SSLConsumer> hc :
100                         ke.getHandshakeConsumers(chc)) {
101                     if (hc.getKey() == SSLHandshake.SERVER_KEY_EXCHANGE.id) {
102                         hc.getValue().consume(context, message);
103                         return;
104                     }
105                 }
106             }
107 
108             // no consumer defined.
109             throw chc.conContext.fatal(Alert.UNEXPECTED_MESSAGE,
110                         "Unexpected ServerKeyExchange handshake message.");
111         }
112     }
113 }

 76     /**
 77      * The "ServerKeyExchange" handshake message consumer.
 78      */
 79     private static final
 80             class ServerKeyExchangeConsumer implements SSLConsumer {
 81         // Prevent instantiation of this class.
 82         private ServerKeyExchangeConsumer() {
 83             // blank
 84         }
 85 
 86         @Override
 87         public void consume(ConnectionContext context,
 88                 ByteBuffer message) throws IOException {
 89             // The consuming happens in client side only.
 90             ClientHandshakeContext chc = (ClientHandshakeContext)context;
 91 
 92             // clean up this consumer
 93             chc.handshakeConsumers.remove(SSLHandshake.SERVER_KEY_EXCHANGE.id);
 94 
 95             SSLConsumer certStatCons = chc.handshakeConsumers.remove(
 96                     SSLHandshake.CERTIFICATE_STATUS.id);
 97             if (certStatCons != null) {
 98                 // Stapling was active but no certificate status message
 99                 // was sent.  We need to run the absence handler which will
100                 // check the certificate chain.
101                 CertificateStatus.handshakeAbsence.absent(context, null);
102             }
103 
104             SSLKeyExchange ke = SSLKeyExchange.valueOf(
105                     chc.negotiatedCipherSuite.keyExchange,
106                     chc.negotiatedProtocol);
107             if (ke != null) {
108                 for (Map.Entry<Byte, SSLConsumer> hc :
109                         ke.getHandshakeConsumers(chc)) {
110                     if (hc.getKey() == SSLHandshake.SERVER_KEY_EXCHANGE.id) {
111                         hc.getValue().consume(context, message);
112                         return;
113                     }
114                 }
115             }
116 
117             // no consumer defined.
118             throw chc.conContext.fatal(Alert.UNEXPECTED_MESSAGE,
119                         "Unexpected ServerKeyExchange handshake message.");
120         }
121     }
122 }
