 26 package sun.security.ssl;
 27 
 28 import java.security.*;
 29 import java.util.*;
 30 import static sun.security.util.SecurityConstants.PROVIDER_VER;
 31 import static sun.security.provider.SunEntries.createAliases;
 32 
 33 /**
 34  * The JSSE provider.
 35  *
 36  * SunJSSE now supports an experimental FIPS compliant mode when used with an
 37  * appropriate FIPS certified crypto provider. In FIPS mode, we:
 38  *  . allow only TLS 1.0 or later
 39  *  . allow only FIPS approved ciphersuites
 40  *  . perform all crypto in the FIPS crypto provider
 41  *
 42  * It is currently not possible to use both FIPS compliant SunJSSE and
 43  * standard JSSE at the same time because of the various static data structures
 44  * we use.
 45  *
 46  * However, we do want to allow FIPS mode to be enabled at runtime and without
 47  * editing the java.security file. That means we need to allow
 48  * Security.removeProvider("SunJSSE") to work, which creates an instance of
 49  * this class in non-FIPS mode. That is why we delay the selection of the mode
 50  * as long as possible. This is until we open an SSL/TLS connection and the
 51  * data structures need to be initialized or until SunJSSE is initialized in
 52  * FIPS mode.
 53  *
 54  */
 55 public class SunJSSE extends java.security.Provider {
 56 
 57     @java.io.Serial
 58     private static final long serialVersionUID = 3231825739635378733L;
 59 
 60     private static String info = "Sun JSSE provider" +
 61         "(PKCS12, SunX509/PKIX key/trust factories, " +
 62         "SSLv3/TLSv1/TLSv1.1/TLSv1.2/TLSv1.3/DTLSv1.0/DTLSv1.2)";
 63 
 64     public SunJSSE() {
 65         super("SunJSSE", PROVIDER_VER, info);
 66         registerAlgorithms();
 67     }
 68 
 69     private void registerAlgorithms() {
 70         AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
 71             doRegister();
 72             return null;
 73         });
 74     }
 75 
 76     private void ps(String type, String algo, String cn,
 77             List<String> aliases, HashMap<String, String> attrs) {
 78         putService(new Provider.Service(this, type, algo, cn, aliases, attrs));
 79     }
 80 
 81     private void doRegister() {
 82         ps("Signature", "MD5andSHA1withRSA",
 83             "sun.security.ssl.RSASignature", null, null);
 84 
 85         ps("KeyManagerFactory", "SunX509",
 86             "sun.security.ssl.KeyManagerFactoryImpl$SunX509", null, null);
 87         ps("KeyManagerFactory", "NewSunX509",
 88             "sun.security.ssl.KeyManagerFactoryImpl$X509",
 89             createAliases("PKIX"), null);
 90 
 91         ps("TrustManagerFactory", "SunX509",
 92             "sun.security.ssl.TrustManagerFactoryImpl$SimpleFactory",
 93             null, null);
 94         ps("TrustManagerFactory", "PKIX",
 95             "sun.security.ssl.TrustManagerFactoryImpl$PKIXFactory",
 96             createAliases("SunPKIX", "X509", "X.509"), null);
 97 
 98         ps("SSLContext", "TLSv1",
 99             "sun.security.ssl.SSLContextImpl$TLS10Context",
100             createAliases("SSLv3"), null);
101         ps("SSLContext", "TLSv1.1",
102             "sun.security.ssl.SSLContextImpl$TLS11Context", null, null);
103         ps("SSLContext", "TLSv1.2",
104             "sun.security.ssl.SSLContextImpl$TLS12Context", null, null);
105         ps("SSLContext", "TLSv1.3",
106             "sun.security.ssl.SSLContextImpl$TLS13Context", null, null);
107         ps("SSLContext", "TLS",
108             "sun.security.ssl.SSLContextImpl$TLSContext",
109             createAliases("SSL"), null);
110 
111         ps("SSLContext", "DTLSv1.0",
112             "sun.security.ssl.SSLContextImpl$DTLS10Context", null, null);
113         ps("SSLContext", "DTLSv1.2",
114             "sun.security.ssl.SSLContextImpl$DTLS12Context", null, null);
115         ps("SSLContext", "DTLS",
116             "sun.security.ssl.SSLContextImpl$DTLSContext", null, null);
117 
118         ps("SSLContext", "Default",
119             "sun.security.ssl.SSLContextImpl$DefaultSSLContext", null, null);
120 
121         /*
122          * KeyStore
123          */
124         ps("KeyStore", "PKCS12",
125             "sun.security.pkcs12.PKCS12KeyStore", null, null);
126     }
127 }
