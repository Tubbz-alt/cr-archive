26 package sun.security.util;
27 
28 /**
29  * An exception that denotes that an operation is pending.
30  * Currently used by LoginContext.
31  *
32  */
33 public class PendingException extends RuntimeException {
34 

35     private static final long serialVersionUID = -5201837247928788640L;
36 
37     /**
38      * Constructs a PendingException with no detail message. A detail
39      * message is a String that describes this particular exception.
40      */
41     public PendingException() {
42         super();
43     }
44 
45     /**
46      * Constructs a PendingException with the specified detail message.
47      * A detail message is a String that describes this particular
48      * exception.
49      *
50      * @param msg the detail message.
51      */
52     public PendingException(String msg) {
53         super(msg);
54     }
55 }
