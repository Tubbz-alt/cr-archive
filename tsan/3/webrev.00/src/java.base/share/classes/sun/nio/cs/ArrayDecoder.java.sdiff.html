28 /*
29  * FastPath byte[]->char[] decoder, REPLACE on malformed or
30  * unmappable input.
31  *
32  * FastPath encoded byte[]-> "String Latin1 coding" byte[] decoder for use when
33  * charset is always decodable to the internal String Latin1 coding byte[], ie. all mappings <=0xff
34  */
35 
36 public interface ArrayDecoder {
37     int decode(byte[] src, int off, int len, char[] dst);
38 
39     default boolean isASCIICompatible() {
40         return false;
41     }
42 
43     // Is always decodable to internal String Latin1 coding, ie. all mappings <= 0xff
44     default boolean isLatin1Decodable() {
45         return false;
46     }
47 
48     // Decode to internal String Latin1 coding byte[] fastpath for when isLatin1Decodable == true
49     default int decodeToLatin1(byte[] src, int sp, int len, byte[] dst) {
50         return 0;
51     }
52 }
