26 package sun.security.util;
27 
28 import sun.security.action.GetPropertyAction;
29 
30 import java.io.FilePermission;
31 import java.security.Permission;
32 import jdk.internal.access.SharedSecrets;
33 
34 /**
35  * Take care of FilePermission compatibility after JDK-8164705.
36  */
37 public class FilePermCompat {
38     /**
39      * New behavior? Keep compatibility? Both default true.
40      */
41     public static final boolean nb;
42     public static final boolean compat;
43 
44     static {
45         String flag = SecurityProperties.privilegedGetOverridable(
46                 "jdk.io.permissionsUseCanonicalPath");
47         if (flag == null) {
48             flag = "false";
49         }
50         switch (flag) {
51             case "true":
52                 nb = false;
53                 compat = false;
54                 break;
55             case "false":
56                 nb = true;
57                 compat = true;
58                 break;
59             default:
60                 throw new RuntimeException(
61                         "Invalid jdk.io.permissionsUseCanonicalPath: " + flag);
62         }
63     }
64 
65     public static Permission newPermPlusAltPath(Permission input) {
66         if (compat && input instanceof FilePermission) {
67             return SharedSecrets.getJavaIOFilePermissionAccess()
68                     .newPermPlusAltPath((FilePermission) input);
69         }
70         return input;
71     }
72 
73     public static Permission newPermUsingAltPath(Permission input) {
74         if (input instanceof FilePermission) {
75             return SharedSecrets.getJavaIOFilePermissionAccess()
76                     .newPermUsingAltPath((FilePermission) input);
77         }
78         return null;
79     }
80 }
