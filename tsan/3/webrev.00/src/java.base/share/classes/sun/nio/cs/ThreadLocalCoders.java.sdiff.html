 89             boolean hasName(Object ob, Object name) {
 90                 if (name instanceof String)
 91                     return (((CharsetDecoder)ob).charset().name().equals(name));
 92                 if (name instanceof Charset)
 93                     return ((CharsetDecoder)ob).charset().equals(name);
