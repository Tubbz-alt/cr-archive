26 /**
27  * IOUtils: A collection of IO-related public static methods.
28  */
29 
30 package sun.security.util;
31 
32 import java.io.EOFException;
33 import java.io.IOException;
34 import java.io.InputStream;

35 
36 public class IOUtils {
37 
38     /**
39      * Read exactly {@code length} of bytes from {@code in}.
40      *
41      * <p> Note that this method is safe to be called with unknown large
42      * {@code length} argument. The memory used is proportional to the
43      * actual bytes available. An exception is thrown if there are not
44      * enough bytes in the stream.
45      *
46      * @param is input stream, must not be null
47      * @param length number of bytes to read


48      * @return bytes read
49      * @throws EOFException if there are not enough bytes in the stream
50      * @throws IOException if an I/O error occurs or {@code length} is negative
51      * @throws OutOfMemoryError if an array of the required size cannot be
52      *         allocated.
53      */
54     public static byte[] readExactlyNBytes(InputStream is, int length)
55             throws IOException {











































56         if (length < 0) {
57             throw new IOException("length cannot be negative: " + length);
58         }
59         byte[] data = is.readNBytes(length);
60         if (data.length < length) {
61             throw new EOFException();
62         }
63         return data;
64     }
65 }
