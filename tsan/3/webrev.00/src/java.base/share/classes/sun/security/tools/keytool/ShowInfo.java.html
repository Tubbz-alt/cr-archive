26 package sun.security.tools.keytool;
27 
28 import javax.net.ssl.SSLContext;
29 import javax.net.ssl.SSLSocket;
30 
31 class ShowInfo {
32 
33     // verbose is not used yet.
34     static void tls(boolean verbose) throws Exception {
35 
36         SSLSocket ssls = (SSLSocket)
37                 SSLContext.getDefault().getSocketFactory().createSocket();
38 
39         System.out.println("Enabled Protocols");
40         System.out.println("-----------------");
41         for (String s : ssls.getEnabledProtocols()) {
42             System.out.println(s);
43         }
44 
45         System.out.println();
46 
47         System.out.println("Enabled Cipher Suites");
48         System.out.println("---------------------");
49         for (String s : ssls.getEnabledCipherSuites()) {
50             System.out.println(s);
51         }
52     }
53 }
