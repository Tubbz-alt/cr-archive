 26 package sun.security.validator;
 27 
 28 import java.security.cert.*;
 29 
 30 /**
 31  * ValidatorException thrown by the Validator. It has optional fields that
 32  * allow better error diagnostics.
 33  *
 34  * @author Andreas Sterbenz
 35  */
 36 public class ValidatorException extends CertificateException {
 37 
 38     @java.io.Serial
 39     private static final long serialVersionUID = -2836879718282292155L;
 40 
 41     public static final Object T_NO_TRUST_ANCHOR =
 42         "No trusted certificate found";
 43 
 44     public static final Object T_EE_EXTENSIONS =
 45         "End entity certificate extension check failed";
 46 
 47     public static final Object T_CA_EXTENSIONS =
 48         "CA certificate extension check failed";
 49 
 50     public static final Object T_CERT_EXPIRED =
 51         "Certificate expired";
 52 
 53     public static final Object T_SIGNATURE_ERROR =
 54         "Certificate signature validation failed";
 55 
 56     public static final Object T_NAME_CHAINING =
 57         "Certificate chaining error";
 58 
 59     public static final Object T_ALGORITHM_DISABLED =
 60         "Certificate signature algorithm disabled";
 61 
 62     public static final Object T_UNTRUSTED_CERT =
 63         "Untrusted certificate";
 64 
 65     @SuppressWarnings("serial") // Not statically typed as Serializable
 66     private Object type;
 67     private X509Certificate cert;
 68 
 69     public ValidatorException(String msg) {
 70         super(msg);
 71     }
 72 
 73     public ValidatorException(String msg, Throwable cause) {
 74         super(msg);
 75         initCause(cause);
 76     }
 77 
 78     public ValidatorException(Object type) {
 79         this(type, null);
 80     }
 81 
 82     public ValidatorException(Object type, X509Certificate cert) {
 83         super((String)type);
 84         this.type = type;
 85         this.cert = cert;
 86     }
 87 
 88     public ValidatorException(Object type, X509Certificate cert,
 89             Throwable cause) {
 90         this(type, cert);
 91         initCause(cause);
 92     }
 93 
 94     public ValidatorException(String msg, Object type, X509Certificate cert) {
 95         super(msg);
 96         this.type = type;
 97         this.cert = cert;
 98     }
 99 
100     public ValidatorException(String msg, Object type, X509Certificate cert,
101             Throwable cause) {
102         this(msg, type, cert);
103         initCause(cause);
104     }
105 
106     /**
107      * Get the type of the failure (one of the T_XXX constants), if
108      * available. This may be helpful when designing a user interface.
109      */
110     public Object getErrorType() {
111         return type;
112     }
113 
114     /**
115      * Get the certificate causing the exception, if available.
116      */
117     public X509Certificate getErrorCertificate() {
118         return cert;
119     }
120 
121 }
