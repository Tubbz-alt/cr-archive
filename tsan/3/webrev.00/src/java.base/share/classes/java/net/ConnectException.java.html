26 package java.net;
27 
28 /**
29  * Signals that an error occurred while attempting to connect a
30  * socket to a remote address and port.  Typically, the connection
31  * was refused remotely (e.g., no process is listening on the
32  * remote address/port).
33  *
34  * @since   1.1
35  */
36 public class ConnectException extends SocketException {
37     @java.io.Serial
38     private static final long serialVersionUID = 3831404271622369215L;
39 
40     /**
41      * Constructs a new ConnectException with the specified detail
42      * message as to why the connect error occurred.
43      * A detail message is a String that gives a specific
44      * description of this error.
45      * @param msg the detail message
46      */
47     public ConnectException(String msg) {
48         super(msg);
49     }
50 
51     /**
52      * Construct a new ConnectException with no detailed message.
53      */
54     public ConnectException() {}
55 }
