 25 package java.lang.constant;
 26 
 27 import java.lang.invoke.MethodHandles;
 28 import java.lang.invoke.MethodType;


 29 import java.util.Arrays;
 30 import java.util.List;
 31 
 32 import static java.util.Objects.requireNonNull;
 33 
 34 /**
 35  * A <a href="package-summary.html#nominal">nominal descriptor</a> for a
 36  * {@link MethodType}.  A {@linkplain MethodTypeDescImpl} corresponds to a
 37  * {@code Constant_MethodType_info} entry in the constant pool of a classfile.
 38  */
 39 final class MethodTypeDescImpl implements MethodTypeDesc {
 40     private final ClassDesc returnType;
 41     private final ClassDesc[] argTypes;
 42 
 43     /**
 44      * Constructs a {@linkplain MethodTypeDesc} with the specified return type
 45      * and parameter types
 46      *
 47      * @param returnType a {@link ClassDesc} describing the return type
 48      * @param argTypes {@link ClassDesc}s describing the parameter types
 49      */
 50     MethodTypeDescImpl(ClassDesc returnType, ClassDesc[] argTypes) {
 51         this.returnType = requireNonNull(returnType);
 52         this.argTypes = requireNonNull(argTypes);
 53 
 54         for (ClassDesc cr : argTypes)
 55             if (cr.isPrimitive() && cr.descriptorString().equals("V"))
 56                 throw new IllegalArgumentException("Void parameters not permitted");
 57     }
 58 
 59     /**
 60      * Creates a {@linkplain MethodTypeDescImpl} given a method descriptor string.
 61      *
 62      * @param descriptor the method descriptor string
 63      * @return a {@linkplain MethodTypeDescImpl} describing the desired method type
 64      * @throws IllegalArgumentException if the descriptor string is not a valid
 65      * method descriptor
 66      * @jvms 4.3.3 Method Descriptors
 67      */
 68     static MethodTypeDescImpl ofDescriptor(String descriptor) {
 69         requireNonNull(descriptor);
 70         List<String> types = ConstantUtils.parseMethodDescriptor(descriptor);
 71         ClassDesc[] paramTypes = types.stream().skip(1).map(ClassDesc::ofDescriptor).toArray(ClassDesc[]::new);
 72         return new MethodTypeDescImpl(ClassDesc.ofDescriptor(types.get(0)), paramTypes);
 73     }
 74 
 75     @Override
 76     public ClassDesc returnType() {
 77         return returnType;
 78     }
 79 
 80     @Override
 81     public int parameterCount() {
 82         return argTypes.length;
 83     }
 84 
 85     @Override
 86     public ClassDesc parameterType(int index) {
 87         return argTypes[index];
 88     }
 89 
 90     @Override
 91     public List<ClassDesc> parameterList() {
 92         return List.of(argTypes);
 93     }
 94 
 95     @Override
 96     public ClassDesc[] parameterArray() {
 97         return argTypes.clone();
 98     }
 99 
100     @Override
101     public MethodTypeDesc changeReturnType(ClassDesc returnType) {
102         return MethodTypeDesc.of(returnType, argTypes);
103     }
104 
105     @Override
106     public MethodTypeDesc changeParameterType(int index, ClassDesc paramType) {
107         ClassDesc[] newArgs = argTypes.clone();
108         newArgs[index] = paramType;
109         return MethodTypeDesc.of(returnType, newArgs);
110     }
111 
112     @Override
113     public MethodTypeDesc dropParameterTypes(int start, int end) {
114         if (start < 0 || start >= argTypes.length || end < 0 || end > argTypes.length)
115             throw new IndexOutOfBoundsException();
116         else if (start > end)
117             throw new IllegalArgumentException(String.format("Range (%d, %d) not valid for size %d", start, end, argTypes.length));
118         ClassDesc[] newArgs = new ClassDesc[argTypes.length - (end - start)];
119         System.arraycopy(argTypes, 0, newArgs, 0, start);
120         System.arraycopy(argTypes, end, newArgs, start, argTypes.length - end);
121         return MethodTypeDesc.of(returnType, newArgs);
122     }
123 
124     @Override
125     public MethodTypeDesc insertParameterTypes(int pos, ClassDesc... paramTypes) {
126         if (pos < 0 || pos > argTypes.length)
127             throw new IndexOutOfBoundsException(pos);
128         ClassDesc[] newArgs = new ClassDesc[argTypes.length + paramTypes.length];
129         System.arraycopy(argTypes, 0, newArgs, 0, pos);
130         System.arraycopy(paramTypes, 0, newArgs, pos, paramTypes.length);
131         System.arraycopy(argTypes, pos, newArgs, pos+paramTypes.length, argTypes.length - pos);
132         return MethodTypeDesc.of(returnType, newArgs);
133     }
134 
135     @Override
136     public MethodType resolveConstantDesc(MethodHandles.Lookup lookup) {
137         return MethodType.fromMethodDescriptorString(descriptorString(), lookup.lookupClass().getClassLoader());













138     }
139 
140     /**
141      * Returns {@code true} if this {@linkplain MethodTypeDescImpl} is
142      * equal to another {@linkplain MethodTypeDescImpl}.  Equality is
143      * determined by the two descriptors having equal return types and argument
144      * types.
145      *
146      * @param o the {@code MethodTypeDescImpl} to compare to this
147      *       {@code MethodTypeDescImpl}
148      * @return {@code true} if the specified {@code MethodTypeDescImpl} is
149      *      equals to this {@code MethodTypeDescImpl}.
150      */
151     @Override
152     public boolean equals(Object o) {
153         if (this == o) return true;
154         if (o == null || getClass() != o.getClass()) return false;
155 
156         MethodTypeDescImpl constant = (MethodTypeDescImpl) o;
157 
158         return returnType.equals(constant.returnType)
159                && Arrays.equals(argTypes, constant.argTypes);
160     }
161 
162     @Override
163     public int hashCode() {
164         int result = returnType.hashCode();
165         result = 31 * result + Arrays.hashCode(argTypes);
166         return result;
167     }
168 
169     @Override
170     public String toString() {
171         return String.format("MethodTypeDesc[%s]", displayDescriptor());
172     }
173 }
