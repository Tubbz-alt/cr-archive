26 package java.security;
27 
28 /**
29  * <p> This exception is thrown by the AccessController to indicate
30  * that a requested access (to a critical system resource such as the
31  * file system or the network) is denied.
32  *
33  * <p> The reason to deny access can vary.  For example, the requested
34  * permission might be of an incorrect type,  contain an invalid
35  * value, or request access that is not allowed according to the
36  * security policy.  Such information should be given whenever
37  * possible at the time the exception is thrown.
38  *
39  * @author Li Gong
40  * @author Roland Schemers
41  * @since 1.2
42  */
43 
44 public class AccessControlException extends SecurityException {
45 
46     @java.io.Serial
47     private static final long serialVersionUID = 5138225684096988535L;
48 
49     // the permission that caused the exception to be thrown.
50     private Permission perm;
51 
52     /**
53      * Constructs an {@code AccessControlException} with the
54      * specified, detailed message.
55      *
56      * @param   s   the detail message.
57      */
58     public AccessControlException(String s) {
59         super(s);
60     }
61 
62     /**
63      * Constructs an {@code AccessControlException} with the
64      * specified, detailed message, and the requested permission that caused
65      * the exception.
66      *
67      * @param   s   the detail message.
68      * @param   p   the permission that caused the exception.
69      */
70     public AccessControlException(String s, Permission p) {
71         super(s);
72         perm = p;
73     }
74 
75     /**
76      * Gets the Permission object associated with this exception, or
77      * null if there was no corresponding Permission object.
78      *
79      * @return the Permission object.
80      */
81     public Permission getPermission() {
82         return perm;
83     }
84 }
