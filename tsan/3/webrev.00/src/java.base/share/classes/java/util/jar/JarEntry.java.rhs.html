 26 package java.util.jar;
 27 
 28 import java.io.IOException;
 29 import java.util.zip.ZipEntry;
 30 import java.security.CodeSigner;
 31 import java.security.cert.Certificate;
 32 
 33 /**
 34  * This class is used to represent a JAR file entry.
 35  *
 36  * @since 1.2
 37  */
 38 public class JarEntry extends ZipEntry {

 39     Attributes attr;
 40     Certificate[] certs;
 41     CodeSigner[] signers;
 42 
 43     /**
 44      * Creates a new {@code JarEntry} for the specified JAR file
 45      * entry name.
 46      *
 47      * @param name the JAR file entry name
 48      * @throws    NullPointerException if the entry name is {@code null}
 49      * @throws    IllegalArgumentException if the entry name is longer than
 50      *            0xFFFF bytes.
 51      */
 52     public JarEntry(String name) {
 53         super(name);
 54     }
 55 
 56     /**
 57      * Creates a new {@code JarEntry} with fields taken from the
 58      * specified {@code ZipEntry} object.
 59      * @param ze the {@code ZipEntry} object to create the
 60      *           {@code JarEntry} from
 61      */
 62     public JarEntry(ZipEntry ze) {
 63         super(ze);
 64     }
 65 
 66     /**
 67      * Creates a new {@code JarEntry} with fields taken from the
 68      * specified {@code JarEntry} object.
 69      *
 70      * @param je the {@code JarEntry} to copy
 71      */
 72     public JarEntry(JarEntry je) {
 73         this((ZipEntry)je);
 74         this.attr = je.attr;
 75         this.certs = je.certs;
 76         this.signers = je.signers;
 77     }
 78 
 79     /**
 80      * Returns the {@code Manifest} {@code Attributes} for this
 81      * entry, or {@code null} if none.
 82      *
 83      * @return the {@code Manifest} {@code Attributes} for this
 84      * entry, or {@code null} if none
 85      * @throws IOException  if an I/O error has occurred
 86      */
 87     public Attributes getAttributes() throws IOException {
 88         return attr;
 89     }
 90 
 91     /**
 92      * Returns the {@code Certificate} objects for this entry, or
 93      * {@code null} if none. This method can only be called once
 94      * the {@code JarEntry} has been completely verified by reading
 95      * from the entry input stream until the end of the stream has been
 96      * reached. Otherwise, this method will return {@code null}.
 97      *
 98      * <p>The returned certificate array comprises all the signer certificates
 99      * that were used to verify this entry. Each signer certificate is
100      * followed by its supporting certificate chain (which may be empty).
101      * Each signer certificate and its supporting certificate chain are ordered
102      * bottom-to-top (i.e., with the signer certificate first and the (root)
103      * certificate authority last).
104      *
105      * @return the {@code Certificate} objects for this entry, or
106      * {@code null} if none.
107      */
108     public Certificate[] getCertificates() {
109         return certs == null ? null : certs.clone();
110     }
111 
112     /**
113      * Returns the {@code CodeSigner} objects for this entry, or
114      * {@code null} if none. This method can only be called once
115      * the {@code JarEntry} has been completely verified by reading
116      * from the entry input stream until the end of the stream has been
117      * reached. Otherwise, this method will return {@code null}.
118      *
119      * <p>The returned array comprises all the code signers that have signed
120      * this entry.
121      *
122      * @return the {@code CodeSigner} objects for this entry, or
123      * {@code null} if none.
124      *
125      * @since 1.5
126      */
127     public CodeSigner[] getCodeSigners() {
128         return signers == null ? null : signers.clone();
129     }
130 
131     /**
132      * Returns the real name of this {@code JarEntry}.
133      *
134      * If this {@code JarEntry} is an entry of a
135      * <a href="JarFile.html#multirelease">multi-release jar file</a> and the
136      * {@code JarFile} is configured to be processed as such, the name returned
137      * by this method is the path name of the versioned entry that the
138      * {@code JarEntry} represents, rather than the path name of the base entry
139      * that {@link #getName()} returns. If the {@code JarEntry} does not represent
140      * a versioned entry of a multi-release {@code JarFile} or the {@code JarFile}
141      * is not configured for processing a multi-release jar file, this method
142      * returns the same name that {@link #getName()} returns.
143      *
144      * @return the real name of the JarEntry
145      *
146      * @since 10
147      */
148     public String getRealName() {
149         return super.getName();
150     }
151 }
