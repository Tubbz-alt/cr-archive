25 package java.io;
26 
27 /**
28  * Base class for character conversion exceptions.
29  *
30  * @author      Asmus Freytag
31  * @since       1.1
32  */
33 public class CharConversionException
34     extends java.io.IOException
35 {
36     private static final long serialVersionUID = -8680016352018427031L;
37 
38     /**
39      * This provides no detailed message.
40      */
41     public CharConversionException() {
42     }
43     /**
44      * This provides a detailed message.
45      *
46      * @param s the detailed message associated with the exception.
47      */
48     public CharConversionException(String s) {
49         super(s);
50     }
51 }
