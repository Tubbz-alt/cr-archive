26 package java.lang.invoke;
27 
28 /**
29  * LambdaConversionException
30  *
31  * @since 1.8
32  */
33 public class LambdaConversionException extends Exception {
34     private static final long serialVersionUID = 292L + 8L;
35 
36     /**
37      * Constructs a {@code LambdaConversionException}.
38      */
39     public LambdaConversionException() {
40     }
41 
42     /**
43      * Constructs a {@code LambdaConversionException} with a message.
44      * @param message the detail message
45      */
46     public LambdaConversionException(String message) {
47         super(message);
48     }
49 
50     /**
51      * Constructs a {@code LambdaConversionException} with a message and cause.
52      * @param message the detail message
53      * @param cause the cause
54      */
55     public LambdaConversionException(String message, Throwable cause) {
56         super(message, cause);
57     }
58 
59     /**
60      * Constructs a {@code LambdaConversionException} with a cause.
61      * @param cause the cause
62      */
63     public LambdaConversionException(Throwable cause) {
64         super(cause);
65     }
66 
67     /**
68      * Constructs a {@code LambdaConversionException} with a message,
69      * cause, and other settings.
70      * @param message the detail message
71      * @param cause the cause
72      * @param enableSuppression whether or not suppressed exceptions are enabled
73      * @param writableStackTrace whether or not the stack trace is writable
74      */
75     public LambdaConversionException(String message, Throwable cause, boolean enableSuppression, boolean writableStackTrace) {
76         super(message, cause, enableSuppression, writableStackTrace);
77     }
78 }
