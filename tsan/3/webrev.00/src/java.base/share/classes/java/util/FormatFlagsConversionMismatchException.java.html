26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when a conversion and flag are incompatible.
30  *
31  * <p> Unless otherwise specified, passing a {@code null} argument to any
32  * method or constructor in this class will cause a {@link
33  * NullPointerException} to be thrown.
34  *
35  * @since 1.5
36  */
37 public class FormatFlagsConversionMismatchException
38     extends IllegalFormatException
39 {
40     @java.io.Serial
41     private static final long serialVersionUID = 19120414L;
42 
43     private String f;
44 
45     private char c;
46 
47     /**
48      * Constructs an instance of this class with the specified flag
49      * and conversion.
50      *
51      * @param  f
52      *         The flag
53      *
54      * @param  c
55      *         The conversion
56      */
57     public FormatFlagsConversionMismatchException(String f, char c) {
58         if (f == null)
59             throw new NullPointerException();
60         this.f = f;
61         this.c = c;
62     }
63 
64     /**
65      * Returns the incompatible flag.
66      *
67      * @return  The flag
68      */
69      public String getFlags() {
70         return f;
71     }
72 
73     /**
74      * Returns the incompatible conversion.
75      *
76      * @return  The conversion
77      */
78     public char getConversion() {
79         return c;
80     }
81 
82     public String getMessage() {
83         return "Conversion = " + c + ", Flags = " + f;
84     }
85 }
