 26 package java.security.cert;
 27 
 28 import java.security.GeneralSecurityException;
 29 
 30 /**
 31  * An exception indicating one of a variety of problems retrieving
 32  * certificates and CRLs from a {@code CertStore}.
 33  * <p>
 34  * A {@code CertStoreException} provides support for wrapping
 35  * exceptions. The {@link #getCause getCause} method returns the throwable,
 36  * if any, that caused this exception to be thrown.
 37  * <p>
 38  * <b>Concurrent Access</b>
 39  * <p>
 40  * Unless otherwise specified, the methods defined in this class are not
 41  * thread-safe. Multiple threads that need to access a single
 42  * object concurrently should synchronize amongst themselves and
 43  * provide the necessary locking. Multiple threads each manipulating
 44  * separate objects need not synchronize.
 45  *
 46  * @see CertStore
 47  *
 48  * @since       1.4
 49  * @author      Sean Mullan
 50  */
 51 public class CertStoreException extends GeneralSecurityException {
 52 
 53     @java.io.Serial
 54     private static final long serialVersionUID = 2395296107471573245L;
 55 
 56     /**
 57      * Creates a {@code CertStoreException} with {@code null} as
 58      * its detail message.
 59      */
 60     public CertStoreException() {
 61         super();
 62     }
 63 
 64     /**
 65      * Creates a {@code CertStoreException} with the given detail
 66      * message. A detail message is a {@code String} that describes this
 67      * particular exception.
 68      *
 69      * @param msg the detail message
 70      */
 71     public CertStoreException(String msg) {
 72         super(msg);
 73     }
 74 
 75     /**
 76      * Creates a {@code CertStoreException} that wraps the specified
 77      * throwable. This allows any exception to be converted into a
 78      * {@code CertStoreException}, while retaining information about the
 79      * cause, which may be useful for debugging. The detail message is
 80      * set to ({@code cause==null ? null : cause.toString()}) (which
 81      * typically contains the class and detail message of cause).
 82      *
 83      * @param cause the cause (which is saved for later retrieval by the
 84      * {@link #getCause getCause()} method). (A {@code null} value is
 85      * permitted, and indicates that the cause is nonexistent or unknown.)
 86      */
 87     public CertStoreException(Throwable cause) {
 88         super(cause);
 89     }
 90 
 91     /**
 92      * Creates a {@code CertStoreException} with the specified detail
 93      * message and cause.
 94      *
 95      * @param msg the detail message
 96      * @param cause the cause (which is saved for later retrieval by the
 97      * {@link #getCause getCause()} method). (A {@code null} value is
 98      * permitted, and indicates that the cause is nonexistent or unknown.)
 99      */
100     public CertStoreException(String msg, Throwable cause) {
101         super(msg, cause);
102     }
103 
104 }
