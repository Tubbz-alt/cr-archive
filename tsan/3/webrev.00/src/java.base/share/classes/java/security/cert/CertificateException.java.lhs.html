26 package java.security.cert;
27 
28 import java.security.GeneralSecurityException;
29 
30 /**
31  * This exception indicates one of a variety of certificate problems.
32  *
33  * @author Hemma Prafullchandra
34  * @since 1.2
35  * @see Certificate
36  */
37 public class CertificateException extends GeneralSecurityException {
38 

39     private static final long serialVersionUID = 3192535253797119798L;
40 
41     /**
42      * Constructs a certificate exception with no detail message. A detail
43      * message is a String that describes this particular exception.
44      */
45     public CertificateException() {
46         super();
47     }
48 
49     /**
50      * Constructs a certificate exception with the given detail
51      * message. A detail message is a String that describes this
52      * particular exception.
53      *
54      * @param msg the detail message.
55      */
56     public CertificateException(String msg) {
57         super(msg);
58     }
59 
60     /**
61      * Creates a {@code CertificateException} with the specified
62      * detail message and cause.
63      *
64      * @param message the detail message (which is saved for later retrieval
65      *        by the {@link #getMessage()} method).
66      * @param cause the cause (which is saved for later retrieval by the
67      *        {@link #getCause()} method).  (A {@code null} value is permitted,
68      *        and indicates that the cause is nonexistent or unknown.)
69      * @since 1.5
70      */
71     public CertificateException(String message, Throwable cause) {
72         super(message, cause);
73     }
74 
75     /**
76      * Creates a {@code CertificateException} with the specified cause
77      * and a detail message of {@code (cause==null ? null : cause.toString())}
78      * (which typically contains the class and detail message of
79      * {@code cause}).
80      *
81      * @param cause the cause (which is saved for later retrieval by the
82      *        {@link #getCause()} method).  (A {@code null} value is permitted,
83      *        and indicates that the cause is nonexistent or unknown.)
84      * @since 1.5
85      */
86     public CertificateException(Throwable cause) {
87         super(cause);
88     }
89 }
