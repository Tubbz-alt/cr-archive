26 package java.lang;
27 
28 /**
29  * Thrown if an application tries to call a specified method of a
30  * class (either static or instance), and that class no longer has a
31  * definition of that method.
32  * <p>
33  * Normally, this error is caught by the compiler; this error can
34  * only occur at run time if the definition of a class has
35  * incompatibly changed.
36  *
37  * @author  unascribed
38  * @since   1.0
39  */
40 public class NoSuchMethodError extends IncompatibleClassChangeError {
41     @java.io.Serial
42     private static final long serialVersionUID = -3765521442372831335L;
43 
44     /**
45      * Constructs a {@code NoSuchMethodError} with no detail message.
46      */
47     public NoSuchMethodError() {
48         super();
49     }
50 
51     /**
52      * Constructs a {@code NoSuchMethodError} with the
53      * specified detail message.
54      *
55      * @param   s   the detail message.
56      */
57     public NoSuchMethodError(String s) {
58         super(s);
59     }
60 }
