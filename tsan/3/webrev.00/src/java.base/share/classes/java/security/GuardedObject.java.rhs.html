 26 package java.security;
 27 
 28 /**
 29  * A GuardedObject is an object that is used to protect access to
 30  * another object.
 31  *
 32  * <p>A GuardedObject encapsulates a target object and a Guard object,
 33  * such that access to the target object is possible
 34  * only if the Guard object allows it.
 35  * Once an object is encapsulated by a GuardedObject,
 36  * access to that object is controlled by the {@code getObject}
 37  * method, which invokes the
 38  * {@code checkGuard} method on the Guard object that is
 39  * guarding access. If access is not allowed,
 40  * an exception is thrown.
 41  *
 42  * @see Guard
 43  * @see Permission
 44  *
 45  * @author Roland Schemers
 46  * @author Li Gong
 47  * @since 1.2
 48  */
 49 
 50 public class GuardedObject implements java.io.Serializable {
 51 
 52     @java.io.Serial
 53     private static final long serialVersionUID = -5240450096227834308L;
 54 
 55     @SuppressWarnings("serial") // Not statically typed as Serializable
 56     private Object object; // the object we are guarding
 57     @SuppressWarnings("serial") // Not statically typed as Serializable
 58     private Guard guard;   // the guard
 59 
 60     /**
 61      * Constructs a GuardedObject using the specified object and guard.
 62      * If the Guard object is null, then no restrictions will
 63      * be placed on who can access the object.
 64      *
 65      * @param object the object to be guarded.
 66      *
 67      * @param guard the Guard object that guards access to the object.
 68      */
 69 
 70     public GuardedObject(Object object, Guard guard)
 71     {
 72         this.guard = guard;
 73         this.object = object;
 74     }
 75 
 76     /**
 77      * Retrieves the guarded object, or throws an exception if access
 78      * to the guarded object is denied by the guard.
 79      *
 80      * @return the guarded object.
 81      *
 82      * @throws    SecurityException if access to the guarded object is
 83      * denied.
 84      */
 85     public Object getObject()
 86         throws SecurityException
 87     {
 88         if (guard != null)
 89             guard.checkGuard(object);
 90 
 91         return object;
 92     }
 93 
 94     /**
 95      * Writes this object out to a stream (i.e., serializes it).
 96      * We check the guard if there is one.
 97      */
 98     @java.io.Serial
 99     private void writeObject(java.io.ObjectOutputStream oos)
100         throws java.io.IOException
101     {
102         if (guard != null)
103             guard.checkGuard(object);
104 
105         oos.defaultWriteObject();
106     }
107 }
