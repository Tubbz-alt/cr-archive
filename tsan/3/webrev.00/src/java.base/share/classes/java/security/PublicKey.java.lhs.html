28 /**
29  * <p>A public key. This interface contains no methods or constants.
30  * It merely serves to group (and provide type safety for) all public key
31  * interfaces.
32  *
33  * Note: The specialized public key interfaces extend this interface.
34  * See, for example, the DSAPublicKey interface in
35  * {@code java.security.interfaces}.
36  *
37  * @since 1.1
38  * @see Key
39  * @see PrivateKey
40  * @see Certificate
41  * @see Signature#initVerify
42  * @see java.security.interfaces.DSAPublicKey
43  * @see java.security.interfaces.RSAPublicKey
44  *
45  */
46 
47 public interface PublicKey extends Key {
48     // Declare serialVersionUID to be compatible with JDK1.1
49     /**
50      * The class fingerprint that is set to indicate serialization
51      * compatibility with a previous version of the class.
52      *
53      * @deprecated A {@code serialVersionUID} field in an interface is
54      * ineffectual. Do not use; no replacement.
55      */
56     @Deprecated
57     @SuppressWarnings("serial")

58     static final long serialVersionUID = 7187392471159151072L;
59 }
