28 /**
29  * The {@code GeneralSecurityException} class is a generic
30  * security exception class that provides type safety for all the
31  * security-related exception classes that extend from it.
32  *
33  * @author Jan Luehe
34  * @since 1.2
35  */
36 
37 public class GeneralSecurityException extends Exception {
38 

39     private static final long serialVersionUID = 894798122053539237L;
40 
41     /**
42      * Constructs a GeneralSecurityException with no detail message.
43      */
44     public GeneralSecurityException() {
45         super();
46     }
47 
48     /**
49      * Constructs a GeneralSecurityException with the specified detail
50      * message.
51      * A detail message is a String that describes this particular
52      * exception.
53      *
54      * @param msg the detail message.
55      */
56     public GeneralSecurityException(String msg) {
57         super(msg);
58     }
