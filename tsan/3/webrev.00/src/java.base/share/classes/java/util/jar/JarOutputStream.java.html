 26 package java.util.jar;
 27 
 28 import java.util.zip.*;
 29 import java.io.*;
 30 
 31 /**
 32  * The {@code JarOutputStream} class is used to write the contents
 33  * of a JAR file to any output stream. It extends the class
 34  * {@code java.util.zip.ZipOutputStream} with support
 35  * for writing an optional {@code Manifest} entry. The
 36  * {@code Manifest} can be used to specify meta-information about
 37  * the JAR file and its entries.
 38  *
 39  * @author  David Connelly
 40  * @see     Manifest
 41  * @see     java.util.zip.ZipOutputStream
 42  * @since   1.2
 43  */
 44 public class JarOutputStream extends ZipOutputStream {
 45     private static final int JAR_MAGIC = 0xCAFE;
 46 
 47     /**
 48      * Creates a new {@code JarOutputStream} with the specified
 49      * {@code Manifest}. The manifest is written as the first
 50      * entry to the output stream.
 51      *
 52      * @param out the actual output stream
 53      * @param man the optional {@code Manifest}
 54      * @throws    IOException if an I/O error has occurred
 55      */
 56     public JarOutputStream(OutputStream out, Manifest man) throws IOException {
 57         super(out);
 58         if (man == null) {
 59             throw new NullPointerException("man");
 60         }
 61         ZipEntry e = new ZipEntry(JarFile.MANIFEST_NAME);
 62         putNextEntry(e);
 63         man.write(new BufferedOutputStream(this));
 64         closeEntry();
 65     }
 66 
 67     /**
 68      * Creates a new {@code JarOutputStream} with no manifest.
 69      * @param out the actual output stream
 70      * @throws    IOException if an I/O error has occurred
 71      */
 72     public JarOutputStream(OutputStream out) throws IOException {
 73         super(out);
 74     }
 75 
 76     /**
 77      * Begins writing a new JAR file entry and positions the stream
 78      * to the start of the entry data. This method will also close
 79      * any previous entry. The default compression method will be
 80      * used if no compression method was specified for the entry.
 81      * The current time will be used if the entry has no set modification
 82      * time.
 83      *
 84      * @param ze the ZIP/JAR entry to be written
 85      * @throws    ZipException if a ZIP error has occurred
 86      * @throws    IOException if an I/O error has occurred
 87      */
 88     public void putNextEntry(ZipEntry ze) throws IOException {
 89         if (firstEntry) {
 90             // Make sure that extra field data for first JAR
 91             // entry includes JAR magic number id.
 92             byte[] edata = ze.getExtra();
 93             if (edata == null || !hasMagic(edata)) {
 94                 if (edata == null) {
 95                     edata = new byte[4];
 96                 } else {
 97                     // Prepend magic to existing extra data
 98                     byte[] tmp = new byte[edata.length + 4];
 99                     System.arraycopy(edata, 0, tmp, 4, edata.length);
100                     edata = tmp;
101                 }
102                 set16(edata, 0, JAR_MAGIC); // extra field id
103                 set16(edata, 2, 0);         // extra field size
104                 ze.setExtra(edata);
105             }
106             firstEntry = false;
107         }
108         super.putNextEntry(ze);
109     }
110 
111     private boolean firstEntry = true;
112 
113     /*
114      * Returns true if specified byte array contains the
115      * jar magic extra field id.
116      */
117     private static boolean hasMagic(byte[] edata) {
118         try {
119             int i = 0;
120             while (i < edata.length) {
121                 if (get16(edata, i) == JAR_MAGIC) {
122                     return true;
123                 }
124                 i += get16(edata, i + 2) + 4;
125             }
126         } catch (ArrayIndexOutOfBoundsException e) {
127             // Invalid extra field data
128         }
129         return false;
130     }
131 
132     /*
133      * Fetches unsigned 16-bit value from byte array at specified offset.
134      * The bytes are assumed to be in Intel (little-endian) byte order.
135      */
136     private static int get16(byte[] b, int off) {
137         return Byte.toUnsignedInt(b[off]) | ( Byte.toUnsignedInt(b[off+1]) << 8);
138     }
139 
140     /*
141      * Sets 16-bit value at specified offset. The bytes are assumed to
142      * be in Intel (little-endian) byte order.
143      */
144     private static void set16(byte[] b, int off, int value) {
145         b[off+0] = (byte)value;
146         b[off+1] = (byte)(value >> 8);
147     }
148 }
