25 package java.security.spec;
26 
27 /**
28  * This immutable class specifies the set of parameters used for
29  * generating elliptic curve (EC) domain parameters.
30  *
31  * @see AlgorithmParameterSpec
32  *
33  * @author Valerie Peng
34  *
35  * @since 1.5
36  */
37 public class ECGenParameterSpec extends NamedParameterSpec {
38 
39     /**
40      * Creates a parameter specification for EC parameter
41      * generation using a standard (or predefined) name
42      * {@code stdName} in order to generate the corresponding
43      * (precomputed) elliptic curve domain parameters. For the
44      * list of supported names, please consult the documentation
45      * of the provider whose implementation will be used.
46      *
47      * @param stdName the standard name of the to-be-generated EC
48      *        domain parameters. See the ParameterSpec Names section in the
49      *        <a href=
50      *        "{@docRoot}/../specs/security/standard-names.html#parameterspec-names">
51      *        Java Security Standard Algorithm Names Specification</a> for
52      *        information about standard names.
53      * @throws NullPointerException if {@code stdName} is null.
54      */
55     public ECGenParameterSpec(String stdName) {
56         super(stdName);
57     }
58 }
