 26 package java.util.spi;
 27 
 28 import java.util.Calendar;
 29 import java.util.Locale;
 30 import java.util.Map;

 32 /**
 33  * An abstract class for service providers that provide localized string
 34  * representations (display names) of {@code Calendar} field values.
 35  *
 36  * <p><a id="calendartypes"><b>Calendar Types</b></a>
 37  *
 38  * <p>Calendar types are used to specify calendar systems for which the {@link
 39  * #getDisplayName(String, int, int, int, Locale) getDisplayName} and {@link
 40  * #getDisplayNames(String, int, int, Locale) getDisplayNames} methods provide
 41  * calendar field value names. See {@link Calendar#getCalendarType()} for details.
 42  *
 43  * <p><b>Calendar Fields</b>
 44  *
 45  * <p>Calendar fields are specified with the constants defined in {@link
 46  * Calendar}. The following are calendar-common fields and their values to be
 47  * supported for each calendar system.
 48  *
 49  * <table class="striped">
 50  * <caption style="display:none">Field values</caption>
 51  * <thead>
 52  *   <tr>
 53  *     <th scope="col">Field</th>
 54  *     <th scope="col">Value</th>
 55  *     <th scope="col">Description</th>
 56  *   </tr>
 57  * </thead>
 58  * <tbody>
 59  *   <tr>
 60  *     <th scope="row" style="vertical-align:top">{@link Calendar#MONTH}</th>
 61  *     <td style="vertical-align:top">{@link Calendar#JANUARY} to {@link Calendar#UNDECIMBER}</td>
 62  *     <td>Month numbering is 0-based (e.g., 0 - January, ..., 11 -
 63  *         December). Some calendar systems have 13 months. Month
 64  *         names need to be supported in both the formatting and
 65  *         stand-alone forms if required by the supported locales. If there's
 66  *         no distinction in the two forms, the same names should be returned
 67  *         in both of the forms.</td>
 68  *   </tr>
 69  *   <tr>
 70  *     <th scope="row" style="vertical-align:top">{@link Calendar#DAY_OF_WEEK}</th>
 71  *     <td style="vertical-align:top">{@link Calendar#SUNDAY} to {@link Calendar#SATURDAY}</td>
 72  *     <td>Day-of-week numbering is 1-based starting from Sunday (i.e., 1 - Sunday,
 73  *         ..., 7 - Saturday).</td>
 74  *   </tr>
 75  *   <tr>
 76  *     <th scope="row" style="vertical-align:top">{@link Calendar#AM_PM}</th>
 77  *     <td style="vertical-align:top">{@link Calendar#AM} to {@link Calendar#PM}</td>
 78  *     <td>0 - AM, 1 - PM</td>
 79  *   </tr>
 80  * </tbody>
 81  * </table>
 82  *
 83  * <p style="margin-top:20px">The following are calendar-specific fields and their values to be supported.
 84  *
 85  * <table class="plain">
 86  * <caption style="display:none">Calendar type and field values</caption>
 87  * <thead>
 88  *   <tr>
 89  *     <th scope="col">Calendar Type</th>
 90  *     <th scope="col">Field</th>
 91  *     <th scope="col">Value</th>
 92  *     <th scope="col">Description</th>
 93  *   </tr>
 94  * </thead>
 95  * <tbody>
 96  *   <tr>
 97  *     <th scope="row"rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@code "gregory"}</th>
 98  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@link Calendar#ERA}</th>
 99  *     <th scope="row" style="font-weight:normal">0</th>
100  *     <td>{@link java.util.GregorianCalendar#BC} (BCE)</td>
101  *   </tr>
102  *   <tr>
103  *     <th scope="row" style="font-weight:normal">1</th>
104  *     <td>{@link java.util.GregorianCalendar#AD} (CE)</td>
105  *   </tr>
106  *   <tr>
107  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@code "buddhist"}</th>
108  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@link Calendar#ERA}</th>
109  *     <th scope="row" style="font-weight:normal">0</th>
110  *     <td>BC (BCE)</td>
111  *   </tr>
112  *   <tr>
113  *     <th scope="row" style="font-weight:normal">1</th>
114  *     <td>B.E. (Buddhist Era)</td>
115  *   </tr>
116  *   <tr>
117  *     <th scope="row" rowspan="7" style="font-weight:normal; text-align:left; vertical-align:top">{@code "japanese"}</th>
118  *     <th scope="row" rowspan="6" style="font-weight:normal; text-align:left; vertical-align:top">{@link Calendar#ERA}</th>
119  *     <th scope="row" style="font-weight:normal">0</th>
120  *     <td>Seireki (Before Meiji)</td>
121  *   </tr>
122  *   <tr>
123  *     <th scope="row" style="font-weight:normal">1</th>
124  *     <td>Meiji</td>
125  *   </tr>
126  *   <tr>
127  *     <th scope="row" style="font-weight:normal">2</th>
128  *     <td>Taisho</td>
129  *   </tr>
130  *   <tr>
131  *     <th scope="row" style="font-weight:normal">3</th>
132  *     <td>Showa</td>
133  *   </tr>
134  *   <tr>
135  *     <th scope="row" style="font-weight:normal">4</th>
136  *     <td>Heisei</td>
137  *   </tr>
138  *   <tr>
139  *     <th scope="row" style="font-weight:normal">5</th>
140  *     <td>Reiwa</td>
141  *   </tr>
142  *   <tr>
143  *     <th scope="row" style="font-weight:normal; text-align:left; vertical-align:top">{@link Calendar#YEAR}</th>
144  *     <th scope="row" style="font-weight:normal">1</th>
145  *     <td>the first year in each era. It should be returned when a long
146  *     style ({@link Calendar#LONG_FORMAT} or {@link Calendar#LONG_STANDALONE}) is
147  *     specified. See also the <a href="../../text/SimpleDateFormat.html#year">
148  *     Year representation in {@code SimpleDateFormat}</a>.</td>
149  *   </tr>
150  *   <tr>
151  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@code "roc"}</th>
152  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@link Calendar#ERA}</th>
153  *     <th scope="row" style="font-weight:normal">0</th>
154  *     <td>Before R.O.C.</td>
155  *   </tr>
156  *   <tr>
157  *     <th scope="row" style="font-weight:normal">1</th>
158  *     <td>R.O.C.</td>
159  *   </tr>
160  *   <tr>
161  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@code "islamic"}</th>
162  *     <th scope="row" rowspan="2" style="font-weight:normal; text-align:left; vertical-align:top">{@link Calendar#ERA}</th>
163  *     <th scope="row" style="font-weight:normal">0</th>
164  *     <td>Before AH</td>
165  *   </tr>
166  *   <tr>
167  *     <th scope="row" style="font-weight:normal">1</th>
168  *     <td>Anno Hijrah (AH)</td>
169  *   </tr>
170  * </tbody>
171  * </table>
172  *
173  * <p>Calendar field value names for {@code "gregory"} must be consistent with
174  * the date-time symbols provided by {@link java.text.spi.DateFormatSymbolsProvider}.
175  *
176  * <p>Time zone names are supported by {@link TimeZoneNameProvider}.
177  *
178  * @author Masayoshi Okutsu
179  * @since 1.8
180  * @see CalendarDataProvider
181  * @see Locale#getUnicodeLocaleType(String)
182  */
183 public abstract class CalendarNameProvider extends LocaleServiceProvider {
184     /**
185      * Sole constructor. (For invocation by subclass constructors, typically
186      * implicit.)
187      */
188     protected CalendarNameProvider() {
189     }
190 
191     /**
192      * Returns the string representation (display name) of the calendar
193      * {@code field value} in the given {@code style} and
194      * {@code locale}.  If no string representation is
195      * applicable, {@code null} is returned.
196      *
197      * <p>{@code field} is a {@code Calendar} field index, such as {@link
198      * Calendar#MONTH}. The time zone fields, {@link Calendar#ZONE_OFFSET} and
199      * {@link Calendar#DST_OFFSET}, are <em>not</em> supported by this
200      * method. {@code null} must be returned if any time zone fields are
201      * specified.
202      *
203      * <p>{@code value} is the numeric representation of the {@code field} value.
204      * For example, if {@code field} is {@link Calendar#DAY_OF_WEEK}, the valid
205      * values are {@link Calendar#SUNDAY} to {@link Calendar#SATURDAY}
206      * (inclusive).
207      *
208      * <p>{@code style} gives the style of the string representation. It is one
209      * of {@link Calendar#SHORT_FORMAT} ({@link Calendar#SHORT SHORT}),
210      * {@link Calendar#SHORT_STANDALONE}, {@link Calendar#LONG_FORMAT}
211      * ({@link Calendar#LONG LONG}), {@link Calendar#LONG_STANDALONE},
212      * {@link Calendar#NARROW_FORMAT}, or {@link Calendar#NARROW_STANDALONE}.
213      *
214      * <p>For example, the following call will return {@code "Sunday"}.
215      * <pre>
216      * getDisplayName("gregory", Calendar.DAY_OF_WEEK, Calendar.SUNDAY,
217      *                Calendar.LONG_STANDALONE, Locale.ENGLISH);
218      * </pre>
219      *
220      * @param calendarType
221      *              the calendar type. (Any calendar type given by {@code locale}
222      *              is ignored.)
223      * @param field
224      *              the {@code Calendar} field index,
225      *              such as {@link Calendar#DAY_OF_WEEK}
226      * @param value
227      *              the value of the {@code Calendar field},
228      *              such as {@link Calendar#MONDAY}
229      * @param style
230      *              the string representation style: one of {@link
231      *              Calendar#SHORT_FORMAT} ({@link Calendar#SHORT SHORT}),
232      *              {@link Calendar#SHORT_STANDALONE}, {@link
233      *              Calendar#LONG_FORMAT} ({@link Calendar#LONG LONG}),
234      *              {@link Calendar#LONG_STANDALONE},
235      *              {@link Calendar#NARROW_FORMAT},
236      *              or {@link Calendar#NARROW_STANDALONE}
237      * @param locale
238      *              the desired locale
239      * @return the string representation of the {@code field value}, or {@code
240      *         null} if the string representation is not applicable or
241      *         the given calendar type is unknown
242      * @throws IllegalArgumentException
243      *         if {@code field} or {@code style} is invalid
244      * @throws NullPointerException if {@code locale} is {@code null}
245      * @see TimeZoneNameProvider
246      * @see java.util.Calendar#get(int)
247      * @see java.util.Calendar#getDisplayName(int, int, Locale)
248      */
249     public abstract String getDisplayName(String calendarType,
250                                           int field, int value,
251                                           int style, Locale locale);
252 
253     /**
254      * Returns a {@code Map} containing all string representations (display
255      * names) of the {@code Calendar} {@code field} in the given {@code style}
256      * and {@code locale} and their corresponding field values.
257      *
258      * <p>{@code field} is a {@code Calendar} field index, such as {@link
259      * Calendar#MONTH}. The time zone fields, {@link Calendar#ZONE_OFFSET} and
260      * {@link Calendar#DST_OFFSET}, are <em>not</em> supported by this
261      * method. {@code null} must be returned if any time zone fields are specified.
262      *
263      * <p>{@code style} gives the style of the string representation. It must be
264      * one of {@link Calendar#ALL_STYLES}, {@link Calendar#SHORT_FORMAT} ({@link
265      * Calendar#SHORT SHORT}), {@link Calendar#SHORT_STANDALONE}, {@link
266      * Calendar#LONG_FORMAT} ({@link Calendar#LONG LONG}), {@link
267      * Calendar#LONG_STANDALONE}, {@link Calendar#NARROW_FORMAT}, or
268      * {@link Calendar#NARROW_STANDALONE}. Note that narrow names may
269      * not be unique due to use of single characters, such as "S" for Sunday
270      * and Saturday, and that no narrow names are included in that case.
271      *
272      * <p>For example, the following call will return a {@code Map} containing
273      * {@code "January"} to {@link Calendar#JANUARY}, {@code "Jan"} to {@link
274      * Calendar#JANUARY}, {@code "February"} to {@link Calendar#FEBRUARY},
275      * {@code "Feb"} to {@link Calendar#FEBRUARY}, and so on.
276      * <pre>
277      * getDisplayNames("gregory", Calendar.MONTH, Calendar.ALL_STYLES, Locale.ENGLISH);
278      * </pre>
279      *
280      * @param calendarType
281      *              the calendar type. (Any calendar type given by {@code locale}
282      *              is ignored.)
283      * @param field
284      *              the calendar field for which the display names are returned
285      * @param style
286      *              the style applied to the display names; one of
287      *              {@link Calendar#ALL_STYLES}, {@link Calendar#SHORT_FORMAT}
288      *              ({@link Calendar#SHORT SHORT}), {@link
289      *              Calendar#SHORT_STANDALONE}, {@link Calendar#LONG_FORMAT}
290      *              ({@link Calendar#LONG LONG}), {@link Calendar#LONG_STANDALONE},
291      *              {@link Calendar#NARROW_FORMAT},
292      *              or {@link Calendar#NARROW_STANDALONE}
293      * @param locale
294      *              the desired locale
295      * @return a {@code Map} containing all display names of {@code field} in
296      *         {@code style} and {@code locale} and their {@code field} values,
297      *         or {@code null} if no display names are defined for {@code field}
298      * @throws NullPointerException
299      *         if {@code locale} is {@code null}
300      * @see Calendar#getDisplayNames(int, int, Locale)
301      */
302     public abstract Map<String, Integer> getDisplayNames(String calendarType,
303                                                          int field, int style,
304                                                          Locale locale);
305 }
