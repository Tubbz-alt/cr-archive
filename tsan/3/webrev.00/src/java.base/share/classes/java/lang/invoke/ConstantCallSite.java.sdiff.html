 26 package java.lang.invoke;

 28 /**
 29  * A {@code ConstantCallSite} is a {@link CallSite} whose target is permanent, and can never be changed.
 30  * An {@code invokedynamic} instruction linked to a {@code ConstantCallSite} is permanently
 31  * bound to the call site's target.
 32  * @author John Rose, JSR 292 EG
 33  * @since 1.7
 34  */
 35 public class ConstantCallSite extends CallSite {
 36     private final boolean isFrozen;



 37 
 38     /**
 39      * Creates a call site with a permanent target.
 40      * @param target the target to be permanently associated with this call site
 41      * @throws NullPointerException if the proposed target is null
 42      */
 43     public ConstantCallSite(MethodHandle target) {
 44         super(target);
 45         isFrozen = true;

 46     }
 47 
 48     /**
 49      * Creates a call site with a permanent target, possibly bound to the call site itself.
 50      * <p>
 51      * During construction of the call site, the {@code createTargetHook} is invoked to
 52      * produce the actual target, as if by a call of the form
 53      * {@code (MethodHandle) createTargetHook.invoke(this)}.
 54      * <p>
 55      * Note that user code cannot perform such an action directly in a subclass constructor,
 56      * since the target must be fixed before the {@code ConstantCallSite} constructor returns.
 57      * <p>
 58      * The hook is said to bind the call site to a target method handle,
 59      * and a typical action would be {@code someTarget.bindTo(this)}.
 60      * However, the hook is free to take any action whatever,
 61      * including ignoring the call site and returning a constant target.
 62      * <p>
 63      * The result returned by the hook must be a method handle of exactly
 64      * the same type as the call site.
 65      * <p>
 66      * While the hook is being called, the new {@code ConstantCallSite}
 67      * object is in a partially constructed state.
 68      * In this state,
 69      * a call to {@code getTarget}, or any other attempt to use the target,
 70      * will result in an {@code IllegalStateException}.
 71      * It is legal at all times to obtain the call site's type using the {@code type} method.
 72      *
 73      * @param targetType the type of the method handle to be permanently associated with this call site
 74      * @param createTargetHook a method handle to invoke (on the call site) to produce the call site's target
 75      * @throws WrongMethodTypeException if the hook cannot be invoked on the required arguments,
 76      *         or if the target returned by the hook is not of the given {@code targetType}
 77      * @throws NullPointerException if the hook returns a null value
 78      * @throws ClassCastException if the hook returns something other than a {@code MethodHandle}
 79      * @throws Throwable anything else thrown by the hook function
 80      */
 81     protected ConstantCallSite(MethodType targetType, MethodHandle createTargetHook) throws Throwable {
 82         super(targetType, createTargetHook);
 83         isFrozen = true;

 84     }
 85 
 86     /**
 87      * Returns the target method of the call site, which behaves
 88      * like a {@code final} field of the {@code ConstantCallSite}.
 89      * That is, the target is always the original value passed
 90      * to the constructor call which created this instance.
 91      *
 92      * @return the immutable linkage state of this call site, a constant method handle
 93      * @throws IllegalStateException if the {@code ConstantCallSite} constructor has not completed
 94      */
 95     @Override public final MethodHandle getTarget() {
 96         if (!isFrozen)  throw new IllegalStateException();
 97         return target;
 98     }

102      * This kind of call site cannot change its target.

 28 import jdk.internal.misc.Unsafe;

 31 /**
 32  * A {@code ConstantCallSite} is a {@link CallSite} whose target is permanent, and can never be changed.
 33  * An {@code invokedynamic} instruction linked to a {@code ConstantCallSite} is permanently
 34  * bound to the call site's target.
 35  * @author John Rose, JSR 292 EG
 36  * @since 1.7
 37  */
 38 public class ConstantCallSite extends CallSite {
 39     private static final Unsafe UNSAFE = Unsafe.getUnsafe();
 40 
 41     @Stable // should NOT be constant folded during instance initialization (isFrozen == false)
 42     /*final*/ private boolean isFrozen; // Note: This field is known to the JVM.
 43 
 44     /**
 45      * Creates a call site with a permanent target.
 46      * @param target the target to be permanently associated with this call site
 47      * @throws NullPointerException if the proposed target is null
 48      */
 49     public ConstantCallSite(MethodHandle target) {
 50         super(target);
 51         isFrozen = true;
 52         UNSAFE.storeStoreFence(); // properly publish isFrozen update
 53     }
 54 
 55     /**
 56      * Creates a call site with a permanent target, possibly bound to the call site itself.
 57      * <p>
 58      * During construction of the call site, the {@code createTargetHook} is invoked to
 59      * produce the actual target, as if by a call of the form
 60      * {@code (MethodHandle) createTargetHook.invoke(this)}.
 61      * <p>
 62      * Note that user code cannot perform such an action directly in a subclass constructor,
 63      * since the target must be fixed before the {@code ConstantCallSite} constructor returns.
 64      * <p>
 65      * The hook is said to bind the call site to a target method handle,
 66      * and a typical action would be {@code someTarget.bindTo(this)}.
 67      * However, the hook is free to take any action whatever,
 68      * including ignoring the call site and returning a constant target.
 69      * <p>
 70      * The result returned by the hook must be a method handle of exactly
 71      * the same type as the call site.
 72      * <p>
 73      * While the hook is being called, the new {@code ConstantCallSite}
 74      * object is in a partially constructed state.
 75      * In this state,
 76      * a call to {@code getTarget}, or any other attempt to use the target,
 77      * will result in an {@code IllegalStateException}.
 78      * It is legal at all times to obtain the call site's type using the {@code type} method.
 79      *
 80      * @param targetType the type of the method handle to be permanently associated with this call site
 81      * @param createTargetHook a method handle to invoke (on the call site) to produce the call site's target
 82      * @throws WrongMethodTypeException if the hook cannot be invoked on the required arguments,
 83      *         or if the target returned by the hook is not of the given {@code targetType}
 84      * @throws NullPointerException if the hook returns a null value
 85      * @throws ClassCastException if the hook returns something other than a {@code MethodHandle}
 86      * @throws Throwable anything else thrown by the hook function
 87      */
 88     protected ConstantCallSite(MethodType targetType, MethodHandle createTargetHook) throws Throwable {
 89         super(targetType, createTargetHook); // "this" instance leaks into createTargetHook
 90         isFrozen = true;
 91         UNSAFE.storeStoreFence(); // properly publish isFrozen
 92     }

104         if (!isFrozen)  throw new IllegalStateException();
105         return target;
106     }
