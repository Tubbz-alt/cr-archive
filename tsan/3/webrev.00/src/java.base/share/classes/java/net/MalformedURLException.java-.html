26 package java.net;
27 
28 import java.io.IOException;
29 
30 /**
31  * Thrown to indicate that a malformed URL has occurred. Either no
32  * legal protocol could be found in a specification string or the
33  * string could not be parsed.
34  *
35  * @author  Arthur van Hoff
36  * @since   1.0
37  */
38 public class MalformedURLException extends IOException {
39     private static final long serialVersionUID = -182787522200415866L;
40 
41     /**
42      * Constructs a {@code MalformedURLException} with no detail message.
43      */
44     public MalformedURLException() {
45     }
46 
47     /**
48      * Constructs a {@code MalformedURLException} with the
49      * specified detail message.
50      *
51      * @param   msg   the detail message.
52      */
53     public MalformedURLException(String msg) {
54         super(msg);
55     }
56 }
