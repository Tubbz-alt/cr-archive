26 package java.security.cert;
27 
28 /**
29  * Certificate Parsing Exception. This is thrown whenever an
30  * invalid DER-encoded certificate is parsed or unsupported DER features
31  * are found in the Certificate.
32  *
33  * @author Hemma Prafullchandra
34  * @since 1.2
35  */
36 public class CertificateParsingException extends CertificateException {
37 

38     private static final long serialVersionUID = -7989222416793322029L;
39 
40     /**
41      * Constructs a CertificateParsingException with no detail message. A
42      * detail message is a String that describes this particular
43      * exception.
44      */
45     public CertificateParsingException() {
46         super();
47     }
48 
49     /**
50      * Constructs a CertificateParsingException with the specified detail
51      * message. A detail message is a String that describes this
52      * particular exception.
53      *
54      * @param message the detail message.
55      */
56     public CertificateParsingException(String message) {
57         super(message);
58     }
59 
60     /**
61      * Creates a {@code CertificateParsingException} with the specified
62      * detail message and cause.
63      *
64      * @param message the detail message (which is saved for later retrieval
65      *        by the {@link #getMessage()} method).
66      * @param cause the cause (which is saved for later retrieval by the
67      *        {@link #getCause()} method).  (A {@code null} value is permitted,
68      *        and indicates that the cause is nonexistent or unknown.)
69      * @since 1.5
70      */
71     public CertificateParsingException(String message, Throwable cause) {
72         super(message, cause);
73     }
74 
75     /**
76      * Creates a {@code CertificateParsingException} with the
77      * specified cause and a detail message of
78      * {@code (cause==null ? null : cause.toString())}
79      * (which typically contains the class and detail message of
80      * {@code cause}).
81      *
82      * @param cause the cause (which is saved for later retrieval by the
83      *        {@link #getCause()} method).  (A {@code null} value is permitted,
84      *        and indicates that the cause is nonexistent or unknown.)
85      * @since 1.5
86      */
87     public CertificateParsingException(Throwable cause) {
88         super(cause);
89     }
90 }
