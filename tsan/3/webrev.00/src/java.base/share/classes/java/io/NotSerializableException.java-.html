26 package java.io;
27 
28 /**
29  * Thrown when an instance is required to have a Serializable interface.
30  * The serialization runtime or the class of the instance can throw
31  * this exception. The argument should be the name of the class.
32  *
33  * @author  unascribed
34  * @since   1.1
35  */
36 public class NotSerializableException extends ObjectStreamException {
37 
38     private static final long serialVersionUID = 2906642554793891381L;
39 
40     /**
41      * Constructs a NotSerializableException object with message string.
42      *
43      * @param classname Class of the instance being serialized/deserialized.
44      */
45     public NotSerializableException(String classname) {
46         super(classname);
47     }
48 
49     /**
50      *  Constructs a NotSerializableException object.
51      */
52     public NotSerializableException() {
53         super();
54     }
55 }
