26 package java.lang;
27 
28 /**
29  * Thrown when creating a {@linkplain ModuleLayer module layer} fails.
30  *
31  * @see ModuleLayer
32  * @since 9
33  * @spec JPMS
34  */
35 public class LayerInstantiationException extends RuntimeException {

36     private static final long serialVersionUID = -906239691613568347L;
37 
38     /**
39      * Constructs a {@code LayerInstantiationException} with no detail message.
40      */
41     public LayerInstantiationException() {
42     }
43 
44     /**
45      * Constructs a {@code LayerInstantiationException} with the given detail
46      * message.
47      *
48      * @param msg
49      *        The detail message; can be {@code null}
50      */
51     public LayerInstantiationException(String msg) {
52         super(msg);
53     }
54 
55     /**
56      * Constructs a {@code LayerInstantiationException} with the given cause.
57      *
58      * @param cause
59      *        The cause; can be {@code null}
60      */
61     public LayerInstantiationException(Throwable cause) {
62         super(cause);
63     }
64 
65     /**
66      * Constructs a {@code LayerInstantiationException} with the given detail
67      * message and cause.
68      *
69      * @param msg
70      *        The detail message; can be {@code null}
71      * @param cause
72      *        The cause; can be {@code null}
73      */
74     public LayerInstantiationException(String msg, Throwable cause) {
75         super(msg, cause);
76     }
77 }
