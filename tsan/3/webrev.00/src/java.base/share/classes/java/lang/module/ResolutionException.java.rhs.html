26 package java.lang.module;
27 
28 /**
29  * Thrown when resolving a set of modules, or resolving a set of modules with
30  * service binding, fails.
31  *
32  * @see Configuration
33  * @since 9
34  * @spec JPMS
35  */
36 public class ResolutionException extends RuntimeException {
37     @java.io.Serial
38     private static final long serialVersionUID = -1031186845316729450L;
39 
40     /**
41      * Constructs a {@code ResolutionException} with no detail message.
42      */
43     public ResolutionException() { }
44 
45     /**
46      * Constructs a {@code ResolutionException} with the given detail
47      * message.
48      *
49      * @param msg
50      *        The detail message; can be {@code null}
51      */
52     public ResolutionException(String msg) {
53         super(msg);
54     }
55 
56     /**
57      * Constructs an instance of this exception with the given cause.
58      *
59      * @param cause
60      *        The cause; can be {@code null}
61      */
62     public ResolutionException(Throwable cause) {
63         super(cause);
64     }
65 
66     /**
67      * Constructs a {@code ResolutionException} with the given detail message
68      * and cause.
69      *
70      * @param msg
71      *        The detail message; can be {@code null}
72      * @param cause
73      *        The cause; can be {@code null}
74      */
75     public ResolutionException(String msg, Throwable cause) {
76         super(msg, cause);
77     }
78 
79 }
