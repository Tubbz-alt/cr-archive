26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when the precision is a negative value other than
30  * {@code -1}, the conversion does not support a precision, or the value is
31  * otherwise unsupported.
32  *
33  * @since 1.5
34  */
35 public class IllegalFormatPrecisionException extends IllegalFormatException {
36 
37     @java.io.Serial
38     private static final long serialVersionUID = 18711008L;
39 
40     private int p;
41 
42     /**
43      * Constructs an instance of this class with the specified precision.
44      *
45      * @param  p
46      *         The precision
47      */
48     public IllegalFormatPrecisionException(int p) {
49         this.p = p;
50     }
51 
52     /**
53      * Returns the precision
54      *
55      * @return  The precision
56      */
57     public int getPrecision() {
58         return p;
59     }
60 
61     public String getMessage() {
62         return Integer.toString(p);
63     }
64 }
