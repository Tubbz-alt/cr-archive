26 package java.net;
27 
28 import java.io.IOException;
29 
30 /**
31  * Thrown to indicate that there is an error in the underlying
32  * protocol, such as a TCP error.
33  *
34  * @author  Chris Warth
35  * @since   1.0
36  */
37 public
38 class ProtocolException extends IOException {
39     private static final long serialVersionUID = -6098449442062388080L;
40 
41     /**
42      * Constructs a new {@code ProtocolException} with the
43      * specified detail message.
44      *
45      * @param   message   the detail message.
46      */
47     public ProtocolException(String message) {
48         super(message);
49     }
50 
51     /**
52      * Constructs a new {@code ProtocolException} with no detail message.
53      */
54     public ProtocolException() {
55     }
56 }
