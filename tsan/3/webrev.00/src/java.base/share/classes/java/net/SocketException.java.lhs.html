26 package java.net;
27 
28 import java.io.IOException;
29 
30 /**
31  * Thrown to indicate that there is an error creating or accessing a Socket.
32  *
33  * @author  Jonathan Payne
34  * @since   1.0
35  */
36 public
37 class SocketException extends IOException {
38     private static final long serialVersionUID = -5935874303556886934L;
39 
40     /**
41      * Constructs a new {@code SocketException} with the
42      * specified detail message.
43      *
44      * @param msg the detail message.
45      */
46     public SocketException(String msg) {
47         super(msg);
48     }
49 
50     /**
51      * Constructs a new {@code SocketException} with no detail message.
52      */
53     public SocketException() {
54     }
55 }
