26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when a conversion and flag are incompatible.
30  *
31  * <p> Unless otherwise specified, passing a {@code null} argument to any
32  * method or constructor in this class will cause a {@link
33  * NullPointerException} to be thrown.
34  *
35  * @since 1.5
36  */
37 public class FormatFlagsConversionMismatchException
38     extends IllegalFormatException
39 {
40     private static final long serialVersionUID = 19120414L;
41 
42     private String f;
43 
44     private char c;
45 
46     /**
47      * Constructs an instance of this class with the specified flag
48      * and conversion.
49      *
50      * @param  f
51      *         The flag
52      *
53      * @param  c
54      *         The conversion
55      */
56     public FormatFlagsConversionMismatchException(String f, char c) {
57         if (f == null)
58             throw new NullPointerException();
59         this.f = f;
60         this.c = c;
61     }
62 
63     /**
64      * Returns the incompatible flag.
65      *
66      * @return  The flag
67      */
68      public String getFlags() {
69         return f;
70     }
71 
72     /**
73      * Returns the incompatible conversion.
74      *
75      * @return  The conversion
76      */
77     public char getConversion() {
78         return c;
79     }
80 
81     public String getMessage() {
82         return "Conversion = " + c + ", Flags = " + f;
83     }
84 }
