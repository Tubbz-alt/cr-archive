26 package java.io;
27 
28 /**
29  * Thrown when the Serialization runtime detects one of the following
30  * problems with a Class.
31  * <UL>
32  * <LI> The serial version of the class does not match that of the class
33  *      descriptor read from the stream
34  * <LI> The class contains unknown datatypes
35  * <LI> The class does not have an accessible no-arg constructor
36  * <LI> The ObjectStreamClass of an enum constant does not represent
37  *      an enum type
38  * <LI> Other conditions given in the <cite>Java Object Serialization
39  *      Specification</cite>
40  * </UL>
41  *
42  * @author  unascribed
43  * @since   1.1
44  */
45 public class InvalidClassException extends ObjectStreamException {
46 
47     @java.io.Serial
48     private static final long serialVersionUID = -4333316296251054416L;
49 
50     /**
51      * Name of the invalid class.
52      *
53      * @serial Name of the invalid class.
54      */
55     public String classname;
56 
57     /**
58      * Report an InvalidClassException for the reason specified.
59      *
60      * @param reason  String describing the reason for the exception.
61      */
62     public InvalidClassException(String reason) {
63         super(reason);
64     }
65 
66     /**
67      * Constructs an InvalidClassException object.
68      *
69      * @param cname   a String naming the invalid class.
70      * @param reason  a String describing the reason for the exception.
71      */
72     public InvalidClassException(String cname, String reason) {
73         super(reason);
74         classname = cname;
75     }
76 
77     /**
78      * Produce the message and include the classname, if present.
79      */
80     public String getMessage() {
81         if (classname == null)
82             return super.getMessage();
83         else
84             return classname + "; " + super.getMessage();
85     }
86 }
