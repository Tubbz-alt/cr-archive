26 package java.lang;
27 
28 /**
29  * Thrown when a particular method cannot be found.
30  *
31  * @author     unascribed
32  * @since      1.0
33  */
34 public
35 class NoSuchMethodException extends ReflectiveOperationException {
36     private static final long serialVersionUID = 5034388446362600923L;
37 
38     /**
39      * Constructs a <code>NoSuchMethodException</code> without a detail message.
40      */
41     public NoSuchMethodException() {
42         super();
43     }
44 
45     /**
46      * Constructs a <code>NoSuchMethodException</code> with a detail message.
47      *
48      * @param      s   the detail message.
49      */
50     public NoSuchMethodException(String s) {
51         super(s);
52     }
53 }
