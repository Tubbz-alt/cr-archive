26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when an unknown conversion is given.
30  *
31  * <p> Unless otherwise specified, passing a {@code null} argument to
32  * any method or constructor in this class will cause a {@link
33  * NullPointerException} to be thrown.
34  *
35  * @since 1.5
36  */
37 public class UnknownFormatConversionException extends IllegalFormatException {
38 
39     private static final long serialVersionUID = 19060418L;
40 
41     private String s;
42 
43     /**
44      * Constructs an instance of this class with the unknown conversion.
45      *
46      * @param  s
47      *         Unknown conversion
48      */
49     public UnknownFormatConversionException(String s) {
50         if (s == null)
51             throw new NullPointerException();
52         this.s = s;
53     }
54 
55     /**
56      * Returns the unknown conversion.
57      *
58      * @return  The unknown conversion.
59      */
60     public String getConversion() {
61         return s;
62     }
63 
64     // javadoc inherited from Throwable.java
65     public String getMessage() {
66         return String.format("Conversion = '%s'", s);
67     }
68 }
