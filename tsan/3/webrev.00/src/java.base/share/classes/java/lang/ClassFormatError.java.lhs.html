26 package java.lang;
27 
28 /**
29  * Thrown when the Java Virtual Machine attempts to read a class
30  * file and determines that the file is malformed or otherwise cannot
31  * be interpreted as a class file.
32  *
33  * @author  unascribed
34  * @since   1.0
35  */
36 public
37 class ClassFormatError extends LinkageError {
38     private static final long serialVersionUID = -8420114879011949195L;
39 
40     /**
41      * Constructs a <code>ClassFormatError</code> with no detail message.
42      */
43     public ClassFormatError() {
44         super();
45     }
46 
47     /**
48      * Constructs a <code>ClassFormatError</code> with the specified
49      * detail message.
50      *
51      * @param   s   the detail message.
52      */
53     public ClassFormatError(String s) {
54         super(s);
55     }
56 }
