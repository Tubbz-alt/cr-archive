25 package java.nio.charset;
26 
27 /**
28  * Constant definitions for the standard {@link Charset charsets}. These
29  * charsets are guaranteed to be available on every implementation of the Java
30  * platform.
31  *
32  * @see <a href="Charset.html#standard">Standard Charsets</a>
33  * @since 1.7
34  */
35 public final class StandardCharsets {
36 
37     // To avoid accidental eager initialization of often unused Charsets
38     // from happening while the VM is booting up, which may delay
39     // initialization of VM components, we should generally avoid depending
40     // on this class from elsewhere in java.base.
41 
42     private StandardCharsets() {
43         throw new AssertionError("No java.nio.charset.StandardCharsets instances for you!");
44     }
45 
46     /**
47      * Seven-bit ASCII, also known as ISO646-US, also known as the
48      * Basic Latin block of the Unicode character set.
49      */
50     public static final Charset US_ASCII = sun.nio.cs.US_ASCII.INSTANCE;
51 
52     /**
53      * ISO Latin Alphabet {@literal No. 1}, also known as ISO-LATIN-1.
54      */
55     public static final Charset ISO_8859_1 = sun.nio.cs.ISO_8859_1.INSTANCE;
56 
57     /**
58      * Eight-bit UCS Transformation Format.
59      */
60     public static final Charset UTF_8 = sun.nio.cs.UTF_8.INSTANCE;
61 
62     /**
63      * Sixteen-bit UCS Transformation Format, big-endian byte order.
64      */
65     public static final Charset UTF_16BE = new sun.nio.cs.UTF_16BE();
66 
67     /**
68      * Sixteen-bit UCS Transformation Format, little-endian byte order.
69      */
70     public static final Charset UTF_16LE = new sun.nio.cs.UTF_16LE();
71 
72     /**
73      * Sixteen-bit UCS Transformation Format, byte order identified by an
74      * optional byte-order mark.
75      */
76     public static final Charset UTF_16 = new sun.nio.cs.UTF_16();
77 }
