26 package java.util.zip;
27 
28 import java.io.FilterOutputStream;
29 import java.io.OutputStream;
30 import java.io.IOException;
31 
32 /**
33  * An output stream that also maintains a checksum of the data being
34  * written. The checksum can then be used to verify the integrity of
35  * the output data.
36  *
37  * @see         Checksum
38  * @author      David Connelly
39  * @since 1.1
40  */
41 public
42 class CheckedOutputStream extends FilterOutputStream {
43     private Checksum cksum;
44 
45     /**
46      * Creates an output stream with the specified Checksum.
47      * @param out the output stream
48      * @param cksum the checksum
49      */
50     public CheckedOutputStream(OutputStream out, Checksum cksum) {
51         super(out);
52         this.cksum = cksum;
53     }
54 
55     /**
56      * Writes a byte. Will block until the byte is actually written.
57      * @param b the byte to be written
58      * @exception IOException if an I/O error has occurred
59      */
60     public void write(int b) throws IOException {
61         out.write(b);
62         cksum.update(b);
63     }
64 
65     /**
66      * Writes an array of bytes. Will block until the bytes are
67      * actually written.
68      * @param b the data to be written
69      * @param off the start offset of the data
70      * @param len the number of bytes to be written
71      * @exception IOException if an I/O error has occurred
72      */
73     public void write(byte[] b, int off, int len) throws IOException {
74         out.write(b, off, len);
75         cksum.update(b, off, len);
76     }
77 
78     /**
79      * Returns the Checksum for this output stream.
80      * @return the Checksum
81      */
82     public Checksum getChecksum() {
83         return cksum;
84     }
85 }
