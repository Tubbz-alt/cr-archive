26 package java.security.cert;
27 
28 /**
29  * Certificate Parsing Exception. This is thrown whenever an
30  * invalid DER-encoded certificate is parsed or unsupported DER features
31  * are found in the Certificate.
32  *
33  * @author Hemma Prafullchandra
34  * @since 1.2
35  */
36 public class CertificateParsingException extends CertificateException {
37 
38     @java.io.Serial
39     private static final long serialVersionUID = -7989222416793322029L;
40 
41     /**
42      * Constructs a CertificateParsingException with no detail message. A
43      * detail message is a String that describes this particular
44      * exception.
45      */
46     public CertificateParsingException() {
47         super();
48     }
49 
50     /**
51      * Constructs a CertificateParsingException with the specified detail
52      * message. A detail message is a String that describes this
53      * particular exception.
54      *
55      * @param message the detail message.
56      */
57     public CertificateParsingException(String message) {
58         super(message);
59     }
60 
61     /**
62      * Creates a {@code CertificateParsingException} with the specified
63      * detail message and cause.
64      *
65      * @param message the detail message (which is saved for later retrieval
66      *        by the {@link #getMessage()} method).
67      * @param cause the cause (which is saved for later retrieval by the
68      *        {@link #getCause()} method).  (A {@code null} value is permitted,
69      *        and indicates that the cause is nonexistent or unknown.)
70      * @since 1.5
71      */
72     public CertificateParsingException(String message, Throwable cause) {
73         super(message, cause);
74     }
75 
76     /**
77      * Creates a {@code CertificateParsingException} with the
78      * specified cause and a detail message of
79      * {@code (cause==null ? null : cause.toString())}
80      * (which typically contains the class and detail message of
81      * {@code cause}).
82      *
83      * @param cause the cause (which is saved for later retrieval by the
84      *        {@link #getCause()} method).  (A {@code null} value is permitted,
85      *        and indicates that the cause is nonexistent or unknown.)
86      * @since 1.5
87      */
88     public CertificateParsingException(Throwable cause) {
89         super(cause);
90     }
91 }
