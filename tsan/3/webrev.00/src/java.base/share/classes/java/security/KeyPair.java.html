26 package java.security;
27 
28 import java.util.*;
29 
30 /**
31  * This class is a simple holder for a key pair (a public key and a
32  * private key). It does not enforce any security, and, when initialized,
33  * should be treated like a PrivateKey.
34  *
35  * @see PublicKey
36  * @see PrivateKey
37  *
38  * @author Benjamin Renaud
39  * @since 1.1
40  */
41 
42 public final class KeyPair implements java.io.Serializable {
43 
44     @java.io.Serial
45     private static final long serialVersionUID = -7565189502268009837L;
46 
47     private PrivateKey privateKey;
48     private PublicKey publicKey;
49 
50     /**
51      * Constructs a key pair from the given public key and private key.
52      *
53      * <p>Note that this constructor only stores references to the public
54      * and private key components in the generated key pair. This is safe,
55      * because {@code Key} objects are immutable.
56      *
57      * @param publicKey the public key.
58      *
59      * @param privateKey the private key.
60      */
61     public KeyPair(PublicKey publicKey, PrivateKey privateKey) {
62         this.publicKey = publicKey;
63         this.privateKey = privateKey;
64     }
65 
66     /**
67      * Returns a reference to the public key component of this key pair.
68      *
69      * @return a reference to the public key.
70      */
71     public PublicKey getPublic() {
72         return publicKey;
73     }
74 
75      /**
76      * Returns a reference to the private key component of this key pair.
77      *
78      * @return a reference to the private key.
79      */
80    public PrivateKey getPrivate() {
81         return privateKey;
82     }
83 }
