 26 package java.net;
 27 
 28 import java.io.FileDescriptor;
 29 import java.io.IOException;
 30 import java.util.Objects;
 31 import java.util.Set;
 32 
 33 /**
 34  * Abstract datagram and multicast socket implementation base class.
 35  * @author Pavani Diwanji
 36  * @since  1.1
 37  */
 38 
 39 public abstract class DatagramSocketImpl implements SocketOptions {
 40 
 41     /**
 42      * The local port number.
 43      */
 44     protected int localPort;
 45 
 46     /**
 47      * The file descriptor object.
 48      */
 49     protected FileDescriptor fd;
 50 
 51     int dataAvailable() {
 52         // default impl returns zero, which disables the calling
 53         // functionality
 54         return 0;
 55     }
 56 
 57     /**
 58      * Creates a datagram socket.
 59      * @throws    SocketException if there is an error in the
 60      * underlying protocol, such as a TCP error.
 61      */
 62     protected abstract void create() throws SocketException;
 63 
 64     /**
 65      * Binds a datagram socket to a local port and address.
 66      * @param     lport the local port
 67      * @param     laddr the local address
 68      * @throws    SocketException if there is an error in the
 69      *            underlying protocol, such as a TCP error.
 70      */
 71     protected abstract void bind(int lport, InetAddress laddr) throws SocketException;
 72 
 73     /**
 74      * Sends a datagram packet. The packet contains the data and the
 75      * destination address to send the packet to.
 76      * @param    p the packet to be sent.
 77      * @throws   IOException if an I/O exception occurs while sending the
 78      *           datagram packet.
 79      * @throws   PortUnreachableException may be thrown if the socket is connected
 80      *           to a currently unreachable destination. Note, there is no guarantee that
 81      *           the exception will be thrown.
 82      */
 83     protected abstract void send(DatagramPacket p) throws IOException;
 84 
 85     /**
 86      * Connects a datagram socket to a remote destination. This associates the remote
 87      * address with the local socket so that datagrams may only be sent to this destination
 88      * and received from this destination. This may be overridden to call a native
 89      * system connect.
 90      *
 91      * <p>If the remote destination to which the socket is connected does not
 92      * exist, or is otherwise unreachable, and if an ICMP destination unreachable
 93      * packet has been received for that address, then a subsequent call to
 94      * send or receive may throw a PortUnreachableException.
 95      * Note, there is no guarantee that the exception will be thrown.
 96      * @param   address the remote InetAddress to connect to
 97      * @param   port the remote port number
 98      * @throws  SocketException may be thrown if the socket cannot be
 99      *          connected to the remote destination
100      * @since   1.4
101      */
102     protected void connect(InetAddress address, int port) throws SocketException {}
103 
104     /**
105      * Disconnects a datagram socket from its remote destination.
106      * @since 1.4
107      */
108     protected void disconnect() {}
109 
110     /**
111      * Peek at the packet to see who it is from. Updates the specified {@code InetAddress}
112      * to the address which the packet came from.
113      * @param     i an InetAddress object
114      * @return    the port number which the packet came from.
115      * @throws    IOException if an I/O exception occurs
116      * @throws    PortUnreachableException may be thrown if the socket is connected
117      *            to a currently unreachable destination. Note, there is no guarantee that the
118      *            exception will be thrown.
119      */
120     protected abstract int peek(InetAddress i) throws IOException;
121 
122     /**
123      * Peek at the packet to see who it is from. The data is copied into the specified
124      * {@code DatagramPacket}. The data is returned,
125      * but not consumed, so that a subsequent peekData/receive operation
126      * will see the same data.
127      * @param     p the Packet Received.
128      * @return    the port number which the packet came from.
129      * @throws    IOException if an I/O exception occurs
130      * @throws    PortUnreachableException may be thrown if the socket is connected
131      *            to a currently unreachable destination. Note, there is no guarantee that the
132      *            exception will be thrown.
133      * @since 1.4
134      */
135     protected abstract int peekData(DatagramPacket p) throws IOException;
136     /**
137      * Receive the datagram packet.
138      * @param     p the Packet Received.
139      * @throws    IOException if an I/O exception occurs
140      *            while receiving the datagram packet.
141      * @throws    PortUnreachableException may be thrown if the socket is connected
142      *            to a currently unreachable destination. Note, there is no guarantee that the
143      *            exception will be thrown.
144      */
145     protected abstract void receive(DatagramPacket p) throws IOException;
146 
147     /**
148      * Set the TTL (time-to-live) option.
149      * @param ttl a byte specifying the TTL value
150      *
151      * @deprecated use setTimeToLive instead.
152      * @throws    IOException if an I/O exception occurs while setting
153      * the time-to-live option.
154      * @see #getTTL()
155      */
156     @Deprecated
157     protected abstract void setTTL(byte ttl) throws IOException;
158 
159     /**
160      * Retrieve the TTL (time-to-live) option.
161      *
162      * @throws    IOException if an I/O exception occurs
163      * while retrieving the time-to-live option
164      * @deprecated use getTimeToLive instead.
165      * @return a byte representing the TTL value
166      * @see #setTTL(byte)
167      */
168     @Deprecated
169     protected abstract byte getTTL() throws IOException;
170 
171     /**
172      * Set the TTL (time-to-live) option.
173      * @param ttl an {@code int} specifying the time-to-live value
174      * @throws    IOException if an I/O exception occurs
175      * while setting the time-to-live option.
176      * @see #getTimeToLive()
177      */
178     protected abstract void setTimeToLive(int ttl) throws IOException;
179 
180     /**
181      * Retrieve the TTL (time-to-live) option.
182      * @throws    IOException if an I/O exception occurs
183      * while retrieving the time-to-live option
184      * @return an {@code int} representing the time-to-live value
185      * @see #setTimeToLive(int)
186      */
187     protected abstract int getTimeToLive() throws IOException;
188 
189     /**
190      * Join the multicast group.
191      * @param inetaddr multicast address to join.
192      * @throws    IOException if an I/O exception occurs
193      * while joining the multicast group.
194      */
195     protected abstract void join(InetAddress inetaddr) throws IOException;
196 
197     /**
198      * Leave the multicast group.
199      * @param inetaddr multicast address to leave.
200      * @throws    IOException if an I/O exception occurs
201      * while leaving the multicast group.
202      */
203     protected abstract void leave(InetAddress inetaddr) throws IOException;
204 
205     /**
206      * Join the multicast group.
207      * @param mcastaddr address to join.
208      * @param netIf specifies the local interface to receive multicast
209      *        datagram packets
210      * @throws IOException if an I/O exception occurs while joining
211      * the multicast group
212      * @since 1.4
213      */
214     protected abstract void joinGroup(SocketAddress mcastaddr,
215                                       NetworkInterface netIf)
216         throws IOException;
217 
218     /**
219      * Leave the multicast group.
220      * @param mcastaddr address to leave.
221      * @param netIf specified the local interface to leave the group at
222      * @throws IOException if an I/O exception occurs while leaving
223      * the multicast group
224      * @since 1.4
225      */
226     protected abstract void leaveGroup(SocketAddress mcastaddr,
227                                        NetworkInterface netIf)
228         throws IOException;
229 
230     /**
231      * Close the socket.
232      */
233     protected abstract void close();
234 
235     /**
236      * Gets the local port.
237      * @return an {@code int} representing the local port value
238      */
239     protected int getLocalPort() {
240         return localPort;
241     }
242 
243     /**
244      * Gets the datagram socket file descriptor.
245      * @return a {@code FileDescriptor} object representing the datagram socket
246      * file descriptor
247      */
248     protected FileDescriptor getFileDescriptor() {
249         return fd;
250     }
251 
252     /**
253      * Called to set a socket option.
254      *
255      * @implSpec
256      * The default implementation of this method first checks that the given
257      * socket option {@code name} is not null, then throws {@code
258      * UnsupportedOperationException}. Subclasses should override this method
259      * with an appropriate implementation.
260      *
261      * @param  <T> The type of the socket option value
262      * @param  name The socket option
263      * @param  value The value of the socket option. A value of {@code null}
264      *              may be valid for some options.
265      *
266      * @throws UnsupportedOperationException if the DatagramSocketImpl does not
267      *         support the option
268      * @throws IllegalArgumentException if the value is not valid for
269      *         the option
270      * @throws IOException if an I/O error occurs, or if the socket is closed
271      * @throws NullPointerException if name is {@code null}
272      *
273      * @since 9
274      */
275     protected <T> void setOption(SocketOption<T> name, T value) throws IOException {
276         Objects.requireNonNull(name);
277         throw new UnsupportedOperationException("'" + name + "' not supported");
278     }
279 
280     /**
281      * Called to get a socket option.
282      *
283      * @implSpec
284      * The default implementation of this method first checks that the given
285      * socket option {@code name} is not null, then throws {@code
286      * UnsupportedOperationException}. Subclasses should override this method
287      * with an appropriate implementation.
288      *
289      * @param  <T> The type of the socket option value
290      * @param  name The socket option
291      * @return the socket option
292      *
293      * @throws UnsupportedOperationException if the DatagramSocketImpl does not
294      *         support the option
295      * @throws IOException if an I/O error occurs, or if the socket is closed
296      * @throws NullPointerException if name is {@code null}
297      *
298      * @since 9
299      */
300     protected <T> T getOption(SocketOption<T> name) throws IOException {
301         Objects.requireNonNull(name);
302         throw new UnsupportedOperationException("'" + name + "' not supported");
303     }
304 
305     /**
306      * Returns a set of SocketOptions supported by this impl
307      * and by this impl's socket (DatagramSocket or MulticastSocket)
308      *
309      * @implSpec
310      * The default implementation of this method returns an empty set.
311      * Subclasses should override this method with an appropriate implementation.
312      *
313      * @return a Set of SocketOptions
314      *
315      * @since 9
316      */
317     protected Set<SocketOption<?>> supportedOptions() {
318         return Set.of();
319     }
320 }
