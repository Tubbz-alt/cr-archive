26 package java.lang;
27 
28 /**
29  * Thrown when creating a {@linkplain ModuleLayer module layer} fails.
30  *
31  * @see ModuleLayer
32  * @since 9
33  * @spec JPMS
34  */
35 public class LayerInstantiationException extends RuntimeException {
36     @java.io.Serial
37     private static final long serialVersionUID = -906239691613568347L;
38 
39     /**
40      * Constructs a {@code LayerInstantiationException} with no detail message.
41      */
42     public LayerInstantiationException() {
43     }
44 
45     /**
46      * Constructs a {@code LayerInstantiationException} with the given detail
47      * message.
48      *
49      * @param msg
50      *        The detail message; can be {@code null}
51      */
52     public LayerInstantiationException(String msg) {
53         super(msg);
54     }
55 
56     /**
57      * Constructs a {@code LayerInstantiationException} with the given cause.
58      *
59      * @param cause
60      *        The cause; can be {@code null}
61      */
62     public LayerInstantiationException(Throwable cause) {
63         super(cause);
64     }
65 
66     /**
67      * Constructs a {@code LayerInstantiationException} with the given detail
68      * message and cause.
69      *
70      * @param msg
71      *        The detail message; can be {@code null}
72      * @param cause
73      *        The cause; can be {@code null}
74      */
75     public LayerInstantiationException(String msg, Throwable cause) {
76         super(msg, cause);
77     }
78 }
