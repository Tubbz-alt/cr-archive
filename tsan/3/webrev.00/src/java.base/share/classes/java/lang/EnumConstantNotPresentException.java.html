26 package java.lang;
27 
28 /**
29  * Thrown when an application tries to access an enum constant by name
30  * and the enum type contains no constant with the specified name.
31  * This exception can be thrown by the {@linkplain
32  * java.lang.reflect.AnnotatedElement API used to read annotations
33  * reflectively}.
34  *
35  * @author  Josh Bloch
36  * @see     java.lang.reflect.AnnotatedElement
37  * @since   1.5
38  */
39 @SuppressWarnings("rawtypes") /* rawtypes are part of the public api */
40 public class EnumConstantNotPresentException extends RuntimeException {
41     @java.io.Serial
42     private static final long serialVersionUID = -6046998521960521108L;
43 
44     /**
45      * The type of the missing enum constant.
46      */
47     private Class<? extends Enum> enumType;
48 
49     /**
50      * The name of the missing enum constant.
51      */
52     private String constantName;
53 
54     /**
55      * Constructs an {@code EnumConstantNotPresentException} for the
56      * specified constant.
57      *
58      * @param enumType the type of the missing enum constant
59      * @param constantName the name of the missing enum constant
60      */
61     public EnumConstantNotPresentException(Class<? extends Enum> enumType,
62                                            String constantName) {
63         super(enumType.getName() + "." + constantName);
64         this.enumType = enumType;
65         this.constantName  = constantName;
66     }
67 
68     /**
69      * Returns the type of the missing enum constant.
70      *
71      * @return the type of the missing enum constant
72      */
73     public Class<? extends Enum> enumType() { return enumType; }
74 
75     /**
76      * Returns the name of the missing enum constant.
77      *
78      * @return the name of the missing enum constant
79      */
80     public String constantName() { return constantName; }
81 }
