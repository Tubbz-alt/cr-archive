26 package java.security;
27 
28 /**
29  * This exception is thrown if an entry in the keystore cannot be recovered.
30  *
31  *
32  * @since 1.5
33  */
34 
35 public class UnrecoverableEntryException extends GeneralSecurityException {
36 

37     private static final long serialVersionUID = -4527142945246286535L;
38 
39     /**
40      * Constructs an UnrecoverableEntryException with no detail message.
41      */
42     public UnrecoverableEntryException() {
43         super();
44     }
45 
46     /**
47      * Constructs an UnrecoverableEntryException with the specified detail
48      * message, which provides more information about why this exception
49      * has been thrown.
50      *
51      * @param msg the detail message.
52      */
53    public UnrecoverableEntryException(String msg) {
54        super(msg);
55     }
56 }
