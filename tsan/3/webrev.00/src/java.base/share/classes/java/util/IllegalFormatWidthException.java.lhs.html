26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when the format width is a negative value other
30  * than {@code -1} or is otherwise unsupported.
31  *
32  * @since 1.5
33  */
34 public class IllegalFormatWidthException extends IllegalFormatException {
35 

36     private static final long serialVersionUID = 16660902L;
37 
38     private int w;
39 
40     /**
41      * Constructs an instance of this class with the specified width.
42      *
43      * @param  w
44      *         The width
45      */
46     public IllegalFormatWidthException(int w) {
47         this.w = w;
48     }
49 
50     /**
51      * Returns the width
52      *
53      * @return  The width
54      */
55     public int getWidth() {
56         return w;
57     }
58 
59     public String getMessage() {
60         return Integer.toString(w);
61     }
62 }
