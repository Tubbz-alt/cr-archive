26 package java.security.spec;
27 
28 import java.security.GeneralSecurityException;
29 
30 /**
31  * This is the exception for invalid key specifications.
32  *
33  * @author Jan Luehe
34  *
35  *
36  * @see KeySpec
37  *
38  * @since 1.2
39  */
40 
41 public class InvalidKeySpecException extends GeneralSecurityException {
42 
43     @java.io.Serial
44     private static final long serialVersionUID = 3546139293998810778L;
45 
46     /**
47      * Constructs an InvalidKeySpecException with no detail message. A
48      * detail message is a String that describes this particular
49      * exception.
50      */
51     public InvalidKeySpecException() {
52         super();
53     }
54 
55     /**
56      * Constructs an InvalidKeySpecException with the specified detail
57      * message. A detail message is a String that describes this
58      * particular exception.
59      *
60      * @param msg the detail message.
61      */
62     public InvalidKeySpecException(String msg) {
63         super(msg);
64     }
65 
66     /**
67      * Creates an {@code InvalidKeySpecException} with the specified
68      * detail message and cause.
69      *
70      * @param message the detail message (which is saved for later retrieval
71      *        by the {@link #getMessage()} method).
72      * @param cause the cause (which is saved for later retrieval by the
73      *        {@link #getCause()} method).  (A {@code null} value is permitted,
74      *        and indicates that the cause is nonexistent or unknown.)
75      * @since 1.5
76      */
77     public InvalidKeySpecException(String message, Throwable cause) {
78         super(message, cause);
79     }
80 
81     /**
82      * Creates an {@code InvalidKeySpecException} with the specified cause
83      * and a detail message of {@code (cause==null ? null : cause.toString())}
84      * (which typically contains the class and detail message of
85      * {@code cause}).
86      *
87      * @param cause the cause (which is saved for later retrieval by the
88      *        {@link #getCause()} method).  (A {@code null} value is permitted,
89      *        and indicates that the cause is nonexistent or unknown.)
90      * @since 1.5
91      */
92     public InvalidKeySpecException(Throwable cause) {
93         super(cause);
94     }
95 }
