 26 package java.io;
 27 
 28 /**
 29  * A {@code FilterInputStream} contains
 30  * some other input stream, which it uses as
 31  * its  basic source of data, possibly transforming
 32  * the data along the way or providing  additional
 33  * functionality. The class {@code FilterInputStream}
 34  * itself simply overrides all  methods of
 35  * {@code InputStream} with versions that
 36  * pass all requests to the contained  input
 37  * stream. Subclasses of {@code FilterInputStream}
 38  * may further override some of  these methods
 39  * and may also provide additional methods
 40  * and fields.
 41  *
 42  * @author  Jonathan Payne
 43  * @since   1.0
 44  */
 45 public class FilterInputStream extends InputStream {

 46     /**
 47      * The input stream to be filtered.
 48      */
 49     protected volatile InputStream in;
 50 
 51     /**
 52      * Creates a {@code FilterInputStream}
 53      * by assigning the  argument {@code in}
 54      * to the field {@code this.in} so as
 55      * to remember it for later use.
 56      *
 57      * @param   in   the underlying input stream, or {@code null} if
 58      *          this instance is to be created without an underlying stream.
 59      */
 60     protected FilterInputStream(InputStream in) {
 61         this.in = in;
 62     }
 63 
 64     /**
 65      * Reads the next byte of data from this input stream. The value
 66      * byte is returned as an {@code int} in the range
 67      * {@code 0} to {@code 255}. If no byte is available
 68      * because the end of the stream has been reached, the value
 69      * {@code -1} is returned. This method blocks until input data
 70      * is available, the end of the stream is detected, or an exception
 71      * is thrown.
 72      * <p>
 73      * This method
 74      * simply performs {@code in.read()} and returns the result.
 75      *
 76      * @return     the next byte of data, or {@code -1} if the end of the
 77      *             stream is reached.
 78      * @throws     IOException  if an I/O error occurs.
 79      * @see        java.io.FilterInputStream#in
 80      */
 81     public int read() throws IOException {
 82         return in.read();
 83     }
 84 
 85     /**
 86      * Reads up to {@code b.length} bytes of data from this
 87      * input stream into an array of bytes. This method blocks until some
 88      * input is available.
 89      * <p>
 90      * This method simply performs the call
 91      * {@code read(b, 0, b.length)} and returns
 92      * the  result. It is important that it does
 93      * <i>not</i> do {@code in.read(b)} instead;
 94      * certain subclasses of  {@code FilterInputStream}
 95      * depend on the implementation strategy actually
 96      * used.
 97      *
 98      * @param      b   the buffer into which the data is read.
 99      * @return     the total number of bytes read into the buffer, or
100      *             {@code -1} if there is no more data because the end of
101      *             the stream has been reached.
102      * @throws     IOException  if an I/O error occurs.
103      * @see        java.io.FilterInputStream#read(byte[], int, int)
104      */
105     public int read(byte b[]) throws IOException {
106         return read(b, 0, b.length);
107     }
108 
109     /**
110      * Reads up to {@code len} bytes of data from this input stream
111      * into an array of bytes. If {@code len} is not zero, the method
112      * blocks until some input is available; otherwise, no
113      * bytes are read and {@code 0} is returned.
114      * <p>
115      * This method simply performs {@code in.read(b, off, len)}
116      * and returns the result.
117      *
118      * @param      b     the buffer into which the data is read.
119      * @param      off   the start offset in the destination array {@code b}
120      * @param      len   the maximum number of bytes read.
121      * @return     the total number of bytes read into the buffer, or
122      *             {@code -1} if there is no more data because the end of
123      *             the stream has been reached.
124      * @throws     NullPointerException If {@code b} is {@code null}.
125      * @throws     IndexOutOfBoundsException If {@code off} is negative,
126      *             {@code len} is negative, or {@code len} is greater than
127      *             {@code b.length - off}
128      * @throws     IOException  if an I/O error occurs.
129      * @see        java.io.FilterInputStream#in
130      */
131     public int read(byte b[], int off, int len) throws IOException {
132         return in.read(b, off, len);
133     }
134 
135     /**
136      * Skips over and discards {@code n} bytes of data from the
137      * input stream. The {@code skip} method may, for a variety of
138      * reasons, end up skipping over some smaller number of bytes,
139      * possibly {@code 0}. The actual number of bytes skipped is
140      * returned.
141      * <p>
142      * This method simply performs {@code in.skip(n)}.
143      *
144      * @param      n   the number of bytes to be skipped.
145      * @return     the actual number of bytes skipped.
146      * @throws     IOException  if {@code in.skip(n)} throws an IOException.
147      */
148     public long skip(long n) throws IOException {
149         return in.skip(n);
150     }
151 
152     /**
153      * Returns an estimate of the number of bytes that can be read (or
154      * skipped over) from this input stream without blocking by the next
155      * caller of a method for this input stream. The next caller might be
156      * the same thread or another thread.  A single read or skip of this
157      * many bytes will not block, but may read or skip fewer bytes.
158      * <p>
159      * This method returns the result of {@link #in in}.available().
160      *
161      * @return     an estimate of the number of bytes that can be read (or skipped
162      *             over) from this input stream without blocking.
163      * @throws     IOException  if an I/O error occurs.
164      */
165     public int available() throws IOException {
166         return in.available();
167     }
168 
169     /**
170      * Closes this input stream and releases any system resources
171      * associated with the stream.
172      * This
173      * method simply performs {@code in.close()}.
174      *
175      * @throws     IOException  if an I/O error occurs.
176      * @see        java.io.FilterInputStream#in
177      */
178     public void close() throws IOException {
179         in.close();
180     }
181 
182     /**
183      * Marks the current position in this input stream. A subsequent
184      * call to the {@code reset} method repositions this stream at
185      * the last marked position so that subsequent reads re-read the same bytes.
186      * <p>
187      * The {@code readlimit} argument tells this input stream to
188      * allow that many bytes to be read before the mark position gets
189      * invalidated.
190      * <p>
191      * This method simply performs {@code in.mark(readlimit)}.
192      *
193      * @param   readlimit   the maximum limit of bytes that can be read before
194      *                      the mark position becomes invalid.
195      * @see     java.io.FilterInputStream#in
196      * @see     java.io.FilterInputStream#reset()
197      */
198     public synchronized void mark(int readlimit) {
199         in.mark(readlimit);
200     }
201 
202     /**
203      * Repositions this stream to the position at the time the
204      * {@code mark} method was last called on this input stream.
205      * <p>
206      * This method
207      * simply performs {@code in.reset()}.
208      * <p>
209      * Stream marks are intended to be used in
210      * situations where you need to read ahead a little to see what's in
211      * the stream. Often this is most easily done by invoking some
212      * general parser. If the stream is of the type handled by the
213      * parse, it just chugs along happily. If the stream is not of
214      * that type, the parser should toss an exception when it fails.
215      * If this happens within readlimit bytes, it allows the outer
216      * code to reset the stream and try another parser.
217      *
218      * @throws     IOException  if the stream has not been marked or if the
219      *               mark has been invalidated.
220      * @see        java.io.FilterInputStream#in
221      * @see        java.io.FilterInputStream#mark(int)
222      */
223     public synchronized void reset() throws IOException {
224         in.reset();
225     }
226 
227     /**
228      * Tests if this input stream supports the {@code mark}
229      * and {@code reset} methods.
230      * This method
231      * simply performs {@code in.markSupported()}.
232      *
233      * @return  {@code true} if this stream type supports the
234      *          {@code mark} and {@code reset} method;
235      *          {@code false} otherwise.
236      * @see     java.io.FilterInputStream#in
237      * @see     java.io.InputStream#mark(int)
238      * @see     java.io.InputStream#reset()
239      */
240     public boolean markSupported() {
241         return in.markSupported();
242     }
243 }
