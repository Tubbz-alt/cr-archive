26 package java.io;
27 
28 /**
29  * Thrown when serialization or deserialization is not active.
30  *
31  * @author  unascribed
32  * @since   1.1
33  */
34 public class NotActiveException extends ObjectStreamException {
35 

36     private static final long serialVersionUID = -3893467273049808895L;
37 
38     /**
39      * Constructor to create a new NotActiveException with the reason given.
40      *
41      * @param reason  a String describing the reason for the exception.
42      */
43     public NotActiveException(String reason) {
44         super(reason);
45     }
46 
47     /**
48      * Constructor to create a new NotActiveException without a reason.
49      */
50     public NotActiveException() {
51         super();
52     }
53 }
