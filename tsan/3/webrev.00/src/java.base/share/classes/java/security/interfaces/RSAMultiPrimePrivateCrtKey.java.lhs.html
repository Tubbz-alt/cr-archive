 26 package java.security.interfaces;
 27 
 28 import java.math.BigInteger;
 29 import java.security.spec.RSAOtherPrimeInfo;
 30 
 31 /**
 32  * The interface to an RSA multi-prime private key, as defined in the
 33  * <a href="https://tools.ietf.org/rfc/rfc8017.txt">PKCS#1 v2.2</a> standard,
 34  * using the <i>Chinese Remainder Theorem</i> (CRT) information values.
 35  *
 36  * @author Valerie Peng
 37  *
 38  *
 39  * @see java.security.spec.RSAPrivateKeySpec
 40  * @see java.security.spec.RSAMultiPrimePrivateCrtKeySpec
 41  * @see RSAPrivateKey
 42  * @see RSAPrivateCrtKey
 43  *
 44  * @since 1.4
 45  */
 46 
 47 public interface RSAMultiPrimePrivateCrtKey extends RSAPrivateKey {
 48 
 49     /**
 50      * The type fingerprint that is set to indicate
 51      * serialization compatibility with a previous
 52      * version of the type.
 53     *
 54     * @deprecated A {@code serialVersionUID} field in an interface is
 55     * ineffectual. Do not use; no replacement.
 56      */
 57     @Deprecated
 58     @SuppressWarnings("serial")

 59     static final long serialVersionUID = 618058533534628008L;
 60 
 61     /**
 62      * Returns the public exponent.
 63      *
 64      * @return the public exponent.
 65      */
 66     public BigInteger getPublicExponent();
 67 
 68     /**
 69      * Returns the primeP.
 70      *
 71      * @return the primeP.
 72      */
 73     public BigInteger getPrimeP();
 74 
 75     /**
 76      * Returns the primeQ.
 77      *
 78      * @return the primeQ.
 79      */
 80     public BigInteger getPrimeQ();
 81 
 82     /**
 83      * Returns the primeExponentP.
 84      *
 85      * @return the primeExponentP.
 86      */
 87     public BigInteger getPrimeExponentP();
 88 
 89     /**
 90      * Returns the primeExponentQ.
 91      *
 92      * @return the primeExponentQ.
 93      */
 94     public BigInteger getPrimeExponentQ();
 95 
 96     /**
 97      * Returns the crtCoefficient.
 98      *
 99      * @return the crtCoefficient.
100      */
101     public BigInteger getCrtCoefficient();
102 
103     /**
104      * Returns the otherPrimeInfo or null if there are only
105      * two prime factors (p and q).
106      *
107      * @return the otherPrimeInfo.
108      */
109     public RSAOtherPrimeInfo[] getOtherPrimeInfo();
110 }
