 26 package java.security.cert;
 27 
 28 import java.security.GeneralSecurityException;
 29 
 30 /**
 31  * An exception indicating one of a variety of problems encountered when
 32  * building a certification path with a {@code CertPathBuilder}.
 33  * <p>
 34  * A {@code CertPathBuilderException} provides support for wrapping
 35  * exceptions. The {@link #getCause getCause} method returns the throwable,
 36  * if any, that caused this exception to be thrown.
 37  * <p>
 38  * <b>Concurrent Access</b>
 39  * <p>
 40  * Unless otherwise specified, the methods defined in this class are not
 41  * thread-safe. Multiple threads that need to access a single
 42  * object concurrently should synchronize amongst themselves and
 43  * provide the necessary locking. Multiple threads each manipulating
 44  * separate objects need not synchronize.
 45  *
 46  * @see CertPathBuilder
 47  *
 48  * @since       1.4
 49  * @author      Sean Mullan
 50  */
 51 public class CertPathBuilderException extends GeneralSecurityException {
 52 
 53     @java.io.Serial
 54     private static final long serialVersionUID = 5316471420178794402L;
 55 
 56     /**
 57      * Creates a {@code CertPathBuilderException} with {@code null}
 58      * as its detail message.
 59      */
 60     public CertPathBuilderException() {
 61         super();
 62     }
 63 
 64     /**
 65      * Creates a {@code CertPathBuilderException} with the given
 66      * detail message. The detail message is a {@code String} that
 67      * describes this particular exception in more detail.
 68      *
 69      * @param msg the detail message
 70      */
 71     public CertPathBuilderException(String msg) {
 72         super(msg);
 73     }
 74 
 75     /**
 76      * Creates a {@code CertPathBuilderException} that wraps the specified
 77      * throwable. This allows any exception to be converted into a
 78      * {@code CertPathBuilderException}, while retaining information
 79      * about the wrapped exception, which may be useful for debugging. The
 80      * detail message is set to ({@code cause==null ? null : cause.toString()})
 81      * (which typically contains the class and detail message of
 82      * cause).
 83      *
 84      * @param cause the cause (which is saved for later retrieval by the
 85      * {@link #getCause getCause()} method). (A {@code null} value is
 86      * permitted, and indicates that the cause is nonexistent or unknown.)
 87      */
 88     public CertPathBuilderException(Throwable cause) {
 89         super(cause);
 90     }
 91 
 92     /**
 93      * Creates a {@code CertPathBuilderException} with the specified
 94      * detail message and cause.
 95      *
 96      * @param msg the detail message
 97      * @param  cause the cause (which is saved for later retrieval by the
 98      * {@link #getCause getCause()} method). (A {@code null} value is
 99      * permitted, and indicates that the cause is nonexistent or unknown.)
100      */
101     public CertPathBuilderException(String msg, Throwable cause) {
102         super(msg, cause);
103     }
104 
105 }
