26 package java.lang;
27 
28 /**
29  * Thrown when an incompatible class change has occurred to some class
30  * definition. The definition of some class, on which the currently
31  * executing method depends, has since changed.
32  *
33  * @author  unascribed
34  * @since   1.0
35  */
36 public class IncompatibleClassChangeError extends LinkageError {
37     @java.io.Serial
38     private static final long serialVersionUID = -4914975503642802119L;
39 
40     /**
41      * Constructs an {@code IncompatibleClassChangeError} with no
42      * detail message.
43      */
44     public IncompatibleClassChangeError () {
45         super();
46     }
47 
48     /**
49      * Constructs an {@code IncompatibleClassChangeError} with the
50      * specified detail message.
51      *
52      * @param   s   the detail message.
53      */
54     public IncompatibleClassChangeError(String s) {
55         super(s);
56     }
57 }
