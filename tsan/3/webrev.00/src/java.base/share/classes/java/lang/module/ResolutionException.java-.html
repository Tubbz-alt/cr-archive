26 package java.lang.module;
27 
28 /**
29  * Thrown when resolving a set of modules, or resolving a set of modules with
30  * service binding, fails.
31  *
32  * @see Configuration
33  * @since 9
34  * @spec JPMS
35  */
36 public class ResolutionException extends RuntimeException {
37     private static final long serialVersionUID = -1031186845316729450L;
38 
39     /**
40      * Constructs a {@code ResolutionException} with no detail message.
41      */
42     public ResolutionException() { }
43 
44     /**
45      * Constructs a {@code ResolutionException} with the given detail
46      * message.
47      *
48      * @param msg
49      *        The detail message; can be {@code null}
50      */
51     public ResolutionException(String msg) {
52         super(msg);
53     }
54 
55     /**
56      * Constructs an instance of this exception with the given cause.
57      *
58      * @param cause
59      *        The cause; can be {@code null}
60      */
61     public ResolutionException(Throwable cause) {
62         super(cause);
63     }
64 
65     /**
66      * Constructs a {@code ResolutionException} with the given detail message
67      * and cause.
68      *
69      * @param msg
70      *        The detail message; can be {@code null}
71      * @param cause
72      *        The cause; can be {@code null}
73      */
74     public ResolutionException(String msg, Throwable cause) {
75         super(msg, cause);
76     }
77 
78 }
