26 package java.lang;
27 
28 /**
29  * Thrown when the "verifier" detects that a class file,
30  * though well formed, contains some sort of internal inconsistency
31  * or security problem.
32  *
33  * @author  unascribed
34  * @since   1.0
35  */
36 public class VerifyError extends LinkageError {
37     @java.io.Serial
38     private static final long serialVersionUID = 7001962396098498785L;
39 
40     /**
41      * Constructs an {@code VerifyError} with no detail message.
42      */
43     public VerifyError() {
44         super();
45     }
46 
47     /**
48      * Constructs an {@code VerifyError} with the specified detail message.
49      *
50      * @param   s   the detail message.
51      */
52     public VerifyError(String s) {
53         super(s);
54     }
55 }
