 25 package java.net;
 26 
 27 import jdk.internal.access.JavaIOFileDescriptorAccess;
 28 import jdk.internal.access.SharedSecrets;
 29 import jdk.internal.ref.CleanerFactory;
 30 import jdk.internal.ref.PhantomCleanable;
 31 
 32 import java.io.FileDescriptor;
 33 import java.io.IOException;
 34 import java.io.UncheckedIOException;
 35 import java.lang.ref.Cleaner;
 36 
 37 import sun.net.ResourceManager;
 38 
 39 /**
 40  * Cleanable for a socket/datagramsocket FileDescriptor when it becomes phantom reachable.
 41  * Create a cleanup if the raw fd != -1. Windows closes sockets using the fd.
 42  * Subclassed from {@code PhantomCleanable} so that {@code clear} can be
 43  * called to disable the cleanup when the socket fd is closed by any means
 44  * other than calling {@link FileDescriptor#close}.
 45  * Otherwise, it might incorrectly close the handle or fd after it has been reused.
 46  */
 47 final class SocketCleanable extends PhantomCleanable<FileDescriptor> {
 48 
 49     // Access to FileDescriptor private fields
 50     private static final JavaIOFileDescriptorAccess fdAccess =
 51             SharedSecrets.getJavaIOFileDescriptorAccess();
 52 
 53     // Native function to call NET_SocketClose(fd)
 54     // Used only for last chance cleanup.
 55     private static native void cleanupClose0(int fd) throws IOException;
 56 
 57     // The raw fd to close
 58     private final int fd;
 59 
 60     // true for socket, false for datagram socket
 61     private final boolean stream;
 62 
 63     /**
 64      * Register a socket specific Cleanable with the FileDescriptor
 65      * if the FileDescriptor is non-null and the raw fd is != -1.
 66      *
 67      * @param fdo     the FileDescriptor; may be null
 68      * @param stream  false for datagram socket
 69      */
 70     static void register(FileDescriptor fdo, boolean stream) {
 71         if (fdo != null && fdo.valid()) {
 72             int fd = fdAccess.get(fdo);
 73             fdAccess.registerCleanup(fdo,
 74                     new SocketCleanable(fdo, CleanerFactory.cleaner(),
 75                                         fd, stream));
 76         }
 77     }
 78 
 79     /**
 80      * Unregister a Cleanable from the FileDescriptor.
 81      * @param fdo the FileDescriptor; may be null
 82      */
 83     static void unregister(FileDescriptor fdo) {
 84         if (fdo != null) {
 85             fdAccess.unregisterCleanup(fdo);
 86         }
 87     }
 88 
 89     /**
 90      * Constructor for a phantom cleanable reference.
 91      *
 92      * @param obj     the object to monitor
 93      * @param cleaner the cleaner
 94      * @param fd      file descriptor to close
 95      * @param stream  false for datagram socket
 96      */
 97     private SocketCleanable(FileDescriptor obj, Cleaner cleaner,
 98                             int fd, boolean stream) {
 99         super(obj, cleaner);
100         this.fd = fd;
101         this.stream = stream;
102     }
103 
104     /**
105      * Close the native handle or fd.
106      */
107     @Override
108     protected void performCleanup() {
109         try {
110             cleanupClose0(fd);
111         } catch (IOException ioe) {
112             throw new UncheckedIOException("close", ioe);
113         } finally {
114             if (!stream) {
115                 ResourceManager.afterUdpClose();
116             }
117         }
118     }
119 }
