26 package java.lang.invoke;
27 
28 /**
29  * LambdaConversionException
30  *
31  * @since 1.8
32  */
33 public class LambdaConversionException extends Exception {
34     @java.io.Serial
35     private static final long serialVersionUID = 292L + 8L;
36 
37     /**
38      * Constructs a {@code LambdaConversionException}.
39      */
40     public LambdaConversionException() {
41     }
42 
43     /**
44      * Constructs a {@code LambdaConversionException} with a message.
45      * @param message the detail message
46      */
47     public LambdaConversionException(String message) {
48         super(message);
49     }
50 
51     /**
52      * Constructs a {@code LambdaConversionException} with a message and cause.
53      * @param message the detail message
54      * @param cause the cause
55      */
56     public LambdaConversionException(String message, Throwable cause) {
57         super(message, cause);
58     }
59 
60     /**
61      * Constructs a {@code LambdaConversionException} with a cause.
62      * @param cause the cause
63      */
64     public LambdaConversionException(Throwable cause) {
65         super(cause);
66     }
67 
68     /**
69      * Constructs a {@code LambdaConversionException} with a message,
70      * cause, and other settings.
71      * @param message the detail message
72      * @param cause the cause
73      * @param enableSuppression whether or not suppressed exceptions are enabled
74      * @param writableStackTrace whether or not the stack trace is writable
75      */
76     public LambdaConversionException(String message, Throwable cause, boolean enableSuppression, boolean writableStackTrace) {
77         super(message, cause, enableSuppression, writableStackTrace);
78     }
79 }
