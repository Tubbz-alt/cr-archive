26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when the format width is a negative value other
30  * than {@code -1} or is otherwise unsupported.
31  *
32  * @since 1.5
33  */
34 public class IllegalFormatWidthException extends IllegalFormatException {
35 
36     @java.io.Serial
37     private static final long serialVersionUID = 16660902L;
38 
39     private int w;
40 
41     /**
42      * Constructs an instance of this class with the specified width.
43      *
44      * @param  w
45      *         The width
46      */
47     public IllegalFormatWidthException(int w) {
48         this.w = w;
49     }
50 
51     /**
52      * Returns the width
53      *
54      * @return  The width
55      */
56     public int getWidth() {
57         return w;
58     }
59 
60     public String getMessage() {
61         return Integer.toString(w);
62     }
63 }
