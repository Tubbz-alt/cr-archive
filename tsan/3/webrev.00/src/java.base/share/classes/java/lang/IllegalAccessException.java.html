26 package java.lang;
27 
28 /**
29  * An IllegalAccessException is thrown when an application tries
30  * to reflectively create an instance (other than an array),
31  * set or get a field, or invoke a method, but the currently
32  * executing method does not have access to the definition of
33  * the specified class, field, method or constructor.
34  *
35  * @author  unascribed
36  * @see     Class#newInstance()
37  * @see     java.lang.reflect.Field#set(Object, Object)
38  * @see     java.lang.reflect.Field#setBoolean(Object, boolean)
39  * @see     java.lang.reflect.Field#setByte(Object, byte)
40  * @see     java.lang.reflect.Field#setShort(Object, short)
41  * @see     java.lang.reflect.Field#setChar(Object, char)
42  * @see     java.lang.reflect.Field#setInt(Object, int)
43  * @see     java.lang.reflect.Field#setLong(Object, long)
44  * @see     java.lang.reflect.Field#setFloat(Object, float)
45  * @see     java.lang.reflect.Field#setDouble(Object, double)
46  * @see     java.lang.reflect.Field#get(Object)
47  * @see     java.lang.reflect.Field#getBoolean(Object)
48  * @see     java.lang.reflect.Field#getByte(Object)
49  * @see     java.lang.reflect.Field#getShort(Object)
50  * @see     java.lang.reflect.Field#getChar(Object)
51  * @see     java.lang.reflect.Field#getInt(Object)
52  * @see     java.lang.reflect.Field#getLong(Object)
53  * @see     java.lang.reflect.Field#getFloat(Object)
54  * @see     java.lang.reflect.Field#getDouble(Object)
55  * @see     java.lang.reflect.Method#invoke(Object, Object[])
56  * @see     java.lang.reflect.Constructor#newInstance(Object[])
57  * @since   1.0
58  */
59 public class IllegalAccessException extends ReflectiveOperationException {
60     private static final long serialVersionUID = 6616958222490762034L;
61 
62     /**
63      * Constructs an {@code IllegalAccessException} without a
64      * detail message.
65      */
66     public IllegalAccessException() {
67         super();
68     }
69 
70     /**
71      * Constructs an {@code IllegalAccessException} with a detail message.
72      *
73      * @param   s   the detail message.
74      */
75     public IllegalAccessException(String s) {
76         super(s);
77     }
78 }
