 26 package java.util.jar;
 27 
 28 import java.util.zip.*;
 29 import java.io.*;
 30 import sun.security.util.ManifestEntryVerifier;
 31 import jdk.internal.util.jar.JarIndex;
 32 
 33 /**
 34  * The {@code JarInputStream} class is used to read the contents of
 35  * a JAR file from any input stream. It extends the class
 36  * {@code java.util.zip.ZipInputStream} with support for reading
 37  * an optional {@code Manifest} entry. The {@code Manifest}
 38  * can be used to store meta-information about the JAR file and its entries.
 39  *
 40  * @author  David Connelly
 41  * @see     Manifest
 42  * @see     java.util.zip.ZipInputStream
 43  * @since   1.2
 44  */
 45 public class JarInputStream extends ZipInputStream {
 46     private Manifest man;
 47     private JarEntry first;
 48     private JarVerifier jv;
 49     private ManifestEntryVerifier mev;
 50     private final boolean doVerify;
 51     private boolean tryManifest;
 52 
 53     /**
 54      * Creates a new {@code JarInputStream} and reads the optional
 55      * manifest. If a manifest is present, also attempts to verify
 56      * the signatures if the JarInputStream is signed.
 57      * @param in the actual input stream
 58      * @throws    IOException if an I/O error has occurred
 59      */
 60     public JarInputStream(InputStream in) throws IOException {
 61         this(in, true);
 62     }
 63 
 64     /**
 65      * Creates a new {@code JarInputStream} and reads the optional
 66      * manifest. If a manifest is present and verify is true, also attempts
 67      * to verify the signatures if the JarInputStream is signed.
 68      *
 69      * @param in the actual input stream
 70      * @param verify whether or not to verify the JarInputStream if
 71      * it is signed.
 72      * @throws    IOException if an I/O error has occurred
 73      */
 74     public JarInputStream(InputStream in, boolean verify) throws IOException {
 75         super(in);
 76         this.doVerify = verify;
 77 
 78         // This implementation assumes the META-INF/MANIFEST.MF entry
 79         // should be either the first or the second entry (when preceded
 80         // by the dir META-INF/). It skips the META-INF/ and then
 81         // "consumes" the MANIFEST.MF to initialize the Manifest object.
 82         JarEntry e = (JarEntry)super.getNextEntry();
 83         if (e != null && e.getName().equalsIgnoreCase("META-INF/"))
 84             e = (JarEntry)super.getNextEntry();
 85         first = checkManifest(e);
 86     }
 87 
 88     private JarEntry checkManifest(JarEntry e)
 89         throws IOException
 90     {
 91         if (e != null && JarFile.MANIFEST_NAME.equalsIgnoreCase(e.getName())) {
 92             man = new Manifest();
 93             byte bytes[] = getBytes(new BufferedInputStream(this));
 94             man.read(new ByteArrayInputStream(bytes));
 95             closeEntry();
 96             if (doVerify) {
 97                 jv = new JarVerifier(bytes);
 98                 mev = new ManifestEntryVerifier(man);
 99             }
100             return (JarEntry)super.getNextEntry();
101         }
102         return e;
103     }
104 
105     private byte[] getBytes(InputStream is)
106         throws IOException
107     {
108         byte[] buffer = new byte[8192];
109         ByteArrayOutputStream baos = new ByteArrayOutputStream(2048);
110         int n;
111         while ((n = is.read(buffer, 0, buffer.length)) != -1) {
112             baos.write(buffer, 0, n);
113         }
114         return baos.toByteArray();
115     }
116 
117     /**
118      * Returns the {@code Manifest} for this JAR file, or
119      * {@code null} if none.
120      *
121      * @return the {@code Manifest} for this JAR file, or
122      *         {@code null} if none.
123      */
124     public Manifest getManifest() {
125         return man;
126     }
127 
128     /**
129      * Reads the next ZIP file entry and positions the stream at the
130      * beginning of the entry data. If verification has been enabled,
131      * any invalid signature detected while positioning the stream for
132      * the next entry will result in an exception.
133      * @throws    ZipException if a ZIP file error has occurred
134      * @throws    IOException if an I/O error has occurred
135      * @throws    SecurityException if any of the jar file entries
136      *         are incorrectly signed.
137      */
138     public ZipEntry getNextEntry() throws IOException {
139         JarEntry e;
140         if (first == null) {
141             e = (JarEntry)super.getNextEntry();
142             if (tryManifest) {
143                 e = checkManifest(e);
144                 tryManifest = false;
145             }
146         } else {
147             e = first;
148             if (first.getName().equalsIgnoreCase(JarIndex.INDEX_NAME))
149                 tryManifest = true;
150             first = null;
151         }
152         if (jv != null && e != null) {
153             // At this point, we might have parsed all the meta-inf
154             // entries and have nothing to verify. If we have
155             // nothing to verify, get rid of the JarVerifier object.
156             if (jv.nothingToVerify() == true) {
157                 jv = null;
158                 mev = null;
159             } else {
160                 jv.beginEntry(e, mev);
161             }
162         }
163         return e;
164     }
165 
166     /**
167      * Reads the next JAR file entry and positions the stream at the
168      * beginning of the entry data. If verification has been enabled,
169      * any invalid signature detected while positioning the stream for
170      * the next entry will result in an exception.
171      * @return the next JAR file entry, or null if there are no more entries
172      * @throws    ZipException if a ZIP file error has occurred
173      * @throws    IOException if an I/O error has occurred
174      * @throws    SecurityException if any of the jar file entries
175      *         are incorrectly signed.
176      */
177     public JarEntry getNextJarEntry() throws IOException {
178         return (JarEntry)getNextEntry();
179     }
180 
181     /**
182      * Reads from the current JAR file entry into an array of bytes.
183      * If {@code len} is not zero, the method
184      * blocks until some input is available; otherwise, no
185      * bytes are read and {@code 0} is returned.
186      * If verification has been enabled, any invalid signature
187      * on the current entry will be reported at some point before the
188      * end of the entry is reached.
189      * @param b the buffer into which the data is read
190      * @param off the start offset in the destination array {@code b}
191      * @param len the maximum number of bytes to read
192      * @return the actual number of bytes read, or -1 if the end of the
193      *         entry is reached
194      * @throws     NullPointerException If {@code b} is {@code null}.
195      * @throws     IndexOutOfBoundsException If {@code off} is negative,
196      * {@code len} is negative, or {@code len} is greater than
197      * {@code b.length - off}
198      * @throws    ZipException if a ZIP file error has occurred
199      * @throws    IOException if an I/O error has occurred
200      * @throws    SecurityException if any of the jar file entries
201      *         are incorrectly signed.
202      */
203     public int read(byte[] b, int off, int len) throws IOException {
204         int n;
205         if (first == null) {
206             n = super.read(b, off, len);
207         } else {
208             n = -1;
209         }
210         if (jv != null) {
211             jv.update(n, b, off, len, mev);
212         }
213         return n;
214     }
215 
216     /**
217      * Creates a new {@code JarEntry} ({@code ZipEntry}) for the
218      * specified JAR file entry name. The manifest attributes of
219      * the specified JAR file entry name will be copied to the new
220      * <CODE>JarEntry</CODE>.
221      *
222      * @param name the name of the JAR/ZIP file entry
223      * @return the {@code JarEntry} object just created
224      */
225     protected ZipEntry createZipEntry(String name) {
226         JarEntry e = new JarEntry(name);
227         if (man != null) {
228             e.attr = man.getAttributes(name);
229         }
230         return e;
231     }
