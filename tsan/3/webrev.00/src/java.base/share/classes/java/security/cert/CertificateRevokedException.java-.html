 26 package java.security.cert;
 27 
 28 import java.io.ObjectInputStream;
 29 import java.io.ObjectOutputStream;
 30 import java.io.IOException;
 31 import java.util.Collections;
 32 import java.util.Date;
 33 import java.util.HashMap;
 34 import java.util.Map;
 35 import javax.security.auth.x500.X500Principal;
 36 
 37 import sun.security.util.IOUtils;
 38 import sun.security.util.ObjectIdentifier;
 39 import sun.security.x509.InvalidityDateExtension;
 40 
 41 /**
 42  * An exception that indicates an X.509 certificate is revoked. A
 43  * {@code CertificateRevokedException} contains additional information
 44  * about the revoked certificate, such as the date on which the
 45  * certificate was revoked and the reason it was revoked.
 46  *
 47  * @author Sean Mullan
 48  * @since 1.7
 49  * @see CertPathValidatorException
 50  */
 51 public class CertificateRevokedException extends CertificateException {
 52 
 53     private static final long serialVersionUID = 7839996631571608627L;
 54 
 55     /**
 56      * @serial the date on which the certificate was revoked
 57      */
 58     private Date revocationDate;
 59     /**
 60      * @serial the revocation reason
 61      */
 62     private final CRLReason reason;
 63     /**
 64      * @serial the {@code X500Principal} that represents the name of the
 65      * authority that signed the certificate's revocation status information
 66      */
 67     private final X500Principal authority;
 68 
 69     private transient Map<String, Extension> extensions;
 70 
 71     /**
 72      * Constructs a {@code CertificateRevokedException} with
 73      * the specified revocation date, reason code, authority name, and map
 74      * of extensions.
 75      *
 76      * @param revocationDate the date on which the certificate was revoked. The
 77      *    date is copied to protect against subsequent modification.
 78      * @param reason the revocation reason
 79      * @param extensions a map of X.509 Extensions. Each key is an OID String
 80      *    that maps to the corresponding Extension. The map is copied to
 81      *    prevent subsequent modification.
 82      * @param authority the {@code X500Principal} that represents the name
 83      *    of the authority that signed the certificate's revocation status
 84      *    information
 85      * @throws NullPointerException if {@code revocationDate},
 86      *    {@code reason}, {@code authority}, or
 87      *    {@code extensions} is {@code null}
 88      * @throws ClassCastException if {@code extensions} contains an incorrectly
 89      *    typed key or value
 90      */
 91     public CertificateRevokedException(Date revocationDate, CRLReason reason,
 92         X500Principal authority, Map<String, Extension> extensions) {
 93         if (revocationDate == null || reason == null || authority == null ||
 94             extensions == null) {
 95             throw new NullPointerException();
 96         }
 97         this.revocationDate = new Date(revocationDate.getTime());
 98         this.reason = reason;
 99         this.authority = authority;
100         // make sure Map only contains correct types
101         this.extensions = Collections.checkedMap(new HashMap<>(),
102                                                  String.class, Extension.class);
103         this.extensions.putAll(extensions);
104     }
105 
106     /**
107      * Returns the date on which the certificate was revoked. A new copy is
108      * returned each time the method is invoked to protect against subsequent
109      * modification.
110      *
111      * @return the revocation date
112      */
113     public Date getRevocationDate() {
114         return (Date) revocationDate.clone();
115     }
116 
117     /**
118      * Returns the reason the certificate was revoked.
119      *
120      * @return the revocation reason
121      */
122     public CRLReason getRevocationReason() {
123         return reason;
124     }
125 
126     /**
127      * Returns the name of the authority that signed the certificate's
128      * revocation status information.
129      *
130      * @return the {@code X500Principal} that represents the name of the
131      *     authority that signed the certificate's revocation status information
132      */
133     public X500Principal getAuthorityName() {
134         return authority;
135     }
136 
137     /**
138      * Returns the invalidity date, as specified in the Invalidity Date
139      * extension of this {@code CertificateRevokedException}. The
140      * invalidity date is the date on which it is known or suspected that the
141      * private key was compromised or that the certificate otherwise became
142      * invalid. This implementation calls {@code getExtensions()} and
143      * checks the returned map for an entry for the Invalidity Date extension
144      * OID ("2.5.29.24"). If found, it returns the invalidity date in the
145      * extension; otherwise null. A new Date object is returned each time the
146      * method is invoked to protect against subsequent modification.
147      *
148      * @return the invalidity date, or {@code null} if not specified
149      */
150     public Date getInvalidityDate() {
151         Extension ext = getExtensions().get("2.5.29.24");
152         if (ext == null) {
153             return null;
154         } else {
155             try {
156                 Date invalidity = InvalidityDateExtension.toImpl(ext).get("DATE");
157                 return new Date(invalidity.getTime());
158             } catch (IOException ioe) {
159                 return null;
160             }
161         }
162     }
163 
164     /**
165      * Returns a map of X.509 extensions containing additional information
166      * about the revoked certificate, such as the Invalidity Date
167      * Extension. Each key is an OID String that maps to the corresponding
168      * Extension.
169      *
170      * @return an unmodifiable map of X.509 extensions, or an empty map
171      *    if there are no extensions
172      */
173     public Map<String, Extension> getExtensions() {
174         return Collections.unmodifiableMap(extensions);
175     }
176 
177     @Override
178     public String getMessage() {
179         return "Certificate has been revoked, reason: "
180                + reason + ", revocation date: " + revocationDate
181                + ", authority: " + authority + ", extension OIDs: "
182                + extensions.keySet();
183     }
184 
185     /**
186      * Serialize this {@code CertificateRevokedException} instance.
187      *
188      * @serialData the size of the extensions map (int), followed by all of
189      * the extensions in the map, in no particular order. For each extension,
190      * the following data is emitted: the OID String (Object), the criticality
191      * flag (boolean), the length of the encoded extension value byte array
192      * (int), and the encoded extension value bytes.
193      */
194     private void writeObject(ObjectOutputStream oos) throws IOException {
195         // Write out the non-transient fields
196         // (revocationDate, reason, authority)
197         oos.defaultWriteObject();
198 
199         // Write out the size (number of mappings) of the extensions map
200         oos.writeInt(extensions.size());
201 
202         // For each extension in the map, the following are emitted (in order):
203         // the OID String (Object), the criticality flag (boolean), the length
204         // of the encoded extension value byte array (int), and the encoded
205         // extension value byte array. The extensions themselves are emitted
206         // in no particular order.
207         for (Map.Entry<String, Extension> entry : extensions.entrySet()) {
208             Extension ext = entry.getValue();
209             oos.writeObject(ext.getId());
210             oos.writeBoolean(ext.isCritical());
211             byte[] extVal = ext.getValue();
212             oos.writeInt(extVal.length);
213             oos.write(extVal);
214         }
215     }
216 
217     /**
218      * Deserialize the {@code CertificateRevokedException} instance.
219      */
220     private void readObject(ObjectInputStream ois)
221         throws IOException, ClassNotFoundException {
222         // Read in the non-transient fields
223         // (revocationDate, reason, authority)
224         ois.defaultReadObject();
225 
226         // Defensively copy the revocation date
227         revocationDate = new Date(revocationDate.getTime());
228 
229         // Read in the size (number of mappings) of the extensions map
230         // and create the extensions map
231         int size = ois.readInt();
232         if (size == 0) {
233             extensions = Collections.emptyMap();
234         } else if (size < 0) {
235             throw new IOException("size cannot be negative");
236         } else {
237             extensions = new HashMap<>(size > 20 ? 20 : size);
238         }
239 
240         // Read in the extensions and put the mappings in the extensions map
241         for (int i = 0; i < size; i++) {
242             String oid = (String) ois.readObject();
243             boolean critical = ois.readBoolean();
244             byte[] extVal = IOUtils.readNBytes(ois, ois.readInt());
245             Extension ext = sun.security.x509.Extension.newExtension
246                 (new ObjectIdentifier(oid), critical, extVal);
247             extensions.put(oid, ext);
248         }
249     }
