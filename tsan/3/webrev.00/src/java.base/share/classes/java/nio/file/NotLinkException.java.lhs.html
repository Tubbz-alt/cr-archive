26 package java.nio.file;
27 
28 /**
29  * Checked exception thrown when a file system operation fails because a file
30  * is not a symbolic link.
31  *
32  * @since 1.7
33  */
34 
35 public class NotLinkException
36     extends FileSystemException
37 {

38     static final long serialVersionUID = -388655596416518021L;
39 
40     /**
41      * Constructs an instance of this class.
42      *
43      * @param   file
44      *          a string identifying the file or {@code null} if not known
45      */
46     public NotLinkException(String file) {
47         super(file);
48     }
49 
50     /**
51      * Constructs an instance of this class.
52      *
53      * @param   file
54      *          a string identifying the file or {@code null} if not known
55      * @param   other
56      *          a string identifying the other file or {@code null} if not known
57      * @param   reason
58      *          a reason message with additional information or {@code null}
59      */
60     public NotLinkException(String file, String other, String reason) {
61         super(file, other, reason);
62     }
63 }
