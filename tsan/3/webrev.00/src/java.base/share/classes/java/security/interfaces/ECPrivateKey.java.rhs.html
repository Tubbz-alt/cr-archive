33  * @author Valerie Peng

35  *

41 public interface ECPrivateKey extends PrivateKey, ECKey {

43     * The class fingerprint that is set to indicate
44     * serialization compatibility.

51     @java.io.Serial
52     static final long serialVersionUID = -7896394956925609184L;
53 
54     /**
55      * Returns the private value S.
56      * @return the private value S.
57      */
58     BigInteger getS();
59 }
