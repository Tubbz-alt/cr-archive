26 package java.util;
27 
28 /**
29  * Unchecked exception thrown when the format width is required.
30  *
31  * <p> Unless otherwise specified, passing a {@code null} argument to any
32  * method or constructor in this class will cause a {@link
33  * NullPointerException} to be thrown.
34  *
35  * @since 1.5
36  */
37 public class MissingFormatWidthException extends IllegalFormatException {
38 
39     @java.io.Serial
40     private static final long serialVersionUID = 15560123L;
41 
42     private String s;
43 
44     /**
45      * Constructs an instance of this class with the specified format
46      * specifier.
47      *
48      * @param  s
49      *         The format specifier which does not have a width
50      */
51     public MissingFormatWidthException(String s) {
52         if (s == null)
53             throw new NullPointerException();
54         this.s = s;
55     }
56 
57     /**
58      * Returns the format specifier which does not have a width.
59      *
60      * @return  The format specifier which does not have a width
61      */
62     public String getFormatSpecifier() {
63         return s;
64     }
65 
66     public String getMessage() {
67         return s;
68     }
69 }
