26 package java.net;
27 
28 /**
29  * Signals that an ICMP Port Unreachable message has been
30  * received on a connected datagram.
31  *
32  * @since   1.4
33  */
34 
35 public class PortUnreachableException extends SocketException {
36     private static final long serialVersionUID = 8462541992376507323L;
37 
38     /**
39      * Constructs a new {@code PortUnreachableException} with a
40      * detail message.
41      * @param msg the detail message
42      */
43     public PortUnreachableException(String msg) {
44         super(msg);
45     }
46 
47     /**
48      * Construct a new {@code PortUnreachableException} with no
49      * detailed message.
50      */
51     public PortUnreachableException() {}
52 }
