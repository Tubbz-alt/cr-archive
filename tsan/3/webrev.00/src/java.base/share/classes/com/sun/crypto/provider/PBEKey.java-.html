 26 package com.sun.crypto.provider;
 27 
 28 import java.lang.ref.Reference;
 29 import java.security.MessageDigest;
 30 import java.security.KeyRep;
 31 import java.security.spec.InvalidKeySpecException;
 32 import java.util.Arrays;
 33 import java.util.Locale;
 34 import javax.crypto.SecretKey;
 35 import javax.crypto.spec.PBEKeySpec;
 36 
 37 import jdk.internal.ref.CleanerFactory;
 38 
 39 /**
 40  * This class represents a PBE key.
 41  *
 42  * @author Jan Luehe
 43  *
 44  */
 45 final class PBEKey implements SecretKey {
 46 
 47     static final long serialVersionUID = -2234768909660948176L;
 48 
 49     private byte[] key;
 50 
 51     private String type;
 52 
 53     /**
 54      * Creates a PBE key from a given PBE key specification.
 55      *
 56      * @param keytype the given PBE key specification
 57      */
 58     PBEKey(PBEKeySpec keySpec, String keytype, boolean useCleaner)
 59             throws InvalidKeySpecException {
 60         char[] passwd = keySpec.getPassword();
 61         if (passwd == null) {
 62             // Should allow an empty password.
 63             passwd = new char[0];
 64         }
 65         // Accept "\0" to signify "zero-length password with no terminator".
 66         if (!(passwd.length == 1 && passwd[0] == 0)) {
 67             for (int i=0; i<passwd.length; i++) {
 68                 if ((passwd[i] < '\u0020') || (passwd[i] > '\u007E')) {
 69                     throw new InvalidKeySpecException("Password is not ASCII");
 70                 }
 71             }
 72         }
 73         this.key = new byte[passwd.length];
 74         for (int i=0; i<passwd.length; i++)
 75             this.key[i] = (byte) (passwd[i] & 0x7f);
 76         Arrays.fill(passwd, '\0');
 77         type = keytype;
 78 
 79         // Use the cleaner to zero the key when no longer referenced
 80         if (useCleaner) {
 81             final byte[] k = this.key;
 82             CleanerFactory.cleaner().register(this,
 83                 () -> Arrays.fill(k, (byte) 0x00));
 84         }
 85     }
 86 
 87     public byte[] getEncoded() {
 88         // The key is zeroized by finalize()
 89         // The reachability fence ensures finalize() isn't called early
 90         byte[] result = key.clone();
 91         Reference.reachabilityFence(this);
 92         return result;
 93     }
 94 
 95     public String getAlgorithm() {
 96         return type;
 97     }
 98 
 99     public String getFormat() {
100         return "RAW";
101     }
102 
103     /**
104      * Calculates a hash code value for the object.
105      * Objects that are equal will also have the same hashcode.
106      */
107     public int hashCode() {
108         int retval = 0;
109         for (int i = 1; i < this.key.length; i++) {
110             retval += this.key[i] * i;
111         }
112         return(retval ^= getAlgorithm().toLowerCase(Locale.ENGLISH).hashCode());
113     }
114 
115     public boolean equals(Object obj) {
116         if (obj == this)
117             return true;
118 
119         if (!(obj instanceof SecretKey))
120             return false;
121 
122         SecretKey that = (SecretKey)obj;
123 
124         if (!(that.getAlgorithm().equalsIgnoreCase(type)))
125             return false;
126 
127         byte[] thatEncoded = that.getEncoded();
128         boolean ret = MessageDigest.isEqual(this.key, thatEncoded);
129         Arrays.fill(thatEncoded, (byte)0x00);
130         return ret;
131     }
132 
133     /**
134      * Clears the internal copy of the key.
135      *
136      */
137     @Override
138     public void destroy() {
139         if (key != null) {
140             Arrays.fill(key, (byte) 0x00);
141             key = null;
142         }
143     }
144 
145     /**
146      * readObject is called to restore the state of this key from
147      * a stream.
148      */
149     private void readObject(java.io.ObjectInputStream s)
150          throws java.io.IOException, ClassNotFoundException
151     {
152         s.defaultReadObject();
153         key = key.clone();
154     }
155 
156 
157     /**
158      * Replace the PBE key to be serialized.
159      *
160      * @return the standard KeyRep object to be serialized
161      *
162      * @throws java.io.ObjectStreamException if a new object representing
163      * this PBE key could not be created
164      */
165     private Object writeReplace() throws java.io.ObjectStreamException {
166         return new KeyRep(KeyRep.Type.SECRET,
167                         getAlgorithm(),
168                         getFormat(),
169                         getEncoded());
170     }
171 }
