 26 package com.sun.crypto.provider;
 27 
 28 import java.lang.ref.Reference;
 29 import java.security.MessageDigest;
 30 import java.security.KeyRep;
 31 import java.security.spec.InvalidKeySpecException;
 32 import java.util.Arrays;
 33 import java.util.Locale;
 34 import javax.crypto.SecretKey;
 35 import javax.crypto.spec.PBEKeySpec;
 36 
 37 import jdk.internal.ref.CleanerFactory;
 38 
 39 /**
 40  * This class represents a PBE key.
 41  *
 42  * @author Jan Luehe
 43  *
 44  */
 45 final class PBEKey implements SecretKey {
 46 
 47     @java.io.Serial
 48     static final long serialVersionUID = -2234768909660948176L;
 49 
 50     private byte[] key;
 51 
 52     private String type;
 53 
 54     /**
 55      * Creates a PBE key from a given PBE key specification.
 56      *
 57      * @param keytype the given PBE key specification
 58      */
 59     PBEKey(PBEKeySpec keySpec, String keytype, boolean useCleaner)
 60             throws InvalidKeySpecException {
 61         char[] passwd = keySpec.getPassword();
 62         if (passwd == null) {
 63             // Should allow an empty password.
 64             passwd = new char[0];
 65         }
 66         // Accept "\0" to signify "zero-length password with no terminator".
 67         if (!(passwd.length == 1 && passwd[0] == 0)) {
 68             for (int i=0; i<passwd.length; i++) {
 69                 if ((passwd[i] < '\u0020') || (passwd[i] > '\u007E')) {
 70                     throw new InvalidKeySpecException("Password is not ASCII");
 71                 }
 72             }
 73         }
 74         this.key = new byte[passwd.length];
 75         for (int i=0; i<passwd.length; i++)
 76             this.key[i] = (byte) (passwd[i] & 0x7f);
 77         Arrays.fill(passwd, '\0');
 78         type = keytype;
 79 
 80         // Use the cleaner to zero the key when no longer referenced
 81         if (useCleaner) {
 82             final byte[] k = this.key;
 83             CleanerFactory.cleaner().register(this,
 84                 () -> Arrays.fill(k, (byte) 0x00));
 85         }
 86     }
 87 
 88     public byte[] getEncoded() {
 89         // The key is zeroized by finalize()
 90         // The reachability fence ensures finalize() isn't called early
 91         byte[] result = key.clone();
 92         Reference.reachabilityFence(this);
 93         return result;
 94     }
 95 
 96     public String getAlgorithm() {
 97         return type;
 98     }
 99 
100     public String getFormat() {
101         return "RAW";
102     }
103 
104     /**
105      * Calculates a hash code value for the object.
106      * Objects that are equal will also have the same hashcode.
107      */
108     public int hashCode() {
109         int retval = 0;
110         for (int i = 1; i < this.key.length; i++) {
111             retval += this.key[i] * i;
112         }
113         return(retval ^= getAlgorithm().toLowerCase(Locale.ENGLISH).hashCode());
114     }
115 
116     public boolean equals(Object obj) {
117         if (obj == this)
118             return true;
119 
120         if (!(obj instanceof SecretKey))
121             return false;
122 
123         SecretKey that = (SecretKey)obj;
124 
125         if (!(that.getAlgorithm().equalsIgnoreCase(type)))
126             return false;
127 
128         byte[] thatEncoded = that.getEncoded();
129         boolean ret = MessageDigest.isEqual(this.key, thatEncoded);
130         Arrays.fill(thatEncoded, (byte)0x00);
131         return ret;
132     }
133 
134     /**
135      * Clears the internal copy of the key.
136      *
137      */
138     @Override
139     public void destroy() {
140         if (key != null) {
141             Arrays.fill(key, (byte) 0x00);
142             key = null;
143         }
144     }
145 
146     /**
147      * readObject is called to restore the state of this key from
148      * a stream.
149      */
150     @java.io.Serial
151     private void readObject(java.io.ObjectInputStream s)
152          throws java.io.IOException, ClassNotFoundException
153     {
154         s.defaultReadObject();
155         key = key.clone();
156     }
157 
158 
159     /**
160      * Replace the PBE key to be serialized.
161      *
162      * @return the standard KeyRep object to be serialized
163      *
164      * @throws java.io.ObjectStreamException if a new object representing
165      * this PBE key could not be created
166      */
167     @java.io.Serial
168     private Object writeReplace() throws java.io.ObjectStreamException {
169         return new KeyRep(KeyRep.Type.SECRET,
170                         getAlgorithm(),
171                         getFormat(),
172                         getEncoded());
173     }
174 }
