26 package com.sun.security.ntlm;
27 
28 import java.security.GeneralSecurityException;
29 
30 /**
31  * An NTLM-related Exception
32  */
33 public final class NTLMException extends GeneralSecurityException {
34     @java.io.Serial
35     private static final long serialVersionUID = -3298539507906689430L;
36 
37     /**
38      * If the incoming packet is invalid.
39      */
40     public static final int PACKET_READ_ERROR = 1;
41 
42     /**
43      * If the client cannot get a domain value from the server and the
44      * caller has not provided one.
45      */
46     public static final int NO_DOMAIN_INFO = 2;
47 
48     /**
49      * If the domain provided by the client does not match the one received
50      * from server.
51      */
52     //public final static int DOMAIN_UNMATCH = 3;
53 
54     /**
55      * If the client name is not found on server's user database.
56      */
57     public static final int USER_UNKNOWN = 3;
58 
59     /**
60      * If authentication fails.
61      */
62     public static final int AUTH_FAILED = 4;
63 
64     /**
65      * If an illegal version string is provided.
66      */
67     public static final int BAD_VERSION = 5;
68 
69     /**
70      * Protocol errors.
71      */
72     public static final int PROTOCOL = 6;
73 
74     private int errorCode;
75 
76     /**
77      * Constructs an NTLMException object.
78      * @param errorCode the error code, which can be retrieved by
79      * the {@link #errorCode() } method.
80      * @param msg the string message, which can be retrived by
81      * the {@link Exception#getMessage() } method.
82      */
83     public NTLMException(int errorCode, String msg) {
84         super(msg);
85         this.errorCode = errorCode;
86     }
87 
88     /**
89      * Returns the error code associated with this NTLMException.
90      * @return the error code
91      */
92     public int errorCode() {
93         return errorCode;
94     }
95 }
