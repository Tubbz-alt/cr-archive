 44 /**
 45  * The X509V1CertImpl class is used as a conversion wrapper around
 46  * sun.security.x509.X509Cert certificates when running under JDK1.1.x.
 47  *
 48  * @deprecated This is the implementation class for the deprecated
 49  *  {@code javax.security.cert.X509Certificate} class. The classes in the
 50  *  {@code java.security.cert} package should be used instead.
 51  *
 52  * @author Jeff Nisewanger
 53  */
 54 @SuppressWarnings("removal")
 55 @Deprecated(since="9", forRemoval=true)
 56 public class X509V1CertImpl extends X509Certificate implements Serializable {
 57     @java.io.Serial
 58     static final long serialVersionUID = -2048442350420423405L;
 59     private java.security.cert.X509Certificate wrappedCert;
 60 
 61     private static synchronized java.security.cert.CertificateFactory
 62     getFactory()
 63     throws java.security.cert.CertificateException
 64     {
 65         return java.security.cert.CertificateFactory.getInstance("X.509");
 66     }
 67 
 68     /**
 69      * Default constructor.
 70      */
 71     public X509V1CertImpl() { }
 72 
 73     /**
 74      * Unmarshals a certificate from its encoded form, parsing the
 75      * encoded bytes.  This form of constructor is used by agents which
 76      * need to examine and use certificate contents.  That is, this is
 77      * one of the more commonly used constructors.  Note that the buffer
 78      * must include only a certificate, and no "garbage" may be left at
 79      * the end.  If you need to ignore data at the end of a certificate,
 80      * use another constructor.
 81      *
 82      * @param certData the encoded bytes, with no trailing padding.
 83      * @exception CertificateException on parsing errors.
 84      */
 85     public X509V1CertImpl(byte[] certData)
 86     throws CertificateException {
 87         try {
 88             ByteArrayInputStream bs;
 89 
 90             bs = new ByteArrayInputStream(certData);
 91             wrappedCert = (java.security.cert.X509Certificate)
 92                 getFactory().generateCertificate(bs);
 93         } catch (java.security.cert.CertificateException e) {
 94             throw new CertificateException(e.getMessage());
 95         }
 96     }
 97 
 98     /**
 99      * unmarshals an X.509 certificate from an input stream.
100      *
101      * @param in an input stream holding at least one certificate
102      * @exception CertificateException on parsing errors.
103      */
104     public X509V1CertImpl(InputStream in)
105     throws CertificateException {
106         try {
107             wrappedCert = (java.security.cert.X509Certificate)
108                 getFactory().generateCertificate(in);
109         } catch (java.security.cert.CertificateException e) {
110             throw new CertificateException(e.getMessage());
111         }
112     }
113 
114     /**
115      * Returns the encoded form of this certificate. It is
116      * assumed that each certificate type would have only a single
117      * form of encoding; for example, X.509 certificates would
118      * be encoded as ASN.1 DER.
119      */
120     public byte[] getEncoded() throws CertificateEncodingException {
121         try {
122             return wrappedCert.getEncoded();
123         } catch (java.security.cert.CertificateEncodingException e) {
124             throw new CertificateEncodingException(e.getMessage());
125         }
126     }
127 
128     /**
129      * Throws an exception if the certificate was not signed using the
130      * verification key provided.  Successfully verifying a certificate
131      * does <em>not</em> indicate that one should trust the entity which
132      * it represents.
133      *
134      * @param key the public key used for verification.
135      */
136     public void verify(PublicKey key)
137         throws CertificateException, NoSuchAlgorithmException,
138         InvalidKeyException, NoSuchProviderException,
139         SignatureException
140     {
141         try {
142             wrappedCert.verify(key);
143         } catch (java.security.cert.CertificateException e) {
144             throw new CertificateException(e.getMessage());
145         }
146     }
147 
148     /**
149      * Throws an exception if the certificate was not signed using the
150      * verification key provided.  Successfully verifying a certificate
151      * does <em>not</em> indicate that one should trust the entity which
152      * it represents.
153      *
154      * @param key the public key used for verification.
155      * @param sigProvider the name of the provider.
156      */
157     public void verify(PublicKey key, String sigProvider)
158         throws CertificateException, NoSuchAlgorithmException,
159         InvalidKeyException, NoSuchProviderException,
160         SignatureException
161     {
162         try {
163             wrappedCert.verify(key, sigProvider);
164         } catch (java.security.cert.CertificateException e) {
165             throw new CertificateException(e.getMessage());
166         }
167     }
168 
169     /**
170      * Checks that the certificate is currently valid, i.e. the current
171      * time is within the specified validity period.
172      */
173     public void checkValidity() throws
174       CertificateExpiredException, CertificateNotYetValidException {
175         checkValidity(new Date());
176     }
177 
178     /**
179      * Checks that the specified date is within the certificate's
180      * validity period, or basically if the certificate would be
181      * valid at the specified date/time.
182      *
183      * @param date the Date to check against to see if this certificate
184      *        is valid at that date/time.
185      */
186     public void checkValidity(Date date) throws
187       CertificateExpiredException, CertificateNotYetValidException {
188         try {
189             wrappedCert.checkValidity(date);
190         } catch (java.security.cert.CertificateNotYetValidException e) {
191             throw new CertificateNotYetValidException(e.getMessage());
192         } catch (java.security.cert.CertificateExpiredException e) {
193             throw new CertificateExpiredException(e.getMessage());
194         }
195     }
196 
197 
198     /**
199      * Returns a printable representation of the certificate.  This does not
200      * contain all the information available to distinguish this from any
201      * other certificate.  The certificate must be fully constructed
202      * before this function may be called.
203      */
204     public String toString() {
205         return wrappedCert.toString();
206     }
207 
208     /**
209      * Gets the publickey from this certificate.
210      *
211      * @return the publickey.
212      */
213     public PublicKey getPublicKey() {
214         PublicKey key = wrappedCert.getPublicKey();
215         return key;
216     }
217

220      *
221      * @return the version number.

225     }

228      * Gets the serial number from the certificate.
229      *
230      * @return the serial number.

232     public BigInteger getSerialNumber() {

234     }

237      * Gets the subject distinguished name from the certificate.
238      *
239      * @return the subject name.

241      */
242     public Principal getSubjectDN() {
243         return wrappedCert.getSubjectDN();
244     }
245 
246     /**
247      * Gets the issuer distinguished name from the certificate.
248      *
249      * @return the issuer name.
250      * @exception CertificateException if a parsing error occurs.
251      */
252     public Principal getIssuerDN() {
253         return wrappedCert.getIssuerDN();
254     }
255 
256     /**
257      * Gets the notBefore date from the validity period of the certificate.
258      *
259      * @return the start date of the validity period.
260      * @exception CertificateException if a parsing error occurs.
261      */
262     public Date getNotBefore() {
263         return wrappedCert.getNotBefore();
264     }
265 
266     /**
267      * Gets the notAfter date from the validity period of the certificate.
268      *
269      * @return the end date of the validity period.
270      * @exception CertificateException if a parsing error occurs.

272     public Date getNotAfter() {
273         return wrappedCert.getNotAfter();
274     }

276     /**
277      * Gets the signature algorithm name for the certificate
278      * signature algorithm.
279      * For example, the string "SHA1/DSA".
280      *
281      * @return the signature algorithm name.
282      * @exception CertificateException if a parsing error occurs.
283      */
284     public String getSigAlgName() {
285         return wrappedCert.getSigAlgName();
286     }

289      * Gets the signature algorithm OID string from the certificate.
290      * For example, the string "1.2.840.10040.4.3"
291      *
292      * @return the signature algorithm oid string.
293      * @exception CertificateException if a parsing error occurs.
294      */
295     public String getSigAlgOID() {
296         return wrappedCert.getSigAlgOID();
297     }
298 
299     /**
300      * Gets the DER encoded signature algorithm parameters from this
301      * certificate's signature algorithm.
302      *
303      * @return the DER encoded signature algorithm parameters, or
304      *         null if no parameters are present.
305      * @exception CertificateException if a parsing error occurs.
306      */
307     public byte[] getSigAlgParams() {
308         return wrappedCert.getSigAlgParams();
309     }
310 
311     @java.io.Serial
312     private synchronized void writeObject(ObjectOutputStream stream)
313         throws IOException {
314         try {
315             stream.write(getEncoded());
316         } catch (CertificateEncodingException e) {
317             throw new IOException("getEncoded failed: " + e.getMessage());
318         }
319     }
320 
321     @java.io.Serial
322     private synchronized void readObject(ObjectInputStream stream)
323         throws IOException {
324         try {
325             wrappedCert = (java.security.cert.X509Certificate)
326                 getFactory().generateCertificate(stream);
327         } catch (java.security.cert.CertificateException e) {
328             throw new IOException("generateCertificate failed: " + e.getMessage());
329         }
330     }
331 
332     public java.security.cert.X509Certificate getX509Certificate() {
333         return wrappedCert;
334     }
