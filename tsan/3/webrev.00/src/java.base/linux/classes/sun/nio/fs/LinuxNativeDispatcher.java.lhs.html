 26 package sun.nio.fs;
 27 
 28 import java.security.AccessController;
 29 import java.security.PrivilegedAction;
 30 
 31 /**
 32  * Linux specific system calls.
 33  */
 34 
 35 class LinuxNativeDispatcher extends UnixNativeDispatcher {
 36     private LinuxNativeDispatcher() { }
 37 
 38    /**
 39     * FILE *setmntent(const char *filename, const char *type);
 40     */
 41     static long setmntent(byte[] filename, byte[] type) throws UnixException {
 42         NativeBuffer pathBuffer = NativeBuffers.asNativeBuffer(filename);
 43         NativeBuffer typeBuffer = NativeBuffers.asNativeBuffer(type);
 44         try {
 45             return setmntent0(pathBuffer.address(), typeBuffer.address());
 46         } finally {
 47             typeBuffer.release();
 48             pathBuffer.release();
 49         }
 50     }
 51     private static native long setmntent0(long pathAddress, long typeAddress)
 52         throws UnixException;
 53 
 54     /**
 55      * int getmntent(FILE *fp, struct mnttab *mp, int len);
 56      */
 57     static native int getmntent(long fp, UnixMountEntry entry)










 58         throws UnixException;
 59 
 60     /**
 61      * int endmntent(FILE* filep);
 62      */
 63     static native void endmntent(long stream) throws UnixException;
 64 
 65     /**
 66      * ssize_t fgetxattr(int filedes, const char *name, void *value, size_t size);
 67      */
 68     static int fgetxattr(int filedes, byte[] name, long valueAddress,
 69                          int valueLen) throws UnixException
 70     {
 71         NativeBuffer buffer = NativeBuffers.asNativeBuffer(name);
 72         try {
 73             return fgetxattr0(filedes, buffer.address(), valueAddress, valueLen);
 74         } finally {
 75             buffer.release();
 76         }
 77     }
 78 
 79     private static native int fgetxattr0(int filedes, long nameAddress,
 80         long valueAddress, int valueLen) throws UnixException;
 81 
 82     /**
 83      *  fsetxattr(int filedes, const char *name, const void *value, size_t size, int flags);
 84      */
 85     static void fsetxattr(int filedes, byte[] name, long valueAddress,
 86         int valueLen) throws UnixException
 87     {
 88         NativeBuffer buffer = NativeBuffers.asNativeBuffer(name);
 89         try {
 90             fsetxattr0(filedes, buffer.address(), valueAddress, valueLen);
 91         } finally {
 92             buffer.release();
 93         }
 94     }
 95 
 96     private static native void fsetxattr0(int filedes, long nameAddress,
 97         long valueAddress, int valueLen) throws UnixException;
 98 
 99     /**
100      * fremovexattr(int filedes, const char *name);
101      */
102     static void fremovexattr(int filedes, byte[] name) throws UnixException {
103         NativeBuffer buffer = NativeBuffers.asNativeBuffer(name);
104         try {
105             fremovexattr0(filedes, buffer.address());
106         } finally {
107             buffer.release();
108         }
109     }
110 
111     private static native void fremovexattr0(int filedes, long nameAddress)
112         throws UnixException;
113 
114     /**
115      * size_t flistxattr(int filedes, const char *list, size_t size)
116      */
117     static native int flistxattr(int filedes, long listAddress, int size)
118         throws UnixException;
119 
120     // initialize
121     private static native void init();
122 
123     static {
124         AccessController.doPrivileged(new PrivilegedAction<>() {
125             public Void run() {
126                 System.loadLibrary("nio");
127                 return null;
128         }});
129         init();
130     }
131 }
