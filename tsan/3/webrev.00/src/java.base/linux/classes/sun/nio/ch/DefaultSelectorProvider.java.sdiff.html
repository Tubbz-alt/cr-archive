26 package sun.nio.ch;
27 
28 import java.security.AccessController;
29 import java.security.PrivilegedAction;
30 
31 /**
32  * Creates this platform's default SelectorProvider
33  */
34 
35 public class DefaultSelectorProvider {
36     private static final SelectorProviderImpl INSTANCE;
37     static {
38         PrivilegedAction<SelectorProviderImpl> pa = EPollSelectorProvider::new;
39         INSTANCE = AccessController.doPrivileged(pa);
40     }
41 
42     /**
43      * Prevent instantiation.
44      */
45     private DefaultSelectorProvider() { }
46 
47     /**
48      * Returns the default SelectorProvider implementation.
49      */
50     public static SelectorProviderImpl get() {
51         return INSTANCE;
52     }
53 }
