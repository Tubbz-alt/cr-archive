26 package sun.nio.fs;
27 
28 import java.io.IOException;
29 import java.nio.file.Path;


30 
31 /**
32  * File type detector that uses a file extension to look up its MIME type
33  * via the Apple Uniform Type Identifier interfaces.
34  */
35 class UTIFileTypeDetector extends AbstractFileTypeDetector {
36     UTIFileTypeDetector() {
37         super();
38     }
39 
40     private native String probe0(String fileExtension) throws IOException;
41 
42     @Override
43     protected String implProbeContentType(Path path) throws IOException {
44         Path fn = path.getFileName();
45         if (fn == null)
46             return null;  // no file name
47 
48         String ext = getExtension(fn.toString());
49         if (ext.isEmpty())
50             return null;  // no extension
51 
52         return probe0(ext);
53     }
54 
55     static {
56         jdk.internal.loader.BootLoader.loadLibrary("nio");






57     }
58 }
