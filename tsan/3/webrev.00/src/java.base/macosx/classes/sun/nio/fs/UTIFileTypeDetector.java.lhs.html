26 package sun.nio.fs;
27 
28 import java.io.IOException;
29 import java.nio.file.Path;
30 import java.security.AccessController;
31 import java.security.PrivilegedAction;
32 
33 /**
34  * File type detector that uses a file extension to look up its MIME type
35  * via the Apple Uniform Type Identifier interfaces.
36  */
37 class UTIFileTypeDetector extends AbstractFileTypeDetector {
38     UTIFileTypeDetector() {
39         super();
40     }
41 
42     private native String probe0(String fileExtension) throws IOException;
43 
44     @Override
45     protected String implProbeContentType(Path path) throws IOException {
46         Path fn = path.getFileName();
47         if (fn == null)
48             return null;  // no file name
49 
50         String ext = getExtension(fn.toString());
51         if (ext.isEmpty())
52             return null;  // no extension
53 
54         return probe0(ext);
55     }
56 
57     static {
58         AccessController.doPrivileged(new PrivilegedAction<>() {
59             @Override
60             public Void run() {
61                 System.loadLibrary("nio");
62                 return null;
63             }
64         });
65     }
66 }
