47     /**
48      * Finds, and returns, the mount entry for the file system where the file
49      * resides.
50      */
51     @Override
52     UnixMountEntry findMountEntry() throws IOException {
53         UnixFileSystem fs = file().getFileSystem();
54 
55         // step 1: get realpath
56         UnixPath path = null;
57         try {
58             byte[] rp = UnixNativeDispatcher.realpath(file());
59             path = new UnixPath(fs, rp);
60         } catch (UnixException x) {
61             x.rethrowAsIOException(file());
62         }
63 
64         // step 2: find mount point
65         UnixPath parent = path.getParent();
66         while (parent != null) {
67             UnixFileAttributes attrs = null;
68             try {
69                 attrs = UnixFileAttributes.get(parent, true);
70             } catch (UnixException x) {
71                 x.rethrowAsIOException(parent);
72             }
73             if (attrs.dev() != dev())
74                 break;
75             path = parent;
76             parent = parent.getParent();
77         }
78 
79         // step 3: lookup mounted file systems
80         byte[] dir = path.asByteArray();
81         for (UnixMountEntry entry: fs.getMountEntries()) {
82             if (Arrays.equals(dir, entry.dir()))
83                 return entry;
84         }
85 
86         throw new IOException("Mount point not found in fstab");
87     }
88 }
