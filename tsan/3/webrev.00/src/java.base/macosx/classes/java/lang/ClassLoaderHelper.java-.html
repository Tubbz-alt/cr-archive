26 package java.lang;
27 
28 import java.io.File;
29 
30 class ClassLoaderHelper {
31 
32     private ClassLoaderHelper() {}
33 
34     /**
35      * Indicates, whether PATH env variable is allowed to contain quoted entries.
36      */
37     static final boolean allowsQuotedPathElements = false;
38 
39     /**
40      * Returns an alternate path name for the given file
41      * such that if the original pathname did not exist, then the
42      * file may be located at the alternate location.
43      * For mac, this replaces the final .dylib suffix with .jnilib
44      */
45     static File mapAlternativeName(File lib) {
46         String name = lib.toString();
47         int index = name.lastIndexOf('.');
48         if (index < 0) {
49             return null;
50         }
51         return new File(name.substring(0, index) + ".jnilib");
52     }
53 }
