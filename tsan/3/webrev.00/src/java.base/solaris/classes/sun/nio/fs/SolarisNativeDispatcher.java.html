26 package sun.nio.fs;
27 
28 /**
29  * Solaris specific system calls.
30  */
31 
32 class SolarisNativeDispatcher extends UnixNativeDispatcher {
33     private SolarisNativeDispatcher() { }
34 
35     /**
36      * int getextmntent(FILE *fp, struct extmnttab *mp, int len);
37      */
38     static native int getextmntent(long fp, UnixMountEntry entry)
39         throws UnixException;
40 
41     /**
42      * int facl(int filedes, int cmd, int nentries, void aclbufp)
43      */
44     static native int facl(int fd, int cmd, int nentries, long aclbufp)
45         throws UnixException;
46 
47 
48     // initialize
49     private static native void init();
50 
51     static {
52         jdk.internal.loader.BootLoader.loadLibrary("nio");
53         init();
54     }
55 }
