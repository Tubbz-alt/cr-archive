26 package sun.nio.fs;
27 
28 import java.security.AccessController;
29 import java.security.PrivilegedAction;
30 
31 /**
32  * Solaris specific system calls.
33  */
34 
35 class SolarisNativeDispatcher extends UnixNativeDispatcher {
36     private SolarisNativeDispatcher() { }
37 
38     /**
39      * int getextmntent(FILE *fp, struct extmnttab *mp, int len);
40      */
41     static native int getextmntent(long fp, UnixMountEntry entry)
42         throws UnixException;
43 
44     /**
45      * int facl(int filedes, int cmd, int nentries, void aclbufp)
46      */
47     static native int facl(int fd, int cmd, int nentries, long aclbufp)
48         throws UnixException;
49 
50 
51     // initialize
52     private static native void init();
53 
54     static {
55         AccessController.doPrivileged(new PrivilegedAction<Void>() {
56             public Void run() {
57                 System.loadLibrary("nio");
58                 return null;
59         }});
60         init();
61     }
62 }
