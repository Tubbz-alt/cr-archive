45     /**
46      * Parse a PATH env variable.
47      *
48      * Empty elements will be replaced by dot.
49      */
50     static String[] parsePath(String ldPath) {
51         char ps = File.pathSeparatorChar;
52         ArrayList<String> paths = new ArrayList<>();
53         int pathStart = 0;
54         int pathEnd;
55         while ((pathEnd = ldPath.indexOf(ps, pathStart)) >= 0) {
56             paths.add((pathStart < pathEnd) ?
57                     ldPath.substring(pathStart, pathEnd) : ".");
58             pathStart = pathEnd + 1;
59         }
60         int ldLen = ldPath.length();
61         paths.add((pathStart < ldLen) ?
62                 ldPath.substring(pathStart, ldLen) : ".");
63         return paths.toArray(new String[paths.size()]);
64     }
