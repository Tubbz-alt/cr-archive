1103     // -- support for copying String with a NativeBuffer --
1104 
1105     private static final Unsafe unsafe = Unsafe.getUnsafe();
1106 
1107     static NativeBuffer asNativeBuffer(String s) {
1108         int stringLengthInBytes = s.length() << 1;
1109         int sizeInBytes = stringLengthInBytes + 2;  // char terminator
1110 
1111         // get a native buffer of sufficient size
1112         NativeBuffer buffer = NativeBuffers.getNativeBufferFromCache(sizeInBytes);
1113         if (buffer == null) {
1114             buffer = NativeBuffers.allocNativeBuffer(sizeInBytes);
1115         } else {
1116             // buffer already contains the string contents
1117             if (buffer.owner() == s)
1118                 return buffer;
1119         }
1120 
1121         // copy into buffer and zero terminate
1122         char[] chars = s.toCharArray();
1123         unsafe.copyMemory(chars, Unsafe.ARRAY_CHAR_BASE_OFFSET, null,
1124             buffer.address(), (long)stringLengthInBytes);
1125         unsafe.putChar(buffer.address() + stringLengthInBytes, (char)0);
1126         buffer.setOwner(s);
1127         return buffer;
1128     }
