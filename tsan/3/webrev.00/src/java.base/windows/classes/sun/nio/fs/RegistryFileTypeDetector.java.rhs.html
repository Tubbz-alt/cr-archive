26 package sun.nio.fs;
27 
28 import java.nio.file.*;
29 import java.io.IOException;


30 
31 /**
32  * File type detector that does lookup of file extension using Windows Registry.
33  */
34 
35 public class RegistryFileTypeDetector
36     extends AbstractFileTypeDetector
37 {
38     public RegistryFileTypeDetector() {
39         super();
40     }
41 
42     @Override
43     public String implProbeContentType(Path file) throws IOException {
44         if (!(file instanceof Path))
45             return null;
46 
47         // get file extension
48         Path name = file.getFileName();
49         if (name == null)
50             return null;
51         String filename = name.toString();
52         int dot = filename.lastIndexOf('.');
53         if ((dot < 0) || (dot == (filename.length()-1)))
54             return null;
55 
56         // query HKEY_CLASSES_ROOT\<ext>
57         String key = filename.substring(dot);
58         NativeBuffer keyBuffer = WindowsNativeDispatcher.asNativeBuffer(key);
59         NativeBuffer nameBuffer = WindowsNativeDispatcher.asNativeBuffer("Content Type");
60         try {
61             return queryStringValue(keyBuffer.address(), nameBuffer.address());
62         } finally {
63             nameBuffer.release();
64             keyBuffer.release();
65         }
66     }
67 
68     private static native String queryStringValue(long subKey, long name);
69 
70     static {
71         // nio.dll has dependency on net.dll
72         jdk.internal.loader.BootLoader.loadLibrary("net");
73         jdk.internal.loader.BootLoader.loadLibrary("nio");





74     }
75 }
