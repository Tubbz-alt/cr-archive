26 package sun.nio.ch;
27 
28 import java.io.*;
29 
30 /**
31  * Allows different platforms to call different native methods
32  * for read and write operations.
33  */
34 
35 class SocketDispatcher extends NativeDispatcher
36 {
37 
38     static {
39         IOUtil.load();
40     }
41 
42     int read(FileDescriptor fd, long address, int len) throws IOException {
43         return read0(fd, address, len);
44     }
45 
46     long readv(FileDescriptor fd, long address, int len) throws IOException {
47         return readv0(fd, address, len);
48     }
49 
50     int write(FileDescriptor fd, long address, int len) throws IOException {
51         return write0(fd, address, len);
52     }
53 
54     long writev(FileDescriptor fd, long address, int len) throws IOException {
55         return writev0(fd, address, len);
56     }
57 
58     void preClose(FileDescriptor fd) throws IOException {
59         preClose0(fd);
60     }
61 
62     void close(FileDescriptor fd) throws IOException {
63         close0(fd);
64     }
65 
66     //-- Native methods
67     static native int read0(FileDescriptor fd, long address, int len)
68         throws IOException;
69 
70     static native long readv0(FileDescriptor fd, long address, int len)
71         throws IOException;
72 
73     static native int write0(FileDescriptor fd, long address, int len)
74         throws IOException;
75 
76     static native long writev0(FileDescriptor fd, long address, int len)
77         throws IOException;
78 
79     static native void preClose0(FileDescriptor fd) throws IOException;
80 
81     static native void close0(FileDescriptor fd) throws IOException;
82 }
