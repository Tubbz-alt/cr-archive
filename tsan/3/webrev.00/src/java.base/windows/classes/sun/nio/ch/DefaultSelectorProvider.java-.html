26 package sun.nio.ch;
27 
28 import java.nio.channels.spi.SelectorProvider;

30 
31 /**
32  * Creates this platform's default SelectorProvider
33  */
34 
35 public class DefaultSelectorProvider {
36 
37     /**
38      * Prevent instantiation.
39      */
40     private DefaultSelectorProvider() { }
41 
42     /**
43      * Returns the default SelectorProvider.
44      */
45     public static SelectorProvider create() {
46         return new sun.nio.ch.WindowsSelectorProvider();
47     }
48 
49 }
