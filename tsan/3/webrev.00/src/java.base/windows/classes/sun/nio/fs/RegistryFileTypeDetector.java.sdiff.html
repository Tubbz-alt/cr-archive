26 package sun.nio.fs;
27 
28 import java.nio.file.*;
29 import java.io.IOException;
30 import java.security.AccessController;
31 import java.security.PrivilegedAction;
32 
33 /**
34  * File type detector that does lookup of file extension using Windows Registry.
35  */
36 
37 public class RegistryFileTypeDetector
38     extends AbstractFileTypeDetector
39 {
40     public RegistryFileTypeDetector() {
41         super();
42     }
43 
44     @Override
45     public String implProbeContentType(Path file) throws IOException {
46         if (!(file instanceof Path))
47             return null;
48 
49         // get file extension
50         Path name = file.getFileName();
51         if (name == null)

53         String filename = name.toString();
54         int dot = filename.lastIndexOf('.');
55         if ((dot < 0) || (dot == (filename.length()-1)))
56             return null;
57 
58         // query HKEY_CLASSES_ROOT\<ext>
59         String key = filename.substring(dot);
60         NativeBuffer keyBuffer = WindowsNativeDispatcher.asNativeBuffer(key);
61         NativeBuffer nameBuffer = WindowsNativeDispatcher.asNativeBuffer("Content Type");
62         try {
63             return queryStringValue(keyBuffer.address(), nameBuffer.address());
64         } finally {
65             nameBuffer.release();
66             keyBuffer.release();
67         }
68     }
69 
70     private static native String queryStringValue(long subKey, long name);
71 
72     static {
73         AccessController.doPrivileged(new PrivilegedAction<Void>() {
74             @Override
75             public Void run() {
76                 // nio.dll has dependency on net.dll
77                 System.loadLibrary("net");
78                 System.loadLibrary("nio");
79                 return null;
80         }});
81     }
82 }
