26 package sun.net.www.protocol.file;
27 
28 import java.io.File;
29 import java.io.FilePermission;
30 import java.net.URL;
31 import java.security.Permission;
32 
33 final class UNCFileURLConnection extends FileURLConnection {
34 
35     private final String effectivePath;
36     private volatile Permission permission;
37 
38     UNCFileURLConnection(URL u, File file, String effectivePath) {
39         super(u, file);
40         this.effectivePath = effectivePath;
41     }
42 
43     @Override
44     public Permission getPermission() {
45         Permission perm = permission;
46         if (perm == null) {
47             permission = perm = new FilePermission(effectivePath, "read");
48         }
49         return perm;
50     }
51 }
