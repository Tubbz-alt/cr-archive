26 package sun.nio.ch;
27 
28 import java.io.*;
29 import java.net.*;
30 
31 /**
32  * Allows different platforms to call different native methods
33  * for read and write operations.
34  */
35 
36 class DatagramDispatcher extends NativeDispatcher
37 {
38     static {
39         IOUtil.load();
40     }
41 
42     int read(FileDescriptor fd, long address, int len) throws IOException {
43         return read0(fd, address, len);
44     }
45 
46     long readv(FileDescriptor fd, long address, int len) throws IOException {
47         return readv0(fd, address, len);
48     }
49 
50     int write(FileDescriptor fd, long address, int len) throws IOException {
51         return write0(fd, address, len);
52     }
53 
54     long writev(FileDescriptor fd, long address, int len) throws IOException {
55         return writev0(fd, address, len);
56     }
57 
58     void close(FileDescriptor fd) throws IOException {
59         SocketDispatcher.close0(fd);
60     }
61 
62     static native int read0(FileDescriptor fd, long address, int len)


63         throws IOException;
64 
65     static native long readv0(FileDescriptor fd, long address, int len)
66         throws IOException;
67 
68     static native int write0(FileDescriptor fd, long address, int len)
69         throws IOException;
70 
71     static native long writev0(FileDescriptor fd, long address, int len)
72         throws IOException;




73 }
