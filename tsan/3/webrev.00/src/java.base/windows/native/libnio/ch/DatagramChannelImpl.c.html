 37 /*
 38  * This function "purges" all outstanding ICMP port unreachable packets
 39  * outstanding on a socket and returns JNI_TRUE if any ICMP messages
 40  * have been purged. The rational for purging is to emulate normal BSD
 41  * behaviour whereby receiving a "connection reset" status resets the
 42  * socket.
 43  */
 44 jboolean purgeOutstandingICMP(JNIEnv *env, jclass clazz, jint fd)
 45 {
 46     jboolean got_icmp = JNI_FALSE;
 47     char buf[1];
 48     fd_set tbl;
 49     struct timeval t = { 0, 0 };
 50     SOCKETADDRESS sa;
 51     int addrlen = sizeof(sa);
 52 
 53     /*
 54      * Peek at the queue to see if there is an ICMP port unreachable. If there
 55      * is then receive it.
 56      */
 57     FD_ZERO(&tbl);
 58     FD_SET((u_int)fd, &tbl);
 59     while(1) {
 60         if (select(/*ignored*/fd+1, &tbl, 0, 0, &t) <= 0) {
 61             break;
 62         }
 63         if (recvfrom(fd, buf, 1, MSG_PEEK,
 64                      &sa.sa, &addrlen) != SOCKET_ERROR) {
 65             break;
 66         }
 67         if (WSAGetLastError() != WSAECONNRESET) {
 68             /* some other error - we don't care here */
 69             break;
 70         }
 71 
 72         recvfrom(fd, buf, 1, 0, &sa.sa, &addrlen);
 73         got_icmp = JNI_TRUE;
 74     }
 75 
 76     return got_icmp;
 77 }

 79 JNIEXPORT void JNICALL
 80 Java_sun_nio_ch_DatagramChannelImpl_disconnect0(JNIEnv *env, jclass clazz,
 81                                                 jobject fdo, jboolean isIPv6)
 82 {
 83     jint fd = fdval(env, fdo);
 84     int rv = 0;
 85     SOCKETADDRESS sa;
 86     int sa_len = sizeof(sa);
 87 
 88     memset(&sa, 0, sa_len);
 89 
 90     rv = connect((SOCKET)fd, &sa.sa, sa_len);
 91     if (rv == SOCKET_ERROR) {
 92         handleSocketError(env, WSAGetLastError());
 93     } else {
 94         /* Disable WSAECONNRESET errors as socket is no longer connected */
 95         BOOL enable = FALSE;
 96         DWORD bytesReturned = 0;
 97         WSAIoctl((SOCKET)fd, SIO_UDP_CONNRESET, &enable, sizeof(enable),
 98                  NULL, 0, &bytesReturned, NULL, NULL);
 99     }
100 }

102 JNIEXPORT jint JNICALL
103 Java_sun_nio_ch_DatagramChannelImpl_receive0(JNIEnv *env, jclass clazz,
104                                              jobject fdo, jlong bufAddress,
105                                              jint len, jlong senderAddress,
106                                              jboolean connected)
107 {
108     jint fd = fdval(env, fdo);
109     void *buf = (void *)jlong_to_ptr(bufAddress);
110     SOCKETADDRESS *sa = (SOCKETADDRESS *)jlong_to_ptr(senderAddress);
111     int sa_len = sizeof(SOCKETADDRESS);
112     BOOL retry = FALSE;
113     jint n;
114 
115     do {
116         retry = FALSE;
117         n = recvfrom((SOCKET)fd,
118                      (char *)buf,
119                      len,
120                      0,
121                      (struct sockaddr *)sa,
122                      &sa_len);
123 
124         if (n == SOCKET_ERROR) {
125             int theErr = (jint)WSAGetLastError();
126             if (theErr == WSAEMSGSIZE) {
127                 /* Spec says the rest of the data will be discarded... */
128                 n = len;
129             } else if (theErr == WSAECONNRESET) {
130                 purgeOutstandingICMP(env, clazz, fd);
131                 if (connected == JNI_FALSE) {
132                     retry = TRUE;
133                 } else {
134                     JNU_ThrowByName(env, JNU_JAVANETPKG "PortUnreachableException", 0);
135                     return IOS_THROWN;
136                 }
137             } else if (theErr == WSAEWOULDBLOCK) {
138                 return IOS_UNAVAILABLE;
139             } else return handleSocketError(env, theErr);
140         }
141     } while (retry);
142 
143     return n;
144 }

146 JNIEXPORT jint JNICALL
147 Java_sun_nio_ch_DatagramChannelImpl_send0(JNIEnv *env, jclass clazz,
148                                           jobject fdo, jlong bufAddress, jint len,
149                                           jlong targetAddress, jint targetAddressLen)
150 {
151     jint fd = fdval(env, fdo);
152     void *buf = (void *)jlong_to_ptr(bufAddress);
153     SOCKETADDRESS *sa = (SOCKETADDRESS *)jlong_to_ptr(targetAddress);
154     int sa_len = targetAddressLen;
155     jint rv;
156 
157     rv = sendto((SOCKET)fd, buf, len, 0,(struct sockaddr *)sa, sa_len);
158     if (rv == SOCKET_ERROR) {
159         int theErr = (jint)WSAGetLastError();
160         if (theErr == WSAEWOULDBLOCK) {
161             return IOS_UNAVAILABLE;
162         }
163         return handleSocketError(env, (jint)WSAGetLastError());
164     }
165     return rv;
166 }
