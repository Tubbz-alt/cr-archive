 26 #include <jni.h>
 27 #include <windows.h>
 28 #include "jni_util.h"

 29 #include <urlmon.h>
 30 
 31 JNIEXPORT jboolean JNICALL Java_sun_net_www_protocol_http_ntlm_NTLMAuthentication_isTrustedSite(JNIEnv *env, jclass clazz, jstring url )







 32 {










 33 




 34     HRESULT hr;
 35     DWORD dwZone;
 36     DWORD  pPolicy = 0;
 37     IInternetSecurityManager *spSecurityManager;
 38     jboolean ret;
 39 



 40     // Create IInternetSecurityManager
 41     hr = CoInternetCreateSecurityManager(NULL, &spSecurityManager, (DWORD)0);
 42     if (FAILED(hr)) {
 43         return JNI_FALSE;
 44     }
 45 
 46     const LPCWSTR bstrURL = (LPCWSTR)((*env)->GetStringChars(env, url, NULL));
 47     if (bstrURL == NULL) {
 48         if (!(*env)->ExceptionCheck(env))
 49             JNU_ThrowOutOfMemoryError(env, NULL);
 50         spSecurityManager->lpVtbl->Release(spSecurityManager);
 51         return JNI_FALSE;
 52     }
 53 
 54     // Determines the policy for the URLACTION_CREDENTIALS_USE action and display
 55     // a user interface, if the policy indicates that the user should be queried
 56     hr = spSecurityManager->lpVtbl->ProcessUrlAction(
 57         spSecurityManager,
 58         bstrURL,
 59         URLACTION_CREDENTIALS_USE,
 60         (LPBYTE)&pPolicy,
 61         sizeof(DWORD), 0, 0, 0, 0);
 62 
 63     if (FAILED(hr)) {
 64         ret = JNI_FALSE;
 65         goto cleanupAndReturn;
 66     }
 67 
 68     // If these two User Authentication Logon options is selected
 69     // Anonymous logon
 70     // Prompt for user name and password
 71     if (pPolicy == URLPOLICY_CREDENTIALS_ANONYMOUS_ONLY ||
 72         pPolicy == URLPOLICY_CREDENTIALS_MUST_PROMPT_USER) {
 73         ret = JNI_FALSE;
 74         goto cleanupAndReturn;
 75     }
 76 
 77     // Option "Automatic logon with current user name and password" is selected
 78     if (pPolicy == URLPOLICY_CREDENTIALS_SILENT_LOGON_OK) {
 79         ret = JNI_TRUE;
 80         goto cleanupAndReturn;
 81     }
 82 
 83     // Option "Automatic logon only in intranet zone" is selected
 84     if (pPolicy == URLPOLICY_CREDENTIALS_CONDITIONAL_PROMPT) {
 85 
 86         // Gets the zone index from the specified URL
 87         hr = spSecurityManager->lpVtbl->MapUrlToZone(
 88                 spSecurityManager, bstrURL, &dwZone, 0);
 89         if (FAILED(hr)) {
 90             ret = JNI_FALSE;
 91             goto cleanupAndReturn;
 92         }
 93 
 94         // Check if the URL is in Local or Intranet zone
 95         if (dwZone == URLZONE_INTRANET || dwZone == URLZONE_LOCAL_MACHINE) {
 96             ret = JNI_TRUE;
 97             goto cleanupAndReturn;
 98         }
 99     }
100     ret = JNI_FALSE;
101 
102 cleanupAndReturn:
103     (*env)->ReleaseStringChars(env, url, bstrURL);
104     spSecurityManager->lpVtbl->Release(spSecurityManager);
105     return ret;
106 }
