42 JNIEXPORT HMODULE JDK_LoadSystemLibrary(const char* name) {
43     HMODULE handle = NULL;
44     char path[MAX_PATH];
45 
46     if (GetSystemDirectory(path, sizeof(path)) != 0) {
47         strcat(path, "\\");
48         strcat(path, name);
49         handle = LoadLibrary(path);
50     }
51 
52     if (handle == NULL) {
53         if (GetWindowsDirectory(path, sizeof(path)) != 0) {
54             strcat(path, "\\");
55             strcat(path, name);
56             handle = LoadLibrary(path);
57         }
58     }
59     return handle;
60 }
