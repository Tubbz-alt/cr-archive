26 #include <windows.h>
27 #include "jdk_util.h"
28 
29 #define JVM_DLL "jvm.dll"
30 











31 JNIEXPORT HMODULE JDK_LoadSystemLibrary(const char* name) {
32     HMODULE handle = NULL;
33     char path[MAX_PATH];
34 
35     if (GetSystemDirectory(path, sizeof(path)) != 0) {
36         strcat(path, "\\");
37         strcat(path, name);
38         handle = LoadLibrary(path);
39     }
40 
41     if (handle == NULL) {
42         if (GetWindowsDirectory(path, sizeof(path)) != 0) {
43             strcat(path, "\\");
44             strcat(path, name);
45             handle = LoadLibrary(path);
46         }
47     }
48     return handle;
49 }
