25 #include <stdio.h>
26 #include <sys/ldr.h>
27 
28 #include "java_md_aix.h"
29 
30 static unsigned char dladdr_buffer[0x4000];
31 
32 static int fill_dll_info(void) {
33     return loadquery(L_GETINFO, dladdr_buffer, sizeof(dladdr_buffer));
34 }
35 
36 static int dladdr_dont_reload(void *addr, Dl_info *info) {
37     const struct ld_info *p = (struct ld_info *)dladdr_buffer;
38     memset((void *)info, 0, sizeof(Dl_info));
39     for (;;) {
40         if (addr >= p->ldinfo_textorg &&
41             addr < (((char*)p->ldinfo_textorg) + p->ldinfo_textsize))
42         {
43             info->dli_fname = p->ldinfo_filename;
44             return 1;
45         }
46         if (!p->ldinfo_next) {
47             break;
48         }
49         p = (struct ld_info *)(((char *)p) + p->ldinfo_next);
50     }
51     return 0;
52 }
53 
54 int dladdr(void *addr, Dl_info *info) {
55     static int loaded = 0;
56     int rc = 0;
57     void *addr0;
58     if (!addr) {
59         return rc;
60     }
61     if (!loaded) {
62         if (fill_dll_info() == -1)
63             return rc;
64         loaded = 1;
65     }
66 
67     // first try with addr on cached data
68     rc = dladdr_dont_reload(addr, info);
69 
70     // addr could be an AIX function descriptor, so try dereferenced version
71     if (rc == 0) {
72         addr0 = *((void **)addr);
73         rc = dladdr_dont_reload(addr0, info);
74     }
75 
76     // if we had no success until now, maybe loadquery info is outdated.
77     // refresh and retry
78     if (rc == 0) {
79         if (fill_dll_info() == -1)
80             return rc;
81         rc = dladdr_dont_reload(addr, info);
82         if (rc == 0) {
83             rc = dladdr_dont_reload(addr0, info);
84         }
85     }
86     return rc;
87 }
