27 package sun.nio.fs;
28 
29 import java.security.AccessController;
30 import java.security.PrivilegedAction;
31 
32 /**
33  * AIX specific system calls.
34  */
35 
36 class AixNativeDispatcher extends UnixNativeDispatcher {
37     private AixNativeDispatcher() { }
38 
39     /**
40      * Special implementation of 'getextmntent' (see SolarisNativeDispatcher)
41      * that returns all entries at once.
42      */
43     static native UnixMountEntry[] getmntctl() throws UnixException;
44 
45     // initialize
46     private static native void init();
47 
48     static {
49         AccessController.doPrivileged(new PrivilegedAction<Void>() {
50             public Void run() {
51                 System.loadLibrary("nio");
52                 return null;
53         }});
54         init();
55     }
56 }
