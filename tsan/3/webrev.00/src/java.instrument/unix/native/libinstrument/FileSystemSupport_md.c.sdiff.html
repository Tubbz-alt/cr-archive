 47 char* basePath(const char* path) {
 48     char* last = strrchr(path, slash);
 49     if (last == NULL) {
 50         return (char*)path;
 51     } else {
 52         int len = last - path;
 53         char* str = (char*)malloc(len+1);
 54         if (str == NULL) {
 55             fprintf(stderr, "OOM error in native tmp buffer allocation");
 56             return NULL;
 57         }
 58         if (len > 0) {
 59             memcpy(str, path, len);
 60         }
 61         str[len] = '\0';
 62         return str;
 63     }
 64 }
