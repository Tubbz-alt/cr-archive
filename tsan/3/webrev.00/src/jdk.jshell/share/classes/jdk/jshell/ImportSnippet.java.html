 26 package jdk.jshell;
 27 
 28 import jdk.jshell.Key.ImportKey;
 29 
 30 /**
 31  * Snippet for an import declaration.
 32  * The Kind is {@link jdk.jshell.Snippet.Kind#IMPORT}.
 33  * <p>
 34  * {@code ImportSnippet} is immutable: an access to
 35  * any of its methods will always return the same result.
 36  * and thus is thread-safe.
 37  *
 38  * @since 9
 39  * @jls 7.5 Import Declarations
 40  */
 41 public class ImportSnippet extends PersistentSnippet {
 42 
 43     final String fullname;
 44     final String fullkey;
 45     final boolean isStatic;
 46     final boolean isStar;
 47 
 48     ImportSnippet(ImportKey key, String userSource, Wrap guts,
 49             String fullname, String name, SubKind subkind, String fullkey,
 50             boolean isStatic, boolean isStar) {
 51         super(key, userSource, guts, name, subkind, null);
 52         this.fullname = fullname;
 53         this.fullkey = fullkey;
 54         this.isStatic = isStatic;
 55         this.isStar = isStar;
 56     }
 57 
 58     /**
 59      * The identifying name of the import. For on-demand imports
 60      * ({@link jdk.jshell.Snippet.SubKind#TYPE_IMPORT_ON_DEMAND_SUBKIND} or
 61      * ({@link jdk.jshell.Snippet.SubKind#STATIC_IMPORT_ON_DEMAND_SUBKIND})
 62      * that is the full specifier including any
 63      * qualifiers and the asterisks. For single imports
 64      * ({@link jdk.jshell.Snippet.SubKind#SINGLE_TYPE_IMPORT_SUBKIND} or
 65      * ({@link jdk.jshell.Snippet.SubKind#SINGLE_STATIC_IMPORT_SUBKIND}),
 66      * it is the imported name. That is, the unqualified name.
 67      * @return the name of the import.
 68      */
 69     @Override
 70     public String name() {
 71         return key().name();
 72     }
 73 
 74     /**
 75      *
 76      * The qualified name of the import. For any imports
 77      * ({@link jdk.jshell.Snippet.SubKind#TYPE_IMPORT_ON_DEMAND_SUBKIND},
 78      * ({@link jdk.jshell.Snippet.SubKind#STATIC_IMPORT_ON_DEMAND_SUBKIND}),
 79      * ({@link jdk.jshell.Snippet.SubKind#SINGLE_TYPE_IMPORT_SUBKIND} or
 80      * ({@link jdk.jshell.Snippet.SubKind#SINGLE_STATIC_IMPORT_SUBKIND})
 81      * that is the full specifier including any
 82      * qualifiers and the asterisks.
 83      * @return the fullname of the import
 84      */
 85     public String fullname() {
 86         return fullname;
 87     }
 88 
 89     /**
 90      * Indicates whether this snippet represents a static import.
 91      *
 92      * @return {@code true} if this snippet represents a static import;
 93      * otherwise {@code false}
 94      */
 95     public boolean isStatic() {
 96         return isStatic;
 97     }
 98 
 99     /**** internal access ****/
100 
101     @Override
102     ImportKey key() {
103         return (ImportKey) super.key();
104     }
105 
106     @Override
107     String importLine(JShell state) {
108         return guts().wrapped();
109     }
110 }
