26 package jdk.jshell;
27 
28 /**
29  * Common interface for all wrappings of snippet source to Java source.
30  *
31  * Snippet index is index into the source of the snippet.  Note: If the snippet is a sub-range of
32  * the source, the index is not the index in the snippet.
33  *
34  * Wrap index is index into the wrapped snippet.
35  */
36 interface GeneralWrap {
37 
38     String wrapped();
39 
40     int snippetIndexToWrapIndex(int sni);
41 
42     int wrapIndexToSnippetIndex(int wi);
43 
44     default int wrapIndexToSnippetIndex(long wi) {
45         return wrapIndexToSnippetIndex((int) wi);
46     }
47 
48     int firstSnippetIndex();
49 
50     int lastSnippetIndex();
51 
52     int snippetLineToWrapLine(int snline);
53 
54     int wrapLineToSnippetLine(int wline);
55 
56     int firstSnippetLine();
57 
58     int lastSnippetLine();
59 
60     default String debugPos(long lpos) {
61         int pos = (int) lpos;
62         int len = wrapped().length();
63         return wrapped().substring(Math.max(0, pos - 10), Math.max(0, Math.min(len, pos)))
64                 + "###"
65                 + wrapped().substring(Math.max(0, Math.min(len, pos)), Math.max(0, Math.min(len, pos + 10)));
66     }
67 }
