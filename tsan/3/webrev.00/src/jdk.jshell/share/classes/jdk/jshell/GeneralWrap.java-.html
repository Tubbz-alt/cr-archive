26 package jdk.jshell;
27 
28 /**
29  * Common interface for all wrappings of snippet source to Java source.
30  *
31  * @author Robert Field
32  */
33 interface GeneralWrap {
34 
35     String wrapped();
36 
37     int snippetIndexToWrapIndex(int sni);
38 
39     int wrapIndexToSnippetIndex(int wi);
40 
41     default int wrapIndexToSnippetIndex(long wi) {
42         return wrapIndexToSnippetIndex((int) wi);
43     }
44 
45     int firstSnippetIndex();
46 
47     int lastSnippetIndex();
48 
49     int snippetLineToWrapLine(int snline);
50 
51     int wrapLineToSnippetLine(int wline);
52 
53     int firstSnippetLine();
54 
55     int lastSnippetLine();
56 
57     default String debugPos(long lpos) {
58         int pos = (int) lpos;
59         int len = wrapped().length();
60         return wrapped().substring(Math.max(0, pos - 10), Math.max(0, Math.min(len, pos)))
61                 + "###"
62                 + wrapped().substring(Math.max(0, Math.min(len, pos)), Math.max(0, Math.min(len, pos + 10)));
63     }
64 }
