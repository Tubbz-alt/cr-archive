25 package org.graalvm.compiler.core.common.spi;
26 
27 import org.graalvm.compiler.core.common.LIRKind;
28 import jdk.internal.vm.compiler.word.LocationIdentity;
29 
30 import jdk.vm.ci.code.ValueKindFactory;
31 
32 /**
33  * Details about a set of supported {@link ForeignCallDescriptor foreign calls}.
34  */
35 public interface ForeignCallsProvider extends ValueKindFactory<LIRKind> {
36 
37     /**
38      * Determines if a given foreign call is side-effect free. Deoptimization cannot return
39      * execution to a point before a foreign call that has a side effect.
40      */
41     boolean isReexecutable(ForeignCallDescriptor descriptor);
42 
43     /**
44      * Gets the set of memory locations killed by a given foreign call. Returning the special value
45      * {@link LocationIdentity#any()} denotes that the call kills all memory locations. Returning
46      * any empty array denotes that the call does not kill any memory locations.
47      */
48     LocationIdentity[] getKilledLocations(ForeignCallDescriptor descriptor);
49 
50     /**
51      * Determines if deoptimization can occur during a given foreign call.
52      */
53     boolean canDeoptimize(ForeignCallDescriptor descriptor);
54 
55     /**
56      * Identifies foreign calls which are guaranteed to include a safepoint check.
57      */
58     boolean isGuaranteedSafepoint(ForeignCallDescriptor descriptor);
59 
60     /**
61      * Gets the linkage for a foreign call.
62      */
63     ForeignCallLinkage lookupForeignCall(ForeignCallDescriptor descriptor);





64 }
