25 package org.graalvm.compiler.core.common.calc;
26 
27 import jdk.vm.ci.meta.Constant;
28 import jdk.vm.ci.meta.ConstantReflectionProvider;
29 import jdk.vm.ci.meta.PrimitiveConstant;
30 
31 public enum CanonicalCondition {
32     EQ(Condition.EQ),
33     LT(Condition.LT),
34     BT(Condition.BT);
35 
36     private final Condition condition;
37 
38     CanonicalCondition(Condition condition) {
39         assert condition.isCanonical();
40         this.condition = condition;
41     }
42 
43     public Condition asCondition() {
44         return condition;
45     }
46 
47     public boolean foldCondition(Constant lt, Constant rt, ConstantReflectionProvider constantReflection, boolean unorderedIsTrue) {
48         return asCondition().foldCondition(lt, rt, constantReflection, unorderedIsTrue);
49     }
50 
51     public boolean foldCondition(PrimitiveConstant lp, PrimitiveConstant rp, boolean unorderedIsTrue) {
52         return asCondition().foldCondition(lp, rp, unorderedIsTrue);
53     }
54 }
