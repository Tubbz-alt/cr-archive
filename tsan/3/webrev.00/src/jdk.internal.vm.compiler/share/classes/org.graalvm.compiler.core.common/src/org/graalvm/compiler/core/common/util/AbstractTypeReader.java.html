25 package org.graalvm.compiler.core.common.util;
26 
27 public abstract class AbstractTypeReader implements TypeReader {
28     @Override
29     public long getSV() {
30         return decodeSign(read());
31     }
32 
33     @Override
34     public long getUV() {
35         return read();
36     }
37 
38     public static long decodeSign(long value) {
39         return (value >>> 1) ^ -(value & 1);
40     }
41 
42     private long read() {
43         int b0 = getU1();
44         if (b0 < UnsafeArrayTypeWriter.NUM_LOW_CODES) {
45             return b0;
46         } else {
47             return readPacked(b0);
48         }
49     }
50 
51     private long readPacked(int b0) {
52         assert b0 >= UnsafeArrayTypeWriter.NUM_LOW_CODES;
53         long sum = b0;
54         long shift = UnsafeArrayTypeWriter.HIGH_WORD_SHIFT;
55         for (int i = 2;; i++) {
56             long b = getU1();
57             sum += b << shift;
58             if (b < UnsafeArrayTypeWriter.NUM_LOW_CODES || i == UnsafeArrayTypeWriter.MAX_BYTES) {
59                 return sum;
60             }
61             shift += UnsafeArrayTypeWriter.HIGH_WORD_SHIFT;
62         }
63     }
64 }
