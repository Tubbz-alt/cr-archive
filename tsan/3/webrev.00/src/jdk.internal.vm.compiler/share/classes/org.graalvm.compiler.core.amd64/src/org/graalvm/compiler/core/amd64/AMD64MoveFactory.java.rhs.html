 27 
 28 import static jdk.vm.ci.code.ValueUtil.isRegister;

 30 import static org.graalvm.compiler.lir.LIRValueUtil.isConstantValue;
 31 import static org.graalvm.compiler.lir.LIRValueUtil.isStackSlotValue;
 32 
 33 import org.graalvm.compiler.asm.amd64.AMD64Assembler;
 34 import org.graalvm.compiler.core.common.NumUtil;
 35 import org.graalvm.compiler.core.common.type.DataPointerConstant;
 36 import org.graalvm.compiler.debug.GraalError;
 37 import org.graalvm.compiler.lir.LIRInstruction;
 38 import org.graalvm.compiler.lir.amd64.AMD64AddressValue;

 41 import org.graalvm.compiler.lir.amd64.AMD64Move.AMD64StackMove;
 42 import org.graalvm.compiler.lir.amd64.AMD64Move.LeaDataOp;
 43 import org.graalvm.compiler.lir.amd64.AMD64Move.LeaOp;
 44 import org.graalvm.compiler.lir.amd64.AMD64Move.MoveFromConstOp;
 45 import org.graalvm.compiler.lir.amd64.AMD64Move.MoveFromRegOp;
 46 import org.graalvm.compiler.lir.amd64.AMD64Move.MoveToRegOp;

 48 import jdk.vm.ci.amd64.AMD64Kind;
 49 import jdk.vm.ci.code.Register;
 50 import jdk.vm.ci.meta.AllocatableValue;
 51 import jdk.vm.ci.meta.Constant;
 52 import jdk.vm.ci.meta.JavaConstant;
 53 import jdk.vm.ci.meta.PrimitiveConstant;
 54 import jdk.vm.ci.meta.Value;
 55 
 56 public abstract class AMD64MoveFactory extends AMD64MoveFactoryBase {
 57 
 58     public AMD64MoveFactory(BackupSlotProvider backupSlotProvider) {
 59         super(backupSlotProvider);
 60     }
 61 
 62     @Override
 63     public boolean canInlineConstant(Constant con) {
 64         if (con instanceof JavaConstant) {
 65             JavaConstant c = (JavaConstant) con;
 66             switch (c.getJavaKind()) {
 67                 case Long:
 68                     return NumUtil.isInt(c.asLong());
 69                 case Float:
 70                 case Double:
 71                     return false;
 72                 case Object:
 73                     return c.isNull();
 74                 default:
 75                     return true;
 76             }
 77         }
 78         return false;
 79     }
 80 
 81     @Override
 82     public boolean mayEmbedConstantLoad(Constant constant) {
 83         // Only consider not inlineable constants here.
 84         return constant instanceof PrimitiveConstant && ((PrimitiveConstant) constant).getJavaKind().isNumericFloat();
 85     }
 86 
 87     @Override
 88     public boolean allowConstantToStackMove(Constant constant) {
 89         if (constant instanceof DataPointerConstant) {
 90             return false;
 91         }
 92         if (constant instanceof JavaConstant && !AMD64Move.canMoveConst2Stack(((JavaConstant) constant))) {
 93             return false;
 94         }
 95         return true;
 96     }
 97 
 98     @Override
 99     public AMD64LIRInstruction createMove(AllocatableValue dst, Value src) {
100         if (src instanceof AMD64AddressValue) {
101             return new LeaOp(dst, (AMD64AddressValue) src, AMD64Assembler.OperandSize.QWORD);
102         } else if (isConstantValue(src)) {
103             return createLoad(dst, asConstant(src));
104         } else if (isRegister(src) || isStackSlotValue(dst)) {
105             return new MoveFromRegOp((AMD64Kind) dst.getPlatformKind(), dst, (AllocatableValue) src);
106         } else {
107             return new MoveToRegOp((AMD64Kind) dst.getPlatformKind(), dst, (AllocatableValue) src);
108         }
109     }
110 
111     @Override
112     public AMD64LIRInstruction createStackMove(AllocatableValue result, AllocatableValue input, Register scratchRegister, AllocatableValue backupSlot) {
113         return new AMD64StackMove(result, input, scratchRegister, backupSlot);
114     }
115 
116     @Override
117     public AMD64LIRInstruction createLoad(AllocatableValue dst, Constant src) {
118         if (src instanceof JavaConstant) {
119             return new MoveFromConstOp(dst, (JavaConstant) src);
120         } else if (src instanceof DataPointerConstant) {
121             return new LeaDataOp(dst, (DataPointerConstant) src);
122         } else {
123             throw GraalError.shouldNotReachHere(String.format("unsupported constant: %s", src));
124         }
125     }
126 
127     @Override
128     public LIRInstruction createStackLoad(AllocatableValue result, Constant input) {
129         if (input instanceof JavaConstant) {
130             return new MoveFromConstOp(result, (JavaConstant) input);
131         } else {
132             throw GraalError.shouldNotReachHere(String.format("unsupported constant for stack load: %s", input));
133         }
134     }
135 }
