 40 public class ProbabilityDirectiveTest extends GraalCompilerTest {
 41 
 42     /**
 43      * Called before a test is compiled.
 44      */
 45     @Override
 46     protected void before(ResolvedJavaMethod method) {
 47         // don't let -Xcomp pollute profile
 48         method.reprofile();
 49     }
 50 
 51     public static int branchProbabilitySnippet(int arg) {
 52         if (GraalDirectives.injectBranchProbability(0.125, arg > 0)) {
 53             GraalDirectives.controlFlowAnchor(); // prevent removal of the if
 54             return 1;
 55         } else {
 56             GraalDirectives.controlFlowAnchor(); // prevent removal of the if
 57             return 2;
 58         }
 59     }
 60 
 61     @Test
 62     public void testBranchProbability() {
 63         test("branchProbabilitySnippet", 5);
 64     }
 65 
 66     public static int branchProbabilitySnippet2(int arg) {
 67         if (!GraalDirectives.injectBranchProbability(0.125, arg <= 0)) {
 68             GraalDirectives.controlFlowAnchor(); // prevent removal of the if
 69             return 2;
 70         } else {
 71             GraalDirectives.controlFlowAnchor(); // prevent removal of the if
 72             return 1;
 73         }
 74     }
 75 
 76     @Test
 77     public void testBranchProbability2() {
 78         test("branchProbabilitySnippet2", 5);
 79     }
 80 
 81     @Override
 82     protected void checkLowTierGraph(StructuredGraph graph) {
 83         NodeIterable<IfNode> ifNodes = graph.getNodes(IfNode.TYPE);
 84         Assert.assertEquals("IfNode count", 1, ifNodes.count());
 85 
 86         IfNode ifNode = ifNodes.first();
 87         AbstractBeginNode oneSuccessor;
 88         if (returnValue(ifNode.trueSuccessor()) == 1) {
 89             oneSuccessor = ifNode.trueSuccessor();
 90         } else {
 91             assert returnValue(ifNode.falseSuccessor()) == 1;
 92             oneSuccessor = ifNode.falseSuccessor();
 93         }
 94         Assert.assertEquals("branch probability of " + ifNode, 0.125, ifNode.probability(oneSuccessor), 0);
 95     }
 96 
 97     private static int returnValue(AbstractBeginNode b) {
 98         ControlFlowAnchorNode anchor = (ControlFlowAnchorNode) b.next();
 99         ReturnNode returnNode = (ReturnNode) anchor.next();
100         return returnNode.result().asJavaConstant().asInt();
101     }
