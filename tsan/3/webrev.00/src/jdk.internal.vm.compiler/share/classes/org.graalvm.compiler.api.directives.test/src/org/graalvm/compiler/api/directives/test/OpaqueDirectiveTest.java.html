 25 package org.graalvm.compiler.api.directives.test;

 27 import java.lang.annotation.ElementType;
 28 import java.lang.annotation.Retention;
 29 import java.lang.annotation.RetentionPolicy;
 30 import java.lang.annotation.Target;
 31 
 32 import org.graalvm.compiler.api.directives.GraalDirectives;
 33 import org.graalvm.compiler.core.test.GraalCompilerTest;
 34 import org.graalvm.compiler.nodes.ConstantNode;
 35 import org.graalvm.compiler.nodes.ReturnNode;
 36 import org.graalvm.compiler.nodes.StructuredGraph;
 37 import org.graalvm.compiler.nodes.calc.AddNode;
 38 import org.graalvm.compiler.nodes.calc.ConditionalNode;
 39 import org.graalvm.compiler.phases.OptimisticOptimizations;
 40 import org.junit.Assert;
 41 import org.junit.Test;
 42 
 43 /**
 44  * Tests for {@link GraalDirectives#opaque}.
 45  *
 46  * There are two snippets for each kind:
 47  * <ul>
 48  * <li>opaque&lt;Kind&gt;Snippet verifies that constant folding is prevented by the opaque
 49  * directive.
 50  * <li>&lt;kind&gt;Snippet verifies that constant folding does happen if the opaque directive is not
 51  * there.
 52  * </ul>
 53  *
 54  */
 55 public class OpaqueDirectiveTest extends GraalCompilerTest {
 56 
 57     @Retention(RetentionPolicy.RUNTIME)
 58     @Target(ElementType.METHOD)
 59     private @interface OpaqueSnippet {
 60         Class<?> expectedReturnNode();
 61     }
 62 
 63     @OpaqueSnippet(expectedReturnNode = ConstantNode.class)
 64     public static boolean booleanSnippet() {
 65         return 5 > 3;
 66     }
 67 
 68     @OpaqueSnippet(expectedReturnNode = ConditionalNode.class)
 69     public static boolean opaqueBooleanSnippet() {
 70         return 5 > GraalDirectives.opaque(3);
 71     }
 72 
 73     @Test
 74     public void testBoolean() {
 75         test("booleanSnippet");
 76         test("opaqueBooleanSnippet");
 77     }
 78 
 79     @OpaqueSnippet(expectedReturnNode = ConstantNode.class)
 80     public static int intSnippet() {
 81         return 5 + 3;
 82     }
 83 
 84     @OpaqueSnippet(expectedReturnNode = AddNode.class)
 85     public static int opaqueIntSnippet() {
 86         return 5 + GraalDirectives.opaque(3);
 87     }
 88 
 89     @Test
 90     public void testInt() {
 91         test("intSnippet");
 92         test("opaqueIntSnippet");
 93     }
 94 
 95     @OpaqueSnippet(expectedReturnNode = ConstantNode.class)
 96     public static double doubleSnippet() {
 97         return 5. + 3.;
 98     }
 99 
100     @OpaqueSnippet(expectedReturnNode = AddNode.class)
101     public static double opaqueDoubleSnippet() {
102         return 5. + GraalDirectives.opaque(3.);
103     }
104 
105     @Test
106     public void testDouble() {
107         test("doubleSnippet");
108         test("opaqueDoubleSnippet");
109     }
110 
111     private static class Dummy {
112     }
113 
114     @OpaqueSnippet(expectedReturnNode = ConstantNode.class)
115     public static boolean objectSnippet() {
116         Object obj = new Dummy();
117         return obj == null;
118     }
119 
120     @OpaqueSnippet(expectedReturnNode = ConditionalNode.class)
121     public static boolean opaqueObjectSnippet() {
122         Object obj = new Dummy();
123         return GraalDirectives.opaque(obj) == null;
124     }
125 
126     @Test
127     public void testObject() {
128         test("objectSnippet");
129         test("opaqueObjectSnippet");
130     }
131 
132     @Override
133     protected OptimisticOptimizations getOptimisticOptimizations() {
134         return OptimisticOptimizations.ALL.remove(OptimisticOptimizations.Optimization.RemoveNeverExecutedCode);
135     }
136 
137     @Override
138     protected void checkLowTierGraph(StructuredGraph graph) {
139         OpaqueSnippet snippet = graph.method().getAnnotation(OpaqueSnippet.class);
140         for (ReturnNode returnNode : graph.getNodes(ReturnNode.TYPE)) {
141             Assert.assertEquals(snippet.expectedReturnNode(), returnNode.result().getClass());
142         }
143     }
144 }
