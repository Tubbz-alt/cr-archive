 25 package org.graalvm.compiler.api.directives.test;
 26 
 27 import java.lang.annotation.ElementType;
 28 import java.lang.annotation.Retention;
 29 import java.lang.annotation.RetentionPolicy;
 30 import java.lang.annotation.Target;
 31 
 32 import org.graalvm.compiler.api.directives.GraalDirectives;
 33 import org.graalvm.compiler.core.test.GraalCompilerTest;
 34 import org.graalvm.compiler.nodes.ParameterNode;
 35 import org.graalvm.compiler.nodes.StructuredGraph;
 36 import org.graalvm.compiler.phases.OptimisticOptimizations;
 37 import org.junit.Assert;
 38 import org.junit.Test;
 39 
 40 /**
 41  * Tests for {@link GraalDirectives#blackhole}.
 42  *
 43  * There are two snippets for each kind:
 44  * <ul>
 45  * <li>blackhole&lt;Kind&gt;Snippet verifies that dead code elimination is prevented by the
 46  * blackhole directive.
 47  * <li>&lt;kind&gt;Snippet verifies that dead code elimination does happen if the blackhole
 48  * directive is not there.
 49  * </ul>
 50  *
 51  */
 52 public class BlackholeDirectiveTest extends GraalCompilerTest {
 53 
 54     @Retention(RetentionPolicy.RUNTIME)
 55     @Target(ElementType.METHOD)
 56     private @interface BlackholeSnippet {
 57         boolean expectParameterUsage();
 58     }
 59 
 60     @BlackholeSnippet(expectParameterUsage = false)
 61     public static int booleanSnippet(int arg) {
 62         boolean b = arg > 3;
 63         if (b) {
 64             return 1;
 65         } else {
 66             return 1;
 67         }
 68     }
 69 
 70     @BlackholeSnippet(expectParameterUsage = true)
 71     public static int blackholeBooleanSnippet(int arg) {
 72         boolean b = arg > 3;
 73         GraalDirectives.blackhole(b);
 74         if (b) {
 75             return 1;
 76         } else {
 77             return 1;
 78         }
 79     }
 80 
 81     @Test
 82     public void testBoolean() {
 83         test("booleanSnippet", 5);
 84         test("blackholeBooleanSnippet", 5);
 85     }
 86 
 87     @BlackholeSnippet(expectParameterUsage = false)
 88     public static int intSnippet(int arg) {
 89         int x = 42 + arg;
 90         return x - arg;
 91     }
 92 
 93     @BlackholeSnippet(expectParameterUsage = true)
 94     public static int blackholeIntSnippet(int arg) {
 95         int x = 42 + arg;
 96         GraalDirectives.blackhole(x);
 97         return x - arg;
 98     }
 99 
100     @Test
101     public void testInt() {
102         test("intSnippet", 17);
103         test("blackholeIntSnippet", 17);
104     }
105 
106     private static class Dummy {
107         private int x = 42;
108     }
109 
110     @BlackholeSnippet(expectParameterUsage = false)
111     public static int objectSnippet(int arg) {
112         Dummy obj = new Dummy();
113         int ret = obj.x;
114         obj.x = arg;
115         return ret;
116     }
117 
118     @BlackholeSnippet(expectParameterUsage = true)
119     public static int blackholeObjectSnippet(int arg) {
120         Dummy obj = new Dummy();
121         int ret = obj.x;
122         obj.x = arg;
123         GraalDirectives.blackhole(obj);
124         return ret;
125     }
126 
127     @Test
128     public void testObject() {
129         test("objectSnippet", 37);
130         test("blackholeObjectSnippet", 37);
131     }
132 
133     @Override
134     protected OptimisticOptimizations getOptimisticOptimizations() {
135         return OptimisticOptimizations.ALL.remove(OptimisticOptimizations.Optimization.RemoveNeverExecutedCode);
136     }
137 
138     @Override
139     protected void checkLowTierGraph(StructuredGraph graph) {
140         BlackholeSnippet snippet = graph.method().getAnnotation(BlackholeSnippet.class);
141         ParameterNode arg = graph.getParameter(0);
142         if (snippet.expectParameterUsage()) {
143             Assert.assertNotNull("couldn't find ParameterNode(0)", arg);
144             Assert.assertFalse("expected usages of " + arg, arg.hasNoUsages());
145         } else {
146             Assert.assertTrue("expected no usages of ParameterNode", arg == null || arg.hasNoUsages());
147         }
148     }
149 }
