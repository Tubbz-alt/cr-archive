25 package org.graalvm.compiler.core.test;
26 
27 import java.util.Arrays;
28 import java.util.HashSet;
29 import java.util.Set;
30 
31 import org.graalvm.compiler.core.common.type.ObjectStamp;
32 import org.graalvm.compiler.nodes.NodeView;
33 import org.graalvm.compiler.nodes.StructuredGraph;
34 import org.graalvm.compiler.nodes.java.MethodCallTargetNode;
35 import org.graalvm.compiler.nodes.spi.CoreProviders;
36 import org.graalvm.compiler.phases.VerifyPhase;
37 import org.graalvm.compiler.serviceprovider.BufferUtil;
38 
39 import jdk.vm.ci.meta.ResolvedJavaMethod;
40 import jdk.vm.ci.meta.ResolvedJavaType;
41 
42 /**
43  * See {@link BufferUtil}.
44  */
45 public class VerifyBufferUsage extends VerifyPhase<CoreProviders> {
46 
47     private final Set<String> bufferTypes = new HashSet<>(Arrays.asList(
48                     "Ljava/nio/Buffer;",
49                     "Ljava/nio/ByteBuffer;",
50                     "Ljava/nio/ShortBuffer;",
51                     "Ljava/nio/CharBuffer;",
52                     "Ljava/nio/IntBuffer;",
53                     "Ljava/nio/LongBuffer;",
54                     "Ljava/nio/FloatBuffer;",
55                     "Ljava/nio/DoubleBuffer;",
56                     "Ljava/nio/MappedByteBuffer;"));
57 
58     private final Set<String> bufferMethods = new HashSet<>(Arrays.asList(
59                     "position",
60                     "limit",
61                     "mark",
62                     "reset",
63                     "clear",
64                     "flip",
65                     "rewind"));
66 
67     @Override
68     protected void verify(StructuredGraph graph, CoreProviders context) {
69         ResolvedJavaMethod caller = graph.method();
70         for (MethodCallTargetNode t : graph.getNodes(MethodCallTargetNode.TYPE)) {
71             ResolvedJavaMethod callee = t.targetMethod();
72             String calleeClassName = callee.getDeclaringClass().getName();
73             String calleeName = callee.getName();
74             if (bufferTypes.contains(calleeClassName) &&
75                             bufferMethods.contains(calleeName) &&
76                             !callee.getSignature().getReturnKind().isPrimitive()) {
77                 StackTraceElement e = caller.asStackTraceElement(t.invoke().bci());
78                 ResolvedJavaType receiverType = ((ObjectStamp) t.arguments().get(0).stamp(NodeView.DEFAULT)).type();
79                 if (!receiverType.getName().equals("Ljava/nio/Buffer;")) {
80                     throw new VerificationError(
81                                     "%s: Cast receiver of type %s to java.nio.Buffer for call to %s to avoid problems with co-variant overloads added by https://bugs.openjdk.java.net/browse/JDK-4774077",
82                                     e, receiverType.toJavaName(),
83                                     callee.format("%H.%n(%p)"));
84                 }
85             }
86         }
87     }
88 }
