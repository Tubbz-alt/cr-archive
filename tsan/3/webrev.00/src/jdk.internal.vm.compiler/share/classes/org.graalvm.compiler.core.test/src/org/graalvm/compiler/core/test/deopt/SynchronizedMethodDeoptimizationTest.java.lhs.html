33 /**
34  * In the following tests, we try to deoptimize out of synchronized methods.
35  */
36 public class SynchronizedMethodDeoptimizationTest extends GraalCompilerTest {
37 
38     public static final TestClassObject testObject = null;
39 
40     public static synchronized Object testMethodSynchronized(Object o) {
41         if (o == null) {
42             // this branch will always deoptimize
43             return testObject.x;
44         }
45         return o;
46     }
47 
48     @Test
49     public void test1() {
50         // https://bugs.openjdk.java.net/browse/JDK-8182755
51         Assume.assumeTrue(Java8OrEarlier);
52 
53         test("testMethodSynchronized", "test");
54         test("testMethodSynchronized", (Object) null);
55     }
56 }
