 43 import jdk.vm.ci.meta.ResolvedJavaMethod;
 44 
 45 public class LockEliminationTest extends GraalCompilerTest {
 46 
 47     static class A {
 48 
 49         int value;
 50 
 51         public synchronized int getValue() {
 52             return value;
 53         }
 54     }
 55 
 56     static int field1;
 57     static int field2;
 58 
 59     public static void testSynchronizedSnippet(A x, A y) {
 60         synchronized (x) {
 61             field1 = x.value;
 62         }
 63         synchronized (x) {
 64             field2 = y.value;
 65         }
 66     }
 67 
 68     @Test
 69     public void testLock() {
 70         test("testSynchronizedSnippet", new A(), new A());
 71 
 72         StructuredGraph graph = getGraph("testSynchronizedSnippet", false);
 73         new CanonicalizerPhase().apply(graph, new PhaseContext(getProviders()));
 74         new LockEliminationPhase().apply(graph);
 75         assertDeepEquals(1, graph.getNodes().filter(RawMonitorEnterNode.class).count());
 76         assertDeepEquals(1, graph.getNodes().filter(MonitorExitNode.class).count());
 77     }
 78 
 79     public static void testSynchronizedMethodSnippet(A x) {
 80         int value1 = x.getValue();
 81         int value2 = x.getValue();
 82         field1 = value1;
 83         field2 = value2;
 84     }
 85 
 86     @Test
 87     public void testSynchronizedMethod() {
 88         test("testSynchronizedMethodSnippet", new A());
 89 
 90         StructuredGraph graph = getGraph("testSynchronizedMethodSnippet", false);
 91         new CanonicalizerPhase().apply(graph, new PhaseContext(getProviders()));
 92         new LockEliminationPhase().apply(graph);
 93         assertDeepEquals(1, graph.getNodes().filter(RawMonitorEnterNode.class).count());
 94         assertDeepEquals(1, graph.getNodes().filter(MonitorExitNode.class).count());
 95     }
 96 
 97     public void testUnrolledSyncSnippet(Object a) {
 98         for (int i = 0; i < 3; i++) {
 99             synchronized (a) {
100 
101             }
102         }
103     }
104 
105     @Test
106     public void testUnrolledSync() {
107         StructuredGraph graph = getGraph("testUnrolledSyncSnippet", false);
108         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
109         canonicalizer.apply(graph, new PhaseContext(getProviders()));
110         HighTierContext context = getDefaultHighTierContext();
111         new LoopFullUnrollPhase(canonicalizer, new DefaultLoopPolicies()).apply(graph, context);
112         new LockEliminationPhase().apply(graph);
113         assertDeepEquals(1, graph.getNodes().filter(RawMonitorEnterNode.class).count());
114         assertDeepEquals(1, graph.getNodes().filter(MonitorExitNode.class).count());
115     }
116 
117     private StructuredGraph getGraph(String snippet, boolean doEscapeAnalysis) {
118         ResolvedJavaMethod method = getResolvedJavaMethod(snippet);
119         StructuredGraph graph = parseEager(method, AllowAssumptions.YES);
120         HighTierContext context = getDefaultHighTierContext();
121         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
122         canonicalizer.apply(graph, context);
123         createInliningPhase().apply(graph, context);
124         new CanonicalizerPhase().apply(graph, context);
125         new DeadCodeEliminationPhase().apply(graph);
126         if (doEscapeAnalysis) {
127             new PartialEscapePhase(true, canonicalizer, graph.getOptions()).apply(graph, context);
128         }
129         new LoweringPhase(new CanonicalizerPhase(), LoweringTool.StandardLoweringStage.HIGH_TIER).apply(graph, context);
130         return graph;
131     }
132 
133     public void testEscapeAnalysisSnippet(A a) {
134         A newA = new A();
135         synchronized (newA) {
136             synchronized (a) {
137                 field1 = a.value;
138             }
139         }
140         /*
141          * Escape analysis removes the synchronization on newA. But lock elimination still must not
142          * combine the two synchronizations on the parameter a because they have a different lock
143          * depth.
144          */
145         synchronized (a) {
146             field2 = a.value;
147         }
148         /*
149          * Lock elimination can combine these synchronizations, since they are both on parameter a
150          * with the same lock depth.
151          */
152         synchronized (a) {
153             field1 = a.value;
154         }
155     }
156 
157     @Test
158     public void testEscapeAnalysis() {
159         StructuredGraph graph = getGraph("testEscapeAnalysisSnippet", true);
160 
161         assertDeepEquals(3, graph.getNodes().filter(RawMonitorEnterNode.class).count());
162         assertDeepEquals(3, graph.getNodes().filter(MonitorExitNode.class).count());
163 
164         new LockEliminationPhase().apply(graph);
165 
166         assertDeepEquals(2, graph.getNodes().filter(RawMonitorEnterNode.class).count());
167         assertDeepEquals(2, graph.getNodes().filter(MonitorExitNode.class).count());
168     }
