 31 import org.graalvm.compiler.phases.Phase;

 32 import org.graalvm.compiler.phases.tiers.Suites;
 33 import org.junit.Assume;
 34 import org.junit.Test;
 35 
 36 public class EnumSwitchTest extends GraalCompilerTest {
 37 
 38     enum E {
 39         E0,
 40         E1,
 41         E2,
 42         E3,
 43         E4,
 44         E5,
 45         E6,
 46         E7,
 47         E8,
 48         E9,
 49         E10,
 50         E11,
 51         E12,
 52         E13,
 53         E14,
 54         E15,
 55         E16,
 56         E17,
 57         E18,
 58         E19,
 59         E20
 60     }
 61 
 62     public int test1Snippet(E e) {
 63         switch (e) {
 64             case E0:
 65                 return 0;
 66             case E1:
 67                 return 1;
 68             case E2:
 69                 return 2;
 70             case E3:
 71                 return 3;
 72             case E4:
 73                 return 4;
 74             case E5:
 75                 return 5;
 76             case E6:
 77                 return 6;
 78             case E7:
 79                 return 7;
 80             case E8:
 81                 return 8;
 82             case E9:
 83                 return 9;
 84             case E10:
 85                 return 10;
 86             case E11:
 87                 return 11;
 88             case E12:
 89                 return 12;
 90             case E13:
 91                 return 13;
 92             case E14:
 93                 return 14;
 94             case E15:
 95                 return 15;
 96             case E16:
 97                 return 16;
 98             case E17:
 99                 return 17;
100             case E18:
101                 return 18;
102             case E19:
103                 return 19;
104             case E20:
105                 return 20;
106             default:
107                 return -1;
108         }
109     }
110 
111     @Test
112     public void test1() {
113         for (E e : E.values()) {
114             test("test1Snippet", e);
115         }
116         test("test1Snippet", new Object[]{null});
117     }
118 
119     public int test2Snippet(E e) {
120         switch (e) {
121             case E5:
122             case E19:
123             case E20:
124                 return 1;
125             case E8:
126             case E9:
127             case E10:
128                 return 2;
129         }
130         return -1;
131     }
132 
133     @Test
134     public void test2() {
135         for (E e : E.values()) {
136             test("test2Snippet", e);
137         }
138         test("test2Snippet", new Object[]{null});
139     }
140 
141     @Override
142     protected Suites createSuites(OptionValues options) {
143         Suites ret = super.createSuites(options);
144         ret.getHighTier().prependPhase(new Phase() {
145             @Override
146             protected void run(StructuredGraph graph) {
147                 Assume.assumeTrue(graph.getNodes().filter(LoadIndexedNode.class).first().array().isConstant());
148                 /* Array load from the enum switch map. */
149                 assertTrue(graph.getNodes().filter(LoadIndexedNode.class).count() == 1);
150                 /* The actual switch. */
151                 assertTrue(graph.getNodes().filter(IntegerSwitchNode.class).count() == 1);
152             }
153 
154             @Override
155             protected CharSequence getName() {
156                 return "CheckGraphPhase";
157             }
158         });
159         ret.getHighTier().addBeforeLast(new Phase() {
160             @Override
161             protected void run(StructuredGraph graph) {
162                 /* Re-writing of the switch cases eliminates the array load. */
163                 assertTrue(graph.getNodes().filter(LoadIndexedNode.class).count() == 0);
164                 /* The switch is still there. */
165                 assertTrue(graph.getNodes().filter(IntegerSwitchNode.class).count() == 1);
166             }
167 
168             @Override
169             protected CharSequence getName() {
170                 return "CheckGraphPhase";
171             }
172         });
173         return ret;
174     }
175 }
