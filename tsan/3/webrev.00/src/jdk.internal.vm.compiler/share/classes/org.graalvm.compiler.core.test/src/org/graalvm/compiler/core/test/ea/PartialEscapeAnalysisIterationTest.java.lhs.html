 25 package org.graalvm.compiler.core.test.ea;
 26 
 27 import java.lang.ref.WeakReference;
 28 import java.util.List;
 29 
 30 import org.graalvm.compiler.nodes.extended.BoxNode;
 31 import org.graalvm.compiler.nodes.extended.UnboxNode;
 32 import org.graalvm.compiler.nodes.java.StoreFieldNode;
 33 import org.graalvm.compiler.nodes.virtual.CommitAllocationNode;
 34 import org.graalvm.compiler.phases.common.CanonicalizerPhase;
 35 import org.graalvm.compiler.virtual.phases.ea.PartialEscapePhase;
 36 import org.junit.Assert;
 37 import org.junit.Test;
 38 
 39 public class PartialEscapeAnalysisIterationTest extends EATestBase {
 40 
 41     // remember boxing nodes from before PEA
 42     private List<BoxNode> boxNodes;
 43 
 44     @Override
 45     protected void canonicalizeGraph() {
 46         super.canonicalizeGraph();
 47         boxNodes = graph.getNodes().filter(BoxNode.class).snapshot();
 48     }
 49 
 50     private static final class AllocatedObject {
 51         private int value;
 52 
 53         AllocatedObject(int value) {
 54             this.value = value;
 55         }
 56 
 57         AllocatedObject() {
 58             // empty
 59         }
 60     }
 61 
 62     public static volatile Object obj1;
 63     public static volatile Double object1 = (double) 123;
 64     public static volatile AllocatedObject object2 = new AllocatedObject(123);
 65 
 66     public static String moveIntoBranchBox(int id) {
 67         Double box = object1 + 1;
 68         if (id == 0) {
 69             obj1 = new WeakReference<>(box);
 70         }
 71         return "value";
 72     }
 73 
 74     public static String moveIntoBranch(int id) {
 75         AllocatedObject box = new AllocatedObject(object2.value + 1);
 76         if (id == 0) {
 77             obj1 = new WeakReference<>(box);
 78         }
 79         return "value";
 80     }
 81 
 82     @Test
 83     public void testJMHBlackholePattern() {
 84         /*
 85          * The overall number of allocations in this methods does not change during PEA, but the
 86          * effects still need to be applied since they move the allocation between blocks.
 87          */
 88 
 89         // test with a boxing object
 90         prepareGraph("moveIntoBranchBox", false);
 91         Assert.assertEquals(1, graph.getNodes().filter(UnboxNode.class).count());
 92         Assert.assertEquals(1, graph.getNodes().filter(BoxNode.class).count());
 93         // the boxing needs to be moved into the branch
 94         Assert.assertTrue(graph.getNodes().filter(BoxNode.class).first().next() instanceof StoreFieldNode);
 95 
 96         // test with a normal object
 97         prepareGraph("moveIntoBranch", false);
 98         Assert.assertEquals(1, graph.getNodes().filter(CommitAllocationNode.class).count());
 99         // the allocation needs to be moved into the branch
100         Assert.assertTrue(graph.getNodes().filter(CommitAllocationNode.class).first().next() instanceof StoreFieldNode);
101     }
102 
103     public static String noLoopIterationBox(int id) {
104         Double box = object1 + 1;
105         for (int i = 0; i < 100; i++) {
106             if (id == i) {
107                 obj1 = new WeakReference<>(box);
108             }
109         }
110         return "value";
111     }
112 
113     public static String noLoopIteration(int id) {
114         AllocatedObject box = new AllocatedObject(object2.value + 1);
115         for (int i = 0; i < 100; i++) {
116             if (id == i) {
117                 obj1 = new WeakReference<>(box);
118             }
119         }
120         return "value";
121     }
122 
123     public static String noLoopIterationEmpty(int id) {
124         AllocatedObject box = new AllocatedObject();
125         for (int i = 0; i < 100; i++) {
126             if (id == i) {
127                 obj1 = new WeakReference<>(box);
128             }
129         }
130         return "value";
131     }
132 
133     @Test
134     public void testNoLoopIteration() {
135         /*
136          * PEA should not apply any effects on this method, since it cannot move the allocation into
137          * the branch anyway (it needs to stay outside the loop).
138          */
139 
140         // test with a boxing object
141         prepareGraph("noLoopIterationBox", true);
142         Assert.assertEquals(1, boxNodes.size());
143         Assert.assertTrue(boxNodes.get(0).isAlive());
144 
145         // test with a normal object (needs one iteration to replace NewInstance with
146         // CommitAllocation)
147         for (String name : new String[]{"noLoopIterationEmpty", "noLoopIteration"}) {
148             prepareGraph(name, false);
149             List<CommitAllocationNode> allocations = graph.getNodes().filter(CommitAllocationNode.class).snapshot();
150             new PartialEscapePhase(true, false, new CanonicalizerPhase(), null, graph.getOptions()).apply(graph, context);
151             Assert.assertEquals(1, allocations.size());
152             Assert.assertTrue(allocations.get(0).isAlive());
153         }
154     }
155 }
