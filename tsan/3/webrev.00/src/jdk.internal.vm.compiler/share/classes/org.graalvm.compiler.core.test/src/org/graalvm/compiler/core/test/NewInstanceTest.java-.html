 71     private StructuredGraph parseAndProcess(Class<?> cl) {
 72         Constructor<?>[] constructors = cl.getConstructors();
 73         Assert.assertTrue(constructors.length == 1);
 74         final ResolvedJavaMethod javaMethod = getMetaAccess().lookupJavaMethod(constructors[0]);
 75         OptionValues options = getInitialOptions();
 76         StructuredGraph graph = new StructuredGraph.Builder(options, getDebugContext(options, null, javaMethod), AllowAssumptions.YES).method(javaMethod).build();
 77 
 78         GraphBuilderConfiguration conf = GraphBuilderConfiguration.getSnippetDefault(getDefaultGraphBuilderPlugins()).withUnresolvedIsError(false);
 79         new GraphBuilderPhase.Instance(getMetaAccess(), getProviders().getStampProvider(), getProviders().getConstantReflection(), getProviders().getConstantFieldProvider(), conf,
 80                         OptimisticOptimizations.ALL, null).apply(graph);
 81         return graph;
 82     }
 83 
 84     private void checkGraph(Class<?> cl, int newInstanceNodeCount, int deoptimizeNodeCount) {
 85         StructuredGraph graph = parseAndProcess(cl);
 86         Assert.assertEquals(newInstanceNodeCount, graph.getNodes().filter(NewInstanceNode.class).count());
 87         Assert.assertEquals(deoptimizeNodeCount, graph.getNodes().filter(DeoptimizeNode.class).count());
 88     }

 90     /**
 91      * Use a custom class loader to generate classes, substitute class names at the allocation site.
 92      */
 93     @Test
 94     public void test1() throws ClassNotFoundException {
 95         checkGraph(Template.class, 1, 0);
 96         ClassTemplateLoader loader1 = new ClassTemplateLoader("CCCC", "AAAA");
 97         checkGraph(loader1.findClass("Template"), 0, 1);
 98         ClassTemplateLoader loader2 = new ClassTemplateLoader("CCCC", "IIII");
 99         checkGraph(loader2.findClass("Template"), 0, 1);
100     }
101 
102     private static class ClassTemplateLoader extends ClassLoader {
103         private final String find;
104         private final String replace;
105 
106         ClassTemplateLoader(String find, String replace) {
107             assert find.length() == replace.length();
108             this.find = find;
109             this.replace = replace;
110         }
111 
112         @Override
113         protected Class<?> findClass(final String name) throws ClassNotFoundException {
114             // copy classfile to byte array
115             byte[] classData = null;
116             try {
117                 InputStream is = NewInstanceTest.class.getResourceAsStream("NewInstanceTest$" + name + ".class");
118                 assert is != null;
119                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
120 
121                 byte[] buf = new byte[1024];
122                 int size;
123                 while ((size = is.read(buf, 0, buf.length)) != -1) {
124                     baos.write(buf, 0, size);
125                 }
126                 baos.flush();
127                 classData = baos.toByteArray();
128             } catch (IOException e) {
129                 Assert.fail("can't access class: " + name);
130             }
131 
132             // replace all occurrences of the template in classfile
133             int index = -1;
134             while ((index = indexOfTemplate(classData, index + 1, find)) != -1) {
135                 replaceTemplate(classData, index, replace);
136             }
137 
138             Class<?> c = defineClass(null, classData, 0, classData.length);
139             return c;
140         }
141 
142         private static int indexOfTemplate(byte[] b, int index, String find) {
143             for (int i = index; i < b.length; i++) {
144                 boolean match = true;
145                 for (int j = i; j < i + find.length(); j++) {
146                     if (b[j] != (byte) find.charAt(j - i)) {
147                         match = false;
148                         break;
149                     }
150                 }
151                 if (match) {
152                     return i;
153                 }
154             }
155             return -1;
156         }
157 
158         private static void replaceTemplate(byte[] b, int index, String replace) {
159             for (int i = index; i < index + replace.length(); i++) {
160                 b[i] = (byte) replace.charAt(i - index);
161             }
162         }
163     }
