25 package org.graalvm.compiler.core.test;
26 
27 import org.graalvm.compiler.api.directives.GraalDirectives;
28 import org.graalvm.compiler.nodes.StructuredGraph;
29 import org.graalvm.compiler.nodes.calc.RightShiftNode;
30 import org.graalvm.compiler.nodes.calc.UnsignedRightShiftNode;
31 import org.junit.Test;
32 
33 public class IntegerDivPowerOf2Test extends GraalCompilerTest {
34 
35     public static int positiveDivByPowerOf2(boolean flag) {
36         int val = flag ? 1 : 10;
37         GraalDirectives.blackhole(val);
38         return val / 8;
39     }
40 
41     @Test
42     public void testPositiveDivByPowerOf2() {
43         StructuredGraph graph = parseForCompile(getResolvedJavaMethod("positiveDivByPowerOf2"));
44         // We expect no rounding is needed
45         assertTrue(countShiftNode(graph) == 1);
46     }
47 
48     private static int countShiftNode(StructuredGraph graph) {
49         return graph.getNodes().filter(node -> node instanceof RightShiftNode || node instanceof UnsignedRightShiftNode).count();
50     }
51 
52     public static int unknownDivByPowerOf2(boolean flag) {
53         int val = flag ? 0x800000F0 : 0x20;
54         GraalDirectives.blackhole(val);
55         return val / 8;
56     }
57 
58     @Test
59     public void testUnknownDivByPowerOf2() {
60         StructuredGraph graph = parseForCompile(getResolvedJavaMethod("unknownDivByPowerOf2"));
61         // We expect no rounding is needed
62         assertTrue(graph.getNodes().filter(RightShiftNode.class).count() <= 1);
63     }
64 
65 }
