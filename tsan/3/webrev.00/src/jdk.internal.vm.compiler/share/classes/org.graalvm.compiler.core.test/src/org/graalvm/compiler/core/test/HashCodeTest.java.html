 25 package org.graalvm.compiler.core.test;
 26 
 27 import org.graalvm.compiler.core.phases.HighTier;
 28 import org.graalvm.compiler.core.phases.MidTier;
 29 import org.graalvm.compiler.nodes.InvokeNode;
 30 import org.graalvm.compiler.nodes.StructuredGraph;
 31 import org.graalvm.compiler.options.OptionValues;
 32 import org.graalvm.compiler.phases.OptimisticOptimizations;
 33 import org.graalvm.compiler.phases.tiers.MidTierContext;
 34 import org.junit.Assert;
 35 import org.junit.Test;
 36 
 37 public class HashCodeTest extends GraalCompilerTest {
 38 
 39     static class OverrideHashCode {
 40         @Override
 41         public int hashCode() {
 42             return 42;
 43         }
 44     }
 45 
 46     static final class DontOverrideHashCode {
 47     }
 48 
 49     public static final Object NonOverridingConstant = new Object();
 50     public static final Object OverridingConstant = new OverrideHashCode();
 51 
 52     private static void initialize(Class<?> c) {
 53         try {
 54             Class.forName(c.getName(), true, c.getClassLoader());
 55         } catch (ClassNotFoundException e) {
 56             throw new AssertionError(e);
 57         }
 58     }
 59 
 60     public static final int hashCodeSnippet01(Object o) {
 61         return o.hashCode();
 62     }
 63 
 64     public static final int systemIdentityHashCodeSnippet01(Object o) {
 65         return System.identityHashCode(o);
 66     }
 67 
 68     public static final int hashCodeFoldSnippet01() {
 69         return NonOverridingConstant.hashCode();
 70     }
 71 
 72     public static final int identityHashCodeFoldSnippet01() {
 73         return System.identityHashCode(NonOverridingConstant);
 74     }
 75 
 76     public static final int identityHashCodeFoldOverridingSnippet01() {
 77         return System.identityHashCode(OverridingConstant);
 78     }
 79 
 80     public static final int dontOverrideHashCodeFinalClass(DontOverrideHashCode o) {
 81         return o.hashCode();
 82     }
 83 
 84     @Test
 85     public void test01() {
 86         test("hashCodeSnippet01", new Object());
 87     }
 88 
 89     @Test
 90     public void test02() {
 91         test("systemIdentityHashCodeSnippet01", new Object());
 92     }
 93 
 94     @Test
 95     public void test03() {
 96         StructuredGraph g = buildGraphAfterMidTier("hashCodeFoldSnippet01");
 97         Assert.assertEquals(0, g.getNodes().filter(InvokeNode.class).count());
 98     }
 99 
100     @Test
101     public void test04() {
102         StructuredGraph g = buildGraphAfterMidTier("identityHashCodeFoldSnippet01");
103         Assert.assertEquals(0, g.getNodes().filter(InvokeNode.class).count());
104     }
105 
106     @Test
107     public void test05() {
108         StructuredGraph g = buildGraphAfterMidTier("identityHashCodeFoldOverridingSnippet01");
109         Assert.assertEquals(0, g.getNodes().filter(InvokeNode.class).count());
110     }
111 
112     @Test
113     public void test06() {
114         initialize(DontOverrideHashCode.class);
115         StructuredGraph g = buildGraphAfterMidTier("dontOverrideHashCodeFinalClass");
116         Assert.assertEquals(0, g.getNodes().filter(InvokeNode.class).count());
117     }
118 
119     @SuppressWarnings("try")
120     private StructuredGraph buildGraphAfterMidTier(String name) {
121         StructuredGraph g = parseForCompile(getResolvedJavaMethod(name));
122         OptionValues options = getInitialOptions();
123         new HighTier(options).apply(g, getDefaultHighTierContext());
124         new MidTier(options).apply(g, new MidTierContext(getProviders(), getTargetProvider(), OptimisticOptimizations.ALL, g.getProfilingInfo()));
125         return g;
126     }
127 
128 }
