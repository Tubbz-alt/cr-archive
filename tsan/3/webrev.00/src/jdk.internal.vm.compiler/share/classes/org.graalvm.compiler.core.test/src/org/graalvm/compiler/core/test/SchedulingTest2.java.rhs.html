 55 public class SchedulingTest2 extends GraphScheduleTest {
 56 
 57     public static int testSnippet() {
 58         return test() + 2;
 59     }
 60 
 61     public static int test() {
 62         return 40;
 63     }
 64 
 65     @Test
 66     public void testValueProxyInputs() {
 67         StructuredGraph graph = parseEager("testSnippet", AllowAssumptions.YES);
 68         DebugContext debug = graph.getDebug();
 69         ReturnNode returnNode = graph.getNodes(ReturnNode.TYPE).first();
 70         BeginNode beginNode = graph.add(new BeginNode());
 71         returnNode.replaceAtPredecessor(beginNode);
 72         beginNode.setNext(returnNode);
 73         debug.dump(DebugContext.BASIC_LEVEL, graph, "Graph");
 74         SchedulePhase schedulePhase = new SchedulePhase(SchedulingStrategy.EARLIEST_WITH_GUARD_ORDER);
 75         schedulePhase.apply(graph);
 76         ScheduleResult schedule = graph.getLastSchedule();
 77         BlockMap<List<Node>> blockToNodesMap = schedule.getBlockToNodesMap();
 78         NodeMap<Block> nodeToBlock = schedule.getNodeToBlockMap();
 79         assertDeepEquals(2, schedule.getCFG().getBlocks().length);
 80         for (BinaryArithmeticNode<?> node : graph.getNodes().filter(BinaryArithmeticNode.class)) {
 81             if (node instanceof AddNode) {
 82                 assertTrue(node.toString() + " expected: " + nodeToBlock.get(beginNode) + " but was: " + nodeToBlock.get(node), nodeToBlock.get(node) != nodeToBlock.get(beginNode));
 83             }
 84         }
 85 
 86         for (FrameState fs : graph.getNodes(FrameState.TYPE)) {
 87             Block block = nodeToBlock.get(fs);
 88             assertTrue(fs.toString(), block == schedule.getCFG().getStartBlock());
 89             for (Node usage : fs.usages()) {
 90                 if (usage instanceof StateSplit && ((StateSplit) usage).stateAfter() == fs) {
 91                     assertTrue(usage.toString(), nodeToBlock.get(usage) == block);
 92                     if (usage != block.getBeginNode()) {
 93                         List<Node> map = blockToNodesMap.get(block);
 94                         assertTrue(map.indexOf(fs) + " < " + map.indexOf(usage), map.indexOf(fs) < map.indexOf(usage));
 95                     }
 96                 }
 97             }
 98         }
 99 
100         CoreProviders context = getProviders();
101         new LoweringPhase(createCanonicalizerPhase(), LoweringTool.StandardLoweringStage.HIGH_TIER).apply(graph, context);
102         new LoweringPhase(createCanonicalizerPhase(), LoweringTool.StandardLoweringStage.MID_TIER).apply(graph, context);
103         MidTierContext midContext = new MidTierContext(getProviders(), getTargetProvider(), OptimisticOptimizations.ALL, graph.getProfilingInfo());
104 
105         new GuardLoweringPhase().apply(graph, midContext);
106         FrameStateAssignmentPhase phase = new FrameStateAssignmentPhase();
107         phase.apply(graph);
108 
109         schedulePhase.apply(graph);
110         schedule = graph.getLastSchedule();
111         blockToNodesMap = schedule.getBlockToNodesMap();
112         nodeToBlock = schedule.getNodeToBlockMap();
113         for (FrameState fs : graph.getNodes(FrameState.TYPE)) {
114             Block block = nodeToBlock.get(fs);
115             assertTrue(fs.toString(), block == schedule.getCFG().getStartBlock());
116             for (Node usage : fs.usages()) {
117                 if ((usage instanceof StateSplit && ((StateSplit) usage).stateAfter() == fs) || (usage instanceof DeoptDuring && ((DeoptDuring) usage).stateDuring() == fs)) {
118                     assertTrue(usage.toString(), nodeToBlock.get(usage) == block);
119                     if (usage != block.getBeginNode()) {
120                         List<Node> map = blockToNodesMap.get(block);
121                         assertTrue(map.indexOf(fs) + " < " + map.indexOf(usage), map.indexOf(fs) < map.indexOf(usage));
122                     }
123                 }
124             }
125         }
126     }
127 }
