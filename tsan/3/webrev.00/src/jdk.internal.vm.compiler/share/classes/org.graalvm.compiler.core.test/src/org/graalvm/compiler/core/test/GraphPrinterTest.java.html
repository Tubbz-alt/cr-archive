25 package org.graalvm.compiler.core.test;
26 
27 import org.graalvm.compiler.printer.CanonicalStringGraphPrinter;
28 import org.junit.Test;
29 
30 import jdk.vm.ci.meta.JavaConstant;
31 
32 /**
33  * Tests related to graph printing.
34  */
35 public class GraphPrinterTest extends GraalCompilerTest {
36 
37     /**
38      * Tests that a self-recursive object does not cause stack overflow when formatted as a string.
39      */
40     @Test
41     public void testGraphPrinterDoesNotStackOverflow() {
42         CanonicalStringGraphPrinter printer = new CanonicalStringGraphPrinter(getSnippetReflection());
43         Object[] topArray = {null};
44         Object[] parent = topArray;
45         Object[] lastArray = null;
46         for (int i = 0; i < 5; i++) {
47             lastArray = new Object[1];
48             parent[0] = lastArray;
49             parent = lastArray;
50         }
51         lastArray[0] = topArray;
52         JavaConstant constant = getSnippetReflection().forObject(topArray);
53         printer.format(constant);
54     }
55 }
