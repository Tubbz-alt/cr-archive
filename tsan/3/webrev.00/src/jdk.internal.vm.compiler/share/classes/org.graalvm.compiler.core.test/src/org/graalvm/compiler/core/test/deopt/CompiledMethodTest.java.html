 41 public class CompiledMethodTest extends GraalCompilerTest {
 42 
 43     public static Object testMethod(Object arg1, Object arg2, Object arg3) {
 44         String res = arg1 + " " + arg2 + " " + arg3;
 45         return GraalDirectives.inCompiledCode() ? res : "interpreter";
 46     }
 47 
 48     Object f1;
 49 
 50     public Object testMethodVirtual(Object arg1, Object arg2, Object arg3) {
 51         return f1 + " " + arg1 + " " + arg2 + " " + arg3;
 52     }
 53 
 54     /**
 55      * Usages of the constant {@code " "} are replaced with the constant {@code "-"} and it is
 56      * verified that executing the compiled code produces a result that the preserves the node
 57      * replacement unless deoptimization occurs (e.g., due to -Xcomp causing profiles to be
 58      * missing).
 59      */
 60     @Test
 61     public void test1() {
 62         final ResolvedJavaMethod javaMethod = getResolvedJavaMethod("testMethod");
 63         final StructuredGraph graph = parseEager(javaMethod, AllowAssumptions.NO);
 64         createCanonicalizerPhase().apply(graph, getProviders());
 65         new DeadCodeEliminationPhase().apply(graph);
 66 
 67         for (ConstantNode node : ConstantNode.getConstantNodes(graph)) {
 68             if (node.getStackKind() == JavaKind.Object && " ".equals(getSnippetReflection().asObject(String.class, node.asJavaConstant()))) {
 69                 node.replace(graph, ConstantNode.forConstant(getSnippetReflection().forObject("-"), getMetaAccess(), graph));
 70             }
 71         }
 72 
 73         InstalledCode compiledMethod = getCode(javaMethod, graph, true);
 74         try {
 75             Object result = compiledMethod.executeVarargs("1", "2", "3");
 76             if (!"1-2-3".equals(result)) {
 77                 // Deoptimization probably occurred
 78                 Assert.assertEquals("interpreter", result);
 79             }
 80         } catch (InvalidInstalledCodeException t) {
 81             Assert.fail("method invalidated");
 82         }
 83     }
 84 
 85     @Test
 86     public void test3() {
 87         final ResolvedJavaMethod javaMethod = getResolvedJavaMethod("testMethod");
 88         InstalledCode compiledMethod = getCode(javaMethod);
 89         try {
 90             Object result = compiledMethod.executeVarargs("1", "2", "3");
 91             Assert.assertEquals("1 2 3", result);
 92         } catch (InvalidInstalledCodeException t) {
 93             Assert.fail("method invalidated");
 94         }
 95     }
 96 
 97     @Test
 98     public void test4() {
 99         final ResolvedJavaMethod javaMethod = getResolvedJavaMethod("testMethodVirtual");
100         InstalledCode compiledMethod = getCode(javaMethod);
101         try {
102             f1 = "0";
103             Object result = compiledMethod.executeVarargs(this, "1", "2", "3");
104             Assert.assertEquals("0 1 2 3", result);
105         } catch (InvalidInstalledCodeException t) {
106             Assert.fail("method invalidated");
107         }
108     }
