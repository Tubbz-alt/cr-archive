 25 package org.graalvm.compiler.core.test;
 26 


 27 import org.graalvm.compiler.nodes.ParameterNode;
 28 import org.graalvm.compiler.nodes.ReturnNode;
 29 import org.graalvm.compiler.nodes.StructuredGraph;
 30 import org.graalvm.compiler.nodes.StructuredGraph.AllowAssumptions;
 31 import org.graalvm.compiler.nodes.ValueNode;
 32 import org.graalvm.compiler.nodes.calc.ConditionalNode;
 33 import org.graalvm.compiler.nodes.calc.IntegerTestNode;
 34 import org.junit.Test;

 35 
 36 public class CompareCanonicalizerTest extends GraalCompilerTest {
 37 
 38     private StructuredGraph getCanonicalizedGraph(String name) {
 39         StructuredGraph graph = parseEager(name, AllowAssumptions.YES);
 40         createCanonicalizerPhase().apply(graph, getProviders());
 41         return graph;
 42     }
 43 
 44     private static ValueNode getResult(StructuredGraph graph) {
 45         assertTrue(graph.start().next() instanceof ReturnNode);
 46         ReturnNode ret = (ReturnNode) graph.start().next();
 47         return ret.result();
 48     }
 49 
 50     @Test
 51     public void testCanonicalComparison() {
 52         StructuredGraph referenceGraph = parseEager("referenceCanonicalComparison", AllowAssumptions.NO);
 53         for (int i = 1; i < 4; i++) {
 54             StructuredGraph graph = parseEager("canonicalCompare" + i, AllowAssumptions.NO);
 55             assertEquals(referenceGraph, graph);
 56         }
 57         createCanonicalizerPhase().apply(referenceGraph, getProviders());
 58         for (int i = 1; i < 4; i++) {
 59             StructuredGraph graph = getCanonicalizedGraph("canonicalCompare" + i);
 60             assertEquals(referenceGraph, graph);
 61         }
 62     }
 63 
 64     public static int referenceCanonicalComparison(int a, int b) {
 65         if (a < b) {
 66             return 1;
 67         } else {
 68             return 2;
 69         }
 70     }
 71 
 72     public static int canonicalCompare1(int a, int b) {
 73         if (a >= b) {
 74             return 2;
 75         } else {
 76             return 1;
 77         }
 78     }
 79 
 80     public static int canonicalCompare2(int a, int b) {
 81         if (b > a) {
 82             return 1;
 83         } else {
 84             return 2;
 85         }
 86     }
 87 
 88     public static int canonicalCompare3(int a, int b) {
 89         if (b <= a) {
 90             return 2;
 91         } else {
 92             return 1;
 93         }
 94     }
 95 
 96     @Test
 97     public void testIntegerTest() {
 98         for (int i = 1; i <= 4; i++) {
 99             StructuredGraph graph = getCanonicalizedGraph("integerTest" + i);
100 
101             ReturnNode returnNode = (ReturnNode) graph.start().next();
102             ConditionalNode conditional = (ConditionalNode) returnNode.result();
103             IntegerTestNode test = (IntegerTestNode) conditional.condition();
104             ParameterNode param0 = graph.getParameter(0);
105             ParameterNode param1 = graph.getParameter(1);
106             assertTrue((test.getX() == param0 && test.getY() == param1) || (test.getX() == param1 && test.getY() == param0));
107         }
108     }
109 
110     public static boolean integerTest1(int x, int y) {
111         return (x & y) == 0;
112     }
113 
114     public static boolean integerTest2(long x, long y) {
115         return 0 == (x & y);
116     }
117 
118     public static boolean integerTest3(long x, long y) {
119         int c = 5;
120         return (c - 5) == (x & y);
121     }
122 
123     public static boolean integerTest4(int x, int y) {
124         int c = 10;
125         return (x & y) == (10 - c);
126     }
127 
128     @Test
129     public void testIntegerTestCanonicalization() {
130         ValueNode result = getResult(getCanonicalizedGraph("integerTestCanonicalization1"));
131         assertTrue(result.isConstant() && result.asJavaConstant().asLong() == 1);
132         result = getResult(getCanonicalizedGraph("integerTestCanonicalization2"));
133         assertTrue(result.isConstant() && result.asJavaConstant().asLong() == 1);
134         StructuredGraph graph = getCanonicalizedGraph("integerTestCanonicalization3");
135         assertDeepEquals(1, graph.getNodes(ReturnNode.TYPE).count());
136         assertTrue(graph.getNodes(ReturnNode.TYPE).first().result() instanceof ConditionalNode);
137     }
138 
139     public static int integerTestCanonicalization1(boolean b) {
140         int x = b ? 128 : 256;
141         if ((x & 8) == 0) {
142             return 1;
143         } else {
144             return 2;
145         }
146     }
147 
148     public static int integerTestCanonicalization2(boolean b) {
149         int x = b ? 128 : 256;
150         int y = b ? 32 : 64;
151         if ((x & y) == 0) {
152             return 1;
153         } else {
154             return 2;
155         }
156     }
157 
158     public static int integerTestCanonicalization3(boolean b) {
159         int x = b ? 128 : 64;
160         int y = b ? 32 : 64;
161         if ((x & y) == 0) {
162             return 1;
163         } else {
164             return 2;
165         }
166     }
