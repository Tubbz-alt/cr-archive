 59     public static void testSynchronizedSnippet(A x, A y) {
 60         synchronized (x) {
 61             field1 = x.value;
 62         }
 63         synchronized (x) {
 64             field2 = y.value;
 65         }

 68     @Test
 69     public void testLock() {
 70         test("testSynchronizedSnippet", new A(), new A());
 71 
 72         StructuredGraph graph = getGraph("testSynchronizedSnippet", false);
 73         new CanonicalizerPhase().apply(graph, new PhaseContext(getProviders()));
 74         new LockEliminationPhase().apply(graph);
 75         assertDeepEquals(1, graph.getNodes().filter(RawMonitorEnterNode.class).count());
 76         assertDeepEquals(1, graph.getNodes().filter(MonitorExitNode.class).count());
 77     }

117     private StructuredGraph getGraph(String snippet, boolean doEscapeAnalysis) {
118         ResolvedJavaMethod method = getResolvedJavaMethod(snippet);
119         StructuredGraph graph = parseEager(method, AllowAssumptions.YES);
120         HighTierContext context = getDefaultHighTierContext();
121         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
122         canonicalizer.apply(graph, context);
123         createInliningPhase().apply(graph, context);
124         new CanonicalizerPhase().apply(graph, context);
125         new DeadCodeEliminationPhase().apply(graph);
126         if (doEscapeAnalysis) {
127             new PartialEscapePhase(true, canonicalizer, graph.getOptions()).apply(graph, context);
128         }
129         new LoweringPhase(new CanonicalizerPhase(), LoweringTool.StandardLoweringStage.HIGH_TIER).apply(graph, context);
130         return graph;
131     }
