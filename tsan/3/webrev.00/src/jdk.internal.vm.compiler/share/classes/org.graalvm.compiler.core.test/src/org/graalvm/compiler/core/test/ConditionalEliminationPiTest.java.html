29 public class ConditionalEliminationPiTest extends ConditionalEliminationTestBase {
30 
31     static int SideEffect;
32 
33     static double oracleValue1 = -0.0;
34     static double oracleValue2;
35 
36     public static double testSnippet1(int a) {
37         double phi;
38         if (a > 0) {
39             double oracle = oracleValue1;
40             if (oracle == 0.0) {
41                 SideEffect = 1;
42             } else {
43                 return 123;
44             }
45             phi = oracle;
46         } else {
47             double oracle = oracleValue2;
48             if (oracle == 0.0) {
49                 SideEffect = 1;
50                 phi = oracle;
51             } else {
52                 return 0;
53             }
54         }
55         if (Double.doubleToRawLongBits(phi) == Double.doubleToRawLongBits(-0.0)) {
56             return 12;
57         }
58         return 2;
59     }
60 
61     @Test
62     public void test1() {
63         test("testSnippet1", 1);
64     }
