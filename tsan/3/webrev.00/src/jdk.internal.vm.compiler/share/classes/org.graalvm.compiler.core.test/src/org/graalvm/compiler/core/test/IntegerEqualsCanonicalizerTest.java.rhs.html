 72     /**
 73      * Tests the canonicalization of (x >>> const) == 0 to x |test| (-1 << const).
 74      */
 75     @Test
 76     public void testShiftEquals() {
 77         test("testShiftEqualsSnippet", "testShiftEqualsReference");
 78     }
 79 
 80     @SuppressWarnings("unused") private static int field;
 81 
 82     public static void testShiftEqualsSnippet(int x, int[] array, int y) {
 83         // optimize
 84         field = (x >>> 10) == 0 ? 1 : 0;
 85         field = (array.length >> 10) == 0 ? 1 : 0;
 86         field = (x << 10) == 0 ? 1 : 0;
 87         // don't optimize
 88         field = (x >> 10) == 0 ? 1 : 0;
 89         field = (x >>> y) == 0 ? 1 : 0;
 90         field = (x >> y) == 0 ? 1 : 0;
 91         field = (x << y) == 0 ? 1 : 0;
 92         field = (x >>> y) == 1 ? 1 : 0;
 93         field = (x >> y) == 1 ? 1 : 0;
 94         field = (x << y) == 1 ? 1 : 0;
 95     }
 96 
 97     public static void testShiftEqualsReference(int x, int[] array, int y) {
 98         field = (x & 0xfffffc00) == 0 ? 1 : 0;
 99         field = (array.length & 0xfffffc00) == 0 ? 1 : 0;
100         field = (x & 0x3fffff) == 0 ? 1 : 0;
101         // don't optimize signed right shifts
102         field = (x >> 10) == 0 ? 1 : 0;
103         // don't optimize no-constant shift amounts
104         field = (x >>> y) == 0 ? 1 : 0;
105         field = (x >> y) == 0 ? 1 : 0;
106         field = (x << y) == 0 ? 1 : 0;
107         // don't optimize non-zero comparisons
108         field = (x >>> y) == 1 ? 1 : 0;
109         field = (x >> y) == 1 ? 1 : 0;
110         field = (x << y) == 1 ? 1 : 0;
111     }

113     @Test
114     public void testCompare() {
115         test("testCompareSnippet", "testCompareReference");
116     }
117 
118     public static void testCompareSnippet(int x, int y, int[] array1, int[] array2) {
119         int tempX = x;
120         int array1Length = array1.length;
121         int array2Length = array2.length;
122         // optimize
123         field = x == tempX ? 1 : 0;
124         field = x != tempX ? 1 : 0;
125         field = array1Length != (-1 - array2Length) ? 1 : 0;
126         field = array1Length == (-1 - array2Length) ? 1 : 0;
127         // don't optimize
128         field = x == y ? 1 : 0;
129         field = array1Length == array2Length ? 1 : 0;
130         field = array1Length == (-array2Length) ? 1 : 0;
131     }
132 
133     public static void testCompareReference(int x, int y, int[] array1, int[] array2) {
134         int array1Length = array1.length;
135         int array2Length = array2.length;
136         // optimize
137         field = 1;
138         field = 0;
139         field = 1;
140         field = 0;
141         // don't optimize (overlapping value ranges)
142         field = x == y ? 1 : 0;
143         field = array1Length == array2Length ? 1 : 0;
144         field = array1Length == (-array2Length) ? 1 : 0;
145     }
146 
147     public static boolean testNormalIntegerTest(int a) {
148         return (a & 8) != 0;
149     }
150 
151     public static boolean testAlternateIntegerTest(int a) {
152         return (a & 8) == 8;
153     }
154 
155     @Test
156     public void testIntegerTest() {
157         test("testNormalIntegerTest", "testAlternateIntegerTest");
158     }
159 
160     private void test(String snippet, String referenceSnippet) {
161         StructuredGraph graph = getCanonicalizedGraph(snippet);
162         StructuredGraph referenceGraph = getCanonicalizedGraph(referenceSnippet);
163         assertEquals(referenceGraph, graph);
164     }
165 
166     private StructuredGraph getCanonicalizedGraph(String snippet) {
167         StructuredGraph graph = parseEager(snippet, AllowAssumptions.YES);
168         createCanonicalizerPhase().apply(graph, getProviders());
169         for (FrameState state : graph.getNodes(FrameState.TYPE).snapshot()) {
170             state.replaceAtUsages(null);
171             state.safeDelete();
172         }
173         return graph;
174     }
