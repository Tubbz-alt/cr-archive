25 package org.graalvm.compiler.core.test;
26 
27 import jdk.vm.ci.meta.ResolvedJavaMethod;
28 import org.graalvm.compiler.nodes.IfNode;
29 import org.graalvm.compiler.nodes.LogicNode;
30 import org.graalvm.compiler.nodes.ReturnNode;
31 import org.graalvm.compiler.nodes.calc.ObjectEqualsNode;
32 import org.junit.Test;
33 
34 import java.util.HashMap;
35 
36 public class HashMapGetTest extends GraalCompilerTest {
37 
38     public static void mapGet(HashMap<Integer, Integer> map, Integer key) {
39         map.get(key);
40     }
41 
42     @Test
43     public void hashMapTest() {
44         HashMap<Integer, Integer> map = new HashMap<>();
45         ResolvedJavaMethod get = getResolvedJavaMethod(HashMapGetTest.class, "mapGet");
46         for (int i = 0; i < 5000; i++) {
47             mapGet(map, i);
48             map.put(i, i);
49             mapGet(map, i);
50         }
51         test(get, null, map, 0);
52         for (IfNode ifNode : lastCompiledGraph.getNodes(IfNode.TYPE)) {
53             LogicNode condition = ifNode.condition();
54             if (ifNode.getTrueSuccessorProbability() < 0.4 && condition instanceof ObjectEqualsNode) {
55                 assertTrue(ifNode.trueSuccessor().next() instanceof ReturnNode, "Expected return.", ifNode.trueSuccessor(), ifNode.trueSuccessor().next());
56             }
57         }
58     }
59 
60 }
