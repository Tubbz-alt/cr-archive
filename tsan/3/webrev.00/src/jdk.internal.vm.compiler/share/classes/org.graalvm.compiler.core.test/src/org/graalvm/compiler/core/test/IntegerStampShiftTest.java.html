26 package org.graalvm.compiler.core.test;

28 import org.junit.Test;
29 
30 public class IntegerStampShiftTest extends GraalCompilerTest {
31 
32     public static int unsignedShiftPositiveInt(boolean f) {
33         int h = f ? 0x7FFFFFF0 : 0x7FFFFF00;
34         return h >>> 8;
35     }
36 
37     @Test
38     public void testUnsignedShiftPositiveInt() {
39         test("unsignedShiftPositiveInt", false);
40     }
41 
42     public static int unsignedShiftNegativeInt(boolean f) {
43         int h = f ? 0xFFFFFFF0 : 0xFFFFFF00;
44         return h >>> 8;
45     }
46 
47     @Test
48     public void testUnsignedShiftNegativeInt() {
49         test("unsignedShiftNegativeInt", false);
50     }
51 
52     public static long unsignedShiftPositiveLong(boolean f) {
53         long h = f ? 0x7FFFFFFFFFFFFFF0L : 0x7FFFFFFFFFFFFF00L;
54         return h >>> 8;
55     }
56 
57     @Test
58     public void testUnsignedShiftPositiveLong() {
59         test("unsignedShiftPositiveLong", false);
60     }
61 
62     public static long unsignedShiftNegativeLong(boolean f) {
63         long h = f ? 0xFFFFFFFFFFFFFFF0L : 0xFFFFFFFFFFFFFF00L;
64         return h >>> 8;
65     }
66 
67     @Test
68     public void testUnsignedShiftNegativeLong() {
69         test("unsignedShiftNegativeLong", false);
70     }
71 }
