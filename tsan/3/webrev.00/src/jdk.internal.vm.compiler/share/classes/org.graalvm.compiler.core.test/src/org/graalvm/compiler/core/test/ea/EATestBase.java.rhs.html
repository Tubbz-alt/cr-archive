 57     public static class TestClassInt {
 58         public int x;
 59         public int y;
 60         public int z;
 61 
 62         public TestClassInt() {
 63             this(0, 0);
 64         }
 65 
 66         public TestClassInt(int x) {
 67             this(x, 0);
 68         }
 69 
 70         public TestClassInt(int x, int y) {
 71             this.x = x;
 72             this.y = y;
 73         }
 74 
 75         @Override
 76         public boolean equals(Object obj) {
 77             TestClassInt other = (TestClassInt) obj;
 78             return x == other.x && y == other.y && z == other.z;
 79         }
 80 
 81         @Override
 82         public String toString() {
 83             return "{" + x + "," + y + "," + z + "}";
 84         }
 85 
 86         @Override
 87         public int hashCode() {
 88             return x + 13 * y;
 89         }
 90 
 91         public static final long fieldOffset1;
 92         public static final long fieldOffset2;
 93         public static final boolean firstFieldIsX;
 94 
 95         static {
 96             try {
 97                 long localFieldOffset1 = UNSAFE.objectFieldOffset(EATestBase.TestClassInt.class.getField("x"));
 98                 // Make the fields 8 byte aligned (Required for testing setLong on Architectures
 99                 // which does not support unaligned memory access
100                 if (localFieldOffset1 % 8 == 0) {
101                     fieldOffset1 = localFieldOffset1;
102                     fieldOffset2 = UNSAFE.objectFieldOffset(EATestBase.TestClassInt.class.getField("y"));
103                     firstFieldIsX = true;
104                 } else {
105                     fieldOffset1 = UNSAFE.objectFieldOffset(EATestBase.TestClassInt.class.getField("y"));
106                     fieldOffset2 = UNSAFE.objectFieldOffset(EATestBase.TestClassInt.class.getField("z"));
107                     firstFieldIsX = false;
108                 }
109                 assert fieldOffset2 == fieldOffset1 + 4;
110             } catch (Exception e) {
111                 throw new RuntimeException(e);
112             }
113         }
114 
115         public void setFirstField(int v) {
116             if (firstFieldIsX) {
117                 x = v;
118             } else {
119                 y = v;
120             }
121         }
122 
123         public void setSecondField(int v) {
124             if (firstFieldIsX) {
125                 y = v;
126             } else {
127                 z = v;
128             }
129         }
130     }

168     protected StructuredGraph graph;
169     protected HighTierContext context;
170     protected List<ReturnNode> returnNodes;

172     /**
173      * Runs Escape Analysis on the given snippet and makes sure that no allocations remain in the
174      * graph.
175      *
176      * @param snippet the name of the method whose graph should be processed
177      * @param expectedConstantResult if this is non-null, the resulting graph needs to have the
178      *            given constant return value
179      * @param iterativeEscapeAnalysis true if escape analysis should be run for more than one
180      *            iteration
181      */
182     protected void testEscapeAnalysis(String snippet, JavaConstant expectedConstantResult, boolean iterativeEscapeAnalysis) {
183         testEscapeAnalysis(snippet, expectedConstantResult, iterativeEscapeAnalysis, 0);
184     }
185 
186     protected void testEscapeAnalysis(String snippet, JavaConstant expectedConstantResult, boolean iterativeEscapeAnalysis, int expectedAllocationCount) {
187         prepareGraph(snippet, iterativeEscapeAnalysis);
188         if (expectedConstantResult != null) {
189             for (ReturnNode returnNode : returnNodes) {
190                 Assert.assertTrue(returnNode.result().toString(), returnNode.result().isConstant());
191                 Assert.assertEquals(expectedConstantResult, returnNode.result().asConstant());
192             }
193         }
194         int newInstanceCount = getAllocationCount();
195         Assert.assertEquals("Expected allocation count does not match", expectedAllocationCount, newInstanceCount);
196         if (expectedAllocationCount == 0) {
197             Assert.assertTrue("Unexpected CommitAllocationNode", graph.getNodes().filter(CommitAllocationNode.class).isEmpty());
198         }
199     }
200 
201     protected int getAllocationCount() {
202         return graph.getNodes().filter(isA(NewInstanceNode.class).or(NewArrayNode.class).or(AllocatedObjectNode.class)).count();
203     }
204 
205     @SuppressWarnings("try")
206     protected void prepareGraph(String snippet, boolean iterativeEscapeAnalysis) {
207         ResolvedJavaMethod method = getResolvedJavaMethod(snippet);
208         DebugContext debug = getDebugContext();
209         try (DebugContext.Scope s = debug.scope(getClass(), method, getCodeCache())) {
210             graph = parseEager(method, AllowAssumptions.YES, debug);
211             context = getDefaultHighTierContext();
212             createInliningPhase().apply(graph, context);
213             new DeadCodeEliminationPhase().apply(graph);
214             canonicalizeGraph();
215             new PartialEscapePhase(iterativeEscapeAnalysis, false, createCanonicalizerPhase(), null, graph.getOptions()).apply(graph, context);
216             postEACanonicalizeGraph();
217             returnNodes = graph.getNodes(ReturnNode.TYPE).snapshot();
218         } catch (Throwable e) {
219             throw debug.handle(e);
220         }
221     }
222 
223     protected void postEACanonicalizeGraph() {
224     }
225 
226     protected void canonicalizeGraph() {
227         this.createCanonicalizerPhase().apply(graph, context);
228     }
