 25 package org.graalvm.compiler.core.test;
 26 


 27 import org.graalvm.compiler.core.common.type.IntegerStamp;
 28 import org.graalvm.compiler.nodes.StructuredGraph;
 29 import org.graalvm.compiler.nodes.StructuredGraph.AllowAssumptions;

 30 import org.graalvm.compiler.phases.common.DeadCodeEliminationPhase;
 31 import org.junit.Test;
 32 
 33 /**
 34  * This class tests some specific patterns the stamp system should be able to canonicalize away
 35  * using {@link IntegerStamp#upMask()}.
 36  */
 37 public class StampCanonicalizerTest extends GraalCompilerTest {
 38 
 39     public static int andStamp(int a, int b) {
 40         int v = (a & 0xffff00) & (b & 0xff0000f);
 41         return v & 1;
 42     }
 43 
 44     @Test
 45     public void testAnd() {
 46         testZeroReturn("andStamp");
 47     }
 48 
 49     public static int shiftLeftStamp1(int a) {
 50         int v = a << 1;
 51         return v & 1;
 52     }
 53 
 54     public static int shiftLeftStamp2(int a) {
 55         int v = a << 1;
 56         if (a == 17) {
 57             v = a * 4;
 58         }
 59         return v & 1;
 60     }
 61 
 62     @Test
 63     public void testShift() {
 64         testZeroReturn("shiftLeftStamp1");
 65         testZeroReturn("shiftLeftStamp2");
 66     }
 67 
 68     public static int upperBoundShiftStamp1(int a) {
 69         int v = a & 0xffff;
 70         return (v << 4) & 0xff00000;
 71     }
 72 
 73     public static int upperBoundShiftStamp2(int a) {
 74         int v = a & 0xffff;
 75         return (v >> 4) & 0xf000;
 76     }
 77 
 78     @Test
 79     public void testUpperBoundShift() {
 80         testZeroReturn("upperBoundShiftStamp1");
 81         testZeroReturn("upperBoundShiftStamp2");
 82     }
 83 
 84     public static int divStamp1(int[] a) {
 85         int v = a.length / 4;
 86         return v & 0x80000000;
 87     }
 88 
 89     public static int divStamp2(int[] a) {
 90         int v = a.length / 5;
 91         return v & 0x80000000;
 92     }
 93 
 94     @Test
 95     public void testDiv() {
 96         testZeroReturn("divStamp1");
 97         testZeroReturn("divStamp2");
 98     }
 99 
100     public static int distinctMask(int a, int b) {
101         int x = a & 0xaaaa;
102         int y = (b & 0x5555) | 0x1;
103         return x == y ? 1 : 0;
104     }
105 
106     @Test
107     public void testDistinctMask() {
108         testZeroReturn("distinctMask");
109     }
110 
111     private void testZeroReturn(String methodName) {
112         StructuredGraph graph = parseEager(methodName, AllowAssumptions.YES);
113         createCanonicalizerPhase().apply(graph, getProviders());
114         new DeadCodeEliminationPhase().apply(graph);
115         assertConstantReturn(graph, 0);
116     }
117 }
