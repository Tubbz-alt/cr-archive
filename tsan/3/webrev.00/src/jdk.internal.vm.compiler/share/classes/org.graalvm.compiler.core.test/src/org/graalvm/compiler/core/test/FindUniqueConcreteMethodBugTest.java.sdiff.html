 27 import jdk.vm.ci.meta.Assumptions.AssumptionResult;

 29 import jdk.vm.ci.meta.ResolvedJavaType;

 35 public class FindUniqueConcreteMethodBugTest extends GraalCompilerTest {
 36 
 37     // To cause a C1 or C2 crash: -DFindUniqueConcreteMethodBugTest.ITERATIONS=10000
 38     private static final int ITERATIONS = Integer.getInteger("FindUniqueConcreteMethodBugTest.ITERATIONS", 100);
 39 
 40     /**
 41      * Executing {@link ResolvedJavaType#findUniqueConcreteMethod(ResolvedJavaMethod)} for the
 42      * method {@link Person#getName()} on the type {@link AbstractPerson} should return null as both
 43      * {@link PersonImpl} and {@link TenantImpl} provide implementations (namely
 44      * {@link PersonImpl#getName()} and {@link Tenant#getName()}).
 45      */
 46     @Test
 47     @Ignore("fix HotSpotResolvedObjectTypeImpl.findUniqueConcreteMethod")
 48     public void test() throws NoSuchMethodException {
 49         ResolvedJavaMethod ifaceMethod = getMetaAccess().lookupJavaMethod(Person.class.getDeclaredMethod("getName"));
 50 
 51         PersonImpl person = new PersonImpl("maya");
 52         TenantImpl tenant = new TenantImpl(0xdeadbeef);
 53 
 54         // Ensure the relevant methods are linked
 55         person.getName();
 56         tenant.getName();
 57 
 58         for (int i = 0; i < ITERATIONS; i++) {
 59             getLabelLength(person);
 60             getLabelLength(tenant);
 61         }
 62 
 63         // Until HotSpotResolvedObjectTypeImpl.findUniqueConcreteMethod is fixed,
 64         // this causes a VM crash as getLabelLength() directly invokes PersonImpl.getName().
 65         test("getLabelLength", tenant);
 66 
 67         ResolvedJavaMethod expected = null;
 68         AssumptionResult<ResolvedJavaMethod> actual = getMetaAccess().lookupJavaType(AbstractPerson.class).findUniqueConcreteMethod(ifaceMethod);
 69         Assert.assertEquals(expected, actual.getResult());
 70 
 71     }
 72 
 73     public int getLabelLength(AbstractPerson person) {
 74         return person.getName().length();
 75     }
 76 
 77     interface Person {
 78         String getName();
 79 
 80         default int getId() {
 81             return -1;
 82         }
 83     }
 84 
 85     interface Tenant extends Person {
 86         @Override
 87         default String getName() {
 88             return getAddress();
 89         }
