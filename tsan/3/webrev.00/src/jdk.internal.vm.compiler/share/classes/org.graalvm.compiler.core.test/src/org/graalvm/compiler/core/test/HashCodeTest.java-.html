141     public static final int hashCodeInterface(Appendable o) {
142         return o.hashCode();
143     }

145     @Test
146     public void test08() {
147         // This test requires profiling information which does not work reliable across platforms
148         // when running with -Xcomp
149         List<String> commandLine = SubprocessUtil.getVMCommandLine();
150         Assume.assumeTrue(commandLine != null);
151         Assume.assumeFalse(commandLine.contains("-Xcomp"));
152         initialize(Appendable.class);
153         checkForGuardedIntrinsicPattern("hashCodeInterface");
154 
155         // Ensure the profile for the dispatch in hashCodeSnippet01
156         // has a receiver type that does not select Object.hashCode intrinsic
157         hashCodeSnippet01(new HashMap<>());
158         checkForGuardedIntrinsicPattern("hashCodeSnippet01");
159     }
160 
161     private void checkForGuardedIntrinsicPattern(String name) {
162         StructuredGraph g = parseForCompile(getResolvedJavaMethod(name));
163         int invokeNodeCount = g.getNodes().filter(InvokeNode.class).count();
164         int invokeWithExceptionNodeCount = g.getNodes().filter(InvokeWithExceptionNode.class).count();
165         Assert.assertEquals(1, invokeNodeCount + invokeWithExceptionNodeCount);
166         Assert.assertEquals(1, g.getNodes().filter(LoadHubNode.class).count());
167         Assert.assertEquals(1, g.getNodes().filter(LoadMethodNode.class).count());
168     }
169 
170     @SuppressWarnings("try")
171     private StructuredGraph buildGraphAfterMidTier(String name) {
172         StructuredGraph g = parseForCompile(getResolvedJavaMethod(name));
173         OptionValues options = getInitialOptions();
174         new HighTier(options).apply(g, getDefaultHighTierContext());
175         new MidTier(options).apply(g, new MidTierContext(getProviders(), getTargetProvider(), OptimisticOptimizations.ALL, g.getProfilingInfo()));
176         return g;
177     }
