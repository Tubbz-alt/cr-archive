25 package org.graalvm.compiler.replacements.amd64;
26 
27 import static org.graalvm.compiler.replacements.NodeIntrinsificationProvider.INJECTED_TARGET;
28 
29 import org.graalvm.compiler.api.replacements.ClassSubstitution;
30 import org.graalvm.compiler.api.replacements.Fold;
31 import org.graalvm.compiler.api.replacements.MethodSubstitution;
32 import org.graalvm.compiler.core.common.SuppressFBWarnings;
33 import org.graalvm.compiler.replacements.nodes.BitScanForwardNode;
34 import org.graalvm.compiler.replacements.nodes.BitScanReverseNode;
35 
36 import jdk.vm.ci.amd64.AMD64;
37 import jdk.vm.ci.code.TargetDescription;
38 
39 @ClassSubstitution(Integer.class)
40 public class AMD64IntegerSubstitutions {
41 
42     @Fold
43     static boolean lzcnt(@Fold.InjectedParameter TargetDescription target) {
44         AMD64 arch = (AMD64) target.arch;
45         return arch.getFeatures().contains(AMD64.CPUFeature.LZCNT) && arch.getFlags().contains(AMD64.Flag.UseCountLeadingZerosInstruction);
46     }
47 
48     @Fold
49     static boolean tzcnt(@Fold.InjectedParameter TargetDescription target) {
50         AMD64 arch = (AMD64) target.arch;
51         return arch.getFeatures().contains(AMD64.CPUFeature.BMI1) && arch.getFlags().contains(AMD64.Flag.UseCountTrailingZerosInstruction);
52     }
53 
54     @MethodSubstitution
55     @SuppressFBWarnings(value = "NP_NULL_PARAM_DEREF_NONVIRTUAL", justification = "foldable method parameters are injected")
56     public static int numberOfLeadingZeros(int i) {
57         if (lzcnt(INJECTED_TARGET)) {
58             return AMD64CountLeadingZerosNode.countLeadingZeros(i);
59         }
60         if (i == 0) {
61             return 32;
62         }
63         return 31 - BitScanReverseNode.unsafeScan(i);
64     }
65 
66     @MethodSubstitution
67     @SuppressFBWarnings(value = "NP_NULL_PARAM_DEREF_NONVIRTUAL", justification = "foldable method parameters are injected")
68     public static int numberOfTrailingZeros(int i) {
69         if (tzcnt(INJECTED_TARGET)) {
70             return AMD64CountTrailingZerosNode.countTrailingZeros(i);
71         }
72         if (i == 0) {
73             return 32;
74         }
75         return BitScanForwardNode.unsafeScan(i);
76     }
77 }
