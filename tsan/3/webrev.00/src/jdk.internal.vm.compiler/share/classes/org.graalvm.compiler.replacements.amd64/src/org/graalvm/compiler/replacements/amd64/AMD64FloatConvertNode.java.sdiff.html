46 /**
47  * This node has the semantics of the AMD64 floating point conversions. It is used in the lowering
48  * of the {@link FloatConvertNode} which, on AMD64 needs a {@link AMD64FloatConvertNode} plus some
49  * fixup code that handles the corner cases that differ between AMD64 and Java.
50  *
51  * Since this node evaluates to a special value if the conversion is inexact, its stamp must be
52  * modified to avoid optimizing away {@link AMD64ConvertSnippets}.
53  */
54 @NodeInfo(cycles = CYCLES_8, size = SIZE_1)
55 public final class AMD64FloatConvertNode extends UnaryArithmeticNode<FloatConvertOp> implements ArithmeticLIRLowerable {
56     public static final NodeClass<AMD64FloatConvertNode> TYPE = NodeClass.create(AMD64FloatConvertNode.class);
57 
58     protected final FloatConvert op;
59 
60     public AMD64FloatConvertNode(FloatConvert op, ValueNode value) {
61         super(TYPE, table -> table.getFloatConvert(op), value);
62         this.op = op;
63         this.stamp = this.stamp.meet(createInexactCaseStamp());
64     }
65 





66     @Override
67     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
68         // nothing to do
69         return this;
70     }
71 
72     @Override
73     public Stamp foldStamp(Stamp newStamp) {
74         // The semantics of the x64 CVTTSS2SI instruction allow returning 0x8000000 in the special
75         // cases.
76         Stamp foldedStamp = super.foldStamp(newStamp);
77         return foldedStamp.meet(createInexactCaseStamp());
78     }
79 
80     private Stamp createInexactCaseStamp() {
81         IntegerStamp intStamp = (IntegerStamp) this.stamp;
82         long inexactValue = intStamp.getBits() <= 32 ? 0x8000_0000L : 0x8000_0000_0000_0000L;
83         return StampFactory.forConstant(JavaConstant.forPrimitiveInt(intStamp.getBits(), inexactValue));
84     }
