25 package micro.benchmarks;
26 
27 import org.openjdk.jmh.annotations.Benchmark;
28 import org.openjdk.jmh.annotations.Param;
29 import org.openjdk.jmh.annotations.Scope;
30 import org.openjdk.jmh.annotations.State;
31 import org.openjdk.jmh.annotations.Threads;
32 import org.openjdk.jmh.annotations.Warmup;
33 
34 /**
35  * Benchmarks cost of ArrayList.
36  */
37 public class ArrayAllocationBenchmark extends BenchmarkBase {
38 
39     @State(Scope.Benchmark)
40     public static class ThreadState {
41         @Param({"128", "256", "512", "1024", "2048", "4096", "8192", "16384", "32768", "65536", "131072"}) int size;
42         byte[] result;
43     }
44 
45     @Benchmark
46     @Threads(8)
47     @Warmup(iterations = 10)
48     public void arrayAllocate(ThreadState state) {
49         state.result = new byte[state.size];
50     }
51 }
