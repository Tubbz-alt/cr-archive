25 package org.graalvm.compiler.options;
26 
27 import java.util.ServiceLoader;
28 
29 public class ModuleSupport {
30 
31     static Iterable<OptionDescriptors> getOptionsLoader() {
32         /*
33          * The Graal module (i.e., jdk.internal.vm.compiler) is loaded by the platform class loader
34          * as of JDK 9. Modules that depend on and extend Graal are loaded by the app class loader.
35          * As such, we need to start the provider search at the app class loader instead of the
36          * platform class loader.
37          */
38         return ServiceLoader.load(OptionDescriptors.class, ClassLoader.getSystemClassLoader());
39     }
40 }
