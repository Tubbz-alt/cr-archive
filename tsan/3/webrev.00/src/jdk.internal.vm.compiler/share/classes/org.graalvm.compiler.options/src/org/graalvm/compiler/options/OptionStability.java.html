25 package org.graalvm.compiler.options;
26 
27 /**
28  * Categorizes options according to their stability.
29  */
30 public enum OptionStability {
31 
32     /**
33      * A stable option is expected to remain available for many releases. End users can rely on such
34      * an option being present. A stable option can still be removed but will go through a clear
35      * deprecating process before being removed.
36      */
37     STABLE,
38 
39     /**
40      * An experimental option has no guarantees of stability and might be removed at any point.
41      */
42     EXPERIMENTAL
43 }
