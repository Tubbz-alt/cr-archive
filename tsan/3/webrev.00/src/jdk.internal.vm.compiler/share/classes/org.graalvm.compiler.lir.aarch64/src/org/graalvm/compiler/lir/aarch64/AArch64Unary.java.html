 25 package org.graalvm.compiler.lir.aarch64;
 26 
 27 import static jdk.vm.ci.code.ValueUtil.asRegister;
 28 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.COMPOSITE;
 29 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.REG;
 30 
 31 import org.graalvm.compiler.asm.aarch64.AArch64Address;
 32 import org.graalvm.compiler.asm.aarch64.AArch64MacroAssembler;
 33 import org.graalvm.compiler.lir.LIRFrameState;
 34 import org.graalvm.compiler.lir.LIRInstructionClass;
 35 import org.graalvm.compiler.lir.StandardOp.ImplicitNullCheck;
 36 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
 37 
 38 import jdk.vm.ci.code.Register;
 39 import jdk.vm.ci.meta.AllocatableValue;
 40 import jdk.vm.ci.meta.Value;
 41 
 42 /**
 43  * AARCH64 LIR instructions that have one input and one output.
 44  */
 45 public class AArch64Unary {
 46 
 47     /**
 48      * Instruction with a {@link AArch64AddressValue memory} operand.
 49      */
 50     public static class MemoryOp extends AArch64LIRInstruction implements ImplicitNullCheck {
 51         public static final LIRInstructionClass<MemoryOp> TYPE = LIRInstructionClass.create(MemoryOp.class);
 52 
 53         private final boolean isSigned;
 54 
 55         @Def({REG}) protected AllocatableValue result;
 56         @Use({COMPOSITE}) protected AArch64AddressValue input;
 57 
 58         @State protected LIRFrameState state;
 59 
 60         private int targetSize;
 61         private int srcSize;
 62 
 63         public MemoryOp(boolean isSigned, int targetSize, int srcSize, AllocatableValue result, AArch64AddressValue input, LIRFrameState state) {
 64             super(TYPE);
 65             this.targetSize = targetSize;
 66             this.srcSize = srcSize;
 67             this.isSigned = isSigned;
 68             this.result = result;
 69             this.input = input;
 70             this.state = state;
 71         }
 72 
 73         @Override
 74         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
 75             int prePosition = masm.position();
 76             AArch64Address address = input.toAddress();
 77             Register dst = asRegister(result);
 78             if (isSigned) {
 79                 masm.ldrs(targetSize, srcSize, dst, address);
 80             } else {
 81                 masm.ldr(srcSize, dst, address);
 82             }
 83 
 84             if (state != null) {
 85                 int implicitExceptionPosition = prePosition;
 86                 // Adjust implicit exception position if this ldr/str has been merged to ldp/stp.
 87                 if (prePosition == masm.position() && masm.isImmLoadStoreMerged()) {
 88                     implicitExceptionPosition = prePosition - 4;
 89                     if (crb.isImplicitExceptionExist(implicitExceptionPosition)) {
 90                         return;
 91                     }
 92                 }
 93                 crb.recordImplicitException(implicitExceptionPosition, state);
 94             }
 95         }
 96 
 97         @Override
 98         public boolean makeNullCheckFor(Value value, LIRFrameState nullCheckState, int implicitNullCheckLimit) {
 99             int displacement = input.getDisplacement();
100             if (state == null && value.equals(input.getBase()) && input.getOffset().equals(Value.ILLEGAL) && displacement >= 0 && displacement < implicitNullCheckLimit) {
101                 state = nullCheckState;
102                 return true;
103             }
104             return false;
105         }
106     }
107 }
