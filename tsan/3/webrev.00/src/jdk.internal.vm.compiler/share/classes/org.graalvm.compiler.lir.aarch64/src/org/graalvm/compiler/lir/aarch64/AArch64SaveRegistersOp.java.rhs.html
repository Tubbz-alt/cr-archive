41 /**
42  * Saves registers to stack slots.
43  */
44 @Opcode("SAVE_REGISTER")
45 public class AArch64SaveRegistersOp extends SaveRegistersOp {
46     public static final LIRInstructionClass<AArch64SaveRegistersOp> TYPE = LIRInstructionClass.create(AArch64SaveRegistersOp.class);
47 















48     /**
49      *
50      * @param savedRegisters the registers saved by this operation which may be subject to
51      *            {@linkplain #remove(EconomicSet) pruning}
52      * @param savedRegisterLocations the slots to which the registers are saved

53      */
54     public AArch64SaveRegistersOp(Register[] savedRegisters, AllocatableValue[] savedRegisterLocations) {
55         super(TYPE, savedRegisters, savedRegisterLocations);








56     }
57 
58     protected void saveRegister(CompilationResultBuilder crb, AArch64MacroAssembler masm, StackSlot result, Register input) {
59         AArch64Move.reg2stack(crb, masm, result, input.asValue());
60     }
61 
62     @Override
63     public void emitCode(CompilationResultBuilder crb) {
64         AArch64MacroAssembler masm = (AArch64MacroAssembler) crb.asm;
65         for (int i = 0; i < savedRegisters.length; i++) {
66             if (savedRegisters[i] != null) {
67                 assert isStackSlot(slots[i]) : "not a StackSlot: " + slots[i];
68                 saveRegister(crb, masm, asStackSlot(slots[i]), savedRegisters[i]);
69             }
70         }
71     }





































































72 }
