 46 public class AArch64Compare {
 47 
 48     public static class CompareOp extends AArch64LIRInstruction {
 49         public static final LIRInstructionClass<CompareOp> TYPE = LIRInstructionClass.create(CompareOp.class);
 50 
 51         @Use protected Value x;
 52         @Use({REG, CONST}) protected Value y;
 53 
 54         public CompareOp(Value x, Value y) {
 55             super(TYPE);
 56             assert ((AArch64Kind) x.getPlatformKind()).isInteger() && ((AArch64Kind) y.getPlatformKind()).isInteger();
 57             assert x.getPlatformKind() == y.getPlatformKind();
 58             this.x = x;
 59             this.y = y;
 60         }
 61 
 62         @Override
 63         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
 64             gpCompare(masm, x, y);
 65         }
 66     }
 67 
 68     /**
 69      * Compares integer values x and y.
 70      *
 71      * @param x integer value to compare. May not be null.
 72      * @param y integer value to compare. May not be null.
 73      */
 74     public static void gpCompare(AArch64MacroAssembler masm, Value x, Value y) {
 75         final int size = x.getPlatformKind().getSizeInBytes() * Byte.SIZE;
 76         if (isRegister(y)) {
 77             masm.cmp(size, asRegister(x), asRegister(y));
 78         } else {
 79             JavaConstant constant = asJavaConstant(y);
 80             if (constant.isDefaultForKind()) {
 81                 masm.cmp(size, asRegister(x), 0);
 82             } else {
 83                 final long longValue = constant.asLong();
 84                 assert NumUtil.isInt(longValue);
 85                 int maskedValue;
 86                 switch (constant.getJavaKind()) {
 87                     case Boolean:
 88                     case Byte:
 89                         maskedValue = (int) (longValue & 0xFF);
 90                         break;
 91                     case Char:
 92                     case Short:
 93                         maskedValue = (int) (longValue & 0xFFFF);
 94                         break;
 95                     case Int:
 96                     case Long:
 97                         maskedValue = (int) longValue;
 98                         break;
 99                     default:
100                         throw GraalError.shouldNotReachHere();
101                 }
102                 masm.cmp(size, asRegister(x), maskedValue);
103             }
104         }
105     }
106 
107     public static class FloatCompareOp extends AArch64LIRInstruction {
108         public static final LIRInstructionClass<FloatCompareOp> TYPE = LIRInstructionClass.create(FloatCompareOp.class);
109 
110         @Use protected Value x;
111         @Use({REG, CONST}) protected Value y;
112         private final Condition condition;
113         private final boolean unorderedIsTrue;
114 
115         public FloatCompareOp(Value x, Value y, Condition condition, boolean unorderedIsTrue) {
116             super(TYPE);
117             assert !isJavaConstant(y) || isFloatCmpConstant(y, condition, unorderedIsTrue);
118             this.x = x;
119             this.y = y;
120             this.condition = condition;
121             this.unorderedIsTrue = unorderedIsTrue;
122         }
123 
124         /**
125          * Checks if val can be used as a constant for the gpCompare operation or not.
126          */
127         public static boolean isFloatCmpConstant(Value val, Condition condition, boolean unorderedIsTrue) {
128             // If the condition is "EQ || unordered" or "NE && unordered" we have to use 2 registers
129             // in any case.
130             if (!(condition == Condition.EQ && unorderedIsTrue || condition == Condition.NE && !unorderedIsTrue)) {
131                 return false;
132             }
133             return isJavaConstant(val) && asJavaConstant(val).isDefaultForKind();
134         }
135 
136         @Override
137         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
138             assert isRegister(x);
139             int size = x.getPlatformKind().getSizeInBytes() * Byte.SIZE;
140             if (isRegister(y)) {
141                 masm.fcmp(size, asRegister(x), asRegister(y));
142                 // There is no condition code for "EQ || unordered" nor one for "NE && unordered",
143                 // so we have to fix them up ourselves.
144                 // In both cases we combine the asked for condition into the EQ, respectively NE
145                 // condition, i.e.
146                 // if EQ && unoreredIsTrue, then the EQ flag will be set if the two values gpCompare
147                 // unequal but are
148                 // unordered.
149                 if (condition == Condition.EQ && unorderedIsTrue) {
150                     // if f1 ordered f2:
151                     // result = f1 == f2
152                     // else:
153                     // result = EQUAL
154                     int nzcv = 0b0100;   // EQUAL -> Z = 1
155                     masm.fccmp(size, asRegister(x), asRegister(y), nzcv, AArch64Assembler.ConditionFlag.VC);
156                 } else if (condition == Condition.NE && !unorderedIsTrue) {
157                     // if f1 ordered f2:
158                     // result = f1 != f2
159                     // else:
160                     // result = !NE == EQUAL
161                     int nzcv = 0b0100;   // EQUAL -> Z = 1
162                     masm.fccmp(size, asRegister(x), asRegister(y), nzcv, AArch64Assembler.ConditionFlag.VC);
163                 }
164             } else {
165                 // cmp against +0.0
166                 masm.fcmpZero(size, asRegister(x));
167             }
168         }
169 
170         @Override
171         public void verify() {
172             assert x.getPlatformKind().equals(y.getPlatformKind()) : "a: " + x + " b: " + y;
173         }
174     }
