25 package org.graalvm.compiler.lir.aarch64;
26 
27 import org.graalvm.compiler.lir.gen.ArithmeticLIRGeneratorTool;
28 
29 import jdk.vm.ci.meta.Value;
30 
31 /**
32  * This interface can be used to generate AArch64 LIR for arithmetic operations.
33  */
34 public interface AArch64ArithmeticLIRGeneratorTool extends ArithmeticLIRGeneratorTool {
35 
36     Value emitCountLeadingZeros(Value value);
37 
38     Value emitCountTrailingZeros(Value value);
39 
40     enum RoundingMode {
41         NEAREST(0),
42         DOWN(1),
43         UP(2),
44         TRUNCATE(3);
45 
46         public final int encoding;
47 
48         RoundingMode(int encoding) {
49             this.encoding = encoding;
50         }
51     }
52 
53     Value emitRound(Value value, RoundingMode mode);
54 }
