 75             if (state != null) {

 79             Register dst = asRegister(result);

 63         public MemoryOp(boolean isSigned, int targetSize, int srcSize, AllocatableValue result, AArch64AddressValue input, LIRFrameState state) {
 64             super(TYPE);
 65             this.targetSize = targetSize;

 70             this.state = state;
 71         }
 72 
 73         @Override
 74         public void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
 75             int prePosition = masm.position();


 76             AArch64Address address = input.toAddress();
 77             Register dst = asRegister(result);
 78             if (isSigned) {
 79                 masm.ldrs(targetSize, srcSize, dst, address);
 80             } else {
 81                 masm.ldr(srcSize, dst, address);
 82             }
 83 
 84             if (state != null) {
 85                 int implicitExceptionPosition = prePosition;
 86                 // Adjust implicit exception position if this ldr/str has been merged to ldp/stp.
 87                 if (prePosition == masm.position() && masm.isImmLoadStoreMerged()) {
 88                     implicitExceptionPosition = prePosition - 4;
 89                     if (crb.isImplicitExceptionExist(implicitExceptionPosition)) {
 90                         return;
 91                     }
 92                 }
 93                 crb.recordImplicitException(implicitExceptionPosition, state);
 94             }
 95         }

103             }
