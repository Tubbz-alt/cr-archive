41 package jdk.internal.vm.compiler.word;
42 
43 /**
44  * The root of the interface hierarchy for machine-word-sized values.
45  *
46  * @since 1.0
47  */
48 public interface WordBase {
49 
50     /**
51      * Conversion to a Java primitive value.
52      *
53      * @since 1.0
54      */
55     long rawValue();
56 
57     /**
58      * This is deprecated because of the easy to mistype name collision between {@link #equals} and
59      * the other word based equality routines. In general you should never be statically calling
60      * this method anyway.
61      *
62      * @since 1.0
63      */
64     @Override
65     @Deprecated
66     boolean equals(Object o);
67 }
