41 package jdk.internal.vm.compiler.word;
42 
43 /**
44  * A machine-word-sized value that can be compared for equality.
45  *
46  * @since 1.0
47  */
48 public interface ComparableWord extends WordBase {
49 
50     /**
51      * Compares this word with the specified value.
52      *
53      * @param val value to which this word is to be compared.
54      * @return {@code this == val}
55      *
56      * @since 1.0
57      */
58     boolean equal(ComparableWord val);
59 
60     /**
61      * Compares this word with the specified value.
62      *
63      * @param val value to which this word is to be compared.
64      * @return {@code this != val}
65      *
66      * @since 1.0
67      */
68     boolean notEqual(ComparableWord val);
69 }
