41 package jdk.internal.vm.compiler.word;
42 
43 /**
44  * Marker interface for all {@link WordBase word types} that have the semantic of a pointer (but not
45  * necessarily all the memory access methods defined in {@link Pointer}).
46  *
47  * @since 19.0
48  */
49 public interface PointerBase extends ComparableWord {
50 
51     /**
52      * Returns true if this pointer is the {@link WordFactory#nullPointer null pointer}.
53      *
54      * @since 19.0
55      */
56     boolean isNull();
57 
58     /**
59      * Returns true if this pointer is not the {@link WordFactory#nullPointer null pointer}.
60      *
61      * @since 19.0
62      */
63     boolean isNonNull();
64 }
