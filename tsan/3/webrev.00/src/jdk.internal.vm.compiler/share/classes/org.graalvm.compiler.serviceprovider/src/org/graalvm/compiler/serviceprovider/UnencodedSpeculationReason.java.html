25 package org.graalvm.compiler.serviceprovider;
26 
27 import java.util.Arrays;
28 
29 import jdk.vm.ci.meta.SpeculationLog.SpeculationReason;
30 
31 /**
32  * An implementation of {@link SpeculationReason} based on direct, unencoded values.
33  */
34 public final class UnencodedSpeculationReason implements SpeculationReason {
35     final int groupId;
36     final String groupName;
37     final Object[] context;
38 
39     UnencodedSpeculationReason(int groupId, String groupName, Object[] context) {
40         this.groupId = groupId;
41         this.groupName = groupName;
42         this.context = context;
43     }
44 
45     @Override
46     public boolean equals(Object obj) {
47         if (obj instanceof UnencodedSpeculationReason) {
48             UnencodedSpeculationReason that = (UnencodedSpeculationReason) obj;
49             return this.groupId == that.groupId && Arrays.equals(this.context, that.context);
50         }
51         return false;
52     }
53 
54     @Override
55     public int hashCode() {
56         return groupId + Arrays.hashCode(this.context);
57     }
58 
59     @Override
60     public String toString() {
61         return String.format("%s@%d%s", groupName, groupId, Arrays.toString(context));
62     }
63 }
