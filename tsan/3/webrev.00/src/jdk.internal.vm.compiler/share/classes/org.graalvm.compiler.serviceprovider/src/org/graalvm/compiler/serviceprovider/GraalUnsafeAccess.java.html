25 package org.graalvm.compiler.serviceprovider;
26 
27 import java.lang.reflect.Field;
28 
29 import sun.misc.Unsafe;
30 
31 /**
32  * Access to the {@link Unsafe} capability. Care must be taken not to leak the {@link #UNSAFE} value
33  * of out code loaded by the JVMCI class loader or encapsulated in the JVMCI or Graal modules into
34  * other code (e.g. via the Polyglot API).
35  */
36 public class GraalUnsafeAccess {
37 
38     private static final Unsafe UNSAFE = initUnsafe();
39 
40     private static Unsafe initUnsafe() {
41         try {
42             // Fast path when we are trusted.
43             return Unsafe.getUnsafe();
44         } catch (SecurityException se) {
45             // Slow path when we are not trusted.
46             try {
47                 Field theUnsafe = Unsafe.class.getDeclaredField("theUnsafe");
48                 theUnsafe.setAccessible(true);
49                 return (Unsafe) theUnsafe.get(Unsafe.class);
50             } catch (Exception e) {
51                 throw new RuntimeException("exception while trying to get Unsafe", e);
52             }
53         }
54     }
55 
56     /**
57      * Gets the {@link Unsafe} singleton.
58      *
59      * @throws SecurityException if a security manager is present and it denies
60      *             {@link RuntimePermission}("accessUnsafe")
61      */
62     public static Unsafe getUnsafe() {
63         SecurityManager sm = System.getSecurityManager();
64         if (sm != null) {
65             sm.checkPermission(new RuntimePermission("accessUnsafe"));
66         }
67         return UNSAFE;
68     }
69 }
