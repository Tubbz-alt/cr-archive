25 package org.graalvm.compiler.serviceprovider;
26 
27 import java.nio.Buffer;
28 
29 /**
30  * Covariant return types for some methods in the {@code java.nio.Buffer} were
31  * <a href="https://bugs.openjdk.java.net/browse/JDK-4774077">introduced in JDK 9</a>.
32  *
33  * If calls to these methods are compiled with javac from JDK 9+ using {@code -target 8 -source 8}
34  * then the call sites will invoke the covariant methods in the subclass. For example:
35  *
36  * <pre>
37  * static void reset(ByteBuffer buf) {
38  *     buf.reset();
39  * }
40  * </pre>
41  *
42  * will result in:
43  *
44  * <pre>
45  *    0: aload_0
46  *    1: invokevirtual #7  // Method java/nio/ByteBuffer.reset:()Ljava/nio/ByteBuffer;
47  *    4: pop
48  *    5: return
49  * </pre>
50  *
51  * This will result in a {@link NoSuchMethodError} when run on JDK 8. The workaround for this is to
52  * {@linkplain #asBaseBuffer(Buffer) coerce} the receiver for calls to the covariant methods to
53  * {@link Buffer}.
54  */
55 public final class BufferUtil {
56 
57     /**
58      * Coerces {@code obj} to be of type {@link Buffer}. This is required instead of a cast as
59      * {@code org.graalvm.compiler.core.test.VerifyBufferUsage} is based on Graal graphs which will
60      * have had redundant casts eliminated by the bytecode parser.
61      */
62     public static Buffer asBaseBuffer(Buffer obj) {
63         return obj;
64     }
65 }
