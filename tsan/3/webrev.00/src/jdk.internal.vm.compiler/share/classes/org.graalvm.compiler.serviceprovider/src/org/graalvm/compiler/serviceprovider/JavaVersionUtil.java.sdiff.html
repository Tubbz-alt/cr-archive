25 package org.graalvm.compiler.serviceprovider;

27 /**
28  * Interface to query which JDK version Graal is running on.
29  */
30 public final class JavaVersionUtil {
31 
32     private static int getJavaSpecificationVersion() {
33         String value = System.getProperty("java.specification.version");
34         if (value.startsWith("1.")) {
35             value = value.substring(2);
36         }
37         return Integer.parseInt(value);
38     }
39 
40     /**
41      * The integer value corresponding to the value of the {@code java.specification.version} system
42      * property after any leading {@code "1."} has been stripped.
43      */
44     public static final int JAVA_SPECIFICATION_VERSION = getJavaSpecificationVersion();
45 
46     /**
47      * Determines if the Java runtime is version 8 or earlier.
48      */
49     public static final boolean Java8OrEarlier = JAVA_SPECIFICATION_VERSION <= 8;
50 
51     /**
52      * Determines if the Java runtime is version 11 or earlier.
53      */
54     public static final boolean Java11OrEarlier = JAVA_SPECIFICATION_VERSION <= 11;
55 
56     private JavaVersionUtil() {
57     }
58 }
