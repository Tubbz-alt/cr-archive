25 package org.graalvm.compiler.serviceprovider;
26 
27 import jdk.vm.ci.services.Services;
28 
29 /**
30  * Interface to query which JDK version Graal is running on.
31  */
32 public final class JavaVersionUtil {
33 
34     private static int getJavaSpecificationVersion() {
35         String value = Services.getSavedProperties().get("java.specification.version");
36         if (value.startsWith("1.")) {
37             value = value.substring(2);
38         }
39         return Integer.parseInt(value);
40     }
41 
42     /**
43      * The integer value corresponding to the value of the {@code java.specification.version} system
44      * property after any leading {@code "1."} has been stripped.
45      */
46     public static final int JAVA_SPEC = getJavaSpecificationVersion();










47 
48     private JavaVersionUtil() {
49     }
50 }
