25 package org.graalvm.util;
26 
27 import java.io.DataInputStream;
28 import java.io.IOException;
29 import java.io.InputStream;
30 
31 /**
32  * A stream that can read (trivial) values using their in-band data type information, intended for
33  * use with {@link TypedDataOutputStream}.
34  */
35 public class TypedDataInputStream extends DataInputStream {
36     public TypedDataInputStream(InputStream in) {
37         super(in);
38     }
39 
40     /**
41      * Reads a single value, using the data type encoded in the stream.
42      *
43      * @return The read value, such as a boxed primitive or a {@link String}.
44      * @exception IOException in case of an I/O error.
45      */
46     public Object readTypedValue() throws IOException {
47         Object value;
48         final byte type = readByte();
49         switch (type) {
50             case 'Z':
51                 value = readBoolean();
52                 break;
53             case 'B':
54                 value = readByte();
55                 break;
56             case 'S':
57                 value = readShort();
58                 break;
59             case 'C':
60                 value = readChar();
61                 break;
62             case 'I':
63                 value = readInt();
64                 break;
65             case 'J':
66                 value = readLong();
67                 break;
68             case 'F':
69                 value = readFloat();
70                 break;
71             case 'D':
72                 value = readDouble();
73                 break;
74             case 'U':
75                 value = readUTF();
76                 break;
77             default:
78                 throw new IOException("Unsupported type: " + Integer.toHexString(type));
79         }
80         return value;
81     }
82 }
