 25 package org.graalvm.util;
 26 
 27 import java.io.DataOutputStream;
 28 import java.io.IOException;
 29 import java.io.OutputStream;
 30 
 31 /**
 32  * A stream that can write (trivial) values together with their data type, for use with
 33  * {@link TypedDataInputStream}.
 34  */
 35 public class TypedDataOutputStream extends DataOutputStream {
 36     /** Determines if {@code value} is supported by {@link #writeTypedValue(Object)}. */
 37     public static boolean isValueSupported(Object value) {
 38         if (value == null) {
 39             return false;
 40         }
 41         Class<?> valueClass = value.getClass();
 42         return valueClass == Boolean.class ||
 43                         valueClass == Byte.class ||
 44                         valueClass == Short.class ||
 45                         valueClass == Character.class ||
 46                         valueClass == Integer.class ||
 47                         valueClass == Long.class ||
 48                         valueClass == Float.class ||
 49                         valueClass == Double.class ||
 50                         valueClass == String.class ||
 51                         value.getClass().isEnum();
 52     }
 53 
 54     public TypedDataOutputStream(OutputStream out) {
 55         super(out);
 56     }
 57 
 58     /**
 59      * Writes the value that is represented by the given non-null object, together with information
 60      * on the value's data type.
 61      *
 62      * @param value A value of a {@linkplain #isValueSupported supported type}.
 63      * @exception IllegalArgumentException when the provided type is not supported.
 64      * @exception IOException in case of an I/O error.
 65      */
 66     public void writeTypedValue(Object value) throws IOException {
 67         Class<?> valueClz = value.getClass();
 68         if (valueClz == Boolean.class) {
 69             this.writeByte('Z');
 70             this.writeBoolean((Boolean) value);
 71         } else if (valueClz == Byte.class) {
 72             this.writeByte('B');
 73             this.writeByte((Byte) value);
 74         } else if (valueClz == Short.class) {
 75             this.writeByte('S');
 76             this.writeShort((Short) value);
 77         } else if (valueClz == Character.class) {
 78             this.writeByte('C');
 79             this.writeChar((Character) value);
 80         } else if (valueClz == Integer.class) {
 81             this.writeByte('I');
 82             this.writeInt((Integer) value);
 83         } else if (valueClz == Long.class) {
 84             this.writeByte('J');
 85             this.writeLong((Long) value);
 86         } else if (valueClz == Float.class) {
 87             this.writeByte('F');
 88             this.writeFloat((Float) value);
 89         } else if (valueClz == Double.class) {
 90             this.writeByte('D');
 91             this.writeDouble((Double) value);
 92         } else if (valueClz == String.class) {
 93             this.writeByte('U');
 94             this.writeUTF((String) value);
 95         } else if (valueClz.isEnum()) {
 96             this.writeByte('U');
 97             this.writeUTF(((Enum<?>) value).name());
 98         } else {
 99             throw new IllegalArgumentException(String.format("Unsupported type: Value: %s, Value type: %s", value, valueClz));
100         }
101     }
102 }
