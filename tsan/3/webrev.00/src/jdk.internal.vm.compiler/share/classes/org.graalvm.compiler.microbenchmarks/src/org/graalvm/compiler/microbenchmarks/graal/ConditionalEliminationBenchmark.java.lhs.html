 36     @MethodSpec(declaringClass = ConditionalEliminationBenchmark.class, name = "nullnessSnippet")
 37     public static class Nullness extends GraphState {
 38     }
 39 
 40     @SuppressWarnings("unused")
 41     public static int nullnessSnippet(Object a, Object b, int n, int m) {
 42         int result = 0;
 43         if (a == null) {
 44             if (a == b) {
 45                 if (b == null) {
 46                     for (int i = 0; i < n; ++i) {
 47                         if (i % 3 == 0) {
 48                             break;
 49                         }
 50                         result <<= 1;
 51                         if (i % 7 == 0) {
 52                             break;
 53                         }
 54                     }
 55                     return 1;
 56                 } else {
 57                     return -2;
 58                 }
 59             } else {
 60                 if (b == null) {
 61                     for (int i = 0; i < n; ++i) {
 62                         if (i % 4 == 0) {
 63                             break;
 64                         }
 65                         result <<= 1;
 66                         if (i % 7 == 0) {
 67                             break;
 68                         }
 69                     }
 70                     return result;
 71                 } else {
 72                     return 4;
 73                 }
 74             }
 75         } else {
 76             if (a == b) {
 77                 if (b == null) {
 78                     for (int i = 0; i < n; ++i) {
 79                         if (i % 5 == 0) {
 80                             break;
 81                         }
 82                         result <<= 1;
 83                         if (i % 7 == 0) {
 84                             break;
 85                         }
 86                     }
 87                     return result;
 88                 } else {
 89                     return 6;
 90                 }
 91             } else {
 92                 if (b == null) {
 93                     for (int i = 0; i < n; ++i) {
 94                         if (i % 6 == 0) {
 95                             break;
 96                         }
 97                         result <<= 1;
 98                         if (i % 7 == 0) {
 99                             break;
100                         }
101                     }
102                     return result;
103                 } else {
104                     return 8;
105                 }
106             }
107         }
108     }
109 
110     @Benchmark
111     public void nullness(Nullness s, GraalState g) {
112         new ConditionalEliminationPhase(false).apply(s.graph, new PhaseContext(g.providers));
113     }
114 
115     @Benchmark
116     public void newDominatorConditionalElimination(Nullness s, GraalState g) {
117         new ConditionalEliminationPhase(false).apply(s.graph, new PhaseContext(g.providers));
118     }
119 
120     @MethodSpec(declaringClass = ConditionalEliminationBenchmark.class, name = "searchSnippet")
121     public static class Search extends GraphState {
122     }
123 
124     static class Entry {
125         final String name;
126 
127         Entry(String name) {
128             this.name = name;
129         }
130     }
131 
132     static class EntryWithNext extends Entry {
133         EntryWithNext(String name, Entry next) {
134             super(name);
135             this.next = next;
136         }
137 
138         final Entry next;
139     }
140 
141     public static Entry searchSnippet(Entry start, String name, Entry alternative) {
142         Entry current = start;
143         do {
144             while (current instanceof EntryWithNext) {
145                 if (name != null && current.name == name) {
146                     current = null;
147                 } else {
148                     Entry next = ((EntryWithNext) current).next;
149                     current = next;
150                 }
151             }
152 
153             if (current != null) {
154                 if (current.name.equals(name)) {
155                     return current;
156                 }
157             }
158             if (current == alternative) {
159                 return null;
160             }
161             current = alternative;
162 
163         } while (true);
164     }
165 
166     @Benchmark
167     public void search(Search s, GraalState g) {
168         new ConditionalEliminationPhase(false).apply(s.graph, new PhaseContext(g.providers));
169     }
