 61 public class MidTier extends PhaseSuite<MidTierContext> {
 62 
 63     public MidTier(OptionValues options) {
 64         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
 65         if (ImmutableCode.getValue(options)) {
 66             canonicalizer.disableReadCanonicalization();
 67         }
 68 
 69         appendPhase(new LockEliminationPhase());
 70 
 71         if (OptFloatingReads.getValue(options)) {
 72             appendPhase(new IncrementalCanonicalizerPhase<>(canonicalizer, new FloatingReadPhase()));
 73         }
 74 
 75         if (ConditionalElimination.getValue(options)) {
 76             appendPhase(new IterativeConditionalEliminationPhase(canonicalizer, true));
 77         }
 78 
 79         appendPhase(new LoopSafepointEliminationPhase());
 80 
 81         appendPhase(new LoopSafepointInsertionPhase());
 82 
 83         appendPhase(new GuardLoweringPhase());
 84 
 85         if (MitigateSpeculativeExecutionAttacks.getValue(options) == GuardTargets || MitigateSpeculativeExecutionAttacks.getValue(options) == NonDeoptGuardTargets) {
 86             appendPhase(new InsertGuardFencesPhase());
 87         }
 88 
 89         if (VerifyHeapAtReturn.getValue(options)) {
 90             appendPhase(new VerifyHeapAtReturnPhase());
 91         }
 92 
 93         appendPhase(new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.MID_TIER));
 94 
 95         appendPhase(new FrameStateAssignmentPhase());
 96 
 97         LoopPolicies loopPolicies = createLoopPolicies();
 98         if (OptLoopTransform.getValue(options)) {
 99             if (PartialUnroll.getValue(options)) {
100                 appendPhase(new LoopPartialUnrollPhase(loopPolicies, canonicalizer));
101             }
102         }
103         if (ReassociateInvariants.getValue(options)) {
104             appendPhase(new ReassociateInvariantPhase());
105         }
106 
107         if (OptDeoptimizationGrouping.getValue(options)) {
108             appendPhase(new DeoptimizationGroupingPhase());
109         }
110 
111         appendPhase(canonicalizer);
112     }
113 
114     public LoopPolicies createLoopPolicies() {
115         return new DefaultLoopPolicies();
116     }
117 }
