25 package org.graalvm.compiler.core.phases;
26 
27 import org.graalvm.compiler.nodes.spi.LoweringTool;
28 import org.graalvm.compiler.options.OptionValues;
29 import org.graalvm.compiler.phases.common.CanonicalizerPhase;
30 import org.graalvm.compiler.phases.common.FrameStateAssignmentPhase;
31 import org.graalvm.compiler.phases.common.GuardLoweringPhase;
32 import org.graalvm.compiler.phases.common.LoopSafepointInsertionPhase;
33 import org.graalvm.compiler.phases.common.LoweringPhase;
34 import org.graalvm.compiler.phases.common.RemoveValueProxyPhase;
35 import org.graalvm.compiler.phases.common.WriteBarrierAdditionPhase;
36 import org.graalvm.compiler.phases.tiers.MidTierContext;
37 
38 public class EconomyMidTier extends BaseTier<MidTierContext> {
39 
40     public EconomyMidTier(OptionValues options) {
41         CanonicalizerPhase canonicalizer = this.createCanonicalizerPhase(options);
42         appendPhase(new RemoveValueProxyPhase());
43         appendPhase(new LoopSafepointInsertionPhase());
44         appendPhase(new GuardLoweringPhase());
45         appendPhase(new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.MID_TIER));
46         appendPhase(new FrameStateAssignmentPhase());
47         appendPhase(new WriteBarrierAdditionPhase());
48     }
49 }
