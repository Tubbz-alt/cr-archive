 42 /**
 43  * A named {@link MatchPattern} along with a {@link MatchGenerator} that can be evaluated to replace
 44  * one or more {@link Node}s with a single {@link Value}.
 45  */
 46 
 47 public class MatchStatement {
 48     private static final CounterKey MatchStatementSuccess = DebugContext.counter("MatchStatementSuccess");
 49 
 50     /**
 51      * A printable name for this statement. Usually it's just the name of the method doing the
 52      * emission.
 53      */
 54     private final String name;
 55 
 56     /**
 57      * The actual match pattern.
 58      */
 59     private final MatchPattern pattern;
 60 

 63      */
 64     private MatchGenerator generatorMethod;
 65 
 66     /**
 67      * The name of arguments in the order they are expected to be passed to the generator method.
 68      */
 69     private String[] arguments;
 70 
 71     public MatchStatement(String name, MatchPattern pattern, MatchGenerator generator, String[] arguments) {
 72         this.name = name;
 73         this.pattern = pattern;
 74         this.generatorMethod = generator;
 75         this.arguments = arguments;
 76     }
 77 
 78     /**
 79      * Attempt to match the current statement against a Node.
 80      *
 81      * @param builder the current builder instance.
 82      * @param node the node to be matched
 83      * @param nodes the nodes in the current block

 84      * @return true if the statement matched something and set a {@link ComplexMatchResult} to be
 85      *         evaluated by the NodeLIRBuilder.
 86      */
 87     public boolean generate(NodeLIRBuilder builder, int index, Node node, List<Node> nodes) {
 88         DebugContext debug = node.getDebug();
 89         assert index == nodes.indexOf(node);
 90         // Check that the basic shape matches
 91         Result result = pattern.matchShape(node, this);
 92         if (result != Result.OK) {
 93             return false;
 94         }
 95         // Now ensure that the other safety constraints are matched.
 96         MatchContext context = new MatchContext(builder, this, index, node, nodes);
 97         result = pattern.matchUsage(node, context);
 98         if (result == Result.OK) {
 99             // Invoke the generator method and set the result if it's non null.
100             ComplexMatchResult value = generatorMethod.match(builder.getNodeMatchRules(), buildArgList(context));
101             if (value != null) {
102                 context.setResult(value);
103                 MatchStatementSuccess.increment(debug);
104                 DebugContext.counter("MatchStatement[%s]", getName()).increment(debug);
105                 return true;
106             }
107             // The pattern matched but some other code generation constraint disallowed code
108             // generation for the pattern.
109             if (LogVerbose.getValue(node.getOptions())) {
110                 debug.log("while matching %s|%s %s %s returned null", context.getRoot().toString(Verbosity.Id), context.getRoot().getClass().getSimpleName(), getName(), generatorMethod.getName());
111                 debug.log("with nodes %s", formatMatch(node));
112             }
113         } else {
114             if (LogVerbose.getValue(node.getOptions()) && result.code != MatchResultCode.WRONG_CLASS) {
115                 debug.log("while matching %s|%s %s %s", context.getRoot().toString(Verbosity.Id), context.getRoot().getClass().getSimpleName(), getName(), result);
116             }
