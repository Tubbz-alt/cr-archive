25 package org.graalvm.compiler.core.phases;
26 
27 import static org.graalvm.compiler.core.common.GraalOptions.ImmutableCode;
28 
29 import org.graalvm.compiler.nodes.spi.LoweringTool;
30 import org.graalvm.compiler.options.OptionValues;
31 import org.graalvm.compiler.phases.PhaseSuite;
32 import org.graalvm.compiler.phases.common.CanonicalizerPhase;
33 import org.graalvm.compiler.phases.common.FrameStateAssignmentPhase;
34 import org.graalvm.compiler.phases.common.GuardLoweringPhase;
35 import org.graalvm.compiler.phases.common.LoopSafepointInsertionPhase;
36 import org.graalvm.compiler.phases.common.LoweringPhase;
37 import org.graalvm.compiler.phases.common.RemoveValueProxyPhase;
38 import org.graalvm.compiler.phases.tiers.MidTierContext;
39 
40 public class EconomyMidTier extends PhaseSuite<MidTierContext> {
41 
42     public EconomyMidTier(OptionValues options) {
43         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
44         if (ImmutableCode.getValue(options)) {
45             canonicalizer.disableReadCanonicalization();
46         }
47         appendPhase(new RemoveValueProxyPhase());
48 
49         appendPhase(new LoopSafepointInsertionPhase());
50 
51         appendPhase(new GuardLoweringPhase());
52 
53         appendPhase(new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.MID_TIER));
54 
55         appendPhase(new FrameStateAssignmentPhase());
56     }
57 }
