25 package org.graalvm.compiler.core;
26 
27 /**
28  * This is a utility class for Threads started by the compiler itself. In certain execution
29  * environments extra work must be done for these threads to execute correctly and this class
30  * provides hooks for this work.
31  */
32 public class GraalServiceThread extends Thread {
33     private final Runnable runnable;
34 
35     public GraalServiceThread(Runnable runnable) {
36         super();
37         this.runnable = runnable;
38     }
39 
40     @Override
41     public final void run() {
42         beforeRun();
43         try {
44             runnable.run();
45         } finally {
46             afterRun();
47         }
48     }
49 
50     /**
51      * Substituted by {@code com.oracle.svm.graal.hotspot.libgraal.
52      * Target_org_graalvm_compiler_truffle_common_TruffleCompilerRuntimeInstance} to attach to the
53      * peer runtime if required.
54      */
55     private void afterRun() {
56     }
57 
58     /**
59      * Substituted by {@code com.oracle.svm.graal.hotspot.libgraal.
60      * Target_org_graalvm_compiler_truffle_common_TruffleCompilerRuntimeInstance} to attach to the
61      * peer runtime if required.
62      */
63     private void beforeRun() {
64     }
65 }
