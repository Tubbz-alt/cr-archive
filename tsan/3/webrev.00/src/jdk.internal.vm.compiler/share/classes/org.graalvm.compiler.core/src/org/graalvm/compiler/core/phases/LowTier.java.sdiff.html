50 public class LowTier extends PhaseSuite<LowTierContext> {
51 
52     static class Options {
53 
54         // @formatter:off
55         @Option(help = "", type = OptionType.Debug)
56         public static final OptionKey<Boolean> ProfileCompiledMethods = new OptionKey<>(false);
57         // @formatter:on
58 
59     }
60 
61     public LowTier(OptionValues options) {
62         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
63         CanonicalizerPhase canonicalizerWithoutGVN = new CanonicalizerPhase();
64         canonicalizerWithoutGVN.disableGVN();
65         if (ImmutableCode.getValue(options)) {
66             canonicalizer.disableReadCanonicalization();
67             canonicalizerWithoutGVN.disableReadCanonicalization();
68         }
69 
70         if (Options.ProfileCompiledMethods.getValue(options)) {
71             appendPhase(new ProfileCompiledMethodsPhase());
72         }
73 
74         appendPhase(new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.LOW_TIER));
75 
76         appendPhase(new OptimizeDivPhase());
77 
78         appendPhase(new ExpandLogicPhase());
79 
80         appendPhase(new FixReadsPhase(true, new SchedulePhase(GraalOptions.StressTestEarlyReads.getValue(options) ? SchedulingStrategy.EARLIEST : SchedulingStrategy.LATEST_OUT_OF_LOOPS)));

81 
82         appendPhase(canonicalizerWithoutGVN);
83 
84         appendPhase(new UseTrappingNullChecksPhase());
85 
86         appendPhase(new DeadCodeEliminationPhase(Required));
87 
88         appendPhase(new PropagateDeoptimizeProbabilityPhase());
89 
90         appendPhase(new SchedulePhase(SchedulePhase.SchedulingStrategy.FINAL_SCHEDULE));



91     }
92 }

49 public class LowTier extends BaseTier<LowTierContext> {
50 
51     static class Options {
52 
53         // @formatter:off
54         @Option(help = "", type = OptionType.Debug)
55         public static final OptionKey<Boolean> ProfileCompiledMethods = new OptionKey<>(false);
56         // @formatter:on
57 
58     }
59 
60     public LowTier(OptionValues options) {
61         CanonicalizerPhase canonicalizer = createCanonicalizerPhase(options);
62         CanonicalizerPhase canonicalizerWithoutGVN = canonicalizer.copyWithoutGVN();





63 
64         if (Options.ProfileCompiledMethods.getValue(options)) {
65             appendPhase(new ProfileCompiledMethodsPhase());
66         }
67 
68         appendPhase(new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.LOW_TIER));
69 


70         appendPhase(new ExpandLogicPhase());
71 
72         appendPhase(new FixReadsPhase(true,
73                         new SchedulePhase(GraalOptions.StressTestEarlyReads.getValue(options) ? SchedulingStrategy.EARLIEST : SchedulingStrategy.LATEST_OUT_OF_LOOPS_IMPLICIT_NULL_CHECKS)));
74 
75         appendPhase(canonicalizerWithoutGVN);
76 
77         appendPhase(new UseTrappingNullChecksPhase());
78 
79         appendPhase(new DeadCodeEliminationPhase(Required));
80 
81         appendPhase(new PropagateDeoptimizeProbabilityPhase());
82 
83         if (LateMembars.getValue(options)) {
84             appendPhase(new InsertMembarsPhase());
85         }
86         appendPhase(new SchedulePhase(SchedulePhase.SchedulingStrategy.LATEST_OUT_OF_LOOPS));
87     }
