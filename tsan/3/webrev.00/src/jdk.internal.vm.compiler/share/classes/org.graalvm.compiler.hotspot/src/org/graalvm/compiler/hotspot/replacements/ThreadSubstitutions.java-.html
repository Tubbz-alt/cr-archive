44 /**
45  * Substitutions for {@link java.lang.Thread} methods.
46  */
47 @ClassSubstitution(Thread.class)
48 public class ThreadSubstitutions {
49 
50     /**
51      * hidden in 9.
52      */
53     @MethodSubstitution(isStatic = false, optional = true)
54     public static boolean isInterrupted(final Thread thisObject, boolean clearInterrupted) {
55         Word javaThread = CurrentJavaThreadNode.get();
56         Object thread = javaThread.readObject(threadObjectOffset(INJECTED_VMCONFIG), JAVA_THREAD_THREAD_OBJECT_LOCATION);
57         if (thisObject == thread) {
58             Word osThread = javaThread.readWord(osThreadOffset(INJECTED_VMCONFIG), JAVA_THREAD_OSTHREAD_LOCATION);
59             boolean interrupted = osThread.readInt(osThreadInterruptedOffset(INJECTED_VMCONFIG), any()) != 0;
60             if (!interrupted || !clearInterrupted) {
61                 return interrupted;
62             }
63         }
64 
65         return threadIsInterruptedStub(THREAD_IS_INTERRUPTED, thisObject, clearInterrupted);
66     }
67 
68     public static final ForeignCallDescriptor THREAD_IS_INTERRUPTED = new ForeignCallDescriptor("thread_is_interrupted", boolean.class, Thread.class, boolean.class);
69 
70     @NodeIntrinsic(ForeignCallNode.class)
71     private static native boolean threadIsInterruptedStub(@ConstantNodeParameter ForeignCallDescriptor descriptor, Thread thread, boolean clearIsInterrupted);
72 }
