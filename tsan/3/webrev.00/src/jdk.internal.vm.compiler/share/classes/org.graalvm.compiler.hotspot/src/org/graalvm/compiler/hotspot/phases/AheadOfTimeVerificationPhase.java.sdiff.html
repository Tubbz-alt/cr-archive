 38 /**
 39  * Checks for {@link #isIllegalObjectConstant(ConstantNode) illegal} object constants in a graph
 40  * processed for AOT compilation.
 41  *
 42  * @see LoadJavaMirrorWithKlassPhase
 43  */
 44 public class AheadOfTimeVerificationPhase extends VerifyPhase<PhaseContext> {
 45 
 46     @Override
 47     protected boolean verify(StructuredGraph graph, PhaseContext context) {
 48         for (ConstantNode node : getConstantNodes(graph)) {
 49             if (isIllegalObjectConstant(node)) {
 50                 throw new VerificationError("illegal object constant: " + node);
 51             }
 52         }
 53         return true;
 54     }
 55 
 56     public static boolean isIllegalObjectConstant(ConstantNode node) {
 57         return isObject(node) && !isNullReference(node) && !isInternedString(node) && !isDirectMethodHandle(node) && !isBoundMethodHandle(node);





 58     }
 59 
 60     private static boolean isObject(ConstantNode node) {
 61         return node.getStackKind() == JavaKind.Object;
 62     }
 63 
 64     private static boolean isNullReference(ConstantNode node) {
 65         return isObject(node) && node.isNullConstant();
 66     }
 67 
 68     private static boolean isDirectMethodHandle(ConstantNode node) {

 69         if (!isObject(node)) {
 70             return false;
 71         }
 72         return "Ljava/lang/invoke/DirectMethodHandle;".equals(StampTool.typeOrNull(node).getName());











 73     }
 74 
 75     private static boolean isBoundMethodHandle(ConstantNode node) {
 76         if (!isObject(node)) {
 77             return false;
 78         }
 79         return StampTool.typeOrNull(node).getName().startsWith("Ljava/lang/invoke/BoundMethodHandle");
 80     }
 81 








 82     private static boolean isInternedString(ConstantNode node) {
 83         if (!isObject(node)) {
 84             return false;
 85         }
 86 
 87         HotSpotObjectConstant c = (HotSpotObjectConstant) node.asConstant();
 88         return c.isInternedString();
 89     }
 90 }

 27 import static org.graalvm.compiler.nodes.ConstantNode.getConstantNodes;
 28 
 29 import org.graalvm.compiler.nodes.ConstantNode;
 30 import org.graalvm.compiler.nodes.StructuredGraph;
 31 import org.graalvm.compiler.nodes.spi.CoreProviders;
 32 import org.graalvm.compiler.nodes.type.StampTool;
 33 import org.graalvm.compiler.phases.VerifyPhase;

 34 
 35 import jdk.vm.ci.hotspot.HotSpotObjectConstant;
 36 import jdk.vm.ci.meta.JavaKind;
 37 
 38 /**
 39  * Checks for {@link #isIllegalObjectConstant(ConstantNode) illegal} object constants in a graph
 40  * processed for AOT compilation.
 41  *
 42  * @see LoadJavaMirrorWithKlassPhase
 43  */
 44 public class AheadOfTimeVerificationPhase extends VerifyPhase<CoreProviders> {
 45 
 46     @Override
 47     protected void verify(StructuredGraph graph, CoreProviders context) {
 48         for (ConstantNode node : getConstantNodes(graph)) {
 49             if (isIllegalObjectConstant(node)) {
 50                 throw new VerificationError("illegal object constant: " + node);
 51             }
 52         }

 53     }
 54 
 55     public static boolean isIllegalObjectConstant(ConstantNode node) {
 56         return isObject(node) &&
 57                         !isNullReference(node) &&
 58                         !isInternedString(node) &&
 59                         !isDirectMethodHandle(node) &&
 60                         !isBoundMethodHandle(node) &&
 61                         !isVarHandle(node);
 62     }
 63 
 64     private static boolean isObject(ConstantNode node) {
 65         return node.getStackKind() == JavaKind.Object;
 66     }
 67 
 68     private static boolean isNullReference(ConstantNode node) {
 69         return isObject(node) && node.isNullConstant();
 70     }
 71 
 72     private static boolean isDirectMethodHandle(ConstantNode node) {
 73         String typeName = StampTool.typeOrNull(node).getName();
 74         if (!isObject(node)) {
 75             return false;
 76         }
 77 
 78         switch (typeName) {
 79             case "Ljava/lang/invoke/DirectMethodHandle;":
 80             case "Ljava/lang/invoke/DirectMethodHandle$StaticAccessor;":
 81             case "Ljava/lang/invoke/DirectMethodHandle$Accessor;":
 82             case "Ljava/lang/invoke/DirectMethodHandle$Constructor;":
 83             case "Ljava/lang/invoke/DirectMethodHandle$Special;":
 84             case "Ljava/lang/invoke/DirectMethodHandle$Interface;":
 85                 return true;
 86             default:
 87                 return false;
 88         }
 89     }
 90 
 91     private static boolean isBoundMethodHandle(ConstantNode node) {
 92         if (!isObject(node)) {
 93             return false;
 94         }
 95         return StampTool.typeOrNull(node).getName().startsWith("Ljava/lang/invoke/BoundMethodHandle");
 96     }
 97 
 98     private static boolean isVarHandle(ConstantNode node) {
 99         if (!isObject(node)) {
100             return false;
101         }
102         String name = StampTool.typeOrNull(node).getName();
103         return name.equals("Ljava/lang/invoke/VarHandle$AccessDescriptor;");
104     }
105 
106     private static boolean isInternedString(ConstantNode node) {
107         if (!isObject(node)) {
108             return false;
109         }
110 
111         HotSpotObjectConstant c = (HotSpotObjectConstant) node.asConstant();
112         return c.isInternedString();
113     }
114 }
