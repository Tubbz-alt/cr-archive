 51 
 52 public class UnsafeSnippets implements Snippets {
 53 
 54     public static final String copyMemoryName = JavaVersionUtil.JAVA_SPEC <= 8 ? "copyMemory" : "copyMemory0";
 55 
 56     @SuppressWarnings("unused")
 57     @Snippet
 58     static void copyMemory(Object receiver, Object srcBase, long srcOffset, Object destBase, long destOffset, long bytes) {
 59         Word srcAddr = WordFactory.unsigned(ComputeObjectAddressNode.get(srcBase, srcOffset));
 60         Word dstAddr = WordFactory.unsigned(ComputeObjectAddressNode.get(destBase, destOffset));
 61         Word size = WordFactory.signed(bytes);
 62 
 63         HotSpotBackend.unsafeArraycopy(srcAddr, dstAddr, size);
 64     }
 65 
 66     @SuppressWarnings("unused")
 67     @Snippet
 68     static void copyMemoryGuarded(Object receiver, Object srcBase, long srcOffset, Object destBase, long destOffset, long bytes) {
 69         Word srcAddr = WordFactory.unsigned(ComputeObjectAddressNode.get(srcBase, srcOffset));
 70         Word dstAddr = WordFactory.unsigned(ComputeObjectAddressNode.get(destBase, destOffset));
 71         Word size = WordFactory.signed(bytes);
 72         Word javaThread = CurrentJavaThreadNode.get();
 73         int offset = doingUnsafeAccessOffset(INJECTED_VMCONFIG);
 74         LocationIdentity any = LocationIdentity.any();
 75 
 76         /* Set doingUnsafeAccess to guard and handle unsafe memory access failures */
 77         javaThread.writeByte(offset, (byte) 1, any);
 78         HotSpotBackend.unsafeArraycopy(srcAddr, dstAddr, size);
 79         /* Reset doingUnsafeAccess */
 80         javaThread.writeByte(offset, (byte) 0, any);
 81     }
 82 
 83     public static class Templates extends AbstractTemplates {
 84         private final SnippetInfo copyMemory = snippet(UnsafeSnippets.class, "copyMemory");
 85         private final SnippetInfo copyMemoryGuarded = snippet(UnsafeSnippets.class, "copyMemoryGuarded");
 86 
 87         public Templates(OptionValues options, Iterable<DebugHandlersFactory> factories, HotSpotProviders providers, TargetDescription target) {
 88             super(options, factories, providers, providers.getSnippetReflection(), target);
 89         }
 90 
 91         public void lower(UnsafeCopyMemoryNode copyMemoryNode, LoweringTool tool) {
 92             StructuredGraph graph = copyMemoryNode.graph();
 93             Arguments args = new Arguments(copyMemoryNode.isGuarded() ? copyMemoryGuarded : copyMemory, graph.getGuardsStage(), tool.getLoweringStage());
 94             args.add("receiver", copyMemoryNode.receiver);
 95             args.add("srcBase", copyMemoryNode.srcBase);
 96             args.add("srcOffset", copyMemoryNode.srcOffset);
 97             args.add("destBase", copyMemoryNode.destBase);
 98             args.add("destOffset", copyMemoryNode.desOffset);
 99             args.add("bytes", copyMemoryNode.bytes);
100             SnippetTemplate template = template(copyMemoryNode, args);
101             template.instantiate(providers.getMetaAccess(), copyMemoryNode, DEFAULT_REPLACER, args);
102         }
103     }
