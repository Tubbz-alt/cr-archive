25 package org.graalvm.compiler.hotspot.gc.shared;
26 
27 import org.graalvm.compiler.graph.NodeClass;
28 import org.graalvm.compiler.hotspot.nodes.WriteBarrier;
29 import org.graalvm.compiler.nodeinfo.InputType;
30 import org.graalvm.compiler.nodeinfo.NodeInfo;
31 import org.graalvm.compiler.nodes.ValueNode;
32 import org.graalvm.compiler.nodes.memory.address.AddressNode;
33 
34 @NodeInfo
35 public abstract class ObjectWriteBarrier extends WriteBarrier {
36 
37     public static final NodeClass<ObjectWriteBarrier> TYPE = NodeClass.create(ObjectWriteBarrier.class);
38     @Input(InputType.Association) protected AddressNode address;
39     @OptionalInput protected ValueNode value;
40     protected final boolean precise;
41 
42     protected ObjectWriteBarrier(NodeClass<? extends ObjectWriteBarrier> c, AddressNode address, ValueNode value, boolean precise) {
43         super(c);
44         this.address = address;
45         this.value = value;
46         this.precise = precise;
47     }
48 
49     public ValueNode getValue() {
50         return value;
51     }
52 
53     public AddressNode getAddress() {
54         return address;
55     }
56 
57     public boolean usePrecise() {
58         return precise;
59     }
60 }
