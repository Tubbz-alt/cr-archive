26 package org.graalvm.compiler.hotspot.meta;
27 
28 import java.util.ListIterator;
29 
30 import org.graalvm.compiler.hotspot.GraalHotSpotVMConfig;
31 import org.graalvm.compiler.hotspot.HotSpotGraalRuntimeProvider;
32 import org.graalvm.compiler.options.OptionValues;
33 import org.graalvm.compiler.phases.BasePhase;
34 import org.graalvm.compiler.phases.Phase;
35 import org.graalvm.compiler.phases.common.ExpandLogicPhase;
36 import org.graalvm.compiler.phases.common.FixReadsPhase;
37 import org.graalvm.compiler.phases.tiers.LowTierContext;
38 import org.graalvm.compiler.phases.tiers.Suites;
39 import org.graalvm.compiler.phases.tiers.SuitesCreator;
40 
41 /**
42  * Subclass to factor out management of address lowering.
43  */
44 public class AddressLoweringHotSpotSuitesProvider extends HotSpotSuitesProvider {
45 
46     private final Phase addressLowering;
47 
48     public AddressLoweringHotSpotSuitesProvider(SuitesCreator defaultSuitesCreator, GraalHotSpotVMConfig config, HotSpotGraalRuntimeProvider runtime,
49                     Phase addressLowering) {
50         super(defaultSuitesCreator, config, runtime);
51         this.addressLowering = addressLowering;
52     }
53 
54     @Override
55     public Suites createSuites(OptionValues options) {
56         Suites suites = super.createSuites(options);
57 
58         ListIterator<BasePhase<? super LowTierContext>> findPhase = suites.getLowTier().findPhase(FixReadsPhase.class);
59         if (findPhase == null) {
60             findPhase = suites.getLowTier().findPhase(ExpandLogicPhase.class);
61         }

62         findPhase.add(addressLowering);
63 
64         return suites;
65     }
66 }
