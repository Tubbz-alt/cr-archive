47 /**
48  * Causes the VM to exit with a description of the current Java location and an optional printf
49  * formatted error message specified.
50  */
51 @NodeInfo(cycles = CYCLES_UNKNOWN, size = SIZE_UNKNOWN)
52 public final class VMErrorNode extends DeoptimizingStubCall implements LIRLowerable {
53 
54     public static final NodeClass<VMErrorNode> TYPE = NodeClass.create(VMErrorNode.class);
55     protected final String format;
56     @Input ValueNode value;
57 
58     public VMErrorNode(String format, ValueNode value) {
59         super(TYPE, StampFactory.forVoid());
60         this.format = format;
61         this.value = value;
62     }
63 
64     @Override
65     public void generate(NodeLIRBuilderTool gen) {
66         String whereString;
67         if (stateBefore() != null) {
68             String nl = CodeUtil.NEW_LINE;
69             StringBuilder sb = new StringBuilder("in compiled code associated with frame state:");
70             FrameState fs = stateBefore();
71             while (fs != null) {
72                 Bytecode.appendLocation(sb.append(nl).append("\t"), fs.getCode(), fs.bci);
73                 fs = fs.outerFrameState();
74             }
75             whereString = sb.toString();
76         } else {
77             ResolvedJavaMethod method = graph().method();
78             whereString = "in compiled code for " + (method == null ? graph().toString() : method.format("%H.%n(%p)"));
79         }
80 
81         LIRKind wordKind = gen.getLIRGeneratorTool().getLIRKind(StampFactory.pointer());
82         Value whereArg = gen.getLIRGeneratorTool().emitConstant(wordKind, new CStringConstant(whereString));
83         Value formatArg = gen.getLIRGeneratorTool().emitConstant(wordKind, new CStringConstant(format));
84 
85         ForeignCallLinkage linkage = gen.getLIRGeneratorTool().getForeignCalls().lookupForeignCall(VM_ERROR);
86         gen.getLIRGeneratorTool().emitForeignCall(linkage, null, whereArg, formatArg, gen.operand(value));
87     }
88 
89     @NodeIntrinsic
90     public static native void vmError(@ConstantNodeParameter String format, long value);
91 }
