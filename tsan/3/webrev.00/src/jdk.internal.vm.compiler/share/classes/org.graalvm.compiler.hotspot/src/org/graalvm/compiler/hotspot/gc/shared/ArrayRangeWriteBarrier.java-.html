25 package org.graalvm.compiler.hotspot.gc.shared;
26 
27 import org.graalvm.compiler.graph.NodeClass;
28 import org.graalvm.compiler.hotspot.nodes.WriteBarrier;
29 import org.graalvm.compiler.nodeinfo.InputType;
30 import org.graalvm.compiler.nodeinfo.NodeInfo;
31 import org.graalvm.compiler.nodes.ValueNode;
32 import org.graalvm.compiler.nodes.memory.address.AddressNode;
33 import org.graalvm.compiler.nodes.spi.Lowerable;
34 
35 @NodeInfo
36 public abstract class ArrayRangeWriteBarrier extends WriteBarrier implements Lowerable {
37 
38     public static final NodeClass<ArrayRangeWriteBarrier> TYPE = NodeClass.create(ArrayRangeWriteBarrier.class);
39     @Input(InputType.Association) AddressNode address;
40     @Input ValueNode length;
41 
42     private final int elementStride;
43 
44     protected ArrayRangeWriteBarrier(NodeClass<? extends ArrayRangeWriteBarrier> c, AddressNode address, ValueNode length, int elementStride) {
45         super(c);
46         this.address = address;
47         this.length = length;
48         this.elementStride = elementStride;
49     }
50 
51     public AddressNode getAddress() {
52         return address;
53     }
54 
55     public ValueNode getLength() {
56         return length;
57     }
58 
59     public int getElementStride() {
60         return elementStride;
61     }
62 }
