27 import jdk.vm.ci.hotspot.HotSpotJVMCICompilerFactory;

29 /**
30  * Determines if a given class is a JVMCI or Graal class for the purpose of
31  * {@link HotSpotGraalCompilerFactory.Options#CompileGraalWithC1Only}.
32  */
33 public class IsGraalPredicate {
34     /**
35      * Module containing {@link HotSpotJVMCICompilerFactory}.
36      */
37     private final Module jvmciModule;
38 
39     /**
40      * Module containing {@link HotSpotGraalCompilerFactory}.
41      */
42     private final Module graalModule;
43 
44     /**
45      * Module containing the {@linkplain CompilerConfigurationFactory#selectFactory selected}
46      * configuration.
47      */
48     private Module compilerConfigurationModule;
49 
50     public IsGraalPredicate() {
51         jvmciModule = HotSpotJVMCICompilerFactory.class.getModule();
52         graalModule = HotSpotGraalCompilerFactory.class.getModule();
53     }
54 
55     void onCompilerConfigurationFactorySelection(CompilerConfigurationFactory factory) {

56         compilerConfigurationModule = factory.getClass().getModule();

57     }
58 

59     boolean apply(Class<?> declaringClass) {
60         Module module = declaringClass.getModule();
61         return jvmciModule == module || graalModule == module || compilerConfigurationModule == module;
62     }






63 }

34 /**
35  * Determines if a given class is a JVMCI or Graal class for the purpose of
36  * {@link HotSpotGraalCompilerFactory.Options#CompileGraalWithC1Only}.
37  */
38 class IsGraalPredicate extends IsGraalPredicateBase {
39     /**
40      * Module containing {@link HotSpotJVMCICompilerFactory}.
41      */
42     private final Module jvmciModule;
43 
44     /**
45      * Module containing {@link HotSpotGraalCompilerFactory}.
46      */
47     private final Module graalModule;
48 
49     /**
50      * Module containing the {@linkplain CompilerConfigurationFactory#selectFactory selected}
51      * configuration.
52      */
53     private Module compilerConfigurationModule;
54 
55     IsGraalPredicate() {
56         jvmciModule = HotSpotJVMCICompilerFactory.class.getModule();
57         graalModule = HotSpotGraalCompilerFactory.class.getModule();
58     }
59 
60     @Override
61     void onCompilerConfigurationFactorySelection(HotSpotJVMCIRuntime runtime, CompilerConfigurationFactory factory) {
62         compilerConfigurationModule = factory.getClass().getModule();
63         runtime.excludeFromJVMCICompilation(jvmciModule, graalModule, compilerConfigurationModule);
64     }
65 
66     @Override
67     boolean apply(Class<?> declaringClass) {
68         throw GraalError.shouldNotReachHere();

69     }
70 
71     @Override
72     HotSpotJVMCICompilerFactory.CompilationLevelAdjustment getCompilationLevelAdjustment() {
73         return None;
74     }
75 
76 }
