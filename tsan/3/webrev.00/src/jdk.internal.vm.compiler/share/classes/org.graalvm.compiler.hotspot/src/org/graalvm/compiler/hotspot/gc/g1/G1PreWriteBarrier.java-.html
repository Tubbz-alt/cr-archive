26 package org.graalvm.compiler.hotspot.gc.g1;
27 
28 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_64;
29 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_64;
30 
31 import org.graalvm.compiler.graph.NodeClass;
32 import org.graalvm.compiler.hotspot.gc.shared.ObjectWriteBarrier;
33 import org.graalvm.compiler.nodeinfo.InputType;
34 import org.graalvm.compiler.nodeinfo.NodeInfo;
35 import org.graalvm.compiler.nodes.DeoptimizingNode;
36 import org.graalvm.compiler.nodes.FrameState;
37 import org.graalvm.compiler.nodes.ValueNode;
38 import org.graalvm.compiler.nodes.memory.address.AddressNode;
39 
40 @NodeInfo(cycles = CYCLES_64, size = SIZE_64)
41 public final class G1PreWriteBarrier extends ObjectWriteBarrier implements DeoptimizingNode.DeoptBefore {
42 
43     public static final NodeClass<G1PreWriteBarrier> TYPE = NodeClass.create(G1PreWriteBarrier.class);
44 
45     @OptionalInput(InputType.State) private FrameState stateBefore;
46     private final boolean nullCheck;
47     private final boolean doLoad;
48 
49     public G1PreWriteBarrier(AddressNode address, ValueNode expectedObject, boolean doLoad, boolean nullCheck) {
50         super(TYPE, address, expectedObject, true);
51         this.doLoad = doLoad;
52         this.nullCheck = nullCheck;
53     }
54 
55     public ValueNode getExpectedObject() {
56         return getValue();
57     }
58 
59     public boolean doLoad() {
60         return doLoad;
61     }
62 
63     public boolean getNullCheck() {
64         return nullCheck;
65     }
66 
67     @Override
68     public boolean canDeoptimize() {
69         return nullCheck;
70     }
71 
72     @Override
73     public FrameState stateBefore() {
74         return stateBefore;
75     }
76 
77     @Override
78     public void setStateBefore(FrameState state) {
79         updateUsages(stateBefore, state);
80         stateBefore = state;
81     }
82 }
