25 package org.graalvm.compiler.hotspot.phases;
26 
27 import static org.graalvm.compiler.nodes.ConstantNode.getConstantNodes;
28 
29 import org.graalvm.compiler.nodes.ConstantNode;
30 import org.graalvm.compiler.nodes.StructuredGraph;
31 import org.graalvm.compiler.nodes.type.StampTool;
32 import org.graalvm.compiler.phases.VerifyPhase;
33 import org.graalvm.compiler.phases.tiers.PhaseContext;
34 
35 import jdk.vm.ci.hotspot.HotSpotObjectConstant;
36 import jdk.vm.ci.meta.JavaKind;
37 
38 /**
39  * Checks for {@link #isIllegalObjectConstant(ConstantNode) illegal} object constants in a graph
40  * processed for AOT compilation.
41  *
42  * @see LoadJavaMirrorWithKlassPhase
43  */
44 public class AheadOfTimeVerificationPhase extends VerifyPhase<PhaseContext> {
45 
46     @Override
47     protected boolean verify(StructuredGraph graph, PhaseContext context) {
48         for (ConstantNode node : getConstantNodes(graph)) {
49             if (isIllegalObjectConstant(node)) {
50                 throw new VerificationError("illegal object constant: " + node);
51             }
52         }
53         return true;
54     }
55 
56     public static boolean isIllegalObjectConstant(ConstantNode node) {
57         return isObject(node) && !isNullReference(node) && !isInternedString(node) && !isDirectMethodHandle(node) && !isBoundMethodHandle(node);
58     }
59 
60     private static boolean isObject(ConstantNode node) {
61         return node.getStackKind() == JavaKind.Object;
62     }
63 
64     private static boolean isNullReference(ConstantNode node) {
65         return isObject(node) && node.isNullConstant();
66     }
67 
68     private static boolean isDirectMethodHandle(ConstantNode node) {
69         if (!isObject(node)) {
70             return false;
71         }
72         return "Ljava/lang/invoke/DirectMethodHandle;".equals(StampTool.typeOrNull(node).getName());
73     }
74 
75     private static boolean isBoundMethodHandle(ConstantNode node) {
76         if (!isObject(node)) {
77             return false;
78         }
79         return StampTool.typeOrNull(node).getName().startsWith("Ljava/lang/invoke/BoundMethodHandle");
80     }
81 
82     private static boolean isInternedString(ConstantNode node) {
83         if (!isObject(node)) {
84             return false;
85         }
86 
87         HotSpotObjectConstant c = (HotSpotObjectConstant) node.asConstant();
88         return c.isInternedString();
89     }
90 }
