 51 /**
 52  * Base class for a stub defined by a snippet.
 53  */
 54 public abstract class SnippetStub extends Stub implements Snippets {
 55 
 56     protected final ResolvedJavaMethod method;
 57 
 58     /**
 59      * Creates a new snippet stub.
 60      *
 61      * @param snippetMethodName name of the single {@link Snippet} annotated method in the class of
 62      *            this object
 63      * @param linkage linkage details for a call to the stub
 64      */
 65     public SnippetStub(String snippetMethodName, OptionValues options, HotSpotProviders providers, HotSpotForeignCallLinkage linkage) {
 66         this(null, snippetMethodName, options, providers, linkage);
 67     }
 68 
 69     /**
 70      * Creates a new snippet stub.
 71      *
 72      * @param snippetDeclaringClass this class in which the {@link Snippet} annotated method is
 73      *            declared. If {@code null}, this the class of this object is used.
 74      * @param snippetMethodName name of the single {@link Snippet} annotated method in
 75      *            {@code snippetDeclaringClass}
 76      * @param linkage linkage details for a call to the stub
 77      */
 78     public SnippetStub(Class<? extends Snippets> snippetDeclaringClass, String snippetMethodName, OptionValues options, HotSpotProviders providers, HotSpotForeignCallLinkage linkage) {
 79         super(options, providers, linkage);
 80         this.method = SnippetTemplate.AbstractTemplates.findMethod(providers.getMetaAccess(), snippetDeclaringClass == null ? getClass() : snippetDeclaringClass, snippetMethodName);
 81         registerSnippet();
 82     }
 83 
 84     protected void registerSnippet() {
 85         providers.getReplacements().registerSnippet(method, null, null, false, options);
 86     }
 87 
 88     @Override
 89     @SuppressWarnings("try")
 90     protected StructuredGraph getGraph(DebugContext debug, CompilationIdentifier compilationId) {
 91         // Stubs cannot have optimistic assumptions since they have
 92         // to be valid for the entire run of the VM.
 93         final StructuredGraph graph = buildInitialGraph(debug, compilationId, makeConstArgs());
 94         try (DebugContext.Scope outer = debug.scope("SnippetStub", graph)) {
 95             for (ParameterNode param : graph.getNodes(ParameterNode.TYPE)) {
 96                 int index = param.index();
 97                 if (method.getParameterAnnotation(NonNullParameter.class, index) != null) {
 98                     param.setStamp(param.stamp(NodeView.DEFAULT).join(StampFactory.objectNonNull()));
 99                 }
100             }
101 
102             graph.setGuardsStage(GuardsStage.FLOATING_GUARDS);
103             CanonicalizerPhase canonicalizer = CanonicalizerPhase.create();
104             canonicalizer.apply(graph, providers);
105             new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.HIGH_TIER).apply(graph, providers);
106         } catch (Throwable e) {
107             throw debug.handle(e);
108         }
109 
110         return graph;
111     }
112 
113     protected StructuredGraph buildInitialGraph(DebugContext debug, CompilationIdentifier compilationId, Object[] args) {
114         return providers.getReplacements().getSnippet(method, null, args, false, null, options).copyWithIdentifier(compilationId, debug);
115     }
116 
117     protected boolean checkConstArg(int index, String expectedName) {
118         assert method.getParameterAnnotation(ConstantParameter.class, index) != null : String.format("parameter %d of %s is expected to be constant", index, method.format("%H.%n(%p)"));
119         LocalVariableTable lvt = method.getLocalVariableTable();
120         if (lvt != null) {
121             Local local = lvt.getLocal(index, 0);
122             assert local != null;
123             String actualName = local.getName();
124             assert actualName.equals(expectedName) : String.format("parameter %d of %s is expected to be named %s, not %s", index, method.format("%H.%n(%p)"), expectedName, actualName);
125         }
126         return true;
127     }
128 
129     protected Object[] makeConstArgs() {
130         int count = method.getSignature().getParameterCount(false);
131         Object[] args = new Object[count];
132         for (int i = 0; i < args.length; i++) {
133             if (method.getParameterAnnotation(ConstantParameter.class, i) != null) {
134                 args[i] = getConstantParameterValue(i, null);
135             }
136         }
137         return args;
138     }
139 
140     protected Object getConstantParameterValue(int index, String name) {
141         throw new GraalError("%s must override getConstantParameterValue() to provide a value for parameter %d%s", getClass().getName(), index, name == null ? "" : " (" + name + ")");
142     }
143 
144     @Override
145     protected Object debugScopeContext() {
146         return getInstalledCodeOwner();
147     }
148 
149     @Override
150     public ResolvedJavaMethod getInstalledCodeOwner() {
151         return method;
152     }
153 
154     @Override
155     public String toString() {
156         return "Stub<" + getInstalledCodeOwner().format("%h.%n") + ">";
157     }
158 
159     public ResolvedJavaMethod getMethod() {
160         return method;
161     }
162 }
