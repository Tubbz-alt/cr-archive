25 package org.graalvm.compiler.hotspot.meta;
26 
27 import org.graalvm.compiler.core.common.type.AbstractPointerStamp;
28 import org.graalvm.compiler.core.common.type.ObjectStamp;

29 import org.graalvm.compiler.hotspot.nodes.type.KlassPointerStamp;
30 import org.graalvm.compiler.hotspot.nodes.type.MethodPointerStamp;
31 import org.graalvm.compiler.nodes.spi.StampProvider;
32 
33 public class HotSpotStampProvider implements StampProvider {
34 
35     @Override
36     public AbstractPointerStamp createHubStamp(ObjectStamp object) {
37         return KlassPointerStamp.klassNonNull();
38     }
39 
40     @Override
41     public AbstractPointerStamp createMethodStamp() {
42         return MethodPointerStamp.methodNonNull();
43     }
44 }
