25 package org.graalvm.compiler.hotspot;
26 
27 import jdk.vm.ci.hotspot.HotSpotJVMCICompilerFactory;
28 import jdk.vm.ci.hotspot.HotSpotJVMCIRuntime;
29 
30 /**
31  * Determines if a given class is a JVMCI or Graal class for the purpose of
32  * {@link HotSpotGraalCompilerFactory.Options#CompileGraalWithC1Only}.
33  */
34 abstract class IsGraalPredicateBase {
35 
36     @SuppressWarnings("unused")
37     void onCompilerConfigurationFactorySelection(HotSpotJVMCIRuntime runtime, CompilerConfigurationFactory factory) {
38     }
39 
40     abstract boolean apply(Class<?> declaringClass);
41 
42     HotSpotJVMCICompilerFactory.CompilationLevelAdjustment getCompilationLevelAdjustment() {
43         return HotSpotJVMCICompilerFactory.CompilationLevelAdjustment.ByHolder;
44     }
45 }
