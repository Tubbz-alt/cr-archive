28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_1;

30 import org.graalvm.compiler.core.common.type.StampFactory;
31 import org.graalvm.compiler.graph.NodeClass;

36 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
37 import org.graalvm.compiler.word.Word;
38 import org.graalvm.compiler.word.WordTypes;
39 

40 import jdk.vm.ci.meta.Value;
41 
42 /**
43  * Reserves a block of memory in the stack frame of a method. The block is reserved in the frame for
44  * the entire execution of the associated method.
45  */
46 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
47 public final class AllocaNode extends FixedWithNextNode implements LIRLowerable {
48 
49     public static final NodeClass<AllocaNode> TYPE = NodeClass.create(AllocaNode.class);
50     /**
51      * The number of slots in block.
52      */
53     protected final int slots;
54 







55     public AllocaNode(@InjectedNodeParameter WordTypes wordTypes, int slots) {
56         super(TYPE, StampFactory.forKind(wordTypes.getWordKind()));




57         this.slots = slots;

58     }
59 
60     @Override
61     public void generate(NodeLIRBuilderTool gen) {
62         VirtualStackSlot array = gen.getLIRGeneratorTool().allocateStackSlots(slots);
63         Value result = gen.getLIRGeneratorTool().emitAddress(array);
64         gen.setResult(this, result);
65     }
66 
67     @NodeIntrinsic
68     public static native Word alloca(@ConstantNodeParameter int slots);
69 }
