25 package org.graalvm.compiler.hotspot;
26 
27 /**
28  * Communicates with an MBean providing a JMX interface to a {@link HotSpotGraalRuntime} instance.
29  * The MBean will be dynamically created when a JMX client attaches or some other event causes the
30  * platform MBean server to be started.
31  */
32 public interface HotSpotGraalManagementRegistration {
33 
34     /**
35      * Completes the initialization of this registration by recording the
36      * {@link HotSpotGraalRuntime} the MBean will provide an JMX interface to.
37      */
38     void initialize(HotSpotGraalRuntime runtime);
39 
40     /**
41      * Polls this registration to see if the MBean is registered in a MBean server.
42      *
43      * @param sync synchronize with other threads that may be processing this registration. This is
44      *            useful when the caller knows the server is active (e.g., it has a reference to
45      *            server) and expects this poll to therefore return a non-null value.
46      * @return an {@link javax.management.ObjectName} that can be used to access the MBean or
47      *         {@code null} if the MBean has not been registered with an MBean server (e.g., no JMX
48      *         client has attached to the VM)
49      */
50     Object poll(boolean sync);
51 }
