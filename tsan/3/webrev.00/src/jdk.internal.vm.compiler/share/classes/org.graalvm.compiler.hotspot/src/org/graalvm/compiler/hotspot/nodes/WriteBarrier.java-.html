25 package org.graalvm.compiler.hotspot.nodes;
26 
27 import org.graalvm.compiler.core.common.type.StampFactory;
28 import org.graalvm.compiler.graph.NodeClass;
29 import org.graalvm.compiler.nodeinfo.NodeInfo;
30 import org.graalvm.compiler.nodes.FixedWithNextNode;
31 import org.graalvm.compiler.nodes.spi.Lowerable;
32 import org.graalvm.compiler.nodes.spi.LoweringTool;
33 
34 @NodeInfo
35 public abstract class WriteBarrier extends FixedWithNextNode implements Lowerable {
36 
37     public static final NodeClass<WriteBarrier> TYPE = NodeClass.create(WriteBarrier.class);
38 
39     protected WriteBarrier(NodeClass<? extends WriteBarrier> c) {
40         super(c, StampFactory.forVoid());
41     }
42 
43     @Override
44     public void lower(LoweringTool tool) {
45         assert graph().getGuardsStage().areFrameStatesAtDeopts();
46         tool.getLowerer().lower(this, tool);
47     }
48 }
