 50 /**
 51  * Extends {@link DebugInfoBuilder} to allocate the extra debug information required for locks.
 52  */
 53 public class HotSpotDebugInfoBuilder extends DebugInfoBuilder {
 54 
 55     private final HotSpotLockStack lockStack;
 56 
 57     private int maxInterpreterFrameSize;
 58 
 59     private HotSpotCodeCacheProvider codeCacheProvider;
 60 
 61     public HotSpotDebugInfoBuilder(NodeValueMap nodeValueMap, HotSpotLockStack lockStack, HotSpotLIRGenerator gen) {
 62         super(nodeValueMap, gen.getResult().getLIR().getDebug());
 63         this.lockStack = lockStack;
 64         this.codeCacheProvider = gen.getProviders().getCodeCache();
 65     }
 66 
 67     public HotSpotLockStack lockStack() {
 68         return lockStack;
 69     }
 70 
 71     public int maxInterpreterFrameSize() {
 72         return maxInterpreterFrameSize;
 73     }
 74 
 75     @Override
 76     protected JavaValue computeLockValue(FrameState state, int lockIndex) {
 77         int lockDepth = lockIndex;
 78         if (state.outerFrameState() != null) {
 79             lockDepth += state.outerFrameState().nestedLockDepth();
 80         }
 81         VirtualStackSlot slot = lockStack.makeLockSlot(lockDepth);
 82         ValueNode lock = state.lockAt(lockIndex);
 83         JavaValue object = toJavaValue(lock);
 84         boolean eliminated = object instanceof VirtualObject || state.monitorIdAt(lockIndex).isEliminated();
 85         assert state.monitorIdAt(lockIndex).getLockDepth() == lockDepth;
 86         return new StackLockValue(object, slot, eliminated);
 87     }
 88 
 89     @Override
 90     protected BytecodeFrame computeFrameForState(FrameState state) {
 91         if (isPlaceholderBci(state.bci) && state.bci != BytecodeFrame.BEFORE_BCI) {
 92             raiseInvalidFrameStateError(state);
 93         }
 94         BytecodeFrame result = super.computeFrameForState(state);
 95         maxInterpreterFrameSize = Math.max(maxInterpreterFrameSize, codeCacheProvider.interpreterFrameSize(result));
 96         return result;
 97     }
 98 
 99     protected void raiseInvalidFrameStateError(FrameState state) throws GraalGraphError {
100         // This is a hard error since an incorrect state could crash hotspot
101         NodeSourcePosition sourcePosition = state.getNodeSourcePosition();
102         List<String> context = new ArrayList<>();
103         ResolvedJavaMethod replacementMethodWithProblematicSideEffect = null;
104         if (sourcePosition != null) {
105             NodeSourcePosition pos = sourcePosition;
106             while (pos != null) {
107                 StringBuilder sb = new StringBuilder("parsing ");
108                 ResolvedJavaMethod method = pos.getMethod();
109                 MetaUtil.appendLocation(sb, method, pos.getBCI());
110                 if (method.getAnnotation(MethodSubstitution.class) != null ||
111                                 method.getAnnotation(Snippet.class) != null) {
112                     replacementMethodWithProblematicSideEffect = method;
113                 }
114                 context.add(sb.toString());
115                 pos = pos.getCaller();
116             }
117         }
118         String message = "Invalid frame state " + state;
119         if (replacementMethodWithProblematicSideEffect != null) {
120             message += " associated with a side effect in " + replacementMethodWithProblematicSideEffect.format("%H.%n(%p)") + " at a position that cannot be deoptimized to";
121         }
122         GraalGraphError error = new GraalGraphError(message);
123         for (String c : context) {
124             error.addContext(c);
125         }
126         throw error;
127     }
128 }
