27 import static org.graalvm.compiler.hotspot.GraalHotSpotVMConfig.INJECTED_VMCONFIG;
28 import static org.graalvm.compiler.hotspot.replacements.HotSpotReplacementsUtil.JAVA_THREAD_OSTHREAD_LOCATION;
29 import static org.graalvm.compiler.hotspot.replacements.HotSpotReplacementsUtil.JAVA_THREAD_THREAD_OBJECT_LOCATION;
30 import static org.graalvm.compiler.hotspot.replacements.HotSpotReplacementsUtil.osThreadInterruptedOffset;
31 import static org.graalvm.compiler.hotspot.replacements.HotSpotReplacementsUtil.osThreadOffset;
32 import static org.graalvm.compiler.hotspot.replacements.HotSpotReplacementsUtil.threadObjectOffset;
33 import static jdk.internal.vm.compiler.word.LocationIdentity.any;
34 
35 import org.graalvm.compiler.api.replacements.ClassSubstitution;
36 import org.graalvm.compiler.api.replacements.MethodSubstitution;
37 import org.graalvm.compiler.hotspot.nodes.CurrentJavaThreadNode;
38 import org.graalvm.compiler.word.Word;
39 
40 /**
41  * Substitutions for {@link java.lang.Thread} methods.
42  */
43 @ClassSubstitution(Thread.class)
44 public class ThreadSubstitutions {
45 
46     /**
47      * hidden in 9.
48      */
49     @MethodSubstitution(isStatic = false, optional = true)
50     public static boolean isInterrupted(final Thread thisObject, boolean clearInterrupted) {
51         Word javaThread = CurrentJavaThreadNode.get();
52         Object thread = javaThread.readObject(threadObjectOffset(INJECTED_VMCONFIG), JAVA_THREAD_THREAD_OBJECT_LOCATION);
53         if (thisObject == thread) {
54             Word osThread = javaThread.readWord(osThreadOffset(INJECTED_VMCONFIG), JAVA_THREAD_OSTHREAD_LOCATION);
55             boolean interrupted = osThread.readInt(osThreadInterruptedOffset(INJECTED_VMCONFIG), any()) != 0;
56             if (!interrupted || !clearInterrupted) {
57                 return interrupted;
58             }
59         }
60         return isInterrupted(thisObject, clearInterrupted);
61     }
62 }
