 56 
 57 public class ObjectSnippets implements Snippets {
 58 
 59     @NodeIntrinsic(ForeignCallNode.class)
 60     public static native boolean fastNotifyStub(@ConstantNodeParameter ForeignCallDescriptor descriptor, Object o);
 61 
 62     @Snippet
 63     public static void fastNotify(Object thisObj) {
 64         if (fastNotifyStub(HotSpotHostForeignCallsProvider.NOTIFY, thisObj)) {
 65             return;
 66         } else {
 67             PiNode.piCastNonNull(thisObj, SnippetAnchorNode.anchor()).notify();
 68         }
 69     }
 70 
 71     @Snippet
 72     public static void fastNotifyAll(Object thisObj) {
 73         if (fastNotifyStub(HotSpotHostForeignCallsProvider.NOTIFY_ALL, thisObj)) {
 74             return;
 75         } else {
 76             PiNode.piCastNonNull(thisObj, SnippetAnchorNode.anchor()).notifyAll();
 77         }
 78     }
 79 
 80     public static class Templates extends AbstractTemplates {
 81         private final SnippetInfo notifySnippet = snippet(ObjectSnippets.class, "fastNotify", originalNotifyCall(false), null);
 82         private final SnippetInfo notifyAllSnippet = snippet(ObjectSnippets.class, "fastNotifyAll", originalNotifyCall(true), null);
 83 
 84         public Templates(OptionValues options, Iterable<DebugHandlersFactory> factories, HotSpotProviders providers, TargetDescription target) {
 85             super(options, factories, providers, providers.getSnippetReflection(), target);
 86         }
 87 
 88         private ResolvedJavaMethod originalNotifyCall(boolean notifyAll) throws GraalError {
 89             if (notifyAll) {
 90                 return findMethod(providers.getMetaAccess(), Object.class, "notifyAll");
 91             } else {
 92                 return findMethod(providers.getMetaAccess(), Object.class, "notify");
 93             }
 94         }
 95 
 96         public void lower(Node n, LoweringTool tool) {
 97             if (n instanceof FastNotifyNode) {
 98                 FastNotifyNode fn = (FastNotifyNode) n;
 99                 StructuredGraph graph = (StructuredGraph) n.graph();
100                 FrameState stateDuringCall = fn.stateDuring();
101                 assert stateDuringCall != null : "Must have valid state for snippet recursive notify call";
102                 Arguments args = new Arguments(fn.isNotifyAll() ? notifyAllSnippet : notifySnippet, graph.getGuardsStage(), tool.getLoweringStage());
103                 args.add("thisObj", fn.object);
104                 SnippetTemplate template = template(fn, args);
105                 graph.getDebug().log("Lowering fast notify in %s: node=%s, template=%s, arguments=%s", graph, fn, template, args);
106                 UnmodifiableEconomicMap<Node, Node> replacements = template.instantiate(providers.getMetaAccess(), fn, DEFAULT_REPLACER, args);
107                 for (Node originalNode : replacements.getKeys()) {
108                     if (originalNode instanceof InvokeNode) {
109                         InvokeNode invoke = (InvokeNode) replacements.get(originalNode);
110                         assert invoke.asNode().graph() == graph;
111                         // Here we need to fix the bci of the invoke
112                         invoke.replaceBci(fn.getBci());
113                         invoke.setStateDuring(null);
114                         invoke.setStateAfter(null);
115                         invoke.setStateDuring(stateDuringCall);
116                     } else if (originalNode instanceof InvokeWithExceptionNode) {
117                         throw new GraalError("unexpected invoke with exception %s in snippet", originalNode);
118                     }
119                 }
120             } else {
121                 GraalError.shouldNotReachHere("Unknown object snippet lowered node");
122             }
123         }
124 
125     }
126 }
