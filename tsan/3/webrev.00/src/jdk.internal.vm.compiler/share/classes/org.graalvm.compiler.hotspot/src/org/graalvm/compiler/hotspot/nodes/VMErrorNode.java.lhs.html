48 /**
49  * Causes the VM to exit with a description of the current Java location and an optional
50  * {@linkplain Log#printf(String, long) formatted} error message specified.
51  */
52 @NodeInfo(cycles = CYCLES_UNKNOWN, size = SIZE_UNKNOWN)
53 public final class VMErrorNode extends DeoptimizingStubCall implements LIRLowerable {
54 
55     public static final NodeClass<VMErrorNode> TYPE = NodeClass.create(VMErrorNode.class);
56     protected final String format;
57     @Input ValueNode value;
58 
59     public VMErrorNode(String format, ValueNode value) {
60         super(TYPE, StampFactory.forVoid());
61         this.format = format;
62         this.value = value;
63     }
64 
65     @Override
66     public void generate(NodeLIRBuilderTool gen) {
67         String whereString;
68         if (stateBefore() != null) {
69             String nl = CodeUtil.NEW_LINE;
70             StringBuilder sb = new StringBuilder("in compiled code associated with frame state:");
71             FrameState fs = stateBefore();
72             while (fs != null) {
73                 Bytecode.appendLocation(sb.append(nl).append("\t"), fs.getCode(), fs.bci);
74                 fs = fs.outerFrameState();
75             }
76             whereString = sb.toString();
77         } else {
78             ResolvedJavaMethod method = graph().method();
79             whereString = "in compiled code for " + (method == null ? graph().toString() : method.format("%H.%n(%p)"));
80         }
81 
82         LIRKind wordKind = gen.getLIRGeneratorTool().getLIRKind(StampFactory.pointer());
83         Value whereArg = gen.getLIRGeneratorTool().emitConstant(wordKind, new CStringConstant(whereString));
84         Value formatArg = gen.getLIRGeneratorTool().emitConstant(wordKind, new CStringConstant(format));
85 
86         ForeignCallLinkage linkage = gen.getLIRGeneratorTool().getForeignCalls().lookupForeignCall(VM_ERROR);
87         gen.getLIRGeneratorTool().emitForeignCall(linkage, null, whereArg, formatArg, gen.operand(value));
88     }
89 
90     @NodeIntrinsic
91     public static native void vmError(@ConstantNodeParameter String format, long value);
92 }
