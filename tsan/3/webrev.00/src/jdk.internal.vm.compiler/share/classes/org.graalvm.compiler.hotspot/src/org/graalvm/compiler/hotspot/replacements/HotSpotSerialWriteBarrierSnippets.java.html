46 
47 public class HotSpotSerialWriteBarrierSnippets extends SerialWriteBarrierSnippets {
48     private final GraalHotSpotVMConfig config;
49 
50     public HotSpotSerialWriteBarrierSnippets(GraalHotSpotVMConfig config) {
51         this.config = config;
52     }
53 
54     @Override
55     public Word cardTableAddress() {
56         return WordFactory.unsigned(GraalHotSpotVMConfigNode.cardTableAddress());
57     }
58 
59     @Override
60     public int cardTableShift() {
61         return HotSpotReplacementsUtil.cardTableShift(INJECTED_VMCONFIG);
62     }
63 
64     @Override
65     public boolean verifyBarrier() {
66         return ReplacementsUtil.REPLACEMENTS_ASSERTIONS_ENABLED || config.verifyBeforeGC || config.verifyAfterGC;
67     }
68 
69     @Override
70     protected byte dirtyCardValue() {
71         return config.dirtyCardValue;
72     }
73 
74     public static class Templates extends AbstractTemplates {
75         private final SnippetInfo serialImpreciseWriteBarrier;
76         private final SnippetInfo serialPreciseWriteBarrier;
77         private final SnippetInfo serialArrayRangeWriteBarrier;
78 
79         private final SerialWriteBarrierLowerer lowerer;
80 
81         public Templates(OptionValues options, Iterable<DebugHandlersFactory> factories, Group.Factory factory, HotSpotProviders providers, TargetDescription target, GraalHotSpotVMConfig config) {
82             super(options, factories, providers, providers.getSnippetReflection(), target);
83             this.lowerer = new SerialWriteBarrierLowerer(factory);
84 
85             HotSpotSerialWriteBarrierSnippets receiver = new HotSpotSerialWriteBarrierSnippets(config);
86             serialImpreciseWriteBarrier = snippet(SerialWriteBarrierSnippets.class, "serialImpreciseWriteBarrier", null, receiver, GC_CARD_LOCATION);
87             serialPreciseWriteBarrier = snippet(SerialWriteBarrierSnippets.class, "serialPreciseWriteBarrier", null, receiver, GC_CARD_LOCATION);
88             serialArrayRangeWriteBarrier = snippet(SerialWriteBarrierSnippets.class, "serialArrayRangeWriteBarrier", null, receiver, GC_CARD_LOCATION);
89         }
90 
91         public void lower(SerialWriteBarrier barrier, LoweringTool tool) {
92             lowerer.lower(this, serialPreciseWriteBarrier, serialImpreciseWriteBarrier, barrier, tool);
93         }
94 
95         public void lower(SerialArrayRangeWriteBarrier barrier, LoweringTool tool) {
96             lowerer.lower(this, serialArrayRangeWriteBarrier, barrier, tool);
97         }
98     }
99 }
