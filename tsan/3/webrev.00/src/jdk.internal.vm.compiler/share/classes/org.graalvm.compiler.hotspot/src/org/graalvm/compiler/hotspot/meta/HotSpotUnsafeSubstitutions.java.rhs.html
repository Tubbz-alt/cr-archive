25 package org.graalvm.compiler.hotspot.meta;
26 


27 import org.graalvm.compiler.api.replacements.ClassSubstitution;
28 import org.graalvm.compiler.api.replacements.MethodSubstitution;
29 import org.graalvm.compiler.hotspot.replacements.UnsafeCopyMemoryNode;
30 import org.graalvm.compiler.serviceprovider.JavaVersionUtil;


31 
32 @ClassSubstitution(className = {"jdk.internal.misc.Unsafe", "sun.misc.Unsafe"})
33 public class HotSpotUnsafeSubstitutions {
34 
35     public static final String copyMemoryName = JavaVersionUtil.JAVA_SPEC <= 8 ? "copyMemory" : "copyMemory0";
36 
37     @SuppressWarnings("unused")
38     @MethodSubstitution(isStatic = false)
39     static void copyMemory(Object receiver, Object srcBase, long srcOffset, Object destBase, long destOffset, long bytes) {
40         UnsafeCopyMemoryNode.copyMemory(false, receiver, srcBase, srcOffset, destBase, destOffset, bytes);
41     }
42 
43     @SuppressWarnings("unused")
44     @MethodSubstitution(value = "copyMemory", isStatic = false)
45     static void copyMemoryGuarded(Object receiver, Object srcBase, long srcOffset, Object destBase, long destOffset, long bytes) {
46         UnsafeCopyMemoryNode.copyMemory(true, receiver, srcBase, srcOffset, destBase, destOffset, bytes);
47     }
48 }
