45 /**
46  * Node that is used to maintain a stack based counter of how many locks are currently held.
47  */
48 @NodeInfo(cycles = NodeCycles.CYCLES_2, size = SIZE_1)
49 public final class MonitorCounterNode extends FloatingNode implements LIRLowerable, Node.ValueNumberable {
50     public static final NodeClass<MonitorCounterNode> TYPE = NodeClass.create(MonitorCounterNode.class);
51 
52     public MonitorCounterNode(@InjectedNodeParameter WordTypes wordTypes) {
53         super(TYPE, StampFactory.forKind(wordTypes.getWordKind()));
54     }
55 
56     @Override
57     public void generate(NodeLIRBuilderTool gen) {
58         assert graph().getNodes().filter(MonitorCounterNode.class).count() == 1 : "monitor counters not canonicalized to single instance";
59         VirtualStackSlot counter = gen.getLIRGeneratorTool().getResult().getFrameMapBuilder().allocateStackSlots(1, new BitSet(0), null);
60         Value result = gen.getLIRGeneratorTool().emitAddress(counter);
61         gen.setResult(this, result);
62     }
63 
64     @NodeIntrinsic
65     public static native Word counter();
66 }
