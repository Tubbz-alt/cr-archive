 97 public class AArch64HotSpotRegisterAllocationConfig extends RegisterAllocationConfig {
 98 










 99     // @formatter:off
100     static final Register[] registerAllocationOrder = {
101         r0,  r1,  r2,  r3,  r4,  r5,  r6,  r7,
102         r8,  r9,  r10, r11, r12, r13, r14, r15,
103         r16, r17, r18, r19, r20, r21, r22, r23,
104         r24, r25, r26, /* r27, */ r28, /* r29, r30, r31 */
105 
106         v0,  v1,  v2,  v3,  v4,  v5,  v6,  v7,
107         v8,  v9,  v10, v11, v12, v13, v14, v15,
108         v16, v17, v18, v19, v20, v21, v22, v23,
109         v24, v25, v26, v27, v28, v29, v30, v31
110     };
111     // @formatter:on
112 
113     public AArch64HotSpotRegisterAllocationConfig(RegisterConfig registerConfig, String[] allocationRestrictedTo) {
114         super(registerConfig, allocationRestrictedTo);
115     }
116 
117     @Override
118     protected RegisterArray initAllocatable(RegisterArray registers) {
119         BitSet regMap = new BitSet(registerConfig.getAllocatableRegisters().size());
120         for (Register reg : registers) {
121             regMap.set(reg.number);
122         }
123 
124         ArrayList<Register> allocatableRegisters = new ArrayList<>(registers.size());
125         for (Register reg : registerAllocationOrder) {
126             if (regMap.get(reg.number)) {
127                 allocatableRegisters.add(reg);
128             }
129         }
130 
131         return super.initAllocatable(new RegisterArray(allocatableRegisters));
132     }
