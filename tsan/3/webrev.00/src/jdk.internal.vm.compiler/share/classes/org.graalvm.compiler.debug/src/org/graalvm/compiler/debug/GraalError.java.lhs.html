 25 package org.graalvm.compiler.debug;
 26 
 27 import java.util.ArrayList;
 28 import java.util.Locale;
 29 
 30 /**
 31  * Indicates a condition that should never occur during normal operation.
 32  */
 33 public class GraalError extends Error {
 34 
 35     private static final long serialVersionUID = 531632331813456233L;
 36     private final ArrayList<String> context = new ArrayList<>();
 37 
 38     public static RuntimeException unimplemented() {
 39         throw new GraalError("unimplemented");
 40     }
 41 
 42     public static RuntimeException unimplemented(String msg) {
 43         throw new GraalError("unimplemented: %s", msg);
 44     }
 45 
 46     public static RuntimeException shouldNotReachHere() {
 47         throw new GraalError("should not reach here");
 48     }
 49 
 50     public static RuntimeException shouldNotReachHere(String msg) {
 51         throw new GraalError("should not reach here: %s", msg);
 52     }
 53 
 54     public static RuntimeException shouldNotReachHere(Throwable cause) {
 55         throw new GraalError(cause);
 56     }
 57 
 58     /**
 59      * Checks a given condition and throws a {@link GraalError} if it is false. Guarantees are
 60      * stronger than assertions in that they are always checked. Error messages for guarantee
 61      * violations should clearly indicate the nature of the problem as well as a suggested solution
 62      * if possible.
 63      *
 64      * @param condition the condition to check
 65      * @param msg the message that will be associated with the error
 66      */
 67     public static void guarantee(boolean condition, String msg) {
 68         if (!condition) {
 69             throw new GraalError("failed guarantee: " + msg);
 70         }
 71     }
 72 
 73     /**
 74      * Checks a given condition and throws a {@link GraalError} if it is false. Guarantees are
 75      * stronger than assertions in that they are always checked. Error messages for guarantee
 76      * violations should clearly indicate the nature of the problem as well as a suggested solution
 77      * if possible.
 78      *
 79      * @param condition the condition to check
 80      * @param msg the message that will be associated with the error, in
 81      *            {@link String#format(String, Object...)} syntax
 82      * @param arg argument to the format string in {@code msg}
 83      */
 84     public static void guarantee(boolean condition, String msg, Object arg) {
 85         if (!condition) {
 86             throw new GraalError("failed guarantee: " + msg, arg);
 87         }
 88     }
 89 
 90     /**
 91      * Checks a given condition and throws a {@link GraalError} if it is false. Guarantees are
 92      * stronger than assertions in that they are always checked. Error messages for guarantee
 93      * violations should clearly indicate the nature of the problem as well as a suggested solution
 94      * if possible.
 95      *
 96      * @param condition the condition to check
 97      * @param msg the message that will be associated with the error, in
 98      *            {@link String#format(String, Object...)} syntax
 99      * @param arg1 argument to the format string in {@code msg}
100      * @param arg2 argument to the format string in {@code msg}
101      */
102     public static void guarantee(boolean condition, String msg, Object arg1, Object arg2) {
103         if (!condition) {
104             throw new GraalError("failed guarantee: " + msg, arg1, arg2);
105         }
106     }
107 
108     /**
109      * Checks a given condition and throws a {@link GraalError} if it is false. Guarantees are
110      * stronger than assertions in that they are always checked. Error messages for guarantee
111      * violations should clearly indicate the nature of the problem as well as a suggested solution
112      * if possible.
113      *
114      * @param condition the condition to check
115      * @param msg the message that will be associated with the error, in
116      *            {@link String#format(String, Object...)} syntax
117      * @param arg1 argument to the format string in {@code msg}
118      * @param arg2 argument to the format string in {@code msg}
119      * @param arg3 argument to the format string in {@code msg}
120      */
121     public static void guarantee(boolean condition, String msg, Object arg1, Object arg2, Object arg3) {
122         if (!condition) {
123             throw new GraalError("failed guarantee: " + msg, arg1, arg2, arg3);
124         }
125     }
126 
127     /**
128      * This override exists to catch cases when {@link #guarantee(boolean, String, Object)} is
129      * called with one argument bound to a varargs method parameter. It will bind to this method
130      * instead of the single arg variant and produce a deprecation warning instead of silently
131      * wrapping the Object[] inside of another Object[].
132      */
133     @Deprecated
134     public static void guarantee(boolean condition, String msg, Object... args) {
135         if (!condition) {
136             throw new GraalError("failed guarantee: " + msg, args);
137         }
138     }
139 
140     /**
141      * This constructor creates a {@link GraalError} with a given message.
142      *
143      * @param msg the message that will be associated with the error
144      */
145     public GraalError(String msg) {
146         super(msg);
147     }
148 
149     /**
150      * This constructor creates a {@link GraalError} with a message assembled via
151      * {@link String#format(String, Object...)}. It always uses the ENGLISH locale in order to
152      * always generate the same output.
153      *
154      * @param msg the message that will be associated with the error, in String.format syntax
155      * @param args parameters to String.format - parameters that implement {@link Iterable} will be
156      *            expanded into a [x, x, ...] representation.
157      */
158     public GraalError(String msg, Object... args) {
159         super(format(msg, args));
160     }
161 
162     /**
163      * This constructor creates a {@link GraalError} for a given causing Throwable instance.
164      *
165      * @param cause the original exception that contains additional information on this error
166      */
167     public GraalError(Throwable cause) {
168         super(cause);
169     }
170 








171     /**
172      * This constructor creates a {@link GraalError} and adds all the
173      * {@linkplain #addContext(String) context} of another {@link GraalError}.
174      *
175      * @param e the original {@link GraalError}
176      */
177     public GraalError(GraalError e) {
178         super(e);
179         context.addAll(e.context);
180     }
181 
182     @Override
183     public String toString() {
184         StringBuilder str = new StringBuilder();
185         str.append(super.toString());
186         str.append(context());
187         return str.toString();
188     }
189 
190     public String context() {
191         StringBuilder str = new StringBuilder();
192         for (String s : context) {
193             str.append("\n\tat ").append(s);
194         }
195         return str.toString();
196     }
197 
198     private static String format(String msg, Object... args) {
199         if (args != null) {
200             // expand Iterable parameters into a list representation
201             for (int i = 0; i < args.length; i++) {
202                 if (args[i] instanceof Iterable<?>) {
203                     ArrayList<Object> list = new ArrayList<>();
204                     for (Object o : (Iterable<?>) args[i]) {
205                         list.add(o);
206                     }
207                     args[i] = list.toString();
208                 }
209             }
210         }
211         return String.format(Locale.ENGLISH, msg, args);
212     }
213 
214     public GraalError addContext(String newContext) {
215         this.context.add(newContext);
216         return this;
217     }
218 
219     public GraalError addContext(String name, Object obj) {
220         return addContext(format("%s: %s", name, obj));
221     }
222 }
