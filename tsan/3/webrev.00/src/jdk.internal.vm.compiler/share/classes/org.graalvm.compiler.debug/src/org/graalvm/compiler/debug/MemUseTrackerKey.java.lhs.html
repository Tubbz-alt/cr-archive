25 package org.graalvm.compiler.debug;
26 
27 import org.graalvm.compiler.serviceprovider.GraalServices;
28 
29 /**
30  * Tracks memory usage within a scope using {@link com.sun.management.ThreadMXBean}. This facility
31  * should be employed using the try-with-resources pattern:
32  *
33  * <pre>
34  * try (DebugCloseable a = memUseTracker.start()) {
35  *     // the code to measure
36  * }
37  * </pre>
38  */
39 public interface MemUseTrackerKey extends MetricKey {
40 
41     /**
42      * Creates a point from which memory usage will be recorded if memory use tracking is
43      * {@linkplain DebugContext#isMemUseTrackingEnabled() enabled}.
44      *
45      * @return an object that must be closed once the activity has completed to add the memory used
46      *         since this call to the total for this tracker
47      */
48     DebugCloseable start(DebugContext debug);
49 
50     /**
51      * Gets the current value of this tracker.
52      */
53     long getCurrentValue(DebugContext debug);
54 
55     @Override
56     MemUseTrackerKey doc(String string);
57 
58     static long getCurrentThreadAllocatedBytes() {
59         return GraalServices.getCurrentThreadAllocatedBytes();
60     }
61 }
