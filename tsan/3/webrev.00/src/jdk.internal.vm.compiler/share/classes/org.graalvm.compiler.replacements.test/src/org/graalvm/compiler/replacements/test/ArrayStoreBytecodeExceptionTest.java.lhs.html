44 @RunWith(Parameterized.class)
45 public class ArrayStoreBytecodeExceptionTest extends BytecodeExceptionTest {
46 
47     private static class Exceptions {
48 
49         private static Object[] array = new Exceptions[1];
50 
51         public static void throwArrayStore(Object obj) {
52             array[0] = obj;
53         }
54     }
55 
56     @Override
57     protected void registerInvocationPlugins(InvocationPlugins invocationPlugins) {
58         invocationPlugins.register(new InvocationPlugin() {
59             @Override
60             public boolean apply(GraphBuilderContext b, ResolvedJavaMethod targetMethod, Receiver receiver, ValueNode obj) {
61                 return throwBytecodeException(b, BytecodeExceptionKind.ARRAY_STORE, obj);
62             }
63         }, Exceptions.class, "throwArrayStore", Object.class);
64         super.registerInvocationPlugins(invocationPlugins);
65     }
66 
67     public static void arrayStoreSnippet(Object obj) {
68         Exceptions.throwArrayStore(obj);
69     }
70 
71     @Parameter(0) public Object object;
72     @Parameter(1) public Class<?> cls;
73 
74     @Parameters(name = "{1}")
75     public static Collection<Object[]> data() {
76         Object[] objects = {"string", 42, new int[0], new String[0], new double[0][]};
77 
78         ArrayList<Object[]> ret = new ArrayList<>(objects.length);
79         for (Object o : objects) {
80             ret.add(new Object[]{o, o.getClass()});
81         }
82         return ret;
83     }
84 
85     @Test
86     public void testArrayStoreException() {
87         test("arrayStoreSnippet", object);
88     }
