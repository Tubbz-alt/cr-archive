25 package org.graalvm.compiler.replacements.test;
26 
27 import org.junit.Test;
28 import org.graalvm.compiler.nodes.extended.BytecodeExceptionNode.BytecodeExceptionKind;
29 import org.graalvm.compiler.nodes.graphbuilderconf.GraphBuilderContext;
30 import org.graalvm.compiler.nodes.graphbuilderconf.InvocationPlugin;
31 import org.graalvm.compiler.nodes.graphbuilderconf.InvocationPlugins;
32 
33 import jdk.vm.ci.meta.ResolvedJavaMethod;
34 
35 public class NullBytecodeExceptionTest extends BytecodeExceptionTest {
36 
37     private static class Exceptions {
38 
39         private static Object obj = null;
40 
41         public static void throwNull() {
42             obj.toString();
43         }
44     }
45 
46     @Override
47     protected void registerInvocationPlugins(InvocationPlugins invocationPlugins) {
48         invocationPlugins.register(new InvocationPlugin() {
49             @Override
50             public boolean apply(GraphBuilderContext b, ResolvedJavaMethod targetMethod, Receiver receiver) {
51                 return throwBytecodeException(b, BytecodeExceptionKind.NULL_POINTER);
52             }
53         }, Exceptions.class, "throwNull");
54         super.registerInvocationPlugins(invocationPlugins);
55     }
56 
57     public static void nullSnippet() {
58         Exceptions.throwNull();
59     }
60 
61     @Test
62     public void testNullPointerException() {
63         test("nullSnippet");
64     }
65 }
