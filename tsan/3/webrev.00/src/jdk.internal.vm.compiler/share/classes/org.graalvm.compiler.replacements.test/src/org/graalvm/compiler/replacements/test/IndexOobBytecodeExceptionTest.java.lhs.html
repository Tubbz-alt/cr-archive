36 import org.graalvm.compiler.api.directives.GraalDirectives;
37 import org.graalvm.compiler.nodes.ValueNode;
38 import org.graalvm.compiler.nodes.extended.BytecodeExceptionNode.BytecodeExceptionKind;
39 import org.graalvm.compiler.nodes.graphbuilderconf.GraphBuilderContext;
40 import org.graalvm.compiler.nodes.graphbuilderconf.InvocationPlugin;
41 import org.graalvm.compiler.nodes.graphbuilderconf.InvocationPlugins;
42 
43 import jdk.vm.ci.meta.ResolvedJavaMethod;
44 
45 @RunWith(Parameterized.class)
46 public class IndexOobBytecodeExceptionTest extends BytecodeExceptionTest {
47 
48     private static class Exceptions {
49 
50         private static Object[] empty = new Object[0];
51 
52         public static void throwOutOfBounds(int idx, int length) {
53             GraalDirectives.blackhole(empty[idx]);
54             GraalDirectives.blackhole(length);
55         }
56     }
57 
58     @Override
59     protected void registerInvocationPlugins(InvocationPlugins invocationPlugins) {
60         invocationPlugins.register(new InvocationPlugin() {
61             @Override
62             public boolean apply(GraphBuilderContext b, ResolvedJavaMethod targetMethod, Receiver receiver, ValueNode idx, ValueNode length) {
63                 return throwBytecodeException(b, BytecodeExceptionKind.OUT_OF_BOUNDS, idx, length);
64             }
65         }, Exceptions.class, "throwOutOfBounds", int.class, int.class);
66         super.registerInvocationPlugins(invocationPlugins);
67     }
68 
69     public static void oobSnippet(int idx, int length) {
70         Exceptions.throwOutOfBounds(idx, length);
71     }
72 
73     @Parameter public int index;
74 
75     @Parameters(name = "{0}")
76     public static Collection<Object[]> data() {
77         int[] values = {Integer.MIN_VALUE, -42, -1, 0, 1, 42, Integer.MAX_VALUE};
78 
79         ArrayList<Object[]> ret = new ArrayList<>(values.length);
80         for (int i : values) {
81             ret.add(new Object[]{i});
82         }
83         return ret;
84     }
85 
86     @Test
87     public void testOutOfBoundsException() {
88         test("oobSnippet", index, Exceptions.empty.length);

89     }
90 }
