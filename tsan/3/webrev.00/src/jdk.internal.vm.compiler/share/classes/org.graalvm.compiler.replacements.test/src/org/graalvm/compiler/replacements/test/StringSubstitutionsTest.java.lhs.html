 36  * Tests {@link StringSubstitutions}.
 37  */
 38 public class StringSubstitutionsTest extends MethodSubstitutionTest {
 39 
 40     public void testSubstitution(String testMethodName, Class<?> intrinsicClass, Class<?> holder, String methodName, boolean optional, Object[] args1, Object[] args2) {
 41         ResolvedJavaMethod realMethod = getResolvedJavaMethod(holder, methodName);
 42         ResolvedJavaMethod testMethod = getResolvedJavaMethod(testMethodName);
 43         StructuredGraph graph = testGraph(testMethodName);
 44 
 45         // Check to see if the resulting graph contains the expected node
 46         StructuredGraph replacement = getReplacements().getSubstitution(realMethod, -1, false, null);
 47         if (replacement == null && !optional) {
 48             assertInGraph(graph, intrinsicClass);
 49         }
 50 
 51         // Force compilation
 52         InstalledCode code = getCode(testMethod);
 53         assert optional || code != null;
 54 
 55         for (int i = 0; i < args1.length; i++) {
 56             Object arg1 = args1[i];
 57             Object arg2 = args2[i];
 58             Object expected = invokeSafe(realMethod, arg1, arg2);
 59             // Verify that the original method and the substitution produce the same value
 60             assertDeepEquals(expected, invokeSafe(testMethod, null, arg1, arg2));
 61             // Verify that the generated code and the original produce the same value
 62             assertDeepEquals(expected, executeVarargsSafe(code, arg1, arg2));
 63         }
 64     }
 65 
 66     @Test
 67     public void testEquals() {
 68         if (!Java8OrEarlier) {
 69             // StringSubstitutions are disabled in 1.9
 70             return;
 71         }
 72 
 73         final int n = 1000;
 74         Object[] args1 = new Object[n];
 75         Object[] args2 = new Object[n];
 76 
 77         // equal strings
 78         String s1 = "";
 79         String s2 = "";
 80         for (int i = 0; i < n / 2; i++) {
 81             args1[i] = s1;
 82             args2[i] = s2;
 83             s1 = s1 + "0";
 84             s2 = s2 + "0";
 85         }
 86 
 87         // non-equal strings
 88         s1 = "";
 89         s2 = "";
 90         for (int i = n / 2; i < n; i++) {
 91             args1[i] = s1;
 92             args2[i] = s2;
 93             s2 = s1 + "1";
 94             s1 = s1 + "0";
 95         }
 96 
 97         testSubstitution("stringEquals", ArrayEqualsNode.class, String.class, "equals", false, args1, args2);
 98     }
