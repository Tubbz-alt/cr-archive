 52 @RunWith(Parameterized.class)
 53 public class ClassCastBytecodeExceptionTest extends BytecodeExceptionTest {
 54 
 55     private static class Exceptions {
 56 
 57         public static void throwClassCast(Object obj, Class<?> cls) {
 58             /*
 59              * We don't use cls.cast(obj) here because that gives a different exception message than
 60              * the checkcast bytecode.
 61              */
 62             if (cls == Double.class) {
 63                 Double cast = (Double) obj;
 64                 GraalDirectives.blackhole(cast);
 65             } else if (cls == byte[].class) {
 66                 byte[] cast = (byte[]) obj;
 67                 GraalDirectives.blackhole(cast);
 68             } else if (cls == String[].class) {
 69                 String[] cast = (String[]) obj;
 70                 GraalDirectives.blackhole(cast);
 71             } else if (cls == Object[][].class) {
 72                 Object[][] cast = (Object[][]) obj;
 73                 GraalDirectives.blackhole(cast);
 74             } else {
 75                 Assert.fail("unexpected class argument");
 76             }
 77         }
 78     }
 79 
 80     @Override
 81     protected void registerInvocationPlugins(InvocationPlugins invocationPlugins) {
 82         invocationPlugins.register(new InvocationPlugin() {
 83             @Override
 84             public boolean apply(GraphBuilderContext b, ResolvedJavaMethod targetMethod, Receiver receiver, ValueNode obj, ValueNode classNode) {
 85                 ResolvedJavaType type = b.getConstantReflection().asJavaType(classNode.asConstant());
 86                 Constant hub = b.getConstantReflection().asObjectHub(type);
 87                 Stamp hubStamp = b.getStampProvider().createHubStamp(StampFactory.object(TypeReference.createExactTrusted(type)));
 88                 ConstantNode hubConst = b.add(ConstantNode.forConstant(hubStamp, hub, b.getMetaAccess()));
 89                 return throwBytecodeException(b, BytecodeExceptionKind.CLASS_CAST, obj, hubConst);
 90             }
 91         }, Exceptions.class, "throwClassCast", Object.class, Class.class);
 92         super.registerInvocationPlugins(invocationPlugins);
 93     }
 94 
 95     @Parameter(0) public Object object;
 96     @Parameter(1) public Class<?> cls;
 97 
 98     @Parameters(name = "{1}")
 99     public static Collection<Object[]> data() {
100         Object[] objects = {"string", 42, new int[0], new Object[0], new double[0][]};
101 
102         ArrayList<Object[]> ret = new ArrayList<>(objects.length);
103         for (Object o : objects) {
104             ret.add(new Object[]{o, o.getClass()});
105         }
106         return ret;
107     }

111     }
