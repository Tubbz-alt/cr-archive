 55 public class EdgesTest extends GraalCompilerTest {
 56 
 57     @NodeInfo(cycles = CYCLES_0, size = SIZE_0)
 58     static final class TestNode extends Node {
 59         public static final NodeClass<TestNode> TYPE = NodeClass.create(TestNode.class);
 60         @Input NodeInputList<ValueNode> itail;
 61         @Input ConstantNode i1;
 62         @Input FloatingNode i2;
 63 
 64         protected TestNode() {
 65             super(TYPE);
 66         }
 67 
 68     }
 69 
 70     protected StructuredGraph createGraph() {
 71         OptionValues options = getInitialOptions();
 72         return new StructuredGraph.Builder(options, getDebugContext(options)).build();
 73     }
 74 
 75     StructuredGraph graph = createGraph();
 76     TestNode node;
 77     ConstantNode i1;
 78     ConstantNode i2;
 79     ConstantNode i3;
 80     ConstantNode i4;
 81     Edges inputs;
 82 
 83     public EdgesTest() {
 84         node = new TestNode();
 85         i1 = ConstantNode.forInt(1, graph);
 86         i2 = ConstantNode.forDouble(1.0d, graph);
 87         i3 = ConstantNode.forInt(4, graph);
 88         i4 = ConstantNode.forInt(14, graph);
 89         node.itail = new NodeInputList<>(node, new ValueNode[]{i3, i4});
 90         node.i1 = i1;
 91         node.i2 = i2;
 92         graph.add(node);
 93         inputs = node.getNodeClass().getInputEdges();
 94     }
 95 
 96     /**
 97      * Checks that there are no checkcasts in the compiled version of
 98      * {@link Edges#getNode(Node, long[], int)}.
 99      */
100     @Test
101     public void test0() {
102         testMethod(getMethod("getNode", Node.class, long[].class, int.class), null, node, inputs.getOffsets(), 0);
103     }
104 
105     /**
106      * Checks that there are no checkcasts in the compiled version of
107      * {@link Edges#getNodeList(Node, long[], int)}.
108      */
109     @Test
110     public void test1() {
111         testMethod(getMethod("getNodeList", Node.class, long[].class, int.class), null, node, inputs.getOffsets(), 2);
112     }
113 
114     /**
115      * Checks that there are no checkcasts in the compiled version of
116      * {@link Edges#setNode(Node, int, Node)}.
117      */
118     @Test
119     public void test2() {
120         testMethod(getMethod("setNode", Node.class, int.class, Node.class), inputs, node, 1, i2);
121     }
122 
123     private void testMethod(Method method, Object receiver, Object... args) {
124         try {
125             // Invoke the method to ensure it has a type profile
126             for (int i = 0; i < 5000; i++) {
127                 method.invoke(receiver, args);
128             }
129         } catch (Exception e) {
130             throw new RuntimeException(e);
131         }
132 
133         ResolvedJavaMethod javaMethod = getMetaAccess().lookupJavaMethod(method);
134         StructuredGraph g = parseProfiled(javaMethod, AllowAssumptions.NO);
135         HighTierContext context = getDefaultHighTierContext();
136         new InliningPhase(new InlineMethodSubstitutionsPolicy(), new CanonicalizerPhase()).apply(g, context);
137         new CanonicalizerPhase().apply(g, context);
138         Assert.assertTrue(g.getNodes().filter(InstanceOfNode.class).isEmpty());
139     }
140 
141     private static Method getMethod(final String name, Class<?>... parameters) {
142         try {
143             return Edges.class.getDeclaredMethod(name, parameters);
144         } catch (NoSuchMethodException | SecurityException e) {
145             throw new RuntimeException(e);
146         }
147     }
