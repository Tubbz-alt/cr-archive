25 package org.graalvm.compiler.replacements.test;
26 
27 import java.util.ArrayList;
28 import java.util.Collection;
29 
30 import org.graalvm.compiler.core.test.GraalCompilerTest;
31 import org.junit.Test;
32 import org.junit.runner.RunWith;
33 import org.junit.runners.Parameterized;
34 
35 @RunWith(value = Parameterized.class)
36 public class StringIndexOfCharTest extends GraalCompilerTest {
37     @Parameterized.Parameter(value = 0) public String sourceString;
38     @Parameterized.Parameter(value = 1) public int constantChar;
39     @Parameterized.Parameter(value = 2) public int fromIndex;
40 
41     @Parameterized.Parameters(name = "{0},{1},{2}")
42     public static Collection<Object[]> data() {
43         ArrayList<Object[]> tests = new ArrayList<>();
44         String longString = "ab";
45         for (int i = 0; i < 15; i++) {
46             longString = longString + longString;
47         }
48         longString = longString + "xx";
49         String longUTF16String = "\u03bb" + longString;
50         String mediumString = "aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaax" +
51                         "aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa";
52         String mediumUTF16String = "\u03bbaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaax" +
53                         "aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa";
54         String[] targets = new String[]{"foobar", "foo", "bar", "\u03bbfoobar", mediumString, mediumUTF16String, longString, longUTF16String};
55         int[] targetChars = new int[]{'f', 'o', 'r', 'x', Character.MIN_SUPPLEMENTARY_CODE_POINT};
56         int[] targetOffsets = new int[18];
57         for (int i = 0; i < 18; i++) {
58             targetOffsets[i] = i - 1;
59         }
60         for (String source : targets) {
61             for (int targetChar : targetChars) {
62                 for (int offset : targetOffsets) {
63                     tests.add(new Object[]{source, targetChar, offset});
64                 }
65             }
66         }
67 
68         return tests;
69     }
70 










71     public int testStringIndexOf(String a, int b) {
72         return a.indexOf(b);
73     }
74 
75     public int testStringIndexOfOffset(String a, int b, int offset) {
76         return a.indexOf(b, offset);
77     }
78 
79     @Test
80     public void testStringIndexOfConstant() {
81         test("testStringIndexOf", this.sourceString, this.constantChar);
82     }
83 
84     @Test
85     public void testStringIndexOfConstantOffset() {
86         test("testStringIndexOfOffset", this.sourceString, this.constantChar, this.fromIndex);
87     }
88 }
