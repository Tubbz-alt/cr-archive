 44     private static final int N = 10;
 45 
 46     private void testEquals(String methodName, Class<?>[] parameterTypes, ArrayBuilder builder) {
 47         Object[] args1 = new Object[N];
 48         Object[] args2 = new Object[N];
 49         int n = 0;
 50         // equal arrays
 51         for (int i = 0; i < N / 2; i++, n++) {
 52             args1[n] = builder.newArray(i, 0, 1);
 53             args2[n] = builder.newArray(i, 0, 1);
 54         }
 55         // non-equal arrays
 56         for (int i = 0; i < N / 2; i++, n++) {
 57             args1[n] = builder.newArray(i, 0, 1);
 58             args2[n] = builder.newArray(i, 1, 1);
 59         }
 60         testSubstitution(methodName, ArrayEqualsNode.class, Arrays.class, "equals", parameterTypes, false, false, args1, args2);
 61     }

101     public static boolean testEqualsNodeGVNSnippet(boolean b) {
102         int[] newIntArray = new int[]{0, 2, 3, 4, 5, 6, 7, 8, 9};
