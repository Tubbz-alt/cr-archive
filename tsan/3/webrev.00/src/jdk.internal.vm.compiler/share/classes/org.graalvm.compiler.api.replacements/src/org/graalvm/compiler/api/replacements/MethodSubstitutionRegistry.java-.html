25 package org.graalvm.compiler.api.replacements;
26 
27 import java.lang.reflect.Type;
28 
29 /**
30  * A registry for {@link MethodSubstitution}s.
31  */
32 public interface MethodSubstitutionRegistry {
33 
34     /**
35      * Gets the type representing the receiver (i.e., {@code this}) argument in a non-static method.
36      */
37     Class<?> getReceiverType();
38 
39     /**
40      * Registers a substitution method.
41      *
42      * @param substituteDeclaringClass the class declaring the substitute method
43      * @param name the name of both the original and substitute method
44      * @param argumentTypes the argument types of the method. Element 0 of this array must be
45      *            {@link #getReceiverType()} iff the method is non-static. Upon returning, element 0
46      *            will have been rewritten to {@code declaringClass}.
47      */
48     default void registerMethodSubstitution(Class<?> substituteDeclaringClass, String name, Type... argumentTypes) {
49         registerMethodSubstitution(substituteDeclaringClass, name, name, argumentTypes);
50     }
51 
52     /**
53      * Registers a substitution method.
54      *
55      * @param substituteDeclaringClass the class declaring the substitute method
56      * @param name the name of both the original method
57      * @param substituteName the name of the substitute method
58      * @param argumentTypes the argument types of the method. Element 0 of this array must be
59      *            {@link #getReceiverType()} iff the method is non-static. Upon returning, element 0
60      *            will have been rewritten to {@code declaringClass}.
61      */
62     void registerMethodSubstitution(Class<?> substituteDeclaringClass, String name, String substituteName, Type... argumentTypes);
63 }
