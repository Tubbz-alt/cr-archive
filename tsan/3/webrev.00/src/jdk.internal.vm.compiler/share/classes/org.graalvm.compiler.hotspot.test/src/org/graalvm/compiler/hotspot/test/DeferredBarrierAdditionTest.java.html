 54 /**
 55  * This tests that barriers which are deferrable because of ReduceInitialCardMarks are properly
 56  * omitted. The rule is simply that only writes to the very last allocated object can skip the card
 57  * mark. By creating references between objects only one write can skip the card mark and the other
 58  * must emit a card mark.
 59  */
 60 public class DeferredBarrierAdditionTest extends HotSpotGraalCompilerTest {
 61 
 62     private final GraalHotSpotVMConfig config = runtime().getVMConfig();
 63 
 64     public static Object testCrossReferences() {
 65         Object[] a = new Object[1];
 66         Object[] b = new Object[1];
 67         a[0] = b;
 68         b[0] = a;
 69         return a;
 70     }
 71 
 72     @Test
 73     public void testGroupAllocation() throws Exception {
 74         testHelper("testCrossReferences", 1, getInitialOptions());
 75     }
 76 
 77     @SuppressWarnings("try")
 78     protected void testHelper(final String snippetName, final int expectedBarriers, OptionValues options) {
 79         ResolvedJavaMethod snippet = getResolvedJavaMethod(snippetName);
 80         DebugContext debug = getDebugContext(options, null, snippet);
 81         try (DebugContext.Scope s = debug.scope("WriteBarrierAdditionTest", snippet)) {
 82             StructuredGraph graph = parseEager(snippet, AllowAssumptions.NO, debug);
 83             HighTierContext highContext = getDefaultHighTierContext();
 84             MidTierContext midContext = new MidTierContext(getProviders(), getTargetProvider(), OptimisticOptimizations.ALL, graph.getProfilingInfo());
 85             new InliningPhase(new InlineEverythingPolicy(), createCanonicalizerPhase()).apply(graph, highContext);
 86             this.createCanonicalizerPhase().apply(graph, highContext);
 87             new PartialEscapePhase(false, createCanonicalizerPhase(), debug.getOptions()).apply(graph, highContext);
 88             new LoweringPhase(createCanonicalizerPhase(), LoweringTool.StandardLoweringStage.HIGH_TIER).apply(graph, highContext);
 89             new GuardLoweringPhase().apply(graph, midContext);
 90             new LoweringPhase(createCanonicalizerPhase(), LoweringTool.StandardLoweringStage.MID_TIER).apply(graph, midContext);
 91             new WriteBarrierAdditionPhase().apply(graph, midContext);
 92             debug.dump(DebugContext.BASIC_LEVEL, graph, "After Write Barrier Addition");
 93 
 94             checkAssumptions(graph);
 95 
 96             int barriers = 0;
 97             if (config.useG1GC) {
 98                 barriers = graph.getNodes().filter(G1ReferentFieldReadBarrier.class).count() + graph.getNodes().filter(G1PreWriteBarrier.class).count() +
 99                                 graph.getNodes().filter(G1PostWriteBarrier.class).count();
100             } else {
101                 barriers = graph.getNodes().filter(SerialWriteBarrier.class).count();
102             }
103             if (expectedBarriers != barriers) {
104                 Assert.assertEquals(getScheduledGraphString(graph), expectedBarriers, barriers);
105             }
106         } catch (Throwable e) {
107             throw debug.handle(e);
108         }
109     }
110 
111     protected void checkAssumptions(StructuredGraph graph) {
112         assumeTrue(graph.getNodes().filter(AbstractNewObjectNode.class).isNotEmpty());
113     }
