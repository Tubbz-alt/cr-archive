 47 /**
 48  * Tests the intrinsification of certain crypto methods.

 52     @Override
 53     protected InstalledCode addMethod(DebugContext debug, ResolvedJavaMethod method, CompilationResult compResult) {
 54         return getBackend().createDefaultInstalledCode(debug, method, compResult);

 57     SecretKey aesKey;
 58     SecretKey desKey;
 59     byte[] input;
 60     ByteArrayOutputStream aesExpected = new ByteArrayOutputStream();
 61     ByteArrayOutputStream desExpected = new ByteArrayOutputStream();
 62 
 63     public HotSpotCryptoSubstitutionTest() throws Exception {
 64         byte[] seed = {0x4, 0x7, 0x1, 0x1};
 65         SecureRandom random = new SecureRandom(seed);
 66         KeyGenerator aesKeyGen = KeyGenerator.getInstance("AES");
 67         KeyGenerator desKeyGen = KeyGenerator.getInstance("DESede");
 68         aesKeyGen.init(128, random);
 69         desKeyGen.init(168, random);
 70         aesKey = aesKeyGen.generateKey();
 71         desKey = desKeyGen.generateKey();
 72         input = readClassfile16(getClass());
 73 
 74         aesExpected.write(runEncryptDecrypt(aesKey, "AES/CBC/NoPadding"));
 75         aesExpected.write(runEncryptDecrypt(aesKey, "AES/CBC/PKCS5Padding"));
 76 
 77         desExpected.write(runEncryptDecrypt(desKey, "DESede/CBC/NoPadding"));
 78         desExpected.write(runEncryptDecrypt(desKey, "DESede/CBC/PKCS5Padding"));
 79     }
 80 
 81     @Test
 82     public void testAESCryptIntrinsics() throws Exception {
 83         String aesEncryptName = HotSpotGraphBuilderPlugins.lookupIntrinsicName(runtime().getVMConfig(), "com/sun/crypto/provider/AESCrypt", "implEncryptBlock", "encryptBlock");
 84         String aesDecryptName = HotSpotGraphBuilderPlugins.lookupIntrinsicName(runtime().getVMConfig(), "com/sun/crypto/provider/AESCrypt", "implDecryptBlock", "decryptBlock");
 85         if (compileAndInstall("com.sun.crypto.provider.AESCrypt", aesEncryptName, aesDecryptName)) {
 86             ByteArrayOutputStream actual = new ByteArrayOutputStream();
 87             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/NoPadding"));
 88             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/PKCS5Padding"));
 89             Assert.assertArrayEquals(aesExpected.toByteArray(), actual.toByteArray());
 90         }
 91     }
 92 
 93     @Test
 94     public void testCipherBlockChainingIntrinsics() throws Exception {
 95         String cbcEncryptName = HotSpotGraphBuilderPlugins.lookupIntrinsicName(runtime().getVMConfig(), "com/sun/crypto/provider/CipherBlockChaining", "implEncrypt", "encrypt");
 96         String cbcDecryptName = HotSpotGraphBuilderPlugins.lookupIntrinsicName(runtime().getVMConfig(), "com/sun/crypto/provider/CipherBlockChaining", "implDecrypt", "decrypt");
 97         if (compileAndInstall("com.sun.crypto.provider.CipherBlockChaining", cbcEncryptName, cbcDecryptName)) {
 98             ByteArrayOutputStream actual = new ByteArrayOutputStream();
 99             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/NoPadding"));
100             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/PKCS5Padding"));
101             Assert.assertArrayEquals(aesExpected.toByteArray(), actual.toByteArray());
102 
103             actual.reset();
104             actual.write(runEncryptDecrypt(desKey, "DESede/CBC/NoPadding"));
105             actual.write(runEncryptDecrypt(desKey, "DESede/CBC/PKCS5Padding"));
106             Assert.assertArrayEquals(desExpected.toByteArray(), actual.toByteArray());
107         }
108     }

110     /**
111      * Compiles and installs the substitution for some specified methods. Once installed, the next
112      * execution of the methods will use the newly installed code.
113      *
114      * @param className the name of the class for which substitutions are available
115      * @param methodNames the names of the substituted methods
116      * @return true if at least one substitution was compiled and installed
117      */
118     private boolean compileAndInstall(String className, String... methodNames) {
119         if (!runtime().getVMConfig().useAESIntrinsics) {
120             return false;
121         }
122         Class<?> c;
123         try {
124             c = Class.forName(className);
125         } catch (ClassNotFoundException e) {
126             // It's ok to not find the class - a different security provider
127             // may have been installed
128             return false;
129         }
130         boolean atLeastOneCompiled = false;
131         for (String methodName : methodNames) {
132             if (compileAndInstallSubstitution(c, methodName) != null) {
133                 atLeastOneCompiled = true;
134             }
135         }
136         return atLeastOneCompiled;
137     }

139     AlgorithmParameters algorithmParameters;
140 
141     private byte[] encrypt(byte[] indata, SecretKey key, String algorithm) throws Exception {
142 
143         byte[] result = indata;
144 
145         Cipher c = Cipher.getInstance(algorithm);
146         c.init(Cipher.ENCRYPT_MODE, key);
147         algorithmParameters = c.getParameters();
148 
149         byte[] r1 = c.update(result);
150         byte[] r2 = c.doFinal();
151 
152         result = new byte[r1.length + r2.length];
153         System.arraycopy(r1, 0, result, 0, r1.length);
154         System.arraycopy(r2, 0, result, r1.length, r2.length);
155 
156         return result;
157     }
158 
159     private byte[] decrypt(byte[] indata, SecretKey key, String algorithm) throws Exception {
160 
161         byte[] result = indata;
162 
163         Cipher c = Cipher.getInstance(algorithm);
164         c.init(Cipher.DECRYPT_MODE, key, algorithmParameters);
165 
166         byte[] r1 = c.update(result);
167         byte[] r2 = c.doFinal();
168 
169         result = new byte[r1.length + r2.length];
170         System.arraycopy(r1, 0, result, 0, r1.length);
171         System.arraycopy(r2, 0, result, r1.length, r2.length);
172         return result;
173     }
174 
175     private static byte[] readClassfile16(Class<? extends HotSpotCryptoSubstitutionTest> c) throws IOException {
176         String classFilePath = "/" + c.getName().replace('.', '/') + ".class";
177         InputStream stream = c.getResourceAsStream(classFilePath);
178         int bytesToRead = stream.available();
179         bytesToRead -= bytesToRead % 16;
180         byte[] classFile = new byte[bytesToRead];
181         new DataInputStream(stream).readFully(classFile);
182         return classFile;
183     }
184 
185     public byte[] runEncryptDecrypt(SecretKey key, String algorithm) throws Exception {
186         byte[] indata = input.clone();
187         byte[] cipher = encrypt(indata, key, algorithm);
188         byte[] plain = decrypt(cipher, key, algorithm);
189         Assert.assertArrayEquals(indata, plain);
190         return plain;
191     }
