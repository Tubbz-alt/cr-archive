 63     public static void assignFieldsConditionally(boolean choice) {

139     static class SomeClass {

141         static void method() {
142         }
143 
144         @BytecodeParserForceInline
145         static void inlinedMethod() {
146         }
147     }
148 
149     public static void invokestatic() {
150         SomeClass.method();
151     }
152 
153     public static void invokestaticInlined() {
154         SomeClass.inlinedMethod();
155     }
156 
157     private void test(String name, int initNodesAfterParse, int initNodesAfterOpt) {
158         StructuredGraph graph = parseEager(name, AllowAssumptions.NO);
159         Assert.assertEquals(initNodesAfterParse, graph.getNodes().filter(InitializeKlassNode.class).count());
160         HighTierContext highTierContext = getDefaultHighTierContext();
161         new EliminateRedundantInitializationPhase().apply(graph, highTierContext);
162         Assert.assertEquals(initNodesAfterOpt, graph.getNodes().filter(InitializeKlassNode.class).count());
163     }
