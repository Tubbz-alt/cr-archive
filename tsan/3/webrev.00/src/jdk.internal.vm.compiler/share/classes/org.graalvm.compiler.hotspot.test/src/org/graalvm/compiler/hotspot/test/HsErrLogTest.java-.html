 49 /**
 50  * Tests that a hs_err crash log contains expected content.
 51  */
 52 public class HsErrLogTest extends GraalCompilerTest {
 53 
 54     @Test
 55     public void test1() throws IOException, InterruptedException {
 56         List<String> args = new ArrayList<>();
 57         if (Java8OrEarlier) {
 58             args.add("-XX:-UseJVMCIClassLoader");
 59         }
 60         args.add("-XX:+UseJVMCICompiler");
 61         args.add("-XX:CompileOnly=" + Crasher.class.getName() + "::tryCrash");
 62         args.add(Crasher.class.getName());
 63         testHelper(args);
 64     }
 65 
 66     private static final boolean VERBOSE = Boolean.getBoolean(HsErrLogTest.class.getSimpleName() + ".verbose");
 67 
 68     private static void testHelper(List<String> extraVmArgs, String... mainClassAndArgs) throws IOException, InterruptedException {
 69         List<String> vmArgs = withoutDebuggerArguments(getVMCommandLine());
 70         vmArgs.removeIf(a -> a.startsWith("-Dgraal."));
 71         vmArgs.remove("-esa");
 72         vmArgs.remove("-ea");
 73         vmArgs.addAll(extraVmArgs);
 74 
 75         Subprocess proc = SubprocessUtil.java(vmArgs, mainClassAndArgs);
 76         if (VERBOSE) {
 77             System.out.println(proc);
 78         }
 79 
 80         Pattern re = Pattern.compile("# +(.*hs_err_pid[\\d]+\\.log)");
 81 
 82         for (String line : proc.output) {
 83             Matcher m = re.matcher(line);
 84             if (m.matches()) {
 85                 File path = new File(m.group(1));
 86                 Assert.assertTrue(path.toString(), path.exists());
 87                 checkHsErr(path);
 88                 return;
 89             }
 90         }
 91 
 92         Assert.fail("Could not find " + re.pattern());
 93     }
 94 
 95     private static void checkHsErr(File hsErrPath) {
 96         try (BufferedReader br = new BufferedReader(new FileReader(hsErrPath))) {
 97             String line = br.readLine();
 98             String sig = Crasher.class.getName() + ".tryCrash(JI)I";
 99             List<String> lines = new ArrayList<>();
100             while (line != null) {
101                 if (line.contains(sig)) {
102                     if (!VERBOSE) {
103                         hsErrPath.delete();
104                     }
105                     return;
106                 }
107                 lines.add(line);
108                 line = br.readLine();
109             }
110             throw new AssertionError("Could not find line containing \"" + sig + "\" in " + hsErrPath +
111                             ":" + System.lineSeparator() + String.join(System.lineSeparator(), lines));
112         } catch (IOException e) {
113             throw new AssertionError(e);
114         }
115     }
116 }
117 
118 class Crasher {
119     public static void main(String[] args) {
120         int iter = 0;
121         long mem = UNSAFE.allocateMemory(1000);
122         while (iter < Integer.MAX_VALUE) {
123             tryCrash(mem, iter);
124             iter++;
125         }
126     }
127 
128     protected static int tryCrash(long mem, int iter) {
129         if (GraalDirectives.inCompiledCode()) {
130             UNSAFE.putInt(0, iter);
131             return 0;
132         } else {
133             UNSAFE.putInt(mem, iter);
134             return UNSAFE.getInt(mem);
135         }
136     }
137 
138     static final Unsafe UNSAFE = initUnsafe();
139 
140     private static Unsafe initUnsafe() {
141         try {
142             // Fast path when we are trusted.
143             return Unsafe.getUnsafe();
144         } catch (SecurityException se) {
145             // Slow path when we are not trusted.
146             try {
147                 Field theUnsafe = Unsafe.class.getDeclaredField("theUnsafe");
148                 theUnsafe.setAccessible(true);
149                 return (Unsafe) theUnsafe.get(Unsafe.class);
150             } catch (Exception e) {
151                 throw new RuntimeException("exception while trying to get Unsafe", e);
152             }
153         }
154     }
