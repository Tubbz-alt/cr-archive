 48 /**
 49  * Tests the intrinsification of certain crypto methods.
 50  */
 51 public class HotSpotCryptoSubstitutionTest extends HotSpotGraalCompilerTest {
 52 
 53     @Override
 54     protected InstalledCode addMethod(DebugContext debug, ResolvedJavaMethod method, CompilationResult compResult) {
 55         return getBackend().createDefaultInstalledCode(debug, method, compResult);
 56     }
 57 
 58     SecretKey aesKey;
 59     SecretKey desKey;
 60     byte[] input;
 61     ByteArrayOutputStream aesExpected = new ByteArrayOutputStream();
 62     ByteArrayOutputStream desExpected = new ByteArrayOutputStream();
 63 
 64     public HotSpotCryptoSubstitutionTest() throws Exception {
 65         byte[] seed = {0x4, 0x7, 0x1, 0x1};
 66         SecureRandom random = new SecureRandom(seed);
 67         KeyGenerator aesKeyGen = KeyGenerator.getInstance("AES");
 68         KeyGenerator desKeyGen = KeyGenerator.getInstance("DESede");
 69         aesKeyGen.init(128, random);
 70         desKeyGen.init(168, random);
 71         aesKey = aesKeyGen.generateKey();
 72         desKey = desKeyGen.generateKey();
 73         input = readClassfile16(getClass());
 74 
 75         aesExpected.write(runEncryptDecrypt(aesKey, "AES/CBC/NoPadding"));
 76         aesExpected.write(runEncryptDecrypt(aesKey, "AES/CBC/PKCS5Padding"));
 77 
 78         desExpected.write(runEncryptDecrypt(desKey, "DESede/CBC/NoPadding"));
 79         desExpected.write(runEncryptDecrypt(desKey, "DESede/CBC/PKCS5Padding"));
 80     }
 81 
 82     @Test
 83     public void testAESCryptIntrinsics() throws Exception {
 84         if (compileAndInstall("com.sun.crypto.provider.AESCrypt", HotSpotGraphBuilderPlugins.aesEncryptName, HotSpotGraphBuilderPlugins.aesDecryptName)) {
 85             ByteArrayOutputStream actual = new ByteArrayOutputStream();
 86             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/NoPadding"));
 87             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/PKCS5Padding"));
 88             Assert.assertArrayEquals(aesExpected.toByteArray(), actual.toByteArray());
 89         }
 90     }
 91 
 92     @Test
 93     public void testCipherBlockChainingIntrinsics() throws Exception {
 94         if (compileAndInstall("com.sun.crypto.provider.CipherBlockChaining", HotSpotGraphBuilderPlugins.cbcEncryptName, HotSpotGraphBuilderPlugins.cbcDecryptName)) {
 95             ByteArrayOutputStream actual = new ByteArrayOutputStream();
 96             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/NoPadding"));
 97             actual.write(runEncryptDecrypt(aesKey, "AES/CBC/PKCS5Padding"));
 98             Assert.assertArrayEquals(aesExpected.toByteArray(), actual.toByteArray());
 99 
100             actual.reset();
101             actual.write(runEncryptDecrypt(desKey, "DESede/CBC/NoPadding"));
102             actual.write(runEncryptDecrypt(desKey, "DESede/CBC/PKCS5Padding"));
103             Assert.assertArrayEquals(desExpected.toByteArray(), actual.toByteArray());
104         }
105     }
106 
107     /**
108      * Compiles and installs the substitution for some specified methods. Once installed, the next
109      * execution of the methods will use the newly installed code.
110      *
111      * @param className the name of the class for which substitutions are available
112      * @param methodNames the names of the substituted methods
113      * @return true if at least one substitution was compiled and installed
114      */
115     private boolean compileAndInstall(String className, String... methodNames) {
116         if (!runtime().getVMConfig().useAESIntrinsics) {
117             return false;
118         }
119         Class<?> c;
120         try {
121             c = Class.forName(className);
122         } catch (ClassNotFoundException e) {
123             // It's ok to not find the class - a different security provider
124             // may have been installed
125             return false;
126         }
127         boolean atLeastOneCompiled = false;
128         for (String methodName : methodNames) {
129             if (compileAndInstallSubstitution(c, methodName) != null) {
130                 atLeastOneCompiled = true;
131             }
132         }
133         return atLeastOneCompiled;
134     }
135 
136     AlgorithmParameters algorithmParameters;
137 
138     private byte[] encrypt(byte[] indata, SecretKey key, String algorithm) throws Exception {
139 
140         byte[] result = indata;
141 
142         Cipher c = Cipher.getInstance(algorithm);
143         c.init(Cipher.ENCRYPT_MODE, key);
144         algorithmParameters = c.getParameters();
145 
146         byte[] r1 = c.update(result);
147         byte[] r2 = c.doFinal();
148 
149         result = new byte[r1.length + r2.length];
150         System.arraycopy(r1, 0, result, 0, r1.length);
151         System.arraycopy(r2, 0, result, r1.length, r2.length);
152 
153         return result;
154     }
155 
156     private byte[] decrypt(byte[] indata, SecretKey key, String algorithm) throws Exception {
157 
158         byte[] result = indata;
159 
160         Cipher c = Cipher.getInstance(algorithm);
161         c.init(Cipher.DECRYPT_MODE, key, algorithmParameters);
162 
163         byte[] r1 = c.update(result);
164         byte[] r2 = c.doFinal();
165 
166         result = new byte[r1.length + r2.length];
167         System.arraycopy(r1, 0, result, 0, r1.length);
168         System.arraycopy(r2, 0, result, r1.length, r2.length);
169         return result;
170     }
171 
172     private static byte[] readClassfile16(Class<? extends HotSpotCryptoSubstitutionTest> c) throws IOException {
173         String classFilePath = "/" + c.getName().replace('.', '/') + ".class";
174         InputStream stream = c.getResourceAsStream(classFilePath);
175         int bytesToRead = stream.available();
176         bytesToRead -= bytesToRead % 16;
177         byte[] classFile = new byte[bytesToRead];
178         new DataInputStream(stream).readFully(classFile);
179         return classFile;
180     }
181 
182     public byte[] runEncryptDecrypt(SecretKey key, String algorithm) throws Exception {
183         byte[] indata = input.clone();
184         byte[] cipher = encrypt(indata, key, algorithm);
185         byte[] plain = decrypt(cipher, key, algorithm);
186         Assert.assertArrayEquals(indata, plain);
187         return plain;
188     }
