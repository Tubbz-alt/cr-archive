45 /**
46  * A Graal compiler test that needs access to the {@link HotSpotGraalRuntimeProvider}.
47  */
48 public abstract class HotSpotGraalCompilerTest extends GraalCompilerTest {
49 
50     /**
51      * Gets the {@link HotSpotGraalRuntimeProvider}.
52      */
53     protected HotSpotGraalRuntimeProvider runtime() {
54         return ((HotSpotBackend) getBackend()).getRuntime();
55     }
56 
57     /**
58      * Checks that the {@code UseJVMCICompiler} flag is false.
59      *
60      * @param message describes the reason the test should be ignored when Graal is the JIT
61      * @throws AssumptionViolatedException if {@code UseJVMCICompiler == true}
62      */
63     public static void assumeGraalIsNotJIT(String message) {
64         HotSpotVMConfigStore configStore = HotSpotJVMCIRuntime.runtime().getConfigStore();
65         HotSpotVMConfigAccess access = new HotSpotVMConfigAccess(configStore);
66         boolean useJVMCICompiler = access.getFlag("UseJVMCICompiler", Boolean.class);
67         Assume.assumeFalse(message, useJVMCICompiler);
68     }
69 
70     protected InstalledCode compileAndInstallSubstitution(Class<?> c, String methodName) {
71         ResolvedJavaMethod method = getMetaAccess().lookupJavaMethod(getMethod(c, methodName));
72         HotSpotGraalRuntimeProvider rt = (HotSpotGraalRuntimeProvider) Graal.getRequiredCapability(RuntimeProvider.class);
73         HotSpotProviders providers = rt.getHostBackend().getProviders();
74         CompilationIdentifier compilationId = runtime().getHostBackend().getCompilationIdentifier(method);
75         OptionValues options = getInitialOptions();
76         StructuredGraph graph = providers.getReplacements().getIntrinsicGraph(method, compilationId, getDebugContext(options));
77         if (graph != null) {
78             return getCode(method, graph, true, true, graph.getOptions());
79         }
80         return null;
81     }
82 }
