 46  * Exercise the execution of the SHA digest substitutions.

 48 public class TestSHASubstitutions extends HotSpotGraalCompilerTest {
 49 
 50     public byte[] testDigest(String name, byte[] data) throws NoSuchAlgorithmException {
 51         MessageDigest digest;
 52         try {
 53             digest = MessageDigest.getInstance(name, "SUN");
 54             digest.update(data);
 55             return digest.digest();
 56         } catch (NoSuchProviderException e) {
 57             return null;
 58         }
 59     }
 60 
 61     byte[] getData() {
 62         byte[] data = new byte[1024 * 16];
 63         for (int i = 0; i < data.length; i++) {
 64             data[i] = (byte) i;
 65         }
 66         return data;
 67     }
 68 
 69     GraalHotSpotVMConfig getConfig() {
 70         HotSpotGraalRuntimeProvider rt = (HotSpotGraalRuntimeProvider) Graal.getRequiredCapability(RuntimeProvider.class);
 71         return rt.getVMConfig();
 72     }
 73 
 74     @Test
 75     public void testSha1() {
 76         if (getConfig().useSHA1Intrinsics()) {
 77             testWithInstalledIntrinsic("sun.security.provider.SHA", SHASubstitutions.implCompressName, "testDigest", "SHA-1", getData());

 78         }
 79     }
 80 
 81     void testWithInstalledIntrinsic(String className, String methodName, String testSnippetName, Object... args) {
 82         Class<?> c;
 83         try {
 84             c = Class.forName(className);
 85         } catch (ClassNotFoundException e) {
 86             // It's ok to not find the class - a different security provider
 87             // may have been installed
 88             return;
 89         }
 90         InstalledCode code = null;
 91         try {
 92             ResolvedJavaMethod method = getResolvedJavaMethod(testSnippetName);
 93             Object receiver = method.isStatic() ? null : this;
 94             Result expect = executeExpected(method, receiver, args);
 95             code = compileAndInstallSubstitution(c, methodName);
 96             assertTrue("Failed to install " + methodName, code != null);
 97             testAgainstExpected(method, expect, receiver, args);
 98         } catch (AssumptionViolatedException e) {
 99             // Suppress so that subsequent calls to this method within the
100             // same Junit @Test annotated method can proceed.
101         }
102         if (code != null) {
103             code.invalidate();
104         }
105     }
106 
107     @Test
108     public void testSha256() {
109         if (getConfig().useSHA256Intrinsics()) {
110             testWithInstalledIntrinsic("sun.security.provider.SHA2", SHA2Substitutions.implCompressName, "testDigest", "SHA-256", getData());

111         }
112     }

114     @Test
115     public void testSha512() {

118         }
119     }
