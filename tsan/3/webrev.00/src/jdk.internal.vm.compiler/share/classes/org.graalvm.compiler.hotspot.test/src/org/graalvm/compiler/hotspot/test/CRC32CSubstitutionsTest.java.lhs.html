25 package org.graalvm.compiler.hotspot.test;
26 


27 import java.io.DataInputStream;
28 import java.io.InputStream;
29 import java.nio.ByteBuffer;
30 import java.util.zip.Checksum;
31 
32 import java.lang.invoke.MethodHandle;
33 import java.lang.invoke.MethodHandles;
34 import java.lang.invoke.MethodType;


35 
36 import org.graalvm.compiler.test.GraalTest;
37 import org.graalvm.compiler.core.test.GraalCompilerTest;
38 
39 import org.junit.Test;
40 
41 import static org.junit.Assume.assumeFalse;
42 
43 /**
44  * Tests compiled calls to {@link java.util.zip.CRC32C}.
45  */
46 @SuppressWarnings("javadoc")
47 public class CRC32CSubstitutionsTest extends GraalCompilerTest {
48 
49     public static long updateBytes(byte[] input, int offset, int end) throws Throwable {
50         Class<?> crcClass = Class.forName("java.util.zip.CRC32C");
51         MethodHandle newMH = MethodHandles.publicLookup().findConstructor(crcClass, MethodType.methodType(void.class));
52         Checksum crc = (Checksum) newMH.invoke();
53         crc.update(input, offset, end);
54         return crc.getValue();
55     }
56 
57     @Test
58     public void test1() throws Throwable {
59         assumeFalse(GraalTest.Java8OrEarlier);
60         String classfileName = CRC32CSubstitutionsTest.class.getSimpleName().replace('.', '/') + ".class";
61         InputStream s = CRC32CSubstitutionsTest.class.getResourceAsStream(classfileName);
62         byte[] buf = new byte[s.available()];
63         new DataInputStream(s).readFully(buf);
64         int end = buf.length;
65         for (int offset = 0; offset < buf.length; offset++) {
66             test("updateBytes", buf, offset, end);
67         }
68     }
69 
70     public static long updateByteBuffer(ByteBuffer buffer) throws Throwable {
71         Class<?> crcClass = Class.forName("java.util.zip.CRC32C");
72         MethodHandle newMH = MethodHandles.publicLookup().findConstructor(crcClass, MethodType.methodType(void.class));
73         MethodHandle updateMH = MethodHandles.publicLookup().findVirtual(crcClass, "update", MethodType.methodType(void.class, ByteBuffer.class));
74         Checksum crc = (Checksum) newMH.invoke();
75         buffer.rewind();
76         updateMH.invokeExact(crc, buffer); // Checksum.update(ByteBuffer) is also available since 9
77         return crc.getValue();
78     }
79 
80     @Test
81     public void test2() throws Throwable {
82         assumeFalse(GraalTest.Java8OrEarlier);
83         String classfileName = CRC32CSubstitutionsTest.class.getSimpleName().replace('.', '/') + ".class";
84         InputStream s = CRC32CSubstitutionsTest.class.getResourceAsStream(classfileName);
85         byte[] buf = new byte[s.available()];
86         new DataInputStream(s).readFully(buf);
87 
88         ByteBuffer directBuf = ByteBuffer.allocateDirect(buf.length);
89         directBuf.put(buf);
90         ByteBuffer heapBuf = ByteBuffer.wrap(buf);
91 
92         test("updateByteBuffer", directBuf);
93         test("updateByteBuffer", heapBuf);
94     }
95 
96 }
