 56 public class HotSpotInvokeDynamicPluginTest extends HotSpotGraalCompilerTest {
 57     @Override
 58     protected Plugins getDefaultGraphBuilderPlugins() {
 59         Plugins plugins = super.getDefaultGraphBuilderPlugins();
 60         plugins.setClassInitializationPlugin(new HotSpotClassInitializationPlugin());
 61         plugins.setInvokeDynamicPlugin(new HotSpotInvokeDynamicPlugin() {
 62             @Override
 63             public boolean isResolvedDynamicInvoke(GraphBuilderContext builder, int index, int opcode) {
 64                 // Allow invokedynamic testing with older JVMCI
 65                 ResolvedJavaMethod m = builder.getMethod();
 66                 if (m.getName().startsWith("invokeDynamic") && m.getDeclaringClass().getName().equals("Lorg/graalvm/compiler/hotspot/test/HotSpotInvokeDynamicPluginTest;")) {
 67                     return false;
 68                 }
 69                 return super.isResolvedDynamicInvoke(builder, index, opcode);
 70             }
 71 
 72             @Override
 73             public boolean supportsDynamicInvoke(GraphBuilderContext builder, int index, int opcode) {
 74                 // Allow invokehandle testing with older JVMCI
 75                 ResolvedJavaMethod m = builder.getMethod();
 76                 if (m.getName().startsWith("invokeHandle") && m.getDeclaringClass().getName().equals("Lorg/graalvm/compiler/hotspot/test/HotSpotInvokeDynamicPluginTest;")) {
 77                     return true;
 78                 }
 79                 return super.supportsDynamicInvoke(builder, index, opcode);
 80             }
 81         });
 82         return plugins;
 83     }
 84 
 85     @Override
 86     protected InlineInfo bytecodeParserShouldInlineInvoke(GraphBuilderContext b, ResolvedJavaMethod method, ValueNode[] args) {
 87         return InlineInfo.DO_NOT_INLINE_NO_EXCEPTION;
 88     }
 89 
 90     private void test(String name, int expectedResolves, int expectedStubCalls) {
 91         StructuredGraph graph = parseEager(name, AllowAssumptions.NO, new OptionValues(getInitialOptions(), GraalOptions.GeneratePIC, true));
 92         MidTierContext midTierContext = new MidTierContext(getProviders(), getTargetProvider(), OptimisticOptimizations.ALL, graph.getProfilingInfo());
 93 
 94         CanonicalizerPhase canonicalizer = new CanonicalizerPhase();
 95         Assert.assertEquals(expectedResolves, graph.getNodes().filter(ResolveDynamicConstantNode.class).count());
 96         Assert.assertEquals(0, graph.getNodes().filter(ResolveDynamicStubCall.class).count());
 97         PhaseContext context = new PhaseContext(getProviders());
 98         new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.HIGH_TIER).apply(graph, context);
 99         new GuardLoweringPhase().apply(graph, midTierContext);
100         new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.MID_TIER).apply(graph, context);
101         new FrameStateAssignmentPhase().apply(graph);
102         new LoweringPhase(canonicalizer, LoweringTool.StandardLoweringStage.LOW_TIER).apply(graph, context);
103         Assert.assertEquals(0, graph.getNodes().filter(ResolveDynamicConstantNode.class).count());
104         Assert.assertEquals(expectedStubCalls, graph.getNodes().filter(ResolveDynamicStubCall.class).count());
105     }

116     static final MethodHandle objToStringMH;
117 
118     static {
119         MethodHandle mh = null;
120         try {
121             mh = MethodHandles.lookup().findVirtual(Object.class, "toString", MethodType.methodType(String.class));
122         } catch (Exception e) {
123         }
124         objToStringMH = mh;
125     }
126 
127     // invokehandle
128     public static String invokeHandle1(Object o) throws Throwable {
129         return (String) objToStringMH.invokeExact(o);
130     }
