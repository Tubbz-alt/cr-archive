29 /**

31  */

33 
34     private static final long serialVersionUID = 792969002851591180L;
35 
36     public BailoutAndRestartBackendException(String msg) {
37         super(msg);
38     }
39 
40     public BailoutAndRestartBackendException(Throwable cause, String msg) {
41         super(cause, msg);
42     }
43 }
