25 package org.graalvm.compiler.lir.framemap;
26 
27 import java.util.BitSet;
28 import java.util.List;
29 
30 import org.graalvm.compiler.lir.VirtualStackSlot;
31 import org.graalvm.compiler.lir.gen.LIRGenerationResult;
32 
33 import jdk.vm.ci.code.CallingConvention;
34 import jdk.vm.ci.code.CodeCacheProvider;
35 import jdk.vm.ci.code.RegisterConfig;
36 import jdk.vm.ci.meta.ValueKind;
37 
38 /**
39  * A {@link FrameMapBuilder} is used to collect all information necessary to
40  * {@linkplain #buildFrameMap create} a {@link FrameMap}.
41  */
42 public abstract class FrameMapBuilder {
43 
44     /**
45      * Reserves a spill slot in the frame of the method being compiled. The returned slot is aligned
46      * on its natural alignment, i.e., an 8-byte spill slot is aligned at an 8-byte boundary, unless
47      * overridden by a subclass.
48      *
49      * @param kind The kind of the spill slot to be reserved.
50      * @return A spill slot denoting the reserved memory area.
51      */
52     public abstract VirtualStackSlot allocateSpillSlot(ValueKind<?> kind);
53 
54     /**
55      * Reserves a number of contiguous slots in the frame of the method being compiled. If the
56      * requested number of slots is 0, this method returns {@code null}.
57      *
58      * @param slots the number of slots to reserve
59      * @param objects specifies the indexes of the object pointer slots. The caller is responsible
60      *            for guaranteeing that each such object pointer slot is initialized before any
61      *            instruction that uses a reference map. Without this guarantee, the garbage
62      *            collector could see garbage object values.
63      * @param outObjectStackSlots if non-null, the object pointer slots allocated are added to this
64      *            list
65      * @return the first reserved stack slot (i.e., at the lowest address)
66      */
67     public abstract VirtualStackSlot allocateStackSlots(int slots, BitSet objects, List<VirtualStackSlot> outObjectStackSlots);
68 
69     public abstract RegisterConfig getRegisterConfig();
70 
71     public abstract CodeCacheProvider getCodeCache();
72 
73     /**
74      * Informs the frame map that the compiled code calls a particular method, which may need stack
75      * space for outgoing arguments.
76      *
77      * @param cc The calling convention for the called method.
78      */
79     public abstract void callsMethod(CallingConvention cc);
80 
81     /**
82      * Creates a {@linkplain FrameMap} based on the information collected by this
83      * {@linkplain FrameMapBuilder}.
84      */
85     public abstract FrameMap buildFrameMap(LIRGenerationResult result);
86 }
