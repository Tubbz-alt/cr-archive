 53 /**
 54  * A FrameMapBuilder that records allocation.
 55  */
 56 public class FrameMapBuilderImpl extends FrameMapBuilderTool {
 57 
 58     private final RegisterConfig registerConfig;
 59     private final CodeCacheProvider codeCache;
 60     private final FrameMap frameMap;
 61     private final List<VirtualStackSlot> stackSlots;
 62     private final List<CallingConvention> calls;
 63     private int numStackSlots;
 64 
 65     public FrameMapBuilderImpl(FrameMap frameMap, CodeCacheProvider codeCache, RegisterConfig registerConfig) {
 66         assert registerConfig != null : "No register config!";
 67         this.registerConfig = registerConfig == null ? codeCache.getRegisterConfig() : registerConfig;
 68         this.codeCache = codeCache;
 69         this.frameMap = frameMap;
 70         this.stackSlots = new ArrayList<>();
 71         this.calls = new ArrayList<>();
 72         this.numStackSlots = 0;
 73     }
 74 
 75     @Override
 76     public VirtualStackSlot allocateSpillSlot(ValueKind<?> kind) {
 77         SimpleVirtualStackSlot slot = new SimpleVirtualStackSlot(numStackSlots++, kind);
 78         stackSlots.add(slot);
 79         return slot;
 80     }
 81 
 82     @Override
 83     public VirtualStackSlot allocateStackSlots(int slots, BitSet objects, List<VirtualStackSlot> outObjectStackSlots) {
 84         if (slots == 0) {
 85             return null;
 86         }
 87         if (outObjectStackSlots != null) {
 88             throw GraalError.unimplemented();
 89         }
 90         VirtualStackSlotRange slot = new VirtualStackSlotRange(numStackSlots++, slots, objects, LIRKind.fromJavaKind(frameMap.getTarget().arch, JavaKind.Object));
 91         stackSlots.add(slot);
 92         return slot;
 93     }
 94 
 95     @Override
 96     public RegisterConfig getRegisterConfig() {
 97         return registerConfig;
 98     }
 99 
100     @Override
101     public CodeCacheProvider getCodeCache() {
102         return codeCache;
103     }
104 
105     @Override
106     public FrameMap getFrameMap() {
107         return frameMap;
108     }
109 
110     @Override
111     public int getNumberOfStackSlots() {
112         return numStackSlots;
113     }
114 
115     @Override
116     public void callsMethod(CallingConvention cc) {
117         calls.add(cc);
118     }
119 
120     @Override
121     @SuppressWarnings("try")
122     public FrameMap buildFrameMap(LIRGenerationResult res) {
123         DebugContext debug = res.getLIR().getDebug();
124         if (debug.areScopesEnabled()) {
125             verifyStackSlotAllocation(res);
126         }
127         for (CallingConvention cc : calls) {
128             frameMap.callsMethod(cc);
129         }
130         frameMap.finish();
131         return frameMap;
132     }
133 
134     private static void verifyStackSlotAllocation(LIRGenerationResult res) {
135         LIR lir = res.getLIR();
136         InstructionValueConsumer verifySlots = (LIRInstruction op, Value value, OperandMode mode, EnumSet<OperandFlag> flags) -> {
137             assert !isVirtualStackSlot(value) : String.format("Instruction %s contains a virtual stack slot %s", op, value);
138         };
139         for (AbstractBlockBase<?> block : lir.getControlFlowGraph().getBlocks()) {
140             lir.getLIRforBlock(block).forEach(op -> {
141                 op.visitEachInput(verifySlots);
142                 op.visitEachAlive(verifySlots);
143                 op.visitEachState(verifySlots);
144 
145                 op.visitEachTemp(verifySlots);
146                 op.visitEachOutput(verifySlots);
147             });
148         }
149     }
150 
151     @Override
152     public List<VirtualStackSlot> getStackSlots() {
153         return stackSlots;
154     }
155 
156 }
