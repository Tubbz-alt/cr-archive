25 package org.graalvm.compiler.lir.asm;
26 
27 import org.graalvm.compiler.asm.Assembler;
28 import org.graalvm.compiler.code.CompilationResult;
29 import org.graalvm.compiler.core.common.spi.ForeignCallsProvider;
30 import org.graalvm.compiler.debug.DebugContext;
31 import org.graalvm.compiler.lir.framemap.FrameMap;
32 import org.graalvm.compiler.options.OptionValues;
33 
34 import jdk.vm.ci.code.CodeCacheProvider;
35 import jdk.vm.ci.code.Register;
36 
37 /**
38  * Factory class for creating {@link CompilationResultBuilder}s.
39  */
40 public interface CompilationResultBuilderFactory {
41 
42     /**
43      * Creates a new {@link CompilationResultBuilder}.
44      */
45     CompilationResultBuilder createBuilder(CodeCacheProvider codeCache, ForeignCallsProvider foreignCalls, FrameMap frameMap, Assembler asm, DataBuilder dataBuilder, FrameContext frameContext,
46                     OptionValues options, DebugContext debug, CompilationResult compilationResult, Register nullRegister);
47 
48     /**
49      * The default factory creates a standard {@link CompilationResultBuilder}.
50      */
51     CompilationResultBuilderFactory Default = new CompilationResultBuilderFactory() {
52 
53         @Override
54         public CompilationResultBuilder createBuilder(CodeCacheProvider codeCache, ForeignCallsProvider foreignCalls, FrameMap frameMap, Assembler asm, DataBuilder dataBuilder,
55                         FrameContext frameContext, OptionValues options, DebugContext debug, CompilationResult compilationResult, Register nullRegister) {
56             return new CompilationResultBuilder(codeCache, foreignCalls, frameMap, asm, dataBuilder, frameContext, options, debug, compilationResult, nullRegister);
57         }
58     };
59 }
