 50 /**
 51  * A FrameMapBuilder that records allocation.
 52  */
 53 public class FrameMapBuilderImpl extends FrameMapBuilderTool {
 54 
 55     private final RegisterConfig registerConfig;
 56     private final CodeCacheProvider codeCache;
 57     private final FrameMap frameMap;
 58     private final List<VirtualStackSlot> stackSlots;
 59     private final List<CallingConvention> calls;
 60     private int numStackSlots;
 61 
 62     public FrameMapBuilderImpl(FrameMap frameMap, CodeCacheProvider codeCache, RegisterConfig registerConfig) {
 63         assert registerConfig != null : "No register config!";
 64         this.registerConfig = registerConfig == null ? codeCache.getRegisterConfig() : registerConfig;
 65         this.codeCache = codeCache;
 66         this.frameMap = frameMap;
 67         this.stackSlots = new ArrayList<>();
 68         this.calls = new ArrayList<>();
 69         this.numStackSlots = 0;
 70     }
 71 
 72     @Override
 73     public VirtualStackSlot allocateSpillSlot(ValueKind<?> kind) {
 74         SimpleVirtualStackSlot slot = new SimpleVirtualStackSlot(numStackSlots++, kind);
 75         stackSlots.add(slot);
 76         return slot;
 77     }
 78 
 79     @Override
 80     public VirtualStackSlot allocateStackSlots(int slots) {
 81         if (slots == 0) {
 82             return null;
 83         }
 84         VirtualStackSlotRange slot = new VirtualStackSlotRange(numStackSlots++, slots, LIRKind.value(frameMap.getTarget().arch.getWordKind()));
 85         stackSlots.add(slot);
 86         return slot;
 87     }
 88 
 89     @Override
 90     public RegisterConfig getRegisterConfig() {
 91         return registerConfig;
 92     }
 93 
 94     @Override
 95     public CodeCacheProvider getCodeCache() {
 96         return codeCache;
 97     }
 98 
 99     @Override
100     public FrameMap getFrameMap() {
101         return frameMap;
102     }
103 
104     @Override
105     public int getNumberOfStackSlots() {
106         return numStackSlots;
107     }
108 
109     @Override
110     public void callsMethod(CallingConvention cc) {
111         calls.add(cc);
112     }
113 
114     @Override
115     @SuppressWarnings("try")
116     public FrameMap buildFrameMap(LIRGenerationResult res) {
117         DebugContext debug = res.getLIR().getDebug();
118         if (debug.areScopesEnabled()) {
119             verifyStackSlotAllocation(res);
120         }
121         for (CallingConvention cc : calls) {
122             frameMap.callsMethod(cc);
123         }
124         frameMap.finish();
125         return frameMap;
126     }
127 
128     private static void verifyStackSlotAllocation(LIRGenerationResult res) {
129         LIR lir = res.getLIR();
130         InstructionValueConsumer verifySlots = (LIRInstruction op, Value value, OperandMode mode, EnumSet<OperandFlag> flags) -> {
131             assert !isVirtualStackSlot(value) : String.format("Instruction %s contains a virtual stack slot %s", op, value);
132         };
133         for (AbstractBlockBase<?> block : lir.getControlFlowGraph().getBlocks()) {
134             lir.getLIRforBlock(block).forEach(op -> {
135                 op.visitEachInput(verifySlots);
136                 op.visitEachAlive(verifySlots);
137                 op.visitEachState(verifySlots);
138 
139                 op.visitEachTemp(verifySlots);
140                 op.visitEachOutput(verifySlots);
141             });
142         }
143     }
144 
145     @Override
146     public List<VirtualStackSlot> getStackSlots() {
147         return stackSlots;
148     }
149 
150 }
