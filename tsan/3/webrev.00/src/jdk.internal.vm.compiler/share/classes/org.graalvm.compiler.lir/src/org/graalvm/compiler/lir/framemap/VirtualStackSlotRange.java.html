25 package org.graalvm.compiler.lir.framemap;
26 
27 import org.graalvm.compiler.core.common.LIRKind;
28 import org.graalvm.compiler.lir.VirtualStackSlot;
29 
30 import jdk.vm.ci.code.TargetDescription;
31 
32 /**
33  * Represents a {@link #getSlots() numbered} range of {@link VirtualStackSlot virtual stack slot} of
34  * size {@link TargetDescription#wordSize}.
35  */
36 public class VirtualStackSlotRange extends VirtualStackSlot {
37 
38     private final int slots;
39 
40     public VirtualStackSlotRange(int id, int slots, LIRKind kind) {
41         super(id, kind);
42         this.slots = slots;
43     }
44 
45     public int getSlots() {
46         return slots;
47     }
48 }
