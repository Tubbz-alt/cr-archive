25 package org.graalvm.compiler.lir;
26 
27 import org.graalvm.compiler.core.common.PermanentBailoutException;
28 import org.graalvm.compiler.options.Option;
29 import org.graalvm.compiler.options.OptionKey;
30 import org.graalvm.compiler.options.OptionType;
31 
32 /**
33  * Restarts the {@link LIR low-level} compilation with a modified configuration.
34  * {@link BailoutAndRestartBackendException.Options#LIRUnlockBackendRestart LIRUnlockBackendRestart}
35  * needs to be enabled. Use only for debugging purposes only.
36  */
37 public abstract class BailoutAndRestartBackendException extends PermanentBailoutException {
38 
39     public static class Options {
40         // @formatter:off
41         @Option(help = "Unlock backend restart feature.", type = OptionType.Debug)
42         public static final OptionKey<Boolean> LIRUnlockBackendRestart = new OptionKey<>(false);
43         // @formatter:on
44     }
45 
46     private static final long serialVersionUID = 792969002851591180L;
47 
48     public BailoutAndRestartBackendException(String msg) {
49         super(msg);
50     }
51 
52     public BailoutAndRestartBackendException(Throwable cause, String msg) {
53         super(cause, msg);
54     }
55 }
