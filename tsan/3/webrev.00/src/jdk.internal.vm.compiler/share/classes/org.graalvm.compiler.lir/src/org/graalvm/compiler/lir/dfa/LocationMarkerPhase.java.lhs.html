 45 /**
 46  * Mark all live references for a frame state. The frame state use this information to build the OOP
 47  * maps.
 48  */
 49 public final class LocationMarkerPhase extends AllocationPhase {
 50 
 51     @Override
 52     protected void run(TargetDescription target, LIRGenerationResult lirGenRes, AllocationContext context) {
 53         new Marker(lirGenRes.getLIR(), lirGenRes.getFrameMap()).build();
 54     }
 55 
 56     static final class Marker extends LocationMarker<RegStackValueSet> {
 57 
 58         private final RegisterAttributes[] registerAttributes;
 59 
 60         private Marker(LIR lir, FrameMap frameMap) {
 61             super(lir, frameMap);
 62             this.registerAttributes = frameMap.getRegisterConfig().getAttributesMap();
 63         }
 64 
 65         @Override
 66         protected RegStackValueSet newLiveValueSet() {
 67             return new RegStackValueSet(frameMap);
 68         }
 69 
 70         @Override
 71         protected boolean shouldProcessValue(Value operand) {
 72             if (isRegister(operand)) {
 73                 Register reg = asRegister(operand);
 74                 if (!reg.mayContainReference() || !attributes(reg).isAllocatable()) {
 75                     // register that's not allocatable or not part of the reference map
 76                     return false;
 77                 }
 78             } else if (!isStackSlot(operand)) {
 79                 // neither register nor stack slot
 80                 return false;
 81             }
 82 
 83             return !operand.getValueKind().equals(LIRKind.Illegal);
 84         }
 85 
 86         /**
 87          * This method does the actual marking.
 88          */
 89         @Override
 90         protected void processState(LIRInstruction op, LIRFrameState info, RegStackValueSet values) {
 91             if (!info.hasDebugInfo()) {
 92                 info.initDebugInfo(frameMap, !op.destroysCallerSavedRegisters() || !frameMap.getRegisterConfig().areAllAllocatableRegistersCallerSaved());
 93             }
 94 
 95             ReferenceMapBuilder refMap = frameMap.newReferenceMapBuilder();
 96             frameMap.addLiveValues(refMap);
 97             values.addLiveValues(refMap);
 98 
 99             info.debugInfo().setReferenceMap(refMap.finish(info));
100         }
101 
102         /**
103          * Gets an object describing the attributes of a given register according to this register
104          * configuration.
105          */
106         private RegisterAttributes attributes(Register reg) {
107             return registerAttributes[reg.number];
108         }
109 
110     }
111 }
