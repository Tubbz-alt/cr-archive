 65         private final int directCount;

 69             this(mode.directCount, mode.values);

 73         public Values(int directCount, ArrayList<ValueFieldInfo> fields) {
 74             super(fields);
 75             this.directCount = directCount;
 76             flags = (EnumSet<OperandFlag>[]) new EnumSet<?>[fields.size()];
 77             for (int i = 0; i < fields.size(); i++) {
 78                 flags[i] = fields.get(i).flags;
 79             }

112     protected static class ValueFieldInfo extends FieldsScanner.FieldInfo {
113 
114         final EnumSet<OperandFlag> flags;
115 
116         public ValueFieldInfo(long offset, String name, Class<?> type, Class<?> declaringClass, EnumSet<OperandFlag> flags) {
117             super(offset, name, type, declaringClass);
118             assert VALUE_ARRAY_CLASS.isAssignableFrom(type) || VALUE_CLASS.isAssignableFrom(type);
119             this.flags = flags;
120         }
121 
122         /**
123          * Sorts non-array fields before array fields.
124          */
125         @Override
126         public int compareTo(FieldsScanner.FieldInfo o) {
127             if (VALUE_ARRAY_CLASS.isAssignableFrom(o.type)) {
128                 if (!VALUE_ARRAY_CLASS.isAssignableFrom(type)) {
129                     return -1;
130                 }
131             } else {
132                 if (VALUE_ARRAY_CLASS.isAssignableFrom(type)) {
133                     return 1;
134                 }
135             }
136             return super.compareTo(o);
137         }
138 
139         @Override
140         public String toString() {
141             return super.toString() + flags;
142         }
143     }

145     protected static class OperandModeAnnotation {
146 
147         /**
148          * Number of non-array fields in {@link #values}.
149          */
150         public int directCount;
151         public final ArrayList<ValueFieldInfo> values = new ArrayList<>();
152     }

154     protected abstract static class LIRFieldsScanner extends FieldsScanner {
155 
156         public final EconomicMap<Class<? extends Annotation>, OperandModeAnnotation> valueAnnotations;
157         public final ArrayList<FieldsScanner.FieldInfo> states = new ArrayList<>();
158 
159         public LIRFieldsScanner(FieldsScanner.CalcOffset calc) {
160             super(calc);
161             valueAnnotations = EconomicMap.create(Equivalence.DEFAULT);
162         }
163 
164         protected OperandModeAnnotation getOperandModeAnnotation(Field field) {
165             OperandModeAnnotation result = null;
166             MapCursor<Class<? extends Annotation>, OperandModeAnnotation> cursor = valueAnnotations.getEntries();
167             while (cursor.advance()) {
168                 Annotation annotation = field.getAnnotation(cursor.getKey());
169                 if (annotation != null) {
170                     assert result == null : "Field has two operand mode annotations: " + field;
171                     result = cursor.getValue();
172                 }
173             }
174             return result;
175         }
176 
177         protected abstract EnumSet<OperandFlag> getFlags(Field field);
178 
179         @Override
180         protected void scanField(Field field, long offset) {
181             Class<?> type = field.getType();
182             if (VALUE_CLASS.isAssignableFrom(type) && !CONSTANT_VALUE_CLASS.isAssignableFrom(type)) {
183                 assert !Modifier.isFinal(field.getModifiers()) : "Value field must not be declared final because it is modified by register allocator: " + field;
184                 OperandModeAnnotation annotation = getOperandModeAnnotation(field);
185                 assert annotation != null : "Field must have operand mode annotation: " + field;
186                 EnumSet<OperandFlag> flags = getFlags(field);
187                 assert verifyFlags(field, type, flags);
188                 annotation.values.add(new ValueFieldInfo(offset, field.getName(), type, field.getDeclaringClass(), flags));
189                 annotation.directCount++;
190             } else if (VALUE_ARRAY_CLASS.isAssignableFrom(type)) {
191                 OperandModeAnnotation annotation = getOperandModeAnnotation(field);
192                 assert annotation != null : "Field must have operand mode annotation: " + field;
193                 EnumSet<OperandFlag> flags = getFlags(field);
194                 assert verifyFlags(field, type.getComponentType(), flags);
195                 annotation.values.add(new ValueFieldInfo(offset, field.getName(), type, field.getDeclaringClass(), flags));
196             } else {
197                 assert getOperandModeAnnotation(field) == null : "Field must not have operand mode annotation: " + field;
198                 assert field.getAnnotation(LIRInstruction.State.class) == null : "Field must not have state annotation: " + field;
199                 super.scanField(field, offset);
200             }
201         }
202 
203         private static boolean verifyFlags(Field field, Class<?> type, EnumSet<OperandFlag> flags) {
204             if (flags.contains(REG)) {
205                 assert type.isAssignableFrom(REGISTER_VALUE_CLASS) || type.isAssignableFrom(VARIABLE_CLASS) : "Cannot assign RegisterValue / Variable to field with REG flag:" + field;
206             }
207             if (flags.contains(STACK)) {
208                 assert type.isAssignableFrom(STACK_SLOT_CLASS) : "Cannot assign StackSlot to field with STACK flag:" + field;
209             }
210             if (flags.contains(CONST)) {
211                 assert type.isAssignableFrom(CONSTANT_VALUE_CLASS) : "Cannot assign Constant to field with CONST flag:" + field;
212             }
213             return true;
214         }
215     }

217     protected static void forEach(LIRInstruction inst, Values values, OperandMode mode, InstructionValueProcedure proc) {
218         for (int i = 0; i < values.getCount(); i++) {
219             assert LIRInstruction.ALLOWED_FLAGS.get(mode).containsAll(values.getFlags(i));
220 
221             if (i < values.getDirectCount()) {
222                 Value value = values.getValue(inst, i);
223                 Value newValue;
224                 if (value instanceof CompositeValue) {
225                     CompositeValue composite = (CompositeValue) value;
226                     newValue = composite.forEachComponent(inst, mode, proc);
227                 } else {
228                     newValue = proc.doValue(inst, value, mode, values.getFlags(i));
229                 }
230                 if (!value.identityEquals(newValue)) {



231                     values.setValue(inst, i, newValue);
232                 }
233             } else {
234                 Value[] valueArray = values.getValueArray(inst, i);
235                 for (int j = 0; j < valueArray.length; j++) {
236                     Value value = valueArray[j];
237                     Value newValue;
238                     if (value instanceof CompositeValue) {
239                         CompositeValue composite = (CompositeValue) value;
240                         newValue = composite.forEachComponent(inst, mode, proc);
241                     } else {
242                         newValue = proc.doValue(inst, value, mode, values.getFlags(i));
243                     }
244                     if (!value.identityEquals(newValue)) {
245                         valueArray[j] = newValue;
246                     }
247                 }
248             }
249         }
250     }

252     protected static void visitEach(LIRInstruction inst, Values values, OperandMode mode, InstructionValueConsumer proc) {
253         for (int i = 0; i < values.getCount(); i++) {
254             assert LIRInstruction.ALLOWED_FLAGS.get(mode).containsAll(values.getFlags(i));
255 
256             if (i < values.getDirectCount()) {
257                 Value value = values.getValue(inst, i);
258                 if (value instanceof CompositeValue) {
259                     CompositeValue composite = (CompositeValue) value;
260                     composite.visitEachComponent(inst, mode, proc);
261                 } else {
262                     proc.visitValue(inst, value, mode, values.getFlags(i));
263                 }
264             } else {
265                 Value[] valueArray = values.getValueArray(inst, i);
266                 for (int j = 0; j < valueArray.length; j++) {
267                     Value value = valueArray[j];
268                     if (value instanceof CompositeValue) {
269                         CompositeValue composite = (CompositeValue) value;
270                         composite.visitEachComponent(inst, mode, proc);
271                     } else {
272                         proc.visitValue(inst, value, mode, values.getFlags(i));
273                     }
274                 }
275             }
276         }
277     }

279     protected static void appendValues(StringBuilder sb, Object obj, String start, String end, String startMultiple, String endMultiple, String[] prefix, Fields... fieldsList) {
280         int total = 0;
281         for (Fields fields : fieldsList) {
282             total += fields.getCount();
283         }
284         if (total == 0) {
285             return;
286         }
287 
288         sb.append(start);
289         if (total > 1) {
290             sb.append(startMultiple);
291         }
292         String sep = "";
293         int i = 0;
294         for (Fields fields : fieldsList) {
295             for (int j = 0; j < fields.getCount(); j++) {
296                 sb.append(sep).append(prefix[i]);
297                 if (total > 1) {
298                     sb.append(fields.getName(j)).append(": ");
299                 }
300                 sb.append(getFieldString(obj, j, fields));
301                 sep = ", ";
302             }
303             i++;
304         }
305         if (total > 1) {
306             sb.append(endMultiple);
307         }
308         sb.append(end);
309     }

311     protected static String getFieldString(Object obj, int index, Fields fields) {
312         Object value = fields.get(obj, index);
313         Class<?> type = fields.getType(index);
314         if (value == null || type.isPrimitive() || !type.isArray()) {
315             return String.valueOf(value);
316         }
317         if (type == int[].class) {
318             return Arrays.toString((int[]) value);
319         } else if (type == double[].class) {
320             return Arrays.toString((double[]) value);
321         } else if (type == byte[].class) {
322             byte[] byteValue = (byte[]) value;
323             if (isPrintableAsciiString(byteValue)) {
324                 return toString(byteValue);
325             } else {
326                 return Arrays.toString(byteValue);
327             }
328         } else if (!type.getComponentType().isPrimitive()) {
329             return Arrays.toString((Object[]) value);
330         }
331         assert false : "unhandled field type: " + type;
332         return "";
333     }

335     /**
336      * Tests if all values in this string are printable ASCII characters or value \0 (b in
337      * [0x20,0x7F]) or b == 0.
338      *
339      * @param array
340      * @return true if there are only printable ASCII characters and \0, false otherwise
341      */
342     private static boolean isPrintableAsciiString(byte[] array) {
343         for (byte b : array) {
344             char c = (char) b;
345             if (c != 0 && (c < 0x20 || c > 0x7F)) {
346                 return false;
347             }
348         }
349         return true;
350     }
351 
352     private static String toString(byte[] bytes) {
353         StringBuilder sb = new StringBuilder();
354         sb.append('"');
355         for (byte b : bytes) {
356             if (b == 0) {
357                 sb.append("\\0");
358             } else if (b == '"') {
359                 sb.append("\\\"");
360             } else if (b == '\n') {
361                 sb.append("\\n");
362             } else {
363                 sb.append((char) b);
364             }
365         }
366         sb.append('"');
367         return sb.toString();
368     }
