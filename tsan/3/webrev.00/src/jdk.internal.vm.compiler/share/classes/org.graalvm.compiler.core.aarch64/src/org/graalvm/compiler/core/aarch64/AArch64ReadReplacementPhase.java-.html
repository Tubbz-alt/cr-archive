27 package org.graalvm.compiler.core.aarch64;
28 
29 import org.graalvm.compiler.graph.Node;
30 import org.graalvm.compiler.nodes.StructuredGraph;
31 import org.graalvm.compiler.nodes.calc.SignExtendNode;
32 import org.graalvm.compiler.nodes.calc.ZeroExtendNode;
33 import org.graalvm.compiler.nodes.memory.ReadNode;
34 import org.graalvm.compiler.phases.Phase;
35 
36 /**
37  * AArch64-specific phase which substitutes certain read nodes with arch-specific variants in order
38  * to allow merging of zero and sign extension into the read operation.
39  */
40 
41 public class AArch64ReadReplacementPhase extends Phase {
42     @Override
43     protected void run(StructuredGraph graph) {
44         for (Node node : graph.getNodes()) {
45             // don't process nodes we just added
46             if (node instanceof AArch64ReadNode) {
47                 continue;
48             }
49             if (node instanceof ReadNode) {
50                 ReadNode readNode = (ReadNode) node;
51                 if (readNode.hasExactlyOneUsage()) {
52                     Node usage = readNode.getUsageAt(0);
53                     if (usage instanceof ZeroExtendNode || usage instanceof SignExtendNode) {
54                         AArch64ReadNode.replace(readNode);
55                     }
56                 }
57             }
58         }
59     }
60 }
