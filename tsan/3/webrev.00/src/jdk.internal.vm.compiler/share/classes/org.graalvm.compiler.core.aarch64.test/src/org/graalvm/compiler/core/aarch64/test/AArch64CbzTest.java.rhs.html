 27 package org.graalvm.compiler.core.aarch64.test;
 28 
 29 import org.graalvm.compiler.lir.LIRInstruction;
 30 import org.graalvm.compiler.lir.aarch64.AArch64ControlFlow;
 31 import org.junit.Test;
 32 
 33 import java.util.function.Predicate;
 34 
 35 public class AArch64CbzTest extends AArch64MatchRuleTest {
 36     private static final Predicate<LIRInstruction> predicate = op -> (op instanceof AArch64ControlFlow.CompareBranchZeroOp);
 37 
 38     public static int equalsTo(int x) {
 39         if (x == 0) {
 40             return 1;
 41         } else {
 42             return x - 1;
 43         }
 44     }
 45 
 46     public static int notEqualsTo(int x) {
 47         if (x != 0) {
 48             return x + 2;
 49         } else {
 50             return 3;
 51         }
 52     }
 53 
 54     public static String isNull(String s) {
 55         if (s == null) {
 56             return "abc";
 57         } else {
 58             return s + "abc";
 59         }
 60     }
 61 
 62     public static String isNotNull(String s) {
 63         if (s != null) {
 64             return s + "abc";
 65         } else {
 66             return "abc";
 67         }
 68     }
 69 








 70     public static String objectEquals(String s1, String s2) {
 71         if (s1.equals(s2)) {
 72             return s1 + "abc";
 73         } else {
 74             return s2 + "abd";
 75         }
 76     }
 77 
 78     @Test
 79     public void testEqualsTo() {
 80         test("equalsTo", 0);
 81         test("equalsTo", 1);
 82         checkLIR("equalsTo", predicate, 1);
 83     }
 84 
 85     @Test
 86     public void testNotEqualsTo() {
 87         test("notEqualsTo", 0);
 88         test("notEqualsTo", 1);
 89         checkLIR("notEqualsTo", predicate, 1);
 90     }
 91 
 92     @Test
 93     public void testIsNull() {
 94         test("isNull", new Object[]{null});
 95         test("isNull", "abc");
 96         checkLIR("isNull", predicate, 1);
 97     }
 98 
 99     @Test
100     public void testIsNotNull() {
101         test("isNotNull", new Object[]{null});
102         test("isNotNull", "abc");
103         checkLIR("isNotNull", predicate, 1);
104     }
105 







106     @Test
107     public void testObjectEquals() {
108         test("objectEquals", "ab", "ac");
109         test("objectEquals", "abc", "abc");
110         checkLIR("objectEquals", predicate, 0);
111     }
112 }
