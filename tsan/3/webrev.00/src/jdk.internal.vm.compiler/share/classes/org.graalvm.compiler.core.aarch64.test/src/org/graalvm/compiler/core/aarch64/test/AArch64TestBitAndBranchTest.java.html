 63     public static long testBit42Snippet(long a, long b, long c) {
 64         if ((a & (1L << 42)) == 0) {
 65             return b;
 66         } else {
 67             return c;
 68         }
 69     }

 71     @Test
 72     public void testBit42() {
 73         test("testBit42Snippet", 1L << 42, Long.MAX_VALUE, Long.MIN_VALUE);
 74         test("testBit42Snippet", ~(1L << 42), Long.MAX_VALUE, Long.MIN_VALUE);
 75         checkLIR("testBit42Snippet", checkForBitTestAndBranchOp, 1);
 76     }

 78     private static final LargeOpSpec largeOpSingleNop = new LargeOpSpec((1 << 14 - 2) - 10, 2);
 79 
 80     /**
 81      * Tests the graceful case, where the estimation for
 82      * {@link CompilationResultBuilder#labelWithinRange(LIRInstruction, org.graalvm.compiler.asm.Label, int)}
 83      * holds.
 84      */
 85     public static int testBitTestAndBranchSingleSnippet(int a) {
 86         int res;
 87         if (a % 2 == 0) {
 88             res = fillOps(largeOpSingleNop, 1);
 89         } else {
 90             res = fillOps(largeOpSingleNop, 2);
 91         }
 92         return GraalDirectives.opaque(res);
 93     }

191     private static class LargeOpSpec extends LIRTestSpecification {
192         private final int n;
193         private final int nopCount;
194 
195         LargeOpSpec(int n, int nopCount) {
196             super();
197             this.n = n;
198             this.nopCount = nopCount;
199         }
200 
201         @Override
202         public void generate(LIRGeneratorTool gen, Value a) {
203             for (int i = 0; i < n; i++) {
204                 gen.append(new NoOp(nopCount));
205             }
206             setResult(a);
207         }
208     }
209 
210     public static class NoOp extends AArch64LIRInstruction {
211         private static final LIRInstructionClass<NoOp> TYPE = LIRInstructionClass.create(NoOp.class);
212         private final int nopCount;
213 
214         public NoOp(int nopCount) {
215             super(TYPE);
216             this.nopCount = nopCount;
217         }
218 
219         @Override
220         protected void emitCode(CompilationResultBuilder crb, AArch64MacroAssembler masm) {
221             for (int i = 0; i < nopCount; i++) {
222                 masm.nop();
223             }
224         }
225     }
226 
227     @LIRIntrinsic
228     public static int fillOps(@SuppressWarnings("unused") LargeOpSpec s, int a) {
229         return a;
230     }
231 
232     @Override
233     protected LIRSuites createLIRSuites(OptionValues options) {
234         LIRSuites suites = super.createLIRSuites(options);
235         suites.getPreAllocationOptimizationStage().appendPhase(new CheckPhase());
236         return suites;
237     }
238 
239     public class CheckPhase extends LIRPhase<PreAllocationOptimizationContext> {
240         @Override
241         protected void run(
242                         TargetDescription target, LIRGenerationResult lirGenRes,
243                         PreAllocationOptimizationContext context) {
244             lir = lirGenRes.getLIR();
245         }
246     }
247 
248     protected void checkLIR(String methodName, Predicate<LIRInstruction> predicate, int expected) {
249         compile(getResolvedJavaMethod(methodName), null);
250         int actualOpNum = 0;
251         for (LIRInstruction ins : lir.getLIRforBlock(lir.codeEmittingOrder()[0])) {
252             if (predicate.test(ins)) {
253                 actualOpNum++;
254             }
255         }
256         Assert.assertEquals(expected, actualOpNum);
257     }
