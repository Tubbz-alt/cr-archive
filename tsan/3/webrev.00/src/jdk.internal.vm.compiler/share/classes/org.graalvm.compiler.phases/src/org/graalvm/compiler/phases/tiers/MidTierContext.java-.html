25 package org.graalvm.compiler.phases.tiers;
26 
27 import org.graalvm.compiler.phases.OptimisticOptimizations;
28 import org.graalvm.compiler.phases.util.Providers;
29 
30 import jdk.vm.ci.code.TargetDescription;
31 import jdk.vm.ci.meta.ProfilingInfo;
32 
33 public class MidTierContext extends PhaseContext {
34 
35     private final TargetProvider target;
36     private final OptimisticOptimizations optimisticOpts;
37     private final ProfilingInfo profilingInfo;
38 
39     public MidTierContext(Providers copyFrom, TargetProvider target, OptimisticOptimizations optimisticOpts, ProfilingInfo profilingInfo) {
40         super(copyFrom);
41         this.target = target;
42         this.optimisticOpts = optimisticOpts;
43         this.profilingInfo = profilingInfo;
44     }
45 
46     public TargetDescription getTarget() {
47         return target.getTarget();
48     }
49 
50     public TargetProvider getTargetProvider() {
51         return target;
52     }
53 
54     public OptimisticOptimizations getOptimisticOptimizations() {
55         return optimisticOpts;
56     }
57 
58     public ProfilingInfo getProfilingInfo() {
59         return profilingInfo;
60     }
61 }
