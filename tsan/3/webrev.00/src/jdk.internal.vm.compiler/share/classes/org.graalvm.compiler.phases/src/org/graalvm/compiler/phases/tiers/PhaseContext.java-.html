25 package org.graalvm.compiler.phases.tiers;
26 
27 import org.graalvm.compiler.core.common.spi.ConstantFieldProvider;
28 import org.graalvm.compiler.nodes.spi.LoweringProvider;
29 import org.graalvm.compiler.nodes.spi.Replacements;
30 import org.graalvm.compiler.nodes.spi.StampProvider;
31 import org.graalvm.compiler.phases.util.Providers;
32 
33 import jdk.vm.ci.meta.ConstantReflectionProvider;
34 import jdk.vm.ci.meta.MetaAccessProvider;
35 import org.graalvm.compiler.core.common.spi.ForeignCallsProvider;
36 
37 public class PhaseContext {
38 
39     private final MetaAccessProvider metaAccess;
40     private final ConstantReflectionProvider constantReflection;
41     private final ConstantFieldProvider constantFieldProvider;
42     private final LoweringProvider lowerer;
43     private final Replacements replacements;
44     private final StampProvider stampProvider;
45     private final ForeignCallsProvider foreignCalls;
46 
47     public PhaseContext(MetaAccessProvider metaAccess, ConstantReflectionProvider constantReflection, ConstantFieldProvider constantFieldProvider, LoweringProvider lowerer, Replacements replacements,
48                     StampProvider stampProvider, ForeignCallsProvider foreignCalls) {
49         this.metaAccess = metaAccess;
50         this.constantReflection = constantReflection;
51         this.constantFieldProvider = constantFieldProvider;
52         this.lowerer = lowerer;
53         this.replacements = replacements;
54         this.stampProvider = stampProvider;
55         this.foreignCalls = foreignCalls;
56     }
57 
58     public PhaseContext(Providers providers) {
59         this(providers.getMetaAccess(), providers.getConstantReflection(), providers.getConstantFieldProvider(), providers.getLowerer(), providers.getReplacements(), providers.getStampProvider(),
60                         providers.getForeignCalls());
61     }
62 
63     public MetaAccessProvider getMetaAccess() {
64         return metaAccess;
65     }
66 
67     public ConstantReflectionProvider getConstantReflection() {
68         return constantReflection;
69     }
70 
71     public ConstantFieldProvider getConstantFieldProvider() {
72         return constantFieldProvider;
73     }
74 
75     public LoweringProvider getLowerer() {
76         return lowerer;
77     }
78 
79     public Replacements getReplacements() {
80         return replacements;
81     }
82 
83     public StampProvider getStampProvider() {
84         return stampProvider;
85     }
86 
87     public ForeignCallsProvider getForeignCalls() {
88         return foreignCalls;
89     }
90 }
