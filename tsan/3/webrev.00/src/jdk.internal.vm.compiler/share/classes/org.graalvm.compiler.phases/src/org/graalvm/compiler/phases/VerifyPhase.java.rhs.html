25 package org.graalvm.compiler.phases;
26 
27 import org.graalvm.compiler.nodes.StructuredGraph;
28 
29 import jdk.vm.ci.meta.MetaAccessProvider;
30 import jdk.vm.ci.meta.ResolvedJavaType;
31 
32 /***
33  * Verifies a {@linkplain #verify graph} or {@linkplain #verifyClass class} against one or more
34  * invariants.


35  */
36 public abstract class VerifyPhase<C> extends BasePhase<C> {
37 
38     /**
39      * Thrown when verification performed by a {@link VerifyPhase} fails.
40      */
41     @SuppressWarnings("serial")
42     public static class VerificationError extends AssertionError {
43 
44         public VerificationError(String message) {
45             super(message);
46         }
47 
48         public VerificationError(String format, Object... args) {
49             super(String.format(format, args));
50         }
51 
52         public VerificationError(String message, Throwable cause) {
53             super(message, cause);
54         }
55     }
56 
57     @Override
58     protected final void run(StructuredGraph graph, C context) {
59         verify(graph, context);
60     }
61 
62     /**
63      * Checks {@code graph} against some invariants.
64      *
65      * @throws VerificationError if the verification fails
66      */
67     protected abstract void verify(StructuredGraph graph, C context);
68 
69     /**
70      * Checks {@code clazz} against some invariants.
71      *
72      * @param clazz the class to verify
73      * @param metaAccess an object to get a {@link ResolvedJavaType} for {@code clazz}
74      * @throws VerificationError if the class violates some invariant
75      */
76     public void verifyClass(Class<?> clazz, MetaAccessProvider metaAccess) {
77     }
78 }
