25 package org.graalvm.compiler.lir.jtt;
26 
27 import org.junit.Test;
28 
29 import org.graalvm.compiler.lir.gen.LIRGeneratorTool;
30 
31 import jdk.vm.ci.meta.Value;
32 
33 public class LIRTestTest extends LIRTest {
34     private static final LIRTestSpecification stackCopy = new LIRTestSpecification() {
35         @Override
36         public void generate(LIRGeneratorTool gen, Value a, Value b) {
37             setOutput("a", a);
38             setOutput("b", b);
39             setResult(a);
40         }
41     };
42 
43     @SuppressWarnings("unused")
44     @LIRIntrinsic
45     public static int copyInt(LIRTestSpecification spec, int a, int b) {
46         return a;
47     }
48 
49     public static int[] testGetOutput(int a, int b, int[] out) {
50         out[0] = copyInt(stackCopy, a, b);
51         out[1] = getOutput(stackCopy, "a", a);
52         out[2] = getOutput(stackCopy, "b", b);
53         return out;
54     }
55 
56     @Test
57     public void runInt() throws Throwable {
58         runTest("testGetOutput", Integer.MIN_VALUE, 0, supply(() -> new int[3]));
59         runTest("testGetOutput", -1, Integer.MAX_VALUE, supply(() -> new int[3]));
60         runTest("testGetOutput", 0, 42, supply(() -> new int[3]));
61         runTest("testGetOutput", 1, -0xFFAA44, supply(() -> new int[3]));
62         runTest("testGetOutput", Integer.MAX_VALUE, -42, supply(() -> new int[3]));
63     }
64 }
