 45 /**
 46  * Tests move from a constant to a wider stack slot (e.g. byte constant to integer stack slot).
 47  */
 48 public class ConstantStackCastTest extends LIRTest {
 49     private static PlatformKind byteKind;
 50     private static final LoadConstantStackSpec stackCopyByte = new LoadConstantStackSpec();
 51 
 52     @Before
 53     public void setup() {
 54         // Necessary to get the PlatformKind on which we're currently running on
 55         byteKind = getBackend().getTarget().arch.getPlatformKind(JavaKind.Byte);
 56         stackCopyByte.dstKind = LIRKind.fromJavaKind(getBackend().getTarget().arch, JavaKind.Int);
 57         stackCopyByte.srcKind = LIRKind.fromJavaKind(getBackend().getTarget().arch, JavaKind.Byte);
 58     }
 59 
 60     private static class LoadConstantStackSpec extends LIRTestSpecification {
 61         LIRKind dstKind;
 62         LIRKind srcKind;
 63 
 64         @Override
 65         public void generate(LIRGeneratorTool gen, Value value) {
 66             FrameMapBuilder frameMapBuilder = gen.getResult().getFrameMapBuilder();
 67             // create slots
 68             VirtualStackSlot s1 = frameMapBuilder.allocateSpillSlot(dstKind);
 69             // move stuff around
 70             Value srcValue;
 71             if (isJavaConstant(value)) {
 72                 srcValue = getConstant(srcKind, asJavaConstant(value));
 73             } else {
 74                 srcValue = value;
 75             }
 76             gen.emitMove(s1, srcValue);
 77             gen.emitBlackhole(s1);
 78             setResult(gen.emitMove(s1));
 79         }
 80 
 81         private static ConstantValue getConstant(LIRKind srcKind, JavaConstant c) {
 82             if (srcKind.getPlatformKind() == byteKind) {
 83                 JavaConstant byteConst = JavaConstant.forByte((byte) c.asInt());
 84                 return new ConstantValue(srcKind, byteConst);
 85             } else {
 86                 throw GraalError.shouldNotReachHere("Kind not supported: " + srcKind);
 87             }
 88         }
 89 
 90     }
 91 
 92     @LIRIntrinsic
 93     public static byte testCopyByte(@SuppressWarnings("unused") LoadConstantStackSpec spec, byte value) {
 94         return value;
 95     }
 96 
 97     public byte testByte(byte value) {
 98         return testCopyByte(stackCopyByte, value);
 99     }
100 
101     @Test
102     public void runByte() throws Throwable {
103         runTest("testByte", (byte) 0);
104     }
105 
106 }
