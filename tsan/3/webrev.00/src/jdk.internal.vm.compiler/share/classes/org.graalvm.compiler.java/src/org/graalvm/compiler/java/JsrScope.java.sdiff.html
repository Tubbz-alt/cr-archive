 25 package org.graalvm.compiler.java;
 26 
 27 public class JsrScope {

 28 








 29     public static final JsrScope EMPTY_SCOPE = new JsrScope();
 30 
 31     private final long scope;


 32 
 33     private JsrScope(long scope) {
 34         this.scope = scope;




 35     }
 36 
 37     public JsrScope() {
 38         this.scope = 0;


 39     }
 40 
 41     public int nextReturnAddress() {
 42         return (int) (scope & 0xffff);
 43     }
 44 
 45     public JsrScope push(int jsrReturnBci) {
 46         if ((scope & 0xffff000000000000L) != 0) {
 47             throw new JsrNotSupportedBailout("only four jsr nesting levels are supported");













 48         }
 49         return new JsrScope((scope << 16) | jsrReturnBci);




 50     }
 51 



 52     public boolean isEmpty() {
 53         return scope == 0;
 54     }
 55 





















 56     public boolean isPrefixOf(JsrScope other) {
 57         return (scope & other.scope) == scope;





 58     }
 59 





 60     public JsrScope pop() {
 61         return new JsrScope(scope >>> 16);



 62     }
 63 
 64     @Override
 65     public int hashCode() {
 66         return (int) (scope ^ (scope >>> 32));






 67     }
 68 
 69     @Override
 70     public boolean equals(Object obj) {
 71         if (this == obj) {
 72             return true;
 73         }
 74         return obj != null && getClass() == obj.getClass() && scope == ((JsrScope) obj).scope;

















 75     }
 76 
 77     @Override
 78     public String toString() {
 79         StringBuilder sb = new StringBuilder();
 80         long tmp = scope;
 81         sb.append(" [");
 82         while (tmp != 0) {
 83             sb.append(", ").append(tmp & 0xffff);
 84             tmp = tmp >>> 16;



 85         }
 86         sb.append(']');
 87         return sb.toString();
 88     }
 89 }

 25 package org.graalvm.compiler.java;
 26 
 27 import org.graalvm.compiler.bytecode.Bytecodes;
 28 import org.graalvm.compiler.java.BciBlockMapping.BciBlock;
 29 
 30 /**
 31  * Represents a subroutine entered via {@link Bytecodes#JSR} and exited via {@link Bytecodes#RET}.
 32  */
 33 public final class JsrScope {
 34 
 35     /**
 36      * The scope outside of any JSR/RET subroutine.
 37      */
 38     public static final JsrScope EMPTY_SCOPE = new JsrScope();
 39 
 40     private final char returnAddress;
 41 
 42     private final JsrScope parent;
 43 
 44     private final BciBlock jsrEntryBlock;
 45 
 46     private JsrScope(int returnBci, BciBlock jsrEntryBlock, JsrScope parent) {
 47         this.returnAddress = (char) returnBci;
 48         this.parent = parent;
 49         this.jsrEntryBlock = jsrEntryBlock;
 50     }
 51 
 52     private JsrScope() {
 53         this.returnAddress = 0;
 54         this.parent = null;
 55         this.jsrEntryBlock = null;
 56     }
 57 
 58     public int nextReturnAddress() {
 59         return returnAddress;
 60     }
 61 
 62     public BciBlock getJsrEntryBlock() {
 63         return jsrEntryBlock;
 64     }
 65 
 66     /**
 67      * Enters a new subroutine from the current scope represented by this object.
 68      *
 69      * @param returnBci the bytecode address returned to when leaving the new scope
 70      * @return an object representing the newly entered scope
 71      */
 72     public JsrScope push(int returnBci, BciBlock newJsrEntryBlock) {
 73         if (returnBci == 0) {
 74             throw new IllegalArgumentException("A bytecode subroutine cannot have a return address of 0");
 75         }
 76         if (returnBci < 1 || returnBci > 0xFFFF) {
 77             throw new IllegalArgumentException("Bytecode subroutine return address cannot be encoded as a char: " + returnBci);
 78         }
 79         return new JsrScope(returnBci, newJsrEntryBlock, this);
 80     }
 81 
 82     public JsrScope push(int returnBci) {
 83         return push(returnBci, null);
 84     }
 85 
 86     /**
 87      * Determines if this is the scope outside of any JSR/RET subroutine.
 88      */
 89     public boolean isEmpty() {
 90         return returnAddress == 0;
 91     }
 92 
 93     /**
 94      * Gets the ancestry of this scope starting with the {@link #returnAddress} of this scope's most
 95      * distant ancestor and ending with the {@link #returnAddress} of this object.
 96      *
 97      * @return a String where each character is a 16-bit BCI. This value can be converted to an
 98      *         {@code int[]} with {@code value.chars().toArray()}.
 99      */
100     public String getAncestry() {
101         String result = "";
102         for (JsrScope s = this; s != null; s = s.parent) {
103             if (!s.isEmpty()) {
104                 result = s.returnAddress + result;
105             }
106         }
107         return result;
108     }
109 
110     /**
111      * Determines if the {@linkplain #getAncestry() ancestry} of this scope is a prefix of the
112      * ancestry of {@code other}.
113      */
114     public boolean isPrefixOf(JsrScope other) {
115         if (isEmpty()) {
116             return true;
117         }
118         String ancestry = getAncestry();
119         String otherAncestry = other.getAncestry();
120         return otherAncestry.startsWith(ancestry);
121     }
122 
123     /**
124      * Gets this scope's parent.
125      *
126      * @return this scope's parent or {@link #EMPTY_SCOPE} if this is the {@link #EMPTY_SCOPE}
127      */
128     public JsrScope pop() {
129         if (isEmpty()) {
130             return this;
131         }
132         return parent;
133     }
134 
135     @Override
136     public int hashCode() {
137         int hc = returnAddress;
138         JsrScope ancestor = parent;
139         while (ancestor != null) {
140             hc = hc ^ ancestor.returnAddress;
141             ancestor = ancestor.parent;
142         }
143         return hc;
144     }
145 
146     @Override
147     public boolean equals(Object obj) {
148         if (this == obj) {
149             return true;
150         }
151         if (obj != null && getClass() == obj.getClass()) {
152             JsrScope ancestor = this;
153             JsrScope otherAncestor = (JsrScope) obj;
154             while (ancestor != null) {
155                 if (otherAncestor == null) {
156                     return false;
157                 }
158                 if (otherAncestor.returnAddress != ancestor.returnAddress) {
159                     return false;
160                 }
161                 ancestor = ancestor.parent;
162                 otherAncestor = otherAncestor.parent;
163             }
164             if (otherAncestor == null) {
165                 return true;
166             }
167         }
168         return false;
169     }
170 
171     @Override
172     public String toString() {
173         StringBuilder sb = new StringBuilder();
174 
175         for (JsrScope ancestor = this; ancestor != null; ancestor = ancestor.parent) {
176             if (!ancestor.isEmpty()) {
177                 if (sb.length() != 0) {
178                     sb.append(", ");
179                 }
180                 sb.append((int) ancestor.returnAddress);
181             }
182         }
183         return "[" + sb + "]";

184     }
185 }
