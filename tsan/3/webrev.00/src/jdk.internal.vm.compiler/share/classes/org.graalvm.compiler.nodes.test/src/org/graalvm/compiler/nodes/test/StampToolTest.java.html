25 package org.graalvm.compiler.nodes.test;
26 
27 import static org.graalvm.compiler.nodes.type.StampTool.stampForTrailingZeros;
28 import static org.graalvm.compiler.test.GraalTest.assertTrue;
29 
30 import org.graalvm.compiler.core.common.type.IntegerStamp;
31 import org.graalvm.compiler.core.common.type.Stamp;
32 import org.graalvm.compiler.core.common.type.StampFactory;
33 import org.graalvm.compiler.graph.test.GraphTest;
34 import org.junit.Test;
35 
36 public class StampToolTest extends GraphTest {
37 
38     @Test
39     public void testStampForTrailingZeros() {
40         assertIntegerStampEquals(stampForTrailingZeros(forInt(0)), 32);
41         assertIntegerStampEquals(stampForTrailingZeros(forInt(1)), 0);
42         assertIntegerStampEquals(stampForTrailingZeros(forInt(-1)), 0);
43         assertIntegerStampEquals(stampForTrailingZeros(forInt(Integer.MIN_VALUE)), 31);
44         assertIntegerStampEquals(stampForTrailingZeros(forInt(Integer.MAX_VALUE)), 0);
45     }
46 
47     private static IntegerStamp forInt(int value) {
48         return StampFactory.forInteger(32, value, value);
49     }
50 
51     private static void assertIntegerStampEquals(Stamp stamp, int value) {
52         assertTrue(stamp instanceof IntegerStamp);
53         IntegerStamp iStamp = (IntegerStamp) stamp;
54         assertTrue(iStamp.lowerBound() == value);
55         assertTrue(iStamp.upperBound() == value);
56     }
57 
58 }
