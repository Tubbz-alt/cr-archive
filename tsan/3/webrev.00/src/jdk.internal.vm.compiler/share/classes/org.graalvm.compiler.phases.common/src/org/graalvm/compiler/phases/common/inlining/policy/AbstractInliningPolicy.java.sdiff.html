 41 public abstract class AbstractInliningPolicy implements InliningPolicy {
 42     public static final float RelevanceCapForInlining = 1.0f;
 43     public static final float CapInheritedRelevance = 1.0f;
 44     protected final Map<Invoke, Double> hints;
 45 
 46     public AbstractInliningPolicy(Map<Invoke, Double> hints) {
 47         this.hints = hints;
 48     }
 49 
 50     protected double computeMaximumSize(double relevance, int configuredMaximum) {
 51         double inlineRatio = Math.min(RelevanceCapForInlining, relevance);
 52         return configuredMaximum * inlineRatio;
 53     }
 54 
 55     protected double getInliningBonus(InlineInfo info) {
 56         if (hints != null && hints.containsKey(info.invoke())) {
 57             return hints.get(info.invoke());
 58         }
 59         return 1;
 60     }
 61 
 62     protected boolean isIntrinsic(Replacements replacements, InlineInfo info) {
 63         if (AlwaysInlineIntrinsics.getValue(info.graph().getOptions())) {
 64             return onlyIntrinsics(replacements, info);
 65         } else {
 66             return onlyForcedIntrinsics(replacements, info);
 67         }
 68     }
 69 
 70     private static boolean onlyIntrinsics(Replacements replacements, InlineInfo info) {
 71         for (int i = 0; i < info.numberOfMethods(); i++) {
 72             if (!InliningUtil.canIntrinsify(replacements, info.methodAt(i), info.invoke().bci())) {
 73                 return false;
 74             }
 75         }
 76         return true;
 77     }
