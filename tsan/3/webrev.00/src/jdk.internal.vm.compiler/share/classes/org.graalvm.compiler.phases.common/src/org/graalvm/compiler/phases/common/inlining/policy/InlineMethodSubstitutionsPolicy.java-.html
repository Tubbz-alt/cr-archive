25 package org.graalvm.compiler.phases.common.inlining.policy;
26 
27 import org.graalvm.compiler.core.common.GraalOptions;
28 import org.graalvm.compiler.nodes.CallTargetNode;
29 import org.graalvm.compiler.nodes.java.MethodCallTargetNode;
30 import org.graalvm.compiler.nodes.spi.Replacements;
31 import org.graalvm.compiler.phases.common.inlining.walker.MethodInvocation;
32 
33 import jdk.vm.ci.meta.ResolvedJavaMethod;
34 
35 /**
36  * Inline every method which would be replaced by a substitution. Useful for testing purposes.
37  */
38 public final class InlineMethodSubstitutionsPolicy extends InlineEverythingPolicy {
39 
40     @Override
41     public Decision isWorthInlining(Replacements replacements, MethodInvocation invocation, int inliningDepth, boolean fullyProcessed) {
42         final boolean isTracing = GraalOptions.TraceInlining.getValue(replacements.getOptions());
43         CallTargetNode callTarget = invocation.callee().invoke().callTarget();
44         if (callTarget instanceof MethodCallTargetNode) {
45             ResolvedJavaMethod calleeMethod = ((MethodCallTargetNode) callTarget).targetMethod();
46             if (replacements.hasSubstitution(calleeMethod, invocation.callee().invoke().bci())) {
47                 return Decision.YES.withReason(isTracing, "has a method subtitution");
48             }
49         }
50         return Decision.NO.withReason(isTracing, "does not have a method substitution");
51     }
52 }
