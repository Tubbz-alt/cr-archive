25 package org.graalvm.compiler.phases.common;
26 
27 import org.graalvm.compiler.debug.DebugContext;
28 import org.graalvm.compiler.graph.Node;
29 import org.graalvm.compiler.nodes.StructuredGraph;
30 import org.graalvm.compiler.nodes.spi.CoreProviders;
31 import org.graalvm.compiler.options.Option;
32 import org.graalvm.compiler.options.OptionKey;
33 import org.graalvm.compiler.options.OptionType;
34 import org.graalvm.compiler.phases.BasePhase;
35 
36 public class NodeCounterPhase extends BasePhase<CoreProviders> {
37 
38     private Stage stage;
39 
40     public NodeCounterPhase(Stage stage) {
41         this.stage = stage;
42     }
43 
44     public enum Stage {
45         INIT,
46         EARLY,
47         LATE
48     }
49 
50     public static class Options {
51         // @formatter:off
52         @Option(help = "Counts the number of instances of each node class.", type = OptionType.Debug)
53         public static final OptionKey<Boolean> NodeCounters = new OptionKey<>(false);
54         // @formatter:on
55     }
56 
57     @Override
58     protected void run(StructuredGraph graph, CoreProviders context) {
59 
60         for (Node node : graph.getNodes()) {
61             String nodeName = node.getNodeClass().getClazz().getSimpleName();
62 
63             DebugContext.counter("NodeCounter_%s_%s", this.stage, nodeName).increment(node.getDebug());
64         }
65     }
66 }
