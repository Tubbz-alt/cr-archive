25 package org.graalvm.compiler.phases.common.util;
26 
27 import org.graalvm.compiler.debug.TTY;
28 import org.graalvm.compiler.graph.Graph.NodeEvent;
29 import org.graalvm.compiler.graph.Graph.NodeEventListener;
30 import org.graalvm.compiler.graph.Node;
31 
32 /**
33  * A simple {@link NodeEventListener} implementation that traces events to TTY for debugging
34  * purposes.
35  */
36 public class TracingNodeEventListener extends NodeEventListener {
37 
38     @Override
39     public void changed(NodeEvent e, Node node) {
40         TTY.println(e.toString() + ": " + node);
41     }
42 }
