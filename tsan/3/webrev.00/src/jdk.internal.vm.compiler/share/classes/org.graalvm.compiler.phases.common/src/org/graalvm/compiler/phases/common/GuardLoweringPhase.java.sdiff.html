 49 /**
 50  * This phase lowers {@link GuardNode GuardNodes} into corresponding control-flow structure and
 51  * {@link DeoptimizeNode DeoptimizeNodes}.
 52  *
 53  * This allow to enter the {@link GuardsStage#FIXED_DEOPTS FIXED_DEOPTS} stage of the graph where
 54  * all node that may cause deoptimization are fixed.
 55  * <p>
 56  * It first makes a schedule in order to know where the control flow should be placed. Then, for
 57  * each block, it applies two passes. The first one tries to replace null-check guards with implicit
 58  * null checks performed by access to the objects that need to be null checked. The second phase
 59  * does the actual control-flow expansion of the remaining {@link GuardNode GuardNodes}.
 60  */
 61 public class GuardLoweringPhase extends BasePhase<MidTierContext> {
 62 
 63     private static class LowerGuards extends ScheduledNodeIterator {
 64 
 65         private final Block block;
 66         private boolean useGuardIdAsDebugId;
 67 
 68         LowerGuards(Block block, boolean useGuardIdAsDebugId) {
 69             this.block = block;
 70             this.useGuardIdAsDebugId = useGuardIdAsDebugId;
 71         }
 72 
 73         @Override
 74         protected void processNode(Node node) {
 75             if (node instanceof GuardNode) {
 76                 GuardNode guard = (GuardNode) node;
 77                 FixedWithNextNode lowered = guard.lowerGuard();
 78                 if (lowered != null) {
 79                     replaceCurrent(lowered);
 80                 } else {
 81                     lowerToIf(guard);
 82                 }
 83             }
 84         }
 85 
 86         @SuppressWarnings("try")
 87         private void lowerToIf(GuardNode guard) {
 88             try (DebugCloseable position = guard.withNodeSourcePosition()) {
 89                 StructuredGraph graph = guard.graph();
 90                 AbstractBeginNode fastPath = graph.add(new BeginNode());
 91                 fastPath.setNodeSourcePosition(guard.getNoDeoptSuccessorPosition());
 92                 @SuppressWarnings("deprecation")
 93                 int debugId = useGuardIdAsDebugId ? guard.getId() : DeoptimizeNode.DEFAULT_DEBUG_ID;
 94                 DeoptimizeNode deopt = graph.add(new DeoptimizeNode(guard.getAction(), guard.getReason(), debugId, guard.getSpeculation(), null));
 95                 AbstractBeginNode deoptBranch = BeginNode.begin(deopt);
 96                 AbstractBeginNode trueSuccessor;
 97                 AbstractBeginNode falseSuccessor;
 98                 insertLoopExits(deopt);
 99                 if (guard.isNegated()) {
100                     trueSuccessor = deoptBranch;
101                     falseSuccessor = fastPath;
102                 } else {
103                     trueSuccessor = fastPath;
104                     falseSuccessor = deoptBranch;
105                 }
106                 IfNode ifNode = graph.add(new IfNode(guard.getCondition(), trueSuccessor, falseSuccessor, trueSuccessor == fastPath ? 1 : 0));
107                 guard.replaceAndDelete(fastPath);
108                 insert(ifNode, fastPath);
109             }
110         }
111 
112         private void insertLoopExits(DeoptimizeNode deopt) {
113             Loop<Block> loop = block.getLoop();
114             StructuredGraph graph = deopt.graph();
115             while (loop != null) {
116                 LoopExitNode exit = graph.add(new LoopExitNode((LoopBeginNode) loop.getHeader().getBeginNode()));
117                 graph.addBeforeFixed(deopt, exit);
118                 loop = loop.getParent();
119             }
120         }
121     }
122 
123     @Override
124     protected void run(StructuredGraph graph, MidTierContext context) {
125         if (graph.getGuardsStage().allowsFloatingGuards()) {
126             SchedulePhase schedulePhase = new SchedulePhase(SchedulingStrategy.EARLIEST_WITH_GUARD_ORDER);
127             schedulePhase.apply(graph);
128             ScheduleResult schedule = graph.getLastSchedule();
129 
130             for (Block block : schedule.getCFG().getBlocks()) {
131                 processBlock(block, schedule);
132             }
133             graph.setGuardsStage(GuardsStage.FIXED_DEOPTS);
134         }
135 
136         assert assertNoGuardsLeft(graph);
137     }
138 
139     private static boolean assertNoGuardsLeft(StructuredGraph graph) {
140         assert graph.getNodes().filter(GuardNode.class).isEmpty();
141         return true;
142     }
143 
144     private static void processBlock(Block block, ScheduleResult schedule) {
145         DebugContext debug = block.getBeginNode().getDebug();
146         new LowerGuards(block, debug.isDumpEnabledForMethod() || debug.isLogEnabledForMethod()).processNodes(block, schedule);
147     }
148 }
