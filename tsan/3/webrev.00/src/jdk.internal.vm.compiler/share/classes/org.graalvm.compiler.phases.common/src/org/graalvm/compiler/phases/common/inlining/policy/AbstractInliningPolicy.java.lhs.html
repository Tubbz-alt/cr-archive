 25 package org.graalvm.compiler.phases.common.inlining.policy;
 26 
 27 import static org.graalvm.compiler.phases.common.inlining.InliningPhase.Options.AlwaysInlineIntrinsics;
 28 
 29 import java.util.Map;
 30 
 31 import org.graalvm.compiler.nodes.Invoke;
 32 import org.graalvm.compiler.nodes.StructuredGraph;
 33 import org.graalvm.compiler.nodes.spi.Replacements;
 34 import org.graalvm.compiler.phases.common.inlining.InliningUtil;
 35 import org.graalvm.compiler.phases.common.inlining.info.InlineInfo;
 36 import org.graalvm.compiler.phases.common.inlining.info.elem.Inlineable;
 37 
 38 import jdk.vm.ci.meta.ProfilingInfo;
 39 import jdk.vm.ci.meta.ResolvedJavaMethod;
 40 
 41 public abstract class AbstractInliningPolicy implements InliningPolicy {
 42     public static final float RelevanceCapForInlining = 1.0f;
 43     public static final float CapInheritedRelevance = 1.0f;
 44     protected final Map<Invoke, Double> hints;
 45 
 46     public AbstractInliningPolicy(Map<Invoke, Double> hints) {
 47         this.hints = hints;
 48     }
 49 
 50     protected double computeMaximumSize(double relevance, int configuredMaximum) {
 51         double inlineRatio = Math.min(RelevanceCapForInlining, relevance);
 52         return configuredMaximum * inlineRatio;
 53     }
 54 
 55     protected double getInliningBonus(InlineInfo info) {
 56         if (hints != null && hints.containsKey(info.invoke())) {
 57             return hints.get(info.invoke());
 58         }
 59         return 1;
 60     }
 61 
 62     protected boolean isIntrinsic(Replacements replacements, InlineInfo info) {
 63         if (AlwaysInlineIntrinsics.getValue(info.graph().getOptions())) {
 64             return onlyIntrinsics(replacements, info);
 65         } else {
 66             return onlyForcedIntrinsics(replacements, info);
 67         }
 68     }
 69 
 70     private static boolean onlyIntrinsics(Replacements replacements, InlineInfo info) {
 71         for (int i = 0; i < info.numberOfMethods(); i++) {
 72             if (!InliningUtil.canIntrinsify(replacements, info.methodAt(i), info.invoke().bci())) {
 73                 return false;
 74             }
 75         }
 76         return true;
 77     }
 78 
 79     private static boolean onlyForcedIntrinsics(Replacements replacements, InlineInfo info) {
 80         if (!onlyIntrinsics(replacements, info)) {
 81             return false;
 82         }
 83         if (!info.shouldInline()) {
 84             return false;
 85         }
 86         return true;
 87     }
 88 
 89     protected int previousLowLevelGraphSize(InlineInfo info) {
 90         int size = 0;
 91         for (int i = 0; i < info.numberOfMethods(); i++) {
 92             ResolvedJavaMethod m = info.methodAt(i);
 93             ProfilingInfo profile = info.graph().getProfilingInfo(m);
 94             int compiledGraphSize = profile.getCompilerIRSize(StructuredGraph.class);
 95             if (compiledGraphSize > 0) {
 96                 size += compiledGraphSize;
 97             }
 98         }
 99         return size;
100     }
101 
102     protected double determineInvokeProbability(InlineInfo info) {
103         double invokeProbability = 0;
104         for (int i = 0; i < info.numberOfMethods(); i++) {
105             Inlineable callee = info.inlineableElementAt(i);
106             Iterable<Invoke> invokes = callee.getInvokes();
107             if (invokes.iterator().hasNext()) {
108                 for (Invoke invoke : invokes) {
109                     invokeProbability += callee.getProbability(invoke);
110                 }
111             }
112         }
113         return invokeProbability;
114     }
115 }
