25 package org.graalvm.compiler.phases.common.inlining.policy;
26 
27 import static org.graalvm.compiler.core.common.GraalOptions.MaximumDesiredSize;
28 
29 import org.graalvm.compiler.core.common.GraalOptions;
30 import org.graalvm.compiler.core.common.PermanentBailoutException;
31 import org.graalvm.compiler.nodes.StructuredGraph;
32 import org.graalvm.compiler.nodes.spi.Replacements;
33 import org.graalvm.compiler.phases.common.inlining.InliningUtil;
34 import org.graalvm.compiler.phases.common.inlining.walker.MethodInvocation;
35 
36 public class InlineEverythingPolicy implements InliningPolicy {
37 
38     @Override
39     public boolean continueInlining(StructuredGraph graph) {
40         if (InliningUtil.getNodeCount(graph) >= MaximumDesiredSize.getValue(graph.getOptions())) {
41             throw new PermanentBailoutException("Inline all calls failed. The resulting graph is too large.");
42         }
43         return true;
44     }
45 
46     @Override
47     public Decision isWorthInlining(Replacements replacements, MethodInvocation invocation, int inliningDepth, boolean fullyProcessed) {
48         return Decision.YES.withReason(GraalOptions.TraceInlining.getValue(replacements.getOptions()), "inline everything");
49     }
50 }
