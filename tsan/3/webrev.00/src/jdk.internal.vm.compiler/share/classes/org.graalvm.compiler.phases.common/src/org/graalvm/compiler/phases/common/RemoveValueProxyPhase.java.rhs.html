25 package org.graalvm.compiler.phases.common;
26 
27 import org.graalvm.compiler.nodes.FrameState;
28 import org.graalvm.compiler.nodes.LoopExitNode;
29 import org.graalvm.compiler.nodes.StructuredGraph;
30 import org.graalvm.compiler.nodes.util.GraphUtil;
31 import org.graalvm.compiler.phases.Phase;
32 
33 public class RemoveValueProxyPhase extends Phase {
34 
35     @Override
36     protected void run(StructuredGraph graph) {
37         for (LoopExitNode exit : graph.getNodes(LoopExitNode.TYPE)) {
38             exit.removeProxies();
39             FrameState frameState = exit.stateAfter();
40             if (frameState != null && frameState.isExceptionHandlingBCI()) {
41                 // The parser will create loop exits with such BCIs on the exception handling path.
42                 // Loop optimizations must avoid duplicating such exits
43                 // We clean them up here otherwise they could survive until code generation
44                 exit.setStateAfter(null);
45                 GraphUtil.tryKillUnused(frameState);
46             }
47         }
48         graph.setHasValueProxies(false);
49     }
50 }
