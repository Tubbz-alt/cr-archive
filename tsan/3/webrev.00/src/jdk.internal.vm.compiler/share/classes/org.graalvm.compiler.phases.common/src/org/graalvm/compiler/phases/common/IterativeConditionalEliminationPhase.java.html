 25 package org.graalvm.compiler.phases.common;
 26 
 27 import org.graalvm.compiler.core.common.GraalOptions;
 28 import org.graalvm.compiler.core.common.PermanentBailoutException;
 29 import org.graalvm.compiler.debug.TTY;
 30 import org.graalvm.compiler.graph.Graph.NodeEventScope;
 31 import org.graalvm.compiler.graph.Node;
 32 import org.graalvm.compiler.nodes.StructuredGraph;
 33 import org.graalvm.compiler.nodes.spi.CoreProviders;
 34 import org.graalvm.compiler.phases.BasePhase;
 35 import org.graalvm.compiler.phases.common.util.EconomicSetNodeEventListener;
 36 import org.graalvm.compiler.phases.common.util.TracingNodeEventListener;
 37 
 38 public class IterativeConditionalEliminationPhase extends BasePhase<CoreProviders> {
 39 
 40     private static final boolean DEBUG_PHASE = false;
 41     private static final int DEBUG_MAX_ITERATIONS = 256;
 42 
 43     private final CanonicalizerPhase canonicalizer;
 44     private final boolean fullSchedule;
 45 
 46     public IterativeConditionalEliminationPhase(CanonicalizerPhase canonicalizer, boolean fullSchedule) {
 47         this.canonicalizer = canonicalizer;
 48         this.fullSchedule = fullSchedule;
 49     }
 50 
 51     @Override
 52     @SuppressWarnings("try")
 53     protected void run(StructuredGraph graph, CoreProviders context) {
 54         final int maxIterations = GraalOptions.ConditionalEliminationMaxIterations.getValue(graph.getOptions());
 55         EconomicSetNodeEventListener listener = new EconomicSetNodeEventListener();
 56         int count = 0;
 57 
 58         while (true) {
 59             count++;
 60             try (NodeEventScope nes = graph.trackNodeEvents(listener)) {
 61                 new ConditionalEliminationPhase(fullSchedule).apply(graph, context);
 62             }
 63             if (listener.getNodes().isEmpty()) {
 64                 break;
 65             }
 66 
 67             canonicalizer.applyIncremental(graph, context, listener.getNodes());
 68             listener.getNodes().clear();
 69 
 70             if (count >= maxIterations) {
 71                 if (DEBUG_PHASE) {
 72                     if (count >= DEBUG_MAX_ITERATIONS - 5) {
 73                         TTY.println();
 74                         TTY.println("------------------------------------");
 75                         TTY.println("Iteration " + count);
 76                         TTY.println("Conditional elimination changed nodes: ");
 77                         for (Node n : listener.getNodes()) {
 78                             TTY.println(n.toString());
 79                             for (Node input : n.inputs()) {
 80                                 TTY.println("    input: " + input);
 81                             }
 82                         }
 83                         TTY.println("Canonicalization with node listener: ");
 84                         try (NodeEventScope debugNes = graph.trackNodeEvents(new TracingNodeEventListener())) {
 85                             canonicalizer.applyIncremental(graph, context, listener.getNodes());
 86                         }
 87                     }
 88                     if (count >= DEBUG_MAX_ITERATIONS) {
 89                         throw new PermanentBailoutException("Number of iterations in ConditionalEliminationPhase phase exceeds %d", DEBUG_MAX_ITERATIONS);
 90                     }
 91                 }
 92                 break;
 93             }
 94         }
 95     }
 96 
 97     @Override
 98     public float codeSizeIncrease() {
 99         return 2.0f;
100     }
101 }
