 46 /**
 47  * This phase lowers {@link GuardNode GuardNodes} into corresponding control-flow structure and
 48  * {@link DeoptimizeNode DeoptimizeNodes}.
 49  *
 50  * This allow to enter the {@link GuardsStage#FIXED_DEOPTS FIXED_DEOPTS} stage of the graph where
 51  * all node that may cause deoptimization are fixed.
 52  * <p>
 53  * It first makes a schedule in order to know where the control flow should be placed. Then, for
 54  * each block, it applies two passes. The first one tries to replace null-check guards with implicit
 55  * null checks performed by access to the objects that need to be null checked. The second phase
 56  * does the actual control-flow expansion of the remaining {@link GuardNode GuardNodes}.
 57  */
 58 public class GuardLoweringPhase extends BasePhase<MidTierContext> {
 59 
 60     private static class LowerGuards extends ScheduledNodeIterator {
 61 
 62         private boolean useGuardIdAsDebugId;
 63 
 64         LowerGuards(boolean useGuardIdAsDebugId) {
 65             this.useGuardIdAsDebugId = useGuardIdAsDebugId;
 66         }
 67 
 68         @Override
 69         protected void processNode(Node node) {
 70             if (node instanceof GuardNode) {
 71                 GuardNode guard = (GuardNode) node;
 72                 FixedWithNextNode lowered = guard.lowerGuard();
 73                 if (lowered != null) {
 74                     replaceCurrent(lowered);
 75                 } else {
 76                     lowerToIf(guard);
 77                 }
 78             }
 79         }
 80 
 81         @SuppressWarnings("try")
 82         private void lowerToIf(GuardNode guard) {
 83             try (DebugCloseable position = guard.withNodeSourcePosition()) {
 84                 StructuredGraph graph = guard.graph();
 85                 AbstractBeginNode fastPath = graph.add(new BeginNode());
 86                 fastPath.setNodeSourcePosition(guard.getNoDeoptSuccessorPosition());
 87                 @SuppressWarnings("deprecation")
 88                 int debugId = useGuardIdAsDebugId ? guard.getId() : DeoptimizeNode.DEFAULT_DEBUG_ID;
 89                 DeoptimizeNode deopt = graph.add(new DeoptimizeNode(guard.getAction(), guard.getReason(), debugId, guard.getSpeculation(), null));
 90                 AbstractBeginNode deoptBranch = BeginNode.begin(deopt);
 91                 AbstractBeginNode trueSuccessor;
 92                 AbstractBeginNode falseSuccessor;
 93                 if (guard.isNegated()) {
 94                     trueSuccessor = deoptBranch;
 95                     falseSuccessor = fastPath;
 96                 } else {
 97                     trueSuccessor = fastPath;
 98                     falseSuccessor = deoptBranch;
 99                 }
100                 IfNode ifNode = graph.add(new IfNode(guard.getCondition(), trueSuccessor, falseSuccessor, trueSuccessor == fastPath ? 1 : 0));
101                 guard.replaceAndDelete(fastPath);
102                 insert(ifNode, fastPath);
103             }
104         }
105     }
106 
107     @Override
108     protected void run(StructuredGraph graph, MidTierContext context) {
109         if (graph.getGuardsStage().allowsFloatingGuards()) {
110             SchedulePhase schedulePhase = new SchedulePhase(SchedulingStrategy.EARLIEST_WITH_GUARD_ORDER);
111             schedulePhase.apply(graph);
112             ScheduleResult schedule = graph.getLastSchedule();
113 
114             for (Block block : schedule.getCFG().getBlocks()) {
115                 processBlock(block, schedule);
116             }
117             graph.setGuardsStage(GuardsStage.FIXED_DEOPTS);
118         }
119 
120         assert assertNoGuardsLeft(graph);
121     }
122 
123     private static boolean assertNoGuardsLeft(StructuredGraph graph) {
124         assert graph.getNodes().filter(GuardNode.class).isEmpty();
125         return true;
126     }
127 
128     private static void processBlock(Block block, ScheduleResult schedule) {
129         DebugContext debug = block.getBeginNode().getDebug();
130         new LowerGuards(debug.isDumpEnabledForMethod() || debug.isLogEnabledForMethod()).processNodes(block, schedule);
131     }
132 }
