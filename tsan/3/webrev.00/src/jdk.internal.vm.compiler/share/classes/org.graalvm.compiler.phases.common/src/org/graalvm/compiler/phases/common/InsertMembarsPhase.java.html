27 package org.graalvm.compiler.phases.common;
28 
29 import org.graalvm.compiler.nodes.StructuredGraph;
30 import org.graalvm.compiler.nodes.extended.MembarNode;
31 import org.graalvm.compiler.nodes.memory.FixedAccessNode;
32 import org.graalvm.compiler.nodes.memory.ReadNode;
33 import org.graalvm.compiler.nodes.memory.VolatileReadNode;
34 import org.graalvm.compiler.nodes.memory.WriteNode;
35 import org.graalvm.compiler.phases.Phase;
36 
37 import static jdk.vm.ci.code.MemoryBarriers.JMM_POST_VOLATILE_READ;
38 import static jdk.vm.ci.code.MemoryBarriers.JMM_POST_VOLATILE_WRITE;
39 import static jdk.vm.ci.code.MemoryBarriers.JMM_PRE_VOLATILE_READ;
40 import static jdk.vm.ci.code.MemoryBarriers.JMM_PRE_VOLATILE_WRITE;
41 
42 public class InsertMembarsPhase extends Phase {
43     @Override
44     protected void run(StructuredGraph graph) {
45         for (FixedAccessNode access : graph.getNodes(FixedAccessNode.TYPE)) {
46             if (access instanceof VolatileReadNode) {
47                 ReadNode read = (ReadNode) access;
48                 MembarNode preMembar = graph.add(new MembarNode(JMM_PRE_VOLATILE_READ));
49                 graph.addBeforeFixed(read, preMembar);
50                 MembarNode postMembar = graph.add(new MembarNode(JMM_POST_VOLATILE_READ));
51                 graph.addAfterFixed(read, postMembar);
52             } else if (access instanceof WriteNode && ((WriteNode) access).isVolatile()) {
53                 WriteNode write = (WriteNode) access;
54                 MembarNode preMembar = graph.add(new MembarNode(JMM_PRE_VOLATILE_WRITE));
55                 graph.addBeforeFixed(write, preMembar);
56                 MembarNode postMembar = graph.add(new MembarNode(JMM_POST_VOLATILE_WRITE));
57                 graph.addAfterFixed(write, postMembar);
58             }
59         }
60     }
61 
62     @Override
63     public float codeSizeIncrease() {
64         return 3f;
65     }
66 }
