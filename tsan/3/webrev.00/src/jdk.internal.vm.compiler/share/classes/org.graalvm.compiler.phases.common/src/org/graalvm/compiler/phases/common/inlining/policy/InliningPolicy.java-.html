25 package org.graalvm.compiler.phases.common.inlining.policy;
26 
27 import org.graalvm.compiler.nodes.StructuredGraph;
28 import org.graalvm.compiler.nodes.spi.Replacements;
29 import org.graalvm.compiler.phases.common.inlining.walker.MethodInvocation;
30 
31 public interface InliningPolicy {
32     class Decision {
33         public static final Decision YES = new Decision(true, "(unknown reason)");
34         public static final Decision NO = new Decision(false, "(unknown reason)");
35 
36         private final boolean shouldInline;
37         private final String reason;
38 
39         private Decision(boolean shouldInline, String reason) {
40             this.shouldInline = shouldInline;
41             this.reason = reason;
42         }
43 
44         public boolean shouldInline() {
45             return shouldInline;
46         }
47 
48         public String getReason() {
49             return reason;
50         }
51 
52         public Decision withReason(boolean isTracing, String newReason, Object... args) {
53             if (isTracing) {
54                 return new Decision(shouldInline, String.format(newReason, args));
55             } else {
56                 return this;
57             }
58         }
59     }
60 
61     boolean continueInlining(StructuredGraph graph);
62 
63     Decision isWorthInlining(Replacements replacements, MethodInvocation invocation, int inliningDepth, boolean fullyProcessed);
64 }
