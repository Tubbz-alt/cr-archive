25 package org.graalvm.compiler.phases.common.inlining.policy;
26 
27 import org.graalvm.compiler.core.common.GraalOptions;
28 import org.graalvm.compiler.nodes.CallTargetNode;
29 import org.graalvm.compiler.nodes.java.MethodCallTargetNode;
30 import org.graalvm.compiler.nodes.spi.Replacements;
31 import org.graalvm.compiler.phases.common.inlining.info.InlineInfo;
32 import org.graalvm.compiler.phases.common.inlining.walker.MethodInvocation;
33 
34 import jdk.vm.ci.meta.ResolvedJavaMethod;
35 
36 /**
37  * Inline every method which would be replaced by a substitution. Useful for testing purposes.
38  */
39 public final class InlineMethodSubstitutionsPolicy extends InlineEverythingPolicy {
40 
41     @Override
42     public Decision isWorthInlining(Replacements replacements, MethodInvocation invocation, InlineInfo calleeInfo, int inliningDepth, boolean fullyProcessed) {
43         final boolean isTracing = GraalOptions.TraceInlining.getValue(calleeInfo.graph().getOptions());
44         CallTargetNode callTarget = invocation.callee().invoke().callTarget();
45         if (callTarget instanceof MethodCallTargetNode) {
46             ResolvedJavaMethod calleeMethod = callTarget.targetMethod();
47             if (replacements.hasSubstitution(calleeMethod, invocation.callee().invoke().bci())) {
48                 return Decision.YES.withReason(isTracing, "has a method subtitution");
49             }
50         }
51         return Decision.NO.withReason(isTracing, "does not have a method substitution");
52     }
53 }
