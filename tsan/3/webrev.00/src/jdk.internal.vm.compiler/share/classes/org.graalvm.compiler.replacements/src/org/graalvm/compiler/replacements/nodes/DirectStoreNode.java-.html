44 /**
45  * A special purpose store node that differs from {@link RawStoreNode} in that it is not a
46  * {@link StateSplit} and takes a computed address instead of an object.
47  */
48 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
49 public final class DirectStoreNode extends FixedWithNextNode implements LIRLowerable {
50 
51     public static final NodeClass<DirectStoreNode> TYPE = NodeClass.create(DirectStoreNode.class);
52     @Input protected ValueNode address;
53     @Input protected ValueNode value;
54     protected final JavaKind kind;
55 
56     public DirectStoreNode(ValueNode address, ValueNode value, JavaKind kind) {
57         super(TYPE, StampFactory.forVoid());
58         this.address = address;
59         this.value = value;
60         this.kind = kind;
61     }
62 
63     @Override
64     public void generate(NodeLIRBuilderTool gen) {
65         Value v = gen.operand(value);
66         LIRKind lirKind = LIRKind.fromJavaKind(gen.getLIRGeneratorTool().target().arch, kind);
67         gen.getLIRGeneratorTool().getArithmetic().emitStore(lirKind, gen.operand(address), v, null);
68     }
69 
70     public ValueNode getAddress() {
71         return address;
72     }
73 
74     public ValueNode getValue() {
75         return value;
76     }
77 
78     @NodeIntrinsic
79     public static native void storeBoolean(long address, boolean value, @ConstantNodeParameter JavaKind kind);
80 }
