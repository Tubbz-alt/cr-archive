28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_8;
29 
30 import org.graalvm.compiler.core.common.type.StampFactory;
31 import org.graalvm.compiler.graph.NodeClass;

33 import org.graalvm.compiler.nodeinfo.NodeInfo;

36 import org.graalvm.compiler.nodes.memory.address.AddressNode;
37 import org.graalvm.compiler.nodes.memory.address.OffsetAddressNode;
38 import org.graalvm.compiler.nodes.spi.LIRLowerable;
39 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
40 import org.graalvm.compiler.word.Word;
41 import jdk.internal.vm.compiler.word.LocationIdentity;
42 
43 /**
44  * Zeros a chunk of memory.
45  */
46 @NodeInfo(nameTemplate = "ZeroMemory#{p#location/s}", allowedUsageTypes = {InputType.Memory}, cycles = CYCLES_8, size = SIZE_8)
47 public class ZeroMemoryNode extends FixedAccessNode implements LIRLowerable {
48     public static final NodeClass<ZeroMemoryNode> TYPE = NodeClass.create(ZeroMemoryNode.class);
49 
50     @Input ValueNode length;
51     private final boolean isAligned;
52 
53     public ZeroMemoryNode(ValueNode address, ValueNode length, boolean isAligned, LocationIdentity locationIdentity) {
54         this(OffsetAddressNode.create(address), length, isAligned, locationIdentity, BarrierType.NONE);
55     }
56 
57     public ZeroMemoryNode(AddressNode address, ValueNode length, boolean isAligned, LocationIdentity locationIdentity, BarrierType type) {
58         super(TYPE, address, locationIdentity, StampFactory.forVoid(), type);
59         this.length = length;
60         this.isAligned = isAligned;
61     }
62 
63     @Override
64     public void generate(NodeLIRBuilderTool gen) {
65         gen.getLIRGeneratorTool().emitZeroMemory(gen.operand(getAddress()), gen.operand(length), isAligned);
66     }
67 
68     @Override
69     public boolean canNullCheck() {
70         return false;
71     }
72 
73     @NodeIntrinsic
74     public static native void zero(Word address, long length, @ConstantNodeParameter boolean isAligned, @ConstantNodeParameter LocationIdentity locationIdentity);
75 }
