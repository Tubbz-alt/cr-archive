 51 /**
 52  * Node representing an exact integer addition that will throw an {@link ArithmeticException} in
 53  * case the addition would overflow the 32 bit range.
 54  */
 55 @NodeInfo(cycles = CYCLES_2, size = SIZE_2)
 56 public final class IntegerAddExactNode extends AddNode implements GuardedNode, IntegerExactArithmeticNode {
 57     public static final NodeClass<IntegerAddExactNode> TYPE = NodeClass.create(IntegerAddExactNode.class);
 58 
 59     @Input(InputType.Guard) protected GuardingNode guard;
 60 
 61     public IntegerAddExactNode(ValueNode x, ValueNode y, GuardingNode guard) {
 62         super(TYPE, x, y);
 63         setStamp(x.stamp(NodeView.DEFAULT).unrestricted());
 64         assert x.stamp(NodeView.DEFAULT).isCompatible(y.stamp(NodeView.DEFAULT)) && x.stamp(NodeView.DEFAULT) instanceof IntegerStamp;
 65         this.guard = guard;
 66     }
 67 
 68     @Override
 69     public boolean inferStamp() {
 70         /*
 71          * Note: it is not allowed to use the foldStamp method of the regular add node as we do not
 72          * know the result stamp of this node if we do not know whether we may deopt. If we know we
 73          * can never overflow we will replace this node with its non overflow checking counterpart
 74          * anyway.
 75          */
 76         return false;
 77     }
 78 
 79     @Override
 80     public Stamp foldStamp(Stamp stampX, Stamp stampY) {
 81         IntegerStamp a = (IntegerStamp) stampX;
 82         IntegerStamp b = (IntegerStamp) stampY;
 83 
 84         int bits = a.getBits();
 85         assert bits == b.getBits();
 86 
 87         long defaultMask = CodeUtil.mask(bits);
 88         long variableBits = (a.downMask() ^ a.upMask()) | (b.downMask() ^ b.upMask());
 89         long variableBitsWithCarry = variableBits | (carryBits(a.downMask(), b.downMask()) ^ carryBits(a.upMask(), b.upMask()));
 90         long newDownMask = (a.downMask() + b.downMask()) & ~variableBitsWithCarry;
 91         long newUpMask = (a.downMask() + b.downMask()) | variableBitsWithCarry;
 92 
 93         newDownMask &= defaultMask;
 94         newUpMask &= defaultMask;
 95 
 96         long newLowerBound;
 97         long newUpperBound;
 98         boolean lowerOverflowsPositively = addOverflowsPositively(a.lowerBound(), b.lowerBound(), bits);
 99         boolean upperOverflowsPositively = addOverflowsPositively(a.upperBound(), b.upperBound(), bits);
100         boolean lowerOverflowsNegatively = addOverflowsNegatively(a.lowerBound(), b.lowerBound(), bits);
101         boolean upperOverflowsNegatively = addOverflowsNegatively(a.upperBound(), b.upperBound(), bits);
102         if (lowerOverflowsPositively) {
103             newLowerBound = CodeUtil.maxValue(bits);
104         } else if (lowerOverflowsNegatively) {
105             newLowerBound = CodeUtil.minValue(bits);
106         } else {
107             newLowerBound = CodeUtil.signExtend((a.lowerBound() + b.lowerBound()) & defaultMask, bits);
108         }
109 
110         if (upperOverflowsPositively) {
111             newUpperBound = CodeUtil.maxValue(bits);
112         } else if (upperOverflowsNegatively) {
113             newUpperBound = CodeUtil.minValue(bits);
114         } else {
115             newUpperBound = CodeUtil.signExtend((a.upperBound() + b.upperBound()) & defaultMask, bits);
116         }
117 
118         IntegerStamp limit = StampFactory.forInteger(bits, newLowerBound, newUpperBound);
119         newUpMask &= limit.upMask();
120         newUpperBound = CodeUtil.signExtend(newUpperBound & newUpMask, bits);
121         newDownMask |= limit.downMask();
122         newLowerBound |= newDownMask;
123         return IntegerStamp.create(bits, newLowerBound, newUpperBound, newDownMask, newUpMask);
124     }
125 
126     @Override
127     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
128         if (forX.isConstant() && !forY.isConstant()) {
129             return new IntegerAddExactNode(forY, forX, guard).canonical(tool);
130         }
131         if (forX.isConstant() && forY.isConstant()) {
132             return canonicalXYconstant(forX, forY);
133         } else if (forY.isConstant()) {
134             long c = forY.asJavaConstant().asLong();
135             if (c == 0) {
136                 return forX;
137             }
138         }
139         if (!IntegerStamp.addCanOverflow((IntegerStamp) forX.stamp(NodeView.DEFAULT), (IntegerStamp) forY.stamp(NodeView.DEFAULT))) {
140             return new AddNode(forX, forY).canonical(tool);
141         }



142         return this;
143     }
144 
145     private ValueNode canonicalXYconstant(ValueNode forX, ValueNode forY) {
146         JavaConstant xConst = forX.asJavaConstant();
147         JavaConstant yConst = forY.asJavaConstant();
148         assert xConst.getJavaKind() == yConst.getJavaKind();
149         try {
150             if (xConst.getJavaKind() == JavaKind.Int) {
151                 return ConstantNode.forInt(Math.addExact(xConst.asInt(), yConst.asInt()));
152             } else {
153                 assert xConst.getJavaKind() == JavaKind.Long;
154                 return ConstantNode.forLong(Math.addExact(xConst.asLong(), yConst.asLong()));
155             }
156         } catch (ArithmeticException ex) {
157             // The operation will result in an overflow exception, so do not canonicalize.
158         }
159         return this;
160     }
161 
162     @Override
163     public GuardingNode getGuard() {
164         return guard;
165     }
166 
167     @Override
168     public void setGuard(GuardingNode guard) {
169         updateUsagesInterface(this.guard, guard);
170         this.guard = guard;
171     }
172 }
