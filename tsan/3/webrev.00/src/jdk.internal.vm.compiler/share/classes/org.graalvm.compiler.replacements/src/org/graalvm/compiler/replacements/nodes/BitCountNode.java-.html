28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_1;

46 import jdk.vm.ci.meta.JavaKind;

48 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
49 public final class BitCountNode extends UnaryNode implements ArithmeticLIRLowerable {
50 
51     public static final NodeClass<BitCountNode> TYPE = NodeClass.create(BitCountNode.class);
52 
53     public BitCountNode(ValueNode value) {
54         super(TYPE, computeStamp(value.stamp(NodeView.DEFAULT), value), value);
55         assert value.getStackKind() == JavaKind.Int || value.getStackKind() == JavaKind.Long;
56     }
57 
58     @Override
59     public Stamp foldStamp(Stamp newStamp) {
60         ValueNode theValue = getValue();
61         return computeStamp(newStamp, theValue);
62     }
63 
64     static Stamp computeStamp(Stamp newStamp, ValueNode theValue) {
65         assert newStamp.isCompatible(theValue.stamp(NodeView.DEFAULT));
66         IntegerStamp valueStamp = (IntegerStamp) newStamp;
67         assert (valueStamp.downMask() & CodeUtil.mask(valueStamp.getBits())) == valueStamp.downMask();
68         assert (valueStamp.upMask() & CodeUtil.mask(valueStamp.getBits())) == valueStamp.upMask();
69         return StampFactory.forInteger(JavaKind.Int, Long.bitCount(valueStamp.downMask()), Long.bitCount(valueStamp.upMask()));
70     }
71 
72     @Override
73     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
74         if (forValue.isConstant()) {
75             JavaConstant c = forValue.asJavaConstant();
76             return ConstantNode.forInt(forValue.getStackKind() == JavaKind.Int ? Integer.bitCount(c.asInt()) : Long.bitCount(c.asLong()));
77         }
78         return this;
79     }
80 
81     @Override
82     public void generate(NodeLIRBuilderTool builder, ArithmeticLIRGeneratorTool gen) {
83         builder.setResult(this, gen.emitBitCount(builder.operand(getValue())));
84     }
85 }
