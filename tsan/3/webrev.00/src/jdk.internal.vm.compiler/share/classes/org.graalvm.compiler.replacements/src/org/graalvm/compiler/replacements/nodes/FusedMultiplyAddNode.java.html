47 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
48 public final class FusedMultiplyAddNode extends TernaryNode implements ArithmeticLIRLowerable {
49 
50     public static final NodeClass<FusedMultiplyAddNode> TYPE = NodeClass.create(FusedMultiplyAddNode.class);
51 
52     public FusedMultiplyAddNode(ValueNode a, ValueNode b, ValueNode c) {
53         super(TYPE, computeStamp(a.stamp(NodeView.DEFAULT), b.stamp(NodeView.DEFAULT), c.stamp(NodeView.DEFAULT)), a, b, c);
54         assert a.getStackKind().isNumericFloat();
55         assert b.getStackKind().isNumericFloat();
56         assert c.getStackKind().isNumericFloat();
57     }
58 
59     @Override
60     public Stamp foldStamp(Stamp stampX, Stamp stampY, Stamp stampZ) {
61         return computeStamp(stampX, stampY, stampZ);
62     }
63 
64     private static Stamp computeStamp(Stamp stampX, Stamp stampY, Stamp stampZ) {
65         Stamp m = FloatStamp.OPS.getMul().foldStamp(stampX, stampY);
66         return FloatStamp.OPS.getAdd().foldStamp(m, stampZ);
67     }
68 
69     @Override
70     public ValueNode canonical(CanonicalizerTool tool, ValueNode a, ValueNode b, ValueNode c) {
71         if (a.isConstant() && b.isConstant() && c.isConstant()) {
72             JavaConstant ca = a.asJavaConstant();
73             JavaConstant cb = b.asJavaConstant();
74             JavaConstant cc = c.asJavaConstant();
75 
76             ValueNode res;
77             if (a.getStackKind() == JavaKind.Float) {
78                 res = ConstantNode.forFloat(GraalServices.fma(ca.asFloat(), cb.asFloat(), cc.asFloat()));
79             } else {
80                 assert a.getStackKind() == JavaKind.Double;
81                 res = ConstantNode.forDouble(GraalServices.fma(ca.asDouble(), cb.asDouble(), cc.asDouble()));
82             }
83             return res;
84         }
85         return this;
86     }
87 
88     @Override
89     public void generate(NodeLIRBuilderTool builder, ArithmeticLIRGeneratorTool gen) {
90         builder.setResult(this, gen.emitFusedMultiplyAdd(builder.operand(getX()), builder.operand(getY()), builder.operand(getZ())));
91     }
92 }
