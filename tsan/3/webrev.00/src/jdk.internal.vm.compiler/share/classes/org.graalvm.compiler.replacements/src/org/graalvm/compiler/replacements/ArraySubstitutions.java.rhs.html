25 package org.graalvm.compiler.replacements;
26 
27 import java.lang.reflect.Array;
28 
29 import org.graalvm.compiler.api.replacements.ClassSubstitution;
30 import org.graalvm.compiler.api.replacements.MethodSubstitution;
31 import org.graalvm.compiler.nodes.DeoptimizeNode;
32 import org.graalvm.compiler.nodes.extended.BranchProbabilityNode;
33 import org.graalvm.compiler.nodes.java.ArrayLengthNode;
34 
35 import jdk.vm.ci.meta.DeoptimizationAction;
36 import jdk.vm.ci.meta.DeoptimizationReason;
37 
38 // JaCoCo Exclude
39 
40 /**
41  * Substitutions for {@link java.lang.reflect.Array} methods.
42  */
43 @ClassSubstitution(Array.class)
44 public class ArraySubstitutions {
45 
46     @MethodSubstitution
47     public static int getLength(Object array) {
48         if (BranchProbabilityNode.probability(BranchProbabilityNode.DEOPT_PROBABILITY, !array.getClass().isArray())) {
49             DeoptimizeNode.deopt(DeoptimizationAction.None, DeoptimizationReason.RuntimeConstraint);
50         }
51         return ArrayLengthNode.arrayLength(array);
52     }
53 
54 }
