 49 /**
 50  * Compares two array regions with a given length.
 51  */
 52 @NodeInfo(cycles = NodeCycles.CYCLES_UNKNOWN, size = NodeSize.SIZE_128)
 53 public final class ArrayRegionEqualsNode extends FixedWithNextNode implements LIRLowerable, MemoryAccess {
 54 
 55     public static final NodeClass<ArrayRegionEqualsNode> TYPE = NodeClass.create(ArrayRegionEqualsNode.class);
 56 
 57     /** {@link JavaKind} of the arrays to compare. */
 58     private final JavaKind kind1;
 59     private final JavaKind kind2;
 60 
 61     /** Pointer to first array region to be tested for equality. */
 62     @Input private ValueNode array1;
 63 
 64     /** Pointer to second array region to be tested for equality. */
 65     @Input private ValueNode array2;
 66 
 67     /** Length of the array region. */
 68     @Input private ValueNode length;
 69 
 70     @OptionalInput(Memory) private MemoryNode lastLocationAccess;
 71 
 72     public ArrayRegionEqualsNode(ValueNode array1, ValueNode array2, ValueNode length, @ConstantNodeParameter JavaKind kind1, @ConstantNodeParameter JavaKind kind2) {
 73         super(TYPE, StampFactory.forKind(JavaKind.Boolean));
 74         this.kind1 = kind1;
 75         this.kind2 = kind2;
 76         this.array1 = array1;
 77         this.array2 = array2;
 78         this.length = length;
 79     }
 80 
 81     public static boolean regionEquals(Pointer array1, Pointer array2, int length, @ConstantNodeParameter JavaKind kind) {
 82         return regionEquals(array1, array2, length, kind, kind);
 83     }
 84 
 85     @NodeIntrinsic
 86     public static native boolean regionEquals(Pointer array1, Pointer array2, int length, @ConstantNodeParameter JavaKind kind1, @ConstantNodeParameter JavaKind kind2);
 87 
 88     @Override
 89     public void generate(NodeLIRBuilderTool gen) {
 90         int constantLength = -1;
 91         if (length.isConstant()) {
 92             constantLength = length.asJavaConstant().asInt();
 93         }
 94         Value result;
 95         if (kind1 == kind2) {
 96             result = gen.getLIRGeneratorTool().emitArrayEquals(kind1, gen.operand(array1), gen.operand(array2), gen.operand(length), constantLength, true);
 97         } else {
 98             result = gen.getLIRGeneratorTool().emitArrayEquals(kind1, kind2, gen.operand(array1), gen.operand(array2), gen.operand(length), constantLength, true);
 99         }
100         gen.setResult(this, result);
101     }
102 
103     @Override
104     public LocationIdentity getLocationIdentity() {
105         return kind1 != kind2 ? LocationIdentity.ANY_LOCATION : NamedLocationIdentity.getArrayLocation(kind1);
106     }
107 
108     @Override
109     public MemoryNode getLastLocationAccess() {
110         return lastLocationAccess;
111     }
112 
113     @Override
114     public void setLastLocationAccess(MemoryNode lla) {
115         updateUsages(ValueNodeUtil.asNode(lastLocationAccess), ValueNodeUtil.asNode(lla));
116         lastLocationAccess = lla;
117     }
118 }
