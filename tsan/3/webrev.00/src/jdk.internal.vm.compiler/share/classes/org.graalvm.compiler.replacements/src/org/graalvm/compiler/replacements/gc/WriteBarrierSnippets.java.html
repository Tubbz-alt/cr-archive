25 package org.graalvm.compiler.replacements.gc;
26 
27 import org.graalvm.compiler.nodes.NamedLocationIdentity;
28 import org.graalvm.compiler.nodes.PiNode;
29 import org.graalvm.compiler.nodes.SnippetAnchorNode;
30 import org.graalvm.compiler.nodes.memory.address.AddressNode.Address;
31 import org.graalvm.compiler.replacements.nodes.AssertionNode;
32 import org.graalvm.compiler.word.Word;
33 import jdk.internal.vm.compiler.word.LocationIdentity;
34 import jdk.internal.vm.compiler.word.WordFactory;
35 
36 public abstract class WriteBarrierSnippets {
37     public static final LocationIdentity GC_CARD_LOCATION = NamedLocationIdentity.mutable("GC-Card");
38 
39     protected static void verifyNotArray(Object object) {
40         if (object != null) {
41             // Manually build the null check and cast because we're in snippet that's lowered late.
42             AssertionNode.assertion(false, !PiNode.piCastNonNull(object, SnippetAnchorNode.anchor()).getClass().isArray(), "imprecise card mark used with array");
43         }
44     }
45 
46     protected static Word getPointerToFirstArrayElement(Address address, int length, int elementStride) {
47         long result = Word.fromAddress(address).rawValue();
48         if (elementStride < 0) {
49             // the address points to the place after the last array element
50             result = result + elementStride * length;
51         }
52         return WordFactory.unsigned(result);
53     }
54 
55     protected static Word getPointerToLastArrayElement(Address address, int length, int elementStride) {
56         long result = Word.fromAddress(address).rawValue();
57         if (elementStride < 0) {
58             // the address points to the place after the last array element
59             result = result + elementStride;
60         } else {
61             result = result + (length - 1) * elementStride;
62         }
63         return WordFactory.unsigned(result);
64     }
65 }
