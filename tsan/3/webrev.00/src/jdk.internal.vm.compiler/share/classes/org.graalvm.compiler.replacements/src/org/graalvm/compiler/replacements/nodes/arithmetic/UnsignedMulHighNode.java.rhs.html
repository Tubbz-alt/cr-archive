28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_2;

46 import jdk.vm.ci.meta.Value;

48 @NodeInfo(shortName = "|*H|", cycles = CYCLES_2, size = SIZE_2)
49 public final class UnsignedMulHighNode extends BinaryArithmeticNode<UMulHigh> implements Canonicalizable.BinaryCommutative<ValueNode> {
50     public static final NodeClass<UnsignedMulHighNode> TYPE = NodeClass.create(UnsignedMulHighNode.class);
51 
52     public UnsignedMulHighNode(ValueNode x, ValueNode y) {
53         super(TYPE, getArithmeticOpTable(x).getUMulHigh(), x, y);
54     }
55 
56     @Override
57     protected BinaryOp<UMulHigh> getOp(ArithmeticOpTable table) {
58         return table.getUMulHigh();
59     }
60 
61     @Override
62     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
63         Value a = nodeValueMap.operand(getX());
64         Value b = nodeValueMap.operand(getY());
65         nodeValueMap.setResult(this, gen.emitUMulHigh(a, b));
66     }
67 
68     @Override
69     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
70         ValueNode ret = super.canonical(tool, forX, forY);
71         if (ret != this) {
72             return ret;
73         }
74 
75         if (forX.isConstant() && !forY.isConstant()) {
76             // we try to swap and canonicalize
77             ValueNode improvement = canonical(tool, forY, forX);
78             if (improvement != this) {
79                 return improvement;
80             }
81             // if this fails we only swap
82             return new UnsignedMulHighNode(forY, forX);
83         }
84         return canonical(this, forY);
85     }
86 
87     private static ValueNode canonical(UnsignedMulHighNode self, ValueNode forY) {
88         if (forY.isConstant()) {
89             Constant c = forY.asConstant();
90             if (c instanceof PrimitiveConstant && ((PrimitiveConstant) c).getJavaKind().isNumericInteger()) {
91                 long i = ((PrimitiveConstant) c).asLong();
92                 if (i == 0 || i == 1) {
93                     return ConstantNode.forIntegerStamp(self.stamp(NodeView.DEFAULT), 0);
94                 }
95             }
96         }
97         return self;
98     }
99 }
