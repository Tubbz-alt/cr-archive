 52  * Node representing an exact integer addition that will throw an {@link ArithmeticException} in
 53  * case the addition would overflow the 32 bit range.
 54  */
 55 @NodeInfo(cycles = CYCLES_2, size = SIZE_2)

 57     public static final NodeClass<IntegerAddExactNode> TYPE = NodeClass.create(IntegerAddExactNode.class);

 60 
 61     public IntegerAddExactNode(ValueNode x, ValueNode y, GuardingNode guard) {
 62         super(TYPE, x, y);
 63         setStamp(x.stamp(NodeView.DEFAULT).unrestricted());
 64         assert x.stamp(NodeView.DEFAULT).isCompatible(y.stamp(NodeView.DEFAULT)) && x.stamp(NodeView.DEFAULT) instanceof IntegerStamp;
 65         this.guard = guard;
 66     }

 72          * know the result stamp of this node if we do not know whether we may deopt. If we know we
 73          * can never overflow we will replace this node with its non overflow checking counterpart
 74          * anyway.
 75          */
 76         return false;

122         newLowerBound |= newDownMask;
123         return IntegerStamp.create(bits, newLowerBound, newUpperBound, newDownMask, newUpMask);
124     }

126     @Override
127     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
128         if (forX.isConstant() && !forY.isConstant()) {
129             return new IntegerAddExactNode(forY, forX, guard).canonical(tool);
130         }
131         if (forX.isConstant() && forY.isConstant()) {
132             return canonicalXYconstant(forX, forY);
133         } else if (forY.isConstant()) {
134             long c = forY.asJavaConstant().asLong();
135             if (c == 0) {
136                 return forX;
137             }
138         }
139         if (!IntegerStamp.addCanOverflow((IntegerStamp) forX.stamp(NodeView.DEFAULT), (IntegerStamp) forY.stamp(NodeView.DEFAULT))) {
140             return new AddNode(forX, forY).canonical(tool);
141         }



142         return this;
143     }
144 
145     private ValueNode canonicalXYconstant(ValueNode forX, ValueNode forY) {
146         JavaConstant xConst = forX.asJavaConstant();
147         JavaConstant yConst = forY.asJavaConstant();
148         assert xConst.getJavaKind() == yConst.getJavaKind();
149         try {
150             if (xConst.getJavaKind() == JavaKind.Int) {
151                 return ConstantNode.forInt(Math.addExact(xConst.asInt(), yConst.asInt()));
152             } else {
153                 assert xConst.getJavaKind() == JavaKind.Long;
154                 return ConstantNode.forLong(Math.addExact(xConst.asLong(), yConst.asLong()));
155             }
156         } catch (ArithmeticException ex) {
157             // The operation will result in an overflow exception, so do not canonicalize.
158         }
159         return this;
160     }
