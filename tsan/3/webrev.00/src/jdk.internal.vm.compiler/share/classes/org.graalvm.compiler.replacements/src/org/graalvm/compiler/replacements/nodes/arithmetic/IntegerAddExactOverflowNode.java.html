 45 import jdk.vm.ci.meta.JavaConstant;
 46 import jdk.vm.ci.meta.JavaKind;

 48 @NodeInfo(cycles = CYCLES_2, size = SIZE_2)
 49 public final class IntegerAddExactOverflowNode extends IntegerExactOverflowNode implements Simplifiable, BinaryCommutative<ValueNode>, IterableNodeType {
 50     public static final NodeClass<IntegerAddExactOverflowNode> TYPE = NodeClass.create(IntegerAddExactOverflowNode.class);
 51 
 52     public IntegerAddExactOverflowNode(ValueNode x, ValueNode y) {
 53         super(TYPE, x, y);
 54     }
 55 
 56     @Override
 57     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 58         if (forX.isConstant() && !forY.isConstant()) {
 59             return new IntegerAddExactOverflowNode(forY, forX).canonical(tool);
 60         }
 61         if (forX.isConstant() && forY.isConstant()) {
 62             return canonicalXYconstant(forX, forY);
 63         } else if (forY.isConstant()) {
 64             long c = forY.asJavaConstant().asLong();
 65             if (c == 0) {
 66                 return LogicConstantNode.forBoolean(false);
 67             }
 68         }
 69         if (!IntegerStamp.addCanOverflow((IntegerStamp) forX.stamp(NodeView.DEFAULT), (IntegerStamp) forY.stamp(NodeView.DEFAULT))) {
 70             return LogicConstantNode.forBoolean(false);
 71         }
 72         return this;
 73     }
 74 
 75     private static LogicConstantNode canonicalXYconstant(ValueNode forX, ValueNode forY) {
 76         JavaConstant xConst = forX.asJavaConstant();
 77         JavaConstant yConst = forY.asJavaConstant();
 78         assert xConst.getJavaKind() == yConst.getJavaKind();
 79         try {
 80             if (xConst.getJavaKind() == JavaKind.Int) {
 81                 Math.addExact(xConst.asInt(), yConst.asInt());
 82             } else {
 83                 assert xConst.getJavaKind() == JavaKind.Long;
 84                 Math.addExact(xConst.asLong(), yConst.asLong());
 85             }
 86         } catch (ArithmeticException ex) {
 87             // Always overflows
 88             return LogicConstantNode.forBoolean(true);
 89         }
 90         // Never overflows
 91         return LogicConstantNode.forBoolean(false);
 92     }
 93 
 94     @Override
 95     protected IntegerExactArithmeticSplitNode createSplit(Stamp splitStamp, AbstractBeginNode next, AbstractBeginNode overflow) {
 96         return new IntegerAddExactSplitNode(splitStamp, x, y, next, overflow);
 97     }
 98 
 99     @Override
100     protected Class<? extends BinaryNode> getCoupledType() {
101         return IntegerAddExactNode.class;
102     }
103 }
