25 package org.graalvm.graphio;
26 
27 final class DefaultGraphTypes implements GraphTypes {
28     static final GraphTypes DEFAULT = new DefaultGraphTypes();
29 
30     private DefaultGraphTypes() {
31     }
32 

33     @Override
34     public Class<?> enumClass(Object enumValue) {
35         if (enumValue instanceof Enum<?>) {
36             return enumValue.getClass();






37         }
38         return null;
39     }
40 
41     @Override
42     public int enumOrdinal(Object obj) {
43         if (obj instanceof Enum<?>) {
44             return ((Enum<?>) obj).ordinal();
45         }
46         return -1;
47     }
48 
49     @SuppressWarnings("unchecked")
50     @Override
51     public String[] enumTypeValues(Object clazz) {
52         if (clazz instanceof Class<?>) {
53             Class<? extends Enum<?>> enumClass = (Class<? extends Enum<?>>) clazz;
54             Enum<?>[] constants = enumClass.getEnumConstants();
55             if (constants != null) {
56                 String[] names = new String[constants.length];
57                 for (int i = 0; i < constants.length; i++) {
58                     names[i] = constants[i].name();
59                 }
60                 return names;
61             }
62         }
63         return null;
64     }
65 
66     @Override
67     public String typeName(Object clazz) {
68         if (clazz instanceof Class<?>) {
69             return ((Class<?>) clazz).getName();
70         }
71         return null;
72     }
73 
74 }
