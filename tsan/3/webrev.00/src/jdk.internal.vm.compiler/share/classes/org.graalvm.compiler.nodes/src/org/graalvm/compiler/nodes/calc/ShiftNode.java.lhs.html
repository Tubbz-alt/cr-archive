 50 /**
 51  * The {@code ShiftOp} class represents shift operations.
 52  */
 53 @NodeInfo(cycles = CYCLES_1, size = SIZE_1)
 54 public abstract class ShiftNode<OP> extends BinaryNode implements ArithmeticOperation, ArithmeticLIRLowerable, NarrowableArithmeticNode {
 55 
 56     @SuppressWarnings("rawtypes") public static final NodeClass<ShiftNode> TYPE = NodeClass.create(ShiftNode.class);
 57 
 58     protected interface SerializableShiftFunction<T> extends Function<ArithmeticOpTable, ShiftOp<T>>, Serializable {
 59     }
 60 
 61     protected final SerializableShiftFunction<OP> getOp;
 62 
 63     /**
 64      * Creates a new shift operation.
 65      *
 66      * @param x the first input value
 67      * @param s the second input value
 68      */
 69     protected ShiftNode(NodeClass<? extends ShiftNode<OP>> c, SerializableShiftFunction<OP> getOp, ValueNode x, ValueNode s) {
 70         super(c, getOp.apply(ArithmeticOpTable.forStamp(x.stamp(NodeView.DEFAULT))).foldStamp(x.stamp(NodeView.DEFAULT), (IntegerStamp) s.stamp(NodeView.DEFAULT)), x, s);
 71         assert ((IntegerStamp) s.stamp(NodeView.DEFAULT)).getBits() == 32;
 72         this.getOp = getOp;
 73     }
 74 


 75     protected final ShiftOp<OP> getOp(ValueNode forValue) {
 76         return getOp.apply(ArithmeticOpTable.forStamp(forValue.stamp(NodeView.DEFAULT)));
 77     }
 78 
 79     @Override
 80     public final ShiftOp<OP> getArithmeticOp() {
 81         return getOp(getX());
 82     }
 83 
 84     @Override
 85     public Stamp foldStamp(Stamp stampX, Stamp stampY) {
 86         return getArithmeticOp().foldStamp(stampX, (IntegerStamp) stampY);
 87     }
 88 
 89     @Override
 90     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 91         NodeView view = NodeView.from(tool);
 92         ValueNode valueNode = canonical(getOp(forX), stamp(NodeView.DEFAULT), forX, forY, view);
 93         if (valueNode != null) {
 94             return valueNode;
 95         }
 96         return this;
 97     }
 98 
 99     @SuppressWarnings("unused")
100     public static <OP> ValueNode canonical(ShiftOp<OP> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
101         if (forX.isConstant() && forY.isConstant()) {
102             JavaConstant amount = forY.asJavaConstant();
103             assert amount.getJavaKind() == JavaKind.Int;
104             return ConstantNode.forPrimitive(stamp, op.foldConstant(forX.asConstant(), amount.asInt()));
105         }
106         return null;
107     }
108 
109     public int getShiftAmountMask() {
110         return getArithmeticOp().getShiftAmountMask(stamp(NodeView.DEFAULT));
111     }
112 
113     @Override
114     public boolean isNarrowable(int resultBits) {
115         assert CodeUtil.isPowerOf2(resultBits);
116         int narrowMask = resultBits - 1;
117         int wideMask = getShiftAmountMask();
118         assert (wideMask & narrowMask) == narrowMask : String.format("wideMask %x should be wider than narrowMask %x", wideMask, narrowMask);
119 
120         /*
121          * Shifts are special because narrowing them also changes the implicit mask of the shift
122          * amount. We can narrow only if (y & wideMask) == (y & narrowMask) for all possible values
123          * of y.
124          */
125         IntegerStamp yStamp = (IntegerStamp) getY().stamp(NodeView.DEFAULT);
126         return (yStamp.upMask() & (wideMask & ~narrowMask)) == 0;
127     }
128 }
