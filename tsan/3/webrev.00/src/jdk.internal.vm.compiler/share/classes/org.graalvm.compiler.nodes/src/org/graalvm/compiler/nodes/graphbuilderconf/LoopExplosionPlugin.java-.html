25 package org.graalvm.compiler.nodes.graphbuilderconf;
26 
27 import jdk.vm.ci.meta.ResolvedJavaMethod;
28 
29 public interface LoopExplosionPlugin extends GraphBuilderPlugin {
30 
31     enum LoopExplosionKind {
32         /**
33          * No loop explosion.
34          */
35         NONE,
36         /**
37          * Fully unroll all loops. The loops must have a known finite number of iterations. If a
38          * loop has multiple loop ends, they are merged so that the subsequent loop iteration is
39          * processed only once. For example, a loop with 4 iterations and 2 loop ends leads to
40          * 1+1+1+1 = 4 copies of the loop body.
41          */
42         FULL_UNROLL,
43         /**
44          * Fully explode all loops. The loops must have a known finite number of iterations. If a
45          * loop has multiple loop ends, they are not merged so that subsequent loop iterations are
46          * processed multiple times. For example, a loop with 4 iterations and 2 loop ends leads to
47          * 1+2+4+8 = 15 copies of the loop body.
48          */
49         FULL_EXPLODE,
50         /**
51          * Like {@link #FULL_EXPLODE}, but in addition explosion does not stop at loop exits. Code
52          * after the loop is duplicated for every loop exit of every loop iteration. For example, a
53          * loop with 4 iterations and 2 loop exits leads to 4 * 2 = 8 copies of the code after the
54          * loop.
55          */
56         FULL_EXPLODE_UNTIL_RETURN,
57         /**
58          * like {@link #FULL_EXPLODE}, but copies of the loop body that have the exact same state
59          * (all local variables have the same value) are merged. This reduces the number of copies
60          * necessary, but can introduce loops again. This kind is useful for bytecode interpreter
61          * loops.
62          */
63         MERGE_EXPLODE
64     }
65 
66     LoopExplosionKind loopExplosionKind(ResolvedJavaMethod method);
67 }
