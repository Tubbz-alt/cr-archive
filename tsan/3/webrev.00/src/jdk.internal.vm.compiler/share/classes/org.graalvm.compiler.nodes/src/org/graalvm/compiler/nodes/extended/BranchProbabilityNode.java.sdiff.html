 53 /**
 54  * Instances of this node class will look for a preceding if node and put the given probability into
 55  * the if node's taken probability. Then the branch probability node will be removed. This node is
 56  * intended primarily for snippets, so that they can define their fast and slow paths.
 57  */
 58 @NodeInfo(cycles = CYCLES_0, cyclesRationale = "Artificial Node", size = SIZE_0)
 59 public final class BranchProbabilityNode extends FloatingNode implements Simplifiable, Lowerable {
 60 
 61     public static final NodeClass<BranchProbabilityNode> TYPE = NodeClass.create(BranchProbabilityNode.class);
 62     public static final double LIKELY_PROBABILITY = 0.6;
 63     public static final double NOT_LIKELY_PROBABILITY = 1 - LIKELY_PROBABILITY;
 64 
 65     public static final double FREQUENT_PROBABILITY = 0.9;
 66     public static final double NOT_FREQUENT_PROBABILITY = 1 - FREQUENT_PROBABILITY;
 67 
 68     public static final double FAST_PATH_PROBABILITY = 0.99;
 69     public static final double SLOW_PATH_PROBABILITY = 1 - FAST_PATH_PROBABILITY;
 70 
 71     public static final double VERY_FAST_PATH_PROBABILITY = 0.999;
 72     public static final double VERY_SLOW_PATH_PROBABILITY = 1 - VERY_FAST_PATH_PROBABILITY;
 73 


 74     /*
 75      * This probability may seem excessive, but it makes a difference in long running loops. Lets
 76      * say a loop is executed 100k times and it has a few null checks with probability 0.999. As
 77      * these probabilities multiply for every loop iteration, the overall loop frequency will be
 78      * calculated as approximately 30 while it should be 100k.
 79      */
 80     public static final double LUDICROUSLY_FAST_PATH_PROBABILITY = 0.999999;
 81     public static final double LUDICROUSLY_SLOW_PATH_PROBABILITY = 1 - LUDICROUSLY_FAST_PATH_PROBABILITY;
 82 
 83     @Input ValueNode probability;
 84     @Input ValueNode condition;
 85 
 86     public BranchProbabilityNode(ValueNode probability, ValueNode condition) {
 87         super(TYPE, StampFactory.forKind(JavaKind.Boolean));
 88         this.probability = probability;
 89         this.condition = condition;
 90     }
 91 
 92     public ValueNode getProbability() {
 93         return probability;
 94     }
 95 
 96     public ValueNode getCondition() {
 97         return condition;
 98     }
