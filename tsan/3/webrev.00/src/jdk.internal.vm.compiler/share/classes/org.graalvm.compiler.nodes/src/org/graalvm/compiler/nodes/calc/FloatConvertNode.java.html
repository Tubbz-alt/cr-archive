 48 /**
 49  * A {@code FloatConvert} converts between integers and floating point numbers according to Java
 50  * semantics.
 51  */
 52 @NodeInfo(cycles = CYCLES_8)
 53 public final class FloatConvertNode extends UnaryArithmeticNode<FloatConvertOp> implements ConvertNode, Lowerable, ArithmeticLIRLowerable {
 54     public static final NodeClass<FloatConvertNode> TYPE = NodeClass.create(FloatConvertNode.class);
 55 
 56     protected final FloatConvert op;
 57 
 58     public FloatConvertNode(FloatConvert op, ValueNode input) {
 59         super(TYPE, getArithmeticOpTable(input).getFloatConvert(op), input);
 60         this.op = op;
 61     }
 62 
 63     public static ValueNode create(FloatConvert op, ValueNode input, NodeView view) {
 64         ValueNode synonym = findSynonym(input, ArithmeticOpTable.forStamp(input.stamp(view)).getFloatConvert(op));
 65         if (synonym != null) {
 66             return synonym;
 67         }
 68         return new FloatConvertNode(op, input);
 69     }
 70 
 71     @Override
 72     protected UnaryOp<FloatConvertOp> getOp(ArithmeticOpTable table) {
 73         return table.getFloatConvert(op);
 74     }
 75 
 76     public FloatConvert getFloatConvert() {
 77         return op;
 78     }
 79 
 80     @Override
 81     public Constant convert(Constant c, ConstantReflectionProvider constantReflection) {
 82         return getArithmeticOp().foldConstant(c);
 83     }
 84 
 85     @Override
 86     public Constant reverse(Constant c, ConstantReflectionProvider constantReflection) {
 87         FloatConvertOp reverse = ArithmeticOpTable.forStamp(stamp(NodeView.DEFAULT)).getFloatConvert(op.reverse());
 88         return reverse.foldConstant(c);
 89     }
 90 
 91     @Override
 92     public boolean isLossless() {
 93         switch (getFloatConvert()) {
 94             case F2D:
 95             case I2D:
 96                 return true;
 97             default:
 98                 return false;
 99         }
100     }
101 
102     @Override
103     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
104         ValueNode ret = super.canonical(tool, forValue);
105         if (ret != this) {
106             return ret;
107         }
108 
109         if (forValue instanceof FloatConvertNode) {
110             FloatConvertNode other = (FloatConvertNode) forValue;
111             if (other.isLossless() && other.op == this.op.reverse()) {
112                 return other.getValue();
113             }
114         }
115         return this;
116     }
117 
118     @Override
119     public void lower(LoweringTool tool) {
120         tool.getLowerer().lower(this, tool);
121     }
122 
123     @Override
124     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
125         nodeValueMap.setResult(this, gen.emitFloatConvert(getFloatConvert(), nodeValueMap.operand(getValue())));
126     }
127 
128     @Override
129     public boolean mayNullCheckSkipConversion() {
130         return false;
131     }
132 }
