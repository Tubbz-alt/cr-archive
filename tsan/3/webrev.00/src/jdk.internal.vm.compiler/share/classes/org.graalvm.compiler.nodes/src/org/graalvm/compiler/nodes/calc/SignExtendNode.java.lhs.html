 45 /**
 46  * The {@code SignExtendNode} converts an integer to a wider integer using sign extension.
 47  */
 48 @NodeInfo(cycles = CYCLES_1)
 49 public final class SignExtendNode extends IntegerConvertNode<SignExtend, Narrow> {
 50 
 51     public static final NodeClass<SignExtendNode> TYPE = NodeClass.create(SignExtendNode.class);
 52 
 53     public SignExtendNode(ValueNode input, int resultBits) {
 54         this(input, PrimitiveStamp.getBits(input.stamp(NodeView.DEFAULT)), resultBits);
 55         assert 0 < PrimitiveStamp.getBits(input.stamp(NodeView.DEFAULT)) && PrimitiveStamp.getBits(input.stamp(NodeView.DEFAULT)) <= resultBits;
 56     }
 57 
 58     public SignExtendNode(ValueNode input, int inputBits, int resultBits) {
 59         super(TYPE, ArithmeticOpTable::getSignExtend, ArithmeticOpTable::getNarrow, inputBits, resultBits, input);
 60     }
 61 
 62     public static ValueNode create(ValueNode input, int resultBits, NodeView view) {
 63         return create(input, PrimitiveStamp.getBits(input.stamp(view)), resultBits, view);
 64     }
 65 
 66     public static ValueNode create(ValueNode input, int inputBits, int resultBits, NodeView view) {
 67         IntegerConvertOp<SignExtend> signExtend = ArithmeticOpTable.forStamp(input.stamp(view)).getSignExtend();
 68         ValueNode synonym = findSynonym(signExtend, input, inputBits, resultBits, signExtend.foldStamp(inputBits, resultBits, input.stamp(view)));
 69         if (synonym != null) {
 70             return synonym;
 71         }
 72         return canonical(null, input, inputBits, resultBits, view);
 73     }
 74 










 75     @Override
 76     public boolean isLossless() {
 77         return true;
 78     }
 79 
 80     @Override
 81     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
 82         NodeView view = NodeView.from(tool);
 83         ValueNode ret = super.canonical(tool, forValue);
 84         if (ret != this) {
 85             return ret;
 86         }
 87 
 88         return canonical(this, forValue, getInputBits(), getResultBits(), view);
 89     }
 90 
 91     private static ValueNode canonical(SignExtendNode self, ValueNode forValue, int inputBits, int resultBits, NodeView view) {
 92         if (forValue instanceof SignExtendNode) {
 93             // sxxx -(sign-extend)-> ssss sxxx -(sign-extend)-> ssssssss sssssxxx
 94             // ==> sxxx -(sign-extend)-> ssssssss sssssxxx
 95             SignExtendNode other = (SignExtendNode) forValue;
 96             return SignExtendNode.create(other.getValue(), other.getInputBits(), resultBits, view);
 97         } else if (forValue instanceof ZeroExtendNode) {
 98             ZeroExtendNode other = (ZeroExtendNode) forValue;
 99             if (other.getResultBits() > other.getInputBits()) {
100                 // sxxx -(zero-extend)-> 0000 sxxx -(sign-extend)-> 00000000 0000sxxx
101                 // ==> sxxx -(zero-extend)-> 00000000 0000sxxx
102                 return ZeroExtendNode.create(other.getValue(), other.getInputBits(), resultBits, view, other.isInputAlwaysPositive());
103             }
104         }
105 
106         if (forValue.stamp(view) instanceof IntegerStamp) {
107             IntegerStamp inputStamp = (IntegerStamp) forValue.stamp(view);
108             if ((inputStamp.upMask() & (1L << (inputBits - 1))) == 0L) {
109                 // 0xxx -(sign-extend)-> 0000 0xxx
110                 // ==> 0xxx -(zero-extend)-> 0000 0xxx
111                 return ZeroExtendNode.create(forValue, inputBits, resultBits, view, true);
112             }
113         }
114         if (forValue instanceof NarrowNode) {
115             NarrowNode narrow = (NarrowNode) forValue;
116             Stamp inputStamp = narrow.getValue().stamp(view);
117             if (inputStamp instanceof IntegerStamp) {
118                 IntegerStamp istamp = (IntegerStamp) inputStamp;
119                 long mask = CodeUtil.mask(PrimitiveStamp.getBits(narrow.stamp(view)) - 1);
120                 if (~mask <= istamp.lowerBound() && istamp.upperBound() <= mask) {
121                     // The original value cannot change because of the narrow and sign extend.
122                     if (istamp.getBits() < resultBits) {
123                         // Need to keep the sign extend, skip the narrow.
124                         return create(narrow.getValue(), resultBits, view);
125                     } else if (istamp.getBits() > resultBits) {
126                         // Need to keep the narrow, skip the sign extend.
127                         return NarrowNode.create(narrow.getValue(), resultBits, view);
128                     } else {
129                         assert istamp.getBits() == resultBits;
130                         // Just return the original value.
131                         return narrow.getValue();
132                     }
133                 }
134             }
135         }
136         return self != null ? self : new SignExtendNode(forValue, inputBits, resultBits);
137     }
138 
139     @Override
140     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
141         nodeValueMap.setResult(this, gen.emitSignExtend(nodeValueMap.operand(getValue()), getInputBits(), getResultBits()));
142     }
143 
144     @Override
145     public boolean mayNullCheckSkipConversion() {
146         return true;
147     }
148 }
