25 package org.graalvm.compiler.nodes.spi;
26 
27 import org.graalvm.compiler.graph.Node;
28 import org.graalvm.compiler.nodes.ValueNode;
29 import org.graalvm.compiler.nodes.memory.address.AddressNode;
30 
31 import jdk.vm.ci.meta.JavaKind;
32 
33 /**
34  * Provides a capability for replacing a higher node with one or more lower level nodes.
35  */
36 public interface LoweringProvider {
37 
38     void lower(Node n, LoweringTool tool);
39 
40     /**
41      * Reconstructs the array index from an address node that was created as a lowering of an
42      * indexed access to an array.
43      *
44      * @param elementKind the {@link JavaKind} of the array elements
45      * @param address an {@link AddressNode} pointing to an element in an array
46      * @return a node that gives the index of the element
47      */
48     ValueNode reconstructArrayIndex(JavaKind elementKind, AddressNode address);
49 
50     /**
51      * Indicates the smallest width for comparing an integer value on the target platform.
52      */
53     default Integer smallestCompareWidth() {
54         // most platforms only support 32 and 64 bit compares
55         return 32;
56     }
57 }
