25 package org.graalvm.compiler.nodes.util;
26 
27 import org.graalvm.compiler.core.common.NumUtil;
28 import org.graalvm.compiler.core.common.type.IntegerStamp;
29 import org.graalvm.compiler.core.common.type.StampFactory;
30 import org.graalvm.compiler.nodes.LogicNode;
31 import org.graalvm.compiler.nodes.NodeView;
32 import org.graalvm.compiler.nodes.ValueNode;
33 import org.graalvm.compiler.nodes.calc.IntegerBelowNode;
34 
35 import jdk.vm.ci.code.CodeUtil;
36 
37 public class UnsignedIntegerHelper extends IntegerHelper {
38     public UnsignedIntegerHelper(int bits) {
39         super(bits);
40     }
41 
42     @Override
43     public long upperBound(IntegerStamp stamp) {
44         assert stamp.getBits() == bits;
45         return stamp.unsignedUpperBound();
46     }
47 
48     @Override
49     public long lowerBound(IntegerStamp stamp) {
50         assert stamp.getBits() == bits;
51         return stamp.unsignedLowerBound();
52     }
53 
54     @Override
55     protected int rawCompare(long a, long b) {
56         return Long.compareUnsigned(a, b);
57     }
58 
59     @Override
60     protected long rawMin(long a, long b) {
61         return NumUtil.minUnsigned(a, b);
62     }
63 
64     @Override
65     protected long rawMax(long a, long b) {
66         return NumUtil.maxUnsigned(a, b);
67     }
68 
69     @Override
70     public long cast(long a) {
71         return CodeUtil.zeroExtend(a, bits);
72     }
73 
74     @Override
75     public long minValue() {
76         return 0;
77     }
78 
79     @Override
80     public long maxValue() {
81         return NumUtil.maxValueUnsigned(bits);
82     }
83 
84     @Override
85     public IntegerStamp stamp(long min, long max) {
86         return StampFactory.forUnsignedInteger(bits, min, max);
87     }
88 
89     @Override
90     public LogicNode createCompareNode(ValueNode x, ValueNode y, NodeView view) {
91         return IntegerBelowNode.create(x, y, view);
92     }
93 }
