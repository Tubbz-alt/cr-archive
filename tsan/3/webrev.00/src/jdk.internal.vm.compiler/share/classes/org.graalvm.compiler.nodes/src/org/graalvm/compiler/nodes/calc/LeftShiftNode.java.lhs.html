 27 import org.graalvm.compiler.core.common.type.ArithmeticOpTable;

 28 import org.graalvm.compiler.core.common.type.ArithmeticOpTable.ShiftOp.Shl;
 29 import org.graalvm.compiler.core.common.type.IntegerStamp;
 30 import org.graalvm.compiler.core.common.type.Stamp;
 31 import org.graalvm.compiler.graph.NodeClass;
 32 import org.graalvm.compiler.graph.spi.CanonicalizerTool;
 33 import org.graalvm.compiler.lir.gen.ArithmeticLIRGeneratorTool;
 34 import org.graalvm.compiler.nodeinfo.NodeInfo;
 35 import org.graalvm.compiler.nodes.ConstantNode;
 36 import org.graalvm.compiler.nodes.NodeView;
 37 import org.graalvm.compiler.nodes.ValueNode;
 38 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
 39 
 40 import jdk.vm.ci.meta.JavaKind;
 41 
 42 @NodeInfo(shortName = "<<")
 43 public final class LeftShiftNode extends ShiftNode<Shl> {
 44 
 45     public static final NodeClass<LeftShiftNode> TYPE = NodeClass.create(LeftShiftNode.class);
 46 
 47     public LeftShiftNode(ValueNode x, ValueNode y) {
 48         super(TYPE, ArithmeticOpTable::getShl, x, y);
 49     }
 50 
 51     public static ValueNode create(ValueNode x, ValueNode y, NodeView view) {
 52         ArithmeticOpTable.ShiftOp<Shl> op = ArithmeticOpTable.forStamp(x.stamp(view)).getShl();
 53         Stamp stamp = op.foldStamp(x.stamp(view), (IntegerStamp) y.stamp(view));
 54         ValueNode value = ShiftNode.canonical(op, stamp, x, y, view);
 55         if (value != null) {
 56             return value;
 57         }
 58 
 59         return canonical(null, op, stamp, x, y);
 60     }
 61 





 62     @Override
 63     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 64         ValueNode ret = super.canonical(tool, forX, forY);
 65         if (ret != this) {
 66             return ret;
 67         }
 68 
 69         return canonical(this, getArithmeticOp(), stamp(NodeView.DEFAULT), forX, forY);
 70     }
 71 
 72     private static ValueNode canonical(LeftShiftNode leftShiftNode, ArithmeticOpTable.ShiftOp<Shl> op, Stamp stamp, ValueNode forX, ValueNode forY) {
 73         LeftShiftNode self = leftShiftNode;
 74         if (forY.isConstant()) {
 75             int amount = forY.asJavaConstant().asInt();
 76             int originalAmount = amount;
 77             int mask = op.getShiftAmountMask(stamp);
 78             amount &= mask;
 79             if (amount == 0) {
 80                 return forX;
 81             }
 82             if (forX instanceof ShiftNode) {
 83                 ShiftNode<?> other = (ShiftNode<?>) forX;
 84                 if (other.getY().isConstant()) {
 85                     int otherAmount = other.getY().asJavaConstant().asInt() & mask;
 86                     if (other instanceof LeftShiftNode) {
 87                         int total = amount + otherAmount;
 88                         if (total != (total & mask)) {
 89                             return ConstantNode.forIntegerKind(stamp.getStackKind(), 0);
 90                         }
 91                         return new LeftShiftNode(other.getX(), ConstantNode.forInt(total));
 92                     } else if ((other instanceof RightShiftNode || other instanceof UnsignedRightShiftNode) && otherAmount == amount) {
 93                         if (stamp.getStackKind() == JavaKind.Long) {
 94                             return new AndNode(other.getX(), ConstantNode.forLong(-1L << amount));
 95                         } else {
 96                             assert stamp.getStackKind() == JavaKind.Int;
 97                             return new AndNode(other.getX(), ConstantNode.forInt(-1 << amount));
 98                         }
 99                     }
100                 }
101             }
102             if (originalAmount != amount) {
103                 return new LeftShiftNode(forX, ConstantNode.forInt(amount));
104             }
105         }
106         if (self == null) {
107             self = new LeftShiftNode(forX, forY);
108         }
109         return self;
110     }
111 
112     @Override
113     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
114         nodeValueMap.setResult(this, gen.emitShl(nodeValueMap.operand(getX()), nodeValueMap.operand(getY())));
115     }
116 }
