 53 /**
 54  * Instances of this node class will look for a preceding if node and put the given probability into
 55  * the if node's taken probability. Then the branch probability node will be removed. This node is
 56  * intended primarily for snippets, so that they can define their fast and slow paths.
 57  */
 58 @NodeInfo(cycles = CYCLES_0, cyclesRationale = "Artificial Node", size = SIZE_0)
 59 public final class BranchProbabilityNode extends FloatingNode implements Simplifiable, Lowerable {
 60 
 61     public static final NodeClass<BranchProbabilityNode> TYPE = NodeClass.create(BranchProbabilityNode.class);
 62     public static final double LIKELY_PROBABILITY = 0.6;
 63     public static final double NOT_LIKELY_PROBABILITY = 1 - LIKELY_PROBABILITY;
 64 
 65     public static final double FREQUENT_PROBABILITY = 0.9;
 66     public static final double NOT_FREQUENT_PROBABILITY = 1 - FREQUENT_PROBABILITY;
 67 
 68     public static final double FAST_PATH_PROBABILITY = 0.99;
 69     public static final double SLOW_PATH_PROBABILITY = 1 - FAST_PATH_PROBABILITY;
 70 
 71     public static final double VERY_FAST_PATH_PROBABILITY = 0.999;
 72     public static final double VERY_SLOW_PATH_PROBABILITY = 1 - VERY_FAST_PATH_PROBABILITY;
 73 


 74     /*
 75      * This probability may seem excessive, but it makes a difference in long running loops. Lets
 76      * say a loop is executed 100k times and it has a few null checks with probability 0.999. As
 77      * these probabilities multiply for every loop iteration, the overall loop frequency will be
 78      * calculated as approximately 30 while it should be 100k.
 79      */
 80     public static final double LUDICROUSLY_FAST_PATH_PROBABILITY = 0.999999;
 81     public static final double LUDICROUSLY_SLOW_PATH_PROBABILITY = 1 - LUDICROUSLY_FAST_PATH_PROBABILITY;
 82 
 83     @Input ValueNode probability;
 84     @Input ValueNode condition;
 85 
 86     public BranchProbabilityNode(ValueNode probability, ValueNode condition) {
 87         super(TYPE, StampFactory.forKind(JavaKind.Boolean));
 88         this.probability = probability;
 89         this.condition = condition;
 90     }
 91 
 92     public ValueNode getProbability() {
 93         return probability;
 94     }
 95 
 96     public ValueNode getCondition() {
 97         return condition;
 98     }
 99 









100     @Override
101     public void simplify(SimplifierTool tool) {
102         if (!hasUsages()) {
103             return;
104         }
105         if (probability.isConstant()) {
106             double probabilityValue = probability.asJavaConstant().asDouble();
107             if (probabilityValue < 0.0) {
108                 throw new GraalError("A negative probability of " + probabilityValue + " is not allowed!");
109             } else if (probabilityValue > 1.0) {
110                 throw new GraalError("A probability of more than 1.0 (" + probabilityValue + ") is not allowed!");
111             } else if (Double.isNaN(probabilityValue)) {
112                 /*
113                  * We allow NaN if the node is in unreachable code that will eventually fall away,
114                  * or else an error will be thrown during lowering since we keep the node around.
115                  */
116                 return;
117             }
118             boolean usageFound = false;
119             for (IntegerEqualsNode node : this.usages().filter(IntegerEqualsNode.class)) {
120                 assert node.condition() == CanonicalCondition.EQ;
121                 ValueNode other = node.getX();
122                 if (node.getX() == this) {
123                     other = node.getY();
124                 }
125                 if (other.isConstant()) {
126                     double probabilityToSet = probabilityValue;
127                     if (other.asJavaConstant().asInt() == 0) {
128                         probabilityToSet = 1.0 - probabilityToSet;
129                     }
130                     for (IfNode ifNodeUsages : node.usages().filter(IfNode.class)) {
131                         usageFound = true;
132                         ifNodeUsages.setTrueSuccessorProbability(probabilityToSet);
133                     }
134                     if (!usageFound) {
135                         usageFound = node.usages().filter(NodePredicates.isA(FixedGuardNode.class).or(ConditionalNode.class)).isNotEmpty();
136                     }
137                 }
138             }
139             if (usageFound) {
140                 ValueNode currentCondition = condition;
141                 IntegerStamp currentStamp = (IntegerStamp) currentCondition.stamp(NodeView.DEFAULT);
142                 if (currentStamp.lowerBound() < 0 || 1 < currentStamp.upperBound()) {
143                     ValueNode narrow = graph().maybeAddOrUnique(NarrowNode.create(currentCondition, 1, NodeView.DEFAULT));
144                     currentCondition = graph().maybeAddOrUnique(ZeroExtendNode.create(narrow, 32, NodeView.DEFAULT));
145                 }
146                 replaceAndDelete(currentCondition);
147                 if (tool != null) {
148                     tool.addToWorkList(currentCondition.usages());
149                 }
150             } else {
151                 if (!isSubstitutionGraph()) {
152                     throw new GraalError("Wrong usage of branch probability injection!");
153                 }
154             }
155         }
156     }
157 
158     private boolean isSubstitutionGraph() {
159         return hasExactlyOneUsage() && usages().first() instanceof ReturnNode;
160     }
161 
162     /**
163      * This intrinsic should only be used for the condition of an if statement. The parameter
164      * condition should also only denote a simple condition and not a combined condition involving
165      * &amp;&amp; or || operators. It injects the probability of the condition into the if
166      * statement.
167      *
168      * @param probability the probability that the given condition is true as a double value between
169      *            0.0 and 1.0.
170      * @param condition the simple condition without any &amp;&amp; or || operators
171      * @return the condition
172      */
173     @NodeIntrinsic
174     public static native boolean probability(double probability, boolean condition);
175 
176     @Override
177     public void lower(LoweringTool tool) {
178         throw new GraalError("Branch probability could not be injected, because the probability value did not reduce to a constant value.");
179     }
180 }
