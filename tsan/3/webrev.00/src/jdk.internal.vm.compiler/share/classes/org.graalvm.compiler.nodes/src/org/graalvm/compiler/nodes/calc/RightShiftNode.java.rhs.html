 45 @NodeInfo(shortName = ">>")
 46 public final class RightShiftNode extends ShiftNode<Shr> {
 47 
 48     public static final NodeClass<RightShiftNode> TYPE = NodeClass.create(RightShiftNode.class);
 49 
 50     public RightShiftNode(ValueNode x, ValueNode y) {
 51         super(TYPE, getArithmeticOpTable(x).getShr(), x, y);
 52     }
 53 
 54     public static ValueNode create(ValueNode x, int y, NodeView view) {
 55         if (y == 0) {
 56             return x;
 57         }
 58         return create(x, ConstantNode.forInt(y), view);
 59     }
 60 
 61     public static ValueNode create(ValueNode x, ValueNode y, NodeView view) {
 62         ArithmeticOpTable.ShiftOp<Shr> op = ArithmeticOpTable.forStamp(x.stamp(view)).getShr();
 63         Stamp stamp = op.foldStamp(x.stamp(view), (IntegerStamp) y.stamp(view));
 64         ValueNode value = ShiftNode.canonical(op, stamp, x, y, view);
 65         if (value != null) {
 66             return value;
 67         }
 68 
 69         return canonical(null, op, stamp, x, y, view);
 70     }
 71 
 72     @Override
 73     protected ShiftOp<Shr> getOp(ArithmeticOpTable table) {
 74         return table.getShr();
 75     }
 76 
 77     @Override
 78     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 79         NodeView view = NodeView.from(tool);
 80         ValueNode ret = super.canonical(tool, forX, forY);
 81         if (ret != this) {
 82             return ret;
 83         }
 84 
 85         return canonical(this, getArithmeticOp(), stamp(view), forX, forY, view);
 86     }
 87 
 88     private static ValueNode canonical(RightShiftNode rightShiftNode, ArithmeticOpTable.ShiftOp<Shr> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
 89         RightShiftNode self = rightShiftNode;
 90         if (forX.stamp(view) instanceof IntegerStamp && ((IntegerStamp) forX.stamp(view)).isPositive()) {
 91             return new UnsignedRightShiftNode(forX, forY);
 92         }
 93 
 94         Stamp xStampGeneric = forX.stamp(view);
 95         if (xStampGeneric instanceof IntegerStamp) {
 96             IntegerStamp xStamp = (IntegerStamp) xStampGeneric;
 97             if (xStamp.lowerBound() >= -1 && xStamp.upperBound() <= 0) {
 98                 // Right shift by any amount does not change any bit.
 99                 return forX;
100             }
101         }
102 
103         if (forY.isConstant()) {
104             int amount = forY.asJavaConstant().asInt();
105             int originalAmout = amount;
106             int mask = op.getShiftAmountMask(stamp);
107             amount &= mask;
108             if (amount == 0) {
109                 return forX;
110             }
111 
112             if (xStampGeneric instanceof IntegerStamp) {
113                 IntegerStamp xStamp = (IntegerStamp) xStampGeneric;
114 
115                 if (xStamp.lowerBound() >> amount == xStamp.upperBound() >> amount) {
116                     // Right shift turns the result of the expression into a constant.
117                     return ConstantNode.forIntegerKind(stamp.getStackKind(), xStamp.lowerBound() >> amount);
118                 }
119             }
120 
121             if (forX instanceof ShiftNode) {
122                 ShiftNode<?> other = (ShiftNode<?>) forX;
123                 if (other.getY().isConstant()) {
124                     int otherAmount = other.getY().asJavaConstant().asInt() & mask;
125                     if (other instanceof RightShiftNode) {
126                         int total = amount + otherAmount;
127                         if (total != (total & mask)) {
128                             assert other.getX().stamp(view) instanceof IntegerStamp;
129                             IntegerStamp istamp = (IntegerStamp) other.getX().stamp(view);
130 
131                             if (istamp.isPositive()) {
132                                 return ConstantNode.forIntegerKind(stamp.getStackKind(), 0);
133                             }
134                             if (istamp.isStrictlyNegative()) {
135                                 return ConstantNode.forIntegerKind(stamp.getStackKind(), -1L);
136                             }
137 
138                             /*
139                              * if we cannot replace both shifts with a constant, replace them by a
140                              * full shift for this kind
141                              */
142                             assert total >= mask;
143                             return new RightShiftNode(other.getX(), ConstantNode.forInt(mask));
144                         }
145                         return new RightShiftNode(other.getX(), ConstantNode.forInt(total));
146                     }
147                 }
148             }
149             if (originalAmout != amount) {
150                 return new RightShiftNode(forX, ConstantNode.forInt(amount));
151             }
152         }
153         if (self == null) {
154             self = new RightShiftNode(forX, forY);
155         }
156         return self;
157     }
158 
159     @Override
160     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
161         nodeValueMap.setResult(this, gen.emitShr(nodeValueMap.operand(getX()), nodeValueMap.operand(getY())));
162     }
163 
164     @Override
165     public boolean isNarrowable(int resultBits) {
166         if (super.isNarrowable(resultBits)) {
167             /*
168              * For signed right shifts, the narrow can be done before the shift if the cut off bits
169              * are all equal to the sign bit of the input. That's equivalent to the condition that
170              * the input is in the signed range of the narrow type.
171              */
172             IntegerStamp inputStamp = (IntegerStamp) getX().stamp(NodeView.DEFAULT);
173             return CodeUtil.minValue(resultBits) <= inputStamp.lowerBound() && inputStamp.upperBound() <= CodeUtil.maxValue(resultBits);
174         } else {
175             return false;
176         }
177     }
178 }
