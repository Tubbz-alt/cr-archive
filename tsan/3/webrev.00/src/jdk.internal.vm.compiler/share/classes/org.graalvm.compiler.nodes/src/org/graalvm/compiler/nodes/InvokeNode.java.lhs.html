 60 /**
 61  * The {@code InvokeNode} represents all kinds of method calls.
 62  */
 63 // @formatter:off
 64 @NodeInfo(nameTemplate = "Invoke#{p#targetMethod/s}",
 65           allowedUsageTypes = {Memory},
 66           cycles = CYCLES_UNKNOWN,
 67           cyclesRationale = "We cannot estimate the runtime cost of a call, it is a blackhole." +
 68                             "However, we can estimate, dynamically, the cost of the call operation itself based on the type of the call.",
 69           size = SIZE_UNKNOWN,
 70           sizeRationale = "We can only dynamically, based on the type of the call (special, static, virtual, interface) decide" +
 71                           "how much code is generated for the call.")
 72 // @formatter:on
 73 public final class InvokeNode extends AbstractMemoryCheckpoint implements Invoke, LIRLowerable, MemoryCheckpoint.Single, UncheckedInterfaceProvider {
 74     public static final NodeClass<InvokeNode> TYPE = NodeClass.create(InvokeNode.class);
 75 
 76     @OptionalInput ValueNode classInit;
 77     @Input(Extension) CallTargetNode callTarget;
 78     @OptionalInput(State) FrameState stateDuring;
 79     protected final int bci;
 80     protected boolean polymorphic;
 81     protected boolean useForInlining;
 82     protected final LocationIdentity identity;
 83 
 84     public InvokeNode(CallTargetNode callTarget, int bci) {
 85         this(callTarget, bci, callTarget.returnStamp().getTrustedStamp());
 86     }
 87 
 88     public InvokeNode(CallTargetNode callTarget, int bci, LocationIdentity identity) {
 89         this(callTarget, bci, callTarget.returnStamp().getTrustedStamp(), identity);
 90     }
 91 
 92     public InvokeNode(CallTargetNode callTarget, int bci, Stamp stamp) {
 93         this(callTarget, bci, stamp, LocationIdentity.any());
 94     }
 95 
 96     public InvokeNode(CallTargetNode callTarget, int bci, Stamp stamp, LocationIdentity identity) {
 97         super(TYPE, stamp);
 98         this.callTarget = callTarget;
 99         this.bci = bci;
100         this.polymorphic = false;
101         this.useForInlining = true;
102         this.identity = identity;
103     }
104 
105     public InvokeNode replaceWithNewBci(int newBci) {
106         InvokeNode newInvoke = graph().add(new InvokeNode(callTarget, newBci, stamp, identity));
107         newInvoke.setUseForInlining(useForInlining);
108         newInvoke.setPolymorphic(polymorphic);
109         newInvoke.setStateAfter(stateAfter);
110         newInvoke.setStateDuring(stateDuring);
111         newInvoke.setClassInit(classInit);
112         graph().replaceFixedWithFixed(this, newInvoke);
113         return newInvoke;




114     }
115 
116     @Override
117     protected void afterClone(Node other) {
118         updateInliningLogAfterClone(other);
119     }
120 
121     @Override
122     public FixedNode asFixedNode() {
123         return this;
124     }
125 
126     @Override
127     public CallTargetNode callTarget() {
128         return callTarget;
129     }
130 
131     void setCallTarget(CallTargetNode callTarget) {
132         updateUsages(this.callTarget, callTarget);
133         this.callTarget = callTarget;
134     }
135 
136     @Override
137     public boolean isPolymorphic() {
138         return polymorphic;
139     }
140 
141     @Override
142     public void setPolymorphic(boolean value) {
143         this.polymorphic = value;
144     }
145 
146     @Override
147     public boolean useForInlining() {
148         return useForInlining;
149     }
150 
151     @Override
152     public void setUseForInlining(boolean value) {
153         this.useForInlining = value;
154     }
155 
156     @Override
157     public boolean isAllowedUsageType(InputType type) {
158         if (!super.isAllowedUsageType(type)) {
159             if (getStackKind() != JavaKind.Void) {
160                 if (callTarget instanceof MethodCallTargetNode && ((MethodCallTargetNode) callTarget).targetMethod().getAnnotation(NodeIntrinsic.class) != null) {
161                     return true;
162                 }
163             }
164             return false;
165         }
166         return true;
167     }
168 
169     @Override
170     public Map<Object, Object> getDebugProperties(Map<Object, Object> map) {
171         Map<Object, Object> debugProperties = super.getDebugProperties(map);
172         if (callTarget != null) {
173             debugProperties.put("targetMethod", callTarget.targetName());
174         }
175         return debugProperties;
176     }
177 
178     @Override
179     public LocationIdentity getLocationIdentity() {
180         return identity;
181     }
182 
183     @Override
184     public void lower(LoweringTool tool) {
185         tool.getLowerer().lower(this, tool);
186     }
187 
188     @Override
189     public void generate(NodeLIRBuilderTool gen) {
190         gen.emitInvoke(this);
191     }
192 
193     @Override
194     public String toString(Verbosity verbosity) {
195         if (verbosity == Verbosity.Long) {
196             return super.toString(Verbosity.Short) + "(bci=" + bci() + ")";
197         } else if (verbosity == Verbosity.Name) {
198             return "Invoke#" + (callTarget == null ? "null" : callTarget().targetName());
199         } else {
200             return super.toString(verbosity);
201         }
202     }
203 
204     @Override
205     public int bci() {
206         return bci;
207     }
208 
209     @Override
210     public boolean canDeoptimize() {
211         return true;
212     }
213 
214     @Override
215     public FrameState stateDuring() {
216         return stateDuring;
217     }
218 
219     @Override
220     public void setStateDuring(FrameState stateDuring) {
221         updateUsages(this.stateDuring, stateDuring);
222         this.stateDuring = stateDuring;
223     }
224 
225     @Override
226     public Stamp uncheckedStamp() {
227         return this.callTarget.returnStamp().getUncheckedStamp();
228     }
229 
230     @Override
231     public void setClassInit(ValueNode classInit) {
232         this.classInit = classInit;
233         updateUsages(null, classInit);
234     }
235 
236     @Override
237     public ValueNode classInit() {
238         return classInit;
239     }
240 
241     @Override
242     public NodeCycles estimatedNodeCycles() {
243         if (callTarget == null) {
244             return CYCLES_UNKNOWN;
245         }
246         switch (callTarget.invokeKind()) {
247             case Interface:
248                 return CYCLES_64;
249             case Special:
250             case Static:
251                 return CYCLES_2;
252             case Virtual:
253                 return CYCLES_8;
254             default:
255                 return CYCLES_UNKNOWN;
256         }
257     }
258 
259     @Override
260     public NodeSize estimatedNodeSize() {
261         if (callTarget == null) {
262             return SIZE_UNKNOWN;
263         }
264         switch (callTarget.invokeKind()) {
265             case Interface:
266                 return SIZE_64;
267             case Special:
268             case Static:
269                 return SIZE_2;
270             case Virtual:
271                 return SIZE_8;
272             default:
273                 return SIZE_UNKNOWN;
274         }
275     }
276 }
