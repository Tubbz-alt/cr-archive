25 package org.graalvm.compiler.nodes.graphbuilderconf;
26 
27 import org.graalvm.compiler.core.common.spi.ConstantFieldProvider;
28 import org.graalvm.compiler.debug.DebugContext;
29 import org.graalvm.compiler.nodes.StructuredGraph;
30 import org.graalvm.compiler.nodes.ValueNode;

31 import org.graalvm.compiler.nodes.spi.StampProvider;
32 import org.graalvm.compiler.options.OptionValues;
33 
34 import jdk.vm.ci.meta.Assumptions;
35 import jdk.vm.ci.meta.ConstantReflectionProvider;
36 import jdk.vm.ci.meta.MetaAccessProvider;
37 
38 /**
39  * Used by a {@link GraphBuilderPlugin} to interface with an object that builds a graph.
40  */
41 public interface GraphBuilderTool {
42 
43     /**
44      * Adds the given node to the graph and also adds recursively all referenced inputs.
45      *
46      * @param value the node to be added to the graph
47      * @return either the node added or an equivalent node
48      */
49     <T extends ValueNode> T append(T value);
50 
51     StampProvider getStampProvider();
52 
53     MetaAccessProvider getMetaAccess();
54 


55     default Assumptions getAssumptions() {
56         return getGraph().getAssumptions();
57     }
58 
59     ConstantReflectionProvider getConstantReflection();
60 
61     ConstantFieldProvider getConstantFieldProvider();
62 
63     /**
64      * Gets the graph being constructed.
65      */
66     StructuredGraph getGraph();
67 
68     default OptionValues getOptions() {
69         return getGraph().getOptions();
70     }
71 
72     default DebugContext getDebug() {
73         return getGraph().getDebug();
74     }
75 
76     /**
77      * Determines if this parsing context is within the bytecode of an intrinsic or a method inlined
78      * by an intrinsic.
79      */
80     boolean parsingIntrinsic();
81 
82     @SuppressWarnings("unused")
83     default boolean canDeferPlugin(GeneratedInvocationPlugin plugin) {
84         // By default generated plugins must be completely processed during parsing.
85         return false;
86     }
87 }
