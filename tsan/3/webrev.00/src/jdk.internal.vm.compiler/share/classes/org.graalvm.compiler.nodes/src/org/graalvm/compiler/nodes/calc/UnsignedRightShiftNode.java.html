 44 import jdk.vm.ci.meta.JavaKind;

 46 @NodeInfo(shortName = ">>>")
 47 public final class UnsignedRightShiftNode extends ShiftNode<UShr> {
 48 
 49     public static final NodeClass<UnsignedRightShiftNode> TYPE = NodeClass.create(UnsignedRightShiftNode.class);
 50 
 51     public UnsignedRightShiftNode(ValueNode x, ValueNode y) {
 52         super(TYPE, getArithmeticOpTable(x).getUShr(), x, y);
 53     }
 54 
 55     public static ValueNode create(ValueNode x, ValueNode y, NodeView view) {
 56         ArithmeticOpTable.ShiftOp<UShr> op = ArithmeticOpTable.forStamp(x.stamp(view)).getUShr();
 57         Stamp stamp = op.foldStamp(x.stamp(view), (IntegerStamp) y.stamp(view));
 58         ValueNode value = ShiftNode.canonical(op, stamp, x, y, view);
 59         if (value != null) {
 60             return value;
 61         }
 62 
 63         return canonical(null, op, stamp, x, y, view);
 64     }
 65 
 66     @Override
 67     protected ShiftOp<UShr> getOp(ArithmeticOpTable table) {
 68         return table.getUShr();
 69     }
 70 
 71     @Override
 72     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 73         NodeView view = NodeView.from(tool);
 74         ValueNode ret = super.canonical(tool, forX, forY);
 75         if (ret != this) {
 76             return ret;
 77         }
 78 
 79         return canonical(this, this.getArithmeticOp(), this.stamp(view), forX, forY, view);
 80     }
 81 
 82     @SuppressWarnings("unused")
 83     private static ValueNode canonical(UnsignedRightShiftNode node, ArithmeticOpTable.ShiftOp<UShr> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
 84         if (forY.isConstant()) {
 85             int amount = forY.asJavaConstant().asInt();
 86             int originalAmout = amount;
 87             int mask = op.getShiftAmountMask(stamp);
 88             amount &= mask;
 89             if (amount == 0) {
 90                 return forX;
 91             }
 92 
 93             Stamp xStampGeneric = forX.stamp(view);
 94             if (xStampGeneric instanceof IntegerStamp) {
 95                 IntegerStamp xStamp = (IntegerStamp) xStampGeneric;
 96                 long xMask = CodeUtil.mask(xStamp.getBits());
 97                 long xLowerBound = xStamp.lowerBound() & xMask;
 98                 long xUpperBound = xStamp.upperBound() & xMask;
 99 
100                 if (xLowerBound >>> amount == xUpperBound >>> amount) {
101                     // The result of the shift is constant.
102                     return ConstantNode.forIntegerKind(stamp.getStackKind(), xLowerBound >>> amount);
103                 }
104 
105                 if (amount == xStamp.getBits() - 1 && xStamp.lowerBound() == -1 && xStamp.upperBound() == 0) {
106                     // Shift is equivalent to a negation, i.e., turns -1 into 1 and keeps 0 at 0.
107                     return NegateNode.create(forX, view);
108                 }
109             }
110 
111             if (forX instanceof ShiftNode) {
112                 ShiftNode<?> other = (ShiftNode<?>) forX;
113                 if (other.getY().isConstant()) {
114                     int otherAmount = other.getY().asJavaConstant().asInt() & mask;
115                     if (other instanceof UnsignedRightShiftNode) {
116                         int total = amount + otherAmount;
117                         if (total != (total & mask)) {
118                             return ConstantNode.forIntegerKind(stamp.getStackKind(), 0);
119                         }
120                         return new UnsignedRightShiftNode(other.getX(), ConstantNode.forInt(total));
121                     } else if (other instanceof LeftShiftNode && otherAmount == amount) {
122                         if (stamp.getStackKind() == JavaKind.Long) {
123                             return new AndNode(other.getX(), ConstantNode.forLong(-1L >>> amount));
124                         } else {
125                             assert stamp.getStackKind() == JavaKind.Int;
126                             return new AndNode(other.getX(), ConstantNode.forInt(-1 >>> amount));
127                         }
128                     }
129                 }
130             }
131             if (originalAmout != amount) {
132                 return new UnsignedRightShiftNode(forX, ConstantNode.forInt(amount));
133             }
134         }
135 
136         if (node != null) {
137             return node;
138         }
139         return new UnsignedRightShiftNode(forX, forY);
140     }
141 
142     @Override
143     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
144         nodeValueMap.setResult(this, gen.emitUShr(nodeValueMap.operand(getX()), nodeValueMap.operand(getY())));
145     }
146 
147     @Override
148     public boolean isNarrowable(int resultBits) {
149         if (super.isNarrowable(resultBits)) {
150             /*
151              * For unsigned right shifts, the narrow can be done before the shift if the cut off
152              * bits are all zero.
153              */
154             IntegerStamp inputStamp = (IntegerStamp) getX().stamp(NodeView.DEFAULT);
155             return (inputStamp.upMask() & ~(resultBits - 1)) == 0;
156         } else {
157             return false;
158         }
159     }
