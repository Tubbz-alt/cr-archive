25 package org.graalvm.compiler.nodes;

27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_1;
28 
29 import org.graalvm.compiler.core.common.type.StampFactory;
30 import org.graalvm.compiler.graph.NodeClass;
31 import org.graalvm.compiler.nodeinfo.NodeInfo;
32 import org.graalvm.compiler.nodeinfo.NodeSize;
33 import org.graalvm.compiler.nodes.calc.SignExtendNode;
34 import org.graalvm.compiler.nodes.debug.ControlFlowAnchored;
35 import org.graalvm.compiler.nodes.spi.Lowerable;
36 import org.graalvm.compiler.nodes.spi.LoweringTool;
37 
38 import jdk.vm.ci.meta.JavaKind;
39 
40 /**
41  * A high-level intrinsic for getting an address inside of an object. During lowering it will be
42  * moved next to any uses to avoid creating a derived pointer that is live across a safepoint.
43  */
44 @NodeInfo(cycles = CYCLES_1, size = NodeSize.SIZE_1)
45 public final class ComputeObjectAddressNode extends FixedWithNextNode implements Lowerable, ControlFlowAnchored {
46     public static final NodeClass<ComputeObjectAddressNode> TYPE = NodeClass.create(ComputeObjectAddressNode.class);
47 
48     @Input ValueNode object;
49     @Input ValueNode offset;
50 
51     public ComputeObjectAddressNode(ValueNode obj, ValueNode offset) {
52         super(TYPE, StampFactory.forKind(JavaKind.Long));
53         this.object = obj;
54         this.offset = SignExtendNode.create(offset, 64, NodeView.DEFAULT);
55     }
56 
57     @NodeIntrinsic
58     public static native long get(Object array, long offset);
59 
60     @Override
61     public void lower(LoweringTool tool) {
62         tool.getLowerer().lower(this, tool);
63     }
64 
65     public ValueNode getObject() {
66         return object;
67     }
68 
69     public ValueNode getOffset() {
70         return offset;
71     }
72 }
