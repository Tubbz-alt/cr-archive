25 package org.graalvm.compiler.nodes;
26 
27 import static org.graalvm.compiler.nodeinfo.InputType.Memory;
28 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_0;
29 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_0;
30 
31 import org.graalvm.compiler.graph.NodeClass;
32 import org.graalvm.compiler.nodeinfo.NodeInfo;
33 import org.graalvm.compiler.nodes.memory.MemoryCheckpoint;
34 import jdk.internal.vm.compiler.word.LocationIdentity;
35 
36 /**
37  * The start node of a graph.
38  */
39 @NodeInfo(allowedUsageTypes = {Memory}, nameTemplate = "Start", cycles = CYCLES_0, size = SIZE_0)
40 public class StartNode extends BeginStateSplitNode implements MemoryCheckpoint.Single {
41     public static final NodeClass<StartNode> TYPE = NodeClass.create(StartNode.class);
42 
43     protected StartNode(NodeClass<? extends StartNode> c) {
44         super(c);
45     }
46 
47     public StartNode() {
48         super(TYPE);
49     }
50 
51     @Override
52     public LocationIdentity getKilledLocationIdentity() {
53         return LocationIdentity.any();
54     }
55 }
