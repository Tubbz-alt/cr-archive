41 /**
42  * Absolute value.
43  */
44 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
45 public final class AbsNode extends UnaryArithmeticNode<Abs> implements ArithmeticLIRLowerable, NarrowableArithmeticNode {
46     public static final NodeClass<AbsNode> TYPE = NodeClass.create(AbsNode.class);
47 
48     public AbsNode(ValueNode x) {
49         super(TYPE, ArithmeticOpTable::getAbs, x);
50     }
51 
52     public static ValueNode create(ValueNode value, NodeView view) {
53         ValueNode synonym = findSynonym(value, view);
54         if (synonym != null) {
55             return synonym;
56         }
57         return new NegateNode(value);
58     }
59 
60     protected static ValueNode findSynonym(ValueNode forValue, NodeView view) {
61         ArithmeticOpTable.UnaryOp<Abs> absOp = ArithmeticOpTable.forStamp(forValue.stamp(view)).getAbs();
62         ValueNode synonym = UnaryArithmeticNode.findSynonym(forValue, absOp);
63         if (synonym != null) {
64             return synonym;
65         }
66         return null;
67     }
68 





69     @Override
70     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
71         ValueNode ret = super.canonical(tool, forValue);
72         if (ret != this) {
73             return ret;
74         }
75         if (forValue instanceof AbsNode) {
76             return forValue;
77         }
78         return this;
79     }
80 
81     @Override
82     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
83         nodeValueMap.setResult(this, gen.emitMathAbs(nodeValueMap.operand(getValue())));
84     }
