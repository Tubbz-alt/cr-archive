 49 @NodeInfo(shortName = "*", cycles = CYCLES_2)
 50 public class MulNode extends BinaryArithmeticNode<Mul> implements NarrowableArithmeticNode, BinaryCommutative<ValueNode> {
 51 
 52     public static final NodeClass<MulNode> TYPE = NodeClass.create(MulNode.class);
 53 
 54     public MulNode(ValueNode x, ValueNode y) {
 55         this(TYPE, x, y);
 56     }
 57 
 58     protected MulNode(NodeClass<? extends MulNode> c, ValueNode x, ValueNode y) {
 59         super(c, getArithmeticOpTable(x).getMul(), x, y);
 60     }
 61 
 62     public static ValueNode create(ValueNode x, ValueNode y, NodeView view) {
 63         BinaryOp<Mul> op = ArithmeticOpTable.forStamp(x.stamp(view)).getMul();
 64         Stamp stamp = op.foldStamp(x.stamp(view), y.stamp(view));
 65         ConstantNode tryConstantFold = tryConstantFold(op, x, y, stamp, view);
 66         if (tryConstantFold != null) {
 67             return tryConstantFold;
 68         }
 69         return canonical(null, op, stamp, x, y, view);
 70     }
 71 
 72     @Override
 73     protected BinaryOp<Mul> getOp(ArithmeticOpTable table) {
 74         return table.getMul();
 75     }
 76 
 77     @Override
 78     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 79         ValueNode ret = super.canonical(tool, forX, forY);
 80         if (ret != this) {
 81             return ret;
 82         }
 83 
 84         if (forX.isConstant() && !forY.isConstant()) {
 85             // we try to swap and canonicalize
 86             ValueNode improvement = canonical(tool, forY, forX);
 87             if (improvement != this) {
 88                 return improvement;
 89             }
 90             // if this fails we only swap
 91             return new MulNode(forY, forX);
 92         }
 93         BinaryOp<Mul> op = getOp(forX, forY);
 94         NodeView view = NodeView.from(tool);
 95         return canonical(this, op, stamp(view), forX, forY, view);
 96     }
 97 
 98     private static ValueNode canonical(MulNode self, BinaryOp<Mul> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
 99         if (forY.isConstant()) {
100             Constant c = forY.asConstant();
101             if (op.isNeutral(c)) {
102                 return forX;
103             }
104 
105             if (c instanceof PrimitiveConstant && ((PrimitiveConstant) c).getJavaKind().isNumericInteger()) {
106                 long i = ((PrimitiveConstant) c).asLong();
107                 ValueNode result = canonical(stamp, forX, i, view);
108                 if (result != null) {
109                     return result;
110                 }
111             }
112 
113             if (op.isAssociative()) {
114                 // canonicalize expressions like "(a * 1) * 2"
115                 return reassociate(self != null ? self : (MulNode) new MulNode(forX, forY).maybeCommuteInputs(), ValueNode.isConstantPredicate(), forX, forY, view);
116             }
117         }
118         return self != null ? self : new MulNode(forX, forY).maybeCommuteInputs();
119     }
120 
121     public static ValueNode canonical(Stamp stamp, ValueNode forX, long i, NodeView view) {
122         if (i == 0) {
123             return ConstantNode.forIntegerStamp(stamp, 0);
124         } else if (i == 1) {
125             return forX;
126         } else if (i == -1) {
127             return NegateNode.create(forX, view);
128         } else if (i > 0) {
129             if (CodeUtil.isPowerOf2(i)) {
130                 return new LeftShiftNode(forX, ConstantNode.forInt(CodeUtil.log2(i)));
131             } else if (CodeUtil.isPowerOf2(i - 1)) {
132                 return AddNode.create(new LeftShiftNode(forX, ConstantNode.forInt(CodeUtil.log2(i - 1))), forX, view);
133             } else if (CodeUtil.isPowerOf2(i + 1)) {
134                 return SubNode.create(new LeftShiftNode(forX, ConstantNode.forInt(CodeUtil.log2(i + 1))), forX, view);
135             } else {
136                 int bitCount = Long.bitCount(i);
137                 long highestBitValue = Long.highestOneBit(i);
138                 if (bitCount == 2) {
139                     // e.g., 0b1000_0010
140                     long lowerBitValue = i - highestBitValue;
141                     assert highestBitValue > 0 && lowerBitValue > 0;
142                     ValueNode left = new LeftShiftNode(forX, ConstantNode.forInt(CodeUtil.log2(highestBitValue)));
143                     ValueNode right = lowerBitValue == 1 ? forX : new LeftShiftNode(forX, ConstantNode.forInt(CodeUtil.log2(lowerBitValue)));
144                     return AddNode.create(left, right, view);
145                 } else {
146                     // e.g., 0b1111_1101
147                     int shiftToRoundUpToPowerOf2 = CodeUtil.log2(highestBitValue) + 1;
148                     long subValue = (1 << shiftToRoundUpToPowerOf2) - i;
149                     if (CodeUtil.isPowerOf2(subValue) && shiftToRoundUpToPowerOf2 < ((IntegerStamp) stamp).getBits()) {
150                         assert CodeUtil.log2(subValue) >= 1;
151                         ValueNode left = new LeftShiftNode(forX, ConstantNode.forInt(shiftToRoundUpToPowerOf2));
152                         ValueNode right = new LeftShiftNode(forX, ConstantNode.forInt(CodeUtil.log2(subValue)));
153                         return SubNode.create(left, right, view);
154                     }
155                 }
156             }
157         } else if (i < 0) {
158             if (CodeUtil.isPowerOf2(-i)) {
159                 return NegateNode.create(LeftShiftNode.create(forX, ConstantNode.forInt(CodeUtil.log2(-i)), view), view);
160             }
161         }
162         return null;
163     }
164 
165     @Override
166     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
167         Value op1 = nodeValueMap.operand(getX());
168         Value op2 = nodeValueMap.operand(getY());
169         if (shouldSwapInputs(nodeValueMap)) {
170             Value tmp = op1;
171             op1 = op2;
172             op2 = tmp;
173         }
174         nodeValueMap.setResult(this, gen.emitMul(op1, op2, false));
175     }
176 }
