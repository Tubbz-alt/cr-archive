 59 /**
 60  * Node for a {@linkplain ForeignCallDescriptor foreign} call.
 61  */
 62 // @formatter:off
 63 @NodeInfo(nameTemplate = "ForeignCall#{p#descriptor/s}",
 64           allowedUsageTypes = Memory,
 65           cycles = CYCLES_2,
 66           cyclesRationale = "Rough estimation of the call operation itself.",
 67           size = SIZE_2,
 68           sizeRationale = "Rough estimation of the call operation itself.")
 69 // @formatter:on
 70 public class ForeignCallNode extends AbstractMemoryCheckpoint implements LIRLowerable, DeoptimizingNode.DeoptDuring, MemoryCheckpoint.Multi {
 71     public static final NodeClass<ForeignCallNode> TYPE = NodeClass.create(ForeignCallNode.class);
 72 
 73     @Input protected NodeInputList<ValueNode> arguments;
 74     @OptionalInput(State) protected FrameState stateDuring;
 75     protected final ForeignCallsProvider foreignCalls;
 76 
 77     protected final ForeignCallDescriptor descriptor;
 78     protected int bci = BytecodeFrame.UNKNOWN_BCI;
 79 
 80     public static boolean intrinsify(GraphBuilderContext b, ResolvedJavaMethod targetMethod, @InjectedNodeParameter Stamp returnStamp, @InjectedNodeParameter ForeignCallsProvider foreignCalls,
 81                     ForeignCallDescriptor descriptor, ValueNode... arguments) {
 82         ForeignCallNode node = new ForeignCallNode(foreignCalls, descriptor, arguments);
 83         node.setStamp(returnStamp);
 84 
 85         assert verifyDescriptor(b, targetMethod, descriptor);
 86 
 87         /*
 88          * Need to update the BCI of a ForeignCallNode so that it gets the stateDuring in the case
 89          * that the foreign call can deoptimize. As with all deoptimization, we need a state in a
 90          * non-intrinsic method.
 91          */
 92         GraphBuilderContext nonIntrinsicAncestor = b.getNonIntrinsicAncestor();
 93         if (nonIntrinsicAncestor != null) {
 94             node.setBci(nonIntrinsicAncestor.bci());
 95         }
 96 
 97         JavaKind returnKind = targetMethod.getSignature().getReturnKind();
 98         if (returnKind == JavaKind.Void) {
 99             b.add(node);
100         } else {
101             b.addPush(returnKind, node);
102         }
103 
104         return true;
105     }
106 
107     static boolean verifyDescriptor(GraphBuilderContext b, ResolvedJavaMethod targetMethod, ForeignCallDescriptor descriptor) {
108         int parameters = 1;
109         for (Class<?> arg : descriptor.getArgumentTypes()) {
110             ResolvedJavaType res = b.getMetaAccess().lookupJavaType(arg);
111             ResolvedJavaType parameterType = (ResolvedJavaType) targetMethod.getSignature().getParameterType(parameters, targetMethod.getDeclaringClass());
112             assert parameterType.equals(res) : descriptor + ": parameter " + parameters + " mismatch: " + res + " != " + parameterType;
113             parameters++;
114         }
115         return true;
116     }

118     public ForeignCallNode(ForeignCallsProvider foreignCalls, ForeignCallDescriptor descriptor, ValueNode... arguments) {
119         this(TYPE, foreignCalls, descriptor, arguments);
120     }
121 
122     public ForeignCallNode(ForeignCallsProvider foreignCalls, ForeignCallDescriptor descriptor, Stamp stamp, List<ValueNode> arguments) {
123         super(TYPE, stamp);
124         this.arguments = new NodeInputList<>(this, arguments);
125         this.descriptor = descriptor;
126         this.foreignCalls = foreignCalls;
127         assert descriptor.getArgumentTypes().length == this.arguments.size() : "wrong number of arguments to " + this;
128     }
129 
130     public ForeignCallNode(ForeignCallsProvider foreignCalls, ForeignCallDescriptor descriptor, Stamp stamp) {
131         super(TYPE, stamp);
132         this.arguments = new NodeInputList<>(this);
133         this.descriptor = descriptor;
134         this.foreignCalls = foreignCalls;
135     }
136 
137     protected ForeignCallNode(NodeClass<? extends ForeignCallNode> c, ForeignCallsProvider foreignCalls, ForeignCallDescriptor descriptor, ValueNode... arguments) {
138         super(c, StampFactory.forKind(JavaKind.fromJavaClass(descriptor.getResultType())));
139         this.arguments = new NodeInputList<>(this, arguments);
140         this.descriptor = descriptor;
141         this.foreignCalls = foreignCalls;
142         assert descriptor.getArgumentTypes().length == this.arguments.size() : "wrong number of arguments to " + this;
143     }
144 
145     @Override
146     public boolean hasSideEffect() {
147         return !foreignCalls.isReexecutable(descriptor);
148     }
149 
150     public ForeignCallDescriptor getDescriptor() {
151         return descriptor;
152     }
153 
154     @Override
155     public LocationIdentity[] getLocationIdentities() {
156         return foreignCalls.getKilledLocations(descriptor);
157     }
158 
159     protected Value[] operands(NodeLIRBuilderTool gen) {
160         Value[] operands = new Value[arguments.size()];
161         for (int i = 0; i < operands.length; i++) {
162             operands[i] = gen.operand(arguments.get(i));
163         }
164         return operands;
165     }
166 
167     @Override
168     public void generate(NodeLIRBuilderTool gen) {
169         ForeignCallLinkage linkage = gen.getLIRGeneratorTool().getForeignCalls().lookupForeignCall(descriptor);
170         Value[] operands = operands(gen);
171         Value result = gen.getLIRGeneratorTool().emitForeignCall(linkage, gen.state(this), operands);
172         if (result != null) {
173             gen.setResult(this, result);
174         }
175     }
176 
177     @Override
178     public void setStateAfter(FrameState x) {
179         assert hasSideEffect() || x == null;
180         super.setStateAfter(x);
181     }
182 
183     @Override
184     public FrameState stateDuring() {
185         return stateDuring;
186     }
187 
188     @Override
189     public void setStateDuring(FrameState stateDuring) {
190         updateUsages(this.stateDuring, stateDuring);
191         this.stateDuring = stateDuring;
192     }
193 
194     public int getBci() {
195         return bci;
196     }
197 
198     /**
199      * Set the {@code bci} of the invoke bytecode for use when converting a stateAfter into a
200      * stateDuring.
201      */
202     public void setBci(int bci) {
203         assert this.bci == BytecodeFrame.UNKNOWN_BCI || this.bci == bci;
204         this.bci = bci;
205     }
206 
207     @Override
208     public void computeStateDuring(FrameState currentStateAfter) {
209         FrameState newStateDuring;
210         if ((currentStateAfter.stackSize() > 0 && currentStateAfter.stackAt(currentStateAfter.stackSize() - 1) == this) ||
211                         (currentStateAfter.stackSize() > 1 && currentStateAfter.stackAt(currentStateAfter.stackSize() - 2) == this)) {
212             // The result of this call is on the top of stack, so roll back to the previous bci.
213             assert bci != BytecodeFrame.UNKNOWN_BCI : this;
214             newStateDuring = currentStateAfter.duplicateModifiedDuringCall(bci, this.getStackKind());
215         } else {
216             newStateDuring = currentStateAfter;
217         }
218         setStateDuring(newStateDuring);
219     }
220 
221     @Override
222     public String toString(Verbosity verbosity) {
223         if (verbosity == Verbosity.Name) {
224             return super.toString(verbosity) + "#" + descriptor;
225         }
226         return super.toString(verbosity);
227     }
228 
229     @Override
230     public boolean canDeoptimize() {
231         return foreignCalls.canDeoptimize(descriptor);
232     }
233 
234     public boolean isGuaranteedSafepoint() {
235         return foreignCalls.isGuaranteedSafepoint(descriptor);
236     }
237 
238     public NodeInputList<ValueNode> getArguments() {
239         return arguments;
240     }
