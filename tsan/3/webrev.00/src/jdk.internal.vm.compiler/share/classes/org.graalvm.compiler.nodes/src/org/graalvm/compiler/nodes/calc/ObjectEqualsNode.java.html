 60 @NodeInfo(shortName = "==")
 61 public final class ObjectEqualsNode extends PointerEqualsNode implements Virtualizable {
 62 
 63     public static final NodeClass<ObjectEqualsNode> TYPE = NodeClass.create(ObjectEqualsNode.class);
 64     private static final ObjectEqualsOp OP = new ObjectEqualsOp();
 65 
 66     public ObjectEqualsNode(ValueNode x, ValueNode y) {
 67         super(TYPE, x, y);
 68         assert x.stamp(NodeView.DEFAULT) instanceof AbstractObjectStamp;
 69         assert y.stamp(NodeView.DEFAULT) instanceof AbstractObjectStamp;
 70     }
 71 
 72     public static LogicNode create(ValueNode x, ValueNode y, ConstantReflectionProvider constantReflection, NodeView view) {
 73         LogicNode result = CompareNode.tryConstantFold(CanonicalCondition.EQ, x, y, constantReflection, false);
 74         if (result != null) {
 75             return result;
 76         } else {
 77             result = findSynonym(x, y, view);
 78             if (result != null) {
 79                 return result;
 80             }
 81             return new ObjectEqualsNode(x, y);
 82         }
 83     }
 84 
 85     public static LogicNode create(ConstantReflectionProvider constantReflection, MetaAccessProvider metaAccess, OptionValues options, ValueNode x, ValueNode y, NodeView view) {
 86         LogicNode result = OP.canonical(constantReflection, metaAccess, options, null, CanonicalCondition.EQ, false, x, y, view);
 87         if (result != null) {
 88             return result;
 89         }
 90         return create(x, y, constantReflection, view);
 91     }
 92 
 93     @Override
 94     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 95         NodeView view = NodeView.from(tool);
 96         ValueNode value = OP.canonical(tool.getConstantReflection(), tool.getMetaAccess(), tool.getOptions(), tool.smallestCompareWidth(), CanonicalCondition.EQ, false, forX, forY, view);
 97         if (value != null) {
 98             return value;
 99         }
100         return this;
101     }
102 
103     public static class ObjectEqualsOp extends PointerEqualsOp {
104 
105         @Override
106         protected LogicNode canonicalizeSymmetricConstant(ConstantReflectionProvider constantReflection, MetaAccessProvider metaAccess, OptionValues options, Integer smallestCompareWidth,
107                         CanonicalCondition condition, Constant constant, ValueNode nonConstant, boolean mirrored, boolean unorderedIsTrue, NodeView view) {
108             ResolvedJavaType type = constantReflection.asJavaType(constant);
109             if (type != null && nonConstant instanceof GetClassNode) {
110                 GetClassNode getClassNode = (GetClassNode) nonConstant;
111                 ValueNode object = getClassNode.getObject();
112                 assert ((ObjectStamp) object.stamp(view)).nonNull();
113                 if (!type.isPrimitive() && (type.isConcrete() || type.isArray())) {
114                     return InstanceOfNode.create(TypeReference.createExactTrusted(type), object);
115                 }
116                 return LogicConstantNode.forBoolean(false);
117             }
118             if (nonConstant instanceof AbstractNewObjectNode || nonConstant instanceof AllocatedObjectNode) {
119                 assert !(nonConstant instanceof BoxNode); // guard against class hierarchy changes
120                 // a constant can never be equals to a new object
121                 return LogicConstantNode.forBoolean(false);
122             }
123             return super.canonicalizeSymmetricConstant(constantReflection, metaAccess, options, smallestCompareWidth, condition, constant, nonConstant, mirrored, unorderedIsTrue, view);
124         }
125 
126         @Override
127         protected CompareNode duplicateModified(ValueNode newX, ValueNode newY, boolean unorderedIsTrue, NodeView view) {
128             if (newX.stamp(view) instanceof ObjectStamp && newY.stamp(view) instanceof ObjectStamp) {
129                 return new ObjectEqualsNode(newX, newY);
130             } else if (newX.stamp(view) instanceof AbstractPointerStamp && newY.stamp(view) instanceof AbstractPointerStamp) {
131                 return new PointerEqualsNode(newX, newY);
132             }
133             throw GraalError.shouldNotReachHere();
134         }
135     }
136 
137     private static LogicNode virtualizeNonVirtualComparison(VirtualObjectNode virtual, ValueNode other, StructuredGraph graph, VirtualizerTool tool) {
138         if (virtual instanceof VirtualBoxingNode && other.isConstant()) {
139             VirtualBoxingNode virtualBoxingNode = (VirtualBoxingNode) virtual;
140             if (virtualBoxingNode.getBoxingKind() == JavaKind.Boolean) {
141                 JavaConstant otherUnboxed = tool.getConstantReflection().unboxPrimitive(other.asJavaConstant());
142                 if (otherUnboxed != null && otherUnboxed.getJavaKind() == JavaKind.Boolean) {
143                     int expectedValue = otherUnboxed.asBoolean() ? 1 : 0;
144                     return new IntegerEqualsNode(virtualBoxingNode.getBoxedValue(tool), ConstantNode.forInt(expectedValue, graph));
145                 } else {
146                     return LogicConstantNode.contradiction(graph);
147                 }
148             }
149         }
150         if (virtual.hasIdentity()) {
151             // one of them is virtual: they can never be the same objects
152             return LogicConstantNode.contradiction(graph);
153         }
154         return null;
155     }
156 
157     public static LogicNode virtualizeComparison(ValueNode x, ValueNode y, StructuredGraph graph, VirtualizerTool tool) {
158         ValueNode xAlias = tool.getAlias(x);
159         ValueNode yAlias = tool.getAlias(y);
160 
161         VirtualObjectNode xVirtual = xAlias instanceof VirtualObjectNode ? (VirtualObjectNode) xAlias : null;
162         VirtualObjectNode yVirtual = yAlias instanceof VirtualObjectNode ? (VirtualObjectNode) yAlias : null;
163 
164         if (xVirtual != null && yVirtual == null) {
165             return virtualizeNonVirtualComparison(xVirtual, yAlias, graph, tool);
166         } else if (xVirtual == null && yVirtual != null) {
167             return virtualizeNonVirtualComparison(yVirtual, xAlias, graph, tool);
168         } else if (xVirtual != null && yVirtual != null) {
169             if (xVirtual.hasIdentity() ^ yVirtual.hasIdentity()) {
170                 /*
171                  * One of the two objects has identity, the other doesn't. In code, this looks like
172                  * "Integer.valueOf(a) == new Integer(b)", which is always false.
173                  *
174                  * In other words: an object created via valueOf can never be equal to one created
175                  * by new in the same compilation unit.
176                  */
177                 return LogicConstantNode.contradiction(graph);
178             } else if (!xVirtual.hasIdentity() && !yVirtual.hasIdentity()) {
179                 ResolvedJavaType type = xVirtual.type();
180                 if (type.equals(yVirtual.type())) {
181                     MetaAccessProvider metaAccess = tool.getMetaAccess();
182                     if (type.equals(metaAccess.lookupJavaType(Integer.class)) || type.equals(metaAccess.lookupJavaType(Long.class))) {
183                         // both are virtual without identity: check contents
184                         assert xVirtual.entryCount() == 1 && yVirtual.entryCount() == 1;
185                         assert xVirtual.entryKind(0).getStackKind() == JavaKind.Int || xVirtual.entryKind(0) == JavaKind.Long;
186                         return new IntegerEqualsNode(tool.getEntry(xVirtual, 0), tool.getEntry(yVirtual, 0));
187                     }
188                 }
189             } else {
190                 // both are virtual with identity: check if they refer to the same object
191                 return LogicConstantNode.forBoolean(xVirtual == yVirtual, graph);
192             }
193         }
194         return null;
195     }
196 
197     @Override
198     public void virtualize(VirtualizerTool tool) {
199         ValueNode node = virtualizeComparison(getX(), getY(), graph(), tool);
200         if (node == null) {
201             return;
202         }
203         if (!node.isAlive()) {
204             tool.addNode(node);
205         }
206         tool.replaceWithValue(node);
207     }
