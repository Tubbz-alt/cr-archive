 48 /**
 49  * The {@code ShiftOp} class represents shift operations.
 50  */
 51 @NodeInfo(cycles = CYCLES_1, size = SIZE_1)
 52 public abstract class ShiftNode<OP> extends BinaryNode implements ArithmeticOperation, ArithmeticLIRLowerable, NarrowableArithmeticNode {
 53 
 54     @SuppressWarnings("rawtypes") public static final NodeClass<ShiftNode> TYPE = NodeClass.create(ShiftNode.class);
 55 





 56     /**
 57      * Creates a new shift operation.
 58      *
 59      * @param x the first input value
 60      * @param s the second input value
 61      */
 62     protected ShiftNode(NodeClass<? extends ShiftNode<OP>> c, ShiftOp<OP> opForStampComputation, ValueNode x, ValueNode s) {
 63         super(c, opForStampComputation.foldStamp(x.stamp(NodeView.DEFAULT), (IntegerStamp) s.stamp(NodeView.DEFAULT)), x, s);
 64         assert ((IntegerStamp) s.stamp(NodeView.DEFAULT)).getBits() == 32;

 65     }
 66 
 67     protected abstract ShiftOp<OP> getOp(ArithmeticOpTable table);
 68 
 69     protected final ShiftOp<OP> getOp(ValueNode forValue) {
 70         return getOp(getArithmeticOpTable(forValue));
 71     }
 72 
 73     @Override
 74     public final ShiftOp<OP> getArithmeticOp() {
 75         return getOp(getX());
 76     }
 77 
 78     @Override
 79     public Stamp foldStamp(Stamp stampX, Stamp stampY) {
 80         return getArithmeticOp().foldStamp(stampX, (IntegerStamp) stampY);
 81     }
 82 
 83     @Override
 84     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 85         NodeView view = NodeView.from(tool);
 86         ValueNode valueNode = canonical(getOp(forX), stamp(NodeView.DEFAULT), forX, forY, view);
 87         if (valueNode != null) {
 88             return valueNode;
 89         }
 90         return this;
 91     }
 92 
 93     @SuppressWarnings("unused")
 94     public static <OP> ValueNode canonical(ShiftOp<OP> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
 95         if (forX.isConstant() && forY.isConstant()) {
 96             JavaConstant amount = forY.asJavaConstant();
 97             assert amount.getJavaKind() == JavaKind.Int;
 98             return ConstantNode.forPrimitive(stamp, op.foldConstant(forX.asConstant(), amount.asInt()));
 99         }
100         return null;
101     }
102 
103     public int getShiftAmountMask() {
104         return getArithmeticOp().getShiftAmountMask(stamp(NodeView.DEFAULT));
105     }
106 
107     @Override
108     public boolean isNarrowable(int resultBits) {
109         assert CodeUtil.isPowerOf2(resultBits);
110         int narrowMask = resultBits - 1;
111         int wideMask = getShiftAmountMask();
112         assert (wideMask & narrowMask) == narrowMask : String.format("wideMask %x should be wider than narrowMask %x", wideMask, narrowMask);
113 
114         /*
115          * Shifts are special because narrowing them also changes the implicit mask of the shift
116          * amount. We can narrow only if (y & wideMask) == (y & narrowMask) for all possible values
117          * of y.
118          */
119         IntegerStamp yStamp = (IntegerStamp) getY().stamp(NodeView.DEFAULT);
120         return (yStamp.upMask() & (wideMask & ~narrowMask)) == 0;
121     }
122 }
