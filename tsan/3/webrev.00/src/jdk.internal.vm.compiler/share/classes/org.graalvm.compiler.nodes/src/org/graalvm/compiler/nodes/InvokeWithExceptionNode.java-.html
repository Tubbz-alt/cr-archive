 49 @NodeInfo(nameTemplate = "Invoke!#{p#targetMethod/s}", allowedUsageTypes = {Memory}, cycles = CYCLES_UNKNOWN, size = SIZE_UNKNOWN)
 50 public final class InvokeWithExceptionNode extends ControlSplitNode implements Invoke, MemoryCheckpoint.Single, LIRLowerable, UncheckedInterfaceProvider {
 51     public static final NodeClass<InvokeWithExceptionNode> TYPE = NodeClass.create(InvokeWithExceptionNode.class);
 52 
 53     private static final double EXCEPTION_PROBA = 1e-5;
 54 
 55     @Successor AbstractBeginNode next;
 56     @Successor AbstractBeginNode exceptionEdge;
 57     @OptionalInput ValueNode classInit;
 58     @Input(Extension) CallTargetNode callTarget;
 59     @OptionalInput(State) FrameState stateDuring;
 60     @OptionalInput(State) FrameState stateAfter;
 61     protected final int bci;
 62     protected boolean polymorphic;
 63     protected boolean useForInlining;
 64     protected double exceptionProbability;
 65 
 66     public InvokeWithExceptionNode(CallTargetNode callTarget, AbstractBeginNode exceptionEdge, int bci) {
 67         super(TYPE, callTarget.returnStamp().getTrustedStamp());
 68         this.exceptionEdge = exceptionEdge;
 69         this.bci = bci;
 70         this.callTarget = callTarget;
 71         this.polymorphic = false;
 72         this.useForInlining = true;
 73         this.exceptionProbability = EXCEPTION_PROBA;
 74     }
 75 
 76     @Override
 77     protected void afterClone(Node other) {
 78         updateInliningLogAfterClone(other);
 79     }
 80 
 81     @Override
 82     public FixedNode asFixedNode() {
 83         return this;
 84     }
 85 
 86     public AbstractBeginNode exceptionEdge() {
 87         return exceptionEdge;
 88     }
 89 
 90     public void setExceptionEdge(AbstractBeginNode x) {
 91         updatePredecessor(exceptionEdge, x);
 92         exceptionEdge = x;
 93     }
 94 
 95     @Override
 96     public AbstractBeginNode next() {
 97         return next;
 98     }
 99 
100     public void setNext(AbstractBeginNode x) {
101         updatePredecessor(next, x);
102         next = x;
103     }
104 
105     @Override
106     public CallTargetNode callTarget() {
107         return callTarget;
108     }
109 
110     void setCallTarget(CallTargetNode callTarget) {
111         updateUsages(this.callTarget, callTarget);
112         this.callTarget = callTarget;
113     }
114 
115     public MethodCallTargetNode methodCallTarget() {
116         return (MethodCallTargetNode) callTarget;
117     }
118 
119     @Override
120     public boolean isPolymorphic() {
121         return polymorphic;
122     }
123 
124     @Override
125     public void setPolymorphic(boolean value) {
126         this.polymorphic = value;
127     }
128 
129     @Override
130     public boolean useForInlining() {
131         return useForInlining;
132     }
133 
134     @Override
135     public void setUseForInlining(boolean value) {
136         this.useForInlining = value;
137     }
138 
139     @Override
140     public String toString(Verbosity verbosity) {
141         if (verbosity == Verbosity.Long) {
142             return super.toString(Verbosity.Short) + "(bci=" + bci() + ")";
143         } else if (verbosity == Verbosity.Name) {
144             return "Invoke#" + (callTarget == null ? "null" : callTarget().targetName());
145         } else {
146             return super.toString(verbosity);
147         }
148     }
149 
150     @Override
151     public int bci() {
152         return bci;
153     }
154 
155     @Override
156     public void setNext(FixedNode x) {
157         if (x != null) {
158             this.setNext(KillingBeginNode.begin(x, getLocationIdentity()));
159         } else {
160             this.setNext(null);
161         }
162     }
163 
164     @Override
165     public void lower(LoweringTool tool) {
166         tool.getLowerer().lower(this, tool);
167     }
168 
169     @Override
170     public void generate(NodeLIRBuilderTool gen) {
171         gen.emitInvoke(this);
172     }
173 
174     @Override
175     public FrameState stateAfter() {
176         return stateAfter;
177     }
178 
179     @Override
180     public void setStateAfter(FrameState stateAfter) {
181         updateUsages(this.stateAfter, stateAfter);
182         this.stateAfter = stateAfter;
183     }
184 
185     @Override
186     public boolean hasSideEffect() {
187         return true;
188     }
189 
190     @Override
191     public LocationIdentity getLocationIdentity() {
192         return LocationIdentity.any();
193     }
194 
195     @Override
196     public Map<Object, Object> getDebugProperties(Map<Object, Object> map) {
197         Map<Object, Object> debugProperties = super.getDebugProperties(map);
198         if (callTarget != null) {
199             debugProperties.put("targetMethod", callTarget.targetName());
200         }
201         return debugProperties;
202     }
203 
204     public void killExceptionEdge() {
205         AbstractBeginNode edge = exceptionEdge();
206         setExceptionEdge(null);
207         GraphUtil.killCFG(edge);
208     }
209 
210     public void replaceWithNewBci(int newBci) {
211         AbstractBeginNode nextNode = next();
212         AbstractBeginNode exceptionObject = exceptionEdge;
213         setExceptionEdge(null);
214         setNext(null);
215         InvokeWithExceptionNode repl = graph().add(new InvokeWithExceptionNode(callTarget(), exceptionObject, newBci));
216         repl.setStateAfter(stateAfter);
217         this.setStateAfter(null);
218         this.replaceAtPredecessor(repl);
219         repl.setNext(nextNode);
220         boolean removed = this.callTarget().removeUsage(this);
221         assert removed;
222         this.replaceAtUsages(repl);
223         this.markDeleted();
224     }
225 
226     @Override
227     public double probability(AbstractBeginNode successor) {
228         return successor == next ? 1 - exceptionProbability : exceptionProbability;
229     }
230 
231     @Override
232     public boolean canDeoptimize() {
233         return true;
234     }
235 
236     @Override
237     public FrameState stateDuring() {
238         return stateDuring;
239     }
240 
241     @Override
242     public void setStateDuring(FrameState stateDuring) {
243         updateUsages(this.stateDuring, stateDuring);
244         this.stateDuring = stateDuring;
245     }
246 
247     @Override
248     public AbstractBeginNode getPrimarySuccessor() {
249         return this.next();
250     }
251 
252     @Override
253     public Stamp uncheckedStamp() {
254         return this.callTarget.returnStamp().getUncheckedStamp();
255     }
256 
257     @Override
258     public void setClassInit(ValueNode classInit) {
259         this.classInit = classInit;
260         updateUsages(null, classInit);
261     }
262 
263     @Override
264     public ValueNode classInit() {
265         return classInit;
266     }
267 
268     @Override
269     public boolean setProbability(AbstractBeginNode successor, double value) {
270         // Cannot set probability for exception invokes.
271         return false;
272     }
273 
274     @Override
275     public int getSuccessorCount() {
276         return 2;
277     }
278 
279     /**
280      * Replaces this InvokeWithExceptionNode with a normal InvokeNode. Kills the exception dispatch
281      * code.
282      */
283     public InvokeNode replaceWithInvoke() {
284         InvokeNode newInvoke = graph().add(new InvokeNode(callTarget, bci, stamp, getLocationIdentity()));
285         newInvoke.setStateAfter(stateAfter);
286         newInvoke.setStateDuring(stateDuring);
287         AbstractBeginNode oldException = this.exceptionEdge;
288         graph().replaceSplitWithFixed(this, newInvoke, this.next());
289         GraphUtil.killCFG(oldException);
290         return newInvoke;
291     }
