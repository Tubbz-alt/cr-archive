 46 @NodeInfo(shortName = "-")
 47 public class SubNode extends BinaryArithmeticNode<Sub> implements NarrowableArithmeticNode {
 48 
 49     public static final NodeClass<SubNode> TYPE = NodeClass.create(SubNode.class);
 50 
 51     public SubNode(ValueNode x, ValueNode y) {
 52         this(TYPE, x, y);
 53     }
 54 
 55     protected SubNode(NodeClass<? extends SubNode> c, ValueNode x, ValueNode y) {
 56         super(c, ArithmeticOpTable::getSub, x, y);
 57     }
 58 
 59     public static ValueNode create(ValueNode x, ValueNode y, NodeView view) {
 60         BinaryOp<Sub> op = ArithmeticOpTable.forStamp(x.stamp(view)).getSub();
 61         Stamp stamp = op.foldStamp(x.stamp(view), y.stamp(view));
 62         ConstantNode tryConstantFold = tryConstantFold(op, x, y, stamp, view);
 63         if (tryConstantFold != null) {
 64             return tryConstantFold;
 65         }
 66         return canonical(null, op, stamp, x, y, view);
 67     }
 68 





 69     private static ValueNode canonical(SubNode subNode, BinaryOp<Sub> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
 70         SubNode self = subNode;
 71         if (GraphUtil.unproxify(forX) == GraphUtil.unproxify(forY)) {
 72             Constant zero = op.getZero(forX.stamp(view));
 73             if (zero != null) {
 74                 return ConstantNode.forPrimitive(stamp, zero);
 75             }
 76         }
 77         boolean associative = op.isAssociative();
 78         if (associative) {
 79             if (forX instanceof AddNode) {
 80                 AddNode x = (AddNode) forX;
 81                 if (x.getY() == forY) {
 82                     // (a + b) - b
 83                     return x.getX();
 84                 }
 85                 if (x.getX() == forY) {
 86                     // (a + b) - a
 87                     return x.getY();
 88                 }
 89             } else if (forX instanceof SubNode) {
 90                 SubNode x = (SubNode) forX;
 91                 if (x.getX() == forY) {
 92                     // (a - b) - a
 93                     return NegateNode.create(x.getY(), view);
 94                 }
 95             }
 96             if (forY instanceof AddNode) {
 97                 AddNode y = (AddNode) forY;
 98                 if (y.getX() == forX) {
 99                     // a - (a + b)
100                     return NegateNode.create(y.getY(), view);
101                 }
102                 if (y.getY() == forX) {
103                     // b - (a + b)
104                     return NegateNode.create(y.getX(), view);
105                 }
106             } else if (forY instanceof SubNode) {
107                 SubNode y = (SubNode) forY;
108                 if (y.getX() == forX) {
109                     // a - (a - b)
110                     return y.getY();
111                 }
112             }
113         }
114         if (forY.isConstant()) {
115             Constant c = forY.asConstant();
116             if (op.isNeutral(c)) {
117                 return forX;
118             }
119             if (associative && self != null) {
120                 ValueNode reassociated = reassociate(self, ValueNode.isConstantPredicate(), forX, forY, view);
121                 if (reassociated != self) {
122                     return reassociated;
123                 }
124             }
125             if (c instanceof PrimitiveConstant && ((PrimitiveConstant) c).getJavaKind().isNumericInteger()) {
126                 long i = ((PrimitiveConstant) c).asLong();
127                 if (i < 0 || ((IntegerStamp) StampFactory.forKind(forY.getStackKind())).contains(-i)) {
128                     // Adding a negative is more friendly to the backend since adds are
129                     // commutative, so prefer add when it fits.
130                     return BinaryArithmeticNode.add(forX, ConstantNode.forIntegerStamp(stamp, -i), view);
131                 }
132             }
133         } else if (forX.isConstant()) {
134             Constant c = forX.asConstant();
135             if (ArithmeticOpTable.forStamp(stamp).getAdd().isNeutral(c)) {
136                 /*
137                  * Note that for floating point numbers, + and - have different neutral elements. We
138                  * have to test for the neutral element of +, because we are doing this
139                  * transformation: 0 - x == (-x) + 0 == -x.
140                  */
141                 return NegateNode.create(forY, view);
142             }
143             if (associative && self != null) {
144                 return reassociate(self, ValueNode.isConstantPredicate(), forX, forY, view);
145             }
146         }
147         if (forY instanceof NegateNode) {
148             return BinaryArithmeticNode.add(forX, ((NegateNode) forY).getValue(), view);
149         }
150         return self != null ? self : new SubNode(forX, forY);
151     }
152 
153     @Override
154     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
155         NodeView view = NodeView.from(tool);
156         ValueNode ret = super.canonical(tool, forX, forY);
157         if (ret != this) {
158             return ret;
159         }
160 
161         BinaryOp<Sub> op = getOp(forX, forY);
162         return canonical(this, op, stamp, forX, forY, view);
163     }
164 
165     @Override
166     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
167         nodeValueMap.setResult(this, gen.emitSub(nodeValueMap.operand(getX()), nodeValueMap.operand(getY()), false));
168     }
