 46 @NodeInfo(shortName = "-")
 47 public class SubNode extends BinaryArithmeticNode<Sub> implements NarrowableArithmeticNode {
 48 
 49     public static final NodeClass<SubNode> TYPE = NodeClass.create(SubNode.class);
 50 
 51     public SubNode(ValueNode x, ValueNode y) {
 52         this(TYPE, x, y);
 53     }
 54 
 55     protected SubNode(NodeClass<? extends SubNode> c, ValueNode x, ValueNode y) {
 56         super(c, getArithmeticOpTable(x).getSub(), x, y);
 57     }
 58 
 59     public static ValueNode create(ValueNode x, ValueNode y, NodeView view) {
 60         BinaryOp<Sub> op = ArithmeticOpTable.forStamp(x.stamp(view)).getSub();
 61         Stamp stamp = op.foldStamp(x.stamp(view), y.stamp(view));
 62         ConstantNode tryConstantFold = tryConstantFold(op, x, y, stamp, view);
 63         if (tryConstantFold != null) {
 64             return tryConstantFold;
 65         }
 66         return canonical(null, op, stamp, x, y, view);
 67     }
 68 
 69     @Override
 70     protected BinaryOp<Sub> getOp(ArithmeticOpTable table) {
 71         return table.getSub();
 72     }
 73 
 74     private static ValueNode canonical(SubNode subNode, BinaryOp<Sub> op, Stamp stamp, ValueNode forX, ValueNode forY, NodeView view) {
 75         SubNode self = subNode;
 76         if (GraphUtil.unproxify(forX) == GraphUtil.unproxify(forY)) {
 77             Constant zero = op.getZero(forX.stamp(view));
 78             if (zero != null) {
 79                 return ConstantNode.forPrimitive(stamp, zero);
 80             }
 81         }
 82         boolean associative = op.isAssociative();
 83         if (associative) {
 84             if (forX instanceof AddNode) {
 85                 AddNode x = (AddNode) forX;
 86                 if (x.getY() == forY) {
 87                     // (a + b) - b
 88                     return x.getX();
 89                 }
 90                 if (x.getX() == forY) {
 91                     // (a + b) - a
 92                     return x.getY();
 93                 }
 94             } else if (forX instanceof SubNode) {
 95                 SubNode x = (SubNode) forX;
 96                 if (x.getX() == forY) {
 97                     // (a - b) - a
 98                     return NegateNode.create(x.getY(), view);
 99                 }
100             }
101             if (forY instanceof AddNode) {
102                 AddNode y = (AddNode) forY;
103                 if (y.getX() == forX) {
104                     // a - (a + b)
105                     return NegateNode.create(y.getY(), view);
106                 }
107                 if (y.getY() == forX) {
108                     // b - (a + b)
109                     return NegateNode.create(y.getX(), view);
110                 }
111             } else if (forY instanceof SubNode) {
112                 SubNode y = (SubNode) forY;
113                 if (y.getX() == forX) {
114                     // a - (a - b)
115                     return y.getY();
116                 }
117             }
118         }
119         if (forY.isConstant()) {
120             Constant c = forY.asConstant();
121             if (op.isNeutral(c)) {
122                 return forX;
123             }
124             if (associative && self != null) {
125                 ValueNode reassociated = reassociate(self, ValueNode.isConstantPredicate(), forX, forY, view);
126                 if (reassociated != self) {
127                     return reassociated;
128                 }
129             }
130             if (c instanceof PrimitiveConstant && ((PrimitiveConstant) c).getJavaKind().isNumericInteger()) {
131                 long i = ((PrimitiveConstant) c).asLong();
132                 if (i < 0 || ((IntegerStamp) StampFactory.forKind(forY.getStackKind())).contains(-i)) {
133                     // Adding a negative is more friendly to the backend since adds are
134                     // commutative, so prefer add when it fits.
135                     return BinaryArithmeticNode.add(forX, ConstantNode.forIntegerStamp(stamp, -i), view);
136                 }
137             }
138         } else if (forX.isConstant()) {
139             Constant c = forX.asConstant();
140             if (ArithmeticOpTable.forStamp(stamp).getAdd().isNeutral(c)) {
141                 /*
142                  * Note that for floating point numbers, + and - have different neutral elements. We
143                  * have to test for the neutral element of +, because we are doing this
144                  * transformation: 0 - x == (-x) + 0 == -x.
145                  */
146                 return NegateNode.create(forY, view);
147             }
148             if (associative && self != null) {
149                 return reassociate(self, ValueNode.isConstantPredicate(), forX, forY, view);
150             }
151         }
152         if (forY instanceof NegateNode) {
153             return BinaryArithmeticNode.add(forX, ((NegateNode) forY).getValue(), view);
154         }
155         return self != null ? self : new SubNode(forX, forY);
156     }
157 
158     @Override
159     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
160         NodeView view = NodeView.from(tool);
161         ValueNode ret = super.canonical(tool, forX, forY);
162         if (ret != this) {
163             return ret;
164         }
165 
166         BinaryOp<Sub> op = getOp(forX, forY);
167         return canonical(this, op, stamp, forX, forY, view);
168     }
169 
170     @Override
171     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
172         nodeValueMap.setResult(this, gen.emitSub(nodeValueMap.operand(getX()), nodeValueMap.operand(getY()), false));
173     }
174 }
