28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_1;

43 /**
44  * Absolute value.
45  */
46 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
47 public final class AbsNode extends UnaryArithmeticNode<Abs> implements ArithmeticLIRLowerable, NarrowableArithmeticNode {
48     public static final NodeClass<AbsNode> TYPE = NodeClass.create(AbsNode.class);
49 
50     public AbsNode(ValueNode x) {
51         super(TYPE, getArithmeticOpTable(x).getAbs(), x);
52     }
53 
54     public static ValueNode create(ValueNode value, NodeView view) {
55         ValueNode synonym = findSynonym(value, view);
56         if (synonym != null) {
57             return synonym;
58         }
59         return new NegateNode(value);
60     }
61 
62     protected static ValueNode findSynonym(ValueNode forValue, NodeView view) {
63         ArithmeticOpTable.UnaryOp<Abs> absOp = ArithmeticOpTable.forStamp(forValue.stamp(view)).getAbs();
64         ValueNode synonym = UnaryArithmeticNode.findSynonym(forValue, absOp);
65         if (synonym != null) {
66             return synonym;
67         }
68         return null;
69     }
70 
71     @Override
72     protected UnaryOp<Abs> getOp(ArithmeticOpTable table) {
73         return table.getAbs();
74     }
75 
76     @Override
77     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
78         ValueNode ret = super.canonical(tool, forValue);
79         if (ret != this) {
80             return ret;
81         }
82         if (forValue instanceof AbsNode) {
83             return forValue;
84         }
85         return this;
86     }
87 
88     @Override
89     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
90         nodeValueMap.setResult(this, gen.emitMathAbs(nodeValueMap.operand(getValue())));
91     }
92 }
