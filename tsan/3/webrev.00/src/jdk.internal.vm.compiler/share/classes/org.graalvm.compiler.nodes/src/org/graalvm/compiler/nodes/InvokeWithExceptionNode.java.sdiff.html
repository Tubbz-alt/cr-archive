210     public void replaceWithNewBci(int newBci) {
211         AbstractBeginNode nextNode = next();
212         AbstractBeginNode exceptionObject = exceptionEdge;
213         setExceptionEdge(null);
214         setNext(null);
215         InvokeWithExceptionNode repl = graph().add(new InvokeWithExceptionNode(callTarget(), exceptionObject, newBci));
216         repl.setStateAfter(stateAfter);
217         this.setStateAfter(null);
218         this.replaceAtPredecessor(repl);
219         repl.setNext(nextNode);
220         boolean removed = this.callTarget().removeUsage(this);
221         assert removed;
222         this.replaceAtUsages(repl);
223         this.markDeleted();
