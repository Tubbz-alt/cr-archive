25 package org.graalvm.compiler.nodes.calc;
26 
27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_2;
28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_2;
29 
30 import org.graalvm.compiler.core.common.type.ArithmeticOpTable;
31 import org.graalvm.compiler.core.common.type.ArithmeticOpTable.BinaryOp;
32 import org.graalvm.compiler.core.common.type.ArithmeticOpTable.BinaryOp.MulHigh;
33 import org.graalvm.compiler.graph.NodeClass;
34 import org.graalvm.compiler.graph.spi.Canonicalizable;
35 import org.graalvm.compiler.graph.spi.CanonicalizerTool;
36 import org.graalvm.compiler.lir.gen.ArithmeticLIRGeneratorTool;
37 import org.graalvm.compiler.nodeinfo.NodeInfo;
38 import org.graalvm.compiler.nodes.ConstantNode;
39 import org.graalvm.compiler.nodes.NodeView;
40 import org.graalvm.compiler.nodes.ValueNode;
41 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
42 
43 import jdk.vm.ci.meta.Constant;
44 import jdk.vm.ci.meta.PrimitiveConstant;
45 import jdk.vm.ci.meta.Value;
46 
47 @NodeInfo(shortName = "*H", cycles = CYCLES_2, size = SIZE_2)
48 public final class IntegerMulHighNode extends BinaryArithmeticNode<MulHigh> implements Canonicalizable.BinaryCommutative<ValueNode> {
49     public static final NodeClass<IntegerMulHighNode> TYPE = NodeClass.create(IntegerMulHighNode.class);
50 
51     public IntegerMulHighNode(ValueNode x, ValueNode y) {
52         super(TYPE, getArithmeticOpTable(x).getMulHigh(), x, y);
53     }
54 
55     @Override
56     protected BinaryOp<MulHigh> getOp(ArithmeticOpTable table) {
57         return table.getMulHigh();
58     }
59 
60     @Override
61     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
62         Value a = nodeValueMap.operand(getX());
63         Value b = nodeValueMap.operand(getY());
64         nodeValueMap.setResult(this, gen.emitMulHigh(a, b));
65     }
66 
67     @Override
68     public ValueNode canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
69         ValueNode ret = super.canonical(tool, forX, forY);
70         if (ret != this) {
71             return ret;
72         }
73 
74         if (forX.isConstant() && !forY.isConstant()) {
75             // we try to swap and canonicalize
76             ValueNode improvement = canonical(tool, forY, forX);
77             if (improvement != this) {
78                 return improvement;
79             }
80             // if this fails we only swap
81             return new IntegerMulHighNode(forY, forX);
82         }
83         return canonical(this, forY);
84     }
85 
86     private static ValueNode canonical(IntegerMulHighNode self, ValueNode forY) {
87         if (forY.isConstant()) {
88             Constant c = forY.asConstant();
89             if (c instanceof PrimitiveConstant && ((PrimitiveConstant) c).getJavaKind().isNumericInteger()) {
90                 long i = ((PrimitiveConstant) c).asLong();
91                 if (i == 0 || i == 1) {
92                     return ConstantNode.forIntegerStamp(self.stamp(NodeView.DEFAULT), 0);
93                 }
94             }
95         }
96         return self;
97     }
98 }
