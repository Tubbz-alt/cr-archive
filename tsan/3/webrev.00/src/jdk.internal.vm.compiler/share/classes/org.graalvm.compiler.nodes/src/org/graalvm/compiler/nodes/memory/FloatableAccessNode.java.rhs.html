25 package org.graalvm.compiler.nodes.memory;
26 
27 import org.graalvm.compiler.core.common.type.Stamp;
28 import org.graalvm.compiler.graph.NodeClass;
29 import org.graalvm.compiler.nodeinfo.NodeInfo;
30 import org.graalvm.compiler.nodes.FrameState;
31 import org.graalvm.compiler.nodes.extended.GuardingNode;
32 import org.graalvm.compiler.nodes.memory.address.AddressNode;
33 import jdk.internal.vm.compiler.word.LocationIdentity;
34 
35 /**
36  * An {@link FixedAccessNode} that can be converted to a {@link FloatingAccessNode}.
37  */
38 @NodeInfo
39 public abstract class FloatableAccessNode extends FixedAccessNode {
40     public static final NodeClass<FloatableAccessNode> TYPE = NodeClass.create(FloatableAccessNode.class);
41 
42     protected FloatableAccessNode(NodeClass<? extends FloatableAccessNode> c, AddressNode address, LocationIdentity location, Stamp stamp) {
43         super(c, address, location, stamp);
44     }
45 
46     protected FloatableAccessNode(NodeClass<? extends FloatableAccessNode> c, AddressNode address, LocationIdentity location, Stamp stamp, GuardingNode guard, BarrierType barrierType) {
47         super(c, address, location, stamp, guard, barrierType, false, null);
48     }
49 
50     protected FloatableAccessNode(NodeClass<? extends FloatableAccessNode> c, AddressNode address, LocationIdentity location, Stamp stamp, GuardingNode guard, BarrierType barrierType,
51                     boolean nullCheck, FrameState stateBefore) {
52         super(c, address, location, stamp, guard, barrierType, nullCheck, stateBefore);
53     }
54 
55     public abstract FloatingAccessNode asFloatingNode();
56 
57     protected boolean forceFixed;
58 
59     public void setForceFixed(boolean flag) {
60         this.forceFixed = flag;
61     }
62 
63     /**
64      * AccessNodes can float only if their location identities are not ANY_LOCATION. Furthermore, in
65      * case G1 is enabled any access (read) to the java.lang.ref.Reference.referent field which has
66      * an attached write barrier with pre-semantics can not also float.
67      */
68     public boolean canFloat() {
69         return !forceFixed && getLocationIdentity().isSingle() && getBarrierType() == BarrierType.NONE;
70     }
71 }
