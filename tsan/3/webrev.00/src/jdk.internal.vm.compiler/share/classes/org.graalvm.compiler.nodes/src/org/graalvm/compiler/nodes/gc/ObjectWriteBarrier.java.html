25 package org.graalvm.compiler.nodes.gc;
26 
27 import org.graalvm.compiler.graph.NodeClass;
28 import org.graalvm.compiler.nodeinfo.NodeInfo;
29 import org.graalvm.compiler.nodes.ValueNode;
30 import org.graalvm.compiler.nodes.memory.address.AddressNode;
31 
32 @NodeInfo
33 public abstract class ObjectWriteBarrier extends WriteBarrier {
34 
35     public static final NodeClass<ObjectWriteBarrier> TYPE = NodeClass.create(ObjectWriteBarrier.class);
36     @OptionalInput protected ValueNode value;
37     protected final boolean precise;
38 
39     protected ObjectWriteBarrier(NodeClass<? extends ObjectWriteBarrier> c, AddressNode address, ValueNode value, boolean precise) {
40         super(c, address);
41         this.value = value;
42         this.precise = precise;
43     }
44 
45     public ValueNode getValue() {
46         return value;
47     }
48 
49     public boolean usePrecise() {
50         return precise;
51     }
52 }
