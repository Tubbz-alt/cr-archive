44 /**
45  * Binary negation of long or integer values.
46  */
47 @NodeInfo(cycles = CYCLES_1, size = SIZE_1)
48 public final class NotNode extends UnaryArithmeticNode<Not> implements ArithmeticLIRLowerable, NarrowableArithmeticNode, StampInverter {
49 
50     public static final NodeClass<NotNode> TYPE = NodeClass.create(NotNode.class);
51 
52     protected NotNode(ValueNode x) {
53         super(TYPE, getArithmeticOpTable(x).getNot(), x);
54     }
55 
56     public static ValueNode create(ValueNode x) {
57         return canonicalize(null, x);
58     }
59 
60     @Override
61     protected UnaryOp<Not> getOp(ArithmeticOpTable table) {
62         return table.getNot();
63     }
64 
65     @Override
66     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
67         ValueNode ret = super.canonical(tool, forValue);
68         if (ret != this) {
69             return ret;
70         }
71         return canonicalize(this, forValue);
72     }
73 
74     private static ValueNode canonicalize(NotNode node, ValueNode x) {
75         if (x instanceof NotNode) {
76             return ((NotNode) x).getValue();
77         }
78         if (node != null) {
79             return node;
80         }
81         return new NotNode(x);
82     }
83 
84     @Override
85     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
86         nodeValueMap.setResult(this, gen.emitNot(nodeValueMap.operand(getValue())));
87     }
88 
89     @Override
90     public Stamp invertStamp(Stamp outStamp) {
91         return getArithmeticOp().foldStamp(outStamp);
92     }
93 }
