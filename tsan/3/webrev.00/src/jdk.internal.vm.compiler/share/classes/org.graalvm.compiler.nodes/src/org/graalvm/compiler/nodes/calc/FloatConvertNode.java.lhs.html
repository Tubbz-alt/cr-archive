 48 /**
 49  * A {@code FloatConvert} converts between integers and floating point numbers according to Java
 50  * semantics.
 51  */
 52 @NodeInfo(cycles = CYCLES_8)
 53 public final class FloatConvertNode extends UnaryArithmeticNode<FloatConvertOp> implements ConvertNode, Lowerable, ArithmeticLIRLowerable {
 54     public static final NodeClass<FloatConvertNode> TYPE = NodeClass.create(FloatConvertNode.class);
 55 
 56     protected final FloatConvert op;
 57 
 58     private static final EnumMap<FloatConvert, SerializableUnaryFunction<FloatConvertOp>> getOps;
 59     static {
 60         getOps = new EnumMap<>(FloatConvert.class);
 61         for (FloatConvert op : FloatConvert.values()) {
 62             getOps.put(op, table -> table.getFloatConvert(op));
 63         }
 64     }
 65 
 66     public FloatConvertNode(FloatConvert op, ValueNode input) {
 67         super(TYPE, getOps.get(op), input);
 68         this.op = op;
 69     }
 70 
 71     public static ValueNode create(FloatConvert op, ValueNode input, NodeView view) {
 72         ValueNode synonym = findSynonym(input, ArithmeticOpTable.forStamp(input.stamp(view)).getFloatConvert(op));
 73         if (synonym != null) {
 74             return synonym;
 75         }
 76         return new FloatConvertNode(op, input);
 77     }
 78 





 79     public FloatConvert getFloatConvert() {
 80         return op;
 81     }
 82 
 83     @Override
 84     public Constant convert(Constant c, ConstantReflectionProvider constantReflection) {
 85         return getArithmeticOp().foldConstant(c);
 86     }
 87 
 88     @Override
 89     public Constant reverse(Constant c, ConstantReflectionProvider constantReflection) {
 90         FloatConvertOp reverse = ArithmeticOpTable.forStamp(stamp(NodeView.DEFAULT)).getFloatConvert(op.reverse());
 91         return reverse.foldConstant(c);
 92     }
 93 
 94     @Override
 95     public boolean isLossless() {
 96         switch (getFloatConvert()) {
 97             case F2D:
 98             case I2D:
 99                 return true;
100             default:
101                 return false;
102         }
103     }
104 
105     @Override
106     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
107         ValueNode ret = super.canonical(tool, forValue);
108         if (ret != this) {
109             return ret;
110         }
111 
112         if (forValue instanceof FloatConvertNode) {
113             FloatConvertNode other = (FloatConvertNode) forValue;
114             if (other.isLossless() && other.op == this.op.reverse()) {
115                 return other.getValue();
116             }
117         }
118         return this;
119     }
120 
121     @Override
122     public void lower(LoweringTool tool) {
123         tool.getLowerer().lower(this, tool);
124     }
125 
126     @Override
127     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
128         nodeValueMap.setResult(this, gen.emitFloatConvert(getFloatConvert(), nodeValueMap.operand(getValue())));
129     }
130 
131     @Override
132     public boolean mayNullCheckSkipConversion() {
133         return false;
134     }
135 }
