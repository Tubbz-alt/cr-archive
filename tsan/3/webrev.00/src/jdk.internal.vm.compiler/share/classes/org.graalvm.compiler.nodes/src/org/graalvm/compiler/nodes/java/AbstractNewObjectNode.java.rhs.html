27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_8;
28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_8;
29 
30 import org.graalvm.compiler.core.common.type.Stamp;
31 import org.graalvm.compiler.graph.NodeClass;
32 import org.graalvm.compiler.nodeinfo.NodeInfo;
33 import org.graalvm.compiler.nodes.DeoptimizingFixedWithNextNode;
34 import org.graalvm.compiler.nodes.FrameState;
35 import org.graalvm.compiler.nodes.extended.MembarNode;
36 import org.graalvm.compiler.nodes.spi.Lowerable;
37 import org.graalvm.compiler.nodes.spi.LoweringTool;
38 
39 import jdk.vm.ci.code.MemoryBarriers;

41 /**
42  * The {@code AbstractNewObjectNode} is the base class for the new instance and new array nodes.
43  */
44 @NodeInfo(cycles = CYCLES_8, cyclesRationale = "tlab alloc + header init", size = SIZE_8)
45 public abstract class AbstractNewObjectNode extends DeoptimizingFixedWithNextNode implements Lowerable {
46 
47     public static final NodeClass<AbstractNewObjectNode> TYPE = NodeClass.create(AbstractNewObjectNode.class);
48     protected final boolean fillContents;
49 
50     /**
51      * Controls whether this allocation emits a {@link MembarNode} with
52      * {@link MemoryBarriers#STORE_STORE} as part of the object initialization.
53      */
54     protected boolean emitMemoryBarrier = true;
55 
56     protected AbstractNewObjectNode(NodeClass<? extends AbstractNewObjectNode> c, Stamp stamp, boolean fillContents, FrameState stateBefore) {
57         super(c, stamp, stateBefore);
58         this.fillContents = fillContents;
59     }
60 
61     /**
62      * @return <code>true</code> if the object's contents should be initialized to zero/null.
63      */
64     public boolean fillContents() {
65         return fillContents;
66     }
67 
68     @Override
69     public void lower(LoweringTool tool) {
70         tool.getLowerer().lower(this, tool);
71     }
72 
73     @Override
74     public boolean canDeoptimize() {
75         return true;
76     }
77 
78     public boolean emitMemoryBarrier() {
79         return emitMemoryBarrier;
80     }
81 
82     public void clearEmitMemoryBarrier() {
83         this.emitMemoryBarrier = false;
84     }
85 }
