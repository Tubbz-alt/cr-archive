 50 @NodeInfo(shortName = "==", cycles = CYCLES_2)
 51 public final class FloatEqualsNode extends CompareNode implements BinaryCommutative<ValueNode> {
 52     public static final NodeClass<FloatEqualsNode> TYPE = NodeClass.create(FloatEqualsNode.class);
 53     private static final FloatEqualsOp OP = new FloatEqualsOp();
 54 
 55     public FloatEqualsNode(ValueNode x, ValueNode y) {
 56         super(TYPE, CanonicalCondition.EQ, false, x, y);
 57         assert x.stamp(NodeView.DEFAULT) instanceof FloatStamp && y.stamp(NodeView.DEFAULT) instanceof FloatStamp : x.stamp(NodeView.DEFAULT) + " " + y.stamp(NodeView.DEFAULT);

 58         assert x.stamp(NodeView.DEFAULT).isCompatible(y.stamp(NodeView.DEFAULT));
 59     }
 60 
 61     public static LogicNode create(ValueNode x, ValueNode y, NodeView view) {
 62         LogicNode result = CompareNode.tryConstantFoldPrimitive(CanonicalCondition.EQ, x, y, false, view);
 63         if (result != null) {
 64             return result;
 65         } else {
 66             return new FloatEqualsNode(x, y).maybeCommuteInputs();
 67         }
 68     }
 69 
 70     public static LogicNode create(ConstantReflectionProvider constantReflection, MetaAccessProvider metaAccess, OptionValues options, Integer smallestCompareWidth,
 71                     ValueNode x, ValueNode y, NodeView view) {
 72         LogicNode value = OP.canonical(constantReflection, metaAccess, options, smallestCompareWidth, CanonicalCondition.EQ, false, x, y, view);
 73         if (value != null) {
 74             return value;
 75         }
 76         return create(x, y, view);
 77     }
 78 
 79     @Override
 80     public boolean isIdentityComparison() {
 81         FloatStamp xStamp = (FloatStamp) x.stamp(NodeView.DEFAULT);
 82         FloatStamp yStamp = (FloatStamp) y.stamp(NodeView.DEFAULT);
 83         /*
 84          * If both stamps have at most one 0.0 and it's the same 0.0 then this is an identity
 85          * comparison. FloatStamp isn't careful about tracking the presence of -0.0 so assume that
 86          * anything that includes 0.0 might include -0.0. So if either one is non-zero then it's an
 87          * identity comparison.
 88          */
 89         return (!xStamp.contains(0.0) || !yStamp.contains(0.0));




 90     }
 91 
 92     @Override
 93     public Node canonical(CanonicalizerTool tool, ValueNode forX, ValueNode forY) {
 94         NodeView view = NodeView.from(tool);
 95         ValueNode value = OP.canonical(tool.getConstantReflection(), tool.getMetaAccess(), tool.getOptions(), tool.smallestCompareWidth(), CanonicalCondition.EQ, unorderedIsTrue, forX, forY, view);
 96         if (value != null) {
 97             return value;
 98         }
 99         return this;
100     }
101 
102     public static class FloatEqualsOp extends CompareOp {
103 
104         @Override
105         public LogicNode canonical(ConstantReflectionProvider constantReflection, MetaAccessProvider metaAccess, OptionValues options, Integer smallestCompareWidth, CanonicalCondition condition,
106                         boolean unorderedIsTrue, ValueNode forX, ValueNode forY, NodeView view) {
107             LogicNode result = super.canonical(constantReflection, metaAccess, options, smallestCompareWidth, condition, unorderedIsTrue, forX, forY, view);
108             if (result != null) {
109                 return result;
110             }
111             Stamp xStampGeneric = forX.stamp(view);
112             Stamp yStampGeneric = forY.stamp(view);
113             if (xStampGeneric instanceof FloatStamp && yStampGeneric instanceof FloatStamp) {
114                 FloatStamp xStamp = (FloatStamp) xStampGeneric;
115                 FloatStamp yStamp = (FloatStamp) yStampGeneric;
116                 if (GraphUtil.unproxify(forX) == GraphUtil.unproxify(forY) && xStamp.isNonNaN() && yStamp.isNonNaN()) {
117                     return LogicConstantNode.tautology();
118                 } else if (xStamp.alwaysDistinct(yStamp)) {
119                     return LogicConstantNode.contradiction();
120                 }
121             }
122             return null;
123         }
124 
125         @Override
126         protected CompareNode duplicateModified(ValueNode newX, ValueNode newY, boolean unorderedIsTrue, NodeView view) {
127             if (newX.stamp(view) instanceof FloatStamp && newY.stamp(view) instanceof FloatStamp) {
128                 return new FloatEqualsNode(newX, newY);
129             } else if (newX.stamp(view) instanceof IntegerStamp && newY.stamp(view) instanceof IntegerStamp) {
130                 return new IntegerEqualsNode(newX, newY);
131             }
132             throw GraalError.shouldNotReachHere();
133         }
134     }
135 
136     @Override
137     public Stamp getSucceedingStampForX(boolean negated, Stamp xStamp, Stamp yStamp) {
138         if (!negated) {
139             return xStamp.join(yStamp);
140         }
141         return null;
142     }
143 
144     @Override
145     public Stamp getSucceedingStampForY(boolean negated, Stamp xStamp, Stamp yStamp) {
146         if (!negated) {
147             return xStamp.join(yStamp);
148         }
149         return null;
150     }
151 
152     @Override
153     public TriState tryFold(Stamp xStampGeneric, Stamp yStampGeneric) {
154         if (xStampGeneric instanceof FloatStamp && yStampGeneric instanceof FloatStamp) {
155             FloatStamp xStamp = (FloatStamp) xStampGeneric;
156             FloatStamp yStamp = (FloatStamp) yStampGeneric;
157             if (xStamp.alwaysDistinct(yStamp)) {
158                 return TriState.FALSE;
159             } else if (xStamp.neverDistinct(yStamp)) {
160                 return TriState.TRUE;
161             }
162         }
163         return TriState.UNKNOWN;
164     }
165 }
