 82         ForeignCallNode node = new ForeignCallNode(foreignCalls, descriptor, arguments);
 83         node.setStamp(returnStamp);
 84 
 85         assert verifyDescriptor(b, targetMethod, descriptor);
 86 
 87         /*
 88          * Need to update the BCI of a ForeignCallNode so that it gets the stateDuring in the case
 89          * that the foreign call can deoptimize. As with all deoptimization, we need a state in a
 90          * non-intrinsic method.
 91          */
 92         GraphBuilderContext nonIntrinsicAncestor = b.getNonIntrinsicAncestor();
 93         if (nonIntrinsicAncestor != null) {
 94             node.setBci(nonIntrinsicAncestor.bci());
 95         }
 96 
 97         JavaKind returnKind = targetMethod.getSignature().getReturnKind();
 98         if (returnKind == JavaKind.Void) {
 99             b.add(node);
100         } else {
101             b.addPush(returnKind, node);

215         } else {
216             newStateDuring = currentStateAfter;
