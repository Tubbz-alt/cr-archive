25 package org.graalvm.compiler.nodes;
26 
27 import org.graalvm.compiler.nodes.spi.NodeWithState;
28 
29 /**
30  * Interface implemented by nodes which may need {@linkplain FrameState deoptimization information}.
31  */
32 public interface DeoptimizingNode extends NodeWithState {
33 
34     /**
35      * Determines if this node needs deoptimization information.
36      */
37     boolean canDeoptimize();
38 
39     /**
40      * Interface for nodes that need a {@link FrameState} for deoptimizing to a point before their
41      * execution.
42      */
43     public interface DeoptBefore extends DeoptimizingNode {
44 
45         /**
46          * Sets the {@link FrameState} describing the program state before the execution of this
47          * node.
48          */
49         void setStateBefore(FrameState state);
50 
51         FrameState stateBefore();
52     }
53 
54     /**
55      * Interface for nodes that need a {@link FrameState} for deoptimizing to a point after their
56      * execution.
57      */
58     public interface DeoptAfter extends DeoptimizingNode, StateSplit {
59     }
60 
61     /**
62      * Interface for nodes that need a special {@link FrameState} for deoptimizing during their
63      * execution (e.g. {@link Invoke}).
64      */
65     public interface DeoptDuring extends DeoptimizingNode, StateSplit {
66 
67         FrameState stateDuring();
68 
69         /**
70          * Sets the {@link FrameState} describing the program state during the execution of this
71          * node.
72          */
73         void setStateDuring(FrameState state);
74 
75         /**
76          * Compute the {@link FrameState} describing the program state during the execution of this
77          * node from an input {@link FrameState} describing the program state after finishing the
78          * execution of this node.
79          */
80         void computeStateDuring(FrameState stateAfter);
81     }
82 }
