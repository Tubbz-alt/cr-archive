25 package org.graalvm.compiler.nodes.gc;
26 
27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_64;
28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_64;
29 
30 import org.graalvm.compiler.graph.NodeClass;
31 import org.graalvm.compiler.nodeinfo.NodeInfo;
32 import org.graalvm.compiler.nodes.ValueNode;
33 import org.graalvm.compiler.nodes.memory.address.AddressNode;
34 
35 @NodeInfo(cycles = CYCLES_64, size = SIZE_64)
36 public class G1PostWriteBarrier extends ObjectWriteBarrier {
37 
38     public static final NodeClass<G1PostWriteBarrier> TYPE = NodeClass.create(G1PostWriteBarrier.class);
39     protected final boolean alwaysNull;
40 
41     public G1PostWriteBarrier(AddressNode address, ValueNode value, boolean precise, boolean alwaysNull) {
42         this(TYPE, address, value, precise, alwaysNull);
43     }
44 
45     private G1PostWriteBarrier(NodeClass<? extends G1PostWriteBarrier> c, AddressNode address, ValueNode value, boolean precise, boolean alwaysNull) {
46         super(c, address, value, precise);
47         this.alwaysNull = alwaysNull;
48     }
49 
50     public boolean alwaysNull() {
51         return alwaysNull;
52     }
53 }
