27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_16;
28 import static org.graalvm.compiler.nodeinfo.NodeSize.SIZE_1;

29 
30 import org.graalvm.compiler.core.common.type.ArithmeticOpTable;

31 import org.graalvm.compiler.core.common.type.ArithmeticOpTable.UnaryOp.Sqrt;
32 import org.graalvm.compiler.graph.NodeClass;
33 import org.graalvm.compiler.lir.gen.ArithmeticLIRGeneratorTool;
34 import org.graalvm.compiler.nodeinfo.NodeInfo;
35 import org.graalvm.compiler.nodes.ConstantNode;
36 import org.graalvm.compiler.nodes.NodeView;
37 import org.graalvm.compiler.nodes.ValueNode;
38 import org.graalvm.compiler.nodes.spi.ArithmeticLIRLowerable;
39 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
40 
41 /**
42  * Square root.
43  */
44 @NodeInfo(cycles = CYCLES_16, size = SIZE_1)
45 public final class SqrtNode extends UnaryArithmeticNode<Sqrt> implements ArithmeticLIRLowerable {
46 
47     public static final NodeClass<SqrtNode> TYPE = NodeClass.create(SqrtNode.class);
48 
49     protected SqrtNode(ValueNode x) {
50         super(TYPE, ArithmeticOpTable::getSqrt, x);
51     }
52 
53     public static ValueNode create(ValueNode x, NodeView view) {
54         if (x.isConstant()) {
55             ArithmeticOpTable.UnaryOp<Sqrt> op = ArithmeticOpTable.forStamp(x.stamp(view)).getSqrt();
56             return ConstantNode.forPrimitive(op.foldStamp(x.stamp(view)), op.foldConstant(x.asConstant()));
57         }
58         return new SqrtNode(x);
59     }
60 





61     @Override
62     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
63         nodeValueMap.setResult(this, gen.emitMathSqrt(nodeValueMap.operand(getValue())));
64     }
65 }
