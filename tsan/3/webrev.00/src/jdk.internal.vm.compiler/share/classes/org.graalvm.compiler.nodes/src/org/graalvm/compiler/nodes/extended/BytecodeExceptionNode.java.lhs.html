 50 /**
 51  * A node that represents an exception thrown implicitly by a Java bytecode. It can be lowered to
 52  * either a {@linkplain ForeignCallDescriptor foreign} call or a pre-allocated exception object.
 53  */
 54 // @formatter:off
 55 @NodeInfo(cycles = CYCLES_8,
 56           cyclesRationale = "Node will be lowered to a foreign call.",
 57           size = SIZE_8)
 58 // @formatter:on
 59 public final class BytecodeExceptionNode extends AbstractMemoryCheckpoint implements Lowerable, MemoryCheckpoint.Single, Canonicalizable {
 60 
 61     public enum BytecodeExceptionKind {
 62         NULL_POINTER(0, NullPointerException.class),
 63         OUT_OF_BOUNDS(2, ArrayIndexOutOfBoundsException.class),
 64         CLASS_CAST(2, ClassCastException.class),
 65         ARRAY_STORE(1, ArrayStoreException.class),
 66         DIVISION_BY_ZERO(0, ArithmeticException.class),
 67         INTEGER_EXACT_OVERFLOW(0, ArithmeticException.class),
 68         LONG_EXACT_OVERFLOW(0, ArithmeticException.class);
 69 
 70         final int numArguments;
 71         final Class<? extends Throwable> exceptionClass;
 72 
 73         BytecodeExceptionKind(int numArguments, Class<? extends Throwable> exceptionClass) {
 74             this.numArguments = numArguments;
 75             this.exceptionClass = exceptionClass;
 76         }
 77     }
 78 
 79     public static final NodeClass<BytecodeExceptionNode> TYPE = NodeClass.create(BytecodeExceptionNode.class);
 80     protected final BytecodeExceptionKind exceptionKind;
 81     @Input NodeInputList<ValueNode> arguments;
 82 
 83     public BytecodeExceptionNode(MetaAccessProvider metaAccess, BytecodeExceptionKind exceptionKind, ValueNode... arguments) {
 84         super(TYPE, StampFactory.objectNonNull(TypeReference.createExactTrusted(metaAccess.lookupJavaType(exceptionKind.exceptionClass))));
 85         this.exceptionKind = exceptionKind;
 86         this.arguments = new NodeInputList<>(this, arguments);
 87         GraalError.guarantee(arguments.length == exceptionKind.numArguments, "Mismatch in argument count for BytecodeExceptionNode");
 88     }
 89 
 90     public BytecodeExceptionKind getExceptionKind() {
 91         return exceptionKind;
 92     }
 93 
 94     @Override
 95     public String toString(Verbosity verbosity) {
 96         if (verbosity == Verbosity.Name) {
 97             return super.toString(verbosity) + "#" + exceptionKind;
 98         }
 99         return super.toString(verbosity);
100     }
101 
102     @Override
103     public LocationIdentity getLocationIdentity() {
104         return LocationIdentity.any();
105     }
106 
107     @Override
108     public Node canonical(CanonicalizerTool tool) {
109         if (tool.allUsagesAvailable() && getUsageCount() == 0) {
110             return null;
111         }
112         return this;
113     }
114 
115     @Override
116     public void lower(LoweringTool tool) {
117         tool.getLowerer().lower(this, tool);
118     }
119 
120     public NodeInputList<ValueNode> getArguments() {
121         return arguments;
122     }









123 }
