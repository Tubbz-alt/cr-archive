25 package org.graalvm.compiler.nodes;
26 
27 import org.graalvm.compiler.core.common.type.Stamp;

28 import org.graalvm.compiler.graph.NodeClass;
29 import org.graalvm.compiler.nodeinfo.NodeInfo;
30 
31 /**
32  * The {@code ControlSplitNode} is a base class for all instructions that split the control flow
33  * (ie. have more than one successor).
34  */
35 @NodeInfo
36 public abstract class ControlSplitNode extends FixedNode {
37     public static final NodeClass<ControlSplitNode> TYPE = NodeClass.create(ControlSplitNode.class);
38 
39     protected ControlSplitNode(NodeClass<? extends ControlSplitNode> c, Stamp stamp) {
40         super(c, stamp);
41     }
42 
43     public abstract double probability(AbstractBeginNode successor);
44 
45     /**
46      * Attempts to set the probability for the given successor to the passed value (which has to be
47      * in the range of 0.0 and 1.0). Returns whether setting the probability was successful.
48      */
49     public abstract boolean setProbability(AbstractBeginNode successor, double value);
50 
51     /**
52      * Primary successor of the control split. Data dependencies on the node have to be scheduled in
53      * the primary successor. Returns null if data dependencies are not expected.
54      *
55      * @return the primary successor
56      */
57     public abstract AbstractBeginNode getPrimarySuccessor();
58 
59     /**
60      * Returns the number of successors.
61      */
62     public abstract int getSuccessorCount();
63 }
