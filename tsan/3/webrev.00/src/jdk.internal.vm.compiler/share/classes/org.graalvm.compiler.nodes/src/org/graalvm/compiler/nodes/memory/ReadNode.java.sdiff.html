 68         this(TYPE, address, location, stamp, null, barrierType, false, null);
 69     }
 70 
 71     protected ReadNode(NodeClass<? extends ReadNode> c, AddressNode address, LocationIdentity location, Stamp stamp, GuardingNode guard, BarrierType barrierType, boolean nullCheck,

 75 
 76     @Override
 77     public void generate(NodeLIRBuilderTool gen) {
 78         LIRKind readKind = gen.getLIRGeneratorTool().getLIRKind(getAccessStamp());
 79         gen.setResult(this, gen.getLIRGeneratorTool().getArithmetic().emitLoad(readKind, gen.operand(address), gen.state(this)));
 80     }
 81 
 82     @Override
 83     public Node canonical(CanonicalizerTool tool) {
 84         if (tool.allUsagesAvailable() && hasNoUsages()) {
 85             // Read without usages or guard can be safely removed.
 86             return null;
 87         }
 88         if (!getNullCheck()) {
 89             return canonicalizeRead(this, getAddress(), getLocationIdentity(), tool);
 90         } else {
 91             // if this read is a null check, then replacing it with the value is incorrect for
 92             // guard-type usages
 93             return this;
 94         }
 95     }
 96 
 97     @SuppressWarnings("try")
 98     @Override
 99     public FloatingAccessNode asFloatingNode(MemoryNode lastLocationAccess) {
100         try (DebugCloseable position = withNodeSourcePosition()) {
101             return graph().unique(new FloatingReadNode(getAddress(), getLocationIdentity(), lastLocationAccess, stamp(NodeView.DEFAULT), getGuard(), getBarrierType()));
102         }
103     }
104 
105     @Override
106     public boolean isAllowedUsageType(InputType type) {
107         return (getNullCheck() && type == InputType.Guard) ? true : super.isAllowedUsageType(type);
108     }
