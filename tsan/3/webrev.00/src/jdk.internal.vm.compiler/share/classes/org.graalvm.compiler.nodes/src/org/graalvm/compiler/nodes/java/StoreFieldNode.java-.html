 27 import static org.graalvm.compiler.nodeinfo.NodeCycles.CYCLES_8;

 38 import org.graalvm.compiler.nodes.spi.VirtualizerTool;
 39 import org.graalvm.compiler.nodes.virtual.VirtualInstanceNode;
 40 import org.graalvm.compiler.nodes.virtual.VirtualObjectNode;
 41 
 42 import jdk.vm.ci.meta.ResolvedJavaField;

 44 /**
 45  * The {@code StoreFieldNode} represents a write to a static or instance field.
 46  */
 47 @NodeInfo(nameTemplate = "StoreField#{p#field/s}")
 48 public final class StoreFieldNode extends AccessFieldNode implements StateSplit, Virtualizable {
 49     public static final NodeClass<StoreFieldNode> TYPE = NodeClass.create(StoreFieldNode.class);
 50 
 51     @Input ValueNode value;
 52     @OptionalInput(InputType.State) FrameState stateAfter;
 53 
 54     @Override
 55     public FrameState stateAfter() {
 56         return stateAfter;
 57     }
 58 
 59     @Override
 60     public void setStateAfter(FrameState x) {
 61         assert x == null || x.isAlive() : "frame state must be in a graph";
 62         updateUsages(stateAfter, x);
 63         stateAfter = x;
 64     }
 65 
 66     @Override
 67     public boolean hasSideEffect() {
 68         return true;
 69     }
 70 
 71     public ValueNode value() {
 72         return value;
 73     }
 74 
 75     public StoreFieldNode(ValueNode object, ResolvedJavaField field, ValueNode value) {
 76         this(object, field, value, field.isVolatile());
 77     }
 78 
 79     public StoreFieldNode(ValueNode object, ResolvedJavaField field, ValueNode value, boolean volatileAccess) {
 80         super(TYPE, StampFactory.forVoid(), object, field, volatileAccess);
 81         this.value = value;
 82     }
 83 
 84     public StoreFieldNode(ValueNode object, ResolvedJavaField field, ValueNode value, FrameState stateAfter, boolean volatileAccess) {
 85         super(TYPE, StampFactory.forVoid(), object, field, volatileAccess);
 86         this.value = value;
 87         this.stateAfter = stateAfter;
 88     }
 89 
 90     @Override
 91     public void virtualize(VirtualizerTool tool) {
 92         ValueNode alias = tool.getAlias(object());
 93         if (alias instanceof VirtualObjectNode) {
 94             VirtualInstanceNode virtual = (VirtualInstanceNode) alias;
 95             int fieldIndex = virtual.fieldIndex(field());
 96             if (fieldIndex != -1) {
 97                 tool.setVirtualEntry(virtual, fieldIndex, value());
 98                 tool.delete();
 99             }
100         }
101     }
102 
103     public FrameState getState() {
104         return stateAfter;
105     }
106 
107     @Override
108     public NodeCycles estimatedNodeCycles() {
109         if (isVolatile()) {
110             return CYCLES_8;
111         }
112         return super.estimatedNodeCycles();
113     }
114 }
