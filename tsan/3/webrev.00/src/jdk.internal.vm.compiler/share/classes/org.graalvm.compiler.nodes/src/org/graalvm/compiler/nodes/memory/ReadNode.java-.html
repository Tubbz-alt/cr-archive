 59 /**
 60  * Reads an {@linkplain FixedAccessNode accessed} value.
 61  */
 62 @NodeInfo(nameTemplate = "Read#{p#location/s}", cycles = CYCLES_2, size = SIZE_1)
 63 public class ReadNode extends FloatableAccessNode implements LIRLowerableAccess, Canonicalizable, Virtualizable, GuardingNode {
 64 
 65     public static final NodeClass<ReadNode> TYPE = NodeClass.create(ReadNode.class);
 66 
 67     public ReadNode(AddressNode address, LocationIdentity location, Stamp stamp, BarrierType barrierType) {
 68         this(TYPE, address, location, stamp, null, barrierType, false, null);
 69     }
 70 
 71     protected ReadNode(NodeClass<? extends ReadNode> c, AddressNode address, LocationIdentity location, Stamp stamp, GuardingNode guard, BarrierType barrierType, boolean nullCheck,
 72                     FrameState stateBefore) {
 73         super(c, address, location, stamp, guard, barrierType, nullCheck, stateBefore);
 74     }
 75 
 76     @Override
 77     public void generate(NodeLIRBuilderTool gen) {
 78         LIRKind readKind = gen.getLIRGeneratorTool().getLIRKind(getAccessStamp());
 79         gen.setResult(this, gen.getLIRGeneratorTool().getArithmetic().emitLoad(readKind, gen.operand(address), gen.state(this)));
 80     }
 81 
 82     @Override
 83     public Node canonical(CanonicalizerTool tool) {
 84         if (tool.allUsagesAvailable() && hasNoUsages()) {
 85             // Read without usages or guard can be safely removed.
 86             return null;
 87         }
 88         if (!getNullCheck()) {
 89             return canonicalizeRead(this, getAddress(), getLocationIdentity(), tool);
 90         } else {
 91             // if this read is a null check, then replacing it with the value is incorrect for
 92             // guard-type usages
 93             return this;
 94         }
 95     }
 96 
 97     @SuppressWarnings("try")
 98     @Override
 99     public FloatingAccessNode asFloatingNode(MemoryNode lastLocationAccess) {
100         try (DebugCloseable position = withNodeSourcePosition()) {
101             return graph().unique(new FloatingReadNode(getAddress(), getLocationIdentity(), lastLocationAccess, stamp(NodeView.DEFAULT), getGuard(), getBarrierType()));
102         }
103     }
104 
105     @Override
106     public boolean isAllowedUsageType(InputType type) {
107         return (getNullCheck() && type == InputType.Guard) ? true : super.isAllowedUsageType(type);
108     }
109 
110     public static ValueNode canonicalizeRead(ValueNode read, AddressNode address, LocationIdentity locationIdentity, CanonicalizerTool tool) {
111         NodeView view = NodeView.from(tool);
112         MetaAccessProvider metaAccess = tool.getMetaAccess();
113         if (tool.canonicalizeReads() && address instanceof OffsetAddressNode) {
114             OffsetAddressNode objAddress = (OffsetAddressNode) address;
115             ValueNode object = objAddress.getBase();
116             if (metaAccess != null && object.isConstant() && !object.isNullConstant() && objAddress.getOffset().isConstant()) {
117                 long displacement = objAddress.getOffset().asJavaConstant().asLong();
118                 int stableDimension = ((ConstantNode) object).getStableDimension();
119                 if (locationIdentity.isImmutable() || stableDimension > 0) {
120                     Constant constant = read.stamp(view).readConstant(tool.getConstantReflection().getMemoryAccessProvider(), object.asConstant(), displacement);
121                     boolean isDefaultStable = locationIdentity.isImmutable() || ((ConstantNode) object).isDefaultStable();
122                     if (constant != null && (isDefaultStable || !constant.isDefaultForKind())) {
123                         return ConstantNode.forConstant(read.stamp(view), constant, Math.max(stableDimension - 1, 0), isDefaultStable, metaAccess);
124                     }
125                 }
126             }
127             if (locationIdentity.equals(ARRAY_LENGTH_LOCATION)) {
128                 ValueNode length = GraphUtil.arrayLength(object, ArrayLengthProvider.FindLengthMode.CANONICALIZE_READ, tool.getConstantReflection());
129                 if (length != null) {
130                     return length;
131                 }
132             }
133             if (locationIdentity instanceof CanonicalizableLocation) {
134                 CanonicalizableLocation canonicalize = (CanonicalizableLocation) locationIdentity;
135                 ValueNode result = canonicalize.canonicalizeRead(read, address, object, tool);
136                 assert result != null && result.stamp(view).isCompatible(read.stamp(view));
137                 return result;
138             }
139 
140         }
141         return read;
142     }
143 
144     @Override
145     public void virtualize(VirtualizerTool tool) {
146         throw GraalError.shouldNotReachHere("unexpected ReadNode before PEA");
147     }
148 
149     @Override
150     public boolean canNullCheck() {
151         return true;
152     }
153 
154     @Override
155     public Stamp getAccessStamp() {
156         return stamp(NodeView.DEFAULT);
157     }
158 }
