42 
43 /**
44  * Square root.
45  */
46 @NodeInfo(cycles = CYCLES_16, size = SIZE_1)
47 public final class SqrtNode extends UnaryArithmeticNode<Sqrt> implements ArithmeticLIRLowerable {
48 
49     public static final NodeClass<SqrtNode> TYPE = NodeClass.create(SqrtNode.class);
50 
51     protected SqrtNode(ValueNode x) {
52         super(TYPE, getArithmeticOpTable(x).getSqrt(), x);
53     }
54 
55     public static ValueNode create(ValueNode x, NodeView view) {
56         if (x.isConstant()) {
57             ArithmeticOpTable.UnaryOp<Sqrt> op = ArithmeticOpTable.forStamp(x.stamp(view)).getSqrt();
58             return ConstantNode.forPrimitive(op.foldStamp(x.stamp(view)), op.foldConstant(x.asConstant()));
59         }
60         return new SqrtNode(x);
61     }
62 
63     @Override
64     protected UnaryOp<Sqrt> getOp(ArithmeticOpTable table) {
65         return table.getSqrt();
66     }
67 
68     @Override
69     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
70         nodeValueMap.setResult(this, gen.emitMathSqrt(nodeValueMap.operand(getValue())));
71     }
72 }
