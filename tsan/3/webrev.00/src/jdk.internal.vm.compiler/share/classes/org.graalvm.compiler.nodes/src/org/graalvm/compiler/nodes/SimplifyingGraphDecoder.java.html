 56 /**
 57  * Graph decoder that simplifies nodes during decoding. The standard
 58  * {@link Canonicalizable#canonical node canonicalization} interface is used to canonicalize nodes
 59  * during decoding. Additionally, {@link IfNode branches} and {@link IntegerSwitchNode switches}
 60  * with constant conditions are simplified.
 61  */
 62 public class SimplifyingGraphDecoder extends GraphDecoder {
 63 
 64     protected final CoreProviders providers;
 65     protected final boolean canonicalizeReads;
 66     protected final CanonicalizerTool canonicalizerTool;
 67 
 68     protected class PECanonicalizerTool implements CanonicalizerTool {
 69 
 70         private final Assumptions assumptions;
 71         private final OptionValues options;
 72 
 73         public PECanonicalizerTool(Assumptions assumptions, OptionValues options) {
 74             this.assumptions = assumptions;
 75             this.options = options;
 76         }
 77 
 78         @Override
 79         public OptionValues getOptions() {
 80             return options;
 81         }
 82 
 83         @Override
 84         public MetaAccessProvider getMetaAccess() {
 85             return providers.getMetaAccess();
 86         }
 87 
 88         @Override
 89         public ConstantReflectionProvider getConstantReflection() {
 90             return providers.getConstantReflection();
 91         }
 92 
 93         @Override
 94         public ConstantFieldProvider getConstantFieldProvider() {
 95             return providers.getConstantFieldProvider();
 96         }
 97 
 98         @Override
 99         public boolean canonicalizeReads() {
100             return canonicalizeReads;
101         }
102 
103         @Override
104         public boolean allUsagesAvailable() {
105             return false;
106         }
107 
108         @Override
109         public Assumptions getAssumptions() {
110             return assumptions;
111         }
112 
113         @Override
114         public Integer smallestCompareWidth() {
115             // to be safe, just report null here
116             // there will be more opportunities for this optimization later
117             return null;
118         }

121     @NodeInfo(cycles = CYCLES_IGNORED, size = SIZE_IGNORED, allowedUsageTypes = {Guard})
122     static class CanonicalizeToNullNode extends FloatingNode implements Canonicalizable, GuardingNode {
123         public static final NodeClass<CanonicalizeToNullNode> TYPE = NodeClass.create(CanonicalizeToNullNode.class);
124 
125         protected CanonicalizeToNullNode(Stamp stamp) {
126             super(TYPE, stamp);
127         }
128 
129         @Override
130         public Node canonical(CanonicalizerTool tool) {
131             return null;
132         }
133     }

135     public SimplifyingGraphDecoder(Architecture architecture, StructuredGraph graph, CoreProviders providers, boolean canonicalizeReads) {
136         super(architecture, graph);
137         this.providers = providers;
138         this.canonicalizeReads = canonicalizeReads;
139         this.canonicalizerTool = new PECanonicalizerTool(graph.getAssumptions(), graph.getOptions());
140     }

142     @Override
143     protected void cleanupGraph(MethodScope methodScope) {
144         GraphUtil.normalizeLoops(graph);
145         super.cleanupGraph(methodScope);
146 
147         for (Node node : graph.getNewNodes(methodScope.methodStartMark)) {
148             if (node instanceof MergeNode) {
149                 MergeNode mergeNode = (MergeNode) node;
150                 if (mergeNode.forwardEndCount() == 1) {
151                     graph.reduceTrivialMerge(mergeNode);
152                 }
153             } else if (node instanceof BeginNode || node instanceof KillingBeginNode) {
154                 if (!(node.predecessor() instanceof ControlSplitNode) && node.hasNoUsages()) {
155                     GraphUtil.unlinkFixedNode((AbstractBeginNode) node);
156                     node.safeDelete();
157                 }
158             }
159         }
160 
161         for (Node node : graph.getNewNodes(methodScope.methodStartMark)) {
162             GraphUtil.tryKillUnused(node);
163         }
164     }
165 
166     @Override
167     protected boolean allowLazyPhis() {
168         /*
169          * We do not need to exactly reproduce the encoded graph, so we want to avoid unnecessary
170          * phi functions.
171          */
172         return true;
173     }
174 
175     @Override
176     protected void handleMergeNode(MergeNode merge) {
177         /*
178          * All inputs of non-loop phi nodes are known by now. We can infer the stamp for the phi, so
179          * that parsing continues with more precise type information.
180          */
181         for (ValuePhiNode phi : merge.valuePhis()) {
182             phi.inferStamp();
183         }
184     }
185 
186     @Override
187     protected void handleFixedNode(MethodScope methodScope, LoopScope loopScope, int nodeOrderId, FixedNode node) {
188         Node canonical = canonicalizeFixedNode(methodScope, node);
189         if (canonical != node) {
190             handleCanonicalization(loopScope, nodeOrderId, node, canonical);
191         }
192     }
193 
194     /**
195      * Canonicalizes the provided node, which was originally a {@link FixedNode} but can already be
196      * canonicalized (and therefore be a non-fixed node).
197      *
198      * @param methodScope The current method.
199      * @param node The node to be canonicalized.
200      */
201     protected Node canonicalizeFixedNode(MethodScope methodScope, Node node) {
202         if (node instanceof LoadFieldNode) {
203             LoadFieldNode loadFieldNode = (LoadFieldNode) node;
204             return loadFieldNode.canonical(canonicalizerTool);
205         } else if (node instanceof FixedGuardNode) {
206             FixedGuardNode guard = (FixedGuardNode) node;
207             if (guard.getCondition() instanceof LogicConstantNode) {
208                 LogicConstantNode condition = (LogicConstantNode) guard.getCondition();
209                 if (condition.getValue() == guard.isNegated()) {
210                     DeoptimizeNode deopt = new DeoptimizeNode(guard.getAction(), guard.getReason(), guard.getSpeculation());
211                     if (guard.stateBefore() != null) {
212                         deopt.setStateBefore(guard.stateBefore());
213                     }
214                     return deopt;
215                 } else {
216                     return null;
217                 }
218             }
219             return node;
220         } else if (node instanceof IfNode) {
221             IfNode ifNode = (IfNode) node;
222             if (ifNode.condition() instanceof LogicNegationNode) {
223                 ifNode.eliminateNegation();
224             }
225             if (ifNode.condition() instanceof LogicConstantNode) {
226                 boolean condition = ((LogicConstantNode) ifNode.condition()).getValue();
227                 AbstractBeginNode survivingSuccessor = ifNode.getSuccessor(condition);
228                 AbstractBeginNode deadSuccessor = ifNode.getSuccessor(!condition);
229 
230                 graph.removeSplit(ifNode, survivingSuccessor);
231                 assert deadSuccessor.next() == null : "must not be parsed yet";
232                 deadSuccessor.safeDelete();
233             }
234             return node;
235         } else if (node instanceof LoadIndexedNode) {
236             LoadIndexedNode loadIndexedNode = (LoadIndexedNode) node;
237             return loadIndexedNode.canonical(canonicalizerTool);
238         } else if (node instanceof ArrayLengthNode) {
239             ArrayLengthNode arrayLengthNode = (ArrayLengthNode) node;
240             return arrayLengthNode.canonical(canonicalizerTool);
241         } else if (node instanceof IntegerSwitchNode && ((IntegerSwitchNode) node).value().isConstant()) {
242             IntegerSwitchNode switchNode = (IntegerSwitchNode) node;
243             int value = switchNode.value().asJavaConstant().asInt();
244             AbstractBeginNode survivingSuccessor = switchNode.successorAtKey(value);
245             List<Node> allSuccessors = switchNode.successors().snapshot();
246 
247             graph.removeSplit(switchNode, survivingSuccessor);
248             for (Node successor : allSuccessors) {
249                 if (successor != survivingSuccessor) {
250                     assert ((AbstractBeginNode) successor).next() == null : "must not be parsed yet";
251                     successor.safeDelete();
252                 }
253             }
254             return node;
255         } else if (node instanceof Canonicalizable) {
256             return ((Canonicalizable) node).canonical(canonicalizerTool);
257         } else {
258             return node;
259         }
260     }
261 
262     private static Node canonicalizeFixedNodeToNull(FixedNode node) {
263         /*
264          * When a node is unnecessary, we must not remove it right away because there might be nodes
265          * that use it as a guard input. Therefore, we replace it with a more lightweight node
266          * (which is floating and has no inputs).
267          */
268         return new CanonicalizeToNullNode(node.stamp);
269     }
270 
271     @SuppressWarnings("try")
272     private void handleCanonicalization(LoopScope loopScope, int nodeOrderId, FixedNode node, Node c) {
273         assert c != node : "unnecessary call";
274         try (DebugCloseable position = graph.withNodeSourcePosition(node)) {
275             Node canonical = c == null ? canonicalizeFixedNodeToNull(node) : c;
276             if (!canonical.isAlive()) {
277                 assert !canonical.isDeleted();
278                 canonical = graph.addOrUniqueWithInputs(canonical);
279                 if (canonical instanceof FixedWithNextNode) {
280                     graph.addBeforeFixed(node, (FixedWithNextNode) canonical);
281                 } else if (canonical instanceof ControlSinkNode) {
282                     FixedWithNextNode predecessor = (FixedWithNextNode) node.predecessor();
283                     predecessor.setNext((ControlSinkNode) canonical);
284                     List<Node> successorSnapshot = node.successors().snapshot();
285                     node.safeDelete();
286                     for (Node successor : successorSnapshot) {
287                         successor.safeDelete();
288                     }
289                 } else {
290                     assert !(canonical instanceof FixedNode);
291                 }
292             }
293             if (!node.isDeleted()) {
294                 GraphUtil.unlinkFixedNode((FixedWithNextNode) node);
295                 node.replaceAtUsagesAndDelete(canonical);
296             }
297             assert lookupNode(loopScope, nodeOrderId) == node;
298             registerNode(loopScope, nodeOrderId, canonical, true, false);
299         }
300     }

302     @Override
303     @SuppressWarnings("try")
304     protected Node handleFloatingNodeBeforeAdd(MethodScope methodScope, LoopScope loopScope, Node node) {
305         if (node instanceof ValueNode) {
306             ((ValueNode) node).inferStamp();
307         }
308         if (node instanceof Canonicalizable) {
309             try (DebugCloseable context = graph.withNodeSourcePosition(node)) {
310                 Node canonical = ((Canonicalizable) node).canonical(canonicalizerTool);
311                 if (canonical == null) {
312                     /*
313                      * This is a possible return value of canonicalization. However, we might need
314                      * to add additional usages later on for which we need a node. Therefore, we
315                      * just do nothing and leave the node in place.
316                      */
317                 } else if (canonical != node) {
318                     if (!canonical.isAlive()) {
319                         assert !canonical.isDeleted();
320                         canonical = graph.addOrUniqueWithInputs(canonical);
321                     }
322                     assert node.hasNoUsages();
323                     return canonical;
324                 }
325             }
326         }
327         return node;
328     }
329 
330     @Override
331     protected Node addFloatingNode(MethodScope methodScope, Node node) {
332         /*
333          * In contrast to the base class implementation, we do not need to exactly reproduce the
334          * encoded graph. Since we do canonicalization, we also want nodes to be unique.
335          */
336         return graph.addOrUnique(node);
337     }
