25 package org.graalvm.compiler.nodes.memory.address;
26 
27 import org.graalvm.compiler.graph.NodeClass;
28 import org.graalvm.compiler.nodeinfo.InputType;
29 import org.graalvm.compiler.nodeinfo.NodeInfo;
30 import org.graalvm.compiler.nodes.ValueNode;
31 import org.graalvm.compiler.nodes.spi.Lowerable;
32 import org.graalvm.compiler.nodes.spi.LoweringTool;
33 
34 import jdk.vm.ci.meta.JavaKind;
35 
36 /**
37  * Represents an address that points to an element of a Java array.
38  */
39 @NodeInfo(allowedUsageTypes = InputType.Association)
40 public class IndexAddressNode extends AddressNode implements Lowerable {
41     public static final NodeClass<IndexAddressNode> TYPE = NodeClass.create(IndexAddressNode.class);
42 
43     @Input ValueNode array;
44     @Input ValueNode index;
45 
46     private final JavaKind arrayKind;
47     private final JavaKind elementKind;
48 
49     public IndexAddressNode(ValueNode array, ValueNode index, JavaKind elementKind) {
50         this(array, index, elementKind, elementKind);
51     }
52 
53     public IndexAddressNode(ValueNode array, ValueNode index, JavaKind arrayKind, JavaKind elementKind) {
54         super(TYPE);
55         this.array = array;
56         this.index = index;
57         this.arrayKind = arrayKind;
58         this.elementKind = elementKind;
59     }
60 
61     @Override
62     public ValueNode getBase() {
63         return array;
64     }
65 
66     public ValueNode getArray() {
67         return array;
68     }
69 
70     @Override
71     public ValueNode getIndex() {
72         return index;
73     }
74 
75     @Override
76     public long getMaxConstantDisplacement() {
77         return Long.MAX_VALUE;
78     }
79 
80     public JavaKind getArrayKind() {
81         return arrayKind;
82     }
83 
84     public JavaKind getElementKind() {
85         return elementKind;
86     }
87 
88     @Override
89     public void lower(LoweringTool tool) {
90         tool.getLowerer().lower(this, tool);
91     }
92 }
