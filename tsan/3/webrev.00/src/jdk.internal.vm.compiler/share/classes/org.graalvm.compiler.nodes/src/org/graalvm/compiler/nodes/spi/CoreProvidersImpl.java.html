25 package org.graalvm.compiler.nodes.spi;
26 
27 import org.graalvm.compiler.core.common.spi.ConstantFieldProvider;
28 import org.graalvm.compiler.core.common.spi.ForeignCallsProvider;
29 
30 import jdk.vm.ci.meta.ConstantReflectionProvider;
31 import jdk.vm.ci.meta.MetaAccessProvider;
32 
33 public class CoreProvidersImpl implements CoreProviders {
34     protected final MetaAccessProvider metaAccess;
35     protected final ConstantReflectionProvider constantReflection;
36     protected final ConstantFieldProvider constantFieldProvider;
37     protected final LoweringProvider lowerer;
38     protected final Replacements replacements;
39     protected final StampProvider stampProvider;
40     protected final ForeignCallsProvider foreignCalls;
41     protected final GCProvider gc;
42 
43     protected CoreProvidersImpl(MetaAccessProvider metaAccess, ConstantReflectionProvider constantReflection, ConstantFieldProvider constantFieldProvider, LoweringProvider lowerer,
44                     Replacements replacements, StampProvider stampProvider, ForeignCallsProvider foreignCalls, GCProvider gc) {
45         this.metaAccess = metaAccess;
46         this.constantReflection = constantReflection;
47         this.constantFieldProvider = constantFieldProvider;
48         this.lowerer = lowerer;
49         this.replacements = replacements;
50         this.stampProvider = stampProvider;
51         this.foreignCalls = foreignCalls;
52         this.gc = gc;
53     }
54 
55     @Override
56     public MetaAccessProvider getMetaAccess() {
57         return metaAccess;
58     }
59 
60     @Override
61     public ConstantReflectionProvider getConstantReflection() {
62         return constantReflection;
63     }
64 
65     @Override
66     public ConstantFieldProvider getConstantFieldProvider() {
67         return constantFieldProvider;
68     }
69 
70     @Override
71     public LoweringProvider getLowerer() {
72         return lowerer;
73     }
74 
75     @Override
76     public Replacements getReplacements() {
77         return replacements;
78     }
79 
80     @Override
81     public StampProvider getStampProvider() {
82         return stampProvider;
83     }
84 
85     @Override
86     public ForeignCallsProvider getForeignCalls() {
87         return foreignCalls;
88     }
89 
90     @Override
91     public GCProvider getGC() {
92         return gc;
93     }
94 }
