25 package org.graalvm.compiler.nodes.gc;
26 
27 import org.graalvm.compiler.core.common.type.StampFactory;
28 import org.graalvm.compiler.graph.NodeClass;
29 import org.graalvm.compiler.nodeinfo.InputType;
30 import org.graalvm.compiler.nodeinfo.NodeInfo;
31 import org.graalvm.compiler.nodes.FixedWithNextNode;
32 import org.graalvm.compiler.nodes.memory.address.AddressNode;
33 import org.graalvm.compiler.nodes.spi.Lowerable;
34 import org.graalvm.compiler.nodes.spi.LoweringTool;
35 
36 @NodeInfo
37 public abstract class WriteBarrier extends FixedWithNextNode implements Lowerable {
38 
39     public static final NodeClass<WriteBarrier> TYPE = NodeClass.create(WriteBarrier.class);
40     @Input(InputType.Association) AddressNode address;
41 
42     protected WriteBarrier(NodeClass<? extends WriteBarrier> c, AddressNode address) {
43         super(c, StampFactory.forVoid());
44         this.address = address;
45     }
46 
47     @Override
48     public void lower(LoweringTool tool) {
49         assert graph().getGuardsStage().areFrameStatesAtDeopts();
50         tool.getLowerer().lower(this, tool);
51     }
52 
53     public AddressNode getAddress() {
54         return address;
55     }
56 }
