 45 /**
 46  * The {@code NegateNode} node negates its operand.
 47  */
 48 @NodeInfo(cycles = CYCLES_2, size = SIZE_1)
 49 public final class NegateNode extends UnaryArithmeticNode<Neg> implements NarrowableArithmeticNode, StampInverter {
 50 
 51     public static final NodeClass<NegateNode> TYPE = NodeClass.create(NegateNode.class);
 52 
 53     public NegateNode(ValueNode value) {
 54         super(TYPE, getArithmeticOpTable(value).getNeg(), value);
 55     }
 56 
 57     public static ValueNode create(ValueNode value, NodeView view) {
 58         ValueNode synonym = findSynonym(value, view);
 59         if (synonym != null) {
 60             return synonym;
 61         }
 62         return new NegateNode(value);
 63     }
 64 
 65     @Override
 66     protected UnaryOp<Neg> getOp(ArithmeticOpTable table) {
 67         return table.getNeg();
 68     }
 69 
 70     @Override
 71     public ValueNode canonical(CanonicalizerTool tool, ValueNode forValue) {
 72         ValueNode synonym = findSynonym(forValue, getOp(forValue));
 73         if (synonym != null) {
 74             return synonym;
 75         }
 76         return this;
 77     }
 78 
 79     protected static ValueNode findSynonym(ValueNode forValue, NodeView view) {
 80         ArithmeticOpTable.UnaryOp<Neg> negOp = ArithmeticOpTable.forStamp(forValue.stamp(view)).getNeg();
 81 
 82         // Folds constants
 83         ValueNode synonym = UnaryArithmeticNode.findSynonym(forValue, negOp);
 84         if (synonym != null) {
 85             return synonym;
 86         }
 87         if (forValue instanceof NegateNode) {
 88             return ((NegateNode) forValue).getValue();
 89         }
 90         if (forValue instanceof SubNode && !(forValue.stamp(view) instanceof FloatStamp)) {
 91             SubNode sub = (SubNode) forValue;
 92             return SubNode.create(sub.getY(), sub.getX(), view);
 93         }
 94         return null;
 95     }
 96 
 97     @Override
 98     public void generate(NodeLIRBuilderTool nodeValueMap, ArithmeticLIRGeneratorTool gen) {
 99         nodeValueMap.setResult(this, gen.emitNegate(nodeValueMap.operand(getValue())));
100     }
101 
102     @Override
103     public Stamp invertStamp(Stamp outStamp) {
104         return getArithmeticOp().foldStamp(outStamp);
105     }
106 }
