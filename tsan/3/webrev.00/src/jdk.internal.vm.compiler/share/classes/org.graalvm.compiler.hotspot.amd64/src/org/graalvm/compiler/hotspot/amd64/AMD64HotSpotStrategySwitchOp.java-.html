25 package org.graalvm.compiler.hotspot.amd64;

27 import org.graalvm.compiler.asm.amd64.AMD64Address;
28 import org.graalvm.compiler.asm.amd64.AMD64MacroAssembler;
29 import org.graalvm.compiler.lir.LIRInstructionClass;
30 import org.graalvm.compiler.lir.LabelRef;
31 import org.graalvm.compiler.lir.SwitchStrategy;
32 import org.graalvm.compiler.lir.amd64.AMD64ControlFlow;
33 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
34 
35 import jdk.vm.ci.code.Register;
36 import jdk.vm.ci.code.ValueUtil;
37 import jdk.vm.ci.hotspot.HotSpotMetaspaceConstant;
38 import jdk.vm.ci.meta.Constant;
39 import jdk.vm.ci.meta.Value;
40 
41 final class AMD64HotSpotStrategySwitchOp extends AMD64ControlFlow.StrategySwitchOp {
42     public static final LIRInstructionClass<AMD64HotSpotStrategySwitchOp> TYPE = LIRInstructionClass.create(AMD64HotSpotStrategySwitchOp.class);
43 
44     AMD64HotSpotStrategySwitchOp(SwitchStrategy strategy, LabelRef[] keyTargets, LabelRef defaultTarget, Value key, Value scratch) {
45         super(TYPE, strategy, keyTargets, defaultTarget, key, scratch);
46     }
47 
48     @Override
49     public void emitCode(final CompilationResultBuilder crb, final AMD64MacroAssembler masm) {
50         strategy.run(new HotSpotSwitchClosure(ValueUtil.asRegister(key), crb, masm));
51     }
52 
53     public class HotSpotSwitchClosure extends SwitchClosure {
54 
55         protected HotSpotSwitchClosure(Register keyRegister, CompilationResultBuilder crb, AMD64MacroAssembler masm) {
56             super(keyRegister, crb, masm);
57         }
58 
59         @Override
60         protected void emitComparison(Constant c) {
61             if (c instanceof HotSpotMetaspaceConstant) {
62                 HotSpotMetaspaceConstant meta = (HotSpotMetaspaceConstant) c;
63                 if (meta.isCompressed()) {
64                     crb.recordInlineDataInCode(meta);
65                     masm.cmpl(keyRegister, 0xDEADDEAD);
66                 } else {
67                     AMD64Address addr = (AMD64Address) crb.recordDataReferenceInCode(meta, 8);
68                     masm.cmpq(keyRegister, addr);
69                 }
70             } else {
71                 super.emitComparison(c);
72             }
73         }
74     }
75 }
