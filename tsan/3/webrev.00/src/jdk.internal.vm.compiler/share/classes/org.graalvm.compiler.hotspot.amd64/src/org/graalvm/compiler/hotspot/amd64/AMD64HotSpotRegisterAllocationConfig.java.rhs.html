 67 class AMD64HotSpotRegisterAllocationConfig extends RegisterAllocationConfig {
 68     /**
 69      * Specify priority of register selection within phases of register allocation. Highest priority
 70      * is first. A useful heuristic is to give registers a low priority when they are required by
 71      * machine instructions, like EAX and EDX on I486, and choose no-save registers before
 72      * save-on-call, & save-on-call before save-on-entry. Registers which participate in fixed
 73      * calling sequences should come last. Registers which are used as pairs must fall on an even
 74      * boundary.
 75      *
 76      * Adopted from x86_64.ad.
 77      */
 78     // @formatter:off
 79     static final Register[] registerAllocationOrder = {
 80         r10, r11, r8, r9, r12, rcx, rbx, rdi, rdx, rsi, rax, rbp, r13, r14, /*r15,*/ /*rsp,*/
 81         xmm0, xmm1, xmm2,  xmm3,  xmm4,  xmm5,  xmm6,  xmm7,
 82         xmm8, xmm9, xmm10, xmm11, xmm12, xmm13, xmm14, xmm15
 83     };
 84     // @formatter:on
 85 
 86     private final boolean useStandardFrameProlog;
 87 
 88     AMD64HotSpotRegisterAllocationConfig(RegisterConfig registerConfig, String[] allocationRestrictedTo, boolean useStandardFrameProlog) {
 89         super(registerConfig, allocationRestrictedTo);
 90         this.useStandardFrameProlog = useStandardFrameProlog;
 91     }
 92 
 93     @Override
 94     protected RegisterArray initAllocatable(RegisterArray registers) {
 95         BitSet regMap = new BitSet(registerConfig.getAllocatableRegisters().size());
 96         for (Register reg : registers) {
 97             regMap.set(reg.number);
 98         }
 99         if (useStandardFrameProlog) {
100             regMap.clear(rbp.number);
101         }
102 
103         ArrayList<Register> allocatableRegisters = new ArrayList<>(registers.size());
104         for (Register reg : registerAllocationOrder) {
105             if (regMap.get(reg.number)) {
106                 allocatableRegisters.add(reg);
107             }
108         }
109 
110         return super.initAllocatable(new RegisterArray(allocatableRegisters));
111     }
112 }
