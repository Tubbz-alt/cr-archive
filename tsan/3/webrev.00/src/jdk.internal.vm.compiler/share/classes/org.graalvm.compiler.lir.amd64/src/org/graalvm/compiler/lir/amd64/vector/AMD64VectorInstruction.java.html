25 package org.graalvm.compiler.lir.amd64.vector;
26 
27 import org.graalvm.compiler.asm.amd64.AVXKind.AVXSize;
28 import org.graalvm.compiler.lir.LIRInstructionClass;
29 import org.graalvm.compiler.lir.amd64.AMD64LIRInstruction;
30 
31 public abstract class AMD64VectorInstruction extends AMD64LIRInstruction {
32 
33     public static final LIRInstructionClass<AMD64VectorInstruction> TYPE = LIRInstructionClass.create(AMD64VectorInstruction.class);
34     protected final AVXSize size;
35 
36     public AMD64VectorInstruction(LIRInstructionClass<? extends AMD64VectorInstruction> c, AVXSize size) {
37         super(c);
38         this.size = size;
39     }
40 
41     @Override
42     public boolean needsClearUpperVectorRegisters() {
43         return size == AVXSize.YMM || size == AVXSize.ZMM;
44     }
45 
46 }
