27 import static jdk.vm.ci.amd64.AMD64.xmm0;
28 import static org.graalvm.compiler.lir.amd64.AMD64HotSpotHelper.registersToValues;

30 import org.graalvm.compiler.core.common.LIRKind;

33 import org.graalvm.compiler.lir.gen.LIRGeneratorTool;

35 import jdk.vm.ci.amd64.AMD64Kind;
36 import jdk.vm.ci.code.Register;
37 import jdk.vm.ci.code.RegisterValue;
38 import jdk.vm.ci.meta.Value;
39 
40 /**
41  * AMD64MathIntrinsicUnaryOp assumes that the input value is stored at the xmm0 register, and will
42  * emit the output value into the xmm0 register as well.
43  * {@link #emitLIRWrapper(LIRGeneratorTool, Value)} is provided for emitting necessary mov LIRs
44  * before and after this LIR instruction.
45  */
46 public abstract class AMD64MathIntrinsicUnaryOp extends AMD64LIRInstruction {
47 
48     @Def protected Value output;
49     @Use protected Value input;
50     @Temp protected Value[] temps;
51 
52     public AMD64MathIntrinsicUnaryOp(LIRInstructionClass<? extends AMD64MathIntrinsicUnaryOp> type, Register... registers) {
53         super(type);
54 
55         input = xmm0.asValue(LIRKind.value(AMD64Kind.DOUBLE));
56         output = xmm0.asValue(LIRKind.value(AMD64Kind.DOUBLE));
57 
58         temps = registersToValues(registers);
59     }
60 
61     public final Variable emitLIRWrapper(LIRGeneratorTool gen, Value value) {
62         LIRKind kind = LIRKind.combine(value);
63         RegisterValue xmm0Value = xmm0.asValue(kind);
64         gen.emitMove(xmm0Value, value);
65         gen.append(this);
66         Variable result = gen.newVariable(kind);
67         gen.emitMove(result, xmm0Value);
68         return result;
69     }
70 
71 }
