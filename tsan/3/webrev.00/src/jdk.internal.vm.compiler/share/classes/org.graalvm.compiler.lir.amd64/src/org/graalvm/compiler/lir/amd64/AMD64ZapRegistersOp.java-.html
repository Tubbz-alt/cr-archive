27 import static org.graalvm.compiler.lir.amd64.AMD64SaveRegistersOp.prune;

29 import jdk.internal.vm.compiler.collections.EconomicSet;

31 import org.graalvm.compiler.lir.LIRInstructionClass;

34 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
35 import org.graalvm.compiler.lir.framemap.FrameMap;
36 
37 import jdk.vm.ci.code.Register;
38 import jdk.vm.ci.code.RegisterSaveLayout;
39 import jdk.vm.ci.meta.JavaConstant;
40 
41 /**
42  * Writes well known garbage values to registers.
43  */
44 @Opcode("ZAP_REGISTER")
45 public final class AMD64ZapRegistersOp extends AMD64LIRInstruction implements SaveRegistersOp {
46     public static final LIRInstructionClass<AMD64ZapRegistersOp> TYPE = LIRInstructionClass.create(AMD64ZapRegistersOp.class);
47 
48     /**
49      * The registers that are zapped.
50      */
51     protected final Register[] zappedRegisters;
52 
53     /**
54      * The garbage values that are written to the registers.
55      */
56     protected final JavaConstant[] zapValues;
57 
58     public AMD64ZapRegistersOp(Register[] zappedRegisters, JavaConstant[] zapValues) {
59         super(TYPE);
60         this.zappedRegisters = zappedRegisters;
61         this.zapValues = zapValues;
62     }
63 
64     @Override
65     public void emitCode(CompilationResultBuilder crb, AMD64MacroAssembler masm) {
66         for (int i = 0; i < zappedRegisters.length; i++) {
67             Register reg = zappedRegisters[i];
68             if (reg != null) {
69                 AMD64Move.const2reg(crb, masm, reg, zapValues[i]);
70             }
71         }
72     }
73 
74     @Override
75     public boolean supportsRemove() {
76         return true;
77     }
78 
79     @Override
80     public int remove(EconomicSet<Register> doNotSave) {
81         return prune(doNotSave, zappedRegisters);
82     }
83 
84     @Override
85     public RegisterSaveLayout getMap(FrameMap frameMap) {
86         return new RegisterSaveLayout(new Register[0], new int[0]);
87     }
88 }
