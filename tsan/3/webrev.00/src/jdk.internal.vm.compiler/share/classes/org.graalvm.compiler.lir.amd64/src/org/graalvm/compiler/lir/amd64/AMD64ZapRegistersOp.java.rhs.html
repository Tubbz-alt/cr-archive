25 package org.graalvm.compiler.lir.amd64;
26 



27 import org.graalvm.compiler.asm.amd64.AMD64MacroAssembler;
28 import org.graalvm.compiler.lir.LIRInstructionClass;
29 import org.graalvm.compiler.lir.Opcode;
30 import org.graalvm.compiler.lir.StandardOp;
31 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;

32 
33 import jdk.vm.ci.amd64.AMD64Kind;
34 import jdk.vm.ci.code.Register;

35 import jdk.vm.ci.meta.JavaConstant;
36 
37 /**
38  * Writes well known garbage values to registers.
39  */
40 @Opcode("ZAP_REGISTER")
41 public final class AMD64ZapRegistersOp extends AMD64LIRInstruction implements StandardOp.ZapRegistersOp {
42     public static final LIRInstructionClass<AMD64ZapRegistersOp> TYPE = LIRInstructionClass.create(AMD64ZapRegistersOp.class);
43 
44     /**
45      * The registers that are zapped.
46      */
47     protected final Register[] zappedRegisters;
48 
49     /**
50      * The garbage values that are written to the registers.
51      */
52     protected final JavaConstant[] zapValues;
53 
54     public AMD64ZapRegistersOp(Register[] zappedRegisters, JavaConstant[] zapValues) {
55         super(TYPE);
56         this.zappedRegisters = zappedRegisters;
57         this.zapValues = zapValues;
58     }
59 
60     @Override
61     public void emitCode(CompilationResultBuilder crb, AMD64MacroAssembler masm) {
62         for (int i = 0; i < zappedRegisters.length; i++) {
63             Register reg = zappedRegisters[i];
64             if (reg != null) {
65                 AMD64Move.const2reg(crb, masm, reg, zapValues[i], AMD64Kind.QWORD);
66             }
67         }
68     }















69 }
