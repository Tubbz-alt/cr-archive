41 package jdk.internal.vm.compiler.collections;
42 
43 /**
44  * Cursor to iterate over a mutable map.
45  *
46  * @since 19.0
47  */
48 public interface MapCursor<K, V> extends UnmodifiableMapCursor<K, V> {
49     /**
50      * Remove the current entry from the map. May only be called once. After calling
51      * {@link #remove()}, it is no longer valid to call {@link #getKey()} or {@link #getValue()} on
52      * the current entry.
53      *
54      * @since 19.0
55      */
56     void remove();
57 }
