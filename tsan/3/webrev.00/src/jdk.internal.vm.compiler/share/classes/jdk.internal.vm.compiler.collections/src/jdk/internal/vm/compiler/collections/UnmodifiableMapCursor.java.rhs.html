41 package jdk.internal.vm.compiler.collections;
42 
43 /**
44  * Cursor to iterate over a map without changing its contents.
45  *
46  * @since 19.0
47  */
48 public interface UnmodifiableMapCursor<K, V> {
49     /**
50      * Advances to the next entry.
51      *
52      * @return {@code true} if a next entry exists, {@code false} if there is no next entry.
53      * @since 19.0
54      */
55     boolean advance();
56 
57     /**
58      * The key of the current entry.
59      *
60      * @since 19.0
61      */
62     K getKey();
63 
64     /**
65      * The value of the current entry.
66      *
67      * @since 19.0
68      */
69     V getValue();
70 }
