 39 /*
 40  * Tests comparison-based canonicalizations for floats.
 41  */
 42 @RunWith(Parameterized.class)
 43 public class TrichotomyFloats extends JTTTest {
 44 
 45     public static int test0(float x, float y) {
 46         return x < y ? -1 : (x == y ? 0 : 1);
 47     }
 48 
 49     public static int test1(float x, float y) {
 50         return x < y ? 1 : (x == y ? 0 : -1);
 51     }
 52 
 53     public static int test2(float x, float y) {
 54         return x == y ? 0 : (x < y ? -1 : 1);
 55     }
 56 
 57     public static int test3(float x, float y) {
 58         return x == y ? 0 : (x < y ? 1 : -1);
 59     }
 60 
 61     public static int test4(float x, float y) {
 62         return x == y ? 0 : (x > y ? -1 : 1);
 63     }
 64 
 65     public static int test5(float x, float y) {
 66         return x == y ? 0 : (x > y ? 1 : -1);
 67     }
 68 
 69     public static int test6(float x, float y) {
 70         return x < y ? 1 : (x > y ? -1 : 0);
 71     }
 72 
 73     public static int test7(float x, float y) {
 74         return x < y ? -1 : (x > y ? 1 : 0);
 75     }
 76 
 77     @Parameter(value = 0) public float x;
 78     @Parameter(value = 1) public float y;
 79 
 80     @Parameters(name = "x = {0}, y = {1}")
 81     public static Collection<Object[]> data() {
 82         List<Object[]> parameters = new ArrayList<>();
 83         float[] floats = {Float.NEGATIVE_INFINITY, Float.POSITIVE_INFINITY, Float.NaN, 0f, -1f, Float.MIN_VALUE, Float.MAX_VALUE};
 84         for (float f1 : floats) {
 85             for (float f2 : floats) {
 86                 parameters.add(new Object[]{f1, f2});
 87             }
 88         }
 89         return parameters;
 90     }
 91 
 92     @Test
 93     public void run0() {
 94         runTest("test0", x, y);
 95     }
 96 
 97     @Test
 98     public void run1() {
 99         runTest("test1", x, y);
100     }
101 
102     @Test
103     public void run2() {
104         runTest("test2", x, y);
105     }
106 
107     @Test
108     public void run3() {
109         runTest("test3", x, y);
110     }
111 
112     @Test
113     public void run4() {
114         runTest("test4", x, y);
115     }
116 
117     @Test
118     public void run5() {
119         runTest("test5", x, y);
120     }
121 
122     @Test
123     public void run6() {
124         runTest("test6", x, y);
125     }
126 
127     @Test
128     public void run7() {
129         runTest("test7", x, y);
130     }
131 }
