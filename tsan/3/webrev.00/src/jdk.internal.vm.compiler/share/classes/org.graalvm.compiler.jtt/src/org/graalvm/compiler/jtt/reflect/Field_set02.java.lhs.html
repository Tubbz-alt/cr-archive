 33 public class Field_set02 extends JTTTest {
 34 
 35     private static class TestClass {
 36         public byte byteField;
 37         public short shortField;
 38         public char charField;
 39         public int intField;
 40         public long longField;
 41         public float floatField;
 42         public double doubleField;
 43         public boolean booleanField;
 44     }
 45 
 46     private static final TestClass object = new TestClass();
 47 
 48     public static boolean test(int arg) throws NoSuchFieldException, IllegalAccessException {
 49         if (arg == 0) {
 50             TestClass.class.getField("byteField").set(object, Byte.valueOf((byte) 11));
 51             return object.byteField == 11;
 52         } else if (arg == 1) {
 53             TestClass.class.getField("shortField").set(object, Short.valueOf((short) 12));
 54             return object.shortField == 12;
 55         } else if (arg == 2) {
 56             TestClass.class.getField("charField").set(object, Character.valueOf((char) 13));
 57             return object.charField == 13;
 58         } else if (arg == 3) {
 59             TestClass.class.getField("intField").set(object, Integer.valueOf(14));
 60             return object.intField == 14;
 61         } else if (arg == 4) {
 62             TestClass.class.getField("longField").set(object, Long.valueOf(15L));
 63             return object.longField == 15;
 64         } else if (arg == 5) {
 65             TestClass.class.getField("floatField").set(object, Float.valueOf(16));
 66             return object.floatField == 16;
 67         } else if (arg == 6) {
 68             TestClass.class.getField("doubleField").set(object, Double.valueOf(17));
 69             return object.doubleField == 17;
 70         } else if (arg == 7) {
 71             TestClass.class.getField("booleanField").set(object, true);
 72             return object.booleanField == true;
 73         }
 74         return false;
 75     }
