25 package org.graalvm.compiler.jtt.lang;
26 
27 import org.graalvm.compiler.jtt.JTTTest;
28 import org.graalvm.compiler.options.OptionValues;

29 
30 import jdk.vm.ci.meta.ResolvedJavaMethod;
31 
32 public abstract class UnaryMath extends JTTTest {
33 
34     private static final long STEP = Long.MAX_VALUE / 1_000_000;
35 
36     /**
37      * Tests a unary {@link Math} method on a wide range of values.
38      */
39     void testManyValues(OptionValues options, ResolvedJavaMethod method) throws AssertionError {
40         if (!Java8OrEarlier) {
41             /*
42              * GR-8276: Allow for variance on JVMCI > 8 until a JVMCI version that includes
43              * https://github.com/graalvm/graal-jvmci-8/commit/
44              * c86fb66f86b8d52a08dd2495d34879d3730f9987 or Graal has stubs that a monotonic with
45              * other HotSpot implementations of these Math routines.
46              */
47             ulpDelta = 2D;
48         } else {
49             /*
50              * Forces the assertion message shows the ulps by which a computed result is wrong.
51              */
52             ulpDelta = 0D;
53         }
54         Object receiver = null;
55         long testIteration = 0;
56         for (long l = Long.MIN_VALUE;; l += STEP) {
57             double d = Double.longBitsToDouble(l);
58             Result expect = executeExpected(method, receiver, d);
59             try {
60                 testAgainstExpected(options, method, expect, EMPTY, receiver, d);
61                 testIteration++;
62             } catch (AssertionError e) {
63                 throw new AssertionError(String.format("%d: While testing %g [long: %d, hex: %x]", testIteration, d, l, l), e);
64             }
65             if (Long.MAX_VALUE - STEP < l) {
66                 break;
67             }
68         }
69     }
70 }
