33 public class System_currentTimeMillis02 extends JTTTest {

35     static void m(long[] times) {
36         times[1] = System.currentTimeMillis() - times[0];
37     }
38 
39     public static boolean test() {
40         long[] times = new long[2];  // { start, delta }
41         times[0] = System.currentTimeMillis();
42         times[1] = 0;
43         // force compilation:
44         for (int i = 0; i < 5000; i++) {
45             m(times);
46         }
47         times[0] = System.currentTimeMillis();
48         times[1] = 0;
49         for (int i = 0; times[1] == 0 && i < 5000000; i++) {
50             m(times);
51             // do nothing.
52         }
53         // better get at least 100 millisecond resolution.
54         return times[1] >= 1 && times[1] < 100;
55     }

59         runTest("test");
60     }

31 /**
32  * Checks that the time between 2 successive calls to {@link System#currentTimeMillis()} is less
33  * than 100 milliseconds at least once in 5_000_000 attempts.
34  */
35 public class System_currentTimeMillis02 extends JTTTest {
36 




37     public static boolean test() {
38         for (int i = 0; i < 5_000_000; i++) {
39             long elapsed = System.currentTimeMillis() - System.currentTimeMillis();
40             if (elapsed < 100) {
41                 return true;
42             }

43         }
44         if (!GraalDirectives.inCompiledCode()) {
45             // We don't care about the result for the interpreter, C1 or C2
46             return true;


47         }
48         return false;

49     }
50 
51     @Test
52     public void run0() throws Throwable {
53         runTest("test");
54     }
55 
56 }
