23 package jdk.vm.ci.hotspot.aarch64;
24 
25 import jdk.vm.ci.hotspot.HotSpotVMConfigAccess;
26 import jdk.vm.ci.hotspot.HotSpotVMConfigStore;
27 import jdk.vm.ci.services.Services;
28 
29 /**
30  * Used to access native configuration details.
31  *
32  * All non-static, public fields in this class are so that they can be compiled as constants.
33  */
34 class AArch64HotSpotVMConfig extends HotSpotVMConfigAccess {
35 
36     AArch64HotSpotVMConfig(HotSpotVMConfigStore config) {
37         super(config);
38     }
39 
40     final boolean linuxOs = Services.getSavedProperty("os.name", "").startsWith("Linux");
41 
42     final boolean useCompressedOops = getFlag("UseCompressedOops", Boolean.class);
43 
44     // CPU Capabilities
45 
46     /*
47      * These flags are set based on the corresponding command line flags.
48      */
49     final boolean useBarriersForVolatile = getFlag("UseBarriersForVolatile", Boolean.class);
50     final boolean useCRC32 = getFlag("UseCRC32", Boolean.class);
51     final boolean useNeon = getFlag("UseNeon", Boolean.class);
52     final boolean useSIMDForMemoryOps = getFlag("UseSIMDForMemoryOps", Boolean.class);
53     final boolean avoidUnalignedAccesses = getFlag("AvoidUnalignedAccesses", Boolean.class);
54     final boolean useLSE = getFlag("UseLSE", Boolean.class);
55     final boolean useBlockZeroing = getFlag("UseBlockZeroing", Boolean.class);
56 
57     final long vmVersionFeatures = getFieldValue("Abstract_VM_Version::_features", Long.class, "uint64_t");
58 
59     /*
60      * These flags are set if the corresponding support is in the hardware.
61      */
62     // Checkstyle: stop
63     final long aarch64FP = getConstant("VM_Version::CPU_FP", Long.class);
64     final long aarch64ASIMD = getConstant("VM_Version::CPU_ASIMD", Long.class);
65     final long aarch64EVTSTRM = getConstant("VM_Version::CPU_EVTSTRM", Long.class);
66     final long aarch64AES = getConstant("VM_Version::CPU_AES", Long.class);
67     final long aarch64PMULL = getConstant("VM_Version::CPU_PMULL", Long.class);
68     final long aarch64SHA1 = getConstant("VM_Version::CPU_SHA1", Long.class);
69     final long aarch64SHA2 = getConstant("VM_Version::CPU_SHA2", Long.class);
70     final long aarch64CRC32 = getConstant("VM_Version::CPU_CRC32", Long.class);
71     final long aarch64LSE = getConstant("VM_Version::CPU_LSE", Long.class);
72     final long aarch64STXR_PREFETCH = getConstant("VM_Version::CPU_STXR_PREFETCH", Long.class);
73     final long aarch64A53MAC = getConstant("VM_Version::CPU_A53MAC", Long.class);
74     final long aarch64DMB_ATOMICS = getConstant("VM_Version::CPU_DMB_ATOMICS", Long.class);
75     // Checkstyle: resume
76 }
