23 package jdk.vm.ci.runtime;
24 
25 import java.util.Formatter;
26 
27 import jdk.vm.ci.common.NativeImageReinitialize;
28 import jdk.vm.ci.services.Services;
29 
30 public class JVMCI {
31 
32     /**
33      * Singleton instance lazily initialized via double-checked locking.
34      */
35     @NativeImageReinitialize private static volatile JVMCIRuntime runtime;
36 
37     @NativeImageReinitialize private static boolean initializing;
38 
39     public static void initialize() {
40         // force static initializer
41     }
42 
43     private static native JVMCIRuntime initializeRuntime();
44 
45     /**
46      * Gets the singleton {@link JVMCIRuntime} instance available to the application.
47      *
48      * @throws UnsupportedOperationException if JVMCI is not supported
49      */
50     public static JVMCIRuntime getRuntime() {
51         JVMCIRuntime result = runtime;
52         if (result == null) {
53             synchronized (JVMCI.class) {
54                 result = runtime;
55                 if (result == null) {
56                     if (initializing) {
57                         // In recursive call from HotSpotJVMCIRuntime.runtime
58                         // so no need to re-enter initializeRuntime below. This
59                         // path is only entered if JVMCI initialization starts
60                         // with JVMCI.getRuntime().
61                         return null;
62                     }
63                     initializing = true;
64                     try {
65                         runtime = result = initializeRuntime();
66                     } catch (UnsatisfiedLinkError e) {
67                         String javaHome = Services.getSavedProperty("java.home");
68                         String vmName = Services.getSavedProperty("java.vm.name");
69                         Formatter errorMessage = new Formatter();
70                         errorMessage.format("The VM does not support the JVMCI API.%n");
71                         errorMessage.format("Currently used Java home directory is %s.%n", javaHome);
72                         errorMessage.format("Currently used VM configuration is: %s", vmName);
73                         throw new UnsupportedOperationException(errorMessage.toString());
74                     } finally {
75                         initializing = false;
76                     }
77                 }
78             }
79         }
80         return result;
81     }
82 }
