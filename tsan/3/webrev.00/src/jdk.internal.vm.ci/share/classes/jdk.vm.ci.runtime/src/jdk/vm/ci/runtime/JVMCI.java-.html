23 package jdk.vm.ci.runtime;
24 
25 import java.util.Formatter;
26 
27 public class JVMCI {
28 
29     private static final JVMCIRuntime runtime;
30 
31     private static native JVMCIRuntime initializeRuntime();
32 
33     public static void initialize() {
34         // force static initializer
35     }
36 
37     /**
38      * Gets the singleton {@link JVMCIRuntime} instance available to the application.
39      *
40      * @throws UnsupportedOperationException if JVMCI is not supported
41      */
42     public static JVMCIRuntime getRuntime() {
43         if (runtime == null) {
44             String javaHome = System.getProperty("java.home");
45             String vmName = System.getProperty("java.vm.name");
46             Formatter errorMessage = new Formatter();
47             errorMessage.format("The VM does not support the JVMCI API.%n");
48             errorMessage.format("Currently used Java home directory is %s.%n", javaHome);
49             errorMessage.format("Currently used VM configuration is: %s", vmName);
50             throw new UnsupportedOperationException(errorMessage.toString());
51         }
52         return runtime;
53     }
54 
55     static {
56         JVMCIRuntime rt = null;
57         try {
58             rt = initializeRuntime();
59         } catch (UnsatisfiedLinkError e) {
60         }
61         runtime = rt;
62     }
63 }
