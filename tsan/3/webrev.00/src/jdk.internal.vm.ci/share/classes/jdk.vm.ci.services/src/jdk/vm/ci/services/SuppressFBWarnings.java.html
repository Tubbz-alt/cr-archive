25 /**
26  * Used to suppress <a href="http://findbugs.sourceforge.net">FindBugs</a> warnings.
27  */
28 @interface SuppressFBWarnings {
29     /**
30      * The set of FindBugs
31      * <a href="http://findbugs.sourceforge.net/bugDescriptions.html">warnings</a> that are to be
32      * suppressed in annotated element. The value can be a bug category, kind or pattern.
33      */
34     String[] value();
35 
36     /**
37      * Reason why the warning is suppressed.
38      */
39     String justification();
40 }
