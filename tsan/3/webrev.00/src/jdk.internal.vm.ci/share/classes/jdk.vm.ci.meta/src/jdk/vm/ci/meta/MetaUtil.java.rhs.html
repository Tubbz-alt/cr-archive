 30     /**
 31      * Extends the functionality of {@link Class#getSimpleName()} to include a non-empty string for
 32      * anonymous and local classes.
 33      *
 34      * @param clazz the class for which the simple name is being requested
 35      * @param withEnclosingClass specifies if the returned name should be qualified with the name(s)
 36      *            of the enclosing class/classes of {@code clazz} (if any). This option is ignored
 37      *            if {@code clazz} denotes an anonymous or local class.
 38      * @return the simple name
 39      */
 40     public static String getSimpleName(Class<?> clazz, boolean withEnclosingClass) {
 41         final String simpleName = safeSimpleName(clazz);
 42         if (simpleName.length() != 0) {
 43             if (withEnclosingClass) {
 44                 String prefix = "";
 45                 Class<?> enclosingClass = clazz;
 46                 while ((enclosingClass = enclosingClass.getEnclosingClass()) != null) {
 47                     prefix = safeSimpleName(enclosingClass) + "." + prefix;
 48                 }
 49                 return prefix + simpleName;
 50             }
 51             return simpleName;
 52         }
 53         // Must be an anonymous or local class
 54         final String name = clazz.getName();
 55         int index = name.indexOf('$');
 56         if (index == -1) {
 57             return name;
 58         }
 59         index = name.lastIndexOf('.', index);
 60         if (index == -1) {
 61             return name;
 62         }
 63         return name.substring(index + 1);
 64     }
 65 
 66     private static String safeSimpleName(Class<?> clazz) {
 67         try {
 68             return clazz.getSimpleName();
 69         } catch (InternalError e) {
 70             // Scala inner class names do not always start with '$',
 71             // causing Class.getSimpleName to throw an InternalError
 72             Class<?> enclosingClass = clazz.getEnclosingClass();
 73             String fqn = clazz.getName();
 74             if (enclosingClass == null) {
 75                 // Should never happen given logic in
 76                 // Class.getSimpleName but best be safe
 77                 return fqn;
 78             }
 79             String enclosingFQN = enclosingClass.getName();
 80             int length = fqn.length();
 81             if (enclosingFQN.length() >= length) {
 82                 // Should also never happen
 83                 return fqn;
 84             }
 85             return fqn.substring(enclosingFQN.length());
 86         }
 87     }

 89     /**
 90      * Classes for lambdas can have {@code /} characters that are not package separators. These are
 91      * distinguished by being followed by a character that is not a
 92      * {@link Character#isJavaIdentifierStart(char)} (e.g.,
 93      * "jdk.vm.ci.runtime.test.TypeUniverse$$Lambda$1/869601985").
 94      */
 95     private static String replacePackageSeparatorsWithDot(String name) {
 96         int length = name.length();
 97         int i = 0;
 98         StringBuilder buf = new StringBuilder(length);
 99         while (i < length - 1) {
100             char ch = name.charAt(i);
101             if (ch == '/' && Character.isJavaIdentifierStart(name.charAt(i + 1))) {
102                 buf.append('.');
103             } else {
104                 buf.append(ch);
105             }
106             i++;
107         }
108         buf.append(name.charAt(length - 1));
109         return buf.toString();
110     }

112     /**
113      * Converts a type name in internal form to an external form.
114      *
115      * @param name the internal name to convert
116      * @param qualified whether the returned name should be qualified with the package name
117      * @param classForNameCompatible specifies if the returned name for array types should be in
118      *            {@link Class#forName(String)} format (e.g., {@code "[Ljava.lang.Object;"},
119      *            {@code "[[I"}) or in Java source code format (e.g., {@code "java.lang.Object[]"},
120      *            {@code "int[][]"} ).
121      */
122     public static String internalNameToJava(String name, boolean qualified, boolean classForNameCompatible) {
123         switch (name.charAt(0)) {
124             case 'L': {
125                 String result = replacePackageSeparatorsWithDot(name.substring(1, name.length() - 1));
126                 if (!qualified) {
127                     final int lastDot = result.lastIndexOf('.');
128                     if (lastDot != -1) {
129                         result = result.substring(lastDot + 1);
130                     }
131                 }
132                 return result;
133             }
134             case '[':
135                 return classForNameCompatible ? replacePackageSeparatorsWithDot(name) : internalNameToJava(name.substring(1), qualified, classForNameCompatible) + "[]";
136             default:
137                 if (name.length() != 1) {
138                     throw new IllegalArgumentException("Illegal internal name: " + name);
139                 }
140                 return JavaKind.fromPrimitiveOrVoidTypeChar(name.charAt(0)).getJavaName();
141         }
142     }

144     /**
145      * Convenient shortcut for calling
146      * {@link #appendLocation(StringBuilder, ResolvedJavaMethod, int)} without having to supply a
147      * {@link StringBuilder} instance and convert the result to a string.
148      */
149     public static String toLocation(ResolvedJavaMethod method, int bci) {
150         return appendLocation(new StringBuilder(), method, bci).toString();
151     }

153     /**
154      * Appends a string representation of a location specified by a given method and bci to a given
155      * {@link StringBuilder}. If a stack trace element with a non-null file name and non-negative
156      * line number is {@linkplain ResolvedJavaMethod#asStackTraceElement(int) available} for the
157      * given method, then the string returned is the {@link StackTraceElement#toString()} value of
158      * the stack trace element, suffixed by the bci location. For example:
159      *
160      * <pre>
161      *     java.lang.String.valueOf(String.java:2930) [bci: 12]
162      * </pre>
163      *
164      * Otherwise, the string returned is the value of applying {@link JavaMethod#format(String)}
165      * with the format string {@code "%H.%n(%p)"}, suffixed by the bci location. For example:
166      *
167      * <pre>
168      *     java.lang.String.valueOf(int) [bci: 12]
169      * </pre>
170      *
171      * @param sb
172      * @param method
173      * @param bci
174      */
175     public static StringBuilder appendLocation(StringBuilder sb, ResolvedJavaMethod method, int bci) {
176         if (method != null) {
177             StackTraceElement ste = method.asStackTraceElement(bci);
178             if (ste.getFileName() != null && ste.getLineNumber() > 0) {
179                 sb.append(ste);
180             } else {
181                 sb.append(method.format("%H.%n(%p)"));
182             }
183         } else {
184             sb.append("Null method");
185         }
186         return sb.append(" [bci: ").append(bci).append(']');
187     }

189     static void appendProfile(StringBuilder buf, AbstractJavaProfile<?, ?> profile, int bci, String type, String sep) {
190         if (profile != null) {
191             AbstractProfiledItem<?>[] pitems = profile.getItems();
192             if (pitems != null) {
193                 buf.append(String.format("%s@%d:", type, bci));
194                 for (int j = 0; j < pitems.length; j++) {
195                     AbstractProfiledItem<?> pitem = pitems[j];
196                     buf.append(String.format(" %.6f (%s)%s", pitem.getProbability(), pitem.getItem(), sep));
197                 }
198                 if (profile.getNotRecordedProbability() != 0) {
199                     buf.append(String.format(" %.6f <other %s>%s", profile.getNotRecordedProbability(), type, sep));
200                 } else {
201                     buf.append(String.format(" <no other %s>%s", type, sep));
202                 }
203             }
204         }
205     }

207     /**
208      * Converts a Java source-language class name into the internal form.
209      *
210      * @param className the class name
211      * @return the internal name form of the class name
212      */
213     public static String toInternalName(String className) {
214         if (className.startsWith("[")) {
215             /* Already in the correct array style. */
216             return className.replace('.', '/');
217         }
218 
219         StringBuilder result = new StringBuilder();
220         String base = className;
221         while (base.endsWith("[]")) {
222             result.append("[");
223             base = base.substring(0, base.length() - 2);
224         }
225 
226         switch (base) {
227             case "boolean":
228                 result.append("Z");
229                 break;
230             case "byte":
231                 result.append("B");
232                 break;
233             case "short":
234                 result.append("S");
235                 break;
236             case "char":
237                 result.append("C");
238                 break;
239             case "int":
240                 result.append("I");
241                 break;
242             case "float":
243                 result.append("F");
244                 break;
245             case "long":
246                 result.append("J");
247                 break;
248             case "double":
249                 result.append("D");
250                 break;
251             case "void":
252                 result.append("V");
253                 break;
254             default:
255                 result.append("L").append(base.replace('.', '/')).append(";");
256                 break;
257         }
258         return result.toString();
259     }

261     /**
262      * Gets a string representation of an object based soley on its class and its
263      * {@linkplain System#identityHashCode(Object) identity hash code}. This avoids and calls to
264      * virtual methods on the object such as {@link Object#hashCode()}.
265      */
266     public static String identityHashCodeString(Object obj) {
267         if (obj == null) {
268             return "null";
269         }
270         return obj.getClass().getName() + "@" + System.identityHashCode(obj);
271     }
