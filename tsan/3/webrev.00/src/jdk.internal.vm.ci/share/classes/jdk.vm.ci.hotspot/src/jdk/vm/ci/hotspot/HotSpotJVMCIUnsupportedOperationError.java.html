23 package jdk.vm.ci.hotspot;
24 
25 import jdk.vm.ci.meta.ResolvedJavaType;
26 import jdk.vm.ci.services.Services;
27 
28 /**
29  * Indicates a path in HotSpot JVMCI related code that is unsupported in the current execution
30  * environment. For example, certain operations are not supported by JVMCI code running in an ahead
31  * of time compiled {@linkplain Services#IS_IN_NATIVE_IMAGE native image}. This usually reflects
32  * functionality only needed in non-native image execution that would require a more complex
33  * implementation to support in a native image.
34  *
35  * An example of such functionality is {@link ResolvedJavaType#isLocal()}. This can be conveniently
36  * implemented when JVMCI is running on the HotSpot heap as we can obtain the {@link Class} mirror
37  * for the {@link ResolvedJavaType} and call {@link Class#isLocalClass()}. In a native image, there
38  * is no {@link Class} mirror available in the native image heap so implementing this would involve
39  * a call into VM native code that in turn would make an upcall into Java code executing on the
40  * HotSpot heap. We have opted to defer implementing functionality such as this until there's a
41  * demonstrated need for it.
42  */
43 public class HotSpotJVMCIUnsupportedOperationError extends Error {
44 
45     public HotSpotJVMCIUnsupportedOperationError(String reason) {
46         super(reason);
47     }
48 
49     public HotSpotJVMCIUnsupportedOperationError(String reason, Throwable cause) {
50         super(reason, cause);
51     }
52 
53     private static final long serialVersionUID = 7782431672678016392L;
54 
55 }
