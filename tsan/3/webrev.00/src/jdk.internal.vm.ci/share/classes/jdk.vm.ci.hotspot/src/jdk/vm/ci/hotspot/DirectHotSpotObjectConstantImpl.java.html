23 package jdk.vm.ci.hotspot;
24 
25 import jdk.vm.ci.meta.JavaConstant;
26 
27 final class DirectHotSpotObjectConstantImpl extends HotSpotObjectConstantImpl {
28 
29     static JavaConstant forObject(Object object, boolean compressed) {
30         if (object == null) {
31             return compressed ? HotSpotCompressedNullConstant.COMPRESSED_NULL : JavaConstant.NULL_POINTER;
32         } else {
33             return new DirectHotSpotObjectConstantImpl(object, compressed);
34         }
35     }
36 
37     static HotSpotObjectConstantImpl forNonNullObject(Object object, boolean compressed) {
38         if (object == null) {
39             throw new NullPointerException();
40         }
41         return new DirectHotSpotObjectConstantImpl(object, compressed);
42     }
43 
44     private DirectHotSpotObjectConstantImpl(Object object, boolean compressed) {
45         super(compressed);
46         assert object != null;
47         this.object = object;
48     }
49 
50     final Object object;
51 
52     @Override
53     public JavaConstant compress() {
54         assert !compressed;
55         return new DirectHotSpotObjectConstantImpl(object, true);
56     }
57 
58     @Override
59     public JavaConstant uncompress() {
60         assert compressed;
61         return new DirectHotSpotObjectConstantImpl(object, false);
62     }
63 
64     @Override
65     public int getIdentityHashCode() {
66         return System.identityHashCode(object);
67     }
68 }
