23 package jdk.vm.ci.hotspot;
24 
25 import jdk.vm.ci.code.InstalledCode;
26 import jdk.vm.ci.code.InvalidInstalledCodeException;
27 import jdk.vm.ci.meta.ResolvedJavaMethod;
28 
29 /**
30  * Implementation of {@link InstalledCode} for code installed as a RuntimeStub.


31  */
32 public class HotSpotRuntimeStub extends HotSpotInstalledCode {
33 
34     public HotSpotRuntimeStub(String name) {
35         super(name);
36     }
37 
38     public ResolvedJavaMethod getMethod() {
39         return null;
40     }
41 
42     @Override
43     public boolean isValid() {
44         return true;
45     }
46 
47     @Override
48     public void invalidate() {
49     }
50 
51     @Override
52     public String toString() {
53         return String.format("InstalledRuntimeStub[stub=%s, codeBlob=0x%x]", name, getAddress());
54     }
55 
56     @Override
57     public Object executeVarargs(Object... args) throws InvalidInstalledCodeException {
58         throw new InternalError("Cannot call stub " + name);
59     }
60 }
