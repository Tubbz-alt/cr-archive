 23 package jdk.vm.ci.hotspot;
 24 
 25 import jdk.vm.ci.code.BytecodeFrame;
 26 import jdk.vm.ci.code.CompiledCode;
 27 import jdk.vm.ci.code.StackSlot;
 28 import jdk.vm.ci.code.VirtualObject;
 29 import jdk.vm.ci.code.site.DataPatch;
 30 import jdk.vm.ci.code.site.Infopoint;
 31 import jdk.vm.ci.code.site.Site;
 32 import jdk.vm.ci.meta.Assumptions.Assumption;
 33 import jdk.vm.ci.meta.ResolvedJavaField;
 34 import jdk.vm.ci.meta.ResolvedJavaMethod;
 35 
 36 /**
 37  * A {@link CompiledCode} with additional HotSpot-specific information required for installing the
 38  * code in HotSpot's code cache.
 39  */
 40 public class HotSpotCompiledCode implements CompiledCode {
 41 
 42     /**
 43      * The name of this compilation unit.
 44      */
 45     protected final String name;
 46 
 47     /**
 48      * The buffer containing the emitted machine code.
 49      */
 50     protected final byte[] targetCode;
 51 
 52     /**
 53      * The leading number of bytes in {@link #targetCode} containing the emitted machine code.
 54      */
 55     protected final int targetCodeSize;
 56 
 57     /**
 58      * A list of code annotations describing special sites in {@link #targetCode}.
 59      */
 60     protected final Site[] sites;
 61 
 62     /**
 63      * A list of {@link Assumption} this code relies on.
 64      */
 65     protected final Assumption[] assumptions;
 66 
 67     /**
 68      * The list of the methods whose bytecodes were used as input to the compilation. If
 69      * {@code null}, then the compilation did not record method dependencies. Otherwise, the first
 70      * element of this array is the root method of the compilation.
 71      */
 72     protected final ResolvedJavaMethod[] methods;
 73 
 74     /**
 75      * A list of comments that will be included in code dumps.
 76      */
 77     protected final Comment[] comments;
 78 
 79     /**
 80      * The data section containing serialized constants for the emitted machine code.
 81      */
 82     protected final byte[] dataSection;
 83 
 84     /**
 85      * The minimum alignment of the data section.
 86      */
 87     protected final int dataSectionAlignment;
 88 
 89     /**
 90      * A list of relocations in the {@link #dataSection}.
 91      */
 92     protected final DataPatch[] dataSectionPatches;
 93 
 94     /**
 95      * A flag determining whether this code is immutable and position independent.
 96      */
 97     protected final boolean isImmutablePIC;
 98 
 99     /**
100      * The total size of the stack frame of this compiled method.
101      */
102     protected final int totalFrameSize;
103 
104     /**
105      * The deopt rescue slot. Must be non-null if there is a safepoint in the method.
106      */
107     protected final StackSlot deoptRescueSlot;
108 
109     public static class Comment {
110 
111         public final String text;
112         public final int pcOffset;
113 
114         public Comment(int pcOffset, String text) {
115             this.text = text;
116             this.pcOffset = pcOffset;
117         }
118     }
119 
120     @SuppressFBWarnings(value = "EI_EXPOSE_REP2", justification = "caller transfers ownership of `sites`, `targetCode`, `comments`, `methods`, `dataSection`, `dataSectionPatches` and `assumptions`")
121     public HotSpotCompiledCode(String name, byte[] targetCode, int targetCodeSize, Site[] sites, Assumption[] assumptions, ResolvedJavaMethod[] methods, Comment[] comments, byte[] dataSection,
122                     int dataSectionAlignment, DataPatch[] dataSectionPatches, boolean isImmutablePIC, int totalFrameSize, StackSlot deoptRescueSlot) {
123         this.name = name;
124         this.targetCode = targetCode;
125         this.targetCodeSize = targetCodeSize;
126         this.sites = sites;
127         this.assumptions = assumptions;
128         this.methods = methods;
129 
130         this.comments = comments;
131         this.dataSection = dataSection;
132         this.dataSectionAlignment = dataSectionAlignment;
133         this.dataSectionPatches = dataSectionPatches;
134         this.isImmutablePIC = isImmutablePIC;
135         this.totalFrameSize = totalFrameSize;
136         this.deoptRescueSlot = deoptRescueSlot;
137 
138         assert validateFrames();
139     }
140 
141     public String getName() {
142         return name;
143     }
144 
145     @Override
146     public String toString() {
147         return name;
148     }
149 
150     /**
151      * Ensure that all the frames passed into the VM are properly formatted with an empty or illegal
152      * slot following double word slots.
153      */
154     private boolean validateFrames() {
155         for (Site site : sites) {
156             if (site instanceof Infopoint) {
157                 Infopoint info = (Infopoint) site;
158                 if (info.debugInfo != null) {
159                     BytecodeFrame frame = info.debugInfo.frame();
160                     assert frame == null || frame.validateFormat();
161                     if (info.debugInfo.getVirtualObjectMapping() != null) {
162                         for (VirtualObject v : info.debugInfo.getVirtualObjectMapping()) {
163                             verifyVirtualObject(v);
164                         }
165                     }
166                 }
167             }
168         }
169         return true;
170     }
171 
172     public static void verifyVirtualObject(VirtualObject v) {
173         v.verifyLayout(new VirtualObject.LayoutVerifier() {
174             @Override
175             public int getOffset(ResolvedJavaField field) {
176                 return field.getOffset();
177             }
178         });
179     }
180 }
