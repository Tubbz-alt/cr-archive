23 package jdk.vm.ci.hotspot;
24 
25 /**
26  * A tag interface indicating that this type is a handledized wrapper around a HotSpot metaspace
27  * object that requires GC interaction to keep alive.
28  *
29  * It would preferable if this were the base class containing the pointer but that would require
30  * mixins since most of the wrapper types have complex supertype hierarchies.
31  */
32 interface MetaspaceHandleObject extends MetaspaceObject {
33 
34     long getMetadataHandle();
35 
36     @Override
37     default long getMetaspacePointer() {
38         return UnsafeAccess.UNSAFE.getLong(getMetadataHandle());
39     }
40 }
