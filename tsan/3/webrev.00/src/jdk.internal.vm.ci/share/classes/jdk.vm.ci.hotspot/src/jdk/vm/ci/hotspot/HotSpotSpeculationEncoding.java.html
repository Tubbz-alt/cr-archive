 23 package jdk.vm.ci.hotspot;
 24 
 25 import java.io.ByteArrayOutputStream;
 26 import java.io.DataOutputStream;
 27 import java.io.IOException;
 28 import java.security.MessageDigest;
 29 import java.security.NoSuchAlgorithmException;
 30 import java.util.Arrays;
 31 
 32 import jdk.vm.ci.meta.ResolvedJavaMethod;
 33 import jdk.vm.ci.meta.ResolvedJavaType;
 34 import jdk.vm.ci.meta.SpeculationLog.SpeculationReasonEncoding;
 35 
 36 /**
 37  * Implements a {@link SpeculationReasonEncoding} that {@linkplain #getByteArray() produces} a byte
 38  * array. Data is added via a {@link DataOutputStream}. When producing the final byte array, if the
 39  * total length of data exceeds the length of a SHA-1 digest and a SHA-1 digest algorithm is
 40  * available, then a SHA-1 digest of the data is produced instead.
 41  */
 42 final class HotSpotSpeculationEncoding extends ByteArrayOutputStream implements SpeculationReasonEncoding {
 43 
 44     private DataOutputStream dos = new DataOutputStream(this);
 45     private byte[] result;
 46 
 47     HotSpotSpeculationEncoding() {
 48         super(SHA1_LENGTH);
 49     }
 50 
 51     private void checkOpen() {
 52         if (result != null) {
 53             throw new IllegalArgumentException("Cannot update closed speculation encoding");
 54         }
 55     }
 56 
 57     private static final int NULL_METHOD = -1;
 58     private static final int NULL_TYPE = -2;
 59     private static final int NULL_STRING = -3;
 60 
 61     @Override
 62     public void addByte(int value) {
 63         checkOpen();
 64         try {
 65             dos.writeByte(value);
 66         } catch (IOException e) {
 67             throw new InternalError(e);
 68         }
 69     }
 70 
 71     @Override
 72     public void addShort(int value) {
 73         checkOpen();
 74         try {
 75             dos.writeShort(value);
 76         } catch (IOException e) {
 77             throw new InternalError(e);
 78         }
 79     }
 80 
 81     @Override
 82     public void addMethod(ResolvedJavaMethod method) {
 83         if (!addNull(method, NULL_METHOD)) {
 84             checkOpen();
 85             if (method instanceof HotSpotResolvedJavaMethodImpl) {
 86                 try {
 87                     dos.writeLong(((HotSpotResolvedJavaMethodImpl) method).getMetaspaceMethod());
 88                 } catch (IOException e) {
 89                     throw new InternalError(e);
 90                 }
 91             } else {
 92                 throw new IllegalArgumentException("Cannot encode unsupported type " + method.getClass().getName() + ": " + method.format("%H.%n(%p)"));
 93             }
 94         }
 95     }
 96 
 97     @Override
 98     public void addType(ResolvedJavaType type) {
 99         if (!addNull(type, NULL_TYPE)) {
100             checkOpen();
101             if (type instanceof HotSpotResolvedObjectTypeImpl) {
102                 try {
103                     dos.writeLong(((HotSpotResolvedObjectTypeImpl) type).getMetaspaceKlass());
104                 } catch (IOException e) {
105                     throw new InternalError(e);
106                 }
107             } else {
108                 throw new IllegalArgumentException("Cannot encode unsupported type " + type.getClass().getName() + ": " + type.toClassName());
109             }
110         }
111     }
112 
113     @Override
114     public void addString(String value) {
115         if (!addNull(value, NULL_STRING)) {
116             checkOpen();
117             try {
118                 dos.writeChars(value);
119             } catch (IOException e) {
120                 throw new InternalError(e);
121             }
122         }
123     }
124 
125     @Override
126     public void addInt(int value) {
127         checkOpen();
128         try {
129             dos.writeInt(value);
130         } catch (IOException e) {
131             throw new InternalError(e);
132         }
133     }
134 
135     @Override
136     public void addLong(long value) {
137         checkOpen();
138         try {
139             dos.writeLong(value);
140         } catch (IOException e) {
141             throw new InternalError(e);
142         }
143     }
144 
145     private boolean addNull(Object o, int nullValue) {
146         if (o == null) {
147             addInt(nullValue);
148             return true;
149         }
150         return false;
151     }
152 
153     /**
154      * Prototype SHA1 digest.
155      */
156     private static final MessageDigest SHA1;
157 
158     /**
159      * Cloning the prototype is quicker than calling {@link MessageDigest#getInstance(String)} every
160      * time.
161      */
162     private static final boolean SHA1_IS_CLONEABLE;
163     private static final int SHA1_LENGTH;
164 
165     static {
166         MessageDigest sha1 = null;
167         boolean sha1IsCloneable = false;
168         try {
169             sha1 = MessageDigest.getInstance("SHA-1");
170             sha1.clone();
171             sha1IsCloneable = true;
172         } catch (NoSuchAlgorithmException e) {
173             // Should never happen given that SHA-1 is mandated in a
174             // compliant Java platform implementation. However, be
175             // conservative and fall back to not using a digest.
176         } catch (CloneNotSupportedException e) {
177         }
178         SHA1 = sha1;
179         SHA1_IS_CLONEABLE = sha1IsCloneable;
180         SHA1_LENGTH = SHA1 == null ? 20 : SHA1.getDigestLength();
181     }
182 
183     /**
184      * Gets the final encoded byte array and closes this encoding such that any further attempts to
185      * update it result in an {@link IllegalArgumentException}.
186      */
187     byte[] getByteArray() {
188         if (result == null) {
189             if (SHA1 != null && count > SHA1_LENGTH) {
190                 try {
191                     MessageDigest md = SHA1_IS_CLONEABLE ? (MessageDigest) SHA1.clone() : MessageDigest.getInstance("SHA-1");
192                     md.update(buf, 0, count);
193                     result = md.digest();
194                 } catch (CloneNotSupportedException | NoSuchAlgorithmException e) {
195                     throw new InternalError(e);
196                 }
197             } else {
198                 if (buf.length == count) {
199                     // No need to copy the byte array
200                     return buf;
201                 }
202                 result = Arrays.copyOf(buf, count);
203             }
204             dos = null;
205         }
206         return result;
207     }
208 }
