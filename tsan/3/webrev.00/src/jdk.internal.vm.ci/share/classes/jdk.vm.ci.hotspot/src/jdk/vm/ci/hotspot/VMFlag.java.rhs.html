23 package jdk.vm.ci.hotspot;
24 
25 /**
26  * Describes a VM flag exposed via {@link HotSpotVMConfigAccess}.
27  */
28 public final class VMFlag {
29 
30     /**
31      * The name of the flag.
32      */
33     public final String name;
34 
35     /**
36      * The C++ type of the flag.
37      */
38     public final String type;
39 
40     /**
41      * The flag's value.
42      */
43     public final Object value;
44 
45     @VMEntryPoint
46     VMFlag(String name, String type, Object value) {
47         this.name = name;
48         this.type = type;
49         this.value = value;
50     }
51 
52     @Override
53     public String toString() {
54         return String.format("Flag[type=%s, name=%s, value=%s]", type, name, value);
55     }
56 }
