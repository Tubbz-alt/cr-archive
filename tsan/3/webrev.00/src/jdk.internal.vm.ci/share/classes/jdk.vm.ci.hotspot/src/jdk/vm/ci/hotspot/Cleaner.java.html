 23 package jdk.vm.ci.hotspot;
 24 
 25 import java.lang.ref.ReferenceQueue;
 26 import java.lang.ref.WeakReference;
 27 
 28 import jdk.vm.ci.common.NativeImageReinitialize;
 29 import jdk.vm.ci.meta.ResolvedJavaType;
 30 
 31 /**
 32  * A cleaner tracks a referent object and includes some {@linkplain #doCleanup() cleanup code} that
 33  * is run some time after the referent object has become weakly-reachable.
 34  *
 35  * This is like {@link java.lang.ref.Cleaner} but with weak semantics instead of phantom. Objects
 36  * referenced by this might be referenced by {@link ResolvedJavaType} which is kept alive by a
 37  * {@link WeakReference} so we need equivalent reference strength.
 38  */
 39 abstract class Cleaner extends WeakReference<Object> {
 40 
 41     /**
 42      * Head of linked list of cleaners.
 43      */
 44     @NativeImageReinitialize private static Cleaner first;
 45 
 46     /**
 47      * Linked list pointers.
 48      */
 49     private Cleaner next = null;
 50     private Cleaner prev = null;
 51 
 52     Cleaner(Object referent) {
 53         super(referent, queue);
 54         add(this);
 55     }
 56 
 57     private static synchronized Cleaner add(Cleaner cl) {
 58         if (first != null) {
 59             clean();
 60         }
 61         if (first != null) {
 62             cl.next = first;
 63             first.prev = cl;
 64         }
 65         first = cl;
 66         return cl;
 67     }
 68 
 69     /**
 70      * Removes {@code cl} from the linked list of cleaners.
 71      */
 72     private static synchronized void remove(Cleaner cl) {
 73         // If already removed, do nothing
 74         if (cl.next == cl) {
 75             return;
 76         }
 77 
 78         // Update list
 79         if (first == cl) {
 80             if (cl.next != null) {
 81                 first = cl.next;
 82             } else {
 83                 first = cl.prev;
 84             }
 85         }
 86         if (cl.next != null) {
 87             cl.next.prev = cl.prev;
 88         }
 89         if (cl.prev != null) {
 90             cl.prev.next = cl.next;
 91         }
 92 
 93         // Indicate removal by pointing the cleaner to itself
 94         cl.next = cl;
 95         cl.prev = cl;
 96     }
 97 
 98     /**
 99      * Performs the cleanup action now that this object's referent has become weakly reachable.
100      */
101     abstract void doCleanup();
102 
103     /**
104      * Remove the cleaners whose referents have become weakly reachable.
105      */
106     static void clean() {
107         Cleaner c = (Cleaner) queue.poll();
108         while (c != null) {
109             remove(c);
110             c.doCleanup();
111             c = (Cleaner) queue.poll();
112         }
113     }
114 
115     /**
116      * The {@link ReferenceQueue} to which {@link Cleaner}s are enqueued once their referents'
117      * become unreachable.
118      */
119     private static final ReferenceQueue<Object> queue = new ReferenceQueue<>();
120 }
