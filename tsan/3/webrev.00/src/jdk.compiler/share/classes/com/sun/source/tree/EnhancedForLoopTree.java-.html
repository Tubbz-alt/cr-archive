26 package com.sun.source.tree;
27 
28 /**
29  * A tree node for an "enhanced" {@code for} loop statement.
30  *
31  * For example:
32  * <pre>
33  *   for ( <em>variable</em> : <em>expression</em> )
34  *       <em>statement</em>
35  * </pre>
36  *
37  * @jls section 14.14.2
38  *
39  * @author Peter von der Ah&eacute;
40  * @author Jonathan Gibbons
41  * @since 1.6
42  */
43 public interface EnhancedForLoopTree extends StatementTree {
44     /**
45      * Returns the control variable for the loop.
46      * @return the control variable
47      */
48     VariableTree getVariable();
49 
50     /**
51      * Returns the expression yielding the values for the control variable.
52      * @return the expression
53      */
54     ExpressionTree getExpression();
55 
56     /**
57      * Returns the body of the loop.
58      * @return the body of the loop
59      */
60     StatementTree getStatement();
61 }
