 97     /** Result codes.
 98      */
 99     public enum Result {
100         OK(0),        // Compilation completed with no errors.
101         ERROR(1),     // Completed but reported errors.
102         CMDERR(2),    // Bad command-line arguments
103         SYSERR(3),    // System error or resource exhaustion.
104         ABNORMAL(4);  // Compiler terminated abnormally
105 
106         Result(int exitCode) {
107             this.exitCode = exitCode;
108         }
109 
110         public boolean isOK() {
111             return (exitCode == 0);
112         }
113 
114         public final int exitCode;
115     }

147     /** Report a usage error.
148      */
149     void reportDiag(DiagnosticInfo diag) {
150         if (apiMode) {
151             String msg = log.localize(diag);
152             throw new PropagatedException(new IllegalStateException(msg));
153         }
154         reportHelper(diag);
155         log.printLines(PrefixKind.JAVAC, "msg.usage", ownName);
156     }
157 
158     /** Report helper.
159      */
160     void reportHelper(DiagnosticInfo diag) {
161         String msg = log.localize(diag);
162         String errorPrefix = log.localize(Errors.Error);
163         msg = msg.startsWith(errorPrefix) ? msg : errorPrefix + msg;
164         log.printRawLines(msg);
165     }

168     /**
169      * Programmatic interface for main function.
170      * @param args  the command line parameters
171      * @return the result of the compilation
172      */
173     public Result compile(String[] args) {
174         Context context = new Context();
175         JavacFileManager.preRegister(context); // can't create it until Log has been set up
176         Result result = compile(args, context);
177         try {
178             // A fresh context was created above, so the file manager can be safely closed:
179             if (fileManager != null)
180                 fileManager.close();
181         } catch (IOException ex) {
182             bugMessage(ex);
183         }
184         return result;
185     }

187     /**
188      * Internal version of compile, allowing context to be provided.
189      * Note that the context needs to have a file manager set up.
190      * @param argv  the command line parameters
191      * @param context the context
192      * @return the result of the compilation
193      */
194     public Result compile(String[] argv, Context context) {
195         if (stdOut != null) {
196             context.put(Log.outKey, stdOut);
197         }
198 
199         if (stdErr != null) {
200             context.put(Log.errKey, stdErr);
201         }
202 
203         log = Log.instance(context);
204 
205         if (argv.length == 0) {
206             OptionHelper h = new OptionHelper.GrumpyHelper(log) {
207                 @Override
208                 public String getOwnName() { return ownName; }
209                 @Override
210                 public void put(String name, String value) { }
211             };
212             try {
213                 Option.HELP.process(h, "-help");
214             } catch (Option.InvalidValueException ignore) {
215             }
216             return Result.CMDERR;
217         }
218 
219         // prefix argv with contents of environment variable and expand @-files
220         try {
221             argv = CommandLine.parse(ENV_OPT_NAME, argv);
222         } catch (UnmatchedQuote ex) {
223             reportDiag(Errors.UnmatchedQuote(ex.variableName));
224             return Result.CMDERR;
225         } catch (FileNotFoundException | NoSuchFileException e) {
226             reportHelper(Errors.FileNotFound(e.getMessage()));
227             return Result.SYSERR;
228         } catch (IOException ex) {
229             log.printLines(PrefixKind.JAVAC, "msg.io");
230             ex.printStackTrace(log.getWriter(WriterKind.NOTICE));
231             return Result.SYSERR;
232         }
233 
234         Arguments args = Arguments.instance(context);
235         args.init(ownName, argv);
236 
237         if (log.nerrors > 0)
238             return Result.CMDERR;
239 
240         Options options = Options.instance(context);
241 
242         // init Log
243         boolean forceStdOut = options.isSet("stdout");
244         if (forceStdOut) {
245             log.flush();
246             log.setWriters(new PrintWriter(System.out, true));
247         }
248 
249         // init CacheFSInfo
250         // allow System property in following line as a Mustang legacy
251         boolean batchMode = (options.isUnset("nonBatchMode")
252                     && System.getProperty("nonBatchMode") == null);
253         if (batchMode)
254             CacheFSInfo.preRegister(context);
255 
256         boolean ok = true;
257 
258         // init file manager
259         fileManager = context.get(JavaFileManager.class);
260         JavaFileManager undel = fileManager instanceof DelegatingJavaFileManager ?
261                 ((DelegatingJavaFileManager) fileManager).getBaseFileManager() : fileManager;
262         if (undel instanceof BaseFileManager) {
263             ((BaseFileManager) undel).setContext(context); // reinit with options
264             ok &= ((BaseFileManager) undel).handleOptions(args.getDeferredFileManagerOptions());
265         }
266 
267         // handle this here so it works even if no other options given
268         String showClass = options.get("showClass");
269         if (showClass != null) {
270             if (showClass.equals("showClass")) // no value given for option
271                 showClass = "com.sun.tools.javac.Main";
272             showClass(showClass);
273         }
274 
275         ok &= args.validate();
276         if (!ok || log.nerrors > 0)
277             return Result.CMDERR;
278 
279         if (args.isEmpty())
280             return Result.OK;
281 
282         // init Dependencies
283         if (options.isSet("debug.completionDeps")) {
284             Dependencies.GraphDependencies.preRegister(context);
285         }
286 
287         // init plugins
288         Set<List<String>> pluginOpts = args.getPluginOpts();
289         if (!pluginOpts.isEmpty() || context.get(PlatformDescription.class) != null) {
290             BasicJavacTask t = (BasicJavacTask) BasicJavacTask.instance(context);
291             t.initPlugins(pluginOpts);
292         }
293 
294         // init multi-release jar handling
295         if (fileManager.isSupportedOption(Option.MULTIRELEASE.primaryName) == 1) {
296             Target target = Target.instance(context);
297             List<String> list = List.of(target.multiReleaseValue());
298             fileManager.handleOption(Option.MULTIRELEASE.primaryName, list.iterator());
299         }
300 
301         // init JavaCompiler
302         JavaCompiler comp = JavaCompiler.instance(context);
303 
304         // init doclint
305         List<String> docLintOpts = args.getDocLintOpts();
306         if (!docLintOpts.isEmpty()) {
307             BasicJavacTask t = (BasicJavacTask) BasicJavacTask.instance(context);
308             t.initDocLint(docLintOpts);
309         }
310 
311         if (options.get(Option.XSTDOUT) != null) {
312             // Stdout reassigned - ask compiler to close it when it is done
313             comp.closeables = comp.closeables.prepend(log.getWriter(WriterKind.NOTICE));
314         }
315 
316         boolean printArgsToFile = options.isSet("printArgsToFile");
317         try {
318             comp.compile(args.getFileObjects(), args.getClassNames(), null, List.nil());
319 
320             if (log.expectDiagKeys != null) {
321                 if (log.expectDiagKeys.isEmpty()) {
322                     log.printRawLines("all expected diagnostics found");
323                     return Result.OK;
324                 } else {
325                     log.printRawLines("expected diagnostic keys not found: " + log.expectDiagKeys);
326                     return Result.ERROR;
327                 }
328             }
329 
330             return (comp.errorCount() == 0) ? Result.OK : Result.ERROR;
331 
332         } catch (OutOfMemoryError | StackOverflowError ex) {
333             resourceMessage(ex);
334             return Result.SYSERR;
335         } catch (FatalError ex) {
336             feMessage(ex, options);
337             return Result.SYSERR;
338         } catch (AnnotationProcessingError ex) {
339             apMessage(ex);
340             return Result.SYSERR;
341         } catch (PropagatedException ex) {
342             // TODO: what about errors from plugins?   should not simply rethrow the error here
343             throw ex.getCause();
344         } catch (IllegalAccessError iae) {
345             if (twoClassLoadersInUse(iae)) {
346                 bugMessage(iae);
347             }
348             printArgsToFile = true;
349             return Result.ABNORMAL;
350         } catch (Throwable ex) {
351             // Nasty.  If we've already reported an error, compensate
352             // for buggy compiler error recovery by swallowing thrown
353             // exceptions.
354             if (comp == null || comp.errorCount() == 0 || options.isSet("dev"))
355                 bugMessage(ex);
356             printArgsToFile = true;
357             return Result.ABNORMAL;
358         } finally {
359             if (printArgsToFile) {
360                 printArgumentsToFile(argv);
361             }
362             if (comp != null) {
363                 try {
364                     comp.close();
365                 } catch (ClientCodeException ex) {
366                     throw new RuntimeException(ex.getCause());
367                 }
368             }
369         }
370     }

372     void printArgumentsToFile(String... params) {
373         Path out = Paths.get(String.format("javac.%s.args",
374                 new SimpleDateFormat("yyyyMMdd_HHmmss").format(Calendar.getInstance().getTime())));
375         String strOut = "";
376         try {
377             try (Writer w = Files.newBufferedWriter(out)) {
378                 for (String param : params) {
379                     param = param.replaceAll("\\\\", "\\\\\\\\");
380                     if (param.matches(".*\\s+.*")) {
381                         param = "\"" + param + "\"";
382                     }
383                     strOut += param + '\n';
384                 }
385                 w.write(strOut);
386             }
387             log.printLines(PrefixKind.JAVAC, "msg.parameters.output", out.toAbsolutePath());
388         } catch (IOException ioe) {
389             log.printLines(PrefixKind.JAVAC, "msg.parameters.output.error", out.toAbsolutePath());
390             System.err.println(strOut);
391             System.err.println();
392         }
393     }

395     private boolean twoClassLoadersInUse(IllegalAccessError iae) {
396         String msg = iae.getMessage();
397         Pattern pattern = Pattern.compile("(?i)(?<=tried to access class )([a-z_$][a-z\\d_$]*\\.)*[a-z_$][a-z\\d_$]*");
398         Matcher matcher = pattern.matcher(msg);
399         if (matcher.find()) {
400             try {
401                 String otherClassName = matcher.group(0);
402                 Class<?> otherClass = Class.forName(otherClassName);
403                 ClassLoader otherClassLoader = otherClass.getClassLoader();
404                 ClassLoader javacClassLoader = this.getClass().getClassLoader();
405                 if (javacClassLoader != otherClassLoader) {
406                     CodeSource otherClassCodeSource = otherClass.getProtectionDomain().getCodeSource();
407                     CodeSource javacCodeSource = this.getClass().getProtectionDomain().getCodeSource();
408                     if (otherClassCodeSource != null && javacCodeSource != null) {
409                         log.printLines(Errors.TwoClassLoaders2(otherClassCodeSource.getLocation(),
410                                 javacCodeSource.getLocation()));
411                     } else {
412                         log.printLines(Errors.TwoClassLoaders1);
413                     }
414                     return true;
415                 }
416             } catch (Throwable t) {
417                 return false;
418             }
419         }
420         return false;
421     }

430     /** Print a message reporting a fatal error.
431      */
432     void feMessage(Throwable ex, Options options) {
433         log.printRawLines(ex.getMessage());
434         if (ex.getCause() != null && options.isSet("dev")) {
435             ex.getCause().printStackTrace(log.getWriter(WriterKind.NOTICE));
436         }

469     /** Display the location and checksum of a class. */
470     void showClass(String className) {
471         PrintWriter pw = log.getWriter(WriterKind.NOTICE);
472         pw.println("javac: show class: " + className);
473 
474         URL url = getClass().getResource('/' + className.replace('.', '/') + ".class");
475         if (url != null) {
476             pw.println("  " + url);
477         }
478 
479         try (InputStream in = getClass().getResourceAsStream('/' + className.replace('.', '/') + ".class")) {
480             final String algorithm = "MD5";
481             byte[] digest;
482             MessageDigest md = MessageDigest.getInstance(algorithm);
483             try (DigestInputStream din = new DigestInputStream(in, md)) {
484                 byte[] buf = new byte[8192];
485                 int n;
486                 do { n = din.read(buf); } while (n > 0);
487                 digest = md.digest();
488             }
489             StringBuilder sb = new StringBuilder();
490             for (byte b: digest)
491                 sb.append(String.format("%02x", b));
492             pw.println("  " + algorithm + " checksum: " + sb);
493         } catch (NoSuchAlgorithmException | IOException e) {
494             pw.println("  cannot compute digest: " + e);
495         }
496     }
