26 package com.sun.source.tree;
27 
28 import java.util.List;
29 
30 /**
31  * A tree node for an expression to create a new instance of an array.
32  *
33  * For example:
34  * <pre>
35  *   new <em>type</em> <em>dimensions</em> <em>initializers</em>
36  *
37  *   new <em>type</em> <em>dimensions</em> [ ] <em>initializers</em>
38  * </pre>
39  *
40  * @jls section 15.10
41  *
42  * @author Peter von der Ah&eacute;
43  * @author Jonathan Gibbons
44  * @since 1.6
45  */
46 public interface NewArrayTree extends ExpressionTree {
47     /**
48      * Returns the base type of the expression.
49      * May be {@code null} for an array initializer expression.
50      * @return the base type
51      */
52     Tree getType();
53 
54     /**
55      * Returns the dimension expressions for the type.
56      *
57      * @return the dimension expressions
58      */
59     List<? extends ExpressionTree> getDimensions();
60 
61     /**
62      * Returns the initializer expressions.
63      *
64      * @return the initializer expressions
65      */
66     List<? extends ExpressionTree> getInitializers();
67 
68     /**
69      * Returns the annotations on the base type.
70      * @return the annotations
71      */
72     List<? extends AnnotationTree> getAnnotations();
73 
74     /**
75      * Returns the annotations on each of the dimension
76      * expressions.
77      * @return the annotations on the dimensions expressions
78      */
79     List<? extends List<? extends AnnotationTree>> getDimAnnotations();
80 }
