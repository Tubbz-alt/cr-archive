26 package com.sun.source.tree;
27 
28 import java.util.List;
29 
30 /**
31  * A tree node for a statement block.
32  *
33  * For example:
34  * <pre>
35  *   { }
36  *
37  *   { <em>statements</em> }
38  *
39  *   static { <em>statements</em> }
40  * </pre>
41  *
42  * @jls section 14.2
43  *
44  * @author Peter von der Ah&eacute;
45  * @author Jonathan Gibbons
46  * @since 1.6
47  */
48 public interface BlockTree extends StatementTree {
49     /**
50      * Returns true if and only if this is a static initializer block.
51      * @return true if this is a static initializer block
52      */
53     boolean isStatic();
54 
55     /**
56      * Returns the statements comprising this block.
57      * @return the statements
58      */
59     List<? extends StatementTree> getStatements();
60 }
