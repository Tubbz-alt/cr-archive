 26 package com.sun.source.tree;
 27 
 28 import java.util.List;
 29 import javax.lang.model.element.Name;
 30 
 31 /**
 32  * A tree node for a method or annotation type element declaration.
 33  *
 34  * For example:
 35  * <pre>
 36  *   <em>modifiers</em> <em>typeParameters</em> <em>type</em> <em>name</em>
 37  *      ( <em>parameters</em> )
 38  *      <em>body</em>
 39  *
 40  *   <em>modifiers</em> <em>type</em> <em>name</em> () default <em>defaultValue</em>
 41  * </pre>
 42  *
 43  * @jls 8.4 Method Declarations
 44  * @jls 8.6 Instance Initializers
 45  * @jls 8.7 Static Initializers
 46  * @jls 9.4 Method Declarations
 47  * @jls 9.6.1 Annotation Type Elements
 48  *
 49  * @author Peter von der Ah&eacute;
 50  * @author Jonathan Gibbons
 51  * @since 1.6
 52  */
 53 public interface MethodTree extends Tree {
 54     /**
 55      * Returns the modifiers, including any annotations for the method being declared.
 56      * @return the modifiers
 57      */
 58     ModifiersTree getModifiers();
 59 
 60     /**
 61      * Returns the name of the method being declared.
 62      * @return the name
 63      */
 64     Name getName();
 65 
 66     /**
 67      * Returns the return type of the method being declared.
 68      * Returns {@code null} for a constructor.
 69      * @return the return type
 70      */
 71     Tree getReturnType();
 72 
 73     /**
 74      * Returns the type parameters of the method being declared.
 75      * @return the type parameters
 76      */
 77     List<? extends TypeParameterTree> getTypeParameters();
 78 
 79     /**
 80      * Returns the parameters of the method being declared.
 81      * @return the parameters
 82      */
 83     List<? extends VariableTree> getParameters();
 84 
 85     /**
 86      * Return an explicit receiver parameter ("this" parameter),
 87      * or {@code null} if none.
 88      *
 89      * @return an explicit receiver parameter ("this" parameter)
 90      * @since 1.8
 91      */
 92     VariableTree getReceiverParameter();
 93 
 94     /**
 95      * Returns the exceptions listed as being thrown by this method.
 96      * @return the exceptions
 97      */
 98     List<? extends ExpressionTree> getThrows();
 99 
100     /**
101      * Returns the method body, or {@code null} if this is an abstract or native method.
102      * @return the method body
103      */
104     BlockTree getBody();
105 
106     /**
107      * Returns the default value, if this is an element within
108      * an annotation type declaration.
109      * Returns {@code null} otherwise.
110      * @return the default value
111      */
112     Tree getDefaultValue(); // for annotation types
113 }
