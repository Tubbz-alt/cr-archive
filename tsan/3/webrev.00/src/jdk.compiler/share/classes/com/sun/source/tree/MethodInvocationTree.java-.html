26 package com.sun.source.tree;
27 
28 import java.util.List;
29 
30 /**
31  * A tree node for a method invocation expression.
32  *
33  * For example:
34  * <pre>
35  *   <em>identifier</em> ( <em>arguments</em> )
36  *
37  *   this . <em>typeArguments</em> <em>identifier</em> ( <em>arguments</em> )
38  * </pre>
39  *
40  * @jls section 15.12
41  *
42  * @author Peter von der Ah&eacute;
43  * @author Jonathan Gibbons
44  * @since 1.6
45  */
46 public interface MethodInvocationTree extends ExpressionTree {
47     /**
48      * Returns the type arguments for this method invocation.
49      * @return the type arguments
50      */
51     List<? extends Tree> getTypeArguments();
52 
53     /**
54      * Returns the expression identifying the method to be invoked.
55      * @return the method selection expression
56      */
57     ExpressionTree getMethodSelect();
58 
59     /**
60      * Returns the arguments for the method invocation.
61      * @return the arguments
62      */
63     List<? extends ExpressionTree> getArguments();
64 }
