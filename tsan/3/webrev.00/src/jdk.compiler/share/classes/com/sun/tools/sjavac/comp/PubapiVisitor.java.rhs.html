 26 package com.sun.tools.sjavac.comp;
 27 
 28 import static javax.lang.model.element.Modifier.PRIVATE;
 29 
 30 import java.util.List;
 31 import java.util.stream.Collectors;
 32 
 33 import javax.lang.model.element.*;




 34 import javax.lang.model.type.TypeMirror;
 35 import javax.lang.model.util.ElementScanner14;
 36 
 37 import com.sun.tools.javac.code.Symbol.ClassSymbol;
 38 import com.sun.tools.javac.util.DefinedBy;
 39 import com.sun.tools.javac.util.DefinedBy.Api;
 40 import com.sun.tools.sjavac.pubapi.PubApi;
 41 import com.sun.tools.sjavac.pubapi.PubApiTypeParam;
 42 import com.sun.tools.sjavac.pubapi.PubMethod;
 43 import com.sun.tools.sjavac.pubapi.PubType;
 44 import com.sun.tools.sjavac.pubapi.PubVar;
 45 import com.sun.tools.sjavac.pubapi.TypeDesc;

 47 /** Utility class that constructs a textual representation
 48  * of the public api of a class.
 49  *
 50  *  <p><b>This is NOT part of any supported API.
 51  *  If you write code that depends on this, you do so at your own risk.
 52  *  This code and its internal interfaces are subject to change or
 53  *  deletion without notice.</b>
 54  */
 55 @SuppressWarnings("preview")
 56 public class PubapiVisitor extends ElementScanner14<Void, Void> {
 57 
 58     private PubApi collectedApi = new PubApi();
 59 
 60     private boolean isNonPrivate(Element e) {
 61         return !e.getModifiers().contains(PRIVATE);
 62     }
 63 
 64     @Override @DefinedBy(Api.LANGUAGE_MODEL)
 65     public Void visitType(TypeElement e, Void p) {
 66         if (isNonPrivate(e)) {
 67             PubApi prevApi = collectedApi;
 68             collectedApi = new PubApi();
 69             super.visitType(e, p);
 70             if (!isAnonymous(e)) {
 71                 String name = ((ClassSymbol) e).flatname.toString();
 72                 PubType t = new PubType(e.getModifiers(),
 73                                         name,
 74                                         //e.getQualifiedName().toString(),
 75                                         collectedApi);
 76                 prevApi.types.put(t.fqName, t);
 77             }
 78             collectedApi = prevApi;
 79         }
 80         return null;
 81     }
 82 
 83     private boolean isAnonymous(TypeElement e) {
 84         return e.getQualifiedName().length() == 0;
 85     }
 86 
 87     private static String encodeChar(int c) {
 88         return String.format("\\u%04x", c);
 89     }
 90 
 91     @Override @DefinedBy(Api.LANGUAGE_MODEL)
 92     public Void visitVariable(VariableElement e, Void p) {
 93         if (isNonPrivate(e)) {
 94             Object constVal = e.getConstantValue();
 95             String constValStr = null;
 96             // TODO: This doesn't seem to be entirely accurate. What if I change
 97             // from, say, 0 to 0L? (And the field is public final static so that
 98             // it could get inlined.)
 99             if (constVal != null) {
100                 if (e.asType().toString().equals("char")) {
101                     // What type is 'value'? Is it already a char?
102                     char c = constVal.toString().charAt(0);
103                     constValStr = "'" + encodeChar(c) + "'";
104                 } else {
105                     constValStr = constVal.toString()
106                                           .chars()
107                                           .mapToObj(PubapiVisitor::encodeChar)
108                                           .collect(Collectors.joining("", "\"", "\""));
109                 }
110             }
111 
112             PubVar v = new PubVar(e.getModifiers(),
113                                   TypeDesc.fromType(e.asType()),
114                                   e.toString(),
115                                   constValStr);
116             collectedApi.variables.put(v.identifier, v);
117         }
118 
119         // Safe to not recurse here, because the only thing
120         // to visit here is the constructor of a variable declaration.
121         // If it happens to contain an anonymous inner class (which it might)
122         // then this class is never visible outside of the package anyway, so
123         // we are allowed to ignore it here.
124         return null;
125     }
126 
127     @SuppressWarnings("preview")
128     @Override @DefinedBy(Api.LANGUAGE_MODEL)
129     public Void visitRecordComponent(RecordComponentElement e, Void p) {
130         PubVar v = new PubVar(e.getModifiers(),
131                 TypeDesc.fromType(e.asType()),
132                 e.toString(),
133                 null);
134         collectedApi.recordComponents.put(v.identifier, v);
135         return null;
136     }
137 
138     @Override @DefinedBy(Api.LANGUAGE_MODEL)
139     public Void visitExecutable(ExecutableElement e, Void p) {
140         if (isNonPrivate(e)) {
141             PubMethod m = new PubMethod(e.getModifiers(),
142                                         getTypeParameters(e.getTypeParameters()),
143                                         TypeDesc.fromType(e.getReturnType()),
144                                         e.getSimpleName().toString(),
145                                         getTypeDescs(getParamTypes(e)),
146                                         getTypeDescs(e.getThrownTypes()));
147             collectedApi.methods.put(m.asSignatureString(), m);
148         }
149         return null;
150     }
151 
152     private List<PubApiTypeParam> getTypeParameters(List<? extends TypeParameterElement> elements) {
153         return elements.stream()
154                        .map(e -> new PubApiTypeParam(e.getSimpleName().toString(), getTypeDescs(e.getBounds())))
155                        .collect(Collectors.toList());
156     }
157 
158     private List<TypeMirror> getParamTypes(ExecutableElement e) {
159         return e.getParameters()
160                 .stream()
161                 .map(VariableElement::asType)
162                 .collect(Collectors.toList());
163     }
164 
165     private List<TypeDesc> getTypeDescs(List<? extends TypeMirror> list) {
166         return list.stream()
167                    .map(TypeDesc::fromType)
168                    .collect(Collectors.toList());
169     }
170 
171     public PubApi getCollectedPubApi() {
172         return collectedApi;
173     }
174 }
