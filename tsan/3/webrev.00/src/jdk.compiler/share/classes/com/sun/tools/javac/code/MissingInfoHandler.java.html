26 package com.sun.tools.javac.code;
27 
28 import com.sun.source.util.ParameterNameProvider;
29 import com.sun.tools.javac.code.Symbol.ParamSymbol;
30 import com.sun.tools.javac.util.Context;
31 import com.sun.tools.javac.util.Name;
32 import com.sun.tools.javac.util.Names;
33 
34 /**
35  * A Context class, that can return additional useful information for Symbols, currently
36  * parameter names. It does so by calling user-supplied {@link ParameterNameProvider}.
37  *
38  * <p><b>This is NOT part of any supported API.
39  * If you write code that depends on this, you do so at your own risk.
40  * This code and its internal interfaces are subject to change or
41  * deletion without notice.</b>
42  */
43 public class MissingInfoHandler {
44     protected static final Context.Key<MissingInfoHandler> missingInfoHandlerWrapperKey = new Context.Key<>();
45 
46     public static MissingInfoHandler instance(Context context) {
47         MissingInfoHandler instance = context.get(missingInfoHandlerWrapperKey);
48         if (instance == null)
49             instance = new MissingInfoHandler(context);
50         return instance;
51     }
52 
53     private final Names names;
54     private ParameterNameProvider parameterNameProvider;
55 
56     protected MissingInfoHandler(Context context) {
57         context.put(missingInfoHandlerWrapperKey, this);
58         names = Names.instance(context);
59     }
60 
61     public Name getParameterName(ParamSymbol parameter) {
62         if (parameterNameProvider != null) {
63             CharSequence name = parameterNameProvider.getParameterName(parameter);
64             if (name != null) {
65                 return names.fromString(name.toString());
66             }
67         }
68 
69         return null;
70     }
71 
72     public void setDelegate(ParameterNameProvider delegate) {
73         this.parameterNameProvider = delegate;
74     }
75 }
