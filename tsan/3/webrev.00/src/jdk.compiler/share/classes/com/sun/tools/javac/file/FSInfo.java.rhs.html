 48 /**

 57 public class FSInfo {
 58 
 59     /** Get the FSInfo instance for this context.
 60      *  @param context the context
 61      *  @return the Paths instance for this context
 62      */
 63     public static FSInfo instance(Context context) {
 64         FSInfo instance = context.get(FSInfo.class);
 65         if (instance == null)
 66             instance = new FSInfo();
 67         return instance;
 68     }

 70     protected FSInfo() {
 71     }

 73     protected FSInfo(Context context) {
 74         context.put(FSInfo.class, this);
 75     }

 77     public Path getCanonicalFile(Path file) {
 78         try {
 79             return file.toRealPath();
 80         } catch (IOException e) {
 81             return file.toAbsolutePath().normalize();
 82         }
 83     }
 84 
 85     public boolean exists(Path file) {
 86         return Files.exists(file);
 87     }
 88 
 89     public boolean isDirectory(Path file) {
 90         return Files.isDirectory(file);
 91     }
 92 
 93     public boolean isFile(Path file) {
 94         return Files.isRegularFile(file);
 95     }

 97     public List<Path> getJarClassPath(Path file) throws IOException {

 98         try (JarFile jarFile = new JarFile(file.toFile())) {
 99             Manifest man = jarFile.getManifest();
100             if (man == null)
101                 return Collections.emptyList();
102 
103             Attributes attr = man.getMainAttributes();
104             if (attr == null)
105                 return Collections.emptyList();
106 
107             String path = attr.getValue(Attributes.Name.CLASS_PATH);
108             if (path == null)
109                 return Collections.emptyList();
110 
111             List<Path> list = new ArrayList<>();
112             URL base = file.toUri().toURL();
113 
114             for (StringTokenizer st = new StringTokenizer(path);
115                  st.hasMoreTokens(); ) {
116                 String elt = st.nextToken();
117                 try {
118                     URL url = tryResolveFile(base, elt);
119                     if (url != null) {
120                         list.add(Path.of(url.toURI()));
121                     }
122                 } catch (URISyntaxException ex) {
123                     throw new IOException(ex);
124                 }
125             }
126 
127             return list;
128         }
129     }
130 
131     /**
132      * Attempt to return a file URL by resolving input against a base file
133      * URL.
134      *
135      * @return the resolved URL or null if the input is an absolute URL with
136      *         a scheme other than file (ignoring case)
137      * @throws MalformedURLException
138      */
139     static URL tryResolveFile(URL base, String input) throws MalformedURLException {
140         URL retVal = new URL(base, input);
141         if (input.indexOf(':') >= 0 && !"file".equalsIgnoreCase(retVal.getProtocol())) {
142             // 'input' contains a ':', which might be a scheme, or might be
143             // a Windows drive letter.  If the resolved file has a protocol
144             // that isn't "file:", it should be ignored.
145             return null;
146         }
147         return retVal;
148     }
149 
150     private FileSystemProvider jarFSProvider;
151 
152     public synchronized FileSystemProvider getJarFSProvider() {
153         if (jarFSProvider != null) {
154             return jarFSProvider;
155         }
156         for (FileSystemProvider provider: FileSystemProvider.installedProviders()) {
157             if (provider.getScheme().equals("jar")) {
158                 return (jarFSProvider = provider);
159             }
160         }
161         return null;
162     }
