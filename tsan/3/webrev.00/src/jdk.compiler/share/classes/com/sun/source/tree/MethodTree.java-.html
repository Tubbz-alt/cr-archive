 26 package com.sun.source.tree;
 27 
 28 import java.util.List;
 29 import javax.lang.model.element.Name;
 30 
 31 /**
 32  * A tree node for a method or annotation type element declaration.
 33  *
 34  * For example:
 35  * <pre>
 36  *   <em>modifiers</em> <em>typeParameters</em> <em>type</em> <em>name</em>
 37  *      ( <em>parameters</em> )
 38  *      <em>body</em>
 39  *
 40  *   <em>modifiers</em> <em>type</em> <em>name</em> () default <em>defaultValue</em>
 41  * </pre>
 42  *
 43  * @jls sections 8.4, 8.6, 8.7, 9.4, and 9.6
 44  *
 45  * @author Peter von der Ah&eacute;
 46  * @author Jonathan Gibbons
 47  * @since 1.6
 48  */
 49 public interface MethodTree extends Tree {
 50     /**
 51      * Returns the modifiers, including any annotations for the method being declared.
 52      * @return the modifiers
 53      */
 54     ModifiersTree getModifiers();
 55 
 56     /**
 57      * Returns the name of the method being declared.
 58      * @return the name
 59      */
 60     Name getName();
 61 
 62     /**
 63      * Returns the return type of the method being declared.
 64      * Returns {@code null} for a constructor.
 65      * @return the return type
 66      */
 67     Tree getReturnType();
 68 
 69     /**
 70      * Returns the type parameters of the method being declared.
 71      * @return the type parameters
 72      */
 73     List<? extends TypeParameterTree> getTypeParameters();
 74 
 75     /**
 76      * Returns the parameters of the method being declared.
 77      * @return the parameters
 78      */
 79     List<? extends VariableTree> getParameters();
 80 
 81     /**
 82      * Return an explicit receiver parameter ("this" parameter),
 83      * or {@code null} if none.
 84      *
 85      * @return an explicit receiver parameter ("this" parameter)
 86      * @since 1.8
 87      */
 88     VariableTree getReceiverParameter();
 89 
 90     /**
 91      * Returns the exceptions listed as being thrown by this method.
 92      * @return the exceptions
 93      */
 94     List<? extends ExpressionTree> getThrows();
 95 
 96     /**
 97      * Returns the method body, or {@code null} if this is an abstract or native method.
 98      * @return the method body
 99      */
100     BlockTree getBody();
101 
102     /**
103      * Returns the default value, if this is an element within
104      * an annotation type declaration.
105      * Returns {@code null} otherwise.
106      * @return the default value
107      */
108     Tree getDefaultValue(); // for annotation types
109 }
