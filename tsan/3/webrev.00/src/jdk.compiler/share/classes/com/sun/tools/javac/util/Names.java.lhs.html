 39     public static final Context.Key<Names> namesKey = new Context.Key<>();

 41     public static Names instance(Context context) {
 42         Names instance = context.get(namesKey);
 43         if (instance == null) {
 44             instance = new Names(context);
 45             context.put(namesKey, instance);
 46         }
 47         return instance;
 48     }

343     protected Name.Table createTable(Options options) {
344         boolean useUnsharedTable = options.isSet("useUnsharedTable");
345         if (useUnsharedTable)
346             return UnsharedNameTable.create(this);
347         else
348             return SharedNameTable.create(this);
349     }
