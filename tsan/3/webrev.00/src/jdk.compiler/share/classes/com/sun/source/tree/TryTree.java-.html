26 package com.sun.source.tree;
27 
28 import java.util.List;
29 
30 /**
31  * A tree node for a {@code try} statement.
32  *
33  * For example:
34  * <pre>
35  *   try
36  *       <em>block</em>
37  *   <em>catches</em>
38  *   finally
39  *       <em>finallyBlock</em>
40  * </pre>
41  *
42  * @jls section 14.20
43  *
44  * @author Peter von der Ah&eacute;
45  * @author Jonathan Gibbons
46  * @since 1.6
47  */
48 public interface TryTree extends StatementTree {
49     /**
50      * Returns the block of the {@code try} statement.
51      * @return the block
52      */
53     BlockTree getBlock();
54 
55     /**
56      * Returns any catch blocks provided in the {@code try} statement.
57      * The result will be an empty list if there are no
58      * catch blocks.
59      * @return the catch blocks
60      */
61     List<? extends CatchTree> getCatches();
62 
63     /**
64      * Returns the finally block provided in the {@code try} statement,
65      * or {@code null} if there is none.
66      * @return the finally block
67      */
68     BlockTree getFinallyBlock();
69 
70 
71     /**
72      * Returns any resource declarations provided in the {@code try} statement.
73      * The result will be an empty list if there are no
74      * resource declarations.
75      * @return the resource declarations
76      */
77     List<? extends Tree> getResources();
78 }
