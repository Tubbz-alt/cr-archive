 92     public List<Path> getJarClassPath(Path file) throws IOException {
 93         Path parent = file.getParent();
 94         try (JarFile jarFile = new JarFile(file.toFile())) {
 95             Manifest man = jarFile.getManifest();
 96             if (man == null)
 97                 return Collections.emptyList();
 98 
 99             Attributes attr = man.getMainAttributes();
100             if (attr == null)
101                 return Collections.emptyList();
102 
103             String path = attr.getValue(Attributes.Name.CLASS_PATH);
104             if (path == null)
105                 return Collections.emptyList();
106 
107             List<Path> list = new ArrayList<>();

108 
109             for (StringTokenizer st = new StringTokenizer(path);
110                  st.hasMoreTokens(); ) {
111                 String elt = st.nextToken();
112                 Path f = FileSystems.getDefault().getPath(elt);
113                 if (!f.isAbsolute() && parent != null)
114                     f = parent.resolve(f).toAbsolutePath();
115                 list.add(f);




116             }
117 
118             return list;
119         }
120     }

 97     public List<Path> getJarClassPath(Path file) throws IOException {

 98         try (JarFile jarFile = new JarFile(file.toFile())) {
 99             Manifest man = jarFile.getManifest();
100             if (man == null)
101                 return Collections.emptyList();
102 
103             Attributes attr = man.getMainAttributes();
104             if (attr == null)
105                 return Collections.emptyList();
106 
107             String path = attr.getValue(Attributes.Name.CLASS_PATH);
108             if (path == null)
109                 return Collections.emptyList();
110 
111             List<Path> list = new ArrayList<>();
112             URL base = file.toUri().toURL();
113 
114             for (StringTokenizer st = new StringTokenizer(path);
115                  st.hasMoreTokens(); ) {
116                 String elt = st.nextToken();
117                 try {
118                     URL url = tryResolveFile(base, elt);
119                     if (url != null) {
120                         list.add(Path.of(url.toURI()));
121                     }
122                 } catch (URISyntaxException ex) {
123                     throw new IOException(ex);
124                 }
125             }
126 
127             return list;
128         }
129     }

131     /**
132      * Attempt to return a file URL by resolving input against a base file
133      * URL.
134      *
135      * @return the resolved URL or null if the input is an absolute URL with
136      *         a scheme other than file (ignoring case)
137      * @throws MalformedURLException
138      */
139     static URL tryResolveFile(URL base, String input) throws MalformedURLException {
140         URL retVal = new URL(base, input);
141         if (input.indexOf(':') >= 0 && !"file".equalsIgnoreCase(retVal.getProtocol())) {
142             // 'input' contains a ':', which might be a scheme, or might be
143             // a Windows drive letter.  If the resolved file has a protocol
144             // that isn't "file:", it should be ignored.
145             return null;
146         }
147         return retVal;
148     }
