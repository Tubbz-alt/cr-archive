37  *
38  *   ? super <em>bound</em>
39  * </pre>
40  *
41  * @jls section 4.5.1
42  *
43  * @author Peter von der Ah&eacute;
44  * @author Jonathan Gibbons
45  * @since 1.6
46  */
47 public interface WildcardTree extends Tree {
48     /**
49      * Returns the bound of the wildcard.
50      * @return the bound
51      */
52     Tree getBound();
53 }
