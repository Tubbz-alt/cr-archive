26 package com.sun.source.tree;
27 
28 /**
29  * A tree node for a {@code catch} block in a {@code try} statement.
30  *
31  * For example:
32  * <pre>
33  *   catch ( <em>parameter</em> )
34  *       <em>block</em>
35  * </pre>
36  *
37  * @jls 14.20 The try statement
38  *
39  * @author Peter von der Ah&eacute;
40  * @author Jonathan Gibbons
41  * @since 1.6
42  */
43 public interface CatchTree extends Tree {
44     /**
45      * Returns the catch variable.
46      * A multi-catch variable will have a
47      * {@link UnionTypeTree UnionTypeTree}
48      * as the type of the variable.
49      * @return the catch variable
50      */
51     VariableTree getParameter();
52 
53     /**
54      * Returns the catch block.
55      * @return the catch block
56      */
57     BlockTree getBlock();
58 }
