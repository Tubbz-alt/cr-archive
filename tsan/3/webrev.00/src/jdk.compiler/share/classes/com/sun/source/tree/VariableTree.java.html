26 package com.sun.source.tree;
27 
28 import javax.lang.model.element.Name;
29 
30 /**
31  * A tree node for a variable declaration.
32  *
33  * For example:
34  * <pre>
35  *   <em>modifiers</em> <em>type</em> <em>name</em> <em>initializer</em> ;
36  *   <em>modifiers</em> <em>type</em> <em>qualified-name</em>.this
37  * </pre>
38  *
39  * @jls 8.3 Field Declarations
40  * @jls 14.4 Local Variable Declaration Statements
41  *
42  * @author Peter von der Ah&eacute;
43  * @author Jonathan Gibbons
44  * @since 1.6
45  */
46 public interface VariableTree extends StatementTree {
47     /**
48      * Returns the modifiers, including any annotations, on the declaration.
49      * @return the modifiers
50      */
51     ModifiersTree getModifiers();
52 
53     /**
54      * Returns the name of the variable being declared.
55      * @return the name
56      */
57     Name getName();
58 
59     /**
60      * Returns the qualified identifier for the name being "declared".
61      * This is only used in certain cases for the receiver of a
62      * method declaration. Returns {@code null} in all other cases.
63      * @return the qualified identifier of a receiver declaration
64      */
65     ExpressionTree getNameExpression();
66 
67     /**
68      * Returns the type of the variable being declared.
69      * @return the type
70      */
71     Tree getType();
72 
73     /**
74      * Returns the initializer for the variable, or {@code null} if none.
75      * @return the initializer
76      */
77     ExpressionTree getInitializer();
78 }
