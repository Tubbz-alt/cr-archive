187     public void initPlugins(Set<List<String>> pluginOpts) {
188         PlatformDescription platformProvider = context.get(PlatformDescription.class);
189 
190         if (platformProvider != null) {
191             for (PluginInfo<Plugin> pluginDesc : platformProvider.getPlugins()) {
192                 java.util.List<String> options =
193                         pluginDesc.getOptions().entrySet().stream()
194                                                           .map(e -> e.getKey() + "=" + e.getValue())
195                                                           .collect(Collectors.toList());
196                 try {
197                     pluginDesc.getPlugin().init(this, options.toArray(new String[options.size()]));
198                 } catch (RuntimeException ex) {
199                     throw new PropagatedException(ex);
200                 }
201             }
202         }
203 
204         if (pluginOpts.isEmpty())
205             return;
206 
207         Set<List<String>> pluginsToCall = new LinkedHashSet<>(pluginOpts);
208         JavacProcessingEnvironment pEnv = JavacProcessingEnvironment.instance(context);
209         ServiceLoader<Plugin> sl = pEnv.getServiceLoader(Plugin.class);

210         for (Plugin plugin : sl) {



211             for (List<String> p : pluginsToCall) {
212                 if (plugin.getName().equals(p.head)) {
213                     pluginsToCall.remove(p);

214                     try {
215                         plugin.init(this, p.tail.toArray(new String[p.tail.size()]));
216                     } catch (RuntimeException ex) {
217                         throw new PropagatedException(ex);
218                     }
219                     break;
220                 }
221             }
222         }
223         for (List<String> p: pluginsToCall) {
224             Log.instance(context).error(Errors.PluginNotFound(p.head));
225         }
















226     }

198     public void initPlugins(Set<List<String>> pluginOpts) {
199         PlatformDescription platformProvider = context.get(PlatformDescription.class);
200 
201         if (platformProvider != null) {
202             for (PluginInfo<Plugin> pluginDesc : platformProvider.getPlugins()) {
203                 java.util.List<String> options =
204                         pluginDesc.getOptions().entrySet().stream()
205                                 .map(e -> e.getKey() + "=" + e.getValue())
206                                 .collect(Collectors.toList());
207                 try {
208                     initPlugin(pluginDesc.getPlugin(), options.toArray(new String[options.size()]));
209                 } catch (RuntimeException ex) {
210                     throw new PropagatedException(ex);
211                 }
212             }
213         }
214 



215         Set<List<String>> pluginsToCall = new LinkedHashSet<>(pluginOpts);
216         JavacProcessingEnvironment pEnv = JavacProcessingEnvironment.instance(context);
217         ServiceLoader<Plugin> sl = pEnv.getServiceLoader(Plugin.class);
218         Set<Plugin> autoStart = new LinkedHashSet<>();
219         for (Plugin plugin : sl) {
220             if (plugin.autoStart()) {
221                 autoStart.add(plugin);
222             }
223             for (List<String> p : pluginsToCall) {
224                 if (plugin.getName().equals(p.head)) {
225                     pluginsToCall.remove(p);
226                     autoStart.remove(plugin);
227                     try {
228                         initPlugin(plugin, p.tail.toArray(new String[p.tail.size()]));
229                     } catch (RuntimeException ex) {
230                         throw new PropagatedException(ex);
231                     }
232                     break;
233                 }
234             }
235         }
236         for (List<String> p : pluginsToCall) {
237             Log.instance(context).error(Errors.PluginNotFound(p.head));
238         }
239         for (Plugin plugin : autoStart) {
240             try {
241                 initPlugin(plugin, new String[0]);
242             } catch (RuntimeException ex) {
243                 throw new PropagatedException(ex);
244             }
245 
246         }
247     }

249     private void initPlugin(Plugin p, String... args) {
250         Module m = p.getClass().getModule();
251         if (m.isNamed() && options.isSet("accessInternalAPI")) {
252             ModuleHelper.addExports(getClass().getModule(), m);
253         }
254         p.init(this, args);
255     }
