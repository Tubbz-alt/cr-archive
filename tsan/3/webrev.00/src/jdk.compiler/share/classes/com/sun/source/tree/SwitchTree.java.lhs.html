26 package com.sun.source.tree;
27 
28 import java.util.List;
29 
30 /**
31  * A tree node for a {@code switch} statement.
32  *
33  * For example:
34  * <pre>
35  *   switch ( <em>expression</em> ) {
36  *     <em>cases</em>
37  *   }
38  * </pre>
39  *
40  * @jls section 14.11
41  *
42  * @author Peter von der Ah&eacute;
43  * @author Jonathan Gibbons
44  * @since 1.6
45  */
46 public interface SwitchTree extends StatementTree {
47     /**
48      * Returns the expression for the {@code switch} statement.
49      * @return the expression
50      */
51     ExpressionTree getExpression();
52 
53     /**
54      * Returns the cases for the {@code switch} statement.
55      * @return the cases
56      */
57     List<? extends CaseTree> getCases();
58 }
