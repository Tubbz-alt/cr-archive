26 package com.sun.source.tree;
27 
28 import javax.lang.model.element.Name;
29 
30 /**
31  * A tree node for a labeled statement.
32  *
33  * For example:
34  * <pre>
35  *   <em>label</em> : <em>statement</em>
36  * </pre>
37  *
38  * @jls section 14.7
39  *
40  * @author Peter von der Ah&eacute;
41  * @author Jonathan Gibbons
42  * @since 1.6
43  */
44 public interface LabeledStatementTree extends StatementTree {
45     /**
46      * Returns the label.
47      * @return the label
48      */
49     Name getLabel();
50 
51     /**
52      * Returns the statement that is labeled.
53      * @return the statement
54      */
55     StatementTree getStatement();
56 }
