33  * <pre>
34  *   <em>operator</em> <em>expression</em>
35  *
36  *   <em>expression</em> <em>operator</em>
37  * </pre>
38  *
39  * @jls 15.14 Postfix Expressions
40  * @jls 15.15 Unary Operators
41  *
42  * @author Peter von der Ah&eacute;
43  * @author Jonathan Gibbons
44  * @since 1.6
45  */
46 public interface UnaryTree extends ExpressionTree {
47     /**
48      * Returns the expression that is the operand of the unary operator.
49      * @return the expression
50      */
51     ExpressionTree getExpression();
52 }
