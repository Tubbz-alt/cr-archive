26 package com.sun.source.tree;
27 
28 import java.util.List;
29 import javax.lang.model.element.Name;
30 
31 /**
32  * A tree node for a class, interface, enum, record, or annotation
33  * type declaration.
34  *
35  * For example:
36  * <pre>
37  *   <em>modifiers</em> class <em>simpleName</em> <em>typeParameters</em>
38  *       extends <em>extendsClause</em>
39  *       implements <em>implementsClause</em>
40  *   {
41  *       <em>members</em>
42  *   }
43  * </pre>
44  *
45  * @jls 8.1 Class Declarations
46  * @jls 8.9 Enum Types
47  * @jls 8.10 Record Types
48  * @jls 9.1 Interface Declarations
49  * @jls 9.6 Annotation Types
50  *
51  * @author Peter von der Ah&eacute;
52  * @author Jonathan Gibbons
53  * @since 1.6
54  */
55 public interface ClassTree extends StatementTree {
56     /**
57      * Returns the modifiers, including any annotations,
58      * for this type declaration.
59      * @return the modifiers
60      */
61     ModifiersTree getModifiers();
62 
63     /**
64      * Returns the simple name of this type declaration.
65      * @return the simple name
66      */
67     Name getSimpleName();
68 
69     /**
70      * Returns any type parameters of this type declaration.
71      * @return the type parameters
72      */
73     List<? extends TypeParameterTree> getTypeParameters();
74 
75     /**
76      * Returns the supertype of this type declaration,
77      * or {@code null} if none is provided.
78      * @return the supertype
79      */
80     Tree getExtendsClause();
81 
82     /**
83      * Returns the interfaces implemented by this type declaration.
84      * @return the interfaces
85      */
86     List<? extends Tree> getImplementsClause();
87 
88     /**
89      * Returns the members declared in this type declaration.
90      * @return the members
91      */
92     List<? extends Tree> getMembers();
93 }
