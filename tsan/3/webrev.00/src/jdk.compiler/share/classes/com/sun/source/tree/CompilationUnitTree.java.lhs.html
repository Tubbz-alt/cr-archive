26 package com.sun.source.tree;
27 
28 import java.util.List;
29 import javax.tools.JavaFileObject;
30 
31 /**
32  * Represents the abstract syntax tree for compilation units (source
33  * files) and package declarations (package-info.java).
34  *
35  * @jls sections 7.3, and 7.4

36  *
37  * @author Peter von der Ah&eacute;
38  * @since 1.6
39  */
40 public interface CompilationUnitTree extends Tree {
41     /**
42      * Returns the annotations listed on any package declaration
43      * at the head of this compilation unit, or {@code null} if there
44      * is no package declaration.
45      * @return the package annotations
46      */
47     List<? extends AnnotationTree> getPackageAnnotations();
48 
49     /**
50      * Returns the name contained in any package declaration
51      * at the head of this compilation unit, or {@code null} if there
52      * is no package declaration.
53      * @return the package name
54      */
55     ExpressionTree getPackageName();
56 
57     /**
58      * Returns the package tree associated with this compilation unit,
59      * or {@code null} if there is no package declaration.
60      * @return the package tree
61      * @since 9
62      */
63     PackageTree getPackage();
64 
65     /**
66      * Returns the import declarations appearing in this compilation unit.
67      * @return the import declarations
68      */
69     List<? extends ImportTree> getImports();
70 
71     /**
72      * Returns the type declarations appearing in this compilation unit.
73      * The list may also include empty statements resulting from
74      * extraneous semicolons.
75      * @return the type declarations
76      */
77     List<? extends Tree> getTypeDecls();
78 
79     /**
80      * Returns the file object containing the source for this compilation unit.
81      * @return the file object
82      */
83     JavaFileObject getSourceFile();
84 
85     /**
86      * Returns the line map for this compilation unit, if available.
87      * Returns {@code null} if the line map is not available.
88      * @return the line map for this compilation unit
89      */
90     LineMap getLineMap();
91 }
