 26 package com.sun.tools.javac.comp;
 27 
 28 import com.sun.tools.javac.code.Symbol;
 29 import com.sun.tools.javac.tree.JCTree;
 30 import com.sun.tools.javac.tree.JCTree.JCFieldAccess;
 31 import com.sun.tools.javac.tree.JCTree.JCIdent;
 32 import com.sun.tools.javac.tree.JCTree.JCLiteral;
 33 import com.sun.tools.javac.tree.JCTree.JCVariableDecl;
 34 import com.sun.tools.javac.tree.TreeInfo;
 35 import com.sun.tools.javac.tree.TreeScanner;
 36 import java.util.Collection;
 37 import java.util.HashMap;
 38 import java.util.Map;
 39 import java.util.Objects;
 40 
 41 /** A tree visitor that computes a hash code. */
 42 public class TreeHasher extends TreeScanner {
 43 
 44     private final Map<Symbol, Integer> symbolHashes;
 45     private int result = 17;
 46 
 47     public TreeHasher(Map<Symbol, Integer> symbolHashes) {
 48         this.symbolHashes = Objects.requireNonNull(symbolHashes);
 49     }
 50 
 51     public static int hash(JCTree tree, Collection<? extends Symbol> symbols) {
 52         if (tree == null) {
 53             return 0;
 54         }
 55         Map<Symbol, Integer> symbolHashes = new HashMap<>();
 56         symbols.forEach(s -> symbolHashes.put(s, symbolHashes.size()));
 57         TreeHasher hasher = new TreeHasher(symbolHashes);
 58         tree.accept(hasher);
 59         return hasher.result;
 60     }
 61 
 62     private void hash(Object object) {
 63         result = 31 * result + Objects.hashCode(object);
 64     }
 65 
 66     @Override
 67     public void scan(JCTree tree) {
 68         if (tree == null) {
 69             return;
 70         }
 71         tree = TreeInfo.skipParens(tree);
 72         if (tree.type != null) {
 73             Object value = tree.type.constValue();
 74             if (value != null) {
 75                 hash(value);
 76                 return;
 77             }
 78         }
 79         hash(tree.getTag());
 80         tree.accept(this);
 81     }
 82 
 83     @Override
 84     public void visitLiteral(JCLiteral tree) {
 85         hash(tree.value);
 86         super.visitLiteral(tree);
 87     }
 88 
 89     @Override
 90     public void visitIdent(JCIdent tree) {
 91         Symbol sym = tree.sym;
 92         if (sym != null) {
 93             Integer hash = symbolHashes.get(sym);
 94             if (hash != null) {
 95                 hash(hash);
 96                 return;
 97             }
 98         }
 99         hash(sym);
100     }
101 
102     @Override
103     public void visitSelect(JCFieldAccess tree) {
104         hash(tree.sym);
105         super.visitSelect(tree);
106     }
107 
108     @Override
109     public void visitBindingPattern(JCTree.JCBindingPattern tree) {
110         symbolHashes.computeIfAbsent(tree.symbol, k -> symbolHashes.size());
111         super.visitBindingPattern(tree);
112     }
113 
114     @Override
115     public void visitVarDef(JCVariableDecl tree) {
116         symbolHashes.computeIfAbsent(tree.sym, k -> symbolHashes.size());
117         super.visitVarDef(tree);
118     }
119 }
