 26 package com.sun.tools.javac.file;
 27 
 28 import java.io.IOException;
 29 import java.nio.file.Files;
 30 import java.nio.file.Path;
 31 import java.nio.file.attribute.BasicFileAttributes;
 32 import java.util.List;
 33 import java.util.Optional;
 34 import java.util.concurrent.ConcurrentHashMap;
 35 
 36 import com.sun.tools.javac.util.Context;
 37 import com.sun.tools.javac.util.Context.Factory;
 38 
 39 /**
 40  * Caching implementation of FSInfo.
 41  *
 42  * <p><b>This is NOT part of any supported API.
 43  * If you write code that depends on this, you do so at your own risk.
 44  * This code and its internal interfaces are subject to change or
 45  * deletion without notice.</b>
 46  */
 47 public class CacheFSInfo extends FSInfo {
 48 
 49     protected final ConcurrentHashMap<Path, Path> canonicalPathCache = new ConcurrentHashMap<>();
 50     protected final ConcurrentHashMap<Path, Optional<BasicFileAttributes>> attributeCache =
 51             new ConcurrentHashMap<>();
 52     protected final ConcurrentHashMap<Path, List<Path>> jarClassPathCache =
 53             new ConcurrentHashMap<>();
 54 
 55     /**
 56      * Register a Context.Factory to create a CacheFSInfo.
 57      */
 58     public static void preRegister(Context context) {
 59         context.put(FSInfo.class, (Factory<FSInfo>)c -> {
 60                 FSInfo instance = new CacheFSInfo();
 61                 c.put(FSInfo.class, instance);
 62                 return instance;
 63             });
 64     }
 65 
 66     public void clearCache() {
 67         canonicalPathCache.clear();
 68         attributeCache.clear();
 69         jarClassPathCache.clear();
 70     }
 71 
 72     @Override
 73     public Path getCanonicalFile(Path file) {
 74         return canonicalPathCache.computeIfAbsent(file, super::getCanonicalFile);

 75     }
 76 
 77     @Override
 78     public boolean exists(Path file) {
 79         return getAttributes(file).isPresent();

 80     }
 81 
 82     @Override
 83     public boolean isDirectory(Path file) {
 84         return getAttributes(file).map(BasicFileAttributes::isDirectory).orElse(false);

 85     }
 86 
 87     @Override
 88     public boolean isFile(Path file) {
 89         return getAttributes(file).map(BasicFileAttributes::isRegularFile).orElse(false);

 90     }
 91 
 92     @Override
 93     public List<Path> getJarClassPath(Path file) throws IOException {
 94         synchronized (jarClassPathCache) {
 95             List<Path> jarClassPath = jarClassPathCache.get(file);
 96             if (jarClassPath == null) {
 97                 jarClassPath = super.getJarClassPath(file);
 98                 jarClassPathCache.put(file, jarClassPath);
 99             }
100             return jarClassPath;














101         }

102     }
103 
104     protected Optional<BasicFileAttributes> getAttributes(Path file) {
105         return attributeCache.computeIfAbsent(file, this::maybeReadAttributes);
106     }
107 
108     protected Optional<BasicFileAttributes> maybeReadAttributes(Path file) {
109         try {
110             return Optional.of(Files.readAttributes(file, BasicFileAttributes.class));
111         } catch (IOException e) {
112             // Ignore; means file not found
113             return Optional.empty();
114         }
115     }
116 }
