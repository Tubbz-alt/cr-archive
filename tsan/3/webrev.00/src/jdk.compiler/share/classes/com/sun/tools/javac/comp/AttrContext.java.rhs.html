 26 package com.sun.tools.javac.comp;
 27 
 28 import com.sun.tools.javac.tree.JCTree;
 29 import com.sun.tools.javac.util.*;
 30 import com.sun.tools.javac.code.*;
 31 import com.sun.tools.javac.code.Scope.WriteableScope;
 32 import com.sun.tools.javac.comp.DeferredAttr.AttributionMode;
 33 
 34 /** Contains information specific to the attribute and enter
 35  *  passes, to be used in place of the generic field in environments.
 36  *
 37  *  <p><b>This is NOT part of any supported API.
 38  *  If you write code that depends on this, you do so at your own risk.
 39  *  This code and its internal interfaces are subject to change or
 40  *  deletion without notice.</b>
 41  */
 42 public class AttrContext {
 43 
 44     /** The scope of local symbols.
 45      */
 46     WriteableScope scope = null;
 47 
 48     /** The number of enclosing `static' modifiers.
 49      */
 50     int staticLevel = 0;
 51 
 52     /** Is this an environment for a this(...) or super(...) call?
 53      */
 54     boolean isSelfCall = false;
 55 
 56     /** Are we evaluating the selector of a `super' or type name?
 57      */
 58     boolean selectSuper = false;
 59 
 60     /** Is the current target of lambda expression or method reference serializable or is this a
 61      *  serializable class?
 62      */
 63     boolean isSerializable = false;
 64 
 65     /** Is this a serializable lambda?
 66      */
 67     boolean isSerializableLambda = false;
 68 
 69     /** Is this a lambda environment?
 70      */
 71     boolean isLambda = false;
 72 
 73     /** Is this a speculative attribution environment?
 74      */
 75     AttributionMode attributionMode = AttributionMode.FULL;
 76 
 77     /**
 78      *  Is this an attribution environment for an anonymous class instantiated using <> ?
 79      */
 80     boolean isAnonymousDiamond = false;
 81 
 82     /**
 83      *  Is this an attribution environment for an instance creation expression?
 84      */
 85     boolean isNewClass = false;
 86 
 87     /** Indicate if the type being visited is a service implementation
 88      */
 89     boolean visitingServiceImplementation = false;
 90 
 91     /** Are arguments to current function applications boxed into an array for varargs?
 92      */
 93     Resolve.MethodResolutionPhase pendingResolutionPhase = null;
 94 
 95     /** A record of the lint/SuppressWarnings currently in effect
 96      */
 97     Lint lint;
 98 
 99     /** The variable whose initializer is being attributed
100      * useful for detecting self-references in variable initializers
101      */
102     Symbol enclVar = null;
103 
104     /** ResultInfo to be used for attributing 'return' statement expressions
105      * (set by Attr.visitMethod and Attr.visitLambda)
106      */
107     Attr.ResultInfo returnResult = null;
108 
109     /** ResultInfo to be used for attributing 'yield' statement expressions
110      * (set by Attr.visitSwitchExpression)
111      */
112     Attr.ResultInfo yieldResult = null;
113 
114     /** Symbol corresponding to the site of a qualified default super call
115      */
116     Type defaultSuperCallSite = null;
117 
118     /** Tree that when non null, is to be preferentially used in diagnostics.
119      *  Usually Env<AttrContext>.tree is the tree to be referred to in messages,
120      *  but this may not be true during the window a method is looked up in enclosing
121      *  contexts (JDK-8145466)
122      */
123     JCTree preferredTreeForDiagnostics;
124 
125     /** Duplicate this context, replacing scope field and copying all others.
126      */
127     AttrContext dup(WriteableScope scope) {
128         AttrContext info = new AttrContext();
129         info.scope = scope;
130         info.staticLevel = staticLevel;
131         info.isSelfCall = isSelfCall;
132         info.selectSuper = selectSuper;
133         info.pendingResolutionPhase = pendingResolutionPhase;
134         info.lint = lint;
135         info.enclVar = enclVar;
136         info.returnResult = returnResult;
137         info.yieldResult = yieldResult;
138         info.defaultSuperCallSite = defaultSuperCallSite;
139         info.isSerializable = isSerializable;
140         info.isLambda = isLambda;
141         info.isSerializableLambda = isSerializableLambda;
142         info.attributionMode = attributionMode;
143         info.isAnonymousDiamond = isAnonymousDiamond;
144         info.isNewClass = isNewClass;
145         info.preferredTreeForDiagnostics = preferredTreeForDiagnostics;
146         info.visitingServiceImplementation = visitingServiceImplementation;
147         return info;
148     }
149 
150     /** Duplicate this context, copying all fields.
151      */
152     AttrContext dup() {
153         return dup(scope);
154     }
155 
156     public Iterable<Symbol> getLocalElements() {
157         if (scope == null)
158             return List.nil();
159         return scope.getSymbols();
160     }
161 
162     boolean lastResolveVarargs() {
163         return pendingResolutionPhase != null &&
164                 pendingResolutionPhase.isVarargsRequired();
165     }
166 
167     @Override
168     public String toString() {
169         return "AttrContext[" + scope.toString() + "]";
170     }
171 }
