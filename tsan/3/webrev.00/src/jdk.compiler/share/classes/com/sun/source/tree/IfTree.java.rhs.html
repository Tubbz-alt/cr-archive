26 package com.sun.source.tree;
27 
28 /**
29  * A tree node for an {@code if} statement.
30  *
31  * For example:
32  * <pre>
33  *   if ( <em>condition</em> )
34  *      <em>thenStatement</em>
35  *
36  *   if ( <em>condition</em> )
37  *       <em>thenStatement</em>
38  *   else
39  *       <em>elseStatement</em>
40  * </pre>
41  *
42  * @jls 14.9 The if Statement
43  *
44  * @author Peter von der Ah&eacute;
45  * @author Jonathan Gibbons
46  * @since 1.6
47  */
48 public interface IfTree extends StatementTree {
49     /**
50      * Returns the condition of the if-statement.
51      * @return the condition
52      */
53     ExpressionTree getCondition();
54 
55     /**
56      * Returns the statement to be executed if the condition is true
57      * @return the statement to be executed if the condition is true
58      */
59     StatementTree getThenStatement();
60 
61     /**
62      * Returns the statement to be executed if the condition is false,
63      * or {@code null} if there is no such statement.
64      * @return the statement to be executed if the condition is false
65      */
66     StatementTree getElseStatement();
67 }
