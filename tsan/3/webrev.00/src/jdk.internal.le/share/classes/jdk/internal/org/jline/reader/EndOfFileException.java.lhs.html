 9 package jdk.internal.org.jline.reader;
10 
11 /**
12  * This exception is thrown by {@link LineReader#readLine} when
13  * user the user types ctrl-D).
14  */
15 public class EndOfFileException extends RuntimeException {
16 
17     private static final long serialVersionUID = 528485360925144689L;
18 
19     public EndOfFileException() {
20     }
21 
22     public EndOfFileException(String message) {
23         super(message);
24     }
25 
26     public EndOfFileException(String message, Throwable cause) {
27         super(message, cause);
28     }
29 
30     public EndOfFileException(Throwable cause) {
31         super(cause);
32     }
33 
34     public EndOfFileException(String message, Throwable cause, boolean enableSuppression, boolean writableStackTrace) {
35         super(message, cause, enableSuppression, writableStackTrace);
36     }
37 }
