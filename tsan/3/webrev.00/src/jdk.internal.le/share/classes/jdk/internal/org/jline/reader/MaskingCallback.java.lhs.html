 9 package jdk.internal.org.jline.reader;
10 
11 /**
12  * Callback used to mask parts of the line
13  */
14 public interface MaskingCallback {
15 
16     /**
17      * Transforms the line before it is displayed so that
18      * some parts can be hidden.
19      *
20      * @param line the current line being edited
21      * @return the modified line to display
22      */
23     String display(String line);
24 
25     /**
26      * Transforms the line before storing in the history.
27      * If the return value is empty or null, it will not be saved
28      * in the history.
29      *
30      * @param line the line to be added to history
31      * @return the modified line
32      */
33     String history(String line);
34 
35 }
