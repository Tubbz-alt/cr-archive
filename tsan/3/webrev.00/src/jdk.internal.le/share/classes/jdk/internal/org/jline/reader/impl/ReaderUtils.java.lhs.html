 9 package jdk.internal.org.jline.reader.impl;
10 
11 import jdk.internal.org.jline.reader.LineReader;
12 
13 public class ReaderUtils {
14 
15     private ReaderUtils() { }
16 
17     public static boolean isSet(LineReader reader, LineReader.Option option) {
18         return reader != null && reader.isSet(option);
19     }
20 
21     public static String getString(LineReader reader, String name, String def) {
22         Object v = reader != null ? reader.getVariable(name) : null;
23         return v != null ? v.toString() : def;
24     }
25 
26     public static boolean getBoolean(LineReader reader, String name, boolean def) {
27         Object v = reader != null ? reader.getVariable(name) : null;
28         if (v instanceof Boolean) {
29             return (Boolean) v;
30         } else if (v != null) {
31             String s = v.toString();
32             return s.isEmpty() || s.equalsIgnoreCase("on")
33                     || s.equalsIgnoreCase("1") || s.equalsIgnoreCase("true");
34         }
35         return def;
36     }
37 
38     public static int getInt(LineReader reader, String name, int def) {
39         int nb = def;
40         Object v = reader != null ? reader.getVariable(name) : null;
41         if (v instanceof Number) {
42             return ((Number) v).intValue();
43         } else if (v != null) {
44             nb = 0;
45             try {
46                 nb = Integer.parseInt(v.toString());
47             } catch (NumberFormatException e) {
48                 // Ignore
49             }
50         }
51         return nb;
52     }
53 
54     public static long getLong(LineReader reader, String name, long def) {
55         long nb = def;
56         Object v = reader != null ? reader.getVariable(name) : null;
57         if (v instanceof Number) {
58             return ((Number) v).longValue();
59         } else if (v != null) {
60             nb = 0;
61             try {
62                 nb = Long.parseLong(v.toString());
63             } catch (NumberFormatException e) {
64                 // Ignore
65             }
66         }
67         return nb;
68     }
69 
70 }
