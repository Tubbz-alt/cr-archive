 9 package jdk.internal.org.jline.reader.impl.completer;
10 
11 import java.util.Arrays;
12 import java.util.Collection;
13 import java.util.List;
14 import java.util.Objects;
15 
16 import jdk.internal.org.jline.reader.Candidate;
17 import jdk.internal.org.jline.reader.Completer;
18 import jdk.internal.org.jline.reader.LineReader;
19 import jdk.internal.org.jline.reader.ParsedLine;
20 
21 /**
22  * Completer which contains multiple completers and aggregates them together.
23  *
24  * @author <a href="mailto:jason@planet57.com">Jason Dillon</a>
25  * @since 2.3
26  */
27 public class AggregateCompleter
28     implements Completer
29 {
30     private final Collection<Completer> completers;
31 
32     /**
33      * Construct an AggregateCompleter with the given completers.
34      * The completers will be used in the order given.
35      *
36      * @param completers the completers
37      */
38     public AggregateCompleter(final Completer... completers) {
39         this(Arrays.asList(completers));
40     }
41 
42     /**
43      * Construct an AggregateCompleter with the given completers.
44      * The completers will be used in the order given.
45      *
46      * @param completers the completers
47      */
48     public AggregateCompleter(Collection<Completer> completers) {
49         assert completers != null;
50         this.completers = completers;
51     }
52 
53     /**
54      * Retrieve the collection of completers currently being aggregated.
55      *
56      * @return the aggregated completers
57      */
58     public Collection<Completer> getCompleters() {
59         return completers;
60     }
61 
62     /**
63      * Perform a completion operation across all aggregated completers.
64      *
65      * The effect is similar to the following code:
66      * <blockquote><pre>{@code completers.forEach(c -> c.complete(reader, line, candidates));}</pre></blockquote>
67      *
68      * @see Completer#complete(LineReader, ParsedLine, List)
69      */
70     public void complete(LineReader reader, final ParsedLine line, final List<Candidate> candidates) {
71         Objects.requireNonNull(line);
72         Objects.requireNonNull(candidates);
73         completers.forEach(c -> c.complete(reader, line, candidates));
74     }
75 
76     /**
77      * @return a string representing the aggregated completers
78      */
79     @Override
80     public String toString() {
81         return getClass().getSimpleName() + "{" +
82             "completers=" + completers +
83             '}';
84     }
85 
86 }
