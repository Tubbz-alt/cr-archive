 9 package jdk.internal.org.jline.terminal;
10 
11 /**
12  * Class holding the cursor position.
13  *
14  * @see Terminal#getCursorPosition(java.util.function.IntConsumer)
15  */
16 public class Cursor {
17 
18     private final int x;
19     private final int y;
20 
21     public Cursor(int x, int y) {
22         this.x = x;
23         this.y = y;
24     }
25 
26     public int getX() {
27         return x;
28     }
29 
30     public int getY() {
31         return y;
32     }
33 
34     @Override
35     public boolean equals(Object o) {
36         if (o instanceof Cursor) {
37             Cursor c = (Cursor) o;
38             return x == c.x && y == c.y;
39         } else {
40             return false;
41         }
42     }
43 
44     @Override
45     public int hashCode() {
46         return x * 31 + y;
47     }
48 
49     @Override
50     public String toString() {
51         return "Cursor[" + "x=" + x + ", y=" + y + ']';
52     }
53 }
