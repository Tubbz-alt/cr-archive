 60     /**
 61      * Helper to support rendering messages.
 62      */
 63     static void render(final PrintStream out, final Object message) {
 64         if (message != null && message.getClass().isArray()) {
 65             Object[] array = (Object[]) message;
 66 
 67             out.print("[");
 68             for (int i = 0; i < array.length; i++) {
 69                 out.print(array[i]);
 70                 if (i + 1 < array.length) {
 71                     out.print(",");
 72                 }
 73             }
 74             out.print("]");
 75         }
 76         else {
 77             out.print(message);
 78         }
 79     }
