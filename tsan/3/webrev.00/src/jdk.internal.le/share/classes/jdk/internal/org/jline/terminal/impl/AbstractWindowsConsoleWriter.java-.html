 9 package jdk.internal.org.jline.terminal.impl;
10 
11 import java.io.IOException;
12 import java.io.Writer;
13 
14 public abstract class AbstractWindowsConsoleWriter extends Writer {
15 
16     protected abstract void writeConsole(char[] text, int len) throws IOException;
17 
18     @Override
19     public void write(char[] cbuf, int off, int len) throws IOException {
20         char[] text = cbuf;
21         if (off != 0) {
22             text = new char[len];
23             System.arraycopy(cbuf, off, text, 0, len);
24         }
25 
26         synchronized (this.lock) {
27             writeConsole(text, len);
28         }
29     }
30 
31     @Override
32     public void flush() {
33     }
34 
35     @Override
36     public void close() {
37     }
38 
39 }
