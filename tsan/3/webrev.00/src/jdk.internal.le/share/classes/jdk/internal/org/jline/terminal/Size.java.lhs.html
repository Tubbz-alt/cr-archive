 9 package jdk.internal.org.jline.terminal;
10 
11 public class Size {
12 
13     private int rows;
14     private int cols;
15 
16     public Size() {
17     }
18 
19     public Size(int columns, int rows) {
20         this();
21         setColumns(columns);
22         setRows(rows);
23     }
24 
25     public int getColumns() {
26         return cols;
27     }
28 
29     public void setColumns(int columns) {
30         cols = (short) columns;
31     }
32 
33     public int getRows() {
34         return rows;
35     }
36 
37     public void setRows(int rows) {
38         this.rows = (short) rows;
39     }
40 
41     /**
42      * A cursor position combines a row number with a column position.
43      * <p>
44      * Note each row has {@code col+1} different column positions,
45      * including the right margin.
46      * </p>
47      *
48      * @param col the new column
49      * @param row the new row
50      * @return the cursor position
51      */
52     public int cursorPos(int row, int col) {
53         return row * (cols+1) + col;
54     }
55 
56     public void copy(Size size) {
57         setColumns(size.getColumns());
58         setRows(size.getRows());
59     }
60 
61     @Override
62     public boolean equals(Object o) {
63         if (o instanceof Size) {
64             Size size = (Size) o;
65             return rows == size.rows && cols == size.cols;
66         } else {
67             return false;
68         }
69     }
70 
71     @Override
72     public int hashCode() {
73         return rows * 31 + cols;
74     }
75 
76     @Override
77     public String toString() {
78         return "Size[" + "cols=" + cols + ", rows=" + rows + ']';
79     }
80 }
