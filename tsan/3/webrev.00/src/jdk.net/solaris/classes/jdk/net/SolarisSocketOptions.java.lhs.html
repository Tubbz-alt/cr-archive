26 package jdk.net;
27 
28 import java.net.SocketException;
29 import java.security.AccessController;
30 import java.security.PrivilegedAction;
31 import jdk.net.ExtendedSocketOptions.PlatformSocketOptions;
32 

33 class SolarisSocketOptions extends PlatformSocketOptions {
34 
35     public SolarisSocketOptions() { }
36 
37     @Override native int setFlowOption(int fd, int priority, long bandwidth)
38             throws SocketException;
39 
40     @Override native int getFlowOption(int fd, SocketFlow f)
41             throws SocketException;
42 
43     @Override native boolean flowSupported();
44 
45     private static native void init();
46 
47     static {
48         AccessController.doPrivileged(new PrivilegedAction<Void>() {
49             public Void run() {
50                 System.loadLibrary("extnet");
51                 return null;
52             }
53         });




54         init();
55     }
56 }
