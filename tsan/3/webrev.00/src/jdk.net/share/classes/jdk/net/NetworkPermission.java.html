26 package jdk.net;
27 
28 import java.security.BasicPermission;
29 
30 /**
31  * Represents permission to access the extended networking capabilities
32  * defined in the jdk.net package. These permissions contain a target
33  * name, but no actions list. Callers either possess the permission or not.
34  * <p>
35  * The following targets are defined:
36  *
37  * <table class="striped"><caption style="display:none">permission target name,
38  *  what the target allows,and associated risks</caption>
39  * <thead>
40  * <tr>
41  *   <th scope="col">Permission Target Name</th>
42  *   <th scope="col">What the Permission Allows</th>
43  *   <th scope="col">Risks of Allowing this Permission</th>
44  * </tr>
45  * </thead>
46  * <tbody>
47  * <tr>
48  *   <th scope="row">setOption.SO_FLOW_SLA</th>
49  *   <td>set the {@link ExtendedSocketOptions#SO_FLOW_SLA SO_FLOW_SLA} option
50  *       on any socket that supports it</td>
51  *   <td>allows caller to set a higher priority or bandwidth allocation
52  *       to sockets it creates, than they might otherwise be allowed.
53  *       This permission is deprecated.</td>
54  * </tr>
55  * <tr>
56  *   <th scope="row">getOption.SO_FLOW_SLA</th>
57  *   <td>retrieve the {@link ExtendedSocketOptions#SO_FLOW_SLA SO_FLOW_SLA}
58  *       setting from any socket that supports the option</td>
59  *   <td>allows caller access to SLA information that it might not
60  *       otherwise have. This permission is deprecated.</td>
61  * </tr>
62  * </tbody>
63  * </table>
64  *
65  * @see jdk.net.ExtendedSocketOptions
66  *
67  * @since 1.8
68  */
69 
70 public final class NetworkPermission extends BasicPermission {
71 
72     private static final long serialVersionUID = -2012939586906722291L;
73 
74     /**
75      * Creates a NetworkPermission with the given target name.
76      *
77      * @param name the permission target name
78      * @throws NullPointerException if {@code name} is {@code null}.
79      * @throws IllegalArgumentException if {@code name} is empty.
80      */
81     public NetworkPermission(String name)
82     {
83         super(name);
84     }
85 
86     /**
87      * Creates a NetworkPermission with the given target name.
88      *
89      * @param name the permission target name
90      * @param actions should be {@code null}. Is ignored if not.
91      * @throws NullPointerException if {@code name} is {@code null}.
92      * @throws IllegalArgumentException if {@code name} is empty.
93      */
94     public NetworkPermission(String name, String actions)
95     {
96         super(name, actions);
97     }
98 }
