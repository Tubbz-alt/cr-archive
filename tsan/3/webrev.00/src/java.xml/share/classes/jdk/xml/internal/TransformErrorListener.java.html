26 package jdk.xml.internal;
27 
28 import javax.xml.transform.ErrorListener;
29 import javax.xml.transform.TransformerException;
30 
31 /**
32  * Implements an ErrorListener for use by the JDK as the default ErrorListener.
33  * For compatibility, this implementation retains the behavior as was implemented
34  * by TransformerFactoryImpl and TransformerImpl where both the error and
35  * fatalError methods re-throw the exception.
36  */
37 public class TransformErrorListener implements ErrorListener {
38     /**
39      * Receives notification of a warning.
40      *
41      * @param e The warning information encapsulated in a TransformerException.
42      * @throws TransformerException not thrown in this implementation
43      */
44     @Override
45     public void warning(TransformerException e)
46         throws TransformerException
47     {
48         // no op
49     }
50 
51     /**
52      * Receives notification of an error.
53      * The transformer may continue the process if the error is recoverable.
54      * It may decide not to if it can not continue after the error.
55      *
56      * @param e The error information encapsulated in a TransformerException.
57      * @throws TransformerException re-throws the exception.
58      */
59     @Override
60     public void error(TransformerException e)
61         throws TransformerException
62     {
63         throw e;
64     }
65 
66     /**
67      * Receives notification of a fatal error.
68      *
69      * @param e The error information encapsulated in a TransformerException.
70      * @throws TransformerException re-throws the exception.
71      */
72     @Override
73     public void fatalError(TransformerException e)
74         throws TransformerException
75     {
76         throw e;
77     }
78 }
