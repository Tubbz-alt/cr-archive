 26 package javax.xml.transform;
 27 
 28 /**
 29  * The listener interface used by a {@link TransformerFactory} or {@link Transformer}
 30  * to notify callers of error messages that occur during a transformation process.
 31  * An ErrorListener receives three levels of messages: warnings, errors and fatal
 32  * errors as classified by their severity. Each of them is handled as described
 33  * in their respective method.
 34  *
 35  * <p>
 36  * An ErrorListener instance can be registered to a {@link TransformerFactory}
 37  * or {@link Transformer} through
 38  * the {@link TransformerFactory#setErrorListener(ErrorListener)}
 39  * or {@link Transformer#setErrorListener(ErrorListener)}
 40  * method to receive errors and warnings reported by the TransformerFactory
 41  * or Transformer.
 42  *
 43  * <p>
 44  * When a listener is registered, the {@link TransformerFactory} or {@link Transformer}
 45  * must use this interface to pass on all warnings and errors to the listener
 46  * and let the application decide how to handle them.
 47  * Note that the {@code TransformerFactory} or {@code Transformer} is not
 48  * required to continue with the transformation after a call to
 49  * {@link #fatalError(TransformerException exception)}.
 50  *
 51  * <p>
 52  * If an application does not provide a listener, the {@link TransformerFactory}
 53  * or {@link Transformer} shall create one on its own. The default {@code ErrorListener}
 54  * may take no action for warnings and recoverable errors, and allow the
 55  * transformation to continue.
 56  * However, the {@code TransformerFactory} or {@code Transformer} may still throw
 57  * {@code TransformerException} when it decides it can not continue processing.
 58  *
 59  * @apiNote It is recommended that applications register and use their own
 60  * {@code ErrorListener} to override the default behavior in order to ensure
 61  * proper handling of warnings and errors.
 62  *
 63  * @since 1.4
 64  */
 65 public interface ErrorListener {
 66 
 67     /**
 68      * Receive notification of a warning.
 69      *
 70      * <p>{@link javax.xml.transform.Transformer} can use this method to report
 71      * conditions that are not errors or fatal errors.  The default behaviour
 72      * is to take no action.</p>
 73      *
 74      * <p>After invoking this method, the Transformer must continue with
 75      * the transformation. It should still be possible for the
 76      * application to process the document through to the end.</p>
 77      *
 78      * @param exception The warning information encapsulated in a
 79      *                  transformer exception.
 80      *
 81      * @throws javax.xml.transform.TransformerException if the application
 82      * chooses to discontinue the transformation.
 83      *
 84      * @see javax.xml.transform.TransformerException
 85      */
 86     public abstract void warning(TransformerException exception)
 87         throws TransformerException;
 88 
 89     /**
 90      * Receive notification of a recoverable error.
 91      *
 92      * <p>The transformer must continue to try and provide normal transformation
 93      * after invoking this method.  It should still be possible for the
 94      * application to process the document through to the end if no other errors
 95      * are encountered.</p>
 96      *
 97      * @param exception The error information encapsulated in a
 98      *                  transformer exception.
 99      *
100      * @throws javax.xml.transform.TransformerException if the application
101      * chooses to discontinue the transformation.
102      *
103      * @see javax.xml.transform.TransformerException
104      */
105     public abstract void error(TransformerException exception)
106         throws TransformerException;
107 
108     /**
109      * <p>Receive notification of a non-recoverable error.</p>
110      *
111      * <p>The processor may choose to continue, but will not normally
112      * proceed to a successful completion.</p>
113      *
114      * <p>The method should throw an exception if it is unable to
115      * process the error, or if it wishes execution to terminate
116      * immediately. The processor will not necessarily honor this
117      * request.</p>
118      *
119      * @param exception The error information encapsulated in a
120      *    <code>TransformerException</code>.
121      *
122      * @throws javax.xml.transform.TransformerException if the application
123      * chooses to discontinue the transformation.
124      *
125      * @see javax.xml.transform.TransformerException
126      */
127     public abstract void fatalError(TransformerException exception)
128         throws TransformerException;
129 }
