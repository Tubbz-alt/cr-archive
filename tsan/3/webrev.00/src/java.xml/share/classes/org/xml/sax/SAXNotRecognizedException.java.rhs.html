26 package org.xml.sax;

29 /**
30  * Exception class for an unrecognized identifier.
31  *







32  * <p>An XMLReader will throw this exception when it finds an
33  * unrecognized feature or property identifier; SAX applications and
34  * extensions may use this class for other, similar purposes.</p>
35  *
36  * @since 1.4, SAX 2.0
37  * @author David Megginson
38  * @see org.xml.sax.SAXNotSupportedException
39  */
40 public class SAXNotRecognizedException extends SAXException
41 {
42 
43     /**
44      * Default constructor.
45      */
46     public SAXNotRecognizedException ()
47     {
48         super();
49     }
50 
51 
52     /**
53      * Construct a new exception with the given message.
54      *
55      * @param message The text message of the exception.
56      */
57     public SAXNotRecognizedException (String message)
58     {
59         super(message);
60     }
61 
62     // Added serialVersionUID to preserve binary compatibility
63     static final long serialVersionUID = 5440506620509557213L;
64 }
