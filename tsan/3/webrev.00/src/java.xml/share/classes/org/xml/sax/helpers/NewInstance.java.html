26 package org.xml.sax.helpers;
27 
28 import java.lang.reflect.InvocationTargetException;
29 import java.util.Objects;
30 
31 /**
32  * Create a new instance of a class by name.
33  *
34  *
35  * <p>This class contains a static method for creating an instance of a
36  * class from an explicit class name.  It tries to use the thread's context
37  * ClassLoader if possible and falls back to using
38  * Class.forName(String).</p>
39  *
40  * <p>This code is designed to compile and run on JDK version 1.1 and later
41  * including versions of Java 2.</p>
42  *
43  * @author Edwin Goei, David Brownell
44  * @version 2.0.1 (sax2r2)
45  */
46 class NewInstance {
47     private static final String DEFAULT_PACKAGE = "com.sun.org.apache.xerces.internal";
48     private static final String DEFAULT_CLASS = "com.sun.org.apache.xerces.internal.parsers.SAXParser";
49     /**
50      * Creates a new instance of the specified class name
51      *
52      * Package private so this code is not exposed at the API level.
53      */
54     static <T> T newInstance (Class<T> type, ClassLoader loader, String clsName)
55         throws ClassNotFoundException, IllegalAccessException,
56             InstantiationException
57     {
58         ClassLoader classLoader = Objects.requireNonNull(loader);
59         String className = Objects.requireNonNull(clsName);
60 
61         // Instantiate directly for the SAX default parser
62         if (className.equals(DEFAULT_CLASS)) {
63             return type.cast(new com.sun.org.apache.xerces.internal.parsers.SAXParser());
64         }
65 
66         // make sure we have access to restricted packages
67         boolean internal = false;
68         if (System.getSecurityManager() != null) {
69             if (className != null && className.startsWith(DEFAULT_PACKAGE)) {
70                 internal = true;
71             }
72         }
73 
74         Class<?> driverClass;
75         if (classLoader == null || internal) {
76             driverClass = Class.forName(className);
77         } else {
78             driverClass = classLoader.loadClass(className);
79         }
80 
81         try {
82             return type.cast(driverClass.getConstructor().newInstance());
83         } catch (NoSuchMethodException | SecurityException | InvocationTargetException ex) {
84             throw new InstantiationException(ex.getMessage());
85         }
86     }
87 
88 }
