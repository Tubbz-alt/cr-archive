 33 package org.xml.sax.helpers;
 34 
 35 import java.lang.reflect.InvocationTargetException;
 36 import java.util.Objects;
 37 
 38 /**
 39  * Create a new instance of a class by name.
 40  *
 41  * <blockquote>
 42  * <em>This module, both source code and documentation, is in the
 43  * Public Domain, and comes with <strong>NO WARRANTY</strong>.</em>
 44  * See <a href='http://www.saxproject.org'>http://www.saxproject.org</a>
 45  * for further information.
 46  * </blockquote>
 47  *
 48  * <p>This class contains a static method for creating an instance of a
 49  * class from an explicit class name.  It tries to use the thread's context
 50  * ClassLoader if possible and falls back to using
 51  * Class.forName(String).</p>
 52  *
 53  * <p>This code is designed to compile and run on JDK version 1.1 and later
 54  * including versions of Java 2.</p>
 55  *
 56  * @author Edwin Goei, David Brownell
 57  * @version 2.0.1 (sax2r2)
 58  */
 59 class NewInstance {
 60     private static final String DEFAULT_PACKAGE = "com.sun.org.apache.xerces.internal";
 61     private static final String DEFAULT_CLASS = "com.sun.org.apache.xerces.internal.parsers.SAXParser";
 62     /**
 63      * Creates a new instance of the specified class name
 64      *
 65      * Package private so this code is not exposed at the API level.
 66      */
 67     static <T> T newInstance (Class<T> type, ClassLoader loader, String clsName)
 68         throws ClassNotFoundException, IllegalAccessException,
 69             InstantiationException
 70     {
 71         ClassLoader classLoader = Objects.requireNonNull(loader);
 72         String className = Objects.requireNonNull(clsName);
 73 
 74         // Instantiate directly for the SAX default parser
 75         if (className.equals(DEFAULT_CLASS)) {
 76             return type.cast(new com.sun.org.apache.xerces.internal.parsers.SAXParser());
 77         }
 78 
 79         // make sure we have access to restricted packages
 80         boolean internal = false;
 81         if (System.getSecurityManager() != null) {
 82             if (className != null && className.startsWith(DEFAULT_PACKAGE)) {
 83                 internal = true;
 84             }
 85         }
 86 
 87         Class<?> driverClass;
 88         if (classLoader == null || internal) {
 89             driverClass = Class.forName(className);
 90         } else {
 91             driverClass = classLoader.loadClass(className);
 92         }
 93 
 94         try {
 95             return type.cast(driverClass.getConstructor().newInstance());
 96         } catch (NoSuchMethodException | SecurityException | InvocationTargetException ex) {
 97             throw new InstantiationException(ex.getMessage());
 98         }
 99     }
100 
101 }
