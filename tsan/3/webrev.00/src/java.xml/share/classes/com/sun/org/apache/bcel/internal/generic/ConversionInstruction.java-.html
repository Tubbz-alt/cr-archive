22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.Const;
25 
26 /**
27  * Super class for the x2y family of instructions.
28  *
29  * @version $Id: ConversionInstruction.java 1747278 2016-06-07 17:28:43Z britter $
30  */
31 public abstract class ConversionInstruction extends Instruction implements TypedInstruction,
32         StackProducer, StackConsumer {
33 
34     /**
35      * Empty constructor needed for the Class.newInstance() statement in
36      * Instruction.readInstruction(). Not to be used otherwise.
37      */
38     ConversionInstruction() {
39     }
40 
41 
42     /**
43      * @param opcode opcode of instruction
44      */
45     protected ConversionInstruction(final short opcode) {
46         super(opcode, (short) 1);
47     }
48 
49 
50     /** @return type associated with the instruction
51      */
52     @Override
53     public Type getType( final ConstantPoolGen cp ) {
54         final short _opcode = super.getOpcode();
55         switch (_opcode) {
56             case Const.D2I:
57             case Const.F2I:
58             case Const.L2I:
59                 return Type.INT;
60             case Const.D2F:
61             case Const.I2F:
62             case Const.L2F:
63                 return Type.FLOAT;
64             case Const.D2L:
65             case Const.F2L:
66             case Const.I2L:
67                 return Type.LONG;
68             case Const.F2D:
69             case Const.I2D:
70             case Const.L2D:
71                 return Type.DOUBLE;
72             case Const.I2B:
73                 return Type.BYTE;
74             case Const.I2C:
75                 return Type.CHAR;
76             case Const.I2S:
77                 return Type.SHORT;
78             default: // Never reached
79                 throw new ClassGenException("Unknown type " + _opcode);
80         }
81     }
82 }
