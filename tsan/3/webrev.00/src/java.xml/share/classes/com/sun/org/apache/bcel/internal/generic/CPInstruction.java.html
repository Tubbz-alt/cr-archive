 20 package com.sun.org.apache.bcel.internal.generic;
 21 
 22 import java.io.DataOutputStream;
 23 import java.io.IOException;
 24 
 25 import com.sun.org.apache.bcel.internal.classfile.Constant;
 26 import com.sun.org.apache.bcel.internal.classfile.ConstantClass;
 27 import com.sun.org.apache.bcel.internal.classfile.ConstantPool;
 28 import com.sun.org.apache.bcel.internal.util.ByteSequence;
 29 
 30 /**
 31  * Abstract super class for instructions that use an index into the
 32  * constant pool such as LDC, INVOKEVIRTUAL, etc.
 33  *
 34  * @see ConstantPoolGen
 35  * @see LDC
 36  * @see INVOKEVIRTUAL
 37  *
 38  * @LastModified: Jan 2020
 39  */
 40 public abstract class CPInstruction extends Instruction implements TypedInstruction,
 41         IndexedInstruction {
 42 
 43     private int index; // index to constant pool
 44 
 45 
 46     /**
 47      * Empty constructor needed for Instruction.readInstruction.
 48      * Not to be used otherwise.
 49      */
 50     CPInstruction() {
 51     }
 52 
 53 
 54     /**
 55      * @param index to constant pool
 56      */
 57     protected CPInstruction(final short opcode, final int index) {
 58         super(opcode, (short) 3);
 59         setIndex(index);
 60     }
 61 
 62 
 63     /**
 64      * Dump instruction as byte code to stream out.
 65      * @param out Output stream
 66      */
 67     @Override
 68     public void dump( final DataOutputStream out ) throws IOException {
 69         out.writeByte(super.getOpcode());
 70         out.writeShort(index);
 71     }
 72 
 73 
 74     /**
 75      * Long output format:
 76      *
 77      * &lt;name of opcode&gt; "["&lt;opcode number&gt;"]"
 78      * "("&lt;length of instruction&gt;")" "&lt;"&lt; constant pool index&gt;"&gt;"
 79      *
 80      * @param verbose long/short format switch
 81      * @return mnemonic for instruction
 82      */
 83     @Override
 84     public String toString( final boolean verbose ) {
 85         return super.toString(verbose) + " " + index;
 86     }
 87 
 88 
 89     /**
 90      * @return mnemonic for instruction with symbolic references resolved
 91      */
 92     @Override
 93     public String toString( final ConstantPool cp ) {
 94         final Constant c = cp.getConstant(index);
 95         String str = cp.constantToString(c);
 96         if (c instanceof ConstantClass) {
 97             str = str.replace('.', '/');
 98         }
 99         return com.sun.org.apache.bcel.internal.Const.getOpcodeName(super.getOpcode()) + " " + str;
100     }
101 
102 
103     /**
104      * Read needed data (i.e., index) from file.
105      * @param bytes input stream
106      * @param wide wide prefix?
107      */
108     @Override
109     protected void initFromFile( final ByteSequence bytes, final boolean wide ) throws IOException {
110         setIndex(bytes.readUnsignedShort());
111         super.setLength(3);
112     }
113 
114 
115     /**
116      * @return index in constant pool referred by this instruction.
117      */
118     @Override
119     public final int getIndex() {
120         return index;
121     }
122 
123 
124     /**
125      * Set the index to constant pool.
126      * @param index in  constant pool.
127      */
128     @Override
129     public void setIndex( final int index ) { // TODO could be package-protected?
130         if (index < 0) {
131             throw new ClassGenException("Negative index value: " + index);
132         }
133         this.index = index;
134     }
135 
136 
137     /** @return type related with this instruction.
138      */
139     @Override
140     public Type getType( final ConstantPoolGen cpg ) {
141         final ConstantPool cp = cpg.getConstantPool();
142         String name = cp.getConstantString(index, com.sun.org.apache.bcel.internal.Const.CONSTANT_Class);
143         if (!name.startsWith("[")) {
144             name = "L" + name + ";";
145         }
146         return Type.getType(name);
147     }
148 }
