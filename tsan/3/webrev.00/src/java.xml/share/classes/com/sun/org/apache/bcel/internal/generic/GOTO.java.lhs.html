 22 package com.sun.org.apache.bcel.internal.generic;
 23 
 24 import java.io.DataOutputStream;
 25 import java.io.IOException;
 26 
 27 /**
 28  * GOTO - Branch always (to relative offset, not absolute address)
 29  *
 30  * @version $Id: GOTO.java 1749603 2016-06-21 20:50:19Z ggregory $
 31  */
 32 public class GOTO extends GotoInstruction implements VariableLengthInstruction {
 33 
 34     /**
 35      * Empty constructor needed for the Class.newInstance() statement in
 36      * Instruction.readInstruction(). Not to be used otherwise.
 37      */
 38     GOTO() {
 39     }
 40 
 41 
 42     public GOTO(final InstructionHandle target) {
 43         super(com.sun.org.apache.bcel.internal.Const.GOTO, target);
 44     }
 45 
 46 
 47     /**
 48      * Dump instruction as byte code to stream out.
 49      * @param out Output stream
 50      */
 51     @Override
 52     public void dump( final DataOutputStream out ) throws IOException {
 53         super.setIndex(getTargetOffset());
 54         final short _opcode = getOpcode();
 55         if (_opcode == com.sun.org.apache.bcel.internal.Const.GOTO) {
 56             super.dump(out);
 57         } else { // GOTO_W
 58             super.setIndex(getTargetOffset());
 59             out.writeByte(_opcode);
 60             out.writeInt(super.getIndex());
 61         }
 62     }
 63 
 64 
 65     /**
 66      * Called in pass 2 of InstructionList.setPositions() in order to update
 67      * the branch target, that may shift due to variable length instructions.
 68      *
 69      * @param offset additional offset caused by preceding (variable length) instructions
 70      * @param max_offset the maximum offset that may be caused by these instructions
 71      * @return additional offset caused by possible change of this instruction's length
 72      */
 73     @Override
 74     protected int updatePosition( final int offset, final int max_offset ) {
 75         final int i = getTargetOffset(); // Depending on old position value
 76         setPosition(getPosition() + offset); // Position may be shifted by preceding expansions
 77         if (Math.abs(i) >= (Short.MAX_VALUE - max_offset)) { // to large for short (estimate)
 78             super.setOpcode(com.sun.org.apache.bcel.internal.Const.GOTO_W);
 79             final short old_length = (short) super.getLength();
 80             super.setLength(5);
 81             return super.getLength() - old_length;
 82         }
 83         return 0;
 84     }
 85 
 86 
 87     /**
 88      * Call corresponding visitor method(s). The order is:
 89      * Call visitor methods of implemented interfaces first, then
 90      * call methods according to the class hierarchy in descending order,
 91      * i.e., the most specific visitXXX() call comes last.
 92      *
 93      * @param v Visitor object
 94      */
 95     @Override
 96     public void accept( final Visitor v ) {
 97         v.visitVariableLengthInstruction(this);
 98         v.visitUnconditionalBranch(this);
 99         v.visitBranchInstruction(this);
100         v.visitGotoInstruction(this);
101         v.visitGOTO(this);
102     }
103 }
