22 package com.sun.org.apache.bcel.internal.util;
23 
24 import java.util.Collection;
25 import java.util.HashMap;
26 import java.util.Map;
27 
28 import com.sun.org.apache.bcel.internal.classfile.JavaClass;
29 
30 /**
31  * Utility class implementing a (typesafe) set of JavaClass objects.
32  * Since JavaClass has no equals() method, the name of the class is
33  * used for comparison.
34  *
35  * @see ClassStack
36  */
37 public class ClassSet {
38 
39     private final Map<String, JavaClass> map = new HashMap<>();
40 
41 
42     public boolean add( final JavaClass clazz ) {
43         boolean result = false;
44         if (!map.containsKey(clazz.getClassName())) {
45             result = true;
46             map.put(clazz.getClassName(), clazz);
47         }
48         return result;
49     }
50 
51 
52     public void remove( final JavaClass clazz ) {
53         map.remove(clazz.getClassName());
54     }
55 
56 
57     public boolean empty() {
58         return map.isEmpty();
59     }
60 
61 
62     public JavaClass[] toArray() {
63         final Collection<JavaClass> values = map.values();
64         final JavaClass[] classes = new JavaClass[values.size()];
65         values.toArray(classes);
66         return classes;
67     }
68 
69 
70     public String[] getClassNames() {
71         return map.keySet().toArray(new String[map.size()]);
72     }
73 }
