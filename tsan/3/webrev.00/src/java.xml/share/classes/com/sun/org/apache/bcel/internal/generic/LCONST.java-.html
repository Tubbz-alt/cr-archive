21 package com.sun.org.apache.bcel.internal.generic;
22 
23 /**
24  * LCONST - Push 0 or 1, other values cause an exception
25  *
26  * <PRE>Stack: ... -&gt; ..., </PRE>
27  *
28  * @version $Id: LCONST.java 1747278 2016-06-07 17:28:43Z britter $
29  */
30 public class LCONST extends Instruction implements ConstantPushInstruction {
31 
32     private long value;
33 
34 
35     /**
36      * Empty constructor needed for the Class.newInstance() statement in
37      * Instruction.readInstruction(). Not to be used otherwise.
38      */
39     LCONST() {
40     }
41 
42 
43     public LCONST(final long l) {
44         super(com.sun.org.apache.bcel.internal.Const.LCONST_0, (short) 1);
45         if (l == 0) {
46             super.setOpcode(com.sun.org.apache.bcel.internal.Const.LCONST_0);
47         } else if (l == 1) {
48             super.setOpcode(com.sun.org.apache.bcel.internal.Const.LCONST_1);
49         } else {
50             throw new ClassGenException("LCONST can be used only for 0 and 1: " + l);
51         }
52         value = l;
53     }
54 
55 
56     @Override
57     public Number getValue() {
58         return Long.valueOf(value);
59     }
60 
61 
62     /** @return Type.LONG
63      */
64     @Override
65     public Type getType( final ConstantPoolGen cp ) {
66         return Type.LONG;
67     }
68 
69 
70     /**
71      * Call corresponding visitor method(s). The order is:
72      * Call visitor methods of implemented interfaces first, then
73      * call methods according to the class hierarchy in descending order,
74      * i.e., the most specific visitXXX() call comes last.
75      *
76      * @param v Visitor object
77      */
78     @Override
79     public void accept( final Visitor v ) {
80         v.visitPushInstruction(this);
81         v.visitStackProducer(this);
82         v.visitTypedInstruction(this);
83         v.visitConstantPushInstruction(this);
84         v.visitLCONST(this);
85     }
86 }
