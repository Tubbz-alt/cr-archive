25 /**
26  * MONITORENTER - Enter monitor for object
27  * <PRE>Stack: ..., objectref -&gt; ...</PRE>
28  *
29  * @LastModified: Jan 2020

30  */
31 public class MONITORENTER extends Instruction implements ExceptionThrower, StackConsumer {
32 
33     public MONITORENTER() {
34         super(com.sun.org.apache.bcel.internal.Const.MONITORENTER, (short) 1);
35     }
36 
37 
38     @Override
39     public Class<?>[] getExceptions() {
40         return new Class<?>[] {
41             ExceptionConst.NULL_POINTER_EXCEPTION
42         };
43     }
44 
45 
46     /**
47      * Call corresponding visitor method(s). The order is:
48      * Call visitor methods of implemented interfaces first, then
49      * call methods according to the class hierarchy in descending order,
50      * i.e., the most specific visitXXX() call comes last.
51      *
52      * @param v Visitor object
53      */
54     @Override
55     public void accept( final Visitor v ) {
56         v.visitExceptionThrower(this);
57         v.visitStackConsumer(this);
58         v.visitMONITORENTER(this);
59     }
60 }
