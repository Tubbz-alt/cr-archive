 22 package com.sun.org.apache.bcel.internal.classfile;
 23 
 24 import java.io.DataInput;
 25 import java.io.DataOutputStream;
 26 import java.io.IOException;
 27 
 28 /**
 29  * base class for parameter annotations
 30  *
 31  * @since 6.0
 32  */
 33 public abstract class ParameterAnnotations extends Attribute {
 34 
 35     /** Table of parameter annotations */
 36     private ParameterAnnotationEntry[] parameter_annotation_table;
 37 
 38     /**
 39      * @param parameter_annotation_type the subclass type of the parameter annotation
 40      * @param name_index Index pointing to the name <em>Code</em>
 41      * @param length Content length in bytes
 42      * @param input Input stream
 43      * @param constant_pool Array of constants
 44      */
 45     ParameterAnnotations(final byte parameter_annotation_type, final int name_index, final int length,
 46             final DataInput input, final ConstantPool constant_pool) throws IOException {
 47         this(parameter_annotation_type, name_index, length, (ParameterAnnotationEntry[]) null,
 48                 constant_pool);
 49         final int num_parameters = input.readUnsignedByte();
 50         parameter_annotation_table = new ParameterAnnotationEntry[num_parameters];
 51         for (int i = 0; i < num_parameters; i++) {
 52             parameter_annotation_table[i] = new ParameterAnnotationEntry(input, constant_pool);
 53         }
 54     }
 55 
 56 
 57     /**
 58      * @param parameter_annotation_type the subclass type of the parameter annotation
 59      * @param name_index Index pointing to the name <em>Code</em>
 60      * @param length Content length in bytes
 61      * @param parameter_annotation_table the actual parameter annotations
 62      * @param constant_pool Array of constants
 63      */
 64     public ParameterAnnotations(final byte parameter_annotation_type, final int name_index, final int length,
 65             final ParameterAnnotationEntry[] parameter_annotation_table, final ConstantPool constant_pool) {
 66         super(parameter_annotation_type, name_index, length, constant_pool);
 67         this.parameter_annotation_table = parameter_annotation_table;
 68     }
 69 
 70 
 71     /**
 72      * Called by objects that are traversing the nodes of the tree implicitely
 73      * defined by the contents of a Java class. I.e., the hierarchy of methods,
 74      * fields, attributes, etc. spawns a tree of objects.
 75      *
 76      * @param v Visitor object
 77      */
 78     @Override
 79     public void accept( final Visitor v ) {
 80         v.visitParameterAnnotation(this);
 81     }
 82 
 83 
 84     /**
 85      * @param parameter_annotation_table the entries to set in this parameter annotation
 86      */
 87     public final void setParameterAnnotationTable(final ParameterAnnotationEntry[] parameter_annotation_table ) {
 88         this.parameter_annotation_table = parameter_annotation_table;
 89     }
 90 
 91 
 92     /**
 93      * @return the parameter annotation entry table
 94      */
 95     public final ParameterAnnotationEntry[] getParameterAnnotationTable() {
 96         return parameter_annotation_table;
 97     }
 98 
 99 
100     /**
101      * returns the array of parameter annotation entries in this parameter annotation
102      */
103     public ParameterAnnotationEntry[] getParameterAnnotationEntries() {
104         return parameter_annotation_table;
105     }
106 
107     @Override
108     public void dump(final DataOutputStream dos) throws IOException
109     {
110         super.dump(dos);
111         dos.writeByte(parameter_annotation_table.length);
112 
113         for (final ParameterAnnotationEntry element : parameter_annotation_table) {
114             element.dump(dos);
115         }
116 
117     }
118 
119     /**
120      * @return deep copy of this attribute
121      */
122     @Override
123     public Attribute copy( final ConstantPool constant_pool ) {
124         return (Attribute) clone();
125     }
126 }
