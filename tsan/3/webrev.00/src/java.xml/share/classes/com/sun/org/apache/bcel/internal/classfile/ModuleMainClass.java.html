 22 package com.sun.org.apache.bcel.internal.classfile;
 23 
 24 import java.io.DataInput;
 25 import java.io.DataOutputStream;
 26 import java.io.IOException;
 27 
 28 import com.sun.org.apache.bcel.internal.Const;
 29 
 30 /**
 31  * This class is derived from <em>Attribute</em> and indicates the main class of a module.
 32  * There may be at most one ModuleMainClass attribute in a ClassFile structure.
 33  *
 34  * @see     Attribute
 35  */
 36 public final class ModuleMainClass extends Attribute {
 37 
 38     private int main_class_index;
 39 
 40 
 41     /**
 42      * Initialize from another object. Note that both objects use the same
 43      * references (shallow copy). Use copy() for a physical copy.
 44      */
 45     public ModuleMainClass(final ModuleMainClass c) {
 46         this(c.getNameIndex(), c.getLength(), c.getHostClassIndex(), c.getConstantPool());
 47     }
 48 
 49 
 50     /**
 51      * @param name_index Index in constant pool
 52      * @param length Content length in bytes
 53      * @param main_class_index Host class index
 54      * @param constant_pool Array of constants
 55      */
 56     public ModuleMainClass(final int name_index, final int length, final int main_class_index,
 57             final ConstantPool constant_pool) {
 58         super(Const.ATTR_NEST_MEMBERS, name_index, length, constant_pool);
 59         this.main_class_index = main_class_index;
 60     }
 61 
 62 
 63     /**
 64      * Construct object from input stream.
 65      * @param name_index Index in constant pool
 66      * @param length Content length in bytes
 67      * @param input Input stream
 68      * @param constant_pool Array of constants
 69      * @throws IOException
 70      */
 71     ModuleMainClass(final int name_index, final int length, final DataInput input, final ConstantPool constant_pool) throws IOException {
 72         this(name_index, length, 0, constant_pool);
 73         main_class_index = input.readUnsignedShort();
 74     }
 75 
 76 
 77     /**
 78      * Called by objects that are traversing the nodes of the tree implicitly
 79      * defined by the contents of a Java class. I.e., the hierarchy of methods,
 80      * fields, attributes, etc. spawns a tree of objects.
 81      *
 82      * @param v Visitor object
 83      */
 84     @Override
 85     public void accept( final Visitor v ) {
 86         v.visitModuleMainClass(this);
 87     }
 88 
 89 
 90     /**
 91      * Dump ModuleMainClass attribute to file stream in binary format.
 92      *
 93      * @param file Output file stream
 94      * @throws IOException if an I/O error occurs.
 95      */
 96     @Override
 97     public void dump( final DataOutputStream file ) throws IOException {
 98         super.dump(file);
 99         file.writeShort(main_class_index);
100     }
101 
102 
103     /**
104      * @return index into constant pool of host class name.
105      */
106     public int getHostClassIndex() {
107         return main_class_index;
108     }
109 
110 
111     /**
112      * @param main_class_index the host class index
113      */
114     public void setHostClassIndex( final int main_class_index ) {
115         this.main_class_index = main_class_index;
116     }
117 
118 
119     /**
120      * @return String representation
121      */
122     @Override
123     public String toString() {
124         final StringBuilder buf = new StringBuilder();
125         buf.append("ModuleMainClass: ");
126         final String class_name = super.getConstantPool().getConstantString(main_class_index, Const.CONSTANT_Class);
127         buf.append(Utility.compactClassName(class_name, false));
128         return buf.toString();
129     }
130 
131 
132     /**
133      * @return deep copy of this attribute
134      */
135     @Override
136     public Attribute copy( final ConstantPool _constant_pool ) {
137         final ModuleMainClass c = (ModuleMainClass) clone();
138         c.setConstantPool(_constant_pool);
139         return c;
140     }
141 }
