 22 package com.sun.org.apache.bcel.internal.generic;
 23 
 24 import java.io.DataOutputStream;
 25 import java.io.IOException;
 26 
 27 import com.sun.org.apache.bcel.internal.ExceptionConst;
 28 import com.sun.org.apache.bcel.internal.classfile.ConstantPool;
 29 import com.sun.org.apache.bcel.internal.util.ByteSequence;
 30 
 31 /**
 32  * MULTIANEWARRAY - Create new mutidimensional array of references
 33  * <PRE>Stack: ..., count1, [count2, ...] -&gt; ..., arrayref</PRE>
 34  *

 35  */
 36 public class MULTIANEWARRAY extends CPInstruction implements LoadClass, AllocationInstruction,
 37         ExceptionThrower {
 38 
 39     private short dimensions;
 40 
 41 
 42     /**
 43      * Empty constructor needed for Instruction.readInstruction.
 44      * Not to be used otherwise.
 45      */
 46     MULTIANEWARRAY() {
 47     }
 48 
 49 
 50     public MULTIANEWARRAY(final int index, final short dimensions) {
 51         super(com.sun.org.apache.bcel.internal.Const.MULTIANEWARRAY, index);
 52         if (dimensions < 1) {
 53             throw new ClassGenException("Invalid dimensions value: " + dimensions);
 54         }
 55         this.dimensions = dimensions;
 56         super.setLength(4);
 57     }
 58 
 59 
 60     /**
 61      * Dump instruction as byte code to stream out.
 62      * @param out Output stream
 63      */
 64     @Override
 65     public void dump( final DataOutputStream out ) throws IOException {
 66         out.writeByte(super.getOpcode());
 67         out.writeShort(super.getIndex());
 68         out.writeByte(dimensions);
 69     }
 70 
 71 
 72     /**
 73      * Read needed data (i.e., no. dimension) from file.
 74      */
 75     @Override
 76     protected void initFromFile( final ByteSequence bytes, final boolean wide ) throws IOException {
 77         super.initFromFile(bytes, wide);
 78         dimensions = bytes.readByte();
 79         super.setLength(4);
 80     }
 81 
 82 
 83     /**
 84      * @return number of dimensions to be created
 85      */
 86     public final short getDimensions() {
 87         return dimensions;
 88     }
 89 
 90 
 91     /**
 92      * @return mnemonic for instruction
 93      */
 94     @Override
 95     public String toString( final boolean verbose ) {
 96         return super.toString(verbose) + " " + super.getIndex() + " " + dimensions;
 97     }
 98 
 99 
100     /**
101      * @return mnemonic for instruction with symbolic references resolved
102      */
103     @Override
104     public String toString( final ConstantPool cp ) {
105         return super.toString(cp) + " " + dimensions;
106     }
107 
108 
109     /**
110      * Also works for instructions whose stack effect depends on the
111      * constant pool entry they reference.
112      * @return Number of words consumed from stack by this instruction
113      */
114     @Override
115     public int consumeStack( final ConstantPoolGen cpg ) {
116         return dimensions;
117     }
118 
119 
120     @Override
121     public Class<?>[] getExceptions() {
122         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION,
123             ExceptionConst.ILLEGAL_ACCESS_ERROR,
124             ExceptionConst.NEGATIVE_ARRAY_SIZE_EXCEPTION);
125     }
126 
127 
128     @Override
129     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
130         Type t = getType(cpg);
131         if (t instanceof ArrayType) {
132             t = ((ArrayType) t).getBasicType();
133         }
134         return (t instanceof ObjectType) ? (ObjectType) t : null;
135     }
136 
137 
138     /**
139      * Call corresponding visitor method(s). The order is:
140      * Call visitor methods of implemented interfaces first, then
141      * call methods according to the class hierarchy in descending order,
142      * i.e., the most specific visitXXX() call comes last.
143      *
144      * @param v Visitor object
145      */
146     @Override
147     public void accept( final Visitor v ) {
148         v.visitLoadClass(this);
149         v.visitAllocationInstruction(this);
150         v.visitExceptionThrower(this);
151         v.visitTypedInstruction(this);
152         v.visitCPInstruction(this);
153         v.visitMULTIANEWARRAY(this);
154     }
155 }
