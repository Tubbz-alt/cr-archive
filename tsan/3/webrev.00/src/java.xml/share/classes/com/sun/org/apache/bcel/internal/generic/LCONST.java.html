21 package com.sun.org.apache.bcel.internal.generic;
22 
23 /**
24  * LCONST - Push 0 or 1, other values cause an exception
25  *
26  * <PRE>Stack: ... -&gt; ..., </PRE>
27  *
28  */
29 public class LCONST extends Instruction implements ConstantPushInstruction {
30 
31     private long value;
32 
33 
34     /**
35      * Empty constructor needed for Instruction.readInstruction.
36      * Not to be used otherwise.
37      */
38     LCONST() {
39     }
40 
41 
42     public LCONST(final long l) {
43         super(com.sun.org.apache.bcel.internal.Const.LCONST_0, (short) 1);
44         if (l == 0) {
45             super.setOpcode(com.sun.org.apache.bcel.internal.Const.LCONST_0);
46         } else if (l == 1) {
47             super.setOpcode(com.sun.org.apache.bcel.internal.Const.LCONST_1);
48         } else {
49             throw new ClassGenException("LCONST can be used only for 0 and 1: " + l);
50         }
51         value = l;
52     }
53 
54 
55     @Override
56     public Number getValue() {
57         return Long.valueOf(value);
58     }
59 
60 
61     /** @return Type.LONG
62      */
63     @Override
64     public Type getType( final ConstantPoolGen cp ) {
65         return Type.LONG;
66     }
67 
68 
69     /**
70      * Call corresponding visitor method(s). The order is:
71      * Call visitor methods of implemented interfaces first, then
72      * call methods according to the class hierarchy in descending order,
73      * i.e., the most specific visitXXX() call comes last.
74      *
75      * @param v Visitor object
76      */
77     @Override
78     public void accept( final Visitor v ) {
79         v.visitPushInstruction(this);
80         v.visitStackProducer(this);
81         v.visitTypedInstruction(this);
82         v.visitConstantPushInstruction(this);
83         v.visitLCONST(this);
84     }
85 }
