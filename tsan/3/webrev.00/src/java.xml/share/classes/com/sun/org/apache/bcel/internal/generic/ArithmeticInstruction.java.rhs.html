22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.Const;
25 
26 /**
27  * Super class for the family of arithmetic instructions.
28  *

29  */
30 public abstract class ArithmeticInstruction extends Instruction implements TypedInstruction,
31         StackProducer, StackConsumer {
32 
33     /**
34      * Empty constructor needed for Instruction.readInstruction.
35      * Not to be used otherwise.
36      */
37     ArithmeticInstruction() {
38     }
39 
40 
41     /**
42      * @param opcode of instruction
43      */
44     protected ArithmeticInstruction(final short opcode) {
45         super(opcode, (short) 1);
46     }
47 
48 
49     /** @return type associated with the instruction
50      */
51     @Override
52     public Type getType( final ConstantPoolGen cp ) {
53         final short _opcode = super.getOpcode();
54         switch (_opcode) {
55             case Const.DADD:
56             case Const.DDIV:
57             case Const.DMUL:
58             case Const.DNEG:
59             case Const.DREM:
60             case Const.DSUB:
61                 return Type.DOUBLE;
62             case Const.FADD:
63             case Const.FDIV:
64             case Const.FMUL:
65             case Const.FNEG:
66             case Const.FREM:
67             case Const.FSUB:
68                 return Type.FLOAT;
69             case Const.IADD:
70             case Const.IAND:
71             case Const.IDIV:
72             case Const.IMUL:
73             case Const.INEG:
74             case Const.IOR:
75             case Const.IREM:
76             case Const.ISHL:
77             case Const.ISHR:
78             case Const.ISUB:
79             case Const.IUSHR:
80             case Const.IXOR:
81                 return Type.INT;
82             case Const.LADD:
83             case Const.LAND:
84             case Const.LDIV:
85             case Const.LMUL:
86             case Const.LNEG:
87             case Const.LOR:
88             case Const.LREM:
89             case Const.LSHL:
90             case Const.LSHR:
91             case Const.LSUB:
92             case Const.LUSHR:
93             case Const.LXOR:
94                 return Type.LONG;
95             default: // Never reached
96                 throw new ClassGenException("Unknown type " + _opcode);
97         }
98     }
99 }
