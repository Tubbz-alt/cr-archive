22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.Const;
25 import com.sun.org.apache.bcel.internal.ExceptionConst;
26 
27 /**
28  * PUTFIELD - Put field in object
29  * <PRE>Stack: ..., objectref, value -&gt; ...</PRE>
30  * OR
31  * <PRE>Stack: ..., objectref, value.word1, value.word2 -&gt; ...</PRE>
32  *
33  * @version $Id: PUTFIELD.java 1747278 2016-06-07 17:28:43Z britter $
34  */
35 public class PUTFIELD extends FieldInstruction implements PopInstruction, ExceptionThrower {
36 
37     /**
38      * Empty constructor needed for the Class.newInstance() statement in
39      * Instruction.readInstruction(). Not to be used otherwise.
40      */
41     PUTFIELD() {
42     }
43 
44 
45     public PUTFIELD(final int index) {
46         super(Const.PUTFIELD, index);
47     }
48 
49 
50     @Override
51     public int consumeStack( final ConstantPoolGen cpg ) {
52         return getFieldSize(cpg) + 1;
53     }
54 
55 
56     @Override
57     public Class<?>[] getExceptions() {
58         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_FIELD_AND_METHOD_RESOLUTION,
59             ExceptionConst.NULL_POINTER_EXCEPTION,
60             ExceptionConst.INCOMPATIBLE_CLASS_CHANGE_ERROR);
61     }
62 
63 
64     /**
65      * Call corresponding visitor method(s). The order is:
66      * Call visitor methods of implemented interfaces first, then
67      * call methods according to the class hierarchy in descending order,
68      * i.e., the most specific visitXXX() call comes last.
69      *
70      * @param v Visitor object
71      */
72     @Override
73     public void accept( final Visitor v ) {
74         v.visitExceptionThrower(this);
75         v.visitStackConsumer(this);
76         v.visitPopInstruction(this);
77         v.visitTypedInstruction(this);
78         v.visitLoadClass(this);
79         v.visitCPInstruction(this);
80         v.visitFieldOrMethod(this);
81         v.visitFieldInstruction(this);
82         v.visitPUTFIELD(this);
83     }
84 }
