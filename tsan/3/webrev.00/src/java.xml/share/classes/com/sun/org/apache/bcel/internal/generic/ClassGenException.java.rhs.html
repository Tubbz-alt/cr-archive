24 /**
25  * Thrown on internal errors. Extends RuntimeException so it hasn't to be declared
26  * in the throws clause every time.
27  *

28  */
29 public class ClassGenException extends RuntimeException {
30 
31     private static final long serialVersionUID = 7247369755051242791L;
32 
33     public ClassGenException() {
34         super();
35     }
36 
37 
38     public ClassGenException(final String s) {
39         super(s);
40     }
41 
42     public ClassGenException(final String s, final Throwable initCause) {
43         super(s, initCause);
44     }
45 }
