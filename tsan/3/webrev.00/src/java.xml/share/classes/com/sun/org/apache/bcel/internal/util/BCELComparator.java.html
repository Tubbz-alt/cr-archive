24 /**
25  * Used for BCEL comparison strategy
26  *
27  * @since 5.2
28  */
29 public interface BCELComparator {
30 
31     /**
32      * Compare two objects and return what THIS.equals(THAT) should return
33      *
34      * @param THIS
35      * @param THAT
36      * @return true if and only if THIS equals THAT
37      */
38     boolean equals( Object THIS, Object THAT );
39 
40 
41     /**
42      * Return hashcode for THIS.hashCode()
43      *
44      * @param THIS
45      * @return hashcode for THIS.hashCode()
46      */
47     int hashCode( Object THIS );
48 }
