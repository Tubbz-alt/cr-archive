22 package com.sun.org.apache.bcel.internal.util;
23 
24 import java.util.Stack;
25 
26 import com.sun.org.apache.bcel.internal.classfile.JavaClass;
27 
28 /**
29  * Utility class implementing a (typesafe) stack of JavaClass objects.
30  *
31  * @see Stack
32  */
33 public class ClassStack {
34 
35     private final Stack<JavaClass> stack = new Stack<>();
36 
37 
38     public void push( final JavaClass clazz ) {
39         stack.push(clazz);
40     }
41 
42 
43     public JavaClass pop() {
44         return stack.pop();
45     }
46 
47 
48     public JavaClass top() {
49         return stack.peek();
50     }
51 
52 
53     public boolean empty() {
54         return stack.empty();
55     }
56 }
