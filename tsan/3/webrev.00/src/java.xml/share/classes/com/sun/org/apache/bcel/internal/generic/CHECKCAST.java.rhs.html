22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.ExceptionConst;
25 
26 /**
27  * CHECKCAST - Check whether object is of given type
28  * <PRE>Stack: ..., objectref -&gt; ..., objectref</PRE>
29  *

30  */
31 public class CHECKCAST extends CPInstruction implements LoadClass, ExceptionThrower, StackProducer,
32         StackConsumer {
33 
34     /**
35      * Empty constructor needed for Instruction.readInstruction.
36      * Not to be used otherwise.
37      */
38     CHECKCAST() {
39     }
40 
41 
42     /** Check whether object is of given type
43      * @param index index to class in constant pool
44      */
45     public CHECKCAST(final int index) {
46         super(com.sun.org.apache.bcel.internal.Const.CHECKCAST, index);
47     }
48 
49 
50     /** @return exceptions this instruction may cause
51      */
52     @Override
53     public Class<?>[] getExceptions() {
54         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION,
55             ExceptionConst.CLASS_CAST_EXCEPTION);
56     }
57 
58 
59     @Override
60     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
61         Type t = getType(cpg);
62         if (t instanceof ArrayType) {
63             t = ((ArrayType) t).getBasicType();
64         }
65         return (t instanceof ObjectType) ? (ObjectType) t : null;
66     }
67 
68 
69     /**
70      * Call corresponding visitor method(s). The order is:
71      * Call visitor methods of implemented interfaces first, then
72      * call methods according to the class hierarchy in descending order,
73      * i.e., the most specific visitXXX() call comes last.
74      *
75      * @param v Visitor object
76      */
77     @Override
78     public void accept( final Visitor v ) {
79         v.visitLoadClass(this);
80         v.visitExceptionThrower(this);
81         v.visitStackProducer(this);
82         v.visitStackConsumer(this);
83         v.visitTypedInstruction(this);
84         v.visitCPInstruction(this);
85         v.visitCHECKCAST(this);
86     }
87 }
