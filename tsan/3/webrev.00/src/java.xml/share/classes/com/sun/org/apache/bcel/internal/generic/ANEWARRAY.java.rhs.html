22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.ExceptionConst;
25 
26 /**
27  * ANEWARRAY -  Create new array of references
28  * <PRE>Stack: ..., count -&gt; ..., arrayref</PRE>
29  *

30  */
31 public class ANEWARRAY extends CPInstruction implements LoadClass, AllocationInstruction,
32         ExceptionThrower, StackConsumer, StackProducer {
33 
34     /**
35      * Empty constructor needed for Instruction.readInstruction.
36      * Not to be used otherwise.
37      */
38     ANEWARRAY() {
39     }
40 
41 
42     public ANEWARRAY(final int index) {
43         super(com.sun.org.apache.bcel.internal.Const.ANEWARRAY, index);
44     }
45 
46 
47     @Override
48     public Class<?>[] getExceptions() {
49         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION,
50             ExceptionConst.NEGATIVE_ARRAY_SIZE_EXCEPTION);
51     }
52 
53 
54     /**
55      * Call corresponding visitor method(s). The order is:
56      * Call visitor methods of implemented interfaces first, then
57      * call methods according to the class hierarchy in descending order,
58      * i.e., the most specific visitXXX() call comes last.
59      *
60      * @param v Visitor object
61      */
62     @Override
63     public void accept( final Visitor v ) {
64         v.visitLoadClass(this);
65         v.visitAllocationInstruction(this);
66         v.visitExceptionThrower(this);
67         v.visitStackProducer(this);
68         v.visitTypedInstruction(this);
69         v.visitCPInstruction(this);
70         v.visitANEWARRAY(this);
71     }
72 
73 
74     @Override
75     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
76         Type t = getType(cpg);
77         if (t instanceof ArrayType) {
78             t = ((ArrayType) t).getBasicType();
79         }
80         return (t instanceof ObjectType) ? (ObjectType) t : null;
81     }
82 }
