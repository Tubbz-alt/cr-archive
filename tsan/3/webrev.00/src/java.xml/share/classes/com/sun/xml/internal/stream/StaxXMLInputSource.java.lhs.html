26 package com.sun.xml.internal.stream;
27 
28 import javax.xml.stream.XMLEventReader;
29 import javax.xml.stream.XMLStreamReader;
30 import com.sun.org.apache.xerces.internal.xni.parser.XMLInputSource;
31 
32 /**
33  *
34  * @author  Neeraj
35  *
36  * This class wraps XMLInputSource and is also capable of telling wether application
37  * returned XMLStreamReader or not when XMLResolver.resolveEntity
38  * was called.
39  */
40 public class StaxXMLInputSource {
41 
42     XMLStreamReader fStreamReader ;
43     XMLEventReader fEventReader ;
44     XMLInputSource fInputSource ;
45 
46     //indicates if the source is created by a resolver
47     boolean fIsCreatedByResolver = false;
48 
49     /** Creates a new instance of StaxXMLInputSource */
50     public StaxXMLInputSource(XMLStreamReader streamReader, boolean byResolver) {
51         fStreamReader = streamReader ;
52     }
53 
54     /** Creates a new instance of StaxXMLInputSource */
55     public StaxXMLInputSource(XMLEventReader eventReader, boolean byResolver) {
56         fEventReader = eventReader ;
57     }
58 
59     public StaxXMLInputSource(XMLInputSource inputSource, boolean byResolver){
60         fInputSource = inputSource ;
61         fIsCreatedByResolver = byResolver;
62     }
63 
64     public XMLStreamReader getXMLStreamReader(){
65         return fStreamReader ;
66     }
67 
68     public XMLEventReader getXMLEventReader(){
69         return fEventReader ;
70     }
71 
72     public XMLInputSource getXMLInputSource(){
73         return fInputSource ;
74     }
75 
76     public boolean hasXMLStreamOrXMLEventReader(){
77         return (fStreamReader == null) && (fEventReader == null) ? false : true ;
78     }
79 
80     public boolean isCreatedByResolver() {
81         return fIsCreatedByResolver;
82     }
83 }
