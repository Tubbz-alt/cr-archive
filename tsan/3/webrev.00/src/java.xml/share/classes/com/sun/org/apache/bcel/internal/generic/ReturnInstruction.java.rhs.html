21 package com.sun.org.apache.bcel.internal.generic;
22 
23 import com.sun.org.apache.bcel.internal.Const;
24 import com.sun.org.apache.bcel.internal.ExceptionConst;
25 
26 /**
27  * Super class for the xRETURN family of instructions.
28  *
29  * @LastModified: Jan 2020

30  */
31 public abstract class ReturnInstruction extends Instruction implements ExceptionThrower,
32         TypedInstruction, StackConsumer {
33 
34     /**
35      * Empty constructor needed for Instruction.readInstruction.
36      * Not to be used otherwise.
37      */
38     ReturnInstruction() {
39     }
40 
41 
42     /**
43      * @param opcode of instruction
44      */
45     protected ReturnInstruction(final short opcode) {
46         super(opcode, (short) 1);
47     }
48 
49 
50     public Type getType() {
51         final short _opcode = super.getOpcode();
52         switch (_opcode) {
53             case Const.IRETURN:
54                 return Type.INT;
55             case Const.LRETURN:
56                 return Type.LONG;
57             case Const.FRETURN:
58                 return Type.FLOAT;
59             case Const.DRETURN:
60                 return Type.DOUBLE;
61             case Const.ARETURN:
62                 return Type.OBJECT;
63             case Const.RETURN:
64                 return Type.VOID;
65             default: // Never reached
66                 throw new ClassGenException("Unknown type " + _opcode);
67         }
68     }
69 
70 
71     @Override
72     public Class<?>[] getExceptions() {
73         return new Class<?>[] {
74             ExceptionConst.ILLEGAL_MONITOR_STATE
75         };
76     }
77 
78 
79     /** @return type associated with the instruction
80      */
81     @Override
82     public Type getType( final ConstantPoolGen cp ) {
83         return getType();
84     }
85 }
