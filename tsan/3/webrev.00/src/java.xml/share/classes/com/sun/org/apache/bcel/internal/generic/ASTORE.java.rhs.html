24 /**
25  * ASTORE - Store reference into local variable
26  * <PRE>Stack ..., objectref -&gt; ... </PRE>
27  *
28  * @LastModified: Jan 2020
29  */
30 public class ASTORE extends StoreInstruction {
31 
32     /**
33      * Empty constructor needed for Instruction.readInstruction.
34      * Not to be used otherwise.
35      */
36     ASTORE() {
37         super(Const.ASTORE, Const.ASTORE_0);
38     }
39 
40 
41     /** Store reference into local variable

42      * @param n index of local variable
43      */
44     public ASTORE(final int n) {
45         super(Const.ASTORE, Const.ASTORE_0, n);
46     }
47 
48 
49     /**
50      * Call corresponding visitor method(s). The order is:
51      * Call visitor methods of implemented interfaces first, then
52      * call methods according to the class hierarchy in descending order,
53      * i.e., the most specific visitXXX() call comes last.
54      *
55      * @param v Visitor object
56      */
57     @Override
58     public void accept( final Visitor v ) {
59         super.accept(v);
60         v.visitASTORE(this);
61     }
62 }
