22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.ExceptionConst;
25 
26 /**
27  * INSTANCEOF - Determine if object is of given type
28  * <PRE>Stack: ..., objectref -&gt; ..., result</PRE>
29  *

30  */
31 public class INSTANCEOF extends CPInstruction implements LoadClass, ExceptionThrower,
32         StackProducer, StackConsumer {
33 
34     /**
35      * Empty constructor needed for Instruction.readInstruction.
36      * Not to be used otherwise.
37      */
38     INSTANCEOF() {
39     }
40 
41 
42     public INSTANCEOF(final int index) {
43         super(com.sun.org.apache.bcel.internal.Const.INSTANCEOF, index);
44     }
45 
46 
47     @Override
48     public Class<?>[] getExceptions() {
49         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION);
50     }
51 
52 
53     @Override
54     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
55         Type t = getType(cpg);
56         if (t instanceof ArrayType) {
57             t = ((ArrayType) t).getBasicType();
58         }
59         return (t instanceof ObjectType) ? (ObjectType) t : null;
60     }
61 
62 
63     /**
64      * Call corresponding visitor method(s). The order is:
65      * Call visitor methods of implemented interfaces first, then
66      * call methods according to the class hierarchy in descending order,
67      * i.e., the most specific visitXXX() call comes last.
68      *
69      * @param v Visitor object
70      */
71     @Override
72     public void accept( final Visitor v ) {
73         v.visitLoadClass(this);
74         v.visitExceptionThrower(this);
75         v.visitStackProducer(this);
76         v.visitStackConsumer(this);
77         v.visitTypedInstruction(this);
78         v.visitCPInstruction(this);
79         v.visitINSTANCEOF(this);
80     }
81 }
