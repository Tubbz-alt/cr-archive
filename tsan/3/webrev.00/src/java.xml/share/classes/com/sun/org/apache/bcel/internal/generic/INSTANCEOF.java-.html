22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.ExceptionConst;
25 
26 /**
27  * INSTANCEOF - Determine if object is of given type
28  * <PRE>Stack: ..., objectref -&gt; ..., result</PRE>
29  *
30  * @version $Id: INSTANCEOF.java 1747278 2016-06-07 17:28:43Z britter $
31  */
32 public class INSTANCEOF extends CPInstruction implements LoadClass, ExceptionThrower,
33         StackProducer, StackConsumer {
34 
35     /**
36      * Empty constructor needed for the Class.newInstance() statement in
37      * Instruction.readInstruction(). Not to be used otherwise.
38      */
39     INSTANCEOF() {
40     }
41 
42 
43     public INSTANCEOF(final int index) {
44         super(com.sun.org.apache.bcel.internal.Const.INSTANCEOF, index);
45     }
46 
47 
48     @Override
49     public Class<?>[] getExceptions() {
50         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION);
51     }
52 
53 
54     @Override
55     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
56         Type t = getType(cpg);
57         if (t instanceof ArrayType) {
58             t = ((ArrayType) t).getBasicType();
59         }
60         return (t instanceof ObjectType) ? (ObjectType) t : null;
61     }
62 
63 
64     /**
65      * Call corresponding visitor method(s). The order is:
66      * Call visitor methods of implemented interfaces first, then
67      * call methods according to the class hierarchy in descending order,
68      * i.e., the most specific visitXXX() call comes last.
69      *
70      * @param v Visitor object
71      */
72     @Override
73     public void accept( final Visitor v ) {
74         v.visitLoadClass(this);
75         v.visitExceptionThrower(this);
76         v.visitStackProducer(this);
77         v.visitStackConsumer(this);
78         v.visitTypedInstruction(this);
79         v.visitCPInstruction(this);
80         v.visitINSTANCEOF(this);
81     }
82 }
