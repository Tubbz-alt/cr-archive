22 package com.sun.org.apache.bcel.internal.generic;
23 
24 /**
25  * Denotes an unparameterized instruction to load a value from a local
26  * variable, e.g. ILOAD.
27  *

28  */
29 public abstract class LoadInstruction extends LocalVariableInstruction implements PushInstruction {
30 
31     /**
32      * Empty constructor needed for Instruction.readInstruction.
33      * Not to be used otherwise.
34      * tag and length are defined in readInstruction and initFromFile, respectively.
35      */
36     LoadInstruction(final short canon_tag, final short c_tag) {
37         super(canon_tag, c_tag);
38     }
39 
40 
41     /**
42      * @param opcode Instruction opcode
43      * @param c_tag Instruction number for compact version, ALOAD_0, e.g.
44      * @param n local variable index (unsigned short)
45      */
46     protected LoadInstruction(final short opcode, final short c_tag, final int n) {
47         super(opcode, c_tag, n);
48     }
49 
50 
51     /**
52      * Call corresponding visitor method(s). The order is:
53      * Call visitor methods of implemented interfaces first, then
54      * call methods according to the class hierarchy in descending order,
55      * i.e., the most specific visitXXX() call comes last.
56      *
57      * @param v Visitor object
58      */
59     @Override
60     public void accept( final Visitor v ) {
61         v.visitStackProducer(this);
62         v.visitPushInstruction(this);
63         v.visitTypedInstruction(this);
64         v.visitLocalVariableInstruction(this);
65         v.visitLoadInstruction(this);
66     }
67 }
