 22 package com.sun.org.apache.bcel.internal.classfile;
 23 
 24 import java.io.DataInput;
 25 import java.io.DataOutputStream;
 26 import java.io.IOException;
 27 
 28 import com.sun.org.apache.bcel.internal.Const;
 29 
 30 /**
 31  * This class represents the type of a local variable or item on stack
 32  * used in the StackMap entries.
 33  *
 34  * @see     StackMapEntry
 35  * @see     StackMap
 36  * @see     Const
 37  */
 38 public final class StackMapType implements Cloneable {
 39 
 40     private byte type;
 41     private int index = -1; // Index to CONSTANT_Class or offset
 42     private ConstantPool constant_pool;
 43 
 44 
 45     /**
 46      * Construct object from file stream.
 47      * @param file Input stream
 48      * @throws IOException
 49      */
 50     StackMapType(final DataInput file, final ConstantPool constant_pool) throws IOException {
 51         this(file.readByte(), -1, constant_pool);
 52         if (hasIndex()) {
 53             this.index = file.readShort();
 54         }
 55         this.constant_pool = constant_pool;
 56     }
 57 
 58 
 59     /**
 60      * @param type type tag as defined in the Constants interface
 61      * @param index index to constant pool, or byte code offset
 62      */
 63     public StackMapType(final byte type, final int index, final ConstantPool constant_pool) {
 64         if ((type < Const.ITEM_Bogus) || (type > Const.ITEM_NewObject)) {
 65             throw new RuntimeException("Illegal type for StackMapType: " + type);
 66         }
 67         this.type = type;
 68         this.index = index;
 69         this.constant_pool = constant_pool;
 70     }
 71 
 72 
 73     public void setType( final byte t ) {
 74         if ((t < Const.ITEM_Bogus) || (t > Const.ITEM_NewObject)) {
 75             throw new RuntimeException("Illegal type for StackMapType: " + t);
 76         }
 77         type = t;
 78     }
 79 
 80 
 81     public byte getType() {
 82         return type;
 83     }
 84 
 85 
 86     public void setIndex( final int t ) {
 87         index = t;
 88     }
 89 
 90 
 91     /** @return index to constant pool if type == ITEM_Object, or offset
 92      * in byte code, if type == ITEM_NewObject, and -1 otherwise
 93      */
 94     public int getIndex() {
 95         return index;
 96     }
 97 
 98 
 99     /**
100      * Dump type entries to file.
101      *
102      * @param file Output file stream
103      * @throws IOException
104      */
105     public void dump( final DataOutputStream file ) throws IOException {
106         file.writeByte(type);
107         if (hasIndex()) {
108             file.writeShort(getIndex());
109         }
110     }
111 
112 
113     /** @return true, if type is either ITEM_Object or ITEM_NewObject
114      */
115     public boolean hasIndex() {
116         return type == Const.ITEM_Object || type == Const.ITEM_NewObject;
117     }
118 
119 
120     private String printIndex() {
121         if (type == Const.ITEM_Object) {
122             if (index < 0) {
123                 return ", class=<unknown>";
124             }
125             return ", class=" + constant_pool.constantToString(index, Const.CONSTANT_Class);
126         } else if (type == Const.ITEM_NewObject) {
127             return ", offset=" + index;
128         } else {
129             return "";
130         }
131     }
132 
133 
134     /**
135      * @return String representation
136      */
137     @Override
138     public String toString() {
139         return "(type=" + Const.getItemName(type) + printIndex() + ")";
140     }
141 
142 
143     /**
144      * @return deep copy of this object
145      */
146     public StackMapType copy() {
147         try {
148             return (StackMapType) clone();
149         } catch (final CloneNotSupportedException e) {
150             // TODO should this throw?
151         }
152         return null;
153     }
154 
155 
156     /**
157      * @return Constant pool used by this object.
158      */
159     public ConstantPool getConstantPool() {
160         return constant_pool;
161     }
162 
163 
164     /**
165      * @param constant_pool Constant pool to be used for this object.
166      */
167     public void setConstantPool( final ConstantPool constant_pool ) {
168         this.constant_pool = constant_pool;
169     }
170 }
