24 import java.io.IOException;

26 import com.sun.org.apache.bcel.internal.util.ByteSequence;

28 /**
29  * LDC_W - Push item from constant pool (wide index)
30  *
31  * <PRE>Stack: ... -&gt; ..., item.word1, item.word2</PRE>
32  *
33  * @version $Id: LDC_W.java 1747278 2016-06-07 17:28:43Z britter $
34  */
35 public class LDC_W extends LDC {
36 
37     /**
38      * Empty constructor needed for the Class.newInstance() statement in
39      * Instruction.readInstruction(). Not to be used otherwise.
40      */
41     LDC_W() {
42     }
43 
44 
45     public LDC_W(final int index) {
46         super(index);
47     }
48 
49 
50     /**
51      * Read needed data (i.e., index) from file.
52      */
53     @Override
54     protected void initFromFile( final ByteSequence bytes, final boolean wide ) throws IOException {
55         setIndex(bytes.readUnsignedShort());
56         // Override just in case it has been changed
57         super.setOpcode(com.sun.org.apache.bcel.internal.Const.LDC_W);
58         super.setLength(3);
59     }
60 }
