22 package com.sun.org.apache.bcel.internal.generic;
23 
24 /**
25  * Super class for JSR - Jump to subroutine
26  *

27  */
28 public abstract class JsrInstruction extends BranchInstruction implements UnconditionalBranch,
29         TypedInstruction, StackProducer {
30 
31     JsrInstruction(final short opcode, final InstructionHandle target) {
32         super(opcode, target);
33     }
34 
35 
36     /**
37      * Empty constructor needed for Instruction.readInstruction.
38      * Not to be used otherwise.
39      */
40     JsrInstruction() {
41     }
42 
43 
44     /** @return return address type
45      */
46     @Override
47     public Type getType( final ConstantPoolGen cp ) {
48         return new ReturnaddressType(physicalSuccessor());
49     }
50 
51 
52     /**
53      * Returns an InstructionHandle to the physical successor
54      * of this JsrInstruction. <B>For this method to work,
55      * this JsrInstruction object must not be shared between
56      * multiple InstructionHandle objects!</B>
57      * Formally, there must not be InstructionHandle objects
58      * i, j where i != j and i.getInstruction() == this ==
59      * j.getInstruction().
60      * @return an InstructionHandle to the "next" instruction that
61      * will be executed when RETurned from a subroutine.
62      */
63     public InstructionHandle physicalSuccessor() {
64         InstructionHandle ih = super.getTarget();
65         // Rewind!
66         while (ih.getPrev() != null) {
67             ih = ih.getPrev();
68         }
69         // Find the handle for "this" JsrInstruction object.
70         while (ih.getInstruction() != this) {
71             ih = ih.getNext();
72         }
73         final InstructionHandle toThis = ih;
74         while (ih != null) {
75             ih = ih.getNext();
76             if ((ih != null) && (ih.getInstruction() == this)) {
77                 throw new RuntimeException("physicalSuccessor() called on a shared JsrInstruction.");
78             }
79         }
80         // Return the physical successor
81         return toThis.getNext();
82     }
83 }
