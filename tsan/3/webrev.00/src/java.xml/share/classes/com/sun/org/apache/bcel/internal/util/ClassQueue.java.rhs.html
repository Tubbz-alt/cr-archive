20 package com.sun.org.apache.bcel.internal.util;
21 
22 import java.util.LinkedList;
23 
24 import com.sun.org.apache.bcel.internal.classfile.JavaClass;
25 
26 /**
27  * Utility class implementing a (typesafe) queue of JavaClass
28  * objects.
29  *
30  * @LastModified: Jan 2020
31  */
32 public class ClassQueue {
33 
34     private final LinkedList<JavaClass> vec = new LinkedList<>();
35 
36 
37     public void enqueue( final JavaClass clazz ) {
38         vec.addLast(clazz);
39     }
40 
41 
42     public JavaClass dequeue() {
43         return vec.removeFirst();
44     }
45 
46 
47     public boolean empty() {
48         return vec.isEmpty();
49     }
50 
51 
52     @Override
53     public String toString() {
54         return vec.toString();
55     }
56 }
