22 package com.sun.org.apache.bcel.internal.generic;
23 
24 /**
25  * Denotes that an instruction may start the process of loading and resolving
26  * the referenced class in the Virtual Machine.
27  *

28  */
29 public interface LoadClass {
30 
31     /**
32      * Returns the ObjectType of the referenced class or interface
33      * that may be loaded and resolved.
34      * @return object type that may be loaded or null if a primitive is
35      * referenced
36      */
37     ObjectType getLoadClassType( ConstantPoolGen cpg );
38 
39 
40     /**
41      * Returns the type associated with this instruction.
42      * LoadClass instances are always typed, but this type
43      * does not always refer to the type of the class or interface
44      * that it possibly forces to load. For example, GETFIELD would
45      * return the type of the field and not the type of the class
46      * where the field is defined.
47      * If no class is forced to be loaded, <B>null</B> is returned.
48      * An example for this is an ANEWARRAY instruction that creates
49      * an int[][].
50      * @see #getLoadClassType(ConstantPoolGen)
51      */
52     Type getType( ConstantPoolGen cpg );
53 }
