 31 /**
 32  * This class is derived from the abstract {@link Constant}
 33  * and represents a reference to a Utf8 encoded string.
 34  *
 35  * @version $Id: ConstantUtf8.java 1750029 2016-06-23 22:14:38Z sebb $
 36  * @see     Constant
 37  */
 38 public final class ConstantUtf8 extends Constant {
 39 
 40     private final String bytes;
 41 
 42     // TODO these should perhaps be AtomicInt?
 43     private static volatile int considered = 0;
 44     private static volatile int hits = 0;
 45     private static volatile int skipped = 0;
 46     private static volatile int created = 0;
 47 
 48     // Set the size to 0 or below to skip caching entirely
 49     private static final int MAX_CACHED_SIZE = 200;
 50     private static final boolean BCEL_STATISTICS = false;
 51 
 52 
 53     private static class CACHE_HOLDER {
 54 
 55         private static final int MAX_CACHE_ENTRIES = 20000;
 56         private static final int INITIAL_CACHE_CAPACITY = (int)(MAX_CACHE_ENTRIES/0.75);
 57 
 58         private static final HashMap<String, ConstantUtf8> CACHE =
 59                 new LinkedHashMap<String, ConstantUtf8>(INITIAL_CACHE_CAPACITY, 0.75f, true) {
 60             private static final long serialVersionUID = -8506975356158971766L;
 61 
 62             @Override
 63             protected boolean removeEldestEntry(final Map.Entry<String, ConstantUtf8> eldest) {
 64                  return size() > MAX_CACHE_ENTRIES;
 65             }
 66         };
 67 
 68     }

 69 
 70     // for accesss by test code
 71     static void printStats() {
 72         System.err.println("Cache hit " + hits + "/" + considered +", " + skipped + " skipped");
 73         System.err.println("Total of " + created + " ConstantUtf8 objects created");
 74     }
 75 
 76     // for accesss by test code
 77     static void clearStats() {
 78         hits = considered = skipped = created = 0;
 79     }
 80 
 81     static {
 82         if (BCEL_STATISTICS) {
 83             Runtime.getRuntime().addShutdownHook(new Thread() {
 84                 @Override
 85                 public void run() {
 86                     printStats();
 87                 }
 88             });
 89         }
 90     }
 91 
 92     /**







 93      * @since 6.0
 94      */
 95     public static ConstantUtf8 getCachedInstance(final String s) {
 96         if (s.length() > MAX_CACHED_SIZE) {
 97             skipped++;
 98             return  new ConstantUtf8(s);
 99         }
100         considered++;
101         synchronized (ConstantUtf8.class) { // might be better with a specific lock object
102             ConstantUtf8 result = CACHE_HOLDER.CACHE.get(s);
103             if (result != null) {
104                     hits++;
105                     return result;
106                 }
107             result = new ConstantUtf8(s);
108             CACHE_HOLDER.CACHE.put(s, result);
109             return result;
110         }
111     }
112 
113     /**








114      * @since 6.0
115      */
116     public static ConstantUtf8 getInstance(final String s) {
117         return new ConstantUtf8(s);
118     }
119 
120     /**







121      * @since 6.0
122      */
123     public static ConstantUtf8 getInstance (final DataInput input)  throws IOException {
124         return getInstance(input.readUTF());
125     }

128      * Initialize from another object.


129      */
130     public ConstantUtf8(final ConstantUtf8 c) {
131         this(c.getBytes());
132     }

135     /**
136      * Initialize instance from file data.
137      *
138      * @param file Input stream
139      * @throws IOException
140      */
141     ConstantUtf8(final DataInput file) throws IOException {
142         super(Const.CONSTANT_Utf8);
143         bytes = file.readUTF();
144         created++;
145     }
146

148     /**
149      * @param bytes Data
150      */
151     public ConstantUtf8(final String bytes) {
152         super(Const.CONSTANT_Utf8);
153         if (bytes == null) {
154             throw new IllegalArgumentException("bytes must not be null!");
155         }
156         this.bytes = bytes;
157         created++;
158     }

161     /**
162      * Called by objects that are traversing the nodes of the tree implicitely
163      * defined by the contents of a Java class. I.e., the hierarchy of methods,
164      * fields, attributes, etc. spawns a tree of objects.
165      *
166      * @param v Visitor object
167      */
168     @Override
169     public void accept( final Visitor v ) {
170         v.visitConstantUtf8(this);
171     }

173 
174     /**
175      * Dump String in Utf8 format to file stream.
176      *
177      * @param file Output file stream
178      * @throws IOException
179      */
180     @Override
181     public final void dump( final DataOutputStream file ) throws IOException {
182         file.writeByte(super.getTag());
183         file.writeUTF(bytes);
184     }
185 
186 
187     /**
188      * @return Data converted to string.
189      */
190     public final String getBytes() {
191         return bytes;
192     }
193 
194 
195     /**
196      * @param bytes the raw bytes of this Utf-8
197      * @deprecated (since 6.0)
198      */
199     @java.lang.Deprecated
200     public final void setBytes( final String bytes ) {
201         throw new UnsupportedOperationException();
202     }
203 
204 
205     /**
206      * @return String representation
207      */
208     @Override
209     public final String toString() {
210         return super.toString() + "(\"" + Utility.replace(bytes, "\n", "\\n") + "\")";
211     }

 31 /**
 32  * Extends the abstract {@link Constant} to represent a reference to a UTF-8 encoded string.

 33  *
 34  * @see Constant
 35  * @LastModified: Jan 2020
 36  */
 37 public final class ConstantUtf8 extends Constant {
 38 
 39     private static class Cache {
 40 
 41         private static final boolean BCEL_STATISTICS = false;
 42         private static final int MAX_ENTRIES = 20000;
 43         private static final int INITIAL_CAPACITY = (int) (MAX_ENTRIES / 0.75);


 44 
 45         private static final HashMap<String, ConstantUtf8> CACHE = new LinkedHashMap<String, ConstantUtf8>(
 46             INITIAL_CAPACITY, 0.75f, true) {

 47 








 48             private static final long serialVersionUID = -8506975356158971766L;
 49 
 50             @Override
 51             protected boolean removeEldestEntry(final Map.Entry<String, ConstantUtf8> eldest) {
 52                 return size() > MAX_ENTRIES;
 53             }
 54         };
 55 
 56         // Set the size to 0 or below to skip caching entirely
 57         private static final int MAX_ENTRY_SIZE = 200;
 58 
 59         static boolean isEnabled() {
 60             return Cache.MAX_ENTRIES > 0 && MAX_ENTRY_SIZE > 0;
 61         }


 62 



 63     }
 64 
 65     /**
 66      * Clears the cache.
 67      *
 68      * @since 6.4.0
 69      */
 70     public static synchronized void clearCache() {
 71         Cache.CACHE.clear();


 72     }
 73 
 74     /**
 75      * Gets a new or cached instance of the given value.
 76      * <p>
 77      * See {@link ConstantUtf8} class Javadoc for details.
 78      * </p>
 79      *
 80      * @param value the value.
 81      * @return a new or cached instance of the given value.
 82      * @since 6.0
 83      */
 84     public static ConstantUtf8 getCachedInstance(final String value) {
 85         if (value.length() > Cache.MAX_ENTRY_SIZE) {
 86             return new ConstantUtf8(value);

 87         }
 88 
 89         synchronized (ConstantUtf8.class) { // might be better with a specific lock object
 90             ConstantUtf8 result = Cache.CACHE.get(value);
 91             if (result != null) {
 92                 return result;
 93             }
 94             result = new ConstantUtf8(value);
 95             Cache.CACHE.put(value, result);

 96             return result;
 97         }
 98     }
 99 
100     /**
101      * Gets a new or cached instance of the given value.
102      * <p>
103      * See {@link ConstantUtf8} class Javadoc for details.
104      * </p>
105      *
106      * @param dataInput the value.
107      * @return a new or cached instance of the given value.
108      * @throws IOException if an I/O error occurs.
109      * @since 6.0
110      */
111     public static ConstantUtf8 getInstance(final DataInput dataInput) throws IOException {
112         return getInstance(dataInput.readUTF());
113     }
114 
115     /**
116      * Gets a new or cached instance of the given value.
117      * <p>
118      * See {@link ConstantUtf8} class Javadoc for details.
119      * </p>
120      *
121      * @param value the value.
122      * @return a new or cached instance of the given value.
123      * @since 6.0
124      */
125     public static ConstantUtf8 getInstance(final String value) {
126         return Cache.isEnabled() ? getCachedInstance(value) : new ConstantUtf8(value);
127     }
128 
129     private final String value;
130 
131     /**
132      * Initializes from another object.
133      *
134      * @param constantUtf8 the value.
135      */
136     public ConstantUtf8(final ConstantUtf8 constantUtf8) {
137         this(constantUtf8.getBytes());
138     }

147         super(Const.CONSTANT_Utf8);
148         value = dataInput.readUTF();

149     }

157             throw new IllegalArgumentException("Value must not be null.");
158         }
159         this.value = value;

160     }

166      * @param v Visitor object

170         v.visitConstantUtf8(this);
171     }

178      */

182         file.writeUTF(value);
