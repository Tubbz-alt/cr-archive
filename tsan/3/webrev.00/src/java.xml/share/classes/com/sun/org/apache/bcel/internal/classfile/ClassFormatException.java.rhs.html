22 package com.sun.org.apache.bcel.internal.classfile;
23 
24 /**
25  * Thrown when the BCEL attempts to read a class file and determines
26  * that the file is malformed or otherwise cannot be interpreted as a
27  * class file.
28  *

29  */
30 public class ClassFormatException extends RuntimeException {
31 
32     private static final long serialVersionUID = -3569097343160139969L;
33 
34     public ClassFormatException() {
35         super();
36     }
37 
38 
39     public ClassFormatException(final String s) {
40         super(s);
41     }
42 
43     /**
44      * @since 6.0
45      */
46     public ClassFormatException(final String message, final Throwable cause) {
47         super(message, cause);
48     }
49 }
