22 package com.sun.org.apache.bcel.internal.generic;
23 
24 /**
25  * Denotes that an instruction may start the process of loading and resolving
26  * the referenced class in the Virtual Machine.
27  *
28  * @version $Id: LoadClass.java 1749597 2016-06-21 20:28:51Z ggregory $
29  */
30 public interface LoadClass {
31 
32     /**
33      * Returns the ObjectType of the referenced class or interface
34      * that may be loaded and resolved.
35      * @return object type that may be loaded or null if a primitive is
36      * referenced
37      */
38     ObjectType getLoadClassType( ConstantPoolGen cpg );
39 
40 
41     /**
42      * Returns the type associated with this instruction.
43      * LoadClass instances are always typed, but this type
44      * does not always refer to the type of the class or interface
45      * that it possibly forces to load. For example, GETFIELD would
46      * return the type of the field and not the type of the class
47      * where the field is defined.
48      * If no class is forced to be loaded, <B>null</B> is returned.
49      * An example for this is an ANEWARRAY instruction that creates
50      * an int[][].
51      * @see #getLoadClassType(ConstantPoolGen)
52      */
53     Type getType( ConstantPoolGen cpg );
54 }
