24 /**
25  * Thrown on internal errors. Extends RuntimeException so it hasn't to be declared
26  * in the throws clause every time.
27  *
28  * @version $Id: ClassGenException.java 1747278 2016-06-07 17:28:43Z britter $
29  */
30 public class ClassGenException extends RuntimeException {
31 
32     private static final long serialVersionUID = 7247369755051242791L;
33 
34     public ClassGenException() {
35         super();
36     }
37 
38 
39     public ClassGenException(final String s) {
40         super(s);
41     }
42 
43     public ClassGenException(final String s, final Throwable initCause) {
44         super(s, initCause);
45     }
46 }
