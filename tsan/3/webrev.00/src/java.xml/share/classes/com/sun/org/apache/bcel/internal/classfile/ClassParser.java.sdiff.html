 34 /**
 35  * Wrapper class that parses a given Java .class file. The method <A
 36  * href ="#parse">parse</A> returns a <A href ="JavaClass.html">
 37  * JavaClass</A> object on success. When an I/O error or an
 38  * inconsistency occurs an appropiate exception is propagated back to
 39  * the caller.
 40  *
 41  * The structure and the names comply, except for a few conveniences,
 42  * exactly with the <A href="http://docs.oracle.com/javase/specs/">
 43  * JVM specification 1.0</a>. See this paper for
 44  * further details about the structure of a bytecode file.
 45  *
 46  * @version $Id: ClassParser.java 1749603 2016-06-21 20:50:19Z ggregory $
 47  */
 48 public final class ClassParser {
 49 
 50     private DataInputStream dataInputStream;
 51     private final boolean fileOwned;
 52     private final String file_name;
 53     private String zip_file;
 54     private int class_name_index;
 55     private int superclass_name_index;
 56     private int major; // Compiler version
 57     private int minor; // Compiler version
 58     private int access_flags; // Access rights of parsed class
 59     private int[] interfaces; // Names of implemented interfaces
 60     private ConstantPool constant_pool; // collection of constants
 61     private Field[] fields; // class fields, i.e., its variables
 62     private Method[] methods; // methods defined in the class
 63     private Attribute[] attributes; // attributes defined in the class
 64     private final boolean is_zip; // Loaded from zip file
 65     private static final int BUFSIZE = 8192;
 66 
 67 
 68     /**
 69      * Parse class from the given stream.
 70      *
 71      * @param inputStream Input stream
 72      * @param file_name File name
 73      */
 74     public ClassParser(final InputStream inputStream, final String file_name) {
 75         this.file_name = file_name;
 76         fileOwned = false;
 77         final String clazz = inputStream.getClass().getName(); // Not a very clean solution ...
 78         is_zip = clazz.startsWith("java.util.zip.") || clazz.startsWith("java.util.jar.");
 79         if (inputStream instanceof DataInputStream) {
 80             this.dataInputStream = (DataInputStream) inputStream;
 81         } else {
 82             this.dataInputStream = new DataInputStream(new BufferedInputStream(inputStream, BUFSIZE));
 83         }
 84     }

 87     /** Parse class from given .class file.
 88      *
 89      * @param file_name file name
 90      */
 91     public ClassParser(final String file_name) {
 92         is_zip = false;
 93         this.file_name = file_name;
 94         fileOwned = true;
 95     }

 99      *
100      * @param zip_file zip file name
101      * @param file_name file name
102      */
103     public ClassParser(final String zip_file, final String file_name) {
104         is_zip = true;

106         this.zip_file = zip_file;
107         this.file_name = file_name;
108     }

111     /**
112      * Parse the given Java class file and return an object that represents
113      * the contained data, i.e., constants, methods, fields and commands.
114      * A <em>ClassFormatException</em> is raised, if the file is not a valid
115      * .class file. (This does not include verification of the byte code as it
116      * is performed by the java interpreter).
117      *
118      * @return Class object representing the parsed class file
119      * @throws  IOException
120      * @throws  ClassFormatException
121      */
122     public JavaClass parse() throws IOException, ClassFormatException {
123         ZipFile zip = null;
124         try {
125             if (fileOwned) {
126                 if (is_zip) {
127                     zip = new ZipFile(zip_file);
128                     final ZipEntry entry = zip.getEntry(file_name);
129 
130                     if (entry == null) {
131                         throw new IOException("File " + file_name + " not found");
132                     }

182                     //ignore close exceptions
183                 }
184             }
185             try {
186                 if (zip != null) {
187                     zip.close();
188                 }
189             } catch (final IOException ioe) {
190                 //ignore close exceptions
191             }
192         }
193         // Return the information we have gathered in a new object
194         return new JavaClass(class_name_index, superclass_name_index, file_name, major, minor,
195                 access_flags, constant_pool, interfaces, fields, methods, attributes, is_zip
196                         ? JavaClass.ZIP
197                         : JavaClass.FILE);
198     }

201     /**
202      * Read information about the attributes of the class.
203      * @throws  IOException
204      * @throws  ClassFormatException
205      */
206     private void readAttributes() throws IOException, ClassFormatException {
207         final int attributes_count = dataInputStream.readUnsignedShort();
208         attributes = new Attribute[attributes_count];
209         for (int i = 0; i < attributes_count; i++) {
210             attributes[i] = Attribute.readAttribute(dataInputStream, constant_pool);
211         }
212     }

215     /**
216      * Read information about the class and its super class.
217      * @throws  IOException
218      * @throws  ClassFormatException
219      */
220     private void readClassInfo() throws IOException, ClassFormatException {
221         access_flags = dataInputStream.readUnsignedShort();
222         /* Interfaces are implicitely abstract, the flag should be set
223          * according to the JVM specification.
224          */
225         if ((access_flags & Const.ACC_INTERFACE) != 0) {
226             access_flags |= Const.ACC_ABSTRACT;
227         }
228         if (((access_flags & Const.ACC_ABSTRACT) != 0)
229                 && ((access_flags & Const.ACC_FINAL) != 0)) {
230             throw new ClassFormatException("Class " + file_name + " can't be both final and abstract");
231         }
232         class_name_index = dataInputStream.readUnsignedShort();
233         superclass_name_index = dataInputStream.readUnsignedShort();
234     }

238      * Read constant pool entries.
239      * @throws  IOException
240      * @throws  ClassFormatException
241      */
242     private void readConstantPool() throws IOException, ClassFormatException {
243         constant_pool = new ConstantPool(dataInputStream);
244     }

250      * @throws  ClassFormatException
251      */
252     private void readFields() throws IOException, ClassFormatException {
253         final int fields_count = dataInputStream.readUnsignedShort();
254         fields = new Field[fields_count];
255         for (int i = 0; i < fields_count; i++) {
256             fields[i] = new Field(dataInputStream, constant_pool);
257         }
258     }

268     private void readID() throws IOException, ClassFormatException {

295         final int methods_count = dataInputStream.readUnsignedShort();
296         methods = new Method[methods_count];
297         for (int i = 0; i < methods_count; i++) {
298             methods[i] = new Method(dataInputStream, constant_pool);
299         }
300     }

304      * Read major and minor version of compiler which created the file.
305      * @throws  IOException
306      * @throws  ClassFormatException
307      */
308     private void readVersion() throws IOException, ClassFormatException {
309         minor = dataInputStream.readUnsignedShort();
310         major = dataInputStream.readUnsignedShort();
311     }
