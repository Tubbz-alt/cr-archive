 22 package com.sun.org.apache.bcel.internal.generic;
 23 
 24 import com.sun.org.apache.bcel.internal.classfile.LineNumber;
 25 
 26 /**
 27  * This class represents a line number within a method, i.e., give an instruction
 28  * a line number corresponding to the source code line.
 29  *
 30  * @version $Id: LineNumberGen.java 1749603 2016-06-21 20:50:19Z ggregory $
 31  * @see     LineNumber
 32  * @see     MethodGen
 33  */
 34 public class LineNumberGen implements InstructionTargeter, Cloneable {
 35 
 36     private InstructionHandle ih;
 37     private int src_line;
 38 
 39 
 40     /**
 41      * Create a line number.
 42      *
 43      * @param ih instruction handle to reference
 44      */
 45     public LineNumberGen(final InstructionHandle ih, final int src_line) {
 46         setInstruction(ih);
 47         setSourceLine(src_line);
 48     }
 49 
 50 
 51     /**
 52      * @return true, if ih is target of this line number
 53      */
 54     @Override
 55     public boolean containsTarget( final InstructionHandle ih ) {
 56         return this.ih == ih;
 57     }
 58 
 59 
 60     /**
 61      * @param old_ih old target
 62      * @param new_ih new target
 63      */
 64     @Override
 65     public void updateTarget( final InstructionHandle old_ih, final InstructionHandle new_ih ) {
 66         if (old_ih != ih) {
 67             throw new ClassGenException("Not targeting " + old_ih + ", but " + ih + "}");
 68         }
 69         setInstruction(new_ih);
 70     }
 71 
 72 
 73     /**
 74      * Get LineNumber attribute .
 75      *
 76      * This relies on that the instruction list has already been dumped to byte code or
 77      * or that the `setPositions' methods has been called for the instruction list.
 78      */
 79     public LineNumber getLineNumber() {
 80         return new LineNumber(ih.getPosition(), src_line);
 81     }
 82 
 83 
 84     public void setInstruction( final InstructionHandle ih ) { // TODO could be package-protected?
 85         if (ih == null) {
 86             throw new NullPointerException("InstructionHandle may not be null");
 87         }
 88         BranchInstruction.notifyTarget(this.ih, ih, this);
 89         this.ih = ih;
 90     }
 91 
 92 
 93     @Override
 94     public Object clone() {
 95         try {
 96             return super.clone();
 97         } catch (final CloneNotSupportedException e) {
 98             throw new Error("Clone Not Supported"); // never happens
 99         }
100     }
101 
102 
103     public InstructionHandle getInstruction() {
104         return ih;
105     }
106 
107 
108     public void setSourceLine( final int src_line ) { // TODO could be package-protected?
109         this.src_line = src_line;
110     }
111 
112 
113     public int getSourceLine() {
114         return src_line;
115     }
116 }
