22 package com.sun.org.apache.bcel.internal.generic;
23 
24 /**
25  * DUP2 - Duplicate two top operand stack words
26  * <PRE>Stack: ..., word2, word1 -&gt; ..., word2, word1, word2, word1</PRE>
27  *
28  */
29 public class DUP2 extends StackInstruction implements PushInstruction {
30 
31     public DUP2() {
32         super(com.sun.org.apache.bcel.internal.Const.DUP2);
33     }
34 
35 
36     /**
37      * Call corresponding visitor method(s). The order is:
38      * Call visitor methods of implemented interfaces first, then
39      * call methods according to the class hierarchy in descending order,
40      * i.e., the most specific visitXXX() call comes last.
41      *
42      * @param v Visitor object
43      */
44     @Override
45     public void accept( final Visitor v ) {
46         v.visitStackProducer(this);
47         v.visitPushInstruction(this);
48         v.visitStackInstruction(this);
49         v.visitDUP2(this);
50     }
51 }
