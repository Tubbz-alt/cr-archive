 21 package com.sun.org.apache.bcel.internal.generic;

 23 import java.io.DataOutputStream;

 26 import com.sun.org.apache.bcel.internal.classfile.Constant;

 31 /**
 32  * Abstract super class for instructions that use an index into the constant
 33  * pool such as LDC, INVOKEVIRTUAL, etc.
 34  *
 35  * @see ConstantPoolGen
 36  * @see LDC
 37  * @see INVOKEVIRTUAL
 38  *
 39  * @version $Id: CPInstruction.java 1749603 2016-06-21 20:50:19Z ggregory $
 40  */
 41 public abstract class CPInstruction extends Instruction implements TypedInstruction,
 42         IndexedInstruction {
 43 
 44     private int index; // index to constant pool
 45 

 46     /**
 47      * Empty constructor needed for the Class.newInstance() statement in
 48      * Instruction.readInstruction(). Not to be used otherwise.
 49      */
 50     CPInstruction() {
 51     }
 52 

 53     /**
 54      * @param index to constant pool
 55      */
 56     protected CPInstruction(final short opcode, final int index) {
 57         super(opcode, (short) 3);
 58         setIndex(index);
 59     }
 60 

 61     /**
 62      * Dump instruction as byte code to stream out.
 63      *
 64      * @param out Output stream
 65      */
 66     @Override
 67     public void dump(final DataOutputStream out) throws IOException {
 68         out.writeByte(super.getOpcode());
 69         out.writeShort(index);
 70     }
 71 

 72     /**
 73      * Long output format:
 74      *
 75      * &lt;name of opcode&gt; "["&lt;opcode number&gt;"]" "("&lt;length of
 76      * instruction&gt;")" "&lt;"&lt; constant pool index&gt;"&gt;"
 77      *
 78      * @param verbose long/short format switch
 79      * @return mnemonic for instruction
 80      */
 81     @Override
 82     public String toString(final boolean verbose) {
 83         return super.toString(verbose) + " " + index;
 84     }
 85 

 86     /**
 87      * @return mnemonic for instruction with symbolic references resolved
 88      */
 89     @Override
 90     public String toString(final ConstantPool cp) {
 91         final Constant c = cp.getConstant(index);
 92         String str = cp.constantToString(c);
 93         if (c instanceof ConstantClass) {
 94             str = str.replace('.', '/');
 95         }
 96         return com.sun.org.apache.bcel.internal.Const.getOpcodeName(super.getOpcode()) + " " + str;
 97     }
 98 

 99     /**
100      * Read needed data (i.e., index) from file.
101      *
102      * @param bytes input stream
103      * @param wide wide prefix?
104      */
105     @Override
106     protected void initFromFile(final ByteSequence bytes, final boolean wide) throws IOException {
107         setIndex(bytes.readUnsignedShort());
108         super.setLength(3);
109     }
110 

111     /**
112      * @return index in constant pool referred by this instruction.
113      */
114     @Override
115     public final int getIndex() {
116         return index;
117     }
118 

119     /**
120      * Set the index to constant pool.
121      *
122      * @param index in constant pool.
123      */
124     @Override
125     public void setIndex(final int index) { // TODO could be package-protected?
126         if (index < 0) {
127             throw new ClassGenException("Negative index value: " + index);
128         }
129         this.index = index;
130     }
131 
132     /**
133      * @return type related with this instruction.
134      */
135     @Override
136     public Type getType(final ConstantPoolGen cpg) {
137         final ConstantPool cp = cpg.getConstantPool();
138         String name = cp.getConstantString(index, com.sun.org.apache.bcel.internal.Const.CONSTANT_Class);
139         if (!name.startsWith("[")) {
140             name = "L" + name + ";";
141         }
142         return Type.getType(name);
143     }
144 }
