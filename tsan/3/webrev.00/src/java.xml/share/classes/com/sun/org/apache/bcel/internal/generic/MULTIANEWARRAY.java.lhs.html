 22 package com.sun.org.apache.bcel.internal.generic;
 23 
 24 import java.io.DataOutputStream;
 25 import java.io.IOException;
 26 
 27 import com.sun.org.apache.bcel.internal.ExceptionConst;
 28 import com.sun.org.apache.bcel.internal.classfile.ConstantPool;
 29 import com.sun.org.apache.bcel.internal.util.ByteSequence;
 30 
 31 /**
 32  * MULTIANEWARRAY - Create new mutidimensional array of references
 33  * <PRE>Stack: ..., count1, [count2, ...] -&gt; ..., arrayref</PRE>
 34  *
 35  * @version $Id: MULTIANEWARRAY.java 1747278 2016-06-07 17:28:43Z britter $
 36  */
 37 public class MULTIANEWARRAY extends CPInstruction implements LoadClass, AllocationInstruction,
 38         ExceptionThrower {
 39 
 40     private short dimensions;
 41 
 42 
 43     /**
 44      * Empty constructor needed for the Class.newInstance() statement in
 45      * Instruction.readInstruction(). Not to be used otherwise.
 46      */
 47     MULTIANEWARRAY() {
 48     }
 49 
 50 
 51     public MULTIANEWARRAY(final int index, final short dimensions) {
 52         super(com.sun.org.apache.bcel.internal.Const.MULTIANEWARRAY, index);
 53         if (dimensions < 1) {
 54             throw new ClassGenException("Invalid dimensions value: " + dimensions);
 55         }
 56         this.dimensions = dimensions;
 57         super.setLength(4);
 58     }
 59 
 60 
 61     /**
 62      * Dump instruction as byte code to stream out.
 63      * @param out Output stream
 64      */
 65     @Override
 66     public void dump( final DataOutputStream out ) throws IOException {
 67         out.writeByte(super.getOpcode());
 68         out.writeShort(super.getIndex());
 69         out.writeByte(dimensions);
 70     }
 71 
 72 
 73     /**
 74      * Read needed data (i.e., no. dimension) from file.
 75      */
 76     @Override
 77     protected void initFromFile( final ByteSequence bytes, final boolean wide ) throws IOException {
 78         super.initFromFile(bytes, wide);
 79         dimensions = bytes.readByte();
 80         super.setLength(4);
 81     }
 82 
 83 
 84     /**
 85      * @return number of dimensions to be created
 86      */
 87     public final short getDimensions() {
 88         return dimensions;
 89     }
 90 
 91 
 92     /**
 93      * @return mnemonic for instruction
 94      */
 95     @Override
 96     public String toString( final boolean verbose ) {
 97         return super.toString(verbose) + " " + super.getIndex() + " " + dimensions;
 98     }
 99 
100 
101     /**
102      * @return mnemonic for instruction with symbolic references resolved
103      */
104     @Override
105     public String toString( final ConstantPool cp ) {
106         return super.toString(cp) + " " + dimensions;
107     }
108 
109 
110     /**
111      * Also works for instructions whose stack effect depends on the
112      * constant pool entry they reference.
113      * @return Number of words consumed from stack by this instruction
114      */
115     @Override
116     public int consumeStack( final ConstantPoolGen cpg ) {
117         return dimensions;
118     }
119 
120 
121     @Override
122     public Class<?>[] getExceptions() {
123         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION,
124             ExceptionConst.ILLEGAL_ACCESS_ERROR,
125             ExceptionConst.NEGATIVE_ARRAY_SIZE_EXCEPTION);
126     }
127 
128 
129     @Override
130     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
131         Type t = getType(cpg);
132         if (t instanceof ArrayType) {
133             t = ((ArrayType) t).getBasicType();
134         }
135         return (t instanceof ObjectType) ? (ObjectType) t : null;
136     }
137 
138 
139     /**
140      * Call corresponding visitor method(s). The order is:
141      * Call visitor methods of implemented interfaces first, then
142      * call methods according to the class hierarchy in descending order,
143      * i.e., the most specific visitXXX() call comes last.
144      *
145      * @param v Visitor object
146      */
147     @Override
148     public void accept( final Visitor v ) {
149         v.visitLoadClass(this);
150         v.visitAllocationInstruction(this);
151         v.visitExceptionThrower(this);
152         v.visitTypedInstruction(this);
153         v.visitCPInstruction(this);
154         v.visitMULTIANEWARRAY(this);
155     }
156 }
