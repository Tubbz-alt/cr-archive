22 package com.sun.org.apache.bcel.internal.generic;

24 import java.io.IOException;

26 import com.sun.org.apache.bcel.internal.util.ByteSequence;

28 /**
29  * LDC_W - Push item from constant pool (wide index)
30  *
31  * <PRE>Stack: ... -&gt; ..., item.word1, item.word2</PRE>
32  *
33  */
34 public class LDC_W extends LDC {
35 
36     /**
37      * Empty constructor needed for Instruction.readInstruction.
38      * Not to be used otherwise.
39      */
40     LDC_W() {
41     }
42 
43 
44     public LDC_W(final int index) {
45         super(index);
46     }
47 
48 
49     /**
50      * Read needed data (i.e., index) from file.
51      */
52     @Override
53     protected void initFromFile( final ByteSequence bytes, final boolean wide ) throws IOException {
54         setIndex(bytes.readUnsignedShort());
55         // Override just in case it has been changed
56         super.setOpcode(com.sun.org.apache.bcel.internal.Const.LDC_W);
57         super.setLength(3);
58     }
59 }
