 22 package com.sun.org.apache.bcel.internal.classfile;
 23 
 24 import java.io.DataInput;
 25 import java.io.DataOutputStream;
 26 import java.io.IOException;
 27 import java.util.ArrayList;
 28 import java.util.Collections;
 29 import java.util.List;
 30 
 31 import com.sun.org.apache.bcel.internal.Const;
 32 
 33 /**
 34  * represents one annotation in the annotation table
 35  *
 36  * @since 6.0
 37  */
 38 public class AnnotationEntry implements Node {
 39 
 40     private final int type_index;
 41     private final ConstantPool constant_pool;
 42     private final boolean isRuntimeVisible;
 43 
 44     private List<ElementValuePair> element_value_pairs;
 45 
 46     /*
 47      * Factory method to create an AnnotionEntry from a DataInput
 48      *
 49      * @param input
 50      * @param constant_pool
 51      * @param isRuntimeVisible
 52      * @return the entry
 53      * @throws IOException
 54      */
 55     public static AnnotationEntry read(final DataInput input, final ConstantPool constant_pool, final boolean isRuntimeVisible) throws IOException {
 56 
 57         final AnnotationEntry annotationEntry = new AnnotationEntry(input.readUnsignedShort(), constant_pool, isRuntimeVisible);
 58         final int num_element_value_pairs = input.readUnsignedShort();
 59         annotationEntry.element_value_pairs = new ArrayList<>();
 60         for (int i = 0; i < num_element_value_pairs; i++) {
 61             annotationEntry.element_value_pairs.add(
 62                     new ElementValuePair(input.readUnsignedShort(), ElementValue.readElementValue(input, constant_pool),
 63                     constant_pool));
 64         }
 65         return annotationEntry;
 66     }
 67 
 68     public AnnotationEntry(final int type_index, final ConstantPool constant_pool, final boolean isRuntimeVisible) {
 69         this.type_index = type_index;
 70         this.constant_pool = constant_pool;
 71         this.isRuntimeVisible = isRuntimeVisible;
 72     }
 73 
 74     public int getTypeIndex() {
 75         return type_index;
 76     }
 77 
 78     public ConstantPool getConstantPool() {
 79         return constant_pool;
 80     }
 81 
 82     public boolean isRuntimeVisible() {
 83         return isRuntimeVisible;
 84     }
 85 
 86     /**
 87      * Called by objects that are traversing the nodes of the tree implicitely defined by the contents of a Java class.
 88      * I.e., the hierarchy of methods, fields, attributes, etc. spawns a tree of objects.
 89      *
 90      * @param v Visitor object
 91      */
 92     @Override
 93     public void accept(final Visitor v) {
 94         v.visitAnnotationEntry(this);
 95     }
 96 
 97     /**
 98      * @return the annotation type name
 99      */
100     public String getAnnotationType() {
101         final ConstantUtf8 c = (ConstantUtf8) constant_pool.getConstant(type_index, Const.CONSTANT_Utf8);
102         return c.getBytes();
103     }
104 
105     /**
106      * @return the annotation type index
107      */
108     public int getAnnotationTypeIndex() {
109         return type_index;
110     }
111 
112     /**
113      * @return the number of element value pairs in this annotation entry
114      */
115     public final int getNumElementValuePairs() {
116         return element_value_pairs.size();
117     }
118 
119     /**
120      * @return the element value pairs in this annotation entry
121      */
122     public ElementValuePair[] getElementValuePairs() {
123         // TODO return List
124         return element_value_pairs.toArray(new ElementValuePair[element_value_pairs.size()]);
125     }
126 
127     public void dump(final DataOutputStream dos) throws IOException {
128         dos.writeShort(type_index); // u2 index of type name in cpool
129         dos.writeShort(element_value_pairs.size()); // u2 element_value pair
130         // count
131         for (final ElementValuePair envp : element_value_pairs) {
132             envp.dump(dos);
133         }
134     }
135 
136     public void addElementNameValuePair(final ElementValuePair elementNameValuePair) {
137         element_value_pairs.add(elementNameValuePair);
138     }
139 
140     public String toShortString() {
141         final StringBuilder result = new StringBuilder();
142         result.append("@");
143         result.append(getAnnotationType());
144         final ElementValuePair[] evPairs = getElementValuePairs();
145         if (evPairs.length > 0) {
146             result.append("(");
147             for (final ElementValuePair element : evPairs) {
148                 result.append(element.toShortString());
149             }
150             result.append(")");
151         }
152         return result.toString();
153     }
154 
155     @Override
156     public String toString() {
157         return toShortString();
158     }
159 
160     public static AnnotationEntry[] createAnnotationEntries(final Attribute[] attrs) {
161         // Find attributes that contain annotation data
162         final List<AnnotationEntry> accumulatedAnnotations = new ArrayList<>(attrs.length);
163         for (final Attribute attribute : attrs) {
164             if (attribute instanceof Annotations) {
165                 final Annotations runtimeAnnotations = (Annotations) attribute;
166                 Collections.addAll(accumulatedAnnotations, runtimeAnnotations.getAnnotationEntries());
167             }
168         }
169         return accumulatedAnnotations.toArray(new AnnotationEntry[accumulatedAnnotations.size()]);
170     }
171 }
