22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import java.io.DataOutputStream;
25 import java.io.IOException;
26 
27 /**
28  * JSR - Jump to subroutine
29  *

30  */
31 public class JSR extends JsrInstruction implements VariableLengthInstruction {
32 
33     /**
34      * Empty constructor needed for Instruction.readInstruction.
35      * Not to be used otherwise.
36      */
37     JSR() {
38     }
39 
40 
41     public JSR(final InstructionHandle target) {
42         super(com.sun.org.apache.bcel.internal.Const.JSR, target);
43     }
44 
45 
46     /**
47      * Dump instruction as byte code to stream out.
48      * @param out Output stream
49      */
50     @Override
51     public void dump( final DataOutputStream out ) throws IOException {
52         super.setIndex(getTargetOffset());
53         if (super.getOpcode() == com.sun.org.apache.bcel.internal.Const.JSR) {
54             super.dump(out);
55         } else { // JSR_W
56             super.setIndex(getTargetOffset());
57             out.writeByte(super.getOpcode());
58             out.writeInt(super.getIndex());
59         }
60     }
61 
62 
63     @Override
64     protected int updatePosition( final int offset, final int max_offset ) {
65         final int i = getTargetOffset(); // Depending on old position value
66         setPosition(getPosition() + offset); // Position may be shifted by preceding expansions
67         if (Math.abs(i) >= (Short.MAX_VALUE - max_offset)) { // to large for short (estimate)
68             super.setOpcode(com.sun.org.apache.bcel.internal.Const.JSR_W);
69             final short old_length = (short) super.getLength();
70             super.setLength(5);
71             return super.getLength() - old_length;
72         }
73         return 0;
74     }
75 
76 
77     /**
78      * Call corresponding visitor method(s). The order is:
79      * Call visitor methods of implemented interfaces first, then
80      * call methods according to the class hierarchy in descending order,
81      * i.e., the most specific visitXXX() call comes last.
82      *
83      * @param v Visitor object
84      */
85     @Override
86     public void accept( final Visitor v ) {
87         v.visitStackProducer(this);
88         v.visitVariableLengthInstruction(this);
89         v.visitBranchInstruction(this);
90         v.visitJsrInstruction(this);
91         v.visitJSR(this);
92     }
93 }
