22 package com.sun.org.apache.bcel.internal.generic;

24 /**
25  * IF_ACMPNE - Branch if reference comparison doesn't succeed
26  *
27  * <PRE>Stack: ..., value1, value2 -&gt; ...</PRE>
28  *
29  */
30 public class IF_ACMPNE extends IfInstruction {
31 
32     /**
33      * Empty constructor needed for Instruction.readInstruction.
34      * Not to be used otherwise.
35      */
36     IF_ACMPNE() {
37     }
38 
39 
40     public IF_ACMPNE(final InstructionHandle target) {
41         super(com.sun.org.apache.bcel.internal.Const.IF_ACMPNE, target);
42     }
43 
44 
45     /**
46      * @return negation of instruction
47      */
48     @Override
49     public IfInstruction negate() {
50         return new IF_ACMPEQ(super.getTarget());
51     }
52 
53 
54     /**
55      * Call corresponding visitor method(s). The order is:
56      * Call visitor methods of implemented interfaces first, then
57      * call methods according to the class hierarchy in descending order,
58      * i.e., the most specific visitXXX() call comes last.
59      *
60      * @param v Visitor object
61      */
62     @Override
63     public void accept( final Visitor v ) {
64         v.visitStackConsumer(this);
65         v.visitBranchInstruction(this);
66         v.visitIfInstruction(this);
67         v.visitIF_ACMPNE(this);
68     }
69 }
