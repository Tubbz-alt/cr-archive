23 import com.sun.org.apache.bcel.internal.ExceptionConst;

25 /**
26  * ATHROW -  Throw exception
27  * <PRE>Stack: ..., objectref -&gt; objectref</PRE>
28  *
29  * @LastModified: Jan 2020

30  */
31 public class ATHROW extends Instruction implements UnconditionalBranch, ExceptionThrower {
32 
33     /**
34      *  Throw exception
35      */
36     public ATHROW() {
37         super(com.sun.org.apache.bcel.internal.Const.ATHROW, (short) 1);
38     }
39 
40 
41     /** @return exceptions this instruction may cause
42      */
43     @Override
44     public Class<?>[] getExceptions() {
45         return new Class<?>[] {
46             ExceptionConst.THROWABLE
47         };
48     }
49 
50 
51     /**
52      * Call corresponding visitor method(s). The order is:
53      * Call visitor methods of implemented interfaces first, then
54      * call methods according to the class hierarchy in descending order,
55      * i.e., the most specific visitXXX() call comes last.
56      *
57      * @param v Visitor object
58      */
59     @Override
60     public void accept( final Visitor v ) {
61         v.visitUnconditionalBranch(this);
62         v.visitExceptionThrower(this);
63         v.visitATHROW(this);
64     }
65 }
