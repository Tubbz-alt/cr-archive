24 /**
25  * Wrapper class for `compound' operations, virtual instructions that
26  * don't exist as byte code, but give a useful meaning. For example,
27  * the (virtual) PUSH instruction takes an arbitray argument and produces the
28  * appropiate code at dump time (ICONST, LDC, BIPUSH, ...). Also you can use the
29  * SWITCH instruction as a useful template for either LOOKUPSWITCH or
30  * TABLESWITCH.
31  *
32  * The interface provides the possibilty for the user to write
33  * `templates' or `macros' for such reuseable code patterns.
34  *
35  * @version $Id: CompoundInstruction.java 1747278 2016-06-07 17:28:43Z britter $
36  * @see PUSH
37  * @see SWITCH
38  */
39 public interface CompoundInstruction {
40 
41     InstructionList getInstructionList();
42 }
