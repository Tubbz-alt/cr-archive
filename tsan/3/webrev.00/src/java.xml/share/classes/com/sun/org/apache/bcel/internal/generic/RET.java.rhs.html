 22 package com.sun.org.apache.bcel.internal.generic;
 23 
 24 import java.io.DataOutputStream;
 25 import java.io.IOException;
 26 
 27 import com.sun.org.apache.bcel.internal.util.ByteSequence;
 28 
 29 /**
 30  * RET - Return from subroutine
 31  *
 32  * <PRE>Stack: ... -&gt; ...</PRE>
 33  *

 34  */
 35 public class RET extends Instruction implements IndexedInstruction, TypedInstruction {
 36 
 37     private boolean wide;
 38     private int index; // index to local variable containg the return address
 39 
 40 
 41     /**
 42      * Empty constructor needed for Instruction.readInstruction.
 43      * Not to be used otherwise.
 44      */
 45     RET() {
 46     }
 47 
 48 
 49     public RET(final int index) {
 50         super(com.sun.org.apache.bcel.internal.Const.RET, (short) 2);
 51         setIndex(index); // May set wide as side effect
 52     }
 53 
 54 
 55     /**
 56      * Dump instruction as byte code to stream out.
 57      * @param out Output stream
 58      */
 59     @Override
 60     public void dump( final DataOutputStream out ) throws IOException {
 61         if (wide) {
 62             out.writeByte(com.sun.org.apache.bcel.internal.Const.WIDE);
 63         }
 64         out.writeByte(super.getOpcode());
 65         if (wide) {
 66             out.writeShort(index);
 67         } else {
 68             out.writeByte(index);
 69         }
 70     }
 71 
 72 
 73     private void setWide() {
 74         wide = index > com.sun.org.apache.bcel.internal.Const.MAX_BYTE;
 75         if (wide) {
 76             super.setLength(4); // Including the wide byte
 77         } else {
 78             super.setLength(2);
 79         }
 80     }
 81 
 82 
 83     /**
 84      * Read needed data (e.g. index) from file.
 85      */
 86     @Override
 87     protected void initFromFile( final ByteSequence bytes, final boolean wide ) throws IOException {
 88         this.wide = wide;
 89         if (wide) {
 90             index = bytes.readUnsignedShort();
 91             super.setLength(4);
 92         } else {
 93             index = bytes.readUnsignedByte();
 94             super.setLength(2);
 95         }
 96     }
 97 
 98 
 99     /**
100      * @return index of local variable containg the return address
101      */
102     @Override
103     public final int getIndex() {
104         return index;
105     }
106 
107 
108     /**
109      * Set index of local variable containg the return address
110      */
111     @Override
112     public final void setIndex( final int n ) {
113         if (n < 0) {
114             throw new ClassGenException("Negative index value: " + n);
115         }
116         index = n;
117         setWide();
118     }
119 
120 
121     /**
122      * @return mnemonic for instruction
123      */
124     @Override
125     public String toString( final boolean verbose ) {
126         return super.toString(verbose) + " " + index;
127     }
128 
129 
130     /** @return return address type
131      */
132     @Override
133     public Type getType( final ConstantPoolGen cp ) {
134         return ReturnaddressType.NO_TARGET;
135     }
136 
137 
138     /**
139      * Call corresponding visitor method(s). The order is:
140      * Call visitor methods of implemented interfaces first, then
141      * call methods according to the class hierarchy in descending order,
142      * i.e., the most specific visitXXX() call comes last.
143      *
144      * @param v Visitor object
145      */
146     @Override
147     public void accept( final Visitor v ) {
148         v.visitRET(this);
149     }
150 }
