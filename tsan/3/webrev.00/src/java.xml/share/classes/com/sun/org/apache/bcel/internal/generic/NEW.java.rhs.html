22 package com.sun.org.apache.bcel.internal.generic;
23 
24 import com.sun.org.apache.bcel.internal.ExceptionConst;
25 
26 /**
27  * NEW - Create new object
28  * <PRE>Stack: ... -&gt; ..., objectref</PRE>
29  *

30  */
31 public class NEW extends CPInstruction implements LoadClass, AllocationInstruction,
32         ExceptionThrower, StackProducer {
33 
34     /**
35      * Empty constructor needed for Instruction.readInstruction.
36      * Not to be used otherwise.
37      */
38     NEW() {
39     }
40 
41 
42     public NEW(final int index) {
43         super(com.sun.org.apache.bcel.internal.Const.NEW, index);
44     }
45 
46 
47     @Override
48     public Class<?>[] getExceptions() {
49         return ExceptionConst.createExceptions(ExceptionConst.EXCS.EXCS_CLASS_AND_INTERFACE_RESOLUTION,
50             ExceptionConst.ILLEGAL_ACCESS_ERROR,
51             ExceptionConst.INSTANTIATION_ERROR);
52     }
53 
54 
55     @Override
56     public ObjectType getLoadClassType( final ConstantPoolGen cpg ) {
57         return (ObjectType) getType(cpg);
58     }
59 
60 
61     /**
62      * Call corresponding visitor method(s). The order is:
63      * Call visitor methods of implemented interfaces first, then
64      * call methods according to the class hierarchy in descending order,
65      * i.e., the most specific visitXXX() call comes last.
66      *
67      * @param v Visitor object
68      */
69     @Override
70     public void accept( final Visitor v ) {
71         v.visitLoadClass(this);
72         v.visitAllocationInstruction(this);
73         v.visitExceptionThrower(this);
74         v.visitStackProducer(this);
75         v.visitTypedInstruction(this);
76         v.visitCPInstruction(this);
77         v.visitNEW(this);
78     }
79 }
