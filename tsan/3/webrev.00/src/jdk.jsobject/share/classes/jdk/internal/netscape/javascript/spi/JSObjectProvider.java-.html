26 package jdk.internal.netscape.javascript.spi;
27 
28 import java.applet.Applet;
29 import netscape.javascript.JSException;
30 import netscape.javascript.JSObject;
31 
32 @SuppressWarnings("deprecation")
33 public interface JSObjectProvider {
34     /**
35      * Return a JSObject for the window containing the given applet.
36      * Implementations of this class should return null if not connected to a
37      * browser, for example, when running in AppletViewer.
38      *
39      * @param applet The applet.
40      * @return JSObject for the window containing the given applet or null if we
41      * are not connected to a browser.
42      * @throws JSException when an error is encountered.
43      */
44     public JSObject getWindow(Applet applet) throws JSException;
45 }
