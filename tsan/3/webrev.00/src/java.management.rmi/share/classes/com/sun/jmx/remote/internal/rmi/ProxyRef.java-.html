 26 package com.sun.jmx.remote.internal.rmi;
 27 
 28 import java.io.IOException;
 29 import java.io.ObjectInput;
 30 import java.io.ObjectOutput;
 31 import java.lang.reflect.Method;
 32 import java.rmi.Remote;
 33 import java.rmi.RemoteException;
 34 import java.rmi.server.RemoteObject;
 35 import java.rmi.server.RemoteRef;
 36 
 37 
 38 @SuppressWarnings("deprecation")
 39 public class ProxyRef implements RemoteRef {
 40     private static final long serialVersionUID = -6503061366316814723L;
 41 
 42     public ProxyRef(RemoteRef ref) {
 43         this.ref = ref;
 44     }
 45 
 46     public void readExternal(ObjectInput in)
 47             throws IOException, ClassNotFoundException {
 48         ref.readExternal(in);
 49     }
 50 
 51     public void writeExternal(ObjectOutput out) throws IOException {
 52         ref.writeExternal(out);
 53     }
 54 
 55     /**
 56      * @deprecated
 57      */
 58     @Deprecated
 59     public void invoke(java.rmi.server.RemoteCall call) throws Exception {
 60         ref.invoke(call);
 61     }
 62 
 63     public Object invoke(Remote obj, Method method, Object[] params,
 64                          long opnum) throws Exception {
 65         return ref.invoke(obj, method, params, opnum);
 66     }
 67 
 68     /**
 69      * @deprecated
 70      */
 71     @Deprecated
 72     public void done(java.rmi.server.RemoteCall call) throws RemoteException {
 73         ref.done(call);
 74     }
 75 
 76     public String getRefClass(ObjectOutput out) {
 77         return ref.getRefClass(out);
 78     }
 79 
 80     /**
 81      * @deprecated
 82      */
 83     @Deprecated
 84     public java.rmi.server.RemoteCall newCall(RemoteObject obj,
 85             java.rmi.server.Operation[] op, int opnum,
 86                               long hash) throws RemoteException {
 87         return ref.newCall(obj, op, opnum, hash);
 88     }
 89 
 90     public boolean remoteEquals(RemoteRef obj) {
 91         return ref.remoteEquals(obj);
 92     }
 93 
 94     public int remoteHashCode() {
 95         return ref.remoteHashCode();
 96     }
 97 
 98     public String remoteToString() {
 99         return ref.remoteToString();
100     }
101 
102     protected RemoteRef ref;
103 }
