26 package jdk.tools.jlink.internal.plugins;
27 





28 import jdk.tools.jlink.plugin.Plugin;
29 import jdk.tools.jlink.plugin.ResourcePool;
30 import jdk.tools.jlink.plugin.ResourcePoolBuilder;
31 
32 /**
33  * Combined debug stripping plugin: Java debug attributes and native debug
34  * symbols.
35  *
36  */
37 public final class DefaultStripDebugPlugin implements Plugin {
38 
39     public static final String NAME = "strip-debug";










40 
41     private final Plugin javaStripPlugin = new StripJavaDebugAttributesPlugin();




42 
43     @Override
44     public String getName() {
45         return NAME;
46     }
47 
48     @Override
49     public String getDescription() {
50         return PluginsResourceBundle.getDescription(NAME);
51     }
52 
53     @Override
54     public ResourcePool transform(ResourcePool in, ResourcePoolBuilder out) {
55         return javaStripPlugin.transform(in, out);



























56     }
57 
58 }
