 25 package jdk.tools.jlink.plugin;
 26 
 27 import java.util.Collections;
 28 import java.util.EnumSet;
 29 import java.util.Map;
 30 import java.util.Set;
 31 import jdk.tools.jlink.internal.plugins.PluginsResourceBundle;
 32 
 33 /**
 34  * Base interface that jlink plugins should implement.
 35  */
 36 public interface Plugin {
 37 
 38     /**
 39      * Order of categories matches the plugin sort order.
 40      * <ol>
 41      * <li>FILTER: Filter in/out resources or files.</li>
 42      * <li>ADDER: Add resources or files.</li>
 43      * <li>TRANSFORMER: Transform resources or files(eg: refactoring, bytecode
 44      * manipulation).</li>
 45      * <li>MODULEINFO_TRANSFORMER: Transform only module-info.class</li>
 46      * <li>SORTER: Sort resources within the resource container.</li>
 47      * <li>METAINFO_ADDER: Added meta info (like release, copyright etc.)</li>
 48      * <li>COMPRESSOR: Compress resource within the resouce containers.</li>
 49      * <li>VERIFIER: Does some image verification.</li>
 50      * <li>PROCESSOR: Does some post processing on image.</li>
 51      * <li>PACKAGER: Final processing</li>
 52      * </ol>
 53      */
 54     public enum Category {
 55         FILTER("FILTER"),
 56         ADDER("ADDER"),
 57         TRANSFORMER("TRANSFORMER"),
 58         MODULEINFO_TRANSFORMER("MODULEINFO_TRANSFORMER"),
 59         SORTER("SORTER"),
 60         METAINFO_ADDER("METAINFO_ADDER"),
 61         COMPRESSOR("COMPRESSOR"),
 62         VERIFIER("VERIFIER"),
 63         PROCESSOR("PROCESSOR"),
 64         PACKAGER("PACKAGER");
 65 
 66         private final String name;
 67 
 68         Category(String name) {
 69             this.name = name;
 70         }
 71 
 72         public String getName() {
 73             return name;
 74         }
 75     }
 76 
 77     /**
 78      * Plugin state:
 79      * <ul>
 80      * <li>DISABLED: The plugin is not exposed in help and will be not called.</li>
 81      * <li>AUTO_ENABLED: The plugin is enabled by default. It doesn't require its
 82      * option to be present to be called.<li>
 83      * <li>FUNCTIONAL: The plugin is properly configured and can operate.
 84      * Non functional plugin must advertise their status in the
 85      * {@link #getStateDescription() getStateDescription} method</li>
 86      * </ul>
 87      */
 88     public enum State {
 89         DISABLED,
 90         AUTO_ENABLED,
 91         FUNCTIONAL
 92     }
 93 
 94     /**
 95      * The type of this plugin.
 96      *
 97      * @return The type of this plugin
 98      */
 99     public default Category getType() {
100         return Category.TRANSFORMER;
101     }
102 
103     /**
104      * The Plugin set of states.
105      * @return The set of states.
106      */
107     public default Set<State> getState() {
108         return EnumSet.of(State.FUNCTIONAL);
109     }
110 
111     /**
112      * The plugin name.
113      * @return The name.
114      */
115     public default String getName() {
116         return getClass().getName().replace('.', '-');
117     }
118 
119     /**
120      * The plugin description.
121      * @return  The description.
122      */
123     public default String getDescription() {
124         return "";
125     }
126 
127     /**
128      * The option that identifies this plugin. This may be null.
129      * "--" is prefixed to the String (when non-null) when invoking
130      * this plugin from jlink command line.
131      *
132      * @return The plugin option.
133      */
134     public default String getOption() {
135         return getName();
136     }
137 
138     /**
139      * Has this plugin require one or more arguments?
140      * A plugin can have one or more optional arguments.
141      * <br>
142      * A plugin option with a single argument is specified as follows:
143      * <pre>
144      *     --plugin-option=arg_value
145      * </pre>
146      * If there are more than arguments, command line option looks like:
147      * <pre>
148      *     --plugin-option=arg_value:arg2=value2:arg3=value3...
149      *</pre>
150      *
151      * @return true if arguments are needed.
152      */
153     public default boolean hasArguments() {
154         return false;
155     }
156 
157     public default boolean hasRawArgument() {
158         return false;
159     }
160 
161     /**
162      * The plugin argument(s) description.
163      * @return  The argument(s) description.
164      */
165     public default String getArgumentsDescription() {
166         return "";
167     }
168 
169     /**
170      * Return a message indicating the status of the provider.
171      *
172      * @return A status description.
173      */
174     public default String getStateDescription() {
175         return getState().contains(State.FUNCTIONAL)
176                 ? PluginsResourceBundle.getMessage("main.status.ok")
177                 : PluginsResourceBundle.getMessage("main.status.not.ok");
178     }
179 
180     /**
181      * Configure the plugin based on the passed configuration.
182      * This method is called prior to invoke the plugin.
183      *
184      * @param config The plugin configuration.
185      * @throws IllegalArgumentException if a mandatory argument is missing or
186      * if an argument has invalid value.
187      */
188     public default void configure(Map<String, String> config) {
189     }
190 
191     /**
192      * Visit the content of the modules that are composing the image.
193      *
194      * @param in Read only content.
195      * @param out The pool to fill with content. This pool must contain
196      * the result of the visit.
197      *
198      * @throws PluginException
199      */
200     public ResourcePool transform(ResourcePool in, ResourcePoolBuilder out);
201 }
