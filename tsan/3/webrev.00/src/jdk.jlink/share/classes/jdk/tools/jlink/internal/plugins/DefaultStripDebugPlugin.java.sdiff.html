 28 import java.util.Map;
 29 
 30 import jdk.tools.jlink.internal.PluginRepository;
 31 import jdk.tools.jlink.internal.ResourcePoolManager;
 32 import jdk.tools.jlink.internal.ResourcePoolManager.ResourcePoolImpl;
 33 import jdk.tools.jlink.plugin.Plugin;
 34 import jdk.tools.jlink.plugin.ResourcePool;
 35 import jdk.tools.jlink.plugin.ResourcePoolBuilder;
 36 
 37 /**
 38  * Combined debug stripping plugin: Java debug attributes and native debug
 39  * symbols.
 40  *
 41  */
 42 public final class DefaultStripDebugPlugin implements Plugin {
 43 
 44     public static final String NAME = "strip-debug";
 45     private static final String STRIP_NATIVE_DEBUG_PLUGIN = "strip-native-debug-symbols";
 46     private static final String EXCLUDE_DEBUGINFO = "exclude-debuginfo-files";
 47 
 48     private final Plugin javaStripPlugin;
 49     private final NativePluginFactory stripNativePluginFactory;
 50 
 51     public DefaultStripDebugPlugin() {
 52         this(new StripJavaDebugAttributesPlugin(),
 53              new DefaultNativePluginFactory());
 54     }
 55 
 56     public DefaultStripDebugPlugin(Plugin javaStripPlugin,
 57                                    NativePluginFactory nativeStripPluginFact) {
 58         this.javaStripPlugin = javaStripPlugin;
 59         this.stripNativePluginFactory = nativeStripPluginFact;
 60     }
 61 
 62     @Override
 63     public String getName() {
 64         return NAME;
 65     }
 66 
 67     @Override
 68     public String getDescription() {
 69         return PluginsResourceBundle.getDescription(NAME);
 70     }
 71 
 72     @Override
 73     public ResourcePool transform(ResourcePool in, ResourcePoolBuilder out) {
 74         Plugin stripNativePlugin = stripNativePluginFactory.create();
 75         if (stripNativePlugin != null) {
 76             Map<String, String> stripNativeConfig = Map.of(
 77                                      STRIP_NATIVE_DEBUG_PLUGIN, EXCLUDE_DEBUGINFO);
 78             stripNativePlugin.configure(stripNativeConfig);
 79             ResourcePoolManager outRes =
 80                                  new ResourcePoolManager(in.byteOrder(),
 81                                                         ((ResourcePoolImpl)in).getStringTable());
 82             ResourcePool strippedJava = javaStripPlugin.transform(in,
 83                                                                   outRes.resourcePoolBuilder());
 84             return stripNativePlugin.transform(strippedJava, out);
 85         } else {
 86             return javaStripPlugin.transform(in, out);
 87         }
 88     }
 89 
 90     public interface NativePluginFactory {
 91         Plugin create();
 92     }
 93 
 94     private static class DefaultNativePluginFactory implements NativePluginFactory {
 95 
 96         @Override
 97         public Plugin create() {
 98             return PluginRepository.getPlugin(STRIP_NATIVE_DEBUG_PLUGIN,
 99                                               ModuleLayer.boot());
100         }
101 
102     }
103 
104 }
