26 package jdk.tools.jlink.internal.plugins;
27 
28 import java.io.*;
29 import java.nio.charset.*;
30 import java.util.*;
31 import java.util.function.*;
32 import java.util.stream.*;
33 import jdk.tools.jlink.plugin.*;
34 
35 /**
36  * Base plugin to add a resource
37  */
38 abstract class AddResourcePlugin implements Plugin {
39 
40     private final String name;
41     private final String path;
42     private String value;
43 
44     protected AddResourcePlugin(String n, String p) {
45         name = n;
46         path = p;
47     }
48 
49     @Override
50     public String getName() {
51         return name;
52     }
53 
54     @Override
55     public String getDescription() {
56         return PluginsResourceBundle.getDescription(name);
57     }
58 
59     @Override
60     public Category getType() {
61         return Category.ADDER;
62     }
63 
64     @Override
65     public boolean hasArguments() {
66         return true;
67     }
68 
69     @Override
70     public boolean hasRawArgument() {
71         return true;
72     }
73 
74     @Override
75     public String getArgumentsDescription() {
76        return PluginsResourceBundle.getArgument(name);
77     }
78 
79     @Override
80     public void configure(Map<String, String> config) {
81         var v = config.get(name);
82         if (v == null)
83             throw new AssertionError();
84         value = v;
85     }
86 
87     @Override
88     public ResourcePool transform(ResourcePool in, ResourcePoolBuilder out) {
89         in.transformAndCopy(Function.identity(), out);
90         out.add(ResourcePoolEntry.create(path,
91                                          value.getBytes(StandardCharsets.UTF_8)));
92         return out.build();
93     }
94 
95 }
