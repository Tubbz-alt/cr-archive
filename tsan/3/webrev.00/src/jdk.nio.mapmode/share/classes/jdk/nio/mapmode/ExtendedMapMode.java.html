26 package jdk.nio.mapmode;
27 
28 import java.nio.channels.FileChannel.MapMode;
29 
30 /**
31  * JDK-specific map modes.
32  *
33  * @since 14
34  * @see java.nio.channels.FileChannel#map
35  */
36 public class ExtendedMapMode {
37     private ExtendedMapMode() { }
38 
39     /**
40      * File mapping mode for a read-only mapping of a file backed by
41      * non-volatile RAM.
42      *
43      * <p> The {@linkplain FileChannel#map map} method throws
44      * {@linkplain UnsupportedOperationException} when this map mode
45      * is used on an implementation that does not support it.
46      *
47      * @implNote On Linux, the {@code MAP_SYNC} and {@code
48      * MAP_SHARED_VALIDATE} flags are specified to {@code mmap} when
49      * mapping the file into memory.
50      */
51     public static final MapMode READ_ONLY_SYNC = jdk.internal.misc.ExtendedMapMode.READ_ONLY_SYNC;
52 
53     /**
54      * File mapping mode for a read-write mapping of a file backed by
55      * non-volatile RAM. {@linkplain MappedByteBufefr#force force}
56      * operations on a buffer created with this mode will be performed
57      * using cache line writeback rather than proceeding via a file
58      * device flush.
59      *
60      * <p> The {@linkplain FileChannel#map map} method throws
61      * {@linkplain UnsupportedOperationException} when this map mode
62      * is used on an implementation that does not support it.
63      *
64      * @implNote On Linux, the {@code MAP_SYNC} and {@code
65      * MAP_SHARED_VALIDATE} flags are specified to {@code mmap} when
66      * mapping the file into memory.
67      */
68     public static final MapMode READ_WRITE_SYNC = jdk.internal.misc.ExtendedMapMode.READ_WRITE_SYNC;
69 
70 }
