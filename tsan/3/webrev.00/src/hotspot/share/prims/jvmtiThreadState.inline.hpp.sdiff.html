 87   JvmtiThreadState *state = thread->jvmti_thread_state();
 88   if (state == NULL) {
 89     MutexLocker mu(JvmtiThreadState_lock);
 90     // check again with the lock held

 93     // Check possible safepoint even if state is non-null.
