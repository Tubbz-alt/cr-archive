35 // Returns the population count of x, i.e., the number of bits set in x.
36 //
37 // Adapted from Hacker's Delight, 2nd Edition, Figure 5-2 and the text that
38 // follows.
39 //
40 // Ideally this should be dispatched per platform to use optimized
41 // instructions when available, such as POPCNT on modern x86/AMD. Our builds
42 // still target and support older architectures that might lack support for
43 // these. For example, with current build configurations, __builtin_popcount(x)
44 // generate a call to a similar but slower 64-bit version when calling with
45 // a 32-bit integer type.
46 template <typename T>
47 inline unsigned population_count(T x) {
48   STATIC_ASSERT(BitsPerWord <= 128);
49   STATIC_ASSERT(BitsPerByte == 8);
50   STATIC_ASSERT(IsIntegral<T>::value);
51   STATIC_ASSERT(!IsSigned<T>::value);
52   // We need to take care with implicit integer promotion when dealing with
53   // integers < 32-bit. We chose to do this by explicitly widening constants
54   // to unsigned
55   typedef typename Conditional<(sizeof(T) < sizeof(unsigned)), unsigned, T>::type P;
56   const T all = ~T(0);           // 0xFF..FF
57   const P fives = all/3;         // 0x55..55
58   const P threes = (all/15) * 3; // 0x33..33
59   const P z_ones = all/255;      // 0x0101..01
60   const P z_effs = z_ones * 15;  // 0x0F0F..0F
61   P r = x;
62   r -= ((r >> 1) & fives);
63   r = (r & threes) + ((r >> 2) & threes);
64   r = ((r + (r >> 4)) & z_effs) * z_ones;
65   // The preceeding multiply by z_ones is the only place where the intermediate
66   // calculations can exceed the range of T. We need to discard any such excess
67   // before the right-shift, hence the conversion back to T.
68   return static_cast<T>(r) >> (((sizeof(T) - 1) * BitsPerByte));
69 }
