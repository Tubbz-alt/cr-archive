 45 // assertions
 46 #ifndef ASSERT
 47 #define vmassert(p, ...)
 48 #else
 49 // Note: message says "assert" rather than "vmassert" for backward
 50 // compatibility with tools that parse/match the message text.
 51 // Note: The signature is vmassert(p, format, ...), but the solaris
 52 // compiler can't handle an empty ellipsis in a macro without a warning.
 53 #define vmassert(p, ...)                                                       \
 54 do {                                                                           \
 55   if (!(p)) {                                                                  \
 56     TOUCH_ASSERT_POISON;                                                       \
 57     report_vm_error(__FILE__, __LINE__, "assert(" #p ") failed", __VA_ARGS__); \
 58     BREAKPOINT;                                                                \
 59   }                                                                            \
 60 } while (0)
 61 #endif
 62 
 63 // For backward compatibility.
 64 #define assert(p, ...) vmassert(p, __VA_ARGS__)
 65 
 66 #ifndef ASSERT
 67 #define vmassert_status(p, status, msg)
 68 #else
 69 // This version of vmassert is for use with checking return status from
 70 // library calls that return actual error values eg. EINVAL,
 71 // ENOMEM etc, rather than returning -1 and setting errno.
 72 // When the status is not what is expected it is very useful to know
 73 // what status was actually returned, so we pass the status variable as
 74 // an extra arg and use strerror to convert it to a meaningful string
 75 // like "Invalid argument", "out of memory" etc
 76 #define vmassert_status(p, status, msg) \
 77 do {                                                                           \
 78   if (!(p)) {                                                                  \
 79     TOUCH_ASSERT_POISON;                                                       \
 80     report_vm_status_error(__FILE__, __LINE__, "assert(" #p ") failed",        \
 81                            status, msg);                                       \
 82     BREAKPOINT;                                                                \
 83   }                                                                            \
 84 } while (0)
 85 #endif
 86 
 87 // For backward compatibility.
 88 #define assert_status(p, status, msg) vmassert_status(p, status, msg)
 89 
 90 // guarantee is like vmassert except it's always executed -- use it for
 91 // cheap tests that catch errors that would otherwise be hard to find.
 92 // guarantee is also used for Verify options.
 93 #define guarantee(p, ...)                                                         \
 94 do {                                                                              \
 95   if (!(p)) {                                                                     \
 96     TOUCH_ASSERT_POISON;                                                          \
 97     report_vm_error(__FILE__, __LINE__, "guarantee(" #p ") failed", __VA_ARGS__); \
 98     BREAKPOINT;                                                                   \
 99   }                                                                               \
100 } while (0)

174 // Compile-time asserts.  Cond must be a compile-time constant expression that
175 // is convertible to bool.  STATIC_ASSERT() can be used anywhere a declaration
176 // may appear.
177 //
178 // Implementation Note: STATIC_ASSERT_FAILURE<true> provides a value member
179 // rather than type member that could be used directly in the typedef, because
180 // a type member would require conditional use of "typename", depending on
181 // whether Cond is dependent or not.  The use of a value member leads to the
182 // use of an array type.
183 
184 template<bool x> struct STATIC_ASSERT_FAILURE;
185 template<> struct STATIC_ASSERT_FAILURE<true> { enum { value = 1 }; };
186 
187 #define STATIC_ASSERT(Cond) \
188   typedef char PASTE_TOKENS(STATIC_ASSERT_DUMMY_TYPE_, __LINE__)[ \
189     STATIC_ASSERT_FAILURE< (Cond) >::value ]
