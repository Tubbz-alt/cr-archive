25 #include "precompiled.hpp"
26 #include "memory/allocation.inline.hpp"
27 #include "runtime/os.hpp"
28 #include "utilities/classpathStream.hpp"

30 const char* ClasspathStream::get_next() {
31   while (_class_path[_end] != '\0' && _class_path[_end] != os::path_separator()[0]) {
32     _end++;
33   }
34   int path_len = _end - _start;
35   char* path = NEW_RESOURCE_ARRAY(char, path_len + 1);
36   strncpy(path, &_class_path[_start], path_len);
37   path[path_len] = '\0';
38 
39   while (_class_path[_end] == os::path_separator()[0]) {
40     _end++;
41   }
42   _start = _end;
43   return path;
44 }
