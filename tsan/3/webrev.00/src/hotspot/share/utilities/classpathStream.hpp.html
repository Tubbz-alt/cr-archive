25 #ifndef SHARE_UTILITIES_CLASSPATHSTREAM_HPP
26 #define SHARE_UTILITIES_CLASSPATHSTREAM_HPP
27 
28 class ClasspathStream : public StackObj {
29   const char* _class_path;
30   int _len;
31   int _start;
32   int _end;
33 
34 public:
35   ClasspathStream(const char* class_path) {
36     _class_path = class_path;
37     _len = (int)strlen(class_path);
38     _start = 0;
39     _end = 0;
40   }
41 
42   bool has_next() {
43     return _start < _len;
44   }
45 
46   const char* get_next();
47 };
48 
49 #endif // SHARE_UTILITIES_CLASSPATHSTREAM_HPP
