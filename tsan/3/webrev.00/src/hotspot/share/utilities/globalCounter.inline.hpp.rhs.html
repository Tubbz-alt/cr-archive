25 #ifndef SHARE_UTILITIES_GLOBALCOUNTER_INLINE_HPP
26 #define SHARE_UTILITIES_GLOBALCOUNTER_INLINE_HPP
27 
28 #include "runtime/atomic.hpp"

29 #include "runtime/thread.inline.hpp"
30 #include "utilities/globalCounter.hpp"
31 
32 inline GlobalCounter::CSContext
33 GlobalCounter::critical_section_begin(Thread *thread) {
34   assert(thread == Thread::current(), "must be current thread");
35   uintx old_cnt = Atomic::load(thread->get_rcu_counter());
36   // Retain the old counter value if already active, e.g. nested.
37   // Otherwise, set the counter to the current version + active bit.
38   uintx new_cnt = old_cnt;
39   if ((new_cnt & COUNTER_ACTIVE) == 0) {
40     new_cnt = Atomic::load(&_global_counter._counter) | COUNTER_ACTIVE;
41   }
42   Atomic::release_store_fence(thread->get_rcu_counter(), new_cnt);
43   return static_cast<CSContext>(old_cnt);
44 }
45 
46 inline void
47 GlobalCounter::critical_section_end(Thread *thread, CSContext context) {
48   assert(thread == Thread::current(), "must be current thread");
49   assert((*thread->get_rcu_counter() & COUNTER_ACTIVE) == COUNTER_ACTIVE, "must be in critical section");
50   // Restore the counter value from before the associated begin.
51   Atomic::release_store(thread->get_rcu_counter(),
52                         static_cast<uintx>(context));
53 }
54 
55 class GlobalCounter::CriticalSection {
56  private:
57   Thread* _thread;
58   CSContext _context;
59  public:
60   inline CriticalSection(Thread* thread) :
61     _thread(thread),
62     _context(GlobalCounter::critical_section_begin(_thread))
63   {}
64 
65   inline  ~CriticalSection() {
66     GlobalCounter::critical_section_end(_thread, _context);
67   }
68 };

70 #endif // SHARE_UTILITIES_GLOBALCOUNTER_INLINE_HPP
