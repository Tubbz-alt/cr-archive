31 // Returns the population count of x, i.e., the number of bits set in x.
32 //
33 // Adapted from Hacker's Delight, 2nd Edition, Figure 5-2.
34 //
35 // Ideally this should be dispatched per platform to use optimized
36 // instructions when available, such as POPCNT on modern x86/AMD. Our builds
37 // still target and support older architectures that might lack support for
38 // these, however. For example, with current build configurations,
39 // __builtin_popcount(x) would generate a call to a similar but slower 64-bit
40 // version of this 32-bit implementation.
41 static uint32_t population_count(uint32_t x) {
42   x -= ((x >> 1) & 0x55555555);
43   x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
44   return (((x + (x >> 4)) & 0x0F0F0F0F) * 0x01010101) >> 24;
45 }
