33 inline ClassLoaderData *ClassLoaderDataGraph::find_or_create(Handle loader) {
34   guarantee(loader() != NULL && oopDesc::is_oop(loader()), "Loader must be oop");
35   // Gets the class loader data out of the java/lang/ClassLoader object, if non-null
36   // it's already in the loader_data, so no need to add
37   ClassLoaderData* loader_data = java_lang_ClassLoader::loader_data_acquire(loader());
38   if (loader_data) {
39      return loader_data;
40   }
41   return ClassLoaderDataGraph::add(loader, false);
42 }

70 bool ClassLoaderDataGraph::should_clean_metaspaces_and_reset() {
71   // Only clean metaspaces after full GC.

73 #if INCLUDE_JVMTI
