35 void oop::register_oop() {
36   assert (CheckUnhandledOops, "should only call when CheckUnhandledOops");
37   if (!Universe::is_fully_initialized()) return;
38   // This gets expensive, which is why checking unhandled oops is on a switch.
39   Thread* t = Thread::current_or_null();
40   if (t != NULL && t->is_Java_thread()) {
41      frame fr = os::current_frame();
42      // This points to the oop creator, I guess current frame points to caller
43      assert (fr.pc(), "should point to a vm frame");
44      t->unhandled_oops()->register_unhandled_oop(this, fr.pc());
45   }
46 }
47 
48 void oop::unregister_oop() {
49   assert (CheckUnhandledOops, "should only call when CheckUnhandledOops");
50   if (!Universe::is_fully_initialized()) return;
51   // This gets expensive, which is why checking unhandled oops is on a switch.
52   Thread* t = Thread::current_or_null();
53   if (t != NULL && t->is_Java_thread()) {
54     t->unhandled_oops()->unregister_unhandled_oop(this);
55   }
56 }

58 bool oop::operator==(const oop o) const {
59   assert(BarrierSet::barrier_set()->oop_equals_operator_allowed(), "Not allowed");
60   return obj() == o.obj();
61 }
62 
63 bool oop::operator!=(const volatile oop o) const {
64   assert(BarrierSet::barrier_set()->oop_equals_operator_allowed(), "Not allowed");
65   return obj() != o.obj();
66 }
