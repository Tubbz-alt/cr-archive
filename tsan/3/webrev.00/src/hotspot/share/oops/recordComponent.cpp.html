35 RecordComponent* RecordComponent::allocate(ClassLoaderData* loader_data,
36                                            u2 name_index, u2 descriptor_index,
37                                            u2 attributes_count,
38                                            u2 generic_signature_index,
39                                            AnnotationArray* annotations,
40                                            AnnotationArray* type_annotations, TRAPS) {
41   return new (loader_data, size(), MetaspaceObj::RecordComponentType, THREAD)
42          RecordComponent(name_index, descriptor_index, attributes_count,
43                          generic_signature_index, annotations, type_annotations);
44 }
45 
46 void RecordComponent::deallocate_contents(ClassLoaderData* loader_data) {
47   if (annotations() != NULL) {
48     MetadataFactory::free_array<u1>(loader_data, annotations());
49   }
50   if (type_annotations() != NULL) {
51     MetadataFactory::free_array<u1>(loader_data, type_annotations());
52   }
53 }
54 
55 void RecordComponent::metaspace_pointers_do(MetaspaceClosure* it) {
56   log_trace(cds)("Iter(RecordComponent): %p", this);
57   it->push(&_annotations);
58   it->push(&_type_annotations);
59 }
60 
61 void RecordComponent::print_value_on(outputStream* st) const {
62   st->print("RecordComponent(" INTPTR_FORMAT ")", p2i(this));
63 }
64 
65 #ifndef PRODUCT
66 void RecordComponent::print_on(outputStream* st) const {
67   st->print("name_index: %d", _name_index);
68   st->print(" - descriptor_index: %d", _descriptor_index);
69   st->print(" - attributes_count: %d", _attributes_count);
70   if (_generic_signature_index != 0) {
71     st->print(" - generic_signature_index: %d", _generic_signature_index);
72   }
73   st->cr();
74   if (_annotations != NULL) {
75     st->print_cr("record component annotations");
76     _annotations->print_value_on(st);
77   }
78   if (_type_annotations != NULL) {
79     st->print_cr("record component type annotations");
80     _type_annotations->print_value_on(st);
81   }
82 }
83 #endif // PRODUCT
