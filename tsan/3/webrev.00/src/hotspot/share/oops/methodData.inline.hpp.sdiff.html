48 inline void RetData::release_set_bci(uint row, int bci) {
49   assert((uint)row < row_limit(), "oob");
50   // 'release' when setting the bci acts as a valid flag for other
51   // threads wrt bci_count and bci_displacement.
52   release_set_int_at(bci0_offset + row * ret_row_cell_count, bci);
