25 #ifndef SHARE_OOPS_FIELDSTREAMS_INLINE_HPP
26 #define SHARE_OOPS_FIELDSTREAMS_INLINE_HPP
27 
28 #include "oops/fieldStreams.hpp"
29 #include "runtime/thread.inline.hpp"
30 
31 FieldStreamBase::FieldStreamBase(Array<u2>* fields, ConstantPool* constants, int start, int limit) : _fields(fields),
32          _constants(constantPoolHandle(Thread::current(), constants)), _index(start) {
33   _index = start;
34   int num_fields = init_generic_signature_start_slot();
35   if (limit < start) {
36     _limit = num_fields;
37   } else {
38     _limit = limit;
39   }
40 }
41 
42 FieldStreamBase::FieldStreamBase(Array<u2>* fields, ConstantPool* constants) : _fields(fields),
43          _constants(constantPoolHandle(Thread::current(), constants)), _index(0) {
44   _limit = init_generic_signature_start_slot();
45 }
46 
47 FieldStreamBase::FieldStreamBase(InstanceKlass* klass) : _fields(klass->fields()),
48          _constants(constantPoolHandle(Thread::current(), klass->constants())), _index(0),
49          _limit(klass->java_fields_count()) {
50   init_generic_signature_start_slot();
51   assert(klass == field_holder(), "");
52 }
53 
54 #endif // SHARE_OOPS_FIELDSTREAMS_INLINE_HPP
