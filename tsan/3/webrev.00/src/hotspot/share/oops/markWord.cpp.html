25 #include "precompiled.hpp"
26 #include "oops/markWord.hpp"
27 #include "runtime/thread.inline.hpp"
28 #include "runtime/objectMonitor.hpp"
29 
30 void markWord::print_on(outputStream* st) const {
31   if (is_marked()) {  // last bits = 11
32     st->print(" marked(" INTPTR_FORMAT ")", value());
33   } else if (has_monitor()) {  // last bits = 10
34     // have to check has_monitor() before is_locked()
35     st->print(" monitor(" INTPTR_FORMAT ")=", value());
36     ObjectMonitor* mon = monitor();
37     if (mon == NULL) {
38       st->print("NULL (this should never be seen!)");
39     } else {
40       mon->print_on(st);
41     }
42   } else if (is_locked()) {  // last bits != 01 => 00
43     // thin locked
44     st->print(" locked(" INTPTR_FORMAT ")", value());
45   } else {
46     st->print(" mark(");
47     // Biased bit is 3rd rightmost bit
48     if (is_neutral()) {   // last bits = 001
49       st->print("is_neutral");
50       if (has_no_hash()) {
51         st->print(" no_hash");
52       } else {
53         st->print(" hash=" INTPTR_FORMAT, hash());
54       }
55     } else if (has_bias_pattern()) {  // last bits = 101
56       st->print("is_biased");
57       JavaThread* jt = biased_locker();
58       st->print(" biased_locker=" INTPTR_FORMAT " epoch=%d", p2i(jt), bias_epoch());
59     } else {
60       st->print("??");
61     }
62     st->print(" age=%d)", age());
63   }
64 }
