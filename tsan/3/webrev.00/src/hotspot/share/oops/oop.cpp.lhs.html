 91 intptr_t oopDesc::slow_identity_hash() {
 92   // slow case; we have to acquire the micro lock in order to locate the header
 93   Thread* THREAD = Thread::current();
 94   ResetNoHandleMark rnm; // Might be called from LEAF/QUICK ENTRY
 95   HandleMark hm(THREAD);
 96   Handle object(THREAD, this);
 97   return ObjectSynchronizer::identity_hash_value_for(object);
 98 }

100 // used only for asserts and guarantees
101 bool oopDesc::is_oop(oop obj, bool ignore_mark_word) {
102   if (!Universe::heap()->is_oop(obj)) {
103     return false;
104   }
105 
106   // Header verification: the mark is typically non-NULL. If we're
107   // at a safepoint, it must not be null.
108   // Outside of a safepoint, the header could be changing (for example,
109   // another thread could be inflating a lock on this object).
110   if (ignore_mark_word) {
111     return true;
112   }
113   if (obj->mark_raw() != NULL) {
114     return true;
115   }
116   return !SafepointSynchronize::is_at_safepoint();
117 }
118 
119 // used only for asserts and guarantees
120 bool oopDesc::is_oop_or_null(oop obj, bool ignore_mark_word) {
121   return obj == NULL ? true : is_oop(obj, ignore_mark_word);
122 }

159 oop oopDesc::decode_oop_raw(narrowOop narrow_oop) {
160   return (oop)(void*)( (uintptr_t)Universe::narrow_oop_base() +
161                       ((uintptr_t)narrow_oop << Universe::narrow_oop_shift()));
162 }
163 
164 void* oopDesc::load_klass_raw(oop obj) {
165   if (UseCompressedClassPointers) {
166     narrowKlass narrow_klass = *(obj->compressed_klass_addr());
167     if (narrow_klass == 0) return NULL;
168     return (void*)Klass::decode_klass_raw(narrow_klass);
169   } else {
170     return *(void**)(obj->klass_addr());
171   }
172 }
173 
174 void* oopDesc::load_oop_raw(oop obj, int offset) {
175   uintptr_t addr = (uintptr_t)(void*)obj + (uint)offset;
176   if (UseCompressedOops) {
177     narrowOop narrow_oop = *(narrowOop*)addr;
178     if (narrow_oop == 0) return NULL;
179     return (void*)decode_oop_raw(narrow_oop);
180   } else {
181     return *(void**)addr;
182   }
183 }

185 bool oopDesc::is_valid(oop obj) {
186   if (!is_object_aligned(obj)) return false;
187   if ((size_t)(oopDesc*)obj < os::min_page_size()) return false;
188 
189   // We need at least the mark and the klass word in the committed region.
190   if (!os::is_readable_range(obj, (oopDesc*)obj + 1)) return false;
191   if (!Universe::heap()->is_in(obj)) return false;
192 
193   Klass* k = (Klass*)load_klass_raw(obj);
194 
195   if (!os::is_readable_range(k, k + 1)) return false;
196   return MetaspaceUtils::is_range_in_committed(k, k + 1);
197 }

199 oop oopDesc::oop_or_null(address addr) {
200   if (is_valid(oop(addr))) {
201     // We were just given an oop directly.
202     return oop(addr);
203   }
204 
205   // Try to find addr using block_start.
206   HeapWord* p = Universe::heap()->block_start(addr);
207   if (p != NULL && Universe::heap()->block_is_obj(p)) {
208     if (!is_valid(oop(p))) return NULL;
209     return oop(p);
210   }
211 
212   // If we can't find it it just may mean that heap wasn't parsable.
213   return NULL;
214 }
