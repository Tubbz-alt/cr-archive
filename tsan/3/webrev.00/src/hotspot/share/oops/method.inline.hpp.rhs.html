25 #ifndef SHARE_OOPS_METHOD_INLINE_HPP
26 #define SHARE_OOPS_METHOD_INLINE_HPP

28 #include "oops/method.hpp"
29 #include "runtime/atomic.hpp"

31 inline address Method::from_compiled_entry() const {
32   return Atomic::load_acquire(&_from_compiled_entry);
33 }
34 
35 inline address Method::from_interpreted_entry() const {
36   return Atomic::load_acquire(&_from_interpreted_entry);
37 }
38 
39 inline void Method::set_method_data(MethodData* data) {
40   // The store into method must be released. On platforms without
41   // total store order (TSO) the reference may become visible before
42   // the initialization of data otherwise.
43   Atomic::release_store(&_method_data, data);
44 }
45 
46 inline CompiledMethod* volatile Method::code() const {
47   assert( check_code(), "" );
48   return Atomic::load_acquire(&_code);
49 }
50 
51 // Write (bci, line number) pair to stream
52 inline void CompressedLineNumberWriteStream::write_pair_regular(int bci_delta, int line_delta) {
53   // bci and line number does not compress into single byte.
54   // Write out escape character and use regular compression for bci and line number.
55   write_byte((jubyte)0xFF);
56   write_signed_int(bci_delta);
57   write_signed_int(line_delta);
58 }
59 
60 inline void CompressedLineNumberWriteStream::write_pair_inline(int bci, int line) {
61   int bci_delta = bci - _bci;
62   int line_delta = line - _line;
63   _bci = bci;
64   _line = line;
65   // Skip (0,0) deltas - they do not add information and conflict with terminator.
66   if (bci_delta == 0 && line_delta == 0) return;
67   // Check if bci is 5-bit and line number 3-bit unsigned.
68   if (((bci_delta & ~0x1F) == 0) && ((line_delta & ~0x7) == 0)) {
69     // Compress into single byte.
70     jubyte value = ((jubyte) bci_delta << 3) | (jubyte) line_delta;
71     // Check that value doesn't match escape character.
72     if (value != 0xFF) {
73       write_byte(value);
74       return;
75     }
76   }
77   write_pair_regular(bci_delta, line_delta);
78 }
79 
80 inline void CompressedLineNumberWriteStream::write_pair(int bci, int line) {
81   write_pair_inline(bci, line);
82 }
83 
84 inline bool Method::has_compiled_code() const { return code() != NULL; }
