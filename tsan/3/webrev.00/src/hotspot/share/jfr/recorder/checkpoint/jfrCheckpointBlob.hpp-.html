25 #ifndef SHARE_JFR_RECORDER_CHECKPOINT_JFRCHECKPOINTBLOB_HPP
26 #define SHARE_JFR_RECORDER_CHECKPOINT_JFRCHECKPOINTBLOB_HPP
27 
28 #include "jfr/utilities/jfrAllocation.hpp"
29 #include "jfr/utilities/jfrRefCountPointer.hpp"
30 
31 class JfrCheckpointBlob;
32 class JfrCheckpointWriter;
33 
34 typedef RefCountPointer<JfrCheckpointBlob, MultiThreadedRefCounter> JfrCheckpointBlobReference;
35 typedef RefCountHandle<JfrCheckpointBlobReference> JfrCheckpointBlobHandle;
36 
37 class JfrCheckpointBlob : public JfrCHeapObj {
38   template <typename, typename>
39   friend class RefCountPointer;
40  private:
41   const u1* _checkpoint;
42   const size_t _size;
43   JfrCheckpointBlobHandle _next;
44   mutable bool _written;
45 
46   JfrCheckpointBlob(const u1* checkpoint, size_t size);
47   ~JfrCheckpointBlob();
48   const JfrCheckpointBlobHandle& next() const;
49   void write_this(JfrCheckpointWriter& writer) const;
50 
51  public:
52   void write(JfrCheckpointWriter& writer) const;
53   void exclusive_write(JfrCheckpointWriter& writer) const;
54   void reset_write_state() const;
55   void set_next(const JfrCheckpointBlobHandle& ref);
56   static JfrCheckpointBlobHandle make(const u1* checkpoint, size_t size);
57 };
58 
59 #endif // SHARE_JFR_RECORDER_CHECKPOINT_JFRCHECKPOINTBLOB_HPP
