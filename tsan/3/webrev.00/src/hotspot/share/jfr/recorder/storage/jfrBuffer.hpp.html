 25 #ifndef SHARE_JFR_RECORDER_STORAGE_JFRBUFFER_HPP
 26 #define SHARE_JFR_RECORDER_STORAGE_JFRBUFFER_HPP
 27 
 28 #include "memory/allocation.hpp"
 29 #include "runtime/atomic.hpp"
 30 
 31 //
 32 // Represents a piece of committed memory.
 33 //
 34 // Use acquire() and/or try_acquire() for exclusive access
 35 // to the buffer (cas identity). This is a precondition
 36 // for attempting stores.
 37 //
 38 // u1* _pos <-- last committed position
 39 // u1* _top <-- next unflushed position
 40 //
 41 // Stores must uphold transactional semantics. This means that _pos
 42 // must be updated only after all intended stores have completed already.
 43 // The relation between _pos and _top must hold atomically,
 44 // e.g. the delta must always be fully parsable.
 45 // _top can move concurrently by other threads but is always <= _pos.
 46 //
 47 // Memory ordering:
 48 //
 49 //  Method                 Owner thread             Other threads
 50 //  ---------------------------------------------------------------
 51 //  acquire()              Acquire semantics (cas)  Acquire semantics (cas)
 52 //  try_acquire()          Acquire semantics (cas)  Acquire semantics (cas)
 53 //  release()              Release semantics        Release semantics
 54 //  pos()                  Plain load               Acquire semantics needed at call sites
 55 //  set_pos()              Release semantics        N/A
 56 //  top()                  Acquire semantics        Acquire semantics
 57 //  set_top()              Release semantics        Release semantics
 58 //  acquire_crit_sec_top() Acquire semantics (cas)  Acquire semantics (cas)
 59 //  release_crit_sec_top() Release semantics        Release semantics
 60 //
 61 
 62 class JfrBuffer {
 63  private:
 64   JfrBuffer* _next;
 65   JfrBuffer* _prev;
 66   const void* _identity;
 67   u1* _pos;
 68   mutable const u1* _top;
 69   u2 _flags;
 70   u2 _header_size;
 71   u4 _size;
 72 
 73   const u1* stable_top() const;
 74 
 75  public:
 76   JfrBuffer();
 77   bool initialize(size_t header_size, size_t size);
 78   void reinitialize(bool exclusion = false);
 79 
 80   JfrBuffer* next() const {
 81     return _next;
 82   }
 83 
 84   JfrBuffer* prev() const {
 85     return _prev;
 86   }
 87 
 88   void set_next(JfrBuffer* next) {
 89     _next = next;
 90   }
 91 
 92   void set_prev(JfrBuffer* prev) {
 93     _prev = prev;
 94   }
 95 
 96   const u1* start() const {
 97     return ((const u1*)this) + _header_size;
 98   }
 99 
100   u1* start() {
101     return ((u1*)this) + _header_size;
102   }
103 
104   const u1* end() const {
105     return start() + size();
106   }
107 
108   // If pos() methods are invoked by a thread that is not the owner,
109   // then acquire semantics must be ensured at the call site.
110   const u1* pos() const {
111     return _pos;
112   }
113 
114   u1* pos() {
115     return _pos;
116   }
117 
118   u1** pos_address() {
119     return &_pos;
120   }
121 
122   void set_pos(u1* new_pos) {
123     assert(new_pos <= end(), "invariant");
124     Atomic::release_store(&_pos, new_pos);
125   }
126 
127   void set_pos(size_t size) {
128     set_pos(pos() + size);
129   }
130 
131   const u1* top() const;
132   void set_top(const u1* new_top);
133 
134   // mutual exclusion
135   const u1* acquire_critical_section_top() const;
136   void release_critical_section_top(const u1* new_top);
137 
138   size_t size() const {
139     return _size * BytesPerWord;
140   }
141 
142   size_t total_size() const {
143     return _header_size + size();
144   }
145 
146   size_t free_size() const {
147     return end() - Atomic::load_acquire(&_pos);
148   }
149 
150   size_t unflushed_size() const;
151 
152   bool empty() const {
153     return Atomic::load_acquire(&_pos) == start();
154   }
155 
156   const void* identity() const {
157     return Atomic::load_acquire(&_identity);
158   }
159 
160   void acquire(const void* id);
161   bool try_acquire(const void* id);
162   bool acquired_by(const void* id) const;
163   bool acquired_by_self() const;
164   void release();
165 
166   size_t discard();
167   void move(JfrBuffer* const to, size_t size);
168 
169   bool transient() const;
170   void set_transient();
171   void clear_transient();
172 
173   bool lease() const;
174   void set_lease();
175   void clear_lease();
176 
177   bool retired() const;
178   void set_retired();
179   void clear_retired();
180 
181   bool excluded() const;
182   void set_excluded();
183   void clear_excluded();
184 };
185 
186 class JfrAgeNode : public JfrBuffer {
187  private:
188   JfrBuffer* _retired;
189  public:
190   JfrAgeNode() : _retired(NULL) {}
191   void set_retired_buffer(JfrBuffer* retired) {
192     _retired = retired;
193   }
194   JfrBuffer* retired_buffer() const {
195     return _retired;
196   }
197 };
