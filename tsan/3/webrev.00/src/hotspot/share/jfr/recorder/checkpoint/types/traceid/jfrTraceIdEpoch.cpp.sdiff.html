30 // Alternating epochs on each rotation allow for concurrent tagging.
31 // The regular epoch shift happens only during a safepoint.
32 // The fence is there only for the emergency dump case which happens outside of safepoint.
33 bool JfrTraceIdEpoch::_epoch_state = false;
34 void JfrTraceIdEpoch::shift_epoch() {











35   _epoch_state = !_epoch_state;
36   if (!SafepointSynchronize::is_at_safepoint()) {
37     OrderAccess::fence();
38   }
39 }

29 bool JfrTraceIdEpoch::_epoch_state = false;
30 bool JfrTraceIdEpoch::_synchronizing = false;
31 volatile bool JfrTraceIdEpoch::_changed_tag_state = false;
32 
33 void JfrTraceIdEpoch::begin_epoch_shift() {
34   assert(SafepointSynchronize::is_at_safepoint(), "invariant");
35   _synchronizing = true;
36   OrderAccess::fence();
37 }
38 
39 void JfrTraceIdEpoch::end_epoch_shift() {
40   assert(SafepointSynchronize::is_at_safepoint(), "invariant");
41   assert(_synchronizing, "invariant");
42   _epoch_state = !_epoch_state;
43   OrderAccess::storestore();
44   _synchronizing = false;

45 }
