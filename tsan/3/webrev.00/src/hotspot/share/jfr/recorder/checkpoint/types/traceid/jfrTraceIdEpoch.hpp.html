 25 #ifndef SHARE_JFR_RECORDER_CHECKPOINT_TYPES_TRACEID_JFRTRACEIDEPOCH_HPP
 26 #define SHARE_JFR_RECORDER_CHECKPOINT_TYPES_TRACEID_JFRTRACEIDEPOCH_HPP
 27 
 28 #include "jfr/utilities/jfrTypes.hpp"
 29 #include "memory/allocation.hpp"
 30 #include "runtime/atomic.hpp"
 31 
 32 #define USED_BIT                             1
 33 #define METHOD_USED_BIT                      (USED_BIT << 2)
 34 #define EPOCH_1_SHIFT                        0
 35 #define EPOCH_2_SHIFT                        1
 36 #define USED_EPOCH_1_BIT                     (USED_BIT << EPOCH_1_SHIFT)
 37 #define USED_EPOCH_2_BIT                     (USED_BIT << EPOCH_2_SHIFT)
 38 #define METHOD_USED_EPOCH_1_BIT              (METHOD_USED_BIT << EPOCH_1_SHIFT)
 39 #define METHOD_USED_EPOCH_2_BIT              (METHOD_USED_BIT << EPOCH_2_SHIFT)
 40 #define METHOD_AND_CLASS_IN_USE_BITS         (METHOD_USED_BIT | USED_BIT)
 41 #define METHOD_AND_CLASS_IN_USE_EPOCH_1_BITS (METHOD_AND_CLASS_IN_USE_BITS << EPOCH_1_SHIFT)
 42 #define METHOD_AND_CLASS_IN_USE_EPOCH_2_BITS (METHOD_AND_CLASS_IN_USE_BITS << EPOCH_2_SHIFT)
 43 
 44  // Epoch alternation on each rotation allow for concurrent tagging.
 45  // The epoch shift happens only during a safepoint.
 46  //
 47  // _synchronizing is a transition state, the purpose of which is to
 48  // have JavaThreads that run _thread_in_native (i.e. Compiler threads)
 49  // respect the current epoch shift in-progress during a safepoint.
 50  //
 51  // _changed_tag_state == true signals an incremental modification to artifact tagging
 52  // (klasses, methods, CLDs, etc), used to request collection of artifacts.
 53  //
 54 class JfrTraceIdEpoch : AllStatic {
 55   friend class JfrCheckpointManager;
 56  private:
 57   static bool _epoch_state;
 58   static bool _synchronizing;
 59   static volatile bool _changed_tag_state;
 60 
 61   static void begin_epoch_shift();
 62   static void end_epoch_shift();
 63 
 64   static bool changed_tag_state() {
 65     return Atomic::load_acquire(&_changed_tag_state);
 66   }
 67 
 68   static void set_tag_state(bool value) {
 69     Atomic::release_store(&_changed_tag_state, value);
 70   }
 71 
 72  public:
 73   static bool epoch() {
 74     return _epoch_state;
 75   }
 76 
 77   static jlong epoch_address() {
 78     return (jlong)&_epoch_state;
 79   }
 80 
 81   static u1 current() {
 82     return _epoch_state ? (u1)1 : (u1)0;
 83   }
 84 
 85   static u1 previous() {
 86     return _epoch_state ? (u1)0 : (u1)1;
 87   }
 88 
 89   static bool is_synchronizing() {
 90     return Atomic::load_acquire(&_synchronizing);
 91   }
 92 
 93   static traceid in_use_this_epoch_bit() {
 94     return _epoch_state ? USED_EPOCH_2_BIT : USED_EPOCH_1_BIT;
 95   }
 96 
 97   static traceid in_use_prev_epoch_bit() {
 98     return _epoch_state ? USED_EPOCH_1_BIT : USED_EPOCH_2_BIT;
 99   }
100 
101   static traceid method_in_use_this_epoch_bit() {
102     return _epoch_state ? METHOD_USED_EPOCH_2_BIT : METHOD_USED_EPOCH_1_BIT;
103   }
104 
105   static traceid method_in_use_prev_epoch_bit() {
106     return _epoch_state ? METHOD_USED_EPOCH_1_BIT : METHOD_USED_EPOCH_2_BIT;
107   }
108 
109   static traceid method_and_class_in_use_this_epoch_bits() {
110     return _epoch_state ? METHOD_AND_CLASS_IN_USE_EPOCH_2_BITS : METHOD_AND_CLASS_IN_USE_EPOCH_1_BITS;
111   }
112 
113   static traceid method_and_class_in_use_prev_epoch_bits() {
114     return _epoch_state ? METHOD_AND_CLASS_IN_USE_EPOCH_1_BITS :  METHOD_AND_CLASS_IN_USE_EPOCH_2_BITS;
115   }
116 
117   static bool has_changed_tag_state() {
118     if (changed_tag_state()) {
119       set_tag_state(false);
120       return true;
121     }
122     return false;
123   }
124 
125   static void set_changed_tag_state() {
126     if (!changed_tag_state()) {
127       set_tag_state(true);
128     }
129   }
130 };
131 
132 #endif // SHARE_JFR_RECORDER_CHECKPOINT_TYPES_TRACEID_JFRTRACEIDEPOCH_HPP
