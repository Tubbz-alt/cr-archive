45   int64_t _last_metadata_offset;

69   int64_t last_metadata_offset() const;

73   int64_t start_ticks() const;
74   int64_t start_nanos() const;
75 
76   int64_t previous_start_ticks() const;
77   int64_t previous_start_nanos() const;
78   int64_t last_chunk_duration() const;

80   void set_time_stamp();

82 
83   void set_path(const char* path);

85 
86   bool is_started() const;
87   bool is_finished() const;
88 
89   int64_t duration() const;
90   u1 generation() const;
91   u1 next_generation() const;
92 };
