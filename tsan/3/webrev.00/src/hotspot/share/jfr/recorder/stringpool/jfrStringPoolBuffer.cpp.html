25 #include "precompiled.hpp"
26 #include "jfr/recorder/stringpool/jfrStringPoolBuffer.hpp"
27 
28 JfrStringPoolBuffer::JfrStringPoolBuffer() : JfrBuffer(), _string_count_pos(0), _string_count_top(0) {}
29 
30 void JfrStringPoolBuffer::reinitialize() {
31   assert(acquired_by_self() || retired(), "invariant");
32   set_string_pos(0);
33   set_string_top(0);
34   JfrBuffer::reinitialize();
35 }
36 
37 uint64_t JfrStringPoolBuffer::string_pos() const {
38   assert(acquired_by_self() || retired(), "invariant");
39   return _string_count_pos;
40 }
41 
42 uint64_t JfrStringPoolBuffer::string_top() const {
43   assert(acquired_by_self() || retired(), "invariant");
44   return _string_count_top;
45 }
46 
47 uint64_t JfrStringPoolBuffer::string_count() const {
48   assert(acquired_by_self() || retired(), "invariant");
49   return string_pos() - string_top();
50 }
51 
52 void JfrStringPoolBuffer::set_string_pos(uint64_t value) {
53   assert(acquired_by_self() || retired(), "invariant");
54   _string_count_pos = value;
55 }
56 
57 void JfrStringPoolBuffer::increment(uint64_t value) {
58   assert(acquired_by_self(), "invariant");
59   ++_string_count_pos;
60 }
61 
62 void JfrStringPoolBuffer::set_string_top(uint64_t value) {
63   assert(acquired_by_self() || retired(), "invariant");
64   _string_count_top = value;
65 }
