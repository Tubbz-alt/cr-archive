 43 template <typename Operation>
 44 inline bool ConcurrentWriteOp<Operation>::process(typename Operation::Type* t) {
 45   const u1* const current_top = t->concurrent_top();
 46   const size_t unflushed_size = t->pos() - current_top;

 47   if (unflushed_size == 0) {
 48     t->set_concurrent_top(current_top);
 49     return true;
 50   }
 51   const bool result = _operation.write(t, current_top, unflushed_size);
 52   t->set_concurrent_top(current_top + unflushed_size);
 53   return result;
 54 }

 56 template <typename Operation>
 57 inline bool ConcurrentWriteOpExcludeRetired<Operation>::process(typename Operation::Type* t) {
 58   if (t->retired()) {
 59     assert(t->empty(), "invariant");
 60     return true;
 61   }
 62   return ConcurrentWriteOp<Operation>::process(t);
 63 }

 65 template <typename Operation>
 66 inline bool MutexedWriteOp<Operation>::process(typename Operation::Type* t) {
 67   assert(t != NULL, "invariant");
 68   const u1* const current_top = t->top();
 69   const size_t unflushed_size = t->pos() - current_top;
 70   if (unflushed_size == 0) {
 71     return true;
 72   }

 74   t->set_top(current_top + unflushed_size);
 75   return result;

 78 template <typename Operation>
 79 inline bool DiscardOp<Operation>::process(typename Operation::Type* t) {
 80   assert(t != NULL, "invariant");
 81   const u1* const current_top = _mode == concurrent ? t->concurrent_top() : t->top();
 82   const size_t unflushed_size = t->pos() - current_top;
 83   if (unflushed_size == 0) {
 84     if (_mode == concurrent) {
 85       t->set_concurrent_top(current_top);
 86     }
 87     return true;
 88   }
 89   const bool result = _operation.discard(t, current_top, unflushed_size);
 90   if (_mode == concurrent) {
 91     t->set_concurrent_top(current_top + unflushed_size);
 92   } else {
 93     t->set_top(current_top + unflushed_size);
 94   }
 95   return result;
 96 }

 47 template <typename Type>
 48 inline size_t get_unflushed_size(const u1* top, Type* t) {
 49   assert(t != NULL, "invariant");
 50   return Atomic::load_acquire(t->pos_address()) - top;
 51 }

 53 template <typename Operation>
 54 inline bool ConcurrentWriteOp<Operation>::process(typename Operation::Type* t) {
 55   // acquire_critical_section_top() must be read before pos() for stable access
 56   const u1* const top = t->acquire_critical_section_top();
 57   const size_t unflushed_size = get_unflushed_size(top, t);
 58   if (unflushed_size == 0) {
 59     t->release_critical_section_top(top);
 60     return true;
 61   }
 62   const bool result = _operation.write(t, top, unflushed_size);
 63   t->release_critical_section_top(top + unflushed_size);
 64   return result;
 65 }

 67 template <typename Operation>
 68 inline bool MutexedWriteOp<Operation>::process(typename Operation::Type* t) {
 69   assert(t != NULL, "invariant");
 70   const u1* const top = t->top();
 71   const size_t unflushed_size = get_unflushed_size(top, t);
 72   if (unflushed_size == 0) {
 73     return true;
 74   }
 75   const bool result = _operation.write(t, top, unflushed_size);
 76   t->set_top(top + unflushed_size);
 77   return result;
 78 }

 80 template <typename Type>
 81 static void retired_sensitive_acquire(Type* t) {
 82   assert(t != NULL, "invariant");
 83   if (t->retired()) {
 84     return;
 85   }
 86   Thread* const thread = Thread::current();
 87   while (!t->try_acquire(thread)) {
 88     if (t->retired()) {
 89       return;
 90     }
 91   }
 92 }

 94 template <typename Operation>
 95 inline bool ExclusiveOp<Operation>::process(typename Operation::Type* t) {
 96   retired_sensitive_acquire(t);
 97   assert(t->acquired_by_self() || t->retired(), "invariant");
 98   // User is required to ensure proper release of the acquisition
 99   return MutexedWriteOp<Operation>::process(t);
100 }

102 template <typename Operation>
103 inline bool DiscardOp<Operation>::process(typename Operation::Type* t) {
104   assert(t != NULL, "invariant");
105   const u1* const top = _mode == concurrent ? t->acquire_critical_section_top() : t->top();
106   const size_t unflushed_size = get_unflushed_size(top, t);
107   if (unflushed_size == 0) {
108     if (_mode == concurrent) {
109       t->release_critical_section_top(top);
110     }
111     return true;
112   }
113   const bool result = _operation.discard(t, top, unflushed_size);
114   if (_mode == concurrent) {
115     t->release_critical_section_top(top + unflushed_size);
116   } else {
117     t->set_top(top + unflushed_size);
118   }
119   return result;
120 }
