46 template <typename Delegate>
47 RootSetClosure<Delegate>::RootSetClosure(Delegate* delegate) : _delegate(delegate) {}

48 
49 template <typename Delegate>
50 void RootSetClosure<Delegate>::do_oop(oop* ref) {
51   assert(ref != NULL, "invariant");











52   assert(is_aligned(ref, HeapWordSize), "invariant");
53   if (*ref != NULL) {
54     _delegate->do_root(UnifiedOopRef::encode_in_native(ref));

55   }
56 }
57 
58 template <typename Delegate>
59 void RootSetClosure<Delegate>::do_oop(narrowOop* ref) {
60   assert(ref != NULL, "invariant");
61   assert(is_aligned(ref, sizeof(narrowOop)), "invariant");
62   if (*ref != 0) {
63     _delegate->do_root(UnifiedOopRef::encode_in_native(ref));







64   }
65 }

67 class RootSetClosureMarkScope : public MarkScope {};






68 
69 template <typename Delegate>
70 void RootSetClosure<Delegate>::process() {
71   RootSetClosureMarkScope mark_scope;
72   CLDToOopClosure cldt_closure(this, ClassLoaderData::_claim_none);

73   ClassLoaderDataGraph::always_strong_cld_do(&cldt_closure);
74   // We don't follow code blob oops, because they have misaligned oops.
75   Threads::oops_do(this, NULL);
76   ObjectSynchronizer::oops_do(this);
77   Universe::oops_do(this);
78   JNIHandles::oops_do(this);
79   JvmtiExport::oops_do(this);
80   SystemDictionary::oops_do(this);
81   Management::oops_do(this);
82   AOTLoader::oops_do(this);

83 }
