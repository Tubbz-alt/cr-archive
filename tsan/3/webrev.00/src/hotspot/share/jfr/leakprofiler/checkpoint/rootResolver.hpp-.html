32 struct RootCallbackInfo {
33   const void* _high;
34   const void* _low;
35   const void* _context;
36   OldObjectRoot::System _system;
37   OldObjectRoot::Type _type;
38 };

40 class RootCallback {

43   virtual int entries() const = 0;

45 };

49   static void resolve(RootCallback& callback);
50 };
