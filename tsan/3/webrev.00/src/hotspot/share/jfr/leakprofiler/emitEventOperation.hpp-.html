28 #include "runtime/vmOperations.hpp"

30 class BFSClosure;
31 class EdgeStore;
32 class EdgeQueue;
33 class JfrThreadData;
34 class ObjectSample;

39 // Safepoint operation for emitting object sample events
40 class EmitEventOperation : public VM_Operation {
41  private:
42   jlong _cutoff_ticks;
43   bool _emit_all;
44   VMThread* _vm_thread;
45   JfrThreadLocal* _vm_thread_local;
46   ObjectSampler* _object_sampler;
47 
48   void write_event(const ObjectSample* sample, EdgeStore* edge_store);
49   int write_events(EdgeStore* edge_store);
50 
51  public:
52   EmitEventOperation(jlong cutoff_ticks, bool emit_all) :
53     _cutoff_ticks(cutoff_ticks),
54     _emit_all(emit_all),
55     _vm_thread(NULL),
56     _vm_thread_local(NULL),
57     _object_sampler(NULL) {
58   }
59 
60   VMOp_Type type() const {
61     return VMOp_GC_HeapInspection;
62   }
63 
64   Mode evaluation_mode() const {
65     return _safepoint;
66   }
67 
68   virtual void doit();
69 };
70 
71 #endif // SHARE_JFR_LEAKPROFILER_EMITEVENTOPERATION_HPP
