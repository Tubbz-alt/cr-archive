25 #ifndef SHARE_JFR_LEAKPROFILER_SAMPLING_OBJECTSAMPLER_HPP
26 #define SHARE_JFR_LEAKPROFILER_SAMPLING_OBJECTSAMPLER_HPP
27 
28 #include "memory/allocation.hpp"
29 #include "jfr/utilities/jfrTime.hpp"
30 
31 typedef u8 traceid;
32 
33 class BoolObjectClosure;
34 class JavaThread;
35 class OopClosure;
36 class ObjectSample;

37 class SampleList;
38 class SamplePriorityQueue;

39 
40 // Class reponsible for holding samples and
41 // making sure the samples are evenly distributed as
42 // new entries are added and removed.
43 class ObjectSampler : public CHeapObj<mtTracing> {
44   friend class LeakProfiler;

45   friend class StartOperation;
46   friend class StopOperation;

47  private:
48   SamplePriorityQueue* _priority_queue;
49   SampleList* _list;
50   JfrTicks _last_sweep;
51   size_t _total_allocated;
52   size_t _threshold;
53   size_t _size;

54   bool _dead_samples;
55 
56   // Lifecycle
57   explicit ObjectSampler(size_t size);
58   ~ObjectSampler();
59   static bool create(size_t size);
60   static bool is_created();
61   static void destroy();
62 
63   // Sampling
64   static void sample(HeapWord* object, size_t size, JavaThread* thread);
65   void add(HeapWord* object, size_t size, traceid thread_id, JavaThread* thread);
66   void scavenge();
67   void remove_dead(ObjectSample* sample);
68 
69   // Called by GC
70   static void weak_oops_do(BoolObjectClosure* is_alive, OopClosure* f);
71 

72   const ObjectSample* item_at(int index) const;
73   ObjectSample* item_at(int index);
74   int item_count() const;
75 
76  public:
77   static ObjectSampler* sampler();
78   // For operations that require exclusive access (non-safepoint)
79   static ObjectSampler* acquire();
80   static void release();
81 
82   const ObjectSample* first() const;
83   ObjectSample* last() const;
84   const ObjectSample* last_resolved() const;
85   void set_last_resolved(const ObjectSample* sample);
86   const JfrTicks& last_sweep() const;
87 };
88 
89 #endif // SHARE_JFR_LEAKPROFILER_SAMPLING_OBJECTSAMPLER_HPP
