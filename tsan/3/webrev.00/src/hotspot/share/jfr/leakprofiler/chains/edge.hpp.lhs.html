25 #ifndef SHARE_JFR_LEAKPROFILER_CHAINS_EDGE_HPP
26 #define SHARE_JFR_LEAKPROFILER_CHAINS_EDGE_HPP
27 

28 #include "memory/allocation.hpp"
29 #include "oops/oopsHierarchy.hpp"
30 
31 class Edge {
32  private:
33   const Edge* _parent;
34   const oop* _reference;
35  public:
36   Edge();
37   Edge(const Edge* parent, const oop* reference);
38 
39   const oop* reference() const {
40     return _reference;
41   }
42   const Edge* parent() const {
43     return _parent;
44   }
45   bool is_root() const {
46     return _parent == NULL;
47   }
48   const oop pointee() const;
49   const Klass* pointee_klass() const;
50   const oop reference_owner() const;
51   const Klass* reference_owner_klass() const;
52   size_t distance_to_root() const;
53 
54   void* operator new (size_t sz, void* here) {
55     return here;
56   }
57 };

59 #endif // SHARE_JFR_LEAKPROFILER_CHAINS_EDGE_HPP
