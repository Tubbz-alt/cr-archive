25 #ifndef SHARE_JFR_LEAKPROFILER_UTILITIES_UNIFIEDOOP_HPP
26 #define SHARE_JFR_LEAKPROFILER_UTILITIES_UNIFIEDOOP_HPP
27 
28 #include "oops/oop.inline.hpp"
29 
30 class UnifiedOop : public AllStatic {
31  public:
32   static const bool is_narrow(const oop* ref) {
33     assert(ref != NULL, "invariant");
34     return 1 == (((u8)ref) & 1);
35   }
36 
37   static const oop* decode(const oop* ref) {
38     assert(ref != NULL, "invariant");
39     return is_narrow(ref) ? (const oop*)(((u8)ref) & ~1) : ref;
40   }
41 
42   static const oop* encode(narrowOop* ref) {
43     assert(ref != NULL, "invariant");
44     return (const oop*)((u8)ref | 1);
45   }
46 
47   static oop dereference(const oop* ref) {
48     assert(ref != NULL, "invariant");
49     return is_narrow(ref) ?
50       (oop)RawAccess<>::oop_load((narrowOop*)decode(ref)) :
51       (oop)RawAccess<>::oop_load(const_cast<oop*>(ref));
52 
53   }
54 };
55 
56 #endif // SHARE_JFR_LEAKPROFILER_UTILITIES_UNIFIEDOOP_HPP
