41 inline int compare_traceid(const traceid& lhs, const traceid& rhs) {
42   return lhs > rhs ? 1 : (lhs < rhs) ? -1 : 0;
43 }
44 
45 inline int sort_traceid(traceid* lhs, traceid* rhs) {
46   return compare_traceid(*lhs, *rhs);
47 }
