25 #ifndef SHARE_JFR_UTILITIES_JFRLOGTAGSETS_HPP
26 #define SHARE_JFR_UTILITIES_JFRLOGTAGSETS_HPP
27 
28 #include "logging/logTag.hpp"
29 
30 /*
31  * This file declares the Unified Logging (UL) tag sets for JFR.
32  *
33  * The JFR_LOG_TAG_SET_LIST macro will generate an enum
34  * (zero-based ordinals) for these log tag sets.
35  * Log tag set order (enum ordinals) is important
36  * because we need to mirror this order in Java.
37  *
38  * Adding a new log tag set:
39  *
40  * 1. Ensure the log tag primitive which you attempt to use in your new log tag set is defined to UL.
41  *    UL log tags are defined in logging/logTag.hpp.
42  * 2. Append the new log tag set to the JFR_LOG_TAG_SET_LIST macro in this file.
43  * 3. Make the corresponding change in Java by updating jdk.jfr.internal.LogTag.java
44  *    to add another enum instance with a corresponding tag id ordinal.
45  *
46  */
47 
48 #define JFR_LOG_TAG_SET_LIST \
49   JFR_LOG_TAG(jfr) \
50   JFR_LOG_TAG(jfr, system) \
51   JFR_LOG_TAG(jfr, system, event) \
52   JFR_LOG_TAG(jfr, system, setting) \
53   JFR_LOG_TAG(jfr, system, bytecode) \
54   JFR_LOG_TAG(jfr, system, parser) \
55   JFR_LOG_TAG(jfr, system, metadata) \
56   JFR_LOG_TAG(jfr, metadata) \
57   JFR_LOG_TAG(jfr, event) \
58   JFR_LOG_TAG(jfr, setting) \
59   JFR_LOG_TAG(jfr, dcmd)
60   /* NEW TAGS, DONT FORGET TO UPDATE JAVA SIDE */
61 
62 #endif // SHARE_JFR_UTILITIES_JFRLOGTAGSETS_HPP
