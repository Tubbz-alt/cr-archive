25 #ifndef SHARE_JFR_UTILITIES_JFRBLOB_HPP
26 #define SHARE_JFR_UTILITIES_JFRBLOB_HPP
27 
28 #include "jfr/utilities/jfrAllocation.hpp"
29 #include "jfr/utilities/jfrRefCountPointer.hpp"
30 
31 class JfrBlob;
32 typedef RefCountPointer<JfrBlob, MultiThreadedRefCounter> JfrBlobReference;
33 typedef RefCountHandle<JfrBlobReference> JfrBlobHandle;
34 
35 class JfrBlob : public JfrCHeapObj {
36   template <typename, typename>
37   friend class RefCountPointer;
38  private:
39   const u1* const _data;
40   JfrBlobHandle _next;
41   const size_t _size;
42   mutable bool _written;
43 
44   JfrBlob(const u1* data, size_t size);
45   ~JfrBlob();
46 
47  public:
48   void set_next(const JfrBlobHandle& ref);
49   void reset_write_state() const;
50   static JfrBlobHandle make(const u1* data, size_t size);
51   template <typename Writer>
52   void write(Writer& writer) const {
53     writer.bytes(_data, _size);
54     if (_next.valid()) {
55       _next->write(writer);
56     }
57   }
58   template <typename Writer>
59   void exclusive_write(Writer& writer) const {
60     if (_written) {
61       return;
62     }
63     writer.bytes(_data, _size);
64     _written = true;
65     if (_next.valid()) {
66       _next->exclusive_write(writer);
67     }
68   }
69 };
70 
71 #endif // SHARE_JFR_UTILITIES_JFRBLOB_HPP
