 40  // incremented during class unloading for each unloaded event class
 41 static jlong unloaded_event_classes = 0;
 42 
 43 jlong JfrEventClasses::unloaded_event_classes_count() {
 44   return unloaded_event_classes;
 45 }
 46 
 47 void JfrEventClasses::increment_unloaded_event_class() {
 48   assert_locked_or_safepoint(ClassLoaderDataGraph_lock);

 49   ++unloaded_event_classes;
 50 }

 52 static jobject empty_java_util_arraylist = NULL;
 53 
 54 static oop new_java_util_arraylist(TRAPS) {
 55   DEBUG_ONLY(JfrJavaSupport::check_java_thread_in_vm(THREAD));
 56   JavaValue result(T_OBJECT);
 57   JfrJavaArguments args(&result, "java/util/ArrayList", "<init>", "()V", CHECK_NULL);
 58   JfrJavaSupport::new_object(&args, CHECK_NULL);
 59   return (oop)result.get_jobject();
 60 }
 61 
 62 static bool initialize(TRAPS) {
 63   static bool initialized = false;
 64   if (!initialized) {
 65     unloaded_event_classes = 0;
 66     assert(NULL == empty_java_util_arraylist, "invariant");
 67     const oop array_list = new_java_util_arraylist(CHECK_false);
 68     empty_java_util_arraylist = JfrJavaSupport::global_jni_handle(array_list, THREAD);
 69     initialized = empty_java_util_arraylist != NULL;
 70   }
 71   return initialized;
 72 }

 74 /*
 75  * Abstract klasses are filtered out unconditionally.
 76  * If a klass is not yet initialized, i.e yet to run its <clinit>
 77  * it is also filtered out so we don't accidentally
 78  * trigger initialization.
 79  */
 80 static bool is_whitelisted(const Klass* k) {
 81   assert(k != NULL, "invariant");
 82   return !(k->is_abstract() || k->should_be_initialized());
 83 }
 84 
 85 static void fill_klasses(GrowableArray<const void*>& event_subklasses, const Klass* event_klass, Thread* thread) {
 86   assert(event_subklasses.length() == 0, "invariant");
 87   assert(event_klass != NULL, "invariant");
 88   DEBUG_ONLY(JfrJavaSupport::check_java_thread_in_vm(thread));
 89 
 90   Stack<const Klass*, mtTracing> mark_stack;
 91   MutexLocker ml(thread, Compile_lock);
 92   mark_stack.push(event_klass->subklass());
 93 
 94   while (!mark_stack.is_empty()) {
 95     const Klass* const current = mark_stack.pop();
 96     assert(current != NULL, "null element in stack!");
 97 
 98     if (is_whitelisted(current)) {
 99       event_subklasses.append(current);
100     }
101 
102     // subclass (depth)
103     const Klass* next_klass = current->subklass();
104     if (next_klass != NULL) {
105       mark_stack.push(next_klass);
106     }
107 
108     // siblings (breadth)
109     next_klass = current->next_sibling();
110     if (next_klass != NULL) {
111       mark_stack.push(next_klass);
112     }
113   }
114   assert(mark_stack.is_empty(), "invariant");
115 }

117  static void transform_klasses_to_local_jni_handles(GrowableArray<const void*>& event_subklasses, Thread* thread) {
118   assert(event_subklasses.is_nonempty(), "invariant");
119   DEBUG_ONLY(JfrJavaSupport::check_java_thread_in_vm(thread));
120 
121   for (int i = 0; i < event_subklasses.length(); ++i) {
122     const InstanceKlass* k = static_cast<const InstanceKlass*>(event_subklasses.at(i));
123     assert(is_whitelisted(k), "invariant");
124     event_subklasses.at_put(i, JfrJavaSupport::local_jni_handle(k->java_mirror(), thread));
125   }
126 }

128 static const int initial_size_growable_array = 64;
129 
130 jobject JfrEventClasses::get_all_event_classes(TRAPS) {
131   DEBUG_ONLY(JfrJavaSupport::check_java_thread_in_vm(THREAD));
132   initialize(THREAD);
133   assert(empty_java_util_arraylist != NULL, "should have been setup already!");
134   static const char jdk_jfr_event_name[] = "jdk/internal/event/Event";
135   Symbol* const event_klass_name = SymbolTable::probe(jdk_jfr_event_name, sizeof jdk_jfr_event_name - 1);

136 
137   if (NULL == event_klass_name) {
138     // not loaded yet
139     return empty_java_util_arraylist;
140   }
141 
142   const Klass* const klass = SystemDictionary::resolve_or_null(event_klass_name, THREAD);
143   assert(klass != NULL, "invariant");
144   assert(JdkJfrEvent::is(klass), "invariant");
145 
146   if (klass->subklass() == NULL) {
147     return empty_java_util_arraylist;
148   }
149 
150   ResourceMark rm(THREAD);
151   GrowableArray<const void*> event_subklasses(THREAD, initial_size_growable_array);
152   fill_klasses(event_subklasses, klass, THREAD);
153 
154   if (event_subklasses.is_empty()) {
155     return empty_java_util_arraylist;
156   }
157 
158   transform_klasses_to_local_jni_handles(event_subklasses, THREAD);
159 
160   Handle h_array_list(THREAD, new_java_util_arraylist(THREAD));
161   assert(h_array_list.not_null(), "invariant");
162 
163   static const char add_method_name[] = "add";
164   static const char add_method_signature[] = "(Ljava/lang/Object;)Z";
165   const Klass* const array_list_klass = JfrJavaSupport::klass(empty_java_util_arraylist);
166   assert(array_list_klass != NULL, "invariant");
167 
168   const Symbol* const add_method_sym = SymbolTable::new_symbol(add_method_name);
169   assert(add_method_sym != NULL, "invariant");
170 
171   const Symbol* const add_method_sig_sym = SymbolTable::new_symbol(add_method_signature);
172   assert(add_method_signature != NULL, "invariant");
173 
174   JavaValue result(T_BOOLEAN);
175   for (int i = 0; i < event_subklasses.length(); ++i) {
176     const jclass clazz = (const jclass)event_subklasses.at(i);
177     assert(JdkJfrEvent::is_subklass(clazz), "invariant");
178     JfrJavaArguments args(&result, array_list_klass, add_method_sym, add_method_sig_sym);
179     args.set_receiver(h_array_list());
180     args.push_jobject(clazz);
181     JfrJavaSupport::call_virtual(&args, THREAD);
182     if (HAS_PENDING_EXCEPTION || JNI_FALSE == result.get_jboolean()) {
183       return empty_java_util_arraylist;
184     }
185   }
186   return JfrJavaSupport::local_jni_handle(h_array_list(), THREAD);
187 }
