 37 void SafepointMechanism::default_initialize() {
 38   if (ThreadLocalHandshakes) {
 39     set_uses_thread_local_poll();
 40 
 41     // Poll bit values
 42     intptr_t poll_armed_value = poll_bit();
 43     intptr_t poll_disarmed_value = 0;
 44 
 45 #ifdef USE_POLL_BIT_ONLY
 46     if (!USE_POLL_BIT_ONLY)
 47 #endif
 48     {
 49       // Polling page
 50       const size_t page_size = os::vm_page_size();
 51       const size_t allocation_size = 2 * page_size;
 52       char* polling_page = os::reserve_memory(allocation_size, NULL, page_size);
 53       os::commit_memory_or_exit(polling_page, allocation_size, false, "Unable to commit Safepoint polling page");
 54       MemTracker::record_virtual_memory_type((address)polling_page, mtSafepoint);
 55 
 56       char* bad_page  = polling_page;
 57       char* good_page = polling_page + page_size;
 58 
 59       os::protect_memory(bad_page,  page_size, os::MEM_PROT_NONE);

 66       intptr_t bad_page_val  = reinterpret_cast<intptr_t>(bad_page),
 67                good_page_val = reinterpret_cast<intptr_t>(good_page);
 68       poll_armed_value    |= bad_page_val;
 69       poll_disarmed_value |= good_page_val;
 70     }
 71 
 72     _poll_armed_value    = reinterpret_cast<void*>(poll_armed_value);
 73     _poll_disarmed_value = reinterpret_cast<void*>(poll_disarmed_value);
 74   } else {
 75     const size_t page_size = os::vm_page_size();
 76     char* polling_page = os::reserve_memory(page_size, NULL, page_size);
 77     os::commit_memory_or_exit(polling_page, page_size, false, "Unable to commit Safepoint polling page");
 78     os::protect_memory(polling_page, page_size, os::MEM_PROT_READ);
 79     MemTracker::record_virtual_memory_type((address)polling_page, mtSafepoint);
 80 
 81     log_info(os)("SafePoint Polling address: " INTPTR_FORMAT, p2i(polling_page));
 82     os::set_polling_page((address)(polling_page));
 83   }

 85 void SafepointMechanism::block_or_handshake(JavaThread *thread) {

 86   if (global_poll()) {
 87     // Any load in ::block must not pass the global poll load.
 88     // Otherwise we might load an old safepoint counter (for example).
 89     OrderAccess::loadload();
 90     SafepointSynchronize::block(thread);
 91   }
 92   if (uses_thread_local_poll() && thread->has_handshake()) {
 93     thread->handshake_process_by_self();
 94   }
 95 }

 97 void SafepointMechanism::block_if_requested_slow(JavaThread *thread) {
 98   // Read global poll and has_handshake after local poll
 99   OrderAccess::loadload();
100 
101   // local poll already checked, if used.
102   block_or_handshake(thread);
103 
104   OrderAccess::loadload();
105 
106   if (uses_thread_local_poll() && local_poll_armed(thread)) {
107     disarm_local_poll_release(thread);
108     // We might have disarmed next safepoint/handshake
109     OrderAccess::storeload();
110     if (global_poll() || thread->has_handshake()) {
111       arm_local_poll(thread);
112     }
113   }
114 
115   OrderAccess::cross_modify_fence();
116 }
