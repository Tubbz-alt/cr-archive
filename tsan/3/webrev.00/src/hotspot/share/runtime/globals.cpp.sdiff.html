52 #define MATERIALIZE_PD_DEVELOPER_FLAG(type, name, doc)
53 #define MATERIALIZE_NOTPRODUCT_FLAG(type, name, value, doc)
54 #else
55 #define MATERIALIZE_DEVELOPER_FLAG(type, name, value, doc)    type name = value;
56 #define MATERIALIZE_PD_DEVELOPER_FLAG(type, name, doc)        type name = pd_##name;
57 #define MATERIALIZE_NOTPRODUCT_FLAG(type, name, value, doc)   type name = value;
58 #endif // PRODUCT
59 #ifdef _LP64
60 #define MATERIALIZE_LP64_PRODUCT_FLAG(type, name, value, doc) type name = value;
