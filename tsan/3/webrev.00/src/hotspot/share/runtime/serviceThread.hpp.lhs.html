25 #ifndef SHARE_RUNTIME_SERVICETHREAD_HPP
26 #define SHARE_RUNTIME_SERVICETHREAD_HPP
27 

28 #include "runtime/thread.hpp"
29 
30 // A JavaThread for low memory detection support and JVMTI
31 // compiled-method-load events.



32 class ServiceThread : public JavaThread {
33   friend class VMStructs;
34  private:
35 
36   static ServiceThread* _instance;


37 
38   static void service_thread_entry(JavaThread* thread, TRAPS);
39   ServiceThread(ThreadFunction entry_point) : JavaThread(entry_point) {};
40 
41  public:
42   static void initialize();
43 
44   // Hide this thread from external view.
45   bool is_hidden_from_external_view() const      { return true; }




46 
47   // Returns true if the passed thread is the service thread.
48   static bool is_service_thread(Thread* thread);

49 };
50 
51 #endif // SHARE_RUNTIME_SERVICETHREAD_HPP
