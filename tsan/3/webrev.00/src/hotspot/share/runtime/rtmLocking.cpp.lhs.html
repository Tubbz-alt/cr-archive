35 // One-shot PeriodicTask subclass for enabling RTM locking
36 uintx RTMLockingCounters::_calculation_flag = 0;
37 
38 class RTMLockingCalculationTask : public PeriodicTask {
39  public:
40   RTMLockingCalculationTask(size_t interval_time) : PeriodicTask(interval_time){  }
41 
42   virtual void task() {
43     RTMLockingCounters::_calculation_flag = 1;
44     // Reclaim our storage and disenroll ourself
45     delete this;
46   }
47 };
48 
49 void RTMLockingCounters::init() {
50   if (UseRTMLocking && RTMLockingCalculationDelay > 0) {
51     RTMLockingCalculationTask* task = new RTMLockingCalculationTask(RTMLockingCalculationDelay);
52     task->enroll();
53   } else {
54     _calculation_flag = 1;
55   }
56 }

58 //------------------------------print_on-------------------------------
59 void RTMLockingCounters::print_on(outputStream* st) {
60   tty->print_cr("# rtm locks total (estimated): " UINTX_FORMAT, _total_count * RTMTotalCountIncrRate);
61   tty->print_cr("# rtm lock aborts  : " UINTX_FORMAT, _abort_count);
62   for (int i = 0; i < ABORT_STATUS_LIMIT; i++) {
63     tty->print_cr("# rtm lock aborts %d: " UINTX_FORMAT, i, _abortX_count[i]);
64   }
65 }
