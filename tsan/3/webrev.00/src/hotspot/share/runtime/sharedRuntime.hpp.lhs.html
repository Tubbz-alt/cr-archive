287   // TSAN uses a 64-bit value to identify code location.
288   // TSAN uses the uppermost 3 bits (63:61) for the internal purposes.
289   // If bit 60 is set, TSAN recognizes that the code location belongs to the
290   // JVM, and will call __tsan_symbolize_external_ex() for symbolization rather
291   // than TSAN's own symbolizer. See __sanitizer::kExternalPCBit and
292   // __tsan::__tsan_symbolize_external_ex() in TSAN for more details.
293   // The lower 60 bits may contain either a packed bytecode location, or an
294   // instruction address inside the code generated by JIT compiler.
295   // A packed code location has the method ID in bits 59:16 and the bytecode
296   //offset within method in bits 15:0. 44 bits (59:16) are enough to encode any
297   // 47-bit 8-byte-aligned address, which is the maximum address space TSAN
298   // allows. The next 16 bits are used for storing the bci.
299   // | Tsan: 3 | TsanJava: 1 | jmethodID: 44 | BCI: 16 |
300   static const int tsan_method_id_alignment_bits = 3;
301   static const int tsan_bci_bits = 16;
302   static const u8 tsan_bci_mask = right_n_bits(tsan_bci_bits);
303   static const int tsan_method_id_shift = tsan_bci_bits -
304       tsan_method_id_alignment_bits;
305   static const u8 tsan_fake_pc_bit = 1L << 60;
306   static void * tsan_code_location(jmethodID jmethod_id_ptr, u2 bci) {
307     return (void *)(tsan_fake_pc_bit |
308       (((u8)(jmethod_id_ptr)) << tsan_method_id_shift) | bci);
309   }
310   static jmethodID tsan_method_id_from_code_location(u8 loc) {
311     return (jmethodID)(
312         (loc & ~(tsan_fake_pc_bit | tsan_bci_mask)) >> tsan_method_id_shift);
313   }
314   static u2 tsan_bci_from_code_location(u8 loc) {
315     return (u2)(loc & tsan_bci_mask);
316   }

448   // compiled Java code would like to put the results.  Values in reg_lo and

454   // the bottom of the frame the first 16 words will be skipped and SharedInfo::stack0
455   // will be just above it. (

478   // Generate I2C and C2I adapters. These adapters are simple argument marshalling
479   // blobs. Unlike adapters in the tiger and earlier releases the code in these
480   // blobs does not create a new frame and are therefore virtually invisible
481   // to the stack walking code. In general these blobs extend the callers stack
482   // as needed for the conversion of argument locations.
483 
484   // When calling a c2i blob the code will always call the interpreter even if
485   // by the time we reach the blob there is compiled code available. This allows
486   // the blob to pass the incoming stack pointer (the sender sp) in a known
487   // location for the interpreter to record. This is used by the frame code
488   // to correct the sender code to match up with the stack pointer when the
489   // thread left the compiled code. In addition it allows the interpreter
490   // to remove the space the c2i adapter allocated to do its argument conversion.
491 
492   // Although a c2i blob will always run interpreted even if compiled code is
493   // present if we see that compiled code is present the compiled call site
494   // will be patched/re-resolved so that later calls will run compiled.
495 
496   // Additionally a c2i blob need to have a unverified entry because it can be reached
497   // in situations where the call site is an inlined cache site and may go megamorphic.
498 
499   // A i2c adapter is simpler than the c2i adapter. This is because it is assumed
500   // that the interpreter before it does any call dispatch will record the current
501   // stack pointer in the interpreter frame. On return it will restore the stack
502   // pointer as needed. This means the i2c adapter code doesn't need any special
503   // handshaking path with compiled code to keep the stack walking correct.
504 
505   static AdapterHandlerEntry* generate_i2c2i_adapters(MacroAssembler *_masm,
506                                                       int total_args_passed,
507                                                       int max_arg,
508                                                       const BasicType *sig_bt,
509                                                       const VMRegPair *regs,
510                                                       AdapterFingerPrint* fingerprint);

678 // ---------------------------------------------------------------------------
679 // Implementation of AdapterHandlerLibrary
680 //
681 // This library manages argument marshaling adapters and native wrappers.
682 // There are 2 flavors of adapters: I2C and C2I.
683 //
684 // The I2C flavor takes a stock interpreted call setup, marshals the
685 // arguments for a Java-compiled call, and jumps to Rmethod-> code()->
686 // code_begin().  It is broken to call it without an nmethod assigned.
687 // The usual behavior is to lift any register arguments up out of the
688 // stack and possibly re-pack the extra arguments to be contiguous.
689 // I2C adapters will save what the interpreter's stack pointer will be
690 // after arguments are popped, then adjust the interpreter's frame
691 // size to force alignment and possibly to repack the arguments.
692 // After re-packing, it jumps to the compiled code start.  There are
693 // no safepoints in this adapter code and a GC cannot happen while
694 // marshaling is in progress.
695 //
696 // The C2I flavor takes a stock compiled call setup plus the target method in
697 // Rmethod, marshals the arguments for an interpreted call and jumps to
698 // Rmethod->_i2i_entry.  On entry, the interpreted frame has not yet been
699 // setup.  Compiled frames are fixed-size and the args are likely not in the
700 // right place.  Hence all the args will likely be copied into the
701 // interpreter's frame, forcing that frame to grow.  The compiled frame's
702 // outgoing stack args will be dead after the copy.
703 //
704 // Native wrappers, like adapters, marshal arguments.  Unlike adapters they
705 // also perform an official frame push & pop.  They have a call to the native
706 // routine in their middles and end in a return (instead of ending in a jump).
707 // The native wrappers are stored in real nmethods instead of the BufferBlobs
708 // used by the adapters.  The code generation happens here because it's very
709 // similar to what the adapters have to do.
710
