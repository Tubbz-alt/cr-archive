 37 // Thread Safe Memory Reclamation (Thread-SMR) support.
 38 //
 39 // ThreadsListHandles are used to safely perform operations on one or more
 40 // threads without the risk of the thread or threads exiting during the
 41 // operation. It is no longer necessary to hold the Threads_lock to safely
 42 // perform an operation on a target thread.
 43 //
 44 // There are several different ways to refer to java.lang.Thread objects
 45 // so we have a few ways to get a protected JavaThread *:
 46 //
 47 // JNI jobject example:
 48 //   jobject jthread = ...;
 49 //   :
 50 //   ThreadsListHandle tlh;
 51 //   JavaThread* jt = NULL;
 52 //   bool is_alive = tlh.cv_internal_thread_to_JavaThread(jthread, &jt, NULL);
 53 //   if (is_alive) {
 54 //     :  // do stuff with 'jt'...
 55 //   }
 56 //
 57 // JVM/TI jthread example:
 58 //   jthread thread = ...;
 59 //   :
 60 //   JavaThread* jt = NULL;
 61 //   ThreadsListHandle tlh;
 62 //   jvmtiError err = JvmtiExport::cv_external_thread_to_JavaThread(tlh.list(), thread, &jt, NULL);
 63 //   if (err != JVMTI_ERROR_NONE) {
 64 //     return err;
 65 //   }
 66 //   :  // do stuff with 'jt'...
 67 //
 68 // JVM/TI oop example (this one should be very rare):
 69 //   oop thread_obj = ...;
 70 //   :
 71 //   JavaThread *jt = NULL;
 72 //   ThreadsListHandle tlh;
 73 //   jvmtiError err = JvmtiExport::cv_oop_to_JavaThread(tlh.list(), thread_obj, &jt);
 74 //   if (err != JVMTI_ERROR_NONE) {
 75 //     return err;
 76 //   }
 77 //   :  // do stuff with 'jt'...
 78 //
 79 // A JavaThread * that is included in the ThreadsList that is held by
 80 // a ThreadsListHandle is protected as long as the ThreadsListHandle
 81 // remains in scope. The target JavaThread * may have logically exited,
 82 // but that target JavaThread * will not be deleted until it is no
 83 // longer protected by a ThreadsListHandle.
 84

 86 // SMR Support for the Threads class.
 87 //
 88 class ThreadsSMRSupport : AllStatic {

 89   friend class SafeThreadsListPtr;  // for _nested_thread_list_max, delete_notify(), release_stable_list_wake_up() access
 90 
 91   // The coordination between ThreadsSMRSupport::release_stable_list() and
 92   // ThreadsSMRSupport::smr_delete() uses the delete_lock in order to
 93   // reduce the traffic on the Threads_lock.
 94   static Monitor* delete_lock() { return ThreadsSMRDelete_lock; }
 95 
 96   // The '_cnt', '_max' and '_times" fields are enabled via
 97   // -XX:+EnableThreadSMRStatistics (see thread.cpp for a
 98   // description about each field):
 99   static uint                  _delete_lock_wait_cnt;
100   static uint                  _delete_lock_wait_max;
101   // The delete_notify flag is used for proper double-check
102   // locking in order to reduce the traffic on the system wide
103   // Thread-SMR delete_lock.
104   static volatile uint         _delete_notify;
105   static volatile uint         _deleted_thread_cnt;
106   static volatile uint         _deleted_thread_time_max;
107   static volatile uint         _deleted_thread_times;
108   static ThreadsList           _bootstrap_list;
109   static ThreadsList* volatile _java_thread_list;
110   static uint64_t              _java_thread_list_alloc_cnt;
111   static uint64_t              _java_thread_list_free_cnt;
112   static uint                  _java_thread_list_max;
113   static uint                  _nested_thread_list_max;
114   static volatile uint         _tlh_cnt;
115   static volatile uint         _tlh_time_max;
116   static volatile uint         _tlh_times;
117   static ThreadsList*          _to_delete_list;
118   static uint                  _to_delete_list_cnt;
119   static uint                  _to_delete_list_max;
120 
121   static ThreadsList *acquire_stable_list_fast_path(Thread *self);
122   static ThreadsList *acquire_stable_list_nested_path(Thread *self);
123   static void add_deleted_thread_times(uint add_value);
124   static void add_tlh_times(uint add_value);
125   static void clear_delete_notify();
126   static bool delete_notify();
127   static void free_list(ThreadsList* threads);
128   static void inc_deleted_thread_cnt();
129   static void inc_java_thread_list_alloc_cnt();
130   static void inc_tlh_cnt();
131   static bool is_a_protected_JavaThread(JavaThread *thread);
132   static void release_stable_list_wake_up(bool is_nested);
133   static void set_delete_notify();
134   static void threads_do(ThreadClosure *tc);
135   static void threads_do(ThreadClosure *tc, ThreadsList *list);
136   static void update_deleted_thread_time_max(uint new_value);
137   static void update_java_thread_list_max(uint new_value);
138   static void update_tlh_time_max(uint new_value);
139   static void verify_hazard_ptr_scanned(Thread *self, ThreadsList *threads);
140   static ThreadsList* xchg_java_thread_list(ThreadsList* new_list);
141 
142  public:
143   static void add_thread(JavaThread *thread);
144   static ThreadsList* get_java_thread_list();
145   static bool is_a_protected_JavaThread_with_lock(JavaThread *thread);
146   static bool is_bootstrap_list(ThreadsList* list);
147   static void remove_thread(JavaThread *thread);
148   static void smr_delete(JavaThread *thread);
149   static void update_tlh_stats(uint millis);
150 
151   // Logging and printing support:
152   static void log_statistics();
153   static void print_info_elements_on(outputStream* st, ThreadsList* t_list);
154   static void print_info_on(outputStream* st);
155   static void print_info_on(const Thread* thread, outputStream* st);
156 };

158 // A fast list of JavaThreads.
159 //
160 class ThreadsList : public CHeapObj<mtThread> {

161   friend class SafeThreadsListPtr;  // for {dec,inc}_nested_handle_cnt() access
162   friend class ThreadsSMRSupport;  // for _nested_handle_cnt, {add,remove}_thread(), {,set_}next_list() access
163 
164   const uint _length;
165   ThreadsList* _next_list;
166   JavaThread *const *const _threads;
167   volatile intx _nested_handle_cnt;
168 
169   template <class T>
170   void threads_do_dispatch(T *cl, JavaThread *const thread) const;
171 
172   ThreadsList *next_list() const        { return _next_list; }
173   void set_next_list(ThreadsList *list) { _next_list = list; }
174 
175   void inc_nested_handle_cnt();
176   void dec_nested_handle_cnt();
177 
178   static ThreadsList* add_thread(ThreadsList* list, JavaThread* java_thread);
179   static ThreadsList* remove_thread(ThreadsList* list, JavaThread* java_thread);
180 
181 public:
182   ThreadsList(int entries);
183   ~ThreadsList();
184 
185   template <class T>
186   void threads_do(T *cl) const;
187 
188   uint length() const                       { return _length; }
189 
190   JavaThread *const thread_at(uint i) const { return _threads[i]; }
191 
192   JavaThread *const *threads() const        { return _threads; }
193 
194   // Returns -1 if target is not found.
195   int find_index_of_JavaThread(JavaThread* target);
196   JavaThread* find_JavaThread_from_java_tid(jlong java_tid) const;
197   bool includes(const JavaThread * const p) const;
198 };

200 // An abstract safe ptr to a ThreadsList comprising either a stable hazard ptr
201 // for leaves, or a retained reference count for nested uses. The user of this
202 // API does not need to know which mechanism is providing the safety.
203 class SafeThreadsListPtr {
204   friend class ThreadsListSetter;
205 
206   SafeThreadsListPtr* _previous;
207   Thread*                 _thread;
208   ThreadsList*            _list;
209   bool                    _has_ref_count;
210   bool                    _needs_release;
211 
212   void acquire_stable_list();
213   void acquire_stable_list_fast_path();
214   void acquire_stable_list_nested_path();
215 
216   void release_stable_list();
217 
218   void verify_hazard_ptr_scanned();
219 
220 public:
221   // Constructor that attaches the list onto a thread.
222   SafeThreadsListPtr(Thread *thread, bool acquire) :
223     _previous(NULL),
224     _thread(thread),
225     _list(NULL),
226     _has_ref_count(false),
227     _needs_release(false)
228   {
229     if (acquire) {
230       acquire_stable_list();
231     }
232   }
233 
234   // Constructor that transfers ownership of the pointer.
235   SafeThreadsListPtr(SafeThreadsListPtr& other) :
236     _previous(other._previous),
237     _thread(other._thread),
238     _list(other._list),
239     _has_ref_count(other._has_ref_count),
240     _needs_release(other._needs_release)
241   {
242     other._needs_release = false;
243   }
244 
245   ~SafeThreadsListPtr() {
246     if (_needs_release) {
247       release_stable_list();
248     }
249   }
250 
251   ThreadsList* list() const { return _list; }
252   SafeThreadsListPtr* previous() const { return _previous; }
253   void print_on(outputStream* st);
254 };

256 // A helper to optionally set the hazard ptr in ourself. This helper can
257 // be used by ourself or by another thread. If the hazard ptr is set(),
258 // then the destructor will release it.
259 //
260 class ThreadsListSetter : public StackObj {
261 private:
262   SafeThreadsListPtr _list_ptr;
263 
264 public:
265   ThreadsListSetter() : _list_ptr(Thread::current(), /* acquire */ false) {}
266   ThreadsList* list() { return _list_ptr.list(); }
267   void set() { _list_ptr.acquire_stable_list(); }
268   bool is_set() { return _list_ptr._needs_release; }
269 };

271 // This stack allocated ThreadsListHandle keeps all JavaThreads in the
272 // ThreadsList from being deleted until it is safe.
273 //
274 class ThreadsListHandle : public StackObj {
275   SafeThreadsListPtr _list_ptr;
276   elapsedTimer _timer;  // Enabled via -XX:+EnableThreadSMRStatistics.
277 
278 public:
279   ThreadsListHandle(Thread *self = Thread::current());
280   ~ThreadsListHandle();
281 
282   ThreadsList *list() const {
283     return _list_ptr.list();
284   }
285 
286   template <class T>
287   void threads_do(T *cl) const {
288     return list()->threads_do(cl);
289   }
290 
291   bool cv_internal_thread_to_JavaThread(jobject jthread, JavaThread ** jt_pp, oop * thread_oop_p);
292 
293   bool includes(JavaThread* p) {
294     return list()->includes(p);
295   }
296 
297   uint length() const {
298     return list()->length();
299   }




300 };

302 // This stack allocated JavaThreadIterator is used to walk the
303 // specified ThreadsList using the following style:
304 //
305 //   JavaThreadIterator jti(t_list);
306 //   for (JavaThread *jt = jti.first(); jt != NULL; jt = jti.next()) {
307 //     ...
308 //   }
309 //
310 class JavaThreadIterator : public StackObj {
311   ThreadsList * _list;
312   uint _index;
313 
314 public:
315   JavaThreadIterator(ThreadsList *list) : _list(list), _index(0) {
316     assert(list != NULL, "ThreadsList must not be NULL.");
317   }
318 
319   JavaThread *first() {
320     _index = 0;
321     return _list->thread_at(_index);
322   }
323 
324   uint length() const {
325     return _list->length();
326   }
327 
328   ThreadsList *list() const {
329     return _list;
330   }
331 
332   JavaThread *next() {
333     if (++_index >= length()) {
334       return NULL;
335     }
336     return _list->thread_at(_index);
337   }
338 };

340 // This stack allocated ThreadsListHandle and JavaThreadIterator combo
341 // is used to walk the ThreadsList in the included ThreadsListHandle
342 // using the following style:
343 //
344 //   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *jt = jtiwh.next(); ) {
345 //     ...
346 //   }
347 //
348 class JavaThreadIteratorWithHandle : public StackObj {
349   ThreadsListHandle _tlh;
350   uint _index;
351 
352 public:
353   JavaThreadIteratorWithHandle() : _index(0) {}
354 
355   uint length() const {
356     return _tlh.length();
357   }
358 
359   ThreadsList *list() const {
360     return _tlh.list();
361   }
362 
363   JavaThread *next() {
364     if (_index >= length()) {
365       return NULL;
366     }
367     return _tlh.list()->thread_at(_index++);
368   }
369 
370   void rewind() {
371     _index = 0;
372   }
373 };
