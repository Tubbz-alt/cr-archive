25 #ifndef SHARE_RUNTIME_SERVICETHREAD_HPP
26 #define SHARE_RUNTIME_SERVICETHREAD_HPP
27 
28 #include "prims/jvmtiImpl.hpp"
29 #include "runtime/thread.hpp"
30 
31 // A hidden from external view JavaThread for JVMTI compiled-method-load
32 // events, oop storage cleanup, and the maintainance of string, symbol,
33 // protection domain, and resolved method tables.
34 class JvmtiDeferredEvent;
35 
36 class ServiceThread : public JavaThread {
37   friend class VMStructs;
38  private:
39   static ServiceThread* _instance;
40   static JvmtiDeferredEvent* _jvmti_event;
41   static JvmtiDeferredEventQueue _jvmti_service_queue;
42 
43   static void service_thread_entry(JavaThread* thread, TRAPS);
44   ServiceThread(ThreadFunction entry_point) : JavaThread(entry_point) {};
45 
46  public:
47   static void initialize();
48 
49   // Hide this thread from external view.
50   bool is_hidden_from_external_view() const      { return true; }
51   bool is_service_thread() const                 { return true; }
52 
53   // Add event to the service thread event queue.
54   static void enqueue_deferred_event(JvmtiDeferredEvent* event);
55 
56   // GC support
57   void oops_do(OopClosure* f, CodeBlobClosure* cf);
58   void nmethods_do(CodeBlobClosure* cf);
59 };
60 
61 #endif // SHARE_RUNTIME_SERVICETHREAD_HPP
