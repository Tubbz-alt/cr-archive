25 #ifndef SHARE_CODE_OOPRECORDER_INLINE_HPP
26 #define SHARE_CODE_OOPRECORDER_INLINE_HPP
27 
28 #include "code/oopRecorder.hpp"
29 #include "memory/universe.hpp"
30 
31 template <class T>
32 bool ValueRecorder<T>::is_real(T h) {
33   return h != NULL && h != (T)Universe::non_oop_word();
34 }
35 
36 bool OopRecorder::is_real(jobject h) {
37   return _oops.is_real(h);
38 }
39 
40 bool OopRecorder::is_real(Metadata* h) {
41   return _metadata.is_real(h);
42 }
43 
44 #endif // SHARE_CODE_OOPRECORDER_INLINE_HPP
