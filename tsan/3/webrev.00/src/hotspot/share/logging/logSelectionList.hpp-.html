24 #ifndef SHARE_LOGGING_LOGSELECTIONLIST_HPP
25 #define SHARE_LOGGING_LOGSELECTIONLIST_HPP

27 #include "logging/logConfiguration.hpp"
28 #include "logging/logSelection.hpp"
29 #include "logging/logTag.hpp"
30 #include "memory/allocation.hpp"
31 #include "utilities/debug.hpp"
32 #include "utilities/globalDefinitions.hpp"
33 
34 class LogTagSet;
35 
36 // Class used to temporary encode a series of log selections during log configuration.
37 // Consists of ordered LogSelections, i.e. "tag1+tag2=level1,tag3*=level2".
38 class LogSelectionList : public StackObj {
39  public:
40   static const size_t MaxSelections = 256;
41 
42  private:
43   friend void LogConfiguration::configure_stdout(LogLevelType, int, ...);
44 
45   size_t _nselections;
46   LogSelection _selections[MaxSelections];
47 
48  public:
49   LogSelectionList() : _nselections(0) {
50   }
51 
52   LogSelectionList(const LogSelection& selection) : _nselections(1) {
53     _selections[0] = selection;
54   }
55 
56   bool parse(const char* str, outputStream* errstream = NULL);
57   LogLevelType level_for(const LogTagSet& ts) const;
58 
59   // Verify that each selection actually selects something.
60   // Returns false if some invalid selection was found. If given an outputstream,
61   // this function will list all the invalid selections on the stream.
62   bool verify_selections(outputStream* out = NULL) const;
63 };
64 
65 #endif // SHARE_LOGGING_LOGSELECTIONLIST_HPP
