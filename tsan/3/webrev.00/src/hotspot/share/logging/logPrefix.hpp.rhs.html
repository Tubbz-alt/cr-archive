 30 // Prefixes prepend each log message for a specified tagset with a given prefix.
 31 // These prefixes are written before the log message but after the log decorations.
 32 //
 33 // A prefix is defined as a function that takes a buffer (with some size) as argument.
 34 // This function will be called for each log message, and should write the prefix
 35 // to the given buffer. The function should return how many characters it wrote,
 36 // which should never exceed the given size.

 39 // Syntax: LOG_PREFIX(<name of prefixer function>, LOG_TAGS(<chosen log tags>))
 40 // Where the prefixer function matches the following signature: size_t (*)(char*, size_t)

 99 // The empty prefix, used when there's no prefix defined.
100 template <LogTagType T0, LogTagType T1, LogTagType T2, LogTagType T3, LogTagType T4, LogTagType GuardTag = LogTag::__NO_TAG>
101 struct LogPrefix : public AllStatic {
102   STATIC_ASSERT(GuardTag == LogTag::__NO_TAG);
103   static size_t prefix(char* buf, size_t len) {
104     return 0;
105   }
106 };
107 
108 #define LOG_PREFIX(fn, ...) \
109 template <> struct LogPrefix<__VA_ARGS__> { \
110   static size_t prefix(char* buf, size_t len) { \
111     size_t ret = fn(buf, len); \
112     /* Either prefix did fit (strlen(buf) == ret && ret < len) */ \
113     /* or the prefix didn't fit in buffer (ret > len && strlen(buf) < len) */ \
114     assert(ret == 0 || strlen(buf) < len, \
115            "Buffer overrun by prefix function."); \
116     assert(ret == 0 || strlen(buf) == ret || ret >= len, \
117            "Prefix function should return length of prefix written," \
118            " or the intended length of prefix if the buffer was too small."); \
119     return ret; \
120   } \
121 };
