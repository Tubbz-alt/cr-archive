185 #define PREFIX_LOG_TAG(T) (LogTag::_##T)
186 
187 // Expand a set of log tags to their prefixed names.
188 // For error detection purposes, the macro passes one more tag than what is supported.
189 // If too many tags are given, a static assert in the log class will fail.
190 #define LOG_TAGS_EXPANDED(T0, T1, T2, T3, T4, T5, ...)  PREFIX_LOG_TAG(T0), PREFIX_LOG_TAG(T1), PREFIX_LOG_TAG(T2), \
191                                                         PREFIX_LOG_TAG(T3), PREFIX_LOG_TAG(T4), PREFIX_LOG_TAG(T5)
192 // The EXPAND_VARARGS macro is required for MSVC, or it will resolve the LOG_TAGS_EXPANDED macro incorrectly.
193 #define EXPAND_VARARGS(x) x
194 #define LOG_TAGS(...) EXPAND_VARARGS(LOG_TAGS_EXPANDED(__VA_ARGS__, _NO_TAG, _NO_TAG, _NO_TAG, _NO_TAG, _NO_TAG, _NO_TAG))
195 
196 // Log tags are used to classify log messages.
197 // Each log message can be assigned between 1 to LogTag::MaxTags number of tags.
198 // Specifying multiple tags for a log message means that only outputs configured
199 // for those exact tags, or a subset of the tags with a wildcard, will see the logging.
200 // Multiple tags should be comma separated, e.g. log_error(tag1, tag2)("msg").
201 class LogTag : public AllStatic {
202  public:
203   // The maximum number of tags that a single log message can have.
204   // E.g. there might be hundreds of different tags available,
205   // but a specific log message can only be tagged with up to MaxTags of those.
206   static const size_t MaxTags = 5;
207 
208   enum type {
209     __NO_TAG,
210 #define LOG_TAG(name) _##name,
211     LOG_TAG_LIST
212 #undef LOG_TAG
213     Count
214   };
215 
216   static const char* name(LogTag::type tag) {
217     return _name[tag];
218   }
219 
220   static LogTag::type from_string(const char *str);
221   static LogTag::type fuzzy_match(const char *tag);
222   static void list_tags(outputStream* out);
223 
224  private:
225   static const char* _name[];
226 };
