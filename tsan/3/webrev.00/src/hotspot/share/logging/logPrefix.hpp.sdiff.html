102 #define LOG_PREFIX(fn, ...) \
103 template <> struct LogPrefix<__VA_ARGS__> { \
104   static size_t prefix(char* buf, size_t len) { \
105     size_t ret = fn(buf, len); \
106     /* Either prefix did fit (strlen(buf) == ret && ret < len) */ \
107     /* or the prefix didn't fit in buffer (ret > len && strlen(buf) < len) */ \
108     assert(ret == 0 || strlen(buf) < len, \
109            "Buffer overrun by prefix function."); \
