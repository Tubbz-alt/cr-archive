 36 class SimpleThreadStackSite;

 38 class SimpleThreadStack {
 39   friend class SimpleThreadStackSite;
 40 private:
 41   address _base;
 42   size_t  _size;
 43 public:
 44   SimpleThreadStack() : _base(NULL), _size(0) { }
 45   bool equals(const SimpleThreadStack& s) const {
 46     return base() == s.base();
 47   }
 48 
 49   size_t  size() const { return _size; }
 50   address base() const { return _base; }
 51 private:
 52   void set_size(size_t size)  { _size = size; }
 53   void set_base(address base) { _base = base; }
 54 };
 55 
 56 class SimpleThreadStackSite : public AllocationSite<SimpleThreadStack> {
 57 public:
 58   SimpleThreadStackSite(address base, size_t size, const NativeCallStack& stack) :
 59     AllocationSite<SimpleThreadStack>(stack, mtThreadStack) {
 60     data()->set_size(size);
 61     data()->set_base(base);
 62   }
 63 
 64   SimpleThreadStackSite(address base, size_t size) :
 65     AllocationSite<SimpleThreadStack>(NativeCallStack::empty_stack(), mtThreadStack) {
 66     data()->set_base(base);
 67     data()->set_size(size);
 68   }
 69 
 70   bool equals(const SimpleThreadStackSite& mts) const {
 71     bool eq = base() == mts.base();
 72     assert(!eq || size() == mts.size(), "Must match");
 73     return eq;
 74   }
 75 
 76   size_t  size() const { return peek()->size(); }
 77   address base() const { return peek()->base(); }
 78 };
 79 
 80   /*
 81    * Most of platforms, that hotspot support, have their thread stacks backed by
 82    * virtual memory by default. For these cases, thread stack tracker simply
 83    * delegates tracking to virtual memory tracker.
 84    * However, there are exceptions, (e.g. AIX), that platforms can provide stacks
 85    * that are not page aligned. A hypothetical VM implementation, it can provide
 86    * it own stacks. In these case, track_as_vm() should return false and manage
 87    * stack tracking by this tracker internally.
 88    * During memory snapshot, tracked thread stacks memory data is walked and stored
 89    * along with malloc'd data inside baseline. The regions are not scanned and assumed
 90    * all committed for now. Can add scanning phase when there is a need.
 91    */
 92 class ThreadStackTracker : AllStatic {
 93 private:
 94   static volatile size_t _thread_count;
 95 
 96   static int compare_thread_stack_base(const SimpleThreadStackSite& s1, const SimpleThreadStackSite& s2);
 97   static SortedLinkedList<SimpleThreadStackSite, compare_thread_stack_base>* _simple_thread_stacks;
 98 public:
 99   // Late phase initialization
100   static bool late_initialize(NMT_TrackingLevel level);
101   static bool transition(NMT_TrackingLevel from, NMT_TrackingLevel to);
102 
103   static void new_thread_stack(void* base, size_t size, const NativeCallStack& stack);
104   static void delete_thread_stack(void* base, size_t size);
105 
106   static bool   track_as_vm()  { return AIX_ONLY(false) NOT_AIX(true); }
107   static size_t thread_count() { return _thread_count; }
108 
109   // Snapshot support. Piggyback thread stack data in malloc slot, NMT always handles
110   // thread stack slot specially since beginning.
111   static bool walk_simple_thread_stack_site(MallocSiteWalker* walker);
112 };
