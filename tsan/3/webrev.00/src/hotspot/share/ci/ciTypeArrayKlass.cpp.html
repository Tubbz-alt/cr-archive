25 #include "precompiled.hpp"
26 #include "ci/ciTypeArrayKlass.hpp"
27 #include "ci/ciUtilities.inline.hpp"
28 #include "memory/universe.hpp"
29 
30 // ciTypeArrayKlass
31 //
32 // This class represents a Klass* in the HotSpot virtual machine
33 // whose Klass part in a TypeArrayKlass.
34 
35 // ------------------------------------------------------------------
36 // ciTypeArrayKlass::ciTypeArrayKlass
37 ciTypeArrayKlass::ciTypeArrayKlass(Klass* k) : ciArrayKlass(k) {
38   assert(get_Klass()->is_typeArray_klass(), "wrong type");
39   assert(element_type() == get_TypeArrayKlass()->element_type(), "");
40 }
41 
42 // ------------------------------------------------------------------
43 // ciTypeArrayKlass::make_impl
44 //
45 // Implementation of make.
46 ciTypeArrayKlass* ciTypeArrayKlass::make_impl(BasicType t) {
47   Klass* k = Universe::typeArrayKlassObj(t);
48   return CURRENT_ENV->get_type_array_klass(k);
49 }
50 
51 // ------------------------------------------------------------------
52 // ciTypeArrayKlass::make
53 //
54 // Make an array klass corresponding to the specified primitive type.
55 ciTypeArrayKlass* ciTypeArrayKlass::make(BasicType t) {
56   GUARDED_VM_ENTRY(return make_impl(t);)
57 }
