30 // ciExceptionHandler
31 //
32 // This class represents an exception handler for a method.
33 
34 // ------------------------------------------------------------------
35 // ciExceptionHandler::catch_klass
36 //
37 // Get the exception klass that this handler catches.
38 ciInstanceKlass* ciExceptionHandler::catch_klass() {
39   VM_ENTRY_MARK;
40   assert(!is_catch_all(), "bad index");
41   if (_catch_klass == NULL) {
42     bool will_link;
43     assert(_loading_klass->get_instanceKlass()->is_linked(), "must be linked before accessing constant pool");
44     constantPoolHandle cpool(_loading_klass->get_instanceKlass()->constants());
45     ciKlass* k = CURRENT_ENV->get_klass_by_index(cpool,
46                                                  _catch_klass_index,
47                                                  will_link,
48                                                  _loading_klass);
49     if (!will_link && k->is_loaded()) {
50       GUARDED_VM_ENTRY(
51         k = CURRENT_ENV->get_unloaded_klass(_loading_klass, k->name());
52       )
53     }
54     _catch_klass = k->as_instance_klass();
55   }
56   return _catch_klass;
57 }
58 
59 // ------------------------------------------------------------------
60 // ciExceptionHandler::print()
61 void ciExceptionHandler::print() {
62   tty->print("<ciExceptionHandler start=%d limit=%d"
63              " handler_bci=%d ex_klass_index=%d",
64              start(), limit(), handler_bci(), catch_klass_index());
65   if (_catch_klass != NULL) {
66     tty->print(" ex_klass=");
67     _catch_klass->print();
68   }
69   tty->print(">");
70 }
