25 #include "precompiled.hpp"
26 #include "ci/ciTypeArrayKlass.hpp"
27 #include "ci/ciUtilities.inline.hpp"

28 
29 // ciTypeArrayKlass
30 //
31 // This class represents a Klass* in the HotSpot virtual machine
32 // whose Klass part in a TypeArrayKlass.
33 
34 // ------------------------------------------------------------------
35 // ciTypeArrayKlass::ciTypeArrayKlass
36 ciTypeArrayKlass::ciTypeArrayKlass(Klass* k) : ciArrayKlass(k) {
37   assert(get_Klass()->is_typeArray_klass(), "wrong type");
38   assert(element_type() == get_TypeArrayKlass()->element_type(), "");
39 }
40 
41 // ------------------------------------------------------------------
42 // ciTypeArrayKlass::make_impl
43 //
44 // Implementation of make.
45 ciTypeArrayKlass* ciTypeArrayKlass::make_impl(BasicType t) {
46   Klass* k = Universe::typeArrayKlassObj(t);
47   return CURRENT_ENV->get_type_array_klass(k);
48 }
49 
50 // ------------------------------------------------------------------
51 // ciTypeArrayKlass::make
52 //
53 // Make an array klass corresponding to the specified primitive type.
54 ciTypeArrayKlass* ciTypeArrayKlass::make(BasicType t) {
55   GUARDED_VM_ENTRY(return make_impl(t);)
56 }
