 30 // ------------------------------------------------------------------
 31 // ciSymbol::ciSymbol
 32 //
 33 // Preallocated symbol variant.  Used with symbols from vmSymbols.
 34 ciSymbol::ciSymbol(Symbol* s, vmSymbols::SID sid)
 35   : _symbol(s), _sid(sid)
 36 {
 37   assert(_symbol != NULL, "adding null symbol");
 38   _symbol->increment_refcount();  // increment ref count
 39   assert(sid_ok(), "must be in vmSymbols");
 40 }
 41 
 42 // Normal case for non-famous symbols.
 43 ciSymbol::ciSymbol(Symbol* s)
 44   : _symbol(s), _sid(vmSymbols::NO_SID)
 45 {
 46   assert(_symbol != NULL, "adding null symbol");
 47   _symbol->increment_refcount();  // increment ref count
 48   assert(sid_ok(), "must not be in vmSymbols");
 49 }

128 // ------------------------------------------------------------------
129 // ciSymbol::make_impl
130 //
131 // Make a ciSymbol from a C string (implementation).
132 ciSymbol* ciSymbol::make_impl(const char* s) {
133   EXCEPTION_CONTEXT;
134   TempNewSymbol sym = SymbolTable::new_symbol(s, THREAD);
135   if (HAS_PENDING_EXCEPTION) {
136     CLEAR_PENDING_EXCEPTION;
137     CURRENT_THREAD_ENV->record_out_of_memory_failure();
138     return ciEnv::_unloaded_cisymbol;
139   }
140   return CURRENT_THREAD_ENV->get_symbol(sym);
141 }
142 
143 // ------------------------------------------------------------------
144 // ciSymbol::make
145 //
146 // Make a ciSymbol from a C string.
147 ciSymbol* ciSymbol::make(const char* s) {
148   GUARDED_VM_ENTRY(return make_impl(s);)
149 }
