32 void ResourceArea::bias_to(MEMFLAGS new_flags) {
33   if (new_flags != _flags) {


34     MemTracker::record_arena_free(_flags);
35     MemTracker::record_new_arena(new_flags);

36     _flags = new_flags;
37   }
38 }

43 // The following routines are declared in allocation.hpp and used everywhere:
44 
45 // Allocation in thread-local resource area
46 extern char* resource_allocate_bytes(size_t size, AllocFailType alloc_failmode) {
47   return Thread::current()->resource_area()->allocate_bytes(size, alloc_failmode);
48 }
49 extern char* resource_allocate_bytes(Thread* thread, size_t size, AllocFailType alloc_failmode) {
50   return thread->resource_area()->allocate_bytes(size, alloc_failmode);
51 }
52 
53 extern char* resource_reallocate_bytes( char *old, size_t old_size, size_t new_size, AllocFailType alloc_failmode){
54   return (char*)Thread::current()->resource_area()->Arealloc(old, old_size, new_size, alloc_failmode);
55 }
56 
57 extern void resource_free_bytes( char *old, size_t size ) {
58   Thread::current()->resource_area()->Afree(old, size);
59 }
