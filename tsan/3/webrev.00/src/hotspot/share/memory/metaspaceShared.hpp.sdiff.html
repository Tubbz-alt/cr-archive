 52 
 53   // CDS support
 54   static ReservedSpace _shared_rs;
 55   static VirtualSpace _shared_vs;

 62   static size_t  _cds_i2i_entry_code_buffers_size;

 64  public:
 65   enum {
 66     // core archive spaces
 67     mc = 0,  // miscellaneous code for method trampolines
 68     rw = 1,  // read-write shared space in the heap
 69     ro = 2,  // read-only shared space in the heap
 70     md = 3,  // miscellaneous data for initializing tables, etc.
 71     num_core_spaces = 4, // number of non-string regions
 72     num_non_heap_spaces = 4,
 73 
 74     // mapped java heap regions
 75     first_closed_archive_heap_region = md + 1,
 76     max_closed_archive_heap_region = 2,
 77     last_closed_archive_heap_region = first_closed_archive_heap_region + max_closed_archive_heap_region - 1,
 78     first_open_archive_heap_region = last_closed_archive_heap_region + 1,
 79     max_open_archive_heap_region = 2,
 80     last_open_archive_heap_region = first_open_archive_heap_region + max_open_archive_heap_region - 1,
 81 
 82     last_valid_region = last_open_archive_heap_region,
 83     n_regions =  last_valid_region + 1 // total number of regions
 84   };
 85 
 86   static void prepare_for_dumping() NOT_CDS_RETURN;
 87   static void preload_and_dump(TRAPS) NOT_CDS_RETURN;

 90 
 91   static GrowableArray<Klass*>* collected_klasses();
 92 
 93   static ReservedSpace* shared_rs() {
 94     CDS_ONLY(return &_shared_rs);
 95     NOT_CDS(return NULL);
 96   }
 97   static void commit_shared_space_to(char* newtop) NOT_CDS_RETURN;
 98   static size_t core_spaces_size() {
 99     assert(DumpSharedSpaces || UseSharedSpaces, "sanity");
100     assert(_core_spaces_size != 0, "sanity");
101     return _core_spaces_size;
102   }


103   static void initialize_dumptime_shared_and_meta_spaces() NOT_CDS_RETURN;
104   static void initialize_runtime_shared_and_meta_spaces() NOT_CDS_RETURN;
105   static void post_initialize(TRAPS) NOT_CDS_RETURN;
106 
107   // Delta of this object from the bottom of the archive.
108   static uintx object_delta_uintx(void* obj) {
109     assert(DumpSharedSpaces, "supported only for dumping");
110     assert(shared_rs()->contains(obj), "must be");
111     address base_address = address(shared_rs()->base());
112     uintx deltax = address(obj) - base_address;
113     return deltax;
114   }
115 
116   static u4 object_delta_u4(void* obj) {
117     // offset is guaranteed to be less than MAX_SHARED_DELTA in DumpRegion::expand_top_to()
118     uintx deltax = object_delta_uintx(obj);
119     guarantee(deltax <= MAX_SHARED_DELTA, "must be 32-bit offset");
120     return (u4)deltax;
121   }
122 
123   static void set_archive_loading_failed() {

125   }

127   static void initialize_shared_spaces() NOT_CDS_RETURN;
128 
129   // Return true if given address is in the shared metaspace regions (i.e., excluding any
130   // mapped shared heap regions.)
131   static bool is_in_shared_metaspace(const void* p) {
132     // If no shared metaspace regions are mapped, MetaspceObj::_shared_metaspace_{base,top} will
133     // both be NULL and all values of p will be rejected quickly.
134     return (p < MetaspaceObj::shared_metaspace_top() && p >= MetaspaceObj::shared_metaspace_base());


135   }

140   static bool is_in_trampoline_frame(address addr) NOT_CDS_RETURN_(false);

143   static intptr_t* clone_cpp_vtables(intptr_t* p);


144   static void zero_cpp_vtable_clones_for_writing();
145   static void patch_cpp_vtable_pointers();


146   static bool is_valid_shared_method(const Method* m) NOT_CDS_RETURN_(false);
147   static void serialize(SerializeClosure* sc) NOT_CDS_RETURN;
148 
149   static MetaspaceSharedStats* stats() {
150     return &_stats;
151   }
152 
153   static void report_out_of_space(const char* name, size_t needed_bytes);
154 
155   // JVM/TI RedefineClasses() support:
156   // Remap the shared readonly space to shared readwrite, private if
157   // sharing is enabled. Simply returns true if sharing is not enabled
158   // or if the remapping has already been done by a prior call.
159   static bool remap_shared_readonly_as_readwrite() NOT_CDS_RETURN_(true);
160   static bool remapped_readwrite() {
161     CDS_ONLY(return _remapped_readwrite);
162     NOT_CDS(return false);
163   }
164 
165   static bool try_link_class(InstanceKlass* ik, TRAPS);
166   static void link_and_cleanup_shared_classes(TRAPS);
167 













168   // Allocate a block of memory from the "mc", "ro", or "rw" regions.
169   static char* misc_code_space_alloc(size_t num_bytes);
170   static char* read_only_space_alloc(size_t num_bytes);
171 
172   template <typename T>
173   static Array<T>* new_ro_array(int length) {
174 #if INCLUDE_CDS

179 #else

193

 57 #if INCLUDE_CDS
 58 class DumpRegion {
 59 private:
 60   const char* _name;
 61   char* _base;
 62   char* _top;
 63   char* _end;
 64   bool _is_packed;
 65 
 66 public:
 67   DumpRegion(const char* name) : _name(name), _base(NULL), _top(NULL), _end(NULL), _is_packed(false) {}
 68 
 69   char* expand_top_to(char* newtop);
 70   char* allocate(size_t num_bytes, size_t alignment=BytesPerWord);
 71 
 72   void append_intptr_t(intptr_t n, bool need_to_mark = false);
 73 
 74   char* base()      const { return _base;        }
 75   char* top()       const { return _top;         }
 76   char* end()       const { return _end;         }
 77   size_t reserved() const { return _end - _base; }
 78   size_t used()     const { return _top - _base; }
 79   bool is_packed()  const { return _is_packed;   }
 80   bool is_allocatable() const {
 81     return !is_packed() && _base != NULL;
 82   }
 83 
 84   void print(size_t total_bytes) const;
 85   void print_out_of_space_msg(const char* failing_region, size_t needed_bytes);
 86 
 87   void init(const ReservedSpace* rs, char* base) {
 88     if (base == NULL) {
 89       base = rs->base();
 90     }
 91     assert(rs->contains(base), "must be");
 92     _base = _top = base;
 93     _end = rs->end();
 94   }
 95   void init(char* b, char* t, char* e) {
 96     _base = b;
 97     _top = t;
 98     _end = e;
 99   }
100 
101   void pack(DumpRegion* next = NULL);
102 
103   bool contains(char* p) {
104     return base() <= p && p < top();
105   }
106 };
