 42 typeArrayOop oopFactory::new_charArray(const char* utf8_str, TRAPS) {
 43   int length = utf8_str == NULL ? 0 : UTF8::unicode_length(utf8_str);
 44   typeArrayOop result = new_charArray(length, CHECK_NULL);
 45   if (length > 0) {
 46     UTF8::convert_to_unicode(utf8_str, result->char_at_addr(0), length);
 47   }
 48   return result;
 49 }
