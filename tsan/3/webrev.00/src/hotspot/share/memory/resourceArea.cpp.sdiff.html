32 void ResourceArea::bias_to(MEMFLAGS new_flags) {
33   if (new_flags != _flags) {
34     size_t size = size_in_bytes();
35     MemTracker::record_arena_size_change(-ssize_t(size), _flags);
36     MemTracker::record_arena_free(_flags);
37     MemTracker::record_new_arena(new_flags);
38     MemTracker::record_arena_size_change(ssize_t(size), new_flags);
39     _flags = new_flags;
40   }
41 }
