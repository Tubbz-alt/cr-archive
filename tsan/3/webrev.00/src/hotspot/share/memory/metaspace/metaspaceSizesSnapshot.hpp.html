26 #ifndef SHARE_MEMORY_METASPACE_METASPACESIZESSNAPSHOT_HPP
27 #define SHARE_MEMORY_METASPACE_METASPACESIZESSNAPSHOT_HPP
28 
29 namespace metaspace {
30 
31 class MetaspaceSizesSnapshot {
32 public:
33   MetaspaceSizesSnapshot();
34 
35   size_t used() const { return _used; }
36   size_t committed() const { return _committed; }
37   size_t non_class_used() const { return _non_class_used; }
38   size_t non_class_committed() const { return _non_class_committed; }
39   size_t class_used() const { return _class_used; }
40   size_t class_committed() const { return _class_committed; }
41 
42 private:
43   const size_t _used;
44   const size_t _committed;
45   const size_t _non_class_used;
46   const size_t _non_class_committed;
47   const size_t _class_used;
48   const size_t _class_committed;
49 };
50 
51 } // namespace metaspace
