36 PrintMetaspaceInfoKlassClosure::PrintMetaspaceInfoKlassClosure(outputStream* out, bool do_print)
37 : _out(out), _do_print(do_print)
38 , _num_classes(0), _num_instance_classes(0), _num_array_classes(0) {
39 }
40 
41 void PrintMetaspaceInfoKlassClosure::do_klass(Klass* k) {
42   _num_classes ++;
43   if (k->is_instance_klass()) {
44     _num_instance_classes ++;
45   } else if (k->is_array_klass()) {
46     _num_array_classes ++;
47   }
48   if (_do_print) {
49     _out->cr_indent();
50     _out->print(UINTX_FORMAT_W(4) ": ", _num_classes);
51     ResourceMark rm;
52     _out->print("%s", k->external_name());
53 
54     // Special treatment for generated core reflection accessor classes: print invocation target.
55     if (ReflectionAccessorImplKlassHelper::is_generated_accessor(k)) {
56       _out->print(" (invokes: ");
57       ReflectionAccessorImplKlassHelper::print_invocation_target(_out, k);
58       _out->print(")");
59     }
60 





61   }
62 }
