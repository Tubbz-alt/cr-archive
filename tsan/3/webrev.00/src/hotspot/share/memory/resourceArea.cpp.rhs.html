32 void ResourceArea::bias_to(MEMFLAGS new_flags) {
33   if (new_flags != _flags) {
34     size_t size = size_in_bytes();
35     MemTracker::record_arena_size_change(-ssize_t(size), _flags);
36     MemTracker::record_arena_free(_flags);
37     MemTracker::record_new_arena(new_flags);
38     MemTracker::record_arena_size_change(ssize_t(size), new_flags);
39     _flags = new_flags;
40   }
41 }

46 // The following routines are declared in allocation.hpp and used everywhere:
47 
48 // Allocation in thread-local resource area
49 extern char* resource_allocate_bytes(size_t size, AllocFailType alloc_failmode) {
50   return Thread::current()->resource_area()->allocate_bytes(size, alloc_failmode);
51 }
52 extern char* resource_allocate_bytes(Thread* thread, size_t size, AllocFailType alloc_failmode) {
53   return thread->resource_area()->allocate_bytes(size, alloc_failmode);
54 }
55 
56 extern char* resource_reallocate_bytes( char *old, size_t old_size, size_t new_size, AllocFailType alloc_failmode){
57   return (char*)Thread::current()->resource_area()->Arealloc(old, old_size, new_size, alloc_failmode);
58 }
59 
60 extern void resource_free_bytes( char *old, size_t size ) {
61   Thread::current()->resource_area()->Afree(old, size);
62 }
