 49 // A dump time sub-graph info for Klass _k. It includes the entry points
 50 // (static fields in _k's mirror) of the archived sub-graphs reachable
 51 // from _k's mirror. It also contains a list of Klasses of the objects
 52 // within the sub-graphs.

108  public:

110     _k(NULL), _entry_field_records(NULL), _subgraph_object_klasses(NULL) {}

112   Klass* klass() const { return _k; }

118 class HeapShared: AllStatic {

120  private:
121 
122 #if INCLUDE_CDS_JAVA_HEAP

124   static bool _open_archive_heap_region_mapped;
125   static bool _archive_heap_region_fixed;
126 
127   static bool oop_equals(oop const& p1, oop const& p2) {
128     return oopDesc::equals(p1, p2);
129   }
130   static unsigned oop_hash(oop const& p);
131 
132   typedef ResourceHashtable<oop, oop,
133       HeapShared::oop_hash,
134       HeapShared::oop_equals,
135       15889, // prime number
136       ResourceObj::C_HEAP> ArchivedObjectCache;
137   static ArchivedObjectCache* _archived_object_cache;
138 
139   static bool klass_equals(Klass* const& p1, Klass* const& p2) {
140     return primitive_equals<Klass*>(p1, p2);
141   }
142 
143   static unsigned klass_hash(Klass* const& klass) {
144     return primitive_hash<address>((address)klass);
145   }
146 
147   class DumpTimeKlassSubGraphInfoTable
148     : public ResourceHashtable<Klass*, KlassSubGraphInfo,

269   static oop archive_reachable_objects_from(int level,
270                                             KlassSubGraphInfo* subgraph_info,
271                                             oop orig_obj,
272                                             bool is_closed_archive,
273                                             TRAPS);
274 
275   static ResourceBitMap calculate_oopmap(MemRegion region);
276 #endif // INCLUDE_CDS_JAVA_HEAP
277 
278  public:
279   static bool is_heap_object_archiving_allowed() {
280     CDS_JAVA_HEAP_ONLY(return (UseG1GC && UseCompressedOops && UseCompressedClassPointers);)
281     NOT_CDS_JAVA_HEAP(return false;)
282   }
283 
284   static bool is_heap_region(int idx) {
285     CDS_JAVA_HEAP_ONLY(return (idx >= MetaspaceShared::first_closed_archive_heap_region &&
286                                idx <= MetaspaceShared::last_open_archive_heap_region));
287     NOT_CDS_JAVA_HEAP_RETURN_(false);
288   }
289   static bool is_closed_archive_heap_region(int idx) {
290     CDS_JAVA_HEAP_ONLY(return (idx >= MetaspaceShared::first_closed_archive_heap_region &&
291                                idx <= MetaspaceShared::last_closed_archive_heap_region));
292     NOT_CDS_JAVA_HEAP_RETURN_(false);
293   }
294   static bool is_open_archive_heap_region(int idx) {
295     CDS_JAVA_HEAP_ONLY(return (idx >= MetaspaceShared::first_open_archive_heap_region &&
296                                idx <= MetaspaceShared::last_open_archive_heap_region));
297     NOT_CDS_JAVA_HEAP_RETURN_(false);
298   }
299 
300   static void set_closed_archive_heap_region_mapped() {
301     CDS_JAVA_HEAP_ONLY(_closed_archive_heap_region_mapped = true);
302     NOT_CDS_JAVA_HEAP_RETURN;
303   }
304   static bool closed_archive_heap_region_mapped() {
305     CDS_JAVA_HEAP_ONLY(return _closed_archive_heap_region_mapped);
306     NOT_CDS_JAVA_HEAP_RETURN_(false);
307   }
308   static void set_open_archive_heap_region_mapped() {
309     CDS_JAVA_HEAP_ONLY(_open_archive_heap_region_mapped = true);
310     NOT_CDS_JAVA_HEAP_RETURN;
311   }
312   static bool open_archive_heap_region_mapped() {
313     CDS_JAVA_HEAP_ONLY(return _open_archive_heap_region_mapped);
314     NOT_CDS_JAVA_HEAP_RETURN_(false);
315   }
316 
317   static void fixup_mapped_heap_regions() NOT_CDS_JAVA_HEAP_RETURN;
318 
319   inline static bool is_archived_object(oop p) NOT_CDS_JAVA_HEAP_RETURN_(false);
320 
321   static void initialize_from_archived_subgraph(Klass* k) NOT_CDS_JAVA_HEAP_RETURN;
322 
323   // NarrowOops stored in the CDS archive may use a different encoding scheme
324   // than Universe::narrow_oop_{base,shift} -- see FileMapInfo::map_heap_regions_impl.
325   // To decode them, do not use CompressedOops::decode_not_null. Use this
326   // function instead.

328 
329   static void init_narrow_oop_decoding(address base, int shift) NOT_CDS_JAVA_HEAP_RETURN;
330 
331   static void patch_archived_heap_embedded_pointers(MemRegion mem, address  oopmap,
332                                                     size_t oopmap_in_bits) NOT_CDS_JAVA_HEAP_RETURN;
333 
334   static void init_subgraph_entry_fields(Thread* THREAD) NOT_CDS_JAVA_HEAP_RETURN;
335   static void write_subgraph_info_table() NOT_CDS_JAVA_HEAP_RETURN;
336   static void serialize_subgraph_info_table_header(SerializeClosure* soc) NOT_CDS_JAVA_HEAP_RETURN;
337 };
