25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHROOTVERIFIER_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHROOTVERIFIER_HPP
27 
28 #include "memory/allocation.hpp"
29 #include "memory/iterator.hpp"
30 
31 class ShenandoahRootVerifier : public StackObj {
32 public:
33   enum RootTypes {
34     None                = 0,
35     SerialRoots         = 1 << 0,
36     ThreadRoots         = 1 << 1,
37     CodeRoots           = 1 << 2,
38     CLDGRoots           = 1 << 3,
39     SerialWeakRoots     = 1 << 4,
40     ConcurrentWeakRoots = 1 << 5,
41     WeakRoots           = (SerialWeakRoots | ConcurrentWeakRoots),
42     StringDedupRoots    = 1 << 6,
43     JNIHandleRoots      = 1 << 7,
44     AllRoots            = (SerialRoots | ThreadRoots | CodeRoots | CLDGRoots | WeakRoots | StringDedupRoots | JNIHandleRoots)
45   };
46 
47 private:
48   RootTypes _types;
49 
50 public:
51   ShenandoahRootVerifier(RootTypes types = AllRoots);
52 
53   void excludes(RootTypes types);
54   void oops_do(OopClosure* cl);
55 
56   // Used to seed ShenandoahVerifier, do not honor root type filter
57   void roots_do(OopClosure* cl);
58   void strong_roots_do(OopClosure* cl);
59 
60   static RootTypes combine(RootTypes t1, RootTypes t2);
61 private:
62   bool verify(RootTypes type) const;
63 
64   void serial_weak_roots_do(OopClosure* cl);
65   void concurrent_weak_roots_do(OopClosure* cl);
66 };
67 
68 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHROOTVERIFIER_HPP
