37 template <class T, bool STRING_DEDUP, bool DEGEN>
38 void ShenandoahTraversalGC::process_oop(T* p, Thread* thread, ShenandoahObjToScanQueue* queue, ShenandoahMarkingContext* const mark_context) {
39   T o = RawAccess<>::oop_load(p);
40   if (!CompressedOops::is_null(o)) {
41     oop obj = CompressedOops::decode_not_null(o);
42     if (DEGEN) {

43       oop forw = ShenandoahBarrierSet::resolve_forwarded_not_null(obj);
44       if (!oopDesc::equals_raw(obj, forw)) {
45         // Update reference.
46         RawAccess<IS_NOT_NULL>::oop_store(p, forw);
47       }
48       obj = forw;
49     } else if (_heap->in_collection_set(obj)) {
50       oop forw = ShenandoahBarrierSet::resolve_forwarded_not_null(obj);
51       if (oopDesc::equals_raw(obj, forw)) {

52         forw = _heap->evacuate_object(obj, thread);
53       }
54       shenandoah_assert_forwarded_except(p, obj, _heap->cancelled_gc());
55       // Update reference.
56       _heap->atomic_compare_exchange_oop(forw, p, obj);




57       obj = forw;
58     }
59 
60     shenandoah_assert_not_forwarded(p, obj);
61     shenandoah_assert_not_in_cset_except(p, obj, _heap->cancelled_gc());
62 
63     if (mark_context->mark(obj)) {
64       bool succeeded = queue->push(ShenandoahMarkTask(obj));
65       assert(succeeded, "must succeed to push to task queue");
66 
67       if (STRING_DEDUP && ShenandoahStringDedup::is_candidate(obj) && !_heap->cancelled_gc()) {
68         assert(ShenandoahStringDedup::is_enabled(), "Must be enabled");
69         // Only dealing with to-space string, so that we can avoid evac-oom protocol, which is costly here.
70         shenandoah_assert_not_in_cset(p, obj);
71         ShenandoahStringDedup::enqueue_candidate(obj);
72       }
73     }
74   }
75 }
