 57 void ShenandoahNMethod::disarm_nmethod(nmethod* nm) {
 58   if (!ShenandoahConcurrentRoots::can_do_concurrent_class_unloading()) {
 59     return;
 60   }
 61 
 62   BarrierSetNMethod* const bs = BarrierSet::barrier_set()->barrier_set_nmethod();
 63   assert(bs != NULL, "Sanity");
 64   if (bs->is_armed(nm)) {
 65     bs->disarm(nm);
 66   }
 67 }

 85 template<bool CSET_FILTER>
 86 void ShenandoahNMethodTableSnapshot::parallel_blobs_do(CodeBlobClosure *f) {
 87   size_t stride = 256; // educated guess
 88 
 89   ShenandoahNMethod** const list = _array;
 90 
 91   size_t max = (size_t)_length;
 92   while (_claimed < max) {
 93     size_t cur = Atomic::fetch_and_add(&_claimed, stride);
 94     size_t start = cur;
 95     size_t end = MIN2(cur + stride, max);
 96     if (start >= max) break;
 97 
 98     for (size_t idx = start; idx < end; idx++) {
 99       ShenandoahNMethod* nmr = list[idx];
100       assert(nmr != NULL, "Sanity");
101       if (nmr->is_unregistered()) {
102         continue;
103       }
104 
105       nmr->assert_alive_and_correct();
106 
107       if (CSET_FILTER && !nmr->has_cset_oops(_heap)) {
108         continue;
109       }
110 
111       f->do_code_blob(nmr->nm());
112     }
113   }
114 }
