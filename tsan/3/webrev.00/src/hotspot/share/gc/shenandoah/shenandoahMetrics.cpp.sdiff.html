136 bool ShenandoahMetricsSnapshot::is_good_progress(const char *label) {
137   // Under the critical threshold? Declare failure.
138   size_t free_actual   = _heap->free_set()->available();
139   size_t free_expected = _heap->max_capacity() / 100 * ShenandoahCriticalFreeThreshold;
140   if (free_actual < free_expected) {
141     log_info(gc, ergo)("Not enough free space (" SIZE_FORMAT "M, need " SIZE_FORMAT "M) after %s",
142                        free_actual / M, free_expected / M, label);



143     return false;
144   }
145 
146   // Freed up enough? Good! Declare victory.
147   size_t progress_actual   = (_used_before > _used_after) ? _used_before - _used_after : 0;
148   size_t progress_expected = ShenandoahHeapRegion::region_size_bytes();
149   if (progress_actual >= progress_expected) {





150     return true;
151   }
152   log_info(gc,ergo)("Not enough progress (" SIZE_FORMAT "M, need " SIZE_FORMAT "M) after %s",
153                     progress_actual / M, progress_expected / M, label);
154 
155   // Internal fragmentation is down? Good! Declare victory.
156   double if_actual = _if_before - _if_after;
157   double if_expected = 0.01; // 1% should be enough
158   if (if_actual > if_expected) {




159     return true;
160   }
161   log_info(gc,ergo)("Not enough internal fragmentation improvement (%.1f%%, need %.1f%%) after %s",
162                     if_actual * 100, if_expected * 100, label);
163 
164   // External fragmentation is down? Good! Declare victory.
165   double ef_actual = _ef_before - _ef_after;
166   double ef_expected = 0.01; // 1% should be enough
167   if (ef_actual > ef_expected) {




168     return true;
169   }
170   log_info(gc,ergo)("Not enough external fragmentation improvement (%.1f%%, need %.1f%%) after %s",
171                     if_actual * 100, if_expected * 100, label);
172 
173   // Nothing good had happened.
174   return false;
175 }

131 bool ShenandoahMetricsSnapshot::is_good_progress() {
132   // Under the critical threshold?






133   size_t free_actual   = _heap->free_set()->available();
134   size_t free_expected = _heap->max_capacity() / 100 * ShenandoahCriticalFreeThreshold;
135   bool prog_free = free_actual >= free_expected;
136   log_info(gc, ergo)("%s progress for free space: " SIZE_FORMAT "%s, need " SIZE_FORMAT "%s",
137                      prog_free ? "Good" : "Bad",
138                      byte_size_in_proper_unit(free_actual),   proper_unit_for_byte_size(free_actual),
139                      byte_size_in_proper_unit(free_expected), proper_unit_for_byte_size(free_expected));
140   if (!prog_free) {
141     return false;
142   }
143 
144   // Freed up enough?
145   size_t progress_actual   = (_used_before > _used_after) ? _used_before - _used_after : 0;
146   size_t progress_expected = ShenandoahHeapRegion::region_size_bytes();
147   bool prog_used = progress_actual >= progress_expected;
148   log_info(gc, ergo)("%s progress for used space: " SIZE_FORMAT "%s, need " SIZE_FORMAT "%s",
149                      prog_used ? "Good" : "Bad",
150                      byte_size_in_proper_unit(progress_actual),   proper_unit_for_byte_size(progress_actual),
151                      byte_size_in_proper_unit(progress_expected), proper_unit_for_byte_size(progress_expected));
152   if (prog_used) {
153     return true;
154   }


155 
156   // Internal fragmentation is down?
157   double if_actual = _if_before - _if_after;
158   double if_expected = 0.01; // 1% should be enough
159   bool prog_if = if_actual >= if_expected;
160   log_info(gc, ergo)("%s progress for internal fragmentation: %.1f%%, need %.1f%%",
161                      prog_if ? "Good" : "Bad",
162                      if_actual * 100, if_expected * 100);
163   if (prog_if) {
164     return true;
165   }


166 
167   // External fragmentation is down?
168   double ef_actual = _ef_before - _ef_after;
169   double ef_expected = 0.01; // 1% should be enough
170   bool prog_ef = ef_actual >= ef_expected;
171   log_info(gc, ergo)("%s progress for external fragmentation: %.1f%%, need %.1f%%",
172                      prog_ef ? "Good" : "Bad",
173                      ef_actual * 100, ef_expected * 100);
174   if (prog_ef) {
175     return true;
176   }


177 
178   // Nothing good had happened.
179   return false;
180 }
