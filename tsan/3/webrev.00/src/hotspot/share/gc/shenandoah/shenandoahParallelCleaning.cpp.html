34 ShenandoahClassUnloadingTask::ShenandoahClassUnloadingTask(BoolObjectClosure* is_alive,
35                                                            uint num_workers,
36                                                            bool unloading_occurred) :
37   AbstractGangTask("Parallel Class Unloading Task"),
38   _unloading_occurred(unloading_occurred),
39   _code_cache_task(num_workers, is_alive, unloading_occurred),
40   _klass_cleaning_task() {
41   assert(SafepointSynchronize::is_at_safepoint(), "Must be at a safepoint");
42 }
43 
44 void ShenandoahClassUnloadingTask::work(uint worker_id) {
45   _code_cache_task.work(worker_id);
46   // Clean all klasses that were not unloaded.
47   // The weak metadata in klass doesn't need to be
48   // processed if there was no unloading.
49   if (_unloading_occurred) {
50     _klass_cleaning_task.work();
51   }
52 }
