24 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHALLOCTRACKER_HPP
25 #define SHARE_GC_SHENANDOAH_SHENANDOAHALLOCTRACKER_HPP
26 
27 #include "gc/shenandoah/shenandoahAllocRequest.hpp"
28 #include "gc/shenandoah/shenandoahNumberSeq.hpp"
29 #include "memory/allocation.hpp"
30 #include "utilities/ostream.hpp"
31 
32 class ShenandoahAllocTracker : public CHeapObj<mtGC> {
33 private:
34   BinaryMagnitudeSeq _alloc_size[ShenandoahAllocRequest::_ALLOC_LIMIT];
35   BinaryMagnitudeSeq _alloc_latency[ShenandoahAllocRequest::_ALLOC_LIMIT];
36 
37 public:
38   void record_alloc_latency(size_t words_size,
39                             ShenandoahAllocRequest::Type _alloc_type,
40                             double latency_us) {
41     _alloc_size[_alloc_type].add(words_size);
42     _alloc_latency[_alloc_type].add((size_t)latency_us);
43   }
44 
45   void print_on(outputStream* out) const;
46 };
47 
48 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHALLOCTRACKER_HPP
