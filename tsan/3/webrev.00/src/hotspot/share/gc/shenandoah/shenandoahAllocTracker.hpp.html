25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHALLOCTRACKER_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHALLOCTRACKER_HPP
27 
28 #include "gc/shenandoah/shenandoahAllocRequest.hpp"
29 #include "gc/shenandoah/shenandoahNumberSeq.hpp"
30 #include "memory/allocation.hpp"
31 #include "utilities/ostream.hpp"
32 
33 class ShenandoahAllocTracker : public CHeapObj<mtGC> {
34 private:
35   BinaryMagnitudeSeq _alloc_size[ShenandoahAllocRequest::_ALLOC_LIMIT];
36   BinaryMagnitudeSeq _alloc_latency[ShenandoahAllocRequest::_ALLOC_LIMIT];
37 
38 public:
39   void record_alloc_latency(size_t words_size,
40                             ShenandoahAllocRequest::Type _alloc_type,
41                             double latency_us) {
42     _alloc_size[_alloc_type].add(words_size);
43     _alloc_latency[_alloc_type].add((size_t)latency_us);
44   }
45 
46   void print_on(outputStream* out) const;
47 };
48 
49 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHALLOCTRACKER_HPP
