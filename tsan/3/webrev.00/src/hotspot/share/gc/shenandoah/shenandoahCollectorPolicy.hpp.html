33 class ShenandoahCollectorPolicy : public CHeapObj<mtGC> {
34 private:
35   size_t _success_concurrent_gcs;
36   size_t _success_degenerated_gcs;
37   size_t _success_full_gcs;
38   size_t _alloc_failure_degenerated;
39   size_t _alloc_failure_degenerated_upgrade_to_full;
40   size_t _alloc_failure_full;
41   size_t _explicit_concurrent;
42   size_t _explicit_full;
43   size_t _implicit_concurrent;
44   size_t _implicit_full;
45   size_t _degen_points[ShenandoahHeap::_DEGENERATED_LIMIT];
46 
47   ShenandoahSharedFlag _in_shutdown;
48 
49   ShenandoahTracer* _tracer;
50 
51   size_t _cycle_counter;
52 
53 public:

57   // These two encompass the entire cycle.
58   void record_cycle_start();
59 
60   void record_success_concurrent();
61   void record_success_degenerated();
62   void record_success_full();
63   void record_alloc_failure_to_degenerated(ShenandoahHeap::ShenandoahDegenPoint point);
64   void record_alloc_failure_to_full();
65   void record_degenerated_upgrade_to_full();
66   void record_explicit_to_concurrent();
67   void record_explicit_to_full();
68   void record_implicit_to_concurrent();
69   void record_implicit_to_full();
70 
71   void record_shutdown();
72   bool is_at_shutdown();
73 
74   ShenandoahTracer* tracer() {return _tracer;}
75 
76   size_t cycle_counter() const;
77 
78   void print_gc_stats(outputStream* out) const;
79 };
