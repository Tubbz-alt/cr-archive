30 bool VM_ShenandoahReferenceOperation::doit_prologue() {
31   Heap_lock->lock();
32   return true;
33 }
34 
35 void VM_ShenandoahReferenceOperation::doit_epilogue() {
36   if (Universe::has_reference_pending_list()) {
37     Heap_lock->notify_all();
38   }
39   Heap_lock->unlock();
40 }
