24 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHCOLLECTIONSET_HPP
25 #define SHARE_GC_SHENANDOAH_SHENANDOAHCOLLECTIONSET_HPP
26 
27 #include "memory/allocation.hpp"
28 #include "gc/shenandoah/shenandoahHeap.hpp"
29 #include "gc/shenandoah/shenandoahHeapRegion.hpp"
30 
31 class ShenandoahCollectionSet : public CHeapObj<mtGC> {
32   friend class ShenandoahHeap;
33 private:
34   size_t const          _map_size;
35   size_t const          _region_size_bytes_shift;
36   jbyte* const          _cset_map;
37   // Bias cset map's base address for fast test if an oop is in cset
38   jbyte* const          _biased_cset_map;
39 
40   ShenandoahHeap* const _heap;
41 
42   size_t                _garbage;
43   size_t                _live_data;
44   size_t                _used;
45   size_t                _region_count;
46 
47   DEFINE_PAD_MINUS_SIZE(0, DEFAULT_CACHE_LINE_SIZE, sizeof(volatile size_t));
48   volatile jint         _current_index;
49   DEFINE_PAD_MINUS_SIZE(1, DEFAULT_CACHE_LINE_SIZE, 0);
50 
51 public:
52   ShenandoahCollectionSet(ShenandoahHeap* heap, HeapWord* heap_base);
53 
54   // Add region to collection set
55   void add_region(ShenandoahHeapRegion* r);
56   bool add_region_check_for_duplicates(ShenandoahHeapRegion* r);
57 
58   // Bring per-region statuses to consistency with this collection.
59   // TODO: This is a transitional interface that bridges the gap between
60   // region statuses and this collection. Should go away after we merge them.
61   void update_region_status();
62 
63   // Remove region from collection set
64   void remove_region(ShenandoahHeapRegion* r);
65 
66   // MT version
67   ShenandoahHeapRegion* claim_next();
68 
69   // Single-thread version
70   ShenandoahHeapRegion* next();
71 
72   size_t count()  const { return _region_count; }
73   bool is_empty() const { return _region_count == 0; }
74 
75   void clear_current_index() {
76     _current_index = 0;
77   }
78 
79   inline bool is_in(ShenandoahHeapRegion* r) const;
80   inline bool is_in(size_t region_number)    const;
81   inline bool is_in(HeapWord* p)             const;
82 
83   void print_on(outputStream* out) const;
84 
85   size_t used()      const { return _used; }
86   size_t live_data() const { return _live_data; }
87   size_t garbage()   const { return _garbage;   }
88   void clear();
89 
90 private:
91   jbyte* biased_map_address() const {
92     return _biased_cset_map;
93   }
94 };
95 
96 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHCOLLECTIONSET_HPP
