31 void ShenandoahRuntime::write_ref_array_pre_oop_entry(oop* dst, size_t length) {
32   ShenandoahBarrierSet *bs = ShenandoahBarrierSet::barrier_set();
33   bs->write_ref_array_pre(dst, length, false);
34 }

36 void ShenandoahRuntime::write_ref_array_pre_narrow_oop_entry(narrowOop* dst, size_t length) {

38   bs->write_ref_array_pre(dst, length, false);
39 }

41 void ShenandoahRuntime::write_ref_array_post_entry(HeapWord* dst, size_t length) {
42   ShenandoahBarrierSet *bs = ShenandoahBarrierSet::barrier_set();
43   bs->ShenandoahBarrierSet::write_ref_array(dst, length);
44 }

46 // Shenandoah pre write barrier slowpath
47 JRT_LEAF(void, ShenandoahRuntime::write_ref_field_pre_entry(oopDesc* orig, JavaThread *thread))
48   if (orig == NULL) {
49     assert(false, "should be optimized out");
50     return;
51   }
52   shenandoah_assert_correct(NULL, orig);
53   // store the original value that was in the field reference
54   ShenandoahThreadLocalData::satb_mark_queue(thread).enqueue(orig);
55 JRT_END
56 
57 JRT_LEAF(oopDesc*, ShenandoahRuntime::write_barrier_JRT(oopDesc* src))
58   oop result = ShenandoahBarrierSet::barrier_set()->write_barrier_mutator(src);
59   return (oopDesc*) result;
60 JRT_END
61 
62 // Shenandoah clone barrier: makes sure that references point to to-space
63 // in cloned objects.
64 JRT_LEAF(void, ShenandoahRuntime::shenandoah_clone_barrier(oopDesc* obj))
65   ShenandoahBarrierSet::barrier_set()->write_region(MemRegion((HeapWord*) obj, obj->size()));
66 JRT_END
