25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHWORKERPOLICY_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHWORKERPOLICY_HPP
27 
28 #include "memory/allocation.hpp"
29 
30 class ShenandoahWorkerPolicy : AllStatic {
31 private:
32   static uint _prev_par_marking;
33   static uint _prev_conc_marking;
34   static uint _prev_conc_root_proc;
35   static uint _prev_conc_evac;
36   static uint _prev_fullgc;
37   static uint _prev_degengc;
38   static uint _prev_stw_traversal;
39   static uint _prev_conc_traversal;
40   static uint _prev_conc_update_ref;
41   static uint _prev_par_update_ref;
42   static uint _prev_conc_cleanup;
43   static uint _prev_conc_reset;
44 
45 public:
46   // Calculate the number of workers for initial marking
47   static uint calc_workers_for_init_marking();
48 
49   // Calculate the number of workers for concurrent marking
50   static uint calc_workers_for_conc_marking();
51 
52   // Calculate the number of workers for final marking
53   static uint calc_workers_for_final_marking();
54 
55   // Calculate workers for concurrent root processing
56   static uint calc_workers_for_conc_root_processing();
57 
58   // Calculate workers for concurrent evacuation (concurrent GC)
59   static uint calc_workers_for_conc_evac();
60 
61   // Calculate workers for parallel full gc
62   static uint calc_workers_for_fullgc();
63 
64   // Calculate workers for parallel degenerated gc
65   static uint calc_workers_for_stw_degenerated();
66 
67   // Calculate workers for Stop-the-world traversal GC
68   static uint calc_workers_for_stw_traversal();
69 
70   // Calculate workers for concurrent traversal GC
71   static uint calc_workers_for_conc_traversal();
72 
73   // Calculate workers for concurrent reference update
74   static uint calc_workers_for_conc_update_ref();
75 
76   // Calculate workers for parallel/final reference update
77   static uint calc_workers_for_final_update_ref();
78 
79   // Calculate workers for concurrent precleaning
80   static uint calc_workers_for_conc_preclean();
81 
82   // Calculate workers for concurrent cleanup
83   static uint calc_workers_for_conc_cleanup();
84 
85   // Calculate workers for concurrent reset
86   static uint calc_workers_for_conc_reset();
87 };
88 
89 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHWORKERPOLICY_HPP
