 47 ShenandoahRootVerifier::ShenandoahRootVerifier(RootTypes types) : _types(types) {
 48 }
 49 
 50 void ShenandoahRootVerifier::excludes(RootTypes types) {
 51   _types = static_cast<ShenandoahRootVerifier::RootTypes>(static_cast<uint>(_types) & (~static_cast<uint>(types)));
 52 }
 53 
 54 bool ShenandoahRootVerifier::verify(RootTypes type) const {
 55   return (_types & type) == type;
 56 }
 57 
 58 ShenandoahRootVerifier::RootTypes ShenandoahRootVerifier::combine(RootTypes t1, RootTypes t2) {
 59   return static_cast<ShenandoahRootVerifier::RootTypes>(static_cast<uint>(t1) | static_cast<uint>(t2));
 60 }

 62 void ShenandoahRootVerifier::oops_do(OopClosure* oops) {
 63   CodeBlobToOopClosure blobs(oops, !CodeBlobToOopClosure::FixRelocations);
 64   if (verify(CodeRoots)) {
 65     shenandoah_assert_locked_or_safepoint(CodeCache_lock);
 66     CodeCache::blobs_do(&blobs);
 67   }
 68 
 69   if (verify(CLDGRoots)) {
 70     shenandoah_assert_locked_or_safepoint(ClassLoaderDataGraph_lock);
 71     CLDToOopClosure clds(oops, ClassLoaderData::_claim_none);
 72     ClassLoaderDataGraph::cld_do(&clds);
 73   }
 74 
 75   if (verify(SerialRoots)) {
 76     shenandoah_assert_safepoint();
 77     Universe::oops_do(oops);
 78     Management::oops_do(oops);
 79     JvmtiExport::oops_do(oops);
 80     ObjectSynchronizer::oops_do(oops);
 81     SystemDictionary::oops_do(oops);
 82   }
 83 
 84   if (verify(JNIHandleRoots)) {
 85     shenandoah_assert_safepoint();
 86     JNIHandles::oops_do(oops);
 87   }
 88 
 89   if (verify(WeakRoots)) {
 90     shenandoah_assert_safepoint();
 91     AlwaysTrueClosure always_true;
 92     WeakProcessor::weak_oops_do(&always_true, oops);
 93   } else if (verify(SerialWeakRoots)) {
 94     shenandoah_assert_safepoint();
 95     serial_weak_roots_do(oops);
 96   } else if (verify(ConcurrentWeakRoots)) {
 97     concurrent_weak_roots_do(oops);
 98   }
 99 
100   if (ShenandoahStringDedup::is_enabled() && verify(StringDedupRoots)) {
101     shenandoah_assert_safepoint();
102     ShenandoahStringDedup::oops_do_slow(oops);
103   }
104 
105   if (verify(ThreadRoots)) {
106     shenandoah_assert_safepoint();
107     // Do thread roots the last. This allows verification code to find
108     // any broken objects from those special roots first, not the accidental
109     // dangling reference from the thread root.
110     Threads::possibly_parallel_oops_do(false, oops, &blobs);
111   }
112 }

114 void ShenandoahRootVerifier::roots_do(OopClosure* oops) {
115   shenandoah_assert_safepoint();
116 
117   CodeBlobToOopClosure blobs(oops, !CodeBlobToOopClosure::FixRelocations);
118   CodeCache::blobs_do(&blobs);
119 
120   CLDToOopClosure clds(oops, ClassLoaderData::_claim_none);
121   ClassLoaderDataGraph::cld_do(&clds);
122 
123   Universe::oops_do(oops);
124   Management::oops_do(oops);
125   JvmtiExport::oops_do(oops);
126   JNIHandles::oops_do(oops);
127   ObjectSynchronizer::oops_do(oops);
128   SystemDictionary::oops_do(oops);
129 
130   AlwaysTrueClosure always_true;
131   WeakProcessor::weak_oops_do(&always_true, oops);
132 
133   if (ShenandoahStringDedup::is_enabled()) {
134     ShenandoahStringDedup::oops_do_slow(oops);
135   }
136 
137   // Do thread roots the last. This allows verification code to find
138   // any broken objects from those special roots first, not the accidental
139   // dangling reference from the thread root.
140   Threads::possibly_parallel_oops_do(false, oops, &blobs);
141 }
142 
143 void ShenandoahRootVerifier::strong_roots_do(OopClosure* oops) {
144   shenandoah_assert_safepoint();
145 
146   CodeBlobToOopClosure blobs(oops, !CodeBlobToOopClosure::FixRelocations);
147 
148   CLDToOopClosure clds(oops, ClassLoaderData::_claim_none);
149   ClassLoaderDataGraph::roots_cld_do(&clds, NULL);
150 
151   Universe::oops_do(oops);
152   Management::oops_do(oops);
153   JvmtiExport::oops_do(oops);
154   JNIHandles::oops_do(oops);
155   ObjectSynchronizer::oops_do(oops);
156   SystemDictionary::oops_do(oops);
157 
158   // Do thread roots the last. This allows verification code to find
159   // any broken objects from those special roots first, not the accidental
160   // dangling reference from the thread root.
161   Threads::possibly_parallel_oops_do(false, oops, &blobs);
162 }

164 void ShenandoahRootVerifier::serial_weak_roots_do(OopClosure* cl) {
165   WeakProcessorPhases::Iterator itr = WeakProcessorPhases::serial_iterator();
166   AlwaysTrueClosure always_true;
167   for ( ; !itr.is_end(); ++itr) {
168     WeakProcessorPhases::processor(*itr)(&always_true, cl);
169   }
170 }
171 
172 void ShenandoahRootVerifier::concurrent_weak_roots_do(OopClosure* cl) {
173   for (OopStorageSet::Iterator it = OopStorageSet::weak_iterator(); !it.is_end(); ++it) {
174     OopStorage* storage = *it;
175     storage->oops_do<OopClosure>(cl);
176   }
