24 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHHEAPREGIONSET_INLINE_HPP
25 #define SHARE_GC_SHENANDOAH_SHENANDOAHHEAPREGIONSET_INLINE_HPP
26 
27 #include "gc/shenandoah/shenandoahHeapRegionSet.hpp"
28 #include "gc/shenandoah/shenandoahHeap.hpp"
29 #include "gc/shenandoah/shenandoahHeap.inline.hpp"
30 #include "gc/shenandoah/shenandoahHeapRegion.hpp"
31 
32 bool ShenandoahHeapRegionSet::is_in(size_t region_number) const {
33   assert(region_number < _heap->num_regions(), "Sanity");
34   return _set_map[region_number] == 1;
35 }
36 
37 bool ShenandoahHeapRegionSet::is_in(ShenandoahHeapRegion* r) const {
38   return is_in(r->region_number());
39 }
40 
41 bool ShenandoahHeapRegionSet::is_in(HeapWord* p) const {
42   assert(_heap->is_in(p), "Must be in the heap");
43   uintx index = ((uintx) p) >> _region_size_bytes_shift;
44   // no need to subtract the bottom of the heap from p,
45   // _biased_set_map is biased
46   return _biased_set_map[index] == 1;
47 }
