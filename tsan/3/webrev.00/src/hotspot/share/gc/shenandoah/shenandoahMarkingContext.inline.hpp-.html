24 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHMARKINGCONTEXT_INLINE_HPP
25 #define SHARE_GC_SHENANDOAH_SHENANDOAHMARKINGCONTEXT_INLINE_HPP
26 
27 #include "gc/shenandoah/shenandoahMarkingContext.hpp"
28 
29 inline MarkBitMap* ShenandoahMarkingContext::mark_bit_map() {
30   return &_mark_bit_map;
31 }
32 
33 inline bool ShenandoahMarkingContext::mark(oop obj) {
34   shenandoah_assert_not_forwarded(NULL, obj);
35   HeapWord* addr = (HeapWord*) obj;
36   return (! allocated_after_mark_start(addr)) && _mark_bit_map.par_mark(addr);
37 }
38 
39 inline bool ShenandoahMarkingContext::is_marked(oop obj) const {
40   HeapWord* addr = (HeapWord*) obj;
41   return allocated_after_mark_start(addr) || _mark_bit_map.is_marked(addr);
42 }
43 
44 inline bool ShenandoahMarkingContext::allocated_after_mark_start(HeapWord* addr) const {
45   uintx index = ((uintx) addr) >> ShenandoahHeapRegion::region_size_bytes_shift();
46   HeapWord* top_at_mark_start = _top_at_mark_starts[index];
47   bool alloc_after_mark_start = addr >= top_at_mark_start;
48   return alloc_after_mark_start;
49 }
50 
51 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHMARKINGCONTEXT_INLINE_HPP
