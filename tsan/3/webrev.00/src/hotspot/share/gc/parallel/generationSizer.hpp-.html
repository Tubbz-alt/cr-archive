26 #define SHARE_GC_PARALLEL_GENERATIONSIZER_HPP

28 #include "gc/shared/collectorPolicy.hpp"
29 
30 // There is a nice batch of tested generation sizing code in
31 // GenCollectorPolicy. Lets reuse it!
32 
33 class GenerationSizer : public GenCollectorPolicy {
34  private:
35   // The alignment used for boundary between young gen and old gen
36   static size_t default_gen_alignment() { return 64 * K * HeapWordSize; }
37 
38  protected:
39 
40   void initialize_alignments();
41   void initialize_flags();
42   void initialize_size_info();
43 
44  public:
45   virtual size_t heap_reserved_size_bytes() const;
46   virtual bool is_hetero_heap() const;
47 };
48 #endif // SHARE_GC_PARALLEL_GENERATIONSIZER_HPP
