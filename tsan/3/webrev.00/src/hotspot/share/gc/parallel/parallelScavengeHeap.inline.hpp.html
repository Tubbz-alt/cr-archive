28 #include "gc/parallel/parallelScavengeHeap.hpp"
29 #include "gc/parallel/psParallelCompact.inline.hpp"
30 #include "gc/parallel/psScavenge.hpp"
31 
32 inline size_t ParallelScavengeHeap::total_invocations() {
33   return PSParallelCompact::total_invocations();
34 }
35 
36 inline bool ParallelScavengeHeap::should_alloc_in_eden(const size_t size) const {
37   const size_t eden_size = young_gen()->eden_space()->capacity_in_words();
38   return size < eden_size / 2;
39 }
40 
41 inline void ParallelScavengeHeap::invoke_scavenge() {
42   PSScavenge::invoke();
43 }
44 
45 inline bool ParallelScavengeHeap::is_in_young(oop p) {
46   // Assumes the the old gen address range is lower than that of the young gen.
47   bool result = cast_from_oop<HeapWord*>(p) >= young_gen()->reserved().start();
48   assert(result == young_gen()->is_in_reserved(p),
49          "incorrect test - result=%d, p=" PTR_FORMAT, result, p2i((void*)p));
50   return result;
51 }
52 #endif // SHARE_GC_PARALLEL_PARALLELSCAVENGEHEAP_INLINE_HPP
