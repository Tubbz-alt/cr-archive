 43 void ParallelArguments::initialize() {
 44   GCArguments::initialize();
 45   assert(UseParallelGC || UseParallelOldGC, "Error");
 46   // Enable ParallelOld unless it was explicitly disabled (cmd line or rc file).
 47   if (FLAG_IS_DEFAULT(UseParallelOldGC)) {
 48     FLAG_SET_DEFAULT(UseParallelOldGC, true);
 49   }
 50   FLAG_SET_DEFAULT(UseParallelGC, true);
 51 
 52   // If no heap maximum was requested explicitly, use some reasonable fraction
 53   // of the physical memory, up to a maximum of 1GB.
 54   FLAG_SET_DEFAULT(ParallelGCThreads,
 55                    WorkerPolicy::parallel_worker_threads());
 56   if (ParallelGCThreads == 0) {
 57     jio_fprintf(defaultStream::error_stream(),
 58         "The Parallel GC can not be combined with -XX:ParallelGCThreads=0\n");
 59     vm_exit(1);
 60   }
 61 
 62   if (UseAdaptiveSizePolicy) {
 63     // We don't want to limit adaptive heap sizing's freedom to adjust the heap
 64     // unless the user actually sets these flags.
 65     if (FLAG_IS_DEFAULT(MinHeapFreeRatio)) {
 66       FLAG_SET_DEFAULT(MinHeapFreeRatio, 0);
 67     }
 68     if (FLAG_IS_DEFAULT(MaxHeapFreeRatio)) {
 69       FLAG_SET_DEFAULT(MaxHeapFreeRatio, 100);
 70     }
 71   }
 72 
 73   // If InitialSurvivorRatio or MinSurvivorRatio were not specified, but the
 74   // SurvivorRatio has been set, reset their default values to SurvivorRatio +
 75   // 2.  By doing this we make SurvivorRatio also work for Parallel Scavenger.
 76   // See CR 6362902 for details.
 77   if (!FLAG_IS_DEFAULT(SurvivorRatio)) {
 78     if (FLAG_IS_DEFAULT(InitialSurvivorRatio)) {
 79        FLAG_SET_DEFAULT(InitialSurvivorRatio, SurvivorRatio + 2);
 80     }
 81     if (FLAG_IS_DEFAULT(MinSurvivorRatio)) {
 82       FLAG_SET_DEFAULT(MinSurvivorRatio, SurvivorRatio + 2);
 83     }
 84   }
 85 
 86   if (UseParallelOldGC) {
 87     // Par compact uses lower default values since they are treated as
 88     // minimums.  These are different defaults because of the different
 89     // interpretation and are not ergonomically set.
 90     if (FLAG_IS_DEFAULT(MarkSweepDeadRatio)) {
 91       FLAG_SET_DEFAULT(MarkSweepDeadRatio, 1);
 92     }
 93   }
 94 }

 96 CollectedHeap* ParallelArguments::create_heap() {
 97   if (AllocateOldGenAt != NULL) {
 98     return create_heap_with_policy<ParallelScavengeHeap, HeterogeneousGenerationSizer>();
























































 99   } else {
100     return create_heap_with_policy<ParallelScavengeHeap, GenerationSizer>();

101   }






























































102 }
