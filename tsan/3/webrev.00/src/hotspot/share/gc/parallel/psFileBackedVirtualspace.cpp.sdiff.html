35 bool PSFileBackedVirtualSpace::initialize() {
36   _fd = os::create_file_for_heap(_file_path);
37   if (_fd == -1) {
38     return false;
39   }
40   // We map the reserved space to a file at initialization.
41   char* ret = os::replace_existing_mapping_with_file_mapping(reserved_low_addr(), reserved_size(), _fd);
42   if (ret != reserved_low_addr()) {
43     os::close(_fd);
44     return false;
45   }
46   // _mapping_succeeded is false if we return before this point.
47   // expand calls later check value of this flag and return error if it is false.
48   _mapping_succeeded = true;
49   _special = true;
50   os::close(_fd);
51   return true;
52 }

54 PSFileBackedVirtualSpace::PSFileBackedVirtualSpace(ReservedSpace rs, const char* path) {
55   PSFileBackedVirtualSpace(rs, os::vm_page_size(), path);
56 }

58 bool PSFileBackedVirtualSpace::expand_by(size_t bytes) {
59   assert(special(), "Since entire space is committed at initialization, _special should always be true for PSFileBackedVirtualSpace");
60 
61   // if mapping did not succeed during intialization return false
62   if (!_mapping_succeeded) {
63     return false;
64   }
65   return PSVirtualSpace::expand_by(bytes);
66 
67 }

71   return PSVirtualSpace::shrink_by(bytes);
72 }

74 size_t PSFileBackedVirtualSpace::expand_into(PSVirtualSpace* space, size_t bytes) {
75   // not supported. Since doing this will change page mapping which will lead to large TLB penalties.
76   assert(false, "expand_into() should not be called for PSFileBackedVirtualSpace");
77   return 0;
