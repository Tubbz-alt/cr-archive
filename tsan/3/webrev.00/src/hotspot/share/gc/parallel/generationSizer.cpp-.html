32 void GenerationSizer::initialize_alignments() {
33   _space_alignment = _gen_alignment = default_gen_alignment();
34   _heap_alignment = compute_heap_alignment();
35 }
36 
37 void GenerationSizer::initialize_flags() {
38   // Do basic sizing work
39   GenCollectorPolicy::initialize_flags();
40 
41   // The survivor ratio's are calculated "raw", unlike the
42   // default gc, which adds 2 to the ratio value. We need to
43   // make sure the values are valid before using them.
44   if (MinSurvivorRatio < 3) {
45     FLAG_SET_ERGO(uintx, MinSurvivorRatio, 3);
46   }
47 
48   if (InitialSurvivorRatio < 3) {
49     FLAG_SET_ERGO(uintx, InitialSurvivorRatio, 3);
50   }
51 }
52 
53 void GenerationSizer::initialize_size_info() {
54   const size_t max_page_sz = os::page_size_for_region_aligned(_max_heap_byte_size, 8);
55   const size_t min_pages = 4; // 1 for eden + 1 for each survivor + 1 for old
56   const size_t min_page_sz = os::page_size_for_region_aligned(_min_heap_byte_size, min_pages);
57   const size_t page_sz = MIN2(max_page_sz, min_page_sz);
58 
59   // Can a page size be something else than a power of two?
60   assert(is_power_of_2((intptr_t)page_sz), "must be a power of 2");
61   size_t new_alignment = align_up(page_sz, _gen_alignment);
62   if (new_alignment != _gen_alignment) {
63     _gen_alignment = new_alignment;
64     _space_alignment = new_alignment;
65     // Redo everything from the start
66     initialize_flags();
67   }
68   GenCollectorPolicy::initialize_size_info();
69 }
