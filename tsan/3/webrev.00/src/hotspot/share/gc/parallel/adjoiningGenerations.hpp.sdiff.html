34 // Contains two generations that both use an AdjoiningVirtualSpaces.
35 // The two generations are adjacent in the reserved space for the
36 // heap.  Each generation has a virtual space and shrinking and
37 // expanding of the generations can still be down with that
38 // virtual space as was previously done.  If expanding of reserved
39 // size of a generation is required, the adjacent generation
40 // must be shrunk.  Adjusting the boundary between the generations
41 // is called for in this class.
42 
43 class AdjoiningGenerations : public CHeapObj<mtGC> {
44   friend class VMStructs;
45  private:
46   // Move boundary up to expand old gen.  Checks are made to
47   // determine if the move can be done with specified limits.
48   void request_old_gen_expansion(size_t desired_change_in_bytes);
49   // Move boundary down to expand young gen.
50   bool request_young_gen_expansion(size_t desired_change_in_bytes);
51 
52  protected:

53    // The young generation and old generation, respectively
54    PSYoungGen* _young_gen;
55    PSOldGen* _old_gen;
56 
57    // The spaces used by the two generations.
58    AdjoiningVirtualSpaces* _virtual_spaces;
59 
60  public:
61   AdjoiningGenerations(ReservedSpace rs, GenerationSizer* policy, size_t alignment);
62   AdjoiningGenerations();
63 
64   // Accessors
65   PSYoungGen* young_gen() { return _young_gen; }
66   PSOldGen* old_gen() { return _old_gen; }
67 
68   AdjoiningVirtualSpaces* virtual_spaces() { return _virtual_spaces; }
69 
70   // Additional space is needed in the old generation.  Check
71   // the available space and attempt to move the boundary if more space
72   // is needed.  The growth is not guaranteed to occur.
73   void adjust_boundary_for_old_gen_needs(size_t desired_change_in_bytes);
74   // Similarly for a growth of the young generation.
75   void adjust_boundary_for_young_gen_needs(size_t eden_size, size_t survivor_size);
76 
77   // Return the total byte size of the reserved space
78   // for the adjoining generations.
79   virtual size_t reserved_byte_size();
80 
81   // Return new AdjoiningGenerations instance based on collector policy (specifically - whether heap is heterogeneous).
82   static AdjoiningGenerations* create_adjoining_generations(ReservedSpace rs, GenerationSizer* policy, size_t alignment);
83 };
84 #endif // SHARE_GC_PARALLEL_ADJOININGGENERATIONS_HPP
