34 void AllocationStats::initialize(bool split_birth)   {

37                                                        CMS_FLSPadding);
38   _desired = 0;
39   _coal_desired = 0;
40   _surplus = 0;
41   _bfr_surp = 0;
42   _prev_sweep = 0;

44   _coal_births = 0;
45   _coal_deaths = 0;
46   _split_births = (split_birth ? 1 : 0);
47   _split_deaths = 0;
48   _returned_bytes = 0;
49 }
