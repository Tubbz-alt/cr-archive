29 #ifndef PRODUCT

31 #define baadbabeHeapWord badHeapWordVal
32 #define deadbeefHeapWord 0xdeadbeef
33 
34 size_t const FreeChunk::header_size() {
35   return sizeof(FreeChunk)/HeapWordSize;
36 }
37 
38 void FreeChunk::mangleAllocated(size_t size) {
39   // mangle all but the header of a just-allocated block
40   // of storage
41   assert(size >= MinChunkSize, "smallest size of object");
42   // we can't assert that _size == size because this may be an
43   // allocation out of a linear allocation block
44   assert(sizeof(FreeChunk) % HeapWordSize == 0,
45          "shouldn't write beyond chunk");
46   HeapWord* addr = (HeapWord*)this;
47   size_t hdr = header_size();
48   Copy::fill_to_words(addr + hdr, size - hdr, baadbabeHeapWord);
49 }
50 
51 void FreeChunk::mangleFreed(size_t sz) {
52   assert(baadbabeHeapWord != deadbeefHeapWord, "Need distinct patterns");
53   // mangle all but the header of a just-freed block of storage
54   // just prior to passing it to the storage dictionary
55   assert(sz >= MinChunkSize, "smallest size of object");
56   assert(sz == size(), "just checking");
57   HeapWord* addr = (HeapWord*)this;
58   size_t hdr = header_size();
59   Copy::fill_to_words(addr + hdr, sz - hdr, deadbeefHeapWord);
60 }
61 
62 void FreeChunk::verifyList() const {
63   FreeChunk* nextFC = next();
64   if (nextFC != NULL) {
65     assert(this == nextFC->prev(), "broken chain");
66     assert(size() == nextFC->size(), "wrong size");
67     nextFC->verifyList();
68   }
69 }

72 void FreeChunk::print_on(outputStream* st) {
73   st->print_cr("Next: " PTR_FORMAT " Prev: " PTR_FORMAT " %s",
74     p2i(next()), p2i(prev()), cantCoalesce() ? "[can't coalesce]" : "");
75 }
