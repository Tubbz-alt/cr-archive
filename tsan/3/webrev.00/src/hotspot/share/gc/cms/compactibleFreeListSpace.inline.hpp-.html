25 #ifndef SHARE_GC_CMS_COMPACTIBLEFREELISTSPACE_INLINE_HPP
26 #define SHARE_GC_CMS_COMPACTIBLEFREELISTSPACE_INLINE_HPP
27 
28 #include "gc/cms/compactibleFreeListSpace.hpp"
29 #include "gc/cms/promotionInfo.inline.hpp"
30 
31 template <typename OopClosureType>
32 void CompactibleFreeListSpace::oop_since_save_marks_iterate(OopClosureType* blk) {
33   _promoInfo.promoted_oops_iterate(blk);
34 
35   // This also restores any displaced headers and removes the elements from
36   // the iteration set as they are processed, so that we have a clean slate
37   // at the end of the iteration. Note, thus, that if new objects are
38   // promoted as a result of the iteration they are iterated over as well.
39   assert(_promoInfo.noPromotions(), "_promoInfo inconsistency");
40 }
41 
42 #endif // SHARE_GC_CMS_COMPACTIBLEFREELISTSPACE_INLINE_HPP
