25 #ifndef SHARE_GC_CMS_CMSLOCKVERIFIER_HPP
26 #define SHARE_GC_CMS_CMSLOCKVERIFIER_HPP
27 
28 #include "runtime/mutex.hpp"
29 
30 ///////////// Locking verification specific to CMS //////////////
31 // Much like "assert_lock_strong()", except
32 // that it relaxes the assertion somewhat for the parallel GC case, where
33 // main GC thread or the CMS thread might hold the lock on behalf of
34 // the parallel threads.
35 class CMSLockVerifier: AllStatic {
36  public:
37   static void assert_locked(const Mutex* lock, const Mutex* p_lock1, const Mutex* p_lock2)
38     PRODUCT_RETURN;
39   static void assert_locked(const Mutex* lock, const Mutex* p_lock) {
40     assert_locked(lock, p_lock, NULL);
41   }
42   static void assert_locked(const Mutex* lock) {
43     assert_locked(lock, NULL);
44   }
45 };
46 
47 #endif // SHARE_GC_CMS_CMSLOCKVERIFIER_HPP
