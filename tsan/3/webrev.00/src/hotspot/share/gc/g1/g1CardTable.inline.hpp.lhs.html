30 void G1CardTable::set_card_claimed(size_t card_index) {
31   jbyte val = _byte_map[card_index];
32   if (val == clean_card_val()) {
33     val = (jbyte)claimed_card_val();
34   } else {
35     val |= (jbyte)claimed_card_val();



















































36   }
37   _byte_map[card_index] = val;
38 }
