25 #ifndef SHARE_GC_G1_G1FREEIDSET_HPP
26 #define SHARE_GC_G1_G1FREEIDSET_HPP
27 

28 #include "runtime/semaphore.hpp"
29 #include "utilities/globalDefinitions.hpp"
30 
31 // Represents a set of small integer ids, from which elements can be
32 // temporarily allocated for exclusive use.  The ids are in a
33 // contiguous range from 'start' to 'start + size'.  Used to obtain a
34 // distinct worker_id value for a mutator thread that doesn't normally
35 // have such an id.
36 class G1FreeIdSet {
37   Semaphore _sem;
38   uint* _next;
39   uint _start;
40   uint _size;
41   uintx _head_index_mask;
42   volatile uintx _head;
43 
44   uint head_index(uintx head) const;
45   uintx make_head(uint index, uintx old_head) const;
46 
47   NONCOPYABLE(G1FreeIdSet);


48 
49 public:
50   G1FreeIdSet(uint start, uint size);
51   ~G1FreeIdSet();
52 
53   // Returns an unclaimed parallel id (waiting for one to be released if
54   // necessary).  Must not safepoint while holding a claimed id.
55   uint claim_par_id();
56 
57   void release_par_id(uint id);
58 
59   struct TestSupport;           // For unit test access.
60 };
61 
62 #endif // SHARE_GC_G1_G1FREEIDSET_HPP
