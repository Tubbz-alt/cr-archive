29 G1BufferNodeList::G1BufferNodeList() :
30   _head(NULL), _tail(NULL), _entry_count(0) {}
31 
32 G1BufferNodeList::G1BufferNodeList(BufferNode* head,
33                                    BufferNode* tail,
34                                    size_t entry_count) :
35   _head(head), _tail(tail), _entry_count(entry_count)
36 {
37   assert((_head == NULL) == (_tail == NULL), "invariant");
38   assert((_head == NULL) == (_entry_count == 0), "invariant");
39 }
