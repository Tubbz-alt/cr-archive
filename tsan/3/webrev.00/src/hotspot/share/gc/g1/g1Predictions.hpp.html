25 #ifndef SHARE_GC_G1_G1PREDICTIONS_HPP
26 #define SHARE_GC_G1_G1PREDICTIONS_HPP
27 
28 #include "utilities/numberSeq.hpp"
29 
30 // Utility class containing various helper methods for prediction.
31 class G1Predictions {
32  private:
33   double _sigma;
34 
35   // This function is used to estimate the stddev of sample sets. There is some
36   // special consideration of small sample sets: the actual stddev for them is
37   // not very useful, so we calculate some value based on the sample average.
38   // Five or more samples yields zero (at that point we use the stddev); fewer
39   // scale the sample set average linearly from two times the average to 0.5 times
40   // it.
41   double stddev_estimate(TruncatedSeq const* seq) const {
42     double estimate = seq->dsd();
43     int const samples = seq->num();
44     if (samples < 5) {
45       estimate = MAX2(seq->davg() * (5 - samples) / 2.0, estimate);
46     }
47     return estimate;
48   }
49  public:
50   G1Predictions(double sigma) : _sigma(sigma) {
51     assert(sigma >= 0.0, "Confidence must be larger than or equal to zero");
52   }
53 
54   // Confidence factor.
55   double sigma() const { return _sigma; }
56 
57   double predict(TruncatedSeq const* seq) const {
58     return seq->davg() + _sigma * stddev_estimate(seq);
59   }
60 
61   double predict_in_unit_interval(TruncatedSeq const* seq) const {
62     return clamp(predict(seq), 0.0, 1.0);
63   }
64 
65   double predict_zero_bounded(TruncatedSeq const* seq) const {
66     return MAX2(predict(seq), 0.0);
67   }
68 };
69 
70 #endif // SHARE_GC_G1_G1PREDICTIONS_HPP
