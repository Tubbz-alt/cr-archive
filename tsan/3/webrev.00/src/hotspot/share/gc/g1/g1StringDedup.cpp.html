38 void G1StringDedup::initialize() {
39   assert(UseG1GC, "String deduplication available with G1");
40   StringDedup::initialize_impl<G1StringDedupQueue, G1StringDedupStat>();
41 }
42 
43 bool G1StringDedup::is_candidate_from_mark(oop obj) {
44   if (java_lang_String::is_instance_inlined(obj)) {
45     bool from_young = G1CollectedHeap::heap()->heap_region_containing(obj)->is_young();
46     if (from_young && obj->age() < StringDeduplicationAgeThreshold) {
47       // Candidate found. String is being evacuated from young to old but has not
48       // reached the deduplication age threshold, i.e. has not previously been a
49       // candidate during its life in the young generation.
50       return true;
51     }
52   }
53 
54   // Not a candidate
55   return false;
56 }
57 
58 void G1StringDedup::enqueue_from_mark(oop java_string, uint worker_id) {
59   assert(is_enabled(), "String deduplication not enabled");
60   if (is_candidate_from_mark(java_string)) {
61     G1StringDedupQueue::push(worker_id, java_string);
62   }
63 }
64 
65 bool G1StringDedup::is_candidate_from_evacuation(bool from_young, bool to_young, oop obj) {
66   if (from_young && java_lang_String::is_instance_inlined(obj)) {
67     if (to_young && obj->age() == StringDeduplicationAgeThreshold) {
68       // Candidate found. String is being evacuated from young to young and just
69       // reached the deduplication age threshold.
70       return true;
71     }
72     if (!to_young && obj->age() < StringDeduplicationAgeThreshold) {
73       // Candidate found. String is being evacuated from young to old but has not
74       // reached the deduplication age threshold, i.e. has not previously been a
75       // candidate during its life in the young generation.
76       return true;
77     }
78   }
79 
80   // Not a candidate
81   return false;
82 }
83 
84 void G1StringDedup::enqueue_from_evacuation(bool from_young, bool to_young, uint worker_id, oop java_string) {
85   assert(is_enabled(), "String deduplication not enabled");
86   if (is_candidate_from_evacuation(from_young, to_young, java_string)) {
87     G1StringDedupQueue::push(worker_id, java_string);
88   }
89 }
