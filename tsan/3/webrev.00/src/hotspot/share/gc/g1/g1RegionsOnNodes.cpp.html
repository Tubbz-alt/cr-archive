25 #include "precompiled.hpp"
26 #include "gc/g1/g1NUMA.hpp"
27 #include "gc/g1/g1RegionsOnNodes.hpp"
28 #include "gc/g1/heapRegion.hpp"
29 
30 G1RegionsOnNodes::G1RegionsOnNodes() : _count_per_node(NULL), _numa(G1NUMA::numa()) {
31   _count_per_node = NEW_C_HEAP_ARRAY(uint, _numa->num_active_nodes(), mtGC);
32   clear();
33 }
34 
35 G1RegionsOnNodes::~G1RegionsOnNodes() {
36   FREE_C_HEAP_ARRAY(uint, _count_per_node);
37 }
38 
39 uint G1RegionsOnNodes::add(HeapRegion* hr) {
40   uint node_index = hr->node_index();
41 
42   // Update only if the node index is valid.
43   if (node_index < _numa->num_active_nodes()) {
44     *(_count_per_node + node_index) += 1;
45     return node_index;
46   }
47 
48   return G1NUMA::UnknownNodeIndex;
49 }
50 
51 void G1RegionsOnNodes::clear() {
52   for (uint i = 0; i < _numa->num_active_nodes(); i++) {
53     _count_per_node[i] = 0;
54   }
55 }
56 
57 uint G1RegionsOnNodes::count(uint node_index) const {
58   return _count_per_node[node_index];
59 }
