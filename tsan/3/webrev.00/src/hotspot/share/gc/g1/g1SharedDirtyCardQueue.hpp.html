25 #ifndef SHARE_GC_G1_G1SHAREDDIRTYCARDQUEUE_HPP
26 #define SHARE_GC_G1_G1SHAREDDIRTYCARDQUEUE_HPP
27 
28 #include "utilities/globalDefinitions.hpp"
29 
30 class G1DirtyCardQueueSet;
31 
32 // A dirty card queue providing thread-safe enqueue.  A shared global
33 // instance can be used for cases where a thread-local dirty card can't
34 // be used.
35 class G1SharedDirtyCardQueue {
36   G1DirtyCardQueueSet* const _qset;
37   void** _buffer;
38   size_t _index;
39 
40   NONCOPYABLE(G1SharedDirtyCardQueue);
41 
42 public:
43   G1SharedDirtyCardQueue(G1DirtyCardQueueSet* qset);
44   ~G1SharedDirtyCardQueue();    // flushes the queue.
45 
46   // Thread-safe addition to shared logging buffer.
47   void enqueue(void* card_ptr);
48 
49   // Flush any pending entries to the qset and remove the buffer.
50   // Not thread-safe.
51   void flush();
52 
53   // Discard any pending entries.
54   // Not thread-safe.
55   void reset();
56 };
57 
58 #endif // SHARE_GC_G1_G1SHAREDDIRTYCARDQUEUE_HPP
