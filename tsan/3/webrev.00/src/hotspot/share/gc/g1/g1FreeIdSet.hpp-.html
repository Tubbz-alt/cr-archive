25 #ifndef SHARE_GC_G1_G1FREEIDSET_HPP
26 #define SHARE_GC_G1_G1FREEIDSET_HPP
27 
28 #include "memory/allocation.hpp"
29 #include "runtime/semaphore.hpp"
30 #include "utilities/globalDefinitions.hpp"
31 
32 // Represents a set of small integer ids, from which elements can be
33 // temporarily allocated for exclusive use.  The ids are in a
34 // contiguous range from 'start' to 'start + size'.  Used to obtain a
35 // distinct worker_id value for a mutator thread that doesn't normally
36 // have such an id.
37 class G1FreeIdSet : public CHeapObj<mtGC> {
38   Semaphore _sem;
39   uint* _next;
40   uint _start;
41   uint _size;
42   uintx _head_index_mask;
43   volatile uintx _head;
44 
45   uint head_index(uintx head) const;
46   uintx make_head(uint index, uintx old_head) const;
47 
48   // Noncopyable.
49   G1FreeIdSet(const G1FreeIdSet&);
50   G1FreeIdSet& operator=(const G1FreeIdSet&);
51 
52 public:
53   G1FreeIdSet(uint start, uint size);
54   ~G1FreeIdSet();
55 
56   // Returns an unclaimed parallel id (waiting for one to be released if
57   // necessary).  Must not safepoint while holding a claimed id.
58   uint claim_par_id();
59 
60   void release_par_id(uint id);
61 
62   struct TestSupport;           // For unit test access.
63 };
64 
65 #endif // SHARE_GC_G1_G1FREEIDSET_HPP
