 32 void G1EvacStats::log_plab_allocation() {
 33   PLABStats::log_plab_allocation();
 34   log_debug(gc, plab)("%s other allocation: "
 35                       "region end waste: " SIZE_FORMAT "B, "
 36                       "regions filled: %u, "
 37                       "direct allocated: " SIZE_FORMAT "B, "
 38                       "failure used: " SIZE_FORMAT "B, "
 39                       "failure wasted: " SIZE_FORMAT "B",
 40                       _description,
 41                       _region_end_waste * HeapWordSize,
 42                       _regions_filled,
 43                       _direct_allocated * HeapWordSize,
 44                       _failure_used * HeapWordSize,
 45                       _failure_waste * HeapWordSize);
 46 }

 48 size_t G1EvacStats::compute_desired_plab_sz() {
 49   // The size of the PLAB caps the amount of space that can be wasted at the
 50   // end of the collection. In the worst case the last PLAB could be completely
 51   // empty.
 52   // This allows us to calculate the new PLAB size to achieve the
 53   // TargetPLABWastePct given the latest memory usage and that the last buffer
 54   // will be G1LastPLABAverageOccupancy full.
 55   //
 56   // E.g. assume that if in the current GC 100 words were allocated and a
 57   // TargetPLABWastePct of 10 had been set.
 58   //
 59   // So we could waste up to 10 words to meet that percentage. Given that we
 60   // also assume that that buffer is typically half-full, the new desired PLAB
 61   // size is set to 20 words.
 62   //
 63   // The amount of allocation performed should be independent of the number of
 64   // threads, so should the maximum waste we can spend in total. So if
 65   // we used n threads to allocate, each of them can spend maximum waste/n words in
 66   // a first rough approximation. The number of threads only comes into play later
 67   // when actually retrieving the actual desired PLAB size.
 68   //
 69   // After calculating this optimal PLAB size the algorithm applies the usual
 70   // exponential decaying average over this value to guess the next PLAB size.
 71   //
 72   // We account region end waste fully to PLAB allocation (in the calculation of
 73   // what we consider as "used_for_waste_calculation" below). This is not
 74   // completely fair, but is a conservative assumption because PLABs may be sized
 75   // flexibly while we cannot adjust inline allocations.
 76   // Allocation during GC will try to minimize region end waste so this impact
 77   // should be minimal.
 78   //
 79   // We need to cover overflow when calculating the amount of space actually used
 80   // by objects in PLABs when subtracting the region end waste.
 81   // Region end waste may be higher than actual allocation. This may occur if many
 82   // threads do not allocate anything but a few rather large objects. In this
 83   // degenerate case the PLAB size would simply quickly tend to minimum PLAB size,
 84   // which is an okay reaction.
 85   size_t const used_for_waste_calculation = used() > _region_end_waste ? used() - _region_end_waste : 0;
 86 
 87   size_t const total_waste_allowed = used_for_waste_calculation * TargetPLABWastePct;
 88   size_t const cur_plab_sz = (size_t)((double)total_waste_allowed / G1LastPLABAverageOccupancy);
 89   return cur_plab_sz;
 90 }
