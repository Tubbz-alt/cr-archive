31 CSpaceCounters::CSpaceCounters(const char* name, int ordinal, size_t max_size,
32                                ContiguousSpace* s, GenerationCounters* gc) :
33    _space(s) {
34 
35   if (UsePerfData) {
36     EXCEPTION_MARK;
37     ResourceMark rm;
38 
39     const char* cns = PerfDataManager::name_space(gc->name_space(), "space",
40                                                   ordinal);
41 
42     _name_space = NEW_C_HEAP_ARRAY(char, strlen(cns)+1, mtGC);
43     strcpy(_name_space, cns);
44 
45     const char* cname = PerfDataManager::counter_name(_name_space, "name");
46     PerfDataManager::create_string_constant(SUN_GC, cname, name, CHECK);
47 
48     cname = PerfDataManager::counter_name(_name_space, "maxCapacity");
49     _max_capacity = PerfDataManager::create_variable(SUN_GC, cname, PerfData::U_Bytes,
50                                      (jlong)max_size, CHECK);
51 
52     cname = PerfDataManager::counter_name(_name_space, "capacity");
53     _capacity = PerfDataManager::create_variable(SUN_GC, cname,
54                                                  PerfData::U_Bytes,
55                                                  _space->capacity(), CHECK);
56 
57     cname = PerfDataManager::counter_name(_name_space, "used");
58     _used = PerfDataManager::create_variable(SUN_GC, cname, PerfData::U_Bytes,
59                                     new ContiguousSpaceUsedHelper(_space),
60                                     CHECK);
61 
62     cname = PerfDataManager::counter_name(_name_space, "initCapacity");
63     PerfDataManager::create_constant(SUN_GC, cname, PerfData::U_Bytes,
64                                      _space->capacity(), CHECK);
65   }
66 }
67 
68 CSpaceCounters::~CSpaceCounters() {
69   FREE_C_HEAP_ARRAY(char, _name_space);
70 }
71 
72 void CSpaceCounters::update_capacity() {
73   _capacity->set_value(_space->capacity());
74 }
75 
76 void CSpaceCounters::update_used() {
77   _used->set_value(_space->used());
78 }
79 
80 void CSpaceCounters::update_all() {
81   update_used();
82   update_capacity();
83 }
84 
85 jlong ContiguousSpaceUsedHelper::take_sample(){
86   return _space->used();
87 }
