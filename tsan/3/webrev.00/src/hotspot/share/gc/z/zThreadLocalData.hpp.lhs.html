24 #ifndef SHARE_GC_Z_ZTHREADLOCALDATA_HPP
25 #define SHARE_GC_Z_ZTHREADLOCALDATA_HPP
26 
27 #include "gc/z/zMarkStack.hpp"
28 #include "gc/z/zGlobals.hpp"
29 #include "runtime/thread.hpp"
30 #include "utilities/debug.hpp"
31 #include "utilities/sizes.hpp"
32 
33 class ZThreadLocalData {
34 private:
35   uintptr_t              _address_bad_mask;
36   ZMarkThreadLocalStacks _stacks;

37 
38   ZThreadLocalData() :
39       _address_bad_mask(0),
40       _stacks() {}

41 
42   static ZThreadLocalData* data(Thread* thread) {
43     return thread->gc_data<ZThreadLocalData>();
44   }
45 
46 public:
47   static void create(Thread* thread) {
48     new (data(thread)) ZThreadLocalData();
49   }
50 
51   static void destroy(Thread* thread) {
52     data(thread)->~ZThreadLocalData();
53   }
54 
55   static void set_address_bad_mask(Thread* thread, uintptr_t mask) {
56     data(thread)->_address_bad_mask = mask;
57   }
58 
59   static ZMarkThreadLocalStacks* stacks(Thread* thread) {
60     return &data(thread)->_stacks;
61   }
62 

















63   static ByteSize address_bad_mask_offset() {
64     return Thread::gc_data_offset() + byte_offset_of(ZThreadLocalData, _address_bad_mask);
65   }
66 
67   static ByteSize nmethod_disarmed_offset() {
68     return address_bad_mask_offset() + in_ByteSize(ZNMethodDisarmedOffset);
69   }
70 };
71 
72 #endif // SHARE_GC_Z_ZTHREADLOCALDATA_HPP
