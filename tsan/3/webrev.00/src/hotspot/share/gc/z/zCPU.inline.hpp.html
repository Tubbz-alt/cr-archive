24 #ifndef SHARE_GC_Z_ZCPU_INLINE_HPP
25 #define SHARE_GC_Z_ZCPU_INLINE_HPP
26 
27 #include "gc/z/zCPU.hpp"
28 #include "runtime/os.hpp"
29 #include "utilities/debug.hpp"
30 
31 inline uint32_t ZCPU::count() {
32   return os::processor_count();
33 }
34 
35 inline uint32_t ZCPU::id() {
36   assert(_affinity != NULL, "Not initialized");
37 
38   // Fast path
39   if (_affinity[_cpu]._thread == _self) {
40     return _cpu;
41   }
42 
43   // Slow path
44   return id_slow();
45 }
46 
47 #endif // SHARE_GC_Z_ZCPU_INLINE_HPP
