24 #ifndef SHARE_GC_Z_ZARRAY_HPP
25 #define SHARE_GC_Z_ZARRAY_HPP
26 
27 #include "memory/allocation.hpp"
28 
29 template <typename T>
30 class ZArray {
31 private:
32   static const size_t initial_capacity = 32;
33 
34   T*     _array;
35   size_t _size;
36   size_t _capacity;
37 
38   // Copy and assignment are not allowed
39   ZArray(const ZArray<T>& array);
40   ZArray<T>& operator=(const ZArray<T>& array);
41 
42   void expand(size_t new_capacity);
43 
44 public:
45   ZArray();
46   ~ZArray();
47 
48   size_t size() const;
49   bool is_empty() const;
50 
51   T at(size_t index) const;
52 
53   void add(T value);
54   void clear();
55 };
56 
57 template <typename T, bool parallel>
58 class ZArrayIteratorImpl : public StackObj {
59 private:
60   ZArray<T>* const _array;
61   size_t           _next;
62 
63 public:
64   ZArrayIteratorImpl(ZArray<T>* array);
65 
66   bool next(T* elem);
67 };
68 
69 // Iterator types
70 #define ZARRAY_SERIAL      false
71 #define ZARRAY_PARALLEL    true
72 
73 template <typename T>
74 class ZArrayIterator : public ZArrayIteratorImpl<T, ZARRAY_SERIAL> {
75 public:
76   ZArrayIterator(ZArray<T>* array) :
77       ZArrayIteratorImpl<T, ZARRAY_SERIAL>(array) {}
78 };
79 
80 template <typename T>
81 class ZArrayParallelIterator : public ZArrayIteratorImpl<T, ZARRAY_PARALLEL> {
82 public:
83   ZArrayParallelIterator(ZArray<T>* array) :
84       ZArrayIteratorImpl<T, ZARRAY_PARALLEL>(array) {}
85 };
86 
87 #endif // SHARE_GC_Z_ZARRAY_HPP
