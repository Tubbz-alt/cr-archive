 44 inline void ZNMethodOopClosure::do_oop(oop* p) {
 45   if (ZResurrection::is_blocked()) {
 46     ZBarrier::keep_alive_barrier_on_phantom_root_oop_field(p);

 48     ZBarrier::load_barrier_on_root_oop_field(p);
 49   }

 70 template <bool finalizable>
 71 inline void ZMarkBarrierOopClosure<finalizable>::do_klass(Klass* k) {
 72   ClassLoaderData* const cld = k->class_loader_data();
 73   ZMarkBarrierOopClosure<finalizable>::do_cld(cld);
 74 }

 76 template <bool finalizable>
 77 inline void ZMarkBarrierOopClosure<finalizable>::do_cld(ClassLoaderData* cld) {
 78   const int claim = finalizable ? ClassLoaderData::_claim_finalizable : ClassLoaderData::_claim_strong;
 79   cld->oops_do(this, claim);
 80 }

 94 inline void ZPhantomCleanOopClosure::do_oop(oop* p) {
 95   // Read the oop once, to make sure the liveness check
 96   // and the later clearing uses the same value.
 97   const oop obj = *(volatile oop*)p;
 98   if (ZBarrier::is_alive_barrier_on_phantom_oop(obj)) {
 99     ZBarrier::keep_alive_barrier_on_phantom_oop_field(p);
100   } else {
101     // The destination could have been modified/reused, in which case
102     // we don't want to clear it. However, no one could write the same
103     // oop here again (the object would be strongly live and we would
104     // not consider clearing such oops), so therefore we don't have an
105     // ABA problem here.
106     Atomic::cmpxchg(oop(NULL), p, obj);
107   }
108 }
