31 ZPerWorker<ThreadLocalAllocStats>* ZThreadLocalAllocBuffer::_stats = NULL;
32 
33 void ZThreadLocalAllocBuffer::initialize() {
34   if (UseTLAB) {
35     assert(_stats == NULL, "Already initialized");
36     _stats = new ZPerWorker<ThreadLocalAllocStats>();
37     reset_statistics();
38   }
39 }
40 
41 void ZThreadLocalAllocBuffer::reset_statistics() {
42   if (UseTLAB) {
43     ZPerWorkerIterator<ThreadLocalAllocStats> iter(_stats);
44     for (ThreadLocalAllocStats* stats; iter.next(&stats);) {
45       stats->reset();
46     }
47   }
48 }
49 
50 void ZThreadLocalAllocBuffer::publish_statistics() {
51   if (UseTLAB) {
52     ThreadLocalAllocStats total;
53 
54     ZPerWorkerIterator<ThreadLocalAllocStats> iter(_stats);
55     for (ThreadLocalAllocStats* stats; iter.next(&stats);) {
56       total.update(*stats);
57     }
58 
59     total.publish();
60   }
61 }
62 
63 static void fixup_address(HeapWord** p) {
64   *p = (HeapWord*)ZAddress::good_or_null((uintptr_t)*p);
65 }
66 
67 void ZThreadLocalAllocBuffer::retire(Thread* thread) {
68   if (UseTLAB && thread->is_Java_thread()) {
69     ThreadLocalAllocStats* const stats = _stats->addr();
70     thread->tlab().addresses_do(fixup_address);
71     thread->tlab().retire(stats);
72     if (ResizeTLAB) {
73       thread->tlab().resize();
74     }
75   }
76 }
77 
78 void ZThreadLocalAllocBuffer::remap(Thread* thread) {
79   if (UseTLAB && thread->is_Java_thread()) {
80     thread->tlab().addresses_do(fixup_address);
81   }
82 }
