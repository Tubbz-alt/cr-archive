 94 inline void ZPhantomCleanOopClosure::do_oop(oop* p) {
 95   // Read the oop once, to make sure the liveness check
 96   // and the later clearing uses the same value.
 97   const oop obj = *(volatile oop*)p;
 98   if (ZBarrier::is_alive_barrier_on_phantom_oop(obj)) {
 99     ZBarrier::keep_alive_barrier_on_phantom_oop_field(p);
100   } else {
101     // The destination could have been modified/reused, in which case
102     // we don't want to clear it. However, no one could write the same
103     // oop here again (the object would be strongly live and we would
104     // not consider clearing such oops), so therefore we don't have an
105     // ABA problem here.
106     Atomic::cmpxchg(oop(NULL), p, obj);
107   }
108 }
