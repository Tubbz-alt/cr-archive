24 #include "precompiled.hpp"

25 #include "gc/z/zGlobals.hpp"
26 #include "gc/z/zVirtualMemory.inline.hpp"

27 #include "services/memTracker.hpp"


28 
29 ZVirtualMemoryManager::ZVirtualMemoryManager() :
30     _manager(),
31     _initialized(false) {
32   // Reserve address space
33   if (!reserve(ZAddressSpaceStart, ZAddressSpaceSize)) {



34     return;
35   }
36 
37   // Make the complete address view free
38   _manager.free(0, ZAddressOffsetMax);



39 
40   // Register address space with native memory tracker
41   nmt_reserve(ZAddressSpaceStart, ZAddressSpaceSize);
42 
43   // Successfully initialized
44   _initialized = true;
45 }
46 
























































































47 void ZVirtualMemoryManager::nmt_reserve(uintptr_t start, size_t size) {
48   MemTracker::record_virtual_memory_reserve((void*)start, size, CALLER_PC);
49   MemTracker::record_virtual_memory_type((void*)start, mtJavaHeap);
50 }
51 
52 bool ZVirtualMemoryManager::is_initialized() const {
53   return _initialized;
54 }
55 
56 ZVirtualMemory ZVirtualMemoryManager::alloc(size_t size, bool alloc_from_front) {
57   uintptr_t start;
58 
59   if (alloc_from_front || size <= ZPageSizeSmall) {
60     // Small page
61     start = _manager.alloc_from_front(size);
62   } else {
63     // Medium/Large page
64     start = _manager.alloc_from_back(size);
65   }
66 
67   return ZVirtualMemory(start, size);
68 }
69 
70 void ZVirtualMemoryManager::free(ZVirtualMemory vmem) {
71   _manager.free(vmem.start(), vmem.size());
72 }
