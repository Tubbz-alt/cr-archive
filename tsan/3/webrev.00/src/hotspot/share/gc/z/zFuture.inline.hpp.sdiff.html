40 template <typename T>
41 inline T ZFuture<T>::get() {
42   // Wait for notification
43   Thread* const thread = Thread::current();
44   if (thread->is_Java_thread()) {
45     _sema.wait_with_safepoint_check((JavaThread*)thread);
46   } else {
47     _sema.wait();
