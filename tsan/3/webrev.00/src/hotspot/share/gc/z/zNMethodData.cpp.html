34 ZNMethodDataOops* ZNMethodDataOops::create(const GrowableArray<oop*>& immediates, bool has_non_immediates) {
35   return ::new (AttachedArray::alloc(immediates.length())) ZNMethodDataOops(immediates, has_non_immediates);
36 }
37 
38 void ZNMethodDataOops::destroy(ZNMethodDataOops* oops) {
39   AttachedArray::free(oops);
40 }
41 
42 ZNMethodDataOops::ZNMethodDataOops(const GrowableArray<oop*>& immediates, bool has_non_immediates) :
43     _immediates(immediates.length()),
44     _has_non_immediates(has_non_immediates) {
45   // Save all immediate oops
46   for (size_t i = 0; i < immediates_count(); i++) {
47     immediates_begin()[i] = immediates.at(int(i));
48   }
49 }
50 
51 size_t ZNMethodDataOops::immediates_count() const {
52   return _immediates.length();
53 }
54 
55 oop** ZNMethodDataOops::immediates_begin() const {
56   return _immediates(this);
57 }
58 
59 oop** ZNMethodDataOops::immediates_end() const {
60   return immediates_begin() + immediates_count();
61 }
62 
63 bool ZNMethodDataOops::has_non_immediates() const {
64   return _has_non_immediates;
65 }
66 
67 ZNMethodData::ZNMethodData() :
68     _lock(),
69     _oops(NULL) {}
70 
71 ZNMethodData::~ZNMethodData() {
72   ZNMethodDataOops::destroy(_oops);
73 }
74 
75 ZReentrantLock* ZNMethodData::lock() {
76   return &_lock;
77 }
78 
79 ZNMethodDataOops* ZNMethodData::oops() const {
80   return Atomic::load_acquire(&_oops);
81 }
82 
83 ZNMethodDataOops* ZNMethodData::swap_oops(ZNMethodDataOops* new_oops) {
84   ZLocker<ZReentrantLock> locker(&_lock);
85   ZNMethodDataOops* const old_oops = _oops;
86   _oops = new_oops;
87   return old_oops;
88 }
