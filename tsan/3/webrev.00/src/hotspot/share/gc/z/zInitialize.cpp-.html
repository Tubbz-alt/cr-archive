38 ZInitialize::ZInitialize(ZBarrierSet* barrier_set) {
39   log_info(gc, init)("Initializing %s", ZGCName);
40   log_info(gc, init)("Version: %s (%s)",
41                      VM_Version::vm_release(),
42                      VM_Version::jdk_debug_level());
43 
44   // Early initialization
45   ZAddressMasks::initialize();
46   ZNUMA::initialize();
47   ZCPU::initialize();
48   ZStatValue::initialize();
49   ZThreadLocalAllocBuffer::initialize();
50   ZTracer::initialize();
51   ZLargePages::initialize();
52   ZBarrierSet::set_barrier_set(barrier_set);
53 }
