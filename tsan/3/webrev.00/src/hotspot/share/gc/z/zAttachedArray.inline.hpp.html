24 #ifndef SHARE_GC_Z_ZATTACHEDARRAY_INLINE_HPP
25 #define SHARE_GC_Z_ZATTACHEDARRAY_INLINE_HPP
26 
27 #include "gc/z/zAttachedArray.hpp"
28 #include "memory/allocation.hpp"
29 #include "utilities/align.hpp"
30 
31 template <typename ObjectT, typename ArrayT>
32 inline size_t ZAttachedArray<ObjectT, ArrayT>::object_size() {
33   return align_up(sizeof(ObjectT), sizeof(ArrayT));
34 }
35 
36 template <typename ObjectT, typename ArrayT>
37 inline void* ZAttachedArray<ObjectT, ArrayT>::alloc(size_t length) {
38   const size_t array_size = sizeof(ArrayT) * length;
39   char* const addr = AllocateHeap(object_size() + array_size, mtGC);
40   ::new (addr + object_size()) ArrayT[length];
41   return addr;
42 }
43 
44 template <typename ObjectT, typename ArrayT>
45 inline void ZAttachedArray<ObjectT, ArrayT>::free(ObjectT* obj) {
46   FreeHeap(obj);
47 }
48 
49 template <typename ObjectT, typename ArrayT>
50 inline ZAttachedArray<ObjectT, ArrayT>::ZAttachedArray(size_t length) :
51     _length(length) {}
52 
53 template <typename ObjectT, typename ArrayT>
54 inline size_t ZAttachedArray<ObjectT, ArrayT>::length() const {
55   return _length;
56 }
57 
58 template <typename ObjectT, typename ArrayT>
59 inline ArrayT* ZAttachedArray<ObjectT, ArrayT>::operator()(const ObjectT* obj) const {
60   return reinterpret_cast<ArrayT*>(reinterpret_cast<uintptr_t>(obj) + object_size());
61 }
62 
63 #endif // SHARE_GC_Z_ZATTACHEDARRAY_INLINE_HPP
