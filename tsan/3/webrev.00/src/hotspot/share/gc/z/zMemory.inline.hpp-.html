24 #ifndef SHARE_GC_Z_ZMEMORY_INLINE_HPP
25 #define SHARE_GC_Z_ZMEMORY_INLINE_HPP
26 
27 #include "gc/z/zMemory.hpp"
28 #include "utilities/debug.hpp"
29 
30 inline ZMemory::ZMemory(uintptr_t start, size_t size) :
31     _start(start),
32     _end(start + size) {}
33 
34 inline uintptr_t ZMemory::start() const {
35   return _start;
36 }
37 
38 inline uintptr_t ZMemory::end() const {
39   return _end;
40 }
41 
42 inline size_t ZMemory::size() const {
43   return end() - start();
44 }
45 
46 inline void ZMemory::shrink_from_front(size_t size) {
47   assert(this->size() > size, "Too small");
48   _start += size;
49 }
50 
51 inline void ZMemory::shrink_from_back(size_t size) {
52   assert(this->size() > size, "Too small");
53   _end -= size;
54 }
55 
56 inline void ZMemory::grow_from_front(size_t size) {
57   assert(start() >= size, "Too big");
58   _start -= size;
59 }
60 
61 inline void ZMemory::grow_from_back(size_t size) {
62   _end += size;
63 }
64 
65 #endif // SHARE_GC_Z_ZMEMORY_INLINE_HPP
