24 #ifndef SHARE_GC_Z_ZMARKTERMINATE_INLINE_HPP
25 #define SHARE_GC_Z_ZMARKTERMINATE_INLINE_HPP
26 
27 #include "runtime/atomic.hpp"
28 
29 inline ZMarkTerminate::ZMarkTerminate() :
30     _nworkers(0),
31     _nworking_stage0(0),
32     _nworking_stage1(0) {}
33 
34 inline bool ZMarkTerminate::enter_stage(volatile uint* nworking_stage) {
35   return Atomic::sub(nworking_stage, 1u) == 0;
36 }
37 
38 inline void ZMarkTerminate::exit_stage(volatile uint* nworking_stage) {
39   Atomic::add(nworking_stage, 1u);
40 }
41 
42 inline bool ZMarkTerminate::try_exit_stage(volatile uint* nworking_stage) {
43   uint nworking = Atomic::load(nworking_stage);
44 
45   for (;;) {
46     if (nworking == 0) {
47       return false;
48     }
49 
50     const uint new_nworking = nworking + 1;
51     const uint prev_nworking = Atomic::cmpxchg(nworking_stage, nworking, new_nworking);
52     if (prev_nworking == nworking) {
53       // Success
54       return true;
55     }
56 
57     // Retry
58     nworking = prev_nworking;
59   }
60 }
61 
62 inline void ZMarkTerminate::reset(uint nworkers) {
63   _nworkers = _nworking_stage0 = _nworking_stage1 = nworkers;
64 }
65 
66 inline bool ZMarkTerminate::enter_stage0() {
67   return enter_stage(&_nworking_stage0);
68 }
69 
70 inline void ZMarkTerminate::exit_stage0() {
71   exit_stage(&_nworking_stage0);
72 }
73 
74 inline bool ZMarkTerminate::try_exit_stage0() {
75   return try_exit_stage(&_nworking_stage0);
76 }
77 
78 inline bool ZMarkTerminate::enter_stage1() {
79   return enter_stage(&_nworking_stage1);
80 }
81 
82 inline bool ZMarkTerminate::try_exit_stage1() {
83   return try_exit_stage(&_nworking_stage1);
84 }
85 
86 #endif // SHARE_GC_Z_ZMARKTERMINATE_INLINE_HPP
