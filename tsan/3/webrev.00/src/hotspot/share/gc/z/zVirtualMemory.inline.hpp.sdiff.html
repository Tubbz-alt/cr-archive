34 inline ZVirtualMemory::ZVirtualMemory(uintptr_t start, size_t size) :
35     _start(start),
36     _end(start + size) {}
37 
38 inline bool ZVirtualMemory::is_null() const {
39   return _start == UINTPTR_MAX;
40 }
41 
42 inline uintptr_t ZVirtualMemory::start() const {
43   return _start;
44 }
45 
46 inline uintptr_t ZVirtualMemory::end() const {
47   return _end;
48 }
49 
50 inline size_t ZVirtualMemory::size() const {
51   return _end - _start;
52 }
53 
54 inline ZVirtualMemory ZVirtualMemory::split(size_t split_size) {
55   assert(split_size <= size(), "precondition");
56   ZVirtualMemory mem(_start, split_size);
57   _start += split_size;
58   return mem;
59 }
60 
61 inline void ZVirtualMemory::clear() {
62   _start = UINTPTR_MAX;
63   _end = UINTPTR_MAX;
64 }
