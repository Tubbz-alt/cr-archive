28 void ZAddress::set_good_mask(uintptr_t mask) {
29   ZAddressGoodMask = mask;
30   ZAddressBadMask = ZAddressGoodMask ^ ZAddressMetadataMask;
31   ZAddressWeakBadMask = (ZAddressGoodMask | ZAddressMetadataRemapped | ZAddressMetadataFinalizable) ^ ZAddressMetadataMask;
32 }
33 
34 void ZAddress::initialize() {
35   ZAddressOffsetBits = ZPlatformAddressOffsetBits();
36   ZAddressOffsetMask = (((uintptr_t)1 << ZAddressOffsetBits) - 1) << ZAddressOffsetShift;
37   ZAddressOffsetMax = (uintptr_t)1 << ZAddressOffsetBits;
38 
39   ZAddressMetadataShift = ZPlatformAddressMetadataShift();
40   ZAddressMetadataMask = (((uintptr_t)1 << ZAddressMetadataBits) - 1) << ZAddressMetadataShift;
41 
42   ZAddressMetadataMarked0 = (uintptr_t)1 << (ZAddressMetadataShift + 0);
43   ZAddressMetadataMarked1 = (uintptr_t)1 << (ZAddressMetadataShift + 1);
44   ZAddressMetadataRemapped = (uintptr_t)1 << (ZAddressMetadataShift + 2);
45   ZAddressMetadataFinalizable = (uintptr_t)1 << (ZAddressMetadataShift + 3);
46 
47   ZAddressMetadataMarked = ZAddressMetadataMarked0;
48   set_good_mask(ZAddressMetadataRemapped);
49 }
50 
51 void ZAddress::flip_to_marked() {
52   ZAddressMetadataMarked ^= (ZAddressMetadataMarked0 | ZAddressMetadataMarked1);
53   set_good_mask(ZAddressMetadataMarked);
54 }
55 
56 void ZAddress::flip_to_remapped() {
57   set_good_mask(ZAddressMetadataRemapped);
58 }
