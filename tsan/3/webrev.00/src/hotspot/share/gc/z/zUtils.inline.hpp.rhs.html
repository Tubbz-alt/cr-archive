35 inline size_t ZUtils::bytes_to_words(size_t size_in_bytes) {
36   assert(is_aligned(size_in_bytes, BytesPerWord), "Size not word aligned");
37   return size_in_bytes >> LogBytesPerWord;
38 }
39 
40 inline size_t ZUtils::words_to_bytes(size_t size_in_words) {
41   return size_in_words << LogBytesPerWord;
42 }
43 
44 inline size_t ZUtils::object_size(uintptr_t addr) {
45   return words_to_bytes(ZOop::from_address(addr)->size());
46 }
47 
48 inline void ZUtils::object_copy(uintptr_t from, uintptr_t to, size_t size) {
49   Copy::aligned_disjoint_words((HeapWord*)from, (HeapWord*)to, bytes_to_words(size));
50 }
