35 void ZVerifyOopClosure::do_oop(oop* p) {
36   guarantee(SafepointSynchronize::is_at_safepoint(), "Must be at a safepoint");
37   guarantee(ZGlobalPhase == ZPhaseMarkCompleted, "Invalid phase");
38   guarantee(!ZResurrection::is_blocked(), "Invalid phase");
39 
40   const oop o = RawAccess<>::oop_load(p);
41   if (o != NULL) {
42     guarantee(ZOop::is_good(o) || ZOop::is_finalizable_good(o),
43               "Bad oop " PTR_FORMAT " found at " PTR_FORMAT ", expected " PTR_FORMAT,
44               p2i(o), p2i(p), p2i(ZOop::good(o)));
45     guarantee(oopDesc::is_oop(ZOop::good(o)),
46               "Bad object " PTR_FORMAT " found at " PTR_FORMAT,
47               p2i(o), p2i(p));
48   }
49 }
