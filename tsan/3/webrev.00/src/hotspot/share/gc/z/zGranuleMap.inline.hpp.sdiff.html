 32 template <typename T>

 41 template <typename T>
 42 inline size_t ZGranuleMap<T>::index_for_addr(uintptr_t addr) const {
 43   assert(!ZAddress::is_null(addr), "Invalid address");
 44 
 45   const size_t index = ZAddress::offset(addr) >> ZGranuleSizeShift;
 46   assert(index < size(), "Invalid index");
 47 
 48   return index;
 49 }
 50 
 51 template <typename T>
 52 inline size_t ZGranuleMap<T>::size() const {
 53   return ZAddressOffsetMax >> ZGranuleSizeShift;

 54 }

 56 template <typename T>
 57 inline T ZGranuleMap<T>::get(uintptr_t addr) const {
 58   const size_t index = index_for_addr(addr);
 59   return _map[index];
 60 }

 63 inline void ZGranuleMap<T>::put(uintptr_t addr, T value) {
 64   const size_t index = index_for_addr(addr);
 65   _map[index] = value;





 66 }

 73 template <typename T>
 74 inline bool ZGranuleMapIterator<T>::next(T* value) {
 75   if (_next < _map->size()) {
 76     *value = _map->_map[_next++];

 33 template <typename T>
 34 inline ZGranuleMap<T>::ZGranuleMap(size_t max_offset) :
 35     _size(max_offset >> ZGranuleSizeShift),
 36     _map(MmapArrayAllocator<T>::allocate(_size, mtGC)) {
 37   assert(is_aligned(max_offset, ZGranuleSize), "Misaligned");
 38 }
 39 
 40 template <typename T>
 41 inline ZGranuleMap<T>::~ZGranuleMap() {
 42   MmapArrayAllocator<T>::free(_map, _size);
 43 }
 44 
 45 template <typename T>
 46 inline size_t ZGranuleMap<T>::index_for_offset(uintptr_t offset) const {
 47   const size_t index = offset >> ZGranuleSizeShift;
 48   assert(index < _size, "Invalid index");


 49 
 50   return index;
 51 }
 52 
 53 template <typename T>
 54 inline T ZGranuleMap<T>::get(uintptr_t offset) const {
 55   const size_t index = index_for_offset(offset);
 56   return _map[index];
 57 }
 58 
 59 template <typename T>
 60 inline void ZGranuleMap<T>::put(uintptr_t offset, T value) {
 61   const size_t index = index_for_offset(offset);
 62   _map[index] = value;
 63 }
 64 
 65 template <typename T>
 66 inline void ZGranuleMap<T>::put(uintptr_t offset, size_t size, T value) {
 67   assert(is_aligned(size, ZGranuleSize), "Misaligned");
 68 
 69   const size_t start_index = index_for_offset(offset);
 70   const size_t end_index = start_index + (size >> ZGranuleSizeShift);
 71   for (size_t index = start_index; index < end_index; index++) {
 72     _map[index] = value;
 73   }
 74 }

 76 template <typename T>
 77 inline ZGranuleMapIterator<T>::ZGranuleMapIterator(const ZGranuleMap<T>* map) :
 78     _map(map),
 79     _next(0) {}

 81 template <typename T>
 82 inline bool ZGranuleMapIterator<T>::next(T* value) {
 83   if (_next < _map->_size) {
 84     *value = _map->_map[_next++];
 85     return true;
 86   }
 87 
 88   // End of map
 89   return false;
 90 }
 91 
 92 template <typename T>
 93 inline bool ZGranuleMapIterator<T>::next(T** value) {
 94   if (_next < _map->_size) {
 95     *value = _map->_map + _next++;
 96     return true;
 97   }
 98 
 99   // End of map
100   return false;
101 }
