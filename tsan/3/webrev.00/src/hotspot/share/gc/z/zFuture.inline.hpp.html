24 #ifndef SHARE_GC_Z_ZFUTURE_INLINE_HPP
25 #define SHARE_GC_Z_ZFUTURE_INLINE_HPP
26 
27 #include "gc/z/zFuture.hpp"
28 #include "runtime/semaphore.inline.hpp"
29 #include "runtime/thread.hpp"
30 
31 template <typename T>
32 inline ZFuture<T>::ZFuture() :
33     _value() {}
34 
35 template <typename T>
36 inline void ZFuture<T>::set(T value) {
37   // Set value
38   _value = value;
39 
40   // Notify waiter
41   _sema.signal();
42 }
43 
44 template <typename T>
45 inline T ZFuture<T>::peek() {
46   return _value;
47 }
48 
49 template <typename T>
50 inline T ZFuture<T>::get() {
51   // Wait for notification
52   Thread* const thread = Thread::current();
53   if (thread->is_Java_thread()) {
54     _sema.wait_with_safepoint_check((JavaThread*)thread);
55   } else {
56     _sema.wait();
57   }
58 
59   // Return value
60   return _value;
61 }
62 
63 #endif // SHARE_GC_Z_ZFUTURE_INLINE_HPP
