24 #ifndef SHARE_GC_Z_ZVIRTUALMEMORY_INLINE_HPP
25 #define SHARE_GC_Z_ZVIRTUALMEMORY_INLINE_HPP
26 
27 #include "gc/z/zMemory.inline.hpp"
28 #include "gc/z/zVirtualMemory.hpp"
29 
30 inline ZVirtualMemory::ZVirtualMemory() :
31     _start(UINTPTR_MAX),
32     _end(UINTPTR_MAX) {}
33 
34 inline ZVirtualMemory::ZVirtualMemory(uintptr_t start, size_t size) :
35     _start(start),
36     _end(start + size) {}
37 
38 inline bool ZVirtualMemory::is_null() const {
39   return _start == UINTPTR_MAX;
40 }
41 
42 inline uintptr_t ZVirtualMemory::start() const {
43   return _start;
44 }
45 
46 inline uintptr_t ZVirtualMemory::end() const {
47   return _end;
48 }
49 
50 inline size_t ZVirtualMemory::size() const {
51   return _end - _start;
52 }
53 
54 inline ZVirtualMemory ZVirtualMemory::split(size_t size) {
55   _start += size;
56   return ZVirtualMemory(_start - size, size);







57 }
58 
59 #endif // SHARE_GC_Z_ZVIRTUALMEMORY_INLINE_HPP
