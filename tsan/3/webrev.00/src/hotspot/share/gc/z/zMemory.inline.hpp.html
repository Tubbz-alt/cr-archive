24 #ifndef SHARE_GC_Z_ZMEMORY_INLINE_HPP
25 #define SHARE_GC_Z_ZMEMORY_INLINE_HPP
26 
27 #include "gc/z/zList.inline.hpp"
28 #include "gc/z/zMemory.hpp"
29 #include "utilities/debug.hpp"
30 
31 inline ZMemory::ZMemory(uintptr_t start, size_t size) :
32     _start(start),
33     _end(start + size) {}
34 
35 inline uintptr_t ZMemory::start() const {
36   return _start;
37 }
38 
39 inline uintptr_t ZMemory::end() const {
40   return _end;
41 }
42 
43 inline size_t ZMemory::size() const {
44   return end() - start();
45 }
46 
47 inline void ZMemory::shrink_from_front(size_t size) {
48   assert(this->size() > size, "Too small");
49   _start += size;
50 }
51 
52 inline void ZMemory::shrink_from_back(size_t size) {
53   assert(this->size() > size, "Too small");
54   _end -= size;
55 }
56 
57 inline void ZMemory::grow_from_front(size_t size) {
58   assert(start() >= size, "Too big");
59   _start -= size;
60 }
61 
62 inline void ZMemory::grow_from_back(size_t size) {
63   _end += size;
64 }
65 
66 #endif // SHARE_GC_Z_ZMEMORY_INLINE_HPP
