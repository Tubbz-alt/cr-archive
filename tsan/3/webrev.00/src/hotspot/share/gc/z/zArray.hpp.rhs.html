24 #ifndef SHARE_GC_Z_ZARRAY_HPP
25 #define SHARE_GC_Z_ZARRAY_HPP
26 
27 #include "memory/allocation.hpp"
28 #include "utilities/globalDefinitions.hpp"
29 
30 template <typename T>
31 class ZArray {
32 private:
33   static const size_t initial_capacity = 32;
34 
35   T*     _array;
36   size_t _size;
37   size_t _capacity;
38 
39   NONCOPYABLE(ZArray);


40 
41   void expand(size_t new_capacity);
42 
43 public:
44   ZArray();
45   ~ZArray();
46 
47   size_t size() const;
48   bool is_empty() const;
49 
50   T at(size_t index) const;
51 
52   void add(T value);
53   void transfer(ZArray<T>* from);
54   void clear();
55 };
56 
57 template <typename T, bool parallel>
58 class ZArrayIteratorImpl : public StackObj {
59 private:
60   ZArray<T>* const _array;
61   size_t           _next;
62 
63 public:
64   ZArrayIteratorImpl(ZArray<T>* array);
65 
66   bool next(T* elem);
67 };
68 
69 // Iterator types
70 #define ZARRAY_SERIAL      false
71 #define ZARRAY_PARALLEL    true
72 
73 template <typename T>
74 class ZArrayIterator : public ZArrayIteratorImpl<T, ZARRAY_SERIAL> {
75 public:
76   ZArrayIterator(ZArray<T>* array);

77 };
78 
79 template <typename T>
80 class ZArrayParallelIterator : public ZArrayIteratorImpl<T, ZARRAY_PARALLEL> {
81 public:
82   ZArrayParallelIterator(ZArray<T>* array);

83 };

85 #endif // SHARE_GC_Z_ZARRAY_HPP
