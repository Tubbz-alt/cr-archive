30 void ZForwardingTable::setup(size_t live_objects) {
31   assert(is_null(), "Should be empty");
32   assert(live_objects > 0, "Invalid size");
33 
34   // Allocate table for linear probing. The size of the table must be
35   // a power of two to allow for quick and inexpensive indexing/masking.
36   // The table is sized to have a load factor of 50%, i.e. sized to have
37   // double the number of entries actually inserted.
38   _size = ZUtils::round_up_power_of_2(live_objects * 2);
39   _table = MallocArrayAllocator<ZForwardingTableEntry>::allocate(_size, mtGC);
40 
41   // Construct table entries
42   for (size_t i = 0; i < _size; i++) {
43     ::new (_table + i) ZForwardingTableEntry();
44   }
45 }
46 
47 void ZForwardingTable::reset() {
48   // Destruct table entries
49   for (size_t i = 0; i < _size; i++) {
50     (_table + i)->~ZForwardingTableEntry();
51   }
52 
53   // Free table
54   MallocArrayAllocator<ZForwardingTableEntry>::free(_table);
55   _table = NULL;
56   _size = 0;
57 }
58 
59 void ZForwardingTable::verify(size_t object_max_count, size_t live_objects) const {
60   size_t count = 0;
61 
62   for (size_t i = 0; i < _size; i++) {
63     const ZForwardingTableEntry entry = _table[i];
64     if (entry.is_empty()) {
65       // Skip empty entries
66       continue;
67     }
68 
69     // Check from index
70     guarantee(entry.from_index() < object_max_count, "Invalid from index");
71 
72     // Check for duplicates
73     for (size_t j = i + 1; j < _size; j++) {
74       const ZForwardingTableEntry other = _table[j];
75       guarantee(entry.from_index() != other.from_index(), "Duplicate from");
76       guarantee(entry.to_offset() != other.to_offset(), "Duplicate to");
77     }
78 
79     count++;
80   }
81 
82   // Check number of non-null entries
83   guarantee(live_objects == count, "Count mismatch");
84 }
