31 
32 ZPageTable::ZPageTable() :
33     _map(ZAddressOffsetMax) {}
34 
35 void ZPageTable::insert(ZPage* page) {
36   const uintptr_t offset = page->start();
37   const size_t size = page->size();
38 
39   // Make sure a newly created page is
40   // visible before updating the page table.
41   OrderAccess::storestore();
42 
43   assert(_map.get(offset) == NULL, "Invalid entry");
44   _map.put(offset, size, page);
45 }
46 
47 void ZPageTable::remove(ZPage* page) {
48   const uintptr_t offset = page->start();
49   const size_t size = page->size();
50 
51   assert(_map.get(offset) == page, "Invalid entry");
52   _map.put(offset, size, NULL);
53 }
