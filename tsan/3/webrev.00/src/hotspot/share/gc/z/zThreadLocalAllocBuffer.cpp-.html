24 #include "precompiled.hpp"

26 #include "gc/z/zThreadLocalAllocBuffer.hpp"

30 ZPerWorker<ThreadLocalAllocStats>* ZThreadLocalAllocBuffer::_stats = NULL;
31 
32 void ZThreadLocalAllocBuffer::initialize() {
33   if (UseTLAB) {
34     assert(_stats == NULL, "Already initialized");
35     _stats = new ZPerWorker<ThreadLocalAllocStats>();
36     reset_statistics();
37   }
38 }
39 
40 void ZThreadLocalAllocBuffer::reset_statistics() {
41   if (UseTLAB) {
42     ZPerWorkerIterator<ThreadLocalAllocStats> iter(_stats);
43     for (ThreadLocalAllocStats* stats; iter.next(&stats);) {
44       stats->reset();
45     }
46   }
47 }
48 
49 void ZThreadLocalAllocBuffer::publish_statistics() {
50   if (UseTLAB) {
51     ThreadLocalAllocStats total;
52 
53     ZPerWorkerIterator<ThreadLocalAllocStats> iter(_stats);
54     for (ThreadLocalAllocStats* stats; iter.next(&stats);) {
55       total.update(*stats);
56     }
57 
58     total.publish();
59   }
60 }
61 
62 static void fixup_address(HeapWord** p) {
63   *p = (HeapWord*)ZAddress::good_or_null((uintptr_t)*p);
64 }
65 
66 void ZThreadLocalAllocBuffer::retire(Thread* thread) {
67   if (UseTLAB && thread->is_Java_thread()) {
68     ThreadLocalAllocStats* const stats = _stats->addr();
69     thread->tlab().addresses_do(fixup_address);
70     thread->tlab().retire(stats);
71     thread->tlab().resize();
72   }
73 }
74 
75 void ZThreadLocalAllocBuffer::remap(Thread* thread) {
76   if (UseTLAB && thread->is_Java_thread()) {
77     thread->tlab().addresses_do(fixup_address);
78   }
79 }
