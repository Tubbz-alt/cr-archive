24 #ifndef SHARE_GC_Z_ZTHREAD_INLINE_HPP
25 #define SHARE_GC_Z_ZTHREAD_INLINE_HPP
26 
27 #include "gc/z/zThread.hpp"
28 #include "utilities/debug.hpp"
29 
30 inline void ZThread::ensure_initialized() {
31   if (!_initialized) {
32     initialize();
33   }
34 }
35 
36 inline uintptr_t ZThread::id() {
37   ensure_initialized();
38   return _id;
39 }
40 
41 inline bool ZThread::is_vm() {
42   ensure_initialized();
43   return _is_vm;
44 }
45 
46 inline bool ZThread::is_java() {
47   ensure_initialized();
48   return _is_java;
49 }
50 
51 inline bool ZThread::is_worker() {
52   ensure_initialized();
53   return _is_worker;
54 }
55 
56 inline bool ZThread::is_runtime_worker() {
57   ensure_initialized();
58   return _is_runtime_worker;
59 }
60 
61 inline uint ZThread::worker_id() {
62   assert(has_worker_id(), "Worker id not initialized");
63   return _worker_id;
64 }
65 
66 #endif // SHARE_GC_Z_ZTHREAD_INLINE_HPP
