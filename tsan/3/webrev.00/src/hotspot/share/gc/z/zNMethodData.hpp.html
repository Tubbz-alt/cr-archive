33 class nmethod;
34 template <typename T> class GrowableArray;
35 
36 class ZNMethodDataOops {
37 private:
38   typedef ZAttachedArray<ZNMethodDataOops, oop*> AttachedArray;
39 
40   const AttachedArray _immediates;
41   const bool          _has_non_immediates;
42 
43   ZNMethodDataOops(const GrowableArray<oop*>& immediates, bool has_non_immediates);
44 
45 public:
46   static ZNMethodDataOops* create(const GrowableArray<oop*>& immediates, bool has_non_immediates);
47   static void destroy(ZNMethodDataOops* oops);
48 
49   size_t immediates_count() const;
50   oop** immediates_begin() const;
51   oop** immediates_end() const;
52 
53   bool has_non_immediates() const;
54 };
55 
56 class ZNMethodData : public CHeapObj<mtGC> {
57 private:
58   ZReentrantLock             _lock;
59   ZNMethodDataOops* volatile _oops;
60 
61 public:
62   ZNMethodData();
63   ~ZNMethodData();
64 
65   ZReentrantLock* lock();
66 
67   ZNMethodDataOops* oops() const;
68   ZNMethodDataOops* swap_oops(ZNMethodDataOops* oops);
69 };
