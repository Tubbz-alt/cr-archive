43 inline bool ZPageTableIterator::next(ZPage** page) {
44   ZPageTableEntry entry;
45 
46   while (_iter.next(&entry)) {
47     ZPage* const next = entry.page();
48     if (next != NULL && next != _prev) {
49       // Next page found
50       *page = _prev = next;
51       return true;
52     }
53   }
54 
55   // No more pages
56   return false;
57 }
