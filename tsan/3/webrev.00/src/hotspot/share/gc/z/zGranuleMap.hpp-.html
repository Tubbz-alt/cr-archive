24 #ifndef SHARE_GC_Z_ZGRANULEMAP_HPP
25 #define SHARE_GC_Z_ZGRANULEMAP_HPP
26 
27 #include "memory/allocation.hpp"
28 
29 template<typename T>
30 class ZGranuleMapIterator;
31 
32 template <typename T>
33 class ZGranuleMap {
34   friend class VMStructs;
35   friend class ZGranuleMapIterator<T>;
36 
37 private:
38   T* const _map;
39 
40   size_t index_for_addr(uintptr_t addr) const;
41   size_t size() const;
42 
43 public:
44   ZGranuleMap();
45   ~ZGranuleMap();
46 
47   T get(uintptr_t addr) const;
48   void put(uintptr_t addr, T value);
49 };
50 
51 template <typename T>
52 class ZGranuleMapIterator : public StackObj {
53 public:
54   const ZGranuleMap<T>* const _map;
55   size_t                      _next;
56 
57 public:
58   ZGranuleMapIterator(const ZGranuleMap<T>* map);
59 
60   bool next(T* value);
61 };
62 
63 #endif // SHARE_GC_Z_ZGRANULEMAP_HPP
