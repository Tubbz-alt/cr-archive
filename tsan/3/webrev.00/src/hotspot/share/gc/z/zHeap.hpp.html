 47 private:

 63   size_t heap_min_size() const;
 64   size_t heap_initial_size() const;

 66   size_t heap_max_reserve_size() const;

 68   void flip_to_marked();

 71   void out_of_memory();

 79   bool is_initialized() const;
 80 
 81   // Heap metrics
 82   size_t min_capacity() const;
 83   size_t max_capacity() const;
 84   size_t soft_max_capacity() const;
 85   size_t capacity() const;
 86   size_t max_reserve() const;
 87   size_t used_high() const;

107   void print_worker_threads_on(outputStream* st) const;
108 
109   // Reference processing
110   ReferenceDiscoverer* reference_discoverer();

112 
113   // Non-strong reference processing
114   void process_non_strong_references();
115 
116   // Page allocation
117   ZPage* alloc_page(uint8_t type, size_t size, ZAllocationFlags flags);

119   void free_page(ZPage* page, bool reclaimed);
120 
121   // Uncommit memory
122   uint64_t uncommit(uint64_t delay);
123 
124   // Object allocation
125   uintptr_t alloc_tlab(size_t size);
126   uintptr_t alloc_object(size_t size);
127   uintptr_t alloc_object_for_relocation(size_t size);
128   void undo_alloc_object_for_relocation(uintptr_t addr, size_t size);
129   bool is_alloc_stalled() const;
130   void check_out_of_memory();
131 
132   // Marking
133   bool is_object_live(uintptr_t addr) const;
134   bool is_object_strongly_live(uintptr_t addr) const;

136   void mark_start();
137   void mark(bool initial);
138   void mark_flush_and_free(Thread* thread);

141 
142   // Relocation set
143   void select_relocation_set();
144   void reset_relocation_set();
145 
146   // Relocation
147   void relocate_start();
148   uintptr_t relocate_object(uintptr_t addr);
149   uintptr_t remap_object(uintptr_t addr);
150   void relocate();
151 
152   // Iteration

154   void pages_do(ZPageClosure* cl);
155 
156   // Serviceability
157   void serviceability_initialize();
158   GCMemoryManager* serviceability_memory_manager();
159   MemoryPool* serviceability_memory_pool();
160   ZServiceabilityCounters* serviceability_counters();
161 
162   // Printing

165   bool print_location(outputStream* st, uintptr_t addr) const;
166 
167   // Verification
168   bool is_oop(uintptr_t addr) const;
169   void verify();
170 };
