24 #include "precompiled.hpp"
25 #include "gc/z/zAddressSpaceLimit.hpp"
26 #include "gc/z/zGlobals.hpp"
27 #include "runtime/os.hpp"
28 #include "utilities/align.hpp"
29 
30 static size_t address_space_limit() {
31   julong limit = 0;
32 
33   if (os::has_allocatable_memory_limit(&limit)) {
34     return (size_t)limit;
35   }
36 
37   // No limit
38   return SIZE_MAX;
39 }
40 
41 size_t ZAddressSpaceLimit::mark_stack() {
42   // Allow mark stacks to occupy 10% of the address space
43   const size_t limit = address_space_limit() / 10;
44   return align_up(limit, ZMarkStackSpaceExpandSize);
45 }
46 
47 size_t ZAddressSpaceLimit::heap_view() {
48   // Allow all heap views to occupy 50% of the address space
49   const size_t limit = address_space_limit() / 2 / ZHeapViews;
50   return align_up(limit, ZGranuleSize);
51 }
