24 #ifndef SHARE_GC_Z_ZPHYSICALMEMORY_INLINE_HPP
25 #define SHARE_GC_Z_ZPHYSICALMEMORY_INLINE_HPP
26 
27 #include "gc/z/zPhysicalMemory.hpp"
28 #include "utilities/debug.hpp"
29 
30 inline ZPhysicalMemorySegment::ZPhysicalMemorySegment() :
31     _start(UINTPTR_MAX),
32     _end(UINTPTR_MAX) {}
33 
34 inline ZPhysicalMemorySegment::ZPhysicalMemorySegment(uintptr_t start, size_t size) :
35     _start(start),
36     _end(start + size) {}
37 
38 inline uintptr_t ZPhysicalMemorySegment::start() const {
39   return _start;
40 }
41 
42 inline uintptr_t ZPhysicalMemorySegment::end() const {
43   return _end;
44 }
45 
46 inline size_t ZPhysicalMemorySegment::size() const {
47   return _end - _start;
48 }
49 
50 inline bool ZPhysicalMemory::is_null() const {
51   return _nsegments == 0;
52 }
53 
54 inline size_t ZPhysicalMemory::nsegments() const {
55   return _nsegments;
56 }
57 
58 inline const ZPhysicalMemorySegment& ZPhysicalMemory::segment(size_t index) const {
59   assert(index < _nsegments, "Invalid segment index");
60   return _segments[index];
61 }
62 
63 #endif // SHARE_GC_Z_ZPHYSICALMEMORY_INLINE_HPP
