29 void ZAddressMasks::set_good_mask(uintptr_t mask) {
30   uintptr_t old_bad_mask = ZAddressBadMask;
31   ZAddressGoodMask = mask;
32   ZAddressBadMask = ZAddressGoodMask ^ ZAddressMetadataMask;
33   ZAddressWeakBadMask = (ZAddressGoodMask | ZAddressMetadataRemapped | ZAddressMetadataFinalizable) ^ ZAddressMetadataMask;
34 }
35 
36 void ZAddressMasks::initialize() {












37   ZAddressMetadataMarked = ZAddressMetadataMarked0;
38   set_good_mask(ZAddressMetadataRemapped);
39 }
40 
41 void ZAddressMasks::flip_to_marked() {
42   ZAddressMetadataMarked ^= (ZAddressMetadataMarked0 | ZAddressMetadataMarked1);
43   set_good_mask(ZAddressMetadataMarked);
44 }
45 
46 void ZAddressMasks::flip_to_remapped() {
47   set_good_mask(ZAddressMetadataRemapped);
48 }
