24 #ifndef SHARE_GC_Z_ZRELOCATIONSET_HPP
25 #define SHARE_GC_Z_ZRELOCATIONSET_HPP
26 
27 #include "memory/allocation.hpp"
28 
29 class ZForwarding;
30 class ZPage;
31 
32 class ZRelocationSet {
33   template <bool> friend class ZRelocationSetIteratorImpl;
34 
35 private:
36   ZForwarding** _forwardings;
37   size_t        _nforwardings;
38 
39 public:
40   ZRelocationSet();
41 
42   void populate(ZPage* const* group0, size_t ngroup0,
43                 ZPage* const* group1, size_t ngroup1);
44   void reset();
45 };
46 
47 template <bool parallel>
48 class ZRelocationSetIteratorImpl : public StackObj {
49 private:
50   ZRelocationSet* const _relocation_set;
51   size_t                _next;
52 
53 public:
54   ZRelocationSetIteratorImpl(ZRelocationSet* relocation_set);
55 
56   bool next(ZForwarding** forwarding);
57 };
58 
59 // Iterator types
60 #define ZRELOCATIONSET_SERIAL      false
61 #define ZRELOCATIONSET_PARALLEL    true
62 
63 class ZRelocationSetIterator : public ZRelocationSetIteratorImpl<ZRELOCATIONSET_SERIAL> {
64 public:
65   ZRelocationSetIterator(ZRelocationSet* relocation_set) :
66       ZRelocationSetIteratorImpl<ZRELOCATIONSET_SERIAL>(relocation_set) {}
67 };
68 
69 class ZRelocationSetParallelIterator : public ZRelocationSetIteratorImpl<ZRELOCATIONSET_PARALLEL> {
70 public:
71   ZRelocationSetParallelIterator(ZRelocationSet* relocation_set) :
72       ZRelocationSetIteratorImpl<ZRELOCATIONSET_PARALLEL>(relocation_set) {}
73 };
74 
75 #endif // SHARE_GC_Z_ZRELOCATIONSET_HPP
