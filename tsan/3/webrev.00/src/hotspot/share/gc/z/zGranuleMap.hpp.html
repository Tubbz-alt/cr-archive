24 #ifndef SHARE_GC_Z_ZGRANULEMAP_HPP
25 #define SHARE_GC_Z_ZGRANULEMAP_HPP
26 
27 #include "memory/allocation.hpp"
28 
29 template<typename T>
30 class ZGranuleMapIterator;
31 
32 template <typename T>
33 class ZGranuleMap {
34   friend class VMStructs;
35   friend class ZGranuleMapIterator<T>;
36 
37 private:
38   const size_t _size;
39   T* const     _map;
40 
41   size_t index_for_offset(uintptr_t offset) const;
42 
43 public:
44   ZGranuleMap(size_t max_offset);
45   ~ZGranuleMap();
46 
47   T get(uintptr_t offset) const;
48   void put(uintptr_t offset, T value);
49   void put(uintptr_t offset, size_t size, T value);
50 };
51 
52 template <typename T>
53 class ZGranuleMapIterator : public StackObj {
54 public:
55   const ZGranuleMap<T>* const _map;
56   size_t                      _next;
57 
58 public:
59   ZGranuleMapIterator(const ZGranuleMap<T>* map);
60 
61   bool next(T* value);
62   bool next(T** value);
63 };
64 
65 #endif // SHARE_GC_Z_ZGRANULEMAP_HPP
