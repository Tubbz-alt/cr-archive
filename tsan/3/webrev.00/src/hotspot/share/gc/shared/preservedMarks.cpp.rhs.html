 34 void PreservedMarks::restore() {
 35   while (!_stack.is_empty()) {
 36     const OopAndMarkWord elem = _stack.pop();
 37     elem.set_mark();
 38   }
 39   assert_empty();
 40 }
 41 
 42 void PreservedMarks::adjust_during_full_gc() {
 43   StackIterator<OopAndMarkWord, mtGC> iter(_stack);
 44   while (!iter.is_empty()) {
 45     OopAndMarkWord* elem = iter.next_addr();
 46 
 47     oop obj = elem->get_oop();
 48     if (obj->is_forwarded()) {
 49       elem->set_oop(obj->forwardee());
 50     }
 51   }
 52 }
 53 
 54 void PreservedMarks::restore_and_increment(volatile size_t* const total_size_addr) {
 55   const size_t stack_size = size();
 56   restore();
 57   // Only do the atomic add if the size is > 0.
 58   if (stack_size > 0) {
 59     Atomic::add(total_size_addr, stack_size);
 60   }
 61 }

 64 void PreservedMarks::assert_empty() {

 66          _stack.size());
 67   assert(_stack.cache_size() == 0,
 68          "stack expected to have no cached segments, cache size = " SIZE_FORMAT,
 69          _stack.cache_size());

 73 void RemoveForwardedPointerClosure::do_object(oop obj) {
 74   if (obj->is_forwarded()) {
 75     PreservedMarks::init_forwarded_mark(obj);
 76   }
 77 }

 79 void PreservedMarksSet::init(uint num) {
 80   assert(_stacks == NULL && _num == 0, "do not re-initialize");
 81   assert(num > 0, "pre-condition");
 82   if (_in_c_heap) {
 83     _stacks = NEW_C_HEAP_ARRAY(Padded<PreservedMarks>, num, mtGC);
 84   } else {
 85     _stacks = NEW_RESOURCE_ARRAY(Padded<PreservedMarks>, num);
 86   }
 87   for (uint i = 0; i < num; i += 1) {
 88     ::new (_stacks + i) PreservedMarks();
 89   }
 90   _num = num;
 91 
 92   assert_empty();
 93 }

 95 class ParRestoreTask : public AbstractGangTask {
 96 private:
 97   PreservedMarksSet* const _preserved_marks_set;
 98   SequentialSubTasksDone _sub_tasks;
 99   volatile size_t* const _total_size_addr;
100 
101 public:
102   virtual void work(uint worker_id) {
103     uint task_id = 0;
104     while (_sub_tasks.try_claim_task(/* reference */ task_id)) {
105       _preserved_marks_set->get(task_id)->restore_and_increment(_total_size_addr);
106     }
107     _sub_tasks.all_tasks_completed();
108   }
109 
110   ParRestoreTask(uint worker_num,
111                  PreservedMarksSet* preserved_marks_set,
112                  volatile size_t* total_size_addr)
113       : AbstractGangTask("Parallel Preserved Mark Restoration"),
114         _preserved_marks_set(preserved_marks_set),
115         _total_size_addr(total_size_addr) {
116     _sub_tasks.set_n_threads(worker_num);
117     _sub_tasks.set_n_tasks(preserved_marks_set->num());
118   }
119 };

121 void PreservedMarksSet::restore(WorkGang* workers) {
122   volatile size_t total_size = 0;
123 
124 #ifdef ASSERT
125   // This is to make sure the total_size we'll calculate below is correct.
126   size_t total_size_before = 0;
127   for (uint i = 0; i < _num; i += 1) {
128     total_size_before += get(i)->size();
129   }
130 #endif // def ASSERT
131 
132   if (workers == NULL) {
133     for (uint i = 0; i < num(); i += 1) {
134       total_size += get(i)->size();
135       get(i)->restore();
136     }
137   } else {
138     ParRestoreTask task(workers->active_workers(), this, &total_size);
139     workers->run_task(&task);
140   }
141 
142   assert_empty();
143 
144   assert(total_size == total_size_before,
145          "total_size = " SIZE_FORMAT " before = " SIZE_FORMAT,
146          total_size, total_size_before);
147 
148   log_trace(gc)("Restored " SIZE_FORMAT " marks", total_size);
149 }
150 
151 void PreservedMarksSet::reclaim() {
152   assert_empty();
153 
154   for (uint i = 0; i < _num; i += 1) {
155     _stacks[i].~Padded<PreservedMarks>();
156   }
157 
158   if (_in_c_heap) {
159     FREE_C_HEAP_ARRAY(Padded<PreservedMarks>, _stacks);
160   } else {
161     // the array was resource-allocated, so nothing to do
162   }
163   _stacks = NULL;
164   _num = 0;
165 }
166 
167 #ifndef PRODUCT
168 void PreservedMarksSet::assert_empty() {
169   assert(_stacks != NULL && _num > 0, "should have been initialized");
170   for (uint i = 0; i < _num; i += 1) {
171     get(i)->assert_empty();
172   }
173 }
