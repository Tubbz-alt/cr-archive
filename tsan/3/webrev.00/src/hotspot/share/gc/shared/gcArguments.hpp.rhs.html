26 #ifndef SHARE_GC_SHARED_GCARGUMENTS_HPP
27 #define SHARE_GC_SHARED_GCARGUMENTS_HPP
28 
29 #include "memory/allocation.hpp"
30 
31 class CollectedHeap;
32 
33 extern size_t HeapAlignment;
34 extern size_t SpaceAlignment;
35 
36 class GCArguments {
37 protected:
38   // Initialize HeapAlignment, SpaceAlignment, and extra alignments (E.g. GenAlignment)
39   virtual void initialize_alignments() = 0;
40   virtual void initialize_heap_flags_and_sizes();
41   virtual void initialize_size_info();
42 
43   DEBUG_ONLY(void assert_flags();)
44   DEBUG_ONLY(void assert_size_info();)
45 
46 public:
47   virtual void initialize();
48   virtual size_t conservative_max_heap_alignment() = 0;
49   virtual CollectedHeap* create_heap() = 0;
50 
51   // Allows GCs to tell external code if it's supported or not in the current setup.
52   virtual bool is_supported() const {
53     return true;
54   }
55 
56   void initialize_heap_sizes();
57 
58   static size_t compute_heap_alignment();
59 
60   static bool check_args_consistency();
61 };
62 
63 #endif // SHARE_GC_SHARED_GCARGUMENTS_HPP
