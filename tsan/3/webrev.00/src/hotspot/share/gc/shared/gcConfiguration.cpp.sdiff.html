 33 GCName GCConfiguration::young_collector() const {
 34   if (UseG1GC) {
 35     return G1New;
 36   }
 37 
 38   if (UseParallelGC) {
 39     return ParallelScavenge;
 40   }
 41 
 42   if (UseConcMarkSweepGC) {
 43     return ParNew;
 44   }
 45 
 46   if (UseZGC || UseShenandoahGC) {
 47     return NA;
 48   }
 49 
 50   return DefNew;
 51 }
