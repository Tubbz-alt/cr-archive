25 #ifndef SHARE_GC_SHARED_GCNAME_HPP
26 #define SHARE_GC_SHARED_GCNAME_HPP
27 
28 #include "utilities/debug.hpp"
29 
30 enum GCName {
31   ParallelOld,
32   SerialOld,
33   PSMarkSweep,
34   ParallelScavenge,
35   DefNew,
36   ParNew,
37   G1New,
38   ConcurrentMarkSweep,
39   G1Old,
40   G1Full,
41   Z,
42   Shenandoah,
43   NA,
44   GCNameEndSentinel
45 };
46 
47 class GCNameHelper {
48  public:
49   static const char* to_string(GCName name) {
50     switch(name) {
51       case ParallelOld: return "ParallelOld";
52       case SerialOld: return "SerialOld";
53       case PSMarkSweep: return "PSMarkSweep";
54       case ParallelScavenge: return "ParallelScavenge";
55       case DefNew: return "DefNew";
56       case ParNew: return "ParNew";
57       case G1New: return "G1New";
58       case ConcurrentMarkSweep: return "ConcurrentMarkSweep";
59       case G1Old: return "G1Old";
60       case G1Full: return "G1Full";
61       case Z: return "Z";
62       case Shenandoah: return "Shenandoah";
63       case NA: return "N/A";
64       default: ShouldNotReachHere(); return NULL;
65     }
66   }
67 };
68 
69 #endif // SHARE_GC_SHARED_GCNAME_HPP
