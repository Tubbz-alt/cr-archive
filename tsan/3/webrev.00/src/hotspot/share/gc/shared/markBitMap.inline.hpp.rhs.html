34 inline HeapWord* MarkBitMap::get_next_marked_addr(const HeapWord* addr,
35                                                 const HeapWord* limit) const {
36   assert(limit != NULL, "limit must not be NULL");
37   // Round addr up to a possible object boundary to be safe.
38   size_t const addr_offset = addr_to_offset(align_up(addr, HeapWordSize << _shifter));
39   size_t const limit_offset = addr_to_offset(limit);
40   size_t const nextOffset = _bm.get_next_one_offset(addr_offset, limit_offset);
41   return offset_to_addr(nextOffset);
42 }
43 
44 inline void MarkBitMap::mark(HeapWord* addr) {
45   check_mark(addr);
46   _bm.set_bit(addr_to_offset(addr));
47 }
48 
49 inline void MarkBitMap::mark(oop obj) {
50   return mark(cast_from_oop<HeapWord*>(obj));
51 }
52 
53 inline void MarkBitMap::clear(HeapWord* addr) {
54   check_mark(addr);
55   _bm.clear_bit(addr_to_offset(addr));
56 }
57 
58 inline bool MarkBitMap::par_mark(HeapWord* addr) {
59   check_mark(addr);
60   return _bm.par_set_bit(addr_to_offset(addr));
61 }
62 
63 inline bool MarkBitMap::par_mark(oop obj) {
64   return par_mark(cast_from_oop<HeapWord*>(obj));
65 }
66 
67 inline bool MarkBitMap::is_marked(oop obj) const{
68   return is_marked(cast_from_oop<HeapWord*>(obj));
69 }
70 
71 inline void MarkBitMap::clear(oop obj) {
72   clear(cast_from_oop<HeapWord*>(obj));
73 }
