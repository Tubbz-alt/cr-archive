31 CollectorCounters::CollectorCounters(const char* name, int ordinal) {
32 
33   if (UsePerfData) {
34     EXCEPTION_MARK;
35     ResourceMark rm;
36 
37     const char* cns = PerfDataManager::name_space("collector", ordinal);
38 
39     _name_space = NEW_C_HEAP_ARRAY(char, strlen(cns)+1, mtGC);
40     strcpy(_name_space, cns);
41 
42     char* cname = PerfDataManager::counter_name(_name_space, "name");
43     PerfDataManager::create_string_constant(SUN_GC, cname, name, CHECK);
44 
45     cname = PerfDataManager::counter_name(_name_space, "invocations");
46     _invocations = PerfDataManager::create_counter(SUN_GC, cname,
47                                                    PerfData::U_Events, CHECK);
48 
49     cname = PerfDataManager::counter_name(_name_space, "time");
50     _time = PerfDataManager::create_counter(SUN_GC, cname, PerfData::U_Ticks,
51                                             CHECK);
52 
53     cname = PerfDataManager::counter_name(_name_space, "lastEntryTime");
54     _last_entry_time = PerfDataManager::create_variable(SUN_GC, cname,
55                                                         PerfData::U_Ticks,
56                                                         CHECK);
57 
58     cname = PerfDataManager::counter_name(_name_space, "lastExitTime");
59     _last_exit_time = PerfDataManager::create_variable(SUN_GC, cname,
60                                                        PerfData::U_Ticks,
61                                                        CHECK);
62   }
63 }
64 
65 CollectorCounters::~CollectorCounters() {
66   if (_name_space != NULL) {
67     FREE_C_HEAP_ARRAY(char, _name_space);
68   }
69 }
70 
71 TraceCollectorStats::TraceCollectorStats(CollectorCounters* c) :
72     PerfTraceTimedEvent(c->time_counter(), c->invocation_counter()),
73     _c(c) {
74 
75   if (UsePerfData) {
76      _c->last_entry_counter()->set_value(os::elapsed_counter());
77   }
78 }
79 
80 TraceCollectorStats::~TraceCollectorStats() {
81   if (UsePerfData) {
82     _c->last_exit_counter()->set_value(os::elapsed_counter());
83   }
84 }
