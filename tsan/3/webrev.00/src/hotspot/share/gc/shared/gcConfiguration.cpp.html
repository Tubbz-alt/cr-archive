 35 GCName GCConfiguration::young_collector() const {
 36   if (UseG1GC) {
 37     return G1New;
 38   }
 39 
 40   if (UseParallelGC) {
 41     return ParallelScavenge;
 42   }
 43 
 44   if (UseZGC || UseShenandoahGC) {
 45     return NA;
 46   }
 47 
 48   return DefNew;
 49 }
 50 
 51 GCName GCConfiguration::old_collector() const {
 52   if (UseG1GC) {
 53     return G1Old;
 54   }
 55 
 56   if (UseParallelGC) {
 57     return ParallelOld;
 58   }
 59 
 60   if (UseZGC) {
 61     return Z;
 62   }
 63 
 64   if (UseShenandoahGC) {
 65     return Shenandoah;
 66   }
 67 
 68   return SerialOld;
 69 }
