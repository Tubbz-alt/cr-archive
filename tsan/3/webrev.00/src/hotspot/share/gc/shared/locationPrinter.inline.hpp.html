25 #ifndef SHARE_GC_SHARED_LOCATIONPRINTER_INLINE_HPP
26 #define SHARE_GC_SHARED_LOCATIONPRINTER_INLINE_HPP
27 
28 #include "gc/shared/locationPrinter.hpp"
29 #include "oops/compressedOops.inline.hpp"
30 #include "oops/oopsHierarchy.hpp"
31 
32 template <typename CollectedHeapT>
33 oop BlockLocationPrinter<CollectedHeapT>::base_oop_or_null(void* addr) {
34   if (is_valid_obj(addr)) {
35     // We were just given an oop directly.
36     return oop(addr);
37   }
38 
39   // Try to find addr using block_start.
40   HeapWord* p = CollectedHeapT::heap()->block_start(addr);
41   if (p != NULL && CollectedHeapT::heap()->block_is_obj(p)) {
42     if (!is_valid_obj(p)) {
43       return NULL;
44     }
45     return oop(p);
46   }
47 
48   return NULL;
49 }
50 
51 template <typename CollectedHeapT>
52 bool BlockLocationPrinter<CollectedHeapT>::print_location(outputStream* st, void* addr) {
53   // Check if addr points into Java heap.
54   if (CollectedHeapT::heap()->is_in(addr)) {
55     oop o = base_oop_or_null(addr);
56     if (o != NULL) {
57       if ((void*)o == addr) {
58         st->print(INTPTR_FORMAT " is an oop: ", p2i(addr));
59       } else {
60         st->print(INTPTR_FORMAT " is pointing into object: " , p2i(addr));
61       }
62       o->print_on(st);
63       return true;
64     }
65   } else if (CollectedHeapT::heap()->is_in_reserved(addr)) {
66     st->print_cr(INTPTR_FORMAT " is an unallocated location in the heap", p2i(addr));
67     return true;
68   }
69 
70   // Compressed oop needs to be decoded first.
71 #ifdef _LP64
72   if (UseCompressedOops && ((uintptr_t)addr &~ (uintptr_t)max_juint) == 0) {
73     narrowOop narrow_oop = (narrowOop)(uintptr_t)addr;
74     oop o = CompressedOops::decode_raw(narrow_oop);
75 
76     if (is_valid_obj(o)) {
77       st->print(UINT32_FORMAT " is a compressed pointer to object: ", narrow_oop);
78       o->print_on(st);
79       return true;
80     }
81   }
82 #endif
83 
84   return false;
85 }
