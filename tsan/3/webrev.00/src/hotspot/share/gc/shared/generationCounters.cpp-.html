30 void GenerationCounters::initialize(const char* name, int ordinal, int spaces,
31                                     size_t min_capacity, size_t max_capacity,
32                                     size_t curr_capacity) {
33   if (UsePerfData) {
34     EXCEPTION_MARK;
35     ResourceMark rm;
36 
37     const char* cns = PerfDataManager::name_space("generation", ordinal);
38 
39     _name_space = NEW_C_HEAP_ARRAY(char, strlen(cns)+1, mtGC);
40     strcpy(_name_space, cns);
41 
42     const char* cname = PerfDataManager::counter_name(_name_space, "name");
43     PerfDataManager::create_string_constant(SUN_GC, cname, name, CHECK);
44 
45     cname = PerfDataManager::counter_name(_name_space, "spaces");
46     PerfDataManager::create_constant(SUN_GC, cname, PerfData::U_None,
47                                      spaces, CHECK);
48 
49     cname = PerfDataManager::counter_name(_name_space, "minCapacity");
50     PerfDataManager::create_constant(SUN_GC, cname, PerfData::U_Bytes,
51                                      min_capacity, CHECK);
52 
53     cname = PerfDataManager::counter_name(_name_space, "maxCapacity");
54     PerfDataManager::create_constant(SUN_GC, cname, PerfData::U_Bytes,
55                                      max_capacity, CHECK);
56 
57     cname = PerfDataManager::counter_name(_name_space, "capacity");
58     _current_size =
59       PerfDataManager::create_variable(SUN_GC, cname, PerfData::U_Bytes,
60                                        curr_capacity, CHECK);
61   }
62 }
63 
64 GenerationCounters::GenerationCounters(const char* name,
65                                        int ordinal, int spaces,
66                                        size_t min_capacity, size_t max_capacity,
67                                        VirtualSpace* v)
68   : _virtual_space(v) {
69   assert(v != NULL, "don't call this constructor if v == NULL");
70   initialize(name, ordinal, spaces,
71              min_capacity, max_capacity, v->committed_size());
72 }
73 
74 GenerationCounters::GenerationCounters(const char* name,
75                                        int ordinal, int spaces,
76                                        size_t min_capacity, size_t max_capacity,
77                                        size_t curr_capacity)
78   : _virtual_space(NULL) {
79   initialize(name, ordinal, spaces, min_capacity, max_capacity, curr_capacity);
80 }
81 
82 GenerationCounters::~GenerationCounters() {
83   if (_name_space != NULL) {
84     FREE_C_HEAP_ARRAY(char, _name_space);
85   }
86 }
87 
88 void GenerationCounters::update_all() {
89   assert(_virtual_space != NULL, "otherwise, override this method");
90   _current_size->set_value(_virtual_space->committed_size());
91 }
