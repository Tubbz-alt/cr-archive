26 #ifndef SHARE_GC_SHARED_GCOVERHEADCHECKER_HPP
27 #define SHARE_GC_SHARED_GCOVERHEADCHECKER_HPP
28 
29 #include "memory/allocation.hpp"
30 #include "gc/shared/gcCause.hpp"
31 #include "runtime/globals.hpp"
32 
33 class SoftRefPolicy;
34 
35 class GCOverheadTester: public StackObj {
36 public:
37   virtual bool is_exceeded() = 0;
38 };
39 
40 class GCOverheadChecker: public CHeapObj<mtGC> {
41   // This is a hint for the heap:  we've detected that GC times
42   // are taking longer than GCTimeLimit allows.
43   bool _gc_overhead_limit_exceeded;
44   // Use for diagnostics only.  If UseGCOverheadLimit is false,
45   // this variable is still set.
46   bool _print_gc_overhead_limit_would_be_exceeded;
47   // Count of consecutive GC that have exceeded the
48   // GC time limit criterion
49   uint _gc_overhead_limit_count;
50   // This flag signals that GCTimeLimit is being exceeded
51   // but may not have done so for the required number of consecutive
52   // collections
53 
54 public:
55   GCOverheadChecker();
56 
57   // This is a hint for the heap:  we've detected that gc times
58   // are taking longer than GCTimeLimit allows.
59   // Most heaps will choose to throw an OutOfMemoryError when
60   // this occurs but it is up to the heap to request this information
61   // of the policy
62   bool gc_overhead_limit_exceeded() {
63     return _gc_overhead_limit_exceeded;
64   }
65   void set_gc_overhead_limit_exceeded(bool v) {
66     _gc_overhead_limit_exceeded = v;
67   }
68 
69   // Tests conditions indicate the GC overhead limit is being approached.
70   bool gc_overhead_limit_near() {
71     return _gc_overhead_limit_count >= (GCOverheadLimitThreshold - 1);
72   }
73   void reset_gc_overhead_limit_count() {
74     _gc_overhead_limit_count = 0;
75   }
76 
77   // Check the conditions for an out-of-memory due to excessive GC time.
78   // Set _gc_overhead_limit_exceeded if all the conditions have been met.
79   void check_gc_overhead_limit(GCOverheadTester* time_overhead,
80                                GCOverheadTester* space_overhead,
81                                bool is_full_gc,
82                                GCCause::Cause gc_cause,
83                                SoftRefPolicy* soft_ref_policy);
84 };
85 
86 #endif // SHARE_GC_SHARED_GCOVERHEADCHECKER_HPP
