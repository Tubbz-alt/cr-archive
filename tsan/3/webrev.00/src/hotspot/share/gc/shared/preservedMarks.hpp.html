 25 #ifndef SHARE_GC_SHARED_PRESERVEDMARKS_HPP
 26 #define SHARE_GC_SHARED_PRESERVEDMARKS_HPP
 27 
 28 #include "memory/allocation.hpp"
 29 #include "memory/padded.hpp"
 30 #include "oops/oop.hpp"
 31 #include "utilities/stack.hpp"
 32 
 33 class PreservedMarksSet;
 34 class WorkGang;
 35 
 36 class PreservedMarks {
 37 private:
 38   class OopAndMarkWord {
 39   private:
 40     oop _o;
 41     markWord _m;
 42 
 43   public:
 44     OopAndMarkWord(oop obj, markWord m) : _o(obj), _m(m) { }
 45 
 46     oop get_oop() { return _o; }
 47     inline void set_mark() const;
 48     void set_oop(oop obj) { _o = obj; }
 49   };
 50   typedef Stack<OopAndMarkWord, mtGC> OopAndMarkWordStack;
 51 
 52   OopAndMarkWordStack _stack;
 53 
 54   inline bool should_preserve_mark(oop obj, markWord m) const;
 55 
 56 public:
 57   size_t size() const { return _stack.size(); }
 58   inline void push(oop obj, markWord m);
 59   inline void push_if_necessary(oop obj, markWord m);
 60   // Iterate over the stack, restore all preserved marks, and
 61   // reclaim the memory taken up by the stack segments.
 62   void restore();
 63   // Iterate over the stack, adjust all preserved marks according
 64   // to their forwarding location stored in the mark.
 65   void adjust_during_full_gc();
 66 
 67   void restore_and_increment(volatile size_t* const _total_size_addr);
 68   inline static void init_forwarded_mark(oop obj);
 69 
 70   // Assert the stack is empty and has no cached segments.
 71   void assert_empty() PRODUCT_RETURN;
 72 
 73   inline PreservedMarks();
 74   ~PreservedMarks() { assert_empty(); }
 75 };
 76 
 77 class RemoveForwardedPointerClosure: public ObjectClosure {
 78 public:
 79   virtual void do_object(oop obj);
 80 };
 81 
 82 class PreservedMarksSet : public CHeapObj<mtGC> {
 83 private:
 84   // true -> _stacks will be allocated in the C heap
 85   // false -> _stacks will be allocated in the resource arena
 86   const bool _in_c_heap;
 87 
 88   // Number of stacks we have allocated (typically, one stack per GC worker).
 89   // This should be >= 1 if the stacks have been initialized,
 90   // or == 0 if they have not.
 91   uint _num;
 92 
 93   // Stack array (typically, one stack per GC worker) of length _num.
 94   // This should be != NULL if the stacks have been initialized,
 95   // or == NULL if they have not.
 96   Padded<PreservedMarks>* _stacks;
 97 
 98 public:
 99   uint num() const { return _num; }
100 
101   // Return the i'th stack.
102   PreservedMarks* get(uint i = 0) const {
103     assert(_num > 0 && _stacks != NULL, "stacks should have been initialized");
104     assert(i < _num, "pre-condition");
105     return (_stacks + i);
106   }
107 
108   // Allocate stack array.
109   void init(uint num);
110 
111   // Iterate over all stacks, restore all preserved marks, and reclaim
112   // the memory taken up by the stack segments using the given WorkGang. If the WorkGang
113   // is NULL, perform the work serially in the current thread.
114   void restore(WorkGang* workers);
115 
116   // Reclaim stack array.
117   void reclaim();
118 
119   // Assert all the stacks are empty and have no cached segments.
120   void assert_empty() PRODUCT_RETURN;
121 
122   PreservedMarksSet(bool in_c_heap)
123       : _in_c_heap(in_c_heap), _num(0), _stacks(NULL) { }
124 
125   ~PreservedMarksSet() {
126     assert(_stacks == NULL && _num == 0, "stacks should have been reclaimed");
127   }
128 };
129 
130 #endif // SHARE_GC_SHARED_PRESERVEDMARKS_HPP
