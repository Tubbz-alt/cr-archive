35 size_t EpsilonArguments::conservative_max_heap_alignment() {
36   return UseLargePages ? os::large_page_size() : os::vm_page_size();
37 }
38 
39 void EpsilonArguments::initialize() {
40   GCArguments::initialize();
41 
42   assert(UseEpsilonGC, "Sanity");
43 
44   // Forcefully exit when OOME is detected. Nothing we can do at that point.
45   if (FLAG_IS_DEFAULT(ExitOnOutOfMemoryError)) {
46     FLAG_SET_DEFAULT(ExitOnOutOfMemoryError, true);
47   }
48 












49   if (EpsilonMaxTLABSize < MinTLABSize) {
50     warning("EpsilonMaxTLABSize < MinTLABSize, adjusting it to " SIZE_FORMAT, MinTLABSize);
51     EpsilonMaxTLABSize = MinTLABSize;
52   }
53 
54   if (!EpsilonElasticTLAB && EpsilonElasticTLABDecay) {
55     warning("Disabling EpsilonElasticTLABDecay because EpsilonElasticTLAB is disabled");
56     FLAG_SET_DEFAULT(EpsilonElasticTLABDecay, false);
57   }
58 
59 #ifdef COMPILER2
60   // Enable loop strip mining: there are still non-GC safepoints, no need to make it worse
61   if (FLAG_IS_DEFAULT(UseCountedLoopSafepoints)) {
62     FLAG_SET_DEFAULT(UseCountedLoopSafepoints, true);
63     if (FLAG_IS_DEFAULT(LoopStripMiningIter)) {
64       FLAG_SET_DEFAULT(LoopStripMiningIter, 1000);
65     }
66   }
67 #endif
68 }
69 







70 CollectedHeap* EpsilonArguments::create_heap() {
71   return create_heap_with_policy<EpsilonHeap, EpsilonCollectorPolicy>();
72 }
