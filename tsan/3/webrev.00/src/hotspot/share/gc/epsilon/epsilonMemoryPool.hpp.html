33 class EpsilonMemoryPool : public CollectedMemoryPool {
34 private:
35   EpsilonHeap* _heap;
36 
37 public:
38   EpsilonMemoryPool(EpsilonHeap* heap);

42   MemoryUsage get_memory_usage();
43 };
