24 #ifndef SHARE_GC_EPSILON_EPSILONTHREADLOCALDATA_HPP
25 #define SHARE_GC_EPSILON_EPSILONTHREADLOCALDATA_HPP
26 
27 #include "runtime/thread.hpp"
28 #include "utilities/debug.hpp"
29 
30 class EpsilonThreadLocalData {
31 private:
32   size_t _ergo_tlab_size;
33   int64_t _last_tlab_time;
34 
35   EpsilonThreadLocalData() :
36           _ergo_tlab_size(0),
37           _last_tlab_time(0) {}
38 
39   static EpsilonThreadLocalData* data(Thread* thread) {
40     assert(UseEpsilonGC, "Sanity");
41     return thread->gc_data<EpsilonThreadLocalData>();
42   }
43 
44 public:
45   static void create(Thread* thread) {
46     new (data(thread)) EpsilonThreadLocalData();
47   }
48 
49   static void destroy(Thread* thread) {
50     data(thread)->~EpsilonThreadLocalData();
51   }
52 
53   static size_t ergo_tlab_size(Thread *thread) {
54     return data(thread)->_ergo_tlab_size;
55   }
56 
57   static int64_t last_tlab_time(Thread *thread) {
58     return data(thread)->_last_tlab_time;
59   }
60 
61   static void set_ergo_tlab_size(Thread *thread, size_t val) {
62     data(thread)->_ergo_tlab_size = val;
63   }
64 
65   static void set_last_tlab_time(Thread *thread, int64_t time) {
66     data(thread)->_last_tlab_time = time;
67   }
68 };
69 
70 #endif // SHARE_GC_EPSILON_EPSILONTHREADLOCALDATA_HPP
