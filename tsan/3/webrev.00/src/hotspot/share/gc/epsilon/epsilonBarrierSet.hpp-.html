24 #ifndef SHARE_GC_EPSILON_EPSILONBARRIERSET_HPP
25 #define SHARE_GC_EPSILON_EPSILONBARRIERSET_HPP
26 
27 #include "gc/shared/barrierSetAssembler.hpp"
28 #include "gc/shared/barrierSet.hpp"
29 
30 // No interaction with application is required for Epsilon, and therefore
31 // the barrier set is empty.
32 class EpsilonBarrierSet: public BarrierSet {
33   friend class VMStructs;
34 
35 public:
36   EpsilonBarrierSet();
37 
38   virtual void print_on(outputStream *st) const {}
39 
40   virtual void on_thread_create(Thread* thread);
41   virtual void on_thread_destroy(Thread* thread);
42 
43   template <DecoratorSet decorators, typename BarrierSetT = EpsilonBarrierSet>
44   class AccessBarrier: public BarrierSet::AccessBarrier<decorators, BarrierSetT> {};
45 };
46 
47 template<>
48 struct BarrierSet::GetName<EpsilonBarrierSet> {
49   static const BarrierSet::Name value = BarrierSet::EpsilonBarrierSet;
50 };
51 
52 template<>
53 struct BarrierSet::GetType<BarrierSet::EpsilonBarrierSet> {
54   typedef ::EpsilonBarrierSet type;
55 };
56 
57 #endif // SHARE_GC_EPSILON_EPSILONBARRIERSET_HPP
