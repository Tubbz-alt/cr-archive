32 VectorSet::VectorSet(Arena *arena) : _size(2),
33     _data(NEW_ARENA_ARRAY(arena, uint32_t, 2)),
34     _data_size(2),
35     _set_arena(arena) {
36   _data[0] = 0;
37   _data[1] = 0;












































38 }
39 

40 // Expand the existing set to a bigger size
41 void VectorSet::grow(uint new_word_capacity) {
42   assert(new_word_capacity < (1U << 30), "");
43   uint x = next_power_of_2(new_word_capacity);
44   if (x > _data_size) {
45     _data = REALLOC_ARENA_ARRAY(_set_arena, uint32_t, _data, _size, x);
46     _data_size = x;































































47   }
48   Copy::zero_to_bytes(_data + _size, (x - _size) * sizeof(uint32_t));
49   _size = x;



















50 }
51 
52 // Insert a member into an existing Set.
53 void VectorSet::insert(uint elem) {
54   uint32_t word = elem >> word_bits;
55   uint32_t mask = 1U << (elem & bit_mask);
56   if (word >= _size) {
57     grow(word);



















































































































58   }
59   _data[word] |= mask;

















60 }
61 
62 // Return true if the set is empty
63 bool VectorSet::is_empty() const {
64   for (uint32_t i = 0; i < _size; i++) {
65     if (_data[i] != 0) {
66       return false;



































67     }
68   }
69   return true;



70 }
