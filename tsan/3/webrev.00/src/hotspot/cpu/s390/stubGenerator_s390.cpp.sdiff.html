 672   // Wrapper which calls oopDesc::is_oop_or_null()
 673   // Only called by MacroAssembler::verify_oop
 674   static void verify_oop_helper(const char* message, oopDesc* o) {
 675     if (!oopDesc::is_oop_or_null(o)) {
