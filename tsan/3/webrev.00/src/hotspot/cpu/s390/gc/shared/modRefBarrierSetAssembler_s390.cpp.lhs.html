26 #include "precompiled.hpp"
27 #include "asm/macroAssembler.inline.hpp"
28 #include "gc/shared/modRefBarrierSetAssembler.hpp"
29 
30 #define __ masm->
31 
32 void ModRefBarrierSetAssembler::gen_write_ref_array_post_barrier(MacroAssembler* masm, DecoratorSet decorators, Register addr, Register count,
33                                                                  bool do_return) {
34   if (do_return) { __ z_br(Z_R14); }
35 }
36 
37 void ModRefBarrierSetAssembler::arraycopy_prologue(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
38                                                    Register src, Register dst, Register count) {
39   if (type == T_OBJECT || type == T_ARRAY) {
40     gen_write_ref_array_pre_barrier(masm, decorators, dst, count);
41   }
42 }
43 
44 void ModRefBarrierSetAssembler::arraycopy_epilogue(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
45                                                    Register dst, Register count, bool do_return) {
46   if (type == T_OBJECT || type == T_ARRAY) {
47     gen_write_ref_array_post_barrier(masm, decorators, dst, count, do_return);
48   } else {
49     if (do_return) { __ z_br(Z_R14); }
50   }
51 }
52 
53 void ModRefBarrierSetAssembler::store_at(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
54                                          const Address& dst, Register val, Register tmp1, Register tmp2, Register tmp3) {
55   if (type == T_OBJECT || type == T_ARRAY) {
56     oop_store_at(masm, decorators, type, dst, val, tmp1, tmp2, tmp3);
57   } else {
58     BarrierSetAssembler::store_at(masm, decorators, type, dst, val, tmp1, tmp2, tmp3);
59   }
60 }
