26 #ifndef CPU_ZERO_ASSEMBLER_ZERO_HPP
27 #define CPU_ZERO_ASSEMBLER_ZERO_HPP
28 
29 // In normal, CPU-specific ports of HotSpot these two classes are used
30 // for generating assembly language.  We don't do any of this in zero,
31 // of course, but we do sneak entry points around in CodeBuffers so we
32 // generate those here.
33 
34 class Assembler : public AbstractAssembler {
35  public:
36   Assembler(CodeBuffer* code) : AbstractAssembler(code) {}
37 
38  public:
39   void pd_patch_instruction(address branch, address target, const char* file, int line);
40 
41   //---<  calculate length of instruction  >---
42   static unsigned int instr_len(unsigned char *instr) { return 1; }
43 
44   //---<  longest instructions  >---
45   static unsigned int instr_maxlen() { return 1; }
46 };
47 
48 class MacroAssembler : public Assembler {
49  public:
50   MacroAssembler(CodeBuffer* code) : Assembler(code) {}
51 
52  public:
53   void align(int modulus);
54   void bang_stack_with_offset(int offset);
55   bool needs_explicit_null_check(intptr_t offset);
56   bool uses_implicit_null_check(void* address);
57   RegisterOrConstant delayed_value_impl(intptr_t* delayed_value_addr,
58                                         Register tmp, int offset);
59  public:
60   void advance(int bytes);
61   void store_oop(jobject obj);
62   void store_Metadata(Metadata* obj);
63 };

66 address ShouldNotCallThisEntry();

68 #endif // CPU_ZERO_ASSEMBLER_ZERO_HPP
