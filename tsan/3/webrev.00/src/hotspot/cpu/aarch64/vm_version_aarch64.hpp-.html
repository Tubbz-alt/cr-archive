 26 #ifndef CPU_AARCH64_VM_VERSION_AARCH64_HPP
 27 #define CPU_AARCH64_VM_VERSION_AARCH64_HPP
 28 
 29 #include "runtime/globals_extension.hpp"
 30 #include "runtime/vm_version.hpp"
 31 #include "utilities/sizes.hpp"
 32 
 33 class VM_Version : public Abstract_VM_Version {
 34   friend class JVMCIVMStructs;
 35 
 36 protected:
 37   static int _cpu;
 38   static int _model;
 39   static int _model2;
 40   static int _variant;
 41   static int _revision;
 42   static int _stepping;
 43 
 44   struct PsrInfo {
 45     uint32_t dczid_el0;
 46     uint32_t ctr_el0;
 47   };
 48   static PsrInfo _psr_info;
 49   static void get_processor_features();
 50 
 51 public:
 52   // Initialization
 53   static void initialize();
 54 
 55   // Asserts
 56   static void assert_is_initialized() {
 57   }
 58 
 59   static bool expensive_load(int ld_size, int scale) {
 60     if (cpu_family() == CPU_ARM) {
 61       // Half-word load with index shift by 1 (aka scale is 2) has
 62       // extra cycle latency, e.g. ldrsh w0, [x1,w2,sxtw #1].
 63       if (ld_size == 2 && scale == 2) {
 64         return true;
 65       }
 66     }
 67     return false;
 68   }
 69 
 70   enum Family {
 71     CPU_ARM       = 'A',
 72     CPU_BROADCOM  = 'B',
 73     CPU_CAVIUM    = 'C',
 74     CPU_DEC       = 'D',
 75     CPU_HISILICON = 'H',
 76     CPU_INFINEON  = 'I',
 77     CPU_MOTOROLA  = 'M',
 78     CPU_NVIDIA    = 'N',
 79     CPU_AMCC      = 'P',
 80     CPU_QUALCOM   = 'Q',
 81     CPU_MARVELL   = 'V',
 82     CPU_INTEL     = 'i',
 83   };
 84 
 85   enum Feature_Flag {
 86     CPU_FP           = (1<<0),
 87     CPU_ASIMD        = (1<<1),
 88     CPU_EVTSTRM      = (1<<2),
 89     CPU_AES          = (1<<3),
 90     CPU_PMULL        = (1<<4),
 91     CPU_SHA1         = (1<<5),
 92     CPU_SHA2         = (1<<6),
 93     CPU_CRC32        = (1<<7),
 94     CPU_LSE          = (1<<8),
 95     CPU_STXR_PREFETCH= (1 << 29),
 96     CPU_A53MAC       = (1 << 30),
 97     CPU_DMB_ATOMICS  = (1 << 31),
 98   };
 99 
100   static int cpu_family()                     { return _cpu; }
101   static int cpu_model()                      { return _model; }
102   static int cpu_model2()                     { return _model2; }
103   static int cpu_variant()                    { return _variant; }
104   static int cpu_revision()                   { return _revision; }
105   static ByteSize dczid_el0_offset() { return byte_offset_of(PsrInfo, dczid_el0); }
106   static ByteSize ctr_el0_offset()   { return byte_offset_of(PsrInfo, ctr_el0); }
107   static bool is_zva_enabled() {
108     // Check the DZP bit (bit 4) of dczid_el0 is zero
109     // and block size (bit 0~3) is not zero.
110     return ((_psr_info.dczid_el0 & 0x10) == 0 &&
111             (_psr_info.dczid_el0 & 0xf) != 0);
112   }
113   static int zva_length() {
114     assert(is_zva_enabled(), "ZVA not available");
115     return 4 << (_psr_info.dczid_el0 & 0xf);
116   }
117   static int icache_line_size() {
118     return (1 << (_psr_info.ctr_el0 & 0x0f)) * 4;
119   }
120   static int dcache_line_size() {
121     return (1 << ((_psr_info.ctr_el0 >> 16) & 0x0f)) * 4;
122   }
123 };
124 
125 #endif // CPU_AARCH64_VM_VERSION_AARCH64_HPP
