 35 #define __ masm->
 36 
 37 #define BUFFER_SIZE 30*wordSize
 38 
 39 // Instead of issuing a LoadLoad barrier we create an address
 40 // dependency between loads; this might be more efficient.
 41 
 42 // Common register usage:
 43 // r0/v0:      result
 44 // c_rarg0:    jni env
 45 // c_rarg1:    obj
 46 // c_rarg2:    jfield id
 47 
 48 static const Register robj          = r3;
 49 static const Register rcounter      = r4;
 50 static const Register roffset       = r5;
 51 static const Register rcounter_addr = r6;
 52 static const Register result        = r7;
 53 
 54 address JNI_FastGetField::generate_fast_get_int_field0(BasicType type) {
 55   const char *name;
 56   switch (type) {
 57     case T_BOOLEAN: name = "jni_fast_GetBooleanField"; break;
 58     case T_BYTE:    name = "jni_fast_GetByteField";    break;
 59     case T_CHAR:    name = "jni_fast_GetCharField";    break;
 60     case T_SHORT:   name = "jni_fast_GetShortField";   break;
 61     case T_INT:     name = "jni_fast_GetIntField";     break;
 62     case T_LONG:    name = "jni_fast_GetLongField";    break;
 63     case T_FLOAT:   name = "jni_fast_GetFloatField";   break;
 64     case T_DOUBLE:  name = "jni_fast_GetDoubleField";  break;
 65     default:        ShouldNotReachHere();
 66       name = NULL;  // unreachable
 67   }
 68   ResourceMark rm;
 69   BufferBlob* blob = BufferBlob::create(name, BUFFER_SIZE);
 70   CodeBuffer cbuf(blob);
 71   MacroAssembler* masm = new MacroAssembler(&cbuf);
 72   address fast_entry = __ pc();
 73 
 74   Label slow;
 75 
 76   unsigned long offset;
 77   __ adrp(rcounter_addr,
 78           SafepointSynchronize::safepoint_counter_addr(), offset);
 79   Address safepoint_counter_addr(rcounter_addr, offset);
 80   __ ldrw(rcounter, safepoint_counter_addr);
 81   __ tbnz(rcounter, 0, slow);
 82   __ eor(robj, c_rarg1, rcounter);
 83   __ eor(robj, robj, rcounter);               // obj, since
 84                                               // robj ^ rcounter ^ rcounter == robj
 85                                               // robj is address dependent on rcounter.
 86 
 87   BarrierSetAssembler* bs = BarrierSet::barrier_set()->barrier_set_assembler();
 88   bs->try_resolve_jobject_in_native(masm, c_rarg0, robj, rscratch1, slow);
 89 
 90   __ lsr(roffset, c_rarg2, 2);                // offset
 91 
 92   assert(count < LIST_CAPACITY, "LIST_CAPACITY too small");
 93   speculative_load_pclist[count] = __ pc();   // Used by the segfault handler
 94   switch (type) {
 95     case T_BOOLEAN: __ ldrb    (result, Address(robj, roffset)); break;
 96     case T_BYTE:    __ ldrsb   (result, Address(robj, roffset)); break;
 97     case T_CHAR:    __ ldrh    (result, Address(robj, roffset)); break;
 98     case T_SHORT:   __ ldrsh   (result, Address(robj, roffset)); break;
 99     case T_FLOAT:   __ ldrw    (result, Address(robj, roffset)); break;
100     case T_INT:     __ ldrsw   (result, Address(robj, roffset)); break;
101     case T_DOUBLE:
102     case T_LONG:    __ ldr     (result, Address(robj, roffset)); break;
103     default:        ShouldNotReachHere();
104   }
105 
106   // counter_addr is address dependent on result.
107   __ eor(rcounter_addr, rcounter_addr, result);
108   __ eor(rcounter_addr, rcounter_addr, result);
109   __ ldrw(rscratch1, safepoint_counter_addr);
110   __ cmpw(rcounter, rscratch1);
111   __ br (Assembler::NE, slow);
112 
113   switch (type) {
114     case T_FLOAT:   __ fmovs(v0, result); break;
115     case T_DOUBLE:  __ fmovd(v0, result); break;
116     default:        __ mov(r0, result);   break;
117   }
118   __ ret(lr);
119 
120   slowcase_entry_pclist[count++] = __ pc();
121   __ bind(slow);
122   address slow_case_addr;
123   switch (type) {
124     case T_BOOLEAN: slow_case_addr = jni_GetBooleanField_addr(); break;
125     case T_BYTE:    slow_case_addr = jni_GetByteField_addr();    break;
126     case T_CHAR:    slow_case_addr = jni_GetCharField_addr();    break;
127     case T_SHORT:   slow_case_addr = jni_GetShortField_addr();   break;
128     case T_INT:     slow_case_addr = jni_GetIntField_addr();     break;
129     case T_LONG:    slow_case_addr = jni_GetLongField_addr();    break;
130     case T_FLOAT:   slow_case_addr = jni_GetFloatField_addr();   break;
131     case T_DOUBLE:  slow_case_addr = jni_GetDoubleField_addr();  break;
132     default:        ShouldNotReachHere();
133       slow_case_addr = NULL;  // unreachable
134   }
135 
136   {
137     __ enter();
138     __ lea(rscratch1, ExternalAddress(slow_case_addr));
139     __ blr(rscratch1);
140     __ maybe_isb();
141     __ leave();
142     __ ret(lr);
143   }
144   __ flush ();
145 
146   return fast_entry;
147 }
