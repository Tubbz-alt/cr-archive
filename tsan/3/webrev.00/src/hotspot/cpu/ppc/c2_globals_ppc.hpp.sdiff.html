 65 define_pd_global(bool, SuperWordLoopUnrollAnalysis,  true);
 66 // GL:
 67 // Detected a problem with unscaled compressed oops and
 68 // narrow_oop_use_complex_address() == false.
 69 // -Djava.io.tmpdir=./tmp -jar SPECjvm2008.jar -ikv -wt 3 -it 3
 70 //   -bt 1 --base compiler.sunflow
 71 // fails in Lower.visitIf->translate->tranlate->translate and
 72 // throws an unexpected NPE. A load and a store seem to be
 73 // reordered.  Java reads about:
 74 //   loc = x.f
 75 //   x.f = 0
 76 //   NullCheck loc
 77 // While assembler reads:
 78 //   x.f = 0
 79 //   loc = x.f
 80 //   NullCheck loc
 81 define_pd_global(bool,     OptoScheduling,               false);
 82 define_pd_global(bool,     IdealizeClearArrayNode,       true);
 83

 85 define_pd_global(uintx,    ReservedCodeCacheSize,        256*M);
 86 define_pd_global(uintx,    NonProfiledCodeHeapSize,      125*M);
 87 define_pd_global(uintx,    ProfiledCodeHeapSize,         126*M);
 88 define_pd_global(uintx,    NonNMethodCodeHeapSize,       5*M  );
 89 define_pd_global(uintx,    CodeCacheExpansionSize,       64*K);
 90 
 91 // Ergonomics related flags
 92 define_pd_global(uint64_t, MaxRAM,                       128ULL*G);
 93 define_pd_global(uintx,    CodeCacheMinBlockLength,      4);
 94 define_pd_global(uintx,    CodeCacheMinimumUseSpace,     400*K);
 95 
 96 define_pd_global(bool,     TrapBasedRangeChecks,          true);
 97 
 98 // Heap related flags
 99 define_pd_global(size_t,   MetaspaceSize,                ScaleForWordSize(16*M));
