30 #define __ masm->
31 
32 void ModRefBarrierSetAssembler::arraycopy_prologue(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
33                                                    Register src, Register dst, Register count, Register preserve1, Register preserve2) {
34   if (type == T_OBJECT) {
35     gen_write_ref_array_pre_barrier(masm, decorators, src, dst, count, preserve1, preserve2);
36 
37     bool checkcast = (decorators & ARRAYCOPY_CHECKCAST) != 0;
38     if (!checkcast) {
39       assert_different_registers(dst, count, R9_ARG7, R10_ARG8);
40       // Save some arguments for epilogue, e.g. disjoint_long_copy_core destroys them.
41       __ mr(R9_ARG7, dst);
42       __ mr(R10_ARG8, count);
43     }
44   }
45 }
46 
47 void ModRefBarrierSetAssembler::arraycopy_epilogue(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
48                                                    Register dst, Register count, Register preserve) {
49   if (type == T_OBJECT) {
50     bool checkcast = (decorators & ARRAYCOPY_CHECKCAST) != 0;
51     if (!checkcast) {
52       gen_write_ref_array_post_barrier(masm, decorators, R9_ARG7, R10_ARG8, preserve);
53     } else {
54       gen_write_ref_array_post_barrier(masm, decorators, dst, count, preserve);
55     }
56   }
57 }
58 
59 void ModRefBarrierSetAssembler::store_at(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
60                                          Register base, RegisterOrConstant ind_or_offs, Register val,
61                                          Register tmp1, Register tmp2, Register tmp3, bool needs_frame) {
62   if (type == T_OBJECT || type == T_ARRAY) {
63     oop_store_at(masm, decorators, type, base, ind_or_offs, val, tmp1, tmp2, tmp3, needs_frame);
64   } else {
65     BarrierSetAssembler::store_at(masm, decorators, type, base, ind_or_offs, val, tmp1, tmp2, tmp3, needs_frame);
66   }
67 }
