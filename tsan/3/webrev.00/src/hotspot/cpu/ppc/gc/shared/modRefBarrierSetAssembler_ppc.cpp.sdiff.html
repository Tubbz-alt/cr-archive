47 void ModRefBarrierSetAssembler::arraycopy_epilogue(MacroAssembler* masm, DecoratorSet decorators, BasicType type,
48                                                    Register dst, Register count, Register preserve) {
49   if (type == T_OBJECT) {
50     bool checkcast = (decorators & ARRAYCOPY_CHECKCAST) != 0;
51     if (!checkcast) {
52       gen_write_ref_array_post_barrier(masm, decorators, R9_ARG7, R10_ARG8, preserve);
53     } else {
54       gen_write_ref_array_post_barrier(masm, decorators, dst, count, preserve);
55     }
56   }
57 }
