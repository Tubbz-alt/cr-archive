 25 #ifndef CPU_ARM_VM_VERSION_ARM_HPP
 26 #define CPU_ARM_VM_VERSION_ARM_HPP
 27 

 28 #include "runtime/globals_extension.hpp"
 29 #include "runtime/vm_version.hpp"
 30 
 31 class VM_Version: public Abstract_VM_Version {
 32   friend class JVMCIVMStructs;
 33 
 34   static bool _has_simd;

 35 
 36  protected:
 37   // Are we done with vm version initialization
 38   static bool _is_initialized;
 39 
 40  public:
 41   static void initialize();
 42   static bool is_initialized()      { return _is_initialized; }
 43 
 44 
 45  protected:
 46   enum Feature_Flag {
 47     vfp = 0,
 48     vfp3_32 = 1,
 49     simd = 2,

 50   };
 51 
 52   enum Feature_Flag_Set {
 53     unknown_m           = 0,
 54     all_features_m      = -1,
 55 
 56     vfp_m     = 1 << vfp,
 57     vfp3_32_m = 1 << vfp3_32,
 58     simd_m    = 1 << simd,

 59   };
 60 
 61   // The value stored by "STR PC, [addr]" instruction can be either
 62   // (address of this instruction + 8) or (address of this instruction + 12)
 63   // depending on hardware implementation.
 64   // This adjustment is calculated in runtime.
 65   static int _stored_pc_adjustment;
 66 
 67   // ARM architecture version: 5 = ARMv5, 6 = ARMv6, 7 = ARMv7 etc.
 68   static int _arm_arch;
 69 
 70   // linux kernel atomic helper function version info
 71   // __kuser_cmpxchg() if version >= 2
 72   // __kuser_cmpxchg64() if version >= 5
 73   static int _kuser_helper_version;
 74 
 75 #define KUSER_HELPER_VERSION_ADDR 0xffff0ffc
 76 #define KUSER_VERSION_CMPXCHG32 2
 77 #define KUSER_VERSION_CMPXCHG64 5
 78 

 80   static void get_os_cpu_info();
 81 
 82  public:
 83   static void early_initialize();
 84 
 85   static int arm_arch()             { return _arm_arch; }
 86   static int stored_pc_adjustment() { return _stored_pc_adjustment; }
 87   static bool supports_rev()        { return _arm_arch >= 6; }
 88   static bool supports_ldrex()      { return _arm_arch >= 6; }
 89   static bool supports_movw()       { return _arm_arch >= 7; }
 90   static bool supports_ldrexd()     { return _arm_arch >= 7; }
 91   static bool supports_compare_and_exchange() { return true; }
 92   static bool supports_kuser_cmpxchg32() { return _kuser_helper_version >= KUSER_VERSION_CMPXCHG32; }
 93   static bool supports_kuser_cmpxchg64() { return _kuser_helper_version >= KUSER_VERSION_CMPXCHG64; }
 94   // Override Abstract_VM_Version implementation
 95   static bool use_biased_locking();
 96 
 97   static bool has_vfp()             { return (_features & vfp_m) != 0; }
 98   static bool has_vfp3_32()         { return (_features & vfp3_32_m) != 0; }
 99   static bool has_simd()            { return (_features & simd_m) != 0; }

100 
101   static bool simd_math_is_compliant() { return false; }
102 
103   static bool prefer_moves_over_load_literal() { return supports_movw(); }
104 
105   friend class VM_Version_StubGenerator;
106 
107 };
