 52 inline void OrderAccess::fence() {
 53   // always use locked addl since mfence is sometimes expensive
 54 #ifdef AMD64
 55   __asm__ volatile ("lock; addl $0,0(%%rsp)" : : : "cc", "memory");
 56 #else
 57   __asm__ volatile ("lock; addl $0,0(%%esp)" : : : "cc", "memory");
 58 #endif
 59   compiler_barrier();
 60 }
