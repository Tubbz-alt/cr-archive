33 // Implementation of class OrderAccess.
34 
35 inline void OrderAccess::loadload()   { acquire(); }
36 inline void OrderAccess::storestore() { release(); }
37 inline void OrderAccess::loadstore()  { acquire(); }
38 inline void OrderAccess::storeload()  { fence(); }
39 
40 inline void OrderAccess::acquire() {
41   READ_MEM_BARRIER;
42 }
43 
44 inline void OrderAccess::release() {
45   WRITE_MEM_BARRIER;
46 }
47 
48 inline void OrderAccess::fence() {
49   FULL_MEM_BARRIER;
50 }
51 
52 template<size_t byte_size>
53 struct OrderAccess::PlatformOrderedLoad<byte_size, X_ACQUIRE>
54 {
55   template <typename T>
56   T operator()(const volatile T* p) const { T data; __atomic_load(p, &data, __ATOMIC_ACQUIRE); return data; }
57 };
58 
59 template<size_t byte_size>
60 struct OrderAccess::PlatformOrderedStore<byte_size, RELEASE_X>
61 {
62   template <typename T>
63   void operator()(T v, volatile T* p) const { __atomic_store(p, &v, __ATOMIC_RELEASE); }
64 };
65 
66 template<size_t byte_size>
67 struct OrderAccess::PlatformOrderedStore<byte_size, RELEASE_X_FENCE>
68 {
69   template <typename T>
70   void operator()(T v, volatile T* p) const { release_store(p, v); fence(); }
71 };
