29 # include <sys/utsname.h>

31 // Use uname() to find the architecture version
32 void VM_Version::get_os_cpu_info() {
33   struct utsname name;
34   static bool done = false;
35 
36   // Support for multiple calls in the init phase
37   if (done) return;
38   done = true;
39 
40   uname(&name);
41   if (strncmp(name.machine, "aarch64", 7) == 0) {
42     _arm_arch = 8;
43   } else if (strncmp(name.machine, "armv", 4) == 0 &&
44       name.machine[4] >= '5' && name.machine[4] <= '9') {
45     _arm_arch = (int)(name.machine[4] - '0');
46   }
47 }
