 74 // macros for restartable system calls
 75 
 76 #define RESTARTABLE(_cmd, _result) do { \
 77     _result = _cmd; \
 78   } while(((int)_result == OS_ERR) && (errno == EINTR))
 79 
 80 #define RESTARTABLE_RETURN_INT(_cmd) do { \
 81   int _result; \
 82   RESTARTABLE(_cmd, _result); \
 83   return _result; \
 84 } while(false)

 89 inline size_t os::write(int fd, const void *buf, unsigned int nBytes) {
 90   size_t res;
 91   RESTARTABLE((size_t) ::write(fd, buf, (size_t) nBytes), res);

 93 }
