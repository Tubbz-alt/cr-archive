31 #include <semaphore.h>

33 class PosixSemaphore : public CHeapObj<mtInternal> {
34   sem_t _semaphore;
35 
36   NONCOPYABLE(PosixSemaphore);
37 
38  public:
39   PosixSemaphore(uint value = 0);
40   ~PosixSemaphore();
41 
42   void signal(uint count = 1);
43 
44   void wait();
45 
46   bool trywait();

48   bool timedwait(struct timespec ts);
49   // wait until the given relative time elapses
50   bool timedwait(int64_t millis);
51 };

55 #endif // OS_POSIX_SEMAPHORE_POSIX_HPP
