 45 inline bool os::uses_stack_guard_pages() {
 46   return true;
 47 }
 48 
 49 inline bool os::must_commit_stack_guard_pages() {
 50   assert(uses_stack_guard_pages(), "sanity check");
 51   int r = thr_main() ;
 52   guarantee (r == 0 || r == 1, "CR6501650 or CR6493689") ;
 53   return r;
 54 }

 74 // macros for restartable system calls
 75 
 76 #define RESTARTABLE(_cmd, _result) do { \
 77   do { \
 78     _result = _cmd; \
 79   } while((_result == OS_ERR) && (errno == EINTR)); \
 80 } while(false)
 81 
 82 #define RESTARTABLE_RETURN_INT(_cmd) do { \
 83   int _result; \
 84   RESTARTABLE(_cmd, _result); \
 85   return _result; \
 86 } while(false)
