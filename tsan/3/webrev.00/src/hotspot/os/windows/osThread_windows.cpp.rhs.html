25 // no precompiled headers


26 #include "runtime/os.hpp"
27 #include "runtime/osThread.hpp"


28 
29 void OSThread::pd_initialize() {
30   set_thread_handle(NULL);
31   set_thread_id(NULL);
32   set_interrupt_event(NULL);
33 }
34 



35 void OSThread::pd_destroy() {
36   if (_interrupt_event != NULL) {
37     CloseHandle(_interrupt_event);
38   }
39 }
40 
41 // We need to specialize this to interact with the _interrupt_event.
42 
43 void OSThread::set_interrupted(bool z) {
44   if (z) {
45     SetEvent(_interrupt_event);
46   }
47   else {
48     // We should only ever clear the interrupt if we are in fact interrupted,
49     // and this can only be done by the current thread on itself.
50     ResetEvent(_interrupt_event);
51   }
52 }
