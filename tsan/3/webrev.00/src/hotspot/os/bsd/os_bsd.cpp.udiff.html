+       uint ebx;
+       uint ecx = i;
+       uint edx;
+ 
+       __asm__ ("cpuid\n\t" : "+a" (eax), "+b" (ebx), "+c" (ecx), "+d" (edx) : );
+ 
+       uint level_type = (ecx >> 8) & 0xFF;
+       if (level_type == 0) {
+         // Invalid level; end of topology
+         break;
+       }
+       uint level_apic_id_shift = eax & ((1u << 5) - 1);
+       total_bits += level_apic_id_shift;
+     }
