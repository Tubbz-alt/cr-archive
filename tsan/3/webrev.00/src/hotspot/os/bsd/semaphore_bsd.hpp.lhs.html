25 #ifndef OS_BSD_SEMAPHORE_BSD_HPP
26 #define OS_BSD_SEMAPHORE_BSD_HPP
27 


28 #ifndef __APPLE__
29 // Use POSIX semaphores.
30 # include "semaphore_posix.hpp"
31 
32 #else
33 // OS X doesn't support unamed POSIX semaphores, so the implementation in os_posix.cpp can't be used.
34 # include "memory/allocation.hpp"
35 # include <mach/semaphore.h>
36 
37 class OSXSemaphore : public CHeapObj<mtInternal>{
38   semaphore_t _semaphore;
39 
40   // Prevent copying and assignment.
41   OSXSemaphore(const OSXSemaphore&);
42   OSXSemaphore& operator=(const OSXSemaphore&);
43 
44  public:
45   OSXSemaphore(uint value = 0);
46   ~OSXSemaphore();
47 
48   void signal(uint count = 1);
49 
50   void wait();
51 
52   bool trywait();
53 
54   // wait until the given relative time elapses
55   bool timedwait(int64_t millis);
56 };
57 
58 typedef OSXSemaphore SemaphoreImpl;
59 
60 #endif // __APPLE__
61 
62 #endif // OS_BSD_SEMAPHORE_BSD_HPP
