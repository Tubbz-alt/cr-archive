30 #include <signal.h>

32 void OSThread::pd_initialize() {
33   assert(this != NULL, "check");
34 #ifdef __APPLE__
35   _thread_id        = 0;
36 #else
37   _thread_id        = NULL;
38 #endif
39   _unique_thread_id = 0;
40   _pthread_id       = NULL;
41   _siginfo          = NULL;
42   _ucontext         = NULL;
43   _expanding_stack  = 0;
44   _alt_sig_stack    = NULL;
45 
46   sigemptyset(&_caller_sigmask);
47 
48   _startThread_lock = new Monitor(Mutex::event, "startThread_lock", true,
49                                   Monitor::_safepoint_check_never);
50   assert(_startThread_lock !=NULL, "check");
51 }
52 
53 // Additional thread_id used to correlate threads in SA
54 void OSThread::set_unique_thread_id() {
55 #ifdef __APPLE__
56   thread_identifier_info_data_t m_ident_info;
57   mach_msg_type_number_t count = THREAD_IDENTIFIER_INFO_COUNT;
58 
59   mach_port_t mach_thread_port = mach_thread_self();
60   guarantee(mach_thread_port != 0, "just checking");
61   thread_info(mach_thread_port, THREAD_IDENTIFIER_INFO,
62               (thread_info_t) &m_ident_info, &count);
63   mach_port_deallocate(mach_task_self(), mach_thread_port);
64 
65   _unique_thread_id = m_ident_info.thread_id;
66 #endif
67 }
68 
69 void OSThread::pd_destroy() {
70   delete _startThread_lock;
71 }
