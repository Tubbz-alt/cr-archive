25 #ifndef OS_BSD_SEMAPHORE_BSD_HPP
26 #define OS_BSD_SEMAPHORE_BSD_HPP
27 
28 #include "utilities/globalDefinitions.hpp"
29 
30 #ifndef __APPLE__
31 // Use POSIX semaphores.
32 # include "semaphore_posix.hpp"
33 
34 #else
35 // OS X doesn't support unamed POSIX semaphores, so the implementation in os_posix.cpp can't be used.
36 # include "memory/allocation.hpp"
37 # include <mach/semaphore.h>
38 
39 class OSXSemaphore : public CHeapObj<mtInternal>{
40   semaphore_t _semaphore;
41 
42   NONCOPYABLE(OSXSemaphore);
43 
44  public:
45   OSXSemaphore(uint value = 0);
46   ~OSXSemaphore();
47 
48   void signal(uint count = 1);
49 
50   void wait();
51 
52   bool trywait();
53 
54   // wait until the given relative time elapses
55   bool timedwait(int64_t millis);
56 };
57 
58 typedef OSXSemaphore SemaphoreImpl;
59 
60 #endif // __APPLE__
61 
62 #endif // OS_BSD_SEMAPHORE_BSD_HPP
