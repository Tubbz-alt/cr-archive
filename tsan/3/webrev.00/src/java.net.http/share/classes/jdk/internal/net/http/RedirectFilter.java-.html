 26 package jdk.internal.net.http;
 27 
 28 import java.io.IOException;
 29 import java.io.UncheckedIOException;
 30 import java.net.URI;
 31 import java.net.http.HttpClient;
 32 import java.net.http.HttpHeaders;
 33 import jdk.internal.net.http.common.Log;
 34 import jdk.internal.net.http.common.Utils;
 35 
 36 class RedirectFilter implements HeaderFilter {
 37 
 38     HttpRequestImpl request;
 39     HttpClientImpl client;
 40     HttpClient.Redirect policy;
 41     String method;
 42     MultiExchange<?> exchange;
 43     static final int DEFAULT_MAX_REDIRECTS = 5;
 44     URI uri;
 45 
 46     /*
 47      * NOT_MODIFIED status code results from a conditional GET where
 48      * the server does not (must not) return a response body because
 49      * the condition specified in the request disallows it
 50      */
 51     static final int HTTP_NOT_MODIFIED = 304;
 52 
 53     static final int max_redirects = Utils.getIntegerNetProperty(
 54             "jdk.httpclient.redirects.retrylimit", DEFAULT_MAX_REDIRECTS
 55     );
 56 
 57     // A public no-arg constructor is required by FilterFactory
 58     public RedirectFilter() {}
 59 
 60     @Override
 61     public synchronized void request(HttpRequestImpl r, MultiExchange<?> e) throws IOException {
 62         this.request = r;
 63         this.client = e.client();
 64         this.policy = client.followRedirects();
 65 
 66         this.method = r.method();
 67         this.uri = r.uri();
 68         this.exchange = e;
 69     }
 70 
 71     @Override
 72     public synchronized HttpRequestImpl response(Response r) throws IOException {
 73         return handleResponse(r);
 74     }
 75 
 76     private static String redirectedMethod(int statusCode, String orig) {
 77         switch (statusCode) {
 78             case 301:
 79             case 302:
 80                 return orig.equals("POST") ? "GET" : orig;
 81             case 303:
 82                 return "GET";
 83             case 307:
 84             case 308:
 85                 return orig;
 86             default:
 87                 // unexpected but return orig
 88                 return orig;
 89         }
 90     }
 91 
 92     /**
 93      * Checks to see if a new request is needed and returns it.
 94      * Null means response is ok to return to user.
 95      */
 96     private HttpRequestImpl handleResponse(Response r) {
 97         int rcode = r.statusCode();
 98         if (rcode == 200 || policy == HttpClient.Redirect.NEVER) {
 99             return null;
100         }
101 
102         if (rcode == HTTP_NOT_MODIFIED)
103             return null;
104 
105         if (rcode >= 300 && rcode <= 399) {
106             URI redir = getRedirectedURI(r.headers());
107             String newMethod = redirectedMethod(rcode, method);
108             Log.logTrace("response code: {0}, redirected URI: {1}", rcode, redir);
109             if (canRedirect(redir) && ++exchange.numberOfRedirects < max_redirects) {
110                 Log.logTrace("redirect to: {0} with method: {1}", redir, newMethod);
111                 return HttpRequestImpl.newInstanceForRedirection(redir, newMethod, request);
112             } else {
113                 Log.logTrace("not redirecting");
114                 return null;
115             }
116         }
117         return null;
118     }
119 
120     private URI getRedirectedURI(HttpHeaders headers) {
121         URI redirectedURI;
122         redirectedURI = headers.firstValue("Location")
123                 .map(URI::create)
124                 .orElseThrow(() -> new UncheckedIOException(
125                         new IOException("Invalid redirection")));
126 
127         // redirect could be relative to original URL, but if not
128         // then redirect is used.
129         redirectedURI = uri.resolve(redirectedURI);
130         return redirectedURI;
131     }
132 
133     private boolean canRedirect(URI redir) {
134         String newScheme = redir.getScheme();
135         String oldScheme = uri.getScheme();
136         switch (policy) {
137             case ALWAYS:
138                 return true;
139             case NEVER:
140                 return false;
141             case NORMAL:
142                 return newScheme.equalsIgnoreCase(oldScheme)
143                         || newScheme.equalsIgnoreCase("https");
144             default:
145                 throw new InternalError();
146         }
147     }
148 }
