 26 package jdk.internal.net.http;
 27 
 28 import java.io.IOException;
 29 import java.io.UncheckedIOException;
 30 import java.net.URI;
 31 import java.net.http.HttpClient;
 32 import java.net.http.HttpHeaders;
 33 import jdk.internal.net.http.common.Log;
 34 import jdk.internal.net.http.common.Utils;
 35 
 36 class RedirectFilter implements HeaderFilter {
 37 
 38     HttpRequestImpl request;
 39     HttpClientImpl client;
 40     HttpClient.Redirect policy;
 41     String method;
 42     MultiExchange<?> exchange;
 43     static final int DEFAULT_MAX_REDIRECTS = 5;
 44     URI uri;
 45 
 46     /*
 47      * NOT_MODIFIED status code results from a conditional GET where
 48      * the server does not (must not) return a response body because
 49      * the condition specified in the request disallows it
 50      */
 51     static final int HTTP_NOT_MODIFIED = 304;
 52 
 53     static final int max_redirects = Utils.getIntegerNetProperty(
 54             "jdk.httpclient.redirects.retrylimit", DEFAULT_MAX_REDIRECTS
 55     );
 56 
 57     // A public no-arg constructor is required by FilterFactory
 58     public RedirectFilter() {}
 59 
 60     @Override
 61     public synchronized void request(HttpRequestImpl r, MultiExchange<?> e) throws IOException {
 62         this.request = r;
 63         this.client = e.client();
 64         this.policy = client.followRedirects();
 65 
 66         this.method = r.method();
 67         this.uri = r.uri();
 68         this.exchange = e;
 69     }
 70 
 71     @Override
 72     public synchronized HttpRequestImpl response(Response r) throws IOException {
 73         return handleResponse(r);
 74     }
 75 
 76     private static String redirectedMethod(int statusCode, String orig) {
 77         switch (statusCode) {
 78             case 301:
 79             case 302:
 80                 return orig.equals("POST") ? "GET" : orig;
 81             case 303:
 82                 return "GET";
 83             case 307:
 84             case 308:
 85                 return orig;
 86             default:
 87                 // unexpected but return orig
 88                 return orig;
 89         }
 90     }
 91 
 92     private static boolean isRedirecting(int statusCode) {
 93         // < 300: not a redirect codes
 94         if (statusCode < 300) return false;
 95         // 309-399 Unassigned => don't follow
 96         // > 399: not a redirect code
 97         if (statusCode > 308) return false;
 98         switch (statusCode) {
 99             // 300: MultipleChoice => don't follow
100             case 300:
101                 return false;
102             // 304: Not Modified => don't follow
103             case 304:
104                 return false;
105             // 305: Proxy Redirect => don't follow.
106             case 305:
107                 return false;
108             // 306: Unused => don't follow
109             case 306:
110                 return false;
111             // 301, 302, 303, 307, 308: OK to follow.
112             default:
113                 return true;
114         }
115     }
116 
117     /**
118      * Checks to see if a new request is needed and returns it.
119      * Null means response is ok to return to user.
120      */
121     private HttpRequestImpl handleResponse(Response r) {
122         int rcode = r.statusCode();
123         if (rcode == 200 || policy == HttpClient.Redirect.NEVER) {
124             return null;
125         }
126 
127         if (rcode == HTTP_NOT_MODIFIED)
128             return null;
129 
130         if (isRedirecting(rcode)) {
131             URI redir = getRedirectedURI(r.headers());
132             String newMethod = redirectedMethod(rcode, method);
133             Log.logTrace("response code: {0}, redirected URI: {1}", rcode, redir);
134             if (canRedirect(redir) && ++exchange.numberOfRedirects < max_redirects) {
135                 Log.logTrace("redirect to: {0} with method: {1}", redir, newMethod);
136                 return HttpRequestImpl.newInstanceForRedirection(redir, newMethod, request, rcode != 303);
137             } else {
138                 Log.logTrace("not redirecting");
139                 return null;
140             }
141         }
142         return null;
143     }
144 
145     private URI getRedirectedURI(HttpHeaders headers) {
146         URI redirectedURI;
147         redirectedURI = headers.firstValue("Location")
148                 .map(URI::create)
149                 .orElseThrow(() -> new UncheckedIOException(
150                         new IOException("Invalid redirection")));
151 
152         // redirect could be relative to original URL, but if not
153         // then redirect is used.
154         redirectedURI = uri.resolve(redirectedURI);
155         return redirectedURI;
156     }
157 
158     private boolean canRedirect(URI redir) {
159         String newScheme = redir.getScheme();
160         String oldScheme = uri.getScheme();
161         switch (policy) {
162             case ALWAYS:
163                 return true;
164             case NEVER:
165                 return false;
166             case NORMAL:
167                 return newScheme.equalsIgnoreCase(oldScheme)
168                         || newScheme.equalsIgnoreCase("https");
169             default:
170                 throw new InternalError();
171         }
172     }
173 }
