26 package sun.tools.jconsole;
27 
28 import java.util.*;
29 
30 public class Worker extends Thread {
31     ArrayList<Runnable> jobs = new ArrayList<Runnable>();
32     private volatile boolean stopped = false;
33 
34     public Worker(String name) {
35         super("Worker-"+name);
36 
37         setPriority(NORM_PRIORITY - 1);
38     }
39 
40     public void run() {
41         while (!stopped) {
42             Runnable job;
43             synchronized(jobs) {
44                 while (!stopped && jobs.size() == 0) {
45                     try {
46                         jobs.wait();
47                     } catch (InterruptedException ex) {
48                     }
49                 }
50 
51                 if(stopped) break;
52 
53                 job = jobs.remove(0);
54             }
55             job.run();
56         }
57     }
58 
59     public void stopWorker() {
60         stopped = true;
61         synchronized(jobs) {
62             jobs.notify();
63         }
64     }
65 
66     public void add(Runnable job) {
67         synchronized(jobs) {
68             jobs.add(job);
69             jobs.notify();
70         }
71     }
72 
73     public boolean queueFull() {
74         synchronized(jobs) {
75             return (jobs.size() > 0);
76         }
77     }
78 }
