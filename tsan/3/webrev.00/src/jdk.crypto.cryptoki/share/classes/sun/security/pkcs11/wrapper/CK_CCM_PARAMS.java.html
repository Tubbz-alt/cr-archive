26 package sun.security.pkcs11.wrapper;
27 
28 /**
29  * This class represents the necessary parameters required by
30  * the CKM_AES_CCM mechanism as defined in CK_CCM_PARAMS structure.<p>
31  * <B>PKCS#11 structure:</B>
32  * <PRE>
33  * typedef struct CK_CCM_PARAMS {
34  *   CK_ULONG ulDataLen;
35  *   CK_BYTE_PTR pNonce;
36  *   CK_ULONG ulNonceLen;
37  *   CK_BYTE_PTR pAAD;
38  *   CK_ULONG ulAADLen;
39  *   CK_ULONG ulMACLen;
40  * } CK_CCM_PARAMS;
41  * </PRE>
42  *
43  * @since   13
44  */
45 public class CK_CCM_PARAMS {
46 
47     private final long dataLen;
48     private final byte[] nonce;
49     private final byte[] aad;
50     private final long macLen;
51 
52     public CK_CCM_PARAMS(int tagLen, byte[] iv, byte[] aad, int dataLen) {
53         this.dataLen = dataLen;
54         this.nonce = iv;
55         this.aad = aad;
56         this.macLen = tagLen;
57     }
58 
59     public String toString() {
60         StringBuilder sb = new StringBuilder();
61 
62         sb.append(Constants.INDENT);
63         sb.append("ulDataLen: ");
64         sb.append(dataLen);
65         sb.append(Constants.NEWLINE);
66 
67         sb.append(Constants.INDENT);
68         sb.append("iv: ");
69         sb.append(Functions.toHexString(nonce));
70         sb.append(Constants.NEWLINE);
71 
72         sb.append(Constants.INDENT);
73         sb.append("aad: ");
74         sb.append(Functions.toHexString(aad));
75         sb.append(Constants.NEWLINE);
76 
77         sb.append(Constants.INDENT);
78         sb.append("tagLen: ");
79         sb.append(macLen);
80 
81         return sb.toString();
82     }
83 }
