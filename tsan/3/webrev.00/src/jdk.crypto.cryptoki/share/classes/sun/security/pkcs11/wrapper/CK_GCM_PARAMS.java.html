26 package sun.security.pkcs11.wrapper;
27 
28 /**
29  * This class represents the necessary parameters required by
30  * the CKM_AES_GCM mechanism as defined in CK_GCM_PARAMS structure.<p>
31  * <B>PKCS#11 structure:</B>
32  * <PRE>
33  * typedef struct CK_GCM_PARAMS {
34  *    CK_BYTE_PTR       pIv;
35  *    CK_ULONG          ulIvLen;
36  *    CK_BYTE_PTR       pAAD;
37  *    CK_ULONG          ulAADLen;
38  *    CK_ULONG          ulTagBits;
39  * } CK_GCM_PARAMS;
40  * </PRE>
41  *
42  * @since   10
43  */
44 public class CK_GCM_PARAMS {
45 
46     private final byte[] iv;
47     private final byte[] aad;
48     private final long tagBits;
49 
50     public CK_GCM_PARAMS(int tagLenInBits, byte[] iv, byte[] aad) {
51         this.iv = iv;
52         this.aad = aad;
53         this.tagBits = tagLenInBits;
54     }
55 
56     public String toString() {
57         StringBuilder sb = new StringBuilder();
58 
59         sb.append(Constants.INDENT);
60         sb.append("iv: ");
61         sb.append(Functions.toHexString(iv));
62         sb.append(Constants.NEWLINE);
63 
64         sb.append(Constants.INDENT);
65         sb.append("aad: ");
66         sb.append(Functions.toHexString(aad));
67         sb.append(Constants.NEWLINE);
68 
69         sb.append(Constants.INDENT);
70         sb.append("tagLen(in bits): ");
71         sb.append(tagBits);
72 
73         return sb.toString();
74     }
75 }
