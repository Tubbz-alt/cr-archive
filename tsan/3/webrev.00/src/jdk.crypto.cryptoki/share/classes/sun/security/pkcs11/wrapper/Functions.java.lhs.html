105     /**
106      * Converts a long value to a hexadecimal String of length 16. Includes
107      * leading zeros if necessary.
108      *
109      * @param value The long value to be converted.
110      * @return The hexadecimal string representation of the long value.
111      */
112     public static String toFullHexString(long value) {
113         long currentValue = value;
114         StringBuilder sb = new StringBuilder(16);
115         for(int j = 0; j < 16; j++) {
116             int currentDigit = (int) currentValue & 0xf;
117             sb.append(HEX_DIGITS[currentDigit]);
118             currentValue >>>= 4;
119         }
120 
121         return sb.reverse().toString();
122     }

124     /**
125      * Converts a int value to a hexadecimal String of length 8. Includes
126      * leading zeros if necessary.
127      *
128      * @param value The int value to be converted.
129      * @return The hexadecimal string representation of the int value.
130      */
131     public static String toFullHexString(int value) {
132         int currentValue = value;
133         StringBuilder sb = new StringBuilder(8);
134         for(int i = 0; i < 8; i++) {
135             int currentDigit = currentValue & 0xf;
136             sb.append(HEX_DIGITS[currentDigit]);
137             currentValue >>>= 4;
138         }
139 
140         return sb.reverse().toString();
141     }

153     /**
154      * Converts a byte array to a hexadecimal String. Each byte is presented by
155      * its two digit hex-code; 0x0A -> "0a", 0x00 -> "00". No leading "0x" is
156      * included in the result.
157      *
158      * @param value the byte array to be converted
159      * @return the hexadecimal string representation of the byte array
160      */
161     public static String toHexString(byte[] value) {
162         if (value == null) {
163             return null;
164         }
165 
166         StringBuilder sb = new StringBuilder(2 * value.length);
167         int          single;
168 
169         for (int i = 0; i < value.length; i++) {
170             single = value[i] & 0xFF;
171 
172             if (single < 0x10) {
173                 sb.append('0');
174             }
175 
176             sb.append(Integer.toString(single, 16));
177         }
178 
179         return sb.toString();
180     }

204     private static class Flags {
205         private final long[] flagIds;
206         private final String[] flagNames;
207         Flags(long[] flagIds, String[] flagNames) {
208             if (flagIds.length != flagNames.length) {
209                 throw new AssertionError("Array lengths do not match");
210             }
211             this.flagIds = flagIds;
212             this.flagNames = flagNames;
213         }
214         String toString(long val) {
215             StringBuilder sb = new StringBuilder();
216             boolean first = true;
217             for (int i = 0; i < flagIds.length; i++) {
218                 if ((val & flagIds[i]) != 0) {
219                     if (first == false) {
220                         sb.append(" | ");
221                     }
222                     sb.append(flagNames[i]);
223                     first = false;
224                 }
225             }
226             return sb.toString();
227         }
228     }

319      * converts the long value state to a SessionState string
320      *
321      * @param state the state to be converted
322      * @return the SessionState string representation of the state
323      */
324     public static String sessionStateToString(long state) {
325         String name;
326 
327         if (state == CKS_RO_PUBLIC_SESSION) {
328             name = "CKS_RO_PUBLIC_SESSION";
329         } else if (state == CKS_RO_USER_FUNCTIONS) {
330             name = "CKS_RO_USER_FUNCTIONS";
331         } else if (state == CKS_RW_PUBLIC_SESSION) {
332             name = "CKS_RW_PUBLIC_SESSION";
333         } else if (state == CKS_RW_USER_FUNCTIONS) {
334             name = "CKS_RW_USER_FUNCTIONS";
335         } else if (state == CKS_RW_SO_FUNCTIONS) {
336             name = "CKS_RW_SO_FUNCTIONS";
337         } else {
338             name = "ERROR: unknown session state 0x" + toFullHexString(state);
339         }
340 
341         return name;
342     }

398     private static String getName(Map<Integer,String> nameMap, long id) {
399         String name = null;
400         if ((id >>> 32) == 0) {
401             name = nameMap.get(Integer.valueOf((int)id));
402         }
403         if (name == null) {
404             name = "Unknown 0x" + toFullHexString(id);




405         }
406         return name;
407     }

469     /**
470      * Check the given dates for equalitiy. This method considers both dates as
471      * equal, if both are <code>null</code> or both contain exactly the same char
472      * values.
473      *
474      * @param date1 The first date.
475      * @param date2 The second date.
476      * @return True, if both dates are <code>null</code> or both contain the same
477      *         char values. False, otherwise.
478      * @preconditions
479      * @postconditions
480      */
481     public static boolean equals(CK_DATE date1, CK_DATE date2) {
482         boolean equal = false;
483 
484         if (date1 == date2) {
485             equal = true;
486         } else if ((date1 != null) && (date2 != null)) {
487             equal = equals(date1.year, date2.year)
488               && equals(date1.month, date2.month)
489               && equals(date1.day, date2.day);
490         } else {
491             equal = false;
492         }
493 
494         return equal ;
495     }

497     /**
498      * Calculate a hash code for the given byte array.
499      *
500      * @param array The byte array.
501      * @return A hash code for the given array.
502      * @preconditions
503      * @postconditions
504      */
505     public static int hashCode(byte[] array) {
506         int hash = 0;
507 
508         if (array != null) {
509             for (int i = 0; (i < 4) && (i < array.length); i++) {
510                 hash ^= (0xFF & array[i]) << ((i%4) << 3);
511             }
512         }
513 
514         return hash ;
515     }

517     /**
518      * Calculate a hash code for the given char array.
519      *
520      * @param array The char array.
521      * @return A hash code for the given array.
522      * @preconditions
523      * @postconditions

525     public static int hashCode(char[] array) {
526         int hash = 0;
527 
528         if (array != null) {
529             for (int i = 0; (i < 4) && (i < array.length); i++) {

531             }
532         }
533 
534         return hash ;

537     /**
538      * Calculate a hash code for the given date object.
539      *
540      * @param date The date object.
541      * @return A hash code for the given date.
542      * @preconditions
543      * @postconditions
544      */
545     public static int hashCode(CK_DATE date) {
546         int hash = 0;
547 
548         if (date != null) {
549             if (date.year.length == 4) {
550                 hash ^= (0xFFFF & date.year[0]) << 16;
551                 hash ^= 0xFFFF & date.year[1];
552                 hash ^= (0xFFFF & date.year[2]) << 16;
553                 hash ^= 0xFFFF & date.year[3];
554             }
555             if (date.month.length == 2) {
556                 hash ^= (0xFFFF & date.month[0]) << 16;
557                 hash ^= 0xFFFF & date.month[1];
558             }
559             if (date.day.length == 2) {
560                 hash ^= (0xFFFF & date.day[0]) << 16;
561                 hash ^= 0xFFFF & date.day[1];
562             }
563         }
564 
565         return hash ;
566     }

568     private static void addMapping(Map<Integer,String> nameMap,
569             Map<String,Integer> idMap, long id, String name) {
570         if ((id >>> 32) != 0) {
571             throw new AssertionError("Id has high bits set: " + id + ", " + name);
572         }
573         Integer intId = Integer.valueOf((int)id);
574         if (nameMap.put(intId, name) != null) {
575             throw new AssertionError("Duplicate id: " + id + ", " + name);
576         }
577         if (idMap.put(name, intId) != null) {
578             throw new AssertionError("Duplicate name: " + id + ", " + name);
579         }
580     }
