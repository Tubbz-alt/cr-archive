 112     /**
 113      * Converts a long value to a hexadecimal String of length 16. Includes
 114      * leading zeros if necessary.
 115      *
 116      * @param value The long value to be converted.
 117      * @return The hexadecimal string representation of the long value.
 118      */
 119     public static String toFullHexString(long value) {
 120         long currentValue = value;
 121         StringBuilder sb = new StringBuilder(16);
 122         for(int j = 0; j < 16; j++) {
 123             int currentDigit = (int) currentValue & 0xf;
 124             sb.append(HEX_DIGITS[currentDigit]);
 125             currentValue >>>= 4;
 126         }
 127 
 128         return sb.reverse().toString();
 129     }

 135      * @param value The int value to be converted.

 138     public static String toFullHexString(int value) {

 140         StringBuilder sb = new StringBuilder(8);
 141         for(int i = 0; i < 8; i++) {
 142             int currentDigit = currentValue & 0xf;
 143             sb.append(HEX_DIGITS[currentDigit]);
 144             currentValue >>>= 4;

 146 
 147         return sb.reverse().toString();

 160     /**
 161      * Converts a byte array to a hexadecimal String. Each byte is presented by
 162      * its two digit hex-code; 0x0A -> "0a", 0x00 -> "00". No leading "0x" is
 163      * included in the result.
 164      *
 165      * @param value the byte array to be converted
 166      * @return the hexadecimal string representation of the byte array
 167      */
 168     public static String toHexString(byte[] value) {
 169         if (value == null) {
 170             return null;
 171         }
 172 
 173         StringBuilder sb = new StringBuilder(2 * value.length);
 174         int          single;
 175 
 176         for (int i = 0; i < value.length; i++) {
 177             single = value[i] & 0xFF;
 178 
 179             if (single < 0x10) {
 180                 sb.append('0');
 181             }
 182 
 183             sb.append(Integer.toString(single, 16));
 184         }
 185 
 186         return sb.toString();
 187     }

 211     private static class Flags {
 212         private final long[] flagIds;
 213         private final String[] flagNames;
 214         Flags(long[] flagIds, String[] flagNames) {
 215             if (flagIds.length != flagNames.length) {
 216                 throw new AssertionError("Array lengths do not match");
 217             }
 218             this.flagIds = flagIds;
 219             this.flagNames = flagNames;
 220         }
 221         String toString(long val) {
 222             StringBuilder sb = new StringBuilder();
 223             boolean first = true;
 224             for (int i = 0; i < flagIds.length; i++) {
 225                 if ((val & flagIds[i]) != 0) {
 226                     if (first == false) {
 227                         sb.append(" | ");
 228                     }
 229                     sb.append(flagNames[i]);
 230                     first = false;
 231                 }
 232             }
 233             return sb.toString();
 234         }
 235     }

 336         } else if (state == CKS_RO_USER_FUNCTIONS) {
 337             name = "CKS_RO_USER_FUNCTIONS";

 339             name = "CKS_RW_PUBLIC_SESSION";

 341             name = "CKS_RW_USER_FUNCTIONS";
 342         } else if (state == CKS_RW_SO_FUNCTIONS) {
 343             name = "CKS_RW_SO_FUNCTIONS";
 344         } else {
 345             name = "ERROR: unknown session state 0x" + toFullHexString(state);
 346         }
 347 
 348         return name;

 405     private static String getName(Map<Integer,String> nameMap, long id) {
 406         String name = null;
 407         if ((id >>> 32) == 0) {
 408             name = nameMap.get(Integer.valueOf((int)id));
 409         }
 410         if (name == null) {
 411             if ((id & CKM_VENDOR_DEFINED) != 0) {
 412                 name = "(Vendor-Specific) 0x" + toFullHexString(id);
 413             } else {
 414                 name = "(Unknown) 0x" + toFullHexString(id);
 415             }
 416         }
 417         return name;
 418     }

 488     /**
 489      * Check the given dates for equalitiy. This method considers both dates as
 490      * equal, if both are <code>null</code> or both contain exactly the same char
 491      * values.
 492      *
 493      * @param date1 The first date.
 494      * @param date2 The second date.
 495      * @return True, if both dates are <code>null</code> or both contain the same
 496      *         char values. False, otherwise.
 497      * @preconditions
 498      * @postconditions
 499      */
 500     public static boolean equals(CK_DATE date1, CK_DATE date2) {
 501         boolean equal = false;
 502 
 503         if (date1 == date2) {
 504             equal = true;
 505         } else if ((date1 != null) && (date2 != null)) {
 506             equal = equals(date1.year, date2.year)
 507               && equals(date1.month, date2.month)
 508               && equals(date1.day, date2.day);
 509         } else {
 510             equal = false;
 511         }
 512 
 513         return equal ;
 514     }

 516     /**
 517      * Calculate a hash code for the given byte array.
 518      *
 519      * @param array The byte array.
 520      * @return A hash code for the given array.
 521      * @preconditions
 522      * @postconditions
 523      */
 524     public static int hashCode(byte[] array) {
 525         int hash = 0;
 526 
 527         if (array != null) {
 528             for (int i = 0; (i < 4) && (i < array.length); i++) {
 529                 hash ^= (0xFF & array[i]) << ((i%4) << 3);
 530             }
 531         }
 532 
 533         return hash ;
 534     }

 556     /**
 557      * Calculate a hash code for the given date object.
 558      *
 559      * @param date The date object.
 560      * @return A hash code for the given date.
 561      * @preconditions
 562      * @postconditions
 563      */
 564     public static int hashCode(CK_DATE date) {
 565         int hash = 0;
 566 
 567         if (date != null) {
 568             if (date.year.length == 4) {
 569                 hash ^= (0xFFFF & date.year[0]) << 16;
 570                 hash ^= 0xFFFF & date.year[1];
 571                 hash ^= (0xFFFF & date.year[2]) << 16;
 572                 hash ^= 0xFFFF & date.year[3];
 573             }
 574             if (date.month.length == 2) {
 575                 hash ^= (0xFFFF & date.month[0]) << 16;
 576                 hash ^= 0xFFFF & date.month[1];
 577             }
 578             if (date.day.length == 2) {
 579                 hash ^= (0xFFFF & date.day[0]) << 16;
 580                 hash ^= 0xFFFF & date.day[1];
 581             }
 582         }
 583 
 584         return hash ;
 585     }

 587     private static void addMapping(Map<Integer,String> nameMap,
 588             Map<String,Integer> idMap, long id, String name) {
 589         if ((id >>> 32) != 0) {
 590             throw new AssertionError("Id has high bits set: " + id + ", " + name);
 591         }
 592         Integer intId = Integer.valueOf((int)id);
 593         if (nameMap.put(intId, name) != null) {
 594             throw new AssertionError("Duplicate id: " + id + ", " + name);
 595         }
 596         if (idMap.put(name, intId) != null) {
 597             throw new AssertionError("Duplicate name: " + id + ", " + name);
 598         }
 599     }
