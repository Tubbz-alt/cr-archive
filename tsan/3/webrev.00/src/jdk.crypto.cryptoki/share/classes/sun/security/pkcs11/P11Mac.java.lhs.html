 26 package sun.security.pkcs11;
 27 
 28 import java.nio.ByteBuffer;
 29 
 30 import java.security.*;
 31 import java.security.spec.AlgorithmParameterSpec;
 32 
 33 import javax.crypto.MacSpi;
 34 
 35 import sun.nio.ch.DirectBuffer;
 36 
 37 import sun.security.pkcs11.wrapper.*;
 38 import static sun.security.pkcs11.wrapper.PKCS11Constants.*;
 39 
 40 /**
 41  * MAC implementation class. This class currently supports HMAC using
 42  * MD5, SHA-1, SHA-224, SHA-256, SHA-384, and SHA-512 and the SSL3 MAC
 43  * using MD5 and SHA-1.
 44  *
 45  * Note that unlike other classes (e.g. Signature), this does not
 46  * composite various operations if the token only supports part of the
 47  * required functionality. The MAC implementations in SunJCE already
 48  * do exactly that by implementing an MAC on top of MessageDigests. We
 49  * could not do any better than they.
 50  *
 51  * @author  Andreas Sterbenz
 52  * @since   1.5
 53  */
 54 final class P11Mac extends MacSpi {
 55 
 56     // token instance
 57     private final Token token;
 58 
 59     // algorithm name
 60     private final String algorithm;
 61 
 62     // mechanism object
 63     private final CK_MECHANISM ckMechanism;
 64 
 65     // length of the MAC in bytes
 66     private final int macLength;
 67 
 68     // key instance used, if operation active
 69     private P11Key p11Key;
 70 
 71     // associated session, if any
 72     private Session session;
 73 
 74     // initialization status
 75     private boolean initialized;
 76 
 77     // one byte buffer for the update(byte) method, initialized on demand
 78     private byte[] oneByte;
 79 
 80     P11Mac(Token token, String algorithm, long mechanism)
 81             throws PKCS11Exception {
 82         super();
 83         this.token = token;
 84         this.algorithm = algorithm;
 85         Long params = null;
 86         switch ((int)mechanism) {
 87         case (int)CKM_MD5_HMAC:
 88             macLength = 16;
 89             break;
 90         case (int)CKM_SHA_1_HMAC:
 91             macLength = 20;
 92             break;
 93         case (int)CKM_SHA224_HMAC:

 94             macLength = 28;
 95             break;
 96         case (int)CKM_SHA256_HMAC:

 97             macLength = 32;
 98             break;
 99         case (int)CKM_SHA384_HMAC:
100             macLength = 48;
101             break;
102         case (int)CKM_SHA512_HMAC:
103             macLength = 64;
104             break;
105         case (int)CKM_SSL3_MD5_MAC:
106             macLength = 16;
107             params = Long.valueOf(16);
108             break;
109         case (int)CKM_SSL3_SHA1_MAC:
110             macLength = 20;
111             params = Long.valueOf(20);
112             break;
113         default:
114             throw new ProviderException("Unknown mechanism: " + mechanism);
115         }
116         ckMechanism = new CK_MECHANISM(mechanism, params);
117     }
118 
119     // reset the states to the pre-initialized values
120     private void reset(boolean doCancel) {
121         if (!initialized) {
122             return;
123         }
124         initialized = false;

125         try {
126             if (session == null) {
127                 return;
128             }

129             if (doCancel && token.explicitCancel) {
130                 cancelOperation();
131             }
132         } finally {
133             p11Key.releaseKeyID();
134             session = token.releaseSession(session);
135         }
136     }
137 
138     private void cancelOperation() {
139         token.ensureValid();
140         if (session.hasObjects() == false) {
141             session = token.killSession(session);
142             return;
143         } else {
144             try {
145                 token.p11.C_SignFinal(session.id(), 0);
146             } catch (PKCS11Exception e) {
147                 throw new ProviderException("Cancel failed", e);
148             }
149         }
150     }
151 
152     private void ensureInitialized() throws PKCS11Exception {
153         if (!initialized) {
154             initialize();
155         }
156     }
157 
158     private void initialize() throws PKCS11Exception {
159         if (p11Key == null) {
160             throw new ProviderException(
161                     "Operation cannot be performed without calling engineInit first");
162         }
163         token.ensureValid();
164         long p11KeyID = p11Key.getKeyID();
165         try {
166             if (session == null) {
167                 session = token.getOpSession();
168             }
169             token.p11.C_SignInit(session.id(), ckMechanism, p11KeyID);
170         } catch (PKCS11Exception e) {
171             p11Key.releaseKeyID();
172             session = token.releaseSession(session);
173             throw e;
174         }
175         initialized = true;
176     }
177 
178     // see JCE spec
179     protected int engineGetMacLength() {
180         return macLength;
181     }
182 
183     // see JCE spec
184     protected void engineReset() {
185         reset(true);
186     }
187 
188     // see JCE spec
189     protected void engineInit(Key key, AlgorithmParameterSpec params)
190             throws InvalidKeyException, InvalidAlgorithmParameterException {
191         if (params != null) {
192             throw new InvalidAlgorithmParameterException
193                 ("Parameters not supported");
194         }
195         reset(true);
196         p11Key = P11SecretKeyFactory.convertKey(token, key, algorithm);
197         try {
198             initialize();
199         } catch (PKCS11Exception e) {
200             throw new InvalidKeyException("init() failed", e);
201         }
202     }
203 
204     // see JCE spec
205     protected byte[] engineDoFinal() {
206         try {
207             ensureInitialized();
208             return token.p11.C_SignFinal(session.id(), 0);
209         } catch (PKCS11Exception e) {
210             reset(true);
211             throw new ProviderException("doFinal() failed", e);
212         } finally {
213             reset(false);
214         }
215     }
216 
217     // see JCE spec
218     protected void engineUpdate(byte input) {
219         if (oneByte == null) {
220            oneByte = new byte[1];
221         }
222         oneByte[0] = input;
223         engineUpdate(oneByte, 0, 1);
224     }
225 
226     // see JCE spec
227     protected void engineUpdate(byte[] b, int ofs, int len) {
228         try {
229             ensureInitialized();
230             token.p11.C_SignUpdate(session.id(), 0, b, ofs, len);
231         } catch (PKCS11Exception e) {
232             throw new ProviderException("update() failed", e);
233         }
234     }
235 
236     // see JCE spec
237     protected void engineUpdate(ByteBuffer byteBuffer) {
238         try {
239             ensureInitialized();
240             int len = byteBuffer.remaining();
241             if (len <= 0) {
242                 return;
243             }
244             if (byteBuffer instanceof DirectBuffer == false) {
245                 super.engineUpdate(byteBuffer);
246                 return;
247             }
248             long addr = ((DirectBuffer)byteBuffer).address();
249             int ofs = byteBuffer.position();
250             token.p11.C_SignUpdate(session.id(), addr + ofs, null, 0, len);
251             byteBuffer.position(ofs + len);
252         } catch (PKCS11Exception e) {
253             throw new ProviderException("update() failed", e);
254         }
255     }
256 }
