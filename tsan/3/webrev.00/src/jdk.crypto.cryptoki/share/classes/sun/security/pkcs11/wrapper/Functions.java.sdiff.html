 398     private static String getName(Map<Integer,String> nameMap, long id) {
 399         String name = null;
 400         if ((id >>> 32) == 0) {
 401             name = nameMap.get(Integer.valueOf((int)id));
 402         }
 403         if (name == null) {
 404             name = "Unknown 0x" + toFullHexString(id);




 405         }
 406         return name;

 405     private static String getName(Map<Integer,String> nameMap, long id) {
 406         String name = null;
 407         if ((id >>> 32) == 0) {
 408             name = nameMap.get(Integer.valueOf((int)id));
 409         }
 410         if (name == null) {
 411             if ((id & CKM_VENDOR_DEFINED) != 0) {
 412                 name = "(Vendor-Specific) 0x" + toFullHexString(id);
 413             } else {
 414                 name = "(Unknown) 0x" + toFullHexString(id);
 415             }
 416         }
 417         return name;
 418     }
