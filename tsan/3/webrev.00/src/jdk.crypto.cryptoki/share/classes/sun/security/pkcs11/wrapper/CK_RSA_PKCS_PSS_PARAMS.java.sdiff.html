 76     /**

 28 import java.security.ProviderException;
 29 import java.security.spec.PSSParameterSpec;
 30 import java.security.spec.MGF1ParameterSpec;

 33 /**
 34  * This class represents the necessary parameters required by the
 35  * CKM_RSA_PKCS_PSS mechanism as defined in CK_RSA_PKCS_PSS_PARAMS structure.<p>
 36  * <B>PKCS#11 structure:</B>
 37  * <PRE>
 38  * typedef struct CK_RSA_PKCS_PSS_PARAMS {
 39  *    CK_MECHANISM_TYPE    hashAlg;
 40  *    CK_RSA_PKCS_MGF_TYPE mgf;
 41  *    CK_ULONG             sLen;
 42  * } CK_RSA_PKCS_PSS_PARAMS;
 43  * </PRE>
 44  *
 45  * @since   13
 46  */
 47 public class CK_RSA_PKCS_PSS_PARAMS {
 48 
 49     private final long hashAlg;
 50     private final long mgf;
 51     private final long sLen;




 52 
 53     public CK_RSA_PKCS_PSS_PARAMS(String hashAlg, String mgfAlg,
 54             String mgfHash, int sLen) {
 55         this.hashAlg = Functions.getHashMechId(hashAlg);
 56         if (!mgfAlg.equals("MGF1")) {
 57             throw new ProviderException("Only MGF1 is supported");
 58         }
 59         // no dash in PKCS#11 mechanism names
 60         this.mgf = Functions.getMGFId("CKG_MGF1_" + mgfHash.replaceFirst("-", ""));
 61         this.sLen = sLen;
 62     }
 63 
 64     @Override
 65     public boolean equals(Object o) {
 66         if (o == this) {
 67             return true;
 68         }


 69 
 70         if (!(o instanceof CK_RSA_PKCS_PSS_PARAMS)) {
 71             return false;
 72         }
 73 
 74         CK_RSA_PKCS_PSS_PARAMS other = (CK_RSA_PKCS_PSS_PARAMS) o;
 75         return ((other.hashAlg == hashAlg) &&
 76                 (other.mgf == mgf) &&
 77                 (other.sLen == sLen));
 78     }
 79 
 80     @Override
 81     public int hashCode() {
 82         return (int)(hashAlg << 2 + mgf << 1 + sLen);
 83     }
 84 
 85     @Override
 86     public String toString() {
 87         StringBuilder sb = new StringBuilder();
 88 
 89         sb.append(Constants.INDENT);
 90         sb.append("hashAlg: ");
 91         sb.append(Functions.toFullHexString(hashAlg));
 92         sb.append(Constants.NEWLINE);
 93 
 94         sb.append(Constants.INDENT);
 95         sb.append("mgf: ");
 96         sb.append(Functions.toFullHexString(mgf));
 97         sb.append(Constants.NEWLINE);
 98 
 99         sb.append(Constants.INDENT);
100         sb.append("sLen(in bytes): ");
101         sb.append(sLen);

102 
103         return sb.toString();
104     }
