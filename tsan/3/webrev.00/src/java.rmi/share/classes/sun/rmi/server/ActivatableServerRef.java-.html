26 package sun.rmi.server;
27 
28 import java.io.IOException;
29 import java.io.NotSerializableException;
30 import java.io.ObjectOutput;
31 import java.rmi.*;
32 import java.rmi.server.*;
33 import java.rmi.activation.ActivationID;
34 import sun.rmi.transport.LiveRef;
35 
36 /**
37  * Server-side ref for a persistent remote impl.
38  *
39  * @author Ann Wollrath
40  */
41 public class ActivatableServerRef extends UnicastServerRef2 {
42 
43     private static final long serialVersionUID = 2002967993223003793L;
44 
45     private ActivationID id;
46 
47     /**
48      * Construct a Unicast server remote reference to be exported
49      * on the specified port.
50      */
51     public ActivatableServerRef(ActivationID id, int port)
52     {
53         this(id, port, null, null);
54     }
55 
56     /**
57      * Construct a Unicast server remote reference to be exported
58      * on the specified port.
59      */
60     public ActivatableServerRef(ActivationID id, int port,
61                                 RMIClientSocketFactory csf,
62                                 RMIServerSocketFactory ssf)
63     {
64         super(new LiveRef(port, csf, ssf));
65         this.id = id;
66     }
67 
68     /**
69      * Returns the class of the ref type to be serialized
70      */
71     public String getRefClass(ObjectOutput out)
72     {
73         return "ActivatableServerRef";
74     }
75 
76     /**
77      * Return the client remote reference for this remoteRef.
78      * In the case of a client RemoteRef "this" is the answer.
79      * For  a server remote reference, a client side one will have to
80      * found or created.
81      */
82     protected RemoteRef getClientRef() {
83         return new ActivatableRef(id, new UnicastRef2(ref));
84     }
85 
86     /**
87      * Prevents serialization (because deserializaion is impossible).
88      */
89     public void writeExternal(ObjectOutput out) throws IOException {
90         throw new NotSerializableException(
91             "ActivatableServerRef not serializable");
92     }
93 }
