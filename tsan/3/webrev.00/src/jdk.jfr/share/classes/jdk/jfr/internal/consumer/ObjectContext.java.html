26 package jdk.jfr.internal.consumer;
27 
28 import java.time.ZoneId;
29 import java.util.HashMap;
30 import java.util.List;
31 import java.util.Map;
32 
33 import jdk.jfr.EventType;
34 import jdk.jfr.ValueDescriptor;
35 
36 public final class ObjectContext {
37     private final Map<ValueDescriptor, ObjectContext> contextLookup;
38     private final TimeConverter timeConverter;
39 
40     public final EventType eventType;
41     public final List<ValueDescriptor> fields;
42 
43     ObjectContext(EventType eventType, List<ValueDescriptor> fields, TimeConverter timeConverter) {
44         this.contextLookup = new HashMap<>();
45         this.eventType = eventType;
46         this.fields = fields;
47         this.timeConverter = timeConverter;
48     }
49 
50     private ObjectContext(ObjectContext parent, ValueDescriptor descriptor) {
51         this.eventType = parent.eventType;
52         this.contextLookup = parent.contextLookup;
53         this.timeConverter = parent.timeConverter;
54         this.fields = descriptor.getFields();
55     }
56 
57     public ObjectContext getInstance(ValueDescriptor descriptor) {
58         ObjectContext context = contextLookup.get(descriptor);
59         if (context == null) {
60             context = new ObjectContext(this, descriptor);
61             contextLookup.put(descriptor, context);
62         }
63         return context;
64     }
65 
66     public long convertTimestamp(long ticks) {
67         return timeConverter.convertTimestamp(ticks);
68     }
69 
70     public long convertTimespan(long ticks) {
71         return timeConverter.convertTimespan(ticks);
72     }
73 
74     public ZoneId getZoneOffset() {
75         return timeConverter.getZoneOffset();
76     }
77 }
