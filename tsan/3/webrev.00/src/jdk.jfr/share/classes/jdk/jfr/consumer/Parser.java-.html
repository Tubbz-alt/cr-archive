26 package jdk.jfr.consumer;
27 
28 import java.io.IOException;
29 
30 import jdk.jfr.internal.consumer.RecordingInput;
31 
32 /**
33  * Base class for parsing data from a {@link RecordingInput}.
34  */
35 abstract class Parser {
36     /**
37      * Parses data from a {@link RecordingInput} and return an object.
38      *
39      * @param input input to read from
40      * @return an object
41      * @throws IOException if operation couldn't be completed due to I/O
42      *         problems
43      */
44     abstract Object parse(RecordingInput input) throws IOException;
45 }
