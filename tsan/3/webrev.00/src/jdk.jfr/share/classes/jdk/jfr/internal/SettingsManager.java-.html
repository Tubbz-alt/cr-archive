 28 import java.util.ArrayList;
 29 import java.util.Collection;
 30 import java.util.Collections;
 31 import java.util.HashMap;
 32 import java.util.HashSet;
 33 import java.util.LinkedHashMap;

 42 final class SettingsManager {
 43 
 44     private static class InternalSetting {
 45 
 46         private final String identifier;
 47         private Map<String, Set<String>> enabledMap = new LinkedHashMap<>(5);
 48         private Map<String, Set<String>> allMap = new LinkedHashMap<>(5);
 49         private boolean enabled;
 50 
 51         /**
 52          * Settings identifier, for example "com.example.HelloWorld" or "56"
 53          * (id of event)
 54          *
 55          * @param settingsId
 56          */
 57         public InternalSetting(String settingsId) {
 58             this.identifier = settingsId;
 59         }
 60 
 61         public Set<String> getValues(String key) {
 62             if (enabled) {
 63                 return enabledMap.get(key);
 64             } else {
 65                 return allMap.get(key);
 66             }
 67         }
 68 
 69         public void add(String attribute, String value) {
 70             if ("enabled".equals(attribute) && "true".equals(value)) {
 71                 enabled = true;
 72                 allMap = null; // no need to keep these around
 73             }
 74             addToMap(enabledMap, attribute, value);
 75             if (allMap != null) {
 76                 addToMap(allMap, attribute, value);
 77             }
 78         }
 79 
 80         private void addToMap(Map<String, Set<String>> map, String attribute, String value) {
 81             Set<String> values = map.get(attribute);
 82             if (values == null) {
 83                 values = new HashSet<String>(5);
 84                 map.put(attribute, values);
 85             }
 86             values.add(value);
 87 
 88         }
 89 
 90         public String getSettingsId() {
 91             return identifier;
 92         }
 93 
 94         public void add(InternalSetting enabled) {
 95             for (Map.Entry<String, Set<String>> entry : enabled.enabledMap.entrySet()) {
 96                 for (String value : entry.getValue()) {
 97                     add(entry.getKey(), value);
 98                 }
 99             }
100         }
101 
102         public boolean isEnabled() {
103             return enabled;
104         }
105 
106         @Override
107         public String toString() {
108             StringBuilder sb = new StringBuilder();
109             sb.append(identifier);
110             sb.append(": ");
111             sb.append(enabledMap.toString());
112             return sb.toString();
113         }
114 
115         public void finish() {
116             if (!enabled) {
117                 // settings from disabled
118                 // events should not impact results, but
119                 // we can't clear enabledMap since enabled=false
120                 // needs be there, so events that are enabled
121                 // by default are turned off
122                 Map<String, Set<String>> disabledMap = new HashMap<>(2);
123                 Set<String> values = new HashSet<>(2);
124                 values.add("false");
125                 disabledMap.put("enabled", values);
126                 enabledMap = disabledMap;
127             }
128         }
129     }
130 
131    private Map<String, InternalSetting> availableSettings = new LinkedHashMap<>();
132 
133     void setSettings(List<Map<String, String>> activeSettings) {
134         // store settings so they are available if a new event class is loaded
135         availableSettings = createSettingsMap(activeSettings);
136         List<EventControl> eventControls = MetadataRepository.getInstance().getEventControls();
137         if (!JVM.getJVM().isRecording()) {
138             for (EventControl ec : eventControls) {
139                 ec.disable();
140             }
141         } else {
142             if (Logger.shouldLog(LogTag.JFR_SETTING, LogLevel.INFO)) {
143                 Collections.sort(eventControls, (x,y) -> x.getEventType().getName().compareTo(y.getEventType().getName()));
144             }
145             for (EventControl ec : eventControls) {
146                 setEventControl(ec);
147             }
148         }
149         if (JVM.getJVM().getAllowedToDoEventRetransforms()) {
150             updateRetransform(JVM.getJVM().getAllEventClasses());
151         }
152     }
153 
154     public void updateRetransform(List<Class<? extends jdk.internal.event.Event>> eventClasses) {
155         List<Class<?>> classes = new ArrayList<>();
156         for(Class<? extends jdk.internal.event.Event> eventClass: eventClasses) {
157             EventHandler eh = Utils.getHandler(eventClass);
158             if (eh != null ) {
159                 PlatformEventType eventType = eh.getPlatformEventType();
160                 if (eventType.isMarkedForInstrumentation()) {
161                     classes.add(eventClass);
162                     eventType.markForInstrumentation(false);
163                     // A bit premature to set it here, but hard to check
164                     // after call to retransformClasses.
165                     eventType.setInstrumented();
166                 }
167             }
168         }
169         if (!classes.isEmpty()) {
170             JVM.getJVM().retransformClasses(classes.toArray(new Class<?>[0]));
171         }
172     }
173 
174     private Map<String, InternalSetting> createSettingsMap(List<Map<String,String>> activeSettings) {
175         Map<String, InternalSetting> map = new LinkedHashMap<>(activeSettings.size());
176         for (Map<String, String> rec : activeSettings) {
177             for (InternalSetting internal : makeInternalSettings(rec)) {
178                 InternalSetting is = map.get(internal.getSettingsId());
179                 if (is == null) {
180                     map.put(internal.getSettingsId(), internal);
181                 } else {
182                     is.add(internal);
183                 }
184             }
185         }
186         return map;
187     }
188 
189     private Collection<InternalSetting> makeInternalSettings(Map<String, String> rec) {
190         Map<String, InternalSetting> internals = new LinkedHashMap<>();
191         for (Map.Entry<String, String> entry : rec.entrySet()) {
192             String key = entry.getKey();
193             String value = entry.getValue();
194             int index = key.indexOf("#");
195             if (index > 1 && index < key.length() - 2) {
196                 String eventName = key.substring(0, index);
197                 eventName = Utils.upgradeLegacyJDKEvent(eventName);
198                 InternalSetting s = internals.get(eventName);
199                 String settingName = key.substring(index + 1).trim();
200                 if (s == null) {
201                     s = new InternalSetting(eventName);
202                     internals.put(eventName, s);
203                 }
204                 s.add(settingName, value);
205             }
206         }
207       for (InternalSetting s : internals.values()) {
208          s.finish();
209       }
210 
211       return internals.values();
212     }

214     void setEventControl(EventControl ec) {
215         InternalSetting is = getInternalSetting(ec);
216         Logger.log(LogTag.JFR_SETTING, LogLevel.INFO, "Applied settings for " + ec.getEventType().getLogName() + " {");
217         for (Entry<String, Control> entry : ec.getEntries()) {
218             Set<String> values = null;
219             String settingName = entry.getKey();
220             if (is != null) {
221                 values = is.getValues(settingName);
222             }
223             Control control = entry.getValue();
224             if (values != null) {
225                 control.apply(values);
226                 String after = control.getLastValue();
227                 if (Logger.shouldLog(LogTag.JFR_SETTING, LogLevel.INFO)) {
228                     if (Utils.isSettingVisible(control, ec.getEventType().hasEventHook())) {
229                         if (values.size() > 1) {
230                             StringJoiner sj = new StringJoiner(", ", "{", "}");
231                             for (String s : values) {
232                                 sj.add("\"" + s + "\"");
233                             }
234                             String message = "  " + settingName + "= " + sj.toString() + " => \"" + after + "\"";
235                             Logger.log(LogTag.JFR_SETTING, LogLevel.INFO, message);
236                         } else {
237                             String message = "  " + settingName + "=\"" + control.getLastValue() + "\"";
238                             Logger.log(LogTag.JFR_SETTING, LogLevel.INFO, message);
239                         }
240                     }
241                 }
242             } else {
243                 control.setDefault();
244                 if (Logger.shouldLog(LogTag.JFR_SETTING, LogLevel.INFO)) {
245                     String message = "  " + settingName + "=\"" + control.getLastValue() + "\"";
246                     Logger.log(LogTag.JFR_SETTING, LogLevel.INFO, message);
247                 }
248             }
249         }
250         ec.writeActiveSettingEvent();
251         Logger.log(LogTag.JFR_SETTING, LogLevel.INFO, "}");
252     }
253 
254     private InternalSetting getInternalSetting(EventControl ec) {
255         String name = ec.getEventType().getName();
256         InternalSetting nameBased = availableSettings.get(name);
257         InternalSetting idBased = availableSettings.get(ec.getSettingsId());
258 
259         if (nameBased == null && idBased == null) {
260             return null;
261         }
262         if (idBased == null) {
263             return nameBased;
264         }
265         if (nameBased == null) {
266             return idBased;
267         }
268         InternalSetting mixed = new InternalSetting(nameBased.getSettingsId());
269         mixed.add(nameBased);
270         mixed.add(idBased);
271         return mixed;
272     }
273 
274     @Override
275     public String toString() {
276         StringBuilder sb = new StringBuilder();
277         for (InternalSetting enabled : availableSettings.values()) {
278             sb.append(enabled.toString());
279             sb.append("\n");
280         }
281         return sb.toString();
282     }
283 
284     boolean isEnabled(String eventName) {
285         InternalSetting is = availableSettings.get(eventName);
286         if (is == null) {
287             return false;
288         }
289         return is.isEnabled();
290     }
