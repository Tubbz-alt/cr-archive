26 package jdk.jfr.consumer;
27 
28 import jdk.jfr.internal.consumer.ObjectContext;



29 
30 /**
31  * A recorded Java class loader.
32  *
33  * @since 9
34  */
35 public final class RecordedClassLoader extends RecordedObject {










36     private final long uniqueId;
37 
38     // package private
39     RecordedClassLoader(ObjectContext objectContext, long id, Object[] values) {
40         super(objectContext, values);
41         this.uniqueId = id;
42     }
43 
44     /**
45      * Returns the class of the class loader.
46      * <P>
47      * If the bootstrap class loader is represented as {@code null} in the Java
48      * Virtual Machine (JVM), then {@code null} is also the return value of this
49      * method.
50      *
51      * @return class of the class loader, can be {@code null}
52      */
53     public RecordedClass getType() {
54         return getTyped("type", RecordedClass.class, null);
55     }
56 
57     /**
58      * Returns the name of the class loader (for example, "boot", "platform", and
59      * "app").
60      *
61      * @return the class loader name, can be {@code null}
62      */
63     public String getName() {
64         return getTyped("name", String.class, null);
65     }
66 
67     /**
68      * Returns a unique ID for the class loader.
69      * <p>
70      * The ID might not be the same between Java Virtual Machine (JVM) instances.
71      *
72      * @return a unique ID
73      */
74     public long getId() {
75         return uniqueId;
76     }
77 }
