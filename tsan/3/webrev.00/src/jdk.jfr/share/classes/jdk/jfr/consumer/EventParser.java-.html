26 package jdk.jfr.consumer;
27 
28 import static jdk.jfr.internal.EventInstrumentation.FIELD_DURATION;
29 
30 import java.io.IOException;
31 import java.util.List;
32 
33 import jdk.jfr.EventType;
34 import jdk.jfr.ValueDescriptor;
35 import jdk.jfr.internal.consumer.RecordingInput;
36 
37 /**
38  * Parses an event and returns a {@link RecordedEvent}.
39  *
40  */
41 final class EventParser extends Parser {
42     private final Parser[] parsers;
43     private final EventType eventType;
44     private final TimeConverter timeConverter;
45     private final boolean hasDuration;
46     private final List<ValueDescriptor> valueDescriptors;
47 
48     EventParser(TimeConverter timeConverter, EventType type, Parser[] parsers) {
49         this.timeConverter = timeConverter;
50         this.parsers = parsers;
51         this.eventType = type;
52         this.hasDuration = type.getField(FIELD_DURATION) != null;
53         this.valueDescriptors = type.getFields();
54     }
55 
56     @Override
57     public Object parse(RecordingInput input) throws IOException {
58         Object[] values = new Object[parsers.length];
59         for (int i = 0; i < parsers.length; i++) {
60             values[i] = parsers[i].parse(input);
61         }
62         Long startTicks = (Long) values[0];
63         long startTime = timeConverter.convertTimestamp(startTicks);
64         if (hasDuration) {
65             long durationTicks = (Long) values[1];
66             long endTime = timeConverter.convertTimestamp(startTicks + durationTicks);
67             return new RecordedEvent(eventType, valueDescriptors, values, startTime, endTime, timeConverter);
68         } else {
69             return new RecordedEvent(eventType, valueDescriptors, values, startTime, startTime, timeConverter);
70         }
71     }
72 }
