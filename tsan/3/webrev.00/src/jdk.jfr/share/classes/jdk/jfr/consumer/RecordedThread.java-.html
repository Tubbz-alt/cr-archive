 26 package jdk.jfr.consumer;
 27 
 28 import java.util.List;
 29 
 30 import jdk.jfr.ValueDescriptor;
 31 import jdk.jfr.internal.Type;
 32 
 33 /**
 34  * A recorded thread.
 35  *
 36  * @since 9
 37  */
 38 public final class RecordedThread extends RecordedObject {
 39 
 40     static ObjectFactory<RecordedThread> createFactory(Type type, TimeConverter timeConverter) {
 41         return new ObjectFactory<RecordedThread>(type) {
 42             @Override
 43             RecordedThread createTyped(List<ValueDescriptor> desc, long id, Object[] object) {
 44                 return new RecordedThread(desc, id, object, timeConverter);
 45             }
 46         };
 47     }
 48 
 49     private final long uniqueId;
 50 
 51     private RecordedThread(List<ValueDescriptor> descriptors, long id, Object[] values,  TimeConverter timeConverter) {
 52         super(descriptors, values, timeConverter);
 53         this.uniqueId = id;
 54     }
 55 
 56     /**
 57      * Returns the thread name used by the operating system.
 58      *
 59      * @return the OS thread name, or {@code null} if doesn't exist
 60      */
 61     public String getOSName() {
 62         return getTyped("osName", String.class, null);
 63     }
 64 
 65     /**
 66      * Returns the thread ID used by the operating system.
 67      *
 68      * @return The Java thread ID, or {@code -1} if doesn't exist
 69      */
 70     public long getOSThreadId() {
 71         Long l = getTyped("osThreadId", Long.class, -1L);
 72         return l.longValue();
 73     }
 74 
 75     /**
 76      * Returns the Java thread group, if available.
 77      *
 78      * @return the thread group, or {@code null} if doesn't exist
 79      */
 80     public RecordedThreadGroup getThreadGroup() {
 81         return getTyped("group", RecordedThreadGroup.class, null);
 82     }
 83 
 84     /**
 85      * Returns the Java thread name, or {@code null} if if doesn't exist.
 86      * <p>
 87      * Returns {@code java.lang.Thread.getName()} if the thread has a Java
 88      * representation. {@code null} otherwise.
 89      *
 90      * @return the Java thread name, or {@code null} if doesn't exist
 91      */
 92     public String getJavaName() {
 93         return getTyped("javaName", String.class, null);
 94     }
 95 
 96     /**
 97      * Returns the Java thread ID, or {@code -1} if it's not a Java thread.
 98      *
 99      * @return the Java thread ID, or {@code -1} if it's not a Java thread
100      */
101     public long getJavaThreadId() {
102         Long l = getTyped("javaThreadId", Long.class, -1L);
103         return l.longValue();
104     }
105 
106     /**
107      * Returns a unique ID for both native threads and Java threads that can't be
108      * reused within the lifespan of the JVM.
109      * <p>
110      * See {@link #getJavaThreadId()} for the ID that is returned by
111      * {@code java.lang.Thread.getId()}
112      *
113      * @return a unique ID for the thread
114      */
115     public long getId() {
116         return uniqueId;
117     }
118 }
