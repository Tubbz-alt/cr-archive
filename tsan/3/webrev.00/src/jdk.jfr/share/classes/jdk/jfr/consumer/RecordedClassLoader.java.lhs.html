26 package jdk.jfr.consumer;
27 
28 import java.util.List;
29 
30 import jdk.jfr.ValueDescriptor;
31 import jdk.jfr.internal.Type;
32 
33 /**
34  * A recorded Java class loader.
35  *
36  * @since 9
37  */
38 public final class RecordedClassLoader extends RecordedObject {
39 
40     static ObjectFactory<RecordedClassLoader> createFactory(Type type, TimeConverter timeConverter) {
41         return new ObjectFactory<RecordedClassLoader>(type) {
42             @Override
43             RecordedClassLoader createTyped(List<ValueDescriptor> desc, long id, Object[] object) {
44                 return new RecordedClassLoader(desc, id, object, timeConverter);
45             }
46         };
47     }
48 
49     private final long uniqueId;
50 
51     // package private
52     private RecordedClassLoader(List<ValueDescriptor> descriptors, long id, Object[] values, TimeConverter timeConverter) {
53         super(descriptors, values, timeConverter);
54         this.uniqueId = id;
55     }
56 
57     /**
58      * Returns the class of the class loader.
59      * <P>
60      * If the bootstrap class loader is represented as {@code null} in the Java
61      * Virtual Machine (JVM), then {@code null} is also the return value of this
62      * method.
63      *
64      * @return class of the class loader, can be {@code null}
65      */
66     public RecordedClass getType() {
67         return getTyped("type", RecordedClass.class, null);
68     }
69 
70     /**
71      * Returns the name of the class loader (for example, "boot", "platform", and
72      * "app").
73      *
74      * @return the class loader name, can be {@code null}
75      */
76     public String getName() {
77         return getTyped("name", String.class, null);
78     }
79 
80     /**
81      * Returns a unique ID for the class loader.
82      * <p>
83      * The ID might not be the same between Java Virtual Machine (JVM) instances.
84      *
85      * @return a unique ID
86      */
87     public long getId() {
88         return uniqueId;
89     }
90 }
