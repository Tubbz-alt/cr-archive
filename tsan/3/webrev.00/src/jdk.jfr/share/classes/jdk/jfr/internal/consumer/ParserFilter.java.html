26 package jdk.jfr.internal.consumer;
27 
28 import java.util.HashMap;
29 import java.util.Map;
30 import java.util.StringJoiner;
31 
32 final class ParserFilter {
33     public static final ParserFilter ACCEPT_ALL = new ParserFilter(true, Map.of());
34 
35     private final Map<String, Long> thresholds;
36     private final boolean acceptAll;
37 
38     public ParserFilter() {
39         this(false, new HashMap<>());
40     }
41 
42     private ParserFilter(boolean acceptAll, Map<String, Long> thresholds) {
43         this.acceptAll = acceptAll;
44         this.thresholds = thresholds;
45     }
46 
47     public void setThreshold(String eventName, long nanos) {
48         Long l = thresholds.get(eventName);
49         if (l != null) {
50             l = Math.min(l, nanos);
51         } else {
52             l = nanos;
53         }
54         thresholds.put(eventName, l);
55     }
56 
57     public long getThreshold(String eventName) {
58         if (acceptAll) {
59             return 0L;
60         }
61         Long l = thresholds.get(eventName);
62         if (l != null) {
63             return l;
64         }
65         return -1;
66     }
67 
68     public String toString() {
69         if (acceptAll) {
70             return "ACCEPT ALL";
71         }
72 
73         StringJoiner sb = new StringJoiner(", ");
74         for (String key : thresholds.keySet().toArray(new String[0])) {
75             Long value = thresholds.get(key);
76             sb.add(key + " = " + value);
77         }
78         return sb.toString();
79     }
80 }
