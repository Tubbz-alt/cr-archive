 41 //Instantiated by native
 42 final class DCmdConfigure extends AbstractDCmd {
 43     /**
 44      * Execute JFR.configure.
 45      *
 46      * @param repositoryPath the path
 47      * @param dumpPath path to dump to on fatal error (oom)
 48      * @param stackDepth depth of stack traces
 49      * @param globalBufferCount number of global buffers
 50      * @param globalBufferSize size of global buffers
 51      * @param threadBufferSize size of thread buffer for events
 52      * @param maxChunkSize threshold at which a new chunk is created in the disk repository
 53      * @param sampleThreads if thread sampling should be enabled

 70             Boolean sampleThreads
 71 
 72     ) throws DCmdException {
 73         if (Logger.shouldLog(LogTag.JFR_DCMD, LogLevel.DEBUG)) {
 74             Logger.log(LogTag.JFR_DCMD, LogLevel.DEBUG, "Executing DCmdConfigure: repositorypath=" + repositoryPath +
 75                     ", dumppath=" + dumpPath +
 76                     ", stackdepth=" + stackDepth +
 77                     ", globalbuffercount=" + globalBufferCount +
 78                     ", globalbuffersize=" + globalBufferSize +
 79                     ", thread_buffer_size" + threadBufferSize +
 80                     ", memorysize" + memorySize +
 81                     ", maxchunksize=" + maxChunkSize +
 82                     ", samplethreads" + sampleThreads);
 83         }
 84 
 85 
 86         boolean updated = false;
 87         if (repositoryPath != null) {
 88             try {
 89                 SafePath s = new SafePath(repositoryPath);
 90                 Repository.getRepository().setBasePath(s);




 91                 Logger.log(LogTag.JFR, LogLevel.INFO, "Base repository path set to " + repositoryPath);
 92             } catch (Exception e) {
 93                 throw new DCmdException("Could not use " + repositoryPath + " as repository. " + e.getMessage(), e);
 94             }
 95             printRepositoryPath();
 96             updated = true;
 97         }
 98 
 99         if (dumpPath != null)  {
100             Options.setDumpPath(new SafePath(dumpPath));
101             Logger.log(LogTag.JFR, LogLevel.INFO, "Emergency dump path set to " + dumpPath);
102             printDumpPath();
103             updated = true;
104         }
105 
106         if (stackDepth != null)  {
107             Options.setStackDepth(stackDepth);
108             Logger.log(LogTag.JFR, LogLevel.INFO, "Stack depth set to " + stackDepth);
109             printStackDepth();
110             updated = true;

 40 /**
 41  * JFR.configure - invoked from native
 42  *
 43  */
 44 //Instantiated by native
 45 final class DCmdConfigure extends AbstractDCmd {
 46     /**
 47      * Execute JFR.configure.
 48      *
 49      * @param repositoryPath the path
 50      * @param dumpPath path to dump to on fatal error (oom)
 51      * @param stackDepth depth of stack traces
 52      * @param globalBufferCount number of global buffers
 53      * @param globalBufferSize size of global buffers
 54      * @param threadBufferSize size of thread buffer for events
 55      * @param maxChunkSize threshold at which a new chunk is created in the disk repository
 56      * @param sampleThreads if thread sampling should be enabled

 73             Boolean sampleThreads
 74 
 75     ) throws DCmdException {
 76         if (Logger.shouldLog(LogTag.JFR_DCMD, LogLevel.DEBUG)) {
 77             Logger.log(LogTag.JFR_DCMD, LogLevel.DEBUG, "Executing DCmdConfigure: repositorypath=" + repositoryPath +
 78                     ", dumppath=" + dumpPath +
 79                     ", stackdepth=" + stackDepth +
 80                     ", globalbuffercount=" + globalBufferCount +
 81                     ", globalbuffersize=" + globalBufferSize +
 82                     ", thread_buffer_size" + threadBufferSize +
 83                     ", memorysize" + memorySize +
 84                     ", maxchunksize=" + maxChunkSize +
 85                     ", samplethreads" + sampleThreads);
 86         }
 87 
 88 
 89         boolean updated = false;
 90         if (repositoryPath != null) {
 91             try {
 92                 SafePath s = new SafePath(repositoryPath);
 93                 if (FlightRecorder.isInitialized()) {
 94                     PrivateAccess.getInstance().getPlatformRecorder().migrate(s);
 95                 } else {
 96                     Repository.getRepository().setBasePath(s);
 97                 }
 98                 Logger.log(LogTag.JFR, LogLevel.INFO, "Base repository path set to " + repositoryPath);
 99             } catch (Exception e) {
100                 throw new DCmdException("Could not use " + repositoryPath + " as repository. " + e.getMessage(), e);
101             }
102             printRepositoryPath();
103             updated = true;
104         }
105 
106         if (dumpPath != null)  {
107             Options.setDumpPath(new SafePath(dumpPath));
108             Logger.log(LogTag.JFR, LogLevel.INFO, "Emergency dump path set to " + dumpPath);
109             printDumpPath();
110             updated = true;
111         }
112 
113         if (stackDepth != null)  {
114             Options.setStackDepth(stackDepth);
115             Logger.log(LogTag.JFR, LogLevel.INFO, "Stack depth set to " + stackDepth);
116             printStackDepth();
117             updated = true;
