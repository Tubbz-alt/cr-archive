 26 package jdk.jfr.consumer;
 27 
 28 import java.lang.reflect.Modifier;

 29 
 30 import jdk.jfr.internal.consumer.ObjectContext;

 31 
 32 /**
 33  * A recorded frame in a stack trace.
 34  *
 35  * @since 9
 36  */
 37 public final class RecordedFrame extends RecordedObject {










 38     // package private
 39     RecordedFrame(ObjectContext objectContext, Object[] values) {
 40         super(objectContext, values);
 41     }
 42 
 43     /**
 44      * Returns {@code true} if this is a Java frame, {@code false} otherwise.
 45      * <p>
 46      * A Java method that has a native modifier is considered a Java frame.
 47      *
 48      * @return {@code true} if this is a Java frame, {@code false} otherwise
 49      *
 50      * @see Modifier#isNative(int)
 51      */
 52     public boolean isJavaFrame() {
 53         // Only Java frames exist today, but this allows
 54         // API to be extended for native frame in the future.
 55         if (hasField("javaFrame")) {
 56             return getTyped("javaFrame", Boolean.class, Boolean.TRUE);
 57         }
 58         return true;
 59     }
 60 
 61     /**
 62      * Returns the bytecode index for the execution point that is represented by
 63      * this recorded frame.
 64      *
 65      * @return byte code index, or {@code -1} if doesn't exist
 66      */
 67     public int getBytecodeIndex() {
 68         return getTyped("bytecodeIndex", Integer.class, Integer.valueOf(-1));
 69     }
 70 
 71     /**
 72      * Returns the line number for the execution point that is represented by this
 73      * recorded frame, or {@code -1} if doesn't exist
 74      *
 75      * @return the line number, or {@code -1} if doesn't exist
 76      */
 77     public int getLineNumber() {
 78         return getTyped("lineNumber", Integer.class, Integer.valueOf(-1));
 79     }
 80 
 81     /**
 82      * Returns the frame type for the execution point that is represented by this
 83      * recorded frame (for example, {@code "Interpreted"}, {@code "JIT compiled"} or
 84      * {@code "Inlined"}).
 85      *
 86      * @return the frame type, or {@code null} if doesn't exist
 87      */
 88     public String getType() {
 89         return getTyped("type", String.class, null);
 90     }
 91 
 92     /**
 93      * Returns the method for the execution point that is represented by this
 94      * recorded frame.
 95      *
 96      * @return the method, not {@code null}
 97      */
 98     public RecordedMethod getMethod() {
 99         return getTyped("method", RecordedMethod.class, null);
100     }
101 }
