 26 package jdk.jfr.consumer;
 27 
 28 import java.lang.reflect.Modifier;
 29 import java.util.List;
 30 
 31 import jdk.jfr.ValueDescriptor;
 32 import jdk.jfr.internal.Type;
 33 
 34 /**
 35  * A recorded Java type, such as a class or an interface.
 36  *
 37  * @since 9
 38  */
 39 public final class RecordedClass extends RecordedObject {
 40 
 41     static ObjectFactory<RecordedClass> createFactory(Type type, TimeConverter timeConverter) {
 42         return new ObjectFactory<RecordedClass>(type) {
 43             @Override
 44             RecordedClass createTyped(List<ValueDescriptor> desc, long id, Object[] object) {
 45                 return new RecordedClass(desc, id, object, timeConverter);
 46             }
 47         };
 48     }
 49 
 50     private final long uniqueId;
 51 
 52     // package private
 53     private RecordedClass(List<ValueDescriptor> descriptors, long id, Object[] values, TimeConverter timeConverter) {
 54         super(descriptors, values, timeConverter);
 55         this.uniqueId = id;
 56     }
 57 
 58     /**
 59      * Returns the modifiers of the class.
 60      * <p>
 61      * See {@link java.lang.reflect.Modifier}
 62      *
 63      * @return the modifiers
 64      *
 65      * @see Modifier
 66      */
 67     public int getModifiers() {
 68         return getTyped("modifiers", Integer.class, -1);
 69     }
 70 
 71     /**
 72      * Returns the class loader that defined the class.
 73      * <P>
 74      * If the bootstrap class loader is represented as {@code null} in the Java
 75      * Virtual Machine (JVM), then {@code null} is also the return value of this method.
 76      *
 77      * @return the class loader defining this class, can be {@code null}
 78      */
 79     public RecordedClassLoader getClassLoader() {
 80         return getTyped("classLoader", RecordedClassLoader.class, null);
 81     }
 82 
 83     /**
 84      * Returns the fully qualified name of the class (for example,
 85      * {@code "java.lang.String"}).
 86      *
 87      * @return the class name, not {@code null}
 88      */
 89     public String getName() {
 90         return getTyped("name", String.class, null).replace("/", ".");
 91     }
 92 
 93     /**
 94      * Returns a unique ID for the class.
 95      * <p>
 96      * The ID might not be the same between Java Virtual Machine (JVM) instances.
 97      *
 98      * @return a unique ID
 99      */
100     public long getId() {
101         return uniqueId;
102     }
103 }
