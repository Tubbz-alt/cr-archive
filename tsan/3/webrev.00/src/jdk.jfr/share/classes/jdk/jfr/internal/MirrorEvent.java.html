26 package jdk.jfr.internal;
27 
28 import java.lang.annotation.Retention;
29 import java.lang.annotation.Target;
30 import java.lang.annotation.RetentionPolicy;
31 import java.lang.annotation.ElementType;
32 
33 @Retention(RetentionPolicy.RUNTIME)
34 @Target({ ElementType.TYPE })
35 public @interface MirrorEvent {
36     /**
37      * Fully qualified name of the class to mirror metadata for (for example,
38      * {@code "jdk.internal.event.Example"})
39      *
40      * @return the fully qualified class name of the event
41      */
42     String className();
43 
44     /**
45      * The module where the event is located, by default {@code "java.base"}.
46      *
47      * @return the module name
48      */
49     String module() default "java.base";
50 }
