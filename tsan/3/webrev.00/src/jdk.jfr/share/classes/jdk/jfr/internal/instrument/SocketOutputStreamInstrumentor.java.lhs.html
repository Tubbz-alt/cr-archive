26 package jdk.jfr.internal.instrument;
27 
28 import java.io.IOException;
29 import java.net.InetAddress;

30 
31 import jdk.jfr.events.SocketWriteEvent;
32 
33 /**
34  * See {@link JITracer} for an explanation of this code.
35  */
36 @JIInstrumentationTarget("java.net.SocketOutputStream")
37 @JITypeMapping(from = "jdk.jfr.internal.instrument.SocketOutputStreamInstrumentor$AbstractPlainSocketImpl",
38             to = "java.net.AbstractPlainSocketImpl")
39 final class SocketOutputStreamInstrumentor {
40 
41     private SocketOutputStreamInstrumentor() {
42     }
43 
44     @SuppressWarnings("deprecation")
45     @JIInstrumentationMethod
46     private void socketWrite(byte b[], int off, int len) throws IOException {
47         SocketWriteEvent event = SocketWriteEvent.EVENT.get();
48         if (!event.isEnabled()) {
49             socketWrite(b, off, len);
50             return;
51         }
52         int bytesWritten = 0;
53         try {
54             event.begin();
55             socketWrite(b, off, len);
56             bytesWritten = len;
57         } finally {
58             event.end() ;
59             if (event.shouldCommit()) {
60                 String hostString  = impl.address.toString();
61                 int delimiterIndex = hostString.lastIndexOf('/');
62 
63                 event.host         = hostString.substring(0, delimiterIndex);
64                 event.address      = hostString.substring(delimiterIndex + 1);
65                 event.port         = impl.port;
66                 event.bytesWritten = bytesWritten < 0 ? 0 : bytesWritten;
67 
68                 event.commit();
69                 event.reset();
70             }
71         }
72     }
73 
74     private AbstractPlainSocketImpl impl = null;
75 
76     void silenceFindBugsUnwrittenField(InetAddress dummy) {
77         impl.address = dummy;
78     }
79 
80     static class AbstractPlainSocketImpl {
81         InetAddress address;
82         int port;
83     }
84 }
