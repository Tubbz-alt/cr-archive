26 package jdk.jfr.consumer;
27 
28 import java.time.DateTimeException;
29 import java.time.ZoneOffset;
30 
31 import jdk.jfr.internal.LogLevel;
32 import jdk.jfr.internal.LogTag;
33 import jdk.jfr.internal.Logger;
34 import jdk.jfr.internal.consumer.ChunkHeader;
35 
36 /**
37  * Converts ticks to nanoseconds
38  */
39 final class TimeConverter {
40     private final long startTicks;
41     private final long startNanos;
42     private final double divisor;
43     private final ZoneOffset zoneOffet;
44 
45     TimeConverter(ChunkHeader chunkHeader, int rawOffset) {
46         this.startTicks = chunkHeader.getStartTicks();
47         this.startNanos = chunkHeader.getStartNanos();
48         this.divisor = chunkHeader.getTicksPerSecond() / 1000_000_000L;
49         this.zoneOffet = zoneOfSet(rawOffset);
50     }
51 
52     private ZoneOffset zoneOfSet(int rawOffset) {
53         try {
54             return ZoneOffset.ofTotalSeconds(rawOffset / 1000);
55         } catch (DateTimeException dte) {
56             Logger.log(LogTag.JFR_SYSTEM_PARSER, LogLevel.INFO, "Could not create ZoneOffset from raw offset " + rawOffset);
57         }
58         return ZoneOffset.UTC;
59     }
60 
61     public long convertTimestamp(long ticks) {
62         return startNanos + (long) ((ticks - startTicks) / divisor);
63     }
64 
65     public long convertTimespan(long ticks) {
66         return (long) (ticks / divisor);
67     }
68 
69     public ZoneOffset getZoneOffset() {
70         return zoneOffet;
71     }
72 }
