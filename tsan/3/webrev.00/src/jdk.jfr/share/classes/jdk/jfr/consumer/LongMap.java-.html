26 package jdk.jfr.consumer;
27 
28 import java.util.HashMap;
29 import java.util.Iterator;
30 
31 /**
32  * Commonly used data structure for looking up objects given an id (long value)
33  *
34  * TODO: Implement without using Map and Long objects, to minimize allocation
35  *
36  * @param <T>
37  */
38 final class LongMap<T> implements Iterable<T> {
39     private final HashMap<Long, T> map;
40 
41     LongMap() {
42         map = new HashMap<>(101);
43     }
44 
45     void put(long id, T object) {
46         map.put(id, object);
47     }
48 
49     T get(long id) {
50         return map.get(id);
51     }
52 
53     @Override
54     public Iterator<T> iterator() {
55         return map.values().iterator();
56     }
57 
58     Iterator<Long> keys() {
59         return map.keySet().iterator();
60     }
61 }
