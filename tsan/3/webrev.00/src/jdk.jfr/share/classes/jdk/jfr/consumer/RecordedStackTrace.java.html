26 package jdk.jfr.consumer;
27 
28 import java.util.Arrays;
29 import java.util.Collections;
30 import java.util.List;
31 
32 import jdk.jfr.internal.consumer.ObjectContext;
33 
34 /**
35  * A recorded stack trace.
36  *
37  * @since 9
38  */
39 public final class RecordedStackTrace extends RecordedObject {
40     // package private
41     RecordedStackTrace(ObjectContext objectContext, Object[] values) {
42         super(objectContext, values);
43     }
44 
45     /**
46      * Returns the frames in the stack trace.
47      *
48      * @return a list of Java stack frames, not {@code null}
49      */
50     @SuppressWarnings("unchecked")
51     public List<RecordedFrame> getFrames() {
52         Object[] array = getTyped("frames", Object[].class, null);
53         if (array == null) {
54             return Collections.EMPTY_LIST;
55         }
56         List<?> list = Arrays.asList(array);
57         return (List<RecordedFrame>) list;
58     }
59 
60     /**
61      * Returns {@code true} if the stack trace is truncated due to its size,
62      * {@code false} otherwise.
63      *
64      * @return {@code true} if the stack trace is truncated, {@code false}
65      *         otherwise
66      */
67     public boolean isTruncated() {
68         return getTyped("truncated", Boolean.class, true);
69     }
70 }
