 63     protected InterfaceTypeImpl(VirtualMachine aVm,long aRef) {
 64         super(aVm, aRef);
 65     }

 67     public List<InterfaceType> superinterfaces() {
 68         List<InterfaceType> superinterfaces = (superinterfacesRef == null) ? null :
 69                                      superinterfacesRef.get();
 70         if (superinterfaces == null) {
 71             superinterfaces = getInterfaces();
 72             superinterfaces = Collections.unmodifiableList(superinterfaces);
 73             superinterfacesRef = new SoftReference<List<InterfaceType>>(superinterfaces);
 74         }
 75         return superinterfaces;
 76     }
 77 
 78     public List<InterfaceType> subinterfaces() {
 79         List<InterfaceType> subs = new ArrayList<InterfaceType>();
 80         for (ReferenceType refType : vm.allClasses()) {
 81             if (refType instanceof InterfaceType) {
 82                 InterfaceType interfaze = (InterfaceType)refType;
 83                 if (interfaze.isPrepared() && interfaze.superinterfaces().contains(this)) {
 84                     subs.add(interfaze);
 85                 }
 86             }
 87         }
 88         return subs;
 89     }

 91     public List<ClassType> implementors() {
 92         List<ClassType> implementors = new ArrayList<ClassType>();
 93         for (ReferenceType refType : vm.allClasses()) {
 94             if (refType instanceof ClassType) {
 95                 ClassType clazz = (ClassType)refType;
 96                 if (clazz.isPrepared() && clazz.interfaces().contains(this)) {
 97                     implementors.add(clazz);
 98                 }
 99             }
100         }
101         return implementors;
102     }
