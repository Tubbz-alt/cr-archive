 26 package com.sun.tools.jdi;
 27 
 28 import java.lang.ref.SoftReference;
 29 import java.util.ArrayList;
 30 import java.util.Collections;
 31 import java.util.List;
 32 
 33 import com.sun.jdi.ClassType;
 34 import com.sun.jdi.InterfaceType;
 35 import com.sun.jdi.Method;
 36 import com.sun.jdi.ReferenceType;
 37 import com.sun.jdi.VirtualMachine;
 38 
 39 final public class InterfaceTypeImpl extends InvokableTypeImpl
 40                                      implements InterfaceType {
 41 
 42     private static class IResult implements InvocationResult {
 43         final private JDWP.InterfaceType.InvokeMethod rslt;
 44 
 45         public IResult(JDWP.InterfaceType.InvokeMethod rslt) {
 46             this.rslt = rslt;
 47         }
 48 
 49         @Override
 50         public ObjectReferenceImpl getException() {
 51             return rslt.exception;
 52         }
 53 
 54         @Override
 55         public ValueImpl getResult() {
 56             return rslt.returnValue;
 57         }
 58 
 59     }
 60 
 61     private SoftReference<List<InterfaceType>> superinterfacesRef = null;
 62 
 63     protected InterfaceTypeImpl(VirtualMachine aVm,long aRef) {
 64         super(aVm, aRef);
 65     }
 66 
 67     public List<InterfaceType> superinterfaces() {
 68         List<InterfaceType> superinterfaces = (superinterfacesRef == null) ? null :
 69                                      superinterfacesRef.get();
 70         if (superinterfaces == null) {
 71             superinterfaces = getInterfaces();
 72             superinterfaces = Collections.unmodifiableList(superinterfaces);
 73             superinterfacesRef = new SoftReference<List<InterfaceType>>(superinterfaces);
 74         }
 75         return superinterfaces;
 76     }
 77 
 78     public List<InterfaceType> subinterfaces() {
 79         List<InterfaceType> subs = new ArrayList<InterfaceType>();
 80         vm.forEachClass(refType -> {
 81             if (refType instanceof InterfaceType) {
 82                 InterfaceType interfaze = (InterfaceType)refType;
 83                 if (interfaze.isPrepared() && interfaze.superinterfaces().contains(this)) {
 84                     subs.add(interfaze);
 85                 }
 86             }
 87         });
 88         return subs;
 89     }
 90 
 91     public List<ClassType> implementors() {
 92         List<ClassType> implementors = new ArrayList<ClassType>();
 93         vm.forEachClass(refType -> {
 94             if (refType instanceof ClassType) {
 95                 ClassType clazz = (ClassType)refType;
 96                 if (clazz.isPrepared() && clazz.interfaces().contains(this)) {
 97                     implementors.add(clazz);
 98                 }
 99             }
100         });
101         return implementors;
102     }
103 
104     public boolean isInitialized() {
105         return isPrepared();
106     }
107 
108     public String toString() {
109        return "interface " + name() + " (" + loaderString() + ")";
110     }
111 
112     @Override
113     InvocationResult waitForReply(PacketStream stream) throws JDWPException {
114         return new IResult(JDWP.InterfaceType.InvokeMethod.waitForReply(vm, stream));
115     }
116 
117     @Override
118     CommandSender getInvokeMethodSender(final ThreadReferenceImpl thread,
119                                         final MethodImpl method,
120                                         final ValueImpl[] args,
121                                         final int options) {
122         return () ->
123             JDWP.InterfaceType.InvokeMethod.enqueueCommand(vm,
124                                                            InterfaceTypeImpl.this,
125                                                            thread,
126                                                            method.ref(),
127                                                            args,
128                                                            options);
129     }
130 
131     @Override
132     ClassType superclass() {
133         return null;
134     }
135 
136     @Override
137     boolean isAssignableTo(ReferenceType type) {
138         if (type.name().equals("java.lang.Object")) {
139             // interfaces are always assignable to j.l.Object
140             return true;
141         }
142         return super.isAssignableTo(type);
143     }
144 
145     @Override
146     List<InterfaceType> interfaces() {
147         return superinterfaces();
148     }
149 
150     @Override
151     boolean canInvoke(Method method) {
152         // method must be directly in this interface
153         return this.equals(method.declaringType());
154     }
155 }
