26 package com.sun.jdi.connect;
27 
28 import java.util.List;
29 import java.util.ArrayList;
30 import java.util.Collections;
31 
32 /**
33  * Thrown to indicate an invalid argument or
34  * inconsistent passed to a {@link Connector}.
35  *
36  * @author Gordon Hirsch
37  * @since  1.3
38  */
39 public class IllegalConnectorArgumentsException extends Exception {
40 
41     private static final long serialVersionUID = -3042212603611350941L;
42 
43     @SuppressWarnings("serial") // Conditionally serializable
44     List<String> names;
45 
46     /**
47      * Construct an <code>IllegalConnectorArgumentsException</code>
48      * with the specified detail message and the name of the argument
49      * which is invalid or inconsistent.
50      * @param s the detailed message.
51      * @param name the name of the invalid or inconsistent argument.
52      */
53     public IllegalConnectorArgumentsException(String s, String name) {
54         super(s);
55         names = new ArrayList<String>(1);
56         names.add(name);
57     }
58 
59     /**
60      * Construct an <code>IllegalConnectorArgumentsException</code>
61      * with the specified detail message and a <code>List</code> of
62      * names of arguments which are invalid or inconsistent.
63      * @param s the detailed message.
64      * @param names a <code>List</code> containing the names of the
65      * invalid or inconsistent argument.
66      */
67     public IllegalConnectorArgumentsException(String s, List<String> names) {
68         super(s);
69         this.names = new ArrayList<String>(names);
70     }
71 
72     /**
73      * Return a <code>List</code> containing the names of the
74      * invalid or inconsistent arguments.
75      * @return a <code>List</code> of argument names.
76      */
77     public List<String> argumentNames() {
78         return Collections.unmodifiableList(names);
79     }
80 }
