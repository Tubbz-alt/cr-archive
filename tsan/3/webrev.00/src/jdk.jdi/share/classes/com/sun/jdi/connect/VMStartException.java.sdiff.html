28 /**
29  * A target VM was successfully launched, but terminated with an
30  * error before a connection could be established. This exception
31  * provides the {@link java.lang.Process} object for the launched
32  * target to help in diagnosing the problem.
33  *
34  * @author Gordon Hirsch
35  * @since  1.3
36  */
37 public class VMStartException extends Exception {
38 
39     private static final long serialVersionUID = 6408644824640801020L;
40 

41     Process process;
42 
43     public VMStartException(Process process) {
44         super();
45         this.process = process;
46     }
47 
48     public VMStartException(String message, Process process) {
49         super(message);
50         this.process = process;
51     }
52 
53     public Process process() {
54         return process;
55     }
56 }
