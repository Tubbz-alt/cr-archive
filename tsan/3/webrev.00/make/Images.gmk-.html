204 ################################################################################
205 # Debug symbols
206 # Since debug symbols are not included in the jmod files, they need to be copied
207 # in manually after generating the images.
208 
209 ALL_JDK_MODULES := $(JDK_MODULES)
210 ALL_JRE_MODULES := $(sort $(JRE_MODULES), $(foreach m, $(JRE_MODULES), \
211     $(call FindTransitiveDepsForModule, $m)))
212 
213 ifeq ($(call isTargetOs, windows), true)
214   LIBS_TARGET_SUBDIR := bin
215 else
216   LIBS_TARGET_SUBDIR := lib
217 endif
218 
219 # Param 1 - dir to find debuginfo files in
220 FindDebuginfoFiles = \
221     $(wildcard $(addprefix $1/*, $(DEBUGINFO_SUFFIXES)) \
222         $(addprefix $1/*/*, $(DEBUGINFO_SUFFIXES)) \
223         $(addprefix $1/*/*/*, $(DEBUGINFO_SUFFIXES)))
224 
225 # Pick the correct debug info files to copy, either zipped or not.
226 ifeq ($(ZIP_EXTERNAL_DEBUG_SYMBOLS), true)
227   DEBUGINFO_SUFFIXES += .diz
228 else
229   DEBUGINFO_SUFFIXES := .debuginfo .pdb .map
230   # On Macosx, if debug symbols have not been zipped, find all files inside *.dSYM
231   # dirs.
232   ifeq ($(call isTargetOs, macosx), true)
233     $(eval $(call FillCacheFind, \
234         $(SUPPORT_OUTPUTDIR)/modules_cmds $(SUPPORT_OUTPUTDIR)/modules_libs))
235     FindDebuginfoFiles = \
236         $(if $(wildcard $1), $(call containing, .dSYM/, $(call CacheFind, $1)))
237   endif
238 endif
239 
240 # Param 1 - either JDK or JRE
241 SetupCopyDebuginfo = \
242     $(foreach m, $(ALL_$1_MODULES), \
243       $(eval $(call SetupCopyFiles, COPY_$1_LIBS_DEBUGINFO_$m, \
244           SRC := $(SUPPORT_OUTPUTDIR)/modules_libs/$m, \
245           DEST := $($1_IMAGE_DIR)/$(LIBS_TARGET_SUBDIR), \
246           FILES := $(call FindDebuginfoFiles, \
247               $(SUPPORT_OUTPUTDIR)/modules_libs/$m), \
248       )) \
249       $(eval $1_TARGETS += $$(COPY_$1_LIBS_DEBUGINFO_$m)) \
250     )
251 
252 # No space before argument to avoid having to put $(strip ) everywhere in
253 # implementation above.
254 $(call SetupCopyDebuginfo,JDK)
255 $(call SetupCopyDebuginfo,JRE)
