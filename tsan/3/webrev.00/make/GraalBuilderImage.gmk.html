26 # This makefile creates a jdk image overlayed with statically linked core
27 # libraries.
28 
29 default: all
30 
31 include $(SPEC)
32 include MakeBase.gmk
33 
34 ################################################################################
35 
36 TARGETS :=
37 
38 $(eval $(call SetupCopyFiles, COPY_JDK_IMG, \
39       SRC := $(JDK_IMAGE_DIR)/, \
40       DEST := $(GRAAL_BUILDER_IMAGE_DIR)/, \
41       FILES := $(call FindFiles, $(JDK_IMAGE_DIR)/), \
42 ))
43 TARGETS += $(COPY_JDK_IMG)
44 
45 $(eval $(call SetupCopyFiles, COPY_STATIC_LIBS, \
46       SRC := $(STATIC_LIBS_IMAGE_DIR)/lib, \
47       DEST := $(GRAAL_BUILDER_IMAGE_DIR)/lib, \
48       FILES := $(filter %$(STATIC_LIBRARY_SUFFIX), \
49           $(call FindFiles, $(STATIC_LIBS_IMAGE_DIR)/lib)), \
50 ))
51 TARGETS += $(COPY_STATIC_LIBS)
52 
53 ################################################################################
54 
55 all: $(TARGETS)
56 
57 .PHONY: all
