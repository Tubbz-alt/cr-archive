38 if [ "$OPENJDK_TARGET_OS" = "linux" ]; then
39   ACCEPTED_BIN_DIFF="
40       ./lib/server/libjvm.so
41       ./hotspot/gtest/server/libjvm.so
42       "
43   STRIP_BEFORE_COMPARE="
44       ./hotspot/gtest/server/libjvm.so
45       "
46 elif [ "$OPENJDK_TARGET_OS" = "solaris" ]; then
47   SKIP_BIN_DIFF="true"
48   SKIP_FULLDUMP_DIFF="true"
49   MAX_KNOWN_DIS_DIFF_SIZE="3000"
50   SORT_SYMBOLS="
51       ./lib/libfontmanager.so
52       ./lib/libjimage.so
53       ./lib/server/libjvm.so
54       ./hotspot/gtest/server/libjvm.so
55       "

57       ./lib/libfontmanager.so
58       ./lib/libsaproc.so
59   "
60   STRIP_TESTS_BEFORE_COMPARE="true"

62   SKIP_BIN_DIFF="true"
63   SKIP_FULLDUMP_DIFF="true"

65       /modules_libs/java.security.jgss/w2k_lsa_auth.pdb

72       ./lib/libosxapp.dylib
73       ./lib/libosxui.dylib
74       ./lib/server/libjvm.dylib
75       ./hotspot/gtest/server/libjvm.dylib
76       "
77   STRIP_TESTS_BEFORE_COMPARE="true"
78 fi
