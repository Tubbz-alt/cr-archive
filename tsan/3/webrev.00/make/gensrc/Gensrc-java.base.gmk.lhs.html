 40 CLDR_DATA_DIR := $(TOPDIR)/make/data/cldr/common
 41 GENSRC_DIR := $(SUPPORT_OUTPUTDIR)/gensrc/java.base
 42 CLDR_GEN_DONE := $(GENSRC_DIR)/_cldr-gensrc.marker
 43 TZ_DATA_DIR := $(TOPDIR)/make/data/tzdata
 44 ZONENAME_TEMPLATE := $(TOPDIR)/src/java.base/share/classes/java/time/format/ZoneName.java.template
 45 
 46 $(CLDR_GEN_DONE): $(wildcard $(CLDR_DATA_DIR)/dtd/*.dtd) \
 47     $(wildcard $(CLDR_DATA_DIR)/main/en*.xml) \
 48     $(wildcard $(CLDR_DATA_DIR)/supplemental/*.xml) \
 49     $(wildcard $(TZ_DATA_DIR)/*) \
 50     $(ZONENAME_TEMPLATE) \
 51     $(BUILD_TOOLS_JDK)
 52 	$(call MakeDir, $(GENSRC_DIR))
 53 	$(call LogInfo, Processing CLDR data for java.base)
 54 	$(call ExecuteWithLog, $@, \
 55 	    $(TOOL_CLDRCONVERTER) -base $(CLDR_DATA_DIR) \
 56 	    -baselocales "en-US" \
 57 	    -o $(GENSRC_DIR) \
 58 	    -basemodule \
 59 	    -zntempfile $(ZONENAME_TEMPLATE) \
 60 	    -tzdatadir $(TZ_DATA_DIR))
 61 	$(TOUCH) $@
 62 
 63 GENSRC_JAVA_BASE += $(CLDR_GEN_DONE)
