103 $(GENSRC_DIR)/_gensrc_proc_done: $(PROC_SRCS) $(PROCESSOR_JARS)
104 	$(call MakeDir, $(@D))
105 	$(eval $(call ListPathsSafely,PROC_SRCS,$(@D)/_gensrc_proc_files))
106 	$(JAVA) $(NEW_JAVAC) \
107 	    -XDignore.symbol.file \
108 	    --upgrade-module-path $(JDK_OUTPUTDIR)/modules --system none \
109 	    $(ADD_EXPORTS) \
110 	    -sourcepath $(SOURCEPATH) \
111 	    -implicit:none \
112 	    -proc:only \
113 	    -processorpath $(PROCESSOR_PATH) \
114 	    -d $(GENSRC_DIR) \
115 	    -s $(GENSRC_DIR) \
116 	    @$(@D)/_gensrc_proc_files
117 	$(TOUCH) $@
118 
119 TARGETS += $(GENSRC_DIR)/_gensrc_proc_done

123 $(GENSRC_DIR)/module-info.java.extra: $(GENSRC_DIR)/_gensrc_proc_done
124 	$(ECHO) "" > $@;
125 	($(CD) $(GENSRC_DIR)/META-INF/providers && \
126 	    p=""; \
127 	    impl=""; \
128 	    for i in $$($(GREP) '^' * | $(SORT) -t ':' -k 2 | $(SED) 's/:.*//'); do \
129 	      c=$$($(CAT) $$i | $(TR) -d '\n\r'); \
130 	      if test x$$p != x$$c; then \
131                 if test x$$p != x; then \
132 	          $(ECHO) "    ;" >> $@; \
133 	        fi; \
134 	        $(ECHO) "provides $$c with" >> $@; \
135                 p=$$c; \
136 	        impl=""; \
137 	      fi; \
138               if test x$$impl != x; then \
139 	        $(ECHO) "  , $$i" >> $@; \
140               else \
141 	        $(ECHO) "    $$i" >> $@; \
142               fi; \
143               impl=$$i; \
144 	    done); \
145             $(ECHO) "    ;" >> $@; \
146 	$(ECHO) "uses org.graalvm.compiler.options.OptionDescriptors;" >> $@; \
147 	$(ECHO) "provides org.graalvm.compiler.options.OptionDescriptors with" >> $@; \
148         impl=""; \
149 	for i in $$($(FIND) $(GENSRC_DIR) -name '*_OptionDescriptors.java' | $(SORT)); do \
150 	    c=$$($(ECHO) $$i | $(SED) 's:.*/jdk\.internal\.vm\.compiler/\(.*\)\.java:\1:' | $(TR) '/' '.'); \
151             if test x$$impl != x; then \
152 	      $(ECHO) "  , $$c" >> $@; \
153             else \
154 	      $(ECHO) "    $$c" >> $@; \
155             fi; \
156             impl=$$c; \
157 	done; \
158 	$(ECHO) "    ;" >> $@;
