 42 # Setup make rules for defining a Java compiler, which is needed to compile
 43 # Java code. This rule generates no output.
 44 #
 45 # Parameter 1 is the name of the compiler definition. This name needs to be
 46 # passed to SetupJavaCompilation. This name is used as variable prefix.
 47 #
 48 # Remaining parameters are named arguments. These include:
 49 #   JVM:=The jvm used to run the javac command
 50 #   JAVAC:=The javac jar and bootstrap classpath changes, or just bin/javac if JVM is left out
 51 #   FLAGS:=Flags to be supplied to javac
 52 #   SERVER_DIR:=Use a javac server (-XDserver) and store the server related files here
 53 #   SERVER_JVM:=Use this JVM for the server. Defaults to the JVM above.
 54 #   DISABLE_SJAVAC:=Set to true if this setup does not support sjavac
 55 SetupJavaCompiler = $(NamedParamsMacroTemplate)
 56 define SetupJavaCompilerBody
 57   # The port file contains the tcp/ip on which the server listens
 58   # and the cookie necessary to talk to the server.
 59   $1_SJAVAC_PORTFILE:=$$($1_SERVER_DIR)/server.port
 60   # You can use a different JVM to run the background javac server.
 61   ifeq ($$($1_SERVER_JVM),)
 62     # It defaults to the same JVM that is used to start the javac command.
 63     $1_SERVER_JVM:=$$($1_JVM)
 64   endif
 65 endef

 67 define add_file_to_copy
 68   # param 1 = BUILD_MYPACKAGE
 69   # parma 2 = The source file to copy.
 70   $2_TARGET:=$2
 71   # Remove the source prefix.
 72   $$(foreach i,$$($1_SRC),$$(eval $$(call remove_string,$$i,$2_TARGET)))
 73   # To allow for automatic overrides, do not create a rule for a target file that
 74   # already has one
 75   ifneq ($$($1_COPY_$$($2_TARGET)), 1)
 76     $1_COPY_$$($2_TARGET) := 1
 77     # Now we can setup the dependency that will trigger the copying.
 78     $$($1_BIN)$$($1_MODULE_SUBDIR)$$($2_TARGET) : $2
 79 	$$(call LogInfo, Copying $$(patsubst $(OUTPUTDIR)/%,%, $$@))
 80 	$$(install-file)
 81 	$(CHMOD) -f ug+w $$@
 82 
 83     # And do not forget this target
 84     $1_ALL_COPY_TARGETS += $$($1_BIN)$$($1_MODULE_SUBDIR)$$($2_TARGET)
 85   endif
 86 endef

 88 # This macro is used only for properties files that are to be
 89 # copied over to the classes directory in cleaned form:
 90 # Previously this was inconsistently done in different repositories.
 91 # This is the new clean standard. Though it is to be superseded by
 92 # a standard annotation processor from with sjavac.
 93 #
 94 # An empty echo ensures that the input to sed always ends with a newline.
 95 # Certain implementations (e.g. Solaris) will skip the last line without
 96 # it.
 97 #
 98 # The sed expression does this:
 99 # 1. Add a backslash before any :, = or ! that do not have a backslash already.
100 # 2. Apply the file unicode2x.sed which does a whole bunch of \u00XX to \xXX
101 #    conversions.
102 # 3. Delete all lines starting with #.
103 # 4. Delete empty lines.
104 # 5. Append lines ending with \ with the next line.
105 # 6. Remove leading and trailing white space. Note that tabs must be explicit
106 #    as sed on macosx does not understand '\t'.
107 # 7. Replace the first \= with just =.
108 # 8. Finally it's all sorted to create a stable output.
109 #
110 # It is assumed that = is the character used for separating names and values.
111 define add_file_to_clean
112   # param 1 = BUILD_MYPACKAGE
113   # parma 2 = The source file to copy and clean.
114   $2_TARGET:=$2
115   # Remove the source prefix.
116   $$(foreach i,$$($1_SRC),$$(eval $$(call remove_string,$$i,$2_TARGET)))
117   # Now we can setup the depency that will trigger the copying.
118   # To allow for automatic overrides, do not create a rule for a target file that
119   # already has one
120   ifneq ($$($1_CLEAN_$$($2_TARGET)), 1)
121     $1_CLEAN_$$($2_TARGET) := 1
122     $$($1_BIN)$$($1_MODULE_SUBDIR)$$($2_TARGET) : $2
123 	$$(call LogInfo, Cleaning $$(patsubst $(OUTPUTDIR)/%,%, $$@))
124 	$$(call MakeTargetDir)
125 	export LC_ALL=C ; ( $(CAT) $$< && $(ECHO) "" ) \
126 	    | $(SED) -e 's/\([^\\]\):/\1\\:/g' -e 's/\([^\\]\)=/\1\\=/g' \
127 	        -e 's/\([^\\]\)!/\1\\!/g' -e 's/^[ 	]*#.*/#/g' \
128 	    | $(SED) -f "$(TOPDIR)/make/common/support/unicode2x.sed" \
129 	    | $(SED) -e '/^#/d' -e '/^$$$$/d' \
130 	        -e :a -e '/\\$$$$/N; s/\\\n//; ta' \
131 	        -e 's/^[ 	]*//;s/[ 	]*$$$$//' \
132 	        -e 's/\\=/=/' \
133 	    | $(SORT) > $$@
134 	$(CHMOD) -f ug+w $$@
135 
136     # And do not forget this target
137     $1_ALL_COPY_CLEAN_TARGETS += $$($1_BIN)$$($1_MODULE_SUBDIR)$$($2_TARGET)
138   endif
139 endef
140 
141 define remove_string
142   $2 := $$(subst $1,,$$($2))
143 endef

145 # Setup make rules for compiling Java source code to class files and/or a
146 # resulting jar file.
147 #
148 # Parameter 1 is the name of the rule. This name is used as variable prefix,
149 # and the targets generated are listed in a variable by that name.
150 #
151 # The target for public API digest is returned in $1_API_TARGET.
152 #
153 # Remaining parameters are named arguments. These include:
154 #   SETUP:=must point to a previously setup java compiler, for example: SETUP:=BOOTJAVAC
155 #   JVM:=path to ..bin/java
156 #   ADD_JAVAC_FLAGS:=javac flags to append to the default ones.
157 #   SRC:=one or more directories to search for sources. The order of the source roots
158 #        is significant. The first found file of a certain name has priority.
159 #   BIN:=store classes here
160 #   MODULE:=Name of module being compiled. If set, classes are put in BIN/MODULE.
161 #   CLASSPATH:=a list of additional entries to set as classpath to javac
162 #   INCLUDES:=myapp.foo means will only compile java files in myapp.foo or any of its sub-packages.
163 #   EXCLUDES:=myapp.foo means will do not compile java files in myapp.foo or any of its sub-packages.
164 #   COPY:=.prp means copy all prp files to the corresponding package in BIN.
165 #   COPY_FILES:=myapp/foo/setting.txt means copy this file over to the package myapp/foo
166 #   CLEAN:=.properties means copy and clean all properties file to the corresponding package in BIN.
167 #   CLEAN_FILES:=myapp/foo/setting.txt means clean this file over to the package myapp/foo
168 #   SRCZIP:=Create a src.zip based on the found sources and copied files.
169 #   INCLUDE_FILES:="com/sun/SolarisFoobar.java" means only compile this file!
170 #   EXCLUDE_FILES:="com/sun/SolarisFoobar.java" means do not compile this particular file!
171 #       "SolarisFoobar.java" means do not compile SolarisFoobar, wherever it is found.
172 #   EXTRA_FILES:=List of extra source files to include in compilation. Can be used to
173 #       specify files that need to be generated by other rules first.
174 #   HEADERS:=path to directory where all generated c-headers are written.
175 #   DEPENDS:=Extra dependecy
176 #   DISABLE_SJAVAC:=Explicitly disable the use of sjavac for this compilation unit.
177 #   KEEP_DUPS:=Do not remove duplicate file names from different source roots.
178 #   FAIL_NO_SRC:=Set to false to not fail the build if no source files are found,
179 #        default is true.
180 #   DEBUG_SYMBOLS:=Set to false to disable generation of debug symbols.
181 #   CREATE_API_DIGEST:=Set to true to use a javac plugin to generate a public API
182 #        hash which can be used for down stream dependencies to only rebuild
183 #        when the API changes. Implicitly used in sjavac.
184 #   KEEP_ALL_TRANSLATIONS:=Set to true to skip translation filtering
185 SetupJavaCompilation = $(NamedParamsMacroTemplate)
186 define SetupJavaCompilationBody
187 
188   # Verify arguments
189   ifeq ($$($1_BIN),)
190     $$(error Must specify BIN (in $1))
191   endif
192 
193   # Extract the info from the java compiler setup.
194   $1_JVM := $$($$($1_SETUP)_JVM)
195   $1_JAVAC := $$($$($1_SETUP)_JAVAC)
196   $1_FLAGS :=
197   ifneq ($$($1_DEBUG_SYMBOLS), false)
198     $1_FLAGS := -g
199   endif
200   $1_FLAGS += $$($$($1_SETUP)_FLAGS) $$($1_ADD_JAVAC_FLAGS) $(JAVAC_FLAGS)
201   ifneq ($$($1_CLASSPATH), )
202     $1_FLAGS += -cp $$(call PathList, $$($1_CLASSPATH))
203   endif
204   ifeq ($$($1_JAVAC),)
205     $$(error The Java compilation $1 refers to a non-existant java compiler setup $$($1_SETUP))
206   endif
207   $1_SJAVAC_PORTFILE := $$($$($1_SETUP)_SJAVAC_PORTFILE)
208   $1_SERVER_JVM := $$($$($1_SETUP)_SERVER_JVM)
209   $1_DISABLE_SJAVAC := $$($$($1_SETUP)_DISABLE_SJAVAC)
210 
211   ifneq ($$($1_MODULE), )
212     $1_MODULE_SUBDIR := /$$($1_MODULE)
213   endif
214 
215   # Make sure the dirs exist, or that one of the EXTRA_FILES, that may not
216   # exist yet, is in it.
217   $$(foreach d, $$($1_SRC), \
218     $$(if $$(wildcard $$d), , \
219       $$(if $$(filter $$d%, $$($1_EXTRA_FILES)), , \
220         $$(error SRC specified to SetupJavaCompilation $1 contains missing directory >$$d<) \
221       ) \
222     ) \
223   )
224   $$(call MakeDir,$$($1_BIN))
225   # Add all source roots to the find cache since we are likely going to run find
226   # on these more than once. The cache will only be updated if necessary.
227   $$(eval $$(call FillCacheFind, $$($1_SRC)))
228   # Find all files in the source trees. Preserve order of source roots so that
229   # the first version in case of multiple instances of the same file is selected.
230   # CacheFind does not preserve order so need to call it for each root.
231   $1_ALL_SRCS += $$($1_EXTRA_FILES) $$(foreach s, $$($1_SRC), $$(call CacheFind, $$s))
232   # Extract the java files.
233   $1_SRCS := $$(filter %.java, $$($1_ALL_SRCS))
234 
235   # Translate include/exclude into patterns
236   ifneq ($$($1_EXCLUDE_FILES), )
237     $1_EXCLUDE_PATTERN := $$(addprefix %, $$($1_EXCLUDE_FILES))
238   endif
239   ifneq ($$($1_INCLUDE_FILES), )
240     $1_INCLUDE_PATTERN := $$(foreach i, $$($1_SRC), $$(addprefix $$i/, $$($1_INCLUDE_FILES)))
241   endif
242   ifneq ($$($1_EXCLUDES), )
243     $1_EXCLUDE_PATTERN += $$(foreach i, $$($1_SRC), $$(addprefix $$i/, $$(addsuffix /%, $$($1_EXCLUDES))))
244   endif
245   ifneq ($$($1_INCLUDES), )
246     $1_INCLUDE_PATTERN += $$(foreach i, $$($1_SRC), $$(addprefix $$i/, $$(addsuffix /%, $$($1_INCLUDES))))
247   endif
248 
249   # Apply include/exclude patterns to java sources
250   ifneq ($$($1_EXCLUDE_PATTERN), )
251     $1_SRCS := $$(filter-out $$($1_EXCLUDE_PATTERN), $$($1_SRCS))
252   endif
253   ifneq ($$($1_INCLUDE_PATTERN), )
254     $1_SRCS := $$(filter $$($1_INCLUDE_PATTERN) $$($1_EXTRA_FILES), $$($1_SRCS))
255   endif
256 
257   ifneq ($$($1_KEEP_DUPS), true)
258     # Remove duplicate source files by keeping the first found of each duplicate.
259     # This allows for automatic overrides with custom or platform specific versions
260     # source files.
261     #
262     # For the smart javac wrapper case, add each removed file to an extra exclude
263     # file list to prevent sjavac from finding duplicate sources.
264     $1_SRCS := $$(strip $$(foreach s, $$($1_SRCS), \
265         $$(eval relative_src := $$(call remove-prefixes, $$($1_SRC), $$(s))) \
266         $$(if $$($1_$$(relative_src)), \
267           $$(eval $1_SJAVAC_EXCLUDE_FILES += $$(s)), \
268           $$(eval $1_$$(relative_src) := 1) $$(s))))
269   endif
270 
271   # Filter out any excluded translations
272   ifneq ($$($1_KEEP_ALL_TRANSLATIONS), true)
273     $1_SRCS := $$(call FilterExcludedTranslations, $$($1_SRCS), .java)
274   endif
275 
276   ifeq ($$(strip $$($1_SRCS)), )
277     ifneq ($$($1_FAIL_NO_SRC), false)
278       $$(error No source files found for $1)
279     endif
280   else
281 
282     $1_SAFE_NAME := $$(strip $$(subst /,_, $1))
283 
284     # All files below META-INF are always copied.
285     $1_ALL_COPIES := $$(filter $$(addsuffix /META-INF%,$$($1_SRC)),$$($1_ALL_SRCS))
286     # Find all files to be copied from source to bin.
287     ifneq (,$$($1_COPY)$$($1_COPY_FILES))
288       # Search for all files to be copied.
289       $1_ALL_COPIES += $$(filter $$(addprefix %,$$($1_COPY)),$$($1_ALL_SRCS))
290       # Copy these explicitly
291       $1_ALL_COPIES += $$($1_COPY_FILES)
292     endif
293     # Copy must also respect filters.
294     ifneq (,$$($1_INCLUDE_PATTERN))
295       $1_ALL_COPIES := $$(filter $$($1_INCLUDE_PATTERN),$$($1_ALL_COPIES))
296     endif
297     ifneq (,$$($1_EXCLUDE_PATTERN))
298       $1_ALL_COPIES := $$(filter-out $$($1_EXCLUDE_PATTERN),$$($1_ALL_COPIES))
299     endif
300     # Filter out any excluded translations
301     ifneq ($$($1_KEEP_ALL_TRANSLATIONS), true)
302       $1_ALL_COPIES := $$(call FilterExcludedTranslations, $$($1_ALL_COPIES), .properties)
303     endif
304     ifneq (,$$($1_ALL_COPIES))
305       # Yep, there are files to be copied!
306       $1_ALL_COPY_TARGETS:=
307           $$(foreach i,$$($1_ALL_COPIES),$$(eval $$(call add_file_to_copy,$1,$$i)))
308       # Now we can depend on $$($1_ALL_COPY_TARGETS) to copy all files!
309     endif
310 
311     # Find all property files to be copied and cleaned from source to bin.
312     ifneq (,$$($1_CLEAN)$$($1_CLEAN_FILES))
313       # Search for all files to be copied.
314       $1_ALL_CLEANS := $$(filter $$(addprefix %,$$($1_CLEAN)),$$($1_ALL_SRCS))
315       # Clean these explicitly
316       $1_ALL_CLEANS += $$($1_CLEAN_FILES)
317       # Copy and clean must also respect filters.
318       ifneq (,$$($1_INCLUDE_PATTERN))
319         $1_ALL_CLEANS := $$(filter $$($1_INCLUDE_PATTERN),$$($1_ALL_CLEANS))
320       endif
321       ifneq (,$$($1_EXCLUDE_PATTERN))
322         $1_ALL_CLEANS := $$(filter-out $$($1_EXCLUDE_PATTERN),$$($1_ALL_CLEANS))
323       endif
324       # Filter out any excluded translations
325       ifneq ($$($1_KEEP_ALL_TRANSLATIONS), true)
326         $1_ALL_CLEANS := $$(call FilterExcludedTranslations, $$($1_ALL_CLEANS), .properties)
327       endif
328       ifneq (,$$($1_ALL_CLEANS))
329         # Yep, there are files to be copied and cleaned!
330         $1_ALL_COPY_CLEAN_TARGETS:=
331             $$(foreach i,$$($1_ALL_CLEANS),$$(eval $$(call add_file_to_clean,$1,$$i)))
332         # Now we can depend on $$($1_ALL_COPY_CLEAN_TARGETS) to copy all files!
333       endif
334     endif
335 
336     # Create a sed expression to remove the source roots and to replace / with .
337     # and remove .java at the end.
338     $1_REWRITE_INTO_CLASSES:=$$(foreach i,$$($1_SRC),-e 's|$$i/||g') -e 's|/|.|g' -e 's|.java$$$$||g'
339 
340     # Create SJAVAC variable from JAVAC variable. Expects $1_JAVAC to be
341     # "bootclasspathprepend -cp .../javac.jar com.sun.tools.javac.Main"
342     # and javac is simply replaced with sjavac.
343     $1_SJAVAC:=$$(subst com.sun.tools.javac.Main,com.sun.tools.sjavac.Main,$$($1_JAVAC))
344 
345     # Set the $1_REMOTE to spawn a background javac server.
346     $1_REMOTE:=--server:portfile=$$($1_SJAVAC_PORTFILE),id=$1,sjavac=$$(subst \
347         $$(SPACE),%20,$$(subst $$(COMMA),%2C,$$(strip $$($1_SERVER_JVM) $$($1_SJAVAC))))
348 
349     $1_COMPILE_TARGET := $$($1_BIN)$$($1_MODULE_SUBDIR)/_the.$1_batch
350     $1_API_TARGET := $$($1_BIN)$$($1_MODULE_SUBDIR)/_the.$1_pubapi
351 
352     ifeq ($$($1_DISABLE_SJAVAC)x$$(ENABLE_SJAVAC),xyes)
353       # Using sjavac to compile.
354 
355       # Create the sjavac wrapper command line. Sjavac doesn't handle patterns that
356       # match the absolute path, only the part inside each src dir. Instead -i and
357       # -x flags apply only to the next -src arg on the command line.
358       $1_EXCLUDE_FILES_ABS := $$(filter /%, $$($1_EXCLUDE_FILES)) $$($1_SJAVAC_EXCLUDE_FILES)
359       $1_EXCLUDE_FILES_REL := $$(filter-out /%, $$($1_EXCLUDE_FILES))
360       $1_SJAVAC_ARGS_STRING := $$(foreach s, $$(patsubst %/, %, $$($1_SRC)), \
361           $$(addprefix -x ,$$(addsuffix /**,$$($1_EXCLUDES))) \
362           $$(addprefix -i ,$$(addsuffix /**,$$($1_INCLUDES))) \
363           $$(addprefix -x **,$$(strip $$($1_EXCLUDE_FILES_REL))) \
364           $$(addprefix -i **,$$(strip $$($1_INCLUDE_FILES))) \
365           $$(addprefix -x , $$(strip $$(patsubst $$(s)/%, %, $$(filter $$(s)/%, $$($1_EXCLUDE_FILES_ABS))))) \
366           -src $$(s))
367 
368       ifneq ($$(word 20, $$($1_SJAVAC_ARGS_STRING)), )
369         $1_SJAVAC_ARGS_FILE := $$($1_BIN)/_the.$1_args
370         $1_SJAVAC_ARGS := @$$($1_SJAVAC_ARGS_FILE)
371       else
372         $1_SJAVAC_ARGS := $$($1_SJAVAC_ARGS_STRING)
373       endif
374 
375 
376       ifneq (,$$($1_HEADERS))
377         $1_HEADERS_ARG := -h $$($1_HEADERS)
378       endif
379 
380       $1_VARDEPS := $$($1_JVM) $$($1_SJAVAC) $$($1_SJAVAC_ARGS_STRING) $$($1_FLAGS) \
381           $$($1_HEADERS_ARG) $$($1_BIN) $$($1_EXCLUDES) $$($1_INCLUDES) \
382           $$($1_EXCLUDE_FILES) $$($1_INCLUDE_FILES)
383       $1_VARDEPS_FILE := $$(call DependOnVariable, $1_VARDEPS, \
384           $$($1_BIN)$$($1_MODULE_SUBDIR)/_the.$1.vardeps)
385 
386       $$($1_COMPILE_TARGET): $$($1_SRCS) $$($1_DEPENDS) $$($1_VARDEPS_FILE)
387 		$$(call MakeDir, $$(@D) $$(dir $$($1_SJAVAC_PORTFILE)))
388 		$$(eval $$(call ListPathsSafely,$1_SRCS, $$@.tmp))
389                 ifneq ($$($1_SJAVAC_ARGS_FILE), )
390 		  $$(eval $$(call ListPathsSafely,$1_SJAVAC_ARGS_STRING, $$($1_SJAVAC_ARGS_FILE)))
391                 endif
392 		$$(call LogWarn, Compiling $1)
393 		$$(call ExecuteWithLog, $$($1_BIN)$$($1_MODULE_SUBDIR)/_the.$$($1_SAFE_NAME)_batch, \
394 		    $$($1_JVM) $$($1_SJAVAC) \
395 		        $$($1_REMOTE) \
396 		        -j 1 \
397 		        --permit-unidentified-artifacts \
398 		        --permit-sources-without-package \
399 		        --compare-found-sources $$@.tmp \
400 		        --log=$(LOG_LEVEL) \
401 		        --state-dir=$$($1_BIN)$$($1_MODULE_SUBDIR) \
402 		        $$($1_SJAVAC_ARGS) \
403 		        $$($1_FLAGS) \
404 		        $$($1_HEADERS_ARG) \
405 		        -d $$($1_BIN)) && \
406 		$(MV) $$@.tmp $$@
407                 # Create a pubapi file that only changes when the pubapi changes. Dependent
408                 # compilations can use this file to only get recompiled when pubapi has changed.
409                 # Grep returns 1 if no matching lines are found. Do not fail for this.
410 		$(GREP) -e "^I" $$($1_BIN)$$($1_MODULE_SUBDIR)/javac_state \
411 		    > $$($1_API_TARGET).tmp || test "$$$$?" = "1"
412 		if [ ! -f $$($1_API_TARGET) ] \
413 		    || [ "`$(DIFF) $$($1_API_TARGET) $$($1_API_TARGET).tmp`" != "" ]; then \
414 		  $(MV) $$($1_API_TARGET).tmp $$($1_API_TARGET); \
415 		fi
416 
417     else
418       # Using plain javac to batch compile everything.
419 
420       # When building in batch, put headers in a temp dir to filter out those that actually
421       # changed before copying them to the real header dir.
422       ifneq (,$$($1_HEADERS))
423         $1_HEADERS_ARG := -h $$($1_HEADERS).$1.tmp
424 
425         $$($1_HEADERS)/_the.$1_headers: $$($1_COMPILE_TARGET)
426 		$$(call MakeTargetDir)
427 		if [ -d "$$($1_HEADERS).$1.tmp" ]; then \
428 		  for f in `$(CD) $$($1_HEADERS).$1.tmp && $(FIND) . -type f`; do \
429 		    if [ ! -f "$$($1_HEADERS)/$$$$f" ] \
430 		        || [ "`$(DIFF) $$($1_HEADERS)/$$$$f $$($1_HEADERS).$1.tmp/$$$$f`" != "" ]; then \
431 		      $(MKDIR) -p `$(DIRNAME) $$($1_HEADERS)/$$$$f`; \
432 		      $(CP) -f $$($1_HEADERS).$1.tmp/$$$$f $$($1_HEADERS)/$$$$f; \
433 		    fi; \
434 		  done; \
435 		fi
436 		$(RM) -r $$($1_HEADERS).$1.tmp
437 		$(TOUCH) $$@
438 
439         $1_HEADER_TARGETS := $$($1_HEADERS)/_the.$1_headers
440       endif
441 
442       $1_VARDEPS := $$($1_JVM) $$($1_JAVAC) $$($1_FLAGS) $$($1_BIN) \
443           $$($1_HEADERS_ARG) $$($1_EXCLUDES) $$($1_INCLUDES) \
444           $$($1_EXCLUDE_FILES) $$($1_INCLUDE_FILES)
445       $1_VARDEPS_FILE := $$(call DependOnVariable, $1_VARDEPS, \
446           $$($1_BIN)$$($1_MODULE_SUBDIR)/_the.$1.vardeps)
447 
448       ifeq ($$($1_DISABLE_SJAVAC)x$(ENABLE_JAVAC_SERVER), xyes)
449         $1_JAVAC_CMD := $$($1_SJAVAC) $$($1_REMOTE)
450       else
451         $1_JAVAC_CMD := $$($1_JAVAC)
452       endif
453 
454       ifeq ($$($1_CREATE_API_DIGEST), true)
455         $1_API_DIGEST_FLAGS := \
456             -classpath $(BUILDTOOLS_OUTPUTDIR)/depend \
457             -Xplugin:"depend $$($1_API_TARGET)" \
458             #
459 
460         $1_EXTRA_DEPS := $(BUILDTOOLS_OUTPUTDIR)/depend/_the.COMPILE_DEPEND_batch
461       endif
462 
463       # When not using sjavac, pass along all sources to javac using an @file.
464       $$($1_COMPILE_TARGET): $$($1_SRCS) $$($1_DEPENDS) $$($1_VARDEPS_FILE) \
465           $$($1_EXTRA_DEPS)
466 		$$(call MakeDir, $$(@D))
467 		$$(eval $$(call ListPathsSafely,$1_SRCS, $$@.tmp))
468 		$$(call LogWarn, Compiling $$(words $$($1_SRCS)) files for $1)
469 		$$(call ExecuteWithLog, $$($1_BIN)$$($1_MODULE_SUBDIR)/_the.$$($1_SAFE_NAME)_batch, \
470 		    $$($1_JVM) $$($1_JAVAC_CMD) $$($1_FLAGS) \
471 		        -implicit:none \
472 		        $$($1_API_DIGEST_FLAGS) \
473 		        -d $$($1_BIN) $$($1_HEADERS_ARG) @$$@.tmp) && \
474 		$(MV) $$@.tmp $$@
475     endif
476 
477     # Add all targets to main variable
478     $1 := $$($1_ALL_COPY_TARGETS) $$($1_ALL_COPY_CLEAN_TARGETS) $$($1_COMPILE_TARGET) \
479         $$($1_HEADER_TARGETS)

481     # Check if a jar file was specified, then setup the rules for the jar.
482     ifneq (,$$($1_JAR))
483       # If no suffixes was explicitly set for this jar file.
484       # Use class and the cleaned/copied properties file suffixes as the default
485       # for the types of files to be put into the jar.
486       ifeq (,$$($1_SUFFIXES))
487         $1_SUFFIXES:=.class $$($1_CLEAN) $$($1_COPY)
488       endif

490       $$(eval $$(call SetupJarArchive, ARCHIVE_$1, \

492           SRCS:=$$($1_BIN)$$($1_MODULE_SUBDIR), \

497           JAR:=$$($1_JAR), \
498           JARMAIN:=$$($1_JARMAIN), \
499           MANIFEST:=$$($1_MANIFEST), \
500           EXTRA_MANIFEST_ATTR:=$$($1_EXTRA_MANIFEST_ATTR), \
501           JARINDEX:=$$($1_JARINDEX), \
502           HEADERS:=$$($1_HEADERS), \
503           SETUP:=$$($1_SETUP), \
504       ))
505 
506       # Add jar to target list
507       $1 += $$($1_JAR)
508     endif
509 
510     # Check if a srczip was specified, then setup the rules for the srczip.
511     ifneq (,$$($1_SRCZIP))
512       $$(eval $$(call SetupZipArchive, ZIP_ARCHIVE_$1, \
513           SRC:=$$($1_SRC), \
514           ZIP:=$$($1_SRCZIP), \
515           INCLUDES:=$$($1_INCLUDES), \
516           EXCLUDES:=$$($1_EXCLUDES), \
517           EXCLUDE_FILES:=$$($1_EXCLUDE_FILES)))
518 
519       # Add zip to target list
520       $1 += $$($1_SRCZIP)
521     endif
522   endif # Source files found
