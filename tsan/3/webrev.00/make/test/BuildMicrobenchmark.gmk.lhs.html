 29 include $(SPEC)
 30 include MakeBase.gmk
 31 include JavaCompilation.gmk
 32 include SetupJavaCompilers.gmk
 33 include TestFilesCompilation.gmk
 34 
 35 ifeq ($(JMH_CORE_JAR), )
 36   $(info Error: JMH is missing. Please use configure --with-jmh.)
 37   $(error Cannot continue)
 38 endif
 39 
 40 #### Variables
 41 
 42 MICROBENCHMARK_SRC := $(TOPDIR)/test/micro
 43 MICROBENCHMARK_IMAGE_DIR := $(TEST_IMAGE_DIR)/micro
 44 MICROBENCHMARK_JAR := $(MICROBENCHMARK_IMAGE_DIR)/benchmarks.jar
 45 
 46 MICROBENCHMARK_OUTPUT := $(SUPPORT_OUTPUTDIR)/test/micro
 47 MICROBENCHMARK_CLASSES := $(MICROBENCHMARK_OUTPUT)/classes
 48 MICROBENCHMARK_JAR_BIN := $(MICROBENCHMARK_OUTPUT)/jar
 49 



 50 JMH_UNPACKED_DIR := $(MICROBENCHMARK_OUTPUT)/jmh_jars
 51 JMH_UNPACKED_JARS_DONE := $(JMH_UNPACKED_DIR)/_unpacked.marker
 52 
 53 # External dependencies
 54 JMH_COMPILE_JARS := $(JMH_CORE_JAR) $(JMH_GENERATOR_JAR)
 55 JMH_RUNTIME_JARS := $(JMH_CORE_JAR) $(JMH_COMMONS_MATH_JAR) $(JMH_JOPT_SIMPLE_JAR)
 56 
 57 MICROBENCHMARK_CLASSPATH := $(call PathList, $(JMH_COMPILE_JARS))
 58 
 59 # Native dependencies
 60 MICROBENCHMARK_NATIVE_SRC_DIRS := $(MICROBENCHMARK_SRC)
 61 MICROBENCHMARK_NATIVE_OUTPUT := $(MICROBENCHMARK_OUTPUT)/native
 62 MICROBENCHMARK_NATIVE_EXCLUDE :=
 63 
 64 ###
 65 
 66 # Need double \n to get new lines and no trailing spaces
 67 MICROBENCHMARK_MANIFEST := Build: $(FULL_VERSION)\n\
 68 \nJMH-Version: $(JMH_VERSION)\n\
 69 \nName: OpenJDK Microbenchmark Suite
 70 








 71 
 72 #### Compile Targets
 73 
 74 # Building microbenchmark requires the jdk.unsupported and java.management modules,
 75 # and to have sjavac disabled.
 76 $(eval $(call SetupJavaCompiler, MICROBENCHMARK_JAVA_COMPILER, \
 77     JVM := $(JAVA) --add-modules jdk.unsupported --limit-modules java.management, \
 78     JAVAC := $(NEW_JAVAC), \
 79     DISABLE_SJAVAC := true, \
 80     FLAGS := --upgrade-module-path $(JDK_OUTPUTDIR)/modules --system none $(DISABLE_WARNINGS), \
 81     SERVER_DIR := $(SJAVAC_SERVER_DIR), \
 82     SERVER_JVM := $(SJAVAC_SERVER_JAVA), \
 83 ))
 84 
 85 # Build microbenchmark suite for the current JDK
 86 $(eval $(call SetupJavaCompilation, BUILD_JDK_MICROBENCHMARK, \
 87     SETUP := MICROBENCHMARK_JAVA_COMPILER, \
 88     ADD_JAVAC_FLAGS := -cp $(MICROBENCHMARK_CLASSPATH) -Xlint -Xlint:-processing -Werror, \
 89     SRC := $(MICROBENCHMARK_SRC), \
 90     BIN := $(MICROBENCHMARK_CLASSES), \
 91 ))
 92 
 93 $(BUILD_JDK_MICROBENCHMARK): $(JMH_COMPILE_JARS)
 94 








 95 # Unpacking dependencies for inclusion in the benchmark JARs
 96 $(JMH_UNPACKED_JARS_DONE): $(JMH_RUNTIME_JARS)
 97 	$(RM) -r $(JMH_UNPACKED_DIR)
 98 	$(MKDIR) -p $(JMH_UNPACKED_DIR)
 99 	$(foreach jar, $(JMH_RUNTIME_JARS), \
100             $$($(UNZIP) -oq $(jar) -d $(JMH_UNPACKED_DIR)))
101 	$(RM) -r $(JMH_UNPACKED_DIR)/META-INF
102 	$(RM) $(JMH_UNPACKED_DIR)/*.xml
103 	$(TOUCH) $@
104 
105 # Create benchmarks JAR file with benchmarks for both the old and new JDK
106 $(eval $(call SetupJarArchive, BUILD_JDK_JAR, \
107     DEPENDENCIES := $(BUILD_JDK_MICROBENCHMARK) $(JMH_UNPACKED_JARS_DONE), \

108     SRCS := $(MICROBENCHMARK_CLASSES) $(JMH_UNPACKED_DIR), \
109     BIN := $(MICROBENCHMARK_JAR_BIN), \
110     SUFFIXES := .*, \
111     EXCLUDE_FILES:= _the.BUILD_JDK_MICROBENCHMARK_batch \
112         _the.BUILD_JDK_MICROBENCHMARK.vardeps _unpacked.marker, \
113     EXTRA_MANIFEST_ATTR := $(MICROBENCHMARK_MANIFEST), \
114     JARMAIN := org.openjdk.jmh.Main, \
115     JAR := $(MICROBENCHMARK_JAR), \
116 ))
117 
118 # Setup compilation of native library dependencies
119 $(eval $(call SetupTestFilesCompilation, BUILD_MICROBENCHMARK_LIBRARIES, \
120     TYPE := LIBRARY, \
121     SOURCE_DIRS := $(MICROBENCHMARK_NATIVE_SRC_DIRS), \
122     OUTPUT_DIR := $(MICROBENCHMARK_NATIVE_OUTPUT), \
123     EXCLUDE := $(MICROBENCHMARK_NATIVE_EXCLUDE), \
124 ))
125 
126 # Setup copy of native dependencies to image output dir
127 $(eval $(call SetupCopyFiles, COPY_MICROBENCHMARK_NATIVE, \
128     SRC := $(MICROBENCHMARK_NATIVE_OUTPUT), \
129     DEST := $(MICROBENCHMARK_IMAGE_DIR)/native, \
130     FILES := $(BUILD_MICROBENCHMARK_LIBRARIES), \
131     FLATTEN := true, \
132 ))
133 
134 all: $(MICROBENCHMARK_JAR) $(BUILD_MICROBENCHMARK_LIBRARIES) $(COPY_MICROBENCHMARK_NATIVE)
