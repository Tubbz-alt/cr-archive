26 package build.tools.jdwpgen;
27 
28 import java.util.*;
29 import java.io.*;
30 
31 class ErrorNode extends AbstractCommandNode {
32 
33     protected static final String NAME_OF_ERROR_TABLE = "Error";
34 
35     ErrorNode() {
36         this(new ArrayList<Node>());
37     }
38 
39     ErrorNode(List<Node> components) {
40         this.kind = "Error";
41         this.components = components;
42         this.lineno = 0;
43     }
44 
45     void constrain(Context ctx) {
46         if (components.size() != 0) {
47             error("Errors have no internal structure");
48         }
49         super.constrain(ctx);
50     }
51 
52     void document(PrintWriter writer) {
53 
54         String com = comment();
55         if (com == null || com.length() == 0) {
56             com = ConstantSetNode.getConstant("Error_" + name);
57         }
58         writer.println("<tr>"
59                 + "<th scope=\"row\">" + "<a href=\"#" + NAME_OF_ERROR_TABLE + "_" + name + "\">" + name + "</a>"
60                 + "<td>" + com + "&nbsp;"
61                 + "</tr>");
62     }
63 
64     void genJavaComment(PrintWriter writer, int depth) {}
65 
66     void genJava(PrintWriter writer, int depth) {}
67 
68     void genCInclude(PrintWriter writer) {}
69 
70     void genJavaDebugWrite(PrintWriter writer, int depth,
71                            String writeLabel) {}
72 
73     void genJavaDebugWrite(PrintWriter writer, int depth,
74                            String writeLabel, String displayValue) {}
75 
76     public void genJavaRead(PrintWriter writer, int depth,
77                             String readLabel) {}
78 
79     void genJavaDebugRead(PrintWriter writer, int depth,
80                           String readLabel, String displayValue) {}
81 
82     void genJavaPreDef(PrintWriter writer, int depth) {}
83 }
