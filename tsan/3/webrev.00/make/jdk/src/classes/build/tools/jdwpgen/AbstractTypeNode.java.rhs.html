26 package build.tools.jdwpgen;
27 
28 
29 import java.util.*;
30 import java.io.*;
31 
32 abstract class AbstractTypeNode extends AbstractNamedNode
33                                 implements TypeNode {
34 
35     abstract String docType();
36 
37     public abstract void genJavaWrite(PrintWriter writer, int depth,
38                                       String writeLabel);
39 
40     abstract String javaRead();
41 
42     void document(PrintWriter writer) {
43         writer.println("<tr>");
44         writer.println("<td>" + indentElement(structIndent, docType()));
45         writer.println("<th scope=\"row\"><i>" + name() + "</i>");
46         writer.println("<td>" + comment() + "&nbsp;");
47         writer.println("</tr>");
48     }
49 
50     String javaType() {
51         return docType(); // default
52     }
53 
54     public void genJavaRead(PrintWriter writer, int depth,
55                             String readLabel) {
56         indent(writer, depth);
57         writer.print(readLabel);
58         writer.print(" = ");
59         writer.print(javaRead());
60         writer.println(";");
61         genJavaDebugRead(writer, depth, readLabel, debugValue(readLabel));
62     }
63 
64     public void genJavaDeclaration(PrintWriter writer, int depth) {
65         writer.println();
66         genJavaComment(writer, depth);
67         indent(writer, depth);
68         writer.print("final ");
69         writer.print(javaType());
70         writer.print(" " + name);
71         writer.println(";");
72     }
73 
74     public String javaParam() {
75         return javaType() + " " + name;
76     }
77 }
