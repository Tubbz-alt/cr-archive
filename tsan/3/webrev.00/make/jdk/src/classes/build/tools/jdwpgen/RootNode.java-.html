26 package build.tools.jdwpgen;
27 
28 import java.util.*;
29 import java.io.*;
30 
31 class RootNode extends AbstractNamedNode {
32 
33     void constrainComponent(Context ctx, Node node) {
34         if (node instanceof CommandSetNode ||
35                     node instanceof ConstantSetNode) {
36             node.constrain(ctx);
37         } else {
38             error("Expected 'CommandSet' item, got: " + node);
39         }
40     }
41 
42     void document(PrintWriter writer) {
43         writer.println("<!DOCTYPE html>");
44         writer.println("<html lang=\"en\">");
45         writer.println("<head>");
46         writer.println("<meta charset=\"utf-8\"/>");
47         writer.println("<title>" + comment() + "</title>");
48         writer.println("<style>");
49         writer.println("body {background-color:white;}");
50         writer.println("table {border: 1px solid grey; border-spacing:0px; border-collapse: separate; width: 90%;}");
51         writer.println("td, th {padding: 3px; border: 1px solid black;}");
52         writer.println("</style>");
53         writer.println("</head>");
54         writer.println("<body>");
55         writer.println("<ul role=\"navigation\">");
56         for (Node node : components) {
57             node.documentIndex(writer);
58         }
59         writer.println("</ul>");
60         writer.println("<div role=\"main\">");
61         for (Node node : components) {
62             node.document(writer);
63         }
64         writer.println("</div>");
65         writer.println("</body></html>");
66     }
67 
68     void genJava(PrintWriter writer, int depth) {
69         writer.println("package com.sun.tools.jdi;");
70         writer.println();
71         writer.println("import com.sun.jdi.*;");
72         writer.println("import java.util.*;");
73         writer.println();
74 
75         genJavaClass(writer, depth);
76     }
77 }
