26 package build.tools.jdwpgen;
27 
28 import java.util.*;
29 import java.io.*;
30 
31 class ErrorNode extends AbstractCommandNode {
32 
33     protected static final String NAME_OF_ERROR_TABLE = "Error";
34 
35     ErrorNode() {
36         this(new ArrayList<Node>());
37     }
38 
39     ErrorNode(List<Node> components) {
40         this.kind = "Error";
41         this.components = components;
42         this.lineno = 0;
43     }
44 
45     void constrain(Context ctx) {
46         if (components.size() != 0) {
47             error("Errors have no internal structure");
48         }
49         super.constrain(ctx);
50     }
51 
52     void document(PrintWriter writer) {
53 
54         String com = comment();
55         if (com == null || com.length() == 0) {
56             com = ConstantSetNode.getConstant("Error_" + name);
57         }
58         writer.println("<tr><td>" + "<a href=\"#" + NAME_OF_ERROR_TABLE + "_" + name + "\">"
59                        + name + "</a></td>" +
60                        "<td>" + com + "&nbsp;</td></tr>");
61     }
62 
63     void genJavaComment(PrintWriter writer, int depth) {}
64 
65     void genJava(PrintWriter writer, int depth) {}
66 
67     void genCInclude(PrintWriter writer) {}
68 
69     void genJavaDebugWrite(PrintWriter writer, int depth,
70                            String writeLabel) {}
71 
72     void genJavaDebugWrite(PrintWriter writer, int depth,
73                            String writeLabel, String displayValue) {}
74 
75     public void genJavaRead(PrintWriter writer, int depth,
76                             String readLabel) {}
77 
78     void genJavaDebugRead(PrintWriter writer, int depth,
79                           String readLabel, String displayValue) {}
80 
81     void genJavaPreDef(PrintWriter writer, int depth) {}
82 }
