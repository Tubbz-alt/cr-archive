26 package build.tools.cldrconverter;
27 
28 class StringListElement extends Container {
29 
30     StringListEntry list;
31     String count;
32     int index;
33 
34     StringListElement(String qName, Container parent, int index, String count) {
35         super(qName, parent);
36         while (!(parent instanceof StringListEntry)) {
37             parent = parent.getParent();
38         }
39         list = (StringListEntry) parent;
40         this.index = index;
41         this.count = count;
42     }
43 
44     @Override
45     void addCharacters(char[] characters, int start, int length) {
46         list.addCharacters(index, count, characters, start, length);
47     }
48 
49 }
