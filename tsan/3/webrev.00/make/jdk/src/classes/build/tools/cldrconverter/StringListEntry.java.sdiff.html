26 package build.tools.cldrconverter;
27 
28 import java.util.ArrayList;
29 import java.util.List;
30 import java.util.stream.IntStream;
31 
32 class StringListEntry extends Entry<List<String>> {
33 
34     private List<String> value;
35 
36     StringListEntry(String qName, Container parent, String key) {
37         super(qName, parent, key);
38         value = new ArrayList<>();
39     }
40 
41     void addCharacters(int index, String count, char[] characters, int start, int length) {
42         int size = value.size();
43         String elem = count + ":" + new String(characters, start, length);
44 
45         // quote embedded spaces, if any
46         elem = elem.replaceAll(" ", "' '");
47 
48         if (size < index) {
49             // fill with empty strings when the patterns start from index > size
50             IntStream.range(size, index).forEach(i -> value.add(i, ""));
51             value.add(index, elem);
52         } else if (size == index) {
53             value.add(index, elem);
54         } else {
55             // concatenate the pattern with the delimiter ' '
56             value.set(index, value.get(index) + " " + elem);
57         }
58     }
59 
60     @Override
61     List<String> getValue() {
62         for (String element : value) {
63             if (element != null) {
64                 return value;
65             }
66         }
67         return null;
68     }
