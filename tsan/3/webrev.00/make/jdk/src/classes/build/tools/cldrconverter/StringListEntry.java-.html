26 package build.tools.cldrconverter;
27 
28 import java.util.ArrayList;
29 import java.util.List;
30 import java.util.stream.IntStream;
31 
32 class StringListEntry extends Entry<List<String>> {
33 
34     private List<String> value;
35 
36     StringListEntry(String qName, Container parent, String key) {
37         super(qName, parent, key);
38         value = new ArrayList<>();
39     }
40 
41     void addCharacters(int index, char[] characters, int start, int length) {
42         // fill with empty strings when the patterns start from index > 0
43         if (value.size() < index) {
44             IntStream.range(0, index).forEach(i -> value.add(i, ""));
45             value.add(index, new String(characters, start, length));
46         } else {
47             value.add(index, new String(characters, start, length));
48         }
49     }
50 
51     @Override
52     List<String> getValue() {
53         for (String element : value) {
54             if (element != null) {
55                 return value;
56             }
57         }
58         return null;
59     }
60 
61 }
