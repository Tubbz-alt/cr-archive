26 package build.tools.jdwpgen;
27 
28 import java.util.*;
29 import java.io.*;
30 
31 class ConstantNode extends AbstractCommandNode {
32 
33     ConstantNode() {
34         this(new ArrayList<Node>());
35     }
36 
37     ConstantNode(List<Node> components) {
38         this.kind = "Constant";
39         this.components = components;
40         this.lineno = 0;
41     }
42 
43     void constrain(Context ctx) {
44         if (components.size() != 0) {
45             error("Constants have no internal structure");
46         }
47         super.constrain(ctx);
48     }
49 
50     void genJava(PrintWriter writer, int depth) {
51         indent(writer, depth);
52         writer.println("static final int " + name + " = " +
53                        nameNode.value() + ";");
54     }
55 
56     void document(PrintWriter writer) {
57 
58         //Add anchor to each constant with format <constant table name>_<constant name>
59         writer.println("<tr><td>" + name + "<td>" + nameNode.value() +
60                        "<td>" + comment() + " &nbsp;");
61     }
62 
63     public String getName(){
64 
65         if (name == null || name.length() == 0) {
66             prune();
67         }
68         return name;
69     }
70 }
