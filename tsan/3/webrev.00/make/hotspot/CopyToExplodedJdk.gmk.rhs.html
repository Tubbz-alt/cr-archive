26 # Copy all built libraries into exploded jdk
27 LIB_TARGETS := $(filter $(LIB_OUTPUTDIR)/%, $(TARGETS))
28 ifeq ($(call isTargetOs, windows), true)
29   $(eval $(call SetupCopyFiles, COPY_LIBS_BIN, \
30       SRC := $(SUPPORT_OUTPUTDIR)/modules_libs/java.base, \
31       DEST := $(JDK_OUTPUTDIR)/bin, \
32       FILES := $(filter-out %.lib, $(LIB_TARGETS)), \
33   ))
34 
35   $(eval $(call SetupCopyFiles, COPY_LIBS_LIB, \
36       SRC := $(SUPPORT_OUTPUTDIR)/modules_libs/java.base, \
37       DEST := $(JDK_OUTPUTDIR)/lib, \
38       FILES := $(filter %.lib, $(LIB_TARGETS))))
39 
40   TARGETS += $(COPY_LIBS_BIN) $(COPY_LIBS_LIB)
41 else
42   $(eval $(call SetupCopyFiles, COPY_LIBS, \
43       SRC := $(SUPPORT_OUTPUTDIR)/modules_libs/java.base, \
44       DEST := $(JDK_OUTPUTDIR)/lib, \
45       FILES := $(filter %$(SHARED_LIBRARY_SUFFIX), $(LIB_TARGETS)), \
46   ))
47   $(eval $(call SetupCopyFiles, LINK_LIBS, \
48       SRC := $(SUPPORT_OUTPUTDIR)/modules_libs/java.base, \
49       DEST := $(JDK_OUTPUTDIR)/lib, \
50       FILES := $(filter-out %$(SHARED_LIBRARY_SUFFIX), $(LIB_TARGETS)), \
51       MACRO := link-file-relative, \
52       LOG_ACTION := Creating symlink, \
53   ))
54 
55   TARGETS += $(COPY_LIBS) $(LINK_LIBS)
56 endif
