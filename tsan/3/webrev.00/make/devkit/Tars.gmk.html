26 # This makefile creates the tar files. It needs to be called with make -r
27 # which is why it's separated out into its own file.
28 
29 # Input parameters
30 # SRC_DIR
31 # TAR_FILE
32 
33 ifeq ($(SRC_DIR), )
34   $(error SRC_DIR not set)
35 endif
36 ifeq ($(TAR_FILE), )
37   $(error TAR_FILE not set)
38 endif
39 
40 default: tars
41 
42 tars : $(TAR_FILE)
43 
44 $(TAR_FILE): $(shell find $(SRC_DIR) -type f)
45 	@echo 'Creating compiler package $@'
46 	cd $(dir $(SRC_DIR)) && tar -czf $@ $(notdir $(SRC_DIR))/*
47 	touch $@
48 
49 .PHONY: default tars
