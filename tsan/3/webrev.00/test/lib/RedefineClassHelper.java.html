24 import java.io.PrintWriter;
25 import java.lang.instrument.Instrumentation;
26 import java.lang.instrument.ClassDefinition;
27 import jdk.test.lib.compiler.InMemoryJavaCompiler;
28 
29 /*
30  * Helper class to write tests that redefine classes.
31  * When main method is run, it will create a redefineagent.jar that can be used
32  * with the -javaagent option to support redefining classes in jtreg tests.
33  *
34  * See sample test in test/testlibrary_tests/RedefineClassTest.java
35  */
36 public class RedefineClassHelper {
37 
38     public static Instrumentation instrumentation;
39     public static void premain(String agentArgs, Instrumentation inst) {
40         instrumentation = inst;
41     }
42 
43     /**
44      * Redefine a class
45      *
46      * @param clazz Class to redefine
47      * @param javacode String with the new java code for the class to be redefined
48      */
49     public static void redefineClass(Class clazz, String javacode) throws Exception {
50         byte[] bytecode = InMemoryJavaCompiler.compile(clazz.getName(), javacode);
51         redefineClass(clazz, bytecode);
52     }
53 
54     /**
55      * Redefine a class
56      *
57      * @param clazz Class to redefine
58      * @param bytecode byte[] with the new class
59      */
60     public static void redefineClass(Class clazz, byte[] bytecode) throws Exception {
61         instrumentation.redefineClasses(new ClassDefinition(clazz, bytecode));
62     }
63 
64     /**
65      * Main method to be invoked before test to create the redefineagent.jar
66      */
67     public static void main(String[] args) throws Exception {
68         String manifest = "Premain-Class: RedefineClassHelper\nCan-Redefine-Classes: true\n";
69         ClassFileInstaller.writeJar("redefineagent.jar", ClassFileInstaller.Manifest.fromString(manifest), "RedefineClassHelper");
70     }
71 }
