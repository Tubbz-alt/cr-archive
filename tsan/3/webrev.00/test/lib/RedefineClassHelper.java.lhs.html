24 import java.io.PrintWriter;
25 import java.lang.instrument.*;

26 import jdk.test.lib.compiler.InMemoryJavaCompiler;
27 
28 /*
29  * Helper class to write tests that redefine classes.
30  * When main method is run, it will create a redefineagent.jar that can be used
31  * with the -javaagent option to support redefining classes in jtreg tests.
32  *
33  * See sample test in test/testlibrary_tests/RedefineClassTest.java
34  */
35 public class RedefineClassHelper {
36 
37     public static Instrumentation instrumentation;
38     public static void premain(String agentArgs, Instrumentation inst) {
39         instrumentation = inst;
40     }
41 
42     /**
43      * Redefine a class
44      *
45      * @param clazz Class to redefine
46      * @param javacode String with the new java code for the class to be redefined
47      */
48     public static void redefineClass(Class clazz, String javacode) throws Exception {
49         byte[] bytecode = InMemoryJavaCompiler.compile(clazz.getName(), javacode);
50         redefineClass(clazz, bytecode);
51     }
52 
53     /**
54      * Redefine a class
55      *
56      * @param clazz Class to redefine
57      * @param bytecode byte[] with the new class
58      */
59     public static void redefineClass(Class clazz, byte[] bytecode) throws Exception {
60         instrumentation.redefineClasses(new ClassDefinition(clazz, bytecode));
61     }
62 
63     /**
64      * Main method to be invoked before test to create the redefineagent.jar
65      */
66     public static void main(String[] args) throws Exception {
67         ClassFileInstaller.main("RedefineClassHelper");
68 
69         PrintWriter pw = new PrintWriter("MANIFEST.MF");
70         pw.println("Premain-Class: RedefineClassHelper");
71         pw.println("Can-Redefine-Classes: true");
72         pw.close();
73 
74         sun.tools.jar.Main jarTool = new sun.tools.jar.Main(System.out, System.err, "jar");
75         if (!jarTool.run(new String[] { "-cmf", "MANIFEST.MF", "redefineagent.jar", "RedefineClassHelper.class" })) {
76             throw new Exception("jar operation failed");
77         }
78     }
79 }
