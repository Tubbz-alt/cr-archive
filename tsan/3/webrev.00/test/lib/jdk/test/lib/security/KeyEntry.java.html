24 package jdk.test.lib.security;
25 
26 /*
27  * An entry in key store.
28  */
29 public class KeyEntry {
30 
31     // The key algorithm
32     public final String keyAlgo;
33 
34     // The PEM-encoded PKCS8 key string
35     public final String keyStr;
36 
37     // The password to protect the key
38     public final String password;
39 
40     // The certificate chain
41     // Every certificate is a PEM-encoded string
42     public final String[] certStrs;
43 
44     public KeyEntry(String keyAlgo, String keyStr, String password,
45             String[] certStrs) {
46         this.keyAlgo = keyAlgo;
47         this.keyStr = keyStr;
48         this.password = password;
49         this.certStrs = certStrs;
50     }
51 
52     public KeyEntry(String keyAlgo, String keyStr, String[] certStrs) {
53         this(keyAlgo, keyStr, null, certStrs);
54     }
55 }
