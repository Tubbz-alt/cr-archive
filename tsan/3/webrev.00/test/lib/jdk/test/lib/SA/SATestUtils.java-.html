23 package jdk.test.lib.SA;
24 
25 import java.io.IOException;
26 import java.util.List;
27 import java.util.ArrayList;
28 import jdk.test.lib.Asserts;
29 import jdk.test.lib.Platform;
30 import java.util.concurrent.TimeUnit;
31 
32 public class SATestUtils {
33 
34     public static boolean canAddPrivileges()
35        throws IOException, InterruptedException {
36        List<String> echoList = new ArrayList<String>();
37        echoList.add("sudo");
38        echoList.add("-E");
39        echoList.add("/bin/echo");
40        echoList.add("'Checking for sudo'");
41        ProcessBuilder pb = new ProcessBuilder(echoList);
42        Process echoProcess = pb.start();
43        if (echoProcess.waitFor(60, TimeUnit.SECONDS) == false) {
44            // 'sudo' has been added but we don't have a no-password
45            // entry for the user in the /etc/sudoers list. Could
46            // have timed out waiting for the password. Skip the
47            // test if there is a timeout here.
48            System.out.println("Timed out waiting for the password to be entered.");
49            echoProcess.destroyForcibly();
50            return false;
51        }
52        if (echoProcess.exitValue() == 0) {
53            return true;
54        }
55        java.io.InputStream is = echoProcess.getErrorStream();
56        String err = new String(is.readAllBytes());
57        System.out.println(err);
58        // 'sudo' has been added but we don't have a no-password
59        // entry for the user in the /etc/sudoers list. Check for
60        // the sudo error message and skip the test.
61        if (err.contains("no tty present") ||
62            err.contains("a password is required")) {
63            return false;
64        } else {
65            throw new Error("Unknown Error from 'sudo'");
66        }
67     }
68 
69     public static List<String> addPrivileges(List<String> cmdStringList)
70         throws IOException {
71         Asserts.assertTrue(Platform.isOSX());
72 
73         System.out.println("Adding 'sudo -E' to the command.");
74         List<String> outStringList = new ArrayList<String>();
75         outStringList.add("sudo");
76         outStringList.add("-E");
77         outStringList.addAll(cmdStringList);
78         return outStringList;
79     }
80 }
