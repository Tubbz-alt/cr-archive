 44 /**
 45  * Helper class for retrieving network interfaces and local addresses
 46  * suitable for testing.
 47  */
 48 public class NetworkConfiguration {
 49 
 50     private Map<NetworkInterface,List<Inet4Address>> ip4Interfaces;
 51     private Map<NetworkInterface,List<Inet6Address>> ip6Interfaces;
 52     private final boolean isIPv6Available;
 53     private boolean has_testableipv6address = false;
 54     private boolean has_sitelocaladdress = false;
 55     private boolean has_linklocaladdress = false;
 56     private boolean has_globaladdress = false;
 57 
 58     private NetworkConfiguration(
 59             Map<NetworkInterface,List<Inet4Address>> ip4Interfaces,
 60             Map<NetworkInterface,List<Inet6Address>> ip6Interfaces) {
 61         this.ip4Interfaces = ip4Interfaces;
 62         this.ip6Interfaces = ip6Interfaces;
 63 
 64         // initialize properties that can be queried
 65         isIPv6Available = !ip6Interfaces().collect(Collectors.toList()).isEmpty();
 66         ip6Interfaces().forEach(nif -> {
 67             ip6Addresses(nif)
 68                 // On Solaris or AIX, a configuration with only local or loopback
 69                 // addresses does not fully enable IPv6 operations.
 70                 // E.g. IPv6 multicasting does not work.
 71                 // So, don't set has_testableipv6address if we only find these.
 72                 .filter(addr -> Platform.isSolaris() || Platform.isAix() ?
 73                     !(addr.isAnyLocalAddress() || addr.isLoopbackAddress()) : true)
 74                 .forEach(ia -> {
 75                     has_testableipv6address = true;
 76                     if (ia.isLinkLocalAddress()) has_linklocaladdress = true;
 77                     if (ia.isSiteLocalAddress()) has_sitelocaladdress = true;
 78 
 79                     if (!ia.isLinkLocalAddress() &&
 80                         !ia.isSiteLocalAddress() &&
 81                         !ia.isLoopbackAddress()) {
 82                         has_globaladdress = true;
 83                     }
 84                 });
 85         });
 86     }
 87 
 88     private static boolean isNotExcludedInterface(NetworkInterface nif) {
 89         if (Platform.isOSX() && nif.getName().contains("awdl")) {
 90             return false;
 91         }
 92         if (Platform.isWindows()) {
 93             String dName = nif.getDisplayName();
 94             if (dName != null && dName.contains("Teredo")) {
 95                 return false;
 96             }
 97         }
 98         return true;
 99     }
100 
101     private static boolean isNotLoopback(NetworkInterface nif) {
102         try {
103             return !nif.isLoopback();
104         } catch (IOException e) {
105             throw new UncheckedIOException(e);
106         }
107     }
108 
109     private boolean hasIp4Addresses(NetworkInterface nif) {
110         return ip4Interfaces.get(nif).stream().anyMatch(a -> !a.isAnyLocalAddress());
111     }
112 
113     private boolean hasIp6Addresses(NetworkInterface nif) {
114         return ip6Interfaces.get(nif).stream().anyMatch(a -> !a.isAnyLocalAddress());
115     }
116 
117     private boolean supportsIp4Multicast(NetworkInterface nif) {
118         try {
119             if (!nif.supportsMulticast()) {
120                 return false;
121             }
122 
123             // On AIX there is a bug:
124             // When IPv6 is enabled on the system, the JDK opens sockets as AF_INET6.
125             // If there's an interface configured with IPv4 addresses only, it should
126             // be able to become the network interface for a multicast socket (that
127             // could be in both, IPv4 or IPv6 space). But both possible setsockopt
128             // calls for either IPV6_MULTICAST_IF or IP_MULTICAST_IF return
129             // EADDRNOTAVAIL. So we must skip such interfaces here.
130             if (Platform.isAix() && isIPv6Available() && !hasIp6Addresses(nif)) {
131                 return false;
132             }
133 
134             return hasIp4Addresses(nif);
135         } catch (IOException e) {
136             throw new UncheckedIOException(e);
137         }
138     }
139 
140     private boolean supportsIp6Multicast(NetworkInterface nif) {
141         try {
142             if (!nif.supportsMulticast()) {
143                 return false;
144             }
145 
146             return hasIp6Addresses(nif);
147         } catch (IOException e) {
148             throw new UncheckedIOException(e);
149         }
150     }

188     /**
189      * Returns a stream of interfaces suitable for functional tests.
190      */
191     public Stream<NetworkInterface> interfaces() {
192         return Stream.concat(ip4Interfaces(), ip6Interfaces())
193                      .distinct();
194     }
195 
196     /**
197      * Returns a stream of interfaces suitable for IPv4 functional tests.
198      */
199     public Stream<NetworkInterface> ip4Interfaces() {
200         return ip4Interfaces.keySet()
201                             .stream()
202                             .filter(NetworkConfiguration::isNotExcludedInterface)
203                             .filter(this::hasIp4Addresses);
204     }
205 
206     /**
207      * Returns a stream of interfaces suitable for IPv6 functional tests.
208      */
209     public Stream<NetworkInterface> ip6Interfaces() {
210         return ip6Interfaces.keySet()
211                             .stream()
212                             .filter(NetworkConfiguration::isNotExcludedInterface)
213                             .filter(this::hasIp6Addresses);
214     }

216     /**
217      * Returns a stream of interfaces suitable for functional tests.
218      */
219     public Stream<NetworkInterface> multicastInterfaces(boolean includeLoopback) {
220         return Stream
221             .concat(ip4MulticastInterfaces(includeLoopback),
222                     ip6MulticastInterfaces(includeLoopback))
223             .distinct();
224     }

226     /**
227      * Returns a stream of interfaces suitable for IPv4 multicast tests.
228      *
229      * The loopback interface will not be included.
230      */
231     public Stream<NetworkInterface> ip4MulticastInterfaces() {
232         return ip4MulticastInterfaces(false);
233     }

235     /**
236      * Returns a stream of interfaces suitable for IPv4 multicast tests.
237      */
238     public Stream<NetworkInterface> ip4MulticastInterfaces(boolean includeLoopback) {
239         return (includeLoopback) ?
240             ip4Interfaces().filter(this::supportsIp4Multicast) :
241             ip4Interfaces().filter(this::supportsIp4Multicast)
242                 .filter(NetworkConfiguration::isNotLoopback);
243     }

251         return ip6MulticastInterfaces(false);
252     }

255      * Returns a stream of interfaces suitable for IPv6 multicast tests.

257     public Stream<NetworkInterface> ip6MulticastInterfaces(boolean includeLoopback) {
258         return (includeLoopback) ?

306     /**
307      * Return a NetworkConfiguration instance.
308      */
309     public static NetworkConfiguration probe() throws IOException {
310         Map<NetworkInterface, List<Inet4Address>> ip4Interfaces = new HashMap<>();
311         Map<NetworkInterface, List<Inet6Address>> ip6Interfaces = new HashMap<>();
312 
313         List<NetworkInterface> nifs = list(getNetworkInterfaces());
314         for (NetworkInterface nif : nifs) {
315             // ignore interfaces that are down
316             if (!nif.isUp() || nif.isPointToPoint()) {
317                 continue;
318             }
319 
320             List<Inet4Address> ip4Addresses = new LinkedList<>();
321             List<Inet6Address> ip6Addresses = new LinkedList<>();
322             ip4Interfaces.put(nif, ip4Addresses);
323             ip6Interfaces.put(nif, ip6Addresses);
324             for (InetAddress addr : list(nif.getInetAddresses())) {
325                 if (addr instanceof Inet4Address) {
326                     ip4Addresses.add((Inet4Address) addr);
327                 } else if (addr instanceof Inet6Address) {
328                     ip6Addresses.add((Inet6Address) addr);
329                 }
330             }
331         }
332         return new NetworkConfiguration(ip4Interfaces, ip6Interfaces);
333     }

335     /** Returns detailed information for the given interface. */
336     public static String interfaceInformation(NetworkInterface nif) {
337         StringBuilder sb = new StringBuilder();
338         try {
339             sb.append("Display name: ")
340               .append(nif.getDisplayName())
341               .append("\n");
342             sb.append("Name: ")
343               .append(nif.getName())
344               .append("\n");
345             for (InetAddress inetAddress : list(nif.getInetAddresses())) {
346                 sb.append("InetAddress: ")
347                   .append(inetAddress)
348                   .append("\n");
349             }
350             sb.append("Up? ")
351               .append(nif.isUp())
352               .append("\n");
353             sb.append("Loopback? ")
354               .append(nif.isLoopback())
355               .append("\n");
356             sb.append("PointToPoint? ")
357               .append(nif.isPointToPoint())
358               .append("\n");
359             sb.append("Supports multicast? ")
360               .append(nif.supportsMulticast())
361               .append("\n");
362             sb.append("Virtual? ")
363               .append(nif.isVirtual())
364               .append("\n");
365             sb.append("Hardware address: ")
366               .append(Arrays.toString(nif.getHardwareAddress()))
367               .append("\n");
368             sb.append("MTU: ")
369               .append(nif.getMTU())
370               .append("\n");
371             sb.append("Index: ")
372               .append(nif.getIndex())
373               .append("\n");
374             sb.append("\n");
375             return sb.toString();
376         } catch (IOException e) {
377             throw new UncheckedIOException(e);
378         }
379     }

381     /** Prints all the system interface information to the give stream. */
382     public static void printSystemConfiguration(PrintStream out) {
383         try {
384             out.println("*** all system network interface configuration ***");
385             for (NetworkInterface nif : list(getNetworkInterfaces())) {
386                 out.print(interfaceInformation(nif));
387             }
388             out.println("*** end ***");
389         } catch (IOException e) {
390             throw new UncheckedIOException(e);
391         }
392     }
