 25 package jdk.test.lib.net;
 26 
 27 import java.net.InetAddress;
 28 import java.net.MalformedURLException;
 29 import java.net.URI;
 30 import java.net.URISyntaxException;
 31 import java.net.URL;
 32 
 33 public class URIBuilder {
 34 
 35     public static URIBuilder newBuilder() {
 36         return new URIBuilder();
 37     }
 38 
 39     private String scheme;
 40     private String userInfo;
 41     private String host;
 42     private int port;
 43     private String path;
 44     private String query;
 45     private String fragment;
 46 
 47     private URIBuilder() {}
 48 
 49     public URIBuilder scheme(String scheme) {
 50         this.scheme = scheme;
 51         return this;
 52     }
 53 
 54     public URIBuilder userInfo(String userInfo) {
 55         this.userInfo = userInfo;
 56         return this;
 57     }
 58 
 59     public URIBuilder host(String host) {
 60         this.host = host;
 61         return this;
 62     }
 63 
 64     public URIBuilder host(InetAddress address) {
 65         String hostaddr = address.isAnyLocalAddress()
 66                ? "localhost" : address.getHostAddress();
 67         return host(hostaddr);
 68     }
 69 
 70     public URIBuilder loopback() {
 71         return host(InetAddress.getLoopbackAddress().getHostAddress());
 72     }
 73 
 74     public URIBuilder port(int port) {
 75         this.port = port;
 76         return this;
 77     }
 78 
 79     public URIBuilder path(String path) {
 80         this.path = path;
 81         return this;
 82     }
 83 
 84     public URIBuilder query(String query) {
 85         this.query = query;
 86         return this;
 87     }
 88 
 89     public URIBuilder fragment(String fragment) {
 90         this.fragment = fragment;
 91         return this;
 92     }
 93 
 94     public URI build() throws URISyntaxException {
 95         return new URI(scheme, userInfo, host, port, path, query, fragment);
 96     }
 97 
 98     public URI buildUnchecked() {
 99         try {
100             return build();
101         } catch (URISyntaxException e) {
102             throw new IllegalArgumentException(e);
103         }
104     }
105 
106     public URL toURL() throws URISyntaxException, MalformedURLException {
107         return build().toURL();
108     }
109 
110     public URL toURLUnchecked() {
111         try {
112             return toURL();
113         } catch (URISyntaxException | MalformedURLException e) {
114             throw new IllegalArgumentException(e);
115         }
116     }
117 }
