+     private int[] pascalsTriangle(int[] source, int n) {
+         if (n > 0) {

+     private int[] verifyPascal(int n) {

+             for (int j = 0; j <= n; j++) {

+                     dataArray[i][0] = 1;
+                 } else if (j == i) {
+                     dataArray[i][i] = 1;
+                 } else if (j < i) {
