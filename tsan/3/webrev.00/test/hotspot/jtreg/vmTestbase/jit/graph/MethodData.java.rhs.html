24 package jit.graph;
25 
26 import java.lang.reflect.Method;
27 
28 import nsk.share.TestFailure;
29 
30 public final class MethodData {
31     public String ClassName;
32     public String MethodName;
33     public Class ClassObject;
34     public Method nextMethod;
35     public int id;
36     public Object instance;

37 
38     MethodData(String ClassName, String MethodName, Class ClassObject, Method nextMethod, int id) {
39         this.ClassName = ClassName;
40         this.MethodName = MethodName;
41         this.nextMethod = nextMethod;
42         this.id = id;
43         this.ClassObject = ClassObject;
44         try {
45             this.instance = ClassObject.newInstance();
46         } catch (InstantiationException e) {
47             throw new TestFailure("Class: " + ClassName + " Instantiation Exception", e);
48         } catch (IllegalAccessException e) {
49             throw new TestFailure("Class: " + ClassName + " Illegal Access Exception", e);
50         }
51     }






52 }
