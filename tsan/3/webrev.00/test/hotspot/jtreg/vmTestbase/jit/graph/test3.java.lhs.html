 23 package jit.graph;

 25 import java.util.*;
 26 import java.lang.reflect.*;
 27 import nsk.share.TestFailure;
 28 



 29 
 30 class test3 extends test1
 31 {
 32 
 33     private final int[] MethodID = {Globals.MethodID_Array[3],Globals.MethodID_Array[4]};
 34     private static Random loopNumGen = new Random(Globals.RANDOM_SEED);
 35 
 36     private final int maxLoops = 10;
 37     private int localNumLoops = loopNumGen.nextInt(maxLoops);
 38 
 39     public void selfRecursion(Vector summation, Vector ID, Long functionDepth, Integer staticFunctionDepth)
 40                 throws InvocationTargetException
 41     {
 42         Globals.appendSumToSumationVector(MethodID[1], summation);
 43 
 44         if (CGT.shouldFinish())
 45             return;

 46 
 47         if (Globals.VERBOSE)
 48             System.out.println("test3.selfRecursion");
 49         if ((functionDepth.longValue() <= 0) && (staticFunctionDepth.intValue() <=  0))
 50             {
 51                 return;
 52             }

 53 
 54         MethodData methodCallStr;
 55         Long numFcalls;
 56         Integer staticFcalls;
 57         if (staticFunctionDepth.intValue() > 0) //make a static call
 58             {
 59                 numFcalls = functionDepth;
 60                 staticFcalls = new Integer(staticFunctionDepth.intValue()-1);
 61                 //methodCallStr = Globals.nextStaticMethod(Globals.getIndexFromID(MethodID[1]));
 62                 methodCallStr = Globals.returnNextStaticMethod(MethodID[1]);
 63             }
 64         else if (localNumLoops > 0)  //make a recursive call
 65             {
 66                 numFcalls = new Long(functionDepth.longValue()-1);
 67                 staticFcalls = staticFunctionDepth;
 68                 Globals.addFunctionIDToVector(MethodID[1], ID);
 69                 localNumLoops--;
 70                 selfRecursion(summation, ID, numFcalls, staticFcalls);
 71                 return;
 72             }
 73         else //make a random call
 74             {
 75                 numFcalls = new Long(functionDepth.longValue() -1);
 76                 staticFcalls = staticFunctionDepth;
 77                 methodCallStr = Globals.nextRandomMethod();
 78 
 79                 localNumLoops = loopNumGen.nextInt(maxLoops);   //get ready for the next call to this method
 80             }
 81         Globals.addFunctionIDToVector(methodCallStr.id, ID);
 82         Globals.callMethod(methodCallStr, summation, ID, numFcalls, staticFcalls);
 83 
 84     }
 85 
 86     public void callMe(Vector summation, Vector ID, Long functionDepth, Integer staticFunctionDepth)
 87                 throws InvocationTargetException
 88     {
 89         Globals.appendSumToSumationVector(MethodID[0], summation);
 90 
 91         if (CGT.shouldFinish())
 92             return;

 93 
 94         if (Globals.VERBOSE)
 95             System.out.println("test3.callMe");

 96 
 97         if ((functionDepth.longValue() <= 0) && (staticFunctionDepth.intValue() <=  0))
 98             {
 99                 return;
100             }
101         MethodData methodCallStr;
102         Long numFcalls;
103         Integer staticFcalls;
104         if (staticFunctionDepth.intValue() > 0)
105             {
106                 numFcalls = functionDepth;
107                 staticFcalls = new Integer(staticFunctionDepth.intValue()-1);
108                 //methodCallStr = Globals.nextStaticMethod(Globals.getIndexFromID(MethodID[0]));
109                 methodCallStr = Globals.returnNextStaticMethod(MethodID[0]);
110             }
111         else
112             {
113                 numFcalls = new Long(functionDepth.longValue() -1);
114                 staticFcalls = staticFunctionDepth;
115                 methodCallStr = Globals.nextRandomMethod();
116             }
117         Globals.addFunctionIDToVector(methodCallStr.id, ID);
118         Globals.callMethod(methodCallStr, summation, ID, numFcalls, staticFcalls);
119 
120     }
121 }
