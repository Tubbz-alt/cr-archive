26 import nsk.share.TestFailure;
27 
28 import java.lang.reflect.InvocationTargetException;
29 import java.util.Vector;
30 
31 class test1 {
32     private final int classID = Globals.MethodID_Array[0];
33 
34     public void callMe(Vector summation, Vector ID, Long functionDepth, Integer staticFunctionDepth)
35             throws InvocationTargetException {
36         Globals.appendSumToSummationVector(classID, summation);

37 
38         if (CGT.shouldFinish()) {
39             return;
40         }
41 
42         if (Globals.VERBOSE) {
43             System.out.println("test1.callMe");
44         }
45 
46         if ((functionDepth.longValue() <= 0) && (staticFunctionDepth.intValue() <= 0)) {

47             return;
48         }
49 
50         MethodData methodCallStr;
51         Long numFcalls;
52         Integer staticFcalls;
53 
54         if (staticFunctionDepth.intValue() > 0) {

55             numFcalls = functionDepth;
56             staticFcalls = new Integer(staticFunctionDepth.intValue() - 1);

57             methodCallStr = Globals.returnNextStaticMethod(classID);
58         } else {
59             numFcalls = new Long(functionDepth.longValue() - 1);


60             staticFcalls = staticFunctionDepth;
61             methodCallStr = Globals.nextRandomMethod();
62         }
63 
64         Globals.addFunctionIDToVector(methodCallStr.id, ID);
65 
66         try {
67             methodCallStr.nextMethod.invoke(methodCallStr.instance,
68                     new Object[]{summation, ID, numFcalls, staticFcalls});
69         } catch (IllegalAccessException iax) {
70             throw new TestFailure("Illegal Access Exception");
71         }
72     }
73 }
