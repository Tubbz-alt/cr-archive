 24 package jit.graph;
 25 import java.util.*;


 26 import nsk.share.TestFailure;
 27 import nsk.share.test.StressOptions;
 28 
 29 class CGT extends Thread
 30 {
 31   private static StressOptions stressOptions = new StressOptions();
 32   private final static String           version = "1.0";
 33   private       static String       ClistPath = "";
 34 
 35   private static long finishTime;
 36 
 37   public CGT( String[] args )
 38   {
 39     parse (args);
 40     Globals.initialize(ClistPath);
 41     outputStats (args);
 42   }
 43 
 44 
 45   public static void main( String[] args )
 46   {
 47     stressOptions.parseCommandLine(args);
 48     CGT jnimt = new CGT(args);
 49     jnimt.start();
 50   }
 51 
 52   public void outputStats( String[] args )
 53   {
 54     System.out.println("CGT command line options:");
 55     for (int i=0; i < args.length; ++i )
 56             System.out.println("# " + args[i] );
 57 
 58     System.out.println();



 59 
 60     System.out.println("CGT parameters");
 61     System.out.println("Seed: " +Globals.RANDOM_SEED);
 62     System.out.println("Number of Threads: " +Globals.NUM_THREADS);
 63     System.out.println("Number of Random Loop iterations: " + Globals.RANDOM_LOOP);
 64     System.out.println("Number of Static Loop iterations: " + Globals.STATIC_LOOP);
 65     System.out.println("Max number of Methods in the Graph: " +  Globals.NUM_TEST_CLASSES);
 66     System.out.println("Verbose function calls: " + Globals.VERBOSE);
 67 
 68     System.out.println();
 69   }



 70 
 71   public void run()
 72   {
 73     finishTime = System.currentTimeMillis() + stressOptions.getTime() * 1000;

 74 
 75     for (int i = 0; i< Globals.NUM_THREADS; i++)
 76       new CGTThread("CGT Thread " + i).start();
 77   }


 78 
 79   public static boolean shouldFinish()
 80   {
 81      return System.currentTimeMillis() >= finishTime;
 82   }







































 83 
 84   public void parse (String args[])
 85   {
 86     for (int i = 0; i<args.length; i++)
 87       {
 88         if ((args[i].equalsIgnoreCase("-help")) || (args[i].equalsIgnoreCase("-h")) || (args[i].equalsIgnoreCase("-?")))
 89           {
 90             usage ();
 91           }
 92         else if (args[i].equalsIgnoreCase("-version"))
 93           {
 94             version();
 95           }
 96         else if (args[i].equalsIgnoreCase("-seed"))
 97           {
 98             int argIndex = i+1;
 99             if (argIndex < args.length)
100               {
101                 try
102                   {
103                     Globals.RANDOM_SEED = Math.abs(Long.parseLong(args[argIndex]));
104                   }
105                 catch (NumberFormatException e)
106                   {
107                     System.out.println("Improper Argument: " + args[i] + " " + args[argIndex]);
108                     usage ();
109                   }
110                 i++;
111               }
112             else
113               {
114                 System.out.println("Improper Argument: " + args[i]);
115                 usage ();
116               }
117 
118           }
119         else if ((args[i].equalsIgnoreCase("-thread")) || (args[i].equalsIgnoreCase("-threads")))
120           {
121             int argIndex = i+1;
122             if (argIndex < args.length)
123               {
124                 try
125                   {
126                     Globals.NUM_THREADS = Math.abs(Integer.parseInt(args[argIndex])) * stressOptions.getThreadsFactor();
127                   }
128                 catch (NumberFormatException e)
129                   {
130                     System.out.println("Improper Argument: " + args[i] + " " + args[argIndex]);
131                     usage ();
132                   }
133                 if(Globals.NUM_THREADS == 0)
134                   Globals.NUM_THREADS = 1;
135                 i++;
136               }
137             else
138               {
139                 System.out.println("Improper Argument: " + args[i]);
140                 usage ();
141               }
142 
143           }
144         else if (args[i].equalsIgnoreCase("-staticLoop"))
145           {
146             int argIndex = i+1;
147             if (argIndex < args.length)
148               {
149                 try
150                   {
151                     Globals.STATIC_LOOP = Math.abs(Integer.parseInt(args[argIndex])) * stressOptions.getIterationsFactor();
152                   }
153                 catch (NumberFormatException e)
154                   {
155                     System.out.println("Improper Argument: " + args[i] + " " + args[argIndex]);
156                     usage ();
157                   }
158                 i++;
159               }
160             else
161               {
162                 System.out.println("Improper Argument: " + args[i]);
163                 usage ();
164               }
165 
166           }
167         else if (args[i].equalsIgnoreCase("-randomLoop"))
168           {
169             int argIndex = i+1;
170             if (argIndex < args.length)
171               {
172                 try
173                   {
174                     Globals.RANDOM_LOOP = Math.abs(Long.parseLong(args[argIndex])) * stressOptions.getIterationsFactor();
175                   }
176                 catch (NumberFormatException e)
177                   {
178                     System.out.println("Improper Argument: " + args[i] + " " + args[argIndex]);
179                     usage ();
180                   }
181                 i++;
182               }
183             else
184               {
185                 System.out.println("Improper Argument: " + args[i]);
186                 usage ();
187               }
188 
189           }
190         else if (args[i].equalsIgnoreCase("-numTestClass"))
191           {
192             int argIndex = i+1;
193             if (argIndex < args.length)
194               {
195                 try
196                   {
197                     Globals.NUM_TEST_CLASSES = Math.abs(Integer.parseInt(args[argIndex]));
198                   }
199                 catch (NumberFormatException e)
200                   {
201                     System.out.println("Improper Argument: " + args[i] + " " + args[argIndex]);
202                     usage ();
203                   }
204                 i++;
205               }
206             else
207               {
208                 System.out.println("Improper Argument: " + args[i]);
209                 usage ();
210               }
211 
212           }
213         else if (args[i].equalsIgnoreCase("-v") || args[i].equalsIgnoreCase("-verbose"))
214           {
215             Globals.VERBOSE = true;
216           }
217         else if (args[i].equalsIgnoreCase("-path"))
218           {
219             int argIndex = i+1;
220             if (argIndex < args.length)
221               {
222                 ClistPath = args[argIndex];
223                 i++;
224               }
225             else
226               {
227                 System.out.println("Improper Argument: " + args[i]);
228                 usage ();
229               }
230           }
231         else if (args[i].startsWith("-stress"))
232           {
233               break;
234           }
235         else
236           {
237             System.out.println("Invalid Argument: " + args[i]);
238             usage ();
239           }
240         }
241         if (ClistPath.equals(""))
242         {
243                 System.out.println("class list path not defined");
244                 usage();
245         }
246   }
247 
248   public void usage ()
249   {
250     System.out.println("usage: java CGT [options]");
251     System.out.println("  -help                               prints out this message");
252     System.out.println("  -numTestClass #                     limits the number of \"Test Methods\" to #");
253     System.out.println("  -randomcLoop #                      # of random function calls");
254     System.out.println("  -seed #                             uses the specified seed rather than the System Time");
255     System.out.println("  -staticLoop #                       # of non-random static function calls");
256     System.out.println("  -threads #                          # number of test threads, NOTE: no maximum value");
257     System.out.println("  -version                            print out the tool version");
258     System.out.println("  -v -verbose                         turn on verbose mode");
259     throw new TestFailure("  -path <path to classlist>           required, argument so program can find classes");
260   }














































































261 
262   public void version ()
263   {
264     throw new TestFailure("CGT version = " + version);
265   }












266 }
