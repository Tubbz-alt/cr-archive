23 package jit.graph;
24 import java.lang.reflect.*;


25 import nsk.share.TestFailure;
26 
27 public final class MethodData
28 {
29   public String ClassName;
30   public String MethodName;
31   public Class  ClassObject;
32   public Method nextMethod;
33   public int id;
34   public Object instance = null;
35 
36   MethodData(String ClassName, String MethodName, Class ClassObject,Method nextMethod, int id)
37   {
38     this.ClassName = ClassName;
39     this.MethodName = MethodName;
40     this.nextMethod = nextMethod;
41     this.id = id;
42     this.ClassObject = ClassObject;
43     try
44       {
45         this.instance = ClassObject.newInstance();
46       }
47     catch (InstantiationException ix)
48       {
49         throw new TestFailure("Class: " +ClassName+ " Instantiation Exception");
50       }
51     catch (IllegalAccessException iax)
52       {
53         throw new TestFailure("Class: " +ClassName+ " Illegal Access Exception");
54       }
55   }
56 }
