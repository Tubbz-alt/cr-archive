25 JAVA_OPTS="$TESTJAVAOPTS $TESTVMOPTS"
26 JAVA="$TESTJAVA/bin/java"
27 
28 log=output.txt
29 tiered="off"
30 
31 if echo "$JAVA_OPTS" | grep "[-]XX:+TieredCompilation"; then
32     tiered="on"
33 elif echo "$JAVA_OPTS" | grep "[-]XX:-TieredCompilation"; then
34     tiered="off"
35 else
36     echo "TEST PASSED: TieredCompilation option is not specified. Nothing to test"
37     exit 0
38 fi
39 
40 echo "Tiered is ${tiered}"
41 
42 $JAVA $JAVA_OPTS -XX:+PrintTieredEvents -version >$log 2>&1
43 
44 if grep "Client VM" $log; then
45     echo "TEST PASSED: Client VM. The test is useless"
46     exit 0
47 fi
48 
49 if grep "TieredCompilation not supported in this VM" $log; then
50     echo "TEST PASSED: Non-tiered Server VM. The test is useless"
51     exit 0
52 fi
53 
54 if ! egrep '^[0-9.]+: \[compile level=[0-9]' $log; then
55     if [ "${tiered}" == "on" ]; then
56         echo "TEST FAILED: No PrintTieredEvents output"
57         exit 2
58     else
59         echo "TEST PASSED: No events with TieredCompilation turned off"
60         exit 0
61     fi
62 else
63     if [ "${tiered}" == "off" ]; then
64         echo "TEST FAILED: PrintTieredEvents output found but TieredCompilation is turned off"
65         exit 2
66     else
67         echo "TEST PASSED: PrintTieredEvents output found"
68         exit 0
69     fi
70 fi
