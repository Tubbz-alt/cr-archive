 52 package vm.mlvm.meth.stress.compiler.deoptimize;
 53 
 54 import java.lang.invoke.MethodHandle;
 55 import java.lang.invoke.MethodHandles;
 56 import java.lang.invoke.MethodType;
 57 
 58 import vm.mlvm.meth.share.Argument;
 59 import vm.mlvm.meth.share.MHTransformationGen;
 60 import vm.mlvm.meth.share.RandomArgumentsGen;
 61 import vm.mlvm.share.Env;
 62 import vm.mlvm.share.MlvmTest;
 63 import vm.mlvm.share.MultiThreadedTest;
 64 
 65 // TODO: check deopt using vm.mlvm.share.comp framework
 66 public class Test extends MultiThreadedTest {
 67 
 68     static class A {
 69         Object m() {
 70             return new Integer(0);
 71         }
 72     }
 73 
 74     static class B extends A {
 75         @Override
 76         Object m() {
 77             return new Integer(1);
 78         }
 79     }
 80 
 81     static class TestData {
 82         final A _a;
 83         final int _expectedResult;
 84 
 85         TestData(A a) {
 86             _a = a;
 87             _expectedResult = (Integer) a.m();
 88         }
 89 
 90         Object m() {
 91             return _a.m();
 92         }
 93     }
 94 
 95     volatile TestData _data = new TestData(new A());
 96 
 97     volatile boolean _testDone = false;
 98 
 99     final MethodHandle _mh;
100     final Argument[] _finalArgs;
101 
102     public Test() throws Throwable {
103         _mh = MethodHandles.lookup().findVirtual(TestData.class, "m", MethodType.methodType(Object.class));
104         _finalArgs = RandomArgumentsGen.createRandomArgs(true, _mh.type());
105     }
106 
107     @Override
108     protected void initializeTest() throws Throwable {
109         super.initializeTest();
110         if (calcThreadNum() < 2) {
111             throw new IllegalArgumentException("Number of threads should be 2 or more");
112         }
113     }
114 
115     private void sleepAndDeoptimize() throws Throwable {
116         try {
117             Thread.sleep(3000);
118             // Force deoptimization in uncommon trap logic
119             Env.traceNormal("Deoptimizing");
120             _data = new TestData((A) Test.class.getClassLoader().loadClass(Test.class.getName() + "$B").newInstance());
121             Thread.sleep(3000);
122         } finally {
123             this._testDone = true;
124         }
125     }
126 
127     @Override
128     protected boolean runThread(int threadNum) throws Throwable {
129         if ( threadNum == 0 ) {
130             sleepAndDeoptimize();
131             return true;
132         }
133 
134         while ( ! Test.this._testDone) {
135             TestData dataSnapshot = _data;
136             Integer expectedResult = (Integer) dataSnapshot._a.m();
137             Argument retVal = new Argument(Object.class, expectedResult);
138             retVal.setPreserved(true);
139 
140             MHTransformationGen.createAndCallSequence(retVal, dataSnapshot, _mh, _finalArgs, true);
141         }
142 
143         return true;
144     }
145 
146     public static void main(String[] args) { MlvmTest.launch(args); }
147 }
