 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.canWatchFieldAccess()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test has three phases and works as follows.              <BR>
 46  * <BR>
 47  * In first phase,                                                      <BR>
 48  * upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 52  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 53  * and waits for the event within the predefined time interval.         <BR>
 54  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 57  * within debuggee's special methodForCommunication().                  <BR>
 58  * <BR>
 59  * In second phase to check the assetion,                               <BR>
 60  * the debugger and the debuggee perform the following loop.            <BR>
 61  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 63  *   and to inform the debugger with the event.                         <BR>
 64  * - Upon getting the BreakpointEvent,                                  <BR>
 65  *   the debugger performs the check case required.                     <BR>
 66  * Note. To inform each other of needed actions, the debugger and       <BR>
 67  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 68  * <BR>
 69  * In third phase when at the end,                                      <BR>
 70  * the debuggee changes the value of the "instruction"                  <BR>
 71  * to inform the debugger of checks finished, and both end.             <BR>
 72  */
 73

152     private int runThis (String argv[], PrintStream out) {
153 
154         argsHandler     = new ArgumentHandler(argv);
155         logHandler      = new Log(out, argsHandler);
156         Binder binder   = new Binder(argsHandler, logHandler);
157 
158         waitTime        = argsHandler.getWaitTime() * 60000;
159 
160         try {
161             log2("launching a debuggee :");
162             log2("       " + debuggeeName);
163             if (argsHandler.verbose()) {
164                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
165             } else {
166                 debuggee = binder.bindToDebugee(debuggeeName);
167             }
168             if (debuggee == null) {
169                 log3("ERROR: no debuggee launched");
170                 return FAILED;
171             }
172             log2("debuggee launched");
173         } catch ( Exception e ) {
174             log3("ERROR: Exception : " + e);
175             log2("       test cancelled");
176             return FAILED;
177         }
178 
179         debuggee.redirectOutput(logHandler);
180 
181         vm = debuggee.VM();
182 
183         eventQueue = vm.eventQueue();
184         if (eventQueue == null) {
185             log3("ERROR: eventQueue == null : TEST ABORTED");
186             vm.exit(PASS_BASE);
187             return FAILED;
188         }
189 
190         log2("invocation of the method runTest()");
191         switch (runTest()) {
192 
193             case 0 :  log2("test phase has finished normally");
194                       log2("   waiting for the debuggee to finish ...");
195                       debuggee.waitFor();
196 
197                       log2("......getting the debuggee's exit status");
198                       int status = debuggee.getStatus();
199                       if (status != PASS_BASE) {
200                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
201                               status + " != PASS_BASE");
202                           testExitCode = FAILED;
203                       } else {
204                           log2("......debuggee returned expected exit status: " +
205                               status + " == PASS_BASE");
206                       }
207                       break;
208 
209             default : log3("ERROR: runTest() returned unexpected value");
210 
211             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
212                       log2("......forcing: vm.exit();");
213                       testExitCode = FAILED;
214                       try {
215                           vm.exit(PASS_BASE);
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : e");
218                       }
219                       break;
220 
221             case 2 :  log3("test cancelled due to VMDisconnectedException");
222                       log2("......trying: vm.process().destroy();");
223                       testExitCode = FAILED;
224                       try {
225                           Process vmProcess = vm.process();
226                           if (vmProcess != null) {
227                               vmProcess.destroy();
228                           }
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233             }
234 
235         return testExitCode;
236     }

240     * Return value: 0 - normal end of the test
241     *               1 - ubnormal end of the test
242     *               2 - VMDisconnectedException while test phase
243     */

245     private int runTest() {
246 
247         try {
248             testRun();
249 
250             log2("waiting for VMDeathEvent");
251             getEventSet();
252             if (eventIterator.nextEvent() instanceof VMDeathEvent)
253                 return 0;
254 
255             log3("ERROR: last event is not the VMDeathEvent");
256             return 1;
257         } catch ( VMDisconnectedException e ) {
258             log3("ERROR: VMDisconnectedException : " + e);
259             return 2;
260         } catch ( Exception e ) {
261             log3("ERROR: Exception : " + e);
262             return 1;
263         }
264 
265     }

267     private void testRun()
268                  throws JDITestRuntimeException, Exception {
269 
270         eventRManager = vm.eventRequestManager();
271 
272         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
273         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
274         cpRequest.addClassFilter(debuggeeName);
275 
276         cpRequest.enable();
277         vm.resume();
278         getEventSet();
279         cpRequest.disable();
280 
281         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
282         debuggeeClass = event.referenceType();
283 
284         if (!debuggeeClass.name().equals(debuggeeName))
285            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
286 
287         log2("      received: ClassPrepareEvent for debuggeeClass");
288 
289         String bPointMethod = "methodForCommunication";
290         String lineForComm  = "lineForComm";
291         BreakpointRequest bpRequest;
292 
293         try {
294             bpRequest = settingBreakpoint(threadByName("main"),
295                                           debuggeeClass,
296                                           bPointMethod, lineForComm, "zero");
297         } catch ( Exception e ) {
298             throw e;
299         }
300         bpRequest.enable();
301 
302     //------------------------------------------------------  testing section
303 
304         log1("     TESTING BEGINS");
305 
306         for (int i = 0; ; i++) {
307 
308             vm.resume();
309             breakpointForCommunication();
310 
311             int instruction = ((IntegerValue)
312                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
313 
314             if (instruction == 0) {
315                 vm.resume();
316                 break;
317             }
318 
319             log1("  new check: # " + i);
320 
321             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
322 
323             log2("...... vm.canWatchFieldAccess()");
324             if ( vm.canWatchFieldAccess() )
325                 log2("      vm.canWatchFieldAccess() == true");
326             else
327                 log2("      vm.canWatchFieldAccess() == false");
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
330         }
331         log1("    TESTING ENDS");
332         return;
333     }

335     private ThreadReference threadByName(String name)
336                  throws JDITestRuntimeException {
337 
338         List         all = vm.allThreads();
339         ListIterator li  = all.listIterator();
340 
341         for (; li.hasNext(); ) {
342             ThreadReference thread = (ThreadReference) li.next();
343             if (thread.name().equals(name))
344                 return thread;
345         }
346         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
347     }

349    /*
350     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
351     *                                             String, String, String)
352     *
353     * It sets up a breakpoint at given line number within a given method in a given class
354     * for a given thread.
355     *
356     * Return value: BreakpointRequest object  in case of success
357     *
358     * JDITestRuntimeException   in case of an Exception thrown within the method
359     */
360 
361     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
362                                                   ReferenceType testedClass,
363                                                   String methodName,
364                                                   String bpLine,
365                                                   String property)
366             throws JDITestRuntimeException {
367 
368         log2("......setting up a breakpoint:");
369         log2("       thread: " + thread + "; class: " + testedClass +
370                         "; method: " + methodName + "; line: " + bpLine);
371 
372         List              alllineLocations = null;
373         Location          lineLocation     = null;
374         BreakpointRequest breakpRequest    = null;
375 
376         try {
377             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
378 
379             alllineLocations = method.allLineLocations();
380 
381             int n =
382                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
383             if (n > alllineLocations.size()) {
384                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
385             } else {
386                 lineLocation = (Location) alllineLocations.get(n);
387                 try {
388                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
389                     breakpRequest.putProperty("number", property);
390                     breakpRequest.addThreadFilter(thread);
391                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
392                 } catch ( Exception e1 ) {
393                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
394                     breakpRequest    = null;
395                 }
396             }
397         } catch ( Exception e2 ) {
398             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
399             breakpRequest    = null;
400         }
401 
402         if (breakpRequest == null) {
403             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
404             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
405         }
406 
407         log2("      a breakpoint has been set up");
408         return breakpRequest;
409     }

412     private void getEventSet()
413                  throws JDITestRuntimeException {
414         try {
415 //            log2("       eventSet = eventQueue.remove(waitTime);");
416             eventSet = eventQueue.remove(waitTime);
417             if (eventSet == null) {
418                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
419             }
420 //            log2("       eventIterator = eventSet.eventIterator;");
421             eventIterator = eventSet.eventIterator();
422         } catch ( Exception e ) {
423             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
424         }
425     }

428     private void breakpointForCommunication()
429                  throws JDITestRuntimeException {
430 
431         log2("breakpointForCommunication");
432         getEventSet();
433 
434         if (eventIterator.nextEvent() instanceof BreakpointEvent)
435             return;
436 
437         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
438     }
