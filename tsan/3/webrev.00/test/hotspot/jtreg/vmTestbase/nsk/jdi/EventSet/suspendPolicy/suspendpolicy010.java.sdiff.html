421     private ThreadReference threadByName(String name)
422                  throws JDITestRuntimeException {
423 
424         List         all = vm.allThreads();
425         ListIterator li  = all.listIterator();
426 
427         for (; li.hasNext(); ) {
428             ThreadReference thread = (ThreadReference) li.next();
429             if (thread.name().equals(name))
430                 return thread;
431         }
432         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
