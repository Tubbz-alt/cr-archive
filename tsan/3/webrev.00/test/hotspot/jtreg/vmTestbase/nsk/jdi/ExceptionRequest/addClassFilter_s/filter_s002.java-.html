 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addClassFilter(String)</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates a TestClass10 object, to use as filter,       <BR>
 68  *   and invokes the methodForCommunication to be suspended and         <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an ExceptionRequest and enables it,                      <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */
 84

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest  eventRequest1      = null;
318 
319         String        property1          = "ExceptionRequest1";
320 
321 
322 
323 
324 
325         for (int i = 0; ; i++) {
326 
327             vm.resume();
328             breakpointForCommunication();
329 
330             int instruction = ((IntegerValue)
331                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
332 
333             if (instruction == 0) {
334                 vm.resume();
335                 break;
336             }
337 
338 
339             log1(":::::: case: # " + i);
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
342 
343             switch (i) {
344 
345               case 0:
346                      eventRequest1 = setting22ExceptionRequest(null, testedClassName1,
347                                               EventRequest.SUSPEND_NONE, property1);
348 
349                      eventRequest1.enable();
350 
351                      try {
352                          log2("......eventRequest1.addClassFilter(testedClassName1);");
353                          log2("        InvalidRequestStateException expected");
354                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
355                          log3("ERROR: no InvalidRequestStateException");
356                          testExitCode = FAILED;
357                      } catch ( InvalidRequestStateException e ) {
358                          log2("        InvalidRequestStateException");
359                      }
360 
361                      log2("......eventRequest1.disable();");
362                      eventRequest1.disable();
363                      try {
364                          log2("......eventRequest1.addClassFilter(testedClassName1);");
365                          log2("        no InvalidRequestStateException expected");
366                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
367                          log2("        no InvalidRequestStateException");
368                      } catch ( InvalidRequestStateException e ) {
369                          log3("ERROR: InvalidRequestStateException");
370                          testExitCode = FAILED;
371                      }
372 
373                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
374                      eventRManager.deleteEventRequest(eventRequest1);
375                      try {
376                          log2("......eventRequest1.addClassFilter(testedClassName1);");
377                          log2("        InvalidRequestStateException expected");
378                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
379                          log3("ERROR: no InvalidRequestStateException");
380                          testExitCode = FAILED;
381                      } catch ( InvalidRequestStateException e ) {
382                          log2("        InvalidRequestStateException");
383                      }
384                       break;
385 
386               default:
387                       throw new JDITestRuntimeException("** default case 2 **");
388             }
389 
390             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
391         }
392         log1("    TESTING ENDS");
393         return;
394     }

396     private ThreadReference threadByName(String name)
397                  throws JDITestRuntimeException {
398 
399         List         all = vm.allThreads();
400         ListIterator li  = all.listIterator();
401 
402         for (; li.hasNext(); ) {
403             ThreadReference thread = (ThreadReference) li.next();
404             if (thread.name().equals(name))
405                 return thread;
406         }
407         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
408     }

410    /*
411     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
412     *                                             String, String, String)
413     *
414     * It sets up a breakpoint at given line number within a given method in a given class
415     * for a given thread.
416     *
417     * Return value: BreakpointRequest object  in case of success
418     *
419     * JDITestRuntimeException   in case of an Exception thrown within the method
420     */
421 
422     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
423                                                   ReferenceType testedClass,
424                                                   String methodName,
425                                                   String bpLine,
426                                                   String property)
427             throws JDITestRuntimeException {
428 
429         log2("......setting up a breakpoint:");
430         log2("       thread: " + thread + "; class: " + testedClass +
431                         "; method: " + methodName + "; line: " + bpLine);
432 
433         List              alllineLocations = null;
434         Location          lineLocation     = null;
435         BreakpointRequest breakpRequest    = null;
436 
437         try {
438             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
439 
440             alllineLocations = method.allLineLocations();
441 
442             int n =
443                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
444             if (n > alllineLocations.size()) {
445                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
446             } else {
447                 lineLocation = (Location) alllineLocations.get(n);
448                 try {
449                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
450                     breakpRequest.putProperty("number", property);
451                     breakpRequest.addThreadFilter(thread);
452                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
453                 } catch ( Exception e1 ) {
454                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
455                     breakpRequest    = null;
456                 }
457             }
458         } catch ( Exception e2 ) {
459             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
460             breakpRequest    = null;
461         }
462 
463         if (breakpRequest == null) {
464             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
465             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
466         }
467 
468         log2("      a breakpoint has been set up");
469         return breakpRequest;
470     }

473     private void getEventSet()
474                  throws JDITestRuntimeException {
475         try {
476 //            log2("       eventSet = eventQueue.remove(waitTime);");
477             eventSet = eventQueue.remove(waitTime);
478             if (eventSet == null) {
479                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
480             }
481 //            log2("       eventIterator = eventSet.eventIterator;");
482             eventIterator = eventSet.eventIterator();
483         } catch ( Exception e ) {
484             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
485         }
486     }

489     private void breakpointForCommunication()
490                  throws JDITestRuntimeException {
491 
492         log2("breakpointForCommunication");
493         getEventSet();
494 
495         if (eventIterator.nextEvent() instanceof BreakpointEvent)
496             return;
497 
498         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
499     }

503     private ExceptionRequest setting22ExceptionRequest ( ThreadReference thread,
504                                                          String          testedClass,
505                                                          int             suspendPolicy,
506                                                          String          property       )
507             throws JDITestRuntimeException {
508         try {
509             log2("......setting up ExceptionRequest:");
510             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
511 
512             ExceptionRequest
513             excr = eventRManager.createExceptionRequest(null, true, true);
514             excr.putProperty("number", property);
515             if (thread != null)
516                 excr.addThreadFilter(thread);
517             excr.addClassFilter(testedClass);
518             excr.setSuspendPolicy(suspendPolicy);
519 
520             log2("      ExceptionRequest has been set up");
521             return excr;
522         } catch ( Exception e ) {
523             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
524             log3("       ExceptionRequest HAS NOT BEEN SET UP");
525             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
526         }
