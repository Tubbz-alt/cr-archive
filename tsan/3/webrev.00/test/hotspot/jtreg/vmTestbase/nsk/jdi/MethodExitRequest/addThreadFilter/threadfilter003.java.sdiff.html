411     private ThreadReference threadByName(String name)
412                  throws JDITestRuntimeException {
413 
414         List         all = vm.allThreads();
415         ListIterator li  = all.listIterator();
416 
417         for (; li.hasNext(); ) {
418             ThreadReference thread = (ThreadReference) li.next();
419             if (thread.name().equals(name))
420                 return thread;
421         }
422         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
423     }
