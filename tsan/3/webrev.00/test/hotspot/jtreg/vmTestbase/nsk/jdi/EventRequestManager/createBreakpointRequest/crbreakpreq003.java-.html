 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createBreakpointRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    - Multiple breakpoints at the same location are permitted.        <BR>
 47  *    - Creates a new disabled BreakpointRequest.                       <BR>
 48  *    - The BreakpointRequest is at the specified location.             <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares testcase and invokes                         <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger performs the checks required.                         <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

247    /*
248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase
251     */
252 
253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = threadByName("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         List         requests = null;
310         ListIterator li       = null;
311 
312         BreakpointRequest bpRequest1 = null;
313         BreakpointRequest bpRequest2 = null;
314 
315 
316         log1("     TESTING BEGINS");
317 
318         for (int i = 0; ; i++) {
319 
320             vm.resume();
321             breakpointForCommunication();
322 
323             int instruction = ((IntegerValue)
324                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
325 
326             if (instruction == 0) {
327                 vm.resume();
328                 break;
329             }
330 
331             log1(":::::: case: # " + i);
332 
333             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
334 
335                 log2("      creating two BreakpointRequests for 'breakpLocation': bpRequest1 & bpRequest2");
336 
337                 bpRequest1 = eventRManager.createBreakpointRequest(breakpLocation);
338                 bpRequest2 = eventRManager.createBreakpointRequest(breakpLocation);
339 
340                 log2("      checking up on: requests are disabled");
341                 if (bpRequest1.isEnabled()) {
342                     testExitCode = FAILED;
343                     log3("ERROR: bpRequest1 is enabled");
344                 }
345                 if (bpRequest2.isEnabled()) {
346                     testExitCode = FAILED;
347                     log3("ERROR: bpRequest2 is enabled");
348                 }
349                 log2("      compareing locations equality");
350                 if ( !bpRequest1.location().equals(breakpLocation) ) {
351                     testExitCode = FAILED;
352                     log3("ERROR: !bpRequest1.location().equals(breakpLocation)");
353                 }
354                 if ( !bpRequest2.location().equals(breakpLocation) ) {
355                     testExitCode = FAILED;
356                     log3("ERROR: !bpRequest2.location().equals(breakpLocation)");
357                 }
358 
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
360         }
361         log1("    TESTING ENDS");
362         return;
363     }

365     private ThreadReference threadByName(String name)
366                  throws JDITestRuntimeException {
367 
368         List         all = vm.allThreads();
369         ListIterator li  = all.listIterator();
370 
371         for (; li.hasNext(); ) {
372             ThreadReference thread = (ThreadReference) li.next();
373             if (thread.name().equals(name))
374                 return thread;
375         }
376         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
377     }

379    /*
380     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
381     *                                             String, String, String)
382     *
383     * It sets up a breakpoint at given line number within a given method in a given class
384     * for a given thread.
385     *
386     * Return value: BreakpointRequest object  in case of success
387     *
388     * JDITestRuntimeException   in case of an Exception thrown within the method
389     */
390 
391     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
392                                                   ReferenceType testedClass,
393                                                   String methodName,
394                                                   String bpLine,
395                                                   String property)
396             throws JDITestRuntimeException {
397 
398         log2("......setting up a breakpoint:");
399         log2("       thread: " + thread + "; class: " + testedClass +
400                         "; method: " + methodName + "; line: " + bpLine);
401 
402         List              alllineLocations = null;
403         Location          lineLocation     = null;
404         BreakpointRequest breakpRequest    = null;
405 
406         try {
407             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
408 
409             alllineLocations = method.allLineLocations();
410 
411             int n =
412                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
413             if (n > alllineLocations.size()) {
414                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
415             } else {
416                 lineLocation = (Location) alllineLocations.get(n);
417 breakpLocation = lineLocation;
418                 try {
419                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
420                     breakpRequest.putProperty("number", property);
421                     breakpRequest.addThreadFilter(thread);
422                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
423                 } catch ( Exception e1 ) {
424                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
425                     breakpRequest    = null;
426                 }
427             }
428         } catch ( Exception e2 ) {
429             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
430             breakpRequest    = null;
431         }
432 
433         if (breakpRequest == null) {
434             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
435             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
436         }
437 
438         log2("      a breakpoint has been set up");
439         return breakpRequest;
440     }

443     private void getEventSet()
444                  throws JDITestRuntimeException {
445         try {
446 //            log2("       eventSet = eventQueue.remove(waitTime);");
447             eventSet = eventQueue.remove(waitTime);
448             if (eventSet == null) {
449                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
450             }
451 //            log2("       eventIterator = eventSet.eventIterator;");
452             eventIterator = eventSet.eventIterator();
453         } catch ( Exception e ) {
454             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
455         }
456     }

459     private void breakpointForCommunication()
460                  throws JDITestRuntimeException {
461 
462         log2("breakpointForCommunication");
463         getEventSet();
464 
465         if (eventIterator.nextEvent() instanceof BreakpointEvent)
466             return;
467 
468         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
469     }
