 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a BreakpointEvent set,              <BR>
 46  * the method returns a value corresponding to one              <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a BreakpointRequest,                          <BR>
 70  *   resumes the debuggee, and waits for the BreakpointEvent.           <BR>
 71  * - The debuggee invokes the special method in which control passes    <BR>
 72  *   through a predefined line to be resulting in the event.            <BR>
 73  * - Upon getting new event, the debugger                               <BR>
 74  *   performs the check corresponding to the event.                     <BR>
 75  * <BR>
 76  * Note. To inform each other of needed actions, the debugger and       <BR>
 77  *       and the debuggee use debuggee's variable "instruction".        <BR>
 78  * In third phase, at the end,                                          <BR>
 79  * the debuggee changes the value of the "instruction"                  <BR>
 80  * to inform the debugger of checks finished, and both end.             <BR>
 81  * <BR>
 82  */
 83

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

246    /*
247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase
250     */

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
260                 log3("ERROR: last event is not the VMDeathEvent");
261                 return 1;
262             }
263 
264             log2("waiting for VMDisconnectEvent");
265             getEventSet();
266             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
267                 log3("ERROR: last event is not the VMDisconnectEvent");
268                 return 1;
269             }
270 
271             return 0;
272 
273         } catch ( VMDisconnectedException e ) {
274             log3("ERROR: VMDisconnectedException : " + e);
275             return 2;
276         } catch ( Exception e ) {
277             log3("ERROR: Exception : " + e);
278             return 1;
279         }
280 
281     }

283     private void testRun()
284                  throws JDITestRuntimeException, Exception {
285 
286         eventRManager = vm.eventRequestManager();
287 
288         log2("......getting ClassPrepareEvent for debuggee's class");
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292         cpRequest.enable();
293         vm.resume();
294         getEventSet();
295         cpRequest.disable();
296 
297         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
298         debuggeeClass = event.referenceType();
299 
300         if (!debuggeeClass.name().equals(debuggeeName))
301            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
305 
306         String            bPointMethod = "methodForCommunication";
307         String            lineForComm  = "lineForComm";
308         BreakpointRequest bpRequest;
309         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
310         bpRequest = settingBreakpoint(mainThread,
311                                       debuggeeClass,
312                                       bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315         vm.resume();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         EventRequest eventRequest1 = null;
322         EventRequest eventRequest2 = null;
323         EventRequest eventRequest3 = null;
324 
325         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
327         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
328         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
329 
330         int policyExpected[] = {    SUSPEND_NONE,
331                                     SUSPEND_THREAD,
332                                     SUSPEND_ALL,
333                                     SUSPEND_THREAD,
334                                     SUSPEND_ALL,
335                                     SUSPEND_ALL,
336                                     SUSPEND_ALL      };
337         int policy = 0;
338 
339 
340         ReferenceType testClassReference = null;
341 
342         String methodName = "method";
343         String bpLineName = "breakpointLine";
344 
345 
346         for (int i = 0; ; i++) {
347 
348 
349             breakpointForCommunication();
350 
351             int instruction = ((IntegerValue)
352                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
353             if (instruction == 0) {
354                 vm.resume();
355                 break;
356             }
357 
358 
359             log1(":::::: case: # " + i);
360 
361             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
362 
363             switch (i) {
364 
365               case 0:
366                       testClassReference =
367                            (ReferenceType) vm.classesByName(testedClassName).get(0);
368 
369                       eventRequest1 = settingBreakpointRequest (mainThread,
370                                              testClassReference, methodName, bpLineName,
371                                              SUSPEND_NONE, "BreakpointRequest1");
372                       eventRequest1.enable();
373                       break;
374 
375               case 1:
376                       eventRequest2 = settingBreakpointRequest (mainThread,
377                                              testClassReference, methodName, bpLineName,
378                                              SUSPEND_THREAD, "BreakpointRequest2");
379                       eventRequest2.enable();
380                       break;
381 
382               case 2:
383                       eventRequest3 = settingBreakpointRequest (mainThread,
384                                              testClassReference, methodName, bpLineName,
385                                              SUSPEND_ALL, "BreakpointRequest3");
386                       eventRequest3.enable();
387                       break;
388 
389               case 3:
390                       eventRequest1 = settingBreakpointRequest (mainThread,
391                                              testClassReference, methodName, bpLineName,
392                                              SUSPEND_NONE, "BreakpointRequest4");
393                       eventRequest1.enable();
394                       eventRequest2 = settingBreakpointRequest (mainThread,
395                                              testClassReference, methodName, bpLineName,
396                                              SUSPEND_THREAD, "BreakpointRequest5");
397                       eventRequest2.enable();
398                       break;
399 
400               case 4:
401                       eventRequest1 = settingBreakpointRequest (mainThread,
402                                              testClassReference, methodName, bpLineName,
403                                              SUSPEND_NONE, "BreakpointRequest6");
404                       eventRequest1.enable();
405                       eventRequest3 = settingBreakpointRequest (mainThread,
406                                              testClassReference, methodName, bpLineName,
407                                              SUSPEND_ALL, "BreakpointRequest7");
408                       eventRequest3.enable();
409                       break;
410 
411               case 5:
412                       eventRequest2 = settingBreakpointRequest (mainThread,
413                                              testClassReference, methodName, bpLineName,
414                                              SUSPEND_THREAD, "BreakpointRequest8");
415                       eventRequest2.enable();
416                       eventRequest3 = settingBreakpointRequest (mainThread,
417                                              testClassReference, methodName, bpLineName,
418                                              SUSPEND_ALL, "BreakpointRequest9");
419                       eventRequest3.enable();
420                       break;
421 
422               case 6:
423                       eventRequest1 = settingBreakpointRequest (mainThread,
424                                              testClassReference, methodName, bpLineName,
425                                              SUSPEND_NONE, "BreakpointRequest10");
426                       eventRequest1.enable();
427                       eventRequest2 = settingBreakpointRequest (mainThread,
428                                              testClassReference, methodName, bpLineName,
429                                              SUSPEND_THREAD, "BreakpointRequest11");
430                       eventRequest2.enable();
431                       eventRequest3 = settingBreakpointRequest (mainThread,
432                                              testClassReference, methodName, bpLineName,
433                                              SUSPEND_ALL, "BreakpointRequest12");
434                       eventRequest3.enable();
435                       break;
436 
437 
438               default:
439                       throw new JDITestRuntimeException("** default case 2 **");
440             }
441 
442             log2("......waiting for new BreakpointEvent : " + i);
443             mainThread.resume();
444             getEventSet();
445 
446             Event newEvent = eventIterator.nextEvent();
447             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
448             if ( !( newEvent instanceof BreakpointEvent)) {
449                  log3("ERROR: new event is not BreakpointEvent");
450                  testExitCode = FAILED;
451             } else {
452                 log2("       Event is instanceof BreakpointEvent");
453                 policy = eventSet.suspendPolicy();
454                 if (policy != policyExpected[i]) {
455                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
456                     log3("       eventSet.suspendPolicy() == " + policy);
457                     log3("       policyExpected           == " + policyExpected[i]);
458                     testExitCode = FAILED;
459                 }
460             }
461 
462             switch (i) {
463               case 0: eventRequest1.disable();                          break;
464               case 1: eventRequest2.disable();                          break;
465               case 2: eventRequest3.disable();                          break;
466               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
467               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
468               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
469               case 6: eventRequest1.disable(); eventRequest2.disable();
470                       eventRequest3.disable();                          break;
471            }
472 
473             switch (policy) {
474               case SUSPEND_NONE   :                      break;
475               case SUSPEND_THREAD : mainThread.resume(); break;
476               case SUSPEND_ALL    : vm.resume();         break;
477               default: throw new JDITestRuntimeException("** default case 3 **");
478             }
479 
480             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
481         }
482         log1("    TESTING ENDS");
483         return;
484     }

486    /*
487     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
488     *                                             String, String, String)
489     *
490     * It sets up a breakpoint at given line number within a given method in a given class
491     * for a given thread.
492     *
493     * Return value: BreakpointRequest object  in case of success
494     *
495     * JDITestRuntimeException   in case of an Exception thrown within the method
496     */
497 
498     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
499                                                   ReferenceType testedClass,
500                                                   String methodName,
501                                                   String bpLine,
502                                                   String property)
503             throws JDITestRuntimeException {
504 
505         log2("......setting up a breakpoint:");
506         log2("       thread: " + thread + "; class: " + testedClass +
507                         "; method: " + methodName + "; line: " + bpLine);
508 
509         List              alllineLocations = null;
510         Location          lineLocation     = null;
511         BreakpointRequest breakpRequest    = null;
512 
513         try {
514             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
515 
516             alllineLocations = method.allLineLocations();
517 
518             int n =
519                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
520             if (n > alllineLocations.size()) {
521                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
522             } else {
523                 lineLocation = (Location) alllineLocations.get(n);
524                 try {
525                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
526                     breakpRequest.putProperty("number", property);
527                     breakpRequest.addThreadFilter(thread);
528                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
529                 } catch ( Exception e1 ) {
530                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
531                     breakpRequest    = null;
532                 }
533             }
534         } catch ( Exception e2 ) {
535             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
536             breakpRequest    = null;
537         }
538 
539         if (breakpRequest == null) {
540             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
541             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
542         }
543 
544         log2("      a breakpoint has been set up");
545         return breakpRequest;
546     }

549     private void getEventSet()
550                  throws JDITestRuntimeException {
551         try {
552 //            log2("       eventSet = eventQueue.remove(waitTime);");
553             eventSet = eventQueue.remove(waitTime);
554             if (eventSet == null) {
555                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
556             }
557 //            log2("       eventIterator = eventSet.eventIterator;");
558             eventIterator = eventSet.eventIterator();
559         } catch ( Exception e ) {
560             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
561         }
562     }

565     private void breakpointForCommunication()
566                  throws JDITestRuntimeException {
567 
568         log2("breakpointForCommunication");
569         getEventSet();
570 
571         if (eventIterator.nextEvent() instanceof BreakpointEvent)
572             return;
573 
574         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
575     }

579     private BreakpointRequest settingBreakpointRequest ( ThreadReference thread,
580                                                          ReferenceType   testedClass,
581                                                          String          methodName,
582                                                          String          bpLine,
583                                                          int             suspendPolicy,
584                                                          String          property        )
585             throws JDITestRuntimeException {
586         try {
587             log2("......setting up a breakpoint:");
588             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
589 
590             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
591             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
592 
593             BreakpointRequest
594             bpr = eventRManager.createBreakpointRequest(loc);
595             bpr.putProperty("number", property);
596             bpr.addThreadFilter(thread);
597             bpr.setSuspendPolicy(suspendPolicy);
598 
599             log2("      a breakpoint has been set up");
600             return bpr;
601         } catch ( Exception e ) {
602             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
603             log3("       BreakpointRequest HAS NOT BEEN SET UP");
604             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
605         }
606     }
