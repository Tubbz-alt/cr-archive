 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.addClassFilter(ReferenceType)</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates a filter_rt002aTestClass10 object, to use as a filter,        <BR>
 68  *   and invokes the methodForCommunication to be suspended and         <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an StepRequest and enables it,                           <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */
 84

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         EventRequest  eventRequest1      = null;
312 
313         String        property1          = "StepRequest1";
314 
315         ReferenceType testClassReference = null;
316 
317         ThreadReference thread1     = null;
318         String          threadName1 = "thread1";
319 
320 
321         for (int i = 0; ; i++) {
322 
323             vm.resume();
324             breakpointForCommunication();
325 
326             int instruction = ((IntegerValue)
327                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
328 
329             if (instruction == 0) {
330                 vm.resume();
331                 break;
332             }
333 
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339             switch (i) {
340 
341               case 0:
342                      testClassReference =
343                           (ReferenceType) vm.classesByName(testedClassName).get(0);
344 
345                      thread1 = debuggee.threadByNameOrThrow(threadName1);
346 
347                      eventRequest1 = setting21StepRequest(thread1, testClassReference,
348                                               EventRequest.SUSPEND_NONE, property1);
349                      eventRequest1.enable();
350 
351                      try {
352                          log2("......eventRequest1.addClassFilter(testClassReference);");
353                          log2("        InvalidRequestStateException expected");
354                          ((StepRequest)eventRequest1).addClassFilter(testClassReference);
355                          log3("ERROR: no InvalidRequestStateException");
356                          testExitCode = FAILED;
357                      } catch ( InvalidRequestStateException e ) {
358                          log2("        InvalidRequestStateException");
359                      }
360 
361                      log2("......eventRequest1.disable();");
362                      eventRequest1.disable();
363                      try {
364                          log2("......eventRequest1.addClassFilter(testClassReference);");
365                          log2("        no InvalidRequestStateException expected");
366                          ((StepRequest)eventRequest1).addClassFilter(testClassReference);
367                          log2("        no InvalidRequestStateException");
368                      } catch ( InvalidRequestStateException e ) {
369                          log3("ERROR: InvalidRequestStateException");
370                          testExitCode = FAILED;
371                      }
372 
373                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
374                      eventRManager.deleteEventRequest(eventRequest1);
375                      try {
376                          log2("......eventRequest1.addClassFilter(testClassReference);");
377                          log2("        InvalidRequestStateException expected");
378                          ((StepRequest)eventRequest1).addClassFilter(testClassReference);
379                          log3("ERROR: no InvalidRequestStateException");
380                          testExitCode = FAILED;
381                      } catch ( InvalidRequestStateException e ) {
382                          log2("        InvalidRequestStateException");
383                      }
384                       break;
385 
386               default:
387                       throw new JDITestRuntimeException("** default case 2 **");
388             }
389 
390             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
391         }
392         log1("    TESTING ENDS");
393         return;
394     }

396    /*
397     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
398     *                                             String, String, String)
399     *
400     * It sets up a breakpoint at given line number within a given method in a given class
401     * for a given thread.
402     *
403     * Return value: BreakpointRequest object  in case of success
404     *
405     * JDITestRuntimeException   in case of an Exception thrown within the method
406     */
407 
408     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
409                                                   ReferenceType testedClass,
410                                                   String methodName,
411                                                   String bpLine,
412                                                   String property)
413             throws JDITestRuntimeException {
414 
415         log2("......setting up a breakpoint:");
416         log2("       thread: " + thread + "; class: " + testedClass +
417                         "; method: " + methodName + "; line: " + bpLine);
418 
419         List              alllineLocations = null;
420         Location          lineLocation     = null;
421         BreakpointRequest breakpRequest    = null;
422 
423         try {
424             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
425 
426             alllineLocations = method.allLineLocations();
427 
428             int n =
429                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
430             if (n > alllineLocations.size()) {
431                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
432             } else {
433                 lineLocation = (Location) alllineLocations.get(n);
434                 try {
435                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
436                     breakpRequest.putProperty("number", property);
437                     breakpRequest.addThreadFilter(thread);
438                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
439                 } catch ( Exception e1 ) {
440                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
441                     breakpRequest    = null;
442                 }
443             }
444         } catch ( Exception e2 ) {
445             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
446             breakpRequest    = null;
447         }
448 
449         if (breakpRequest == null) {
450             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
451             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
452         }
453 
454         log2("      a breakpoint has been set up");
455         return breakpRequest;
456     }

459     private void getEventSet()
460                  throws JDITestRuntimeException {
461         try {
462 //            log2("       eventSet = eventQueue.remove(waitTime);");
463             eventSet = eventQueue.remove(waitTime);
464             if (eventSet == null) {
465                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
466             }
467 //            log2("       eventIterator = eventSet.eventIterator;");
468             eventIterator = eventSet.eventIterator();
469         } catch ( Exception e ) {
470             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
471         }
472     }

475     private void breakpointForCommunication()
476                  throws JDITestRuntimeException {
477 
478         log2("breakpointForCommunication");
479         getEventSet();
480 
481         if (eventIterator.nextEvent() instanceof BreakpointEvent)
482             return;
483 
484         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
485     }

489     private StepRequest setting21StepRequest ( ThreadReference thread,
490                                                ReferenceType   testedClass,
491                                                int             suspendPolicy,
492                                                String          property        )
493             throws JDITestRuntimeException {
494         try {
495             log2("......setting up StepRequest:");
496             log2("       thread: " + thread + "; property: " + property);
497 
498             StepRequest
499             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_INTO);
500             str.putProperty("number", property);
501             str.addCountFilter(2);
502             str.setSuspendPolicy(suspendPolicy);
503             str.addClassFilter(testedClass);
504 
505             log2("      StepRequest has been set up");
506             return str;
507         } catch ( Exception e ) {
508             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
509             log3("       StepRequest HAS NOT BEEN SET UP");
510             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
511         }
