397     private ThreadReference threadByName(String name)
398                  throws JDITestRuntimeException {
399 
400         List         all = vm.allThreads();
401         ListIterator li  = all.listIterator();
402 
403         for (; li.hasNext(); ) {
404             ThreadReference thread = (ThreadReference) li.next();
405             if (thread.name().equals(name))
406                 return thread;
407         }
408         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
