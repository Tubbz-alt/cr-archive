 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.eventIterator()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for each type of the following Events:  <BR>
 46  * AccessWatchpoint, ModificationWatchpoint, Breakpoint,        <BR>
 47  * Exception, MethodEntry, MethodExit, and Step,                <BR>
 48  *  - the method returns non-null object, and                   <BR>
 49  *  - object's class is subclass of class Iterator.             <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase the debugger and debuggee perform the following loop.<BR>
 66  * - The debugger creates new Request, resumes the debuggee, and        <BR>
 67  *   waits for corresponding Event.                                     <BR>
 68  * - The debuggee performes an action resulting in Event required.      <BR>
 69  * - Upon getting the Event, the debugger checks up on EventSet.        <BR>
 70  * <BR>
 71  * In third phase, at the end                                           <BR>
 72  * the debuggee changes the value of the "instruction"                  <BR>
 73  * to inform the debugger of checks finished, and both end.             <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * <BR>
 78  */

155     private int runThis (String argv[], PrintStream out) {
156 
157         argsHandler     = new ArgumentHandler(argv);
158         logHandler      = new Log(out, argsHandler);
159         Binder binder   = new Binder(argsHandler, logHandler);
160 
161         waitTime        = argsHandler.getWaitTime() * 60000;
162 
163         try {
164             log2("launching a debuggee :");
165             log2("       " + debuggeeName);
166             if (argsHandler.verbose()) {
167                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
168             } else {
169                 debuggee = binder.bindToDebugee(debuggeeName);
170             }
171             if (debuggee == null) {
172                 log3("ERROR: no debuggee launched");
173                 return FAILED;
174             }
175             log2("debuggee launched");
176         } catch ( Exception e ) {
177             log3("ERROR: Exception : " + e);
178             log2("       test cancelled");
179             return FAILED;
180         }
181 
182         debuggee.redirectOutput(logHandler);
183 
184         vm = debuggee.VM();
185 
186         eventQueue = vm.eventQueue();
187         if (eventQueue == null) {
188             log3("ERROR: eventQueue == null : TEST ABORTED");
189             vm.exit(PASS_BASE);
190             return FAILED;
191         }
192 
193         log2("invocation of the method runTest()");
194         switch (runTest()) {
195 
196             case 0 :  log2("test phase has finished normally");
197                       log2("   waiting for the debuggee to finish ...");
198                       debuggee.waitFor();
199 
200                       log2("......getting the debuggee's exit status");
201                       int status = debuggee.getStatus();
202                       if (status != PASS_BASE) {
203                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
204                               status + " != PASS_BASE");
205                           testExitCode = FAILED;
206                       } else {
207                           log2("......debuggee returned expected exit status: " +
208                               status + " == PASS_BASE");
209                       }
210                       break;
211 
212             default : log3("ERROR: runTest() returned unexpected value");
213 
214             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
215                       log2("......forcing: vm.exit();");
216                       testExitCode = FAILED;
217                       try {
218                           vm.exit(PASS_BASE);
219                       } catch ( Exception e ) {
220                           log3("ERROR: Exception : e");
221                       }
222                       break;
223 
224             case 2 :  log3("test cancelled due to VMDisconnectedException");
225                       log2("......trying: vm.process().destroy();");
226                       testExitCode = FAILED;
227                       try {
228                           Process vmProcess = vm.process();
229                           if (vmProcess != null) {
230                               vmProcess.destroy();
231                           }
232                       } catch ( Exception e ) {
233                           log3("ERROR: Exception : e");
234                       }
235                       break;
236             }
237 
238         return testExitCode;
239     }

242    /*
243     * Return value: 0 - normal end of the test
244     *               1 - ubnormal end of the test
245     *               2 - VMDisconnectedException while test phase
246     */

248     private int runTest() {
249 
250         try {
251             testRun();
252 
253             log2("waiting for VMDeathEvent");
254             getEventSet();
255 
256             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
257                 log3("ERROR: last event is not the VMDeathEvent");
258                 return 1;
259             }
260 
261             log2("waiting for VMDisconnectEvent");
262             getEventSet();
263 
264             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
265                 log3("ERROR: last event is not the VMDisconnectEvent");
266                 return 1;
267             }
268 
269             return 0;
270 
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278 
279     }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304         BreakpointRequest bpRequest;
305 
306         ThreadReference mainThread = debuggee.threadByNameOrThrow("main");
307 
308         bpRequest = settingBreakpoint(mainThread,
309                                       debuggeeClass,
310                                       bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317             String accWatchpointName = "var1";
318             String modWatchpointName = "var2";
319             String bpLineName        = "breakpointLine";
320             String bpMethodName      = "method";
321             String awFieldName       = "awFieldName";
322             String mwFieldName       = "mwFieldName";
323             String excName           = "method";
324             String menName           = "method";
325             String mexName           = "method";
326 
327         EventRequest eRequest = null;
328 
329         for (int i = 0; ; i++) {
330 
331             vm.resume();
332             breakpointForCommunication();
333 
334             int instruction = ((IntegerValue)
335                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
336 
337             if (instruction == 0) {
338                 vm.resume();
339                 break;
340             }
341 
342             log1(":::::: case: # " + i);
343 
344             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
345 
346             log2("......ReferenceType testClass = (ReferenceType) (vm.classesByName(testedClassName)).get(0);");
347             ReferenceType testClass = (ReferenceType) (vm.classesByName(testedClassName)).get(0);
348 
349             switch (i) {
350 
351               case 0:
352                       eRequest = null;
353                       if (vm.canWatchFieldAccess()) {
354                           String awName = ( (StringReference) testClass.getValue(
355                                             testClass.fieldByName(awFieldName))).value();
356                           eRequest = settingAccessWatchpointRequest(mainThread,
357                                             testClass, awName, "AccessWatchpointRequest");
358                           eRequest.enable();
359                       }
360                       break;
361 
362               case 1:
363                       eRequest = null;
364                       if (vm.canWatchFieldModification() ) {
365                           String mwName = ( (StringReference) testClass.getValue(
366                                             testClass.fieldByName(mwFieldName))).value();
367                           eRequest = settingModificationWatchpointRequest(mainThread,
368                                             testClass, mwName, "ModificationWatchpointRequest");
369                           eRequest.enable();
370                       }
371                       break;
372 
373               case 2:
374                       eRequest = settingBreakpointRequest(mainThread, testClass,
375                                         bpMethodName, bpLineName,
376                                         EventRequest.SUSPEND_EVENT_THREAD, "BreakpointRequest");
377                       eRequest.enable();
378                       break;
379 
380               case 3:
381                       eRequest = settingExceptionRequest(mainThread, testClass,
382                                                                 "ExceptionRequest");
383                       eRequest.enable();
384                       break;
385 
386               case 4:
387                       eRequest = settingMethodEntryRequest(mainThread, testClass,
388                                                                 "MethodEntryRequest");
389                       eRequest.enable();
390                       break;
391 
392               case 5:
393                       eRequest = settingMethodExitRequest(mainThread, testClass,
394                                                                 "MethodExitRequest");
395                       eRequest.enable();
396                       break;
397 
398               case 6:
399                       eRequest = settingStepRequest(mainThread, "StepRequest");
400                       eRequest.enable();
401                       break;
402             }
403 
404                                  //  see setting up Acceess&Modification requests above
405             if (eRequest == null)
406                 continue;
407 
408             log2("......getting new Event and checking up on EventIterator");
409             vm.resume();
410             getEventSet();
411             eRequest.disable();
412 
413             if (eventIterator == null) {
414                 testExitCode = FAILED;
415                 log3("ERROR: eventIterator == null");
416             }
417             if ( !(eventIterator instanceof Iterator) ) {
418                 testExitCode = FAILED;
419                 log3("ERROR: eventIterator is NOT instanceof Iterator");
420             }
421 
422             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
423         }
424         log1("    TESTING ENDS");
425         return;
426     }

428    /*
429     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
430     *                                             String, String, String)
431     *
432     * It sets up a breakpoint at given line number within a given method in a given class
433     * for a given thread.
434     *
435     * Return value: BreakpointRequest object  in case of success
436     *
437     * JDITestRuntimeException   in case of an Exception thrown within the method
438     */
439 
440     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
441                                                   ReferenceType testedClass,
442                                                   String methodName,
443                                                   String bpLine,
444                                                   String property)
445             throws JDITestRuntimeException {
446 
447         log2("......setting up a breakpoint:");
448         log2("       thread: " + thread + "; class: " + testedClass +
449                         "; method: " + methodName + "; line: " + bpLine);
450 
451         List              alllineLocations = null;
452         Location          lineLocation     = null;
453         BreakpointRequest breakpRequest    = null;
454 
455         try {
456             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
457 
458             alllineLocations = method.allLineLocations();
459 
460             int n =
461                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
462             if (n > alllineLocations.size()) {
463                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
464             } else {
465                 lineLocation = (Location) alllineLocations.get(n);
466                 try {
467                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
468                     breakpRequest.putProperty("number", property);
469                     breakpRequest.addThreadFilter(thread);
470                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
471                 } catch ( Exception e1 ) {
472                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
473                     breakpRequest    = null;
474                 }
475             }
476         } catch ( Exception e2 ) {
477             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
478             breakpRequest    = null;
479         }
480 
481         if (breakpRequest == null) {
482             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
483             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
484         }
485 
486         log2("      a breakpoint has been set up");
487         return breakpRequest;
488     }

491     private void getEventSet()
492                  throws JDITestRuntimeException {
493         try {
494 //            log2("       eventSet = eventQueue.remove(waitTime);");
495             eventSet = eventQueue.remove(waitTime);
496             if (eventSet == null) {
497                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
498             }
499 //            log2("       eventIterator = eventSet.eventIterator;");
500             eventIterator = eventSet.eventIterator();
501         } catch ( Exception e ) {
502             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
503         }
504     }

508                  throws JDITestRuntimeException {
509 
510         log2("breakpointForCommunication");
511         getEventSet();
512 
513         if (eventIterator.nextEvent() instanceof BreakpointEvent)
514             return;
515 
516         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
517     }
