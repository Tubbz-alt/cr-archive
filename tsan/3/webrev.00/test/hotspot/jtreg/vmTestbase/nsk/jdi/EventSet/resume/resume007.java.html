 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for MethodExitEvent sets.             <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a MethodExitRequest, resumes                  <BR>
 74  *   the debuggee, and waits for the MethodExitEvent.                   <BR>
 75  * - The debuggee invokes the special method which makes access to      <BR>
 76  *   a predefined variable to be resulting in the event.                <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         String bPointMethod = "methodForCommunication";
301         String lineForComm  = "lineForComm";
302         BreakpointRequest bpRequest;
303         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
304         bpRequest = settingBreakpoint(mainThread,
305                                       debuggeeClass,
306                                       bPointMethod, lineForComm, "zero");
307         bpRequest.enable();
308 
309         vm.resume();
310 
311     //------------------------------------------------------  testing section
312 
313         log1("     TESTING BEGINS");
314 
315         EventRequest eventRequest1 = null;
316         EventRequest eventRequest2 = null;
317         EventRequest eventRequest3 = null;
318 
319         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
320         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
321         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
322 
323         ReferenceType testClassReference = null;
324 
325 
326         for (int i = 0; ; i++) {
327 
328             breakpointForCommunication();
329 
330             int instruction = ((IntegerValue)
331                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
332 
333             if (instruction == 0) {
334                 vm.resume();
335                 break;
336             }
337 
338             log1(":::::: case: # " + i);
339 
340             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
341 
342             switch (i) {
343 
344               case 0:
345                       testClassReference =
346                            (ReferenceType) vm.classesByName(testedClassName).get(0);
347 
348                       eventRequest1 = settingMethodExitRequest (mainThread,
349                                              testClassReference,
350                                              SUSPEND_NONE, "MethodExitRequest1");
351                       eventRequest1.enable();
352                       break;
353 
354               case 1:
355                       eventRequest2 = settingMethodExitRequest (mainThread,
356                                              testClassReference,
357                                              SUSPEND_THREAD, "MethodExitRequest2");
358                       eventRequest2.enable();
359                       break;
360 
361               case 2:
362                       eventRequest3 = settingMethodExitRequest (mainThread,
363                                              testClassReference,
364                                              SUSPEND_ALL, "MethodExitRequest3");
365                       eventRequest3.enable();
366                       break;
367 
368 
369               default:
370                       throw new JDITestRuntimeException("** default case 2 **");
371             }
372 
373             log2("......waiting for new MethodExitEvent : " + i);
374             mainThread.resume();
375             getEventSet();
376 
377             Event newEvent = eventIterator.nextEvent();
378             if ( !(newEvent instanceof MethodExitEvent)) {
379                  log3("ERROR: new event is not MethodExitEvent");
380                  testExitCode = FAILED;
381             } else {
382 
383                 String property = (String) newEvent.request().getProperty("number");
384                 log2("       got new MethodExitEvent with propety 'number' == " + property);
385 
386                 log2("......checking up on EventSet.resume()");
387                 log2("......--> vm.suspend();");
388                 vm.suspend();
389 
390                 log2("        getting : Map<String, Integer> suspendsCounts1");
391 
392                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
393                 for (ThreadReference threadReference : vm.allThreads()) {
394                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
395                 }
396 
397                 log2("        eventSet.resume;");
398                 eventSet.resume();
399 
400                 log2("        getting : Map<String, Integer> suspendsCounts2");
401                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
402                 for (ThreadReference threadReference : vm.allThreads()) {
403                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
404                 }
405 
406                 log2(suspendsCounts1.toString());
407                 log2(suspendsCounts2.toString());
408 
409                 log2("        getting : int policy = eventSet.suspendPolicy();");
410                 int policy = eventSet.suspendPolicy();
411 
412                 switch (policy) {
413 
414                   case SUSPEND_NONE   :
415                        log2("        case SUSPEND_NONE");
416                        for (String threadName : suspendsCounts1.keySet()) {
417                            log2("        checking " + threadName);
418                            if (!suspendsCounts2.containsKey(threadName)) {
419                                log3("ERROR: couldn't get ThreadReference for " + threadName);
420                                testExitCode = FAILED;
421                                break;
422                            }
423                            int count1 = suspendsCounts1.get(threadName);
424                            int count2 = suspendsCounts2.get(threadName);
425                            if (count1 != count2) {
426                                log3("ERROR: suspendCounts don't match for : " + threadName);
427                                log3("before resuming : " + count1);
428                                log3("after  resuming : " + count2);
429                                testExitCode = FAILED;
430                                break;
431                            }
432                        }
433                        eventRequest1.disable();
434                        break;
435 
436                   case SUSPEND_THREAD :
437                        log2("        case SUSPEND_THREAD");
438                        for (String threadName : suspendsCounts1.keySet()) {
439                            log2("checking " + threadName);
440                            if (!suspendsCounts2.containsKey(threadName)) {
441                                log3("ERROR: couldn't get ThreadReference for " + threadName);
442                                testExitCode = FAILED;
443                                break;
444                            }
445                            int count1 = suspendsCounts1.get(threadName);
446                            int count2 = suspendsCounts2.get(threadName);
447                            String eventThreadName = event.thread().name();
448                            int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
449                            if (count1 != expectedValue) {
450                                log3("ERROR: suspendCounts don't match for : " + threadName);
451                                log3("before resuming : " + count1);
452                                log3("after  resuming : " + count2);
453                                testExitCode = FAILED;
454                                break;
455                            }
456                        }
457                        eventRequest2.disable();
458                        break;
459 
460                   case SUSPEND_ALL    :
461 
462                         log2("        case SUSPEND_ALL");
463                         for (String threadName : suspendsCounts1.keySet()) {
464                             log2("checking " + threadName);
465                             if (!event.request().equals(eventRequest3))
466                                 break;
467                             if (!suspendsCounts2.containsKey(threadName)) {
468                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
469                                 testExitCode = FAILED;
470                                 break;
471                             }
472                             int count1 = suspendsCounts1.get(threadName);
473                             int count2 = suspendsCounts2.get(threadName);
474                             if (count1 != count2 + 1) {
475                                 log3("ERROR: suspendCounts don't match for : " + threadName);
476                                 log3("before resuming : " + count1);
477                                 log3("after  resuming : " + count2);
478                                 testExitCode = FAILED;
479                                 break;
480                             }
481                         }
482                         eventRequest3.disable();
483                         break;
484 
485                   default: throw new JDITestRuntimeException("** default case 1 **");
486                 }
487                 informDebuggeeTestCase(i);
488             }
489 
490             log2("......--> vm.resume()");
491             vm.resume();
492             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
493         }
494         log1("    TESTING ENDS");
495         return;
496     }

498    /*
499     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
500     *                                             String, String, String)
501     *
502     * It sets up a breakpoint at given line number within a given method in a given class
503     * for a given thread.
504     *
505     * Return value: BreakpointRequest object  in case of success
506     *
507     * JDITestRuntimeException   in case of an Exception thrown within the method
508     */
509 
510     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
511                                                   ReferenceType testedClass,
512                                                   String methodName,
513                                                   String bpLine,
514                                                   String property)
515             throws JDITestRuntimeException {
516 
517         log2("......setting up a breakpoint:");
518         log2("       thread: " + thread + "; class: " + testedClass +
519                         "; method: " + methodName + "; line: " + bpLine);
520 
521         List              alllineLocations = null;
522         Location          lineLocation     = null;
523         BreakpointRequest breakpRequest    = null;
524 
525         try {
526             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
527 
528             alllineLocations = method.allLineLocations();
529 
530             int n =
531                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
532             if (n > alllineLocations.size()) {
533                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
534             } else {
535                 lineLocation = (Location) alllineLocations.get(n);
536                 try {
537                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
538                     breakpRequest.putProperty("number", property);
539                     breakpRequest.addThreadFilter(thread);
540                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
541                 } catch ( Exception e1 ) {
542                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
543                     breakpRequest    = null;
544                 }
545             }
546         } catch ( Exception e2 ) {
547             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
548             breakpRequest    = null;
549         }
550 
551         if (breakpRequest == null) {
552             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
553             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
554         }
555 
556         log2("      a breakpoint has been set up");
557         return breakpRequest;
558     }

561     private void getEventSet()
562                  throws JDITestRuntimeException {
563         try {
564 //            log2("       eventSet = eventQueue.remove(waitTime);");
565             eventSet = eventQueue.remove(waitTime);
566             if (eventSet == null) {
567                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
568             }
569 //            log2("       eventIterator = eventSet.eventIterator;");
570             eventIterator = eventSet.eventIterator();
571         } catch ( Exception e ) {
572             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
573         }
574     }

577     private void breakpointForCommunication()
578                  throws JDITestRuntimeException {
579 
580         log2("breakpointForCommunication");
581         getEventSet();
582 
583         if (eventIterator.nextEvent() instanceof BreakpointEvent)
584             return;
585 
586         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
587     }

591     private MethodExitRequest settingMethodExitRequest ( ThreadReference thread,
592                                                          ReferenceType   testedClass,
593                                                          int             suspendPolicy,
594                                                          String          property        )
595             throws JDITestRuntimeException {
596         try {
597             log2("......setting up MethodExitRequest:");
598             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
599 
600             MethodExitRequest
601             mexr = eventRManager.createMethodExitRequest();
602             mexr.putProperty("number", property);
603             mexr.addThreadFilter(thread);
604             mexr.addClassFilter(testedClass);
605             mexr.setSuspendPolicy(suspendPolicy);
606 
607         log2("      MethodExitRequest has been set up");
608         return mexr;
609         } catch ( Exception e ) {
610             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
611             log3("       MethodExitRequest HAS NOT BEEN SET UP");
612             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
613         }

615     /**
616      * Inform debuggee which thread test the debugger has completed.
617      * Used for synchronization, so the debuggee does not move too quickly.
618      * @param testCase index of just completed test
619      */
620     void informDebuggeeTestCase(int testCase) {
621         try {
622             ((ClassType)debuggeeClass)
623                 .setValue(debuggeeClass.fieldByName("testCase"),
624                           vm.mirrorOf(testCase));
625         } catch (InvalidTypeException ite) {
626             throw new Failure("** FAILURE setting testCase  **");
627         } catch (ClassNotLoadedException cnle) {
628             throw new Failure("** FAILURE notifying debuggee  **");
629         } catch (VMDisconnectedException e) {
630             throw new Failure("** FAILURE debuggee connection **");
631         }
632     }
