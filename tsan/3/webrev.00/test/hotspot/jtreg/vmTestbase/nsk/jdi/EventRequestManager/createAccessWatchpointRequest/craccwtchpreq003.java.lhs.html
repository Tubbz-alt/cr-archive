 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createAccessWatchpointRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    - Multiple watchpoints on the same field are permitted.   <BR>
 47  *    - Creates a new disabled watchpoint.                      <BR>
 48  *    - The watchpoint watches accesses to the specified field. <BR>
 49  *    - Throws: UnsupportedOperationException -                 <BR>
 50  *              if the target virtual machine does not support  <BR>
 51  *              this operation.                                 <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee prepares testcase and invokes                         <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent,                                  <BR>
 74  *   the debugger performs the checks required.                         <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = threadByName("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         String fieldName = "testField";
310 
311         Field field = null;
312 
313         List         requests = null;
314         ListIterator li       = null;
315 
316         AccessWatchpointRequest awRequest1 = null;
317         AccessWatchpointRequest awRequest2 = null;
318 
319 
320         log1("     TESTING BEGINS");
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339             field = debuggeeClass.fieldByName(fieldName);
340 
341             if ( vm.canWatchFieldAccess() ) {
342                 log2("......vm.canWatchFieldAccess == true");
343                 log2("      creating two AccessWatchpointRequests for 'field': awRequest1 & awRequest2");
344 
345                 awRequest1 = eventRManager.createAccessWatchpointRequest(field);
346                 awRequest2 = eventRManager.createAccessWatchpointRequest(field);
347 
348                 log2("      checking up on: requests are disabled");
349                 if (awRequest1.isEnabled()) {
350                     testExitCode = FAILED;
351                     log3("ERROR: awRequest1 is enabled");
352                 }
353                 if (awRequest2.isEnabled()) {
354                     testExitCode = FAILED;
355                     log3("ERROR: awRequest2 is enabled");
356                 }
357                 log2("      compareing field equality");
358                 if ( !awRequest1.field().equals(field) ) {
359                     testExitCode = FAILED;
360                     log3("ERROR: !awRequest1.field().equals(field)");
361                 }
362                 if ( !awRequest2.field().equals(field) ) {
363                     testExitCode = FAILED;
364                     log3("ERROR: !awRequest2.field().equals(field)");
365                 }
366 
367             } else {
368                 log2("......vm.canWatchFieldAccess == false");
369                 log2("      UnsupportedOperationException is expected");
370                 try {
371                     awRequest1 = eventRManager.createAccessWatchpointRequest(field);
372                     testExitCode = FAILED;
373                     log3("ERROR: NO UnsupportedOperationException");
374                 } catch ( UnsupportedOperationException e ) {
375                 }
376             }
377 
378             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
379         }
380         log1("    TESTING ENDS");
381         return;
382     }

384     private ThreadReference threadByName(String name)
385                  throws JDITestRuntimeException {
386 
387         List         all = vm.allThreads();
388         ListIterator li  = all.listIterator();
389 
390         for (; li.hasNext(); ) {
391             ThreadReference thread = (ThreadReference) li.next();
392             if (thread.name().equals(name))
393                 return thread;
394         }
395         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
396     }

398    /*
399     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
400     *                                             String, String, String)
401     *
402     * It sets up a breakpoint at given line number within a given method in a given class
403     * for a given thread.
404     *
405     * Return value: BreakpointRequest object  in case of success
406     *
407     * JDITestRuntimeException   in case of an Exception thrown within the method
408     */
409 
410     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
411                                                   ReferenceType testedClass,
412                                                   String methodName,
413                                                   String bpLine,
414                                                   String property)
415             throws JDITestRuntimeException {
416 
417         log2("......setting up a breakpoint:");
418         log2("       thread: " + thread + "; class: " + testedClass +
419                         "; method: " + methodName + "; line: " + bpLine);
420 
421         List              alllineLocations = null;
422         Location          lineLocation     = null;
423         BreakpointRequest breakpRequest    = null;
424 
425         try {
426             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
427 
428             alllineLocations = method.allLineLocations();
429 
430             int n =
431                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
432             if (n > alllineLocations.size()) {
433                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
434             } else {
435                 lineLocation = (Location) alllineLocations.get(n);
436                 try {
437                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
438                     breakpRequest.putProperty("number", property);
439                     breakpRequest.addThreadFilter(thread);
440                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
441                 } catch ( Exception e1 ) {
442                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
443                     breakpRequest    = null;
444                 }
445             }
446         } catch ( Exception e2 ) {
447             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
448             breakpRequest    = null;
449         }
450 
451         if (breakpRequest == null) {
452             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
453             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
454         }
455 
456         log2("      a breakpoint has been set up");
457         return breakpRequest;
458     }

461     private void getEventSet()
462                  throws JDITestRuntimeException {
463         try {
464 //            log2("       eventSet = eventQueue.remove(waitTime);");
465             eventSet = eventQueue.remove(waitTime);
466             if (eventSet == null) {
467                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
468             }
469 //            log2("       eventIterator = eventSet.eventIterator;");
470             eventIterator = eventSet.eventIterator();
471         } catch ( Exception e ) {
472             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
473         }
474     }
