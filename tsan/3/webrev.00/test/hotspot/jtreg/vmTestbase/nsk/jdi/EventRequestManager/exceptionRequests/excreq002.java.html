 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.exceptionRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled exception requests.           <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

151     private int runThis (String argv[], PrintStream out) {
152 
153         argsHandler     = new ArgumentHandler(argv);
154         logHandler      = new Log(out, argsHandler);
155         Binder binder   = new Binder(argsHandler, logHandler);
156 
157         waitTime        = argsHandler.getWaitTime() * 60000;
158 
159         try {
160             log2("launching a debuggee :");
161             log2("       " + debuggeeName);
162             if (argsHandler.verbose()) {
163                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
164             } else {
165                 debuggee = binder.bindToDebugee(debuggeeName);
166             }
167             if (debuggee == null) {
168                 log3("ERROR: no debuggee launched");
169                 return FAILED;
170             }
171             log2("debuggee launched");
172         } catch ( Exception e ) {
173             log3("ERROR: Exception : " + e);
174             log2("       test cancelled");
175             return FAILED;
176         }
177 
178         debuggee.redirectOutput(logHandler);
179 
180         vm = debuggee.VM();
181 
182         eventQueue = vm.eventQueue();
183         if (eventQueue == null) {
184             log3("ERROR: eventQueue == null : TEST ABORTED");
185             vm.exit(PASS_BASE);
186             return FAILED;
187         }
188 
189         log2("invocation of the method runTest()");
190         switch (runTest()) {
191 
192             case 0 :  log2("test phase has finished normally");
193                       log2("   waiting for the debuggee to finish ...");
194                       debuggee.waitFor();
195 
196                       log2("......getting the debuggee's exit status");
197                       int status = debuggee.getStatus();
198                       if (status != PASS_BASE) {
199                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
200                               status + " != PASS_BASE");
201                           testExitCode = FAILED;
202                       } else {
203                           log2("......debuggee returned expected exit status: " +
204                               status + " == PASS_BASE");
205                       }
206                       break;
207 
208             default : log3("ERROR: runTest() returned unexpected value");
209 
210             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
211                       log2("......forcing: vm.exit();");
212                       testExitCode = FAILED;
213                       try {
214                           vm.exit(PASS_BASE);
215                       } catch ( Exception e ) {
216                           log3("ERROR: Exception : " + e);
217                       }
218                       break;
219 
220             case 2 :  log3("test cancelled due to VMDisconnectedException");
221                       log2("......trying: vm.process().destroy();");
222                       testExitCode = FAILED;
223                       try {
224                           Process vmProcess = vm.process();
225                           if (vmProcess != null) {
226                               vmProcess.destroy();
227                           }
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : " + e);
230                       }
231                       break;
232             }
233 
234         return testExitCode;
235     }

244     private int runTest() {
245 
246         try {
247             testRun();
248 
249             log2("waiting for VMDeathEvent");
250             getEventSet();
251             if (eventIterator.nextEvent() instanceof VMDeathEvent)
252                 return 0;
253 
254             log3("ERROR: last event is not the VMDeathEvent");
255             return 1;
256         } catch ( VMDisconnectedException e ) {
257             log3("ERROR: VMDisconnectedException : " + e);
258             return 2;
259         } catch ( Exception e ) {
260             log3("ERROR: Exception : " + e);
261             return 1;
262         }
263 
264     }

266     private void testRun()
267                  throws JDITestRuntimeException, Exception {
268 
269         eventRManager = vm.eventRequestManager();
270 
271         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
272         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
273         cpRequest.addClassFilter(debuggeeName);
274 
275         cpRequest.enable();
276         vm.resume();
277         getEventSet();
278         cpRequest.disable();
279 
280         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
281         debuggeeClass = event.referenceType();
282 
283         if (!debuggeeClass.name().equals(debuggeeName))
284            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
285 
286         log2("      received: ClassPrepareEvent for debuggeeClass");
287 
288         String bPointMethod = "methodForCommunication";
289         String lineForComm  = "lineForComm";
290 
291         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
292 
293         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
294                                              debuggeeClass,
295                                             bPointMethod, lineForComm, "zero");
296         bpRequest.enable();
297 
298     //------------------------------------------------------  testing section
299 
300         String fieldName = "testField";
301 
302         List         requests = null;
303         ListIterator li       = null;
304 
305         ExceptionRequest request = null;
306         ExceptionRequest cuRequests[] = { null, null, null, null, null,
307                                           null, null, null, null, null };
308         int listSize;
309         int flag;
310 
311 
312         log1("     TESTING BEGINS");
313 
314         for (int i = 0; ; i++) {
315 
316             vm.resume();
317             breakpointForCommunication();
318 
319             int instruction = ((IntegerValue)
320                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
321 
322             if (instruction == 0) {
323                 vm.resume();
324                 break;
325             }
326 
327             log1(":::::: case: # " + i);
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
330 
331             ReferenceType refType = (ReferenceType)
332                                     debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName)).type();
333 
334             log2("......creating ExceptionRequests");
335             for (int i1 = 0; i1 < cuRequests.length; i1++) {
336                 cuRequests[i1] = eventRManager.createExceptionRequest(refType, false, false);
337                 cuRequests[i1].putProperty("number", String.valueOf(i1));
338 
339                 log2("......checking up on returned List after creating new ExceptionRequest");
340                 requests = eventRManager.exceptionRequests();
341                 listSize = requests.size();
342                 if ( listSize != (i1 + 1) ) {
343                     testExitCode = FAILED;
344                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
345                 }
346                 flag = 0;
347                 li = requests.listIterator();
348                 while (li.hasNext()) {
349                     request = (ExceptionRequest) li.next();
350                     if ( !request.isEnabled() ) {
351                         flag++;
352                         if (flag > 1) {
353                             testExitCode = FAILED;
354                             log3("ERROR: # of disabled requests > 1 : " + flag);
355                         }
356                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
357                             testExitCode = FAILED;
358                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
359                         }
360                     } else {
361                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
362                             testExitCode = FAILED;
363                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
364                         }
365                     }
366                 }
367 
368                 log2("      enabling created ExceptionRequest");
369                 cuRequests[i1].enable();
370                 requests = eventRManager.exceptionRequests();
371                 listSize = requests.size();
372                 if ( listSize != (i1 + 1) ) {
373                     testExitCode = FAILED;
374                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
375                 }
376 
377                 li = requests.listIterator();
378                 while (li.hasNext()) {
379                     request = (ExceptionRequest) li.next();
380                     if ( !request.isEnabled() ) {
381                         testExitCode = FAILED;
382                         log3("ERROR: returned List contains disabled request : " + request);
383                     }
384                 }
385 
386                 log2("      removing item from the List; UnsupportedOperationException is expected");
387                 try {
388                     requests.remove(i1);
389                     testExitCode = FAILED;
390                     log3("ERROR: NO exception");
391                 } catch ( UnsupportedOperationException e ) {
392                     log2("        UnsupportedOperationException ");
393                 }
394             }
395 
396             log2("......deleting ExceptionRequests");
397             for (int i2 = cuRequests.length -1; i2 >= 0; i2--) {
398                 eventRManager.deleteEventRequest(cuRequests[i2]);
399                 requests = eventRManager.exceptionRequests();
400                 listSize = requests.size();
401                 if ( listSize != i2 ) {
402                     testExitCode = FAILED;
403                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " + i2);
404                 }
405                 log2("      removing item from the List; UnsupportedOperationException is expected");
406                 try {
407                     requests.remove(i2);
408                     testExitCode = FAILED;
409                     log3("ERROR: NO exception");
410                 } catch ( UnsupportedOperationException e ) {
411                     log2("        UnsupportedOperationException ");
412                 }
413             }
414 
415             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
416         }
417         log1("    TESTING ENDS");
418         return;
419     }

421    /*
422     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
423     *                                             String, String, String)
424     *
425     * It sets up a breakpoint at given line number within a given method in a given class
426     * for a given thread.
427     *
428     * Return value: BreakpointRequest object  in case of success
429     *
430     * JDITestRuntimeException   in case of an Exception thrown within the method
431     */
432 
433     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
434                                                   ReferenceType testedClass,
435                                                   String methodName,
436                                                   String bpLine,
437                                                   String property)
438             throws JDITestRuntimeException {
439 
440         log2("......setting up a breakpoint:");
441         log2("       thread: " + thread + "; class: " + testedClass +
442                         "; method: " + methodName + "; line: " + bpLine);
443 
444         List              alllineLocations = null;
445         Location          lineLocation     = null;
446         BreakpointRequest breakpRequest    = null;
447 
448         try {
449             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
450 
451             alllineLocations = method.allLineLocations();
452 
453             int n =
454                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
455             if (n > alllineLocations.size()) {
456                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
457             } else {
458                 lineLocation = (Location) alllineLocations.get(n);
459                 try {
460                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
461                     breakpRequest.putProperty("number", property);
462                     breakpRequest.addThreadFilter(thread);
463                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
464                 } catch ( Exception e1 ) {
465                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
466                     breakpRequest    = null;
467                 }
468             }
469         } catch ( Exception e2 ) {
470             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
471             breakpRequest    = null;
472         }
473 
474         if (breakpRequest == null) {
475             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
476             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
477         }
478 
479         log2("      a breakpoint has been set up");
480         return breakpRequest;
481     }

484     private void getEventSet()
485                  throws JDITestRuntimeException {
486         try {
487 //            log2("       eventSet = eventQueue.remove(waitTime);");
488             eventSet = eventQueue.remove(waitTime);
489             if (eventSet == null) {
490                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
491             }
492 //            log2("       eventIterator = eventSet.eventIterator;");
493             eventIterator = eventSet.eventIterator();
494         } catch ( Exception e ) {
495             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
496         }
497     }

500     private void breakpointForCommunication()
501                  throws JDITestRuntimeException {
502 
503         log2("breakpointForCommunication");
504         getEventSet();
505 
506         if (eventIterator.nextEvent() instanceof BreakpointEvent)
507             return;
508 
509         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
510     }
