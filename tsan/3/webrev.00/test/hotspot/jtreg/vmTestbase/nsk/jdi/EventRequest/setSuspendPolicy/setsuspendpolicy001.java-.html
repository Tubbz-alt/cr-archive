 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.setSuspendPolicy()</code>     <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on following assertions:                  <BR>
 46  * - a value of policy is equal to one assigned by last method  <BR>
 47  *   setSuspendPolicy(int);                                     <BR>
 48  * - InvalidRequestStateException is thrown                     <BR>
 49  *   if this request is currently enabled;                      <BR>
 50  * - InvalidRequestStateException is thrown                     <BR>
 51  *   if this request has been deleted.                          <BR>
 52  * The cases to check include Requests of all sub-types.        <BR>
 53  * <BR>
 54  * The test has three phases and works as follows.              <BR>
 55  * <BR>
 56  * In first phase,                                                      <BR>
 57  * upon launching debuggee's VM which will be suspended,                <BR>
 58  * a debugger waits for the VMStartEvent within a predefined            <BR>
 59  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 60  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 61  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 62  * and waits for the event within the predefined time interval.         <BR>
 63  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 64  * Upon getting the ClassPrepareEvent,                                  <BR>
 65  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 66  * within debuggee's special methodForCommunication().                  <BR>
 67  * <BR>
 68  * In second phase to check the assertion,                              <BR>
 69  * the debugger and the debuggee perform the following loop.            <BR>
 70  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 71  * - The debuggee prepares new check case (as if) and invokes           <BR>
 72  *   the methodForCommunication to be suspended and                     <BR>
 73  *   to inform the debugger with the event.                             <BR>
 74  * - Upon getting the BreakpointEvent,                                  <BR>
 75  *   the debugger performs the check case required.                     <BR>
 76  * <BR>
 77  * In third phase, at the end of the test, the debuggee changes         <BR>
 78  * the value of the "instruction" which the debugger and debuggee       <BR>
 79  * use to inform each other of needed actions,  and both end.           <BR>
 80  * <BR>
 81  */

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278 
279     }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315 
316         EventRequest  eventRequest1 = null;
317 
318         String        fieldName1    = "var101";
319         String        fieldName2    = "excObj";
320 
321         ThreadReference thread1     = null;
322         String          threadName1 = "thread1";
323 
324         ReferenceType testClassReference = null;
325 
326         String propertyName   = "name";
327         String propertyValue1 = "value1";
328         String propertyValue2 = "value2";
329 
330 
331         log1("     TESTING BEGINS");
332 
333         for (int i = 0; ; i++) {
334 
335             vm.resume();
336             breakpointForCommunication();
337 
338             int instruction = ((IntegerValue)
339                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
340 
341             if (instruction == 0) {
342                 vm.resume();
343                 break;
344             }
345 
346             log1(":::::: case: # " + i);
347 
348             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
349 
350 
351             switch (i) {
352 
353               case 0:
354                      thread1 = threadByName(threadName1);
355 
356                      log2("......setting up StepRequest");
357                      eventRequest1 = eventRManager.createStepRequest
358                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
359                      break;
360 
361               case 1:
362                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
363 
364                      log2("......setting up AccessWatchpointRequest");
365                      eventRequest1 = eventRManager.createAccessWatchpointRequest
366                                     (testClassReference.fieldByName(fieldName1));
367                      break;
368 
369               case 2:
370                      log2(".....setting up ModificationWatchpointRequest");
371                      eventRequest1 = eventRManager.createModificationWatchpointRequest
372                                     (testClassReference.fieldByName(fieldName1));
373                      break;
374 
375               case 3:
376                      log2(".....setting up ClassPrepareRequest");
377                      eventRequest1 = eventRManager.createClassPrepareRequest();
378                      break;
379 
380               case 4:
381                      log2(".....setting up ClassUnloadRequest");
382                      eventRequest1 = eventRManager.createClassUnloadRequest();
383                      break;
384 
385               case 5:
386                      log2(".....setting up MethodEntryRequest");
387                      eventRequest1 = eventRManager.createMethodEntryRequest();
388                      break;
389 
390               case 6:
391                      log2(".....setting up MethodExitRequest");
392                      eventRequest1 = eventRManager.createMethodExitRequest();
393                      break;
394 
395               case 7:
396                      log2(".....setting up ThreadDeathRequest");
397                      eventRequest1 = eventRManager.createThreadDeathRequest();
398                      break;

402                      eventRequest1 = eventRManager.createThreadStartRequest();

405               case 9:

471                 log2("     checking up on eventRequest1; InvalidRequestStateException is expected");
472                 eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
473                 testExitCode = FAILED;
474                 log3("ERROR: NO InvalidRequestStateException");
475             } catch ( InvalidRequestStateException e ) {
476                 log2("        InvalidRequestStateException");
477             }
478 
479             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
480         }
481         log1("    TESTING ENDS");
482         return;

485     private ThreadReference threadByName(String name)
486                  throws JDITestRuntimeException {
487 
488         List         all = vm.allThreads();
489         ListIterator li  = all.listIterator();
490 
491         for (; li.hasNext(); ) {
492             ThreadReference thread = (ThreadReference) li.next();
493             if (thread.name().equals(name))
494                 return thread;
495         }
496         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
497     }

499    /*
500     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
501     *                                             String, String, String)
502     *
503     * It sets up a breakpoint at given line number within a given method in a given class
504     * for a given thread.
505     *
506     * Return value: BreakpointRequest object  in case of success
507     *
508     * JDITestRuntimeException   in case of an Exception thrown within the method
509     */
510 
511     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
512                                                   ReferenceType testedClass,
513                                                   String methodName,
514                                                   String bpLine,
515                                                   String property)
516             throws JDITestRuntimeException {
517 
518         log2("......setting up a breakpoint:");
519         log2("       thread: " + thread + "; class: " + testedClass +
520                         "; method: " + methodName + "; line: " + bpLine);
521 
522         List              alllineLocations = null;
523         Location          lineLocation     = null;
524         BreakpointRequest breakpRequest    = null;
525 
526         try {
527             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
528 
529             alllineLocations = method.allLineLocations();
530 
531             int n =
532                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
533             if (n > alllineLocations.size()) {
534                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
535             } else {
536                 lineLocation = (Location) alllineLocations.get(n);
537 
538 // this is only for this test to get Location object
539 location = lineLocation;
540 
541                 try {
542                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
543                     breakpRequest.putProperty("number", property);
544                     breakpRequest.addThreadFilter(thread);
545                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
546                 } catch ( Exception e1 ) {
547                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
548                     breakpRequest    = null;
549                 }
550             }
551         } catch ( Exception e2 ) {
552             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
553             breakpRequest    = null;
554         }
555 
556         if (breakpRequest == null) {
557             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
558             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
559         }
560 
561         log2("      a breakpoint has been set up");
562         return breakpRequest;
563     }

566     private void getEventSet()
567                  throws JDITestRuntimeException {
568         try {
569 //            log2("       eventSet = eventQueue.remove(waitTime);");
570             eventSet = eventQueue.remove(waitTime);
571             if (eventSet == null) {
572                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
573             }
574 //            log2("       eventIterator = eventSet.eventIterator;");
575             eventIterator = eventSet.eventIterator();
576         } catch ( Exception e ) {
577             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
578         }
579     }
