 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ModificationWatchpointEvent.                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of methods in the interface     <BR>
 42  * <code>com.sun.jdi.ModificationWatchpointEvent</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    The number of ModificationWatchpointEvents received is    <BR>
 47  *    equal to the number of ModificationWatchpointRequests     <BR>
 48  *    set up by a debugger.                                     <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates an object foo of CheckedClass and invokes     <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger:                    <BR>
 71  *   - set up enabled ModificationWatchpointRequests for all fields,    <BR>
 72  *   - resumes the debuggee, and                                        <BR>
 73  *   - waiting for all expected events to receive.                      <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugee(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : " + e);
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : " + e);
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

249     * Return value: 0 - normal end of the test
250     *               1 - ubnormal end of the test
251     *               2 - VMDisconnectedException while test phase

254     private int runTest() {
255 
256         try {
257             testRun();
258 
259             log2("waiting for VMDeathEvent");
260             getEventSet();
261             if (eventIterator.nextEvent() instanceof VMDeathEvent)
262                 return 0;
263 
264             log3("ERROR: last event is not the VMDeathEvent");
265             return 1;
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273 
274     }

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         if ( !vm.canWatchFieldModification() ) {
280             log2("......vm.canWatchFieldModification == false :: test cancelled");
281             vm.exit(PASS_BASE);
282             return;
283         }
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = threadByName("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         ReferenceType refType = null;
319         List          fields  = null;
320         ListIterator  li      = null;
321         Field         field   = null;
322 
323         ModificationWatchpointRequest mwRequest = null;
324 
325         int requestsCount = 0;
326         int mwEventsCount = 0;
327 
328 
329         vm.resume();            //   !!!!!
330         for (int i = 0; ; i++) {
331 
332             breakpointForCommunication();
333 
334             int instruction = ((IntegerValue)
335                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
336 
337             if (instruction == 0) {
338                 vm.resume();
339                 break;
340             }
341 
342             log1(":::::: case: # " + i);
343 
344             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
345 
346             refType = (ReferenceType) vm.classesByName(testedClassName).get(0);
347             fields = refType.fields();
348             li = fields.listIterator();
349 
350             log2("......setting up ModificationWatchpointRequests");
351             while (li.hasNext()) {
352                 field = (Field) li.next();
353                 mwRequest = eventRManager.createModificationWatchpointRequest(field);
354                 mwRequest.setSuspendPolicy(EventRequest.SUSPEND_NONE);
355                 mwRequest.enable();
356                 requestsCount++;
357             }
358             log2("       # mwRequests set up : " + requestsCount);
359 
360             log2("......vm.resume();");
361             vm.resume();
362 
363             log2("......waiting for ModificationWatchpointEvents");
364             while ( mwEventsCount < requestsCount ) {
365                 getEventSet();
366                 Event newEvent = eventIterator.nextEvent();
367                 if ( !( newEvent instanceof ModificationWatchpointEvent)) {
368                     testExitCode = FAILED;
369                     log3("ERROR: new event is not ModificationWatchpointEvent");
370                     break;
371                 }
372                 mwEventsCount++;
373             }
374             log2("      # mwEventsCount == " + mwEventsCount);
375 
376             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
377         }
378         log1("    TESTING ENDS");
379         return;
380     }

382     private ThreadReference threadByName(String name)
383                  throws JDITestRuntimeException {
384 
385         List         all = vm.allThreads();
386         ListIterator li  = all.listIterator();
387 
388         for (; li.hasNext(); ) {
389             ThreadReference thread = (ThreadReference) li.next();
390             if (thread.name().equals(name))
391                 return thread;
392         }
393         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
394     }

396    /*
397     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
398     *                                             String, String, String)
399     *
400     * It sets up a breakpoint at given line number within a given method in a given class
401     * for a given thread.
402     *
403     * Return value: BreakpointRequest object  in case of success
404     *
405     * JDITestRuntimeException   in case of an Exception thrown within the method
406     */
407 
408     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
409                                                   ReferenceType testedClass,
410                                                   String methodName,
411                                                   String bpLine,
412                                                   String property)
413             throws JDITestRuntimeException {
414 
415         log2("......setting up a breakpoint:");
416         log2("       thread: " + thread + "; class: " + testedClass +
417                         "; method: " + methodName + "; line: " + bpLine);
418 
419         List              alllineLocations = null;
420         Location          lineLocation     = null;
421         BreakpointRequest breakpRequest    = null;
422 
423         try {
424             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
425 
426             alllineLocations = method.allLineLocations();
427 
428             int n =
429                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
430             if (n > alllineLocations.size()) {
431                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
432             } else {
433                 lineLocation = (Location) alllineLocations.get(n);
434                 try {
435                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
436                     breakpRequest.putProperty("number", property);
437                     breakpRequest.addThreadFilter(thread);
438                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
439                 } catch ( Exception e1 ) {
440                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
441                     breakpRequest    = null;
442                 }
443             }
444         } catch ( Exception e2 ) {
445             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
446             breakpRequest    = null;
447         }
448 
449         if (breakpRequest == null) {
450             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
451             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
452         }
453 
454         log2("      a breakpoint has been set up");
455         return breakpRequest;
456     }

459     private void getEventSet()
460                  throws JDITestRuntimeException {
461         try {
462 //            log2("       eventSet = eventQueue.remove(waitTime);");
463             eventSet = eventQueue.remove(waitTime);
464             if (eventSet == null) {
465                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
466             }
467 //            log2("       eventIterator = eventSet.eventIterator;");
468             eventIterator = eventSet.eventIterator();
469         } catch ( Exception e ) {
470             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
471         }
472     }

475     private void breakpointForCommunication()
476                  throws JDITestRuntimeException {
477 
478         log2("breakpointForCommunication");
479         getEventSet();
480 
481         if (eventIterator.nextEvent() instanceof BreakpointEvent)
482             return;
483 
484         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
