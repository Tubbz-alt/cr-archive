 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventQueue.                                                  <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.EventQueue.remove(int)</code>              <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks two assertions:                              <BR>
 46  * - Returns: null if there is a timeout.                       <BR>
 47  * - Returns: the next EventSet if no timeout.                  <BR>
 48  * It uses the WAITTINE value which is the standard parameter for<BR>
 49  * time intervals in the testbase_nsk.                          <BR>
 50  * <BR>
 51  * The test works as follows.                                   <BR>
 52  * Upon launching debuggee's VM which will be suspended,        <BR>
 53  * a debugger waits for the VMStartEvent within a predefined    <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.<BR>
 55  * Upon getting the VMStartEvent, it makes the request          <BR>
 56  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,  <BR>
 57  * resumes the VM, and waits for the event within the predefined<BR>
 58  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 59  * Upon getting the ClassPrepareEvent,                          <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,<BR>
 61  * the debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * The debuggee firstly invokes the methodForCommunication              <BR>
 63  * to be suspended and to inform the debugger with the event.           <BR>
 64  * The case for testing consists of of two steps.                       <BR>
 65  * In the first one the first assertion is checked up on as follows:    <BR>
 66  * the debugger sleeps for some time;                                   <BR>
 67  * hence, no event is expected in the debugger within WAITTIME, and     <BR>
 68  * debugger's method breakpointForCommunication() should get null.      <BR>
 69  * After WAITTIME, the debugger just expects to get normal breakpoint event.<BR>
 70  */
 71

151     private int runThis (String argv[], PrintStream out) {
152 
153         argsHandler     = new ArgumentHandler(argv);
154         logHandler      = new Log(out, argsHandler);
155         Binder binder   = new Binder(argsHandler, logHandler);
156 
157         waitTime        = Utils.adjustTimeout(argsHandler.getWaitTime() * 1000);
158 
159         try {
160             log2("launching a debuggee :");
161             log2("       " + debuggeeName);
162             if (argsHandler.verbose()) {
163                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
164             } else {
165                 debuggee = binder.bindToDebugee(debuggeeName);
166             }
167             if (debuggee == null) {
168                 log3("ERROR: no debuggee launched");
169                 return FAILED;
170             }
171             log2("debuggee launched");
172         } catch ( Exception e ) {
173             log3("ERROR: Exception : " + e);
174             log2("       test cancelled");
175             return FAILED;
176         }
177 
178         debuggee.redirectOutput(logHandler);
179 
180         vm = debuggee.VM();
181 
182         eventQueue = vm.eventQueue();
183         if (eventQueue == null) {
184             log3("ERROR: eventQueue == null : TEST ABORTED");
185             vm.exit(PASS_BASE);
186             return FAILED;
187         }
188 
189         log2("invocation of the method runTest()");
190         switch (runTest()) {
191 
192             case 0 :  log2("test phase has finished normally");
193                       log2("   waiting for the debuggee to finish ...");
194                       debuggee.waitFor();
195 
196                       log2("......getting the debuggee's exit status");
197                       int status = debuggee.getStatus();
198                       if (status != PASS_BASE) {
199                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
200                               status + " != PASS_BASE");
201                           testExitCode = FAILED;
202                       } else {
203                           log2("......debuggee returned expected exit status: " +
204                               status + " == PASS_BASE");
205                       }
206                       break;
207 
208             default : log3("ERROR: runTest() returned unexpected value");
209 
210             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
211                       log2("......forcing: vm.exit();");
212                       testExitCode = FAILED;
213                       try {
214                           vm.exit(PASS_BASE);
215                       } catch ( Exception e ) {
216                           log3("ERROR: Exception : e");
217                       }
218                       break;
219 
220             case 2 :  log3("test cancelled due to VMDisconnectedException");
221                       log2("......trying: vm.process().destroy();");
222                       testExitCode = FAILED;
223                       try {
224                           Process vmProcess = vm.process();
225                           if (vmProcess != null) {
226                               vmProcess.destroy();
227                           }
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232             }
233 
234         return testExitCode;
235     }

238    /*
239     * Return value: 0 - normal end of the test
240     *               1 - ubnormal end of the test
241     *               2 - VMDisconnectedException while test phase
242     */

244     private int runTest() {
245 
246         try {
247             testRun();
248 
249             log2("waiting for VMDeathEvent");
250             getEventSet();
251             if (eventIterator.nextEvent() instanceof VMDeathEvent)
252                 return 0;
253 
254             log3("ERROR: last event is not the VMDeathEvent");
255             return 1;
256         } catch ( VMDisconnectedException e ) {
257             log3("ERROR: VMDisconnectedException : " + e);
258             return 2;
259         } catch ( Exception e ) {
260             log3("ERROR: Exception : " + e);
261             return 1;
262         }
263 
264     }

266     private void testRun()
267                  throws JDITestRuntimeException, Exception {
268 
269         eventRManager = vm.eventRequestManager();
270 
271         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
272         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
273         cpRequest.addClassFilter(debuggeeName);
274 
275         cpRequest.enable();
276         vm.resume();
277         getEventSet();
278         cpRequest.disable();
279 
280         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
281         debuggeeClass = event.referenceType();
282 
283         if (!debuggeeClass.name().equals(debuggeeName))
284            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
285 
286         trueValue = debuggeeClass.getValue(debuggeeClass.fieldByName("BOOLEAN_TRUE_VALUE"));
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292         BreakpointRequest bpRequest;
293 
294         try {
295             bpRequest = settingBreakpoint(threadByName("main"),
296                                           debuggeeClass,
297                                           bPointMethod, lineForComm, "zero");
298         } catch ( Exception e ) {
299             throw e;
300         }
301         bpRequest.enable();
302 
303     //------------------------------------------------------  testing section
304 
305         log1("     TESTING BEGINS");
306 
307 
308             vm.resume();
309             breakpointForCommunication();
310 
311             int instruction = ((IntegerValue)
312                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
313 
314             if (instruction != 1) {
315                 testExitCode = FAILED;
316                 throw new JDITestRuntimeException("** unexpected instruction **");
317             }
318 
319             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
320 
321             vm.resume();
322             try {
323                 log2("......invoking: getEventSet();");
324                 getEventSet();
325             } catch ( JDITestRuntimeException e ) {
326                 String str = e.getMessage();
327                 log2("      JDITestRuntimeException : " + str);
328                 if (str.lastIndexOf("TIMEOUT") <0) {
329                     log3("ERROR: no TIMEOUT");
330                     testExitCode = FAILED;
331                     throw new JDITestRuntimeException("** unexpected Exception **");
332                 }
333 
334                 // Signal to debuggee to stop sleeping
335                 ((ClassType) debuggeeClass).setValue(debuggeeClass.fieldByName("stopSleeping"),
336                         trueValue);
337 
338             }
339             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
340 
341             log2("...... waiting for breakpoint event");
342 
343             breakpointForCommunication();
344 
345             instruction = ((IntegerValue)
346                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
347             if (instruction != 0) {
348                 testExitCode = FAILED;
349                 throw new JDITestRuntimeException("** unexpected instruction **");
350             }
351             vm.resume();
352 
353         log1("    TESTING ENDS");
354         return;
355     }

357     private ThreadReference threadByName(String name)
358                  throws JDITestRuntimeException {
359 
360         List         all = vm.allThreads();
361         ListIterator li  = all.listIterator();
362 
363         for (; li.hasNext(); ) {
364             ThreadReference thread = (ThreadReference) li.next();
365             if (thread.name().equals(name))
366                 return thread;
367         }
368         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
369     }

371    /*
372     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
373     *                                             String, String, String)
374     *
375     * It sets up a breakpoint at given line number within a given method in a given class
376     * for a given thread.
377     *
378     * Return value: BreakpointRequest object  in case of success
379     *
380     * JDITestRuntimeException   in case of an Exception thrown within the method
381     */
382 
383     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
384                                                   ReferenceType testedClass,
385                                                   String methodName,
386                                                   String bpLine,
387                                                   String property)
388             throws JDITestRuntimeException {
389 
390         log2("......setting up a breakpoint:");
391         log2("       thread: " + thread + "; class: " + testedClass +
392                         "; method: " + methodName + "; line: " + bpLine);
393 
394         List              alllineLocations = null;
395         Location          lineLocation     = null;
396         BreakpointRequest breakpRequest    = null;
397 
398         try {
399             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
400 
401             alllineLocations = method.allLineLocations();
402 
403             int n =
404                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
405             if (n > alllineLocations.size()) {
406                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
407             } else {
408                 lineLocation = (Location) alllineLocations.get(n);
409                 try {
410                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
411                     breakpRequest.putProperty("number", property);
412                     breakpRequest.addThreadFilter(thread);
413                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
414                 } catch ( Exception e1 ) {
415                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
416                     breakpRequest    = null;
417                 }
418             }
419         } catch ( Exception e2 ) {
420             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
421             breakpRequest    = null;
422         }
423 
424         if (breakpRequest == null) {
425             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
426             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
427         }
428 
429         log2("      a breakpoint has been set up");
430         return breakpRequest;
431     }

434     private void getEventSet()
435                  throws JDITestRuntimeException {
436         try {
437 //            log2("       eventSet = eventQueue.remove(waitTime);");
438             eventSet = eventQueue.remove(waitTime);
439             if (eventSet == null) {
440                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
441             }
442 //            log2("       eventIterator = eventSet.eventIterator;");
443             eventIterator = eventSet.eventIterator();
444         } catch ( Exception e ) {
445             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
446         }
447     }

450     private void breakpointForCommunication()
451                  throws JDITestRuntimeException {
452 
453         log2("breakpointForCommunication");
454         getEventSet();
455 
456         if (eventIterator.nextEvent() instanceof BreakpointEvent)
457             return;
458 
459         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
460     }
