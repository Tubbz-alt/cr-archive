 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VMDeathEvent.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that the interface                           <BR>
 42  * <code>com.sun.jdi.VMDeathEvent</code>                        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The case to test includes receiving the VMDeathEvent         <BR>
 46  * as a response to VirtualMachine.exit(int);                   <BR>
 47  * <BR>
 48  * The test works as follows.                                   <BR>
 49  * <BR>
 50  * Upon launching debuggee's VM which will be suspended,                <BR>
 51  * a debugger waits for the VMStartEvent within a predefined            <BR>
 52  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 53  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 54  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 55  * and waits for the event within the predefined time interval.         <BR>
 56  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 57  * Upon getting the ClassPrepareEvent,                                  <BR>
 58  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 59  * within debuggee's special methodForCommunication().                  <BR>
 60  * <BR>
 61  * Then to test the interface,                                          <BR>
 62  * the debugger invokes the method VirtualMachine.exit() and checks up  <BR>
 63  * on the event received in the response to.                            <BR>
 64  * <BR>
 65  */

145     private int runThis (String argv[], PrintStream out) {
146 
147         argsHandler     = new ArgumentHandler(argv);
148         logHandler      = new Log(out, argsHandler);
149         Binder binder   = new Binder(argsHandler, logHandler);
150 
151         waitTime        = argsHandler.getWaitTime() * 60000;
152 
153         try {
154             log2("launching a debuggee :");
155             log2("       " + debuggeeName);
156             if (argsHandler.verbose()) {
157                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
158             } else {
159                 debuggee = binder.bindToDebugee(debuggeeName);
160             }
161             if (debuggee == null) {
162                 log3("ERROR: no debuggee launched");
163                 return FAILED;
164             }
165             log2("debuggee launched");
166         } catch ( Exception e ) {
167             log3("ERROR: Exception : " + e);
168             log2("       test cancelled");
169             return FAILED;
170         }
171 
172         debuggee.redirectOutput(logHandler);
173 
174         vm = debuggee.VM();
175 
176         eventQueue = vm.eventQueue();
177         if (eventQueue == null) {
178             log3("ERROR: eventQueue == null : TEST ABORTED");
179             vm.exit(PASS_BASE);
180             return FAILED;
181         }
182 
183         log2("invocation of the method runTest()");
184         switch (runTest()) {
185 
186             case 0 :  log2("test phase has finished normally");
187                       log2("   waiting for the debuggee to finish ...");
188                       debuggee.waitFor();
189 
190                       log2("......getting the debuggee's exit status");
191                       int status = debuggee.getStatus();
192                       if (status != PASS_BASE) {
193                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
194                               status + " != PASS_BASE");
195                           testExitCode = FAILED;
196                       } else {
197                           log2("......debuggee returned expected exit status: " +
198                               status + " == PASS_BASE");
199                       }
200                       break;
201 
202             default : log3("ERROR: runTest() returned unexpected value");
203 
204             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
205                       log2("......forcing: vm.exit();");
206                       testExitCode = FAILED;
207                       try {
208                           vm.exit(PASS_BASE);
209                       } catch ( Exception e ) {
210                           log3("ERROR: Exception : " + e);
211                       }
212                       break;
213 
214             case 2 :  log3("test cancelled due to VMDisconnectedException");
215                       log2("......trying: vm.process().destroy();");
216                       testExitCode = FAILED;
217                       try {
218                           Process vmProcess = vm.process();
219                           if (vmProcess != null) {
220                               vmProcess.destroy();
221                           }
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : " + e);
224                       }
225                       break;
226             }
227 
228         return testExitCode;
229     }

233     * Return value: 0 - normal end of the test
234     *               1 - ubnormal end of the test
235     *               2 - VMDisconnectedException while test phase

238     private int runTest() {
239 
240         try {
241             testRun();
242 
243             log2("waiting for VMDeathEvent");
244             getEventSet();
245             Event event = eventIterator.nextEvent();
246             if ( !(event instanceof VMDeathEvent) ) {
247                 testExitCode = FAILED;
248                 log3("ERROR: last event is not the VMDeathEvent");
249 
250                 if (event instanceof VMDisconnectEvent)
251                    log3("      event is the VMDisconnectEvent     ????");
252                 else
253                    log3("      event is NOT the VMDisconnectEvent ????!!!!!!!");
254                 return 2;
255 
256             }
257             return 0;
258         } catch ( VMDisconnectedException e ) {
259             log3("ERROR: VMDisconnectedException : " + e);
260             return 2;
261         } catch ( Exception e ) {
262             log3("ERROR: Exception : " + e);
263             return 1;
264         }
265 
266     }

268     private void testRun()
269                  throws JDITestRuntimeException, Exception {
270 
271         eventRManager = vm.eventRequestManager();
272 
273         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
274         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
275         cpRequest.addClassFilter(debuggeeName);
276 
277         cpRequest.enable();
278         vm.resume();
279         getEventSet();
280         cpRequest.disable();
281 
282         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
283         debuggeeClass = event.referenceType();
284 
285         if (!debuggeeClass.name().equals(debuggeeName))
286            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292         BreakpointRequest bpRequest;
293 
294         bpRequest = settingBreakpoint(threadByName("main"),
295                                       debuggeeClass,
296                                       bPointMethod, lineForComm, "zero");
297         bpRequest.enable();
298 
299     //------------------------------------------------------  testing section
300 
301         log1("     TESTING BEGINS");
302 
303         for (int i = 0; ; i++) {
304 
305             vm.resume();
306             breakpointForCommunication();
307 
308             int instruction = ((IntegerValue)
309                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
310 
311             if (instruction == 0) {
312                 vm.resume();
313                 break;
314             }
315 
316             log1(":::::: case: # " + i);
317 
318             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
319 
320             //vm.eventRequestManager().createVMDeathRequest().enable();
321 
322             log2("......vm.exit(PASS_BASE);  :: VMDeathEvent is excpected");
323 
324             //vm.exit(PASS_BASE);
325             break;
326 
327             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
328         }
329         log1("    TESTING ENDS");
330         return;
331     }

333     private ThreadReference threadByName(String name)
334                  throws JDITestRuntimeException {
335 
336         List         all = vm.allThreads();
337         ListIterator li  = all.listIterator();
338 
339         for (; li.hasNext(); ) {
340             ThreadReference thread = (ThreadReference) li.next();
341             if (thread.name().equals(name))
342                 return thread;
343         }
344         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
345     }

347    /*
348     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
349     *                                             String, String, String)
350     *
351     * It sets up a breakpoint at given line number within a given method in a given class
352     * for a given thread.
353     *
354     * Return value: BreakpointRequest object  in case of success
355     *
356     * JDITestRuntimeException   in case of an Exception thrown within the method
357     */
358 
359     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
360                                                   ReferenceType testedClass,
361                                                   String methodName,
362                                                   String bpLine,
363                                                   String property)
364             throws JDITestRuntimeException {
365 
366         log2("......setting up a breakpoint:");
367         log2("       thread: " + thread + "; class: " + testedClass +
368                         "; method: " + methodName + "; line: " + bpLine);
369 
370         List              alllineLocations = null;
371         Location          lineLocation     = null;
372         BreakpointRequest breakpRequest    = null;
373 
374         try {
375             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
376 
377             alllineLocations = method.allLineLocations();
378 
379             int n =
380                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
381             if (n > alllineLocations.size()) {
382                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
383             } else {
384                 lineLocation = (Location) alllineLocations.get(n);
385                 try {
386                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
387                     breakpRequest.putProperty("number", property);
388                     breakpRequest.addThreadFilter(thread);
389                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
390                 } catch ( Exception e1 ) {
391                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
392                     breakpRequest    = null;
393                 }
394             }
395         } catch ( Exception e2 ) {
396             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
397             breakpRequest    = null;
398         }
399 
400         if (breakpRequest == null) {
401             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
402             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
403         }
404 
405         log2("      a breakpoint has been set up");
406         return breakpRequest;
407     }
