 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a ClassPrepareEvent set,            <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * The debugger then prepares an array containing sets of               <BR>
 67  * enabled requests for each combination to check.                      <BR>
 68  * <BR>
 69  * In second phase to check the above,                                  <BR>
 70  * the debugger and the debuggee perform the following.                 <BR>
 71  * - The debugger resumes the debuggee and waits for ClassPrepareEvents.<BR>
 72  * - The debuggee makes new checked classes prepared and invokes then   <BR>
 73  *   the methodForCommunication to be suspended and                     <BR>
 74  *   to inform the debugger with the event.                             <BR>
 75  * - Upon getting the ClassPrepareEvents,                               <BR>
 76  *   the debugger performs the check required.                          <BR>
 77  * Note. To inform each other of needed actions, the debugger and       <BR>
 78  *       and the debuggee use debuggee's variable "instruction".        <BR>
 79  * <BR>
 80  * In third phase when at the end,                                      <BR>
 81  * the debuggee changes the value of the "instruction"                  <BR>
 82  * to inform the debugger of checks finished, and both end.             <BR>
 83  * <BR>
 84  */
 85

108     public static int run (String argv[], PrintStream out) {
109 
110         int exitCode = new suspendpolicy001().runThis(argv, out);
111 
112         if (exitCode != PASSED) {
113             System.out.println("TEST FAILED");
114         }
115         return testExitCode;
116     }

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : e");
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : e");
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
259                 log3("ERROR: last event is not the VMDeathEvent");
260                 return 1;
261             }
262 
263             log2("waiting for VMDisconnectEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
266                 log3("ERROR: last event is not the VMDisconnectEvent");
267                 return 1;
268             }
269 
270             return 0;
271 
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279 
280     }

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         log2("......getting ClassPrepareEvent for debuggee's class");
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
304 
305         String            bPointMethod = "methodForCommunication";
306         String            lineForComm  = "lineForComm";
307         BreakpointRequest bpRequest;
308         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
309         bpRequest = settingBreakpoint(mainThread,
310                                       debuggeeClass,
311                                       bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         int classFlags = 0;
319 
320         String classNames[] = {
321             "nsk.jdi.EventSet.suspendPolicy.TestClass0",
322             "nsk.jdi.EventSet.suspendPolicy.TestClass1",
323             "nsk.jdi.EventSet.suspendPolicy.TestClass2",
324             "nsk.jdi.EventSet.suspendPolicy.TestClass3",
325             "nsk.jdi.EventSet.suspendPolicy.TestClass4",
326             "nsk.jdi.EventSet.suspendPolicy.TestClass5",
327             "nsk.jdi.EventSet.suspendPolicy.TestClass6"
328         };
329 
330         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
331         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
332         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
333         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
334 
335         int policyExpected[] = {    SUSPEND_NONE,
336                                     SUSPEND_THREAD,
337                                     SUSPEND_ALL,
338                                     SUSPEND_THREAD,
339                                     SUSPEND_ALL,
340                                     SUSPEND_ALL,
341                                     SUSPEND_ALL      };
342         int policy = 0;
343 
344         EventRequest eventRequests[][] = new EventRequest[classNames.length][3];
345 
346         for ( int nr = 0; nr < eventRequests.length; nr++) {
347             log2("......setting up ClassPrepareRequest for TestClass" + nr);
348             switch (nr) {
349 
350               case 0:
351                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
352                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
353                       eventRequests[nr][0].enable();
354                       break;
355 
356               case 1:
357                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
358                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
359                       eventRequests[nr][1].enable();
360                       break;
361 
362               case 2:
363                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
364                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
365                       eventRequests[nr][2].enable();
366                       break;
367 
368               case 3:
369                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
370                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
371                       eventRequests[nr][0].enable();
372                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
373                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
374                       eventRequests[nr][1].enable();
375                       break;
376 
377               case 4:
378                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
379                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
380                       eventRequests[nr][0].enable();
381                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
382                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
383                       eventRequests[nr][2].enable();
384                       break;
385 
386               case 5:
387                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
388                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
389                       eventRequests[nr][1].enable();
390                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
391                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
392                       eventRequests[nr][2].enable();
393                       break;
394 
395               case 6:
396                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
397                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
398                       eventRequests[nr][0].enable();
399                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
400                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
401                       eventRequests[nr][1].enable();
402                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
403                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
404                       eventRequests[nr][2].enable();
405                       break;
406 
407               default:
408                       throw new JDITestRuntimeException("** default case 2 **");
409             }
410             classFlags |= 1 << nr;
411         }
412 
413         log2("......checking up on eventSets");
414 
415         vm.resume();
416 
417         for (int ne = 0; ne < eventRequests.length; ne++) {
418 
419             getEventSet();
420 
421             Event ev = eventIterator.nextEvent();
422             if ( !( ev instanceof ClassPrepareEvent)) {
423                  log3("ERROR: new event is not ClassPrepareEvent");
424                  testExitCode = FAILED;
425                  break;
426             }
427 
428             String property = (String) ev.request().getProperty("number");
429 
430             int np = 0;
431             for (; np < eventRequests.length; np++) {
432                 if (property.equals("ClassPrepareRequest" + np))
433                     break;
434             }
435 
436             policy = eventSet.suspendPolicy();
437 
438             if (policy != policyExpected[np]) {
439                 log3("ERROR: eventSet.suspendPolicy() != policyExpected");
440                 log3("       eventSet.suspendPolicy() == " + policy);
441                 log3("       policyExpected           == " + policyExpected[np]);
442                 testExitCode = FAILED;
443             }
444 
445             switch (np) {
446               case 0: eventRequests[np][0].disable();                                 break;
447               case 1: eventRequests[np][1].disable();                                 break;
448               case 2: eventRequests[np][2].disable();                                 break;
449               case 3: eventRequests[np][0].disable(); eventRequests[np][1].disable(); break;
450               case 4: eventRequests[np][0].disable(); eventRequests[np][2].disable(); break;
451               case 5: eventRequests[np][1].disable(); eventRequests[np][2].disable(); break;
452               case 6: eventRequests[np][0].disable(); eventRequests[np][1].disable();
453                       eventRequests[np][2].disable();                                 break;
454            }
455 
456             switch (policy) {
457               case SUSPEND_NONE   :                      break;
458               case SUSPEND_THREAD : mainThread.resume(); break;
459               case SUSPEND_ALL    : vm.resume();         break;
460               default: throw new JDITestRuntimeException("** default case 1 **");
461             }
462             classFlags &= 1 << ne;
463         }
464         if (classFlags != 0) {
465             log3("ERROR: not all events received : " + classFlags);
466             testExitCode = FAILED;
467         }
468 
469         for (int i = 0; ; i++) {
470 
471             log1(":::::: case: # " + i);
472 
473             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
474 
475             breakpointForCommunication();
476 
477             int instruction = ((IntegerValue)
478                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
479             if (instruction == 0) {
480                 vm.resume();
481                 break;
482             }
483             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
484         }
485         log1("    TESTING ENDS");
486         return;
487     }

489    /*
490     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
491     *                                             String, String, String)
492     *
493     * It sets up a breakpoint at given line number within a given method in a given class
494     * for a given thread.
495     *
496     * Return value: BreakpointRequest object  in case of success
497     *
498     * JDITestRuntimeException   in case of an Exception thrown within the method
499     */
500 
501     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
502                                                   ReferenceType testedClass,
503                                                   String methodName,
504                                                   String bpLine,
505                                                   String property)
506             throws JDITestRuntimeException {
507 
508         log2("......setting up a breakpoint:");
509         log2("       thread: " + thread + "; class: " + testedClass +
510                         "; method: " + methodName + "; line: " + bpLine);
511 
512         List              alllineLocations = null;
513         Location          lineLocation     = null;
514         BreakpointRequest breakpRequest    = null;
515 
516         try {
517             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
518 
519             alllineLocations = method.allLineLocations();
520 
521             int n =
522                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
523             if (n > alllineLocations.size()) {
524                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
525             } else {
526                 lineLocation = (Location) alllineLocations.get(n);
527                 try {
528                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
529                     breakpRequest.putProperty("number", property);
530                     breakpRequest.addThreadFilter(thread);
531                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
532                 } catch ( Exception e1 ) {
533                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
534                     breakpRequest    = null;
535                 }
536             }
537         } catch ( Exception e2 ) {
538             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
539             breakpRequest    = null;
540         }
541 
542         if (breakpRequest == null) {
543             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
544             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
545         }
546 
547         log2("      a breakpoint has been set up");
548         return breakpRequest;
549     }

552     private void getEventSet()
553                  throws JDITestRuntimeException {
554         try {
555 //            log2("       eventSet = eventQueue.remove(waitTime);");
556             eventSet = eventQueue.remove(waitTime);
557             if (eventSet == null) {
558                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
559             }
560 //            log2("       eventIterator = eventSet.eventIterator;");
561             eventIterator = eventSet.eventIterator();
562         } catch ( Exception e ) {
563             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
564         }
565     }

568     private void breakpointForCommunication()
569                  throws JDITestRuntimeException {
570 
571         log2("breakpointForCommunication");
572         getEventSet();
573 
574         if (eventIterator.nextEvent() instanceof BreakpointEvent)
575             return;
576 
577         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
578     }

582     private ClassPrepareRequest settingClassPrepareRequest ( String  testedClass,
583                                                              int     suspendPolicy,
584                                                              String  property       )
585             throws JDITestRuntimeException {
586         try {
587             log2("......setting up ClassPrepareRequest:");
588             log2("       class: " + testedClass + "; property: " + property);
589 
590             ClassPrepareRequest
591             cpr = eventRManager.createClassPrepareRequest();
592             cpr.putProperty("number", property);
593             cpr.addClassFilter(testedClass);
594             cpr.setSuspendPolicy(suspendPolicy);
595 
596             log2("      ClassPrepareRequest has been set up");
597             return cpr;
598         } catch ( Exception e ) {
599             log3("ERROR: ATTENTION: Exception within settingClassPrepareRequest() : " + e);
600             log3("       ClassPreparenRequest HAS NOT BEEN SET UP");
601             throw new JDITestRuntimeException("** FAILURE to set up ClassPrepareRequest **");
602         }
603     }
