 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.addClassFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates and starts a thread, thread1, and             <BR>
 68  *   invokes the methodForCommunication to be suspended                 <BR>
 69  *   and to inform the debugger with the event.                         <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a StepRequest and enables it,                            <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */
 84

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest  eventRequest1      = null;
318 
319         String        property1          = "StepRequest1";
320 
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335 
336             log1(":::::: case: # " + i);
337 
338             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
339 
340             switch (i) {
341 
342               case 0:
343                      ThreadReference thread1 = threadByName("thread1");
344                      eventRequest1 = setting23StepRequest(thread1, testedClassName1,
345                                               EventRequest.SUSPEND_NONE, property1);
346 
347                      eventRequest1.enable();
348 
349                      try {
350                          log2("......eventRequest1.addClassFilter(testedClassName1);");
351                          log2("        InvalidRequestStateException expected");
352                          ((StepRequest)eventRequest1).addClassFilter(testedClassName1);
353                          log3("ERROR: no InvalidRequestStateException");
354                          testExitCode = FAILED;
355                      } catch ( InvalidRequestStateException e ) {
356                          log2("        InvalidRequestStateException");
357                      }
358 
359                      log2("......eventRequest1.disable();");
360                      eventRequest1.disable();
361                      try {
362                          log2("......eventRequest1.addClassFilter(testedClassName1);");
363                          log2("        no InvalidRequestStateException expected");
364                          ((StepRequest)eventRequest1).addClassFilter(testedClassName1);
365                          log2("        no InvalidRequestStateException");
366                      } catch ( InvalidRequestStateException e ) {
367                          log3("ERROR: InvalidRequestStateException");
368                          testExitCode = FAILED;
369                      }
370 
371                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
372                      eventRManager.deleteEventRequest(eventRequest1);
373                      try {
374                          log2("......eventRequest1.addClassFilter(testedClassName1);");
375                          log2("        InvalidRequestStateException expected");
376                          ((StepRequest)eventRequest1).addClassFilter(testedClassName1);
377                          log3("ERROR: no InvalidRequestStateException");
378                          testExitCode = FAILED;
379                      } catch ( InvalidRequestStateException e ) {
380                          log2("        InvalidRequestStateException");
381                      }
382                       break;
383 
384               default:
385                       throw new JDITestRuntimeException("** default case 2 **");
386             }
387 
388             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
389         }
390         log1("    TESTING ENDS");
391         return;
392     }

394     private ThreadReference threadByName(String name)
395                  throws JDITestRuntimeException {
396 
397         List         all = vm.allThreads();
398         ListIterator li  = all.listIterator();
399 
400         for (; li.hasNext(); ) {
401             ThreadReference thread = (ThreadReference) li.next();
402             if (thread.name().equals(name))
403                 return thread;
404         }
405         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
406     }

408    /*
409     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
410     *                                             String, String, String)
411     *
412     * It sets up a breakpoint at given line number within a given method in a given class
413     * for a given thread.
414     *
415     * Return value: BreakpointRequest object  in case of success
416     *
417     * JDITestRuntimeException   in case of an Exception thrown within the method
418     */
419 
420     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
421                                                   ReferenceType testedClass,
422                                                   String methodName,
423                                                   String bpLine,
424                                                   String property)
425             throws JDITestRuntimeException {
426 
427         log2("......setting up a breakpoint:");
428         log2("       thread: " + thread + "; class: " + testedClass +
429                         "; method: " + methodName + "; line: " + bpLine);
430 
431         List              alllineLocations = null;
432         Location          lineLocation     = null;
433         BreakpointRequest breakpRequest    = null;
434 
435         try {
436             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
437 
438             alllineLocations = method.allLineLocations();
439 
440             int n =
441                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
442             if (n > alllineLocations.size()) {
443                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
444             } else {
445                 lineLocation = (Location) alllineLocations.get(n);
446                 try {
447                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
448                     breakpRequest.putProperty("number", property);
449                     breakpRequest.addThreadFilter(thread);
450                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
451                 } catch ( Exception e1 ) {
452                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
453                     breakpRequest    = null;
454                 }
455             }
456         } catch ( Exception e2 ) {
457             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
458             breakpRequest    = null;
459         }
460 
461         if (breakpRequest == null) {
462             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
463             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
464         }
465 
466         log2("      a breakpoint has been set up");
467         return breakpRequest;
468     }

471     private void getEventSet()
472                  throws JDITestRuntimeException {
473         try {
474 //            log2("       eventSet = eventQueue.remove(waitTime);");
475             eventSet = eventQueue.remove(waitTime);
476             if (eventSet == null) {
477                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
478             }
479 //            log2("       eventIterator = eventSet.eventIterator;");
480             eventIterator = eventSet.eventIterator();
481         } catch ( Exception e ) {
482             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
483         }
484     }
