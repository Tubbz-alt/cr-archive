 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createThreadStartRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *  - Creates a new disabled ThreadDeathRequest.                <BR>
 47  * <BR>
 48  * The test has three phases and works as follows.              <BR>
 49  * <BR>
 50  * In first phase,                                              <BR>
 51  * upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 55  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 56  * and waits for the event within the predefined time interval.         <BR>
 57  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 58  * Upon getting the ClassPrepareEvent,                                  <BR>
 59  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 60  * within debuggee's special methodForCommunication().                  <BR>
 61  * <BR>
 62  * In second phase to check the assertion,                              <BR>
 63  * the debugger and the debuggee perform the following.                 <BR>
 64  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 65  * - The debuggee prepares new check case and invokes                   <BR>
 66  *   the methodForCommunication to be suspended and                     <BR>
 67  *   to inform the debugger with the event.                             <BR>
 68  * - Upon getting the BreakpointEvent,                                  <BR>
 69  *   the debugger performs the check required.                          <BR>
 70  * <BR>
 71  * In third phase, at the end of the test, the debuggee changes         <BR>
 72  * the value of the "instruction" which the debugger and debuggee       <BR>
 73  * use to inform each other of needed actions,  and both end.           <BR>
 74  * <BR>
 75  */
 76

149     private int runThis (String argv[], PrintStream out) {
150 
151         argsHandler     = new ArgumentHandler(argv);
152         logHandler      = new Log(out, argsHandler);
153         Binder binder   = new Binder(argsHandler, logHandler);
154 
155         waitTime        = argsHandler.getWaitTime() * 60000;
156 
157         try {
158             log2("launching a debuggee :");
159             log2("       " + debuggeeName);
160             if (argsHandler.verbose()) {
161                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
162             } else {
163                 debuggee = binder.bindToDebugee(debuggeeName);
164             }
165             if (debuggee == null) {
166                 log3("ERROR: no debuggee launched");
167                 return FAILED;
168             }
169             log2("debuggee launched");
170         } catch ( Exception e ) {
171             log3("ERROR: Exception : " + e);
172             log2("       test cancelled");
173             return FAILED;
174         }
175 
176         debuggee.redirectOutput(logHandler);
177 
178         vm = debuggee.VM();
179 
180         eventQueue = vm.eventQueue();
181         if (eventQueue == null) {
182             log3("ERROR: eventQueue == null : TEST ABORTED");
183             vm.exit(PASS_BASE);
184             return FAILED;
185         }
186 
187         log2("invocation of the method runTest()");
188         switch (runTest()) {
189 
190             case 0 :  log2("test phase has finished normally");
191                       log2("   waiting for the debuggee to finish ...");
192                       debuggee.waitFor();
193 
194                       log2("......getting the debuggee's exit status");
195                       int status = debuggee.getStatus();
196                       if (status != PASS_BASE) {
197                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
198                               status + " != PASS_BASE");
199                           testExitCode = FAILED;
200                       } else {
201                           log2("......debuggee returned expected exit status: " +
202                               status + " == PASS_BASE");
203                       }
204                       break;
205 
206             default : log3("ERROR: runTest() returned unexpected value");
207 
208             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
209                       log2("......forcing: vm.exit();");
210                       testExitCode = FAILED;
211                       try {
212                           vm.exit(PASS_BASE);
213                       } catch ( Exception e ) {
214                           log3("ERROR: Exception : " + e);
215                       }
216                       break;
217 
218             case 2 :  log3("test cancelled due to VMDisconnectedException");
219                       log2("......trying: vm.process().destroy();");
220                       testExitCode = FAILED;
221                       try {
222                           Process vmProcess = vm.process();
223                           if (vmProcess != null) {
224                               vmProcess.destroy();
225                           }
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : " + e);
228                       }
229                       break;
230             }
231 
232         return testExitCode;
233     }

236    /*
237     * Return value: 0 - normal end of the test
238     *               1 - ubnormal end of the test
239     *               2 - VMDisconnectedException while test phase
240     */

242     private int runTest() {
243 
244         try {
245             testRun();
246 
247             log2("waiting for VMDeathEvent");
248             getEventSet();
249             if (eventIterator.nextEvent() instanceof VMDeathEvent)
250                 return 0;
251 
252             log3("ERROR: last event is not the VMDeathEvent");
253             return 1;
254         } catch ( VMDisconnectedException e ) {
255             log3("ERROR: VMDisconnectedException : " + e);
256             return 2;
257         } catch ( Exception e ) {
258             log3("ERROR: Exception : " + e);
259             return 1;
260         }
261 
262     }

264     private void testRun()
265                  throws JDITestRuntimeException, Exception {
266 
267         eventRManager = vm.eventRequestManager();
268 
269         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
270         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
271         cpRequest.addClassFilter(debuggeeName);
272 
273         cpRequest.enable();
274         vm.resume();
275         getEventSet();
276         cpRequest.disable();
277 
278         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
279         debuggeeClass = event.referenceType();
280 
281         if (!debuggeeClass.name().equals(debuggeeName))
282            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
283 
284         log2("      received: ClassPrepareEvent for debuggeeClass");
285 
286         String bPointMethod = "methodForCommunication";
287         String lineForComm  = "lineForComm";
288 
289         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
290 
291         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
292                                              debuggeeClass,
293                                             bPointMethod, lineForComm, "zero");
294         bpRequest.enable();
295 
296     //------------------------------------------------------  testing section
297 
298         ThreadStartRequest tsRequest1 = null;
299 
300 
301         log1("     TESTING BEGINS");
302 
303         for (int i = 0; ; i++) {
304 
305             vm.resume();
306             breakpointForCommunication();
307 
308             int instruction = ((IntegerValue)
309                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
310 
311             if (instruction == 0) {
312                 vm.resume();
313                 break;
314             }
315 
316             log1(":::::: case: # " + i);
317 
318             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
319 
320             log2("......creating: tsRequest1 = eventRManager.createThreadStartRequest();");
321             tsRequest1 = eventRManager.createThreadStartRequest();
322             log2("      checking up on if request is disabled");
323             if (tsRequest1.isEnabled()) {
324                 testExitCode = FAILED;
325                 log3("ERROR: request is not disabled");
326             }
327 
328             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
329         }
330         log1("    TESTING ENDS");
331         return;
332     }

334    /*
335     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
336     *                                             String, String, String)
337     *
338     * It sets up a breakpoint at given line number within a given method in a given class
339     * for a given thread.
340     *
341     * Return value: BreakpointRequest object  in case of success
342     *
343     * JDITestRuntimeException   in case of an Exception thrown within the method
344     */
345 
346     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
347                                                   ReferenceType testedClass,
348                                                   String methodName,
349                                                   String bpLine,
350                                                   String property)
351             throws JDITestRuntimeException {
352 
353         log2("......setting up a breakpoint:");
354         log2("       thread: " + thread + "; class: " + testedClass +
355                         "; method: " + methodName + "; line: " + bpLine);
356 
357         List              alllineLocations = null;
358         Location          lineLocation     = null;
359         BreakpointRequest breakpRequest    = null;
360 
361         try {
362             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
363 
364             alllineLocations = method.allLineLocations();
365 
366             int n =
367                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
368             if (n > alllineLocations.size()) {
369                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
370             } else {
371                 lineLocation = (Location) alllineLocations.get(n);
372                 try {
373                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
374                     breakpRequest.putProperty("number", property);
375                     breakpRequest.addThreadFilter(thread);
376                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
377                 } catch ( Exception e1 ) {
378                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
379                     breakpRequest    = null;
380                 }
381             }
382         } catch ( Exception e2 ) {
383             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
384             breakpRequest    = null;
385         }
386 
387         if (breakpRequest == null) {
388             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
389             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
390         }
391 
392         log2("      a breakpoint has been set up");
393         return breakpRequest;
394     }

397     private void getEventSet()
398                  throws JDITestRuntimeException {
399         try {
400 //            log2("       eventSet = eventQueue.remove(waitTime);");
401             eventSet = eventQueue.remove(waitTime);
402             if (eventSet == null) {
403                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
404             }
405 //            log2("       eventIterator = eventSet.eventIterator;");
406             eventIterator = eventSet.eventIterator();
407         } catch ( Exception e ) {
408             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
409         }
410     }

413     private void breakpointForCommunication()
414                  throws JDITestRuntimeException {
415 
416         log2("breakpointForCommunication");
417         getEventSet();
418 
419         if (eventIterator.nextEvent() instanceof BreakpointEvent)
420             return;
421 
422         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
423     }
