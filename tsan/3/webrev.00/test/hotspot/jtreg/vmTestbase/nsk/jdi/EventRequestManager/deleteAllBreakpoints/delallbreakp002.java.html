 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.deleteAllBreakpoints()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - All breakpoints are deleted, that is, following call to            <BR>
 47  *       EventRequestManager.breakpointRequests()                       <BR>
 48  *   returns empty list.                                                <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debugger resumes         <BR>
 74  * the debuggee,  and both end.                                         <BR>
 75  * <BR>
 76  */

153     private int runThis (String argv[], PrintStream out) {
154 
155         argsHandler     = new ArgumentHandler(argv);
156         logHandler      = new Log(out, argsHandler);
157         Binder binder   = new Binder(argsHandler, logHandler);
158 
159         waitTime        = argsHandler.getWaitTime() * 60000;
160 
161         try {
162             log2("launching a debuggee :");
163             log2("       " + debuggeeName);
164             if (argsHandler.verbose()) {
165                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
166             } else {
167                 debuggee = binder.bindToDebugee(debuggeeName);
168             }
169             if (debuggee == null) {
170                 log3("ERROR: no debuggee launched");
171                 return FAILED;
172             }
173             log2("debuggee launched");
174         } catch ( Exception e ) {
175             log3("ERROR: Exception : " + e);
176             log2("       test cancelled");
177             return FAILED;
178         }
179 
180         debuggee.redirectOutput(logHandler);
181 
182         vm = debuggee.VM();
183 
184         eventQueue = vm.eventQueue();
185         if (eventQueue == null) {
186             log3("ERROR: eventQueue == null : TEST ABORTED");
187             vm.exit(PASS_BASE);
188             return FAILED;
189         }
190 
191         log2("invocation of the method runTest()");
192         switch (runTest()) {
193 
194             case 0 :  log2("test phase has finished normally");
195                       log2("   waiting for the debuggee to finish ...");
196                       debuggee.waitFor();
197 
198                       log2("......getting the debuggee's exit status");
199                       int status = debuggee.getStatus();
200                       if (status != PASS_BASE) {
201                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
202                               status + " != PASS_BASE");
203                           testExitCode = FAILED;
204                       } else {
205                           log2("......debuggee returned expected exit status: " +
206                               status + " == PASS_BASE");
207                       }
208                       break;
209 
210             default : log3("ERROR: runTest() returned unexpected value");
211 
212             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
213                       log2("......forcing: vm.exit();");
214                       testExitCode = FAILED;
215                       try {
216                           vm.exit(PASS_BASE);
217                       } catch ( Exception e ) {
218                           log3("ERROR: Exception : " + e);
219                       }
220                       break;
221 
222             case 2 :  log3("test cancelled due to VMDisconnectedException");
223                       log2("......trying: vm.process().destroy();");
224                       testExitCode = FAILED;
225                       try {
226                           Process vmProcess = vm.process();
227                           if (vmProcess != null) {
228                               vmProcess.destroy();
229                           }
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234             }
235 
236         return testExitCode;
237     }

246     private int runTest() {
247 
248         try {
249             testRun();
250 
251             log2("waiting for VMDeathEvent");
252             getEventSet();
253             if (eventIterator.nextEvent() instanceof VMDeathEvent)
254                 return 0;
255 
256             log3("ERROR: last event is not the VMDeathEvent");
257             return 1;
258         } catch ( VMDisconnectedException e ) {
259             log3("ERROR: VMDisconnectedException : " + e);
260             return 2;
261         } catch ( Exception e ) {
262             log3("ERROR: Exception : " + e);
263             return 1;
264         }
265 
266     }

268     private void testRun()
269                  throws JDITestRuntimeException, Exception {
270 
271         eventRManager = vm.eventRequestManager();
272 
273         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
274         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
275         cpRequest.addClassFilter(debuggeeName);
276 
277         cpRequest.enable();
278         vm.resume();
279         getEventSet();
280         cpRequest.disable();
281 
282         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
283         debuggeeClass = event.referenceType();
284 
285         if (!debuggeeClass.name().equals(debuggeeName))
286            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292 
293         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
294 
295         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
296                                              debuggeeClass,
297                                             bPointMethod, lineForComm, "zero");
298         bpRequest.enable();
299 
300     //------------------------------------------------------  testing section
301 
302         List         requests = null;
303         ListIterator li       = null;
304 
305         BreakpointRequest request = null;
306         BreakpointRequest bpRequests[] = { null, null, null, null, null,
307                                            null, null, null, null, null };
308         int listSize;
309         int flag;
310 
311 
312         log1("     TESTING BEGINS");
313 
314             vm.resume();
315             breakpointForCommunication();
316 
317             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
318 
319             log2("......creating BreakpointRequests");
320             for (int i1 = 0; i1 < bpRequests.length; i1++) {
321                 bpRequests[i1] = eventRManager.createBreakpointRequest(breakpLocation);
322                 bpRequests[i1].putProperty("number", String.valueOf(i1));
323             }
324 
325             log2("......deleting all BreakpointRequests");
326             eventRManager.deleteAllBreakpoints();
327             log2("...... getting new list of BreakpointRequests");
328             requests = eventRManager.breakpointRequests();
329 
330             log2("...... checking up on size of the list;  0 is expected");
331             listSize = requests.size();
332             if ( listSize != 0 ) {
333                 testExitCode = FAILED;
334                 log3("ERROR: size of returned List is not equal to expected 0 ");
335             }
336 
337             vm.resume();
338 
339             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
340 
341         log1("    TESTING ENDS");
342         return;
343     }

345    /*
346     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
347     *                                             String, String, String)
348     *
349     * It sets up a breakpoint at given line number within a given method in a given class
350     * for a given thread.
351     *
352     * Return value: BreakpointRequest object  in case of success
353     *
354     * JDITestRuntimeException   in case of an Exception thrown within the method
355     */
356 
357     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
358                                                   ReferenceType testedClass,
359                                                   String methodName,
360                                                   String bpLine,
361                                                   String property)
362             throws JDITestRuntimeException {
363 
364         log2("......setting up a breakpoint:");
365         log2("       thread: " + thread + "; class: " + testedClass +
366                         "; method: " + methodName + "; line: " + bpLine);
367 
368         List              alllineLocations = null;
369         Location          lineLocation     = null;
370         BreakpointRequest breakpRequest    = null;
371 
372         try {
373             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
374 
375             alllineLocations = method.allLineLocations();
376 
377             int n =
378                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
379             if (n > alllineLocations.size()) {
380                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
381             } else {
382                 lineLocation = (Location) alllineLocations.get(n);
383 breakpLocation = lineLocation;
384                 try {
385                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
386                     breakpRequest.putProperty("number", property);
387                     breakpRequest.addThreadFilter(thread);
388                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
389                 } catch ( Exception e1 ) {
390                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
391                     breakpRequest    = null;
392                 }
393             }
394         } catch ( Exception e2 ) {
395             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
396             breakpRequest    = null;
397         }
398 
399         if (breakpRequest == null) {
400             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
401             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
402         }
403 
404         log2("      a breakpoint has been set up");
405         return breakpRequest;
406     }

409     private void getEventSet()
410                  throws JDITestRuntimeException {
411         try {
412 //            log2("       eventSet = eventQueue.remove(waitTime);");
413             eventSet = eventQueue.remove(waitTime);
414             if (eventSet == null) {
415                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
416             }
417 //            log2("       eventIterator = eventSet.eventIterator;");
418             eventIterator = eventSet.eventIterator();
419         } catch ( Exception e ) {
420             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
421         }
422     }

425     private void breakpointForCommunication()
426                  throws JDITestRuntimeException {
427 
428         log2("breakpointForCommunication");
429         getEventSet();
430 
431         if (eventIterator.nextEvent() instanceof BreakpointEvent)
432             return;
433 
434         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
435     }
