 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.allClasses()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test check up that new class and interface types <BR>
 46  * are in returned List only after corresponding objects<BR>
 47  * have been created within the debuggee.               <BR>
 48  * The sequence of the cases for testing is as follows: <BR>
 49  * <BR>
 50  *      Objects# == 0, Classes# == 0, Interfaces# == 0  <BR>
 51  * hence, the returned List must contain no mirrors of  <BR>
 52  * tested reference types;                              <BR>
 53  *      Objects# == 1, Classes# == 1, Interfaces# == 0  <BR>
 54  * the returned list must contain only one expected     <BR>
 55  * object mirroring a tested Class type;                <BR>
 56  *      Objects# == 2, Classes# == 2, Interfaces# == 1  <BR>
 57  * the returned list must contain three expected objects<BR>
 58  * mirroring two Class and one Interface types;         <BR>
 59  *      Objects# == 3, Classes# == 2, Interfaces# == 1, <BR>
 60  * the returned list must contain three expected objects<BR>
 61  * mirroring two Class and one Interface types;         <BR>
 62  * repetirion of previous check after one more object   <BR>
 63  * has been created;                                    <BR>
 64  *      Objects# == 4, Classes# == 3, Interfaces# == 1, <BR>
 65  * the returned list must contain four expected objects <BR>
 66  * mirroring three Class and one Interface types;       <BR>
 67  * <BR>
 68  */

151     private int runThis (String argv[], PrintStream out) {
152 
153         argsHandler     = new ArgumentHandler(argv);
154         logHandler      = new Log(out, argsHandler);
155         Binder binder   = new Binder(argsHandler, logHandler);
156 
157         waitTime        = argsHandler.getWaitTime() * 60000;
158 
159         try {
160             log2("launching a debuggee :");
161             log2("       " + debuggeeName);
162             if (argsHandler.verbose()) {
163                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
164             } else {
165                 debuggee = binder.bindToDebugee(debuggeeName);
166             }
167             if (debuggee == null) {
168                 log3("ERROR: no debuggee launched");
169                 return FAILED;
170             }
171             log2("debuggee launched");
172         } catch ( Exception e ) {
173             log3("ERROR: Exception : " + e);
174             log2("       test cancelled");
175             return FAILED;
176         }
177 
178         debuggee.redirectOutput(logHandler);
179 
180         vm = debuggee.VM();
181 
182         eventQueue = vm.eventQueue();
183         if (eventQueue == null) {
184             log3("ERROR: eventQueue == null : TEST ABORTED");
185             vm.exit(PASS_BASE);
186             return FAILED;
187         }
188 
189         log2("invocation of the method runTest()");
190         switch (runTest()) {
191 
192             case 0 :  log2("test phase has finished normally");
193                       log2("   waiting for the debuggee to finish ...");
194                       debuggee.waitFor();
195 
196                       log2("......getting the debuggee's exit status");
197                       int status = debuggee.getStatus();
198                       if (status != PASS_BASE) {
199                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
200                               status + " != PASS_BASE");
201                           testExitCode = FAILED;
202                       } else {
203                           log2("......debuggee returned expected exit status: " +
204                               status + " == PASS_BASE");
205                       }
206                       break;
207 
208             default : log3("ERROR: runTest() returned unexpected value");
209 
210             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
211                       log2("......forcing: vm.exit();");
212                       testExitCode = FAILED;
213                       try {
214                           vm.exit(PASS_BASE);
215                       } catch ( Exception e ) {
216                           log3("ERROR: Exception : e");
217                       }
218                       break;
219 
220             case 2 :  log3("test cancelled due to VMDisconnectedException");
221                       log2("......trying: vm.process().destroy();");
222                       testExitCode = FAILED;
223                       try {
224                           Process vmProcess = vm.process();
225                           if (vmProcess != null) {
226                               vmProcess.destroy();
227                           }
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232             }
233 
234         return testExitCode;
235     }

239     * Return value: 0 - normal end of the test
240     *               1 - ubnormal end of the test
241     *               2 - VMDisconnectedException while test phase
242     */

244     private int runTest() {
245 
246         try {
247             testRun();
248 
249             log2("waiting for VMDeathEvent");
250             getEventSet();
251             if (eventIterator.nextEvent() instanceof VMDeathEvent)
252                 return 0;
253 
254             log3("ERROR: last event is not the VMDeathEvent");
255             return 1;
256         } catch ( VMDisconnectedException e ) {
257             log3("ERROR: VMDisconnectedException : " + e);
258             return 2;
259         } catch ( Exception e ) {
260             log3("ERROR: Exception : " + e);
261             return 1;
262         }
263 
264     }

266     private void testRun()
267                  throws JDITestRuntimeException, Exception {
268 
269         eventRManager = vm.eventRequestManager();
270 
271         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
272         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
273         cpRequest.addClassFilter(debuggeeName);
274 
275         cpRequest.enable();
276         vm.resume();
277         getEventSet();
278         cpRequest.disable();
279 
280         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
281         debuggeeClass = event.referenceType();
282 
283         if (!debuggeeClass.name().equals(debuggeeName))
284            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
285 
286         log2("      received: ClassPrepareEvent for debuggeeClass");
287 
288         String bPointMethod = "methodForCommunication";
289         String lineForComm  = "lineForComm";
290         BreakpointRequest bpRequest;
291 
292         try {
293             bpRequest = settingBreakpoint(threadByName("main"),
294                                           debuggeeClass,
295                                           bPointMethod, lineForComm, "zero");
296         } catch ( Exception e ) {
297             throw e;
298         }
299         bpRequest.enable();
300 
301     //------------------------------------------------------  testing section
302 
303         log1("     TESTING BEGINS");
304 
305         for (int i = 0; ; i++) {
306 
307             vm.resume();
308             breakpointForCommunication();
309 
310             int instruction = ((IntegerValue) (
311                                debuggeeClass.getValue(
312                                debuggeeClass.fieldByName("instruction")))).value();
313 
314             if (instruction == 0) {
315                 vm.resume();
316                 break;
317             }
318 
319             log1("  new check: # " + i);
320 
321             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
322 
323             int expresult = returnCode0;
324 
325             String name[] = { "Class1ForCheck",
326                               "Class2ForCheck",
327                               "Class3ForCheck",
328                               "InterfaceForCheck" };
329 
330 
331             switch (i) {

333                 case 0:

336                             log3("ERROR: Class in the List: " + name[0]);

338                         }
339                         if (refTypeYes(name[1]) != 0) {
340                             log3("ERROR: Class in the List: " + name[1]);
341                             expresult = returnCode1;

451     private ThreadReference threadByName(String name)
452                             throws JDITestRuntimeException {
453 
454         List         all = vm.allThreads();
455         ListIterator li  = all.listIterator();
456 
457         for (; li.hasNext(); ) {
458             ThreadReference thread = (ThreadReference) li.next();
459             if (thread.name().equals(name))
460                 return thread;
461         }
462         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
463     }

465    /*
466     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
467     *                                             String, String, String)
468     *
469     * It sets up a breakpoint at given line number within a given method in a given class
470     * for a given thread.
471     *
472     * Return value: BreakpointRequest object  in case of success
473     *
474     * JDITestRuntimeException   in case of an Exception thrown within the method
475     */
476 
477     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
478                                                   ReferenceType testedClass,
479                                                   String methodName,
480                                                   String bpLine,
481                                                   String property)
482             throws JDITestRuntimeException, Exception {
483 
484         log2("......setting up a breakpoint:");
485         log2("       thread: " + thread + "; class: " + testedClass +
486                         "; method: " + methodName + "; line: " + bpLine);
487 
488         List              alllineLocations = null;
489         Location          lineLocation     = null;
490         BreakpointRequest breakpRequest    = null;
491 
492         try {
493             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
494 
495             alllineLocations = method.allLineLocations();
496 
497             int n =
498                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
499             if (n > alllineLocations.size()) {
500                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
501             } else {
502                 lineLocation = (Location) alllineLocations.get(n);
503                 try {
504                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
505                     breakpRequest.putProperty("number", property);
506                     breakpRequest.addThreadFilter(thread);
507                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
508                 } catch ( Exception e1 ) {
509                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
510 //                    breakpRequest    = null;
511                     throw e1;
512                 }
513             }
514         } catch ( Exception e2 ) {
515             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
516 //            breakpRequest    = null;
517             throw e2;
518         }
519 
520         if (breakpRequest == null) {
521             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
522             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
523         }
524 
525         log2("      a breakpoint has been set up");
526         return breakpRequest;
527     }

530     private void getEventSet()
531                  throws JDITestRuntimeException {
532         try {
533 //            log2("       eventSet = eventQueue.remove(waitTime);");
534             eventSet = eventQueue.remove(waitTime);
535             if (eventSet == null) {
536                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
537             }
538 //            log2("       eventIterator = eventSet.eventIterator;");
539             eventIterator = eventSet.eventIterator();
540         } catch ( Exception e ) {
541             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
542         }
543     }

546     private void breakpointForCommunication()
547                  throws JDITestRuntimeException {
548 
549         log2("breakpointForCommunication");
550         getEventSet();
551 
552         if (eventIterator.nextEvent() instanceof BreakpointEvent)
553             return;
554 
555         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
556     }

560     private int refTypeYes ( String name ) {
561 
562         List vmClasses = vm.allClasses();
563         ListIterator li = vmClasses.listIterator();
564 
565         for ( int i = 0; li.hasNext(); i++) {
566             ReferenceType obj = (ReferenceType) li.next();
567             if ( obj.name().equals(namePrefix + name) )
568                 return 1;
569         }
570         return 0;
571     }
