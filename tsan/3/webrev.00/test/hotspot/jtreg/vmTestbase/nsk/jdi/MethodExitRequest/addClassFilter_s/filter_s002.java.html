 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodExitRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodExitRequest.addClassFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 68  *   and to inform the debugger with the event.                         <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates an MethodExitRequest and enables it,                     <BR>
 71  *   - invokes the method addClassFilter() on the request               <BR>
 72  *     expecting to catch InvalidRequestStateException,                 <BR>
 73  *   - disables the request and invokes the method addClassFilter()     <BR>
 74  *     expecting to catch no InvalidRequestStateException,              <BR>
 75  *   - deletes the request and invokes the method addClassFilter()      <BR>
 76  *     once again expecting to catch InvalidRequestStateException.      <BR>
 77  * <BR>
 78  * In third phase, at the end of the test, the debuggee changes         <BR>
 79  * the value of the "instruction" which the debugger and debuggee       <BR>
 80  * use to inform each other of needed actions,  and both end.           <BR>
 81  * <BR>
 82  */
 83

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }

240             }
241 
242         return testExitCode;
243     }

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         log1("     TESTING BEGINS");
309 
310         EventRequest  eventRequest1      = null;
311 
312         String        property1          = "MethodExitRequest1";
313 
314 
315         for (int i = 0; ; i++) {
316 
317             vm.resume();
318             breakpointForCommunication();
319 
320             int instruction = ((IntegerValue)
321                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
322 
323             if (instruction == 0) {
324                 vm.resume();
325                 break;
326             }
327 
328 
329             log1(":::::: case: # " + i);
330 
331             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
332 
333             switch (i) {
334 
335               case 0:
336                      eventRequest1 = setting22MethodExitRequest(null, testedClassName1,
337                                               EventRequest.SUSPEND_NONE, property1);
338 
339                      eventRequest1.enable();
340 
341                      try {
342                          log2("......eventRequest1.addClassFilter(testedClassName1);");
343                          log2("        InvalidRequestStateException expected");
344                          ((MethodExitRequest)eventRequest1).addClassFilter(testedClassName1);
345                          log3("ERROR: no InvalidRequestStateException");
346                          testExitCode = FAILED;
347                      } catch ( InvalidRequestStateException e ) {
348                          log2("        InvalidRequestStateException");
349                      }
350 
351                      log2("......eventRequest1.disable();");
352                      eventRequest1.disable();
353                      try {
354                          log2("......eventRequest1.addClassFilter(testedClassName1);");
355                          log2("        no InvalidRequestStateException expected");
356                          ((MethodExitRequest)eventRequest1).addClassFilter(testedClassName1);
357                          log2("        no InvalidRequestStateException");
358                      } catch ( InvalidRequestStateException e ) {
359                          log3("ERROR: InvalidRequestStateException");
360                          testExitCode = FAILED;
361                      }
362 
363                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
364                      eventRManager.deleteEventRequest(eventRequest1);
365                      try {
366                          log2("......eventRequest1.addClassFilter(testedClassName1);");
367                          log2("        InvalidRequestStateException expected");
368                          ((MethodExitRequest)eventRequest1).addClassFilter(testedClassName1);
369                          log3("ERROR: no InvalidRequestStateException");
370                          testExitCode = FAILED;
371                      } catch ( InvalidRequestStateException e ) {
372                          log2("        InvalidRequestStateException");
373                      }
374                       break;
375 
376               default:
377                       throw new JDITestRuntimeException("** default case 2 **");
378             }
379 
380             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
381         }
382         log1("    TESTING ENDS");
383         return;
384     }

386    /*
387     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
388     *                                             String, String, String)
389     *
390     * It sets up a breakpoint at given line number within a given method in a given class
391     * for a given thread.
392     *
393     * Return value: BreakpointRequest object  in case of success
394     *
395     * JDITestRuntimeException   in case of an Exception thrown within the method
396     */
397 
398     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
399                                                   ReferenceType testedClass,
400                                                   String methodName,
401                                                   String bpLine,
402                                                   String property)
403             throws JDITestRuntimeException {
404 
405         log2("......setting up a breakpoint:");
406         log2("       thread: " + thread + "; class: " + testedClass +
407                         "; method: " + methodName + "; line: " + bpLine);
408 
409         List              alllineLocations = null;
410         Location          lineLocation     = null;
411         BreakpointRequest breakpRequest    = null;
412 
413         try {
414             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
415 
416             alllineLocations = method.allLineLocations();
417 
418             int n =
419                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
420             if (n > alllineLocations.size()) {
421                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
422             } else {
423                 lineLocation = (Location) alllineLocations.get(n);
424                 try {
425                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
426                     breakpRequest.putProperty("number", property);
427                     breakpRequest.addThreadFilter(thread);
428                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
429                 } catch ( Exception e1 ) {
430                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
431                     breakpRequest    = null;
432                 }
433             }
434         } catch ( Exception e2 ) {
435             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
436             breakpRequest    = null;
437         }
438 
439         if (breakpRequest == null) {
440             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
441             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
442         }
443 
444         log2("      a breakpoint has been set up");
445         return breakpRequest;
446     }

449     private void getEventSet()
450                  throws JDITestRuntimeException {
451         try {
452 //            log2("       eventSet = eventQueue.remove(waitTime);");
453             eventSet = eventQueue.remove(waitTime);
454             if (eventSet == null) {
455                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
456             }
457 //            log2("       eventIterator = eventSet.eventIterator;");
458             eventIterator = eventSet.eventIterator();
459         } catch ( Exception e ) {
460             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
461         }
462     }

465     private void breakpointForCommunication()
466                  throws JDITestRuntimeException {
467 
468         log2("breakpointForCommunication");
469         getEventSet();
470 
471         if (eventIterator.nextEvent() instanceof BreakpointEvent)
472             return;
473 
474         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
475     }

479     private MethodExitRequest setting22MethodExitRequest ( ThreadReference thread,
480                                                              String          testedClass,
481                                                              int             suspendPolicy,
482                                                              String          property       )
483             throws JDITestRuntimeException {
484         try {
485             log2("......setting up MethodExitRequest:");
486             log2("       thread: " + thread + "; class: " + testedClass +  "; property: " + property);
487 
488             MethodExitRequest
489             menr = eventRManager.createMethodExitRequest();
490             menr.putProperty("number", property);
491             if (thread != null)
492                 menr.addThreadFilter(thread);
493             menr.addClassFilter(testedClass);
494             menr.setSuspendPolicy(suspendPolicy);
495 
496             log2("      a MethodExitRequest has been set up");
497             return menr;
498         } catch ( Exception e ) {
499             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
500             log3("       MethodExitRequest HAS NOT BEEN SET UP");
501             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
502         }
