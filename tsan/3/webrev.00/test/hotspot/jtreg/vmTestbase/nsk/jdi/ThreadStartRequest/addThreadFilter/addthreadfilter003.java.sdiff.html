402     private ThreadReference threadByName(String name)
403                  throws JDITestRuntimeException {
404 
405         List         all = vm.allThreads();
406         ListIterator li  = all.listIterator();
407 
408         for (; li.hasNext(); ) {
409             ThreadReference thread = (ThreadReference) li.next();
410             if (thread.name().equals(name))
411                 return thread;
412         }
413         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
414     }
