 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.field()</code>           <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  *  The case to test include ModificationWatchpointRequest.             <BR>
 46  *  The test checks if a Field object, an argument of the method        <BR>
 47  *     EventRequestManager.createModificationWatchpointRequest(Field),  <BR>
 48  *  is equal to one returned by the method WatchpointRequest.field()    <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 68  *   and to inform the debugger with the event.                         <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates an ModificationWatchpointRequest and                     <BR>
 71  *     gets Field object filed1,                                        <BR>
 72  *   - gets Field object field2 with the method WatchpointRequest.field(),<BR>
 73  *   - compares the Filed objects.                                      <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         log1("     TESTING BEGINS");
309 
310         EventRequest  eventRequest1 = null;
311         String        property1     = "ModificationWatchpointRequest1";
312         String        fieldName1    = "var101";
313 
314         ReferenceType testClassReference = null;
315 
316 
317         for (int i = 0; ; i++) {
318 
319             vm.resume();
320             breakpointForCommunication();
321 
322             int instruction = ((IntegerValue)
323                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
324 
325             if (instruction == 0) {
326                 vm.resume();
327                 break;
328             }
329 
330 
331             log1(":::::: case: # " + i);
332 
333             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
334 
335             switch (i) {
336 
337               case 0:
338                      testClassReference =
339                            (ReferenceType) vm.classesByName(testedClassName1).get(0);
340 
341                      log2(".....setting up ModificationWatchpointRequest and getting Field object field1");
342                      eventRequest1 = setting21ModificationWatchpointRequest (null,
343                                              testClassReference, fieldName1,
344                                              EventRequest.SUSPEND_NONE, property1);
345 
346                      log2("......getting: field2 = ((WatchpointRequest) eventRequest1).field();");
347                      field2 = ((WatchpointRequest) eventRequest1).field();
348 
349                      log2("      checking up on equality of field1 and field2");
350                      if ( !field1.equals(field2) ) {
351                          testExitCode = FAILED;
352                          log3("ERROR: Field objects are not equal");
353                      }
354 
355                       break;
356 
357               default:
358                       throw new JDITestRuntimeException("** default case 2 **");
359             }
360 
361             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
362         }
363         log1("    TESTING ENDS");
364         return;
365     }

367    /*
368     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
369     *                                             String, String, String)
370     *
371     * It sets up a breakpoint at given line number within a given method in a given class
372     * for a given thread.
373     *
374     * Return value: BreakpointRequest object  in case of success
375     *
376     * JDITestRuntimeException   in case of an Exception thrown within the method
377     */
378 
379     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
380                                                   ReferenceType testedClass,
381                                                   String methodName,
382                                                   String bpLine,
383                                                   String property)
384             throws JDITestRuntimeException {
385 
386         log2("......setting up a breakpoint:");
387         log2("       thread: " + thread + "; class: " + testedClass +
388                         "; method: " + methodName + "; line: " + bpLine);
389 
390         List              alllineLocations = null;
391         Location          lineLocation     = null;
392         BreakpointRequest breakpRequest    = null;
393 
394         try {
395             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
396 
397             alllineLocations = method.allLineLocations();
398 
399             int n =
400                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
401             if (n > alllineLocations.size()) {
402                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
403             } else {
404                 lineLocation = (Location) alllineLocations.get(n);
405                 try {
406                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
407                     breakpRequest.putProperty("number", property);
408                     breakpRequest.addThreadFilter(thread);
409                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
410                 } catch ( Exception e1 ) {
411                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
412                     breakpRequest    = null;
413                 }
414             }
415         } catch ( Exception e2 ) {
416             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
417             breakpRequest    = null;
418         }
419 
420         if (breakpRequest == null) {
421             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
422             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
423         }
424 
425         log2("      a breakpoint has been set up");
426         return breakpRequest;
427     }

430     private void getEventSet()
431                  throws JDITestRuntimeException {
432         try {
433 //            log2("       eventSet = eventQueue.remove(waitTime);");
434             eventSet = eventQueue.remove(waitTime);
435             if (eventSet == null) {
436                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
437             }
438 //            log2("       eventIterator = eventSet.eventIterator;");
439             eventIterator = eventSet.eventIterator();
440         } catch ( Exception e ) {
441             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
442         }
443     }

446     private void breakpointForCommunication()
447                  throws JDITestRuntimeException {
448 
449         log2("breakpointForCommunication");
450         getEventSet();
451 
452         if (eventIterator.nextEvent() instanceof BreakpointEvent)
453             return;
454 
455         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
456     }
