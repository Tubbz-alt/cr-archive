 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createModificationWatchpointRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    - Multiple watchpoints on the same field are permitted.           <BR>
 47  *    - Creates a new disabled watchpoint.                              <BR>
 48  *    - The watchpoint watches modification of the specified field.     <BR>
 49  *    - Throws: UnsupportedOperationException -                         <BR>
 50  *       if the target virtual machine does not support this operation. <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares testcase and invokes                         <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the checks required.                         <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

153     private int runThis (String argv[], PrintStream out) {
154 
155         argsHandler     = new ArgumentHandler(argv);
156         logHandler      = new Log(out, argsHandler);
157         Binder binder   = new Binder(argsHandler, logHandler);
158 
159         waitTime        = argsHandler.getWaitTime() * 60000;
160 
161         try {
162             log2("launching a debuggee :");
163             log2("       " + debuggeeName);
164             if (argsHandler.verbose()) {
165                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
166             } else {
167                 debuggee = binder.bindToDebugee(debuggeeName);
168             }
169             if (debuggee == null) {
170                 log3("ERROR: no debuggee launched");
171                 return FAILED;
172             }
173             log2("debuggee launched");
174         } catch ( Exception e ) {
175             log3("ERROR: Exception : " + e);
176             log2("       test cancelled");
177             return FAILED;
178         }
179 
180         debuggee.redirectOutput(logHandler);
181 
182         vm = debuggee.VM();
183 
184         eventQueue = vm.eventQueue();
185         if (eventQueue == null) {
186             log3("ERROR: eventQueue == null : TEST ABORTED");
187             vm.exit(PASS_BASE);
188             return FAILED;
189         }
190 
191         log2("invocation of the method runTest()");
192         switch (runTest()) {
193 
194             case 0 :  log2("test phase has finished normally");
195                       log2("   waiting for the debuggee to finish ...");
196                       debuggee.waitFor();
197 
198                       log2("......getting the debuggee's exit status");
199                       int status = debuggee.getStatus();
200                       if (status != PASS_BASE) {
201                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
202                               status + " != PASS_BASE");
203                           testExitCode = FAILED;
204                       } else {
205                           log2("......debuggee returned expected exit status: " +
206                               status + " == PASS_BASE");
207                       }
208                       break;
209 
210             default : log3("ERROR: runTest() returned unexpected value");
211 
212             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
213                       log2("......forcing: vm.exit();");
214                       testExitCode = FAILED;
215                       try {
216                           vm.exit(PASS_BASE);
217                       } catch ( Exception e ) {
218                           log3("ERROR: Exception : " + e);
219                       }
220                       break;
221 
222             case 2 :  log3("test cancelled due to VMDisconnectedException");
223                       log2("......trying: vm.process().destroy();");
224                       testExitCode = FAILED;
225                       try {
226                           Process vmProcess = vm.process();
227                           if (vmProcess != null) {
228                               vmProcess.destroy();
229                           }
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234             }
235 
236         return testExitCode;
237     }

242     *               1 - ubnormal end of the test
243     *               2 - VMDisconnectedException while test phase

246     private int runTest() {
247 
248         try {
249             testRun();
250 
251             log2("waiting for VMDeathEvent");
252             getEventSet();
253             if (eventIterator.nextEvent() instanceof VMDeathEvent)
254                 return 0;
255 
256             log3("ERROR: last event is not the VMDeathEvent");
257             return 1;
258         } catch ( VMDisconnectedException e ) {
259             log3("ERROR: VMDisconnectedException : " + e);
260             return 2;
261         } catch ( Exception e ) {
262             log3("ERROR: Exception : " + e);
263             return 1;
264         }
265 
266     }

268     private void testRun()
269                  throws JDITestRuntimeException, Exception {
270 
271         eventRManager = vm.eventRequestManager();
272 
273         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
274         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
275         cpRequest.addClassFilter(debuggeeName);
276 
277         cpRequest.enable();
278         vm.resume();
279         getEventSet();
280         cpRequest.disable();
281 
282         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
283         debuggeeClass = event.referenceType();
284 
285         if (!debuggeeClass.name().equals(debuggeeName))
286            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292 
293         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
294 
295         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
296                                              debuggeeClass,
297                                             bPointMethod, lineForComm, "zero");
298         bpRequest.enable();
299 
300     //------------------------------------------------------  testing section
301 
302         String fieldName = "testField";
303 
304         Field field = null;
305 
306         List         requests = null;
307         ListIterator li       = null;
308 
309         ModificationWatchpointRequest mwRequest1 = null;
310         ModificationWatchpointRequest mwRequest2 = null;
311 
312 
313         log1("     TESTING BEGINS");
314 
315         for (int i = 0; ; i++) {
316 
317             vm.resume();
318             breakpointForCommunication();
319 
320             int instruction = ((IntegerValue)
321                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
322 
323             if (instruction == 0) {
324                 vm.resume();
325                 break;
326             }
327 
328             log1(":::::: case: # " + i);
329 
330             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
331 
332             field = debuggeeClass.fieldByName(fieldName);
333 
334             if ( vm.canWatchFieldModification() ) {
335                 log2("......vm.canWatchFieldModification == true");
336                 log2("      creating two ModificationWatchpointRequests for 'field': mwRequest1 & mwRequest2");
337 
338                 mwRequest1 = eventRManager.createModificationWatchpointRequest(field);
339                 mwRequest2 = eventRManager.createModificationWatchpointRequest(field);
340 
341                 log2("      checking up on: requests are disabled");
342                 if (mwRequest1.isEnabled()) {
343                     testExitCode = FAILED;
344                     log3("ERROR: mwRequest1 is enabled");
345                 }
346                 if (mwRequest2.isEnabled()) {
347                     testExitCode = FAILED;
348                     log3("ERROR: mwRequest2 is enabled");
349                 }
350                 log2("      compareing field equality");
351                 if ( !mwRequest1.field().equals(field) ) {
352                     testExitCode = FAILED;
353                     log3("ERROR: !mwRequest1.field().equals(field)");
354                 }
355                 if ( !mwRequest2.field().equals(field) ) {
356                     testExitCode = FAILED;
357                     log3("ERROR: !mwRequest2.field().equals(field)");
358                 }
359 
360             } else {
361                 log2("......vm.canWatchFieldModification == false");
362                 log2("      UnsupportedOperationException is expected");
363                 try {
364                     mwRequest1 = eventRManager.createModificationWatchpointRequest(field);
365                     testExitCode = FAILED;
366                     log3("ERROR: NO UnsupportedOperationException");
367                 } catch ( UnsupportedOperationException e ) {
368                 }
369             }
370 
371             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
372         }
373         log1("    TESTING ENDS");
374         return;
375     }

378    /*
379     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
380     *                                             String, String, String)
381     *
382     * It sets up a breakpoint at given line number within a given method in a given class
383     * for a given thread.
384     *
385     * Return value: BreakpointRequest object  in case of success
386     *
387     * JDITestRuntimeException   in case of an Exception thrown within the method
388     */
389 
390     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
391                                                   ReferenceType testedClass,
392                                                   String methodName,
393                                                   String bpLine,
394                                                   String property)
395             throws JDITestRuntimeException {
396 
397         log2("......setting up a breakpoint:");
398         log2("       thread: " + thread + "; class: " + testedClass +
399                         "; method: " + methodName + "; line: " + bpLine);
400 
401         List              alllineLocations = null;
402         Location          lineLocation     = null;
403         BreakpointRequest breakpRequest    = null;
404 
405         try {
406             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
407 
408             alllineLocations = method.allLineLocations();
409 
410             int n =
411                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
412             if (n > alllineLocations.size()) {
413                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
414             } else {
415                 lineLocation = (Location) alllineLocations.get(n);
416                 try {
417                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
418                     breakpRequest.putProperty("number", property);
419                     breakpRequest.addThreadFilter(thread);
420                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
421                 } catch ( Exception e1 ) {
422                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
423                     breakpRequest    = null;
424                 }
425             }
426         } catch ( Exception e2 ) {
427             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
428             breakpRequest    = null;
429         }
430 
431         if (breakpRequest == null) {
432             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
433             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
434         }
435 
436         log2("      a breakpoint has been set up");
437         return breakpRequest;
438     }

441     private void getEventSet()
442                  throws JDITestRuntimeException {
443         try {
444 //            log2("       eventSet = eventQueue.remove(waitTime);");
445             eventSet = eventQueue.remove(waitTime);
446             if (eventSet == null) {
447                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
448             }
449 //            log2("       eventIterator = eventSet.eventIterator;");
450             eventIterator = eventSet.eventIterator();
451         } catch ( Exception e ) {
452             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
453         }
454     }

457     private void breakpointForCommunication()
458                  throws JDITestRuntimeException {
459 
460         log2("breakpointForCommunication");
461         getEventSet();
462 
463         if (eventIterator.nextEvent() instanceof BreakpointEvent)
464             return;
465 
466         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
467     }
