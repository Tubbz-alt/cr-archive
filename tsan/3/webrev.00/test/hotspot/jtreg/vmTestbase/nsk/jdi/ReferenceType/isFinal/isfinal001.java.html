127     private int runThis (String argv[], PrintStream out) {
128 
129         argsHandler     = new ArgumentHandler(argv);
130         logHandler      = new Log(out, argsHandler);
131         Binder binder   = new Binder(argsHandler, logHandler);
132 
133         waitTime        = argsHandler.getWaitTime() * 60000;
134 
135         try {
136             log2("launching a debuggee :");
137             log2("       " + debuggeeName);
138             if (argsHandler.verbose()) {
139                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
140             } else {
141                 debuggee = binder.bindToDebugee(debuggeeName);
142             }
143             if (debuggee == null) {
144                 log3("ERROR: no debuggee launched");
145                 return FAILED;
146             }
147             log2("debuggee launched");
148         } catch ( Exception e ) {
149             log3("ERROR: Exception : " + e);
150             log2("       test cancelled");
151             return FAILED;
152         }
153 
154         debuggee.redirectOutput(logHandler);
155 
156         vm = debuggee.VM();
157 
158         eventQueue = vm.eventQueue();
159         if (eventQueue == null) {
160             log3("ERROR: eventQueue == null : TEST ABORTED");
161             vm.exit(PASS_BASE);
162             return FAILED;
163         }
164 
165         log2("invocation of the method runTest()");
166         switch (runTest()) {
167 
168             case 0 :  log2("test phase has finished normally");
169                       log2("   waiting for the debuggee to finish ...");
170                       debuggee.waitFor();
171 
172                       log2("......getting the debuggee's exit status");
173                       int status = debuggee.getStatus();
174                       if (status != PASS_BASE) {
175                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
176                               status + " != PASS_BASE");
177                           testExitCode = FAILED;
178                       } else {
179                           log2("......debuggee returned expected exit status: " +
180                               status + " == PASS_BASE");
181                       }
182                       break;
183 
184             default : log3("ERROR: runTest() returned unexpected value");
185 
186             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
187                       log2("......forcing: vm.exit();");
188                       testExitCode = FAILED;
189                       try {
190                           vm.exit(PASS_BASE);
191                       } catch ( Exception e ) {
192                           log3("ERROR: Exception : e");
193                       }
194                       break;
195 
196             case 2 :  log3("test cancelled due to VMDisconnectedException");
197                       log2("......trying: vm.process().destroy();");
198                       testExitCode = FAILED;
199                       try {
200                           Process vmProcess = vm.process();
201                           if (vmProcess != null) {
202                               vmProcess.destroy();
203                           }
204                       } catch ( Exception e ) {
205                           log3("ERROR: Exception : e");
206                       }
207                       break;
208             }
209 
210         return testExitCode;
211     }

215     * Return value: 0 - normal end of the test
216     *               1 - ubnormal end of the test
217     *               2 - VMDisconnectedException while test phase
218     */

220     private int runTest() {
221 
222         try {
223             testRun();
224 
225             log2("waiting for VMDeathEvent");
226             getEventSet();
227             if (eventIterator.nextEvent() instanceof VMDeathEvent)
228                 return 0;
229 
230             log3("ERROR: last event is not the VMDeathEvent");
231             return 1;
232         } catch ( VMDisconnectedException e ) {
233             log3("ERROR: VMDisconnectedException : " + e);
234             return 2;
235         } catch ( Exception e ) {
236             log3("ERROR: Exception : " + e);
237             return 1;
238         }
239 
240     }

242     private void testRun()
243                  throws JDITestRuntimeException, Exception {
244 
245         eventRManager = vm.eventRequestManager();
246 
247         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
248         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
249         cpRequest.addClassFilter(debuggeeName);
250 
251         cpRequest.enable();
252         vm.resume();
253         getEventSet();
254         cpRequest.disable();
255 
256         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
257         debuggeeClass = event.referenceType();
258 
259         if (!debuggeeClass.name().equals(debuggeeName))
260            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
261 
262         log2("      received: ClassPrepareEvent for debuggeeClass");
263 
264         String bPointMethod = "methodForCommunication";
265         String lineForComm  = "lineForComm";
266         BreakpointRequest bpRequest;
267 
268         try {
269             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
270                                           debuggeeClass,
271                                           bPointMethod, lineForComm, "zero");
272         } catch ( Exception e ) {
273             throw e;
274         }
275         bpRequest.enable();
276 
277     //------------------------------------------------------  testing section
278 
279         log1("     TESTING BEGINS");
280 
281         for (int i = 0; ; i++) {
282 
283             vm.resume();
284             breakpointForCommunication();
285 
286             int instruction = ((IntegerValue)
287                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
288 
289             if (instruction == 0) {
290                 vm.resume();
291                 break;
292             }
293 
294             log1("  new check: # " + i);
295 
296             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
297 
298             String names1[] = {
299 
300                       ".TestClass$InnerClass",        "f",
301                       ".TestClass$InnerClass[]",      "t",
302 
303                       ".TestClass$FinalInnerClass",   "t",
304                       ".TestClass$FinalInnerClass[]", "t",
305 
306                       ".TestClass$NestedIface",        "f",
307                       ".TestClass$NestedIface[]",      "t",
308 
309                       ".OuterClass",        "f",
310                       ".OuterClass[]",      "t",
311                       ".OuterIface",        "f",
312                       ".OuterIface[]",      "t",
313 
314                       ".FinalOuterClass",   "t",
315                       ".FinalOuterClass[]", "t"
316 
317                       };
318 
319             String names2[] = {
320 
321                       "boolean[][][][]", "t",
322                       "byte[][][][]",    "t",
323                       "char[][][][]",    "t",
324                       "double[][][][]",  "t",
325                       "float[][][][]",   "t",
326                       "int[][][][]",     "t",
327                       "long[][][][]",    "t",
328                       "short[][][][]",   "t"
329 
330                       };
331 
332             String names3[] = {
333 
334                       "java.lang.Boolean",   "t",
335                       "java.lang.Byte",      "t",
336                       "java.lang.Character", "t",
337                       "java.lang.Double",    "t",
338                       "java.lang.Float",     "t",
339                       "java.lang.Integer",   "t",
340                       "java.lang.Long",      "t",
341                       "java.lang.Short",     "t"
342 
343                       };
344 
345             log2("----- Cases for testing: ReferenceTypes");
346             for (int i1 = 0; i1 < names1.length; i1 += 2) {
347 
348                 String typeForCheck = mName + names1[i1];
349                 log2("......typeForCheck: " + typeForCheck);
350 
351                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
352 
353                 List classList = vm.classesByName(typeForCheck);
354                 if (classList.size() != 1) {
355                     log3("ERROR: classList.size() != 1  : " + classList.size());
356                     testExitCode = FAILED;
357                     continue;
358                 }
359 
360                 log2("      getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
361                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
362 
363                 log2("      getting: boolean isFinal = referenceType.isfinal();");
364                 boolean isFinal = referenceType.isFinal();
365 
366                 if ( names1[i1+1].equals("t") ) {
367                     log2("      expected value of isFinal is 'true'");
368                     if (!isFinal) {
369                         log3("ERROR: isfinal != true for: " + typeForCheck);
370                         testExitCode = FAILED;
371                     }
372                 } else {
373                     log2("      expected value of isFinal is 'false'");
374                     if (isFinal) {
375                         log3("ERROR: isfinal != false for: " + typeForCheck);
376                         testExitCode = FAILED;
377                     }
378                 }
379             }
380 
381             log2("----- Cases for testing: primitive type arrays");
382             for (int i1 = 0; i1 < names2.length; i1 += 2) {
383

385                 log2("......typeForCheck: " + typeForCheck);
386

395                 log2("       getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
396                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
397 
398                 log2("       getting: boolean isFinal = referenceType.isfinal();");
399                 boolean isFinal = referenceType.isFinal();
400 
401                 log2("      expected value of isFinal is 'true'");
402                 if (!isFinal) {
403                     log3("ERROR: isfinal != true for: " + typeForCheck);
404                     testExitCode = FAILED;
405                 }
406 
407             }
408 
409             log2("----- Cases for testing: primitive classes");
410             for (int i1 = 0; i1 < names3.length; i1 += 2) {

412                 String typeForCheck = names3[i1];

432                     testExitCode = FAILED;
433                 }

435             }

442    /*
443     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
444     *                                             String, String, String)
445     *
446     * It sets up a breakpoint at given line number within a given method in a given class
447     * for a given thread.
448     *
449     * Return value: BreakpointRequest object  in case of success
450     *
451     * JDITestRuntimeException   in case of an Exception thrown within the method
452     */
453 
454     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
455                                                   ReferenceType testedClass,
456                                                   String methodName,
457                                                   String bpLine,
458                                                   String property)
459             throws JDITestRuntimeException {
460 
461         log2("......setting up a breakpoint:");
462         log2("       thread: " + thread + "; class: " + testedClass +
463                         "; method: " + methodName + "; line: " + bpLine);
464 
465         List              alllineLocations = null;
466         Location          lineLocation     = null;
467         BreakpointRequest breakpRequest    = null;
468 
469         try {
470             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
471 
472             alllineLocations = method.allLineLocations();
473 
474             int n =
475                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
476             if (n > alllineLocations.size()) {
477                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
478             } else {
479                 lineLocation = (Location) alllineLocations.get(n);
480                 try {
481                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
482                     breakpRequest.putProperty("number", property);
483                     breakpRequest.addThreadFilter(thread);
484                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
485                 } catch ( Exception e1 ) {
486                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
487                     breakpRequest    = null;
488                 }
489             }
490         } catch ( Exception e2 ) {
491             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
492             breakpRequest    = null;
493         }
494 
495         if (breakpRequest == null) {
496             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
497             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
498         }
499 
500         log2("      a breakpoint has been set up");
501         return breakpRequest;
502     }

505     private void getEventSet()
506                  throws JDITestRuntimeException {
507         try {
508 //            log2("       eventSet = eventQueue.remove(waitTime);");
509             eventSet = eventQueue.remove(waitTime);
510             if (eventSet == null) {
511                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
512             }
513 //            log2("       eventIterator = eventSet.eventIterator;");
514             eventIterator = eventSet.eventIterator();
515         } catch ( Exception e ) {
516             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
517         }
518     }

521     private void breakpointForCommunication()
522                  throws JDITestRuntimeException {
523 
524         log2("breakpointForCommunication");
525         getEventSet();
526 
527         if (eventIterator.nextEvent() instanceof BreakpointEvent)
528             return;
529 
530         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
531     }
