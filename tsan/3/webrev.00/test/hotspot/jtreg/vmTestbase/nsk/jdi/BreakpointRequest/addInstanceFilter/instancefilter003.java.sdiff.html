389     private ThreadReference threadByName(String name)
390                  throws JDITestRuntimeException {
391 
392         List         all = vm.allThreads();
393         ListIterator li  = all.listIterator();
394 
395         for (; li.hasNext(); ) {
396             ThreadReference thread = (ThreadReference) li.next();
397             if (thread.name().equals(name))
398                 return thread;
399         }
400         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
