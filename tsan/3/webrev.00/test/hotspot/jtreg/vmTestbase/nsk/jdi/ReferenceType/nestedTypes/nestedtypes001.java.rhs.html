 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.nestedTypes()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Cases for testing include only ClassTypes and InterfaceTypes.<BR>
 46  * <BR>
 47  * The test works as follows.                                   <BR>
 48  * Upon launching debuggee's VM which will be suspended,        <BR>
 49  * a debugger waits for the VMStartEvent within a predefined    <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.<BR>
 51  * Upon getting the VMStartEvent, it makes the request          <BR>
 52  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,  <BR>
 53  * resumes the VM, and waits for the event within the predefined<BR>
 54  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 55  * Upon getting the ClassPrepareEvent,                          <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,<BR>
 57  * the debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 58  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 59  * to be suspended and to inform the debugger with the event.           <BR>
 60  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 61  * At the end, the debuggee changes the value of the "instruction"      <BR>
 62  * to inform the debugger of checks finished, and both end.             <BR>
 63  */

133     private int runThis (String argv[], PrintStream out) {
134 
135         argsHandler     = new ArgumentHandler(argv);
136         logHandler      = new Log(out, argsHandler);
137         Binder binder   = new Binder(argsHandler, logHandler);
138 
139         waitTime        = argsHandler.getWaitTime() * 60000;
140 
141         try {
142             log2("launching a debuggee :");
143             log2("       " + debuggeeName);
144             if (argsHandler.verbose()) {
145                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
146             } else {
147                 debuggee = binder.bindToDebugee(debuggeeName);
148             }
149             if (debuggee == null) {
150                 log3("ERROR: no debuggee launched");
151                 return FAILED;
152             }
153             log2("debuggee launched");
154         } catch ( Exception e ) {
155             log3("ERROR: Exception : " + e);
156             log2("       test cancelled");
157             return FAILED;
158         }
159 
160         debuggee.redirectOutput(logHandler);
161 
162         vm = debuggee.VM();
163 
164         eventQueue = vm.eventQueue();
165         if (eventQueue == null) {
166             log3("ERROR: eventQueue == null : TEST ABORTED");
167             vm.exit(PASS_BASE);
168             return FAILED;
169         }
170 
171         log2("invocation of the method runTest()");
172         switch (runTest()) {
173 
174             case 0 :  log2("test phase has finished normally");
175                       log2("   waiting for the debuggee to finish ...");
176                       debuggee.waitFor();
177 
178                       log2("......getting the debuggee's exit status");
179                       int status = debuggee.getStatus();
180                       if (status != PASS_BASE) {
181                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
182                               status + " != PASS_BASE");
183                           testExitCode = FAILED;
184                       } else {
185                           log2("......debuggee returned expected exit status: " +
186                               status + " == PASS_BASE");
187                       }
188                       break;
189 
190             default : log3("ERROR: runTest() returned unexpected value");
191 
192             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
193                       log2("......forcing: vm.exit();");
194                       testExitCode = FAILED;
195                       try {
196                           vm.exit(PASS_BASE);
197                       } catch ( Exception e ) {
198                           log3("ERROR: Exception : e");
199                       }
200                       break;
201 
202             case 2 :  log3("test cancelled due to VMDisconnectedException");
203                       log2("......trying: vm.process().destroy();");
204                       testExitCode = FAILED;
205                       try {
206                           Process vmProcess = vm.process();
207                           if (vmProcess != null) {
208                               vmProcess.destroy();
209                           }
210                       } catch ( Exception e ) {
211                           log3("ERROR: Exception : e");
212                       }
213                       break;
214             }
215 
216         return testExitCode;
217     }

220    /*
221     * Return value: 0 - normal end of the test
222     *               1 - ubnormal end of the test
223     *               2 - VMDisconnectedException while test phase
224     */
225 
226     private int runTest() {
227 
228         try {
229             testRun();
230 
231             log2("waiting for VMDeathEvent");
232             getEventSet();
233             if (eventIterator.nextEvent() instanceof VMDeathEvent)
234                 return 0;
235 
236             log3("ERROR: last event is not the VMDeathEvent");
237             return 1;
238         } catch ( VMDisconnectedException e ) {
239             log3("ERROR: VMDisconnectedException : " + e);
240             return 2;
241         } catch ( Exception e ) {
242             log3("ERROR: Exception : " + e);
243             return 1;
244         }
245 
246     }

248     private void testRun()
249                  throws JDITestRuntimeException, Exception {
250 
251         eventRManager = vm.eventRequestManager();
252 
253         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
254         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
255         cpRequest.addClassFilter(debuggeeName);
256 
257         cpRequest.enable();
258         vm.resume();
259         getEventSet();
260         cpRequest.disable();
261 
262         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
263         debuggeeClass = event.referenceType();
264 
265         if (!debuggeeClass.name().equals(debuggeeName))
266            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
267 
268         log2("      received: ClassPrepareEvent for debuggeeClass");
269 
270         String bPointMethod = "methodForCommunication";
271         String lineForComm  = "lineForComm";
272         BreakpointRequest bpRequest;
273 
274         try {
275             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
276                                           debuggeeClass,
277                                           bPointMethod, lineForComm, "zero");
278         } catch ( Exception e ) {
279             throw e;
280         }
281         bpRequest.enable();
282 
283     //------------------------------------------------------  testing section
284 
285         log1("     TESTING BEGINS");
286 
287         for (int i = 0; ; i++) {
288 
289             vm.resume();
290             breakpointForCommunication();
291 
292             int instruction = ((IntegerValue)
293                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
294 
295             if (instruction == 0) {
296                 vm.resume();
297                 break;
298             }
299 
300             log1("  new check: # " + i);
301 
302             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
303 
304 
305            String names1[] = {
306 
307                       ".nestedtypes001aTestClass$NestedClass",
308                       ".nestedtypes001aTestClass$NestedClass[]",
309 
310                       ".nestedtypes001aTestClass$NestedIface",
311                       ".nestedtypes001aTestClass$NestedIface[]",
312 
313 
314                       ".nestedtypes001aTestClass$StaticNestedClass",
315                       ".nestedtypes001aTestClass$StaticNestedClass[]"
316 
317                       };
318 
319             String typeForCheck = mName + ".nestedtypes001aTestClass";
320             log2("......typeForCheck: " + typeForCheck);
321 
322             log2("      getting: List classList = vm.classesByName(typeForCheck);");
323             List classes = vm.classesByName(typeForCheck);
324             if (classes.size() != 1) {
325                 log3("ERROR: classes.size() != 1  : " + classes.size());
326                 testExitCode = FAILED;
327                 continue;
328             }
329 
330             log2("      getting: ReferenceType testedType  = (ReferenceType) classList.get(0);");
331             ReferenceType testedType  = (ReferenceType) classes.get(0);
332 
333             log2("      getting: List netsedTypes = testedType.nestedTypes();");
334             List nestedTypes = testedType.nestedTypes();
335             if (nestedTypes.size() != names1.length) {
336                 log3("ERROR: nestedTypes.size() != names1.length : " + nestedTypes.size());
337                 testExitCode = FAILED;
338                 continue;
339             }
340 
341 
342             log2("----- Cases for testing: ReferenceTypes");
343             for (int i1 = 0; i1 < names1.length; i1++) {
344 
345                 typeForCheck = mName + names1[i1];
346                 log2("......typeForCheck: " + typeForCheck);
347 
348                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
349                 List classList = vm.classesByName(typeForCheck);
350                 if (classList.size() != 1) {
351                     log3("ERROR: classList.size() != 1  : " + classList.size());
352                     testExitCode = FAILED;
353                     continue;
354                 }
355 
356                 log2("      getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
357                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
358 
359                 boolean equals = false;
360 
361                 ListIterator li = nestedTypes.listIterator();
362                 for (; li.hasNext(); ) {
363                     if (li.next().equals(referenceType)) {
364                         equals = true;
365                         break;
366                     }
367                 }
368 
369                 if (!equals) {
370                     log3("ERROR: referenceType is not in the List: " + typeForCheck);
371                     testExitCode = FAILED;
372                 }
373             }
374 
375             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
376         }
377         log1("    TESTING ENDS");
378         return;
379     }

381    /*
382     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
383     *                                             String, String, String)
384     *
385     * It sets up a breakpoint at given line number within a given method in a given class
386     * for a given thread.
387     *
388     * Return value: BreakpointRequest object  in case of success
389     *
390     * JDITestRuntimeException   in case of an Exception thrown within the method
391     */
392 
393     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
394                                                   ReferenceType testedClass,
395                                                   String methodName,
396                                                   String bpLine,
397                                                   String property)
398             throws JDITestRuntimeException {
399 
400         log2("......setting up a breakpoint:");
401         log2("       thread: " + thread + "; class: " + testedClass +
402                         "; method: " + methodName + "; line: " + bpLine);
403 
404         List              alllineLocations = null;
405         Location          lineLocation     = null;
406         BreakpointRequest breakpRequest    = null;
407 
408         try {
409             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
410 
411             alllineLocations = method.allLineLocations();
412 
413             int n =
414                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
415             if (n > alllineLocations.size()) {
416                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
417             } else {
418                 lineLocation = (Location) alllineLocations.get(n);
419                 try {
420                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
421                     breakpRequest.putProperty("number", property);
422                     breakpRequest.addThreadFilter(thread);
423                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
424                 } catch ( Exception e1 ) {
425                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
426                     breakpRequest    = null;
427                 }
428             }
429         } catch ( Exception e2 ) {
430             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
431             breakpRequest    = null;
432         }
433 
434         if (breakpRequest == null) {
435             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
436             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
437         }
438 
439         log2("      a breakpoint has been set up");
440         return breakpRequest;
441     }

444     private void getEventSet()
445                  throws JDITestRuntimeException {
446         try {
447 //            log2("       eventSet = eventQueue.remove(waitTime);");
448             eventSet = eventQueue.remove(waitTime);
449             if (eventSet == null) {
450                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
451             }
452 //            log2("       eventIterator = eventSet.eventIterator;");
453             eventIterator = eventSet.eventIterator();
454         } catch ( Exception e ) {
455             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
456         }
457     }

460     private void breakpointForCommunication()
461                  throws JDITestRuntimeException {
462 
463         log2("breakpointForCommunication");
464         getEventSet();
465 
466         if (eventIterator.nextEvent() instanceof BreakpointEvent)
467             return;
468 
469         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
470     }
