 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.thread()</code>                <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a ThreadReference object,                         <BR>
 46  * an argument of the method                                            <BR>
 47  *    EventRequestManager.createAccessWatchpointRequest(),              <BR>
 48  * is equal to one returned by the method StepRequest.thread()          <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates a tested thread                               <BR>
 68  *   and invokes the methodForCommunication to be suspended and         <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an StepRequest with the thread1,                         <BR>
 72  *   - gets the thread of the Request, the thread2,                     <BR>
 73  *     with the method StepRequest.thread(),                            <BR>
 74  *   - and compares the objects thrtead1 and thread2.                   <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

157     private int runThis (String argv[], PrintStream out) {
158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : " + e);
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

244    /*
245     * Return value: 0 - normal end of the test
246     *               1 - ubnormal end of the test
247     *               2 - VMDisconnectedException while test phase
248     */

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if (eventIterator.nextEvent() instanceof VMDeathEvent)
258                 return 0;
259 
260             log3("ERROR: last event is not the VMDeathEvent");
261             return 1;
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }
269 
270     }

272     private void testRun()
273                  throws JDITestRuntimeException, Exception {
274 
275         eventRManager = vm.eventRequestManager();
276 
277         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
278         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
279         cpRequest.addClassFilter(debuggeeName);
280 
281         cpRequest.enable();
282         vm.resume();
283         getEventSet();
284         cpRequest.disable();
285 
286         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
287         debuggeeClass = event.referenceType();
288 
289         if (!debuggeeClass.name().equals(debuggeeName))
290            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
291 
292         log2("      received: ClassPrepareEvent for debuggeeClass");
293 
294         String bPointMethod = "methodForCommunication";
295         String lineForComm  = "lineForComm";
296 
297         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
298 
299         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
300                                              debuggeeClass,
301                                             bPointMethod, lineForComm, "zero");
302         bpRequest.enable();
303 
304     //------------------------------------------------------  testing section
305 
306         log1("     TESTING BEGINS");
307 
308         EventRequest  eventRequest1 = null;
309 
310         ThreadReference thread1     = null;
311         String          threadName1 = "thread1";
312 
313         ThreadReference thread2;
314 
315 
316         for (int i = 0; ; i++) {
317 
318             vm.resume();
319             breakpointForCommunication();
320 
321             int instruction = ((IntegerValue)
322                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
323 
324             if (instruction == 0) {
325                 vm.resume();
326                 break;
327             }
328 
329 
330             log1(":::::: case: # " + i);
331 
332             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
333 
334             switch (i) {
335 
336               case 0:
337                      thread1 = debuggee.threadByNameOrThrow(threadName1);
338 
339                      log2("......setting up StepRequest with size StepRequest.STEP_MIN");
340                      eventRequest1 = setting24StepRequest(thread1, StepRequest.STEP_MIN,
341                                               StepRequest.STEP_INTO);
342 
343                      log2("......getting: size = ((StepRequest) eventRequest1).thread();");
344                      thread2 = ((StepRequest) eventRequest1).thread();
345 
346                      log2("      checking up on equality of values");
347                      if ( !thread2.equals(thread1) ) {
348                          testExitCode = FAILED;
349                          log3("ERROR: !thread2.equals(thread1) : " + thread2);
350                      }
351 
352                       break;
353 
354               default:
355                       throw new JDITestRuntimeException("** default case 2 **");
356             }
357 
358             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
359         }
360         log1("    TESTING ENDS");
361         return;
362     }

364    /*
365     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
366     *                                             String, String, String)
367     *
368     * It sets up a breakpoint at given line number within a given method in a given class
369     * for a given thread.
370     *
371     * Return value: BreakpointRequest object  in case of success
372     *
373     * JDITestRuntimeException   in case of an Exception thrown within the method
374     */
375 
376     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
377                                                   ReferenceType testedClass,
378                                                   String methodName,
379                                                   String bpLine,
380                                                   String property)
381             throws JDITestRuntimeException {
382 
383         log2("......setting up a breakpoint:");
384         log2("       thread: " + thread + "; class: " + testedClass +
385                         "; method: " + methodName + "; line: " + bpLine);
386 
387         List              alllineLocations = null;
388         Location          lineLocation     = null;
389         BreakpointRequest breakpRequest    = null;
390 
391         try {
392             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
393 
394             alllineLocations = method.allLineLocations();
395 
396             int n =
397                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
398             if (n > alllineLocations.size()) {
399                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
400             } else {
401                 lineLocation = (Location) alllineLocations.get(n);
402                 try {
403                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
404                     breakpRequest.putProperty("number", property);
405                     breakpRequest.addThreadFilter(thread);
406                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
407                 } catch ( Exception e1 ) {
408                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
409                     breakpRequest    = null;
410                 }
411             }
412         } catch ( Exception e2 ) {
413             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
414             breakpRequest    = null;
415         }
416 
417         if (breakpRequest == null) {
418             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
419             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
420         }
421 
422         log2("      a breakpoint has been set up");
423         return breakpRequest;
424     }

427     private void getEventSet()
428                  throws JDITestRuntimeException {
429         try {
430 //            log2("       eventSet = eventQueue.remove(waitTime);");
431             eventSet = eventQueue.remove(waitTime);
432             if (eventSet == null) {
433                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
434             }
435 //            log2("       eventIterator = eventSet.eventIterator;");
436             eventIterator = eventSet.eventIterator();
437         } catch ( Exception e ) {
438             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
439         }
440     }

443     private void breakpointForCommunication()
444                  throws JDITestRuntimeException {
445 
446         log2("breakpointForCommunication");
447         getEventSet();
448 
449         if (eventIterator.nextEvent() instanceof BreakpointEvent)
450             return;
451 
452         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
453     }
