 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a ThreadDeathEvent set,             <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up one to three new ThreadDeathRequests,         <BR>
 70  *    resumes the debuggee, and waits for the ThreadDeathEvents.        <BR>
 71  * - The debuggee starts new thread to be resulting in the events.      <BR>
 72  * - Upon getting new ThreadDeathEvent,                                 <BR>
 73  *   the debugger performs the check corresponding to the EventSet.     <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * In third phase, at the end,                                          <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  */

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : e");
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : e");
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
259                 log3("ERROR: last event is not the VMDeathEvent");
260                 return 1;
261             }
262 
263             log2("waiting for VMDisconnectEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
266                 log3("ERROR: last event is not the VMDisconnectEvent");
267                 return 1;
268             }
269 
270             return 0;
271 
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279 
280     }

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         log2("......getting ClassPrepareEvent for debuggee's class");
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
304 
305         String            bPointMethod = "methodForCommunication";
306         String            lineForComm  = "lineForComm";
307         BreakpointRequest bpRequest;
308         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
309         bpRequest = settingBreakpoint(mainThread,
310                                       debuggeeClass,
311                                       bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314         vm.resume();
315 
316     //------------------------------------------------------  testing section
317 
318         log1("     TESTING BEGINS");
319 
320         EventRequest eventRequest1 = null;
321         EventRequest eventRequest2 = null;
322         EventRequest eventRequest3 = null;
323 
324         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
325         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
327         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
328 
329         int policyExpected[] = {    SUSPEND_NONE,
330                                     SUSPEND_THREAD,
331                                     SUSPEND_ALL,
332                                     SUSPEND_THREAD,
333                                     SUSPEND_ALL,
334                                     SUSPEND_ALL,
335                                     SUSPEND_ALL      };
336         int policy = 0;
337 
338 
339         for (int i = 0; ; i++) {
340 
341             breakpointForCommunication();
342 
343             int instruction = ((IntegerValue)
344                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
345             if (instruction == 0) {
346                 vm.resume();
347                 break;
348             }
349 
350 
351             log1(":::::: case: # " + i);
352 
353             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
354 
355             switch (i) {
356 
357               case 0:
358                       eventRequest1 = settingThreadDeathRequest (
359                                              SUSPEND_NONE, "ThreadDeathRequest1");
360                       eventRequest1.enable();
361                       break;
362 
363               case 1:
364                       eventRequest2 = settingThreadDeathRequest (
365                                              SUSPEND_THREAD, "ThreadDeathRequest2");
366                       eventRequest2.enable();
367                       break;
368 
369               case 2:
370                       eventRequest3 = settingThreadDeathRequest (
371                                              SUSPEND_ALL, "ThreadDeathRequest3");
372                       eventRequest3.enable();
373                       break;
374 
375               case 3:
376                       eventRequest1 = settingThreadDeathRequest (
377                                              SUSPEND_NONE, "ThreadDeathRequest1");
378                       eventRequest1.enable();
379                       eventRequest2 = settingThreadDeathRequest (
380                                              SUSPEND_THREAD, "ThreadDeathRequest2");
381                       eventRequest2.enable();
382                       break;
383 
384               case 4:
385                       eventRequest1 = settingThreadDeathRequest (
386                                              SUSPEND_NONE, "ThreadDeathRequest1");
387                       eventRequest1.enable();
388                       eventRequest3 = settingThreadDeathRequest (
389                                              SUSPEND_ALL, "ThreadDeathRequest3");
390                       eventRequest3.enable();
391                       break;
392 
393               case 5:
394                       eventRequest2 = settingThreadDeathRequest (
395                                              SUSPEND_THREAD, "ThreadDeathRequest2");
396                       eventRequest2.enable();
397                       eventRequest3 = settingThreadDeathRequest (
398                                              SUSPEND_ALL, "ThreadDeathRequest3");
399                       eventRequest3.enable();
400                       break;
401 
402               case 6:
403                       eventRequest1 = settingThreadDeathRequest (
404                                              SUSPEND_NONE, "ThreadDeathRequest1");
405                       eventRequest1.enable();
406                       eventRequest2 = settingThreadDeathRequest (
407                                              SUSPEND_THREAD, "ThreadDeathRequest2");
408                       eventRequest2.enable();
409                       eventRequest3 = settingThreadDeathRequest (
410                                              SUSPEND_ALL, "ThreadDeathRequest3");
411                       eventRequest3.enable();
412                       break;
413 
414 
415               default:
416                       throw new JDITestRuntimeException("** default case 2 **");
417             }
418 
419             mainThread.resume();
420             getEventSet();
421 
422             if ( !(eventIterator.nextEvent() instanceof ThreadDeathEvent)) {
423                  log3("ERROR: new event is not ThreadDeathEvent");
424                  testExitCode = FAILED;
425             } else {
426                 log2("......got : instanceof ThreadDeathEvent");
427                 policy = eventSet.suspendPolicy();
428                 if (policy != policyExpected[i]) {
429                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
430                     log3("       eventSet.suspendPolicy() == " + policy);
431                     log3("       policyExpected           == " + policyExpected[i]);
432                     testExitCode = FAILED;
433                 }
434             }
435 
436             switch (i) {
437               case 0: eventRequest1.disable();                          break;
438               case 1: eventRequest2.disable();                          break;
439               case 2: eventRequest3.disable();                          break;
440               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
441               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
442               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
443               case 6: eventRequest1.disable(); eventRequest2.disable();
444                       eventRequest3.disable();                          break;
445            }
446 
447             switch (policy) {
448               case SUSPEND_NONE   :              break;
449               case SUSPEND_THREAD : vm.resume(); break;
450               case SUSPEND_ALL    : vm.resume(); break;
451               default: throw new JDITestRuntimeException("** default case 3 **");
452             }
453 
454             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
455         }
456         log1("    TESTING ENDS");
457         return;
458     }

460    /*
461     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
462     *                                             String, String, String)
463     *
464     * It sets up a breakpoint at given line number within a given method in a given class
465     * for a given thread.
466     *
467     * Return value: BreakpointRequest object  in case of success
468     *
469     * JDITestRuntimeException   in case of an Exception thrown within the method
470     */
471 
472     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
473                                                   ReferenceType testedClass,
474                                                   String methodName,
475                                                   String bpLine,
476                                                   String property)
477             throws JDITestRuntimeException {
478 
479         log2("......setting up a breakpoint:");
480         log2("       thread: " + thread + "; class: " + testedClass +
481                         "; method: " + methodName + "; line: " + bpLine);
482 
483         List              alllineLocations = null;
484         Location          lineLocation     = null;
485         BreakpointRequest breakpRequest    = null;
486 
487         try {
488             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
489 
490             alllineLocations = method.allLineLocations();
491 
492             int n =
493                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
494             if (n > alllineLocations.size()) {
495                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
496             } else {
497                 lineLocation = (Location) alllineLocations.get(n);
498                 try {
499                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
500                     breakpRequest.putProperty("number", property);
501                     breakpRequest.addThreadFilter(thread);
502                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
503                 } catch ( Exception e1 ) {
504                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
505                     breakpRequest    = null;
506                 }
507             }
508         } catch ( Exception e2 ) {
509             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
510             breakpRequest    = null;
511         }
512 
513         if (breakpRequest == null) {
514             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
515             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
516         }
517 
518         log2("      a breakpoint has been set up");
519         return breakpRequest;
520     }

523     private void getEventSet()
524                  throws JDITestRuntimeException {
525         try {
526 //            log2("       eventSet = eventQueue.remove(waitTime);");
527             eventSet = eventQueue.remove(waitTime);
528             if (eventSet == null) {
529                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
530             }
531 //            log2("       eventIterator = eventSet.eventIterator;");
532             eventIterator = eventSet.eventIterator();
533         } catch ( Exception e ) {
534             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
535         }
536     }

539     private void breakpointForCommunication()
540                  throws JDITestRuntimeException {
541 
542         log2("breakpointForCommunication");
543         getEventSet();
544 
545         if (eventIterator.nextEvent() instanceof BreakpointEvent)
546             return;
547 
548         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
549     }

555             throws JDITestRuntimeException {
556         try {
557             ThreadDeathRequest tsr = eventRManager.createThreadDeathRequest();
558 //            tsr.addThreadFilter(mainThread);

564             log3("ERROR: ATTENTION: Exception within settingThreadDeathRequest() : " + e);
565             log3("       ThreadDeathRequest HAS NOT BEEN SET UP");
566             throw new JDITestRuntimeException("** FAILURE to set up ThreadDeathRequest **");
567         }
