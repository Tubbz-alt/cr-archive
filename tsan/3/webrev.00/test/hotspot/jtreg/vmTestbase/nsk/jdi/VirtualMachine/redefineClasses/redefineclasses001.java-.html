 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.redefineClasses()</code>    <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    If any redefined methods have active stack frames,        <BR>
 47  *    those active frames continue to run the bytecodes of      <BR>
 48  *    the previous method.                                      <BR>
 49  *    The redefined methods will be used on new invokes.        <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                              <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assetion,                               <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new tested class object, objRF1, and invokes  <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the breakpointForCommunication Event,                 <BR>
 72  *   the debugger sets up the breakpoint bpRequest2 in the tested method<BR>
 73  *   redefineclasses001b.m2(), resumes the debuggee.                    <BR>
 74  * - The debuggee invokes objRF1.m1() method to be suspended            <BR>
 75  *   at the breakpoint in the method m2().                              <BR>
 76  * - Upon getting the breakpointInMethod Event,                         <BR>
 77  *   the debugger prepares mapClassToByte, invokes                      <BR>
 78  *         VirtualMachine.redefineClasses(mapClassToByte)               <BR>
 79  *   and resumes the debuggee.                                          <BR>
 80  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 81  *   once again and to inform the debugger with the event.              <BR>
 82  * - Upon getting seconf breakpointForCommunication Event,              <BR>
 83  *   the debugger sets up another breakpoint in the redefined method m2()<BR>
 84  *   and resumes the debuggee.                                          <BR>
 85  * - The debuggee invokes the method m1() to invoke m2() second time,   <BR>
 86  *   wher it will be suspended once again.                              <BR>
 87  * - Upon getting second breakpointInMethod Event,                      <BR>
 88  *   the debugger checks up the value of the class variable "i1"        <BR>
 89  *   which should be equal to the expected value.                       <BR>
 90  * <BR>
 91  * In third phase when at the end,                                      <BR>
 92  * the debuggee changes the value of the "instruction"                  <BR>
 93  * to inform the debugger of checks finished, and both end.             <BR>
 94  * <BR>
 95  */

174     private int runThis (String argv[], PrintStream out) {
175 
176         argsHandler     = new ArgumentHandler(argv);
177         logHandler      = new Log(out, argsHandler);
178         Binder binder   = new Binder(argsHandler, logHandler);
179 
180         waitTime        = argsHandler.getWaitTime() * 60000;
181 
182         try {
183             log2("launching a debuggee :");
184             log2("       " + debuggeeName);
185 
186             debuggee = binder.bindToDebugee(debuggeeName);
187 
188             if (debuggee == null) {
189                 log3("ERROR: no debuggee launched");
190                 return FAILED;
191             }
192             log2("debuggee launched");
193         } catch ( Exception e ) {
194             log3("ERROR: Exception : " + e);
195             log2("       test cancelled");
196             return FAILED;
197         }
198 
199         debuggee.redirectOutput(logHandler);
200 
201         vm = debuggee.VM();
202 
203         eventQueue = vm.eventQueue();
204         if (eventQueue == null) {
205             log3("ERROR: eventQueue == null : TEST ABORTED");
206             vm.exit(PASS_BASE);
207             return FAILED;
208         }
209 
210         log2("invocation of the method runTest()");
211         switch (runTest()) {
212 
213             case 0 :  log2("test phase has finished normally");
214                       log2("   waiting for the debuggee to finish ...");
215                       debuggee.waitFor();
216 
217                       log2("......getting the debuggee's exit status");
218                       int status = debuggee.getStatus();
219                       if (status != PASS_BASE) {
220                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
221                               status + " != PASS_BASE");
222                           testExitCode = FAILED;
223                       } else {
224                           log2("......debuggee returned expected exit status: " +
225                               status + " == PASS_BASE");
226                       }
227                       break;
228 
229             default : log3("ERROR: runTest() returned unexpected value");
230 
231             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
232                       log2("......forcing: vm.exit();");
233                       testExitCode = FAILED;
234                       try {
235                           vm.exit(PASS_BASE);
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240 
241             case 2 :  log3("test cancelled due to VMDisconnectedException");
242                       log2("......trying: vm.process().destroy();");
243                       testExitCode = FAILED;
244                       try {
245                           Process vmProcess = vm.process();
246                           if (vmProcess != null) {
247                               vmProcess.destroy();
248                           }
249                       } catch ( Exception e ) {
250                           log3("ERROR: Exception : e");
251                       }
252                       break;
253             }
254 
255         return testExitCode;
256     }

259    /*
260     * Return value: 0 - normal end of the test
261     *               1 - ubnormal end of the test
262     *               2 - VMDisconnectedException while test phase
263     */
264 
265     private int runTest() {
266 
267         try {
268             testRun();
269 
270             log2("waiting for VMDeathEvent");
271             getEventSet();
272             if (eventIterator.nextEvent() instanceof VMDeathEvent)
273                 return 0;
274 
275             log3("ERROR: last event is not the VMDeathEvent");
276             return 1;
277         } catch ( VMDisconnectedException e ) {
278             log3("ERROR: VMDisconnectedException : " + e);
279             return 2;
280         } catch ( Exception e ) {
281             log3("ERROR: Exception : " + e);
282             return 1;
283         }
284 
285     }

287     private void testRun()
288                  throws JDITestRuntimeException, Exception {
289 
290         eventRManager = vm.eventRequestManager();
291 
292         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
293         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
294         cpRequest.addClassFilter(debuggeeName);
295 
296         cpRequest.enable();
297         vm.resume();
298         getEventSet();
299         cpRequest.disable();
300 
301         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
302         debuggeeClass = event.referenceType();
303 
304         if (!debuggeeClass.name().equals(debuggeeName))
305            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
306 
307         log2("      received: ClassPrepareEvent for debuggeeClass");
308 
309 
310         if ( !vm.canRedefineClasses() ) {
311             log2("......vm.canRedefineClasses() == false : test is cancelled");
312             vm.resume();
313             return;
314         }
315 
316 
317         String bPointMethod = "methodForCommunication";
318         String lineForComm  = "lineForComm";
319         BreakpointRequest bpRequest;
320 
321         bpRequest = settingBreakpoint(threadByName("main"),
322                                       debuggeeClass,
323                                       bPointMethod, lineForComm, "zero");
324         bpRequest.enable();
325 
326     //------------------------------------------------------  testing section
327 
328         log1("     TESTING BEGINS");
329 
330         String bpClassName  = "nsk.jdi.VirtualMachine.redefineClasses.redefineclasses001b";
331         String bpMethodName = "m2";
332         String bpLineName   = "bpline";
333         String varName      = "i1";
334         int    varValue     = 2;
335 
336         BreakpointRequest bpRequest2 = null;
337         BreakpointRequest bpRequest3 = null;
338         ReferenceType     bpClass    = null;
339 
340 
341         for (int i = 0; ; i++) {
342 
343             vm.resume();
344             breakpointForCommunication();
345 
346             int instruction = ((IntegerValue)
347                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
348 
349             if (instruction == 0) {
350                 vm.resume();
351                 break;
352             }
353 
354             log1(":::::: case: # " + i);
355 
356             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
357 
358             switch (i) {
359               case 0:
360                   List          classes = vm.classesByName(bpClassName);
361                   bpClass = (ReferenceType) classes.get(0);
362                   bpRequest2 = settingBreakpoint(threadByName("main"),
363                                           bpClass,
364                                           bpMethodName, bpLineName, "one");
365                   bpRequest2.enable();
366 
367                   vm.resume();
368                   breakpointInMethod("one");
369                   bpRequest2.disable();
370 
371                   try {
372                       log2("......vm.redefineClasses(mapClassToBytes());");
373                       vm.redefineClasses(mapClassToBytes());
374                   } catch ( Exception e ) {
375                       log3("ERROR:  Exception: " + e);
376                       testExitCode = FAILED;
377                       throw e;
378                   } catch ( Error e ) {
379                       log3("ERROR:  Error: " + e);
380                       testExitCode = FAILED;
381                       throw e;
382                   }
383                   break;
384 
385               case 1:
386 
387                   bpRequest3 = settingBreakpoint(threadByName("main"),
388                                           bpClass,
389                                           bpMethodName, bpLineName, "one");
390 
391                   bpRequest3.enable();
392 
393                   vm.resume();
394                   breakpointInMethod("one");
395                   bpRequest3.disable();
396 
397                   int i1 = ( (IntegerValue)
398                       bpClass.getValue(bpClass.fieldByName(varName) ) ).value();
399 
400                   if (i1 != varValue) {
401                       testExitCode = FAILED;
402                       log3("ERROR: value of i1 != 2 (expected) but : " + i1);
403                   }
404                   break;
405 
406               default:
407                   throw new JDITestRuntimeException ("** default case **");
408             }
409             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
410         }
411 // this is for debugging
412 // testExitCode = FAILED;
413 
414         vm.resume();
415         log1("    TESTING ENDS");
416         return;
417     }

419     private ThreadReference threadByName(String name)
420                  throws JDITestRuntimeException {
421 
422         List         all = vm.allThreads();
423         ListIterator li  = all.listIterator();
424 
425         for (; li.hasNext(); ) {
426             ThreadReference thread = (ThreadReference) li.next();
427             if (thread.name().equals(name))
428                 return thread;
429         }
430         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
431     }

433    /*
434     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
435     *                                             String, String, String)
436     *
437     * It sets up a breakpoint at given line number within a given method in a given class
438     * for a given thread.
439     *
440     * Return value: BreakpointRequest object  in case of success
441     *
442     * JDITestRuntimeException   in case of an Exception thrown within the method
443     */
444 
445     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
446                                                   ReferenceType testedClass,
447                                                   String methodName,
448                                                   String bpLine,
449                                                   String property)
450             throws JDITestRuntimeException {
451 
452         log2("......setting up a breakpoint:");
453         log2("       thread: " + thread + "; class: " + testedClass +
454                         "; method: " + methodName + "; line: " + bpLine);
455 
456         List              alllineLocations = null;
457         Location          lineLocation     = null;
458         BreakpointRequest breakpRequest    = null;
459 
460         try {
461             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
462             alllineLocations = method.allLineLocations();
463             int n =
464                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
465 
466             if (n > alllineLocations.size()) {
467                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
468             } else {
469                 lineLocation = (Location) alllineLocations.get(n);
470                 try {
471                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
472                     breakpRequest.putProperty("number", property);
473                     breakpRequest.addThreadFilter(thread);
474                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
475                 } catch ( Exception e1 ) {
476                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
477                     breakpRequest    = null;
478                 }
479             }
480         } catch ( Exception e2 ) {
481             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
482             breakpRequest    = null;
483         }
484 
485         if (breakpRequest == null) {
486             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
487             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
488         }
489 
490         log2("      a breakpoint has been set up");
491         return breakpRequest;
492     }

495     private void getEventSet()
496                  throws JDITestRuntimeException {
497         try {
498 //            log2("       eventSet = eventQueue.remove(waitTime);");
499             eventSet = eventQueue.remove(waitTime);
500             if (eventSet == null) {
501                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
502             }
503 //            log2("       eventIterator = eventSet.eventIterator;");
504             eventIterator = eventSet.eventIterator();
505         } catch ( Exception e ) {
506             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
507         }
508     }

511     private void breakpointForCommunication()
512                  throws JDITestRuntimeException {
513 
514         log2("breakpointForCommunication");
515         getEventSet();
516 
517         if (eventIterator.nextEvent() instanceof BreakpointEvent)
518             return;
519 
520         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
521     }

525     private void breakpointInMethod(String number)
526                  throws JDITestRuntimeException {
527 
528         Event event;
529 
530         getEventSet();
531         event = eventIterator.nextEvent();
532 
533         if ( !(event instanceof BreakpointEvent) )
534             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
535 
536         log2("---->: request().getProperty == " +
537              event.request().getProperty("number"));
538 
539         if ( event.request().getProperty("number").equals(number) )
540             return;
541 
542         throw new JDITestRuntimeException("** UNEXPECTED breakpoint **");
543     }

546     private Map<? extends com.sun.jdi.ReferenceType,byte[]> mapClassToBytes()
547                 throws JDITestRuntimeException {
548 
549         byte []arrayToRedefine;
550 
551 //   -----------------------------
552         log2("......getting: String[] args = argsHandler.getArguments();");
553         String[] args = argsHandler.getArguments();
554         if (args.length <= 0) {
555             log3("ERROR: args.length <= 0");
556             testExitCode = FAILED;
557             throw new JDITestRuntimeException("** args.length <= 0 **");
558         }
559         String testDir = args[0];
560         log2("...... testDir = " + testDir);
561 
562 
563         String filePrefix = File.separator + "nsk"
564                           + File.separator + "jdi"
565                           + File.separator + "VirtualMachine"
566                           + File.separator + "redefineClasses";
567 
568         String fileToRedefineName    = testDir
569                                        + File.separator + "newclass"
570                                        + filePrefix
571                                        + File.separator + "redefineclasses001b.class";
572 
573         log2("...... fileToRedefineName = " + fileToRedefineName);
574 //   -----------------------------
575 
576         try {
577             log2("......File fileToRedefine = new File(fileToRedefineName);");
578             File fileToRedefine    = new File(fileToRedefineName);
579             if (fileToRedefine.exists())
580                 log2("       fileToRedefine.exists()");
581             else {
582                 log3("ERROR: fileToRedefine doesn't exist");
583                 testExitCode = FAILED;
584             }
585 
586             if (testExitCode == FAILED)
587                 throw new JDITestRuntimeException("** testExitCode == FAILED **");
588 
589             int fileToRedefineLength = (int) fileToRedefine.length();
590             log2("       fileToRedefineLength == " + fileToRedefineLength);
591 
592             FileInputStream inputFile = new FileInputStream(fileToRedefine);
593             arrayToRedefine = new byte[fileToRedefineLength];
594             inputFile.read(arrayToRedefine);
595             inputFile.close();
596         } catch ( IOException e ) {
597             log3("ERROR: IOException: " + e);
598             throw new JDITestRuntimeException("** IOException **");
599         }
600 
601         String testClassName = "nsk.jdi.VirtualMachine.redefineClasses.redefineclasses001b";
602         List classes = vm.classesByName(testClassName);
603         ReferenceType testClass = (ReferenceType) classes.get(0);
604 
605         HashMap<com.sun.jdi.ReferenceType,byte[]> mapForClass = new HashMap<com.sun.jdi.ReferenceType,byte[]>();
606         mapForClass.put(testClass, arrayToRedefine);
607 
608         return mapForClass;
609     }
