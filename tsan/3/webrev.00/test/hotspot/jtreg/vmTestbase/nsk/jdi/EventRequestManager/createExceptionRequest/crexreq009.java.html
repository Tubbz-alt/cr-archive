 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createExceptionRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - Creates a new disabled ExceptionRequest.                           <BR>
 47  * - A specific exception type and its subclasses                       <BR>
 48  *   can be selected for exception events.                              <BR>
 49  * - Caught exceptions, uncaught exceptions, or both can be selected.   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee prepares new testcase and invokes                     <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent,                                  <BR>
 72  *   the debugger performs the checks required.                         <BR>
 73  * <BR>
 74  * In third phase, at the end of the test, the debuggee changes         <BR>
 75  * the value of the "instruction" which the debugger and debuggee       <BR>
 76  * use to inform each other of needed actions,  and both end.           <BR>
 77  * <BR>
 78  */

152     private int runThis (String argv[], PrintStream out) {
153 
154         argsHandler     = new ArgumentHandler(argv);
155         logHandler      = new Log(out, argsHandler);
156         Binder binder   = new Binder(argsHandler, logHandler);
157 
158         waitTime        = argsHandler.getWaitTime() * 60000;
159 
160         try {
161             log2("launching a debuggee :");
162             log2("       " + debuggeeName);
163             if (argsHandler.verbose()) {
164                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
165             } else {
166                 debuggee = binder.bindToDebugee(debuggeeName);
167             }
168             if (debuggee == null) {
169                 log3("ERROR: no debuggee launched");
170                 return FAILED;
171             }
172             log2("debuggee launched");
173         } catch ( Exception e ) {
174             log3("ERROR: Exception : " + e);
175             log2("       test cancelled");
176             return FAILED;
177         }
178 
179         debuggee.redirectOutput(logHandler);
180 
181         vm = debuggee.VM();
182 
183         eventQueue = vm.eventQueue();
184         if (eventQueue == null) {
185             log3("ERROR: eventQueue == null : TEST ABORTED");
186             vm.exit(PASS_BASE);
187             return FAILED;
188         }
189 
190         log2("invocation of the method runTest()");
191         switch (runTest()) {
192 
193             case 0 :  log2("test phase has finished normally");
194                       log2("   waiting for the debuggee to finish ...");
195                       debuggee.waitFor();
196 
197                       log2("......getting the debuggee's exit status");
198                       int status = debuggee.getStatus();
199                       if (status != PASS_BASE) {
200                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
201                               status + " != PASS_BASE");
202                           testExitCode = FAILED;
203                       } else {
204                           log2("......debuggee returned expected exit status: " +
205                               status + " == PASS_BASE");
206                       }
207                       break;
208 
209             default : log3("ERROR: runTest() returned unexpected value");
210 
211             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
212                       log2("......forcing: vm.exit();");
213                       testExitCode = FAILED;
214                       try {
215                           vm.exit(PASS_BASE);
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : " + e);
218                       }
219                       break;
220 
221             case 2 :  log3("test cancelled due to VMDisconnectedException");
222                       log2("......trying: vm.process().destroy();");
223                       testExitCode = FAILED;
224                       try {
225                           Process vmProcess = vm.process();
226                           if (vmProcess != null) {
227                               vmProcess.destroy();
228                           }
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : " + e);
231                       }
232                       break;
233             }
234 
235         return testExitCode;
236     }

240     * Return value: 0 - normal end of the test
241     *               1 - ubnormal end of the test
242     *               2 - VMDisconnectedException while test phase
243     */

245     private int runTest() {
246 
247         try {
248             testRun();
249 
250             log2("waiting for VMDeathEvent");
251             getEventSet();
252             if (eventIterator.nextEvent() instanceof VMDeathEvent)
253                 return 0;
254 
255             log3("ERROR: last event is not the VMDeathEvent");
256             return 1;
257         } catch ( VMDisconnectedException e ) {
258             log3("ERROR: VMDisconnectedException : " + e);
259             return 2;
260         } catch ( Exception e ) {
261             log3("ERROR: Exception : " + e);
262             return 1;
263         }
264 
265     }

267     private void testRun()
268                  throws JDITestRuntimeException, Exception {
269 
270         eventRManager = vm.eventRequestManager();
271 
272         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
273         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
274         cpRequest.addClassFilter(debuggeeName);
275 
276         cpRequest.enable();
277         vm.resume();
278         getEventSet();
279         cpRequest.disable();
280 
281         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
282         debuggeeClass = event.referenceType();
283 
284         if (!debuggeeClass.name().equals(debuggeeName))
285            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
286 
287         log2("      received: ClassPrepareEvent for debuggeeClass");
288 
289         String bPointMethod = "methodForCommunication";
290         String lineForComm  = "lineForComm";
291 
292         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
293 
294         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
295                                              debuggeeClass,
296                                             bPointMethod, lineForComm, "zero");
297         bpRequest.enable();
298 
299     //------------------------------------------------------  testing section
300 
301         String fieldName = "testField";
302 
303         ExceptionRequest exRequest = null;
304 
305 
306         log1("     TESTING BEGINS");
307 
308         for (int i = 0; ; i++) {
309 
310             vm.resume();
311             breakpointForCommunication();
312 
313             int instruction = ((IntegerValue)
314                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
315 
316             if (instruction == 0) {
317                 vm.resume();
318                 break;
319             }
320 
321             log1(":::::: case: # " + i);
322 
323             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
324 
325             ReferenceType refType = (ReferenceType)
326                                     debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName)).type();
327 
328             log2("......creating: exRequest = eventRManager.createExceptionRequest(refType, false, true);");
329             exRequest = eventRManager.createExceptionRequest(refType, false, true);
330             log2("       check on if request is disabled");
331             if (exRequest.isEnabled()) {
332                 log3("ERROR: request is enabled");
333                 testExitCode = FAILED;
334             }
335 
336             log2("......creating: exRequest = eventRManager.createExceptionRequest(null, true, false);");
337             exRequest = eventRManager.createExceptionRequest(null, true, false);
338             log2("       check on if request is disabled");
339             if (exRequest.isEnabled()) {
340                 log3("ERROR: request is enabled");
341                 testExitCode = FAILED;
342             }
343 
344             log2("......creating: exRequest = eventRManager.createExceptionRequest(null, true, true);");
345             exRequest = eventRManager.createExceptionRequest(null, true, true);
346             log2("       check on if request is disabled");
347             if (exRequest.isEnabled()) {
348                 log3("ERROR: request is enabled");
349                 testExitCode = FAILED;
350             }
351 
352             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
353         }
354         log1("    TESTING ENDS");
355         return;
356     }

358    /*
359     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
360     *                                             String, String, String)
361     *
362     * It sets up a breakpoint at given line number within a given method in a given class
363     * for a given thread.
364     *
365     * Return value: BreakpointRequest object  in case of success
366     *
367     * JDITestRuntimeException   in case of an Exception thrown within the method
368     */
369 
370     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
371                                                   ReferenceType testedClass,
372                                                   String methodName,
373                                                   String bpLine,
374                                                   String property)
375             throws JDITestRuntimeException {
376 
377         log2("......setting up a breakpoint:");
378         log2("       thread: " + thread + "; class: " + testedClass +
379                         "; method: " + methodName + "; line: " + bpLine);
380 
381         List              alllineLocations = null;
382         Location          lineLocation     = null;
383         BreakpointRequest breakpRequest    = null;
384 
385         try {
386             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
387 
388             alllineLocations = method.allLineLocations();
389 
390             int n =
391                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
392             if (n > alllineLocations.size()) {
393                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
394             } else {
395                 lineLocation = (Location) alllineLocations.get(n);
396                 try {
397                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
398                     breakpRequest.putProperty("number", property);
399                     breakpRequest.addThreadFilter(thread);
400                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
401                 } catch ( Exception e1 ) {
402                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
403                     breakpRequest    = null;
404                 }
405             }
406         } catch ( Exception e2 ) {
407             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
408             breakpRequest    = null;
409         }
410 
411         if (breakpRequest == null) {
412             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
413             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
414         }
415 
416         log2("      a breakpoint has been set up");
417         return breakpRequest;
418     }

421     private void getEventSet()
422                  throws JDITestRuntimeException {
423         try {
424 //            log2("       eventSet = eventQueue.remove(waitTime);");
425             eventSet = eventQueue.remove(waitTime);
426             if (eventSet == null) {
427                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
428             }
429 //            log2("       eventIterator = eventSet.eventIterator;");
430             eventIterator = eventSet.eventIterator();
431         } catch ( Exception e ) {
432             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
433         }
434     }

437     private void breakpointForCommunication()
438                  throws JDITestRuntimeException {
439 
440         log2("breakpointForCommunication");
441         getEventSet();
442 
443         if (eventIterator.nextEvent() instanceof BreakpointEvent)
444             return;
445 
446         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
447     }
