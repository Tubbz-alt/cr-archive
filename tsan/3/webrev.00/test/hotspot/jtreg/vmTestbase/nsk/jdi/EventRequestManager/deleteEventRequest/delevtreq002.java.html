 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.deleteEventRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Once the eventRequest is deleted, no operations                   <BR>
 47  *    (for example, EventRequest.setEnabled(boolean)) are permitted -   <BR>
 48  *    attempts to do so will generally cause                            <BR>
 49  *    an InvalidRequestStateException.                                  <BR>
 50  * Note: only operations specified to throw InvalidRequestStateException <BR>
 51  * are not permitted.                                                   <BR>
 52  * Testcases unclude all 12 subclasses of EventRequest created by       <BR>
 53  * EventRequestManager.                                                 <BR>
 54  * <BR>
 55  * The test has three phases and works as follows.                      <BR>
 56  * <BR>
 57  * In first phase,                                                      <BR>
 58  * upon launching debuggee's VM which will be suspended,                <BR>
 59  * a debugger waits for the VMStartEvent within a predefined            <BR>
 60  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 61  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 62  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 63  * and waits for the event within the predefined time interval.         <BR>
 64  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 65  * Upon getting the ClassPrepareEvent,                                  <BR>
 66  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 67  * within debuggee's special methodForCommunication().                  <BR>
 68  * <BR>
 69  * In second phase to check the assertion,                              <BR>
 70  * the debugger and the debuggee perform the following.                 <BR>
 71  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 72  * - The debuggee prepares testcases and invokes                        <BR>
 73  *   the methodForCommunication to be suspended and                     <BR>
 74  *   to inform the debugger with the event.                             <BR>
 75  * - Upon getting the BreakpointEvent,                                  <BR>
 76  *   the debugger performs the checks required.                         <BR>
 77  * <BR>
 78  * In third phase, at the end of the test, the debuggee changes         <BR>
 79  * the value of the "instruction" which the debugger and debuggee       <BR>
 80  * use to inform each other of needed actions,  and both end.           <BR>
 81  * <BR>
 82  */
 83

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         EventRequest requestArray[] = new EventRequest[12];
309 
310         String fieldName1 = "testField1";
311         String fieldName2 = "testField2";
312         String fieldName3 = "testField3";
313 
314 
315         log1("     TESTING BEGINS");
316 
317         for (int i = 0; ; i++) {
318 
319             vm.resume();
320             breakpointForCommunication();
321 
322             int instruction = ((IntegerValue)
323                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
324 
325             if (instruction == 0) {
326                 vm.resume();
327                 break;
328             }
329 
330             log1(":::::: case: # " + i);
331 
332             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
333 
334 
335             Field field1 = debuggeeClass.fieldByName(fieldName1);
336 
337             ReferenceType refType = (ReferenceType)
338                                 debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2)).type();
339 
340             ThreadReference threadRef = (ThreadReference)
341                                 debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName3));
342 
343             String requests[] = { "AccessWatchpoint", "ModificationWatchpoint",
344                                   "Breakpoint",       "ClassPrepare",
345                                   "ClassUnload",      "Exception",
346                                   "MethodEntry",      "MethodExit",
347                                   "Step",             "ThreadDeath",
348                                   "ThreadStart",      "VMDeath"        };
349 
350             requestArray[0]  = eventRManager.createAccessWatchpointRequest(field1);
351             requestArray[1]  = eventRManager.createModificationWatchpointRequest(field1);
352             requestArray[2]  = eventRManager.createBreakpointRequest(breakpLocation);
353             requestArray[3]  = eventRManager.createClassPrepareRequest();
354             requestArray[4]  = eventRManager.createClassUnloadRequest();
355             requestArray[5]  = eventRManager.createExceptionRequest(refType, false, false);
356             requestArray[6]  = eventRManager.createMethodEntryRequest();
357             requestArray[7]  = eventRManager.createMethodExitRequest();
358             requestArray[8]  = eventRManager.createStepRequest(threadRef, StepRequest.STEP_MIN, StepRequest.STEP_OUT);
359             requestArray[9]  = eventRManager.createThreadDeathRequest();
360             requestArray[10] = eventRManager.createThreadStartRequest();
361             requestArray[11] = eventRManager.createVMDeathRequest();
362 
363             for (int i1 = 0; i1 < requestArray.length; i1++) {
364 
365                 log2("......eventRManager.deleteEventRequest(requestArray[i1]); :: " + requests[i1]);
366                 eventRManager.deleteEventRequest(requestArray[i1]);
367 
368                 try {
369                     requestArray[i1].addCountFilter(1);
370                     testExitCode = FAILED;
371                     log3("ERROR: NO InvalidRequestStateException for addCountFilter(1);");
372                 } catch ( InvalidRequestStateException e ) {
373                 }
374                 try {
375                     requestArray[i1].disable();
376                     testExitCode = FAILED;
377                     log3("ERROR: NO InvalidRequestStateException for disable();");
378                 } catch ( InvalidRequestStateException e ) {
379                 }
380                 try {
381                     requestArray[i1].enable();
382                     testExitCode = FAILED;
383                     log3("ERROR: NO InvalidRequestStateException for enable();");
384                 } catch ( InvalidRequestStateException e ) {
385                 }
386                 try {
387                     requestArray[i1].getProperty("number");
388                 } catch ( InvalidRequestStateException e ) {
389                     testExitCode = FAILED;
390                     log3("ERROR: InvalidRequestStateException for getProperty('number');");
391                 }
392                 try {
393                     requestArray[i1].isEnabled();
394                 } catch ( InvalidRequestStateException e ) {
395                     testExitCode = FAILED;
396                     log3("ERROR: InvalidRequestStateException for isEnabled();");
397                 }
398                 try {
399                     requestArray[i1].putProperty("number", "request" + i1);
400                 } catch ( InvalidRequestStateException e ) {
401                     testExitCode = FAILED;
402                     log3("ERROR: InvalidRequestStateException for putProperty('number', 'request' + i1);");
403                 }
404                 try {
405                     requestArray[i1].setEnabled(true);
406                     testExitCode = FAILED;
407                     log3("ERROR: NO InvalidRequestStateException for setEnabled(true);");
408                 } catch ( InvalidRequestStateException e ) {
409                 }
410                 try {
411                     requestArray[i1].setSuspendPolicy(EventRequest.SUSPEND_NONE);
412                     testExitCode = FAILED;
413                     log3("ERROR: NO InvalidRequestStateException for setSuspendPolicy(EventRequest.SUSPEND_NONE);");
414                 } catch ( InvalidRequestStateException e ) {
415                 }
416                 try {
417                     requestArray[i1].suspendPolicy();
418                 } catch ( InvalidRequestStateException e ) {
419                     testExitCode = FAILED;
420                     log3("ERROR: InvalidRequestStateException for suspendPolicy();");
421                 }
422             }
423 
424             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
425         }
426         log1("    TESTING ENDS");
427         return;
428     }

430    /*
431     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
432     *                                             String, String, String)
433     *
434     * It sets up a breakpoint at given line number within a given method in a given class
435     * for a given thread.
436     *
437     * Return value: BreakpointRequest object  in case of success
438     *
439     * JDITestRuntimeException   in case of an Exception thrown within the method
440     */
441 
442     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
443                                                   ReferenceType testedClass,
444                                                   String methodName,
445                                                   String bpLine,
446                                                   String property)
447             throws JDITestRuntimeException {
448 
449         log2("......setting up a breakpoint:");
450         log2("       thread: " + thread + "; class: " + testedClass +
451                         "; method: " + methodName + "; line: " + bpLine);
452 
453         List              alllineLocations = null;
454         Location          lineLocation     = null;
455         BreakpointRequest breakpRequest    = null;
456 
457         try {
458             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
459 
460             alllineLocations = method.allLineLocations();
461 
462             int n =
463                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
464             if (n > alllineLocations.size()) {
465                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
466             } else {
467                 lineLocation = (Location) alllineLocations.get(n);
468 breakpLocation = lineLocation;
469                 try {
470                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
471                     breakpRequest.putProperty("number", property);
472                     breakpRequest.addThreadFilter(thread);
473                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
474                 } catch ( Exception e1 ) {
475                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
476                     breakpRequest    = null;
477                 }
478             }
479         } catch ( Exception e2 ) {
480             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
481             breakpRequest    = null;
482         }
483 
484         if (breakpRequest == null) {
485             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
486             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
487         }
488 
489         log2("      a breakpoint has been set up");
490         return breakpRequest;
491     }

494     private void getEventSet()
495                  throws JDITestRuntimeException {
496         try {
497 //            log2("       eventSet = eventQueue.remove(waitTime);");
498             eventSet = eventQueue.remove(waitTime);
499             if (eventSet == null) {
500                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
501             }
502 //            log2("       eventIterator = eventSet.eventIterator;");
503             eventIterator = eventSet.eventIterator();
504         } catch ( Exception e ) {
505             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
506         }
507     }

510     private void breakpointForCommunication()
511                  throws JDITestRuntimeException {
512 
513         log2("breakpointForCommunication");
514         getEventSet();
515 
516         if (eventIterator.nextEvent() instanceof BreakpointEvent)
517             return;
518 
519         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
520     }
