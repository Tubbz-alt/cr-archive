 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for StepEvent sets.                   <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a StepRequest, resumes                        <BR>
 74  *   the debuggee, and waits for the StepEvent.                         <BR>
 75  * - The debuggee invokes the special method                            <BR>
 76  *   to be resulting in the event.                                      <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         String bPointMethod = "methodForCommunication";
301         String lineForComm  = "lineForComm";
302         BreakpointRequest bpRequest;
303         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
304         bpRequest = settingBreakpoint(mainThread,
305                                       debuggeeClass,
306                                       bPointMethod, lineForComm, "zero");
307         bpRequest.enable();
308 
309         vm.resume();
310 
311     //------------------------------------------------------  testing section
312 
313         log1("     TESTING BEGINS");
314 
315         EventRequest eventRequest1 = null;
316         EventRequest eventRequest2 = null;
317         EventRequest eventRequest3 = null;
318 
319         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
320         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
321         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
322 
323         ReferenceType testClassReference = null;
324 
325         for (int i = 0; ; i++) {
326 
327             breakpointForCommunication();
328 
329             int instruction = ((IntegerValue)
330                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
331 
332             if (instruction == 0) {
333                 vm.resume();
334                 break;
335             }
336 
337             log1(":::::: case: # " + i);
338 
339             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
340 
341             switch (i) {
342 
343               case 0:
344                       eventRequest1 = settingStepRequest (mainThread,
345                                              SUSPEND_NONE, "StepRequest1");
346                       eventRequest1.enable();
347                       break;
348 
349               case 1:
350                       eventRequest2 = settingStepRequest (mainThread,
351                                              SUSPEND_THREAD, "StepRequest2");
352                       eventRequest2.enable();
353                       break;
354 
355               case 2:
356                       eventRequest3 = settingStepRequest (mainThread,
357 
358                                              SUSPEND_ALL, "StepRequest3");
359                       eventRequest3.enable();
360                       break;
361 
362 
363               default:
364                       throw new JDITestRuntimeException("** default case 2 **");
365             }
366 
367             log2("......waiting for new StepEvent : " + i);
368             mainThread.resume();
369             getEventSet();
370 
371             Event newEvent = eventIterator.nextEvent();
372             if ( !(newEvent instanceof StepEvent)) {
373                  log3("ERROR: new event is not StepEvent");
374                  testExitCode = FAILED;
375             } else {
376 
377                 String property = (String) newEvent.request().getProperty("number");
378                 log2("       got new StepEvent with propety 'number' == " + property);
379 
380                 log2("......checking up on EventSet.resume()");
381                 log2("......--> vm.suspend();");
382                 vm.suspend();
383 
384                 log2("        getting : Map<String, Integer> suspendsCounts1");
385 
386                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
387                 for (ThreadReference threadReference : vm.allThreads()) {
388                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
389                 }
390 
391                 log2("        eventSet.resume;");
392                 eventSet.resume();
393 
394                 log2("        getting : Map<String, Integer> suspendsCounts2");
395 
396                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
397                 for (ThreadReference threadReference : vm.allThreads()) {
398                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
399                 }
400 
401                 log2(suspendsCounts1.toString());
402                 log2(suspendsCounts2.toString());
403 
404                 log2("        getting : int policy = eventSet.suspendPolicy();");
405                 int policy = eventSet.suspendPolicy();
406 
407                 switch (policy) {
408 
409                     case SUSPEND_NONE:
410 
411                         log2("        case SUSPEND_NONE");
412                         for (String threadName : suspendsCounts1.keySet()) {
413                             log2("        checking " + threadName);
414                             if (!suspendsCounts2.containsKey(threadName)) {
415                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
416                                 testExitCode = FAILED;
417                                 break;
418                             }
419                             int count1 = suspendsCounts1.get(threadName);
420                             int count2 = suspendsCounts2.get(threadName);
421                             if (count1 != count2) {
422                                 log3("ERROR: suspendCounts don't match for : " + threadName);
423                                 log3("before resuming : " + count1);
424                                 log3("after  resuming : " + count2);
425                                 testExitCode = FAILED;
426                                 break;
427                             }
428                         }
429                         eventRequest1.disable();
430                         break;
431 
432                     case SUSPEND_THREAD:
433 
434                         log2("        case SUSPEND_THREAD");
435                         for (String threadName : suspendsCounts1.keySet()) {
436                             log2("checking " + threadName);
437                             if (!suspendsCounts2.containsKey(threadName)) {
438                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
439                                 testExitCode = FAILED;
440                                 break;
441                             }
442                             int count1 = suspendsCounts1.get(threadName);
443                             int count2 = suspendsCounts2.get(threadName);
444                             String eventThreadName = ((StepEvent) newEvent).thread().name();
445                             int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
446                             if (count1 != expectedValue) {
447                                 log3("ERROR: suspendCounts don't match for : " + threadName);
448                                 log3("before resuming : " + count1);
449                                 log3("after  resuming : " + count2);
450                                 testExitCode = FAILED;
451                                 break;
452                             }
453                         }
454                         eventRequest2.disable();
455                         break;
456 
457                     case SUSPEND_ALL:
458 
459                         log2("        case SUSPEND_ALL");
460                         for (String threadName : suspendsCounts1.keySet()) {
461                             log2("checking " + threadName);
462                             if (!event.request().equals(eventRequest3))
463                                 break;
464                             if (!suspendsCounts2.containsKey(threadName)) {
465                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
466                                 testExitCode = FAILED;
467                                 break;
468                             }
469                             int count1 = suspendsCounts1.get(threadName);
470                             int count2 = suspendsCounts2.get(threadName);
471                             if (count1 != count2 + 1) {
472                                 log3("ERROR: suspendCounts don't match for : " + threadName);
473                                 log3("before resuming : " + count1);
474                                 log3("after  resuming : " + count2);
475                                 testExitCode = FAILED;
476                                 break;
477                             }
478                         }
479                         eventRequest3.disable();
480                         break;
481 
482                     default:
483                         throw new JDITestRuntimeException("** default case 1 **");
484                 }
485                 informDebuggeeTestCase(i);
486             }
487 
488             log2("......--> vm.resume()");
489             vm.resume();
490             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
491         }
492         log1("    TESTING ENDS");
493         return;
494     }

496    /*
497     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
498     *                                             String, String, String)
499     *
500     * It sets up a breakpoint at given line number within a given method in a given class
501     * for a given thread.
502     *
503     * Return value: BreakpointRequest object  in case of success
504     *
505     * JDITestRuntimeException   in case of an Exception thrown within the method
506     */
507 
508     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
509                                                   ReferenceType testedClass,
510                                                   String methodName,
511                                                   String bpLine,
512                                                   String property)
513             throws JDITestRuntimeException {
514 
515         log2("......setting up a breakpoint:");
516         log2("       thread: " + thread + "; class: " + testedClass +
517                         "; method: " + methodName + "; line: " + bpLine);
518 
519         List              alllineLocations = null;
520         Location          lineLocation     = null;
521         BreakpointRequest breakpRequest    = null;
522 
523         try {
524             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
525 
526             alllineLocations = method.allLineLocations();
527 
528             int n =
529                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
530             if (n > alllineLocations.size()) {
531                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
532             } else {
533                 lineLocation = (Location) alllineLocations.get(n);
534                 try {
535                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
536                     breakpRequest.putProperty("number", property);
537                     breakpRequest.addThreadFilter(thread);
538                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
539                 } catch ( Exception e1 ) {
540                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
541                     breakpRequest    = null;
542                 }
543             }
544         } catch ( Exception e2 ) {
545             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
546             breakpRequest    = null;
547         }
548 
549         if (breakpRequest == null) {
550             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
551             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
552         }
553 
554         log2("      a breakpoint has been set up");
555         return breakpRequest;
556     }

559     private void getEventSet()
560                  throws JDITestRuntimeException {
561         try {
562 //            log2("       eventSet = eventQueue.remove(waitTime);");
563             eventSet = eventQueue.remove(waitTime);
564             if (eventSet == null) {
565                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
566             }
567 //            log2("       eventIterator = eventSet.eventIterator;");
568             eventIterator = eventSet.eventIterator();
569         } catch ( Exception e ) {
570             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
571         }
572     }

589     private StepRequest settingStepRequest ( ThreadReference thread,
590                                              int             suspendPolicy,
591                                              String          property        )
592             throws JDITestRuntimeException {
593         try {
594             log2("......setting up StepRequest:");
595             log2("       thread: " + thread + "; property: " + property);
596 
597             StepRequest
598             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OVER);
599             str.putProperty("number", property);
600             str.addCountFilter(1);
601             str.setSuspendPolicy(suspendPolicy);
602 
603             log2("      StepRequest has been set up");
604             return str;
605         } catch ( Exception e ) {
606             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
607             log3("       StepRequest HAS NOT BEEN SET UP");
608             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
609         }
610     }
611     /**
612      * Inform debuggee which thread test the debugger has completed.
613      * Used for synchronization, so the debuggee does not move too quickly.
614      * @param testCase index of just completed test
615      */
616     void informDebuggeeTestCase(int testCase) {
617         try {
618             ((ClassType)debuggeeClass)
619                 .setValue(debuggeeClass.fieldByName("testCase"),
620                           vm.mirrorOf(testCase));
621         } catch (InvalidTypeException ite) {
622             throw new Failure("** FAILURE setting testCase  **");
623         } catch (ClassNotLoadedException cnle) {
624             throw new Failure("** FAILURE notifying debuggee  **");
625         } catch (VMDisconnectedException e) {
626             throw new Failure("** FAILURE debuggee connection **");
627         }
628     }
