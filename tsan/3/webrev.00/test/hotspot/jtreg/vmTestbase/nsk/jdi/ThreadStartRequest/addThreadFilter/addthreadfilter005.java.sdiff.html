409     private ThreadReference threadByName(String name)
410                  throws JDITestRuntimeException {
411 
412         List         all = vm.allThreads();
413         ListIterator li  = all.listIterator();
414 
415         for (; li.hasNext(); ) {
416             ThreadReference thread = (ThreadReference) li.next();
417             if (thread.name().equals(name))
418                 return thread;
419         }
420         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
421     }
