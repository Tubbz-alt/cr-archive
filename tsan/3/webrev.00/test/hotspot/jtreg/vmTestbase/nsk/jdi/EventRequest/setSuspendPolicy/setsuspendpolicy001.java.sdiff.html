485     private ThreadReference threadByName(String name)
486                  throws JDITestRuntimeException {
487 
488         List         all = vm.allThreads();
489         ListIterator li  = all.listIterator();
490 
491         for (; li.hasNext(); ) {
492             ThreadReference thread = (ThreadReference) li.next();
493             if (thread.name().equals(name))
494                 return thread;
495         }
496         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
497     }
