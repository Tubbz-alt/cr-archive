 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a StepEvent set,                    <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one thread with all possible combinations:                   <BR>
 50  *    NONE, THREAD, ALL                                         <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the above,                                  <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger sets up a StepRequest, resumes                        <BR>
 69  *   the debuggee, and waits for the StepEvent.                         <BR>
 70  * - The debuggee invokes the special method to be resulting in the event.<BR>
 71  * - Upon getting new event, the debugger                               <BR>
 72  *   performs the check corresponding to the event.                     <BR>
 73  * <BR>
 74  * Note. To inform each other of needed actions, the debugger and       <BR>
 75  *       and the debuggee use debuggee's variable "instruction".        <BR>
 76  * In third phase, at the end,                                          <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */
 81

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

250    /*
251     * Return value: 0 - normal end of the test
252     *               1 - ubnormal end of the test
253     *               2 - VMDisconnectedException while test phase
254     */
255 
256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
264                 log3("ERROR: last event is not the VMDeathEvent");
265                 return 1;
266             }
267 
268             log2("waiting for VMDisconnectEvent");
269             getEventSet();
270             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
271                 log3("ERROR: last event is not the VMDisconnectEvent");
272                 return 1;
273             }
274 
275             return 0;
276 
277         } catch ( VMDisconnectedException e ) {
278             log3("ERROR: VMDisconnectedException : " + e);
279             return 2;
280         } catch ( Exception e ) {
281             log3("ERROR: Exception : " + e);
282             return 1;
283         }
284

287     private void testRun()
288                  throws JDITestRuntimeException, Exception {
289 
290         eventRManager = vm.eventRequestManager();
291 
292         log2("......getting ClassPrepareEvent for debuggee's class");
293         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
294         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
295         cpRequest.addClassFilter(debuggeeName);
296         cpRequest.enable();
297         vm.resume();
298         getEventSet();
299         cpRequest.disable();
300 
301         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
302         debuggeeClass = event.referenceType();
303 
304         if (!debuggeeClass.name().equals(debuggeeName))
305            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
306         log2("      received: ClassPrepareEvent for debuggeeClass");
307 
308         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
309 
310         String            bPointMethod = "methodForCommunication";
311         String            lineForComm  = "lineForComm";
312         BreakpointRequest bpRequest;
313         ThreadReference   mainThread = threadByName("main");
314         bpRequest = settingBreakpoint(mainThread,
315                                       debuggeeClass,
316                                       bPointMethod, lineForComm, "zero");
317         bpRequest.enable();
318 
319         vm.resume();
320 
321     //------------------------------------------------------  testing section
322 
323         log1("     TESTING BEGINS");
324 
325         EventRequest eventRequest1 = null;
326         EventRequest eventRequest2 = null;
327         EventRequest eventRequest3 = null;
328 
329         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
330         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
331         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
332         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
333 
334         int policyExpected[] = {    SUSPEND_NONE,
335                                     SUSPEND_THREAD,
336                                     SUSPEND_ALL,
337                                     SUSPEND_THREAD,
338                                     SUSPEND_ALL,
339                                     SUSPEND_ALL,
340                                     SUSPEND_ALL      };
341         int policy = 0;
342 
343 
344         for (int i = 0; ; i++) {
345 
346             breakpointForCommunication();
347 
348             int instruction = ((IntegerValue)
349                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
350             if (instruction == 0) {
351                 vm.resume();
352                 break;
353             }
354 
355             log1(":::::: case: # " + i);
356 
357             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
358 
359             switch (i) {
360 
361               case 0:
362                       eventRequest1 = settingStepRequest (mainThread,
363                                              SUSPEND_NONE, "StepRequest1");
364                       eventRequest1.enable();
365                       break;
366 
367               case 1:
368                       eventRequest2 = settingStepRequest (mainThread,
369                                              SUSPEND_THREAD, "StepRequest2");
370                       eventRequest2.enable();
371                       break;
372 
373               case 2:
374                       eventRequest3 = settingStepRequest (mainThread,
375 
376                                              SUSPEND_ALL, "StepRequest3");
377                       eventRequest3.enable();
378                       break;
379 
380 
381               default:
382                       throw new JDITestRuntimeException("** default case 2 **");
383             }
384 
385             mainThread.resume();
386             getEventSet();
387 
388             if ( !(eventIterator.nextEvent() instanceof StepEvent)) {
389                  log3("ERROR: new event is not StepEvent");
390                  testExitCode = FAILED;
391             } else {
392                 log2("......got : instanceof StepEvent");
393                 policy = eventSet.suspendPolicy();
394                 if (policy != policyExpected[i]) {
395                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
396                     log3("       eventSet.suspendPolicy() == " + policy);
397                     log3("       policyExpected           == " + policyExpected[i]);
398                     testExitCode = FAILED;
399                 }
400             }
401 
402             switch (i) {
403               case 0: eventRequest1.disable();                          break;
404               case 1: eventRequest2.disable();                          break;
405               case 2: eventRequest3.disable();                          break;
406            }
407 
408             switch (policy) {
409               case SUSPEND_NONE   :                      break;
410               case SUSPEND_THREAD : mainThread.resume(); break;
411               case SUSPEND_ALL    : vm.resume();         break;
412               default: throw new JDITestRuntimeException("** default case 3 **");
413             }
414 
415             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
416         }
417         log1("    TESTING ENDS");
418         return;
419     }

421     private ThreadReference threadByName(String name)
422                  throws JDITestRuntimeException {
423 
424         List         all = vm.allThreads();
425         ListIterator li  = all.listIterator();
426 
427         for (; li.hasNext(); ) {
428             ThreadReference thread = (ThreadReference) li.next();
429             if (thread.name().equals(name))
430                 return thread;
431         }
432         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
433     }

435    /*
436     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
437     *                                             String, String, String)
438     *
439     * It sets up a breakpoint at given line number within a given method in a given class
440     * for a given thread.
441     *
442     * Return value: BreakpointRequest object  in case of success
443     *
444     * JDITestRuntimeException   in case of an Exception thrown within the method
445     */
446 
447     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
448                                                   ReferenceType testedClass,
449                                                   String methodName,
450                                                   String bpLine,
451                                                   String property)
452             throws JDITestRuntimeException {
453 
454         log2("......setting up a breakpoint:");
455         log2("       thread: " + thread + "; class: " + testedClass +
456                         "; method: " + methodName + "; line: " + bpLine);
457 
458         List              alllineLocations = null;
459         Location          lineLocation     = null;
460         BreakpointRequest breakpRequest    = null;
461 
462         try {
463             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
464 
465             alllineLocations = method.allLineLocations();
466 
467             int n =
468                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
469             if (n > alllineLocations.size()) {
470                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
471             } else {
472                 lineLocation = (Location) alllineLocations.get(n);
473                 try {
474                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
475                     breakpRequest.putProperty("number", property);
476                     breakpRequest.addThreadFilter(thread);
477                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
478                 } catch ( Exception e1 ) {
479                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
480                     breakpRequest    = null;
481                 }
482             }
483         } catch ( Exception e2 ) {
484             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
485             breakpRequest    = null;
486         }
487 
488         if (breakpRequest == null) {
489             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
490             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
491         }
492 
493         log2("      a breakpoint has been set up");
494         return breakpRequest;
495     }

498     private void getEventSet()
499                  throws JDITestRuntimeException {
500         try {
501 //            log2("       eventSet = eventQueue.remove(waitTime);");
502             eventSet = eventQueue.remove(waitTime);
503             if (eventSet == null) {
504                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
505             }
506 //            log2("       eventIterator = eventSet.eventIterator;");
507             eventIterator = eventSet.eventIterator();
508         } catch ( Exception e ) {
509             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
510         }
511     }

514     private void breakpointForCommunication()
515                  throws JDITestRuntimeException {
516 
517         log2("breakpointForCommunication");
518         getEventSet();
519 
520         if (eventIterator.nextEvent() instanceof BreakpointEvent)
521             return;
522 
523         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
524     }

528     private StepRequest settingStepRequest ( ThreadReference thread,
529                                              int             suspendPolicy,
530                                              String          property        )
531             throws JDITestRuntimeException {
532         try {
533             log2("......setting up StepRequest:");
534             log2("       thread: " + thread + "; property: " + property);
535 
536             StepRequest
537             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OVER);
538             str.putProperty("number", property);
539             str.addCountFilter(1);
540             str.setSuspendPolicy(suspendPolicy);
541 
542             log2("      StepRequest has been set up");
543             return str;
544         } catch ( Exception e ) {
545             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
546             log3("       StepRequest HAS NOT BEEN SET UP");
547             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
548         }
