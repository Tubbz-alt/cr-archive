 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                              <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new threads, thread1, and invokes             <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a BreakpointRequest and enables it,                      <BR>
 72  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addThreadFilter()    <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

167     private int runThis (String argv[], PrintStream out) {
168 
169         argsHandler     = new ArgumentHandler(argv);
170         logHandler      = new Log(out, argsHandler);
171         Binder binder   = new Binder(argsHandler, logHandler);
172 
173         waitTime        = argsHandler.getWaitTime() * 60000;
174 
175         try {
176             log2("launching a debuggee :");
177             log2("       " + debuggeeName);
178             if (argsHandler.verbose()) {
179                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
180             } else {
181                 debuggee = binder.bindToDebugee(debuggeeName);
182             }
183             if (debuggee == null) {
184                 log3("ERROR: no debuggee launched");
185                 return FAILED;
186             }
187             log2("debuggee launched");
188         } catch ( Exception e ) {
189             log3("ERROR: Exception : " + e);
190             log2("       test cancelled");
191             return FAILED;
192         }
193 
194         debuggee.redirectOutput(logHandler);
195 
196         vm = debuggee.VM();
197 
198         eventQueue = vm.eventQueue();
199         if (eventQueue == null) {
200             log3("ERROR: eventQueue == null : TEST ABORTED");
201             vm.exit(PASS_BASE);
202             return FAILED;
203         }
204 
205         log2("invocation of the method runTest()");
206         switch (runTest()) {
207 
208             case 0 :  log2("test phase has finished normally");
209                       log2("   waiting for the debuggee to finish ...");
210                       debuggee.waitFor();
211 
212                       log2("......getting the debuggee's exit status");
213                       int status = debuggee.getStatus();
214                       if (status != PASS_BASE) {
215                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
216                               status + " != PASS_BASE");
217                           testExitCode = FAILED;
218                       } else {
219                           log2("......debuggee returned expected exit status: " +
220                               status + " == PASS_BASE");
221                       }
222                       break;
223 
224             default : log3("ERROR: runTest() returned unexpected value");
225 
226             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
227                       log2("......forcing: vm.exit();");
228                       testExitCode = FAILED;
229                       try {
230                           vm.exit(PASS_BASE);
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235 
236             case 2 :  log3("test cancelled due to VMDisconnectedException");
237                       log2("......trying: vm.process().destroy();");
238                       testExitCode = FAILED;
239                       try {
240                           Process vmProcess = vm.process();
241                           if (vmProcess != null) {
242                               vmProcess.destroy();
243                           }
244                       } catch ( Exception e ) {
245                           log3("ERROR: Exception : " + e);
246                       }
247                       break;
248             }
249 
250         return testExitCode;
251     }

255     * Return value: 0 - normal end of the test
256     *               1 - ubnormal end of the test
257     *               2 - VMDisconnectedException while test phase
258     */

260     private int runTest() {
261 
262         try {
263             testRun();
264 
265             log2("waiting for VMDeathEvent");
266             getEventSet();
267             if (eventIterator.nextEvent() instanceof VMDeathEvent)
268                 return 0;
269 
270             log3("ERROR: last event is not the VMDeathEvent");
271             return 1;
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = threadByName("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         EventRequest eventRequest1 = null;
319 
320         ThreadReference thread1     = null;
321         String          thread1Name = "thread1";
322 
323         String property1 = "BreakpointRequest1";
324 
325         String methodName = "method";
326         String bpLineName = "breakpointLine";
327 
328         ReferenceType testClassReference = null;
329 
330         for (int i = 0; ; i++) {
331 
332             vm.resume();
333             breakpointForCommunication();
334 
335             int instruction = ((IntegerValue)
336                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
337 
338             if (instruction == 0) {
339                 vm.resume();
340                 break;
341             }
342 
343             log1(":::::: case: # " + i);
344 
345             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
346 
347             switch (i) {
348 
349               case 0:
350                      testClassReference =
351                             (ReferenceType) vm.classesByName(testedClassName).get(0);
352 
353                      thread1 = (ThreadReference) debuggeeClass.getValue(
354                                         debuggeeClass.fieldByName(thread1Name));
355 
356                      eventRequest1 = setting2BreakpointRequest (null,
357                                              testClassReference, methodName, bpLineName,
358                                              EventRequest.SUSPEND_NONE, property1);
359 
360                      log2("......eventRequest1.enable();");
361                      eventRequest1.enable();
362 
363                      try {
364                          log2("......eventRequest1.addThreadFilter(thread1);");
365                          log2("        InvalidRequestStateException expected");
366                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
367                          log3("ERROR: no InvalidRequestStateException");
368                          testExitCode = FAILED;
369                      } catch ( InvalidRequestStateException e ) {
370                          log2("        InvalidRequestStateException");
371                      }
372 
373                      log2("......eventRequest1.disable();");
374                      eventRequest1.disable();
375                      try {
376                          log2("......eventRequest1.addThreadFilter(thread1);");
377                          log2("        no InvalidRequestStateException expected");
378                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
379                          log2("        no InvalidRequestStateException");
380                      } catch ( InvalidRequestStateException e ) {
381                          log3("ERROR: InvalidRequestStateException");
382                          testExitCode = FAILED;
383                      }
384 
385                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
386                      eventRManager.deleteEventRequest(eventRequest1);
387                      try {
388                          log2("......eventRequest1.addThreadFilter(thread1);");
389                          log2("        InvalidRequestStateException expected");
390                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
391                          log3("ERROR: no InvalidRequestStateException");
392                          testExitCode = FAILED;
393                      } catch ( InvalidRequestStateException e ) {
394                          log2("        InvalidRequestStateException");
395                      }
396 
397                      break;
398 
399               default:
400                       throw new JDITestRuntimeException("** default case 2 **");
401             }
402             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
403         }
404         log1("    TESTING ENDS");
405         return;
406     }

408     private ThreadReference threadByName(String name)
409                  throws JDITestRuntimeException {
410 
411         List         all = vm.allThreads();
412         ListIterator li  = all.listIterator();
413 
414         for (; li.hasNext(); ) {
415             ThreadReference thread = (ThreadReference) li.next();
416             if (thread.name().equals(name))
417                 return thread;
418         }
419         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
420     }

422    /*
423     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
424     *                                             String, String, String)
425     *
426     * It sets up a breakpoint at given line number within a given method in a given class
427     * for a given thread.
428     *
429     * Return value: BreakpointRequest object  in case of success
430     *
431     * JDITestRuntimeException   in case of an Exception thrown within the method
432     */
433 
434     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
435                                                   ReferenceType testedClass,
436                                                   String methodName,
437                                                   String bpLine,
438                                                   String property)
439             throws JDITestRuntimeException {
440 
441         log2("......setting up a breakpoint:");
442         log2("       thread: " + thread + "; class: " + testedClass +
443                         "; method: " + methodName + "; line: " + bpLine);
444 
445         List              alllineLocations = null;
446         Location          lineLocation     = null;
447         BreakpointRequest breakpRequest    = null;
448 
449         try {
450             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
451 
452             alllineLocations = method.allLineLocations();
453 
454             int n =
455                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
456             if (n > alllineLocations.size()) {
457                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
458             } else {
459                 lineLocation = (Location) alllineLocations.get(n);
460                 try {
461                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
462                     breakpRequest.putProperty("number", property);
463                     breakpRequest.addThreadFilter(thread);
464                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
465                 } catch ( Exception e1 ) {
466                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
467                     breakpRequest    = null;
468                 }
469             }
470         } catch ( Exception e2 ) {
471             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
472             breakpRequest    = null;
473         }
474 
475         if (breakpRequest == null) {
476             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
477             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
478         }
479 
480         log2("      a breakpoint has been set up");
481         return breakpRequest;
482     }

485     private void getEventSet()
486                  throws JDITestRuntimeException {
487         try {
488 //            log2("       eventSet = eventQueue.remove(waitTime);");
489             eventSet = eventQueue.remove(waitTime);
490             if (eventSet == null) {
491                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
492             }
493 //            log2("       eventIterator = eventSet.eventIterator;");
494             eventIterator = eventSet.eventIterator();
495         } catch ( Exception e ) {
496             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
497         }
498     }

501     private void breakpointForCommunication()
502                  throws JDITestRuntimeException {
503 
504         log2("breakpointForCommunication");
505         getEventSet();
506 
507         if (eventIterator.nextEvent() instanceof BreakpointEvent)
508             return;
509 
510         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
511     }

515     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
516                                                           ReferenceType   testedClass,
517                                                           String          methodName,
518                                                           String          bpLine,
519                                                           int             suspendPolicy,
520                                                           String          property        )
521             throws JDITestRuntimeException {
522         try {
523             log2("......setting up a breakpoint:");
524             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
525 
526             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
527             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
528 
529             BreakpointRequest
530             bpr = eventRManager.createBreakpointRequest(loc);
531             bpr.putProperty("number", property);
532             if (thread != null)
533                 bpr.addThreadFilter(thread);
534             bpr.setSuspendPolicy(suspendPolicy);
535 
536             log2("      a breakpoint has been set up");
537             return bpr;
538         } catch ( Exception e ) {
539             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
540             log3("       BreakpointRequest HAS NOT BEEN SET UP");
541             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
542         }
