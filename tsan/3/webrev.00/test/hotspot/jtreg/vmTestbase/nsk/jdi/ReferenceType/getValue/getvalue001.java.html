 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.getValue()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * Case for testing includes fields mirroring primitive type values.    <BR>
 45  * The test checks that Values returned by the method,                  <BR>
 46  * are casted to PrimitiveValue(s).                                     <BR>
 47  * <BR>
 48  * The test works as follows.                                           <BR>
 49  * Upon launching debuggee's VM which will be suspended,                <BR>
 50  * a debugger waits for the VMStartEvent within a predefined            <BR>
 51  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 52  * Upon getting the VMStartEvent, it makes the request                  <BR>
 53  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 54  * resumes the VM, and waits for the event within the predefined        <BR>
 55  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 56  * Upon getting the ClassPrepareEvent,                                  <BR>
 57  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 58  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 59  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 60  * to be suspended and to inform the debugger with the event.           <BR>
 61  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 62  * At the end, the debuggee changes the value of the "instruction"      <BR>
 63  * to inform the debugger of checks finished, and both end.             <BR>
 64  */

 89 
 90         int exitCode = new getvalue001().runThis(argv, out);

 92         if (exitCode != PASSED) {
 93             System.out.println("TEST FAILED");
 94         }
 95         return testExitCode;

140     private int runThis (String argv[], PrintStream out) {
141 
142         argsHandler     = new ArgumentHandler(argv);
143         logHandler      = new Log(out, argsHandler);
144         Binder binder   = new Binder(argsHandler, logHandler);
145 
146         waitTime        = argsHandler.getWaitTime() * 60000;
147 
148         try {
149             log2("launching a debuggee :");
150             log2("       " + debuggeeName);
151             if (argsHandler.verbose()) {
152                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
153             } else {
154                 debuggee = binder.bindToDebugee(debuggeeName);
155             }
156             if (debuggee == null) {
157                 log3("ERROR: no debuggee launched");
158                 return FAILED;
159             }
160             log2("debuggee launched");
161         } catch ( Exception e ) {
162             log3("ERROR: Exception : " + e);
163             log2("       test cancelled");
164             return FAILED;
165         }
166 
167         debuggee.redirectOutput(logHandler);
168 
169         vm = debuggee.VM();
170 
171         eventQueue = vm.eventQueue();
172         if (eventQueue == null) {
173             log3("ERROR: eventQueue == null : TEST ABORTED");
174             vm.exit(PASS_BASE);
175             return FAILED;
176         }
177 
178         log2("invocation of the method runTest()");
179         switch (runTest()) {
180 
181             case 0 :  log2("test phase has finished normally");
182                       log2("   waiting for the debuggee to finish ...");
183                       debuggee.waitFor();
184 
185                       log2("......getting the debuggee's exit status");
186                       int status = debuggee.getStatus();
187                       if (status != PASS_BASE) {
188                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
189                               status + " != PASS_BASE");
190                           testExitCode = FAILED;
191                       } else {
192                           log2("......debuggee returned expected exit status: " +
193                               status + " == PASS_BASE");
194                       }
195                       break;
196 
197             default : log3("ERROR: runTest() returned unexpected value");
198 
199             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
200                       log2("......forcing: vm.exit();");
201                       testExitCode = FAILED;
202                       try {
203                           vm.exit(PASS_BASE);
204                       } catch ( Exception e ) {
205                           log3("ERROR: Exception : e");
206                       }
207                       break;
208 
209             case 2 :  log3("test cancelled due to VMDisconnectedException");
210                       log2("......trying: vm.process().destroy();");
211                       testExitCode = FAILED;
212                       try {
213                           Process vmProcess = vm.process();
214                           if (vmProcess != null) {
215                               vmProcess.destroy();
216                           }
217                       } catch ( Exception e ) {
218                           log3("ERROR: Exception : e");
219                       }
220                       break;
221             }
222 
223         return testExitCode;
224     }

227    /*
228     * Return value: 0 - normal end of the test
229     *               1 - ubnormal end of the test
230     *               2 - VMDisconnectedException while test phase
231     */
232 
233     private int runTest() {
234 
235         try {
236             testRun();
237 
238             log2("waiting for VMDeathEvent");
239             getEventSet();
240             if (eventIterator.nextEvent() instanceof VMDeathEvent)
241                 return 0;
242 
243             log3("ERROR: last event is not the VMDeathEvent");
244             return 1;
245         } catch ( VMDisconnectedException e ) {
246             log3("ERROR: VMDisconnectedException : " + e);
247             return 2;
248         } catch ( Exception e ) {
249             log3("ERROR: Exception : " + e);
250             return 1;
251         }
252 
253     }

255     private void testRun()
256                  throws JDITestRuntimeException, Exception {
257 
258         eventRManager = vm.eventRequestManager();
259 
260         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
261         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
262         cpRequest.addClassFilter(debuggeeName);
263 
264         cpRequest.enable();
265         vm.resume();
266         getEventSet();
267         cpRequest.disable();
268 
269         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
270         debuggeeClass = event.referenceType();
271 
272         if (!debuggeeClass.name().equals(debuggeeName))
273            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
274 
275         log2("      received: ClassPrepareEvent for debuggeeClass");
276 
277         String bPointMethod = "methodForCommunication";
278         String lineForComm  = "lineForComm";
279         BreakpointRequest bpRequest;
280 
281         try {
282             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
283                                           debuggeeClass,
284                                           bPointMethod, lineForComm, "zero");
285         } catch ( Exception e ) {
286             throw e;
287         }
288         bpRequest.enable();
289 
290     //------------------------------------------------------  testing section
291 
292         log1("     TESTING BEGINS");
293 
294         for (int i = 0; ; i++) {
295 
296             vm.resume();
297             breakpointForCommunication();
298 
299             int instruction = ((IntegerValue)
300                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
301 
302             if (instruction == 0) {
303                 vm.resume();
304                 break;
305             }
306 
307             log1("  new check: # " + i);
308 
309             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
310 
311             log2("      name of tested class : " + testedClassName);
312             log2("      getting: List classes = vm.classByName(testedClassName);");
313             List classes = vm.classesByName(testedClassName);
314             if (classes.size() != 1) {
315                 log3("ERROR: classes.size() != 1 : " + classes.size());
316                 testExitCode = FAILED;
317                 break;
318             }
319             log2("      getting: ReferenceType testedClass = (ReferenceType) classes.get(0);");
320             ReferenceType testedClass = (ReferenceType) classes.get(0);
321 
322             log2("------case for testing: primitive values");
323 
324             String fieldNames0[] = {
325                        "bl0",
326                        "bt0",
327                        "ch0",
328                        "db0",
329                        "fl0",
330                        "in0",
331                        "ln0",
332                        "sh0"      };
333 
334             for (int i1 = 0; i1 < 8; i1++) {
335 
336                 log2("......field tested : " + fieldNames0[i1]);
337                 log2("      getting: Field field = testedClass.fieldByName(fieldNames0[i1]);");
338                 Field field = testedClass.fieldByName(fieldNames0[i1]);
339                 if (field == null) {
340                     log3("ERROR: field == null");
341                     testExitCode = FAILED;
342                     continue;
343                 }
344                 log2("      getting: Value value = testedClass.getValue(field);");
345                 Value value = testedClass.getValue(field);
346                 if (value == null) {
347                     log3("ERROR: value == null");
348                     testExitCode = FAILED;
349                     continue;
350                 }
351                 try {
352                     log2("      getting Class cast: PrimitiveValue primitiveValue = (PrimitiveValue) value;");
353                     PrimitiveValue primitiveValue = (PrimitiveValue) value;
354                 } catch ( ClassCastException e ) {
355                     log3("ERROR: ClassCastException");
356                     testExitCode = FAILED;
357                 }
358             }
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
360         }
361         log1("    TESTING ENDS");
362         return;
363     }

365    /*
366     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
367     *                                             String, String, String)
368     *
369     * It sets up a breakpoint at given line number within a given method in a given class
370     * for a given thread.
371     *
372     * Return value: BreakpointRequest object  in case of success
373     *
374     * JDITestRuntimeException   in case of an Exception thrown within the method
375     */
376 
377     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
378                                                   ReferenceType testedClass,
379                                                   String methodName,
380                                                   String bpLine,
381                                                   String property)
382             throws JDITestRuntimeException {
383 
384         log2("......setting up a breakpoint:");
385         log2("       thread: " + thread + "; class: " + testedClass +
386                         "; method: " + methodName + "; line: " + bpLine);
387 
388         List              alllineLocations = null;
389         Location          lineLocation     = null;
390         BreakpointRequest breakpRequest    = null;
391 
392         try {
393             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
394 
395             alllineLocations = method.allLineLocations();
396 
397             int n =
398                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
399             if (n > alllineLocations.size()) {
400                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
401             } else {
402                 lineLocation = (Location) alllineLocations.get(n);
403                 try {
404                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
405                     breakpRequest.putProperty("number", property);
406                     breakpRequest.addThreadFilter(thread);
407                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
408                 } catch ( Exception e1 ) {
409                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
410                     breakpRequest    = null;
411                 }
412             }
413         } catch ( Exception e2 ) {
414             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
415             breakpRequest    = null;
416         }
417 
418         if (breakpRequest == null) {
419             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
420             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
421         }
422 
423         log2("      a breakpoint has been set up");
424         return breakpRequest;
425     }

428     private void getEventSet()
429                  throws JDITestRuntimeException {
430         try {
431 //            log2("       eventSet = eventQueue.remove(waitTime);");
432             eventSet = eventQueue.remove(waitTime);
433             if (eventSet == null) {
434                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
435             }
436 //            log2("       eventIterator = eventSet.eventIterator;");
437             eventIterator = eventSet.eventIterator();
438         } catch ( Exception e ) {
439             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
440         }
441     }

444     private void breakpointForCommunication()
445                  throws JDITestRuntimeException {
446 
447         log2("breakpointForCommunication");
448         getEventSet();
449 
450         if (eventIterator.nextEvent() instanceof BreakpointEvent)
451             return;
452 
453         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
454     }
