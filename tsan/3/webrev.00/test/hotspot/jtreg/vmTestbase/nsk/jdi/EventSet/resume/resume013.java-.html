 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test case includes SUSPEND_ALL for VMDeathEvent set.         <BR>
 46  * <BR>
 47  * The test works as follows.                                   <BR>
 48  * <BR>
 49  * Upon launching debuggee's VM which will be suspended,                <BR>
 50  * a debugger waits for the VMStartEvent within a predefined            <BR>
 51  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 52  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 53  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 54  * and waits for the event within the predefined time interval.         <BR>
 55  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 56  * Upon getting the ClassPrepareEvent,                                  <BR>
 57  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 58  * within debuggee's special methodForCommunication().                  <BR>
 59  * <BR>
 60  * Then to check the above,                                             <BR>
 61  * the debugger and the debuggee perform the following.                 <BR>
 62  * - The debugger sets up a VMDeathRequest, resumes                     <BR>
 63  *   the debuggee, and waits for the VMDeathEvent.                      <BR>
 64  * - The debuggee ends to be resulting in the VMDeathEvent.             <BR>
 65  * - Upon getting the event, the debugger                               <BR>
 66  *   performs the check corresponding to the event.                     <BR>
 67  * <BR>
 68  * Note. To inform each other of needed actions, the debugger and       <BR>
 69  *       and the debuggee use debuggee's variable "instruction".        <BR>
 70  * <BR>
 71  */
 72

154     private int runThis (String argv[], PrintStream out) {
155 
156         argsHandler     = new ArgumentHandler(argv);
157         logHandler      = new Log(out, argsHandler);
158         Binder binder   = new Binder(argsHandler, logHandler);
159 
160         waitTime        = argsHandler.getWaitTime() * 60000;
161 
162         try {
163             log2("launching a debuggee :");
164             log2("       " + debuggeeName);
165             if (argsHandler.verbose()) {
166                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
167             } else {
168                 debuggee = binder.bindToDebugee(debuggeeName);
169             }
170             if (debuggee == null) {
171                 log3("ERROR: no debuggee launched");
172                 return FAILED;
173             }
174             log2("debuggee launched");
175         } catch ( Exception e ) {
176             log3("ERROR: Exception : " + e);
177             log2("       test cancelled");
178             return FAILED;
179         }
180 
181         debuggee.redirectOutput(logHandler);
182 
183         vm = debuggee.VM();
184 
185         eventQueue = vm.eventQueue();
186         if (eventQueue == null) {
187             log3("ERROR: eventQueue == null : TEST ABORTED");
188             vm.exit(PASS_BASE);
189             return FAILED;
190         }
191 
192         log2("invocation of the method runTest()");
193         switch (runTest()) {
194 
195             case 0 :  log2("test phase has finished normally");
196                       log2("   waiting for the debuggee to finish ...");
197                       debuggee.waitFor();
198 
199                       log2("......getting the debuggee's exit status");
200                       int status = debuggee.getStatus();
201                       if (status != PASS_BASE) {
202                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
203                               status + " != PASS_BASE");
204                           testExitCode = FAILED;
205                       } else {
206                           log2("......debuggee returned expected exit status: " +
207                               status + " == PASS_BASE");
208                       }
209                       break;
210 
211             default : log3("ERROR: runTest() returned unexpected value");
212 
213             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
214                       log2("......forcing: vm.exit();");
215                       testExitCode = FAILED;
216                       try {
217                           vm.exit(PASS_BASE);
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : e");
220                       }
221                       break;
222 
223             case 2 :  log3("test cancelled due to VMDisconnectedException");
224                       log2("......trying: vm.process().destroy();");
225                       testExitCode = FAILED;
226                       try {
227                           Process vmProcess = vm.process();
228                           if (vmProcess != null) {
229                               vmProcess.destroy();
230                           }
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : e");
233                       }
234                       break;
235             }
236 
237         return testExitCode;
238     }

241    /*
242     * Return value: 0 - normal end of the test
243     *               1 - ubnormal end of the test
244     *               2 - VMDisconnectedException while test phase
245     */

247     private int runTest() {
248 
249         try {
250             testRun();
251 
252             log2("waiting for VMDeathEvent");
253             getEventSet();
254             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
255                 log3("ERROR: last event is not the VMDeathEvent");
256                 return 1;
257             }
258             check();
259 
260             log2("waiting for VMDisconnectEvent");
261             getEventSet();
262             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
263                 log3("ERROR: last event is not the VMDisconnectEvent");
264                 return 1;
265             }
266 
267             return 0;
268 
269         } catch ( VMDisconnectedException e ) {
270             log3("ERROR: VMDisconnectedException : " + e);
271             return 2;
272         } catch ( Exception e ) {
273             log3("ERROR: Exception : " + e);
274             return 1;
275         }
276 
277     }

279     private void testRun()
280                  throws JDITestRuntimeException, Exception {
281 
282         if ( !vm.canRequestVMDeathEvent() ) {
283             log2("......vm.canRequestVMDeathEvent == false :: test cancelled");
284             vm.exit(PASS_BASE);
285             return;
286         }
287 
288 
289         eventRManager = vm.eventRequestManager();
290 
291         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
292         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
293         cpRequest.addClassFilter(debuggeeName);
294 
295         cpRequest.enable();
296         vm.resume();
297         getEventSet();
298         cpRequest.disable();
299 
300         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
301         debuggeeClass = event.referenceType();
302 
303         if (!debuggeeClass.name().equals(debuggeeName))
304            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
305 
306         log2("      received: ClassPrepareEvent for debuggeeClass");
307 
308         String bPointMethod = "methodForCommunication";
309         String lineForComm  = "lineForComm";
310         BreakpointRequest bpRequest;
311         ThreadReference   mainThread = threadByName("main");
312         bpRequest = settingBreakpoint(mainThread,
313                                       debuggeeClass,
314                                       bPointMethod, lineForComm, "zero");
315         bpRequest.enable();
316 
317         vm.resume();
318 
319     //------------------------------------------------------  testing section
320 
321         log1("     TESTING BEGINS");
322 
323 //        EventRequest eventRequest1 = null;
324 //        EventRequest eventRequest2 = null;
325         EventRequest eventRequest3 = null;
326 
327         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
328         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
329         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
330 
331 
332         breakpointForCommunication();
333 
334             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
335 
336         eventRequest3 = settingVMDeathRequest ( SUSPEND_ALL, "VMDeathRequest3");
337         eventRequest3.enable();
338 
339         mainThread.resume();
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
342 
343         log1("    TESTING ENDS");
344         return;
345     }

347     private ThreadReference threadByName(String name)
348                  throws JDITestRuntimeException {
349 
350         List         all = vm.allThreads();
351         ListIterator li  = all.listIterator();
352 
353         for (; li.hasNext(); ) {
354             ThreadReference thread = (ThreadReference) li.next();
355             if (thread.name().equals(name))
356                 return thread;
357         }
358         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
359     }

361    /*
362     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
363     *                                             String, String, String)
364     *
365     * It sets up a breakpoint at given line number within a given method in a given class
366     * for a given thread.
367     *
368     * Return value: BreakpointRequest object  in case of success
369     *
370     * JDITestRuntimeException   in case of an Exception thrown within the method
371     */
372 
373     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
374                                                   ReferenceType testedClass,
375                                                   String methodName,
376                                                   String bpLine,
377                                                   String property)
378             throws JDITestRuntimeException {
379 
380         log2("......setting up a breakpoint:");
381         log2("       thread: " + thread + "; class: " + testedClass +
382                         "; method: " + methodName + "; line: " + bpLine);
383 
384         List              alllineLocations = null;
385         Location          lineLocation     = null;
386         BreakpointRequest breakpRequest    = null;
387 
388         try {
389             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
390 
391             alllineLocations = method.allLineLocations();
392 
393             int n =
394                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
395             if (n > alllineLocations.size()) {
396                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
397             } else {
398                 lineLocation = (Location) alllineLocations.get(n);
399                 try {
400                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
401                     breakpRequest.putProperty("number", property);
402                     breakpRequest.addThreadFilter(thread);
403                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
404                 } catch ( Exception e1 ) {
405                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
406                     breakpRequest    = null;
407                 }
408             }
409         } catch ( Exception e2 ) {
410             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
411             breakpRequest    = null;
412         }
413 
414         if (breakpRequest == null) {
415             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
416             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
417         }
418 
419         log2("      a breakpoint has been set up");
420         return breakpRequest;
421     }

424     private void getEventSet()
425                  throws JDITestRuntimeException {
426         try {
427 //            log2("       eventSet = eventQueue.remove(waitTime);");
428             eventSet = eventQueue.remove(waitTime);
429             if (eventSet == null) {
430                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
431             }
432 //            log2("       eventIterator = eventSet.eventIterator;");
433             eventIterator = eventSet.eventIterator();
434         } catch ( Exception e ) {
435             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
436         }
437     }

440     private void breakpointForCommunication()
441                  throws JDITestRuntimeException {
442 
443         log2("breakpointForCommunication");
444         getEventSet();
445 
446         if (eventIterator.nextEvent() instanceof BreakpointEvent)
447             return;
448 
449         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
450     }

454     private VMDeathRequest settingVMDeathRequest( int    suspendPolicy,
455                                                   String property       )
456             throws JDITestRuntimeException {
457         try {
458             log2("......setting up VMDeathRequest:");
459             log2("      suspendPolicy: " + suspendPolicy + "; property: " + property);
460 
461             VMDeathRequest
462             vmdr = eventRManager.createVMDeathRequest();
463             vmdr.putProperty("number", property);
464             vmdr.setSuspendPolicy(suspendPolicy);
465 
466             return vmdr;
467         } catch ( Exception e ) {
468             log3("ERROR: ATTENTION: Exception within settingVMDeathRequest() : " + e);
469             log3("       VMDeathRequest HAS NOT BEEN SET UP");
470             throw new JDITestRuntimeException("** FAILURE to set up a VMDeathRequest **");
471         }
472     }

474     void check() {
475 
476         log2("......checking up on EventSet.resume()");
477         try {
478             log2("        eventSet.resume();  :: no Exception expected");
479             eventSet.resume();
480         } catch ( Exception e ) {
481             log3("ERROR: Exception : " + e);
482             testExitCode = FAILED;
483             log2("        vm.resume;");
484             vm.resume();
485         }
