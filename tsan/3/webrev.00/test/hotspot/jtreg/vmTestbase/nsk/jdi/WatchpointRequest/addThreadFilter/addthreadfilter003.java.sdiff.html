416     private ThreadReference threadByName(String name)
417                  throws JDITestRuntimeException {
418 
419         List         all = vm.allThreads();
420         ListIterator li  = all.listIterator();
421 
422         for (; li.hasNext(); ) {
423             ThreadReference thread = (ThreadReference) li.next();
424             if (thread.name().equals(name))
425                 return thread;
426         }
427         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
428     }
