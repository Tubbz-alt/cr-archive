 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.deleteEventRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *     After invoking the tested method,                        <BR>
 47  *     the methods returning Lists of requests,                 <BR>
 48  *     such as EventRequestManager.accessWatchpointRequests(),  <BR>
 49  *     return a List containing no eventRequests                <BR>
 50  *     deleted by the tested method.                            <BR>
 51  * Testcases include all subclasses of EventRequest.            <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee prepares testcases and invokes                        <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent,                                  <BR>
 74  *   the debugger performs the checks required.                         <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : " + e);
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : " + e);
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

505     private ThreadReference threadByName(String name)
506                  throws JDITestRuntimeException {
507 
508         List         all = vm.allThreads();
509         ListIterator li  = all.listIterator();
510 
511         for (; li.hasNext(); ) {
512             ThreadReference thread = (ThreadReference) li.next();
513             if (thread.name().equals(name))
514                 return thread;
515         }
516         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
517     }

519    /*
520     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
521     *                                             String, String, String)
522     *
523     * It sets up a breakpoint at given line number within a given method in a given class
524     * for a given thread.
525     *
526     * Return value: BreakpointRequest object  in case of success
527     *
528     * JDITestRuntimeException   in case of an Exception thrown within the method
529     */
530 
531     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
532                                                   ReferenceType testedClass,
533                                                   String methodName,
534                                                   String bpLine,
535                                                   String property)
536             throws JDITestRuntimeException {
537 
538         log2("......setting up a breakpoint:");
539         log2("       thread: " + thread + "; class: " + testedClass +
540                         "; method: " + methodName + "; line: " + bpLine);
541 
542         List              alllineLocations = null;
543         Location          lineLocation     = null;
544         BreakpointRequest breakpRequest    = null;
545 
546         try {
547             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
548 
549             alllineLocations = method.allLineLocations();
550 
551             int n =
552                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
553             if (n > alllineLocations.size()) {
554                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
555             } else {
556                 lineLocation = (Location) alllineLocations.get(n);
557 breakpLocation = lineLocation;
558                 try {
559                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
560                     breakpRequest.putProperty("number", property);
561                     breakpRequest.addThreadFilter(thread);
562                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
563                 } catch ( Exception e1 ) {
564                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
565                     breakpRequest    = null;
566                 }
567             }
568         } catch ( Exception e2 ) {
569             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
570             breakpRequest    = null;
571         }
572 
573         if (breakpRequest == null) {
574             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
575             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
576         }
577 
578         log2("      a breakpoint has been set up");
579         return breakpRequest;
580     }

583     private void getEventSet()
584                  throws JDITestRuntimeException {
585         try {
586 //            log2("       eventSet = eventQueue.remove(waitTime);");
587             eventSet = eventQueue.remove(waitTime);
588             if (eventSet == null) {
589                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
590             }
591 //            log2("       eventIterator = eventSet.eventIterator;");
592             eventIterator = eventSet.eventIterator();
593         } catch ( Exception e ) {
594             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
595         }
596     }

607
