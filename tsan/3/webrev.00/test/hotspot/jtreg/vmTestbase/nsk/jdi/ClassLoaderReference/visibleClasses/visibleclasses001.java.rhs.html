 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ClassLoader.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ClassLoader.visibleClasses()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *   Returns a list of all classes for which this class loader  <BR>
 47  *   has been recorded as the initiating loader in the target VM.<BR>
 48  *   The list contains ReferenceTypes defined directly by       <BR>
 49  *   this loader (as returned by definedClasses()) and          <BR>
 50  *   any types for which loading was delegated by               <BR>
 51  *   this class loader to another class loader.                 <BR>
 52  * <BR>
 53  * The case to check includes a class loader with no            <BR>
 54  * delegated types, that is, the expected returned value should <BR>
 55  * be equal to one returned by ClassLoader.definedClasses(),    <BR>
 56  * except primitive arrays defined by bootstrap class loader    <BR>
 57  * and visible.                                                 <BR>
 58  * <BR>
 59  * The test has three phases and works as follows.              <BR>
 60  * <BR>
 61  * In first phase,                                              <BR>
 62  * upon launching debuggee's VM which will be suspended,                <BR>
 63  * a debugger waits for the VMStartEvent within a predefined            <BR>
 64  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 65  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 66  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 67  * and waits for the event within the predefined time interval.         <BR>
 68  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 69  * Upon getting the ClassPrepareEvent,                                  <BR>
 70  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 71  * within debuggee's special methodForCommunication().                  <BR>
 72  * <BR>
 73  * In second phase to check the assetion,                               <BR>
 74  * the debugger and the debuggee perform the following.                 <BR>
 75  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 76  * - The debuggee prepares a ClassLoader object and invokes             <BR>
 77  *   the methodForCommunication to be suspended and                     <BR>
 78  *   to inform the debugger with the event.                             <BR>
 79  * - Upon getting the BreakpointEvent,                                  <BR>
 80  *   the debugger performs the check.                                   <BR>
 81  * Note. To inform each other of needed actions, the debugger and       <BR>
 82  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 83  * <BR>
 84  * In third phase when at the end,                                      <BR>
 85  * the debuggee changes the value of the "instruction"                  <BR>
 86  * to inform the debugger of checks finished, and both end.             <BR>
 87  * <BR>
 88  */

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

252    /*
253     * Return value: 0 - normal end of the test
254     *               1 - ubnormal end of the test
255     *               2 - VMDisconnectedException while test phase
256     */
257 
258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if (eventIterator.nextEvent() instanceof VMDeathEvent)
266                 return 0;
267 
268             log3("ERROR: last event is not the VMDeathEvent");
269             return 1;
270         } catch ( VMDisconnectedException e ) {
271             log3("ERROR: VMDisconnectedException : " + e);
272             return 2;
273         } catch ( Exception e ) {
274             log3("ERROR: Exception : " + e);
275             return 1;
276         }
277 
278     }

280     private void testRun()
281                  throws JDITestRuntimeException, Exception {
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304         BreakpointRequest bpRequest;
305 
306         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
307                                       debuggeeClass,
308                                       bPointMethod, lineForComm, "zero");
309         bpRequest.enable();
310 
311     //------------------------------------------------------  testing section
312 
313         log1("     TESTING BEGINS");
314 
315         for (int i = 0; ; i++) {
316 
317             vm.resume();
318             breakpointForCommunication();
319 
320             int instruction = ((IntegerValue)
321                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
322 
323             if (instruction == 0) {
324                 vm.resume();
325                 break;
326             }
327 
328             log1(":::::: case: # 0");
329 
330             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
331 
332             log2("......getting: List classes = vm.allClasses();");
333             List<ReferenceType> classes = vm.allClasses();
334             log2("               classes.size() == " + classes.size());
335 
336             vm.resume();
337             breakpointForCommunication();
338 
339             log1(":::::: case: # 1");
340 
341             String ClassLoaderObjName = "classLoader";
342 
343             log2("......getting: Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));");
344             Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));
345 
346             log2("......getting: ClassLoaderReference clRef = (ClassLoaderReference) val;");
347             ClassLoaderReference clRef = (ClassLoaderReference) val;
348 
349             log2("......getting: List definedClasses = clRef.definedClasses();");
350             List<ReferenceType> definedClasses = clRef.definedClasses();
351 
352             log2("......getting: List visibleClasses = clRef.visibleClasses();");
353             List<ReferenceType> visibleClasses = clRef.visibleClasses();
354 
355             log2("......checking up on: visibleClasses.size() == definedClasses.size()");
356 
357             if (visibleClasses.size() != definedClasses.size()) {
358                 log2("     : visibleClasses.size() != definedClasses.size()");
359                 log2("     : definedClasses.size() == " + definedClasses.size());
360                 log2("     : visibleClasses.size() == " + visibleClasses.size());
361 
362                 for (ReferenceType vcl : visibleClasses) {
363 
364                     int vclIndex = visibleClasses.indexOf(vcl);
365                     String vclName = vcl.name();
366 
367                     if (primitiveArraysNamesPatterns.stream().anyMatch(vcl.name()::matches)) {
368                         log2("     : visibleClasses[" + vclIndex + "].name() == " + vclName +
369                             " Correct - primitive arrays are visible for class loader");
370                     } else {
371                         log3("     : visibleClasses[" + vclIndex + "].name() == " + vclName);
372 
373                         classes.stream().filter(cl -> cl.name().equals(vclName))
374                                         .forEach(cl -> log3("     :  List classes contains an object with the name: " + cl.name()));
375 
376                         testExitCode = FAILED;
377                     }
378                 }
379             }
380 
381             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
382         }
383         log1("    TESTING ENDS");
384         return;
385     }

387    /*
388     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
389     *                                             String, String, String)
390     *
391     * It sets up a breakpoint at given line number within a given method in a given class
392     * for a given thread.
393     *
394     * Return value: BreakpointRequest object  in case of success
395     *
396     * JDITestRuntimeException   in case of an Exception thrown within the method
397     */
398 
399     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
400                                                   ReferenceType testedClass,
401                                                   String methodName,
402                                                   String bpLine,
403                                                   String property)
404             throws JDITestRuntimeException {
405 
406         log2("......setting up a breakpoint:");
407         log2("       thread: " + thread + "; class: " + testedClass +
408                         "; method: " + methodName + "; line: " + bpLine);
409 
410         List              alllineLocations = null;
411         Location          lineLocation     = null;
412         BreakpointRequest breakpRequest    = null;
413 
414         try {
415             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
416 
417             alllineLocations = method.allLineLocations();
418 
419             int n =
420                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
421             if (n > alllineLocations.size()) {
422                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
423             } else {
424                 lineLocation = (Location) alllineLocations.get(n);
425                 try {
426                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
427                     breakpRequest.putProperty("number", property);
428                     breakpRequest.addThreadFilter(thread);
429                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
430                 } catch ( Exception e1 ) {
431                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
432                     breakpRequest    = null;
433                 }
434             }
435         } catch ( Exception e2 ) {
436             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
437             breakpRequest    = null;
438         }
439 
440         if (breakpRequest == null) {
441             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
442             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
443         }
444 
445         log2("      a breakpoint has been set up");
446         return breakpRequest;
447     }

450     private void getEventSet()
451                  throws JDITestRuntimeException {
452         try {
453 //            log2("       eventSet = eventQueue.remove(waitTime);");
454             eventSet = eventQueue.remove(waitTime);
455             if (eventSet == null) {
456                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
457             }
458 //            log2("       eventIterator = eventSet.eventIterator;");
459             eventIterator = eventSet.eventIterator();
460         } catch ( Exception e ) {
461             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
462         }
463     }

466     private void breakpointForCommunication()
467                  throws JDITestRuntimeException {
468 
469         log2("breakpointForCommunication");
470         getEventSet();
471 
472         if (eventIterator.nextEvent() instanceof BreakpointEvent)
473             return;
474 
475         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
476     }
