 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodExitRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodExitRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Restricts the events generated by this request            <BR>
 47  *    to those in the given thread.                             <BR>
 48  * The cases to test include three thread's states:             <BR>
 49  * not started, running, dead.                                  <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates MethodExitRequest1 and                                   <BR>
 72  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 73  *   - creates MethodExitRequest2,                                      <BR>
 74  *     resumes the debuggee to get the thread1 started, and             <BR>
 75  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 76  *   - creates MethodExitRequest3, waits the thread1 to die, and        <BR>
 77  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugee(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : " + e);
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : " + e);
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

249     * Return value: 0 - normal end of the test
250     *               1 - ubnormal end of the test
251     *               2 - VMDisconnectedException while test phase
252     */

254     private int runTest() {
255 
256         try {
257             testRun();
258 
259             log2("waiting for VMDeathEvent");
260             getEventSet();
261             if (eventIterator.nextEvent() instanceof VMDeathEvent)
262                 return 0;
263 
264             log3("ERROR: last event is not the VMDeathEvent");
265             return 1;
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273 
274     }

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         eventRManager = vm.eventRequestManager();
280 
281         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
282         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
283         cpRequest.addClassFilter(debuggeeName);
284 
285         cpRequest.enable();
286         vm.resume();
287         getEventSet();
288         cpRequest.disable();
289 
290         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
291         debuggeeClass = event.referenceType();
292 
293         if (!debuggeeClass.name().equals(debuggeeName))
294            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
295 
296         log2("      received: ClassPrepareEvent for debuggeeClass");
297 
298         String bPointMethod = "methodForCommunication";
299         String lineForComm  = "lineForComm";
300 
301         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
302 
303         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
304                                              debuggeeClass,
305                                             bPointMethod, lineForComm, "zero");
306         bpRequest.enable();
307 
308     //------------------------------------------------------  testing section
309 
310         log1("     TESTING BEGINS");
311 
312 
313         EventRequest eventRequest1 = null;
314         EventRequest eventRequest2 = null;
315         EventRequest eventRequest3 = null;
316 
317         ThreadReference thread1     = null;
318         String          thread1Name = "thread1";
319 
320         String property1 = "MethodExitRequest1";
321         String property2 = "MethodExitRequest2";
322         String property3 = "MethodExitRequest3";
323 
324         ReferenceType testClassReference = null;
325 
326         for (int i = 0; ; i++) {
327 
328             vm.resume();
329             breakpointForCommunication();
330 
331             int instruction = ((IntegerValue)
332                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
333 
334             if (instruction == 0) {
335                 vm.resume();
336                 break;
337             }
338 
339             log1(":::::: case: # " + i);
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
342 
343             switch (i) {
344               case 0:
345                      testClassReference =
346                             (ReferenceType) vm.classesByName(testedClassName).get(0);
347                      thread1 = (ThreadReference) debuggeeClass.getValue(
348                                         debuggeeClass.fieldByName(thread1Name));
349 
350                      eventRequest1 = setting2MethodExitRequest (null,
351                                              testClassReference,
352                                              EventRequest.SUSPEND_NONE, property1);
353 
354                      try {
355                          log2("......eventRequest1.addThreadFilter(thread1);");
356                          log2("        no Exception expected");
357                          ((MethodExitRequest)eventRequest1).addThreadFilter(thread1);
358                      } catch ( Exception e ) {
359                          log3("ERROR: Exception : " + e);
360                          testExitCode = FAILED;
361                      }
362                      break;
363 
364               case 1:
365                      eventRequest2 = setting2MethodExitRequest (null,
366                                              testClassReference,
367                                              EventRequest.SUSPEND_NONE, property2);
368 
369                      try {
370                          log2("......eventRequest2.addThreadFilter(thread1);");
371                          log2("        no Exception expected");
372                          ((MethodExitRequest)eventRequest2).addThreadFilter(thread1);
373                      } catch ( Exception e ) {
374                          log3("ERROR: Exception : " + e);
375                          testExitCode = FAILED;
376                      }
377                      break;
378 
379               case 2:
380                      eventRequest3 = setting2MethodExitRequest (null,
381                                              testClassReference,
382                                              EventRequest.SUSPEND_NONE, property3);
383 
384                      try {
385                          log2("......eventRequest3.addThreadFilter(thread1);");
386                          log2("        no Exception expected");
387                          ((MethodExitRequest)eventRequest3).addThreadFilter(thread1);
388                      } catch ( Exception e ) {
389                          log3("ERROR: Exception : " + e);
390                          testExitCode = FAILED;
391                      }
392                      break;
393 
394 
395               default:
396                       throw new JDITestRuntimeException("** default case 2 **");
397             }
398 
399             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
400         }
401         log1("    TESTING ENDS");
402         return;

405    /*
406     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
407     *                                             String, String, String)
408     *
409     * It sets up a breakpoint at given line number within a given method in a given class
410     * for a given thread.
411     *
412     * Return value: BreakpointRequest object  in case of success
413     *
414     * JDITestRuntimeException   in case of an Exception thrown within the method
415     */
416 
417     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
418                                                   ReferenceType testedClass,
419                                                   String methodName,
420                                                   String bpLine,
421                                                   String property)
422             throws JDITestRuntimeException {
423 
424         log2("......setting up a breakpoint:");
425         log2("       thread: " + thread + "; class: " + testedClass +
426                         "; method: " + methodName + "; line: " + bpLine);
427 
428         List              alllineLocations = null;
429         Location          lineLocation     = null;
430         BreakpointRequest breakpRequest    = null;
431 
432         try {
433             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
434 
435             alllineLocations = method.allLineLocations();
436 
437             int n =
438                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
439             if (n > alllineLocations.size()) {
440                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
441             } else {
442                 lineLocation = (Location) alllineLocations.get(n);
443                 try {
444                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
445                     breakpRequest.putProperty("number", property);
446                     breakpRequest.addThreadFilter(thread);
447                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
448                 } catch ( Exception e1 ) {
449                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
450                     breakpRequest    = null;
451                 }
452             }
453         } catch ( Exception e2 ) {
454             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
455             breakpRequest    = null;
456         }
457 
458         if (breakpRequest == null) {
459             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
460             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
461         }
462 
463         log2("      a breakpoint has been set up");
464         return breakpRequest;
465     }

468     private void getEventSet()
469                  throws JDITestRuntimeException {
470         try {
471 //            log2("       eventSet = eventQueue.remove(waitTime);");
472             eventSet = eventQueue.remove(waitTime);
473             if (eventSet == null) {
474                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
475             }
476 //            log2("       eventIterator = eventSet.eventIterator;");
477             eventIterator = eventSet.eventIterator();
478         } catch ( Exception e ) {
479             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
480         }
481     }

484     private void breakpointForCommunication()
485                  throws JDITestRuntimeException {
486 
487         log2("breakpointForCommunication");
488         getEventSet();
489 
490         if (eventIterator.nextEvent() instanceof BreakpointEvent)
491             return;
492 
493         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
494     }

498     private MethodExitRequest setting2MethodExitRequest ( ThreadReference thread,
499                                                          ReferenceType   testedClass,
500                                                          int             suspendPolicy,
501                                                          String          property        )
502             throws JDITestRuntimeException {
503         try {
504             log2("......setting up MethodExitRequest:");
505             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
506 
507             MethodExitRequest
508             mexr = eventRManager.createMethodExitRequest();
509             mexr.putProperty("number", property);
510             if (thread != null)
511                 mexr.addThreadFilter(thread);
512             mexr.addClassFilter(testedClass);
513             mexr.setSuspendPolicy(suspendPolicy);
514 
515         log2("      MethodExitRequest has been set up");
516         return mexr;
517         } catch ( Exception e ) {
518             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
519             log3("       MethodExitRequest HAS NOT BEEN SET UP");
520             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
521         }
522     }
