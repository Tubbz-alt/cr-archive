 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.location()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a Location object, an argument of the method      <BR>
 46  *    EventRequestManager.createBreakpointRequest(Location),            <BR>
 47  * is equal to one returned by the method BreakpointRequest.location()  <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.                      <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assertion,                              <BR>
 64  * the debugger and the debuggee perform the following.                 <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee creates new threads, thread1, and invokes             <BR>
 67  *   the methodForCommunication to be suspended and                     <BR>
 68  *   to inform the debugger with the event.                             <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates a BreakpointRequest and gets Location object location1,  <BR>
 71  *   - gets Location object location2 with the method                   <BR>
 72  *     BreakpointRequest.location(),                                    <BR>
 73  *   - compares the Location objects.                                   <BR>
 74  * <BR>
 75  * In third phase, at the end of the test,                              <BR>
 76  * the debuggee changes the value of the "instruction"  which           <BR>
 77  * the debugger and debuggee use to inform each other of needed actions,<BR>
 78  * and both end.
 79  * <BR>
 80  */
 81

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

253    /*
254     * Return value: 0 - normal end of the test
255     *               1 - ubnormal end of the test
256     *               2 - VMDisconnectedException while test phase
257     */

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278 
279     }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest eventRequest1 = null;
318 
319         ThreadReference thread1     = null;
320         String          thread1Name = "thread1";
321 
322         String property1 = "BreakpointRequest1";
323 
324         String methodName = "method";
325         String bpLineName = "breakpointLine";
326 
327         ReferenceType testClassReference = null;
328 
329         for (int i = 0; ; i++) {
330 
331             vm.resume();
332             breakpointForCommunication();
333 
334             int instruction = ((IntegerValue)
335                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
336 
337             if (instruction == 0) {
338                 vm.resume();
339                 break;
340             }
341 
342             log1(":::::: case: # " + i);
343 
344             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
345 
346             switch (i) {
347 
348               case 0:
349                      testClassReference =
350                             (ReferenceType) vm.classesByName(testedClassName).get(0);
351 
352                      thread1 = (ThreadReference) debuggeeClass.getValue(
353                                         debuggeeClass.fieldByName(thread1Name));
354 
355                      log2(".....setting up BreakpointRequest and getting Location object location1");
356                      eventRequest1 = setting2BreakpointRequest (null,
357                                              testClassReference, methodName, bpLineName,
358                                              EventRequest.SUSPEND_NONE, property1);
359 
360                      log2("......getting: location2 = ((BreakpointRequest) eventRequest1).location();");
361                      location2 = ((BreakpointRequest) eventRequest1).location();
362 
363                      log2("      checking up on equality of location1 and location2");
364                      if ( !location1.equals(location2) ) {
365                          testExitCode = FAILED;
366                          log3("ERROR: Location objects are not equal");
367                      }
368 
369                      break;
370 
371               default:
372                       throw new JDITestRuntimeException("** default case 2 **");
373             }
374             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
375         }
376         log1("    TESTING ENDS");
377         return;
378     }

380     private ThreadReference threadByName(String name)
381                  throws JDITestRuntimeException {
382 
383         List         all = vm.allThreads();
384         ListIterator li  = all.listIterator();
385 
386         for (; li.hasNext(); ) {
387             ThreadReference thread = (ThreadReference) li.next();
388             if (thread.name().equals(name))
389                 return thread;
390         }
391         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
392     }

394    /*
395     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
396     *                                             String, String, String)
397     *
398     * It sets up a breakpoint at given line number within a given method in a given class
399     * for a given thread.
400     *
401     * Return value: BreakpointRequest object  in case of success
402     *
403     * JDITestRuntimeException   in case of an Exception thrown within the method
404     */
405 
406     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
407                                                   ReferenceType testedClass,
408                                                   String methodName,
409                                                   String bpLine,
410                                                   String property)
411             throws JDITestRuntimeException {
412 
413         log2("......setting up a breakpoint:");
414         log2("       thread: " + thread + "; class: " + testedClass +
415                         "; method: " + methodName + "; line: " + bpLine);
416 
417         List              alllineLocations = null;
418         Location          lineLocation     = null;
419         BreakpointRequest breakpRequest    = null;
420 
421         try {
422             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
423 
424             alllineLocations = method.allLineLocations();
425 
426             int n =
427                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
428             if (n > alllineLocations.size()) {
429                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
430             } else {
431                 lineLocation = (Location) alllineLocations.get(n);
432                 try {
433                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
434                     breakpRequest.putProperty("number", property);
435                     breakpRequest.addThreadFilter(thread);
436                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
437                 } catch ( Exception e1 ) {
438                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
439                     breakpRequest    = null;
440                 }
441             }
442         } catch ( Exception e2 ) {
443             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
444             breakpRequest    = null;
445         }
446 
447         if (breakpRequest == null) {
448             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
449             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
450         }
451 
452         log2("      a breakpoint has been set up");
453         return breakpRequest;
454     }

457     private void getEventSet()
458                  throws JDITestRuntimeException {
459         try {
460 //            log2("       eventSet = eventQueue.remove(waitTime);");
461             eventSet = eventQueue.remove(waitTime);
462             if (eventSet == null) {
463                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
464             }
465 //            log2("       eventIterator = eventSet.eventIterator;");
466             eventIterator = eventSet.eventIterator();
467         } catch ( Exception e ) {
468             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
469         }
470     }

473     private void breakpointForCommunication()
474                  throws JDITestRuntimeException {
475 
476         log2("breakpointForCommunication");
477         getEventSet();
478 
479         if (eventIterator.nextEvent() instanceof BreakpointEvent)
480             return;
481 
482         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
483     }

487     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
488                                                           ReferenceType   testedClass,
489                                                           String          methodName,
490                                                           String          bpLine,
491                                                           int             suspendPolicy,
492                                                           String          property        )
493             throws JDITestRuntimeException {
494         try {
495             log2("......setting up a breakpoint:");
496             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
497 
498             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
499             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
500 
501             location1 = loc;
502 
503             BreakpointRequest
504             bpr = eventRManager.createBreakpointRequest(loc);
505             bpr.putProperty("number", property);
506             if (thread != null)
507                 bpr.addThreadFilter(thread);
508             bpr.setSuspendPolicy(suspendPolicy);
509 
510             log2("      a breakpoint has been set up");
511             return bpr;
512         } catch ( Exception e ) {
513             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
514             log3("       BreakpointRequest HAS NOT BEEN SET UP");
515             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
516         }
