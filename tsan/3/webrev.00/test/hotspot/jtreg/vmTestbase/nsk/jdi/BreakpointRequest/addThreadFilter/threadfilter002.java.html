 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                              <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new threads, thread1, and invokes             <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a BreakpointRequest and enables it,                      <BR>
 72  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addThreadFilter()    <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugee(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : " + e);
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : " + e);
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

251     *               2 - VMDisconnectedException while test phase

254     private int runTest() {
255 
256         try {
257             testRun();
258 
259             log2("waiting for VMDeathEvent");
260             getEventSet();
261             if (eventIterator.nextEvent() instanceof VMDeathEvent)
262                 return 0;
263 
264             log3("ERROR: last event is not the VMDeathEvent");
265             return 1;
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         eventRManager = vm.eventRequestManager();
280 
281         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
282         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
283         cpRequest.addClassFilter(debuggeeName);
284 
285         cpRequest.enable();
286         vm.resume();
287         getEventSet();
288         cpRequest.disable();
289 
290         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
291         debuggeeClass = event.referenceType();
292 
293         if (!debuggeeClass.name().equals(debuggeeName))
294            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
295 
296         log2("      received: ClassPrepareEvent for debuggeeClass");
297 
298         String bPointMethod = "methodForCommunication";
299         String lineForComm  = "lineForComm";
300 
301         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
302 
303         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
304                                              debuggeeClass,
305                                             bPointMethod, lineForComm, "zero");
306         bpRequest.enable();
307 
308     //------------------------------------------------------  testing section
309 
310         log1("     TESTING BEGINS");
311 
312         EventRequest eventRequest1 = null;
313 
314         ThreadReference thread1     = null;
315         String          thread1Name = "thread1";
316 
317         String property1 = "BreakpointRequest1";
318 
319         String methodName = "method";
320         String bpLineName = "breakpointLine";
321 
322         ReferenceType testClassReference = null;
323 
324         for (int i = 0; ; i++) {
325 
326             vm.resume();
327             breakpointForCommunication();
328 
329             int instruction = ((IntegerValue)
330                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
331 
332             if (instruction == 0) {
333                 vm.resume();
334                 break;
335             }
336 
337             log1(":::::: case: # " + i);
338 
339             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
340 
341             switch (i) {
342 
343               case 0:
344                      testClassReference =
345                             (ReferenceType) vm.classesByName(testedClassName).get(0);
346 
347                      thread1 = (ThreadReference) debuggeeClass.getValue(
348                                         debuggeeClass.fieldByName(thread1Name));
349 
350                      eventRequest1 = setting2BreakpointRequest (null,
351                                              testClassReference, methodName, bpLineName,
352                                              EventRequest.SUSPEND_NONE, property1);
353 
354                      log2("......eventRequest1.enable();");
355                      eventRequest1.enable();
356 
357                      try {
358                          log2("......eventRequest1.addThreadFilter(thread1);");
359                          log2("        InvalidRequestStateException expected");
360                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
361                          log3("ERROR: no InvalidRequestStateException");
362                          testExitCode = FAILED;
363                      } catch ( InvalidRequestStateException e ) {
364                          log2("        InvalidRequestStateException");
365                      }
366 
367                      log2("......eventRequest1.disable();");
368                      eventRequest1.disable();
369                      try {
370                          log2("......eventRequest1.addThreadFilter(thread1);");
371                          log2("        no InvalidRequestStateException expected");
372                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
373                          log2("        no InvalidRequestStateException");
374                      } catch ( InvalidRequestStateException e ) {
375                          log3("ERROR: InvalidRequestStateException");
376                          testExitCode = FAILED;
377                      }
378 
379                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
380                      eventRManager.deleteEventRequest(eventRequest1);
381                      try {
382                          log2("......eventRequest1.addThreadFilter(thread1);");
383                          log2("        InvalidRequestStateException expected");
384                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
385                          log3("ERROR: no InvalidRequestStateException");
386                          testExitCode = FAILED;
387                      } catch ( InvalidRequestStateException e ) {
388                          log2("        InvalidRequestStateException");
389                      }
390 
391                      break;
392 
393               default:
394                       throw new JDITestRuntimeException("** default case 2 **");
395             }
396             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
397         }
398         log1("    TESTING ENDS");
399         return;
400     }

403    /*
404     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
405     *                                             String, String, String)
406     *
407     * It sets up a breakpoint at given line number within a given method in a given class
408     * for a given thread.
409     *
410     * Return value: BreakpointRequest object  in case of success
411     *
412     * JDITestRuntimeException   in case of an Exception thrown within the method
413     */
414 
415     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
416                                                   ReferenceType testedClass,
417                                                   String methodName,
418                                                   String bpLine,
419                                                   String property)
420             throws JDITestRuntimeException {
421 
422         log2("......setting up a breakpoint:");
423         log2("       thread: " + thread + "; class: " + testedClass +
424                         "; method: " + methodName + "; line: " + bpLine);
425 
426         List              alllineLocations = null;
427         Location          lineLocation     = null;
428         BreakpointRequest breakpRequest    = null;
429 
430         try {
431             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
432 
433             alllineLocations = method.allLineLocations();
434 
435             int n =
436                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
437             if (n > alllineLocations.size()) {
438                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
439             } else {
440                 lineLocation = (Location) alllineLocations.get(n);
441                 try {
442                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
443                     breakpRequest.putProperty("number", property);
444                     breakpRequest.addThreadFilter(thread);
445                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
446                 } catch ( Exception e1 ) {
447                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
448                     breakpRequest    = null;
449                 }
450             }
451         } catch ( Exception e2 ) {
452             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
453             breakpRequest    = null;
454         }
455 
456         if (breakpRequest == null) {
457             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
458             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
459         }
460 
461         log2("      a breakpoint has been set up");
462         return breakpRequest;
463     }

466     private void getEventSet()
467                  throws JDITestRuntimeException {
468         try {
469 //            log2("       eventSet = eventQueue.remove(waitTime);");
470             eventSet = eventQueue.remove(waitTime);
471             if (eventSet == null) {
472                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
473             }
474 //            log2("       eventIterator = eventSet.eventIterator;");
475             eventIterator = eventSet.eventIterator();
476         } catch ( Exception e ) {
477             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
478         }
479     }

482     private void breakpointForCommunication()
483                  throws JDITestRuntimeException {
484 
485         log2("breakpointForCommunication");
486         getEventSet();
487 
488         if (eventIterator.nextEvent() instanceof BreakpointEvent)
489             return;
490 
491         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
492     }

496     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
497                                                           ReferenceType   testedClass,
498                                                           String          methodName,
499                                                           String          bpLine,
500                                                           int             suspendPolicy,
501                                                           String          property        )
502             throws JDITestRuntimeException {
503         try {
504             log2("......setting up a breakpoint:");
505             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
506 
507             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
508             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
509 
510             BreakpointRequest
511             bpr = eventRManager.createBreakpointRequest(loc);
512             bpr.putProperty("number", property);
513             if (thread != null)
514                 bpr.addThreadFilter(thread);
515             bpr.setSuspendPolicy(suspendPolicy);
516 
517             log2("      a breakpoint has been set up");
518             return bpr;
519         } catch ( Exception e ) {
520             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
521             log3("       BreakpointRequest HAS NOT BEEN SET UP");
522             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
523         }
