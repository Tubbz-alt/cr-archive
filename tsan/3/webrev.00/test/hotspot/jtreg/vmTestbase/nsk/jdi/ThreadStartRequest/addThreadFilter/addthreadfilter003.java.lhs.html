 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadStartRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadStartRequest.addThreadFilter()</code>                        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  *            Filters may be added only to disabled requests.   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates an ThreadStartRequest and enables it,                    <BR>
 73  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 74  *     to catch InvalidRequestStateException,                           <BR>
 75  *   - disables the request and invokes the method addThreadFilter()    <BR>
 76  *     expecting to catch no InvalidRequestStateException,              <BR>
 77  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 78  *     once again expecting to catch InvalidRequestStateException.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  *
 86 
 87  */
 88

170     private int runThis (String argv[], PrintStream out) {
171 
172         argsHandler     = new ArgumentHandler(argv);
173         logHandler      = new Log(out, argsHandler);
174         Binder binder   = new Binder(argsHandler, logHandler);
175 
176         waitTime        = argsHandler.getWaitTime() * 60000;
177 
178         try {
179             log2("launching a debuggee :");
180             log2("       " + debuggeeName);
181             if (argsHandler.verbose()) {
182                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
183             } else {
184                 debuggee = binder.bindToDebugee(debuggeeName);
185             }
186             if (debuggee == null) {
187                 log3("ERROR: no debuggee launched");
188                 return FAILED;
189             }
190             log2("debuggee launched");
191         } catch ( Exception e ) {
192             log3("ERROR: Exception : " + e);
193             log2("       test cancelled");
194             return FAILED;
195         }
196 
197         debuggee.redirectOutput(logHandler);
198 
199         vm = debuggee.VM();
200 
201         eventQueue = vm.eventQueue();
202         if (eventQueue == null) {
203             log3("ERROR: eventQueue == null : TEST ABORTED");
204             vm.exit(PASS_BASE);
205             return FAILED;
206         }
207 
208         log2("invocation of the method runTest()");
209         switch (runTest()) {
210 
211             case 0 :  log2("test phase has finished normally");
212                       log2("   waiting for the debuggee to finish ...");
213                       debuggee.waitFor();
214 
215                       log2("......getting the debuggee's exit status");
216                       int status = debuggee.getStatus();
217                       if (status != PASS_BASE) {
218                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
219                               status + " != PASS_BASE");
220                           testExitCode = FAILED;
221                       } else {
222                           log2("......debuggee returned expected exit status: " +
223                               status + " == PASS_BASE");
224                       }
225                       break;
226 
227             default : log3("ERROR: runTest() returned unexpected value");
228 
229             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
230                       log2("......forcing: vm.exit();");
231                       testExitCode = FAILED;
232                       try {
233                           vm.exit(PASS_BASE);
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : e");
236                       }
237                       break;
238 
239             case 2 :  log3("test cancelled due to VMDisconnectedException");
240                       log2("......trying: vm.process().destroy();");
241                       testExitCode = FAILED;
242                       try {
243                           Process vmProcess = vm.process();
244                           if (vmProcess != null) {
245                               vmProcess.destroy();
246                           }
247                       } catch ( Exception e ) {
248                           log3("ERROR: Exception : e");
249                       }
250                       break;
251             }
252 
253         return testExitCode;
254     }

263     private int runTest() {
264 
265         try {
266             testRun();
267 
268             log2("waiting for VMDeathEvent");
269             getEventSet();
270             if (eventIterator.nextEvent() instanceof VMDeathEvent)
271                 return 0;
272 
273             log3("ERROR: last event is not the VMDeathEvent");
274             return 1;
275         } catch ( VMDisconnectedException e ) {
276             log3("ERROR: VMDisconnectedException : " + e);
277             return 2;
278         } catch ( Exception e ) {
279             log3("ERROR: Exception : " + e);
280             return 1;
281         }

285     private void testRun()
286                  throws JDITestRuntimeException, Exception {
287 
288         eventRManager = vm.eventRequestManager();
289 
290         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
291         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
292         cpRequest.addClassFilter(debuggeeName);
293 
294         cpRequest.enable();
295         vm.resume();
296         getEventSet();
297         cpRequest.disable();
298 
299         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
300         debuggeeClass = event.referenceType();
301 
302         if (!debuggeeClass.name().equals(debuggeeName))
303            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
304 
305         log2("      received: ClassPrepareEvent for debuggeeClass");
306 
307         String bPointMethod = "methodForCommunication";
308         String lineForComm  = "lineForComm";
309         BreakpointRequest bpRequest;
310 
311         bpRequest = settingBreakpoint(threadByName("main"),
312                                       debuggeeClass,
313                                       bPointMethod, lineForComm, "zero");
314         bpRequest.enable();
315 
316     //------------------------------------------------------  testing section
317 
318         log1("     TESTING BEGINS");
319 
320         EventRequest eventRequest1 = null;
321 
322         ThreadReference thread1    = null;
323         String         thread1Name = "thread1";
324 
325         String property1 = "ThreadStartRequest1";
326 
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             switch (i) {
346 
347               case 0:
348                       thread1 = (ThreadReference) debuggeeClass.getValue(
349                                                   debuggeeClass.fieldByName(thread1Name));
350 
351                       eventRequest1 = setting2ThreadStartRequest (null,
352                                              EventRequest.SUSPEND_NONE, "ThreadStartRequest1");
353 
354                      log2("......eventRequest1.enable();");
355                      eventRequest1.enable();
356 
357                      try {
358                          log2("......eventRequest1.addThreadFilter(thread1);");
359                          log2("        InvalidRequestStateException expected");
360                          ((ThreadStartRequest)eventRequest1).addThreadFilter(thread1);
361                          log3("ERROR: no InvalidRequestStateException");
362                          testExitCode = FAILED;
363                      } catch ( InvalidRequestStateException e ) {
364                          log2("        InvalidRequestStateException");
365                      }
366 
367                      log2("......eventRequest1.disable();");
368                      eventRequest1.disable();
369                      try {
370                          log2("......eventRequest1.addThreadFilter(thread1);");
371                          log2("        no InvalidRequestStateException expected");
372                          ((ThreadStartRequest)eventRequest1).addThreadFilter(thread1);
373                          log2("        no InvalidRequestStateException");
374                      } catch ( InvalidRequestStateException e ) {
375                          log3("ERROR: InvalidRequestStateException");
376                          testExitCode = FAILED;
377                      }
378 
379                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
380                      eventRManager.deleteEventRequest(eventRequest1);
381                      try {
382                          log2("......eventRequest1.addThreadFilter(thread1);");
383                          log2("        InvalidRequestStateException expected");
384                          ((ThreadStartRequest)eventRequest1).addThreadFilter(thread1);
385                          log3("ERROR: no InvalidRequestStateException");
386                          testExitCode = FAILED;
387                      } catch ( InvalidRequestStateException e ) {
388                          log2("        InvalidRequestStateException");
389                      }
390                       break;
391 
392               default:
393                       throw new JDITestRuntimeException("** default case 2 **");
394             }
395 
396             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
397         }
398         log1("    TESTING ENDS");
399         return;
400     }

402     private ThreadReference threadByName(String name)
403                  throws JDITestRuntimeException {
404 
405         List         all = vm.allThreads();
406         ListIterator li  = all.listIterator();
407 
408         for (; li.hasNext(); ) {
409             ThreadReference thread = (ThreadReference) li.next();
410             if (thread.name().equals(name))
411                 return thread;
412         }
413         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
414     }

416    /*
417     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
418     *                                             String, String, String)
419     *
420     * It sets up a breakpoint at given line number within a given method in a given class
421     * for a given thread.
422     *
423     * Return value: BreakpointRequest object  in case of success
424     *
425     * JDITestRuntimeException   in case of an Exception thrown within the method
426     */
427 
428     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
429                                                   ReferenceType testedClass,
430                                                   String methodName,
431                                                   String bpLine,
432                                                   String property)
433             throws JDITestRuntimeException {
434 
435         log2("......setting up a breakpoint:");
436         log2("       thread: " + thread + "; class: " + testedClass +
437                         "; method: " + methodName + "; line: " + bpLine);
438 
439         List              alllineLocations = null;
440         Location          lineLocation     = null;
441         BreakpointRequest breakpRequest    = null;
442 
443         try {
444             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
445 
446             alllineLocations = method.allLineLocations();
447 
448             int n =
449                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
450             if (n > alllineLocations.size()) {
451                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
452             } else {
453                 lineLocation = (Location) alllineLocations.get(n);
454                 try {
455                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
456                     breakpRequest.putProperty("number", property);
457                     breakpRequest.addThreadFilter(thread);
458                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
459                 } catch ( Exception e1 ) {
460                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
461                     breakpRequest    = null;
462                 }
463             }
464         } catch ( Exception e2 ) {
465             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
466             breakpRequest    = null;
467         }
468 
469         if (breakpRequest == null) {
470             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
471             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
472         }
473 
474         log2("      a breakpoint has been set up");
475         return breakpRequest;
476     }

479     private void getEventSet()
480                  throws JDITestRuntimeException {
481         try {
482 //            log2("       eventSet = eventQueue.remove(waitTime);");
483             eventSet = eventQueue.remove(waitTime);
484             if (eventSet == null) {
485                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
486             }
487 //            log2("       eventIterator = eventSet.eventIterator;");
488             eventIterator = eventSet.eventIterator();
489         } catch ( Exception e ) {
490             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
491         }
492     }

496                  throws JDITestRuntimeException {

499         getEventSet();
500

502             return;
503 
504         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");

509     private ThreadStartRequest setting2ThreadStartRequest( ThreadReference thread,
510                                                            int             suspendPolicy,
511                                                            String          property)
512             throws JDITestRuntimeException {
513         try {
514             ThreadStartRequest tsr = eventRManager.createThreadStartRequest();
515             if (thread != null)
516                 tsr.addThreadFilter(thread);
517             tsr.addCountFilter(1);
518             tsr.setSuspendPolicy(suspendPolicy);
519             tsr.putProperty("number", property);
520             return tsr;
521         } catch ( Exception e ) {
522             log3("ERROR: ATTENTION: Exception within settingThreadStartRequest() : " + e);
523             log3("       ThreadStartRequest HAS NOT BEEN SET UP");
524             throw new JDITestRuntimeException("** FAILURE to set up ThreadStartRequest **");
525         }
