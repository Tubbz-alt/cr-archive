 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * LongType.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that for a mirror of a long value, the casts <BR>
 42  * from Type to LongType and from PrimitiveType to LongType     <BR>
 43  * don't throw ClassCastException.                              <BR>
 44  * <BR>
 45  * The test has three phases and works as follows.              <BR>
 46  * <BR>
 47  * In first phase,                                                      <BR>
 48  * upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 52  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 53  * and waits for the event within the predefined time interval.         <BR>
 54  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 57  * within debuggee's special methodForCommunication().                  <BR>
 58  * <BR>
 59  * In second phase to check the casts,                                  <BR>
 60  * the debugger and the debuggee perform the following.                 <BR>
 61  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * - The debuggee prepares a long field and invokes                     <BR>
 63  *   the methodForCommunication to be suspended and                     <BR>
 64  *   to inform the debugger with the event.                             <BR>
 65  * - Upon getting the BreakpointEvent,                                  <BR>
 66  *   the debugger performs the checks.                                  <BR>
 67  * Note. To inform each other of needed actions, the debugger and       <BR>
 68  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 69  * <BR>
 70  * In third phase when at the end,                                      <BR>
 71  * the debuggee changes the value of the "instruction"                  <BR>
 72  * to inform the debugger of checks finished, and both end.             <BR>
 73  * <BR>
 74  */
 75

 98     public static int run (String argv[], PrintStream out) {
 99 
100         int exitCode = new longtype001().runThis(argv, out);

102         if (exitCode != PASSED) {
103             System.out.println("TEST FAILED");
104         }
105         return testExitCode;
106     }

148     private int runThis (String argv[], PrintStream out) {
149 
150         argsHandler     = new ArgumentHandler(argv);
151         logHandler      = new Log(out, argsHandler);
152         Binder binder   = new Binder(argsHandler, logHandler);
153 
154         waitTime        = argsHandler.getWaitTime() * 60000;
155 
156         try {
157             log2("launching a debuggee :");
158             log2("       " + debuggeeName);
159             if (argsHandler.verbose()) {
160                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
161             } else {
162                 debuggee = binder.bindToDebugee(debuggeeName);
163             }
164             if (debuggee == null) {
165                 log3("ERROR: no debuggee launched");
166                 return FAILED;
167             }
168             log2("debuggee launched");
169         } catch ( Exception e ) {
170             log3("ERROR: Exception : " + e);
171             log2("       test cancelled");
172             return FAILED;
173         }
174 
175         debuggee.redirectOutput(logHandler);
176 
177         vm = debuggee.VM();
178 
179         eventQueue = vm.eventQueue();
180         if (eventQueue == null) {
181             log3("ERROR: eventQueue == null : TEST ABORTED");
182             vm.exit(PASS_BASE);
183             return FAILED;
184         }
185 
186         log2("invocation of the method runTest()");
187         switch (runTest()) {
188 
189             case 0 :  log2("test phase has finished normally");
190                       log2("   waiting for the debuggee to finish ...");
191                       debuggee.waitFor();
192 
193                       log2("......getting the debuggee's exit status");
194                       int status = debuggee.getStatus();
195                       if (status != PASS_BASE) {
196                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
197                               status + " != PASS_BASE");
198                           testExitCode = FAILED;
199                       } else {
200                           log2("......debuggee returned expected exit status: " +
201                               status + " == PASS_BASE");
202                       }
203                       break;
204 
205             default : log3("ERROR: runTest() returned unexpected value");
206 
207             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
208                       log2("......forcing: vm.exit();");
209                       testExitCode = FAILED;
210                       try {
211                           vm.exit(PASS_BASE);
212                       } catch ( Exception e ) {
213                           log3("ERROR: Exception : e");
214                       }
215                       break;
216 
217             case 2 :  log3("test cancelled due to VMDisconnectedException");
218                       log2("......trying: vm.process().destroy();");
219                       testExitCode = FAILED;
220                       try {
221                           Process vmProcess = vm.process();
222                           if (vmProcess != null) {
223                               vmProcess.destroy();
224                           }
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : e");
227                       }
228                       break;
229             }
230 
231         return testExitCode;
232     }

235    /*
236     * Return value: 0 - normal end of the test
237     *               1 - ubnormal end of the test
238     *               2 - VMDisconnectedException while test phase
239     */
240 
241     private int runTest() {
242 
243         try {
244             testRun();
245 
246             log2("waiting for VMDeathEvent");
247             getEventSet();
248             if (eventIterator.nextEvent() instanceof VMDeathEvent)
249                 return 0;
250 
251             log3("ERROR: last event is not the VMDeathEvent");
252             return 1;
253         } catch ( VMDisconnectedException e ) {
254             log3("ERROR: VMDisconnectedException : " + e);
255             return 2;
256         } catch ( Exception e ) {
257             log3("ERROR: Exception : " + e);
258             return 1;
259         }
260 
261     }

263     private void testRun()
264                  throws JDITestRuntimeException, Exception {
265 
266         eventRManager = vm.eventRequestManager();
267 
268         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
269         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
270         cpRequest.addClassFilter(debuggeeName);
271 
272         cpRequest.enable();
273         vm.resume();
274         getEventSet();
275         cpRequest.disable();
276 
277         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
278         debuggeeClass = event.referenceType();
279 
280         if (!debuggeeClass.name().equals(debuggeeName))
281            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
282 
283         log2("      received: ClassPrepareEvent for debuggeeClass");
284 
285         String bPointMethod = "methodForCommunication";
286         String lineForComm  = "lineForComm";
287         BreakpointRequest bpRequest;
288 
289         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
290                                       debuggeeClass,
291                                       bPointMethod, lineForComm, "zero");
292         bpRequest.enable();
293 
294     //------------------------------------------------------  testing section
295 
296         log1("     TESTING BEGINS");
297 
298         for (int i = 0; ; i++) {
299 
300             vm.resume();
301             breakpointForCommunication();
302 
303             int instruction = ((IntegerValue)
304                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
305 
306             if (instruction == 0) {
307                 vm.resume();
308                 break;
309             }
310 
311             log1(":::::: case: # " + i);
312 
313             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
314 
315             log2("......String name = 'ln';");
316             String name = "ln";
317 
318             log2("      getting: Type type = debuggeeClass.getValue(debuggeeClass.fieldByName(name)).type();");
319             Type type = debuggeeClass.getValue(debuggeeClass.fieldByName(name)).type();
320             log2("      PrimitiveType primitiveType = (PrimitiveType) type;");
321             PrimitiveType primitiveType = (PrimitiveType) type;
322 
323             LongType longType = null;
324             try {
325                 log2("      checking up on cast: LongType longType = (LongType) type;");
326                     longType = (LongType) type;
327             } catch ( ClassCastException e ) {
328                     testExitCode = FAILED;
329                     log3("ERROR: ClassCastException");
330             }
331             try {
332                 log2("      checking up on cast: LongType longType = (LongType) primitiveType;");
333                     longType = (LongType) primitiveType;
334             } catch ( ClassCastException e ) {
335                     testExitCode = FAILED;
336                     log3("ERROR: ClassCastException");
337             }
338 
339             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
340         }
341         log1("    TESTING ENDS");
342         return;
343     }

345    /*
346     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
347     *                                             String, String, String)
348     *
349     * It sets up a breakpoint at given line number within a given method in a given class
350     * for a given thread.
351     *
352     * Return value: BreakpointRequest object  in case of success
353     *
354     * JDITestRuntimeException   in case of an Exception thrown within the method
355     */
356 
357     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
358                                                   ReferenceType testedClass,
359                                                   String methodName,
360                                                   String bpLine,
361                                                   String property)
362             throws JDITestRuntimeException {
363 
364         log2("......setting up a breakpoint:");
365         log2("       thread: " + thread + "; class: " + testedClass +
366                         "; method: " + methodName + "; line: " + bpLine);
367 
368         List              alllineLocations = null;
369         Location          lineLocation     = null;
370         BreakpointRequest breakpRequest    = null;
371 
372         try {
373             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
374 
375             alllineLocations = method.allLineLocations();
376 
377             int n =
378                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
379             if (n > alllineLocations.size()) {
380                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
381             } else {
382                 lineLocation = (Location) alllineLocations.get(n);
383                 try {
384                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
385                     breakpRequest.putProperty("number", property);
386                     breakpRequest.addThreadFilter(thread);
387                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
388                 } catch ( Exception e1 ) {
389                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
390                     breakpRequest    = null;
391                 }
392             }
393         } catch ( Exception e2 ) {
394             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
395             breakpRequest    = null;
396         }
397 
398         if (breakpRequest == null) {
399             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
400             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
401         }
402 
403         log2("      a breakpoint has been set up");
404         return breakpRequest;
405     }

408     private void getEventSet()
409                  throws JDITestRuntimeException {
410         try {
411 //            log2("       eventSet = eventQueue.remove(waitTime);");
412             eventSet = eventQueue.remove(waitTime);
413             if (eventSet == null) {
414                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
415             }
416 //            log2("       eventIterator = eventSet.eventIterator;");
417             eventIterator = eventSet.eventIterator();
418         } catch ( Exception e ) {
419             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
420         }
421     }

424     private void breakpointForCommunication()
425                  throws JDITestRuntimeException {
426 
427         log2("breakpointForCommunication");
428         getEventSet();
429 
430         if (eventIterator.nextEvent() instanceof BreakpointEvent)
431             return;
432 
433         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
434     }
