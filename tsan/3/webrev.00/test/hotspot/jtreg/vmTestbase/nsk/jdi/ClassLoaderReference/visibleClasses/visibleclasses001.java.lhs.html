 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ClassLoader.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ClassLoader.visibleClasses()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *   Returns a list of all classes for which this class loader  <BR>
 47  *   has been recorded as the initiating loader in the target VM.<BR>
 48  *   The list contains ReferenceTypes defined directly by       <BR>
 49  *   this loader (as returned by definedClasses()) and          <BR>
 50  *   any types for which loading was delegated by               <BR>
 51  *   this class loader to another class loader.                 <BR>
 52  * <BR>
 53  * The case to check includes a class loader with no            <BR>
 54  * delegated types, that is, the expected returned value should <BR>
 55  * be equal to one returned by ClassLoader.definedClasses(),    <BR>
 56  * except primitive arrays defined by bootstrap class loader    <BR>
 57  * and visible.                                                 <BR>
 58  * <BR>
 59  * The test has three phases and works as follows.              <BR>
 60  * <BR>
 61  * In first phase,                                              <BR>
 62  * upon launching debuggee's VM which will be suspended,                <BR>
 63  * a debugger waits for the VMStartEvent within a predefined            <BR>
 64  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 65  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 66  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 67  * and waits for the event within the predefined time interval.         <BR>
 68  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 69  * Upon getting the ClassPrepareEvent,                                  <BR>
 70  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 71  * within debuggee's special methodForCommunication().                  <BR>
 72  * <BR>
 73  * In second phase to check the assetion,                               <BR>
 74  * the debugger and the debuggee perform the following.                 <BR>
 75  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 76  * - The debuggee prepares a ClassLoader object and invokes             <BR>
 77  *   the methodForCommunication to be suspended and                     <BR>
 78  *   to inform the debugger with the event.                             <BR>
 79  * - Upon getting the BreakpointEvent,                                  <BR>
 80  *   the debugger performs the check.                                   <BR>
 81  * Note. To inform each other of needed actions, the debugger and       <BR>
 82  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 83  * <BR>
 84  * In third phase when at the end,                                      <BR>
 85  * the debuggee changes the value of the "instruction"                  <BR>
 86  * to inform the debugger of checks finished, and both end.             <BR>
 87  * <BR>
 88  */
 89

171     private int runThis (String argv[], PrintStream out) {
172 
173         argsHandler     = new ArgumentHandler(argv);
174         logHandler      = new Log(out, argsHandler);
175         Binder binder   = new Binder(argsHandler, logHandler);
176 
177         waitTime        = argsHandler.getWaitTime() * 60000;
178 
179         try {
180             log2("launching a debuggee :");
181             log2("       " + debuggeeName);
182             if (argsHandler.verbose()) {
183                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
184             } else {
185                 debuggee = binder.bindToDebugee(debuggeeName);
186             }
187             if (debuggee == null) {
188                 log3("ERROR: no debuggee launched");
189                 return FAILED;
190             }
191             log2("debuggee launched");
192         } catch ( Exception e ) {
193             log3("ERROR: Exception : " + e);
194             log2("       test cancelled");
195             return FAILED;
196         }
197 
198         debuggee.redirectOutput(logHandler);
199 
200         vm = debuggee.VM();
201 
202         eventQueue = vm.eventQueue();
203         if (eventQueue == null) {
204             log3("ERROR: eventQueue == null : TEST ABORTED");
205             vm.exit(PASS_BASE);
206             return FAILED;
207         }
208 
209         log2("invocation of the method runTest()");
210         switch (runTest()) {
211 
212             case 0 :  log2("test phase has finished normally");
213                       log2("   waiting for the debuggee to finish ...");
214                       debuggee.waitFor();
215 
216                       log2("......getting the debuggee's exit status");
217                       int status = debuggee.getStatus();
218                       if (status != PASS_BASE) {
219                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
220                               status + " != PASS_BASE");
221                           testExitCode = FAILED;
222                       } else {
223                           log2("......debuggee returned expected exit status: " +
224                               status + " == PASS_BASE");
225                       }
226                       break;
227 
228             default : log3("ERROR: runTest() returned unexpected value");
229 
230             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
231                       log2("......forcing: vm.exit();");
232                       testExitCode = FAILED;
233                       try {
234                           vm.exit(PASS_BASE);
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : e");
237                       }
238                       break;
239 
240             case 2 :  log3("test cancelled due to VMDisconnectedException");
241                       log2("......trying: vm.process().destroy();");
242                       testExitCode = FAILED;
243                       try {
244                           Process vmProcess = vm.process();
245                           if (vmProcess != null) {
246                               vmProcess.destroy();
247                           }
248                       } catch ( Exception e ) {
249                           log3("ERROR: Exception : e");
250                       }
251                       break;
252             }
253 
254         return testExitCode;
255     }

258    /*
259     * Return value: 0 - normal end of the test
260     *               1 - ubnormal end of the test
261     *               2 - VMDisconnectedException while test phase
262     */
263 
264     private int runTest() {
265 
266         try {
267             testRun();
268 
269             log2("waiting for VMDeathEvent");
270             getEventSet();
271             if (eventIterator.nextEvent() instanceof VMDeathEvent)
272                 return 0;
273 
274             log3("ERROR: last event is not the VMDeathEvent");
275             return 1;
276         } catch ( VMDisconnectedException e ) {
277             log3("ERROR: VMDisconnectedException : " + e);
278             return 2;
279         } catch ( Exception e ) {
280             log3("ERROR: Exception : " + e);
281             return 1;
282         }
283 
284     }

286     private void testRun()
287                  throws JDITestRuntimeException, Exception {
288 
289         eventRManager = vm.eventRequestManager();
290 
291         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
292         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
293         cpRequest.addClassFilter(debuggeeName);
294 
295         cpRequest.enable();
296         vm.resume();
297         getEventSet();
298         cpRequest.disable();
299 
300         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
301         debuggeeClass = event.referenceType();
302 
303         if (!debuggeeClass.name().equals(debuggeeName))
304            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
305 
306         log2("      received: ClassPrepareEvent for debuggeeClass");
307 
308         String bPointMethod = "methodForCommunication";
309         String lineForComm  = "lineForComm";
310         BreakpointRequest bpRequest;
311 
312         bpRequest = settingBreakpoint(threadByName("main"),
313                                       debuggeeClass,
314                                       bPointMethod, lineForComm, "zero");
315         bpRequest.enable();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         for (int i = 0; ; i++) {
322 
323             vm.resume();
324             breakpointForCommunication();
325 
326             int instruction = ((IntegerValue)
327                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
328 
329             if (instruction == 0) {
330                 vm.resume();
331                 break;
332             }
333 
334             log1(":::::: case: # 0");
335 
336             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
337 
338             log2("......getting: List classes = vm.allClasses();");
339             List<ReferenceType> classes = vm.allClasses();
340             log2("               classes.size() == " + classes.size());
341 
342             vm.resume();
343             breakpointForCommunication();
344 
345             log1(":::::: case: # 1");
346 
347             String ClassLoaderObjName = "classLoader";
348 
349             log2("......getting: Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));");
350             Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));
351 
352             log2("......getting: ClassLoaderReference clRef = (ClassLoaderReference) val;");
353             ClassLoaderReference clRef = (ClassLoaderReference) val;
354 
355             log2("......getting: List definedClasses = clRef.definedClasses();");
356             List<ReferenceType> definedClasses = clRef.definedClasses();
357 
358             log2("......getting: List visibleClasses = clRef.visibleClasses();");
359             List<ReferenceType> visibleClasses = clRef.visibleClasses();
360 
361             log2("......checking up on: visibleClasses.size() == definedClasses.size()");
362 
363             if (visibleClasses.size() != definedClasses.size()) {
364                 log2("     : visibleClasses.size() != definedClasses.size()");
365                 log2("     : definedClasses.size() == " + definedClasses.size());
366                 log2("     : visibleClasses.size() == " + visibleClasses.size());
367 
368                 for (ReferenceType vcl : visibleClasses) {
369 
370                     int vclIndex = visibleClasses.indexOf(vcl);
371                     String vclName = vcl.name();
372 
373                     if (primitiveArraysNamesPatterns.stream().anyMatch(vcl.name()::matches)) {
374                         log2("     : visibleClasses[" + vclIndex + "].name() == " + vclName +
375                             " Correct - primitive arrays are visible for class loader");
376                     } else {
377                         log3("     : visibleClasses[" + vclIndex + "].name() == " + vclName);
378 
379                         classes.stream().filter(cl -> cl.name().equals(vclName))
380                                         .forEach(cl -> log3("     :  List classes contains an object with the name: " + cl.name()));
381 
382                         testExitCode = FAILED;
383                     }
384                 }
385             }
386 
387             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
388         }
389         log1("    TESTING ENDS");
390         return;
391     }

393     private ThreadReference threadByName(String name)
394                  throws JDITestRuntimeException {
395 
396         List         all = vm.allThreads();
397         ListIterator li  = all.listIterator();
398 
399         for (; li.hasNext(); ) {
400             ThreadReference thread = (ThreadReference) li.next();
401             if (thread.name().equals(name))
402                 return thread;
403         }
404         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
405     }

408    /*
409     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
410     *                                             String, String, String)
411     *
412     * It sets up a breakpoint at given line number within a given method in a given class
413     * for a given thread.
414     *
415     * Return value: BreakpointRequest object  in case of success
416     *
417     * JDITestRuntimeException   in case of an Exception thrown within the method
418     */
419 
420     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
421                                                   ReferenceType testedClass,
422                                                   String methodName,
423                                                   String bpLine,
424                                                   String property)
425             throws JDITestRuntimeException {
426 
427         log2("......setting up a breakpoint:");
428         log2("       thread: " + thread + "; class: " + testedClass +
429                         "; method: " + methodName + "; line: " + bpLine);
430 
431         List              alllineLocations = null;
432         Location          lineLocation     = null;
433         BreakpointRequest breakpRequest    = null;
434 
435         try {
436             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
437 
438             alllineLocations = method.allLineLocations();
439 
440             int n =
441                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
442             if (n > alllineLocations.size()) {
443                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
444             } else {
445                 lineLocation = (Location) alllineLocations.get(n);
446                 try {
447                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
448                     breakpRequest.putProperty("number", property);
449                     breakpRequest.addThreadFilter(thread);
450                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
451                 } catch ( Exception e1 ) {
452                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
453                     breakpRequest    = null;
454                 }
455             }
456         } catch ( Exception e2 ) {
457             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
458             breakpRequest    = null;
459         }
460 
461         if (breakpRequest == null) {
462             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
463             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
464         }
465 
466         log2("      a breakpoint has been set up");
467         return breakpRequest;
468     }

471     private void getEventSet()
472                  throws JDITestRuntimeException {
473         try {
474 //            log2("       eventSet = eventQueue.remove(waitTime);");
475             eventSet = eventQueue.remove(waitTime);
476             if (eventSet == null) {
477                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
478             }
479 //            log2("       eventIterator = eventSet.eventIterator;");
480             eventIterator = eventSet.eventIterator();
481         } catch ( Exception e ) {
482             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
483         }
484     }

487     private void breakpointForCommunication()
488                  throws JDITestRuntimeException {
489 
490         log2("breakpointForCommunication");
491         getEventSet();
492 
493         if (eventIterator.nextEvent() instanceof BreakpointEvent)
494             return;
495 
496         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
497     }
