 37 /* scaffold objects */
 38 static jvmtiEnv *jvmti = NULL;
 39 static jlong timeout = 0;
 40 static jrawMonitorID syncLock = NULL;
 41 
 42 static int methodLoadCount = 0;
 43 static int methodUnloadCount = 0;
 44 
 45 #define NAME_LENGTH 50
 46 const void *plist = NULL;
 47 static volatile int callbacksEnabled = NSK_TRUE;
 48 
 49 typedef struct nsk_jvmti_CompiledMethodIDStruct {
 50     jmethodID method;
 51     const void* code_addr;
 52     char name[NAME_LENGTH];
 53 } nsk_jvmti_CompiledMethod;

 58 /* callbacks */
 59 void JNICALL
 60 cbCompiledMethodLoad(jvmtiEnv *jvmti_env, jmethodID method, jint code_size,
 61                 const void* code_addr, jint map_length,
 62                 const jvmtiAddrLocationMap* map, const void* compile_info) {
 63     char *name;
 64     char *sign;
 65     char *genc;
 66 
 67     jvmti->RawMonitorEnter(syncLock);
 68     if (!callbacksEnabled) {
 69         jvmti->RawMonitorExit(syncLock);
 70         return;
 71     }
 72 
 73     if (!NSK_JVMTI_VERIFY(jvmti_env->GetMethodName(method, &name, &sign, &genc))) {
 74         nsk_jvmti_setFailStatus();
 75         jvmti->RawMonitorExit(syncLock);
 76         return;
 77     }
 78 
 79     if (!strncmp(name,"javaMethod", 8)) {
 80         nsk_jvmti_CompiledMethod *rec =
 81             (nsk_jvmti_CompiledMethod *)malloc(sizeof(nsk_jvmti_CompiledMethod));
 82 
 83         rec->method = method;
 84         rec->code_addr = code_addr;
 85         strncpy(rec->name, name, NAME_LENGTH);
 86         rec->name[NAME_LENGTH - 1] = '\0';
 87 
 88         if (!NSK_VERIFY(nsk_list_add(plist, rec))) {
 89             nsk_jvmti_setFailStatus();
 90             free((void *)rec);
 91         } else {
 92             NSK_DISPLAY0(">>>JVMTI_EVENT_COMPILED_METHOD_LOAD received for\n");
 93             NSK_DISPLAY1("\t\tmethod: %s\n", rec->name);
 94 
 95             if (!NSK_JVMTI_VERIFY(jvmti->RawMonitorEnter(syncLock)))
 96                 nsk_jvmti_setFailStatus();
 97 
 98             methodLoadCount++;
 99 
100             if (!NSK_JVMTI_VERIFY(jvmti->RawMonitorExit(syncLock)))
101                 nsk_jvmti_setFailStatus();
102 
103         }
104     }
105 
106     if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char*)name))) {
107         nsk_jvmti_setFailStatus();
108     }
109     if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char*)sign))) {
110         nsk_jvmti_setFailStatus();
111     }
112     if (genc != NULL)
113         if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char*)genc))) {
114             nsk_jvmti_setFailStatus();
115         }
116 
117     jvmti->RawMonitorExit(syncLock);
118 
119 }
120 
121 void JNICALL
122 cbCompiledMethodUnload(jvmtiEnv *jvmti_env, jmethodID method,
123                 const void* code_addr) {
124 
125     nsk_jvmti_CompiledMethod *rec;
126 
127     jvmti->RawMonitorEnter(syncLock);
128     if (!callbacksEnabled) {
129         jvmti->RawMonitorExit(syncLock);
130         return;
131     }
132     int count = nsk_list_getCount(plist);
133 
134     int i;
135 
136     for (i = 0; i < count; i ++) {
137         rec = (nsk_jvmti_CompiledMethod *)nsk_list_get(plist, i);
138         if ((rec->code_addr == code_addr) && (rec->method == method)) {
139             NSK_DISPLAY0(">>>JVMTI_EVENT_COMPILED_METHOD_UNLOAD received for\n");
140             NSK_DISPLAY1("\t\tmethod: %s\n", rec->name);
141 



142             methodUnloadCount++;
143 



144             free(rec);
145             nsk_list_remove(plist, i);
146             jvmti->RawMonitorExit(syncLock);
147             return;
148         }
149 
150     }
151     jvmti->RawMonitorExit(syncLock);
152 }

156 static int
157 enableEvent(jvmtiEventMode enable, jvmtiEvent event) {
158     if (!NSK_JVMTI_VERIFY(jvmti->SetEventNotificationMode(enable, event, NULL))) {
159         nsk_jvmti_setFailStatus();
160         return NSK_FALSE;
161     }
162 
163     return NSK_TRUE;
164 }
165 
166 int checkEvents() {
167 
168     int result = methodUnloadCount <= methodLoadCount;
169 
170     if (result) {
171         NSK_DISPLAY0("Received correct number of events:\n");
172         NSK_DISPLAY1("\t\tCOMPILED_METHOD_LOAD events number = %d\n",
173                                 methodLoadCount);
174         NSK_DISPLAY1("\t\tCOMPILED_METHOD_UNLOAD events number = %d\n",
175                                 methodUnloadCount);
176     } else {
177         NSK_COMPLAIN0("Received incorrect number of events:\n");
178         NSK_COMPLAIN1("\t\tCOMPILED_METHOD_LOAD events number = %d\n",
179                                 methodLoadCount);
180         NSK_COMPLAIN1("\t\tCOMPILED_METHOD_UNLOAD events number = %d\n",
181                                 methodUnloadCount);
182     }
183 
184     return result;
185 }

189 static int
190 setCallBacks() {
191 
192     jvmtiEventCallbacks eventCallbacks;
193 
194     memset(&eventCallbacks, 0, sizeof(eventCallbacks));
195 
196     eventCallbacks.CompiledMethodLoad        = cbCompiledMethodLoad;
197     eventCallbacks.CompiledMethodUnload      = cbCompiledMethodUnload;
198 
199     return NSK_JVMTI_VERIFY(jvmti->SetEventCallbacks(&eventCallbacks, sizeof(eventCallbacks)));
200 }

204 /** Agent algorithm. */
205 static void JNICALL
206 agentProc(jvmtiEnv* jvmti, JNIEnv* agentJNI, void* arg) {
207 
208     int i;
209 
210     int attempts = nsk_jvmti_findOptionIntValue("attempts", 1);
211 
212     for (i = 0; i < attempts; i++) {
213 
214         if (!nsk_jvmti_waitForSync(timeout))
215             return;
216 
217         if (!checkEvents())
218             nsk_jvmti_setFailStatus();
219 
220         NSK_DISPLAY0("Let debuggee to continue\n");
221         if (!nsk_jvmti_resumeSync())
222             return;
223     }
224 
225     jvmti->RawMonitorEnter(syncLock);
226     callbacksEnabled = NSK_FALSE;
227 
228     {
229         int count = nsk_list_getCount(plist);
230 
231         while (count > 0) {
232             free((void *)nsk_list_get(plist, 0));
233             nsk_list_remove(plist, 0);
234             count = nsk_list_getCount(plist);
235         }
236 
237     }
238 
239     jvmti->RawMonitorExit(syncLock);
240 
241     if (!NSK_JVMTI_VERIFY(jvmti->DestroyRawMonitor(syncLock)))
242         nsk_jvmti_setFailStatus();
243 
244 }

248 /** Agent library initialization. */
249 #ifdef STATIC_BUILD
250 JNIEXPORT jint JNICALL Agent_OnLoad_em07t002(JavaVM *jvm, char *options, void *reserved) {
251     return Agent_Initialize(jvm, options, reserved);
252 }
253 JNIEXPORT jint JNICALL Agent_OnAttach_em07t002(JavaVM *jvm, char *options, void *reserved) {
254     return Agent_Initialize(jvm, options, reserved);
255 }
256 JNIEXPORT jint JNI_OnLoad_em07t002(JavaVM *jvm, char *options, void *reserved) {
257     return JNI_VERSION_1_8;
258 }
259 #endif
260 jint Agent_Initialize(JavaVM *jvm, char *options, void *reserved) {
261 
262     if (!NSK_VERIFY(nsk_jvmti_parseOptions(options)))
263         return JNI_ERR;
264 
265     timeout = nsk_jvmti_getWaitTime() * 60 * 1000;
266 
267     if (!NSK_VERIFY((jvmti = nsk_jvmti_createJVMTIEnv(jvm, reserved)) != NULL))
268         return JNI_ERR;
269 
270     if (!NSK_JVMTI_VERIFY(jvmti->CreateRawMonitor("_syncLock", &syncLock))) {
271         nsk_jvmti_setFailStatus();
272         return JNI_ERR;
273     }
274 
275     if (!NSK_VERIFY((plist = (const void *)nsk_list_create()) != NULL))
276         return JNI_ERR;
277 
278     {
279         jvmtiCapabilities caps;
280         memset(&caps, 0, sizeof(caps));
281 
282         caps.can_generate_compiled_method_load_events = 1;
283         if (!NSK_JVMTI_VERIFY(jvmti->AddCapabilities(&caps)))
284             return JNI_ERR;
285     }
286 
287     if (!setCallBacks()) {
288         return JNI_ERR;
289     }
290 
291     if (!enableEvent(JVMTI_ENABLE, JVMTI_EVENT_COMPILED_METHOD_LOAD) ||
292                 !enableEvent(JVMTI_ENABLE, JVMTI_EVENT_COMPILED_METHOD_UNLOAD)) {
293         return JNI_ERR;
294     }
295 
296     if (!NSK_VERIFY(nsk_jvmti_setAgentProc(agentProc, NULL)))
297         return JNI_ERR;
298 
299     return JNI_OK;
300 }
