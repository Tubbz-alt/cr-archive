 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.getValue()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * Case for testing includes fields mirroring primitive type values.    <BR>
 45  * The test checks that Values returned by the method,                  <BR>
 46  * are casted to PrimitiveValue(s).                                     <BR>
 47  * <BR>
 48  * The test works as follows.                                           <BR>
 49  * Upon launching debuggee's VM which will be suspended,                <BR>
 50  * a debugger waits for the VMStartEvent within a predefined            <BR>
 51  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 52  * Upon getting the VMStartEvent, it makes the request                  <BR>
 53  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 54  * resumes the VM, and waits for the event within the predefined        <BR>
 55  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 56  * Upon getting the ClassPrepareEvent,                                  <BR>
 57  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 58  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 59  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 60  * to be suspended and to inform the debugger with the event.           <BR>
 61  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 62  * At the end, the debuggee changes the value of the "instruction"      <BR>
 63  * to inform the debugger of checks finished, and both end.             <BR>
 64  */

 89 
 90         int exitCode = new getvalue001().runThis(argv, out);

 92         if (exitCode != PASSED) {

 94         }
 95         return testExitCode;

146     private int runThis (String argv[], PrintStream out) {
147 
148         argsHandler     = new ArgumentHandler(argv);
149         logHandler      = new Log(out, argsHandler);
150         Binder binder   = new Binder(argsHandler, logHandler);
151 
152         waitTime        = argsHandler.getWaitTime() * 60000;
153 
154         try {
155             log2("launching a debuggee :");
156             log2("       " + debuggeeName);
157             if (argsHandler.verbose()) {
158                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
159             } else {
160                 debuggee = binder.bindToDebugee(debuggeeName);
161             }
162             if (debuggee == null) {
163                 log3("ERROR: no debuggee launched");
164                 return FAILED;
165             }
166             log2("debuggee launched");
167         } catch ( Exception e ) {
168             log3("ERROR: Exception : " + e);
169             log2("       test cancelled");
170             return FAILED;
171         }
172 
173         debuggee.redirectOutput(logHandler);
174 
175         vm = debuggee.VM();
176 
177         eventQueue = vm.eventQueue();
178         if (eventQueue == null) {
179             log3("ERROR: eventQueue == null : TEST ABORTED");
180             vm.exit(PASS_BASE);
181             return FAILED;
182         }
183 
184         log2("invocation of the method runTest()");
185         switch (runTest()) {
186 
187             case 0 :  log2("test phase has finished normally");
188                       log2("   waiting for the debuggee to finish ...");
189                       debuggee.waitFor();
190 
191                       log2("......getting the debuggee's exit status");
192                       int status = debuggee.getStatus();
193                       if (status != PASS_BASE) {
194                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
195                               status + " != PASS_BASE");
196                           testExitCode = FAILED;
197                       } else {
198                           log2("......debuggee returned expected exit status: " +
199                               status + " == PASS_BASE");
200                       }
201                       break;
202 
203             default : log3("ERROR: runTest() returned unexpected value");
204 
205             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
206                       log2("......forcing: vm.exit();");
207                       testExitCode = FAILED;
208                       try {
209                           vm.exit(PASS_BASE);
210                       } catch ( Exception e ) {
211                           log3("ERROR: Exception : e");
212                       }
213                       break;
214 
215             case 2 :  log3("test cancelled due to VMDisconnectedException");
216                       log2("......trying: vm.process().destroy();");
217                       testExitCode = FAILED;
218                       try {
219                           Process vmProcess = vm.process();
220                           if (vmProcess != null) {
221                               vmProcess.destroy();
222                           }
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227             }
228 
229         return testExitCode;
230     }

234     * Return value: 0 - normal end of the test
235     *               1 - ubnormal end of the test
236     *               2 - VMDisconnectedException while test phase
237     */

239     private int runTest() {
240 
241         try {
242             testRun();
243 
244             log2("waiting for VMDeathEvent");
245             getEventSet();
246             if (eventIterator.nextEvent() instanceof VMDeathEvent)
247                 return 0;
248 
249             log3("ERROR: last event is not the VMDeathEvent");
250             return 1;
251         } catch ( VMDisconnectedException e ) {
252             log3("ERROR: VMDisconnectedException : " + e);
253             return 2;
254         } catch ( Exception e ) {
255             log3("ERROR: Exception : " + e);
256             return 1;
257         }
258 
259     }

261     private void testRun()
262                  throws JDITestRuntimeException, Exception {
263 
264         eventRManager = vm.eventRequestManager();
265 
266         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
267         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
268         cpRequest.addClassFilter(debuggeeName);
269 
270         cpRequest.enable();
271         vm.resume();
272         getEventSet();
273         cpRequest.disable();
274 
275         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
276         debuggeeClass = event.referenceType();
277 
278         if (!debuggeeClass.name().equals(debuggeeName))
279            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
280 
281         log2("      received: ClassPrepareEvent for debuggeeClass");
282 
283         String bPointMethod = "methodForCommunication";
284         String lineForComm  = "lineForComm";
285         BreakpointRequest bpRequest;
286 
287         try {
288             bpRequest = settingBreakpoint(threadByName("main"),
289                                           debuggeeClass,
290                                           bPointMethod, lineForComm, "zero");
291         } catch ( Exception e ) {
292             throw e;
293         }
294         bpRequest.enable();
295 
296     //------------------------------------------------------  testing section
297 
298         log1("     TESTING BEGINS");
299 
300         for (int i = 0; ; i++) {
301 
302             vm.resume();
303             breakpointForCommunication();
304 
305             int instruction = ((IntegerValue)
306                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
307 
308             if (instruction == 0) {
309                 vm.resume();
310                 break;
311             }
312 
313             log1("  new check: # " + i);
314 
315             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
316 
317             log2("      name of tested class : " + testedClassName);
318             log2("      getting: List classes = vm.classByName(testedClassName);");
319             List classes = vm.classesByName(testedClassName);
320             if (classes.size() != 1) {
321                 log3("ERROR: classes.size() != 1 : " + classes.size());
322                 testExitCode = FAILED;
323                 break;
324             }
325             log2("      getting: ReferenceType testedClass = (ReferenceType) classes.get(0);");
326             ReferenceType testedClass = (ReferenceType) classes.get(0);
327 
328             log2("------case for testing: primitive values");
329 
330             String fieldNames0[] = {
331                        "bl0",
332                        "bt0",
333                        "ch0",
334                        "db0",
335                        "fl0",
336                        "in0",
337                        "ln0",
338                        "sh0"      };
339 
340             for (int i1 = 0; i1 < 8; i1++) {
341 
342                 log2("......field tested : " + fieldNames0[i1]);
343                 log2("      getting: Field field = testedClass.fieldByName(fieldNames0[i1]);");
344                 Field field = testedClass.fieldByName(fieldNames0[i1]);
345                 if (field == null) {
346                     log3("ERROR: field == null");
347                     testExitCode = FAILED;
348                     continue;
349                 }
350                 log2("      getting: Value value = testedClass.getValue(field);");
351                 Value value = testedClass.getValue(field);
352                 if (value == null) {
353                     log3("ERROR: value == null");
354                     testExitCode = FAILED;
355                     continue;
356                 }
357                 try {
358                     log2("      getting Class cast: PrimitiveValue primitiveValue = (PrimitiveValue) value;");
359                     PrimitiveValue primitiveValue = (PrimitiveValue) value;
360                 } catch ( ClassCastException e ) {
361                     log3("ERROR: ClassCastException");
362                     testExitCode = FAILED;
363                 }
364             }
365             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
366         }
367         log1("    TESTING ENDS");
368         return;
369     }

371     private ThreadReference threadByName(String name)
372                  throws JDITestRuntimeException {
373 
374         List         all = vm.allThreads();
375         ListIterator li  = all.listIterator();
376 
377         for (; li.hasNext(); ) {
378             ThreadReference thread = (ThreadReference) li.next();
379             if (thread.name().equals(name))
380                 return thread;
381         }
382         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
383     }

385    /*
386     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
387     *                                             String, String, String)
388     *
389     * It sets up a breakpoint at given line number within a given method in a given class
390     * for a given thread.
391     *
392     * Return value: BreakpointRequest object  in case of success
393     *
394     * JDITestRuntimeException   in case of an Exception thrown within the method
395     */
396 
397     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
398                                                   ReferenceType testedClass,
399                                                   String methodName,
400                                                   String bpLine,
401                                                   String property)
402             throws JDITestRuntimeException {
403 
404         log2("......setting up a breakpoint:");
405         log2("       thread: " + thread + "; class: " + testedClass +
406                         "; method: " + methodName + "; line: " + bpLine);
407 
408         List              alllineLocations = null;
409         Location          lineLocation     = null;
410         BreakpointRequest breakpRequest    = null;
411 
412         try {
413             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
414 
415             alllineLocations = method.allLineLocations();
416 
417             int n =
418                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
419             if (n > alllineLocations.size()) {
420                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
421             } else {
422                 lineLocation = (Location) alllineLocations.get(n);
423                 try {
424                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
425                     breakpRequest.putProperty("number", property);
426                     breakpRequest.addThreadFilter(thread);
427                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
428                 } catch ( Exception e1 ) {
429                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
430                     breakpRequest    = null;
431                 }
432             }
433         } catch ( Exception e2 ) {
434             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
435             breakpRequest    = null;
436         }
437 
438         if (breakpRequest == null) {
439             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
440             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
441         }
442 
443         log2("      a breakpoint has been set up");
444         return breakpRequest;
445     }

448     private void getEventSet()
449                  throws JDITestRuntimeException {
450         try {
451 //            log2("       eventSet = eventQueue.remove(waitTime);");
452             eventSet = eventQueue.remove(waitTime);
453             if (eventSet == null) {
454                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
455             }
456 //            log2("       eventIterator = eventSet.eventIterator;");
457             eventIterator = eventSet.eventIterator();
458         } catch ( Exception e ) {
459             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
460         }
461     }

464     private void breakpointForCommunication()
465                  throws JDITestRuntimeException {
466 
467         log2("breakpointForCommunication");
468         getEventSet();
469 
470         if (eventIterator.nextEvent() instanceof BreakpointEvent)
471             return;
472 
473         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
474     }
