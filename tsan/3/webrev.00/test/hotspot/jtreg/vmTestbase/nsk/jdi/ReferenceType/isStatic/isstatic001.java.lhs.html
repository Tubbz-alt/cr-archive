 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.isStatic()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Cases for testing include :                                  <BR>
 46  * - nested and outer ClassTypes and InterfaceTypes,            <BR>
 47  *   and nested ArrayTypes;                                     <BR>
 48  * - arrays of primitive values;                                <BR>
 49  * - primitive classes.                                         <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assetion,                               <BR>
 66  * the debugger and the debuggee perform the following loop.            <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee prepares new check case and invokes                   <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent,                                  <BR>
 72  *   the debugger performs the check case required.                     <BR>
 73  * Note. To inform each other of needed actions, the debugger and       <BR>
 74  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 75  * <BR>
 76  * In third phase when at the end,                                      <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */

162     private int runThis (String argv[], PrintStream out) {
163 
164         argsHandler     = new ArgumentHandler(argv);
165         logHandler      = new Log(out, argsHandler);
166         Binder binder   = new Binder(argsHandler, logHandler);
167 
168         waitTime        = argsHandler.getWaitTime() * 60000;
169 
170         try {
171             log2("launching a debuggee :");
172             log2("       " + debuggeeName);
173             if (argsHandler.verbose()) {
174                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
175             } else {
176                 debuggee = binder.bindToDebugee(debuggeeName);
177             }
178             if (debuggee == null) {
179                 log3("ERROR: no debuggee launched");
180                 return FAILED;
181             }
182             log2("debuggee launched");
183         } catch ( Exception e ) {
184             log3("ERROR: Exception : " + e);
185             log2("       test cancelled");
186             return FAILED;
187         }
188 
189         debuggee.redirectOutput(logHandler);
190 
191         vm = debuggee.VM();
192 
193         eventQueue = vm.eventQueue();
194         if (eventQueue == null) {
195             log3("ERROR: eventQueue == null : TEST ABORTED");
196             vm.exit(PASS_BASE);
197             return FAILED;
198         }
199 
200         log2("invocation of the method runTest()");
201         switch (runTest()) {
202 
203             case 0 :  log2("test phase has finished normally");
204                       log2("   waiting for the debuggee to finish ...");
205                       debuggee.waitFor();
206 
207                       log2("......getting the debuggee's exit status");
208                       int status = debuggee.getStatus();
209                       if (status != PASS_BASE) {
210                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
211                               status + " != PASS_BASE");
212                           testExitCode = FAILED;
213                       } else {
214                           log2("......debuggee returned expected exit status: " +
215                               status + " == PASS_BASE");
216                       }
217                       break;
218 
219             default : log3("ERROR: runTest() returned unexpected value");
220 
221             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
222                       log2("......forcing: vm.exit();");
223                       testExitCode = FAILED;
224                       try {
225                           vm.exit(PASS_BASE);
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : e");
228                       }
229                       break;
230 
231             case 2 :  log3("test cancelled due to VMDisconnectedException");
232                       log2("......trying: vm.process().destroy();");
233                       testExitCode = FAILED;
234                       try {
235                           Process vmProcess = vm.process();
236                           if (vmProcess != null) {
237                               vmProcess.destroy();
238                           }
239                       } catch ( Exception e ) {
240                           log3("ERROR: Exception : e");
241                       }
242                       break;
243             }
244 
245         return testExitCode;
246     }

250     * Return value: 0 - normal end of the test
251     *               1 - ubnormal end of the test
252     *               2 - VMDisconnectedException while test phase
253     */

255     private int runTest() {
256 
257         try {
258             testRun();
259 
260             log2("waiting for VMDeathEvent");
261             getEventSet();
262             if (eventIterator.nextEvent() instanceof VMDeathEvent)
263                 return 0;
264 
265             log3("ERROR: last event is not the VMDeathEvent");
266             return 1;
267         } catch ( VMDisconnectedException e ) {
268             log3("ERROR: VMDisconnectedException : " + e);
269             return 2;
270         } catch ( Exception e ) {
271             log3("ERROR: Exception : " + e);
272             return 1;
273         }
274 
275     }

277     private void testRun()
278                  throws JDITestRuntimeException, Exception {
279 
280         eventRManager = vm.eventRequestManager();
281 
282         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
283         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
284         cpRequest.addClassFilter(debuggeeName);
285 
286         cpRequest.enable();
287         vm.resume();
288         getEventSet();
289         cpRequest.disable();
290 
291         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
292         debuggeeClass = event.referenceType();
293 
294         if (!debuggeeClass.name().equals(debuggeeName))
295            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
296 
297         log2("      received: ClassPrepareEvent for debuggeeClass");
298 
299         String bPointMethod = "methodForCommunication";
300         String lineForComm  = "lineForComm";
301         BreakpointRequest bpRequest;
302 
303         bpRequest = settingBreakpoint(threadByName("main"),
304                                       debuggeeClass,
305                                       bPointMethod, lineForComm, "zero");
306         bpRequest.enable();
307 
308     //------------------------------------------------------  testing section
309 
310         log1("     TESTING BEGINS");
311 
312         for (int i = 0; ; i++) {
313 
314             vm.resume();
315             breakpointForCommunication();
316 
317             int instruction = ((IntegerValue)
318                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
319 
320             if (instruction == 0) {
321                 vm.resume();
322                 break;
323             }
324 
325             log1(":::::: case: # " + i);
326 
327             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
328 
329            String names1[] = {
330 
331                       ".isstatic001aTestClass$NestedClass",         "f",
332                       ".isstatic001aTestClass$NestedClass[]",       "f",
333                       ".isstatic001aTestClass$NestedIface",         "t",
334                       ".isstatic001aTestClass$StaticNestedClass",   "t",
335                       ".isstatic001aOuterClass",                    "f",
336                       ".isstatic001aOuterClass[]",                  "f",
337                       ".isstatic001aOuterIface",                    "f",
338                       ".isstatic001aOuterIface[]",                  "f",
339                       ".isstatic001aOuterIface$InnerIface",         "t",
340                       ".isstatic001aOuterIface$StaticInnerIface",   "t"
341 
342                       };
343 
344             String names2[] = {
345 
346                       "boolean[][][][]", "f",
347                       "byte[][][][]",    "f",
348                       "char[][][][]",    "f",
349                       "double[][][][]",  "f",
350                       "float[][][][]",   "f",
351                       "int[][][][]",     "f",
352                       "long[][][][]",    "f",
353                       "short[][][][]",   "f"
354 
355                       };
356 
357             String names3[] = {
358 
359                       "java.lang.Boolean",   "f",
360                       "java.lang.Byte",      "f",
361                       "java.lang.Character", "f",
362                       "java.lang.Double",    "f",
363                       "java.lang.Float",     "f",
364                       "java.lang.Integer",   "f",
365                       "java.lang.Long",      "f",
366                       "java.lang.Short",     "f"
367 
368                       };
369 
370             log2("----- Cases for testing: ReferenceTypes");
371             for (int i1 = 0; i1 < names1.length; i1 += 2) {
372 
373                 String typeForCheck = mName + names1[i1];
374                 log2("......typeForCheck: " + typeForCheck);
375 
376                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
377                 List classList = vm.classesByName(typeForCheck);
378                 if (classList.size() != 1) {
379                     log3("ERROR: classList.size() != 1  : " + classList.size());
380                     testExitCode = FAILED;
381                     continue;
382                 }
383 
384                 log2("      getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
385                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
386 
387                 log2("      getting: boolean isStatic = referenceType.isStatic();");
388                 boolean isStatic = referenceType.isStatic();
389 
390                 if ( names1[i1+1].equals("t") ) {
391                     log2("      expected value of isStatic is 'true'");
392                     if (!isStatic) {
393                         log3("ERROR: isStatic != true for: " + typeForCheck);
394                         testExitCode = FAILED;
395                     }
396                 } else {
397                     log2("      expected value of isStatic is 'false'");
398                     if (isStatic) {
399                         log3("ERROR: isStatic != false for: " + typeForCheck);
400                         testExitCode = FAILED;
401                     }
402                 }
403             }
404 
405             log2("----- Cases for testing: primitive type arrays");
406             for (int i1 = 0; i1 < names2.length; i1 += 2) {
407 
408                 String typeForCheck = names2[i1];
409                 log2("......typeForCheck: " + typeForCheck);
410 
411                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
412                 List classList = vm.classesByName(typeForCheck);
413                 if (classList.size() != 1) {
414                     log3("ERROR: classList.size() != 1  : " + classList.size());
415                     testExitCode = FAILED;
416                     continue;
417                 }
418 
419                 log2("       getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
420                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
421 
422                 log2("       getting: boolean isStatic = referenceType.isStatic();");
423                 boolean isStatic = referenceType.isStatic();
424 
425                 log2("      expected value of isStatic is 'false'");
426                 if (isStatic) {
427                     log3("ERROR: isStatic != false for: " + typeForCheck);
428                     testExitCode = FAILED;
429                 }
430 
431             }
432 
433             log2("----- Cases for testing: primitive classes");
434             for (int i1 = 0; i1 < names3.length; i1 += 2) {
435 
436                 String typeForCheck = names3[i1];
437                 log2("......typeForCheck: " + typeForCheck);
438 
439                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
440                 List classList = vm.classesByName(typeForCheck);
441                 if (classList.size() != 1) {
442                     log3("ERROR: classList.size() != 1  : " + classList.size());
443                     testExitCode = FAILED;
444                     continue;
445                 }
446 
447                 log2("      getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
448                 ReferenceType referenceType  = (ReferenceType) classList.get(0);

450                 log2("      getting: boolean isStatic = referenceType.isStatic();");
451                 boolean isStatic = referenceType.isStatic();
452 
453                 log2("      expected value of isStatic is 'false'");
454                 if (isStatic) {
455                     log3("ERROR: isStatic != false for: " + typeForCheck);
456                     testExitCode = FAILED;
457                 }
458             }
459 
460             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
461         }
462         log1("    TESTING ENDS");
463         return;
464     }

466     private ThreadReference threadByName(String name)
467                  throws JDITestRuntimeException {
468 
469         List         all = vm.allThreads();
470         ListIterator li  = all.listIterator();
471 
472         for (; li.hasNext(); ) {
473             ThreadReference thread = (ThreadReference) li.next();
474             if (thread.name().equals(name))
475                 return thread;
476         }
477         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
478     }

480    /*
481     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
482     *                                             String, String, String)
483     *
484     * It sets up a breakpoint at given line number within a given method in a given class
485     * for a given thread.
486     *
487     * Return value: BreakpointRequest object  in case of success
488     *
489     * JDITestRuntimeException   in case of an Exception thrown within the method
490     */
491 
492     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
493                                                   ReferenceType testedClass,
494                                                   String methodName,
495                                                   String bpLine,
496                                                   String property)
497             throws JDITestRuntimeException {
498 
499         log2("......setting up a breakpoint:");
500         log2("       thread: " + thread + "; class: " + testedClass +
501                         "; method: " + methodName + "; line: " + bpLine);
502 
503         List              alllineLocations = null;
504         Location          lineLocation     = null;
505         BreakpointRequest breakpRequest    = null;
506 
507         try {
508             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
509 
510             alllineLocations = method.allLineLocations();
511 
512             int n =
513                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
514             if (n > alllineLocations.size()) {
515                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
516             } else {
517                 lineLocation = (Location) alllineLocations.get(n);
518                 try {
519                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
520                     breakpRequest.putProperty("number", property);
521                     breakpRequest.addThreadFilter(thread);
522                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
523                 } catch ( Exception e1 ) {
524                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
525                     breakpRequest    = null;
526                 }
527             }
528         } catch ( Exception e2 ) {
529             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
530             breakpRequest    = null;
531         }
532 
533         if (breakpRequest == null) {
534             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
535             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
536         }
537 
538         log2("      a breakpoint has been set up");
539         return breakpRequest;
540     }

543     private void getEventSet()
544                  throws JDITestRuntimeException {
545         try {
546 //            log2("       eventSet = eventQueue.remove(waitTime);");
547             eventSet = eventQueue.remove(waitTime);
548             if (eventSet == null) {
549                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
550             }
551 //            log2("       eventIterator = eventSet.eventIterator;");
552             eventIterator = eventSet.eventIterator();
553         } catch ( Exception e ) {
554             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
555         }
556     }

559     private void breakpointForCommunication()
560                  throws JDITestRuntimeException {
561 
562         log2("breakpointForCommunication");
563         getEventSet();
564 
565         if (eventIterator.nextEvent() instanceof BreakpointEvent)
566             return;
567 
568         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
569     }
