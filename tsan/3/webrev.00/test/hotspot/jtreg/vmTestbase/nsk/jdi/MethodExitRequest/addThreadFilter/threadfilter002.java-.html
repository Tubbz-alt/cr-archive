 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodExitRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodExitRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  *            Filters may be added only to disabled requests.   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new threads, thread1, and invokes             <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an MethodExitRequest and enables it,                     <BR>
 72  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addThreadFilter()    <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

167     private int runThis (String argv[], PrintStream out) {
168 
169         argsHandler     = new ArgumentHandler(argv);
170         logHandler      = new Log(out, argsHandler);
171         Binder binder   = new Binder(argsHandler, logHandler);
172 
173         waitTime        = argsHandler.getWaitTime() * 60000;
174 
175         try {
176             log2("launching a debuggee :");
177             log2("       " + debuggeeName);
178             if (argsHandler.verbose()) {
179                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
180             } else {
181                 debuggee = binder.bindToDebugee(debuggeeName);
182             }
183             if (debuggee == null) {
184                 log3("ERROR: no debuggee launched");
185                 return FAILED;
186             }
187             log2("debuggee launched");
188         } catch ( Exception e ) {
189             log3("ERROR: Exception : " + e);
190             log2("       test cancelled");
191             return FAILED;
192         }
193 
194         debuggee.redirectOutput(logHandler);
195 
196         vm = debuggee.VM();
197 
198         eventQueue = vm.eventQueue();
199         if (eventQueue == null) {
200             log3("ERROR: eventQueue == null : TEST ABORTED");
201             vm.exit(PASS_BASE);
202             return FAILED;
203         }
204 
205         log2("invocation of the method runTest()");
206         switch (runTest()) {
207 
208             case 0 :  log2("test phase has finished normally");
209                       log2("   waiting for the debuggee to finish ...");
210                       debuggee.waitFor();
211 
212                       log2("......getting the debuggee's exit status");
213                       int status = debuggee.getStatus();
214                       if (status != PASS_BASE) {
215                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
216                               status + " != PASS_BASE");
217                           testExitCode = FAILED;
218                       } else {
219                           log2("......debuggee returned expected exit status: " +
220                               status + " == PASS_BASE");
221                       }
222                       break;
223 
224             default : log3("ERROR: runTest() returned unexpected value");
225 
226             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
227                       log2("......forcing: vm.exit();");
228                       testExitCode = FAILED;
229                       try {
230                           vm.exit(PASS_BASE);
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235 
236             case 2 :  log3("test cancelled due to VMDisconnectedException");
237                       log2("......trying: vm.process().destroy();");
238                       testExitCode = FAILED;
239                       try {
240                           Process vmProcess = vm.process();
241                           if (vmProcess != null) {
242                               vmProcess.destroy();
243                           }
244                       } catch ( Exception e ) {
245                           log3("ERROR: Exception : " + e);
246                       }
247                       break;
248             }
249 
250         return testExitCode;
251     }

255     * Return value: 0 - normal end of the test
256     *               1 - ubnormal end of the test
257     *               2 - VMDisconnectedException while test phase
258     */

260     private int runTest() {
261 
262         try {
263             testRun();
264 
265             log2("waiting for VMDeathEvent");
266             getEventSet();
267             if (eventIterator.nextEvent() instanceof VMDeathEvent)
268                 return 0;
269 
270             log3("ERROR: last event is not the VMDeathEvent");
271             return 1;
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = threadByName("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318 
319         EventRequest eventRequest1 = null;
320 
321         ThreadReference thread1     = null;
322         String          thread1Name = "thread1";
323 
324         String property1 = "MethodExitRequest1";
325 
326         ReferenceType testClassReference = null;
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             switch (i) {
346               case 0:
347                      testClassReference =
348                             (ReferenceType) vm.classesByName(testedClassName).get(0);
349                      thread1 = (ThreadReference) debuggeeClass.getValue(
350                                         debuggeeClass.fieldByName(thread1Name));
351 
352                      eventRequest1 = setting2MethodExitRequest (null,
353                                              testClassReference,
354                                              EventRequest.SUSPEND_NONE, property1);
355 
356                      log2("......eventRequest1.enable();");
357                      eventRequest1.enable();
358 
359                      try {
360                          log2("......eventRequest1.addThreadFilter(thread1);");
361                          log2("        InvalidRequestStateException expected");
362                          ((MethodExitRequest)eventRequest1).addThreadFilter(thread1);
363                          log3("ERROR: no InvalidRequestStateException");
364                          testExitCode = FAILED;
365                      } catch ( InvalidRequestStateException e ) {
366                          log2("        InvalidRequestStateException");
367                      }
368 
369                      log2("......eventRequest1.disable();");
370                      eventRequest1.disable();
371                      try {
372                          log2("......eventRequest1.addThreadFilter(thread1);");
373                          log2("        no InvalidRequestStateException expected");
374                          ((MethodExitRequest)eventRequest1).addThreadFilter(thread1);
375                          log2("        no InvalidRequestStateException");
376                      } catch ( InvalidRequestStateException e ) {
377                          log3("ERROR: InvalidRequestStateException");
378                          testExitCode = FAILED;
379                      }
380 
381                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
382                      eventRManager.deleteEventRequest(eventRequest1);
383                      try {
384                          log2("......eventRequest1.addThreadFilter(thread1);");
385                          log2("        InvalidRequestStateException expected");
386                          ((MethodExitRequest)eventRequest1).addThreadFilter(thread1);
387                          log3("ERROR: no InvalidRequestStateException");
388                          testExitCode = FAILED;
389                      } catch ( InvalidRequestStateException e ) {
390                          log2("        InvalidRequestStateException");
391                      }
392 
393                      break;
394 
395               default:
396                       throw new JDITestRuntimeException("** default case 2 **");
397             }
398 
399             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
400         }
401         log1("    TESTING ENDS");
402         return;
403     }

405     private ThreadReference threadByName(String name)
406                  throws JDITestRuntimeException {
407 
408         List         all = vm.allThreads();
409         ListIterator li  = all.listIterator();
410 
411         for (; li.hasNext(); ) {
412             ThreadReference thread = (ThreadReference) li.next();
413             if (thread.name().equals(name))
414                 return thread;
415         }
416         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
417     }

419    /*
420     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
421     *                                             String, String, String)
422     *
423     * It sets up a breakpoint at given line number within a given method in a given class
424     * for a given thread.
425     *
426     * Return value: BreakpointRequest object  in case of success
427     *
428     * JDITestRuntimeException   in case of an Exception thrown within the method
429     */
430 
431     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
432                                                   ReferenceType testedClass,
433                                                   String methodName,
434                                                   String bpLine,
435                                                   String property)
436             throws JDITestRuntimeException {
437 
438         log2("......setting up a breakpoint:");
439         log2("       thread: " + thread + "; class: " + testedClass +
440                         "; method: " + methodName + "; line: " + bpLine);
441 
442         List              alllineLocations = null;
443         Location          lineLocation     = null;
444         BreakpointRequest breakpRequest    = null;
445 
446         try {
447             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
448 
449             alllineLocations = method.allLineLocations();
450 
451             int n =
452                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
453             if (n > alllineLocations.size()) {
454                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
455             } else {
456                 lineLocation = (Location) alllineLocations.get(n);
457                 try {
458                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
459                     breakpRequest.putProperty("number", property);
460                     breakpRequest.addThreadFilter(thread);
461                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
462                 } catch ( Exception e1 ) {
463                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
464                     breakpRequest    = null;
465                 }
466             }
467         } catch ( Exception e2 ) {
468             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
469             breakpRequest    = null;
470         }
471 
472         if (breakpRequest == null) {
473             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
474             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
475         }
476 
477         log2("      a breakpoint has been set up");
478         return breakpRequest;
479     }

482     private void getEventSet()
483                  throws JDITestRuntimeException {
484         try {
485 //            log2("       eventSet = eventQueue.remove(waitTime);");
486             eventSet = eventQueue.remove(waitTime);
487             if (eventSet == null) {
488                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
489             }
490 //            log2("       eventIterator = eventSet.eventIterator;");
491             eventIterator = eventSet.eventIterator();
492         } catch ( Exception e ) {
493             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
494         }
495     }

498     private void breakpointForCommunication()
499                  throws JDITestRuntimeException {
500 
501         log2("breakpointForCommunication");
502         getEventSet();
503 
504         if (eventIterator.nextEvent() instanceof BreakpointEvent)
505             return;
506 
507         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
508     }

512     private MethodExitRequest setting2MethodExitRequest ( ThreadReference thread,
513                                                          ReferenceType   testedClass,
514                                                          int             suspendPolicy,
515                                                          String          property        )
516             throws JDITestRuntimeException {
517         try {
518             log2("......setting up MethodExitRequest:");
519             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
520 
521             MethodExitRequest
522             mexr = eventRManager.createMethodExitRequest();
523             mexr.putProperty("number", property);
524             if (thread != null)
525                 mexr.addThreadFilter(thread);
526             mexr.addClassFilter(testedClass);
527             mexr.setSuspendPolicy(suspendPolicy);
528 
529         log2("      MethodExitRequest has been set up");
530         return mexr;
531         } catch ( Exception e ) {
532             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
533             log3("       MethodExitRequest HAS NOT BEEN SET UP");
534             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
535         }
