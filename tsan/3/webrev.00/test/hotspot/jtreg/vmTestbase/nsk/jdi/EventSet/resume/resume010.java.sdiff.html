502     private ThreadReference threadByName(String name)
503                  throws JDITestRuntimeException {
504 
505         List         all = vm.allThreads();
506         ListIterator li  = all.listIterator();
507 
508         for (; li.hasNext(); ) {
509             ThreadReference thread = (ThreadReference) li.next();
510             if (thread.name().equals(name))
511                 return thread;
512         }
513         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
514     }
