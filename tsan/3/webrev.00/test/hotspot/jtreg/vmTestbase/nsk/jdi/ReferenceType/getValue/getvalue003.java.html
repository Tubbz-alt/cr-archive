 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.getValue()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * Case for testing includes arrays of primitive types declared <BR>
 45  * in a superinterface and an implemented interface.            <BR>
 46  * <BR>
 47  * The test works as follows.                                           <BR>
 48  * Upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request                  <BR>
 52  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 53  * resumes the VM, and waits for the event within the predefined        <BR>
 54  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 57  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 58  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 59  * to be suspended and to inform the debugger with the event.           <BR>
 60  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 61  * At the end, the debuggee changes the value of the "instruction"      <BR>
 62  * to inform the debugger of checks finished, and both end.             <BR>
 63  */

139     private int runThis (String argv[], PrintStream out) {
140 
141         argsHandler     = new ArgumentHandler(argv);
142         logHandler      = new Log(out, argsHandler);
143         Binder binder   = new Binder(argsHandler, logHandler);
144 
145         waitTime        = argsHandler.getWaitTime() * 60000;
146 
147         try {
148             log2("launching a debuggee :");
149             log2("       " + debuggeeName);
150             if (argsHandler.verbose()) {
151                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
152             } else {
153                 debuggee = binder.bindToDebugee(debuggeeName);
154             }
155             if (debuggee == null) {
156                 log3("ERROR: no debuggee launched");
157                 return FAILED;
158             }
159             log2("debuggee launched");
160         } catch ( Exception e ) {
161             log3("ERROR: Exception : " + e);
162             log2("       test cancelled");
163             return FAILED;
164         }
165 
166         debuggee.redirectOutput(logHandler);
167 
168         vm = debuggee.VM();
169 
170         eventQueue = vm.eventQueue();
171         if (eventQueue == null) {
172             log3("ERROR: eventQueue == null : TEST ABORTED");
173             vm.exit(PASS_BASE);
174             return FAILED;
175         }
176 
177         log2("invocation of the method runTest()");
178         switch (runTest()) {
179 
180             case 0 :  log2("test phase has finished normally");
181                       log2("   waiting for the debuggee to finish ...");
182                       debuggee.waitFor();
183 
184                       log2("......getting the debuggee's exit status");
185                       int status = debuggee.getStatus();
186                       if (status != PASS_BASE) {
187                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
188                               status + " != PASS_BASE");
189                           testExitCode = FAILED;
190                       } else {
191                           log2("......debuggee returned expected exit status: " +
192                               status + " == PASS_BASE");
193                       }
194                       break;
195 
196             default : log3("ERROR: runTest() returned unexpected value");
197 
198             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
199                       log2("......forcing: vm.exit();");
200                       testExitCode = FAILED;
201                       try {
202                           vm.exit(PASS_BASE);
203                       } catch ( Exception e ) {
204                           log3("ERROR: Exception : e");
205                       }
206                       break;
207 
208             case 2 :  log3("test cancelled due to VMDisconnectedException");
209                       log2("......trying: vm.process().destroy();");
210                       testExitCode = FAILED;
211                       try {
212                           Process vmProcess = vm.process();
213                           if (vmProcess != null) {
214                               vmProcess.destroy();
215                           }
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : e");
218                       }
219                       break;
220             }
221 
222         return testExitCode;
223     }

226    /*
227     * Return value: 0 - normal end of the test
228     *               1 - ubnormal end of the test
229     *               2 - VMDisconnectedException while test phase
230     */
231 
232     private int runTest() {
233 
234         try {
235             testRun();
236 
237             log2("waiting for VMDeathEvent");
238             getEventSet();
239             if (eventIterator.nextEvent() instanceof VMDeathEvent)
240                 return 0;
241 
242             log3("ERROR: last event is not the VMDeathEvent");
243             return 1;
244         } catch ( VMDisconnectedException e ) {
245             log3("ERROR: VMDisconnectedException : " + e);
246             return 2;
247         } catch ( Exception e ) {
248             log3("ERROR: Exception : " + e);
249             return 1;
250         }
251 
252     }

254     private void testRun()
255                  throws JDITestRuntimeException, Exception {
256 
257         eventRManager = vm.eventRequestManager();
258 
259         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
260         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
261         cpRequest.addClassFilter(debuggeeName);
262 
263         cpRequest.enable();
264         vm.resume();
265         getEventSet();
266         cpRequest.disable();
267 
268         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
269         debuggeeClass = event.referenceType();
270 
271         if (!debuggeeClass.name().equals(debuggeeName))
272            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
273 
274         log2("      received: ClassPrepareEvent for debuggeeClass");
275 
276         String bPointMethod = "methodForCommunication";
277         String lineForComm  = "lineForComm";
278         BreakpointRequest bpRequest;
279 
280         try {
281             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
282                                           debuggeeClass,
283                                           bPointMethod, lineForComm, "zero");
284         } catch ( Exception e ) {
285             throw e;
286         }
287         bpRequest.enable();
288 
289     //------------------------------------------------------  testing section
290 
291         log1("     TESTING BEGINS");
292 
293         for (int i = 0; ; i++) {
294 
295             vm.resume();
296             breakpointForCommunication();
297 
298             int instruction = ((IntegerValue)
299                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
300 
301             if (instruction == 0) {
302                 vm.resume();
303                 break;
304             }
305 
306             log1("  new check: # " + i);
307 
308             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
309 
310             log2("      name of tested class : " + testedClassName);
311             log2("      getting: List classes = vm.classByName(testedClassName);");
312             List classes = vm.classesByName(testedClassName);
313             if (classes.size() != 1) {
314                 log3("ERROR: classes.size() != 1 : " + classes.size());
315                 testExitCode = FAILED;
316                 break;
317             }
318             log2("      getting: ReferenceType testedClass = (ReferenceType) classes.get(0);");
319             ReferenceType testedClass = (ReferenceType) classes.get(0);
320 
321 
322             log2("------case for testing: arrays of primitive values");
323 
324             String fieldNames[] = {
325                        "bl1",
326                        "bt1",
327                        "ch1",
328                        "db1",
329                        "fl1"
330                        //"in1",
331                        //"ln1",
332                        //"sh1"
333             };
334 
335             for (int i1 = 0; i1 < fieldNames.length; i1++) {
336 
337                 log2("......field tested : " + fieldNames[i1]);
338                 log2("      getting: Field field = testedClass.fieldByName(fieldNames[i1]);");
339                 Field field = testedClass.fieldByName(fieldNames[i1]);
340                 if (field == null) {
341                     log3("ERROR: field == null");
342                     testExitCode = FAILED;
343                     continue;
344                 }
345                 log2("      getting: Value value = testedClass.getValue(field);");
346                 Value value = testedClass.getValue(field);
347                 if (value == null) {
348                     log3("ERROR: value == null");
349                     testExitCode = FAILED;
350                     continue;
351                 }
352                 try {
353                     log2("      getting Class cast: ObjectReference objectReferenceValue = (ObjectReference) value;");
354                     ObjectReference objectReferenceValue = (ObjectReference) value;
355                 } catch ( ClassCastException e ) {
356                     log3("ERROR: ClassCastException");
357                     testExitCode = FAILED;
358                 }
359             }
360             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
361         }
362         log1("    TESTING ENDS");
363         return;
364     }

366    /*
367     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
368     *                                             String, String, String)
369     *
370     * It sets up a breakpoint at given line number within a given method in a given class
371     * for a given thread.
372     *
373     * Return value: BreakpointRequest object  in case of success
374     *
375     * JDITestRuntimeException   in case of an Exception thrown within the method
376     */
377 
378     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
379                                                   ReferenceType testedClass,
380                                                   String methodName,
381                                                   String bpLine,
382                                                   String property)
383             throws JDITestRuntimeException {
384 
385         log2("......setting up a breakpoint:");
386         log2("       thread: " + thread + "; class: " + testedClass +
387                         "; method: " + methodName + "; line: " + bpLine);
388 
389         List              alllineLocations = null;
390         Location          lineLocation     = null;
391         BreakpointRequest breakpRequest    = null;
392 
393         try {
394             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
395 
396             alllineLocations = method.allLineLocations();
397 
398             int n =
399                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
400             if (n > alllineLocations.size()) {
401                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
402             } else {
403                 lineLocation = (Location) alllineLocations.get(n);
404                 try {
405                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
406                     breakpRequest.putProperty("number", property);
407                     breakpRequest.addThreadFilter(thread);
408                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
409                 } catch ( Exception e1 ) {
410                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
411                     breakpRequest    = null;
412                 }
413             }
414         } catch ( Exception e2 ) {
415             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
416             breakpRequest    = null;
417         }
418 
419         if (breakpRequest == null) {
420             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
421             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
422         }
423 
424         log2("      a breakpoint has been set up");
425         return breakpRequest;
426     }

429     private void getEventSet()
430                  throws JDITestRuntimeException {
431         try {
432 //            log2("       eventSet = eventQueue.remove(waitTime);");
433             eventSet = eventQueue.remove(waitTime);
434             if (eventSet == null) {
435                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
436             }
437 //            log2("       eventIterator = eventSet.eventIterator;");
438             eventIterator = eventSet.eventIterator();
439         } catch ( Exception e ) {
440             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
441         }
442     }

445     private void breakpointForCommunication()
446                  throws JDITestRuntimeException {
447 
448         log2("breakpointForCommunication");
449         getEventSet();
450 
451         if (eventIterator.nextEvent() instanceof BreakpointEvent)
452             return;
453 
454         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
455     }
