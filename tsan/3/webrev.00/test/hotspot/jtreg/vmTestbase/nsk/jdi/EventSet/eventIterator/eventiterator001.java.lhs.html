 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.eventIterator()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for ClassPrepareEvent set               <BR>
 46  *  - the method returns a non-null object, and                 <BR>
 47  *  - object's class is the subclass of class Iterator.         <BR>
 48  * <BR>
 49  * The test works as follows.                                           <BR>
 50  * <BR>
 51  * Upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent,                                       <BR>
 55  * the debugger sets up two requests, both with SUSPEND_EVENT_THREAD,   <BR>
 56  * one for debuggee's ClassPrepareEvent,                                <BR>
 57  * another one for a special TestClass,  resumes the VM,                <BR>
 58  * and waits for the events within the predefined time interval.        <BR>
 59  * If no the events received, the test is FAILED.                       <BR>
 60  * Upon getting the debuggeeClass's ClassPrepareEvent,                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * Upon getting TestClass's ClassPrepareEvent,                          <BR>
 64  * the debugger performs the check.                                     <BR>
 65  * <BR>
 66  * At the end the debuggee changes the value of the "instruction"       <BR>
 67  * to inform the debugger of checks finished, and both ends.            <BR>
 68  * <BR>
 69  * Note. To inform each other of needed actions, the debugger and       <BR>
 70  *       and the debuggee use debuggee's variable "instruction".        <BR>
 71  * <BR>
 72  */
 73

156     private int runThis (String argv[], PrintStream out) {
157 
158         argsHandler     = new ArgumentHandler(argv);
159         logHandler      = new Log(out, argsHandler);
160         Binder binder   = new Binder(argsHandler, logHandler);
161 
162         waitTime        = argsHandler.getWaitTime() * 60000;
163 
164         try {
165             log2("launching a debuggee :");
166             log2("       " + debuggeeName);
167             if (argsHandler.verbose()) {
168                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
169             } else {
170                 debuggee = binder.bindToDebugee(debuggeeName);
171             }
172             if (debuggee == null) {
173                 log3("ERROR: no debuggee launched");
174                 return FAILED;
175             }
176             log2("debuggee launched");
177         } catch ( Exception e ) {
178             log3("ERROR: Exception : " + e);
179             log2("       test cancelled");
180             return FAILED;
181         }
182 
183         debuggee.redirectOutput(logHandler);
184 
185         vm = debuggee.VM();
186 
187         eventQueue = vm.eventQueue();
188         if (eventQueue == null) {
189             log3("ERROR: eventQueue == null : TEST ABORTED");
190             vm.exit(PASS_BASE);
191             return FAILED;
192         }
193 
194         log2("invocation of the method runTest()");
195         switch (runTest()) {
196 
197             case 0 :  log2("test phase has finished normally");
198                       log2("   waiting for the debuggee to finish ...");
199                       debuggee.waitFor();
200 
201                       log2("......getting the debuggee's exit status");
202                       int status = debuggee.getStatus();
203                       if (status != PASS_BASE) {
204                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
205                               status + " != PASS_BASE");
206                           testExitCode = FAILED;
207                       } else {
208                           log2("......debuggee returned expected exit status: " +
209                               status + " == PASS_BASE");
210                       }
211                       break;
212 
213             default : log3("ERROR: runTest() returned unexpected value");
214 
215             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
216                       log2("......forcing: vm.exit();");
217                       testExitCode = FAILED;
218                       try {
219                           vm.exit(PASS_BASE);
220                       } catch ( Exception e ) {
221                           log3("ERROR: Exception : e");
222                       }
223                       break;
224 
225             case 2 :  log3("test cancelled due to VMDisconnectedException");
226                       log2("......trying: vm.process().destroy();");
227                       testExitCode = FAILED;
228                       try {
229                           Process vmProcess = vm.process();
230                           if (vmProcess != null) {
231                               vmProcess.destroy();
232                           }
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237             }
238 
239         return testExitCode;
240     }

243    /*
244     * Return value: 0 - normal end of the test
245     *               1 - ubnormal end of the test
246     *               2 - VMDisconnectedException while test phase
247     */
248 
249     private int runTest() {
250 
251         try {
252             testRun();
253 
254             log2("waiting for VMDeathEvent");
255             getEventSet();
256 
257             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
258                 log3("ERROR: last event is not the VMDeathEvent");
259                 return 1;
260             }
261 
262             log2("waiting for VMDisconnectEvent");
263             getEventSet();
264 
265             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
266                 log3("ERROR: last event is not the VMDisconnectEvent");
267                 return 1;
268             }
269 
270             return 0;
271 
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279 
280     }

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290         cpRequest.putProperty("number", "debuggeeClass");
291         cpRequest.enable();
292 
293         ClassPrepareRequest tcRequest = settingClassPrepareRequest(testedClassName,
294                                                     EventRequest.SUSPEND_EVENT_THREAD,
295                                                     "TestClassPrepareRequest");
296         tcRequest.enable();
297 
298         int nn1 = 0;
299         int nn2 = 0;
300         for (int nn = 0; nn < 2; nn++) {
301 
302             vm.resume();
303             getEventSet();
304 
305             ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
306 
307             log2("String property = (String) event.request().getProperty('number');");
308             String property = (String) event.request().getProperty("number");
309 
310             if (property.equals("debuggeeClass")) {
311 
312                 log2("      received: ClassPrepareEvent for debuggeeClass");
313 
314                 nn1++;
315                 if (nn1 > 1)
316                     throw new JDITestRuntimeException("** 2-nd event for debuggeeClass **");
317 
318                 cpRequest.disable();
319                 debuggeeClass = event.referenceType();
320 
321                 String          bPointMethod = "methodForCommunication";
322                 String          lineForComm  = "lineForComm";
323                 ThreadReference mainThread   = threadByName("main");
324 
325                 BreakpointRequest bpRequest = settingBreakpoint(mainThread,
326                                               debuggeeClass,
327                                               bPointMethod, lineForComm, "zero");
328                 bpRequest.enable();
329 
330             } else if (property.equals("TestClassPrepareRequest")) {
331                 nn2++;
332                 if (nn2 > 1)
333                     throw new JDITestRuntimeException("** 2-nd event for TestClass **");
334 
335                 if (eventIterator == null) {
336                     testExitCode = FAILED;
337                     log3("ERROR: eventIterator == null");
338                 }
339                 if ( !(eventIterator instanceof Iterator) ) {
340                     testExitCode = FAILED;
341                     log3("ERROR: eventIterator is NOT instanceof Iterator");
342                 }
343             } else {
344                 log3("ERROR: unexpected Event :  property == " + property);
345                 throw new JDITestRuntimeException("** UNEXPECTED Event **");
346             }
347         }
348 
349     //------------------------------------------------------  testing section
350 
351         log1("     TESTING BEGINS");
352 
353         vm.resume();
354         breakpointForCommunication();
355 
356         int instruction = ((IntegerValue)
357                            (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
358 
359         if (instruction != 0) {
360             throw new JDITestRuntimeException("** instruction != 0 **");
361         }
362         vm.resume();
363             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
364             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
365 
366         log1("    TESTING ENDS");
367         return;
368     }

370     private ThreadReference threadByName(String name)
371                  throws JDITestRuntimeException {
372 
373         List         all = vm.allThreads();
374         ListIterator li  = all.listIterator();
375 
376         for (; li.hasNext(); ) {
377             ThreadReference thread = (ThreadReference) li.next();
378             if (thread.name().equals(name))
379                 return thread;
380         }
381         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
382     }

384    /*
385     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
386     *                                             String, String, String)
387     *
388     * It sets up a breakpoint at given line number within a given method in a given class
389     * for a given thread.
390     *
391     * Return value: BreakpointRequest object  in case of success
392     *
393     * JDITestRuntimeException   in case of an Exception thrown within the method
394     */
395 
396     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
397                                                   ReferenceType testedClass,
398                                                   String methodName,
399                                                   String bpLine,
400                                                   String property)
401             throws JDITestRuntimeException {
402 
403         log2("......setting up a breakpoint:");
404         log2("       thread: " + thread + "; class: " + testedClass +
405                         "; method: " + methodName + "; line: " + bpLine);
406 
407         List              alllineLocations = null;
408         Location          lineLocation     = null;
409         BreakpointRequest breakpRequest    = null;
410 
411         try {
412             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
413 
414             alllineLocations = method.allLineLocations();
415 
416             int n =
417                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
418             if (n > alllineLocations.size()) {
419                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
420             } else {
421                 lineLocation = (Location) alllineLocations.get(n);
422                 try {
423                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
424                     breakpRequest.putProperty("number", property);
425                     breakpRequest.addThreadFilter(thread);
426                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
427                 } catch ( Exception e1 ) {
428                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
429                     breakpRequest    = null;
430                 }
431             }
432         } catch ( Exception e2 ) {
433             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
434             breakpRequest    = null;
435         }
436 
437         if (breakpRequest == null) {
438             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
439             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
440         }
441 
442         log2("      a breakpoint has been set up");
443         return breakpRequest;
444     }

447     private void getEventSet()
448                  throws JDITestRuntimeException {
449         try {
450 //            log2("       eventSet = eventQueue.remove(waitTime);");
451             eventSet = eventQueue.remove(waitTime);
452             if (eventSet == null) {
453                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
454             }
455 //            log2("       eventIterator = eventSet.eventIterator;");
456             eventIterator = eventSet.eventIterator();
457         } catch ( Exception e ) {
458             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
459         }
460     }

463     private void breakpointForCommunication()
464                  throws JDITestRuntimeException {
465 
466         log2("breakpointForCommunication");
467         getEventSet();
468 
469         if (eventIterator.nextEvent() instanceof BreakpointEvent)
470             return;
471 
472         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
473     }

479                                                              String  property       )
480             throws JDITestRuntimeException {
481         try {
482             log2("......setting up ClassPrepareRequest:");

484 
485             ClassPrepareRequest
486             cpr = eventRManager.createClassPrepareRequest();
487             cpr.putProperty("number", property);
488             cpr.addClassFilter(testedClass);
489             cpr.setSuspendPolicy(suspendPolicy);
490 
491             log2("      ClassPrepareRequest has been set up");
492             return cpr;
493         } catch ( Exception e ) {
494             log3("ERROR: ATTENTION: Exception within settingClassPrepareRequest() : " + e);
495             log3("       ClassPreparenRequest HAS NOT BEEN SET UP");
496             throw new JDITestRuntimeException("** FAILURE to set up ClassPrepareRequest **");
497         }
