 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createExceptionRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *  - If neither caught exception nor uncaught exception is selected,   <BR>
 47  *  that is, both notifyCaught and notifyUncaught arguments are false,  <BR>
 48  *  returned value is null.                                             <BR>
 49  * Note. In condition of no Exception for such testcase is defined in   <BR>
 50  * the spec,   null is the only reasonable returned value.              <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares a testcase and invokes                       <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the check required.                          <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

246    /*
247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase
250     */
251 
252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = threadByName("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         String fieldName = "testField";
309 
310         ExceptionRequest exRequest = null;
311 
312 
313         log1("     TESTING BEGINS");
314 
315         for (int i = 0; ; i++) {
316 
317             vm.resume();
318             breakpointForCommunication();
319 
320             int instruction = ((IntegerValue)
321                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
322 
323             if (instruction == 0) {
324                 vm.resume();
325                 break;
326             }
327 
328             log1(":::::: case: # " + i);
329 
330             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
331 
332             ReferenceType refType = (ReferenceType)
333                                     debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName)).type();
334 
335 
336             try {
337                 log2("......creating: exRequest = eventRManager.createExceptionRequest(refType, false, false);");
338                 exRequest = eventRManager.createExceptionRequest(refType, false, false);
339                 log2("      check on if returned value is null");
340                 if (exRequest == null) {
341                     log3("ERROR: returned value is null!");
342                     testExitCode = FAILED;
343                 }
344             } catch (Exception e) {
345                 log3("ERROR: exception caught: " + e);
346                 testExitCode = FAILED;
347             }
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
350         }
351         log1("    TESTING ENDS");
352         return;
353     }

355     private ThreadReference threadByName(String name)
356                  throws JDITestRuntimeException {
357 
358         List         all = vm.allThreads();
359         ListIterator li  = all.listIterator();
360 
361         for (; li.hasNext(); ) {
362             ThreadReference thread = (ThreadReference) li.next();
363             if (thread.name().equals(name))
364                 return thread;
365         }
366         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
367     }

369    /*
370     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
371     *                                             String, String, String)
372     *
373     * It sets up a breakpoint at given line number within a given method in a given class
374     * for a given thread.
375     *
376     * Return value: BreakpointRequest object  in case of success
377     *
378     * JDITestRuntimeException   in case of an Exception thrown within the method
379     */
380 
381     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
382                                                   ReferenceType testedClass,
383                                                   String methodName,
384                                                   String bpLine,
385                                                   String property)
386             throws JDITestRuntimeException {
387 
388         log2("......setting up a breakpoint:");
389         log2("       thread: " + thread + "; class: " + testedClass +
390                         "; method: " + methodName + "; line: " + bpLine);
391 
392         List              alllineLocations = null;
393         Location          lineLocation     = null;
394         BreakpointRequest breakpRequest    = null;
395 
396         try {
397             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
398 
399             alllineLocations = method.allLineLocations();
400 
401             int n =
402                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
403             if (n > alllineLocations.size()) {
404                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
405             } else {
406                 lineLocation = (Location) alllineLocations.get(n);
407                 try {
408                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
409                     breakpRequest.putProperty("number", property);
410                     breakpRequest.addThreadFilter(thread);
411                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
412                 } catch ( Exception e1 ) {
413                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
414                     breakpRequest    = null;
415                 }
416             }
417         } catch ( Exception e2 ) {
418             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
419             breakpRequest    = null;
420         }
421 
422         if (breakpRequest == null) {
423             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
424             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
425         }
426 
427         log2("      a breakpoint has been set up");
428         return breakpRequest;
429     }

432     private void getEventSet()
433                  throws JDITestRuntimeException {
434         try {
435 //            log2("       eventSet = eventQueue.remove(waitTime);");
436             eventSet = eventQueue.remove(waitTime);
437             if (eventSet == null) {
438                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
439             }
440 //            log2("       eventIterator = eventSet.eventIterator;");
441             eventIterator = eventSet.eventIterator();
442         } catch ( Exception e ) {
443             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
444         }
445     }

448     private void breakpointForCommunication()
449                  throws JDITestRuntimeException {
450 
451         log2("breakpointForCommunication");
452         getEventSet();
453 
454         if (eventIterator.nextEvent() instanceof BreakpointEvent)
455             return;
456 
457         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
458     }
