 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ClassLoader.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ClassLoader.definedClasses()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    The list has length 0 if no types have been defined by    <BR>
 47  *    this classloader.                                         <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.              <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assetion,                               <BR>
 64  * the debugger and the debuggee perform the following.                 <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee prepares a ClassLoader object and invokes             <BR>
 67  *   the methodForCommunication to be suspended and                     <BR>
 68  *   to inform the debugger with the event.                             <BR>
 69  * - Upon getting the BreakpointEvent,                                  <BR>
 70  *   the debugger performs the check.                                   <BR>
 71  * Note. To inform each other of needed actions, the debugger and       <BR>
 72  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 73  * <BR>
 74  * In third phase when at the end,                                      <BR>
 75  * the debuggee changes the value of the "instruction"                  <BR>
 76  * to inform the debugger of checks finished, and both end.             <BR>
 77  * <BR>
 78  */
 79

 82     //----------------------------------------------------- templete section
 83     static final int PASSED = 0;
 84     static final int FAILED = 2;
 85     static final int PASS_BASE = 95;
 86 
 87     //----------------------------------------------------- templete parameters
 88     static final String
 89     sHeader1 = "\n==> nsk/jdi/ClassLoaderReference/definedClasses/definedclasses001 ",
 90     sHeader2 = "--> debugger: ",
 91     sHeader3 = "##> debugger: ";

 95     public static void main (String argv[]) {

 97         int result = run(argv, System.out);

 99         System.exit(result + PASS_BASE);

102     public static int run (String argv[], PrintStream out) {
103 
104         int exitCode = new definedclasses001().runThis(argv, out);
105 
106         if (exitCode != PASSED) {
107             System.out.println("TEST FAILED");
108         }
109         return testExitCode;
110     }

152     private int runThis (String argv[], PrintStream out) {
153 
154         argsHandler     = new ArgumentHandler(argv);
155         logHandler      = new Log(out, argsHandler);
156         Binder binder   = new Binder(argsHandler, logHandler);
157 
158         waitTime        = argsHandler.getWaitTime() * 60000;
159 
160         try {
161             log2("launching a debuggee :");
162             log2("       " + debuggeeName);
163             if (argsHandler.verbose()) {
164                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
165             } else {
166                 debuggee = binder.bindToDebugee(debuggeeName);
167             }
168             if (debuggee == null) {
169                 log3("ERROR: no debuggee launched");
170                 return FAILED;
171             }
172             log2("debuggee launched");
173         } catch ( Exception e ) {
174             log3("ERROR: Exception : " + e);
175             log2("       test cancelled");
176             return FAILED;
177         }
178 
179         debuggee.redirectOutput(logHandler);
180 
181         vm = debuggee.VM();
182 
183         eventQueue = vm.eventQueue();
184         if (eventQueue == null) {
185             log3("ERROR: eventQueue == null : TEST ABORTED");
186             vm.exit(PASS_BASE);
187             return FAILED;
188         }
189 
190         log2("invocation of the method runTest()");
191         switch (runTest()) {
192 
193             case 0 :  log2("test phase has finished normally");
194                       log2("   waiting for the debuggee to finish ...");
195                       debuggee.waitFor();
196 
197                       log2("......getting the debuggee's exit status");
198                       int status = debuggee.getStatus();
199                       if (status != PASS_BASE) {
200                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
201                               status + " != PASS_BASE");
202                           testExitCode = FAILED;
203                       } else {
204                           log2("......debuggee returned expected exit status: " +
205                               status + " == PASS_BASE");
206                       }
207                       break;
208 
209             default : log3("ERROR: runTest() returned unexpected value");
210 
211             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
212                       log2("......forcing: vm.exit();");
213                       testExitCode = FAILED;
214                       try {
215                           vm.exit(PASS_BASE);
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : e");
218                       }
219                       break;
220 
221             case 2 :  log3("test cancelled due to VMDisconnectedException");
222                       log2("......trying: vm.process().destroy();");
223                       testExitCode = FAILED;
224                       try {
225                           Process vmProcess = vm.process();
226                           if (vmProcess != null) {
227                               vmProcess.destroy();
228                           }
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233             }
234 
235         return testExitCode;
236     }

239    /*
240     * Return value: 0 - normal end of the test
241     *               1 - ubnormal end of the test
242     *               2 - VMDisconnectedException while test phase
243     */
244 
245     private int runTest() {
246 
247         try {
248             testRun();
249 
250             log2("waiting for VMDeathEvent");
251             getEventSet();
252             if (eventIterator.nextEvent() instanceof VMDeathEvent)
253                 return 0;
254 
255             log3("ERROR: last event is not the VMDeathEvent");
256             return 1;
257         } catch ( VMDisconnectedException e ) {
258             log3("ERROR: VMDisconnectedException : " + e);
259             return 2;
260         } catch ( Exception e ) {
261             log3("ERROR: Exception : " + e);
262             return 1;
263         }
264 
265     }

267     private void testRun()
268                  throws JDITestRuntimeException, Exception {
269 
270         eventRManager = vm.eventRequestManager();
271 
272         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
273         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
274         cpRequest.addClassFilter(debuggeeName);
275 
276         cpRequest.enable();
277         vm.resume();
278         getEventSet();
279         cpRequest.disable();
280 
281         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
282         debuggeeClass = event.referenceType();
283 
284         if (!debuggeeClass.name().equals(debuggeeName))
285            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
286 
287         log2("      received: ClassPrepareEvent for debuggeeClass");
288 
289         String bPointMethod = "methodForCommunication";
290         String lineForComm  = "lineForComm";
291         BreakpointRequest bpRequest;
292 
293         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
294                                       debuggeeClass,
295                                       bPointMethod, lineForComm, "zero");
296         bpRequest.enable();
297 
298     //------------------------------------------------------  testing section
299 
300         log1("     TESTING BEGINS");
301 
302         for (int i = 0; ; i++) {
303 
304             vm.resume();
305             breakpointForCommunication();
306 
307             int instruction = ((IntegerValue)
308                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
309 
310             if (instruction == 0) {
311                 vm.resume();
312                 break;
313             }
314 
315             log1(":::::: case: # " + i);
316 
317             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
318 
319             String ClassLoaderObjName = "classLoader";
320 
321             log2("......getting: Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));");
322             Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));
323 
324             log2("......getting: ClassLoaderReference clRef = (ClassLoaderReference) val;");
325             ClassLoaderReference clRef = (ClassLoaderReference) val;
326 
327             log2("......getting: List definedClasses = clRef.definedClasses();");
328             List definedClasses = clRef.definedClasses();
329             log2("......checking up on definedClasses.size() :  0 is expected");
330             if (definedClasses.size() != 0) {
331                 log3("ERROR: definedClasses.size() != 0");
332                 testExitCode = FAILED;
333             }
334 
335             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
336         }
337         log1("    TESTING ENDS");
338         return;
339     }

341    /*
342     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
343     *                                             String, String, String)
344     *
345     * It sets up a breakpoint at given line number within a given method in a given class
346     * for a given thread.
347     *
348     * Return value: BreakpointRequest object  in case of success
349     *
350     * JDITestRuntimeException   in case of an Exception thrown within the method
351     */
352 
353     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
354                                                   ReferenceType testedClass,
355                                                   String methodName,
356                                                   String bpLine,
357                                                   String property)
358             throws JDITestRuntimeException {
359 
360         log2("......setting up a breakpoint:");
361         log2("       thread: " + thread + "; class: " + testedClass +
362                         "; method: " + methodName + "; line: " + bpLine);
363 
364         List              alllineLocations = null;
365         Location          lineLocation     = null;
366         BreakpointRequest breakpRequest    = null;
367 
368         try {
369             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
370 
371             alllineLocations = method.allLineLocations();
372 
373             int n =
374                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
375             if (n > alllineLocations.size()) {
376                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
377             } else {
378                 lineLocation = (Location) alllineLocations.get(n);
379                 try {
380                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
381                     breakpRequest.putProperty("number", property);
382                     breakpRequest.addThreadFilter(thread);
383                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
384                 } catch ( Exception e1 ) {
385                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
386                     breakpRequest    = null;
387                 }
388             }
389         } catch ( Exception e2 ) {
390             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
391             breakpRequest    = null;
392         }
393 
394         if (breakpRequest == null) {
395             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
396             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
397         }
398 
399         log2("      a breakpoint has been set up");
400         return breakpRequest;
401     }

404     private void getEventSet()
405                  throws JDITestRuntimeException {
406         try {
407 //            log2("       eventSet = eventQueue.remove(waitTime);");
408             eventSet = eventQueue.remove(waitTime);
409             if (eventSet == null) {
410                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
411             }
412 //            log2("       eventIterator = eventSet.eventIterator;");
413             eventIterator = eventSet.eventIterator();
414         } catch ( Exception e ) {
415             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
416         }
417     }

420     private void breakpointForCommunication()
421                  throws JDITestRuntimeException {
422 
423         log2("breakpointForCommunication");
424         getEventSet();
425 
426         if (eventIterator.nextEvent() instanceof BreakpointEvent)
427             return;
428 
429         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
430     }
