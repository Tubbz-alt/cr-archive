 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createBreakpointRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    - Multiple breakpoints at the same location are permitted.        <BR>
 47  *    - Creates a new disabled BreakpointRequest.                       <BR>
 48  *    - The BreakpointRequest is at the specified location.             <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares testcase and invokes                         <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger performs the checks required.                         <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */
 78

154     private int runThis (String argv[], PrintStream out) {
155 
156         argsHandler     = new ArgumentHandler(argv);
157         logHandler      = new Log(out, argsHandler);
158         Binder binder   = new Binder(argsHandler, logHandler);
159 
160         waitTime        = argsHandler.getWaitTime() * 60000;
161 
162         try {
163             log2("launching a debuggee :");
164             log2("       " + debuggeeName);
165             if (argsHandler.verbose()) {
166                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
167             } else {
168                 debuggee = binder.bindToDebugee(debuggeeName);
169             }
170             if (debuggee == null) {
171                 log3("ERROR: no debuggee launched");
172                 return FAILED;
173             }
174             log2("debuggee launched");
175         } catch ( Exception e ) {
176             log3("ERROR: Exception : " + e);
177             log2("       test cancelled");
178             return FAILED;
179         }
180 
181         debuggee.redirectOutput(logHandler);
182 
183         vm = debuggee.VM();
184 
185         eventQueue = vm.eventQueue();
186         if (eventQueue == null) {
187             log3("ERROR: eventQueue == null : TEST ABORTED");
188             vm.exit(PASS_BASE);
189             return FAILED;
190         }
191 
192         log2("invocation of the method runTest()");
193         switch (runTest()) {
194 
195             case 0 :  log2("test phase has finished normally");
196                       log2("   waiting for the debuggee to finish ...");
197                       debuggee.waitFor();
198 
199                       log2("......getting the debuggee's exit status");
200                       int status = debuggee.getStatus();
201                       if (status != PASS_BASE) {
202                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
203                               status + " != PASS_BASE");
204                           testExitCode = FAILED;
205                       } else {
206                           log2("......debuggee returned expected exit status: " +
207                               status + " == PASS_BASE");
208                       }
209                       break;
210 
211             default : log3("ERROR: runTest() returned unexpected value");
212 
213             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
214                       log2("......forcing: vm.exit();");
215                       testExitCode = FAILED;
216                       try {
217                           vm.exit(PASS_BASE);
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : " + e);
220                       }
221                       break;
222 
223             case 2 :  log3("test cancelled due to VMDisconnectedException");
224                       log2("......trying: vm.process().destroy();");
225                       testExitCode = FAILED;
226                       try {
227                           Process vmProcess = vm.process();
228                           if (vmProcess != null) {
229                               vmProcess.destroy();
230                           }
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235             }
236 
237         return testExitCode;
238     }

241    /*
242     * Return value: 0 - normal end of the test
243     *               1 - ubnormal end of the test
244     *               2 - VMDisconnectedException while test phase
245     */

247     private int runTest() {
248 
249         try {
250             testRun();
251 
252             log2("waiting for VMDeathEvent");
253             getEventSet();
254             if (eventIterator.nextEvent() instanceof VMDeathEvent)
255                 return 0;
256 
257             log3("ERROR: last event is not the VMDeathEvent");
258             return 1;
259         } catch ( VMDisconnectedException e ) {
260             log3("ERROR: VMDisconnectedException : " + e);
261             return 2;
262         } catch ( Exception e ) {
263             log3("ERROR: Exception : " + e);
264             return 1;
265         }
266 
267     }

269     private void testRun()
270                  throws JDITestRuntimeException, Exception {
271 
272         eventRManager = vm.eventRequestManager();
273 
274         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
275         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
276         cpRequest.addClassFilter(debuggeeName);
277 
278         cpRequest.enable();
279         vm.resume();
280         getEventSet();
281         cpRequest.disable();
282 
283         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
284         debuggeeClass = event.referenceType();
285 
286         if (!debuggeeClass.name().equals(debuggeeName))
287            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
288 
289         log2("      received: ClassPrepareEvent for debuggeeClass");
290 
291         String bPointMethod = "methodForCommunication";
292         String lineForComm  = "lineForComm";
293 
294         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
295 
296         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
297                                              debuggeeClass,
298                                             bPointMethod, lineForComm, "zero");
299         bpRequest.enable();
300 
301     //------------------------------------------------------  testing section
302 
303         List         requests = null;
304         ListIterator li       = null;
305 
306         BreakpointRequest bpRequest1 = null;
307         BreakpointRequest bpRequest2 = null;
308 
309 
310         log1("     TESTING BEGINS");
311 
312         for (int i = 0; ; i++) {
313 
314             vm.resume();
315             breakpointForCommunication();
316 
317             int instruction = ((IntegerValue)
318                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
319 
320             if (instruction == 0) {
321                 vm.resume();
322                 break;
323             }
324 
325             log1(":::::: case: # " + i);
326 
327             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
328 
329                 log2("      creating two BreakpointRequests for 'breakpLocation': bpRequest1 & bpRequest2");
330 
331                 bpRequest1 = eventRManager.createBreakpointRequest(breakpLocation);
332                 bpRequest2 = eventRManager.createBreakpointRequest(breakpLocation);
333 
334                 log2("      checking up on: requests are disabled");
335                 if (bpRequest1.isEnabled()) {
336                     testExitCode = FAILED;
337                     log3("ERROR: bpRequest1 is enabled");
338                 }
339                 if (bpRequest2.isEnabled()) {
340                     testExitCode = FAILED;
341                     log3("ERROR: bpRequest2 is enabled");
342                 }
343                 log2("      compareing locations equality");
344                 if ( !bpRequest1.location().equals(breakpLocation) ) {
345                     testExitCode = FAILED;
346                     log3("ERROR: !bpRequest1.location().equals(breakpLocation)");
347                 }
348                 if ( !bpRequest2.location().equals(breakpLocation) ) {
349                     testExitCode = FAILED;
350                     log3("ERROR: !bpRequest2.location().equals(breakpLocation)");
351                 }
352 
353             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
354         }
355         log1("    TESTING ENDS");
356         return;
357     }

359    /*
360     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
361     *                                             String, String, String)
362     *
363     * It sets up a breakpoint at given line number within a given method in a given class
364     * for a given thread.
365     *
366     * Return value: BreakpointRequest object  in case of success
367     *
368     * JDITestRuntimeException   in case of an Exception thrown within the method
369     */
370 
371     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
372                                                   ReferenceType testedClass,
373                                                   String methodName,
374                                                   String bpLine,
375                                                   String property)
376             throws JDITestRuntimeException {
377 
378         log2("......setting up a breakpoint:");
379         log2("       thread: " + thread + "; class: " + testedClass +
380                         "; method: " + methodName + "; line: " + bpLine);
381 
382         List              alllineLocations = null;
383         Location          lineLocation     = null;
384         BreakpointRequest breakpRequest    = null;
385 
386         try {
387             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
388 
389             alllineLocations = method.allLineLocations();
390 
391             int n =
392                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
393             if (n > alllineLocations.size()) {
394                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
395             } else {
396                 lineLocation = (Location) alllineLocations.get(n);
397 breakpLocation = lineLocation;
398                 try {
399                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
400                     breakpRequest.putProperty("number", property);
401                     breakpRequest.addThreadFilter(thread);
402                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
403                 } catch ( Exception e1 ) {
404                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
405                     breakpRequest    = null;
406                 }
407             }
408         } catch ( Exception e2 ) {
409             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
410             breakpRequest    = null;
411         }
412 
413         if (breakpRequest == null) {
414             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
415             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
416         }
417 
418         log2("      a breakpoint has been set up");
419         return breakpRequest;
420     }

423     private void getEventSet()
424                  throws JDITestRuntimeException {
425         try {
426 //            log2("       eventSet = eventQueue.remove(waitTime);");
427             eventSet = eventQueue.remove(waitTime);
428             if (eventSet == null) {
429                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
430             }
431 //            log2("       eventIterator = eventSet.eventIterator;");
432             eventIterator = eventSet.eventIterator();
433         } catch ( Exception e ) {
434             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
435         }
436     }

439     private void breakpointForCommunication()
440                  throws JDITestRuntimeException {
441 
442         log2("breakpointForCommunication");
443         getEventSet();
444 
445         if (eventIterator.nextEvent() instanceof BreakpointEvent)
446             return;
447 
448         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
449     }
