 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.classPrepareRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled class prepare requests.       <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

157     private int runThis (String argv[], PrintStream out) {
158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : " + e);
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

246     *               1 - ubnormal end of the test
247     *               2 - VMDisconnectedException while test phase

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if (eventIterator.nextEvent() instanceof VMDeathEvent)
258                 return 0;
259 
260             log3("ERROR: last event is not the VMDeathEvent");
261             return 1;
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }
269 
270     }

272     private void testRun()
273                  throws JDITestRuntimeException, Exception {
274 
275         eventRManager = vm.eventRequestManager();
276 
277         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
278         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
279         cpRequest.addClassFilter(debuggeeName);
280 
281         cpRequest.enable();
282         vm.resume();
283         getEventSet();
284         cpRequest.disable();
285 
286         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
287         debuggeeClass = event.referenceType();
288 
289         if (!debuggeeClass.name().equals(debuggeeName))
290            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
291 
292         log2("      received: ClassPrepareEvent for debuggeeClass");
293 
294         String bPointMethod = "methodForCommunication";
295         String lineForComm  = "lineForComm";
296 
297         ThreadReference   mainThread = threadByName("main");
298 
299         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
300                                              debuggeeClass,
301                                             bPointMethod, lineForComm, "zero");
302         bpRequest.enable();
303 
304     //------------------------------------------------------  testing section
305 
306         List         requests = null;
307         ListIterator li       = null;
308 
309         ClassPrepareRequest request = null;
310         ClassPrepareRequest cpRequests[] = { null, null, null, null, null,
311                                              null, null, null, null, null };
312         int listSize;
313         int flag;
314 
315 
316         log1("     TESTING BEGINS");
317 
318         for (int i = 0; ; i++) {
319 
320             vm.resume();
321             breakpointForCommunication();
322 
323             int instruction = ((IntegerValue)
324                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
325 
326             if (instruction == 0) {
327                 vm.resume();
328                 break;
329             }
330 
331             log1(":::::: case: # " + i);
332 
333             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
334 
335             cpRequest.putProperty("number", "zero");
336             cpRequest.enable();
337 
338             log2("......creating ClassPrepareRequests");
339             for (int i1 = 0; i1 < cpRequests.length; i1++) {
340                 cpRequests[i1] = eventRManager.createClassPrepareRequest();
341                 cpRequests[i1].putProperty("number", String.valueOf(i1));
342 
343                 log2("......checking up on returned List after creating new ClassPrepareRequest");
344                 requests = eventRManager.classPrepareRequests();
345                 listSize = requests.size();
346                 if ( listSize != (i1 + 2) ) {
347                     testExitCode = FAILED;
348                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 2));
349                 }
350                 flag = 0;
351                 li = requests.listIterator();
352                 while (li.hasNext()) {
353                     request = (ClassPrepareRequest) li.next();
354                     if ( request.getProperty("number").equals("zero") )
355                         continue;
356                     if ( !request.isEnabled() ) {
357                         flag++;
358                         if (flag > 1) {
359                             testExitCode = FAILED;
360                             log3("ERROR: # of disabled requests > 1 : " + flag);
361                         }
362                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
363                             testExitCode = FAILED;
364                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
365                         }
366                     } else {
367                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
368                             testExitCode = FAILED;
369                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
370                         }
371                     }
372                 }
373 
374                 log2("      enabling created ClassPrepareRequest");
375                 cpRequests[i1].enable();
376                 requests = eventRManager.classPrepareRequests();
377                 listSize = requests.size();
378                 if ( listSize != (i1 + 2) ) {
379                     testExitCode = FAILED;
380                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 2));
381                 }
382 
383                 li = requests.listIterator();
384                 while (li.hasNext()) {
385                     request = (ClassPrepareRequest) li.next();
386                     if ( !request.isEnabled() ) {
387                         testExitCode = FAILED;
388                         log3("ERROR: returned List contains disabled request : " + request);
389                     }
390                 }
391 
392                 log2("      removing item from the List; UnsupportedOperationException is expected");
393                 try {
394                     requests.remove(i1);
395                     testExitCode = FAILED;
396                     log3("ERROR: NO exception");
397                 } catch ( UnsupportedOperationException e ) {
398                     log2("        UnsupportedOperationException ");
399                 }
400             }
401 
402             log2("......deleting ClassPrepareRequests");
403             for (int i2 = cpRequests.length -1; i2 >= 0; i2--) {
404                 eventRManager.deleteEventRequest(cpRequests[i2]);
405                 requests = eventRManager.classPrepareRequests();
406                 listSize = requests.size();
407                 if ( listSize != (i2 + 1) ) {
408                     testExitCode = FAILED;
409                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " + (i2 + 1));
410                 }
411                 log2("      removing item from the List; UnsupportedOperationException is expected");
412                 try {
413                     requests.remove(i2);
414                     testExitCode = FAILED;
415                     log3("ERROR: NO exception");
416                 } catch ( UnsupportedOperationException e ) {
417                     log2("        UnsupportedOperationException ");
418                 }
419             }
420 
421             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
422         }
423         log1("    TESTING ENDS");
424         return;
425     }

427     private ThreadReference threadByName(String name)
428                  throws JDITestRuntimeException {
429 
430         List         all = vm.allThreads();
431         ListIterator li  = all.listIterator();
432 
433         for (; li.hasNext(); ) {
434             ThreadReference thread = (ThreadReference) li.next();
435             if (thread.name().equals(name))
436                 return thread;
437         }
438         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
439     }

441    /*
442     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
443     *                                             String, String, String)
444     *
445     * It sets up a breakpoint at given line number within a given method in a given class
446     * for a given thread.
447     *
448     * Return value: BreakpointRequest object  in case of success
449     *
450     * JDITestRuntimeException   in case of an Exception thrown within the method
451     */
452 
453     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
454                                                   ReferenceType testedClass,
455                                                   String methodName,
456                                                   String bpLine,
457                                                   String property)
458             throws JDITestRuntimeException {
459 
460         log2("......setting up a breakpoint:");
461         log2("       thread: " + thread + "; class: " + testedClass +
462                         "; method: " + methodName + "; line: " + bpLine);
463 
464         List              alllineLocations = null;
465         Location          lineLocation     = null;
466         BreakpointRequest breakpRequest    = null;
467 
468         try {
469             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
470 
471             alllineLocations = method.allLineLocations();
472 
473             int n =
474                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
475             if (n > alllineLocations.size()) {
476                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
477             } else {
478                 lineLocation = (Location) alllineLocations.get(n);
479                 try {
480                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
481                     breakpRequest.putProperty("number", property);
482                     breakpRequest.addThreadFilter(thread);
483                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
484                 } catch ( Exception e1 ) {
485                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
486                     breakpRequest    = null;
487                 }
488             }
489         } catch ( Exception e2 ) {
490             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
491             breakpRequest    = null;
492         }
493 
494         if (breakpRequest == null) {
495             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
496             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
497         }
498 
499         log2("      a breakpoint has been set up");
500         return breakpRequest;
501     }

504     private void getEventSet()
505                  throws JDITestRuntimeException {
506         try {
507 //            log2("       eventSet = eventQueue.remove(waitTime);");
508             eventSet = eventQueue.remove(waitTime);
509             if (eventSet == null) {
510                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
511             }
512 //            log2("       eventIterator = eventSet.eventIterator;");
513             eventIterator = eventSet.eventIterator();
514         } catch ( Exception e ) {
515             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
516         }
517     }

520     private void breakpointForCommunication()
521                  throws JDITestRuntimeException {
522         log2("breakpointForCommunication");
523 
524         do {
525             getEventSet();
526 
527             Event event = eventIterator.nextEvent();
528             if (event instanceof BreakpointEvent)
529                 return;
530 
531             log2("      received: " + event);
532 
533             if (EventFilters.filtered(event, debuggeeName)) {
534                 eventSet.resume();
535             }
536             else {
537                 break;
538             }
539         } while (true);
540 
541         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
542     }
