 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Finally blocks are not executed.                          <BR>
 47  * <BR>
 48  * The test has two phases and works as follows.                <BR>
 49  * <BR>
 50  * In first phase,                                                      <BR>
 51  * upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 55  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 56  * and waits for the event within the predefined time interval.         <BR>
 57  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 58  * Upon getting the ClassPrepareEvent,                                  <BR>
 59  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 60  * within debuggee's special methodForCommunication().                  <BR>
 61  * <BR>
 62  * In second phase to check the assetion,                               <BR>
 63  * the debugger and the debuggee perform the following.                 <BR>
 64  * - The debugger resumes the debuggee and waits for the BreakpointEvent<BR>
 65  * in the methodForCommunication().                                     <BR>
 66  * - The debuggee creates second thread, the thread2, which will        <BR>
 67  * call a tested poppedMethod() two times from its run() method.        <BR>
 68  * The poppedMethod will call the special breakpointMethod,             <BR>
 69  * first time from try clause, second time from catch clause.           <BR>
 70  * The finally clause contains assignment of 1 to the variable testVar, <BR>
 71  * whose initial value is 0; hence, if the finally clause is executed,  <BR>
 72  * the value of the testVar is changed as well.                         <BR>
 73  * After creating the thread2, the debuggee invokes                     <BR>
 74  * the methodForCommunication to be suspended and                       <BR>
 75  * to inform the debugger with the event.                               <BR>
 76  * - Upon getting the BreakpointEvent in the methodForCommunication,    <BR>
 77  * the debugger sets up a breakpoint in the breakpointMethod,           <BR>
 78  * resumes the debuggee, waits for BreakpointEvents in the              <BR>
 79  * breakpointMethod and, and upon getting them, invokes popFrames() and <BR>
 80  * checks up on the value of the testVar.                               <BR>
 81  * <BR>
 82  */
 83

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

253     * Return value: 0 - normal end of the test
254     *               1 - ubnormal end of the test
255     *               2 - VMDisconnectedException while test phase
256     */

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if (eventIterator.nextEvent() instanceof VMDeathEvent)
266                 return 0;
267 
268             log3("ERROR: last event is not the VMDeathEvent");
269             return 1;
270         } catch ( VMDisconnectedException e ) {
271             log3("ERROR: VMDisconnectedException : " + e);
272             return 2;
273         } catch ( Exception e ) {
274             log3("ERROR: Exception : " + e);
275             return 1;
276         }
277

280     private void testRun()
281                  throws JDITestRuntimeException, Exception {
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         if ( !vm.canPopFrames() ) {
303             log2("......vm.canPopFrames() == false : test is cancelled");
304             vm.resume();
305             return;
306         }
307 
308         String bPointMethod = "methodForCommunication";
309         String lineForComm  = "lineForComm";
310 
311         log2("......setting BreakpointRequest (bpRequest) in main thread");
312         bpRequest = settingBreakpoint(threadByName("main"),
313                                           debuggeeClass,
314                                           bPointMethod, lineForComm, "zero");
315         log2("bpRequest.enable();");
316         bpRequest.enable();
317 
318     //------------------------------------------------------  testing section
319 
320         log1("     TESTING BEGINS");
321 
322         label0:
323         {
324             vm.resume();
325             breakpointForCommunication();
326 
327             String thread2Name         = "thread2";
328             ThreadReference thread2Ref = threadByName(thread2Name);
329 
330             StackFrame stackFrame = null;
331             int var;
332 
333             String breakpointMethod = "breakpointMethod";
334             String breakpointLine    = "breakpointLine";
335 
336             String testVar = "testVar";
337 
338             log2("......setting breakpoint in breakpointMethod");
339             breakpointRequest2 = settingBreakpoint(thread2Ref, debuggeeClass,
340                                           breakpointMethod, breakpointLine, "one");
341             log2("......breakpointRequest2.enable();");
342             breakpointRequest2.enable();
343 
344             {  // to get mainThread suspended; otherwise its end results in
345                // no suspention for breakpointRequest2 (bug or not?), end of test,
346                // and VMDisconnectedException
347             thread2Ref.suspend();
348             log2("......eventSet.resume();");
349             eventSet.resume();
350             breakpointForCommunication();
351             log2("bpRequest.disable();");
352             bpRequest.disable();
353             thread2Ref.resume();
354             }
355 
356             log2("......breakpointInMethod(breakpointRequest2);");
357             breakpointInMethod("one");
358 
359             log2(" ");
360             log2("......TEST CASE: try clause");
361             log2(" ");
362             log2("......stackFrame = thread2Ref.frame(1);");
363             stackFrame = thread2Ref.frame(1);
364 
365             log2("......thread2Ref.popFrames(stackFrame);");
366             try {
367                 thread2Ref.popFrames(stackFrame);
368             } catch ( IncompatibleThreadStateException e ) {
369                 log3("ERROR: IncompatibleThreadStateException");
370                 testExitCode = FAILED;
371                 break label0;
372             }
373             var = ((IntegerValue) debuggeeClass.getValue(
374                          debuggeeClass.fieldByName(testVar))).value();
375             if  (var != 0) {
376                 log3("ERROR:  testVar != 0");
377                 testExitCode = FAILED;
378             }
379 
380             thread2Ref.resume();
381             log2("......breakpointInMethod(breakpointRequest2);");
382             breakpointInMethod("one");
383 
384             thread2Ref.resume();
385             log2("......breakpointInMethod(breakpointRequest2);");
386             breakpointInMethod("one");
387 
388             log2(" ");
389             log2("......TEST CASE: catch clause");
390             log2(" ");
391 
392             log2("......StackFrame stackFrame = thread2Ref.frame(1);");
393             stackFrame = thread2Ref.frame(1);
394 
395             log2("......thread2Ref.popFrames(stackFrame);");
396             try {
397                 thread2Ref.popFrames(stackFrame);
398             } catch ( IncompatibleThreadStateException e ) {
399                 log3("ERROR: IncompatibleThreadStateException");
400                 testExitCode = FAILED;
401                 break label0;
402             }
403             var = ((IntegerValue) debuggeeClass.getValue(
404                           debuggeeClass.fieldByName(testVar))).value();
405             if  (var != 0) {
406                 log3("ERROR:  testVar != 0");
407                 testExitCode = FAILED;
408             }
409 
410             log2("......breakpointRequest2.disable();");
411             breakpointRequest2.disable();
412 
413             vm.resume();
414 
415             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
416         }
417         log1("    TESTING ENDS");
418         return;
419     }

421     private ThreadReference threadByName(String name)
422                  throws JDITestRuntimeException {
423 
424         List         all = vm.allThreads();
425         ListIterator li  = all.listIterator();
426 
427         for (; li.hasNext(); ) {
428             ThreadReference thread = (ThreadReference) li.next();
429             if (thread.name().equals(name))
430                 return thread;
431         }
432         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
433     }

435    /*
436     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
437     *                                             String, String, String)
438     *
439     * It sets up a breakpoint at given line number within a given method in a given class
440     * for a given thread.
441     *
442     * Return value: BreakpointRequest object  in case of success
443     *
444     * JDITestRuntimeException   in case of an Exception thrown within the method
445     */
446 
447     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
448                                                   ReferenceType testedClass,
449                                                   String methodName,
450                                                   String bpLine,
451                                                   String property)
452             throws JDITestRuntimeException {
453 
454         log2("......setting up a breakpoint:");
455         log2("       thread: " + thread + "; class: " + testedClass +
456                         "; method: " + methodName + "; line: " + bpLine);
457 
458         List              alllineLocations = null;
459         Location          lineLocation     = null;
460         BreakpointRequest breakpRequest    = null;
461 
462         try {
463             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
464 
465             alllineLocations = method.allLineLocations();
466 
467             int n =
468                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
469             if (n > alllineLocations.size()) {
470                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
471             } else {
472                 lineLocation = (Location) alllineLocations.get(n);
473                 try {
474                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
475                     breakpRequest.putProperty("number", property);
476                     breakpRequest.addThreadFilter(thread);
477                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
478                 } catch ( Exception e1 ) {
479                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
480                     breakpRequest    = null;
481                 }
482             }
483         } catch ( Exception e2 ) {
484             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
485             breakpRequest    = null;
486         }
487 
488         if (breakpRequest == null) {
489             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
490             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
491         }
492 
493         log2("      a breakpoint has been set up");
494         return breakpRequest;
495     }

499     private void getEventSet()
500                  throws JDITestRuntimeException {
501         try {
502 //            log2("       eventSet = eventQueue.remove(waitTime);");
503             eventSet = eventQueue.remove(waitTime);
504             if (eventSet == null) {
505                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
506             }
507 //            log2("       eventIterator = eventSet.eventIterator;");
508             eventIterator = eventSet.eventIterator();
509         } catch ( Exception e ) {
510             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
511         }
512     }

515     private void breakpointForCommunication()
516                  throws JDITestRuntimeException {
517 
518         log2("breakpointForCommunication");
519         getEventSet();
520 
521         if (eventIterator.nextEvent() instanceof BreakpointEvent)
522             return;
523 
524         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
525     }

529     private void breakpointInMethod(String number)
530                  throws JDITestRuntimeException {
531 
532         Event event;
533 
534         getEventSet();
535         event = eventIterator.nextEvent();
536 
537         if ( !(event instanceof BreakpointEvent) )
538             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
539 
540         log2("---->: request().getProperty == " +
541              event.request().getProperty("number"));
542 
543         if ( event.request().getProperty("number").equals(number) )
544             return;
545 
546         throw new JDITestRuntimeException("** UNEXPECTED breakpoint **");
547     }
