 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a BreakpointRequest and enables it,                      <BR>
 72  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         if ( !vm.canUseInstanceFilters() ) {
279             log2("......vm.canUseInstanceFilters == false :: test cancelled");
280             vm.exit(PASS_BASE);
281             return;
282         }
283 
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         EventRequest eventRequest1 = null;
319 
320         ThreadReference thread1     = null;
321         String          thread1Name = "thread1";
322 
323         String property1 = "BreakpointRequest1";
324 
325         String methodName = "method";
326         String bpLineName = "breakpointLine";
327 
328         ReferenceType testClassReference = null;
329 
330         String         arrayName = "objTC";
331         ObjectReference instance = null;
332 
333 
334         for (int i = 0; ; i++) {
335 
336             vm.resume();
337             breakpointForCommunication();
338 
339             int instruction = ((IntegerValue)
340                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
341 
342             if (instruction == 0) {
343                 vm.resume();
344                 break;
345             }
346 
347             log1(":::::: case: # " + i);
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351             switch (i) {
352 
353               case 0:
354                       testClassReference =
355                            (ReferenceType) vm.classesByName(testedClassName).get(0);
356 
357                       thread1 = (ThreadReference) debuggeeClass.getValue(
358                                                   debuggeeClass.fieldByName(thread1Name));
359 
360                       eventRequest1 = setting2BreakpointRequest (null,
361                                              testClassReference, methodName, bpLineName,
362                                              EventRequest.SUSPEND_NONE, property1);
363 
364                       instance = (ObjectReference)
365                                  ((ArrayReference) debuggeeClass.getValue(
366                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
367 
368                       eventRequest1.enable();
369 
370                      try {
371                          log2("......eventRequest1.addInstanceFilter(instance);");
372                          log2("        InvalidRequestStateException expected");
373                          ((BreakpointRequest) eventRequest1).addInstanceFilter(instance);
374                          log3("ERROR: no InvalidRequestStateException");
375                          testExitCode = FAILED;
376                      } catch ( InvalidRequestStateException e ) {
377                          log2("        InvalidRequestStateException");
378                      }
379 
380                      log2("......eventRequest1.disable();");
381                      eventRequest1.disable();
382                      try {
383                          log2("......eventRequest1.addInstanceFilter(instance);");
384                          log2("        no InvalidRequestStateException expected");
385                          ((BreakpointRequest) eventRequest1).addInstanceFilter(instance);
386                          log2("        no InvalidRequestStateException");
387                      } catch ( InvalidRequestStateException e ) {
388                          log3("ERROR: InvalidRequestStateException");
389                          testExitCode = FAILED;
390                      }
391 
392                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
393                      eventRManager.deleteEventRequest(eventRequest1);
394                      try {
395                          log2("......eventRequest1.addInstanceFilter(instance);");
396                          log2("        InvalidRequestStateException expected");
397                          ((BreakpointRequest) eventRequest1).addInstanceFilter(instance);
398                          log3("ERROR: no InvalidRequestStateException");
399                          testExitCode = FAILED;
400                      } catch ( InvalidRequestStateException e ) {
401                          log2("        InvalidRequestStateException");
402                      }
403 
404                       break;
405 
406               default:
407                       throw new JDITestRuntimeException("** default case 2 **");
408             }
409 
410             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
411         }
412         log1("    TESTING ENDS");
413         return;
414     }

416    /*
417     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
418     *                                             String, String, String)
419     *
420     * It sets up a breakpoint at given line number within a given method in a given class
421     * for a given thread.
422     *
423     * Return value: BreakpointRequest object  in case of success
424     *
425     * JDITestRuntimeException   in case of an Exception thrown within the method
426     */
427 
428     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
429                                                   ReferenceType testedClass,
430                                                   String methodName,
431                                                   String bpLine,
432                                                   String property)
433             throws JDITestRuntimeException {
434 
435         log2("......setting up a breakpoint:");
436         log2("       thread: " + thread + "; class: " + testedClass +
437                         "; method: " + methodName + "; line: " + bpLine);
438 
439         List              alllineLocations = null;
440         Location          lineLocation     = null;
441         BreakpointRequest breakpRequest    = null;
442 
443         try {
444             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
445 
446             alllineLocations = method.allLineLocations();
447 
448             int n =
449                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
450             if (n > alllineLocations.size()) {
451                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
452             } else {
453                 lineLocation = (Location) alllineLocations.get(n);
454                 try {
455                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
456                     breakpRequest.putProperty("number", property);
457                     breakpRequest.addThreadFilter(thread);
458                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
459                 } catch ( Exception e1 ) {
460                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
461                     breakpRequest    = null;
462                 }
463             }
464         } catch ( Exception e2 ) {
465             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
466             breakpRequest    = null;
467         }
468 
469         if (breakpRequest == null) {
470             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
471             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
472         }
473 
474         log2("      a breakpoint has been set up");
475         return breakpRequest;
476     }

479     private void getEventSet()
480                  throws JDITestRuntimeException {
481         try {
482 //            log2("       eventSet = eventQueue.remove(waitTime);");
483             eventSet = eventQueue.remove(waitTime);
484             if (eventSet == null) {
485                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
486             }
487 //            log2("       eventIterator = eventSet.eventIterator;");
488             eventIterator = eventSet.eventIterator();
489         } catch ( Exception e ) {
490             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
491         }
492     }

495     private void breakpointForCommunication()
496                  throws JDITestRuntimeException {
497 
498         log2("breakpointForCommunication");
499         getEventSet();
500 
501         if (eventIterator.nextEvent() instanceof BreakpointEvent)
502             return;
503 
504         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
505     }

509     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
510                                                           ReferenceType   testedClass,
511                                                           String          methodName,
512                                                           String          bpLine,
513                                                           int             suspendPolicy,
514                                                           String          property        )
515             throws JDITestRuntimeException {
516         try {
517             log2("......setting up a breakpoint:");
518             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
519 
520             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
521             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
522 
523             BreakpointRequest
524             bpr = eventRManager.createBreakpointRequest(loc);
525             bpr.putProperty("number", property);
526             if (thread != null)
527                 bpr.addThreadFilter(thread);
528             bpr.setSuspendPolicy(suspendPolicy);
529 
530             log2("      a breakpoint has been set up");
531             return bpr;
532         } catch ( Exception e ) {
533             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
534             log3("       BreakpointRequest HAS NOT BEEN SET UP");
535             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
536         }
537     }
