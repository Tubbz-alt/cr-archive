 23 package nsk.jvmti.ResourceExhausted;
 24 
 25 import java.io.File;
 26 import java.io.FileInputStream;
 27 import java.io.PrintStream;
 28 import java.security.ProtectionDomain;


 29 
 30 import nsk.share.Consts;
 31 import nsk.share.test.Stresser;
 32 
 33 public class resexhausted003 {
 34 
 35     static final long MAX_ITERATIONS = Long.MAX_VALUE;
 36 
 37     static class MyClassLoader extends ClassLoader {
 38         Class<?> loadUm(String name, byte[] bytes, ProtectionDomain pd) throws ClassFormatError {
 39             Class k = defineClass(name, bytes, 0, bytes.length, pd);
 40             resolveClass(k);
 41             return k;
 42         }
 43 
 44         Class loadClass(String name, byte[] bytes) {
 45             try {
 46                 Class k = loadUm(name, bytes, getClass().getProtectionDomain());
 47                 k.newInstance(); // prepare class
 48                 return k;
 49             } catch (Exception exc) {
 50                 throw new RuntimeException("Exception in loadClass: " + name, exc);
 51             }
 52         }
 53     };
 54 
 55     static class Node {
 56         MyClassLoader loader = new MyClassLoader();
 57         Node next;
 58     };
 59 
 60 
 61     static byte[] fileBytes(String fileName) {
 62         try {
 63             File f = new File(fileName);
 64             FileInputStream fi = new FileInputStream(f);
 65             byte[] bytes;
 66             try {
 67                 bytes = new byte[(int) f.length()];
 68                 fi.read(bytes);
 69             } finally {
 70                 fi.close();
 71             }
 72             return bytes;
 73         } catch (Exception e) {
 74             throw new RuntimeException("Exception when reading file '" + fileName + "'", e);
 75         }
 76     }
 77 
 78 
 79     public static int run(String args[], PrintStream out) {
 80         if ( args == null || args.length < 1 ) {
 81             System.err.println("TEST BUG: Classes directory should be the first argument. Check .cfg file.");










 82             return Consts.TEST_FAILED;
 83         }
 84 
 85         String classesDir = args[0];
 86         Stresser stress = new Stresser(args);
 87 
 88         String className = Helper.class.getName();
 89         byte[] bloatBytes = fileBytes(classesDir + File.separator + className.replace('.', '/') + ".class");
 90 
 91         int count = 0;
 92         Helper.resetExhaustedEvent();
 93 
 94         out.println("Loading classes...");
 95         stress.start(MAX_ITERATIONS);
 96         try {
 97             Node list = null;
 98 
 99             while ( stress.iteration() ) {
100                 Node n = new Node();
101                 n.next = list;
102                 list = n;
103                 n.loader.loadClass(className, bloatBytes);
104                 ++count;
105             }
106 
107             System.out.println("Can't reproduce OOME due to a limit on iterations/execution time. Test was useless.");
108             return Consts.TEST_PASSED;
109 
110         } catch (OutOfMemoryError e) {
111             // that is what we are waiting for
112         } finally {
113             stress.finish();
114         }
115 
116         System.gc();
117         if ( ! Helper.checkResult("loading " + count + " classes of " + bloatBytes.length + " bytes") )
118             return Consts.TEST_FAILED;
119 
120         return Consts.TEST_PASSED;
121     }
122 
123     public static void main(String[] args) {
124         args = nsk.share.jvmti.JVMTITest.commonInit(args);
125 
126         int result = run(args, System.out);
127         System.out.println(result == Consts.TEST_PASSED ? "TEST PASSED" : "TEST FAILED");
128         System.exit(result + Consts.JCK_STATUS_BASE);
129     }
130 }
