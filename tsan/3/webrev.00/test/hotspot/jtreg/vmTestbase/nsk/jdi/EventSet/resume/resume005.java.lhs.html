 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for ExceptionEvent sets.              <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a ExceptionRequest, resumes                   <BR>
 74  *   the debuggee, and waits for the ExceptionEvent.                    <BR>
 75  * - The debuggee invokes the special method which makes access to      <BR>
 76  *   a predefined variable to be resulting in the event.                <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

169     private int runThis (String argv[], PrintStream out) {
170 
171         argsHandler     = new ArgumentHandler(argv);
172         logHandler      = new Log(out, argsHandler);
173         Binder binder   = new Binder(argsHandler, logHandler);
174 
175         waitTime        = argsHandler.getWaitTime() * 60000;
176 
177         try {
178             log2("launching a debuggee :");
179             log2("       " + debuggeeName);
180             if (argsHandler.verbose()) {
181                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
182             } else {
183                 debuggee = binder.bindToDebugee(debuggeeName);
184             }
185             if (debuggee == null) {
186                 log3("ERROR: no debuggee launched");
187                 return FAILED;
188             }
189             log2("debuggee launched");
190         } catch ( Exception e ) {
191             log3("ERROR: Exception : " + e);
192             log2("       test cancelled");
193             return FAILED;
194         }
195 
196         debuggee.redirectOutput(logHandler);
197 
198         vm = debuggee.VM();
199 
200         eventQueue = vm.eventQueue();
201         if (eventQueue == null) {
202             log3("ERROR: eventQueue == null : TEST ABORTED");
203             vm.exit(PASS_BASE);
204             return FAILED;
205         }
206 
207         log2("invocation of the method runTest()");
208         switch (runTest()) {
209 
210             case 0 :  log2("test phase has finished normally");
211                       log2("   waiting for the debuggee to finish ...");
212                       debuggee.waitFor();
213 
214                       log2("......getting the debuggee's exit status");
215                       int status = debuggee.getStatus();
216                       if (status != PASS_BASE) {
217                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
218                               status + " != PASS_BASE");
219                           testExitCode = FAILED;
220                       } else {
221                           log2("......debuggee returned expected exit status: " +
222                               status + " == PASS_BASE");
223                       }
224                       break;
225 
226             default : log3("ERROR: runTest() returned unexpected value");
227 
228             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
229                       log2("......forcing: vm.exit();");
230                       testExitCode = FAILED;
231                       try {
232                           vm.exit(PASS_BASE);
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237 
238             case 2 :  log3("test cancelled due to VMDisconnectedException");
239                       log2("......trying: vm.process().destroy();");
240                       testExitCode = FAILED;
241                       try {
242                           Process vmProcess = vm.process();
243                           if (vmProcess != null) {
244                               vmProcess.destroy();
245                           }
246                       } catch ( Exception e ) {
247                           log3("ERROR: Exception : e");
248                       }
249                       break;
250             }
251 
252         return testExitCode;
253     }

262     private int runTest() {
263 
264         try {
265             testRun();
266 
267             log2("waiting for VMDeathEvent");
268             getEventSet();
269             if (eventIterator.nextEvent() instanceof VMDeathEvent)
270                 return 0;
271 
272             log3("ERROR: last event is not the VMDeathEvent");
273             return 1;
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         eventRManager = vm.eventRequestManager();
288 
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292 
293         cpRequest.enable();
294         vm.resume();
295         getEventSet();
296         cpRequest.disable();
297 
298         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
299         debuggeeClass = event.referenceType();
300 
301         if (!debuggeeClass.name().equals(debuggeeName))
302            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
303 
304         log2("      received: ClassPrepareEvent for debuggeeClass");
305 
306         String bPointMethod = "methodForCommunication";
307         String lineForComm  = "lineForComm";
308         BreakpointRequest bpRequest;
309         ThreadReference   mainThread = threadByName("main");
310         bpRequest = settingBreakpoint(mainThread,
311                                       debuggeeClass,
312                                       bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315         vm.resume();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         EventRequest eventRequest1 = null;
322         EventRequest eventRequest2 = null;
323         EventRequest eventRequest3 = null;
324 
325         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
327         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
328 
329         String methodName = "method";
330         String bpLineName = "breakpointLine";
331 
332         ReferenceType testClassReference = null;
333 
334 
335         for (int i = 0; ; i++) {
336 
337             breakpointForCommunication();
338 
339             int instruction = ((IntegerValue)
340                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
341 
342             if (instruction == 0) {
343                 vm.resume();
344                 break;
345             }
346 
347             log1(":::::: case: # " + i);
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351             switch (i) {
352 
353               case 0:
354                       testClassReference =
355                            (ReferenceType) vm.classesByName(testedClassName).get(0);
356 
357                       eventRequest1 = settingExceptionRequest (mainThread,
358                                              debuggeeClass, // testClassReference,
359                                              SUSPEND_NONE, "ExceptionRequest1");
360                       eventRequest1.enable();
361                       break;
362 
363               case 1:
364                       eventRequest2 = settingExceptionRequest (mainThread,
365                                              debuggeeClass, // testClassReference,
366                                              SUSPEND_THREAD, "ExceptionRequest2");
367                       eventRequest2.enable();
368                       break;
369 
370               case 2:
371                       eventRequest3 = settingExceptionRequest (mainThread,
372                                              debuggeeClass, // testClassReference,
373                                              SUSPEND_ALL, "ExceptionRequest3");
374                       eventRequest3.enable();
375                       break;
376 
377 
378               default:
379                       throw new JDITestRuntimeException("** default case 2 **");
380             }
381 
382             log2("......waiting for new ExceptionEvent : " + i);
383             mainThread.resume();
384             getEventSet();
385 
386             Event newEvent = eventIterator.nextEvent();
387             if ( !(newEvent instanceof ExceptionEvent)) {
388                  log3("ERROR: new event is not ExceptionEvent");
389                  testExitCode = FAILED;
390             } else {
391 
392                 String property = (String) newEvent.request().getProperty("number");
393                 log2("       got new ExceptionEvent with propety 'number' == " + property);
394 
395                 log2("......checking up on EventSet.resume()");
396                 log2("......--> vm.suspend();");
397                 vm.suspend();
398 
399                 log2("        getting : Map<String, Integer> suspendsCounts1");
400 
401                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
402                 for (ThreadReference threadReference : vm.allThreads()) {
403                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
404                 }
405 
406                 log2("        eventSet.resume;");
407                 eventSet.resume();
408 
409                 log2("        getting : Map<String, Integer> suspendsCounts2");
410                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
411                 for (ThreadReference threadReference : vm.allThreads()) {
412                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
413                 }
414 
415                 log2(suspendsCounts1.toString());
416                 log2(suspendsCounts2.toString());
417 
418                 log2("        getting : int policy = eventSet.suspendPolicy();");
419                 int policy = eventSet.suspendPolicy();
420 
421                 switch (policy) {
422 
423                   case SUSPEND_NONE   :
424                        log2("        case SUSPEND_NONE");
425                        for (String threadName : suspendsCounts1.keySet()) {
426                            log2("        checking " + threadName);
427                            if (!suspendsCounts2.containsKey(threadName)) {
428                                log3("ERROR: couldn't get ThreadReference for " + threadName);
429                                testExitCode = FAILED;
430                                break;
431                            }
432                            int count1 = suspendsCounts1.get(threadName);
433                            int count2 = suspendsCounts2.get(threadName);
434                            if (count1 != count2) {
435                                log3("ERROR: suspendCounts don't match for : " + threadName);
436                                log3("before resuming : " + count1);
437                                log3("after  resuming : " + count2);
438                                testExitCode = FAILED;
439                                break;
440                            }
441                        }
442                        eventRequest1.disable();
443                        break;
444 
445                   case SUSPEND_THREAD :
446                        log2("        case SUSPEND_THREAD");
447                        for (String threadName : suspendsCounts1.keySet()) {
448                            log2("checking " + threadName);
449                            if (!suspendsCounts2.containsKey(threadName)) {
450                                log3("ERROR: couldn't get ThreadReference for " + threadName);
451                                testExitCode = FAILED;
452                                break;
453                            }
454                            int count1 = suspendsCounts1.get(threadName);
455                            int count2 = suspendsCounts2.get(threadName);
456                            String eventThreadName = ((ExceptionEvent)newEvent).thread().name();
457                            int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
458                            if (count1 != expectedValue) {
459                                log3("ERROR: suspendCounts don't match for : " + threadName);
460                                log3("before resuming : " + count1);
461                                log3("after  resuming : " + count2);
462                                testExitCode = FAILED;
463                                break;
464                            }
465                        }
466                        eventRequest2.disable();
467                        break;
468 
469                   case SUSPEND_ALL    :
470 
471                         log2("        case SUSPEND_ALL");
472                         for (String threadName : suspendsCounts1.keySet()) {
473                             log2("checking " + threadName);
474                             if (!event.request().equals(eventRequest3))
475                                 break;
476                             if (!suspendsCounts2.containsKey(threadName)) {
477                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
478                                 testExitCode = FAILED;
479                                 break;
480                             }
481                             int count1 = suspendsCounts1.get(threadName);
482                             int count2 = suspendsCounts2.get(threadName);
483                             if (count1 != count2 + 1) {
484                                 log3("ERROR: suspendCounts don't match for : " + threadName);
485                                 log3("before resuming : " + count1);
486                                 log3("after  resuming : " + count2);
487                                 testExitCode = FAILED;
488                                 break;
489                             }
490                         }
491                         eventRequest3.disable();
492                         break;
493 
494                   default: throw new JDITestRuntimeException("** default case 1 **");
495                 }
496                 informDebuggeeTestCase(i);
497             }
498 
499             log2("......--> vm.resume()");
500             vm.resume();
501             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
502         }
503         log1("    TESTING ENDS");
504         return;
505     }

507     private ThreadReference threadByName(String name)
508                  throws JDITestRuntimeException {
509 
510         List         all = vm.allThreads();
511         ListIterator li  = all.listIterator();
512 
513         for (; li.hasNext(); ) {
514             ThreadReference thread = (ThreadReference) li.next();
515             if (thread.name().equals(name))
516                 return thread;
517         }
518         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
519     }

521    /*
522     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
523     *                                             String, String, String)
524     *
525     * It sets up a breakpoint at given line number within a given method in a given class
526     * for a given thread.
527     *
528     * Return value: BreakpointRequest object  in case of success
529     *
530     * JDITestRuntimeException   in case of an Exception thrown within the method
531     */
532 
533     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
534                                                   ReferenceType testedClass,
535                                                   String methodName,
536                                                   String bpLine,
537                                                   String property)
538             throws JDITestRuntimeException {
539 
540         log2("......setting up a breakpoint:");
541         log2("       thread: " + thread + "; class: " + testedClass +
542                         "; method: " + methodName + "; line: " + bpLine);
543 
544         List              alllineLocations = null;
545         Location          lineLocation     = null;
546         BreakpointRequest breakpRequest    = null;
547 
548         try {
549             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
550 
551             alllineLocations = method.allLineLocations();
552 
553             int n =
554                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
555             if (n > alllineLocations.size()) {
556                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
557             } else {
558                 lineLocation = (Location) alllineLocations.get(n);
559                 try {
560                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
561                     breakpRequest.putProperty("number", property);
562                     breakpRequest.addThreadFilter(thread);
563                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
564                 } catch ( Exception e1 ) {
565                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
566                     breakpRequest    = null;
567                 }
568             }
569         } catch ( Exception e2 ) {
570             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
571             breakpRequest    = null;
572         }
573 
574         if (breakpRequest == null) {
575             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
576             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
577         }
578 
579         log2("      a breakpoint has been set up");
580         return breakpRequest;
581     }

584     private void getEventSet()
585                  throws JDITestRuntimeException {
586         try {
587 //            log2("       eventSet = eventQueue.remove(waitTime);");
588             eventSet = eventQueue.remove(waitTime);
589             if (eventSet == null) {
590                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
591             }
592 //            log2("       eventIterator = eventSet.eventIterator;");
593             eventIterator = eventSet.eventIterator();
594         } catch ( Exception e ) {
595             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
596         }
597     }

600     private void breakpointForCommunication()
601                  throws JDITestRuntimeException {
602 
603         log2("breakpointForCommunication");
604         getEventSet();
605 
606         if (eventIterator.nextEvent() instanceof BreakpointEvent)
607             return;
608 
609         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
610     }

614     private ExceptionRequest settingExceptionRequest ( ThreadReference thread,
615                                                        ReferenceType   testedClass,
616                                                        int             suspendPolicy,
617                                                        String          property       )
618             throws JDITestRuntimeException {
619         try {
620             log2("......setting up ExceptionRequest:");
621             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
622 
623             ExceptionRequest
624             excr = eventRManager.createExceptionRequest(null, true, true);
625             excr.putProperty("number", property);
626             excr.addThreadFilter(thread);
627             excr.addClassFilter(testedClass);
628             excr.setSuspendPolicy(suspendPolicy);
629 
630             log2("      ExceptionRequest has been set up");
631             return excr;
632         } catch ( Exception e ) {
633             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
634             log3("       ExceptionRequest HAS NOT BEEN SET UP");
635             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
636         }

638     /**
639      * Inform debuggee which thread test the debugger has completed.
640      * Used for synchronization, so the debuggee does not move too quickly.
641      * @param testCase index of just completed test
642      */
643     void informDebuggeeTestCase(int testCase) {
644         try {
645             ((ClassType)debuggeeClass)
646                 .setValue(debuggeeClass.fieldByName("testCase"),
647                           vm.mirrorOf(testCase));
648         } catch (InvalidTypeException ite) {
649             throw new Failure("** FAILURE setting testCase  **");
650         } catch (ClassNotLoadedException cnle) {
651             throw new Failure("** FAILURE notifying debuggee  **");
652         } catch (VMDisconnectedException e) {
653             throw new Failure("** FAILURE debuggee connection **");
654         }
655     }
