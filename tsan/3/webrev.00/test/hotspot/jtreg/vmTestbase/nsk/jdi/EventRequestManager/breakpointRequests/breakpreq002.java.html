 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.breakpointRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled breakpoint requests.          <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

154     private int runThis (String argv[], PrintStream out) {
155 
156         argsHandler     = new ArgumentHandler(argv);
157         logHandler      = new Log(out, argsHandler);
158         Binder binder   = new Binder(argsHandler, logHandler);
159 
160         waitTime        = argsHandler.getWaitTime() * 60000;
161 
162         try {
163             log2("launching a debuggee :");
164             log2("       " + debuggeeName);
165             if (argsHandler.verbose()) {
166                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
167             } else {
168                 debuggee = binder.bindToDebugee(debuggeeName);
169             }
170             if (debuggee == null) {
171                 log3("ERROR: no debuggee launched");
172                 return FAILED;
173             }
174             log2("debuggee launched");
175         } catch ( Exception e ) {
176             log3("ERROR: Exception : " + e);
177             log2("       test cancelled");
178             return FAILED;
179         }
180 
181         debuggee.redirectOutput(logHandler);
182 
183         vm = debuggee.VM();
184 
185         eventQueue = vm.eventQueue();
186         if (eventQueue == null) {
187             log3("ERROR: eventQueue == null : TEST ABORTED");
188             vm.exit(PASS_BASE);
189             return FAILED;
190         }
191 
192         log2("invocation of the method runTest()");
193         switch (runTest()) {
194 
195             case 0 :  log2("test phase has finished normally");
196                       log2("   waiting for the debuggee to finish ...");
197                       debuggee.waitFor();
198 
199                       log2("......getting the debuggee's exit status");
200                       int status = debuggee.getStatus();
201                       if (status != PASS_BASE) {
202                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
203                               status + " != PASS_BASE");
204                           testExitCode = FAILED;
205                       } else {
206                           log2("......debuggee returned expected exit status: " +
207                               status + " == PASS_BASE");
208                       }
209                       break;
210 
211             default : log3("ERROR: runTest() returned unexpected value");
212 
213             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
214                       log2("......forcing: vm.exit();");
215                       testExitCode = FAILED;
216                       try {
217                           vm.exit(PASS_BASE);
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : " + e);
220                       }
221                       break;
222 
223             case 2 :  log3("test cancelled due to VMDisconnectedException");
224                       log2("......trying: vm.process().destroy();");
225                       testExitCode = FAILED;
226                       try {
227                           Process vmProcess = vm.process();
228                           if (vmProcess != null) {
229                               vmProcess.destroy();
230                           }
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235             }
236 
237         return testExitCode;
238     }

243     *               1 - ubnormal end of the test
244     *               2 - VMDisconnectedException while test phase

247     private int runTest() {
248 
249         try {
250             testRun();
251 
252             log2("waiting for VMDeathEvent");
253             getEventSet();
254             if (eventIterator.nextEvent() instanceof VMDeathEvent)
255                 return 0;
256 
257             log3("ERROR: last event is not the VMDeathEvent");
258             return 1;
259         } catch ( VMDisconnectedException e ) {
260             log3("ERROR: VMDisconnectedException : " + e);
261             return 2;
262         } catch ( Exception e ) {
263             log3("ERROR: Exception : " + e);
264             return 1;
265         }
266 
267     }

269     private void testRun()
270                  throws JDITestRuntimeException, Exception {
271 
272         eventRManager = vm.eventRequestManager();
273 
274         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
275         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
276         cpRequest.addClassFilter(debuggeeName);
277 
278         cpRequest.enable();
279         vm.resume();
280         getEventSet();
281         cpRequest.disable();
282 
283         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
284         debuggeeClass = event.referenceType();
285 
286         if (!debuggeeClass.name().equals(debuggeeName))
287            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
288 
289         log2("      received: ClassPrepareEvent for debuggeeClass");
290 
291         String bPointMethod = "methodForCommunication";
292         String lineForComm  = "lineForComm";
293 
294         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
295 
296         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
297                                              debuggeeClass,
298                                             bPointMethod, lineForComm, "zero");
299         bpRequest.enable();
300 
301     //------------------------------------------------------  testing section
302 
303         List         requests = null;
304         ListIterator li       = null;
305 
306         BreakpointRequest request = null;
307         BreakpointRequest bpRequests[] = { null, null, null, null, null,
308                                            null, null, null, null, null };
309         int listSize;
310         int flag;
311 
312 
313         log1("     TESTING BEGINS");
314 
315         for (int i = 0; ; i++) {
316 
317             vm.resume();
318             breakpointForCommunication();
319 
320             int instruction = ((IntegerValue)
321                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
322 
323             if (instruction == 0) {
324                 vm.resume();
325                 break;
326             }
327 
328             log1(":::::: case: # " + i);
329 
330             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
331 
332             log2("......creating BreakpointRequests");
333             for (int i1 = 0; i1 < bpRequests.length; i1++) {
334                 bpRequests[i1] = eventRManager.createBreakpointRequest(breakpLocation);
335                 bpRequests[i1].putProperty("number", String.valueOf(i1));
336 
337                 log2("......checking up on returned List after creating new BreakpointRequest");
338                 requests = eventRManager.breakpointRequests();
339                 listSize = requests.size();
340                 if ( listSize != (i1 + 2) ) {
341                     testExitCode = FAILED;
342                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 2));
343                 }
344                 flag = 0;
345                 li = requests.listIterator();
346                 while (li.hasNext()) {
347                     request = (BreakpointRequest) li.next();
348                     if ( request.getProperty("number").equals("zero") )
349                         continue;
350                     if ( !request.isEnabled() ) {
351                         flag++;
352                         if (flag > 1) {
353                             testExitCode = FAILED;
354                             log3("ERROR: # of disabled requests > 1 : " + flag);
355                         }
356                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
357                             testExitCode = FAILED;
358                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
359                         }
360                     } else {
361                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
362                             testExitCode = FAILED;
363                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
364                         }
365                     }
366                 }
367 
368                 log2("      enabling created BreakpointRequest");
369                 bpRequests[i1].enable();
370                 requests = eventRManager.breakpointRequests();
371                 listSize = requests.size();
372                 if ( listSize != (i1 + 2) ) {
373                     testExitCode = FAILED;
374                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 2));
375                 }
376 
377                 li = requests.listIterator();
378                 while (li.hasNext()) {
379                     request = (BreakpointRequest) li.next();
380                     if ( !request.isEnabled() ) {
381                         testExitCode = FAILED;
382                         log3("ERROR: returned List contains disabled request : " + request);
383                     }
384                 }
385 
386                 log2("      removing item from the List; UnsupportedOperationException is expected");
387                 try {
388                     requests.remove(i1);
389                     testExitCode = FAILED;
390                     log3("ERROR: NO exception");
391                 } catch ( UnsupportedOperationException e ) {
392                     log2("        UnsupportedOperationException ");
393                 }
394             }
395 
396             log2("......deleting BreakpointRequests");
397             for (int i2 = bpRequests.length -1; i2 >= 0; i2--) {
398                 eventRManager.deleteEventRequest(bpRequests[i2]);
399                 requests = eventRManager.breakpointRequests();
400                 listSize = requests.size();
401                 if ( listSize != (i2 + 1) ) {
402                     testExitCode = FAILED;
403                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i2 + 1));
404                 }
405                 log2("      removing item from the List; UnsupportedOperationException is expected");
406                 try {
407                     requests.remove(i2);
408                     testExitCode = FAILED;
409                     log3("ERROR: NO exception");
410                 } catch ( UnsupportedOperationException e ) {
411                     log2("        UnsupportedOperationException ");
412                 }
413             }
414 
415             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
416         }
417         log1("    TESTING ENDS");
418         return;
419     }

421    /*
422     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
423     *                                             String, String, String)
424     *
425     * It sets up a breakpoint at given line number within a given method in a given class
426     * for a given thread.
427     *
428     * Return value: BreakpointRequest object  in case of success
429     *
430     * JDITestRuntimeException   in case of an Exception thrown within the method
431     */
432 
433     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
434                                                   ReferenceType testedClass,
435                                                   String methodName,
436                                                   String bpLine,
437                                                   String property)
438             throws JDITestRuntimeException {
439 
440         log2("......setting up a breakpoint:");
441         log2("       thread: " + thread + "; class: " + testedClass +
442                         "; method: " + methodName + "; line: " + bpLine);
443 
444         List              alllineLocations = null;
445         Location          lineLocation     = null;
446         BreakpointRequest breakpRequest    = null;
447 
448         try {
449             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
450 
451             alllineLocations = method.allLineLocations();
452 
453             int n =
454                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
455             if (n > alllineLocations.size()) {
456                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
457             } else {
458                 lineLocation = (Location) alllineLocations.get(n);
459 breakpLocation = lineLocation;
460                 try {
461                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
462                     breakpRequest.putProperty("number", property);
463                     breakpRequest.addThreadFilter(thread);
464                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
465                 } catch ( Exception e1 ) {
466                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
467                     breakpRequest    = null;
468                 }
469             }
470         } catch ( Exception e2 ) {
471             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
472             breakpRequest    = null;
473         }
474 
475         if (breakpRequest == null) {
476             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
477             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
478         }
479 
480         log2("      a breakpoint has been set up");
481         return breakpRequest;
482     }

485     private void getEventSet()
486                  throws JDITestRuntimeException {
487         try {
488 //            log2("       eventSet = eventQueue.remove(waitTime);");
489             eventSet = eventQueue.remove(waitTime);
490             if (eventSet == null) {
491                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
492             }
493 //            log2("       eventIterator = eventSet.eventIterator;");
494             eventIterator = eventSet.eventIterator();
495         } catch ( Exception e ) {
496             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
497         }
498     }

501     private void breakpointForCommunication()
502                  throws JDITestRuntimeException {
503 
504         log2("breakpointForCommunication");
505         getEventSet();
506 
507         if (eventIterator.nextEvent() instanceof BreakpointEvent)
508             return;
509 
510         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
511     }
