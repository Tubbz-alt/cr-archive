398     private ThreadReference threadByName(String name)
399                  throws JDITestRuntimeException {
400 
401         List         all = vm.allThreads();
402         ListIterator li  = all.listIterator();
403 
404         for (; li.hasNext(); ) {
405             ThreadReference thread = (ThreadReference) li.next();
406             if (thread.name().equals(name))
407                 return thread;
408         }
409         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
