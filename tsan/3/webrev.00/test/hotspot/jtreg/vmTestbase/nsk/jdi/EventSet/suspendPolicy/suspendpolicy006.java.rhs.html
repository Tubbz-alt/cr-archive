 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a MethodEntryEvent set,             <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a MethodEntryRequest, resumes                 <BR>
 70  *   the debuggee, and waits for the MethodEntryEvent.                  <BR>
 71  * - The debuggee invokes the special method to be resulting in the event.<BR>
 72  * - Upon getting new event, the debugger                               <BR>
 73  *   performs the check corresponding to the event.                     <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * In third phase, at the end,                                          <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  */
 82

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : e");
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : e");
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
259                 log3("ERROR: last event is not the VMDeathEvent");
260                 return 1;
261             }
262 
263             log2("waiting for VMDisconnectEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
266                 log3("ERROR: last event is not the VMDisconnectEvent");
267                 return 1;
268             }
269 
270             return 0;
271 
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279 
280     }

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         log2("......getting ClassPrepareEvent for debuggee's class");
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
304 
305         String            bPointMethod = "methodForCommunication";
306         String            lineForComm  = "lineForComm";
307         BreakpointRequest bpRequest;
308         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
309         bpRequest = settingBreakpoint(mainThread,
310                                       debuggeeClass,
311                                       bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314         vm.resume();
315 
316     //------------------------------------------------------  testing section
317 
318         log1("     TESTING BEGINS");
319 
320         EventRequest eventRequest1 = null;
321         EventRequest eventRequest2 = null;
322         EventRequest eventRequest3 = null;
323 
324         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
325         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
327         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
328 
329         int policyExpected[] = {    SUSPEND_NONE,
330                                     SUSPEND_THREAD,
331                                     SUSPEND_ALL,
332                                     SUSPEND_THREAD,
333                                     SUSPEND_ALL,
334                                     SUSPEND_ALL,
335                                     SUSPEND_ALL      };
336         int policy = 0;
337 
338 
339         ReferenceType testClassReference = null;
340 
341         String fieldName = "var1";
342 
343 
344         for (int i = 0; ; i++) {
345 
346 
347             breakpointForCommunication();
348 
349             int instruction = ((IntegerValue)
350                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
351             if (instruction == 0) {
352                 vm.resume();
353                 break;
354             }
355 
356 
357             log1(":::::: case: # " + i);
358 
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
360 
361             switch (i) {
362 
363               case 0:
364                       testClassReference =
365                            (ReferenceType) vm.classesByName(testedClassName).get(0);
366 
367                       eventRequest1 = settingMethodEntryRequest (mainThread,
368                                              testClassReference,
369                                              SUSPEND_NONE, "MethodEntryRequest1");
370                       eventRequest1.enable();
371                       break;
372 
373               case 1:
374                       eventRequest2 = settingMethodEntryRequest (mainThread,
375                                              testClassReference,
376                                              SUSPEND_THREAD, "MethodEntryRequest2");
377                       eventRequest2.enable();
378                       break;
379 
380               case 2:
381                       eventRequest3 = settingMethodEntryRequest (mainThread,
382                                              testClassReference,
383                                              SUSPEND_ALL, "MethodEntryRequest3");
384                       eventRequest3.enable();
385                       break;
386 
387               case 3:
388                       eventRequest1 = settingMethodEntryRequest (mainThread,
389                                              testClassReference,
390                                              SUSPEND_NONE, "MethodEntryRequest4");
391                       eventRequest1.enable();
392                       eventRequest2 = settingMethodEntryRequest (mainThread,
393                                              testClassReference,
394                                              SUSPEND_THREAD, "MethodEntryRequest5");
395                       eventRequest2.enable();
396                       break;
397 
398               case 4:
399                       eventRequest1 = settingMethodEntryRequest (mainThread,
400                                              testClassReference,
401                                              SUSPEND_NONE, "MethodEntryRequest6");
402                       eventRequest1.enable();
403                       eventRequest3 = settingMethodEntryRequest (mainThread,
404                                              testClassReference,
405                                              SUSPEND_ALL, "MethodEntryRequest7");
406                       eventRequest3.enable();
407                       break;
408 
409               case 5:
410                       eventRequest2 = settingMethodEntryRequest (mainThread,
411                                              testClassReference,
412                                              SUSPEND_THREAD, "MethodEntryRequest8");
413                       eventRequest2.enable();
414                       eventRequest3 = settingMethodEntryRequest (mainThread,
415                                              testClassReference,
416                                              SUSPEND_ALL, "MethodEntryRequest9");
417                       eventRequest3.enable();
418                       break;
419 
420               case 6:
421                       eventRequest1 = settingMethodEntryRequest (mainThread,
422                                              testClassReference,
423                                              SUSPEND_NONE, "MethodEntryRequest10");
424                       eventRequest1.enable();
425                       eventRequest2 = settingMethodEntryRequest (mainThread,
426                                              testClassReference,
427                                              SUSPEND_THREAD, "MethodEntryRequest11");
428                       eventRequest2.enable();
429                       eventRequest3 = settingMethodEntryRequest (mainThread,
430                                              testClassReference,
431                                              SUSPEND_ALL, "MethodEntryRequest12");
432                       eventRequest3.enable();
433                       break;
434 
435 
436               default:
437                       throw new JDITestRuntimeException("** default case 2 **");
438             }
439 
440             log2("......waiting for new MethodEntryEvent : " + i);
441             mainThread.resume();
442             getEventSet();
443 
444             Event newEvent = eventIterator.nextEvent();
445             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
446             if ( !( newEvent instanceof MethodEntryEvent)) {
447                  log3("ERROR: new event is not MethodEntryEvent");
448                  testExitCode = FAILED;
449             } else {
450                 log2("       Event is instanceof MethodEntryEvent");
451                 policy = eventSet.suspendPolicy();
452                 if (policy != policyExpected[i]) {
453                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
454                     log3("       eventSet.suspendPolicy() == " + policy);
455                     log3("       policyExpected           == " + policyExpected[i]);
456                     testExitCode = FAILED;
457                 }
458             }
459 
460             switch (i) {
461               case 0: eventRequest1.disable();                          break;
462               case 1: eventRequest2.disable();                          break;
463               case 2: eventRequest3.disable();                          break;
464               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
465               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
466               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
467               case 6: eventRequest1.disable(); eventRequest2.disable();
468                       eventRequest3.disable();                          break;
469            }
470 
471             switch (policy) {
472               case SUSPEND_NONE   :                      break;
473               case SUSPEND_THREAD : mainThread.resume(); break;
474               case SUSPEND_ALL    : vm.resume();         break;
475               default: throw new JDITestRuntimeException("** default case 3 **");
476             }
477 
478             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
479         }
480         log1("    TESTING ENDS");
481         return;
482     }

484    /*
485     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
486     *                                             String, String, String)
487     *
488     * It sets up a breakpoint at given line number within a given method in a given class
489     * for a given thread.
490     *
491     * Return value: BreakpointRequest object  in case of success
492     *
493     * JDITestRuntimeException   in case of an Exception thrown within the method
494     */
495 
496     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
497                                                   ReferenceType testedClass,
498                                                   String methodName,
499                                                   String bpLine,
500                                                   String property)
501             throws JDITestRuntimeException {
502 
503         log2("......setting up a breakpoint:");
504         log2("       thread: " + thread + "; class: " + testedClass +
505                         "; method: " + methodName + "; line: " + bpLine);
506 
507         List              alllineLocations = null;
508         Location          lineLocation     = null;
509         BreakpointRequest breakpRequest    = null;
510 
511         try {
512             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
513 
514             alllineLocations = method.allLineLocations();
515 
516             int n =
517                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
518             if (n > alllineLocations.size()) {
519                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
520             } else {
521                 lineLocation = (Location) alllineLocations.get(n);
522                 try {
523                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
524                     breakpRequest.putProperty("number", property);
525                     breakpRequest.addThreadFilter(thread);
526                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
527                 } catch ( Exception e1 ) {
528                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
529                     breakpRequest    = null;
530                 }
531             }
532         } catch ( Exception e2 ) {
533             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
534             breakpRequest    = null;
535         }
536 
537         if (breakpRequest == null) {
538             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
539             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
540         }
541 
542         log2("      a breakpoint has been set up");
543         return breakpRequest;
544     }

547     private void getEventSet()
548                  throws JDITestRuntimeException {
549         try {
550 //            log2("       eventSet = eventQueue.remove(waitTime);");
551             eventSet = eventQueue.remove(waitTime);
552             if (eventSet == null) {
553                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
554             }
555 //            log2("       eventIterator = eventSet.eventIterator;");
556             eventIterator = eventSet.eventIterator();
557         } catch ( Exception e ) {
558             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
559         }
560     }

563     private void breakpointForCommunication()
564                  throws JDITestRuntimeException {
565 
566         log2("breakpointForCommunication");

568 
569         if (eventIterator.nextEvent() instanceof BreakpointEvent)

571 
572         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
573     }

577     private MethodEntryRequest settingMethodEntryRequest ( ThreadReference thread,
578                                                            ReferenceType   testedClass,
579                                                            int             suspendPolicy,
580                                                            String          property       )
581             throws JDITestRuntimeException {
582         try {
583             log2("......setting up MethodEntryRequest:");
584             log2("       thread: " + thread + "; class: " + testedClass +  "; property: " + property);
585 
586             MethodEntryRequest
587             menr = eventRManager.createMethodEntryRequest();
588             menr.putProperty("number", property);
589             menr.addThreadFilter(thread);
590             menr.addClassFilter(testedClass);
591             menr.setSuspendPolicy(suspendPolicy);
592 
593             log2("      a MethodEntryRequest has been set up");
594             return menr;
595         } catch ( Exception e ) {
596             log3("ERROR: ATTENTION: Exception within settingMethodEntryRequest() : " + e);
597             log3("       MethodEntryRequest HAS NOT BEEN SET UP");
598             throw new JDITestRuntimeException("** FAILURE to set up MethodEntryRequest **");
599         }
