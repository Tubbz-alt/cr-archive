 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a MethodExitEvent set,              <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a MethodExitRequest, resumes          <BR>
 70  *   the debuggee, and waits for the MethodExitEvent.                   <BR>
 71  * - The debuggee invokes the special method to be resulting in the event.<BR>
 72  * - Upon getting new event, the debugger                               <BR>
 73  *   performs the check corresponding to the event.                     <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * In third phase, at the end,                                          <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  */
 82

106 
107         int exitCode = new suspendpolicy007().runThis(argv, out);

111         }

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : e");
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : e");
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
259                 log3("ERROR: last event is not the VMDeathEvent");
260                 return 1;
261             }
262 
263             log2("waiting for VMDisconnectEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
266                 log3("ERROR: last event is not the VMDisconnectEvent");
267                 return 1;
268             }
269 
270             return 0;
271 
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         log2("......getting ClassPrepareEvent for debuggee's class");
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
304 
305         String            bPointMethod = "methodForCommunication";
306         String            lineForComm  = "lineForComm";
307         BreakpointRequest bpRequest;
308         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
309         bpRequest = settingBreakpoint(mainThread,
310                                       debuggeeClass,
311                                       bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314         vm.resume();
315 
316     //------------------------------------------------------  testing section
317 
318         log1("     TESTING BEGINS");
319 
320         EventRequest eventRequest1 = null;
321         EventRequest eventRequest2 = null;
322         EventRequest eventRequest3 = null;
323 
324         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
325         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
327         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
328 
329         int policyExpected[] = {    SUSPEND_NONE,
330                                     SUSPEND_THREAD,
331                                     SUSPEND_ALL,
332                                     SUSPEND_THREAD,
333                                     SUSPEND_ALL,
334                                     SUSPEND_ALL,
335                                     SUSPEND_ALL      };
336         int policy = 0;
337 
338         ReferenceType testClassReference = null;
339 
340         String fieldName = "var1";
341 
342 
343         for (int i = 0; ; i++) {
344 
345 
346             breakpointForCommunication();
347 
348             int instruction = ((IntegerValue)
349                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
350             if (instruction == 0) {
351                 vm.resume();
352                 break;
353             }
354 
355 
356             log1(":::::: case: # " + i);
357 
358             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
359 
360             switch (i) {
361 
362               case 0:
363                       testClassReference =
364                            (ReferenceType) vm.classesByName(testedClassName).get(0);
365 
366                       eventRequest1 = settingMethodExitRequest (mainThread,
367                                              testClassReference,
368                                              SUSPEND_NONE, "MethodExitRequest1");
369                       eventRequest1.enable();
370                       break;
371 
372               case 1:
373                       eventRequest2 = settingMethodExitRequest (mainThread,
374                                              testClassReference,
375                                              SUSPEND_THREAD, "MethodExitRequest2");
376                       eventRequest2.enable();
377                       break;
378 
379               case 2:
380                       eventRequest3 = settingMethodExitRequest (mainThread,
381                                              testClassReference,
382                                              SUSPEND_ALL, "MethodExitRequest3");
383                       eventRequest3.enable();
384                       break;
385 
386               case 3:
387                       eventRequest1 = settingMethodExitRequest (mainThread,
388                                              testClassReference,
389                                              SUSPEND_NONE, "MethodExitRequest4");
390                       eventRequest1.enable();
391                       eventRequest2 = settingMethodExitRequest (mainThread,
392                                              testClassReference,
393                                              SUSPEND_THREAD, "MethodExitRequest5");
394                       eventRequest2.enable();
395                       break;
396 
397               case 4:
398                       eventRequest1 = settingMethodExitRequest (mainThread,
399                                              testClassReference,
400                                              SUSPEND_NONE, "MethodExitRequest6");
401                       eventRequest1.enable();
402                       eventRequest3 = settingMethodExitRequest (mainThread,
403                                              testClassReference,
404                                              SUSPEND_ALL, "MethodExitRequest7");
405                       eventRequest3.enable();
406                       break;
407 
408               case 5:
409                       eventRequest2 = settingMethodExitRequest (mainThread,
410                                              testClassReference,
411                                              SUSPEND_THREAD, "MethodExitRequest8");
412                       eventRequest2.enable();
413                       eventRequest3 = settingMethodExitRequest (mainThread,
414                                              testClassReference,
415                                              SUSPEND_ALL, "MethodExitRequest9");
416                       eventRequest3.enable();
417                       break;
418 
419               case 6:
420                       eventRequest1 = settingMethodExitRequest (mainThread,
421                                              testClassReference,
422                                              SUSPEND_NONE, "MethodExitRequest10");
423                       eventRequest1.enable();
424                       eventRequest2 = settingMethodExitRequest (mainThread,
425                                              testClassReference,
426                                              SUSPEND_THREAD, "MethodExitRequest11");
427                       eventRequest2.enable();
428                       eventRequest3 = settingMethodExitRequest (mainThread,
429                                              testClassReference,
430                                              SUSPEND_ALL, "MethodExitRequest12");
431                       eventRequest3.enable();
432                       break;
433 
434 
435               default:
436                       throw new JDITestRuntimeException("** default case 2 **");
437             }
438 
439             log2("......waiting for new MethodExitEvent : " + i);
440             mainThread.resume();
441             getEventSet();
442 
443             Event newEvent = eventIterator.nextEvent();
444             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
445             if ( !( newEvent instanceof MethodExitEvent)) {
446                  log3("ERROR: new event is not MethodExitEvent");
447                  testExitCode = FAILED;
448             } else {
449                 log2("       Event is instanceof MethodExitEvent");
450                 policy = eventSet.suspendPolicy();
451                 if (policy != policyExpected[i]) {
452                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
453                     log3("       eventSet.suspendPolicy() == " + policy);
454                     log3("       policyExpected           == " + policyExpected[i]);
455                     testExitCode = FAILED;
456                 }
457             }
458 
459             switch (i) {
460               case 0: eventRequest1.disable();                          break;
461               case 1: eventRequest2.disable();                          break;
462               case 2: eventRequest3.disable();                          break;
463               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
464               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
465               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
466               case 6: eventRequest1.disable(); eventRequest2.disable();
467                       eventRequest3.disable();                          break;
468            }
469 
470             switch (policy) {
471               case SUSPEND_NONE   :                      break;
472               case SUSPEND_THREAD : mainThread.resume(); break;
473               case SUSPEND_ALL    : vm.resume();         break;
474               default: throw new JDITestRuntimeException("** default case 3 **");
475             }
476 
477             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
478         }
479         log1("    TESTING ENDS");
480         return;
481     }

483    /*
484     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
485     *                                             String, String, String)
486     *
487     * It sets up a breakpoint at given line number within a given method in a given class
488     * for a given thread.
489     *
490     * Return value: BreakpointRequest object  in case of success
491     *
492     * JDITestRuntimeException   in case of an Exception thrown within the method
493     */
494 
495     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
496                                                   ReferenceType testedClass,
497                                                   String methodName,
498                                                   String bpLine,
499                                                   String property)
500             throws JDITestRuntimeException {
501 
502         log2("......setting up a breakpoint:");
503         log2("       thread: " + thread + "; class: " + testedClass +
504                         "; method: " + methodName + "; line: " + bpLine);
505 
506         List              alllineLocations = null;
507         Location          lineLocation     = null;
508         BreakpointRequest breakpRequest    = null;
509 
510         try {
511             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
512 
513             alllineLocations = method.allLineLocations();
514 
515             int n =
516                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
517             if (n > alllineLocations.size()) {
518                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
519             } else {
520                 lineLocation = (Location) alllineLocations.get(n);
521                 try {
522                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
523                     breakpRequest.putProperty("number", property);
524                     breakpRequest.addThreadFilter(thread);
525                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
526                 } catch ( Exception e1 ) {
527                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
528                     breakpRequest    = null;
529                 }
530             }
531         } catch ( Exception e2 ) {
532             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
533             breakpRequest    = null;
534         }
535 
536         if (breakpRequest == null) {
537             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
538             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
539         }
540 
541         log2("      a breakpoint has been set up");
542         return breakpRequest;
543     }

546     private void getEventSet()
547                  throws JDITestRuntimeException {
548         try {
549 //            log2("       eventSet = eventQueue.remove(waitTime);");
550             eventSet = eventQueue.remove(waitTime);
551             if (eventSet == null) {
552                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
553             }
554 //            log2("       eventIterator = eventSet.eventIterator;");
555             eventIterator = eventSet.eventIterator();
556         } catch ( Exception e ) {
557             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
558         }
559     }

562     private void breakpointForCommunication()
563                  throws JDITestRuntimeException {
564 
565         log2("breakpointForCommunication");
566         getEventSet();
567 
568         if (eventIterator.nextEvent() instanceof BreakpointEvent)
569             return;
570 
571         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
572     }

576     private MethodExitRequest settingMethodExitRequest ( ThreadReference thread,
577                                                          ReferenceType   testedClass,
578                                                          int             suspendPolicy,
579                                                          String          property        )
580             throws JDITestRuntimeException {
581         try {
582             log2("......setting up MethodExitRequest:");
583             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
584 
585             MethodExitRequest
586             mexr = eventRManager.createMethodExitRequest();
587             mexr.putProperty("number", property);
588             mexr.addThreadFilter(thread);
589             mexr.addClassFilter(testedClass);
590             mexr.setSuspendPolicy(suspendPolicy);
591 
592         log2("      MethodExitRequest has been set up");
593         return mexr;
594         } catch ( Exception e ) {
595             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
596             log3("       MethodExitRequest HAS NOT BEEN SET UP");
597             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
598         }
