 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for an AccessWatchpointEvent set,       <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a AccessWatchpointRequest, resumes            <BR>
 70  *   the debuggee, and waits for the AccessWatchpointEvent.             <BR>
 71  * - The debuggee invokes the special method which makes access to      <BR>
 72  *   a predefined variable to be resulting in the event.                <BR>
 73  * - Upon getting new event, the debugger                               <BR>
 74  *   performs the check corresponding to the event.                     <BR>
 75  * <BR>
 76  * Note. To inform each other of needed actions, the debugger and       <BR>
 77  *       and the debuggee use debuggee's variable "instruction".        <BR>
 78  * In third phase, at the end,                                          <BR>
 79  * the debuggee changes the value of the "instruction"                  <BR>
 80  * to inform the debugger of checks finished, and both end.             <BR>
 81  * <BR>
 82  */
 83

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
266                 log3("ERROR: last event is not the VMDeathEvent");
267                 return 1;
268             }
269 
270             return 0;
271 
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         if ( !vm.canWatchFieldAccess() ) {
286             log2("......vm.canWatchFieldAccess == false :: test cancelled");
287             vm.exit(PASS_BASE);
288             return;
289         }
290 
291 
292         eventRManager = vm.eventRequestManager();
293 
294         log2("......getting ClassPrepareEvent for debuggee's class");
295         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
296         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
297         cpRequest.addClassFilter(debuggeeName);
298         cpRequest.enable();
299         vm.resume();
300         getEventSet();
301         cpRequest.disable();
302 
303         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
304         debuggeeClass = event.referenceType();
305 
306         if (!debuggeeClass.name().equals(debuggeeName))
307            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
311 
312         String            bPointMethod = "methodForCommunication";
313         String            lineForComm  = "lineForComm";
314         BreakpointRequest bpRequest;
315         ThreadReference   mainThread = threadByName("main");
316         bpRequest = settingBreakpoint(mainThread,
317                                       debuggeeClass,
318                                       bPointMethod, lineForComm, "zero");
319         bpRequest.enable();
320 
321         vm.resume();
322 
323     //------------------------------------------------------  testing section
324 
325         log1("     TESTING BEGINS");
326 
327         EventRequest eventRequest1 = null;
328         EventRequest eventRequest2 = null;
329         EventRequest eventRequest3 = null;
330 
331         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
332         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
333         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
334         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
335 
336         int policyExpected[] = {    SUSPEND_NONE,
337                                     SUSPEND_THREAD,
338                                     SUSPEND_ALL,
339                                     SUSPEND_THREAD,
340                                     SUSPEND_ALL,
341                                     SUSPEND_ALL,
342                                     SUSPEND_ALL      };
343         int policy = 0;
344 
345         ReferenceType testClassReference = null;
346 
347         String fieldName = "var1";
348 
349 
350         for (int i = 0; ; i++) {
351 
352 
353             breakpointForCommunication();
354 
355             int instruction = ((IntegerValue)
356                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
357             if (instruction == 0) {
358                 vm.resume();
359                 break;
360             }
361 
362 
363             log1(":::::: case: # " + i);
364 
365             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
366 
367             switch (i) {
368 
369               case 0:
370                       testClassReference =
371                            (ReferenceType) vm.classesByName(testedClassName).get(0);
372 
373                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
374                                              testClassReference, fieldName,
375                                              SUSPEND_NONE, "AccessWatchpointRequest1");
376                       eventRequest1.enable();
377                       break;
378 
379               case 1:
380                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
381                                              testClassReference, fieldName,
382                                              SUSPEND_THREAD, "AccessWatchpointRequest2");
383                       eventRequest2.enable();
384                       break;
385 
386               case 2:
387                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
388                                              testClassReference, fieldName,
389                                              SUSPEND_ALL, "AccessWatchpointRequest3");
390                       eventRequest3.enable();
391                       break;
392 
393               case 3:
394                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
395                                              testClassReference, fieldName,
396                                              SUSPEND_NONE, "AccessWatchpointRequest4");
397                       eventRequest1.enable();
398                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
399                                              testClassReference, fieldName,
400                                              SUSPEND_THREAD, "AccessWatchpointRequest5");
401                       eventRequest2.enable();
402                       break;
403 
404               case 4:
405                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
406                                              testClassReference, fieldName,
407                                              SUSPEND_NONE, "AccessWatchpointRequest6");
408                       eventRequest1.enable();
409                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
410                                              testClassReference, fieldName,
411                                              SUSPEND_ALL, "AccessWatchpointRequest7");
412                       eventRequest3.enable();
413                       break;
414 
415               case 5:
416                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
417                                              testClassReference, fieldName,
418                                              SUSPEND_THREAD, "AccessWatchpointRequest8");
419                       eventRequest2.enable();
420                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
421                                              testClassReference, fieldName,
422                                              SUSPEND_ALL, "AccessWatchpointRequest9");
423                       eventRequest3.enable();
424                       break;
425 
426               case 6:
427                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
428                                              testClassReference, fieldName,
429                                              SUSPEND_NONE, "AccessWatchpointRequest10");
430                       eventRequest1.enable();
431                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
432                                              testClassReference, fieldName,
433                                              SUSPEND_THREAD, "AccessWatchpointRequest11");
434                       eventRequest2.enable();
435                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
436                                              testClassReference, fieldName,
437                                              SUSPEND_ALL, "AccessWatchpointRequest12");
438                       eventRequest3.enable();
439                       break;
440 
441 
442               default:
443                       throw new JDITestRuntimeException("** default case 2 **");
444             }
445 
446             log2("......waiting for new AccessWatchpointEvent : " + i);
447             mainThread.resume();
448             getEventSet();
449 
450             Event newEvent = eventIterator.nextEvent();
451             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
452             if ( !( newEvent instanceof AccessWatchpointEvent)) {
453                  log3("ERROR: new event is not AccessWatchpointEvent");
454                  testExitCode = FAILED;
455             } else {
456                 log2("       Event is instanceof AccessWatchpointEvent");
457                 policy = eventSet.suspendPolicy();
458                 if (policy != policyExpected[i]) {
459                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
460                     log3("       eventSet.suspendPolicy() == " + policy);
461                     log3("       policyExpected           == " + policyExpected[i]);
462                     testExitCode = FAILED;
463                 }
464             }
465 
466             switch (i) {
467               case 0: eventRequest1.disable();                          break;
468               case 1: eventRequest2.disable();                          break;
469               case 2: eventRequest3.disable();                          break;
470               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
471               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
472               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
473               case 6: eventRequest1.disable(); eventRequest2.disable();
474                       eventRequest3.disable();                          break;
475            }
476 
477             switch (policy) {
478               case SUSPEND_NONE   :                      break;
479               case SUSPEND_THREAD : mainThread.resume(); break;
480               case SUSPEND_ALL    : vm.resume();         break;
481               default: throw new JDITestRuntimeException("** default case 3 **");
482             }
483 
484             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
485         }
486         log1("    TESTING ENDS");
487         return;
488     }

490     private ThreadReference threadByName(String name)
491                  throws JDITestRuntimeException {
492 
493         List         all = vm.allThreads();
494         ListIterator li  = all.listIterator();
495 
496         for (; li.hasNext(); ) {
497             ThreadReference thread = (ThreadReference) li.next();
498             if (thread.name().equals(name))
499                 return thread;
500         }
501         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
502     }

504    /*
505     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
506     *                                             String, String, String)
507     *
508     * It sets up a breakpoint at given line number within a given method in a given class
509     * for a given thread.
510     *
511     * Return value: BreakpointRequest object  in case of success
512     *
513     * JDITestRuntimeException   in case of an Exception thrown within the method
514     */
515 
516     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
517                                                   ReferenceType testedClass,
518                                                   String methodName,
519                                                   String bpLine,
520                                                   String property)
521             throws JDITestRuntimeException {
522 
523         log2("......setting up a breakpoint:");
524         log2("       thread: " + thread + "; class: " + testedClass +
525                         "; method: " + methodName + "; line: " + bpLine);
526 
527         List              alllineLocations = null;
528         Location          lineLocation     = null;
529         BreakpointRequest breakpRequest    = null;
530 
531         try {
532             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
533 
534             alllineLocations = method.allLineLocations();
535 
536             int n =
537                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
538             if (n > alllineLocations.size()) {
539                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
540             } else {
541                 lineLocation = (Location) alllineLocations.get(n);
542                 try {
543                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
544                     breakpRequest.putProperty("number", property);
545                     breakpRequest.addThreadFilter(thread);
546                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
547                 } catch ( Exception e1 ) {
548                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
549                     breakpRequest    = null;
550                 }
551             }
552         } catch ( Exception e2 ) {
553             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
554             breakpRequest    = null;
555         }
556 
557         if (breakpRequest == null) {
558             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
559             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
560         }
561 
562         log2("      a breakpoint has been set up");
563         return breakpRequest;
564     }

567     private void getEventSet()
568                  throws JDITestRuntimeException {
569         try {
570 //            log2("       eventSet = eventQueue.remove(waitTime);");
571             eventSet = eventQueue.remove(waitTime);
572             if (eventSet == null) {
573                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
574             }
575 //            log2("       eventIterator = eventSet.eventIterator;");
576             eventIterator = eventSet.eventIterator();
577         } catch ( Exception e ) {
578             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
579         }
580     }

583     private void breakpointForCommunication()
584                  throws JDITestRuntimeException {
585 
586         log2("breakpointForCommunication");
587         getEventSet();
588 
589         if (eventIterator.nextEvent() instanceof BreakpointEvent)
590             return;
591 
592         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
593     }

597     private AccessWatchpointRequest settingAccessWatchpointRequest (
598                                                   ThreadReference thread,
599                                                   ReferenceType   testedClass,
600                                                   String          fieldName,
601                                                   int             suspendPolicy,
602                                                   String          property        )
603             throws JDITestRuntimeException {
604         try {
605             log2("......setting up AccessWatchpointRequest:");
606             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
607             Field field = testedClass.fieldByName(fieldName);
608 
609             AccessWatchpointRequest
610             awr = eventRManager.createAccessWatchpointRequest(field);
611             awr.putProperty("number", property);
612             awr.addThreadFilter(thread);
613             awr.setSuspendPolicy(suspendPolicy);
614 
615             log2("      AccessWatchpointRequest has been set up");
616             return awr;
617         } catch ( Exception e ) {
618             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
619             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
620             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
621         }
