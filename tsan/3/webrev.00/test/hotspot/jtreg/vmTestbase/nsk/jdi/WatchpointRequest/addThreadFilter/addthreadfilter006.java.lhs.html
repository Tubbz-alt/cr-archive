 38 /**
 39  * The test for the implementation of an object of the type     <BR>
 40  * WatchpointRequest.                                           <BR>
 41  *                                                              <BR>
 42  * The test checks that results of the method                   <BR>
 43  * <code>com.sun.jdi.WatchpointRequest.addThreadFilter()</code> <BR>
 44  * complies with its spec.                                      <BR>
 45  * <BR>
 46  * The test checks up on the following assertion:               <BR>
 47  *    Restricts the events generated by this request            <BR>
 48  *    to those in the given thread.                             <BR>
 49  * The cases to test include three thread's states:             <BR>
 50  * not started, running, dead.                                  <BR>
 51  * The tested Request is ModificationWatchpointRequest.         <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee creates new thread, thread1, and invokes              <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 74  *   - creates ModificationWatchpointRequest1 and                       <BR>
 75  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 76  *   - creates ModificationWatchpointRequest2,                          <BR>
 77  *     resumes the debuggee to get the thread1 started, and             <BR>
 78  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 79  *   - creates ModificationWatchpointRequest3, waits the thread1 to die,<BR>
 80  *     and invokes the method addThreadFilter(thread) on the request3.  <BR>
 81  * <BR>
 82  * In third phase, at the end of the test,                              <BR>
 83  * the debuggee changes the value of the "instruction"  which           <BR>
 84  * the debugger and debuggee use to inform each other of needed actions,<BR>
 85  * and both end.
 86  * <BR>
 87  */

170     private int runThis (String argv[], PrintStream out) {
171 
172         argsHandler     = new ArgumentHandler(argv);
173         logHandler      = new Log(out, argsHandler);
174         Binder binder   = new Binder(argsHandler, logHandler);
175 
176         waitTime        = argsHandler.getWaitTime() * 60000;
177 
178         try {
179             log2("launching a debuggee :");
180             log2("       " + debuggeeName);
181             if (argsHandler.verbose()) {
182                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
183             } else {
184                 debuggee = binder.bindToDebugee(debuggeeName);
185             }
186             if (debuggee == null) {
187                 log3("ERROR: no debuggee launched");
188                 return FAILED;
189             }
190             log2("debuggee launched");
191         } catch ( Exception e ) {
192             log3("ERROR: Exception : " + e);
193             log2("       test cancelled");
194             return FAILED;
195         }
196 
197         debuggee.redirectOutput(logHandler);
198 
199         vm = debuggee.VM();
200 
201         eventQueue = vm.eventQueue();
202         if (eventQueue == null) {
203             log3("ERROR: eventQueue == null : TEST ABORTED");
204             vm.exit(PASS_BASE);
205             return FAILED;
206         }
207 
208         log2("invocation of the method runTest()");
209         switch (runTest()) {
210 
211             case 0 :  log2("test phase has finished normally");
212                       log2("   waiting for the debuggee to finish ...");
213                       debuggee.waitFor();
214 
215                       log2("......getting the debuggee's exit status");
216                       int status = debuggee.getStatus();
217                       if (status != PASS_BASE) {
218                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
219                               status + " != PASS_BASE");
220                           testExitCode = FAILED;
221                       } else {
222                           log2("......debuggee returned expected exit status: " +
223                               status + " == PASS_BASE");
224                       }
225                       break;
226 
227             default : log3("ERROR: runTest() returned unexpected value");
228 
229             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
230                       log2("......forcing: vm.exit();");
231                       testExitCode = FAILED;
232                       try {
233                           vm.exit(PASS_BASE);
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238 
239             case 2 :  log3("test cancelled due to VMDisconnectedException");
240                       log2("......trying: vm.process().destroy();");
241                       testExitCode = FAILED;
242                       try {
243                           Process vmProcess = vm.process();
244                           if (vmProcess != null) {
245                               vmProcess.destroy();
246                           }
247                       } catch ( Exception e ) {
248                           log3("ERROR: Exception : " + e);
249                       }
250                       break;
251             }
252 
253         return testExitCode;
254     }

263     private int runTest() {
264 
265         try {
266             testRun();
267 
268             log2("waiting for VMDeathEvent");
269             getEventSet();
270             if (eventIterator.nextEvent() instanceof VMDeathEvent)
271                 return 0;
272 
273             log3("ERROR: last event is not the VMDeathEvent");
274             return 1;
275         } catch ( VMDisconnectedException e ) {
276             log3("ERROR: VMDisconnectedException : " + e);
277             return 2;
278         } catch ( Exception e ) {
279             log3("ERROR: Exception : " + e);
280             return 1;
281         }

285     private void testRun()
286                  throws JDITestRuntimeException, Exception {
287 
288         if ( !vm.canWatchFieldModification() ) {
289             log2("......vm.canWatchFieldModification == false :: test cancelled");
290             vm.exit(PASS_BASE);
291             return;
292         }
293 
294 
295         eventRManager = vm.eventRequestManager();
296 
297         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
298         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
299         cpRequest.addClassFilter(debuggeeName);
300 
301         cpRequest.enable();
302         vm.resume();
303         getEventSet();
304         cpRequest.disable();
305 
306         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
307         debuggeeClass = event.referenceType();
308 
309         if (!debuggeeClass.name().equals(debuggeeName))
310            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
311 
312         log2("      received: ClassPrepareEvent for debuggeeClass");
313 
314         String bPointMethod = "methodForCommunication";
315         String lineForComm  = "lineForComm";
316 
317         ThreadReference   mainThread = threadByName("main");
318 
319         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
320                                              debuggeeClass,
321                                             bPointMethod, lineForComm, "zero");
322         bpRequest.enable();
323 
324     //------------------------------------------------------  testing section
325 
326         log1("     TESTING BEGINS");
327 
328         EventRequest eventRequest1 = null;
329         EventRequest eventRequest2 = null;
330         EventRequest eventRequest3 = null;
331 
332         ThreadReference thread1    = null;
333         String         thread1Name = "thread1";
334 
335         String property1 = "ModificationWatchpointRequest1";
336         String property2 = "ModificationWatchpointRequest2";
337         String property3 = "ModificationWatchpointRequest3";
338 
339         String fieldName = "var1";
340 
341         ReferenceType testClassReference = null;
342 
343         for (int i = 0; ; i++) {
344 
345             vm.resume();
346             breakpointForCommunication();
347 
348             int instruction = ((IntegerValue)
349                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
350 
351             if (instruction == 0) {
352                 vm.resume();
353                 break;
354             }
355 
356             log1(":::::: case: # " + i);
357 
358             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
359 
360             switch (i) {
361 
362               case 0:
363                       testClassReference =
364                            (ReferenceType) vm.classesByName(testedClassName).get(0);
365                       thread1 = (ThreadReference) debuggeeClass.getValue(
366                                                   debuggeeClass.fieldByName(thread1Name));
367 
368                       eventRequest1 = setting2ModificationWatchpointRequest (null,
369                                              testClassReference, fieldName,
370                                              EventRequest.SUSPEND_NONE, "ModificationWatchpointRequest1");
371                       try {
372                           log2("......eventRequest1.addThreadFilter(thread1);");
373                           log2("        no Exception expected");
374                           ((ModificationWatchpointRequest)eventRequest1).addThreadFilter(thread1);
375                       } catch ( Exception e ) {
376                           log3("ERROR: Exception : " + e);
377                           testExitCode = FAILED;
378                       }
379                       break;
380 
381               case 1:
382                       eventRequest2 = setting2ModificationWatchpointRequest (null,
383                                              testClassReference, fieldName,
384                                              EventRequest.SUSPEND_NONE, "ModificationWatchpointRequest2");
385                       try {
386                           log2("......eventRequest2.addThreadFilter(thread1);");
387                           log2("        no Exception expected");
388                           ((ModificationWatchpointRequest)eventRequest2).addThreadFilter(thread1);
389                       } catch ( Exception e ) {
390                           log3("ERROR: Exception : " + e);
391                           testExitCode = FAILED;
392                       }
393                       break;
394 
395               case 2:
396                       eventRequest3 = setting2ModificationWatchpointRequest (null,
397                                              testClassReference, fieldName,
398                                              EventRequest.SUSPEND_NONE, "ModificationWatchpointRequest3");
399                       try {
400                           log2("......eventRequest3.addThreadFilter(thread1);");
401                           log2("        no Exception expected");
402                           ((ModificationWatchpointRequest)eventRequest3).addThreadFilter(thread1);
403                       } catch ( Exception e ) {
404                           log3("ERROR: Exception : " + e);

419     private ThreadReference threadByName(String name)
420                  throws JDITestRuntimeException {
421 
422         List         all = vm.allThreads();
423         ListIterator li  = all.listIterator();
424 
425         for (; li.hasNext(); ) {
426             ThreadReference thread = (ThreadReference) li.next();
427             if (thread.name().equals(name))
428                 return thread;
429         }
430         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
431     }

433    /*
434     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
435     *                                             String, String, String)
436     *
437     * It sets up a breakpoint at given line number within a given method in a given class
438     * for a given thread.
439     *
440     * Return value: BreakpointRequest object  in case of success
441     *
442     * JDITestRuntimeException   in case of an Exception thrown within the method
443     */
444 
445     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
446                                                   ReferenceType testedClass,
447                                                   String methodName,
448                                                   String bpLine,
449                                                   String property)
450             throws JDITestRuntimeException {
451 
452         log2("......setting up a breakpoint:");
453         log2("       thread: " + thread + "; class: " + testedClass +
454                         "; method: " + methodName + "; line: " + bpLine);
455 
456         List              alllineLocations = null;
457         Location          lineLocation     = null;
458         BreakpointRequest breakpRequest    = null;
459 
460         try {
461             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
462 
463             alllineLocations = method.allLineLocations();
464 
465             int n =
466                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
467             if (n > alllineLocations.size()) {
468                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
469             } else {
470                 lineLocation = (Location) alllineLocations.get(n);
471                 try {
472                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
473                     breakpRequest.putProperty("number", property);
474                     breakpRequest.addThreadFilter(thread);
475                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
476                 } catch ( Exception e1 ) {
477                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
478                     breakpRequest    = null;
479                 }
480             }
481         } catch ( Exception e2 ) {
482             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
483             breakpRequest    = null;
484         }
485 
486         if (breakpRequest == null) {
487             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
488             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
489         }
490 
491         log2("      a breakpoint has been set up");
492         return breakpRequest;
493     }

496     private void getEventSet()
497                  throws JDITestRuntimeException {
498         try {
499 //            log2("       eventSet = eventQueue.remove(waitTime);");
500             eventSet = eventQueue.remove(waitTime);
501             if (eventSet == null) {
502                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
503             }
504 //            log2("       eventIterator = eventSet.eventIterator;");
505             eventIterator = eventSet.eventIterator();
506         } catch ( Exception e ) {
507             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
508         }
509     }

512     private void breakpointForCommunication()
513                  throws JDITestRuntimeException {
514 
515         log2("breakpointForCommunication");
516         getEventSet();
517 
518         if (eventIterator.nextEvent() instanceof BreakpointEvent)
519             return;
520 
521         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
522     }

526     private ModificationWatchpointRequest setting2ModificationWatchpointRequest (
527                                                   ThreadReference thread,
528                                                   ReferenceType   testedClass,
529                                                   String          fieldName,
530                                                   int             suspendPolicy,
531                                                   String          property        )
532             throws JDITestRuntimeException {
533         try {
534             log2("......setting up ModificationWatchpointRequest:");
535             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
536             Field field = testedClass.fieldByName(fieldName);
537 
538             ModificationWatchpointRequest
539             awr = eventRManager.createModificationWatchpointRequest(field);
540             awr.putProperty("number", property);
541 
542             if (thread != null)
543                 awr.addThreadFilter(thread);
544             awr.setSuspendPolicy(suspendPolicy);
545 
546             log2("      ModificationWatchpointRequest has been set up");
547             return awr;
548         } catch ( Exception e ) {
549             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
550             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
551             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
552         }
