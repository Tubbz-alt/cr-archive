414     private ThreadReference threadByName(String name)
415                  throws JDITestRuntimeException {
416 
417         List         all = vm.allThreads();
418         ListIterator li  = all.listIterator();
419 
420         for (; li.hasNext(); ) {
421             ThreadReference thread = (ThreadReference) li.next();
422             if (thread.name().equals(name))
423                 return thread;
424         }
425         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
426     }
