 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.stepRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled step requests.                <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

151     private int runThis (String argv[], PrintStream out) {
152 
153         argsHandler     = new ArgumentHandler(argv);
154         logHandler      = new Log(out, argsHandler);
155         Binder binder   = new Binder(argsHandler, logHandler);
156 
157         waitTime        = argsHandler.getWaitTime() * 60000;
158 
159         try {
160             log2("launching a debuggee :");
161             log2("       " + debuggeeName);
162             if (argsHandler.verbose()) {
163                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
164             } else {
165                 debuggee = binder.bindToDebugee(debuggeeName);
166             }
167             if (debuggee == null) {
168                 log3("ERROR: no debuggee launched");
169                 return FAILED;
170             }
171             log2("debuggee launched");
172         } catch ( Exception e ) {
173             log3("ERROR: Exception : " + e);
174             log2("       test cancelled");
175             return FAILED;
176         }
177 
178         debuggee.redirectOutput(logHandler);
179 
180         vm = debuggee.VM();
181 
182         eventQueue = vm.eventQueue();
183         if (eventQueue == null) {
184             log3("ERROR: eventQueue == null : TEST ABORTED");
185             vm.exit(PASS_BASE);
186             return FAILED;
187         }
188 
189         log2("invocation of the method runTest()");
190         switch (runTest()) {
191 
192             case 0 :  log2("test phase has finished normally");
193                       log2("   waiting for the debuggee to finish ...");
194                       debuggee.waitFor();
195 
196                       log2("......getting the debuggee's exit status");
197                       int status = debuggee.getStatus();
198                       if (status != PASS_BASE) {
199                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
200                               status + " != PASS_BASE");
201                           testExitCode = FAILED;
202                       } else {
203                           log2("......debuggee returned expected exit status: " +
204                               status + " == PASS_BASE");
205                       }
206                       break;
207 
208             default : log3("ERROR: runTest() returned unexpected value");
209 
210             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
211                       log2("......forcing: vm.exit();");
212                       testExitCode = FAILED;
213                       try {
214                           vm.exit(PASS_BASE);
215                       } catch ( Exception e ) {
216                           log3("ERROR: Exception : " + e);
217                       }
218                       break;
219 
220             case 2 :  log3("test cancelled due to VMDisconnectedException");
221                       log2("......trying: vm.process().destroy();");
222                       testExitCode = FAILED;
223                       try {
224                           Process vmProcess = vm.process();
225                           if (vmProcess != null) {
226                               vmProcess.destroy();
227                           }
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : " + e);
230                       }
231                       break;
232             }
233 
234         return testExitCode;
235     }

244     private int runTest() {
245 
246         try {
247             testRun();
248 
249             log2("waiting for VMDeathEvent");
250             getEventSet();
251             if (eventIterator.nextEvent() instanceof VMDeathEvent)
252                 return 0;
253 
254             log3("ERROR: last event is not the VMDeathEvent");
255             return 1;
256         } catch ( VMDisconnectedException e ) {
257             log3("ERROR: VMDisconnectedException : " + e);
258             return 2;
259         } catch ( Exception e ) {
260             log3("ERROR: Exception : " + e);
261             return 1;
262         }
263 
264     }

266     private void testRun()
267                  throws JDITestRuntimeException, Exception {
268 
269         eventRManager = vm.eventRequestManager();
270 
271         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
272         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
273         cpRequest.addClassFilter(debuggeeName);
274 
275         cpRequest.enable();
276         vm.resume();
277         getEventSet();
278         cpRequest.disable();
279 
280         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
281         debuggeeClass = event.referenceType();
282 
283         if (!debuggeeClass.name().equals(debuggeeName))
284            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
285 
286         log2("      received: ClassPrepareEvent for debuggeeClass");
287 
288         String bPointMethod = "methodForCommunication";
289         String lineForComm  = "lineForComm";
290 
291         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
292 
293         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
294                                              debuggeeClass,
295                                             bPointMethod, lineForComm, "zero");
296         bpRequest.enable();
297 
298     //------------------------------------------------------  testing section
299 
300         String fieldName = "testField";
301 
302         List         requests = null;
303         ListIterator li       = null;
304 
305         StepRequest request = null;
306         StepRequest stRequests[] = { null, null, null, null, null,
307                                      null, null, null, null, null };
308         int listSize;
309         int flag;
310 
311 
312         log1("     TESTING BEGINS");
313 
314         for (int i = 0; ; i++) {
315 
316             vm.resume();
317             breakpointForCommunication();
318 
319             int instruction = ((IntegerValue)
320                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
321 
322             if (instruction == 0) {
323                 vm.resume();
324                 break;
325             }
326 
327             log1(":::::: case: # " + i);
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
330 
331             ArrayReference threads = (ArrayReference)
332                                      debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName));
333 
334 
335             log2("......creating StepRequests");
336             for (int i1 = 0; i1 < stRequests.length; i1++) {
337                 stRequests[i1] = eventRManager.createStepRequest( (ThreadReference) threads.getValue(i1),
338                                                                  StepRequest.STEP_MIN, StepRequest.STEP_OUT);
339                 stRequests[i1].putProperty("number", String.valueOf(i1));
340 
341                 log2("......checking up on returned List after creating new StepRequest");
342                 requests = eventRManager.stepRequests();
343                 listSize = requests.size();
344                 if ( listSize != (i1 + 1) ) {
345                     testExitCode = FAILED;
346                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
347                 }
348                 flag = 0;
349                 li = requests.listIterator();
350                 while (li.hasNext()) {
351                     request = (StepRequest) li.next();
352                     if ( !request.isEnabled() ) {
353                         flag++;
354                         if (flag > 1) {
355                             testExitCode = FAILED;
356                             log3("ERROR: # of disabled requests > 1 : " + flag);
357                         }
358                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
359                             testExitCode = FAILED;
360                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
361                         }
362                     } else {
363                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
364                             testExitCode = FAILED;
365                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
366                         }
367                     }
368                 }
369 
370                 log2("      enabling created StepRequest");
371                 stRequests[i1].enable();
372                 requests = eventRManager.stepRequests();
373                 listSize = requests.size();
374                 if ( listSize != (i1 + 1) ) {
375                     testExitCode = FAILED;
376                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
377                 }
378 
379                 li = requests.listIterator();
380                 while (li.hasNext()) {
381                     request = (StepRequest) li.next();
382                     if ( !request.isEnabled() ) {
383                         testExitCode = FAILED;
384                         log3("ERROR: returned List contains disabled request : " + request);
385                     }
386                 }
387 
388                 log2("      removing item from the List; UnsupportedOperationException is expected");
389                 try {
390                     requests.remove(i1);
391                     testExitCode = FAILED;
392                     log3("ERROR: NO exception");
393                 } catch ( UnsupportedOperationException e ) {
394                     log2("        UnsupportedOperationException ");
395                 }
396             }
397 
398             log2("......deleting StepRequests");
399             for (int i2 = stRequests.length -1; i2 >= 0; i2--) {
400                 eventRManager.deleteEventRequest(stRequests[i2]);
401                 requests = eventRManager.stepRequests();
402                 listSize = requests.size();
403                 if ( listSize != i2 ) {
404                     testExitCode = FAILED;
405                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " + i2);
406                 }
407                 log2("      removing item from the List; UnsupportedOperationException is expected");
408                 try {
409                     requests.remove(i2);
410                     testExitCode = FAILED;
411                     log3("ERROR: NO exception");
412                 } catch ( UnsupportedOperationException e ) {
413                     log2("        UnsupportedOperationException ");
414                 }
415             }
416 
417             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
418         }
419         log1("    TESTING ENDS");
420         return;
421     }

423    /*
424     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
425     *                                             String, String, String)
426     *
427     * It sets up a breakpoint at given line number within a given method in a given class
428     * for a given thread.
429     *
430     * Return value: BreakpointRequest object  in case of success
431     *
432     * JDITestRuntimeException   in case of an Exception thrown within the method
433     */
434 
435     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
436                                                   ReferenceType testedClass,
437                                                   String methodName,
438                                                   String bpLine,
439                                                   String property)
440             throws JDITestRuntimeException {
441 
442         log2("......setting up a breakpoint:");
443         log2("       thread: " + thread + "; class: " + testedClass +
444                         "; method: " + methodName + "; line: " + bpLine);
445 
446         List              alllineLocations = null;
447         Location          lineLocation     = null;
448         BreakpointRequest breakpRequest    = null;
449 
450         try {
451             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
452 
453             alllineLocations = method.allLineLocations();
454 
455             int n =
456                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
457             if (n > alllineLocations.size()) {
458                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
459             } else {
460                 lineLocation = (Location) alllineLocations.get(n);
461                 try {
462                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
463                     breakpRequest.putProperty("number", property);
464                     breakpRequest.addThreadFilter(thread);
465                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
466                 } catch ( Exception e1 ) {
467                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
468                     breakpRequest    = null;
469                 }
470             }
471         } catch ( Exception e2 ) {
472             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
473             breakpRequest    = null;
474         }
475 
476         if (breakpRequest == null) {
477             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
478             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
479         }
480 
481         log2("      a breakpoint has been set up");
482         return breakpRequest;
483     }

486     private void getEventSet()
487                  throws JDITestRuntimeException {
488         try {
489 //            log2("       eventSet = eventQueue.remove(waitTime);");
490             eventSet = eventQueue.remove(waitTime);
491             if (eventSet == null) {
492                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
493             }
494 //            log2("       eventIterator = eventSet.eventIterator;");
495             eventIterator = eventSet.eventIterator();
496         } catch ( Exception e ) {
497             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
498         }
499     }

502     private void breakpointForCommunication()
503                  throws JDITestRuntimeException {
504 
505         log2("breakpointForCommunication");
506         getEventSet();
507 
508         if (eventIterator.nextEvent() instanceof BreakpointEvent)
509             return;
510 
511         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
512     }
