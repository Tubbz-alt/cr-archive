390     private ThreadReference threadByName(String name)
391                  throws JDITestRuntimeException {
392 
393         List         all = vm.allThreads();
394         ListIterator li  = all.listIterator();
395 
396         for (; li.hasNext(); ) {
397             ThreadReference thread = (ThreadReference) li.next();
398             if (thread.name().equals(name))
399                 return thread;
400         }
401         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
402     }
