 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.redefineClasses()</code>    <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    If any redefined methods have active stack frames,        <BR>
 47  *    those active frames continue to run the bytecodes of      <BR>
 48  *    the previous method.                                      <BR>
 49  *    The redefined methods will be used on new invokes.        <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                              <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assetion,                               <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new tested class object, objRF1, and invokes  <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the breakpointForCommunication Event,                 <BR>
 72  *   the debugger sets up the breakpoint bpRequest2 in the tested method<BR>
 73  *   redefineclasses001b.m2(), resumes the debuggee.                    <BR>
 74  * - The debuggee invokes objRF1.m1() method to be suspended            <BR>
 75  *   at the breakpoint in the method m2().                              <BR>
 76  * - Upon getting the breakpointInMethod Event,                         <BR>
 77  *   the debugger prepares mapClassToByte, invokes                      <BR>
 78  *         VirtualMachine.redefineClasses(mapClassToByte)               <BR>
 79  *   and resumes the debuggee.                                          <BR>
 80  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 81  *   once again and to inform the debugger with the event.              <BR>
 82  * - Upon getting seconf breakpointForCommunication Event,              <BR>
 83  *   the debugger sets up another breakpoint in the redefined method m2()<BR>
 84  *   and resumes the debuggee.                                          <BR>
 85  * - The debuggee invokes the method m1() to invoke m2() second time,   <BR>
 86  *   wher it will be suspended once again.                              <BR>
 87  * - Upon getting second breakpointInMethod Event,                      <BR>
 88  *   the debugger checks up the value of the class variable "i1"        <BR>
 89  *   which should be equal to the expected value.                       <BR>
 90  * <BR>
 91  * In third phase when at the end,                                      <BR>
 92  * the debuggee changes the value of the "instruction"                  <BR>
 93  * to inform the debugger of checks finished, and both end.             <BR>
 94  * <BR>
 95  */
 96

168     private int runThis (String argv[], PrintStream out) {
169 
170         argsHandler     = new ArgumentHandler(argv);
171         logHandler      = new Log(out, argsHandler);
172         Binder binder   = new Binder(argsHandler, logHandler);
173 
174         waitTime        = argsHandler.getWaitTime() * 60000;
175 
176         try {
177             log2("launching a debuggee :");
178             log2("       " + debuggeeName);
179 
180             debuggee = binder.bindToDebugee(debuggeeName);
181 
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : e");
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : e");
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

254     * Return value: 0 - normal end of the test
255     *               1 - ubnormal end of the test
256     *               2 - VMDisconnectedException while test phase

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278 
279     }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303 
304         if ( !vm.canRedefineClasses() ) {
305             log2("......vm.canRedefineClasses() == false : test is cancelled");
306             vm.resume();
307             return;
308         }
309 
310 
311         String bPointMethod = "methodForCommunication";
312         String lineForComm  = "lineForComm";
313         BreakpointRequest bpRequest;
314 
315         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
316                                       debuggeeClass,
317                                       bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320     //------------------------------------------------------  testing section
321 
322         log1("     TESTING BEGINS");
323 
324         String bpClassName  = "nsk.jdi.VirtualMachine.redefineClasses.redefineclasses001b";
325         String bpMethodName = "m2";
326         String bpLineName   = "bpline";
327         String varName      = "i1";
328         int    varValue     = 2;
329 
330         BreakpointRequest bpRequest2 = null;
331         BreakpointRequest bpRequest3 = null;
332         ReferenceType     bpClass    = null;
333 
334 
335         for (int i = 0; ; i++) {
336 
337             vm.resume();
338             breakpointForCommunication();
339 
340             int instruction = ((IntegerValue)
341                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
342 
343             if (instruction == 0) {
344                 vm.resume();
345                 break;
346             }
347 
348             log1(":::::: case: # " + i);
349 
350             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
351 
352             switch (i) {
353               case 0:
354                   List          classes = vm.classesByName(bpClassName);
355                   bpClass = (ReferenceType) classes.get(0);
356                   bpRequest2 = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
357                                           bpClass,
358                                           bpMethodName, bpLineName, "one");
359                   bpRequest2.enable();
360 
361                   vm.resume();
362                   breakpointInMethod("one");
363                   bpRequest2.disable();
364 
365                   try {
366                       log2("......vm.redefineClasses(mapClassToBytes());");
367                       vm.redefineClasses(mapClassToBytes());
368                   } catch ( Exception e ) {
369                       log3("ERROR:  Exception: " + e);
370                       testExitCode = FAILED;
371                       throw e;
372                   } catch ( Error e ) {
373                       log3("ERROR:  Error: " + e);
374                       testExitCode = FAILED;
375                       throw e;
376                   }
377                   break;
378 
379               case 1:
380 
381                   bpRequest3 = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
382                                           bpClass,
383                                           bpMethodName, bpLineName, "one");
384 
385                   bpRequest3.enable();
386 
387                   vm.resume();
388                   breakpointInMethod("one");
389                   bpRequest3.disable();
390 
391                   int i1 = ( (IntegerValue)
392                       bpClass.getValue(bpClass.fieldByName(varName) ) ).value();
393 
394                   if (i1 != varValue) {
395                       testExitCode = FAILED;
396                       log3("ERROR: value of i1 != 2 (expected) but : " + i1);
397                   }
398                   break;
399 
400               default:
401                   throw new JDITestRuntimeException ("** default case **");
402             }
403             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
404         }
405 // this is for debugging
406 // testExitCode = FAILED;
407 
408         vm.resume();
409         log1("    TESTING ENDS");
410         return;
411     }

413    /*
414     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
415     *                                             String, String, String)
416     *
417     * It sets up a breakpoint at given line number within a given method in a given class
418     * for a given thread.
419     *
420     * Return value: BreakpointRequest object  in case of success
421     *
422     * JDITestRuntimeException   in case of an Exception thrown within the method
423     */
424 
425     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
426                                                   ReferenceType testedClass,
427                                                   String methodName,
428                                                   String bpLine,
429                                                   String property)
430             throws JDITestRuntimeException {
431 
432         log2("......setting up a breakpoint:");
433         log2("       thread: " + thread + "; class: " + testedClass +
434                         "; method: " + methodName + "; line: " + bpLine);
435 
436         List              alllineLocations = null;
437         Location          lineLocation     = null;
438         BreakpointRequest breakpRequest    = null;
439 
440         try {
441             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
442             alllineLocations = method.allLineLocations();
443             int n =
444                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
445 
446             if (n > alllineLocations.size()) {
447                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
448             } else {
449                 lineLocation = (Location) alllineLocations.get(n);
450                 try {
451                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
452                     breakpRequest.putProperty("number", property);
453                     breakpRequest.addThreadFilter(thread);
454                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
455                 } catch ( Exception e1 ) {
456                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
457                     breakpRequest    = null;
458                 }
459             }
460         } catch ( Exception e2 ) {
461             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
462             breakpRequest    = null;
463         }
464 
465         if (breakpRequest == null) {
466             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
467             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
468         }
469 
470         log2("      a breakpoint has been set up");
471         return breakpRequest;
472     }

475     private void getEventSet()
476                  throws JDITestRuntimeException {
477         try {
478 //            log2("       eventSet = eventQueue.remove(waitTime);");
479             eventSet = eventQueue.remove(waitTime);
480             if (eventSet == null) {
481                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
482             }
483 //            log2("       eventIterator = eventSet.eventIterator;");
484             eventIterator = eventSet.eventIterator();
485         } catch ( Exception e ) {
486             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
487         }
488     }

491     private void breakpointForCommunication()
492                  throws JDITestRuntimeException {
493 
494         log2("breakpointForCommunication");
495         getEventSet();
496 
497         if (eventIterator.nextEvent() instanceof BreakpointEvent)
498             return;
499 
500         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
501     }

505     private void breakpointInMethod(String number)
506                  throws JDITestRuntimeException {
507 
508         Event event;
509 
510         getEventSet();
511         event = eventIterator.nextEvent();
512 
513         if ( !(event instanceof BreakpointEvent) )
514             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
515 
516         log2("---->: request().getProperty == " +
517              event.request().getProperty("number"));
518 
519         if ( event.request().getProperty("number").equals(number) )
520             return;
521 
522         throw new JDITestRuntimeException("** UNEXPECTED breakpoint **");
523     }

526     private Map<? extends com.sun.jdi.ReferenceType,byte[]> mapClassToBytes()
527                 throws JDITestRuntimeException {
528 
529         byte []arrayToRedefine;
530 
531 //   -----------------------------
532         log2("......getting: String[] args = argsHandler.getArguments();");
533         String[] args = argsHandler.getArguments();
534         if (args.length <= 0) {
535             log3("ERROR: args.length <= 0");
536             testExitCode = FAILED;
537             throw new JDITestRuntimeException("** args.length <= 0 **");
538         }
539         String testDir = args[0];
540         log2("...... testDir = " + testDir);
541 
542 
543         String filePrefix = File.separator + "nsk"
544                           + File.separator + "jdi"
545                           + File.separator + "VirtualMachine"
546                           + File.separator + "redefineClasses";
547 
548         String fileToRedefineName    = testDir
549                                        + File.separator + "newclass"
550                                        + filePrefix
551                                        + File.separator + "redefineclasses001b.class";
552 
553         log2("...... fileToRedefineName = " + fileToRedefineName);
554 //   -----------------------------
555 
556         try {
557             log2("......File fileToRedefine = new File(fileToRedefineName);");
558             File fileToRedefine    = new File(fileToRedefineName);
559             if (fileToRedefine.exists())
560                 log2("       fileToRedefine.exists()");
561             else {
562                 log3("ERROR: fileToRedefine doesn't exist");
563                 testExitCode = FAILED;
564             }
565 
566             if (testExitCode == FAILED)
567                 throw new JDITestRuntimeException("** testExitCode == FAILED **");
568 
569             int fileToRedefineLength = (int) fileToRedefine.length();
570             log2("       fileToRedefineLength == " + fileToRedefineLength);
571 
572             FileInputStream inputFile = new FileInputStream(fileToRedefine);
573             arrayToRedefine = new byte[fileToRedefineLength];
574             inputFile.read(arrayToRedefine);
575             inputFile.close();
576         } catch ( IOException e ) {
577             log3("ERROR: IOException: " + e);
578             throw new JDITestRuntimeException("** IOException **");
579         }
580 
581         String testClassName = "nsk.jdi.VirtualMachine.redefineClasses.redefineclasses001b";
582         List classes = vm.classesByName(testClassName);
583         ReferenceType testClass = (ReferenceType) classes.get(0);
584 
585         HashMap<com.sun.jdi.ReferenceType,byte[]> mapForClass = new HashMap<com.sun.jdi.ReferenceType,byte[]>();
586         mapForClass.put(testClass, arrayToRedefine);
587 
588         return mapForClass;
589     }
