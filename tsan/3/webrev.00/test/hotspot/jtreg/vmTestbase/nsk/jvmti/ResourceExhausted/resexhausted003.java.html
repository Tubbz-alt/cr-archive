 23 package nsk.jvmti.ResourceExhausted;
 24 
 25 import java.io.File;
 26 import java.io.FileInputStream;
 27 import java.io.PrintStream;
 28 import java.security.ProtectionDomain;
 29 import java.util.regex.Pattern;
 30 import java.util.regex.Matcher;
 31 
 32 import nsk.share.Consts;
 33 import nsk.share.test.Stresser;
 34 
 35 public class resexhausted003 {
 36 
 37     static final long MAX_ITERATIONS = Long.MAX_VALUE;
 38 
 39     static class MyClassLoader extends ClassLoader {
 40         Class<?> loadUm(String name, byte[] bytes, ProtectionDomain pd) throws ClassFormatError {
 41             Class k = defineClass(name, bytes, 0, bytes.length, pd);
 42             resolveClass(k);
 43             return k;
 44         }
 45 
 46         Class loadClass(String name, byte[] bytes) {
 47             try {
 48                 Class k = loadUm(name, bytes, getClass().getProtectionDomain());
 49                 k.newInstance(); // prepare class
 50                 return k;
 51             } catch (Exception exc) {
 52                 throw new RuntimeException("Exception in loadClass: " + name, exc);
 53             }
 54         }
 55     };
 56 
 57     static class Node {
 58         MyClassLoader loader = new MyClassLoader();
 59         Node next;
 60     };
 61 
 62 
 63     static byte[] fileBytes(String fileName) {
 64         try {
 65             File f = new File(fileName);
 66             FileInputStream fi = new FileInputStream(f);
 67             byte[] bytes;
 68             try {
 69                 bytes = new byte[(int) f.length()];
 70                 fi.read(bytes);
 71             } finally {
 72                 fi.close();
 73             }
 74             return bytes;
 75         } catch (Exception e) {
 76             throw new RuntimeException("Exception when reading file '" + fileName + "'", e);
 77         }
 78     }
 79 
 80 
 81     public static int run(String args[], PrintStream out) {
 82         String testclasspath = System.getProperty("test.class.path");
 83         String [] testpaths = testclasspath.split(System.getProperty("path.separator"));
 84         String classesDir = "";
 85 
 86         Pattern pattern = Pattern.compile("^(.*)classes(.*)vmTestbase(.*)$");
 87         for (int i = 0 ; i < testpaths.length; i++) {
 88             if (pattern.matcher(testpaths[i]).matches()) {
 89                 classesDir = testpaths[i];
 90             }
 91         }
 92         if (classesDir.equals("")) {
 93             System.err.println("TEST BUG: Classes directory not found in test,class.path.");
 94             return Consts.TEST_FAILED;
 95         }
 96         Stresser stress = new Stresser(args);
 97 
 98         String className = Helper.class.getName();
 99         byte[] bloatBytes = fileBytes(classesDir + File.separator + className.replace('.', '/') + ".class");
100 
101         int count = 0;
102         Helper.resetExhaustedEvent();
103 
104         out.println("Loading classes...");
105         stress.start(MAX_ITERATIONS);
106         try {
107             Node list = null;
108 
109             while ( stress.iteration() ) {
110                 Node n = new Node();
111                 n.next = list;
112                 list = n;
113                 n.loader.loadClass(className, bloatBytes);
114                 ++count;
115             }
116 
117             System.out.println("Can't reproduce OOME due to a limit on iterations/execution time. Test was useless.");
118             return Consts.TEST_PASSED;
119 
120         } catch (OutOfMemoryError e) {
121             // that is what we are waiting for
122         } finally {
123             stress.finish();
124         }
125 
126         System.gc();
127         if ( ! Helper.checkResult("loading " + count + " classes of " + bloatBytes.length + " bytes") )
128             return Consts.TEST_FAILED;
129 
130         return Consts.TEST_PASSED;
131     }
132 
133     public static void main(String[] args) {
134         args = nsk.share.jvmti.JVMTITest.commonInit(args);
135 
136         int result = run(args, System.out);
137         System.out.println(result == Consts.TEST_PASSED ? "TEST PASSED" : "TEST FAILED");
138         System.exit(result + Consts.JCK_STATUS_BASE);
139     }
140 }
