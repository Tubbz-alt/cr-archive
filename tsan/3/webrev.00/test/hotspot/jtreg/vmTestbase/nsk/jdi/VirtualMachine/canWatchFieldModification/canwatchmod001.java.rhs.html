 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.canWatchFieldModification()</code>  <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test has three phases and works as follows.              <BR>
 46  * <BR>
 47  * In first phase,                                                      <BR>
 48  * upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 52  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 53  * and waits for the event within the predefined time interval.         <BR>
 54  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 57  * within debuggee's special methodForCommunication().                  <BR>
 58  * <BR>
 59  * In second phase to check the assetion,                               <BR>
 60  * the debugger and the debuggee perform the following loop.            <BR>
 61  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 63  *   and to inform the debugger with the event.                         <BR>
 64  * - Upon getting the BreakpointEvent,                                  <BR>
 65  *   the debugger performs the check case required.                     <BR>
 66  * Note. To inform each other of needed actions, the debugger and       <BR>
 67  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 68  * <BR>
 69  * In third phase when at the end,                                      <BR>
 70  * the debuggee changes the value of the "instruction"                  <BR>
 71  * to inform the debugger of checks finished, and both end.             <BR>
 72  * <BR>
 73  */
 74

 99         int exitCode = new canwatchmod001().runThis(argv, out);

101         if (exitCode != PASSED) {
102             System.out.println("TEST FAILED");
103         }
104         return testExitCode;
105     }

147     private int runThis (String argv[], PrintStream out) {
148 
149         argsHandler     = new ArgumentHandler(argv);
150         logHandler      = new Log(out, argsHandler);
151         Binder binder   = new Binder(argsHandler, logHandler);
152 
153         waitTime        = argsHandler.getWaitTime() * 60000;
154 
155         try {
156             log2("launching a debuggee :");
157             log2("       " + debuggeeName);
158             if (argsHandler.verbose()) {
159                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
160             } else {
161                 debuggee = binder.bindToDebugee(debuggeeName);
162             }
163             if (debuggee == null) {
164                 log3("ERROR: no debuggee launched");
165                 return FAILED;
166             }
167             log2("debuggee launched");
168         } catch ( Exception e ) {
169             log3("ERROR: Exception : " + e);
170             log2("       test cancelled");
171             return FAILED;
172         }
173 
174         debuggee.redirectOutput(logHandler);
175 
176         vm = debuggee.VM();
177 
178         eventQueue = vm.eventQueue();
179         if (eventQueue == null) {
180             log3("ERROR: eventQueue == null : TEST ABORTED");
181             vm.exit(PASS_BASE);
182             return FAILED;
183         }
184 
185         log2("invocation of the method runTest()");
186         switch (runTest()) {
187 
188             case 0 :  log2("test phase has finished normally");
189                       log2("   waiting for the debuggee to finish ...");
190                       debuggee.waitFor();
191 
192                       log2("......getting the debuggee's exit status");
193                       int status = debuggee.getStatus();
194                       if (status != PASS_BASE) {
195                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
196                               status + " != PASS_BASE");
197                           testExitCode = FAILED;
198                       } else {
199                           log2("......debuggee returned expected exit status: " +
200                               status + " == PASS_BASE");
201                       }
202                       break;
203 
204             default : log3("ERROR: runTest() returned unexpected value");
205 
206             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
207                       log2("......forcing: vm.exit();");
208                       testExitCode = FAILED;
209                       try {
210                           vm.exit(PASS_BASE);
211                       } catch ( Exception e ) {
212                           log3("ERROR: Exception : e");
213                       }
214                       break;
215 
216             case 2 :  log3("test cancelled due to VMDisconnectedException");
217                       log2("......trying: vm.process().destroy();");
218                       testExitCode = FAILED;
219                       try {
220                           Process vmProcess = vm.process();
221                           if (vmProcess != null) {
222                               vmProcess.destroy();
223                           }
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : e");
226                       }
227                       break;
228             }
229 
230         return testExitCode;
231     }

234    /*
235     * Return value: 0 - normal end of the test
236     *               1 - ubnormal end of the test
237     *               2 - VMDisconnectedException while test phase
238     */
239 
240     private int runTest() {
241 
242         try {
243             testRun();
244 
245             log2("waiting for VMDeathEvent");
246             getEventSet();
247             if (eventIterator.nextEvent() instanceof VMDeathEvent)
248                 return 0;
249 
250             log3("ERROR: last event is not the VMDeathEvent");
251             return 1;
252         } catch ( VMDisconnectedException e ) {
253             log3("ERROR: VMDisconnectedException : " + e);
254             return 2;
255         } catch ( Exception e ) {
256             log3("ERROR: Exception : " + e);
257             return 1;
258         }
259 
260     }

262     private void testRun()
263                  throws JDITestRuntimeException, Exception {
264 
265         eventRManager = vm.eventRequestManager();
266 
267         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
268         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
269         cpRequest.addClassFilter(debuggeeName);
270 
271         cpRequest.enable();
272         vm.resume();
273         getEventSet();
274         cpRequest.disable();
275 
276         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
277         debuggeeClass = event.referenceType();
278 
279         if (!debuggeeClass.name().equals(debuggeeName))
280            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
281 
282         log2("      received: ClassPrepareEvent for debuggeeClass");
283 
284         String bPointMethod = "methodForCommunication";
285         String lineForComm  = "lineForComm";
286         BreakpointRequest bpRequest;
287 
288         try {
289             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
290                                           debuggeeClass,
291                                           bPointMethod, lineForComm, "zero");
292         } catch ( Exception e ) {
293             throw e;
294         }
295         bpRequest.enable();
296 
297     //------------------------------------------------------  testing section
298 
299         log1("     TESTING BEGINS");
300 
301         for (int i = 0; ; i++) {
302 
303             vm.resume();
304             breakpointForCommunication();
305 
306             int instruction = ((IntegerValue)
307                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
308 
309             if (instruction == 0) {
310                 vm.resume();
311                 break;
312             }
313 
314             log1("  new check: # " + i);
315 
316             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
317 
318             log2("...... vm.canWatchFieldModification()");
319             if ( vm.canWatchFieldModification() )
320                 log2("      vm.canWatchFieldModification() == true");
321             else
322                 log2("      vm.canWatchFieldModification() == false");
323 
324             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
325         }
326         log1("    TESTING ENDS");
327         return;
328     }

330    /*
331     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
332     *                                             String, String, String)
333     *
334     * It sets up a breakpoint at given line number within a given method in a given class
335     * for a given thread.
336     *
337     * Return value: BreakpointRequest object  in case of success
338     *
339     * JDITestRuntimeException   in case of an Exception thrown within the method
340     */
341 
342     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
343                                                   ReferenceType testedClass,
344                                                   String methodName,
345                                                   String bpLine,
346                                                   String property)
347             throws JDITestRuntimeException {
348 
349         log2("......setting up a breakpoint:");
350         log2("       thread: " + thread + "; class: " + testedClass +
351                         "; method: " + methodName + "; line: " + bpLine);
352 
353         List              alllineLocations = null;
354         Location          lineLocation     = null;
355         BreakpointRequest breakpRequest    = null;
356 
357         try {
358             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
359 
360             alllineLocations = method.allLineLocations();
361 
362             int n =
363                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
364             if (n > alllineLocations.size()) {
365                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
366             } else {
367                 lineLocation = (Location) alllineLocations.get(n);
368                 try {
369                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
370                     breakpRequest.putProperty("number", property);
371                     breakpRequest.addThreadFilter(thread);
372                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
373                 } catch ( Exception e1 ) {
374                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
375                     breakpRequest    = null;
376                 }
377             }
378         } catch ( Exception e2 ) {
379             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
380             breakpRequest    = null;
381         }
382 
383         if (breakpRequest == null) {
384             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
385             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
386         }
387 
388         log2("      a breakpoint has been set up");
389         return breakpRequest;
390     }

393     private void getEventSet()
394                  throws JDITestRuntimeException {
395         try {
396 //            log2("       eventSet = eventQueue.remove(waitTime);");
397             eventSet = eventQueue.remove(waitTime);
398             if (eventSet == null) {
399                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
400             }
401 //            log2("       eventIterator = eventSet.eventIterator;");
402             eventIterator = eventSet.eventIterator();
403         } catch ( Exception e ) {
404             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
405         }
406     }

408 
409     private void breakpointForCommunication()
410                  throws JDITestRuntimeException {
411 
412         log2("breakpointForCommunication");
413         getEventSet();
414 
415         if (eventIterator.nextEvent() instanceof BreakpointEvent)
416             return;
417 
418         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
419     }
