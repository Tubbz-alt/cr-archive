 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.thread()</code>                <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a ThreadReference object,                         <BR>
 46  * an argument of the method                                            <BR>
 47  *    EventRequestManager.createAccessWatchpointRequest(),              <BR>
 48  * is equal to one returned by the method StepRequest.thread()          <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates a tested thread                               <BR>
 68  *   and invokes the methodForCommunication to be suspended and         <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an StepRequest with the thread1,                         <BR>
 72  *   - gets the thread of the Request, the thread2,                     <BR>
 73  *     with the method StepRequest.thread(),                            <BR>
 74  *   - and compares the objects thrtead1 and thread2.                   <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : " + e);
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : " + e);
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

251     * Return value: 0 - normal end of the test
252     *               1 - ubnormal end of the test
253     *               2 - VMDisconnectedException while test phase
254     */

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         String bPointMethod = "methodForCommunication";
301         String lineForComm  = "lineForComm";
302 
303         ThreadReference   mainThread = threadByName("main");
304 
305         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
306                                              debuggeeClass,
307                                             bPointMethod, lineForComm, "zero");
308         bpRequest.enable();
309 
310     //------------------------------------------------------  testing section
311 
312         log1("     TESTING BEGINS");
313 
314         EventRequest  eventRequest1 = null;
315 
316         ThreadReference thread1     = null;
317         String          threadName1 = "thread1";
318 
319         ThreadReference thread2;
320 
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335 
336             log1(":::::: case: # " + i);
337 
338             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
339 
340             switch (i) {
341 
342               case 0:
343                      thread1 = threadByName(threadName1);
344 
345                      log2("......setting up StepRequest with size StepRequest.STEP_MIN");
346                      eventRequest1 = setting24StepRequest(thread1, StepRequest.STEP_MIN,
347                                               StepRequest.STEP_INTO);
348 
349                      log2("......getting: size = ((StepRequest) eventRequest1).thread();");
350                      thread2 = ((StepRequest) eventRequest1).thread();
351 
352                      log2("      checking up on equality of values");
353                      if ( !thread2.equals(thread1) ) {
354                          testExitCode = FAILED;
355                          log3("ERROR: !thread2.equals(thread1) : " + thread2);
356                      }
357 
358                       break;
359 
360               default:
361                       throw new JDITestRuntimeException("** default case 2 **");
362             }
363 
364             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
365         }
366         log1("    TESTING ENDS");
367         return;
368     }

370     private ThreadReference threadByName(String name)
371                  throws JDITestRuntimeException {
372 
373         List         all = vm.allThreads();
374         ListIterator li  = all.listIterator();
375 
376         for (; li.hasNext(); ) {
377             ThreadReference thread = (ThreadReference) li.next();
378             if (thread.name().equals(name))
379                 return thread;
380         }
381         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
382     }

384    /*
385     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
386     *                                             String, String, String)
387     *
388     * It sets up a breakpoint at given line number within a given method in a given class
389     * for a given thread.
390     *
391     * Return value: BreakpointRequest object  in case of success
392     *
393     * JDITestRuntimeException   in case of an Exception thrown within the method
394     */
395 
396     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
397                                                   ReferenceType testedClass,
398                                                   String methodName,
399                                                   String bpLine,
400                                                   String property)
401             throws JDITestRuntimeException {
402 
403         log2("......setting up a breakpoint:");
404         log2("       thread: " + thread + "; class: " + testedClass +
405                         "; method: " + methodName + "; line: " + bpLine);
406 
407         List              alllineLocations = null;
408         Location          lineLocation     = null;
409         BreakpointRequest breakpRequest    = null;
410 
411         try {
412             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
413 
414             alllineLocations = method.allLineLocations();
415 
416             int n =
417                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
418             if (n > alllineLocations.size()) {
419                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
420             } else {
421                 lineLocation = (Location) alllineLocations.get(n);
422                 try {
423                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
424                     breakpRequest.putProperty("number", property);
425                     breakpRequest.addThreadFilter(thread);
426                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
427                 } catch ( Exception e1 ) {
428                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
429                     breakpRequest    = null;
430                 }
431             }
432         } catch ( Exception e2 ) {
433             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
434             breakpRequest    = null;
435         }
436 
437         if (breakpRequest == null) {
438             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
439             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
440         }
441 
442         log2("      a breakpoint has been set up");
443         return breakpRequest;
444     }

447     private void getEventSet()
448                  throws JDITestRuntimeException {
449         try {
450 //            log2("       eventSet = eventQueue.remove(waitTime);");
451             eventSet = eventQueue.remove(waitTime);
452             if (eventSet == null) {
453                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
454             }
455 //            log2("       eventIterator = eventSet.eventIterator;");
456             eventIterator = eventSet.eventIterator();
457         } catch ( Exception e ) {
458             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
459         }
460     }

463     private void breakpointForCommunication()
464                  throws JDITestRuntimeException {
465 
466         log2("breakpointForCommunication");
467         getEventSet();
468 
469         if (eventIterator.nextEvent() instanceof BreakpointEvent)
470             return;
471 
472         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
473     }
