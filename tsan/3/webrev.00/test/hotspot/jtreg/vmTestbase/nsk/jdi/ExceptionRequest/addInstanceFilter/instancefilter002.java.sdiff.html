418     private ThreadReference threadByName(String name)
419                  throws JDITestRuntimeException {
420 
421         List         all = vm.allThreads();
422         ListIterator li  = all.listIterator();
423 
424         for (; li.hasNext(); ) {
425             ThreadReference thread = (ThreadReference) li.next();
426             if (thread.name().equals(name))
427                 return thread;
428         }
429         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
