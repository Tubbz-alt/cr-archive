 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.suspendPolicy()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on following assertions:                  <BR>
 46  *   - Returns: the current suspend mode for this request       <BR>
 47  * The cases to check include Requests of all sub-types.        <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.              <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assertion,                              <BR>
 64  * the debugger and the debuggee perform the following loop.            <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee prepares new check case (as if) and invokes           <BR>
 67  *   the methodForCommunication to be suspended and                     <BR>
 68  *   to inform the debugger with the event.                             <BR>
 69  * - Upon getting the BreakpointEvent,                                  <BR>
 70  *   the debugger performs the check case required.                     <BR>
 71  * <BR>
 72  * In third phase, at the end of the test, the debuggee changes         <BR>
 73  * the value of the "instruction" which the debugger and debuggee       <BR>
 74  * use to inform each other of needed actions,  and both end.           <BR>
 75  * <BR>
 76  */

155     private int runThis (String argv[], PrintStream out) {
156 
157         argsHandler     = new ArgumentHandler(argv);
158         logHandler      = new Log(out, argsHandler);
159         Binder binder   = new Binder(argsHandler, logHandler);
160 
161         waitTime        = argsHandler.getWaitTime() * 60000;
162 
163         try {
164             log2("launching a debuggee :");
165             log2("       " + debuggeeName);
166             if (argsHandler.verbose()) {
167                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
168             } else {
169                 debuggee = binder.bindToDebugee(debuggeeName);
170             }
171             if (debuggee == null) {
172                 log3("ERROR: no debuggee launched");
173                 return FAILED;
174             }
175             log2("debuggee launched");
176         } catch ( Exception e ) {
177             log3("ERROR: Exception : " + e);
178             log2("       test cancelled");
179             return FAILED;
180         }
181 
182         debuggee.redirectOutput(logHandler);
183 
184         vm = debuggee.VM();
185 
186         eventQueue = vm.eventQueue();
187         if (eventQueue == null) {
188             log3("ERROR: eventQueue == null : TEST ABORTED");
189             vm.exit(PASS_BASE);
190             return FAILED;
191         }
192 
193         log2("invocation of the method runTest()");
194         switch (runTest()) {
195 
196             case 0 :  log2("test phase has finished normally");
197                       log2("   waiting for the debuggee to finish ...");
198                       debuggee.waitFor();
199 
200                       log2("......getting the debuggee's exit status");
201                       int status = debuggee.getStatus();
202                       if (status != PASS_BASE) {
203                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
204                               status + " != PASS_BASE");
205                           testExitCode = FAILED;
206                       } else {
207                           log2("......debuggee returned expected exit status: " +
208                               status + " == PASS_BASE");
209                       }
210                       break;
211 
212             default : log3("ERROR: runTest() returned unexpected value");
213 
214             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
215                       log2("......forcing: vm.exit();");
216                       testExitCode = FAILED;
217                       try {
218                           vm.exit(PASS_BASE);
219                       } catch ( Exception e ) {
220                           log3("ERROR: Exception : " + e);
221                       }
222                       break;
223 
224             case 2 :  log3("test cancelled due to VMDisconnectedException");
225                       log2("......trying: vm.process().destroy();");
226                       testExitCode = FAILED;
227                       try {
228                           Process vmProcess = vm.process();
229                           if (vmProcess != null) {
230                               vmProcess.destroy();
231                           }
232                       } catch ( Exception e ) {
233                           log3("ERROR: Exception : " + e);
234                       }
235                       break;
236             }
237 
238         return testExitCode;
239     }

243     * Return value: 0 - normal end of the test
244     *               1 - ubnormal end of the test
245     *               2 - VMDisconnectedException while test phase

248     private int runTest() {
249 
250         try {
251             testRun();
252 
253             log2("waiting for VMDeathEvent");
254             getEventSet();
255             if (eventIterator.nextEvent() instanceof VMDeathEvent)
256                 return 0;
257 
258             log3("ERROR: last event is not the VMDeathEvent");
259             return 1;
260         } catch ( VMDisconnectedException e ) {
261             log3("ERROR: VMDisconnectedException : " + e);
262             return 2;
263         } catch ( Exception e ) {
264             log3("ERROR: Exception : " + e);
265             return 1;
266         }
267 
268     }

270     private void testRun()
271                  throws JDITestRuntimeException, Exception {
272 
273         eventRManager = vm.eventRequestManager();
274 
275         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
276         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
277         cpRequest.addClassFilter(debuggeeName);
278 
279         cpRequest.enable();
280         vm.resume();
281         getEventSet();
282         cpRequest.disable();
283 
284         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
285         debuggeeClass = event.referenceType();
286 
287         if (!debuggeeClass.name().equals(debuggeeName))
288            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
289 
290         log2("      received: ClassPrepareEvent for debuggeeClass");
291 
292         String bPointMethod = "methodForCommunication";
293         String lineForComm  = "lineForComm";
294 
295         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
296 
297         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
298                                              debuggeeClass,
299                                             bPointMethod, lineForComm, "zero");
300         bpRequest.enable();
301 
302     //------------------------------------------------------  testing section
303 
304 
305         EventRequest  eventRequest1 = null;
306 
307         String        fieldName1    = "var101";
308         String        fieldName2    = "excObj";
309 
310         ThreadReference thread1     = null;
311         String          threadName1 = "thread1";
312 
313         ReferenceType testClassReference = null;
314 
315         String propertyName   = "name";
316         String propertyValue1 = "value1";
317         String propertyValue2 = "value2";
318 
319 
320         log1("     TESTING BEGINS");
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339 
340             switch (i) {
341 
342               case 0:
343                      thread1 = debuggee.threadByNameOrThrow(threadName1);
344 
345                      log2("......setting up StepRequest");
346                      eventRequest1 = eventRManager.createStepRequest
347                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
348                      break;
349 
350               case 1:
351                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
352 
353                      log2("......setting up AccessWatchpointRequest");
354                      eventRequest1 = eventRManager.createAccessWatchpointRequest
355                                     (testClassReference.fieldByName(fieldName1));
356                      break;
357 
358               case 2:
359                      log2(".....setting up ModificationWatchpointRequest");
360                      eventRequest1 = eventRManager.createModificationWatchpointRequest
361                                     (testClassReference.fieldByName(fieldName1));
362                      break;
363 
364               case 3:
365                      log2(".....setting up ClassPrepareRequest");
366                      eventRequest1 = eventRManager.createClassPrepareRequest();
367                      break;
368 
369               case 4:
370                      log2(".....setting up ClassUnloadRequest");
371                      eventRequest1 = eventRManager.createClassUnloadRequest();
372                      break;
373 
374               case 5:
375                      log2(".....setting up MethodEntryRequest");
376                      eventRequest1 = eventRManager.createMethodEntryRequest();
377                      break;
378 
379               case 6:
380                      log2(".....setting up MethodExitRequest");
381                      eventRequest1 = eventRManager.createMethodExitRequest();
382                      break;
383 
384               case 7:
385                      log2(".....setting up ThreadDeathRequest");
386                      eventRequest1 = eventRManager.createThreadDeathRequest();
387                      break;
388 
389               case 8:
390                      log2(".....setting up ThreadStartRequest");
391                      eventRequest1 = eventRManager.createThreadStartRequest();
392                      break;
393 
394               case 9:
395                      log2(".....setting up VMDeathRequest");
396                      eventRequest1 = eventRManager.createVMDeathRequest();
397                      break;
398 
399               case 10:
400                      log2(".....setting up ExceptionRequest");
401                      eventRequest1 = eventRManager.createExceptionRequest(
402                                      (ReferenceType)
403                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
404                                      true, true );
405                      break;
406 
407               case 11:
408                      log2(".....setting up BreakpointRequest");
409                      eventRequest1 = eventRManager.createBreakpointRequest(location);
410                      break;
411 
412 
413               default:
414                       throw new JDITestRuntimeException("** default case 2 **");
415             }
416 
417             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_ALL);");
418             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
419 
420             log2("      checking up on eventRequest1");
421             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_ALL) {
422                 testExitCode = FAILED;
423                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_ALL");
424             }
425 
426             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD );");
427             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD );
428 
429             log2("      checking up on eventRequest1");
430             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_EVENT_THREAD ) {
431                 testExitCode = FAILED;
432                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_EVENT_THREAD ");
433             }
434 
435             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);");
436             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
437 
438             log2("      checking up on eventRequest1");
439             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_NONE) {
440                 testExitCode = FAILED;
441                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_NONE");
442             }
443 
444             eventRManager.deleteEventRequest(eventRequest1);
445 
446             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
447         }
448         log1("    TESTING ENDS");
449         return;
450     }

453    /*
454     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
455     *                                             String, String, String)
456     *
457     * It sets up a breakpoint at given line number within a given method in a given class
458     * for a given thread.
459     *
460     * Return value: BreakpointRequest object  in case of success
461     *
462     * JDITestRuntimeException   in case of an Exception thrown within the method
463     */
464 
465     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
466                                                   ReferenceType testedClass,
467                                                   String methodName,
468                                                   String bpLine,
469                                                   String property)
470             throws JDITestRuntimeException {
471 
472         log2("......setting up a breakpoint:");
473         log2("       thread: " + thread + "; class: " + testedClass +
474                         "; method: " + methodName + "; line: " + bpLine);
475 
476         List              alllineLocations = null;
477         Location          lineLocation     = null;
478         BreakpointRequest breakpRequest    = null;
479 
480         try {
481             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
482 
483             alllineLocations = method.allLineLocations();
484 
485             int n =
486                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
487             if (n > alllineLocations.size()) {
488                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
489             } else {
490                 lineLocation = (Location) alllineLocations.get(n);
491 
492 // this is only for this test to get Location object
493 location = lineLocation;
494 
495                 try {
496                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
497                     breakpRequest.putProperty("number", property);
498                     breakpRequest.addThreadFilter(thread);
499                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
500                 } catch ( Exception e1 ) {
501                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
502                     breakpRequest    = null;
503                 }
504             }
505         } catch ( Exception e2 ) {
506             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
507             breakpRequest    = null;
508         }
509 
510         if (breakpRequest == null) {
511             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
512             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
513         }
514 
515         log2("      a breakpoint has been set up");
516         return breakpRequest;
517     }

520     private void getEventSet()
521                  throws JDITestRuntimeException {
522         try {
523 //            log2("       eventSet = eventQueue.remove(waitTime);");
524             eventSet = eventQueue.remove(waitTime);
525             if (eventSet == null) {
526                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
527             }
528 //            log2("       eventIterator = eventSet.eventIterator;");
529             eventIterator = eventSet.eventIterator();
530         } catch ( Exception e ) {
531             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
532         }
533     }

537                  throws JDITestRuntimeException {
538

540         getEventSet();
541 
542         if (eventIterator.nextEvent() instanceof BreakpointEvent)
543             return;
544 
545         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
546     }
