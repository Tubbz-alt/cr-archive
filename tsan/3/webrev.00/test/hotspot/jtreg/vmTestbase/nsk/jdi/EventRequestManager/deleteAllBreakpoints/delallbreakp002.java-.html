 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.deleteAllBreakpoints()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - All breakpoints are deleted, that is, following call to            <BR>
 47  *       EventRequestManager.breakpointRequests()                       <BR>
 48  *   returns empty list.                                                <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debugger resumes         <BR>
 74  * the debuggee,  and both end.                                         <BR>
 75  * <BR>
 76  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase
250     */

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = threadByName("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         List         requests = null;
309         ListIterator li       = null;
310 
311         BreakpointRequest request = null;
312         BreakpointRequest bpRequests[] = { null, null, null, null, null,
313                                            null, null, null, null, null };
314         int listSize;
315         int flag;
316 
317 
318         log1("     TESTING BEGINS");
319 
320             vm.resume();
321             breakpointForCommunication();
322 
323             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
324 
325             log2("......creating BreakpointRequests");
326             for (int i1 = 0; i1 < bpRequests.length; i1++) {
327                 bpRequests[i1] = eventRManager.createBreakpointRequest(breakpLocation);
328                 bpRequests[i1].putProperty("number", String.valueOf(i1));
329             }
330 
331             log2("......deleting all BreakpointRequests");
332             eventRManager.deleteAllBreakpoints();
333             log2("...... getting new list of BreakpointRequests");
334             requests = eventRManager.breakpointRequests();
335 
336             log2("...... checking up on size of the list;  0 is expected");
337             listSize = requests.size();
338             if ( listSize != 0 ) {
339                 testExitCode = FAILED;
340                 log3("ERROR: size of returned List is not equal to expected 0 ");
341             }
342 
343             vm.resume();
344 
345             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
346 
347         log1("    TESTING ENDS");
348         return;
349     }

351     private ThreadReference threadByName(String name)
352                  throws JDITestRuntimeException {
353 
354         List         all = vm.allThreads();
355         ListIterator li  = all.listIterator();
356 
357         for (; li.hasNext(); ) {
358             ThreadReference thread = (ThreadReference) li.next();
359             if (thread.name().equals(name))
360                 return thread;
361         }
362         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
363     }

365    /*
366     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
367     *                                             String, String, String)
368     *
369     * It sets up a breakpoint at given line number within a given method in a given class
370     * for a given thread.
371     *
372     * Return value: BreakpointRequest object  in case of success
373     *
374     * JDITestRuntimeException   in case of an Exception thrown within the method
375     */
376 
377     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
378                                                   ReferenceType testedClass,
379                                                   String methodName,
380                                                   String bpLine,
381                                                   String property)
382             throws JDITestRuntimeException {
383 
384         log2("......setting up a breakpoint:");
385         log2("       thread: " + thread + "; class: " + testedClass +
386                         "; method: " + methodName + "; line: " + bpLine);
387 
388         List              alllineLocations = null;
389         Location          lineLocation     = null;
390         BreakpointRequest breakpRequest    = null;
391 
392         try {
393             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
394 
395             alllineLocations = method.allLineLocations();
396 
397             int n =
398                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
399             if (n > alllineLocations.size()) {
400                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
401             } else {
402                 lineLocation = (Location) alllineLocations.get(n);
403 breakpLocation = lineLocation;
404                 try {
405                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
406                     breakpRequest.putProperty("number", property);
407                     breakpRequest.addThreadFilter(thread);
408                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
409                 } catch ( Exception e1 ) {
410                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
411                     breakpRequest    = null;
412                 }
413             }
414         } catch ( Exception e2 ) {
415             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
416             breakpRequest    = null;
417         }
418 
419         if (breakpRequest == null) {
420             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
421             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
422         }
423 
424         log2("      a breakpoint has been set up");
425         return breakpRequest;
426     }

429     private void getEventSet()
430                  throws JDITestRuntimeException {
431         try {
432 //            log2("       eventSet = eventQueue.remove(waitTime);");
433             eventSet = eventQueue.remove(waitTime);
434             if (eventSet == null) {
435                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
436             }
437 //            log2("       eventIterator = eventSet.eventIterator;");
438             eventIterator = eventSet.eventIterator();
439         } catch ( Exception e ) {
440             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
441         }
442     }

445     private void breakpointForCommunication()
446                  throws JDITestRuntimeException {
447 
448         log2("breakpointForCommunication");
449         getEventSet();

452             return;
