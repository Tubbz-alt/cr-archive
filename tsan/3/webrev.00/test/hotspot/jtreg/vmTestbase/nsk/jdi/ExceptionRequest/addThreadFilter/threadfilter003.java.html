 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addThreadFilter()</code>  <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Restricts the events generated by this request            <BR>
 47  *    to those in the given thread.                             <BR>
 48  * The cases to test include three thread's states:             <BR>
 49  * not started, running, dead.                                  <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new thread, thread1, and invokes              <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates ExceptionRequest1 and                                    <BR>
 73  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 74  *   - creates ExceptiontRequest2,                                      <BR>
 75  *     resumes the debuggee to get the thread1 started, and             <BR>
 76  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 77  *   - creates ExceptiontRequest3, waits the thread1 to die, and        <BR>
 78  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  */
 86

109     public static int run (String argv[], PrintStream out) {
110 
111         int exitCode = new threadfilter003().runThis(argv, out);

113         if (exitCode != PASSED) {
114             System.out.println("TEST FAILED");
115         }
116         return testExitCode;
117     }

162     private int runThis (String argv[], PrintStream out) {
163 
164         argsHandler     = new ArgumentHandler(argv);
165         logHandler      = new Log(out, argsHandler);
166         Binder binder   = new Binder(argsHandler, logHandler);
167 
168         waitTime        = argsHandler.getWaitTime() * 60000;
169 
170         try {
171             log2("launching a debuggee :");
172             log2("       " + debuggeeName);
173             if (argsHandler.verbose()) {
174                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
175             } else {
176                 debuggee = binder.bindToDebugee(debuggeeName);
177             }
178             if (debuggee == null) {
179                 log3("ERROR: no debuggee launched");
180                 return FAILED;
181             }
182             log2("debuggee launched");
183         } catch ( Exception e ) {
184             log3("ERROR: Exception : " + e);
185             log2("       test cancelled");
186             return FAILED;
187         }
188 
189         debuggee.redirectOutput(logHandler);
190 
191         vm = debuggee.VM();
192 
193         eventQueue = vm.eventQueue();
194         if (eventQueue == null) {
195             log3("ERROR: eventQueue == null : TEST ABORTED");
196             vm.exit(PASS_BASE);
197             return FAILED;
198         }
199 
200         log2("invocation of the method runTest()");
201         switch (runTest()) {
202 
203             case 0 :  log2("test phase has finished normally");
204                       log2("   waiting for the debuggee to finish ...");
205                       debuggee.waitFor();
206 
207                       log2("......getting the debuggee's exit status");
208                       int status = debuggee.getStatus();
209                       if (status != PASS_BASE) {
210                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
211                               status + " != PASS_BASE");
212                           testExitCode = FAILED;
213                       } else {
214                           log2("......debuggee returned expected exit status: " +
215                               status + " == PASS_BASE");
216                       }
217                       break;
218 
219             default : log3("ERROR: runTest() returned unexpected value");
220 
221             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
222                       log2("......forcing: vm.exit();");
223                       testExitCode = FAILED;
224                       try {
225                           vm.exit(PASS_BASE);
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : " + e);
228                       }
229                       break;
230 
231             case 2 :  log3("test cancelled due to VMDisconnectedException");
232                       log2("......trying: vm.process().destroy();");
233                       testExitCode = FAILED;
234                       try {
235                           Process vmProcess = vm.process();
236                           if (vmProcess != null) {
237                               vmProcess.destroy();
238                           }
239                       } catch ( Exception e ) {
240                           log3("ERROR: Exception : " + e);
241                       }
242                       break;
243             }
244 
245         return testExitCode;
246     }

249    /*
250     * Return value: 0 - normal end of the test
251     *               1 - ubnormal end of the test
252     *               2 - VMDisconnectedException while test phase
253     */

255     private int runTest() {
256 
257         try {
258             testRun();
259 
260             log2("waiting for VMDeathEvent");
261             getEventSet();
262             if (eventIterator.nextEvent() instanceof VMDeathEvent)
263                 return 0;
264 
265             log3("ERROR: last event is not the VMDeathEvent");
266             return 1;
267         } catch ( VMDisconnectedException e ) {
268             log3("ERROR: VMDisconnectedException : " + e);
269             return 2;
270         } catch ( Exception e ) {
271             log3("ERROR: Exception : " + e);
272             return 1;
273         }
274 
275     }

277     private void testRun()
278                  throws JDITestRuntimeException, Exception {
279 
280         eventRManager = vm.eventRequestManager();
281 
282         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
283         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
284         cpRequest.addClassFilter(debuggeeName);
285 
286         cpRequest.enable();
287         vm.resume();
288         getEventSet();
289         cpRequest.disable();
290 
291         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
292         debuggeeClass = event.referenceType();
293 
294         if (!debuggeeClass.name().equals(debuggeeName))
295            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
296 
297         log2("      received: ClassPrepareEvent for debuggeeClass");
298 
299         String bPointMethod = "methodForCommunication";
300         String lineForComm  = "lineForComm";
301 
302         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
303 
304         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
305                                              debuggeeClass,
306                                             bPointMethod, lineForComm, "zero");
307         bpRequest.enable();
308 
309     //------------------------------------------------------  testing section
310 
311         log1("     TESTING BEGINS");
312 
313         EventRequest eventRequest1 = null;
314         EventRequest eventRequest2 = null;
315         EventRequest eventRequest3 = null;
316 
317         ThreadReference thread1     = null;
318         String          thread1Name = "thread1";
319 
320         String property1 = "ExceptionRequest1";
321         String property2 = "ExceptionRequest2";
322         String property3 = "ExceptionRequest3";
323 
324         ReferenceType testClassReference = null;
325 
326         for (int i = 0; ; i++) {
327 
328             vm.resume();
329             breakpointForCommunication();
330 
331             int instruction = ((IntegerValue)
332                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
333 
334             if (instruction == 0) {
335                 vm.resume();
336                 break;
337             }
338 
339             log1(":::::: case: # " + i);
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
342 
343             switch (i) {
344 
345               case 0:
346                      testClassReference =
347                             (ReferenceType) vm.classesByName(testedClassName).get(0);
348 
349                      thread1 = (ThreadReference) debuggeeClass.getValue(
350                                         debuggeeClass.fieldByName(thread1Name));
351 
352                      eventRequest1 = setting2ExceptionRequest (null,
353                                              testClassReference,
354                                              EventRequest.SUSPEND_NONE, property1);
355 
356                      try {
357                          log2("......eventRequest1.addThreadFilter(thread1);");
358                          log2("        no Exception expected");
359                          ((ExceptionRequest)eventRequest1).addThreadFilter(thread1);
360                      } catch ( Exception e ) {
361                          log3("ERROR: Exception :: " + e);
362                          testExitCode = FAILED;
363                      }
364                      break;
365 
366               case 1:
367                      eventRequest2 = setting2ExceptionRequest (null,
368                                              testClassReference,
369                                              EventRequest.SUSPEND_NONE, property2);
370                      try {
371                          log2("......eventRequest2.addThreadFilter(thread1);");
372                          log2("        no Exception expected");
373                          ((ExceptionRequest)eventRequest2).addThreadFilter(thread1);
374                      } catch ( Exception e ) {
375                          log3("ERROR: Exception :: " + e);
376                          testExitCode = FAILED;
377                      }
378                      break;
379 
380               case 2:
381                      eventRequest3 = setting2ExceptionRequest (null,
382                                              testClassReference,
383                                              EventRequest.SUSPEND_NONE, property3);
384                      try {
385                          log2("......eventRequest3.addThreadFilter(thread1);");
386                          log2("        no Exception expected");
387                          ((ExceptionRequest)eventRequest3).addThreadFilter(thread1);
388                      } catch ( Exception e ) {
389                          log3("ERROR: Exception :: " + e);
390                          testExitCode = FAILED;
391                      }
392                      break;
393 
394               default:
395                       throw new JDITestRuntimeException("** default case 2 **");
396             }
397 
398             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
399         }
400         log1("    TESTING ENDS");
401         return;
402     }

404    /*
405     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
406     *                                             String, String, String)
407     *
408     * It sets up a breakpoint at given line number within a given method in a given class
409     * for a given thread.
410     *
411     * Return value: BreakpointRequest object  in case of success
412     *
413     * JDITestRuntimeException   in case of an Exception thrown within the method
414     */
415 
416     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
417                                                   ReferenceType testedClass,
418                                                   String methodName,
419                                                   String bpLine,
420                                                   String property)
421             throws JDITestRuntimeException {
422 
423         log2("......setting up a breakpoint:");
424         log2("       thread: " + thread + "; class: " + testedClass +
425                         "; method: " + methodName + "; line: " + bpLine);
426 
427         List              alllineLocations = null;
428         Location          lineLocation     = null;
429         BreakpointRequest breakpRequest    = null;
430 
431         try {
432             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
433 
434             alllineLocations = method.allLineLocations();
435 
436             int n =
437                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
438             if (n > alllineLocations.size()) {
439                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
440             } else {
441                 lineLocation = (Location) alllineLocations.get(n);
442                 try {
443                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
444                     breakpRequest.putProperty("number", property);
445                     breakpRequest.addThreadFilter(thread);
446                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
447                 } catch ( Exception e1 ) {
448                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
449                     breakpRequest    = null;
450                 }
451             }
452         } catch ( Exception e2 ) {
453             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
454             breakpRequest    = null;
455         }
456 
457         if (breakpRequest == null) {
458             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
459             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
460         }
461 
462         log2("      a breakpoint has been set up");
463         return breakpRequest;
464     }

467     private void getEventSet()
468                  throws JDITestRuntimeException {
469         try {
470 //            log2("       eventSet = eventQueue.remove(waitTime);");
471             eventSet = eventQueue.remove(waitTime);
472             if (eventSet == null) {
473                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
474             }
475 //            log2("       eventIterator = eventSet.eventIterator;");
476             eventIterator = eventSet.eventIterator();
477         } catch ( Exception e ) {
478             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
479         }
480     }

483     private void breakpointForCommunication()
484                  throws JDITestRuntimeException {
485 
486         log2("breakpointForCommunication");
487         getEventSet();
488 
489         if (eventIterator.nextEvent() instanceof BreakpointEvent)
490             return;
491 
492         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
493     }

497     private ExceptionRequest setting2ExceptionRequest ( ThreadReference thread,
498                                                        ReferenceType   testedClass,
499                                                        int             suspendPolicy,
500                                                        String          property       )
501             throws JDITestRuntimeException {
502         try {
503             log2("......setting up ExceptionRequest:");
504             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
505 
506             ExceptionRequest
507             excr = eventRManager.createExceptionRequest(null, true, true);
508             excr.putProperty("number", property);
509             if (thread != null)
510                 excr.addThreadFilter(thread);
511             excr.addClassFilter(testedClass);
512             excr.setSuspendPolicy(suspendPolicy);
513 
514             log2("      ExceptionRequest has been set up");
515             return excr;
516         } catch ( Exception e ) {
517             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
518             log3("       ExceptionRequest HAS NOT BEEN SET UP");
519             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
520         }
