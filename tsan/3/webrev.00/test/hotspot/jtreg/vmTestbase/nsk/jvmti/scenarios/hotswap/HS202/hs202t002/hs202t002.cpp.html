 43 static jint redefineNumber = 0;
 44 static jvmtiEnv * jvmti = NULL;

 46 static volatile bool thread_suspend_error = false;

 48 void JNICALL callbackMethodExit(jvmtiEnv *jvmti_env,
 49                                 JNIEnv* jni_env,
 50                                 jthread thread,
 51                                 jmethodID method,
 52                                 jboolean was_popped_by_exception,
 53                                 jvalue return_value) {
 54     if (was_popped_by_exception) {
 55         char * name;
 56         char * signature;
 57         char * generic ;
 58         jvmtiError err;
 59         err= JVMTI_ERROR_NONE;
 60         jvmti_env->GetMethodName(method, &name, &signature, &generic);
 61         if (strcmp(name,METHOD_NAME) == 0) {
 62             jclass cls;
 63             char fileName[512];
 64             nsk_jvmti_getFileName(redefineNumber, FILE_NAME, fileName,
 65                                   sizeof(fileName)/sizeof(char));
 66             jvmti_env->GetMethodDeclaringClass(method, &cls);
 67             if (nsk_jvmti_redefineClass(jvmti_env, cls,fileName)) {
 68                 nsk_printf(" Agent:: redefine class success ..\n");
 69                 nsk_printf("Agent::SUSPENDING>> \n");
 70                 err=jvmti_env->SuspendThread(thread);
 71                 if (err == JVMTI_ERROR_NONE) {
 72                   // we don't get here until we are resumed
 73                     nsk_printf("Agent:: Thread successfully suspended and was resumed\n");
 74                 } else {
 75                     thread_suspend_error = true;
 76                     nsk_printf(" ## Error occured %s \n",TranslateError(err));
 77                 }
 78             }
 79         }
 80     }
 81 }

 94 jint  Agent_Initialize(JavaVM *vm, char *options, void *reserved) {
 95     jint rc ;
 96     nsk_printf("Agent:: VM.. Started..\n");
 97     redefineNumber=0;
 98     rc=vm->GetEnv((void **)&jvmti, JVMTI_VERSION_1_1);
 99     if (rc != JNI_OK) {
100         nsk_printf("Agent:: Could not load JVMTI interface \n");
101         return JNI_ERR;
102     } else {
103         jvmtiCapabilities caps;
104         jvmtiEventCallbacks eventCallbacks;
105         memset(&caps, 0, sizeof(caps));
106         if (!nsk_jvmti_parseOptions(options)) {
107             nsk_printf("# error agent Failed to parse options \n");
108             return JNI_ERR;
109         }
110         caps.can_redefine_classes = 1;
111         caps.can_suspend = 1;
112         caps.can_pop_frame = 1;
113         caps.can_generate_method_exit_events = 1;
114         jvmti->AddCapabilities(&caps);
115         memset(&eventCallbacks, 0, sizeof(eventCallbacks));
116         eventCallbacks.MethodExit = callbackMethodExit;
117         rc=jvmti->SetEventCallbacks(&eventCallbacks, sizeof(eventCallbacks));
118         if (rc != JVMTI_ERROR_NONE) {
119             nsk_printf(" Agent:: Error occured while setting event callbacks \n");
120             return JNI_ERR;
121         }
122         if (nsk_jvmti_enableNotification(jvmti,JVMTI_EVENT_METHOD_EXIT, NULL)) {
123             nsk_printf(" Agent :: NOTIFICATIONS ARE ENABLED \n");
124         } else {
125             nsk_printf(" Agent :: Error Enabling Notifications..");
126         }
127     }
128     return JNI_OK;
129 }

131 JNIEXPORT jboolean JNICALL
132 Java_nsk_jvmti_scenarios_hotswap_HS202_hs202t002_hs202t002_popThreadFrame(JNIEnv * jni,
133                                                                           jclass clas,
134                                                                           jthread thread) {
135     jvmtiError err = JVMTI_ERROR_NONE;
136     jboolean retvalue = JNI_FALSE;
137     jint state;
138     nsk_printf("Agent:: POPPING THE FRAME..\n");
139     jvmti->GetThreadState(thread, &state);
140     if (state & JVMTI_THREAD_STATE_SUSPENDED) {
141         err = jvmti->PopFrame(thread);
142         if (err == JVMTI_ERROR_NONE) {
143             nsk_printf("Agent:: PopFrame succeeded..\n");
144             return JNI_TRUE;
145         } else  {
146             nsk_printf(" ## Error occured %s \n",TranslateError(err));
147         }
148     } else {
149         nsk_printf("Agent:: Thread was not suspened.. check for capabilities, and java method signature ");
150     }
151     return retvalue;
152 }

154 JNIEXPORT jboolean JNICALL
155 Java_nsk_jvmti_scenarios_hotswap_HS202_hs202t002_hs202t002_resumeThread(JNIEnv * jni,
156                                                                         jclass clas,
157                                                                         jthread thread) {
158     jvmtiError err = JVMTI_ERROR_NONE;
159     jboolean retvalue = JNI_FALSE;
160 
161     // disable notifications before resuming thread
162     // to avoid recursion on PopFrame issued reinvoke
163     if (nsk_jvmti_disableNotification(jvmti,JVMTI_EVENT_METHOD_EXIT, NULL)) {
164         nsk_printf("Agent :: nsk_jvmti_disabled notifications..\n");
165     } else {
166         nsk_printf("Agent :: Failed to disable notifications..");
167         return JNI_FALSE;
168     }
169 
170     err = jvmti->ResumeThread(thread);
171     if (err == JVMTI_ERROR_NONE) {
172         retvalue = JNI_TRUE;
173         nsk_printf(" Agent:: Thread Resumed.. \n");
174     } else {
175         nsk_printf(" Agent:: Failed.. to Resume the thread.\n");
176         retvalue = JNI_FALSE;
177     }
178     return retvalue;
179 }

181 JNIEXPORT jboolean JNICALL
182 Java_nsk_jvmti_scenarios_hotswap_HS202_hs202t002_hs202t002_isThreadSuspended(JNIEnv* jni,
183                                                                              jclass clas,
184                                                                              jthread thread) {
185     if (thread_suspend_error) {
186         jclass ex_class = jni->FindClass("java/lang/IllegalThreadStateException");
187         jni->ThrowNew(ex_class, "Thread has failed to self suspend");
188         return JNI_FALSE;
189     }
190 
191     // There is an inherent race here if the suspend fails for some reason but
192     // thread_suspend_error is not yet set. But as long as we report the suspend
193     // state correctly there is no problem as the Java code will simply loop and call
194     // this again until we see thread_suspend_error is true.
195 
196     jint state = 0;
197     // No errors possible here: thread is valid, and state is not NULL
198     jvmti->GetThreadState(thread, &state);
199     return (state & JVMTI_THREAD_STATE_SUSPENDED) != 0;
200 }
