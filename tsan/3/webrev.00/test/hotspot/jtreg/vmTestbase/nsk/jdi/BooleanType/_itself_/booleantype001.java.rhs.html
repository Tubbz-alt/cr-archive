 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BooleanType.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that for a mirror of a boolean value the casts<BR>
 42  * from Type to BooleanType and from PrimitiveType to BooleanType<BR>
 43  * don't throw ClassCastException.                              <BR>
 44  * <BR>
 45  * The test has three phases and works as follows.              <BR>
 46  * <BR>
 47  * In first phase,                                                      <BR>
 48  * upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 52  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 53  * and waits for the event within the predefined time interval.         <BR>
 54  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 57  * within debuggee's special methodForCommunication().                  <BR>
 58  * <BR>
 59  * In second phase to check the casts,                                  <BR>
 60  * the debugger and the debuggee perform the following.                 <BR>
 61  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * - The debuggee prepares a boolean field and invokes                  <BR>
 63  *   the methodForCommunication to be suspended and                     <BR>
 64  *   to inform the debugger with the event.                             <BR>
 65  * - Upon getting the BreakpointEvent,                                  <BR>
 66  *   the debugger performs the checks.                                  <BR>
 67  * Note. To inform each other of needed actions, the debugger and       <BR>
 68  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 69  * <BR>
 70  * In third phase when at the end,                                      <BR>
 71  * the debuggee changes the value of the "instruction"                  <BR>
 72  * to inform the debugger of checks finished, and both end.             <BR>
 73  * <BR>
 74  */
 75

 80     static final int FAILED = 2;
 81     static final int PASS_BASE = 95;

 98     public static int run (String argv[], PrintStream out) {
 99 
100         int exitCode = new booleantype001().runThis(argv, out);
101 
102         if (exitCode != PASSED) {
103             System.out.println("TEST FAILED");
104         }
105         return testExitCode;
106     }

147     private int runThis (String argv[], PrintStream out) {
148 
149         argsHandler     = new ArgumentHandler(argv);
150         logHandler      = new Log(out, argsHandler);
151         Binder binder   = new Binder(argsHandler, logHandler);
152 
153         waitTime        = argsHandler.getWaitTime() * 60000;
154 
155         try {
156             log2("launching a debuggee :");
157             log2("       " + debuggeeName);
158             if (argsHandler.verbose()) {
159                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
160             } else {
161                 debuggee = binder.bindToDebugee(debuggeeName);
162             }
163             if (debuggee == null) {
164                 log3("ERROR: no debuggee launched");
165                 return FAILED;
166             }
167             log2("debuggee launched");
168         } catch ( Exception e ) {
169             log3("ERROR: Exception : " + e);
170             log2("       test cancelled");
171             return FAILED;
172         }
173 
174         debuggee.redirectOutput(logHandler);
175 
176         vm = debuggee.VM();
177 
178         eventQueue = vm.eventQueue();
179         if (eventQueue == null) {
180             log3("ERROR: eventQueue == null : TEST ABORTED");
181             vm.exit(PASS_BASE);
182             return FAILED;
183         }
184 
185         log2("invocation of the method runTest()");
186         switch (runTest()) {
187 
188             case 0 :  log2("test phase has finished normally");
189                       log2("   waiting for the debuggee to finish ...");
190                       debuggee.waitFor();
191 
192                       log2("......getting the debuggee's exit status");
193                       int status = debuggee.getStatus();
194                       if (status != PASS_BASE) {
195                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
196                               status + " != PASS_BASE");
197                           testExitCode = FAILED;
198                       } else {
199                           log2("......debuggee returned expected exit status: " +
200                               status + " == PASS_BASE");
201                       }
202                       break;
203 
204             default : log3("ERROR: runTest() returned unexpected value");
205 
206             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
207                       log2("......forcing: vm.exit();");
208                       testExitCode = FAILED;
209                       try {
210                           vm.exit(PASS_BASE);
211                       } catch ( Exception e ) {
212                           log3("ERROR: Exception : e");
213                       }
214                       break;
215 
216             case 2 :  log3("test cancelled due to VMDisconnectedException");
217                       log2("......trying: vm.process().destroy();");
218                       testExitCode = FAILED;
219                       try {
220                           Process vmProcess = vm.process();
221                           if (vmProcess != null) {
222                               vmProcess.destroy();
223                           }
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : e");
226                       }
227                       break;
228             }
229 
230         return testExitCode;
231     }

234    /*
235     * Return value: 0 - normal end of the test
236     *               1 - ubnormal end of the test
237     *               2 - VMDisconnectedException while test phase
238     */
239 
240     private int runTest() {
241 
242         try {
243             testRun();
244 
245             log2("waiting for VMDeathEvent");
246             getEventSet();
247             if (eventIterator.nextEvent() instanceof VMDeathEvent)
248                 return 0;
249 
250             log3("ERROR: last event is not the VMDeathEvent");
251             return 1;
252         } catch ( VMDisconnectedException e ) {
253             log3("ERROR: VMDisconnectedException : " + e);
254             return 2;
255         } catch ( Exception e ) {
256             log3("ERROR: Exception : " + e);
257             return 1;
258         }
259 
260     }

262     private void testRun()
263                  throws JDITestRuntimeException, Exception {
264 
265         eventRManager = vm.eventRequestManager();
266 
267         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
268         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
269         cpRequest.addClassFilter(debuggeeName);
270 
271         cpRequest.enable();
272         vm.resume();
273         getEventSet();
274         cpRequest.disable();
275 
276         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
277         debuggeeClass = event.referenceType();
278 
279         if (!debuggeeClass.name().equals(debuggeeName))
280            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
281 
282         log2("      received: ClassPrepareEvent for debuggeeClass");
283 
284         String bPointMethod = "methodForCommunication";
285         String lineForComm  = "lineForComm";
286         BreakpointRequest bpRequest;
287 
288         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
289                                       debuggeeClass,
290                                       bPointMethod, lineForComm, "zero");
291         bpRequest.enable();
292 
293     //------------------------------------------------------  testing section
294 
295         log1("     TESTING BEGINS");
296 
297         for (int i = 0; ; i++) {
298 
299             vm.resume();
300             breakpointForCommunication();
301 
302             int instruction = ((IntegerValue)
303                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
304 
305             if (instruction == 0) {
306                 vm.resume();
307                 break;
308             }
309 
310             log1(":::::: case: # " + i);
311 
312             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
313 
314             log2("......String name = 'bl';");
315             String name = "bl";
316 
317             log2("      getting: Type type = debuggeeClass.getValue(debuggeeClass.fieldByName(name)).type();");
318             Type type = debuggeeClass.getValue(debuggeeClass.fieldByName(name)).type();
319             log2("      PrimitiveType primitiveType = (PrimitiveType) type;");
320             PrimitiveType primitiveType = (PrimitiveType) type;
321 
322             BooleanType booleanType = null;
323             try {
324                 log2("      checking up on cast: BooleanType booleanType = (BooleanType) type;");
325                     booleanType = (BooleanType) type;
326             } catch ( ClassCastException e ) {
327                     testExitCode = FAILED;
328                     log3("ERROR: ClassCastException");
329             }
330             try {
331                 log2("      checking up on cast: BooleanType booleanType = (BooleanType) primitiveType;");
332                     booleanType = (BooleanType) primitiveType;
333             } catch ( ClassCastException e ) {
334                     testExitCode = FAILED;
335                     log3("ERROR: ClassCastException");
336             }
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
338         }
339         log1("    TESTING ENDS");
340         return;
341     }

343    /*
344     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
345     *                                             String, String, String)
346     *
347     * It sets up a breakpoint at given line number within a given method in a given class
348     * for a given thread.
349     *
350     * Return value: BreakpointRequest object  in case of success
351     *
352     * JDITestRuntimeException   in case of an Exception thrown within the method
353     */
354 
355     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
356                                                   ReferenceType testedClass,
357                                                   String methodName,
358                                                   String bpLine,
359                                                   String property)
360             throws JDITestRuntimeException {
361 
362         log2("......setting up a breakpoint:");
363         log2("       thread: " + thread + "; class: " + testedClass +
364                         "; method: " + methodName + "; line: " + bpLine);
365 
366         List              alllineLocations = null;
367         Location          lineLocation     = null;
368         BreakpointRequest breakpRequest    = null;
369 
370         try {
371             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
372 
373             alllineLocations = method.allLineLocations();
374 
375             int n =
376                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
377             if (n > alllineLocations.size()) {
378                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
379             } else {
380                 lineLocation = (Location) alllineLocations.get(n);
381                 try {
382                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
383                     breakpRequest.putProperty("number", property);
384                     breakpRequest.addThreadFilter(thread);
385                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
386                 } catch ( Exception e1 ) {
387                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
388                     breakpRequest    = null;
389                 }
390             }
391         } catch ( Exception e2 ) {
392             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
393             breakpRequest    = null;
394         }
395 
396         if (breakpRequest == null) {
397             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
398             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
399         }
400 
401         log2("      a breakpoint has been set up");
402         return breakpRequest;
403     }

406     private void getEventSet()
407                  throws JDITestRuntimeException {
408         try {
409 //            log2("       eventSet = eventQueue.remove(waitTime);");
410             eventSet = eventQueue.remove(waitTime);
411             if (eventSet == null) {
412                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
413             }
414 //            log2("       eventIterator = eventSet.eventIterator;");
415             eventIterator = eventSet.eventIterator();
416         } catch ( Exception e ) {
417             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
418         }
419     }

422     private void breakpointForCommunication()
423                  throws JDITestRuntimeException {
424 
425         log2("breakpointForCommunication");
426         getEventSet();
427 
428         if (eventIterator.nextEvent() instanceof BreakpointEvent)
429             return;
430 
431         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
432     }
