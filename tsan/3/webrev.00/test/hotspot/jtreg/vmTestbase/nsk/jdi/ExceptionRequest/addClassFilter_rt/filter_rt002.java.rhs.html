 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addClassFilter(ReferenceType)</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates a TestClass10 object, to use as filter,       <BR>
 68  *   and invokes the methodForCommunication to be suspended and         <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an ExceptionRequest and enables it,                      <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

247    /*
248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase
251     */

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         EventRequest  eventRequest1      = null;
312 
313         String        property1          = "ExceptionRequest1";
314 
315         ReferenceType testClassReference = null;
316 
317 
318         for (int i = 0; ; i++) {
319 
320             vm.resume();
321             breakpointForCommunication();
322 
323             int instruction = ((IntegerValue)
324                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
325 
326             if (instruction == 0) {
327                 vm.resume();
328                 break;
329             }
330 
331 
332             log1(":::::: case: # " + i);
333 
334             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
335 
336             switch (i) {
337 
338               case 0:
339                      testClassReference =
340                           (ReferenceType) vm.classesByName(testedClassName).get(0);
341 
342                      eventRequest1 = setting21ExceptionRequest(null, testClassReference,
343                                               EventRequest.SUSPEND_NONE, property1);
344                      eventRequest1.enable();
345 
346                      try {
347                          log2("......eventRequest1.addClassFilter(testClassReference);");
348                          log2("        InvalidRequestStateException expected");
349                          ((ExceptionRequest)eventRequest1).addClassFilter(testClassReference);
350                          log3("ERROR: no InvalidRequestStateException");
351                          testExitCode = FAILED;
352                      } catch ( InvalidRequestStateException e ) {
353                          log2("        InvalidRequestStateException");
354                      }
355 
356                      log2("......eventRequest1.disable();");
357                      eventRequest1.disable();
358                      try {
359                          log2("......eventRequest1.addClassFilter(testClassReference);");
360                          log2("        no InvalidRequestStateException expected");
361                          ((ExceptionRequest)eventRequest1).addClassFilter(testClassReference);
362                          log2("        no InvalidRequestStateException");
363                      } catch ( InvalidRequestStateException e ) {
364                          log3("ERROR: InvalidRequestStateException");
365                          testExitCode = FAILED;
366                      }
367 
368                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
369                      eventRManager.deleteEventRequest(eventRequest1);
370                      try {
371                          log2("......eventRequest1.addClassFilter(testClassReference);");
372                          log2("        InvalidRequestStateException expected");
373                          ((ExceptionRequest)eventRequest1).addClassFilter(testClassReference);
374                          log3("ERROR: no InvalidRequestStateException");
375                          testExitCode = FAILED;
376                      } catch ( InvalidRequestStateException e ) {
377                          log2("        InvalidRequestStateException");
378                      }
379                       break;
380 
381               default:
382                       throw new JDITestRuntimeException("** default case 2 **");
383             }
384 
385             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
386         }
387         log1("    TESTING ENDS");
388         return;
389     }

391    /*
392     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
393     *                                             String, String, String)
394     *
395     * It sets up a breakpoint at given line number within a given method in a given class
396     * for a given thread.
397     *
398     * Return value: BreakpointRequest object  in case of success
399     *
400     * JDITestRuntimeException   in case of an Exception thrown within the method
401     */
402 
403     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
404                                                   ReferenceType testedClass,
405                                                   String methodName,
406                                                   String bpLine,
407                                                   String property)
408             throws JDITestRuntimeException {
409 
410         log2("......setting up a breakpoint:");
411         log2("       thread: " + thread + "; class: " + testedClass +
412                         "; method: " + methodName + "; line: " + bpLine);
413 
414         List              alllineLocations = null;
415         Location          lineLocation     = null;
416         BreakpointRequest breakpRequest    = null;
417 
418         try {
419             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
420 
421             alllineLocations = method.allLineLocations();
422 
423             int n =
424                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
425             if (n > alllineLocations.size()) {
426                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
427             } else {
428                 lineLocation = (Location) alllineLocations.get(n);
429                 try {
430                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
431                     breakpRequest.putProperty("number", property);
432                     breakpRequest.addThreadFilter(thread);
433                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
434                 } catch ( Exception e1 ) {
435                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
436                     breakpRequest    = null;
437                 }
438             }
439         } catch ( Exception e2 ) {
440             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
441             breakpRequest    = null;
442         }
443 
444         if (breakpRequest == null) {
445             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
446             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
447         }
448 
449         log2("      a breakpoint has been set up");
450         return breakpRequest;
451     }

454     private void getEventSet()
455                  throws JDITestRuntimeException {
456         try {
457 //            log2("       eventSet = eventQueue.remove(waitTime);");
458             eventSet = eventQueue.remove(waitTime);
459             if (eventSet == null) {
460                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
461             }
462 //            log2("       eventIterator = eventSet.eventIterator;");
463             eventIterator = eventSet.eventIterator();
464         } catch ( Exception e ) {
465             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
466         }
467     }

470     private void breakpointForCommunication()
471                  throws JDITestRuntimeException {
472 
473         log2("breakpointForCommunication");
474         getEventSet();
475 
476         if (eventIterator.nextEvent() instanceof BreakpointEvent)
477             return;
478 
479         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
480     }

484     private ExceptionRequest setting21ExceptionRequest ( ThreadReference thread,
485                                                          ReferenceType   testedClass,
486                                                          int             suspendPolicy,
487                                                          String          property       )
488             throws JDITestRuntimeException {
489         try {
490             log2("......setting up ExceptionRequest:");
491             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
492 
493             ExceptionRequest
494             excr = eventRManager.createExceptionRequest(null, true, true);
495             excr.putProperty("number", property);
496             if (thread != null)
497                 excr.addThreadFilter(thread);
498             excr.addClassFilter(testedClass);
499             excr.setSuspendPolicy(suspendPolicy);
500 
501             log2("      ExceptionRequest has been set up");
502             return excr;
503         } catch ( Exception e ) {
504             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
505             log3("       ExceptionRequest HAS NOT BEEN SET UP");
506             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
507         }
