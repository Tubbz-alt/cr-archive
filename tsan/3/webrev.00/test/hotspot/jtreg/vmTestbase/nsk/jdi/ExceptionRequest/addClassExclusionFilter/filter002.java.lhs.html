 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addClassExclusionFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 68  *   and to inform the debugger with the event.                         <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates an ExceptionRequest and enables it,                      <BR>
 71  *   - invokes the method addClassExclusionFilter() on the request      <BR>
 72  *     expecting to catch InvalidRequestStateException,                         <BR>
 73  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 74  *     expecting to catch no InvalidRequestStateException,                      <BR>
 75  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 76  *     once again expecting to catch InvalidRequestStateException.              <BR>
 77  * <BR>
 78  * In third phase, at the end of the test, the debuggee changes         <BR>
 79  * the value of the "instruction" which the debugger and debuggee       <BR>
 80  * use to inform each other of needed actions,  and both end.           <BR>
 81  * <BR>
 82  */

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : " + e);
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : " + e);
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if (eventIterator.nextEvent() instanceof VMDeathEvent)
266                 return 0;
267 
268             log3("ERROR: last event is not the VMDeathEvent");
269             return 1;
270         } catch ( VMDisconnectedException e ) {
271             log3("ERROR: VMDisconnectedException : " + e);
272             return 2;
273         } catch ( Exception e ) {
274             log3("ERROR: Exception : " + e);
275             return 1;
276         }
277

280     private void testRun()
281                  throws JDITestRuntimeException, Exception {
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304 
305         ThreadReference   mainThread = threadByName("main");
306 
307         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
308                                              debuggeeClass,
309                                             bPointMethod, lineForComm, "zero");
310         bpRequest.enable();
311 
312     //------------------------------------------------------  testing section
313 
314         log1("     TESTING BEGINS");
315 
316         EventRequest  eventRequest1      = null;
317 
318         String        property1          = "ExceptionRequest1";
319 
320 
321 
322 
323 
324         for (int i = 0; ; i++) {
325 
326             vm.resume();
327             breakpointForCommunication();
328 
329             int instruction = ((IntegerValue)
330                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
331 
332             if (instruction == 0) {
333                 vm.resume();
334                 break;
335             }
336 
337 
338             log1(":::::: case: # " + i);
339 
340             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
341 
342             switch (i) {
343 
344               case 0:
345                      eventRequest1 = setting23ExceptionRequest(null, testedClassName1,
346                                               EventRequest.SUSPEND_NONE, property1);
347 
348                      eventRequest1.enable();
349 
350                      try {
351                          log2("......eventRequest1.addClassFilter(testedClassName1);");
352                          log2("        InvalidRequestStateException expected");
353                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
354                          log3("ERROR: no InvalidRequestStateException");
355                          testExitCode = FAILED;
356                      } catch ( InvalidRequestStateException e ) {
357                          log2("        InvalidRequestStateException");
358                      }
359 
360                      log2("......eventRequest1.disable();");
361                      eventRequest1.disable();
362                      try {
363                          log2("......eventRequest1.addClassFilter(testedClassName1);");
364                          log2("        no InvalidRequestStateException expected");
365                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
366                          log2("        no InvalidRequestStateException");
367                      } catch ( InvalidRequestStateException e ) {
368                          log3("ERROR: InvalidRequestStateException");
369                          testExitCode = FAILED;
370                      }
371 
372                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
373                      eventRManager.deleteEventRequest(eventRequest1);
374                      try {
375                          log2("......eventRequest1.addClassFilter(testedClassName1);");
376                          log2("        InvalidRequestStateException expected");
377                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
378                          log3("ERROR: no InvalidRequestStateException");
379                          testExitCode = FAILED;
380                      } catch ( InvalidRequestStateException e ) {
381                          log2("        InvalidRequestStateException");
382                      }
383                       break;
384 
385               default:
386                       throw new JDITestRuntimeException("** default case 2 **");
387             }
388 
389             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
390         }
391         log1("    TESTING ENDS");
392         return;
393     }

395     private ThreadReference threadByName(String name)
396                  throws JDITestRuntimeException {
397 
398         List         all = vm.allThreads();
399         ListIterator li  = all.listIterator();
400 
401         for (; li.hasNext(); ) {
402             ThreadReference thread = (ThreadReference) li.next();
403             if (thread.name().equals(name))
404                 return thread;
405         }
406         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
407     }

409    /*
410     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
411     *                                             String, String, String)
412     *
413     * It sets up a breakpoint at given line number within a given method in a given class
414     * for a given thread.
415     *
416     * Return value: BreakpointRequest object  in case of success
417     *
418     * JDITestRuntimeException   in case of an Exception thrown within the method
419     */
420 
421     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
422                                                   ReferenceType testedClass,
423                                                   String methodName,
424                                                   String bpLine,
425                                                   String property)
426             throws JDITestRuntimeException {
427 
428         log2("......setting up a breakpoint:");
429         log2("       thread: " + thread + "; class: " + testedClass +
430                         "; method: " + methodName + "; line: " + bpLine);
431 
432         List              alllineLocations = null;
433         Location          lineLocation     = null;
434         BreakpointRequest breakpRequest    = null;
435 
436         try {
437             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
438 
439             alllineLocations = method.allLineLocations();
440 
441             int n =
442                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
443             if (n > alllineLocations.size()) {
444                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
445             } else {
446                 lineLocation = (Location) alllineLocations.get(n);
447                 try {
448                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
449                     breakpRequest.putProperty("number", property);
450                     breakpRequest.addThreadFilter(thread);
451                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
452                 } catch ( Exception e1 ) {
453                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
454                     breakpRequest    = null;
455                 }
456             }
457         } catch ( Exception e2 ) {
458             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
459             breakpRequest    = null;
460         }
461 
462         if (breakpRequest == null) {
463             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
464             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
465         }
466 
467         log2("      a breakpoint has been set up");
468         return breakpRequest;
469     }

472     private void getEventSet()
473                  throws JDITestRuntimeException {
474         try {
475 //            log2("       eventSet = eventQueue.remove(waitTime);");
476             eventSet = eventQueue.remove(waitTime);
477             if (eventSet == null) {
478                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
479             }
480 //            log2("       eventIterator = eventSet.eventIterator;");
481             eventIterator = eventSet.eventIterator();
482         } catch ( Exception e ) {
483             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
484         }
485     }

488     private void breakpointForCommunication()
489                  throws JDITestRuntimeException {
490 
491         log2("breakpointForCommunication");
492         getEventSet();
493 
494         if (eventIterator.nextEvent() instanceof BreakpointEvent)
495             return;
496 
497         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
498     }

502     private ExceptionRequest setting23ExceptionRequest ( ThreadReference thread,
503                                                          String          testedClass,
504                                                          int             suspendPolicy,
505                                                          String          property       )
506             throws JDITestRuntimeException {
507         try {
508             log2("......setting up ExceptionRequest:");
509             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
510 
511             ExceptionRequest
512             excr = eventRManager.createExceptionRequest(null, true, true);
513             excr.putProperty("number", property);
514             if (thread != null)
515                 excr.addThreadFilter(thread);
516             excr.addClassExclusionFilter(testedClass);
517             excr.setSuspendPolicy(suspendPolicy);
518 
519             log2("      ExceptionRequest has been set up");
520             return excr;
521         } catch ( Exception e ) {
522             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
523             log3("       ExceptionRequest HAS NOT BEEN SET UP");
524             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
525         }
