 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VoidType.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that for a mirror of a "void" value the cast <BR>
 42  * from Type to VoidType doesn't throw ClassCastException.      <BR>
 43  * <BR>
 44  * The test has three phases and works as follows.              <BR>
 45  * <BR>
 46  * In first phase,                                                      <BR>
 47  * upon launching debuggee's VM which will be suspended,                <BR>
 48  * a debugger waits for the VMStartEvent within a predefined            <BR>
 49  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 50  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 51  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 52  * and waits for the event within the predefined time interval.         <BR>
 53  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 54  * Upon getting the ClassPrepareEvent,                                  <BR>
 55  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 56  * within debuggee's special methodForCommunication().                  <BR>
 57  * <BR>
 58  * In second phase to check the cast,                                   <BR>
 59  * the debugger and the debuggee perform the following.                 <BR>
 60  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 61  * - The debuggee prepares new check case and invokes                   <BR>
 62  *   the methodForCommunication to be suspended and                     <BR>
 63  *   to inform the debugger with the event.                             <BR>
 64  * - Upon getting the BreakpointEvent,                                  <BR>
 65  *   the debugger performs the check.                                   <BR>
 66  * Note. To inform each other of needed actions, the debugger and       <BR>
 67  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 68  * <BR>
 69  * In third phase when at the end,                                      <BR>
 70  * the debuggee changes the value of the "instruction"                  <BR>
 71  * to inform the debugger of checks finished, and both end.             <BR>
 72  * <BR>
 73  */
 74

 79     static final int FAILED = 2;

 97     public static int run (String argv[], PrintStream out) {
 98 
 99         int exitCode = new voidtype001().runThis(argv, out);
100 
101         if (exitCode != PASSED) {
102             System.out.println("TEST FAILED");
103         }
104         return testExitCode;
105     }

147     private int runThis (String argv[], PrintStream out) {
148 
149         argsHandler     = new ArgumentHandler(argv);
150         logHandler      = new Log(out, argsHandler);
151         Binder binder   = new Binder(argsHandler, logHandler);
152 
153         waitTime        = argsHandler.getWaitTime() * 60000;
154 
155         try {
156             log2("launching a debuggee :");
157             log2("       " + debuggeeName);
158             if (argsHandler.verbose()) {
159                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
160             } else {
161                 debuggee = binder.bindToDebugee(debuggeeName);
162             }
163             if (debuggee == null) {
164                 log3("ERROR: no debuggee launched");
165                 return FAILED;
166             }
167             log2("debuggee launched");
168         } catch ( Exception e ) {
169             log3("ERROR: Exception : " + e);
170             log2("       test cancelled");
171             return FAILED;
172         }
173 
174         debuggee.redirectOutput(logHandler);
175 
176         vm = debuggee.VM();
177 
178         eventQueue = vm.eventQueue();
179         if (eventQueue == null) {
180             log3("ERROR: eventQueue == null : TEST ABORTED");
181             vm.exit(PASS_BASE);
182             return FAILED;
183         }
184 
185         log2("invocation of the method runTest()");
186         switch (runTest()) {
187 
188             case 0 :  log2("test phase has finished normally");
189                       log2("   waiting for the debuggee to finish ...");
190                       debuggee.waitFor();
191 
192                       log2("......getting the debuggee's exit status");
193                       int status = debuggee.getStatus();
194                       if (status != PASS_BASE) {
195                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
196                               status + " != PASS_BASE");
197                           testExitCode = FAILED;
198                       } else {
199                           log2("......debuggee returned expected exit status: " +
200                               status + " == PASS_BASE");
201                       }
202                       break;
203 
204             default : log3("ERROR: runTest() returned unexpected value");
205 
206             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
207                       log2("......forcing: vm.exit();");
208                       testExitCode = FAILED;
209                       try {
210                           vm.exit(PASS_BASE);
211                       } catch ( Exception e ) {
212                           log3("ERROR: Exception : e");
213                       }
214                       break;
215 
216             case 2 :  log3("test cancelled due to VMDisconnectedException");
217                       log2("......trying: vm.process().destroy();");
218                       testExitCode = FAILED;
219                       try {
220                           Process vmProcess = vm.process();
221                           if (vmProcess != null) {
222                               vmProcess.destroy();
223                           }
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : e");
226                       }
227                       break;
228             }
229 
230         return testExitCode;
231     }

234    /*
235     * Return value: 0 - normal end of the test
236     *               1 - ubnormal end of the test
237     *               2 - VMDisconnectedException while test phase
238     */
239 
240     private int runTest() {
241 
242         try {
243             testRun();
244 
245             log2("waiting for VMDeathEvent");
246             getEventSet();
247             if (eventIterator.nextEvent() instanceof VMDeathEvent)
248                 return 0;
249 
250             log3("ERROR: last event is not the VMDeathEvent");
251             return 1;
252         } catch ( VMDisconnectedException e ) {
253             log3("ERROR: VMDisconnectedException : " + e);
254             return 2;
255         } catch ( Exception e ) {
256             log3("ERROR: Exception : " + e);
257             return 1;
258         }
259 
260     }

262     private void testRun()
263                  throws JDITestRuntimeException, Exception {
264 
265         eventRManager = vm.eventRequestManager();
266 
267         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
268         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
269         cpRequest.addClassFilter(debuggeeName);
270 
271         cpRequest.enable();
272         vm.resume();
273         getEventSet();
274         cpRequest.disable();
275 
276         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
277         debuggeeClass = event.referenceType();
278 
279         if (!debuggeeClass.name().equals(debuggeeName))
280            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
281 
282         log2("      received: ClassPrepareEvent for debuggeeClass");
283 
284         String bPointMethod = "methodForCommunication";
285         String lineForComm  = "lineForComm";
286         BreakpointRequest bpRequest;
287 
288         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
289                                       debuggeeClass,
290                                       bPointMethod, lineForComm, "zero");
291         bpRequest.enable();
292 
293     //------------------------------------------------------  testing section
294 
295         log1("     TESTING BEGINS");
296 
297         for (int i = 0; ; i++) {
298 
299             vm.resume();
300             breakpointForCommunication();
301 
302             int instruction = ((IntegerValue)
303                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
304 
305             if (instruction == 0) {
306                 vm.resume();
307                 break;
308             }
309 
310             log1(":::::: case: # " + i);
311 
312             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
313 
314             log2("......getting: List methods = debuggeeClass.methodsByName('testMethod');");
315             List methods = debuggeeClass.methodsByName("testMethod");
316             if (methods.size() != 1) {
317                 testExitCode = FAILED;
318                 log3("ERROR: methods.size() != 1");
319                 break ;
320             }
321 
322             log2("......getting: Method testMethod = (Method) methods.get(0);");
323             Method testMethod = (Method) methods.get(0);
324 
325             try {
326                  log2("......getting: Type testMethodReturnType = testMethod.returnType();");
327                  Type testMethodReturnType = testMethod.returnType();
328 
329                  log2("......VoidType voidType = (VoidType) testMethodReturnType;");
330                  VoidType voidType = (VoidType) testMethodReturnType;
331             } catch ( ClassNotLoadedException e ) {
332                  log3("ERROR: testmethodReturnType = testmethod.returnType();");
333                  testExitCode = FAILED;
334                  break ;
335             } catch ( ClassCastException e ) {
336                  log3("ERROR: ClassCastException");
337                  testExitCode = FAILED;
338                  break ;
339             }
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
342         }
343         log1("    TESTING ENDS");
344         return;
345     }

347    /*
348     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
349     *                                             String, String, String)
350     *
351     * It sets up a breakpoint at given line number within a given method in a given class
352     * for a given thread.
353     *
354     * Return value: BreakpointRequest object  in case of success
355     *
356     * JDITestRuntimeException   in case of an Exception thrown within the method
357     */
358 
359     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
360                                                   ReferenceType testedClass,
361                                                   String methodName,
362                                                   String bpLine,
363                                                   String property)
364             throws JDITestRuntimeException {
365 
366         log2("......setting up a breakpoint:");
367         log2("       thread: " + thread + "; class: " + testedClass +
368                         "; method: " + methodName + "; line: " + bpLine);
369 
370         List              alllineLocations = null;
371         Location          lineLocation     = null;
372         BreakpointRequest breakpRequest    = null;
373 
374         try {
375             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
376 
377             alllineLocations = method.allLineLocations();
378 
379             int n =
380                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
381             if (n > alllineLocations.size()) {
382                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
383             } else {
384                 lineLocation = (Location) alllineLocations.get(n);
385                 try {
386                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
387                     breakpRequest.putProperty("number", property);
388                     breakpRequest.addThreadFilter(thread);
389                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
390                 } catch ( Exception e1 ) {
391                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
392                     breakpRequest    = null;
393                 }
394             }
395         } catch ( Exception e2 ) {
396             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
397             breakpRequest    = null;
398         }
399 
400         if (breakpRequest == null) {
401             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
402             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
403         }
404 
405         log2("      a breakpoint has been set up");
406         return breakpRequest;
407     }

410     private void getEventSet()
411                  throws JDITestRuntimeException {
412         try {
413 //            log2("       eventSet = eventQueue.remove(waitTime);");
414             eventSet = eventQueue.remove(waitTime);
415             if (eventSet == null) {
416                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
417             }
418 //            log2("       eventIterator = eventSet.eventIterator;");
419             eventIterator = eventSet.eventIterator();
420         } catch ( Exception e ) {
421             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
422         }
423     }

426     private void breakpointForCommunication()
427                  throws JDITestRuntimeException {
428 
429         log2("breakpointForCommunication");
430         getEventSet();
431 
432         if (eventIterator.nextEvent() instanceof BreakpointEvent)
433             return;
434 
435         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
436     }
