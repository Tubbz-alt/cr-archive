 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for MethodExitEvent sets.             <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a MethodExitRequest, resumes                  <BR>
 74  *   the debuggee, and waits for the MethodExitEvent.                   <BR>
 75  * - The debuggee invokes the special method which makes access to      <BR>
 76  *   a predefined variable to be resulting in the event.                <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */

169     private int runThis (String argv[], PrintStream out) {
170 
171         argsHandler     = new ArgumentHandler(argv);
172         logHandler      = new Log(out, argsHandler);
173         Binder binder   = new Binder(argsHandler, logHandler);
174 
175         waitTime        = argsHandler.getWaitTime() * 60000;
176 
177         try {
178             log2("launching a debuggee :");
179             log2("       " + debuggeeName);
180             if (argsHandler.verbose()) {
181                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
182             } else {
183                 debuggee = binder.bindToDebugee(debuggeeName);
184             }
185             if (debuggee == null) {
186                 log3("ERROR: no debuggee launched");
187                 return FAILED;
188             }
189             log2("debuggee launched");
190         } catch ( Exception e ) {
191             log3("ERROR: Exception : " + e);
192             log2("       test cancelled");
193             return FAILED;
194         }
195 
196         debuggee.redirectOutput(logHandler);
197 
198         vm = debuggee.VM();
199 
200         eventQueue = vm.eventQueue();
201         if (eventQueue == null) {
202             log3("ERROR: eventQueue == null : TEST ABORTED");
203             vm.exit(PASS_BASE);
204             return FAILED;
205         }
206 
207         log2("invocation of the method runTest()");
208         switch (runTest()) {
209 
210             case 0 :  log2("test phase has finished normally");
211                       log2("   waiting for the debuggee to finish ...");
212                       debuggee.waitFor();
213 
214                       log2("......getting the debuggee's exit status");
215                       int status = debuggee.getStatus();
216                       if (status != PASS_BASE) {
217                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
218                               status + " != PASS_BASE");
219                           testExitCode = FAILED;
220                       } else {
221                           log2("......debuggee returned expected exit status: " +
222                               status + " == PASS_BASE");
223                       }
224                       break;
225 
226             default : log3("ERROR: runTest() returned unexpected value");
227 
228             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
229                       log2("......forcing: vm.exit();");
230                       testExitCode = FAILED;
231                       try {
232                           vm.exit(PASS_BASE);
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237 
238             case 2 :  log3("test cancelled due to VMDisconnectedException");
239                       log2("......trying: vm.process().destroy();");
240                       testExitCode = FAILED;
241                       try {
242                           Process vmProcess = vm.process();
243                           if (vmProcess != null) {
244                               vmProcess.destroy();
245                           }
246                       } catch ( Exception e ) {
247                           log3("ERROR: Exception : e");
248                       }
249                       break;
250             }
251 
252         return testExitCode;
253     }

262     private int runTest() {
263 
264         try {
265             testRun();
266 
267             log2("waiting for VMDeathEvent");
268             getEventSet();
269             if (eventIterator.nextEvent() instanceof VMDeathEvent)
270                 return 0;
271 
272             log3("ERROR: last event is not the VMDeathEvent");
273             return 1;
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         eventRManager = vm.eventRequestManager();
288 
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292 
293         cpRequest.enable();
294         vm.resume();
295         getEventSet();
296         cpRequest.disable();
297 
298         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
299         debuggeeClass = event.referenceType();
300 
301         if (!debuggeeClass.name().equals(debuggeeName))
302            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
303 
304         log2("      received: ClassPrepareEvent for debuggeeClass");
305 
306         String bPointMethod = "methodForCommunication";
307         String lineForComm  = "lineForComm";
308         BreakpointRequest bpRequest;
309         ThreadReference   mainThread = threadByName("main");
310         bpRequest = settingBreakpoint(mainThread,
311                                       debuggeeClass,
312                                       bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315         vm.resume();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         EventRequest eventRequest1 = null;
322         EventRequest eventRequest2 = null;
323         EventRequest eventRequest3 = null;
324 
325         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
327         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
328 
329         ReferenceType testClassReference = null;
330 
331 
332         for (int i = 0; ; i++) {
333 
334             breakpointForCommunication();
335 
336             int instruction = ((IntegerValue)
337                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
338 
339             if (instruction == 0) {
340                 vm.resume();
341                 break;
342             }
343 
344             log1(":::::: case: # " + i);
345 
346             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
347 
348             switch (i) {
349 
350               case 0:
351                       testClassReference =
352                            (ReferenceType) vm.classesByName(testedClassName).get(0);
353 
354                       eventRequest1 = settingMethodExitRequest (mainThread,
355                                              testClassReference,
356                                              SUSPEND_NONE, "MethodExitRequest1");
357                       eventRequest1.enable();
358                       break;
359 
360               case 1:
361                       eventRequest2 = settingMethodExitRequest (mainThread,
362                                              testClassReference,
363                                              SUSPEND_THREAD, "MethodExitRequest2");
364                       eventRequest2.enable();
365                       break;
366 
367               case 2:
368                       eventRequest3 = settingMethodExitRequest (mainThread,
369                                              testClassReference,
370                                              SUSPEND_ALL, "MethodExitRequest3");
371                       eventRequest3.enable();
372                       break;
373 
374 
375               default:
376                       throw new JDITestRuntimeException("** default case 2 **");
377             }
378 
379             log2("......waiting for new MethodExitEvent : " + i);
380             mainThread.resume();
381             getEventSet();
382 
383             Event newEvent = eventIterator.nextEvent();
384             if ( !(newEvent instanceof MethodExitEvent)) {
385                  log3("ERROR: new event is not MethodExitEvent");
386                  testExitCode = FAILED;
387             } else {
388 
389                 String property = (String) newEvent.request().getProperty("number");
390                 log2("       got new MethodExitEvent with propety 'number' == " + property);
391 
392                 log2("......checking up on EventSet.resume()");
393                 log2("......--> vm.suspend();");
394                 vm.suspend();
395 
396                 log2("        getting : Map<String, Integer> suspendsCounts1");
397 
398                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
399                 for (ThreadReference threadReference : vm.allThreads()) {
400                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
401                 }
402 
403                 log2("        eventSet.resume;");
404                 eventSet.resume();
405 
406                 log2("        getting : Map<String, Integer> suspendsCounts2");
407                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
408                 for (ThreadReference threadReference : vm.allThreads()) {
409                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
410                 }
411 
412                 log2(suspendsCounts1.toString());
413                 log2(suspendsCounts2.toString());
414 
415                 log2("        getting : int policy = eventSet.suspendPolicy();");
416                 int policy = eventSet.suspendPolicy();
417 
418                 switch (policy) {
419 
420                   case SUSPEND_NONE   :
421                        log2("        case SUSPEND_NONE");
422                        for (String threadName : suspendsCounts1.keySet()) {
423                            log2("        checking " + threadName);
424                            if (!suspendsCounts2.containsKey(threadName)) {
425                                log3("ERROR: couldn't get ThreadReference for " + threadName);
426                                testExitCode = FAILED;
427                                break;
428                            }
429                            int count1 = suspendsCounts1.get(threadName);
430                            int count2 = suspendsCounts2.get(threadName);
431                            if (count1 != count2) {
432                                log3("ERROR: suspendCounts don't match for : " + threadName);
433                                log3("before resuming : " + count1);
434                                log3("after  resuming : " + count2);
435                                testExitCode = FAILED;
436                                break;
437                            }
438                        }
439                        eventRequest1.disable();
440                        break;
441 
442                   case SUSPEND_THREAD :
443                        log2("        case SUSPEND_THREAD");
444                        for (String threadName : suspendsCounts1.keySet()) {
445                            log2("checking " + threadName);
446                            if (!suspendsCounts2.containsKey(threadName)) {
447                                log3("ERROR: couldn't get ThreadReference for " + threadName);
448                                testExitCode = FAILED;
449                                break;
450                            }
451                            int count1 = suspendsCounts1.get(threadName);
452                            int count2 = suspendsCounts2.get(threadName);
453                            String eventThreadName = event.thread().name();
454                            int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
455                            if (count1 != expectedValue) {
456                                log3("ERROR: suspendCounts don't match for : " + threadName);
457                                log3("before resuming : " + count1);
458                                log3("after  resuming : " + count2);
459                                testExitCode = FAILED;
460                                break;
461                            }
462                        }
463                        eventRequest2.disable();
464                        break;
465 
466                   case SUSPEND_ALL    :
467 
468                         log2("        case SUSPEND_ALL");
469                         for (String threadName : suspendsCounts1.keySet()) {
470                             log2("checking " + threadName);
471                             if (!event.request().equals(eventRequest3))
472                                 break;
473                             if (!suspendsCounts2.containsKey(threadName)) {
474                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
475                                 testExitCode = FAILED;
476                                 break;
477                             }
478                             int count1 = suspendsCounts1.get(threadName);
479                             int count2 = suspendsCounts2.get(threadName);
480                             if (count1 != count2 + 1) {
481                                 log3("ERROR: suspendCounts don't match for : " + threadName);
482                                 log3("before resuming : " + count1);
483                                 log3("after  resuming : " + count2);
484                                 testExitCode = FAILED;
485                                 break;
486                             }
487                         }
488                         eventRequest3.disable();
489                         break;
490 
491                   default: throw new JDITestRuntimeException("** default case 1 **");
492                 }
493                 informDebuggeeTestCase(i);
494             }
495 
496             log2("......--> vm.resume()");
497             vm.resume();
498             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
499         }
500         log1("    TESTING ENDS");
501         return;
502     }

504     private ThreadReference threadByName(String name)
505                  throws JDITestRuntimeException {
506 
507         List         all = vm.allThreads();
508         ListIterator li  = all.listIterator();
509 
510         for (; li.hasNext(); ) {
511             ThreadReference thread = (ThreadReference) li.next();
512             if (thread.name().equals(name))
513                 return thread;
514         }
515         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
516     }

518    /*
519     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
520     *                                             String, String, String)
521     *
522     * It sets up a breakpoint at given line number within a given method in a given class
523     * for a given thread.
524     *
525     * Return value: BreakpointRequest object  in case of success
526     *
527     * JDITestRuntimeException   in case of an Exception thrown within the method
528     */
529 
530     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
531                                                   ReferenceType testedClass,
532                                                   String methodName,
533                                                   String bpLine,
534                                                   String property)
535             throws JDITestRuntimeException {
536 
537         log2("......setting up a breakpoint:");
538         log2("       thread: " + thread + "; class: " + testedClass +
539                         "; method: " + methodName + "; line: " + bpLine);
540 
541         List              alllineLocations = null;
542         Location          lineLocation     = null;
543         BreakpointRequest breakpRequest    = null;
544 
545         try {
546             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
547 
548             alllineLocations = method.allLineLocations();
549 
550             int n =
551                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
552             if (n > alllineLocations.size()) {
553                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
554             } else {
555                 lineLocation = (Location) alllineLocations.get(n);
556                 try {
557                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
558                     breakpRequest.putProperty("number", property);
559                     breakpRequest.addThreadFilter(thread);
560                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
561                 } catch ( Exception e1 ) {
562                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
563                     breakpRequest    = null;
564                 }
565             }
566         } catch ( Exception e2 ) {
567             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
568             breakpRequest    = null;
569         }
570 
571         if (breakpRequest == null) {
572             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
573             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
574         }
575 
576         log2("      a breakpoint has been set up");
577         return breakpRequest;
578     }

581     private void getEventSet()
582                  throws JDITestRuntimeException {
583         try {
584 //            log2("       eventSet = eventQueue.remove(waitTime);");
585             eventSet = eventQueue.remove(waitTime);
586             if (eventSet == null) {
587                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
588             }
589 //            log2("       eventIterator = eventSet.eventIterator;");
590             eventIterator = eventSet.eventIterator();
591         } catch ( Exception e ) {
592             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
593         }
594     }

597     private void breakpointForCommunication()
598                  throws JDITestRuntimeException {
599 
600         log2("breakpointForCommunication");
601         getEventSet();
602 
603         if (eventIterator.nextEvent() instanceof BreakpointEvent)
604             return;
605 
606         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
607     }

611     private MethodExitRequest settingMethodExitRequest ( ThreadReference thread,
612                                                          ReferenceType   testedClass,
613                                                          int             suspendPolicy,
614                                                          String          property        )
615             throws JDITestRuntimeException {
616         try {
617             log2("......setting up MethodExitRequest:");
618             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
619 
620             MethodExitRequest
621             mexr = eventRManager.createMethodExitRequest();
622             mexr.putProperty("number", property);
623             mexr.addThreadFilter(thread);
624             mexr.addClassFilter(testedClass);
625             mexr.setSuspendPolicy(suspendPolicy);
626 
627         log2("      MethodExitRequest has been set up");
628         return mexr;
629         } catch ( Exception e ) {
630             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
631             log3("       MethodExitRequest HAS NOT BEEN SET UP");
632             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
633         }

635     /**
636      * Inform debuggee which thread test the debugger has completed.
637      * Used for synchronization, so the debuggee does not move too quickly.
638      * @param testCase index of just completed test
639      */
640     void informDebuggeeTestCase(int testCase) {
641         try {
642             ((ClassType)debuggeeClass)
643                 .setValue(debuggeeClass.fieldByName("testCase"),
644                           vm.mirrorOf(testCase));
645         } catch (InvalidTypeException ite) {
646             throw new Failure("** FAILURE setting testCase  **");
647         } catch (ClassNotLoadedException cnle) {
648             throw new Failure("** FAILURE notifying debuggee  **");
649         } catch (VMDisconnectedException e) {
650             throw new Failure("** FAILURE debuggee connection **");
651         }
652     }
