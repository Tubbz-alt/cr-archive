 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadDeathRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadDeathRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Restricts the events generated by this request            <BR>
 47  *    to those in the given thread.                             <BR>
 48  * The cases to test include three thread's states:             <BR>
 49  * not started, running, dead.                                  <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new thread, thread1, and invokes              <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates ThreadDeathRequest1 and                                  <BR>
 73  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 74  *   - creates ThreadDeathRequest2,                                     <BR>
 75  *     resumes the debuggee to get the thread1 started, and             <BR>
 76  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 77  *   - creates ThreadDeathRequest3, waits the thread1 to die, and       <BR>
 78  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  *
 86 
 87  */
 88

170     private int runThis (String argv[], PrintStream out) {
171 
172         argsHandler     = new ArgumentHandler(argv);
173         logHandler      = new Log(out, argsHandler);
174         Binder binder   = new Binder(argsHandler, logHandler);
175 
176         waitTime        = argsHandler.getWaitTime() * 60000;
177 
178         try {
179             log2("launching a debuggee :");
180             log2("       " + debuggeeName);
181             if (argsHandler.verbose()) {
182                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
183             } else {
184                 debuggee = binder.bindToDebugee(debuggeeName);
185             }
186             if (debuggee == null) {
187                 log3("ERROR: no debuggee launched");
188                 return FAILED;
189             }
190             log2("debuggee launched");
191         } catch ( Exception e ) {
192             log3("ERROR: Exception : " + e);
193             log2("       test cancelled");
194             return FAILED;
195         }
196 
197         debuggee.redirectOutput(logHandler);
198 
199         vm = debuggee.VM();
200 
201         eventQueue = vm.eventQueue();
202         if (eventQueue == null) {
203             log3("ERROR: eventQueue == null : TEST ABORTED");
204             vm.exit(PASS_BASE);
205             return FAILED;
206         }
207 
208         log2("invocation of the method runTest()");
209         switch (runTest()) {
210 
211             case 0 :  log2("test phase has finished normally");
212                       log2("   waiting for the debuggee to finish ...");
213                       debuggee.waitFor();
214 
215                       log2("......getting the debuggee's exit status");
216                       int status = debuggee.getStatus();
217                       if (status != PASS_BASE) {
218                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
219                               status + " != PASS_BASE");
220                           testExitCode = FAILED;
221                       } else {
222                           log2("......debuggee returned expected exit status: " +
223                               status + " == PASS_BASE");
224                       }
225                       break;
226 
227             default : log3("ERROR: runTest() returned unexpected value");
228 
229             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
230                       log2("......forcing: vm.exit();");
231                       testExitCode = FAILED;
232                       try {
233                           vm.exit(PASS_BASE);
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : e");
236                       }
237                       break;
238 
239             case 2 :  log3("test cancelled due to VMDisconnectedException");
240                       log2("......trying: vm.process().destroy();");
241                       testExitCode = FAILED;
242                       try {
243                           Process vmProcess = vm.process();
244                           if (vmProcess != null) {
245                               vmProcess.destroy();
246                           }
247                       } catch ( Exception e ) {
248                           log3("ERROR: Exception : e");
249                       }
250                       break;
251             }
252 
253         return testExitCode;
254     }

259     *               1 - ubnormal end of the test
260     *               2 - VMDisconnectedException while test phase

263     private int runTest() {
264 
265         try {
266             testRun();
267 
268             log2("waiting for VMDeathEvent");
269             getEventSet();
270             if (eventIterator.nextEvent() instanceof VMDeathEvent)
271                 return 0;
272 
273             log3("ERROR: last event is not the VMDeathEvent");
274             return 1;
275         } catch ( VMDisconnectedException e ) {
276             log3("ERROR: VMDisconnectedException : " + e);
277             return 2;
278         } catch ( Exception e ) {
279             log3("ERROR: Exception : " + e);
280             return 1;
281         }
282

285     private void testRun()
286                  throws JDITestRuntimeException, Exception {
287 
288         eventRManager = vm.eventRequestManager();
289 
290         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
291         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
292         cpRequest.addClassFilter(debuggeeName);
293 
294         cpRequest.enable();
295         vm.resume();
296         getEventSet();
297         cpRequest.disable();
298 
299         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
300         debuggeeClass = event.referenceType();
301 
302         if (!debuggeeClass.name().equals(debuggeeName))
303            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
304 
305         log2("      received: ClassPrepareEvent for debuggeeClass");
306 
307         String bPointMethod = "methodForCommunication";
308         String lineForComm  = "lineForComm";
309 
310         ThreadReference   mainThread = threadByName("main");
311 
312         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
313                                              debuggeeClass,
314                                             bPointMethod, lineForComm, "zero");
315         bpRequest.enable();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         EventRequest eventRequest1 = null;
322         EventRequest eventRequest2 = null;
323         EventRequest eventRequest3 = null;
324 
325         ThreadReference thread1 = null;
326 
327         String thread1Name = "thread1";
328 
329         String property1 = "ThreadDeathRequest1";
330         String property2 = "ThreadDeathRequest2";
331         String property3 = "ThreadDeathRequest3";
332 
333         ReferenceType testClassReference = null;
334 
335 
336         for (int i = 0; ; i++) {
337 
338             vm.resume();
339             breakpointForCommunication();
340 
341             int instruction = ((IntegerValue)
342                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
343 
344             if (instruction == 0) {
345                 vm.resume();
346                 break;
347             }
348 
349             log1(":::::: case: # " + i);
350 
351             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
352 
353             switch (i) {
354 
355               case 0:
356                       testClassReference =
357                            (ReferenceType) vm.classesByName(testedClassName).get(0);
358 
359                       thread1 = (ThreadReference) debuggeeClass.getValue(
360                                                   debuggeeClass.fieldByName(thread1Name));
361 
362                       eventRequest1 = setting2ThreadDeathRequest (null,
363                                              EventRequest.SUSPEND_NONE, property1);
364 
365                       try {
366                           log2("......eventRequest1.addThreadFilter(thread1);");
367                           log2("        no Exception expected");
368                           ((ThreadDeathRequest)eventRequest1).addThreadFilter(thread1);
369                       } catch ( Exception e ) {
370                           log3("ERROR: Exception : " + e);
371                           testExitCode = FAILED;
372                       }
373                       break;
374 
375               case 1:
376                       eventRequest2 = setting2ThreadDeathRequest (null,
377                                              EventRequest.SUSPEND_NONE, property2);
378 
379                       try {
380                           log2("......eventRequest2.addThreadFilter(thread1);");
381                           log2("        no Exception expected");
382                           ((ThreadDeathRequest)eventRequest2).addThreadFilter(thread1);
383                       } catch ( Exception e ) {
384                           log3("ERROR: Exception : " + e);
385                           testExitCode = FAILED;
386                       }
387                       break;
388 
389               case 2:
390                       eventRequest3 = setting2ThreadDeathRequest (null,
391                                              EventRequest.SUSPEND_NONE, property3);
392 
393                       try {
394                           log2("......eventRequest3.addThreadFilter(thread1);");
395                           log2("        no Exception expected");
396                           ((ThreadDeathRequest)eventRequest3).addThreadFilter(thread1);
397                       } catch ( Exception e ) {
398                           log3("ERROR: Exception : " + e);
399                           testExitCode = FAILED;
400                       }
401                       break;
402 
403 
404               default:
405                       throw new JDITestRuntimeException("** default case 2 **");
406             }
407 
408             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
409         }
410         log1("    TESTING ENDS");
411         return;
412     }

414     private ThreadReference threadByName(String name)
415                  throws JDITestRuntimeException {
416 
417         List         all = vm.allThreads();
418         ListIterator li  = all.listIterator();
419 
420         for (; li.hasNext(); ) {
421             ThreadReference thread = (ThreadReference) li.next();
422             if (thread.name().equals(name))
423                 return thread;
424         }
425         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
426     }

428    /*
429     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
430     *                                             String, String, String)
431     *
432     * It sets up a breakpoint at given line number within a given method in a given class
433     * for a given thread.
434     *
435     * Return value: BreakpointRequest object  in case of success
436     *
437     * JDITestRuntimeException   in case of an Exception thrown within the method
438     */
439 
440     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
441                                                   ReferenceType testedClass,
442                                                   String methodName,
443                                                   String bpLine,
444                                                   String property)
445             throws JDITestRuntimeException {
446 
447         log2("......setting up a breakpoint:");
448         log2("       thread: " + thread + "; class: " + testedClass +
449                         "; method: " + methodName + "; line: " + bpLine);
450 
451         List              alllineLocations = null;
452         Location          lineLocation     = null;
453         BreakpointRequest breakpRequest    = null;
454 
455         try {
456             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
457 
458             alllineLocations = method.allLineLocations();
459 
460             int n =
461                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
462             if (n > alllineLocations.size()) {
463                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
464             } else {
465                 lineLocation = (Location) alllineLocations.get(n);
466                 try {
467                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
468                     breakpRequest.putProperty("number", property);
469                     breakpRequest.addThreadFilter(thread);
470                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
471                 } catch ( Exception e1 ) {
472                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
473                     breakpRequest    = null;
474                 }
475             }
476         } catch ( Exception e2 ) {
477             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
478             breakpRequest    = null;
479         }
480 
481         if (breakpRequest == null) {
482             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
483             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
484         }
485 
486         log2("      a breakpoint has been set up");
487         return breakpRequest;
488     }

491     private void getEventSet()
492                  throws JDITestRuntimeException {
493         try {
494 //            log2("       eventSet = eventQueue.remove(waitTime);");
495             eventSet = eventQueue.remove(waitTime);
496             if (eventSet == null) {
497                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
498             }
499 //            log2("       eventIterator = eventSet.eventIterator;");
500             eventIterator = eventSet.eventIterator();
501         } catch ( Exception e ) {
502             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
503         }
504     }

507     private void breakpointForCommunication()
508                  throws JDITestRuntimeException {
509 
510         log2("breakpointForCommunication");
511         getEventSet();
512 
513         if (eventIterator.nextEvent() instanceof BreakpointEvent)
514             return;
515 
516         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
517     }

521     private ThreadDeathRequest setting2ThreadDeathRequest( ThreadReference thread,
522                                                            int             suspendPolicy,
523                                                            String          property)
524             throws JDITestRuntimeException {
525         try {
526             ThreadDeathRequest tsr = eventRManager.createThreadDeathRequest();
527             if (thread != null)
528                 tsr.addThreadFilter(thread);
529             tsr.addCountFilter(1);
530             tsr.setSuspendPolicy(suspendPolicy);
531             tsr.putProperty("number", property);
532             return tsr;
533         } catch ( Exception e ) {
534             log3("ERROR: ATTENTION: Exception within settingThreadDeathRequest() : " + e);
535             log3("       ThreadDeathRequest HAS NOT BEEN SET UP");
536             throw new JDITestRuntimeException("** FAILURE to set up ThreadDeathRequest **");
537         }
538     }
