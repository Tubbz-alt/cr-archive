 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.deleteEventRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *     After invoking the tested method,                        <BR>
 47  *     the methods returning Lists of requests,                 <BR>
 48  *     such as EventRequestManager.accessWatchpointRequests(),  <BR>
 49  *     return a List containing no eventRequests                <BR>
 50  *     deleted by the tested method.                            <BR>
 51  * Testcases include all subclasses of EventRequest.            <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee prepares testcases and invokes                        <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent,                                  <BR>
 74  *   the debugger performs the checks required.                         <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : " + e);
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if (eventIterator.nextEvent() instanceof VMDeathEvent)
258                 return 0;
259 
260             log3("ERROR: last event is not the VMDeathEvent");
261             return 1;
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }

499    /*
500     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
501     *                                             String, String, String)
502     *
503     * It sets up a breakpoint at given line number within a given method in a given class
504     * for a given thread.
505     *
506     * Return value: BreakpointRequest object  in case of success
507     *
508     * JDITestRuntimeException   in case of an Exception thrown within the method
509     */
510 
511     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
512                                                   ReferenceType testedClass,
513                                                   String methodName,
514                                                   String bpLine,
515                                                   String property)
516             throws JDITestRuntimeException {
517 
518         log2("......setting up a breakpoint:");
519         log2("       thread: " + thread + "; class: " + testedClass +
520                         "; method: " + methodName + "; line: " + bpLine);
521 
522         List              alllineLocations = null;
523         Location          lineLocation     = null;
524         BreakpointRequest breakpRequest    = null;
525 
526         try {
527             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
528 
529             alllineLocations = method.allLineLocations();
530 
531             int n =
532                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
533             if (n > alllineLocations.size()) {
534                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
535             } else {
536                 lineLocation = (Location) alllineLocations.get(n);
537 breakpLocation = lineLocation;
538                 try {
539                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
540                     breakpRequest.putProperty("number", property);
541                     breakpRequest.addThreadFilter(thread);
542                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
543                 } catch ( Exception e1 ) {
544                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
545                     breakpRequest    = null;
546                 }
547             }
548         } catch ( Exception e2 ) {
549             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
550             breakpRequest    = null;
551         }
552 
553         if (breakpRequest == null) {
554             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
555             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
556         }
557 
558         log2("      a breakpoint has been set up");
559         return breakpRequest;
560     }

563     private void getEventSet()
564                  throws JDITestRuntimeException {
565         try {
566 //            log2("       eventSet = eventQueue.remove(waitTime);");
567             eventSet = eventQueue.remove(waitTime);
568             if (eventSet == null) {
569                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
570             }
571 //            log2("       eventIterator = eventSet.eventIterator;");
572             eventIterator = eventSet.eventIterator();
573         } catch ( Exception e ) {
574             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
575         }
576     }

579     private void breakpointForCommunication()
580                  throws JDITestRuntimeException {
581 
582         log2("breakpointForCommunication");
583         getEventSet();
584 
585         if (eventIterator.nextEvent() instanceof BreakpointEvent)
586             return;
587 
588         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
589     }
