 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadDeathRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadDeathRequest.addThreadFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The cases to check are as follows:
 46  * (1) if this request is currently enabled,                    <BR>
 47  *   - addThreadFilter(ThreadReference) results in throwing     <BR>
 48  *     InvalidRequestStateException;                            <BR>
 49  *   - addThreadFilter(null) results in throwing        <BR>
 50  *     NullPointerException or InvalidRequestStateException;    <BR>
 51  * (2) if this request is currently disabled,                   <BR>
 52  *     addThreadFilter(null) results in throwing                <BR>
 53  *     NullPointerException.                                    <BR>
 54  * <BR>
 55  * The test has three phases and works as follows.              <BR>
 56  * <BR>
 57  * In first phase,                                                      <BR>
 58  * upon launching debuggee's VM which will be suspended,                <BR>
 59  * a debugger waits for the VMStartEvent within a predefined            <BR>
 60  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 61  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 62  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 63  * and waits for the event within the predefined time interval.         <BR>
 64  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 65  * Upon getting the ClassPrepareEvent,                                  <BR>
 66  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 67  * within debuggee's special methodForCommunication().                  <BR>
 68  * <BR>
 69  * In second phase, the debugger creates ThreadDeathRequest             <BR>
 70  * and performs the above checks.                                       <BR>
 71  * <BR>
 72  * In third phase, at the end of the test,                              <BR>
 73  * the debuggee changes the value of the "instruction"                  <BR>
 74  * to inform the debugger of checks finished, and both end.             <BR>
 75  * <BR>
 76  */

150     private int runThis (String argv[], PrintStream out) {
151 
152         argsHandler     = new ArgumentHandler(argv);
153         logHandler      = new Log(out, argsHandler);
154         Binder binder   = new Binder(argsHandler, logHandler);
155 
156         waitTime        = argsHandler.getWaitTime() * 60000;
157 
158         try {
159             log2("launching a debuggee :");
160             log2("       " + debuggeeName);
161             if (argsHandler.verbose()) {
162                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
163             } else {
164                 debuggee = binder.bindToDebugee(debuggeeName);
165             }
166             if (debuggee == null) {
167                 log3("ERROR: no debuggee launched");
168                 return FAILED;
169             }
170             log2("debuggee launched");
171         } catch ( Exception e ) {
172             log3("ERROR: Exception : " + e);
173             log2("       test cancelled");
174             return FAILED;
175         }
176 
177         debuggee.redirectOutput(logHandler);
178 
179         vm = debuggee.VM();
180 
181         eventQueue = vm.eventQueue();
182         if (eventQueue == null) {
183             log3("ERROR: eventQueue == null : TEST ABORTED");
184             vm.exit(PASS_BASE);
185             return FAILED;
186         }
187 
188         log2("invocation of the method runTest()");
189         switch (runTest()) {
190 
191             case 0 :  log2("test phase has finished normally");
192                       log2("   waiting for the debuggee to finish ...");
193                       debuggee.waitFor();
194 
195                       log2("......getting the debuggee's exit status");
196                       int status = debuggee.getStatus();
197                       if (status != PASS_BASE) {
198                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
199                               status + " != PASS_BASE");
200                           testExitCode = FAILED;
201                       } else {
202                           log2("......debuggee returned expected exit status: " +
203                               status + " == PASS_BASE");
204                       }
205                       break;
206 
207             default : log3("ERROR: runTest() returned unexpected value");
208 
209             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
210                       log2("......forcing: vm.exit();");
211                       testExitCode = FAILED;
212                       try {
213                           vm.exit(PASS_BASE);
214                       } catch ( Exception e ) {
215                           log3("ERROR: Exception : e");
216                       }
217                       break;
218 
219             case 2 :  log3("test cancelled due to VMDisconnectedException");
220                       log2("......trying: vm.process().destroy();");
221                       testExitCode = FAILED;
222                       try {
223                           Process vmProcess = vm.process();
224                           if (vmProcess != null) {
225                               vmProcess.destroy();
226                           }
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231             }
232 
233         return testExitCode;
234     }

238     * Return value: 0 - normal end of the test
239     *               1 - ubnormal end of the test
240     *               2 - VMDisconnectedException while test phase
241     */

243     private int runTest() {
244 
245         try {
246             testRun();
247 
248             log2("waiting for VMDeathEvent");
249             getEventSet();
250             if (eventIterator.nextEvent() instanceof VMDeathEvent)
251                 return 0;
252 
253             log3("ERROR: last event is not the VMDeathEvent");
254             return 1;
255         } catch ( VMDisconnectedException e ) {
256             log3("ERROR: VMDisconnectedException : " + e);
257             return 2;
258         } catch ( Exception e ) {
259             log3("ERROR: Exception : " + e);
260             return 1;
261         }
262 
263     }

265     private void testRun()
266                  throws JDITestRuntimeException, Exception {
267 
268         eventRManager = vm.eventRequestManager();
269 
270         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
271         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
272         cpRequest.addClassFilter(debuggeeName);
273 
274         cpRequest.enable();
275         vm.resume();
276         getEventSet();
277         cpRequest.disable();
278 
279         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
280         debuggeeClass = event.referenceType();
281 
282         if (!debuggeeClass.name().equals(debuggeeName))
283            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
284 
285         log2("      received: ClassPrepareEvent for debuggeeClass");
286 
287         String bPointMethod = "methodForCommunication";
288         String lineForComm  = "lineForComm";
289         BreakpointRequest bpRequest;
290 
291         ThreadReference mainThread = debuggee.threadByNameOrThrow("main");
292 
293         bpRequest = settingBreakpoint(mainThread,
294                                       debuggeeClass,
295                                       bPointMethod, lineForComm, "zero");
296         bpRequest.enable();
297 
298     //------------------------------------------------------  testing section
299 
300         log1("     TESTING BEGINS");
301 
302         for (int i = 0; ; i++) {
303 
304             vm.resume();
305             breakpointForCommunication();
306 
307             int instruction = ((IntegerValue)
308                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
309 
310             if (instruction == 0) {
311                 vm.resume();
312                 break;
313             }
314 
315             log1(":::::: case: # " + i);
316 
317             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
318 
319             log2("......setting up ThreadDeathRequest");
320             log2("...... ThreadDeathRequest tdr1 = eventRManager.createThreadDeathRequest();");
321             ThreadDeathRequest tdr1 = eventRManager.createThreadDeathRequest();
322             tdr1.addCountFilter(1);
323             tdr1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
324             tdr1.putProperty("number", "ThreadDeathRequest1");
325 
326             log2("...... tdr1.enable();");
327             tdr1.enable();
328 
329             try {
330                 log2("...... tdr1.addThreadFilter(mainThread);");
331                 log2("         InvalidRequestStateException is expected");
332                 tdr1.addThreadFilter(mainThread);
333                 log3("ERROR: no InvalidRequestStateException ");
334                 testExitCode = FAILED;
335             } catch ( InvalidRequestStateException e ) {
336                 log2("          InvalidRequestStateException");
337             } catch ( Exception e ) {
338                 log3("ERROR: unexpected Exception : " + e);
339                 testExitCode = FAILED;
340             }
341             try {
342                 log2("...... tdr1.addThreadFilter(null);");
343                 log2("         NullPointerException is expected");
344                 tdr1.addThreadFilter(null);
345                 log3("ERROR: no NullPointerException ");
346                 testExitCode = FAILED;
347             } catch ( NullPointerException e ) {
348                 log2("          NullPointerException");
349             } catch ( Exception e ) {
350                 log3("ERROR: unexpected Exception : " + e);
351                 testExitCode = FAILED;
352             }
353 
354             log2("...... tdr1.disable()");
355             tdr1.disable();
356 
357             try {
358                 log2("...... tdr1.addThreadFilter(null);");
359                 log2("         NullPointerException or InvalidRequestStateException is expected");
360                 tdr1.addThreadFilter(null);
361                 log3("ERROR: no Exception thrown");
362                 testExitCode = FAILED;
363             } catch ( NullPointerException e ) {
364                 log2("          NullPointerException");
365             } catch ( InvalidRequestStateException e ) {
366                 log2("          InvalidRequestStateException");
367             } catch ( Exception e ) {
368                 log3("ERROR: unexpected Exception : " + e);
369                 testExitCode = FAILED;
370             }
371             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
372         }
373         log1("    TESTING ENDS");
374         return;
375     }

377    /*
378     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
379     *                                             String, String, String)
380     *
381     * It sets up a breakpoint at given line number within a given method in a given class
382     * for a given thread.
383     *
384     * Return value: BreakpointRequest object  in case of success
385     *
386     * JDITestRuntimeException   in case of an Exception thrown within the method
387     */
388 
389     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
390                                                   ReferenceType testedClass,
391                                                   String methodName,
392                                                   String bpLine,
393                                                   String property)
394             throws JDITestRuntimeException {
395 
396         log2("......setting up a breakpoint:");
397         log2("       thread: " + thread + "; class: " + testedClass +
398                         "; method: " + methodName + "; line: " + bpLine);
399 
400         List              alllineLocations = null;
401         Location          lineLocation     = null;
402         BreakpointRequest breakpRequest    = null;
403 
404         try {
405             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
406 
407             alllineLocations = method.allLineLocations();
408 
409             int n =
410                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
411             if (n > alllineLocations.size()) {
412                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
413             } else {
414                 lineLocation = (Location) alllineLocations.get(n);
415                 try {
416                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
417                     breakpRequest.putProperty("number", property);
418                     breakpRequest.addThreadFilter(thread);
419                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
420                 } catch ( Exception e1 ) {
421                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
422                     breakpRequest    = null;
423                 }
424             }
425         } catch ( Exception e2 ) {
426             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
427             breakpRequest    = null;
428         }
429 
430         if (breakpRequest == null) {
431             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
432             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
433         }
434 
435         log2("      a breakpoint has been set up");
436         return breakpRequest;
437     }

440     private void getEventSet()
441                  throws JDITestRuntimeException {
442         try {
443 //            log2("       eventSet = eventQueue.remove(waitTime);");
444             eventSet = eventQueue.remove(waitTime);
445             if (eventSet == null) {
446                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
447             }
448 //            log2("       eventIterator = eventSet.eventIterator;");
449             eventIterator = eventSet.eventIterator();
450         } catch ( Exception e ) {
451             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
452         }
453     }

456     private void breakpointForCommunication()
457                  throws JDITestRuntimeException {
458 
459         log2("breakpointForCommunication");
460         getEventSet();
461 
462         if (eventIterator.nextEvent() instanceof BreakpointEvent)
463             return;
464 
465         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
466     }
