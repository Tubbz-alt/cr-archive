 36 /**
 37  * The test for the implementation of an object of the type     <BR>
 38  * StepRequest.                                                 <BR>
 39  *                                                              <BR>
 40  * The test checks that results of the method                   <BR>
 41  * <code>com.sun.jdi.StepRequest.addClassExclusionFilter()</code> <BR>
 42  * complies with its spec.                                              <BR>
 43  * <BR>
 44  * The test checks up on the following assertion:                       <BR>
 45  *    Throws: InvalidRequestStateException -                            <BR>
 46  *            if this request is currently enabled or has been deleted. <BR>
 47  *            Filters may be added only to disabled requests.           <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.                      <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assertion,                              <BR>
 64  * the debugger and the debuggee perform the following.                 <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 67  *   and to inform the debugger with the event.                         <BR>
 68  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 69  *   - creates an StepRequest and enables it,                           <BR>
 70  *   - invokes the method addClassExclusionFilter() on the request      <BR>
 71  *     expecting to catch InvalidRequestStateException,                 <BR>
 72  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 73  *     expecting to catch no InvalidRequestStateException,                      <BR>
 74  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 75  *     once again expecting to catch InvalidRequestStateException.              <BR>
 76  * <BR>
 77  * In third phase, at the end of the test, the debuggee changes         <BR>
 78  * the value of the "instruction" which the debugger and debuggee       <BR>
 79  * use to inform each other of needed actions,  and both end.           <BR>
 80  * <BR>
 81  */

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : " + e);
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : " + e);
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

253     *               1 - ubnormal end of the test

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if (eventIterator.nextEvent() instanceof VMDeathEvent)
265                 return 0;
266 
267             log3("ERROR: last event is not the VMDeathEvent");
268             return 1;
269         } catch ( VMDisconnectedException e ) {
270             log3("ERROR: VMDisconnectedException : " + e);
271             return 2;
272         } catch ( Exception e ) {
273             log3("ERROR: Exception : " + e);
274             return 1;
275         }
276

279     private void testRun()
280                  throws JDITestRuntimeException, Exception {
281 
282         eventRManager = vm.eventRequestManager();
283 
284         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
285         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
286         cpRequest.addClassFilter(debuggeeName);
287 
288         cpRequest.enable();
289         vm.resume();
290         getEventSet();
291         cpRequest.disable();
292 
293         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
294         debuggeeClass = event.referenceType();
295 
296         if (!debuggeeClass.name().equals(debuggeeName))
297            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
298 
299         log2("      received: ClassPrepareEvent for debuggeeClass");
300 
301         String bPointMethod = "methodForCommunication";
302         String lineForComm  = "lineForComm";
303 
304         ThreadReference   mainThread = threadByName("main");
305 
306         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
307                                              debuggeeClass,
308                                             bPointMethod, lineForComm, "zero");
309         bpRequest.enable();
310 
311     //------------------------------------------------------  testing section
312 
313         log1("     TESTING BEGINS");
314 
315         EventRequest  eventRequest1      = null;
316 
317         String        property1          = "StepRequest1";
318 
319 
320         for (int i = 0; ; i++) {
321 
322             vm.resume();
323             breakpointForCommunication();
324 
325             int instruction = ((IntegerValue)
326                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
327 
328             if (instruction == 0) {
329                 vm.resume();
330                 break;
331             }
332 
333             log1(":::::: case: # " + i);
334 
335             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
336 
337             switch (i) {
338 
339               case 0:
340                      ThreadReference thread1 = threadByName("main");
341                      eventRequest1 = setting23StepRequest(thread1, testedClassName1,
342                                               EventRequest.SUSPEND_NONE, property1);
343 
344                      eventRequest1.enable();
345 
346                      try {
347                          log2("......eventRequest1.addClassExclusionFilter(testedClassName1);");
348                          log2("        InvalidRequestStateException expected");
349                          ((StepRequest)eventRequest1).addClassExclusionFilter(testedClassName1);
350                          log3("ERROR: no InvalidRequestStateException");
351                          testExitCode = FAILED;
352                      } catch ( InvalidRequestStateException e ) {
353                          log2("        InvalidRequestStateException");
354                      }
355 
356                      log2("......eventRequest1.disable();");
357                      eventRequest1.disable();
358                      try {
359                          log2("......eventRequest1.addClassExclusionFilter(testedClassName1);");
360                          log2("        no InvalidRequestStateException expected");
361                          ((StepRequest)eventRequest1).addClassExclusionFilter(testedClassName1);
362                          log2("        no InvalidRequestStateException");
363                      } catch ( InvalidRequestStateException e ) {
364                          log3("ERROR: InvalidRequestStateException");
365                          testExitCode = FAILED;
366                      }
367 
368                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
369                      eventRManager.deleteEventRequest(eventRequest1);
370                      try {
371                          log2("......eventRequest1.addClassExclusionFilter(testedClassName1);");
372                          log2("        InvalidRequestStateException expected");
373                          ((StepRequest)eventRequest1).addClassExclusionFilter(testedClassName1);
374                          log3("ERROR: no InvalidRequestStateException");
375                          testExitCode = FAILED;
376                      } catch ( InvalidRequestStateException e ) {
377                          log2("        InvalidRequestStateException");
378                      }
379                       break;
380 
381               default:
382                       throw new JDITestRuntimeException("** default case 2 **");
383             }
384 
385             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
386         }
387         log1("    TESTING ENDS");
388         return;
389     }

391     private ThreadReference threadByName(String name)
392                  throws JDITestRuntimeException {
393 
394         List         all = vm.allThreads();
395         ListIterator li  = all.listIterator();
396 
397         for (; li.hasNext(); ) {
398             ThreadReference thread = (ThreadReference) li.next();
399             if (thread.name().equals(name))
400                 return thread;
401         }
402         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
403     }

405    /*
406     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
407     *                                             String, String, String)
408     *
409     * It sets up a breakpoint at given line number within a given method in a given class
410     * for a given thread.
411     *
412     * Return value: BreakpointRequest object  in case of success
413     *
414     * JDITestRuntimeException   in case of an Exception thrown within the method
415     */
416 
417     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
418                                                   ReferenceType testedClass,
419                                                   String methodName,
420                                                   String bpLine,
421                                                   String property)
422             throws JDITestRuntimeException {
423 
424         log2("......setting up a breakpoint:");
425         log2("       thread: " + thread + "; class: " + testedClass +
426                         "; method: " + methodName + "; line: " + bpLine);
427 
428         List              alllineLocations = null;
429         Location          lineLocation     = null;
430         BreakpointRequest breakpRequest    = null;
431 
432         try {
433             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
434 
435             alllineLocations = method.allLineLocations();
436 
437             int n =
438                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
439             if (n > alllineLocations.size()) {
440                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
441             } else {
442                 lineLocation = (Location) alllineLocations.get(n);
443                 try {
444                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
445                     breakpRequest.putProperty("number", property);
446                     breakpRequest.addThreadFilter(thread);
447                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
448                 } catch ( Exception e1 ) {
449                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
450                     breakpRequest    = null;
451                 }
452             }
453         } catch ( Exception e2 ) {
454             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
455             breakpRequest    = null;
456         }
457 
458         if (breakpRequest == null) {
459             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
460             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
461         }
462 
463         log2("      a breakpoint has been set up");
464         return breakpRequest;
465     }

468     private void getEventSet()
469                  throws JDITestRuntimeException {
470         try {
471 //            log2("       eventSet = eventQueue.remove(waitTime);");
472             eventSet = eventQueue.remove(waitTime);
473             if (eventSet == null) {
474                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
475             }
476 //            log2("       eventIterator = eventSet.eventIterator;");
477             eventIterator = eventSet.eventIterator();
478         } catch ( Exception e ) {
479             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
480         }
481     }

484     private void breakpointForCommunication()
485                  throws JDITestRuntimeException {
486 
487         log2("breakpointForCommunication");
488         getEventSet();
489         Event evt = eventIterator.nextEvent();
490 
491         if (evt instanceof BreakpointEvent)
492             return;
493 
494         log3("Didn't get expected BreakpointEvent, events are:");
495         while(true) {
496             log3("    event = " + evt);
497             if (!eventIterator.hasNext()) {
498                 break;
499             }
500             evt = eventIterator.nextEvent();
501         }
502 
503         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
504     }

508     private StepRequest setting23StepRequest ( ThreadReference thread,
509                                                String          testedClass,
510                                                int             suspendPolicy,
511                                                String          property        )
512             throws JDITestRuntimeException {
513         try {
514             log2("......setting up StepRequest:");
515             log2("       thread: " + thread + "; property: " + property);
516 
517             StepRequest
518             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_INTO);
519             str.putProperty("number", property);
520             str.addCountFilter(1);
521             str.setSuspendPolicy(suspendPolicy);
522             str.addClassExclusionFilter(testedClass);
523 
524             log2("      StepRequest has been set up");
525             return str;
526         } catch ( Exception e ) {
527             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
528             log3("       StepRequest HAS NOT BEEN SET UP");
529             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
530         }
