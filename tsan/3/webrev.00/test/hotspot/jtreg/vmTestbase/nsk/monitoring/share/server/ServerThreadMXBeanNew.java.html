24 package nsk.monitoring.share.server;
25 
26 import javax.management.*;
27 import com.sun.management.*;
28 
29 /**
30  * com.sun.management.ThreadMXBean implementation that delegates functionality to MBeanServer.
31  */
32 public class ServerThreadMXBeanNew extends ServerThreadMXBean implements ThreadMXBean{
33 
34     public ServerThreadMXBeanNew(MBeanServer mbeanServer) {
35         super(mbeanServer);
36     }
37 
38     public long[] getThreadUserTime(long[] ids) {
39         return (long[]) invokeMethod("getThreadUserTime",
40                 new Object[] { ids },
41                 new String[] { long[].class.getName() });
42     }
43 
44     public long[] getThreadCpuTime(long[] ids) {
45         return (long[]) invokeMethod("getThreadCpuTime",
46                 new Object[] { ids },
47                 new String[] { long[].class.getName() });
48     }
49 
50     public long[] getThreadAllocatedBytes(long[] ids) {
51         return (long[]) invokeMethod("getThreadAllocatedBytes",
52                 new Object[] { ids },
53                 new String[] { long[].class.getName() });
54     }
55 
56     public long getThreadAllocatedBytes(long id) {
57         return (Long) invokeMethod("getThreadAllocatedBytes",
58             new Object[] { id },
59             new String[] { long.class.getName() });
60     }
61 
62     public long getCurrentThreadAllocatedBytes() {
63         return getLongAttribute("CurrentThreadAllocatedBytes");
64     }
65 
66     public void setThreadAllocatedMemoryEnabled(boolean enabled) {
67         setBooleanAttribute("ThreadAllocatedMemoryEnabled", enabled);
68     }
69 
70     public boolean isThreadAllocatedMemorySupported() {
71         return getBooleanAttribute("ThreadAllocatedMemorySupported");
72     }
73 
74     public boolean isThreadAllocatedMemoryEnabled() {
75         return getBooleanAttribute("ThreadAllocatedMemoryEnabled");
76     }
77 }
