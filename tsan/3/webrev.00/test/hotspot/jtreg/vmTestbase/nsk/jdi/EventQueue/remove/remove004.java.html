 38 /**
 39  * The test for the implementation of an object of the type     <BR>
 40  * EventQueue.                                                  <BR>
 41  *                                                              <BR>
 42  * The test checks up that results of the method                <BR>
 43  * <code>com.sun.jdi.EventQueue.remove()</code>                 <BR>
 44  * complies with its spec.                                      <BR>
 45  * <BR>
 46  * Since method's assertion is                                  <BR>
 47  *     "Waits forever for the next available event."            <BR>
 48  * in the condition of a limited time for testing,              <BR>
 49  * the test case limits time for waiting for the next event     <BR>
 50  * to the standard value of WAITTIME set up in the testbase_nsk.<BR>
 51  * <BR>
 52  * The test works as follows.                                   <BR>
 53  * Upon launching debuggee's VM which will be suspended,        <BR>
 54  * a debugger waits for the VMStartEvent within a predefined    <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.<BR>
 56  * Upon getting the VMStartEvent, it makes the request          <BR>
 57  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,  <BR>
 58  * resumes the VM, and waits for the event within the predefined<BR>
 59  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 60  * Upon getting the ClassPrepareEvent,                          <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,<BR>
 62  * the debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 63  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 64  * to be suspended and to inform the debugger with the event.           <BR>
 65  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 66  * At the end, the debuggee changes the value of the "instruction"      <BR>
 67  * to inform the debugger of checks finished, and both end.             <BR>
 68  * <BR>
 69  * The check includes two steps.                                        <BR>
 70  * In first one, second thread waits for any incoming event from the    <BR>
 71  * debugger which is sleeping for some time; hence,                     <BR>
 72  * no events are expected to be received at the debugger end.           <BR>
 73  * In second, second thread is interrupted, and the debugger waits for  <BR>
 74  * a breakpoint event after the debuggee finishes sleeping.             <BR>
 75  */

243     private int runTest() {
244 
245         try {
246             testRun();
247 
248             log2("waiting for VMDeathEvent");
249             getEventSet();
250             if (eventIterator.nextEvent() instanceof VMDeathEvent)
251                 return 0;

254             return 1;
255         } catch ( VMDisconnectedException e ) {
256             log3("ERROR: VMDisconnectedException : " + e);
257             return 2;
258         } catch ( Exception e ) {
259             log3("ERROR: Exception : " + e);
260             return 1;
261         }

265     private void testRun()
266                  throws JDITestRuntimeException, Exception {
267 
268         eventRManager = vm.eventRequestManager();
269 
270         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
271         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
272         cpRequest.addClassFilter(debuggeeName);
273 
274         cpRequest.enable();
275         vm.resume();
276         getEventSet();
277         cpRequest.disable();
278 
279         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
280         debuggeeClass = event.referenceType();
281 
282         if (!debuggeeClass.name().equals(debuggeeName))
283            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
284 
285         trueValue = debuggeeClass.getValue(debuggeeClass.fieldByName("BOOLEAN_TRUE_VALUE"));
286 
287         log2("      received: ClassPrepareEvent for debuggeeClass");
288 
289         String bPointMethod = "methodForCommunication";
290         String lineForComm  = "lineForComm";
291         BreakpointRequest bpRequest;
292 
293         try {
294             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
295                                           debuggeeClass,
296                                           bPointMethod, lineForComm, "zero");
297         } catch ( Exception e ) {
298             throw e;
299         }
300         bpRequest.enable();
301 
302     //------------------------------------------------------  testing section
303 
304         log1("     TESTING BEGINS");
305         vm.resume();
306 
307         for (int i = 0; ; i++) {
308 
309             breakpointForCommunication();
310 
311             int instruction = ((IntegerValue)
312                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
313 
314             if (instruction == 0) {
315                 vm.resume();
316                 break;
317             }
318 
319             log1("  new check: # " + i);
320 
321             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
322 
323             vm.resume();
324 
325             Threadremove004 thread2 = new Threadremove004("thread2");
326             log2("       thread2 is created");
327 
328             synchronized (lockingObject) {
329                 synchronized (waitnotifyObj) {
330                     log2("       before: thread2.start()");
331                     thread2.start();
332 
333                     try {
334                         log2("       before:   waitnotifyObj.wait();");
335                         waitnotifyObj.wait();
336                     } catch ( Exception e) {
337                         log3("ERROR:  Exception : " + e );
338                         testExitCode = FAILED;
339                         throw new JDITestRuntimeException("** Exception while waiting: notify() **");
340                     }
341                 }
342             }
343             log2("mainThread is out of: synchronized (lockingObject)");
344 
345             Object waitObj = new Object();
346 
347             synchronized (waitObj) {
348                 waitObj.wait(waitTime);
349                 if (!thread2.isAlive()) {
350                     log3("ERROR:  thread2 is not alive");
351                     testExitCode = FAILED;
352                     break;
353                 }
354 
355                 thread2.interrupt();
356 
357                 for (int i2 = 0; i2 < waitTime; ) {
358                     waitObj.wait(1000);
359                     if (!thread2.isAlive()) {
360                         break;
361                     }
362                     i2 += 1000;
363                 }
364                 if (thread2.isAlive()) {
365                     log3("ERROR: thread2 is still alive");
366                     testExitCode = FAILED;
367                     break;
368                 }
369             }
370 
371             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
372         }
373         log1("    TESTING ENDS");
374         return;
375     }

377    /*
378     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
379     *                                             String, String, String)
380     *
381     * It sets up a breakpoint at given line number within a given method in a given class
382     * for a given thread.
383     *
384     * Return value: BreakpointRequest object  in case of success
385     *
386     * JDITestRuntimeException   in case of an Exception thrown within the method
387     */
388 
389     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
390                                                   ReferenceType testedClass,
391                                                   String methodName,
392                                                   String bpLine,
393                                                   String property)
394             throws JDITestRuntimeException {
395 
396         log2("......setting up a breakpoint:");
397         log2("       thread: " + thread + "; class: " + testedClass +
398                         "; method: " + methodName + "; line: " + bpLine);
399 
400         List              alllineLocations = null;
401         Location          lineLocation     = null;
402         BreakpointRequest breakpRequest    = null;
403 
404         try {
405             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
406 
407             alllineLocations = method.allLineLocations();
408 
409             int n =
410                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
411             if (n > alllineLocations.size()) {
412                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
413             } else {
414                 lineLocation = (Location) alllineLocations.get(n);
415                 try {
416                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
417                     breakpRequest.putProperty("number", property);
418                     breakpRequest.addThreadFilter(thread);
419                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
420                 } catch ( Exception e1 ) {
421                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
422                     breakpRequest    = null;
423                 }
424             }
425         } catch ( Exception e2 ) {
426             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
427             breakpRequest    = null;
428         }
429 
430         if (breakpRequest == null) {
431             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
432             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
433         }
434 
435         log2("      a breakpoint has been set up");
436         return breakpRequest;
437     }

440     private void getEventSet()
441                  throws JDITestRuntimeException {
442         try {
443 //            log2("       eventSet = eventQueue.remove(waitTime);");
444             eventSet = eventQueue.remove(waitTime);
445             if (eventSet == null) {
446                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
447             }
448 //            log2("       eventIterator = eventSet.eventIterator;");
449             eventIterator = eventSet.eventIterator();
450         } catch ( Exception e ) {
451             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
452         }
453     }

456     private void breakpointForCommunication()
457                  throws JDITestRuntimeException {
458 
459         log2("breakpointForCommunication");
460         getEventSet();
461 
462         if (eventIterator.nextEvent() instanceof BreakpointEvent)
463             return;
464 
465         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
466     }

480         public void run() {
481             log2("-----t2: method 'run' enter");

484                 waitnotifyObj.notify();
485             }
486             log2("-----t2: exited from block:  synchronized (waitnotifyObj)");
487             synchronized (lockingObject)  {
488                 log2("-----t2: entered into block:  synchronized (lockingObject)");
489             }
490             log2("-----t2: exited from block:  synchronized (lockingObject)");
491 
492             try {
493                 log2("-----t2:  eventSet = eventQueue.remove(); expects: InterruptedException");
494                 eventSet = eventQueue.remove();
495                 throw new JDITestRuntimeException("** return from eventQueue.remove(); **");
496             } catch (InterruptedException e1) {
497                 log2("-----t2:            InterruptedException");
498                 // Signal to debuggee to stop sleeping
499                 try {
500                     ((ClassType) debuggeeClass).setValue(debuggeeClass.fieldByName("stopSleeping"),

502                 } catch (InvalidTypeException | ClassNotLoadedException e) {
503                     log3("ERROR: -----t2: Exception : " + e);
504                     testExitCode = FAILED;
505                 }
506             } catch (Exception e) {
507                 log3("ERROR: -----t2: Exception : " + e);
508                 testExitCode = FAILED;
509             }
510 
511             log2("-----t2: method 'run' exit");
512             return;
513         }
