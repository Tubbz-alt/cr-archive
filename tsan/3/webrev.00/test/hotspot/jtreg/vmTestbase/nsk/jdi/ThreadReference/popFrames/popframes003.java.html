 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * <BR>
 46  * The test checks up that locks acquired by a popped frame,    <BR>
 47  * are released when it is popped.                              <BR>
 48  * This applies to synchronized methods being popped.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to perform the above check,                          <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates two threads, thread2 and thread3, which will  <BR>
 68  * call a tested synchronized poppedMethod() from their run() methods.  <BR>
 69  * After getting the threads started, the debuggee invokes the          <BR>
 70  * methodForCommunication() to inform the debugger of the threads.      <BR>
 71  * - The debugger sets up Breakpoint and MethodEntry Requests           <BR>
 72  * within the poppedMethod(), first one for the thread2 and             <BR>
 73  * second one for the thread3, and resumes debuggee's main thread.      <BR>
 74  * - The main thread releases the thread2 and the thread3 in the order  <BR>
 75  * which makes the thread2 to be suspended at the breakpoint within     <BR>
 76  * the poppedMethod and the thread3 to be waiting for releasing the     <BR>
 77  * synchronized poppedMethod by the thread2. Then                       <BR>
 78  * the main thread informs the debugger via methodForCommunication().   <BR>
 79  * - After getting both events, from the thread2 and the main,          <BR>
 80  * the debugger invokes the popFrames() method and waits for            <BR>
 81  * the MethodEntry event in poppedMethod from the thread3.              <BR>
 82  * If no the event, the test FAILED.                                    <BR>
 83  */
 84

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300 
301         if ( !vm.canPopFrames() ) {
302             log2("......vm.canPopFrames() == false : test is cancelled");
303             vm.resume();
304             return;
305         }
306 
307         String bPointMethod = "methodForCommunication";
308         String lineForComm  = "lineForComm";
309 
310         ThreadReference threadMainRef = debuggee.threadByNameOrThrow("main");
311         try {
312             bpRequest = settingBreakpoint(threadMainRef,
313                                           debuggeeClass,
314                                           bPointMethod, lineForComm, "zero");
315         } catch ( Exception e ) {
316             throw e;
317         }
318         bpRequest.enable();
319 
320     //------------------------------------------------------  testing section
321 
322         log1("     TESTING BEGINS");
323 
324         EventSet mainThreadEventSet = null;
325 
326         int flag = 0;
327 
328 
329         vm.resume();
330         breakpointForCommunication();
331 
332         log2("......setting MethodEntryRequest (meRequest) in ForCommunication.methodForCommunication");
333         meRequest = settingMethodEntryRequest(threadMainRef,
334                                               debuggeeName + "$ForCommunication",
335                                               "zero");
336         log2("meRequest.enable();");
337         meRequest.enable();
338 
339 
340         String thread2Name         = "thread2";
341         ThreadReference thread2Ref = debuggee.threadByNameOrThrow(thread2Name);
342 
343         String thread3Name         = "thread3";
344         ThreadReference thread3Ref = debuggee.threadByNameOrThrow(thread3Name);
345 
346         String poppedMethod    = "poppedMethod";
347         String breakpointLine  = "breakpointLine";
348         String testVariable    = "testVar1";
349 
350         log2("......setting BreakpointRequest (bpRequest2) in poppedMethod");
351         List classes = vm.classesByName(debuggeeName + "$Popped");
352         ReferenceType poppedClass = (ReferenceType) classes.get(0);
353         bpRequest2 = settingBreakpoint(thread2Ref, poppedClass,
354                                           poppedMethod, breakpointLine, "one");
355         log2("......bpRequest2.enable();");
356         bpRequest2.enable();
357 
358         log2("......setting MethodEntryRequest (meRequest3) in poppedMethod");
359         meRequest3 = settingMethodEntryRequest(thread3Ref, debuggeeName + "$Popped",
360                                           "two");
361         log2("......meRequest3.enable();");
362         meRequest3.enable();
363 
364         log2("......eventSet.resume();");
365         eventSet.resume();
366 
367         log2(".....waiting for Breakpoint(thread2) and MethodEntry(main) Events");
368         for ( ; ; ) {
369 
370             Event event1;
371 
372             if (flag == 3)
373                 break;
374 
375             getEventSet();
376             event1 = eventIterator.nextEvent();
377 
378             if ( event1 instanceof MethodEntryEvent ) {
379                 mainThreadEventSet = eventSet;
380                 flag |= 1;
381                 log2("......event1 instanceof MethodEntryEvent");
382             } else if ( event1 instanceof BreakpointEvent ) {
383                 flag |= 2;
384                 log2("......event1 instanceof BreakpointEvent");
385             } else
386                 throw new JDITestRuntimeException("** event1 IS NOT Breakpoint or MethodEntry **");
387         }
388 
389         log2("......bpRequest2.disable();");
390         bpRequest2.disable();
391 
392         label0:
393         {
394 //            log2("        THREAD_STATUS_RUNNING == " + ThreadReference.THREAD_STATUS_RUNNING );
395 //            log2("        thread2's status      == " + thread2Ref.status());
396 
397             if (thread2Ref.isSuspended())
398                 log2("         thread2Ref.isSuspended()");
399             else {
400                 log3("ERROR:  !thread2Ref.isSuspended()");
401                 testExitCode = FAILED;
402                 break label0;
403             }
404 
405             log2("......mainThreadEventSet.resume();");
406             mainThreadEventSet.resume();
407 
408             log2("......up to WAITTIME loop of sleeping");
409             log2("         to get thread3 waiting on monitor after calling poppedMethod");
410             int thread3Status = 0;
411             for ( int i1 = 0; i1 < waitTime; i1 += 10000) {
412                 if (thread3Ref.status() == ThreadReference.THREAD_STATUS_MONITOR) {
413                     thread3Status = 1;
414                     break;
415                 }
416                 Thread.sleep(10000);
417             }
418             if (thread3Status != 1) {
419                 log3("ERROR: thread3Ref.status() != THREAD_STATUS_MONITOR");
420                 testExitCode = FAILED;
421                 break label0;
422             }
423 
424             log2("......StackFrame stackFrame = thread2Ref.frame(0);");
425             StackFrame stackFrame = thread2Ref.frame(0);
426 
427             log2("......thread2Ref.popFrames(stackFrame);");
428             try {
429                 thread2Ref.popFrames(stackFrame);
430             } catch ( IncompatibleThreadStateException e ) {
431                 log3("ERROR: IncompatibleThreadStateException");
432                 testExitCode = FAILED;
433                 break label0;
434             }
435 
436             log2(".....waiting for MethodEntry (main and thread3) Events");
437             {
438                 flag = 0;
439 
440                 for ( ; ; ) {
441 
442                     Event event1;
443 
444                     if (flag == 3)
445                         break;
446 
447                     getEventSet();
448                     event1 = eventIterator.nextEvent();
449 
450                     if ( !(event1 instanceof MethodEntryEvent) )
451                         throw new JDITestRuntimeException("** event IS NOT MethodEntry **");
452 
453                     if ( event1.request().getProperty("number").equals("zero") ) {
454                         flag |= 1;
455                         log2("......MethodEntry in mainThread");
456                     } else {
457                         flag |= 2;
458                         log2("......MethodEntry in thread3");
459                     }
460                 }
461             }
462 
463         }
464 
465         log2("......disabling requests and resuming vm");
466         meRequest3.disable();
467         vm.resume();
468 
469     //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
470 
471         log1("    TESTING ENDS");
472         return;
473     }

475   /*
476     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
477     *                                             String, String, String)
478     *
479     * It sets up a breakpoint at given line number within a given method in a given class
480     * for a given thread.
481     *
482     * Return value: BreakpointRequest object  in case of success
483     *
484     * JDITestRuntimeException   in case of an Exception thrown within the method
485     */
486 
487     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
488                                                   ReferenceType testedClass,
489                                                   String methodName,
490                                                   String bpLine,
491                                                   String property)
492             throws JDITestRuntimeException {
493 
494         log2("......setting up a breakpoint:");
495         log2("       thread: " + thread + "; class: " + testedClass +
496                         "; method: " + methodName + "; line: " + bpLine);
497 
498         List              alllineLocations = null;
499         Location          lineLocation     = null;
500         BreakpointRequest breakpRequest    = null;
501 
502         try {
503             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
504 
505             alllineLocations = method.allLineLocations();
506 
507             int n =
508                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
509             if (n > alllineLocations.size()) {
510                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
511             } else {
512                 lineLocation = (Location) alllineLocations.get(n);
513                 try {
514                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
515                     breakpRequest.putProperty("number", property);
516                     breakpRequest.addThreadFilter(thread);
517                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
518                 } catch ( Exception e1 ) {
519                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
520                     breakpRequest    = null;
521                 }
522             }
523         } catch ( Exception e2 ) {
524             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
525             breakpRequest    = null;
526         }
527 
528         if (breakpRequest == null) {
529             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
530             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
531         }
532 
533         log2("      a breakpoint has been set up");
534         return breakpRequest;
535     }

538    /*
539     * private MethodEntryRequest settingMethodEntryRequest(ThreadReference, ReferenceType,
540     *                                             String)
541     *
542     * It sets up a MethodEntryRequest in a given class for a given thread.
543     *
544     * Return value: MethodEntryRequest object  in case of success
545     *
546     * JDITestRuntimeException   in case of an Exception thrown within the method
547     */
548 
549     private MethodEntryRequest settingMethodEntryRequest ( ThreadReference thread,
550                                                          String className,
551                                                          String property)
552             throws JDITestRuntimeException {
553 
554         log2("......setting up a methodEntryRequest:");
555         log2("       thread: " + thread + "; className: " + className +
556                         "; property: " + property);
557 
558         List classList = vm.classesByName(className);
559         ReferenceType classRef = (ReferenceType) classList.get(0);
560 
561         MethodEntryRequest methodEntryRequest = null;
562 
563         try {
564              methodEntryRequest = eventRManager.createMethodEntryRequest();
565              methodEntryRequest.putProperty("number", property);
566              methodEntryRequest.addThreadFilter(thread);
567              methodEntryRequest.addClassFilter(classRef);
568              methodEntryRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
569         } catch ( Exception e ) {
570              log3("ERROR: Exception within methodEntryRequest() : " + e);
571              throw new JDITestRuntimeException("** FAILURE to set up a MethodEntryRequest **");
572         }
573 
574         log2("      a methodEntryRequest has been set up");
575         return methodEntryRequest;
576     }
