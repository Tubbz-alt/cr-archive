 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.deleteEventRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Once the eventRequest is deleted, no operations                   <BR>
 47  *    (for example, EventRequest.setEnabled(boolean)) are permitted -   <BR>
 48  *    attempts to do so will generally cause                            <BR>
 49  *    an InvalidRequestStateException.                                  <BR>
 50  * Note: only operations specified to throw InvalidRequestStateException <BR>
 51  * are not permitted.                                                   <BR>
 52  * Testcases unclude all 12 subclasses of EventRequest created by       <BR>
 53  * EventRequestManager.                                                 <BR>
 54  * <BR>
 55  * The test has three phases and works as follows.                      <BR>
 56  * <BR>
 57  * In first phase,                                                      <BR>
 58  * upon launching debuggee's VM which will be suspended,                <BR>
 59  * a debugger waits for the VMStartEvent within a predefined            <BR>
 60  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 61  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 62  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 63  * and waits for the event within the predefined time interval.         <BR>
 64  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 65  * Upon getting the ClassPrepareEvent,                                  <BR>
 66  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 67  * within debuggee's special methodForCommunication().                  <BR>
 68  * <BR>
 69  * In second phase to check the assertion,                              <BR>
 70  * the debugger and the debuggee perform the following.                 <BR>
 71  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 72  * - The debuggee prepares testcases and invokes                        <BR>
 73  *   the methodForCommunication to be suspended and                     <BR>
 74  *   to inform the debugger with the event.                             <BR>
 75  * - Upon getting the BreakpointEvent,                                  <BR>
 76  *   the debugger performs the checks required.                         <BR>
 77  * <BR>
 78  * In third phase, at the end of the test, the debuggee changes         <BR>
 79  * the value of the "instruction" which the debugger and debuggee       <BR>
 80  * use to inform each other of needed actions,  and both end.           <BR>
 81  * <BR>
 82  */

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : " + e);
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : " + e);
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if (eventIterator.nextEvent() instanceof VMDeathEvent)
266                 return 0;
267 
268             log3("ERROR: last event is not the VMDeathEvent");
269             return 1;
270         } catch ( VMDisconnectedException e ) {
271             log3("ERROR: VMDisconnectedException : " + e);
272             return 2;
273         } catch ( Exception e ) {
274             log3("ERROR: Exception : " + e);
275             return 1;
276         }

280     private void testRun()
281                  throws JDITestRuntimeException, Exception {
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304 
305         ThreadReference   mainThread = threadByName("main");
306 
307         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
308                                              debuggeeClass,
309                                             bPointMethod, lineForComm, "zero");
310         bpRequest.enable();
311 
312     //------------------------------------------------------  testing section
313 
314         EventRequest requestArray[] = new EventRequest[12];
315 
316         String fieldName1 = "testField1";
317         String fieldName2 = "testField2";
318         String fieldName3 = "testField3";
319 
320 
321         log1("     TESTING BEGINS");
322 
323         for (int i = 0; ; i++) {
324 
325             vm.resume();
326             breakpointForCommunication();
327 
328             int instruction = ((IntegerValue)
329                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
330 
331             if (instruction == 0) {
332                 vm.resume();
333                 break;
334             }
335 
336             log1(":::::: case: # " + i);
337 
338             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
339 
340 
341             Field field1 = debuggeeClass.fieldByName(fieldName1);
342 
343             ReferenceType refType = (ReferenceType)
344                                 debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2)).type();
345 
346             ThreadReference threadRef = (ThreadReference)
347                                 debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName3));
348 
349             String requests[] = { "AccessWatchpoint", "ModificationWatchpoint",
350                                   "Breakpoint",       "ClassPrepare",
351                                   "ClassUnload",      "Exception",
352                                   "MethodEntry",      "MethodExit",
353                                   "Step",             "ThreadDeath",
354                                   "ThreadStart",      "VMDeath"        };
355 
356             requestArray[0]  = eventRManager.createAccessWatchpointRequest(field1);
357             requestArray[1]  = eventRManager.createModificationWatchpointRequest(field1);
358             requestArray[2]  = eventRManager.createBreakpointRequest(breakpLocation);
359             requestArray[3]  = eventRManager.createClassPrepareRequest();
360             requestArray[4]  = eventRManager.createClassUnloadRequest();
361             requestArray[5]  = eventRManager.createExceptionRequest(refType, false, false);
362             requestArray[6]  = eventRManager.createMethodEntryRequest();
363             requestArray[7]  = eventRManager.createMethodExitRequest();
364             requestArray[8]  = eventRManager.createStepRequest(threadRef, StepRequest.STEP_MIN, StepRequest.STEP_OUT);
365             requestArray[9]  = eventRManager.createThreadDeathRequest();
366             requestArray[10] = eventRManager.createThreadStartRequest();
367             requestArray[11] = eventRManager.createVMDeathRequest();
368 
369             for (int i1 = 0; i1 < requestArray.length; i1++) {
370 
371                 log2("......eventRManager.deleteEventRequest(requestArray[i1]); :: " + requests[i1]);
372                 eventRManager.deleteEventRequest(requestArray[i1]);
373 
374                 try {
375                     requestArray[i1].addCountFilter(1);
376                     testExitCode = FAILED;
377                     log3("ERROR: NO InvalidRequestStateException for addCountFilter(1);");
378                 } catch ( InvalidRequestStateException e ) {
379                 }
380                 try {
381                     requestArray[i1].disable();
382                     testExitCode = FAILED;
383                     log3("ERROR: NO InvalidRequestStateException for disable();");
384                 } catch ( InvalidRequestStateException e ) {
385                 }
386                 try {
387                     requestArray[i1].enable();
388                     testExitCode = FAILED;
389                     log3("ERROR: NO InvalidRequestStateException for enable();");
390                 } catch ( InvalidRequestStateException e ) {
391                 }
392                 try {
393                     requestArray[i1].getProperty("number");
394                 } catch ( InvalidRequestStateException e ) {
395                     testExitCode = FAILED;
396                     log3("ERROR: InvalidRequestStateException for getProperty('number');");
397                 }
398                 try {
399                     requestArray[i1].isEnabled();
400                 } catch ( InvalidRequestStateException e ) {
401                     testExitCode = FAILED;
402                     log3("ERROR: InvalidRequestStateException for isEnabled();");
403                 }
404                 try {
405                     requestArray[i1].putProperty("number", "request" + i1);
406                 } catch ( InvalidRequestStateException e ) {
407                     testExitCode = FAILED;
408                     log3("ERROR: InvalidRequestStateException for putProperty('number', 'request' + i1);");
409                 }
410                 try {
411                     requestArray[i1].setEnabled(true);
412                     testExitCode = FAILED;
413                     log3("ERROR: NO InvalidRequestStateException for setEnabled(true);");
414                 } catch ( InvalidRequestStateException e ) {
415                 }
416                 try {
417                     requestArray[i1].setSuspendPolicy(EventRequest.SUSPEND_NONE);
418                     testExitCode = FAILED;
419                     log3("ERROR: NO InvalidRequestStateException for setSuspendPolicy(EventRequest.SUSPEND_NONE);");
420                 } catch ( InvalidRequestStateException e ) {
421                 }
422                 try {
423                     requestArray[i1].suspendPolicy();
424                 } catch ( InvalidRequestStateException e ) {
425                     testExitCode = FAILED;
426                     log3("ERROR: InvalidRequestStateException for suspendPolicy();");
427                 }
428             }
429 
430             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
431         }
432         log1("    TESTING ENDS");
433         return;
434     }

436     private ThreadReference threadByName(String name)
437                  throws JDITestRuntimeException {
438 
439         List         all = vm.allThreads();
440         ListIterator li  = all.listIterator();
441 
442         for (; li.hasNext(); ) {
443             ThreadReference thread = (ThreadReference) li.next();
444             if (thread.name().equals(name))
445                 return thread;
446         }
447         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
448     }

450    /*
451     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
452     *                                             String, String, String)
453     *
454     * It sets up a breakpoint at given line number within a given method in a given class
455     * for a given thread.
456     *
457     * Return value: BreakpointRequest object  in case of success
458     *
459     * JDITestRuntimeException   in case of an Exception thrown within the method
460     */
461 
462     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
463                                                   ReferenceType testedClass,
464                                                   String methodName,
465                                                   String bpLine,
466                                                   String property)
467             throws JDITestRuntimeException {
468 
469         log2("......setting up a breakpoint:");
470         log2("       thread: " + thread + "; class: " + testedClass +
471                         "; method: " + methodName + "; line: " + bpLine);
472 
473         List              alllineLocations = null;
474         Location          lineLocation     = null;
475         BreakpointRequest breakpRequest    = null;
476 
477         try {
478             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
479 
480             alllineLocations = method.allLineLocations();
481 
482             int n =
483                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
484             if (n > alllineLocations.size()) {
485                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
486             } else {
487                 lineLocation = (Location) alllineLocations.get(n);
488 breakpLocation = lineLocation;
489                 try {
490                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
491                     breakpRequest.putProperty("number", property);
492                     breakpRequest.addThreadFilter(thread);
493                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
494                 } catch ( Exception e1 ) {
495                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
496                     breakpRequest    = null;
497                 }
498             }
499         } catch ( Exception e2 ) {
500             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
501             breakpRequest    = null;
502         }
503 
504         if (breakpRequest == null) {
505             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
506             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
507         }
508 
509         log2("      a breakpoint has been set up");
510         return breakpRequest;
511     }

514     private void getEventSet()
515                  throws JDITestRuntimeException {
516         try {
517 //            log2("       eventSet = eventQueue.remove(waitTime);");
518             eventSet = eventQueue.remove(waitTime);
519             if (eventSet == null) {
520                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
521             }
522 //            log2("       eventIterator = eventSet.eventIterator;");
523             eventIterator = eventSet.eventIterator();
524         } catch ( Exception e ) {
525             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
526         }
527     }
