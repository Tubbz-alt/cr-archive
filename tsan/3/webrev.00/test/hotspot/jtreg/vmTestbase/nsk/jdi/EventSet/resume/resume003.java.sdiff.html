514     private ThreadReference threadByName(String name)
515                  throws JDITestRuntimeException {
516 
517         List         all = vm.allThreads();
518         ListIterator li  = all.listIterator();
519 
520         for (; li.hasNext(); ) {
521             ThreadReference thread = (ThreadReference) li.next();
522             if (thread.name().equals(name))
523                 return thread;
524         }
525         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
526     }
