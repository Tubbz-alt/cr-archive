355     private ThreadReference threadByName(String name)
356                  throws JDITestRuntimeException {
357 
358         List         all = vm.allThreads();
359         ListIterator li  = all.listIterator();
360 
361         for (; li.hasNext(); ) {
362             ThreadReference thread = (ThreadReference) li.next();
363             if (thread.name().equals(name))
364                 return thread;
365         }
366         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
367     }
