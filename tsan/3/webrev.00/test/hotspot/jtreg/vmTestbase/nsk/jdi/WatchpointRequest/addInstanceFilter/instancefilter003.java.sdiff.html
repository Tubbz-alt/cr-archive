426     private ThreadReference threadByName(String name)
427                  throws JDITestRuntimeException {
428 
429         List         all = vm.allThreads();
430         ListIterator li  = all.listIterator();
431 
432         for (; li.hasNext(); ) {
433             ThreadReference thread = (ThreadReference) li.next();
434             if (thread.name().equals(name))
435                 return thread;
436         }
437         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
438     }
