347     private ThreadReference threadByName(String name)
348                  throws JDITestRuntimeException {
349 
350         List         all = vm.allThreads();
351         ListIterator li  = all.listIterator();
352 
353         for (; li.hasNext(); ) {
354             ThreadReference thread = (ThreadReference) li.next();
355             if (thread.name().equals(name))
356                 return thread;
357         }
358         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
359     }
