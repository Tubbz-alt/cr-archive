 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadStartRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadStartRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that the method restricts the events         <BR>
 46  * genereted by this request to those in the thread to filter.  <BR>
 47  * <BR>
 48  * The test has three phases and works as follows.              <BR>
 49  * <BR>
 50  * In first phase,                                                      <BR>
 51  * upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 55  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 56  * and waits for the event within the predefined time interval.         <BR>
 57  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 58  * Upon getting the ClassPrepareEvent,                                  <BR>
 59  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 60  * within debuggee's special methodForCommunication().                  <BR>
 61  * <BR>
 62  * In second phase to check the assertion,                              <BR>
 63  * the debugger and the debuggee perform the following.                 <BR>
 64  * - The debugger creates three ThreadStartRequests with                <BR>
 65  *   addThreadFilter(debuggee's thread2) invoked,                       <BR>
 66  *   resumes the debuggee and waits for the ThreadStartEvent.           <BR>
 67  * - Debuggee's main thread creates and starts new thread, thread2, and <BR>
 68  *   invokes the methodForCommunication                                 <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting ThreadStartEvent, the debugger checks                 <BR>
 71  *   if EventSet contains all three events; if no, the test FAILED.     <BR>
 72  * Note. To inform each other of needed actions, the debugger and       <BR>
 73  *       and the debuggee use debuggee's variable "instruction".        <BR>
 74  * <BR>
 75  * In third phase, at the end of the test,                              <BR>
 76  * the debuggee changes the value of the "instruction"                  <BR>
 77  * to inform the debugger of checks finished, and both end.             <BR>
 78  * <BR>
 79  */

153     private int runThis (String argv[], PrintStream out) {
154 
155         argsHandler     = new ArgumentHandler(argv);
156         logHandler      = new Log(out, argsHandler);
157         Binder binder   = new Binder(argsHandler, logHandler);
158 
159         waitTime        = argsHandler.getWaitTime() * 60000;
160 
161         try {
162             log2("launching a debuggee :");
163             log2("       " + debuggeeName);
164             if (argsHandler.verbose()) {
165                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
166             } else {
167                 debuggee = binder.bindToDebugee(debuggeeName);
168             }
169             if (debuggee == null) {
170                 log3("ERROR: no debuggee launched");
171                 return FAILED;
172             }
173             log2("debuggee launched");
174         } catch ( Exception e ) {
175             log3("ERROR: Exception : " + e);
176             log2("       test cancelled");
177             return FAILED;
178         }
179 
180         debuggee.redirectOutput(logHandler);
181 
182         vm = debuggee.VM();
183 
184         eventQueue = vm.eventQueue();
185         if (eventQueue == null) {
186             log3("ERROR: eventQueue == null : TEST ABORTED");
187             vm.exit(PASS_BASE);
188             return FAILED;
189         }
190 
191         log2("invocation of the method runTest()");
192         switch (runTest()) {
193 
194             case 0 :  log2("test phase has finished normally");
195                       log2("   waiting for the debuggee to finish ...");
196                       debuggee.waitFor();
197 
198                       log2("......getting the debuggee's exit status");
199                       int status = debuggee.getStatus();
200                       if (status != PASS_BASE) {
201                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
202                               status + " != PASS_BASE");
203                           testExitCode = FAILED;
204                       } else {
205                           log2("......debuggee returned expected exit status: " +
206                               status + " == PASS_BASE");
207                       }
208                       break;
209 
210             default : log3("ERROR: runTest() returned unexpected value");
211 
212             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
213                       log2("......forcing: vm.exit();");
214                       testExitCode = FAILED;
215                       try {
216                           vm.exit(PASS_BASE);
217                       } catch ( Exception e ) {
218                           log3("ERROR: Exception : e");
219                       }
220                       break;
221 
222             case 2 :  log3("test cancelled due to VMDisconnectedException");
223                       log2("......trying: vm.process().destroy();");
224                       testExitCode = FAILED;
225                       try {
226                           Process vmProcess = vm.process();
227                           if (vmProcess != null) {
228                               vmProcess.destroy();
229                           }
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : e");
232                       }
233                       break;
234             }
235 
236         return testExitCode;
237     }

241     * Return value: 0 - normal end of the test
242     *               1 - ubnormal end of the test
243     *               2 - VMDisconnectedException while test phase
244     */

246     private int runTest() {
247 
248         try {
249             testRun();
250 
251             log2("waiting for VMDeathEvent");
252             getEventSet();
253             if (eventIterator.nextEvent() instanceof VMDeathEvent)
254                 return 0;
255 
256             log3("ERROR: last event is not the VMDeathEvent");
257             return 1;
258         } catch ( VMDisconnectedException e ) {
259             log3("ERROR: VMDisconnectedException : " + e);
260             return 2;
261         } catch ( Exception e ) {
262             log3("ERROR: Exception : " + e);
263             return 1;
264         }
265 
266     }

268     private void testRun()
269                  throws JDITestRuntimeException, Exception {
270 
271         eventRManager = vm.eventRequestManager();
272 
273         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
274         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
275         cpRequest.addClassFilter(debuggeeName);
276 
277         cpRequest.enable();
278         vm.resume();
279         getEventSet();
280         cpRequest.disable();
281 
282         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
283         debuggeeClass = event.referenceType();
284 
285         if (!debuggeeClass.name().equals(debuggeeName))
286            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292         BreakpointRequest bpRequest;
293 
294         ThreadReference mainThread = debuggee.threadByNameOrThrow("main");
295 
296         bpRequest = settingBreakpoint(mainThread,
297                                       debuggeeClass,
298                                       bPointMethod, lineForComm, "zero");
299         bpRequest.enable();
300 
301     //------------------------------------------------------  testing section
302 
303         for (int i = 0; ; i++) {
304 
305             vm.resume();
306             breakpointForCommunication();
307 
308             int instruction = ((IntegerValue)
309                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
310 
311             if (instruction == 0) {
312                 vm.resume();
313                 break;
314             }
315 
316             log1(":::::: case: # " + i);
317 
318             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
319 
320             ThreadReference testThread = (ThreadReference)
321                                   debuggeeClass.getValue(debuggeeClass.fieldByName("thread2"));
322 
323             log1("     TESTING BEGINS");
324 
325             log2("......setting up ThreadStartRequest");
326             ThreadStartRequest tsr1 = eventRManager.createThreadStartRequest();
327             tsr1.addThreadFilter(testThread);
328             tsr1.addCountFilter(1);
329             tsr1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
330             tsr1.putProperty("number", "ThreadStartRequest1");
331             tsr1.enable();
332 
333             ThreadStartRequest tsr2 = eventRManager.createThreadStartRequest();
334             tsr2.addThreadFilter(testThread);
335             tsr2.addCountFilter(1);
336             tsr2.setSuspendPolicy(EventRequest.SUSPEND_ALL);
337             tsr2.putProperty("number", "ThreadStartRequest2");
338             tsr2.enable();
339 
340             ThreadStartRequest tsr3 = eventRManager.createThreadStartRequest();
341             tsr3.addThreadFilter(testThread);
342             tsr3.addCountFilter(1);
343             tsr3.setSuspendPolicy(EventRequest.SUSPEND_ALL);
344             tsr3.putProperty("number", "ThreadStartRequest3");
345             tsr3.enable();
346 
347             log2("......vm.resume();");
348             vm.resume();
349 
350             log2("......waiting for ThreadStartEvent");
351             getEventSet();
352             EventSet eventSet1 = eventSet;
353             if ( !(eventIterator.nextEvent() instanceof ThreadStartEvent) ) {
354                 testExitCode = FAILED;
355                 log3("ERROR: new event is not ThreadStartEvent");
356                 return;
357             }
358             tsr1.disable();
359             tsr2.disable();
360             tsr3.disable();
361 
362             int setSize = eventSet1.size();
363             log2("......eventSet1.size() == " + setSize);
364             if (setSize != 3) {
365                 log3("ERROR: eventSet1.size() != 3  :: " + setSize);
366                 testExitCode = FAILED;
367 
368                 EventIterator evI1 = eventSet1.eventIterator();
369                 while (evI1.hasNext()) {
370                     Event ev1 = evI1.nextEvent();
371                     log2("........event property : " + ev1.request().getProperty("number"));
372                 }
373             }
374             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
375         }
376         log1("    TESTING ENDS");
377         return;
378     }

380    /*
381     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
382     *                                             String, String, String)
383     *
384     * It sets up a breakpoint at given line number within a given method in a given class
385     * for a given thread.
386     *
387     * Return value: BreakpointRequest object  in case of success
388     *
389     * JDITestRuntimeException   in case of an Exception thrown within the method
390     */
391 
392     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
393                                                   ReferenceType testedClass,
394                                                   String methodName,
395                                                   String bpLine,
396                                                   String property)
397             throws JDITestRuntimeException {
398 
399         log2("......setting up a breakpoint:");
400         log2("       thread: " + thread + "; class: " + testedClass +
401                         "; method: " + methodName + "; line: " + bpLine);
402 
403         List              alllineLocations = null;
404         Location          lineLocation     = null;
405         BreakpointRequest breakpRequest    = null;
406 
407         try {
408             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
409 
410             alllineLocations = method.allLineLocations();
411 
412             int n =
413                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
414             if (n > alllineLocations.size()) {
415                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
416             } else {
417                 lineLocation = (Location) alllineLocations.get(n);
418                 try {
419                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
420                     breakpRequest.putProperty("number", property);
421                     breakpRequest.addThreadFilter(thread);
422                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
423                 } catch ( Exception e1 ) {
424                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
425                     breakpRequest    = null;
426                 }
427             }
428         } catch ( Exception e2 ) {
429             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
430             breakpRequest    = null;
431         }
432 
433         if (breakpRequest == null) {
434             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
435             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
436         }
437 
438         log2("      a breakpoint has been set up");
439         return breakpRequest;
440     }

443     private void getEventSet()
444                  throws JDITestRuntimeException {
445         try {
446 //            log2("       eventSet = eventQueue.remove(waitTime);");
447             eventSet = eventQueue.remove(waitTime);
448             if (eventSet == null) {
449                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
450             }
451 //            log2("       eventIterator = eventSet.eventIterator;");
452             eventIterator = eventSet.eventIterator();
453         } catch ( Exception e ) {
454             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
455         }
456     }

463         getEventSet();

466             return;
467
