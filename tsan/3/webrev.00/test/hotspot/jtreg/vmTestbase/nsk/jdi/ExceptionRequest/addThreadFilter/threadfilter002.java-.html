 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addThreadFilter()</code>  <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  *            Filters may be added only to disabled requests.   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates an ExceptionRequest and enables it,                      <BR>
 73  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 74  *     to catch InvalidRequestStateException,                           <BR>
 75  *   - disables the request and invokes the method addThreadFilter()    <BR>
 76  *     expecting to catch no InvalidRequestStateException,              <BR>
 77  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 78  *     once again expecting to catch InvalidRequestStateException.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  */

168     private int runThis (String argv[], PrintStream out) {
169 
170         argsHandler     = new ArgumentHandler(argv);
171         logHandler      = new Log(out, argsHandler);
172         Binder binder   = new Binder(argsHandler, logHandler);
173 
174         waitTime        = argsHandler.getWaitTime() * 60000;
175 
176         try {
177             log2("launching a debuggee :");
178             log2("       " + debuggeeName);
179             if (argsHandler.verbose()) {
180                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
181             } else {
182                 debuggee = binder.bindToDebugee(debuggeeName);
183             }
184             if (debuggee == null) {
185                 log3("ERROR: no debuggee launched");
186                 return FAILED;
187             }
188             log2("debuggee launched");
189         } catch ( Exception e ) {
190             log3("ERROR: Exception : " + e);
191             log2("       test cancelled");
192             return FAILED;
193         }
194 
195         debuggee.redirectOutput(logHandler);
196 
197         vm = debuggee.VM();
198 
199         eventQueue = vm.eventQueue();
200         if (eventQueue == null) {
201             log3("ERROR: eventQueue == null : TEST ABORTED");
202             vm.exit(PASS_BASE);
203             return FAILED;
204         }
205 
206         log2("invocation of the method runTest()");
207         switch (runTest()) {
208 
209             case 0 :  log2("test phase has finished normally");
210                       log2("   waiting for the debuggee to finish ...");
211                       debuggee.waitFor();
212 
213                       log2("......getting the debuggee's exit status");
214                       int status = debuggee.getStatus();
215                       if (status != PASS_BASE) {
216                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
217                               status + " != PASS_BASE");
218                           testExitCode = FAILED;
219                       } else {
220                           log2("......debuggee returned expected exit status: " +
221                               status + " == PASS_BASE");
222                       }
223                       break;
224 
225             default : log3("ERROR: runTest() returned unexpected value");
226 
227             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
228                       log2("......forcing: vm.exit();");
229                       testExitCode = FAILED;
230                       try {
231                           vm.exit(PASS_BASE);
232                       } catch ( Exception e ) {
233                           log3("ERROR: Exception : " + e);
234                       }
235                       break;
236 
237             case 2 :  log3("test cancelled due to VMDisconnectedException");
238                       log2("......trying: vm.process().destroy();");
239                       testExitCode = FAILED;
240                       try {
241                           Process vmProcess = vm.process();
242                           if (vmProcess != null) {
243                               vmProcess.destroy();
244                           }
245                       } catch ( Exception e ) {
246                           log3("ERROR: Exception : " + e);
247                       }
248                       break;
249             }
250 
251         return testExitCode;
252     }

256     * Return value: 0 - normal end of the test
257     *               1 - ubnormal end of the test
258     *               2 - VMDisconnectedException while test phase
259     */

261     private int runTest() {
262 
263         try {
264             testRun();
265 
266             log2("waiting for VMDeathEvent");
267             getEventSet();
268             if (eventIterator.nextEvent() instanceof VMDeathEvent)
269                 return 0;
270 
271             log3("ERROR: last event is not the VMDeathEvent");
272             return 1;
273         } catch ( VMDisconnectedException e ) {
274             log3("ERROR: VMDisconnectedException : " + e);
275             return 2;
276         } catch ( Exception e ) {
277             log3("ERROR: Exception : " + e);
278             return 1;
279         }
280

283     private void testRun()
284                  throws JDITestRuntimeException, Exception {
285 
286         eventRManager = vm.eventRequestManager();
287 
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291 
292         cpRequest.enable();
293         vm.resume();
294         getEventSet();
295         cpRequest.disable();
296 
297         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
298         debuggeeClass = event.referenceType();
299 
300         if (!debuggeeClass.name().equals(debuggeeName))
301            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
302 
303         log2("      received: ClassPrepareEvent for debuggeeClass");
304 
305         String bPointMethod = "methodForCommunication";
306         String lineForComm  = "lineForComm";
307 
308         ThreadReference   mainThread = threadByName("main");
309 
310         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
311                                              debuggeeClass,
312                                             bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315     //------------------------------------------------------  testing section
316 
317         log1("     TESTING BEGINS");
318 
319         EventRequest eventRequest1 = null;
320 
321         ThreadReference thread1     = null;
322         String          thread1Name = "thread1";
323 
324         String property1 = "ExceptionRequest1";
325 
326         ReferenceType testClassReference = null;
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             switch (i) {
346               case 0:
347                      testClassReference =
348                             (ReferenceType) vm.classesByName(testedClassName).get(0);
349                      thread1 = (ThreadReference) debuggeeClass.getValue(
350                                         debuggeeClass.fieldByName(thread1Name));
351 
352                      eventRequest1 = setting2ExceptionRequest (null,
353                                              testClassReference,
354                                              EventRequest.SUSPEND_NONE, property1);
355 
356                      log2("......eventRequest1.enable();");
357                      eventRequest1.enable();
358 
359                      try {
360                          log2("......eventRequest1.addThreadFilter(thread1);");
361                          log2("        InvalidRequestStateException expected");
362                          ((ExceptionRequest)eventRequest1).addThreadFilter(thread1);
363                          log3("ERROR: no InvalidRequestStateException");
364                          testExitCode = FAILED;
365                      } catch ( InvalidRequestStateException e ) {
366                          log2("        InvalidRequestStateException");
367                      }
368 
369                      log2("......eventRequest1.disable();");
370                      eventRequest1.disable();
371                      try {
372                          log2("......eventRequest1.addThreadFilter(thread1);");
373                          log2("        no InvalidRequestStateException expected");
374                          ((ExceptionRequest)eventRequest1).addThreadFilter(thread1);
375                          log2("        no InvalidRequestStateException");
376                      } catch ( InvalidRequestStateException e ) {
377                          log3("ERROR: InvalidRequestStateException");
378                          testExitCode = FAILED;
379                      }
380 
381                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
382                      eventRManager.deleteEventRequest(eventRequest1);
383                      try {
384                          log2("......eventRequest1.addThreadFilter(thread1);");
385                          log2("        InvalidRequestStateException expected");
386                          ((ExceptionRequest)eventRequest1).addThreadFilter(thread1);
387                          log3("ERROR: no InvalidRequestStateException");
388                          testExitCode = FAILED;
389                      } catch ( InvalidRequestStateException e ) {
390                          log2("        InvalidRequestStateException");
391                      }
392 
393                      break;
394 
395               default:
396                       throw new JDITestRuntimeException("** default case 2 **");
397             }
398 
399             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
400         }
401         log1("    TESTING ENDS");
402         return;
403     }

405     private ThreadReference threadByName(String name)
406                  throws JDITestRuntimeException {
407 
408         List         all = vm.allThreads();
409         ListIterator li  = all.listIterator();
410 
411         for (; li.hasNext(); ) {
412             ThreadReference thread = (ThreadReference) li.next();
413             if (thread.name().equals(name))
414                 return thread;
415         }
416         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
417     }

419    /*
420     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
421     *                                             String, String, String)
422     *
423     * It sets up a breakpoint at given line number within a given method in a given class
424     * for a given thread.
425     *
426     * Return value: BreakpointRequest object  in case of success
427     *
428     * JDITestRuntimeException   in case of an Exception thrown within the method
429     */
430 
431     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
432                                                   ReferenceType testedClass,
433                                                   String methodName,
434                                                   String bpLine,
435                                                   String property)
436             throws JDITestRuntimeException {
437 
438         log2("......setting up a breakpoint:");
439         log2("       thread: " + thread + "; class: " + testedClass +
440                         "; method: " + methodName + "; line: " + bpLine);
441 
442         List              alllineLocations = null;
443         Location          lineLocation     = null;
444         BreakpointRequest breakpRequest    = null;
445 
446         try {
447             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
448 
449             alllineLocations = method.allLineLocations();
450 
451             int n =
452                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
453             if (n > alllineLocations.size()) {
454                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
455             } else {
456                 lineLocation = (Location) alllineLocations.get(n);
457                 try {
458                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
459                     breakpRequest.putProperty("number", property);
460                     breakpRequest.addThreadFilter(thread);
461                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
462                 } catch ( Exception e1 ) {
463                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
464                     breakpRequest    = null;
465                 }
466             }
467         } catch ( Exception e2 ) {
468             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
469             breakpRequest    = null;
470         }
471 
472         if (breakpRequest == null) {
473             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
474             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
475         }
476 
477         log2("      a breakpoint has been set up");
478         return breakpRequest;
479     }

482     private void getEventSet()
483                  throws JDITestRuntimeException {
484         try {
485 //            log2("       eventSet = eventQueue.remove(waitTime);");
486             eventSet = eventQueue.remove(waitTime);
487             if (eventSet == null) {
488                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
489             }
490 //            log2("       eventIterator = eventSet.eventIterator;");
491             eventIterator = eventSet.eventIterator();
492         } catch ( Exception e ) {
493             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
494         }
495     }

498     private void breakpointForCommunication()
499                  throws JDITestRuntimeException {
500 
501         log2("breakpointForCommunication");
502         getEventSet();
503 
504         if (eventIterator.nextEvent() instanceof BreakpointEvent)
505             return;
506 
507         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
508     }

512     private ExceptionRequest setting2ExceptionRequest ( ThreadReference thread,
513                                                        ReferenceType   testedClass,
514                                                        int             suspendPolicy,
515                                                        String          property       )
516             throws JDITestRuntimeException {
517         try {
518             log2("......setting up ExceptionRequest:");
519             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
520 
521             ExceptionRequest
522             excr = eventRManager.createExceptionRequest(null, true, true);
523             excr.putProperty("number", property);
524             if (thread != null)
525                 excr.addThreadFilter(thread);
526             excr.addClassFilter(testedClass);
527             excr.setSuspendPolicy(suspendPolicy);
528 
529             log2("      ExceptionRequest has been set up");
530             return excr;
531         } catch ( Exception e ) {
532             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
533             log3("       ExceptionRequest HAS NOT BEEN SET UP");
534             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
535         }
