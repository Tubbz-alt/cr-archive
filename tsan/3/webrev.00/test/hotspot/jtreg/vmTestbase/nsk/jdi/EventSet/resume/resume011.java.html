 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test case includes SUSPEND_NONE for a VMDeathEvent set.      <BR>
 46  * <BR>
 47  * The test has three phases and works as follows.              <BR>
 48  * <BR>
 49  * Upon launching debuggee's VM which will be suspended,                <BR>
 50  * a debugger waits for the VMStartEvent within a predefined            <BR>
 51  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 52  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 53  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 54  * and waits for the event within the predefined time interval.         <BR>
 55  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 56  * Upon getting the ClassPrepareEvent,                                  <BR>
 57  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 58  * within debuggee's special methodForCommunication().                  <BR>
 59  * <BR>
 60  * Then to check the above,                                             <BR>
 61  * the debugger and the debuggee perform the following.                 <BR>
 62  * - The debugger sets up a VMDeathRequest, resumes                     <BR>
 63  *   the debuggee, and waits for the VMDeathEvent.                      <BR>
 64  * - The debuggee ends to be resulting in the VMDeathEvent.                     <BR>
 65  * - Upon getting the event, the debugger performs the check.           <BR>
 66  * <BR>
 67  * Note. To inform each other of needed actions, the debugger and       <BR>
 68  *       and the debuggee use debuggee's variable "instruction".        <BR>
 69  * <BR>
 70  */
 71

 94     public static int run (String argv[], PrintStream out) {
 95 
 96         int exitCode = new resume011().runThis(argv, out);
 97 
 98         if (exitCode != PASSED) {
 99             System.out.println("TEST FAILED");
100         }
101         return testExitCode;
102     }

147     private int runThis (String argv[], PrintStream out) {
148 
149         argsHandler     = new ArgumentHandler(argv);
150         logHandler      = new Log(out, argsHandler);
151         Binder binder   = new Binder(argsHandler, logHandler);
152 
153         waitTime        = argsHandler.getWaitTime() * 60000;
154 
155         try {
156             log2("launching a debuggee :");
157             log2("       " + debuggeeName);
158             if (argsHandler.verbose()) {
159                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
160             } else {
161                 debuggee = binder.bindToDebugee(debuggeeName);
162             }
163             if (debuggee == null) {
164                 log3("ERROR: no debuggee launched");
165                 return FAILED;
166             }
167             log2("debuggee launched");
168         } catch ( Exception e ) {
169             log3("ERROR: Exception : " + e);
170             log2("       test cancelled");
171             return FAILED;
172         }
173 
174         debuggee.redirectOutput(logHandler);
175 
176         vm = debuggee.VM();
177 
178         eventQueue = vm.eventQueue();
179         if (eventQueue == null) {
180             log3("ERROR: eventQueue == null : TEST ABORTED");
181             vm.exit(PASS_BASE);
182             return FAILED;
183         }
184 
185         log2("invocation of the method runTest()");
186         switch (runTest()) {
187 
188             case 0 :  log2("test phase has finished normally");
189                       log2("   waiting for the debuggee to finish ...");
190                       debuggee.waitFor();
191 
192                       log2("......getting the debuggee's exit status");
193                       int status = debuggee.getStatus();
194                       if (status != PASS_BASE) {
195                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
196                               status + " != PASS_BASE");
197                           testExitCode = FAILED;
198                       } else {
199                           log2("......debuggee returned expected exit status: " +
200                               status + " == PASS_BASE");
201                       }
202                       break;
203 
204             default : log3("ERROR: runTest() returned unexpected value");
205 
206             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
207                       log2("......forcing: vm.exit();");
208                       testExitCode = FAILED;
209                       try {
210                           vm.exit(PASS_BASE);
211                       } catch ( Exception e ) {
212                           log3("ERROR: Exception : e");
213                       }
214                       break;
215 
216             case 2 :  log3("test cancelled due to VMDisconnectedException");
217                       log2("......trying: vm.process().destroy();");
218                       testExitCode = FAILED;
219                       try {
220                           Process vmProcess = vm.process();
221                           if (vmProcess != null) {
222                               vmProcess.destroy();
223                           }
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : e");
226                       }
227                       break;
228             }
229 
230         return testExitCode;
231     }

234    /*
235     * Return value: 0 - normal end of the test
236     *               1 - ubnormal end of the test
237     *               2 - VMDisconnectedException while test phase
238     */
239 
240     private int runTest() {
241 
242         try {
243             testRun();
244 
245             log2("waiting for VMDeathEvent");
246             getEventSet();
247             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
248                 log3("ERROR: last event is not the VMDeathEvent");
249                 return 1;
250             }
251             check();
252 
253             log2("waiting for VMDisconnectEvent");
254             getEventSet();
255             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
256                 log3("ERROR: last event is not the VMDisconnectEvent");
257                 return 1;
258             }
259 
260             return 0;
261 
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }
269 
270     }

272     private void testRun()
273                  throws JDITestRuntimeException, Exception {
274 
275         if ( !vm.canRequestVMDeathEvent() ) {
276             log2("......vm.canRequestVMDeathEvent == false :: test cancelled");
277             vm.exit(PASS_BASE);
278             return;
279         }
280 
281 
282         eventRManager = vm.eventRequestManager();
283 
284         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
285         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
286         cpRequest.addClassFilter(debuggeeName);
287 
288         cpRequest.enable();
289         vm.resume();
290         getEventSet();
291         cpRequest.disable();
292 
293         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
294         debuggeeClass = event.referenceType();
295 
296         if (!debuggeeClass.name().equals(debuggeeName))
297            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
298 
299         log2("      received: ClassPrepareEvent for debuggeeClass");
300 
301         String bPointMethod = "methodForCommunication";
302         String lineForComm  = "lineForComm";
303         BreakpointRequest bpRequest;
304         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
305         bpRequest = settingBreakpoint(mainThread,
306                                       debuggeeClass,
307                                       bPointMethod, lineForComm, "zero");
308         bpRequest.enable();
309 
310         vm.resume();
311 
312     //------------------------------------------------------  testing section
313 
314         log1("     TESTING BEGINS");
315 
316         EventRequest eventRequest1 = null;
317 //        EventRequest eventRequest2 = null;
318 //        EventRequest eventRequest3 = null;
319 
320         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
321         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
322         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
323 
324 
325 
326         breakpointForCommunication();
327 
328             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
329 
330         eventRequest1 = settingVMDeathRequest ( SUSPEND_NONE, "VMDeathRequest1");
331         eventRequest1.enable();
332 
333         mainThread.resume();
334 
335             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
336 
337         log1("    TESTING ENDS");
338         return;
339     }

341    /*
342     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
343     *                                             String, String, String)
344     *
345     * It sets up a breakpoint at given line number within a given method in a given class
346     * for a given thread.
347     *
348     * Return value: BreakpointRequest object  in case of success
349     *
350     * JDITestRuntimeException   in case of an Exception thrown within the method
351     */
352 
353     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
354                                                   ReferenceType testedClass,
355                                                   String methodName,
356                                                   String bpLine,
357                                                   String property)
358             throws JDITestRuntimeException {
359 
360         log2("......setting up a breakpoint:");
361         log2("       thread: " + thread + "; class: " + testedClass +
362                         "; method: " + methodName + "; line: " + bpLine);
363 
364         List              alllineLocations = null;
365         Location          lineLocation     = null;
366         BreakpointRequest breakpRequest    = null;
367 
368         try {
369             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
370 
371             alllineLocations = method.allLineLocations();
372 
373             int n =
374                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
375             if (n > alllineLocations.size()) {
376                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
377             } else {
378                 lineLocation = (Location) alllineLocations.get(n);
379                 try {
380                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
381                     breakpRequest.putProperty("number", property);
382                     breakpRequest.addThreadFilter(thread);
383                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
384                 } catch ( Exception e1 ) {
385                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
386                     breakpRequest    = null;
387                 }
388             }
389         } catch ( Exception e2 ) {
390             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
391             breakpRequest    = null;
392         }
393 
394         if (breakpRequest == null) {
395             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
396             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
397         }
398 
399         log2("      a breakpoint has been set up");
400         return breakpRequest;
401     }

404     private void getEventSet()
405                  throws JDITestRuntimeException {
406         try {
407 //            log2("       eventSet = eventQueue.remove(waitTime);");
408             eventSet = eventQueue.remove(waitTime);
409             if (eventSet == null) {
410                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
411             }
412 //            log2("       eventIterator = eventSet.eventIterator;");
413             eventIterator = eventSet.eventIterator();
414         } catch ( Exception e ) {
415             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
416         }
417     }

420     private void breakpointForCommunication()
421                  throws JDITestRuntimeException {
422 
423         log2("breakpointForCommunication");
424         getEventSet();
425 
426         if (eventIterator.nextEvent() instanceof BreakpointEvent)
427             return;
428 
429         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
430     }

434     private VMDeathRequest settingVMDeathRequest( int    suspendPolicy,
435                                                   String property       )
436             throws JDITestRuntimeException {
437         try {
438             log2("......setting up VMDeathRequest:");
439             log2("      suspendPolicy: " + suspendPolicy + "; property: " + property);
440 
441             VMDeathRequest
442             vmdr = eventRManager.createVMDeathRequest();
443             vmdr.putProperty("number", property);
444             vmdr.setSuspendPolicy(suspendPolicy);
445 
446             return vmdr;
447         } catch ( Exception e ) {
448             log3("ERROR: ATTENTION: Exception within settingVMDeathRequest() : " + e);
449             log3("       VMDeathRequest HAS NOT BEEN SET UP");
450             throw new JDITestRuntimeException("** FAILURE to set up a VMDeathRequest **");
451         }
452     }

461             testExitCode = FAILED;
462         }
