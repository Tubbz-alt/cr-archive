 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.addInstanceFilter()</code>     <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a StepRequest and enables it,                            <BR>
 72  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

254     * Return value: 0 - normal end of the test
255     *               1 - ubnormal end of the test
256     *               2 - VMDisconnectedException while test phase

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         if ( !vm.canUseInstanceFilters() ) {
285             log2("......vm.canUseInstanceFilters == false :: test cancelled");
286             vm.exit(PASS_BASE);
287             return;
288         }
289 
290 
291         eventRManager = vm.eventRequestManager();
292 
293         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
294         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
295         cpRequest.addClassFilter(debuggeeName);
296 
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307 
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         String bPointMethod = "methodForCommunication";
311         String lineForComm  = "lineForComm";
312 
313         ThreadReference   mainThread = threadByName("main");
314 
315         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
316                                              debuggeeClass,
317                                             bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320     //------------------------------------------------------  testing section
321 
322         log1("     TESTING BEGINS");
323 
324         EventRequest eventRequest1 = null;
325 
326         ThreadReference thread1 = null;
327         ThreadReference thread2 = null;
328 
329         String thread1Name = "thread1";
330         String thread2Name = "thread2";
331 
332         String property1 = "StepRequest1";
333 
334         ReferenceType testClassReference = null;
335 
336         String         arrayName = "objTC";
337         ObjectReference instance = null;
338 
339 
340         for (int i = 0; ; i++) {
341 
342             vm.resume();
343             breakpointForCommunication();
344 
345             int instruction = ((IntegerValue)
346                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
347 
348             if (instruction == 0) {
349                 vm.resume();
350                 break;
351             }
352 
353             log1(":::::: case: # " + i);
354 
355             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
356 
357             switch (i) {
358 
359               case 0:
360                       testClassReference =
361                            (ReferenceType) vm.classesByName(testedClassName).get(0);
362 
363                       thread1 = (ThreadReference) debuggeeClass.getValue(
364                                                   debuggeeClass.fieldByName(thread1Name));
365 
366                       eventRequest1 = setting2StepRequest (thread1,
367                                              EventRequest.SUSPEND_NONE, property1);
368 
369                       instance = (ObjectReference)
370                                  ((ArrayReference) debuggeeClass.getValue(
371                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
372 
373                       ((StepRequest) eventRequest1).addInstanceFilter(instance);
374 
375                       eventRequest1.enable();
376 
377                      try {
378                          log2("......eventRequest1.addInstanceFilter(instance);");
379                          log2("        InvalidRequestStateException expected");
380                          ((StepRequest) eventRequest1).addInstanceFilter(instance);
381                          log3("ERROR: no InvalidRequestStateException");
382                          testExitCode = FAILED;
383                      } catch ( InvalidRequestStateException e ) {
384                          log2("        InvalidRequestStateException");
385                      }
386 
387                      log2("......eventRequest1.disable();");
388                      eventRequest1.disable();
389                      try {
390                          log2("......eventRequest1.addInstanceFilter(instance);");
391                          log2("        no InvalidRequestStateException expected");
392                          ((StepRequest) eventRequest1).addInstanceFilter(instance);
393                          log2("        no InvalidRequestStateException");
394                      } catch ( InvalidRequestStateException e ) {
395                          log3("ERROR: InvalidRequestStateException");
396                          testExitCode = FAILED;
397                      }
398 
399                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
400                      eventRManager.deleteEventRequest(eventRequest1);
401                      try {
402                          log2("......eventRequest1.addInstanceFilter(instance);");
403                          log2("        InvalidRequestStateException expected");
404                          ((StepRequest) eventRequest1).addInstanceFilter(instance);
405                          log3("ERROR: no InvalidRequestStateException");
406                          testExitCode = FAILED;
407                      } catch ( InvalidRequestStateException e ) {
408                          log2("        InvalidRequestStateException");
409                      }
410                       break;
411 
412               default:
413                       throw new JDITestRuntimeException("** default case 2 **");
414             }
415 
416             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
417         }
418         log1("    TESTING ENDS");
419         return;
420     }

422     private ThreadReference threadByName(String name)
423                  throws JDITestRuntimeException {
424 
425         List         all = vm.allThreads();
426         ListIterator li  = all.listIterator();
427 
428         for (; li.hasNext(); ) {
429             ThreadReference thread = (ThreadReference) li.next();
430             if (thread.name().equals(name))
431                 return thread;
432         }
433         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
434     }

436    /*
437     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
438     *                                             String, String, String)
439     *
440     * It sets up a breakpoint at given line number within a given method in a given class
441     * for a given thread.
442     *
443     * Return value: BreakpointRequest object  in case of success
444     *
445     * JDITestRuntimeException   in case of an Exception thrown within the method
446     */
447 
448     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
449                                                   ReferenceType testedClass,
450                                                   String methodName,
451                                                   String bpLine,
452                                                   String property)
453             throws JDITestRuntimeException {
454 
455         log2("......setting up a breakpoint:");
456         log2("       thread: " + thread + "; class: " + testedClass +
457                         "; method: " + methodName + "; line: " + bpLine);
458 
459         List              alllineLocations = null;
460         Location          lineLocation     = null;
461         BreakpointRequest breakpRequest    = null;
462 
463         try {
464             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
465 
466             alllineLocations = method.allLineLocations();
467 
468             int n =
469                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
470             if (n > alllineLocations.size()) {
471                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
472             } else {
473                 lineLocation = (Location) alllineLocations.get(n);
474                 try {
475                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
476                     breakpRequest.putProperty("number", property);
477                     breakpRequest.addThreadFilter(thread);
478                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
479                 } catch ( Exception e1 ) {
480                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
481                     breakpRequest    = null;
482                 }
483             }
484         } catch ( Exception e2 ) {
485             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
486             breakpRequest    = null;
487         }
488 
489         if (breakpRequest == null) {
490             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
491             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
492         }
493 
494         log2("      a breakpoint has been set up");
495         return breakpRequest;
496     }

499     private void getEventSet()
500                  throws JDITestRuntimeException {
501         try {
502 //            log2("       eventSet = eventQueue.remove(waitTime);");
503             eventSet = eventQueue.remove(waitTime);
504             if (eventSet == null) {
505                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
506             }
507 //            log2("       eventIterator = eventSet.eventIterator;");
508             eventIterator = eventSet.eventIterator();
509         } catch ( Exception e ) {
510             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
511         }
512     }

515     private void breakpointForCommunication()
516                  throws JDITestRuntimeException {
517 
518         log2("breakpointForCommunication");
519         getEventSet();
520 
521         if (eventIterator.nextEvent() instanceof BreakpointEvent)
522             return;
523 
524         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
525     }

529     private StepRequest setting2StepRequest ( ThreadReference thread,
530                                               int             suspendPolicy,
531                                               String          property        )
532             throws JDITestRuntimeException {
533         try {
534             log2("......setting up StepRequest:");
535             log2("       thread: " + thread + "; property: " + property);
536 
537             StepRequest
538             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_INTO);
539             str.putProperty("number", property);
540             str.addCountFilter(2);
541             str.setSuspendPolicy(suspendPolicy);
542 
543             log2("      StepRequest has been set up");
544             return str;
545         } catch ( Exception e ) {
546             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
547             log3("       StepRequest HAS NOT BEEN SET UP");
548             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
549         }
