 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.eventIterator()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for each type of the following Events:  <BR>
 46  * AccessWatchpoint, ModificationWatchpoint, Breakpoint,        <BR>
 47  * Exception, MethodEntry, MethodExit, and Step,                <BR>
 48  *  - the method returns non-null object, and                   <BR>
 49  *  - object's class is subclass of class Iterator.             <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase the debugger and debuggee perform the following loop.<BR>
 66  * - The debugger creates new Request, resumes the debuggee, and        <BR>
 67  *   waits for corresponding Event.                                     <BR>
 68  * - The debuggee performes an action resulting in Event required.      <BR>
 69  * - Upon getting the Event, the debugger checks up on EventSet.        <BR>
 70  * <BR>
 71  * In third phase, at the end                                           <BR>
 72  * the debuggee changes the value of the "instruction"                  <BR>
 73  * to inform the debugger of checks finished, and both end.             <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * <BR>
 78  */

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugee(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : e");
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : e");
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

249     * Return value: 0 - normal end of the test
250     *               1 - ubnormal end of the test
251     *               2 - VMDisconnectedException while test phase
252     */

254     private int runTest() {
255 
256         try {
257             testRun();
258 
259             log2("waiting for VMDeathEvent");
260             getEventSet();
261 
262             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
263                 log3("ERROR: last event is not the VMDeathEvent");
264                 return 1;
265             }
266 
267             log2("waiting for VMDisconnectEvent");
268             getEventSet();
269 
270             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
271                 log3("ERROR: last event is not the VMDisconnectEvent");
272                 return 1;
273             }
274 
275             return 0;
276 
277         } catch ( VMDisconnectedException e ) {
278             log3("ERROR: VMDisconnectedException : " + e);
279             return 2;
280         } catch ( Exception e ) {
281             log3("ERROR: Exception : " + e);
282             return 1;
283         }
284

287     private void testRun()
288                  throws JDITestRuntimeException, Exception {
289 
290         eventRManager = vm.eventRequestManager();
291 
292         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
293         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
294         cpRequest.addClassFilter(debuggeeName);
295         cpRequest.enable();
296         vm.resume();
297         getEventSet();
298         cpRequest.disable();
299 
300         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
301         debuggeeClass = event.referenceType();
302 
303         if (!debuggeeClass.name().equals(debuggeeName))
304            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
305 
306         log2("      received: ClassPrepareEvent for debuggeeClass");
307 
308         String bPointMethod = "methodForCommunication";
309         String lineForComm  = "lineForComm";
310         BreakpointRequest bpRequest;
311 
312         ThreadReference mainThread = threadByName("main");
313 
314         bpRequest = settingBreakpoint(mainThread,
315                                       debuggeeClass,
316                                       bPointMethod, lineForComm, "zero");
317         bpRequest.enable();
318 
319     //------------------------------------------------------  testing section
320 
321         log1("     TESTING BEGINS");
322 
323             String accWatchpointName = "var1";
324             String modWatchpointName = "var2";
325             String bpLineName        = "breakpointLine";
326             String bpMethodName      = "method";
327             String awFieldName       = "awFieldName";
328             String mwFieldName       = "mwFieldName";
329             String excName           = "method";
330             String menName           = "method";
331             String mexName           = "method";
332 
333         EventRequest eRequest = null;
334 
335         for (int i = 0; ; i++) {
336 
337             vm.resume();
338             breakpointForCommunication();
339 
340             int instruction = ((IntegerValue)
341                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
342 
343             if (instruction == 0) {
344                 vm.resume();
345                 break;
346             }
347 
348             log1(":::::: case: # " + i);
349 
350             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
351 
352             log2("......ReferenceType testClass = (ReferenceType) (vm.classesByName(testedClassName)).get(0);");
353             ReferenceType testClass = (ReferenceType) (vm.classesByName(testedClassName)).get(0);
354 
355             switch (i) {
356 
357               case 0:
358                       eRequest = null;
359                       if (vm.canWatchFieldAccess()) {
360                           String awName = ( (StringReference) testClass.getValue(
361                                             testClass.fieldByName(awFieldName))).value();
362                           eRequest = settingAccessWatchpointRequest(mainThread,
363                                             testClass, awName, "AccessWatchpointRequest");
364                           eRequest.enable();
365                       }
366                       break;
367 
368               case 1:
369                       eRequest = null;
370                       if (vm.canWatchFieldModification() ) {
371                           String mwName = ( (StringReference) testClass.getValue(
372                                             testClass.fieldByName(mwFieldName))).value();
373                           eRequest = settingModificationWatchpointRequest(mainThread,
374                                             testClass, mwName, "ModificationWatchpointRequest");
375                           eRequest.enable();
376                       }
377                       break;
378 
379               case 2:
380                       eRequest = settingBreakpointRequest(mainThread, testClass,
381                                         bpMethodName, bpLineName,
382                                         EventRequest.SUSPEND_EVENT_THREAD, "BreakpointRequest");
383                       eRequest.enable();
384                       break;
385 
386               case 3:
387                       eRequest = settingExceptionRequest(mainThread, testClass,
388                                                                 "ExceptionRequest");
389                       eRequest.enable();
390                       break;
391 
392               case 4:
393                       eRequest = settingMethodEntryRequest(mainThread, testClass,
394                                                                 "MethodEntryRequest");
395                       eRequest.enable();
396                       break;
397 
398               case 5:
399                       eRequest = settingMethodExitRequest(mainThread, testClass,
400                                                                 "MethodExitRequest");
401                       eRequest.enable();
402                       break;
403 
404               case 6:
405                       eRequest = settingStepRequest(mainThread, "StepRequest");
406                       eRequest.enable();
407                       break;
408             }
409 
410                                  //  see setting up Acceess&Modification requests above
411             if (eRequest == null)
412                 continue;
413 
414             log2("......getting new Event and checking up on EventIterator");
415             vm.resume();
416             getEventSet();
417             eRequest.disable();
418 
419             if (eventIterator == null) {
420                 testExitCode = FAILED;
421                 log3("ERROR: eventIterator == null");
422             }
423             if ( !(eventIterator instanceof Iterator) ) {
424                 testExitCode = FAILED;
425                 log3("ERROR: eventIterator is NOT instanceof Iterator");
426             }
427 
428             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
429         }
430         log1("    TESTING ENDS");
431         return;
432     }

434     private ThreadReference threadByName(String name)
435                  throws JDITestRuntimeException {
436 
437         List         all = vm.allThreads();
438         ListIterator li  = all.listIterator();
439 
440         for (; li.hasNext(); ) {
441             ThreadReference thread = (ThreadReference) li.next();
442             if (thread.name().equals(name))
443                 return thread;
444         }
445         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
446     }

448    /*
449     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
450     *                                             String, String, String)
451     *
452     * It sets up a breakpoint at given line number within a given method in a given class
453     * for a given thread.
454     *
455     * Return value: BreakpointRequest object  in case of success
456     *
457     * JDITestRuntimeException   in case of an Exception thrown within the method
458     */
459 
460     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
461                                                   ReferenceType testedClass,
462                                                   String methodName,
463                                                   String bpLine,
464                                                   String property)
465             throws JDITestRuntimeException {
466 
467         log2("......setting up a breakpoint:");
468         log2("       thread: " + thread + "; class: " + testedClass +
469                         "; method: " + methodName + "; line: " + bpLine);
470 
471         List              alllineLocations = null;
472         Location          lineLocation     = null;
473         BreakpointRequest breakpRequest    = null;
474 
475         try {
476             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
477 
478             alllineLocations = method.allLineLocations();
479 
480             int n =
481                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
482             if (n > alllineLocations.size()) {
483                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
484             } else {
485                 lineLocation = (Location) alllineLocations.get(n);
486                 try {
487                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
488                     breakpRequest.putProperty("number", property);
489                     breakpRequest.addThreadFilter(thread);
490                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
491                 } catch ( Exception e1 ) {
492                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
493                     breakpRequest    = null;
494                 }
495             }
496         } catch ( Exception e2 ) {
497             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
498             breakpRequest    = null;
499         }
500 
501         if (breakpRequest == null) {
502             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
503             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
504         }
505 
506         log2("      a breakpoint has been set up");
507         return breakpRequest;
508     }

511     private void getEventSet()
512                  throws JDITestRuntimeException {
513         try {
514 //            log2("       eventSet = eventQueue.remove(waitTime);");
515             eventSet = eventQueue.remove(waitTime);
516             if (eventSet == null) {
517                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
518             }
519 //            log2("       eventIterator = eventSet.eventIterator;");
520             eventIterator = eventSet.eventIterator();
521         } catch ( Exception e ) {
522             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
523         }
524     }
