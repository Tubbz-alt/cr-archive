 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * PathSearchingVirtualMachine.                                 <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.PathSearchingVirtualMachine.classPath()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  * A List returned by the method contains only String elements  <BR>
 47  * and each element is a substring of String returned by        <BR>
 48  * the method System.getProperty("java.class.path")             <BR>
 49  * invoked in a debuggee.                                       <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assetion,                               <BR>
 66  * the debugger and the debuggee perform the following loop.            <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee prepares the check case and invokes                   <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent,                                  <BR>
 72  *   the debugger performs the check case required.                     <BR>
 73  * Note. To inform each other of needed actions, the debugger and       <BR>
 74  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 75  * <BR>
 76  * In third phase when at the end,                                      <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : e");
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : e");
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

247    /*
248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase
251     */
252 
253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299         BreakpointRequest bpRequest;
300 
301         try {
302             bpRequest = settingBreakpoint(threadByName("main"),
303                                           debuggeeClass,
304                                           bPointMethod, lineForComm, "zero");
305         } catch ( Exception e ) {
306             throw e;
307         }
308         bpRequest.enable();
309 
310     //------------------------------------------------------  testing section
311 
312         log1("     TESTING BEGINS");
313 
314         for (int i = 0; ; i++) {
315 
316             vm.resume();
317             breakpointForCommunication();
318 
319             int instruction = ((IntegerValue)
320                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
321 
322             if (instruction == 0) {
323                 vm.resume();
324                 break;
325             }
326 
327             log1("  new check: # " + i);
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
330 
331             try {
332                 PathSearchingVirtualMachine pvm;
333 
334                 String debuggeeClassPathName = "javaClassPath";
335 
336                 String debuggeeClassPath = ((StringReference) debuggeeClass.getValue(
337                                              debuggeeClass.fieldByName(debuggeeClassPathName))
338                                            ).value();
339 
340                 log2("......getting: pvm = (PathSearchingVirtualMachine) vm;");
341                 pvm = (PathSearchingVirtualMachine) vm;
342 
343                 log2("......getting: List classPath = pvm.classPath();");
344                 List classPath = pvm.classPath();
345 
346                 if (classPath == null) {
347                     log3("ERROR: classPath == null");
348                     testExitCode = FAILED;
349                 }
350 
351                 log2("......getting: ListIterator li = classPath.listIterator();");
352                 ListIterator li = classPath.listIterator();
353 
354                 log2("......loop of checking up on ListIterator elements; no Exception is expected");
355                 String str;
356                 for (; li.hasNext(); ) {
357 
358                     str = (String) li.next();
359                     log2("           : " + str);
360 
361                     if (debuggeeClassPath.indexOf(str) == -1) {
362                         log3("ERROR: element is not a substring of debuggeeClassPath");
363                         log2("       element == " + str);
364                         testExitCode = FAILED;
365                     }
366                 }
367 
368             } catch ( Exception e) {
369                 log3("ERROR: unspecified exception thrown: " + e);
370                 testExitCode = FAILED;
371             }
372 
373             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
374         }
375         log1("    TESTING ENDS");
376         return;
377     }

379     private ThreadReference threadByName(String name)
380                  throws JDITestRuntimeException {
381 
382         List         all = vm.allThreads();
383         ListIterator li  = all.listIterator();
384 
385         for (; li.hasNext(); ) {
386             ThreadReference thread = (ThreadReference) li.next();
387             if (thread.name().equals(name))
388                 return thread;
389         }
390         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
391     }

393    /*
394     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
395     *                                             String, String, String)
396     *
397     * It sets up a breakpoint at given line number within a given method in a given class
398     * for a given thread.
399     *
400     * Return value: BreakpointRequest object  in case of success
401     *
402     * JDITestRuntimeException   in case of an Exception thrown within the method
403     */
404 
405     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
406                                                   ReferenceType testedClass,
407                                                   String methodName,
408                                                   String bpLine,
409                                                   String property)
410             throws JDITestRuntimeException {
411 
412         log2("......setting up a breakpoint:");
413         log2("       thread: " + thread + "; class: " + testedClass +
414                         "; method: " + methodName + "; line: " + bpLine);
415 
416         List              alllineLocations = null;
417         Location          lineLocation     = null;
418         BreakpointRequest breakpRequest    = null;
419 
420         try {
421             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
422 
423             alllineLocations = method.allLineLocations();
424 
425             int n =
426                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
427             if (n > alllineLocations.size()) {
428                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
429             } else {
430                 lineLocation = (Location) alllineLocations.get(n);
431                 try {
432                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
433                     breakpRequest.putProperty("number", property);
434                     breakpRequest.addThreadFilter(thread);
435                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
436                 } catch ( Exception e1 ) {
437                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
438                     breakpRequest    = null;
439                 }
440             }
441         } catch ( Exception e2 ) {
442             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
443             breakpRequest    = null;
444         }
445 
446         if (breakpRequest == null) {
447             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
448             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
449         }
450 
451         log2("      a breakpoint has been set up");
452         return breakpRequest;
453     }

456     private void getEventSet()
457                  throws JDITestRuntimeException {
458         try {
459 //            log2("       eventSet = eventQueue.remove(waitTime);");
460             eventSet = eventQueue.remove(waitTime);
461             if (eventSet == null) {
462                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
463             }
464 //            log2("       eventIterator = eventSet.eventIterator;");
465             eventIterator = eventSet.eventIterator();
466         } catch ( Exception e ) {
467             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
468         }
469     }

472     private void breakpointForCommunication()
473                  throws JDITestRuntimeException {
474 
475         log2("breakpointForCommunication");
476         getEventSet();
477 
478         if (eventIterator.nextEvent() instanceof BreakpointEvent)
479             return;
480 
481         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
482     }
