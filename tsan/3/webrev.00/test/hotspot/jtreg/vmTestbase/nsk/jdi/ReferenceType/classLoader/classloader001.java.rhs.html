 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.classLoader()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The case for testing includes a classLoader object for type  <BR>
 46  * loaded through the bootstrap class loader.                   <BR>
 47  * <BR>
 48  * The test works as follows.                                           <BR>
 49  * Upon launching debuggee's VM which will be suspended,                <BR>
 50  * a debugger waits for the VMStartEvent within a predefined            <BR>
 51  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 52  * Upon getting the VMStartEvent, it makes the request                  <BR>
 53  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 54  * resumes the VM, and waits for the event within the predefined        <BR>
 55  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 56  * Upon getting the ClassPrepareEvent,                                  <BR>
 57  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 58  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 59  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 60  * to be suspended and to inform the debugger with the event.           <BR>
 61  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 62  * At the end, the debuggee changes the value of the "instruction"      <BR>
 63  * to inform the debugger of checks finished, and both end.             <BR>
 64  */

 88     public static int run (String argv[], PrintStream out) {
 89 
 90         int exitCode = new classloader001().runThis(argv, out);

 92         if (exitCode != PASSED) {
 93             System.out.println("TEST FAILED");
 94         }
 95         return testExitCode;
 96     }

139     private int runThis (String argv[], PrintStream out) {
140 
141         argsHandler     = new ArgumentHandler(argv);
142         logHandler      = new Log(out, argsHandler);
143         Binder binder   = new Binder(argsHandler, logHandler);
144 
145         waitTime        = argsHandler.getWaitTime() * 60000;
146 
147         try {
148             log2("launching a debuggee :");
149             log2("       " + debuggeeName);
150             if (argsHandler.verbose()) {
151                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
152             } else {
153                 debuggee = binder.bindToDebugee(debuggeeName);
154             }
155             if (debuggee == null) {
156                 log3("ERROR: no debuggee launched");
157                 return FAILED;
158             }
159             log2("debuggee launched");
160         } catch ( Exception e ) {
161             log3("ERROR: Exception : " + e);
162             log2("       test cancelled");
163             return FAILED;
164         }
165 
166         debuggee.redirectOutput(logHandler);
167 
168         vm = debuggee.VM();
169 
170         eventQueue = vm.eventQueue();
171         if (eventQueue == null) {
172             log3("ERROR: eventQueue == null : TEST ABORTED");
173             vm.exit(PASS_BASE);
174             return FAILED;
175         }
176 
177         log2("invocation of the method runTest()");
178         switch (runTest()) {
179 
180             case 0 :  log2("test phase has finished normally");
181                       log2("   waiting for the debuggee to finish ...");
182                       debuggee.waitFor();
183 
184                       log2("......getting the debuggee's exit status");
185                       int status = debuggee.getStatus();
186                       if (status != PASS_BASE) {
187                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
188                               status + " != PASS_BASE");
189                           testExitCode = FAILED;
190                       } else {
191                           log2("......debuggee returned expected exit status: " +
192                               status + " == PASS_BASE");
193                       }
194                       break;
195 
196             default : log3("ERROR: runTest() returned unexpected value");
197 
198             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
199                       log2("......forcing: vm.exit();");
200                       testExitCode = FAILED;
201                       try {
202                           vm.exit(PASS_BASE);
203                       } catch ( Exception e ) {
204                           log3("ERROR: Exception : e");
205                       }
206                       break;
207 
208             case 2 :  log3("test cancelled due to VMDisconnectedException");
209                       log2("......trying: vm.process().destroy();");
210                       testExitCode = FAILED;
211                       try {
212                           Process vmProcess = vm.process();
213                           if (vmProcess != null) {
214                               vmProcess.destroy();
215                           }
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : e");
218                       }
219                       break;
220             }
221 
222         return testExitCode;
223     }

226    /*
227     * Return value: 0 - normal end of the test
228     *               1 - ubnormal end of the test
229     *               2 - VMDisconnectedException while test phase
230     */
231 
232     private int runTest() {
233 
234         try {
235             testRun();
236 
237             log2("waiting for VMDeathEvent");
238             getEventSet();
239             if (eventIterator.nextEvent() instanceof VMDeathEvent)
240                 return 0;
241 
242             log3("ERROR: last event is not the VMDeathEvent");
243             return 1;
244         } catch ( VMDisconnectedException e ) {
245             log3("ERROR: VMDisconnectedException : " + e);
246             return 2;
247         } catch ( Exception e ) {
248             log3("ERROR: Exception : " + e);
249             return 1;
250         }
251 
252     }

254     private void testRun()
255                  throws JDITestRuntimeException, Exception {
256 
257         eventRManager = vm.eventRequestManager();
258 
259         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
260         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
261         cpRequest.addClassFilter(debuggeeName);
262 
263         cpRequest.enable();
264         vm.resume();
265         getEventSet();
266         cpRequest.disable();
267 
268         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
269         debuggeeClass = event.referenceType();
270 
271         if (!debuggeeClass.name().equals(debuggeeName))
272            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
273 
274         log2("      received: ClassPrepareEvent for debuggeeClass");
275 
276         String bPointMethod = "methodForCommunication";
277         String lineForComm  = "lineForComm";
278         BreakpointRequest bpRequest;
279 
280         try {
281             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
282                                           debuggeeClass,
283                                           bPointMethod, lineForComm, "zero");
284         } catch ( Exception e ) {
285             throw e;
286         }
287         bpRequest.enable();
288 
289     //------------------------------------------------------  testing section
290 
291         log1("     TESTING BEGINS");
292 
293         for (int i = 0; ; i++) {
294 
295             vm.resume();
296             breakpointForCommunication();
297 
298             int instruction = ((IntegerValue)
299                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
300 
301             if (instruction == 0) {
302                 vm.resume();
303                 break;
304             }
305 
306             log1("  new check: # " + i);
307 
308             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
309 
310             switch (i) {
311                 case 0:
312                      checkClassLoader (testedClassName0, false);
313                      break;
314 
315                 case 1:
316                      checkClassLoader (testedClassName1, true);
317                      break;
318 
319                  default :
320                      throw new JDITestRuntimeException("Wrong test case :" + i);
321              }
322             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
323         }
324         log1("    TESTING ENDS");
325         return;
326     }

328     private void checkClassLoader (String testedClassName, boolean nullExpected) {
329          log2("      name of tested class : " + testedClassName);
330          List classes = vm.classesByName(testedClassName);
331          if (classes.size() != 1) {
332              log3("ERROR: classes.size() != 1 : " + classes.size());
333              testExitCode = FAILED;
334              return;
335          }
336          ReferenceType testedClass = (ReferenceType) classes.get(0);
337          if (testedClass == null) {
338              log3("ERROR: testedClass == null");
339              testExitCode = FAILED;
340          }
341          if (nullExpected) {
342              log2("      checking: testedClass.classLoader(); expected result is null");
343              if (testedClass.classLoader() != null) {
344                  log3("ERROR: method classLoader() returns not null for " + testedClass.name());
345                  testExitCode = FAILED;
346              }
347          } else {
348              log2("      checking: testedClass.classLoader(); expected result is not null");
349              if (testedClass.classLoader() == null) {
350                  log3("ERROR: method classLoader() returns null for " + testedClass.name());
351                  testExitCode = FAILED;
352              }
353          }
354     }

356    /*
357     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
358     *                                             String, String, String)
359     *
360     * It sets up a breakpoint at given line number within a given method in a given class
361     * for a given thread.
362     *
363     * Return value: BreakpointRequest object  in case of success
364     *
365     * JDITestRuntimeException   in case of an Exception thrown within the method
366     */
367 
368     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
369                                                   ReferenceType testedClass,
370                                                   String methodName,
371                                                   String bpLine,
372                                                   String property)
373             throws JDITestRuntimeException {
374 
375         log2("......setting up a breakpoint:");
376         log2("       thread: " + thread + "; class: " + testedClass +
377                         "; method: " + methodName + "; line: " + bpLine);
378 
379         List              alllineLocations = null;
380         Location          lineLocation     = null;
381         BreakpointRequest breakpRequest    = null;
382 
383         try {
384             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
385 
386             alllineLocations = method.allLineLocations();
387 
388             int n =
389                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
390             if (n > alllineLocations.size()) {
391                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
392             } else {
393                 lineLocation = (Location) alllineLocations.get(n);
394                 try {
395                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
396                     breakpRequest.putProperty("number", property);
397                     breakpRequest.addThreadFilter(thread);
398                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
399                 } catch ( Exception e1 ) {
400                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
401                     breakpRequest    = null;
402                 }
403             }
404         } catch ( Exception e2 ) {
405             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
406             breakpRequest    = null;
407         }
408 
409         if (breakpRequest == null) {
410             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
411             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
412         }
413 
414         log2("      a breakpoint has been set up");
415         return breakpRequest;
416     }

419     private void getEventSet()
420                  throws JDITestRuntimeException {
421         try {
422 //            log2("       eventSet = eventQueue.remove(waitTime);");
423             eventSet = eventQueue.remove(waitTime);
424             if (eventSet == null) {
425                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
426             }
427 //            log2("       eventIterator = eventSet.eventIterator;");
428             eventIterator = eventSet.eventIterator();
429         } catch ( Exception e ) {
430             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
431         }
432     }

435     private void breakpointForCommunication()
436                  throws JDITestRuntimeException {
437 
438         log2("breakpointForCommunication");
439         getEventSet();
440 
441         if (eventIterator.nextEvent() instanceof BreakpointEvent)
442             return;
443 
444         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
445     }
