427     private ThreadReference threadByName(String name)
428                  throws JDITestRuntimeException {
429 
430         List         all = vm.allThreads();
431         ListIterator li  = all.listIterator();
432 
433         for (; li.hasNext(); ) {
434             ThreadReference thread = (ThreadReference) li.next();
435             if (thread.name().equals(name))
436                 return thread;
437         }
438         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
439     }
