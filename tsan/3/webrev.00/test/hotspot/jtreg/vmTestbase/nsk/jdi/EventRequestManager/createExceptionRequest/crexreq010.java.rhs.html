 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createExceptionRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *  - If neither caught exception nor uncaught exception is selected,   <BR>
 47  *  that is, both notifyCaught and notifyUncaught arguments are false,  <BR>
 48  *  returned value is null.                                             <BR>
 49  * Note. In condition of no Exception for such testcase is defined in   <BR>
 50  * the spec,   null is the only reasonable returned value.              <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares a testcase and invokes                       <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the check required.                          <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

153     private int runThis (String argv[], PrintStream out) {
154 
155         argsHandler     = new ArgumentHandler(argv);
156         logHandler      = new Log(out, argsHandler);
157         Binder binder   = new Binder(argsHandler, logHandler);
158 
159         waitTime        = argsHandler.getWaitTime() * 60000;
160 
161         try {
162             log2("launching a debuggee :");
163             log2("       " + debuggeeName);
164             if (argsHandler.verbose()) {
165                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
166             } else {
167                 debuggee = binder.bindToDebugee(debuggeeName);
168             }
169             if (debuggee == null) {
170                 log3("ERROR: no debuggee launched");
171                 return FAILED;
172             }
173             log2("debuggee launched");
174         } catch ( Exception e ) {
175             log3("ERROR: Exception : " + e);
176             log2("       test cancelled");
177             return FAILED;
178         }
179 
180         debuggee.redirectOutput(logHandler);
181 
182         vm = debuggee.VM();
183 
184         eventQueue = vm.eventQueue();
185         if (eventQueue == null) {
186             log3("ERROR: eventQueue == null : TEST ABORTED");
187             vm.exit(PASS_BASE);
188             return FAILED;
189         }
190 
191         log2("invocation of the method runTest()");
192         switch (runTest()) {
193 
194             case 0 :  log2("test phase has finished normally");
195                       log2("   waiting for the debuggee to finish ...");
196                       debuggee.waitFor();
197 
198                       log2("......getting the debuggee's exit status");
199                       int status = debuggee.getStatus();
200                       if (status != PASS_BASE) {
201                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
202                               status + " != PASS_BASE");
203                           testExitCode = FAILED;
204                       } else {
205                           log2("......debuggee returned expected exit status: " +
206                               status + " == PASS_BASE");
207                       }
208                       break;
209 
210             default : log3("ERROR: runTest() returned unexpected value");
211 
212             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
213                       log2("......forcing: vm.exit();");
214                       testExitCode = FAILED;
215                       try {
216                           vm.exit(PASS_BASE);
217                       } catch ( Exception e ) {
218                           log3("ERROR: Exception : " + e);
219                       }
220                       break;
221 
222             case 2 :  log3("test cancelled due to VMDisconnectedException");
223                       log2("......trying: vm.process().destroy();");
224                       testExitCode = FAILED;
225                       try {
226                           Process vmProcess = vm.process();
227                           if (vmProcess != null) {
228                               vmProcess.destroy();
229                           }
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234             }
235 
236         return testExitCode;
237     }

240    /*
241     * Return value: 0 - normal end of the test
242     *               1 - ubnormal end of the test
243     *               2 - VMDisconnectedException while test phase
244     */

246     private int runTest() {
247 
248         try {
249             testRun();
250 
251             log2("waiting for VMDeathEvent");
252             getEventSet();
253             if (eventIterator.nextEvent() instanceof VMDeathEvent)
254                 return 0;
255 
256             log3("ERROR: last event is not the VMDeathEvent");
257             return 1;
258         } catch ( VMDisconnectedException e ) {
259             log3("ERROR: VMDisconnectedException : " + e);
260             return 2;
261         } catch ( Exception e ) {
262             log3("ERROR: Exception : " + e);
263             return 1;
264         }
265 
266     }

268     private void testRun()
269                  throws JDITestRuntimeException, Exception {
270 
271         eventRManager = vm.eventRequestManager();
272 
273         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
274         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
275         cpRequest.addClassFilter(debuggeeName);
276 
277         cpRequest.enable();
278         vm.resume();
279         getEventSet();
280         cpRequest.disable();
281 
282         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
283         debuggeeClass = event.referenceType();
284 
285         if (!debuggeeClass.name().equals(debuggeeName))
286            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292 
293         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
294 
295         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
296                                              debuggeeClass,
297                                             bPointMethod, lineForComm, "zero");
298         bpRequest.enable();
299 
300     //------------------------------------------------------  testing section
301 
302         String fieldName = "testField";
303 
304         ExceptionRequest exRequest = null;
305 
306 
307         log1("     TESTING BEGINS");
308 
309         for (int i = 0; ; i++) {
310 
311             vm.resume();
312             breakpointForCommunication();
313 
314             int instruction = ((IntegerValue)
315                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
316 
317             if (instruction == 0) {
318                 vm.resume();
319                 break;
320             }
321 
322             log1(":::::: case: # " + i);
323 
324             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
325 
326             ReferenceType refType = (ReferenceType)
327                                     debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName)).type();
328 
329 
330             try {
331                 log2("......creating: exRequest = eventRManager.createExceptionRequest(refType, false, false);");
332                 exRequest = eventRManager.createExceptionRequest(refType, false, false);
333                 log2("      check on if returned value is null");
334                 if (exRequest == null) {
335                     log3("ERROR: returned value is null!");
336                     testExitCode = FAILED;
337                 }
338             } catch (Exception e) {
339                 log3("ERROR: exception caught: " + e);
340                 testExitCode = FAILED;
341             }
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
344         }
345         log1("    TESTING ENDS");
346         return;
347     }

349    /*
350     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
351     *                                             String, String, String)
352     *
353     * It sets up a breakpoint at given line number within a given method in a given class
354     * for a given thread.
355     *
356     * Return value: BreakpointRequest object  in case of success
357     *
358     * JDITestRuntimeException   in case of an Exception thrown within the method
359     */
360 
361     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
362                                                   ReferenceType testedClass,
363                                                   String methodName,
364                                                   String bpLine,
365                                                   String property)
366             throws JDITestRuntimeException {
367 
368         log2("......setting up a breakpoint:");
369         log2("       thread: " + thread + "; class: " + testedClass +
370                         "; method: " + methodName + "; line: " + bpLine);
371 
372         List              alllineLocations = null;
373         Location          lineLocation     = null;
374         BreakpointRequest breakpRequest    = null;
375 
376         try {
377             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
378 
379             alllineLocations = method.allLineLocations();
380 
381             int n =
382                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
383             if (n > alllineLocations.size()) {
384                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
385             } else {
386                 lineLocation = (Location) alllineLocations.get(n);
387                 try {
388                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
389                     breakpRequest.putProperty("number", property);
390                     breakpRequest.addThreadFilter(thread);
391                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
392                 } catch ( Exception e1 ) {
393                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
394                     breakpRequest    = null;
395                 }
396             }
397         } catch ( Exception e2 ) {
398             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
399             breakpRequest    = null;
400         }
401 
402         if (breakpRequest == null) {
403             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
404             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
405         }
406 
407         log2("      a breakpoint has been set up");
408         return breakpRequest;
409     }

412     private void getEventSet()
413                  throws JDITestRuntimeException {
414         try {
415 //            log2("       eventSet = eventQueue.remove(waitTime);");
416             eventSet = eventQueue.remove(waitTime);
417             if (eventSet == null) {
418                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
419             }
420 //            log2("       eventIterator = eventSet.eventIterator;");
421             eventIterator = eventSet.eventIterator();
422         } catch ( Exception e ) {
423             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
424         }
425     }

428     private void breakpointForCommunication()
429                  throws JDITestRuntimeException {
430 
431         log2("breakpointForCommunication");
432         getEventSet();
433 
434         if (eventIterator.nextEvent() instanceof BreakpointEvent)
435             return;
436 
437         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
438     }
