 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.threadDeathRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled thread death requests.        <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

151     private int runThis (String argv[], PrintStream out) {
152 
153         argsHandler     = new ArgumentHandler(argv);
154         logHandler      = new Log(out, argsHandler);
155         Binder binder   = new Binder(argsHandler, logHandler);
156 
157         waitTime        = argsHandler.getWaitTime() * 60000;
158 
159         try {
160             log2("launching a debuggee :");
161             log2("       " + debuggeeName);
162             if (argsHandler.verbose()) {
163                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
164             } else {
165                 debuggee = binder.bindToDebugee(debuggeeName);
166             }
167             if (debuggee == null) {
168                 log3("ERROR: no debuggee launched");
169                 return FAILED;
170             }
171             log2("debuggee launched");
172         } catch ( Exception e ) {
173             log3("ERROR: Exception : " + e);
174             log2("       test cancelled");
175             return FAILED;
176         }
177 
178         debuggee.redirectOutput(logHandler);
179 
180         vm = debuggee.VM();
181 
182         eventQueue = vm.eventQueue();
183         if (eventQueue == null) {
184             log3("ERROR: eventQueue == null : TEST ABORTED");
185             vm.exit(PASS_BASE);
186             return FAILED;
187         }
188 
189         log2("invocation of the method runTest()");
190         switch (runTest()) {
191 
192             case 0 :  log2("test phase has finished normally");
193                       log2("   waiting for the debuggee to finish ...");
194                       debuggee.waitFor();
195 
196                       log2("......getting the debuggee's exit status");
197                       int status = debuggee.getStatus();
198                       if (status != PASS_BASE) {
199                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
200                               status + " != PASS_BASE");
201                           testExitCode = FAILED;
202                       } else {
203                           log2("......debuggee returned expected exit status: " +
204                               status + " == PASS_BASE");
205                       }
206                       break;
207 
208             default : log3("ERROR: runTest() returned unexpected value");
209 
210             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
211                       log2("......forcing: vm.exit();");
212                       testExitCode = FAILED;
213                       try {
214                           vm.exit(PASS_BASE);
215                       } catch ( Exception e ) {
216                           log3("ERROR: Exception : " + e);
217                       }
218                       break;
219 
220             case 2 :  log3("test cancelled due to VMDisconnectedException");
221                       log2("......trying: vm.process().destroy();");
222                       testExitCode = FAILED;
223                       try {
224                           Process vmProcess = vm.process();
225                           if (vmProcess != null) {
226                               vmProcess.destroy();
227                           }
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : " + e);
230                       }
231                       break;
232             }
233 
234         return testExitCode;
235     }

240     *               1 - ubnormal end of the test

244     private int runTest() {
245 
246         try {
247             testRun();
248 
249             log2("waiting for VMDeathEvent");
250             getEventSet();
251             if (eventIterator.nextEvent() instanceof VMDeathEvent)
252                 return 0;
253 
254             log3("ERROR: last event is not the VMDeathEvent");
255             return 1;
256         } catch ( VMDisconnectedException e ) {
257             log3("ERROR: VMDisconnectedException : " + e);
258             return 2;
259         } catch ( Exception e ) {
260             log3("ERROR: Exception : " + e);
261             return 1;
262         }
263 
264     }

266     private void testRun()
267                  throws JDITestRuntimeException, Exception {
268 
269         eventRManager = vm.eventRequestManager();
270 
271         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
272         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
273         cpRequest.addClassFilter(debuggeeName);
274 
275         cpRequest.enable();
276         vm.resume();
277         getEventSet();
278         cpRequest.disable();
279 
280         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
281         debuggeeClass = event.referenceType();
282 
283         if (!debuggeeClass.name().equals(debuggeeName))
284            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
285 
286         log2("      received: ClassPrepareEvent for debuggeeClass");
287 
288         String bPointMethod = "methodForCommunication";
289         String lineForComm  = "lineForComm";
290 
291         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
292 
293         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
294                                              debuggeeClass,
295                                             bPointMethod, lineForComm, "zero");
296         bpRequest.enable();
297 
298     //------------------------------------------------------  testing section
299 
300         List         requests = null;
301         ListIterator li       = null;
302 
303         ThreadDeathRequest request = null;
304         ThreadDeathRequest tdRequests[] = { null, null, null, null, null,
305                                             null, null, null, null, null };
306         int listSize;
307         int flag;
308 
309 
310         log1("     TESTING BEGINS");
311 
312         for (int i = 0; ; i++) {
313 
314             vm.resume();
315             breakpointForCommunication();
316 
317             int instruction = ((IntegerValue)
318                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
319 
320             if (instruction == 0) {
321                 vm.resume();
322                 break;
323             }
324 
325             log1(":::::: case: # " + i);
326 
327             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
328 
329             log2("......creating ThreadDeathRequests");
330             for (int i1 = 0; i1 < tdRequests.length; i1++) {
331                 tdRequests[i1] = eventRManager.createThreadDeathRequest();
332                 tdRequests[i1].putProperty("number", String.valueOf(i1));
333 
334                 log2("......checking up on returned List after creating new ThreadDeathRequest");
335                 requests = eventRManager.threadDeathRequests();
336                 listSize = requests.size();
337                 if ( listSize != (i1 + 1) ) {
338                     testExitCode = FAILED;
339                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
340                 }
341                 flag = 0;
342                 li = requests.listIterator();
343                 while (li.hasNext()) {
344                     request = (ThreadDeathRequest) li.next();
345                     if ( !request.isEnabled() ) {
346                         flag++;
347                         if (flag > 1) {
348                             testExitCode = FAILED;
349                             log3("ERROR: # of disabled requests > 1 : " + flag);
350                         }
351                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
352                             testExitCode = FAILED;
353                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
354                         }
355                     } else {
356                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
357                             testExitCode = FAILED;
358                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
359                         }
360                     }
361                 }
362 
363                 log2("      enabling created ThreadDeathRequest");
364                 tdRequests[i1].enable();
365                 requests = eventRManager.threadDeathRequests();
366                 listSize = requests.size();
367                 if ( listSize != (i1 + 1) ) {
368                     testExitCode = FAILED;
369                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
370                 }
371 
372                 li = requests.listIterator();
373                 while (li.hasNext()) {
374                     request = (ThreadDeathRequest) li.next();
375                     if ( !request.isEnabled() ) {
376                         testExitCode = FAILED;
377                         log3("ERROR: returned List contains disabled request : " + request);
378                     }
379                 }
380 
381                 log2("      removing item from the List; UnsupportedOperationException is expected");
382                 try {
383                     requests.remove(i1);
384                     testExitCode = FAILED;
385                     log3("ERROR: NO exception");
386                 } catch ( UnsupportedOperationException e ) {
387                     log2("        UnsupportedOperationException ");
388                 }
389             }
390 
391             log2("......deleting ThreadDeathRequests");
392             for (int i2 = tdRequests.length -1; i2 >= 0; i2--) {
393                 eventRManager.deleteEventRequest(tdRequests[i2]);
394                 requests = eventRManager.threadDeathRequests();
395                 listSize = requests.size();
396                 if ( listSize != i2 ) {
397                     testExitCode = FAILED;
398                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " + i2);
399                 }
400                 log2("      removing item from the List; UnsupportedOperationException is expected");
401                 try {
402                     requests.remove(i2);
403                     testExitCode = FAILED;
404                     log3("ERROR: NO exception");
405                 } catch ( UnsupportedOperationException e ) {
406                     log2("        UnsupportedOperationException ");
407                 }
408             }
409 
410             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
411         }
412         log1("    TESTING ENDS");
413         return;
414     }

416    /*
417     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
418     *                                             String, String, String)
419     *
420     * It sets up a breakpoint at given line number within a given method in a given class
421     * for a given thread.
422     *
423     * Return value: BreakpointRequest object  in case of success
424     *
425     * JDITestRuntimeException   in case of an Exception thrown within the method
426     */
427 
428     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
429                                                   ReferenceType testedClass,
430                                                   String methodName,
431                                                   String bpLine,
432                                                   String property)
433             throws JDITestRuntimeException {
434 
435         log2("......setting up a breakpoint:");
436         log2("       thread: " + thread + "; class: " + testedClass +
437                         "; method: " + methodName + "; line: " + bpLine);
438 
439         List              alllineLocations = null;
440         Location          lineLocation     = null;
441         BreakpointRequest breakpRequest    = null;
442 
443         try {
444             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
445 
446             alllineLocations = method.allLineLocations();
447 
448             int n =
449                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
450             if (n > alllineLocations.size()) {
451                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
452             } else {
453                 lineLocation = (Location) alllineLocations.get(n);
454                 try {
455                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
456                     breakpRequest.putProperty("number", property);
457                     breakpRequest.addThreadFilter(thread);
458                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
459                 } catch ( Exception e1 ) {
460                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
461                     breakpRequest    = null;
462                 }
463             }
464         } catch ( Exception e2 ) {
465             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
466             breakpRequest    = null;
467         }
468 
469         if (breakpRequest == null) {
470             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
471             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
472         }
473 
474         log2("      a breakpoint has been set up");
475         return breakpRequest;
476     }

479     private void getEventSet()
480                  throws JDITestRuntimeException {
481         try {
482 //            log2("       eventSet = eventQueue.remove(waitTime);");
483             eventSet = eventQueue.remove(waitTime);
484             if (eventSet == null) {
485                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
486             }
487 //            log2("       eventIterator = eventSet.eventIterator;");
488             eventIterator = eventSet.eventIterator();
489         } catch ( Exception e ) {
490             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
491         }
492     }

495     private void breakpointForCommunication()
496                  throws JDITestRuntimeException {
497 
498         log2("breakpointForCommunication");
499         getEventSet();
500 
501         if (eventIterator.nextEvent() instanceof BreakpointEvent)
502             return;
503 
504         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
505     }
