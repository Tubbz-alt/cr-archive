 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a MethodExitEvent set,              <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a MethodExitRequest, resumes          <BR>
 70  *   the debuggee, and waits for the MethodExitEvent.                   <BR>
 71  * - The debuggee invokes the special method to be resulting in the event.<BR>
 72  * - Upon getting new event, the debugger                               <BR>
 73  *   performs the check corresponding to the event.                     <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * In third phase, at the end,                                          <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  */
 82

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : e");
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
265                 log3("ERROR: last event is not the VMDeathEvent");
266                 return 1;
267             }
268 
269             log2("waiting for VMDisconnectEvent");
270             getEventSet();
271             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
272                 log3("ERROR: last event is not the VMDisconnectEvent");
273                 return 1;
274             }
275 
276             return 0;
277 
278         } catch ( VMDisconnectedException e ) {
279             log3("ERROR: VMDisconnectedException : " + e);
280             return 2;
281         } catch ( Exception e ) {
282             log3("ERROR: Exception : " + e);
283             return 1;
284         }
285

288     private void testRun()
289                  throws JDITestRuntimeException, Exception {
290 
291         eventRManager = vm.eventRequestManager();
292 
293         log2("......getting ClassPrepareEvent for debuggee's class");
294         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
295         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
296         cpRequest.addClassFilter(debuggeeName);
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307         log2("      received: ClassPrepareEvent for debuggeeClass");
308 
309         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
310 
311         String            bPointMethod = "methodForCommunication";
312         String            lineForComm  = "lineForComm";
313         BreakpointRequest bpRequest;
314         ThreadReference   mainThread = threadByName("main");
315         bpRequest = settingBreakpoint(mainThread,
316                                       debuggeeClass,
317                                       bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320         vm.resume();
321 
322     //------------------------------------------------------  testing section
323 
324         log1("     TESTING BEGINS");
325 
326         EventRequest eventRequest1 = null;
327         EventRequest eventRequest2 = null;
328         EventRequest eventRequest3 = null;
329 
330         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
331         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
332         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
333         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
334 
335         int policyExpected[] = {    SUSPEND_NONE,
336                                     SUSPEND_THREAD,
337                                     SUSPEND_ALL,
338                                     SUSPEND_THREAD,
339                                     SUSPEND_ALL,
340                                     SUSPEND_ALL,
341                                     SUSPEND_ALL      };
342         int policy = 0;
343 
344         ReferenceType testClassReference = null;
345 
346         String fieldName = "var1";
347 
348 
349         for (int i = 0; ; i++) {
350 
351 
352             breakpointForCommunication();
353 
354             int instruction = ((IntegerValue)
355                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
356             if (instruction == 0) {
357                 vm.resume();
358                 break;
359             }
360 
361 
362             log1(":::::: case: # " + i);
363 
364             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
365 
366             switch (i) {
367 
368               case 0:
369                       testClassReference =
370                            (ReferenceType) vm.classesByName(testedClassName).get(0);
371 
372                       eventRequest1 = settingMethodExitRequest (mainThread,
373                                              testClassReference,
374                                              SUSPEND_NONE, "MethodExitRequest1");
375                       eventRequest1.enable();
376                       break;
377 
378               case 1:
379                       eventRequest2 = settingMethodExitRequest (mainThread,
380                                              testClassReference,
381                                              SUSPEND_THREAD, "MethodExitRequest2");
382                       eventRequest2.enable();
383                       break;
384 
385               case 2:
386                       eventRequest3 = settingMethodExitRequest (mainThread,
387                                              testClassReference,
388                                              SUSPEND_ALL, "MethodExitRequest3");
389                       eventRequest3.enable();
390                       break;
391 
392               case 3:
393                       eventRequest1 = settingMethodExitRequest (mainThread,
394                                              testClassReference,
395                                              SUSPEND_NONE, "MethodExitRequest4");
396                       eventRequest1.enable();
397                       eventRequest2 = settingMethodExitRequest (mainThread,
398                                              testClassReference,
399                                              SUSPEND_THREAD, "MethodExitRequest5");
400                       eventRequest2.enable();
401                       break;
402 
403               case 4:
404                       eventRequest1 = settingMethodExitRequest (mainThread,
405                                              testClassReference,
406                                              SUSPEND_NONE, "MethodExitRequest6");
407                       eventRequest1.enable();
408                       eventRequest3 = settingMethodExitRequest (mainThread,
409                                              testClassReference,
410                                              SUSPEND_ALL, "MethodExitRequest7");
411                       eventRequest3.enable();
412                       break;
413 
414               case 5:
415                       eventRequest2 = settingMethodExitRequest (mainThread,
416                                              testClassReference,
417                                              SUSPEND_THREAD, "MethodExitRequest8");
418                       eventRequest2.enable();
419                       eventRequest3 = settingMethodExitRequest (mainThread,
420                                              testClassReference,
421                                              SUSPEND_ALL, "MethodExitRequest9");
422                       eventRequest3.enable();
423                       break;
424 
425               case 6:
426                       eventRequest1 = settingMethodExitRequest (mainThread,
427                                              testClassReference,
428                                              SUSPEND_NONE, "MethodExitRequest10");
429                       eventRequest1.enable();
430                       eventRequest2 = settingMethodExitRequest (mainThread,
431                                              testClassReference,
432                                              SUSPEND_THREAD, "MethodExitRequest11");
433                       eventRequest2.enable();
434                       eventRequest3 = settingMethodExitRequest (mainThread,
435                                              testClassReference,
436                                              SUSPEND_ALL, "MethodExitRequest12");
437                       eventRequest3.enable();
438                       break;
439 
440 
441               default:
442                       throw new JDITestRuntimeException("** default case 2 **");
443             }
444 
445             log2("......waiting for new MethodExitEvent : " + i);
446             mainThread.resume();
447             getEventSet();
448 
449             Event newEvent = eventIterator.nextEvent();
450             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
451             if ( !( newEvent instanceof MethodExitEvent)) {
452                  log3("ERROR: new event is not MethodExitEvent");
453                  testExitCode = FAILED;
454             } else {
455                 log2("       Event is instanceof MethodExitEvent");
456                 policy = eventSet.suspendPolicy();
457                 if (policy != policyExpected[i]) {
458                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
459                     log3("       eventSet.suspendPolicy() == " + policy);
460                     log3("       policyExpected           == " + policyExpected[i]);
461                     testExitCode = FAILED;
462                 }
463             }
464 
465             switch (i) {
466               case 0: eventRequest1.disable();                          break;
467               case 1: eventRequest2.disable();                          break;
468               case 2: eventRequest3.disable();                          break;
469               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
470               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
471               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
472               case 6: eventRequest1.disable(); eventRequest2.disable();
473                       eventRequest3.disable();                          break;
474            }
475 
476             switch (policy) {
477               case SUSPEND_NONE   :                      break;
478               case SUSPEND_THREAD : mainThread.resume(); break;
479               case SUSPEND_ALL    : vm.resume();         break;
480               default: throw new JDITestRuntimeException("** default case 3 **");
481             }
482 
483             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
484         }
485         log1("    TESTING ENDS");
486         return;
487     }

489     private ThreadReference threadByName(String name)
490                  throws JDITestRuntimeException {
491 
492         List         all = vm.allThreads();
493         ListIterator li  = all.listIterator();
494 
495         for (; li.hasNext(); ) {
496             ThreadReference thread = (ThreadReference) li.next();
497             if (thread.name().equals(name))
498                 return thread;
499         }
500         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
501     }

503    /*
504     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
505     *                                             String, String, String)
506     *
507     * It sets up a breakpoint at given line number within a given method in a given class
508     * for a given thread.
509     *
510     * Return value: BreakpointRequest object  in case of success
511     *
512     * JDITestRuntimeException   in case of an Exception thrown within the method
513     */
514 
515     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
516                                                   ReferenceType testedClass,
517                                                   String methodName,
518                                                   String bpLine,
519                                                   String property)
520             throws JDITestRuntimeException {
521 
522         log2("......setting up a breakpoint:");
523         log2("       thread: " + thread + "; class: " + testedClass +
524                         "; method: " + methodName + "; line: " + bpLine);
525 
526         List              alllineLocations = null;
527         Location          lineLocation     = null;
528         BreakpointRequest breakpRequest    = null;
529 
530         try {
531             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
532 
533             alllineLocations = method.allLineLocations();
534 
535             int n =
536                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
537             if (n > alllineLocations.size()) {
538                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
539             } else {
540                 lineLocation = (Location) alllineLocations.get(n);
541                 try {
542                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
543                     breakpRequest.putProperty("number", property);
544                     breakpRequest.addThreadFilter(thread);
545                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
546                 } catch ( Exception e1 ) {
547                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
548                     breakpRequest    = null;
549                 }
550             }
551         } catch ( Exception e2 ) {
552             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
553             breakpRequest    = null;
554         }
555 
556         if (breakpRequest == null) {
557             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
558             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
559         }
560 
561         log2("      a breakpoint has been set up");
562         return breakpRequest;
563     }

566     private void getEventSet()
567                  throws JDITestRuntimeException {
568         try {
569 //            log2("       eventSet = eventQueue.remove(waitTime);");
570             eventSet = eventQueue.remove(waitTime);
571             if (eventSet == null) {
572                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
573             }
574 //            log2("       eventIterator = eventSet.eventIterator;");
575             eventIterator = eventSet.eventIterator();
576         } catch ( Exception e ) {
577             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
578         }
579     }

583                  throws JDITestRuntimeException {
584 
585         log2("breakpointForCommunication");
586         getEventSet();
587 
588         if (eventIterator.nextEvent() instanceof BreakpointEvent)
589             return;
590 
591         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
592     }

596     private MethodExitRequest settingMethodExitRequest ( ThreadReference thread,
597                                                          ReferenceType   testedClass,
598                                                          int             suspendPolicy,
599                                                          String          property        )
600             throws JDITestRuntimeException {
601         try {
602             log2("......setting up MethodExitRequest:");
603             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
604 
605             MethodExitRequest
606             mexr = eventRManager.createMethodExitRequest();
607             mexr.putProperty("number", property);
608             mexr.addThreadFilter(thread);
609             mexr.addClassFilter(testedClass);
610             mexr.setSuspendPolicy(suspendPolicy);
611 
612         log2("      MethodExitRequest has been set up");
613         return mexr;
614         } catch ( Exception e ) {
615             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
616             log3("       MethodExitRequest HAS NOT BEEN SET UP");
617             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
618         }
