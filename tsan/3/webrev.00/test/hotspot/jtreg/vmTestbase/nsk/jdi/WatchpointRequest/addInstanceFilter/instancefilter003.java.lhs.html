 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * This request is AccessWatchpointRequest.                     <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new thread, thread1, and invokes              <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates an AccessWatchpointRequest and enables it,               <BR>
 73  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 74  *     to catch InvalidRequestStateException,                           <BR>
 75  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 76  *     expecting to catch no InvalidRequestStateException,              <BR>
 77  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 78  *     once again expecting to catch InvalidRequestStateException.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test, the debuggee changes         <BR>
 81  * the value of the "instruction" which the debugger and debuggee       <BR>
 82  * use to inform each other of needed actions,  and both end.           <BR>
 83  * <BR>
 84  */

167     private int runThis (String argv[], PrintStream out) {
168 
169         argsHandler     = new ArgumentHandler(argv);
170         logHandler      = new Log(out, argsHandler);
171         Binder binder   = new Binder(argsHandler, logHandler);
172 
173         waitTime        = argsHandler.getWaitTime() * 60000;
174 
175         try {
176             log2("launching a debuggee :");
177             log2("       " + debuggeeName);
178             if (argsHandler.verbose()) {
179                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
180             } else {
181                 debuggee = binder.bindToDebugee(debuggeeName);
182             }
183             if (debuggee == null) {
184                 log3("ERROR: no debuggee launched");
185                 return FAILED;
186             }
187             log2("debuggee launched");
188         } catch ( Exception e ) {
189             log3("ERROR: Exception : " + e);
190             log2("       test cancelled");
191             return FAILED;
192         }
193 
194         debuggee.redirectOutput(logHandler);
195 
196         vm = debuggee.VM();
197 
198         eventQueue = vm.eventQueue();
199         if (eventQueue == null) {
200             log3("ERROR: eventQueue == null : TEST ABORTED");
201             vm.exit(PASS_BASE);
202             return FAILED;
203         }
204 
205         log2("invocation of the method runTest()");
206         switch (runTest()) {
207 
208             case 0 :  log2("test phase has finished normally");
209                       log2("   waiting for the debuggee to finish ...");
210                       debuggee.waitFor();
211 
212                       log2("......getting the debuggee's exit status");
213                       int status = debuggee.getStatus();
214                       if (status != PASS_BASE) {
215                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
216                               status + " != PASS_BASE");
217                           testExitCode = FAILED;
218                       } else {
219                           log2("......debuggee returned expected exit status: " +
220                               status + " == PASS_BASE");
221                       }
222                       break;
223 
224             default : log3("ERROR: runTest() returned unexpected value");
225 
226             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
227                       log2("......forcing: vm.exit();");
228                       testExitCode = FAILED;
229                       try {
230                           vm.exit(PASS_BASE);
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235 
236             case 2 :  log3("test cancelled due to VMDisconnectedException");
237                       log2("......trying: vm.process().destroy();");
238                       testExitCode = FAILED;
239                       try {
240                           Process vmProcess = vm.process();
241                           if (vmProcess != null) {
242                               vmProcess.destroy();
243                           }
244                       } catch ( Exception e ) {
245                           log3("ERROR: Exception : " + e);
246                       }
247                       break;
248             }
249 
250         return testExitCode;
251     }

256     *               1 - ubnormal end of the test
257     *               2 - VMDisconnectedException while test phase

260     private int runTest() {
261 
262         try {
263             testRun();
264 
265             log2("waiting for VMDeathEvent");
266             getEventSet();
267             if (eventIterator.nextEvent() instanceof VMDeathEvent)
268                 return 0;
269 
270             log3("ERROR: last event is not the VMDeathEvent");
271             return 1;
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         if ( !vm.canUseInstanceFilters() ) {
286             log2("......vm.canUseInstanceFilters == false :: test cancelled");
287             vm.exit(PASS_BASE);
288             return;
289         }
290 
291 
292         eventRManager = vm.eventRequestManager();
293 
294         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
295         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
296         cpRequest.addClassFilter(debuggeeName);
297 
298         cpRequest.enable();
299         vm.resume();
300         getEventSet();
301         cpRequest.disable();
302 
303         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
304         debuggeeClass = event.referenceType();
305 
306         if (!debuggeeClass.name().equals(debuggeeName))
307            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
308 
309         log2("      received: ClassPrepareEvent for debuggeeClass");
310 
311         String bPointMethod = "methodForCommunication";
312         String lineForComm  = "lineForComm";
313 
314         ThreadReference   mainThread = threadByName("main");
315 
316         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
317                                              debuggeeClass,
318                                             bPointMethod, lineForComm, "zero");
319         bpRequest.enable();
320 
321     //------------------------------------------------------  testing section
322 
323         log1("     TESTING BEGINS");
324 
325         EventRequest eventRequest1 = null;
326 
327         ThreadReference thread1 = null;
328         ThreadReference thread2 = null;
329 
330         String thread1Name = "thread1";
331         String thread2Name = "thread2";
332 
333         String property1 = "AccessWatchpointRequest1";
334 
335         String fieldName = "var1";
336 
337         ReferenceType testClassReference = null;
338 
339         String         arrayName = "objTC";
340         ObjectReference instance = null;
341 
342 
343         for (int i = 0; ; i++) {
344 
345             vm.resume();
346             breakpointForCommunication();
347 
348             int instruction = ((IntegerValue)
349                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
350 
351             if (instruction == 0) {
352                 vm.resume();
353                 break;
354             }
355 
356             log1(":::::: case: # " + i);
357 
358             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
359 
360             switch (i) {
361 
362               case 0:
363                       testClassReference =
364                            (ReferenceType) vm.classesByName(testedClassName).get(0);
365 
366                       thread1 = (ThreadReference) debuggeeClass.getValue(
367                                                   debuggeeClass.fieldByName(thread1Name));
368 
369                       eventRequest1 = setting2AccessWatchpointRequest (thread1,
370                                              testClassReference, fieldName,
371                                              EventRequest.SUSPEND_NONE, property1);
372 
373                       instance = (ObjectReference)
374                                  ((ArrayReference) debuggeeClass.getValue(
375                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
376 
377                       ((WatchpointRequest) eventRequest1).addInstanceFilter(instance);
378 
379                       eventRequest1.enable();
380 
381                      try {
382                          log2("......eventRequest1.addInstanceFilter(instance);");
383                          log2("        InvalidRequestStateException expected");
384                          ((WatchpointRequest) eventRequest1).addInstanceFilter(instance);
385                          log3("ERROR: no InvalidRequestStateException");
386                          testExitCode = FAILED;
387                      } catch ( InvalidRequestStateException e ) {
388                          log2("        InvalidRequestStateException");
389                      }
390 
391                      log2("......eventRequest1.disable();");
392                      eventRequest1.disable();
393                      try {
394                          log2("......eventRequest1.addInstanceFilter(instance);");
395                          log2("        no InvalidRequestStateException expected");
396                          ((WatchpointRequest) eventRequest1).addInstanceFilter(instance);
397                          log2("        no InvalidRequestStateException");
398                      } catch ( InvalidRequestStateException e ) {
399                          log3("ERROR: InvalidRequestStateException");
400                          testExitCode = FAILED;
401                      }
402 
403                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
404                      eventRManager.deleteEventRequest(eventRequest1);
405                      try {
406                          log2("......eventRequest1.addInstanceFilter(instance);");
407                          log2("        InvalidRequestStateException expected");
408                          ((WatchpointRequest) eventRequest1).addInstanceFilter(instance);
409                          log3("ERROR: no InvalidRequestStateException");
410                          testExitCode = FAILED;
411                      } catch ( InvalidRequestStateException e ) {
412                          log2("        InvalidRequestStateException");
413                      }
414                       break;
415 
416               default:
417                       throw new JDITestRuntimeException("** default case 2 **");
418             }
419 
420             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
421         }
422         log1("    TESTING ENDS");
423         return;
424     }

426     private ThreadReference threadByName(String name)
427                  throws JDITestRuntimeException {
428 
429         List         all = vm.allThreads();
430         ListIterator li  = all.listIterator();
431 
432         for (; li.hasNext(); ) {
433             ThreadReference thread = (ThreadReference) li.next();
434             if (thread.name().equals(name))
435                 return thread;
436         }
437         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
438     }

440    /*
441     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
442     *                                             String, String, String)
443     *
444     * It sets up a breakpoint at given line number within a given method in a given class
445     * for a given thread.
446     *
447     * Return value: BreakpointRequest object  in case of success
448     *
449     * JDITestRuntimeException   in case of an Exception thrown within the method
450     */
451 
452     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
453                                                   ReferenceType testedClass,
454                                                   String methodName,
455                                                   String bpLine,
456                                                   String property)
457             throws JDITestRuntimeException {
458 
459         log2("......setting up a breakpoint:");
460         log2("       thread: " + thread + "; class: " + testedClass +
461                         "; method: " + methodName + "; line: " + bpLine);
462 
463         List              alllineLocations = null;
464         Location          lineLocation     = null;
465         BreakpointRequest breakpRequest    = null;
466 
467         try {
468             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
469 
470             alllineLocations = method.allLineLocations();
471 
472             int n =
473                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
474             if (n > alllineLocations.size()) {
475                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
476             } else {
477                 lineLocation = (Location) alllineLocations.get(n);
478                 try {
479                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
480                     breakpRequest.putProperty("number", property);
481                     breakpRequest.addThreadFilter(thread);
482                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
483                 } catch ( Exception e1 ) {
484                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
485                     breakpRequest    = null;
486                 }
487             }
488         } catch ( Exception e2 ) {
489             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
490             breakpRequest    = null;
491         }
492 
493         if (breakpRequest == null) {
494             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
495             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
496         }
497 
498         log2("      a breakpoint has been set up");
499         return breakpRequest;
500     }

503     private void getEventSet()
504                  throws JDITestRuntimeException {
505         try {
506 //            log2("       eventSet = eventQueue.remove(waitTime);");
507             eventSet = eventQueue.remove(waitTime);
508             if (eventSet == null) {
509                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
510             }
511 //            log2("       eventIterator = eventSet.eventIterator;");
512             eventIterator = eventSet.eventIterator();
513         } catch ( Exception e ) {
514             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
515         }
516     }

519     private void breakpointForCommunication()
520                  throws JDITestRuntimeException {
521 
522         log2("breakpointForCommunication");
523         getEventSet();
524 
525         if (eventIterator.nextEvent() instanceof BreakpointEvent)
526             return;
527 
528         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
529     }

533     private AccessWatchpointRequest setting2AccessWatchpointRequest (
534                                                   ThreadReference thread,
535                                                   ReferenceType   testedClass,
536                                                   String          fieldName,
537                                                   int             suspendPolicy,
538                                                   String          property        )
539             throws JDITestRuntimeException {
540         try {
541             log2("......setting up AccessWatchpointRequest:");
542             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
543             Field field = testedClass.fieldByName(fieldName);
544 
545             AccessWatchpointRequest
546             awr = eventRManager.createAccessWatchpointRequest(field);
547             awr.putProperty("number", property);
548 
549             if (thread != null)
550                 awr.addThreadFilter(thread);
551             awr.setSuspendPolicy(suspendPolicy);
552 
553             log2("      AccessWatchpointRequest has been set up");
554             return awr;
555         } catch ( Exception e ) {
556             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
557             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
558             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
559         }
