395     private ThreadReference threadByName(String name)
396                  throws JDITestRuntimeException {
397 
398         List         all = vm.allThreads();
399         ListIterator li  = all.listIterator();
400 
401         for (; li.hasNext(); ) {
402             ThreadReference thread = (ThreadReference) li.next();
403             if (thread.name().equals(name))
404                 return thread;
405         }
406         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
