 38 /**
 39  * The test for the implementation of an object of the type     <BR>
 40  * WatchpointRequest.                                           <BR>
 41  *                                                              <BR>
 42  * The test checks that results of the method                   <BR>
 43  * <code>com.sun.jdi.WatchpointRequest.addThreadFilter()</code> <BR>
 44  * complies with its spec.                                      <BR>
 45  * <BR>
 46  * The test checks up on the following assertion:               <BR>
 47  *    Restricts the events generated by this request            <BR>
 48  *    to those in the given thread.                             <BR>
 49  * The cases to test include three thread's states:             <BR>
 50  * not started, running, dead.                                  <BR>
 51  * The tested Request is AccessWatchpointRequest.               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee creates new thread, thread1, and invokes              <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 74  *   - creates AccessWatchpointRequest1 and                             <BR>
 75  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 76  *   - creates AccessWatchpointRequest2,                                <BR>
 77  *     resumes the debuggee to get the thread1 started, and             <BR>
 78  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 79  *   - creates AccessWatchpointRequest3, waits the thread1 to die, and  <BR>
 80  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 81  * <BR>
 82  * In third phase, at the end of the test,                              <BR>
 83  * the debuggee changes the value of the "instruction"  which           <BR>
 84  * the debugger and debuggee use to inform each other of needed actions,<BR>
 85  * and both end.
 86  * <BR>
 87  */
 88

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : " + e);
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : " + e);
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

253     *               1 - ubnormal end of the test
254     *               2 - VMDisconnectedException while test phase

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if (eventIterator.nextEvent() instanceof VMDeathEvent)
265                 return 0;
266 
267             log3("ERROR: last event is not the VMDeathEvent");
268             return 1;
269         } catch ( VMDisconnectedException e ) {
270             log3("ERROR: VMDisconnectedException : " + e);
271             return 2;
272         } catch ( Exception e ) {
273             log3("ERROR: Exception : " + e);
274             return 1;
275         }

279     private void testRun()
280                  throws JDITestRuntimeException, Exception {
281 
282         if ( !vm.canWatchFieldAccess() ) {
283             log2("......vm.canWatchFieldAccess == false :: test cancelled");
284             vm.exit(PASS_BASE);
285             return;
286         }
287 
288 
289         eventRManager = vm.eventRequestManager();
290 
291         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
292         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
293         cpRequest.addClassFilter(debuggeeName);
294 
295         cpRequest.enable();
296         vm.resume();
297         getEventSet();
298         cpRequest.disable();
299 
300         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
301         debuggeeClass = event.referenceType();
302 
303         if (!debuggeeClass.name().equals(debuggeeName))
304            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
305 
306         log2("      received: ClassPrepareEvent for debuggeeClass");
307 
308         String bPointMethod = "methodForCommunication";
309         String lineForComm  = "lineForComm";
310 
311         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
312 
313         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
314                                              debuggeeClass,
315                                             bPointMethod, lineForComm, "zero");
316         bpRequest.enable();
317 
318     //------------------------------------------------------  testing section
319 
320         log1("     TESTING BEGINS");
321 
322         EventRequest eventRequest1 = null;
323         EventRequest eventRequest2 = null;
324         EventRequest eventRequest3 = null;
325 
326         ThreadReference thread1    = null;
327         String         thread1Name = "thread1";
328 
329         String property1 = "AccessWatchpointRequest1";
330         String property2 = "AccessWatchpointRequest2";
331         String property3 = "AccessWatchpointRequest3";
332 
333         String fieldName = "var1";
334 
335         ReferenceType testClassReference = null;
336 
337         for (int i = 0; ; i++) {
338 
339             vm.resume();
340             breakpointForCommunication();
341 
342             int instruction = ((IntegerValue)
343                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
344 
345             if (instruction == 0) {
346                 vm.resume();
347                 break;
348             }
349 
350             log1(":::::: case: # " + i);
351 
352             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
353 
354             switch (i) {
355 
356               case 0:
357                       testClassReference =
358                            (ReferenceType) vm.classesByName(testedClassName).get(0);
359                       thread1 = (ThreadReference) debuggeeClass.getValue(
360                                                   debuggeeClass.fieldByName(thread1Name));
361 
362                       eventRequest1 = setting2AccessWatchpointRequest (null,
363                                              testClassReference, fieldName,
364                                              EventRequest.SUSPEND_NONE, "AccessWatchpointRequest1");
365                       try {
366                           log2("......eventRequest1.addThreadFilter(thread1);");
367                           log2("        no Exception expected");
368                           ((AccessWatchpointRequest)eventRequest1).addThreadFilter(thread1);
369                       } catch ( Exception e ) {
370                           log3("ERROR: Exception : " + e);
371                           testExitCode = FAILED;
372                       }
373                       break;
374 
375               case 1:
376                       eventRequest2 = setting2AccessWatchpointRequest (null,
377                                              testClassReference, fieldName,
378                                              EventRequest.SUSPEND_NONE, "AccessWatchpointRequest2");
379                       try {
380                           log2("......eventRequest2.addThreadFilter(thread1);");
381                           log2("        no Exception expected");
382                           ((AccessWatchpointRequest)eventRequest2).addThreadFilter(thread1);
383                       } catch ( Exception e ) {
384                           log3("ERROR: Exception : " + e);
385                           testExitCode = FAILED;
386                       }
387                       break;
388 
389               case 2:
390                       eventRequest3 = setting2AccessWatchpointRequest (null,
391                                              testClassReference, fieldName,
392                                              EventRequest.SUSPEND_NONE, "AccessWatchpointRequest3");
393                       try {
394                           log2("......eventRequest3.addThreadFilter(thread1);");
395                           log2("        no Exception expected");
396                           ((AccessWatchpointRequest)eventRequest3).addThreadFilter(thread1);
397                       } catch ( Exception e ) {
398                           log3("ERROR: Exception : " + e);
399                           testExitCode = FAILED;
400                       }
401                       break;
402 
403               default:
404                       throw new JDITestRuntimeException("** default case 2 **");
405             }
406 
407             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
408         }
409         log1("    TESTING ENDS");
410         return;

413    /*
414     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
415     *                                             String, String, String)
416     *
417     * It sets up a breakpoint at given line number within a given method in a given class
418     * for a given thread.
419     *
420     * Return value: BreakpointRequest object  in case of success
421     *
422     * JDITestRuntimeException   in case of an Exception thrown within the method
423     */
424 
425     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
426                                                   ReferenceType testedClass,
427                                                   String methodName,
428                                                   String bpLine,
429                                                   String property)
430             throws JDITestRuntimeException {
431 
432         log2("......setting up a breakpoint:");
433         log2("       thread: " + thread + "; class: " + testedClass +
434                         "; method: " + methodName + "; line: " + bpLine);
435 
436         List              alllineLocations = null;
437         Location          lineLocation     = null;
438         BreakpointRequest breakpRequest    = null;
439 
440         try {
441             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
442 
443             alllineLocations = method.allLineLocations();
444 
445             int n =
446                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
447             if (n > alllineLocations.size()) {
448                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
449             } else {
450                 lineLocation = (Location) alllineLocations.get(n);
451                 try {
452                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
453                     breakpRequest.putProperty("number", property);
454                     breakpRequest.addThreadFilter(thread);
455                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
456                 } catch ( Exception e1 ) {
457                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
458                     breakpRequest    = null;
459                 }
460             }
461         } catch ( Exception e2 ) {
462             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
463             breakpRequest    = null;
464         }
465 
466         if (breakpRequest == null) {
467             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
468             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
469         }
470 
471         log2("      a breakpoint has been set up");
472         return breakpRequest;
473     }

476     private void getEventSet()
477                  throws JDITestRuntimeException {
478         try {
479 //            log2("       eventSet = eventQueue.remove(waitTime);");
480             eventSet = eventQueue.remove(waitTime);
481             if (eventSet == null) {
482                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
483             }
484 //            log2("       eventIterator = eventSet.eventIterator;");
485             eventIterator = eventSet.eventIterator();
486         } catch ( Exception e ) {
487             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
488         }
489     }

492     private void breakpointForCommunication()
493                  throws JDITestRuntimeException {
494 
495         log2("breakpointForCommunication");
496         getEventSet();
497 
498         if (eventIterator.nextEvent() instanceof BreakpointEvent)
499             return;
500 
501         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
502     }

506     private AccessWatchpointRequest setting2AccessWatchpointRequest (
507                                                   ThreadReference thread,
508                                                   ReferenceType   testedClass,
509                                                   String          fieldName,
510                                                   int             suspendPolicy,
511                                                   String          property        )
512             throws JDITestRuntimeException {
513         try {
514             log2("......setting up AccessWatchpointRequest:");
515             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
516             Field field = testedClass.fieldByName(fieldName);
517 
518             AccessWatchpointRequest
519             awr = eventRManager.createAccessWatchpointRequest(field);
520             awr.putProperty("number", property);
521 
522             if (thread != null)
523                 awr.addThreadFilter(thread);
524             awr.setSuspendPolicy(suspendPolicy);
525 
526             log2("      AccessWatchpointRequest has been set up");
527             return awr;
528         } catch ( Exception e ) {
529             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
530             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
531             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
532         }
533     }
