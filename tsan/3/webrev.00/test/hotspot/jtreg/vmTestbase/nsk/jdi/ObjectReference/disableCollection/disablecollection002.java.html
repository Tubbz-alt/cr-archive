 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ObjectReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ObjectReference.disableCollection()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Prevents garbage collection for this object.              <BR>
 47  *    By default all ObjectReference values returned by JDI     <BR>
 48  *    may be collected at any time the target VM is running.    <BR>
 49  *    A call to this method guarantees that                     <BR>
 50  *    the object will not be collected.                         <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assetion,                               <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new array object "arr1" to be tested on      <BR>
 70  *   collection and invokes the methodForCommunication to be suspended  <BR>
 71  *   and to inform the debugger with the event.                         <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger gets ObjectReference mirroring arr1,                  <BR>
 74  *   disables collecting "arr1" resumes the debuggee and waits for new  <BR>
 75  *   BreakpointEvent.                                                   <BR>
 76  * - The debuggee assigns null to arr1 to make object's memory available<BR>
 77  *   to be collected and pertforms a loop in which it creates new       <BR>
 78  *   objects until VM throws OutOfMemoryError. Then it informs the      <BR>
 79  *   debugger of the ERROR with new BreakPoint event.                   <BR>
 80  * - Upon getting the event,                                            <BR>
 81  *   the debugger gets new ObjectReference mirroring arr1,              <BR>
 82  *   resumes the debuggee, and waits for new breakpoint.                <BR>
 83  * - The debuggee invokes a garbage collector and informs the debugger  <BR>
 84  *   with new breakpoint event.                                         <BR>
 85  * - Upon getting the event,                                            <BR>
 86  *   the debugger gets new ObjectReference mirroring arr1 once again.   <BR>
 87  * <BR>
 88  * If no ObjectCollectedexception is thrown during the above cooperation<BR>
 89  * the test is PASSED, othewise, FAILED.                                <BR>
 90  * <BR>
 91  * Note. To inform each other of needed actions, the debugger and       <BR>
 92  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 93  * <BR>
 94  * In third phase when at the end,                                      <BR>
 95  * the debuggee changes the value of the "instruction"                  <BR>
 96  * to inform the debugger of checks finished, and both end.             <BR>
 97  * <BR>
 98  */

172     private int runThis (String argv[], PrintStream out) {
173 
174         argsHandler     = new ArgumentHandler(argv);
175         logHandler      = new Log(out, argsHandler);
176         Binder binder   = new Binder(argsHandler, logHandler);
177 
178         waitTime        = argsHandler.getWaitTime() * 60000;
179 
180         try {
181             log2("launching a debuggee :");
182             log2("       " + debuggeeName);
183             if (argsHandler.verbose()) {
184                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
185             } else {
186                 debuggee = binder.bindToDebugee(debuggeeName);
187             }
188             if (debuggee == null) {
189                 log3("ERROR: no debuggee launched");
190                 return FAILED;
191             }
192             log2("debuggee launched");
193         } catch ( Exception e ) {
194             log3("ERROR: Exception : " + e);
195             log2("       test cancelled");
196             return FAILED;
197         }
198 
199         debuggee.redirectOutput(logHandler);
200 
201         vm = debuggee.VM();
202 
203         eventQueue = vm.eventQueue();
204         if (eventQueue == null) {
205             log3("ERROR: eventQueue == null : TEST ABORTED");
206             vm.exit(PASS_BASE);
207             return FAILED;
208         }
209 
210         log2("invocation of the method runTest()");
211         switch (runTest()) {
212 
213             case 0 :  log2("test phase has finished normally");
214                       log2("   waiting for the debuggee to finish ...");
215                       debuggee.waitFor();
216 
217                       log2("......getting the debuggee's exit status");
218                       int status = debuggee.getStatus();
219                       if (status != PASS_BASE) {
220                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
221                               status + " != PASS_BASE");
222                           testExitCode = FAILED;
223                       } else {
224                           log2("......debuggee returned expected exit status: " +
225                               status + " == PASS_BASE");
226                       }
227                       break;
228 
229             default : log3("ERROR: runTest() returned unexpected value");
230 
231             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
232                       log2("......forcing: vm.exit();");
233                       testExitCode = FAILED;
234                       try {
235                           vm.exit(PASS_BASE);
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240 
241             case 2 :  log3("test cancelled due to VMDisconnectedException");
242                       log2("......trying: vm.process().destroy();");
243                       testExitCode = FAILED;
244                       try {
245                           Process vmProcess = vm.process();
246                           if (vmProcess != null) {
247                               vmProcess.destroy();
248                           }
249                       } catch ( Exception e ) {
250                           log3("ERROR: Exception : e");
251                       }
252                       break;
253             }
254 
255         return testExitCode;
256     }

259    /*
260     * Return value: 0 - normal end of the test
261     *               1 - ubnormal end of the test
262     *               2 - VMDisconnectedException while test phase
263     */
264 
265     private int runTest() {
266 
267         try {
268             testRun();
269 
270             log2("waiting for VMDeathEvent");
271             getEventSet();
272             if (eventIterator.nextEvent() instanceof VMDeathEvent)
273                 return 0;
274 
275             log3("ERROR: last event is not the VMDeathEvent");
276             return 1;
277         } catch ( VMDisconnectedException e ) {
278             log3("ERROR: VMDisconnectedException : " + e);
279             return 2;
280         } catch ( Exception e ) {
281             log3("ERROR: Exception : " + e);
282             return 1;
283         }
284 
285     }

287     private void testRun()
288                  throws JDITestRuntimeException, Exception {
289 
290         eventRManager = vm.eventRequestManager();
291 
292         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
293         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
294         cpRequest.addClassFilter(debuggeeName);
295 
296         cpRequest.enable();
297         vm.resume();
298         getEventSet();
299         cpRequest.disable();
300 
301         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
302         debuggeeClass = event.referenceType();
303 
304         if (!debuggeeClass.name().equals(debuggeeName))
305            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
306 
307         log2("      received: ClassPrepareEvent for debuggeeClass");
308 
309         String bPointMethod = "methodForCommunication";
310         String lineForComm  = "lineForComm";
311         BreakpointRequest bpRequest;
312 
313         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
314                                       debuggeeClass,
315                                       bPointMethod, lineForComm, "zero");
316         bpRequest.enable();
317 
318     //------------------------------------------------------  testing section
319 
320         log1("     TESTING BEGINS");
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339             String name = "arr1";
340 
341             ReferenceType   arr1RefType = null;
342             ObjectReference arr1ObjRef  = null;
343 
344             try {
345                 log2("......getting: arr1ObjRef = (ObjectReference) debuggeeClass.getValue(debuggeeClass.fieldByName(name));");
346                 arr1ObjRef = (ObjectReference)
347                              debuggeeClass.getValue(debuggeeClass.fieldByName(name));
348 
349                 log2("......         arr1ObjRef.disableCollection();");
350                 arr1ObjRef.disableCollection();
351 
352                 log2("......vm.resume();");
353                 vm.resume();
354                 breakpointForCommunication();
355                 log2("      after breakpointForCommunication();");
356 
357                 log2("......getting: arr1RefType = arr1ObjRef.referenceType();");
358                 arr1RefType = arr1ObjRef.referenceType();
359 
360                 log2("......vm.resume();");
361                 vm.resume();
362                 breakpointForCommunication();
363                 log2("      after breakpointForCommunication();");
364 
365                 log2("......after debuggee invoked runTime.gc() :");
366                 log2("      getting: arr1RefType = arr1ObjRef.referenceType();");
367                 arr1RefType = arr1ObjRef.referenceType();
368 
369             } catch ( ObjectCollectedException e ) {
370                 log3("ERROR: caught ObjectCollectedException");
371                 throw new JDITestRuntimeException("** unexpected ObjectCollectedException **");
372             }
373 
374             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
375         }
376         log1("    TESTING ENDS");
377         return;
378     }

380    /*
381     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
382     *                                             String, String, String)
383     *
384     * It sets up a breakpoint at given line number within a given method in a given class
385     * for a given thread.
386     *
387     * Return value: BreakpointRequest object  in case of success
388     *
389     * JDITestRuntimeException   in case of an Exception thrown within the method
390     */
391 
392     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
393                                                   ReferenceType testedClass,
394                                                   String methodName,
395                                                   String bpLine,
396                                                   String property)
397             throws JDITestRuntimeException {
398 
399         log2("......setting up a breakpoint:");
400         log2("       thread: " + thread + "; class: " + testedClass +
401                         "; method: " + methodName + "; line: " + bpLine);
402 
403         List              alllineLocations = null;
404         Location          lineLocation     = null;
405         BreakpointRequest breakpRequest    = null;
406 
407         try {
408             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
409 
410             alllineLocations = method.allLineLocations();
411 
412             int n =
413                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
414             if (n > alllineLocations.size()) {
415                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
416             } else {
417                 lineLocation = (Location) alllineLocations.get(n);
418                 try {
419                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
420                     breakpRequest.putProperty("number", property);
421                     breakpRequest.addThreadFilter(thread);
422                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
423                 } catch ( Exception e1 ) {
424                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
425                     breakpRequest    = null;
426                 }
427             }
428         } catch ( Exception e2 ) {
429             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
430             breakpRequest    = null;
431         }
432 
433         if (breakpRequest == null) {
434             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
435             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
436         }
437 
438         log2("      a breakpoint has been set up");
439         return breakpRequest;
440     }

443     private void getEventSet()
444                  throws JDITestRuntimeException {
445         try {
446 //            log2("       eventSet = eventQueue.remove(waitTime);");
447             eventSet = eventQueue.remove(waitTime);
448             if (eventSet == null) {
449                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
450             }
451 //            log2("       eventIterator = eventSet.eventIterator;");
452             eventIterator = eventSet.eventIterator();
453         } catch ( Exception e ) {
454             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
455         }
456     }

459     private void breakpointForCommunication()
460                  throws JDITestRuntimeException {
461 
462         log2("breakpointForCommunication");
463         getEventSet();
464 
465         if (eventIterator.nextEvent() instanceof BreakpointEvent)
466             return;
467 
468         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
469     }
