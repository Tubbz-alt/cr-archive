 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.eventIterator()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for ClassPrepareEvent set               <BR>
 46  *  - the method returns a non-null object, and                 <BR>
 47  *  - object's class is the subclass of class Iterator.         <BR>
 48  * <BR>
 49  * The test works as follows.                                           <BR>
 50  * <BR>
 51  * Upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent,                                       <BR>
 55  * the debugger sets up two requests, both with SUSPEND_EVENT_THREAD,   <BR>
 56  * one for debuggee's ClassPrepareEvent,                                <BR>
 57  * another one for a special TestClass,  resumes the VM,                <BR>
 58  * and waits for the events within the predefined time interval.        <BR>
 59  * If no the events received, the test is FAILED.                       <BR>
 60  * Upon getting the debuggeeClass's ClassPrepareEvent,                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * Upon getting TestClass's ClassPrepareEvent,                          <BR>
 64  * the debugger performs the check.                                     <BR>
 65  * <BR>
 66  * At the end the debuggee changes the value of the "instruction"       <BR>
 67  * to inform the debugger of checks finished, and both ends.            <BR>
 68  * <BR>
 69  * Note. To inform each other of needed actions, the debugger and       <BR>
 70  *       and the debuggee use debuggee's variable "instruction".        <BR>
 71  * <BR>
 72  */
 73

 96     public static int run (String argv[], PrintStream out) {
 97 
 98         int exitCode = new eventiterator001().runThis(argv, out);

100         if (exitCode != PASSED) {
101             System.out.println("TEST FAILED");
102         }
103         return testExitCode;
104     }

150     private int runThis (String argv[], PrintStream out) {
151 
152         argsHandler     = new ArgumentHandler(argv);
153         logHandler      = new Log(out, argsHandler);
154         Binder binder   = new Binder(argsHandler, logHandler);
155 
156         waitTime        = argsHandler.getWaitTime() * 60000;
157 
158         try {
159             log2("launching a debuggee :");
160             log2("       " + debuggeeName);
161             if (argsHandler.verbose()) {
162                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
163             } else {
164                 debuggee = binder.bindToDebugee(debuggeeName);
165             }
166             if (debuggee == null) {
167                 log3("ERROR: no debuggee launched");
168                 return FAILED;
169             }
170             log2("debuggee launched");
171         } catch ( Exception e ) {
172             log3("ERROR: Exception : " + e);
173             log2("       test cancelled");
174             return FAILED;
175         }
176 
177         debuggee.redirectOutput(logHandler);
178 
179         vm = debuggee.VM();
180 
181         eventQueue = vm.eventQueue();
182         if (eventQueue == null) {
183             log3("ERROR: eventQueue == null : TEST ABORTED");
184             vm.exit(PASS_BASE);
185             return FAILED;
186         }
187 
188         log2("invocation of the method runTest()");
189         switch (runTest()) {
190 
191             case 0 :  log2("test phase has finished normally");
192                       log2("   waiting for the debuggee to finish ...");
193                       debuggee.waitFor();
194 
195                       log2("......getting the debuggee's exit status");
196                       int status = debuggee.getStatus();
197                       if (status != PASS_BASE) {
198                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
199                               status + " != PASS_BASE");
200                           testExitCode = FAILED;
201                       } else {
202                           log2("......debuggee returned expected exit status: " +
203                               status + " == PASS_BASE");
204                       }
205                       break;
206 
207             default : log3("ERROR: runTest() returned unexpected value");
208 
209             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
210                       log2("......forcing: vm.exit();");
211                       testExitCode = FAILED;
212                       try {
213                           vm.exit(PASS_BASE);
214                       } catch ( Exception e ) {
215                           log3("ERROR: Exception : e");
216                       }
217                       break;
218 
219             case 2 :  log3("test cancelled due to VMDisconnectedException");
220                       log2("......trying: vm.process().destroy();");
221                       testExitCode = FAILED;
222                       try {
223                           Process vmProcess = vm.process();
224                           if (vmProcess != null) {
225                               vmProcess.destroy();
226                           }
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231             }
232 
233         return testExitCode;
234     }

237    /*
238     * Return value: 0 - normal end of the test
239     *               1 - ubnormal end of the test
240     *               2 - VMDisconnectedException while test phase
241     */
242 
243     private int runTest() {
244 
245         try {
246             testRun();
247 
248             log2("waiting for VMDeathEvent");
249             getEventSet();
250 
251             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
252                 log3("ERROR: last event is not the VMDeathEvent");
253                 return 1;
254             }
255 
256             log2("waiting for VMDisconnectEvent");
257             getEventSet();
258 
259             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
260                 log3("ERROR: last event is not the VMDisconnectEvent");
261                 return 1;
262             }
263 
264             return 0;
265 
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273 
274     }

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         eventRManager = vm.eventRequestManager();
280 
281         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
282         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
283         cpRequest.addClassFilter(debuggeeName);
284         cpRequest.putProperty("number", "debuggeeClass");
285         cpRequest.enable();
286 
287         ClassPrepareRequest tcRequest = settingClassPrepareRequest(testedClassName,
288                                                     EventRequest.SUSPEND_EVENT_THREAD,
289                                                     "TestClassPrepareRequest");
290         tcRequest.enable();
291 
292         int nn1 = 0;
293         int nn2 = 0;
294         for (int nn = 0; nn < 2; nn++) {
295 
296             vm.resume();
297             getEventSet();
298 
299             ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
300 
301             log2("String property = (String) event.request().getProperty('number');");
302             String property = (String) event.request().getProperty("number");
303 
304             if (property.equals("debuggeeClass")) {
305 
306                 log2("      received: ClassPrepareEvent for debuggeeClass");
307 
308                 nn1++;
309                 if (nn1 > 1)
310                     throw new JDITestRuntimeException("** 2-nd event for debuggeeClass **");
311 
312                 cpRequest.disable();
313                 debuggeeClass = event.referenceType();
314 
315                 String          bPointMethod = "methodForCommunication";
316                 String          lineForComm  = "lineForComm";
317                 ThreadReference mainThread   = debuggee.threadByNameOrThrow("main");
318 
319                 BreakpointRequest bpRequest = settingBreakpoint(mainThread,
320                                               debuggeeClass,
321                                               bPointMethod, lineForComm, "zero");
322                 bpRequest.enable();
323 
324             } else if (property.equals("TestClassPrepareRequest")) {
325                 nn2++;
326                 if (nn2 > 1)
327                     throw new JDITestRuntimeException("** 2-nd event for TestClass **");
328 
329                 if (eventIterator == null) {
330                     testExitCode = FAILED;
331                     log3("ERROR: eventIterator == null");
332                 }
333                 if ( !(eventIterator instanceof Iterator) ) {
334                     testExitCode = FAILED;
335                     log3("ERROR: eventIterator is NOT instanceof Iterator");
336                 }
337             } else {
338                 log3("ERROR: unexpected Event :  property == " + property);
339                 throw new JDITestRuntimeException("** UNEXPECTED Event **");
340             }
341         }
342 
343     //------------------------------------------------------  testing section
344 
345         log1("     TESTING BEGINS");
346 
347         vm.resume();
348         breakpointForCommunication();
349 
350         int instruction = ((IntegerValue)
351                            (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
352 
353         if (instruction != 0) {
354             throw new JDITestRuntimeException("** instruction != 0 **");
355         }
356         vm.resume();
357             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
358             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
359 
360         log1("    TESTING ENDS");
361         return;
362     }

364    /*
365     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
366     *                                             String, String, String)
367     *
368     * It sets up a breakpoint at given line number within a given method in a given class
369     * for a given thread.
370     *
371     * Return value: BreakpointRequest object  in case of success
372     *
373     * JDITestRuntimeException   in case of an Exception thrown within the method
374     */
375 
376     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
377                                                   ReferenceType testedClass,
378                                                   String methodName,
379                                                   String bpLine,
380                                                   String property)
381             throws JDITestRuntimeException {
382 
383         log2("......setting up a breakpoint:");
384         log2("       thread: " + thread + "; class: " + testedClass +
385                         "; method: " + methodName + "; line: " + bpLine);
386 
387         List              alllineLocations = null;
388         Location          lineLocation     = null;
389         BreakpointRequest breakpRequest    = null;
390 
391         try {
392             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
393 
394             alllineLocations = method.allLineLocations();
395 
396             int n =
397                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
398             if (n > alllineLocations.size()) {
399                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
400             } else {
401                 lineLocation = (Location) alllineLocations.get(n);
402                 try {
403                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
404                     breakpRequest.putProperty("number", property);
405                     breakpRequest.addThreadFilter(thread);
406                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
407                 } catch ( Exception e1 ) {
408                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
409                     breakpRequest    = null;
410                 }
411             }
412         } catch ( Exception e2 ) {
413             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
414             breakpRequest    = null;
415         }
416 
417         if (breakpRequest == null) {
418             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
419             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
420         }
421 
422         log2("      a breakpoint has been set up");
423         return breakpRequest;
424     }

427     private void getEventSet()
428                  throws JDITestRuntimeException {
429         try {
430 //            log2("       eventSet = eventQueue.remove(waitTime);");
431             eventSet = eventQueue.remove(waitTime);
432             if (eventSet == null) {
433                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
434             }
435 //            log2("       eventIterator = eventSet.eventIterator;");
436             eventIterator = eventSet.eventIterator();
437         } catch ( Exception e ) {
438             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
439         }
440     }

443     private void breakpointForCommunication()
444                  throws JDITestRuntimeException {
445 
446         log2("breakpointForCommunication");
447         getEventSet();
448 
449         if (eventIterator.nextEvent() instanceof BreakpointEvent)
450             return;
451 
452         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
453     }

457     private ClassPrepareRequest settingClassPrepareRequest ( String  testedClass,
458                                                              int     suspendPolicy,
459                                                              String  property       )
460             throws JDITestRuntimeException {
461         try {
462             log2("......setting up ClassPrepareRequest:");
463             log2("       class: " + testedClass + "; property: " + property);
464 
465             ClassPrepareRequest
466             cpr = eventRManager.createClassPrepareRequest();
467             cpr.putProperty("number", property);
468             cpr.addClassFilter(testedClass);
469             cpr.setSuspendPolicy(suspendPolicy);
470 
471             log2("      ClassPrepareRequest has been set up");
472             return cpr;
473         } catch ( Exception e ) {
474             log3("ERROR: ATTENTION: Exception within settingClassPrepareRequest() : " + e);
475             log3("       ClassPreparenRequest HAS NOT BEEN SET UP");
476             throw new JDITestRuntimeException("** FAILURE to set up ClassPrepareRequest **");
477         }
