 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.eventIterator()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for VMStart, VMDeath, and               <BR>
 46  * VMDisconnect Events:                                         <BR>
 47  *  - the method returns non-null object, and                   <BR>
 48  *  - object's class is subclass of class Iterator.             <BR>
 49  * <BR>
 50  * The test works as follows.                                   <BR>
 51  * <BR>
 52  * Upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it checks up on the method and        <BR>
 56  * sets up the request for debuggee's                                   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * <BR>
 61  * The debugger then resumes the debuggee which will normally end       <BR>
 62  * that will result in VMDeathEvent and waits for the event.            <BR>
 63  * Upon getting the VMDeathEvent, it checks up on the method and        <BR>
 64  * waits for VMDisconnectEvent, and upon getting it, checks up on this. <BR>
 65  * <BR>
 66  */

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugeeNoWait(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugeeNoWait(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : e");
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : e");
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

248    /*
249     * Return value: 0 - normal end of the test
250     *               1 - ubnormal end of the test
251     *               2 - VMDisconnectedException while test phase
252     */
253 
254     private int runTest() {
255 
256         try {
257             log2("waiting for VMStartEvent");
258             getEventSet();
259             check();
260 
261 
262             if (eventIterator.nextEvent() instanceof VMStartEvent) {
263                 log2("VMStartEvent received; test begins");
264 
265                 testRun();
266 
267                 log2("waiting for VMDeathEvent");
268                 getEventSet();
269                 check();
270 
271                 if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
272                     log3("ERROR: last event is not the VMDeathEvent");
273                     return 1;
274                 }
275 
276                 log2("waiting for VMDisconnectEvent");
277                 getEventSet();
278                 check();
279                 if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
280                     log3("ERROR: last event is not the VMDisconnectEvent");
281                     return 1;
282                 }
283 
284                 return 0;
285             } else {
286                 log3("ERROR: first event is not the VMStartEvent");
287                 return 1;
288             }
289         } catch ( VMDisconnectedException e ) {
290             log3("ERROR: VMDisconnectedException : " + e);
291             return 2;
292         } catch ( Exception e ) {
293             log3("ERROR: Exception : " + e);
294             return 1;
295         }
296

299     private void testRun()
300                  throws JDITestRuntimeException, Exception {
301 
302         eventRManager = vm.eventRequestManager();
303 
304         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
305         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
306         cpRequest.addClassFilter(debuggeeName);
307 
308         log2("......setting up ClassPrepareRequest");
309         eRequests[7] = cpRequest;
310 
311         cpRequest.enable();
312         vm.resume();
313 
314         getEventSet();
315         eventSets[7] = eventSet;
316 
317         cpRequest.disable();
318 
319         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
320         debuggeeClass = event.referenceType();
321 
322         if (!debuggeeClass.name().equals(debuggeeName))
323            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
324 
325         log2("      received: ClassPrepareEvent for debuggeeClass");
326 
327         log2("......setting up ClassPrepareEvent");
328 
329         String bPointMethod = "methodForCommunication";
330         String lineForComm  = "lineForComm";
331         BreakpointRequest bpRequest;
332 
333         ThreadReference mainThread = threadByName("main");
334 
335         bpRequest = settingBreakpoint(mainThread,
336                                       debuggeeClass,
337                                       bPointMethod, lineForComm, "zero");
338         bpRequest.enable();
339 
340     //------------------------------------------------------  testing section
341 
342         log1("     TESTING BEGINS");
343 
344         for (int i = 0; ; i++) {
345 
346             vm.resume();
347             breakpointForCommunication();
348 
349             int instruction = ((IntegerValue)
350                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
351 
352             if (instruction == 0) {
353                 vm.resume();
354                 break;
355             }
356 
357             log1(":::::: case: # " + i);
358 
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
360             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
361         }
362         log1("    TESTING ENDS");
363         return;
364     }

366     private ThreadReference threadByName(String name)
367                  throws JDITestRuntimeException {
368 
369         List         all = vm.allThreads();
370         ListIterator li  = all.listIterator();
371 
372         for (; li.hasNext(); ) {
373             ThreadReference thread = (ThreadReference) li.next();
374             if (thread.name().equals(name))
375                 return thread;
376         }
377         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
378     }

380    /*
381     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
382     *                                             String, String, String)
383     *
384     * It sets up a breakpoint at given line number within a given method in a given class
385     * for a given thread.
386     *
387     * Return value: BreakpointRequest object  in case of success
388     *
389     * JDITestRuntimeException   in case of an Exception thrown within the method
390     */
391 
392     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
393                                                   ReferenceType testedClass,
394                                                   String methodName,
395                                                   String bpLine,
396                                                   String property)
397             throws JDITestRuntimeException {
398 
399         log2("......setting up a breakpoint:");
400         log2("       thread: " + thread + "; class: " + testedClass +
401                         "; method: " + methodName + "; line: " + bpLine);
402 
403         List              alllineLocations = null;
404         Location          lineLocation     = null;
405         BreakpointRequest breakpRequest    = null;
406 
407         try {
408             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
409 
410             alllineLocations = method.allLineLocations();
411 
412             int n =
413                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
414             if (n > alllineLocations.size()) {
415                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
416             } else {
417                 lineLocation = (Location) alllineLocations.get(n);
418                 try {
419                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
420                     breakpRequest.putProperty("number", property);
421                     breakpRequest.addThreadFilter(thread);
422                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
423                 } catch ( Exception e1 ) {
424                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
425                     breakpRequest    = null;
426                 }
427             }
428         } catch ( Exception e2 ) {
429             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
430             breakpRequest    = null;
431         }
432 
433         if (breakpRequest == null) {
434             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
435             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
436         }
437 
438         log2("      a breakpoint has been set up");
439         return breakpRequest;
440     }

443     private void getEventSet()
444                  throws JDITestRuntimeException {
445         try {
446 //            log2("       eventSet = eventQueue.remove(waitTime);");
447             eventSet = eventQueue.remove(waitTime);
448             if (eventSet == null) {
449                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
450             }
451 //            log2("       eventIterator = eventSet.eventIterator;");
452             eventIterator = eventSet.eventIterator();
453         } catch ( Exception e ) {
454             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
455         }
456     }
