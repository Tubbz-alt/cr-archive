 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a ThreadDeathEvent set,             <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up one to three new ThreadDeathRequests,         <BR>
 70  *    resumes the debuggee, and waits for the ThreadDeathEvents.        <BR>
 71  * - The debuggee starts new thread to be resulting in the events.      <BR>
 72  * - Upon getting new ThreadDeathEvent,                                 <BR>
 73  *   the debugger performs the check corresponding to the EventSet.     <BR>
 74  * <BR>
 75  * Note. To inform each other of needed actions, the debugger and       <BR>
 76  *       and the debuggee use debuggee's variable "instruction".        <BR>
 77  * In third phase, at the end,                                          <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  */
 82

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : e");
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

252     * Return value: 0 - normal end of the test
253     *               1 - ubnormal end of the test
254     *               2 - VMDisconnectedException while test phase
255     */

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
265                 log3("ERROR: last event is not the VMDeathEvent");
266                 return 1;
267             }
268 
269             log2("waiting for VMDisconnectEvent");
270             getEventSet();
271             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
272                 log3("ERROR: last event is not the VMDisconnectEvent");
273                 return 1;
274             }
275 
276             return 0;
277 
278         } catch ( VMDisconnectedException e ) {
279             log3("ERROR: VMDisconnectedException : " + e);
280             return 2;
281         } catch ( Exception e ) {
282             log3("ERROR: Exception : " + e);
283             return 1;
284         }
285

288     private void testRun()
289                  throws JDITestRuntimeException, Exception {
290 
291         eventRManager = vm.eventRequestManager();
292 
293         log2("......getting ClassPrepareEvent for debuggee's class");
294         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
295         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
296         cpRequest.addClassFilter(debuggeeName);
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307         log2("      received: ClassPrepareEvent for debuggeeClass");
308 
309         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
310 
311         String            bPointMethod = "methodForCommunication";
312         String            lineForComm  = "lineForComm";
313         BreakpointRequest bpRequest;
314         ThreadReference   mainThread = threadByName("main");
315         bpRequest = settingBreakpoint(mainThread,
316                                       debuggeeClass,
317                                       bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320         vm.resume();
321 
322     //------------------------------------------------------  testing section
323 
324         log1("     TESTING BEGINS");
325 
326         EventRequest eventRequest1 = null;
327         EventRequest eventRequest2 = null;
328         EventRequest eventRequest3 = null;
329 
330         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
331         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
332         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
333         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
334 
335         int policyExpected[] = {    SUSPEND_NONE,
336                                     SUSPEND_THREAD,
337                                     SUSPEND_ALL,
338                                     SUSPEND_THREAD,
339                                     SUSPEND_ALL,
340                                     SUSPEND_ALL,
341                                     SUSPEND_ALL      };
342         int policy = 0;
343 
344 
345         for (int i = 0; ; i++) {
346 
347             breakpointForCommunication();
348 
349             int instruction = ((IntegerValue)
350                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
351             if (instruction == 0) {
352                 vm.resume();
353                 break;
354             }
355 
356 
357             log1(":::::: case: # " + i);
358 
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
360 
361             switch (i) {
362 
363               case 0:
364                       eventRequest1 = settingThreadDeathRequest (
365                                              SUSPEND_NONE, "ThreadDeathRequest1");
366                       eventRequest1.enable();
367                       break;
368 
369               case 1:
370                       eventRequest2 = settingThreadDeathRequest (
371                                              SUSPEND_THREAD, "ThreadDeathRequest2");
372                       eventRequest2.enable();
373                       break;
374 
375               case 2:
376                       eventRequest3 = settingThreadDeathRequest (
377                                              SUSPEND_ALL, "ThreadDeathRequest3");
378                       eventRequest3.enable();
379                       break;
380 
381               case 3:
382                       eventRequest1 = settingThreadDeathRequest (
383                                              SUSPEND_NONE, "ThreadDeathRequest1");
384                       eventRequest1.enable();
385                       eventRequest2 = settingThreadDeathRequest (
386                                              SUSPEND_THREAD, "ThreadDeathRequest2");
387                       eventRequest2.enable();
388                       break;
389 
390               case 4:
391                       eventRequest1 = settingThreadDeathRequest (
392                                              SUSPEND_NONE, "ThreadDeathRequest1");
393                       eventRequest1.enable();
394                       eventRequest3 = settingThreadDeathRequest (
395                                              SUSPEND_ALL, "ThreadDeathRequest3");
396                       eventRequest3.enable();
397                       break;
398 
399               case 5:
400                       eventRequest2 = settingThreadDeathRequest (
401                                              SUSPEND_THREAD, "ThreadDeathRequest2");
402                       eventRequest2.enable();
403                       eventRequest3 = settingThreadDeathRequest (
404                                              SUSPEND_ALL, "ThreadDeathRequest3");
405                       eventRequest3.enable();
406                       break;
407 
408               case 6:
409                       eventRequest1 = settingThreadDeathRequest (
410                                              SUSPEND_NONE, "ThreadDeathRequest1");
411                       eventRequest1.enable();
412                       eventRequest2 = settingThreadDeathRequest (
413                                              SUSPEND_THREAD, "ThreadDeathRequest2");
414                       eventRequest2.enable();
415                       eventRequest3 = settingThreadDeathRequest (
416                                              SUSPEND_ALL, "ThreadDeathRequest3");
417                       eventRequest3.enable();
418                       break;
419 
420 
421               default:
422                       throw new JDITestRuntimeException("** default case 2 **");
423             }
424 
425             mainThread.resume();
426             getEventSet();
427 
428             if ( !(eventIterator.nextEvent() instanceof ThreadDeathEvent)) {
429                  log3("ERROR: new event is not ThreadDeathEvent");
430                  testExitCode = FAILED;
431             } else {
432                 log2("......got : instanceof ThreadDeathEvent");
433                 policy = eventSet.suspendPolicy();
434                 if (policy != policyExpected[i]) {
435                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
436                     log3("       eventSet.suspendPolicy() == " + policy);
437                     log3("       policyExpected           == " + policyExpected[i]);
438                     testExitCode = FAILED;
439                 }
440             }
441 
442             switch (i) {
443               case 0: eventRequest1.disable();                          break;
444               case 1: eventRequest2.disable();                          break;
445               case 2: eventRequest3.disable();                          break;
446               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
447               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
448               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
449               case 6: eventRequest1.disable(); eventRequest2.disable();
450                       eventRequest3.disable();                          break;
451            }
452 
453             switch (policy) {
454               case SUSPEND_NONE   :              break;
455               case SUSPEND_THREAD : vm.resume(); break;
456               case SUSPEND_ALL    : vm.resume(); break;
457               default: throw new JDITestRuntimeException("** default case 3 **");
458             }
459 
460             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
461         }
462         log1("    TESTING ENDS");
463         return;
464     }

466     private ThreadReference threadByName(String name)
467                  throws JDITestRuntimeException {
468 
469         List         all = vm.allThreads();
470         ListIterator li  = all.listIterator();
471 
472         for (; li.hasNext(); ) {
473             ThreadReference thread = (ThreadReference) li.next();
474             if (thread.name().equals(name))
475                 return thread;
476         }
477         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
478     }

480    /*
481     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
482     *                                             String, String, String)
483     *
484     * It sets up a breakpoint at given line number within a given method in a given class
485     * for a given thread.
486     *
487     * Return value: BreakpointRequest object  in case of success
488     *
489     * JDITestRuntimeException   in case of an Exception thrown within the method
490     */
491 
492     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
493                                                   ReferenceType testedClass,
494                                                   String methodName,
495                                                   String bpLine,
496                                                   String property)
497             throws JDITestRuntimeException {
498 
499         log2("......setting up a breakpoint:");
500         log2("       thread: " + thread + "; class: " + testedClass +
501                         "; method: " + methodName + "; line: " + bpLine);
502 
503         List              alllineLocations = null;
504         Location          lineLocation     = null;
505         BreakpointRequest breakpRequest    = null;
506 
507         try {
508             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
509 
510             alllineLocations = method.allLineLocations();
511 
512             int n =
513                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
514             if (n > alllineLocations.size()) {
515                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
516             } else {
517                 lineLocation = (Location) alllineLocations.get(n);
518                 try {
519                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
520                     breakpRequest.putProperty("number", property);
521                     breakpRequest.addThreadFilter(thread);
522                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
523                 } catch ( Exception e1 ) {
524                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
525                     breakpRequest    = null;
526                 }
527             }
528         } catch ( Exception e2 ) {
529             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
530             breakpRequest    = null;
531         }
532 
533         if (breakpRequest == null) {
534             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
535             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
536         }
537 
538         log2("      a breakpoint has been set up");
539         return breakpRequest;
540     }

544                  throws JDITestRuntimeException {
545         try {
546 //            log2("       eventSet = eventQueue.remove(waitTime);");

553         } catch ( Exception e ) {
554             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
555         }
556     }

587         }
