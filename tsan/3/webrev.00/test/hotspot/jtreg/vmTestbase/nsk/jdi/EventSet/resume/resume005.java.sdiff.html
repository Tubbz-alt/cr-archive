507     private ThreadReference threadByName(String name)
508                  throws JDITestRuntimeException {
509 
510         List         all = vm.allThreads();
511         ListIterator li  = all.listIterator();
512 
513         for (; li.hasNext(); ) {
514             ThreadReference thread = (ThreadReference) li.next();
515             if (thread.name().equals(name))
516                 return thread;
517         }
518         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
519     }
