 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.nestedTypes()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Cases for testing include ArrayType and primitive classes.   <BR>
 46  * <BR>
 47  * The test works as follows.                                   <BR>
 48  * Upon launching debuggee's VM which will be suspended,        <BR>
 49  * a debugger waits for the VMStartEvent within a predefined    <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.<BR>
 51  * Upon getting the VMStartEvent, it makes the request          <BR>
 52  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,  <BR>
 53  * resumes the VM, and waits for the event within the predefined<BR>
 54  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 57  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 58  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 59  * to be suspended and to inform the debugger with the event.           <BR>
 60  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 61  * At the end, the debuggee changes the value of the "instruction"      <BR>
 62  * to inform the debugger of checks finished, and both end.             <BR>
 63  */
 64

139     private int runThis (String argv[], PrintStream out) {
140 
141         argsHandler     = new ArgumentHandler(argv);
142         logHandler      = new Log(out, argsHandler);
143         Binder binder   = new Binder(argsHandler, logHandler);
144 
145         waitTime        = argsHandler.getWaitTime() * 60000;
146 
147         try {
148             log2("launching a debuggee :");
149             log2("       " + debuggeeName);
150             if (argsHandler.verbose()) {
151                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
152             } else {
153                 debuggee = binder.bindToDebugee(debuggeeName);
154             }
155             if (debuggee == null) {
156                 log3("ERROR: no debuggee launched");
157                 return FAILED;
158             }
159             log2("debuggee launched");
160         } catch ( Exception e ) {
161             log3("ERROR: Exception : " + e);
162             log2("       test cancelled");
163             return FAILED;
164         }
165 
166         debuggee.redirectOutput(logHandler);
167 
168         vm = debuggee.VM();
169 
170         eventQueue = vm.eventQueue();
171         if (eventQueue == null) {
172             log3("ERROR: eventQueue == null : TEST ABORTED");
173             vm.exit(PASS_BASE);
174             return FAILED;
175         }
176 
177         log2("invocation of the method runTest()");
178         switch (runTest()) {
179 
180             case 0 :  log2("test phase has finished normally");
181                       log2("   waiting for the debuggee to finish ...");
182                       debuggee.waitFor();
183 
184                       log2("......getting the debuggee's exit status");
185                       int status = debuggee.getStatus();
186                       if (status != PASS_BASE) {
187                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
188                               status + " != PASS_BASE");
189                           testExitCode = FAILED;
190                       } else {
191                           log2("......debuggee returned expected exit status: " +
192                               status + " == PASS_BASE");
193                       }
194                       break;
195 
196             default : log3("ERROR: runTest() returned unexpected value");
197 
198             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
199                       log2("......forcing: vm.exit();");
200                       testExitCode = FAILED;
201                       try {
202                           vm.exit(PASS_BASE);
203                       } catch ( Exception e ) {
204                           log3("ERROR: Exception : e");
205                       }
206                       break;
207 
208             case 2 :  log3("test cancelled due to VMDisconnectedException");
209                       log2("......trying: vm.process().destroy();");
210                       testExitCode = FAILED;
211                       try {
212                           Process vmProcess = vm.process();
213                           if (vmProcess != null) {
214                               vmProcess.destroy();
215                           }
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : e");
218                       }
219                       break;
220             }
221 
222         return testExitCode;
223     }

226    /*
227     * Return value: 0 - normal end of the test
228     *               1 - ubnormal end of the test
229     *               2 - VMDisconnectedException while test phase
230     */
231 
232     private int runTest() {
233 
234         try {
235             testRun();
236 
237             log2("waiting for VMDeathEvent");
238             getEventSet();
239             if (eventIterator.nextEvent() instanceof VMDeathEvent)
240                 return 0;
241 
242             log3("ERROR: last event is not the VMDeathEvent");
243             return 1;
244         } catch ( VMDisconnectedException e ) {
245             log3("ERROR: VMDisconnectedException : " + e);
246             return 2;
247         } catch ( Exception e ) {
248             log3("ERROR: Exception : " + e);
249             return 1;
250         }
251 
252     }

256     private void testRun()
257                  throws JDITestRuntimeException, Exception {
258 
259         eventRManager = vm.eventRequestManager();
260 
261         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
262         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
263         cpRequest.addClassFilter(debuggeeName);
264 
265         cpRequest.enable();
266         vm.resume();
267         getEventSet();
268         cpRequest.disable();
269 
270         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
271         debuggeeClass = event.referenceType();
272 
273         if (!debuggeeClass.name().equals(debuggeeName))
274            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
275 
276         log2("      received: ClassPrepareEvent for debuggeeClass");
277 
278         String bPointMethod = "methodForCommunication";
279         String lineForComm  = "lineForComm";
280         BreakpointRequest bpRequest;
281 
282         try {
283             bpRequest = settingBreakpoint(threadByName("main"),
284                                           debuggeeClass,
285                                           bPointMethod, lineForComm, "zero");
286         } catch ( Exception e ) {
287             throw e;
288         }
289         bpRequest.enable();
290 
291     //------------------------------------------------------  testing section
292 
293         log1("     TESTING BEGINS");
294 
295         for (int i = 0; ; i++) {
296 
297             vm.resume();
298             breakpointForCommunication();
299 
300             int instruction = ((IntegerValue)
301                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
302 
303             if (instruction == 0) {
304                 vm.resume();
305                 break;
306             }
307 
308             log1("  new check: # " + i);
309 
310             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
311 
312             ReferenceType testedType  = null;
313             List          nestedTypes = null;
314             String        typeForCheck;
315 
316 
317             log2("----- Case for testing: ArrayType");
318 
319             typeForCheck = mName + ".nestedtypes002aTestClass[]";
320             log2("......typeForCheck: " + typeForCheck);
321 
322             log2("      getting: List classList = vm.classesByName(typeForCheck);");
323             List classes = vm.classesByName(typeForCheck);
324             if (classes.size() != 1) {
325                 log3("ERROR: classes.size() != 1  : " + classes.size());
326                 testExitCode = FAILED;
327                 continue;
328             }
329 
330             log2("      getting: ReferenceType testedType  = (ReferenceType) classList.get(0);");
331             testedType  = (ReferenceType) classes.get(0);
332 
333             log2("      getting: List nestedTypes = testedType.nestedTypes();");
334             nestedTypes = testedType.nestedTypes();
335             if (nestedTypes.size() != 0) {
336                 log3("ERROR: nestedTypes.size() != 0 : " + nestedTypes.size());
337                 testExitCode = FAILED;
338                 continue;
339             }
340 
341             log2("----- Cases for testing: primitive classes");
342 
343             String names[] = {
344                                  "bl",
345                                  "bt",
346                                  "ch",
347                                  "db",
348                                  "fl",
349                                  "in",
350                                  "ln",
351                                  "sh"
352                              };
353 
354             Method testMethod = (Method)debuggeeClass.methodsByName("main").get(0);
355             // bpEvent should be assigned while getting of BreakpointEvent
356             if (bpEvent == null) {
357                 throw new JDITestRuntimeException("bpEvent is null");
358             }
359 
360             // get stack farme with main method
361             StackFrame frame = null;
362             try {
363                 frame = bpEvent.thread().frame(1);
364             } catch (Exception e) {
365                 throw new JDITestRuntimeException("Unexpected exception while getting stack frame: " + e );
366             }
367             if (frame.location().method() != testMethod) {
368                 throw new JDITestRuntimeException("Cannot take frame of main method");
369             }
370 
371             for (int i1 = 0; i1 < names.length; i1++) {
372 
373                 log2("......check for field : " + names[i]);
374                 LocalVariable lVar = null;
375                 try {
376                     lVar = frame.visibleVariableByName(names[i1]);
377                 } catch (Exception e ) {
378                     throw new JDITestRuntimeException("Unexpected exception while searching for field " + names[i1] + " : " + e  );
379                 }
380                 Value val = null;
381                 try {
382                     val = frame.getValue(lVar);
383                 } catch (Exception e ) {
384                     throw new JDITestRuntimeException("Unexpected exception while getting value of field " + names[i1] + " : " + e  );
385                 }
386 
387                 testedType = ((ClassObjectReference)val).reflectedType();
388 
389                 log2("      checking nestedTypes() for ClassObjectReference of : " + testedType.name());
390                 nestedTypes = testedType.nestedTypes();
391                 if (nestedTypes.size() != 0) {
392                     log3("ERROR: nestedTypes.size() != 0 : " + nestedTypes.size());
393                     testExitCode = FAILED;
394                     continue;
395                 }
396             }
397 
398             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
399         }
400         log1("    TESTING ENDS");
401         return;
402     }

404     private ThreadReference threadByName(String name)
405                  throws JDITestRuntimeException {
406 
407         List         all = vm.allThreads();
408         ListIterator li  = all.listIterator();
409 
410         for (; li.hasNext(); ) {
411             ThreadReference thread = (ThreadReference) li.next();
412             if (thread.name().equals(name))
413                 return thread;
414         }
415         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
416     }

418    /*
419     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
420     *                                             String, String, String)
421     *
422     * It sets up a breakpoint at given line number within a given method in a given class
423     * for a given thread.
424     *
425     * Return value: BreakpointRequest object  in case of success
426     *
427     * JDITestRuntimeException   in case of an Exception thrown within the method
428     */
429 
430     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
431                                                   ReferenceType testedClass,
432                                                   String methodName,
433                                                   String bpLine,
434                                                   String property)
435             throws JDITestRuntimeException {
436 
437         log2("......setting up a breakpoint:");
438         log2("       thread: " + thread + "; class: " + testedClass +
439                         "; method: " + methodName + "; line: " + bpLine);
440 
441         List              alllineLocations = null;
442         Location          lineLocation     = null;
443         BreakpointRequest breakpRequest    = null;
444 
445         try {
446             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
447 
448             alllineLocations = method.allLineLocations();
449 
450             int n =
451                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
452             if (n > alllineLocations.size()) {
453                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
454             } else {
455                 lineLocation = (Location) alllineLocations.get(n);
456                 try {
457                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
458                     breakpRequest.putProperty("number", property);
459                     breakpRequest.addThreadFilter(thread);
460                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
461                 } catch ( Exception e1 ) {
462                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
463                     breakpRequest    = null;
464                 }
465             }
466         } catch ( Exception e2 ) {
467             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
468             breakpRequest    = null;
469         }
470 
471         if (breakpRequest == null) {
472             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
473             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
474         }
475 
476         log2("      a breakpoint has been set up");
477         return breakpRequest;
478     }

481     private void getEventSet()
482                  throws JDITestRuntimeException {
483         try {
484 //            log2("       eventSet = eventQueue.remove(waitTime);");
485             eventSet = eventQueue.remove(waitTime);
486             if (eventSet == null) {
487                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
488             }
489 //            log2("       eventIterator = eventSet.eventIterator;");
490             eventIterator = eventSet.eventIterator();
491         } catch ( Exception e ) {
492             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
493         }
494     }

497     private void breakpointForCommunication()
498                  throws JDITestRuntimeException {
499 
500         log2("breakpointForCommunication");
501         getEventSet();
502 
503         Event event = eventIterator.nextEvent();
504         if ( event instanceof BreakpointEvent) {
505             bpEvent = (BreakpointEvent)event;
506             return;
507         }
508         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
509     }
