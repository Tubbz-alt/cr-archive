437     private ThreadReference threadByName(String name)
438                  throws JDITestRuntimeException {
439 
440         List         all = vm.allThreads();
441         ListIterator li  = all.listIterator();
442 
443         for (; li.hasNext(); ) {
444             ThreadReference thread = (ThreadReference) li.next();
445             if (thread.name().equals(name))
446                 return thread;
447         }
448         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
449     }
