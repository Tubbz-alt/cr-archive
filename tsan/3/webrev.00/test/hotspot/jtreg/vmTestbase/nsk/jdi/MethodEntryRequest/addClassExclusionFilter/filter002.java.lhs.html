 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodEntryRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodEntryRequest.addClassExclusionFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 68  *   and to inform the debugger with the event.                         <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates an MethodEntryRequest and enables it,                    <BR>
 71  *   - invokes the method addClassExclusionFilter() on the request      <BR>
 72  *     expecting to catch InvalidRequestStateException,                 <BR>
 73  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 74  *     expecting to catch no InvalidRequestStateException,                      <BR>
 75  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 76  *     once again expecting to catch InvalidRequestStateException.              <BR>
 77  * <BR>
 78  * In third phase, at the end of the test, the debuggee changes         <BR>
 79  * the value of the "instruction" which the debugger and debuggee       <BR>
 80  * use to inform each other of needed actions,  and both end.           <BR>
 81  * <BR>
 82  */

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : " + e);
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : " + e);
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if (eventIterator.nextEvent() instanceof VMDeathEvent)
266                 return 0;
267 
268             log3("ERROR: last event is not the VMDeathEvent");
269             return 1;
270         } catch ( VMDisconnectedException e ) {
271             log3("ERROR: VMDisconnectedException : " + e);
272             return 2;
273         } catch ( Exception e ) {
274             log3("ERROR: Exception : " + e);
275             return 1;
276         }

280     private void testRun()
281                  throws JDITestRuntimeException, Exception {
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304 
305         ThreadReference   mainThread = threadByName("main");
306 
307         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
308                                              debuggeeClass,
309                                             bPointMethod, lineForComm, "zero");
310         bpRequest.enable();
311 
312     //------------------------------------------------------  testing section
313 
314         log1("     TESTING BEGINS");
315 
316         EventRequest  eventRequest1      = null;
317 
318         String        property1          = "MethodEntryRequest1";
319 
320 
321         for (int i = 0; ; i++) {
322 
323             vm.resume();
324             breakpointForCommunication();
325 
326             int instruction = ((IntegerValue)
327                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
328 
329             if (instruction == 0) {
330                 vm.resume();
331                 break;
332             }
333 
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339             switch (i) {
340 
341               case 0:
342                      eventRequest1 = setting23MethodEntryRequest(null, testedClassName1,
343                                               EventRequest.SUSPEND_NONE, property1);
344 
345                      eventRequest1.enable();
346 
347                      try {
348                          log2("......eventRequest1.addClassFilter(testedClassName1);");
349                          log2("        InvalidRequestStateException expected");
350                          ((MethodEntryRequest)eventRequest1).addClassFilter(testedClassName1);
351                          log3("ERROR: no InvalidRequestStateException");
352                          testExitCode = FAILED;
353                      } catch ( InvalidRequestStateException e ) {
354                          log2("        InvalidRequestStateException");
355                      }
356 
357                      log2("......eventRequest1.disable();");
358                      eventRequest1.disable();
359                      try {
360                          log2("......eventRequest1.addClassFilter(testedClassName1);");
361                          log2("        no InvalidRequestStateException expected");
362                          ((MethodEntryRequest)eventRequest1).addClassFilter(testedClassName1);
363                          log2("        no InvalidRequestStateException");
364                      } catch ( InvalidRequestStateException e ) {
365                          log3("ERROR: InvalidRequestStateException");
366                          testExitCode = FAILED;
367                      }
368 
369                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
370                      eventRManager.deleteEventRequest(eventRequest1);
371                      try {
372                          log2("......eventRequest1.addClassFilter(testedClassName1);");
373                          log2("        InvalidRequestStateException expected");
374                          ((MethodEntryRequest)eventRequest1).addClassFilter(testedClassName1);
375                          log3("ERROR: no InvalidRequestStateException");
376                          testExitCode = FAILED;
377                      } catch ( InvalidRequestStateException e ) {
378                          log2("        InvalidRequestStateException");
379                      }
380                       break;
381 
382               default:
383                       throw new JDITestRuntimeException("** default case 2 **");
384             }
385 
386             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
387         }
388         log1("    TESTING ENDS");
389         return;
390     }

392     private ThreadReference threadByName(String name)
393                  throws JDITestRuntimeException {
394 
395         List         all = vm.allThreads();
396         ListIterator li  = all.listIterator();
397 
398         for (; li.hasNext(); ) {
399             ThreadReference thread = (ThreadReference) li.next();
400             if (thread.name().equals(name))
401                 return thread;
402         }
403         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
404     }

406    /*
407     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
408     *                                             String, String, String)
409     *
410     * It sets up a breakpoint at given line number within a given method in a given class
411     * for a given thread.
412     *
413     * Return value: BreakpointRequest object  in case of success
414     *
415     * JDITestRuntimeException   in case of an Exception thrown within the method
416     */
417 
418     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
419                                                   ReferenceType testedClass,
420                                                   String methodName,
421                                                   String bpLine,
422                                                   String property)
423             throws JDITestRuntimeException {
424 
425         log2("......setting up a breakpoint:");
426         log2("       thread: " + thread + "; class: " + testedClass +
427                         "; method: " + methodName + "; line: " + bpLine);
428 
429         List              alllineLocations = null;
430         Location          lineLocation     = null;
431         BreakpointRequest breakpRequest    = null;
432 
433         try {
434             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
435 
436             alllineLocations = method.allLineLocations();
437 
438             int n =
439                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
440             if (n > alllineLocations.size()) {
441                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
442             } else {
443                 lineLocation = (Location) alllineLocations.get(n);
444                 try {
445                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
446                     breakpRequest.putProperty("number", property);
447                     breakpRequest.addThreadFilter(thread);
448                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
449                 } catch ( Exception e1 ) {
450                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
451                     breakpRequest    = null;
452                 }
453             }
454         } catch ( Exception e2 ) {
455             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
456             breakpRequest    = null;
457         }
458 
459         if (breakpRequest == null) {
460             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
461             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
462         }
463 
464         log2("      a breakpoint has been set up");
465         return breakpRequest;
466     }

469     private void getEventSet()
470                  throws JDITestRuntimeException {
471         try {
472 //            log2("       eventSet = eventQueue.remove(waitTime);");
473             eventSet = eventQueue.remove(waitTime);
474             if (eventSet == null) {
475                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
476             }
477 //            log2("       eventIterator = eventSet.eventIterator;");
478             eventIterator = eventSet.eventIterator();
479         } catch ( Exception e ) {
480             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
481         }
482     }

485     private void breakpointForCommunication()
486                  throws JDITestRuntimeException {
487         log2("breakpointForCommunication");
488 
489         do {
490             getEventSet();
491 
492             Event event = eventIterator.nextEvent();
493             if (event instanceof BreakpointEvent)
494                 return;
495 
496             log2("      received: " + event);
497 
498             if (EventFilters.filtered(event, debuggeeName)) {
499                 eventSet.resume();
500             }
501             else {
502                 break;
503             }
504         } while (true);
505 
506         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
507     }

511     private MethodEntryRequest setting23MethodEntryRequest ( ThreadReference thread,
512                                                             String          testedClass,
513                                                             int             suspendPolicy,
514                                                             String          property       )
515             throws JDITestRuntimeException {
516         try {
517             log2("......setting up MethodEntryRequest:");
518             log2("       thread: " + thread + "; class: " + testedClass +  "; property: " + property);
519 
520             MethodEntryRequest
521             menr = eventRManager.createMethodEntryRequest();
522             menr.putProperty("number", property);
523             if (thread != null)
524                 menr.addThreadFilter(thread);
525             menr.addClassExclusionFilter(testedClass);
526             menr.setSuspendPolicy(suspendPolicy);
527             menr.addCountFilter(1);
528 
529             log2("      a MethodEntryRequest has been set up");
530             return menr;
531         } catch ( Exception e ) {
532             log3("ERROR: ATTENTION: Exception within settingMethodEntryRequest() : " + e);
533             log3("       MethodEntryRequest HAS NOT BEEN SET UP");
534             throw new JDITestRuntimeException("** FAILURE to set up MethodEntryRequest **");
535         }
