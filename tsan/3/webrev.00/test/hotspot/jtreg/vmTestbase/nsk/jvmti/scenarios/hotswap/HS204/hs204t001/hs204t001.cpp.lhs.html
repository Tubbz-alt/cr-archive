 51 char *getClassName(jvmtiEnv *jvmti, jclass  klass) {
 52     char * className;
 53     char * generic;
 54     if (!NSK_JVMTI_VERIFY(jvmti->GetClassSignature(klass, &className, &generic))) {
 55         nsk_jvmti_setFailStatus();
 56     }
 57     return className;
 58 }

138 static void JNICALL agentProc(jvmtiEnv* jvmti, JNIEnv* agentJNI, void* arg) {
139     redefineNumber = 0;
140     jni = agentJNI;
141     testStep=1;
142     NSK_DISPLAY0("\n\n>>>> Debugge started, waiting for class loading \n");
143     jni->DeleteGlobalRef(testClass);
144     jni->DeleteGlobalRef(testedThread);
145     NSK_DISPLAY0("Waiting for debuggee to become ready\n");
146     if (!nsk_jvmti_waitForSync(timeout)) {
147         return;
148     }
149     testStep = 1;
150     NSK_DISPLAY0("\n\n>>>> Debugge started, waiting for class loading \n");
151     if (!nsk_jvmti_resumeSync())
152         return;
153     NSK_DISPLAY0("Waiting for debuggee's threads to finish\n");
154     if (!nsk_jvmti_waitForSync(timeout))
155         return;
156     jni->DeleteGlobalRef(testClass);
157     jni->DeleteGlobalRef(testedThread);
158     NSK_DISPLAY0("Let debuggee to finish\n");
159     if (!nsk_jvmti_resumeSync())
160         return;
161 }

174 jint  Agent_Initialize(JavaVM *vm, char *options, void *reserved) {
175     jint rc ;
176     NSK_DISPLAY0(" VM.. Started..\n");
177     rc=vm->GetEnv((void **)&jvmti, JVMTI_VERSION_1_1);
178     if (rc != JNI_OK) {
179         NSK_COMPLAIN0(" Could not load JVMTI interface \n");
180     } else {
181         /* Open simple block for better memor usage. */
182         jvmtiCapabilities caps;
183         memset(&caps, 0, sizeof(caps));
184 
185         /*
186            set capabilities of
187            1.ClassFileLoadHock,
188            2.ClassLoad(doesn;t require any capabilities to set).
189            3.ClassPrepare (doesn;t require any capabilitiesto set).
190            4.Redefine (default).
191            5.PopFrame.
192          */
193         caps.can_generate_all_class_hook_events=1;
194         caps.can_access_local_variables = 1;
195         caps.can_generate_single_step_events=1;
196         caps.can_redefine_classes = 1;
197         caps.can_suspend = 1;
198         caps.can_pop_frame=1;
199         caps.can_generate_all_class_hook_events=1;
200         jvmti->AddCapabilities(&caps);
201         /*
202            set the method and other functions..
203          */
204         {
205             jvmtiEventCallbacks eventCallbacks;
206             memset(&eventCallbacks, 0, sizeof(eventCallbacks));
207             eventCallbacks.ClassLoad = callbackClassLoad;
208             eventCallbacks.ClassPrepare=callbackClassPrepare;
209             eventCallbacks.ClassFileLoadHook=callbackClassFileLoadHock;
210             rc=jvmti->SetEventCallbacks(&eventCallbacks, sizeof(eventCallbacks));
211             if (rc != JVMTI_ERROR_NONE) {
212                 NSK_COMPLAIN0("Error setting event callbacks");
213                 return JNI_ERR;
214             }
215         }
216         {
217             nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_SINGLE_STEP, testedThread);
218             nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_CLASS_LOAD, testedThread);
219             nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_CLASS_FILE_LOAD_HOOK, testedThread);
220             nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_CLASS_PREPARE, testedThread);
221             nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_CLASS_FILE_LOAD_HOOK,testedThread);
222         }
223 
224         if (!nsk_jvmti_setAgentProc(agentProc, NULL)) {
225             NSK_COMPLAIN0("setAgentProc failed");
226         }
227         if (!nsk_jvmti_parseOptions(options)) {
228             NSK_COMPLAIN0("Cannot parse options");
229         }
230         NSK_DISPLAY1("Wait time: %d\n",nsk_jvmti_getWaitTime());
231         timeout=nsk_jvmti_getWaitTime();
232         NSK_DISPLAY1(" returning back.. enter timeout-->%d  \n",timeout);
233         return JNI_OK;
234     }
235     // TODO: shouldn't we return JNI_ERR if GetEnv failed?
236     return JNI_OK;
237 }

250 JNIEXPORT jboolean JNICALL
251 Java_nsk_jvmti_scenarios_hotswap_HS204_hs204t001_hs204t001_suspendThread(JNIEnv * env,
252         jclass klass,
253         jobject thread) {
254     jint state;
255     NSK_DISPLAY0("---suspend thread .. \n");
256     if (jvmti->GetThreadState(thread, &state) == JVMTI_ERROR_NONE) {
257         NSK_DISPLAY0(" No Errors in finding state of the thread.\n");
258         if (state & JVMTI_THREAD_STATE_ALIVE) {
259             NSK_DISPLAY0(" Thread state is alive .. So can be suspend should be possible ..\n");
260             nsk_jvmti_disableNotification(jvmti, JVMTI_EVENT_SINGLE_STEP, thread);
261             if (!NSK_JVMTI_VERIFY(jvmti->SuspendThread(thread))) {
262                 NSK_COMPLAIN0("TEST FAILED: unable to suspend the thread \n");
263                 nsk_jvmti_setFailStatus();
264                 return NSK_FALSE;
265             } else {
266                 NSK_DISPLAY0(" Sucessfully suspended Thread..\n");
267             }
268         } else {
269             NSK_COMPLAIN0("Was not able to suspend a thread..\n");
270             return NSK_FALSE;
271         }
272     }
273     return NSK_TRUE;
274 }

276 JNIEXPORT jboolean JNICALL
277 Java_nsk_jvmti_scenarios_hotswap_HS204_hs204t001_hs204t001_popFrame(JNIEnv * env,
278         jclass klass,
279         jthread thread) {
280     jint state;
281     NSK_DISPLAY0("Inside pop_Frame method.....\n");
282     if (jvmti->GetThreadState(thread, &state) == JVMTI_ERROR_NONE) {
283         NSK_DISPLAY0(" Got the state of thread \n");
284         if (state & JVMTI_THREAD_STATE_SUSPENDED) {
285             NSK_DISPLAY0(" Thread is already in suspended mode..\n");
286             if (!NSK_JVMTI_VERIFY(jvmti->PopFrame(thread))) {
287                 NSK_COMPLAIN0(" TEST FAILED: UNABLE TO POP FRAME \n");
288                 nsk_jvmti_setFailStatus();
289                 return NSK_FALSE;
290             } else {
291                 NSK_DISPLAY0(" Poped frame safely..");
292             }
293             /* We should resume that thread for next execution.. */
294             if (!NSK_JVMTI_VERIFY(jvmti->ResumeThread(thread))) {
295                 NSK_COMPLAIN0(" TEST FAILED: UNABLE TO Resume thread \n");
296                 nsk_jvmti_setFailStatus();
297                 return NSK_FALSE;
298             } else {
299                 NSK_DISPLAY0(" Resumed.. thread for next set of executions...");
300             }
301         } else {
302             NSK_DISPLAY0(" Thread is not in Suspened State for poping its status..");
303         }
304     }
305     return NSK_TRUE;
306 }
