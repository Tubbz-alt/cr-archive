 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.suspendPolicy()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on following assertions:                  <BR>
 46  *   - Returns: the current suspend mode for this request       <BR>
 47  * The cases to check include Requests of all sub-types.        <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.              <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assertion,                              <BR>
 64  * the debugger and the debuggee perform the following loop.            <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee prepares new check case (as if) and invokes           <BR>
 67  *   the methodForCommunication to be suspended and                     <BR>
 68  *   to inform the debugger with the event.                             <BR>
 69  * - Upon getting the BreakpointEvent,                                  <BR>
 70  *   the debugger performs the check case required.                     <BR>
 71  * <BR>
 72  * In third phase, at the end of the test, the debuggee changes         <BR>
 73  * the value of the "instruction" which the debugger and debuggee       <BR>
 74  * use to inform each other of needed actions,  and both end.           <BR>
 75  * <BR>
 76  */

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugee(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : " + e);
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : " + e);
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

249     * Return value: 0 - normal end of the test
250     *               1 - ubnormal end of the test
251     *               2 - VMDisconnectedException while test phase

254     private int runTest() {
255 
256         try {
257             testRun();
258 
259             log2("waiting for VMDeathEvent");
260             getEventSet();
261             if (eventIterator.nextEvent() instanceof VMDeathEvent)
262                 return 0;
263 
264             log3("ERROR: last event is not the VMDeathEvent");
265             return 1;
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273 
274     }

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         eventRManager = vm.eventRequestManager();
280 
281         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
282         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
283         cpRequest.addClassFilter(debuggeeName);
284 
285         cpRequest.enable();
286         vm.resume();
287         getEventSet();
288         cpRequest.disable();
289 
290         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
291         debuggeeClass = event.referenceType();
292 
293         if (!debuggeeClass.name().equals(debuggeeName))
294            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
295 
296         log2("      received: ClassPrepareEvent for debuggeeClass");
297 
298         String bPointMethod = "methodForCommunication";
299         String lineForComm  = "lineForComm";
300 
301         ThreadReference   mainThread = threadByName("main");
302 
303         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
304                                              debuggeeClass,
305                                             bPointMethod, lineForComm, "zero");
306         bpRequest.enable();
307 
308     //------------------------------------------------------  testing section
309 
310 
311         EventRequest  eventRequest1 = null;
312 
313         String        fieldName1    = "var101";
314         String        fieldName2    = "excObj";
315 
316         ThreadReference thread1     = null;
317         String          threadName1 = "thread1";
318 
319         ReferenceType testClassReference = null;
320 
321         String propertyName   = "name";
322         String propertyValue1 = "value1";
323         String propertyValue2 = "value2";
324 
325 
326         log1("     TESTING BEGINS");
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345 
346             switch (i) {
347 
348               case 0:
349                      thread1 = threadByName(threadName1);
350 
351                      log2("......setting up StepRequest");
352                      eventRequest1 = eventRManager.createStepRequest
353                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
354                      break;
355 
356               case 1:
357                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
358 
359                      log2("......setting up AccessWatchpointRequest");
360                      eventRequest1 = eventRManager.createAccessWatchpointRequest
361                                     (testClassReference.fieldByName(fieldName1));
362                      break;
363 
364               case 2:
365                      log2(".....setting up ModificationWatchpointRequest");
366                      eventRequest1 = eventRManager.createModificationWatchpointRequest
367                                     (testClassReference.fieldByName(fieldName1));
368                      break;
369 
370               case 3:
371                      log2(".....setting up ClassPrepareRequest");
372                      eventRequest1 = eventRManager.createClassPrepareRequest();
373                      break;
374 
375               case 4:
376                      log2(".....setting up ClassUnloadRequest");
377                      eventRequest1 = eventRManager.createClassUnloadRequest();
378                      break;
379 
380               case 5:
381                      log2(".....setting up MethodEntryRequest");
382                      eventRequest1 = eventRManager.createMethodEntryRequest();
383                      break;
384 
385               case 6:
386                      log2(".....setting up MethodExitRequest");
387                      eventRequest1 = eventRManager.createMethodExitRequest();
388                      break;
389 
390               case 7:
391                      log2(".....setting up ThreadDeathRequest");
392                      eventRequest1 = eventRManager.createThreadDeathRequest();
393                      break;
394 
395               case 8:
396                      log2(".....setting up ThreadStartRequest");
397                      eventRequest1 = eventRManager.createThreadStartRequest();
398                      break;
399 
400               case 9:
401                      log2(".....setting up VMDeathRequest");
402                      eventRequest1 = eventRManager.createVMDeathRequest();
403                      break;
404 
405               case 10:
406                      log2(".....setting up ExceptionRequest");
407                      eventRequest1 = eventRManager.createExceptionRequest(
408                                      (ReferenceType)
409                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
410                                      true, true );
411                      break;
412 
413               case 11:
414                      log2(".....setting up BreakpointRequest");
415                      eventRequest1 = eventRManager.createBreakpointRequest(location);
416                      break;
417 
418 
419               default:
420                       throw new JDITestRuntimeException("** default case 2 **");
421             }
422 
423             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_ALL);");
424             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
425 
426             log2("      checking up on eventRequest1");
427             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_ALL) {
428                 testExitCode = FAILED;
429                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_ALL");
430             }
431 
432             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD );");
433             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD );
434 
435             log2("      checking up on eventRequest1");
436             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_EVENT_THREAD ) {
437                 testExitCode = FAILED;
438                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_EVENT_THREAD ");
439             }
440 
441             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);");
442             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
443 
444             log2("      checking up on eventRequest1");
445             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_NONE) {
446                 testExitCode = FAILED;
447                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_NONE");
448             }
449 
450             eventRManager.deleteEventRequest(eventRequest1);
451 
452             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
453         }
454         log1("    TESTING ENDS");
455         return;
456     }

458     private ThreadReference threadByName(String name)
459                  throws JDITestRuntimeException {
460 
461         List         all = vm.allThreads();
462         ListIterator li  = all.listIterator();
463 
464         for (; li.hasNext(); ) {
465             ThreadReference thread = (ThreadReference) li.next();
466             if (thread.name().equals(name))
467                 return thread;
468         }
469         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
470     }

472    /*
473     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
474     *                                             String, String, String)
475     *
476     * It sets up a breakpoint at given line number within a given method in a given class
477     * for a given thread.
478     *
479     * Return value: BreakpointRequest object  in case of success
480     *
481     * JDITestRuntimeException   in case of an Exception thrown within the method
482     */
483 
484     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
485                                                   ReferenceType testedClass,
486                                                   String methodName,
487                                                   String bpLine,
488                                                   String property)
489             throws JDITestRuntimeException {
490 
491         log2("......setting up a breakpoint:");
492         log2("       thread: " + thread + "; class: " + testedClass +
493                         "; method: " + methodName + "; line: " + bpLine);
494 
495         List              alllineLocations = null;
496         Location          lineLocation     = null;
497         BreakpointRequest breakpRequest    = null;
498 
499         try {
500             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
501 
502             alllineLocations = method.allLineLocations();
503 
504             int n =
505                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
506             if (n > alllineLocations.size()) {
507                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
508             } else {
509                 lineLocation = (Location) alllineLocations.get(n);
510 
511 // this is only for this test to get Location object
512 location = lineLocation;
513 
514                 try {
515                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
516                     breakpRequest.putProperty("number", property);
517                     breakpRequest.addThreadFilter(thread);
518                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
519                 } catch ( Exception e1 ) {
520                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
521                     breakpRequest    = null;
522                 }
523             }
524         } catch ( Exception e2 ) {
525             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
526             breakpRequest    = null;
527         }
528 
529         if (breakpRequest == null) {
530             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
531             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
532         }
533 
534         log2("      a breakpoint has been set up");
535         return breakpRequest;
536     }

539     private void getEventSet()
540                  throws JDITestRuntimeException {
541         try {
542 //            log2("       eventSet = eventQueue.remove(waitTime);");
543             eventSet = eventQueue.remove(waitTime);
544             if (eventSet == null) {
545                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
546             }
547 //            log2("       eventIterator = eventSet.eventIterator;");
548             eventIterator = eventSet.eventIterator();
549         } catch ( Exception e ) {
550             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
551         }
552     }

555     private void breakpointForCommunication()
556                  throws JDITestRuntimeException {
557 
558         log2("breakpointForCommunication");
559         getEventSet();
560 
561         if (eventIterator.nextEvent() instanceof BreakpointEvent)
562             return;
563 
564         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
565     }
