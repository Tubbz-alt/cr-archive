29 public class exceptionjni001 {
30     static {
31         try {
32             System.loadLibrary("exceptionjni001");
33         } catch (UnsatisfiedLinkError ule) {
34             System.err.println("Could not load exceptionjni001 library");
35             System.err.println("java.library.path:"
36                 + System.getProperty("java.library.path"));
37             throw ule;
38         }
39     }
40 
41     // A field the JNI code can try to get.
42     int anInteger;
43 
44     // Check the exception checking code.
45     native static boolean check();
46 
47     public static void main(String args[]) {
48         if (!check()) {
49           throw new RuntimeException("Problem with ExceptionCheckingJniEnv");
50         }
51     }
52 }
