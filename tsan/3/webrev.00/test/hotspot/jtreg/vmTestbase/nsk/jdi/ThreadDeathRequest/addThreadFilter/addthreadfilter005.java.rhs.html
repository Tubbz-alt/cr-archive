 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadDeathRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadDeathRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Restricts the events generated by this request            <BR>
 47  *    to those in the given thread.                             <BR>
 48  * The cases to test include three thread's states:             <BR>
 49  * not started, running, dead.                                  <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new thread, thread1, and invokes              <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates ThreadDeathRequest1 and                                  <BR>
 73  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 74  *   - creates ThreadDeathRequest2,                                     <BR>
 75  *     resumes the debuggee to get the thread1 started, and             <BR>
 76  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 77  *   - creates ThreadDeathRequest3, waits the thread1 to die, and       <BR>
 78  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  *
 86 
 87  */
 88

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : e");
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

252     * Return value: 0 - normal end of the test
253     *               1 - ubnormal end of the test
254     *               2 - VMDisconnectedException while test phase

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if (eventIterator.nextEvent() instanceof VMDeathEvent)
265                 return 0;
266 
267             log3("ERROR: last event is not the VMDeathEvent");
268             return 1;
269         } catch ( VMDisconnectedException e ) {
270             log3("ERROR: VMDisconnectedException : " + e);
271             return 2;
272         } catch ( Exception e ) {
273             log3("ERROR: Exception : " + e);
274             return 1;
275         }
276 
277     }

279     private void testRun()
280                  throws JDITestRuntimeException, Exception {
281 
282         eventRManager = vm.eventRequestManager();
283 
284         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
285         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
286         cpRequest.addClassFilter(debuggeeName);
287 
288         cpRequest.enable();
289         vm.resume();
290         getEventSet();
291         cpRequest.disable();
292 
293         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
294         debuggeeClass = event.referenceType();
295 
296         if (!debuggeeClass.name().equals(debuggeeName))
297            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
298 
299         log2("      received: ClassPrepareEvent for debuggeeClass");
300 
301         String bPointMethod = "methodForCommunication";
302         String lineForComm  = "lineForComm";
303 
304         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
305 
306         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
307                                              debuggeeClass,
308                                             bPointMethod, lineForComm, "zero");
309         bpRequest.enable();
310 
311     //------------------------------------------------------  testing section
312 
313         log1("     TESTING BEGINS");
314 
315         EventRequest eventRequest1 = null;
316         EventRequest eventRequest2 = null;
317         EventRequest eventRequest3 = null;
318 
319         ThreadReference thread1 = null;
320 
321         String thread1Name = "thread1";
322 
323         String property1 = "ThreadDeathRequest1";
324         String property2 = "ThreadDeathRequest2";
325         String property3 = "ThreadDeathRequest3";
326 
327         ReferenceType testClassReference = null;
328 
329 
330         for (int i = 0; ; i++) {
331 
332             vm.resume();
333             breakpointForCommunication();
334 
335             int instruction = ((IntegerValue)
336                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
337 
338             if (instruction == 0) {
339                 vm.resume();
340                 break;
341             }
342 
343             log1(":::::: case: # " + i);
344 
345             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
346 
347             switch (i) {
348 
349               case 0:
350                       testClassReference =
351                            (ReferenceType) vm.classesByName(testedClassName).get(0);
352 
353                       thread1 = (ThreadReference) debuggeeClass.getValue(
354                                                   debuggeeClass.fieldByName(thread1Name));
355 
356                       eventRequest1 = setting2ThreadDeathRequest (null,
357                                              EventRequest.SUSPEND_NONE, property1);
358 
359                       try {
360                           log2("......eventRequest1.addThreadFilter(thread1);");
361                           log2("        no Exception expected");
362                           ((ThreadDeathRequest)eventRequest1).addThreadFilter(thread1);
363                       } catch ( Exception e ) {
364                           log3("ERROR: Exception : " + e);
365                           testExitCode = FAILED;
366                       }
367                       break;
368 
369               case 1:
370                       eventRequest2 = setting2ThreadDeathRequest (null,
371                                              EventRequest.SUSPEND_NONE, property2);
372 
373                       try {
374                           log2("......eventRequest2.addThreadFilter(thread1);");
375                           log2("        no Exception expected");
376                           ((ThreadDeathRequest)eventRequest2).addThreadFilter(thread1);
377                       } catch ( Exception e ) {
378                           log3("ERROR: Exception : " + e);
379                           testExitCode = FAILED;
380                       }
381                       break;
382 
383               case 2:
384                       eventRequest3 = setting2ThreadDeathRequest (null,
385                                              EventRequest.SUSPEND_NONE, property3);
386 
387                       try {
388                           log2("......eventRequest3.addThreadFilter(thread1);");
389                           log2("        no Exception expected");
390                           ((ThreadDeathRequest)eventRequest3).addThreadFilter(thread1);
391                       } catch ( Exception e ) {
392                           log3("ERROR: Exception : " + e);
393                           testExitCode = FAILED;
394                       }
395                       break;
396 
397 
398               default:
399                       throw new JDITestRuntimeException("** default case 2 **");
400             }
401 
402             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
403         }
404         log1("    TESTING ENDS");
405         return;

408    /*
409     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
410     *                                             String, String, String)
411     *
412     * It sets up a breakpoint at given line number within a given method in a given class
413     * for a given thread.
414     *
415     * Return value: BreakpointRequest object  in case of success
416     *
417     * JDITestRuntimeException   in case of an Exception thrown within the method
418     */
419 
420     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
421                                                   ReferenceType testedClass,
422                                                   String methodName,
423                                                   String bpLine,
424                                                   String property)
425             throws JDITestRuntimeException {
426 
427         log2("......setting up a breakpoint:");
428         log2("       thread: " + thread + "; class: " + testedClass +
429                         "; method: " + methodName + "; line: " + bpLine);
430 
431         List              alllineLocations = null;
432         Location          lineLocation     = null;
433         BreakpointRequest breakpRequest    = null;
434 
435         try {
436             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
437 
438             alllineLocations = method.allLineLocations();
439 
440             int n =
441                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
442             if (n > alllineLocations.size()) {
443                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
444             } else {
445                 lineLocation = (Location) alllineLocations.get(n);
446                 try {
447                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
448                     breakpRequest.putProperty("number", property);
449                     breakpRequest.addThreadFilter(thread);
450                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
451                 } catch ( Exception e1 ) {
452                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
453                     breakpRequest    = null;
454                 }
455             }
456         } catch ( Exception e2 ) {
457             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
458             breakpRequest    = null;
459         }
460 
461         if (breakpRequest == null) {
462             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
463             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
464         }
465 
466         log2("      a breakpoint has been set up");
467         return breakpRequest;
468     }

471     private void getEventSet()
472                  throws JDITestRuntimeException {
473         try {
474 //            log2("       eventSet = eventQueue.remove(waitTime);");
475             eventSet = eventQueue.remove(waitTime);
476             if (eventSet == null) {
477                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
478             }
479 //            log2("       eventIterator = eventSet.eventIterator;");
480             eventIterator = eventSet.eventIterator();
481         } catch ( Exception e ) {
482             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
483         }
484     }

487     private void breakpointForCommunication()
488                  throws JDITestRuntimeException {
489 
490         log2("breakpointForCommunication");
491         getEventSet();
492 
493         if (eventIterator.nextEvent() instanceof BreakpointEvent)
494             return;
495 
496         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
497     }

501     private ThreadDeathRequest setting2ThreadDeathRequest( ThreadReference thread,
502                                                            int             suspendPolicy,
503                                                            String          property)
504             throws JDITestRuntimeException {
505         try {
506             ThreadDeathRequest tsr = eventRManager.createThreadDeathRequest();
507             if (thread != null)
508                 tsr.addThreadFilter(thread);
509             tsr.addCountFilter(1);
510             tsr.setSuspendPolicy(suspendPolicy);
511             tsr.putProperty("number", property);
512             return tsr;
513         } catch ( Exception e ) {
514             log3("ERROR: ATTENTION: Exception within settingThreadDeathRequest() : " + e);
515             log3("       ThreadDeathRequest HAS NOT BEEN SET UP");
516             throw new JDITestRuntimeException("** FAILURE to set up ThreadDeathRequest **");
517         }
518     }
