 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.notifyCaught()</code>     <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a boolean value, an argument of the method        <BR>
 46  *    EventRequestManager.createExceptionRequest(),                     <BR>
 47  * is equal to one returned by the method                               <BR>
 48  *    ExceptionRequest.notifyCaught()                                   <BR>
 49  * The cases to check include both true and false.                      <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger                                                       <BR>
 68  *   - sets up two ExceptionRequests with true and false values of      <BR>
 69  *     the notifyCaught argument,                                       <BR>
 70  *   - gets the values of the Request with the method                   <BR>
 71  *     StepRequest.notifyCaught(), and                                  <BR>
 72  *   - compares the values.                                             <BR>
 73  * <BR>
 74  * In third phase, at the end of the test, the debuggee changes         <BR>
 75  * the value of the "instruction" which the debugger and debuggee       <BR>
 76  * use to inform each other of needed actions,  and both end.           <BR>
 77  * <BR>
 78  */
 79

152     private int runThis (String argv[], PrintStream out) {
153 
154         argsHandler     = new ArgumentHandler(argv);
155         logHandler      = new Log(out, argsHandler);
156         Binder binder   = new Binder(argsHandler, logHandler);
157 
158         waitTime        = argsHandler.getWaitTime() * 60000;
159 
160         try {
161             log2("launching a debuggee :");
162             log2("       " + debuggeeName);
163             if (argsHandler.verbose()) {
164                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
165             } else {
166                 debuggee = binder.bindToDebugee(debuggeeName);
167             }
168             if (debuggee == null) {
169                 log3("ERROR: no debuggee launched");
170                 return FAILED;
171             }
172             log2("debuggee launched");
173         } catch ( Exception e ) {
174             log3("ERROR: Exception : " + e);
175             log2("       test cancelled");
176             return FAILED;
177         }
178 
179         debuggee.redirectOutput(logHandler);
180 
181         vm = debuggee.VM();
182 
183         eventQueue = vm.eventQueue();
184         if (eventQueue == null) {
185             log3("ERROR: eventQueue == null : TEST ABORTED");
186             vm.exit(PASS_BASE);
187             return FAILED;
188         }
189 
190         log2("invocation of the method runTest()");
191         switch (runTest()) {
192 
193             case 0 :  log2("test phase has finished normally");
194                       log2("   waiting for the debuggee to finish ...");
195                       debuggee.waitFor();
196 
197                       log2("......getting the debuggee's exit status");
198                       int status = debuggee.getStatus();
199                       if (status != PASS_BASE) {
200                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
201                               status + " != PASS_BASE");
202                           testExitCode = FAILED;
203                       } else {
204                           log2("......debuggee returned expected exit status: " +
205                               status + " == PASS_BASE");
206                       }
207                       break;
208 
209             default : log3("ERROR: runTest() returned unexpected value");
210 
211             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
212                       log2("......forcing: vm.exit();");
213                       testExitCode = FAILED;
214                       try {
215                           vm.exit(PASS_BASE);
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : " + e);
218                       }
219                       break;
220 
221             case 2 :  log3("test cancelled due to VMDisconnectedException");
222                       log2("......trying: vm.process().destroy();");
223                       testExitCode = FAILED;
224                       try {
225                           Process vmProcess = vm.process();
226                           if (vmProcess != null) {
227                               vmProcess.destroy();
228                           }
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : " + e);
231                       }
232                       break;
233             }
234 
235         return testExitCode;
236     }

239    /*
240     * Return value: 0 - normal end of the test
241     *               1 - ubnormal end of the test
242     *               2 - VMDisconnectedException while test phase
243     */
244 
245     private int runTest() {
246 
247         try {
248             testRun();
249 
250             log2("waiting for VMDeathEvent");
251             getEventSet();
252             if (eventIterator.nextEvent() instanceof VMDeathEvent)
253                 return 0;
254 
255             log3("ERROR: last event is not the VMDeathEvent");
256             return 1;
257         } catch ( VMDisconnectedException e ) {
258             log3("ERROR: VMDisconnectedException : " + e);
259             return 2;
260         } catch ( Exception e ) {
261             log3("ERROR: Exception : " + e);
262             return 1;
263         }
264 
265     }

267     private void testRun()
268                  throws JDITestRuntimeException, Exception {
269 
270         eventRManager = vm.eventRequestManager();
271 
272         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
273         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
274         cpRequest.addClassFilter(debuggeeName);
275 
276         cpRequest.enable();
277         vm.resume();
278         getEventSet();
279         cpRequest.disable();
280 
281         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
282         debuggeeClass = event.referenceType();
283 
284         if (!debuggeeClass.name().equals(debuggeeName))
285            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
286 
287         log2("      received: ClassPrepareEvent for debuggeeClass");
288 
289         String bPointMethod = "methodForCommunication";
290         String lineForComm  = "lineForComm";
291 
292         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
293 
294         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
295                                              debuggeeClass,
296                                             bPointMethod, lineForComm, "zero");
297         bpRequest.enable();
298 
299     //------------------------------------------------------  testing section
300 
301         log1("     TESTING BEGINS");
302 
303         EventRequest  eventRequest1 = null;
304         EventRequest  eventRequest2 = null;
305 
306         boolean bool;
307 
308 
309         for (int i = 0; ; i++) {
310 
311             vm.resume();
312 
313             log1(":::::: case: # " + i);
314 
315             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
316 
317             switch (i) {
318 
319               case 0:
320 
321 
322                      log2("......setting up: eventRequest1 = setting24ExceptionRequest(true,  true);");
323                      eventRequest1 = setting24ExceptionRequest(true,  true);
324 
325                      log2("......getting: bool = ((ExceptionRequest) eventRequest1).notifyCaught();");
326                      bool = ((ExceptionRequest) eventRequest1).notifyCaught();
327 
328                      log2("      compareing boolean values");
329                      if (bool != true) {
330                          testExitCode = FAILED;
331                          log3("ERROR: values are not equal : true and false");
332                      }
333 
334                      log2("......setting up: eventRequest2 = setting24ExceptionRequest(false,  true);");
335                      eventRequest2 = setting24ExceptionRequest(false,  true);
336 
337                      log2("......getting: bool = ((ExceptionRequest) eventRequest2).notifyCaught();");
338                      bool = ((ExceptionRequest) eventRequest2).notifyCaught();
339 
340                      log2("      compareing boolean values");
341                      if (bool != false) {
342                          testExitCode = FAILED;
343                          log3("ERROR: values are not equal : true and false");
344                      }
345 
346                       break;
347 
348               default:
349                       throw new JDITestRuntimeException("** default case 2 **");
350             }
351 
352             breakpointForCommunication();
353 
354             int instruction = ((IntegerValue)
355                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
356 
357             if (instruction == 0) {
358                 vm.resume();
359                 break;
360             }
361 
362             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
363         }
364         log1("    TESTING ENDS");
365         return;
366     }

368    /*
369     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
370     *                                             String, String, String)
371     *
372     * It sets up a breakpoint at given line number within a given method in a given class
373     * for a given thread.
374     *
375     * Return value: BreakpointRequest object  in case of success
376     *
377     * JDITestRuntimeException   in case of an Exception thrown within the method
378     */
379 
380     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
381                                                   ReferenceType testedClass,
382                                                   String methodName,
383                                                   String bpLine,
384                                                   String property)
385             throws JDITestRuntimeException {
386 
387         log2("......setting up a breakpoint:");
388         log2("       thread: " + thread + "; class: " + testedClass +
389                         "; method: " + methodName + "; line: " + bpLine);
390 
391         List              alllineLocations = null;
392         Location          lineLocation     = null;
393         BreakpointRequest breakpRequest    = null;
394 
395         try {
396             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
397 
398             alllineLocations = method.allLineLocations();
399 
400             int n =
401                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
402             if (n > alllineLocations.size()) {
403                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
404             } else {
405                 lineLocation = (Location) alllineLocations.get(n);
406                 try {
407                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
408                     breakpRequest.putProperty("number", property);
409                     breakpRequest.addThreadFilter(thread);
410                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
411                 } catch ( Exception e1 ) {
412                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
413                     breakpRequest    = null;
414                 }
415             }
416         } catch ( Exception e2 ) {
417             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
418             breakpRequest    = null;
419         }
420 
421         if (breakpRequest == null) {
422             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
423             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
424         }
425 
426         log2("      a breakpoint has been set up");
427         return breakpRequest;
428     }

431     private void getEventSet()
432                  throws JDITestRuntimeException {
433         try {
434 //            log2("       eventSet = eventQueue.remove(waitTime);");
435             eventSet = eventQueue.remove(waitTime);
436             if (eventSet == null) {
437                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
438             }
439 //            log2("       eventIterator = eventSet.eventIterator;");
440             eventIterator = eventSet.eventIterator();
441         } catch ( Exception e ) {
442             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
443         }
444     }

447     private void breakpointForCommunication()
448                  throws JDITestRuntimeException {
449 
450         log2("breakpointForCommunication");
451         getEventSet();
452 
453         if (eventIterator.nextEvent() instanceof BreakpointEvent)
454             return;
455 
456         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
457     }
