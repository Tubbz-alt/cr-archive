374     private ThreadReference threadByName(String name)
375                  throws JDITestRuntimeException {
376 
377         List         all = vm.allThreads();
378         ListIterator li  = all.listIterator();
379 
380         for (; li.hasNext(); ) {
381             ThreadReference thread = (ThreadReference) li.next();
382             if (thread.name().equals(name))
383                 return thread;
384         }
385         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
386     }
