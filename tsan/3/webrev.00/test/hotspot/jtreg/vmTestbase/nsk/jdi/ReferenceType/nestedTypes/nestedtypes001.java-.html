 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.nestedTypes()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Cases for testing include only ClassTypes and InterfaceTypes.<BR>
 46  * <BR>
 47  * The test works as follows.                                   <BR>
 48  * Upon launching debuggee's VM which will be suspended,        <BR>
 49  * a debugger waits for the VMStartEvent within a predefined    <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.<BR>
 51  * Upon getting the VMStartEvent, it makes the request          <BR>
 52  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,  <BR>
 53  * resumes the VM, and waits for the event within the predefined<BR>
 54  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 55  * Upon getting the ClassPrepareEvent,                          <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,<BR>
 57  * the debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 58  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 59  * to be suspended and to inform the debugger with the event.           <BR>
 60  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 61  * At the end, the debuggee changes the value of the "instruction"      <BR>
 62  * to inform the debugger of checks finished, and both end.             <BR>
 63  */
 64

139     private int runThis (String argv[], PrintStream out) {
140 
141         argsHandler     = new ArgumentHandler(argv);
142         logHandler      = new Log(out, argsHandler);
143         Binder binder   = new Binder(argsHandler, logHandler);
144 
145         waitTime        = argsHandler.getWaitTime() * 60000;
146 
147         try {
148             log2("launching a debuggee :");
149             log2("       " + debuggeeName);
150             if (argsHandler.verbose()) {
151                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
152             } else {
153                 debuggee = binder.bindToDebugee(debuggeeName);
154             }
155             if (debuggee == null) {
156                 log3("ERROR: no debuggee launched");
157                 return FAILED;
158             }
159             log2("debuggee launched");
160         } catch ( Exception e ) {
161             log3("ERROR: Exception : " + e);
162             log2("       test cancelled");
163             return FAILED;
164         }
165 
166         debuggee.redirectOutput(logHandler);
167 
168         vm = debuggee.VM();
169 
170         eventQueue = vm.eventQueue();
171         if (eventQueue == null) {
172             log3("ERROR: eventQueue == null : TEST ABORTED");
173             vm.exit(PASS_BASE);
174             return FAILED;
175         }
176 
177         log2("invocation of the method runTest()");
178         switch (runTest()) {
179 
180             case 0 :  log2("test phase has finished normally");
181                       log2("   waiting for the debuggee to finish ...");
182                       debuggee.waitFor();
183 
184                       log2("......getting the debuggee's exit status");
185                       int status = debuggee.getStatus();
186                       if (status != PASS_BASE) {
187                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
188                               status + " != PASS_BASE");
189                           testExitCode = FAILED;
190                       } else {
191                           log2("......debuggee returned expected exit status: " +
192                               status + " == PASS_BASE");
193                       }
194                       break;
195 
196             default : log3("ERROR: runTest() returned unexpected value");
197 
198             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
199                       log2("......forcing: vm.exit();");
200                       testExitCode = FAILED;
201                       try {
202                           vm.exit(PASS_BASE);
203                       } catch ( Exception e ) {
204                           log3("ERROR: Exception : e");
205                       }
206                       break;
207 
208             case 2 :  log3("test cancelled due to VMDisconnectedException");
209                       log2("......trying: vm.process().destroy();");
210                       testExitCode = FAILED;
211                       try {
212                           Process vmProcess = vm.process();
213                           if (vmProcess != null) {
214                               vmProcess.destroy();
215                           }
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : e");
218                       }
219                       break;
220             }
221 
222         return testExitCode;
223     }

226    /*
227     * Return value: 0 - normal end of the test
228     *               1 - ubnormal end of the test
229     *               2 - VMDisconnectedException while test phase
230     */
231 
232     private int runTest() {
233 
234         try {
235             testRun();
236 
237             log2("waiting for VMDeathEvent");
238             getEventSet();
239             if (eventIterator.nextEvent() instanceof VMDeathEvent)
240                 return 0;
241 
242             log3("ERROR: last event is not the VMDeathEvent");
243             return 1;
244         } catch ( VMDisconnectedException e ) {
245             log3("ERROR: VMDisconnectedException : " + e);
246             return 2;
247         } catch ( Exception e ) {
248             log3("ERROR: Exception : " + e);
249             return 1;
250         }
251 
252     }

254     private void testRun()
255                  throws JDITestRuntimeException, Exception {
256 
257         eventRManager = vm.eventRequestManager();
258 
259         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
260         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
261         cpRequest.addClassFilter(debuggeeName);
262 
263         cpRequest.enable();
264         vm.resume();
265         getEventSet();
266         cpRequest.disable();
267 
268         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
269         debuggeeClass = event.referenceType();
270 
271         if (!debuggeeClass.name().equals(debuggeeName))
272            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
273 
274         log2("      received: ClassPrepareEvent for debuggeeClass");
275 
276         String bPointMethod = "methodForCommunication";
277         String lineForComm  = "lineForComm";
278         BreakpointRequest bpRequest;
279 
280         try {
281             bpRequest = settingBreakpoint(threadByName("main"),
282                                           debuggeeClass,
283                                           bPointMethod, lineForComm, "zero");
284         } catch ( Exception e ) {
285             throw e;
286         }
287         bpRequest.enable();
288 
289     //------------------------------------------------------  testing section
290 
291         log1("     TESTING BEGINS");
292 
293         for (int i = 0; ; i++) {
294 
295             vm.resume();
296             breakpointForCommunication();
297 
298             int instruction = ((IntegerValue)
299                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
300 
301             if (instruction == 0) {
302                 vm.resume();
303                 break;
304             }
305 
306             log1("  new check: # " + i);
307 
308             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
309 
310 
311            String names1[] = {
312 
313                       ".nestedtypes001aTestClass$NestedClass",
314                       ".nestedtypes001aTestClass$NestedClass[]",
315 
316                       ".nestedtypes001aTestClass$NestedIface",
317                       ".nestedtypes001aTestClass$NestedIface[]",
318 
319 
320                       ".nestedtypes001aTestClass$StaticNestedClass",
321                       ".nestedtypes001aTestClass$StaticNestedClass[]"
322 
323                       };
324 
325             String typeForCheck = mName + ".nestedtypes001aTestClass";
326             log2("......typeForCheck: " + typeForCheck);
327 
328             log2("      getting: List classList = vm.classesByName(typeForCheck);");
329             List classes = vm.classesByName(typeForCheck);
330             if (classes.size() != 1) {
331                 log3("ERROR: classes.size() != 1  : " + classes.size());
332                 testExitCode = FAILED;
333                 continue;
334             }
335 
336             log2("      getting: ReferenceType testedType  = (ReferenceType) classList.get(0);");
337             ReferenceType testedType  = (ReferenceType) classes.get(0);
338 
339             log2("      getting: List netsedTypes = testedType.nestedTypes();");
340             List nestedTypes = testedType.nestedTypes();
341             if (nestedTypes.size() != names1.length) {
342                 log3("ERROR: nestedTypes.size() != names1.length : " + nestedTypes.size());
343                 testExitCode = FAILED;
344                 continue;
345             }
346 
347 
348             log2("----- Cases for testing: ReferenceTypes");
349             for (int i1 = 0; i1 < names1.length; i1++) {
350 
351                 typeForCheck = mName + names1[i1];
352                 log2("......typeForCheck: " + typeForCheck);
353 
354                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
355                 List classList = vm.classesByName(typeForCheck);
356                 if (classList.size() != 1) {
357                     log3("ERROR: classList.size() != 1  : " + classList.size());
358                     testExitCode = FAILED;
359                     continue;
360                 }
361 
362                 log2("      getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
363                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
364 
365                 boolean equals = false;
366 
367                 ListIterator li = nestedTypes.listIterator();
368                 for (; li.hasNext(); ) {
369                     if (li.next().equals(referenceType)) {
370                         equals = true;
371                         break;
372                     }
373                 }
374 
375                 if (!equals) {
376                     log3("ERROR: referenceType is not in the List: " + typeForCheck);
377                     testExitCode = FAILED;
378                 }
379             }
380 
381             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
382         }
383         log1("    TESTING ENDS");
384         return;
385     }

387     private ThreadReference threadByName(String name)
388                  throws JDITestRuntimeException {
389 
390         List         all = vm.allThreads();
391         ListIterator li  = all.listIterator();
392 
393         for (; li.hasNext(); ) {
394             ThreadReference thread = (ThreadReference) li.next();
395             if (thread.name().equals(name))
396                 return thread;
397         }
398         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
399     }

401    /*
402     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
403     *                                             String, String, String)
404     *
405     * It sets up a breakpoint at given line number within a given method in a given class
406     * for a given thread.
407     *
408     * Return value: BreakpointRequest object  in case of success
409     *
410     * JDITestRuntimeException   in case of an Exception thrown within the method
411     */
412 
413     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
414                                                   ReferenceType testedClass,
415                                                   String methodName,
416                                                   String bpLine,
417                                                   String property)
418             throws JDITestRuntimeException {
419 
420         log2("......setting up a breakpoint:");
421         log2("       thread: " + thread + "; class: " + testedClass +
422                         "; method: " + methodName + "; line: " + bpLine);
423 
424         List              alllineLocations = null;
425         Location          lineLocation     = null;
426         BreakpointRequest breakpRequest    = null;
427 
428         try {
429             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
430 
431             alllineLocations = method.allLineLocations();
432 
433             int n =
434                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
435             if (n > alllineLocations.size()) {
436                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
437             } else {
438                 lineLocation = (Location) alllineLocations.get(n);
439                 try {
440                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
441                     breakpRequest.putProperty("number", property);
442                     breakpRequest.addThreadFilter(thread);
443                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
444                 } catch ( Exception e1 ) {
445                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
446                     breakpRequest    = null;
447                 }
448             }
449         } catch ( Exception e2 ) {
450             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
451             breakpRequest    = null;
452         }
453 
454         if (breakpRequest == null) {
455             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
456             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
457         }
458 
459         log2("      a breakpoint has been set up");
460         return breakpRequest;
461     }

464     private void getEventSet()
465                  throws JDITestRuntimeException {
466         try {
467 //            log2("       eventSet = eventQueue.remove(waitTime);");
468             eventSet = eventQueue.remove(waitTime);
469             if (eventSet == null) {
470                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
471             }
472 //            log2("       eventIterator = eventSet.eventIterator;");
473             eventIterator = eventSet.eventIterator();
474         } catch ( Exception e ) {
475             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
476         }
477     }

480     private void breakpointForCommunication()
481                  throws JDITestRuntimeException {
482 
483         log2("breakpointForCommunication");
484         getEventSet();
485 
486         if (eventIterator.nextEvent() instanceof BreakpointEvent)
487             return;
488 
489         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
490     }
