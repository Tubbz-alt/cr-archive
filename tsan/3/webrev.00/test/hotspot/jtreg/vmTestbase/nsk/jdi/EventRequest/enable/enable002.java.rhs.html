 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.enable()</code>               <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request has been deleted.                 <BR>
 48  *            IllegalThreadStateException -                     <BR>
 49  *            if this is a StepRequest and                      <BR>
 50  *            the thread named in the request has died.         <BR>
 51  *  The cases to test include all sub-types of EventRequest.    <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee prepares new check case (as if) and invokes           <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent,                                  <BR>
 74  *   the debugger performs the check case required.                     <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

246    /*
247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase
250     */

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308 
309         EventRequest  eventRequest1 = null;
310 
311         String        fieldName1    = "var101";
312         String        fieldName2    = "excObj";
313 
314         ThreadReference thread1     = null;
315         String          threadName1 = "thread1";
316 
317         ReferenceType testClassReference = null;
318 
319 
320         log1("     TESTING BEGINS");
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339 
340             switch (i) {
341 
342               case 0:
343                      thread1 = debuggee.threadByNameOrThrow(threadName1);
344 
345                      log2("......setting up StepRequest");
346                      eventRequest1 = eventRManager.createStepRequest
347                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
348                      vm.resume();
349                      breakpointForCommunication();
350                      break;
351 
352               case 1:
353                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
354 
355                      log2("......setting up AccessWatchpointRequest");
356                      eventRequest1 = eventRManager.createAccessWatchpointRequest
357                                     (testClassReference.fieldByName(fieldName1));
358                      break;
359 
360               case 2:
361                      log2(".....setting up ModificationWatchpointRequest");
362                      eventRequest1 = eventRManager.createModificationWatchpointRequest
363                                     (testClassReference.fieldByName(fieldName1));
364                      break;
365 
366               case 3:
367                      log2(".....setting up ClassPrepareRequest");
368                      eventRequest1 = eventRManager.createClassPrepareRequest();
369                      break;
370 
371               case 4:
372                      log2(".....setting up ClassUnloadRequest");
373                      eventRequest1 = eventRManager.createClassUnloadRequest();
374                      break;
375 
376               case 5:
377                      log2(".....setting up MethodEntryRequest");
378                      eventRequest1 = eventRManager.createMethodEntryRequest();
379                      break;
380 
381               case 6:
382                      log2(".....setting up MethodExitRequest");
383                      eventRequest1 = eventRManager.createMethodExitRequest();
384                      break;
385 
386               case 7:
387                      log2(".....setting up ThreadDeathRequest");
388                      eventRequest1 = eventRManager.createThreadDeathRequest();
389                      break;
390 
391               case 8:
392                      log2(".....setting up ThreadStartRequest");
393                      eventRequest1 = eventRManager.createThreadStartRequest();
394                      break;
395 
396               case 9:
397                      log2(".....setting up VMDeathRequest");
398                      eventRequest1 = eventRManager.createVMDeathRequest();
399                      break;
400 
401               case 10:
402                      log2(".....setting up ExceptionRequest");
403                      eventRequest1 = eventRManager.createExceptionRequest(
404                                      (ReferenceType)
405                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
406                                      true, true );
407                      break;
408 
409               case 11:
410                      log2(".....setting up BreakpointRequest");
411                      eventRequest1 = eventRManager.createBreakpointRequest(location);
412                      break;
413 
414 
415               default:
416                       throw new JDITestRuntimeException("** default case 2 **");
417             }
418 
419             if (eventRequest1 instanceof StepRequest) {
420                 try {
421                     log2("......eventRequest1.enable(); IllegalThreadStateException is expected");
422                     eventRequest1.enable();
423                     testExitCode = FAILED;
424                     log3("ERROR:  NO  IllegalThreadStateException for StepRequest");
425                 } catch ( IllegalThreadStateException e ) {
426                     log2("       IllegalThreadStateException");
427                 }
428             }
429 
430             log2("......eventRManager.deleteEventRequest(eventRequest1);");
431             eventRManager.deleteEventRequest(eventRequest1);
432             try {
433                 log2("......eventRequest1.enable();  InvalidRequestStateException is expected");
434                 eventRequest1.enable();
435                 testExitCode = FAILED;
436                 log3("ERROR:  NO  InvalidRequestStateException");
437             } catch ( InvalidRequestStateException e ) {
438                     log2("       InvalidRequestStateException");
439             }
440 
441             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
442         }
443         log1("    TESTING ENDS");
444         return;
445     }

447    /*
448     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
449     *                                             String, String, String)
450     *
451     * It sets up a breakpoint at given line number within a given method in a given class
452     * for a given thread.
453     *
454     * Return value: BreakpointRequest object  in case of success
455     *
456     * JDITestRuntimeException   in case of an Exception thrown within the method
457     */
458 
459     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
460                                                   ReferenceType testedClass,
461                                                   String methodName,
462                                                   String bpLine,
463                                                   String property)
464             throws JDITestRuntimeException {
465 
466         log2("......setting up a breakpoint:");
467         log2("       thread: " + thread + "; class: " + testedClass +
468                         "; method: " + methodName + "; line: " + bpLine);
469 
470         List              alllineLocations = null;
471         Location          lineLocation     = null;
472         BreakpointRequest breakpRequest    = null;
473 
474         try {
475             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
476 
477             alllineLocations = method.allLineLocations();
478 
479             int n =
480                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
481             if (n > alllineLocations.size()) {
482                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
483             } else {
484                 lineLocation = (Location) alllineLocations.get(n);
485 
486 // this is only for this test to get Location object
487 location = lineLocation;
488 
489                 try {
490                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
491                     breakpRequest.putProperty("number", property);
492                     breakpRequest.addThreadFilter(thread);
493                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
494                 } catch ( Exception e1 ) {
495                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
496                     breakpRequest    = null;
497                 }
498             }
499         } catch ( Exception e2 ) {
500             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
501             breakpRequest    = null;
502         }
503 
504         if (breakpRequest == null) {
505             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
506             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
507         }
508 
509         log2("      a breakpoint has been set up");
510         return breakpRequest;
511     }

514     private void getEventSet()
515                  throws JDITestRuntimeException {
516         try {
517 //            log2("       eventSet = eventQueue.remove(waitTime);");
518             eventSet = eventQueue.remove(waitTime);
519             if (eventSet == null) {
520                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
521             }
522 //            log2("       eventIterator = eventSet.eventIterator;");
523             eventIterator = eventSet.eventIterator();
524         } catch ( Exception e ) {
525             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
526         }
527     }

530     private void breakpointForCommunication()
531                  throws JDITestRuntimeException {
532 
533         log2("breakpointForCommunication");
534         getEventSet();
535 
536         if (eventIterator.nextEvent() instanceof BreakpointEvent)
537             return;
538 
539         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
540     }
