 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.addClassFilter(ReferenceType)</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates a filter_rt002aTestClass10 object, to use as a filter,        <BR>
 68  *   and invokes the methodForCommunication to be suspended and         <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an StepRequest and enables it,                           <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */
 84

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest  eventRequest1      = null;
318 
319         String        property1          = "StepRequest1";
320 
321         ReferenceType testClassReference = null;
322 
323         ThreadReference thread1     = null;
324         String          threadName1 = "thread1";
325 
326 
327         for (int i = 0; ; i++) {
328 
329             vm.resume();
330             breakpointForCommunication();
331 
332             int instruction = ((IntegerValue)
333                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
334 
335             if (instruction == 0) {
336                 vm.resume();
337                 break;
338             }
339 
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             switch (i) {
346 
347               case 0:
348                      testClassReference =
349                           (ReferenceType) vm.classesByName(testedClassName).get(0);
350 
351                      thread1 = threadByName(threadName1);
352 
353                      eventRequest1 = setting21StepRequest(thread1, testClassReference,
354                                               EventRequest.SUSPEND_NONE, property1);
355                      eventRequest1.enable();
356 
357                      try {
358                          log2("......eventRequest1.addClassFilter(testClassReference);");
359                          log2("        InvalidRequestStateException expected");
360                          ((StepRequest)eventRequest1).addClassFilter(testClassReference);
361                          log3("ERROR: no InvalidRequestStateException");
362                          testExitCode = FAILED;
363                      } catch ( InvalidRequestStateException e ) {
364                          log2("        InvalidRequestStateException");
365                      }
366 
367                      log2("......eventRequest1.disable();");
368                      eventRequest1.disable();
369                      try {
370                          log2("......eventRequest1.addClassFilter(testClassReference);");
371                          log2("        no InvalidRequestStateException expected");
372                          ((StepRequest)eventRequest1).addClassFilter(testClassReference);
373                          log2("        no InvalidRequestStateException");
374                      } catch ( InvalidRequestStateException e ) {
375                          log3("ERROR: InvalidRequestStateException");
376                          testExitCode = FAILED;
377                      }
378 
379                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
380                      eventRManager.deleteEventRequest(eventRequest1);
381                      try {
382                          log2("......eventRequest1.addClassFilter(testClassReference);");
383                          log2("        InvalidRequestStateException expected");
384                          ((StepRequest)eventRequest1).addClassFilter(testClassReference);
385                          log3("ERROR: no InvalidRequestStateException");
386                          testExitCode = FAILED;
387                      } catch ( InvalidRequestStateException e ) {
388                          log2("        InvalidRequestStateException");
389                      }
390                       break;
391 
392               default:
393                       throw new JDITestRuntimeException("** default case 2 **");
394             }
395 
396             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
397         }
398         log1("    TESTING ENDS");
399         return;
400     }

402     private ThreadReference threadByName(String name)
403                  throws JDITestRuntimeException {
404 
405         List         all = vm.allThreads();
406         ListIterator li  = all.listIterator();
407 
408         for (; li.hasNext(); ) {
409             ThreadReference thread = (ThreadReference) li.next();
410             if (thread.name().equals(name))
411                 return thread;
412         }
413         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
414     }

416    /*
417     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
418     *                                             String, String, String)
419     *
420     * It sets up a breakpoint at given line number within a given method in a given class
421     * for a given thread.
422     *
423     * Return value: BreakpointRequest object  in case of success
424     *
425     * JDITestRuntimeException   in case of an Exception thrown within the method
426     */
427 
428     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
429                                                   ReferenceType testedClass,
430                                                   String methodName,
431                                                   String bpLine,
432                                                   String property)
433             throws JDITestRuntimeException {
434 
435         log2("......setting up a breakpoint:");
436         log2("       thread: " + thread + "; class: " + testedClass +
437                         "; method: " + methodName + "; line: " + bpLine);
438 
439         List              alllineLocations = null;
440         Location          lineLocation     = null;
441         BreakpointRequest breakpRequest    = null;
442 
443         try {
444             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
445 
446             alllineLocations = method.allLineLocations();
447 
448             int n =
449                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
450             if (n > alllineLocations.size()) {
451                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
452             } else {
453                 lineLocation = (Location) alllineLocations.get(n);
454                 try {
455                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
456                     breakpRequest.putProperty("number", property);
457                     breakpRequest.addThreadFilter(thread);
458                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
459                 } catch ( Exception e1 ) {
460                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
461                     breakpRequest    = null;
462                 }
463             }
464         } catch ( Exception e2 ) {
465             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
466             breakpRequest    = null;
467         }
468 
469         if (breakpRequest == null) {
470             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
471             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
472         }
473 
474         log2("      a breakpoint has been set up");
475         return breakpRequest;
476     }

479     private void getEventSet()
480                  throws JDITestRuntimeException {
481         try {
482 //            log2("       eventSet = eventQueue.remove(waitTime);");
483             eventSet = eventQueue.remove(waitTime);
484             if (eventSet == null) {
485                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
486             }
487 //            log2("       eventIterator = eventSet.eventIterator;");
488             eventIterator = eventSet.eventIterator();
489         } catch ( Exception e ) {
490             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
491         }
492     }

495     private void breakpointForCommunication()
496                  throws JDITestRuntimeException {
497 
498         log2("breakpointForCommunication");
499         getEventSet();
500 
501         if (eventIterator.nextEvent() instanceof BreakpointEvent)
502             return;
503 
504         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
505     }

509     private StepRequest setting21StepRequest ( ThreadReference thread,
510                                                ReferenceType   testedClass,
511                                                int             suspendPolicy,
512                                                String          property        )
513             throws JDITestRuntimeException {
514         try {
515             log2("......setting up StepRequest:");
516             log2("       thread: " + thread + "; property: " + property);
517 
518             StepRequest
519             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_INTO);
520             str.putProperty("number", property);
521             str.addCountFilter(2);
522             str.setSuspendPolicy(suspendPolicy);
523             str.addClassFilter(testedClass);
524 
525             log2("      StepRequest has been set up");
526             return str;
527         } catch ( Exception e ) {
528             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
529             log3("       StepRequest HAS NOT BEEN SET UP");
530             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
531         }
