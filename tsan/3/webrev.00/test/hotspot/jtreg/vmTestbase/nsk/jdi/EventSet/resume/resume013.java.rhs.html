 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test case includes SUSPEND_ALL for VMDeathEvent set.         <BR>
 46  * <BR>
 47  * The test works as follows.                                   <BR>
 48  * <BR>
 49  * Upon launching debuggee's VM which will be suspended,                <BR>
 50  * a debugger waits for the VMStartEvent within a predefined            <BR>
 51  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 52  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 53  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 54  * and waits for the event within the predefined time interval.         <BR>
 55  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 56  * Upon getting the ClassPrepareEvent,                                  <BR>
 57  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 58  * within debuggee's special methodForCommunication().                  <BR>
 59  * <BR>
 60  * Then to check the above,                                             <BR>
 61  * the debugger and the debuggee perform the following.                 <BR>
 62  * - The debugger sets up a VMDeathRequest, resumes                     <BR>
 63  *   the debuggee, and waits for the VMDeathEvent.                      <BR>
 64  * - The debuggee ends to be resulting in the VMDeathEvent.             <BR>
 65  * - Upon getting the event, the debugger                               <BR>
 66  *   performs the check corresponding to the event.                     <BR>
 67  * <BR>
 68  * Note. To inform each other of needed actions, the debugger and       <BR>
 69  *       and the debuggee use debuggee's variable "instruction".        <BR>
 70  * <BR>
 71  */
 72

148     private int runThis (String argv[], PrintStream out) {
149 
150         argsHandler     = new ArgumentHandler(argv);
151         logHandler      = new Log(out, argsHandler);
152         Binder binder   = new Binder(argsHandler, logHandler);
153 
154         waitTime        = argsHandler.getWaitTime() * 60000;
155 
156         try {
157             log2("launching a debuggee :");
158             log2("       " + debuggeeName);
159             if (argsHandler.verbose()) {
160                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
161             } else {
162                 debuggee = binder.bindToDebugee(debuggeeName);
163             }
164             if (debuggee == null) {
165                 log3("ERROR: no debuggee launched");
166                 return FAILED;
167             }
168             log2("debuggee launched");
169         } catch ( Exception e ) {
170             log3("ERROR: Exception : " + e);
171             log2("       test cancelled");
172             return FAILED;
173         }
174 
175         debuggee.redirectOutput(logHandler);
176 
177         vm = debuggee.VM();
178 
179         eventQueue = vm.eventQueue();
180         if (eventQueue == null) {
181             log3("ERROR: eventQueue == null : TEST ABORTED");
182             vm.exit(PASS_BASE);
183             return FAILED;
184         }
185 
186         log2("invocation of the method runTest()");
187         switch (runTest()) {
188 
189             case 0 :  log2("test phase has finished normally");
190                       log2("   waiting for the debuggee to finish ...");
191                       debuggee.waitFor();
192 
193                       log2("......getting the debuggee's exit status");
194                       int status = debuggee.getStatus();
195                       if (status != PASS_BASE) {
196                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
197                               status + " != PASS_BASE");
198                           testExitCode = FAILED;
199                       } else {
200                           log2("......debuggee returned expected exit status: " +
201                               status + " == PASS_BASE");
202                       }
203                       break;
204 
205             default : log3("ERROR: runTest() returned unexpected value");
206 
207             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
208                       log2("......forcing: vm.exit();");
209                       testExitCode = FAILED;
210                       try {
211                           vm.exit(PASS_BASE);
212                       } catch ( Exception e ) {
213                           log3("ERROR: Exception : e");
214                       }
215                       break;
216 
217             case 2 :  log3("test cancelled due to VMDisconnectedException");
218                       log2("......trying: vm.process().destroy();");
219                       testExitCode = FAILED;
220                       try {
221                           Process vmProcess = vm.process();
222                           if (vmProcess != null) {
223                               vmProcess.destroy();
224                           }
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : e");
227                       }
228                       break;
229             }
230 
231         return testExitCode;
232     }

235    /*
236     * Return value: 0 - normal end of the test
237     *               1 - ubnormal end of the test
238     *               2 - VMDisconnectedException while test phase
239     */

241     private int runTest() {
242 
243         try {
244             testRun();
245 
246             log2("waiting for VMDeathEvent");
247             getEventSet();
248             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
249                 log3("ERROR: last event is not the VMDeathEvent");
250                 return 1;
251             }
252             check();
253 
254             log2("waiting for VMDisconnectEvent");
255             getEventSet();
256             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
257                 log3("ERROR: last event is not the VMDisconnectEvent");
258                 return 1;
259             }
260 
261             return 0;
262 
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         if ( !vm.canRequestVMDeathEvent() ) {
277             log2("......vm.canRequestVMDeathEvent == false :: test cancelled");
278             vm.exit(PASS_BASE);
279             return;
280         }
281 
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304         BreakpointRequest bpRequest;
305         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
306         bpRequest = settingBreakpoint(mainThread,
307                                       debuggeeClass,
308                                       bPointMethod, lineForComm, "zero");
309         bpRequest.enable();
310 
311         vm.resume();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317 //        EventRequest eventRequest1 = null;
318 //        EventRequest eventRequest2 = null;
319         EventRequest eventRequest3 = null;
320 
321         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
322         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
323         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
324 
325 
326         breakpointForCommunication();
327 
328             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
329 
330         eventRequest3 = settingVMDeathRequest ( SUSPEND_ALL, "VMDeathRequest3");
331         eventRequest3.enable();
332 
333         mainThread.resume();
334 
335             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
336 
337         log1("    TESTING ENDS");
338         return;
339     }

341    /*
342     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
343     *                                             String, String, String)
344     *
345     * It sets up a breakpoint at given line number within a given method in a given class
346     * for a given thread.
347     *
348     * Return value: BreakpointRequest object  in case of success
349     *
350     * JDITestRuntimeException   in case of an Exception thrown within the method
351     */
352 
353     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
354                                                   ReferenceType testedClass,
355                                                   String methodName,
356                                                   String bpLine,
357                                                   String property)
358             throws JDITestRuntimeException {
359 
360         log2("......setting up a breakpoint:");
361         log2("       thread: " + thread + "; class: " + testedClass +
362                         "; method: " + methodName + "; line: " + bpLine);
363 
364         List              alllineLocations = null;
365         Location          lineLocation     = null;
366         BreakpointRequest breakpRequest    = null;
367 
368         try {
369             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
370 
371             alllineLocations = method.allLineLocations();
372 
373             int n =
374                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
375             if (n > alllineLocations.size()) {
376                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
377             } else {
378                 lineLocation = (Location) alllineLocations.get(n);
379                 try {
380                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
381                     breakpRequest.putProperty("number", property);
382                     breakpRequest.addThreadFilter(thread);
383                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
384                 } catch ( Exception e1 ) {
385                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
386                     breakpRequest    = null;
387                 }
388             }
389         } catch ( Exception e2 ) {
390             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
391             breakpRequest    = null;
392         }
393 
394         if (breakpRequest == null) {
395             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
396             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
397         }
398 
399         log2("      a breakpoint has been set up");
400         return breakpRequest;
401     }

404     private void getEventSet()
405                  throws JDITestRuntimeException {
406         try {
407 //            log2("       eventSet = eventQueue.remove(waitTime);");
408             eventSet = eventQueue.remove(waitTime);
409             if (eventSet == null) {
410                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
411             }
412 //            log2("       eventIterator = eventSet.eventIterator;");
413             eventIterator = eventSet.eventIterator();
414         } catch ( Exception e ) {
415             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
416         }
417     }

420     private void breakpointForCommunication()
421                  throws JDITestRuntimeException {
422 
423         log2("breakpointForCommunication");
424         getEventSet();
425 
426         if (eventIterator.nextEvent() instanceof BreakpointEvent)
427             return;
428 
429         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
430     }

434     private VMDeathRequest settingVMDeathRequest( int    suspendPolicy,
435                                                   String property       )
436             throws JDITestRuntimeException {
437         try {
438             log2("......setting up VMDeathRequest:");
439             log2("      suspendPolicy: " + suspendPolicy + "; property: " + property);
440 
441             VMDeathRequest
442             vmdr = eventRManager.createVMDeathRequest();
443             vmdr.putProperty("number", property);
444             vmdr.setSuspendPolicy(suspendPolicy);
445 
446             return vmdr;
447         } catch ( Exception e ) {
448             log3("ERROR: ATTENTION: Exception within settingVMDeathRequest() : " + e);
449             log3("       VMDeathRequest HAS NOT BEEN SET UP");
450             throw new JDITestRuntimeException("** FAILURE to set up a VMDeathRequest **");
451         }
452     }

454     void check() {
455 
456         log2("......checking up on EventSet.resume()");
457         try {
458             log2("        eventSet.resume();  :: no Exception expected");
459             eventSet.resume();
460         } catch ( Exception e ) {
461             log3("ERROR: Exception : " + e);
462             testExitCode = FAILED;
463             log2("        vm.resume;");
464             vm.resume();
465         }
