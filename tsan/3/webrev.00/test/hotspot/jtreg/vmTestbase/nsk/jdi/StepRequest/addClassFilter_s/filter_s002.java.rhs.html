 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * StepRequest.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.StepRequest.addClassFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates and starts a thread, thread1, and             <BR>
 68  *   invokes the methodForCommunication to be suspended                 <BR>
 69  *   and to inform the debugger with the event.                         <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a StepRequest and enables it,                            <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         EventRequest  eventRequest1      = null;
312 
313         String        property1          = "StepRequest1";
314 
315 
316         for (int i = 0; ; i++) {
317 
318             vm.resume();
319             breakpointForCommunication();
320 
321             int instruction = ((IntegerValue)
322                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
323 
324             if (instruction == 0) {
325                 vm.resume();
326                 break;
327             }
328 
329 
330             log1(":::::: case: # " + i);
331 
332             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
333 
334             switch (i) {
335 
336               case 0:
337                      ThreadReference thread1 = debuggee.threadByNameOrThrow("thread1");
338                      eventRequest1 = setting23StepRequest(thread1, testedClassName1,
339                                               EventRequest.SUSPEND_NONE, property1);
340 
341                      eventRequest1.enable();
342 
343                      try {
344                          log2("......eventRequest1.addClassFilter(testedClassName1);");
345                          log2("        InvalidRequestStateException expected");
346                          ((StepRequest)eventRequest1).addClassFilter(testedClassName1);
347                          log3("ERROR: no InvalidRequestStateException");
348                          testExitCode = FAILED;
349                      } catch ( InvalidRequestStateException e ) {
350                          log2("        InvalidRequestStateException");
351                      }
352 
353                      log2("......eventRequest1.disable();");
354                      eventRequest1.disable();
355                      try {
356                          log2("......eventRequest1.addClassFilter(testedClassName1);");
357                          log2("        no InvalidRequestStateException expected");
358                          ((StepRequest)eventRequest1).addClassFilter(testedClassName1);
359                          log2("        no InvalidRequestStateException");
360                      } catch ( InvalidRequestStateException e ) {
361                          log3("ERROR: InvalidRequestStateException");
362                          testExitCode = FAILED;
363                      }
364 
365                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
366                      eventRManager.deleteEventRequest(eventRequest1);
367                      try {
368                          log2("......eventRequest1.addClassFilter(testedClassName1);");
369                          log2("        InvalidRequestStateException expected");
370                          ((StepRequest)eventRequest1).addClassFilter(testedClassName1);
371                          log3("ERROR: no InvalidRequestStateException");
372                          testExitCode = FAILED;
373                      } catch ( InvalidRequestStateException e ) {
374                          log2("        InvalidRequestStateException");
375                      }
376                       break;
377 
378               default:
379                       throw new JDITestRuntimeException("** default case 2 **");
380             }
381 
382             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
383         }
384         log1("    TESTING ENDS");
385         return;
386     }

388    /*
389     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
390     *                                             String, String, String)
391     *
392     * It sets up a breakpoint at given line number within a given method in a given class
393     * for a given thread.
394     *
395     * Return value: BreakpointRequest object  in case of success
396     *
397     * JDITestRuntimeException   in case of an Exception thrown within the method
398     */
399 
400     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
401                                                   ReferenceType testedClass,
402                                                   String methodName,
403                                                   String bpLine,
404                                                   String property)
405             throws JDITestRuntimeException {
406 
407         log2("......setting up a breakpoint:");
408         log2("       thread: " + thread + "; class: " + testedClass +
409                         "; method: " + methodName + "; line: " + bpLine);
410 
411         List              alllineLocations = null;
412         Location          lineLocation     = null;
413         BreakpointRequest breakpRequest    = null;
414 
415         try {
416             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
417 
418             alllineLocations = method.allLineLocations();
419 
420             int n =
421                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
422             if (n > alllineLocations.size()) {
423                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
424             } else {
425                 lineLocation = (Location) alllineLocations.get(n);
426                 try {
427                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
428                     breakpRequest.putProperty("number", property);
429                     breakpRequest.addThreadFilter(thread);
430                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
431                 } catch ( Exception e1 ) {
432                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
433                     breakpRequest    = null;
434                 }
435             }
436         } catch ( Exception e2 ) {
437             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
438             breakpRequest    = null;
439         }
440 
441         if (breakpRequest == null) {
442             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
443             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
444         }
445 
446         log2("      a breakpoint has been set up");
447         return breakpRequest;
448     }

451     private void getEventSet()
452                  throws JDITestRuntimeException {
453         try {
454 //            log2("       eventSet = eventQueue.remove(waitTime);");
455             eventSet = eventQueue.remove(waitTime);
456             if (eventSet == null) {
457                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
458             }
459 //            log2("       eventIterator = eventSet.eventIterator;");
460             eventIterator = eventSet.eventIterator();
461         } catch ( Exception e ) {
462             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
463         }
464     }
