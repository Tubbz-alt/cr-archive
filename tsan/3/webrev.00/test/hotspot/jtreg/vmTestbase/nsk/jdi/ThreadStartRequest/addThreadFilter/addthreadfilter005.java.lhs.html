 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadStartRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks up on the following assertion:               <BR>
 42  *    Restricts the events generated by this request            <BR>
 43  *    to those in the given thread.                             <BR>
 44  * The cases to test include three thread's states:             <BR>
 45  * not started, running, dead.                                  <BR>
 46  * <BR>
 47  * The test has three phases and works as follows.                      <BR>
 48  * <BR>
 49  * In first phase,                                                      <BR>
 50  * upon launching debuggee's VM which will be suspended,                <BR>
 51  * a debugger waits for the VMStartEvent within a predefined            <BR>
 52  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 53  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 54  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 55  * and waits for the event within the predefined time interval.         <BR>
 56  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 57  * Upon getting the ClassPrepareEvent,                                  <BR>
 58  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 59  * within debuggee's special methodForCommunication().                  <BR>
 60  * <BR>
 61  * In second phase to check the assertion,                              <BR>
 62  * the debugger and the debuggee perform the following.                 <BR>
 63  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 64  * - The debuggee creates new thread, thread1, and invokes              <BR>
 65  *   the methodForCommunication to be suspended and                     <BR>
 66  *   to inform the debugger with the event.                             <BR>
 67  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 68  *   - creates ThreadStartRequest1 and                                  <BR>
 69  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 70  *   - creates ThreadStartRequest2,                                     <BR>
 71  *     resumes the debuggee to get the thread1 started, and             <BR>
 72  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 73  *   - creates ThreadStartRequest3, waits the thread1 to die, and       <BR>
 74  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 75  * <BR>
 76  * In third phase, at the end of the test,                              <BR>
 77  * the debuggee changes the value of the "instruction"  which           <BR>
 78  * the debugger and debuggee use to inform each other of needed actions,<BR>
 79  * and both end.
 80  * <BR>
 81  *
 82 
 83  */
 84

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : e");
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : e");
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278 
279     }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest eventRequest1 = null;
318         EventRequest eventRequest2 = null;
319         EventRequest eventRequest3 = null;
320 
321         ThreadReference thread1 = null;
322 
323         String thread1Name = "thread1";
324 
325         String property1 = "ThreadStartRequest1";
326         String property2 = "ThreadStartRequest2";
327         String property3 = "ThreadStartRequest3";
328 
329         ReferenceType testClassReference = null;
330 
331 
332         for (int i = 0; ; i++) {
333 
334             vm.resume();
335             breakpointForCommunication();
336 
337             int instruction = ((IntegerValue)
338                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
339 
340             if (instruction == 0) {
341                 vm.resume();
342                 break;
343             }
344 
345             log1(":::::: case: # " + i);
346 
347             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
348 
349             switch (i) {
350 
351               case 0:
352                       testClassReference =
353                            (ReferenceType) vm.classesByName(testedClassName).get(0);
354 
355                       thread1 = (ThreadReference) debuggeeClass.getValue(
356                                                   debuggeeClass.fieldByName(thread1Name));
357 
358                       eventRequest1 = setting2ThreadStartRequest (null,
359                                              EventRequest.SUSPEND_NONE, property1);
360 
361                       try {
362                           log2("......eventRequest1.addThreadFilter(thread1);");
363                           log2("        no Exception expected");
364                           ((ThreadStartRequest)eventRequest1).addThreadFilter(thread1);
365                       } catch ( Exception e ) {
366                           log3("ERROR: Exception : " + e);
367                           testExitCode = FAILED;
368                       }
369                       break;
370 
371               case 1:
372                       eventRequest2 = setting2ThreadStartRequest (null,
373                                              EventRequest.SUSPEND_NONE, property2);
374 
375                       try {
376                           log2("......eventRequest2.addThreadFilter(thread1);");
377                           log2("        no Exception expected");
378                           ((ThreadStartRequest)eventRequest2).addThreadFilter(thread1);
379                       } catch ( Exception e ) {
380                           log3("ERROR: Exception : " + e);
381                           testExitCode = FAILED;
382                       }
383                       break;
384 
385               case 2:
386                       eventRequest3 = setting2ThreadStartRequest (null,
387                                              EventRequest.SUSPEND_NONE, property3);
388 
389                       try {
390                           log2("......eventRequest3.addThreadFilter(thread1);");
391                           log2("        no Exception expected");
392                           ((ThreadStartRequest)eventRequest3).addThreadFilter(thread1);
393                       } catch ( Exception e ) {
394                           log3("ERROR: Exception : " + e);
395                           testExitCode = FAILED;
396                       }
397                       break;
398 
399               default:
400                       throw new JDITestRuntimeException("** default case 2 **");
401             }
402 
403             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
404         }
405         log1("    TESTING ENDS");
406         return;
407     }

409     private ThreadReference threadByName(String name)
410                  throws JDITestRuntimeException {
411 
412         List         all = vm.allThreads();
413         ListIterator li  = all.listIterator();
414 
415         for (; li.hasNext(); ) {
416             ThreadReference thread = (ThreadReference) li.next();
417             if (thread.name().equals(name))
418                 return thread;
419         }
420         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
421     }

423    /*
424     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
425     *                                             String, String, String)
426     *
427     * It sets up a breakpoint at given line number within a given method in a given class
428     * for a given thread.
429     *
430     * Return value: BreakpointRequest object  in case of success
431     *
432     * JDITestRuntimeException   in case of an Exception thrown within the method
433     */
434 
435     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
436                                                   ReferenceType testedClass,
437                                                   String methodName,
438                                                   String bpLine,
439                                                   String property)
440             throws JDITestRuntimeException {
441 
442         log2("......setting up a breakpoint:");
443         log2("       thread: " + thread + "; class: " + testedClass +
444                         "; method: " + methodName + "; line: " + bpLine);
445 
446         List              alllineLocations = null;
447         Location          lineLocation     = null;
448         BreakpointRequest breakpRequest    = null;
449 
450         try {
451             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
452 
453             alllineLocations = method.allLineLocations();
454 
455             int n =
456                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
457             if (n > alllineLocations.size()) {
458                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
459             } else {
460                 lineLocation = (Location) alllineLocations.get(n);
461                 try {
462                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
463                     breakpRequest.putProperty("number", property);
464                     breakpRequest.addThreadFilter(thread);
465                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
466                 } catch ( Exception e1 ) {
467                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
468                     breakpRequest    = null;
469                 }
470             }
471         } catch ( Exception e2 ) {
472             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
473             breakpRequest    = null;
474         }
475 
476         if (breakpRequest == null) {
477             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
478             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
479         }
480 
481         log2("      a breakpoint has been set up");
482         return breakpRequest;
483     }

486     private void getEventSet()
487                  throws JDITestRuntimeException {
488         try {
489 //            log2("       eventSet = eventQueue.remove(waitTime);");
490             eventSet = eventQueue.remove(waitTime);
491             if (eventSet == null) {
492                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
493             }
494 //            log2("       eventIterator = eventSet.eventIterator;");
495             eventIterator = eventSet.eventIterator();
496         } catch ( Exception e ) {
497             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
498         }
499     }

502     private void breakpointForCommunication()
503                  throws JDITestRuntimeException {
504 
505         log2("breakpointForCommunication");
506         getEventSet();
507 
508         if (eventIterator.nextEvent() instanceof BreakpointEvent)
509             return;
510 
511         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
512     }

516     private ThreadStartRequest setting2ThreadStartRequest( ThreadReference thread,
517                                                            int             suspendPolicy,
518                                                            String          property)
519             throws JDITestRuntimeException {
520         try {
521             ThreadStartRequest tsr = eventRManager.createThreadStartRequest();
522             if (thread != null)
523                 tsr.addThreadFilter(thread);
524             tsr.addCountFilter(1);
525             tsr.setSuspendPolicy(suspendPolicy);
526             tsr.putProperty("number", property);
527             return tsr;
528         } catch ( Exception e ) {
529             log3("ERROR: ATTENTION: Exception within settingThreadStartRequest() : " + e);
530             log3("       ThreadStartRequest HAS NOT BEEN SET UP");
531             throw new JDITestRuntimeException("** FAILURE to set up ThreadStartRequest **");
532         }
533     }
