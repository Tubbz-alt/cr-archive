 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ObjectReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ObjectReference.disableCollection()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Prevents garbage collection for this object.              <BR>
 47  *    By default all ObjectReference values returned by JDI     <BR>
 48  *    may be collected at any time the target VM is running.    <BR>
 49  *    A call to this method guarantees that                     <BR>
 50  *    the object will not be collected.                         <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assetion,                               <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new array object "arr1" to be tested on      <BR>
 70  *   collection and invokes the methodForCommunication to be suspended  <BR>
 71  *   and to inform the debugger with the event.                         <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger gets ObjectReference mirroring arr1,                  <BR>
 74  *   disables collecting "arr1" resumes the debuggee and waits for new  <BR>
 75  *   BreakpointEvent.                                                   <BR>
 76  * - The debuggee assigns null to arr1 to make object's memory available<BR>
 77  *   to be collected and pertforms a loop in which it creates new       <BR>
 78  *   objects until VM throws OutOfMemoryError. Then it informs the      <BR>
 79  *   debugger of the ERROR with new BreakPoint event.                   <BR>
 80  * - Upon getting the event,                                            <BR>
 81  *   the debugger gets new ObjectReference mirroring arr1,              <BR>
 82  *   resumes the debuggee, and waits for new breakpoint.                <BR>
 83  * - The debuggee invokes a garbage collector and informs the debugger  <BR>
 84  *   with new breakpoint event.                                         <BR>
 85  * - Upon getting the event,                                            <BR>
 86  *   the debugger gets new ObjectReference mirroring arr1 once again.   <BR>
 87  * <BR>
 88  * If no ObjectCollectedexception is thrown during the above cooperation<BR>
 89  * the test is PASSED, othewise, FAILED.                                <BR>
 90  * <BR>
 91  * Note. To inform each other of needed actions, the debugger and       <BR>
 92  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 93  * <BR>
 94  * In third phase when at the end,                                      <BR>
 95  * the debuggee changes the value of the "instruction"                  <BR>
 96  * to inform the debugger of checks finished, and both end.             <BR>
 97  * <BR>
 98  */

178     private int runThis (String argv[], PrintStream out) {
179 
180         argsHandler     = new ArgumentHandler(argv);
181         logHandler      = new Log(out, argsHandler);
182         Binder binder   = new Binder(argsHandler, logHandler);
183 
184         waitTime        = argsHandler.getWaitTime() * 60000;
185 
186         try {
187             log2("launching a debuggee :");
188             log2("       " + debuggeeName);
189             if (argsHandler.verbose()) {
190                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
191             } else {
192                 debuggee = binder.bindToDebugee(debuggeeName);
193             }
194             if (debuggee == null) {
195                 log3("ERROR: no debuggee launched");
196                 return FAILED;
197             }
198             log2("debuggee launched");
199         } catch ( Exception e ) {
200             log3("ERROR: Exception : " + e);
201             log2("       test cancelled");
202             return FAILED;
203         }
204 
205         debuggee.redirectOutput(logHandler);
206 
207         vm = debuggee.VM();
208 
209         eventQueue = vm.eventQueue();
210         if (eventQueue == null) {
211             log3("ERROR: eventQueue == null : TEST ABORTED");
212             vm.exit(PASS_BASE);
213             return FAILED;
214         }
215 
216         log2("invocation of the method runTest()");
217         switch (runTest()) {
218 
219             case 0 :  log2("test phase has finished normally");
220                       log2("   waiting for the debuggee to finish ...");
221                       debuggee.waitFor();
222 
223                       log2("......getting the debuggee's exit status");
224                       int status = debuggee.getStatus();
225                       if (status != PASS_BASE) {
226                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
227                               status + " != PASS_BASE");
228                           testExitCode = FAILED;
229                       } else {
230                           log2("......debuggee returned expected exit status: " +
231                               status + " == PASS_BASE");
232                       }
233                       break;
234 
235             default : log3("ERROR: runTest() returned unexpected value");
236 
237             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
238                       log2("......forcing: vm.exit();");
239                       testExitCode = FAILED;
240                       try {
241                           vm.exit(PASS_BASE);
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246 
247             case 2 :  log3("test cancelled due to VMDisconnectedException");
248                       log2("......trying: vm.process().destroy();");
249                       testExitCode = FAILED;
250                       try {
251                           Process vmProcess = vm.process();
252                           if (vmProcess != null) {
253                               vmProcess.destroy();
254                           }
255                       } catch ( Exception e ) {
256                           log3("ERROR: Exception : e");
257                       }
258                       break;
259             }
260 
261         return testExitCode;
262     }

266     * Return value: 0 - normal end of the test
267     *               1 - ubnormal end of the test
268     *               2 - VMDisconnectedException while test phase

271     private int runTest() {
272 
273         try {
274             testRun();
275 
276             log2("waiting for VMDeathEvent");
277             getEventSet();
278             if (eventIterator.nextEvent() instanceof VMDeathEvent)
279                 return 0;
280 
281             log3("ERROR: last event is not the VMDeathEvent");
282             return 1;
283         } catch ( VMDisconnectedException e ) {
284             log3("ERROR: VMDisconnectedException : " + e);
285             return 2;
286         } catch ( Exception e ) {
287             log3("ERROR: Exception : " + e);
288             return 1;
289         }
290 
291     }

293     private void testRun()
294                  throws JDITestRuntimeException, Exception {
295 
296         eventRManager = vm.eventRequestManager();
297 
298         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
299         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
300         cpRequest.addClassFilter(debuggeeName);
301 
302         cpRequest.enable();
303         vm.resume();
304         getEventSet();
305         cpRequest.disable();
306 
307         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
308         debuggeeClass = event.referenceType();
309 
310         if (!debuggeeClass.name().equals(debuggeeName))
311            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
312 
313         log2("      received: ClassPrepareEvent for debuggeeClass");
314 
315         String bPointMethod = "methodForCommunication";
316         String lineForComm  = "lineForComm";
317         BreakpointRequest bpRequest;
318 
319         bpRequest = settingBreakpoint(threadByName("main"),
320                                       debuggeeClass,
321                                       bPointMethod, lineForComm, "zero");
322         bpRequest.enable();
323 
324     //------------------------------------------------------  testing section
325 
326         log1("     TESTING BEGINS");
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             String name = "arr1";
346 
347             ReferenceType   arr1RefType = null;
348             ObjectReference arr1ObjRef  = null;
349 
350             try {
351                 log2("......getting: arr1ObjRef = (ObjectReference) debuggeeClass.getValue(debuggeeClass.fieldByName(name));");
352                 arr1ObjRef = (ObjectReference)
353                              debuggeeClass.getValue(debuggeeClass.fieldByName(name));
354 
355                 log2("......         arr1ObjRef.disableCollection();");
356                 arr1ObjRef.disableCollection();
357 
358                 log2("......vm.resume();");
359                 vm.resume();
360                 breakpointForCommunication();
361                 log2("      after breakpointForCommunication();");
362 
363                 log2("......getting: arr1RefType = arr1ObjRef.referenceType();");
364                 arr1RefType = arr1ObjRef.referenceType();
365 
366                 log2("......vm.resume();");
367                 vm.resume();
368                 breakpointForCommunication();
369                 log2("      after breakpointForCommunication();");
370 
371                 log2("......after debuggee invoked runTime.gc() :");
372                 log2("      getting: arr1RefType = arr1ObjRef.referenceType();");
373                 arr1RefType = arr1ObjRef.referenceType();
374 
375             } catch ( ObjectCollectedException e ) {
376                 log3("ERROR: caught ObjectCollectedException");
377                 throw new JDITestRuntimeException("** unexpected ObjectCollectedException **");
378             }
379 
380             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
381         }
382         log1("    TESTING ENDS");
383         return;
384     }

386     private ThreadReference threadByName(String name)
387                  throws JDITestRuntimeException {
388 
389         List         all = vm.allThreads();
390         ListIterator li  = all.listIterator();
391 
392         for (; li.hasNext(); ) {
393             ThreadReference thread = (ThreadReference) li.next();
394             if (thread.name().equals(name))
395                 return thread;
396         }
397         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
398     }

400    /*
401     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
402     *                                             String, String, String)
403     *
404     * It sets up a breakpoint at given line number within a given method in a given class
405     * for a given thread.
406     *
407     * Return value: BreakpointRequest object  in case of success
408     *
409     * JDITestRuntimeException   in case of an Exception thrown within the method
410     */
411 
412     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
413                                                   ReferenceType testedClass,
414                                                   String methodName,
415                                                   String bpLine,
416                                                   String property)
417             throws JDITestRuntimeException {
418 
419         log2("......setting up a breakpoint:");
420         log2("       thread: " + thread + "; class: " + testedClass +
421                         "; method: " + methodName + "; line: " + bpLine);
422 
423         List              alllineLocations = null;
424         Location          lineLocation     = null;
425         BreakpointRequest breakpRequest    = null;
426 
427         try {
428             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
429 
430             alllineLocations = method.allLineLocations();
431 
432             int n =
433                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
434             if (n > alllineLocations.size()) {
435                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
436             } else {
437                 lineLocation = (Location) alllineLocations.get(n);
438                 try {
439                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
440                     breakpRequest.putProperty("number", property);
441                     breakpRequest.addThreadFilter(thread);
442                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
443                 } catch ( Exception e1 ) {
444                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
445                     breakpRequest    = null;
446                 }
447             }
448         } catch ( Exception e2 ) {
449             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
450             breakpRequest    = null;
451         }
452 
453         if (breakpRequest == null) {
454             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
455             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
456         }
457 
458         log2("      a breakpoint has been set up");
459         return breakpRequest;
460     }

463     private void getEventSet()
464                  throws JDITestRuntimeException {
465         try {
466 //            log2("       eventSet = eventQueue.remove(waitTime);");
467             eventSet = eventQueue.remove(waitTime);
468             if (eventSet == null) {
469                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
470             }
471 //            log2("       eventIterator = eventSet.eventIterator;");
472             eventIterator = eventSet.eventIterator();
473         } catch ( Exception e ) {
474             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
475         }
476     }

479     private void breakpointForCommunication()
480                  throws JDITestRuntimeException {
481 
482         log2("breakpointForCommunication");
483         getEventSet();
484 
485         if (eventIterator.nextEvent() instanceof BreakpointEvent)
486             return;
487 
488         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
489     }
