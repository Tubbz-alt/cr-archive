 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VoidType.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that for a mirror of a "void" value the cast <BR>
 42  * from Type to VoidType doesn't throw ClassCastException.      <BR>
 43  * <BR>
 44  * The test has three phases and works as follows.              <BR>
 45  * <BR>
 46  * In first phase,                                                      <BR>
 47  * upon launching debuggee's VM which will be suspended,                <BR>
 48  * a debugger waits for the VMStartEvent within a predefined            <BR>
 49  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 50  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 51  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 52  * and waits for the event within the predefined time interval.         <BR>
 53  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 54  * Upon getting the ClassPrepareEvent,                                  <BR>
 55  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 56  * within debuggee's special methodForCommunication().                  <BR>
 57  * <BR>
 58  * In second phase to check the cast,                                   <BR>
 59  * the debugger and the debuggee perform the following.                 <BR>
 60  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 61  * - The debuggee prepares new check case and invokes                   <BR>
 62  *   the methodForCommunication to be suspended and                     <BR>
 63  *   to inform the debugger with the event.                             <BR>
 64  * - Upon getting the BreakpointEvent,                                  <BR>
 65  *   the debugger performs the check.                                   <BR>
 66  * Note. To inform each other of needed actions, the debugger and       <BR>
 67  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 68  * <BR>
 69  * In third phase when at the end,                                      <BR>
 70  * the debuggee changes the value of the "instruction"                  <BR>
 71  * to inform the debugger of checks finished, and both end.             <BR>
 72  * <BR>
 73  */
 74

 97     public static int run (String argv[], PrintStream out) {
 98 
 99         int exitCode = new voidtype001().runThis(argv, out);
100 
101         if (exitCode != PASSED) {
102             System.out.println("TEST FAILED");
103         }
104         return testExitCode;
105     }

153     private int runThis (String argv[], PrintStream out) {
154 
155         argsHandler     = new ArgumentHandler(argv);
156         logHandler      = new Log(out, argsHandler);
157         Binder binder   = new Binder(argsHandler, logHandler);
158 
159         waitTime        = argsHandler.getWaitTime() * 60000;
160 
161         try {
162             log2("launching a debuggee :");
163             log2("       " + debuggeeName);
164             if (argsHandler.verbose()) {
165                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
166             } else {
167                 debuggee = binder.bindToDebugee(debuggeeName);
168             }
169             if (debuggee == null) {
170                 log3("ERROR: no debuggee launched");
171                 return FAILED;
172             }
173             log2("debuggee launched");
174         } catch ( Exception e ) {
175             log3("ERROR: Exception : " + e);
176             log2("       test cancelled");
177             return FAILED;
178         }
179 
180         debuggee.redirectOutput(logHandler);
181 
182         vm = debuggee.VM();
183 
184         eventQueue = vm.eventQueue();
185         if (eventQueue == null) {
186             log3("ERROR: eventQueue == null : TEST ABORTED");
187             vm.exit(PASS_BASE);
188             return FAILED;
189         }
190 
191         log2("invocation of the method runTest()");
192         switch (runTest()) {
193 
194             case 0 :  log2("test phase has finished normally");
195                       log2("   waiting for the debuggee to finish ...");
196                       debuggee.waitFor();
197 
198                       log2("......getting the debuggee's exit status");
199                       int status = debuggee.getStatus();
200                       if (status != PASS_BASE) {
201                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
202                               status + " != PASS_BASE");
203                           testExitCode = FAILED;
204                       } else {
205                           log2("......debuggee returned expected exit status: " +
206                               status + " == PASS_BASE");
207                       }
208                       break;
209 
210             default : log3("ERROR: runTest() returned unexpected value");
211 
212             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
213                       log2("......forcing: vm.exit();");
214                       testExitCode = FAILED;
215                       try {
216                           vm.exit(PASS_BASE);
217                       } catch ( Exception e ) {
218                           log3("ERROR: Exception : e");
219                       }
220                       break;
221 
222             case 2 :  log3("test cancelled due to VMDisconnectedException");
223                       log2("......trying: vm.process().destroy();");
224                       testExitCode = FAILED;
225                       try {
226                           Process vmProcess = vm.process();
227                           if (vmProcess != null) {
228                               vmProcess.destroy();
229                           }
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : e");
232                       }
233                       break;
234             }
235 
236         return testExitCode;
237     }

240    /*
241     * Return value: 0 - normal end of the test
242     *               1 - ubnormal end of the test
243     *               2 - VMDisconnectedException while test phase
244     */
245 
246     private int runTest() {
247 
248         try {
249             testRun();
250 
251             log2("waiting for VMDeathEvent");
252             getEventSet();
253             if (eventIterator.nextEvent() instanceof VMDeathEvent)
254                 return 0;
255 
256             log3("ERROR: last event is not the VMDeathEvent");
257             return 1;
258         } catch ( VMDisconnectedException e ) {
259             log3("ERROR: VMDisconnectedException : " + e);
260             return 2;
261         } catch ( Exception e ) {
262             log3("ERROR: Exception : " + e);
263             return 1;
264         }
265 
266     }

268     private void testRun()
269                  throws JDITestRuntimeException, Exception {
270 
271         eventRManager = vm.eventRequestManager();
272 
273         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
274         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
275         cpRequest.addClassFilter(debuggeeName);
276 
277         cpRequest.enable();
278         vm.resume();
279         getEventSet();
280         cpRequest.disable();
281 
282         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
283         debuggeeClass = event.referenceType();
284 
285         if (!debuggeeClass.name().equals(debuggeeName))
286            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
287 
288         log2("      received: ClassPrepareEvent for debuggeeClass");
289 
290         String bPointMethod = "methodForCommunication";
291         String lineForComm  = "lineForComm";
292         BreakpointRequest bpRequest;
293 
294         bpRequest = settingBreakpoint(threadByName("main"),
295                                       debuggeeClass,
296                                       bPointMethod, lineForComm, "zero");
297         bpRequest.enable();
298 
299     //------------------------------------------------------  testing section
300 
301         log1("     TESTING BEGINS");
302 
303         for (int i = 0; ; i++) {
304 
305             vm.resume();
306             breakpointForCommunication();
307 
308             int instruction = ((IntegerValue)
309                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
310 
311             if (instruction == 0) {
312                 vm.resume();
313                 break;
314             }
315 
316             log1(":::::: case: # " + i);
317 
318             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
319 
320             log2("......getting: List methods = debuggeeClass.methodsByName('testMethod');");
321             List methods = debuggeeClass.methodsByName("testMethod");
322             if (methods.size() != 1) {
323                 testExitCode = FAILED;
324                 log3("ERROR: methods.size() != 1");
325                 break ;
326             }
327 
328             log2("......getting: Method testMethod = (Method) methods.get(0);");
329             Method testMethod = (Method) methods.get(0);
330 
331             try {
332                  log2("......getting: Type testMethodReturnType = testMethod.returnType();");
333                  Type testMethodReturnType = testMethod.returnType();
334 
335                  log2("......VoidType voidType = (VoidType) testMethodReturnType;");
336                  VoidType voidType = (VoidType) testMethodReturnType;
337             } catch ( ClassNotLoadedException e ) {
338                  log3("ERROR: testmethodReturnType = testmethod.returnType();");
339                  testExitCode = FAILED;
340                  break ;
341             } catch ( ClassCastException e ) {
342                  log3("ERROR: ClassCastException");
343                  testExitCode = FAILED;
344                  break ;
345             }
346 
347             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
348         }
349         log1("    TESTING ENDS");
350         return;
351     }

353     private ThreadReference threadByName(String name)
354                  throws JDITestRuntimeException {
355 
356         List         all = vm.allThreads();
357         ListIterator li  = all.listIterator();
358 
359         for (; li.hasNext(); ) {
360             ThreadReference thread = (ThreadReference) li.next();
361             if (thread.name().equals(name))
362                 return thread;
363         }
364         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
365     }

367    /*
368     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
369     *                                             String, String, String)
370     *
371     * It sets up a breakpoint at given line number within a given method in a given class
372     * for a given thread.
373     *
374     * Return value: BreakpointRequest object  in case of success
375     *
376     * JDITestRuntimeException   in case of an Exception thrown within the method
377     */
378 
379     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
380                                                   ReferenceType testedClass,
381                                                   String methodName,
382                                                   String bpLine,
383                                                   String property)
384             throws JDITestRuntimeException {
385 
386         log2("......setting up a breakpoint:");
387         log2("       thread: " + thread + "; class: " + testedClass +
388                         "; method: " + methodName + "; line: " + bpLine);
389 
390         List              alllineLocations = null;
391         Location          lineLocation     = null;
392         BreakpointRequest breakpRequest    = null;
393 
394         try {
395             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
396 
397             alllineLocations = method.allLineLocations();
398 
399             int n =
400                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
401             if (n > alllineLocations.size()) {
402                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
403             } else {
404                 lineLocation = (Location) alllineLocations.get(n);
405                 try {
406                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
407                     breakpRequest.putProperty("number", property);
408                     breakpRequest.addThreadFilter(thread);
409                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
410                 } catch ( Exception e1 ) {
411                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
412                     breakpRequest    = null;
413                 }
414             }
415         } catch ( Exception e2 ) {
416             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
417             breakpRequest    = null;
418         }
419 
420         if (breakpRequest == null) {
421             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
422             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
423         }
424 
425         log2("      a breakpoint has been set up");
426         return breakpRequest;
427     }

430     private void getEventSet()
431                  throws JDITestRuntimeException {
432         try {
433 //            log2("       eventSet = eventQueue.remove(waitTime);");
434             eventSet = eventQueue.remove(waitTime);
435             if (eventSet == null) {
436                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
437             }
438 //            log2("       eventIterator = eventSet.eventIterator;");
439             eventIterator = eventSet.eventIterator();
440         } catch ( Exception e ) {
441             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
442         }
443     }

446     private void breakpointForCommunication()
447                  throws JDITestRuntimeException {
448 
449         log2("breakpointForCommunication");
450         getEventSet();
451 
452         if (eventIterator.nextEvent() instanceof BreakpointEvent)
453             return;
454 
455         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
456     }
