 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.setSuspendPolicy()</code>     <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on following assertions:                  <BR>
 46  * - a value of policy is equal to one assigned by last method  <BR>
 47  *   setSuspendPolicy(int);                                     <BR>
 48  * - InvalidRequestStateException is thrown                     <BR>
 49  *   if this request is currently enabled;                      <BR>
 50  * - InvalidRequestStateException is thrown                     <BR>
 51  *   if this request has been deleted.                          <BR>
 52  * The cases to check include Requests of all sub-types.        <BR>
 53  * <BR>
 54  * The test has three phases and works as follows.              <BR>
 55  * <BR>
 56  * In first phase,                                                      <BR>
 57  * upon launching debuggee's VM which will be suspended,                <BR>
 58  * a debugger waits for the VMStartEvent within a predefined            <BR>
 59  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 60  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 61  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 62  * and waits for the event within the predefined time interval.         <BR>
 63  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 64  * Upon getting the ClassPrepareEvent,                                  <BR>
 65  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 66  * within debuggee's special methodForCommunication().                  <BR>
 67  * <BR>
 68  * In second phase to check the assertion,                              <BR>
 69  * the debugger and the debuggee perform the following loop.            <BR>
 70  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 71  * - The debuggee prepares new check case (as if) and invokes           <BR>
 72  *   the methodForCommunication to be suspended and                     <BR>
 73  *   to inform the debugger with the event.                             <BR>
 74  * - Upon getting the BreakpointEvent,                                  <BR>
 75  *   the debugger performs the check case required.                     <BR>
 76  * <BR>
 77  * In third phase, at the end of the test, the debuggee changes         <BR>
 78  * the value of the "instruction" which the debugger and debuggee       <BR>
 79  * use to inform each other of needed actions,  and both end.           <BR>
 80  * <BR>
 81  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

247    /*
248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase
251     */
252 
253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309 
310         EventRequest  eventRequest1 = null;
311 
312         String        fieldName1    = "var101";
313         String        fieldName2    = "excObj";
314 
315         ThreadReference thread1     = null;
316         String          threadName1 = "thread1";
317 
318         ReferenceType testClassReference = null;
319 
320         String propertyName   = "name";
321         String propertyValue1 = "value1";
322         String propertyValue2 = "value2";
323 
324 
325         log1("     TESTING BEGINS");
326 
327         for (int i = 0; ; i++) {
328 
329             vm.resume();
330             breakpointForCommunication();
331 
332             int instruction = ((IntegerValue)
333                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
334 
335             if (instruction == 0) {
336                 vm.resume();
337                 break;
338             }
339 
340             log1(":::::: case: # " + i);
341 
342             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
343 
344 
345             switch (i) {
346 
347               case 0:
348                      thread1 = debuggee.threadByNameOrThrow(threadName1);
349 
350                      log2("......setting up StepRequest");
351                      eventRequest1 = eventRManager.createStepRequest
352                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
353                      break;
354 
355               case 1:
356                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
357 
358                      log2("......setting up AccessWatchpointRequest");
359                      eventRequest1 = eventRManager.createAccessWatchpointRequest
360                                     (testClassReference.fieldByName(fieldName1));
361                      break;
362 
363               case 2:
364                      log2(".....setting up ModificationWatchpointRequest");
365                      eventRequest1 = eventRManager.createModificationWatchpointRequest
366                                     (testClassReference.fieldByName(fieldName1));
367                      break;
368 
369               case 3:
370                      log2(".....setting up ClassPrepareRequest");
371                      eventRequest1 = eventRManager.createClassPrepareRequest();
372                      break;
373 
374               case 4:
375                      log2(".....setting up ClassUnloadRequest");
376                      eventRequest1 = eventRManager.createClassUnloadRequest();
377                      break;
378 
379               case 5:
380                      log2(".....setting up MethodEntryRequest");
381                      eventRequest1 = eventRManager.createMethodEntryRequest();
382                      break;
383 
384               case 6:
385                      log2(".....setting up MethodExitRequest");
386                      eventRequest1 = eventRManager.createMethodExitRequest();
387                      break;
388 
389               case 7:
390                      log2(".....setting up ThreadDeathRequest");
391                      eventRequest1 = eventRManager.createThreadDeathRequest();
392                      break;
393 
394               case 8:
395                      log2(".....setting up ThreadStartRequest");
396                      eventRequest1 = eventRManager.createThreadStartRequest();
397                      break;
398 
399               case 9:
400                      log2(".....setting up VMDeathRequest");
401                      eventRequest1 = eventRManager.createVMDeathRequest();
402                      break;
403 
404               case 10:
405                      log2(".....setting up ExceptionRequest");
406                      eventRequest1 = eventRManager.createExceptionRequest(
407                                      (ReferenceType)
408                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
409                                      true, true );
410                      break;
411 
412               case 11:
413                      log2(".....setting up BreakpointRequest");
414                      eventRequest1 = eventRManager.createBreakpointRequest(location);
415                      break;
416 
417 
418               default:
419                       throw new JDITestRuntimeException("** default case 2 **");
420             }
421 
422             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);");
423             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
424 
425             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_ALL);");
426             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
427 
428             log2("      checking up on eventRequest1");
429             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_ALL) {
430                 testExitCode = FAILED;
431                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_ALL");
432             }
433 
434             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD );");
435             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD );
436 
437             log2("      checking up on eventRequest1");
438             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_EVENT_THREAD ) {
439                 testExitCode = FAILED;
440                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_EVENT_THREAD ");
441             }
442 
443             log2("......eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);");
444             eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
445 
446             log2("      checking up on eventRequest1");
447             if ( eventRequest1.suspendPolicy() != EventRequest.SUSPEND_NONE) {
448                 testExitCode = FAILED;
449                 log3("ERROR: suspendPolicy() != EventRequest.SUSPEND_NONE");
450             }
451 
452             log2("......eventRequest1.setEnabled(true);");
453             eventRequest1.setEnabled(true);
454             try {
455                 log2("     checking up on eventRequest1; InvalidRequestStateException is expected");
456                 eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
457                 testExitCode = FAILED;
458                 log3("ERROR: NO InvalidRequestStateException");
459             } catch ( InvalidRequestStateException e ) {
460                 log2("        InvalidRequestStateException");
461             }
462             log2("......eventRManager.deleteEventRequest(eventRequest1);");
463             eventRManager.deleteEventRequest(eventRequest1);
464             try {
465                 log2("     checking up on eventRequest1; InvalidRequestStateException is expected");
466                 eventRequest1.setSuspendPolicy(EventRequest.SUSPEND_NONE);
467                 testExitCode = FAILED;
468                 log3("ERROR: NO InvalidRequestStateException");
469             } catch ( InvalidRequestStateException e ) {
470                 log2("        InvalidRequestStateException");
471             }
472 
473             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
474         }
475         log1("    TESTING ENDS");
476         return;
477     }

479    /*
480     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
481     *                                             String, String, String)
482     *
483     * It sets up a breakpoint at given line number within a given method in a given class
484     * for a given thread.
485     *
486     * Return value: BreakpointRequest object  in case of success
487     *
488     * JDITestRuntimeException   in case of an Exception thrown within the method
489     */
490 
491     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
492                                                   ReferenceType testedClass,
493                                                   String methodName,
494                                                   String bpLine,
495                                                   String property)
496             throws JDITestRuntimeException {
497 
498         log2("......setting up a breakpoint:");
499         log2("       thread: " + thread + "; class: " + testedClass +
500                         "; method: " + methodName + "; line: " + bpLine);
501 
502         List              alllineLocations = null;
503         Location          lineLocation     = null;
504         BreakpointRequest breakpRequest    = null;
505 
506         try {
507             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
508 
509             alllineLocations = method.allLineLocations();
510 
511             int n =
512                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
513             if (n > alllineLocations.size()) {
514                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
515             } else {
516                 lineLocation = (Location) alllineLocations.get(n);
517 
518 // this is only for this test to get Location object
519 location = lineLocation;
520 
521                 try {
522                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
523                     breakpRequest.putProperty("number", property);
524                     breakpRequest.addThreadFilter(thread);
525                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
526                 } catch ( Exception e1 ) {
527                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
528                     breakpRequest    = null;
529                 }
530             }
531         } catch ( Exception e2 ) {
532             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
533             breakpRequest    = null;
534         }
535 
536         if (breakpRequest == null) {
537             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
538             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
539         }
540 
541         log2("      a breakpoint has been set up");
542         return breakpRequest;
543     }

546     private void getEventSet()
547                  throws JDITestRuntimeException {
548         try {
549 //            log2("       eventSet = eventQueue.remove(waitTime);");
550             eventSet = eventQueue.remove(waitTime);
551             if (eventSet == null) {
552                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
553             }
554 //            log2("       eventIterator = eventSet.eventIterator;");
555             eventIterator = eventSet.eventIterator();
556         } catch ( Exception e ) {
557             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
558         }
559     }

563                  throws JDITestRuntimeException {
564

567
