 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createAccessWatchpointRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    - Multiple watchpoints on the same field are permitted.   <BR>
 47  *    - Creates a new disabled watchpoint.                      <BR>
 48  *    - The watchpoint watches accesses to the specified field. <BR>
 49  *    - Throws: UnsupportedOperationException -                 <BR>
 50  *              if the target virtual machine does not support  <BR>
 51  *              this operation.                                 <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee prepares testcase and invokes                         <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent,                                  <BR>
 74  *   the debugger performs the checks required.                         <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

154     private int runThis (String argv[], PrintStream out) {
155 
156         argsHandler     = new ArgumentHandler(argv);
157         logHandler      = new Log(out, argsHandler);
158         Binder binder   = new Binder(argsHandler, logHandler);
159 
160         waitTime        = argsHandler.getWaitTime() * 60000;
161 
162         try {
163             log2("launching a debuggee :");
164             log2("       " + debuggeeName);
165             if (argsHandler.verbose()) {
166                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
167             } else {
168                 debuggee = binder.bindToDebugee(debuggeeName);
169             }
170             if (debuggee == null) {
171                 log3("ERROR: no debuggee launched");
172                 return FAILED;
173             }
174             log2("debuggee launched");
175         } catch ( Exception e ) {
176             log3("ERROR: Exception : " + e);
177             log2("       test cancelled");
178             return FAILED;
179         }
180 
181         debuggee.redirectOutput(logHandler);
182 
183         vm = debuggee.VM();
184 
185         eventQueue = vm.eventQueue();
186         if (eventQueue == null) {
187             log3("ERROR: eventQueue == null : TEST ABORTED");
188             vm.exit(PASS_BASE);
189             return FAILED;
190         }
191 
192         log2("invocation of the method runTest()");
193         switch (runTest()) {
194 
195             case 0 :  log2("test phase has finished normally");
196                       log2("   waiting for the debuggee to finish ...");
197                       debuggee.waitFor();
198 
199                       log2("......getting the debuggee's exit status");
200                       int status = debuggee.getStatus();
201                       if (status != PASS_BASE) {
202                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
203                               status + " != PASS_BASE");
204                           testExitCode = FAILED;
205                       } else {
206                           log2("......debuggee returned expected exit status: " +
207                               status + " == PASS_BASE");
208                       }
209                       break;
210 
211             default : log3("ERROR: runTest() returned unexpected value");
212 
213             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
214                       log2("......forcing: vm.exit();");
215                       testExitCode = FAILED;
216                       try {
217                           vm.exit(PASS_BASE);
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : " + e);
220                       }
221                       break;
222 
223             case 2 :  log3("test cancelled due to VMDisconnectedException");
224                       log2("......trying: vm.process().destroy();");
225                       testExitCode = FAILED;
226                       try {
227                           Process vmProcess = vm.process();
228                           if (vmProcess != null) {
229                               vmProcess.destroy();
230                           }
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235             }
236 
237         return testExitCode;
238     }

242     * Return value: 0 - normal end of the test
243     *               1 - ubnormal end of the test
244     *               2 - VMDisconnectedException while test phase

247     private int runTest() {
248 
249         try {
250             testRun();
251 
252             log2("waiting for VMDeathEvent");
253             getEventSet();
254             if (eventIterator.nextEvent() instanceof VMDeathEvent)
255                 return 0;
256 
257             log3("ERROR: last event is not the VMDeathEvent");
258             return 1;
259         } catch ( VMDisconnectedException e ) {
260             log3("ERROR: VMDisconnectedException : " + e);
261             return 2;
262         } catch ( Exception e ) {
263             log3("ERROR: Exception : " + e);
264             return 1;
265         }
266 
267     }

269     private void testRun()
270                  throws JDITestRuntimeException, Exception {
271 
272         eventRManager = vm.eventRequestManager();
273 
274         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
275         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
276         cpRequest.addClassFilter(debuggeeName);
277 
278         cpRequest.enable();
279         vm.resume();
280         getEventSet();
281         cpRequest.disable();
282 
283         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
284         debuggeeClass = event.referenceType();
285 
286         if (!debuggeeClass.name().equals(debuggeeName))
287            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
288 
289         log2("      received: ClassPrepareEvent for debuggeeClass");
290 
291         String bPointMethod = "methodForCommunication";
292         String lineForComm  = "lineForComm";
293 
294         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
295 
296         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
297                                              debuggeeClass,
298                                             bPointMethod, lineForComm, "zero");
299         bpRequest.enable();
300 
301     //------------------------------------------------------  testing section
302 
303         String fieldName = "testField";
304 
305         Field field = null;
306 
307         List         requests = null;
308         ListIterator li       = null;
309 
310         AccessWatchpointRequest awRequest1 = null;
311         AccessWatchpointRequest awRequest2 = null;
312 
313 
314         log1("     TESTING BEGINS");
315 
316         for (int i = 0; ; i++) {
317 
318             vm.resume();
319             breakpointForCommunication();
320 
321             int instruction = ((IntegerValue)
322                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
323 
324             if (instruction == 0) {
325                 vm.resume();
326                 break;
327             }
328 
329             log1(":::::: case: # " + i);
330 
331             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
332 
333             field = debuggeeClass.fieldByName(fieldName);
334 
335             if ( vm.canWatchFieldAccess() ) {
336                 log2("......vm.canWatchFieldAccess == true");
337                 log2("      creating two AccessWatchpointRequests for 'field': awRequest1 & awRequest2");
338 
339                 awRequest1 = eventRManager.createAccessWatchpointRequest(field);
340                 awRequest2 = eventRManager.createAccessWatchpointRequest(field);
341 
342                 log2("      checking up on: requests are disabled");
343                 if (awRequest1.isEnabled()) {
344                     testExitCode = FAILED;
345                     log3("ERROR: awRequest1 is enabled");
346                 }
347                 if (awRequest2.isEnabled()) {
348                     testExitCode = FAILED;
349                     log3("ERROR: awRequest2 is enabled");
350                 }
351                 log2("      compareing field equality");
352                 if ( !awRequest1.field().equals(field) ) {
353                     testExitCode = FAILED;
354                     log3("ERROR: !awRequest1.field().equals(field)");
355                 }
356                 if ( !awRequest2.field().equals(field) ) {
357                     testExitCode = FAILED;
358                     log3("ERROR: !awRequest2.field().equals(field)");
359                 }
360 
361             } else {
362                 log2("......vm.canWatchFieldAccess == false");
363                 log2("      UnsupportedOperationException is expected");
364                 try {
365                     awRequest1 = eventRManager.createAccessWatchpointRequest(field);
366                     testExitCode = FAILED;
367                     log3("ERROR: NO UnsupportedOperationException");
368                 } catch ( UnsupportedOperationException e ) {
369                 }
370             }
371 
372             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
373         }
374         log1("    TESTING ENDS");
375         return;
376     }

378    /*
379     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
380     *                                             String, String, String)
381     *
382     * It sets up a breakpoint at given line number within a given method in a given class
383     * for a given thread.
384     *
385     * Return value: BreakpointRequest object  in case of success
386     *
387     * JDITestRuntimeException   in case of an Exception thrown within the method
388     */
389 
390     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
391                                                   ReferenceType testedClass,
392                                                   String methodName,
393                                                   String bpLine,
394                                                   String property)
395             throws JDITestRuntimeException {
396 
397         log2("......setting up a breakpoint:");
398         log2("       thread: " + thread + "; class: " + testedClass +
399                         "; method: " + methodName + "; line: " + bpLine);
400 
401         List              alllineLocations = null;
402         Location          lineLocation     = null;
403         BreakpointRequest breakpRequest    = null;
404 
405         try {
406             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
407 
408             alllineLocations = method.allLineLocations();
409 
410             int n =
411                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
412             if (n > alllineLocations.size()) {
413                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
414             } else {
415                 lineLocation = (Location) alllineLocations.get(n);
416                 try {
417                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
418                     breakpRequest.putProperty("number", property);
419                     breakpRequest.addThreadFilter(thread);
420                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
421                 } catch ( Exception e1 ) {
422                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
423                     breakpRequest    = null;
424                 }
425             }
426         } catch ( Exception e2 ) {
427             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
428             breakpRequest    = null;
429         }
430 
431         if (breakpRequest == null) {
432             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
433             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
434         }
435 
436         log2("      a breakpoint has been set up");
437         return breakpRequest;
438     }

441     private void getEventSet()
442                  throws JDITestRuntimeException {
443         try {
444 //            log2("       eventSet = eventQueue.remove(waitTime);");
445             eventSet = eventQueue.remove(waitTime);
446             if (eventSet == null) {
447                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
448             }
449 //            log2("       eventIterator = eventSet.eventIterator;");
450             eventIterator = eventSet.eventIterator();
451         } catch ( Exception e ) {
452             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
453         }
454     }

457     private void breakpointForCommunication()
458                  throws JDITestRuntimeException {
459 
460         log2("breakpointForCommunication");
461         getEventSet();
462 
463         if (eventIterator.nextEvent() instanceof BreakpointEvent)
464             return;
465 
466         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
467     }
