 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *     Restricts the events generated by this request           <BR>
 47  *     to those in the given thread.                            <BR>
 48  *  The cases to test include three thread's states:            <BR>
 49  *  not started, running, dead.                                 <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                              <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates a BreakpointRequest1 and                                 <BR>
 73  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 74  *   - creates a BreakpointRequest2, starts the thread2, and            <BR>
 75  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 76  *   - creates a BreakpointRequest3, waits the thread2 to die, and      <BR>
 77  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

167     private int runThis (String argv[], PrintStream out) {
168 
169         argsHandler     = new ArgumentHandler(argv);
170         logHandler      = new Log(out, argsHandler);
171         Binder binder   = new Binder(argsHandler, logHandler);
172 
173         waitTime        = argsHandler.getWaitTime() * 60000;
174 
175         try {
176             log2("launching a debuggee :");
177             log2("       " + debuggeeName);
178             if (argsHandler.verbose()) {
179                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
180             } else {
181                 debuggee = binder.bindToDebugee(debuggeeName);
182             }
183             if (debuggee == null) {
184                 log3("ERROR: no debuggee launched");
185                 return FAILED;
186             }
187             log2("debuggee launched");
188         } catch ( Exception e ) {
189             log3("ERROR: Exception : " + e);
190             log2("       test cancelled");
191             return FAILED;
192         }
193 
194         debuggee.redirectOutput(logHandler);
195 
196         vm = debuggee.VM();
197 
198         eventQueue = vm.eventQueue();
199         if (eventQueue == null) {
200             log3("ERROR: eventQueue == null : TEST ABORTED");
201             vm.exit(PASS_BASE);
202             return FAILED;
203         }
204 
205         log2("invocation of the method runTest()");
206         switch (runTest()) {
207 
208             case 0 :  log2("test phase has finished normally");
209                       log2("   waiting for the debuggee to finish ...");
210                       debuggee.waitFor();
211 
212                       log2("......getting the debuggee's exit status");
213                       int status = debuggee.getStatus();
214                       if (status != PASS_BASE) {
215                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
216                               status + " != PASS_BASE");
217                           testExitCode = FAILED;
218                       } else {
219                           log2("......debuggee returned expected exit status: " +
220                               status + " == PASS_BASE");
221                       }
222                       break;
223 
224             default : log3("ERROR: runTest() returned unexpected value");
225 
226             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
227                       log2("......forcing: vm.exit();");
228                       testExitCode = FAILED;
229                       try {
230                           vm.exit(PASS_BASE);
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235 
236             case 2 :  log3("test cancelled due to VMDisconnectedException");
237                       log2("......trying: vm.process().destroy();");
238                       testExitCode = FAILED;
239                       try {
240                           Process vmProcess = vm.process();
241                           if (vmProcess != null) {
242                               vmProcess.destroy();
243                           }
244                       } catch ( Exception e ) {
245                           log3("ERROR: Exception : " + e);
246                       }
247                       break;
248             }
249 
250         return testExitCode;
251     }

255     * Return value: 0 - normal end of the test
256     *               1 - ubnormal end of the test
257     *               2 - VMDisconnectedException while test phase
258     */

260     private int runTest() {
261 
262         try {
263             testRun();
264 
265             log2("waiting for VMDeathEvent");
266             getEventSet();
267             if (eventIterator.nextEvent() instanceof VMDeathEvent)
268                 return 0;
269 
270             log3("ERROR: last event is not the VMDeathEvent");
271             return 1;
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = threadByName("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         EventRequest eventRequest1 = null;
319         EventRequest eventRequest2 = null;
320         EventRequest eventRequest3 = null;
321 
322         ThreadReference thread1     = null;
323         String          thread1Name = "thread1";
324 
325         String property1 = "BreakpointRequest1";
326         String property2 = "BreakpointRequest2";
327         String property3 = "BreakpointRequest3";
328 
329         String methodName = "method";
330         String bpLineName = "breakpointLine";
331 
332         ReferenceType testClassReference = null;
333 
334         for (int i = 0; ; i++) {
335 
336             vm.resume();
337             breakpointForCommunication();
338 
339             int instruction = ((IntegerValue)
340                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
341 
342             if (instruction == 0) {
343                 vm.resume();
344                 break;
345             }
346 
347             log1(":::::: case: # " + i);
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351             switch (i) {
352 
353               case 0:
354                      testClassReference =
355                             (ReferenceType) vm.classesByName(testedClassName).get(0);
356 
357                      thread1 = (ThreadReference) debuggeeClass.getValue(
358                                         debuggeeClass.fieldByName(thread1Name));
359 
360                      eventRequest1 = setting2BreakpointRequest (null,
361                                              testClassReference, methodName, bpLineName,
362                                              EventRequest.SUSPEND_NONE, property1);
363 
364                      try {
365                          log2("......eventRequest1.addThreadFilter(thread1);");
366                          log2("        no Exception expected");
367                          ((BreakpointRequest)eventRequest1).addThreadFilter(thread1);
368                          log2("        no Exception");
369                      } catch ( Exception e ) {
370                          log3("ERROR: Exception : " + e);
371                          testExitCode = FAILED;
372                      }
373 
374                      break;
375 
376               case 1:
377                      eventRequest2 = setting2BreakpointRequest (null,
378                                              testClassReference, methodName, bpLineName,
379                                              EventRequest.SUSPEND_NONE, property2);
380 
381                      try {
382                          log2("......eventRequest2.addThreadFilter(thread1);");
383                          log2("        no Exception expected");
384                          ((BreakpointRequest)eventRequest2).addThreadFilter(thread1);
385                          log2("        no Exception");
386                      } catch ( Exception e ) {
387                          log3("ERROR: Exception : " + e);
388                          testExitCode = FAILED;
389                      }
390                      break;
391 
392               case 2:
393                      eventRequest3 = setting2BreakpointRequest (null,
394                                              testClassReference, methodName, bpLineName,
395                                              EventRequest.SUSPEND_NONE, property3);
396 
397                      try {
398                          log2("......eventRequest3.addThreadFilter(thread1);");
399                          log2("        no Exception expected");
400                          ((BreakpointRequest)eventRequest3).addThreadFilter(thread1);
401                          log2("        no Exception");
402                      } catch ( Exception e ) {
403                          log3("ERROR: Exception : " + e);
404                          testExitCode = FAILED;
405                      }
406                      break;
407 
408               default:
409                       throw new JDITestRuntimeException("** default case 2 **");
410             }
411             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
412         }
413         log1("    TESTING ENDS");
414         return;

417     private ThreadReference threadByName(String name)
418                  throws JDITestRuntimeException {
419 
420         List         all = vm.allThreads();
421         ListIterator li  = all.listIterator();
422 
423         for (; li.hasNext(); ) {
424             ThreadReference thread = (ThreadReference) li.next();
425             if (thread.name().equals(name))
426                 return thread;
427         }
428         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
429     }

431    /*
432     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
433     *                                             String, String, String)
434     *
435     * It sets up a breakpoint at given line number within a given method in a given class
436     * for a given thread.
437     *
438     * Return value: BreakpointRequest object  in case of success
439     *
440     * JDITestRuntimeException   in case of an Exception thrown within the method
441     */
442 
443     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
444                                                   ReferenceType testedClass,
445                                                   String methodName,
446                                                   String bpLine,
447                                                   String property)
448             throws JDITestRuntimeException {
449 
450         log2("......setting up a breakpoint:");
451         log2("       thread: " + thread + "; class: " + testedClass +
452                         "; method: " + methodName + "; line: " + bpLine);
453 
454         List              alllineLocations = null;
455         Location          lineLocation     = null;
456         BreakpointRequest breakpRequest    = null;
457 
458         try {
459             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
460 
461             alllineLocations = method.allLineLocations();
462 
463             int n =
464                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
465             if (n > alllineLocations.size()) {
466                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
467             } else {
468                 lineLocation = (Location) alllineLocations.get(n);
469                 try {
470                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
471                     breakpRequest.putProperty("number", property);
472                     breakpRequest.addThreadFilter(thread);
473                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
474                 } catch ( Exception e1 ) {
475                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
476                     breakpRequest    = null;
477                 }
478             }
479         } catch ( Exception e2 ) {
480             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
481             breakpRequest    = null;
482         }
483 
484         if (breakpRequest == null) {
485             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
486             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
487         }
488 
489         log2("      a breakpoint has been set up");
490         return breakpRequest;
491     }

494     private void getEventSet()
495                  throws JDITestRuntimeException {
496         try {
497 //            log2("       eventSet = eventQueue.remove(waitTime);");
498             eventSet = eventQueue.remove(waitTime);
499             if (eventSet == null) {
500                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
501             }
502 //            log2("       eventIterator = eventSet.eventIterator;");
503             eventIterator = eventSet.eventIterator();
504         } catch ( Exception e ) {
505             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
506         }
507     }

510     private void breakpointForCommunication()
511                  throws JDITestRuntimeException {
512 
513         log2("breakpointForCommunication");
514         getEventSet();
515 
516         if (eventIterator.nextEvent() instanceof BreakpointEvent)
517             return;
518 
519         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
520     }

524     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
525                                                           ReferenceType   testedClass,
526                                                           String          methodName,
527                                                           String          bpLine,
528                                                           int             suspendPolicy,
529                                                           String          property        )
530             throws JDITestRuntimeException {
531         try {
532             log2("......setting up a breakpoint:");
533             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
534 
535             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
536             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
537 
538             BreakpointRequest
539             bpr = eventRManager.createBreakpointRequest(loc);
540             bpr.putProperty("number", property);
541             if (thread != null)
542                 bpr.addThreadFilter(thread);
543             bpr.setSuspendPolicy(suspendPolicy);
544 
545             log2("      a breakpoint has been set up");
546             return bpr;
547         } catch ( Exception e ) {
548             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
549             log3("       BreakpointRequest HAS NOT BEEN SET UP");
550             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
551         }
