 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ClassPrepareRequest.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                           <BR>
 42  * <code>com.sun.jdi.ClassPrepareRequest.addClassExclusionFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *   Throws: InvalidRequestStateException -                             <BR>
 47  *           if this request is currently enabled or has been deleted.  <BR>
 48  *           Filters may be added only to disabled requests.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger                                                       <BR>
 67  *   - creates an ClassPrepareRequest and enables it,                           <BR>
 68  *   - invokes the method addClassExclusionFilter() on the request              <BR>
 69  *     expecting to catch InvalidRequestStateException,                         <BR>
 70  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 71  *     expecting to catch no InvalidRequestStateException,                      <BR>
 72  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 73  *     once again expecting to catch InvalidRequestStateException.              <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

162     private int runThis (String argv[], PrintStream out) {
163 
164         argsHandler     = new ArgumentHandler(argv);
165         logHandler      = new Log(out, argsHandler);
166         Binder binder   = new Binder(argsHandler, logHandler);
167 
168         waitTime        = argsHandler.getWaitTime() * 60000;
169 
170         try {
171             log2("launching a debuggee :");
172             log2("       " + debuggeeName);
173             if (argsHandler.verbose()) {
174                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
175             } else {
176                 debuggee = binder.bindToDebugee(debuggeeName);
177             }
178             if (debuggee == null) {
179                 log3("ERROR: no debuggee launched");
180                 return FAILED;
181             }
182             log2("debuggee launched");
183         } catch ( Exception e ) {
184             log3("ERROR: Exception : " + e);
185             log2("       test cancelled");
186             return FAILED;
187         }
188 
189         debuggee.redirectOutput(logHandler);
190 
191         vm = debuggee.VM();
192 
193         eventQueue = vm.eventQueue();
194         if (eventQueue == null) {
195             log3("ERROR: eventQueue == null : TEST ABORTED");
196             vm.exit(PASS_BASE);
197             return FAILED;
198         }
199 
200         log2("invocation of the method runTest()");
201         switch (runTest()) {
202 
203             case 0 :  log2("test phase has finished normally");
204                       log2("   waiting for the debuggee to finish ...");
205                       debuggee.waitFor();
206 
207                       log2("......getting the debuggee's exit status");
208                       int status = debuggee.getStatus();
209                       if (status != PASS_BASE) {
210                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
211                               status + " != PASS_BASE");
212                           testExitCode = FAILED;
213                       } else {
214                           log2("......debuggee returned expected exit status: " +
215                               status + " == PASS_BASE");
216                       }
217                       break;
218 
219             default : log3("ERROR: runTest() returned unexpected value");
220 
221             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
222                       log2("......forcing: vm.exit();");
223                       testExitCode = FAILED;
224                       try {
225                           vm.exit(PASS_BASE);
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : " + e);
228                       }
229                       break;
230 
231             case 2 :  log3("test cancelled due to VMDisconnectedException");
232                       log2("......trying: vm.process().destroy();");
233                       testExitCode = FAILED;
234                       try {
235                           Process vmProcess = vm.process();
236                           if (vmProcess != null) {
237                               vmProcess.destroy();
238                           }
239                       } catch ( Exception e ) {
240                           log3("ERROR: Exception : " + e);
241                       }
242                       break;
243             }
244 
245         return testExitCode;
246     }

255     private int runTest() {
256 
257         try {
258             testRun();
259 
260             log2("waiting for VMDeathEvent");
261             getEventSet();
262             if (eventIterator.nextEvent() instanceof VMDeathEvent)
263                 return 0;
264 
265             log3("ERROR: last event is not the VMDeathEvent");
266             return 1;
267         } catch ( VMDisconnectedException e ) {
268             log3("ERROR: VMDisconnectedException : " + e);
269             return 2;
270         } catch ( Exception e ) {
271             log3("ERROR: Exception : " + e);
272             return 1;
273         }

277     private void testRun()
278                  throws JDITestRuntimeException, Exception {
279 
280         eventRManager = vm.eventRequestManager();
281 
282         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
283         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
284         cpRequest.addClassFilter(debuggeeName);
285         cpRequest.enable();
286 
287         vm.resume();
288         getEventSet();
289         cpRequest.disable();
290 
291         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
292         debuggeeClass = event.referenceType();
293 
294         if (!debuggeeClass.name().equals(debuggeeName))
295            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
296 
297         log2("      received: ClassPrepareEvent for debuggeeClass");
298 
299         String bPointMethod = "methodForCommunication";
300         String lineForComm  = "lineForComm";
301 
302 
303         ThreadReference   mainThread = threadByName("main");
304 
305         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
306                                              debuggeeClass,
307                                             bPointMethod, lineForComm, "zero");
308         bpRequest.enable();
309 
310     //------------------------------------------------------  testing section
311 
312         log1("     TESTING BEGINS");
313 
314         EventRequest  eventRequest1      = null;
315 
316         String        property1          = "ClassPrepareRequest1";
317 
318         eventRequest1 = setting22ClassPrepareRequest(testedClassName1,
319                                  EventRequest.SUSPEND_NONE, property1);
320 
321 
322         for (int i = 0; ; i++) {
323 
324             log1(":::::: case: # " + i);
325 
326             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
327 
328             switch (i) {
329 
330               case 0:
331                      log2("......eventRequest1.enable();");
332                      eventRequest1.enable();
333                      try {
334                          log2("......eventRequest1.addClassFilter(testedClassName1);");
335                          log2("        InvalidRequestStateException expected");
336                          ((ClassPrepareRequest)eventRequest1).addClassFilter(testedClassName1);
337                          log3("ERROR: no InvalidRequestStateException");
338                          testExitCode = FAILED;
339                      } catch ( InvalidRequestStateException e ) {
340                          log2("        InvalidRequestStateException");
341                      }
342 
343                      log2("......eventRequest1.disable();");
344                      eventRequest1.disable();
345                      try {
346                          log2("......eventRequest1.addClassFilter(testedClassName1);");
347                          log2("        no InvalidRequestStateException expected");
348                          ((ClassPrepareRequest)eventRequest1).addClassFilter(testedClassName1);
349                          log2("        no InvalidRequestStateException");
350                      } catch ( InvalidRequestStateException e ) {
351                          log3("ERROR: InvalidRequestStateException");
352                          testExitCode = FAILED;
353                      }
354 
355                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
356                      eventRManager.deleteEventRequest(eventRequest1);
357                      try {
358                          log2("......eventRequest1.addClassFilter(testedClassName1);");
359                          log2("        InvalidRequestStateException expected");
360                          ((ClassPrepareRequest)eventRequest1).addClassFilter(testedClassName1);
361                          log3("ERROR: no InvalidRequestStateException");
362                          testExitCode = FAILED;
363                      } catch ( InvalidRequestStateException e ) {
364                          log2("        InvalidRequestStateException");
365                      }
366 
367                       break;
368 
369               default:
370                       throw new JDITestRuntimeException("** default case 2 **");
371             }
372 
373 
374             vm.resume();
375             breakpointForCommunication();
376 
377             int instruction = ((IntegerValue)
378                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
379 
380             if (instruction == 0) {
381                 vm.resume();
382                 break;
383             }
384 
385             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
386         }
387         log1("    TESTING ENDS");
388         return;
389     }

391     private ThreadReference threadByName(String name)
392                  throws JDITestRuntimeException {
393 
394         List         all = vm.allThreads();
395         ListIterator li  = all.listIterator();
396 
397         for (; li.hasNext(); ) {
398             ThreadReference thread = (ThreadReference) li.next();
399             if (thread.name().equals(name))
400                 return thread;
401         }
402         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
403     }

405    /*
406     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
407     *                                             String, String, String)
408     *
409     * It sets up a breakpoint at given line number within a given method in a given class
410     * for a given thread.
411     *
412     * Return value: BreakpointRequest object  in case of success
413     *
414     * JDITestRuntimeException   in case of an Exception thrown within the method
415     */
416 
417     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
418                                                   ReferenceType testedClass,
419                                                   String methodName,
420                                                   String bpLine,
421                                                   String property)
422             throws JDITestRuntimeException {
423 
424         log2("......setting up a breakpoint:");
425         log2("       thread: " + thread + "; class: " + testedClass +
426                         "; method: " + methodName + "; line: " + bpLine);
427 
428         List              alllineLocations = null;
429         Location          lineLocation     = null;
430         BreakpointRequest breakpRequest    = null;
431 
432         try {
433             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
434 
435             alllineLocations = method.allLineLocations();
436 
437             int n =
438                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
439             if (n > alllineLocations.size()) {
440                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
441             } else {
442                 lineLocation = (Location) alllineLocations.get(n);
443                 try {
444                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
445                     breakpRequest.putProperty("number", property);
446                     breakpRequest.addThreadFilter(thread);
447                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
448                 } catch ( Exception e1 ) {
449                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
450                     breakpRequest    = null;
451                 }
452             }
453         } catch ( Exception e2 ) {
454             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
455             breakpRequest    = null;
456         }
457 
458         if (breakpRequest == null) {
459             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
460             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
461         }
462 
463         log2("      a breakpoint has been set up");
464         return breakpRequest;
465     }

468     private void getEventSet()
469                  throws JDITestRuntimeException {
470         try {
471 //            log2("       eventSet = eventQueue.remove(waitTime);");
472             eventSet = eventQueue.remove(waitTime);
473             if (eventSet == null) {
474                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
475             }
476 //            log2("       eventIterator = eventSet.eventIterator;");
477             eventIterator = eventSet.eventIterator();
478         } catch ( Exception e ) {
479             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
480         }
481     }

484     private void breakpointForCommunication()
485                  throws JDITestRuntimeException {
486         log2("breakpointForCommunication");
487 
488         do {
489             getEventSet();
490 
491             Event event = eventIterator.nextEvent();
492             if (event instanceof BreakpointEvent)
493                 return;
494 
495             log2("      received: " + event);
496 
497             if (EventFilters.filtered(event, debuggeeName)) {
498                 eventSet.resume();
499             }
500             else {
501                 break;
502             }
503         } while (true);
504 
505         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
506     }
