 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.canRedefineClasses()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test works as follows.                                           <BR>
 46  * Upon launching debuggee's VM which will be suspended,                <BR>
 47  * a debugger waits for the VMStartEvent within a predefined            <BR>
 48  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 49  * Upon getting the VMStartEvent, it makes the request                  <BR>
 50  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 51  * resumes the VM, and waits for the event within the predefined        <BR>
 52  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 53  * Upon getting the ClassPrepareEvent,                                  <BR>
 54  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 55  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 56  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 57  * to be suspended and to inform the debugger with the event.           <BR>
 58  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 59  * At the end, the debuggee changes the value of the "instruction"      <BR>
 60  * to inform the debugger of checks finished, and both end.             <BR>
 61  */
 62

 67     static final int FAILED = 2;

 85     public static int run (String argv[], PrintStream out) {
 86 
 87         int exitCode = new canredefineclasses001().runThis(argv, out);
 88 
 89         if (exitCode != PASSED) {
 90             System.out.println("TEST FAILED");
 91         }
 92         return testExitCode;
 93     }

135     private int runThis (String argv[], PrintStream out) {
136 
137         argsHandler     = new ArgumentHandler(argv);
138         logHandler      = new Log(out, argsHandler);
139         Binder binder   = new Binder(argsHandler, logHandler);
140 
141         waitTime        = argsHandler.getWaitTime() * 60000;
142 
143         try {
144             log2("launching a debuggee :");
145             log2("       " + debuggeeName);
146             if (argsHandler.verbose()) {
147                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
148             } else {
149                 debuggee = binder.bindToDebugee(debuggeeName);
150             }
151             if (debuggee == null) {
152                 log3("ERROR: no debuggee launched");
153                 return FAILED;
154             }
155             log2("debuggee launched");
156         } catch ( Exception e ) {
157             log3("ERROR: Exception : " + e);
158             log2("       test cancelled");
159             return FAILED;
160         }
161 
162         debuggee.redirectOutput(logHandler);
163 
164         vm = debuggee.VM();
165 
166         eventQueue = vm.eventQueue();
167         if (eventQueue == null) {
168             log3("ERROR: eventQueue == null : TEST ABORTED");
169             vm.exit(PASS_BASE);
170             return FAILED;
171         }
172 
173         log2("invocation of the method runTest()");
174         switch (runTest()) {
175 
176             case 0 :  log2("test phase has finished normally");
177                       log2("   waiting for the debuggee to finish ...");
178                       debuggee.waitFor();
179 
180                       log2("......getting the debuggee's exit status");
181                       int status = debuggee.getStatus();
182                       if (status != PASS_BASE) {
183                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
184                               status + " != PASS_BASE");
185                           testExitCode = FAILED;
186                       } else {
187                           log2("......debuggee returned expected exit status: " +
188                               status + " == PASS_BASE");
189                       }
190                       break;
191 
192             default : log3("ERROR: runTest() returned unexpected value");
193 
194             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
195                       log2("......forcing: vm.exit();");
196                       testExitCode = FAILED;
197                       try {
198                           vm.exit(PASS_BASE);
199                       } catch ( Exception e ) {
200                           log3("ERROR: Exception : e");
201                       }
202                       break;
203 
204             case 2 :  log3("test cancelled due to VMDisconnectedException");
205                       log2("......trying: vm.process().destroy();");
206                       testExitCode = FAILED;
207                       try {
208                           Process vmProcess = vm.process();
209                           if (vmProcess != null) {
210                               vmProcess.destroy();
211                           }
212                       } catch ( Exception e ) {
213                           log3("ERROR: Exception : e");
214                       }
215                       break;
216             }
217 
218         return testExitCode;
219     }

222    /*
223     * Return value: 0 - normal end of the test
224     *               1 - ubnormal end of the test
225     *               2 - VMDisconnectedException while test phase
226     */
227 
228     private int runTest() {
229 
230         try {
231             testRun();
232 
233             log2("waiting for VMDeathEvent");
234             getEventSet();
235             if (eventIterator.nextEvent() instanceof VMDeathEvent)
236                 return 0;
237 
238             log3("ERROR: last event is not the VMDeathEvent");
239             return 1;
240         } catch ( VMDisconnectedException e ) {
241             log3("ERROR: VMDisconnectedException : " + e);
242             return 2;
243         } catch ( Exception e ) {
244             log3("ERROR: Exception : " + e);
245             return 1;
246         }
247 
248     }

250     private void testRun()
251                  throws JDITestRuntimeException, Exception {
252 
253         eventRManager = vm.eventRequestManager();
254 
255         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
256         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
257         cpRequest.addClassFilter(debuggeeName);
258 
259         cpRequest.enable();
260         vm.resume();
261         getEventSet();
262         cpRequest.disable();
263 
264         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
265         debuggeeClass = event.referenceType();
266 
267         if (!debuggeeClass.name().equals(debuggeeName))
268            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
269 
270         log2("      received: ClassPrepareEvent for debuggeeClass");
271 
272         String bPointMethod = "methodForCommunication";
273         String lineForComm  = "lineForComm";
274         BreakpointRequest bpRequest;
275 
276         try {
277             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
278                                           debuggeeClass,
279                                           bPointMethod, lineForComm, "zero");
280         } catch ( Exception e ) {
281             throw e;
282         }
283         bpRequest.enable();
284 
285     //------------------------------------------------------  testing section
286 
287         log1("     TESTING BEGINS");
288 
289         for (int i = 0; ; i++) {
290 
291             vm.resume();
292             breakpointForCommunication();
293 
294             int instruction = ((IntegerValue)
295                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
296 
297             if (instruction == 0) {
298                 vm.resume();
299                 break;
300             }
301 
302             log1("  new check: # " + i);
303 
304             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
305 
306             log2("...... vm.canRedefineClasses()");
307             if ( vm.canRedefineClasses() )
308                 log2("      vm.canRedefineClasses() == true");
309             else
310                 log2("      vm.canRedefineClasses() == false");
311 
312             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
313         }
314         log1("    TESTING ENDS");
315         return;
316     }

318    /*
319     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
320     *                                             String, String, String)
321     *
322     * It sets up a breakpoint at given line number within a given method in a given class
323     * for a given thread.
324     *
325     * Return value: BreakpointRequest object  in case of success
326     *
327     * JDITestRuntimeException   in case of an Exception thrown within the method
328     */
329 
330     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
331                                                   ReferenceType testedClass,
332                                                   String methodName,
333                                                   String bpLine,
334                                                   String property)
335             throws JDITestRuntimeException {
336 
337         log2("......setting up a breakpoint:");
338         log2("       thread: " + thread + "; class: " + testedClass +
339                         "; method: " + methodName + "; line: " + bpLine);
340 
341         List              alllineLocations = null;
342         Location          lineLocation     = null;
343         BreakpointRequest breakpRequest    = null;
344 
345         try {
346             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
347 
348             alllineLocations = method.allLineLocations();
349 
350             int n =
351                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
352             if (n > alllineLocations.size()) {
353                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
354             } else {
355                 lineLocation = (Location) alllineLocations.get(n);
356                 try {
357                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
358                     breakpRequest.putProperty("number", property);
359                     breakpRequest.addThreadFilter(thread);
360                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
361                 } catch ( Exception e1 ) {
362                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
363                     breakpRequest    = null;
364                 }
365             }
366         } catch ( Exception e2 ) {
367             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
368             breakpRequest    = null;
369         }
370 
371         if (breakpRequest == null) {
372             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
373             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
374         }
375 
376         log2("      a breakpoint has been set up");
377         return breakpRequest;
378     }

381     private void getEventSet()
382                  throws JDITestRuntimeException {
383         try {
384 //            log2("       eventSet = eventQueue.remove(waitTime);");
385             eventSet = eventQueue.remove(waitTime);
386             if (eventSet == null) {
387                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
388             }
389 //            log2("       eventIterator = eventSet.eventIterator;");
390             eventIterator = eventSet.eventIterator();
391         } catch ( Exception e ) {
392             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
393         }
394     }

396 
397     private void breakpointForCommunication()
398                  throws JDITestRuntimeException {
399 
400         log2("breakpointForCommunication");
401         getEventSet();
402 
403         if (eventIterator.nextEvent() instanceof BreakpointEvent)
404             return;
405 
406         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
407     }
