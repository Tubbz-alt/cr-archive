 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addClassExclusionFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 68  *   and to inform the debugger with the event.                         <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates an ExceptionRequest and enables it,                      <BR>
 71  *   - invokes the method addClassExclusionFilter() on the request      <BR>
 72  *     expecting to catch InvalidRequestStateException,                         <BR>
 73  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 74  *     expecting to catch no InvalidRequestStateException,                      <BR>
 75  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 76  *     once again expecting to catch InvalidRequestStateException.              <BR>
 77  * <BR>
 78  * In third phase, at the end of the test, the debuggee changes         <BR>
 79  * the value of the "instruction" which the debugger and debuggee       <BR>
 80  * use to inform each other of needed actions,  and both end.           <BR>
 81  * <BR>
 82  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         log1("     TESTING BEGINS");
309 
310         EventRequest  eventRequest1      = null;
311 
312         String        property1          = "ExceptionRequest1";
313 
314 
315 
316 
317 
318         for (int i = 0; ; i++) {
319 
320             vm.resume();
321             breakpointForCommunication();
322 
323             int instruction = ((IntegerValue)
324                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
325 
326             if (instruction == 0) {
327                 vm.resume();
328                 break;
329             }
330 
331 
332             log1(":::::: case: # " + i);
333 
334             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
335 
336             switch (i) {
337 
338               case 0:
339                      eventRequest1 = setting23ExceptionRequest(null, testedClassName1,
340                                               EventRequest.SUSPEND_NONE, property1);
341 
342                      eventRequest1.enable();
343 
344                      try {
345                          log2("......eventRequest1.addClassFilter(testedClassName1);");
346                          log2("        InvalidRequestStateException expected");
347                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
348                          log3("ERROR: no InvalidRequestStateException");
349                          testExitCode = FAILED;
350                      } catch ( InvalidRequestStateException e ) {
351                          log2("        InvalidRequestStateException");
352                      }
353 
354                      log2("......eventRequest1.disable();");
355                      eventRequest1.disable();
356                      try {
357                          log2("......eventRequest1.addClassFilter(testedClassName1);");
358                          log2("        no InvalidRequestStateException expected");
359                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
360                          log2("        no InvalidRequestStateException");
361                      } catch ( InvalidRequestStateException e ) {
362                          log3("ERROR: InvalidRequestStateException");
363                          testExitCode = FAILED;
364                      }
365 
366                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
367                      eventRManager.deleteEventRequest(eventRequest1);
368                      try {
369                          log2("......eventRequest1.addClassFilter(testedClassName1);");
370                          log2("        InvalidRequestStateException expected");
371                          ((ExceptionRequest)eventRequest1).addClassFilter(testedClassName1);
372                          log3("ERROR: no InvalidRequestStateException");
373                          testExitCode = FAILED;
374                      } catch ( InvalidRequestStateException e ) {
375                          log2("        InvalidRequestStateException");
376                      }
377                       break;
378 
379               default:
380                       throw new JDITestRuntimeException("** default case 2 **");
381             }
382 
383             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
384         }
385         log1("    TESTING ENDS");
386         return;
387     }

389    /*
390     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
391     *                                             String, String, String)
392     *
393     * It sets up a breakpoint at given line number within a given method in a given class
394     * for a given thread.
395     *
396     * Return value: BreakpointRequest object  in case of success
397     *
398     * JDITestRuntimeException   in case of an Exception thrown within the method
399     */
400 
401     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
402                                                   ReferenceType testedClass,
403                                                   String methodName,
404                                                   String bpLine,
405                                                   String property)
406             throws JDITestRuntimeException {
407 
408         log2("......setting up a breakpoint:");
409         log2("       thread: " + thread + "; class: " + testedClass +
410                         "; method: " + methodName + "; line: " + bpLine);
411 
412         List              alllineLocations = null;
413         Location          lineLocation     = null;
414         BreakpointRequest breakpRequest    = null;
415 
416         try {
417             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
418 
419             alllineLocations = method.allLineLocations();
420 
421             int n =
422                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
423             if (n > alllineLocations.size()) {
424                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
425             } else {
426                 lineLocation = (Location) alllineLocations.get(n);
427                 try {
428                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
429                     breakpRequest.putProperty("number", property);
430                     breakpRequest.addThreadFilter(thread);
431                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
432                 } catch ( Exception e1 ) {
433                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
434                     breakpRequest    = null;
435                 }
436             }
437         } catch ( Exception e2 ) {
438             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
439             breakpRequest    = null;
440         }
441 
442         if (breakpRequest == null) {
443             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
444             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
445         }
446 
447         log2("      a breakpoint has been set up");
448         return breakpRequest;
449     }

452     private void getEventSet()
453                  throws JDITestRuntimeException {
454         try {
455 //            log2("       eventSet = eventQueue.remove(waitTime);");
456             eventSet = eventQueue.remove(waitTime);
457             if (eventSet == null) {
458                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
459             }
460 //            log2("       eventIterator = eventSet.eventIterator;");
461             eventIterator = eventSet.eventIterator();
462         } catch ( Exception e ) {
463             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
464         }
465     }

468     private void breakpointForCommunication()
469                  throws JDITestRuntimeException {
470 
471         log2("breakpointForCommunication");
472         getEventSet();
473 
474         if (eventIterator.nextEvent() instanceof BreakpointEvent)
475             return;
476 
477         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
478     }

482     private ExceptionRequest setting23ExceptionRequest ( ThreadReference thread,
483                                                          String          testedClass,
484                                                          int             suspendPolicy,
485                                                          String          property       )
486             throws JDITestRuntimeException {
487         try {
488             log2("......setting up ExceptionRequest:");
489             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
490 
491             ExceptionRequest
492             excr = eventRManager.createExceptionRequest(null, true, true);
493             excr.putProperty("number", property);
494             if (thread != null)
495                 excr.addThreadFilter(thread);
496             excr.addClassExclusionFilter(testedClass);
497             excr.setSuspendPolicy(suspendPolicy);
498 
499             log2("      ExceptionRequest has been set up");
500             return excr;
501         } catch ( Exception e ) {
502             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
503             log3("       ExceptionRequest HAS NOT BEEN SET UP");
504             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
505         }
