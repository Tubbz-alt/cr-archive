 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for ModificationWatchpointEvent sets. <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a ModificationWatchpointRequest, resumes      <BR>
 74  *   the debuggee, and waits for the ModificationWatchpointEvent.       <BR>
 75  * - The debuggee invokes the special method which makes access to      <BR>
 76  *   a predefined variable to be resulting in the event.                <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

169     private int runThis (String argv[], PrintStream out) {
170 
171         argsHandler     = new ArgumentHandler(argv);
172         logHandler      = new Log(out, argsHandler);
173         Binder binder   = new Binder(argsHandler, logHandler);
174 
175         waitTime        = argsHandler.getWaitTime() * 60000;
176 
177         try {
178             log2("launching a debuggee :");
179             log2("       " + debuggeeName);
180             if (argsHandler.verbose()) {
181                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
182             } else {
183                 debuggee = binder.bindToDebugee(debuggeeName);
184             }
185             if (debuggee == null) {
186                 log3("ERROR: no debuggee launched");
187                 return FAILED;
188             }
189             log2("debuggee launched");
190         } catch ( Exception e ) {
191             log3("ERROR: Exception : " + e);
192             log2("       test cancelled");
193             return FAILED;
194         }
195 
196         debuggee.redirectOutput(logHandler);
197 
198         vm = debuggee.VM();
199 
200         eventQueue = vm.eventQueue();
201         if (eventQueue == null) {
202             log3("ERROR: eventQueue == null : TEST ABORTED");
203             vm.exit(PASS_BASE);
204             return FAILED;
205         }
206 
207         log2("invocation of the method runTest()");
208         switch (runTest()) {
209 
210             case 0 :  log2("test phase has finished normally");
211                       log2("   waiting for the debuggee to finish ...");
212                       debuggee.waitFor();
213 
214                       log2("......getting the debuggee's exit status");
215                       int status = debuggee.getStatus();
216                       if (status != PASS_BASE) {
217                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
218                               status + " != PASS_BASE");
219                           testExitCode = FAILED;
220                       } else {
221                           log2("......debuggee returned expected exit status: " +
222                               status + " == PASS_BASE");
223                       }
224                       break;
225 
226             default : log3("ERROR: runTest() returned unexpected value");
227 
228             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
229                       log2("......forcing: vm.exit();");
230                       testExitCode = FAILED;
231                       try {
232                           vm.exit(PASS_BASE);
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237 
238             case 2 :  log3("test cancelled due to VMDisconnectedException");
239                       log2("......trying: vm.process().destroy();");
240                       testExitCode = FAILED;
241                       try {
242                           Process vmProcess = vm.process();
243                           if (vmProcess != null) {
244                               vmProcess.destroy();
245                           }
246                       } catch ( Exception e ) {
247                           log3("ERROR: Exception : e");
248                       }
249                       break;
250             }
251 
252         return testExitCode;
253     }

262     private int runTest() {
263 
264         try {
265             testRun();
266 
267             log2("waiting for VMDeathEvent");
268             getEventSet();
269             if (eventIterator.nextEvent() instanceof VMDeathEvent)
270                 return 0;
271 
272             log3("ERROR: last event is not the VMDeathEvent");
273             return 1;
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         if ( !vm.canWatchFieldModification() ) {
288             log2("......vm.canWatchFieldModification == false :: test cancelled");
289             vm.exit(PASS_BASE);
290             return;
291         }
292 
293 
294         eventRManager = vm.eventRequestManager();
295 
296         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
297         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
298         cpRequest.addClassFilter(debuggeeName);
299 
300         cpRequest.enable();
301         vm.resume();
302         getEventSet();
303         cpRequest.disable();
304 
305         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
306         debuggeeClass = event.referenceType();
307 
308         if (!debuggeeClass.name().equals(debuggeeName))
309            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
310 
311         log2("      received: ClassPrepareEvent for debuggeeClass");
312 
313         String bPointMethod = "methodForCommunication";
314         String lineForComm  = "lineForComm";
315         BreakpointRequest bpRequest;
316         ThreadReference   mainThread = threadByName("main");
317         bpRequest = settingBreakpoint(mainThread,
318                                       debuggeeClass,
319                                       bPointMethod, lineForComm, "zero");
320         bpRequest.enable();
321 
322         vm.resume();
323 
324     //------------------------------------------------------  testing section
325 
326         log1("     TESTING BEGINS");
327 
328         EventRequest eventRequest1 = null;
329         EventRequest eventRequest2 = null;
330         EventRequest eventRequest3 = null;
331 
332         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
333         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
334         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
335 
336         String fieldName = "var1";
337 
338         ReferenceType testClassReference = null;
339 
340 
341         for (int i = 0; ; i++) {
342 
343             breakpointForCommunication();
344 
345             int instruction = ((IntegerValue)
346                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
347 
348             if (instruction == 0) {
349                 vm.resume();
350                 break;
351             }
352 
353             log1(":::::: case: # " + i);
354 
355             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
356 
357             switch (i) {
358 
359               case 0:
360                       testClassReference =
361                            (ReferenceType) vm.classesByName(testedClassName).get(0);
362 
363                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
364                                              testClassReference, fieldName,
365                                              SUSPEND_NONE, "ModificationWatchpointRequest1");
366                       eventRequest1.enable();
367                       break;
368 
369               case 1:
370                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
371                                              testClassReference, fieldName,
372                                              SUSPEND_THREAD, "ModificationWatchpointRequest2");
373                       eventRequest2.enable();
374                       break;
375 
376               case 2:
377                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
378                                              testClassReference, fieldName,
379                                              SUSPEND_ALL, "ModificationWatchpointRequest3");
380                       eventRequest3.enable();
381                       break;
382 
383 
384               default:
385                       throw new JDITestRuntimeException("** default case 2 **");
386             }
387 
388             log2("......waiting for new ModificationWatchpointEvent : " + i);
389             mainThread.resume();
390             getEventSet();
391 
392             Event newEvent = eventIterator.nextEvent();
393             if ( !(newEvent instanceof ModificationWatchpointEvent)) {
394                  log3("ERROR: new event is not ModificationWatchpointEvent");
395                  testExitCode = FAILED;
396             } else {
397 
398                 String property = (String) newEvent.request().getProperty("number");
399                 log2("       got new ModificationWatchpointEvent with propety 'number' == " + property);
400 
401                 log2("......checking up on EventSet.resume()");
402                 log2("......--> vm.suspend();");
403                 vm.suspend();
404 
405                 log2("        getting : Map<String, Integer> suspendsCounts1");
406 
407                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
408                 for (ThreadReference threadReference : vm.allThreads()) {
409                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
410                 }
411 
412                 log2("        eventSet.resume;");
413                 eventSet.resume();
414 
415                 log2("        getting : Map<String, Integer> suspendsCounts2");
416                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
417                 for (ThreadReference threadReference : vm.allThreads()) {
418                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
419                 }
420 
421                 log2(suspendsCounts1.toString());
422                 log2(suspendsCounts2.toString());
423 
424                 log2("        getting : int policy = eventSet.suspendPolicy();");
425                 int policy = eventSet.suspendPolicy();
426 
427                 switch (policy) {
428 
429                   case SUSPEND_NONE   :
430                        log2("        case SUSPEND_NONE");
431                        for (String threadName : suspendsCounts1.keySet()) {
432                            log2("        checking " + threadName);
433                            if (!suspendsCounts2.containsKey(threadName)) {
434                                log3("ERROR: couldn't get ThreadReference for " + threadName);
435                                testExitCode = FAILED;
436                                break;
437                            }
438                            int count1 = suspendsCounts1.get(threadName);
439                            int count2 = suspendsCounts2.get(threadName);
440                            if (count1 != count2) {
441                                log3("ERROR: suspendCounts don't match for : " + threadName);
442                                log3("before resuming : " + count1);
443                                log3("after  resuming : " + count2);
444                                testExitCode = FAILED;
445                                break;
446                            }
447                        }
448                        eventRequest1.disable();
449                        break;
450 
451                   case SUSPEND_THREAD :
452                        log2("        case SUSPEND_THREAD");
453                        for (String threadName : suspendsCounts1.keySet()) {
454                            log2("checking " + threadName);
455                            if (!suspendsCounts2.containsKey(threadName)) {
456                                log3("ERROR: couldn't get ThreadReference for " + threadName);
457                                testExitCode = FAILED;
458                                break;
459                            }
460                            int count1 = suspendsCounts1.get(threadName);
461                            int count2 = suspendsCounts2.get(threadName);
462                            String eventThreadName = ((ModificationWatchpointEvent)newEvent).thread().name();
463                            int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
464                            if (count1 != expectedValue) {
465                                log3("ERROR: suspendCounts don't match for : " + threadName);
466                                log3("before resuming : " + count1);
467                                log3("after  resuming : " + count2);
468                                testExitCode = FAILED;
469                                break;
470                            }
471                        }
472                        eventRequest2.disable();
473                        break;
474 
475                   case SUSPEND_ALL    :
476 
477                         log2("        case SUSPEND_ALL");
478                         for (String threadName : suspendsCounts1.keySet()) {
479                             log2("checking " + threadName);
480                             if (!event.request().equals(eventRequest3))
481                                 break;
482                             if (!suspendsCounts2.containsKey(threadName)) {
483                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
484                                 testExitCode = FAILED;
485                                 break;
486                             }
487                             int count1 = suspendsCounts1.get(threadName);
488                             int count2 = suspendsCounts2.get(threadName);
489                             if (count1 != count2 + 1) {
490                                 log3("ERROR: suspendCounts don't match for : " + threadName);
491                                 log3("before resuming : " + count1);
492                                 log3("after  resuming : " + count2);
493                                 testExitCode = FAILED;
494                                 break;
495                             }
496                         }
497                         eventRequest3.disable();
498                         break;
499 
500                   default: throw new JDITestRuntimeException("** default case 1 **");
501                 }
502             }
503 
504             log2("......--> vm.resume()");
505             vm.resume();
506             informDebuggeeTestCase(i);
507 
508             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
509         }
510         log1("    TESTING ENDS");
511         return;
512     }

514     private ThreadReference threadByName(String name)
515                  throws JDITestRuntimeException {
516 
517         List         all = vm.allThreads();
518         ListIterator li  = all.listIterator();
519 
520         for (; li.hasNext(); ) {
521             ThreadReference thread = (ThreadReference) li.next();
522             if (thread.name().equals(name))
523                 return thread;
524         }
525         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
526     }

528    /*
529     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
530     *                                             String, String, String)
531     *
532     * It sets up a breakpoint at given line number within a given method in a given class
533     * for a given thread.
534     *
535     * Return value: BreakpointRequest object  in case of success
536     *
537     * JDITestRuntimeException   in case of an Exception thrown within the method
538     */
539 
540     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
541                                                   ReferenceType testedClass,
542                                                   String methodName,
543                                                   String bpLine,
544                                                   String property)
545             throws JDITestRuntimeException {
546 
547         log2("......setting up a breakpoint:");
548         log2("       thread: " + thread + "; class: " + testedClass +
549                         "; method: " + methodName + "; line: " + bpLine);
550 
551         List              alllineLocations = null;
552         Location          lineLocation     = null;
553         BreakpointRequest breakpRequest    = null;
554 
555         try {
556             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
557 
558             alllineLocations = method.allLineLocations();
559 
560             int n =
561                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
562             if (n > alllineLocations.size()) {
563                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
564             } else {
565                 lineLocation = (Location) alllineLocations.get(n);
566                 try {
567                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
568                     breakpRequest.putProperty("number", property);
569                     breakpRequest.addThreadFilter(thread);
570                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
571                 } catch ( Exception e1 ) {
572                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
573                     breakpRequest    = null;
574                 }
575             }
576         } catch ( Exception e2 ) {
577             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
578             breakpRequest    = null;
579         }
580 
581         if (breakpRequest == null) {
582             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
583             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
584         }
585 
586         log2("      a breakpoint has been set up");
587         return breakpRequest;
588     }

591     private void getEventSet()
592                  throws JDITestRuntimeException {
593         try {
594 //            log2("       eventSet = eventQueue.remove(waitTime);");
595             eventSet = eventQueue.remove(waitTime);
596             if (eventSet == null) {
597                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
598             }
599 //            log2("       eventIterator = eventSet.eventIterator;");
600             eventIterator = eventSet.eventIterator();
601         } catch ( Exception e ) {
602             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
603         }
604     }

607     private void breakpointForCommunication()
608                  throws JDITestRuntimeException {
609 
610         log2("breakpointForCommunication");
611         getEventSet();
612 
613         if (eventIterator.nextEvent() instanceof BreakpointEvent)
614             return;
615 
616         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
617     }

621     private ModificationWatchpointRequest settingModificationWatchpointRequest (
622                                                   ThreadReference thread,
623                                                   ReferenceType   testedClass,
624                                                   String          fieldName,
625                                                   int             suspendPolicy,
626                                                   String          property        )
627             throws JDITestRuntimeException {
628         try {
629             log2("......setting up ModificationWatchpointRequest:");
630             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
631             Field field = testedClass.fieldByName(fieldName);
632 
633             ModificationWatchpointRequest
634             awr = eventRManager.createModificationWatchpointRequest(field);
635             awr.putProperty("number", property);
636             awr.addThreadFilter(thread);
637             awr.setSuspendPolicy(suspendPolicy);
638 
639             log2("      ModificationWatchpointRequest has been set up");
640             return awr;
641         } catch ( Exception e ) {
642             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
643             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
644             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
645         }
646     }
647     /**
648      * Inform debuggee which thread test the debugger has completed.
649      * Used for synchronization, so the debuggee does not move too quickly.
650      * @param testCase index of just completed test
651      */
652     void informDebuggeeTestCase(int testCase) {
653         try {
654             ((ClassType)debuggeeClass)
655                 .setValue(debuggeeClass.fieldByName("testCase"),
656                           vm.mirrorOf(testCase));
657         } catch (InvalidTypeException ite) {
658             throw new Failure("** FAILURE setting testCase  **");
659         } catch (ClassNotLoadedException cnle) {
660             throw new Failure("** FAILURE notifying debuggee  **");
661         } catch (VMDisconnectedException e) {
662             throw new Failure("** FAILURE debuggee connection **");
663         }
664     }
