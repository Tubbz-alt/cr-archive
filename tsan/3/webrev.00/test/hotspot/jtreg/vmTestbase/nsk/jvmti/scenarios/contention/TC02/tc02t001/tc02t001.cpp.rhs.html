 47 static jint findLineNumber(jvmtiEnv *jvmti, jthread thread) {
 48     jmethodID method = NULL;
 49     jlocation location;
 50     jvmtiLineNumberEntry* table = NULL;
 51     jint count = 0;
 52     jint line = 0;
 53     int i;
 54 
 55     if (!NSK_JVMTI_VERIFY(jvmti->GetFrameLocation(thread, 0, &method, &location)))
 56         return 0;
 57 
 58     if (!NSK_VERIFY(method != NULL))
 59         return 0;
 60 
 61     if (!NSK_VERIFY(location != -1))
 62         return 0;
 63 
 64     if (!NSK_JVMTI_VERIFY(jvmti->GetLineNumberTable(method, &count, &table)))
 65         return 0;
 66 
 67     if (!NSK_VERIFY(table != NULL))
 68         return 0;
 69 
 70     if (!NSK_VERIFY(count > 0))
 71         return 0;
 72 
 73     for (i = 0; i < count; i++) {
 74         if (location < table[i].start_location) {
 75             break;
 76         }
 77     }
 78 
 79     line = table[i-1].line_number;
 80 
 81     if (table != NULL) {
 82         if (!NSK_JVMTI_VERIFY(jvmti->Deallocate((unsigned char*)table)))
 83             return 0;
 84     }
 85 
 86     return line;
 87 }

 91 void JNICALL
 92 MonitorContendedEnter(jvmtiEnv *jvmti, JNIEnv* jni, jthread thr, jobject obj) {
 93     jint line = 0;
 94 
 95     if (!NSK_VERIFY(thr != NULL)) {
 96         nsk_jvmti_setFailStatus();
 97         NSK_COMPLAIN1("MonitorContendedEnter event: thread=%p\n", thr);
 98         return;
 99     }
100 
101     if (!NSK_VERIFY(obj != NULL)) {
102         nsk_jvmti_setFailStatus();
103         NSK_COMPLAIN1("MonitorContendedEnter event: object=%p\n", obj);
104         return;
105     }
106 
107     /* check if event is for tested thread and object */
108     if (jni->IsSameObject(thread, thr) &&
109             jni->IsSameObject(object_M, obj)) {
110 
111         line = findLineNumber(jvmti, thread);
112         if (!line) {
113             nsk_jvmti_setFailStatus();
114             NSK_COMPLAIN2("MonitorContendedEnter event: thread=%p, object=%p\n",
115                 thr, obj);
116             return;
117         }
118 
119         NSK_DISPLAY3("MonitorContendedEnter event: thread=%p, object=%p, line=%d\n",
120             thr, obj, line);
121 
122         /* workaround of 4527285 bug: in -Xint mode GetFrameLocation
123            returns the location after the monitor enter.
124          */
125         if (!NSK_VERIFY(line == lines[enterEventsCount] ||
126                 line == (lines[enterEventsCount] + 1))) {
127             nsk_jvmti_setFailStatus();
128             NSK_COMPLAIN3("MonitorContendedEnter event: thread=%p, object=%p, line=%d\n",
129                 thr, obj, line);
130         }
131 
132         enterEventsCount++;
133     }
134 }

183 static int prepare(jvmtiEnv* jvmti, JNIEnv* jni) {
184     const char* THREAD_NAME = "Debuggee Thread";
185     const char* FIELD_SIG = "Ljava/lang/Object;";
186     jvmtiThreadInfo info;
187     jthread *threads = NULL;
188     jint threads_count = 0;
189     jfieldID field = NULL;
190     jclass klass = NULL;
191     int i;
192 
193     NSK_DISPLAY0("Prepare: find tested thread\n");
194 
195     /* get all live threads */
196     if (!NSK_JVMTI_VERIFY(jvmti->GetAllThreads(&threads_count, &threads)))
197         return NSK_FALSE;
198 
199     if (!NSK_VERIFY(threads_count > 0 && threads != NULL))
200         return NSK_FALSE;
201 
202     /* find tested thread */
203     for (i = 0; i < threads_count; i++) {
204         if (!NSK_VERIFY(threads[i] != NULL))
205             return NSK_FALSE;
206 
207         /* get thread information */
208         if (!NSK_JVMTI_VERIFY(jvmti->GetThreadInfo(threads[i], &info)))
209             return NSK_FALSE;
210 
211         NSK_DISPLAY3("    thread #%d (%s): %p\n", i, info.name, threads[i]);
212 
213         /* find by name */
214         if (info.name != NULL && (strcmp(info.name, THREAD_NAME) == 0)) {
215             thread = threads[i];
216         }
217 
218         if (info.name != NULL) {
219             if (!NSK_JVMTI_VERIFY(jvmti->Deallocate((unsigned char*)info.name)))
220                 return NSK_FALSE;
221         }
222     }
223 
224     /* deallocate threads list */
225     if (!NSK_JVMTI_VERIFY(jvmti->Deallocate((unsigned char*)threads)))
226         return NSK_FALSE;
227 
228     if (thread == NULL) {
229         NSK_COMPLAIN0("Debuggee thread not found");
230         return NSK_FALSE;
231     }
232 
233     /* make thread accessable for a long time */
234     if (!NSK_JNI_VERIFY(jni, (thread = jni->NewGlobalRef(thread)) != NULL))
235         return NSK_FALSE;
236 
237     /* get tested thread class */
238     if (!NSK_JNI_VERIFY(jni, (klass = jni->GetObjectClass(thread)) != NULL))
239         return NSK_FALSE;
240 
241     /* get tested thread field 'M' */
242     if (!NSK_JNI_VERIFY(jni, (field = jni->GetFieldID(klass, "M", FIELD_SIG)) != NULL))
243         return NSK_FALSE;
244 
245     if (!NSK_JNI_VERIFY(jni, (object_M = jni->GetObjectField(thread, field)) != NULL))
246         return NSK_FALSE;
247 
248     /* make object accessable for a long time */
249     if (!NSK_JNI_VERIFY(jni, (object_M = jni->NewGlobalRef(object_M)) != NULL))
250         return NSK_FALSE;
251 
252     /* enable MonitorContendedEntered event */
253     if (!NSK_JVMTI_VERIFY(
254             jvmti->SetEventNotificationMode(
255                 JVMTI_ENABLE, JVMTI_EVENT_MONITOR_CONTENDED_ENTER, NULL)))
256         return NSK_FALSE;
257 
258     /* enable MonitorContendedEntered event */
259     if (!NSK_JVMTI_VERIFY(
260             jvmti->SetEventNotificationMode(
261                 JVMTI_ENABLE, JVMTI_EVENT_MONITOR_CONTENDED_ENTERED, NULL)))
262         return NSK_FALSE;
263 
264     return NSK_TRUE;
265 }

267 static int clean(jvmtiEnv* jvmti, JNIEnv* jni) {
268 
269     /* disable MonitorContendedEntered event */
270     if (!NSK_JVMTI_VERIFY(
271             jvmti->SetEventNotificationMode(
272                 JVMTI_DISABLE, JVMTI_EVENT_MONITOR_CONTENDED_ENTERED, NULL)))
273         nsk_jvmti_setFailStatus();
274 
275     return NSK_TRUE;
276 }

280 /* agent algorithm */
281 static void JNICALL
282 agentProc(jvmtiEnv* jvmti, JNIEnv* jni, void* arg) {
283 
284     /* wait for initial sync */
285     if (!nsk_jvmti_waitForSync(timeout))
286         return;
287 
288     if (!prepare(jvmti, jni)) {
289         nsk_jvmti_setFailStatus();
290         return;
291     }
292 
293     /* resume debugee to catch MonitorContendedEntered events */
294     if (!(NSK_VERIFY(nsk_jvmti_resumeSync()) &&
295           NSK_VERIFY(nsk_jvmti_waitForSync(timeout))))
296         return;
297 
298     NSK_DISPLAY1("Number of MonitorContendedEnter events: %d\n",
299         enterEventsCount);
300 
301     if (!(NSK_VERIFY(enterEventsCount == 3))) {
302         nsk_jvmti_setFailStatus();
303     }
304 
305     NSK_DISPLAY1("Number of MonitorContendedEntered events: %d\n",
306         enteredEventsCount);
307 
308     if (!(NSK_VERIFY(enteredEventsCount == 3))) {
309         nsk_jvmti_setFailStatus();
310     }
311 
312     if (!clean(jvmti, jni)) {
313         nsk_jvmti_setFailStatus();
314         return;
315     }
316 
317     /* resume debugee after last sync */
318     if (!nsk_jvmti_resumeSync())
319         return;
320 }

324 /* agent library initialization */
325 #ifdef STATIC_BUILD
326 JNIEXPORT jint JNICALL Agent_OnLoad_tc02t001(JavaVM *jvm, char *options, void *reserved) {
327     return Agent_Initialize(jvm, options, reserved);
328 }
329 JNIEXPORT jint JNICALL Agent_OnAttach_tc02t001(JavaVM *jvm, char *options, void *reserved) {
330     return Agent_Initialize(jvm, options, reserved);
331 }
332 JNIEXPORT jint JNI_OnLoad_tc02t001(JavaVM *jvm, char *options, void *reserved) {
333     return JNI_VERSION_1_8;
334 }
335 #endif
336 jint Agent_Initialize(JavaVM *jvm, char *options, void *reserved) {
337     jvmtiEnv* jvmti = NULL;
338     jvmtiCapabilities caps;
339     jvmtiEventCallbacks callbacks;
340 
341     /* init framework and parse options */
342     if (!NSK_VERIFY(nsk_jvmti_parseOptions(options)))
343         return JNI_ERR;
344 
345     timeout = nsk_jvmti_getWaitTime() * 60000;
346     NSK_DISPLAY1("Timeout: %d msc\n", (int)timeout);
347 
348     /* create JVMTI environment */
349     if (!NSK_VERIFY((jvmti =
350             nsk_jvmti_createJVMTIEnv(jvm, reserved)) != NULL))
351         return JNI_ERR;
352 
353     /* add capabilities */
354     memset(&caps, 0, sizeof(caps));
355     caps.can_generate_monitor_events = 1;
356     caps.can_get_line_numbers = 1;
357     if (!NSK_JVMTI_VERIFY(jvmti->AddCapabilities(&caps)))
358         return JNI_ERR;
359 
360     memset(&callbacks, 0, sizeof(callbacks));
361     callbacks.MonitorContendedEnter = &MonitorContendedEnter;
362     callbacks.MonitorContendedEntered = &MonitorContendedEntered;
363     if (!NSK_JVMTI_VERIFY(jvmti->SetEventCallbacks(&callbacks, sizeof(callbacks))))
364         return JNI_ERR;
365 
366     /* register agent proc and arg */
367     if (!NSK_VERIFY(nsk_jvmti_setAgentProc(agentProc, NULL)))
368         return JNI_ERR;
369 
370     return JNI_OK;
371 }
