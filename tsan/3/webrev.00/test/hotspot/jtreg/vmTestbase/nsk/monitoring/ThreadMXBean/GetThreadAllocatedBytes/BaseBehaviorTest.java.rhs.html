 23 package nsk.monitoring.ThreadMXBean.GetThreadAllocatedBytes;
 24 
 25 import nsk.share.*;
 26 import nsk.monitoring.share.*;
 27 import nsk.monitoring.ThreadMXBean.*;
 28 
 29 /**
 30  * Tests getCurrentThreadAllocatedBytes(), getThreadAllocatedBytes(long id).
 31  * and getThreadAllocatedBytes(long[] ids),
 32  * functions of com.sun.management.ThreadMXBean
 33  * <p>
 34  * All methods should
 35  * <br>
 36  * -  return -1 if ThreadAllocatedMemory allocation is
 37  * not enabled
 38  *  <br>
 39  * - return -1 for not started thread
 40  *  <br>
 41  * - return > 0 value for any running thread
 42  *  <br>
 43  * - return -1 for finished thread
 44  */
 45 public class BaseBehaviorTest extends ThreadMXBeanTestBase {
 46 
 47     /**
 48      * Actually runs the test
 49      */
 50     public void run() {
 51         if (threadMXBean == null)
 52             return;
 53 
 54         // Expect -1 if thread allocated memory is disabled
 55         threadMXBean.setThreadAllocatedMemoryEnabled(false);
 56         long result = threadMXBean.getCurrentThreadAllocatedBytes();
 57         if (result != -1)
 58             throw new TestFailure("Failure! getCurrentThreadAllocatedBytes() should "
 59                     + "return -1 if ThreadAllocatedMemoryEnabled is set to false. "
 60                     + "Received : " + result);
 61         threadMXBean.setThreadAllocatedMemoryEnabled(true);
 62         // Expect >= 0 value for current thread
 63         result = threadMXBean.getCurrentThreadAllocatedBytes();
 64         if (result < 0)
 65             throw new TestFailure("Failure! getCurrentThreadAllocatedBytes() should "
 66                     + "return >= 0 value for current thread. Received : " + result);
 67         // Expect >= 0 value for current thread from getThreadAllocatedBytes(id)
 68         result = threadMXBean.getThreadAllocatedBytes(Thread.currentThread().getId());
 69         if (result < 0)
 70             throw new TestFailure("Failure! getThreadAllocatedBytes(id) should "
 71                     + "return >= 0 value for current thread. Received : " + result);
 72 
 73         MXBeanTestThread thread = new MXBeanTestThread();
 74         long id = thread.getId();
 75         long[] idArr = new long[] { id };

 76         long[] resultArr;
 77 
 78         // Expect -1 for not started threads
 79         result = threadMXBean.getThreadAllocatedBytes(id);
 80         if (result != -1)
 81             throw new TestFailure("Failure! getThreadAllocatedBytes(long id) should "
 82                     + "return -1 for not started threads. Recieved : " + result);
 83         resultArr = threadMXBean.getThreadAllocatedBytes(idArr);
 84         if (resultArr[0] != -1)
 85             throw new TestFailure("Failure! getThreadAllocatedBytes(long[] ids) should "
 86                     + "return -1 for not started threads. Recieved : " + resultArr[0]);
 87         BarrierHandler handler = startThreads(thread);
 88         try {
 89             handler.proceed();
 90             // Expect -1 for running thread if ThreadAllocatedMemory (CpuTime) is disabled
 91             threadMXBean.setThreadAllocatedMemoryEnabled(false);
 92             result = threadMXBean.getThreadAllocatedBytes(id);
 93             if (result != -1)
 94                 throw new TestFailure("Failure! getThreadAllocatedBytes(long id) should "
 95                     + "return -1 if ThreadAllocatedMemoryEnabled is set to false. "
 96                     + "Recieved : " + result);
 97             resultArr = threadMXBean.getThreadAllocatedBytes(idArr);
 98             if (resultArr[0] != -1)
 99                 throw new TestFailure("Failure! getThreadAllocatedBytes(long[] ids) should "
100                     + "return -1 if ThreadAllocatedMemoryEnabled is set to false. "
101                     + "Recieved : " + resultArr[0]);
102 
103             threadMXBean.setThreadAllocatedMemoryEnabled(true);
104             // Expect >= 0 value for running threads
105             result = threadMXBean.getThreadAllocatedBytes(id);
106             if (result < 0)
107             throw new TestFailure("Failure! getThreadAllocatedBytes(long id) should "
108                     + "return > 0 value for RUNNING thread. Recieved : " + result);
109             resultArr = threadMXBean.getThreadAllocatedBytes(idArr);
110             if (resultArr[0] < 0)
111                 throw new TestFailure("Failure! getThreadAllocatedBytes(long[] ids) should "
112                     + "return > 0 value for RUNNING thread. Recieved : " + resultArr[0]);
113         } finally {
114             // Let thread finish
115             handler.finish();
116         }
117         try {
118             thread.join();
119         } catch (InterruptedException e) {}
120         // Expect -1 for finished thread
121         result = threadMXBean.getThreadAllocatedBytes(id);
122         if (result != -1)
123             throw new TestFailure("Failure! getThreadAllocatedBytes(long id) should "
124                     + "return -1 for finished threads. Recieved : " + result);
125         resultArr = threadMXBean.getThreadAllocatedBytes(idArr);
126         if (resultArr[0] != -1)
127             throw new TestFailure("Failure! getThreadAllocatedBytes(long[] ids) should "
128                     + "return -1 for finished threads. Recieved : " + resultArr[0]);
129         log.info("BaseBehaviorTest passed.");
130     }
131 
132     /**
133      * Entry point for java program
134      * @param args sets the test configuration
135      */
136     public static void main(String[] args) {
137         ThreadMXBeanTestBase test = new BaseBehaviorTest();
138         Monitoring.runTest(test, test.setGarbageProducer(args));
139     }
140 }
