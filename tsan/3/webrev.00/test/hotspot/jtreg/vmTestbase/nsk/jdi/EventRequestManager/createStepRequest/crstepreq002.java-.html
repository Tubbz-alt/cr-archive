 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createStepRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *  - Creates a new disabled StepRequest.                               <BR>
 47  *  - STEP_MIN, STEP_LINE, STEP_INTO, STEP_OVER, STEP_OUT are valid     <BR>
 48  *    arguments, that is, they don't throw IllegalArgumentException.    <BR>
 49  *  - Values larger and lesser corresponding above are invalid arguments,<BR>
 50  *    that is, they do throw IllegalArgumentException.                  <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertions,                             <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new testcase and invokes                     <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the checks required.                         <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase
250     */

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = threadByName("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         ThreadReference thread     = null;
309         String          threadName = "thread1";
310 
311         StepRequest  stRequest = null;
312 
313         int minDepth, maxDepth;
314         int minSize,  maxSize;
315 
316 
317         log1("     TESTING BEGINS");
318 
319         for (int i = 0; ; i++) {
320 
321             vm.resume();
322             breakpointForCommunication();
323 
324             int instruction = ((IntegerValue)
325                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
326 
327             if (instruction == 0) {
328                 vm.resume();
329                 break;
330             }
331 
332             log1(":::::: case: # " + i);
333 
334             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
335 
336             thread = threadByName(threadName);
337 
338             if (StepRequest.STEP_MIN > StepRequest.STEP_LINE) {
339                 maxSize = StepRequest.STEP_MIN;
340                 minSize = StepRequest.STEP_LINE;
341             } else {
342                 minSize = StepRequest.STEP_MIN;
343                 maxSize = StepRequest.STEP_LINE;
344             }
345 
346             if ( (StepRequest.STEP_INTO > StepRequest.STEP_OVER) &&
347                  (StepRequest.STEP_INTO > StepRequest.STEP_OUT) )
348                 maxDepth = StepRequest.STEP_INTO;
349 
350             if ( (StepRequest.STEP_OVER > StepRequest.STEP_INTO) &&
351                  (StepRequest.STEP_OVER > StepRequest.STEP_OUT) )
352                 maxDepth = StepRequest.STEP_OVER;
353 
354             if ( (StepRequest.STEP_OUT > StepRequest.STEP_OVER) &&
355                  (StepRequest.STEP_OUT > StepRequest.STEP_INTO) )
356                 maxDepth = StepRequest.STEP_OUT;
357 
358             if ( (StepRequest.STEP_INTO < StepRequest.STEP_OVER) &&
359                  (StepRequest.STEP_INTO < StepRequest.STEP_OUT) )
360                 minDepth = StepRequest.STEP_INTO;
361 
362             if ( (StepRequest.STEP_OVER < StepRequest.STEP_INTO) &&
363                  (StepRequest.STEP_OVER < StepRequest.STEP_OUT) )
364                 minDepth = StepRequest.STEP_OVER;
365 
366             if ( (StepRequest.STEP_OUT < StepRequest.STEP_OVER) &&
367                  (StepRequest.STEP_OUT < StepRequest.STEP_INTO) )
368                 minDepth = StepRequest.STEP_OUT;
369 
370             try {
371                 log2("......creating: eventRManager.createStepRequest(thread, StepRequest.STEP_MIN, StepRequest.STEP_INTO);");
372                 stRequest = eventRManager.createStepRequest(thread, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
373                 log2("      checking up on if request is disabled");
374                 if (stRequest.isEnabled()) {
375                     testExitCode = FAILED;
376                     log3("ERROR: request is enabled");
377                 }
378             } catch ( IllegalArgumentException e ) {
379                 testExitCode = FAILED;
380                 log3("ERROR: IllegalArgumentException for STEP_MIN - STEP_INTO");
381             }
382             eventRManager.deleteEventRequest(stRequest);
383             try {
384                 log2("......creating: eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OVER);");
385                 stRequest = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OVER);
386                 log2("      checking up on if request is disabled");
387                 if (stRequest.isEnabled()) {
388                     testExitCode = FAILED;
389                     log3("ERROR: request is enabled");
390                 }
391             } catch ( IllegalArgumentException e ) {
392                 testExitCode = FAILED;
393                 log3("ERROR: IllegalArgumentException for STEP_LINE - STEP_OVER");
394             }
395             eventRManager.deleteEventRequest(stRequest);
396             try {
397                 log2("......creating: eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OUT);");
398                 stRequest = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OUT);
399                 log2("      checking up on if request is disabled");
400                 if (stRequest.isEnabled()) {
401                     testExitCode = FAILED;
402                     log3("ERROR: request is enabled");
403                 }
404             } catch ( IllegalArgumentException e ) {
405                 testExitCode = FAILED;
406                 log3("ERROR: IllegalArgumentException for STEP_LINE - STEP_OUT");
407             }
408             eventRManager.deleteEventRequest(stRequest);
409             log2("......creating: eventRManager.createStepRequest(thread, minSize -1, StepRequest.STEP_OUT);");
410             log2("                 IllegalArgumentException is expected");
411             try {
412                 stRequest = eventRManager.createStepRequest(thread, minSize -1, StepRequest.STEP_OUT);
413                 testExitCode = FAILED;
414                 log3("ERROR:  no IllegalArgumentException thrown");
415             } catch ( IllegalArgumentException e ) {
416             }
417             eventRManager.deleteEventRequest(stRequest);
418             log2("......creating: eventRManager.createStepRequest(thread, maxSize +1, StepRequest.STEP_OUT);");
419             log2("                 IllegalArgumentException is expected");
420             try {
421                 stRequest = eventRManager.createStepRequest(thread, maxSize +1, StepRequest.STEP_OUT);
422                 testExitCode = FAILED;
423                 log3("ERROR:  no IllegalArgumentException thrown");
424             } catch ( IllegalArgumentException e ) {
425             }
426             eventRManager.deleteEventRequest(stRequest);
427             log2("......creating: eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, minDepth -1);");
428             log2("                 IllegalArgumentException is expected");
429             try {
430                 stRequest = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, minDepth -1);
431                 testExitCode = FAILED;
432                 log3("ERROR:  no IllegalArgumentException thrown");
433             } catch ( IllegalArgumentException e ) {
434             }
435             eventRManager.deleteEventRequest(stRequest);
436             log2("......creating: eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, maxDepth +1);");
437             log2("                 IllegalArgumentException is expected");
438             try {
439                 stRequest = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, maxDepth +1);
440                 testExitCode = FAILED;
441                 log3("ERROR:  no IllegalArgumentException thrown");
442             } catch ( IllegalArgumentException e ) {
443             }
444 
445             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
446         }
447         log1("    TESTING ENDS");
448         return;
449     }

451     private ThreadReference threadByName(String name)
452                  throws JDITestRuntimeException {
453 
454         List         all = vm.allThreads();
455         ListIterator li  = all.listIterator();
456 
457         for (; li.hasNext(); ) {
458             ThreadReference thread = (ThreadReference) li.next();
459             if (thread.name().equals(name))
460                 return thread;
461         }
462         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
463     }

465    /*
466     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
467     *                                             String, String, String)
468     *
469     * It sets up a breakpoint at given line number within a given method in a given class
470     * for a given thread.
471     *
472     * Return value: BreakpointRequest object  in case of success
473     *
474     * JDITestRuntimeException   in case of an Exception thrown within the method
475     */
476 
477     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
478                                                   ReferenceType testedClass,
479                                                   String methodName,
480                                                   String bpLine,
481                                                   String property)
482             throws JDITestRuntimeException {
483 
484         log2("......setting up a breakpoint:");
485         log2("       thread: " + thread + "; class: " + testedClass +
486                         "; method: " + methodName + "; line: " + bpLine);
487 
488         List              alllineLocations = null;
489         Location          lineLocation     = null;
490         BreakpointRequest breakpRequest    = null;
491 
492         try {
493             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
494 
495             alllineLocations = method.allLineLocations();
496 
497             int n =
498                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
499             if (n > alllineLocations.size()) {
500                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
501             } else {
502                 lineLocation = (Location) alllineLocations.get(n);
503                 try {
504                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
505                     breakpRequest.putProperty("number", property);
506                     breakpRequest.addThreadFilter(thread);
507                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
508                 } catch ( Exception e1 ) {
509                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
510                     breakpRequest    = null;
511                 }
512             }
513         } catch ( Exception e2 ) {
514             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
515             breakpRequest    = null;
516         }
517 
518         if (breakpRequest == null) {
519             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
520             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
521         }
522 
523         log2("      a breakpoint has been set up");
524         return breakpRequest;
525     }

528     private void getEventSet()
529                  throws JDITestRuntimeException {
530         try {
531 //            log2("       eventSet = eventQueue.remove(waitTime);");
532             eventSet = eventQueue.remove(waitTime);
533             if (eventSet == null) {
534                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
535             }
536 //            log2("       eventIterator = eventSet.eventIterator;");
537             eventIterator = eventSet.eventIterator();
538         } catch ( Exception e ) {
539             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
540         }
541     }

544     private void breakpointForCommunication()
545                  throws JDITestRuntimeException {
546 
547         log2("breakpointForCommunication");
548         getEventSet();
549 
550         if (eventIterator.nextEvent() instanceof BreakpointEvent)
551             return;
552 
553         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
554     }
