 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.enable()</code>               <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request has been deleted.                 <BR>
 48  *            IllegalThreadStateException -                     <BR>
 49  *            if this is a StepRequest and                      <BR>
 50  *            the thread named in the request has died.         <BR>
 51  *  The cases to test include all sub-types of EventRequest.    <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee prepares new check case (as if) and invokes           <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent,                                  <BR>
 74  *   the debugger performs the check case required.                     <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : " + e);
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : " + e);
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

253     * Return value: 0 - normal end of the test
254     *               1 - ubnormal end of the test
255     *               2 - VMDisconnectedException while test phase
256     */

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if (eventIterator.nextEvent() instanceof VMDeathEvent)
266                 return 0;
267 
268             log3("ERROR: last event is not the VMDeathEvent");
269             return 1;
270         } catch ( VMDisconnectedException e ) {
271             log3("ERROR: VMDisconnectedException : " + e);
272             return 2;
273         } catch ( Exception e ) {
274             log3("ERROR: Exception : " + e);
275             return 1;
276         }
277 
278     }

280     private void testRun()
281                  throws JDITestRuntimeException, Exception {
282 
283         eventRManager = vm.eventRequestManager();
284 
285         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
286         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
287         cpRequest.addClassFilter(debuggeeName);
288 
289         cpRequest.enable();
290         vm.resume();
291         getEventSet();
292         cpRequest.disable();
293 
294         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
295         debuggeeClass = event.referenceType();
296 
297         if (!debuggeeClass.name().equals(debuggeeName))
298            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
299 
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304 
305         ThreadReference   mainThread = threadByName("main");
306 
307         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
308                                              debuggeeClass,
309                                             bPointMethod, lineForComm, "zero");
310         bpRequest.enable();
311 
312     //------------------------------------------------------  testing section
313 
314 
315         EventRequest  eventRequest1 = null;
316 
317         String        fieldName1    = "var101";
318         String        fieldName2    = "excObj";
319 
320         ThreadReference thread1     = null;
321         String          threadName1 = "thread1";
322 
323         ReferenceType testClassReference = null;
324 
325 
326         log1("     TESTING BEGINS");
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345 
346             switch (i) {
347 
348               case 0:
349                      thread1 = threadByName(threadName1);
350 
351                      log2("......setting up StepRequest");
352                      eventRequest1 = eventRManager.createStepRequest
353                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
354                      vm.resume();
355                      breakpointForCommunication();
356                      break;
357 
358               case 1:
359                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
360 
361                      log2("......setting up AccessWatchpointRequest");
362                      eventRequest1 = eventRManager.createAccessWatchpointRequest
363                                     (testClassReference.fieldByName(fieldName1));
364                      break;
365 
366               case 2:
367                      log2(".....setting up ModificationWatchpointRequest");
368                      eventRequest1 = eventRManager.createModificationWatchpointRequest
369                                     (testClassReference.fieldByName(fieldName1));
370                      break;
371 
372               case 3:
373                      log2(".....setting up ClassPrepareRequest");
374                      eventRequest1 = eventRManager.createClassPrepareRequest();
375                      break;
376 
377               case 4:
378                      log2(".....setting up ClassUnloadRequest");
379                      eventRequest1 = eventRManager.createClassUnloadRequest();
380                      break;
381 
382               case 5:
383                      log2(".....setting up MethodEntryRequest");
384                      eventRequest1 = eventRManager.createMethodEntryRequest();
385                      break;
386 
387               case 6:
388                      log2(".....setting up MethodExitRequest");
389                      eventRequest1 = eventRManager.createMethodExitRequest();
390                      break;
391 
392               case 7:
393                      log2(".....setting up ThreadDeathRequest");
394                      eventRequest1 = eventRManager.createThreadDeathRequest();
395                      break;
396 
397               case 8:
398                      log2(".....setting up ThreadStartRequest");
399                      eventRequest1 = eventRManager.createThreadStartRequest();
400                      break;
401 
402               case 9:
403                      log2(".....setting up VMDeathRequest");
404                      eventRequest1 = eventRManager.createVMDeathRequest();
405                      break;
406 
407               case 10:
408                      log2(".....setting up ExceptionRequest");
409                      eventRequest1 = eventRManager.createExceptionRequest(
410                                      (ReferenceType)
411                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
412                                      true, true );
413                      break;
414 
415               case 11:
416                      log2(".....setting up BreakpointRequest");
417                      eventRequest1 = eventRManager.createBreakpointRequest(location);
418                      break;
419 
420 
421               default:
422                       throw new JDITestRuntimeException("** default case 2 **");
423             }
424 
425             if (eventRequest1 instanceof StepRequest) {
426                 try {
427                     log2("......eventRequest1.enable(); IllegalThreadStateException is expected");
428                     eventRequest1.enable();
429                     testExitCode = FAILED;
430                     log3("ERROR:  NO  IllegalThreadStateException for StepRequest");
431                 } catch ( IllegalThreadStateException e ) {
432                     log2("       IllegalThreadStateException");
433                 }
434             }
435 
436             log2("......eventRManager.deleteEventRequest(eventRequest1);");
437             eventRManager.deleteEventRequest(eventRequest1);
438             try {
439                 log2("......eventRequest1.enable();  InvalidRequestStateException is expected");
440                 eventRequest1.enable();
441                 testExitCode = FAILED;
442                 log3("ERROR:  NO  InvalidRequestStateException");
443             } catch ( InvalidRequestStateException e ) {
444                     log2("       InvalidRequestStateException");
445             }
446 
447             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
448         }
449         log1("    TESTING ENDS");
450         return;
451     }

453     private ThreadReference threadByName(String name)
454                  throws JDITestRuntimeException {
455 
456         List         all = vm.allThreads();
457         ListIterator li  = all.listIterator();
458 
459         for (; li.hasNext(); ) {
460             ThreadReference thread = (ThreadReference) li.next();
461             if (thread.name().equals(name))
462                 return thread;
463         }
464         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
465     }

467    /*
468     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
469     *                                             String, String, String)
470     *
471     * It sets up a breakpoint at given line number within a given method in a given class
472     * for a given thread.
473     *
474     * Return value: BreakpointRequest object  in case of success
475     *
476     * JDITestRuntimeException   in case of an Exception thrown within the method
477     */
478 
479     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
480                                                   ReferenceType testedClass,
481                                                   String methodName,
482                                                   String bpLine,
483                                                   String property)
484             throws JDITestRuntimeException {
485 
486         log2("......setting up a breakpoint:");
487         log2("       thread: " + thread + "; class: " + testedClass +
488                         "; method: " + methodName + "; line: " + bpLine);
489 
490         List              alllineLocations = null;
491         Location          lineLocation     = null;
492         BreakpointRequest breakpRequest    = null;
493 
494         try {
495             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
496 
497             alllineLocations = method.allLineLocations();
498 
499             int n =
500                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
501             if (n > alllineLocations.size()) {
502                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
503             } else {
504                 lineLocation = (Location) alllineLocations.get(n);
505 
506 // this is only for this test to get Location object
507 location = lineLocation;
508 
509                 try {
510                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
511                     breakpRequest.putProperty("number", property);
512                     breakpRequest.addThreadFilter(thread);
513                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
514                 } catch ( Exception e1 ) {
515                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
516                     breakpRequest    = null;
517                 }
518             }
519         } catch ( Exception e2 ) {
520             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
521             breakpRequest    = null;
522         }
523 
524         if (breakpRequest == null) {
525             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
526             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
527         }
528 
529         log2("      a breakpoint has been set up");
530         return breakpRequest;
531     }

534     private void getEventSet()
535                  throws JDITestRuntimeException {
536         try {
537 //            log2("       eventSet = eventQueue.remove(waitTime);");
538             eventSet = eventQueue.remove(waitTime);
539             if (eventSet == null) {
540                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
541             }
542 //            log2("       eventIterator = eventSet.eventIterator;");
543             eventIterator = eventSet.eventIterator();
544         } catch ( Exception e ) {
545             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
546         }
547     }

550     private void breakpointForCommunication()
551                  throws JDITestRuntimeException {
552 
553         log2("breakpointForCommunication");
554         getEventSet();
555 
556         if (eventIterator.nextEvent() instanceof BreakpointEvent)
557             return;
558 
559         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
560     }
