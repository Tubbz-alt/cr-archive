 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BooleanType.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks that for a mirror of a boolean value the casts<BR>
 42  * from Type to BooleanType and from PrimitiveType to BooleanType<BR>
 43  * don't throw ClassCastException.                              <BR>
 44  * <BR>
 45  * The test has three phases and works as follows.              <BR>
 46  * <BR>
 47  * In first phase,                                                      <BR>
 48  * upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 52  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 53  * and waits for the event within the predefined time interval.         <BR>
 54  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 57  * within debuggee's special methodForCommunication().                  <BR>
 58  * <BR>
 59  * In second phase to check the casts,                                  <BR>
 60  * the debugger and the debuggee perform the following.                 <BR>
 61  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * - The debuggee prepares a boolean field and invokes                  <BR>
 63  *   the methodForCommunication to be suspended and                     <BR>
 64  *   to inform the debugger with the event.                             <BR>
 65  * - Upon getting the BreakpointEvent,                                  <BR>
 66  *   the debugger performs the checks.                                  <BR>
 67  * Note. To inform each other of needed actions, the debugger and       <BR>
 68  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 69  * <BR>
 70  * In third phase when at the end,                                      <BR>
 71  * the debuggee changes the value of the "instruction"                  <BR>
 72  * to inform the debugger of checks finished, and both end.             <BR>
 73  * <BR>
 74  */
 75

 98     public static int run (String argv[], PrintStream out) {
 99 
100         int exitCode = new booleantype001().runThis(argv, out);

102         if (exitCode != PASSED) {

104         }
105         return testExitCode;
106     }

154     private int runThis (String argv[], PrintStream out) {
155 
156         argsHandler     = new ArgumentHandler(argv);
157         logHandler      = new Log(out, argsHandler);
158         Binder binder   = new Binder(argsHandler, logHandler);
159 
160         waitTime        = argsHandler.getWaitTime() * 60000;
161 
162         try {
163             log2("launching a debuggee :");
164             log2("       " + debuggeeName);
165             if (argsHandler.verbose()) {
166                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
167             } else {
168                 debuggee = binder.bindToDebugee(debuggeeName);
169             }
170             if (debuggee == null) {
171                 log3("ERROR: no debuggee launched");
172                 return FAILED;
173             }
174             log2("debuggee launched");
175         } catch ( Exception e ) {
176             log3("ERROR: Exception : " + e);
177             log2("       test cancelled");
178             return FAILED;
179         }
180 
181         debuggee.redirectOutput(logHandler);
182 
183         vm = debuggee.VM();
184 
185         eventQueue = vm.eventQueue();
186         if (eventQueue == null) {
187             log3("ERROR: eventQueue == null : TEST ABORTED");
188             vm.exit(PASS_BASE);
189             return FAILED;
190         }
191 
192         log2("invocation of the method runTest()");
193         switch (runTest()) {
194 
195             case 0 :  log2("test phase has finished normally");
196                       log2("   waiting for the debuggee to finish ...");
197                       debuggee.waitFor();
198 
199                       log2("......getting the debuggee's exit status");
200                       int status = debuggee.getStatus();
201                       if (status != PASS_BASE) {
202                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
203                               status + " != PASS_BASE");
204                           testExitCode = FAILED;
205                       } else {
206                           log2("......debuggee returned expected exit status: " +
207                               status + " == PASS_BASE");
208                       }
209                       break;
210 
211             default : log3("ERROR: runTest() returned unexpected value");
212 
213             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
214                       log2("......forcing: vm.exit();");
215                       testExitCode = FAILED;
216                       try {
217                           vm.exit(PASS_BASE);
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : e");
220                       }
221                       break;
222 
223             case 2 :  log3("test cancelled due to VMDisconnectedException");
224                       log2("......trying: vm.process().destroy();");
225                       testExitCode = FAILED;
226                       try {
227                           Process vmProcess = vm.process();
228                           if (vmProcess != null) {
229                               vmProcess.destroy();
230                           }
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : e");
233                       }
234                       break;
235             }
236 
237         return testExitCode;
238     }

241    /*
242     * Return value: 0 - normal end of the test
243     *               1 - ubnormal end of the test
244     *               2 - VMDisconnectedException while test phase
245     */
246 
247     private int runTest() {
248 
249         try {
250             testRun();
251 
252             log2("waiting for VMDeathEvent");
253             getEventSet();
254             if (eventIterator.nextEvent() instanceof VMDeathEvent)
255                 return 0;
256 
257             log3("ERROR: last event is not the VMDeathEvent");
258             return 1;
259         } catch ( VMDisconnectedException e ) {
260             log3("ERROR: VMDisconnectedException : " + e);
261             return 2;
262         } catch ( Exception e ) {
263             log3("ERROR: Exception : " + e);
264             return 1;
265         }
266 
267     }

269     private void testRun()
270                  throws JDITestRuntimeException, Exception {
271 
272         eventRManager = vm.eventRequestManager();
273 
274         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
275         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
276         cpRequest.addClassFilter(debuggeeName);
277 
278         cpRequest.enable();
279         vm.resume();
280         getEventSet();
281         cpRequest.disable();
282 
283         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
284         debuggeeClass = event.referenceType();
285 
286         if (!debuggeeClass.name().equals(debuggeeName))
287            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
288 
289         log2("      received: ClassPrepareEvent for debuggeeClass");
290 
291         String bPointMethod = "methodForCommunication";
292         String lineForComm  = "lineForComm";
293         BreakpointRequest bpRequest;
294 
295         bpRequest = settingBreakpoint(threadByName("main"),
296                                       debuggeeClass,
297                                       bPointMethod, lineForComm, "zero");
298         bpRequest.enable();
299 
300     //------------------------------------------------------  testing section
301 
302         log1("     TESTING BEGINS");
303 
304         for (int i = 0; ; i++) {
305 
306             vm.resume();
307             breakpointForCommunication();
308 
309             int instruction = ((IntegerValue)
310                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
311 
312             if (instruction == 0) {
313                 vm.resume();
314                 break;
315             }
316 
317             log1(":::::: case: # " + i);
318 
319             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
320 
321             log2("......String name = 'bl';");
322             String name = "bl";
323 
324             log2("      getting: Type type = debuggeeClass.getValue(debuggeeClass.fieldByName(name)).type();");
325             Type type = debuggeeClass.getValue(debuggeeClass.fieldByName(name)).type();
326             log2("      PrimitiveType primitiveType = (PrimitiveType) type;");
327             PrimitiveType primitiveType = (PrimitiveType) type;
328 
329             BooleanType booleanType = null;
330             try {
331                 log2("      checking up on cast: BooleanType booleanType = (BooleanType) type;");
332                     booleanType = (BooleanType) type;
333             } catch ( ClassCastException e ) {
334                     testExitCode = FAILED;
335                     log3("ERROR: ClassCastException");
336             }
337             try {
338                 log2("      checking up on cast: BooleanType booleanType = (BooleanType) primitiveType;");
339                     booleanType = (BooleanType) primitiveType;
340             } catch ( ClassCastException e ) {
341                     testExitCode = FAILED;
342                     log3("ERROR: ClassCastException");
343             }
344             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
345         }
346         log1("    TESTING ENDS");
347         return;
348     }

350     private ThreadReference threadByName(String name)
351                  throws JDITestRuntimeException {
352 
353         List         all = vm.allThreads();
354         ListIterator li  = all.listIterator();
355 
356         for (; li.hasNext(); ) {
357             ThreadReference thread = (ThreadReference) li.next();
358             if (thread.name().equals(name))
359                 return thread;
360         }
361         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
362     }

364    /*
365     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
366     *                                             String, String, String)
367     *
368     * It sets up a breakpoint at given line number within a given method in a given class
369     * for a given thread.
370     *
371     * Return value: BreakpointRequest object  in case of success
372     *
373     * JDITestRuntimeException   in case of an Exception thrown within the method
374     */
375 
376     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
377                                                   ReferenceType testedClass,
378                                                   String methodName,
379                                                   String bpLine,
380                                                   String property)
381             throws JDITestRuntimeException {
382 
383         log2("......setting up a breakpoint:");
384         log2("       thread: " + thread + "; class: " + testedClass +
385                         "; method: " + methodName + "; line: " + bpLine);
386 
387         List              alllineLocations = null;
388         Location          lineLocation     = null;
389         BreakpointRequest breakpRequest    = null;
390 
391         try {
392             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
393 
394             alllineLocations = method.allLineLocations();
395 
396             int n =
397                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
398             if (n > alllineLocations.size()) {
399                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
400             } else {
401                 lineLocation = (Location) alllineLocations.get(n);
402                 try {
403                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
404                     breakpRequest.putProperty("number", property);
405                     breakpRequest.addThreadFilter(thread);
406                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
407                 } catch ( Exception e1 ) {
408                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
409                     breakpRequest    = null;
410                 }
411             }
412         } catch ( Exception e2 ) {
413             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
414             breakpRequest    = null;
415         }
416 
417         if (breakpRequest == null) {
418             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
419             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
420         }
421 
422         log2("      a breakpoint has been set up");
423         return breakpRequest;
424     }

427     private void getEventSet()
428                  throws JDITestRuntimeException {
429         try {
430 //            log2("       eventSet = eventQueue.remove(waitTime);");
431             eventSet = eventQueue.remove(waitTime);
432             if (eventSet == null) {
433                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
434             }
435 //            log2("       eventIterator = eventSet.eventIterator;");
436             eventIterator = eventSet.eventIterator();
437         } catch ( Exception e ) {
438             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
439         }
440     }

443     private void breakpointForCommunication()
444                  throws JDITestRuntimeException {
445 
446         log2("breakpointForCommunication");
447         getEventSet();
448 
449         if (eventIterator.nextEvent() instanceof BreakpointEvent)
450             return;
451 
452         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
453     }
