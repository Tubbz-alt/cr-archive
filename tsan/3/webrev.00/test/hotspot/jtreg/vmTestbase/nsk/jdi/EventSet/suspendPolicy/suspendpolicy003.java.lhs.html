 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a ModificationWatchpointEvent set,  <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a ModificationWatchpointRequest, resumes      <BR>
 70  *   the debuggee, and waits for the ModificationWatchpointEvent.       <BR>
 71  * - The debuggee invokes the special method which makes modification   <BR>
 72  *   of a predefined variable to be resulting in the event.             <BR>
 73  * - Upon getting new event, the debugger                               <BR>
 74  *   performs the check corresponding to the event.                     <BR>
 75  * <BR>
 76  * Note. To inform each other of needed actions, the debugger and       <BR>
 77  *       and the debuggee use debuggee's variable "instruction".        <BR>
 78  * In third phase, at the end,                                          <BR>
 79  * the debuggee changes the value of the "instruction"                  <BR>
 80  * to inform the debugger of checks finished, and both end.             <BR>
 81  * <BR>
 82  */
 83

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
266                 log3("ERROR: last event is not the VMDeathEvent");
267                 return 1;
268             }
269 
270             log2("waiting for VMDisconnectEvent");
271             getEventSet();
272             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
273                 log3("ERROR: last event is not the VMDisconnectEvent");
274                 return 1;
275             }
276 
277             return 0;
278 
279         } catch ( VMDisconnectedException e ) {
280             log3("ERROR: VMDisconnectedException : " + e);
281             return 2;
282         } catch ( Exception e ) {
283             log3("ERROR: Exception : " + e);
284             return 1;
285         }
286

289     private void testRun()
290                  throws JDITestRuntimeException, Exception {
291 
292         if ( !vm.canWatchFieldModification() ) {
293             log2("......vm.canWatchFieldModification == false :: test cancelled");
294             vm.exit(PASS_BASE);
295             return;
296         }
297 
298         eventRManager = vm.eventRequestManager();
299 
300         log2("......getting ClassPrepareEvent for debuggee's class");
301         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
302         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
303         cpRequest.addClassFilter(debuggeeName);
304         cpRequest.enable();
305         vm.resume();
306         getEventSet();
307         cpRequest.disable();
308 
309         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
310         debuggeeClass = event.referenceType();
311 
312         if (!debuggeeClass.name().equals(debuggeeName))
313            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
314         log2("      received: ClassPrepareEvent for debuggeeClass");
315 
316         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
317 
318         String            bPointMethod = "methodForCommunication";
319         String            lineForComm  = "lineForComm";
320         BreakpointRequest bpRequest;
321         ThreadReference   mainThread = threadByName("main");
322         bpRequest = settingBreakpoint(mainThread,
323                                       debuggeeClass,
324                                       bPointMethod, lineForComm, "zero");
325         bpRequest.enable();
326 
327         vm.resume();
328 
329     //------------------------------------------------------  testing section
330 
331         log1("     TESTING BEGINS");
332 
333         EventRequest eventRequest1 = null;
334         EventRequest eventRequest2 = null;
335         EventRequest eventRequest3 = null;
336 
337         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
338         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
339         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
340         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
341 
342         int policyExpected[] = {    SUSPEND_NONE,
343                                     SUSPEND_THREAD,
344                                     SUSPEND_ALL,
345                                     SUSPEND_THREAD,
346                                     SUSPEND_ALL,
347                                     SUSPEND_ALL,
348                                     SUSPEND_ALL      };
349         int policy = 0;
350 
351 
352         ReferenceType testClassReference = null;
353 
354         String fieldName = "var2";
355 
356 
357         for (int i = 0; ; i++) {
358 
359 
360             breakpointForCommunication();
361 
362             int instruction = ((IntegerValue)
363                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
364             if (instruction == 0) {
365                 vm.resume();
366                 break;
367             }
368 
369 
370             log1(":::::: case: # " + i);
371 
372             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
373 
374             switch (i) {
375 
376               case 0:
377                       testClassReference =
378                            (ReferenceType) vm.classesByName(testedClassName).get(0);
379 
380                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
381                                              testClassReference, fieldName,
382                                              SUSPEND_NONE, "ModificationWatchpointRequest1");
383                       eventRequest1.enable();
384                       break;
385 
386               case 1:
387                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
388                                              testClassReference, fieldName,
389                                              SUSPEND_THREAD, "ModificationWatchpointRequest2");
390                       eventRequest2.enable();
391                       break;
392 
393               case 2:
394                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
395                                              testClassReference, fieldName,
396                                              SUSPEND_ALL, "ModificationWatchpointRequest3");
397                       eventRequest3.enable();
398                       break;
399 
400               case 3:
401                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
402                                              testClassReference, fieldName,
403                                              SUSPEND_NONE, "ModificationWatchpointRequest4");
404                       eventRequest1.enable();
405                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
406                                              testClassReference, fieldName,
407                                              SUSPEND_THREAD, "ModificationWatchpointRequest5");
408                       eventRequest2.enable();
409                       break;
410 
411               case 4:
412                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
413                                              testClassReference, fieldName,
414                                              SUSPEND_NONE, "ModificationWatchpointRequest6");
415                       eventRequest1.enable();
416                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
417                                              testClassReference, fieldName,
418                                              SUSPEND_ALL, "ModificationWatchpointRequest7");
419                       eventRequest3.enable();
420                       break;
421 
422               case 5:
423                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
424                                              testClassReference, fieldName,
425                                              SUSPEND_THREAD, "ModificationWatchpointRequest8");
426                       eventRequest2.enable();
427                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
428                                              testClassReference, fieldName,
429                                              SUSPEND_ALL, "ModificationWatchpointRequest9");
430                       eventRequest3.enable();
431                       break;
432 
433               case 6:
434                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
435                                              testClassReference, fieldName,
436                                              SUSPEND_NONE, "ModificationWatchpointRequest10");
437                       eventRequest1.enable();
438                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
439                                              testClassReference, fieldName,
440                                              SUSPEND_THREAD, "ModificationWatchpointRequest11");
441                       eventRequest2.enable();
442                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
443                                              testClassReference, fieldName,
444                                              SUSPEND_ALL, "ModificationWatchpointRequest12");
445                       eventRequest3.enable();
446                       break;
447 
448 
449               default:
450                       throw new JDITestRuntimeException("** default case 2 **");
451             }
452 
453             log2("......waiting for new AccessWatchpointEvent : " + i);
454             mainThread.resume();
455             getEventSet();
456 
457             Event newEvent = eventIterator.nextEvent();
458             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
459             if ( !( newEvent instanceof ModificationWatchpointEvent)) {
460                  log3("ERROR: new event is not ModificationWatchpointEvent");
461                  testExitCode = FAILED;
462             } else {
463                 log2("       Event is instanceof ModificationWatchpointEvent");
464                 policy = eventSet.suspendPolicy();
465                 if (policy != policyExpected[i]) {
466                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
467                     log3("       eventSet.suspendPolicy() == " + policy);
468                     log3("       policyExpected           == " + policyExpected[i]);
469                     testExitCode = FAILED;
470                 }
471             }
472 
473             switch (i) {
474               case 0: eventRequest1.disable();                          break;
475               case 1: eventRequest2.disable();                          break;
476               case 2: eventRequest3.disable();                          break;
477               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
478               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
479               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
480               case 6: eventRequest1.disable(); eventRequest2.disable();
481                       eventRequest3.disable();                          break;
482            }
483 
484             switch (policy) {
485               case SUSPEND_NONE   :                      break;
486               case SUSPEND_THREAD : mainThread.resume(); break;
487               case SUSPEND_ALL    : vm.resume();         break;
488               default: throw new JDITestRuntimeException("** default case 3 **");
489             }
490 
491             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
492         }
493         log1("    TESTING ENDS");
494         return;
495     }

497     private ThreadReference threadByName(String name)
498                  throws JDITestRuntimeException {
499 
500         List         all = vm.allThreads();
501         ListIterator li  = all.listIterator();
502 
503         for (; li.hasNext(); ) {
504             ThreadReference thread = (ThreadReference) li.next();
505             if (thread.name().equals(name))
506                 return thread;
507         }
508         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
509     }

511    /*
512     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
513     *                                             String, String, String)
514     *
515     * It sets up a breakpoint at given line number within a given method in a given class
516     * for a given thread.
517     *
518     * Return value: BreakpointRequest object  in case of success
519     *
520     * JDITestRuntimeException   in case of an Exception thrown within the method
521     */
522 
523     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
524                                                   ReferenceType testedClass,
525                                                   String methodName,
526                                                   String bpLine,
527                                                   String property)
528             throws JDITestRuntimeException {
529 
530         log2("......setting up a breakpoint:");
531         log2("       thread: " + thread + "; class: " + testedClass +
532                         "; method: " + methodName + "; line: " + bpLine);
533 
534         List              alllineLocations = null;
535         Location          lineLocation     = null;
536         BreakpointRequest breakpRequest    = null;
537 
538         try {
539             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
540 
541             alllineLocations = method.allLineLocations();
542 
543             int n =
544                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
545             if (n > alllineLocations.size()) {
546                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
547             } else {
548                 lineLocation = (Location) alllineLocations.get(n);
549                 try {
550                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
551                     breakpRequest.putProperty("number", property);
552                     breakpRequest.addThreadFilter(thread);
553                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
554                 } catch ( Exception e1 ) {
555                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
556                     breakpRequest    = null;
557                 }
558             }
559         } catch ( Exception e2 ) {
560             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
561             breakpRequest    = null;
562         }
563 
564         if (breakpRequest == null) {
565             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
566             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
567         }
568 
569         log2("      a breakpoint has been set up");
570         return breakpRequest;
571     }

574     private void getEventSet()
575                  throws JDITestRuntimeException {
576         try {
577 //            log2("       eventSet = eventQueue.remove(waitTime);");
578             eventSet = eventQueue.remove(waitTime);
579             if (eventSet == null) {
580                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
581             }
582 //            log2("       eventIterator = eventSet.eventIterator;");
583             eventIterator = eventSet.eventIterator();
584         } catch ( Exception e ) {
585             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
586         }
587     }

590     private void breakpointForCommunication()
591                  throws JDITestRuntimeException {
592 
593         log2("breakpointForCommunication");
594         getEventSet();
595 
596         if (eventIterator.nextEvent() instanceof BreakpointEvent)
597             return;
598 
599         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
600     }

604     private ModificationWatchpointRequest settingModificationWatchpointRequest (
605                                                   ThreadReference thread,
606                                                   ReferenceType   testedClass,
607                                                   String          fieldName,
608                                                   int             suspendPolicy,
609                                                   String          property        )
610             throws JDITestRuntimeException {
611         try {
612             log2("......setting up ModificationWatchpointRequest:");
613             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
614             Field field = testedClass.fieldByName(fieldName);
615 
616             ModificationWatchpointRequest
617             awr = eventRManager.createModificationWatchpointRequest(field);
618             awr.putProperty("number", property);
619             awr.addThreadFilter(thread);
620             awr.setSuspendPolicy(suspendPolicy);
621 
622             log2("      ModificationWatchpointRequest has been set up");
623             return awr;
624         } catch ( Exception e ) {
625             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
626             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
627             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
628         }
