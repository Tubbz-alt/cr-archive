353     private ThreadReference threadByName(String name)
354                  throws JDITestRuntimeException {
355 
356         List         all = vm.allThreads();
357         ListIterator li  = all.listIterator();
358 
359         for (; li.hasNext(); ) {
360             ThreadReference thread = (ThreadReference) li.next();
361             if (thread.name().equals(name))
362                 return thread;
363         }
364         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
365     }
