 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a StepEvent set,                    <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one thread with all possible combinations:                   <BR>
 50  *    NONE, THREAD, ALL                                         <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the above,                                  <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger sets up a StepRequest, resumes                        <BR>
 69  *   the debuggee, and waits for the StepEvent.                         <BR>
 70  * - The debuggee invokes the special method to be resulting in the event.<BR>
 71  * - Upon getting new event, the debugger                               <BR>
 72  *   performs the check corresponding to the event.                     <BR>
 73  * <BR>
 74  * Note. To inform each other of needed actions, the debugger and       <BR>
 75  *       and the debuggee use debuggee's variable "instruction".        <BR>
 76  * In third phase, at the end,                                          <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */

157     private int runThis (String argv[], PrintStream out) {
158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : e");
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : e");
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

244    /*
245     * Return value: 0 - normal end of the test
246     *               1 - ubnormal end of the test
247     *               2 - VMDisconnectedException while test phase
248     */

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
258                 log3("ERROR: last event is not the VMDeathEvent");
259                 return 1;
260             }
261 
262             log2("waiting for VMDisconnectEvent");
263             getEventSet();
264             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
265                 log3("ERROR: last event is not the VMDisconnectEvent");
266                 return 1;
267             }
268 
269             return 0;
270 
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         log2("......getting ClassPrepareEvent for debuggee's class");
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300         log2("      received: ClassPrepareEvent for debuggeeClass");
301 
302         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
303 
304         String            bPointMethod = "methodForCommunication";
305         String            lineForComm  = "lineForComm";
306         BreakpointRequest bpRequest;
307         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
308         bpRequest = settingBreakpoint(mainThread,
309                                       debuggeeClass,
310                                       bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313         vm.resume();
314 
315     //------------------------------------------------------  testing section
316 
317         log1("     TESTING BEGINS");
318 
319         EventRequest eventRequest1 = null;
320         EventRequest eventRequest2 = null;
321         EventRequest eventRequest3 = null;
322 
323         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
324         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
325         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
326         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
327 
328         int policyExpected[] = {    SUSPEND_NONE,
329                                     SUSPEND_THREAD,
330                                     SUSPEND_ALL,
331                                     SUSPEND_THREAD,
332                                     SUSPEND_ALL,
333                                     SUSPEND_ALL,
334                                     SUSPEND_ALL      };
335         int policy = 0;
336 
337 
338         for (int i = 0; ; i++) {
339 
340             breakpointForCommunication();
341 
342             int instruction = ((IntegerValue)
343                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
344             if (instruction == 0) {
345                 vm.resume();
346                 break;
347             }
348 
349             log1(":::::: case: # " + i);
350 
351             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
352 
353             switch (i) {
354 
355               case 0:
356                       eventRequest1 = settingStepRequest (mainThread,
357                                              SUSPEND_NONE, "StepRequest1");
358                       eventRequest1.enable();
359                       break;
360 
361               case 1:
362                       eventRequest2 = settingStepRequest (mainThread,
363                                              SUSPEND_THREAD, "StepRequest2");
364                       eventRequest2.enable();
365                       break;
366 
367               case 2:
368                       eventRequest3 = settingStepRequest (mainThread,
369 
370                                              SUSPEND_ALL, "StepRequest3");
371                       eventRequest3.enable();
372                       break;
373 
374 
375               default:
376                       throw new JDITestRuntimeException("** default case 2 **");
377             }
378 
379             mainThread.resume();
380             getEventSet();
381 
382             if ( !(eventIterator.nextEvent() instanceof StepEvent)) {
383                  log3("ERROR: new event is not StepEvent");
384                  testExitCode = FAILED;
385             } else {
386                 log2("......got : instanceof StepEvent");
387                 policy = eventSet.suspendPolicy();
388                 if (policy != policyExpected[i]) {
389                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
390                     log3("       eventSet.suspendPolicy() == " + policy);
391                     log3("       policyExpected           == " + policyExpected[i]);
392                     testExitCode = FAILED;
393                 }
394             }
395 
396             switch (i) {
397               case 0: eventRequest1.disable();                          break;
398               case 1: eventRequest2.disable();                          break;
399               case 2: eventRequest3.disable();                          break;
400            }
401 
402             switch (policy) {
403               case SUSPEND_NONE   :                      break;
404               case SUSPEND_THREAD : mainThread.resume(); break;
405               case SUSPEND_ALL    : vm.resume();         break;
406               default: throw new JDITestRuntimeException("** default case 3 **");
407             }
408 
409             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
410         }
411         log1("    TESTING ENDS");
412         return;
413     }

415    /*
416     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
417     *                                             String, String, String)
418     *
419     * It sets up a breakpoint at given line number within a given method in a given class
420     * for a given thread.
421     *
422     * Return value: BreakpointRequest object  in case of success
423     *
424     * JDITestRuntimeException   in case of an Exception thrown within the method
425     */
426 
427     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
428                                                   ReferenceType testedClass,
429                                                   String methodName,
430                                                   String bpLine,
431                                                   String property)
432             throws JDITestRuntimeException {
433 
434         log2("......setting up a breakpoint:");
435         log2("       thread: " + thread + "; class: " + testedClass +
436                         "; method: " + methodName + "; line: " + bpLine);
437 
438         List              alllineLocations = null;
439         Location          lineLocation     = null;
440         BreakpointRequest breakpRequest    = null;
441 
442         try {
443             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
444 
445             alllineLocations = method.allLineLocations();
446 
447             int n =
448                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
449             if (n > alllineLocations.size()) {
450                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
451             } else {
452                 lineLocation = (Location) alllineLocations.get(n);
453                 try {
454                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
455                     breakpRequest.putProperty("number", property);
456                     breakpRequest.addThreadFilter(thread);
457                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
458                 } catch ( Exception e1 ) {
459                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
460                     breakpRequest    = null;
461                 }
462             }
463         } catch ( Exception e2 ) {
464             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
465             breakpRequest    = null;
466         }
467 
468         if (breakpRequest == null) {
469             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
470             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
471         }
472 
473         log2("      a breakpoint has been set up");
474         return breakpRequest;
475     }

478     private void getEventSet()
479                  throws JDITestRuntimeException {
480         try {
481 //            log2("       eventSet = eventQueue.remove(waitTime);");
482             eventSet = eventQueue.remove(waitTime);
483             if (eventSet == null) {
484                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
485             }
486 //            log2("       eventIterator = eventSet.eventIterator;");
487             eventIterator = eventSet.eventIterator();
488         } catch ( Exception e ) {
489             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
490         }
491     }

495                  throws JDITestRuntimeException {

501             return;

504     }

508     private StepRequest settingStepRequest ( ThreadReference thread,
509                                              int             suspendPolicy,
510                                              String          property        )
511             throws JDITestRuntimeException {
512         try {
513             log2("......setting up StepRequest:");
514             log2("       thread: " + thread + "; property: " + property);
515 
516             StepRequest
517             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OVER);
518             str.putProperty("number", property);
519             str.addCountFilter(1);
520             str.setSuspendPolicy(suspendPolicy);
521 
522             log2("      StepRequest has been set up");
523             return str;
524         } catch ( Exception e ) {
525             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
526             log3("       StepRequest HAS NOT BEEN SET UP");
527             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
528         }
529     }
