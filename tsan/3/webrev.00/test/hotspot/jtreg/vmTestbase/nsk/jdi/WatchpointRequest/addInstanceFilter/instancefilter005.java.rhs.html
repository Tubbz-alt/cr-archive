 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: UnsupportedOperationException -                   <BR>
 47  *       if the target virtual machine                          <BR>
 48  *       does not support this operation.                       <BR>
 49  * This request is AccessWatchpointRequest.                     <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger creates             <BR>
 72  *   a WatchpointRequest and if addInstanceFilter() is not supported,   <BR>
 73  *   invokes the method on the request expecting                        <BR>
 74  *     to catch UnsupportedOperationException.                          <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

157     private int runThis (String argv[], PrintStream out) {
158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : " + e);
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

244    /*
245     * Return value: 0 - normal end of the test
246     *               1 - ubnormal end of the test
247     *               2 - VMDisconnectedException while test phase
248     */

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if (eventIterator.nextEvent() instanceof VMDeathEvent)
258                 return 0;
259 
260             log3("ERROR: last event is not the VMDeathEvent");
261             return 1;
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }
269 
270     }

272     private void testRun()
273                  throws JDITestRuntimeException, Exception {
274 
275         eventRManager = vm.eventRequestManager();
276 
277         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
278         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
279         cpRequest.addClassFilter(debuggeeName);
280 
281         cpRequest.enable();
282         vm.resume();
283         getEventSet();
284         cpRequest.disable();
285 
286         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
287         debuggeeClass = event.referenceType();
288 
289         if (!debuggeeClass.name().equals(debuggeeName))
290            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
291 
292         log2("      received: ClassPrepareEvent for debuggeeClass");
293 
294         String bPointMethod = "methodForCommunication";
295         String lineForComm  = "lineForComm";
296 
297         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
298 
299         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
300                                              debuggeeClass,
301                                             bPointMethod, lineForComm, "zero");
302         bpRequest.enable();
303 
304     //------------------------------------------------------  testing section
305 
306         log1("     TESTING BEGINS");
307 
308         EventRequest eventRequest1 = null;
309 
310         ThreadReference thread1 = null;
311         ThreadReference thread2 = null;
312 
313         String thread1Name = "thread1";
314         String thread2Name = "thread2";
315 
316         String property1 = "AccessWatchpointRequest1";
317 
318         String fieldName = "var1";
319 
320         ReferenceType testClassReference = null;
321 
322         String         arrayName = "objTC";
323         ObjectReference instance = null;
324 
325 
326         for (int i = 0; ; i++) {
327 
328             vm.resume();
329             breakpointForCommunication();
330 
331             int instruction = ((IntegerValue)
332                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
333 
334             if (instruction == 0) {
335                 vm.resume();
336                 break;
337             }
338 
339             log1(":::::: case: # " + i);
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
342 
343             switch (i) {
344 
345               case 0:
346                       testClassReference =
347                            (ReferenceType) vm.classesByName(testedClassName).get(0);
348 
349                       thread1 = (ThreadReference) debuggeeClass.getValue(
350                                                   debuggeeClass.fieldByName(thread1Name));
351 
352                       eventRequest1 = setting2AccessWatchpointRequest (thread1,
353                                              testClassReference, fieldName,
354                                              EventRequest.SUSPEND_NONE, property1);
355 
356                       instance = (ObjectReference)
357                                  ((ArrayReference) debuggeeClass.getValue(
358                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
359 
360                       if ( vm.canUseInstanceFilters() ) {
361                           log2("......vm.canUseInstanceFilters == true :: test cancelled");
362                       } else {
363 
364                           try {
365                               log2("......eventRequest1.addInstanceFilter(instance);");
366                               log2("        UnsupportedOperationException  expected");
367                               ((WatchpointRequest) eventRequest1).addInstanceFilter(instance);
368                               log3("ERROR: no UnsupportedOperationException ");
369                               testExitCode = FAILED;
370                           } catch ( InvalidRequestStateException e ) {
371                               log2("        UnsupportedOperationException ");
372                           }
373                       }
374 
375                       break;
376 
377               default:
378                       throw new JDITestRuntimeException("** default case 2 **");
379             }
380 
381             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
382         }
383         log1("    TESTING ENDS");
384         return;
385     }

387    /*
388     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
389     *                                             String, String, String)
390     *
391     * It sets up a breakpoint at given line number within a given method in a given class
392     * for a given thread.
393     *
394     * Return value: BreakpointRequest object  in case of success
395     *
396     * JDITestRuntimeException   in case of an Exception thrown within the method
397     */
398 
399     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
400                                                   ReferenceType testedClass,
401                                                   String methodName,
402                                                   String bpLine,
403                                                   String property)
404             throws JDITestRuntimeException {
405 
406         log2("......setting up a breakpoint:");
407         log2("       thread: " + thread + "; class: " + testedClass +
408                         "; method: " + methodName + "; line: " + bpLine);
409 
410         List              alllineLocations = null;
411         Location          lineLocation     = null;
412         BreakpointRequest breakpRequest    = null;
413 
414         try {
415             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
416 
417             alllineLocations = method.allLineLocations();
418 
419             int n =
420                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
421             if (n > alllineLocations.size()) {
422                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
423             } else {
424                 lineLocation = (Location) alllineLocations.get(n);
425                 try {
426                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
427                     breakpRequest.putProperty("number", property);
428                     breakpRequest.addThreadFilter(thread);
429                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
430                 } catch ( Exception e1 ) {
431                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
432                     breakpRequest    = null;
433                 }
434             }
435         } catch ( Exception e2 ) {
436             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
437             breakpRequest    = null;
438         }
439 
440         if (breakpRequest == null) {
441             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
442             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
443         }
444 
445         log2("      a breakpoint has been set up");
446         return breakpRequest;
447     }

450     private void getEventSet()
451                  throws JDITestRuntimeException {
452         try {
453 //            log2("       eventSet = eventQueue.remove(waitTime);");
454             eventSet = eventQueue.remove(waitTime);
455             if (eventSet == null) {
456                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
457             }
458 //            log2("       eventIterator = eventSet.eventIterator;");
459             eventIterator = eventSet.eventIterator();
460         } catch ( Exception e ) {
461             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
462         }
463     }

466     private void breakpointForCommunication()
467                  throws JDITestRuntimeException {
468 
469         log2("breakpointForCommunication");
470         getEventSet();
471 
472         if (eventIterator.nextEvent() instanceof BreakpointEvent)
473             return;
474 
475         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
476     }

480     private AccessWatchpointRequest setting2AccessWatchpointRequest (
481                                                   ThreadReference thread,
482                                                   ReferenceType   testedClass,
483                                                   String          fieldName,
484                                                   int             suspendPolicy,
485                                                   String          property        )
486             throws JDITestRuntimeException {
487         try {
488             log2("......setting up AccessWatchpointRequest:");
489             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
490             Field field = testedClass.fieldByName(fieldName);
491 
492             AccessWatchpointRequest
493             awr = eventRManager.createAccessWatchpointRequest(field);
494             awr.putProperty("number", property);
495 
496             if (thread != null)
497                 awr.addThreadFilter(thread);
498             awr.setSuspendPolicy(suspendPolicy);
499 
500             log2("      AccessWatchpointRequest has been set up");
501             return awr;
502         } catch ( Exception e ) {
503             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
504             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
505             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
506         }
507     }
