 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for an ExceptionEvent set,              <BR>
 46  * the method returns a value corresponding to one              <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a ExceptionRequest,                           <BR>
 70  *   resumes the debuggee, and waits for the ExceptionEvent.            <BR>
 71  * - The debuggee invokes the special method in which                   <BR>
 72  *   NullPointerException is thrown to be resulting in the event.       <BR>
 73  * - Upon getting new event, the debugger                               <BR>
 74  *   performs the check corresponding to the event.                     <BR>
 75  * <BR>
 76  * Note. To inform each other of needed actions, the debugger and       <BR>
 77  *       and the debuggee use debuggee's variable "instruction".        <BR>
 78  * In third phase, at the end,                                          <BR>
 79  * the debuggee changes the value of the "instruction"                  <BR>
 80  * to inform the debugger of checks finished, and both end.             <BR>
 81  * <BR>
 82  */
 83

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

253     * Return value: 0 - normal end of the test
254     *               1 - ubnormal end of the test
255     *               2 - VMDisconnectedException while test phase

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
266                 log3("ERROR: last event is not the VMDeathEvent");
267                 return 1;
268             }
269 
270             log2("waiting for VMDisconnectEvent");
271             getEventSet();
272             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
273                 log3("ERROR: last event is not the VMDisconnectEvent");
274                 return 1;
275             }
276 
277             return 0;
278 
279         } catch ( VMDisconnectedException e ) {
280             log3("ERROR: VMDisconnectedException : " + e);
281             return 2;
282         } catch ( Exception e ) {
283             log3("ERROR: Exception : " + e);
284             return 1;
285         }
286 
287     }

289     private void testRun()
290                  throws JDITestRuntimeException, Exception {
291 
292         eventRManager = vm.eventRequestManager();
293 
294         log2("......getting ClassPrepareEvent for debuggee's class");
295         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
296         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
297         cpRequest.addClassFilter(debuggeeName);
298         cpRequest.enable();
299         vm.resume();
300         getEventSet();
301         cpRequest.disable();
302 
303         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
304         debuggeeClass = event.referenceType();
305 
306         if (!debuggeeClass.name().equals(debuggeeName))
307            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
311 
312         String            bPointMethod = "methodForCommunication";
313         String            lineForComm  = "lineForComm";
314         BreakpointRequest bpRequest;
315         ThreadReference   mainThread = threadByName("main");
316         bpRequest = settingBreakpoint(mainThread,
317                                       debuggeeClass,
318                                       bPointMethod, lineForComm, "zero");
319         bpRequest.enable();
320 
321         vm.resume();
322 
323     //------------------------------------------------------  testing section
324 
325         log1("     TESTING BEGINS");
326 
327         EventRequest eventRequest1 = null;
328         EventRequest eventRequest2 = null;
329         EventRequest eventRequest3 = null;
330 
331         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
332         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
333         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
334         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
335 
336         int policyExpected[] = {    SUSPEND_NONE,
337                                     SUSPEND_THREAD,
338                                     SUSPEND_ALL,
339                                     SUSPEND_THREAD,
340                                     SUSPEND_ALL,
341                                     SUSPEND_ALL,
342                                     SUSPEND_ALL      };
343         int policy = 0;
344 
345 
346         ReferenceType testClassReference = null;
347 
348         String methodName = "method";
349         String bpLineName = "breakpointLine";
350 
351 
352         for (int i = 0; ; i++) {
353 
354 
355             breakpointForCommunication();
356 
357             int instruction = ((IntegerValue)
358                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
359             if (instruction == 0) {
360                 vm.resume();
361                 break;
362             }
363 
364 
365             log1(":::::: case: # " + i);
366 
367             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
368 
369             switch (i) {
370 
371               case 0:
372                       testClassReference =
373                            (ReferenceType) vm.classesByName(testedClassName).get(0);
374 
375                       eventRequest1 = settingExceptionRequest (mainThread,
376                                              debuggeeClass, //testClassReference,
377                                              SUSPEND_NONE, "ExceptionRequest1");
378                       eventRequest1.enable();
379                       break;
380 
381               case 1:
382                       eventRequest2 = settingExceptionRequest (mainThread,
383                                              debuggeeClass, //testClassReference,
384                                              SUSPEND_THREAD, "ExceptionRequest2");
385                       eventRequest2.enable();
386                       break;
387 
388               case 2:
389                       eventRequest3 = settingExceptionRequest (mainThread,
390                                              debuggeeClass, //testClassReference,
391                                              SUSPEND_ALL, "ExceptionRequest3");
392                       eventRequest3.enable();
393                       break;
394 
395               case 3:
396                       eventRequest1 = settingExceptionRequest (mainThread,
397                                              debuggeeClass, // testClassReference,
398                                              SUSPEND_NONE, "ExceptionRequest4");
399                       eventRequest1.enable();
400                       eventRequest2 = settingExceptionRequest (mainThread,
401                                              debuggeeClass, //testClassReference,
402                                              SUSPEND_THREAD, "ExceptionRequest5");
403                       eventRequest2.enable();
404                       break;
405 
406               case 4:
407                       eventRequest1 = settingExceptionRequest (mainThread,
408                                              debuggeeClass, //testClassReference,
409                                              SUSPEND_NONE, "ExceptionRequest6");
410                       eventRequest1.enable();
411                       eventRequest3 = settingExceptionRequest (mainThread,
412                                              debuggeeClass, //testClassReference,
413                                              SUSPEND_ALL, "ExceptionRequest7");
414                       eventRequest3.enable();
415                       break;
416 
417               case 5:
418                       eventRequest2 = settingExceptionRequest (mainThread,
419                                              debuggeeClass, //testClassReference,
420                                              SUSPEND_THREAD, "ExceptionRequest8");
421                       eventRequest2.enable();
422                       eventRequest3 = settingExceptionRequest (mainThread,
423                                              debuggeeClass, //testClassReference,
424                                              SUSPEND_ALL, "ExceptionRequest9");
425                       eventRequest3.enable();
426                       break;
427 
428               case 6:
429                       eventRequest1 = settingExceptionRequest (mainThread,
430                                              debuggeeClass, //testClassReference,
431                                              SUSPEND_NONE, "ExceptionRequest10");
432                       eventRequest1.enable();
433                       eventRequest2 = settingExceptionRequest (mainThread,
434                                              debuggeeClass, //testClassReference,
435                                              SUSPEND_THREAD, "ExceptionRequest11");
436                       eventRequest2.enable();
437                       eventRequest3 = settingExceptionRequest (mainThread,
438                                              debuggeeClass, //testClassReference,
439                                              SUSPEND_ALL, "ExceptionRequest12");
440                       eventRequest3.enable();
441                       break;
442 
443 
444               default:
445                       throw new JDITestRuntimeException("** default case 2 **");
446             }
447 
448             log2("......waiting for new ExceptionEvent : " + i);
449             mainThread.resume();
450             getEventSet();
451 
452             Event newEvent = eventIterator.nextEvent();
453             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
454             if ( !( newEvent instanceof ExceptionEvent)) {
455                  log3("ERROR: new event is not ExceptionEvent");
456                  testExitCode = FAILED;
457             } else {
458                 log2("       Event is instanceof ExceptionEvent");
459                 policy = eventSet.suspendPolicy();
460                 if (policy != policyExpected[i]) {
461                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
462                     log3("       eventSet.suspendPolicy() == " + policy);
463                     log3("       policyExpected           == " + policyExpected[i]);
464                     testExitCode = FAILED;
465                 }
466             }
467 
468             switch (i) {
469               case 0: eventRequest1.disable();                          break;
470               case 1: eventRequest2.disable();                          break;
471               case 2: eventRequest3.disable();                          break;
472               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
473               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
474               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
475               case 6: eventRequest1.disable(); eventRequest2.disable();
476                       eventRequest3.disable();                          break;
477            }
478 
479             switch (policy) {
480               case SUSPEND_NONE   :                      break;
481               case SUSPEND_THREAD : mainThread.resume(); break;
482               case SUSPEND_ALL    : vm.resume();         break;
483               default: throw new JDITestRuntimeException("** default case 3 **");
484             }
485 
486             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
487         }
488         log1("    TESTING ENDS");
489         return;
490     }

492     private ThreadReference threadByName(String name)
493                  throws JDITestRuntimeException {
494 
495         List         all = vm.allThreads();
496         ListIterator li  = all.listIterator();
497 
498         for (; li.hasNext(); ) {
499             ThreadReference thread = (ThreadReference) li.next();
500             if (thread.name().equals(name))
501                 return thread;
502         }
503         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
504     }

506    /*
507     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
508     *                                             String, String, String)
509     *
510     * It sets up a breakpoint at given line number within a given method in a given class
511     * for a given thread.
512     *
513     * Return value: BreakpointRequest object  in case of success
514     *
515     * JDITestRuntimeException   in case of an Exception thrown within the method
516     */
517 
518     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
519                                                   ReferenceType testedClass,
520                                                   String methodName,
521                                                   String bpLine,
522                                                   String property)
523             throws JDITestRuntimeException {
524 
525         log2("......setting up a breakpoint:");
526         log2("       thread: " + thread + "; class: " + testedClass +
527                         "; method: " + methodName + "; line: " + bpLine);
528 
529         List              alllineLocations = null;
530         Location          lineLocation     = null;
531         BreakpointRequest breakpRequest    = null;
532 
533         try {
534             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
535 
536             alllineLocations = method.allLineLocations();
537 
538             int n =
539                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
540             if (n > alllineLocations.size()) {
541                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
542             } else {
543                 lineLocation = (Location) alllineLocations.get(n);
544                 try {
545                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
546                     breakpRequest.putProperty("number", property);
547                     breakpRequest.addThreadFilter(thread);
548                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
549                 } catch ( Exception e1 ) {
550                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
551                     breakpRequest    = null;
552                 }
553             }
554         } catch ( Exception e2 ) {
555             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
556             breakpRequest    = null;
557         }
558 
559         if (breakpRequest == null) {
560             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
561             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
562         }
563 
564         log2("      a breakpoint has been set up");
565         return breakpRequest;
566     }

569     private void getEventSet()
570                  throws JDITestRuntimeException {
571         try {
572 //            log2("       eventSet = eventQueue.remove(waitTime);");
573             eventSet = eventQueue.remove(waitTime);
574             if (eventSet == null) {
575                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
576             }
577 //            log2("       eventIterator = eventSet.eventIterator;");
578             eventIterator = eventSet.eventIterator();
579         } catch ( Exception e ) {
580             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
581         }
582     }

585     private void breakpointForCommunication()
586                  throws JDITestRuntimeException {
587 
588         log2("breakpointForCommunication");
589         getEventSet();
590 
591         if (eventIterator.nextEvent() instanceof BreakpointEvent)
592             return;
593 
594         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
595     }

599     private ExceptionRequest settingExceptionRequest ( ThreadReference thread,
600                                                        ReferenceType   testedClass,
601                                                        int             suspendPolicy,
602                                                        String          property       )
603             throws JDITestRuntimeException {
604         try {
605             log2("......setting up ExceptionRequest:");
606             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
607 
608             ExceptionRequest
609             excr = eventRManager.createExceptionRequest(null, true, true);
610             excr.putProperty("number", property);
611             excr.addThreadFilter(thread);
612             excr.addClassFilter(testedClass);
613             excr.setSuspendPolicy(suspendPolicy);
614 
615             log2("      ExceptionRequest has been set up");
616             return excr;
617         } catch ( Exception e ) {
618             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
619             log3("       ExceptionRequest HAS NOT BEEN SET UP");
620             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
621         }
