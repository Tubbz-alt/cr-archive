 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadDeathRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadDeathRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  *            Filters may be added only to disabled requests.   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates an ThreadDeathRequest and enables it,                    <BR>
 73  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 74  *     to catch InvalidRequestStateException,                           <BR>
 75  *   - disables the request and invokes the method addThreadFilter()    <BR>
 76  *     expecting to catch no InvalidRequestStateException,              <BR>
 77  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 78  *     once again expecting to catch InvalidRequestStateException.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  *
 86 
 87  */
 88

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : e");
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if (eventIterator.nextEvent() instanceof VMDeathEvent)
265                 return 0;
266 
267             log3("ERROR: last event is not the VMDeathEvent");
268             return 1;
269         } catch ( VMDisconnectedException e ) {
270             log3("ERROR: VMDisconnectedException : " + e);
271             return 2;
272         } catch ( Exception e ) {
273             log3("ERROR: Exception : " + e);
274             return 1;
275         }
276

279     private void testRun()
280                  throws JDITestRuntimeException, Exception {
281 
282         eventRManager = vm.eventRequestManager();
283 
284         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
285         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
286         cpRequest.addClassFilter(debuggeeName);
287 
288         cpRequest.enable();
289         vm.resume();
290         getEventSet();
291         cpRequest.disable();
292 
293         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
294         debuggeeClass = event.referenceType();
295 
296         if (!debuggeeClass.name().equals(debuggeeName))
297            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
298 
299         log2("      received: ClassPrepareEvent for debuggeeClass");
300 
301         String bPointMethod = "methodForCommunication";
302         String lineForComm  = "lineForComm";
303         BreakpointRequest bpRequest;
304 
305         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
306                                       debuggeeClass,
307                                       bPointMethod, lineForComm, "zero");
308         bpRequest.enable();
309 
310     //------------------------------------------------------  testing section
311 
312         log1("     TESTING BEGINS");
313 
314         EventRequest eventRequest1 = null;
315 
316         ThreadReference thread1    = null;
317         String         thread1Name = "thread1";
318 
319         String property1 = "ThreadDeathRequest1";
320 
321 
322         for (int i = 0; ; i++) {
323 
324             vm.resume();
325             breakpointForCommunication();
326 
327             int instruction = ((IntegerValue)
328                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
329 
330             if (instruction == 0) {
331                 vm.resume();
332                 break;
333             }
334 
335             log1(":::::: case: # " + i);
336 
337             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
338 
339             switch (i) {
340 
341               case 0:
342                       thread1 = (ThreadReference) debuggeeClass.getValue(
343                                                   debuggeeClass.fieldByName(thread1Name));
344 
345                       eventRequest1 = setting2ThreadDeathRequest (null,
346                                              EventRequest.SUSPEND_NONE, "ThreadDeathRequest1");
347 
348                      log2("......eventRequest1.enable();");
349                      eventRequest1.enable();
350 
351                      try {
352                          log2("......eventRequest1.addThreadFilter(thread1);");
353                          log2("        InvalidRequestStateException expected");
354                          ((ThreadDeathRequest)eventRequest1).addThreadFilter(thread1);
355                          log3("ERROR: no InvalidRequestStateException");
356                          testExitCode = FAILED;
357                      } catch ( InvalidRequestStateException e ) {
358                          log2("        InvalidRequestStateException");
359                      }
360 
361                      log2("......eventRequest1.disable();");
362                      eventRequest1.disable();
363                      try {
364                          log2("......eventRequest1.addThreadFilter(thread1);");
365                          log2("        no InvalidRequestStateException expected");
366                          ((ThreadDeathRequest)eventRequest1).addThreadFilter(thread1);
367                          log2("        no InvalidRequestStateException");
368                      } catch ( InvalidRequestStateException e ) {
369                          log3("ERROR: InvalidRequestStateException");
370                          testExitCode = FAILED;
371                      }
372 
373                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
374                      eventRManager.deleteEventRequest(eventRequest1);
375                      try {
376                          log2("......eventRequest1.addThreadFilter(thread1);");
377                          log2("        InvalidRequestStateException expected");
378                          ((ThreadDeathRequest)eventRequest1).addThreadFilter(thread1);
379                          log3("ERROR: no InvalidRequestStateException");
380                          testExitCode = FAILED;
381                      } catch ( InvalidRequestStateException e ) {
382                          log2("        InvalidRequestStateException");
383                      }
384                       break;
385 
386               default:
387                       throw new JDITestRuntimeException("** default case 2 **");
388             }
389 
390             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
391         }
392         log1("    TESTING ENDS");
393         return;

396    /*
397     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
398     *                                             String, String, String)
399     *
400     * It sets up a breakpoint at given line number within a given method in a given class
401     * for a given thread.
402     *
403     * Return value: BreakpointRequest object  in case of success
404     *
405     * JDITestRuntimeException   in case of an Exception thrown within the method
406     */
407 
408     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
409                                                   ReferenceType testedClass,
410                                                   String methodName,
411                                                   String bpLine,
412                                                   String property)
413             throws JDITestRuntimeException {
414 
415         log2("......setting up a breakpoint:");
416         log2("       thread: " + thread + "; class: " + testedClass +
417                         "; method: " + methodName + "; line: " + bpLine);
418 
419         List              alllineLocations = null;
420         Location          lineLocation     = null;
421         BreakpointRequest breakpRequest    = null;
422 
423         try {
424             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
425 
426             alllineLocations = method.allLineLocations();
427 
428             int n =
429                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
430             if (n > alllineLocations.size()) {
431                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
432             } else {
433                 lineLocation = (Location) alllineLocations.get(n);
434                 try {
435                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
436                     breakpRequest.putProperty("number", property);
437                     breakpRequest.addThreadFilter(thread);
438                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
439                 } catch ( Exception e1 ) {
440                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
441                     breakpRequest    = null;
442                 }
443             }
444         } catch ( Exception e2 ) {
445             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
446             breakpRequest    = null;
447         }
448 
449         if (breakpRequest == null) {
450             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
451             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
452         }
453 
454         log2("      a breakpoint has been set up");
455         return breakpRequest;
456     }

459     private void getEventSet()
460                  throws JDITestRuntimeException {
461         try {
462 //            log2("       eventSet = eventQueue.remove(waitTime);");
463             eventSet = eventQueue.remove(waitTime);
464             if (eventSet == null) {
465                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
466             }
467 //            log2("       eventIterator = eventSet.eventIterator;");
468             eventIterator = eventSet.eventIterator();
469         } catch ( Exception e ) {
470             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
471         }
472     }

475     private void breakpointForCommunication()
476                  throws JDITestRuntimeException {
477 
478         log2("breakpointForCommunication");
479         getEventSet();
480 
481         if (eventIterator.nextEvent() instanceof BreakpointEvent)
482             return;
483 
484         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
485     }

489     private ThreadDeathRequest setting2ThreadDeathRequest( ThreadReference thread,
490                                                            int             suspendPolicy,
491                                                            String          property)
492             throws JDITestRuntimeException {
493         try {
494             ThreadDeathRequest tsr = eventRManager.createThreadDeathRequest();
495             if (thread != null)
496                 tsr.addThreadFilter(thread);
497             tsr.addCountFilter(1);
498             tsr.setSuspendPolicy(suspendPolicy);
499             tsr.putProperty("number", property);
500             return tsr;
501         } catch ( Exception e ) {
502             log3("ERROR: ATTENTION: Exception within settingThreadDeathRequest() : " + e);
503             log3("       ThreadDeathRequest HAS NOT BEEN SET UP");
504             throw new JDITestRuntimeException("** FAILURE to set up ThreadDeathRequest **");
505         }
506     }
