 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.canRequestVMDeathEvent()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test works as follows.                                   <BR>
 46  * Upon launching debuggee's VM which will be suspended,        <BR>
 47  * a debugger waits for the VMStartEvent within a predefined    <BR>
 48  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 49  * Upon getting the VMStartEvent, it makes the request                  <BR>
 50  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 51  * resumes the VM, and waits for the event within the predefined        <BR>
 52  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 53  * Upon getting the ClassPrepareEvent,                                  <BR>
 54  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 55  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 56  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 57  * to be suspended and to inform the debugger with the event.           <BR>
 58  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 59  * At the end, the debuggee changes the value of the "instruction"      <BR>
 60  * to inform the debugger of checks finished, and both end.             <BR>
 61  */
 62

141     private int runThis (String argv[], PrintStream out) {
142 
143         argsHandler     = new ArgumentHandler(argv);
144         logHandler      = new Log(out, argsHandler);
145         Binder binder   = new Binder(argsHandler, logHandler);
146 
147         waitTime        = argsHandler.getWaitTime() * 60000;
148 
149         try {
150             log2("launching a debuggee :");
151             log2("       " + debuggeeName);
152             if (argsHandler.verbose()) {
153                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
154             } else {
155                 debuggee = binder.bindToDebugee(debuggeeName);
156             }
157             if (debuggee == null) {
158                 log3("ERROR: no debuggee launched");
159                 return FAILED;
160             }
161             log2("debuggee launched");
162         } catch ( Exception e ) {
163             log3("ERROR: Exception : " + e);
164             log2("       test cancelled");
165             return FAILED;
166         }
167 
168         debuggee.redirectOutput(logHandler);
169 
170         vm = debuggee.VM();
171 
172         eventQueue = vm.eventQueue();
173         if (eventQueue == null) {
174             log3("ERROR: eventQueue == null : TEST ABORTED");
175             vm.exit(PASS_BASE);
176             return FAILED;
177         }
178 
179         log2("invocation of the method runTest()");
180         switch (runTest()) {
181 
182             case 0 :  log2("test phase has finished normally");
183                       log2("   waiting for the debuggee to finish ...");
184                       debuggee.waitFor();
185 
186                       log2("......getting the debuggee's exit status");
187                       int status = debuggee.getStatus();
188                       if (status != PASS_BASE) {
189                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
190                               status + " != PASS_BASE");
191                           testExitCode = FAILED;
192                       } else {
193                           log2("......debuggee returned expected exit status: " +
194                               status + " == PASS_BASE");
195                       }
196                       break;
197 
198             default : log3("ERROR: runTest() returned unexpected value");
199 
200             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
201                       log2("......forcing: vm.exit();");
202                       testExitCode = FAILED;
203                       try {
204                           vm.exit(PASS_BASE);
205                       } catch ( Exception e ) {
206                           log3("ERROR: Exception : e");
207                       }
208                       break;
209 
210             case 2 :  log3("test cancelled due to VMDisconnectedException");
211                       log2("......trying: vm.process().destroy();");
212                       testExitCode = FAILED;
213                       try {
214                           Process vmProcess = vm.process();
215                           if (vmProcess != null) {
216                               vmProcess.destroy();
217                           }
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : e");
220                       }
221                       break;
222             }
223 
224         return testExitCode;
225     }

229     * Return value: 0 - normal end of the test
230     *               1 - ubnormal end of the test
231     *               2 - VMDisconnectedException while test phase
232     */
233 
234     private int runTest() {
235 
236         try {
237             testRun();
238 
239             log2("waiting for VMDeathEvent");
240             getEventSet();
241             if (eventIterator.nextEvent() instanceof VMDeathEvent)
242                 return 0;
243 
244             log3("ERROR: last event is not the VMDeathEvent");
245             return 1;
246         } catch ( VMDisconnectedException e ) {
247             log3("ERROR: VMDisconnectedException : " + e);
248             return 2;
249         } catch ( Exception e ) {
250             log3("ERROR: Exception : " + e);
251             return 1;
252         }
253 
254     }

256     private void testRun()
257                  throws JDITestRuntimeException, Exception {
258 
259         eventRManager = vm.eventRequestManager();
260 
261         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
262         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
263         cpRequest.addClassFilter(debuggeeName);
264 
265         cpRequest.enable();
266         vm.resume();
267         getEventSet();
268         cpRequest.disable();
269 
270         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
271         debuggeeClass = event.referenceType();
272 
273         if (!debuggeeClass.name().equals(debuggeeName))
274            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
275 
276         log2("      received: ClassPrepareEvent for debuggeeClass");
277 
278         String bPointMethod = "methodForCommunication";
279         String lineForComm  = "lineForComm";
280         BreakpointRequest bpRequest;
281 
282         try {
283             bpRequest = settingBreakpoint(threadByName("main"),
284                                           debuggeeClass,
285                                           bPointMethod, lineForComm, "zero");
286         } catch ( Exception e ) {
287             throw e;
288         }
289         bpRequest.enable();
290 
291     //------------------------------------------------------  testing section
292 
293         log1("     TESTING BEGINS");
294 
295         for (int i = 0; ; i++) {
296 
297             vm.resume();
298             breakpointForCommunication();
299 
300             int instruction = ((IntegerValue)
301                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
302 
303             if (instruction == 0) {
304                 vm.resume();
305                 break;
306             }
307 
308             log1("  new check: # " + i);
309 
310             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
311 
312             log2("...... canRequestVMDeathEvent()");
313             if ( vm.canRequestVMDeathEvent() )
314                 log2("      vm.canRequestVMDeathEvent() == true");
315             else
316                 log2("      vm.canRequestVMDeathEvent() == false");
317 
318             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
319         }
320         log1("    TESTING ENDS");
321         return;
322     }

324     private ThreadReference threadByName(String name)
325                  throws JDITestRuntimeException {
326 
327         List         all = vm.allThreads();
328         ListIterator li  = all.listIterator();
329 
330         for (; li.hasNext(); ) {
331             ThreadReference thread = (ThreadReference) li.next();
332             if (thread.name().equals(name))
333                 return thread;
334         }
335         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
336     }

338    /*
339     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
340     *                                             String, String, String)
341     *
342     * It sets up a breakpoint at given line number within a given method in a given class
343     * for a given thread.
344     *
345     * Return value: BreakpointRequest object  in case of success
346     *
347     * JDITestRuntimeException   in case of an Exception thrown within the method
348     */
349 
350     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
351                                                   ReferenceType testedClass,
352                                                   String methodName,
353                                                   String bpLine,
354                                                   String property)
355             throws JDITestRuntimeException {
356 
357         log2("......setting up a breakpoint:");
358         log2("       thread: " + thread + "; class: " + testedClass +
359                         "; method: " + methodName + "; line: " + bpLine);
360 
361         List              alllineLocations = null;
362         Location          lineLocation     = null;
363         BreakpointRequest breakpRequest    = null;
364 
365         try {
366             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
367 
368             alllineLocations = method.allLineLocations();
369 
370             int n =
371                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
372             if (n > alllineLocations.size()) {
373                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
374             } else {
375                 lineLocation = (Location) alllineLocations.get(n);
376                 try {
377                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
378                     breakpRequest.putProperty("number", property);
379                     breakpRequest.addThreadFilter(thread);
380                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
381                 } catch ( Exception e1 ) {
382                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
383                     breakpRequest    = null;
384                 }
385             }
386         } catch ( Exception e2 ) {
387             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
388             breakpRequest    = null;
389         }
390 
391         if (breakpRequest == null) {
392             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
393             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
394         }
395 
396         log2("      a breakpoint has been set up");
397         return breakpRequest;
398     }

401     private void getEventSet()
402                  throws JDITestRuntimeException {
403         try {
404 //            log2("       eventSet = eventQueue.remove(waitTime);");
405             eventSet = eventQueue.remove(waitTime);
406             if (eventSet == null) {
407                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
408             }
409 //            log2("       eventIterator = eventSet.eventIterator;");
410             eventIterator = eventSet.eventIterator();
411         } catch ( Exception e ) {
412             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
413         }
414     }

417     private void breakpointForCommunication()
418                  throws JDITestRuntimeException {
419 
420         log2("breakpointForCommunication");
421         getEventSet();
422 
423         if (eventIterator.nextEvent() instanceof BreakpointEvent)
424             return;
425 
426         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
427     }
