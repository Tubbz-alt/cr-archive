352     private ThreadReference threadByName(String name)
353                  throws JDITestRuntimeException {
354 
355         List         all = vm.allThreads();
356         ListIterator li  = all.listIterator();
357 
358         for (; li.hasNext(); ) {
359             ThreadReference thread = (ThreadReference) li.next();
360             if (thread.name().equals(name))
361                 return thread;
362         }
363         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
364     }
