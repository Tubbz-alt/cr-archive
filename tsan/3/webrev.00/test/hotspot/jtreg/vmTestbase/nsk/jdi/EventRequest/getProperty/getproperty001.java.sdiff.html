449     private ThreadReference threadByName(String name)
450                  throws JDITestRuntimeException {
451 
452         List         all = vm.allThreads();
453         ListIterator li  = all.listIterator();
454 
455         for (; li.hasNext(); ) {
456             ThreadReference thread = (ThreadReference) li.next();
457             if (thread.name().equals(name))
458                 return thread;
459         }
460         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
461     }
