 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *     Restricts the events generated by this request           <BR>
 47  *     to those in the given thread.                            <BR>
 48  *  The cases to test include three thread's states:            <BR>
 49  *  not started, running, dead.                                 <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                              <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates a BreakpointRequest1 and                                 <BR>
 73  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 74  *   - creates a BreakpointRequest2, starts the thread2, and            <BR>
 75  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 76  *   - creates a BreakpointRequest3, waits the thread2 to die, and      <BR>
 77  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

161     private int runThis (String argv[], PrintStream out) {
162 
163         argsHandler     = new ArgumentHandler(argv);
164         logHandler      = new Log(out, argsHandler);
165         Binder binder   = new Binder(argsHandler, logHandler);
166 
167         waitTime        = argsHandler.getWaitTime() * 60000;
168 
169         try {
170             log2("launching a debuggee :");
171             log2("       " + debuggeeName);
172             if (argsHandler.verbose()) {
173                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
174             } else {
175                 debuggee = binder.bindToDebugee(debuggeeName);
176             }
177             if (debuggee == null) {
178                 log3("ERROR: no debuggee launched");
179                 return FAILED;
180             }
181             log2("debuggee launched");
182         } catch ( Exception e ) {
183             log3("ERROR: Exception : " + e);
184             log2("       test cancelled");
185             return FAILED;
186         }
187 
188         debuggee.redirectOutput(logHandler);
189 
190         vm = debuggee.VM();
191 
192         eventQueue = vm.eventQueue();
193         if (eventQueue == null) {
194             log3("ERROR: eventQueue == null : TEST ABORTED");
195             vm.exit(PASS_BASE);
196             return FAILED;
197         }
198 
199         log2("invocation of the method runTest()");
200         switch (runTest()) {
201 
202             case 0 :  log2("test phase has finished normally");
203                       log2("   waiting for the debuggee to finish ...");
204                       debuggee.waitFor();
205 
206                       log2("......getting the debuggee's exit status");
207                       int status = debuggee.getStatus();
208                       if (status != PASS_BASE) {
209                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
210                               status + " != PASS_BASE");
211                           testExitCode = FAILED;
212                       } else {
213                           log2("......debuggee returned expected exit status: " +
214                               status + " == PASS_BASE");
215                       }
216                       break;
217 
218             default : log3("ERROR: runTest() returned unexpected value");
219 
220             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
221                       log2("......forcing: vm.exit();");
222                       testExitCode = FAILED;
223                       try {
224                           vm.exit(PASS_BASE);
225                       } catch ( Exception e ) {
226                           log3("ERROR: Exception : " + e);
227                       }
228                       break;
229 
230             case 2 :  log3("test cancelled due to VMDisconnectedException");
231                       log2("......trying: vm.process().destroy();");
232                       testExitCode = FAILED;
233                       try {
234                           Process vmProcess = vm.process();
235                           if (vmProcess != null) {
236                               vmProcess.destroy();
237                           }
238                       } catch ( Exception e ) {
239                           log3("ERROR: Exception : " + e);
240                       }
241                       break;
242             }
243 
244         return testExitCode;
245     }

254     private int runTest() {
255 
256         try {
257             testRun();
258 
259             log2("waiting for VMDeathEvent");
260             getEventSet();
261             if (eventIterator.nextEvent() instanceof VMDeathEvent)
262                 return 0;
263 
264             log3("ERROR: last event is not the VMDeathEvent");
265             return 1;
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         eventRManager = vm.eventRequestManager();
280 
281         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
282         cpRequest.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD);
283         cpRequest.addClassFilter(debuggeeName);
284 
285         cpRequest.enable();
286         vm.resume();
287         getEventSet();
288         cpRequest.disable();
289 
290         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
291         debuggeeClass = event.referenceType();
292 
293         if (!debuggeeClass.name().equals(debuggeeName))
294             throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
295 
296         log2("      received: ClassPrepareEvent for debuggeeClass");
297 
298         String bPointMethod = "methodForCommunication";
299         String lineForComm = "lineForComm";
300 
301         ThreadReference mainThread = debuggee.threadByNameOrThrow("main");
302 
303         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
304                 debuggeeClass,
305                 bPointMethod, lineForComm, "zero");
306         bpRequest.enable();
307 
308         //------------------------------------------------------  testing section
309 
310         log1("     TESTING BEGINS");
311 
312         EventRequest eventRequest1 = null;
313         EventRequest eventRequest2 = null;
314         EventRequest eventRequest3 = null;
315 
316         ThreadReference thread1 = null;
317         String thread1Name = "thread1";
318 
319         String property1 = "BreakpointRequest1";
320         String property2 = "BreakpointRequest2";
321         String property3 = "BreakpointRequest3";
322 
323         String methodName = "method";
324         String bpLineName = "breakpointLine";
325 
326         ReferenceType testClassReference = null;
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                     (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             switch (i) {
346 
347                 case 0:
348                     testClassReference =
349                             (ReferenceType) vm.classesByName(testedClassName).get(0);
350 
351                     thread1 = (ThreadReference) debuggeeClass.getValue(
352                             debuggeeClass.fieldByName(thread1Name));
353 
354                     eventRequest1 = setting2BreakpointRequest(null,
355                             testClassReference, methodName, bpLineName,
356                             EventRequest.SUSPEND_NONE, property1);
357 
358                     try {
359                         log2("......eventRequest1.addThreadFilter(thread1);");
360                         log2("        no Exception expected");
361                         ((BreakpointRequest) eventRequest1).addThreadFilter(thread1);
362                         log2("        no Exception");
363                     } catch (Exception e) {
364                         log3("ERROR: Exception : " + e);
365                         testExitCode = FAILED;
366                     }
367 
368                     break;
369 
370                 case 1:
371                     eventRequest2 = setting2BreakpointRequest(null,
372                             testClassReference, methodName, bpLineName,
373                             EventRequest.SUSPEND_NONE, property2);
374 
375                     try {
376                         log2("......eventRequest2.addThreadFilter(thread1);");
377                         log2("        no Exception expected");
378                         ((BreakpointRequest) eventRequest2).addThreadFilter(thread1);
379                         log2("        no Exception");
380                     } catch (Exception e) {
381                         log3("ERROR: Exception : " + e);
382                         testExitCode = FAILED;
383                     }
384                     break;
385 
386                 case 2:
387                     eventRequest3 = setting2BreakpointRequest(null,
388                             testClassReference, methodName, bpLineName,
389                             EventRequest.SUSPEND_NONE, property3);
390 
391                     try {
392                         log2("......eventRequest3.addThreadFilter(thread1);");
393                         log2("        no Exception expected");
394                         ((BreakpointRequest) eventRequest3).addThreadFilter(thread1);
395                         log2("        no Exception");
396                     } catch (Exception e) {
397                         log3("ERROR: Exception : " + e);
398                         testExitCode = FAILED;
399                     }
400                     break;
401 
402                 default:
403                     throw new JDITestRuntimeException("** default case 2 **");
404             }
405             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
406         }
407         log1("    TESTING ENDS");
408         return;

411    /*
412     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
413     *                                             String, String, String)
414     *
415     * It sets up a breakpoint at given line number within a given method in a given class
416     * for a given thread.
417     *
418     * Return value: BreakpointRequest object  in case of success
419     *
420     * JDITestRuntimeException   in case of an Exception thrown within the method
421     */
422 
423     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
424                                                   ReferenceType testedClass,
425                                                   String methodName,
426                                                   String bpLine,
427                                                   String property)
428             throws JDITestRuntimeException {
429 
430         log2("......setting up a breakpoint:");
431         log2("       thread: " + thread + "; class: " + testedClass +
432                         "; method: " + methodName + "; line: " + bpLine);
433 
434         List              alllineLocations = null;
435         Location          lineLocation     = null;
436         BreakpointRequest breakpRequest    = null;
437 
438         try {
439             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
440 
441             alllineLocations = method.allLineLocations();
442 
443             int n =
444                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
445             if (n > alllineLocations.size()) {
446                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
447             } else {
448                 lineLocation = (Location) alllineLocations.get(n);
449                 try {
450                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
451                     breakpRequest.putProperty("number", property);
452                     breakpRequest.addThreadFilter(thread);
453                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
454                 } catch ( Exception e1 ) {
455                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
456                     breakpRequest    = null;
457                 }
458             }
459         } catch ( Exception e2 ) {
460             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
461             breakpRequest    = null;
462         }
463 
464         if (breakpRequest == null) {
465             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
466             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
467         }
468 
469         log2("      a breakpoint has been set up");
470         return breakpRequest;
471     }

474     private void getEventSet()
475                  throws JDITestRuntimeException {
476         try {
477 //            log2("       eventSet = eventQueue.remove(waitTime);");
478             eventSet = eventQueue.remove(waitTime);
479             if (eventSet == null) {
480                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
481             }
482 //            log2("       eventIterator = eventSet.eventIterator;");
483             eventIterator = eventSet.eventIterator();
484         } catch ( Exception e ) {
485             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
486         }
487     }

490     private void breakpointForCommunication()
491                  throws JDITestRuntimeException {
492 
493         log2("breakpointForCommunication");
494         getEventSet();
495 
496         if (eventIterator.nextEvent() instanceof BreakpointEvent)
497             return;
498 
499         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
500     }

504     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
505                                                           ReferenceType   testedClass,
506                                                           String          methodName,
507                                                           String          bpLine,
508                                                           int             suspendPolicy,
509                                                           String          property        )
510             throws JDITestRuntimeException {
511         try {
512             log2("......setting up a breakpoint:");
513             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
514 
515             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
516             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
517 
518             BreakpointRequest
519             bpr = eventRManager.createBreakpointRequest(loc);
520             bpr.putProperty("number", property);
521             if (thread != null)
522                 bpr.addThreadFilter(thread);
523             bpr.setSuspendPolicy(suspendPolicy);
524 
525             log2("      a breakpoint has been set up");
526             return bpr;
527         } catch ( Exception e ) {
528             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
529             log3("       BreakpointRequest HAS NOT BEEN SET UP");
530             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
531         }
