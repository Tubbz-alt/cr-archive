 88     private static Process startDebuggee(String[] jdiArgs, String transport, String suspend) {
 89         List<String> cmd = new ArrayList<>();
 90         Class<?> debuggeeClass = attach004t.class;
 91         cmd.add(JDKToolFinder.getJDKTool("java"));
 92         Collections.addAll(cmd, Utils.addTestJavaOpts(
 93                 "-cp",
 94                 Utils.TEST_CLASS_PATH,
 95                 "-Xdebug",
 96                 "-agentlib:jdwp=transport=" + transport + ",server=y,suspend=" + suspend,
 97                 "-Dmy.little.cookie=" + ProcessHandle.current().pid(),
 98                 debuggeeClass.getName()));
 99         Collections.addAll(cmd, jdiArgs);
100         cmd.add("-testWorkDir");
101         cmd.add(".");
102 
103         System.out.println("Starting debuggee [" + String.join(",", cmd) + "]");
104         Process p;
105         try {
106             p = new ProcessBuilder(cmd).redirectErrorStream(true).start();
107         } catch (IOException e) {
108             throw new Error("can't start debuggee");
109         }
110 
111         Thread t = new Thread(() -> dumpStream("debuggee>> ", p.getInputStream()));
112         t.setDaemon(true);
113         t.start();
114 
115         return p;
116     }
