 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.exception()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a boolean value, an argument of the method        <BR>
 46  *    EventRequestManager.createExceptionRequest(),                     <BR>
 47  * is equal to one returned by the method                               <BR>
 48  *    ExceptionRequest.exception()                                      <BR>
 49  * The cases to test include a ReferenceType                            <BR>
 50  * mirroring NullPointerException, and null.                            <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger                                                       <BR>
 69  *   - gets ReferenceType value mirroring NullPointerException, <BR>
 70  *   - sets up two ExceptionRequests with the ReferenceType and null    <BR>
 71  *     values of the exception arguments,                               <BR>
 72  *   - gets the values of the Request with the method                   <BR>
 73  *     StepRequest.exception(), and                                     <BR>
 74  *   - compares the values.                                             <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */
 81

154     private int runThis (String argv[], PrintStream out) {
155 
156         argsHandler     = new ArgumentHandler(argv);
157         logHandler      = new Log(out, argsHandler);
158         Binder binder   = new Binder(argsHandler, logHandler);
159 
160         waitTime        = argsHandler.getWaitTime() * 60000;
161 
162         try {
163             log2("launching a debuggee :");
164             log2("       " + debuggeeName);
165             if (argsHandler.verbose()) {
166                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
167             } else {
168                 debuggee = binder.bindToDebugee(debuggeeName);
169             }
170             if (debuggee == null) {
171                 log3("ERROR: no debuggee launched");
172                 return FAILED;
173             }
174             log2("debuggee launched");
175         } catch ( Exception e ) {
176             log3("ERROR: Exception : " + e);
177             log2("       test cancelled");
178             return FAILED;
179         }
180 
181         debuggee.redirectOutput(logHandler);
182 
183         vm = debuggee.VM();
184 
185         eventQueue = vm.eventQueue();
186         if (eventQueue == null) {
187             log3("ERROR: eventQueue == null : TEST ABORTED");
188             vm.exit(PASS_BASE);
189             return FAILED;
190         }
191 
192         log2("invocation of the method runTest()");
193         switch (runTest()) {
194 
195             case 0 :  log2("test phase has finished normally");
196                       log2("   waiting for the debuggee to finish ...");
197                       debuggee.waitFor();
198 
199                       log2("......getting the debuggee's exit status");
200                       int status = debuggee.getStatus();
201                       if (status != PASS_BASE) {
202                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
203                               status + " != PASS_BASE");
204                           testExitCode = FAILED;
205                       } else {
206                           log2("......debuggee returned expected exit status: " +
207                               status + " == PASS_BASE");
208                       }
209                       break;
210 
211             default : log3("ERROR: runTest() returned unexpected value");
212 
213             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
214                       log2("......forcing: vm.exit();");
215                       testExitCode = FAILED;
216                       try {
217                           vm.exit(PASS_BASE);
218                       } catch ( Exception e ) {
219                           log3("ERROR: Exception : " + e);
220                       }
221                       break;
222 
223             case 2 :  log3("test cancelled due to VMDisconnectedException");
224                       log2("......trying: vm.process().destroy();");
225                       testExitCode = FAILED;
226                       try {
227                           Process vmProcess = vm.process();
228                           if (vmProcess != null) {
229                               vmProcess.destroy();
230                           }
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235             }
236 
237         return testExitCode;
238     }

241    /*
242     * Return value: 0 - normal end of the test
243     *               1 - ubnormal end of the test
244     *               2 - VMDisconnectedException while test phase
245     */

247     private int runTest() {
248 
249         try {
250             testRun();
251 
252             log2("waiting for VMDeathEvent");
253             getEventSet();
254             if (eventIterator.nextEvent() instanceof VMDeathEvent)
255                 return 0;
256 
257             log3("ERROR: last event is not the VMDeathEvent");
258             return 1;
259         } catch ( VMDisconnectedException e ) {
260             log3("ERROR: VMDisconnectedException : " + e);
261             return 2;
262         } catch ( Exception e ) {
263             log3("ERROR: Exception : " + e);
264             return 1;
265         }
266 
267     }

269     private void testRun()
270                  throws JDITestRuntimeException, Exception {
271 
272         eventRManager = vm.eventRequestManager();
273 
274         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
275         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
276         cpRequest.addClassFilter(debuggeeName);
277 
278         cpRequest.enable();
279         vm.resume();
280         getEventSet();
281         cpRequest.disable();
282 
283         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
284         debuggeeClass = event.referenceType();
285 
286         if (!debuggeeClass.name().equals(debuggeeName))
287            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
288 
289         log2("      received: ClassPrepareEvent for debuggeeClass");
290 
291         String bPointMethod = "methodForCommunication";
292         String lineForComm  = "lineForComm";
293 
294         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
295 
296         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
297                                              debuggeeClass,
298                                             bPointMethod, lineForComm, "zero");
299         bpRequest.enable();
300         vm.resume();
301 
302     //------------------------------------------------------  testing section
303 
304         log1("     TESTING BEGINS");
305 
306         EventRequest  eventRequest1 = null;
307         EventRequest  eventRequest2 = null;
308 
309         String fieldName = "obj";
310         Value value = null;
311 
312         ReferenceType refObject = null;
313         ReferenceType exception = null;
314 
315         for (int i = 0; ; i++) {
316 
317             breakpointForCommunication();
318 
319             int instruction = ((IntegerValue)
320                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
321 
322             if (instruction == 0) {
323                 vm.resume();
324                 break;
325             }
326 
327             log1(":::::: case: # " + i);
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
330 
331             switch (i) {
332 
333               case 0:
334                      log2("......vm.resume();");
335                      vm.resume();
336 
337                      log2("......getting: refObject = debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName));");
338                      value = debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName));
339                      log2("      getting: refObject = (ReferenceType) value.type();");
340                      refObject = (ReferenceType) value.type();
341 
342                      log2("......setting up: eventRequest1 = setting25ExceptionRequest(refObject);");
343                      eventRequest1 = setting25ExceptionRequest(refObject);
344 
345                      log2("......getting: exception = ((ExceptionRequest) eventRequest1).exception();");
346                      exception = ((ExceptionRequest) eventRequest1).exception();
347 
348                      log2("      comparing ReferenceType objects");
349                      if ( !exception.equals(refObject) ) {
350                          testExitCode = FAILED;
351                          log3("ERROR: objects are not equal : " + exception);
352                      }
353 
354                      log2("......setting up: eventRequest2 = setting25ExceptionRequest(null);");
355                      eventRequest2 = setting25ExceptionRequest(null);
356 
357                      log2("......getting: exception = ((ExceptionRequest) eventRequest2).exception();");
358                      exception = ((ExceptionRequest) eventRequest2).exception();
359 
360                      log2("      comparing to null");
361                      if ( exception != null ) {
362                          testExitCode = FAILED;
363                          log3("ERROR: exception is not the null : " + exception);
364                      }
365 
366                       break;
367 
368               default:
369                       throw new JDITestRuntimeException("** default case 2 **");
370             }
371 
372 
373             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
374         }
375         log1("    TESTING ENDS");
376         return;
377     }

379    /*
380     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
381     *                                             String, String, String)
382     *
383     * It sets up a breakpoint at given line number within a given method in a given class
384     * for a given thread.
385     *
386     * Return value: BreakpointRequest object  in case of success
387     *
388     * JDITestRuntimeException   in case of an Exception thrown within the method
389     */
390 
391     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
392                                                   ReferenceType testedClass,
393                                                   String methodName,
394                                                   String bpLine,
395                                                   String property)
396             throws JDITestRuntimeException {
397 
398         log2("......setting up a breakpoint:");
399         log2("       thread: " + thread + "; class: " + testedClass +
400                         "; method: " + methodName + "; line: " + bpLine);
401 
402         List              alllineLocations = null;
403         Location          lineLocation     = null;
404         BreakpointRequest breakpRequest    = null;
405 
406         try {
407             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
408 
409             alllineLocations = method.allLineLocations();
410 
411             int n =
412                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
413             if (n > alllineLocations.size()) {
414                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
415             } else {
416                 lineLocation = (Location) alllineLocations.get(n);
417                 try {
418                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
419                     breakpRequest.putProperty("number", property);
420                     breakpRequest.addThreadFilter(thread);
421                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
422                 } catch ( Exception e1 ) {
423                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
424                     breakpRequest    = null;
425                 }
426             }
427         } catch ( Exception e2 ) {
428             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
429             breakpRequest    = null;
430         }
431 
432         if (breakpRequest == null) {
433             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
434             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
435         }
436 
437         log2("      a breakpoint has been set up");
438         return breakpRequest;
439     }

442     private void getEventSet()
443                  throws JDITestRuntimeException {
444         try {
445 //            log2("       eventSet = eventQueue.remove(waitTime);");
446             eventSet = eventQueue.remove(waitTime);
447             if (eventSet == null) {
448                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
449             }
450 //            log2("       eventIterator = eventSet.eventIterator;");
451             eventIterator = eventSet.eventIterator();
452         } catch ( Exception e ) {
453             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
454         }
455     }

458     private void breakpointForCommunication()
459                  throws JDITestRuntimeException {
460 
461         log2("breakpointForCommunication");
462         getEventSet();
463 
464         if (eventIterator.nextEvent() instanceof BreakpointEvent)
465             return;
466 
467         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
468     }
