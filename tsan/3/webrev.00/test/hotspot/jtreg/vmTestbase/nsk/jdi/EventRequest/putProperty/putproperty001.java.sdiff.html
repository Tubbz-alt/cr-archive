454     private ThreadReference threadByName(String name)
455                  throws JDITestRuntimeException {
456 
457         List         all = vm.allThreads();
458         ListIterator li  = all.listIterator();
459 
460         for (; li.hasNext(); ) {
461             ThreadReference thread = (ThreadReference) li.next();
462             if (thread.name().equals(name))
463                 return thread;
464         }
465         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
466     }
