 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for StepEvent sets.                   <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a StepRequest, resumes                        <BR>
 74  *   the debuggee, and waits for the StepEvent.                         <BR>
 75  * - The debuggee invokes the special method                            <BR>
 76  *   to be resulting in the event.                                      <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

169     private int runThis (String argv[], PrintStream out) {
170 
171         argsHandler     = new ArgumentHandler(argv);
172         logHandler      = new Log(out, argsHandler);
173         Binder binder   = new Binder(argsHandler, logHandler);
174 
175         waitTime        = argsHandler.getWaitTime() * 60000;
176 
177         try {
178             log2("launching a debuggee :");
179             log2("       " + debuggeeName);
180             if (argsHandler.verbose()) {
181                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
182             } else {
183                 debuggee = binder.bindToDebugee(debuggeeName);
184             }
185             if (debuggee == null) {
186                 log3("ERROR: no debuggee launched");
187                 return FAILED;
188             }
189             log2("debuggee launched");
190         } catch ( Exception e ) {
191             log3("ERROR: Exception : " + e);
192             log2("       test cancelled");
193             return FAILED;
194         }
195 
196         debuggee.redirectOutput(logHandler);
197 
198         vm = debuggee.VM();
199 
200         eventQueue = vm.eventQueue();
201         if (eventQueue == null) {
202             log3("ERROR: eventQueue == null : TEST ABORTED");
203             vm.exit(PASS_BASE);
204             return FAILED;
205         }
206 
207         log2("invocation of the method runTest()");
208         switch (runTest()) {
209 
210             case 0 :  log2("test phase has finished normally");
211                       log2("   waiting for the debuggee to finish ...");
212                       debuggee.waitFor();
213 
214                       log2("......getting the debuggee's exit status");
215                       int status = debuggee.getStatus();
216                       if (status != PASS_BASE) {
217                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
218                               status + " != PASS_BASE");
219                           testExitCode = FAILED;
220                       } else {
221                           log2("......debuggee returned expected exit status: " +
222                               status + " == PASS_BASE");
223                       }
224                       break;
225 
226             default : log3("ERROR: runTest() returned unexpected value");
227 
228             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
229                       log2("......forcing: vm.exit();");
230                       testExitCode = FAILED;
231                       try {
232                           vm.exit(PASS_BASE);
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237 
238             case 2 :  log3("test cancelled due to VMDisconnectedException");
239                       log2("......trying: vm.process().destroy();");
240                       testExitCode = FAILED;
241                       try {
242                           Process vmProcess = vm.process();
243                           if (vmProcess != null) {
244                               vmProcess.destroy();
245                           }
246                       } catch ( Exception e ) {
247                           log3("ERROR: Exception : e");
248                       }
249                       break;
250             }
251 
252         return testExitCode;
253     }

262     private int runTest() {
263 
264         try {
265             testRun();
266 
267             log2("waiting for VMDeathEvent");
268             getEventSet();
269             if (eventIterator.nextEvent() instanceof VMDeathEvent)
270                 return 0;
271 
272             log3("ERROR: last event is not the VMDeathEvent");
273             return 1;
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         eventRManager = vm.eventRequestManager();
288 
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292 
293         cpRequest.enable();
294         vm.resume();
295         getEventSet();
296         cpRequest.disable();
297 
298         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
299         debuggeeClass = event.referenceType();
300 
301         if (!debuggeeClass.name().equals(debuggeeName))
302            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
303 
304         log2("      received: ClassPrepareEvent for debuggeeClass");
305 
306         String bPointMethod = "methodForCommunication";
307         String lineForComm  = "lineForComm";
308         BreakpointRequest bpRequest;
309         ThreadReference   mainThread = threadByName("main");
310         bpRequest = settingBreakpoint(mainThread,
311                                       debuggeeClass,
312                                       bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315         vm.resume();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         EventRequest eventRequest1 = null;
322         EventRequest eventRequest2 = null;
323         EventRequest eventRequest3 = null;
324 
325         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
327         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
328 
329         ReferenceType testClassReference = null;
330 
331         for (int i = 0; ; i++) {
332 
333             breakpointForCommunication();
334 
335             int instruction = ((IntegerValue)
336                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
337 
338             if (instruction == 0) {
339                 vm.resume();
340                 break;
341             }
342 
343             log1(":::::: case: # " + i);
344 
345             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
346 
347             switch (i) {
348 
349               case 0:
350                       eventRequest1 = settingStepRequest (mainThread,
351                                              SUSPEND_NONE, "StepRequest1");
352                       eventRequest1.enable();
353                       break;
354 
355               case 1:
356                       eventRequest2 = settingStepRequest (mainThread,
357                                              SUSPEND_THREAD, "StepRequest2");
358                       eventRequest2.enable();
359                       break;
360 
361               case 2:
362                       eventRequest3 = settingStepRequest (mainThread,
363 
364                                              SUSPEND_ALL, "StepRequest3");
365                       eventRequest3.enable();
366                       break;
367 
368 
369               default:
370                       throw new JDITestRuntimeException("** default case 2 **");
371             }
372 
373             log2("......waiting for new StepEvent : " + i);
374             mainThread.resume();
375             getEventSet();
376 
377             Event newEvent = eventIterator.nextEvent();
378             if ( !(newEvent instanceof StepEvent)) {
379                  log3("ERROR: new event is not StepEvent");
380                  testExitCode = FAILED;
381             } else {
382 
383                 String property = (String) newEvent.request().getProperty("number");
384                 log2("       got new StepEvent with propety 'number' == " + property);
385 
386                 log2("......checking up on EventSet.resume()");
387                 log2("......--> vm.suspend();");
388                 vm.suspend();
389 
390                 log2("        getting : Map<String, Integer> suspendsCounts1");
391 
392                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
393                 for (ThreadReference threadReference : vm.allThreads()) {
394                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
395                 }
396 
397                 log2("        eventSet.resume;");
398                 eventSet.resume();
399 
400                 log2("        getting : Map<String, Integer> suspendsCounts2");
401 
402                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
403                 for (ThreadReference threadReference : vm.allThreads()) {
404                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
405                 }
406 
407                 log2(suspendsCounts1.toString());
408                 log2(suspendsCounts2.toString());
409 
410                 log2("        getting : int policy = eventSet.suspendPolicy();");
411                 int policy = eventSet.suspendPolicy();
412 
413                 switch (policy) {
414 
415                     case SUSPEND_NONE:
416 
417                         log2("        case SUSPEND_NONE");
418                         for (String threadName : suspendsCounts1.keySet()) {
419                             log2("        checking " + threadName);
420                             if (!suspendsCounts2.containsKey(threadName)) {
421                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
422                                 testExitCode = FAILED;
423                                 break;
424                             }
425                             int count1 = suspendsCounts1.get(threadName);
426                             int count2 = suspendsCounts2.get(threadName);
427                             if (count1 != count2) {
428                                 log3("ERROR: suspendCounts don't match for : " + threadName);
429                                 log3("before resuming : " + count1);
430                                 log3("after  resuming : " + count2);
431                                 testExitCode = FAILED;
432                                 break;
433                             }
434                         }
435                         eventRequest1.disable();
436                         break;
437 
438                     case SUSPEND_THREAD:
439 
440                         log2("        case SUSPEND_THREAD");
441                         for (String threadName : suspendsCounts1.keySet()) {
442                             log2("checking " + threadName);
443                             if (!suspendsCounts2.containsKey(threadName)) {
444                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
445                                 testExitCode = FAILED;
446                                 break;
447                             }
448                             int count1 = suspendsCounts1.get(threadName);
449                             int count2 = suspendsCounts2.get(threadName);
450                             String eventThreadName = ((StepEvent) newEvent).thread().name();
451                             int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
452                             if (count1 != expectedValue) {
453                                 log3("ERROR: suspendCounts don't match for : " + threadName);
454                                 log3("before resuming : " + count1);
455                                 log3("after  resuming : " + count2);
456                                 testExitCode = FAILED;
457                                 break;
458                             }
459                         }
460                         eventRequest2.disable();
461                         break;
462 
463                     case SUSPEND_ALL:
464 
465                         log2("        case SUSPEND_ALL");
466                         for (String threadName : suspendsCounts1.keySet()) {
467                             log2("checking " + threadName);
468                             if (!event.request().equals(eventRequest3))
469                                 break;
470                             if (!suspendsCounts2.containsKey(threadName)) {
471                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
472                                 testExitCode = FAILED;
473                                 break;
474                             }
475                             int count1 = suspendsCounts1.get(threadName);
476                             int count2 = suspendsCounts2.get(threadName);
477                             if (count1 != count2 + 1) {
478                                 log3("ERROR: suspendCounts don't match for : " + threadName);
479                                 log3("before resuming : " + count1);
480                                 log3("after  resuming : " + count2);
481                                 testExitCode = FAILED;
482                                 break;
483                             }
484                         }
485                         eventRequest3.disable();
486                         break;
487 
488                     default:
489                         throw new JDITestRuntimeException("** default case 1 **");
490                 }
491                 informDebuggeeTestCase(i);
492             }
493 
494             log2("......--> vm.resume()");
495             vm.resume();
496             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
497         }
498         log1("    TESTING ENDS");
499         return;
500     }

502     private ThreadReference threadByName(String name)
503                  throws JDITestRuntimeException {
504 
505         List         all = vm.allThreads();
506         ListIterator li  = all.listIterator();
507 
508         for (; li.hasNext(); ) {
509             ThreadReference thread = (ThreadReference) li.next();
510             if (thread.name().equals(name))
511                 return thread;
512         }
513         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
514     }

516    /*
517     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
518     *                                             String, String, String)
519     *
520     * It sets up a breakpoint at given line number within a given method in a given class
521     * for a given thread.
522     *
523     * Return value: BreakpointRequest object  in case of success
524     *
525     * JDITestRuntimeException   in case of an Exception thrown within the method
526     */
527 
528     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
529                                                   ReferenceType testedClass,
530                                                   String methodName,
531                                                   String bpLine,
532                                                   String property)
533             throws JDITestRuntimeException {
534 
535         log2("......setting up a breakpoint:");
536         log2("       thread: " + thread + "; class: " + testedClass +
537                         "; method: " + methodName + "; line: " + bpLine);
538 
539         List              alllineLocations = null;
540         Location          lineLocation     = null;
541         BreakpointRequest breakpRequest    = null;
542 
543         try {
544             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
545 
546             alllineLocations = method.allLineLocations();
547 
548             int n =
549                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
550             if (n > alllineLocations.size()) {
551                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
552             } else {
553                 lineLocation = (Location) alllineLocations.get(n);
554                 try {
555                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
556                     breakpRequest.putProperty("number", property);
557                     breakpRequest.addThreadFilter(thread);
558                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
559                 } catch ( Exception e1 ) {
560                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
561                     breakpRequest    = null;
562                 }
563             }
564         } catch ( Exception e2 ) {
565             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
566             breakpRequest    = null;
567         }
568 
569         if (breakpRequest == null) {
570             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
571             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
572         }
573 
574         log2("      a breakpoint has been set up");
575         return breakpRequest;
576     }

579     private void getEventSet()
580                  throws JDITestRuntimeException {
581         try {
582 //            log2("       eventSet = eventQueue.remove(waitTime);");
583             eventSet = eventQueue.remove(waitTime);
584             if (eventSet == null) {
585                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
586             }
587 //            log2("       eventIterator = eventSet.eventIterator;");
588             eventIterator = eventSet.eventIterator();
589         } catch ( Exception e ) {
590             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
591         }
592     }

595     private void breakpointForCommunication()
596                  throws JDITestRuntimeException {
597 
598         log2("breakpointForCommunication");
599         getEventSet();
600 
601         if (eventIterator.nextEvent() instanceof BreakpointEvent)
602             return;
603 
604         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
605     }

609     private StepRequest settingStepRequest ( ThreadReference thread,
610                                              int             suspendPolicy,
611                                              String          property        )
612             throws JDITestRuntimeException {
613         try {
614             log2("......setting up StepRequest:");
615             log2("       thread: " + thread + "; property: " + property);
616 
617             StepRequest
618             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_OVER);
619             str.putProperty("number", property);
620             str.addCountFilter(1);
621             str.setSuspendPolicy(suspendPolicy);
622 
623             log2("      StepRequest has been set up");
624             return str;
625         } catch ( Exception e ) {
626             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
627             log3("       StepRequest HAS NOT BEEN SET UP");
628             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
629         }
630     }
631     /**
632      * Inform debuggee which thread test the debugger has completed.
633      * Used for synchronization, so the debuggee does not move too quickly.
634      * @param testCase index of just completed test
635      */
636     void informDebuggeeTestCase(int testCase) {
637         try {
638             ((ClassType)debuggeeClass)
639                 .setValue(debuggeeClass.fieldByName("testCase"),
640                           vm.mirrorOf(testCase));
641         } catch (InvalidTypeException ite) {
642             throw new Failure("** FAILURE setting testCase  **");
643         } catch (ClassNotLoadedException cnle) {
644             throw new Failure("** FAILURE notifying debuggee  **");
645         } catch (VMDisconnectedException e) {
646             throw new Failure("** FAILURE debuggee connection **");
647         }
648     }
