 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.putProperty()</code>          <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on following assertions:                  <BR>
 46  * - a value of property is equal to one assigned by            <BR>
 47  *   last method putProperty(key, value);                       <BR>
 48  * - after calling putProperty(key, null)                       <BR>
 49  *   the value of the property 'key' is also null.              <BR>
 50  * The cases to check include Requests of all sub-types.        <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new check case (as if) and invokes           <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the check case required.                     <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : " + e);
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : " + e);
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if (eventIterator.nextEvent() instanceof VMDeathEvent)
259                 return 0;
260 
261             log3("ERROR: last event is not the VMDeathEvent");
262             return 1;
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         eventRManager = vm.eventRequestManager();
277 
278         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
279         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
280         cpRequest.addClassFilter(debuggeeName);
281 
282         cpRequest.enable();
283         vm.resume();
284         getEventSet();
285         cpRequest.disable();
286 
287         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
288         debuggeeClass = event.referenceType();
289 
290         if (!debuggeeClass.name().equals(debuggeeName))
291            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
292 
293         log2("      received: ClassPrepareEvent for debuggeeClass");
294 
295         String bPointMethod = "methodForCommunication";
296         String lineForComm  = "lineForComm";
297 
298         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
299 
300         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
301                                              debuggeeClass,
302                                             bPointMethod, lineForComm, "zero");
303         bpRequest.enable();
304 
305     //------------------------------------------------------  testing section
306 
307 
308         EventRequest  eventRequest1 = null;
309 
310         String        fieldName1    = "var101";
311         String        fieldName2    = "excObj";
312 
313         ThreadReference thread1     = null;
314         String          threadName1 = "thread1";
315 
316         ReferenceType testClassReference = null;
317 
318         String propertyName   = "name";
319         String propertyValue1 = "value1";
320         String propertyValue2 = "value2";
321 
322 
323         log1("     TESTING BEGINS");
324 
325         for (int i = 0; ; i++) {
326 
327             vm.resume();
328             breakpointForCommunication();
329 
330             int instruction = ((IntegerValue)
331                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
332 
333             if (instruction == 0) {
334                 vm.resume();
335                 break;
336             }
337 
338             log1(":::::: case: # " + i);
339 
340             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
341 
342 
343             switch (i) {
344 
345               case 0:
346                      thread1 = debuggee.threadByNameOrThrow(threadName1);
347 
348                      log2("......setting up StepRequest");
349                      eventRequest1 = eventRManager.createStepRequest
350                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
351                      break;
352 
353               case 1:
354                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
355 
356                      log2("......setting up AccessWatchpointRequest");
357                      eventRequest1 = eventRManager.createAccessWatchpointRequest
358                                     (testClassReference.fieldByName(fieldName1));
359                      break;
360 
361               case 2:
362                      log2(".....setting up ModificationWatchpointRequest");
363                      eventRequest1 = eventRManager.createModificationWatchpointRequest
364                                     (testClassReference.fieldByName(fieldName1));
365                      break;
366 
367               case 3:
368                      log2(".....setting up ClassPrepareRequest");
369                      eventRequest1 = eventRManager.createClassPrepareRequest();
370                      break;
371 
372               case 4:
373                      log2(".....setting up ClassUnloadRequest");
374                      eventRequest1 = eventRManager.createClassUnloadRequest();
375                      break;
376 
377               case 5:
378                      log2(".....setting up MethodEntryRequest");
379                      eventRequest1 = eventRManager.createMethodEntryRequest();
380                      break;
381 
382               case 6:
383                      log2(".....setting up MethodExitRequest");
384                      eventRequest1 = eventRManager.createMethodExitRequest();
385                      break;
386 
387               case 7:
388                      log2(".....setting up ThreadDeathRequest");
389                      eventRequest1 = eventRManager.createThreadDeathRequest();
390                      break;
391 
392               case 8:
393                      log2(".....setting up ThreadStartRequest");
394                      eventRequest1 = eventRManager.createThreadStartRequest();
395                      break;
396 
397               case 9:
398                      log2(".....setting up VMDeathRequest");
399                      eventRequest1 = eventRManager.createVMDeathRequest();
400                      break;
401 
402               case 10:
403                      log2(".....setting up ExceptionRequest");
404                      eventRequest1 = eventRManager.createExceptionRequest(
405                                      (ReferenceType)
406                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
407                                      true, true );
408                      break;
409 
410               case 11:
411                      log2(".....setting up BreakpointRequest");
412                      eventRequest1 = eventRManager.createBreakpointRequest(location);
413                      break;
414 
415 
416               default:
417                       throw new JDITestRuntimeException("** default case 2 **");
418             }
419 
420             log2("......eventRequest1.putProperty(propertyName, propertyValue1);");
421             eventRequest1.putProperty(propertyName, propertyValue1);
422             log2("      eventRequest1.putProperty(propertyName, propertyValue2);");
423             eventRequest1.putProperty(propertyName, propertyValue2);
424 
425             log2("      checking up on eventRequest1");
426             if ( !eventRequest1.getProperty(propertyName).equals(propertyValue2) ) {
427                 testExitCode = FAILED;
428                 log3("ERROR: property is not expected");
429             }
430 
431             log2("......eventRequest1.putProperty(propertyName, null);");
432             eventRequest1.putProperty(propertyName, null);
433 
434             log2("      checking up on eventRequest1");
435             if ( eventRequest1.getProperty(propertyName)!= null ) {
436                 testExitCode = FAILED;
437                 log3("ERROR: property is not the null");
438             }
439 
440             eventRequest1.setEnabled(false);
441 
442             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
443         }
444         log1("    TESTING ENDS");
445         return;
446     }

448    /*
449     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
450     *                                             String, String, String)
451     *
452     * It sets up a breakpoint at given line number within a given method in a given class
453     * for a given thread.
454     *
455     * Return value: BreakpointRequest object  in case of success
456     *
457     * JDITestRuntimeException   in case of an Exception thrown within the method
458     */
459 
460     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
461                                                   ReferenceType testedClass,
462                                                   String methodName,
463                                                   String bpLine,
464                                                   String property)
465             throws JDITestRuntimeException {
466 
467         log2("......setting up a breakpoint:");
468         log2("       thread: " + thread + "; class: " + testedClass +
469                         "; method: " + methodName + "; line: " + bpLine);
470 
471         List              alllineLocations = null;
472         Location          lineLocation     = null;
473         BreakpointRequest breakpRequest    = null;
474 
475         try {
476             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
477 
478             alllineLocations = method.allLineLocations();
479 
480             int n =
481                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
482             if (n > alllineLocations.size()) {
483                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
484             } else {
485                 lineLocation = (Location) alllineLocations.get(n);
486 
487 // this is only for this test to get Location object
488 location = lineLocation;
489 
490                 try {
491                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
492                     breakpRequest.putProperty("number", property);
493                     breakpRequest.addThreadFilter(thread);
494                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
495                 } catch ( Exception e1 ) {
496                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
497                     breakpRequest    = null;
498                 }
499             }
500         } catch ( Exception e2 ) {
501             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
502             breakpRequest    = null;
503         }
504 
505         if (breakpRequest == null) {
506             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
507             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
508         }
509 
510         log2("      a breakpoint has been set up");
511         return breakpRequest;
512     }

515     private void getEventSet()
516                  throws JDITestRuntimeException {
517         try {
518 //            log2("       eventSet = eventQueue.remove(waitTime);");
519             eventSet = eventQueue.remove(waitTime);
520             if (eventSet == null) {
521                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
522             }
523 //            log2("       eventIterator = eventSet.eventIterator;");
524             eventIterator = eventSet.eventIterator();
525         } catch ( Exception e ) {
526             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
527         }
528     }

531     private void breakpointForCommunication()
532                  throws JDITestRuntimeException {
533 
534         log2("breakpointForCommunication");
535         getEventSet();
536 
537         if (eventIterator.nextEvent() instanceof BreakpointEvent)
538             return;
539 
540         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
541     }
