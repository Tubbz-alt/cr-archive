 23 #include <jni.h>
 24 #include <jvmti.h>
 25 #include "agent_common.h"
 26 #include <string.h>
 27 #include "jvmti_tools.h"
 28 
 29 extern "C" {
 30 #define FILE_NAME "nsk/jvmti/scenarios/hotswap/HS301/hs301t003/MyClass"
 31 #define CLASS_NAME "Lnsk/jvmti/scenarios/hotswap/HS301/hs301t003/MyClass;"
 32 
 33 void JNICALL callbackClassPrepare(jvmtiEnv *jvmti_env,
 34                                JNIEnv* jni_env,
 35                                jthread thread,
 36                                jclass klass) {
 37   char * className;
 38   char * generic;
 39   if (!NSK_JVMTI_VERIFY (jvmti_env->GetClassSignature(klass, &className, &generic))) {
 40     nsk_printf(" Agent:: Error while getting ClassFileName Signature ");
 41   } else {
 42       if (strcmp(className, CLASS_NAME) == 0) {
 43           int redefineNumber =0;
 44           char fileName[512];
 45           nsk_jvmti_getFileName(redefineNumber, FILE_NAME,
 46                   fileName, sizeof(fileName)/sizeof(char));
 47           nsk_jvmti_disableNotification(jvmti_env,
 48                   JVMTI_EVENT_CLASS_LOAD, NULL);
 49           if (nsk_jvmti_redefineClass(jvmti_env, klass, fileName)) {
 50               nsk_printf("\n Redefine successful.\n");
 51           } else {
 52               nsk_printf("\n Redefine failed.\n");
 53           }
 54       }
 55   }
 56 }
 57 
 58 #ifdef STATIC_BUILD
 59 JNIEXPORT jint JNICALL Agent_OnLoad_hs301t003(JavaVM *jvm, char *options, void *reserved) {
 60     return Agent_Initialize(jvm, options, reserved);
 61 }
 62 JNIEXPORT jint JNICALL Agent_OnAttach_hs301t003(JavaVM *jvm, char *options, void *reserved) {
 63     return Agent_Initialize(jvm, options, reserved);
 64 }
 65 JNIEXPORT jint JNI_OnLoad_hs301t003(JavaVM *jvm, char *options, void *reserved) {
 66     return JNI_VERSION_1_8;
 67 }
 68 #endif
 69 jint  Agent_Initialize(JavaVM *vm, char *options, void *reserved) {
 70     jvmtiEnv * jvmti;
 71     nsk_printf("Agent:: Agent_OnLoad.\n");
 72     if (!NSK_VERIFY (JNI_OK == vm->GetEnv((void **)&jvmti, JVMTI_VERSION_1_1))) {
 73         nsk_printf("Agent:: Could not load JVMTI interface.\n");
 74         return JNI_ERR;
 75     } else {
 76         jvmtiCapabilities caps;
 77         jvmtiEventCallbacks eventCallbacks;
 78         if (!nsk_jvmti_parseOptions(options)) {
 79             nsk_printf("# error agent Failed to parse options.\n");
 80             return JNI_ERR;
 81         }
 82         memset(&caps, 0, sizeof(caps));
 83         caps.can_redefine_classes = 1;
 84         caps.can_generate_all_class_hook_events=1;
 85         if (!NSK_JVMTI_VERIFY (jvmti->AddCapabilities(&caps)))  {
 86             nsk_printf(" Agent:: Error occured while adding capabilities.\n");
 87             return JNI_ERR;
 88         }
 89         memset(&eventCallbacks, 0, sizeof(eventCallbacks));
 90         eventCallbacks.ClassPrepare = &callbackClassPrepare;
 91         if (!NSK_JVMTI_VERIFY(jvmti->SetEventCallbacks(&eventCallbacks, sizeof(eventCallbacks)))) {
 92             nsk_printf(" Agent:: Error occured while setting event call back.\n");
 93             return JNI_ERR;
 94         }
 95         if (nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_CLASS_PREPARE, NULL)) {
 96             nsk_printf(" Agent:: Enabled notification.\n");
 97         } else {
 98             nsk_printf(" Agent:: Failed to enable notification.\n");
 99         }
100     }
101     return JNI_OK;
102 }
103 }
