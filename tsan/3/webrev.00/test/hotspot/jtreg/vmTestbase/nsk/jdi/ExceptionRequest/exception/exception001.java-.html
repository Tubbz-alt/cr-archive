 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.exception()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a boolean value, an argument of the method        <BR>
 46  *    EventRequestManager.createExceptionRequest(),                     <BR>
 47  * is equal to one returned by the method                               <BR>
 48  *    ExceptionRequest.exception()                                      <BR>
 49  * The cases to test include a ReferenceType                            <BR>
 50  * mirroring NullPointerException, and null.                            <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger                                                       <BR>
 69  *   - gets ReferenceType value mirroring NullPointerException, <BR>
 70  *   - sets up two ExceptionRequests with the ReferenceType and null    <BR>
 71  *     values of the exception arguments,                               <BR>
 72  *   - gets the values of the Request with the method                   <BR>
 73  *     StepRequest.exception(), and                                     <BR>
 74  *   - compares the values.                                             <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase
251     */

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = threadByName("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306         vm.resume();
307 
308     //------------------------------------------------------  testing section
309 
310         log1("     TESTING BEGINS");
311 
312         EventRequest  eventRequest1 = null;
313         EventRequest  eventRequest2 = null;
314 
315         String fieldName = "obj";
316         Value value = null;
317 
318         ReferenceType refObject = null;
319         ReferenceType exception = null;
320 
321         for (int i = 0; ; i++) {
322 
323             breakpointForCommunication();
324 
325             int instruction = ((IntegerValue)
326                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
327 
328             if (instruction == 0) {
329                 vm.resume();
330                 break;
331             }
332 
333             log1(":::::: case: # " + i);
334 
335             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
336 
337             switch (i) {
338 
339               case 0:
340                      log2("......vm.resume();");
341                      vm.resume();
342 
343                      log2("......getting: refObject = debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName));");
344                      value = debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName));
345                      log2("      getting: refObject = (ReferenceType) value.type();");
346                      refObject = (ReferenceType) value.type();
347 
348                      log2("......setting up: eventRequest1 = setting25ExceptionRequest(refObject);");
349                      eventRequest1 = setting25ExceptionRequest(refObject);
350 
351                      log2("......getting: exception = ((ExceptionRequest) eventRequest1).exception();");
352                      exception = ((ExceptionRequest) eventRequest1).exception();
353 
354                      log2("      comparing ReferenceType objects");
355                      if ( !exception.equals(refObject) ) {
356                          testExitCode = FAILED;
357                          log3("ERROR: objects are not equal : " + exception);
358                      }
359 
360                      log2("......setting up: eventRequest2 = setting25ExceptionRequest(null);");
361                      eventRequest2 = setting25ExceptionRequest(null);
362 
363                      log2("......getting: exception = ((ExceptionRequest) eventRequest2).exception();");
364                      exception = ((ExceptionRequest) eventRequest2).exception();
365 
366                      log2("      comparing to null");
367                      if ( exception != null ) {
368                          testExitCode = FAILED;
369                          log3("ERROR: exception is not the null : " + exception);
370                      }
371 
372                       break;
373 
374               default:
375                       throw new JDITestRuntimeException("** default case 2 **");
376             }
377 
378 
379             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
380         }
381         log1("    TESTING ENDS");
382         return;
383     }

385     private ThreadReference threadByName(String name)
386                  throws JDITestRuntimeException {
387 
388         List         all = vm.allThreads();
389         ListIterator li  = all.listIterator();
390 
391         for (; li.hasNext(); ) {
392             ThreadReference thread = (ThreadReference) li.next();
393             if (thread.name().equals(name))
394                 return thread;
395         }
396         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
397     }

399    /*
400     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
401     *                                             String, String, String)
402     *
403     * It sets up a breakpoint at given line number within a given method in a given class
404     * for a given thread.
405     *
406     * Return value: BreakpointRequest object  in case of success
407     *
408     * JDITestRuntimeException   in case of an Exception thrown within the method
409     */
410 
411     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
412                                                   ReferenceType testedClass,
413                                                   String methodName,
414                                                   String bpLine,
415                                                   String property)
416             throws JDITestRuntimeException {
417 
418         log2("......setting up a breakpoint:");
419         log2("       thread: " + thread + "; class: " + testedClass +
420                         "; method: " + methodName + "; line: " + bpLine);
421 
422         List              alllineLocations = null;
423         Location          lineLocation     = null;
424         BreakpointRequest breakpRequest    = null;
425 
426         try {
427             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
428 
429             alllineLocations = method.allLineLocations();
430 
431             int n =
432                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
433             if (n > alllineLocations.size()) {
434                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
435             } else {
436                 lineLocation = (Location) alllineLocations.get(n);
437                 try {
438                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
439                     breakpRequest.putProperty("number", property);
440                     breakpRequest.addThreadFilter(thread);
441                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
442                 } catch ( Exception e1 ) {
443                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
444                     breakpRequest    = null;
445                 }
446             }
447         } catch ( Exception e2 ) {
448             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
449             breakpRequest    = null;
450         }
451 
452         if (breakpRequest == null) {
453             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
454             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
455         }
456 
457         log2("      a breakpoint has been set up");
458         return breakpRequest;
459     }

462     private void getEventSet()
463                  throws JDITestRuntimeException {
464         try {
465 //            log2("       eventSet = eventQueue.remove(waitTime);");
466             eventSet = eventQueue.remove(waitTime);
467             if (eventSet == null) {
468                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
469             }
470 //            log2("       eventIterator = eventSet.eventIterator;");
471             eventIterator = eventSet.eventIterator();
472         } catch ( Exception e ) {
473             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
474         }
475     }

478     private void breakpointForCommunication()
479                  throws JDITestRuntimeException {
480 
481         log2("breakpointForCommunication");
482         getEventSet();
483 
484         if (eventIterator.nextEvent() instanceof BreakpointEvent)
485             return;
486 
487         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
488     }
