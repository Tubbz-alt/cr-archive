 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadStartRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks up on the following assertion:               <BR>
 42  *    Restricts the events generated by this request            <BR>
 43  *    to those in the given thread.                             <BR>
 44  * The cases to test include three thread's states:             <BR>
 45  * not started, running, dead.                                  <BR>
 46  * <BR>
 47  * The test has three phases and works as follows.                      <BR>
 48  * <BR>
 49  * In first phase,                                                      <BR>
 50  * upon launching debuggee's VM which will be suspended,                <BR>
 51  * a debugger waits for the VMStartEvent within a predefined            <BR>
 52  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 53  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 54  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 55  * and waits for the event within the predefined time interval.         <BR>
 56  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 57  * Upon getting the ClassPrepareEvent,                                  <BR>
 58  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 59  * within debuggee's special methodForCommunication().                  <BR>
 60  * <BR>
 61  * In second phase to check the assertion,                              <BR>
 62  * the debugger and the debuggee perform the following.                 <BR>
 63  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 64  * - The debuggee creates new thread, thread1, and invokes              <BR>
 65  *   the methodForCommunication to be suspended and                     <BR>
 66  *   to inform the debugger with the event.                             <BR>
 67  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 68  *   - creates ThreadStartRequest1 and                                  <BR>
 69  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 70  *   - creates ThreadStartRequest2,                                     <BR>
 71  *     resumes the debuggee to get the thread1 started, and             <BR>
 72  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 73  *   - creates ThreadStartRequest3, waits the thread1 to die, and       <BR>
 74  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 75  * <BR>
 76  * In third phase, at the end of the test,                              <BR>
 77  * the debuggee changes the value of the "instruction"  which           <BR>
 78  * the debugger and debuggee use to inform each other of needed actions,<BR>
 79  * and both end.
 80  * <BR>
 81  *
 82 
 83  */
 84

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : e");
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : e");
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         EventRequest eventRequest1 = null;
312         EventRequest eventRequest2 = null;
313         EventRequest eventRequest3 = null;
314 
315         ThreadReference thread1 = null;
316 
317         String thread1Name = "thread1";
318 
319         String property1 = "ThreadStartRequest1";
320         String property2 = "ThreadStartRequest2";
321         String property3 = "ThreadStartRequest3";
322 
323         ReferenceType testClassReference = null;
324 
325 
326         for (int i = 0; ; i++) {
327 
328             vm.resume();
329             breakpointForCommunication();
330 
331             int instruction = ((IntegerValue)
332                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
333 
334             if (instruction == 0) {
335                 vm.resume();
336                 break;
337             }
338 
339             log1(":::::: case: # " + i);
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
342 
343             switch (i) {
344 
345               case 0:
346                       testClassReference =
347                            (ReferenceType) vm.classesByName(testedClassName).get(0);
348 
349                       thread1 = (ThreadReference) debuggeeClass.getValue(
350                                                   debuggeeClass.fieldByName(thread1Name));
351 
352                       eventRequest1 = setting2ThreadStartRequest (null,
353                                              EventRequest.SUSPEND_NONE, property1);
354 
355                       try {
356                           log2("......eventRequest1.addThreadFilter(thread1);");
357                           log2("        no Exception expected");
358                           ((ThreadStartRequest)eventRequest1).addThreadFilter(thread1);
359                       } catch ( Exception e ) {
360                           log3("ERROR: Exception : " + e);
361                           testExitCode = FAILED;
362                       }
363                       break;
364 
365               case 1:
366                       eventRequest2 = setting2ThreadStartRequest (null,
367                                              EventRequest.SUSPEND_NONE, property2);
368 
369                       try {
370                           log2("......eventRequest2.addThreadFilter(thread1);");
371                           log2("        no Exception expected");
372                           ((ThreadStartRequest)eventRequest2).addThreadFilter(thread1);
373                       } catch ( Exception e ) {
374                           log3("ERROR: Exception : " + e);
375                           testExitCode = FAILED;
376                       }
377                       break;
378 
379               case 2:
380                       eventRequest3 = setting2ThreadStartRequest (null,
381                                              EventRequest.SUSPEND_NONE, property3);
382 
383                       try {
384                           log2("......eventRequest3.addThreadFilter(thread1);");
385                           log2("        no Exception expected");
386                           ((ThreadStartRequest)eventRequest3).addThreadFilter(thread1);
387                       } catch ( Exception e ) {
388                           log3("ERROR: Exception : " + e);
389                           testExitCode = FAILED;
390                       }
391                       break;
392 
393               default:
394                       throw new JDITestRuntimeException("** default case 2 **");
395             }
396 
397             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
398         }
399         log1("    TESTING ENDS");
400         return;

403    /*
404     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
405     *                                             String, String, String)
406     *
407     * It sets up a breakpoint at given line number within a given method in a given class
408     * for a given thread.
409     *
410     * Return value: BreakpointRequest object  in case of success
411     *
412     * JDITestRuntimeException   in case of an Exception thrown within the method
413     */
414 
415     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
416                                                   ReferenceType testedClass,
417                                                   String methodName,
418                                                   String bpLine,
419                                                   String property)
420             throws JDITestRuntimeException {
421 
422         log2("......setting up a breakpoint:");
423         log2("       thread: " + thread + "; class: " + testedClass +
424                         "; method: " + methodName + "; line: " + bpLine);
425 
426         List              alllineLocations = null;
427         Location          lineLocation     = null;
428         BreakpointRequest breakpRequest    = null;
429 
430         try {
431             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
432 
433             alllineLocations = method.allLineLocations();
434 
435             int n =
436                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
437             if (n > alllineLocations.size()) {
438                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
439             } else {
440                 lineLocation = (Location) alllineLocations.get(n);
441                 try {
442                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
443                     breakpRequest.putProperty("number", property);
444                     breakpRequest.addThreadFilter(thread);
445                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
446                 } catch ( Exception e1 ) {
447                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
448                     breakpRequest    = null;
449                 }
450             }
451         } catch ( Exception e2 ) {
452             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
453             breakpRequest    = null;
454         }
455 
456         if (breakpRequest == null) {
457             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
458             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
459         }
460 
461         log2("      a breakpoint has been set up");
462         return breakpRequest;
463     }

466     private void getEventSet()
467                  throws JDITestRuntimeException {
468         try {
469 //            log2("       eventSet = eventQueue.remove(waitTime);");
470             eventSet = eventQueue.remove(waitTime);
471             if (eventSet == null) {
472                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
473             }
474 //            log2("       eventIterator = eventSet.eventIterator;");
475             eventIterator = eventSet.eventIterator();
476         } catch ( Exception e ) {
477             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
478         }
479     }

482     private void breakpointForCommunication()
483                  throws JDITestRuntimeException {
484 
485         log2("breakpointForCommunication");
486         getEventSet();
487 
488         if (eventIterator.nextEvent() instanceof BreakpointEvent)
489             return;
490 
491         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
492     }

496     private ThreadStartRequest setting2ThreadStartRequest( ThreadReference thread,
497                                                            int             suspendPolicy,
498                                                            String          property)
499             throws JDITestRuntimeException {
500         try {
501             ThreadStartRequest tsr = eventRManager.createThreadStartRequest();
502             if (thread != null)
503                 tsr.addThreadFilter(thread);
504             tsr.addCountFilter(1);
505             tsr.setSuspendPolicy(suspendPolicy);
506             tsr.putProperty("number", property);
507             return tsr;
508         } catch ( Exception e ) {
509             log3("ERROR: ATTENTION: Exception within settingThreadStartRequest() : " + e);
510             log3("       ThreadStartRequest HAS NOT BEEN SET UP");
511             throw new JDITestRuntimeException("** FAILURE to set up ThreadStartRequest **");
512         }
513     }
