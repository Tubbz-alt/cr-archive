 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                              <BR>
 53  * upon launching debuggee's VM which will be suspended,        <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an ExceptionRequest and enables it,                      <BR>
 72  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

255     *               1 - ubnormal end of the test
256     *               2 - VMDisconnectedException while test phase

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         if ( !vm.canUseInstanceFilters() ) {
285             log2("......vm.canUseInstanceFilters == false :: test cancelled");
286             vm.exit(PASS_BASE);
287             return;
288         }
289 
290 
291         eventRManager = vm.eventRequestManager();
292 
293         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
294         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
295         cpRequest.addClassFilter(debuggeeName);
296 
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307 
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         String bPointMethod = "methodForCommunication";
311         String lineForComm  = "lineForComm";
312 
313         ThreadReference   mainThread = threadByName("main");
314 
315         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
316                                              debuggeeClass,
317                                             bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320     //------------------------------------------------------  testing section
321 
322         log1("     TESTING BEGINS");
323 
324         EventRequest eventRequest1 = null;
325 
326         ThreadReference thread1     = null;
327         String          thread1Name = "thread1";
328 
329         String property1 = "ExceptionRequest1";
330 
331         ReferenceType testClassReference = null;
332 
333         String         arrayName = "objTC";
334         ObjectReference instance = null;
335 
336 
337         for (int i = 0; ; i++) {
338 
339             vm.resume();
340             breakpointForCommunication();
341 
342             int instruction = ((IntegerValue)
343                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
344 
345             if (instruction == 0) {
346                 vm.resume();
347                 break;
348             }
349 
350             log1(":::::: case: # " + i);
351 
352             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
353 
354             switch (i) {
355 
356               case 0:
357                       testClassReference =
358                            (ReferenceType) vm.classesByName(testedClassName).get(0);
359 
360                       thread1 = (ThreadReference) debuggeeClass.getValue(
361                                                   debuggeeClass.fieldByName(thread1Name));
362 
363                       eventRequest1 = setting2ExceptionRequest (thread1,
364                                              testClassReference,
365                                              EventRequest.SUSPEND_NONE, "ExceptionRequest1");
366 
367                       instance = (ObjectReference)
368                                  ((ArrayReference) debuggeeClass.getValue(
369                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
370 
371                       eventRequest1.enable();
372 
373                      try {
374                          log2("......eventRequest1.addInstanceFilter(instance);");
375                          log2("        InvalidRequestStateException expected");
376                          ((ExceptionRequest) eventRequest1).addInstanceFilter(instance);
377                          log3("ERROR: no InvalidRequestStateException");
378                          testExitCode = FAILED;
379                      } catch ( InvalidRequestStateException e ) {
380                          log2("        InvalidRequestStateException");
381                      }
382 
383                      log2("......eventRequest1.disable();");
384                      eventRequest1.disable();
385                      try {
386                          log2("......eventRequest1.addInstanceFilter(instance);");
387                          log2("        no InvalidRequestStateException expected");
388                          ((ExceptionRequest) eventRequest1).addInstanceFilter(instance);
389                          log2("        no InvalidRequestStateException");
390                      } catch ( InvalidRequestStateException e ) {
391                          log3("ERROR: InvalidRequestStateException");
392                          testExitCode = FAILED;
393                      }
394 
395                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
396                      eventRManager.deleteEventRequest(eventRequest1);
397                      try {
398                          log2("......eventRequest1.addInstanceFilter(instance);");
399                          log2("        InvalidRequestStateException expected");
400                          ((ExceptionRequest) eventRequest1).addInstanceFilter(instance);
401                          log3("ERROR: no InvalidRequestStateException");
402                          testExitCode = FAILED;
403                      } catch ( InvalidRequestStateException e ) {
404                          log2("        InvalidRequestStateException");
405                      }
406 
407                       break;
408 
409               default:
410                       throw new JDITestRuntimeException("** default case 2 **");
411             }
412             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
413         }
414         log1("    TESTING ENDS");
415         return;
416     }

418     private ThreadReference threadByName(String name)
419                  throws JDITestRuntimeException {
420 
421         List         all = vm.allThreads();
422         ListIterator li  = all.listIterator();
423 
424         for (; li.hasNext(); ) {
425             ThreadReference thread = (ThreadReference) li.next();
426             if (thread.name().equals(name))
427                 return thread;
428         }
429         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
430     }

432    /*
433     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
434     *                                             String, String, String)
435     *
436     * It sets up a breakpoint at given line number within a given method in a given class
437     * for a given thread.
438     *
439     * Return value: BreakpointRequest object  in case of success
440     *
441     * JDITestRuntimeException   in case of an Exception thrown within the method
442     */
443 
444     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
445                                                   ReferenceType testedClass,
446                                                   String methodName,
447                                                   String bpLine,
448                                                   String property)
449             throws JDITestRuntimeException {
450 
451         log2("......setting up a breakpoint:");
452         log2("       thread: " + thread + "; class: " + testedClass +
453                         "; method: " + methodName + "; line: " + bpLine);
454 
455         List              alllineLocations = null;
456         Location          lineLocation     = null;
457         BreakpointRequest breakpRequest    = null;
458 
459         try {
460             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
461 
462             alllineLocations = method.allLineLocations();
463 
464             int n =
465                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
466             if (n > alllineLocations.size()) {
467                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
468             } else {
469                 lineLocation = (Location) alllineLocations.get(n);
470                 try {
471                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
472                     breakpRequest.putProperty("number", property);
473                     breakpRequest.addThreadFilter(thread);
474                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
475                 } catch ( Exception e1 ) {
476                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
477                     breakpRequest    = null;
478                 }
479             }
480         } catch ( Exception e2 ) {
481             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
482             breakpRequest    = null;
483         }
484 
485         if (breakpRequest == null) {
486             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
487             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
488         }
489 
490         log2("      a breakpoint has been set up");
491         return breakpRequest;
492     }

495     private void getEventSet()
496                  throws JDITestRuntimeException {
497         try {
498 //            log2("       eventSet = eventQueue.remove(waitTime);");
499             eventSet = eventQueue.remove(waitTime);
500             if (eventSet == null) {
501                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
502             }
503 //            log2("       eventIterator = eventSet.eventIterator;");
504             eventIterator = eventSet.eventIterator();
505         } catch ( Exception e ) {
506             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
507         }
508     }

511     private void breakpointForCommunication()
512                  throws JDITestRuntimeException {
513 
514         log2("breakpointForCommunication");
515         getEventSet();
516 
517         if (eventIterator.nextEvent() instanceof BreakpointEvent)
518             return;
519 
520         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
521     }

525     private ExceptionRequest setting2ExceptionRequest ( ThreadReference thread,
526                                                        ReferenceType   testedClass,
527                                                        int             suspendPolicy,
528                                                        String          property       )
529             throws JDITestRuntimeException {
530         try {
531             log2("......setting up ExceptionRequest:");
532             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
533 
534             ExceptionRequest
535             excr = eventRManager.createExceptionRequest(null, true, true);
536             excr.putProperty("number", property);
537             if (thread != null)
538                 excr.addThreadFilter(thread);
539             excr.addClassFilter(testedClass);
540             excr.setSuspendPolicy(suspendPolicy);
541 
542             log2("      ExceptionRequest has been set up");
543             return excr;
544         } catch ( Exception e ) {
545             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
546             log3("       ExceptionRequest HAS NOT BEEN SET UP");
547             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
548         }
