 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.field()</code>           <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  *  The case to test include AccessWatchpointRequest.                   <BR>
 46  *  The test checks if a Field object, an argument of the method        <BR>
 47  *     EventRequestManager.createAccessWatchpointRequest(Field),        <BR>
 48  *  is equal to one returned by the method WatchpointRequest.field()    <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 68  *   and to inform the debugger with the event.                         <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates an AccessWatchpointRequest and gets Field object filed1, <BR>
 71  *   - gets Field object field2 with the method WatchpointRequest.field(),<BR>
 72  *   - compares the Filed objects.                                      <BR>
 73  * <BR>
 74  * In third phase, at the end of the test, the debuggee changes         <BR>
 75  * the value of the "instruction" which the debugger and debuggee       <BR>
 76  * use to inform each other of needed actions,  and both end.           <BR>
 77  * <BR>
 78  */

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : " + e);
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : " + e);
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if (eventIterator.nextEvent() instanceof VMDeathEvent)
259                 return 0;
260 
261             log3("ERROR: last event is not the VMDeathEvent");
262             return 1;
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         eventRManager = vm.eventRequestManager();
277 
278         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
279         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
280         cpRequest.addClassFilter(debuggeeName);
281 
282         cpRequest.enable();
283         vm.resume();
284         getEventSet();
285         cpRequest.disable();
286 
287         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
288         debuggeeClass = event.referenceType();
289 
290         if (!debuggeeClass.name().equals(debuggeeName))
291            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
292 
293         log2("      received: ClassPrepareEvent for debuggeeClass");
294 
295         String bPointMethod = "methodForCommunication";
296         String lineForComm  = "lineForComm";
297 
298         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
299 
300         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
301                                              debuggeeClass,
302                                             bPointMethod, lineForComm, "zero");
303         bpRequest.enable();
304 
305     //------------------------------------------------------  testing section
306 
307         log1("     TESTING BEGINS");
308 
309         EventRequest  eventRequest1 = null;
310         String        property1     = "AccessWatchpointRequest1";
311         String        fieldName1    = "var101";
312 
313         ReferenceType testClassReference = null;
314 
315 
316         for (int i = 0; ; i++) {
317 
318             vm.resume();
319             breakpointForCommunication();
320 
321             int instruction = ((IntegerValue)
322                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
323 
324             if (instruction == 0) {
325                 vm.resume();
326                 break;
327             }
328 
329 
330             log1(":::::: case: # " + i);
331 
332             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
333 
334             switch (i) {
335 
336               case 0:
337                      testClassReference =
338                            (ReferenceType) vm.classesByName(testedClassName1).get(0);
339 
340                      log2("......setting up AccessWatchpointRequest and getting Field object field1");
341                      eventRequest1 = setting21AccessWatchpointRequest (null,
342                                              testClassReference, fieldName1,
343                                              EventRequest.SUSPEND_NONE, property1);
344 
345                      log2("......getting: field2 = ((WatchpointRequest) eventRequest1).field();");
346                      field2 = ((WatchpointRequest) eventRequest1).field();
347 
348                      log2("      checking up on equality of field1 and field2");
349                      if ( !field1.equals(field2) ) {
350                          testExitCode = FAILED;
351                          log3("ERROR: Field objects are not equal");
352                      }
353 
354                       break;
355 
356               default:
357                       throw new JDITestRuntimeException("** default case 2 **");
358             }
359 
360             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
361         }
362         log1("    TESTING ENDS");
363         return;
364     }

366    /*
367     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
368     *                                             String, String, String)
369     *
370     * It sets up a breakpoint at given line number within a given method in a given class
371     * for a given thread.
372     *
373     * Return value: BreakpointRequest object  in case of success
374     *
375     * JDITestRuntimeException   in case of an Exception thrown within the method
376     */
377 
378     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
379                                                   ReferenceType testedClass,
380                                                   String methodName,
381                                                   String bpLine,
382                                                   String property)
383             throws JDITestRuntimeException {
384 
385         log2("......setting up a breakpoint:");
386         log2("       thread: " + thread + "; class: " + testedClass +
387                         "; method: " + methodName + "; line: " + bpLine);
388 
389         List              alllineLocations = null;
390         Location          lineLocation     = null;
391         BreakpointRequest breakpRequest    = null;
392 
393         try {
394             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
395 
396             alllineLocations = method.allLineLocations();
397 
398             int n =
399                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
400             if (n > alllineLocations.size()) {
401                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
402             } else {
403                 lineLocation = (Location) alllineLocations.get(n);
404                 try {
405                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
406                     breakpRequest.putProperty("number", property);
407                     breakpRequest.addThreadFilter(thread);
408                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
409                 } catch ( Exception e1 ) {
410                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
411                     breakpRequest    = null;
412                 }
413             }
414         } catch ( Exception e2 ) {
415             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
416             breakpRequest    = null;
417         }
418 
419         if (breakpRequest == null) {
420             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
421             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
422         }
423 
424         log2("      a breakpoint has been set up");
425         return breakpRequest;
426     }

429     private void getEventSet()
430                  throws JDITestRuntimeException {
431         try {
432 //            log2("       eventSet = eventQueue.remove(waitTime);");
433             eventSet = eventQueue.remove(waitTime);
434             if (eventSet == null) {
435                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
436             }
437 //            log2("       eventIterator = eventSet.eventIterator;");
438             eventIterator = eventSet.eventIterator();
439         } catch ( Exception e ) {
440             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
441         }
442     }

445     private void breakpointForCommunication()
446                  throws JDITestRuntimeException {
447 
448         log2("breakpointForCommunication");
449         getEventSet();
450 
451         if (eventIterator.nextEvent() instanceof BreakpointEvent)
452             return;
453 
454         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
455     }

459     private AccessWatchpointRequest setting21AccessWatchpointRequest (
460                                                   ThreadReference thread,
461                                                   ReferenceType   fieldClass,
462                                                   String          fieldName,
463                                                   int             suspendPolicy,
464                                                   String          property        )
465             throws JDITestRuntimeException {
466         try {
467             log2("......setting up AccessWatchpointRequest:");
468             log2("       thread: " + thread + "; fieldClass: " + fieldClass + "; fieldName: " + fieldName);
469             Field field = fieldClass.fieldByName(fieldName);
470 
471             field1 = field;
472 
473             AccessWatchpointRequest
474             awr = eventRManager.createAccessWatchpointRequest(field);
475             awr.putProperty("number", property);
476 
477             if (thread != null)
478                 awr.addThreadFilter(thread);
479             awr.setSuspendPolicy(suspendPolicy);
480 
481             log2("      AccessWatchpointRequest has been set up");
482             return awr;
483         } catch ( Exception e ) {
484             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
485             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
486             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
487         }
