 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodExitRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodExitRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a MethodExitRequest and enables it,                      <BR>
 72  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         if ( !vm.canUseInstanceFilters() ) {
279             log2("......vm.canUseInstanceFilters == false :: test cancelled");
280             vm.exit(PASS_BASE);
281             return;
282         }
283 
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         EventRequest eventRequest1 = null;
319 
320         ThreadReference thread1 = null;
321         ThreadReference thread2 = null;
322 
323         String thread1Name = "thread1";
324         String thread2Name = "thread2";
325 
326         String property1 = "MethodExitRequest1";
327 
328         ReferenceType testClassReference = null;
329 
330         String         arrayName = "objTC";
331         ObjectReference instance = null;
332 
333 
334         for (int i = 0; ; i++) {
335 
336             vm.resume();
337             breakpointForCommunication();
338 
339             int instruction = ((IntegerValue)
340                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
341 
342             if (instruction == 0) {
343                 vm.resume();
344                 break;
345             }
346 
347             log1(":::::: case: # " + i);
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351             switch (i) {
352 
353               case 0:
354                       testClassReference =
355                            (ReferenceType) vm.classesByName(testedClassName).get(0);
356 
357                       thread1 = (ThreadReference) debuggeeClass.getValue(
358                                                   debuggeeClass.fieldByName(thread1Name));
359 
360                       eventRequest1 = setting2MethodExitRequest (null,
361                                              testClassReference,
362                                              EventRequest.SUSPEND_NONE, property1);
363 
364                       instance = (ObjectReference)
365                                  ((ArrayReference) debuggeeClass.getValue(
366                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
367 
368                       ((MethodExitRequest) eventRequest1).addInstanceFilter(instance);
369 
370                       eventRequest1.enable();
371 
372                      try {
373                          log2("......eventRequest1.addInstanceFilter(instance);");
374                          log2("        InvalidRequestStateException expected");
375                          ((MethodExitRequest) eventRequest1).addInstanceFilter(instance);
376                          log3("ERROR: no InvalidRequestStateException");
377                          testExitCode = FAILED;
378                      } catch ( InvalidRequestStateException e ) {
379                          log2("        InvalidRequestStateException");
380                      }
381 
382                      log2("......eventRequest1.disable();");
383                      eventRequest1.disable();
384                      try {
385                          log2("......eventRequest1.addInstanceFilter(instance);");
386                          log2("        no InvalidRequestStateException expected");
387                          ((MethodExitRequest) eventRequest1).addInstanceFilter(instance);
388                          log2("        no InvalidRequestStateException");
389                      } catch ( InvalidRequestStateException e ) {
390                          log3("ERROR: InvalidRequestStateException");
391                          testExitCode = FAILED;
392                      }
393 
394                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
395                      eventRManager.deleteEventRequest(eventRequest1);
396                      try {
397                          log2("......eventRequest1.addInstanceFilter(instance);");
398                          log2("        InvalidRequestStateException expected");
399                          ((MethodExitRequest) eventRequest1).addInstanceFilter(instance);
400                          log3("ERROR: no InvalidRequestStateException");
401                          testExitCode = FAILED;
402                      } catch ( InvalidRequestStateException e ) {
403                          log2("        InvalidRequestStateException");
404                      }
405                       break;
406 
407               default:
408                       throw new JDITestRuntimeException("** default case 2 **");
409             }
410 
411             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
412         }
413         log1("    TESTING ENDS");
414         return;
415     }

417    /*
418     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
419     *                                             String, String, String)
420     *
421     * It sets up a breakpoint at given line number within a given method in a given class
422     * for a given thread.
423     *
424     * Return value: BreakpointRequest object  in case of success
425     *
426     * JDITestRuntimeException   in case of an Exception thrown within the method
427     */
428 
429     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
430                                                   ReferenceType testedClass,
431                                                   String methodName,
432                                                   String bpLine,
433                                                   String property)
434             throws JDITestRuntimeException {
435 
436         log2("......setting up a breakpoint:");
437         log2("       thread: " + thread + "; class: " + testedClass +
438                         "; method: " + methodName + "; line: " + bpLine);
439 
440         List              alllineLocations = null;
441         Location          lineLocation     = null;
442         BreakpointRequest breakpRequest    = null;
443 
444         try {
445             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
446 
447             alllineLocations = method.allLineLocations();
448 
449             int n =
450                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
451             if (n > alllineLocations.size()) {
452                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
453             } else {
454                 lineLocation = (Location) alllineLocations.get(n);
455                 try {
456                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
457                     breakpRequest.putProperty("number", property);
458                     breakpRequest.addThreadFilter(thread);
459                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
460                 } catch ( Exception e1 ) {
461                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
462                     breakpRequest    = null;
463                 }
464             }
465         } catch ( Exception e2 ) {
466             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
467             breakpRequest    = null;
468         }
469 
470         if (breakpRequest == null) {
471             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
472             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
473         }
474 
475         log2("      a breakpoint has been set up");
476         return breakpRequest;
477     }

480     private void getEventSet()
481                  throws JDITestRuntimeException {
482         try {
483 //            log2("       eventSet = eventQueue.remove(waitTime);");
484             eventSet = eventQueue.remove(waitTime);
485             if (eventSet == null) {
486                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
487             }
488 //            log2("       eventIterator = eventSet.eventIterator;");
489             eventIterator = eventSet.eventIterator();
490         } catch ( Exception e ) {
491             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
492         }
493     }

496     private void breakpointForCommunication()
497                  throws JDITestRuntimeException {
498 
499         log2("breakpointForCommunication");
500         getEventSet();
501 
502         if (eventIterator.nextEvent() instanceof BreakpointEvent)
503             return;
504 
505         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
506     }

510     private MethodExitRequest setting2MethodExitRequest ( ThreadReference thread,
511                                                           ReferenceType   testedClass,
512                                                           int             suspendPolicy,
513                                                           String          property       )
514             throws JDITestRuntimeException {
515         try {
516             log2("......setting up MethodExitRequest:");
517             log2("       thread: " + thread + "; class: " + testedClass +  "; property: " + property);
518 
519             MethodExitRequest
520             menr = eventRManager.createMethodExitRequest();
521             menr.putProperty("number", property);
522             if (thread != null)
523                 menr.addThreadFilter(thread);
524             menr.addClassFilter(testedClass);
525             menr.setSuspendPolicy(suspendPolicy);
526 
527             log2("      a MethodExitRequest has been set up");
528             return menr;
529         } catch ( Exception e ) {
530             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
531             log3("       MethodExitRequest HAS NOT BEEN SET UP");
532             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
533         }
