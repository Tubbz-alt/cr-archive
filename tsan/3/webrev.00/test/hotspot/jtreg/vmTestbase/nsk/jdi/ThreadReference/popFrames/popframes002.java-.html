 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up that                                      <BR>
 46  * a BreakpointRequest in debuggee's third thread               <BR>
 47  * is not affected when                                         <BR>
 48  * its second thread is suspended at a breakpoint in a method   <BR>
 49  * and method's frame is popped by a debugger.                  <BR>
 50  * <BR>
 51  * The test has two phases and works as follows.                <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to perform the above check,                          <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates two threads, thread2 and thread3, which will  <BR>
 69  * call a tested synchronized poppedMethod() from their run() methods.  <BR>
 70  * After getting the threads started, the debuggee invokes the          <BR>
 71  * methodForCommunication() to inform the debugger of the threads.      <BR>
 72  * - The debugger sets up two breakpoints within the poppedMethod(),    <BR>
 73  * one for the thread2 and another one for the thread3, and             <BR>
 74  * resumes debuggee's main thread.                                      <BR>
 75  * - The main thread releases the thread2 and the thread3 in the order  <BR>
 76  * which makes the thread2 to be suspended at the breakpoint within     <BR>
 77  * the poppedMethod and the thread3 to be waiting for releasing the     <BR>
 78  * synchronized poppedMethod by the thread2. Then                       <BR>
 79  * the main thread informs the debugger via methodForCommunication().   <BR>
 80  * - After getting both breakpoint events, from the thread2 and the main,<BR>
 81  * the debugger invokes the popFrames() method and waits for            <BR>
 82  * the breakpoint event in poppedMethod from the thread3.               <BR>
 83  * If no the event, the test FAILED.                                    <BR>
 84  */
 85

168     private int runThis (String argv[], PrintStream out) {
169 
170         argsHandler     = new ArgumentHandler(argv);
171         logHandler      = new Log(out, argsHandler);
172         Binder binder   = new Binder(argsHandler, logHandler);
173 
174         waitTime        = argsHandler.getWaitTime() * 60000;
175 
176         try {
177             log2("launching a debuggee :");
178             log2("       " + debuggeeName);
179             if (argsHandler.verbose()) {
180                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
181             } else {
182                 debuggee = binder.bindToDebugee(debuggeeName);
183             }
184             if (debuggee == null) {
185                 log3("ERROR: no debuggee launched");
186                 return FAILED;
187             }
188             log2("debuggee launched");
189         } catch ( Exception e ) {
190             log3("ERROR: Exception : " + e);
191             log2("       test cancelled");
192             return FAILED;
193         }
194 
195         debuggee.redirectOutput(logHandler);
196 
197         vm = debuggee.VM();
198 
199         eventQueue = vm.eventQueue();
200         if (eventQueue == null) {
201             log3("ERROR: eventQueue == null : TEST ABORTED");
202             vm.exit(PASS_BASE);
203             return FAILED;
204         }
205 
206         log2("invocation of the method runTest()");
207         switch (runTest()) {
208 
209             case 0 :  log2("test phase has finished normally");
210                       log2("   waiting for the debuggee to finish ...");
211                       debuggee.waitFor();
212 
213                       log2("......getting the debuggee's exit status");
214                       int status = debuggee.getStatus();
215                       if (status != PASS_BASE) {
216                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
217                               status + " != PASS_BASE");
218                           testExitCode = FAILED;
219                       } else {
220                           log2("......debuggee returned expected exit status: " +
221                               status + " == PASS_BASE");
222                       }
223                       break;
224 
225             default : log3("ERROR: runTest() returned unexpected value");
226 
227             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
228                       log2("......forcing: vm.exit();");
229                       testExitCode = FAILED;
230                       try {
231                           vm.exit(PASS_BASE);
232                       } catch ( Exception e ) {
233                           log3("ERROR: Exception : e");
234                       }
235                       break;
236 
237             case 2 :  log3("test cancelled due to VMDisconnectedException");
238                       log2("......trying: vm.process().destroy();");
239                       testExitCode = FAILED;
240                       try {
241                           Process vmProcess = vm.process();
242                           if (vmProcess != null) {
243                               vmProcess.destroy();
244                           }
245                       } catch ( Exception e ) {
246                           log3("ERROR: Exception : e");
247                       }
248                       break;
249             }
250 
251         return testExitCode;
252     }

261     private int runTest() {
262 
263         try {
264             testRun();
265 
266             log2("waiting for VMDeathEvent");
267             getEventSet();
268             if (eventIterator.nextEvent() instanceof VMDeathEvent)
269                 return 0;
270 
271             log3("ERROR: last event is not the VMDeathEvent");
272             return 1;
273         } catch ( VMDisconnectedException e ) {
274             log3("ERROR: VMDisconnectedException : " + e);
275             return 2;
276         } catch ( Exception e ) {
277             log3("ERROR: Exception : " + e);
278             return 1;
279         }
280

283     private void testRun()
284                  throws JDITestRuntimeException, Exception {
285 
286         eventRManager = vm.eventRequestManager();
287 
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291 
292         cpRequest.enable();
293         vm.resume();
294         getEventSet();
295         cpRequest.disable();
296 
297         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
298         debuggeeClass = event.referenceType();
299 
300         if (!debuggeeClass.name().equals(debuggeeName))
301            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
302 
303         log2("      received: ClassPrepareEvent for debuggeeClass");
304 
305         if ( !vm.canPopFrames() ) {
306             log2("......vm.canPopFrames() == false : test is cancelled");
307             vm.resume();
308             return;
309         }
310 
311 
312         String bPointMethod = "methodForCommunication";
313         String lineForComm  = "lineForComm";
314 
315         log2("......setting BreakpointRequest (bpRequest) in main thread");
316         bpRequest = settingBreakpoint(threadByName("main"),
317                                           debuggeeClass,
318                                           bPointMethod, lineForComm, "zero");
319         log2("bpRequest.enable();");
320         bpRequest.enable();
321 
322     //------------------------------------------------------  testing section
323 
324         log1("     TESTING BEGINS");
325 
326         label0:
327         {
328             vm.resume();
329             breakpointForCommunication();
330 
331             String thread2Name         = "thread2";
332             ThreadReference thread2Ref = threadByName(thread2Name);
333 
334             String thread3Name         = "thread3";
335             ThreadReference thread3Ref = threadByName(thread3Name);
336 
337             String poppedMethod    = "poppedMethod";
338             String breakpointLine  = "breakpointLine";
339             String testVariable    = "testVar1";
340 
341             log2("......setting breakpoints in poppedMethod");
342             try {
343                 breakpointRequest2 = settingBreakpoint(thread2Ref, debuggeeClass,
344                                           poppedMethod, breakpointLine, "one");
345             } catch ( Exception e ) {
346                 throw e;
347             }
348             log2("......breakpointRequest2.enable();");
349             breakpointRequest2.enable();
350             try {
351                 breakpointRequest3 = settingBreakpoint(thread3Ref, debuggeeClass,
352                                           poppedMethod, breakpointLine, "two");
353             } catch ( Exception e ) {
354                 throw e;
355             }
356             log2("......breakpointRequest3.enable();");
357             breakpointRequest3.enable();
358 
359 
360             log2("......eventSet.resume();");
361             eventSet.resume();
362             log2("......breakpointInMethod(breakpointRequest2);");
363             breakpointInMethod("one");
364 
365             if ( bpRequestFlag == 0 )
366                 breakpointForCommunication();
367 
368             log2("......StackFrame stackFrame = thread2Ref.frame(0);");
369             StackFrame stackFrame = thread2Ref.frame(0);
370 
371             log2("......thread2Ref.popFrames(stackFrame);");
372             try {
373                 thread2Ref.popFrames(stackFrame);
374             } catch ( IncompatibleThreadStateException e ) {
375                 log3("ERROR: IncompatibleThreadStateException");
376                 testExitCode = FAILED;
377                 break label0;
378             }
379 
380             log2("......breakpointInMethod(breakpointRequest3);");
381             try {
382                 breakpointInMethod("two");
383             } catch ( JDITestRuntimeException e ) {
384                 log3("ERROR:  timeout while waiting for thread3's BreakpointEvent");
385                 log2("        THREAD_STATUS_MONITOR == " + ThreadReference.THREAD_STATUS_MONITOR );
386                 log2("        thread3's status      == " + thread3Ref.status());
387                 testExitCode = FAILED;
388             }
389 
390             if (!thread3Ref.isSuspended()) {
391                 log3("ERROR: thread3 is not suspended at the breakpoint");
392                 testExitCode = FAILED;
393                 break label0;
394             }
395         }
396 
397         log2("......disabling breakpointRequests 2&3");
398         breakpointRequest2.disable();
399         breakpointRequest3.disable();
400 
401         vm.resume();
402 
403     //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
404 
405 
406         log1("    TESTING ENDS");
407         return;
408     }

410     private ThreadReference threadByName(String name)
411                  throws JDITestRuntimeException {
412 
413         List         all = vm.allThreads();
414         ListIterator li  = all.listIterator();
415 
416         for (; li.hasNext(); ) {
417             ThreadReference thread = (ThreadReference) li.next();
418             if (thread.name().equals(name))
419                 return thread;
420         }
421         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
422     }

424    /*
425     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
426     *                                             String, String, String)
427     *
428     * It sets up a breakpoint at given line number within a given method in a given class
429     * for a given thread.
430     *
431     * Return value: BreakpointRequest object  in case of success
432     *
433     * JDITestRuntimeException   in case of an Exception thrown within the method
434     */
435 
436     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
437                                                   ReferenceType testedClass,
438                                                   String methodName,
439                                                   String bpLine,
440                                                   String property)
441             throws JDITestRuntimeException {
442 
443         log2("......setting up a breakpoint:");
444         log2("       thread: " + thread + "; class: " + testedClass +
445                         "; method: " + methodName + "; line: " + bpLine);
446 
447         List              alllineLocations = null;
448         Location          lineLocation     = null;
449         BreakpointRequest breakpRequest    = null;
450 
451         try {
452             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
453 
454             alllineLocations = method.allLineLocations();
455 
456             int n =
457                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
458             if (n > alllineLocations.size()) {
459                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
460             } else {
461                 lineLocation = (Location) alllineLocations.get(n);
462                 try {
463                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
464                     breakpRequest.putProperty("number", property);
465                     breakpRequest.addThreadFilter(thread);
466                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
467                 } catch ( Exception e1 ) {
468                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
469                     breakpRequest    = null;
470                 }
471             }
472         } catch ( Exception e2 ) {
473             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
474             breakpRequest    = null;
475         }
476 
477         if (breakpRequest == null) {
478             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
479             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
480         }
481 
482         log2("      a breakpoint has been set up");
483         return breakpRequest;
484     }

487     private void getEventSet()
488                  throws JDITestRuntimeException {
489         try {
490 //            log2("       eventSet = eventQueue.remove(waitTime);");
491             eventSet = eventQueue.remove(waitTime);
492             if (eventSet == null) {
493                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
494             }
495 //            log2("       eventIterator = eventSet.eventIterator;");
496             eventIterator = eventSet.eventIterator();
497         } catch ( Exception e ) {
498             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
499         }
500     }

503     private void breakpointForCommunication()
504                  throws JDITestRuntimeException {
505 
506         log2("breakpointForCommunication");
507         getEventSet();
508 
509         if (eventIterator.nextEvent() instanceof BreakpointEvent)
510             return;
511 
512         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
513     }

517     static int bpRequestFlag = 0;
518 
519     private void breakpointInMethod(String number)
520                  throws JDITestRuntimeException {
521 
522         Event event;
523 
524         for ( ; ; ) {
525             getEventSet();
526             event = eventIterator.nextEvent();
527 
528             if ( !(event instanceof BreakpointEvent) )
529                 throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
530 
531             log2("---->: request().getProperty == " +
532                  event.request().getProperty("number"));
533 
534             if ( event.request().getProperty("number").equals(number) )
535                 return;
536 
537             if ( !event.request().getProperty("number").equals("zero") )
538                  throw new JDITestRuntimeException("** unexpected breakpoint **");
539             bpRequestFlag = 1;
540         }
