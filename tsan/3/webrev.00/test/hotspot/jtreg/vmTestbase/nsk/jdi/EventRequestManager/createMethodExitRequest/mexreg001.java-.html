 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createMethodExitRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *  - Creates a new disabled MethodExitRequest.                 <BR>
 47  * <BR>
 48  * The test has three phases and works as follows.              <BR>
 49  * <BR>
 50  * In first phase,                                              <BR>
 51  * upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 55  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 56  * and waits for the event within the predefined time interval.         <BR>
 57  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 58  * Upon getting the ClassPrepareEvent,                                  <BR>
 59  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 60  * within debuggee's special methodForCommunication().                  <BR>
 61  * <BR>
 62  * In second phase to check the assertion,                              <BR>
 63  * the debugger and the debuggee perform the following.                 <BR>
 64  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 65  * - The debuggee prepares new check case and invokes                   <BR>
 66  *   the methodForCommunication to be suspended and                     <BR>
 67  *   to inform the debugger with the event.                             <BR>
 68  * - Upon getting the BreakpointEvent,                                  <BR>
 69  *   the debugger performs the check required.                          <BR>
 70  * <BR>
 71  * In third phase, at the end of the test, the debuggee changes         <BR>
 72  * the value of the "instruction" which the debugger and debuggee       <BR>
 73  * use to inform each other of needed actions,  and both end.           <BR>
 74  * <BR>
 75  */
 76

155     private int runThis (String argv[], PrintStream out) {
156 
157         argsHandler     = new ArgumentHandler(argv);
158         logHandler      = new Log(out, argsHandler);
159         Binder binder   = new Binder(argsHandler, logHandler);
160 
161         waitTime        = argsHandler.getWaitTime() * 60000;
162 
163         try {
164             log2("launching a debuggee :");
165             log2("       " + debuggeeName);
166             if (argsHandler.verbose()) {
167                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
168             } else {
169                 debuggee = binder.bindToDebugee(debuggeeName);
170             }
171             if (debuggee == null) {
172                 log3("ERROR: no debuggee launched");
173                 return FAILED;
174             }
175             log2("debuggee launched");
176         } catch ( Exception e ) {
177             log3("ERROR: Exception : " + e);
178             log2("       test cancelled");
179             return FAILED;
180         }
181 
182         debuggee.redirectOutput(logHandler);
183 
184         vm = debuggee.VM();
185 
186         eventQueue = vm.eventQueue();
187         if (eventQueue == null) {
188             log3("ERROR: eventQueue == null : TEST ABORTED");
189             vm.exit(PASS_BASE);
190             return FAILED;
191         }
192 
193         log2("invocation of the method runTest()");
194         switch (runTest()) {
195 
196             case 0 :  log2("test phase has finished normally");
197                       log2("   waiting for the debuggee to finish ...");
198                       debuggee.waitFor();
199 
200                       log2("......getting the debuggee's exit status");
201                       int status = debuggee.getStatus();
202                       if (status != PASS_BASE) {
203                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
204                               status + " != PASS_BASE");
205                           testExitCode = FAILED;
206                       } else {
207                           log2("......debuggee returned expected exit status: " +
208                               status + " == PASS_BASE");
209                       }
210                       break;
211 
212             default : log3("ERROR: runTest() returned unexpected value");
213 
214             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
215                       log2("......forcing: vm.exit();");
216                       testExitCode = FAILED;
217                       try {
218                           vm.exit(PASS_BASE);
219                       } catch ( Exception e ) {
220                           log3("ERROR: Exception : " + e);
221                       }
222                       break;
223 
224             case 2 :  log3("test cancelled due to VMDisconnectedException");
225                       log2("......trying: vm.process().destroy();");
226                       testExitCode = FAILED;
227                       try {
228                           Process vmProcess = vm.process();
229                           if (vmProcess != null) {
230                               vmProcess.destroy();
231                           }
232                       } catch ( Exception e ) {
233                           log3("ERROR: Exception : " + e);
234                       }
235                       break;
236             }
237 
238         return testExitCode;
239     }

242    /*
243     * Return value: 0 - normal end of the test
244     *               1 - ubnormal end of the test
245     *               2 - VMDisconnectedException while test phase
246     */
247 
248     private int runTest() {
249 
250         try {
251             testRun();
252 
253             log2("waiting for VMDeathEvent");
254             getEventSet();
255             if (eventIterator.nextEvent() instanceof VMDeathEvent)
256                 return 0;
257 
258             log3("ERROR: last event is not the VMDeathEvent");
259             return 1;
260         } catch ( VMDisconnectedException e ) {
261             log3("ERROR: VMDisconnectedException : " + e);
262             return 2;
263         } catch ( Exception e ) {
264             log3("ERROR: Exception : " + e);
265             return 1;
266         }
267 
268     }

270     private void testRun()
271                  throws JDITestRuntimeException, Exception {
272 
273         eventRManager = vm.eventRequestManager();
274 
275         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
276         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
277         cpRequest.addClassFilter(debuggeeName);
278 
279         cpRequest.enable();
280         vm.resume();
281         getEventSet();
282         cpRequest.disable();
283 
284         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
285         debuggeeClass = event.referenceType();
286 
287         if (!debuggeeClass.name().equals(debuggeeName))
288            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
289 
290         log2("      received: ClassPrepareEvent for debuggeeClass");
291 
292         String bPointMethod = "methodForCommunication";
293         String lineForComm  = "lineForComm";
294 
295         ThreadReference   mainThread = threadByName("main");
296 
297         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
298                                              debuggeeClass,
299                                             bPointMethod, lineForComm, "zero");
300         bpRequest.enable();
301 
302     //------------------------------------------------------  testing section
303 
304         MethodExitRequest mexRequest1 = null;
305 
306 
307         log1("     TESTING BEGINS");
308 
309         for (int i = 0; ; i++) {
310 
311             vm.resume();
312             breakpointForCommunication();
313 
314             int instruction = ((IntegerValue)
315                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
316 
317             if (instruction == 0) {
318                 vm.resume();
319                 break;
320             }
321 
322             log1(":::::: case: # " + i);
323 
324             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
325 
326             log2("......creating: mexRequest1 = eventRManager.createMethodExitRequest();");
327             mexRequest1 = eventRManager.createMethodExitRequest();
328             log2("      checking up on if request is disabled");
329             if (mexRequest1.isEnabled()) {
330                 testExitCode = FAILED;
331                 log3("ERROR: request is not disabled");
332             }
333 
334             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
335         }
336         log1("    TESTING ENDS");
337         return;
338     }

340     private ThreadReference threadByName(String name)
341                  throws JDITestRuntimeException {
342 
343         List         all = vm.allThreads();
344         ListIterator li  = all.listIterator();
345 
346         for (; li.hasNext(); ) {
347             ThreadReference thread = (ThreadReference) li.next();
348             if (thread.name().equals(name))
349                 return thread;
350         }
351         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
352     }

354    /*
355     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
356     *                                             String, String, String)
357     *
358     * It sets up a breakpoint at given line number within a given method in a given class
359     * for a given thread.
360     *
361     * Return value: BreakpointRequest object  in case of success
362     *
363     * JDITestRuntimeException   in case of an Exception thrown within the method
364     */
365 
366     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
367                                                   ReferenceType testedClass,
368                                                   String methodName,
369                                                   String bpLine,
370                                                   String property)
371             throws JDITestRuntimeException {
372 
373         log2("......setting up a breakpoint:");
374         log2("       thread: " + thread + "; class: " + testedClass +
375                         "; method: " + methodName + "; line: " + bpLine);
376 
377         List              alllineLocations = null;
378         Location          lineLocation     = null;
379         BreakpointRequest breakpRequest    = null;
380 
381         try {
382             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
383 
384             alllineLocations = method.allLineLocations();
385 
386             int n =
387                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
388             if (n > alllineLocations.size()) {
389                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
390             } else {
391                 lineLocation = (Location) alllineLocations.get(n);
392                 try {
393                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
394                     breakpRequest.putProperty("number", property);
395                     breakpRequest.addThreadFilter(thread);
396                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
397                 } catch ( Exception e1 ) {
398                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
399                     breakpRequest    = null;
400                 }
401             }
402         } catch ( Exception e2 ) {
403             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
404             breakpRequest    = null;
405         }
406 
407         if (breakpRequest == null) {
408             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
409             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
410         }
411 
412         log2("      a breakpoint has been set up");
413         return breakpRequest;
414     }

417     private void getEventSet()
418                  throws JDITestRuntimeException {
419         try {
420 //            log2("       eventSet = eventQueue.remove(waitTime);");
421             eventSet = eventQueue.remove(waitTime);
422             if (eventSet == null) {
423                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
424             }
425 //            log2("       eventIterator = eventSet.eventIterator;");
426             eventIterator = eventSet.eventIterator();
427         } catch ( Exception e ) {
428             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
429         }
430     }

433     private void breakpointForCommunication()
434                  throws JDITestRuntimeException {
435 
436         log2("breakpointForCommunication");
437         getEventSet();
438 
439         if (eventIterator.nextEvent() instanceof BreakpointEvent)
440             return;
441 
442         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
443     }
