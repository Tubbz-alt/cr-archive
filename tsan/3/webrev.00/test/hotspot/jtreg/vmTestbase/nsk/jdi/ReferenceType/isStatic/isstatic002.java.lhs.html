 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.isStatic()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Cases for testing include :                                  <BR>
 46  * - nested and outer ClassTypes and InterfaceTypes,            <BR>
 47  *   and nested ArrayTypes;                                     <BR>
 48  * - arrays of primitive values;                                <BR>
 49  * - primitive classes.                                         <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assetion,                               <BR>
 66  * the debugger and the debuggee perform the following loop.            <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee prepares new check case and invokes                   <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent,                                  <BR>
 72  *   the debugger performs the check case required.                     <BR>
 73  * Note. To inform each other of needed actions, the debugger and       <BR>
 74  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 75  * <BR>
 76  * In third phase when at the end,                                      <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */

104     public static int run (String argv[], PrintStream out) {
105 
106         int exitCode = new isstatic002().runThis(argv, out);

108         if (exitCode != PASSED) {

110         }
111         return testExitCode;
112     }

162     private int runThis (String argv[], PrintStream out) {
163 
164         argsHandler     = new ArgumentHandler(argv);
165         logHandler      = new Log(out, argsHandler);
166         Binder binder   = new Binder(argsHandler, logHandler);
167 
168         waitTime        = argsHandler.getWaitTime() * 60000;
169 
170         try {
171             log2("launching a debuggee :");
172             log2("       " + debuggeeName);
173             if (argsHandler.verbose()) {
174                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
175             } else {
176                 debuggee = binder.bindToDebugee(debuggeeName);
177             }
178             if (debuggee == null) {
179                 log3("ERROR: no debuggee launched");
180                 return FAILED;
181             }
182             log2("debuggee launched");
183         } catch ( Exception e ) {
184             log3("ERROR: Exception : " + e);
185             log2("       test cancelled");
186             return FAILED;
187         }
188 
189         debuggee.redirectOutput(logHandler);
190 
191         vm = debuggee.VM();
192 
193         eventQueue = vm.eventQueue();
194         if (eventQueue == null) {
195             log3("ERROR: eventQueue == null : TEST ABORTED");
196             vm.exit(PASS_BASE);
197             return FAILED;
198         }
199 
200         log2("invocation of the method runTest()");
201         switch (runTest()) {
202 
203             case 0 :  log2("test phase has finished normally");
204                       log2("   waiting for the debuggee to finish ...");
205                       debuggee.waitFor();
206 
207                       log2("......getting the debuggee's exit status");
208                       int status = debuggee.getStatus();
209                       if (status != PASS_BASE) {
210                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
211                               status + " != PASS_BASE");
212                           testExitCode = FAILED;
213                       } else {
214                           log2("......debuggee returned expected exit status: " +
215                               status + " == PASS_BASE");
216                       }
217                       break;
218 
219             default : log3("ERROR: runTest() returned unexpected value");
220 
221             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
222                       log2("......forcing: vm.exit();");
223                       testExitCode = FAILED;
224                       try {
225                           vm.exit(PASS_BASE);
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : e");
228                       }
229                       break;
230 
231             case 2 :  log3("test cancelled due to VMDisconnectedException");
232                       log2("......trying: vm.process().destroy();");
233                       testExitCode = FAILED;
234                       try {
235                           Process vmProcess = vm.process();
236                           if (vmProcess != null) {
237                               vmProcess.destroy();
238                           }
239                       } catch ( Exception e ) {
240                           log3("ERROR: Exception : e");
241                       }
242                       break;
243             }
244 
245         return testExitCode;
246     }

249    /*
250     * Return value: 0 - normal end of the test
251     *               1 - ubnormal end of the test
252     *               2 - VMDisconnectedException while test phase
253     */
254 
255     private int runTest() {
256 
257         try {
258             testRun();
259 
260             log2("waiting for VMDeathEvent");
261             getEventSet();
262             if (eventIterator.nextEvent() instanceof VMDeathEvent)
263                 return 0;
264 
265             log3("ERROR: last event is not the VMDeathEvent");
266             return 1;
267         } catch ( VMDisconnectedException e ) {
268             log3("ERROR: VMDisconnectedException : " + e);
269             return 2;
270         } catch ( Exception e ) {
271             log3("ERROR: Exception : " + e);
272             return 1;
273         }
274 
275     }

277     private void testRun()
278                  throws JDITestRuntimeException, Exception {
279 
280         eventRManager = vm.eventRequestManager();
281 
282         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
283         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
284         cpRequest.addClassFilter(debuggeeName);
285 
286         cpRequest.enable();
287         vm.resume();
288         getEventSet();
289         cpRequest.disable();
290 
291         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
292         debuggeeClass = event.referenceType();
293 
294         if (!debuggeeClass.name().equals(debuggeeName))
295            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
296 
297         log2("      received: ClassPrepareEvent for debuggeeClass");
298 
299         String bPointMethod = "methodForCommunication";
300         String lineForComm  = "lineForComm";
301         BreakpointRequest bpRequest;
302 
303         bpRequest = settingBreakpoint(threadByName("main"),
304                                       debuggeeClass,
305                                       bPointMethod, lineForComm, "zero");
306         bpRequest.enable();
307 
308     //------------------------------------------------------  testing section
309 
310         log1("     TESTING BEGINS");
311 
312         for (int i = 0; ; i++) {
313 
314             vm.resume();
315             breakpointForCommunication();
316 
317             int instruction = ((IntegerValue)
318                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
319 
320             if (instruction == 0) {
321                 vm.resume();
322                 break;
323             }
324 
325             log1(":::::: case: # " + i);
326 
327             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
328 
329             String names1[] = {
330 
331                       ".isstatic002aTestClass$NestedIface1[]",      "f",
332                       ".isstatic002aTestClass$NestedIface2[]",      "f",
333                       ".isstatic002aTestClass$StaticNestedClass[]", "f"
334             };
335 
336             log2("----- Cases for testing: ReferenceTypes");
337             for (int i1 = 0; i1 < names1.length; i1 += 2) {
338 
339                 String typeForCheck = mName + names1[i1];
340                 log2("......typeForCheck: " + typeForCheck);
341 
342                 log2("      getting: List classList = vm.classesByName(typeForCheck);");
343                 List classList = vm.classesByName(typeForCheck);
344                 if (classList.size() != 1) {
345                     log3("ERROR: classList.size() != 1  : " + classList.size());
346                     testExitCode = FAILED;
347                     continue;
348                 }
349 
350                 log2("      getting: ReferenceType referenceType  = (ReferenceType) classList.get(0);");
351                 ReferenceType referenceType  = (ReferenceType) classList.get(0);
352 
353                 log2("      getting: boolean isStatic = referenceType.isStatic();");
354                 boolean isStatic = referenceType.isStatic();
355 
356                 if ( names1[i1+1].equals("t") ) {
357                     log2("      expected value of isStatic is 'true'");
358                     if (!isStatic) {
359                         log3("ERROR: isStatic != true for: " + typeForCheck);
360                         testExitCode = FAILED;
361                     }
362                 } else {
363                     log2("      expected value of isStatic is 'false'");
364                     if (isStatic) {
365                         log3("ERROR: isStatic != false for: " + typeForCheck);
366                         testExitCode = FAILED;
367                     }
368                 }
369             }
370 
371             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
372         }
373         log1("    TESTING ENDS");
374         return;
375     }

377     private ThreadReference threadByName(String name)
378                  throws JDITestRuntimeException {
379 
380         List         all = vm.allThreads();
381         ListIterator li  = all.listIterator();
382 
383         for (; li.hasNext(); ) {
384             ThreadReference thread = (ThreadReference) li.next();
385             if (thread.name().equals(name))
386                 return thread;
387         }
388         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
389     }

391    /*
392     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
393     *                                             String, String, String)
394     *
395     * It sets up a breakpoint at given line number within a given method in a given class
396     * for a given thread.
397     *
398     * Return value: BreakpointRequest object  in case of success
399     *
400     * JDITestRuntimeException   in case of an Exception thrown within the method
401     */
402 
403     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
404                                                   ReferenceType testedClass,
405                                                   String methodName,
406                                                   String bpLine,
407                                                   String property)
408             throws JDITestRuntimeException {
409 
410         log2("......setting up a breakpoint:");
411         log2("       thread: " + thread + "; class: " + testedClass +
412                         "; method: " + methodName + "; line: " + bpLine);
413 
414         List              alllineLocations = null;
415         Location          lineLocation     = null;
416         BreakpointRequest breakpRequest    = null;
417 
418         try {
419             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
420 
421             alllineLocations = method.allLineLocations();
422 
423             int n =
424                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
425             if (n > alllineLocations.size()) {
426                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
427             } else {
428                 lineLocation = (Location) alllineLocations.get(n);
429                 try {
430                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
431                     breakpRequest.putProperty("number", property);
432                     breakpRequest.addThreadFilter(thread);
433                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
434                 } catch ( Exception e1 ) {
435                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
436                     breakpRequest    = null;
437                 }
438             }
439         } catch ( Exception e2 ) {
440             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
441             breakpRequest    = null;
442         }
443 
444         if (breakpRequest == null) {
445             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
446             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
447         }
448 
449         log2("      a breakpoint has been set up");
450         return breakpRequest;
451     }

454     private void getEventSet()
455                  throws JDITestRuntimeException {
456         try {
457 //            log2("       eventSet = eventQueue.remove(waitTime);");
458             eventSet = eventQueue.remove(waitTime);
459             if (eventSet == null) {
460                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
461             }
462 //            log2("       eventIterator = eventSet.eventIterator;");
463             eventIterator = eventSet.eventIterator();
464         } catch ( Exception e ) {
465             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
466         }
467     }

470     private void breakpointForCommunication()
471                  throws JDITestRuntimeException {
472 
473         log2("breakpointForCommunication");
474         getEventSet();
475 
476         if (eventIterator.nextEvent() instanceof BreakpointEvent)
477             return;
478 
479         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
480     }
