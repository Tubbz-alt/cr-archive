 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createModificationWatchpointRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    - Multiple watchpoints on the same field are permitted.           <BR>
 47  *    - Creates a new disabled watchpoint.                              <BR>
 48  *    - The watchpoint watches modification of the specified field.     <BR>
 49  *    - Throws: UnsupportedOperationException -                         <BR>
 50  *       if the target virtual machine does not support this operation. <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.                      <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following.                 <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares testcase and invokes                         <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the checks required.                         <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : " + e);
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : " + e);
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

248     *               1 - ubnormal end of the test

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298 
299         ThreadReference   mainThread = threadByName("main");
300 
301         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
302                                              debuggeeClass,
303                                             bPointMethod, lineForComm, "zero");
304         bpRequest.enable();
305 
306     //------------------------------------------------------  testing section
307 
308         String fieldName = "testField";
309 
310         Field field = null;
311 
312         List         requests = null;
313         ListIterator li       = null;
314 
315         ModificationWatchpointRequest mwRequest1 = null;
316         ModificationWatchpointRequest mwRequest2 = null;
317 
318 
319         log1("     TESTING BEGINS");
320 
321         for (int i = 0; ; i++) {
322 
323             vm.resume();
324             breakpointForCommunication();
325 
326             int instruction = ((IntegerValue)
327                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
328 
329             if (instruction == 0) {
330                 vm.resume();
331                 break;
332             }
333 
334             log1(":::::: case: # " + i);
335 
336             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
337 
338             field = debuggeeClass.fieldByName(fieldName);
339 
340             if ( vm.canWatchFieldModification() ) {
341                 log2("......vm.canWatchFieldModification == true");
342                 log2("      creating two ModificationWatchpointRequests for 'field': mwRequest1 & mwRequest2");
343 
344                 mwRequest1 = eventRManager.createModificationWatchpointRequest(field);
345                 mwRequest2 = eventRManager.createModificationWatchpointRequest(field);
346 
347                 log2("      checking up on: requests are disabled");
348                 if (mwRequest1.isEnabled()) {
349                     testExitCode = FAILED;
350                     log3("ERROR: mwRequest1 is enabled");
351                 }
352                 if (mwRequest2.isEnabled()) {
353                     testExitCode = FAILED;
354                     log3("ERROR: mwRequest2 is enabled");
355                 }
356                 log2("      compareing field equality");
357                 if ( !mwRequest1.field().equals(field) ) {
358                     testExitCode = FAILED;
359                     log3("ERROR: !mwRequest1.field().equals(field)");
360                 }
361                 if ( !mwRequest2.field().equals(field) ) {
362                     testExitCode = FAILED;
363                     log3("ERROR: !mwRequest2.field().equals(field)");
364                 }
365 
366             } else {
367                 log2("......vm.canWatchFieldModification == false");
368                 log2("      UnsupportedOperationException is expected");
369                 try {
370                     mwRequest1 = eventRManager.createModificationWatchpointRequest(field);
371                     testExitCode = FAILED;
372                     log3("ERROR: NO UnsupportedOperationException");
373                 } catch ( UnsupportedOperationException e ) {
374                 }
375             }
376 
377             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
378         }
379         log1("    TESTING ENDS");
380         return;
381     }

383     private ThreadReference threadByName(String name)
384                  throws JDITestRuntimeException {
385 
386         List         all = vm.allThreads();
387         ListIterator li  = all.listIterator();
388 
389         for (; li.hasNext(); ) {
390             ThreadReference thread = (ThreadReference) li.next();
391             if (thread.name().equals(name))
392                 return thread;
393         }
394         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
395     }

397    /*
398     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
399     *                                             String, String, String)
400     *
401     * It sets up a breakpoint at given line number within a given method in a given class
402     * for a given thread.
403     *
404     * Return value: BreakpointRequest object  in case of success
405     *
406     * JDITestRuntimeException   in case of an Exception thrown within the method
407     */
408 
409     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
410                                                   ReferenceType testedClass,
411                                                   String methodName,
412                                                   String bpLine,
413                                                   String property)
414             throws JDITestRuntimeException {
415 
416         log2("......setting up a breakpoint:");
417         log2("       thread: " + thread + "; class: " + testedClass +
418                         "; method: " + methodName + "; line: " + bpLine);
419 
420         List              alllineLocations = null;
421         Location          lineLocation     = null;
422         BreakpointRequest breakpRequest    = null;
423 
424         try {
425             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
426 
427             alllineLocations = method.allLineLocations();
428 
429             int n =
430                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
431             if (n > alllineLocations.size()) {
432                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
433             } else {
434                 lineLocation = (Location) alllineLocations.get(n);
435                 try {
436                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
437                     breakpRequest.putProperty("number", property);
438                     breakpRequest.addThreadFilter(thread);
439                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
440                 } catch ( Exception e1 ) {
441                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
442                     breakpRequest    = null;
443                 }
444             }
445         } catch ( Exception e2 ) {
446             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
447             breakpRequest    = null;
448         }
449 
450         if (breakpRequest == null) {
451             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
452             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
453         }
454 
455         log2("      a breakpoint has been set up");
456         return breakpRequest;
457     }

460     private void getEventSet()
461                  throws JDITestRuntimeException {
462         try {
463 //            log2("       eventSet = eventQueue.remove(waitTime);");
464             eventSet = eventQueue.remove(waitTime);
465             if (eventSet == null) {
466                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
467             }
468 //            log2("       eventIterator = eventSet.eventIterator;");
469             eventIterator = eventSet.eventIterator();
470         } catch ( Exception e ) {
471             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
472         }
473     }

476     private void breakpointForCommunication()
477                  throws JDITestRuntimeException {
478 
479         log2("breakpointForCommunication");
480         getEventSet();
481 
482         if (eventIterator.nextEvent() instanceof BreakpointEvent)
483             return;
484 
485         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
