 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodEntryRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodEntryRequest.addThreadFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Restricts the events generated by this request            <BR>
 47  *    to those in the given thread.                             <BR>
 48  * The cases to test include three thread's states:             <BR>
 49  * not started, running, dead.                                  <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                              <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new thread, thread1, and invokes              <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 72  *   - creates MethodEntryRequest1 and                                  <BR>
 73  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 74  *   - creates MethodEntryRequest2,                                     <BR>
 75  *     resumes the debuggee to get the thread1 started, and             <BR>
 76  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 77  *   - creates MethodEntryRequest3, waits the thread1 to die, and       <BR>
 78  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 79  * <BR>
 80  * In third phase, at the end of the test,                              <BR>
 81  * the debuggee changes the value of the "instruction"  which           <BR>
 82  * the debugger and debuggee use to inform each other of needed actions,<BR>
 83  * and both end.
 84  * <BR>
 85  */
 86

168     private int runThis (String argv[], PrintStream out) {
169 
170         argsHandler     = new ArgumentHandler(argv);
171         logHandler      = new Log(out, argsHandler);
172         Binder binder   = new Binder(argsHandler, logHandler);
173 
174         waitTime        = argsHandler.getWaitTime() * 60000;
175 
176         try {
177             log2("launching a debuggee :");
178             log2("       " + debuggeeName);
179             if (argsHandler.verbose()) {
180                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
181             } else {
182                 debuggee = binder.bindToDebugee(debuggeeName);
183             }
184             if (debuggee == null) {
185                 log3("ERROR: no debuggee launched");
186                 return FAILED;
187             }
188             log2("debuggee launched");
189         } catch ( Exception e ) {
190             log3("ERROR: Exception : " + e);
191             log2("       test cancelled");
192             return FAILED;
193         }
194 
195         debuggee.redirectOutput(logHandler);
196 
197         vm = debuggee.VM();
198 
199         eventQueue = vm.eventQueue();
200         if (eventQueue == null) {
201             log3("ERROR: eventQueue == null : TEST ABORTED");
202             vm.exit(PASS_BASE);
203             return FAILED;
204         }
205 
206         log2("invocation of the method runTest()");
207         switch (runTest()) {
208 
209             case 0 :  log2("test phase has finished normally");
210                       log2("   waiting for the debuggee to finish ...");
211                       debuggee.waitFor();
212 
213                       log2("......getting the debuggee's exit status");
214                       int status = debuggee.getStatus();
215                       if (status != PASS_BASE) {
216                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
217                               status + " != PASS_BASE");
218                           testExitCode = FAILED;
219                       } else {
220                           log2("......debuggee returned expected exit status: " +
221                               status + " == PASS_BASE");
222                       }
223                       break;
224 
225             default : log3("ERROR: runTest() returned unexpected value");
226 
227             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
228                       log2("......forcing: vm.exit();");
229                       testExitCode = FAILED;
230                       try {
231                           vm.exit(PASS_BASE);
232                       } catch ( Exception e ) {
233                           log3("ERROR: Exception : " + e);
234                       }
235                       break;
236 
237             case 2 :  log3("test cancelled due to VMDisconnectedException");
238                       log2("......trying: vm.process().destroy();");
239                       testExitCode = FAILED;
240                       try {
241                           Process vmProcess = vm.process();
242                           if (vmProcess != null) {
243                               vmProcess.destroy();
244                           }
245                       } catch ( Exception e ) {
246                           log3("ERROR: Exception : " + e);
247                       }
248                       break;
249             }
250 
251         return testExitCode;
252     }

261     private int runTest() {
262 
263         try {
264             testRun();
265 
266             log2("waiting for VMDeathEvent");
267             getEventSet();
268             if (eventIterator.nextEvent() instanceof VMDeathEvent)
269                 return 0;
270 
271             log3("ERROR: last event is not the VMDeathEvent");
272             return 1;
273         } catch ( VMDisconnectedException e ) {
274             log3("ERROR: VMDisconnectedException : " + e);
275             return 2;
276         } catch ( Exception e ) {
277             log3("ERROR: Exception : " + e);
278             return 1;
279         }

283     private void testRun()
284                  throws JDITestRuntimeException, Exception {
285 
286         eventRManager = vm.eventRequestManager();
287 
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291 
292         cpRequest.enable();
293         vm.resume();
294         getEventSet();
295         cpRequest.disable();
296 
297         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
298         debuggeeClass = event.referenceType();
299 
300         if (!debuggeeClass.name().equals(debuggeeName))
301            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
302 
303         log2("      received: ClassPrepareEvent for debuggeeClass");
304 
305         String bPointMethod = "methodForCommunication";
306         String lineForComm  = "lineForComm";
307 
308         ThreadReference   mainThread = threadByName("main");
309 
310         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
311                                              debuggeeClass,
312                                             bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315     //------------------------------------------------------  testing section
316 
317         log1("     TESTING BEGINS");
318 
319         EventRequest eventRequest1 = null;
320         EventRequest eventRequest2 = null;
321         EventRequest eventRequest3 = null;
322 
323         ThreadReference thread1     = null;
324         String          thread1Name = "thread1";
325 
326         String property1 = "MethodEntryRequest1";
327         String property2 = "MethodEntryRequest2";
328         String property3 = "MethodEntryRequest3";
329 
330         ReferenceType testClassReference = null;
331 
332         for (int i = 0; ; i++) {
333 
334             vm.resume();
335             breakpointForCommunication();
336 
337             int instruction = ((IntegerValue)
338                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
339 
340             if (instruction == 0) {
341                 vm.resume();
342                 break;
343             }
344 
345             log1(":::::: case: # " + i);
346 
347             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
348 
349             switch (i) {
350               case 0:
351                      testClassReference =
352                             (ReferenceType) vm.classesByName(testedClassName).get(0);
353                      thread1 = (ThreadReference) debuggeeClass.getValue(
354                                         debuggeeClass.fieldByName(thread1Name));
355 
356                      eventRequest1 = setting2MethodEntryRequest (null,
357                                              testClassReference,
358                                              EventRequest.SUSPEND_NONE, property1);
359 
360                      try {
361                          log2("......eventRequest1.addThreadFilter(thread1);");
362                          log2("        no Exception expected");
363                          ((MethodEntryRequest)eventRequest1).addThreadFilter(thread1);
364                      } catch ( Exception e ) {
365                          log3("ERROR: Exception : " + e);
366                          testExitCode = FAILED;
367                      }
368                      break;
369 
370               case 1:
371                      eventRequest2 = setting2MethodEntryRequest (null,
372                                              testClassReference,
373                                              EventRequest.SUSPEND_NONE, property2);
374                      try {
375                          log2("......eventRequest2.addThreadFilter(thread1);");
376                          log2("        no Exception expected");
377                          ((MethodEntryRequest)eventRequest2).addThreadFilter(thread1);
378                      } catch ( Exception e ) {
379                          log3("ERROR: Exception : " + e);
380                          testExitCode = FAILED;
381                      }
382                      break;
383 
384               case 2:
385                      eventRequest3 = setting2MethodEntryRequest (null,
386                                              testClassReference,
387                                              EventRequest.SUSPEND_NONE, property3);
388                      try {
389                          log2("......eventRequest3.addThreadFilter(thread1);");
390                          log2("        no Exception expected");
391                          ((MethodEntryRequest)eventRequest3).addThreadFilter(thread1);
392                      } catch ( Exception e ) {
393                          log3("ERROR: Exception : " + e);
394                          testExitCode = FAILED;
395                      }
396                      break;
397 
398               default:
399                       throw new JDITestRuntimeException("** default case 2 **");
400             }
401 
402             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
403         }
404         log1("    TESTING ENDS");
405         return;
406     }

408     private ThreadReference threadByName(String name)
409                  throws JDITestRuntimeException {
410 
411         List         all = vm.allThreads();
412         ListIterator li  = all.listIterator();
413 
414         for (; li.hasNext(); ) {
415             ThreadReference thread = (ThreadReference) li.next();
416             if (thread.name().equals(name))
417                 return thread;
418         }
419         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
420     }

422    /*
423     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
424     *                                             String, String, String)
425     *
426     * It sets up a breakpoint at given line number within a given method in a given class
427     * for a given thread.
428     *
429     * Return value: BreakpointRequest object  in case of success
430     *
431     * JDITestRuntimeException   in case of an Exception thrown within the method
432     */
433 
434     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
435                                                   ReferenceType testedClass,
436                                                   String methodName,
437                                                   String bpLine,
438                                                   String property)
439             throws JDITestRuntimeException {
440 
441         log2("......setting up a breakpoint:");
442         log2("       thread: " + thread + "; class: " + testedClass +
443                         "; method: " + methodName + "; line: " + bpLine);
444 
445         List              alllineLocations = null;
446         Location          lineLocation     = null;
447         BreakpointRequest breakpRequest    = null;
448 
449         try {
450             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
451 
452             alllineLocations = method.allLineLocations();
453 
454             int n =
455                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
456             if (n > alllineLocations.size()) {
457                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
458             } else {
459                 lineLocation = (Location) alllineLocations.get(n);
460                 try {
461                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
462                     breakpRequest.putProperty("number", property);
463                     breakpRequest.addThreadFilter(thread);
464                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
465                 } catch ( Exception e1 ) {
466                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
467                     breakpRequest    = null;
468                 }
469             }
470         } catch ( Exception e2 ) {
471             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
472             breakpRequest    = null;
473         }
474 
475         if (breakpRequest == null) {
476             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
477             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
478         }
479 
480         log2("      a breakpoint has been set up");
481         return breakpRequest;
482     }

485     private void getEventSet()
486                  throws JDITestRuntimeException {
487         try {
488 //            log2("       eventSet = eventQueue.remove(waitTime);");
489             eventSet = eventQueue.remove(waitTime);
490             if (eventSet == null) {
491                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
492             }
493 //            log2("       eventIterator = eventSet.eventIterator;");
494             eventIterator = eventSet.eventIterator();
495         } catch ( Exception e ) {
496             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
497         }
498     }

501     private void breakpointForCommunication()
502                  throws JDITestRuntimeException {
503 
504         log2("breakpointForCommunication");
505         getEventSet();
506 
507         if (eventIterator.nextEvent() instanceof BreakpointEvent)
508             return;
509 
510         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
511     }

515     private MethodEntryRequest setting2MethodEntryRequest ( ThreadReference thread,
516                                                             ReferenceType   testedClass,
517                                                             int             suspendPolicy,
518                                                             String          property       )
519             throws JDITestRuntimeException {
520         try {
521             log2("......setting up MethodEntryRequest:");
522             log2("       thread: " + thread + "; class: " + testedClass +  "; property: " + property);
523 
524             MethodEntryRequest
525             menr = eventRManager.createMethodEntryRequest();
526             menr.putProperty("number", property);
527             if (thread != null)
528                 menr.addThreadFilter(thread);
529             menr.addClassFilter(testedClass);
530             menr.setSuspendPolicy(suspendPolicy);
531 
532             log2("      a MethodEntryRequest has been set up");
533             return menr;
534         } catch ( Exception e ) {
535             log3("ERROR: ATTENTION: Exception within settingMethodEntryRequest() : " + e);
536             log3("       MethodEntryRequest HAS NOT BEEN SET UP");
537             throw new JDITestRuntimeException("** FAILURE to set up MethodEntryRequest **");
538         }
