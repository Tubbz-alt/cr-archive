 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.enable()</code>               <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on if a value returned by the method      <BR>
 46  * EventRequest.isEnabled() matchs a value set up with          <BR>
 47  * the method EventRequest.enable().                            <BR>
 48  * The cases to test include all sub-types of EventRequest.     <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following loop.            <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares new check case (as if) and invokes           <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger performs the check case required.                     <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

162     private int runThis (String argv[], PrintStream out) {
163 
164         argsHandler     = new ArgumentHandler(argv);
165         logHandler      = new Log(out, argsHandler);
166         Binder binder   = new Binder(argsHandler, logHandler);
167 
168         waitTime        = argsHandler.getWaitTime() * 60000;
169 
170         try {
171             log2("launching a debuggee :");
172             log2("       " + debuggeeName);
173             if (argsHandler.verbose()) {
174                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
175             } else {
176                 debuggee = binder.bindToDebugee(debuggeeName);
177             }
178             if (debuggee == null) {
179                 log3("ERROR: no debuggee launched");
180                 return FAILED;
181             }
182             log2("debuggee launched");
183         } catch ( Exception e ) {
184             log3("ERROR: Exception : " + e);
185             log2("       test cancelled");
186             return FAILED;
187         }
188 
189         debuggee.redirectOutput(logHandler);
190 
191         vm = debuggee.VM();
192 
193         eventQueue = vm.eventQueue();
194         if (eventQueue == null) {
195             log3("ERROR: eventQueue == null : TEST ABORTED");
196             vm.exit(PASS_BASE);
197             return FAILED;
198         }
199 
200         log2("invocation of the method runTest()");
201         switch (runTest()) {
202 
203             case 0 :  log2("test phase has finished normally");
204                       log2("   waiting for the debuggee to finish ...");
205                       debuggee.waitFor();
206 
207                       log2("......getting the debuggee's exit status");
208                       int status = debuggee.getStatus();
209                       if (status != PASS_BASE) {
210                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
211                               status + " != PASS_BASE");
212                           testExitCode = FAILED;
213                       } else {
214                           log2("......debuggee returned expected exit status: " +
215                               status + " == PASS_BASE");
216                       }
217                       break;
218 
219             default : log3("ERROR: runTest() returned unexpected value");
220 
221             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
222                       log2("......forcing: vm.exit();");
223                       testExitCode = FAILED;
224                       try {
225                           vm.exit(PASS_BASE);
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : " + e);
228                       }
229                       break;
230 
231             case 2 :  log3("test cancelled due to VMDisconnectedException");
232                       log2("......trying: vm.process().destroy();");
233                       testExitCode = FAILED;
234                       try {
235                           Process vmProcess = vm.process();
236                           if (vmProcess != null) {
237                               vmProcess.destroy();
238                           }
239                       } catch ( Exception e ) {
240                           log3("ERROR: Exception : " + e);
241                       }
242                       break;
243             }
244 
245         return testExitCode;
246     }

249    /*
250     * Return value: 0 - normal end of the test
251     *               1 - ubnormal end of the test
252     *               2 - VMDisconnectedException while test phase
253     */
254 
255     private int runTest() {
256 
257         try {
258             testRun();
259 
260             log2("waiting for VMDeathEvent");
261             getEventSet();
262             if (eventIterator.nextEvent() instanceof VMDeathEvent)
263                 return 0;
264 
265             log3("ERROR: last event is not the VMDeathEvent");
266             return 1;
267         } catch ( VMDisconnectedException e ) {
268             log3("ERROR: VMDisconnectedException : " + e);
269             return 2;
270         } catch ( Exception e ) {
271             log3("ERROR: Exception : " + e);
272             return 1;
273         }
274 
275     }

277     private void testRun()
278                  throws JDITestRuntimeException, Exception {
279 
280         eventRManager = vm.eventRequestManager();
281 
282         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
283         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
284         cpRequest.addClassFilter(debuggeeName);
285 
286         cpRequest.enable();
287         vm.resume();
288         getEventSet();
289         cpRequest.disable();
290 
291         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
292         debuggeeClass = event.referenceType();
293 
294         if (!debuggeeClass.name().equals(debuggeeName))
295            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
296 
297         log2("      received: ClassPrepareEvent for debuggeeClass");
298 
299         String bPointMethod = "methodForCommunication";
300         String lineForComm  = "lineForComm";
301 
302         ThreadReference   mainThread = threadByName("main");
303 
304         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
305                                              debuggeeClass,
306                                             bPointMethod, lineForComm, "zero");
307         bpRequest.enable();
308 
309     //------------------------------------------------------  testing section
310 
311 
312         EventRequest  eventRequest1 = null;
313 
314         String        fieldName1    = "var101";
315         String        fieldName2    = "excObj";
316 
317         ThreadReference thread1     = null;
318         String          threadName1 = "thread1";
319 
320         ReferenceType testClassReference = null;
321 
322 
323         log1("     TESTING BEGINS");
324 
325         for (int i = 0; ; i++) {
326 
327             vm.resume();
328             breakpointForCommunication();
329 
330             int instruction = ((IntegerValue)
331                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
332 
333             if (instruction == 0) {
334                 vm.resume();
335                 break;
336             }
337 
338             log1(":::::: case: # " + i);
339 
340             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
341 
342 
343             switch (i) {
344 
345               case 0:
346                      thread1 = threadByName(threadName1);
347 
348                      log2("......setting up StepRequest");
349                      eventRequest1 = eventRManager.createStepRequest
350                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
351                      break;
352 
353               case 1:
354                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
355 
356                      log2("......setting up AccessWatchpointRequest");
357                      eventRequest1 = eventRManager.createAccessWatchpointRequest
358                                     (testClassReference.fieldByName(fieldName1));
359                      break;
360 
361               case 2:
362                      log2(".....setting up ModificationWatchpointRequest");
363                      eventRequest1 = eventRManager.createModificationWatchpointRequest
364                                     (testClassReference.fieldByName(fieldName1));
365                      break;
366 
367               case 3:
368                      log2(".....setting up ClassPrepareRequest");
369                      eventRequest1 = eventRManager.createClassPrepareRequest();
370                      break;
371 
372               case 4:
373                      log2(".....setting up ClassUnloadRequest");
374                      eventRequest1 = eventRManager.createClassUnloadRequest();
375                      break;
376 
377               case 5:
378                      log2(".....setting up MethodEntryRequest");
379                      eventRequest1 = eventRManager.createMethodEntryRequest();
380                      break;
381 
382               case 6:
383                      log2(".....setting up MethodExitRequest");
384                      eventRequest1 = eventRManager.createMethodExitRequest();
385                      break;
386 
387               case 7:
388                      log2(".....setting up ThreadDeathRequest");
389                      eventRequest1 = eventRManager.createThreadDeathRequest();
390                      break;
391 
392               case 8:
393                      log2(".....setting up ThreadStartRequest");
394                      eventRequest1 = eventRManager.createThreadStartRequest();
395                      break;
396 
397               case 9:
398                      log2(".....setting up VMDeathRequest");
399                      eventRequest1 = eventRManager.createVMDeathRequest();
400                      break;
401 
402               case 10:
403                      log2(".....setting up ExceptionRequest");
404                      eventRequest1 = eventRManager.createExceptionRequest(
405                                      (ReferenceType)
406                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
407                                      true, true );
408                      break;
409 
410               case 11:
411                      log2(".....setting up BreakpointRequest");
412                      eventRequest1 = eventRManager.createBreakpointRequest(location);
413                      break;
414 
415 
416               default:
417                       throw new JDITestRuntimeException("** default case 2 **");
418             }
419 
420             log2("......eventRequest1.enable();");
421             eventRequest1.enable();
422             log2("      checking up on eventRequest1");
423             if ( !eventRequest1.isEnabled() ) {
424                 testExitCode = FAILED;
425                 log3("ERROR: EventRequest is not enabled");
426             }
427             eventRequest1.disable();
428 
429             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
430         }
431         log1("    TESTING ENDS");
432         return;
433     }

435     private ThreadReference threadByName(String name)
436                  throws JDITestRuntimeException {
437 
438         List         all = vm.allThreads();
439         ListIterator li  = all.listIterator();
440 
441         for (; li.hasNext(); ) {
442             ThreadReference thread = (ThreadReference) li.next();
443             if (thread.name().equals(name))
444                 return thread;
445         }
446         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
447     }

449    /*
450     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
451     *                                             String, String, String)
452     *
453     * It sets up a breakpoint at given line number within a given method in a given class
454     * for a given thread.
455     *
456     * Return value: BreakpointRequest object  in case of success
457     *
458     * JDITestRuntimeException   in case of an Exception thrown within the method
459     */
460 
461     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
462                                                   ReferenceType testedClass,
463                                                   String methodName,
464                                                   String bpLine,
465                                                   String property)
466             throws JDITestRuntimeException {
467 
468         log2("......setting up a breakpoint:");
469         log2("       thread: " + thread + "; class: " + testedClass +
470                         "; method: " + methodName + "; line: " + bpLine);
471 
472         List              alllineLocations = null;
473         Location          lineLocation     = null;
474         BreakpointRequest breakpRequest    = null;
475 
476         try {
477             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
478 
479             alllineLocations = method.allLineLocations();
480 
481             int n =
482                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
483             if (n > alllineLocations.size()) {
484                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
485             } else {
486                 lineLocation = (Location) alllineLocations.get(n);
487 
488 // this is only for this test to get Location object
489 location = lineLocation;
490 
491                 try {
492                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
493                     breakpRequest.putProperty("number", property);
494                     breakpRequest.addThreadFilter(thread);
495                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
496                 } catch ( Exception e1 ) {
497                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
498                     breakpRequest    = null;
499                 }
500             }
501         } catch ( Exception e2 ) {
502             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
503             breakpRequest    = null;
504         }
505 
506         if (breakpRequest == null) {
507             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
508             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
509         }
510 
511         log2("      a breakpoint has been set up");
512         return breakpRequest;
513     }

516     private void getEventSet()
517                  throws JDITestRuntimeException {
518         try {
519 //            log2("       eventSet = eventQueue.remove(waitTime);");
520             eventSet = eventQueue.remove(waitTime);
521             if (eventSet == null) {
522                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
523             }
524 //            log2("       eventIterator = eventSet.eventIterator;");
525             eventIterator = eventSet.eventIterator();
526         } catch ( Exception e ) {
527             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
528         }
529     }

532     private void breakpointForCommunication()
533                  throws JDITestRuntimeException {
534 
535         log2("breakpointForCommunication");
536         getEventSet();
537 
538         if (eventIterator.nextEvent() instanceof BreakpointEvent)
539             return;
540 
541         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
