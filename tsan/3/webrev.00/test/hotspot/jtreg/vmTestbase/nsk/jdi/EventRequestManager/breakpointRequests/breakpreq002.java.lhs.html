 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.breakpointRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled breakpoint requests.          <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = threadByName("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         List         requests = null;
310         ListIterator li       = null;
311 
312         BreakpointRequest request = null;
313         BreakpointRequest bpRequests[] = { null, null, null, null, null,
314                                            null, null, null, null, null };
315         int listSize;
316         int flag;
317 
318 
319         log1("     TESTING BEGINS");
320 
321         for (int i = 0; ; i++) {
322 
323             vm.resume();
324             breakpointForCommunication();
325 
326             int instruction = ((IntegerValue)
327                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
328 
329             if (instruction == 0) {
330                 vm.resume();
331                 break;
332             }
333 
334             log1(":::::: case: # " + i);
335 
336             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
337 
338             log2("......creating BreakpointRequests");
339             for (int i1 = 0; i1 < bpRequests.length; i1++) {
340                 bpRequests[i1] = eventRManager.createBreakpointRequest(breakpLocation);
341                 bpRequests[i1].putProperty("number", String.valueOf(i1));
342 
343                 log2("......checking up on returned List after creating new BreakpointRequest");
344                 requests = eventRManager.breakpointRequests();
345                 listSize = requests.size();
346                 if ( listSize != (i1 + 2) ) {
347                     testExitCode = FAILED;
348                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 2));
349                 }
350                 flag = 0;
351                 li = requests.listIterator();
352                 while (li.hasNext()) {
353                     request = (BreakpointRequest) li.next();
354                     if ( request.getProperty("number").equals("zero") )
355                         continue;
356                     if ( !request.isEnabled() ) {
357                         flag++;
358                         if (flag > 1) {
359                             testExitCode = FAILED;
360                             log3("ERROR: # of disabled requests > 1 : " + flag);
361                         }
362                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
363                             testExitCode = FAILED;
364                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
365                         }
366                     } else {
367                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
368                             testExitCode = FAILED;
369                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
370                         }
371                     }
372                 }
373 
374                 log2("      enabling created BreakpointRequest");
375                 bpRequests[i1].enable();
376                 requests = eventRManager.breakpointRequests();
377                 listSize = requests.size();
378                 if ( listSize != (i1 + 2) ) {
379                     testExitCode = FAILED;
380                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 2));
381                 }
382 
383                 li = requests.listIterator();
384                 while (li.hasNext()) {
385                     request = (BreakpointRequest) li.next();
386                     if ( !request.isEnabled() ) {
387                         testExitCode = FAILED;
388                         log3("ERROR: returned List contains disabled request : " + request);
389                     }
390                 }
391 
392                 log2("      removing item from the List; UnsupportedOperationException is expected");
393                 try {
394                     requests.remove(i1);
395                     testExitCode = FAILED;
396                     log3("ERROR: NO exception");
397                 } catch ( UnsupportedOperationException e ) {
398                     log2("        UnsupportedOperationException ");
399                 }
400             }
401 
402             log2("......deleting BreakpointRequests");
403             for (int i2 = bpRequests.length -1; i2 >= 0; i2--) {
404                 eventRManager.deleteEventRequest(bpRequests[i2]);
405                 requests = eventRManager.breakpointRequests();
406                 listSize = requests.size();
407                 if ( listSize != (i2 + 1) ) {
408                     testExitCode = FAILED;
409                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i2 + 1));
410                 }
411                 log2("      removing item from the List; UnsupportedOperationException is expected");
412                 try {
413                     requests.remove(i2);
414                     testExitCode = FAILED;
415                     log3("ERROR: NO exception");
416                 } catch ( UnsupportedOperationException e ) {
417                     log2("        UnsupportedOperationException ");
418                 }
419             }
420 
421             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
422         }
423         log1("    TESTING ENDS");
424         return;
425     }

427     private ThreadReference threadByName(String name)
428                  throws JDITestRuntimeException {
429 
430         List         all = vm.allThreads();
431         ListIterator li  = all.listIterator();
432 
433         for (; li.hasNext(); ) {
434             ThreadReference thread = (ThreadReference) li.next();
435             if (thread.name().equals(name))
436                 return thread;
437         }
438         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
439     }

441    /*
442     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
443     *                                             String, String, String)
444     *
445     * It sets up a breakpoint at given line number within a given method in a given class
446     * for a given thread.
447     *
448     * Return value: BreakpointRequest object  in case of success
449     *
450     * JDITestRuntimeException   in case of an Exception thrown within the method
451     */
452 
453     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
454                                                   ReferenceType testedClass,
455                                                   String methodName,
456                                                   String bpLine,
457                                                   String property)
458             throws JDITestRuntimeException {
459 
460         log2("......setting up a breakpoint:");
461         log2("       thread: " + thread + "; class: " + testedClass +
462                         "; method: " + methodName + "; line: " + bpLine);
463 
464         List              alllineLocations = null;
465         Location          lineLocation     = null;
466         BreakpointRequest breakpRequest    = null;
467 
468         try {
469             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
470 
471             alllineLocations = method.allLineLocations();
472 
473             int n =
474                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
475             if (n > alllineLocations.size()) {
476                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
477             } else {
478                 lineLocation = (Location) alllineLocations.get(n);
479 breakpLocation = lineLocation;
480                 try {
481                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
482                     breakpRequest.putProperty("number", property);
483                     breakpRequest.addThreadFilter(thread);
484                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
485                 } catch ( Exception e1 ) {
486                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
487                     breakpRequest    = null;
488                 }
489             }
490         } catch ( Exception e2 ) {
491             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
492             breakpRequest    = null;
493         }
494 
495         if (breakpRequest == null) {
496             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
497             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
498         }
499 
500         log2("      a breakpoint has been set up");
501         return breakpRequest;
502     }

505     private void getEventSet()
506                  throws JDITestRuntimeException {
507         try {
508 //            log2("       eventSet = eventQueue.remove(waitTime);");
509             eventSet = eventQueue.remove(waitTime);
510             if (eventSet == null) {
511                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
512             }
513 //            log2("       eventIterator = eventSet.eventIterator;");
514             eventIterator = eventSet.eventIterator();
515         } catch ( Exception e ) {
516             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
517         }
518     }
