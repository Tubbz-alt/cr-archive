 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.notifyCaught()</code>     <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a boolean value, an argument of the method        <BR>
 46  *    EventRequestManager.createExceptionRequest(),                     <BR>
 47  * is equal to one returned by the method                               <BR>
 48  *    ExceptionRequest.notifyCaught()                                   <BR>
 49  * The cases to check include both true and false.                      <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger                                                       <BR>
 68  *   - sets up two ExceptionRequests with true and false values of      <BR>
 69  *     the notifyCaught argument,                                       <BR>
 70  *   - gets the values of the Request with the method                   <BR>
 71  *     StepRequest.notifyCaught(), and                                  <BR>
 72  *   - compares the values.                                             <BR>
 73  * <BR>
 74  * In third phase, at the end of the test, the debuggee changes         <BR>
 75  * the value of the "instruction" which the debugger and debuggee       <BR>
 76  * use to inform each other of needed actions,  and both end.           <BR>
 77  * <BR>
 78  */
 79

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : " + e);
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : " + e);
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if (eventIterator.nextEvent() instanceof VMDeathEvent)
259                 return 0;
260 
261             log3("ERROR: last event is not the VMDeathEvent");
262             return 1;
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         eventRManager = vm.eventRequestManager();
277 
278         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
279         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
280         cpRequest.addClassFilter(debuggeeName);
281 
282         cpRequest.enable();
283         vm.resume();
284         getEventSet();
285         cpRequest.disable();
286 
287         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
288         debuggeeClass = event.referenceType();
289 
290         if (!debuggeeClass.name().equals(debuggeeName))
291            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
292 
293         log2("      received: ClassPrepareEvent for debuggeeClass");
294 
295         String bPointMethod = "methodForCommunication";
296         String lineForComm  = "lineForComm";
297 
298         ThreadReference   mainThread = threadByName("main");
299 
300         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
301                                              debuggeeClass,
302                                             bPointMethod, lineForComm, "zero");
303         bpRequest.enable();
304 
305     //------------------------------------------------------  testing section
306 
307         log1("     TESTING BEGINS");
308 
309         EventRequest  eventRequest1 = null;
310         EventRequest  eventRequest2 = null;
311 
312         boolean bool;
313 
314 
315         for (int i = 0; ; i++) {
316 
317             vm.resume();
318 
319             log1(":::::: case: # " + i);
320 
321             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
322 
323             switch (i) {
324 
325               case 0:
326 
327 
328                      log2("......setting up: eventRequest1 = setting24ExceptionRequest(true,  true);");
329                      eventRequest1 = setting24ExceptionRequest(true,  true);
330 
331                      log2("......getting: bool = ((ExceptionRequest) eventRequest1).notifyCaught();");
332                      bool = ((ExceptionRequest) eventRequest1).notifyCaught();
333 
334                      log2("      compareing boolean values");
335                      if (bool != true) {
336                          testExitCode = FAILED;
337                          log3("ERROR: values are not equal : true and false");
338                      }
339 
340                      log2("......setting up: eventRequest2 = setting24ExceptionRequest(false,  true);");
341                      eventRequest2 = setting24ExceptionRequest(false,  true);
342 
343                      log2("......getting: bool = ((ExceptionRequest) eventRequest2).notifyCaught();");
344                      bool = ((ExceptionRequest) eventRequest2).notifyCaught();
345 
346                      log2("      compareing boolean values");
347                      if (bool != false) {
348                          testExitCode = FAILED;
349                          log3("ERROR: values are not equal : true and false");
350                      }
351 
352                       break;
353 
354               default:
355                       throw new JDITestRuntimeException("** default case 2 **");
356             }
357 
358             breakpointForCommunication();
359 
360             int instruction = ((IntegerValue)
361                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
362 
363             if (instruction == 0) {
364                 vm.resume();
365                 break;
366             }
367 
368             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
369         }
370         log1("    TESTING ENDS");
371         return;
372     }

374     private ThreadReference threadByName(String name)
375                  throws JDITestRuntimeException {
376 
377         List         all = vm.allThreads();
378         ListIterator li  = all.listIterator();
379 
380         for (; li.hasNext(); ) {
381             ThreadReference thread = (ThreadReference) li.next();
382             if (thread.name().equals(name))
383                 return thread;
384         }
385         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
386     }

388    /*
389     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
390     *                                             String, String, String)
391     *
392     * It sets up a breakpoint at given line number within a given method in a given class
393     * for a given thread.
394     *
395     * Return value: BreakpointRequest object  in case of success
396     *
397     * JDITestRuntimeException   in case of an Exception thrown within the method
398     */
399 
400     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
401                                                   ReferenceType testedClass,
402                                                   String methodName,
403                                                   String bpLine,
404                                                   String property)
405             throws JDITestRuntimeException {
406 
407         log2("......setting up a breakpoint:");
408         log2("       thread: " + thread + "; class: " + testedClass +
409                         "; method: " + methodName + "; line: " + bpLine);
410 
411         List              alllineLocations = null;
412         Location          lineLocation     = null;
413         BreakpointRequest breakpRequest    = null;
414 
415         try {
416             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
417 
418             alllineLocations = method.allLineLocations();
419 
420             int n =
421                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
422             if (n > alllineLocations.size()) {
423                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
424             } else {
425                 lineLocation = (Location) alllineLocations.get(n);
426                 try {
427                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
428                     breakpRequest.putProperty("number", property);
429                     breakpRequest.addThreadFilter(thread);
430                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
431                 } catch ( Exception e1 ) {
432                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
433                     breakpRequest    = null;
434                 }
435             }
436         } catch ( Exception e2 ) {
437             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
438             breakpRequest    = null;
439         }
440 
441         if (breakpRequest == null) {
442             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
443             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
444         }
445 
446         log2("      a breakpoint has been set up");
447         return breakpRequest;
448     }

451     private void getEventSet()
452                  throws JDITestRuntimeException {
453         try {
454 //            log2("       eventSet = eventQueue.remove(waitTime);");
455             eventSet = eventQueue.remove(waitTime);
456             if (eventSet == null) {
457                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
458             }
459 //            log2("       eventIterator = eventSet.eventIterator;");
460             eventIterator = eventSet.eventIterator();
461         } catch ( Exception e ) {
462             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
463         }
464     }

467     private void breakpointForCommunication()
468                  throws JDITestRuntimeException {
469 
470         log2("breakpointForCommunication");
471         getEventSet();
472 
473         if (eventIterator.nextEvent() instanceof BreakpointEvent)
474             return;
475 
476         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
477     }
