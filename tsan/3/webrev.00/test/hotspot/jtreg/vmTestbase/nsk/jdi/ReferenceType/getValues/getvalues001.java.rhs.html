 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.getValues()</code>           <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Case for testing includes fields mirroring primitive type values     <BR>
 46  * declared in this type, a superclass, a superinterface, and           <BR>
 47  * an implemented interface.                                            <BR>
 48  * The test checks that Map returned by the method,                     <BR>
 49  * contains all the tested variables.                                   <BR>
 50  * <BR>
 51  * The test works as follows.                                           <BR>
 52  * Upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request                  <BR>
 56  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 57  * resumes the VM, and waits for the event within the predefined        <BR>
 58  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 61  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 62  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 63  * to be suspended and to inform the debugger with the event.           <BR>
 64  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 65  * At the end, the debuggee changes the value of the "instruction"      <BR>
 66  * to inform the debugger of checks finished, and both end.             <BR>
 67  */

143     private int runThis (String argv[], PrintStream out) {
144 
145         argsHandler     = new ArgumentHandler(argv);
146         logHandler      = new Log(out, argsHandler);
147         Binder binder   = new Binder(argsHandler, logHandler);
148 
149         waitTime        = argsHandler.getWaitTime() * 60000;
150 
151         try {
152             log2("launching a debuggee :");
153             log2("       " + debuggeeName);
154             if (argsHandler.verbose()) {
155                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
156             } else {
157                 debuggee = binder.bindToDebugee(debuggeeName);
158             }
159             if (debuggee == null) {
160                 log3("ERROR: no debuggee launched");
161                 return FAILED;
162             }
163             log2("debuggee launched");
164         } catch ( Exception e ) {
165             log3("ERROR: Exception : " + e);
166             log2("       test cancelled");
167             return FAILED;
168         }
169 
170         debuggee.redirectOutput(logHandler);
171 
172         vm = debuggee.VM();
173 
174         eventQueue = vm.eventQueue();
175         if (eventQueue == null) {
176             log3("ERROR: eventQueue == null : TEST ABORTED");
177             vm.exit(PASS_BASE);
178             return FAILED;
179         }
180 
181         log2("invocation of the method runTest()");
182         switch (runTest()) {
183 
184             case 0 :  log2("test phase has finished normally");
185                       log2("   waiting for the debuggee to finish ...");
186                       debuggee.waitFor();
187 
188                       log2("......getting the debuggee's exit status");
189                       int status = debuggee.getStatus();
190                       if (status != PASS_BASE) {
191                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
192                               status + " != PASS_BASE");
193                           testExitCode = FAILED;
194                       } else {
195                           log2("......debuggee returned expected exit status: " +
196                               status + " == PASS_BASE");
197                       }
198                       break;
199 
200             default : log3("ERROR: runTest() returned unexpected value");
201 
202             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
203                       log2("......forcing: vm.exit();");
204                       testExitCode = FAILED;
205                       try {
206                           vm.exit(PASS_BASE);
207                       } catch ( Exception e ) {
208                           log3("ERROR: Exception : e");
209                       }
210                       break;
211 
212             case 2 :  log3("test cancelled due to VMDisconnectedException");
213                       log2("......trying: vm.process().destroy();");
214                       testExitCode = FAILED;
215                       try {
216                           Process vmProcess = vm.process();
217                           if (vmProcess != null) {
218                               vmProcess.destroy();
219                           }
220                       } catch ( Exception e ) {
221                           log3("ERROR: Exception : e");
222                       }
223                       break;
224             }
225 
226         return testExitCode;
227     }

230    /*
231     * Return value: 0 - normal end of the test
232     *               1 - ubnormal end of the test
233     *               2 - VMDisconnectedException while test phase
234     */

236     private int runTest() {
237 
238         try {
239             testRun();
240 
241             log2("waiting for VMDeathEvent");
242             getEventSet();
243             if (eventIterator.nextEvent() instanceof VMDeathEvent)
244                 return 0;
245 
246             log3("ERROR: last event is not the VMDeathEvent");
247             return 1;
248         } catch ( VMDisconnectedException e ) {
249             log3("ERROR: VMDisconnectedException : " + e);
250             return 2;
251         } catch ( Exception e ) {
252             log3("ERROR: Exception : " + e);
253             return 1;
254         }
255 
256     }

258     private void testRun()
259                  throws JDITestRuntimeException, Exception {
260 
261         eventRManager = vm.eventRequestManager();
262 
263         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
264         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
265         cpRequest.addClassFilter(debuggeeName);
266 
267         cpRequest.enable();
268         vm.resume();
269         getEventSet();
270         cpRequest.disable();
271 
272         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
273         debuggeeClass = event.referenceType();
274 
275         if (!debuggeeClass.name().equals(debuggeeName))
276            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
277 
278         log2("      received: ClassPrepareEvent for debuggeeClass");
279 
280         String bPointMethod = "methodForCommunication";
281         String lineForComm  = "lineForComm";
282         BreakpointRequest bpRequest;
283 
284         try {
285             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
286                                           debuggeeClass,
287                                           bPointMethod, lineForComm, "zero");
288         } catch ( Exception e ) {
289             throw e;
290         }
291         bpRequest.enable();
292 
293     //------------------------------------------------------  testing section
294 
295         log1("     TESTING BEGINS");
296 
297         for (int i = 0; ; i++) {
298 
299             vm.resume();
300             breakpointForCommunication();
301 
302             int instruction = ((IntegerValue)
303                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
304 
305             if (instruction == 0) {
306                 vm.resume();
307                 break;
308             }
309 
310             log1("  new check: # " + i);
311 
312             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
313 
314             log2("      name of tested class : " + testedClassName);
315             log2("      getting: List classes = vm.classByName(testedClassName);");
316             List classes = vm.classesByName(testedClassName);
317             if (classes.size() != 1) {
318                 log3("ERROR: classes.size() != 1 : " + classes.size());
319                 testExitCode = FAILED;
320                 break;
321             }
322             log2("      getting: ReferenceType testedClass = (ReferenceType) classes.get(0);");
323             ReferenceType testedClass = (ReferenceType) classes.get(0);
324 
325             log2("------case for testing: primitive values");
326 
327             String fieldNames[] = {
328                        "bl0",
329                        "bt0",
330                        "ch0",
331                        "db0",
332                        "fl0",
333                        "in0",
334                        "ln0",
335                        "sh0"
336             };
337 
338             Field [] fArray = {
339                 testedClass.fieldByName(fieldNames[0]),
340                 testedClass.fieldByName(fieldNames[1]),
341                 testedClass.fieldByName(fieldNames[2]),
342                 testedClass.fieldByName(fieldNames[3]),
343                 testedClass.fieldByName(fieldNames[4]),
344                 testedClass.fieldByName(fieldNames[5]),
345                 testedClass.fieldByName(fieldNames[6]),
346                 testedClass.fieldByName(fieldNames[7]),
347             };
348 
349             log2("      getting: List fields = testedClass.visibleFields();");
350             List<Field> fields = testedClass.visibleFields();
351 
352             log2("      getting: Map valuesMap = testedClass.getValues(fields);");
353             Map<Field, Value> valuesMap        = testedClass.getValues(fields);
354             Set<Field> keySet           = valuesMap.keySet();
355             Iterator<Field> setIterator = keySet.iterator();
356 
357             int flag        = 0;
358             int controlFlag = 0xFF;
359 
360             Field f1 = null;
361 
362             log2("      loop of testing the Map");
363 
364             for (int ifor = 0; setIterator.hasNext(); ifor++) {
365 
366                 f1 = setIterator.next();
367 
368                 int flag1 = 0;
369                 for (int ifor1 = 0; ifor1 < fieldNames.length; ifor1++) {
370 
371                     if (f1.equals(fArray[ifor1])) {
372                         if (flag1 == 0) {
373                             flag1++;
374                             flag |= 1 << ifor1;
375                             break;
376                         } else {
377                             log3("ERROR: duplication in the Map; ifor = " + ifor + "  ifor1 = " + ifor1);
378                             testExitCode = FAILED;
379                         }
380                     }
381                 }
382             }
383             if (flag != controlFlag) {
384                 log3("ERROR: returned Map object doesn't contain all tested fields");
385                 testExitCode = FAILED;
386             }
387             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
388         }
389         log1("    TESTING ENDS");
390         return;
391     }

393    /*
394     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
395     *                                             String, String, String)
396     *
397     * It sets up a breakpoint at given line number within a given method in a given class
398     * for a given thread.
399     *
400     * Return value: BreakpointRequest object  in case of success
401     *
402     * JDITestRuntimeException   in case of an Exception thrown within the method
403     */
404 
405     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
406                                                   ReferenceType testedClass,
407                                                   String methodName,
408                                                   String bpLine,
409                                                   String property)
410             throws JDITestRuntimeException {
411 
412         log2("......setting up a breakpoint:");
413         log2("       thread: " + thread + "; class: " + testedClass +
414                         "; method: " + methodName + "; line: " + bpLine);
415 
416         List              alllineLocations = null;
417         Location          lineLocation     = null;
418         BreakpointRequest breakpRequest    = null;
419 
420         try {
421             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
422 
423             alllineLocations = method.allLineLocations();
424 
425             int n =
426                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
427             if (n > alllineLocations.size()) {
428                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
429             } else {
430                 lineLocation = (Location) alllineLocations.get(n);
431                 try {
432                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
433                     breakpRequest.putProperty("number", property);
434                     breakpRequest.addThreadFilter(thread);
435                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
436                 } catch ( Exception e1 ) {
437                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
438                     breakpRequest    = null;
439                 }
440             }
441         } catch ( Exception e2 ) {
442             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
443             breakpRequest    = null;
444         }
445 
446         if (breakpRequest == null) {
447             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
448             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
449         }
450 
451         log2("      a breakpoint has been set up");
452         return breakpRequest;
453     }

456     private void getEventSet()
457                  throws JDITestRuntimeException {
458         try {
459 //            log2("       eventSet = eventQueue.remove(waitTime);");
460             eventSet = eventQueue.remove(waitTime);
461             if (eventSet == null) {
462                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
463             }
464 //            log2("       eventIterator = eventSet.eventIterator;");
465             eventIterator = eventSet.eventIterator();
466         } catch ( Exception e ) {
467             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
468         }
469     }

472     private void breakpointForCommunication()
473                  throws JDITestRuntimeException {
474 
475         log2("breakpointForCommunication");
476         getEventSet();
477 
478         if (eventIterator.nextEvent() instanceof BreakpointEvent)
479             return;
480 
481         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
482     }
