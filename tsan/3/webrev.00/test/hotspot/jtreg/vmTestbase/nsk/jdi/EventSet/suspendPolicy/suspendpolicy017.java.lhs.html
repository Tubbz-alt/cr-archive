 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a VMDeathEvent set,                 <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event set containing three more   <BR>
 49  * VMDeathRequests with suspendPolicies                         <BR>
 50  *      SUSPEND_ALL, SUSPEND_NONE, and SUSPEND_EVENT_THREAD.    <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the above,                                  <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger sets up VMDeathRequests, resumes                      <BR>
 69  *   the debuggee, and waits for the VMDeathEvent.                      <BR>
 70  * - The debuggee ends to be resulting in the event.                    <BR>
 71  * - Upon getting new event, the debugger                               <BR>
 72  *   performs the check corresponding to the event.                     <BR>
 73  * <BR>
 74  * Note. To inform each other of needed actions, the debugger and       <BR>
 75  *       and the debuggee use debuggee's variable "instruction".        <BR>
 76  * In third phase, at the end,                                          <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */
 81

165     private int runThis (String argv[], PrintStream out) {
166 
167         argsHandler     = new ArgumentHandler(argv);
168         logHandler      = new Log(out, argsHandler);
169         Binder binder   = new Binder(argsHandler, logHandler);
170 
171         waitTime        = argsHandler.getWaitTime() * 60000;
172 
173         try {
174             log2("launching a debuggee :");
175             log2("       " + debuggeeName);
176             if (argsHandler.verbose()) {
177                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
178             } else {
179                 debuggee = binder.bindToDebugee(debuggeeName);
180             }
181             if (debuggee == null) {
182                 log3("ERROR: no debuggee launched");
183                 return FAILED;
184             }
185             log2("debuggee launched");
186         } catch ( Exception e ) {
187             log3("ERROR: Exception : " + e);
188             log2("       test cancelled");
189             return FAILED;
190         }
191 
192         debuggee.redirectOutput(logHandler);
193 
194         vm = debuggee.VM();
195 
196         eventQueue = vm.eventQueue();
197         if (eventQueue == null) {
198             log3("ERROR: eventQueue == null : TEST ABORTED");
199             vm.exit(PASS_BASE);
200             return FAILED;
201         }
202 
203         log2("invocation of the method runTest()");
204         switch (runTest()) {
205 
206             case 0 :  log2("test phase has finished normally");
207                       log2("   waiting for the debuggee to finish ...");
208                       debuggee.waitFor();
209 
210                       log2("......getting the debuggee's exit status");
211                       int status = debuggee.getStatus();
212                       if (status != PASS_BASE) {
213                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
214                               status + " != PASS_BASE");
215                           testExitCode = FAILED;
216                       } else {
217                           log2("......debuggee returned expected exit status: " +
218                               status + " == PASS_BASE");
219                       }
220                       break;
221 
222             default : log3("ERROR: runTest() returned unexpected value");
223 
224             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
225                       log2("......forcing: vm.exit();");
226                       testExitCode = FAILED;
227                       try {
228                           vm.exit(PASS_BASE);
229                       } catch ( Exception e ) {
230                           log3("ERROR: Exception : e");
231                       }
232                       break;
233 
234             case 2 :  log3("test cancelled due to VMDisconnectedException");
235                       log2("......trying: vm.process().destroy();");
236                       testExitCode = FAILED;
237                       try {
238                           Process vmProcess = vm.process();
239                           if (vmProcess != null) {
240                               vmProcess.destroy();
241                           }
242                       } catch ( Exception e ) {
243                           log3("ERROR: Exception : e");
244                       }
245                       break;
246             }
247 
248         return testExitCode;
249     }

253     * Return value: 0 - normal end of the test
254     *               1 - ubnormal end of the test
255     *               2 - VMDisconnectedException while test phase

258     private int runTest() {
259 
260         try {
261             testRun();
262 
263             log2("waiting for VMDeathEvent");
264             getEventSet();
265             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
266                 log3("ERROR: last event is not the VMDeathEvent");
267                 return 1;
268             }
269             check();
270 
271             log2("waiting for VMDisconnectEvent");
272             getEventSet();
273             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
274                 log3("ERROR: last event is not the VMDisconnectEvent");
275                 return 1;
276             }
277 
278             return 0;
279 
280         } catch ( VMDisconnectedException e ) {
281             log3("ERROR: VMDisconnectedException : " + e);
282             return 2;
283         } catch ( Exception e ) {
284             log3("ERROR: Exception : " + e);
285             return 1;
286         }
287 
288     }

290     private void testRun()
291                  throws JDITestRuntimeException, Exception {
292 
293         if ( !vm.canRequestVMDeathEvent() ) {
294             log2("......vm.canRequestVMDeathEvent == false :: test cancelled");
295             vm.exit(PASS_BASE);
296             return;
297         }
298 
299 
300         eventRManager = vm.eventRequestManager();
301 
302         log2("......getting ClassPrepareEvent for debuggee's class");
303         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
304         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
305         cpRequest.addClassFilter(debuggeeName);
306         cpRequest.enable();
307         vm.resume();
308         getEventSet();
309         cpRequest.disable();
310 
311         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
312         debuggeeClass = event.referenceType();
313 
314         if (!debuggeeClass.name().equals(debuggeeName))
315            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
316         log2("      received: ClassPrepareEvent for debuggeeClass");
317 
318         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
319 
320         String            bPointMethod = "methodForCommunication";
321         String            lineForComm  = "lineForComm";
322         BreakpointRequest bpRequest;
323         ThreadReference   mainThread = threadByName("main");
324         bpRequest = settingBreakpoint(mainThread,
325                                       debuggeeClass,
326                                       bPointMethod, lineForComm, "zero");
327         bpRequest.enable();
328 
329         vm.resume();
330 
331     //------------------------------------------------------  testing section
332 
333         log1("     TESTING BEGINS");
334 
335         EventRequest eventRequest1 = null;
336         EventRequest eventRequest2 = null;
337         EventRequest eventRequest3 = null;
338 
339         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
340         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
341         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
342         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
343 
344         int policyExpected[] = {    SUSPEND_NONE,
345                                     SUSPEND_THREAD,
346                                     SUSPEND_ALL,
347                                     SUSPEND_THREAD,
348                                     SUSPEND_ALL,
349                                     SUSPEND_ALL,
350                                     SUSPEND_ALL      };
351         int policy = 0;
352 
353         breakpointForCommunication();
354 
355             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
356 
357         eventRequest1 = settingVMDeathRequest (SUSPEND_NONE, "VMDeathRequest1");
358         eventRequest1.enable();
359         eventRequest2 = settingVMDeathRequest (SUSPEND_THREAD, "VMDeathRequest2");
360         eventRequest2.enable();
361         eventRequest3 = settingVMDeathRequest (SUSPEND_ALL, "VMDeathRequest3");
362         eventRequest3.enable();
363 
364         policyToCheck = policyExpected[6];
365         mainThread.resume();
366 
367             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
368 
369         log1("    TESTING ENDS");
370         return;
371     }

373     private ThreadReference threadByName(String name)
374                  throws JDITestRuntimeException {
375 
376         List         all = vm.allThreads();
377         ListIterator li  = all.listIterator();
378 
379         for (; li.hasNext(); ) {
380             ThreadReference thread = (ThreadReference) li.next();
381             if (thread.name().equals(name))
382                 return thread;
383         }
384         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
385     }

387    /*
388     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
389     *                                             String, String, String)
390     *
391     * It sets up a breakpoint at given line number within a given method in a given class
392     * for a given thread.
393     *
394     * Return value: BreakpointRequest object  in case of success
395     *
396     * JDITestRuntimeException   in case of an Exception thrown within the method
397     */
398 
399     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
400                                                   ReferenceType testedClass,
401                                                   String methodName,
402                                                   String bpLine,
403                                                   String property)
404             throws JDITestRuntimeException {
405 
406         log2("......setting up a breakpoint:");
407         log2("       thread: " + thread + "; class: " + testedClass +
408                         "; method: " + methodName + "; line: " + bpLine);
409 
410         List              alllineLocations = null;
411         Location          lineLocation     = null;
412         BreakpointRequest breakpRequest    = null;
413 
414         try {
415             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
416 
417             alllineLocations = method.allLineLocations();
418 
419             int n =
420                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
421             if (n > alllineLocations.size()) {
422                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
423             } else {
424                 lineLocation = (Location) alllineLocations.get(n);
425                 try {
426                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
427                     breakpRequest.putProperty("number", property);
428                     breakpRequest.addThreadFilter(thread);
429                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
430                 } catch ( Exception e1 ) {
431                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
432                     breakpRequest    = null;
433                 }
434             }
435         } catch ( Exception e2 ) {
436             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
437             breakpRequest    = null;
438         }
439 
440         if (breakpRequest == null) {
441             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
442             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
443         }
444 
445         log2("      a breakpoint has been set up");
446         return breakpRequest;
447     }

450     private void getEventSet()
451                  throws JDITestRuntimeException {
452         try {
453 //            log2("       eventSet = eventQueue.remove(waitTime);");
454             eventSet = eventQueue.remove(waitTime);
455             if (eventSet == null) {
456                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
457             }
458 //            log2("       eventIterator = eventSet.eventIterator;");
459             eventIterator = eventSet.eventIterator();
460         } catch ( Exception e ) {
461             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
462         }
463     }

466     private void breakpointForCommunication()
467                  throws JDITestRuntimeException {
468 
469         log2("breakpointForCommunication");
470         getEventSet();
471 
472         if (eventIterator.nextEvent() instanceof BreakpointEvent)
473             return;
474 
475         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
476     }

480     private VMDeathRequest settingVMDeathRequest( int    suspendPolicy,
481                                                   String property       )
482             throws JDITestRuntimeException {
483         try {
484             log2("......setting up VMDeathRequest:");
485             log2("      suspendPolicy: " + suspendPolicy + "; property: " + property);
486 
487             VMDeathRequest
488             vmdr = eventRManager.createVMDeathRequest();
489             vmdr.putProperty("number", property);
490             vmdr.setSuspendPolicy(suspendPolicy);
491 
492             return vmdr;
493         } catch ( Exception e ) {
494             log3("ERROR: ATTENTION: Exception within settingVMDeathRequest() : " + e);
495             log3("       VMDeathRequest HAS NOT BEEN SET UP");
496             throw new JDITestRuntimeException("** FAILURE to set up a VMDeathRequest **");
497         }
498     }

500     void check() {
501         log2("......checking up on eventSet.suspendPolicy()");
502         int policy = eventSet.suspendPolicy();
503         if (policy != policyToCheck) {
504             log3("ERROR: eventSet.suspendPolicy() != policyExpected");
505             log3("       eventSet.suspendPolicy() == " + policy);
506             log3("       policyExpected           == " + policyToCheck);
507             testExitCode = FAILED;
508         }
509         vm.resume();
510     }
