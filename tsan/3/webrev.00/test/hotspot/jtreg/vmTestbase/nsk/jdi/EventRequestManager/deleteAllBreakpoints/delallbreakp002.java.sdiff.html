351     private ThreadReference threadByName(String name)
352                  throws JDITestRuntimeException {
353 
354         List         all = vm.allThreads();
355         ListIterator li  = all.listIterator();
356 
357         for (; li.hasNext(); ) {
358             ThreadReference thread = (ThreadReference) li.next();
359             if (thread.name().equals(name))
360                 return thread;
361         }
362         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
363     }
