 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.addClassExclusionFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * The cases to test include ModificationWatchpointRequest.             <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 69  *   and to inform the debugger with the event.                         <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an ModificationWatchpointRequest and enables it,         <BR>
 72  *   - invokes the method addClassExclusionFilter() on the request              <BR>
 73  *     expecting to catch InvalidRequestStateException,                         <BR>
 74  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 75  *     expecting to catch no InvalidRequestStateException,                      <BR>
 76  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 77  *     once again expecting to catch InvalidRequestStateException.              <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest  eventRequest1 = null;
318         String        property1     = "ModificationWatchpointRequest1";
319         String        fieldName1    = "var101";
320 
321         ReferenceType testClassReference = null;
322 
323 
324         for (int i = 0; ; i++) {
325 
326             vm.resume();
327             breakpointForCommunication();
328 
329             int instruction = ((IntegerValue)
330                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
331 
332             if (instruction == 0) {
333                 vm.resume();
334                 break;
335             }
336 
337 
338             log1(":::::: case: # " + i);
339 
340             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
341 
342             switch (i) {
343 
344               case 0:
345                      testClassReference =
346                            (ReferenceType) vm.classesByName(testedClassName1).get(0);
347 
348                      eventRequest1 = setting23ModificationWatchpointRequest (null,
349                                              testClassReference, fieldName1,
350                                              testedClassName1,
351                                              EventRequest.SUSPEND_NONE, property1);
352 
353                      eventRequest1.enable();
354 
355                      try {
356                          log2("......eventRequest1.addClassFilter(testedClassName1);");
357                          log2("        InvalidRequestStateException expected");
358                          ((ModificationWatchpointRequest)eventRequest1).addClassFilter(testedClassName1);
359                          log3("ERROR: no InvalidRequestStateException");
360                          testExitCode = FAILED;
361                      } catch ( InvalidRequestStateException e ) {
362                          log2("        InvalidRequestStateException");
363                      }
364 
365                      log2("......eventRequest1.disable();");
366                      eventRequest1.disable();
367                      try {
368                          log2("......eventRequest1.addClassFilter(testedClassName1);");
369                          log2("        no InvalidRequestStateException expected");
370                          ((ModificationWatchpointRequest)eventRequest1).addClassFilter(testedClassName1);
371                          log2("        no InvalidRequestStateException");
372                      } catch ( InvalidRequestStateException e ) {
373                          log3("ERROR: InvalidRequestStateException");
374                          testExitCode = FAILED;
375                      }
376 
377                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
378                      eventRManager.deleteEventRequest(eventRequest1);
379                      try {
380                          log2("......eventRequest1.addClassFilter(testedClassName1);");
381                          log2("        InvalidRequestStateException expected");
382                          ((ModificationWatchpointRequest)eventRequest1).addClassFilter(testedClassName1);
383                          log3("ERROR: no InvalidRequestStateException");
384                          testExitCode = FAILED;
385                      } catch ( InvalidRequestStateException e ) {
386                          log2("        InvalidRequestStateException");
387                      }
388                       break;
389 
390               default:
391                       throw new JDITestRuntimeException("** default case 2 **");
392             }
393 
394             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
395         }
396         log1("    TESTING ENDS");
397         return;
398     }

400     private ThreadReference threadByName(String name)
401                  throws JDITestRuntimeException {
402 
403         List         all = vm.allThreads();
404         ListIterator li  = all.listIterator();
405 
406         for (; li.hasNext(); ) {
407             ThreadReference thread = (ThreadReference) li.next();
408             if (thread.name().equals(name))
409                 return thread;
410         }
411         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
412     }

414    /*
415     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
416     *                                             String, String, String)
417     *
418     * It sets up a breakpoint at given line number within a given method in a given class
419     * for a given thread.
420     *
421     * Return value: BreakpointRequest object  in case of success
422     *
423     * JDITestRuntimeException   in case of an Exception thrown within the method
424     */
425 
426     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
427                                                   ReferenceType testedClass,
428                                                   String methodName,
429                                                   String bpLine,
430                                                   String property)
431             throws JDITestRuntimeException {
432 
433         log2("......setting up a breakpoint:");
434         log2("       thread: " + thread + "; class: " + testedClass +
435                         "; method: " + methodName + "; line: " + bpLine);
436 
437         List              alllineLocations = null;
438         Location          lineLocation     = null;
439         BreakpointRequest breakpRequest    = null;
440 
441         try {
442             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
443 
444             alllineLocations = method.allLineLocations();
445 
446             int n =
447                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
448             if (n > alllineLocations.size()) {
449                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
450             } else {
451                 lineLocation = (Location) alllineLocations.get(n);
452                 try {
453                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
454                     breakpRequest.putProperty("number", property);
455                     breakpRequest.addThreadFilter(thread);
456                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
457                 } catch ( Exception e1 ) {
458                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
459                     breakpRequest    = null;
460                 }
461             }
462         } catch ( Exception e2 ) {
463             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
464             breakpRequest    = null;
465         }
466 
467         if (breakpRequest == null) {
468             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
469             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
470         }
471 
472         log2("      a breakpoint has been set up");
473         return breakpRequest;
474     }

477     private void getEventSet()
478                  throws JDITestRuntimeException {
479         try {
480 //            log2("       eventSet = eventQueue.remove(waitTime);");
481             eventSet = eventQueue.remove(waitTime);
482             if (eventSet == null) {
483                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
484             }
485 //            log2("       eventIterator = eventSet.eventIterator;");
486             eventIterator = eventSet.eventIterator();
487         } catch ( Exception e ) {
488             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
489         }
490     }

507     private ModificationWatchpointRequest setting23ModificationWatchpointRequest (
508                                                   ThreadReference thread,
509                                                   ReferenceType   fieldClass,
510                                                   String          fieldName,
511                                                   String          testedClass,
512                                                   int             suspendPolicy,
513                                                   String          property        )
514             throws JDITestRuntimeException {
515         try {
516             log2("......setting up ModificationWatchpointRequest:");
517             log2("       thread: " + thread + "; fieldClass: " + fieldClass + "; fieldName: " + fieldName);
518             Field field = fieldClass.fieldByName(fieldName);
519 
520             ModificationWatchpointRequest
521             awr = eventRManager.createModificationWatchpointRequest(field);
522             awr.putProperty("number", property);
523 
524             if (thread != null)
525                 awr.addThreadFilter(thread);
526             awr.setSuspendPolicy(suspendPolicy);
527             awr.addClassExclusionFilter(testedClass);
528 
529             log2("      ModificationWatchpointRequest has been set up");
530             return awr;
531         } catch ( Exception e ) {
532             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
533             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
534             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
535         }
