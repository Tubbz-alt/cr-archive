340     private ThreadReference threadByName(String name)
341                  throws JDITestRuntimeException {
342 
343         List         all = vm.allThreads();
344         ListIterator li  = all.listIterator();
345 
346         for (; li.hasNext(); ) {
347             ThreadReference thread = (ThreadReference) li.next();
348             if (thread.name().equals(name))
349                 return thread;
350         }
351         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
352     }
