 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for BreakpointEvent sets.             <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a BreakpointRequest, resumes                  <BR>
 74  *   the debuggee, and waits for the BreakpointEvent.                   <BR>
 75  * - The debuggee invokes the special method which makes access to      <BR>
 76  *   a predefined variable to be resulting in the event.                <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         String bPointMethod = "methodForCommunication";
301         String lineForComm  = "lineForComm";
302         BreakpointRequest bpRequest;
303         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
304         bpRequest = settingBreakpoint(mainThread,
305                                       debuggeeClass,
306                                       bPointMethod, lineForComm, "zero");
307         bpRequest.enable();
308 
309         vm.resume();
310 
311     //------------------------------------------------------  testing section
312 
313         log1("     TESTING BEGINS");
314 
315         EventRequest eventRequest1 = null;
316         EventRequest eventRequest2 = null;
317         EventRequest eventRequest3 = null;
318 
319         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
320         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
321         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
322 
323         String methodName = "method";
324         String bpLineName = "breakpointLine";
325 
326         ReferenceType testClassReference = null;
327 
328 
329         for (int i = 0; ; i++) {
330 
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             switch (i) {
346 
347               case 0:
348                       testClassReference =
349                            (ReferenceType) vm.classesByName(testedClassName).get(0);
350 
351                       eventRequest1 = settingBreakpointRequest (mainThread,
352                                              testClassReference, methodName, bpLineName,
353                                              SUSPEND_NONE, "BreakpointRequest1");
354                       eventRequest1.enable();
355                       break;
356 
357               case 1:
358                       eventRequest2 = settingBreakpointRequest (mainThread,
359                                              testClassReference, methodName, bpLineName,
360                                              SUSPEND_THREAD, "BreakpointRequest2");
361                       eventRequest2.enable();
362                       break;
363 
364               case 2:
365                       eventRequest3 = settingBreakpointRequest (mainThread,
366                                              testClassReference, methodName, bpLineName,
367                                              SUSPEND_ALL, "BreakpointRequest3");
368                       eventRequest3.enable();
369                       break;
370 
371 
372               default:
373                       throw new JDITestRuntimeException("** default case 2 **");
374             }
375 
376             log2("......waiting for new BreakpointEvent : " + i);
377             mainThread.resume();
378             getEventSet();
379 
380             Event newEvent = eventIterator.nextEvent();
381             if ( !(newEvent instanceof BreakpointEvent)) {
382                  log3("ERROR: new event is not BreakpointEvent");
383                  testExitCode = FAILED;
384             } else {
385 
386                 String property = (String) newEvent.request().getProperty("number");
387                 log2("       got new BreakpointEvent with propety 'number' == " + property);
388 
389                 log2("......checking up on EventSet.resume()");
390                 log2("......--> vm.suspend();");
391                 vm.suspend();
392 
393                 log2("        getting : Map<String, Integer> suspendsCounts1");
394 
395                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
396                 for (ThreadReference threadReference : vm.allThreads()) {
397                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
398                 }
399 
400                 log2("        eventSet.resume;");
401                 eventSet.resume();
402 
403                 log2("        getting : Map<String, Integer> suspendsCounts2");
404                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
405                 for (ThreadReference threadReference : vm.allThreads()) {
406                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
407                 }
408 
409                 log2(suspendsCounts1.toString());
410                 log2(suspendsCounts2.toString());
411 
412                 log2("        getting : int policy = eventSet.suspendPolicy();");
413                 int policy = eventSet.suspendPolicy();
414 
415                 switch (policy) {
416 
417                   case SUSPEND_NONE   :
418                        log2("        case SUSPEND_NONE");
419                        for (String threadName : suspendsCounts1.keySet()) {
420                            log2("        checking " + threadName);
421                            if (!suspendsCounts2.containsKey(threadName)) {
422                                log3("ERROR: couldn't get ThreadReference for " + threadName);
423                                testExitCode = FAILED;
424                                break;
425                            }
426                            int count1 = suspendsCounts1.get(threadName);
427                            int count2 = suspendsCounts2.get(threadName);
428                            if (count1 != count2) {
429                                log3("ERROR: suspendCounts don't match for : " + threadName);
430                                log3("before resuming : " + count1);
431                                log3("after  resuming : " + count2);
432                                testExitCode = FAILED;
433                                break;
434                            }
435                        }
436                        eventRequest1.disable();
437                        break;
438 
439                   case SUSPEND_THREAD :
440                        log2("        case SUSPEND_THREAD");
441                        for (String threadName : suspendsCounts1.keySet()) {
442                            log2("checking " + threadName);
443                            if (!suspendsCounts2.containsKey(threadName)) {
444                                log3("ERROR: couldn't get ThreadReference for " + threadName);
445                                testExitCode = FAILED;
446                                break;
447                            }
448                            int count1 = suspendsCounts1.get(threadName);
449                            int count2 = suspendsCounts2.get(threadName);
450                            String eventThreadName = ((BreakpointEvent)newEvent).thread().name();
451                            int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
452                            if (count1 != expectedValue) {
453                                log3("ERROR: suspendCounts don't match for : " + threadName);
454                                log3("before resuming : " + count1);
455                                log3("after  resuming : " + count2);
456                                testExitCode = FAILED;
457                                break;
458                            }
459                        }
460                        eventRequest2.disable();
461                        break;
462 
463                   case SUSPEND_ALL    :
464 
465                         log2("        case SUSPEND_ALL");
466                         for (String threadName : suspendsCounts1.keySet()) {
467                             log2("checking " + threadName);
468                             if (!event.request().equals(eventRequest3))
469                                 break;
470                             if (!suspendsCounts2.containsKey(threadName)) {
471                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
472                                 testExitCode = FAILED;
473                                 break;
474                             }
475                             int count1 = suspendsCounts1.get(threadName);
476                             int count2 = suspendsCounts2.get(threadName);
477                             if (count1 != count2 + 1) {
478                                 log3("ERROR: suspendCounts don't match for : " + threadName);
479                                 log3("before resuming : " + count1);
480                                 log3("after  resuming : " + count2);
481                                 testExitCode = FAILED;
482                                 break;
483                             }
484                         }
485                         eventRequest3.disable();
486                         break;
487 
488                   default: throw new JDITestRuntimeException("** default case 1 **");
489                 }
490                 informDebuggeeTestCase(i);
491             }
492 
493             log2("......--> vm.resume()");
494             vm.resume();
495             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
496         }
497         log1("    TESTING ENDS");
498         return;
499     }

501    /*
502     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
503     *                                             String, String, String)
504     *
505     * It sets up a breakpoint at given line number within a given method in a given class
506     * for a given thread.
507     *
508     * Return value: BreakpointRequest object  in case of success
509     *
510     * JDITestRuntimeException   in case of an Exception thrown within the method
511     */
512 
513     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
514                                                   ReferenceType testedClass,
515                                                   String methodName,
516                                                   String bpLine,
517                                                   String property)
518             throws JDITestRuntimeException {
519 
520         log2("......setting up a breakpoint:");
521         log2("       thread: " + thread + "; class: " + testedClass +
522                         "; method: " + methodName + "; line: " + bpLine);
523 
524         List              alllineLocations = null;
525         Location          lineLocation     = null;
526         BreakpointRequest breakpRequest    = null;
527 
528         try {
529             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
530 
531             alllineLocations = method.allLineLocations();
532 
533             int n =
534                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
535             if (n > alllineLocations.size()) {
536                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
537             } else {
538                 lineLocation = (Location) alllineLocations.get(n);
539                 try {
540                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
541                     breakpRequest.putProperty("number", property);
542                     breakpRequest.addThreadFilter(thread);
543                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
544                 } catch ( Exception e1 ) {
545                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
546                     breakpRequest    = null;
547                 }
548             }
549         } catch ( Exception e2 ) {
550             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
551             breakpRequest    = null;
552         }
553 
554         if (breakpRequest == null) {
555             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
556             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
557         }
558 
559         log2("      a breakpoint has been set up");
560         return breakpRequest;
561     }

564     private void getEventSet()
565                  throws JDITestRuntimeException {
566         try {
567 //            log2("       eventSet = eventQueue.remove(waitTime);");
568             eventSet = eventQueue.remove(waitTime);
569             if (eventSet == null) {
570                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
571             }
572 //            log2("       eventIterator = eventSet.eventIterator;");
573             eventIterator = eventSet.eventIterator();
574         } catch ( Exception e ) {
575             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
576         }
577     }

581                  throws JDITestRuntimeException {

594     private BreakpointRequest settingBreakpointRequest ( ThreadReference thread,
595                                                          ReferenceType   testedClass,
596                                                          String          methodName,
597                                                          String          bpLine,
598                                                          int             suspendPolicy,
599                                                          String          property        )
600             throws JDITestRuntimeException {
601         try {
602             log2("......setting up a breakpoint:");
603             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
604 
605             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
606             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
607 
608             BreakpointRequest
609             bpr = eventRManager.createBreakpointRequest(loc);
610             bpr.putProperty("number", property);
611             bpr.addThreadFilter(thread);
612             bpr.setSuspendPolicy(suspendPolicy);
613 
614             log2("      a breakpoint has been set up");
615             return bpr;
616         } catch ( Exception e ) {
617             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
618             log3("       BreakpointRequest HAS NOT BEEN SET UP");
619             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
620         }
621     }
622     /**
623      * Inform debuggee which thread test the debugger has completed.
624      * Used for synchronization, so the debuggee does not move too quickly.
625      * @param testCase index of just completed test
626      */
627     void informDebuggeeTestCase(int testCase) {
628         try {
629             ((ClassType)debuggeeClass)
630                 .setValue(debuggeeClass.fieldByName("testCase"),
631                           vm.mirrorOf(testCase));
632         } catch (InvalidTypeException ite) {
633             throw new Failure("** FAILURE setting testCase  **");
634         } catch (ClassNotLoadedException cnle) {
635             throw new Failure("** FAILURE notifying debuggee  **");
636         } catch (VMDisconnectedException e) {
637             throw new Failure("** FAILURE debuggee connection **");
638         }
639     }
