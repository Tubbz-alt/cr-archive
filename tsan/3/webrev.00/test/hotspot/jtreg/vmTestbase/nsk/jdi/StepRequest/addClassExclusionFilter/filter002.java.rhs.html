 36 /**
 37  * The test for the implementation of an object of the type     <BR>
 38  * StepRequest.                                                 <BR>
 39  *                                                              <BR>
 40  * The test checks that results of the method                   <BR>
 41  * <code>com.sun.jdi.StepRequest.addClassExclusionFilter()</code> <BR>
 42  * complies with its spec.                                              <BR>
 43  * <BR>
 44  * The test checks up on the following assertion:                       <BR>
 45  *    Throws: InvalidRequestStateException -                            <BR>
 46  *            if this request is currently enabled or has been deleted. <BR>
 47  *            Filters may be added only to disabled requests.           <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.                      <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assertion,                              <BR>
 64  * the debugger and the debuggee perform the following.                 <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 67  *   and to inform the debugger with the event.                         <BR>
 68  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 69  *   - creates an StepRequest and enables it,                           <BR>
 70  *   - invokes the method addClassExclusionFilter() on the request      <BR>
 71  *     expecting to catch InvalidRequestStateException,                 <BR>
 72  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 73  *     expecting to catch no InvalidRequestStateException,                      <BR>
 74  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 75  *     once again expecting to catch InvalidRequestStateException.              <BR>
 76  * <BR>
 77  * In third phase, at the end of the test, the debuggee changes         <BR>
 78  * the value of the "instruction" which the debugger and debuggee       <BR>
 79  * use to inform each other of needed actions,  and both end.           <BR>
 80  * <BR>
 81  */

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : " + e);
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : " + e);
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if (eventIterator.nextEvent() instanceof VMDeathEvent)
259                 return 0;
260 
261             log3("ERROR: last event is not the VMDeathEvent");
262             return 1;
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         eventRManager = vm.eventRequestManager();
277 
278         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
279         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
280         cpRequest.addClassFilter(debuggeeName);
281 
282         cpRequest.enable();
283         vm.resume();
284         getEventSet();
285         cpRequest.disable();
286 
287         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
288         debuggeeClass = event.referenceType();
289 
290         if (!debuggeeClass.name().equals(debuggeeName))
291            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
292 
293         log2("      received: ClassPrepareEvent for debuggeeClass");
294 
295         String bPointMethod = "methodForCommunication";
296         String lineForComm  = "lineForComm";
297 
298         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
299 
300         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
301                                              debuggeeClass,
302                                             bPointMethod, lineForComm, "zero");
303         bpRequest.enable();
304 
305     //------------------------------------------------------  testing section
306 
307         log1("     TESTING BEGINS");
308 
309         EventRequest  eventRequest1      = null;
310 
311         String        property1          = "StepRequest1";
312 
313 
314         for (int i = 0; ; i++) {
315 
316             vm.resume();
317             breakpointForCommunication();
318 
319             int instruction = ((IntegerValue)
320                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
321 
322             if (instruction == 0) {
323                 vm.resume();
324                 break;
325             }
326 
327             log1(":::::: case: # " + i);
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
330 
331             switch (i) {
332 
333               case 0:
334                      ThreadReference thread1 = debuggee.threadByNameOrThrow("main");
335                      eventRequest1 = setting23StepRequest(thread1, testedClassName1,
336                                               EventRequest.SUSPEND_NONE, property1);
337 
338                      eventRequest1.enable();
339 
340                      try {
341                          log2("......eventRequest1.addClassExclusionFilter(testedClassName1);");
342                          log2("        InvalidRequestStateException expected");
343                          ((StepRequest)eventRequest1).addClassExclusionFilter(testedClassName1);
344                          log3("ERROR: no InvalidRequestStateException");
345                          testExitCode = FAILED;
346                      } catch ( InvalidRequestStateException e ) {
347                          log2("        InvalidRequestStateException");
348                      }
349 
350                      log2("......eventRequest1.disable();");
351                      eventRequest1.disable();
352                      try {
353                          log2("......eventRequest1.addClassExclusionFilter(testedClassName1);");
354                          log2("        no InvalidRequestStateException expected");
355                          ((StepRequest)eventRequest1).addClassExclusionFilter(testedClassName1);
356                          log2("        no InvalidRequestStateException");
357                      } catch ( InvalidRequestStateException e ) {
358                          log3("ERROR: InvalidRequestStateException");
359                          testExitCode = FAILED;
360                      }
361 
362                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
363                      eventRManager.deleteEventRequest(eventRequest1);
364                      try {
365                          log2("......eventRequest1.addClassExclusionFilter(testedClassName1);");
366                          log2("        InvalidRequestStateException expected");
367                          ((StepRequest)eventRequest1).addClassExclusionFilter(testedClassName1);
368                          log3("ERROR: no InvalidRequestStateException");
369                          testExitCode = FAILED;
370                      } catch ( InvalidRequestStateException e ) {
371                          log2("        InvalidRequestStateException");
372                      }
373                       break;
374 
375               default:
376                       throw new JDITestRuntimeException("** default case 2 **");
377             }
378 
379             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
380         }
381         log1("    TESTING ENDS");
382         return;
383     }

385    /*
386     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
387     *                                             String, String, String)
388     *
389     * It sets up a breakpoint at given line number within a given method in a given class
390     * for a given thread.
391     *
392     * Return value: BreakpointRequest object  in case of success
393     *
394     * JDITestRuntimeException   in case of an Exception thrown within the method
395     */
396 
397     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
398                                                   ReferenceType testedClass,
399                                                   String methodName,
400                                                   String bpLine,
401                                                   String property)
402             throws JDITestRuntimeException {
403 
404         log2("......setting up a breakpoint:");
405         log2("       thread: " + thread + "; class: " + testedClass +
406                         "; method: " + methodName + "; line: " + bpLine);
407 
408         List              alllineLocations = null;
409         Location          lineLocation     = null;
410         BreakpointRequest breakpRequest    = null;
411 
412         try {
413             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
414 
415             alllineLocations = method.allLineLocations();
416 
417             int n =
418                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
419             if (n > alllineLocations.size()) {
420                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
421             } else {
422                 lineLocation = (Location) alllineLocations.get(n);
423                 try {
424                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
425                     breakpRequest.putProperty("number", property);
426                     breakpRequest.addThreadFilter(thread);
427                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
428                 } catch ( Exception e1 ) {
429                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
430                     breakpRequest    = null;
431                 }
432             }
433         } catch ( Exception e2 ) {
434             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
435             breakpRequest    = null;
436         }
437 
438         if (breakpRequest == null) {
439             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
440             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
441         }
442 
443         log2("      a breakpoint has been set up");
444         return breakpRequest;
445     }

448     private void getEventSet()
449                  throws JDITestRuntimeException {
450         try {
451 //            log2("       eventSet = eventQueue.remove(waitTime);");
452             eventSet = eventQueue.remove(waitTime);
453             if (eventSet == null) {
454                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
455             }
456 //            log2("       eventIterator = eventSet.eventIterator;");
457             eventIterator = eventSet.eventIterator();
458         } catch ( Exception e ) {
459             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
460         }
461     }

464     private void breakpointForCommunication()
465                  throws JDITestRuntimeException {
466 
467         log2("breakpointForCommunication");
468         getEventSet();
469         Event evt = eventIterator.nextEvent();
470 
471         if (evt instanceof BreakpointEvent)
472             return;
473 
474         log3("Didn't get expected BreakpointEvent, events are:");
475         while(true) {
476             log3("    event = " + evt);
477             if (!eventIterator.hasNext()) {
478                 break;
479             }
480             evt = eventIterator.nextEvent();
481         }
482 
483         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
484     }

488     private StepRequest setting23StepRequest ( ThreadReference thread,
489                                                String          testedClass,
490                                                int             suspendPolicy,
491                                                String          property        )
492             throws JDITestRuntimeException {
493         try {
494             log2("......setting up StepRequest:");
495             log2("       thread: " + thread + "; property: " + property);
496 
497             StepRequest
498             str = eventRManager.createStepRequest(thread, StepRequest.STEP_LINE, StepRequest.STEP_INTO);
499             str.putProperty("number", property);
500             str.addCountFilter(1);
501             str.setSuspendPolicy(suspendPolicy);
502             str.addClassExclusionFilter(testedClass);
503 
504             log2("      StepRequest has been set up");
505             return str;
506         } catch ( Exception e ) {
507             log3("ERROR: ATTENTION: Exception within settingStepRequest() : " + e);
508             log3("       StepRequest HAS NOT BEEN SET UP");
509             throw new JDITestRuntimeException("** FAILURE to set up StepRequest **");
510         }
