 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ClassLoader.                                                 <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ClassLoader.definedClasses()</code>        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    The list has length 0 if no types have been defined by    <BR>
 47  *    this classloader.                                         <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.              <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assetion,                               <BR>
 64  * the debugger and the debuggee perform the following.                 <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee prepares a ClassLoader object and invokes             <BR>
 67  *   the methodForCommunication to be suspended and                     <BR>
 68  *   to inform the debugger with the event.                             <BR>
 69  * - Upon getting the BreakpointEvent,                                  <BR>
 70  *   the debugger performs the check.                                   <BR>
 71  * Note. To inform each other of needed actions, the debugger and       <BR>
 72  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 73  * <BR>
 74  * In third phase when at the end,                                      <BR>
 75  * the debuggee changes the value of the "instruction"                  <BR>
 76  * to inform the debugger of checks finished, and both end.             <BR>
 77  * <BR>
 78  */
 79

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : e");
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : e");
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

245    /*
246     * Return value: 0 - normal end of the test
247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase
249     */
250 
251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if (eventIterator.nextEvent() instanceof VMDeathEvent)
259                 return 0;
260 
261             log3("ERROR: last event is not the VMDeathEvent");
262             return 1;
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         eventRManager = vm.eventRequestManager();
277 
278         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
279         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
280         cpRequest.addClassFilter(debuggeeName);
281 
282         cpRequest.enable();
283         vm.resume();
284         getEventSet();
285         cpRequest.disable();
286 
287         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
288         debuggeeClass = event.referenceType();
289 
290         if (!debuggeeClass.name().equals(debuggeeName))
291            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
292 
293         log2("      received: ClassPrepareEvent for debuggeeClass");
294 
295         String bPointMethod = "methodForCommunication";
296         String lineForComm  = "lineForComm";
297         BreakpointRequest bpRequest;
298 
299         bpRequest = settingBreakpoint(threadByName("main"),
300                                       debuggeeClass,
301                                       bPointMethod, lineForComm, "zero");
302         bpRequest.enable();
303 
304     //------------------------------------------------------  testing section
305 
306         log1("     TESTING BEGINS");
307 
308         for (int i = 0; ; i++) {
309 
310             vm.resume();
311             breakpointForCommunication();
312 
313             int instruction = ((IntegerValue)
314                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
315 
316             if (instruction == 0) {
317                 vm.resume();
318                 break;
319             }
320 
321             log1(":::::: case: # " + i);
322 
323             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
324 
325             String ClassLoaderObjName = "classLoader";
326 
327             log2("......getting: Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));");
328             Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(ClassLoaderObjName));
329 
330             log2("......getting: ClassLoaderReference clRef = (ClassLoaderReference) val;");
331             ClassLoaderReference clRef = (ClassLoaderReference) val;
332 
333             log2("......getting: List definedClasses = clRef.definedClasses();");
334             List definedClasses = clRef.definedClasses();
335             log2("......checking up on definedClasses.size() :  0 is expected");
336             if (definedClasses.size() != 0) {
337                 log3("ERROR: definedClasses.size() != 0");
338                 testExitCode = FAILED;
339             }
340 
341             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
342         }
343         log1("    TESTING ENDS");
344         return;
345     }

347     private ThreadReference threadByName(String name)
348                  throws JDITestRuntimeException {
349 
350         List         all = vm.allThreads();
351         ListIterator li  = all.listIterator();
352 
353         for (; li.hasNext(); ) {
354             ThreadReference thread = (ThreadReference) li.next();
355             if (thread.name().equals(name))
356                 return thread;
357         }
358         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
359     }

361    /*
362     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
363     *                                             String, String, String)
364     *
365     * It sets up a breakpoint at given line number within a given method in a given class
366     * for a given thread.
367     *
368     * Return value: BreakpointRequest object  in case of success
369     *
370     * JDITestRuntimeException   in case of an Exception thrown within the method
371     */
372 
373     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
374                                                   ReferenceType testedClass,
375                                                   String methodName,
376                                                   String bpLine,
377                                                   String property)
378             throws JDITestRuntimeException {
379 
380         log2("......setting up a breakpoint:");
381         log2("       thread: " + thread + "; class: " + testedClass +
382                         "; method: " + methodName + "; line: " + bpLine);
383 
384         List              alllineLocations = null;
385         Location          lineLocation     = null;
386         BreakpointRequest breakpRequest    = null;
387 
388         try {
389             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
390 
391             alllineLocations = method.allLineLocations();
392 
393             int n =
394                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
395             if (n > alllineLocations.size()) {
396                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
397             } else {
398                 lineLocation = (Location) alllineLocations.get(n);
399                 try {
400                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
401                     breakpRequest.putProperty("number", property);
402                     breakpRequest.addThreadFilter(thread);
403                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
404                 } catch ( Exception e1 ) {
405                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
406                     breakpRequest    = null;
407                 }
408             }
409         } catch ( Exception e2 ) {
410             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
411             breakpRequest    = null;
412         }
413 
414         if (breakpRequest == null) {
415             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
416             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
417         }
418 
419         log2("      a breakpoint has been set up");
420         return breakpRequest;
421     }

424     private void getEventSet()
425                  throws JDITestRuntimeException {
426         try {
427 //            log2("       eventSet = eventQueue.remove(waitTime);");
428             eventSet = eventQueue.remove(waitTime);
429             if (eventSet == null) {
430                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
431             }
432 //            log2("       eventIterator = eventSet.eventIterator;");
433             eventIterator = eventSet.eventIterator();
434         } catch ( Exception e ) {
435             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
436         }
437     }

440     private void breakpointForCommunication()
441                  throws JDITestRuntimeException {
442 
443         log2("breakpointForCommunication");
444         getEventSet();
445 
446         if (eventIterator.nextEvent() instanceof BreakpointEvent)
447             return;
448 
449         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
450     }
