 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *  After this operation,                                       <BR>
 47  *  this thread will be suspended at the invoke instruction of  <BR>
 48  *  the target method that created frame.                       <BR>
 49  *  The frame's method can be reentered with a                  <BR>
 50  *  step into the instruction.                                  <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assetion,                               <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new check case and invokes                   <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the check case required.                     <BR>
 74  * Note. To inform each other of needed actions, the debugger and       <BR>
 75  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 76  * <BR>
 77  * In third phase when at the end,                                      <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  * In the second phase, the debuggee creates second thread which        <BR>
 82  * will call a tested poppedMethod() from its run() method.             <BR>
 83  * Being called, the poppedMethod() increments the value of             <BR>
 84  * the global variable testVar1.                                        <BR>
 85  * The debugger sets up a breakpoint within the poppedMethod() and,     <BR>
 86  * after getting this brakpoint event, invokes the popFrames() method,  <BR>
 87  * checks the value of testVar1 after first incerment,                  <BR>
 88  * and resumes debuggee's second thread to reenter the poppedMethod().  <BR>
 89  * After getting the breakpoint event second time,                      <BR>
 90  * the debugger just checks the value of testVar1 which                 <BR>
 91  * has to be incremented one more time.                                 <BR>
 92  */

167     private int runThis (String argv[], PrintStream out) {
168 
169         argsHandler     = new ArgumentHandler(argv);
170         logHandler      = new Log(out, argsHandler);
171         Binder binder   = new Binder(argsHandler, logHandler);
172 
173         waitTime        = argsHandler.getWaitTime() * 60000;
174 
175         try {
176             log2("launching a debuggee :");
177             log2("       " + debuggeeName);
178             if (argsHandler.verbose()) {
179                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
180             } else {
181                 debuggee = binder.bindToDebugee(debuggeeName);
182             }
183             if (debuggee == null) {
184                 log3("ERROR: no debuggee launched");
185                 return FAILED;
186             }
187             log2("debuggee launched");
188         } catch ( Exception e ) {
189             log3("ERROR: Exception : " + e);
190             log2("       test cancelled");
191             return FAILED;
192         }
193 
194         debuggee.redirectOutput(logHandler);
195 
196         vm = debuggee.VM();
197 
198         eventQueue = vm.eventQueue();
199         if (eventQueue == null) {
200             log3("ERROR: eventQueue == null : TEST ABORTED");
201             vm.exit(PASS_BASE);
202             return FAILED;
203         }
204 
205         log2("invocation of the method runTest()");
206         switch (runTest()) {
207 
208             case 0 :  log2("test phase has finished normally");
209                       log2("   waiting for the debuggee to finish ...");
210                       debuggee.waitFor();
211 
212                       log2("......getting the debuggee's exit status");
213                       int status = debuggee.getStatus();
214                       if (status != PASS_BASE) {
215                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
216                               status + " != PASS_BASE");
217                           testExitCode = FAILED;
218                       } else {
219                           log2("......debuggee returned expected exit status: " +
220                               status + " == PASS_BASE");
221                       }
222                       break;
223 
224             default : log3("ERROR: runTest() returned unexpected value");
225 
226             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
227                       log2("......forcing: vm.exit();");
228                       testExitCode = FAILED;
229                       try {
230                           vm.exit(PASS_BASE);
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : e");
233                       }
234                       break;
235 
236             case 2 :  log3("test cancelled due to VMDisconnectedException");
237                       log2("......trying: vm.process().destroy();");
238                       testExitCode = FAILED;
239                       try {
240                           Process vmProcess = vm.process();
241                           if (vmProcess != null) {
242                               vmProcess.destroy();
243                           }
244                       } catch ( Exception e ) {
245                           log3("ERROR: Exception : e");
246                       }
247                       break;
248             }
249 
250         return testExitCode;
251     }

260     private int runTest() {
261 
262         try {
263             testRun();
264 
265             log2("waiting for VMDeathEvent");
266             getEventSet();
267             if (eventIterator.nextEvent() instanceof VMDeathEvent)
268                 return 0;
269 
270             log3("ERROR: last event is not the VMDeathEvent");
271             return 1;
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306         BreakpointRequest bpRequest;
307 
308         try {
309             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
310                                           debuggeeClass,
311                                           bPointMethod, lineForComm, "zero");
312         } catch ( Exception e ) {
313             throw e;
314         }
315         bpRequest.enable();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         log2("......vm.resume();  resuming debuggee's main thread");
322         vm.resume();
323 
324         for (int i = 0; ; i++) {
325 
326 //            vm.resume();
327 
328             breakpointForCommunication();
329 
330             int instruction;
331 
332             log2("......instruction = ((IntegerValue)...; no Exception expected");
333             try {
334                 instruction = ((IntegerValue)
335                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
336             } catch ( Exception e ) {
337                 log3("ERROR: Exception when 'instruction = ((IntegerValue)...' : " + e);
338                 testExitCode = FAILED;
339                 return;
340             }
341 
342             if (instruction == 0) {
343                 vm.resume();
344                 break;
345             }
346 
347             log1("  new check: # " + i);
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351             if ( !vm.canPopFrames() ) {
352                 log2("......vm.canPopFrames() == false : test is cancelled");
353                 return;
354             }
355 
356             String thread2Name         = "thread2";
357             ThreadReference thread2Ref = debuggee.threadByNameOrThrow(thread2Name);
358 
359 
360             String poppedMethod    = "poppedMethod";
361             String breakpointLine  = "breakpointLine";
362             String testVariable    = "testVar1";
363 
364             log2("......setting breakpoint in poppedMethod");
365             try {
366                 breakpointRequest = settingBreakpoint(debuggee.threadByNameOrThrow(thread2Name),
367                                           debuggeeClass,
368                                           poppedMethod, breakpointLine, "one");
369             } catch ( Exception e ) {
370                 throw e;
371             }
372             log2("......breakpointRequest.enable();");
373             breakpointRequest.enable();
374 
375             log2("......eventSet.resume();   resuming debuggee's main thread");
376             log2("                           to get new rendevous at the breakpointForCommunication");
377             eventSet.resume();
378             log2("......breakpointInMethod();");
379             breakpointInMethod();
380 
381             log2("......getting value of 'testVar1'; 1 is expected");
382             Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(testVariable));
383             int intVal = ((IntegerValue) val).value();
384             if (intVal != 1) {
385                 log3("ERROR: intVal != 1");
386                 testExitCode = FAILED;
387                 break;
388             }
389 
390             log2("......StackFrame stackFrame = thread2Ref.frame(0);");
391             StackFrame stackFrame = thread2Ref.frame(0);
392 
393             log2("......thread2Ref.popFrames(stackFrame);");
394             try {
395                 thread2Ref.popFrames(stackFrame);
396             } catch ( IncompatibleThreadStateException e ) {
397                 log3("ERROR: IncompatibleThreadStateException");
398                 testExitCode = FAILED;
399                 break;
400             }
401 
402             log2("......thread2Ref.resume();");
403             thread2Ref.resume();
404             log2("......breakpointInMethod();");
405             breakpointInMethod();
406 
407             log2("......getting value of 'testVar1'; 2 is expected");
408             val = debuggeeClass.getValue(debuggeeClass.fieldByName(testVariable));
409             intVal = ((IntegerValue) val).value();
410             if (intVal != 2) {
411                 log3("ERROR: intVal != 2");
412                 testExitCode = FAILED;
413                 break;
414             }
415 
416             log2("......thread2Ref.resume();");
417             thread2Ref.resume();
418 
419             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
420         }
421         log1("    TESTING ENDS");
422         return;
423     }

425    /*
426     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
427     *                                             String, String, String)
428     *
429     * It sets up a breakpoint at given line number within a given method in a given class
430     * for a given thread.
431     *
432     * Return value: BreakpointRequest object  in case of success
433     *
434     * JDITestRuntimeException   in case of an Exception thrown within the method
435     */
436 
437     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
438                                                   ReferenceType testedClass,
439                                                   String methodName,
440                                                   String bpLine,
441                                                   String property)
442             throws JDITestRuntimeException {
443 
444         log2("......setting up a breakpoint:");
445         log2("       thread: " + thread + "; class: " + testedClass +
446                         "; method: " + methodName + "; line: " + bpLine);
447 
448         List              alllineLocations = null;
449         Location          lineLocation     = null;
450         BreakpointRequest breakpRequest    = null;
451 
452         try {
453             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
454 
455             alllineLocations = method.allLineLocations();
456 
457             int n =
458                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
459             if (n > alllineLocations.size()) {
460                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
461             } else {
462                 lineLocation = (Location) alllineLocations.get(n);
463                 try {
464                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
465                     breakpRequest.putProperty("number", property);
466                     breakpRequest.addThreadFilter(thread);
467                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
468                 } catch ( Exception e1 ) {
469                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
470                     breakpRequest    = null;
471                 }
472             }
473         } catch ( Exception e2 ) {
474             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
475             breakpRequest    = null;
476         }
477 
478         if (breakpRequest == null) {
479             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
480             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
481         }
482 
483         log2("      a breakpoint has been set up");
484         return breakpRequest;
485     }

488     private void getEventSet()
489                  throws JDITestRuntimeException {
490         try {
491 //            log2("       eventSet = eventQueue.remove(waitTime);");
492             eventSet = eventQueue.remove(waitTime);
493             if (eventSet == null) {
494                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
495             }
496 //            log2("       eventIterator = eventSet.eventIterator;");
497             eventIterator = eventSet.eventIterator();
498         } catch ( Exception e ) {
499             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
500         }
501     }

504     private void breakpointForCommunication()
505                  throws JDITestRuntimeException {
506 
507         log2("breakpointForCommunication");
508         getEventSet();
509 
510         if (eventIterator.nextEvent() instanceof BreakpointEvent)
511             return;
512 
513         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
514     }

518     private void breakpointInMethod()
519                  throws JDITestRuntimeException {
520 
521         log2("breakpointInMethod");
522         getEventSet();
523         Event event = eventIterator.nextEvent();
524 
525         if ( !(event instanceof BreakpointEvent) )
526             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
527 
528         if ( !event.request().equals(breakpointRequest) )
529             throw new JDITestRuntimeException("** unexpected breakpoint **");
530     }
