 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertions:              <BR>
 46  * -  All StackFrame objects for this thread are invalidated.   <BR>
 47  * -  The frame previous to the parameter frame                 <BR>
 48  *    will become the current frame.                            <BR>
 49  * <BR>
 50  * The test has two phases and works as follows.                <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assetion,                               <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent<BR>
 67  * in the methodForCommunication().                                     <BR>
 68  * - The debuggee creates second thread, the thread2, which will        <BR>
 69  * call a tested poppedMethod() from its run() method.                  <BR>
 70  * The poppedMethod will call the special breakpointMethod,             <BR>
 71  * After creating the thread2, the debuggee invokes                     <BR>
 72  * the methodForCommunication to be suspended and                       <BR>
 73  * to inform the debugger with the event.                               <BR>
 74  * - Upon getting the BreakpointEvent in the methodForCommunication,    <BR>
 75  * the debugger sets up a breakpoint in the breakpointMethod,           <BR>
 76  * resumes the debuggee, waits for BreakpointEvent in the               <BR>
 77  * breakpointMethod and, and upon getting it,                           <BR>
 78  * (1) to check up on first assertion, gets thread2.frame(2),           <BR>
 79  * invokes popFrames() and checks up on equality of thread2.frame(2)    <BR>
 80  * and thread2.frame(0). Since thread2.frame(2) has to be invalid,      <BR>
 81  * the expected result is false; otherwise, the test FAILED.            <BR>
 82  * (2) to check up on second assertion, gets thread2.frame(1)           <BR>
 83  * which has to result in the IndexOutOfBoundsException because         <BR>
 84  * the current frame is the only one in FramesStack.                    <BR>
 85  * <BR>
 86  */
 87

169     private int runThis (String argv[], PrintStream out) {
170 
171         argsHandler     = new ArgumentHandler(argv);
172         logHandler      = new Log(out, argsHandler);
173         Binder binder   = new Binder(argsHandler, logHandler);
174 
175         waitTime        = argsHandler.getWaitTime() * 60000;
176 
177         try {
178             log2("launching a debuggee :");
179             log2("       " + debuggeeName);
180             if (argsHandler.verbose()) {
181                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
182             } else {
183                 debuggee = binder.bindToDebugee(debuggeeName);
184             }
185             if (debuggee == null) {
186                 log3("ERROR: no debuggee launched");
187                 return FAILED;
188             }
189             log2("debuggee launched");
190         } catch ( Exception e ) {
191             log3("ERROR: Exception : " + e);
192             log2("       test cancelled");
193             return FAILED;
194         }
195 
196         debuggee.redirectOutput(logHandler);
197 
198         vm = debuggee.VM();
199 
200         eventQueue = vm.eventQueue();
201         if (eventQueue == null) {
202             log3("ERROR: eventQueue == null : TEST ABORTED");
203             vm.exit(PASS_BASE);
204             return FAILED;
205         }
206 
207         log2("invocation of the method runTest()");
208         switch (runTest()) {
209 
210             case 0 :  log2("test phase has finished normally");
211                       log2("   waiting for the debuggee to finish ...");
212                       debuggee.waitFor();
213 
214                       log2("......getting the debuggee's exit status");
215                       int status = debuggee.getStatus();
216                       if (status != PASS_BASE) {
217                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
218                               status + " != PASS_BASE");
219                           testExitCode = FAILED;
220                       } else {
221                           log2("......debuggee returned expected exit status: " +
222                               status + " == PASS_BASE");
223                       }
224                       break;
225 
226             default : log3("ERROR: runTest() returned unexpected value");
227 
228             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
229                       log2("......forcing: vm.exit();");
230                       testExitCode = FAILED;
231                       try {
232                           vm.exit(PASS_BASE);
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237 
238             case 2 :  log3("test cancelled due to VMDisconnectedException");
239                       log2("......trying: vm.process().destroy();");
240                       testExitCode = FAILED;
241                       try {
242                           Process vmProcess = vm.process();
243                           if (vmProcess != null) {
244                               vmProcess.destroy();
245                           }
246                       } catch ( Exception e ) {
247                           log3("ERROR: Exception : e");
248                       }
249                       break;
250             }
251 
252         return testExitCode;
253     }

258     *               1 - ubnormal end of the test
259     *               2 - VMDisconnectedException while test phase

262     private int runTest() {
263 
264         try {
265             testRun();
266 
267             log2("waiting for VMDeathEvent");
268             getEventSet();
269             if (eventIterator.nextEvent() instanceof VMDeathEvent)
270                 return 0;
271 
272             log3("ERROR: last event is not the VMDeathEvent");
273             return 1;
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }
281

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         eventRManager = vm.eventRequestManager();
288 
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292 
293         cpRequest.enable();
294         vm.resume();
295         getEventSet();
296         cpRequest.disable();
297 
298         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
299         debuggeeClass = event.referenceType();
300 
301         if (!debuggeeClass.name().equals(debuggeeName))
302            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
303 
304         log2("      received: ClassPrepareEvent for debuggeeClass");
305 
306         if ( !vm.canPopFrames() ) {
307             log2("......vm.canPopFrames() == false : test is cancelled");
308             vm.resume();
309             return;
310         }
311 
312         String bPointMethod = "methodForCommunication";
313         String lineForComm  = "lineForComm";
314 
315         log2("......setting BreakpointRequest (bpRequest) in main thread");
316         bpRequest = settingBreakpoint(threadByName("main"),
317                                           debuggeeClass,
318                                           bPointMethod, lineForComm, "zero");
319         log2("bpRequest.enable();");
320         bpRequest.enable();
321 
322     //------------------------------------------------------  testing section
323 
324         log1("     TESTING BEGINS");
325 
326         label0:
327         {
328             vm.resume();
329             breakpointForCommunication();
330 
331             String thread2Name         = "thread2";
332             ThreadReference thread2Ref = threadByName(thread2Name);
333 
334             StackFrame stackFrame0 = null;
335             StackFrame stackFrame1 = null;
336             StackFrame stackFrame2 = null;
337 
338             String breakpointMethod = "breakpointMethod";
339             String breakpointLine    = "breakpointLine";
340 
341             log2("......setting breakpoint in breakpointMethod");
342             breakpointRequest2 = settingBreakpoint(thread2Ref, debuggeeClass,
343                                           breakpointMethod, breakpointLine, "one");
344             log2("......breakpointRequest2.enable();");
345             breakpointRequest2.enable();
346 
347             {  // to get mainThread suspended; otherwise its end results in
348                // no suspention for breakpointRequest2 (bug or not?), end of test,
349                // and VMDisconnectedException
350             thread2Ref.suspend();
351             log2("......eventSet.resume();");
352             eventSet.resume();
353             breakpointForCommunication();
354             log2("bpRequest.disable();");
355             bpRequest.disable();
356             thread2Ref.resume();
357             }
358 
359             log2("......breakpointInMethod(breakpointRequest2);");
360             breakpointInMethod("one");
361 
362             log2("......stackFrame2 = thread2Ref.frame(2);");
363             stackFrame2 = thread2Ref.frame(2);
364 
365             log2("......stackFrame1 = thread2Ref.frame(1);");
366             stackFrame1 = thread2Ref.frame(1);
367 
368             log2("......thread2Ref.popFrames(stackFrame1);");
369             try {
370                 thread2Ref.popFrames(stackFrame1);
371             } catch ( IncompatibleThreadStateException e ) {
372                 log3("ERROR: IncompatibleThreadStateException");
373                 testExitCode = FAILED;
374                 break label0;
375             }
376             log2("......checking up on equality");
377             if ( thread2Ref.frame(0).equals(stackFrame2) ) {
378                 log3("ERROR: thread2Ref.frame(0).equals(stackFrame2)");
379                 testExitCode = FAILED;
380             }
381 
382             log2("......stackFrame1 = thread2Ref.frame(1);");
383             try {
384                 stackFrame1 = thread2Ref.frame(1);
385                 log3("ERROR: no IndexOutOfBoundsException");
386                 testExitCode = FAILED;
387             } catch ( IndexOutOfBoundsException e ) {
388                 log2("      IndexOutOfBoundsException");
389             }
390 
391 
392             log2("......breakpointRequest2.disable();");
393             breakpointRequest2.disable();
394 
395             vm.resume();
396 
397             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
398         }
399         log1("    TESTING ENDS");
400         return;
401     }

403     private ThreadReference threadByName(String name)
404                  throws JDITestRuntimeException {
405 
406         List         all = vm.allThreads();
407         ListIterator li  = all.listIterator();
408 
409         for (; li.hasNext(); ) {
410             ThreadReference thread = (ThreadReference) li.next();
411             if (thread.name().equals(name))
412                 return thread;
413         }
414         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
415     }

417    /*
418     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
419     *                                             String, String, String)
420     *
421     * It sets up a breakpoint at given line number within a given method in a given class
422     * for a given thread.
423     *
424     * Return value: BreakpointRequest object  in case of success
425     *
426     * JDITestRuntimeException   in case of an Exception thrown within the method
427     */
428 
429     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
430                                                   ReferenceType testedClass,
431                                                   String methodName,
432                                                   String bpLine,
433                                                   String property)
434             throws JDITestRuntimeException {
435 
436         log2("......setting up a breakpoint:");
437         log2("       thread: " + thread + "; class: " + testedClass +
438                         "; method: " + methodName + "; line: " + bpLine);
439 
440         List              alllineLocations = null;
441         Location          lineLocation     = null;
442         BreakpointRequest breakpRequest    = null;
443 
444         try {
445             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
446 
447             alllineLocations = method.allLineLocations();
448 
449             int n =
450                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
451             if (n > alllineLocations.size()) {
452                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
453             } else {
454                 lineLocation = (Location) alllineLocations.get(n);
455                 try {
456                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
457                     breakpRequest.putProperty("number", property);
458                     breakpRequest.addThreadFilter(thread);
459                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
460                 } catch ( Exception e1 ) {
461                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
462                     breakpRequest    = null;
463                 }
464             }
465         } catch ( Exception e2 ) {
466             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
467             breakpRequest    = null;
468         }
469 
470         if (breakpRequest == null) {
471             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
472             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
473         }
474 
475         log2("      a breakpoint has been set up");
476         return breakpRequest;
477     }

481     private void getEventSet()
482                  throws JDITestRuntimeException {
483         try {
484 //            log2("       eventSet = eventQueue.remove(waitTime);");
485             eventSet = eventQueue.remove(waitTime);
486             if (eventSet == null) {
487                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
488             }
489 //            log2("       eventIterator = eventSet.eventIterator;");
490             eventIterator = eventSet.eventIterator();
491         } catch ( Exception e ) {
492             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
493         }
494     }

497     private void breakpointForCommunication()
498                  throws JDITestRuntimeException {
499 
500         log2("breakpointForCommunication");
501         getEventSet();
502 
503         if (eventIterator.nextEvent() instanceof BreakpointEvent)
504             return;
505 
506         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
507     }

511     private void breakpointInMethod(String number)
512                  throws JDITestRuntimeException {
513 
514         Event event;
515 
516         getEventSet();
517         event = eventIterator.nextEvent();
518 
519         if ( !(event instanceof BreakpointEvent) )
520             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
521 
522         log2("---->: request().getProperty == " +
523              event.request().getProperty("number"));
524 
525         if ( event.request().getProperty("number").equals(number) )
526             return;
527 
528         throw new JDITestRuntimeException("** UNEXPECTED breakpoint **");
529     }
