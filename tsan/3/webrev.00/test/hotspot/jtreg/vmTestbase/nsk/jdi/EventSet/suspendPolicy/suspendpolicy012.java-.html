 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a VMDeathEvent set,                 <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event set containing one more     <BR>
 49  * VMDeathRequest with suspendPolicy SUSPEND_EVENT_THREAD.      <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the above,                                  <BR>
 66  * the debugger and the debuggee perform the following loop.            <BR>
 67  * - The debugger sets up a VMDeathRequest, resumes                     <BR>
 68  *   the debuggee, and waits for the VMDeathEvent.                      <BR>
 69  * - The debuggee ends to be resulting in the event.                    <BR>
 70  * - Upon getting new event, the debugger                               <BR>
 71  *   performs the check corresponding to the event.                     <BR>
 72  * <BR>
 73  * Note. To inform each other of needed actions, the debugger and       <BR>
 74  *       and the debuggee use debuggee's variable "instruction".        <BR>
 75  * In third phase, at the end,                                          <BR>
 76  * the debuggee changes the value of the "instruction"                  <BR>
 77  * to inform the debugger of checks finished, and both end.             <BR>
 78  * <BR>
 79  */

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : e");
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

252     * Return value: 0 - normal end of the test
253     *               1 - ubnormal end of the test

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
265                 log3("ERROR: last event is not the VMDeathEvent");
266                 return 1;
267             }
268             check();
269 
270             log2("waiting for VMDisconnectEvent");
271             getEventSet();
272             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
273                 log3("ERROR: last event is not the VMDisconnectEvent");
274                 return 1;
275             }
276 
277             return 0;
278 
279         } catch ( VMDisconnectedException e ) {
280             log3("ERROR: VMDisconnectedException : " + e);
281             return 2;
282         } catch ( Exception e ) {
283             log3("ERROR: Exception : " + e);
284             return 1;
285         }
286 
287     }

289     private void testRun()
290                  throws JDITestRuntimeException, Exception {
291 
292         if ( !vm.canRequestVMDeathEvent() ) {
293             log2("......vm.canRequestVMDeathEvent == false :: test cancelled");
294             vm.exit(PASS_BASE);
295             return;
296         }
297 
298 
299         eventRManager = vm.eventRequestManager();
300 
301         log2("......getting ClassPrepareEvent for debuggee's class");
302         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
303         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
304         cpRequest.addClassFilter(debuggeeName);
305         cpRequest.enable();
306         vm.resume();
307         getEventSet();
308         cpRequest.disable();
309 
310         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
311         debuggeeClass = event.referenceType();
312 
313         if (!debuggeeClass.name().equals(debuggeeName))
314            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
315         log2("      received: ClassPrepareEvent for debuggeeClass");
316 
317         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
318 
319         String            bPointMethod = "methodForCommunication";
320         String            lineForComm  = "lineForComm";
321         BreakpointRequest bpRequest;
322         ThreadReference   mainThread = threadByName("main");
323         bpRequest = settingBreakpoint(mainThread,
324                                       debuggeeClass,
325                                       bPointMethod, lineForComm, "zero");
326         bpRequest.enable();
327 
328         vm.resume();
329 
330     //------------------------------------------------------  testing section
331 
332         log1("     TESTING BEGINS");
333 
334         EventRequest eventRequest1 = null;
335         EventRequest eventRequest2 = null;
336         EventRequest eventRequest3 = null;
337 
338         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
339         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
340         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
341         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
342 
343         int policyExpected[] = {    SUSPEND_NONE,
344                                     SUSPEND_THREAD,
345                                     SUSPEND_ALL,
346                                     SUSPEND_THREAD,
347                                     SUSPEND_ALL,
348                                     SUSPEND_ALL,
349                                     SUSPEND_ALL      };
350         int policy = 0;
351 
352         breakpointForCommunication();
353 
354             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
355 
356         eventRequest2 = settingVMDeathRequest ( SUSPEND_THREAD, "VMDeathRequest2");
357         eventRequest2.enable();
358 
359         policyToCheck = policyExpected[1];
360         mainThread.resume();
361 
362             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
363 
364         log1("    TESTING ENDS");
365         return;
366     }

368     private ThreadReference threadByName(String name)
369                  throws JDITestRuntimeException {
370 
371         List         all = vm.allThreads();
372         ListIterator li  = all.listIterator();
373 
374         for (; li.hasNext(); ) {
375             ThreadReference thread = (ThreadReference) li.next();
376             if (thread.name().equals(name))
377                 return thread;
378         }
379         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
380     }

382    /*
383     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
384     *                                             String, String, String)
385     *
386     * It sets up a breakpoint at given line number within a given method in a given class
387     * for a given thread.
388     *
389     * Return value: BreakpointRequest object  in case of success
390     *
391     * JDITestRuntimeException   in case of an Exception thrown within the method
392     */
393 
394     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
395                                                   ReferenceType testedClass,
396                                                   String methodName,
397                                                   String bpLine,
398                                                   String property)
399             throws JDITestRuntimeException {
400 
401         log2("......setting up a breakpoint:");
402         log2("       thread: " + thread + "; class: " + testedClass +
403                         "; method: " + methodName + "; line: " + bpLine);
404 
405         List              alllineLocations = null;
406         Location          lineLocation     = null;
407         BreakpointRequest breakpRequest    = null;
408 
409         try {
410             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
411 
412             alllineLocations = method.allLineLocations();
413 
414             int n =
415                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
416             if (n > alllineLocations.size()) {
417                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
418             } else {
419                 lineLocation = (Location) alllineLocations.get(n);
420                 try {
421                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
422                     breakpRequest.putProperty("number", property);
423                     breakpRequest.addThreadFilter(thread);
424                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
425                 } catch ( Exception e1 ) {
426                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
427                     breakpRequest    = null;
428                 }
429             }
430         } catch ( Exception e2 ) {
431             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
432             breakpRequest    = null;
433         }
434 
435         if (breakpRequest == null) {
436             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
437             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
438         }
439 
440         log2("      a breakpoint has been set up");
441         return breakpRequest;
442     }

445     private void getEventSet()
446                  throws JDITestRuntimeException {
447         try {
448 //            log2("       eventSet = eventQueue.remove(waitTime);");
449             eventSet = eventQueue.remove(waitTime);
450             if (eventSet == null) {
451                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
452             }
453 //            log2("       eventIterator = eventSet.eventIterator;");
454             eventIterator = eventSet.eventIterator();
455         } catch ( Exception e ) {
456             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
457         }
458     }

461     private void breakpointForCommunication()
462                  throws JDITestRuntimeException {
463 
464         log2("breakpointForCommunication");
465         getEventSet();
466 
467         if (eventIterator.nextEvent() instanceof BreakpointEvent)
468             return;
469 
470         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
471     }

475     private VMDeathRequest settingVMDeathRequest( int    suspendPolicy,
476                                                   String property       )
477             throws JDITestRuntimeException {
478         try {
479             log2("......setting up VMDeathRequest:");
480             log2("      suspendPolicy: " + suspendPolicy + "; property: " + property);
481 
482             VMDeathRequest
483             vmdr = eventRManager.createVMDeathRequest();
484             vmdr.putProperty("number", property);
485             vmdr.setSuspendPolicy(suspendPolicy);
486 
487             return vmdr;
488         } catch ( Exception e ) {
489             log3("ERROR: ATTENTION: Exception within settingVMDeathRequest() : " + e);
490             log3("       VMDeathRequest HAS NOT BEEN SET UP");
491             throw new JDITestRuntimeException("** FAILURE to set up a VMDeathRequest **");
492         }
493     }

495     void check() {
496         log2("......checking up on eventSet.suspendPolicy()");
497         int policy = eventSet.suspendPolicy();
498         if (policy != policyToCheck) {
499             log3("ERROR: eventSet.suspendPolicy() != policyExpected");
500             log3("       eventSet.suspendPolicy() == " + policy);
501             log3("       policyExpected           == " + policyToCheck);
502             testExitCode = FAILED;
503         }
504         vm.resume();
505     }
