 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadStartRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadStartRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The cases to check are as follows:
 46  * (1) if this request is currently enabled,                    <BR>
 47  *   - addThreadFilter(ThreadReference) results in throwing     <BR>
 48  *     InvalidRequestStateException;                            <BR>
 49  *   - addThreadFilter(null) results in throwing                <BR>
 50  *     NullPointerException or InvalidRequestStateException;    <BR>
 51  * (2) if this request is currently disabled,                   <BR>
 52  *     addThreadFilter(null) results in throwing                <BR>
 53  *     NullPointerException.                                    <BR>
 54  * <BR>
 55  * The test has three phases and works as follows.              <BR>
 56  * <BR>
 57  * In first phase,                                                      <BR>
 58  * upon launching debuggee's VM which will be suspended,                <BR>
 59  * a debugger waits for the VMStartEvent within a predefined            <BR>
 60  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 61  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 62  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 63  * and waits for the event within the predefined time interval.         <BR>
 64  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 65  * Upon getting the ClassPrepareEvent,                                  <BR>
 66  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 67  * within debuggee's special methodForCommunication().                  <BR>
 68  * <BR>
 69  * In second phase, the debugger creates ThreadStartRequest             <BR>
 70  * and performs the above checks.                                       <BR>
 71  * <BR>
 72  * In third phase, at the end of the test,                              <BR>
 73  * the debuggee changes the value of the "instruction"                  <BR>
 74  * to inform the debugger of checks finished, and both end.             <BR>
 75  * <BR>
 76  */

150     private int runThis (String argv[], PrintStream out) {
151 
152         argsHandler     = new ArgumentHandler(argv);
153         logHandler      = new Log(out, argsHandler);
154         Binder binder   = new Binder(argsHandler, logHandler);
155 
156         waitTime        = argsHandler.getWaitTime() * 60000;
157 
158         try {
159             log2("launching a debuggee :");
160             log2("       " + debuggeeName);
161             if (argsHandler.verbose()) {
162                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
163             } else {
164                 debuggee = binder.bindToDebugee(debuggeeName);
165             }
166             if (debuggee == null) {
167                 log3("ERROR: no debuggee launched");
168                 return FAILED;
169             }
170             log2("debuggee launched");
171         } catch ( Exception e ) {
172             log3("ERROR: Exception : " + e);
173             log2("       test cancelled");
174             return FAILED;
175         }
176 
177         debuggee.redirectOutput(logHandler);
178 
179         vm = debuggee.VM();
180 
181         eventQueue = vm.eventQueue();
182         if (eventQueue == null) {
183             log3("ERROR: eventQueue == null : TEST ABORTED");
184             vm.exit(PASS_BASE);
185             return FAILED;
186         }
187 
188         log2("invocation of the method runTest()");
189         switch (runTest()) {
190 
191             case 0 :  log2("test phase has finished normally");
192                       log2("   waiting for the debuggee to finish ...");
193                       debuggee.waitFor();
194 
195                       log2("......getting the debuggee's exit status");
196                       int status = debuggee.getStatus();
197                       if (status != PASS_BASE) {
198                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
199                               status + " != PASS_BASE");
200                           testExitCode = FAILED;
201                       } else {
202                           log2("......debuggee returned expected exit status: " +
203                               status + " == PASS_BASE");
204                       }
205                       break;
206 
207             default : log3("ERROR: runTest() returned unexpected value");
208 
209             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
210                       log2("......forcing: vm.exit();");
211                       testExitCode = FAILED;
212                       try {
213                           vm.exit(PASS_BASE);
214                       } catch ( Exception e ) {
215                           log3("ERROR: Exception : e");
216                       }
217                       break;
218 
219             case 2 :  log3("test cancelled due to VMDisconnectedException");
220                       log2("......trying: vm.process().destroy();");
221                       testExitCode = FAILED;
222                       try {
223                           Process vmProcess = vm.process();
224                           if (vmProcess != null) {
225                               vmProcess.destroy();
226                           }
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231             }
232 
233         return testExitCode;
234     }

237    /*
238     * Return value: 0 - normal end of the test
239     *               1 - ubnormal end of the test
240     *               2 - VMDisconnectedException while test phase
241     */
242 
243     private int runTest() {
244 
245         try {
246             testRun();
247 
248             log2("waiting for VMDeathEvent");
249             getEventSet();
250             if (eventIterator.nextEvent() instanceof VMDeathEvent)
251                 return 0;
252 
253             log3("ERROR: last event is not the VMDeathEvent");
254             return 1;
255         } catch ( VMDisconnectedException e ) {
256             log3("ERROR: VMDisconnectedException : " + e);
257             return 2;
258         } catch ( Exception e ) {
259             log3("ERROR: Exception : " + e);
260             return 1;
261         }
262 
263     }

266     private void testRun()
267                  throws JDITestRuntimeException, Exception {
268 
269         eventRManager = vm.eventRequestManager();
270 
271         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
272         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
273         cpRequest.addClassFilter(debuggeeName);
274 
275         cpRequest.enable();
276         vm.resume();
277         getEventSet();
278         cpRequest.disable();
279 
280         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
281         debuggeeClass = event.referenceType();
282 
283         if (!debuggeeClass.name().equals(debuggeeName))
284            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
285 
286         log2("      received: ClassPrepareEvent for debuggeeClass");
287 
288         String bPointMethod = "methodForCommunication";
289         String lineForComm  = "lineForComm";
290         BreakpointRequest bpRequest;
291 
292         ThreadReference mainThread = debuggee.threadByNameOrThrow("main");
293 
294         bpRequest = settingBreakpoint(mainThread,
295                                       debuggeeClass,
296                                       bPointMethod, lineForComm, "zero");
297         bpRequest.enable();
298 
299     //------------------------------------------------------  testing section
300 
301         log1("     TESTING BEGINS");
302 
303 
304         for (int i = 0; ; i++) {
305 
306             vm.resume();
307             breakpointForCommunication();
308 
309             int instruction = ((IntegerValue)
310                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
311 
312             if (instruction == 0) {
313                 vm.resume();
314                 break;
315             }
316 
317             log1(":::::: case: # " + i);
318 
319             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
320 
321             log2("......setting up ThreadStartRequest");
322             log2("...... ThreadStartRequest tsr1 = eventRManager.createThreadStartRequest();");
323             ThreadStartRequest tsr1 = eventRManager.createThreadStartRequest();
324             tsr1.addCountFilter(1);
325             tsr1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
326             tsr1.putProperty("number", "ThreadStartRequest1");
327 
328             log2("...... tsr1.enable();");
329             tsr1.enable();
330 
331             try {
332                 log2("...... tsr1.addThreadFilter(mainThread);");
333                 log2("         InvalidRequestStateException is expected");
334                 tsr1.addThreadFilter(mainThread);
335                 log3("ERROR: no InvalidRequestStateException ");
336                 testExitCode = FAILED;
337             } catch ( InvalidRequestStateException e ) {
338                 log2("          InvalidRequestStateException");
339             } catch ( Exception e ) {
340                 log3("ERROR: unexpected Exception : " + e);
341                 testExitCode = FAILED;
342             }
343             try {
344                 log2("...... tsr1.addThreadFilter(null);");
345                 log2("         NullPointerException or InvalidRequestStateException is expected");
346                 tsr1.addThreadFilter(null);
347                 log3("ERROR: no Exception thrown");
348                 testExitCode = FAILED;
349             } catch ( NullPointerException e ) {
350                 log2("          NullPointerException");
351             } catch ( InvalidRequestStateException e ) {
352                 log2("          InvalidRequestStateException");
353             } catch ( Exception e ) {
354                 log3("ERROR: unexpected Exception : " + e);
355                 testExitCode = FAILED;
356             }
357 
358             log2("...... tsr1.disable()");
359             tsr1.disable();
360 
361             try {
362                 log2("...... tsr1.addThreadFilter(null);");
363                 log2("         NullPointerException is expected");
364                 tsr1.addThreadFilter(null);
365                 log3("ERROR: no NullPointerException ");
366                 testExitCode = FAILED;
367             } catch ( NullPointerException e ) {
368                 log2("          NullPointerException");
369             } catch ( Exception e ) {
370                 log3("ERROR: unexpected Exception : " + e);
371                 testExitCode = FAILED;
372             }
373             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
374         }
375         log1("    TESTING ENDS");
376         return;
377     }

379    /*
380     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
381     *                                             String, String, String)
382     *
383     * It sets up a breakpoint at given line number within a given method in a given class
384     * for a given thread.
385     *
386     * Return value: BreakpointRequest object  in case of success
387     *
388     * JDITestRuntimeException   in case of an Exception thrown within the method
389     */
390 
391     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
392                                                   ReferenceType testedClass,
393                                                   String methodName,
394                                                   String bpLine,
395                                                   String property)
396             throws JDITestRuntimeException {
397 
398         log2("......setting up a breakpoint:");
399         log2("       thread: " + thread + "; class: " + testedClass +
400                         "; method: " + methodName + "; line: " + bpLine);
401 
402         List              alllineLocations = null;
403         Location          lineLocation     = null;
404         BreakpointRequest breakpRequest    = null;
405 
406         try {
407             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
408 
409             alllineLocations = method.allLineLocations();
410 
411             int n =
412                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
413             if (n > alllineLocations.size()) {
414                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
415             } else {
416                 lineLocation = (Location) alllineLocations.get(n);
417                 try {
418                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
419                     breakpRequest.putProperty("number", property);
420                     breakpRequest.addThreadFilter(thread);
421                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
422                 } catch ( Exception e1 ) {
423                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
424                     breakpRequest    = null;
425                 }
426             }
427         } catch ( Exception e2 ) {
428             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
429             breakpRequest    = null;
430         }
431 
432         if (breakpRequest == null) {
433             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
434             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
435         }
436 
437         log2("      a breakpoint has been set up");
438         return breakpRequest;
439     }

442     private void getEventSet()
443                  throws JDITestRuntimeException {
444         try {
445 //            log2("       eventSet = eventQueue.remove(waitTime);");
446             eventSet = eventQueue.remove(waitTime);
447             if (eventSet == null) {
448                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
449             }
450 //            log2("       eventIterator = eventSet.eventIterator;");
451             eventIterator = eventSet.eventIterator();
452         } catch ( Exception e ) {
453             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
454         }
455     }

458     private void breakpointForCommunication()
459                  throws JDITestRuntimeException {
460 
461         log2("breakpointForCommunication");
462         while (true) {
463             getEventSet();
464             while (eventIterator.hasNext()) {
465                 Event event = eventIterator.nextEvent();
466                 if (event instanceof BreakpointEvent) {
467                     return;
468                 } else if (event instanceof ThreadStartEvent) {
469                     // It might be the case that while the thread start request was enabled
470                     // some threads not related to the test ( e.g. JVMCI threads) were started
471                     // and generated thread start events. We ignore these thread start events
472                     // and keep waiting for a breakpoint event.
473                     ThreadStartEvent tse = (ThreadStartEvent) event;
474                     log2("ThreadStartEvent is received while waiting for a breakpoint" +
475                             " event, thread: : " + tse.thread().name());
476                     continue;
477                 }
478                 throw new JDITestRuntimeException("** event IS NOT a breakpoint or a thread start **");
479             }
480         }
481     }
