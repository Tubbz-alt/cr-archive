 36 #define STATUS_FAILED 2
 37 #define PASSED 0
 38 
 39 /* tested methods */
 40 #define METH_NUM 4
 41 static const char *METHODS[][2] = {
 42     { "bpMethod", "()V" },
 43     { "nativeMethod", "()V" },
 44     { "anotherNativeMethod", "(I)V" },
 45     { "runThis", "([Ljava/lang/String;Ljava/io/PrintStream;)I" }
 46 };
 47 
 48 /* event counters for the tested methods and expected numbers
 49  of the events */
 50 static volatile long stepEv[][2] = {
 51     { 0, 1 },
 52     { 0, 0 },
 53     { 0, 0 },
 54     { 0, 1 }
 55 };
 56 
 57 static const char *CLASS_SIG =
 58     "Lnsk/jvmti/SingleStep/singlestep003;";
 59 
 60 static volatile jint result = PASSED;
 61 static jvmtiEnv *jvmti = NULL;
 62 static jvmtiEventCallbacks callbacks;
 63 
 64 static volatile int callbacksEnabled = NSK_FALSE;
 65 static jrawMonitorID agent_lock;
 66 
 67 static void setBP(jvmtiEnv *jvmti_env, JNIEnv *env, jclass klass) {
 68     jmethodID mid;
 69 
 70     if (!NSK_JNI_VERIFY(env, (mid = env->GetMethodID(klass, METHODS[0][0], METHODS[0][1])) != NULL))
 71         env->FatalError("failed to get ID for the java method\n");
 72 
 73     if (!NSK_JVMTI_VERIFY(jvmti_env->SetBreakpoint(mid, 0)))
 74         env->FatalError("failed to set breakpoint\n");
 75 }

 77 /** callback functions **/
 78 void JNICALL
 79 ClassLoad(jvmtiEnv *jvmti_env, JNIEnv *env, jthread thread, jclass klass) {
 80     char *sig, *generic;
 81 
 82     jvmti->RawMonitorEnter(agent_lock);
 83 
 84     if (!callbacksEnabled) {
 85         jvmti->RawMonitorExit(agent_lock);
 86         return;
 87     }
 88 
 89     if (!NSK_JVMTI_VERIFY(jvmti_env->GetClassSignature(klass, &sig, &generic)))
 90         env->FatalError("failed to obtain a class signature\n");
 91 
 92     if (sig != NULL && (strcmp(sig, CLASS_SIG) == 0)) {
 93         NSK_DISPLAY1(
 94             "ClassLoad event received for the class \"%s\"\n"
 95             "\tsetting breakpoint ...\n",
 96             sig);
 97         setBP(jvmti_env, env, klass);
 98     }
 99 
100     jvmti->RawMonitorExit(agent_lock);
101 }

103 void JNICALL
104 VMStart(jvmtiEnv *jvmti_env, JNIEnv* jni_env) {
105     jvmti->RawMonitorEnter(agent_lock);
106 
107     callbacksEnabled = NSK_TRUE;
108 
109     jvmti->RawMonitorExit(agent_lock);
110 }

114 VMDeath(jvmtiEnv *jvmti_env, JNIEnv* jni_env) {

118

122 void JNICALL
123 Breakpoint(jvmtiEnv *jvmti_env, JNIEnv *env, jthread thr, jmethodID method,
124         jlocation loc) {
125     jclass klass;
126     char *sig, *generic;
127 
128     jvmti->RawMonitorEnter(agent_lock);
129 
130     if (!callbacksEnabled) {
131         jvmti->RawMonitorExit(agent_lock);
132         return;
133     }
134 
135     NSK_DISPLAY0("Breakpoint event received\n");
136     if (!NSK_JVMTI_VERIFY(jvmti_env->GetMethodDeclaringClass(method, &klass)))
137         NSK_COMPLAIN0("TEST FAILURE: unable to get method declaring class\n\n");
138 
139     if (!NSK_JVMTI_VERIFY(jvmti_env->GetClassSignature(klass, &sig, &generic)))
140         env->FatalError("Breakpoint: failed to obtain a class signature\n");
141 
142     if (sig != NULL && (strcmp(sig, CLASS_SIG) == 0)) {
143         NSK_DISPLAY1("method declaring class \"%s\"\n\tenabling SingleStep events ...\n",
144             sig);
145         if (!NSK_JVMTI_VERIFY(jvmti_env->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_SINGLE_STEP, thr))) {
146             result = STATUS_FAILED;
147             NSK_COMPLAIN0("TEST FAILURE: cannot enable SingleStep events\n\n");
148         }
149     } else {
150         result = STATUS_FAILED;
151         NSK_COMPLAIN1("TEST FAILURE: unexpected breakpoint event in method of class \"%s\"\n\n",
152             sig);
153     }
154 
155     jvmti->RawMonitorExit(agent_lock);
156 }

158 void JNICALL
159 SingleStep(jvmtiEnv *jvmti_env, JNIEnv* jni_env, jthread thread,
160         jmethodID method, jlocation location) {
161     jclass klass;
162     char *sig, *generic, *methNam, *methSig;
163     int i;
164 
165     if (result == STATUS_FAILED) {
166         return;
167     }
168 
169     NSK_DISPLAY0(">>>> SingleStep event received\n");
170 
171     if (!NSK_JVMTI_VERIFY(jvmti_env->GetMethodName(method, &methNam, &methSig, NULL))) {
172         result = STATUS_FAILED;
173         NSK_COMPLAIN0("TEST FAILED: unable to get method name during SingleStep callback\n\n");
174         return;
175     }
176     if (!NSK_JVMTI_VERIFY(jvmti_env->GetMethodDeclaringClass(method, &klass))) {
177         result = STATUS_FAILED;
178         NSK_COMPLAIN0("TEST FAILED: unable to get method declaring class during SingleStep callback\n\n");
179         return;
180     }
181     if (!NSK_JVMTI_VERIFY(jvmti_env->GetClassSignature(klass, &sig, &generic))) {
182         result = STATUS_FAILED;
183         NSK_COMPLAIN0("TEST FAILED: unable to obtain a class signature during SingleStep callback\n\n");
184         return;
185     }
186 
187     if (sig != NULL) {
188         if (stepEv[METH_NUM-1][0] == 1) {
189             result = STATUS_FAILED;
190             NSK_COMPLAIN0("TEST FAILED: SingleStep event received after disabling the event generation\n\n");
191             return;
192         }
193 
194         for (i=0; i<METH_NUM; i++) {
195             if ((strcmp(methNam,METHODS[i][0]) == 0) &&
196                     (strcmp(methSig,METHODS[i][1]) == 0) &&
197                     (strcmp(sig,CLASS_SIG) == 0)) {
198                 stepEv[i][0]++;
199 
200                 if (stepEv[i][1] == 1) {
201                     NSK_DISPLAY3(
202                         "CHECK PASSED: SingleStep event received for the method:\n"
203                         "\t \"%s %s\" of class \"%s\"\n"
204                         "\tas expected\n",
205                         methNam, methSig, sig);
206                 } else {
207                     result = STATUS_FAILED;
208                     NSK_COMPLAIN3(
209                         "TEST FAILED: SingleStep event received for the method:\n"
210                         "\t \"%s %s\" of class \"%s\"\n",
211                         methNam, methSig, sig);
212                 }
213 
214                 if (i == (METH_NUM-1)) {
215                     NSK_DISPLAY0("Disabling the single step event generation\n");
216                     if (!NSK_JVMTI_VERIFY(jvmti_env->SetEventNotificationMode(JVMTI_DISABLE, JVMTI_EVENT_SINGLE_STEP, thread))) {
217                         result = STATUS_FAILED;
218                         NSK_COMPLAIN0("TEST FAILED: cannot disable SingleStep events\n\n");
219                     }
220                 }
221             }
222         }
223     }
224 
225     if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char*) methNam))) {
226         result = STATUS_FAILED;
227         NSK_COMPLAIN0("TEST FAILED: unable to deallocate memory pointed to method name\n\n");
228     }
229     if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char*) methSig))) {
230         result = STATUS_FAILED;
231         NSK_COMPLAIN0("TEST FAILED: unable to deallocate memory pointed to method signature\n\n");
232     }
233 
234     NSK_DISPLAY0("<<<<\n\n");
235 }

257 JNIEXPORT jint JNICALL
258 Java_nsk_jvmti_SingleStep_singlestep003_check(
259         JNIEnv *env, jobject obj) {
260     int i;
261 
262     for (i=0; i<METH_NUM; i++)
263         if (stepEv[i][0] == 0) {
264             if (stepEv[i][1] == 0) {
265                 NSK_DISPLAY1("CHECK PASSED: no SingleStep events for the method \"%s\" as expected\n\n",
266                     METHODS[i][0]);
267             }
268             else {
269                 result = STATUS_FAILED;
270                 NSK_COMPLAIN1("TEST FAILED: no SingleStep events for the method \"%s\"\n\n",
271                     METHODS[i][0]);
272             }
273         }
274 
275     return result;
276 }

289 jint Agent_Initialize(JavaVM *jvm, char *options, void *reserved) {
290     jvmtiCapabilities caps;
291 
292     /* init framework and parse options */
293     if (!NSK_VERIFY(nsk_jvmti_parseOptions(options)))
294         return JNI_ERR;
295 
296     /* create JVMTI environment */
297     if (!NSK_VERIFY((jvmti =
298             nsk_jvmti_createJVMTIEnv(jvm, reserved)) != NULL))
299         return JNI_ERR;
300 
301     /* add capability to generate compiled method events */
302     memset(&caps, 0, sizeof(jvmtiCapabilities));
303     caps.can_generate_breakpoint_events = 1;
304     caps.can_generate_single_step_events = 1;
305     if (!NSK_JVMTI_VERIFY(jvmti->AddCapabilities(&caps)))
306         return JNI_ERR;
307 
308     if (!NSK_JVMTI_VERIFY(jvmti->GetCapabilities(&caps)))
309         return JNI_ERR;
310     if (!caps.can_generate_single_step_events)
311         NSK_DISPLAY0("Warning: generation of single step events is not implemented\n");
312 
313     /* set event callback */
314     NSK_DISPLAY0("setting event callbacks ...\n");
315     (void) memset(&callbacks, 0, sizeof(callbacks));
316     callbacks.ClassLoad = &ClassLoad;
317     callbacks.Breakpoint = &Breakpoint;
318     callbacks.SingleStep = &SingleStep;
319     callbacks.VMStart = &VMStart;
320     callbacks.VMDeath = &VMDeath;
321 
322     if (!NSK_JVMTI_VERIFY(jvmti->SetEventCallbacks(&callbacks, sizeof(callbacks))))
323         return JNI_ERR;
324 
325     NSK_DISPLAY0("setting event callbacks done\nenabling JVMTI events ...\n");
326     if (!NSK_JVMTI_VERIFY(jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_VM_START, NULL)))
327         return JNI_ERR;
328     if (!NSK_JVMTI_VERIFY(jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_VM_DEATH, NULL)))
329         return JNI_ERR;
330     if (!NSK_JVMTI_VERIFY(jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_CLASS_LOAD, NULL)))
331         return JNI_ERR;
332     if (!NSK_JVMTI_VERIFY(jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_BREAKPOINT, NULL)))
333         return JNI_ERR;
334     NSK_DISPLAY0("enabling the events done\n\n");
335 
336     if (jvmti->CreateRawMonitor("agent_lock", &agent_lock) != JVMTI_ERROR_NONE) {
337         return JNI_ERR;
338     }
339 
340     return JNI_OK;
341 }
