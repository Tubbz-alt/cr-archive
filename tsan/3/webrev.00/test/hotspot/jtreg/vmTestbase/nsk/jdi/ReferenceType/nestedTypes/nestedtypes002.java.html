 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.nestedTypes()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Cases for testing include ArrayType and primitive classes.   <BR>
 46  * <BR>
 47  * The test works as follows.                                   <BR>
 48  * Upon launching debuggee's VM which will be suspended,        <BR>
 49  * a debugger waits for the VMStartEvent within a predefined    <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.<BR>
 51  * Upon getting the VMStartEvent, it makes the request          <BR>
 52  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,  <BR>
 53  * resumes the VM, and waits for the event within the predefined<BR>
 54  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 57  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 58  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 59  * to be suspended and to inform the debugger with the event.           <BR>
 60  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 61  * At the end, the debuggee changes the value of the "instruction"      <BR>
 62  * to inform the debugger of checks finished, and both end.             <BR>
 63  */
 64

133     private int runThis (String argv[], PrintStream out) {
134 
135         argsHandler     = new ArgumentHandler(argv);
136         logHandler      = new Log(out, argsHandler);
137         Binder binder   = new Binder(argsHandler, logHandler);
138 
139         waitTime        = argsHandler.getWaitTime() * 60000;
140 
141         try {
142             log2("launching a debuggee :");
143             log2("       " + debuggeeName);
144             if (argsHandler.verbose()) {
145                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
146             } else {
147                 debuggee = binder.bindToDebugee(debuggeeName);
148             }
149             if (debuggee == null) {
150                 log3("ERROR: no debuggee launched");
151                 return FAILED;
152             }
153             log2("debuggee launched");
154         } catch ( Exception e ) {
155             log3("ERROR: Exception : " + e);
156             log2("       test cancelled");
157             return FAILED;
158         }
159 
160         debuggee.redirectOutput(logHandler);
161 
162         vm = debuggee.VM();
163 
164         eventQueue = vm.eventQueue();
165         if (eventQueue == null) {
166             log3("ERROR: eventQueue == null : TEST ABORTED");
167             vm.exit(PASS_BASE);
168             return FAILED;
169         }
170 
171         log2("invocation of the method runTest()");
172         switch (runTest()) {
173 
174             case 0 :  log2("test phase has finished normally");
175                       log2("   waiting for the debuggee to finish ...");
176                       debuggee.waitFor();
177 
178                       log2("......getting the debuggee's exit status");
179                       int status = debuggee.getStatus();
180                       if (status != PASS_BASE) {
181                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
182                               status + " != PASS_BASE");
183                           testExitCode = FAILED;
184                       } else {
185                           log2("......debuggee returned expected exit status: " +
186                               status + " == PASS_BASE");
187                       }
188                       break;
189 
190             default : log3("ERROR: runTest() returned unexpected value");
191 
192             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
193                       log2("......forcing: vm.exit();");
194                       testExitCode = FAILED;
195                       try {
196                           vm.exit(PASS_BASE);
197                       } catch ( Exception e ) {
198                           log3("ERROR: Exception : e");
199                       }
200                       break;
201 
202             case 2 :  log3("test cancelled due to VMDisconnectedException");
203                       log2("......trying: vm.process().destroy();");
204                       testExitCode = FAILED;
205                       try {
206                           Process vmProcess = vm.process();
207                           if (vmProcess != null) {
208                               vmProcess.destroy();
209                           }
210                       } catch ( Exception e ) {
211                           log3("ERROR: Exception : e");
212                       }
213                       break;
214             }
215 
216         return testExitCode;
217     }

220    /*
221     * Return value: 0 - normal end of the test
222     *               1 - ubnormal end of the test
223     *               2 - VMDisconnectedException while test phase
224     */
225 
226     private int runTest() {
227 
228         try {
229             testRun();
230 
231             log2("waiting for VMDeathEvent");
232             getEventSet();
233             if (eventIterator.nextEvent() instanceof VMDeathEvent)
234                 return 0;
235 
236             log3("ERROR: last event is not the VMDeathEvent");
237             return 1;
238         } catch ( VMDisconnectedException e ) {
239             log3("ERROR: VMDisconnectedException : " + e);
240             return 2;
241         } catch ( Exception e ) {
242             log3("ERROR: Exception : " + e);
243             return 1;
244         }
245 
246     }

248     private static BreakpointEvent bpEvent;
249 
250     private void testRun()
251                  throws JDITestRuntimeException, Exception {
252 
253         eventRManager = vm.eventRequestManager();
254 
255         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
256         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
257         cpRequest.addClassFilter(debuggeeName);
258 
259         cpRequest.enable();
260         vm.resume();
261         getEventSet();
262         cpRequest.disable();
263 
264         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
265         debuggeeClass = event.referenceType();
266 
267         if (!debuggeeClass.name().equals(debuggeeName))
268            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
269 
270         log2("      received: ClassPrepareEvent for debuggeeClass");
271 
272         String bPointMethod = "methodForCommunication";
273         String lineForComm  = "lineForComm";
274         BreakpointRequest bpRequest;
275 
276         try {
277             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
278                                           debuggeeClass,
279                                           bPointMethod, lineForComm, "zero");
280         } catch ( Exception e ) {
281             throw e;
282         }
283         bpRequest.enable();
284 
285     //------------------------------------------------------  testing section
286 
287         log1("     TESTING BEGINS");
288 
289         for (int i = 0; ; i++) {
290 
291             vm.resume();
292             breakpointForCommunication();
293 
294             int instruction = ((IntegerValue)
295                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
296 
297             if (instruction == 0) {
298                 vm.resume();
299                 break;
300             }
301 
302             log1("  new check: # " + i);
303 
304             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
305 
306             ReferenceType testedType  = null;
307             List          nestedTypes = null;
308             String        typeForCheck;
309 
310 
311             log2("----- Case for testing: ArrayType");
312 
313             typeForCheck = mName + ".nestedtypes002aTestClass[]";
314             log2("......typeForCheck: " + typeForCheck);
315 
316             log2("      getting: List classList = vm.classesByName(typeForCheck);");
317             List classes = vm.classesByName(typeForCheck);
318             if (classes.size() != 1) {
319                 log3("ERROR: classes.size() != 1  : " + classes.size());
320                 testExitCode = FAILED;
321                 continue;
322             }
323 
324             log2("      getting: ReferenceType testedType  = (ReferenceType) classList.get(0);");
325             testedType  = (ReferenceType) classes.get(0);
326 
327             log2("      getting: List nestedTypes = testedType.nestedTypes();");
328             nestedTypes = testedType.nestedTypes();
329             if (nestedTypes.size() != 0) {
330                 log3("ERROR: nestedTypes.size() != 0 : " + nestedTypes.size());
331                 testExitCode = FAILED;
332                 continue;
333             }
334 
335             log2("----- Cases for testing: primitive classes");
336 
337             String names[] = {
338                                  "bl",
339                                  "bt",
340                                  "ch",
341                                  "db",
342                                  "fl",
343                                  "in",
344                                  "ln",
345                                  "sh"
346                              };
347 
348             Method testMethod = (Method)debuggeeClass.methodsByName("main").get(0);
349             // bpEvent should be assigned while getting of BreakpointEvent
350             if (bpEvent == null) {
351                 throw new JDITestRuntimeException("bpEvent is null");
352             }
353 
354             // get stack farme with main method
355             StackFrame frame = null;
356             try {
357                 frame = bpEvent.thread().frame(1);
358             } catch (Exception e) {
359                 throw new JDITestRuntimeException("Unexpected exception while getting stack frame: " + e );
360             }
361             if (frame.location().method() != testMethod) {
362                 throw new JDITestRuntimeException("Cannot take frame of main method");
363             }
364 
365             for (int i1 = 0; i1 < names.length; i1++) {
366 
367                 log2("......check for field : " + names[i]);
368                 LocalVariable lVar = null;
369                 try {
370                     lVar = frame.visibleVariableByName(names[i1]);
371                 } catch (Exception e ) {
372                     throw new JDITestRuntimeException("Unexpected exception while searching for field " + names[i1] + " : " + e  );
373                 }
374                 Value val = null;
375                 try {
376                     val = frame.getValue(lVar);
377                 } catch (Exception e ) {
378                     throw new JDITestRuntimeException("Unexpected exception while getting value of field " + names[i1] + " : " + e  );
379                 }
380 
381                 testedType = ((ClassObjectReference)val).reflectedType();
382 
383                 log2("      checking nestedTypes() for ClassObjectReference of : " + testedType.name());
384                 nestedTypes = testedType.nestedTypes();
385                 if (nestedTypes.size() != 0) {
386                     log3("ERROR: nestedTypes.size() != 0 : " + nestedTypes.size());
387                     testExitCode = FAILED;
388                     continue;
389                 }
390             }
391 
392             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
393         }
394         log1("    TESTING ENDS");
395         return;
396     }

398    /*
399     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
400     *                                             String, String, String)
401     *
402     * It sets up a breakpoint at given line number within a given method in a given class
403     * for a given thread.
404     *
405     * Return value: BreakpointRequest object  in case of success
406     *
407     * JDITestRuntimeException   in case of an Exception thrown within the method
408     */
409 
410     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
411                                                   ReferenceType testedClass,
412                                                   String methodName,
413                                                   String bpLine,
414                                                   String property)
415             throws JDITestRuntimeException {
416 
417         log2("......setting up a breakpoint:");
418         log2("       thread: " + thread + "; class: " + testedClass +
419                         "; method: " + methodName + "; line: " + bpLine);
420 
421         List              alllineLocations = null;
422         Location          lineLocation     = null;
423         BreakpointRequest breakpRequest    = null;
424 
425         try {
426             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
427 
428             alllineLocations = method.allLineLocations();
429 
430             int n =
431                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
432             if (n > alllineLocations.size()) {
433                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
434             } else {
435                 lineLocation = (Location) alllineLocations.get(n);
436                 try {
437                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
438                     breakpRequest.putProperty("number", property);
439                     breakpRequest.addThreadFilter(thread);
440                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
441                 } catch ( Exception e1 ) {
442                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
443                     breakpRequest    = null;
444                 }
445             }
446         } catch ( Exception e2 ) {
447             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
448             breakpRequest    = null;
449         }
450 
451         if (breakpRequest == null) {
452             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
453             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
454         }
455 
456         log2("      a breakpoint has been set up");
457         return breakpRequest;
458     }

461     private void getEventSet()
462                  throws JDITestRuntimeException {
463         try {
464 //            log2("       eventSet = eventQueue.remove(waitTime);");
465             eventSet = eventQueue.remove(waitTime);
466             if (eventSet == null) {
467                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
468             }
469 //            log2("       eventIterator = eventSet.eventIterator;");
470             eventIterator = eventSet.eventIterator();
471         } catch ( Exception e ) {
472             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
473         }
474     }

477     private void breakpointForCommunication()
478                  throws JDITestRuntimeException {
479 
480         log2("breakpointForCommunication");
481         getEventSet();
482 
483         Event event = eventIterator.nextEvent();
484         if ( event instanceof BreakpointEvent) {
485             bpEvent = (BreakpointEvent)event;
486             return;
487         }
488         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
489     }
