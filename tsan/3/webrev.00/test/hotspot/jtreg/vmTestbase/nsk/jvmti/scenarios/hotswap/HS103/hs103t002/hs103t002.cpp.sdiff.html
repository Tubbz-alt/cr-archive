109 jint Agent_Initialize(JavaVM *vm, char *options, void *reserved) {
110   jint rc;
111 
112   nsk_printf("Agent:: VM.. Started..\n");
113 
114   rc = vm->GetEnv((void **)&jvmti, JVMTI_VERSION_1_1);
115   if (rc != JNI_OK) {
116     nsk_printf("Agent:: Could not load JVMTI interface \n");
117     return JNI_ERR;
118   } else {
119     jvmtiCapabilities caps;
120     if (nsk_jvmti_parseOptions(options) == NSK_FALSE) {
121       nsk_printf("# error agent Failed to parse options \n");
122       return JNI_ERR;
123     }
124     memset(&caps, 0, sizeof(caps));
125     caps.can_redefine_classes = 1;
126     jvmti->AddCapabilities(&caps);
127   }
128   return JNI_OK;
129 }
