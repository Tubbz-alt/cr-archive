 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a ModificationWatchpointEvent set,  <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a ModificationWatchpointRequest, resumes      <BR>
 70  *   the debuggee, and waits for the ModificationWatchpointEvent.       <BR>
 71  * - The debuggee invokes the special method which makes modification   <BR>
 72  *   of a predefined variable to be resulting in the event.             <BR>
 73  * - Upon getting new event, the debugger                               <BR>
 74  *   performs the check corresponding to the event.                     <BR>
 75  * <BR>
 76  * Note. To inform each other of needed actions, the debugger and       <BR>
 77  *       and the debuggee use debuggee's variable "instruction".        <BR>
 78  * In third phase, at the end,                                          <BR>
 79  * the debuggee changes the value of the "instruction"                  <BR>
 80  * to inform the debugger of checks finished, and both end.             <BR>
 81  * <BR>
 82  */
 83

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
260                 log3("ERROR: last event is not the VMDeathEvent");
261                 return 1;
262             }
263 
264             log2("waiting for VMDisconnectEvent");
265             getEventSet();
266             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
267                 log3("ERROR: last event is not the VMDisconnectEvent");
268                 return 1;
269             }
270 
271             return 0;
272 
273         } catch ( VMDisconnectedException e ) {
274             log3("ERROR: VMDisconnectedException : " + e);
275             return 2;
276         } catch ( Exception e ) {
277             log3("ERROR: Exception : " + e);
278             return 1;
279         }
280 
281     }

283     private void testRun()
284                  throws JDITestRuntimeException, Exception {
285 
286         if ( !vm.canWatchFieldModification() ) {
287             log2("......vm.canWatchFieldModification == false :: test cancelled");
288             vm.exit(PASS_BASE);
289             return;
290         }
291 
292         eventRManager = vm.eventRequestManager();
293 
294         log2("......getting ClassPrepareEvent for debuggee's class");
295         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
296         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
297         cpRequest.addClassFilter(debuggeeName);
298         cpRequest.enable();
299         vm.resume();
300         getEventSet();
301         cpRequest.disable();
302 
303         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
304         debuggeeClass = event.referenceType();
305 
306         if (!debuggeeClass.name().equals(debuggeeName))
307            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
311 
312         String            bPointMethod = "methodForCommunication";
313         String            lineForComm  = "lineForComm";
314         BreakpointRequest bpRequest;
315         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
316         bpRequest = settingBreakpoint(mainThread,
317                                       debuggeeClass,
318                                       bPointMethod, lineForComm, "zero");
319         bpRequest.enable();
320 
321         vm.resume();
322 
323     //------------------------------------------------------  testing section
324 
325         log1("     TESTING BEGINS");
326 
327         EventRequest eventRequest1 = null;
328         EventRequest eventRequest2 = null;
329         EventRequest eventRequest3 = null;
330 
331         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
332         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
333         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
334         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
335 
336         int policyExpected[] = {    SUSPEND_NONE,
337                                     SUSPEND_THREAD,
338                                     SUSPEND_ALL,
339                                     SUSPEND_THREAD,
340                                     SUSPEND_ALL,
341                                     SUSPEND_ALL,
342                                     SUSPEND_ALL      };
343         int policy = 0;
344 
345 
346         ReferenceType testClassReference = null;
347 
348         String fieldName = "var2";
349 
350 
351         for (int i = 0; ; i++) {
352 
353 
354             breakpointForCommunication();
355 
356             int instruction = ((IntegerValue)
357                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
358             if (instruction == 0) {
359                 vm.resume();
360                 break;
361             }
362 
363 
364             log1(":::::: case: # " + i);
365 
366             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
367 
368             switch (i) {
369 
370               case 0:
371                       testClassReference =
372                            (ReferenceType) vm.classesByName(testedClassName).get(0);
373 
374                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
375                                              testClassReference, fieldName,
376                                              SUSPEND_NONE, "ModificationWatchpointRequest1");
377                       eventRequest1.enable();
378                       break;
379 
380               case 1:
381                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
382                                              testClassReference, fieldName,
383                                              SUSPEND_THREAD, "ModificationWatchpointRequest2");
384                       eventRequest2.enable();
385                       break;
386 
387               case 2:
388                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
389                                              testClassReference, fieldName,
390                                              SUSPEND_ALL, "ModificationWatchpointRequest3");
391                       eventRequest3.enable();
392                       break;
393 
394               case 3:
395                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
396                                              testClassReference, fieldName,
397                                              SUSPEND_NONE, "ModificationWatchpointRequest4");
398                       eventRequest1.enable();
399                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
400                                              testClassReference, fieldName,
401                                              SUSPEND_THREAD, "ModificationWatchpointRequest5");
402                       eventRequest2.enable();
403                       break;
404 
405               case 4:
406                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
407                                              testClassReference, fieldName,
408                                              SUSPEND_NONE, "ModificationWatchpointRequest6");
409                       eventRequest1.enable();
410                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
411                                              testClassReference, fieldName,
412                                              SUSPEND_ALL, "ModificationWatchpointRequest7");
413                       eventRequest3.enable();
414                       break;
415 
416               case 5:
417                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
418                                              testClassReference, fieldName,
419                                              SUSPEND_THREAD, "ModificationWatchpointRequest8");
420                       eventRequest2.enable();
421                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
422                                              testClassReference, fieldName,
423                                              SUSPEND_ALL, "ModificationWatchpointRequest9");
424                       eventRequest3.enable();
425                       break;
426 
427               case 6:
428                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
429                                              testClassReference, fieldName,
430                                              SUSPEND_NONE, "ModificationWatchpointRequest10");
431                       eventRequest1.enable();
432                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
433                                              testClassReference, fieldName,
434                                              SUSPEND_THREAD, "ModificationWatchpointRequest11");
435                       eventRequest2.enable();
436                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
437                                              testClassReference, fieldName,
438                                              SUSPEND_ALL, "ModificationWatchpointRequest12");
439                       eventRequest3.enable();
440                       break;
441 
442 
443               default:
444                       throw new JDITestRuntimeException("** default case 2 **");
445             }
446 
447             log2("......waiting for new AccessWatchpointEvent : " + i);
448             mainThread.resume();
449             getEventSet();
450 
451             Event newEvent = eventIterator.nextEvent();
452             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
453             if ( !( newEvent instanceof ModificationWatchpointEvent)) {
454                  log3("ERROR: new event is not ModificationWatchpointEvent");
455                  testExitCode = FAILED;
456             } else {
457                 log2("       Event is instanceof ModificationWatchpointEvent");
458                 policy = eventSet.suspendPolicy();
459                 if (policy != policyExpected[i]) {
460                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
461                     log3("       eventSet.suspendPolicy() == " + policy);
462                     log3("       policyExpected           == " + policyExpected[i]);
463                     testExitCode = FAILED;
464                 }
465             }
466 
467             switch (i) {
468               case 0: eventRequest1.disable();                          break;
469               case 1: eventRequest2.disable();                          break;
470               case 2: eventRequest3.disable();                          break;
471               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
472               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
473               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
474               case 6: eventRequest1.disable(); eventRequest2.disable();
475                       eventRequest3.disable();                          break;
476            }
477 
478             switch (policy) {
479               case SUSPEND_NONE   :                      break;
480               case SUSPEND_THREAD : mainThread.resume(); break;
481               case SUSPEND_ALL    : vm.resume();         break;
482               default: throw new JDITestRuntimeException("** default case 3 **");
483             }
484 
485             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
486         }
487         log1("    TESTING ENDS");
488         return;
489     }

491    /*
492     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
493     *                                             String, String, String)
494     *
495     * It sets up a breakpoint at given line number within a given method in a given class
496     * for a given thread.
497     *
498     * Return value: BreakpointRequest object  in case of success
499     *
500     * JDITestRuntimeException   in case of an Exception thrown within the method
501     */
502 
503     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
504                                                   ReferenceType testedClass,
505                                                   String methodName,
506                                                   String bpLine,
507                                                   String property)
508             throws JDITestRuntimeException {
509 
510         log2("......setting up a breakpoint:");
511         log2("       thread: " + thread + "; class: " + testedClass +
512                         "; method: " + methodName + "; line: " + bpLine);
513 
514         List              alllineLocations = null;
515         Location          lineLocation     = null;
516         BreakpointRequest breakpRequest    = null;
517 
518         try {
519             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
520 
521             alllineLocations = method.allLineLocations();
522 
523             int n =
524                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
525             if (n > alllineLocations.size()) {
526                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
527             } else {
528                 lineLocation = (Location) alllineLocations.get(n);
529                 try {
530                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
531                     breakpRequest.putProperty("number", property);
532                     breakpRequest.addThreadFilter(thread);
533                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
534                 } catch ( Exception e1 ) {
535                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
536                     breakpRequest    = null;
537                 }
538             }
539         } catch ( Exception e2 ) {
540             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
541             breakpRequest    = null;
542         }
543 
544         if (breakpRequest == null) {
545             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
546             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
547         }
548 
549         log2("      a breakpoint has been set up");
550         return breakpRequest;
551     }

554     private void getEventSet()
555                  throws JDITestRuntimeException {
556         try {
557 //            log2("       eventSet = eventQueue.remove(waitTime);");
558             eventSet = eventQueue.remove(waitTime);
559             if (eventSet == null) {
560                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
561             }
562 //            log2("       eventIterator = eventSet.eventIterator;");
563             eventIterator = eventSet.eventIterator();
564         } catch ( Exception e ) {
565             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
566         }
567     }

570     private void breakpointForCommunication()
571                  throws JDITestRuntimeException {
572 
573         log2("breakpointForCommunication");
574         getEventSet();
575 
576         if (eventIterator.nextEvent() instanceof BreakpointEvent)
577             return;
578 
579         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
580     }

584     private ModificationWatchpointRequest settingModificationWatchpointRequest (
585                                                   ThreadReference thread,
586                                                   ReferenceType   testedClass,
587                                                   String          fieldName,
588                                                   int             suspendPolicy,
589                                                   String          property        )
590             throws JDITestRuntimeException {
591         try {
592             log2("......setting up ModificationWatchpointRequest:");
593             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
594             Field field = testedClass.fieldByName(fieldName);
595 
596             ModificationWatchpointRequest
597             awr = eventRManager.createModificationWatchpointRequest(field);
598             awr.putProperty("number", property);
599             awr.addThreadFilter(thread);
600             awr.setSuspendPolicy(suspendPolicy);
601 
602             log2("      ModificationWatchpointRequest has been set up");
603             return awr;
604         } catch ( Exception e ) {
605             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
606             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
607             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
608         }
609     }
