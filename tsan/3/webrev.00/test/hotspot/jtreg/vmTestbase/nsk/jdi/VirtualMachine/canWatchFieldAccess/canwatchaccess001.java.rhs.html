 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VirtualMachine.                                              <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.VirtualMachine.canWatchFieldAccess()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test has three phases and works as follows.              <BR>
 46  * <BR>
 47  * In first phase,                                                      <BR>
 48  * upon launching debuggee's VM which will be suspended,                <BR>
 49  * a debugger waits for the VMStartEvent within a predefined            <BR>
 50  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 51  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 52  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 53  * and waits for the event within the predefined time interval.         <BR>
 54  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 55  * Upon getting the ClassPrepareEvent,                                  <BR>
 56  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 57  * within debuggee's special methodForCommunication().                  <BR>
 58  * <BR>
 59  * In second phase to check the assetion,                               <BR>
 60  * the debugger and the debuggee perform the following loop.            <BR>
 61  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 62  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 63  *   and to inform the debugger with the event.                         <BR>
 64  * - Upon getting the BreakpointEvent,                                  <BR>
 65  *   the debugger performs the check case required.                     <BR>
 66  * Note. To inform each other of needed actions, the debugger and       <BR>
 67  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 68  * <BR>
 69  * In third phase when at the end,                                      <BR>
 70  * the debuggee changes the value of the "instruction"                  <BR>
 71  * to inform the debugger of checks finished, and both end.             <BR>
 72  */
 73

 98         int exitCode = new canwatchaccess001().runThis(argv, out);

102         }
103         return testExitCode;

146     private int runThis (String argv[], PrintStream out) {
147 
148         argsHandler     = new ArgumentHandler(argv);
149         logHandler      = new Log(out, argsHandler);
150         Binder binder   = new Binder(argsHandler, logHandler);
151 
152         waitTime        = argsHandler.getWaitTime() * 60000;
153 
154         try {
155             log2("launching a debuggee :");
156             log2("       " + debuggeeName);
157             if (argsHandler.verbose()) {
158                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
159             } else {
160                 debuggee = binder.bindToDebugee(debuggeeName);
161             }
162             if (debuggee == null) {
163                 log3("ERROR: no debuggee launched");
164                 return FAILED;
165             }
166             log2("debuggee launched");
167         } catch ( Exception e ) {
168             log3("ERROR: Exception : " + e);
169             log2("       test cancelled");
170             return FAILED;
171         }
172 
173         debuggee.redirectOutput(logHandler);
174 
175         vm = debuggee.VM();
176 
177         eventQueue = vm.eventQueue();
178         if (eventQueue == null) {
179             log3("ERROR: eventQueue == null : TEST ABORTED");
180             vm.exit(PASS_BASE);
181             return FAILED;
182         }
183 
184         log2("invocation of the method runTest()");
185         switch (runTest()) {
186 
187             case 0 :  log2("test phase has finished normally");
188                       log2("   waiting for the debuggee to finish ...");
189                       debuggee.waitFor();
190 
191                       log2("......getting the debuggee's exit status");
192                       int status = debuggee.getStatus();
193                       if (status != PASS_BASE) {
194                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
195                               status + " != PASS_BASE");
196                           testExitCode = FAILED;
197                       } else {
198                           log2("......debuggee returned expected exit status: " +
199                               status + " == PASS_BASE");
200                       }
201                       break;
202 
203             default : log3("ERROR: runTest() returned unexpected value");
204 
205             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
206                       log2("......forcing: vm.exit();");
207                       testExitCode = FAILED;
208                       try {
209                           vm.exit(PASS_BASE);
210                       } catch ( Exception e ) {
211                           log3("ERROR: Exception : e");
212                       }
213                       break;
214 
215             case 2 :  log3("test cancelled due to VMDisconnectedException");
216                       log2("......trying: vm.process().destroy();");
217                       testExitCode = FAILED;
218                       try {
219                           Process vmProcess = vm.process();
220                           if (vmProcess != null) {
221                               vmProcess.destroy();
222                           }
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227             }
228 
229         return testExitCode;
230     }

233    /*
234     * Return value: 0 - normal end of the test
235     *               1 - ubnormal end of the test
236     *               2 - VMDisconnectedException while test phase
237     */
238 
239     private int runTest() {
240 
241         try {
242             testRun();
243 
244             log2("waiting for VMDeathEvent");
245             getEventSet();
246             if (eventIterator.nextEvent() instanceof VMDeathEvent)
247                 return 0;
248 
249             log3("ERROR: last event is not the VMDeathEvent");
250             return 1;
251         } catch ( VMDisconnectedException e ) {
252             log3("ERROR: VMDisconnectedException : " + e);
253             return 2;
254         } catch ( Exception e ) {
255             log3("ERROR: Exception : " + e);
256             return 1;
257         }
258 
259     }

261     private void testRun()
262                  throws JDITestRuntimeException, Exception {
263 
264         eventRManager = vm.eventRequestManager();
265 
266         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
267         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
268         cpRequest.addClassFilter(debuggeeName);
269 
270         cpRequest.enable();
271         vm.resume();
272         getEventSet();
273         cpRequest.disable();
274 
275         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
276         debuggeeClass = event.referenceType();
277 
278         if (!debuggeeClass.name().equals(debuggeeName))
279            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
280 
281         log2("      received: ClassPrepareEvent for debuggeeClass");
282 
283         String bPointMethod = "methodForCommunication";
284         String lineForComm  = "lineForComm";
285         BreakpointRequest bpRequest;
286 
287         try {
288             bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
289                                           debuggeeClass,
290                                           bPointMethod, lineForComm, "zero");
291         } catch ( Exception e ) {
292             throw e;
293         }
294         bpRequest.enable();
295 
296     //------------------------------------------------------  testing section
297 
298         log1("     TESTING BEGINS");
299 
300         for (int i = 0; ; i++) {
301 
302             vm.resume();
303             breakpointForCommunication();
304 
305             int instruction = ((IntegerValue)
306                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
307 
308             if (instruction == 0) {
309                 vm.resume();
310                 break;
311             }
312 
313             log1("  new check: # " + i);
314 
315             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
316 
317             log2("...... vm.canWatchFieldAccess()");
318             if ( vm.canWatchFieldAccess() )
319                 log2("      vm.canWatchFieldAccess() == true");
320             else
321                 log2("      vm.canWatchFieldAccess() == false");
322 
323             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
324         }
325         log1("    TESTING ENDS");
326         return;
327     }

329    /*
330     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
331     *                                             String, String, String)
332     *
333     * It sets up a breakpoint at given line number within a given method in a given class
334     * for a given thread.
335     *
336     * Return value: BreakpointRequest object  in case of success
337     *
338     * JDITestRuntimeException   in case of an Exception thrown within the method
339     */
340 
341     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
342                                                   ReferenceType testedClass,
343                                                   String methodName,
344                                                   String bpLine,
345                                                   String property)
346             throws JDITestRuntimeException {
347 
348         log2("......setting up a breakpoint:");
349         log2("       thread: " + thread + "; class: " + testedClass +
350                         "; method: " + methodName + "; line: " + bpLine);
351 
352         List              alllineLocations = null;
353         Location          lineLocation     = null;
354         BreakpointRequest breakpRequest    = null;
355 
356         try {
357             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
358 
359             alllineLocations = method.allLineLocations();
360 
361             int n =
362                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
363             if (n > alllineLocations.size()) {
364                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
365             } else {
366                 lineLocation = (Location) alllineLocations.get(n);
367                 try {
368                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
369                     breakpRequest.putProperty("number", property);
370                     breakpRequest.addThreadFilter(thread);
371                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
372                 } catch ( Exception e1 ) {
373                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
374                     breakpRequest    = null;
375                 }
376             }
377         } catch ( Exception e2 ) {
378             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
379             breakpRequest    = null;
380         }
381 
382         if (breakpRequest == null) {
383             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
384             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
385         }
386 
387         log2("      a breakpoint has been set up");
388         return breakpRequest;
389     }

392     private void getEventSet()
393                  throws JDITestRuntimeException {
394         try {
395 //            log2("       eventSet = eventQueue.remove(waitTime);");
396             eventSet = eventQueue.remove(waitTime);
397             if (eventSet == null) {
398                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
399             }
400 //            log2("       eventIterator = eventSet.eventIterator;");
401             eventIterator = eventSet.eventIterator();
402         } catch ( Exception e ) {
403             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
404         }
405     }

408     private void breakpointForCommunication()
409                  throws JDITestRuntimeException {
410 
411         log2("breakpointForCommunication");
412         getEventSet();
413 
414         if (eventIterator.nextEvent() instanceof BreakpointEvent)
415             return;
416 
417         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
418     }
