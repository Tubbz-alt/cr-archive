 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.modificationWatchpointRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled access watchpoint requests.   <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checks up on assertions.                              <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

157     private int runThis (String argv[], PrintStream out) {
158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : " + e);
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if (eventIterator.nextEvent() instanceof VMDeathEvent)
258                 return 0;
259 
260             log3("ERROR: last event is not the VMDeathEvent");
261             return 1;
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }

272     private void testRun()
273                  throws JDITestRuntimeException, Exception {
274 
275         if ( !vm.canWatchFieldAccess() ) {
276             log2("......vm.canWatchFieldAccess == false :: test cancelled");
277             vm.exit(PASS_BASE);
278             return;
279         }
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         String bPointMethod = "methodForCommunication";
301         String lineForComm  = "lineForComm";
302 
303         ThreadReference   mainThread = threadByName("main");
304 
305         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
306                                              debuggeeClass,
307                                             bPointMethod, lineForComm, "zero");
308         bpRequest.enable();
309 
310     //------------------------------------------------------  testing section
311 
312         String fieldName = "testField";
313 
314         Field field = null;
315 
316         List         requests = null;
317         ListIterator li       = null;
318 
319         ModificationWatchpointRequest request = null;
320         ModificationWatchpointRequest awRequests[] = { null, null, null, null, null,
321                                                        null, null, null, null, null };
322         int listSize;
323         int flag;
324 
325 
326         log1("     TESTING BEGINS");
327 
328         for (int i = 0; ; i++) {
329 
330             vm.resume();
331             breakpointForCommunication();
332 
333             int instruction = ((IntegerValue)
334                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
335 
336             if (instruction == 0) {
337                 vm.resume();
338                 break;
339             }
340 
341             log1(":::::: case: # " + i);
342 
343             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
344 
345             field = debuggeeClass.fieldByName(fieldName);
346 
347             log2("......creating ModificationWatchpointRequests");
348             for (int i1 = 0; i1 < awRequests.length; i1++) {
349                 awRequests[i1] = eventRManager.createModificationWatchpointRequest(field);
350                 awRequests[i1].putProperty("number", String.valueOf(i1));
351 
352                 log2("......checking up on returned List after creating new ModificationWatchpointRequest");
353                 requests = eventRManager.modificationWatchpointRequests();
354                 listSize = requests.size();
355                 if ( listSize != (i1 + 1) ) {
356                     testExitCode = FAILED;
357                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
358                 }
359                 flag = 0;
360                 li = requests.listIterator();
361                 while (li.hasNext()) {
362                     request = (ModificationWatchpointRequest) li.next();
363                     if ( !request.isEnabled() ) {
364                         flag++;
365                         if (flag > 1) {
366                             testExitCode = FAILED;
367                             log3("ERROR: # of disabled requests > 1 : " + flag);
368                         }
369                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
370                             testExitCode = FAILED;
371                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
372                         }
373                     } else {
374                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
375                             testExitCode = FAILED;
376                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
377                         }
378                     }
379                 }
380 
381                 log2("      enabling created ModificationWatchpointRequest");
382                 awRequests[i1].enable();
383                 requests = eventRManager.modificationWatchpointRequests();
384                 listSize = requests.size();
385                 if ( listSize != (i1 + 1) ) {
386                     testExitCode = FAILED;
387                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
388                 }
389 
390                 li = requests.listIterator();
391                 while (li.hasNext()) {
392                     request = (ModificationWatchpointRequest) li.next();
393                     if ( !request.isEnabled() ) {
394                         testExitCode = FAILED;
395                         log3("ERROR: returned List contains disabled request : " + request);
396                     }
397                 }
398 
399                 log2("      removing item from the List; UnsupportedOperationException is expected");
400                 try {
401                     requests.remove(i1);
402                     testExitCode = FAILED;
403                     log3("ERROR: NO exception");
404                 } catch ( UnsupportedOperationException e ) {
405                     log2("        UnsupportedOperationException ");
406                 }
407             }
408 
409             log2("......deleting ModificationWatchpointRequests");
410             for (int i2 = awRequests.length -1; i2 >= 0; i2--) {
411                 eventRManager.deleteEventRequest(awRequests[i2]);
412                 requests = eventRManager.modificationWatchpointRequests();
413                 listSize = requests.size();
414                 if ( listSize != i2 ) {
415                     testExitCode = FAILED;
416                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " + i2);
417                 }
418                 log2("      removing item from the List; UnsupportedOperationException is expected");
419                 try {
420                     requests.remove(i2);
421                     testExitCode = FAILED;
422                     log3("ERROR: NO exception");
423                 } catch ( UnsupportedOperationException e ) {
424                     log2("        UnsupportedOperationException ");
425                 }
426             }
427 
428             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
429         }
430         log1("    TESTING ENDS");
431         return;
432     }

434     private ThreadReference threadByName(String name)
435                  throws JDITestRuntimeException {
436 
437         List         all = vm.allThreads();
438         ListIterator li  = all.listIterator();
439 
440         for (; li.hasNext(); ) {
441             ThreadReference thread = (ThreadReference) li.next();
442             if (thread.name().equals(name))
443                 return thread;
444         }
445         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
446     }

448    /*
449     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
450     *                                             String, String, String)
451     *
452     * It sets up a breakpoint at given line number within a given method in a given class
453     * for a given thread.
454     *
455     * Return value: BreakpointRequest object  in case of success
456     *
457     * JDITestRuntimeException   in case of an Exception thrown within the method
458     */
459 
460     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
461                                                   ReferenceType testedClass,
462                                                   String methodName,
463                                                   String bpLine,
464                                                   String property)
465             throws JDITestRuntimeException {
466 
467         log2("......setting up a breakpoint:");
468         log2("       thread: " + thread + "; class: " + testedClass +
469                         "; method: " + methodName + "; line: " + bpLine);
470 
471         List              alllineLocations = null;
472         Location          lineLocation     = null;
473         BreakpointRequest breakpRequest    = null;
474 
475         try {
476             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
477 
478             alllineLocations = method.allLineLocations();
479 
480             int n =
481                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
482             if (n > alllineLocations.size()) {
483                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
484             } else {
485                 lineLocation = (Location) alllineLocations.get(n);
486                 try {
487                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
488                     breakpRequest.putProperty("number", property);
489                     breakpRequest.addThreadFilter(thread);
490                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
491                 } catch ( Exception e1 ) {
492                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
493                     breakpRequest    = null;
494                 }
495             }
496         } catch ( Exception e2 ) {
497             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
498             breakpRequest    = null;
499         }
500 
501         if (breakpRequest == null) {
502             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
503             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
504         }
505 
506         log2("      a breakpoint has been set up");
507         return breakpRequest;
508     }

511     private void getEventSet()
512                  throws JDITestRuntimeException {
513         try {
514 //            log2("       eventSet = eventQueue.remove(waitTime);");
515             eventSet = eventQueue.remove(waitTime);
516             if (eventSet == null) {
517                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
518             }
519 //            log2("       eventIterator = eventSet.eventIterator;");
520             eventIterator = eventSet.eventIterator();
521         } catch ( Exception e ) {
522             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
523         }
524     }

527     private void breakpointForCommunication()
528                  throws JDITestRuntimeException {
529 
530         log2("breakpointForCommunication");
531         getEventSet();
532 
533         if (eventIterator.nextEvent() instanceof BreakpointEvent)
534             return;
535 
536         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
537     }
