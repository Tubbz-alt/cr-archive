 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodExitRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodExitRequest.addThreadFilter()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Restricts the events generated by this request            <BR>
 47  *    to those in the given thread.                             <BR>
 48  * The cases to test include three thread's states:             <BR>
 49  * not started, running, dead.                                  <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates MethodExitRequest1 and                                   <BR>
 72  *     invokes the method addThreadFilter(tread1) on the request1,      <BR>
 73  *   - creates MethodExitRequest2,                                      <BR>
 74  *     resumes the debuggee to get the thread1 started, and             <BR>
 75  *     invokes the method addThreadFilter(thread) on the request2,      <BR>
 76  *   - creates MethodExitRequest3, waits the thread1 to die, and        <BR>
 77  *     invokes the method addThreadFilter(thread) on the request3.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test,                              <BR>
 80  * the debuggee changes the value of the "instruction"  which           <BR>
 81  * the debugger and debuggee use to inform each other of needed actions,<BR>
 82  * and both end.
 83  * <BR>
 84  */
 85

167     private int runThis (String argv[], PrintStream out) {
168 
169         argsHandler     = new ArgumentHandler(argv);
170         logHandler      = new Log(out, argsHandler);
171         Binder binder   = new Binder(argsHandler, logHandler);
172 
173         waitTime        = argsHandler.getWaitTime() * 60000;
174 
175         try {
176             log2("launching a debuggee :");
177             log2("       " + debuggeeName);
178             if (argsHandler.verbose()) {
179                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
180             } else {
181                 debuggee = binder.bindToDebugee(debuggeeName);
182             }
183             if (debuggee == null) {
184                 log3("ERROR: no debuggee launched");
185                 return FAILED;
186             }
187             log2("debuggee launched");
188         } catch ( Exception e ) {
189             log3("ERROR: Exception : " + e);
190             log2("       test cancelled");
191             return FAILED;
192         }
193 
194         debuggee.redirectOutput(logHandler);
195 
196         vm = debuggee.VM();
197 
198         eventQueue = vm.eventQueue();
199         if (eventQueue == null) {
200             log3("ERROR: eventQueue == null : TEST ABORTED");
201             vm.exit(PASS_BASE);
202             return FAILED;
203         }
204 
205         log2("invocation of the method runTest()");
206         switch (runTest()) {
207 
208             case 0 :  log2("test phase has finished normally");
209                       log2("   waiting for the debuggee to finish ...");
210                       debuggee.waitFor();
211 
212                       log2("......getting the debuggee's exit status");
213                       int status = debuggee.getStatus();
214                       if (status != PASS_BASE) {
215                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
216                               status + " != PASS_BASE");
217                           testExitCode = FAILED;
218                       } else {
219                           log2("......debuggee returned expected exit status: " +
220                               status + " == PASS_BASE");
221                       }
222                       break;
223 
224             default : log3("ERROR: runTest() returned unexpected value");
225 
226             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
227                       log2("......forcing: vm.exit();");
228                       testExitCode = FAILED;
229                       try {
230                           vm.exit(PASS_BASE);
231                       } catch ( Exception e ) {
232                           log3("ERROR: Exception : " + e);
233                       }
234                       break;
235 
236             case 2 :  log3("test cancelled due to VMDisconnectedException");
237                       log2("......trying: vm.process().destroy();");
238                       testExitCode = FAILED;
239                       try {
240                           Process vmProcess = vm.process();
241                           if (vmProcess != null) {
242                               vmProcess.destroy();
243                           }
244                       } catch ( Exception e ) {
245                           log3("ERROR: Exception : " + e);
246                       }
247                       break;
248             }
249 
250         return testExitCode;
251     }

254    /*
255     * Return value: 0 - normal end of the test
256     *               1 - ubnormal end of the test
257     *               2 - VMDisconnectedException while test phase
258     */
259 
260     private int runTest() {
261 
262         try {
263             testRun();
264 
265             log2("waiting for VMDeathEvent");
266             getEventSet();
267             if (eventIterator.nextEvent() instanceof VMDeathEvent)
268                 return 0;
269 
270             log3("ERROR: last event is not the VMDeathEvent");
271             return 1;
272         } catch ( VMDisconnectedException e ) {
273             log3("ERROR: VMDisconnectedException : " + e);
274             return 2;
275         } catch ( Exception e ) {
276             log3("ERROR: Exception : " + e);
277             return 1;
278         }
279 
280     }

282     private void testRun()
283                  throws JDITestRuntimeException, Exception {
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = threadByName("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318 
319         EventRequest eventRequest1 = null;
320         EventRequest eventRequest2 = null;
321         EventRequest eventRequest3 = null;
322 
323         ThreadReference thread1     = null;
324         String          thread1Name = "thread1";
325 
326         String property1 = "MethodExitRequest1";
327         String property2 = "MethodExitRequest2";
328         String property3 = "MethodExitRequest3";
329 
330         ReferenceType testClassReference = null;
331 
332         for (int i = 0; ; i++) {
333 
334             vm.resume();
335             breakpointForCommunication();
336 
337             int instruction = ((IntegerValue)
338                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
339 
340             if (instruction == 0) {
341                 vm.resume();
342                 break;
343             }
344 
345             log1(":::::: case: # " + i);
346 
347             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
348 
349             switch (i) {
350               case 0:
351                      testClassReference =
352                             (ReferenceType) vm.classesByName(testedClassName).get(0);
353                      thread1 = (ThreadReference) debuggeeClass.getValue(
354                                         debuggeeClass.fieldByName(thread1Name));
355 
356                      eventRequest1 = setting2MethodExitRequest (null,
357                                              testClassReference,
358                                              EventRequest.SUSPEND_NONE, property1);
359 
360                      try {
361                          log2("......eventRequest1.addThreadFilter(thread1);");
362                          log2("        no Exception expected");
363                          ((MethodExitRequest)eventRequest1).addThreadFilter(thread1);
364                      } catch ( Exception e ) {
365                          log3("ERROR: Exception : " + e);
366                          testExitCode = FAILED;
367                      }
368                      break;
369 
370               case 1:
371                      eventRequest2 = setting2MethodExitRequest (null,
372                                              testClassReference,
373                                              EventRequest.SUSPEND_NONE, property2);
374 
375                      try {
376                          log2("......eventRequest2.addThreadFilter(thread1);");
377                          log2("        no Exception expected");
378                          ((MethodExitRequest)eventRequest2).addThreadFilter(thread1);
379                      } catch ( Exception e ) {
380                          log3("ERROR: Exception : " + e);
381                          testExitCode = FAILED;
382                      }
383                      break;
384 
385               case 2:
386                      eventRequest3 = setting2MethodExitRequest (null,
387                                              testClassReference,
388                                              EventRequest.SUSPEND_NONE, property3);
389 
390                      try {
391                          log2("......eventRequest3.addThreadFilter(thread1);");
392                          log2("        no Exception expected");
393                          ((MethodExitRequest)eventRequest3).addThreadFilter(thread1);
394                      } catch ( Exception e ) {
395                          log3("ERROR: Exception : " + e);
396                          testExitCode = FAILED;
397                      }
398                      break;
399 
400 
401               default:
402                       throw new JDITestRuntimeException("** default case 2 **");
403             }
404 
405             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
406         }
407         log1("    TESTING ENDS");
408         return;
409     }

411     private ThreadReference threadByName(String name)
412                  throws JDITestRuntimeException {
413 
414         List         all = vm.allThreads();
415         ListIterator li  = all.listIterator();
416 
417         for (; li.hasNext(); ) {
418             ThreadReference thread = (ThreadReference) li.next();
419             if (thread.name().equals(name))
420                 return thread;
421         }
422         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
423     }

425    /*
426     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
427     *                                             String, String, String)
428     *
429     * It sets up a breakpoint at given line number within a given method in a given class
430     * for a given thread.
431     *
432     * Return value: BreakpointRequest object  in case of success
433     *
434     * JDITestRuntimeException   in case of an Exception thrown within the method
435     */
436 
437     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
438                                                   ReferenceType testedClass,
439                                                   String methodName,
440                                                   String bpLine,
441                                                   String property)
442             throws JDITestRuntimeException {
443 
444         log2("......setting up a breakpoint:");
445         log2("       thread: " + thread + "; class: " + testedClass +
446                         "; method: " + methodName + "; line: " + bpLine);
447 
448         List              alllineLocations = null;
449         Location          lineLocation     = null;
450         BreakpointRequest breakpRequest    = null;
451 
452         try {
453             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
454 
455             alllineLocations = method.allLineLocations();
456 
457             int n =
458                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
459             if (n > alllineLocations.size()) {
460                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
461             } else {
462                 lineLocation = (Location) alllineLocations.get(n);
463                 try {
464                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
465                     breakpRequest.putProperty("number", property);
466                     breakpRequest.addThreadFilter(thread);
467                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
468                 } catch ( Exception e1 ) {
469                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
470                     breakpRequest    = null;
471                 }
472             }
473         } catch ( Exception e2 ) {
474             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
475             breakpRequest    = null;
476         }
477 
478         if (breakpRequest == null) {
479             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
480             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
481         }
482 
483         log2("      a breakpoint has been set up");
484         return breakpRequest;
485     }

488     private void getEventSet()
489                  throws JDITestRuntimeException {
490         try {
491 //            log2("       eventSet = eventQueue.remove(waitTime);");
492             eventSet = eventQueue.remove(waitTime);
493             if (eventSet == null) {
494                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
495             }
496 //            log2("       eventIterator = eventSet.eventIterator;");
497             eventIterator = eventSet.eventIterator();
498         } catch ( Exception e ) {
499             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
500         }
501     }

504     private void breakpointForCommunication()
505                  throws JDITestRuntimeException {
506 
507         log2("breakpointForCommunication");
508         getEventSet();
509 
510         if (eventIterator.nextEvent() instanceof BreakpointEvent)
511             return;
512 
513         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
514     }

518     private MethodExitRequest setting2MethodExitRequest ( ThreadReference thread,
519                                                          ReferenceType   testedClass,
520                                                          int             suspendPolicy,
521                                                          String          property        )
522             throws JDITestRuntimeException {
523         try {
524             log2("......setting up MethodExitRequest:");
525             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
526 
527             MethodExitRequest
528             mexr = eventRManager.createMethodExitRequest();
529             mexr.putProperty("number", property);
530             if (thread != null)
531                 mexr.addThreadFilter(thread);
532             mexr.addClassFilter(testedClass);
533             mexr.setSuspendPolicy(suspendPolicy);
534 
535         log2("      MethodExitRequest has been set up");
536         return mexr;
537         } catch ( Exception e ) {
538             log3("ERROR: ATTENTION: Exception within settingMethodExitRequest() : " + e);
539             log3("       MethodExitRequest HAS NOT BEEN SET UP");
540             throw new JDITestRuntimeException("** FAILURE to set up MethodExitRequest **");
541         }
