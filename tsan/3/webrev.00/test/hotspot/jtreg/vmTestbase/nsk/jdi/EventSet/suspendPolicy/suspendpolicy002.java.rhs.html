 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for an AccessWatchpointEvent set,       <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the above,                                  <BR>
 68  * the debugger and the debuggee perform the following loop.            <BR>
 69  * - The debugger sets up a AccessWatchpointRequest, resumes            <BR>
 70  *   the debuggee, and waits for the AccessWatchpointEvent.             <BR>
 71  * - The debuggee invokes the special method which makes access to      <BR>
 72  *   a predefined variable to be resulting in the event.                <BR>
 73  * - Upon getting new event, the debugger                               <BR>
 74  *   performs the check corresponding to the event.                     <BR>
 75  * <BR>
 76  * Note. To inform each other of needed actions, the debugger and       <BR>
 77  *       and the debuggee use debuggee's variable "instruction".        <BR>
 78  * In third phase, at the end,                                          <BR>
 79  * the debuggee changes the value of the "instruction"                  <BR>
 80  * to inform the debugger of checks finished, and both end.             <BR>
 81  * <BR>
 82  */
 83

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
260                 log3("ERROR: last event is not the VMDeathEvent");
261                 return 1;
262             }
263 
264             return 0;
265 
266         } catch ( VMDisconnectedException e ) {
267             log3("ERROR: VMDisconnectedException : " + e);
268             return 2;
269         } catch ( Exception e ) {
270             log3("ERROR: Exception : " + e);
271             return 1;
272         }
273 
274     }

276     private void testRun()
277                  throws JDITestRuntimeException, Exception {
278 
279         if ( !vm.canWatchFieldAccess() ) {
280             log2("......vm.canWatchFieldAccess == false :: test cancelled");
281             vm.exit(PASS_BASE);
282             return;
283         }
284 
285 
286         eventRManager = vm.eventRequestManager();
287 
288         log2("......getting ClassPrepareEvent for debuggee's class");
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292         cpRequest.enable();
293         vm.resume();
294         getEventSet();
295         cpRequest.disable();
296 
297         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
298         debuggeeClass = event.referenceType();
299 
300         if (!debuggeeClass.name().equals(debuggeeName))
301            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
305 
306         String            bPointMethod = "methodForCommunication";
307         String            lineForComm  = "lineForComm";
308         BreakpointRequest bpRequest;
309         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
310         bpRequest = settingBreakpoint(mainThread,
311                                       debuggeeClass,
312                                       bPointMethod, lineForComm, "zero");
313         bpRequest.enable();
314 
315         vm.resume();
316 
317     //------------------------------------------------------  testing section
318 
319         log1("     TESTING BEGINS");
320 
321         EventRequest eventRequest1 = null;
322         EventRequest eventRequest2 = null;
323         EventRequest eventRequest3 = null;
324 
325         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
326         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
327         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
328         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
329 
330         int policyExpected[] = {    SUSPEND_NONE,
331                                     SUSPEND_THREAD,
332                                     SUSPEND_ALL,
333                                     SUSPEND_THREAD,
334                                     SUSPEND_ALL,
335                                     SUSPEND_ALL,
336                                     SUSPEND_ALL      };
337         int policy = 0;
338 
339         ReferenceType testClassReference = null;
340 
341         String fieldName = "var1";
342 
343 
344         for (int i = 0; ; i++) {
345 
346 
347             breakpointForCommunication();
348 
349             int instruction = ((IntegerValue)
350                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
351             if (instruction == 0) {
352                 vm.resume();
353                 break;
354             }
355 
356 
357             log1(":::::: case: # " + i);
358 
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
360 
361             switch (i) {
362 
363               case 0:
364                       testClassReference =
365                            (ReferenceType) vm.classesByName(testedClassName).get(0);
366 
367                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
368                                              testClassReference, fieldName,
369                                              SUSPEND_NONE, "AccessWatchpointRequest1");
370                       eventRequest1.enable();
371                       break;
372 
373               case 1:
374                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
375                                              testClassReference, fieldName,
376                                              SUSPEND_THREAD, "AccessWatchpointRequest2");
377                       eventRequest2.enable();
378                       break;
379 
380               case 2:
381                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
382                                              testClassReference, fieldName,
383                                              SUSPEND_ALL, "AccessWatchpointRequest3");
384                       eventRequest3.enable();
385                       break;
386 
387               case 3:
388                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
389                                              testClassReference, fieldName,
390                                              SUSPEND_NONE, "AccessWatchpointRequest4");
391                       eventRequest1.enable();
392                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
393                                              testClassReference, fieldName,
394                                              SUSPEND_THREAD, "AccessWatchpointRequest5");
395                       eventRequest2.enable();
396                       break;
397 
398               case 4:
399                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
400                                              testClassReference, fieldName,
401                                              SUSPEND_NONE, "AccessWatchpointRequest6");
402                       eventRequest1.enable();
403                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
404                                              testClassReference, fieldName,
405                                              SUSPEND_ALL, "AccessWatchpointRequest7");
406                       eventRequest3.enable();
407                       break;
408 
409               case 5:
410                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
411                                              testClassReference, fieldName,
412                                              SUSPEND_THREAD, "AccessWatchpointRequest8");
413                       eventRequest2.enable();
414                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
415                                              testClassReference, fieldName,
416                                              SUSPEND_ALL, "AccessWatchpointRequest9");
417                       eventRequest3.enable();
418                       break;
419 
420               case 6:
421                       eventRequest1 = settingAccessWatchpointRequest (mainThread,
422                                              testClassReference, fieldName,
423                                              SUSPEND_NONE, "AccessWatchpointRequest10");
424                       eventRequest1.enable();
425                       eventRequest2 = settingAccessWatchpointRequest (mainThread,
426                                              testClassReference, fieldName,
427                                              SUSPEND_THREAD, "AccessWatchpointRequest11");
428                       eventRequest2.enable();
429                       eventRequest3 = settingAccessWatchpointRequest (mainThread,
430                                              testClassReference, fieldName,
431                                              SUSPEND_ALL, "AccessWatchpointRequest12");
432                       eventRequest3.enable();
433                       break;
434 
435 
436               default:
437                       throw new JDITestRuntimeException("** default case 2 **");
438             }
439 
440             log2("......waiting for new AccessWatchpointEvent : " + i);
441             mainThread.resume();
442             getEventSet();
443 
444             Event newEvent = eventIterator.nextEvent();
445             log2("       got new Event with propety 'number' == " + newEvent.request().getProperty("number"));
446             if ( !( newEvent instanceof AccessWatchpointEvent)) {
447                  log3("ERROR: new event is not AccessWatchpointEvent");
448                  testExitCode = FAILED;
449             } else {
450                 log2("       Event is instanceof AccessWatchpointEvent");
451                 policy = eventSet.suspendPolicy();
452                 if (policy != policyExpected[i]) {
453                     log3("ERROR: eventSet.suspendPolicy() != policyExpected");
454                     log3("       eventSet.suspendPolicy() == " + policy);
455                     log3("       policyExpected           == " + policyExpected[i]);
456                     testExitCode = FAILED;
457                 }
458             }
459 
460             switch (i) {
461               case 0: eventRequest1.disable();                          break;
462               case 1: eventRequest2.disable();                          break;
463               case 2: eventRequest3.disable();                          break;
464               case 3: eventRequest1.disable(); eventRequest2.disable(); break;
465               case 4: eventRequest1.disable(); eventRequest3.disable(); break;
466               case 5: eventRequest2.disable(); eventRequest3.disable(); break;
467               case 6: eventRequest1.disable(); eventRequest2.disable();
468                       eventRequest3.disable();                          break;
469            }
470 
471             switch (policy) {
472               case SUSPEND_NONE   :                      break;
473               case SUSPEND_THREAD : mainThread.resume(); break;
474               case SUSPEND_ALL    : vm.resume();         break;
475               default: throw new JDITestRuntimeException("** default case 3 **");
476             }
477 
478             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
479         }
480         log1("    TESTING ENDS");
481         return;
482     }

484    /*
485     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
486     *                                             String, String, String)
487     *
488     * It sets up a breakpoint at given line number within a given method in a given class
489     * for a given thread.
490     *
491     * Return value: BreakpointRequest object  in case of success
492     *
493     * JDITestRuntimeException   in case of an Exception thrown within the method
494     */
495 
496     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
497                                                   ReferenceType testedClass,
498                                                   String methodName,
499                                                   String bpLine,
500                                                   String property)
501             throws JDITestRuntimeException {
502 
503         log2("......setting up a breakpoint:");
504         log2("       thread: " + thread + "; class: " + testedClass +
505                         "; method: " + methodName + "; line: " + bpLine);
506 
507         List              alllineLocations = null;
508         Location          lineLocation     = null;
509         BreakpointRequest breakpRequest    = null;
510 
511         try {
512             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
513 
514             alllineLocations = method.allLineLocations();
515 
516             int n =
517                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
518             if (n > alllineLocations.size()) {
519                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
520             } else {
521                 lineLocation = (Location) alllineLocations.get(n);
522                 try {
523                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
524                     breakpRequest.putProperty("number", property);
525                     breakpRequest.addThreadFilter(thread);
526                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
527                 } catch ( Exception e1 ) {
528                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
529                     breakpRequest    = null;
530                 }
531             }
532         } catch ( Exception e2 ) {
533             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
534             breakpRequest    = null;
535         }
536 
537         if (breakpRequest == null) {
538             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
539             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
540         }
541 
542         log2("      a breakpoint has been set up");
543         return breakpRequest;
544     }

547     private void getEventSet()
548                  throws JDITestRuntimeException {
549         try {
550 //            log2("       eventSet = eventQueue.remove(waitTime);");
551             eventSet = eventQueue.remove(waitTime);
552             if (eventSet == null) {
553                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
554             }
555 //            log2("       eventIterator = eventSet.eventIterator;");
556             eventIterator = eventSet.eventIterator();
557         } catch ( Exception e ) {
558             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
559         }
560     }

563     private void breakpointForCommunication()
564                  throws JDITestRuntimeException {
565 
566         log2("breakpointForCommunication");
567         getEventSet();
568 
569         if (eventIterator.nextEvent() instanceof BreakpointEvent)
570             return;
571 
572         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
573     }

577     private AccessWatchpointRequest settingAccessWatchpointRequest (
578                                                   ThreadReference thread,
579                                                   ReferenceType   testedClass,
580                                                   String          fieldName,
581                                                   int             suspendPolicy,
582                                                   String          property        )
583             throws JDITestRuntimeException {
584         try {
585             log2("......setting up AccessWatchpointRequest:");
586             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
587             Field field = testedClass.fieldByName(fieldName);
588 
589             AccessWatchpointRequest
590             awr = eventRManager.createAccessWatchpointRequest(field);
591             awr.putProperty("number", property);
592             awr.addThreadFilter(thread);
593             awr.setSuspendPolicy(suspendPolicy);
594 
595             log2("      AccessWatchpointRequest has been set up");
596             return awr;
597         } catch ( Exception e ) {
598             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
599             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
600             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
601         }
602     }
