 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertions:              <BR>
 46  * -  All StackFrame objects for this thread are invalidated.   <BR>
 47  * -  The frame previous to the parameter frame                 <BR>
 48  *    will become the current frame.                            <BR>
 49  * <BR>
 50  * The test has two phases and works as follows.                <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assetion,                               <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent<BR>
 67  * in the methodForCommunication().                                     <BR>
 68  * - The debuggee creates second thread, the thread2, which will        <BR>
 69  * call a tested poppedMethod() from its run() method.                  <BR>
 70  * The poppedMethod will call the special breakpointMethod,             <BR>
 71  * After creating the thread2, the debuggee invokes                     <BR>
 72  * the methodForCommunication to be suspended and                       <BR>
 73  * to inform the debugger with the event.                               <BR>
 74  * - Upon getting the BreakpointEvent in the methodForCommunication,    <BR>
 75  * the debugger sets up a breakpoint in the breakpointMethod,           <BR>
 76  * resumes the debuggee, waits for BreakpointEvent in the               <BR>
 77  * breakpointMethod and, and upon getting it,                           <BR>
 78  * (1) to check up on first assertion, gets thread2.frame(2),           <BR>
 79  * invokes popFrames() and checks up on equality of thread2.frame(2)    <BR>
 80  * and thread2.frame(0). Since thread2.frame(2) has to be invalid,      <BR>
 81  * the expected result is false; otherwise, the test FAILED.            <BR>
 82  * (2) to check up on second assertion, gets thread2.frame(1)           <BR>
 83  * which has to result in the IndexOutOfBoundsException because         <BR>
 84  * the current frame is the only one in FramesStack.                    <BR>
 85  * <BR>
 86  */
 87

111 
112         int exitCode = new popframes005().runThis(argv, out);

114         if (exitCode != PASSED) {
115             System.out.println("TEST FAILED");
116         }
117         return testExitCode;

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

250    /*
251     * Return value: 0 - normal end of the test
252     *               1 - ubnormal end of the test
253     *               2 - VMDisconnectedException while test phase
254     */

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }
275 
276     }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         if ( !vm.canPopFrames() ) {
301             log2("......vm.canPopFrames() == false : test is cancelled");
302             vm.resume();
303             return;
304         }
305 
306         String bPointMethod = "methodForCommunication";
307         String lineForComm  = "lineForComm";
308 
309         log2("......setting BreakpointRequest (bpRequest) in main thread");
310         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
311                                           debuggeeClass,
312                                           bPointMethod, lineForComm, "zero");
313         log2("bpRequest.enable();");
314         bpRequest.enable();
315 
316     //------------------------------------------------------  testing section
317 
318         log1("     TESTING BEGINS");
319 
320         label0:
321         {
322             vm.resume();
323             breakpointForCommunication();
324 
325             String thread2Name         = "thread2";
326             ThreadReference thread2Ref = debuggee.threadByNameOrThrow(thread2Name);
327 
328             StackFrame stackFrame0 = null;
329             StackFrame stackFrame1 = null;
330             StackFrame stackFrame2 = null;
331 
332             String breakpointMethod = "breakpointMethod";
333             String breakpointLine    = "breakpointLine";
334 
335             log2("......setting breakpoint in breakpointMethod");
336             breakpointRequest2 = settingBreakpoint(thread2Ref, debuggeeClass,
337                                           breakpointMethod, breakpointLine, "one");
338             log2("......breakpointRequest2.enable();");
339             breakpointRequest2.enable();
340 
341             {  // to get mainThread suspended; otherwise its end results in
342                // no suspention for breakpointRequest2 (bug or not?), end of test,
343                // and VMDisconnectedException
344             thread2Ref.suspend();
345             log2("......eventSet.resume();");
346             eventSet.resume();
347             breakpointForCommunication();
348             log2("bpRequest.disable();");
349             bpRequest.disable();
350             thread2Ref.resume();
351             }
352 
353             log2("......breakpointInMethod(breakpointRequest2);");
354             breakpointInMethod("one");
355 
356             log2("......stackFrame2 = thread2Ref.frame(2);");
357             stackFrame2 = thread2Ref.frame(2);
358 
359             log2("......stackFrame1 = thread2Ref.frame(1);");
360             stackFrame1 = thread2Ref.frame(1);
361 
362             log2("......thread2Ref.popFrames(stackFrame1);");
363             try {
364                 thread2Ref.popFrames(stackFrame1);
365             } catch ( IncompatibleThreadStateException e ) {
366                 log3("ERROR: IncompatibleThreadStateException");
367                 testExitCode = FAILED;
368                 break label0;
369             }
370             log2("......checking up on equality");
371             if ( thread2Ref.frame(0).equals(stackFrame2) ) {
372                 log3("ERROR: thread2Ref.frame(0).equals(stackFrame2)");
373                 testExitCode = FAILED;
374             }
375 
376             log2("......stackFrame1 = thread2Ref.frame(1);");
377             try {
378                 stackFrame1 = thread2Ref.frame(1);
379                 log3("ERROR: no IndexOutOfBoundsException");
380                 testExitCode = FAILED;
381             } catch ( IndexOutOfBoundsException e ) {
382                 log2("      IndexOutOfBoundsException");
383             }
384 
385 
386             log2("......breakpointRequest2.disable();");
387             breakpointRequest2.disable();
388 
389             vm.resume();
390 
391             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
392         }
393         log1("    TESTING ENDS");
394         return;
395     }

397    /*
398     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
399     *                                             String, String, String)
400     *
401     * It sets up a breakpoint at given line number within a given method in a given class
402     * for a given thread.
403     *
404     * Return value: BreakpointRequest object  in case of success
405     *
406     * JDITestRuntimeException   in case of an Exception thrown within the method
407     */
408 
409     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
410                                                   ReferenceType testedClass,
411                                                   String methodName,
412                                                   String bpLine,
413                                                   String property)
414             throws JDITestRuntimeException {
415 
416         log2("......setting up a breakpoint:");
417         log2("       thread: " + thread + "; class: " + testedClass +
418                         "; method: " + methodName + "; line: " + bpLine);
419 
420         List              alllineLocations = null;
421         Location          lineLocation     = null;
422         BreakpointRequest breakpRequest    = null;
423 
424         try {
425             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
426 
427             alllineLocations = method.allLineLocations();
428 
429             int n =
430                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
431             if (n > alllineLocations.size()) {
432                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
433             } else {
434                 lineLocation = (Location) alllineLocations.get(n);
435                 try {
436                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
437                     breakpRequest.putProperty("number", property);
438                     breakpRequest.addThreadFilter(thread);
439                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
440                 } catch ( Exception e1 ) {
441                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
442                     breakpRequest    = null;
443                 }
444             }
445         } catch ( Exception e2 ) {
446             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
447             breakpRequest    = null;
448         }
449 
450         if (breakpRequest == null) {
451             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
452             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
453         }
454 
455         log2("      a breakpoint has been set up");
456         return breakpRequest;
457     }

461     private void getEventSet()
462                  throws JDITestRuntimeException {
463         try {
464 //            log2("       eventSet = eventQueue.remove(waitTime);");
465             eventSet = eventQueue.remove(waitTime);
466             if (eventSet == null) {
467                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
468             }
469 //            log2("       eventIterator = eventSet.eventIterator;");
470             eventIterator = eventSet.eventIterator();
471         } catch ( Exception e ) {
472             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
473         }
474     }

477     private void breakpointForCommunication()
478                  throws JDITestRuntimeException {
479 
480         log2("breakpointForCommunication");
481         getEventSet();
482 
483         if (eventIterator.nextEvent() instanceof BreakpointEvent)
484             return;
485 
486         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
487     }

491     private void breakpointInMethod(String number)
492                  throws JDITestRuntimeException {
493 
494         Event event;
495 
496         getEventSet();
497         event = eventIterator.nextEvent();
498 
499         if ( !(event instanceof BreakpointEvent) )
500             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
501 
502         log2("---->: request().getProperty == " +
503              event.request().getProperty("number"));
504 
505         if ( event.request().getProperty("number").equals(number) )
506             return;
507 
508         throw new JDITestRuntimeException("** UNEXPECTED breakpoint **");
509     }
