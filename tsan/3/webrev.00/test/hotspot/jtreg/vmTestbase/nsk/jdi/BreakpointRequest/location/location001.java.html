 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * BreakpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.BreakpointRequest.location()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks if a Location object, an argument of the method      <BR>
 46  *    EventRequestManager.createBreakpointRequest(Location),            <BR>
 47  * is equal to one returned by the method BreakpointRequest.location()  <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.                      <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase to check the assertion,                              <BR>
 64  * the debugger and the debuggee perform the following.                 <BR>
 65  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 66  * - The debuggee creates new threads, thread1, and invokes             <BR>
 67  *   the methodForCommunication to be suspended and                     <BR>
 68  *   to inform the debugger with the event.                             <BR>
 69  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 70  *   - creates a BreakpointRequest and gets Location object location1,  <BR>
 71  *   - gets Location object location2 with the method                   <BR>
 72  *     BreakpointRequest.location(),                                    <BR>
 73  *   - compares the Location objects.                                   <BR>
 74  * <BR>
 75  * In third phase, at the end of the test,                              <BR>
 76  * the debuggee changes the value of the "instruction"  which           <BR>
 77  * the debugger and debuggee use to inform each other of needed actions,<BR>
 78  * and both end.
 79  * <BR>
 80  */
 81

104     public static int run (String argv[], PrintStream out) {
105 
106         int exitCode = new location001().runThis(argv, out);

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

247    /*
248     * Return value: 0 - normal end of the test
249     *               1 - ubnormal end of the test
250     *               2 - VMDisconnectedException while test phase
251     */
252 
253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }
272 
273     }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         EventRequest eventRequest1 = null;
312 
313         ThreadReference thread1     = null;
314         String          thread1Name = "thread1";
315 
316         String property1 = "BreakpointRequest1";
317 
318         String methodName = "method";
319         String bpLineName = "breakpointLine";
320 
321         ReferenceType testClassReference = null;
322 
323         for (int i = 0; ; i++) {
324 
325             vm.resume();
326             breakpointForCommunication();
327 
328             int instruction = ((IntegerValue)
329                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
330 
331             if (instruction == 0) {
332                 vm.resume();
333                 break;
334             }
335 
336             log1(":::::: case: # " + i);
337 
338             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
339 
340             switch (i) {
341 
342               case 0:
343                      testClassReference =
344                             (ReferenceType) vm.classesByName(testedClassName).get(0);
345 
346                      thread1 = (ThreadReference) debuggeeClass.getValue(
347                                         debuggeeClass.fieldByName(thread1Name));
348 
349                      log2(".....setting up BreakpointRequest and getting Location object location1");
350                      eventRequest1 = setting2BreakpointRequest (null,
351                                              testClassReference, methodName, bpLineName,
352                                              EventRequest.SUSPEND_NONE, property1);
353 
354                      log2("......getting: location2 = ((BreakpointRequest) eventRequest1).location();");
355                      location2 = ((BreakpointRequest) eventRequest1).location();
356 
357                      log2("      checking up on equality of location1 and location2");
358                      if ( !location1.equals(location2) ) {
359                          testExitCode = FAILED;
360                          log3("ERROR: Location objects are not equal");
361                      }
362 
363                      break;
364 
365               default:
366                       throw new JDITestRuntimeException("** default case 2 **");
367             }
368             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
369         }
370         log1("    TESTING ENDS");
371         return;
372     }

374    /*
375     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
376     *                                             String, String, String)
377     *
378     * It sets up a breakpoint at given line number within a given method in a given class
379     * for a given thread.
380     *
381     * Return value: BreakpointRequest object  in case of success
382     *
383     * JDITestRuntimeException   in case of an Exception thrown within the method
384     */
385 
386     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
387                                                   ReferenceType testedClass,
388                                                   String methodName,
389                                                   String bpLine,
390                                                   String property)
391             throws JDITestRuntimeException {
392 
393         log2("......setting up a breakpoint:");
394         log2("       thread: " + thread + "; class: " + testedClass +
395                         "; method: " + methodName + "; line: " + bpLine);
396 
397         List              alllineLocations = null;
398         Location          lineLocation     = null;
399         BreakpointRequest breakpRequest    = null;
400 
401         try {
402             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
403 
404             alllineLocations = method.allLineLocations();
405 
406             int n =
407                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
408             if (n > alllineLocations.size()) {
409                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
410             } else {
411                 lineLocation = (Location) alllineLocations.get(n);
412                 try {
413                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
414                     breakpRequest.putProperty("number", property);
415                     breakpRequest.addThreadFilter(thread);
416                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
417                 } catch ( Exception e1 ) {
418                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
419                     breakpRequest    = null;
420                 }
421             }
422         } catch ( Exception e2 ) {
423             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
424             breakpRequest    = null;
425         }
426 
427         if (breakpRequest == null) {
428             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
429             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
430         }
431 
432         log2("      a breakpoint has been set up");
433         return breakpRequest;
434     }

437     private void getEventSet()
438                  throws JDITestRuntimeException {
439         try {
440 //            log2("       eventSet = eventQueue.remove(waitTime);");
441             eventSet = eventQueue.remove(waitTime);
442             if (eventSet == null) {
443                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
444             }
445 //            log2("       eventIterator = eventSet.eventIterator;");
446             eventIterator = eventSet.eventIterator();
447         } catch ( Exception e ) {
448             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
449         }
450     }

453     private void breakpointForCommunication()
454                  throws JDITestRuntimeException {
455 
456         log2("breakpointForCommunication");
457         getEventSet();
458 
459         if (eventIterator.nextEvent() instanceof BreakpointEvent)
460             return;
461 
462         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
463     }

467     private BreakpointRequest setting2BreakpointRequest ( ThreadReference thread,
468                                                           ReferenceType   testedClass,
469                                                           String          methodName,
470                                                           String          bpLine,
471                                                           int             suspendPolicy,
472                                                           String          property        )
473             throws JDITestRuntimeException {
474         try {
475             log2("......setting up a breakpoint:");
476             log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
477 
478             int n = ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
479             Location loc = (Location) ((Method) testedClass.methodsByName(methodName).get(0)).allLineLocations().get(n);
480 
481             location1 = loc;
482 
483             BreakpointRequest
484             bpr = eventRManager.createBreakpointRequest(loc);
485             bpr.putProperty("number", property);
486             if (thread != null)
487                 bpr.addThreadFilter(thread);
488             bpr.setSuspendPolicy(suspendPolicy);
489 
490             log2("      a breakpoint has been set up");
491             return bpr;
492         } catch ( Exception e ) {
493             log3("ERROR: ATTENTION: Exception within settingBreakpointRequest() : " + e);
494             log3("       BreakpointRequest HAS NOT BEEN SET UP");
495             throw new JDITestRuntimeException("** FAILURE to set up BreakpointRequest **");
496         }
