 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.createExceptionRequest()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - Creates a new disabled ExceptionRequest.                           <BR>
 47  * - A specific exception type and its subclasses                       <BR>
 48  *   can be selected for exception events.                              <BR>
 49  * - Caught exceptions, uncaught exceptions, or both can be selected.   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee prepares new testcase and invokes                     <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent,                                  <BR>
 72  *   the debugger performs the checks required.                         <BR>
 73  * <BR>
 74  * In third phase, at the end of the test, the debuggee changes         <BR>
 75  * the value of the "instruction" which the debugger and debuggee       <BR>
 76  * use to inform each other of needed actions,  and both end.           <BR>
 77  * <BR>
 78  */

158     private int runThis (String argv[], PrintStream out) {
159 
160         argsHandler     = new ArgumentHandler(argv);
161         logHandler      = new Log(out, argsHandler);
162         Binder binder   = new Binder(argsHandler, logHandler);
163 
164         waitTime        = argsHandler.getWaitTime() * 60000;
165 
166         try {
167             log2("launching a debuggee :");
168             log2("       " + debuggeeName);
169             if (argsHandler.verbose()) {
170                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
171             } else {
172                 debuggee = binder.bindToDebugee(debuggeeName);
173             }
174             if (debuggee == null) {
175                 log3("ERROR: no debuggee launched");
176                 return FAILED;
177             }
178             log2("debuggee launched");
179         } catch ( Exception e ) {
180             log3("ERROR: Exception : " + e);
181             log2("       test cancelled");
182             return FAILED;
183         }
184 
185         debuggee.redirectOutput(logHandler);
186 
187         vm = debuggee.VM();
188 
189         eventQueue = vm.eventQueue();
190         if (eventQueue == null) {
191             log3("ERROR: eventQueue == null : TEST ABORTED");
192             vm.exit(PASS_BASE);
193             return FAILED;
194         }
195 
196         log2("invocation of the method runTest()");
197         switch (runTest()) {
198 
199             case 0 :  log2("test phase has finished normally");
200                       log2("   waiting for the debuggee to finish ...");
201                       debuggee.waitFor();
202 
203                       log2("......getting the debuggee's exit status");
204                       int status = debuggee.getStatus();
205                       if (status != PASS_BASE) {
206                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
207                               status + " != PASS_BASE");
208                           testExitCode = FAILED;
209                       } else {
210                           log2("......debuggee returned expected exit status: " +
211                               status + " == PASS_BASE");
212                       }
213                       break;
214 
215             default : log3("ERROR: runTest() returned unexpected value");
216 
217             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
218                       log2("......forcing: vm.exit();");
219                       testExitCode = FAILED;
220                       try {
221                           vm.exit(PASS_BASE);
222                       } catch ( Exception e ) {
223                           log3("ERROR: Exception : " + e);
224                       }
225                       break;
226 
227             case 2 :  log3("test cancelled due to VMDisconnectedException");
228                       log2("......trying: vm.process().destroy();");
229                       testExitCode = FAILED;
230                       try {
231                           Process vmProcess = vm.process();
232                           if (vmProcess != null) {
233                               vmProcess.destroy();
234                           }
235                       } catch ( Exception e ) {
236                           log3("ERROR: Exception : " + e);
237                       }
238                       break;
239             }
240 
241         return testExitCode;
242     }

247     *               1 - ubnormal end of the test
248     *               2 - VMDisconnectedException while test phase

251     private int runTest() {
252 
253         try {
254             testRun();
255 
256             log2("waiting for VMDeathEvent");
257             getEventSet();
258             if (eventIterator.nextEvent() instanceof VMDeathEvent)
259                 return 0;
260 
261             log3("ERROR: last event is not the VMDeathEvent");
262             return 1;
263         } catch ( VMDisconnectedException e ) {
264             log3("ERROR: VMDisconnectedException : " + e);
265             return 2;
266         } catch ( Exception e ) {
267             log3("ERROR: Exception : " + e);
268             return 1;
269         }
270 
271     }

273     private void testRun()
274                  throws JDITestRuntimeException, Exception {
275 
276         eventRManager = vm.eventRequestManager();
277 
278         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
279         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
280         cpRequest.addClassFilter(debuggeeName);
281 
282         cpRequest.enable();
283         vm.resume();
284         getEventSet();
285         cpRequest.disable();
286 
287         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
288         debuggeeClass = event.referenceType();
289 
290         if (!debuggeeClass.name().equals(debuggeeName))
291            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
292 
293         log2("      received: ClassPrepareEvent for debuggeeClass");
294 
295         String bPointMethod = "methodForCommunication";
296         String lineForComm  = "lineForComm";
297 
298         ThreadReference   mainThread = threadByName("main");
299 
300         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
301                                              debuggeeClass,
302                                             bPointMethod, lineForComm, "zero");
303         bpRequest.enable();
304 
305     //------------------------------------------------------  testing section
306 
307         String fieldName = "testField";
308 
309         ExceptionRequest exRequest = null;
310 
311 
312         log1("     TESTING BEGINS");
313 
314         for (int i = 0; ; i++) {
315 
316             vm.resume();
317             breakpointForCommunication();
318 
319             int instruction = ((IntegerValue)
320                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
321 
322             if (instruction == 0) {
323                 vm.resume();
324                 break;
325             }
326 
327             log1(":::::: case: # " + i);
328 
329             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
330 
331             ReferenceType refType = (ReferenceType)
332                                     debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName)).type();
333 
334             log2("......creating: exRequest = eventRManager.createExceptionRequest(refType, false, true);");
335             exRequest = eventRManager.createExceptionRequest(refType, false, true);
336             log2("       check on if request is disabled");
337             if (exRequest.isEnabled()) {
338                 log3("ERROR: request is enabled");
339                 testExitCode = FAILED;
340             }
341 
342             log2("......creating: exRequest = eventRManager.createExceptionRequest(null, true, false);");
343             exRequest = eventRManager.createExceptionRequest(null, true, false);
344             log2("       check on if request is disabled");
345             if (exRequest.isEnabled()) {
346                 log3("ERROR: request is enabled");
347                 testExitCode = FAILED;
348             }
349 
350             log2("......creating: exRequest = eventRManager.createExceptionRequest(null, true, true);");
351             exRequest = eventRManager.createExceptionRequest(null, true, true);
352             log2("       check on if request is disabled");
353             if (exRequest.isEnabled()) {
354                 log3("ERROR: request is enabled");
355                 testExitCode = FAILED;
356             }
357 
358             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
359         }
360         log1("    TESTING ENDS");
361         return;
362     }

364     private ThreadReference threadByName(String name)
365                  throws JDITestRuntimeException {
366 
367         List         all = vm.allThreads();
368         ListIterator li  = all.listIterator();
369 
370         for (; li.hasNext(); ) {
371             ThreadReference thread = (ThreadReference) li.next();
372             if (thread.name().equals(name))
373                 return thread;
374         }
375         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
376     }

378    /*
379     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
380     *                                             String, String, String)
381     *
382     * It sets up a breakpoint at given line number within a given method in a given class
383     * for a given thread.
384     *
385     * Return value: BreakpointRequest object  in case of success
386     *
387     * JDITestRuntimeException   in case of an Exception thrown within the method
388     */
389 
390     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
391                                                   ReferenceType testedClass,
392                                                   String methodName,
393                                                   String bpLine,
394                                                   String property)
395             throws JDITestRuntimeException {
396 
397         log2("......setting up a breakpoint:");
398         log2("       thread: " + thread + "; class: " + testedClass +
399                         "; method: " + methodName + "; line: " + bpLine);
400 
401         List              alllineLocations = null;
402         Location          lineLocation     = null;
403         BreakpointRequest breakpRequest    = null;
404 
405         try {
406             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
407 
408             alllineLocations = method.allLineLocations();
409 
410             int n =
411                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
412             if (n > alllineLocations.size()) {
413                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
414             } else {
415                 lineLocation = (Location) alllineLocations.get(n);
416                 try {
417                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
418                     breakpRequest.putProperty("number", property);
419                     breakpRequest.addThreadFilter(thread);
420                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
421                 } catch ( Exception e1 ) {
422                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
423                     breakpRequest    = null;
424                 }
425             }
426         } catch ( Exception e2 ) {
427             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
428             breakpRequest    = null;
429         }
430 
431         if (breakpRequest == null) {
432             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
433             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
434         }
435 
436         log2("      a breakpoint has been set up");
437         return breakpRequest;
438     }

441     private void getEventSet()
442                  throws JDITestRuntimeException {
443         try {
444 //            log2("       eventSet = eventQueue.remove(waitTime);");
445             eventSet = eventQueue.remove(waitTime);
446             if (eventSet == null) {
447                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
448             }
449 //            log2("       eventIterator = eventSet.eventIterator;");
450             eventIterator = eventSet.eventIterator();
451         } catch ( Exception e ) {
452             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
453         }
454     }

457     private void breakpointForCommunication()
458                  throws JDITestRuntimeException {
459 
460         log2("breakpointForCommunication");
461         getEventSet();
462 
463         if (eventIterator.nextEvent() instanceof BreakpointEvent)
464             return;
465 
466         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
467     }
