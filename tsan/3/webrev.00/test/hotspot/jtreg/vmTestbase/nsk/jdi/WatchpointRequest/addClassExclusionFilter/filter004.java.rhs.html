 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.addClassExclusionFilter()</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * The cases to test include ModificationWatchpointRequest.             <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 69  *   and to inform the debugger with the event.                         <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an ModificationWatchpointRequest and enables it,         <BR>
 72  *   - invokes the method addClassExclusionFilter() on the request              <BR>
 73  *     expecting to catch InvalidRequestStateException,                         <BR>
 74  *   - disables the request and invokes the method addClassExclusionFilter()    <BR>
 75  *     expecting to catch no InvalidRequestStateException,                      <BR>
 76  *   - deletes the request and invokes the method addClassExclusionFilter()     <BR>
 77  *     once again expecting to catch InvalidRequestStateException.              <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         eventRManager = vm.eventRequestManager();
279 
280         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
281         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
282         cpRequest.addClassFilter(debuggeeName);
283 
284         cpRequest.enable();
285         vm.resume();
286         getEventSet();
287         cpRequest.disable();
288 
289         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
290         debuggeeClass = event.referenceType();
291 
292         if (!debuggeeClass.name().equals(debuggeeName))
293            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
294 
295         log2("      received: ClassPrepareEvent for debuggeeClass");
296 
297         String bPointMethod = "methodForCommunication";
298         String lineForComm  = "lineForComm";
299 
300         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
301 
302         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
303                                              debuggeeClass,
304                                             bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         EventRequest  eventRequest1 = null;
312         String        property1     = "ModificationWatchpointRequest1";
313         String        fieldName1    = "var101";
314 
315         ReferenceType testClassReference = null;
316 
317 
318         for (int i = 0; ; i++) {
319 
320             vm.resume();
321             breakpointForCommunication();
322 
323             int instruction = ((IntegerValue)
324                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
325 
326             if (instruction == 0) {
327                 vm.resume();
328                 break;
329             }
330 
331 
332             log1(":::::: case: # " + i);
333 
334             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
335 
336             switch (i) {
337 
338               case 0:
339                      testClassReference =
340                            (ReferenceType) vm.classesByName(testedClassName1).get(0);
341 
342                      eventRequest1 = setting23ModificationWatchpointRequest (null,
343                                              testClassReference, fieldName1,
344                                              testedClassName1,
345                                              EventRequest.SUSPEND_NONE, property1);
346 
347                      eventRequest1.enable();
348 
349                      try {
350                          log2("......eventRequest1.addClassFilter(testedClassName1);");
351                          log2("        InvalidRequestStateException expected");
352                          ((ModificationWatchpointRequest)eventRequest1).addClassFilter(testedClassName1);
353                          log3("ERROR: no InvalidRequestStateException");
354                          testExitCode = FAILED;
355                      } catch ( InvalidRequestStateException e ) {
356                          log2("        InvalidRequestStateException");
357                      }
358 
359                      log2("......eventRequest1.disable();");
360                      eventRequest1.disable();
361                      try {
362                          log2("......eventRequest1.addClassFilter(testedClassName1);");
363                          log2("        no InvalidRequestStateException expected");
364                          ((ModificationWatchpointRequest)eventRequest1).addClassFilter(testedClassName1);
365                          log2("        no InvalidRequestStateException");
366                      } catch ( InvalidRequestStateException e ) {
367                          log3("ERROR: InvalidRequestStateException");
368                          testExitCode = FAILED;
369                      }
370 
371                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
372                      eventRManager.deleteEventRequest(eventRequest1);
373                      try {
374                          log2("......eventRequest1.addClassFilter(testedClassName1);");
375                          log2("        InvalidRequestStateException expected");
376                          ((ModificationWatchpointRequest)eventRequest1).addClassFilter(testedClassName1);
377                          log3("ERROR: no InvalidRequestStateException");
378                          testExitCode = FAILED;
379                      } catch ( InvalidRequestStateException e ) {
380                          log2("        InvalidRequestStateException");
381                      }
382                       break;
383 
384               default:
385                       throw new JDITestRuntimeException("** default case 2 **");
386             }
387 
388             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
389         }
390         log1("    TESTING ENDS");
391         return;

394    /*
395     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
396     *                                             String, String, String)
397     *
398     * It sets up a breakpoint at given line number within a given method in a given class
399     * for a given thread.
400     *
401     * Return value: BreakpointRequest object  in case of success
402     *
403     * JDITestRuntimeException   in case of an Exception thrown within the method
404     */
405 
406     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
407                                                   ReferenceType testedClass,
408                                                   String methodName,
409                                                   String bpLine,
410                                                   String property)
411             throws JDITestRuntimeException {
412 
413         log2("......setting up a breakpoint:");
414         log2("       thread: " + thread + "; class: " + testedClass +
415                         "; method: " + methodName + "; line: " + bpLine);
416 
417         List              alllineLocations = null;
418         Location          lineLocation     = null;
419         BreakpointRequest breakpRequest    = null;
420 
421         try {
422             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
423 
424             alllineLocations = method.allLineLocations();
425 
426             int n =
427                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
428             if (n > alllineLocations.size()) {
429                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
430             } else {
431                 lineLocation = (Location) alllineLocations.get(n);
432                 try {
433                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
434                     breakpRequest.putProperty("number", property);
435                     breakpRequest.addThreadFilter(thread);
436                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
437                 } catch ( Exception e1 ) {
438                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
439                     breakpRequest    = null;
440                 }
441             }
442         } catch ( Exception e2 ) {
443             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
444             breakpRequest    = null;
445         }
446 
447         if (breakpRequest == null) {
448             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
449             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
450         }
451 
452         log2("      a breakpoint has been set up");
453         return breakpRequest;
454     }

457     private void getEventSet()
458                  throws JDITestRuntimeException {
459         try {
460 //            log2("       eventSet = eventQueue.remove(waitTime);");
461             eventSet = eventQueue.remove(waitTime);
462             if (eventSet == null) {
463                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
464             }
465 //            log2("       eventIterator = eventSet.eventIterator;");
466             eventIterator = eventSet.eventIterator();
467         } catch ( Exception e ) {
468             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
469         }
470     }

473     private void breakpointForCommunication()
474                  throws JDITestRuntimeException {
475 
476         log2("breakpointForCommunication");
477         getEventSet();
478 
479         if (eventIterator.nextEvent() instanceof BreakpointEvent)
480             return;
481 
482         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
483     }

487     private ModificationWatchpointRequest setting23ModificationWatchpointRequest (
488                                                   ThreadReference thread,
489                                                   ReferenceType   fieldClass,
490                                                   String          fieldName,
491                                                   String          testedClass,
492                                                   int             suspendPolicy,
493                                                   String          property        )
494             throws JDITestRuntimeException {
495         try {
496             log2("......setting up ModificationWatchpointRequest:");
497             log2("       thread: " + thread + "; fieldClass: " + fieldClass + "; fieldName: " + fieldName);
498             Field field = fieldClass.fieldByName(fieldName);
499 
500             ModificationWatchpointRequest
501             awr = eventRManager.createModificationWatchpointRequest(field);
502             awr.putProperty("number", property);
503 
504             if (thread != null)
505                 awr.addThreadFilter(thread);
506             awr.setSuspendPolicy(suspendPolicy);
507             awr.addClassExclusionFilter(testedClass);
508 
509             log2("      ModificationWatchpointRequest has been set up");
510             return awr;
511         } catch ( Exception e ) {
512             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
513             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
514             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
515         }
516     }
