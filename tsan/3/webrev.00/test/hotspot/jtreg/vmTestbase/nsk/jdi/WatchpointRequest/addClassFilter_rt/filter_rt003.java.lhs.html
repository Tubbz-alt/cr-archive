 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.addClassFilter(ReferenceType)</code> <BR>
 43  * complies with its spec.                                              <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  *    Throws: InvalidRequestStateException -                            <BR>
 47  *            if this request is currently enabled or has been deleted. <BR>
 48  *            Filters may be added only to disabled requests.           <BR>
 49  * The cases to test include AccessWatchpointRequest.                   <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.                      <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee invokes the methodForCommunication to be suspended    <BR>
 69  *   and to inform the debugger with the event.                         <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an AccessWatchpointRequest and enables it,               <BR>
 72  *   - invokes the method addClassFilter() on the request               <BR>
 73  *     expecting to catch InvalidRequestStateException,                 <BR>
 74  *   - disables the request and invokes the method addClassFilter()     <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addClassFilter()      <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */
 84

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278 
279     }

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         eventRManager = vm.eventRequestManager();
285 
286         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
287         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
288         cpRequest.addClassFilter(debuggeeName);
289 
290         cpRequest.enable();
291         vm.resume();
292         getEventSet();
293         cpRequest.disable();
294 
295         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
296         debuggeeClass = event.referenceType();
297 
298         if (!debuggeeClass.name().equals(debuggeeName))
299            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
300 
301         log2("      received: ClassPrepareEvent for debuggeeClass");
302 
303         String bPointMethod = "methodForCommunication";
304         String lineForComm  = "lineForComm";
305 
306         ThreadReference   mainThread = threadByName("main");
307 
308         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
309                                              debuggeeClass,
310                                             bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313     //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         EventRequest  eventRequest1 = null;
318         String        property1     = "AccessWatchpointRequest1";
319         String        fieldName1    = "var101";
320 
321         ReferenceType testClassReference = null;
322 
323 
324         for (int i = 0; ; i++) {
325 
326             vm.resume();
327             breakpointForCommunication();
328 
329             int instruction = ((IntegerValue)
330                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
331 
332             if (instruction == 0) {
333                 vm.resume();
334                 break;
335             }
336 
337 
338             log1(":::::: case: # " + i);
339 
340             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
341 
342             switch (i) {
343 
344               case 0:
345                      testClassReference =
346                            (ReferenceType) vm.classesByName(testedClassName1).get(0);
347 
348                      eventRequest1 = setting21AccessWatchpointRequest (null,
349                                              testClassReference, fieldName1,
350                                              EventRequest.SUSPEND_NONE, property1);
351 
352                      eventRequest1.enable();
353 
354                      try {
355                          log2("......eventRequest1.addClassFilter(testedClassName1);");
356                          log2("        InvalidRequestStateException expected");
357                          ((AccessWatchpointRequest)eventRequest1).addClassFilter(testClassReference);
358                          log3("ERROR: no InvalidRequestStateException");
359                          testExitCode = FAILED;
360                      } catch ( InvalidRequestStateException e ) {
361                          log2("        InvalidRequestStateException");
362                      }
363 
364                      log2("......eventRequest1.disable();");
365                      eventRequest1.disable();
366                      try {
367                          log2("......eventRequest1.addClassFilter(testedClassName1);");
368                          log2("        no InvalidRequestStateException expected");
369                          ((AccessWatchpointRequest)eventRequest1).addClassFilter(testClassReference);
370                          log2("        no InvalidRequestStateException");
371                      } catch ( InvalidRequestStateException e ) {
372                          log3("ERROR: InvalidRequestStateException");
373                          testExitCode = FAILED;
374                      }
375 
376                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
377                      eventRManager.deleteEventRequest(eventRequest1);
378                      try {
379                          log2("......eventRequest1.addClassFilter(testedClassName1);");
380                          log2("        InvalidRequestStateException expected");
381                          ((AccessWatchpointRequest)eventRequest1).addClassFilter(testClassReference);
382                          log3("ERROR: no InvalidRequestStateException");
383                          testExitCode = FAILED;
384                      } catch ( InvalidRequestStateException e ) {
385                          log2("        InvalidRequestStateException");
386                      }
387                       break;
388 
389               default:
390                       throw new JDITestRuntimeException("** default case 2 **");
391             }
392 
393             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
394         }
395         log1("    TESTING ENDS");
396         return;
397     }

399     private ThreadReference threadByName(String name)
400                  throws JDITestRuntimeException {
401 
402         List         all = vm.allThreads();
403         ListIterator li  = all.listIterator();
404 
405         for (; li.hasNext(); ) {
406             ThreadReference thread = (ThreadReference) li.next();
407             if (thread.name().equals(name))
408                 return thread;
409         }
410         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
411     }

413    /*
414     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
415     *                                             String, String, String)
416     *
417     * It sets up a breakpoint at given line number within a given method in a given class
418     * for a given thread.
419     *
420     * Return value: BreakpointRequest object  in case of success
421     *
422     * JDITestRuntimeException   in case of an Exception thrown within the method
423     */
424 
425     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
426                                                   ReferenceType testedClass,
427                                                   String methodName,
428                                                   String bpLine,
429                                                   String property)
430             throws JDITestRuntimeException {
431 
432         log2("......setting up a breakpoint:");
433         log2("       thread: " + thread + "; class: " + testedClass +
434                         "; method: " + methodName + "; line: " + bpLine);
435 
436         List              alllineLocations = null;
437         Location          lineLocation     = null;
438         BreakpointRequest breakpRequest    = null;
439 
440         try {
441             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
442 
443             alllineLocations = method.allLineLocations();
444 
445             int n =
446                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
447             if (n > alllineLocations.size()) {
448                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
449             } else {
450                 lineLocation = (Location) alllineLocations.get(n);
451                 try {
452                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
453                     breakpRequest.putProperty("number", property);
454                     breakpRequest.addThreadFilter(thread);
455                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
456                 } catch ( Exception e1 ) {
457                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
458                     breakpRequest    = null;
459                 }
460             }
461         } catch ( Exception e2 ) {
462             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
463             breakpRequest    = null;
464         }
465 
466         if (breakpRequest == null) {
467             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
468             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
469         }
470 
471         log2("      a breakpoint has been set up");
472         return breakpRequest;
473     }

476     private void getEventSet()
477                  throws JDITestRuntimeException {
478         try {
479 //            log2("       eventSet = eventQueue.remove(waitTime);");
480             eventSet = eventQueue.remove(waitTime);
481             if (eventSet == null) {
482                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
483             }
484 //            log2("       eventIterator = eventSet.eventIterator;");
485             eventIterator = eventSet.eventIterator();
486         } catch ( Exception e ) {
487             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
488         }
489     }

492     private void breakpointForCommunication()
493                  throws JDITestRuntimeException {
494 
495         log2("breakpointForCommunication");
496         getEventSet();
497 
498         if (eventIterator.nextEvent() instanceof BreakpointEvent)
499             return;
500 
501         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
502     }

506     private AccessWatchpointRequest setting21AccessWatchpointRequest (
507                                                   ThreadReference thread,
508                                                   ReferenceType   fieldClass,
509                                                   String          fieldName,
510                                                   int             suspendPolicy,
511                                                   String          property        )
512             throws JDITestRuntimeException {
513         try {
514             log2("......setting up AccessWatchpointRequest:");
515             log2("       thread: " + thread + "; fieldClass: " + fieldClass + "; fieldName: " + fieldName);
516             Field field = fieldClass.fieldByName(fieldName);
517 
518             AccessWatchpointRequest
519             awr = eventRManager.createAccessWatchpointRequest(field);
520             awr.putProperty("number", property);
521 
522             if (thread != null)
523                 awr.addThreadFilter(thread);
524             awr.setSuspendPolicy(suspendPolicy);
525 
526             log2("      AccessWatchpointRequest has been set up");
527             return awr;
528         } catch ( Exception e ) {
529             log3("ERROR: ATTENTION: Exception within settingAccessWatchpointRequest() : " + e);
530             log3("       AccessWatchpointRequest HAS NOT BEEN SET UP");
531             throw new JDITestRuntimeException("** FAILURE to set up AccessWatchpointRequest **");
532         }
533     }
