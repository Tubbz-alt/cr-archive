 38 /**
 39  * The test for the implementation of an object of the type     <BR>
 40  * WatchpointRequest.                                           <BR>
 41  *                                                              <BR>
 42  * The test checks that results of the method                   <BR>
 43  * <code>com.sun.jdi.WatchpointRequest.addThreadFilter()</code> <BR>
 44  * complies with its spec.                                      <BR>
 45  * <BR>
 46  * The test checks up on the following assertion:               <BR>
 47  *    Throws: InvalidRequestStateException -                    <BR>
 48  *            if this request is currently enabled              <BR>
 49  *            or has been deleted.                              <BR>
 50  *            Filters may be added only to disabled requests.   <BR>
 51  * The test case includes ModificationWatchpointRequest.        <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * <BR>
 67  * In second phase to check the assertion,                              <BR>
 68  * the debugger and the debuggee perform the following.                 <BR>
 69  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 70  * - The debuggee creates new threads, thread1, and invokes             <BR>
 71  *   the methodForCommunication to be suspended and                     <BR>
 72  *   to inform the debugger with the event.                             <BR>
 73  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 74  *   - creates an ModificationWatchpointRequest and enables it,         <BR>
 75  *   - invokes the method addThreadFilter() on the request expecting    <BR>
 76  *     to catch InvalidRequestStateException,                           <BR>
 77  *   - disables the request and invokes the method addThreadFilter()    <BR>
 78  *     expecting to catch no InvalidRequestStateException,              <BR>
 79  *   - deletes the request and invokes the method addThreadFilter()     <BR>
 80  *     once again expecting to catch InvalidRequestStateException.      <BR>
 81  * <BR>
 82  * In third phase, at the end of the test,                              <BR>
 83  * the debuggee changes the value of the "instruction"  which           <BR>
 84  * the debugger and debuggee use to inform each other of needed actions,<BR>
 85  * and both end.
 86  * <BR>
 87  */

170     private int runThis (String argv[], PrintStream out) {
171 
172         argsHandler     = new ArgumentHandler(argv);
173         logHandler      = new Log(out, argsHandler);
174         Binder binder   = new Binder(argsHandler, logHandler);
175 
176         waitTime        = argsHandler.getWaitTime() * 60000;
177 
178         try {
179             log2("launching a debuggee :");
180             log2("       " + debuggeeName);
181             if (argsHandler.verbose()) {
182                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
183             } else {
184                 debuggee = binder.bindToDebugee(debuggeeName);
185             }
186             if (debuggee == null) {
187                 log3("ERROR: no debuggee launched");
188                 return FAILED;
189             }
190             log2("debuggee launched");
191         } catch ( Exception e ) {
192             log3("ERROR: Exception : " + e);
193             log2("       test cancelled");
194             return FAILED;
195         }
196 
197         debuggee.redirectOutput(logHandler);
198 
199         vm = debuggee.VM();
200 
201         eventQueue = vm.eventQueue();
202         if (eventQueue == null) {
203             log3("ERROR: eventQueue == null : TEST ABORTED");
204             vm.exit(PASS_BASE);
205             return FAILED;
206         }
207 
208         log2("invocation of the method runTest()");
209         switch (runTest()) {
210 
211             case 0 :  log2("test phase has finished normally");
212                       log2("   waiting for the debuggee to finish ...");
213                       debuggee.waitFor();
214 
215                       log2("......getting the debuggee's exit status");
216                       int status = debuggee.getStatus();
217                       if (status != PASS_BASE) {
218                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
219                               status + " != PASS_BASE");
220                           testExitCode = FAILED;
221                       } else {
222                           log2("......debuggee returned expected exit status: " +
223                               status + " == PASS_BASE");
224                       }
225                       break;
226 
227             default : log3("ERROR: runTest() returned unexpected value");
228 
229             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
230                       log2("......forcing: vm.exit();");
231                       testExitCode = FAILED;
232                       try {
233                           vm.exit(PASS_BASE);
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238 
239             case 2 :  log3("test cancelled due to VMDisconnectedException");
240                       log2("......trying: vm.process().destroy();");
241                       testExitCode = FAILED;
242                       try {
243                           Process vmProcess = vm.process();
244                           if (vmProcess != null) {
245                               vmProcess.destroy();
246                           }
247                       } catch ( Exception e ) {
248                           log3("ERROR: Exception : " + e);
249                       }
250                       break;
251             }
252 
253         return testExitCode;
254     }

263     private int runTest() {
264 
265         try {
266             testRun();
267 
268             log2("waiting for VMDeathEvent");
269             getEventSet();
270             if (eventIterator.nextEvent() instanceof VMDeathEvent)
271                 return 0;
272 
273             log3("ERROR: last event is not the VMDeathEvent");
274             return 1;
275         } catch ( VMDisconnectedException e ) {
276             log3("ERROR: VMDisconnectedException : " + e);
277             return 2;
278         } catch ( Exception e ) {
279             log3("ERROR: Exception : " + e);
280             return 1;
281         }

285     private void testRun()
286                  throws JDITestRuntimeException, Exception {
287 
288         if ( !vm.canWatchFieldModification() ) {
289             log2("......vm.canWatchFieldModification == false :: test cancelled");
290             vm.exit(PASS_BASE);
291             return;
292         }
293 
294 
295         eventRManager = vm.eventRequestManager();
296 
297         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
298         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
299         cpRequest.addClassFilter(debuggeeName);
300 
301         cpRequest.enable();
302         vm.resume();
303         getEventSet();
304         cpRequest.disable();
305 
306         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
307         debuggeeClass = event.referenceType();
308 
309         if (!debuggeeClass.name().equals(debuggeeName))
310            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
311 
312         log2("      received: ClassPrepareEvent for debuggeeClass");
313 
314         String bPointMethod = "methodForCommunication";
315         String lineForComm  = "lineForComm";
316 
317         ThreadReference   mainThread = threadByName("main");
318 
319         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
320                                              debuggeeClass,
321                                             bPointMethod, lineForComm, "zero");
322         bpRequest.enable();
323 
324     //------------------------------------------------------  testing section
325 
326         log1("     TESTING BEGINS");
327 
328         EventRequest eventRequest1 = null;
329 
330         ThreadReference thread1    = null;
331         String         thread1Name = "thread1";
332 
333         String property1 = "ModificationWatchpointRequest1";
334 
335         String fieldName = "var1";
336 
337         ReferenceType testClassReference = null;
338 
339         for (int i = 0; ; i++) {
340 
341             vm.resume();
342             breakpointForCommunication();
343 
344             int instruction = ((IntegerValue)
345                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
346 
347             if (instruction == 0) {
348                 vm.resume();
349                 break;
350             }
351 
352             log1(":::::: case: # " + i);
353 
354             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
355 
356             switch (i) {
357 
358               case 0:
359                       testClassReference =
360                            (ReferenceType) vm.classesByName(testedClassName).get(0);
361                       thread1 = (ThreadReference) debuggeeClass.getValue(
362                                                   debuggeeClass.fieldByName(thread1Name));
363 
364                       eventRequest1 = setting2ModificationWatchpointRequest (null,
365                                              testClassReference, fieldName,
366                                              EventRequest.SUSPEND_NONE, "ModificationWatchpointRequest1");
367 
368                      log2("......eventRequest1.enable();");
369                      eventRequest1.enable();
370 
371                      try {
372                          log2("......eventRequest1.addThreadFilter(thread1);");
373                          log2("        InvalidRequestStateException expected");
374                          ((ModificationWatchpointRequest)eventRequest1).addThreadFilter(thread1);
375                          log3("ERROR: no InvalidRequestStateException");
376                          testExitCode = FAILED;
377                      } catch ( InvalidRequestStateException e ) {
378                          log2("        InvalidRequestStateException");
379                      }
380 
381                      log2("......eventRequest1.disable();");
382                      eventRequest1.disable();
383                      try {
384                          log2("......eventRequest1.addThreadFilter(thread1);");
385                          log2("        no InvalidRequestStateException expected");
386                          ((ModificationWatchpointRequest)eventRequest1).addThreadFilter(thread1);
387                          log2("        no InvalidRequestStateException");
388                      } catch ( InvalidRequestStateException e ) {
389                          log3("ERROR: InvalidRequestStateException");
390                          testExitCode = FAILED;
391                      }
392 
393                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
394                      eventRManager.deleteEventRequest(eventRequest1);
395                      try {
396                          log2("......eventRequest1.addThreadFilter(thread1);");
397                          log2("        InvalidRequestStateException expected");
398                          ((ModificationWatchpointRequest)eventRequest1).addThreadFilter(thread1);
399                          log3("ERROR: no InvalidRequestStateException");
400                          testExitCode = FAILED;
401                      } catch ( InvalidRequestStateException e ) {
402                          log2("        InvalidRequestStateException");
403                      }
404                       break;
405 
406               default:
407                       throw new JDITestRuntimeException("** default case 2 **");
408             }
409 
410             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
411         }
412         log1("    TESTING ENDS");
413         return;
414     }

416     private ThreadReference threadByName(String name)
417                  throws JDITestRuntimeException {
418 
419         List         all = vm.allThreads();
420         ListIterator li  = all.listIterator();
421 
422         for (; li.hasNext(); ) {
423             ThreadReference thread = (ThreadReference) li.next();
424             if (thread.name().equals(name))
425                 return thread;
426         }
427         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
428     }

430    /*
431     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
432     *                                             String, String, String)
433     *
434     * It sets up a breakpoint at given line number within a given method in a given class
435     * for a given thread.
436     *
437     * Return value: BreakpointRequest object  in case of success
438     *
439     * JDITestRuntimeException   in case of an Exception thrown within the method
440     */
441 
442     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
443                                                   ReferenceType testedClass,
444                                                   String methodName,
445                                                   String bpLine,
446                                                   String property)
447             throws JDITestRuntimeException {
448 
449         log2("......setting up a breakpoint:");
450         log2("       thread: " + thread + "; class: " + testedClass +
451                         "; method: " + methodName + "; line: " + bpLine);
452 
453         List              alllineLocations = null;
454         Location          lineLocation     = null;
455         BreakpointRequest breakpRequest    = null;
456 
457         try {
458             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
459 
460             alllineLocations = method.allLineLocations();
461 
462             int n =
463                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
464             if (n > alllineLocations.size()) {
465                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
466             } else {
467                 lineLocation = (Location) alllineLocations.get(n);
468                 try {
469                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
470                     breakpRequest.putProperty("number", property);
471                     breakpRequest.addThreadFilter(thread);
472                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
473                 } catch ( Exception e1 ) {
474                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
475                     breakpRequest    = null;
476                 }
477             }
478         } catch ( Exception e2 ) {
479             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
480             breakpRequest    = null;
481         }
482 
483         if (breakpRequest == null) {
484             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
485             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
486         }
487 
488         log2("      a breakpoint has been set up");
489         return breakpRequest;
490     }

493     private void getEventSet()
494                  throws JDITestRuntimeException {
495         try {
496 //            log2("       eventSet = eventQueue.remove(waitTime);");
497             eventSet = eventQueue.remove(waitTime);
498             if (eventSet == null) {
499                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
500             }
501 //            log2("       eventIterator = eventSet.eventIterator;");
502             eventIterator = eventSet.eventIterator();
503         } catch ( Exception e ) {
504             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
505         }
506     }

509     private void breakpointForCommunication()
510                  throws JDITestRuntimeException {
511 
512         log2("breakpointForCommunication");
513         getEventSet();
514 
515         if (eventIterator.nextEvent() instanceof BreakpointEvent)
516             return;
517 
518         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
519     }

523     private ModificationWatchpointRequest setting2ModificationWatchpointRequest (
524                                                   ThreadReference thread,
525                                                   ReferenceType   testedClass,
526                                                   String          fieldName,
527                                                   int             suspendPolicy,
528                                                   String          property        )
529             throws JDITestRuntimeException {
530         try {
531             log2("......setting up ModificationWatchpointRequest:");
532             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
533             Field field = testedClass.fieldByName(fieldName);
534 
535             ModificationWatchpointRequest
536             awr = eventRManager.createModificationWatchpointRequest(field);
537             awr.putProperty("number", property);
538 
539             if (thread != null)
540                 awr.addThreadFilter(thread);
541             awr.setSuspendPolicy(suspendPolicy);
542 
543             log2("      ModificationWatchpointRequest has been set up");
544             return awr;
545         } catch ( Exception e ) {
546             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
547             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
548             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
549         }
