 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * <BR>
 46  * The test checks up that locks acquired by a popped frame,    <BR>
 47  * are released when it is popped.                              <BR>
 48  * This applies to synchronized methods being popped.           <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to perform the above check,                          <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates two threads, thread2 and thread3, which will  <BR>
 68  * call a tested synchronized poppedMethod() from their run() methods.  <BR>
 69  * After getting the threads started, the debuggee invokes the          <BR>
 70  * methodForCommunication() to inform the debugger of the threads.      <BR>
 71  * - The debugger sets up Breakpoint and MethodEntry Requests           <BR>
 72  * within the poppedMethod(), first one for the thread2 and             <BR>
 73  * second one for the thread3, and resumes debuggee's main thread.      <BR>
 74  * - The main thread releases the thread2 and the thread3 in the order  <BR>
 75  * which makes the thread2 to be suspended at the breakpoint within     <BR>
 76  * the poppedMethod and the thread3 to be waiting for releasing the     <BR>
 77  * synchronized poppedMethod by the thread2. Then                       <BR>
 78  * the main thread informs the debugger via methodForCommunication().   <BR>
 79  * - After getting both events, from the thread2 and the main,          <BR>
 80  * the debugger invokes the popFrames() method and waits for            <BR>
 81  * the MethodEntry event in poppedMethod from the thread3.              <BR>
 82  * If no the event, the test FAILED.                                    <BR>
 83  */

169     private int runThis (String argv[], PrintStream out) {
170 
171         argsHandler     = new ArgumentHandler(argv);
172         logHandler      = new Log(out, argsHandler);
173         Binder binder   = new Binder(argsHandler, logHandler);
174 
175         waitTime        = argsHandler.getWaitTime() * 60000;
176 
177         try {
178             log2("launching a debuggee :");
179             log2("       " + debuggeeName);
180             if (argsHandler.verbose()) {
181                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
182             } else {
183                 debuggee = binder.bindToDebugee(debuggeeName);
184             }
185             if (debuggee == null) {
186                 log3("ERROR: no debuggee launched");
187                 return FAILED;
188             }
189             log2("debuggee launched");
190         } catch ( Exception e ) {
191             log3("ERROR: Exception : " + e);
192             log2("       test cancelled");
193             return FAILED;
194         }
195 
196         debuggee.redirectOutput(logHandler);
197 
198         vm = debuggee.VM();
199 
200         eventQueue = vm.eventQueue();
201         if (eventQueue == null) {
202             log3("ERROR: eventQueue == null : TEST ABORTED");
203             vm.exit(PASS_BASE);
204             return FAILED;
205         }
206 
207         log2("invocation of the method runTest()");
208         switch (runTest()) {
209 
210             case 0 :  log2("test phase has finished normally");
211                       log2("   waiting for the debuggee to finish ...");
212                       debuggee.waitFor();
213 
214                       log2("......getting the debuggee's exit status");
215                       int status = debuggee.getStatus();
216                       if (status != PASS_BASE) {
217                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
218                               status + " != PASS_BASE");
219                           testExitCode = FAILED;
220                       } else {
221                           log2("......debuggee returned expected exit status: " +
222                               status + " == PASS_BASE");
223                       }
224                       break;
225 
226             default : log3("ERROR: runTest() returned unexpected value");
227 
228             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
229                       log2("......forcing: vm.exit();");
230                       testExitCode = FAILED;
231                       try {
232                           vm.exit(PASS_BASE);
233                       } catch ( Exception e ) {
234                           log3("ERROR: Exception : e");
235                       }
236                       break;
237 
238             case 2 :  log3("test cancelled due to VMDisconnectedException");
239                       log2("......trying: vm.process().destroy();");
240                       testExitCode = FAILED;
241                       try {
242                           Process vmProcess = vm.process();
243                           if (vmProcess != null) {
244                               vmProcess.destroy();
245                           }
246                       } catch ( Exception e ) {
247                           log3("ERROR: Exception : e");
248                       }
249                       break;
250             }
251 
252         return testExitCode;
253     }

262     private int runTest() {
263 
264         try {
265             testRun();
266 
267             log2("waiting for VMDeathEvent");
268             getEventSet();
269             if (eventIterator.nextEvent() instanceof VMDeathEvent)
270                 return 0;
271 
272             log3("ERROR: last event is not the VMDeathEvent");
273             return 1;
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         eventRManager = vm.eventRequestManager();
288 
289         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
290         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
291         cpRequest.addClassFilter(debuggeeName);
292 
293         cpRequest.enable();
294         vm.resume();
295         getEventSet();
296         cpRequest.disable();
297 
298         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
299         debuggeeClass = event.referenceType();
300 
301         if (!debuggeeClass.name().equals(debuggeeName))
302            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
303 
304         log2("      received: ClassPrepareEvent for debuggeeClass");
305 
306 
307         if ( !vm.canPopFrames() ) {
308             log2("......vm.canPopFrames() == false : test is cancelled");
309             vm.resume();
310             return;
311         }
312 
313         String bPointMethod = "methodForCommunication";
314         String lineForComm  = "lineForComm";
315 
316         ThreadReference threadMainRef = threadByName("main");
317         try {
318             bpRequest = settingBreakpoint(threadMainRef,
319                                           debuggeeClass,
320                                           bPointMethod, lineForComm, "zero");
321         } catch ( Exception e ) {
322             throw e;
323         }
324         bpRequest.enable();
325 
326     //------------------------------------------------------  testing section
327 
328         log1("     TESTING BEGINS");
329 
330         EventSet mainThreadEventSet = null;
331 
332         int flag = 0;
333 
334 
335         vm.resume();
336         breakpointForCommunication();
337 
338         log2("......setting MethodEntryRequest (meRequest) in ForCommunication.methodForCommunication");
339         meRequest = settingMethodEntryRequest(threadMainRef,
340                                               debuggeeName + "$ForCommunication",
341                                               "zero");
342         log2("meRequest.enable();");
343         meRequest.enable();
344 
345 
346         String thread2Name         = "thread2";
347         ThreadReference thread2Ref = threadByName(thread2Name);
348 
349         String thread3Name         = "thread3";
350         ThreadReference thread3Ref = threadByName(thread3Name);
351 
352         String poppedMethod    = "poppedMethod";
353         String breakpointLine  = "breakpointLine";
354         String testVariable    = "testVar1";
355 
356         log2("......setting BreakpointRequest (bpRequest2) in poppedMethod");
357         List classes = vm.classesByName(debuggeeName + "$Popped");
358         ReferenceType poppedClass = (ReferenceType) classes.get(0);
359         bpRequest2 = settingBreakpoint(thread2Ref, poppedClass,
360                                           poppedMethod, breakpointLine, "one");
361         log2("......bpRequest2.enable();");
362         bpRequest2.enable();
363 
364         log2("......setting MethodEntryRequest (meRequest3) in poppedMethod");
365         meRequest3 = settingMethodEntryRequest(thread3Ref, debuggeeName + "$Popped",
366                                           "two");
367         log2("......meRequest3.enable();");
368         meRequest3.enable();
369 
370         log2("......eventSet.resume();");
371         eventSet.resume();
372 
373         log2(".....waiting for Breakpoint(thread2) and MethodEntry(main) Events");
374         for ( ; ; ) {
375 
376             Event event1;
377 
378             if (flag == 3)
379                 break;
380 
381             getEventSet();
382             event1 = eventIterator.nextEvent();
383 
384             if ( event1 instanceof MethodEntryEvent ) {
385                 mainThreadEventSet = eventSet;
386                 flag |= 1;
387                 log2("......event1 instanceof MethodEntryEvent");
388             } else if ( event1 instanceof BreakpointEvent ) {
389                 flag |= 2;
390                 log2("......event1 instanceof BreakpointEvent");
391             } else
392                 throw new JDITestRuntimeException("** event1 IS NOT Breakpoint or MethodEntry **");
393         }
394 
395         log2("......bpRequest2.disable();");
396         bpRequest2.disable();
397 
398         label0:
399         {
400 //            log2("        THREAD_STATUS_RUNNING == " + ThreadReference.THREAD_STATUS_RUNNING );
401 //            log2("        thread2's status      == " + thread2Ref.status());
402 
403             if (thread2Ref.isSuspended())
404                 log2("         thread2Ref.isSuspended()");
405             else {
406                 log3("ERROR:  !thread2Ref.isSuspended()");
407                 testExitCode = FAILED;
408                 break label0;
409             }
410 
411             log2("......mainThreadEventSet.resume();");
412             mainThreadEventSet.resume();
413 
414             log2("......up to WAITTIME loop of sleeping");
415             log2("         to get thread3 waiting on monitor after calling poppedMethod");
416             int thread3Status = 0;
417             for ( int i1 = 0; i1 < waitTime; i1 += 10000) {
418                 if (thread3Ref.status() == ThreadReference.THREAD_STATUS_MONITOR) {
419                     thread3Status = 1;
420                     break;
421                 }
422                 Thread.sleep(10000);
423             }
424             if (thread3Status != 1) {
425                 log3("ERROR: thread3Ref.status() != THREAD_STATUS_MONITOR");
426                 testExitCode = FAILED;
427                 break label0;
428             }
429 
430             log2("......StackFrame stackFrame = thread2Ref.frame(0);");
431             StackFrame stackFrame = thread2Ref.frame(0);
432 
433             log2("......thread2Ref.popFrames(stackFrame);");
434             try {
435                 thread2Ref.popFrames(stackFrame);
436             } catch ( IncompatibleThreadStateException e ) {
437                 log3("ERROR: IncompatibleThreadStateException");
438                 testExitCode = FAILED;
439                 break label0;
440             }
441 
442             log2(".....waiting for MethodEntry (main and thread3) Events");
443             {
444                 flag = 0;
445 
446                 for ( ; ; ) {
447 
448                     Event event1;
449 
450                     if (flag == 3)
451                         break;
452 
453                     getEventSet();
454                     event1 = eventIterator.nextEvent();
455 
456                     if ( !(event1 instanceof MethodEntryEvent) )
457                         throw new JDITestRuntimeException("** event IS NOT MethodEntry **");
458 
459                     if ( event1.request().getProperty("number").equals("zero") ) {
460                         flag |= 1;
461                         log2("......MethodEntry in mainThread");
462                     } else {
463                         flag |= 2;
464                         log2("......MethodEntry in thread3");
465                     }
466                 }
467             }
468 
469         }
470 
471         log2("......disabling requests and resuming vm");
472         meRequest3.disable();
473         vm.resume();
474 
475     //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
476 
477         log1("    TESTING ENDS");
478         return;
479     }

481     private ThreadReference threadByName(String name)
482                  throws JDITestRuntimeException {
483 
484         List         all = vm.allThreads();
485         ListIterator li  = all.listIterator();
486 
487         for (; li.hasNext(); ) {
488             ThreadReference thread = (ThreadReference) li.next();
489             if (thread.name().equals(name))
490                 return thread;
491         }
492         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
493     }

496   /*
497     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
498     *                                             String, String, String)
499     *
500     * It sets up a breakpoint at given line number within a given method in a given class
501     * for a given thread.
502     *
503     * Return value: BreakpointRequest object  in case of success
504     *
505     * JDITestRuntimeException   in case of an Exception thrown within the method
506     */
507 
508     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
509                                                   ReferenceType testedClass,
510                                                   String methodName,
511                                                   String bpLine,
512                                                   String property)
513             throws JDITestRuntimeException {
514 
515         log2("......setting up a breakpoint:");
516         log2("       thread: " + thread + "; class: " + testedClass +
517                         "; method: " + methodName + "; line: " + bpLine);
518 
519         List              alllineLocations = null;
520         Location          lineLocation     = null;
521         BreakpointRequest breakpRequest    = null;
522 
523         try {
524             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
525 
526             alllineLocations = method.allLineLocations();
527 
528             int n =
529                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
530             if (n > alllineLocations.size()) {
531                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
532             } else {
533                 lineLocation = (Location) alllineLocations.get(n);
534                 try {
535                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
536                     breakpRequest.putProperty("number", property);
537                     breakpRequest.addThreadFilter(thread);
538                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
539                 } catch ( Exception e1 ) {
540                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
541                     breakpRequest    = null;
542                 }
543             }
544         } catch ( Exception e2 ) {
545             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
546             breakpRequest    = null;
547         }
548 
549         if (breakpRequest == null) {
550             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
551             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
552         }
553 
554         log2("      a breakpoint has been set up");
555         return breakpRequest;
556     }

559    /*
560     * private MethodEntryRequest settingMethodEntryRequest(ThreadReference, ReferenceType,
561     *                                             String)
562     *
563     * It sets up a MethodEntryRequest in a given class for a given thread.
564     *
565     * Return value: MethodEntryRequest object  in case of success
566     *
567     * JDITestRuntimeException   in case of an Exception thrown within the method
568     */

570     private MethodEntryRequest settingMethodEntryRequest ( ThreadReference thread,
571                                                          String className,
572                                                          String property)
573             throws JDITestRuntimeException {
574 
575         log2("......setting up a methodEntryRequest:");
576         log2("       thread: " + thread + "; className: " + className +
577                         "; property: " + property);
578 
579         List classList = vm.classesByName(className);
580         ReferenceType classRef = (ReferenceType) classList.get(0);
581 
582         MethodEntryRequest methodEntryRequest = null;
583 
584         try {
585              methodEntryRequest = eventRManager.createMethodEntryRequest();
586              methodEntryRequest.putProperty("number", property);
587              methodEntryRequest.addThreadFilter(thread);
588              methodEntryRequest.addClassFilter(classRef);
589              methodEntryRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
590         } catch ( Exception e ) {
591              log3("ERROR: Exception within methodEntryRequest() : " + e);
592              throw new JDITestRuntimeException("** FAILURE to set up a MethodEntryRequest **");
593         }
594 
595         log2("      a methodEntryRequest has been set up");
596         return methodEntryRequest;
597     }
