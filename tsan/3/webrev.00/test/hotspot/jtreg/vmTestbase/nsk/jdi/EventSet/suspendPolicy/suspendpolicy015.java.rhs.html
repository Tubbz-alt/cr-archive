 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a VMDeathEvent set,                 <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event set containing two more     <BR>
 49  * VMDeathRequests with suspendPolicies                         <BR>
 50  *      SUSPEND_ALL and SUSPEND_NONE.                           <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the above,                                  <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger sets up VMDeathRequests, resumes                      <BR>
 69  *   the debuggee, and waits for the VMDeathEvent.                      <BR>
 70  * - The debuggee ends to be resulting in the event.                    <BR>
 71  * - Upon getting new event, the debugger                               <BR>
 72  *   performs the check corresponding to the event.                     <BR>
 73  * <BR>
 74  * Note. To inform each other of needed actions, the debugger and       <BR>
 75  *       and the debuggee use debuggee's variable "instruction".        <BR>
 76  * In third phase, at the end,                                          <BR>
 77  * the debuggee changes the value of the "instruction"                  <BR>
 78  * to inform the debugger of checks finished, and both end.             <BR>
 79  * <BR>
 80  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
260                 log3("ERROR: last event is not the VMDeathEvent");
261                 return 1;
262             }
263             check();
264 
265             log2("waiting for VMDisconnectEvent");
266             getEventSet();
267             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
268                 log3("ERROR: last event is not the VMDisconnectEvent");
269                 return 1;
270             }
271 
272             return 0;
273 
274         } catch ( VMDisconnectedException e ) {
275             log3("ERROR: VMDisconnectedException : " + e);
276             return 2;
277         } catch ( Exception e ) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }
281 
282     }

284     private void testRun()
285                  throws JDITestRuntimeException, Exception {
286 
287         if ( !vm.canRequestVMDeathEvent() ) {
288             log2("......vm.canRequestVMDeathEvent == false :: test cancelled");
289             vm.exit(PASS_BASE);
290             return;
291         }
292 
293 
294         eventRManager = vm.eventRequestManager();
295 
296         log2("......getting ClassPrepareEvent for debuggee's class");
297         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
298         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
299         cpRequest.addClassFilter(debuggeeName);
300         cpRequest.enable();
301         vm.resume();
302         getEventSet();
303         cpRequest.disable();
304 
305         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
306         debuggeeClass = event.referenceType();
307 
308         if (!debuggeeClass.name().equals(debuggeeName))
309            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
310         log2("      received: ClassPrepareEvent for debuggeeClass");
311 
312         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
313 
314         String            bPointMethod = "methodForCommunication";
315         String            lineForComm  = "lineForComm";
316         BreakpointRequest bpRequest;
317         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
318         bpRequest = settingBreakpoint(mainThread,
319                                       debuggeeClass,
320                                       bPointMethod, lineForComm, "zero");
321         bpRequest.enable();
322 
323         vm.resume();
324 
325     //------------------------------------------------------  testing section
326 
327         log1("     TESTING BEGINS");
328 
329         EventRequest eventRequest1 = null;
330         EventRequest eventRequest2 = null;
331         EventRequest eventRequest3 = null;
332 
333         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
334         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
335         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
336         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
337 
338         int policyExpected[] = {    SUSPEND_NONE,
339                                     SUSPEND_THREAD,
340                                     SUSPEND_ALL,
341                                     SUSPEND_THREAD,
342                                     SUSPEND_ALL,
343                                     SUSPEND_ALL,
344                                     SUSPEND_ALL      };
345         int policy = 0;
346 
347         breakpointForCommunication();
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351         eventRequest1 = settingVMDeathRequest (SUSPEND_NONE, "VMDeathRequest1");
352         eventRequest1.enable();
353         eventRequest3 = settingVMDeathRequest (SUSPEND_ALL, "VMDeathRequest3");
354         eventRequest3.enable();
355 
356         policyToCheck = policyExpected[4];
357         mainThread.resume();
358 
359             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
360 
361         log1("    TESTING ENDS");
362         return;
363     }

365    /*
366     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
367     *                                             String, String, String)
368     *
369     * It sets up a breakpoint at given line number within a given method in a given class
370     * for a given thread.
371     *
372     * Return value: BreakpointRequest object  in case of success
373     *
374     * JDITestRuntimeException   in case of an Exception thrown within the method
375     */
376 
377     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
378                                                   ReferenceType testedClass,
379                                                   String methodName,
380                                                   String bpLine,
381                                                   String property)
382             throws JDITestRuntimeException {
383 
384         log2("......setting up a breakpoint:");
385         log2("       thread: " + thread + "; class: " + testedClass +
386                         "; method: " + methodName + "; line: " + bpLine);
387 
388         List              alllineLocations = null;
389         Location          lineLocation     = null;
390         BreakpointRequest breakpRequest    = null;
391 
392         try {
393             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
394 
395             alllineLocations = method.allLineLocations();
396 
397             int n =
398                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
399             if (n > alllineLocations.size()) {
400                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
401             } else {
402                 lineLocation = (Location) alllineLocations.get(n);
403                 try {
404                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
405                     breakpRequest.putProperty("number", property);
406                     breakpRequest.addThreadFilter(thread);
407                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
408                 } catch ( Exception e1 ) {
409                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
410                     breakpRequest    = null;
411                 }
412             }
413         } catch ( Exception e2 ) {
414             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
415             breakpRequest    = null;
416         }
417 
418         if (breakpRequest == null) {
419             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
420             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
421         }
422 
423         log2("      a breakpoint has been set up");
424         return breakpRequest;
425     }

428     private void getEventSet()
429                  throws JDITestRuntimeException {
430         try {
431 //            log2("       eventSet = eventQueue.remove(waitTime);");
432             eventSet = eventQueue.remove(waitTime);
433             if (eventSet == null) {
434                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
435             }
436 //            log2("       eventIterator = eventSet.eventIterator;");
437             eventIterator = eventSet.eventIterator();
438         } catch ( Exception e ) {
439             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
440         }
441     }

444     private void breakpointForCommunication()
445                  throws JDITestRuntimeException {
446 
447         log2("breakpointForCommunication");
448         getEventSet();
449 
450         if (eventIterator.nextEvent() instanceof BreakpointEvent)
451             return;
452 
453         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
454     }

458     private VMDeathRequest settingVMDeathRequest( int    suspendPolicy,
459                                                   String property       )
460             throws JDITestRuntimeException {
461         try {
462             log2("......setting up VMDeathRequest:");
463             log2("      suspendPolicy: " + suspendPolicy + "; property: " + property);
464 
465             VMDeathRequest
466             vmdr = eventRManager.createVMDeathRequest();
467             vmdr.putProperty("number", property);
468             vmdr.setSuspendPolicy(suspendPolicy);
469 
470             return vmdr;
471         } catch ( Exception e ) {
472             log3("ERROR: ATTENTION: Exception within settingVMDeathRequest() : " + e);
473             log3("       VMDeathRequest HAS NOT BEEN SET UP");
474             throw new JDITestRuntimeException("** FAILURE to set up a VMDeathRequest **");
475         }
476     }

478     void check() {
479         log2("......checking up on eventSet.suspendPolicy()");
480         int policy = eventSet.suspendPolicy();
481         if (policy != policyToCheck) {
482             log3("ERROR: eventSet.suspendPolicy() != policyExpected");
483             log3("       eventSet.suspendPolicy() == " + policy);
484             log3("       policyExpected           == " + policyToCheck);
485             testExitCode = FAILED;
486         }
487         vm.resume();
488     }
