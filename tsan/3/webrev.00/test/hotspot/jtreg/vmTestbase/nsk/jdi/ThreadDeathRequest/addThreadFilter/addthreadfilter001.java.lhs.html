 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadDeathRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ThreadDeathRequest.addThreadFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that the method restricts the events         <BR>
 46  * genereted by this request to those in the thread to filter.  <BR>
 47  * <BR>
 48  * The test has three phases and works as follows.              <BR>
 49  * <BR>
 50  * In first phase,                                                      <BR>
 51  * upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 55  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 56  * and waits for the event within the predefined time interval.         <BR>
 57  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 58  * Upon getting the ClassPrepareEvent,                                  <BR>
 59  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 60  * within debuggee's special methodForCommunication().                  <BR>
 61  * <BR>
 62  * In second phase to check the assetion,                               <BR>
 63  * the debugger and the debuggee perform the following.                 <BR>
 64  * - The debugger creates three ThreadDeathRequests with                <BR>
 65  *   addThreadFilter(debuggee's thread2) invoked only on third Request, <BR>
 66  *   resumes the debuggee and waits for the ThreadDeathEvent.           <BR>
 67  * - Debuggee's main thread creates and starts new thread, thread2, and <BR>
 68  *   invokes the methodForCommunication                                 <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting ThreadDeathEvent, the debugger checks                 <BR>
 71  *   if EventSet contains all three events; if no, the test FAILED.     <BR>
 72  * Note. To inform each other of needed actions, the debugger and       <BR>
 73  *       and the debuggee use debuggee's variable "instruction".        <BR>
 74  * <BR>
 75  * In third phase, at the end of the test,                              <BR>
 76  * the debuggee changes the value of the "instruction"                  <BR>
 77  * to inform the debugger of checks finished, and both end.             <BR>
 78  * <BR>
 79  */

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         String bPointMethod = "methodForCommunication";
297         String lineForComm  = "lineForComm";
298         BreakpointRequest bpRequest;
299 
300         ThreadReference mainThread = threadByName("main");
301 
302         bpRequest = settingBreakpoint(mainThread,
303                                       debuggeeClass,
304                                       bPointMethod, lineForComm, "zero");
305         bpRequest.enable();
306 
307     //------------------------------------------------------  testing section
308 
309         log1("     TESTING BEGINS");
310 
311         for (int i = 0; ; i++) {
312 
313             vm.resume();
314             breakpointForCommunication();
315 
316             int instruction = ((IntegerValue)
317                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
318 
319             if (instruction == 0) {
320                 vm.resume();
321                 break;
322             }
323 
324             log1(":::::: case: # " + i);
325 
326             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
327 
328             ThreadReference testThread = (ThreadReference)
329                                   debuggeeClass.getValue(debuggeeClass.fieldByName("thread2"));
330 
331         log2("......setting up ThreadDeathRequest");
332         ThreadDeathRequest tdr1 = eventRManager.createThreadDeathRequest();
333 //        tdr1.addThreadFilter(mainThread);
334         tdr1.addCountFilter(1);
335         tdr1.setSuspendPolicy(EventRequest.SUSPEND_ALL);
336         tdr1.putProperty("number", "ThreadDeathRequest1");
337         tdr1.enable();
338 
339         ThreadDeathRequest tdr2 = eventRManager.createThreadDeathRequest();
340 //        tsr2.addThreadFilter(mainThread);
341         tdr2.addCountFilter(1);
342         tdr2.setSuspendPolicy(EventRequest.SUSPEND_ALL);
343         tdr2.putProperty("number", "ThreadDeathRequest2");
344         tdr2.enable();
345 
346         ThreadDeathRequest tdr3 = eventRManager.createThreadDeathRequest();
347         tdr3.addThreadFilter(testThread);
348         tdr3.addCountFilter(1);
349         tdr3.setSuspendPolicy(EventRequest.SUSPEND_ALL);
350         tdr3.putProperty("number", "ThreadDeathRequest3");
351         tdr3.enable();
352 
353         log2("......vm.resume();");
354         vm.resume();
355 
356         log2("......waiting for ThreadDeathEvent");
357         getEventSet();
358         EventSet eventSet1 = eventSet;
359         if ( !(eventIterator.nextEvent() instanceof ThreadDeathEvent) ) {
360             testExitCode = FAILED;
361             log3("ERROR: new event is not ThreadDeathEvent");
362             return;
363         }
364         tdr1.disable();
365         tdr2.disable();
366         tdr3.disable();
367 
368         int setSize = eventSet1.size();
369         log2("......eventSet1.size() == " + setSize);
370         if (setSize != 3) {
371             log3("ERROR: eventSet1.size() != 3  :: " + setSize);
372             testExitCode = FAILED;
373 
374             EventIterator evI1 = eventSet1.eventIterator();
375             while (evI1.hasNext()) {
376                 Event ev1 = evI1.nextEvent();
377                 log2("........event property : " + ev1.request().getProperty("number"));
378             }
379         }
380 
381             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
382         }
383         log1("    TESTING ENDS");
384         return;
385     }

387     private ThreadReference threadByName(String name)
388                  throws JDITestRuntimeException {
389 
390         List         all = vm.allThreads();
391         ListIterator li  = all.listIterator();
392 
393         for (; li.hasNext(); ) {
394             ThreadReference thread = (ThreadReference) li.next();
395             if (thread.name().equals(name))
396                 return thread;
397         }
398         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
399     }

401    /*
402     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
403     *                                             String, String, String)
404     *
405     * It sets up a breakpoint at given line number within a given method in a given class
406     * for a given thread.
407     *
408     * Return value: BreakpointRequest object  in case of success
409     *
410     * JDITestRuntimeException   in case of an Exception thrown within the method
411     */
412 
413     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
414                                                   ReferenceType testedClass,
415                                                   String methodName,
416                                                   String bpLine,
417                                                   String property)
418             throws JDITestRuntimeException {
419 
420         log2("......setting up a breakpoint:");
421         log2("       thread: " + thread + "; class: " + testedClass +
422                         "; method: " + methodName + "; line: " + bpLine);
423 
424         List              alllineLocations = null;
425         Location          lineLocation     = null;
426         BreakpointRequest breakpRequest    = null;
427 
428         try {
429             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
430 
431             alllineLocations = method.allLineLocations();
432 
433             int n =
434                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
435             if (n > alllineLocations.size()) {
436                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
437             } else {
438                 lineLocation = (Location) alllineLocations.get(n);
439                 try {
440                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
441                     breakpRequest.putProperty("number", property);
442                     breakpRequest.addThreadFilter(thread);
443                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
444                 } catch ( Exception e1 ) {
445                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
446                     breakpRequest    = null;
447                 }
448             }
449         } catch ( Exception e2 ) {
450             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
451             breakpRequest    = null;
452         }
453 
454         if (breakpRequest == null) {
455             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
456             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
457         }
458 
459         log2("      a breakpoint has been set up");
460         return breakpRequest;
461     }

464     private void getEventSet()
465                  throws JDITestRuntimeException {
466         try {
467 //            log2("       eventSet = eventQueue.remove(waitTime);");
468             eventSet = eventQueue.remove(waitTime);
469             if (eventSet == null) {
470                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
471             }
472 //            log2("       eventIterator = eventSet.eventIterator;");
473             eventIterator = eventSet.eventIterator();
474         } catch ( Exception e ) {
475             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
476         }
477     }

480     private void breakpointForCommunication()
481                  throws JDITestRuntimeException {
482 
483         log2("breakpointForCommunication");
484         getEventSet();
485 
486         if (eventIterator.nextEvent() instanceof BreakpointEvent)
487             return;
488 
489         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
