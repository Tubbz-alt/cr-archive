 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * VMDeathEvent.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that the interface                           <BR>
 42  * <code>com.sun.jdi.VMDeathEvent</code>                        <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The case to test includes receiving the VMDeathEvent         <BR>
 46  * as a response to VirtualMachine.exit(int);                   <BR>
 47  * <BR>
 48  * The test works as follows.                                   <BR>
 49  * <BR>
 50  * Upon launching debuggee's VM which will be suspended,                <BR>
 51  * a debugger waits for the VMStartEvent within a predefined            <BR>
 52  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 53  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 54  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 55  * and waits for the event within the predefined time interval.         <BR>
 56  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 57  * Upon getting the ClassPrepareEvent,                                  <BR>
 58  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 59  * within debuggee's special methodForCommunication().                  <BR>
 60  * <BR>
 61  * Then to test the interface,                                          <BR>
 62  * the debugger invokes the method VirtualMachine.exit() and checks up  <BR>
 63  * on the event received in the response to.                            <BR>
 64  * <BR>
 65  */

139     private int runThis (String argv[], PrintStream out) {
140 
141         argsHandler     = new ArgumentHandler(argv);
142         logHandler      = new Log(out, argsHandler);
143         Binder binder   = new Binder(argsHandler, logHandler);
144 
145         waitTime        = argsHandler.getWaitTime() * 60000;
146 
147         try {
148             log2("launching a debuggee :");
149             log2("       " + debuggeeName);
150             if (argsHandler.verbose()) {
151                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
152             } else {
153                 debuggee = binder.bindToDebugee(debuggeeName);
154             }
155             if (debuggee == null) {
156                 log3("ERROR: no debuggee launched");
157                 return FAILED;
158             }
159             log2("debuggee launched");
160         } catch ( Exception e ) {
161             log3("ERROR: Exception : " + e);
162             log2("       test cancelled");
163             return FAILED;
164         }
165 
166         debuggee.redirectOutput(logHandler);
167 
168         vm = debuggee.VM();
169 
170         eventQueue = vm.eventQueue();
171         if (eventQueue == null) {
172             log3("ERROR: eventQueue == null : TEST ABORTED");
173             vm.exit(PASS_BASE);
174             return FAILED;
175         }
176 
177         log2("invocation of the method runTest()");
178         switch (runTest()) {
179 
180             case 0 :  log2("test phase has finished normally");
181                       log2("   waiting for the debuggee to finish ...");
182                       debuggee.waitFor();
183 
184                       log2("......getting the debuggee's exit status");
185                       int status = debuggee.getStatus();
186                       if (status != PASS_BASE) {
187                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
188                               status + " != PASS_BASE");
189                           testExitCode = FAILED;
190                       } else {
191                           log2("......debuggee returned expected exit status: " +
192                               status + " == PASS_BASE");
193                       }
194                       break;
195 
196             default : log3("ERROR: runTest() returned unexpected value");
197 
198             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
199                       log2("......forcing: vm.exit();");
200                       testExitCode = FAILED;
201                       try {
202                           vm.exit(PASS_BASE);
203                       } catch ( Exception e ) {
204                           log3("ERROR: Exception : " + e);
205                       }
206                       break;
207 
208             case 2 :  log3("test cancelled due to VMDisconnectedException");
209                       log2("......trying: vm.process().destroy();");
210                       testExitCode = FAILED;
211                       try {
212                           Process vmProcess = vm.process();
213                           if (vmProcess != null) {
214                               vmProcess.destroy();
215                           }
216                       } catch ( Exception e ) {
217                           log3("ERROR: Exception : " + e);
218                       }
219                       break;
220             }
221 
222         return testExitCode;
223     }

226    /*
227     * Return value: 0 - normal end of the test
228     *               1 - ubnormal end of the test
229     *               2 - VMDisconnectedException while test phase
230     */
231 
232     private int runTest() {
233 
234         try {
235             testRun();
236 
237             log2("waiting for VMDeathEvent");
238             getEventSet();
239             Event event = eventIterator.nextEvent();
240             if ( !(event instanceof VMDeathEvent) ) {
241                 testExitCode = FAILED;
242                 log3("ERROR: last event is not the VMDeathEvent");
243 
244                 if (event instanceof VMDisconnectEvent)
245                    log3("      event is the VMDisconnectEvent     ????");
246                 else
247                    log3("      event is NOT the VMDisconnectEvent ????!!!!!!!");
248                 return 2;
249 
250             }
251             return 0;
252         } catch ( VMDisconnectedException e ) {
253             log3("ERROR: VMDisconnectedException : " + e);
254             return 2;
255         } catch ( Exception e ) {
256             log3("ERROR: Exception : " + e);
257             return 1;
258         }
259 
260     }

262     private void testRun()
263                  throws JDITestRuntimeException, Exception {
264 
265         eventRManager = vm.eventRequestManager();
266 
267         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
268         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
269         cpRequest.addClassFilter(debuggeeName);
270 
271         cpRequest.enable();
272         vm.resume();
273         getEventSet();
274         cpRequest.disable();
275 
276         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
277         debuggeeClass = event.referenceType();
278 
279         if (!debuggeeClass.name().equals(debuggeeName))
280            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
281 
282         log2("      received: ClassPrepareEvent for debuggeeClass");
283 
284         String bPointMethod = "methodForCommunication";
285         String lineForComm  = "lineForComm";
286         BreakpointRequest bpRequest;
287 
288         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
289                                       debuggeeClass,
290                                       bPointMethod, lineForComm, "zero");
291         bpRequest.enable();
292 
293     //------------------------------------------------------  testing section
294 
295         log1("     TESTING BEGINS");
296 
297         for (int i = 0; ; i++) {
298 
299             vm.resume();
300             breakpointForCommunication();
301 
302             int instruction = ((IntegerValue)
303                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
304 
305             if (instruction == 0) {
306                 vm.resume();
307                 break;
308             }
309 
310             log1(":::::: case: # " + i);
311 
312             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
313 
314             //vm.eventRequestManager().createVMDeathRequest().enable();
315 
316             log2("......vm.exit(PASS_BASE);  :: VMDeathEvent is excpected");
317 
318             //vm.exit(PASS_BASE);
319             break;
320 
321             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
322         }
323         log1("    TESTING ENDS");
324         return;
325     }

328    /*
329     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
330     *                                             String, String, String)
331     *
332     * It sets up a breakpoint at given line number within a given method in a given class
333     * for a given thread.
334     *
335     * Return value: BreakpointRequest object  in case of success
336     *
337     * JDITestRuntimeException   in case of an Exception thrown within the method
338     */
339 
340     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
341                                                   ReferenceType testedClass,
342                                                   String methodName,
343                                                   String bpLine,
344                                                   String property)
345             throws JDITestRuntimeException {
346 
347         log2("......setting up a breakpoint:");
348         log2("       thread: " + thread + "; class: " + testedClass +
349                         "; method: " + methodName + "; line: " + bpLine);
350 
351         List              alllineLocations = null;
352         Location          lineLocation     = null;
353         BreakpointRequest breakpRequest    = null;
354 
355         try {
356             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
357 
358             alllineLocations = method.allLineLocations();
359 
360             int n =
361                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
362             if (n > alllineLocations.size()) {
363                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
364             } else {
365                 lineLocation = (Location) alllineLocations.get(n);
366                 try {
367                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
368                     breakpRequest.putProperty("number", property);
369                     breakpRequest.addThreadFilter(thread);
370                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
371                 } catch ( Exception e1 ) {
372                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
373                     breakpRequest    = null;
374                 }
375             }
376         } catch ( Exception e2 ) {
377             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
378             breakpRequest    = null;
379         }
380 
381         if (breakpRequest == null) {
382             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
383             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
384         }
385 
386         log2("      a breakpoint has been set up");
387         return breakpRequest;
388     }
