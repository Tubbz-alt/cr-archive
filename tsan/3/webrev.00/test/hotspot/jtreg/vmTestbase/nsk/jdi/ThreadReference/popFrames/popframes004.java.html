 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Finally blocks are not executed.                          <BR>
 47  * <BR>
 48  * The test has two phases and works as follows.                <BR>
 49  * <BR>
 50  * In first phase,                                                      <BR>
 51  * upon launching debuggee's VM which will be suspended,                <BR>
 52  * a debugger waits for the VMStartEvent within a predefined            <BR>
 53  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 54  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 55  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 56  * and waits for the event within the predefined time interval.         <BR>
 57  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 58  * Upon getting the ClassPrepareEvent,                                  <BR>
 59  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 60  * within debuggee's special methodForCommunication().                  <BR>
 61  * <BR>
 62  * In second phase to check the assetion,                               <BR>
 63  * the debugger and the debuggee perform the following.                 <BR>
 64  * - The debugger resumes the debuggee and waits for the BreakpointEvent<BR>
 65  * in the methodForCommunication().                                     <BR>
 66  * - The debuggee creates second thread, the thread2, which will        <BR>
 67  * call a tested poppedMethod() two times from its run() method.        <BR>
 68  * The poppedMethod will call the special breakpointMethod,             <BR>
 69  * first time from try clause, second time from catch clause.           <BR>
 70  * The finally clause contains assignment of 1 to the variable testVar, <BR>
 71  * whose initial value is 0; hence, if the finally clause is executed,  <BR>
 72  * the value of the testVar is changed as well.                         <BR>
 73  * After creating the thread2, the debuggee invokes                     <BR>
 74  * the methodForCommunication to be suspended and                       <BR>
 75  * to inform the debugger with the event.                               <BR>
 76  * - Upon getting the BreakpointEvent in the methodForCommunication,    <BR>
 77  * the debugger sets up a breakpoint in the breakpointMethod,           <BR>
 78  * resumes the debuggee, waits for BreakpointEvents in the              <BR>
 79  * breakpointMethod and, and upon getting them, invokes popFrames() and <BR>
 80  * checks up on the value of the testVar.                               <BR>
 81  * <BR>
 82  */
 83

106     public static int run (String argv[], PrintStream out) {
107 
108         int exitCode = new popframes004().runThis(argv, out);

112         }
113         return testExitCode;

159     private int runThis (String argv[], PrintStream out) {
160 
161         argsHandler     = new ArgumentHandler(argv);
162         logHandler      = new Log(out, argsHandler);
163         Binder binder   = new Binder(argsHandler, logHandler);
164 
165         waitTime        = argsHandler.getWaitTime() * 60000;
166 
167         try {
168             log2("launching a debuggee :");
169             log2("       " + debuggeeName);
170             if (argsHandler.verbose()) {
171                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
172             } else {
173                 debuggee = binder.bindToDebugee(debuggeeName);
174             }
175             if (debuggee == null) {
176                 log3("ERROR: no debuggee launched");
177                 return FAILED;
178             }
179             log2("debuggee launched");
180         } catch ( Exception e ) {
181             log3("ERROR: Exception : " + e);
182             log2("       test cancelled");
183             return FAILED;
184         }
185 
186         debuggee.redirectOutput(logHandler);
187 
188         vm = debuggee.VM();
189 
190         eventQueue = vm.eventQueue();
191         if (eventQueue == null) {
192             log3("ERROR: eventQueue == null : TEST ABORTED");
193             vm.exit(PASS_BASE);
194             return FAILED;
195         }
196 
197         log2("invocation of the method runTest()");
198         switch (runTest()) {
199 
200             case 0 :  log2("test phase has finished normally");
201                       log2("   waiting for the debuggee to finish ...");
202                       debuggee.waitFor();
203 
204                       log2("......getting the debuggee's exit status");
205                       int status = debuggee.getStatus();
206                       if (status != PASS_BASE) {
207                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
208                               status + " != PASS_BASE");
209                           testExitCode = FAILED;
210                       } else {
211                           log2("......debuggee returned expected exit status: " +
212                               status + " == PASS_BASE");
213                       }
214                       break;
215 
216             default : log3("ERROR: runTest() returned unexpected value");
217 
218             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
219                       log2("......forcing: vm.exit();");
220                       testExitCode = FAILED;
221                       try {
222                           vm.exit(PASS_BASE);
223                       } catch ( Exception e ) {
224                           log3("ERROR: Exception : e");
225                       }
226                       break;
227 
228             case 2 :  log3("test cancelled due to VMDisconnectedException");
229                       log2("......trying: vm.process().destroy();");
230                       testExitCode = FAILED;
231                       try {
232                           Process vmProcess = vm.process();
233                           if (vmProcess != null) {
234                               vmProcess.destroy();
235                           }
236                       } catch ( Exception e ) {
237                           log3("ERROR: Exception : e");
238                       }
239                       break;
240             }
241 
242         return testExitCode;
243     }

246    /*
247     * Return value: 0 - normal end of the test
248     *               1 - ubnormal end of the test
249     *               2 - VMDisconnectedException while test phase
250     */

252     private int runTest() {
253 
254         try {
255             testRun();
256 
257             log2("waiting for VMDeathEvent");
258             getEventSet();
259             if (eventIterator.nextEvent() instanceof VMDeathEvent)
260                 return 0;
261 
262             log3("ERROR: last event is not the VMDeathEvent");
263             return 1;
264         } catch ( VMDisconnectedException e ) {
265             log3("ERROR: VMDisconnectedException : " + e);
266             return 2;
267         } catch ( Exception e ) {
268             log3("ERROR: Exception : " + e);
269             return 1;
270         }
271 
272     }

274     private void testRun()
275                  throws JDITestRuntimeException, Exception {
276 
277         eventRManager = vm.eventRequestManager();
278 
279         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
280         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
281         cpRequest.addClassFilter(debuggeeName);
282 
283         cpRequest.enable();
284         vm.resume();
285         getEventSet();
286         cpRequest.disable();
287 
288         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
289         debuggeeClass = event.referenceType();
290 
291         if (!debuggeeClass.name().equals(debuggeeName))
292            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
293 
294         log2("      received: ClassPrepareEvent for debuggeeClass");
295 
296         if ( !vm.canPopFrames() ) {
297             log2("......vm.canPopFrames() == false : test is cancelled");
298             vm.resume();
299             return;
300         }
301 
302         String bPointMethod = "methodForCommunication";
303         String lineForComm  = "lineForComm";
304 
305         log2("......setting BreakpointRequest (bpRequest) in main thread");
306         bpRequest = settingBreakpoint(debuggee.threadByNameOrThrow("main"),
307                                           debuggeeClass,
308                                           bPointMethod, lineForComm, "zero");
309         log2("bpRequest.enable();");
310         bpRequest.enable();
311 
312     //------------------------------------------------------  testing section
313 
314         log1("     TESTING BEGINS");
315 
316         label0:
317         {
318             vm.resume();
319             breakpointForCommunication();
320 
321             String thread2Name         = "thread2";
322             ThreadReference thread2Ref = debuggee.threadByNameOrThrow(thread2Name);
323 
324             StackFrame stackFrame = null;
325             int var;
326 
327             String breakpointMethod = "breakpointMethod";
328             String breakpointLine    = "breakpointLine";
329 
330             String testVar = "testVar";
331 
332             log2("......setting breakpoint in breakpointMethod");
333             breakpointRequest2 = settingBreakpoint(thread2Ref, debuggeeClass,
334                                           breakpointMethod, breakpointLine, "one");
335             log2("......breakpointRequest2.enable();");
336             breakpointRequest2.enable();
337 
338             {  // to get mainThread suspended; otherwise its end results in
339                // no suspention for breakpointRequest2 (bug or not?), end of test,
340                // and VMDisconnectedException
341             thread2Ref.suspend();
342             log2("......eventSet.resume();");
343             eventSet.resume();
344             breakpointForCommunication();
345             log2("bpRequest.disable();");
346             bpRequest.disable();
347             thread2Ref.resume();
348             }
349 
350             log2("......breakpointInMethod(breakpointRequest2);");
351             breakpointInMethod("one");
352 
353             log2(" ");
354             log2("......TEST CASE: try clause");
355             log2(" ");
356             log2("......stackFrame = thread2Ref.frame(1);");
357             stackFrame = thread2Ref.frame(1);
358 
359             log2("......thread2Ref.popFrames(stackFrame);");
360             try {
361                 thread2Ref.popFrames(stackFrame);
362             } catch ( IncompatibleThreadStateException e ) {
363                 log3("ERROR: IncompatibleThreadStateException");
364                 testExitCode = FAILED;
365                 break label0;
366             }
367             var = ((IntegerValue) debuggeeClass.getValue(
368                          debuggeeClass.fieldByName(testVar))).value();
369             if  (var != 0) {
370                 log3("ERROR:  testVar != 0");
371                 testExitCode = FAILED;
372             }
373 
374             thread2Ref.resume();
375             log2("......breakpointInMethod(breakpointRequest2);");
376             breakpointInMethod("one");
377 
378             thread2Ref.resume();
379             log2("......breakpointInMethod(breakpointRequest2);");
380             breakpointInMethod("one");
381 
382             log2(" ");
383             log2("......TEST CASE: catch clause");
384             log2(" ");
385 
386             log2("......StackFrame stackFrame = thread2Ref.frame(1);");
387             stackFrame = thread2Ref.frame(1);
388 
389             log2("......thread2Ref.popFrames(stackFrame);");
390             try {
391                 thread2Ref.popFrames(stackFrame);
392             } catch ( IncompatibleThreadStateException e ) {
393                 log3("ERROR: IncompatibleThreadStateException");
394                 testExitCode = FAILED;
395                 break label0;
396             }
397             var = ((IntegerValue) debuggeeClass.getValue(
398                           debuggeeClass.fieldByName(testVar))).value();
399             if  (var != 0) {
400                 log3("ERROR:  testVar != 0");
401                 testExitCode = FAILED;
402             }
403 
404             log2("......breakpointRequest2.disable();");
405             breakpointRequest2.disable();
406 
407             vm.resume();
408 
409             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
410         }
411         log1("    TESTING ENDS");
412         return;
413     }

415    /*
416     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
417     *                                             String, String, String)
418     *
419     * It sets up a breakpoint at given line number within a given method in a given class
420     * for a given thread.
421     *
422     * Return value: BreakpointRequest object  in case of success
423     *
424     * JDITestRuntimeException   in case of an Exception thrown within the method
425     */
426 
427     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
428                                                   ReferenceType testedClass,
429                                                   String methodName,
430                                                   String bpLine,
431                                                   String property)
432             throws JDITestRuntimeException {
433 
434         log2("......setting up a breakpoint:");
435         log2("       thread: " + thread + "; class: " + testedClass +
436                         "; method: " + methodName + "; line: " + bpLine);
437 
438         List              alllineLocations = null;
439         Location          lineLocation     = null;
440         BreakpointRequest breakpRequest    = null;
441 
442         try {
443             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
444 
445             alllineLocations = method.allLineLocations();
446 
447             int n =
448                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
449             if (n > alllineLocations.size()) {
450                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
451             } else {
452                 lineLocation = (Location) alllineLocations.get(n);
453                 try {
454                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
455                     breakpRequest.putProperty("number", property);
456                     breakpRequest.addThreadFilter(thread);
457                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
458                 } catch ( Exception e1 ) {
459                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
460                     breakpRequest    = null;
461                 }
462             }
463         } catch ( Exception e2 ) {
464             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
465             breakpRequest    = null;
466         }
467 
468         if (breakpRequest == null) {
469             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
470             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
471         }
472 
473         log2("      a breakpoint has been set up");
474         return breakpRequest;
475     }

479     private void getEventSet()
480                  throws JDITestRuntimeException {
481         try {
482 //            log2("       eventSet = eventQueue.remove(waitTime);");
483             eventSet = eventQueue.remove(waitTime);
484             if (eventSet == null) {
485                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
486             }
487 //            log2("       eventIterator = eventSet.eventIterator;");
488             eventIterator = eventSet.eventIterator();
489         } catch ( Exception e ) {
490             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
491         }
492     }

495     private void breakpointForCommunication()
496                  throws JDITestRuntimeException {
497 
498         log2("breakpointForCommunication");
499         getEventSet();
500 
501         if (eventIterator.nextEvent() instanceof BreakpointEvent)
502             return;
503 
504         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
505     }

509     private void breakpointInMethod(String number)
510                  throws JDITestRuntimeException {
511 
512         Event event;
513 
514         getEventSet();
515         event = eventIterator.nextEvent();
516 
517         if ( !(event instanceof BreakpointEvent) )
518             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
519 
520         log2("---->: request().getProperty == " +
521              event.request().getProperty("number"));
522 
523         if ( event.request().getProperty("number").equals(number) )
524             return;
525 
526         throw new JDITestRuntimeException("** UNEXPECTED breakpoint **");
527     }
