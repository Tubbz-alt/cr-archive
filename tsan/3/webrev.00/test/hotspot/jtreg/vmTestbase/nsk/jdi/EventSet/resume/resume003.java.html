 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.resume()</code>                   <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Test cases include all three possible suspensions, NONE,     <BR>
 46  * EVENT_THREAD, and ALL, for ModificationWatchpointEvent sets. <BR>
 47  * <BR>
 48  * To check up on the method, a debugger,                       <BR>
 49  * upon getting new set for the EventSet,                       <BR>
 50  * suspends VM with the method VirtualMachine.suspend(),        <BR>
 51  * gets the List of debuggee's threads calling VM.allThreads(), <BR>
 52  * invokes the method EventSet.resume(), and                    <BR>
 53  * gets another List of debuggee's threads.                     <BR>
 54  * The debugger then compares values of                         <BR>
 55  * each thread's suspendCount from first and second Lists.      <BR>
 56  * <BR>
 57  * The test has three phases and works as follows.              <BR>
 58  * <BR>
 59  * In first phase,                                                      <BR>
 60  * upon launching debuggee's VM which will be suspended,                <BR>
 61  * a debugger waits for the VMStartEvent within a predefined            <BR>
 62  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 63  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 64  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 65  * and waits for the event within the predefined time interval.         <BR>
 66  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 67  * Upon getting the ClassPrepareEvent,                                  <BR>
 68  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 69  * within debuggee's special methodForCommunication().                  <BR>
 70  * <BR>
 71  * In second phase to check the above,                                  <BR>
 72  * the debugger and the debuggee perform the following loop.            <BR>
 73  * - The debugger sets up a ModificationWatchpointRequest, resumes      <BR>
 74  *   the debuggee, and waits for the ModificationWatchpointEvent.       <BR>
 75  * - The debuggee invokes the special method which makes access to      <BR>
 76  *   a predefined variable to be resulting in the event.                <BR>
 77  * - Upon getting new event, the debugger                               <BR>
 78  *   performs the check corresponding to the event.                     <BR>
 79  * <BR>
 80  * Note. To inform each other of needed actions, the debugger and       <BR>
 81  *       and the debuggee use debuggee's variable "instruction".        <BR>
 82  * In third phase, at the end,                                          <BR>
 83  * the debuggee changes the value of the "instruction"                  <BR>
 84  * to inform the debugger of checks finished, and both end.             <BR>
 85  * <BR>
 86  */
 87

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : e");
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : e");
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         if ( !vm.canWatchFieldModification() ) {
282             log2("......vm.canWatchFieldModification == false :: test cancelled");
283             vm.exit(PASS_BASE);
284             return;
285         }
286 
287 
288         eventRManager = vm.eventRequestManager();
289 
290         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
291         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
292         cpRequest.addClassFilter(debuggeeName);
293 
294         cpRequest.enable();
295         vm.resume();
296         getEventSet();
297         cpRequest.disable();
298 
299         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
300         debuggeeClass = event.referenceType();
301 
302         if (!debuggeeClass.name().equals(debuggeeName))
303            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
304 
305         log2("      received: ClassPrepareEvent for debuggeeClass");
306 
307         String bPointMethod = "methodForCommunication";
308         String lineForComm  = "lineForComm";
309         BreakpointRequest bpRequest;
310         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
311         bpRequest = settingBreakpoint(mainThread,
312                                       debuggeeClass,
313                                       bPointMethod, lineForComm, "zero");
314         bpRequest.enable();
315 
316         vm.resume();
317 
318     //------------------------------------------------------  testing section
319 
320         log1("     TESTING BEGINS");
321 
322         EventRequest eventRequest1 = null;
323         EventRequest eventRequest2 = null;
324         EventRequest eventRequest3 = null;
325 
326         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
327         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
328         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
329 
330         String fieldName = "var1";
331 
332         ReferenceType testClassReference = null;
333 
334 
335         for (int i = 0; ; i++) {
336 
337             breakpointForCommunication();
338 
339             int instruction = ((IntegerValue)
340                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
341 
342             if (instruction == 0) {
343                 vm.resume();
344                 break;
345             }
346 
347             log1(":::::: case: # " + i);
348 
349             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
350 
351             switch (i) {
352 
353               case 0:
354                       testClassReference =
355                            (ReferenceType) vm.classesByName(testedClassName).get(0);
356 
357                       eventRequest1 = settingModificationWatchpointRequest (mainThread,
358                                              testClassReference, fieldName,
359                                              SUSPEND_NONE, "ModificationWatchpointRequest1");
360                       eventRequest1.enable();
361                       break;
362 
363               case 1:
364                       eventRequest2 = settingModificationWatchpointRequest (mainThread,
365                                              testClassReference, fieldName,
366                                              SUSPEND_THREAD, "ModificationWatchpointRequest2");
367                       eventRequest2.enable();
368                       break;
369 
370               case 2:
371                       eventRequest3 = settingModificationWatchpointRequest (mainThread,
372                                              testClassReference, fieldName,
373                                              SUSPEND_ALL, "ModificationWatchpointRequest3");
374                       eventRequest3.enable();
375                       break;
376 
377 
378               default:
379                       throw new JDITestRuntimeException("** default case 2 **");
380             }
381 
382             log2("......waiting for new ModificationWatchpointEvent : " + i);
383             mainThread.resume();
384             getEventSet();
385 
386             Event newEvent = eventIterator.nextEvent();
387             if ( !(newEvent instanceof ModificationWatchpointEvent)) {
388                  log3("ERROR: new event is not ModificationWatchpointEvent");
389                  testExitCode = FAILED;
390             } else {
391 
392                 String property = (String) newEvent.request().getProperty("number");
393                 log2("       got new ModificationWatchpointEvent with propety 'number' == " + property);
394 
395                 log2("......checking up on EventSet.resume()");
396                 log2("......--> vm.suspend();");
397                 vm.suspend();
398 
399                 log2("        getting : Map<String, Integer> suspendsCounts1");
400 
401                 Map<String, Integer> suspendsCounts1 = new HashMap<String, Integer>();
402                 for (ThreadReference threadReference : vm.allThreads()) {
403                     suspendsCounts1.put(threadReference.name(), threadReference.suspendCount());
404                 }
405 
406                 log2("        eventSet.resume;");
407                 eventSet.resume();
408 
409                 log2("        getting : Map<String, Integer> suspendsCounts2");
410                 Map<String, Integer> suspendsCounts2 = new HashMap<String, Integer>();
411                 for (ThreadReference threadReference : vm.allThreads()) {
412                     suspendsCounts2.put(threadReference.name(), threadReference.suspendCount());
413                 }
414 
415                 log2(suspendsCounts1.toString());
416                 log2(suspendsCounts2.toString());
417 
418                 log2("        getting : int policy = eventSet.suspendPolicy();");
419                 int policy = eventSet.suspendPolicy();
420 
421                 switch (policy) {
422 
423                   case SUSPEND_NONE   :
424                        log2("        case SUSPEND_NONE");
425                        for (String threadName : suspendsCounts1.keySet()) {
426                            log2("        checking " + threadName);
427                            if (!suspendsCounts2.containsKey(threadName)) {
428                                log3("ERROR: couldn't get ThreadReference for " + threadName);
429                                testExitCode = FAILED;
430                                break;
431                            }
432                            int count1 = suspendsCounts1.get(threadName);
433                            int count2 = suspendsCounts2.get(threadName);
434                            if (count1 != count2) {
435                                log3("ERROR: suspendCounts don't match for : " + threadName);
436                                log3("before resuming : " + count1);
437                                log3("after  resuming : " + count2);
438                                testExitCode = FAILED;
439                                break;
440                            }
441                        }
442                        eventRequest1.disable();
443                        break;
444 
445                   case SUSPEND_THREAD :
446                        log2("        case SUSPEND_THREAD");
447                        for (String threadName : suspendsCounts1.keySet()) {
448                            log2("checking " + threadName);
449                            if (!suspendsCounts2.containsKey(threadName)) {
450                                log3("ERROR: couldn't get ThreadReference for " + threadName);
451                                testExitCode = FAILED;
452                                break;
453                            }
454                            int count1 = suspendsCounts1.get(threadName);
455                            int count2 = suspendsCounts2.get(threadName);
456                            String eventThreadName = ((ModificationWatchpointEvent)newEvent).thread().name();
457                            int expectedValue = count2 + (eventThreadName.equals(threadName) ? 1 : 0);
458                            if (count1 != expectedValue) {
459                                log3("ERROR: suspendCounts don't match for : " + threadName);
460                                log3("before resuming : " + count1);
461                                log3("after  resuming : " + count2);
462                                testExitCode = FAILED;
463                                break;
464                            }
465                        }
466                        eventRequest2.disable();
467                        break;
468 
469                   case SUSPEND_ALL    :
470 
471                         log2("        case SUSPEND_ALL");
472                         for (String threadName : suspendsCounts1.keySet()) {
473                             log2("checking " + threadName);
474                             if (!event.request().equals(eventRequest3))
475                                 break;
476                             if (!suspendsCounts2.containsKey(threadName)) {
477                                 log3("ERROR: couldn't get ThreadReference for " + threadName);
478                                 testExitCode = FAILED;
479                                 break;
480                             }
481                             int count1 = suspendsCounts1.get(threadName);
482                             int count2 = suspendsCounts2.get(threadName);
483                             if (count1 != count2 + 1) {
484                                 log3("ERROR: suspendCounts don't match for : " + threadName);
485                                 log3("before resuming : " + count1);
486                                 log3("after  resuming : " + count2);
487                                 testExitCode = FAILED;
488                                 break;
489                             }
490                         }
491                         eventRequest3.disable();
492                         break;
493 
494                   default: throw new JDITestRuntimeException("** default case 1 **");
495                 }
496             }
497 
498             log2("......--> vm.resume()");
499             vm.resume();
500             informDebuggeeTestCase(i);
501 
502             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
503         }
504         log1("    TESTING ENDS");
505         return;
506     }

508    /*
509     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
510     *                                             String, String, String)
511     *
512     * It sets up a breakpoint at given line number within a given method in a given class
513     * for a given thread.
514     *
515     * Return value: BreakpointRequest object  in case of success
516     *
517     * JDITestRuntimeException   in case of an Exception thrown within the method
518     */
519 
520     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
521                                                   ReferenceType testedClass,
522                                                   String methodName,
523                                                   String bpLine,
524                                                   String property)
525             throws JDITestRuntimeException {
526 
527         log2("......setting up a breakpoint:");
528         log2("       thread: " + thread + "; class: " + testedClass +
529                         "; method: " + methodName + "; line: " + bpLine);
530 
531         List              alllineLocations = null;
532         Location          lineLocation     = null;
533         BreakpointRequest breakpRequest    = null;
534 
535         try {
536             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
537 
538             alllineLocations = method.allLineLocations();
539 
540             int n =
541                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
542             if (n > alllineLocations.size()) {
543                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
544             } else {
545                 lineLocation = (Location) alllineLocations.get(n);
546                 try {
547                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
548                     breakpRequest.putProperty("number", property);
549                     breakpRequest.addThreadFilter(thread);
550                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
551                 } catch ( Exception e1 ) {
552                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
553                     breakpRequest    = null;
554                 }
555             }
556         } catch ( Exception e2 ) {
557             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
558             breakpRequest    = null;
559         }
560 
561         if (breakpRequest == null) {
562             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
563             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
564         }
565 
566         log2("      a breakpoint has been set up");
567         return breakpRequest;
568     }

571     private void getEventSet()
572                  throws JDITestRuntimeException {
573         try {
574 //            log2("       eventSet = eventQueue.remove(waitTime);");
575             eventSet = eventQueue.remove(waitTime);
576             if (eventSet == null) {
577                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
578             }
579 //            log2("       eventIterator = eventSet.eventIterator;");
580             eventIterator = eventSet.eventIterator();
581         } catch ( Exception e ) {
582             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
583         }
584     }

587     private void breakpointForCommunication()
588                  throws JDITestRuntimeException {
589 
590         log2("breakpointForCommunication");
591         getEventSet();
592 
593         if (eventIterator.nextEvent() instanceof BreakpointEvent)
594             return;
595 
596         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
597     }

601     private ModificationWatchpointRequest settingModificationWatchpointRequest (
602                                                   ThreadReference thread,
603                                                   ReferenceType   testedClass,
604                                                   String          fieldName,
605                                                   int             suspendPolicy,
606                                                   String          property        )
607             throws JDITestRuntimeException {
608         try {
609             log2("......setting up ModificationWatchpointRequest:");
610             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
611             Field field = testedClass.fieldByName(fieldName);
612 
613             ModificationWatchpointRequest
614             awr = eventRManager.createModificationWatchpointRequest(field);
615             awr.putProperty("number", property);
616             awr.addThreadFilter(thread);
617             awr.setSuspendPolicy(suspendPolicy);
618 
619             log2("      ModificationWatchpointRequest has been set up");
620             return awr;
621         } catch ( Exception e ) {
622             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
623             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
624             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
625         }
626     }
627     /**
628      * Inform debuggee which thread test the debugger has completed.
629      * Used for synchronization, so the debuggee does not move too quickly.
630      * @param testCase index of just completed test
631      */
632     void informDebuggeeTestCase(int testCase) {
633         try {
634             ((ClassType)debuggeeClass)
635                 .setValue(debuggeeClass.fieldByName("testCase"),
636                           vm.mirrorOf(testCase));
637         } catch (InvalidTypeException ite) {
638             throw new Failure("** FAILURE setting testCase  **");
639         } catch (ClassNotLoadedException cnle) {
640             throw new Failure("** FAILURE notifying debuggee  **");
641         } catch (VMDisconnectedException e) {
642             throw new Failure("** FAILURE debuggee connection **");
643         }
644     }
