 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequestManager.                                         <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequestManager.stepRequests()</code> <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:                       <BR>
 46  * - The list is unmodifiable.                                          <BR>
 47  * - The list of the enabled and disabled step requests.                <BR>
 48  * - This list is changes as requests are added and deleted.            <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.                      <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee prepares the check case and invokes                   <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent,                                  <BR>
 71  *   the debugger checking up on assertions.                            <BR>
 72  * <BR>
 73  * In third phase, at the end of the test, the debuggee changes         <BR>
 74  * the value of the "instruction" which the debugger and debuggee       <BR>
 75  * use to inform each other of needed actions,  and both end.           <BR>
 76  * <BR>
 77  */

157     private int runThis (String argv[], PrintStream out) {
158 
159         argsHandler     = new ArgumentHandler(argv);
160         logHandler      = new Log(out, argsHandler);
161         Binder binder   = new Binder(argsHandler, logHandler);
162 
163         waitTime        = argsHandler.getWaitTime() * 60000;
164 
165         try {
166             log2("launching a debuggee :");
167             log2("       " + debuggeeName);
168             if (argsHandler.verbose()) {
169                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
170             } else {
171                 debuggee = binder.bindToDebugee(debuggeeName);
172             }
173             if (debuggee == null) {
174                 log3("ERROR: no debuggee launched");
175                 return FAILED;
176             }
177             log2("debuggee launched");
178         } catch ( Exception e ) {
179             log3("ERROR: Exception : " + e);
180             log2("       test cancelled");
181             return FAILED;
182         }
183 
184         debuggee.redirectOutput(logHandler);
185 
186         vm = debuggee.VM();
187 
188         eventQueue = vm.eventQueue();
189         if (eventQueue == null) {
190             log3("ERROR: eventQueue == null : TEST ABORTED");
191             vm.exit(PASS_BASE);
192             return FAILED;
193         }
194 
195         log2("invocation of the method runTest()");
196         switch (runTest()) {
197 
198             case 0 :  log2("test phase has finished normally");
199                       log2("   waiting for the debuggee to finish ...");
200                       debuggee.waitFor();
201 
202                       log2("......getting the debuggee's exit status");
203                       int status = debuggee.getStatus();
204                       if (status != PASS_BASE) {
205                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
206                               status + " != PASS_BASE");
207                           testExitCode = FAILED;
208                       } else {
209                           log2("......debuggee returned expected exit status: " +
210                               status + " == PASS_BASE");
211                       }
212                       break;
213 
214             default : log3("ERROR: runTest() returned unexpected value");
215 
216             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
217                       log2("......forcing: vm.exit();");
218                       testExitCode = FAILED;
219                       try {
220                           vm.exit(PASS_BASE);
221                       } catch ( Exception e ) {
222                           log3("ERROR: Exception : " + e);
223                       }
224                       break;
225 
226             case 2 :  log3("test cancelled due to VMDisconnectedException");
227                       log2("......trying: vm.process().destroy();");
228                       testExitCode = FAILED;
229                       try {
230                           Process vmProcess = vm.process();
231                           if (vmProcess != null) {
232                               vmProcess.destroy();
233                           }
234                       } catch ( Exception e ) {
235                           log3("ERROR: Exception : " + e);
236                       }
237                       break;
238             }
239 
240         return testExitCode;
241     }

246     *               1 - ubnormal end of the test
247     *               2 - VMDisconnectedException while test phase

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257             if (eventIterator.nextEvent() instanceof VMDeathEvent)
258                 return 0;
259 
260             log3("ERROR: last event is not the VMDeathEvent");
261             return 1;
262         } catch ( VMDisconnectedException e ) {
263             log3("ERROR: VMDisconnectedException : " + e);
264             return 2;
265         } catch ( Exception e ) {
266             log3("ERROR: Exception : " + e);
267             return 1;
268         }
269 
270     }

272     private void testRun()
273                  throws JDITestRuntimeException, Exception {
274 
275         eventRManager = vm.eventRequestManager();
276 
277         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
278         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
279         cpRequest.addClassFilter(debuggeeName);
280 
281         cpRequest.enable();
282         vm.resume();
283         getEventSet();
284         cpRequest.disable();
285 
286         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
287         debuggeeClass = event.referenceType();
288 
289         if (!debuggeeClass.name().equals(debuggeeName))
290            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
291 
292         log2("      received: ClassPrepareEvent for debuggeeClass");
293 
294         String bPointMethod = "methodForCommunication";
295         String lineForComm  = "lineForComm";
296 
297         ThreadReference   mainThread = threadByName("main");
298 
299         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
300                                              debuggeeClass,
301                                             bPointMethod, lineForComm, "zero");
302         bpRequest.enable();
303 
304     //------------------------------------------------------  testing section
305 
306         String fieldName = "testField";
307 
308         List         requests = null;
309         ListIterator li       = null;
310 
311         StepRequest request = null;
312         StepRequest stRequests[] = { null, null, null, null, null,
313                                      null, null, null, null, null };
314         int listSize;
315         int flag;
316 
317 
318         log1("     TESTING BEGINS");
319 
320         for (int i = 0; ; i++) {
321 
322             vm.resume();
323             breakpointForCommunication();
324 
325             int instruction = ((IntegerValue)
326                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
327 
328             if (instruction == 0) {
329                 vm.resume();
330                 break;
331             }
332 
333             log1(":::::: case: # " + i);
334 
335             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
336 
337             ArrayReference threads = (ArrayReference)
338                                      debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName));
339 
340 
341             log2("......creating StepRequests");
342             for (int i1 = 0; i1 < stRequests.length; i1++) {
343                 stRequests[i1] = eventRManager.createStepRequest( (ThreadReference) threads.getValue(i1),
344                                                                  StepRequest.STEP_MIN, StepRequest.STEP_OUT);
345                 stRequests[i1].putProperty("number", String.valueOf(i1));
346 
347                 log2("......checking up on returned List after creating new StepRequest");
348                 requests = eventRManager.stepRequests();
349                 listSize = requests.size();
350                 if ( listSize != (i1 + 1) ) {
351                     testExitCode = FAILED;
352                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
353                 }
354                 flag = 0;
355                 li = requests.listIterator();
356                 while (li.hasNext()) {
357                     request = (StepRequest) li.next();
358                     if ( !request.isEnabled() ) {
359                         flag++;
360                         if (flag > 1) {
361                             testExitCode = FAILED;
362                             log3("ERROR: # of disabled requests > 1 : " + flag);
363                         }
364                         if ( !request.getProperty("number").equals(String.valueOf(i1)) ) {
365                             testExitCode = FAILED;
366                             log3("ERROR: in the List, disabled is request expected to be enabled : # == " + i1);
367                         }
368                     } else {
369                         if ( request.getProperty("number").equals(String.valueOf(i1)) ) {
370                             testExitCode = FAILED;
371                             log3("ERROR: in the List, enabled is newly created disabled request : # == " + i1);
372                         }
373                     }
374                 }
375 
376                 log2("      enabling created StepRequest");
377                 stRequests[i1].enable();
378                 requests = eventRManager.stepRequests();
379                 listSize = requests.size();
380                 if ( listSize != (i1 + 1) ) {
381                     testExitCode = FAILED;
382                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " +  (i1 + 1));
383                 }
384 
385                 li = requests.listIterator();
386                 while (li.hasNext()) {
387                     request = (StepRequest) li.next();
388                     if ( !request.isEnabled() ) {
389                         testExitCode = FAILED;
390                         log3("ERROR: returned List contains disabled request : " + request);
391                     }
392                 }
393 
394                 log2("      removing item from the List; UnsupportedOperationException is expected");
395                 try {
396                     requests.remove(i1);
397                     testExitCode = FAILED;
398                     log3("ERROR: NO exception");
399                 } catch ( UnsupportedOperationException e ) {
400                     log2("        UnsupportedOperationException ");
401                 }
402             }
403 
404             log2("......deleting StepRequests");
405             for (int i2 = stRequests.length -1; i2 >= 0; i2--) {
406                 eventRManager.deleteEventRequest(stRequests[i2]);
407                 requests = eventRManager.stepRequests();
408                 listSize = requests.size();
409                 if ( listSize != i2 ) {
410                     testExitCode = FAILED;
411                     log3("ERROR: size of returned List is not equal to expected : " + listSize + " != " + i2);
412                 }
413                 log2("      removing item from the List; UnsupportedOperationException is expected");
414                 try {
415                     requests.remove(i2);
416                     testExitCode = FAILED;
417                     log3("ERROR: NO exception");
418                 } catch ( UnsupportedOperationException e ) {
419                     log2("        UnsupportedOperationException ");
420                 }
421             }
422 
423             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
424         }
425         log1("    TESTING ENDS");
426         return;
427     }

429     private ThreadReference threadByName(String name)
430                  throws JDITestRuntimeException {
431 
432         List         all = vm.allThreads();
433         ListIterator li  = all.listIterator();
434 
435         for (; li.hasNext(); ) {
436             ThreadReference thread = (ThreadReference) li.next();
437             if (thread.name().equals(name))
438                 return thread;
439         }
440         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
441     }

443    /*
444     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
445     *                                             String, String, String)
446     *
447     * It sets up a breakpoint at given line number within a given method in a given class
448     * for a given thread.
449     *
450     * Return value: BreakpointRequest object  in case of success
451     *
452     * JDITestRuntimeException   in case of an Exception thrown within the method
453     */
454 
455     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
456                                                   ReferenceType testedClass,
457                                                   String methodName,
458                                                   String bpLine,
459                                                   String property)
460             throws JDITestRuntimeException {
461 
462         log2("......setting up a breakpoint:");
463         log2("       thread: " + thread + "; class: " + testedClass +
464                         "; method: " + methodName + "; line: " + bpLine);
465 
466         List              alllineLocations = null;
467         Location          lineLocation     = null;
468         BreakpointRequest breakpRequest    = null;
469 
470         try {
471             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
472 
473             alllineLocations = method.allLineLocations();
474 
475             int n =
476                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
477             if (n > alllineLocations.size()) {
478                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
479             } else {
480                 lineLocation = (Location) alllineLocations.get(n);
481                 try {
482                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
483                     breakpRequest.putProperty("number", property);
484                     breakpRequest.addThreadFilter(thread);
485                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
486                 } catch ( Exception e1 ) {
487                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
488                     breakpRequest    = null;
489                 }
490             }
491         } catch ( Exception e2 ) {
492             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
493             breakpRequest    = null;
494         }
495 
496         if (breakpRequest == null) {
497             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
498             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
499         }
500 
501         log2("      a breakpoint has been set up");
502         return breakpRequest;
503     }

506     private void getEventSet()
507                  throws JDITestRuntimeException {
508         try {
509 //            log2("       eventSet = eventQueue.remove(waitTime);");
510             eventSet = eventQueue.remove(waitTime);
511             if (eventSet == null) {
512                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
513             }
514 //            log2("       eventIterator = eventSet.eventIterator;");
515             eventIterator = eventSet.eventIterator();
516         } catch ( Exception e ) {
517             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
518         }
519     }

523                  throws JDITestRuntimeException {

529             return;
530 
531         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
