 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * MethodEntryRequest.                                          <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.MethodEntryRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                                      <BR>
 53  * upon launching debuggee's VM which will be suspended,                <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates a MethodEntryRequest and enables it,                     <BR>
 72  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */

166     private int runThis (String argv[], PrintStream out) {
167 
168         argsHandler     = new ArgumentHandler(argv);
169         logHandler      = new Log(out, argsHandler);
170         Binder binder   = new Binder(argsHandler, logHandler);
171 
172         waitTime        = argsHandler.getWaitTime() * 60000;
173 
174         try {
175             log2("launching a debuggee :");
176             log2("       " + debuggeeName);
177             if (argsHandler.verbose()) {
178                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
179             } else {
180                 debuggee = binder.bindToDebugee(debuggeeName);
181             }
182             if (debuggee == null) {
183                 log3("ERROR: no debuggee launched");
184                 return FAILED;
185             }
186             log2("debuggee launched");
187         } catch ( Exception e ) {
188             log3("ERROR: Exception : " + e);
189             log2("       test cancelled");
190             return FAILED;
191         }
192 
193         debuggee.redirectOutput(logHandler);
194 
195         vm = debuggee.VM();
196 
197         eventQueue = vm.eventQueue();
198         if (eventQueue == null) {
199             log3("ERROR: eventQueue == null : TEST ABORTED");
200             vm.exit(PASS_BASE);
201             return FAILED;
202         }
203 
204         log2("invocation of the method runTest()");
205         switch (runTest()) {
206 
207             case 0 :  log2("test phase has finished normally");
208                       log2("   waiting for the debuggee to finish ...");
209                       debuggee.waitFor();
210 
211                       log2("......getting the debuggee's exit status");
212                       int status = debuggee.getStatus();
213                       if (status != PASS_BASE) {
214                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
215                               status + " != PASS_BASE");
216                           testExitCode = FAILED;
217                       } else {
218                           log2("......debuggee returned expected exit status: " +
219                               status + " == PASS_BASE");
220                       }
221                       break;
222 
223             default : log3("ERROR: runTest() returned unexpected value");
224 
225             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
226                       log2("......forcing: vm.exit();");
227                       testExitCode = FAILED;
228                       try {
229                           vm.exit(PASS_BASE);
230                       } catch ( Exception e ) {
231                           log3("ERROR: Exception : " + e);
232                       }
233                       break;
234 
235             case 2 :  log3("test cancelled due to VMDisconnectedException");
236                       log2("......trying: vm.process().destroy();");
237                       testExitCode = FAILED;
238                       try {
239                           Process vmProcess = vm.process();
240                           if (vmProcess != null) {
241                               vmProcess.destroy();
242                           }
243                       } catch ( Exception e ) {
244                           log3("ERROR: Exception : " + e);
245                       }
246                       break;
247             }
248 
249         return testExitCode;
250     }

254     * Return value: 0 - normal end of the test
255     *               1 - ubnormal end of the test
256     *               2 - VMDisconnectedException while test phase
257     */

259     private int runTest() {
260 
261         try {
262             testRun();
263 
264             log2("waiting for VMDeathEvent");
265             getEventSet();
266             if (eventIterator.nextEvent() instanceof VMDeathEvent)
267                 return 0;
268 
269             log3("ERROR: last event is not the VMDeathEvent");
270             return 1;
271         } catch ( VMDisconnectedException e ) {
272             log3("ERROR: VMDisconnectedException : " + e);
273             return 2;
274         } catch ( Exception e ) {
275             log3("ERROR: Exception : " + e);
276             return 1;
277         }
278

281     private void testRun()
282                  throws JDITestRuntimeException, Exception {
283 
284         if ( !vm.canUseInstanceFilters() ) {
285             log2("......vm.canUseInstanceFilters == false :: test cancelled");
286             vm.exit(PASS_BASE);
287             return;
288         }
289 
290 
291         eventRManager = vm.eventRequestManager();
292 
293         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
294         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
295         cpRequest.addClassFilter(debuggeeName);
296 
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307 
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         String bPointMethod = "methodForCommunication";
311         String lineForComm  = "lineForComm";
312 
313         ThreadReference   mainThread = threadByName("main");
314 
315         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
316                                              debuggeeClass,
317                                             bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320     //------------------------------------------------------  testing section
321 
322         log1("     TESTING BEGINS");
323 
324         EventRequest eventRequest1 = null;
325 
326         ThreadReference thread1 = null;
327         ThreadReference thread2 = null;
328 
329         String thread1Name = "thread1";
330         String thread2Name = "thread2";
331 
332         String property1 = "MethodEntryRequest1";
333 
334         ReferenceType testClassReference = null;
335 
336         String         arrayName = "objTC";
337         ObjectReference instance = null;
338 
339 
340         for (int i = 0; ; i++) {
341 
342             vm.resume();
343             breakpointForCommunication();
344 
345             int instruction = ((IntegerValue)
346                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
347 
348             if (instruction == 0) {
349                 vm.resume();
350                 break;
351             }
352 
353             log1(":::::: case: # " + i);
354 
355             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
356 
357             switch (i) {
358 
359               case 0:
360                       testClassReference =
361                            (ReferenceType) vm.classesByName(testedClassName).get(0);
362 
363                       thread1 = (ThreadReference) debuggeeClass.getValue(
364                                                   debuggeeClass.fieldByName(thread1Name));
365 
366                       eventRequest1 = setting2MethodEntryRequest (null,
367                                              testClassReference,
368                                              EventRequest.SUSPEND_NONE, property1);
369 
370                       instance = (ObjectReference)
371                                  ((ArrayReference) debuggeeClass.getValue(
372                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
373 
374                       ((MethodEntryRequest) eventRequest1).addInstanceFilter(instance);
375 
376                       eventRequest1.enable();
377 
378                      try {
379                          log2("......eventRequest1.addInstanceFilter(instance);");
380                          log2("        InvalidRequestStateException expected");
381                          ((MethodEntryRequest) eventRequest1).addInstanceFilter(instance);
382                          log3("ERROR: no InvalidRequestStateException");
383                          testExitCode = FAILED;
384                      } catch ( InvalidRequestStateException e ) {
385                          log2("        InvalidRequestStateException");
386                      }
387 
388                      log2("......eventRequest1.disable();");
389                      eventRequest1.disable();
390                      try {
391                          log2("......eventRequest1.addInstanceFilter(instance);");
392                          log2("        no InvalidRequestStateException expected");
393                          ((MethodEntryRequest) eventRequest1).addInstanceFilter(instance);
394                          log2("        no InvalidRequestStateException");
395                      } catch ( InvalidRequestStateException e ) {
396                          log3("ERROR: InvalidRequestStateException");
397                          testExitCode = FAILED;
398                      }
399 
400                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
401                      eventRManager.deleteEventRequest(eventRequest1);
402                      try {
403                          log2("......eventRequest1.addInstanceFilter(instance);");
404                          log2("        InvalidRequestStateException expected");
405                          ((MethodEntryRequest) eventRequest1).addInstanceFilter(instance);
406                          log3("ERROR: no InvalidRequestStateException");
407                          testExitCode = FAILED;
408                      } catch ( InvalidRequestStateException e ) {
409                          log2("        InvalidRequestStateException");
410                      }
411                       break;
412 
413               default:
414                       throw new JDITestRuntimeException("** default case 2 **");
415             }
416 
417             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
418         }
419         log1("    TESTING ENDS");
420         return;
421     }

423     private ThreadReference threadByName(String name)
424                  throws JDITestRuntimeException {
425 
426         List         all = vm.allThreads();
427         ListIterator li  = all.listIterator();
428 
429         for (; li.hasNext(); ) {
430             ThreadReference thread = (ThreadReference) li.next();
431             if (thread.name().equals(name))
432                 return thread;
433         }
434         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
435     }

437    /*
438     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
439     *                                             String, String, String)
440     *
441     * It sets up a breakpoint at given line number within a given method in a given class
442     * for a given thread.
443     *
444     * Return value: BreakpointRequest object  in case of success
445     *
446     * JDITestRuntimeException   in case of an Exception thrown within the method
447     */
448 
449     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
450                                                   ReferenceType testedClass,
451                                                   String methodName,
452                                                   String bpLine,
453                                                   String property)
454             throws JDITestRuntimeException {
455 
456         log2("......setting up a breakpoint:");
457         log2("       thread: " + thread + "; class: " + testedClass +
458                         "; method: " + methodName + "; line: " + bpLine);
459 
460         List              alllineLocations = null;
461         Location          lineLocation     = null;
462         BreakpointRequest breakpRequest    = null;
463 
464         try {
465             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
466 
467             alllineLocations = method.allLineLocations();
468 
469             int n =
470                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
471             if (n > alllineLocations.size()) {
472                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
473             } else {
474                 lineLocation = (Location) alllineLocations.get(n);
475                 try {
476                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
477                     breakpRequest.putProperty("number", property);
478                     breakpRequest.addThreadFilter(thread);
479                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
480                 } catch ( Exception e1 ) {
481                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
482                     breakpRequest    = null;
483                 }
484             }
485         } catch ( Exception e2 ) {
486             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
487             breakpRequest    = null;
488         }
489 
490         if (breakpRequest == null) {
491             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
492             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
493         }
494 
495         log2("      a breakpoint has been set up");
496         return breakpRequest;
497     }

500     private void getEventSet()
501                  throws JDITestRuntimeException {
502         try {
503 //            log2("       eventSet = eventQueue.remove(waitTime);");
504             eventSet = eventQueue.remove(waitTime);
505             if (eventSet == null) {
506                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
507             }
508 //            log2("       eventIterator = eventSet.eventIterator;");
509             eventIterator = eventSet.eventIterator();
510         } catch ( Exception e ) {
511             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
512         }
513     }

516     private void breakpointForCommunication()
517                  throws JDITestRuntimeException {
518 
519         log2("breakpointForCommunication");
520         getEventSet();
521 
522         if (eventIterator.nextEvent() instanceof BreakpointEvent)
523             return;
524 
525         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
526     }

530     private MethodEntryRequest setting2MethodEntryRequest ( ThreadReference thread,
531                                                             ReferenceType   testedClass,
532                                                             int             suspendPolicy,
533                                                             String          property       )
534             throws JDITestRuntimeException {
535         try {
536             log2("......setting up MethodEntryRequest:");
537             log2("       thread: " + thread + "; class: " + testedClass +  "; property: " + property);
538 
539             MethodEntryRequest
540             menr = eventRManager.createMethodEntryRequest();
541             menr.putProperty("number", property);
542             if (thread != null)
543                 menr.addThreadFilter(thread);
544             menr.addClassFilter(testedClass);
545             menr.setSuspendPolicy(suspendPolicy);
546 
547             log2("      a MethodEntryRequest has been set up");
548             return menr;
549         } catch ( Exception e ) {
550             log3("ERROR: ATTENTION: Exception within settingMethodEntryRequest() : " + e);
551             log3("       MethodEntryRequest HAS NOT BEEN SET UP");
552             throw new JDITestRuntimeException("** FAILURE to set up MethodEntryRequest **");
553         }
