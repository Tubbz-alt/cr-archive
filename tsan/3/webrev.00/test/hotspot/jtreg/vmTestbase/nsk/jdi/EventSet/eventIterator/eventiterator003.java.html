 36 /**
 37  * The test for the implementation of an object of the type     <BR>
 38  * EventSet.                                                    <BR>
 39  *                                                              <BR>
 40  * The test checks that results of the method                   <BR>
 41  * <code>com.sun.jdi.EventSet.eventIterator()</code>            <BR>
 42  * complies with its spec.                                      <BR>
 43  * <BR>
 44  * The test checks that for ThreadStartEvent and                <BR>
 45  * ThreadDeathEvent sets:                                       <BR>
 46  *  - the method returns non-null object;                       <BR>
 47  *  - object's class is subclass of class Iterator.             <BR>
 48  * <BR>
 49  * The test has three phases and works as follows.              <BR>
 50  * <BR>
 51  * In first phase,                                                      <BR>
 52  * upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 56  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 57  * and waits for the event within the predefined time interval.         <BR>
 58  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 61  * within debuggee's special methodForCommunication().                  <BR>
 62  * <BR>
 63  * In second phase the debugger and the debuggee perform the following. <BR>
 64  * - The debugger creates ThreadStartRequest,resumes the debuggee, and  <BR>
 65  *   waits for corresponding ThreadStartEvent.                          <BR>
 66  * - The debuggee starts new thread,  named "thread2",                  <BR>
 67  *   whose running creates the needed events.                           <BR>
 68  * - Upon getting the ThreadStartEvent, the debugger checks up on it,   <BR>
 69  *   creates ThreadDeathRequest, resumes the debuggee, and              <BR>
 70  *   waits for corresponding ThreadDeathEvent.                          <BR>
 71  *   Upon getting the ThreadDeathEvent, the debugger checks up on it.   <BR>
 72  * <BR>
 73  * In third phase, the debugger and the debuggee end.                   <BR>
 74  * <BR>
 75  */

156     private int runThis(String argv[], PrintStream out) {
157 
158         argsHandler = new ArgumentHandler(argv);
159         logHandler = new Log(out, argsHandler);
160         Binder binder = new Binder(argsHandler, logHandler);
161 
162         waitTime = argsHandler.getWaitTime() * 60000;
163 
164         try {
165             log2("launching a debuggee :");
166             log2("       " + debuggeeName);
167             if (argsHandler.verbose()) {
168                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
169             } else {
170                 debuggee = binder.bindToDebugee(debuggeeName);
171             }
172             if (debuggee == null) {
173                 log3("ERROR: no debuggee launched");
174                 return FAILED;
175             }
176             log2("debuggee launched");
177         } catch (Exception e) {
178             log3("ERROR: Exception : " + e);
179             log2("       test cancelled");
180             return FAILED;
181         }
182 
183         debuggee.redirectOutput(logHandler);
184 
185         vm = debuggee.VM();
186 
187         eventQueue = vm.eventQueue();
188         if (eventQueue == null) {
189             log3("ERROR: eventQueue == null : TEST ABORTED");
190             vm.exit(PASS_BASE);
191             return FAILED;
192         }
193 
194         log2("invocation of the method runTest()");
195         switch (runTest()) {
196 
197         case 0:
198             log2("test phase has finished normally");
199             log2("   waiting for the debuggee to finish ...");
200             debuggee.waitFor();
201 
202             log2("......getting the debuggee's exit status");
203             int status = debuggee.getStatus();
204             if (status != PASS_BASE) {
205                 log3("ERROR: debuggee returned UNEXPECTED exit status: " + status + " != PASS_BASE");
206                 testExitCode = FAILED;
207             } else {
208                 log2("......debuggee returned expected exit status: " + status + " == PASS_BASE");
209             }
210             break;
211 
212         default:
213             log3("ERROR: runTest() returned unexpected value");
214 
215         case 1:
216             log3("test phase has not finished normally: debuggee is still alive");
217             log2("......forcing: vm.exit();");
218             testExitCode = FAILED;
219             try {
220                 vm.exit(PASS_BASE);
221             } catch (Exception e) {
222                 log3("ERROR: Exception : e");
223             }
224             break;
225 
226         case 2:
227             log3("test cancelled due to VMDisconnectedException");
228             log2("......trying: vm.process().destroy();");
229             testExitCode = FAILED;
230             try {
231                 Process vmProcess = vm.process();
232                 if (vmProcess != null) {
233                     vmProcess.destroy();
234                 }
235             } catch (Exception e) {
236                 log3("ERROR: Exception : e");
237             }
238             break;
239         }
240 
241         return testExitCode;
242     }

245      * Return value: 0 - normal end of the test
246      *               1 - ubnormal end of the test
247      *               2 - VMDisconnectedException while test phase
248      */

250     private int runTest() {
251 
252         try {
253             testRun();
254 
255             log2("waiting for VMDeathEvent");
256             getEventSet();
257 
258             if (!(eventIterator.nextEvent() instanceof VMDeathEvent)) {
259                 log3("ERROR: last event is not the VMDeathEvent");
260                 return 1;
261             }
262 
263             log2("waiting for VMDisconnectEvent");
264             getEventSet();
265 
266             if (!(eventIterator.nextEvent() instanceof VMDisconnectEvent)) {
267                 log3("ERROR: last event is not the VMDisconnectEvent");
268                 return 1;
269             }
270 
271             return 0;
272 
273         } catch (VMDisconnectedException e) {
274             log3("ERROR: VMDisconnectedException : " + e);
275             e.printStackTrace(logHandler.getOutStream());
276             return 2;
277         } catch (Exception e) {
278             log3("ERROR: Exception : " + e);
279             return 1;
280         }
281 
282     }

284     private void testRun() throws JDITestRuntimeException, Exception {
285 
286         eventRManager = vm.eventRequestManager();
287 
288         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
289         cpRequest.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD);
290         cpRequest.addClassFilter(debuggeeName);
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300             throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm = "lineForComm";
306         BreakpointRequest bpRequest;
307 
308         ThreadReference mainThread = debuggee.threadByNameOrThrow("main");
309 
310         bpRequest = settingBreakpoint(mainThread, debuggeeClass, bPointMethod, lineForComm, "zero");
311         bpRequest.enable();
312 
313         //------------------------------------------------------  testing section
314 
315         log1("     TESTING BEGINS");
316 
317         String thread2Name = "thread2";
318 
319         EventRequest eRequest = null;
320 
321         label0: {
322 
323             vm.resume();
324             breakpointForCommunication();
325 
326             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
327 
328             ThreadReference tr = (ThreadReference) debuggeeClass.getValue(debuggeeClass.fieldByName(thread2Name));
329 
330             log2("......setting up ThreadStartRequest");
331             eRequest = settingThreadStartRequest(tr, EventRequest.SUSPEND_EVENT_THREAD, "ThreadStartRequest");
332             eRequest.enable();
333             vm.resume();
334             log2("......getting EventSet");
335             getEventSet();
336             eRequest.disable();
337 
338             String str = (String) eventIterator.nextEvent().request().getProperty("number");
339             if (!str.equals("ThreadStartRequest")) {
340                 testExitCode = FAILED;
341                 log3("ERROR: new event doesn't corresponds to ThreadStartRequest required");
342                 break label0;
343             }
344 
345             log2("......checking up on EventIterator");
346             if (eventIterator == null) {
347                 testExitCode = FAILED;
348                 log3("ERROR: eventIterator == null");
349             }
350             if (!(eventIterator instanceof Iterator)) {
351                 testExitCode = FAILED;
352                 log3("ERROR: eventIterator is NOT instanceof Iterator");
353             }
354 
355             log2("......setting up ThreadDeathRequest");
356             // suspend policy 'SUSPEND_ALL' should be used for ThreadDeathRequest (see 6609499)
357             eRequest = settingThreadDeathRequest(tr, EventRequest.SUSPEND_ALL, "ThreadDeathRequest");
358             eRequest.enable();
359             eventSet.resume();
360             log2("......getting EventSet");
361             getEventSet();
362             eRequest.disable();
363 
364             str = (String) eventIterator.nextEvent().request().getProperty("number");
365             if (!str.equals("ThreadDeathRequest")) {
366                 testExitCode = FAILED;
367                 log3("ERROR: new event doesn't corresponds to ThreadDeathRequest required");
368                 break label0;
369             }
370 
371             log2("......checking up on EventIterator");
372             if (eventIterator == null) {
373                 testExitCode = FAILED;
374                 log3("ERROR: eventIterator == null");
375             }
376             if (!(eventIterator instanceof Iterator)) {
377                 testExitCode = FAILED;
378                 log3("ERROR: eventIterator is NOT instanceof Iterator");
379             }
380 
381             eventSet.resume();
382             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
383         }
384 
385         log1("    TESTING ENDS");
386         return;
387     }

389     /*
390      * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
391      *                                             String, String, String)
392      *
393      * It sets up a breakpoint at given line number within a given method in a given class
394      * for a given thread.
395      *
396      * Return value: BreakpointRequest object  in case of success
397      *
398      * JDITestRuntimeException   in case of an Exception thrown within the method
399      */
400 
401     private BreakpointRequest settingBreakpoint(ThreadReference thread, ReferenceType testedClass, String methodName,
402             String bpLine, String property) throws JDITestRuntimeException {
403 
404         log2("......setting up a breakpoint:");
405         log2("       thread: " + thread + "; class: " + testedClass + "; method: " + methodName + "; line: " + bpLine);
406 
407         List alllineLocations = null;
408         Location lineLocation = null;
409         BreakpointRequest breakpRequest = null;
410 
411         try {
412             Method method = (Method) testedClass.methodsByName(methodName).get(0);
413 
414             alllineLocations = method.allLineLocations();
415 
416             int n = ((IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine))).value();
417             if (n > alllineLocations.size()) {
418                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
419             } else {
420                 lineLocation = (Location) alllineLocations.get(n);
421                 try {
422                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
423                     breakpRequest.putProperty("number", property);
424                     breakpRequest.addThreadFilter(thread);
425                     breakpRequest.setSuspendPolicy(EventRequest.SUSPEND_EVENT_THREAD);
426                 } catch (Exception e1) {
427                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
428                     breakpRequest = null;
429                 }
430             }
431         } catch (Exception e2) {
432             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
433             breakpRequest = null;
434         }
435 
436         if (breakpRequest == null) {
437             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
438             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
439         }
440 
441         log2("      a breakpoint has been set up");
442         return breakpRequest;
443     }

445     private void getEventSet() throws JDITestRuntimeException {
446         try {
447             eventSet = eventQueue.remove(waitTime);
448             if (eventSet == null) {
449                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
450             }
451             eventIterator = eventSet.eventIterator();
452         } catch (Exception e) {
453             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
454         }
455     }

457     private void breakpointForCommunication() throws JDITestRuntimeException {
458 
459         log2("breakpointForCommunication");
460         getEventSet();
461 
462         if (eventIterator.nextEvent() instanceof BreakpointEvent)
463             return;
464 
465         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
466     }

470     private ThreadStartRequest settingThreadStartRequest(ThreadReference thread, int suspendPolicy, String property)
471             throws JDITestRuntimeException {
472         try {
473             ThreadStartRequest tsr = eventRManager.createThreadStartRequest();
474             tsr.addThreadFilter(thread);
475             tsr.addCountFilter(1);
476             tsr.setSuspendPolicy(suspendPolicy);
477             tsr.putProperty("number", property);
478             return tsr;
479         } catch (Exception e) {
480             log3("ERROR: ATTENTION: Exception within settingThreadStartRequest() : " + e);
481             log3("       ThreadStartRequest HAS NOT BEEN SET UP");
482             throw new JDITestRuntimeException("** FAILURE to set up ThreadStartRequest **");
483         }
484     }
