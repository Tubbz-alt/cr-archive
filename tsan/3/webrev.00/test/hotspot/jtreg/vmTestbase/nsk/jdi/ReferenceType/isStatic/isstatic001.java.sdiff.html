466     private ThreadReference threadByName(String name)
467                  throws JDITestRuntimeException {
468 
469         List         all = vm.allThreads();
470         ListIterator li  = all.listIterator();
471 
472         for (; li.hasNext(); ) {
473             ThreadReference thread = (ThreadReference) li.next();
474             if (thread.name().equals(name))
475                 return thread;
476         }
477         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
478     }
