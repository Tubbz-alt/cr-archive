 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventSet.                                                    <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventSet.suspendPolicy()</code>            <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks that for a ClassPrepareEvent set,            <BR>
 46  * the method returns values corresponding to one               <BR>
 47  * which suspends the most threads.                             <BR>
 48  * The cases to check include event sets containing             <BR>
 49  * one to three threads with all possible combinations:         <BR>
 50  *    NONE, THREAD, ALL, NONE+THREAD, NONE+ALL,                 <BR>
 51  *    THREAD+ALL, NONE+THREAD+ALL                               <BR>
 52  * <BR>
 53  * The test has three phases and works as follows.              <BR>
 54  * <BR>
 55  * In first phase,                                                      <BR>
 56  * upon launching debuggee's VM which will be suspended,                <BR>
 57  * a debugger waits for the VMStartEvent within a predefined            <BR>
 58  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 59  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 60  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 61  * and waits for the event within the predefined time interval.         <BR>
 62  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 63  * Upon getting the ClassPrepareEvent,                                  <BR>
 64  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 65  * within debuggee's special methodForCommunication().                  <BR>
 66  * The debugger then prepares an array containing sets of               <BR>
 67  * enabled requests for each combination to check.                      <BR>
 68  * <BR>
 69  * In second phase to check the above,                                  <BR>
 70  * the debugger and the debuggee perform the following.                 <BR>
 71  * - The debugger resumes the debuggee and waits for ClassPrepareEvents.<BR>
 72  * - The debuggee makes new checked classes prepared and invokes then   <BR>
 73  *   the methodForCommunication to be suspended and                     <BR>
 74  *   to inform the debugger with the event.                             <BR>
 75  * - Upon getting the ClassPrepareEvents,                               <BR>
 76  *   the debugger performs the check required.                          <BR>
 77  * Note. To inform each other of needed actions, the debugger and       <BR>
 78  *       and the debuggee use debuggee's variable "instruction".        <BR>
 79  * <BR>
 80  * In third phase when at the end,                                      <BR>
 81  * the debuggee changes the value of the "instruction"                  <BR>
 82  * to inform the debugger of checks finished, and both end.             <BR>
 83  * <BR>
 84  */
 85

109 
110         int exitCode = new suspendpolicy001().runThis(argv, out);

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : e");
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : e");
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

251    /*
252     * Return value: 0 - normal end of the test
253     *               1 - ubnormal end of the test
254     *               2 - VMDisconnectedException while test phase
255     */

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if ( !(eventIterator.nextEvent() instanceof VMDeathEvent) ) {
265                 log3("ERROR: last event is not the VMDeathEvent");
266                 return 1;
267             }
268 
269             log2("waiting for VMDisconnectEvent");
270             getEventSet();
271             if ( !(eventIterator.nextEvent() instanceof VMDisconnectEvent) ) {
272                 log3("ERROR: last event is not the VMDisconnectEvent");
273                 return 1;
274             }
275 
276             return 0;
277 
278         } catch ( VMDisconnectedException e ) {
279             log3("ERROR: VMDisconnectedException : " + e);
280             return 2;
281         } catch ( Exception e ) {
282             log3("ERROR: Exception : " + e);
283             return 1;
284         }
285 
286     }

288     private void testRun()
289                  throws JDITestRuntimeException, Exception {
290 
291         eventRManager = vm.eventRequestManager();
292 
293         log2("......getting ClassPrepareEvent for debuggee's class");
294         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
295         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
296         cpRequest.addClassFilter(debuggeeName);
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307         log2("      received: ClassPrepareEvent for debuggeeClass");
308 
309         log2("......setting up ClassPrepareEvent for breakpointForCommunication");
310 
311         String            bPointMethod = "methodForCommunication";
312         String            lineForComm  = "lineForComm";
313         BreakpointRequest bpRequest;
314         ThreadReference   mainThread = threadByName("main");
315         bpRequest = settingBreakpoint(mainThread,
316                                       debuggeeClass,
317                                       bPointMethod, lineForComm, "zero");
318         bpRequest.enable();
319 
320     //------------------------------------------------------  testing section
321 
322         log1("     TESTING BEGINS");
323 
324         int classFlags = 0;
325 
326         String classNames[] = {
327             "nsk.jdi.EventSet.suspendPolicy.TestClass0",
328             "nsk.jdi.EventSet.suspendPolicy.TestClass1",
329             "nsk.jdi.EventSet.suspendPolicy.TestClass2",
330             "nsk.jdi.EventSet.suspendPolicy.TestClass3",
331             "nsk.jdi.EventSet.suspendPolicy.TestClass4",
332             "nsk.jdi.EventSet.suspendPolicy.TestClass5",
333             "nsk.jdi.EventSet.suspendPolicy.TestClass6"
334         };
335 
336         final int SUSPEND_POLICY = EventRequest.SUSPEND_NONE;
337         final int SUSPEND_NONE   = EventRequest.SUSPEND_NONE;
338         final int SUSPEND_THREAD = EventRequest.SUSPEND_EVENT_THREAD;
339         final int SUSPEND_ALL    = EventRequest.SUSPEND_ALL;
340 
341         int policyExpected[] = {    SUSPEND_NONE,
342                                     SUSPEND_THREAD,
343                                     SUSPEND_ALL,
344                                     SUSPEND_THREAD,
345                                     SUSPEND_ALL,
346                                     SUSPEND_ALL,
347                                     SUSPEND_ALL      };
348         int policy = 0;
349 
350         EventRequest eventRequests[][] = new EventRequest[classNames.length][3];
351 
352         for ( int nr = 0; nr < eventRequests.length; nr++) {
353             log2("......setting up ClassPrepareRequest for TestClass" + nr);
354             switch (nr) {
355 
356               case 0:
357                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
358                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
359                       eventRequests[nr][0].enable();
360                       break;
361 
362               case 1:
363                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
364                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
365                       eventRequests[nr][1].enable();
366                       break;
367 
368               case 2:
369                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
370                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
371                       eventRequests[nr][2].enable();
372                       break;
373 
374               case 3:
375                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
376                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
377                       eventRequests[nr][0].enable();
378                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
379                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
380                       eventRequests[nr][1].enable();
381                       break;
382 
383               case 4:
384                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
385                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
386                       eventRequests[nr][0].enable();
387                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
388                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
389                       eventRequests[nr][2].enable();
390                       break;
391 
392               case 5:
393                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
394                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
395                       eventRequests[nr][1].enable();
396                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
397                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
398                       eventRequests[nr][2].enable();
399                       break;
400 
401               case 6:
402                       eventRequests[nr][0] = settingClassPrepareRequest(classNames[nr],
403                                               SUSPEND_NONE, "ClassPrepareRequest" + nr);
404                       eventRequests[nr][0].enable();
405                       eventRequests[nr][1] = settingClassPrepareRequest(classNames[nr],
406                                               SUSPEND_THREAD, "ClassPrepareRequest" + nr);
407                       eventRequests[nr][1].enable();
408                       eventRequests[nr][2] = settingClassPrepareRequest(classNames[nr],
409                                               SUSPEND_ALL, "ClassPrepareRequest" + nr);
410                       eventRequests[nr][2].enable();
411                       break;
412 
413               default:
414                       throw new JDITestRuntimeException("** default case 2 **");
415             }
416             classFlags |= 1 << nr;
417         }
418 
419         log2("......checking up on eventSets");
420 
421         vm.resume();
422 
423         for (int ne = 0; ne < eventRequests.length; ne++) {
424 
425             getEventSet();
426 
427             Event ev = eventIterator.nextEvent();
428             if ( !( ev instanceof ClassPrepareEvent)) {
429                  log3("ERROR: new event is not ClassPrepareEvent");
430                  testExitCode = FAILED;
431                  break;
432             }
433 
434             String property = (String) ev.request().getProperty("number");
435 
436             int np = 0;
437             for (; np < eventRequests.length; np++) {
438                 if (property.equals("ClassPrepareRequest" + np))
439                     break;
440             }
441 
442             policy = eventSet.suspendPolicy();
443 
444             if (policy != policyExpected[np]) {
445                 log3("ERROR: eventSet.suspendPolicy() != policyExpected");
446                 log3("       eventSet.suspendPolicy() == " + policy);
447                 log3("       policyExpected           == " + policyExpected[np]);
448                 testExitCode = FAILED;
449             }
450 
451             switch (np) {
452               case 0: eventRequests[np][0].disable();                                 break;
453               case 1: eventRequests[np][1].disable();                                 break;
454               case 2: eventRequests[np][2].disable();                                 break;
455               case 3: eventRequests[np][0].disable(); eventRequests[np][1].disable(); break;
456               case 4: eventRequests[np][0].disable(); eventRequests[np][2].disable(); break;
457               case 5: eventRequests[np][1].disable(); eventRequests[np][2].disable(); break;
458               case 6: eventRequests[np][0].disable(); eventRequests[np][1].disable();
459                       eventRequests[np][2].disable();                                 break;
460            }
461 
462             switch (policy) {
463               case SUSPEND_NONE   :                      break;
464               case SUSPEND_THREAD : mainThread.resume(); break;
465               case SUSPEND_ALL    : vm.resume();         break;
466               default: throw new JDITestRuntimeException("** default case 1 **");
467             }
468             classFlags &= 1 << ne;
469         }
470         if (classFlags != 0) {
471             log3("ERROR: not all events received : " + classFlags);
472             testExitCode = FAILED;
473         }
474 
475         for (int i = 0; ; i++) {
476 
477             log1(":::::: case: # " + i);
478 
479             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
480 
481             breakpointForCommunication();
482 
483             int instruction = ((IntegerValue)
484                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
485             if (instruction == 0) {
486                 vm.resume();
487                 break;
488             }
489             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
490         }
491         log1("    TESTING ENDS");
492         return;
493     }

495     private ThreadReference threadByName(String name)
496                  throws JDITestRuntimeException {
497 
498         List         all = vm.allThreads();
499         ListIterator li  = all.listIterator();
500 
501         for (; li.hasNext(); ) {
502             ThreadReference thread = (ThreadReference) li.next();
503             if (thread.name().equals(name))
504                 return thread;
505         }
506         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
507     }

509    /*
510     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
511     *                                             String, String, String)
512     *
513     * It sets up a breakpoint at given line number within a given method in a given class
514     * for a given thread.
515     *
516     * Return value: BreakpointRequest object  in case of success
517     *
518     * JDITestRuntimeException   in case of an Exception thrown within the method
519     */
520 
521     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
522                                                   ReferenceType testedClass,
523                                                   String methodName,
524                                                   String bpLine,
525                                                   String property)
526             throws JDITestRuntimeException {
527 
528         log2("......setting up a breakpoint:");
529         log2("       thread: " + thread + "; class: " + testedClass +
530                         "; method: " + methodName + "; line: " + bpLine);
531 
532         List              alllineLocations = null;
533         Location          lineLocation     = null;
534         BreakpointRequest breakpRequest    = null;
535 
536         try {
537             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
538 
539             alllineLocations = method.allLineLocations();
540 
541             int n =
542                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
543             if (n > alllineLocations.size()) {
544                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
545             } else {
546                 lineLocation = (Location) alllineLocations.get(n);
547                 try {
548                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
549                     breakpRequest.putProperty("number", property);
550                     breakpRequest.addThreadFilter(thread);
551                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
552                 } catch ( Exception e1 ) {
553                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
554                     breakpRequest    = null;
555                 }
556             }
557         } catch ( Exception e2 ) {
558             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
559             breakpRequest    = null;
560         }
561 
562         if (breakpRequest == null) {
563             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
564             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
565         }
566 
567         log2("      a breakpoint has been set up");
568         return breakpRequest;
569     }

572     private void getEventSet()
573                  throws JDITestRuntimeException {
574         try {
575 //            log2("       eventSet = eventQueue.remove(waitTime);");
576             eventSet = eventQueue.remove(waitTime);
577             if (eventSet == null) {
578                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
579             }
580 //            log2("       eventIterator = eventSet.eventIterator;");
581             eventIterator = eventSet.eventIterator();
582         } catch ( Exception e ) {
583             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
584         }
585     }

588     private void breakpointForCommunication()
589                  throws JDITestRuntimeException {
590 
591         log2("breakpointForCommunication");
592         getEventSet();
593 
594         if (eventIterator.nextEvent() instanceof BreakpointEvent)
595             return;
596 
597         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
598     }

602     private ClassPrepareRequest settingClassPrepareRequest ( String  testedClass,
603                                                              int     suspendPolicy,
604                                                              String  property       )
605             throws JDITestRuntimeException {
606         try {
607             log2("......setting up ClassPrepareRequest:");
608             log2("       class: " + testedClass + "; property: " + property);
609 
610             ClassPrepareRequest
611             cpr = eventRManager.createClassPrepareRequest();
612             cpr.putProperty("number", property);
613             cpr.addClassFilter(testedClass);
614             cpr.setSuspendPolicy(suspendPolicy);
615 
616             log2("      ClassPrepareRequest has been set up");
617             return cpr;
618         } catch ( Exception e ) {
619             log3("ERROR: ATTENTION: Exception within settingClassPrepareRequest() : " + e);
620             log3("       ClassPreparenRequest HAS NOT BEEN SET UP");
621             throw new JDITestRuntimeException("** FAILURE to set up ClassPrepareRequest **");
622         }
623     }
