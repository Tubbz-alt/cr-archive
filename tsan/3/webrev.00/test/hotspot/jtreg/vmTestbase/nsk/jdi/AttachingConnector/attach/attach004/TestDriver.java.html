 24 package nsk.jdi.AttachingConnector.attach.attach004;

 26 import jdk.test.lib.JDKToolFinder;
 27 import jdk.test.lib.Utils;
 28 import jtreg.SkippedException;

 31 import java.io.BufferedReader;
 32 import java.io.IOException;
 33 import java.io.InputStream;
 34 import java.io.InputStreamReader;
 35 import java.util.ArrayList;
 36 import java.util.Arrays;
 37 import java.util.Collections;
 38 import java.util.List;
 39 
 40 public class TestDriver {
 41     public static void main(String[] args) throws InterruptedException {
 42         String arch = args[0];
 43         String transport = args[1];
 44         String jdiTransport;
 45         // Need convert transport argument to string recognizable by nsk.share.jdi.ArgumentHandler
 46         switch (transport) {
 47             case "dt_socket":
 48                 jdiTransport = "socket";
 49                 break;
 50             case "dt_shmem":
 51                 jdiTransport = "shmem";
 52                 break;
 53             default:
 54                 throw new Error("Unexpected transport " + args[0]
 55                         + ", expected is dt_socket or dt_shmem");
 56         }
 57         String[] jdiArgs = new String[]{
 58                 "-connector=attaching",
 59                 "-transport=" + jdiTransport,
 60                 "-verbose",
 61                 "-waittime=5",
 62                 "-debugee.vmkind=java",
 63                 "-transport.address=dynamic",
 64                 "-arch=" + arch
 65         };
 66 
 67         if (!isTransportSupported(jdiArgs)) {
 68             throw new SkippedException("Transport isn't supported on this platform");
 69         }
 70 
 71         System.out.println("Transport is supported on this platform, execute test");
 72         String suspend = args[2];
 73         Process debuggee = startDebuggee(jdiArgs, transport, suspend);
 74         Process debugger = startDebugger(jdiArgs, Arrays.copyOfRange(args, 3, args.length), debuggee.pid());
 75 
 76         int debuggerExit = debugger.waitFor();
 77         if (debuggerExit != 95) {
 78             throw new Error("debugger exit code is " + debuggerExit);
 79         }
 80 
 81         int debuggeeExit = debuggee.waitFor();
 82         if (debuggeeExit != 95) {
 83             throw new Error("debuggee exit code is " + debuggeeExit);
 84         }
 85     }
 86 
 87 
 88     private static Process startDebuggee(String[] jdiArgs, String transport, String suspend) {
 89         List<String> cmd = new ArrayList<>();
 90         Class<?> debuggeeClass = attach004t.class;
 91         cmd.add(JDKToolFinder.getJDKTool("java"));
 92         Collections.addAll(cmd, Utils.prependTestJavaOpts(
 93                 "-cp",
 94                 Utils.TEST_CLASS_PATH,
 95                 "-Xdebug",
 96                 "-agentlib:jdwp=transport=" + transport + ",server=y,suspend=" + suspend,
 97                 "-Dmy.little.cookie=" + ProcessHandle.current().pid(),
 98                 debuggeeClass.getName()));
 99         Collections.addAll(cmd, jdiArgs);
100         cmd.add("-testWorkDir");
101         cmd.add(".");
102 
103         System.out.println("Starting debuggee [" + String.join(",", cmd) + "]");
104         Process p;
105         try {
106             p = new ProcessBuilder(cmd).redirectErrorStream(true).start();
107         } catch (IOException e) {
108             throw new Error("can't start debuggee");
109         }
110 
111         Thread t = new Thread(() -> dumpStream("debuggee>> ", p.getInputStream()));
112         t.setDaemon(true);
113         t.start();
114 
115         return p;
116     }
117 
118     private static Process startDebugger(String[] jdiArgs, String[] debuggerArgs, long debuggeePid) {
119         List<String> cmd = new ArrayList<>();
120         Class<?> debuggerClass = attach004.class;
121         cmd.add(JDKToolFinder.getJDKTool("java"));
122         Collections.addAll(cmd, Utils.prependTestJavaOpts(
123                 "-cp",
124                 Utils.TEST_CLASS_PATH,
125                 debuggerClass.getName(),
126                 "-debuggeePID",
127                 "" + debuggeePid));
128         Collections.addAll(cmd, debuggerArgs);
129         Collections.addAll(cmd, jdiArgs);
130         cmd.add("-testWorkDir");
131         cmd.add(".");
132 
133         System.out.println("Starting debugger [" + String.join(",", cmd) + "]");
134         Process p;
135         try {
136             p = new ProcessBuilder(cmd).redirectErrorStream(true).start();
137         } catch (IOException e) {
138             throw new Error("can't start debugger");
139         }
140 
141         Thread t = new Thread(() -> dumpStream("debugger>> ", p.getInputStream()));
142         t.setDaemon(true);
143         t.start();
144 
145         return p;
146     }
147 
148     private static void dumpStream(String prefix, InputStream is) {
149 
150         byte[] buffer = new byte[1024];
151         int n;
152         try (BufferedReader r = new BufferedReader(new InputStreamReader(is))) {
153             String line;
154             while ((line = r.readLine()) != null) {
155                 System.out.println(prefix + line);
156                 System.out.flush();
157             }
158         } catch (IOException e) {
159             e.printStackTrace();
160         }
161     }
162 
163     private static boolean isTransportSupported(String[] jdiArgs) {
164         ArgumentHandler argHandler = new ArgumentHandler(jdiArgs);
165 
166         boolean result;
167 
168         if (argHandler.isShmemTransport()) {
169             result = !argHandler.shouldPass("dt_shmem");
170         } else {
171             result = !argHandler.shouldPass("dt_socket");
172         }
173 
174         if (result) {
175             System.out.println("Transport " + argHandler.getTransportType()
176                     + " is result at " + argHandler.getArch());
177         } else {
178             System.out.println("Transport " + argHandler.getTransportType()
179                     + " isn't result at " + argHandler.getArch());
180         }
181 
182         return result;
183     }
