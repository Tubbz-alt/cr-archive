 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ThreadReference.                                             <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ThreadReference.popFrames()</code>         <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *  After this operation,                                       <BR>
 47  *  this thread will be suspended at the invoke instruction of  <BR>
 48  *  the target method that created frame.                       <BR>
 49  *  The frame's method can be reentered with a                  <BR>
 50  *  step into the instruction.                                  <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,
 63  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assetion,                               <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new check case and invokes                   <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the check case required.                     <BR>
 74  * Note. To inform each other of needed actions, the debugger and       <BR>
 75  *       and the debuggee use debuggeee's variable "instruction".       <BR>
 76  * <BR>
 77  * In third phase when at the end,                                      <BR>
 78  * the debuggee changes the value of the "instruction"                  <BR>
 79  * to inform the debugger of checks finished, and both end.             <BR>
 80  * <BR>
 81  * In the second phase, the debuggee creates second thread which        <BR>
 82  * will call a tested poppedMethod() from its run() method.             <BR>
 83  * Being called, the poppedMethod() increments the value of             <BR>
 84  * the global variable testVar1.                                        <BR>
 85  * The debugger sets up a breakpoint within the poppedMethod() and,     <BR>
 86  * after getting this brakpoint event, invokes the popFrames() method,  <BR>
 87  * checks the value of testVar1 after first incerment,                  <BR>
 88  * and resumes debuggee's second thread to reenter the poppedMethod().  <BR>
 89  * After getting the breakpoint event second time,                      <BR>
 90  * the debugger just checks the value of testVar1 which                 <BR>
 91  * has to be incremented one more time.                                 <BR>
 92  */

173     private int runThis (String argv[], PrintStream out) {
174 
175         argsHandler     = new ArgumentHandler(argv);
176         logHandler      = new Log(out, argsHandler);
177         Binder binder   = new Binder(argsHandler, logHandler);
178 
179         waitTime        = argsHandler.getWaitTime() * 60000;
180 
181         try {
182             log2("launching a debuggee :");
183             log2("       " + debuggeeName);
184             if (argsHandler.verbose()) {
185                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
186             } else {
187                 debuggee = binder.bindToDebugee(debuggeeName);
188             }
189             if (debuggee == null) {
190                 log3("ERROR: no debuggee launched");
191                 return FAILED;
192             }
193             log2("debuggee launched");
194         } catch ( Exception e ) {
195             log3("ERROR: Exception : " + e);
196             log2("       test cancelled");
197             return FAILED;
198         }
199 
200         debuggee.redirectOutput(logHandler);
201 
202         vm = debuggee.VM();
203 
204         eventQueue = vm.eventQueue();
205         if (eventQueue == null) {
206             log3("ERROR: eventQueue == null : TEST ABORTED");
207             vm.exit(PASS_BASE);
208             return FAILED;
209         }
210 
211         log2("invocation of the method runTest()");
212         switch (runTest()) {
213 
214             case 0 :  log2("test phase has finished normally");
215                       log2("   waiting for the debuggee to finish ...");
216                       debuggee.waitFor();
217 
218                       log2("......getting the debuggee's exit status");
219                       int status = debuggee.getStatus();
220                       if (status != PASS_BASE) {
221                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
222                               status + " != PASS_BASE");
223                           testExitCode = FAILED;
224                       } else {
225                           log2("......debuggee returned expected exit status: " +
226                               status + " == PASS_BASE");
227                       }
228                       break;
229 
230             default : log3("ERROR: runTest() returned unexpected value");
231 
232             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
233                       log2("......forcing: vm.exit();");
234                       testExitCode = FAILED;
235                       try {
236                           vm.exit(PASS_BASE);
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : e");
239                       }
240                       break;
241 
242             case 2 :  log3("test cancelled due to VMDisconnectedException");
243                       log2("......trying: vm.process().destroy();");
244                       testExitCode = FAILED;
245                       try {
246                           Process vmProcess = vm.process();
247                           if (vmProcess != null) {
248                               vmProcess.destroy();
249                           }
250                       } catch ( Exception e ) {
251                           log3("ERROR: Exception : e");
252                       }
253                       break;
254             }
255 
256         return testExitCode;
257     }

266     private int runTest() {
267 
268         try {
269             testRun();
270 
271             log2("waiting for VMDeathEvent");
272             getEventSet();
273             if (eventIterator.nextEvent() instanceof VMDeathEvent)
274                 return 0;
275 
276             log3("ERROR: last event is not the VMDeathEvent");
277             return 1;
278         } catch ( VMDisconnectedException e ) {
279             log3("ERROR: VMDisconnectedException : " + e);
280             return 2;
281         } catch ( Exception e ) {
282             log3("ERROR: Exception : " + e);
283             return 1;
284         }
285

288     private void testRun()
289                  throws JDITestRuntimeException, Exception {
290 
291         eventRManager = vm.eventRequestManager();
292 
293         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
294         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
295         cpRequest.addClassFilter(debuggeeName);
296 
297         cpRequest.enable();
298         vm.resume();
299         getEventSet();
300         cpRequest.disable();
301 
302         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
303         debuggeeClass = event.referenceType();
304 
305         if (!debuggeeClass.name().equals(debuggeeName))
306            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
307 
308         log2("      received: ClassPrepareEvent for debuggeeClass");
309 
310         String bPointMethod = "methodForCommunication";
311         String lineForComm  = "lineForComm";
312         BreakpointRequest bpRequest;
313 
314         try {
315             bpRequest = settingBreakpoint(threadByName("main"),
316                                           debuggeeClass,
317                                           bPointMethod, lineForComm, "zero");
318         } catch ( Exception e ) {
319             throw e;
320         }
321         bpRequest.enable();
322 
323     //------------------------------------------------------  testing section
324 
325         log1("     TESTING BEGINS");
326 
327         log2("......vm.resume();  resuming debuggee's main thread");
328         vm.resume();
329 
330         for (int i = 0; ; i++) {
331 
332 //            vm.resume();
333 
334             breakpointForCommunication();
335 
336             int instruction;
337 
338             log2("......instruction = ((IntegerValue)...; no Exception expected");
339             try {
340                 instruction = ((IntegerValue)
341                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
342             } catch ( Exception e ) {
343                 log3("ERROR: Exception when 'instruction = ((IntegerValue)...' : " + e);
344                 testExitCode = FAILED;
345                 return;
346             }
347 
348             if (instruction == 0) {
349                 vm.resume();
350                 break;
351             }
352 
353             log1("  new check: # " + i);
354 
355             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
356 
357             if ( !vm.canPopFrames() ) {
358                 log2("......vm.canPopFrames() == false : test is cancelled");
359                 return;
360             }
361 
362             String thread2Name         = "thread2";
363             ThreadReference thread2Ref = threadByName(thread2Name);
364 
365 
366             String poppedMethod    = "poppedMethod";
367             String breakpointLine  = "breakpointLine";
368             String testVariable    = "testVar1";
369 
370             log2("......setting breakpoint in poppedMethod");
371             try {
372                 breakpointRequest = settingBreakpoint(threadByName(thread2Name),
373                                           debuggeeClass,
374                                           poppedMethod, breakpointLine, "one");
375             } catch ( Exception e ) {
376                 throw e;
377             }
378             log2("......breakpointRequest.enable();");
379             breakpointRequest.enable();
380 
381             log2("......eventSet.resume();   resuming debuggee's main thread");
382             log2("                           to get new rendevous at the breakpointForCommunication");
383             eventSet.resume();
384             log2("......breakpointInMethod();");
385             breakpointInMethod();
386 
387             log2("......getting value of 'testVar1'; 1 is expected");
388             Value val = debuggeeClass.getValue(debuggeeClass.fieldByName(testVariable));
389             int intVal = ((IntegerValue) val).value();
390             if (intVal != 1) {
391                 log3("ERROR: intVal != 1");
392                 testExitCode = FAILED;
393                 break;
394             }
395 
396             log2("......StackFrame stackFrame = thread2Ref.frame(0);");
397             StackFrame stackFrame = thread2Ref.frame(0);
398 
399             log2("......thread2Ref.popFrames(stackFrame);");
400             try {
401                 thread2Ref.popFrames(stackFrame);
402             } catch ( IncompatibleThreadStateException e ) {
403                 log3("ERROR: IncompatibleThreadStateException");
404                 testExitCode = FAILED;
405                 break;
406             }
407 
408             log2("......thread2Ref.resume();");
409             thread2Ref.resume();
410             log2("......breakpointInMethod();");
411             breakpointInMethod();
412 
413             log2("......getting value of 'testVar1'; 2 is expected");
414             val = debuggeeClass.getValue(debuggeeClass.fieldByName(testVariable));
415             intVal = ((IntegerValue) val).value();
416             if (intVal != 2) {
417                 log3("ERROR: intVal != 2");
418                 testExitCode = FAILED;
419                 break;
420             }
421 
422             log2("......thread2Ref.resume();");
423             thread2Ref.resume();
424 
425             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
426         }
427         log1("    TESTING ENDS");
428         return;
429     }

431     private ThreadReference threadByName(String name)
432                  throws JDITestRuntimeException {
433 
434         List         all = vm.allThreads();
435         ListIterator li  = all.listIterator();
436 
437         for (; li.hasNext(); ) {
438             ThreadReference thread = (ThreadReference) li.next();
439             if (thread.name().equals(name))
440                 return thread;
441         }
442         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
443     }

445    /*
446     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
447     *                                             String, String, String)
448     *
449     * It sets up a breakpoint at given line number within a given method in a given class
450     * for a given thread.
451     *
452     * Return value: BreakpointRequest object  in case of success
453     *
454     * JDITestRuntimeException   in case of an Exception thrown within the method
455     */
456 
457     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
458                                                   ReferenceType testedClass,
459                                                   String methodName,
460                                                   String bpLine,
461                                                   String property)
462             throws JDITestRuntimeException {
463 
464         log2("......setting up a breakpoint:");
465         log2("       thread: " + thread + "; class: " + testedClass +
466                         "; method: " + methodName + "; line: " + bpLine);
467 
468         List              alllineLocations = null;
469         Location          lineLocation     = null;
470         BreakpointRequest breakpRequest    = null;
471 
472         try {
473             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
474 
475             alllineLocations = method.allLineLocations();
476 
477             int n =
478                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
479             if (n > alllineLocations.size()) {
480                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
481             } else {
482                 lineLocation = (Location) alllineLocations.get(n);
483                 try {
484                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
485                     breakpRequest.putProperty("number", property);
486                     breakpRequest.addThreadFilter(thread);
487                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
488                 } catch ( Exception e1 ) {
489                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
490                     breakpRequest    = null;
491                 }
492             }
493         } catch ( Exception e2 ) {
494             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
495             breakpRequest    = null;
496         }
497 
498         if (breakpRequest == null) {
499             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
500             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
501         }
502 
503         log2("      a breakpoint has been set up");
504         return breakpRequest;
505     }

508     private void getEventSet()
509                  throws JDITestRuntimeException {
510         try {
511 //            log2("       eventSet = eventQueue.remove(waitTime);");
512             eventSet = eventQueue.remove(waitTime);
513             if (eventSet == null) {
514                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
515             }
516 //            log2("       eventIterator = eventSet.eventIterator;");
517             eventIterator = eventSet.eventIterator();
518         } catch ( Exception e ) {
519             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
520         }
521     }

524     private void breakpointForCommunication()
525                  throws JDITestRuntimeException {
526 
527         log2("breakpointForCommunication");
528         getEventSet();
529 
530         if (eventIterator.nextEvent() instanceof BreakpointEvent)
531             return;
532 
533         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
534     }

538     private void breakpointInMethod()
539                  throws JDITestRuntimeException {
540 
541         log2("breakpointInMethod");
542         getEventSet();
543         Event event = eventIterator.nextEvent();
544 
545         if ( !(event instanceof BreakpointEvent) )
546             throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
547 
548         if ( !event.request().equals(breakpointRequest) )
549             throw new JDITestRuntimeException("** unexpected breakpoint **");
550     }
