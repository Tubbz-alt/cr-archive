133     private int runThis (String argv[], PrintStream out) {
134 
135         argsHandler     = new ArgumentHandler(argv);
136         logHandler      = new Log(out, argsHandler);
137         Binder binder   = new Binder(argsHandler, logHandler);
138 
139         waitTime        = argsHandler.getWaitTime() * 60000;
140 
141         try {
142             log2("launching a debuggee :");
143             log2("       " + debuggeeName);
144             if (argsHandler.verbose()) {
145                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
146             } else {
147                 debuggee = binder.bindToDebugee(debuggeeName);
148             }
149             if (debuggee == null) {
150                 log3("ERROR: no debuggee launched");
151                 return FAILED;
152             }
153             log2("debuggee launched");
154         } catch ( Exception e ) {
155             log3("ERROR: Exception : " + e);
156             log2("       test cancelled");
157             return FAILED;
158         }
159 
160         debuggee.redirectOutput(logHandler);
161 
162         vm = debuggee.VM();
163 
164         eventQueue = vm.eventQueue();
165         if (eventQueue == null) {
166             log3("ERROR: eventQueue == null : TEST ABORTED");
167             vm.exit(PASS_BASE);
168             return FAILED;
169         }
170 
171         log2("invocation of the method runTest()");
172         switch (runTest()) {
173 
174             case 0 :  log2("test phase has finished normally");
175                       log2("   waiting for the debuggee to finish ...");
176                       debuggee.waitFor();
177 
178                       log2("......getting the debuggee's exit status");
179                       int status = debuggee.getStatus();
180                       if (status != PASS_BASE) {
181                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
182                               status + " != PASS_BASE");
183                           testExitCode = FAILED;
184                       } else {
185                           log2("......debuggee returned expected exit status: " +
186                               status + " == PASS_BASE");
187                       }
188                       break;
189 
190             default : log3("ERROR: runTest() returned unexpected value");
191 
192             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
193                       log2("......forcing: vm.exit();");
194                       testExitCode = FAILED;
195                       try {
196                           vm.exit(PASS_BASE);
197                       } catch ( Exception e ) {
198                           log3("ERROR: Exception : e");
199                       }
200                       break;
201 
202             case 2 :  log3("test cancelled due to VMDisconnectedException");
203                       log2("......trying: vm.process().destroy();");
204                       testExitCode = FAILED;
205                       try {
206                           Process vmProcess = vm.process();
207                           if (vmProcess != null) {
208                               vmProcess.destroy();
209                           }
210                       } catch ( Exception e ) {
211                           log3("ERROR: Exception : e");
212                       }
213                       break;
214             }
215 
216         return testExitCode;
217     }

226     private int runTest() {
227 
228         try {
229             testRun();
230 
231             log2("waiting for VMDeathEvent");
232             getEventSet();
233             if (eventIterator.nextEvent() instanceof VMDeathEvent)
234                 return 0;
235 
236             log3("ERROR: last event is not the VMDeathEvent");
237             return 1;
238         } catch ( VMDisconnectedException e ) {
239             log3("ERROR: VMDisconnectedException : " + e);
240             return 2;
241         } catch ( Exception e ) {
242             log3("ERROR: Exception : " + e);
243             return 1;
244         }
245 
246     }

248     private void testRun()
249                  throws JDITestRuntimeException, Exception {
250 
251         eventRManager = vm.eventRequestManager();
252 
253         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
254         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
255         cpRequest.addClassFilter(debuggeeName);
256 
257         cpRequest.enable();
258         vm.resume();
259         getEventSet();
260         cpRequest.disable();
261 
262         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
263         debuggeeClass = event.referenceType();
264 
265         if (!debuggeeClass.name().equals(debuggeeName))
266            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
267 
268         log2("      received: ClassPrepareEvent for debuggeeClass");
269 
270         String bPointMethod = "methodForCommunication";
271         String lineForComm  = "lineForComm";
272         BreakpointRequest bpRequest;
273 
274         try {
275             bpRequest = settingBreakpoint(threadByName("main"),
276                                           debuggeeClass,
277                                           bPointMethod, lineForComm, "zero");
278         } catch ( Exception e ) {
279             throw e;
280         }
281         bpRequest.enable();
282 
283     //------------------------------------------------------  testing section
284 
285         log1("     TESTING BEGINS");
286 
287         for (int i = 0; ; i++) {
288 
289             vm.resume();
290             breakpointForCommunication();
291 
292             int instruction = ((IntegerValue)
293                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();

295             if (instruction == 0) {
296                 vm.resume();
297                 break;
298             }
299 
300             log1("  new check: # " + i);

448     private ThreadReference threadByName(String name)
449                  throws JDITestRuntimeException {
450 
451         List         all = vm.allThreads();
452         ListIterator li  = all.listIterator();
453 
454         for (; li.hasNext(); ) {
455             ThreadReference thread = (ThreadReference) li.next();
456             if (thread.name().equals(name))
457                 return thread;
458         }
459         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
460     }

462    /*
463     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
464     *                                             String, String, String)
465     *
466     * It sets up a breakpoint at given line number within a given method in a given class
467     * for a given thread.
468     *
469     * Return value: BreakpointRequest object  in case of success
470     *
471     * JDITestRuntimeException   in case of an Exception thrown within the method
472     */
473 
474     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
475                                                   ReferenceType testedClass,
476                                                   String methodName,
477                                                   String bpLine,
478                                                   String property)
479             throws JDITestRuntimeException {
480 
481         log2("......setting up a breakpoint:");
482         log2("       thread: " + thread + "; class: " + testedClass +
483                         "; method: " + methodName + "; line: " + bpLine);
484 
485         List              alllineLocations = null;
486         Location          lineLocation     = null;
487         BreakpointRequest breakpRequest    = null;
488 
489         try {
490             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
491 
492             alllineLocations = method.allLineLocations();
493 
494             int n =
495                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
496             if (n > alllineLocations.size()) {
497                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
498             } else {
499                 lineLocation = (Location) alllineLocations.get(n);
500                 try {
501                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
502                     breakpRequest.putProperty("number", property);
503                     breakpRequest.addThreadFilter(thread);
504                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
505                 } catch ( Exception e1 ) {
506                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
507                     breakpRequest    = null;
508                 }
509             }
510         } catch ( Exception e2 ) {
511             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
512             breakpRequest    = null;
513         }
514 
515         if (breakpRequest == null) {
516             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
517             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
518         }
519 
520         log2("      a breakpoint has been set up");
521         return breakpRequest;
522     }

525     private void getEventSet()
526                  throws JDITestRuntimeException {
527         try {
528 //            log2("       eventSet = eventQueue.remove(waitTime);");
529             eventSet = eventQueue.remove(waitTime);
530             if (eventSet == null) {
531                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
532             }
533 //            log2("       eventIterator = eventSet.eventIterator;");
534             eventIterator = eventSet.eventIterator();
535         } catch ( Exception e ) {
536             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
537         }
538     }

541     private void breakpointForCommunication()
542                  throws JDITestRuntimeException {
543 
544         log2("breakpointForCommunication");
545         getEventSet();
546 
547         if (eventIterator.nextEvent() instanceof BreakpointEvent)
548             return;
549 
550         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
551     }
