505     private ThreadReference threadByName(String name)
506                  throws JDITestRuntimeException {
507 
508         List         all = vm.allThreads();
509         ListIterator li  = all.listIterator();
510 
511         for (; li.hasNext(); ) {
512             ThreadReference thread = (ThreadReference) li.next();
513             if (thread.name().equals(name))
514                 return thread;
515         }
516         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
517     }
