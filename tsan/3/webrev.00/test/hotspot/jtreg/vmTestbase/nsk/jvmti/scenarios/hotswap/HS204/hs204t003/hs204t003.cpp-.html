 35 #define FILE_NAME "nsk/jvmti/scenarios/hotswap/HS204/hs204t003/MyThread"
 36 #define CLASS_NAME  "Lnsk/jvmti/scenarios/hotswap/HS204/hs204t003/MyThread;"
 37 #define FIELDNAME "intState"
 38 #define TYPE "I"

 40 static jint redefineNumber;
 41 static jvmtiEnv * jvmti;
 42 static jclass watchFieldClass;

 44 JNIEXPORT void JNICALL callbackClassPrepare(jvmtiEnv *jvmti_env,
 45         JNIEnv* jni,
 46         jthread thread,
 47         jclass klass) {
 48     char * className;
 49     char * generic;
 50 
 51     className = NULL;
 52     generic   = NULL;
 53     redefineNumber=0;
 54     if (!NSK_JVMTI_VERIFY(jvmti_env->GetClassSignature(klass, &className, &generic))) {
 55         NSK_DISPLAY0(" Agent :: Failed get class signature.\n");
 56         nsk_jvmti_agentFailed();
 57     } else {
 58         if ((strcmp(className, CLASS_NAME) == 0)) {
 59             jfieldID fieldId;
 60             if (!NSK_JNI_VERIFY(jni, (fieldId = jni->GetStaticFieldID(klass, FIELDNAME, TYPE)) != NULL)) {
 61                     NSK_DISPLAY0(" Agent :: Failed to get FieldId.\n");
 62                     nsk_jvmti_agentFailed();
 63             } else {
 64                 if (!NSK_JVMTI_VERIFY(jvmti_env->SetFieldAccessWatch(klass, fieldId))) {
 65                     NSK_DISPLAY0(" Agent :: Failed to set watch point on a field.\n");
 66                     nsk_jvmti_agentFailed();
 67                 } else {
 68                     nsk_jvmti_enableNotification(jvmti_env, JVMTI_EVENT_FIELD_ACCESS, NULL);
 69                     if (!NSK_JNI_VERIFY(jni, (watchFieldClass = (jclass) jni->NewGlobalRef(klass)) != NULL)) {
 70                         NSK_DISPLAY0(" Agent :: Failed to get global reference for class.\n");
 71                         nsk_jvmti_agentFailed();
 72                     }
 73                     NSK_DISPLAY0(" Agent :: SetFieldAccessWatch.\n");
 74                 }
 75             }
 76             NSK_DISPLAY1(" Agent :: Leaving callbackClassPrepare for class = %s .\n", className);
 77         }
 78     }
 79 
 80     if (className != NULL) {
 81         if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char *)className))) {
 82             NSK_DISPLAY1(" Agent :: #error failed to Deallocate className = %s.", className);
 83             nsk_jvmti_agentFailed();
 84         }
 85     }
 86 
 87     if (generic != NULL) {
 88         if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char *)generic))) {
 89             NSK_DISPLAY1(" Agent :: #error failed to Deallocate class signature = %s.", generic);
 90             nsk_jvmti_agentFailed();
 91         }
 92     }
 93     return;
 94 }

 96 JNIEXPORT void JNICALL callbackFieldAccess(jvmtiEnv *jvmti_env,
 97         JNIEnv* jni,
 98         jthread thread,
 99         jmethodID method,
100         jlocation location,
101         jclass field_klass,
102         jobject object,
103         jfieldID field) {
104     char fileName[512];
105     char * className;
106     char * generic;
107 
108     className = NULL;
109     generic   = NULL;
110     if (redefineNumber != 0) {
111         return;
112     }
113     if (!NSK_JVMTI_VERIFY(jvmti_env->GetClassSignature(field_klass, &className, &generic))) {
114         NSK_DISPLAY0(" Agent :: Failed get class signature.\n");
115         nsk_jvmti_agentFailed();
116     } else {
117         if ((strcmp(className, CLASS_NAME) == 0)) {
118             jvmtiThreadInfo info;
119             nsk_jvmti_getFileName(redefineNumber, FILE_NAME, fileName,
120                     sizeof(fileName)/sizeof(char));
121             if (nsk_jvmti_redefineClass(jvmti_env, field_klass, fileName) == NSK_TRUE) {
122                 NSK_DISPLAY0(" Agent :: Successfully redefined.\n");
123                 redefineNumber++;
124             } else {
125                 NSK_DISPLAY0(" Agent :: Failed to redefine.\n");
126                 nsk_jvmti_agentFailed();
127             }
128             NSK_DISPLAY0(" Agent :: Before attempting thread suspend.\n");
129             if (!NSK_JVMTI_VERIFY(jvmti_env->GetThreadInfo(thread, &info))) {
130                 NSK_DISPLAY0(" Agent :: error getting thread info ");
131                 nsk_jvmti_agentFailed();
132             } else {
133                 NSK_DISPLAY1(" Agent :: Thread Name = %s .\n", info.name);
134             }
135             if (!NSK_JVMTI_VERIFY(jvmti_env->SuspendThread(thread))) {
136                 NSK_DISPLAY0(" Agent :: Failed to suspend thread.\n");
137                 nsk_jvmti_agentFailed();
138             }
139         }
140     }
141 
142     if (className != NULL) {
143         if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char *)className))) {
144             NSK_DISPLAY1(" Agent :: #error failed to Deallocate className = %s.", className);
145             nsk_jvmti_agentFailed();
146         }
147     }
148 
149     if (generic != NULL) {
150         if (!NSK_JVMTI_VERIFY(jvmti_env->Deallocate((unsigned char *)generic))) {
151             NSK_DISPLAY1(" Agent :: #error failed to Deallocate class signature = %s.", generic);
152             nsk_jvmti_agentFailed();
153         }
154     }
155 }

164 JNIEXPORT jint JNI_OnLoad_hs204t003(JavaVM *jvm, char *options, void *reserved) {
165     return JNI_VERSION_1_8;
166 }
167 #endif
168 jint  Agent_Initialize(JavaVM *vm, char *options, void *reserved) {
169     if (!NSK_VERIFY (JNI_OK == vm->GetEnv((void **)&jvmti, JVMTI_VERSION_1_1))) {
170         NSK_DISPLAY0("Agent :: Could not load JVMTI interface \n");
171         return JNI_ERR;
172     } else {
173         jvmtiCapabilities caps;
174         jvmtiEventCallbacks eventCallbacks;
175         memset(&caps, 0, sizeof(caps));
176         if (!nsk_jvmti_parseOptions(options)) {
177             NSK_DISPLAY0(" NSK Failed to parse..");
178             return JNI_ERR;
179         }
180         caps.can_redefine_classes             = 1;
181         caps.can_generate_field_access_events = 1;
182         caps.can_pop_frame                    = 1;
183         caps.can_suspend                      = 1;
184         if (!NSK_JVMTI_VERIFY(jvmti->AddCapabilities(&caps))) {
185             NSK_DISPLAY0(" Agent :: Failed add required capabilities\n.");
186             return JNI_ERR;
187         }
188         memset(&eventCallbacks, 0, sizeof(eventCallbacks));
189         eventCallbacks.ClassPrepare = callbackClassPrepare;
190         eventCallbacks.FieldAccess  = callbackFieldAccess;
191         if (!NSK_JVMTI_VERIFY(jvmti->SetEventCallbacks(&eventCallbacks, sizeof(eventCallbacks)))) {
192             NSK_DISPLAY0(" Agent :: Error occured while setting event call back \n");
193             return JNI_ERR;
194         }
195         nsk_jvmti_enableNotification(jvmti, JVMTI_EVENT_CLASS_PREPARE, NULL);
196     }
197     return JNI_OK;
198 }

200 JNIEXPORT jboolean JNICALL
201 Java_nsk_jvmti_scenarios_hotswap_HS204_hs204t003_hs204t003_popFrame(JNIEnv * jni,
202         jobject object,
203         jthread thread) {
204     jboolean retvalue;
205     jint state;
206     retvalue = JNI_FALSE;
207     if (!NSK_JVMTI_VERIFY(jvmti->GetThreadState(thread, &state))) {
208         NSK_DISPLAY0(" Agent :: Error getting thread state.\n");
209         nsk_jvmti_agentFailed();
210     } else {
211         if (state & JVMTI_THREAD_STATE_SUSPENDED) {
212             NSK_DISPLAY0(" Agent :: Thread state = JVMTI_THREAD_STATE_SUSPENDED.\n");
213             if (!NSK_JVMTI_VERIFY (jvmti->PopFrame(thread))) {
214                 NSK_DISPLAY0("#error Agent :: Jvmti failed to do popFrame.\n");
215                 nsk_jvmti_agentFailed();
216             } else {
217                 if (!NSK_JVMTI_VERIFY (jvmti->ResumeThread(thread))) {
218                     NSK_DISPLAY0(" Agent :: Error occured in resuming a thread.\n");
219                     nsk_jvmti_agentFailed();
220                 } else {
221                     jfieldID fieldId = jni->GetStaticFieldID(watchFieldClass, FIELDNAME, TYPE);
222                     if (!NSK_JNI_VERIFY(jni, fieldId != NULL)) {
223                         NSK_DISPLAY0(" Agent :: Failed to get FieldId before droping watchers.\n");
224                         nsk_jvmti_agentFailed();
225                     } else {
226                         if (!NSK_JVMTI_VERIFY (jvmti->ClearFieldAccessWatch(watchFieldClass, fieldId))) {
227                             NSK_DISPLAY0(" Agent :: failed to drop field watces.\n");
228                             nsk_jvmti_agentFailed();
229                         } else {
230                             NSK_DISPLAY0(" Agent :: Sucessfully droped watches.\n");
231                             retvalue = JNI_TRUE;
232                         }
233                     }
234                 }
235             }
236         } else {
237             NSK_DISPLAY0(" Agent :: Thread should be suspended to its pop frame.\n");
238         }
239     }
240     return retvalue;
241 }
