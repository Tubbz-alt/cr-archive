513     private ThreadReference threadByName(String name)
514                  throws JDITestRuntimeException {
515 
516         List         all = vm.allThreads();
517         ListIterator li  = all.listIterator();
518 
519         for (; li.hasNext(); ) {
520             ThreadReference thread = (ThreadReference) li.next();
521             if (thread.name().equals(name))
522                 return thread;
523         }
524         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
525     }
