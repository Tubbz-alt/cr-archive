 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ExceptionRequest.                                            <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.ExceptionRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: InvalidRequestStateException -                    <BR>
 47  *            if this request is currently enabled              <BR>
 48  *            or has been deleted.                              <BR>
 49  * <BR>
 50  * The test has three phases and works as follows.              <BR>
 51  * <BR>
 52  * In first phase,                                              <BR>
 53  * upon launching debuggee's VM which will be suspended,        <BR>
 54  * a debugger waits for the VMStartEvent within a predefined            <BR>
 55  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 56  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 57  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 58  * and waits for the event within the predefined time interval.         <BR>
 59  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 60  * Upon getting the ClassPrepareEvent,                                  <BR>
 61  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 62  * within debuggee's special methodForCommunication().                  <BR>
 63  * <BR>
 64  * In second phase to check the assertion,                              <BR>
 65  * the debugger and the debuggee perform the following.                 <BR>
 66  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 67  * - The debuggee creates new thread, thread1, and invokes              <BR>
 68  *   the methodForCommunication to be suspended and                     <BR>
 69  *   to inform the debugger with the event.                             <BR>
 70  * - Upon getting the BreakpointEvent, the debugger                     <BR>
 71  *   - creates an ExceptionRequest and enables it,                      <BR>
 72  *   - invokes the method addInstanceFilter() on the request expecting  <BR>
 73  *     to catch InvalidRequestStateException,                           <BR>
 74  *   - disables the request and invokes the method addInstanceFilter()  <BR>
 75  *     expecting to catch no InvalidRequestStateException,              <BR>
 76  *   - deletes the request and invokes the method addInstanceFilter()   <BR>
 77  *     once again expecting to catch InvalidRequestStateException.      <BR>
 78  * <BR>
 79  * In third phase, at the end of the test, the debuggee changes         <BR>
 80  * the value of the "instruction" which the debugger and debuggee       <BR>
 81  * use to inform each other of needed actions,  and both end.           <BR>
 82  * <BR>
 83  */
 84

160     private int runThis (String argv[], PrintStream out) {
161 
162         argsHandler     = new ArgumentHandler(argv);
163         logHandler      = new Log(out, argsHandler);
164         Binder binder   = new Binder(argsHandler, logHandler);
165 
166         waitTime        = argsHandler.getWaitTime() * 60000;
167 
168         try {
169             log2("launching a debuggee :");
170             log2("       " + debuggeeName);
171             if (argsHandler.verbose()) {
172                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
173             } else {
174                 debuggee = binder.bindToDebugee(debuggeeName);
175             }
176             if (debuggee == null) {
177                 log3("ERROR: no debuggee launched");
178                 return FAILED;
179             }
180             log2("debuggee launched");
181         } catch ( Exception e ) {
182             log3("ERROR: Exception : " + e);
183             log2("       test cancelled");
184             return FAILED;
185         }
186 
187         debuggee.redirectOutput(logHandler);
188 
189         vm = debuggee.VM();
190 
191         eventQueue = vm.eventQueue();
192         if (eventQueue == null) {
193             log3("ERROR: eventQueue == null : TEST ABORTED");
194             vm.exit(PASS_BASE);
195             return FAILED;
196         }
197 
198         log2("invocation of the method runTest()");
199         switch (runTest()) {
200 
201             case 0 :  log2("test phase has finished normally");
202                       log2("   waiting for the debuggee to finish ...");
203                       debuggee.waitFor();
204 
205                       log2("......getting the debuggee's exit status");
206                       int status = debuggee.getStatus();
207                       if (status != PASS_BASE) {
208                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
209                               status + " != PASS_BASE");
210                           testExitCode = FAILED;
211                       } else {
212                           log2("......debuggee returned expected exit status: " +
213                               status + " == PASS_BASE");
214                       }
215                       break;
216 
217             default : log3("ERROR: runTest() returned unexpected value");
218 
219             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
220                       log2("......forcing: vm.exit();");
221                       testExitCode = FAILED;
222                       try {
223                           vm.exit(PASS_BASE);
224                       } catch ( Exception e ) {
225                           log3("ERROR: Exception : " + e);
226                       }
227                       break;
228 
229             case 2 :  log3("test cancelled due to VMDisconnectedException");
230                       log2("......trying: vm.process().destroy();");
231                       testExitCode = FAILED;
232                       try {
233                           Process vmProcess = vm.process();
234                           if (vmProcess != null) {
235                               vmProcess.destroy();
236                           }
237                       } catch ( Exception e ) {
238                           log3("ERROR: Exception : " + e);
239                       }
240                       break;
241             }
242 
243         return testExitCode;
244     }

253     private int runTest() {
254 
255         try {
256             testRun();
257 
258             log2("waiting for VMDeathEvent");
259             getEventSet();
260             if (eventIterator.nextEvent() instanceof VMDeathEvent)
261                 return 0;
262 
263             log3("ERROR: last event is not the VMDeathEvent");
264             return 1;
265         } catch ( VMDisconnectedException e ) {
266             log3("ERROR: VMDisconnectedException : " + e);
267             return 2;
268         } catch ( Exception e ) {
269             log3("ERROR: Exception : " + e);
270             return 1;
271         }

275     private void testRun()
276                  throws JDITestRuntimeException, Exception {
277 
278         if ( !vm.canUseInstanceFilters() ) {
279             log2("......vm.canUseInstanceFilters == false :: test cancelled");
280             vm.exit(PASS_BASE);
281             return;
282         }
283 
284 
285         eventRManager = vm.eventRequestManager();
286 
287         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
288         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
289         cpRequest.addClassFilter(debuggeeName);
290 
291         cpRequest.enable();
292         vm.resume();
293         getEventSet();
294         cpRequest.disable();
295 
296         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
297         debuggeeClass = event.referenceType();
298 
299         if (!debuggeeClass.name().equals(debuggeeName))
300            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
301 
302         log2("      received: ClassPrepareEvent for debuggeeClass");
303 
304         String bPointMethod = "methodForCommunication";
305         String lineForComm  = "lineForComm";
306 
307         ThreadReference   mainThread = debuggee.threadByNameOrThrow("main");
308 
309         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
310                                              debuggeeClass,
311                                             bPointMethod, lineForComm, "zero");
312         bpRequest.enable();
313 
314     //------------------------------------------------------  testing section
315 
316         log1("     TESTING BEGINS");
317 
318         EventRequest eventRequest1 = null;
319 
320         ThreadReference thread1     = null;
321         String          thread1Name = "thread1";
322 
323         String property1 = "ExceptionRequest1";
324 
325         ReferenceType testClassReference = null;
326 
327         String         arrayName = "objTC";
328         ObjectReference instance = null;
329 
330 
331         for (int i = 0; ; i++) {
332 
333             vm.resume();
334             breakpointForCommunication();
335 
336             int instruction = ((IntegerValue)
337                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
338 
339             if (instruction == 0) {
340                 vm.resume();
341                 break;
342             }
343 
344             log1(":::::: case: # " + i);
345 
346             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
347 
348             switch (i) {
349 
350               case 0:
351                       testClassReference =
352                            (ReferenceType) vm.classesByName(testedClassName).get(0);
353 
354                       thread1 = (ThreadReference) debuggeeClass.getValue(
355                                                   debuggeeClass.fieldByName(thread1Name));
356 
357                       eventRequest1 = setting2ExceptionRequest (thread1,
358                                              testClassReference,
359                                              EventRequest.SUSPEND_NONE, "ExceptionRequest1");
360 
361                       instance = (ObjectReference)
362                                  ((ArrayReference) debuggeeClass.getValue(
363                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
364 
365                       eventRequest1.enable();
366 
367                      try {
368                          log2("......eventRequest1.addInstanceFilter(instance);");
369                          log2("        InvalidRequestStateException expected");
370                          ((ExceptionRequest) eventRequest1).addInstanceFilter(instance);
371                          log3("ERROR: no InvalidRequestStateException");
372                          testExitCode = FAILED;
373                      } catch ( InvalidRequestStateException e ) {
374                          log2("        InvalidRequestStateException");
375                      }
376 
377                      log2("......eventRequest1.disable();");
378                      eventRequest1.disable();
379                      try {
380                          log2("......eventRequest1.addInstanceFilter(instance);");
381                          log2("        no InvalidRequestStateException expected");
382                          ((ExceptionRequest) eventRequest1).addInstanceFilter(instance);
383                          log2("        no InvalidRequestStateException");
384                      } catch ( InvalidRequestStateException e ) {
385                          log3("ERROR: InvalidRequestStateException");
386                          testExitCode = FAILED;
387                      }
388 
389                      log2("......eventRManager.deleteEventRequest(eventRequest1);");
390                      eventRManager.deleteEventRequest(eventRequest1);
391                      try {
392                          log2("......eventRequest1.addInstanceFilter(instance);");
393                          log2("        InvalidRequestStateException expected");
394                          ((ExceptionRequest) eventRequest1).addInstanceFilter(instance);
395                          log3("ERROR: no InvalidRequestStateException");
396                          testExitCode = FAILED;
397                      } catch ( InvalidRequestStateException e ) {
398                          log2("        InvalidRequestStateException");
399                      }
400 
401                       break;
402 
403               default:
404                       throw new JDITestRuntimeException("** default case 2 **");
405             }
406             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
407         }
408         log1("    TESTING ENDS");
409         return;
410     }

412    /*
413     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
414     *                                             String, String, String)
415     *
416     * It sets up a breakpoint at given line number within a given method in a given class
417     * for a given thread.
418     *
419     * Return value: BreakpointRequest object  in case of success
420     *
421     * JDITestRuntimeException   in case of an Exception thrown within the method
422     */
423 
424     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
425                                                   ReferenceType testedClass,
426                                                   String methodName,
427                                                   String bpLine,
428                                                   String property)
429             throws JDITestRuntimeException {
430 
431         log2("......setting up a breakpoint:");
432         log2("       thread: " + thread + "; class: " + testedClass +
433                         "; method: " + methodName + "; line: " + bpLine);
434 
435         List              alllineLocations = null;
436         Location          lineLocation     = null;
437         BreakpointRequest breakpRequest    = null;
438 
439         try {
440             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
441 
442             alllineLocations = method.allLineLocations();
443 
444             int n =
445                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
446             if (n > alllineLocations.size()) {
447                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
448             } else {
449                 lineLocation = (Location) alllineLocations.get(n);
450                 try {
451                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
452                     breakpRequest.putProperty("number", property);
453                     breakpRequest.addThreadFilter(thread);
454                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
455                 } catch ( Exception e1 ) {
456                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
457                     breakpRequest    = null;
458                 }
459             }
460         } catch ( Exception e2 ) {
461             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
462             breakpRequest    = null;
463         }
464 
465         if (breakpRequest == null) {
466             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
467             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
468         }
469 
470         log2("      a breakpoint has been set up");
471         return breakpRequest;
472     }

475     private void getEventSet()
476                  throws JDITestRuntimeException {
477         try {
478 //            log2("       eventSet = eventQueue.remove(waitTime);");
479             eventSet = eventQueue.remove(waitTime);
480             if (eventSet == null) {
481                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
482             }
483 //            log2("       eventIterator = eventSet.eventIterator;");
484             eventIterator = eventSet.eventIterator();
485         } catch ( Exception e ) {
486             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
487         }
488     }

491     private void breakpointForCommunication()
492                  throws JDITestRuntimeException {
493 
494         log2("breakpointForCommunication");
495         getEventSet();
496 
497         if (eventIterator.nextEvent() instanceof BreakpointEvent)
498             return;
499 
500         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
501     }

505     private ExceptionRequest setting2ExceptionRequest ( ThreadReference thread,
506                                                        ReferenceType   testedClass,
507                                                        int             suspendPolicy,
508                                                        String          property       )
509             throws JDITestRuntimeException {
510         try {
511             log2("......setting up ExceptionRequest:");
512             log2("       thread: " + thread + "; class: " + testedClass + "; property: " + property);
513 
514             ExceptionRequest
515             excr = eventRManager.createExceptionRequest(null, true, true);
516             excr.putProperty("number", property);
517             if (thread != null)
518                 excr.addThreadFilter(thread);
519             excr.addClassFilter(testedClass);
520             excr.setSuspendPolicy(suspendPolicy);
521 
522             log2("      ExceptionRequest has been set up");
523             return excr;
524         } catch ( Exception e ) {
525             log3("ERROR: ATTENTION: Exception within settingExceptionRequest() : " + e);
526             log3("       ExceptionRequest HAS NOT BEEN SET UP");
527             throw new JDITestRuntimeException("** FAILURE to set up ExceptionRequest **");
528         }
