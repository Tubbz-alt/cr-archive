 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * ReferenceType.                                               <BR>
 40  *                                                              <BR>
 41  * The test checks up that results of the method                <BR>
 42  * <code>com.sun.jdi.ReferenceType.getValues()</code>           <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * Case for testing includes fields mirroring primitive type values     <BR>
 46  * declared in this type, a superclass, a superinterface, and           <BR>
 47  * an implemented interface.                                            <BR>
 48  * The test checks that Map returned by the method,                     <BR>
 49  * contains all the tested variables.                                   <BR>
 50  * <BR>
 51  * The test works as follows.                                           <BR>
 52  * Upon launching debuggee's VM which will be suspended,                <BR>
 53  * a debugger waits for the VMStartEvent within a predefined            <BR>
 54  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 55  * Upon getting the VMStartEvent, it makes the request                  <BR>
 56  * for debuggee's ClassPrepareEvent with SUSPEND_EVENT_THREAD,          <BR>
 57  * resumes the VM, and waits for the event within the predefined        <BR>
 58  * time interval. If no the ClassPrepareEvent received, the test is FAILED.<BR>
 59  * Upon getting the ClassPrepareEvent,                                  <BR>
 60  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD,       <BR>
 61  * the debugger resumes the debuggee and waits for the BreakpointEvent. <BR>
 62  * The debuggee prepares new check and invokes the methodForCommunication<BR>
 63  * to be suspended and to inform the debugger with the event.           <BR>
 64  * Upon getting the BreakpointEvent, the debugger performs the check.   <BR>
 65  * At the end, the debuggee changes the value of the "instruction"      <BR>
 66  * to inform the debugger of checks finished, and both end.             <BR>
 67  */

 98         return testExitCode;

149     private int runThis (String argv[], PrintStream out) {
150 
151         argsHandler     = new ArgumentHandler(argv);
152         logHandler      = new Log(out, argsHandler);
153         Binder binder   = new Binder(argsHandler, logHandler);
154 
155         waitTime        = argsHandler.getWaitTime() * 60000;
156 
157         try {
158             log2("launching a debuggee :");
159             log2("       " + debuggeeName);
160             if (argsHandler.verbose()) {
161                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
162             } else {
163                 debuggee = binder.bindToDebugee(debuggeeName);
164             }
165             if (debuggee == null) {
166                 log3("ERROR: no debuggee launched");
167                 return FAILED;
168             }
169             log2("debuggee launched");
170         } catch ( Exception e ) {
171             log3("ERROR: Exception : " + e);
172             log2("       test cancelled");
173             return FAILED;
174         }
175 
176         debuggee.redirectOutput(logHandler);
177 
178         vm = debuggee.VM();
179 
180         eventQueue = vm.eventQueue();
181         if (eventQueue == null) {
182             log3("ERROR: eventQueue == null : TEST ABORTED");
183             vm.exit(PASS_BASE);
184             return FAILED;
185         }
186 
187         log2("invocation of the method runTest()");
188         switch (runTest()) {
189 
190             case 0 :  log2("test phase has finished normally");
191                       log2("   waiting for the debuggee to finish ...");
192                       debuggee.waitFor();
193 
194                       log2("......getting the debuggee's exit status");
195                       int status = debuggee.getStatus();
196                       if (status != PASS_BASE) {
197                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
198                               status + " != PASS_BASE");
199                           testExitCode = FAILED;
200                       } else {
201                           log2("......debuggee returned expected exit status: " +
202                               status + " == PASS_BASE");
203                       }
204                       break;
205 
206             default : log3("ERROR: runTest() returned unexpected value");
207 
208             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
209                       log2("......forcing: vm.exit();");
210                       testExitCode = FAILED;
211                       try {
212                           vm.exit(PASS_BASE);
213                       } catch ( Exception e ) {
214                           log3("ERROR: Exception : e");
215                       }
216                       break;
217 
218             case 2 :  log3("test cancelled due to VMDisconnectedException");
219                       log2("......trying: vm.process().destroy();");
220                       testExitCode = FAILED;
221                       try {
222                           Process vmProcess = vm.process();
223                           if (vmProcess != null) {
224                               vmProcess.destroy();
225                           }
226                       } catch ( Exception e ) {
227                           log3("ERROR: Exception : e");
228                       }
229                       break;
230             }
231 
232         return testExitCode;
233     }

237     * Return value: 0 - normal end of the test
238     *               1 - ubnormal end of the test
239     *               2 - VMDisconnectedException while test phase
240     */

242     private int runTest() {
243 
244         try {
245             testRun();
246 
247             log2("waiting for VMDeathEvent");
248             getEventSet();
249             if (eventIterator.nextEvent() instanceof VMDeathEvent)
250                 return 0;
251 
252             log3("ERROR: last event is not the VMDeathEvent");
253             return 1;
254         } catch ( VMDisconnectedException e ) {
255             log3("ERROR: VMDisconnectedException : " + e);
256             return 2;
257         } catch ( Exception e ) {
258             log3("ERROR: Exception : " + e);
259             return 1;
260         }
261 
262     }

264     private void testRun()
265                  throws JDITestRuntimeException, Exception {
266 
267         eventRManager = vm.eventRequestManager();
268 
269         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
270         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
271         cpRequest.addClassFilter(debuggeeName);
272 
273         cpRequest.enable();
274         vm.resume();
275         getEventSet();
276         cpRequest.disable();
277 
278         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
279         debuggeeClass = event.referenceType();
280 
281         if (!debuggeeClass.name().equals(debuggeeName))
282            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
283 
284         log2("      received: ClassPrepareEvent for debuggeeClass");
285 
286         String bPointMethod = "methodForCommunication";
287         String lineForComm  = "lineForComm";
288         BreakpointRequest bpRequest;
289 
290         try {
291             bpRequest = settingBreakpoint(threadByName("main"),
292                                           debuggeeClass,
293                                           bPointMethod, lineForComm, "zero");
294         } catch ( Exception e ) {
295             throw e;
296         }
297         bpRequest.enable();
298 
299     //------------------------------------------------------  testing section
300 
301         log1("     TESTING BEGINS");
302 
303         for (int i = 0; ; i++) {
304 
305             vm.resume();
306             breakpointForCommunication();
307 
308             int instruction = ((IntegerValue)
309                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
310 
311             if (instruction == 0) {
312                 vm.resume();
313                 break;
314             }
315 
316             log1("  new check: # " + i);
317 
318             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
319 
320             log2("      name of tested class : " + testedClassName);
321             log2("      getting: List classes = vm.classByName(testedClassName);");
322             List classes = vm.classesByName(testedClassName);
323             if (classes.size() != 1) {
324                 log3("ERROR: classes.size() != 1 : " + classes.size());
325                 testExitCode = FAILED;
326                 break;
327             }
328             log2("      getting: ReferenceType testedClass = (ReferenceType) classes.get(0);");
329             ReferenceType testedClass = (ReferenceType) classes.get(0);
330 
331             log2("------case for testing: primitive values");
332 
333             String fieldNames[] = {
334                        "bl0",
335                        "bt0",
336                        "ch0",
337                        "db0",
338                        "fl0",
339                        "in0",
340                        "ln0",
341                        "sh0"
342             };
343 
344             Field [] fArray = {
345                 testedClass.fieldByName(fieldNames[0]),
346                 testedClass.fieldByName(fieldNames[1]),
347                 testedClass.fieldByName(fieldNames[2]),
348                 testedClass.fieldByName(fieldNames[3]),
349                 testedClass.fieldByName(fieldNames[4]),
350                 testedClass.fieldByName(fieldNames[5]),
351                 testedClass.fieldByName(fieldNames[6]),
352                 testedClass.fieldByName(fieldNames[7]),
353             };
354 
355             log2("      getting: List fields = testedClass.visibleFields();");
356             List<Field> fields = testedClass.visibleFields();
357 
358             log2("      getting: Map valuesMap = testedClass.getValues(fields);");
359             Map<Field, Value> valuesMap        = testedClass.getValues(fields);
360             Set<Field> keySet           = valuesMap.keySet();
361             Iterator<Field> setIterator = keySet.iterator();
362 
363             int flag        = 0;
364             int controlFlag = 0xFF;
365 
366             Field f1 = null;
367 
368             log2("      loop of testing the Map");
369 
370             for (int ifor = 0; setIterator.hasNext(); ifor++) {
371 
372                 f1 = setIterator.next();
373 
374                 int flag1 = 0;
375                 for (int ifor1 = 0; ifor1 < fieldNames.length; ifor1++) {
376 
377                     if (f1.equals(fArray[ifor1])) {
378                         if (flag1 == 0) {
379                             flag1++;
380                             flag |= 1 << ifor1;
381                             break;
382                         } else {
383                             log3("ERROR: duplication in the Map; ifor = " + ifor + "  ifor1 = " + ifor1);
384                             testExitCode = FAILED;
385                         }
386                     }
387                 }
388             }
389             if (flag != controlFlag) {
390                 log3("ERROR: returned Map object doesn't contain all tested fields");
391                 testExitCode = FAILED;
392             }
393             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
394         }
395         log1("    TESTING ENDS");
396         return;
397     }

399     private ThreadReference threadByName(String name)
400                  throws JDITestRuntimeException {
401 
402         List         all = vm.allThreads();
403         ListIterator li  = all.listIterator();
404 
405         for (; li.hasNext(); ) {
406             ThreadReference thread = (ThreadReference) li.next();
407             if (thread.name().equals(name))
408                 return thread;
409         }
410         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
411     }

413    /*
414     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
415     *                                             String, String, String)
416     *
417     * It sets up a breakpoint at given line number within a given method in a given class
418     * for a given thread.
419     *
420     * Return value: BreakpointRequest object  in case of success
421     *
422     * JDITestRuntimeException   in case of an Exception thrown within the method
423     */
424 
425     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
426                                                   ReferenceType testedClass,
427                                                   String methodName,
428                                                   String bpLine,
429                                                   String property)
430             throws JDITestRuntimeException {
431 
432         log2("......setting up a breakpoint:");
433         log2("       thread: " + thread + "; class: " + testedClass +
434                         "; method: " + methodName + "; line: " + bpLine);
435 
436         List              alllineLocations = null;
437         Location          lineLocation     = null;
438         BreakpointRequest breakpRequest    = null;
439 
440         try {
441             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
442 
443             alllineLocations = method.allLineLocations();
444 
445             int n =
446                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
447             if (n > alllineLocations.size()) {
448                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
449             } else {
450                 lineLocation = (Location) alllineLocations.get(n);
451                 try {
452                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
453                     breakpRequest.putProperty("number", property);
454                     breakpRequest.addThreadFilter(thread);
455                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
456                 } catch ( Exception e1 ) {
457                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
458                     breakpRequest    = null;
459                 }
460             }
461         } catch ( Exception e2 ) {
462             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
463             breakpRequest    = null;
464         }
465 
466         if (breakpRequest == null) {
467             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
468             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
469         }
470 
471         log2("      a breakpoint has been set up");
472         return breakpRequest;
473     }

476     private void getEventSet()
477                  throws JDITestRuntimeException {
478         try {
479 //            log2("       eventSet = eventQueue.remove(waitTime);");
480             eventSet = eventQueue.remove(waitTime);
481             if (eventSet == null) {
482                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
483             }
484 //            log2("       eventIterator = eventSet.eventIterator;");
485             eventIterator = eventSet.eventIterator();
486         } catch ( Exception e ) {
487             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
488         }
489     }

492     private void breakpointForCommunication()
493                  throws JDITestRuntimeException {
494 
495         log2("breakpointForCommunication");
496         getEventSet();
497 
498         if (eventIterator.nextEvent() instanceof BreakpointEvent)
499             return;
500 
501         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
502     }
