 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * WatchpointRequest.                                           <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.WatchpointRequest.addInstanceFilter()</code><BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on the following assertion:               <BR>
 46  *    Throws: UnsupportedOperationException -                   <BR>
 47  *       if the target virtual machine                          <BR>
 48  *       does not support this operation.                       <BR>
 49  * This request is ModificationWatchpointRequest.               <BR>
 50  * <BR>
 51  * The test has three phases and works as follows.              <BR>
 52  * <BR>
 53  * In first phase,                                                      <BR>
 54  * upon launching debuggee's VM which will be suspended,                <BR>
 55  * a debugger waits for the VMStartEvent within a predefined            <BR>
 56  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 57  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 58  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 59  * and waits for the event within the predefined time interval.         <BR>
 60  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 61  * Upon getting the ClassPrepareEvent,                                  <BR>
 62  * the debugger sets up the breakpoint with SUSPEND_EVENT_THREAD        <BR>
 63  * within debuggee's special methodForCommunication().                  <BR>
 64  * <BR>
 65  * In second phase to check the assertion,                              <BR>
 66  * the debugger and the debuggee perform the following.                 <BR>
 67  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 68  * - The debuggee creates new threads, thread1, and invokes             <BR>
 69  *   the methodForCommunication to be suspended and                     <BR>
 70  *   to inform the debugger with the event.                             <BR>
 71  * - Upon getting the BreakpointEvent, the debugger creates             <BR>
 72  *   a ModificationWatchpointRequest and if addInstanceFilter()         <BR>
 73  *   is not supported, invokes the method on the request expecting      <BR>
 74  *   to catch UnsupportedOperationException.                            <BR>
 75  * <BR>
 76  * In third phase, at the end of the test, the debuggee changes         <BR>
 77  * the value of the "instruction" which the debugger and debuggee       <BR>
 78  * use to inform each other of needed actions,  and both end.           <BR>
 79  * <BR>
 80  */

163     private int runThis (String argv[], PrintStream out) {
164 
165         argsHandler     = new ArgumentHandler(argv);
166         logHandler      = new Log(out, argsHandler);
167         Binder binder   = new Binder(argsHandler, logHandler);
168 
169         waitTime        = argsHandler.getWaitTime() * 60000;
170 
171         try {
172             log2("launching a debuggee :");
173             log2("       " + debuggeeName);
174             if (argsHandler.verbose()) {
175                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
176             } else {
177                 debuggee = binder.bindToDebugee(debuggeeName);
178             }
179             if (debuggee == null) {
180                 log3("ERROR: no debuggee launched");
181                 return FAILED;
182             }
183             log2("debuggee launched");
184         } catch ( Exception e ) {
185             log3("ERROR: Exception : " + e);
186             log2("       test cancelled");
187             return FAILED;
188         }
189 
190         debuggee.redirectOutput(logHandler);
191 
192         vm = debuggee.VM();
193 
194         eventQueue = vm.eventQueue();
195         if (eventQueue == null) {
196             log3("ERROR: eventQueue == null : TEST ABORTED");
197             vm.exit(PASS_BASE);
198             return FAILED;
199         }
200 
201         log2("invocation of the method runTest()");
202         switch (runTest()) {
203 
204             case 0 :  log2("test phase has finished normally");
205                       log2("   waiting for the debuggee to finish ...");
206                       debuggee.waitFor();
207 
208                       log2("......getting the debuggee's exit status");
209                       int status = debuggee.getStatus();
210                       if (status != PASS_BASE) {
211                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
212                               status + " != PASS_BASE");
213                           testExitCode = FAILED;
214                       } else {
215                           log2("......debuggee returned expected exit status: " +
216                               status + " == PASS_BASE");
217                       }
218                       break;
219 
220             default : log3("ERROR: runTest() returned unexpected value");
221 
222             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
223                       log2("......forcing: vm.exit();");
224                       testExitCode = FAILED;
225                       try {
226                           vm.exit(PASS_BASE);
227                       } catch ( Exception e ) {
228                           log3("ERROR: Exception : " + e);
229                       }
230                       break;
231 
232             case 2 :  log3("test cancelled due to VMDisconnectedException");
233                       log2("......trying: vm.process().destroy();");
234                       testExitCode = FAILED;
235                       try {
236                           Process vmProcess = vm.process();
237                           if (vmProcess != null) {
238                               vmProcess.destroy();
239                           }
240                       } catch ( Exception e ) {
241                           log3("ERROR: Exception : " + e);
242                       }
243                       break;
244             }
245 
246         return testExitCode;
247     }

251     * Return value: 0 - normal end of the test
252     *               1 - ubnormal end of the test
253     *               2 - VMDisconnectedException while test phase

256     private int runTest() {
257 
258         try {
259             testRun();
260 
261             log2("waiting for VMDeathEvent");
262             getEventSet();
263             if (eventIterator.nextEvent() instanceof VMDeathEvent)
264                 return 0;
265 
266             log3("ERROR: last event is not the VMDeathEvent");
267             return 1;
268         } catch ( VMDisconnectedException e ) {
269             log3("ERROR: VMDisconnectedException : " + e);
270             return 2;
271         } catch ( Exception e ) {
272             log3("ERROR: Exception : " + e);
273             return 1;
274         }

278     private void testRun()
279                  throws JDITestRuntimeException, Exception {
280 
281         eventRManager = vm.eventRequestManager();
282 
283         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
284         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
285         cpRequest.addClassFilter(debuggeeName);
286 
287         cpRequest.enable();
288         vm.resume();
289         getEventSet();
290         cpRequest.disable();
291 
292         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
293         debuggeeClass = event.referenceType();
294 
295         if (!debuggeeClass.name().equals(debuggeeName))
296            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
297 
298         log2("      received: ClassPrepareEvent for debuggeeClass");
299 
300         String bPointMethod = "methodForCommunication";
301         String lineForComm  = "lineForComm";
302 
303         ThreadReference   mainThread = threadByName("main");
304 
305         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
306                                              debuggeeClass,
307                                             bPointMethod, lineForComm, "zero");
308         bpRequest.enable();
309 
310     //------------------------------------------------------  testing section
311 
312         log1("     TESTING BEGINS");
313 
314         EventRequest eventRequest1 = null;
315 
316         ThreadReference thread1 = null;
317         ThreadReference thread2 = null;
318 
319         String thread1Name = "thread1";
320         String thread2Name = "thread2";
321 
322         String property1 = "ModificationWatchpointRequest1";
323 
324         String fieldName = "var1";
325 
326         ReferenceType testClassReference = null;
327 
328         String         arrayName = "objTC";
329         ObjectReference instance = null;
330 
331 
332         for (int i = 0; ; i++) {
333 
334             vm.resume();
335             breakpointForCommunication();
336 
337             int instruction = ((IntegerValue)
338                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
339 
340             if (instruction == 0) {
341                 vm.resume();
342                 break;
343             }
344 
345             log1(":::::: case: # " + i);
346 
347             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
348 
349             switch (i) {
350 
351               case 0:
352                       testClassReference =
353                            (ReferenceType) vm.classesByName(testedClassName).get(0);
354 
355                       thread1 = (ThreadReference) debuggeeClass.getValue(
356                                                   debuggeeClass.fieldByName(thread1Name));
357 
358                       eventRequest1 = setting2ModificationWatchpointRequest (thread1,
359                                              testClassReference, fieldName,
360                                              EventRequest.SUSPEND_NONE, property1);
361 
362                       instance = (ObjectReference)
363                                  ((ArrayReference) debuggeeClass.getValue(
364                                  debuggeeClass.fieldByName(arrayName)) ).getValue(0);
365 
366                       if ( vm.canUseInstanceFilters() ) {
367                           log2("......vm.canUseInstanceFilters == true :: test cancelled");
368                       } else {
369 
370                           try {
371                               log2("......eventRequest1.addInstanceFilter(instance);");
372                               log2("        UnsupportedOperationException  expected");
373                               ((WatchpointRequest) eventRequest1).addInstanceFilter(instance);
374                               log3("ERROR: no UnsupportedOperationException ");
375                               testExitCode = FAILED;
376                           } catch ( InvalidRequestStateException e ) {
377                               log2("        UnsupportedOperationException ");
378                           }
379                       }
380 
381                       break;
382 
383               default:
384                       throw new JDITestRuntimeException("** default case 2 **");
385             }
386 
387             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
388         }
389         log1("    TESTING ENDS");
390         return;
391     }

393     private ThreadReference threadByName(String name)
394                  throws JDITestRuntimeException {
395 
396         List         all = vm.allThreads();
397         ListIterator li  = all.listIterator();
398 
399         for (; li.hasNext(); ) {
400             ThreadReference thread = (ThreadReference) li.next();
401             if (thread.name().equals(name))
402                 return thread;
403         }
404         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
405     }

407    /*
408     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
409     *                                             String, String, String)
410     *
411     * It sets up a breakpoint at given line number within a given method in a given class
412     * for a given thread.
413     *
414     * Return value: BreakpointRequest object  in case of success
415     *
416     * JDITestRuntimeException   in case of an Exception thrown within the method
417     */
418 
419     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
420                                                   ReferenceType testedClass,
421                                                   String methodName,
422                                                   String bpLine,
423                                                   String property)
424             throws JDITestRuntimeException {
425 
426         log2("......setting up a breakpoint:");
427         log2("       thread: " + thread + "; class: " + testedClass +
428                         "; method: " + methodName + "; line: " + bpLine);
429 
430         List              alllineLocations = null;
431         Location          lineLocation     = null;
432         BreakpointRequest breakpRequest    = null;
433 
434         try {
435             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
436 
437             alllineLocations = method.allLineLocations();
438 
439             int n =
440                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
441             if (n > alllineLocations.size()) {
442                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
443             } else {
444                 lineLocation = (Location) alllineLocations.get(n);
445                 try {
446                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
447                     breakpRequest.putProperty("number", property);
448                     breakpRequest.addThreadFilter(thread);
449                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
450                 } catch ( Exception e1 ) {
451                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
452                     breakpRequest    = null;
453                 }
454             }
455         } catch ( Exception e2 ) {
456             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
457             breakpRequest    = null;
458         }
459 
460         if (breakpRequest == null) {
461             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
462             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
463         }
464 
465         log2("      a breakpoint has been set up");
466         return breakpRequest;
467     }

470     private void getEventSet()
471                  throws JDITestRuntimeException {
472         try {
473 //            log2("       eventSet = eventQueue.remove(waitTime);");
474             eventSet = eventQueue.remove(waitTime);
475             if (eventSet == null) {
476                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
477             }
478 //            log2("       eventIterator = eventSet.eventIterator;");
479             eventIterator = eventSet.eventIterator();
480         } catch ( Exception e ) {
481             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
482         }
483     }

486     private void breakpointForCommunication()
487                  throws JDITestRuntimeException {
488 
489         log2("breakpointForCommunication");
490         getEventSet();
491 
492         if (eventIterator.nextEvent() instanceof BreakpointEvent)
493             return;
494 
495         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
496     }

500     private ModificationWatchpointRequest setting2ModificationWatchpointRequest (
501                                                   ThreadReference thread,
502                                                   ReferenceType   testedClass,
503                                                   String          fieldName,
504                                                   int             suspendPolicy,
505                                                   String          property        )
506             throws JDITestRuntimeException {
507         try {
508             log2("......setting up ModificationWatchpointRequest:");
509             log2("       thread: " + thread + "; class: " + testedClass + "; fieldName: " + fieldName);
510             Field field = testedClass.fieldByName(fieldName);
511 
512             ModificationWatchpointRequest
513             awr = eventRManager.createModificationWatchpointRequest(field);
514             awr.putProperty("number", property);
515 
516             if (thread != null)
517                 awr.addThreadFilter(thread);
518             awr.setSuspendPolicy(suspendPolicy);
519 
520             log2("      ModificationWatchpointRequest has been set up");
521             return awr;
522         } catch ( Exception e ) {
523             log3("ERROR: ATTENTION: Exception within settingModificationWatchpointRequest() : " + e);
524             log3("       ModificationWatchpointRequest HAS NOT BEEN SET UP");
525             throw new JDITestRuntimeException("** FAILURE to set up ModificationWatchpointRequest **");
526         }
