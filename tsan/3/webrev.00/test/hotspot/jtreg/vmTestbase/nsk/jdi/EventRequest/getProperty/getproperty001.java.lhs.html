 37 /**
 38  * The test for the implementation of an object of the type     <BR>
 39  * EventRequest.                                                <BR>
 40  *                                                              <BR>
 41  * The test checks that results of the method                   <BR>
 42  * <code>com.sun.jdi.EventRequest.getProperty()</code>          <BR>
 43  * complies with its spec.                                      <BR>
 44  * <BR>
 45  * The test checks up on following assertions:                  <BR>
 46  * - a value of property is equal to one assigned with method   <BR>
 47  *   putProperty(key, value);
 48  * - the value of the property which is not assign with method  <BR>
 49  *   putProperty(key, value),  is the null.                     <BR>
 50  * The cases to check include Requests of all sub-types.        <BR>
 51  * <BR>
 52  * The test has three phases and works as follows.              <BR>
 53  * <BR>
 54  * In first phase,                                                      <BR>
 55  * upon launching debuggee's VM which will be suspended,                <BR>
 56  * a debugger waits for the VMStartEvent within a predefined            <BR>
 57  * time interval. If no the VMStartEvent received, the test is FAILED.  <BR>
 58  * Upon getting the VMStartEvent, it makes the request for debuggee's   <BR>
 59  * ClassPrepareEvent with SUSPEND_EVENT_THREAD, resumes the VM,         <BR>
 60  * and waits for the event within the predefined time interval.         <BR>
 61  * If no the ClassPrepareEvent received, the test is FAILED.            <BR>
 62  * Upon getting the ClassPrepareEvent,                                  <BR>
 63  * the debugger sets up the breakpoint with SUSPEND_ALL <BR>
 64  * within debuggee's special methodForCommunication().                  <BR>
 65  * <BR>
 66  * In second phase to check the assertion,                              <BR>
 67  * the debugger and the debuggee perform the following loop.            <BR>
 68  * - The debugger resumes the debuggee and waits for the BreakpointEvent.<BR>
 69  * - The debuggee prepares new check case (as if) and invokes           <BR>
 70  *   the methodForCommunication to be suspended and                     <BR>
 71  *   to inform the debugger with the event.                             <BR>
 72  * - Upon getting the BreakpointEvent,                                  <BR>
 73  *   the debugger performs the check case required.                     <BR>
 74  * <BR>
 75  * In third phase, at the end of the test, the debuggee changes         <BR>
 76  * the value of the "instruction" which the debugger and debuggee       <BR>
 77  * use to inform each other of needed actions,  and both end.           <BR>
 78  * <BR>
 79  */

164     private int runThis (String argv[], PrintStream out) {
165 
166         argsHandler     = new ArgumentHandler(argv);
167         logHandler      = new Log(out, argsHandler);
168         Binder binder   = new Binder(argsHandler, logHandler);
169 
170         waitTime        = argsHandler.getWaitTime() * 60000;
171 
172         try {
173             log2("launching a debuggee :");
174             log2("       " + debuggeeName);
175             if (argsHandler.verbose()) {
176                 debuggee = binder.bindToDebugee(debuggeeName + " -vbs");
177             } else {
178                 debuggee = binder.bindToDebugee(debuggeeName);
179             }
180             if (debuggee == null) {
181                 log3("ERROR: no debuggee launched");
182                 return FAILED;
183             }
184             log2("debuggee launched");
185         } catch ( Exception e ) {
186             log3("ERROR: Exception : " + e);
187             log2("       test cancelled");
188             return FAILED;
189         }
190 
191         debuggee.redirectOutput(logHandler);
192 
193         vm = debuggee.VM();
194 
195         eventQueue = vm.eventQueue();
196         if (eventQueue == null) {
197             log3("ERROR: eventQueue == null : TEST ABORTED");
198             vm.exit(PASS_BASE);
199             return FAILED;
200         }
201 
202         log2("invocation of the method runTest()");
203         switch (runTest()) {
204 
205             case 0 :  log2("test phase has finished normally");
206                       log2("   waiting for the debuggee to finish ...");
207                       debuggee.waitFor();
208 
209                       log2("......getting the debuggee's exit status");
210                       int status = debuggee.getStatus();
211                       if (status != PASS_BASE) {
212                           log3("ERROR: debuggee returned UNEXPECTED exit status: " +
213                               status + " != PASS_BASE");
214                           testExitCode = FAILED;
215                       } else {
216                           log2("......debuggee returned expected exit status: " +
217                               status + " == PASS_BASE");
218                       }
219                       break;
220 
221             default : log3("ERROR: runTest() returned unexpected value");
222 
223             case 1 :  log3("test phase has not finished normally: debuggee is still alive");
224                       log2("......forcing: vm.exit();");
225                       testExitCode = FAILED;
226                       try {
227                           vm.exit(PASS_BASE);
228                       } catch ( Exception e ) {
229                           log3("ERROR: Exception : " + e);
230                       }
231                       break;
232 
233             case 2 :  log3("test cancelled due to VMDisconnectedException");
234                       log2("......trying: vm.process().destroy();");
235                       testExitCode = FAILED;
236                       try {
237                           Process vmProcess = vm.process();
238                           if (vmProcess != null) {
239                               vmProcess.destroy();
240                           }
241                       } catch ( Exception e ) {
242                           log3("ERROR: Exception : " + e);
243                       }
244                       break;
245             }
246 
247         return testExitCode;
248     }

252     * Return value: 0 - normal end of the test
253     *               1 - ubnormal end of the test
254     *               2 - VMDisconnectedException while test phase
255     */

257     private int runTest() {
258 
259         try {
260             testRun();
261 
262             log2("waiting for VMDeathEvent");
263             getEventSet();
264             if (eventIterator.nextEvent() instanceof VMDeathEvent)
265                 return 0;
266 
267             log3("ERROR: last event is not the VMDeathEvent");
268             return 1;
269         } catch ( VMDisconnectedException e ) {
270             log3("ERROR: VMDisconnectedException : " + e);
271             return 2;
272         } catch ( Exception e ) {
273             log3("ERROR: Exception : " + e);
274             return 1;
275         }
276 
277     }

279     private void testRun()
280                  throws JDITestRuntimeException, Exception {
281 
282         eventRManager = vm.eventRequestManager();
283 
284         ClassPrepareRequest cpRequest = eventRManager.createClassPrepareRequest();
285         cpRequest.setSuspendPolicy( EventRequest.SUSPEND_EVENT_THREAD);
286         cpRequest.addClassFilter(debuggeeName);
287 
288         cpRequest.enable();
289         vm.resume();
290         getEventSet();
291         cpRequest.disable();
292 
293         ClassPrepareEvent event = (ClassPrepareEvent) eventIterator.next();
294         debuggeeClass = event.referenceType();
295 
296         if (!debuggeeClass.name().equals(debuggeeName))
297            throw new JDITestRuntimeException("** Unexpected ClassName for ClassPrepareEvent **");
298 
299         log2("      received: ClassPrepareEvent for debuggeeClass");
300 
301         String bPointMethod = "methodForCommunication";
302         String lineForComm  = "lineForComm";
303 
304         ThreadReference   mainThread = threadByName("main");
305 
306         BreakpointRequest bpRequest = settingBreakpoint(mainThread,
307                                              debuggeeClass,
308                                             bPointMethod, lineForComm, "zero");
309         bpRequest.enable();
310 
311     //------------------------------------------------------  testing section
312 
313 
314         EventRequest  eventRequest1 = null;
315 
316         String        fieldName1    = "var101";
317         String        fieldName2    = "excObj";
318 
319         ThreadReference thread1     = null;
320         String          threadName1 = "thread1";
321 
322         ReferenceType testClassReference = null;
323 
324         String propertyName1  = "name1";
325         String propertyName2  = "name2";
326         String propertyValue1 = "value1";
327 
328 
329         log1("     TESTING BEGINS");
330 
331         for (int i = 0; ; i++) {
332 
333             vm.resume();
334             breakpointForCommunication();
335 
336             int instruction = ((IntegerValue)
337                                (debuggeeClass.getValue(debuggeeClass.fieldByName("instruction")))).value();
338 
339             if (instruction == 0) {
340                 vm.resume();
341                 break;
342             }
343 
344             log1(":::::: case: # " + i);
345 
346             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ variable part
347 
348 
349             switch (i) {
350 
351               case 0:
352                      thread1 = threadByName(threadName1);
353 
354                      log2("......setting up StepRequest");
355                      eventRequest1 = eventRManager.createStepRequest
356                                      (thread1, StepRequest.STEP_MIN, StepRequest.STEP_INTO);
357                      break;
358 
359               case 1:
360                      testClassReference = (ReferenceType) vm.classesByName(testedClassName).get(0);
361 
362                      log2("......setting up AccessWatchpointRequest");
363                      eventRequest1 = eventRManager.createAccessWatchpointRequest
364                                     (testClassReference.fieldByName(fieldName1));
365                      break;
366 
367               case 2:
368                      log2(".....setting up ModificationWatchpointRequest");
369                      eventRequest1 = eventRManager.createModificationWatchpointRequest
370                                     (testClassReference.fieldByName(fieldName1));
371                      break;
372 
373               case 3:
374                      log2(".....setting up ClassPrepareRequest");
375                      eventRequest1 = eventRManager.createClassPrepareRequest();
376                      break;
377 
378               case 4:
379                      log2(".....setting up ClassUnloadRequest");
380                      eventRequest1 = eventRManager.createClassUnloadRequest();
381                      break;
382 
383               case 5:
384                      log2(".....setting up MethodEntryRequest");
385                      eventRequest1 = eventRManager.createMethodEntryRequest();
386                      break;
387 
388               case 6:
389                      log2(".....setting up MethodExitRequest");
390                      eventRequest1 = eventRManager.createMethodExitRequest();
391                      break;
392 
393               case 7:
394                      log2(".....setting up ThreadDeathRequest");
395                      eventRequest1 = eventRManager.createThreadDeathRequest();
396                      break;
397 
398               case 8:
399                      log2(".....setting up ThreadStartRequest");
400                      eventRequest1 = eventRManager.createThreadStartRequest();
401                      break;
402 
403               case 9:
404                      log2(".....setting up VMDeathRequest");
405                      eventRequest1 = eventRManager.createVMDeathRequest();
406                      break;
407 
408               case 10:
409                      log2(".....setting up ExceptionRequest");
410                      eventRequest1 = eventRManager.createExceptionRequest(
411                                      (ReferenceType)
412                                      (debuggeeClass.getValue(debuggeeClass.fieldByName(fieldName2))).type(),
413                                      true, true );
414                      break;
415 
416               case 11:
417                      log2(".....setting up BreakpointRequest");
418                      eventRequest1 = eventRManager.createBreakpointRequest(location);
419                      break;
420 
421 
422               default:
423                       throw new JDITestRuntimeException("** default case 2 **");
424             }
425 
426             log2("......eventRequest1.putProperty(propertyName1, propertyValue1);");
427             eventRequest1.putProperty(propertyName1, propertyValue1);
428 
429             log2("      checking up on property1");
430             if ( !eventRequest1.getProperty(propertyName1).equals(propertyValue1) ) {
431                 testExitCode = FAILED;
432                 log3("ERROR: property is not expected");
433             };
434 
435             log2("      checking up on property2");
436             if ( eventRequest1.getProperty(propertyName2)!= null ) {
437                 testExitCode = FAILED;
438                 log3("ERROR: property is not the null");
439             }
440 
441             eventRequest1.setEnabled(false);
442 
443             //~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
444         }
445         log1("    TESTING ENDS");
446         return;
447     }

449     private ThreadReference threadByName(String name)
450                  throws JDITestRuntimeException {
451 
452         List         all = vm.allThreads();
453         ListIterator li  = all.listIterator();
454 
455         for (; li.hasNext(); ) {
456             ThreadReference thread = (ThreadReference) li.next();
457             if (thread.name().equals(name))
458                 return thread;
459         }
460         throw new JDITestRuntimeException("** Thread IS NOT found ** : " + name);
461     }

463    /*
464     * private BreakpointRequest settingBreakpoint(ThreadReference, ReferenceType,
465     *                                             String, String, String)
466     *
467     * It sets up a breakpoint at given line number within a given method in a given class
468     * for a given thread.
469     *
470     * Return value: BreakpointRequest object  in case of success
471     *
472     * JDITestRuntimeException   in case of an Exception thrown within the method
473     */
474 
475     private BreakpointRequest settingBreakpoint ( ThreadReference thread,
476                                                   ReferenceType testedClass,
477                                                   String methodName,
478                                                   String bpLine,
479                                                   String property)
480             throws JDITestRuntimeException {
481 
482         log2("......setting up a breakpoint:");
483         log2("       thread: " + thread + "; class: " + testedClass +
484                         "; method: " + methodName + "; line: " + bpLine);
485 
486         List              alllineLocations = null;
487         Location          lineLocation     = null;
488         BreakpointRequest breakpRequest    = null;
489 
490         try {
491             Method  method  = (Method) testedClass.methodsByName(methodName).get(0);
492 
493             alllineLocations = method.allLineLocations();
494 
495             int n =
496                 ( (IntegerValue) testedClass.getValue(testedClass.fieldByName(bpLine) ) ).value();
497             if (n > alllineLocations.size()) {
498                 log3("ERROR:  TEST_ERROR_IN_settingBreakpoint(): number is out of bound of method's lines");
499             } else {
500                 lineLocation = (Location) alllineLocations.get(n);
501 
502 // this is only for this test to get Location object
503 location = lineLocation;
504 
505                 try {
506                     breakpRequest = eventRManager.createBreakpointRequest(lineLocation);
507                     breakpRequest.putProperty("number", property);
508                     breakpRequest.addThreadFilter(thread);
509                     breakpRequest.setSuspendPolicy( EventRequest.SUSPEND_ALL);
510                 } catch ( Exception e1 ) {
511                     log3("ERROR: inner Exception within settingBreakpoint() : " + e1);
512                     breakpRequest    = null;
513                 }
514             }
515         } catch ( Exception e2 ) {
516             log3("ERROR: ATTENTION:  outer Exception within settingBreakpoint() : " + e2);
517             breakpRequest    = null;
518         }
519 
520         if (breakpRequest == null) {
521             log2("      A BREAKPOINT HAS NOT BEEN SET UP");
522             throw new JDITestRuntimeException("**FAILURE to set up a breakpoint**");
523         }
524 
525         log2("      a breakpoint has been set up");
526         return breakpRequest;
527     }

530     private void getEventSet()
531                  throws JDITestRuntimeException {
532         try {
533 //            log2("       eventSet = eventQueue.remove(waitTime);");
534             eventSet = eventQueue.remove(waitTime);
535             if (eventSet == null) {
536                 throw new JDITestRuntimeException("** TIMEOUT while waiting for event **");
537             }
538 //            log2("       eventIterator = eventSet.eventIterator;");
539             eventIterator = eventSet.eventIterator();
540         } catch ( Exception e ) {
541             throw new JDITestRuntimeException("** EXCEPTION while waiting for event ** : " + e);
542         }
543     }

547                  throws JDITestRuntimeException {

553             return;
554 
555         throw new JDITestRuntimeException("** event IS NOT a breakpoint **");
