24 /*
25  * @test Verifies that negative pids are correctly rejected
26  * @bug 8229957
27  * @requires os.family != "windows"
28  * @library /test/lib
29  * @modules jdk.attach/com.sun.tools.attach
30  * @run main AttachNegativePidTest
31  */
32 
33 import java.io.IOException;
34 
35 import com.sun.tools.attach.VirtualMachine;
36 import com.sun.tools.attach.AttachNotSupportedException;
37 
38 import jdk.test.lib.apps.LingeredApp;
39 
40 public class AttachNegativePidTest {
41 
42     public static void main(String... args) throws Exception {
43         LingeredApp app = LingeredApp.startApp();
44         String strPID = Long.toString(-1 * app.getPid());
45         try {
46             VirtualMachine.attach(strPID);
47         } catch (AttachNotSupportedException anse) {
48             // Passed
49             return;
50         }
51         throw new RuntimeException("There is no expected AttachNotSupportedException for " + strPID);
52     }
53 
54 }
