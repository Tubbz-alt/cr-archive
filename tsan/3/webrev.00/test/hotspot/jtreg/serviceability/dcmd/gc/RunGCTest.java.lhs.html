24 import org.testng.annotations.Test;
25 import org.testng.Assert;
26 
27 import java.io.IOException;
28 import java.nio.file.Files;
29 import java.nio.file.Path;
30 import java.nio.file.Paths;
31 
32 import jdk.test.lib.process.OutputAnalyzer;
33 import jdk.test.lib.dcmd.CommandExecutor;
34 import jdk.test.lib.dcmd.JMXExecutor;
35 
36 /*
37  * @test
38  * @summary Test of diagnostic command GC.run
39  * @requires vm.gc != "Z"
40  * @library /test/lib
41  * @modules java.base/jdk.internal.misc
42  *          java.compiler
43  *          java.management
44  *          jdk.internal.jvmstat/sun.jvmstat.monitor
45  * @run testng/othervm -Xlog:gc=debug:RunGC.gclog -XX:-ExplicitGCInvokesConcurrent RunGCTest
46  */
47 public class RunGCTest {
48     public void run(CommandExecutor executor) {
49         executor.execute("GC.run");
50 
51         Path gcLogPath = Paths.get("RunGC.gclog").toAbsolutePath();
52         String gcLog = null;
53 
54         try {
55             gcLog = new String(Files.readAllBytes(gcLogPath));
56         } catch (IOException e) {
57             Assert.fail("Test error: Could not read GC log file: " + gcLogPath, e);
58         }
59 
60         OutputAnalyzer output = new OutputAnalyzer(gcLog, "");
61         output.shouldContain("Pause Full (Diagnostic Command)");
62     }
63 
64     @Test
65     public void jmx() {
66         run(new JMXExecutor());
67     }
68 }
