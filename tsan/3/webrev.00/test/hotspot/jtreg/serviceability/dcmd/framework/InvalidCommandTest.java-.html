24 import jdk.test.lib.process.OutputAnalyzer;
25 import jdk.test.lib.dcmd.CommandExecutor;
26 import jdk.test.lib.dcmd.PidJcmdExecutor;
27 import jdk.test.lib.dcmd.MainClassJcmdExecutor;
28 import jdk.test.lib.dcmd.FileJcmdExecutor;
29 import jdk.test.lib.dcmd.JMXExecutor;
30 import org.testng.annotations.Test;
31 
32 /*
33  * @test
34  * @summary Test of invalid diagnostic command (tests all DCMD executors)
35  * @library /test/lib
36  *          /vmTestbase
37  * @modules java.base/jdk.internal.misc
38  *          java.compiler
39  *          java.management
40  *          jdk.internal.jvmstat/sun.jvmstat.monitor
41  * @run testng/othervm -XX:+UsePerfData InvalidCommandTest
42  */
43 public class InvalidCommandTest {
44 
45     public void run(CommandExecutor executor) {
46         OutputAnalyzer output = executor.execute("asdf");
47         output.shouldContain("Unknown diagnostic command");
48     }
49 
50     @Test
51     public void pid() {
52         run(new PidJcmdExecutor());
53     }
54 
55     @Test
56     public void mainClass() {
57         TestProcessLauncher t = new TestProcessLauncher(Process.class.getName());
58         try {
59             t.launch();
60             run(new MainClassJcmdExecutor(Process.class.getName()));
61         } finally {
62             t.quit();
63         }
64     }
65 
66     @Test
67     public void file() {
68         run(new FileJcmdExecutor());
69     }
70 
71     @Test
72     public void jmx() {
73         run(new JMXExecutor());
74     }
75 
76     private static class Process extends TestJavaProcess {
77     }
78 }
