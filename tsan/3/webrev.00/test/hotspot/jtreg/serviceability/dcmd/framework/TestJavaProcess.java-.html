24 import nsk.share.jpda.*;
25 import nsk.share.jdi.*;
26 
27 /**
28  * A simple process that connects to a pipe and waits for command "quit" to
29  * be received.
30  *
31  * Usage: java TestJavaProcess -pipe.port <PIPE_PORT_NUMBER>
32  */
33 
34 public class TestJavaProcess {
35 
36     static final int PASSED = 0;
37     static final int FAILED = 2;
38 
39     public static void main(String argv[]) {
40 
41         log("Test Java process started!");
42 
43         ArgumentHandler argHandler = new ArgumentHandler(argv);
44         IOPipe pipe = argHandler.createDebugeeIOPipe();
45         pipe.println("ready");
46         log("Waiting for the quit command from the test ...");
47         String cmd = pipe.readln();
48         int exitCode = PASSED;
49         if (cmd.equals("quit")) {
50             log("'quit' received");
51         } else {
52             log("Invalid command received " + cmd);
53             exitCode = FAILED;
54         }
55         System.exit(exitCode);
56     }
57 
58     private static void log(String message) {
59         System.out.println(message);
60     }
61 }
