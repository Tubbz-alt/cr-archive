24 package process;
25 
26 import nsk.share.jdi.ArgumentHandler;
27 import nsk.share.jpda.IOPipe;
28 
29 /**
30  * A simple process that connects to a pipe and waits for command "quit" to
31  * be received.
32  *
33  * Usage: java TestJavaProcess -pipe.port <PIPE_PORT_NUMBER>
34  */
35 
36 public class TestJavaProcess {
37 
38     static final int PASSED = 0;
39     static final int FAILED = 2;
40 
41     public static void main(String argv[]) {
42 
43         log("Test Java process started!");
44 
45         ArgumentHandler argHandler = new ArgumentHandler(argv);
46         IOPipe pipe = argHandler.createDebugeeIOPipe();
47         pipe.println("ready");
48         log("Waiting for the quit command from the test ...");
49         String cmd = pipe.readln();
50         int exitCode = PASSED;
51         if (cmd.equals("quit")) {
52             log("'quit' received");
53         } else {
54             log("Invalid command received " + cmd);
55             exitCode = FAILED;
56         }
57         System.exit(exitCode);
58     }
59 
60     private static void log(String message) {
61         System.out.println(message);
62     }
63 }
