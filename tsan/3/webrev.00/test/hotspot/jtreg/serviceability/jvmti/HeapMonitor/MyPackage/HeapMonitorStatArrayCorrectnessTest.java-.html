25 package MyPackage;
26 
27 /**
28  * @test
29  * @build Frame HeapMonitor
30  * @summary Verifies the JVMTI Heap Monitor interval when allocating arrays.
31  * @compile HeapMonitorStatArrayCorrectnessTest.java
32  * @run main/othervm/native -agentlib:HeapMonitorTest MyPackage.HeapMonitorStatArrayCorrectnessTest
33  */
34 
35 public class HeapMonitorStatArrayCorrectnessTest {
36 
37   // Do 100000 iterations and expect maxIteration / multiplier samples.
38   private static final int maxIteration = 100000;
39   private static int array[];
40 
41   private static void allocate(int size) {
42     for (int j = 0; j < maxIteration; j++) {
43       array = new int[size];
44     }
45   }
46 
47   public static void main(String[] args) {
48     int sizes[] = {1000, 10000, 100000};
49 
50     for (int currentSize : sizes) {
51       System.out.println("Testing size " + currentSize);
52 
53       HeapMonitor.resetEventStorage();
54       if (!HeapMonitor.eventStorageIsEmpty()) {
55         throw new RuntimeException("Should not have any events stored yet.");
56       }
57 
58       HeapMonitor.enableSamplingEvents();
59 
60       // 111 is as good a number as any.
61       final int samplingMultiplier = 111;
62       HeapMonitor.setSamplingInterval(samplingMultiplier * currentSize);
63 
64       allocate(currentSize);
65 
66       HeapMonitor.disableSamplingEvents();
67 
68       // For simplifications, we ignore the array memory usage for array internals (with the array
69       // sizes requested, it should be a negligible oversight).
70       //
71       // That means that with maxIterations, the loop in the method allocate requests:
72       //    maxIterations * currentSize * 4 bytes (4 for integers)
73       //
74       // Via the enable sampling, the code requests a sample every samplingMultiplier * currentSize bytes.
75       //
76       // Therefore, the expected sample number is:
77       //   (maxIterations * currentSize * 4) / (samplingMultiplier * currentSize);
78       double expected = maxIteration;
79       expected *= 4;
80       expected /= samplingMultiplier;
81 
82       // 10% error ensures a sanity test without becoming flaky.
83       // Flakiness is due to the fact that this test is dependent on the sampling interval, which is a
84       // statistical geometric variable around the sampling interval. This means that the test could be
85       // unlucky and not achieve the mean average fast enough for the test case.
86       if (!HeapMonitor.statsHaveExpectedNumberSamples((int) expected, 10)) {
87         throw new RuntimeException("Statistics should show about " + expected + " samples.");
88       }
89     }
90   }
91 }
