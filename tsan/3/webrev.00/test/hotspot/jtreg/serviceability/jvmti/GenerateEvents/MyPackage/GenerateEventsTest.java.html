24 /**
25  * @test
26  * @bug 8222072
27  * @summary Send CompiledMethodLoad events only to the environment requested it with GenerateEvents
28  * @compile GenerateEventsTest.java
29  * @run main/othervm/native -agentlib:GenerateEvents1 -agentlib:GenerateEvents2 MyPackage.GenerateEventsTest
30  */
31 
32 package MyPackage;
33 
34 public class GenerateEventsTest {
35   static native void agent1GenerateEvents();
36   static native void agent2SetThread(Thread thread);
37   static native boolean agent1FailStatus();
38   static native boolean agent2FailStatus();
39 
40   public static void main(String[] args) {
41       agent2SetThread(Thread.currentThread());
42       agent1GenerateEvents(); // Re-generate CompiledMethodLoad events
43       if (agent1FailStatus()|| agent2FailStatus()) {
44          throw new RuntimeException("GenerateEventsTest failed!");
45       }
46       System.out.println("GenerateEventsTest passed!");
47   }
48 }
