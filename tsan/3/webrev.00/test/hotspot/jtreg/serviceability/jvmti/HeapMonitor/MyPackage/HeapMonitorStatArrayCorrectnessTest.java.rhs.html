 25 package MyPackage;
 26 
 27 /**
 28  * @test
 29  * @build Frame HeapMonitor
 30  * @summary Verifies the JVMTI Heap Monitor interval when allocating arrays.
 31  * @compile HeapMonitorStatArrayCorrectnessTest.java
 32  * @run main/othervm/native -agentlib:HeapMonitorTest MyPackage.HeapMonitorStatArrayCorrectnessTest
 33  */
 34 
 35 public class HeapMonitorStatArrayCorrectnessTest {
 36 
 37   private static final int maxCount = 10;
 38   // Do 100000 iterations and expect maxIteration / multiplier samples.
 39   private static final int maxIteration = 100000;
 40   private static int array[];
 41 
 42   private static void allocate(int size) {
 43     for (int j = 0; j < maxIteration; j++) {
 44       array = new int[size];
 45     }
 46   }
 47 
 48   public static void main(String[] args) {
 49     int sizes[] = {1000, 10000, 100000};
 50     double expected = 0;
 51     int count = 0;
 52 
 53     for (int currentSize : sizes) {
 54       System.out.println("Testing size " + currentSize);
 55 
 56       HeapMonitor.resetEventStorage();
 57       if (!HeapMonitor.eventStorageIsEmpty()) {
 58         throw new RuntimeException("Should not have any events stored yet.");
 59       }
 60 
 61       for (count = 1; count < maxCount; count++) {
 62         // 111 is as good a number as any.
 63         final int samplingMultiplier = 111;
 64         HeapMonitor.setSamplingInterval(samplingMultiplier * currentSize);
 65 
 66         HeapMonitor.enableSamplingEvents();


 67 
 68         allocate(currentSize);
 69 
 70         HeapMonitor.disableSamplingEvents();
 71 
 72         // For simplifications, we ignore the array memory usage for array internals (with the array
 73         // sizes requested, it should be a negligible oversight).
 74         //
 75         // That means that with maxIterations, the loop in the method allocate requests:
 76         //    maxIterations * currentSize * 4 bytes (4 for integers)
 77         //
 78         // Via the enable sampling, the code requests a sample every samplingMultiplier * currentSize bytes.
 79         //
 80         // Therefore, the expected sample number is:
 81         //   count * (maxIterations * currentSize * 4) / (samplingMultiplier * currentSize);
 82         //   (count because we can do this multiple times in order to converge).
 83         expected = maxIteration * count;
 84         expected *= 4;
 85         expected /= samplingMultiplier;
 86 
 87         // 10% error ensures a sanity test without becoming flaky.
 88         // Flakiness is due to the fact that this test is dependent on the sampling interval, which is a
 89         // statistical geometric variable around the sampling interval. This means that the test could be
 90         // unlucky and not achieve the mean average fast enough for the test case.
 91         if (HeapMonitor.statsHaveExpectedNumberSamples((int) expected, 10)) {
 92           break;
 93         }
 94       }
 95 
 96       // If we failed maxCount times, throw the exception.
 97       if (count == maxCount) {
 98         throw new RuntimeException("Statistics should show about " + expected + " samples; "
 99             + " but have " + HeapMonitor.sampledEvents() + " instead for the size "
100             + currentSize);
101       }
102     }
103   }
104 }
