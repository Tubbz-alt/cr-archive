30  * @summary Verifies the JVMTI Heap Monitor interval when allocating arrays.
31  * @compile HeapMonitorArrayAllSampledTest.java
32  * @run main/othervm/native -agentlib:HeapMonitorTest MyPackage.HeapMonitorArrayAllSampledTest
33  */
34 
35 public class HeapMonitorArrayAllSampledTest {
36 
37   // Do 1000 iterations and expect maxIteration samples.
38   private static final int maxIteration = 1000;
39   private static int array[];
40 
41   private static void allocate(int size) {
42     for (int j = 0; j < maxIteration; j++) {
43       array = new int[size];
44     }
45   }
46 
47   public static void main(String[] args) {
48     int sizes[] = {1000, 10000, 100000, 1000000};
49 
50     HeapMonitor.setSamplingInterval(0);
51     HeapMonitor.enableSamplingEvents();
52 
53     for (int currentSize : sizes) {
54       System.out.println("Testing size " + currentSize);
55 
56       HeapMonitor.resetEventStorage();
57       allocate(currentSize);
58 
59       // 10% error ensures a sanity test without becoming flaky.
60       // Flakiness is due to the fact that this test is dependent on the sampling interval, which is a
61       // statistical geometric variable around the sampling interval. This means that the test could be
62       // unlucky and not achieve the mean average fast enough for the test case.
63       if (!HeapMonitor.statsHaveExpectedNumberSamples(maxIteration, 10)) {
64         throw new RuntimeException("Statistics should show about " + maxIteration + " samples.");
65       }
66     }
67   }
68 }
