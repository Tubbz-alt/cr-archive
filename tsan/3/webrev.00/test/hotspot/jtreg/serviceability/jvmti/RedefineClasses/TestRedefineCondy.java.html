24 /*
25  * @test
26  * @bug 8228596
27  * @summary Test redefining a class with a condy in its constant pool
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  * @modules java.compiler
31  *          java.instrument
32  *          jdk.jartool/sun.tools.jar
33  * @compile RedefineCondy.jasm
34  * @run main RedefineClassHelper
35  * @run main/othervm -javaagent:redefineagent.jar TestRedefineCondy
36  */
37 
38 import jdk.test.lib.compiler.InMemoryJavaCompiler;
39 
40 // Test redefining a class that has a constant dynamic in its constant pool
41 // to a class that does not have a constant dynamic in its constant pool.
42 public class TestRedefineCondy {
43 
44     static final String DEST = System.getProperty("test.classes");
45     static String newClass =
46         "public class RedefineCondy { " +
47         "public RedefineCondy(java.lang.invoke.MethodHandles.Lookup l, java.lang.String s, java.lang.Class c) { } " +
48     "public static void main(String argv[]) { } } ";
49 
50     public static void main(String[] args) throws Exception  {
51         Class<?> classWithCondy = Class.forName("RedefineCondy");
52 
53         try {
54             byte[] classBytes = InMemoryJavaCompiler.compile("RedefineCondy", newClass);
55             RedefineClassHelper.redefineClass(classWithCondy, classBytes);
56         } catch (Exception e) {
57             throw new RuntimeException("Unexpected exception: " + e.getMessage());
58         }
59     }
60 }
