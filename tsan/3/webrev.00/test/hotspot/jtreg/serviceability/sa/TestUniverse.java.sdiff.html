 59     private static void testClhsdbForUniverse(long lingeredAppPid,
 60                                               String gc) throws Exception {
 61 
 62         ClhsdbLauncher launcher = new ClhsdbLauncher();
 63         List<String> cmds = List.of("universe");
 64         Map<String, List<String>> expStrMap = new HashMap<>();
 65         List<String> expStrings = new ArrayList<String>();
 66         expStrings.add("Heap Parameters");
 67 
 68         if (gc.contains("UseZGC")) {
 69             expStrings.add("ZHeap");
 70         }
 71         if (gc.contains("G1GC")) {
 72             expStrings.add("garbage-first heap");
 73             expStrings.add("region size");
 74             expStrings.add("G1 Young Generation:");
 75             expStrings.add("regions  =");
 76         }
 77         if (gc.contains("UseConcMarkSweepGC")) {
 78             expStrings.add("Gen 1: concurrent mark-sweep generation");
 79         }
 80         if (gc.contains("UseSerialGC")) {
 81             expStrings.add("Gen 1:   old");
 82         }
 83         if (gc.contains("UseParallelGC")) {

 84             expStrings.add("ParallelScavengeHeap");
 85             expStrings.add("PSYoungGen");
 86             expStrings.add("eden");
 87         }
 88         if (gc.contains("UseEpsilonGC")) {








 89             expStrings.add("Epsilon heap");
 90             expStrings.add("reserved");
 91             expStrings.add("committed");
 92             expStrings.add("used");









 93         }

 94         expStrMap.put("universe", expStrings);
 95         launcher.run(lingeredAppPid, cmds, expStrMap, null);
 96     }

 98     public static void test(String gc) throws Exception {
 99         LingeredApp app = null;
100         try {
101             List<String> vmArgs = new ArrayList<String>();
102             vmArgs.add("-XX:+UnlockExperimentalVMOptions"); // unlock experimental GCs
103             vmArgs.add(gc);
104             app = LingeredApp.startApp(vmArgs);
105             System.out.println ("Started LingeredApp with the GC option " + gc +
106                                 " and pid " + app.getPid());
107             testClhsdbForUniverse(app.getPid(), gc);
108         } finally {
109             LingeredApp.stopApp(app);
110         }
111     }

113     public static void main (String... args) throws Exception {
114         System.out.println("Starting TestUniverse test");
115         try {
116             test("-XX:+UseG1GC");
117             test("-XX:+UseParallelGC");
118             test("-XX:+UseSerialGC");
119             if (!Compiler.isGraalEnabled()) { // Graal does not support all GCs
120                 test("-XX:+UseConcMarkSweepGC");
121                 if (args[0].equals("withZ")) {
122                     test("-XX:+UseZGC");
123                 }
124                 test("-XX:+UseEpsilonGC");
125             }

127             throw se;
128         } catch (Exception e) {
129             System.out.println(e.getMessage());

 48     private static void testClhsdbForUniverse(long lingeredAppPid, GC gc) throws Exception {


 49         ClhsdbLauncher launcher = new ClhsdbLauncher();
 50         List<String> cmds = List.of("universe");
 51         Map<String, List<String>> expStrMap = new HashMap<>();
 52         List<String> expStrings = new ArrayList<String>();
 53         expStrings.add("Heap Parameters");
 54 
 55         switch (gc) {
 56         case Serial:











 57             expStrings.add("Gen 1:   old");
 58             break;
 59 
 60         case Parallel:
 61             expStrings.add("ParallelScavengeHeap");
 62             expStrings.add("PSYoungGen");
 63             expStrings.add("eden");
 64             break;
 65 
 66         case G1:
 67             expStrings.add("garbage-first heap");
 68             expStrings.add("region size");
 69             expStrings.add("G1 Young Generation:");
 70             expStrings.add("regions  =");
 71             break;
 72 
 73         case Epsilon:
 74             expStrings.add("Epsilon heap");
 75             expStrings.add("reserved");
 76             expStrings.add("committed");
 77             expStrings.add("used");
 78             break;
 79 
 80         case Z:
 81             expStrings.add("ZHeap");
 82             break;
 83 
 84         case Shenandoah:
 85             expStrings.add("Shenandoah heap");
 86             break;
 87         }
 88 
 89         expStrMap.put("universe", expStrings);
 90         launcher.run(lingeredAppPid, cmds, expStrMap, null);
 91     }
 92 
 93     private static void test(GC gc) throws Exception {
 94         LingeredApp app = null;
 95         try {
 96             app = LingeredApp.startApp("-XX:+UnlockExperimentalVMOptions", "-XX:+Use" + gc + "GC");
 97             System.out.println ("Started LingeredApp with " + gc + "GC and pid " + app.getPid());




 98             testClhsdbForUniverse(app.getPid(), gc);
 99         } finally {
100             LingeredApp.stopApp(app);
101         }
102     }

104     private static boolean isSelectedAndSupported(GC gc) {
105         if (!gc.isSelected()) {
106             // Not selected
107             return false;
108         }
109 
110         if (Compiler.isGraalEnabled()) {
111             if (gc == GC.Epsilon || gc == GC.Z || gc == GC.Shenandoah) {
112                 // Not supported
113                 System.out.println ("Skipped testing of " + gc + "GC, not supported by Graal");
114                 return false;
115             }
116         }
117 
118         // Selected and supported
119         return true;
120     }

122     public static void main (String... args) throws Exception {
123         System.out.println("Starting TestUniverse");
124         try {
125             for (GC gc: GC.values()) {
126                 if (isSelectedAndSupported(gc)) {
127                     test(gc);




128                 }

129             }
130         } catch (SkippedException se) {
131             throw se;
132         } catch (Exception e) {
133             System.out.println(e.getMessage());
134             e.printStackTrace();
135             throw new Error("Test failed with " + e);
136         }
137     }
