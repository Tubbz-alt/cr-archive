 48  *          jdk.hotspot.agent/sun.jvm.hotspot.oops

 53 public class TestInstanceKlassSizeForInterface {
 54 
 55     private static LingeredAppWithInterface theApp = null;
 56 
 57     private static void SAInstanceKlassSize(int lingeredAppPid,
 58                                             String[] instanceKlassNames) {
 59 
 60         HotSpotAgent agent = new HotSpotAgent();
 61         try {
 62             agent.attach(lingeredAppPid);
 63         }
 64         catch (DebuggerException e) {
 65             System.out.println(e.getMessage());
 66             System.err.println("Unable to connect to process ID: " + lingeredAppPid);
 67 
 68             agent.detach();
 69             e.printStackTrace();
 70         }
 71 
 72         for (String instanceKlassName : instanceKlassNames) {
 73             InstanceKlass iKlass = SystemDictionaryHelper.findInstanceKlass(
 74                                        instanceKlassName);
 75             Asserts.assertNotNull(iKlass,
 76                 String.format("Unable to find instance klass for %s", instanceKlassName));
 77             System.out.println("SA: The size of " + instanceKlassName +
 78                                " is " + iKlass.getSize());
 79         }
 80         agent.detach();
 81     }
 82 
 83     private static String getJcmdInstanceKlassSize(OutputAnalyzer output,
 84                                                    String instanceKlassName) {
 85         for (String s : output.asLines()) {
 86             if (s.contains(instanceKlassName)) {
 87                 String tokens[];
 88                 System.out.println(s);
 89                 tokens = s.split("\\s+");
 90                 return tokens[3];
 91             }
 92         }
 93         return null;
 94     }
 95 
 96     private static void createAnotherToAttach(
 97                             String[] instanceKlassNames,
 98                             int lingeredAppPid) throws Exception {
 99 
100         String[] toolArgs = {
101             "--add-modules=jdk.hotspot.agent",
102             "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot=ALL-UNNAMED",
103             "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot.utilities=ALL-UNNAMED",
104             "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot.oops=ALL-UNNAMED",
105             "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot.debugger=ALL-UNNAMED",
106             "TestInstanceKlassSizeForInterface",
107             Integer.toString(lingeredAppPid)
108         };
109 
110         // Start a new process to attach to the LingeredApp process
111         ProcessBuilder processBuilder = ProcessTools
112                   .createJavaProcessBuilder(toolArgs);
113         OutputAnalyzer SAOutput = ProcessTools.executeProcess(processBuilder);
114         SAOutput.shouldHaveExitValue(0);
115         System.out.println(SAOutput.getOutput());
116 
117         // Run jcmd on the LingeredApp process
118         ProcessBuilder pb = new ProcessBuilder();
119         pb.command(new String[] {
120                           JDKToolFinder.getJDKTool("jcmd"),
121                           Long.toString(lingeredAppPid),
122                           "GC.class_stats",
123                           "VTab,ITab,OopMap,KlassBytes"
124                       }
125                   );
126 
127         OutputAnalyzer jcmdOutput = new OutputAnalyzer(pb.start());
128         System.out.println(jcmdOutput.getOutput());
129 
130         // Match the sizes from both the output streams
131         for (String instanceKlassName : instanceKlassNames) {
132             System.out.println ("Trying to match for " + instanceKlassName);
133             String jcmdInstanceKlassSize = getJcmdInstanceKlassSize(
134                                                       jcmdOutput,
135                                                       instanceKlassName);
136             Asserts.assertNotNull(jcmdInstanceKlassSize,
137                 "Could not get the instance klass size from the jcmd output");
138             for (String s : SAOutput.asLines()) {
139                 if (s.contains(instanceKlassName)) {
140                    Asserts.assertTrue(
141                       s.contains(jcmdInstanceKlassSize),
142                       "The size computed by SA for " +
143                       instanceKlassName + " does not match.");
144                 }
145             }
146         }
147     }
148 
149     public static void main (String... args) throws Exception {
150         String[] instanceKlassNames = new String[] {
151                                           "Language",
152                                           "ParselTongue",
153                                           "LingeredAppWithInterface$1"
154                                       };
155 
156         if (args == null || args.length == 0) {
157             try {
158                 List<String> vmArgs = new ArrayList<String>();
159                 vmArgs.addAll(Utils.getVmOptions());
160 
161                 theApp = new LingeredAppWithInterface();
162                 LingeredApp.startApp(vmArgs, theApp);
163                 createAnotherToAttach(instanceKlassNames,
164                                       (int)theApp.getPid());
165             } finally {
166                 LingeredApp.stopApp(theApp);
167             }
168         } else {
169             SAInstanceKlassSize(Integer.parseInt(args[0]), instanceKlassNames);
170         }
171     }
