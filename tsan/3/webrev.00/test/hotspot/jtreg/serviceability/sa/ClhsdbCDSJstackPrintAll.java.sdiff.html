 44     public static void main(String[] args) throws Exception {
 45         System.out.println("Starting ClhsdbCDSJstackPrintAll test");
 46         String sharedArchiveName = "ArchiveForClhsdbJstackPrintAll.jsa";
 47         LingeredApp theApp = null;
 48 
 49         try {
 50             CDSOptions opts = (new CDSOptions()).setArchiveName(sharedArchiveName);
 51             CDSTestUtils.createArchiveAndCheck(opts);
 52 
 53             ClhsdbLauncher test = new ClhsdbLauncher();
 54             List<String> vmArgs = Arrays.asList(
 55                 "-XX:+UnlockDiagnosticVMOptions",
 56                 "-XX:SharedArchiveFile=" + sharedArchiveName,
 57                 "-Xshare:auto");
 58             theApp = LingeredApp.startApp(vmArgs);
 59             System.out.println("Started LingeredApp with pid " + theApp.getPid());
 60 
 61             // Ensure that UseSharedSpaces is turned on.
 62             List<String> cmds = List.of("flags UseSharedSpaces");
 63 
 64             String useSharedSpacesOutput = test.run(theApp.getPid(), cmds,
 65                                                     null, null);
 66

 68                 LingeredApp.stopApp(theApp);
 69                 // Attach permission issues.

 71             }
 72

 74                 // CDS archive is not mapped. Skip the rest of the test.
