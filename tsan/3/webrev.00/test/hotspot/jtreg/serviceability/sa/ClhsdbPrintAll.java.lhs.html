24 import java.util.HashMap;
25 import java.util.List;
26 import java.util.Map;
27 import jdk.test.lib.apps.LingeredApp;
28 import jtreg.SkippedException;
29 
30 /**
31  * @test
32  * @bug 8175384
33  * @summary Test clhsdb 'printall' command
34  * @requires vm.hasSA
35  * @library /test/lib
36  * @run main/othervm/timeout=2400 -Xmx1g ClhsdbPrintAll
37  */
38 
39 public class ClhsdbPrintAll {
40 
41     public static void main(String[] args) throws Exception {
42         System.out.println("Starting ClhsdbPrintAll test");
43 
44         LingeredAppWithEnum theApp = null;
45         try {
46             ClhsdbLauncher test = new ClhsdbLauncher();
47 
48             theApp = new LingeredAppWithEnum();
49             LingeredApp.startApp(null, theApp);
50             System.out.println("Started LingeredAppWithEnum with pid " + theApp.getPid());
51 
52             List<String> cmds = List.of("printall");
53 
54             Map<String, List<String>> expStrMap = new HashMap<>();
55             Map<String, List<String>> unExpStrMap = new HashMap<>();
56             expStrMap.put("printall", List.of(
57                 "aload_0",
58                 "Constant Pool of",
59                 "public static void main\\(java.lang.String\\[\\]\\)",
60                 "Bytecode",
61                 "\\[enum\\] class Song \\[signature Ljava/lang/Enum\\<LSong;\\>;\\]",
62                 "Method java.lang.Object clone\\(\\)",
63                 "public static Song\\[\\] values\\(\\)",
64                 "invokevirtual",
65                 "checkcast",
66                 "Field Song HAVANA",
67                 "Exception Table",
68                 "invokedynamic"));
69             unExpStrMap.put("printall", List.of(
70                 "cannot be cast to"));
71             test.run(theApp.getPid(), cmds, expStrMap, unExpStrMap);
72         } catch (SkippedException se) {
73             throw se;
74         } catch (Exception ex) {
75             throw new RuntimeException("Test ERROR " + ex, ex);
76         } finally {
77             LingeredApp.stopApp(theApp);
78         }
79         System.out.println("Test PASSED");
80     }
81 }
