 56 public class TestInstanceKlassSize {
 57 
 58     private static String getJcmdInstanceKlassSize(OutputAnalyzer output,
 59                                                    String instanceKlassName) {
 60         for (String s : output.asLines()) {
 61             if (s.contains(instanceKlassName)) {
 62                 String tokens[];
 63                 System.out.println(s);
 64                 tokens = s.split("\\s+");
 65                 return tokens[3];
 66             }
 67         }
 68         return null;
 69     }
 70 
 71     private static OutputAnalyzer jcmd(Long pid,
 72                                        String... toolArgs) throws Exception {
 73         ProcessBuilder processBuilder = new ProcessBuilder();
 74         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("jcmd");
 75         launcher.addToolArg(Long.toString(pid));
 76         if (toolArgs != null) {
 77             for (String toolArg : toolArgs) {
 78                 launcher.addToolArg(toolArg);
 79             }
 80         }
 81 
 82         processBuilder.command(launcher.getCommand());
 83         System.out.println(
 84             processBuilder.command().stream().collect(Collectors.joining(" ")));
 85         return ProcessTools.executeProcess(processBuilder);
 86     }



 87 
 88     private static void startMeWithArgs() throws Exception {
 89 
 90         LingeredApp app = null;
 91         OutputAnalyzer output = null;
 92         try {
 93             List<String> vmArgs = new ArrayList<String>();
 94             vmArgs.add("-XX:+UsePerfData");
 95             vmArgs.addAll(Utils.getVmOptions());
 96             app = LingeredApp.startApp(vmArgs);
 97             System.out.println ("Started LingeredApp with pid " + app.getPid());
 98         } catch (Exception ex) {
 99             ex.printStackTrace();
100             throw new RuntimeException(ex);
101         }
102         try {
103             String[] instanceKlassNames = new String[] {
104                                               " java.lang.Object",
105                                               " java.util.Vector",
106                                               " java.lang.String",
107                                               " java.lang.Thread",
108                                               " java.lang.Byte",
109                                           };
110             String[] toolArgs = {
111                 "--add-modules=jdk.hotspot.agent",
112                 "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot=ALL-UNNAMED",
113                 "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot.utilities=ALL-UNNAMED",
114                 "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot.oops=ALL-UNNAMED",
115                 "--add-exports=jdk.hotspot.agent/sun.jvm.hotspot.debugger=ALL-UNNAMED",



116                 "TestInstanceKlassSize",
117                 Long.toString(app.getPid())
118             };
119 
120             OutputAnalyzer jcmdOutput = jcmd(
121                            app.getPid(),
122                            "GC.class_stats", "VTab,ITab,OopMap,KlassBytes");
123             ProcessBuilder processBuilder = ProcessTools
124                                             .createJavaProcessBuilder(toolArgs);
125             output = ProcessTools.executeProcess(processBuilder);
126             System.out.println(output.getOutput());
127             output.shouldHaveExitValue(0);
128 
129             // Check whether the size matches that which jcmd outputs
130             for (String instanceKlassName : instanceKlassNames) {
131                 System.out.println ("Trying to match for" + instanceKlassName);
132                 String jcmdInstanceKlassSize = getJcmdInstanceKlassSize(
133                                                       jcmdOutput,
134                                                       instanceKlassName);
135                 Asserts.assertNotNull(jcmdInstanceKlassSize,
136                     "Could not get the instance klass size from the jcmd output");
137                 for (String s : output.asLines()) {
138                     if (s.contains(instanceKlassName)) {
139                        Asserts.assertTrue(
140                           s.contains(jcmdInstanceKlassSize),
141                           "The size computed by SA for" +
142                           instanceKlassName + " does not match.");

143                     }
144                 }

145             }
146         } finally {
147             LingeredApp.stopApp(app);
148         }
149     }
150 
151     private static void SAInstanceKlassSize(int pid,
152                                             String[] SAInstanceKlassNames) {
153         HotSpotAgent agent = new HotSpotAgent();
154         try {
155             agent.attach(pid);
156         }
157         catch (DebuggerException e) {
158             System.out.println(e.getMessage());
159             System.err.println("Unable to connect to process ID: " + pid);
160 
161             agent.detach();
162             e.printStackTrace();
163         }
164 
165         for (String SAInstanceKlassName : SAInstanceKlassNames) {
166             InstanceKlass ik = SystemDictionaryHelper.findInstanceKlass(
167                                SAInstanceKlassName);
168             Asserts.assertNotNull(ik,
169                 String.format("Unable to find instance klass for %s", SAInstanceKlassName));
170             System.out.println("SA: The size of " + SAInstanceKlassName +
171                                " is " + ik.getSize());
172         }
173         agent.detach();
174     }
175 
176     public static void main(String[] args) throws Exception {
177 
178         if (args == null || args.length == 0) {
179             System.out.println ("No args run. Starting with args now.");
180             startMeWithArgs();
181         } else {
182             String[] SAInstanceKlassNames = new String[] {
183                                                 "java.lang.Object",
184                                                 "java.util.Vector",
185                                                 "java.lang.String",
186                                                 "java.lang.Thread",
187                                                 "java.lang.Byte"
188                                              };
189             SAInstanceKlassSize(Integer.parseInt(args[0]), SAInstanceKlassNames);
190         }
191     }
