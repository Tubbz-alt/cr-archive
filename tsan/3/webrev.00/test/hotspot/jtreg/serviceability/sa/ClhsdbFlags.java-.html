 24 import java.util.ArrayList;
 25 import java.util.HashMap;
 26 import java.util.List;
 27 import java.util.Map;
 28 
 29 import jdk.test.lib.apps.LingeredApp;
 30 import jdk.test.lib.Utils;
 31 import jtreg.SkippedException;
 32 
 33 /**
 34  * @test
 35  * @bug 8190198
 36  * @bug 8217612
 37  * @bug 8217845
 38  * @summary Test clhsdb flags command
 39  * @requires vm.hasSA
 40  * @library /test/lib
 41  * @run main/othervm ClhsdbFlags
 42  */
 43 
 44 public class ClhsdbFlags {
 45 
 46     public static void runBasicTest() throws Exception {
 47         System.out.println("Starting ClhsdbFlags basic test");
 48 
 49         LingeredApp theApp = null;
 50         try {
 51             ClhsdbLauncher test = new ClhsdbLauncher();
 52             List<String> vmArgs = new ArrayList<String>();
 53             vmArgs.add("-XX:+UnlockExperimentalVMOptions");
 54             vmArgs.add("-XX:+UnlockDiagnosticVMOptions");
 55             vmArgs.add("-XX:-MaxFDLimit");
 56             vmArgs.addAll(Utils.getVmOptions());
 57             theApp = LingeredApp.startApp(vmArgs);
 58             System.out.println("Started LingeredApp with pid " + theApp.getPid());
 59 
 60             List<String> cmds = List.of(
 61                     "flags", "flags -nd",
 62                     "flags UnlockDiagnosticVMOptions", "flags MaxFDLimit",
 63                     "flags MaxJavaStackTraceDepth");
 64 
 65             Map<String, List<String>> expStrMap = new HashMap<>();
 66             expStrMap.put("flags", List.of(
 67                     "command line", "ergonomic", "default",
 68                     "UnlockDiagnosticVMOptions = true",
 69                     "MaxFDLimit = false",
 70                     "MaxJavaStackTraceDepth = 1024",
 71                     "VerifyMergedCPBytecodes",
 72                     "ConcGCThreads", "UseThreadPriorities",
 73                     "ShowHiddenFrames"));
 74             expStrMap.put("flags -nd", List.of(
 75                     "UnlockDiagnosticVMOptions = true",
 76                     "MaxFDLimit = false",
 77                     "InitialHeapSize",
 78                     "MaxHeapSize"));
 79             expStrMap.put("flags UnlockDiagnosticVMOptions", List.of(
 80                     "UnlockDiagnosticVMOptions = true"));
 81             expStrMap.put("flags MaxFDLimit", List.of(
 82                     "MaxFDLimit = false"));
 83             expStrMap.put("flags MaxJavaStackTraceDepth", List.of(
 84                     "MaxJavaStackTraceDepth = 1024"));
 85 
 86             test.run(theApp.getPid(), cmds, expStrMap, null);
 87         } catch (SkippedException se) {
 88             throw se;
 89         } catch (Exception ex) {
 90             throw new RuntimeException("Test ERROR " + ex, ex);
 91         } finally {
 92             LingeredApp.stopApp(theApp);
 93         }
 94         System.out.println("Test PASSED");
 95     }
 96 
 97     public static void runAllTypesTest() throws Exception {
 98         System.out.println("Starting ClhsdbFlags all types test");
 99 
100         LingeredApp theApp = null;
101         try {
102             ClhsdbLauncher test = new ClhsdbLauncher();
103             List<String> vmArgs = new ArrayList<String>();
104             vmArgs.add("-XX:+UnlockDiagnosticVMOptions");   // bool
105             vmArgs.add("-XX:ActiveProcessorCount=1");       // int
106             vmArgs.add("-XX:ParallelGCThreads=1");          // uint
107             vmArgs.add("-XX:MaxJavaStackTraceDepth=1024");  // intx
108             vmArgs.add("-XX:LogEventsBufferEntries=10");    // uintx
109             vmArgs.add("-XX:HeapSizePerGCThread=32m");      // size_t
110             vmArgs.add("-XX:NativeMemoryTracking=off");     // ccstr
111             vmArgs.add("-XX:OnError='echo error'");         // ccstrlist
112             vmArgs.add("-XX:CompileThresholdScaling=1.0");  // double
113             vmArgs.add("-XX:ErrorLogTimeout=120");          // uint64_t
114             vmArgs.addAll(Utils.getVmOptions());
115             theApp = LingeredApp.startApp(vmArgs);
116             System.out.println("Started LingeredApp with pid " + theApp.getPid());
117 
118             List<String> cmds = List.of("flags");
119 
120             Map<String, List<String>> expStrMap = new HashMap<>();
121             expStrMap.put("flags", List.of(
122                     "UnlockDiagnosticVMOptions = true",
123                     "ActiveProcessorCount = 1",
124                     "ParallelGCThreads = 1",
125                     "MaxJavaStackTraceDepth = 1024",
126                     "LogEventsBufferEntries = 10",
127                     "HeapSizePerGCThread = 3",
128                     "NativeMemoryTracking = \"off\"",
129                     "OnError = \"'echo error'\"",
130                     "CompileThresholdScaling = 1.0",
131                     "ErrorLogTimeout = 120"));
132 
133             test.run(theApp.getPid(), cmds, expStrMap, null);
134         } catch (Exception ex) {
135             throw new RuntimeException("Test ERROR " + ex, ex);
136         } finally {
137             LingeredApp.stopApp(theApp);
138         }
139         System.out.println("Test PASSED");
140     }
141 
142     public static void main(String[] args) throws Exception {
143         runBasicTest();
144         runAllTypesTest();
145     }
146 }
