 24 /**
 25  * @test
 26  * @bug 8174994
 27  * @summary Test the clhsdb commands 'jstack', 'printall', 'where' with CDS enabled
 28  * @requires vm.hasSA & vm.cds
 29  * @library /test/lib
 30  * @run main/othervm/timeout=2400 -Xmx1g ClhsdbCDSJstackPrintAll
 31  */
 32 
 33 import java.util.List;
 34 import java.util.Arrays;
 35 import java.util.Map;
 36 import java.util.HashMap;
 37 import jdk.test.lib.cds.CDSTestUtils;
 38 import jdk.test.lib.cds.CDSOptions;
 39 import jdk.test.lib.apps.LingeredApp;
 40 import jtreg.SkippedException;
 41 
 42 public class ClhsdbCDSJstackPrintAll {
 43 
 44     public static void main(String[] args) throws Exception {
 45         System.out.println("Starting ClhsdbCDSJstackPrintAll test");
 46         String sharedArchiveName = "ArchiveForClhsdbJstackPrintAll.jsa";
 47         LingeredApp theApp = null;
 48 
 49         try {
 50             CDSOptions opts = (new CDSOptions()).setArchiveName(sharedArchiveName);
 51             CDSTestUtils.createArchiveAndCheck(opts);
 52 
 53             ClhsdbLauncher test = new ClhsdbLauncher();
 54             List<String> vmArgs = Arrays.asList(
 55                 "-XX:+UnlockDiagnosticVMOptions",
 56                 "-XX:SharedArchiveFile=" + sharedArchiveName,
 57                 "-Xshare:auto");
 58             theApp = LingeredApp.startApp(vmArgs);
 59             System.out.println("Started LingeredApp with pid " + theApp.getPid());
 60 
 61             // Ensure that UseSharedSpaces is turned on.
 62             List<String> cmds = List.of("flags UseSharedSpaces");
 63 
 64             String useSharedSpacesOutput = test.run(theApp.getPid(), cmds,
 65                                                     null, null);
 66 
 67             if (useSharedSpacesOutput == null) {
 68                 LingeredApp.stopApp(theApp);
 69                 // Attach permission issues.
 70                 throw new SkippedException("Could not determine the UseSharedSpaces value");
 71             }
 72 
 73             if (!useSharedSpacesOutput.contains("true")) {
 74                 // CDS archive is not mapped. Skip the rest of the test.
 75                 LingeredApp.stopApp(theApp);
 76                 throw new SkippedException("The CDS archive is not mapped");
 77             }
 78 
 79             cmds = List.of("jstack -v", "printall", "where -a");
 80 
 81             Map<String, List<String>> expStrMap = new HashMap<>();
 82             Map<String, List<String>> unExpStrMap = new HashMap<>();
 83             expStrMap.put("jstack -v", List.of(
 84                 "No deadlocks found",
 85                 "Common-Cleaner",
 86                 "Signal Dispatcher",
 87                 "Method*",
 88                 "LingeredApp.main"));
 89             unExpStrMap.put("jstack -v", List.of(
 90                 "sun.jvm.hotspot.types.WrongTypeException",
 91                 "No suitable match for type of address"));
 92             expStrMap.put("printall", List.of(
 93                 "aload_0",
 94                 "_nofast_aload_0",
 95                 "_nofast_getfield",
 96                 "_nofast_putfield",
 97                 "Constant Pool of",
 98                 "public static void main\\(java.lang.String\\[\\]\\)",
 99                 "Bytecode",
100                 "invokevirtual",
101                 "checkcast",
102                 "Exception Table",
103                 "invokedynamic"));
104             unExpStrMap.put("printall", List.of(
105                 "No suitable match for type of address",
106                 "illegal code",
107                 "Failure occurred at bci"));
108             expStrMap.put("where -a", List.of(
109                 "Java Stack Trace for main",
110                 "public static void main"));
111             unExpStrMap.put("where -a", List.of(
112                 "illegal code",
113                 "Failure occurred at bci"));
114             test.run(theApp.getPid(), cmds, expStrMap, unExpStrMap);
115         } catch (SkippedException e) {
116             throw e;
117         } catch (Exception ex) {
118             throw new RuntimeException("Test ERROR " + ex, ex);
119         } finally {
120             LingeredApp.stopApp(theApp);
121         }
122         System.out.println("Test PASSED");
123     }
124 }
