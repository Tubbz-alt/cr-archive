24 /**
25  * @test
26  * @summary Checks that the jshdb debugd utility sucessfully starts
27  *          and tries to attach to a running process
28  * @requires vm.hasSAandCanAttach
29  * @modules java.base/jdk.internal.misc
30  * @library /test/lib
31  *
32  * @run main/othervm SADebugDTest
33  */
34 
35 import java.io.File;
36 import java.util.concurrent.CountDownLatch;
37 import java.io.InputStreamReader;
38 import java.io.BufferedReader;
39 import java.io.Reader;
40 import java.util.concurrent.TimeUnit;
41 import java.util.function.Predicate;
42 import static jdk.test.lib.Asserts.assertTrue;
43 import static jdk.test.lib.process.ProcessTools.startProcess;
44 
45 public class SADebugDTest {
46 
47     private static final String GOLDEN = "Attaching to process ID %d and starting RMI services, please wait...";
48 
49     private static final String JAVA_HOME = (System.getProperty("test.jdk") != null)
50             ? System.getProperty("test.jdk") : System.getProperty("java.home");
51 
52     private static final String JAVA_BIN_DIR
53             = JAVA_HOME + File.separator + "bin" + File.separator;
54 
55     private static final String JHSDB = JAVA_BIN_DIR + "jhsdb";
56 
57     public static void main(String[] args) throws Exception {
58 
59         long ourPid = ProcessHandle.current().pid();
60 
61         // The string we are expecting in the debugd ouput
62         String golden = String.format(GOLDEN, ourPid);
63 
64         // We are going to run 'jhsdb debugd <our pid>'
65         // The startProcess will block untl the 'golden' string appears in either process' stdout or stderr
66         // In case of timeout startProcess kills the debugd process
67         ProcessBuilder pb = new ProcessBuilder();
68         pb.command(JHSDB, "debugd", String.valueOf(ourPid));
69         Process debugd = startProcess("debugd", pb, null, (line) -> line.trim().contains(golden), 0, TimeUnit.SECONDS);
70 
71         // If we are here, this means we have received the golden line and the test has passed
72         // The debugd remains running, we have to kill it
73         debugd.destroy();
74 
75     }
76 
77     private static void log(String string) {
78         System.out.println(string);
79     }
80 
81 }
