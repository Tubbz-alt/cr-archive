24 import java.util.HashMap;
25 import java.util.List;
26 import java.util.Map;
27 
28 import jdk.test.lib.apps.LingeredApp;
29 import jtreg.SkippedException;
30 
31 /**
32  * @test
33  * @bug 8191658
34  * @summary Test clhsdb attach, detach, reattach commands
35  * @requires vm.hasSA
36  * @library /test/lib
37  * @run main/othervm ClhsdbAttach
38  */
39 
40 public class ClhsdbAttach {
41 
42     public static void main(String[] args) throws Exception {
43         System.out.println("Starting ClhsdbAttach test");
44 
45         LingeredApp theApp = null;
46         try {
47             ClhsdbLauncher test = new ClhsdbLauncher();
48             theApp = LingeredApp.startApp();
49             System.out.println("Started LingeredApp with pid " + theApp.getPid());
50             String attach = "attach " + theApp.getPid();
51 
52             List<String> cmds = List.of(
53                     "where",
54                     attach,
55                     "flags MaxJavaStackTraceDepth",
56                     "detach",
57                     "universe",
58                     "reattach",
59                     "longConstant markWord::locked_value");
60 
61             Map<String, List<String>> expStrMap = new HashMap<>();
62             expStrMap.put("where", List.of(
63                     "Command not valid until attached to a VM"));
64             expStrMap.put("flags MaxJavaStackTraceDepth", List.of(
65                     "MaxJavaStackTraceDepth = "));
66             expStrMap.put("universe", List.of(
67                     "Command not valid until attached to a VM"));
68             expStrMap.put("longConstant markWord::locked_value", List.of(
69                     "longConstant markWord::locked_value"));
70 
71             test.run(-1, cmds, expStrMap, null);
72         } catch (SkippedException se) {
73             throw se;
74         } catch (Exception ex) {
75             throw new RuntimeException("Test ERROR " + ex, ex);
76         } finally {
77             LingeredApp.stopApp(theApp);
78         }
79         System.out.println("Test PASSED");
80     }
81 }
