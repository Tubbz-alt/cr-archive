 24 /**
 25  * @test
 26  * @bug 8192985
 27  * @summary Test the clhsdb 'scanoops' command
 28  * @requires vm.hasSA
 29  * @library /test/lib
 30  * @run main/othervm/timeout=1200 ClhsdbScanOops
 31  */
 32 
 33 import java.util.HashMap;
 34 import java.util.List;
 35 import java.util.Map;
 36 import java.util.ArrayList;
 37 import jdk.test.lib.Utils;
 38 import jdk.test.lib.apps.LingeredApp;
 39 import jtreg.SkippedException;
 40 
 41 public class ClhsdbScanOops {
 42 
 43     private static void testWithGcType(String gc) throws Exception {
 44 
 45         LingeredApp theApp = null;
 46 
 47         try {
 48             ClhsdbLauncher test = new ClhsdbLauncher();
 49             theApp = LingeredApp.startApp(gc);
 50 
 51             System.out.println ("Started LingeredApp with the GC option " + gc +
 52                                 " and pid " + theApp.getPid());
 53 
 54             // Run the 'universe' command to get the address ranges
 55             List<String> cmds = List.of("universe");
 56 
 57             String universeOutput = test.run(theApp.getPid(), cmds, null, null);
 58 
 59             cmds = new ArrayList<String>();
 60             Map<String, List<String>> expStrMap = new HashMap<>();
 61             Map<String, List<String>> unExpStrMap = new HashMap<>();
 62 
 63             String startAddress = null;
 64             String endAddress = null;
 65             String[] snippets = null;
 66 
 67             if (gc.contains("UseParallelGC")) {
 68                 snippets = universeOutput.split("eden =  ");
 69             } else {
 70                 snippets = universeOutput.split("eden \\[");
 71             }
 72             String[] words = snippets[1].split(",");
 73             // Get the addresses from Eden
 74             startAddress = words[0].replace("[", "");
 75             endAddress = words[1];
 76             String cmd = "scanoops " + startAddress + " " + endAddress;
 77             cmds.add(cmd);
 78 
 79             expStrMap.put(cmd, List.of
 80                 ("java/lang/Object", "java/lang/Class", "java/lang/Thread",
 81                  "java/lang/String", "\\[B", "\\[I"));
 82 
 83             // Test the 'type' option also
 84             // scanoops <start addr> <end addr> java/lang/String
 85             // Ensure that only the java/lang/String oops are printed.
 86             cmd = cmd + " java/lang/String";
 87             cmds.add(cmd);
 88             expStrMap.put(cmd, List.of("java/lang/String"));
 89             unExpStrMap.put(cmd, List.of("java/lang/Thread"));
 90 
 91             test.run(theApp.getPid(), cmds, expStrMap, unExpStrMap);
 92         } catch (SkippedException e) {
 93             throw e;
 94         } catch (Exception ex) {
 95             throw new RuntimeException("Test ERROR " + ex, ex);
 96         } finally {
 97             LingeredApp.stopApp(theApp);
 98         }
 99     }
100 
101     public static void main(String[] args) throws Exception {
102         System.out.println("Starting the ClhsdbScanOops test");
103         testWithGcType("-XX:+UseParallelGC");
104         testWithGcType("-XX:+UseSerialGC");
105         System.out.println("Test PASSED");
106     }
107 }
