24 import java.util.ArrayList;
25 import java.util.List;
26 import java.util.Map;
27 import java.util.HashMap;
28 import jdk.test.lib.apps.LingeredApp;
29 import jdk.test.lib.Utils;
30 import jtreg.SkippedException;
31 
32 /**
33  * @test
34  * @library /test/lib
35  * @requires vm.hasSAandCanAttach
36  * @requires vm.flavor == "server" & !vm.emulatedClient & !(vm.opt.TieredStopAtLevel == 1)
37  * @build jdk.test.lib.apps.*
38  * @run main/othervm TestPrintMdo
39  */
40 
41 public class TestPrintMdo {
42 
43     public static void main (String... args) throws Exception {
44 
45         System.out.println("Starting TestPrintMdo test");
46         LingeredApp app = null;
47         try {
48             ClhsdbLauncher test = new ClhsdbLauncher();
49             List<String> vmArgs = new ArrayList<String>();
50             vmArgs.add("-XX:+ProfileInterpreter");
51             vmArgs.addAll(Utils.getVmOptions());
52 
53             app = LingeredApp.startApp(vmArgs);
54             System.out.println ("Started LingeredApp with pid " + app.getPid());
55             List<String> cmds = List.of("printmdo -a");
56 
57             Map<String, List<String>> expStrMap = new HashMap<>();
58             Map<String, List<String>> unExpStrMap = new HashMap<>();
59             expStrMap.put("printmdo -a", List.of(
60                 "VirtualCallData",
61                 "CounterData",
62                 "ReceiverTypeData",
63                 "bci",
64                 "MethodData",
65                 "java/lang/Object"));
66             unExpStrMap.put("printmdo -a", List.of(
67                             "missing reason for "));
68             test.run(app.getPid(), cmds, expStrMap, unExpStrMap);
69         } catch (SkippedException se) {
70             throw se;
71         } catch (Exception ex) {
72             throw new RuntimeException("Test ERROR " + ex, ex);
73         } finally {
74             LingeredApp.stopApp(app);
75         }
76 
77         System.out.println("Test PASSED");
78     }
79 }
