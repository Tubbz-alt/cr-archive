 29 import jdk.test.lib.apps.LingeredApp;
 30 import jtreg.SkippedException;
 31 
 32 /**
 33  * @test
 34  * @bug 8193124
 35  * @summary Test the clhsdb 'findpc' command
 36  * @requires vm.hasSA
 37  * @requires vm.compiler1.enabled
 38  * @library /test/lib
 39  * @run main/othervm/timeout=480 ClhsdbFindPC
 40  */
 41 
 42 public class ClhsdbFindPC {
 43 
 44     private static void testFindPC(boolean withXcomp) throws Exception {
 45         LingeredApp theApp = null;
 46         try {
 47             ClhsdbLauncher test = new ClhsdbLauncher();
 48             theApp = withXcomp ? LingeredApp.startApp(List.of("-Xcomp"))
 49                                : LingeredApp.startApp(List.of("-Xint"));





 50             System.out.print("Started LingeredApp ");
 51             if (withXcomp) {
 52                 System.out.print("(-Xcomp) ");
 53             } else {
 54                 System.out.print("(-Xint) ");
 55             }
 56             System.out.println("with pid " + theApp.getPid());
 57 
 58             // Run 'jstack -v' command to get the pc
 59             List<String> cmds = List.of("jstack -v");
 60             String output = test.run(theApp.getPid(), cmds, null, null);
 61 
 62             // Test the 'findpc' command passing in the pc obtained from
 63             // the 'jstack -v' command
 64             cmds = new ArrayList<String>();
 65 
 66             // Output could be null if the test was skipped due to
 67             // attach permission issues.
 68             if (output != null) {
 69                 String cmdStr = null;
 70                 String[] parts = output.split("LingeredApp.main");
 71                 String[] tokens = parts[1].split(" ");
 72                 for (String token : tokens) {
 73                     if (token.contains("pc")) {
 74                         String[] address = token.split("=");
 75                         // address[1] represents the address of the Method
 76                         cmdStr = "findpc " + address[1].replace(",","");
 77                         cmds.add(cmdStr);
 78                         break;
 79                     }
 80                 }
 81 
 82                 Map<String, List<String>> expStrMap = new HashMap<>();
 83                 if (withXcomp) {
 84                     expStrMap.put(cmdStr, List.of(
 85                             "In code in NMethod for jdk/test/lib/apps/LingeredApp.main",
 86                             "content:",
 87                             "oops:",
 88                             "frame size:"));
 89                 } else {
 90                     expStrMap.put(cmdStr, List.of(
 91                             "In interpreter codelet"));
 92                 }
 93 
 94                 test.run(theApp.getPid(), cmds, expStrMap, null);
 95             }
 96         } catch (SkippedException se) {
 97             throw se;
 98         } catch (Exception ex) {
 99             throw new RuntimeException("Test ERROR " + ex, ex);
100         } finally {
101             LingeredApp.stopApp(theApp);
102         }
103     }
104 
105     public static void main(String[] args) throws Exception {
106         System.out.println("Starting the ClhsdbFindPC test");
107         testFindPC(true);
108         testFindPC(false);
109         System.out.println("Test PASSED");
110     }
111 }
