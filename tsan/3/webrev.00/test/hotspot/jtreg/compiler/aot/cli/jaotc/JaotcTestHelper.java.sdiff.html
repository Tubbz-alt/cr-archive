31 import jdk.test.lib.process.ExitCode;
32 import jdk.test.lib.Platform;
33 import jdk.test.lib.JDKToolLauncher;
34 import jdk.test.lib.Utils;

39 public class JaotcTestHelper {
40     public static final String DEFAULT_LIB_PATH = "./unnamed." + Platform.sharedLibraryExt();
41     public static final String DEFAULT_LIBRARY_LOAD_MESSAGE = "loaded    " + DEFAULT_LIB_PATH
42             + "  aot library";

43     private static final String ENABLE_AOT = "-XX:+UseAOT";
44     private static final String AOT_LIBRARY = "-XX:AOTLibrary=" + DEFAULT_LIB_PATH;
45     private static final String PRINT_AOT = "-XX:+PrintAOT";
46 
47     public static OutputAnalyzer compileLibrary(String... args) {
48         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("jaotc");
49         for (String vmOpt : Utils.getTestJavaOpts()) {
50             launcher.addVMArg(vmOpt);
51         }
52         launcher.addToolArg("--compile-with-assertions");
53         for (String arg : args) {
54             launcher.addToolArg(arg);
55         }
56         String linker = AotCompiler.resolveLinker();
57         if (linker != null) {
58             launcher.addToolArg("--linker-path");
59             launcher.addToolArg(linker);
60         }
61         String[] cmd = launcher.getCommand();
62         try {
63             return ProcessTools.executeCommand(cmd);
64         } catch (Throwable e) {
65             throw new Error("Can't start test process: " + e, e);
66         }
67     }
68 
69     public static void checkLibraryUsage(String classToRun) {
70         checkLibraryUsage(classToRun, new String[]{DEFAULT_LIBRARY_LOAD_MESSAGE}, null);
71     }
72 
73     public static void checkLibraryUsage(String classToRun, String[] expectedOutput,
74             String[] unexpectedOutput) {
75         try {
76             CommandLineOptionTest.verifyJVMStartup(expectedOutput, unexpectedOutput,
77                     "Unexpected exit code", "Unexpected output", ExitCode.OK,
78                     /* addTestVMOpts */ true, ENABLE_AOT, AOT_LIBRARY, PRINT_AOT, classToRun);

79         } catch (Throwable t) {
80             throw new Error("Library usage verification failed: " + t, t);
81         }
82     }
83 
84     public static String getClassAotCompilationFilename(Class<?> classToCompile) {
85         return classToCompile.getName().replaceAll("\\.","/") + ".class";
86     }
87 
88     public static String getClassAotCompilationName(Class<?> classToCompile) {
89         return classToCompile.getName();
90     }
91 }
