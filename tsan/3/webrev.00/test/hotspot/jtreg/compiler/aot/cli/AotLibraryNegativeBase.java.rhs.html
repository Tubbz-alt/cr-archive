24 package compiler.aot.cli;
25 
26 import compiler.aot.HelloWorldPrinter;
27 import jdk.test.lib.process.ExitCode;
28 import jdk.test.lib.cli.CommandLineOptionTest;
29 
30 public class AotLibraryNegativeBase {
31     private static final String[] UNEXPECTED_MESSAGES = new String[] {
32         HelloWorldPrinter.MESSAGE
33     };
34 
35     public static void launchTest(String option, String expectedMessages[]) {
36         try {
37             boolean addTestVMOptions = true;
38             CommandLineOptionTest.verifyJVMStartup(expectedMessages,
39                     UNEXPECTED_MESSAGES,
40                     "Unexpected exit code using " + option,
41                     "Unexpected output using " + option, ExitCode.FAIL,
42                     addTestVMOptions, "-XX:+UnlockExperimentalVMOptions", "-XX:+UseAOT",
43                     "-XX:+PrintAOT", option, HelloWorldPrinter.class.getName());
44         } catch (Throwable t) {
45             throw new Error("Problems executing test using " + option
46                     + ": " + t, t);
47         }
48     }
49 }
