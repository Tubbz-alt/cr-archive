 53  * @summary check if recompilation after aot goes fine

 56  /* having whitebox-related options for aot compiler is a temporary solution,
 57     because of JDK-8146201
 58  */
 59 
 60 package compiler.aot;
 61 
 62 import compiler.whitebox.CompilerWhiteBoxTest;
 63 import java.lang.reflect.Executable;
 64 import jdk.test.lib.Asserts;
 65 
 66 public final class RecompilationTest extends CompilerWhiteBoxTest {
 67     private static final int CHECK_LEVEL = Integer.getInteger(
 68                 "compiler.aot.RecompilationTest.check_level");
 69 
 70     public static void main(String args[]) {
 71         CompilerWhiteBoxTest.main(RecompilationTest::new, args);
 72     }
 73 
 74     private RecompilationTest(TestCase testCase) {
 75         super(testCase);
 76     }
 77 
 78     @Override
 79     protected void test() throws Exception {
 80         if (testCase.isOsr()) {
 81             /* aot compiler is not using osr compilation */
 82             System.out.println("Skipping OSR case");
 83             return;
 84         }
 85         Executable e = testCase.getExecutable();
 86         Asserts.assertTrue(WHITE_BOX.isMethodCompiled(e),
 87                 testCase.name() +  ": an executable expected to be compiled");
 88         Asserts.assertEQ(WHITE_BOX.getMethodCompilationLevel(e),
 89                 COMP_LEVEL_AOT,
 90                 String.format("%s: unexpected compilation level at start",
 91                         testCase.name()));
 92         compile();
 93         Asserts.assertTrue(WHITE_BOX.isMethodCompiled(e), testCase.name()
 94                 + ": method expected to be compiled");
 95         /* a case with AOT'ed code checks exact compilation level equality
 96            while another case checks minimum level and if method compiled
 97            because there might be different compilation level transitions */
 98         if (CHECK_LEVEL != COMP_LEVEL_AOT) {
 99             Asserts.assertGTE(WHITE_BOX.getMethodCompilationLevel(e),
100                 CHECK_LEVEL,
101                 String.format("%s: expected compilation level"
102                         + " after compilation to be no less than %d for %s",
103                         testCase.name(), CHECK_LEVEL, testCase.name()));
104         } else {
105             Asserts.assertEQ(WHITE_BOX.getMethodCompilationLevel(e),
106                 COMP_LEVEL_AOT, String.format("%s: expected compilation"
107                         + " level after compilation to be equal to %d for %s",
108                         testCase.name(), COMP_LEVEL_AOT, testCase.name()));
109         }
110     }
111 }
