 24 /*
 25  * @test
 26  * @summary AOT methods should be swept if a super class has changed.
 27  * @library /test/lib /
 28  * @modules java.base/jdk.internal.misc
 29  *          java.management
 30  * @requires vm.aot
 31  * @build compiler.aot.fingerprint.SuperChanged
 32  *        compiler.aot.AotCompiler
 33  *
 34  * @run main
 35  *      compiler.aot.fingerprint.SuperChanged WRITE-UNMODIFIED-CLASS
 36  * @run driver compiler.aot.AotCompiler -libname libSuperChanged.so
 37  *      -class compiler.aot.fingerprint.Foo
 38  *
 39  * @run main
 40  *      compiler.aot.fingerprint.SuperChanged TEST-UNMODIFIED
 41  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseAOT -XX:+PrintAOT
 42  *      -XX:AOTLibrary=./libSuperChanged.so
 43  *      -Xlog:aot+class+fingerprint=trace -Xlog:aot+class+load=trace
 44  *      compiler.aot.fingerprint.SuperChanged TEST-UNMODIFIED
 45   *
 46  * @run main
 47  *      compiler.aot.fingerprint.SuperChanged WRITE-MODIFIED-CLASS
 48  * @run main
 49  *      compiler.aot.fingerprint.SuperChanged TEST-MODIFIED
 50  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseAOT -XX:+PrintAOT
 51  *      -XX:AOTLibrary=./libSuperChanged.so
 52  *      -Xlog:aot+class+fingerprint=trace -Xlog:aot+class+load=trace
 53  *      compiler.aot.fingerprint.SuperChanged TEST-MODIFIED
 54  */
 55 
 56 package compiler.aot.fingerprint;
 57 
 58 import jdk.test.lib.Asserts;
 59 import jdk.test.lib.compiler.InMemoryJavaCompiler;
 60 
 61 import java.io.*;
 62 
 63 class Bar {
 64     volatile int x = 0;
 65     volatile int y = 1;
 66 }
 67 
 68 class Foo extends Bar {
 69 
 70     volatile int z;
 71     int getX() {
 72         for (z = 0; z < 10000; z++) {
 73             if (z % 7 == 1) {
 74                 z += 2;
 75             }
 76         }
 77         return x;
 78     }
 79 }
 80 
 81 public class SuperChanged {
 82     public static void main(String args[]) throws Throwable {
 83         Foo f = new Foo();
 84         System.out.println("f.getX = " + f.getX());
 85         switch (args[0]) {
 86         case "WRITE-UNMODIFIED-CLASS":
 87             compileClass(false);
 88             break;
 89         case "WRITE-MODIFIED-CLASS":
 90             compileClass(true);
 91             break;
 92         case "TEST-UNMODIFIED":
 93             Asserts.assertTrue(f.getX() == 0, "getX from unmodified Foo class should return 0");
 94             break;
 95         case "TEST-MODIFIED":
 96             Asserts.assertTrue(f.getX() == 1, "getX from modified Foo class should return 1");
 97             break;
 98         default:
 99             throw new RuntimeException("unexpected option: " + args[0]);
100         }
101     }
102 
103     static void compileClass(boolean isModified) throws Throwable {
104         String class_src_0 = "package compiler.aot.fingerprint; class Bar {volatile int x = 0;  volatile int y = 1;}";
105         String class_src_1 = "package compiler.aot.fingerprint; class Bar {volatile int y = 0;  volatile int x = 1;}";
106         String src = (isModified) ? class_src_1 : class_src_0;
107 
108         String filename = System.getProperty("test.classes") + "/compiler/aot/fingerprint/Bar.class";
109         FileOutputStream fos = new FileOutputStream(filename);
110         fos.write(InMemoryJavaCompiler.compile("compiler.aot.fingerprint.Bar", src));
111         fos.close();
112     }
113 }
