 24 /*
 25  * @test
 26  * @summary AOT methods should be swept if a super class has changed.
 27  * @library /test/lib /
 28  * @modules java.base/jdk.internal.misc
 29  *          java.management
 30  * @requires vm.aot
 31  * @build compiler.aot.fingerprint.SelfChanged
 32  *        compiler.aot.AotCompiler
 33  *
 34  * @run main
 35  *      compiler.aot.fingerprint.SelfChanged WRITE-UNMODIFIED-CLASS
 36  * @run driver compiler.aot.AotCompiler -libname libSelfChanged.so
 37  *      -class compiler.aot.fingerprint.Blah
 38  *
 39  * @run main/othervm
 40  *      compiler.aot.fingerprint.SelfChanged TEST-UNMODIFIED
 41  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseAOT -XX:+PrintAOT
 42  *      -XX:AOTLibrary=./libSelfChanged.so
 43  *      -Xlog:aot+class+fingerprint=trace -Xlog:aot+class+load=trace
 44  *      compiler.aot.fingerprint.SelfChanged TEST-UNMODIFIED
 45  *
 46  * @run main
 47  *      compiler.aot.fingerprint.SelfChanged WRITE-MODIFIED-CLASS
 48  * @run main
 49  *      compiler.aot.fingerprint.SelfChanged TEST-MODIFIED
 50  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseAOT -XX:+PrintAOT
 51  *      -XX:AOTLibrary=./libSelfChanged.so
 52  *      -Xlog:aot+class+fingerprint=trace -Xlog:aot+class+load=trace
 53  *      compiler.aot.fingerprint.SelfChanged TEST-MODIFIED
 54  */
 55 
 56 package compiler.aot.fingerprint;
 57 
 58 import jdk.test.lib.Asserts;
 59 import jdk.test.lib.compiler.InMemoryJavaCompiler;
 60 
 61 import java.io.*;
 62 
 63 class Blah {
 64     volatile int z;
 65     int getX() {
 66         for (z = 0; z < 10000; z++) {
 67             if (z % 7 == 1) {
 68                 z += 2;
 69             }
 70         }
 71         return 0;
 72     }
 73 }
 74 
 75 public class SelfChanged {
 76     public static void main(String args[]) throws Throwable {
 77         Blah f = new Blah();
 78         System.out.println("f.getX = " + f.getX());
 79         switch (args[0]) {
 80         case "WRITE-UNMODIFIED-CLASS":
 81             compileClass(false);
 82             break;
 83         case "WRITE-MODIFIED-CLASS":
 84             compileClass(true);
 85             break;
 86         case "TEST-UNMODIFIED":
 87             Asserts.assertTrue(f.getX() == 0, "getX from unmodified Blah class should return 0");
 88             break;
 89         case "TEST-MODIFIED":
 90             Asserts.assertTrue(f.getX() == 1, "getX from modified Blah class should return 1");
 91             break;
 92         default:
 93             throw new RuntimeException("unexpected option: " + args[0]);
 94         }
 95     }
 96 
 97     static void compileClass(boolean isModified) throws Throwable {
 98         String src =
 99                "package compiler.aot.fingerprint;"
100              + "public class Blah {"
101              + "    volatile int z;"
102              + "    int getX() {"
103              + "        for (z = 0; z < 10000; z++) {"
104              + "            if (z % 7 == 1) {"
105              + "                z += 2;"
106              + "            }"
107              + "        }"
108              + "        return " + ((isModified) ? "1" : "0") + ";"
109              + "    }"
110              + "    int getY() {return 255;}"
111 
112             // The following is for the SelfChangedCDS.java test case. We always load an unmodified
113             // version of Blah from the CDS archive. However, we would load an AOT library that
114             // was compiled using a modified version of Blah. The getX method in this AOT library should
115             // not be used.
116 
117             + "    public static void main(String args[]) {"
118              + "        Blah b = new Blah();"
119              + "        int n = b.getX();"
120              + "        if (n != 0) {"
121              + "            throw new RuntimeException(args[0] +  \" : \" + n);"
122              + "        }"
123              + "        System.out.println(\"PASSED\");"
124              + "    }"
125              + "}";
126 
127         String filename = System.getProperty("test.classes") + "/compiler/aot/fingerprint/Blah.class";
128         FileOutputStream fos = new FileOutputStream(filename);
129         fos.write(InMemoryJavaCompiler.compile("compiler.aot.fingerprint.Blah", src));
130         fos.close();
131     }
132 }
