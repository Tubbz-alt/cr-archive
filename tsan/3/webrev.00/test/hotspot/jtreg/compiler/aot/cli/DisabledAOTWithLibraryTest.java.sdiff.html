43 public class DisabledAOTWithLibraryTest {
44     private final static String LIB_NAME = "libDisabledAOTWithLibraryTest.so";
45     private final static String[] UNEXPECTED_MESSAGES = new String[] {
46         LIB_NAME + "  aot library"
47     };
48 
49     private final static String[] EXPECTED_MESSAGES = new String[] {

51     };
52
