41 package compiler.aot;
42 
43 import jdk.test.lib.Asserts;
44 import jdk.test.lib.process.ExitCode;
45 import jdk.test.lib.Utils;
46 import jdk.test.lib.cli.CommandLineOptionTest;
47 
48 public class SharedUsageTest {
49     private static final String HELLO_MSG = "HelloWorld";
50     private static final boolean ADD_TEST_VM_OPTION = false;
51     private static boolean shouldBeFalseInParent = false;
52     private static final boolean IS_PARENT = Boolean.getBoolean(
53         "compiler.aot.SharedUsageTest.parent");
54 
55     public static void main(String args[]) throws Throwable {
56         Asserts.assertFalse(shouldBeFalseInParent,
57                 "A test invariant is broken");
58         if (IS_PARENT) {
59             /* An output of .so being used is verified after launch.
60                A respective message is triggered by PrintAOT option. */
61             CommandLineOptionTest.verifyJVMStartup(
62                     new String[]{"libSharedUsageTest.so  aot library",
63                         HELLO_MSG}, null, "Unexpected exit code",
64                     "Unexpected output", ExitCode.OK, ADD_TEST_VM_OPTION,
65                     "-XX:+UseAOT", "-XX:+PrintAOT",
66                     "-Dtest.jdk=" + Utils.TEST_JDK,
67                     "-XX:AOTLibrary=./libSharedUsageTest.so",
68                     SharedUsageTest.class.getName());
69             Asserts.assertFalse(shouldBeFalseInParent, "A static member got "
70                     + "unexpectedly changed");
71         } else {
72             shouldBeFalseInParent = true;
73             Asserts.assertTrue(shouldBeFalseInParent, "A static member wasn't"
74                     + "changed as expected");
75             System.out.println(HELLO_MSG);
76         }
77     }
78 }
