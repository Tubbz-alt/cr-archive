24 /*
25  * @test
26  * @bug 8224658
27  * @summary Test compilation of unsafe access with zero address.
28  * @modules java.base/jdk.internal.misc:+open
29  * @run main/othervm -Xcomp
30  *                   -XX:CompileCommand=compileonly,compiler.unsafe.TestUnsafeLoadWithZeroAddress::*
31  *                   compiler.unsafe.TestUnsafeLoadWithZeroAddress
32  * @run main/othervm -Xcomp -XX:+IgnoreUnrecognizedVMOptions -XX:+AlwaysIncrementalInline
33  *                   -XX:CompileCommand=compileonly,compiler.unsafe.TestUnsafeLoadWithZeroAddress::*
34  *                   compiler.unsafe.TestUnsafeLoadWithZeroAddress
35  */
36 
37 package compiler.unsafe;
38 
39 import jdk.internal.misc.Unsafe;
40 
41 public class TestUnsafeLoadWithZeroAddress {
42     static final Unsafe UNSAFE = Unsafe.getUnsafe();
43     static boolean f;
44 
45     public static void test1() {
46         if (f) {
47             // This branch is never executed but compiled due to -Xcomp
48             UNSAFE.getInt(0);
49         }
50     }
51 
52     public static void test2() {
53         if (f) {
54             // This branch is never executed but compiled due to -Xcomp
55             UNSAFE.putInt(0, 0);
56         }
57     }
58 
59     private static int getAddress() {
60         return 0;
61     }
62 
63     public static void test3() {
64         if (f) {
65             // This branch is never executed but compiled due to -Xcomp
66             UNSAFE.getInt(getAddress());
67         }
68     }
69 
70     public static void test4() {
71         if (f) {
72             // This branch is never executed but compiled due to -Xcomp
73             UNSAFE.putInt(getAddress(), 0);
74         }
75     }
76 
77     static public void main(String[] args) {
78         test1();
79         test2();
80         test3();
81         test4();
82     }
83 }
