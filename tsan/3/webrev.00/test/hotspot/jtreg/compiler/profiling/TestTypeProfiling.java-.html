 24  /**
 25   * @test
 26   * @bug 8189439
 27   * @summary Parameters type profiling is not performed from aarch64 interpreter
 28   * @requires os.arch != "arm"
 29   * @requires vm.flavor == "server" & vm.compMode == "Xmixed" & !vm.emulatedClient & !vm.graal.enabled
 30   * @library /test/lib /
 31   * @build sun.hotspot.WhiteBox
 32   * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
 33   * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI
 34   *                   -XX:-BackgroundCompilation -XX:-UseOnStackReplacement
 35   *                   -XX:CompileThreshold=10000
 36   *                   -server -XX:-TieredCompilation -XX:TypeProfileLevel=020
 37   *                    compiler.profiling.TestTypeProfiling
 38   * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI
 39   *                   -XX:-BackgroundCompilation -XX:-UseOnStackReplacement
 40   *                   -XX:CompileThreshold=10000
 41   *                   -server -XX:-TieredCompilation -XX:TypeProfileLevel=200
 42   *                    compiler.profiling.TestTypeProfiling
 43   */
 44 
 45 package compiler.profiling;
 46 
 47 import jdk.test.lib.Platform;
 48 import sun.hotspot.WhiteBox;
 49 import compiler.whitebox.CompilerWhiteBoxTest;
 50 import java.lang.reflect.Method;
 51 
 52 public class TestTypeProfiling {
 53 
 54     public static int[] mParamTypeCheck(Object in) {
 55         try {
 56             return (int[]) in;
 57         } catch (ClassCastException cce) {
 58             return null;
 59         }
 60     }
 61 
 62     static Object x2(Object src) {
 63         return src;
 64     }
 65 
 66     public static int[] mRetTypeCheck(Object in) {
 67         try {
 68             Object out = x2(in);
 69             return (int[]) out;
 70         } catch (ClassCastException cce) {
 71             return null;
 72         }
 73     }
 74 
 75     private static final WhiteBox WHITE_BOX = WhiteBox.getWhiteBox();
 76     private static final int TIERED_STOP_AT_LEVEL = WHITE_BOX.getIntxVMFlag("TieredStopAtLevel").intValue();
 77 
 78     static boolean deoptimize(Method method, Object src_obj) throws Exception {
 79         for (int i = 0; i < 10; i++) {
 80             method.invoke(null, src_obj);
 81             if (!WHITE_BOX.isMethodCompiled(method)) {
 82                 return true;
 83             }
 84         }
 85         return false;
 86     }
 87 
 88     static public void main(String[] args) throws Exception {
 89         if (!Platform.isServer() || Platform.isEmulatedClient()) {
 90             throw new Error("TESTBUG: Not server mode");
 91         }
 92         // Only execute if C2 is available
 93         if (TIERED_STOP_AT_LEVEL != CompilerWhiteBoxTest.COMP_LEVEL_FULL_OPTIMIZATION) {
 94             throw new RuntimeException("please enable C2");
 95         }
 96 
 97         Method method;
 98         if (WHITE_BOX.getUintxVMFlag("TypeProfileLevel") == 20) {
 99             method = TestTypeProfiling.class.getMethod("mRetTypeCheck", Object.class);
100         } else
101         if (WHITE_BOX.getUintxVMFlag("TypeProfileLevel") == 200) {
102             method = TestTypeProfiling.class.getMethod("mParamTypeCheck", Object.class);
103         } else {
104             throw new RuntimeException("please setup method return/params type profilation: -XX:TypeProfileLevel=020/200");
105         }
106 
107         int[] src = new int[10];
108         Object src_obj = new Object();
109 
110         // Warm up & make sure we collect type profiling
111         for (int i = 0; i < 20000; i++) {
112             mParamTypeCheck(src);
113             mRetTypeCheck(src);
114         }
115 
116         // And make sure the method is compiled by C2
117         WHITE_BOX.enqueueMethodForCompilation(method, CompilerWhiteBoxTest.COMP_LEVEL_FULL_OPTIMIZATION);
118         if (!WHITE_BOX.isMethodCompiled(method)) {
119             throw new RuntimeException(method.getName() + " is not compiled");
120         }
121 
122         // should deoptimize for speculative type check
123         // Intepreter will also add actual type check trap information into MDO
124         // when it throw ClassCastException
125         if (!deoptimize(method, src_obj)) {
126             throw new RuntimeException(method.getName() + " is not deoptimized");
127         }
128 
129         // compile again
130         // c2 will generate throw instead of uncommon trap because
131         // actual type check trap information is present in MDO
132         WHITE_BOX.enqueueMethodForCompilation(method, CompilerWhiteBoxTest.COMP_LEVEL_FULL_OPTIMIZATION);
133         if (!WHITE_BOX.isMethodCompiled(method)) {
134             throw new RuntimeException(method.getName() + " is not recompiled");
135         }
136 
137         // this time new parameter type should not force deoptimization
138         if (deoptimize(method, src_obj)) {
139             throw new RuntimeException(method.getName() + " is deoptimized again");
140         }
141     }
142 }
