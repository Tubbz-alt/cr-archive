41 package compiler.profiling.spectrapredefineclass;
42 
43 import jdk.test.lib.JDKToolLauncher;
44 import jdk.test.lib.process.OutputAnalyzer;
45 
46 import java.io.File;
47 import java.io.IOException;
48 import java.io.PrintWriter;
49 
50 public class Launcher {
51     private static final String MANIFEST = "MANIFEST.MF";
52     public static void main(String[] args) throws Exception  {
53         try (PrintWriter pw = new PrintWriter(MANIFEST)) {
54             pw.println("Agent-Class: " + Agent.class.getName());
55             pw.println("Can-Retransform-Classes: true");
56         }
57 
58         JDKToolLauncher jar = JDKToolLauncher.create("jar")
59                 .addToolArg("cmf")
60                 .addToolArg(MANIFEST)
61                 .addToolArg(Agent.AGENT_JAR)
62                 .addToolArg(Agent.class.getName().replace('.', File.separatorChar) + ".class");
63 
64         ProcessBuilder pb = new ProcessBuilder(jar.getCommand());
65         try {
66             OutputAnalyzer output = new OutputAnalyzer(pb.start());
67             output.shouldHaveExitValue(0);
68         } catch (IOException ex) {
69             throw new Error("TESTBUG: jar failed.", ex);
70         }
71     }
72 }
