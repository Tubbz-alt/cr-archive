44 package compiler.profiling.spectrapredefineclass_classloaders;
45 
46 import jdk.test.lib.JDKToolLauncher;
47 import jdk.test.lib.process.OutputAnalyzer;
48 
49 import java.io.File;
50 import java.io.IOException;
51 import java.io.PrintWriter;
52 
53 public class Launcher {
54     private static final String MANIFEST = "MANIFEST.MF";
55     public static void main(String[] args) throws Exception  {
56         try (PrintWriter pw = new PrintWriter(MANIFEST)) {
57             pw.println("Agent-Class: " + Agent.class.getName());
58             pw.println("Can-Retransform-Classes: true");
59         }
60 
61         JDKToolLauncher jar = JDKToolLauncher.create("jar")
62                 .addToolArg("cmf")
63                 .addToolArg(MANIFEST)
64                 .addToolArg(Agent.AGENT_JAR)
65                 .addToolArg(Agent.class.getName().replace('.', File.separatorChar) + ".class");
66 
67         ProcessBuilder pb = new ProcessBuilder(jar.getCommand());
68         try {
69             OutputAnalyzer output = new OutputAnalyzer(pb.start());
70             output.shouldHaveExitValue(0);
71         } catch (IOException ex) {
72             throw new Error("TESTBUG: jar failed.", ex);
73         }
74     }
75 }
