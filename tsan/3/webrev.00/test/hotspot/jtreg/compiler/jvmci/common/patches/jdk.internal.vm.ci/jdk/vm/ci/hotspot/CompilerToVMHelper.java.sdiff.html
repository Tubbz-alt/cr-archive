164     public static int installCode(TargetDescription target,
165             HotSpotCompiledCode compiledCode, InstalledCode code, HotSpotSpeculationLog speculationLog) {
166         byte[] speculations;
167         long failedSpeculationsAddress;
168         if (speculationLog != null) {
169             speculations = speculationLog.getFlattenedSpeculations(true);
170             failedSpeculationsAddress = speculationLog.getFailedSpeculationsAddress();
171         } else {
172             speculations = new byte[0];
173             failedSpeculationsAddress = 0L;
174         }
175         return CTVM.installCode(target, compiledCode, code, failedSpeculationsAddress, speculations);
176     }
