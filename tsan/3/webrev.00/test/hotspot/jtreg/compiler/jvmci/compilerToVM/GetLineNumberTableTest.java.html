45 package compiler.jvmci.compilerToVM;
46 
47 import compiler.jvmci.common.CTVMUtilities;
48 import compiler.jvmci.common.testcases.TestCase;
49 import jdk.test.lib.Asserts;
50 import jdk.vm.ci.hotspot.CompilerToVMHelper;
51 import jdk.vm.ci.hotspot.HotSpotResolvedJavaMethod;
52 
53 import java.lang.reflect.Executable;
54 import java.util.Arrays;
55 import java.util.Map;
56 
57 public class GetLineNumberTableTest {
58     public static void main(String[] args) {
59         TestCase.getAllExecutables()
60                 .forEach(GetLineNumberTableTest::runSanityTest);
61     }
62 
63     public static void runSanityTest(Executable aMethod) {
64         HotSpotResolvedJavaMethod method = CTVMUtilities
65                 .getResolvedMethod(aMethod);
66         long[] lineNumbers = CompilerToVMHelper.getLineNumberTable(method);
67         long[] expectedLineNumbers = getExpectedLineNumbers(aMethod);
68 
69         Asserts.assertTrue(Arrays.equals(lineNumbers, expectedLineNumbers),
70                 String.format("%s : unequal table values : %n%s%n%s%n",
71                         aMethod,
72                         Arrays.toString(lineNumbers),
73                         Arrays.toString(expectedLineNumbers)));
74     }
75 
76     public static long[] getExpectedLineNumbers(Executable aMethod) {
77         Map<Integer, Integer> bciToLine = CTVMUtilities
78                 .getBciToLineNumber(aMethod);
79         long[] result = null;
80         if (!bciToLine.isEmpty()) {
81             result = new long[2 * bciToLine.size()];
82             int i = 0;
83             for (Integer key : bciToLine.keySet()) {
84                 result[i++] = key.longValue();
85                 result[i++] = bciToLine.get(key).longValue();
86             }
87         }
88         // compilerToVM::getLineNumberTable returns null in case empty table
89         return result;
90     }
91 
92 }
