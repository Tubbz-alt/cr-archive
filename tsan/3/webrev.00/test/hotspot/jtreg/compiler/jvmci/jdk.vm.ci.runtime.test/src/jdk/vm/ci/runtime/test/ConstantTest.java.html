34 
35 import jdk.vm.ci.meta.JavaConstant;
36 import org.junit.Assert;
37 import org.junit.Test;
38 
39 public class ConstantTest extends FieldUniverse {
40 
41     @Test
42     public void testNegativeZero() {
43         Assert.assertTrue("Constant for 0.0f must be different from -0.0f", JavaConstant.FLOAT_0 != JavaConstant.forFloat(-0.0F));
44         Assert.assertTrue("Constant for 0.0d must be different from -0.0d", JavaConstant.DOUBLE_0 != JavaConstant.forDouble(-0.0d));
45     }
46 
47     @Test
48     public void testNullIsNull() {
49         Assert.assertTrue(JavaConstant.NULL_POINTER.isNull());
50     }
