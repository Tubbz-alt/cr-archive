34 package jdk.vm.ci.hotspot.test;
35 
36 import java.lang.reflect.Field;
37 import java.lang.reflect.Method;
38 import java.util.HashMap;
39 import java.util.Map;
40 
41 import org.testng.Assert;
42 import org.testng.annotations.Test;
43 
44 import jdk.vm.ci.services.Services;
45 
46 public class TestServices {
47 
48     @SuppressWarnings("unchecked")
49     @Test
50     public void serializeSavedPropertiesTest() throws Exception {
51 
52         Field f = Services.class.getDeclaredField("MAX_UTF8_PROPERTY_STRING_LENGTH");
53         f.setAccessible(true);
54         int maxUtf8PropertyStringLength = (int) f.get(null);
55 
56         Method serializeProperties = Services.class.getDeclaredMethod("serializeProperties", Map.class);
57         Method deserializeProperties = Services.class.getDeclaredMethod("deserializeProperties", byte[].class);
58         serializeProperties.setAccessible(true);
59         deserializeProperties.setAccessible(true);
60 
61         Map<String, String> props = new HashMap<>(Services.getSavedProperties());
62         String[] names = {
63                         new String(new char[maxUtf8PropertyStringLength - 100]).replace('\0', 'x'),
64                         new String(new char[maxUtf8PropertyStringLength - 1]).replace('\0', 'x'),
65                         new String(new char[maxUtf8PropertyStringLength]).replace('\0', 'y'),
66                         new String(new char[maxUtf8PropertyStringLength + 1]).replace('\0', 'z'),
67                         new String(new char[maxUtf8PropertyStringLength + 100]).replace('\0', 'z')
68         };
69         String[] values = {
70                         new String(new char[maxUtf8PropertyStringLength - 100]).replace('\0', '1'),
71                         new String(new char[maxUtf8PropertyStringLength - 1]).replace('\0', '1'),
72                         new String(new char[maxUtf8PropertyStringLength]).replace('\0', '2'),
73                         new String(new char[maxUtf8PropertyStringLength + 1]).replace('\0', '1'),
74                         new String(new char[maxUtf8PropertyStringLength + 100]).replace('\0', '3')
75         };
76         for (String name : names) {
77             for (String value : values) {
78                 props.put(name, value);
79             }
80         }
81 
82         byte[] data = (byte[]) serializeProperties.invoke(null, props);
83 
84         Map<String, String> newProps = (Map<String, String>) deserializeProperties.invoke(null, data);
85 
86         Assert.assertEquals(props.size(), newProps.size());
87         for (String name : props.keySet()) {
88             String expect = props.get(name);
89             String actual = newProps.get(name);
90             Assert.assertEquals(expect, actual);
91         }
92     }
93 }
