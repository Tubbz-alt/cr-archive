 43 package compiler.jvmci.compilerToVM;
 44 
 45 import jdk.test.lib.Asserts;
 46 import jdk.vm.ci.runtime.JVMCI;
 47 
 48 import java.lang.reflect.Method;
 49 
 50 public class JVM_RegisterJVMCINatives {
 51     private static final boolean IS_POSITIVE = Boolean.getBoolean(
 52             "compiler.jvmci.compilerToVM.JVM_RegisterJVMCINatives.positive");
 53 
 54     private final Method registerNatives;
 55 
 56     public static void main(String[] args) {
 57         new JVM_RegisterJVMCINatives().runTest();
 58     }
 59 
 60     private void runTest() {
 61         Object result;
 62         try {
 63             result = invoke();
 64         } catch (InternalError e) {
 65             if (IS_POSITIVE) {
 66                 throw new AssertionError("unexpected exception", e);
 67             }
 68             return;
 69         }
 70         if (!IS_POSITIVE) {
 71             throw new AssertionError("didn't get expected exception");
 72         }
 73         Asserts.assertNull(result,
 74                 "registerNatives()V returned non-null");
 75         Asserts.assertEQ(result, invoke(),
 76                 "registerNatives returns different results");
 77 
 78     }
 79     private Object invoke() {
 80         Object result;
 81         try {
 82             result = registerNatives.invoke(JVMCI.class);
 83         } catch (ReflectiveOperationException e) {
 84             throw new Error("can't invoke registerNatives", e);
 85         }
 86         return result;
 87     }
 88 
 89     private JVM_RegisterJVMCINatives() {
 90         Method method;
 91         try {
 92             method = Class.forName("jdk.vm.ci.hotspot.CompilerToVM",
 93                     /* initialize = */ false,
 94                     this.getClass().getClassLoader())
 95                     .getDeclaredMethod("registerNatives");
 96             method.setAccessible(true);
 97         } catch (ReflectiveOperationException e) {
 98             throw new Error("can't find CompilerToVM::registerNatives", e);
 99         }
100         registerNatives = method;
101     }
102 }
