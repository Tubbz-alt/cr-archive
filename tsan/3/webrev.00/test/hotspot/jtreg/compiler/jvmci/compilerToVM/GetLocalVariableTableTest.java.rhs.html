 49 package compiler.jvmci.compilerToVM;
 50 
 51 import compiler.jvmci.common.CTVMUtilities;
 52 import jdk.test.lib.Asserts;
 53 import jdk.vm.ci.hotspot.CompilerToVMHelper;
 54 import jdk.vm.ci.hotspot.HotSpotResolvedJavaMethod;
 55 
 56 import java.lang.reflect.Executable;
 57 import java.util.HashMap;
 58 import java.util.Map;
 59 
 60 public class GetLocalVariableTableTest {
 61 
 62     public static final int MAIN_LOCALS_COUNT = 0;
 63     public static final int INSTANCE_LOCALS_COUNT = 4;
 64     public static final int EMPTY_INSTANCE_COUNT = 1;
 65     public static final int EMPTY_STATIC_COUNT = 0;
 66     public static final int ABSTRACT_INHERIT_LOCALS_COUNT = 2;
 67     public static final int DEFAULTFUNC_LOCALS_COUNT = 4;
 68 
 69     public static void main(String[] args) {
 70         Map<Executable, Integer> testCases = createTestCases();
 71         testCases.forEach(GetLocalVariableTableTest::runSanityTest);
 72     }
 73 
 74     private static Map<Executable, Integer> createTestCases() {
 75         HashMap<Executable, Integer> methods = new HashMap<>();
 76         try {
 77             Class<?> aClass;
 78 
 79             aClass = GetLocalVariableTableTest.class;
 80             methods.put(aClass.getDeclaredMethod("main", String[].class),
 81                     MAIN_LOCALS_COUNT);
 82 
 83             aClass = DummyClass.class;
 84             methods.put(aClass.getMethod("dummyInstanceFunction"),
 85                     INSTANCE_LOCALS_COUNT);
 86             methods.put(aClass.getMethod("dummyEmptyInstanceFunction"),
 87                     EMPTY_INSTANCE_COUNT);
 88             methods.put(aClass.getMethod("dummyEmptyStaticFunction"),
 89                     EMPTY_STATIC_COUNT);
 90             methods.put(aClass.getMethod("dummyFunction"),
 91                     EMPTY_INSTANCE_COUNT);
 92             methods.put(aClass.getMethod("dummyAbstractFunction"),
 93                     ABSTRACT_INHERIT_LOCALS_COUNT);
 94 
 95             aClass = DummyInterface.class;
 96             methods.put(aClass.getMethod("dummyFunction"), EMPTY_STATIC_COUNT);
 97             methods.put(aClass.getMethod("dummyDefaultFunction", int.class,
 98                     int.class), DEFAULTFUNC_LOCALS_COUNT);
 99 
100             aClass = DummyAbstractClass.class;
101             methods.put(aClass.getMethod("dummyAbstractFunction"), 0);
102         } catch (NoSuchMethodException e) {
103             throw new Error("TEST BUG", e);
104         }
105         return methods;
106     }
107 
108     private static void runSanityTest(Executable aMethod,
109                                       Integer expectedTableLength) {
110         HotSpotResolvedJavaMethod method = CTVMUtilities
111                 .getResolvedMethod(aMethod);
112 
113         int tblLength = CompilerToVMHelper.getLocalVariableTableLength(method);
114         Asserts.assertEQ(tblLength, expectedTableLength, aMethod + " : incorrect "
115                 + "local variable table length.");
116 
117         long tblStart = CompilerToVMHelper.getLocalVariableTableStart(method);
118         if (tblLength > 0) {
119             Asserts.assertNE(tblStart, 0L, aMethod + " : local variable table starts"
120                     + " at 0 with length " + tblLength);
121         }
122     }
123 }
