39 package jdk.vm.ci.code.test;
40 
41 import jdk.vm.ci.code.Register;
42 import org.junit.Test;
43 
44 /**
45  * Test simple code installation.
46  */
47 public class SimpleCodeInstallationTest extends CodeInstallationTest {
48 
49     public static int add(int a, int b) {
50         return a + b;
51     }
52 
53     private static void compileAdd(TestAssembler asm) {
54         Register arg0 = asm.emitIntArg0();
55         Register arg1 = asm.emitIntArg1();
56         Register ret = asm.emitIntAdd(arg0, arg1);
57         asm.emitIntRet(ret);
58     }
59 
60     @Test
61     public void test() {
62         test(SimpleCodeInstallationTest::compileAdd, getMethod("add", int.class, int.class), 5, 7);
63     }
64 }
