 45 package compiler.jvmci.compilerToVM;
 46 
 47 import compiler.jvmci.common.CTVMUtilities;
 48 import jdk.test.lib.Asserts;
 49 import jdk.vm.ci.hotspot.CompilerToVMHelper;
 50 import jdk.vm.ci.hotspot.HotSpotResolvedJavaMethod;
 51 
 52 import java.lang.reflect.Executable;
 53 import java.lang.reflect.Method;
 54 import java.util.HashMap;
 55 import java.util.Map;
 56 
 57 public class MethodIsIgnoredBySecurityStackWalkTest {
 58 
 59     public static void main(String[] args) {
 60         Map<Executable, Boolean> testCases = createTestCases();
 61         testCases.forEach(
 62                 MethodIsIgnoredBySecurityStackWalkTest::runSanityTest);
 63     }
 64 
 65     private static void runSanityTest(Executable aMethod, Boolean expected) {
 66         HotSpotResolvedJavaMethod method
 67                 = CTVMUtilities.getResolvedMethod(aMethod);
 68         boolean isIgnored = CompilerToVMHelper
 69                 .methodIsIgnoredBySecurityStackWalk(method);
 70         String msg = String.format("%s is%s ignored but must%s", aMethod,
 71                 isIgnored ? "" : " not",
 72                 expected ? "" : " not");
 73         Asserts.assertEQ(isIgnored, expected, msg);
 74     }
 75 
 76     private static Map<Executable, Boolean> createTestCases() {
 77         Map<Executable, Boolean> testCases = new HashMap<>();
 78 
 79         try {
 80             Class<?> aClass = Method.class;
 81             testCases.put(aClass.getMethod("invoke", Object.class,
 82                     Object[].class), true);
 83 
 84             aClass = Class.forName("jdk.internal.reflect.NativeMethodAccessorImpl");
 85             testCases.put(aClass.getMethod("invoke", Object.class,
 86                     Object[].class), true);
 87             testCases.put(aClass.getDeclaredMethod("invoke0", Method.class,
 88                     Object.class, Object[].class), true);
 89 
 90             aClass = MethodIsIgnoredBySecurityStackWalkTest.class;
 91             for (Executable method : aClass.getMethods()) {
 92                 testCases.put(method, false);
 93             }
 94             for (Executable method : aClass.getDeclaredMethods()) {
 95                 testCases.put(method, false);
 96             }
 97             for (Executable method : aClass.getConstructors()) {
 98                 testCases.put(method, false);
 99             }
100         } catch (NoSuchMethodException | ClassNotFoundException e) {
101             throw new Error("TEST BUG " + e.getMessage(), e);
102         }
103         return testCases;
104     }
105 }
