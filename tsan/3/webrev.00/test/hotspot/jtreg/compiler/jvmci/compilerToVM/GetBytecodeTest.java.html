 44 package compiler.jvmci.compilerToVM;
 45 
 46 import compiler.jvmci.common.CTVMUtilities;
 47 import compiler.jvmci.common.testcases.TestCase;
 48 import jdk.internal.org.objectweb.asm.Opcodes;
 49 import jdk.test.lib.Asserts;
 50 import jdk.vm.ci.hotspot.CompilerToVMHelper;
 51 import jdk.vm.ci.hotspot.HotSpotResolvedJavaMethod;
 52 
 53 import java.lang.reflect.Executable;
 54 import java.lang.reflect.Modifier;
 55 
 56 public class GetBytecodeTest {
 57 
 58     public static void main(String[] args) {
 59         TestCase.getAllExecutables()
 60                 .forEach(GetBytecodeTest::runSanityTest);
 61     }
 62 
 63     private static void runSanityTest(Executable aMethod) {
 64         HotSpotResolvedJavaMethod method = CTVMUtilities
 65                 .getResolvedMethod(aMethod);
 66         byte[] bytecode = CompilerToVMHelper.getBytecode(method);
 67 
 68         int mods = aMethod.getModifiers();
 69         boolean shouldHasZeroLength = Modifier.isAbstract(mods)
 70                 || Modifier.isNative(mods);
 71         boolean correctLength = (bytecode.length == 0 && shouldHasZeroLength)
 72                 || (bytecode.length > 0 && !shouldHasZeroLength);
 73 
 74         Asserts.assertTrue(correctLength, "Bytecode of '" + aMethod + "' has "
 75                 + bytecode.length + " length");
 76 
 77         if (!shouldHasZeroLength) {
 78             Asserts.assertTrue(containsReturn(bytecode), "Bytecode of '"
 79                     + aMethod + "' doesn't have any return statement");
 80         }
 81     }
 82 
 83     private static boolean containsReturn(byte[] bytecode) {
 84         for (byte b : bytecode) {
 85             //  cast unsigned byte to int
 86             int value = (int) b & 0x000000FF;
 87             switch (value) {
 88                 case Opcodes.RET:
 89                 case Opcodes.ARETURN:
 90                 case Opcodes.IRETURN:
 91                 case Opcodes.LRETURN:
 92                 case Opcodes.FRETURN:
 93                 case Opcodes.DRETURN:
 94                 case Opcodes.RETURN:
 95                     return true;
 96             }
 97         }
 98         return false;
 99     }
100 }
