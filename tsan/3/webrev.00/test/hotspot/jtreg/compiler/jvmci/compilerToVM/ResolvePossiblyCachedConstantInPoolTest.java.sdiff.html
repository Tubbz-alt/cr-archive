 65 public class ResolvePossiblyCachedConstantInPoolTest {
 66 
 67     public static void main(String[] args) throws Exception {
 68         Map<ConstantTypes, Validator> typeTests = new HashMap<>();
 69         typeTests.put(CONSTANT_STRING, ResolvePossiblyCachedConstantInPoolTest::validateString);


 70         ConstantPoolTestCase testCase = new ConstantPoolTestCase(typeTests);
 71         // The next "Class.forName" is here for the following reason.
 72         // When class is initialized, constant pool cache is available.
 73         // This method works only with cached constant pool.





 74         for (DummyClasses dummy : DummyClasses.values()) {
 75             Class.forName(dummy.klass.getName());
 76         }
 77         testCase.test();
 78     }
 79 
 80     private static void validateString(ConstantPool constantPoolCTVM,
 81                                        ConstantTypes cpType,
 82                                        DummyClasses dummyClass,
 83                                        int cpi) {
 84         TestedCPEntry entry = cpType.getTestedCPEntry(dummyClass, cpi);
 85         if (entry == null) {
 86             return;
 87         }
 88         int index = cpi;
 89         String cached = "";
 90         int cpci = dummyClass.getCPCacheIndex(cpi);
 91         if (cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
 92             index = cpci;
 93             cached = "cached ";
 94         }
 95         Object constantInPool = CompilerToVMHelper.resolvePossiblyCachedConstantInPool(constantPoolCTVM, index);
 96         String stringToVerify = (String) constantInPool;
 97         String stringToRefer = entry.name;
 98         if (stringToRefer.equals("") && cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
 99             stringToRefer = null; // tested method returns null for cached empty strings
100         }
101         String msg = String.format("Wrong string accessed by %sconstant pool index %d", cached, index);
102         Asserts.assertEQ(stringToRefer, stringToVerify, msg);
103     }

 67 /**
 68  * Test for {@code jdk.vm.ci.hotspot.CompilerToVM.resolvePossiblyCachedConstantInPool} method
 69  */
 70 public class ResolvePossiblyCachedConstantInPoolTest {
 71 
 72     public static void main(String[] args) throws Exception {
 73         Map<ConstantTypes, Validator> typeTests = new HashMap<>();
 74         typeTests.put(CONSTANT_STRING, ResolvePossiblyCachedConstantInPoolTest::validateString);
 75         typeTests.put(CONSTANT_METHODHANDLE, ResolvePossiblyCachedConstantInPoolTest::validateMethodHandle);
 76         typeTests.put(CONSTANT_METHODTYPE, ResolvePossiblyCachedConstantInPoolTest::validateMethodType);
 77         ConstantPoolTestCase testCase = new ConstantPoolTestCase(typeTests);
 78         testCase.test();
 79         // The next "Class.forName" and repeating "testCase.test()"
 80         // are here for the following reason.
 81         // The first test run is without dummy class initialization,
 82         // which means no constant pool cache exists.
 83         // The second run is with initialized class (with constant pool cache available).
 84         // Some CompilerToVM methods require different input
 85         // depending on whether CP cache exists or not.
 86         for (DummyClasses dummy : DummyClasses.values()) {
 87             Class.forName(dummy.klass.getName());
 88         }
 89         testCase.test();
 90     }
 91 
 92     private static void validateString(ConstantPool constantPoolCTVM,
 93                                        ConstantTypes cpType,
 94                                        DummyClasses dummyClass,
 95                                        int cpi) {
 96         TestedCPEntry entry = cpType.getTestedCPEntry(dummyClass, cpi);
 97         if (entry == null) {
 98             return;
 99         }
100         int index = cpi;
101         String cached = "";
102         int cpci = dummyClass.getCPCacheIndex(cpi);
103         if (cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
104             index = cpci;
105             cached = "cached ";
106         }
107         Object constantInPool = CompilerToVMHelper.resolvePossiblyCachedConstantInPool(constantPoolCTVM, index);
108         String stringToVerify = (String) constantInPool;
109         String stringToRefer = entry.name;
110         if (stringToRefer.equals("") && cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
111             stringToRefer = null; // tested method returns null for cached empty strings
112         }
113         String msg = String.format("Wrong string accessed by %sconstant pool index %d", cached, index);
114         Asserts.assertEQ(stringToRefer, stringToVerify, msg);
115     }
116 
117     private static final String NOT_NULL_MSG
118             = "Object returned by resolvePossiblyCachedConstantInPool method should not be null";
119 
120 
121     private static void validateMethodHandle(ConstantPool constantPoolCTVM,
122                                              ConstantTypes cpType,
123                                              DummyClasses dummyClass,
124                                              int index) {
125         Object constantInPool = CompilerToVMHelper.resolvePossiblyCachedConstantInPool(constantPoolCTVM, index);
126         String msg = String.format("%s for index %d", NOT_NULL_MSG, index);
127         Asserts.assertNotNull(constantInPool, msg);
128         if (!(constantInPool instanceof MethodHandle)) {
129             msg = String.format("Wrong constant pool entry accessed by index"
130                                         + " %d: %s, but should be subclass of %s",
131                                 index,
132                                 constantInPool.getClass(),
133                                 MethodHandle.class.getName());
134             throw new AssertionError(msg);
135         }
136     }
137 
138     private static void validateMethodType(ConstantPool constantPoolCTVM,
139                                            ConstantTypes cpType,
140                                            DummyClasses dummyClass,
141                                            int index) {
142         Object constantInPool = CompilerToVMHelper.resolvePossiblyCachedConstantInPool(constantPoolCTVM, index);
143         String msg = String.format("%s for index %d", NOT_NULL_MSG, index);
144         Asserts.assertNotNull(constantInPool, msg);
145         Class mtToVerify = constantInPool.getClass();
146         Class mtToRefer = MethodType.class;
147         msg = String.format("Wrong method type class accessed by"
148                                     + " constant pool index %d",
149                             index);
150         Asserts.assertEQ(mtToRefer, mtToVerify, msg);
151     }
