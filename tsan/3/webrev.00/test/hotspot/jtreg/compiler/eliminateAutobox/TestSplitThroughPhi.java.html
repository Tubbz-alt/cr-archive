24 /*
25  * @test
26  * @bug 8224957
27  * @summary Test splitting a boxed field load through a loop Phi.
28  * @run main/othervm -XX:-TieredCompilation -XX:-ProfileInterpreter
29  *                   -XX:CompileOnly=compiler.eliminateAutobox.TestSplitThroughPhi::test
30  *                   compiler.eliminateAutobox.TestSplitThroughPhi
31  */
32 
33 package compiler.eliminateAutobox;
34 
35 public class TestSplitThroughPhi {
36 
37   static volatile boolean loop = true;
38 
39   public static void test(Object obj) {
40       for (int i = 0; i == 0; ) {
41           // OSR compilation generates Region with 3 input paths
42           while (true) {
43               // Load from boxed int
44               int val = (Integer)obj;
45               if (val == i++) break;
46               if (!loop) break;
47           }
48       }
49   }
50 
51   public static void main(String[] args) {
52       test(100_000);
53   }
54 }
