24 /*
25  * @test
26  * @bug 4093292
27  * @summary Test for correct code generation by the JIT
28  * @run main compiler.codegen.ShiftTest
29  */
30 
31 package compiler.codegen;
32 
33 public class ShiftTest {
34     static final int w = 32;
35 
36     private static void doTest(long ct) throws Exception {
37         int S22 = 0xc46cf7c2;
38         int S23 = 0xcfda9162;
39         int S24 = 0xd029aa4c;
40         int S25 = 0x17cf1801;
41         int A = (int)(ct & 0xffffffffL);
42         int B = (int)(ct >>> 32);
43         int x, y;
44         x = B - S25;
45         y = A & (w-1);
46         B = ((x >>> y) | (x << (w-y))) ^ A;
47         x = A - S24;
48         y = B & (w-1);
49         A = ((x >>> y) | (x << (w-y))) ^ B;
50         x = B - S23;
51         y = A & (w-1);
52         B = ((x >>> y) | (x << (w-y))) ^ A;
53         x = A - S22;
54         y = B & (w-1);
55         A = ((x >>> y) | (x << (w-y))) ^ B;
56         String astr = Integer.toHexString(A);
57         String bstr = Integer.toHexString(B);
58         System.err.println("A = " + astr + " B = " + bstr);
59         if ((!astr.equals("dcb38144")) ||
60             (!bstr.equals("1916de73"))) {
61             throw new RuntimeException("Unexpected shift results!");
62         }
63         System.err.println("Test passed");
64     }
65 
66     public static void main(String[] args) throws Exception {
67         doTest(0x496def29b74be041L);
68     }
69 }
