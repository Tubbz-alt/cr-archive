24 /**
25  * @test
26  * @bug 8221083
27  * @requires vm.gc.Serial
28  * @requires vm.bits == 64 & vm.opt.final.UseCompressedOops == true
29  * @summary On ppc64, C1 erroneously emits a 32-bit compare instruction for oop compares.
30  * @modules java.base/jdk.internal.misc:+open
31  * @library /test/lib /
32  * @build sun.hotspot.WhiteBox
33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
34  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
35  * @run main/othervm -Xbatch -XX:-UseTLAB -Xmx4m -XX:+UseSerialGC -XX:HeapBaseMinAddress=0x700000000
36  *      -XX:CompileCommand=compileonly,compiler.codegen.TestOopCmp::nullTest
37  *      -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
38  *      compiler.codegen.TestOopCmp
39  * @author volker.simonis@gmail.com
40  */
41 
42 package compiler.codegen;
43 
44 import sun.hotspot.WhiteBox;
45 
46 public class TestOopCmp {
47 
48     private static Object nullObj = null;
49 
50     public static boolean nullTest(Object o) {
51         if (o == nullObj) {
52             return true;
53         } else {
54             return false;
55         }
56     }
57 
58     public static void main(String args[]) {
59 
60         WhiteBox WB = WhiteBox.getWhiteBox();
61 
62         // The test is started with -XX:HeapBaseMinAddress=0x700000000 and a
63         // small heap of only 4mb. This works pretty reliable and at least on
64         // Linux/Windows/Solaris we will get a heap starting at 0x700000000.
65         // The test also runs with -XX:+UseSerialGC which means that we'll get
66         // eden starting at 0x700000000.
67         // Calling 'System.gc()' will clean up all the objects from eden, so if
68         // eden starts at 0x700000000 the first allocation right after the
69         // system GC will be allcoated right at address 0x700000000.
70         System.gc();
71         String s = new String("I'm not null!!!");
72         if (WB.getObjectAddress(s) == 0x700000000L) {
73             System.out.println("Got object at address 0x700000000");
74         }
75 
76         // We call 'nullTest()' with the newly allocated String object. If it was
77         // allocated at 0x700000000, its 32 least-significant bits will be 0 and a
78         // 32-bit comparison with 'nullObj' (which is 'null') will yield true and
79         // result in a test failure.
80         // If the code generated for 'nullTest()' correctly performs a 64-bit
81         // comparison or if we didn't manage to allcoate 's' at 0x700000000 the
82         // test will always succeed.
83         for (int i = 0; i < 30_000; i++) {
84             if (nullTest(s)) {
85                 throw new RuntimeException("Comparing non-null object with null returned 'true'");
86             }
87         }
88     }
89 }
