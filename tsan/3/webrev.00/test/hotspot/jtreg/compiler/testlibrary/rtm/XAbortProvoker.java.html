24 package compiler.testlibrary.rtm;
25 
26 /**
27  * Current RTM locking implementation force transaction abort
28  * before native method call by explicit xabort(0) call.
29  */
30 public class XAbortProvoker extends AbortProvoker {
31 
32     static {
33         try {
34             System.loadLibrary("XAbortProvoker");
35         } catch (UnsatisfiedLinkError e) {
36             System.out.println("Could not load native library: " + e);
37         }
38     }
39 
40     public native int doAbort();
41 
42     // Following field have to be static in order to avoid escape analysis.
43     @SuppressWarnings("UnsuedDeclaration")
44     private static int field = 0;
45 
46     public XAbortProvoker() {
47         this(new Object());
48     }
49 
50     public XAbortProvoker(Object monitor) {
51         super(monitor);
52     }
53 
54     @Override
55     public void forceAbort() {
56         synchronized(monitor) {
57             XAbortProvoker.field = doAbort();
58         }
59     }
60 
61     @Override
62     public String[] getMethodsToCompileNames() {
63         return new String[] {
64                 getMethodWithLockName(),
65                 XAbortProvoker.class.getName() + "::doAbort"
66         };
67     }
68 }
