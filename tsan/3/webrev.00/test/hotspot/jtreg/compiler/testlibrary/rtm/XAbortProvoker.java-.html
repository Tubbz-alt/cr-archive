24 package compiler.testlibrary.rtm;
25 
26 import jdk.internal.misc.Unsafe;
27 
28 /**
29  * Current RTM locking implementation force transaction abort
30  * before native method call by explicit xabort(0) call.
31  */
32 class XAbortProvoker extends AbortProvoker {
33     // Following field have to be static in order to avoid escape analysis.
34     @SuppressWarnings("UnsuedDeclaration")
35     private static int field = 0;
36     private static final Unsafe UNSAFE = Unsafe.getUnsafe();
37 
38     public XAbortProvoker() {
39         this(new Object());
40     }
41 
42     public XAbortProvoker(Object monitor) {
43         super(monitor);
44     }
45 
46     @Override
47     public void forceAbort() {
48         synchronized(monitor) {
49             XAbortProvoker.field = UNSAFE.pageSize();
50         }
51     }
52 
53     @Override
54     public String[] getMethodsToCompileNames() {
55         return new String[] {
56                 getMethodWithLockName(),
57                 Unsafe.class.getName() + "::pageSize"
58         };
59     }
60 }
