25 /*
26  * @test
27  * @bug 8237524
28  * @summary String.compareTo() may return incorrect result in this case
29  *
30  * @run main/othervm compiler.intrinsics.Test8237524
31  */
32 
33 package compiler.intrinsics;
34 
35 import java.lang.reflect.Constructor;
36 
37 public class Test8237524 {
38 
39   private static int stringCompareTo(String s1, String s2) {
40     return s1.compareTo(s2);
41   }
42 
43   public static void main(String[] args) throws Exception {
44     Constructor<String> c = String.class.getDeclaredConstructor(byte[].class, byte.class);
45     c.setAccessible(true);
46 
47     byte[] bytes = new byte[] {
48       'Y', 'm', '_', 'l', 'V', 'n', 'W', 'S', 'w', 'm', 'W', 'S'
49     };
50 
51     String s1 = c.newInstance(bytes, (byte) 0); // a Latin string
52     String s2 = c.newInstance(bytes, (byte) 1); // a Chinese string
53 
54     for (int i = 0; i < 50000; i++) {
55       if (stringCompareTo(s1, s2) >= 0) {
56         System.out.println("FAIL. s1 should be less than s2 according to Java API Spec");
57         System.exit(1);
58       }
59     }
60   }
61 }
