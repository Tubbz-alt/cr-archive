24 /**
25  * @test
26  * @bug 8229855
27  * @summary Test jump table with key value that gets out of bounds after loop unrolling.
28  * @run main/othervm -XX:CompileCommand=dontinline,compiler.c2.TestJumpTable::test*
29  *                   -Xbatch -XX:+UnlockDiagnosticVMOptions -XX:-TieredCompilation -XX:-UseSwitchProfiling
30  *                   compiler.c2.TestJumpTable
31  */
32 
33 package compiler.c2;
34 
35 public class TestJumpTable {
36 
37     public static int test() {
38         int res = 0;
39         for (int i = 10; i < 50; ++i) {
40             switch (i * 5) {
41                 case 15:
42                 case 25:
43                 case 40:
44                 case 101:
45                     return 42;
46                 case 45:
47                 case 51:
48                 case 60:
49                     res++;
50                     break;
51             }
52         }
53         return res;
54     }
55 
56     public static void main(String[] args) {
57         for (int i = 0; i < 20_000; ++i) {
58             test();
59         }
60     }
61 }
