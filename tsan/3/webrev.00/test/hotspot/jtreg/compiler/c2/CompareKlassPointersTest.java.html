24 /*
25  * @test
26  * @bug 8220416
27  * @summary This test uses a class pointer comparison optimization which was not applied anymore since JDK-6964458 resulting in a better IR.
28  *
29  * @run main/othervm -Xbatch -XX:-TieredCompilation -XX:+IgnoreUnrecognizedVMOptions -XX:+AlwaysIncrementalInline
30  *                   -XX:CompileOnly=compiler.c2.CompareKlassPointersTest::test compiler.c2.CompareKlassPointersTest
31  */
32 
33 package compiler.c2;
34 
35 
36 
37 public class CompareKlassPointersTest {
38 
39     static A a;
40 
41     public static void main(String[] args) {
42         a = new C();
43         for (int i = 0; i < 10_000; i++) {
44             test();
45         }
46     }
47 
48     public static int test() {
49         Object a = getA();
50 
51         /*
52          * This check is now optimized away which was not the case before anymore since JDK-6964458:
53          * The klass pointer comparison optimization sees that the check is always false since a and B are always unrelated klasses
54          */
55         if (a instanceof B) {
56             return 1;
57         }
58         return 0;
59     }
60 
61     private static Object getA() {
62         return a;
63     }
64 }
65 
66 class A { }
67 
68 class B { }
69 
70 class C extends A { }
