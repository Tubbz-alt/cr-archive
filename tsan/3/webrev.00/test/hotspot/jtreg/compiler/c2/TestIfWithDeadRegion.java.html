33 package compiler.c2;
34 
35 import java.util.function.Supplier;
36 
37 public class TestIfWithDeadRegion {
38 
39     static String msg;
40 
41     static String getString(String s, int i) {
42         String current = s + String.valueOf(i);
43         System.nanoTime();
44         return current;
45     }
46 
47     static void test(Supplier<String> supplier) {
48         msg = supplier.get();
49     }
50 
51     public static void main(String[] args) {
52         for (int i = 0; i < 20_000; ++i) {
53             test(() -> getString("Test1", 42));
54             test(() -> getString("Test2", 42));
55         }
56     }
57 }
