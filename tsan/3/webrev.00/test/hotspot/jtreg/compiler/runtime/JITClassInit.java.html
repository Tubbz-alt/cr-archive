41 
42 class TokenTable {
43     public TokenTable() {
44         new TokenTypeIterator(this);
45     }
46 
47     public void for_token_type(Token t) {
48         t.keyword_character_class();
49     }
50 }
51 
52 class Token {
53     public Object keyword_character_class() {
54         return new Object();
55     }
56 }
57 
58 class NameOrKeywordToken extends Token {
59     static TokenTable kt = new TokenTable();
60     public Object keyword_character_class() {
61         return new Object();
62     }
63 }
64 
65 class CapKeywordToken extends NameOrKeywordToken {
66     public Object keyword_character_class() {
67         return new Object();
68     }
69 };
70 
71 
72 class TokenTypeIterator {
73     public TokenTypeIterator(TokenTable c) {
74         c.for_token_type(new CapKeywordToken());
75         c.for_token_type(new NameOrKeywordToken());
76     }
77 }
