25 /*
26  * @test
27  * @bug 8231223
28  * @summary Test conditional move optimization encountering an Opaque4Node.
29  * @run main/othervm -Xbatch -XX:-TieredCompilation
30  *                   -XX:CompileCommand=inline,compiler.loopopts.TestCMovWithOpaque::test
31  *                   compiler.loopopts.TestCMovWithOpaque
32  */
33 
34 package compiler.loopopts;
35 
36 public class TestCMovWithOpaque {
37 
38     public static void test(int array[]) {
39         for (int i = 1; i < 8; i += 3) {
40             for (int j = 0; j < 4; ++j) {
41                 switch (i % 4) {
42                 case 0:
43                     break;
44                 case 1:
45                     break;
46                 case 2:
47                     break;
48                 case 3:
49                     array[j] += 42;
50                     break;
51                 }
52             }
53         }
54     }
55 
56     public static void main(String[] args) {
57         int[] array = new int[4];
58         for (int i = 0; i < 20_000; ++i) {
59             test(array);
60         }
61     }
62 }
