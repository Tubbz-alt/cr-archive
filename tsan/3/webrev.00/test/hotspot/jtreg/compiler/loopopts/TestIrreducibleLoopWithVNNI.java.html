35 package compiler.loopopts;
36 
37 public class TestIrreducibleLoopWithVNNI {
38     public static boolean condition = false;
39 
40     public static void main(String[] strArr) {
41         TestIrreducibleLoopWithVNNI _instance = new TestIrreducibleLoopWithVNNI();
42         for (int i = 0; i < 10; i++) {
43             _instance.mainTest();
44         }
45     }
46 
47     public void mainTest() {
48         int a = 1, b = 1, c = 1, d = 51;
49         for (b = 0; b < 100; b++) {
50             a = a + b * 342;
51             for (c = 0; c < 100; c++) {
52                 for (d = 0; d < 1; d++)
53                     a = d;
54                 if (condition)
55                     break;
56             }
57             a = d * a;
58         }
59     }
