 34 package compiler.c2;
 35 
 36 public class TestFoldNECompares {
 37 
 38     public static int getNarrowInt(boolean b, int lo, int hi) {
 39         return b ? lo : hi;
 40     }
 41 
 42     public static void test1(boolean b) {
 43         int i = getNarrowInt(b, 42, 142);
 44         // i: 42..142
 45         if (i != 42) {
 46             // i: 43..142
 47             if (i <= 42) {
 48                 throw new RuntimeException("Should not reach here");
 49             }
 50         }
 51     }
 52
