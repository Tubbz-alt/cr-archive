 23 package com.oracle.mxtool.junit;
 24 
 25 import java.io.PrintStream;
 26 
 27 import org.junit.internal.JUnitSystem;
 28 import org.junit.internal.TextListener;
 29 import org.junit.runner.Description;
 30 import org.junit.runner.Result;
 31 import org.junit.runner.notification.Failure;
 32 import org.junit.runner.notification.RunListener;
 33 
 34 class TextRunListener implements MxRunListener {
 35 
 36     private final PrintStream fWriter;
 37     protected Failure lastFailure;
 38 
 39     TextRunListener(JUnitSystem system) {
 40         this(system.out());
 41     }
 42 
 43     TextRunListener(PrintStream writer) {
 44         fWriter = writer;
 45     }
 46 
 47     public boolean beVerbose() {
 48         return false;
 49     }
 50 
 51     @Override
 52     public PrintStream getWriter() {
 53         return fWriter;
 54     }
 55 
 56     public Failure getLastFailure() {
 57         return lastFailure;
 58     }
 59 
 60     @Override
 61     public void testStarted(Description description) {
 62         getWriter().print('.');
 63     }
 64 
 65     @Override
 66     public void testFailed(Failure failure) {
 67         getWriter().print('E');
 68         lastFailure = failure;
 69     }
 70 
 71     @Override
 72     public void testIgnored(Description description) {
 73         getWriter().print('I');
 74     }
 75 
 76     public static RunListener createRunListener(MxRunListener l) {
 77         PrintStream theWriter = l.getWriter();
 78         return new TextListener(theWriter) {
 79             private Class<?> lastClass;
 80             private int passedInLastClass;
 81             private int failedInLastClass;
 82             private boolean failed;
 83 
 84             @Override
 85             public void testStarted(Description description) {
 86                 Class<?> currentClass = description.getTestClass();
 87                 if (currentClass != lastClass) {
 88                     if (lastClass != null) {
 89                         l.testClassFinished(lastClass, passedInLastClass, failedInLastClass);
 90                         l.testClassFinishedDelimiter();
 91                     }
 92                     lastClass = currentClass;
 93                     passedInLastClass = 0;
 94                     failedInLastClass = 0;
 95                     l.testClassStarted(currentClass);
 96                     l.testClassStartedDelimiter();
 97                 }
 98                 failed = false;
 99                 l.testStarted(description);
100                 l.testStartedDelimiter();
101             }
102 
103             @Override
104             public void testFailure(Failure failure) {
105                 failed = true;
106                 failedInLastClass++;
107                 l.testFailed(failure);
108             }
109 
110             @Override
111             public void testFinished(Description description) {
112                 // we have to do this because there is no callback for successful tests
113                 if (!failed) {
114                     l.testSucceeded(description);
115                     passedInLastClass++;
116                 }
117                 l.testFinished(description);
118                 l.testFinishedDelimiter();
119             }
120 
121             @Override
122             public void testIgnored(Description description) {
123                 testStarted(description);
124                 l.testIgnored(description);
125                 l.testFinished(description);
126                 l.testFinishedDelimiter();
127             }
128 
129             @Override
130             public void testRunStarted(Description description) {
131                 l.testRunStarted(description);
132             }
133 
134             @Override
135             public void testRunFinished(Result result) {
136                 if (lastClass != null) {
137                     l.testClassFinished(lastClass, passedInLastClass, failedInLastClass);
138                 }
139                 l.testRunFinished(result);
140                 super.testRunFinished(result);
141             }
142 
143             @Override
144             public void testAssumptionFailure(Failure failure) {
145                 l.testAssumptionFailure(failure);
146             }
147 
148             @Override
149             protected void printFailure(Failure each, String prefix) {
150                 // Print out the test message in the same format used to run a single test:
151                 // my.package.MyClass#methodName
152                 String header = each.getDescription().getClassName() + "#" + each.getDescription().getMethodName();
153                 theWriter.println(prefix + ") " + header);
154                 theWriter.print(each.getTrace());
155             }
156         };
157     }
158 }
