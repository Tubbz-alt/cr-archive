25 class Test {
26 
27     static boolean loadLib(String libName){
28         try {
29             System.loadLibrary(libName);
30             System.out.println("Loaded library "+ libName + ".");
31             return true;
32         } catch (SecurityException e) {
33             System.out.println("loadLibrary(\"" + libName + "\") throws: " + e + "\n");
34         } catch (UnsatisfiedLinkError e) {
35             System.out.println("loadLibrary(\"" + libName + "\") throws: " + e + "\n");
36         }
37         return false;
38     }
39 
40     public static int counter = 1;
41 
42     static int Runner() {
43         counter = counter * -1;
44         int i = counter;
45         if (counter < 2) counter += Runner();
46         return i;
47     }
48 
49     public static int run() {
50         try{
51             Runner();
52         } catch (StackOverflowError e) {
53             System.out.println("Caught stack overflow error.");
54             return 0;
55         } catch (OutOfMemoryError e) {
56             return 0;
57         }
58         return 2;
59     }
60 
61     public static void main(String argv[]) {
62         loadLib(argv[0]);
63         System.exit(run());
64     }
65 }
