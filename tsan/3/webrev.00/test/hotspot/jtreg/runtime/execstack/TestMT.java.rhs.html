25 class TestMT {
26 
27     static boolean loadLib(String libName) {
28         try {
29             System.loadLibrary(libName);
30             System.out.println("Loaded library "+ libName + ".");
31             return true;
32         } catch (SecurityException e) {
33             System.out.println("loadLibrary(\"" + libName + "\") throws: " + e + "\n");
34         } catch (UnsatisfiedLinkError e) {
35             System.out.println("loadLibrary(\"" + libName + "\") throws: " + e + "\n");
36         }
37         return false;
38     }
39 
40     public static int counter = 1;
41     static int Runner() {
42         counter = counter * -1;
43         int i = counter;
44         if (counter < 2) counter += Runner();
45         return i;
46     }
47 
48     public static int run(String msg) {
49         try {
50             Runner();
51         } catch (StackOverflowError e) {
52             System.out.println(msg + " caught stack overflow error.");
53             return 0;
54         } catch (OutOfMemoryError e) {
55             return 0;
56         }
57         return 2;
58     }
59 
60     public static void main(String argv[]) {
61         try {
62             for (int i = 0; i < 20; i++) {
63                 Thread t = new DoStackOverflow("SpawnedThread " + i);
64                 t.start();
65             }
66             run("Main thread");
67             loadLib("test-rwx");
68             run("Main thread");
69         } catch (Exception e) {
70             System.out.println(e);
71         }
72     }
73 
74     static class DoStackOverflow extends Thread {
75         public DoStackOverflow(String name) {
76             super(name);
77         }
78         public void run() {
79             for (int i = 0; i < 10; ++i) {
80                 TestMT.run(getName());
81                 Thread.yield();
82             }
83         }
84     }
85 }
