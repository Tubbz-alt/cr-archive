 24 /*
 25  * @test
 26  * @compile --enable-preview --source ${jdk.version} recordReflectionTest.java
 27  * @run main/othervm --enable-preview recordReflectionTest
 28  */
 29 
 30 
 31 import java.lang.reflect.Method;
 32 import java.lang.reflect.RecordComponent;
 33 
 34 public class recordReflectionTest {
 35 
 36     record recordNames(int intX, String stringY) {
 37         public RecordComponent[] getComponents() throws Throwable {
 38             return recordNames.class.getRecordComponents();
 39         }
 40     }
 41 
 42     class notRecord {
 43         public RecordComponent[] getComponents() throws Throwable {
 44             return notRecord.class.getRecordComponents();
 45         }
 46     }
 47 
 48     record recordGeneric<T>(T xx, String yy) {
 49         public RecordComponent[] getComponents() throws Throwable {
 50             return recordGeneric.class.getRecordComponents();
 51         }
 52     }
 53 
 54     record recordEmpty() {
 55         public RecordComponent[] getComponents() throws Throwable {
 56             return recordEmpty.class.getRecordComponents();
 57         }
 58     }
 59 
 60     notRecord nr = new notRecord();
 61 
 62     public static boolean hasComponent(RecordComponent[] components, String enclosingClass,
 63                                        String name, String type, String method, String genericSig) {
 64         for (int x = 0; x < components.length; x++) {
 65             RecordComponent component = components[x];
 66             if (component.getName() == name &&
 67                 component.getDeclaringRecord().toString().contains(enclosingClass) &&
 68                 component.getType().toString().contains(type) &&
 69                 component.getAccessor().toString().contains(method) &&
 70                 component.getGenericSignature() == genericSig) {
 71                 return true;
 72             }
 73         }
 74         return false;
 75     }
 76 
 77     public static void main(String... args) throws Throwable {
 78         recordNames rn = new recordNames(3, "abc");
 79         RecordComponent[] components = rn.getComponents();
 80         if (components.length != 2 ||
 81             !hasComponent(components, "recordNames", "intX", "int",
 82                           "int recordReflectionTest$recordNames.intX()", null) ||
 83             !hasComponent(components, "recordNames", "stringY", "java.lang.String",
 84                           "java.lang.String recordReflectionTest$recordNames.stringY()", null)) {
 85             throw new RuntimeException("Bad component accessors returned for recordNames");
 86         }
 87 
 88         recordReflectionTest rft = new recordReflectionTest();
 89         components = rft.nr.getComponents();
 90         if (components != null) {
 91             throw new RuntimeException("Non-null component accessors returned for notRecord");
 92         }
 93 
 94         recordGeneric rg = new recordGeneric(35, "abcd");
 95         components = rg.getComponents();
 96         if (components.length != 2 ||
 97             !hasComponent(components, "recordGeneric", "xx", "java.lang.Object",
 98                           "java.lang.Object recordReflectionTest$recordGeneric.xx()", "TT;") ||
 99             !hasComponent(components, "recordGeneric", "yy", "java.lang.String",
100                           "java.lang.String recordReflectionTest$recordGeneric.yy()", null)) {
101             throw new RuntimeException("Bad component accessors returned for recordGeneric");
102         }
103 
104         recordEmpty re = new recordEmpty();
105         components = re.getComponents();
106         if (components.length != 0) {
107             throw new RuntimeException("Non-empty component accessors returned for recordEmpty");
108         }
109     }
110 }
