24 /*
25  * @test
26  * @compile abstractRecord.jcod notFinalRecord.jcod oldRecordAttribute.jcod superNotJLRecord.jcod
27  * @compile shortRecordAttribute.jcod twoRecordAttributes.jcod badRecordAttribute.jcod
28  *
29  * @run main/othervm --enable-preview recordAttributeTest
30  */
31 
32 
33 import java.lang.reflect.Method;
34 
35 public class recordAttributeTest {
36 
37     public static void runTest(String className, String cfeMessage) {
38         try {
39             Class newClass = Class.forName(className);
40             throw new RuntimeException("Expected ClasFormatError exception not thrown");
41         } catch (java.lang.ClassFormatError e) {
42             String eMsg = e.getMessage();
43             if (!eMsg.contains(cfeMessage)) {
44                 throw new RuntimeException("Unexpected exception: " + eMsg);
45             }
46         } catch (java.lang.ClassNotFoundException f) {
47             throw new RuntimeException("Unexpected exception: " + f.getMessage());
48         }
49     }
50 
51     public static void main(String... args) throws Throwable {
52 
53         // Test loading a class with two Record attributes. This should throw ClassFormatError.
54         runTest("twoRecordAttributes",
55                 "Multiple Record attributes in class");
56 
57         // Test loading a Record type marked abstract. This should throw ClassFormatError.
58         runTest("abstractRecord",
59                 "Record attribute in non-final or abstract class");
60 
61         // Test loading a Record type that is not final. This should throw ClassFormatError.
62         runTest("notFinalRecord",
63                 "Record attribute in non-final or abstract class");
64 
65         // Test loading a Record type that is badly formed. This should throw ClassFormatError.
66         runTest("badRecordAttribute",
67                 "Invalid constant pool index 13 for descriptor in Record attribute");
68 
69         // Test loading a Record type that is too small. This should throw ClassFormatError.
70         runTest("shortRecordAttribute", "Truncated class file");
71 
72         // Test that loading a class with an old class file version ignores a
73         // badly formed Record attribute. No exception should be thrown.
74         Class newClass = Class.forName("oldRecordAttribute");
75 
76         // Test that loading a class whose super class is not java.lang.Record
77         // ignores a badly formed Record attribute. No exception should be thrown.
78         newClass = Class.forName("superNotJLRecord");
79     }
80 }
