24 import jdk.test.lib.process.ProcessTools;
25 import jdk.test.lib.process.OutputAnalyzer;
26 
27 /*
28  * @test TestNullTerminatedFlags
29  * @bug 6522873
30  * @summary Test that the VM don't allow random junk characters at the end of valid command line flags.
31  * @library /test/lib
32  * @modules java.base/jdk.internal.misc
33  *          java.management
34  * @run driver TestNullTerminatedFlags
35  */
36 public class TestNullTerminatedFlags {
37    public static String[] options = {
38             "-Xnoclassgc",
39             "-Xbatch",
40             "-green",
41             "-native",
42             "-Xrs",
43             "-Xinternalversion",
44             "-Xprintflags",
45             "-Xint",
46             "-Xmixed",
47             "-Xcomp",
48             "-Xshare:dump",
49             "-Xshare:on",
50             "-Xshare:auto",
51             "-Xshare:off",
52             "-Xdebug",
53             "-Xnoagent"
54         };
55 
56     public static void main(String args[]) throws Exception{
57         for (String option : options) {
58             String testOption = option + "junk";
59             ProcessBuilder pb =
60                 ProcessTools.createJavaProcessBuilder(testOption, "-version");
61             new OutputAnalyzer(pb.start())
62                     .shouldContain("Unrecognized option: " + testOption)
63                     .shouldHaveExitValue(1);
64         }
65     }
66 }
