25 import sun.hotspot.WhiteBox;
26 
27 class LoadMe {
28     static String getValue() {
29         return "beforeHook";
30     }
31     static String getOtherValue() {
32         return "abc-beforeHook-xyz";
33     }
34 }
35 
36 public class ClassFileLoadHook {
37     public enum TestCaseId {
38         SHARING_OFF_CFLH_ON,   // test case to establish a baseline
39         SHARING_ON_CFLH_OFF,
40         SHARING_AUTO_CFLH_ON,
41         SHARING_ON_CFLH_ON
42     }
43 
44     public static void main(String args[]) {
45         TestCaseId testCase = TestCaseId.valueOf(args[0]);
46         WhiteBox wb = WhiteBox.getWhiteBox();
47 
48         System.out.println("====== ClassFileLoadHook.main():testCase = " + testCase);
49         System.out.println("getValue():" + LoadMe.getValue());
50         System.out.println("getOtherValue():" + LoadMe.getOtherValue());
51 
52         switch (testCase) {
53             case SHARING_OFF_CFLH_ON:
54                 assertTrue("after_Hook".equals(LoadMe.getValue()) &&
55                            "abc-after_Hook-xyz".equals(LoadMe.getOtherValue()),
56                            "Not sharing, this test should replace beforeHook " +
57                            "with after_Hook");
58             break;
59 
60             case SHARING_ON_CFLH_OFF:
61                 assertTrue(wb.isSharedClass(LoadMe.class),
62                     "LoadMe should be shared, but is not");
63                 assertTrue("beforeHook".equals(LoadMe.getValue()) &&
64                            "abc-beforeHook-xyz".equals(LoadMe.getOtherValue()),
65                            "CFLH off, bug values are redefined");
66             break;
67 
68             case SHARING_AUTO_CFLH_ON:
69             case SHARING_ON_CFLH_ON:
70                 // LoadMe is rewritten on CFLH
71                 assertFalse(wb.isSharedClass(LoadMe.class),
72                     "LoadMe should not be shared if CFLH has modified the class");
73                 assertFalse("beforeHook".equals(LoadMe.getValue()) &&
74                            "abc-beforeHook-xyz".equals(LoadMe.getOtherValue()),
75                            "Class contents should be changed if CFLH is enabled");
76              break;
77 
78              default:
79                  throw new RuntimeException("Invalid testcase");
80 
81         }
82     }
83 
84     private static void assertTrue(boolean expr, String msg) {
85         if (!expr)
86             throw new RuntimeException(msg);
87     }
88 
89     private static void assertFalse(boolean expr, String msg) {
90         if (expr)
91             throw new RuntimeException(msg);
92     }
93 }
