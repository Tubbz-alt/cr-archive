39 
40 import jdk.test.lib.process.OutputAnalyzer;
41 import sun.hotspot.WhiteBox;
42 
43 public class MirrorWithReferenceFieldsTest {
44     public static void main(String[] args) throws Exception {
45         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
46         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
47         String appJar = ClassFileInstaller.getJarPath("app.jar");
48 
49         String classlist[] = new String[] {
50             "MirrorWithReferenceFieldsApp",
51         };
52 
53         TestCommon.testDump(appJar, classlist, use_whitebox_jar);
54         OutputAnalyzer output = TestCommon.exec(appJar, use_whitebox_jar,
55                                                 "-XX:+UnlockDiagnosticVMOptions",
56                                                 "-XX:+WhiteBoxAPI",
57                                                 "-XX:+VerifyAfterGC",
58                                                 "MirrorWithReferenceFieldsApp");
59         try {
60             TestCommon.checkExec(output, "Done");
61         } catch (Exception e) {
62             output.shouldContain("Archived open_archive_heap objects are not mapped");
63         }
64     }
65 }
