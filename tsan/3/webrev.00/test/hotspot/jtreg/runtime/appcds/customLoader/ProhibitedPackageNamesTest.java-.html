25 /*
26  * @test
27  * @summary Make sure prohibited packages cannot be stored into archive for custom loaders.
28  * @requires vm.cds
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  *          jdk.jartool/sun.tools.jar
34  * @compile ClassListFormatBase.java test-classes/Hello.java test-classes/InProhibitedPkg.java
35  * @run driver ProhibitedPackageNamesTest
36  */
37 
38 public class ProhibitedPackageNamesTest extends ClassListFormatBase {
39     static {
40         // Uncomment the following line to run only one of the test cases
41         // ClassListFormatBase.RUN_ONLY_TEST = "TESTCASE PPN1";
42     }
43 
44     public static void main(String[] args) throws Throwable {
45         String appJar = JarBuilder.getOrCreateHelloJar();
46         String customJarPath = JarBuilder.build("ProhibitedPackageNames_custom", "java/InProhibitedPkg");
47 
48         dumpShouldPass(
49             "TESTCASE PPN1: prohibited package name without loader:",
50             appJar, classlist(
51                 "Hello",
52                 "java/lang/Object id: 1",
53                 // Without "loader:" keyword.
54                 "java/InProhibitedPkg id: 2 super: 1 source: " + customJarPath
55             ),
56             "Prohibited package for non-bootstrap classes: java/InProhibitedPkg.class");
57     }
58 }
