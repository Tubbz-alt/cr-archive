25 import java.lang.*;
26 
27 import java.lang.reflect.Method;
28 import jdk.internal.access.JavaLangAccess;
29 import jdk.internal.access.SharedSecrets;
30 
31 class EmptyClassHelper {
32     static final JavaLangAccess jla = SharedSecrets.getJavaLangAccess();
33     static final String USE_APP = "useAppLoader";
34     public static void main(String[] args) throws Exception {
35         Class cls = null;
36         Method m = null;
37         ClassLoader appLoader = ClassLoader.getSystemClassLoader();
38         String className = "com.sun.tools.javac.Main";
39         if (args[0].equals(USE_APP)) {
40             cls = appLoader.loadClass(className);
41             System.out.println("appLoader loaded class");
42             try {
43                 m = cls.getMethod("main", String[].class);
44                 System.out.println("appLoader found method main");
45             } catch(NoSuchMethodException ex) {
46                 System.out.println(ex.toString());
47             }
48         } else {
49             cls = jla.findBootstrapClassOrNull(appLoader, className);
50             System.out.println("bootLoader loaded class");
51             System.out.println("cls = " + cls);
52             try {
53                 m = cls.getMethod("main", String[].class);
54                 System.out.println("bootLoader found method main");
55             } catch(NoSuchMethodException ex) {
56                 System.out.println(ex.toString());
57             }
58         }
59     }
60 }
