25 /*
26  * @test
27  * @summary AppCDS handling of protection domain.
28  * @requires vm.cds
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/ProtDomain.java
34  * @compile test-classes/ProtDomainB.java
35  * @compile test-classes/JimageClassProtDomain.java
36  * @run driver ProtectionDomain
37  */
38 
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class ProtectionDomain {
42   public static void main(String[] args) throws Exception {
43     JarBuilder.build("prot_domain", "ProtDomain", "ProtDomainB", "ProtDomainOther",
44                      "ProtDomainBOther", "JimageClassProtDomain");
45 
46     String appJar = TestCommon.getTestJar("prot_domain.jar");
47     TestCommon.testDump(appJar,
48          TestCommon.list("ProtDomain",
49                          "ProtDomainBOther",
50                          "java/util/Dictionary",
51                          "com/sun/tools/javac/Main",
52                          "jdk/nio/zipfs/ZipInfo",
53                          "java/net/URL",
54                          "sun/rmi/rmic/Main",
55                          "com/sun/jndi/dns/DnsName"));
56 
57     // First class is loaded from CDS, second class is loaded from JAR
58     TestCommon.run("-cp", appJar, "ProtDomain")
59         .assertNormalExit("Protection Domains match");
60 
61     // First class is loaded from JAR, second class is loaded from CDS
62     TestCommon.run("-cp", appJar, "ProtDomainB")
63         .assertNormalExit("Protection Domains match");
64 
65     // Test ProtectionDomain for application and extension module classes from the
66     // "modules" jimage
67     TestCommon.run("-cp", appJar, "JimageClassProtDomain")
68         .assertNormalExit(output -> output.shouldNotContain(
69                           "Failed: Protection Domains do not match"));
70   }
71 }
