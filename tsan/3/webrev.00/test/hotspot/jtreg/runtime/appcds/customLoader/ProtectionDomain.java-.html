25 /*
26  * @test
27  * @summary AppCDS handling of protection domain in custom loaders.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  *
32  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  *          jdk.jartool/sun.tools.jar
36  * @compile test-classes/ProtDomain.java
37  * @run driver ProtectionDomain
38  */
39 
40 public class ProtectionDomain {
41     public static void main(String[] args) throws Exception {
42         String appJar = JarBuilder.build("ProtectionDomain-app", "ProtDomain");
43 
44         String customJar = JarBuilder.build("ProtectionDomain-custom",
45             "ProtDomainClassForArchive", "ProtDomainNotForArchive");
46         String[] classlist = new String[] {
47             "java/lang/Object id: 1",
48             "ProtDomain id: 2 super: 1 source: " + appJar,
49             "ProtDomainClassForArchive id: 3 super: 1 source: " + customJar
50         };
51 
52         TestCommon.testDump(appJar, classlist);
53 
54         // First class is loaded from CDS, second class is loaded from JAR
55         TestCommon.checkExec(TestCommon.exec(appJar, "-verbose:class", "ProtDomain", customJar),
56             "[class,load] ProtDomainClassForArchive source: shared objects file",
57             "[class,load] ProtDomainNotForArchive source: file");
58     }
59 }
