25 /*
26  * @test
27  * @summary Load app classes from CDS archive in parallel threads
28  * @library /test/lib
29  * @requires vm.cds
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/ParallelLoad.java
34  * @compile test-classes/ParallelClasses.java
35  * @run driver ParallelLoadTest
36  */
37 
38 public class ParallelLoadTest {
39     public static final int MAX_CLASSES = 40;
40 
41     public static void main(String[] args) throws Exception {
42         JarBuilder.build("parallel_load", getClassList(true));
43         String appJar = TestCommon.getTestJar("parallel_load.jar");
44         TestCommon.test(appJar, getClassList(false), "ParallelLoad");
45     }
46 
47     private static String[] getClassList(boolean includeWatchdog) {
48         int extra = includeWatchdog ? 3 : 2;
49         String[] classList = new String[MAX_CLASSES + extra];
50 
51         int i;
52         for (i=0; i<MAX_CLASSES; i++) {
53             classList[i] = "ParallelClass" + i;
54         }
55 
56         classList[i++] = "ParallelLoad";
57         classList[i++] = "ParallelLoadThread";
58         if (includeWatchdog)
59             classList[i++] = "ParallelLoadWatchdog";
60 
61         return classList;
62     }
63 }
