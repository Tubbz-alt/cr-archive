25 /*
26  * @test
27  * @summary classpath mismatch between dump time and execution time
28  * @requires vm.cds
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/Hello.java
34  * @run driver WrongClasspath
35  */
36 
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class WrongClasspath {
40 
41   public static void main(String[] args) throws Exception {
42     String appJar = JarBuilder.getOrCreateHelloJar();
43 
44     // Dump an archive with a specified JAR file in -classpath
45     TestCommon.testDump(appJar, TestCommon.list("Hello"));
46 
47     // Then try to execute the archive without -classpath -- it should fail
48     TestCommon.run(
49         /* "-cp", appJar, */ // <- uncomment this and the execution should succeed
50         "Hello")
51       .assertAbnormalExit("Unable to use shared archive",
52                           "shared class paths mismatch");
53   }
54 }
