38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class WideIloadTest {
41     public static void main(String args[]) throws Exception {
42         JarBuilder.build("iload_w", "Iloadw", "IloadwMain");
43         String appJar = TestCommon.getTestJar("iload_w.jar");
44         OutputAnalyzer dumpOutput = TestCommon.dump(appJar, TestCommon.list(
45                                         "Iloadw", "IloadwMain"));
46         TestCommon.checkDump(dumpOutput);
47         OutputAnalyzer execOutput = TestCommon.exec(appJar, "IloadwMain");
48         TestCommon.checkExec(execOutput, "Passed");
49     }
