25 /*
26  * @test
27  * @summary Load app classes from CDS archive in parallel threads. Similar to ParallelLoad.java, but each class in its own JAR
28  * @requires vm.cds
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/ParallelLoad.java
34  * @compile test-classes/ParallelClasses.java
35  * @run driver ParallelLoad2
36  */
37 
38 import java.io.File;
39 
40 public class ParallelLoad2 {
41   public static int MAX_CLASSES = 40;
42   public static void main(String[] args) throws Exception {
43     JarBuilder.build("parallel_load2", "ParallelLoad", "ParallelLoadThread", "ParallelLoadWatchdog");
44     for (int i=0; i<MAX_CLASSES; i++) {
45       JarBuilder.build("parallel_load2_" + i, "ParallelClass" + i);
46     }
47 
48     String cp = TestCommon.getTestJar("parallel_load2.jar");
49     for (int i=0; i<MAX_CLASSES; i++) {
50       cp += File.pathSeparator + TestCommon.getTestJar("parallel_load2_" + i + ".jar");
51     }
52 
53     String[] class_list = new String[MAX_CLASSES + 2];
54     for (int i=0; i<MAX_CLASSES; i++) {
55       class_list[i] = "ParallelClass" + i;
56     }
57     class_list[class_list.length - 1] = "ParallelLoad";
58     class_list[class_list.length - 2] = "ParallelLoadThread";
59 
60     TestCommon.test(cp, class_list,
61                           "ParallelLoad");
62   }
63 }
