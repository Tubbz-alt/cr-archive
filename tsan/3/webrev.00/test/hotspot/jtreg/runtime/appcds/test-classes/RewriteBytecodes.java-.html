25 import java.io.File;
26 import sun.hotspot.WhiteBox;
27 
28 public class RewriteBytecodes {
29   public static void main(String args[]) throws Throwable {
30     String from = "___xxx___";
31     String to   = "___yyy___";
32     File clsFile = new File(args[0]);
33     Class superClass = Util.defineModifiedClass(RewriteBytecodes.class.getClassLoader(), clsFile, from, to);
34 
35     Child child = new Child();
36 
37     if (child.getClass().getSuperclass() != superClass) {
38       throw new RuntimeException("Mismatched super class");
39     }
40     // Even if the Super class is not loaded from the CDS archive, make sure the Child class
41     // can still be loaded successfully, and properly inherits from the rewritten version
42     // of Super.
43     if (!child.toString().equals(to)) {
44       throw new RuntimeException("Wrong output, expected: " + to + ", but got: " + child.toString());
45     }
46 
47     WhiteBox wb = WhiteBox.getWhiteBox();
48     if (wb.isSharedClass(superClass)) {
49       throw new RuntimeException("wb.isSharedClass(superClass) should be false");
50     }
51     if (wb.isSharedClass(child.getClass())) {
52       throw new RuntimeException("wb.isSharedClass(child.getClass()) should be false");
53     }
54   }
55 }
