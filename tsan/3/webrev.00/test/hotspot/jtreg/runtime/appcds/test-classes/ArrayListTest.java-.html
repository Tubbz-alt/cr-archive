25 import java.util.*;
26 
27 // This is a test case executed by DumpClassList.java to load classes
28 // from various places to ensure that they are not written to the class list.
29 public class ArrayListTest {
30     public static void main(String args[]) throws Exception {
31         // The following lambda usage should generate various classes like
32         // java.lang.invoke.LambdaForm$MH/1146743572. All of them should be excluded from
33         // the class list.
34         List<String> a = new ArrayList<>();
35         a.add("hello world.");
36         a.forEach(str -> System.out.println(str));
37 
38         System.out.println(Class.forName("java.lang.NewClass")); // should be excluded from the class list.
39         System.out.println(Class.forName("boot.append.Foo"));    // should be excluded from the class list.
40     }
41 }
