24 /**
25  * @test
26  * @summary CDS dumping with JVMTI agent.
27  * @requires vm.cds
28  * @requires vm.flavor != "minimal"
29  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
30  * @modules jdk.jartool/sun.tools.jar
31  * @compile ../../test-classes/Hello.java
32  * @run main/othervm/native DumpingWithJvmtiAgent
33  */
34 
35 import java.io.File;
36 import java.nio.file.Path;
37 import java.nio.file.Paths;
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class DumpingWithJvmtiAgent {
41     private static final String AGENT_LIB_ONLOAD = "AddToSystemCLSearchOnLoad";
42 
43     public static void main(String[] args) throws Exception {
44         String appJar = JarBuilder.getOrCreateHelloJar();
45 
46         // CDS dump with a JVMTI agent with the AllowArchivingWithJavaAgent option.
47         // vm should exit with an error message.
48         OutputAnalyzer out = TestCommon.dump(
49            appJar,
50            TestCommon.list("Hello"),
51            "-XX:+UnlockDiagnosticVMOptions", "-XX:+AllowArchivingWithJavaAgent",
52            "-agentlib:" + AGENT_LIB_ONLOAD + "=" + appJar,
53            "-Djava.library.path=" + System.getProperty("java.library.path"));
54         out.shouldContain("CDS dumping does not support native JVMTI agent, name: " + AGENT_LIB_ONLOAD)
55            .shouldHaveExitValue(1);
56 
57         // CDS dump with a JVMTI agent without the AllowArchivingWithJavaAgent option.
58         // vm should exit with an error message.
59         out = TestCommon.dump(
60            appJar,
61            TestCommon.list("Hello"),
62            "-agentlib:" + AGENT_LIB_ONLOAD + "=" + appJar,
63            "-Djava.library.path=" + System.getProperty("java.library.path"));
64         out.shouldContain("CDS dumping does not support native JVMTI agent, name: " + AGENT_LIB_ONLOAD)
65            .shouldHaveExitValue(1);
66     }
67 }
