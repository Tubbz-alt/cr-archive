25 /*
26  * @test
27  * @summary Testing the loading of a class with the same name in two different class loaders.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  *
32  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  *          jdk.jartool/sun.tools.jar
36  * @compile test-classes/CustomLoadee.java
37  *     test-classes/CustomLoadee3.java
38  *     test-classes/SameNameUnrelatedLoaders.java
39  * @build sun.hotspot.WhiteBox
40  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
41  * @run driver SameNameInTwoLoadersTest
42  */
43 
44 import jdk.test.lib.process.OutputAnalyzer;
45 import sun.hotspot.WhiteBox;
46 
47 
48 public class SameNameInTwoLoadersTest {
49     private static String appJar;
50     private static String customJar;
51     private static String useWbParam;
52 
53     public static void main(String[] args) throws Exception {
54         appJar = JarBuilder.build("SameNameInTwoLoadersTest",
55             "SameNameUnrelatedLoaders");
56 
57         customJar = JarBuilder.build("SameNameInTwoLoadersTest_custom", "CustomLoadee", "CustomLoadee3");
58 
59         useWbParam = "-Xbootclasspath/a:" +
60             JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");;
61 
62         // ====== unrelated loaders
63         executeTestCase(getClassList_FP(),
64             "SameNameUnrelatedLoaders", "FpBoth");
65     }
66 
67     private static void executeTestCase(String[] classlist,
68         String testClass, String testCaseId) throws Exception {
69         classlist[0] = testClass;
70 
71         TestCommon.testDump(appJar, classlist, useWbParam);
72 
73         OutputAnalyzer output = TestCommon.exec(appJar,
74                                  // command-line arguments ...
75                                  "--add-opens=java.base/java.security=ALL-UNNAMED",
76                                  useWbParam,
77                                  "-XX:+UnlockDiagnosticVMOptions",
78                                  "-XX:+WhiteBoxAPI",
79                                  testClass,
80                                  customJar, testCaseId);
81         TestCommon.checkExec(output);
82     }
83 
84     // Single entry, no loader specified (FP method)
85     private static String[] getClassList_FP() {
86         return new String[] {
87             "SameNameUnrelatedLoaders",
88             "java/lang/Object id: 1",
89             "CustomLoadee id: 10 super: 1 source: " + customJar,
90         };
91     }
92 }
