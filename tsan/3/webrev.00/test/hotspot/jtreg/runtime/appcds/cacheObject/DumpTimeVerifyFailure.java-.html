25 /*
26  * @test
27  * @summary Dump time should not crash if any class with shared strings fails verification due to missing dependencies.
28  * @bug 8186789
29  * @requires vm.cds.archived.java.heap
30  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
31  * @modules java.base/jdk.internal.misc
32  * @modules java.management
33  *          jdk.jartool/sun.tools.jar
34  * @compile MyOuter.java MyException.java
35  * @run driver DumpTimeVerifyFailure
36  */
37 
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class DumpTimeVerifyFailure {
41     public static void main(String[] args) throws Exception {
42         // App classes (see MyOuter.java):
43         //   MyOuter
44         //   MyInnder$MyOuter extends MyOuter
45         //   MyException
46         //
47         // MyOuter$MyInner.test() throws MyException.
48         // The missingMyException.jar file only includes MyOuter and
49         // MyOuter$MyInner classes, but not the MyException class.
50         // At dump time, MyOuter and MyOuter$MyInner classes fail
51         // verification due to missing MyException class.
52         String[] ARCHIVE_CLASSES = {"MyOuter", "MyOuter$MyInner"};
53         String appJar = JarBuilder.build("missingMyException", ARCHIVE_CLASSES);
54 
55         OutputAnalyzer dumpOutput = TestCommon.dump(
56                 appJar, ARCHIVE_CLASSES,
57                 "-Xlog:verification",
58                 "-XX:SharedArchiveConfigFile=" + TestCommon.getSourceFile("DumpTimeVerifyFailure.config.txt"));
59         TestCommon.checkDump(dumpOutput, "Loading classes to share");
60     }
61 }
