24 #include <stdio.h>
25 #include <string.h>
26 #include <jvmti.h>
27 
28 #ifdef __cplusplus
29 extern "C" {
30 #endif
31 
32     static jvmtiEnv *jvmti = NULL;
33 
34     JNIEXPORT jint JNICALL Agent_OnLoad(JavaVM *jvm, char *options, void *reserved) {
35         int err = (*jvm)->GetEnv(jvm, (void**) &jvmti, JVMTI_VERSION_9);
36         if (err != JNI_OK) {
37             return JNI_ERR;
38         }
39         err = (*jvmti)->AddToSystemClassLoaderSearch(jvmti, (const char*)options);
40         if (err != JVMTI_ERROR_NONE) {
41             return JNI_ERR;
42         }
43         return JNI_OK;
44     }
45 
46 #ifdef __cplusplus
47 }
48 #endif
