25 import sun.hotspot.WhiteBox;
26 
27 // This class should be loaded from a shared archive.
28 public class MultiProcClass {
29     private static String instanceLabel;
30 
31     public static void main(String args[]) throws Exception {
32         instanceLabel = args[0];
33         String checkPmap = args[1];
34 
35         long pid = ProcessHandle.current().pid();
36         System.out.println(inst("========================== Starting MultiProcClass"));
37         System.out.println(inst("My PID: " + pid ));
38         System.out.println(inst("checkPmap = <" + checkPmap + ">" ));
39 
40         if ("true".equals(checkPmap)) {
41             if (runPmap(pid, true) != 0)
42                 System.out.println("MultiProcClass: Pmap failed");
43         }
44 
45         WhiteBox wb = WhiteBox.getWhiteBox();
46         if (!wb.isSharedClass(MultiProcClass.class)) {
47             throw new RuntimeException(inst("MultiProcClass should be shared but is not."));
48         }
49 
50         System.out.println(inst("========================== Leaving MultiProcClass"));
51     }
52 
53     // A convenience method to append process instance label
54     private static String inst(String msg) {
55         return "process-" + instanceLabel + " : " + msg;
56     }
57 
58     // Use on Linux-only; requires jdk-9 for Process.pid()
59     public static int runPmap(long pid, boolean inheritIO) throws Exception {
60         ProcessBuilder pmapPb = new ProcessBuilder("pmap", "" + pid);
61         if (inheritIO)
62             pmapPb.inheritIO();
63 
64         return pmapPb.start().waitFor();
65     }
66 }
