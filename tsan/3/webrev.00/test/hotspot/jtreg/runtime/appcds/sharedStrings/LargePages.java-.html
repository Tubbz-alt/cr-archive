25 /*
26  * @test
27  * @summary Basic shared string test with large pages
28  * @requires vm.cds.archived.java.heap
29  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
30  * @modules jdk.jartool/sun.tools.jar
31  * @build HelloString
32  * @run driver LargePages
33  */
34 public class LargePages {
35     static final String CDS_LOGGING = "-Xlog:cds,cds+hashtables";
36 
37     public static void main(String[] args) throws Exception {
38         SharedStringsUtils.run(args, LargePages::test);
39     }
40 
41     public static void test(String[] args) throws Exception {
42         SharedStringsUtils.buildJar("HelloString");
43 
44         SharedStringsUtils.dump(TestCommon.list("HelloString"),
45             "SharedStringsBasic.txt", "-XX:+UseLargePages", CDS_LOGGING);
46         SharedStringsUtils.runWithArchive("HelloString", "-XX:+UseLargePages");
47 
48         SharedStringsUtils.dump(TestCommon.list("HelloString"),
49             "SharedStringsBasic.txt", CDS_LOGGING,
50             "-XX:+UseLargePages", "-XX:+UseLargePagesInMetaspace");
51         SharedStringsUtils.runWithArchive("HelloString",
52             "-XX:+UseLargePages", "-XX:+UseLargePagesInMetaspace");
53     }
54 }
