25 /*
26  * @test
27  * @summary Use a shared string allocated in a humongous G1 region.
28  * @comment -- the following implies that G1 is used (by command-line or by default)
29  * @requires vm.cds.archived.java.heap
30  *
31  * @library /test/hotspot/jtreg/runtime/appcds /test/lib
32  * @modules jdk.jartool/sun.tools.jar
33  * @build HelloString
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. SharedStringsHumongous
37  */
38 import java.io.File;
39 import java.io.FileOutputStream;
40 import java.io.OutputStreamWriter;
41 import java.io.PrintWriter;
42 import sun.hotspot.WhiteBox;
43 import jdk.test.lib.process.OutputAnalyzer;
44 import jdk.test.lib.process.ProcessTools;
45 import jdk.test.lib.Asserts;
46 
47 public class SharedStringsHumongous {
48     static String sharedArchiveConfigFile = System.getProperty("user.dir") + File.separator + "SharedStringsHumongous_gen.txt";
49 
50     public static void main(String[] args) throws Exception {
51         WhiteBox wb = WhiteBox.getWhiteBox();
52 
53         try (FileOutputStream fos = new FileOutputStream(sharedArchiveConfigFile)) {
54             PrintWriter out = new PrintWriter(new OutputStreamWriter(fos));
55             out.println("VERSION: 1.0");
56             out.println("@SECTION: String");
57             out.println("31: shared_test_string_unique_14325");
58             int region_size = wb.g1RegionSize();
59             char body[] = new char[region_size + region_size / 2];
60             for (int i = 0; i < body.length; i++) {
61               body[i] = 'x';
62             }
63             Asserts.assertTrue(wb.g1IsHumongous(body));
64             String prefix = "generated_string (followed by " + body.length + " 'x') ";
65 
66             System.out.println("G1 region size: " + region_size);
67             System.out.println("Using a humongous string: " + prefix);
68 
69             String s = prefix + new String(body);
70             out.println(s.length() + ": " + s);
71             out.close();
72         }
73 
74         SharedStringsUtils.run(args, SharedStringsHumongous::test);
75     }
76 
77     public static void test(String[] args) throws Exception {
78         String vmOptionsPrefix[] = SharedStringsUtils.getChildVMOptionsPrefix();
79         String appJar = JarBuilder.build("SharedStringsHumongous", "HelloString");
80 
81         OutputAnalyzer dumpOutput = TestCommon.dump(appJar, TestCommon.list("HelloString"),
82                 TestCommon.concat(vmOptionsPrefix,
83                     "-XX:SharedArchiveConfigFile=" + sharedArchiveConfigFile,
84                     "-Xlog:gc+region+cds",
85                     "-Xlog:gc+region=trace"));
86         TestCommon.checkDump(dumpOutput, "extra interned string ignored; size too large");
87         // Extra strings that are humongous are not kelp alive, so they should be GC'ed
88         // before dumping the string table. That means the heap should contain no
89         // humongous regions.
90         dumpOutput.shouldNotMatch("gc,region,cds. HeapRegion 0x[0-9a-f]* HUM");
91 
92         OutputAnalyzer execOutput = TestCommon.exec(appJar,
93             TestCommon.concat(vmOptionsPrefix, "HelloString"));
94         TestCommon.checkExec(execOutput);
95     }
96 }
