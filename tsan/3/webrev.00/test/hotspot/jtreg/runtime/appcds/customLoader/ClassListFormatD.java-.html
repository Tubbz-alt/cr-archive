25 /*
26  * @test
27  * @summary Tests the format checking of hotspot/src/closed/share/vm/classfile/classListParser.cpp.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
32  * @modules java.base/jdk.internal.misc
33  *          java.management
34  *          jdk.jartool/sun.tools.jar
35  * @compile test-classes/Hello.java test-classes/CustomLoadee.java test-classes/CustomLoadee2.java
36  *          test-classes/CustomInterface2_ia.java test-classes/CustomInterface2_ib.java
37  * @run driver ClassListFormatD
38  */
39 
40 public class ClassListFormatD extends ClassListFormatBase {
41     static {
42         // Uncomment the following line to run only one of the test cases
43         // ClassListFormatBase.RUN_ONLY_TEST = "TESTCASE D1";
44     }
45 
46     public static void main(String[] args) throws Throwable {
47         String appJar = JarBuilder.getOrCreateHelloJar();
48         String customJarPath = JarBuilder.build("ClassListFormatD", "CustomLoadee",
49                                                 "CustomLoadee2", "CustomInterface2_ia",
50                                                 "CustomInterface2_ib");
51 
52         //----------------------------------------------------------------------
53         // TESTGROUP D: bad use of IDs
54         //----------------------------------------------------------------------
55         dumpShouldFail(
56             "TESTCASE D1: duplicated id:",
57             appJar, classlist(
58                 "Hello",
59                 "java/lang/Object id: 1",
60                 "CustomLoadee id: 1 super: 1 source: " + customJarPath
61             ),
62             "Duplicated ID 1 for class CustomLoadee");
63 
64         dumpShouldFail(
65             "TESTCASE D2: bad ID for super:",
66             appJar, classlist(
67                 "Hello",
68                 "java/lang/Object id: 1",
69                 "CustomLoadee id: 2 super: 2 source: " + customJarPath
70             ),
71             "Super class id 2 is not yet loaded");
72 
73         dumpShouldFail(
74             "TESTCASE D3: bad ID in interfaces:",
75             appJar, classlist(
76                 "Hello",
77                 "java/lang/Object id: 1",
78                 "CustomLoadee id: 2 super: 1 interfaces: 2 source: " + customJarPath
79             ),
80             "Interface id 2 is not yet loaded");
81     }
82 }
