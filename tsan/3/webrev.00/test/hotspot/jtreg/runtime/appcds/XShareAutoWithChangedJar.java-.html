25 /*
26  * @test
27  * @summary Test -Xshare:auto for AppCDS
28  * @requires vm.cds
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/Hello.java
34  * @run driver XShareAutoWithChangedJar
35  */
36 
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class XShareAutoWithChangedJar {
40   public static void main(String[] args) throws Exception {
41     String appJar = JarBuilder.build("XShareAutoWithChangedJar", "Hello");
42 
43     // 1. dump
44     OutputAnalyzer output = TestCommon.dump(appJar, TestCommon.list("Hello"));
45     TestCommon.checkDump(output);
46 
47     // 2. change the jar
48     JarBuilder.build("XShareAutoWithChangedJar", "Hello");
49 
50     // 3. exec
51     output = TestCommon.execAuto("-cp", appJar, "Hello");
52     output.shouldContain("Hello World");
53   }
54 }
