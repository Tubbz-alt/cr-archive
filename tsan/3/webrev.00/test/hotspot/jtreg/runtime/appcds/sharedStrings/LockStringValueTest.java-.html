25 import java.lang.reflect.*;
26 import sun.hotspot.WhiteBox;
27 
28 /*
29  * Lock the 'value' field of a known shared string, java.lang.Object
30  */
31 public class LockStringValueTest {
32     public static void main(String args[]) {
33         String s = "LiveOak";
34         WhiteBox wb = WhiteBox.getWhiteBox();
35 
36         if (wb.areSharedStringsIgnored()) {
37             System.out.println("The shared strings are ignored");
38             System.out.println("LockStringValueTest: PASS");
39             return;
40         }
41 
42         if (!wb.isShared(s)) {
43             throw new RuntimeException("LockStringValueTest Failed: String is not shared.");
44         }
45 
46         Class c = s.getClass();
47         try {
48             Field f = c.getDeclaredField("value");
49             f.setAccessible(true);
50             Object v = f.get(s);
51             lock(v);
52         } catch (NoSuchFieldException nfe) {
53         } catch (IllegalAccessException iae) {}
54     }
55 
56     public static void lock(Object o) {
57         synchronized (o) {
58             System.out.println("LockStringValueTest: PASS");
59         }
60     }
61 }
