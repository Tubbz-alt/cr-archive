25 import java.lang.instrument.ClassFileTransformer;
26 import java.lang.instrument.IllegalClassFormatException;
27 import java.security.ProtectionDomain;
28 
29 // Note: Util is from /test/hotspot/jtreg/runtime/appcds/test-classes/TestCommon.java
30 
31 public class InstrumentationClassFileTransformer implements ClassFileTransformer {
32     public byte[] transform(ClassLoader loader, String name, Class<?> classBeingRedefined,
33                             ProtectionDomain pd, byte[] buffer) throws IllegalClassFormatException {
34 
35         if (name.startsWith("InstrumentationApp$") && !name.equals("InstrumentationApp$NotTransformed")) {
36             System.out.println("Transforming: " + name + " class = " + classBeingRedefined);
37             try {
38                 if (classBeingRedefined == null) {
39                     // Initial transform
40                     replace(buffer, "buzz", "fuzz");
41                 } else {
42                     replace(buffer, "buzz", "guzz"); // Retransform
43                     replace(buffer, "huzz", "quzz"); // Redefine
44                 }
45             } catch (Throwable t) {
46                 t.printStackTrace();
47             }
48             return buffer;
49         }
50         return null;
51     }
52 
53     static void replace(byte[] buffer, String from, String to) {
54         int n = Util.replace(buffer, from, to);
55         System.out.println("..... replaced " + n + " occurrence(s) of '" + from + "' to '" + to + "'");
56     }
57 }
