25 /*
26  * @test
27  * @summary Ensure that support for AppCDS custom class loaders are not enabled on unsupported platforms.
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/SimpleHello.java
34  * @run driver UnsupportedPlatforms
35  */
36 
37 import jdk.test.lib.Platform;
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class UnsupportedPlatforms {
41     public static String PLATFORM_NOT_SUPPORTED_WARNING =
42         "AppCDS custom class loaders not supported on this platform";
43 
44     public static void main(String[] args) throws Exception {
45         String appJar = JarBuilder.build("UnsupportedPlatforms", "SimpleHello");
46 
47         // Dump the archive
48         String classlist[] = new String[] {
49             "SimpleHello",
50             "java/lang/Object id: 1",
51             "CustomLoadee id: 2 super: 1 source: " + appJar
52         };
53 
54         OutputAnalyzer out = TestCommon.dump(appJar, classlist);
55 
56         if (Platform.areCustomLoadersSupportedForCDS()) {
57             out.shouldNotContain(PLATFORM_NOT_SUPPORTED_WARNING);
58             out.shouldHaveExitValue(0);
59         } else {
60             out.shouldContain(PLATFORM_NOT_SUPPORTED_WARNING);
61             out.shouldHaveExitValue(1);
62         }
63     }
64 }
