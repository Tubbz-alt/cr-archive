25 /*
26  * @test
27  * @summary Test the behavior when shared classes loaded by custom loaders are
28  *          unloaded.
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  * @requires vm.opt.final.ClassUnloading
32  * @library /test/lib /test/hotspot/jtreg/runtime/appcds /test/hotspot/jtreg/runtime/testlibrary
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  *          jdk.jartool/sun.tools.jar
36  * @build sun.hotspot.WhiteBox ClassUnloadCommon
37  * @compile test-classes/UnloadUnregisteredLoader.java test-classes/CustomLoadee.java
38  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
39  * @run driver ClassFileInstaller ClassUnloadCommon
40  * @run driver ClassFileInstaller ClassUnloadCommon$1
41  * @run driver ClassFileInstaller ClassUnloadCommon$TestFailure
42  * @run driver UnloadUnregisteredLoaderTest
43  */
44 
45 import jdk.test.lib.process.OutputAnalyzer;
46 import sun.hotspot.WhiteBox;
47 
48 public class UnloadUnregisteredLoaderTest {
49     public static void main(String[] args) throws Exception {
50         String appJar1 = JarBuilder.build("UnloadUnregisteredLoader_app1", "UnloadUnregisteredLoader");
51         String appJar2 = JarBuilder.build(true, "UnloadUnregisteredLoader_app2",
52                                           "ClassUnloadCommon", "ClassUnloadCommon$1", "ClassUnloadCommon$TestFailure");
53         String customJarPath = JarBuilder.build("UnloadUnregisteredLoader_custom", "CustomLoadee");
54         String wbJar = JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
55         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
56 
57         String classpath = TestCommon.concatPaths(appJar1, appJar2);
58         String classlist[] = new String[] {
59             "UnloadUnregisteredLoader",
60             "ClassUnloadCommon",
61             "ClassUnloadCommon$1",
62             "ClassUnloadCommon$TestFailure",
63             "java/lang/Object id: 1",
64             "CustomLoadee id: 2 super: 1 source: " + customJarPath,
65         };
66 
67         OutputAnalyzer output;
68         TestCommon.testDump(classpath, classlist,
69                             // command-line arguments ...
70                             use_whitebox_jar);
71 
72         output = TestCommon.exec(classpath,
73                                  // command-line arguments ...
74                                  use_whitebox_jar,
75                                  "-XX:+UnlockDiagnosticVMOptions",
76                                  "-XX:+WhiteBoxAPI",
77                                  "UnloadUnregisteredLoader",
78                                  customJarPath);
79         TestCommon.checkExec(output);
80     }
81 }
