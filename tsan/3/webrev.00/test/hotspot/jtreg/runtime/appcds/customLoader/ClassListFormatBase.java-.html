25 import jdk.test.lib.process.OutputAnalyzer;
26 
27 /**
28  * Base class for ClassListFormat[A,B,C...].java
29  */
30 public class ClassListFormatBase {
31     protected static String RUN_ONLY_TEST = null;
32 
33     static void dumpShouldFail(String caseHelp, String appJar, String[] appClasses,
34                                String... expected_errors) throws Throwable {
35         if (RUN_ONLY_TEST != null && !caseHelp.startsWith(RUN_ONLY_TEST)) {
36             System.out.println("Skipped via RUN_ONLY_TEST: " + caseHelp);
37             return;
38         }
39         System.out.println("------------------------------");
40         System.out.println(caseHelp);
41         System.out.println("------------------------------");
42 
43         try {
44             OutputAnalyzer output = TestCommon.dump(appJar, appClasses);
45             output.shouldHaveExitValue(1);
46             for (String s : expected_errors) {
47                 output.shouldContain(s);
48             }
49         } catch (Throwable t) {
50             System.out.println("FAILED CASE: " + caseHelp);
51             throw t;
52         }
53     }
54 
55     static void dumpShouldPass(String caseHelp, String appJar, String[] appClasses,
56                                String... expected_msgs) throws Throwable {
57         if (RUN_ONLY_TEST != null && !caseHelp.startsWith(RUN_ONLY_TEST)) {
58             System.out.println("Skipped via RUN_ONLY_TEST: " + caseHelp);
59             return;
60         }
61         System.out.println("------------------------------");
62         System.out.println(caseHelp);
63         System.out.println("------------------------------");
64 
65         try {
66             OutputAnalyzer output = TestCommon.dump(appJar, appClasses, "-Xlog:cds");
67             output.shouldHaveExitValue(0);
68             output.shouldContain("Dumping");
69             for (String s : expected_msgs) {
70                 output.shouldContain(s);
71             }
72         } catch (Throwable t) {
73             System.out.println("FAILED CASE: " + caseHelp);
74             throw t;
75         }
76     }
77 
78     static String[] classlist(String... args) {
79         return TestCommon.list(args);
80     }
81 }
