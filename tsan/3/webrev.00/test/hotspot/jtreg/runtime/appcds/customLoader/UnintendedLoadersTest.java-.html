25 /*
26  * @test
27  * @summary Make sure classes intended for custom loaders cannot be loaded by BOOT/EXT/APP loaders
28  * @requires vm.cds
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  *          jdk.jartool/sun.tools.jar
34  * @compile test-classes/UnintendedLoaders.java test-classes/CustomLoadee.java
35  * @build sun.hotspot.WhiteBox
36  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
37  * @run driver UnintendedLoadersTest
38  */
39 
40 import jdk.test.lib.process.OutputAnalyzer;
41 import sun.hotspot.WhiteBox;
42 
43 public class UnintendedLoadersTest {
44     public static void main(String[] args) throws Exception {
45         String wbJar = JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
46         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
47 
48         String appJar = JarBuilder.build("UnintendedLoaders_app", "UnintendedLoaders");
49         String customJarPath = JarBuilder.build("UnintendedLoaders_custom", "CustomLoadee");
50 
51         // Dump the archive
52         String classlist[] = new String[] {
53             "UnintendedLoadersTest",
54             "java/lang/Object id: 1",
55 
56             // Without "loader:" keyword.
57             "CustomLoadee id: 2 super: 1 source: " + customJarPath,
58         };
59 
60         OutputAnalyzer output;
61         TestCommon.testDump(appJar, classlist,
62                             // command-line arguments ...
63                             use_whitebox_jar);
64 
65         output = TestCommon.exec(appJar,
66                                  // command-line arguments ...
67                                  use_whitebox_jar,
68                                  "-XX:+UnlockDiagnosticVMOptions",
69                                  "-XX:+WhiteBoxAPI",
70                                  "UnintendedLoaders");
71         TestCommon.checkExec(output);
72     }
73 }
