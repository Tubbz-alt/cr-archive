25 /*
26  * @test
27  * @requires vm.cds
28  * @summary a simple test to ensure that a directory in the --patch-module
29  *          option does not affect dump process
30  * @library ../..
31  * @library /test/hotspot/jtreg/testlibrary
32  * @library /test/lib
33  * @modules java.base/jdk.internal.misc
34  *          jdk.jartool/sun.tools.jar
35  * @build PatchMain
36  * @run driver PatchDir
37  */
38 
39 import java.io.File;
40 import jdk.test.lib.compiler.InMemoryJavaCompiler;
41 
42 public class PatchDir {
43     private static String moduleJar;
44 
45     public static void main(String args[]) throws Throwable {
46 
47         // Create a class file in the module java.naming. This class file
48         // will be put in the javanaming.jar file.
49         String source = "package javax.naming.spi; "                +
50                         "public class NamingManager { "             +
51                         "    static { "                             +
52                         "        System.out.println(\"I pass!\"); " +
53                         "    } "                                    +
54                         "}";
55 
56         String classDir = System.getProperty("test.classes");
57         ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
58              InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
59              classDir);
60 
61         JarBuilder.build("javanaming", "javax/naming/spi/NamingManager");
62         moduleJar = TestCommon.getTestJar("javanaming.jar");
63 
64         System.out.println("Test dumping with --patch-module");
65         TestCommon.dump(null,
66             TestCommon.list("javax/naming/spi/NamingManager"),
67             "--patch-module=java.naming=" + moduleJar,
68             "-Xlog:class+load",
69             "PatchMain", "javax.naming.spi.NamingManager")
70             .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module")
71             .shouldHaveExitValue(1);
72     }
73 }
