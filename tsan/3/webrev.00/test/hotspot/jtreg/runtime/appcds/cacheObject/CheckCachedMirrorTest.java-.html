40  * @run driver CheckCachedMirrorTest

43 import jdk.test.lib.process.OutputAnalyzer;
44 import sun.hotspot.WhiteBox;
45 
46 public class CheckCachedMirrorTest {
47     public static void main(String[] args) throws Exception {
48         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
49         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
50         String appJar = ClassFileInstaller.getJarPath("app.jar");
51         String helloJarPath = ClassFileInstaller.getJarPath("hello.jar");
52 
53         String classlist[] = new String[] {
54             "CheckCachedMirrorApp",            // built-in app loader
55             "java/lang/Object id: 1",          // boot loader
56             "Hello id: 2 super: 1 source: " + helloJarPath // custom loader
57         };
58 
59         TestCommon.testDump(appJar, classlist, use_whitebox_jar);
60         OutputAnalyzer output = TestCommon.exec(appJar, use_whitebox_jar,
61                                                 "-XX:+UnlockDiagnosticVMOptions",
62                                                 "-XX:+WhiteBoxAPI",
63                                                 "-Xlog:cds=debug",
64                                                 "CheckCachedMirrorApp",
65                                                 helloJarPath);
66         TestCommon.checkExec(output);
67     }
68 }
