25 /*
26  * @test
27  * @summary
28  * @requires vm.cds.archived.java.heap
29  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
30  * @modules java.base/jdk.internal.misc

33  * @build sun.hotspot.WhiteBox
34  * @compile GCStressApp.java
35  * @run driver ClassFileInstaller -jar gcstress.jar GCStressApp
36  * @run driver ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
37  * @run driver GCStressTest
38  */
39 
40 import jdk.test.lib.process.OutputAnalyzer;
41 
42 public class GCStressTest {
43     public static void main(String[] args) throws Exception {
44         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
45         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
46         String appJar = ClassFileInstaller.getJarPath("gcstress.jar");
47         String appClasses[] = TestCommon.list("GCStressApp");
48 
49         OutputAnalyzer output = TestCommon.dump(appJar, appClasses,
50                                                 use_whitebox_jar,
51                                                 "-Xms20M", "-Xmx20M");
52         output = TestCommon.exec(appJar, use_whitebox_jar,
53                                  "-Xlog:cds=info",
54                                  "-Xms20M", "-Xmx20M",
55                                  "-XX:+UnlockDiagnosticVMOptions",
56                                  "-XX:+WhiteBoxAPI","GCStressApp");
57         TestCommon.checkExec(output);
58     }
59 }
