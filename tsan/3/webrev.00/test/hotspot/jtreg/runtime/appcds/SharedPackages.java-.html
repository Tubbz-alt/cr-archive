25 /*
26  * @test
27  * @summary AppCDS handling of package.
28  * @requires vm.cds
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @compile test-classes/PackageTest.java
34  * @compile test-classes/JimageClassPackage.java
35  * @run driver SharedPackages
36  */
37 
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class SharedPackages {
41     public static void main(String[] args) throws Exception {
42         JarBuilder.build("pkg", "p/PackageTest", "JimageClassPackage");
43 
44         String appJar = TestCommon.getTestJar("pkg.jar");
45         TestCommon.testDump(appJar, TestCommon.list("p/PackageTest",
46                                                     "java/util/Dictionary",
47                                                     "sun/tools/javac/Main",
48                                                     "jdk/nio/zipfs/ZipInfo",
49                                                     "java/net/URL",
50                                                     "sun/rmi/rmic/Main",
51                                                     "com/sun/jndi/dns/DnsName"));
52 
53         OutputAnalyzer output;
54 
55         // Test 1: shared class from Jar on the -cp
56         output = TestCommon.exec(appJar, "-verbose:class", "p.PackageTest");
57         TestCommon.checkExec(output, "Expected package");
58         if (!TestCommon.isUnableToMap(output))
59             output.shouldContain("Package is not sealed");
60 
61         // Test 2: shared classes from "modules" jimage
62         output = TestCommon.exec(appJar, "-verbose:class",
63                                  "JimageClassPackage");
64         if (!TestCommon.isUnableToMap(output)) {
65             output.shouldNotContain("Unexpected package");
66             output.shouldNotContain("Package is not sealed");
67         }
68 
69         // Test 3: shared class from Jar on the -Xbootclasspath/a
70         TestCommon.dump(
71             appJar, TestCommon.list("p/PackageTest"), "-Xbootclasspath/a:" + appJar);
72         output = TestCommon.exec(appJar, "-Xbootclasspath/a:" + appJar, "p.PackageTest");
73         if (!TestCommon.isUnableToMap(output)) {
74             output.shouldNotContain("Unexpected package");
75             output.shouldContain("Package is not sealed");
76         }
77     }
78 }
