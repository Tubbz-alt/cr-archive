 25 /*
 26  * @test
 27  * @summary Tests the format checking of hotspot/src/closed/share/vm/classfile/classListParser.cpp.
 28  *
 29  * @requires vm.cds
 30  * @requires vm.cds.custom.loaders
 31  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
 32  * @modules java.base/jdk.internal.misc
 33  *          java.management
 34  *          jdk.jartool/sun.tools.jar
 35  * @compile test-classes/Hello.java test-classes/CustomLoadee.java test-classes/CustomLoadee2.java
 36  *          test-classes/CustomInterface2_ia.java test-classes/CustomInterface2_ib.java
 37  * @run driver ClassListFormatE
 38  */
 39 
 40 public class ClassListFormatE extends ClassListFormatBase {
 41     static {
 42         // Uncomment the following line to run only one of the test cases
 43         // ClassListFormatBase.RUN_ONLY_TEST = "TESTCASE E1";
 44     }
 45 
 46     public static void main(String[] args) throws Throwable {
 47         String appJar = JarBuilder.getOrCreateHelloJar();
 48         String customJarPath = JarBuilder.build("ClassListFormatE", "CustomLoadee",
 49                                                 "CustomLoadee2", "CustomInterface2_ia",
 50                                                 "CustomInterface2_ib");
 51 
 52         //----------------------------------------------------------------------
 53         // TESTGROUP E: super class and interfaces
 54         //----------------------------------------------------------------------
 55         dumpShouldFail(
 56             "TESTCASE E1: missing interfaces: keyword",
 57             appJar, classlist(
 58                 "Hello",
 59                 "java/lang/Object id: 1",
 60                 "CustomLoadee2 id: 1 super: 1 source: " + customJarPath
 61             ),
 62             "Class CustomLoadee2 implements the interface CustomInterface2_ia, but no interface has been specified in the input line");
 63 
 64         dumpShouldFail(
 65             "TESTCASE E2: missing one interface",
 66             appJar, classlist(
 67                 "Hello",
 68                 "java/lang/Object id: 1",
 69                 "CustomInterface2_ia id: 2 super: 1 source: " + customJarPath,
 70                 "CustomInterface2_ib id: 3 super: 1 source: " + customJarPath,
 71                 "CustomLoadee2 id: 4 super: 1 interfaces: 2 source: " + customJarPath
 72             ),
 73             "The interface CustomInterface2_ib implemented by class CustomLoadee2 does not match any of the specified interface IDs");
 74 
 75         dumpShouldFail(
 76             "TESTCASE E3: specifying an interface that's not implemented by the class",
 77             appJar, classlist(
 78                 "Hello",
 79                 "java/lang/Object id: 1",
 80                 "CustomInterface2_ia id: 2 super: 1 source: " + customJarPath,
 81                 "CustomLoadee id: 2 super: 1 interfaces: 2 source: " + customJarPath
 82             ),
 83             "The number of interfaces (1) specified in class list does not match the class file (0)");
 84 
 85         dumpShouldFail(
 86             "TESTCASE E4: repeating an ID in the interfaces: keyword",
 87             appJar, classlist(
 88                 "Hello",
 89                 "java/lang/Object id: 1",
 90                 "CustomInterface2_ia id: 2 super: 1 source: " + customJarPath,
 91                 "CustomInterface2_ib id: 3 super: 1 source: " + customJarPath,
 92                 "CustomLoadee2 id: 4 super: 1 interfaces: 2 2 3 source: " + customJarPath
 93             ),
 94             "The number of interfaces (3) specified in class list does not match the class file (2)");
 95 
 96         dumpShouldFail(
 97             "TESTCASE E5: wrong super class",
 98             appJar, classlist(
 99                 "Hello",
100                 "java/lang/Object id: 1",
101                 "CustomInterface2_ia id: 2 super: 1 source: " + customJarPath,
102                 "CustomInterface2_ib id: 3 super: 1 source: " + customJarPath,
103                 "CustomLoadee id: 4 super: 1 source: " + customJarPath,
104                 "CustomLoadee2 id: 5 super: 4 interfaces: 2 3 source: " + customJarPath
105             ),
106             "The specified super class CustomLoadee (id 4) does not match actual super class java.lang.Object");
107     }
108 }
