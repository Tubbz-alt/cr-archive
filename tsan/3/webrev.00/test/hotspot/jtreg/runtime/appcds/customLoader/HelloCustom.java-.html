25 /*
26  * @test
27  * @summary Hello World test for AppCDS custom loader support
28  * @requires vm.cds
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  * @compile test-classes/Hello.java test-classes/CustomLoadee.java
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller -jar hello.jar Hello
36  * @run driver ClassFileInstaller -jar hello_custom.jar CustomLoadee
37  * @run driver ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
38  * @run driver HelloCustom
39  */
40 
41 import jdk.test.lib.process.OutputAnalyzer;
42 import sun.hotspot.WhiteBox;
43 
44 public class HelloCustom {
45     public static void main(String[] args) throws Exception {
46         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
47         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
48 
49         String appJar = ClassFileInstaller.getJarPath("hello.jar");
50         String customJarPath = ClassFileInstaller.getJarPath("hello_custom.jar");
51 
52         // Dump the archive
53         String classlist[] = new String[] {
54             "Hello",
55             "java/lang/Object id: 1",
56             "CustomLoadee id: 2 super: 1 source: " + customJarPath
57         };
58 
59         OutputAnalyzer output;
60         TestCommon.testDump(appJar, classlist,
61                             // command-line arguments ...
62                             use_whitebox_jar);
63 
64         output = TestCommon.exec(appJar,
65                                  // command-line arguments ...
66                                  use_whitebox_jar,
67                                  "-XX:+UnlockDiagnosticVMOptions",
68                                  "-XX:+WhiteBoxAPI",
69                                  "Hello", customJarPath);
70         TestCommon.checkExec(output);
71     }
72 }
