25 import java.io.File;
26 import java.net.URL;
27 import sun.hotspot.WhiteBox;
28 
29 //
30 // - Test static final String field with initial value in cached mirror should be also archived.
31 // - GC should not crash when reference fields in cached mirror are updated at runtime
32 //     - Reference fields are updated to point to runtime created objects
33 //     - Reference fields are nullified
34 //
35 public class MirrorWithReferenceFieldsApp {
36 
37     // Static String field with initial value
38     static final String archived_field = "abc";
39 
40     // Static object field
41     static Object non_archived_field_1;
42 
43     // Instance field
44     Integer non_archived_field_2;
45 
46     public MirrorWithReferenceFieldsApp() {
47         non_archived_field_1 = new Object();
48         non_archived_field_2 = new Integer(1);
49     }
50 
51     public static void main(String args[]) throws Exception {
52         WhiteBox wb = WhiteBox.getWhiteBox();
53 
54         if (!wb.areOpenArchiveHeapObjectsMapped()) {
55             System.out.println("Archived open_archive_heap objects are not mapped.");
56             System.out.println("This may happen during normal operation. Test Skipped.");
57             return;
58         }
59 
60         MirrorWithReferenceFieldsApp m = new MirrorWithReferenceFieldsApp();
61         m.test(wb);
62     }
63 
64     public void test(WhiteBox wb) {
65         Class c = MirrorWithReferenceFieldsApp.class;
66         if (wb.isSharedClass(c)) {
67             // Check if the Class object is cached
68             if (wb.isShared(c)) {
69                 System.out.println(c + " mirror is cached. Expected.");
70             } else {
71                 throw new RuntimeException(
72                     "FAILED. " + c + " mirror should be cached.");
73             }
74 
75             // Check fields
76 
77             if (wb.isShared(archived_field)) {
78                 System.out.println("archived_field is archived as excepted");
79             } else {
80                 throw new RuntimeException(
81                     "FAILED. archived_field is not archived.");
82             }
83 
84             // GC should not crash
85             System.gc();
86             System.gc();
87             System.gc();
88 
89             non_archived_field_1 = null;
90             non_archived_field_2 = null;
91 
92             System.gc();
93             System.gc();
94             System.gc();
95 
96             System.out.println("Done.");
97         }
98     }
99 }
