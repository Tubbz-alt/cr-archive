25 /*
26  * @test
27  * @summary Regression test for JDK-8098821
28  * @bug 8098821
29  * @requires vm.cds.archived.java.heap
30  * @modules jdk.jartool/sun.tools.jar
31  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
32  * @run driver SysDictCrash
33  */
34 
35 import jdk.test.lib.process.OutputAnalyzer;
36 import jdk.test.lib.process.ProcessTools;
37 
38 public class SysDictCrash {
39     public static void main(String[] args) throws Exception {
40         SharedStringsUtils.run(args, SysDictCrash::test);
41     }
42 
43     public static void test(String[] args) throws Exception {
44         String vmOptionsPrefix[] = SharedStringsUtils.getChildVMOptionsPrefix();
45 
46         // SharedBaseAddress=0 puts the archive at a very high address on solaris,
47         // which provokes the crash.
48         ProcessBuilder dumpPb = ProcessTools.createJavaProcessBuilder(true,
49           TestCommon.concat(vmOptionsPrefix,
50             "-XX:+UseG1GC", "-XX:MaxRAMPercentage=12.5",
51             "-cp", ".",
52             "-XX:SharedBaseAddress=0", "-XX:SharedArchiveFile=./SysDictCrash.jsa",
53             "-Xshare:dump",
54             "-showversion", "-Xlog:cds,cds+hashtables"));
55 
56         TestCommon.checkDump(TestCommon.executeAndLog(dumpPb, "dump"));
57 
58         ProcessBuilder runPb = ProcessTools.createJavaProcessBuilder(true,
59           TestCommon.concat(vmOptionsPrefix,
60             "-XX:+UseG1GC", "-XX:MaxRAMPercentage=12.5",
61             "-XX:SharedArchiveFile=./SysDictCrash.jsa",
62             "-Xshare:on",
63             "-version"));
64 
65         TestCommon.checkExec(TestCommon.executeAndLog(runPb, "exec"));
66     }
67 }
