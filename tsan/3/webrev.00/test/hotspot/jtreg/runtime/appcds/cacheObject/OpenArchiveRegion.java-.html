25 /*
26  * @test
27  * @summary Test open archive heap regions
28  * @requires vm.cds.archived.java.heap
29  * @comment This test explicitly chooses the type of GC to be used by sub-processes. It may conflict with the GC type set
30  * via the -vmoptions command line option of JTREG. vm.gc==null will help the test case to discard the explicitly passed
31  * vm options.
32  * @requires (vm.gc=="null")
33  * @library /test/lib /test/hotspot/jtreg/runtime/appcds
34  * @modules java.base/jdk.internal.misc
35  * @modules java.management
36  *          jdk.jartool/sun.tools.jar
37  * @compile ../test-classes/Hello.java
38  * @run driver OpenArchiveRegion
39  */
40 
41 import jdk.test.lib.process.OutputAnalyzer;
42 
43 public class OpenArchiveRegion {
44     public static void main(String[] args) throws Exception {
45         JarBuilder.getOrCreateHelloJar();
46         String appJar = TestCommon.getTestJar("hello.jar");
47         String appClasses[] = TestCommon.list("Hello");
48 
49         // Dump with open archive heap region, requires G1 GC
50         OutputAnalyzer output = TestCommon.dump(appJar, appClasses);
51         TestCommon.checkDump(output, "oa0 space:");
52         output.shouldNotContain("oa0 space:         0 [");
53         output = TestCommon.exec(appJar, "Hello");
54         TestCommon.checkExec(output, "Hello World");
55         output = TestCommon.exec(appJar, "-XX:+UseSerialGC", "Hello");
56         TestCommon.checkExec(output, "Hello World");
57 
58         // Dump with open archive heap region disabled when G1 GC is not in use
59         output = TestCommon.dump(appJar, appClasses, "-XX:+UseParallelGC");
60         TestCommon.checkDump(output);
61         output.shouldNotContain("oa0 space:");
62         output = TestCommon.exec(appJar, "Hello");
63         TestCommon.checkExec(output, "Hello World");
64     }
65 }
