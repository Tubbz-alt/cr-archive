31 public class VerifierTest0 {
32   public static void main(String args[]) {
33     boolean good = true;
34     good &= mustBeInvalid("VerifierTestA");
35     good &= mustBeInvalid("VerifierTestB");
36     good &= mustBeInvalid("VerifierTestC");
37     good &= mustBeInvalid("VerifierTestD");
38     good &= mustBeInvalid("VerifierTestE");
39     if (!good) {
40       System.out.println("VerifierTest0 failed");
41       System.exit(1);
42     }
43   }
44 
45   /** @return false means error */
46   static boolean mustBeInvalid(String className) {
47     System.out.println("Testing: " + className);
48     try {
49       Class.forName(className);
50       System.out.println("ERROR: class " + className + " was loaded unexpectedly.");
51       return false;
52     } catch (Throwable t) {
53       System.out.println("Expected exception:");
54       t.printStackTrace();
55       return true;
56     }
57   }
58 }
