37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 // This is a test for the handling of multi-dimensional Arrays in MetaspaceClosure.
40 //
41 // We choose FieldAnnotations because they happen to be implemented as a multi-dimension
42 // Array (Annotations::_fields_annotations, which is of type Array<Array<unsigned char>*>*,
43 // and is handled by the template class PointerArrayRef<T> in metaspaceClosure.hpp).
44 //
45 // Specifically, we are testing the following C code, where _fields_annotations is non-NULL:
46 //
47 // void Annotations::metaspace_pointers_do(MetaspaceClosure* it) {
48 //   ...
49 //   it->push(&_fields_annotations);
50 //
51 // which will be matched with the function
52 //
53 // template <typename T> void MetaspaceClosure::push(Array<T*>** mpp, Writability w = _default)
54 //
55 public class FieldAnnotationsTest {
56     public static void main(String[] args) throws Exception {
57         String[] ARCHIVE_CLASSES = {"FieldAnnotationsApp", "MyAnnotation"};
58         String appJar = JarBuilder.build("FieldAnnotationsTest", ARCHIVE_CLASSES);
59 
60         OutputAnalyzer dumpOutput = TestCommon.dump(
61                 appJar, ARCHIVE_CLASSES);
62         TestCommon.checkDump(dumpOutput);
63 
64         OutputAnalyzer execOutput = TestCommon.exec(appJar, "FieldAnnotationsApp");
65         TestCommon.checkExec(execOutput, "Field annotations are OK.");
66     }
67 }
