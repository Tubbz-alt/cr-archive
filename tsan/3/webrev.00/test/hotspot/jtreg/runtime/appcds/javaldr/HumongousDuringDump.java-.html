25 /*
26  * @test
27  * @summary Test how CDS dumping handles the existence of humongous G1 regions.
28  * @library /test/lib /test/hotspot/jtreg/runtime/appcds /test/hotspot/jtreg/runtime/appcds/test-classes
29  * @requires vm.cds.archived.java.heap
30  * @requires vm.flavor != "minimal"
31  * @modules java.base/jdk.internal.misc
32  *          jdk.jartool/sun.tools.jar
33  *          java.management
34  * @build HumongousDuringDumpTransformer Hello
35  * @run main/othervm/timeout=240 HumongousDuringDump
36  */
37 
38 import jdk.test.lib.cds.CDSOptions;
39 import jdk.test.lib.process.OutputAnalyzer;
40 import jdk.test.lib.process.ProcessTools;
41 
42 public class HumongousDuringDump {
43     public static String appClasses[] = {
44         "Hello",
45     };
46     public static String agentClasses[] = {
47         "HumongousDuringDumpTransformer",
48     };
49 
50     public static void main(String[] args) throws Throwable {
51         String agentJar =
52             ClassFileInstaller.writeJar("HumongousDuringDumpTransformer.jar",
53                                         ClassFileInstaller.Manifest.fromSourceFile("HumongousDuringDumpTransformer.mf"),
54                                         agentClasses);
55 
56         String appJar =
57             ClassFileInstaller.writeJar("HumongousDuringDumpApp.jar", appClasses);
58 
59         String gcLog = Boolean.getBoolean("test.cds.verbose.gc") ?
60             "-Xlog:gc*=info,gc+region=trace,gc+alloc+region=debug" : "-showversion";
61 
62         String extraArg = "-javaagent:" + agentJar;
63         String extraOption = "-XX:+AllowArchivingWithJavaAgent";
64 
65         OutputAnalyzer out =
66           TestCommon.testDump(appJar, TestCommon.list("Hello"),
67                               "-XX:+UnlockDiagnosticVMOptions", extraOption,
68                               "-Xlog:gc+region+cds",
69                               "-Xlog:gc+region=trace",
70                               extraArg, "-Xmx64m", gcLog);
71         out.shouldContain("(Unmovable) humongous regions have been found and may lead to fragmentation");
72         out.shouldContain("All free regions should be at the top end of the heap, but we found holes.");
73         out.shouldMatch("gc,region,cds. HeapRegion .* HUM. hole");
74         String pattern = "gc,region,cds. HeapRegion .*hole";
75         out.shouldMatch(pattern);
76         out.shouldNotMatch(pattern + ".*unexpected");
77 
78         TestCommon.run(
79                 "-cp", appJar,
80                 "-verbose",
81                 "-Xmx64m",
82                 "-XX:+PrintSharedSpaces",
83                 "-XX:+UnlockDiagnosticVMOptions", extraOption,
84                 gcLog,
85                 "Hello")
86               .assertNormalExit();
87     }
88 }
