25 package p;
26 
27 public class PackageTest {
28     public static void main(String args[]) {
29         (new PackageTest()).test();
30     }
31 
32     private void test() {
33         ClassLoader cl = PackageTest.class.getClassLoader();
34         Package pkg_from_loader;
35         if (cl != null) {
36             pkg_from_loader = cl.getDefinedPackage("p");
37         } else {
38             pkg_from_loader = Package.getPackage("p");
39         }
40 
41         Package pkg = PackageTest.class.getPackage();
42         if (pkg_from_loader != null && pkg == pkg_from_loader &&
43             pkg.getName().equals("p")) {
44             System.out.println("Expected package: " + pkg);
45         } else {
46             System.out.println("Unexpected package: " + pkg);
47             System.exit(1);
48         }
49         if (pkg.isSealed()) {
50             System.out.println("Package is sealed");
51             System.exit(1);
52         } else {
53             System.out.println("Package is not sealed");
54         }
55     }
56 }
