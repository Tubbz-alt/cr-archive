25 /*
26  * @test
27  * @summary Basic (sanity) test for JDK-under-test inside a docker image.
28  * @requires docker.support
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *          jdk.jartool/sun.tools.jar
33  * @build HelloDocker
34  * @run driver DockerBasicTest
35  */
36 import jdk.test.lib.containers.docker.DockerRunOptions;
37 import jdk.test.lib.containers.docker.DockerTestUtils;
38 import jdk.test.lib.Platform;
39 import jdk.test.lib.Utils;
40 
41 
42 public class DockerBasicTest {
43     private static final String imageNameAndTag = "jdk10-internal:test";
44     // Diganostics: set to false to examine image after the test
45     private static final boolean removeImageAfterTest = true;
46 
47     public static void main(String[] args) throws Exception {
48         if (!DockerTestUtils.canTestDocker()) {
49             return;
50         }
51 
52         DockerTestUtils.buildJdkDockerImage(imageNameAndTag, "Dockerfile-BasicTest", "jdk-docker");
53 
54         try {
55             testJavaVersion();
56             testHelloDocker();
57         } finally {
58             if (removeImageAfterTest)
59                 DockerTestUtils.removeDockerImage(imageNameAndTag);
60         }
61     }
62 
63 
64     private static void testJavaVersion() throws Exception {
65         DockerRunOptions opts =
66             new DockerRunOptions(imageNameAndTag, "/jdk/bin/java", "-version");
67 
68         DockerTestUtils.dockerRunJava(opts)
69             .shouldHaveExitValue(0)
70             .shouldContain(Platform.vmName);
71     }
72 
73 
74     private static void testHelloDocker() throws Exception {
75         DockerRunOptions opts =
76             new DockerRunOptions(imageNameAndTag, "/jdk/bin/java", "HelloDocker")
77             .addJavaOpts("-cp", "/test-classes/")
78             .addDockerOpts("--volume", Utils.TEST_CLASSES + ":/test-classes/");
79 
80         DockerTestUtils.dockerRunJava(opts)
81             .shouldHaveExitValue(0)
82             .shouldContain("Hello Docker");
83     }
84 }
