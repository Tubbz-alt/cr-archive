24 import sun.hotspot.WhiteBox;
25 
26 public class CheckContainerized {
27     public static String OUTSIDE_OF_CONTAINER =
28         "CheckContainerized: Running outside of a container";
29     public static String INSIDE_A_CONTAINER =
30         "CheckContainerized: Running inside a container";
31 
32     public static void main(String[] args) {
33         System.out.println("CheckContainerized: Entering");
34         WhiteBox wb = WhiteBox.getWhiteBox();
35 
36         if (wb.isContainerized()) {
37             System.out.println(INSIDE_A_CONTAINER);
38 
39         } else {
40             System.out.println(OUTSIDE_OF_CONTAINER);
41         }
42     }
43 }
