24 /*
25  * @test
26  * @bug 8225789
27  * @summary Test that an unqualified name of "L;" causes a ClassFormatError exception.
28  * @compile EmptyUnqName.jasm
29  * @run main/othervm -Xverify:remote TestEmptyUnqName
30  */
31 
32 public class TestEmptyUnqName {
33     public static void main(String args[]) throws Throwable {
34 
35         System.out.println("Regression test for bug 8225789");
36 
37         try {
38             Class newClass = Class.forName("EmptyUnqName");
39             throw new RuntimeException("Expected ClassFormatError exception not thrown");
40         } catch (java.lang.ClassFormatError e) {
41             if (!e.getMessage().contains("Class name is empty or contains illegal character")) {
42                 throw new RuntimeException("Wrong ClassFormatError: " + e.getMessage());
43             }
44         }
45     }
46 }
