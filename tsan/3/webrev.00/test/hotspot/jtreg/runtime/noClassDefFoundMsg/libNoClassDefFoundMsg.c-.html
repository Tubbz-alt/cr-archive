24 #include <jni.h>
25 
26 JNIEXPORT void JNICALL
27 Java_NoClassDefFoundMsg_callDefineClass(JNIEnv *env, jclass klass, jstring className) {
28     const char *c_name = (*env)->GetStringUTFChars(env, className, NULL);
29     (*env)->DefineClass(env, c_name, NULL, NULL, 0);
30 }
31 
32 JNIEXPORT void JNICALL
33 Java_NoClassDefFoundMsg_callFindClass(JNIEnv *env, jclass klass, jstring className) {
34     const char *c_name;
35     jclass cls;
36     if (className == NULL) {
37         c_name = NULL;
38     } else {
39         c_name = (*env)->GetStringUTFChars(env, className, NULL);
40     }
41     cls = (*env)->FindClass(env, c_name);
42 }
