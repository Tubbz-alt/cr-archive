 25 import jdk.test.lib.process.ProcessTools;
 26 import jdk.test.lib.process.OutputAnalyzer;
 27 import jdk.test.lib.JDKToolFinder;
 28 
 29 /*
 30  * @test
 31  * @key metaspace jcmd
 32  * @summary Test the VM.metaspace command
 33  * @requires vm.gc != "Z" & vm.bits != "32"
 34  * @library /test/lib
 35  * @modules java.base/jdk.internal.misc
 36  *          java.management
 37  * @run main/othervm -XX:MaxMetaspaceSize=201M -Xmx100M -XX:+UseCompressedOops -XX:+UseCompressedClassPointers PrintMetaspaceDcmd with-compressed-class-space
 38  * @run main/othervm -XX:MaxMetaspaceSize=201M -Xmx100M -XX:-UseCompressedOops -XX:-UseCompressedClassPointers PrintMetaspaceDcmd without-compressed-class-space
 39  */
 40 /*
 41  * @test
 42  * @key metaspace jcmd
 43  * @summary Test the VM.metaspace command
 44  * @requires vm.gc != "Z" & vm.bits == "32"
 45  * @library /test/lib
 46  * @modules java.base/jdk.internal.misc
 47  *          java.management
 48  * @run main/othervm -XX:MaxMetaspaceSize=201M -Xmx100M PrintMetaspaceDcmd without-compressed-class-space
 49  */
 50 
 51 public class PrintMetaspaceDcmd {
 52 
 53     // Run jcmd VM.metaspace against a VM with CompressedClassPointers on.
 54     // The report should detail Non-Class and Class portions separately.
 55     private static void doTheTest(boolean usesCompressedClassSpace) throws Exception {
 56         ProcessBuilder pb = new ProcessBuilder();
 57         OutputAnalyzer output;
 58         // Grab my own PID
 59         String pid = Long.toString(ProcessTools.getProcessId());
 60 
 61         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "basic"});
 62         output = new OutputAnalyzer(pb.start());
 63         output.shouldHaveExitValue(0);
 64         if (usesCompressedClassSpace) {
 65             output.shouldContain("Non-Class:");
 66             output.shouldContain("Class:");
 67         }
 68         output.shouldContain("Virtual space:");
 69         output.shouldContain("Chunk freelists:");
 70         output.shouldMatch("MaxMetaspaceSize:.*201.00.*MB");
 71 
 72         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace"});
 73         output = new OutputAnalyzer(pb.start());
 74         output.shouldHaveExitValue(0);
 75         if (usesCompressedClassSpace) {
 76             output.shouldContain("Non-Class:");
 77             output.shouldContain("Class:");
 78         }
 79         output.shouldContain("Virtual space:");
 80         output.shouldContain("Chunk freelist");
 81         output.shouldContain("Waste");
 82         output.shouldMatch("MaxMetaspaceSize:.*201.00.*MB");
 83 
 84         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "show-loaders"});
 85         output = new OutputAnalyzer(pb.start());
 86         output.shouldHaveExitValue(0);
 87         output.shouldMatch("CLD.*<bootstrap>");
 88 
 89         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "by-chunktype"});
 90         output = new OutputAnalyzer(pb.start());
 91         output.shouldHaveExitValue(0);
 92         output.shouldContain("specialized:");
 93         output.shouldContain("small:");
 94         output.shouldContain("medium:");
 95         output.shouldContain("humongous:");
 96 
 97         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "vslist"});
 98         output = new OutputAnalyzer(pb.start());
 99         output.shouldHaveExitValue(0);
100         output.shouldContain("Virtual space list");
101         output.shouldMatch("node.*reserved.*committed.*used.*");
102 
103         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "vsmap"});
104         output = new OutputAnalyzer(pb.start());
105         output.shouldHaveExitValue(0);
106         output.shouldContain("Virtual space map:");
107         output.shouldContain("HHHHHHHHHHH");
108 
109         // Test with different scales
110         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "scale=G"});
111         output = new OutputAnalyzer(pb.start());
112         output.shouldHaveExitValue(0);
113         output.shouldMatch("MaxMetaspaceSize:.*0.2.*GB");
114 
115         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "scale=K"});
116         output = new OutputAnalyzer(pb.start());
117         output.shouldHaveExitValue(0);
118         output.shouldMatch("MaxMetaspaceSize:.*205824.00 KB");
119 
120         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.metaspace", "scale=1"});
121         output = new OutputAnalyzer(pb.start());
122         output.shouldHaveExitValue(0);
123         output.shouldMatch("MaxMetaspaceSize:.*210763776 bytes");
124     }
125 
126     public static void main(String args[]) throws Exception {
127         boolean testForCompressedClassSpace = false;
128         if (args[0].equals("with-compressed-class-space")) {
129             testForCompressedClassSpace = true;
130         } else if (args[0].equals("without-compressed-class-space")) {
131             testForCompressedClassSpace = false;
132         } else {
133             throw new IllegalArgumentException("Invalid argument: " + args[0]);
134         }
135         doTheTest(testForCompressedClassSpace);
136     }
137 }
