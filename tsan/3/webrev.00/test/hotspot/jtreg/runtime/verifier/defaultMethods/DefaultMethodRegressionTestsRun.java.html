24 /**
25  * @test
26  * @bug 8003639
27  * @summary defaultMethod resolution and verification using an URLClassLoader
28  * @modules jdk.compiler
29  *          jdk.zipfs
30  * @compile -XDignore.symbol.file=true DefaultMethodRegressionTestsRun.java
31  * @run main DefaultMethodRegressionTestsRun
32  */
33 import java.io.File;
34 import java.lang.reflect.Method;
35 import java.net.URL;
36 import java.net.URLClassLoader;
37 import java.nio.file.DirectoryStream;
38 import java.nio.file.Files;
39 import java.nio.file.Path;
40 /**
41  * This test is a variant of DefaultMethodRegressionTests, this one creates
42  * an URLClassLoader to load the support classes.
43  *
44  */
45 public class DefaultMethodRegressionTestsRun {
46     public static void main(String... args) throws Exception {
47         File scratchDir = new File(".");
48         File testDir = new File(scratchDir, "testdir");
49         testDir.mkdirs();
50         File srcFile = new File(new File(System.getProperty("test.src")),
51                 "DefaultMethodRegressionTests.java");
52         String[] javacargs = {
53             srcFile.getAbsolutePath(),
54             "-d",
55             testDir.getAbsolutePath()
56         };
57         com.sun.tools.javac.Main.compile(javacargs);
58         runClass(testDir, "DefaultMethodRegressionTests");
59     }
60     static void runClass(
61             File classPath,
62             String classname) throws Exception {
63         URL[] urls = {classPath.toURI().toURL()};
64         ClassLoader loader = new URLClassLoader(urls);
65         Class<?> c = loader.loadClass(classname);
66 
67         Class<?>[] argTypes = new Class<?>[]{String[].class};
68         Object[] methodArgs = new Object[]{null};
69 
70         Method method = c.getMethod("main", argTypes);
71         method.invoke(c, methodArgs);
72     }
73 }
