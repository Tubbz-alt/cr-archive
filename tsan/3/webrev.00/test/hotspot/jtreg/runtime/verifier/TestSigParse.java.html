24 /*
25  * @test
26  * @bug 8129579
27  * @summary Test that signatures are properly parsed when verification of local
28  *          classes is requested but verification of remote classes is not.
29  * @compile BadSignatures.jcod
30  * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+BytecodeVerificationLocal -XX:-BytecodeVerificationRemote TestSigParse
31  */
32 
33 public class TestSigParse {
34 
35     public static void main(String args[]) throws Throwable {
36         System.out.println("Regression test for bug 819579");
37 
38         // Test a FieldRef with a bad signature.
39         try {
40             Class newClass = Class.forName("BadFieldRef");
41             throw new RuntimeException("Expected ClasFormatError exception not thrown");
42         } catch (java.lang.ClassFormatError e) {
43             String eMsg = e.getMessage();
44             if (!eMsg.contains("Field") || !eMsg.contains("has illegal signature")) {
45                 throw new RuntimeException("Unexpected exception: " + eMsg);
46             }
47         }
48 
49         // Test a MethodRef with a bad signature.
50         try {
51             Class newClass = Class.forName("BadMethodRef");
52             throw new RuntimeException("Expected ClasFormatError exception not thrown");
53         } catch (java.lang.ClassFormatError e) {
54             String eMsg = e.getMessage();
55             if (!eMsg.contains("Method") || !eMsg.contains("has illegal signature")) {
56                 throw new RuntimeException("Unexpected exception: " + eMsg);
57             }
58         }
59 
60         // Test a method in a class with a bad signature.
61         try {
62             Class newClass = Class.forName("BadMethodSig");
63             throw new RuntimeException("Expected ClasFormatError exception not thrown");
64         } catch (java.lang.ClassFormatError e) {
65             String eMsg = e.getMessage();
66             if (!eMsg.contains("Class name is empty or contains illegal character")) {
67                 throw new RuntimeException("Unexpected exception: " + eMsg);
68             }
69         }
70     }
71 
72 }
