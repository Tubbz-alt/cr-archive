25 package shared;
26 
27 import java.util.HashMap;
28 import java.util.Map;
29 
30 /*******************************************************************/
31 // Class loader which has local class file storage in memory
32 /*******************************************************************/
33 
34 public class ByteArrayClassLoader extends ClassLoader {
35     private Map<String, byte[]> classes;
36 
37     public ByteArrayClassLoader() {
38         classes = new HashMap<String, byte[]>();
39     }
40 
41     public ByteArrayClassLoader(Map<String, byte[]> classes) {
42         this.classes = classes;
43     }
44 
45     public void appendClass(String name, byte[] classFile) {
46         classes.put(name, classFile);
47     }
48 
49     public Class findClass (String name) throws ClassNotFoundException {
50         if (classes.containsKey(name)) {
51             byte[] classData = classes.get(name);
52             return defineClass(name, classData, 0, classData.length);
53         } else {
54             throw new ClassNotFoundException("Can't find requested class: " + name);
55         }
56     }
57 }
