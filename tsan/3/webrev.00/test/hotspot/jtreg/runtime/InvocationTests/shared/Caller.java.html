25 package shared;
26 
27 import java.lang.reflect.InvocationTargetException;
28 
29 /*******************************************************************/
30 // Invoke different target method callers
31 /*******************************************************************/
32 
33 public class Caller {
34     private ClassLoader loader;
35     private Class paramClass;
36     private Class targetClass;
37     private boolean passed = true;
38     private Checker checker;
39 
40     public Caller(ClassLoader loader, Checker checker,
41                   Class paramClass, Class targetClass) {
42         this.loader = loader;
43         this.paramClass = paramClass;
44         this.targetClass = targetClass;
45         this.checker = checker;
46     }
47 
48     public boolean isPassed() {
49         return passed;
50     }
51 
52     public String call(String invoker) {
53         try {
54             Class clazz = loader.loadClass(invoker);
55 
56             String expectedBehavior = checker.check(clazz);
57 
58             String result = null;
59             Throwable exc = null;
60             try {
61                 java.lang.reflect.Method m = clazz.getDeclaredMethod("call", paramClass);
62                 result = (String) m.invoke(null, targetClass.newInstance());
63             } catch (InvocationTargetException e) {
64                 exc = e.getCause();
65             } catch (Throwable e) {
66                 exc = e;
67             }
68 
69             if (result == null) {
70                 if (exc != null) {
71                     result = exc.getClass().getName();
72                 } else {
73                     result = "null";
74                 }
75             }
76 
77             if (!(result.equals(expectedBehavior) || "".equals(expectedBehavior)) ) {
78                 passed = false;
79                 result = String.format("%s/%s", result, expectedBehavior);
80             }
81 
82             return Checker.abbreviateResult(result);
83         } catch (Exception e) {
84             throw new RuntimeException(e);
85         }
86     }
87 }
