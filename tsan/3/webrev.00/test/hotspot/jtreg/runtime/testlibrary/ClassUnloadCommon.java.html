 31 import java.io.File;
 32 import java.io.FileInputStream;
 33 import java.io.IOException;
 34 import java.net.MalformedURLException;
 35 import java.net.URL;
 36 import java.net.URLClassLoader;
 37 import java.nio.file.Path;
 38 import java.nio.file.Paths;
 39 import java.util.ArrayList;
 40 import java.util.stream.Stream;

 42 public class ClassUnloadCommon {
 43     public static class TestFailure extends RuntimeException {
 44         TestFailure(String msg) {
 45             super(msg);
 46         }
 47     }
 48 
 49     public static void failIf(boolean value, String msg) {
 50         if (value) throw new TestFailure("Test failed: " + msg);
 51     }
 52 
 53     private static volatile Object dummy = null;
 54     private static void allocateMemory(int kilobytes) {
 55         ArrayList<byte[]> l = new ArrayList<>();
 56         dummy = l;
 57         for (int i = kilobytes; i > 0; i -= 1) {
 58             l.add(new byte[1024]);
 59         }
 60         l = null;
 61         dummy = null;
 62     }
 63 
 64     public static void triggerUnloading() {
 65         allocateMemory(16 * 1024); // force young collection
 66         System.gc();
 67     }
 68 
 69     /**
 70      * Creates a class loader that loads classes from {@code ${test.class.path}}
 71      * before delegating to the system class loader.
 72      */
 73     public static ClassLoader newClassLoader() {
 74         String cp = System.getProperty("test.class.path", ".");
 75         URL[] urls = Stream.of(cp.split(File.pathSeparator))
 76                 .map(Paths::get)
 77                 .map(ClassUnloadCommon::toURL)
 78                 .toArray(URL[]::new);
 79         return new URLClassLoader("ClassUnloadCommonClassLoader", urls, new ClassUnloadCommon().getClass().getClassLoader()) {
 80             @Override
 81             public Class<?> loadClass(String cn, boolean resolve)
 82                 throws ClassNotFoundException
 83             {
 84                 synchronized (getClassLoadingLock(cn)) {
 85                     Class<?> c = findLoadedClass(cn);
 86                     if (c == null) {
 87                         try {
 88                             c = findClass(cn);
 89                         } catch (ClassNotFoundException e) {
 90                             c = getParent().loadClass(cn);
 91                         }
 92 
 93                     }
 94                     if (resolve) {
 95                         resolveClass(c);
 96                     }
 97                     return c;
 98                 }
 99             }
100         };
101     }
102 
103     static URL toURL(Path path) {
104         try {
105             return path.toUri().toURL();
106         } catch (MalformedURLException e) {
107             throw new RuntimeException(e);
108         }
109     }
110 
111     // Get data for pre-compiled class file to load.
112     public static byte[] getClassData(String name) {
113         try {
114            String TempName = name.replaceAll("\\.", "/");
115            String currentDir = System.getProperty("test.classes");
116            String filename = currentDir + File.separator + TempName + ".class";
117            System.out.println("filename is " + filename);
118            FileInputStream fis = new FileInputStream(filename);
119            byte[] b = new byte[5000];
120            int cnt = fis.read(b, 0, 5000);
121            byte[] c = new byte[cnt];
122            for (int i=0; i<cnt; i++) c[i] = b[i];
123               return c;
124         } catch (IOException e) {
125            return null;
126         }
127     }
