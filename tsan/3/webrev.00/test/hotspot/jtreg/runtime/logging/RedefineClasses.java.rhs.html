24 /*
25  * @test
26  * @bug 8197901 8209758
27  * @summary Redefine classes with enabling logging to verify Klass:external_name() during GC.
28  * @comment This test is simplified version of serviceability/jvmti/RedefineClasses/RedefineRunningMethods.java.
29  * @library /test/lib
30  * @modules java.compiler
31  *          java.instrument

32  * @run main RedefineClassHelper
33  * @run main/othervm -Xmx256m -XX:MaxMetaspaceSize=64m -javaagent:redefineagent.jar -Xlog:all=trace:file=all.log RedefineClasses
34  */
35 
36 // package access top-level class to avoid problem with RedefineClassHelper
37 // and nested types.
38 class RedefineClasses_B {
39     public static void test() {
40     }
41 }
42 
43 public class RedefineClasses {
44     static Object[] obj = new Object[20];
45     public static String newB =
46             "class RedefineClasses_B {" +
47             "    public static void test() { " +
48             "    }" +
49             "}";
50 
51     public static void main(String[] args) throws Exception {
52         RedefineClassHelper.redefineClass(RedefineClasses_B.class, newB);
53         RedefineClasses_B.test();
54         for (int i = 0; i < 20 ; i++) {
55             obj[i] = new byte[1024 * 1024];
56             System.gc();
57         }
58     }
59 }
