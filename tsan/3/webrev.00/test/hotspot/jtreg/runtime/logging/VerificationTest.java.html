24 /*
25  * @test
26  * @bug 8150083 8234656
27  * @summary test enabling and disabling verification logging and verification log levels
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  * @run driver VerificationTest
32  */
33 
34 import jdk.test.lib.process.OutputAnalyzer;
35 import jdk.test.lib.process.ProcessTools;
36 
37 public class VerificationTest {
38 
39     static void analyzeOutputOn(ProcessBuilder pb, boolean isLogLevelInfo) throws Exception {
40         OutputAnalyzer output = new OutputAnalyzer(pb.start());
41         output.shouldContain("[verification]");
42         output.shouldContain("Verifying class VerificationTest$InternalClass with new format");
43         output.shouldContain("Verifying method VerificationTest$InternalClass.<init>()V");
44         output.shouldContain("End class verification for: VerificationTest$InternalClass");
45 
46         if (isLogLevelInfo) {
47             // logging level 'info' should not output stack map and opcode data.
48             output.shouldNotContain("[verification] StackMapTable: frame_count");
49             output.shouldNotContain("[verification] offset = 0,  opcode =");
50 
51         } else { // log level debug
52             output.shouldContain("[debug][verification] StackMapTable: frame_count");
53             output.shouldContain("[debug][verification] offset = 0,  opcode =");
54         }
55         output.shouldHaveExitValue(0);
56     }
57 
58     static void analyzeOutputOff(ProcessBuilder pb) throws Exception {
59         OutputAnalyzer output = new OutputAnalyzer(pb.start());
60         output.shouldNotContain("[verification]");
61         output.shouldHaveExitValue(0);
62     }
63 
64     public static void main(String[] args) throws Exception {
65         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-Xlog:verification=info",
66                                                                   InternalClass.class.getName());
67         analyzeOutputOn(pb, true);
68 
69         pb = ProcessTools.createJavaProcessBuilder("-Xlog:verification=off",
70                                                    InternalClass.class.getName());
71         analyzeOutputOff(pb);
72 
73         // logging level 'debug' should output stackmaps and bytecode data.
74         pb = ProcessTools.createJavaProcessBuilder("-Xlog:verification=debug",
75                                                    InternalClass.class.getName());
76         analyzeOutputOn(pb, false);
77     }
78 
79     public static class InternalClass {
80         public static void main(String[] args) throws Exception {
81             System.out.println("VerificationTest");
82         }
83     }
84 }
