 24 /**
 25  * @test
 26  * @bug 8167108
 27  * @summary Stress test java.lang.Thread.suspend() at thread exit.
 28  * @run main/othervm -Xlog:thread+smr=debug SuspendAtExit
 29  * @run main/othervm -Xlog:thread+smr=debug -XX:+UnlockDiagnosticVMOptions -XX:GuaranteedSafepointInterval=1 -XX:+HandshakeALot SuspendAtExit
 30  */
 31 
 32 import java.util.concurrent.CountDownLatch;
 33 
 34 public class SuspendAtExit extends Thread {
 35     final static int N_THREADS = 32;
 36     final static int N_LATE_CALLS = 10000;
 37 
 38     public CountDownLatch exitSyncObj = new CountDownLatch(1);
 39     public CountDownLatch startSyncObj = new CountDownLatch(1);
 40 
 41     @Override
 42     public void run() {
 43         // Tell main thread we have started.
 44         startSyncObj.countDown();
 45         try {
 46             // Wait for main thread to interrupt us so we
 47             // can race to exit.
 48             exitSyncObj.await();
 49         } catch (InterruptedException e) {
 50             // ignore because we expect one
 51         }
 52     }
 53 
 54     public static void main(String[] args) {
 55         SuspendAtExit threads[] = new SuspendAtExit[N_THREADS];
 56 
 57         for (int i = 0; i < N_THREADS; i++ ) {
 58             threads[i] = new SuspendAtExit();
 59             int late_count = 1;
 60             threads[i].start();
 61             try {
 62                 // Wait for the worker thread to get going.
 63                 threads[i].startSyncObj.await();
 64 
 65                 // This interrupt() call will break the worker out
 66                 // of the exitSyncObj.await() call and the suspend()
 67                 // calls will come in during thread exit.
 68                 threads[i].interrupt();
 69                 for (; late_count <= N_LATE_CALLS; late_count++) {
 70                     threads[i].suspend();
 71 
 72                     if (!threads[i].isAlive()) {
 73                         // Done with Thread.suspend() calls since
 74                         // thread is not alive.
 75                         break;
 76                     }
 77                     threads[i].resume();
 78                 }
 79             } catch (InterruptedException e) {
 80                 throw new Error("Unexpected: " + e);
 81             }
 82 
 83             System.out.println("INFO: thread #" + i + ": made " + late_count +
 84                                " late calls to java.lang.Thread.suspend()");
 85             System.out.println("INFO: thread #" + i + ": N_LATE_CALLS==" +
 86                                N_LATE_CALLS + " value is " +
 87                                ((late_count >= N_LATE_CALLS) ? "NOT " : "") +
 88                                "large enough to cause a Thread.suspend() " +
 89                                "call after thread exit.");
 90 
 91             try {
 92                 threads[i].join();
 93             } catch (InterruptedException e) {
 94                 throw new Error("Unexpected: " + e);
 95             }
 96             threads[i].suspend();
 97             threads[i].resume();
 98             if (threads[i].isAlive()) {
 99                 throw new Error("Expected !Thread.isAlive() after thread #" +
100                                 i + " has been join()'ed");
101             }
102         }
103 
104         String cmd = System.getProperty("sun.java.command");
105         if (cmd != null && !cmd.startsWith("com.sun.javatest.regtest.agent.MainWrapper")) {
106             // Exit with success in a non-JavaTest environment:
107             System.exit(0);
108         }
109     }
110 }
