 24 /**
 25  * @test
 26  * @bug 8167108
 27  * @summary Stress test java.lang.Thread.suspend() at thread exit.
 28  * @run main/othervm -Xlog:thread+smr=debug SuspendAtExit

 29  */
 30 
 31 import java.util.concurrent.CountDownLatch;
 32 
 33 public class SuspendAtExit extends Thread {
 34     final static int N_THREADS = 32;
 35     final static int N_LATE_CALLS = 10000;
 36 
 37     public CountDownLatch exitSyncObj = new CountDownLatch(1);
 38     public CountDownLatch startSyncObj = new CountDownLatch(1);
 39 
 40     @Override
 41     public void run() {
 42         // Tell main thread we have started.
 43         startSyncObj.countDown();
 44         try {
 45             // Wait for main thread to interrupt us so we
 46             // can race to exit.
 47             exitSyncObj.await();
 48         } catch (InterruptedException e) {
 49             // ignore because we expect one
 50         }
 51     }
 52 
 53     public static void main(String[] args) {
 54         SuspendAtExit threads[] = new SuspendAtExit[N_THREADS];
 55 
 56         for (int i = 0; i < N_THREADS; i++ ) {
 57             threads[i] = new SuspendAtExit();
 58             int late_count = 1;
 59             threads[i].start();
 60             try {
 61                 // Wait for the worker thread to get going.
 62                 threads[i].startSyncObj.await();
 63 
 64                 // This interrupt() call will break the worker out
 65                 // of the exitSyncObj.await() call and the suspend()
 66                 // calls will come in during thread exit.
 67                 threads[i].interrupt();
 68                 for (; late_count <= N_LATE_CALLS; late_count++) {
 69                     threads[i].suspend();
 70 
 71                     if (!threads[i].isAlive()) {
 72                         // Done with Thread.suspend() calls since
 73                         // thread is not alive.
 74                         break;
 75                     }
 76                     threads[i].resume();
 77                 }
 78             } catch (InterruptedException e) {
 79                 throw new Error("Unexpected: " + e);
 80             }
 81 
 82             System.out.println("INFO: thread #" + i + ": made " + late_count +
 83                                " late calls to java.lang.Thread.suspend()");
 84             System.out.println("INFO: thread #" + i + ": N_LATE_CALLS==" +
 85                                N_LATE_CALLS + " value is " +
 86                                ((late_count >= N_LATE_CALLS) ? "NOT " : "") +
 87                                "large enough to cause a Thread.suspend() " +
 88                                "call after thread exit.");
 89 
 90             try {
 91                 threads[i].join();
 92             } catch (InterruptedException e) {
 93                 throw new Error("Unexpected: " + e);
 94             }
 95             threads[i].suspend();
 96             threads[i].resume();
 97             if (threads[i].isAlive()) {
 98                 throw new Error("Expected !Thread.isAlive() after thread #" +
 99                                 i + " has been join()'ed");
100             }
101         }
102 
103         String cmd = System.getProperty("sun.java.command");
104         if (cmd != null && !cmd.startsWith("com.sun.javatest.regtest.agent.MainWrapper")) {
105             // Exit with success in a non-JavaTest environment:
106             System.exit(0);
107         }
108     }
109 }
