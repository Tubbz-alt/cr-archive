25 /*
26  * @test
27  * @summary Tests the format checking of hotspot/src/closed/share/vm/classfile/classListParser.cpp.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
32  * @compile ../test-classes/Hello.java test-classes/CustomLoadee.java test-classes/CustomLoadee2.java
33  *          test-classes/CustomInterface2_ia.java test-classes/CustomInterface2_ib.java
34  * @run driver ClassListFormatD
35  */
36 
37 public class ClassListFormatD extends ClassListFormatBase {
38     static {
39         // Uncomment the following line to run only one of the test cases
40         // ClassListFormatBase.RUN_ONLY_TEST = "TESTCASE D1";
41     }
42 
43     public static void main(String[] args) throws Throwable {
44         String appJar = JarBuilder.getOrCreateHelloJar();
45         String customJarPath = JarBuilder.build("ClassListFormatD", "CustomLoadee",
46                                                 "CustomLoadee2", "CustomInterface2_ia",
47                                                 "CustomInterface2_ib");
48 
49         //----------------------------------------------------------------------
50         // TESTGROUP D: bad use of IDs
51         //----------------------------------------------------------------------
52         dumpShouldFail(
53             "TESTCASE D1: duplicated id:",
54             appJar, classlist(
55                 "Hello",
56                 "java/lang/Object id: 1",
57                 "CustomLoadee id: 1 super: 1 source: " + customJarPath
58             ),
59             "Duplicated ID 1 for class CustomLoadee");
60 
61         dumpShouldFail(
62             "TESTCASE D2: bad ID for super:",
63             appJar, classlist(
64                 "Hello",
65                 "java/lang/Object id: 1",
66                 "CustomLoadee id: 2 super: 2 source: " + customJarPath
67             ),
68             "Super class id 2 is not yet loaded");
69 
70         dumpShouldFail(
71             "TESTCASE D3: bad ID in interfaces:",
72             appJar, classlist(
73                 "Hello",
74                 "java/lang/Object id: 1",
75                 "CustomLoadee id: 2 super: 1 interfaces: 2 source: " + customJarPath
76             ),
77             "Interface id 2 is not yet loaded");
78     }
79 }
