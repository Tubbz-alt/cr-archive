23 import java.io.File;
24 import java.util.List;
25 import java.util.Scanner;
26 import sun.hotspot.WhiteBox;
27 
28 public class LoadClasses {
29 
30     public static void main (String[] args) throws Throwable {
31         String classList = args[0];
32         Scanner sc = new Scanner(new File(classList));
33         WhiteBox wb = WhiteBox.getWhiteBox();
34         int count = 0;
35         while (sc.hasNextLine()) {
36             String cn = sc.nextLine().replace('/', '.');
37             try {
38                 Class<?> cls = Class.forName(cn, false, LoadClasses.class.getClassLoader());
39                 wb.linkClass(cls);
40                 count++;
41             } catch (Throwable ex) {
42                 System.out.println("Loading failed: " + cn);
43                 System.out.println(ex.toString());
44             }
45         }
46         System.out.println("Loaded classes = " + count);
47     }
48 }
