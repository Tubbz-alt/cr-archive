25 /*
26  * @test
27  * @summary Test the -XX:+IgnoreEmptyClassPaths flag
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/Hello.java
31  * @compile test-classes/HelloMore.java
32  * @run driver IgnoreEmptyClassPaths
33  */
34 
35 import java.io.File;
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class IgnoreEmptyClassPaths {
39 
40   public static void main(String[] args) throws Exception {
41     String jar1 = JarBuilder.getOrCreateHelloJar();
42     String jar2 = JarBuilder.build("IgnoreEmptyClassPaths_more", "HelloMore");
43 
44     String sep = File.pathSeparator;
45     String cp_dump = jar1 + sep + jar2 + sep;
46     String cp_exec = sep + jar1 + sep + sep + jar2 + sep;
47 
48     TestCommon.testDump(cp_dump, TestCommon.list("Hello", "HelloMore"),
49                         "-XX:+TraceClassPaths", "-XX:+IgnoreEmptyClassPaths");
50 
51     TestCommon.run(
52         "-verbose:class",
53         "-cp", cp_exec,
54         "-XX:+IgnoreEmptyClassPaths", // should affect classpath even if placed after the "-cp" argument
55         "-XX:+TraceClassPaths",
56         "HelloMore")
57       .assertNormalExit();
58   }
59 }
