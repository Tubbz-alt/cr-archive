25 /*
26  * @test
27  * @summary Test for field annotations.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/FieldAnnotationsApp.java test-classes/MyAnnotation.java
31  * @run driver FieldAnnotationsTest
32  */
33 
34 import jdk.test.lib.process.OutputAnalyzer;
35 
36 // This is a test for the handling of multi-dimensional Arrays in MetaspaceClosure.
37 //
38 // We choose FieldAnnotations because they happen to be implemented as a multi-dimension
39 // Array (Annotations::_fields_annotations, which is of type Array<Array<unsigned char>*>*,
40 // and is handled by the template class PointerArrayRef<T> in metaspaceClosure.hpp).
41 //
42 // Specifically, we are testing the following C code, where _fields_annotations is non-NULL:
43 //
44 // void Annotations::metaspace_pointers_do(MetaspaceClosure* it) {
45 //   ...
46 //   it->push(&_fields_annotations);
47 //
48 // which will be matched with the function
49 //
50 // template <typename T> void MetaspaceClosure::push(Array<T*>** mpp, Writability w = _default)
51 //
52 public class FieldAnnotationsTest {
53     public static void main(String[] args) throws Exception {
54         String[] ARCHIVE_CLASSES = {"FieldAnnotationsApp", "MyAnnotation"};
55         String appJar = JarBuilder.build("FieldAnnotationsTest", ARCHIVE_CLASSES);
56 
57         OutputAnalyzer dumpOutput = TestCommon.dump(
58                 appJar, ARCHIVE_CLASSES);
59         TestCommon.checkDump(dumpOutput);
60 
61         OutputAnalyzer execOutput = TestCommon.exec(appJar, "FieldAnnotationsApp");
62         TestCommon.checkExec(execOutput, "Field annotations are OK.");
63     }
64 }
