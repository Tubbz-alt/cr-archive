25 /*
26  * @test
27  * @summary Abort dumping if any of the new jigsaw vm options is specified.
28  * @requires vm.cds
29  * @library /test/lib ..
30  * @compile ../test-classes/Hello.java
31  * @run driver CheckUnsupportedDumpingOptions
32  */
33 
34 import jdk.test.lib.compiler.InMemoryJavaCompiler;
35 import jdk.test.lib.process.OutputAnalyzer;
36 
37 public class CheckUnsupportedDumpingOptions {
38     private static final String[] jigsawOptions = {
39         "--limit-modules",
40         "--upgrade-module-path",
41         "--patch-module"
42     };
43     private static final String[] optionValues = {
44         "mymod",
45         ".",
46         "java.naming=javax.naming.spi.NamingManger"
47     };
48 
49     public static void main(String[] args) throws Exception {
50         String source = "package javax.naming.spi; "                +
51                         "public class NamingManager { "             +
52                         "    static { "                             +
53                         "        System.out.println(\"I pass!\"); " +
54                         "    } "                                    +
55                         "}";
56         ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
57             InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
58             "mods/java.naming");
59 
60         JarBuilder.build("hello", "Hello");
61         String appJar = TestCommon.getTestJar("hello.jar");
62         String appClasses[] = {"Hello"};
63         for (int i = 0; i < jigsawOptions.length; i++) {
64             OutputAnalyzer output;
65             output = TestCommon.dump(appJar, appClasses, "-Xlog:cds,cds+hashtables",
66                                      jigsawOptions[i], optionValues[i]);
67             output.shouldContain("Cannot use the following option " +
68                 "when dumping the shared archive: " + jigsawOptions[i])
69                   .shouldHaveExitValue(1);
70         }
71     }
72 }
