25 /*
26  * @test
27  * @requires vm.cds
28  * @summary a simple test to ensure that a directory in the --patch-module
29  *          option does not affect dump process
30  * @library ../..
31  * @library /test/hotspot/jtreg/testlibrary
32  * @library /test/lib
33  * @build PatchMain
34  * @run driver PatchDir
35  */
36 
37 import java.io.File;
38 import jdk.test.lib.compiler.InMemoryJavaCompiler;
39 
40 public class PatchDir {
41     private static String moduleJar;
42 
43     public static void main(String args[]) throws Throwable {
44 
45         // Create a class file in the module java.naming. This class file
46         // will be put in the javanaming.jar file.
47         String source = "package javax.naming.spi; "                +
48                         "public class NamingManager { "             +
49                         "    static { "                             +
50                         "        System.out.println(\"I pass!\"); " +
51                         "    } "                                    +
52                         "}";
53 
54         String classDir = System.getProperty("test.classes");
55         ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
56              InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
57              classDir);
58 
59         JarBuilder.build("javanaming", "javax/naming/spi/NamingManager");
60         moduleJar = TestCommon.getTestJar("javanaming.jar");
61 
62         System.out.println("Test dumping with --patch-module");
63         TestCommon.dump(null,
64             TestCommon.list("javax/naming/spi/NamingManager"),
65             "--patch-module=java.naming=" + moduleJar,
66             "-Xlog:class+load",
67             "PatchMain", "javax.naming.spi.NamingManager")
68             .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module")
69             .shouldHaveExitValue(1);
70     }
71 }
