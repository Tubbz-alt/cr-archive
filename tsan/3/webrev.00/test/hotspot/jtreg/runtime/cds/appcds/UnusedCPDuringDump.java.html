25 /*
26  * @test
27  * @bug 8209385
28  * @summary non-empty dir in -cp should be fine during dump time if only classes
29  *          from the system modules are being loaded even though some are
30  *          defined to the PlatformClassLoader and AppClassLoader.
31  * @requires vm.cds
32  * @library /test/lib
33  * @compile test-classes/Hello.java
34  * @run main/othervm -Dtest.cds.copy.child.stdout=false UnusedCPDuringDump
35  */
36 
37 import java.io.File;
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class UnusedCPDuringDump {
41 
42     public static void main(String[] args) throws Exception {
43         File dir = new File(System.getProperty("user.dir"));
44         File emptydir = new File(dir, "emptydir");
45         emptydir.mkdir();
46         String appJar = JarBuilder.getOrCreateHelloJar();
47 
48         OutputAnalyzer output = TestCommon.dump(dir.getPath(),
49             TestCommon.list("sun/util/resources/cldr/provider/CLDRLocaleDataMetaInfo",
50                             "com/sun/tools/sjavac/client/ClientMain"),
51                             "-Xlog:class+path=info,class+load=debug");
52         TestCommon.checkDump(output,
53                              "[class,load] sun.util.resources.cldr.provider.CLDRLocaleDataMetaInfo",
54                              "for instance a 'jdk/internal/loader/ClassLoaders$PlatformClassLoader'",
55                              "[class,load] com.sun.tools.sjavac.client.ClientMain",
56                              "for instance a 'jdk/internal/loader/ClassLoaders$AppClassLoader'");
57 
58         String jsaOpt = "-XX:SharedArchiveFile=" + TestCommon.getCurrentArchiveName();
59         TestCommon.run("-cp", appJar, jsaOpt, "-Xlog:class+path=info,class+load=debug", "Hello")
60             .assertNormalExit("Hello World");
61   }
62 }
