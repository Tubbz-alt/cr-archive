24 /**
25  * @test Default CDS archive file
26  * @summary JDK platforms/binaries do not support default CDS archive should
27  *          not contain classes.jsa in the default location.
28  * @requires vm.cds
29  * @library /test/lib
30  * @build sun.hotspot.WhiteBox
31  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
32  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
33  *      -XX:+WhiteBoxAPI CheckDefaultArchiveFile
34  */
35 import java.nio.file.Files;
36 import java.nio.file.Path;
37 import java.nio.file.Paths;
38 import jdk.test.lib.Platform;
39 import jdk.test.lib.cds.CDSTestUtils;
40 import jtreg.SkippedException;
41 import sun.hotspot.WhiteBox;
42 
43 public class CheckDefaultArchiveFile {
44     public static void main(String[] args) throws Exception {
45         WhiteBox wb = WhiteBox.getWhiteBox();
46         String osArch = Platform.getOsArch();
47         String vmName = System.getProperty("java.vm.name");
48         String vmString = vmName + "(" + osArch + ")";
49         String jsaString = wb.getDefaultArchivePath();
50         System.out.println("classes.jsa location:" + jsaString);
51         if (jsaString == null) {
52             if (Platform.isDefaultCDSArchiveSupported()) {
53                 throw new RuntimeException("default CDS archive supported, but classes.jsa path null");
54             }
55         } else {
56             Path jsa = Paths.get(jsaString);
57             if (Platform.isDefaultCDSArchiveSupported()) {
58                 if (Files.exists(jsa)) {
59                     System.out.println("Passed. " + vmString +
60                                        ": has default classes.jsa file");
61                 } else {
62                     throw new RuntimeException(vmString + "has no " + jsaString);
63                 }
64             } else {
65                 throw new SkippedException("Default CDS archive is not supported");
66             }
67         }
68     }
69 }
