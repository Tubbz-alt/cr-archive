 25 /*
 26  * @test
 27  * @summary Adding extra symbols into CDS archive using -XX:SharedArchiveConfigFile
 28  * @requires vm.cds
 29  * @library /test/lib
 30  * @compile test-classes/Hello.java
 31  * @run driver ExtraSymbols
 32  */
 33 
 34 import java.io.*;
 35 import jdk.test.lib.process.OutputAnalyzer;
 36 import jdk.test.lib.cds.CDSTestUtils;
 37 
 38 public class ExtraSymbols {
 39     static final String CDS_LOGGING = "-Xlog:cds,cds+hashtables";
 40     public static void main(String[] args) throws Exception {
 41         String appJar = JarBuilder.getOrCreateHelloJar();
 42 
 43         // 1. Dump without extra symbols.
 44         OutputAnalyzer output = TestCommon.dump(appJar, TestCommon.list("Hello"),
 45                                                 CDS_LOGGING);
 46         checkOutput(output);
 47         int numEntries1 = numOfEntries(output);
 48 
 49         // 2. Dump an archive with lots of extra symbols.
 50         output = TestCommon.dump(appJar, TestCommon.list("Hello"), CDS_LOGGING,
 51             "-XX:SharedArchiveConfigFile=" + makeLotsExtraSymbols());
 52         checkOutput(output);
 53         int numEntries2 = numOfEntries(output);
 54         if (numEntries2 <= numEntries1) {
 55             throw new RuntimeException("No extra symbols added to archive");
 56         }
 57         output = TestCommon.exec(appJar, "Hello");
 58         TestCommon.checkExec(output);
 59 
 60         // 3. Dump with invalid symbol files. Dumping should fail.
 61         String invalid_symbol_files[] = {"ExtraSymbols.invalid_1.txt",
 62                                          "ExtraSymbols.invalid_2.txt",
 63                                          "ExtraSymbols.invalid_3.txt"};
 64         String err_msgs[] = {"Corrupted at line",
 65                              "wrong version of hashtable dump file",
 66                              "Corrupted at line"};
 67         for (int i = 0; i < invalid_symbol_files.length; i++) {
 68             output = TestCommon.dump(appJar, TestCommon.list("Hello"),
 69                                      "-XX:SharedArchiveConfigFile=" +
 70                                      TestCommon.getSourceFile(invalid_symbol_files[i]));
 71             output.shouldContain("Error occurred during initialization of VM");
 72             output.shouldContain(err_msgs[i]);
 73         }
 74     }
 75 
 76     static int numOfEntries(OutputAnalyzer output) {
 77         String s = output.firstMatch("Number of entries       : .*");
 78         String subs[] = s.split("[:]");
 79         int numEntries = Integer.parseInt(subs[1].trim());
 80         return numEntries;
 81     }
 82 
 83     static void checkOutput(OutputAnalyzer output) throws Exception {
 84         output.shouldContain("Loading classes to share");
 85         output.shouldContain("Shared symbol table stats -------- base:");
 86         output.shouldHaveExitValue(0);
 87     }
 88 
 89     static String makeLotsExtraSymbols() throws Exception {
 90         String fileName = "LotsExtraSymbols.txt";
 91         File f = new File(fileName);
 92         try (FileWriter fw = new FileWriter(f)) {
 93             fw.write("VERSION: 1.0\n");
 94             fw.write("@SECTION: Symbol\n");
 95             appendSymbol(fw, "This file is auto-generated by test/hotspot/jtreg/runtime/cds/appcds/ExtraSymbols.java. Do not edit.");
 96             appendSymbol(fw, "Hello");
 97             appendSymbol(fw, ""); // empty string
 98             appendSymbol(fw, "Hello_\u0001");   // <128 escaping with \x
 99             appendSymbol(fw, "Hello_\u00ff");   // <256 escaping with \x
100             appendSymbol(fw, "Hello_\u1234");   // >= 256 escaping with \x
101             appendSymbol(fw, "Hello_\uffff");   // >= 256 escaping with \x
102 
103             for (int i = 0; i < 10000; i++) {
104                 appendSymbol(fw, "NewSymbol" + Integer.toString(i));
105             }
106         }
107         return fileName;
108     }
109 
110     private static void appendSymbol(FileWriter fw, String symbol) throws Exception {
111         fw.write(CDSTestUtils.formatArchiveConfigSymbol(symbol));
112         fw.write("\n");
113     }
114 }
