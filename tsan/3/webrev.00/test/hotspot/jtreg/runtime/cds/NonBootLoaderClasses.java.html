24 /*
25  * @test NonBootLoaderClasses
26  * @summary Test to ensure platform and app classes are archived when specified in classlist
27  * @requires vm.cds
28  * @library /test/lib
29  * @run driver NonBootLoaderClasses
30  */
31 
32 import jdk.test.lib.cds.CDSOptions;
33 import jdk.test.lib.cds.CDSTestUtils;
34 import jdk.test.lib.process.OutputAnalyzer;
35 
36 public class NonBootLoaderClasses {
37     public static void main(String[] args) throws Exception {
38         final String PLATFORM_CLASS = "jdk/dynalink/DynamicLinker";
39         final String APP_CLASS = "com/sun/tools/javac/Main";
40         String[] classes = {PLATFORM_CLASS, APP_CLASS};
41         String classList =
42             CDSTestUtils.makeClassList(classes).getPath();
43         String archiveName = "NonBootLoaderClasses.jsa";
44         CDSOptions opts = (new CDSOptions())
45             .addPrefix("-XX:ExtraSharedClassListFile=" + classList, "-Djava.class.path=")
46             .setArchiveName(archiveName);
47         CDSTestUtils.createArchiveAndCheck(opts);
48 
49         // Print the shared dictionary and inspect the output
50         opts = (new CDSOptions())
51             .setUseVersion(false)
52             .addSuffix("-Djava.class.path=",
53                 "-XX:+UnlockDiagnosticVMOptions", "-XX:SharedArchiveFile=./" + archiveName,
54                 "-XX:+PrintSharedArchiveAndExit", "-XX:+PrintSharedDictionary");
55         CDSTestUtils.run(opts)
56             .assertNormalExit(output -> {
57                 output.shouldContain("archive is valid");
58                 output.shouldContain(PLATFORM_CLASS.replace('/', '.'));
59                 output.shouldContain(APP_CLASS.replace('/', '.'));
60             });
61    }
62 }
