 25 /*
 26  * @test
 27  * @summary Handling of directories in -cp is based on the classlist
 28  * @requires vm.cds
 29  * @library /test/lib
 30  * @compile test-classes/Hello.java
 31  * @compile test-classes/Super.java
 32  * @run driver DirClasspathTest
 33  */
 34 
 35 import jdk.test.lib.Platform;
 36 import jdk.test.lib.process.OutputAnalyzer;
 37 import java.io.File;
 38 import java.nio.file.Files;
 39 import java.nio.file.Path;
 40 import java.nio.file.Paths;
 41 import java.util.Arrays;
 42 
 43 public class DirClasspathTest {
 44     private static final int MAX_PATH = 260;
 45 
 46     // We add helloJar into the classpath to be compatible with TestCommon.DYNAMIC_DUMP
 47     static OutputAnalyzer doDump(String path, String classList[],
 48                                         String... suffix) throws Exception {
 49         String helloJar = JarBuilder.getOrCreateHelloJar();
 50         return TestCommon.dump(helloJar + File.pathSeparator + path, classList, suffix);
 51     }
 52 
 53     public static void main(String[] args) throws Exception {
 54         File dir = new File(System.getProperty("user.dir"));
 55         File emptydir = new File(dir, "emptydir");
 56         emptydir.mkdir();
 57 
 58 
 59         /////////////////////////////////////////////////////////////////
 60         // The classlist only contains boot class in following test cases
 61         /////////////////////////////////////////////////////////////////
 62         String bootClassList[] = {"java/lang/Object"};
 63 
 64         // Empty dir in -cp: should be OK
 65         OutputAnalyzer output;
 66         output = doDump(emptydir.getPath(), bootClassList, "-Xlog:class+path=info");
 67         TestCommon.checkDump(output);
 68 
 69         // Long path to empty dir in -cp: should be OK
 70         Path classDir = Paths.get(System.getProperty("test.classes"));
 71         Path destDir = classDir;
 72         int subDirLen = MAX_PATH - classDir.toString().length() - 2;
 73         if (subDirLen > 0) {
 74             char[] chars = new char[subDirLen];
 75             Arrays.fill(chars, 'x');
 76             String subPath = new String(chars);
 77             destDir = Paths.get(System.getProperty("test.classes"), subPath);
 78         }
 79         File longDir = destDir.toFile();
 80         longDir.mkdir();
 81         File subDir = new File(longDir, "subdir");
 82         subDir.mkdir();
 83         output = doDump(subDir.getPath(), bootClassList, "-Xlog:class+path=info");
 84         TestCommon.checkDump(output);
 85 
 86         // Non-empty dir in -cp: should be OK
 87         // <dir> is not empty because it has at least one subdirectory, i.e., <emptydir>
 88         output = doDump(dir.getPath(), bootClassList, "-Xlog:class+path=info");
 89         TestCommon.checkDump(output);
 90 
 91         // Long path to non-empty dir in -cp: should be OK
 92         // <dir> is not empty because it has at least one subdirectory, i.e., <emptydir>
 93         output = doDump(longDir.getPath(), bootClassList, "-Xlog:class+path=info");
 94         TestCommon.checkDump(output);
 95 
 96         /////////////////////////////////////////////////////////////////
 97         // The classlist contains non-boot class in following test cases
 98         /////////////////////////////////////////////////////////////////
 99         String appClassList[] = {"java/lang/Object", "com/sun/tools/javac/Main"};
100 
101         // Non-empty dir in -cp: should be OK (as long as no classes were loaded from there)
102         output = doDump(dir.getPath(), appClassList, "-Xlog:class+path=info");
103         TestCommon.checkDump(output);
104 
105         // Long path to non-empty dir in -cp: should be OK (as long as no classes were loaded from there)
106         output = doDump(longDir.getPath(), appClassList, "-Xlog:class+path=info");
107         TestCommon.checkDump(output);
108 
109         /////////////////////////////////////////////////////////////////
110         // Loading an app class from a directory
111         /////////////////////////////////////////////////////////////////
112         String appClassList2[] = {"Super", "java/lang/Object", "com/sun/tools/javac/Main"};
113         // Non-empty dir in -cp: should report error if a class is loaded from it
114         output = doDump(classDir.toString(), appClassList2, "-Xlog:class+path=info,class+load=trace");
115         output.shouldNotHaveExitValue(0);
116         output.shouldContain("Cannot have non-empty directory in paths");
117 
118         // Long path to non-empty dir in -cp: should report error if a class is loaded from it
119         File srcClass = new File(classDir.toFile(), "Super.class");
120         File destClass = new File(longDir, "Super.class");
121         Files.copy(srcClass.toPath(), destClass.toPath());
122         output = doDump(longDir.getPath(), appClassList2, "-Xlog:class+path=info");
123         output.shouldNotHaveExitValue(0);
124         output.shouldContain("Cannot have non-empty directory in paths");
125     }
126 }
