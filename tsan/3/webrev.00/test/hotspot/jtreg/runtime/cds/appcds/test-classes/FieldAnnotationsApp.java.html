25 import java.lang.annotation.Annotation;
26 import java.lang.reflect.Field;
27 
28 public class FieldAnnotationsApp {
29     @MyAnnotation(name="myField1",  value="myValue1")
30     public String myField1 = null;
31 
32     @MyAnnotation(name="myField2",  value="myValue2")
33     public String myField2 = null;
34 
35     public static void main(String args[]) throws Exception {
36         for (int i=1; i<=2; i++) {
37             Field field = FieldAnnotationsApp.class.getField("myField" + i);
38             Annotation[] annotations = field.getDeclaredAnnotations();
39 
40             for (Annotation anno : annotations){
41                 if (anno instanceof MyAnnotation){
42                     MyAnnotation myAnno = (MyAnnotation) anno;
43                     String name = myAnno.name();
44                     String value = myAnno.value();
45 
46                     System.out.println("Field         : " + field.getName());
47                     System.out.println("  myAnno.name : " + name);
48                     System.out.println("  myAnno.value: " + value);
49 
50                     if (!(name.equals("myField" + i) && value.equals("myValue" + i))) {
51                         throw new Exception("Unexpected annotation values: " + i + " = " + value);
52                     }
53                 }
54             }
55         }
56         System.out.println("Field annotations are OK.");
57     }
58 }
