25 /*
26  * @test
27  * @summary test the ability to archive array classes and load them from the archive
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  * @build sun.hotspot.WhiteBox
31  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
32  * @run driver ArrayTest
33  */
34 
35 import java.util.List;
36 import java.util.ArrayList;
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class ArrayTest {
40 
41     static String arrayClasses[] = {
42         ArrayTestHelper.class.getName(),
43         "[Ljava/lang/Comparable;",
44         "[I",
45         "[[[Ljava/lang/Object;",
46         "[[B"
47     };
48 
49     public static void main(String[] args) throws Exception {
50         JarBuilder.build("arrayTestHelper", "ArrayTestHelper");
51 
52         String appJar = TestCommon.getTestJar("arrayTestHelper.jar");
53         JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
54         String whiteBoxJar = TestCommon.getTestJar("WhiteBox.jar");
55         String bootClassPath = "-Xbootclasspath/a:" + whiteBoxJar;
56 
57         // create an archive containing array classes
58         OutputAnalyzer output = TestCommon.dump(appJar, TestCommon.list(arrayClasses), bootClassPath);
59         // we currently don't support array classes during CDS dump
60         output.shouldContain("Preload Warning: Cannot find [Ljava/lang/Comparable;")
61               .shouldContain("Preload Warning: Cannot find [I")
62               .shouldContain("Preload Warning: Cannot find [[[Ljava/lang/Object;")
63               .shouldContain("Preload Warning: Cannot find [[B");
64 
65         List<String> argsList = new ArrayList<String>();
66         argsList.add("-XX:+UnlockDiagnosticVMOptions");
67         argsList.add("-XX:+WhiteBoxAPI");
68         argsList.add("-cp");
69         argsList.add(appJar);
70         argsList.add(bootClassPath);
71         argsList.add(ArrayTestHelper.class.getName());
72         // the following are input args to the ArrayTestHelper.
73         // skip checking array classes during run time
74         for (int i = 0; i < 1; i++) {
75             argsList.add(arrayClasses[i]);
76         }
77         String[] opts = new String[argsList.size()];
78         opts = argsList.toArray(opts);
79         TestCommon.run(opts).assertNormalExit();
80     }
81 }
