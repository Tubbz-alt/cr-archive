25 import java.lang.instrument.ClassFileTransformer;
26 import java.lang.instrument.Instrumentation;
27 import java.lang.instrument.IllegalClassFormatException;
28 import java.security.ProtectionDomain;
29 
30 public class AnonVmClassesDuringDumpTransformer implements ClassFileTransformer {
31     public byte[] transform(ClassLoader loader, String name, Class<?> classBeingRedefined,
32                             ProtectionDomain pd, byte[] buffer) throws IllegalClassFormatException {
33         return null;
34     }
35 
36     private static Instrumentation savedInstrumentation;
37 
38     public static void premain(String agentArguments, Instrumentation instrumentation) {
39         System.out.println("ClassFileTransformer.premain() is called");
40         instrumentation.addTransformer(new AnonVmClassesDuringDumpTransformer(), /*canRetransform=*/true);
41         savedInstrumentation = instrumentation;
42 
43         // This will create a Lambda, which will result in some Anonymous VM Classes
44         // being generated.
45         //
46         // Look for something like these in the STDOUT:
47         // ----------------
48         // ClassFileTransformer.premain() is called
49         // Dumping class files to DUMP_CLASS_FILES/...
50         // dump: DUMP_CLASS_FILES/java/lang/invoke/LambdaForm$MH000.class
51         // dump: DUMP_CLASS_FILES/java/lang/invoke/LambdaForm$MH001.class
52         // Invoked inside a Lambda
53         // ----------------
54         Runnable r = () -> {
55             System.out.println("Invoked inside a Lambda");
56         };
57         r.run();
58     }
59 
60     public static Instrumentation getInstrumentation() {
61         return savedInstrumentation;
62     }
63 
64     public static void agentmain(String args, Instrumentation inst) throws Exception {
65         premain(args, inst);
66     }
67 }
