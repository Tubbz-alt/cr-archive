25 /*
26  * @test
27  * @summary Write a lots of shared strings.
28  * @requires vm.cds.archived.java.heap
29  * @library /test/hotspot/jtreg/runtime/cds/appcds /test/lib
30  * @build HelloString
31  * @run driver/timeout=500 SharedStringsStress
32  */
33 import java.io.File;
34 import java.io.FileOutputStream;
35 import java.io.OutputStreamWriter;
36 import java.io.PrintWriter;
37 import jdk.test.lib.process.OutputAnalyzer;
38 import jdk.test.lib.process.ProcessTools;
39 
40 public class SharedStringsStress {
41     static String sharedArchiveConfigFile = System.getProperty("user.dir") + File.separator + "SharedStringsStress_gen.txt";
42 
43     public static void main(String[] args) throws Exception {
44         try (FileOutputStream fos = new FileOutputStream(sharedArchiveConfigFile)) {
45             PrintWriter out = new PrintWriter(new OutputStreamWriter(fos));
46             out.println("VERSION: 1.0");
47             out.println("@SECTION: String");
48             out.println("31: shared_test_string_unique_14325");
49             for (int i=0; i<200000; i++) {
50                 String s = "generated_string " + i;
51                 out.println(s.length() + ": " + s);
52             }
53             out.close();
54         }
55 
56         SharedStringsUtils.run(args, SharedStringsStress::test);
57     }
58 
59     public static void test(String[] args) throws Exception {
60         String vmOptionsPrefix[] = SharedStringsUtils.getChildVMOptionsPrefix();
61         String appJar = JarBuilder.build("SharedStringsStress", "HelloString");
62 
63         String test_cases[][] = {
64             // default heap size
65             {},
66 
67             // Test for handling of heap fragmentation. With sharedArchiveConfigFile, we will dump about
68             // 18MB of shared objects on 64 bit VM (smaller on 32-bit).
69             //
70             // During dump time, an extra copy of these objects are allocated,
71             // so we need about 36MB, plus a few MB for other system data. So 64MB total heap
72             // should be enough.
73             //
74             // The VM should executed a full GC to maximize contiguous free space and
75             // avoid fragmentation.
76             {"-Xmx64m"},
77         };
78 
79         for (String[] extra_opts: test_cases) {
80             vmOptionsPrefix = TestCommon.concat(vmOptionsPrefix, extra_opts);
81 
82             OutputAnalyzer dumpOutput = TestCommon.dump(appJar, TestCommon.list("HelloString"),
83                 TestCommon.concat(vmOptionsPrefix,
84                     "-XX:SharedArchiveConfigFile=" + sharedArchiveConfigFile,
85                     "-Xlog:gc+region+cds",
86                     "-Xlog:gc+region=trace"));
87             TestCommon.checkDump(dumpOutput);
88             OutputAnalyzer execOutput = TestCommon.exec(appJar,
89                 TestCommon.concat(vmOptionsPrefix, "HelloString"));
90             TestCommon.checkExec(execOutput);
91         }
92     }
93 }
