25 /*
26  * @test
27  * @summary Load app classes from CDS archive in parallel threads. Similar to ParallelLoad.java, but each class in its own JAR
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/ParallelLoad.java
31  * @compile test-classes/ParallelClasses.java
32  * @run driver ParallelLoad2
33  */
34 
35 import java.io.File;
36 
37 public class ParallelLoad2 {
38   public static int MAX_CLASSES = 40;
39   public static void main(String[] args) throws Exception {
40     JarBuilder.build("parallel_load2", "ParallelLoad", "ParallelLoadThread", "ParallelLoadWatchdog");
41     for (int i=0; i<MAX_CLASSES; i++) {
42       JarBuilder.build("parallel_load2_" + i, "ParallelClass" + i);
43     }
44 
45     String cp = TestCommon.getTestJar("parallel_load2.jar");
46     for (int i=0; i<MAX_CLASSES; i++) {
47       cp += File.pathSeparator + TestCommon.getTestJar("parallel_load2_" + i + ".jar");
48     }
49 
50     String[] class_list = new String[MAX_CLASSES + 2];
51     for (int i=0; i<MAX_CLASSES; i++) {
52       class_list[i] = "ParallelClass" + i;
53     }
54     class_list[class_list.length - 1] = "ParallelLoad";
55     class_list[class_list.length - 2] = "ParallelLoadThread";
56 
57     TestCommon.test(cp, class_list,
58                           "ParallelLoad");
59   }
60 }
