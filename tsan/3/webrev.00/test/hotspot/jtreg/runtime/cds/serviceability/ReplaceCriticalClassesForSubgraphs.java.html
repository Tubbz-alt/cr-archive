25 /*
26  * @test
27  * @summary Tests how CDS works when critical library classes are replaced with JVMTI ClassFileLoadHook
28  * @library /test/lib
29  * @requires vm.cds.archived.java.heap
30  * @build sun.hotspot.WhiteBox
31  * @run driver ClassFileInstaller -jar whitebox.jar sun.hotspot.WhiteBox
32  * @run main/othervm/native ReplaceCriticalClassesForSubgraphs
33  */
34 
35 public class ReplaceCriticalClassesForSubgraphs extends ReplaceCriticalClasses {
36     public static void main(String args[]) throws Throwable {
37         ReplaceCriticalClassesForSubgraphs rcc = new ReplaceCriticalClassesForSubgraphs();
38         rcc.process(args);
39     }
40 
41     public String[] getTests() {
42         String tests[] = {
43             // Try to replace classes that are used by the archived subgraph graphs.
44             "-early -notshared -subgraph java/lang/module/ResolvedModule jdk.internal.module.ArchivedModuleGraph",
45             "-early -notshared -subgraph java/lang/Long java.lang.Long$LongCache",
46             "-subgraph java/lang/Long java.lang.Long$LongCache",
47         };
48         return tests;
49     }
50 }
