25 /*
26  * @test
27  * @summary Test archived mirror with reference fields
28  * @requires vm.cds.archived.java.heap
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  * @build sun.hotspot.WhiteBox
31  * @compile MirrorWithReferenceFieldsApp.java
32  * @run driver ClassFileInstaller -jar app.jar MirrorWithReferenceFieldsApp
33  * @run driver ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
34  * @run driver MirrorWithReferenceFieldsTest
35  */
36 
37 import jdk.test.lib.process.OutputAnalyzer;
38 import sun.hotspot.WhiteBox;
39 
40 public class MirrorWithReferenceFieldsTest {
41     public static void main(String[] args) throws Exception {
42         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
43         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
44         String appJar = ClassFileInstaller.getJarPath("app.jar");
45 
46         String classlist[] = new String[] {
47             "MirrorWithReferenceFieldsApp",
48         };
49 
50         TestCommon.testDump(appJar, classlist, use_whitebox_jar);
51         OutputAnalyzer output = TestCommon.exec(appJar, use_whitebox_jar,
52                                                 "-XX:+UnlockDiagnosticVMOptions",
53                                                 "-XX:+WhiteBoxAPI",
54                                                 "-XX:+VerifyAfterGC",
55                                                 "MirrorWithReferenceFieldsApp");
56         try {
57             TestCommon.checkExec(output, "Done");
58         } catch (Exception e) {
59             output.shouldContain("Archived open_archive_heap objects are not mapped");
60         }
61     }
62 }
