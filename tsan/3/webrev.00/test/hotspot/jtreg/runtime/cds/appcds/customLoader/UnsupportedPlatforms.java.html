25 /*
26  * @test
27  * @summary Ensure that support for AppCDS custom class loaders are not enabled on unsupported platforms.
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  * @compile test-classes/SimpleHello.java
31  * @run driver UnsupportedPlatforms
32  */
33 
34 import jdk.test.lib.Platform;
35 import jdk.test.lib.process.OutputAnalyzer;
36 
37 public class UnsupportedPlatforms {
38     public static String PLATFORM_NOT_SUPPORTED_WARNING =
39         "AppCDS custom class loaders not supported on this platform";
40 
41     public static void main(String[] args) throws Exception {
42         String appJar = JarBuilder.build("UnsupportedPlatforms", "SimpleHello");
43 
44         // Dump the archive
45         String classlist[] = new String[] {
46             "SimpleHello",
47             "java/lang/Object id: 1",
48             "CustomLoadee id: 2 super: 1 source: " + appJar
49         };
50 
51         OutputAnalyzer out = TestCommon.dump(appJar, classlist);
52 
53         if (Platform.areCustomLoadersSupportedForCDS()) {
54             out.shouldNotContain(PLATFORM_NOT_SUPPORTED_WARNING);
55             out.shouldHaveExitValue(0);
56         } else {
57             out.shouldContain(PLATFORM_NOT_SUPPORTED_WARNING);
58             out.shouldHaveExitValue(1);
59         }
60     }
61 }
