25 /*
26  * @test
27  * @summary Tests the format checking of hotspot/src/closed/share/vm/classfile/classListParser.cpp.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
32  * @compile ../test-classes/Hello.java test-classes/CustomLoadee.java test-classes/CustomLoadee2.java
33  *          test-classes/CustomInterface2_ia.java test-classes/CustomInterface2_ib.java
34  * @run driver ClassListFormatB
35  */
36 
37 public class ClassListFormatB extends ClassListFormatBase {
38     static {
39         // Uncomment the following line to run only one of the test cases
40         // ClassListFormatBase.RUN_ONLY_TEST = "TESTCASE B1";
41     }
42 
43     public static void main(String[] args) throws Throwable {
44         String appJar = JarBuilder.getOrCreateHelloJar();
45         String customJarPath = JarBuilder.build("ClassListFormatB", "CustomLoadee",
46                                             "CustomLoadee2", "CustomInterface2_ia", "CustomInterface2_ib");
47         //----------------------------------------------------------------------
48         // TESTGROUP B if source IS specified
49         //----------------------------------------------------------------------
50         dumpShouldFail(
51             "TESTCASE B1: if source: is specified, must specify super:",
52             appJar, classlist(
53                 "Hello",
54                 "java/lang/Object id: 1",
55                 "CustomLoadee id: 2 source: " + customJarPath
56             ),
57             "If source location is specified, super class must be also specified");
58 
59         dumpShouldFail(
60             "TESTCASE B2: if source: is specified, must specify id:",
61             appJar, classlist(
62                 "Hello",
63                 "java/lang/Object id: 1",
64                 "CustomLoadee super: 1 source: " + customJarPath
65             ),
66             "If source location is specified, id must be also specified");
67     }
68 }
