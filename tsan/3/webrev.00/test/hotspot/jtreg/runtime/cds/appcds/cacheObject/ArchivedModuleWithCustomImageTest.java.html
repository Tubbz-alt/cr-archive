 24 /*
 25  * @test
 26  * @summary Test archived module graph with custom runtime image
 27  * @requires vm.cds.archived.java.heap
 28  * @library /test/jdk/lib/testlibrary /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 29  * @build sun.hotspot.WhiteBox
 30  * @compile CheckArchivedModuleApp.java
 31  * @run driver ClassFileInstaller -jar app.jar CheckArchivedModuleApp
 32  * @run driver ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
 33  * @run driver ArchivedModuleWithCustomImageTest
 34  */
 35 
 36 import java.io.File;
 37 import java.nio.file.Files;
 38 import java.nio.file.Path;
 39 import java.nio.file.Paths;
 40 
 41 import jdk.test.lib.compiler.CompilerUtils;
 42 import jdk.test.lib.process.OutputAnalyzer;
 43 import jdk.test.lib.process.ProcessTools;
 44 
 45 public class ArchivedModuleWithCustomImageTest {
 46     private static final String JAVA_HOME = System.getProperty("java.home");
 47     private static final String TEST_MODULE = "test";
 48     private static final Path jdkHome = Paths.get(System.getProperty("test.jdk"));
 49     private static final Path jdkMods = jdkHome.resolve("jmods");
 50     private static final Path testSrc = Paths.get(System.getProperty("test.src"));
 51     private static final Path src = testSrc.resolve("src").resolve(TEST_MODULE);
 52     private static final Path classes = Paths.get("classes");
 53     private static final Path jmods = Paths.get("jmods");
 54 
 55     public static void main(String[] args) throws Throwable {
 56         if (Files.notExists(jdkMods)) {
 57             System.out.println("No jmods/ in test JDK, not supported.");
 58             return;
 59         }
 60 
 61         // compile test module class
 62         if (!CompilerUtils.compile(src, classes)) {
 63             throw new RuntimeException("Compilation failure.");
 64         }
 65 
 66         // create custom runtime image named 'myimage'
 67         Files.createDirectories(jmods);
 68         Path image = Paths.get("myimage");
 69         runJmod(classes.toString(), TEST_MODULE);
 70         runJlink(image, TEST_MODULE);
 71 
 72         // test using 'myimage'
 73         testArchivedModuleUsingImage(image);
 74 
 75         Files.delete(jmods.resolve(TEST_MODULE + ".jmod"));
 76     }
 77 
 78     private static void runJlink(Path image, String modName) throws Throwable {
 79         Path jlink = Paths.get(JAVA_HOME, "bin", "jlink");
 80         OutputAnalyzer output = ProcessTools.executeProcess(jlink.toString(),
 81                         "--output", image.toString(),
 82                         "--add-modules", modName,
 83                         "--module-path", jdkMods + File.pathSeparator + jmods);
 84         output.shouldHaveExitValue(0);
 85     }
 86 
 87     private static void runJmod(String cp, String modName) throws Throwable {
 88         Path jmod = Paths.get(JAVA_HOME, "bin", "jmod");
 89         OutputAnalyzer output = ProcessTools.executeProcess(jmod.toString(),
 90                        "create",
 91                        "--class-path", cp,
 92                        "--module-version", "1.0",
 93                        "--main-class", "jdk.test.Test",
 94                        jmods.resolve(modName + ".jmod").toString());
 95         output.shouldHaveExitValue(0);
 96     }
 97 
 98     private static void testArchivedModuleUsingImage(Path image)
 99                             throws Throwable {
100         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
101         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
102         String appJar = ClassFileInstaller.getJarPath("app.jar");
103         Path customJava = Paths.get(image.toString(), "bin", "java");
104 
105         // -Xshare:dump with custom runtime image
106         String[] dumpCmd = {
107             customJava.toString(),
108             "-XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa",
109             "-Xshare:dump", "-Xlog:cds"};
110         printCommand(dumpCmd);
111         ProcessBuilder pbDump = new ProcessBuilder();
112         pbDump.command(dumpCmd);
113         OutputAnalyzer output = TestCommon.executeAndLog(
114             pbDump, "custom.runtime.image.dump");
115         TestCommon.checkDump(output);
116 
117         // Test case 1):
118         // test archived module graph objects are used with custome runtime image
119         System.out.println("------------------- Test case 1 -------------------");
120         String[] runCmd = {customJava.toString(),
121                            use_whitebox_jar,
122                            "-XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa",
123                            "-cp",
124                            appJar,
125                            "-Xshare:on",
126                            "-XX:+UnlockDiagnosticVMOptions",
127                            "-XX:+WhiteBoxAPI",
128                            "CheckArchivedModuleApp",
129                            "yes",
130                            "yes"};
131         printCommand(runCmd);
132         ProcessBuilder pbRun = new ProcessBuilder();
133         pbRun.command(runCmd);
134         output = TestCommon.executeAndLog(pbRun, "custom.runtime.image.run");
135         output.shouldHaveExitValue(0);
136 
137 
138         // Test case 2):
139         // verify --show-module-resolution output
140         System.out.println("------------------- Test case 2 -------------------");
141 
142         // myimage/bin/java -Xshare:off --show-module-resolution -version
143         String[] showModuleCmd1 = {customJava.toString(),
144                                    "-Xshare:off",
145                                    "--show-module-resolution",
146                                    "-version"};
147         printCommand(showModuleCmd1);
148         pbRun = new ProcessBuilder();
149         pbRun.command(showModuleCmd1);
150         output = TestCommon.executeAndLog(
151             pbRun, "custom.runtime.image.showModuleResolution.nocds");
152         output.shouldHaveExitValue(0);
153         String moduleResolutionOut1 = output.getStdout();
154 
155         // myimage/bin/java -Xshare:on --show-module-resolution -version
156         //    -XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa
157         String[] showModuleCmd2 = {
158             customJava.toString(),
159             "-XX:SharedArchiveFile=./ArchivedModuleWithCustomImageTest.jsa",
160             "-Xshare:on",
161             "--show-module-resolution",
162             "-version"};
163         printCommand(showModuleCmd2);
164         pbRun = new ProcessBuilder();
165         pbRun.command(showModuleCmd2);
166         output = TestCommon.executeAndLog(
167             pbRun, "custom.runtime.image.showModuleResolution.cds");
168         if (output.getStderr().contains("sharing")) {
169             String moduleResolutionOut2 = output.getStdout();
170             TestCommon.checkOutputStrings(
171                 moduleResolutionOut1, moduleResolutionOut2, "\n");
172         }
173     }
174 
175     private static void printCommand(String opts[]) {
176         StringBuilder cmdLine = new StringBuilder();
177         for (String cmd : opts)
178             cmdLine.append(cmd).append(' ');
179         System.out.println("Command line: [" + cmdLine.toString() + "]");
180     }
181 }
