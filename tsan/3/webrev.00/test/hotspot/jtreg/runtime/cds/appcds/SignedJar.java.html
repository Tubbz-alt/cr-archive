25 /*
26  * @test
27  * @summary AppCDS handling of signed JAR.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/Hello.java
31  * @run driver SignedJar
32  */
33 
34 import jdk.test.lib.process.OutputAnalyzer;
35 import java.io.File;
36 
37 public class SignedJar {
38     public static void main(String[] args) throws Exception {
39         String unsignedJar = JarBuilder.getOrCreateHelloJar();
40         JarBuilder.signJar();
41 
42         // Test class exists in signed JAR
43         String signedJar = TestCommon.getTestJar("signed_hello.jar");
44         OutputAnalyzer output;
45         output = TestCommon.dump(signedJar, TestCommon.list("Hello"));
46         TestCommon.checkDump(output, "Skipping Hello: Signed JAR");
47 
48         // At runtime, the Hello class should be loaded from the jar file
49         // instead of from the shared archive since a class from a signed
50         // jar shouldn't be dumped into the archive.
51         output = TestCommon.exec(signedJar, "-verbose:class", "Hello");
52         String expectedOutput = ".class,load. Hello source: file:.*signed_hello.jar";
53 
54         try {
55             output.shouldMatch(expectedOutput);
56         } catch (Exception e) {
57             TestCommon.checkCommonExecExceptions(output, e);
58         }
59 
60         // Test class exists in both signed JAR and unsigned JAR
61         String jars = signedJar + System.getProperty("path.separator") + unsignedJar;
62         output = TestCommon.dump(jars, TestCommon.list("Hello"));
63         TestCommon.checkDump(output, "Skipping Hello: Signed JAR");
64     }
65 }
