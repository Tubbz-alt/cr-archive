 25 /*
 26  * @test
 27  * @summary Test dumping with limited metaspace with loading of JVMCI related classes.
 28  *          VM should not crash but CDS dump will abort upon failure in allocating metaspace.
 29  * @requires vm.cds & vm.graal.enabled & vm.compMode == "Xmixed"
 30  * @library /test/lib
 31  * @build UseAppCDS_Test
 32  * @run driver ClassFileInstaller -jar test.jar UseAppCDS_Test
 33  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+EnableJVMCI
 34  *      -XX:+TieredCompilation -XX:+UseJVMCICompiler -Djvmci.Compiler=graal
 35  *      GraalWithLimitedMetaspace
 36  */
 37 
 38 import jdk.test.lib.cds.CDSTestUtils;
 39 import jdk.test.lib.process.OutputAnalyzer;
 40 import jdk.test.lib.process.ProcessTools;
 41 
 42 import java.util.ArrayList;
 43 import java.util.List;
 44 import java.io.*;
 45 
 46 public class GraalWithLimitedMetaspace {
 47 
 48     // Class UseAppCDS_Test is loaded by the App loader
 49 
 50     static final String TEST_OUT = "UseAppCDS_Test.main--executed";
 51 
 52     private static final String TESTJAR = "./test.jar";
 53     private static final String TESTNAME = "UseAppCDS_Test";
 54     private static final String TESTCLASS = TESTNAME + ".class";
 55 
 56     private static final String CLASSLIST_FILE = "./GraalWithLimitedMetaspace.classlist";
 57     private static final String ARCHIVE_FILE = "./GraalWithLimitedMetaspace.jsa";
 58     private static final String BOOTCLASS = "java.lang.Class";
 59 
 60     public static void main(String[] args) throws Exception {
 61 
 62         // dump loaded classes into a classlist file
 63         dumpLoadedClasses(new String[] { BOOTCLASS, TESTNAME });
 64 
 65 
 66         // create an archive using the classlist
 67         dumpArchive();
 68 
 69     }
 70 
 71     public static List<String> toClassNames(String filename) throws IOException {
 72         ArrayList<String> classes = new ArrayList<>();
 73         try (BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(filename)))) {
 74             for (; ; ) {
 75                 String line = br.readLine();
 76                 if (line == null) {
 77                     break;
 78                 }
 79                 classes.add(line.replaceAll("/", "."));
 80             }
 81         }
 82         return classes;
 83     }
 84 
 85     static void dumpLoadedClasses(String[] expectedClasses) throws Exception {
 86         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(true,
 87             "-XX:DumpLoadedClassList=" + CLASSLIST_FILE,
 88             // trigger JVMCI runtime init so that JVMCI classes will be
 89             // included in the classlist
 90             "-XX:+UnlockExperimentalVMOptions",
 91             "-XX:+EnableJVMCI",
 92             "-XX:+EagerJVMCI",
 93             "-cp",
 94             TESTJAR,
 95             TESTNAME,
 96             TEST_OUT);
 97 
 98         OutputAnalyzer output = TestCommon.executeAndLog(pb, "dump-loaded-classes")
 99             .shouldHaveExitValue(0)
100             .shouldContain(TEST_OUT);
101 
102         List<String> dumpedClasses = toClassNames(CLASSLIST_FILE);
103 
104         for (String clazz : expectedClasses) {
105             if (!dumpedClasses.contains(clazz)) {
106                 throw new RuntimeException(clazz + " missing in " +
107                                            CLASSLIST_FILE);
108             }
109         }
110     }
111 
112     static void dumpArchive() throws Exception {
113         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(true,
114             "-cp",
115             TESTJAR,
116             "-XX:SharedClassListFile=" + CLASSLIST_FILE,
117             "-XX:SharedArchiveFile=" + ARCHIVE_FILE,
118             "-Xlog:cds",
119             "-Xshare:dump",
120             "-XX:MetaspaceSize=12M",
121             "-XX:MaxMetaspaceSize=12M");
122 
123         OutputAnalyzer output = TestCommon.executeAndLog(pb, "dump-archive");
124         int exitValue = output.getExitValue();
125         if (exitValue == 1) {
126             output.shouldContain("Failed allocating metaspace object type");
127         } else if (exitValue == 0) {
128             output.shouldContain("Loading classes to share");
129         } else {
130             throw new RuntimeException("Unexpected exit value " + exitValue);
131         }
132     }
133 }
