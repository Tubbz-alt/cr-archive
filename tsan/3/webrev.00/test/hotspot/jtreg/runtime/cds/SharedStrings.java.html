24 /**
25  * @test
26  * @summary Check to make sure that shared strings in the bootstrap CDS archive
27  *          are actually shared
28  * @requires vm.cds.archived.java.heap
29  * @library /test/lib
30  * @build SharedStringsWb sun.hotspot.WhiteBox
31  * @run driver ClassFileInstaller -jar whitebox.jar sun.hotspot.WhiteBox
32  * @run driver SharedStrings
33  */
34 
35 import jdk.test.lib.cds.CDSTestUtils;
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class SharedStrings {
40     public static void main(String[] args) throws Exception {
41         // Note: This is a basic sanity test for Shared Strings feature.
42         // This also serves as a reference on how to use this feature,
43         // hence the command lines are spelled out instead of using the
44         // test utils methods.
45         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
46             "-XX:+UnlockDiagnosticVMOptions",
47             "-XX:SharedArchiveFile=./SharedStrings.jsa",
48             "-Xlog:cds,cds+hashtables",
49             // Needed for bootclasspath match, for CDS to work with WhiteBox API
50             "-Xbootclasspath/a:" + ClassFileInstaller.getJarPath("whitebox.jar"),
51             "-Xshare:dump");
52 
53         OutputAnalyzer out = CDSTestUtils.executeAndLog(pb, "dump");
54         CDSTestUtils.checkDump(out, "Shared string table stats");
55 
56 
57         pb = ProcessTools.createJavaProcessBuilder(
58                 "-XX:+UnlockDiagnosticVMOptions",
59                 "-XX:SharedArchiveFile=./SharedStrings.jsa",
60                 // needed for access to white box test API
61                 "-Xbootclasspath/a:" + ClassFileInstaller.getJarPath("whitebox.jar"),
62                 "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
63                 "-Xshare:on", "-showversion", "SharedStringsWb");
64 
65         out = CDSTestUtils.executeAndLog(pb, "exec");
66         CDSTestUtils.checkExec(out);
67     }
68 }
