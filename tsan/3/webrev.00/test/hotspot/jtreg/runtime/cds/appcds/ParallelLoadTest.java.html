25 /*
26  * @test
27  * @summary Load app classes from CDS archive in parallel threads
28  * @library /test/lib
29  * @requires vm.cds
30  * @compile test-classes/ParallelLoad.java
31  * @compile test-classes/ParallelClasses.java
32  * @run driver ParallelLoadTest
33  */
34 
35 import java.io.File;
36 
37 public class ParallelLoadTest {
38     public static final int MAX_CLASSES = 40;
39 
40     /* For easy stress testing, do this:
41 
42        i=0; while jtreg -DParallelLoadTest.app.loops=100 -DParallelLoadTest.boot.loops=100 \
43            ParallelLoadTest.java; do i=$(expr $i + 1); echo =====$i; done
44 
45      */
46 
47     private static final int APP_LOOPS  = Integer.parseInt(System.getProperty("ParallelLoadTest.app.loops", "1"));
48     private static final int BOOT_LOOPS = Integer.parseInt(System.getProperty("ParallelLoadTest.boot.loops", "1"));
49 
50     public static void main(String[] args) throws Exception {
51         JarBuilder.build("parallel_load_app", "ParallelLoad", "ParallelLoadThread", "ParallelLoadWatchdog");
52         JarBuilder.build("parallel_load_classes", getClassList());
53         String appJar     = TestCommon.getTestJar("parallel_load_app.jar");
54         String classesJar = TestCommon.getTestJar("parallel_load_classes.jar");
55 
56         // (1) Load the classes from app class loader
57         String CP = appJar + File.pathSeparator + classesJar;
58         TestCommon.testDump(CP, getClassList());
59         for (int i = 0; i < APP_LOOPS; i++) {
60             TestCommon.run("-cp", CP,  "ParallelLoad").assertNormalExit();
61         }
62 
63         // (2) Load the classes from boot class loader
64         String bootcp = "-Xbootclasspath/a:" + classesJar;
65         TestCommon.testDump(appJar, getClassList(), bootcp);
66         for (int i = 0; i < BOOT_LOOPS; i++) {
67             TestCommon.run(bootcp, "-cp", appJar,
68                            // "-Xlog:class+load=debug",
69                            "ParallelLoad").assertNormalExit();
70         }
71     }
72 
73     private static String[] getClassList() {
74         String[] classList = new String[MAX_CLASSES];
75 
76         int i;
77         for (i = 0; i < MAX_CLASSES; i++) {
78             classList[i] = "ParallelClass" + i;
79         }
80 
81         return classList;
82     }
83 }
