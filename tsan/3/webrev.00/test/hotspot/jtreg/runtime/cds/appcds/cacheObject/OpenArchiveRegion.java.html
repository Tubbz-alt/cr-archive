25 /*
26  * @test
27  * @summary Test open archive heap regions
28  * @requires vm.cds.archived.java.heap
29  * @comment This test explicitly chooses the type of GC to be used by sub-processes. It may conflict with the GC type set
30  * via the -vmoptions command line option of JTREG. vm.gc==null will help the test case to discard the explicitly passed
31  * vm options.
32  * @requires (vm.gc=="null")
33  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
34  * @compile ../test-classes/Hello.java
35  * @run driver OpenArchiveRegion
36  */
37 
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class OpenArchiveRegion {
41     public static void main(String[] args) throws Exception {
42         JarBuilder.getOrCreateHelloJar();
43         String appJar = TestCommon.getTestJar("hello.jar");
44         String appClasses[] = TestCommon.list("Hello");
45 
46         // Dump with open archive heap region, requires G1 GC
47         OutputAnalyzer output = TestCommon.dump(appJar, appClasses, "-Xlog:cds=debug");
48         TestCommon.checkDump(output, "oa0 space:");
49         output.shouldNotContain("oa0 space:         0 [");
50         output = TestCommon.exec(appJar, "Hello");
51         TestCommon.checkExec(output, "Hello World");
52         output = TestCommon.exec(appJar, "-XX:+UseSerialGC", "Hello");
53         TestCommon.checkExec(output, "Hello World");
54 
55         // Dump with open archive heap region disabled when G1 GC is not in use
56         output = TestCommon.dump(appJar, appClasses, "-XX:+UseParallelGC");
57         TestCommon.checkDump(output);
58         output.shouldNotContain("oa0 space:");
59         output = TestCommon.exec(appJar, "Hello");
60         TestCommon.checkExec(output, "Hello World");
61     }
62 }
