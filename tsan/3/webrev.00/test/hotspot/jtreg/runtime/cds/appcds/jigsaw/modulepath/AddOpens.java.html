 25 /**
 26  * @test
 27  * @requires vm.cds
 28  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 29  * @run driver AddOpens
 30  * @summary sanity test the --add-opens option
 31  */
 32 
 33 import java.io.File;
 34 import java.nio.file.Files;
 35 import java.nio.file.Path;
 36 import java.nio.file.Paths;
 37 
 38 import jdk.test.lib.process.OutputAnalyzer;
 39 
 40 public class AddOpens {
 41 
 42     private static final Path USER_DIR = Paths.get(System.getProperty("user.dir"));
 43 
 44     private static final String TEST_SRC = System.getProperty("test.src");
 45 
 46     private static final Path SRC_DIR = Paths.get(TEST_SRC, "src");
 47     private static final Path MODS_DIR = Paths.get("mods");
 48 
 49     // the module name of the test module
 50     private static final String TEST_MODULE1 = "com.simple";
 51 
 52     // the module main class
 53     private static final String MAIN_CLASS = "com.simple.Main";
 54 
 55     private static Path moduleDir = null;
 56     private static Path moduleDir2 = null;
 57     private static Path destJar = null;
 58 
 59     public static void buildTestModule() throws Exception {
 60 
 61         // javac -d mods/$TESTMODULE --module-path MOD_DIR src/$TESTMODULE/**
 62         JarBuilder.compileModule(SRC_DIR.resolve(TEST_MODULE1),
 63                                  MODS_DIR.resolve(TEST_MODULE1),
 64                                  MODS_DIR.toString());
 65 
 66         moduleDir = Files.createTempDirectory(USER_DIR, "mlib");
 67         moduleDir2 = Files.createTempDirectory(USER_DIR, "mlib2");
 68 
 69         Path srcJar = moduleDir.resolve(TEST_MODULE1 + ".jar");
 70         destJar = moduleDir2.resolve(TEST_MODULE1 + ".jar");
 71         String classes = MODS_DIR.resolve(TEST_MODULE1).toString();
 72         JarBuilder.createModularJar(srcJar.toString(), classes, MAIN_CLASS);
 73         Files.copy(srcJar, destJar);
 74 
 75     }
 76 
 77     public static void main(String... args) throws Exception {
 78         // compile the modules and create the modular jar files
 79         buildTestModule();
 80         String appClasses[] = {MAIN_CLASS};
 81         // create an archive with both -cp and --module-path in the command line.
 82         // Only the class in the modular jar in the --module-path will be archived;
 83         // the class in the modular jar in the -cp won't be archived.
 84         OutputAnalyzer output = TestCommon.createArchive(
 85                                         destJar.toString(), appClasses,
 86                                         "--module-path", moduleDir.toString(),
 87                                         "-m", TEST_MODULE1);
 88         TestCommon.checkDump(output);
 89 
 90         // run with the archive using the same command line as in dump time
 91         // plus the "--add-opens java.base/java.lang=com.simple" option.
 92         // The main class should be loaded from the archive.
 93         // The setaccessible(true) on the ClassLoader.defineClass method should
 94         // be successful.
 95         TestCommon.run( "-Xlog:class+load=trace",
 96                         "-cp", destJar.toString(),
 97                         "--add-opens", "java.base/java.lang=" + TEST_MODULE1,
 98                         "--module-path", moduleDir.toString(),
 99                         "-m", TEST_MODULE1, "with_add_opens")
100             .assertNormalExit(
101                 "[class,load] com.simple.Main source: shared objects file",
102                 "method.setAccessible succeeded!");
103 
104     }
105 }
