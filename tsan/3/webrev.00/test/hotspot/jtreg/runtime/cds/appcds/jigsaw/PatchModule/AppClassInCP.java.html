 25 /*
 26  * @test
 27  * @requires vm.cds
 28  * @summary a test to demonstrate that an application class in the -cp
 29  *          will be archived although --patch-module is specified. The class in
 30  *          the -cp has no dependencies on the class in the --patch-module.
 31  * @library ../..
 32  * @library /test/hotspot/jtreg/testlibrary
 33  * @library /test/lib
 34  * @build PatchMain
 35  * @run driver AppClassInCP
 36  */
 37 
 38 import java.io.File;
 39 import jdk.test.lib.compiler.InMemoryJavaCompiler;
 40 import jdk.test.lib.process.OutputAnalyzer;
 41 
 42 public class AppClassInCP {
 43     private static String moduleJar;
 44     private static String appJar;
 45 
 46     public static void main(String args[]) throws Throwable {
 47 
 48         // Create a class file in the module java.naming. This class file
 49         // will be put in the javanaming.jar file.
 50         String source = "package javax.naming.spi; "                +
 51                         "public class NamingManager { "             +
 52                         "    static { "                             +
 53                         "        System.out.println(\"I pass!\"); " +
 54                         "    } "                                    +
 55                         "}";
 56 
 57         String classDir = System.getProperty("test.classes");
 58 
 59         ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
 60              InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
 61              classDir);
 62 
 63         // Build the jar file that will be used for the module "java.naming".
 64         JarBuilder.build("javanaming", "javax/naming/spi/NamingManager");
 65         moduleJar = TestCommon.getTestJar("javanaming.jar");
 66 
 67         String source2 = "package mypackage; "                +
 68                         "public class Hello { "             +
 69                         "    static { "                             +
 70                         "        System.out.println(\"Hello!\"); " +
 71                         "    } "                                    +
 72                         "}";
 73         ClassFileInstaller.writeClassToDisk("mypackage/Hello",
 74              InMemoryJavaCompiler.compile("mypackage.Hello", source2),
 75              classDir);
 76 
 77         JarBuilder.build("hello", "mypackage/Hello");
 78         appJar = TestCommon.getTestJar("hello.jar");
 79 
 80         System.out.println("Test dumping with --patch-module");
 81         OutputAnalyzer output =
 82             TestCommon.dump(appJar,
 83                 TestCommon.list("javax/naming/spi/NamingManager", "mypackage/Hello"),
 84                 "--patch-module=java.naming=" + moduleJar,
 85                 "-Xlog:class+load",
 86                 "PatchMain", "javax.naming.spi.NamingManager", "mypackage.Hello");
 87         output.shouldHaveExitValue(1)
 88               .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module");
 89 
 90         String classPath = appJar + File.pathSeparator + classDir;
 91         System.out.println("classPath: " + classPath);
 92         TestCommon.run(
 93             "-XX:+UnlockDiagnosticVMOptions",
 94             "-cp", classPath,
 95             "--patch-module=java.naming=" + moduleJar,
 96             "-Xlog:class+load",
 97             "PatchMain", "javax.naming.spi.NamingManager", "mypackage.Hello")
 98             .assertSilentlyDisabledCDS(0, "I pass!", "Hello!");
 99     }
100 }
