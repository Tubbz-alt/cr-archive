25 /*
26  * @test
27  * @summary a simple test for loading a class using the platform class loader
28  *          (which used to be called the "extension loader) in AppCDS
29  * @requires vm.cds
30  * @library /test/lib
31  * @compile test-classes/HelloExt.java
32  * @build sun.hotspot.WhiteBox
33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
34  * @run driver HelloExtTest
35  */
36 
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class HelloExtTest {
40 
41   public static void main(String[] args) throws Exception {
42     JarBuilder.build("helloExt", "HelloExt");
43 
44     String appJar = TestCommon.getTestJar("helloExt.jar");
45     JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
46     String whiteBoxJar = TestCommon.getTestJar("WhiteBox.jar");
47     String bootClassPath = "-Xbootclasspath/a:" + whiteBoxJar;
48 
49     TestCommon.dump(appJar,
50         TestCommon.list("javax/annotation/processing/FilerException", "[Ljava/lang/Comparable;"),
51         bootClassPath);
52 
53     String prefix = ".class.load. ";
54     String class_pattern = ".*LambdaForm[$]MH[/][0123456789].*";
55     String suffix = ".*source: shared objects file.*";
56     String pattern = prefix + class_pattern + suffix;
57 
58     TestCommon.run("-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
59             "-cp", appJar, bootClassPath, "-Xlog:class+load", "HelloExt")
60         .assertNormalExit(output -> output.shouldNotMatch(pattern));
61 
62 
63     TestCommon.run("-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
64             "-cp", appJar, bootClassPath, "-Xlog:class+load",
65             "-XX:+PrintSharedArchiveAndExit", "-XX:+PrintSharedDictionary",
66             "HelloExt")
67         .assertNormalExit(output ->  output.shouldNotMatch(class_pattern));
68   }
69 }
