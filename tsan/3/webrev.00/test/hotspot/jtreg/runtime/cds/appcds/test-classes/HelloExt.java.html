25 import sun.hotspot.WhiteBox;
26 
27 public class HelloExt {
28    public static void main(String[] args) throws Throwable {
29 
30        String className = "javax.annotation.processing.FilerException";
31        Class cls = Class.forName(className);
32 
33        ClassLoader loader = cls.getClassLoader();
34        if (loader != ClassLoader.getPlatformClassLoader()) {
35            throw new java.lang.RuntimeException(className + " should be load by PlatformClassLoader but it is loaded by " + loader);
36        }
37 
38        WhiteBox wb = WhiteBox.getWhiteBox();
39        if (wb.isSharedClass(cls)) {
40            System.out.println("As expected, " + className + " is in shared space.");
41        } else {
42            throw new java.lang.RuntimeException(className + " is not in shared space.");
43        }
44 
45        className = "[Ljava.lang.Comparable;";
46        cls = Class.forName(className);
47        loader = cls.getClassLoader();
48        if (loader != null) {
49            throw new java.lang.RuntimeException(className + " should be load by the NULL class loader but it is loaded by " + loader);
50        }
51 
52        if (wb.isSharedClass(cls)) {
53            System.out.println("As expected, " + className + " is in shared space.");
54        } else {
55            throw new java.lang.RuntimeException(className + " is not in shared space.");
56        }
57    }
58 }
