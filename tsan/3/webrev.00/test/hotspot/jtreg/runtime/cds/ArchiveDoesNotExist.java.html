24 /**
25  * @test ArchiveDoesNotExist
26  * @summary Test how VM handles "file does not exist" situation while
27  *          attempting to use CDS archive. JVM should exit gracefully
28  *          when sharing mode is ON, and continue w/o sharing if sharing
29  *          mode is AUTO.
30  * @requires vm.cds
31  * @library /test/lib
32  * @run driver ArchiveDoesNotExist
33  */
34 
35 import jdk.test.lib.cds.CDSOptions;
36 import jdk.test.lib.cds.CDSTestUtils;
37 import jdk.test.lib.process.OutputAnalyzer;
38 import java.io.File;
39 
40 public class ArchiveDoesNotExist {
41     public static void main(String[] args) throws Exception {
42         String fileName = "ArchiveDoesNotExist.jsa";
43 
44         File cdsFile = new File(fileName);
45         if (cdsFile.exists())
46             throw new RuntimeException("Test error: cds file already exists");
47 
48         CDSOptions opts = (new CDSOptions()).setArchiveName(fileName);
49 
50         // -Xshare=on
51         OutputAnalyzer out = CDSTestUtils.runWithArchive(opts);
52         CDSTestUtils.checkMappingFailure(out);
53         out.shouldContain("Specified shared archive not found")
54             .shouldHaveExitValue(1);
55 
56         // -Xshare=auto
57         opts.setXShareMode("auto");
58         out = CDSTestUtils.runWithArchive(opts);
59         CDSTestUtils.checkMappingFailure(out);
60         out.shouldMatch("(java|openjdk) version")
61             .shouldNotContain("sharing")
62             .shouldHaveExitValue(0);
63     }
64 }
