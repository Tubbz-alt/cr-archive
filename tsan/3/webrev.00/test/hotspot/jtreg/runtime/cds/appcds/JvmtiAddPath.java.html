 25 /*
 26  * @test
 27  * @summary JvmtiEnv::AddToBootstrapClassLoaderSearch and JvmtiEnv::AddToSystemClassLoaderSearch should disable AppCDS
 28  * @requires vm.cds
 29  * @bug 8060592
 30  * @library /test/lib
 31  * @build sun.hotspot.WhiteBox
 32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 33  * @compile test-classes/Hello.java
 34  * @compile test-classes/JvmtiApp.java
 35  * @run driver JvmtiAddPath
 36  */
 37 
 38 import java.io.File;
 39 import jdk.test.lib.process.OutputAnalyzer;
 40 import sun.hotspot.WhiteBox;
 41 
 42 public class JvmtiAddPath {
 43     static String use_whitebox_jar;
 44     static String[] no_extra_matches = {};
 45     static String[] check_appcds_enabled = {
 46         "[class,load] ExtraClass source: shared object"
 47     };
 48     static String[] check_appcds_disabled = {
 49         "[class,load] ExtraClass source: file:"
 50     };
 51 
 52     static void run(String cp, String... args) throws Exception {
 53         run(no_extra_matches, cp, args);
 54     }
 55 
 56     static void run(String[] extra_matches, String cp, String... args) throws Exception {
 57         String[] opts = {"-cp", cp, "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI", use_whitebox_jar};
 58         opts = TestCommon.concat(opts, args);
 59         TestCommon.run(opts).assertNormalExit(extra_matches);
 60     }
 61 
 62     public static void main(String[] args) throws Exception {
 63         JarBuilder.build("jvmti_addboot", "Hello");
 64         JarBuilder.build("jvmti_addapp", "Hello");
 65         JarBuilder.build("jvmti_app", "JvmtiApp", "ExtraClass");
 66         JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
 67 
 68         // In all the test cases below, appJar does not contain Hello.class. Instead, we
 69         // append JAR file(s) that contain Hello.class to the boot classpath, the app
 70         // classpath, or both, and verify that Hello.class is loaded by the expected ClassLoader.
 71         String appJar = TestCommon.getTestJar("jvmti_app.jar");         // contains JvmtiApp.class
 72         String addappJar = TestCommon.getTestJar("jvmti_addapp.jar");   // contains Hello.class
 73         String addbootJar = TestCommon.getTestJar("jvmti_addboot.jar"); // contains Hello.class
 74         String twoAppJars = appJar + File.pathSeparator + addappJar;
 75         String wbJar = TestCommon.getTestJar("WhiteBox.jar");
 76         use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
 77 
 78         TestCommon.testDump(appJar, TestCommon.list("JvmtiApp", "ExtraClass"), use_whitebox_jar);
 79 
 80         System.out.println("Test case 1: not adding any paths - Hello.class should not be found");
 81         run(check_appcds_enabled, appJar, "-Xlog:class+load", "JvmtiApp", "noadd"); // appcds should be enabled
 82 
 83         System.out.println("Test case 2: add to boot classpath only - should find Hello.class in boot loader");
 84         String[] toCheck = (TestCommon.isDynamicArchive()) ? check_appcds_enabled
 85                                                            : check_appcds_disabled;
 86         run(toCheck, appJar, "-Xlog:class+load", "JvmtiApp", "bootonly", addbootJar); // appcds should be disabled
 87 
 88         System.out.println("Test case 3: add to app classpath only - should find Hello.class in app loader");
 89         run(appJar, "JvmtiApp", "apponly", addappJar);
 90 
 91         System.out.println("Test case 4: add to boot and app paths - should find Hello.class in boot loader");
 92         run(appJar, "JvmtiApp", "appandboot", addbootJar, addappJar);
 93 
 94         System.out.println("Test case 5: add to app using -cp, but add to boot using JVMTI - should find Hello.class in boot loader");
 95         run(twoAppJars, "JvmtiApp", "bootonly", addappJar);
 96 
 97         System.out.println("Test case 6: add to app using AppCDS, but add to boot using JVMTI - should find Hello.class in boot loader");
 98         TestCommon.testDump(twoAppJars, TestCommon.list("JvmtiApp", "ExtraClass", "Hello"), use_whitebox_jar);
 99         if (!TestCommon.isDynamicArchive()) {
100             // skip for dynamic archive, the Hello class will be loaded from
101             // the dynamic archive
102             run(twoAppJars, "JvmtiApp", "bootonly", addappJar);
103         }
104 
105         System.out.println("Test case 7: add to app using AppCDS, no JVMTI calls - should find Hello.class in app loader");
106         run(twoAppJars, "JvmtiApp", "noadd-appcds");
107     }
108 }
