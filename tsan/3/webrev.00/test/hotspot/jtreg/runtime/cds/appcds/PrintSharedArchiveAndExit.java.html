 25 /*
 26  * @test
 27  * @summary test the -XX:+PrintSharedArchiveAndExit flag
 28  * @requires vm.cds
 29  * @library /test/lib
 30  * @compile test-classes/Hello.java
 31  * @compile test-classes/HelloMore.java
 32  * @run main/othervm/timeout=3600 PrintSharedArchiveAndExit
 33  */
 34 
 35 import java.io.File;
 36 import jdk.test.lib.process.OutputAnalyzer;
 37 
 38 public class PrintSharedArchiveAndExit {
 39   private static void check(OutputAnalyzer output, int ret, boolean checkContain, String... matches) throws Exception {
 40     // Tests specific to this test
 41     TestCommon.checkExecReturn(output, ret, checkContain, matches);
 42 
 43     // In all test case, we should never print out the following due to
 44     // PrintSharedArchiveAndExit. JVM should have been terminated
 45     // before reaching these outputs.
 46     TestCommon.checkExecReturn(output, ret, false,
 47                                "Usage:",            // JVM help message
 48                                "java version",      // JVM version
 49                                "Hello World");      // output from the Hello.class in hello.jar
 50   }
 51 
 52   private static void log(String msg) {
 53     System.out.println(">---------------------------------------------------------------------");
 54     System.out.println(msg);
 55     System.out.println("<---------------------------------------------------------------------");
 56   }
 57 
 58   public static void main(String[] args) throws Exception {
 59     String appJar = JarBuilder.getOrCreateHelloJar();
 60     String appJar2 = JarBuilder.build("PrintSharedArchiveAndExit-more", "HelloMore");
 61 
 62     String cp = appJar + File.pathSeparator + appJar2;
 63     String lastCheckMsg = "checking shared classpath entry: " + appJar2; // the last JAR to check
 64 
 65     TestCommon.testDump(cp, TestCommon.list("Hello", "HelloMore"));
 66 
 67     log("Normal execution -- all the JAR paths should be checked");
 68     TestCommon.run(
 69         "-cp", cp,
 70         "-XX:+PrintSharedArchiveAndExit")
 71       .ifNoMappingFailure(output -> check(output, 0, true, lastCheckMsg));
 72 
 73     TestCommon.run(
 74         "-cp", cp,
 75         "-XX:+PrintSharedArchiveAndExit",
 76         "-XX:+PrintSharedDictionary")  // Test PrintSharedDictionary as well.
 77       .ifNoMappingFailure(output -> check(output, 0, true, lastCheckMsg, "java.lang.Object"));
 78 
 79     log("Normal execution -- Make sure -version, help message and app main()\n" +
 80         "class are not invoked. These are checked inside check().");
 81     TestCommon.run("-cp", cp, "-XX:+PrintSharedArchiveAndExit", "-version")
 82       .ifNoMappingFailure(output -> check(output, 0, true, lastCheckMsg));
 83 
 84     TestCommon.run("-cp", cp, "-XX:+PrintSharedArchiveAndExit", "-help")
 85       .ifNoMappingFailure(output -> check(output, 0, true, lastCheckMsg));
 86 
 87     TestCommon.run("-cp", cp, "-XX:+PrintSharedArchiveAndExit", "Hello")
 88       .ifNoMappingFailure(output -> check(output, 0, true, lastCheckMsg));
 89 
 90     log("Execution with simple errors -- with 'simple' errors like missing or modified\n" +
 91         "JAR files, the VM should try to continue to print the remaining information.\n" +
 92         "Use an invalid Boot CP -- all the JAR paths should be checked");
 93     TestCommon.run(
 94         "-cp", cp,
 95         "-Xbootclasspath/a:foo.jar",
 96         "-XX:+PrintSharedArchiveAndExit")
 97       .ifNoMappingFailure(output -> check(output, 1, true, lastCheckMsg, "[BOOT classpath mismatch, "));
 98 
 99     log("Use an App CP shorter than the one at dump time -- all the JAR paths should be checked");
100     TestCommon.run(
101         "-cp", ".",
102         "-XX:+PrintSharedArchiveAndExit")
103       .ifNoMappingFailure(output -> check(output, 1, true, lastCheckMsg, "Run time APP classpath is shorter than the one at dump time: ."));
104 
105     log("Use an invalid App CP -- all the JAR paths should be checked.\n" +
106         "Non-existing jar files will be ignored.");
107     String invalidCP = "non-existing-dir" + File.pathSeparator + cp;
108     TestCommon.run(
109         "-cp", invalidCP,
110         "-XX:+PrintSharedArchiveAndExit")
111       .ifNoMappingFailure(output -> check(output, 0, true, lastCheckMsg));
112 
113     log("Changed modification time of hello.jar -- all the JAR paths should be checked");
114     (new File(appJar)).setLastModified(System.currentTimeMillis() + 2000);
115     TestCommon.run(
116         "-cp", cp,
117         "-XX:+PrintSharedArchiveAndExit")
118       .ifNoMappingFailure(output -> check(output, 1, true, lastCheckMsg, "[Timestamp mismatch]"));
119 
120     log("Even if hello.jar is out of date, we should still be able to print the dictionary.");
121     TestCommon.run(
122         "-cp", cp,
123         "-XX:+PrintSharedArchiveAndExit",
124         "-XX:+PrintSharedDictionary")  // Test PrintSharedDictionary as well.
125       .ifNoMappingFailure(output -> check(output, 1, true, lastCheckMsg, "java.lang.Object"));
126 
127 
128     log("Remove hello.jar -- all the JAR paths should be checked");
129     (new File(appJar)).delete();
130     TestCommon.run(
131         "-cp", cp,
132         "-XX:+PrintSharedArchiveAndExit")
133       .ifNoMappingFailure(output -> check(output, 1, true, lastCheckMsg, "[Required classpath entry does not exist: " + appJar + "]"));
134 
135     log("Execution with major errors -- with 'major' errors like the JSA file\n" +
136         "is missing, we should stop immediately to avoid crashing the JVM.");
137     TestCommon.run(
138         "-cp", cp,
139         "-XX:+PrintSharedArchiveAndExit",
140         "-XX:SharedArchiveFile=./no-such-fileappcds.jsa")
141       .ifNoMappingFailure(output -> check(output, 1, false, lastCheckMsg));
142   }
143 }
