 25 /*
 26  * @test
 27  * @summary Similar to GCDuringDumping.java, this test adds the -XX:SharedArchiveConfigFile
 28  *          option for testing the interaction with GC and shared strings.
 29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes
 30  * @requires vm.cds.archived.java.heap
 31  * @build sun.hotspot.WhiteBox
 32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 33  * @run driver/timeout=480 GCSharedStringsDuringDump
 34  */
 35 
 36 import java.io.File;
 37 import java.io.FileOutputStream;
 38 import java.io.OutputStreamWriter;
 39 import java.io.PrintWriter;
 40 import jdk.test.lib.cds.CDSOptions;
 41 import jdk.test.lib.process.OutputAnalyzer;
 42 import jdk.test.lib.process.ProcessTools;
 43 
 44 public class GCSharedStringsDuringDump {
 45     public static String appClasses[] = {
 46         GCSharedStringsDuringDumpWb.class.getName(),
 47     };
 48     public static String agentClasses[] = {
 49         GCDuringDumpTransformer.class.getName(),
 50     };
 51 
 52     public static void main(String[] args) throws Throwable {
 53         String agentJar =
 54             ClassFileInstaller.writeJar("GCDuringDumpTransformer.jar",
 55                                         ClassFileInstaller.Manifest.fromSourceFile("GCDuringDumpTransformer.mf"),
 56                                         agentClasses);
 57 
 58         String appJar =
 59             ClassFileInstaller.writeJar("GCSharedStringsDuringDumpApp.jar", appClasses);
 60 
 61         String gcLog = Boolean.getBoolean("test.cds.verbose.gc") ?
 62             "-Xlog:gc*=info,gc+region=trace,gc+alloc+region=debug" : "-showversion";
 63 
 64         String sharedArchiveCfgFile =
 65             System.getProperty("user.dir") + File.separator + "GCSharedStringDuringDump_gen.txt";
 66         try (FileOutputStream fos = new FileOutputStream(sharedArchiveCfgFile)) {
 67             PrintWriter out = new PrintWriter(new OutputStreamWriter(fos));
 68             out.println("VERSION: 1.0");
 69             out.println("@SECTION: String");
 70             out.println("31: shared_test_string_unique_14325");
 71             for (int i=0; i<100000; i++) {
 72                 String s = "generated_string " + i;
 73                 out.println(s.length() + ": " + s);
 74             }
 75             out.close();
 76         }
 77 
 78         JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
 79         String whiteBoxJar = TestCommon.getTestJar("WhiteBox.jar");
 80         String bootClassPath = "-Xbootclasspath/a:" + whiteBoxJar;
 81 
 82         for (int i=0; i<2; i++) {
 83             // i = 0 -- run without agent = no extra GCs
 84             // i = 1 -- run with agent = cause extra GCs
 85 
 86             String extraArg = (i == 0) ? "-showversion" : "-javaagent:" + agentJar;
 87             String extraOption = (i == 0) ? "-showversion" : "-XX:+AllowArchivingWithJavaAgent";
 88             OutputAnalyzer output = TestCommon.dump(
 89                                 appJar, TestCommon.list(GCSharedStringsDuringDumpWb.class.getName()),
 90                                 bootClassPath, extraArg, "-Xmx32m", gcLog,
 91                                 "-XX:SharedArchiveConfigFile=" + sharedArchiveCfgFile,
 92                                 "-XX:+UnlockDiagnosticVMOptions", extraOption);
 93 
 94             if (output.getStdout().contains("Too many string space regions") ||
 95                 output.getStderr().contains("Unable to write archive heap memory regions") ||
 96                 output.getStdout().contains("Try increasing NewSize") ||
 97                 !output.getStdout().contains("oa0 space:") ||
 98                 output.getExitValue() != 0) {
 99                 // Try again with larger heap and NewSize, this should increase the
100                 // G1 heap region size to 2M
101                 TestCommon.testDump(
102                     appJar, TestCommon.list(GCSharedStringsDuringDumpWb.class.getName()),
103                     bootClassPath, extraArg, "-Xmx8g", "-XX:NewSize=8m", gcLog,
104                     "-XX:SharedArchiveConfigFile=" + sharedArchiveCfgFile,
105                     "-XX:+UnlockDiagnosticVMOptions", extraOption);
106             }
107 
108             TestCommon.run(
109                 "-cp", appJar,
110                 bootClassPath,
111                 extraArg,
112                 "-Xlog:cds=info,class+path=info",
113                 "-Xmx32m",
114                 "-XX:+PrintSharedSpaces",
115                 "-XX:+UnlockDiagnosticVMOptions",
116                 extraOption,
117                 "-XX:+WhiteBoxAPI",
118                 "-XX:SharedReadOnlySize=30m",
119                 gcLog,
120                 GCSharedStringsDuringDumpWb.class.getName())
121               .assertNormalExit();
122         }
123     }
124 }
