 25 /**
 26  * @test
 27  * @requires vm.cds & !vm.graal.enabled
 28  * @library ../..
 29  * @library /test/lib
 30  * @compile src/jdk/test/Main.java
 31  * @compile src/com/sun/tools/javac/Main.jasm
 32  * @compile src/com/sun/tools/javac/MyMain.jasm
 33  * @compile ../../../javax/annotation/processing/FilerException.jasm
 34  * @run driver ClassPathTests
 35  * @summary AppCDS tests for testing classpath/package conflicts
 36  */
 37 
 38 /*
 39  * These tests will verify that AppCDS will correctly handle archived classes
 40  * on the classpath that are in a package that is also exported by the jimage.
 41  * These classes should fail to load unless --limit-modules is used to hide the
 42  * package exported by the jimage. There are 8 variants of this test:
 43  *   - With a jimage app package and with a jimage ext package
 44  *   - With --limit-modules and without --limit-modules
 45  *   - With AppCDS and without AppCDS (to verify behaviour is the same for both).
 46  *
 47  * There is also a 9th test to verify that when --limit-modules is used, a jimage
 48  * class in the archive can be replaced by a classpath class with the
 49  * same name and package.
 50  */
 51 
 52 import java.lang.reflect.InvocationTargetException;
 53 import java.lang.reflect.Method;
 54 import java.nio.file.Path;
 55 import java.nio.file.Paths;
 56 
 57 import jdk.test.lib.Asserts;
 58 import jdk.test.lib.cds.CDSOptions;
 59 import jdk.test.lib.cds.CDSTestUtils;

 63 import jtreg.SkippedException;

 66 public class ClassPathTests {
 67     private static final String TEST_SRC = System.getProperty("test.src");
 68     private static final Path SRC_DIR = Paths.get(TEST_SRC, "src");
 69     private static final Path CLASSES_DIR = Paths.get("classes");
 70 
 71     // the test module
 72     private static final String MAIN_CLASS = "jdk.test.Main";
 73     private static final String LIMITMODS_MAIN_CLASS = "jdk.test.LimitModsMain";
 74 
 75     // test classes to archive. These are both in UPGRADED_MODULES
 76     private static final String JIMAGE_CLASS      = "com/sun/tools/javac/Main";
 77     private static final String APP_ARCHIVE_CLASS = "com/sun/tools/javac/MyMain";
 78     private static final String PLATFORM_ARCHIVE_CLASS = "javax/annotation/processing/FilerException";
 79     private static final String[] ARCHIVE_CLASSES = {APP_ARCHIVE_CLASS, PLATFORM_ARCHIVE_CLASS, JIMAGE_CLASS};
 80     private static final int NUMBER_OF_TEST_CASES = 10;
 81 
 82     private static String appJar;
 83     private static String testArchiveName;
 84 
 85 
 86     public static void main(String[] args) throws Throwable {
 87         ClassPathTests tests = new ClassPathTests();
 88         tests.dumpArchive();
 89 
 90         Method[] methods = tests.getClass().getDeclaredMethods();
 91         int numOfTestMethodsRun = 0;
 92         for (Method m : methods) {
 93             if (m.getName().startsWith("test")) {
 94                 System.out.println("About to run test method: " + m.getName());
 95                 try {
 96                     m.invoke(tests);
 97                 } catch (InvocationTargetException ite) {
 98                     Throwable throwable = ite.getCause();
 99                     if (throwable instanceof SkippedException) {
100                         throw throwable;
101                     } else {
102                         throw ite;
103                     }
104                 }
105                 numOfTestMethodsRun++;
106             }
107         }
108 
109         Asserts.assertTrue((numOfTestMethodsRun == NUMBER_OF_TEST_CASES),
110             "Expected " + NUMBER_OF_TEST_CASES + " test methods to run, actual number is "
111             + numOfTestMethodsRun);
112     }
113 
114     private void dumpArchive() throws Exception {
115         // Create a jar file with all the classes related to this test.
116         JarBuilder.build( "classpathtests",
117                           APP_ARCHIVE_CLASS, PLATFORM_ARCHIVE_CLASS, JIMAGE_CLASS,
118                           "jdk/test/Main");
119         appJar = TestCommon.getTestJar("classpathtests.jar");
120 
121         // dump the archive with altnernate jdk.comiler and jdk.activation classes in the class list
122         OutputAnalyzer output1  = TestCommon.dump(appJar, TestCommon.list(ARCHIVE_CLASSES));
123         TestCommon.checkDump(output1);
124         // The PLATFORM_ARCHIVE_CLASS belongs to the java.compiler
125         // module will be found from the module during dumping.
126         // The JIMAGE_CLASS will be found from the jdk.compiler module during
127         // dumping.
128         // The APP_ARCHIVE_CLASS, which belongs to a package within the
129         // jdk.compiler module, will not be found during dump time.
130         for (String archiveClass : ARCHIVE_CLASSES) {
131             if (archiveClass.equals(APP_ARCHIVE_CLASS)) {
132                 output1.shouldContain("Preload Warning: Cannot find " + archiveClass);
133             } else {
134                 output1.shouldNotContain("Preload Warning: Cannot find " + archiveClass);
135             }
136         }
137 
138         testArchiveName = TestCommon.getCurrentArchiveName();
139     }

141     // #1: Archived classpath class in same package as jimage app class. With AppCDS.
142     // Should fail to load.
143     public void testAppClassWithAppCDS() throws Exception {
144         OutputAnalyzer output = TestCommon.exec(
145             appJar, MAIN_CLASS,
146             "Test #1", APP_ARCHIVE_CLASS, "false"); // last 3 args passed to test
147         TestCommon.checkExec(output);
148     }
149 
150     // #2: Archived classpath class in same package as jimage app class. Without AppCDS.
151     // Should fail to load.
152     public void testAppClassWithoutAppCDS() throws Exception {
153         CDSOptions opts = (new CDSOptions())
154             .addPrefix("-cp", appJar)
155             .setArchiveName(testArchiveName)
156             .addSuffix(MAIN_CLASS, "Test #2", APP_ARCHIVE_CLASS, "false");
157 
158         CDSTestUtils.runWithArchiveAndCheck(opts);
159     }

181     // #5: Archived classpath class in same package as jimage app class. With AppCDS.
182     // Should load with CDS disabled because --limit-modules is used.
183     public void testAppClassWithLimitModsWithAppCDS() throws Exception {
184         TestCommon.run("-cp", appJar,
185                        "--limit-modules", "java.base",
186                        MAIN_CLASS,
187                        "Test #5", APP_ARCHIVE_CLASS, "true") // last 3 args passed to test
188             .assertSilentlyDisabledCDS(out -> {
