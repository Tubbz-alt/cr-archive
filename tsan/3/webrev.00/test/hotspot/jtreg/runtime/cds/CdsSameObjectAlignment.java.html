24 /**
25  * @test CdsSameObjectAlignment
26  * @summary Testing CDS (class data sharing) using varying object alignment.
27  *          Using same object alignment for each dump/load pair
28  * @requires vm.cds
29  * @requires vm.bits == 64
30  * @library /test/lib
31  * @run driver CdsSameObjectAlignment
32  */
33 
34 import jdk.test.lib.Platform;
35 import jdk.test.lib.cds.CDSTestUtils;
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class CdsSameObjectAlignment {
39     public static void main(String[] args) throws Exception {
40         dumpAndLoadSharedArchive(8);
41         dumpAndLoadSharedArchive(16);
42         dumpAndLoadSharedArchive(32);
43         dumpAndLoadSharedArchive(64);
44     }
45 
46     private static void
47     dumpAndLoadSharedArchive(int objectAlignmentInBytes) throws Exception {
48         String objectAlignmentArg = "-XX:ObjectAlignmentInBytes="
49             + objectAlignmentInBytes;
50         System.out.println("dumpAndLoadSharedArchive(): objectAlignmentInBytes = "
51             + objectAlignmentInBytes);
52 
53         CDSTestUtils.createArchiveAndCheck(objectAlignmentArg);
54         CDSTestUtils.runWithArchiveAndCheck(objectAlignmentArg);
55     }
56 }
