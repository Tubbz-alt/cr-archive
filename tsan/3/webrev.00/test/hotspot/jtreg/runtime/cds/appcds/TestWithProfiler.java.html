25 /**
26  * @test
27  * @bug 8145221
28  * @summary After creating an AppCDS archive, run the test with the JFR profiler
29  *          enabled, and keep calling a method in the archive in a tight loop.
30  *          This is to test the safe handling of trampoline functions by the
31  *          profiler.
32  * @requires vm.hasJFR & vm.cds
33  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
34  * @compile test-classes/MyThread.java
35  * @compile test-classes/TestWithProfilerHelper.java
36  * @run driver TestWithProfiler
37  */
38 
39 import jdk.test.lib.BuildHelper;
40 import jdk.test.lib.process.OutputAnalyzer;
41 
42 public class TestWithProfiler {
43     public static void main(String[] args) throws Exception {
44         JarBuilder.build("myThread", "MyThread", "TestWithProfilerHelper");
45         String appJar = TestCommon.getTestJar("myThread.jar");
46         OutputAnalyzer output = TestCommon.dump(appJar,
47             TestCommon.list("MyThread", "TestWithProfilerHelper"));
48         TestCommon.checkDump(output);
49         output = TestCommon.exec(appJar,
50             "-XX:+UnlockDiagnosticVMOptions",
51             "-Xint",
52             "-XX:+FlightRecorder",
53             "-XX:StartFlightRecording=duration=15s,filename=myrecording.jfr,settings=profile,dumponexit=true",
54             "TestWithProfilerHelper");
55         TestCommon.checkExec(output);
56     }
57 }
