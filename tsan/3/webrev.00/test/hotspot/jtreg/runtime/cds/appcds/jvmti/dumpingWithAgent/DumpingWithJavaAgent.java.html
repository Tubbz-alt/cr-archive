 25 /*
 26  * @test
 27  * @summary CDS dumping with java agent.
 28  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes
 29  * @requires vm.cds
 30  * @requires vm.flavor != "minimal"
 31  * @build SimpleAgent Hello
 32  * @run main/othervm DumpingWithJavaAgent
 33  */
 34 
 35 import jdk.test.lib.cds.CDSOptions;
 36 import jdk.test.lib.process.OutputAnalyzer;
 37 import jdk.test.lib.process.ProcessTools;
 38 
 39 public class DumpingWithJavaAgent {
 40     public static String appClasses[] = {
 41         "Hello",
 42     };
 43     public static String agentClasses[] = {
 44         "SimpleAgent",
 45     };
 46 
 47     public static String warningMessages[] = {
 48         "This archive was created with AllowArchivingWithJavaAgent",
 49         "It should be used for testing purposes only and should not be used in a production environment",
 50     };
 51 
 52     public static String errorMessage =
 53         "The setting of the AllowArchivingWithJavaAgent is different from the setting in the shared archive.";
 54 
 55 
 56     public static String diagnosticOption = "-XX:+AllowArchivingWithJavaAgent";
 57 
 58     public static void main(String[] args) throws Throwable {
 59         String agentJar =
 60             ClassFileInstaller.writeJar("SimpleAgent.jar",
 61                                         ClassFileInstaller.Manifest.fromSourceFile("SimpleAgent.mf"),
 62                                         agentClasses);
 63 
 64         String appJar =
 65             ClassFileInstaller.writeJar("DumpingWithJavaAgent.jar", appClasses);
 66 
 67         // CDS dumping with a java agent with the AllowArchvingWithJavaAgent diagnostic option.
 68         OutputAnalyzer output = TestCommon.testDump(appJar, TestCommon.list("Hello"),
 69             "-XX:+UnlockDiagnosticVMOptions", diagnosticOption,
 70             "-javaagent:" + agentJar);
 71         TestCommon.checkDump(output);
 72         output.shouldContain(warningMessages[0]);
 73         output.shouldContain(warningMessages[1]);
 74         output.shouldContain("inside SimpleAgent");
 75 
 76         // Using the archive with the AllowArchvingWithJavaAgent diagnostic option.
 77         output = TestCommon.exec(
 78             appJar,
 79             "-Xlog:class+load=trace",
 80             "-XX:+UnlockDiagnosticVMOptions", diagnosticOption,
 81             "Hello");
 82         if (!TestCommon.isUnableToMap(output)) {
 83             output.shouldHaveExitValue(0);
 84             output.shouldContain(warningMessages[0]);
 85             output.shouldContain(warningMessages[1]);
 86             output.shouldContain("[class,load] Hello source: shared objects file");
 87         }
 88 
 89         // Using the archive with -Xshare:on without the diagnostic option.
 90         // VM should exit with an error message.
 91         output = TestCommon.exec(
 92             appJar,
 93             "Hello");
 94         output.shouldHaveExitValue(1);
 95         output.shouldContain(errorMessage);
 96 
 97         // Using the archive with -Xshare:auto without the diagnostic option.
 98         // VM should continue execution with a warning message. The archive
 99         // will not be used.
100         output = TestCommon.execAuto(
101             "-cp", appJar,
102             "-Xlog:class+load=trace,cds=info",
103             "Hello");
104         if (!TestCommon.isUnableToMap(output)) {
105             output.shouldHaveExitValue(0);
106             output.shouldContain(errorMessage);
107             output.shouldMatch(".class.load.* Hello source:.*DumpingWithJavaAgent.jar");
108 
109         // CDS dumping with a java agent without the AllowArchvingWithJavaAgent diagnostic option.
110         // VM will exit with an error message.
111         output = TestCommon.dump(appJar, TestCommon.list("Hello"),
112             "-javaagent:" + agentJar);
113         }
114         output.shouldContain("Must enable AllowArchivingWithJavaAgent in order to run Java agent during CDS dumping")
115             .shouldHaveExitValue(1);
116     }
117 }
