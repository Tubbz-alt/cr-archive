25 /*
26  * @test
27  * @summary Initiating and defining classloader test.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/Hello.java
31  * @compile test-classes/HelloWB.java
32  * @compile test-classes/ForNameTest.java
33  * @compile test-classes/BootClassPathAppendHelper.java
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
36  * @run driver ClassLoaderTest
37  */
38 
39 import java.io.File;
40 import jdk.test.lib.process.OutputAnalyzer;
41 
42 public class ClassLoaderTest {
43     public static void main(String[] args) throws Exception {
44         JarBuilder.build(true, "ClassLoaderTest-WhiteBox", "sun/hotspot/WhiteBox");
45         JarBuilder.getOrCreateHelloJar();
46         JarBuilder.build("ClassLoaderTest-HelloWB", "HelloWB");
47         JarBuilder.build("ClassLoaderTest-ForName", "ForNameTest");
48         ClassLoaderTest test = new ClassLoaderTest();
49         test.testBootLoader();
50         test.testDefiningLoader();
51     }
52 
53     public void testBootLoader() throws Exception {
54         String appJar = TestCommon.getTestJar("ClassLoaderTest-HelloWB.jar");
55         String appClasses[] = {"HelloWB"};
56         String whiteBoxJar = TestCommon.getTestJar("ClassLoaderTest-WhiteBox.jar");
57         String bootClassPath = "-Xbootclasspath/a:" + appJar +
58             File.pathSeparator + whiteBoxJar;
59 
60         TestCommon.dump(appJar, appClasses, bootClassPath);
61 
62         TestCommon.run(
63             "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
64             "-cp", appJar, bootClassPath, "HelloWB")
65           .assertNormalExit(output -> output.shouldContain("HelloWB.class.getClassLoader() = null"));
66     }
67 
68     public void testDefiningLoader() throws Exception {
69         // The boot loader should be used to load the class when it's
70         // on the bootclasspath, regardless who is the initiating classloader.
71         // In this test case, the AppClassLoader is the initiating classloader.
72         String helloJar = TestCommon.getTestJar("hello.jar");
73         String appJar = helloJar + System.getProperty("path.separator") +
74                         TestCommon.getTestJar("ClassLoaderTest-ForName.jar");
75         String whiteBoxJar = TestCommon.getTestJar("ClassLoaderTest-WhiteBox.jar");
76         String bootClassPath = "-Xbootclasspath/a:" + helloJar +
77             File.pathSeparator + whiteBoxJar;
78 
79         // Archive the "Hello" class from the appended bootclasspath
80         TestCommon.dump(helloJar, TestCommon.list("Hello"), bootClassPath);
81 
82         TestCommon.run("-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
83             "-cp", appJar, bootClassPath, "-Xlog:class+path=trace", "ForNameTest")
84           .assertNormalExit();
85     }
86 }
