25 /*
26  * @test
27  * @requires vm.cds
28  * @summary sharing is disabled if java.base is patch at runtime
29  * @library ../..
30  * @library /test/hotspot/jtreg/testlibrary
31  * @library /test/lib
32  * @build PatchMain
33  * @run driver PatchJavaBase
34  */
35 
36 import jdk.test.lib.compiler.InMemoryJavaCompiler;
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class PatchJavaBase {
40     private static String moduleJar;
41 
42     public static void main(String args[]) throws Throwable {
43 
44         String source = "package java.lang; "                       +
45                         "public class NewClass { "                  +
46                         "    static { "                             +
47                         "        System.out.println(\"I pass!\"); " +
48                         "    } "                                    +
49                         "}";
50 
51         ClassFileInstaller.writeClassToDisk("java/lang/NewClass",
52              InMemoryJavaCompiler.compile("java.lang.NewClass", source, "--patch-module=java.base"),
53              System.getProperty("test.classes"));
54 
55         JarBuilder.build("javabase", "java/lang/NewClass");
56         moduleJar = TestCommon.getTestJar("javabase.jar");
57 
58         System.out.println("Test dumping with --patch-module");
59         String runError = "Unable to use shared archive: CDS is disabled when java.base module is patched";
60         String dumpingError = "Cannot use the following option when dumping the shared archive: --patch-module";
61         String errMsg;
62         if (TestCommon.isDynamicArchive()) {
63             errMsg = runError;
64         } else {
65             errMsg = dumpingError;
66         }
67         OutputAnalyzer output =
68             TestCommon.dump(null, null,
69                 "--patch-module=java.base=" + moduleJar,
70                 "PatchMain", "java.lang.NewClass");
71         output.shouldHaveExitValue(1)
72               .shouldContain(errMsg);
73 
74         TestCommon.run(
75             "-XX:+UnlockDiagnosticVMOptions",
76             "--patch-module=java.base=" + moduleJar,
77             "PatchMain", "java.lang.NewClass")
78           .assertAbnormalExit(runError);
79     }
80 }
