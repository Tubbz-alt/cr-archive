25 /*
26  * @test
27  * @summary Make sure prohibited packages cannot be stored into archive for custom loaders.
28  * @requires vm.cds
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
31  * @compile ClassListFormatBase.java ../test-classes/Hello.java test-classes/InProhibitedPkg.java
32  * @run driver ProhibitedPackageNamesTest
33  */
34 
35 public class ProhibitedPackageNamesTest extends ClassListFormatBase {
36     static {
37         // Uncomment the following line to run only one of the test cases
38         // ClassListFormatBase.RUN_ONLY_TEST = "TESTCASE PPN1";
39     }
40 
41     public static void main(String[] args) throws Throwable {
42         String appJar = JarBuilder.getOrCreateHelloJar();
43         String customJarPath = JarBuilder.build("ProhibitedPackageNames_custom", "java/InProhibitedPkg");
44 
45         dumpShouldPass(
46             "TESTCASE PPN1: prohibited package name without loader:",
47             appJar, classlist(
48                 "Hello",
49                 "java/lang/Object id: 1",
50                 // Without "loader:" keyword.
51                 "java/InProhibitedPkg id: 2 super: 1 source: " + customJarPath
52             ),
53             "Prohibited package for non-bootstrap classes: java/InProhibitedPkg.class");
54     }
55 }
