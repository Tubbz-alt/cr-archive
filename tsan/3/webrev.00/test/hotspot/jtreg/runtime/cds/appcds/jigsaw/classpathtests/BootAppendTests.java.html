 95     static void dumpArchive() throws Exception {
 96         JarBuilder.build("classpathtests", "jdk/test/Main");
 97         appJar = TestCommon.getTestJar("classpathtests.jar");
 98 
 99         JarBuilder.build("bootAppend",
100                          APP_MODULE_CLASS, BOOT_APPEND_MODULE_CLASS, BOOT_APPEND_CLASS);
101         bootAppendJar = TestCommon.getTestJar("bootAppend.jar");
102 
103         OutputAnalyzer output1  = TestCommon.dump(
104             appJar, TestCommon.list(ARCHIVE_CLASSES), "-Xbootclasspath/a:" + bootAppendJar);
105         TestCommon.checkDump(output1);
106 
107         if (!TestCommon.isUnableToMap(output1)) {
108             // Make sure all the classes were successfully archived.
109             for (String archiveClass : ARCHIVE_CLASSES) {
110                 output1.shouldNotContain("Preload Warning: Cannot find " + archiveClass);
111             }
112         }
113 
114         testArchiveName = TestCommon.getCurrentArchiveName();
115     }

117     // Test #1: A class in package defined in boot module
118     //    - should not be loaded from the -Xbootclasspath/a without AppCDS
119     public static void testBootAppendModuleClassWithoutAppCDS() throws Exception {
120         CDSOptions opts = (new CDSOptions())
121             .addPrefix("-Xbootclasspath/a:" + bootAppendJar, "-cp", appJar)
122             .setArchiveName(testArchiveName)
123             .addSuffix(MAIN_CLASS, "Test #1", BOOT_APPEND_MODULE_CLASS, "false");
124 
125         CDSTestUtils.runWithArchiveAndCheck(opts);
126     }
127

129     //          from -Xbootclasspath/a
130     //     - should not be loaded by AppCDS
131     public static void testBootAppendModuleClassWithAppCDS() throws Exception {
132         OutputAnalyzer output = TestCommon.exec(

134             "-Xbootclasspath/a:" + bootAppendJar,
135             MAIN_CLASS,
136             "Test #2", BOOT_APPEND_MODULE_CLASS, "false");
137         TestCommon.checkExec(output);

141     // Test #3: A class in excluded package defined in boot module
142     //     - should be loaded from the -Xbootclasspath/a by the boot classloader
143     public static void testBootAppendExcludedModuleClassWithoutAppCDS() throws Exception {
144         TestCommon.run(
145             "-Xbootclasspath/a:" + bootAppendJar, "-cp", appJar,
146             "-Xlog:class+load=info",
147             "--limit-modules", "java.base",
148             MAIN_CLASS, "Test #3", BOOT_APPEND_MODULE_CLASS, "true", "BOOT")
149             .assertSilentlyDisabledCDS(out -> {
150                 out.shouldHaveExitValue(0)
151                    .shouldMatch(".class.load. sun.nio.cs.ext.MyClass source:.*bootAppend.jar");
152             });
153     }
