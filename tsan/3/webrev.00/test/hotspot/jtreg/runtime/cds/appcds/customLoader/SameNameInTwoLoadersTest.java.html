25 /*
26  * @test
27  * @summary Testing the loading of a class with the same name in two different class loaders.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  *
32  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
33  * @compile test-classes/CustomLoadee.java
34  *     test-classes/CustomLoadee3.java
35  *     test-classes/SameNameUnrelatedLoaders.java
36  * @build sun.hotspot.WhiteBox
37  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
38  * @run driver SameNameInTwoLoadersTest
39  */
40 
41 import jdk.test.lib.process.OutputAnalyzer;
42 import sun.hotspot.WhiteBox;
43 
44 
45 public class SameNameInTwoLoadersTest {
46     private static String appJar;
47     private static String customJar;
48     private static String useWbParam;
49 
50     public static void main(String[] args) throws Exception {
51         appJar = JarBuilder.build("SameNameInTwoLoadersTest",
52             "SameNameUnrelatedLoaders");
53 
54         customJar = JarBuilder.build("SameNameInTwoLoadersTest_custom", "CustomLoadee", "CustomLoadee3");
55 
56         useWbParam = "-Xbootclasspath/a:" +
57             JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");;
58 
59         // ====== unrelated loaders
60         executeTestCase(getClassList_FP(),
61             "SameNameUnrelatedLoaders", "FpBoth");
62     }
63 
64     private static void executeTestCase(String[] classlist,
65         String testClass, String testCaseId) throws Exception {
66         classlist[0] = testClass;
67 
68         TestCommon.testDump(appJar, classlist, useWbParam);
69 
70         OutputAnalyzer output = TestCommon.exec(appJar,
71                                  // command-line arguments ...
72                                  "--add-opens=java.base/java.security=ALL-UNNAMED",
73                                  useWbParam,
74                                  "-XX:+UnlockDiagnosticVMOptions",
75                                  "-XX:+WhiteBoxAPI",
76                                  testClass,
77                                  customJar, testCaseId);
78         TestCommon.checkExec(output);
79     }
80 
81     // Single entry, no loader specified (FP method)
82     private static String[] getClassList_FP() {
83         return new String[] {
84             "SameNameUnrelatedLoaders",
85             "java/lang/Object id: 1",
86             "CustomLoadee id: 10 super: 1 source: " + customJar,
87         };
88     }
89 }
