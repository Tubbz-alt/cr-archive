 25 /*
 26  * @test
 27  * @summary classes with major version < JDK_6 (50) should not be included in CDS
 28  * @requires vm.cds
 29  * @library /test/lib
 30  * @modules java.base/jdk.internal.org.objectweb.asm
 31  * @compile test-classes/Hello.java
 32  * @run build TestCommon JarBuilder
 33  * @run driver OldClassTest
 34  */
 35 
 36 import java.io.File;
 37 import java.io.FileOutputStream;
 38 import jdk.test.lib.process.OutputAnalyzer;
 39 import java.nio.file.Files;
 40 
 41 import java.util.*;
 42 import jdk.internal.org.objectweb.asm.*;
 43 
 44 public class OldClassTest implements Opcodes {
 45 
 46   public static void main(String[] args) throws Exception {
 47     File jarSrcFile = new File(JarBuilder.getOrCreateHelloJar());
 48 
 49     File dir = new File(System.getProperty("test.classes", "."));
 50     File jarFile = new File(dir, "OldClassTest_old.jar");
 51     String jar = jarFile.getPath();
 52 
 53     if (!jarFile.exists() || jarFile.lastModified() < jarSrcFile.lastModified()) {
 54       createTestJarFile(jarSrcFile, jarFile);
 55     } else {
 56       System.out.println("Already up-to-date: " + jarFile);
 57     }
 58 
 59     String appClasses[] = TestCommon.list("Hello");
 60 
 61     // CASE 1: pre-JDK 6 compiled classes should be excluded from the dump
 62     OutputAnalyzer output = TestCommon.dump(jar, appClasses);
 63     TestCommon.checkExecReturn(output, 0, true, "Pre JDK 6 class not supported by CDS");
 64 
 65     TestCommon.run(
 66         "-cp", jar,
 67         "Hello")
 68       .assertNormalExit("Hello Unicode world (Old)");
 69 
 70     // CASE 2: if we exlcude old version of this class, we should not pick up
 71     //         the newer version of this class in a subsequent classpath element.
 72     String classpath = jar + File.pathSeparator + jarSrcFile.getPath();
 73     output = TestCommon.dump(classpath, appClasses);
 74     TestCommon.checkExecReturn(output, 0, true, "Pre JDK 6 class not supported by CDS");
 75 
 76     TestCommon.run(
 77         "-cp", classpath,
 78         "Hello")
 79       .assertNormalExit("Hello Unicode world (Old)");
 80   }
 81 
 82   static void createTestJarFile(File jarSrcFile, File jarFile) throws Exception {
 83     jarFile.delete();
 84     Files.copy(jarSrcFile.toPath(), jarFile.toPath());
 85 
 86     File dir = new File(System.getProperty("test.classes", "."));
 87     File outdir = new File(dir, "old_class_test_classes");
 88     outdir.delete();
 89     outdir.mkdir();
 90 
 91     writeClassFile(new File(outdir, "Hello.class"), makeOldHello());
 92 
 93     JarBuilder.update(jarFile.getPath(), outdir.getPath());
 94   }
 95 
 96   static void writeClassFile(File file, byte bytecodes[]) throws Exception {
 97     try (FileOutputStream fos = new FileOutputStream(file)) {
 98         fos.write(bytecodes);
 99       }
100   }
101 
102 /* makeOldHello() was obtained using JDK8. We use a method name > 128 that would
103    trigger a call to java.lang.Character.isJavaIdentifierStart() during class
104    file parsing.
105 
106 cat > Hello.java <<EOF
107 public class Hello {
108     public static void main(String args[]) {
109         System.out.println(\u1234());
110     }
111     static String \u1234() {
112         return "Hello Unicode world (Old)";
113     }
114 }
115 EOF
116 javac Hello.java
117 java jdk.internal.org.objectweb.asm.util.ASMifier Hello.class
118 
119  */
120 
121   static byte[] makeOldHello() throws Exception {
122     ClassWriter cw = new ClassWriter(0);
123     FieldVisitor fv;
124     MethodVisitor mv;
125     AnnotationVisitor av0;
126 
127       cw.visit(V1_5, ACC_PUBLIC + ACC_SUPER, "Hello", null, "java/lang/Object", null);
128 
129     {
130       mv = cw.visitMethod(ACC_PUBLIC, "<init>", "()V", null, null);
131       mv.visitCode();
132       mv.visitVarInsn(ALOAD, 0);
133       mv.visitMethodInsn(INVOKESPECIAL, "java/lang/Object", "<init>", "()V", false);
134       mv.visitInsn(RETURN);
135       mv.visitMaxs(1, 1);
136       mv.visitEnd();
137     }
138     {
139       mv = cw.visitMethod(ACC_PUBLIC + ACC_STATIC, "main", "([Ljava/lang/String;)V", null, null);
140       mv.visitCode();
141       mv.visitFieldInsn(GETSTATIC, "java/lang/System", "out", "Ljava/io/PrintStream;");
142       mv.visitMethodInsn(INVOKESTATIC, "Hello", "\u1234", "()Ljava/lang/String;", false);
143       mv.visitMethodInsn(INVOKEVIRTUAL, "java/io/PrintStream", "println", "(Ljava/lang/String;)V", false);
144       mv.visitInsn(RETURN);
145       mv.visitMaxs(2, 1);
146       mv.visitEnd();
147     }
148     {
149       mv = cw.visitMethod(ACC_STATIC, "\u1234", "()Ljava/lang/String;", null, null);
150       mv.visitCode();
151       mv.visitLdcInsn("Hello Unicode world (Old)");
152       mv.visitInsn(ARETURN);
153       mv.visitMaxs(1, 0);
154       mv.visitEnd();
155     }
156     cw.visitEnd();
157 
158     return cw.toByteArray();
159   }
