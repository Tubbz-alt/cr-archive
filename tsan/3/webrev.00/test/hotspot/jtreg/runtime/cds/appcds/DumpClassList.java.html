 25 /*
 26  * @test
 27  * @summary DumpLoadedClassList should exclude generated classes, classes in bootclasspath/a and
 28  *          --patch-module.
 29  * @requires vm.cds
 30  * @library /test/lib
 31  * @compile test-classes/ArrayListTest.java
 32  * @run driver DumpClassList
 33  */
 34 
 35 import jdk.test.lib.compiler.InMemoryJavaCompiler;
 36 import jdk.test.lib.process.OutputAnalyzer;
 37 import jdk.test.lib.process.ProcessTools;
 38 
 39 public class DumpClassList {
 40     public static void main(String[] args) throws Exception {
 41         // build The app
 42         String[] appClass = new String[] {"ArrayListTest"};
 43         String classList = "app.list";
 44 
 45         JarBuilder.build("app", appClass[0]);
 46         String appJar = TestCommon.getTestJar("app.jar");
 47 
 48         // build patch-module
 49         String source = "package java.lang; "                       +
 50                         "public class NewClass { "                  +
 51                         "    static { "                             +
 52                         "        System.out.println(\"NewClass\"); "+
 53                         "    } "                                    +
 54                         "}";
 55 
 56         ClassFileInstaller.writeClassToDisk("java/lang/NewClass",
 57              InMemoryJavaCompiler.compile("java.lang.NewClass", source, "--patch-module=java.base"),
 58              System.getProperty("test.classes"));
 59 
 60         String patchJar = JarBuilder.build("javabase", "java/lang/NewClass");
 61 
 62         // build bootclasspath/a
 63         String source2 = "package boot.append; "                 +
 64                         "public class Foo { "                    +
 65                         "    static { "                          +
 66                         "        System.out.println(\"Foo\"); "  +
 67                         "    } "                                 +
 68                         "}";
 69 
 70         ClassFileInstaller.writeClassToDisk("boot/append/Foo",
 71              InMemoryJavaCompiler.compile("boot.append.Foo", source2),
 72              System.getProperty("test.classes"));
 73 
 74         String appendJar = JarBuilder.build("bootappend", "boot/append/Foo");
 75 
 76         // dump class list
 77         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
 78             true,
 79             "-XX:DumpLoadedClassList=" + classList,
 80             "--patch-module=java.base=" + patchJar,
 81             "-Xbootclasspath/a:" + appendJar,
 82             "-cp",
 83             appJar,
 84             appClass[0]);
 85         OutputAnalyzer output = TestCommon.executeAndLog(pb, "dumpClassList");
 86         TestCommon.checkExecReturn(output, 0, true,
 87                                    "hello world",
 88                                    "skip writing class java/lang/NewClass") // skip classes outside of jrt image
 89             .shouldNotContain("skip writing class boot/append/Foo");        // but classes on -Xbootclasspath/a should not be skipped
 90 
 91         output = TestCommon.createArchive(appJar, appClass,
 92                                           "-Xbootclasspath/a:" + appendJar,
 93                                           "-Xlog:class+load",
 94                                           "-XX:SharedClassListFile=" + classList);
 95         TestCommon.checkDump(output)
 96             .shouldNotContain("Preload Warning: Cannot find java/lang/invoke/LambdaForm")
 97             .shouldNotContain("Preload Warning: Cannot find boot/append/Foo")
 98             .shouldContain("[info][class,load] boot.append.Foo");
 99     }
100 }
