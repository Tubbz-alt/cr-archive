25 /*
26  * @test
27  * @summary AppCDS handling of protection domain in custom loaders.
28  *
29  * @requires vm.cds
30  * @requires vm.cds.custom.loaders
31  *
32  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
33  * @compile test-classes/ProtDomain.java
34  * @run driver ProtectionDomain
35  */
36 
37 public class ProtectionDomain {
38     public static void main(String[] args) throws Exception {
39         String appJar = JarBuilder.build("ProtectionDomain-app", "ProtDomain");
40 
41         String customJar = JarBuilder.build("ProtectionDomain-custom",
42             "ProtDomainClassForArchive", "ProtDomainNotForArchive");
43         String[] classlist = new String[] {
44             "java/lang/Object id: 1",
45             "ProtDomain id: 2 super: 1 source: " + appJar,
46             "ProtDomainClassForArchive id: 3 super: 1 source: " + customJar
47         };
48 
49         TestCommon.testDump(appJar, classlist);
50 
51         // First class is loaded from CDS, second class is loaded from JAR
52         TestCommon.checkExec(TestCommon.exec(appJar, "-verbose:class", "ProtDomain", customJar),
53             "[class,load] ProtDomainClassForArchive source: shared objects file",
54             "[class,load] ProtDomainNotForArchive source: file");
55     }
56 }
