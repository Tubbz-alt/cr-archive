 25 /**
 26  * @test
 27  * @requires vm.cds
 28  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 29  * @run driver ModulePathAndCP
 30  * @summary 2 sets of tests: one with only --module-path in the command line;
 31  *          another with both -cp and --module-path in the command line.
 32  */
 33 
 34 import java.io.File;
 35 import java.nio.file.Files;
 36 import java.nio.file.Path;
 37 import java.nio.file.Paths;
 38 
 39 import jdk.test.lib.process.OutputAnalyzer;
 40 
 41 public class ModulePathAndCP {
 42 
 43     private static final Path USER_DIR = Paths.get(System.getProperty("user.dir"));
 44 
 45     private static final String TEST_SRC = System.getProperty("test.src");
 46 
 47     private static final Path SRC_DIR = Paths.get(TEST_SRC, "src");
 48     private static final Path MODS_DIR = Paths.get("mods");
 49 
 50     // the module name of the test module
 51     private static final String MAIN_MODULE = "com.greetings";
 52     private static final String APP_MODULE = "org.astro";
 53 
 54     // the module main class
 55     private static final String MAIN_CLASS = "com.greetings.Main";
 56     private static final String APP_CLASS = "org.astro.World";
 57 
 58     private static Path moduleDir = null;
 59     private static Path moduleDir2 = null;
 60     private static Path subJar = null;
 61     private static Path mainJar = null;
 62     private static Path destJar = null;
 63 
 64     public static void buildTestModule() throws Exception {
 65 
 66         // javac -d mods/$TESTMODULE src/$TESTMODULE/**
 67         JarBuilder.compileModule(SRC_DIR.resolve(APP_MODULE),
 68                                  MODS_DIR.resolve(APP_MODULE),
 69                                  null);
 70 
 71         // javac -d mods/$TESTMODULE --module-path MOD_DIR src/$TESTMODULE/**
 72         JarBuilder.compileModule(SRC_DIR.resolve(MAIN_MODULE),
 73                                  MODS_DIR.resolve(MAIN_MODULE),
 74                                  MODS_DIR.toString());
 75 
 76         moduleDir = Files.createTempDirectory(USER_DIR, "mlib");
 77         moduleDir2 = Files.createTempDirectory(USER_DIR, "mlib2");
 78         subJar = moduleDir.resolve(APP_MODULE + ".jar");
 79         destJar = moduleDir2.resolve(APP_MODULE + ".jar");
 80         String classes = MODS_DIR.resolve(APP_MODULE).toString();
 81         JarBuilder.createModularJar(subJar.toString(), classes, null);
 82         Files.copy(subJar, destJar);
 83 
 84         mainJar = moduleDir.resolve(MAIN_MODULE + ".jar");
 85         Path mainJar2 = moduleDir2.resolve(MAIN_MODULE + ".jar");
 86         classes = MODS_DIR.resolve(MAIN_MODULE).toString();
 87         JarBuilder.createModularJar(mainJar.toString(), classes, MAIN_CLASS);
 88         Files.copy(mainJar, mainJar2);
 89 
 90     }
 91 
 92     public static void main(String... args) throws Exception {
 93         run();
 94     }
 95 
 96     public static void run(String... extra_runtime_args) throws Exception {
 97         // compile the modules and create the modular jar files
 98         buildTestModule();
 99         String appClasses[] = {MAIN_CLASS, APP_CLASS};
100         // create an archive with the classes in the modules built in the
101         // previous step
102         OutputAnalyzer output = TestCommon.createArchive(
103                                         null, appClasses,
104                                         "--module-path", moduleDir.toString(),
105                                         "-m", MAIN_MODULE);
106         TestCommon.checkDump(output);
107         String prefix[] = {"-Djava.class.path=", "-Xlog:class+load=trace"};
108         prefix = TestCommon.concat(prefix, extra_runtime_args);
109 
110         // run with the archive with the --module-path the same as the one during
111         // dump time. The classes should be loaded from the archive.
112         TestCommon.runWithModules(prefix,
113                                   null, // --upgrade-module-path
114                                   moduleDir.toString(), // --module-path
115                                   MAIN_MODULE) // -m
116             .assertNormalExit(out -> {
117                 out.shouldContain("[class,load] com.greetings.Main source: shared objects file")
118                    .shouldContain("[class,load] org.astro.World source: shared objects file");
119             });
120 
121         // run with the archive with the --module-path different from the one during
122         // dump time. The classes should be loaded from the jar files.
123         TestCommon.runWithModules(prefix,
124                                   null, // --upgrade-module-path
125                                   moduleDir2.toString(), // --module-path
126                                   MAIN_MODULE) // -m
127             .assertNormalExit(out -> {
128                 out.shouldMatch(".class.load. com.greetings.Main source:.*com.greetings.jar")
129                    .shouldMatch(".class.load. org.astro.World source:.*org.astro.jar");
130             });
131 
132         // create an archive with modular jar files in both -cp and --module-path
133         String jars = subJar.toString() + System.getProperty("path.separator") +
134                       mainJar.toString();
135         output = TestCommon.createArchive( jars, appClasses,
136                                            "-Xlog:class+load=trace",
137                                            "--module-path", moduleDir.toString(),
138                                            "-m", MAIN_MODULE);
139         TestCommon.checkDump(output);
140 
141         // run with archive with the main class name specified before
142         // the module name with the -m option. Since the -m option was specified
143         // during dump time, the classes in the jar files after the -cp won't be
144         // archived. Therefore, the classes won't be loaded from the archive but
145         // will be loaded from the jar files.
146         TestCommon.run("-Xlog:class+load=trace",
147                        "-cp", jars,
148                        "--module-path", moduleDir.toString(),
149                        MAIN_CLASS, "-m", MAIN_MODULE)
150             .assertNormalExit(out -> {
151                 out.shouldMatch(".class.load. com.greetings.Main source:.*com.greetings.jar")
152                    .shouldMatch(".class.load. org.astro.World source:.*org.astro.jar");
153             });
154 
155         // similar to the above case but without the main class name. The classes
156         // should be loaded from the archive.
157         TestCommon.run("-Xlog:class+load=trace",
158                        "-cp", jars,
159                        "--module-path", moduleDir.toString(),
160                        "-m", MAIN_MODULE)
161             .assertNormalExit(
162               "[class,load] com.greetings.Main source: shared objects file",
163               "[class,load] org.astro.World source: shared objects file");
164 
165         // create an archive with two modular jars in the --module-path
166         output = TestCommon.createArchive(
167                                         null, appClasses,
168                                         "--module-path", jars,
169                                         "-m", MAIN_MODULE);
170         TestCommon.checkDump(output);
171 
172         // run with the above archive but with the modular jar containing the
173         // org.astro module in a different location.
174         // The org.astro.World class should be loaded from the jar.
175         // The Main class should still be loaded from the archive.
176         jars = destJar.toString() + System.getProperty("path.separator") +
177                       mainJar.toString();
178         TestCommon.runWithModules(prefix,
179                                   null, // --upgrade-module-path
180                                   jars, // --module-path
181                                   MAIN_MODULE) // -m
182             .assertNormalExit(out -> {
183                 out.shouldContain("[class,load] com.greetings.Main source: shared objects file")
184                    .shouldMatch(".class.load. org.astro.World source:.*org.astro.jar");
185             });
186     }
187 }
