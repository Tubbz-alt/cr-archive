24 /**
25  * @test CdsDifferentObjectAlignment
26  * @summary Testing CDS (class data sharing) using varying object alignment.
27  *          Using different object alignment for each dump/load pair.
28  *          This is a negative test; using  object alignment for loading that
29  *          is different from object alignment for creating a CDS file
30  *          should fail when loading.
31  * @requires vm.cds
32  * @requires vm.bits == 64
33  * @library /test/lib
34  * @bug 8025642
35  * @run driver CdsDifferentObjectAlignment
36  */
37 
38 import jdk.test.lib.cds.CDSTestUtils;
39 import jdk.test.lib.process.OutputAnalyzer;
40 import jdk.test.lib.Platform;
41 
42 public class CdsDifferentObjectAlignment {
43 
44     public static void main(String[] args) throws Exception {
45         createAndLoadSharedArchive(16, 64);
46         createAndLoadSharedArchive(64, 32);
47     }
48 
49     // Parameters are object alignment expressed in bytes
50     private static void
51     createAndLoadSharedArchive(int createAlignment, int loadAlignment)
52     throws Exception {
53         String createAlignmentArgument = "-XX:ObjectAlignmentInBytes=" +
54             createAlignment;
55         String loadAlignmentArgument = "-XX:ObjectAlignmentInBytes=" +
56             loadAlignment;
57         String expectedErrorMsg =
58             String.format(
59             "The shared archive file's ObjectAlignmentInBytes of %d " +
60             "does not equal the current ObjectAlignmentInBytes of %d",
61             createAlignment,
62             loadAlignment);
63 
64         CDSTestUtils.createArchiveAndCheck(createAlignmentArgument);
65 
66         OutputAnalyzer out = CDSTestUtils.runWithArchive(loadAlignmentArgument);
67         CDSTestUtils.checkExecExpectError(out, 1, expectedErrorMsg);
68     }
69 }
