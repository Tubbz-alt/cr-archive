25 /*
26  * @test
27  * @summary Basic test for shared strings
28  * @requires vm.cds.archived.java.heap
29  * @library /test/hotspot/jtreg/runtime/cds/appcds /test/lib
30  * @build HelloString
31  * @run driver SharedStringsBasic
32  */
33 import jdk.test.lib.process.OutputAnalyzer;
34 import jdk.test.lib.process.ProcessTools;
35 
36 // This test does not use SharedStringsUtils.dumpXXX()
37 // and SharedStringsUtils.runWithXXX() intentionally:
38 // - in order to demonstrate the basic use of the functionality
39 // - to provide sanity check and catch potential problems in the utils
40 public class SharedStringsBasic {
41     public static void main(String[] args) throws Exception {
42         SharedStringsUtils.run(args, SharedStringsBasic::test);
43     }
44 
45     public static void test(String[] args) throws Exception {
46         String vmOptionsPrefix[] = SharedStringsUtils.getChildVMOptionsPrefix();
47 
48         String appJar = JarBuilder.build("SharedStringsBasic", "HelloString");
49 
50         String sharedArchiveConfigFile =
51             TestCommon.getSourceFile("SharedStringsBasic.txt").toString();
52 
53         ProcessBuilder dumpPb = ProcessTools.createJavaProcessBuilder(true,
54           TestCommon.concat(vmOptionsPrefix,
55             "-cp", appJar,
56             "-XX:SharedArchiveConfigFile=" + sharedArchiveConfigFile,
57             "-XX:SharedArchiveFile=./SharedStringsBasic.jsa",
58             "-Xshare:dump",
59             "-Xlog:cds,cds+hashtables"));
60 
61         TestCommon.executeAndLog(dumpPb, "dump")
62             .shouldContain("Shared string table stats")
63             .shouldHaveExitValue(0);
64 
65         ProcessBuilder runPb = ProcessTools.createJavaProcessBuilder(true,
66           TestCommon.concat(vmOptionsPrefix,
67             "-cp", appJar,
68             "-XX:SharedArchiveFile=./SharedStringsBasic.jsa",
69             "-Xshare:auto",
70             "-showversion",
71             "HelloString"));
72 
73         TestCommon.executeAndLog(runPb, "run").shouldHaveExitValue(0);
74     }
75 }
