24 import sun.hotspot.WhiteBox;
25 
26 // This class is used by the test SharedStrings.java
27 // It should be launched in CDS mode
28 public class SharedStringsWb {
29     public static void main(String[] args) throws Exception {
30         WhiteBox wb = WhiteBox.getWhiteBox();
31 
32         if (wb.areSharedStringsIgnored()) {
33             System.out.println("Shared strings are ignored, assuming PASS");
34             return;
35         }
36 
37         // The string below is known to be added to CDS archive
38         String s = "<init>";
39         String internedS = s.intern();
40 
41         // Check that it's a valid string
42         if (s.getClass() != String.class || !(s instanceof String)) {
43             throw new RuntimeException("Shared string is not a valid String: FAIL");
44         }
45 
46         if (wb.isShared(internedS)) {
47             System.out.println("Found shared string, result: PASS");
48         } else {
49             throw new RuntimeException("String is not shared, result: FAIL");
50         }
51     }
52 }
