 25 /**
 26  * @test
 27  * @requires vm.cds
 28  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 29  * @run driver AddReads
 30  * @summary sanity test the --add-reads option
 31  */
 32 
 33 import java.io.File;
 34 import java.nio.file.Files;
 35 import java.nio.file.Path;
 36 import java.nio.file.Paths;
 37 
 38 import jdk.test.lib.process.OutputAnalyzer;
 39 import jdk.test.lib.Asserts;
 40 
 41 public class AddReads {
 42 
 43     private static final Path USER_DIR = Paths.get(System.getProperty("user.dir"));
 44 
 45     private static final String TEST_SRC = System.getProperty("test.src");
 46 
 47     private static final Path SRC_DIR = Paths.get(TEST_SRC, "src");
 48     private static final Path MODS_DIR = Paths.get("mods");
 49 
 50     // the module name of the test module
 51     private static final String MAIN_MODULE = "com.norequires";
 52     private static final String SUB_MODULE = "org.astro";
 53 
 54     // the module main class
 55     private static final String MAIN_CLASS = "com.norequires.Main";
 56     private static final String APP_CLASS = "org.astro.World";
 57 
 58     private static Path moduleDir = null;
 59     private static Path subJar = null;
 60     private static Path mainJar = null;
 61 
 62     public static void buildTestModule() throws Exception {
 63 
 64         // javac -d mods/$TESTMODULE src/$TESTMODULE/**
 65         JarBuilder.compileModule(SRC_DIR.resolve(SUB_MODULE),
 66                                        MODS_DIR.resolve(SUB_MODULE),
 67                                        null);
 68 
 69         Asserts.assertTrue(CompilerUtils
 70             .compile(SRC_DIR.resolve(MAIN_MODULE),
 71                      MODS_DIR.resolve(MAIN_MODULE),
 72                      "-cp", MODS_DIR.resolve(SUB_MODULE).toString(),
 73                      "--add-reads", "com.norequires=ALL-UNNAMED"));
 74 
 75         moduleDir = Files.createTempDirectory(USER_DIR, "mlib");
 76         subJar = moduleDir.resolve(SUB_MODULE + ".jar");
 77         String classes = MODS_DIR.resolve(SUB_MODULE).toString();
 78         JarBuilder.createModularJar(subJar.toString(), classes, null);
 79 
 80         mainJar = moduleDir.resolve(MAIN_MODULE + ".jar");
 81         classes = MODS_DIR.resolve(MAIN_MODULE).toString();
 82         JarBuilder.createModularJar(mainJar.toString(), classes, MAIN_CLASS);
 83     }
 84 
 85     public static void main(String... args) throws Exception {
 86         // compile the modules and create the modular jar files
 87         buildTestModule();
 88         String appClasses[] = {MAIN_CLASS, APP_CLASS};
 89         // create an archive with the classes in the modules built in the
 90         // previous step
 91         OutputAnalyzer output = TestCommon.createArchive(
 92                                         null, appClasses,
 93                                         "--module-path", moduleDir.toString(),
 94                                         "--add-modules", SUB_MODULE,
 95                                         "--add-reads", "com.norequires=org.astro",
 96                                         "-m", MAIN_MODULE);
 97         TestCommon.checkDump(output);
 98         String prefix[] = {"-Djava.class.path=", "-Xlog:class+load=trace",
 99                            "--add-modules", SUB_MODULE,
100                            "--add-reads", "com.norequires=org.astro"};
101 
102         // run the com.norequires module with the archive with the same args
103         // used during dump time.
104         // The classes should be loaded from the archive.
105         TestCommon.runWithModules(prefix,
106                                   null, // --upgrade-module-path
107                                   moduleDir.toString(), // --module-path
108                                   MAIN_MODULE) // -m
109             .assertNormalExit(out -> {
110                 out.shouldContain("[class,load] com.norequires.Main source: shared objects file")
111                    .shouldContain("[class,load] org.astro.World source: shared objects file");
112             });
113 
114         // create an archive with -cp pointing to the jar file containing the
115         // org.astro module and --module-path pointing to the main module
116         output = TestCommon.createArchive(
117                                         subJar.toString(), appClasses,
118                                         "--module-path", moduleDir.toString(),
119                                         "--add-modules", SUB_MODULE,
120                                         "--add-reads", "com.norequires=org.astro",
121                                         "-m", MAIN_MODULE);
122         TestCommon.checkDump(output);
123         // run the com.norequires module with the archive with the sub-module
124         // in the -cp and with -add-reads=com.norequires=ALL-UNNAMED
125         // The main class should be loaded from the archive.
126         // The org.astro.World should be loaded from the jar.
127         String prefix2[] = {"-cp", subJar.toString(), "-Xlog:class+load=trace",
128                            "--add-reads", "com.norequires=ALL-UNNAMED"};
129         TestCommon.runWithModules(prefix2,
130                                   null, // --upgrade-module-path
131                                   moduleDir.toString(), // --module-path
132                                   MAIN_MODULE) // -m
133             .assertNormalExit(out -> {
134                 out.shouldContain("[class,load] com.norequires.Main source: shared objects file")
135                    .shouldMatch(".class.load. org.astro.World source:.*org.astro.jar");
136             });
137 
138     }
139 }
