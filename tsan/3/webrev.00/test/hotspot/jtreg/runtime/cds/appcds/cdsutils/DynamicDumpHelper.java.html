25 package cdsutils;
26 
27 import java.io.BufferedReader;
28 import java.io.File;
29 import java.io.FileReader;
30 
31 /**
32  * This class is used to simulate -Xshare:dump with -XX:ArchiveClassesAtExit.
33  * It loads all classes specified in a classlist file. See patchJarForDynamicDump()
34  * in ../TestCommon.java for details.
35  */
36 public class DynamicDumpHelper {
37     public static void main(String args[]) throws Throwable {
38         File file = new File(args[0]);
39 
40         System.out.println("Loading classes to share...");
41         try (BufferedReader br = new BufferedReader(new FileReader(file))) {
42             String line;
43             while ((line = br.readLine()) != null) {
44                 //System.out.println("Loading class: " + line);
45                 line = line.replace('/', '.');
46                 try {
47                     Class.forName(line);
48                 } catch (java.lang.ClassNotFoundException ex) {
49                     try {
50                         Class.forName(line, true, null);
51                     } catch (java.lang.ClassNotFoundException cnfe) {
52                         System.out.println("Preload Warning: Cannot find " + line.replace('.', '/'));
53                     }
54                 } catch (Throwable t) {
55                     System.out.println("Error: failed to load \"" + line + "\": " + t);
56                 }
57             }
58         }
59         System.out.println("Loading classes to share: done.");
60     }
61 }
