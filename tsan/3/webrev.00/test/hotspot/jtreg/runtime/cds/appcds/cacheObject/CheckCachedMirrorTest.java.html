25 /*
26  * @test
27  * @summary Test archived mirror
28  * @requires vm.cds.archived.java.heap
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
31  * @build sun.hotspot.WhiteBox
32  * @compile CheckCachedMirrorApp.java
33  * @compile ../test-classes/Hello.java
34  * @run driver ClassFileInstaller -jar app.jar CheckCachedMirrorApp
35  * @run driver ClassFileInstaller -jar hello.jar Hello
36  * @run driver ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
37  * @run driver CheckCachedMirrorTest
38  */
39 
40 import jdk.test.lib.process.OutputAnalyzer;
41 import sun.hotspot.WhiteBox;
42 
43 public class CheckCachedMirrorTest {
44     public static void main(String[] args) throws Exception {
45         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
46         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
47         String appJar = ClassFileInstaller.getJarPath("app.jar");
48         String helloJarPath = ClassFileInstaller.getJarPath("hello.jar");
49 
50         String classlist[] = new String[] {
51             "CheckCachedMirrorApp",            // built-in app loader
52             "java/lang/Object id: 1",          // boot loader
53             "Hello id: 2 super: 1 source: " + helloJarPath // custom loader
54         };
55 
56         TestCommon.testDump(appJar, classlist, use_whitebox_jar);
57         OutputAnalyzer output = TestCommon.exec(appJar, use_whitebox_jar,
58                                                 "-XX:+UnlockDiagnosticVMOptions",
59                                                 "-XX:+WhiteBoxAPI",
60                                                 "-Xlog:cds=debug",
61                                                 "CheckCachedMirrorApp",
62                                                 helloJarPath);
63         TestCommon.checkExec(output);
64     }
65 }
