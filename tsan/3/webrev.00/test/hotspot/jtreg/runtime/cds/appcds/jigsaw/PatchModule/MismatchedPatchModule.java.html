25 /*
26  * @test
27  * @requires vm.cds
28  * @summary different settings of --patch-module at dump time and runtime are
29  *          acceptable. The class found in runtime --patch-module entry should
30  *          be used.
31  * @library ../..
32  * @library /test/hotspot/jtreg/testlibrary
33  * @library /test/lib
34  * @build PatchMain
35  * @run driver MismatchedPatchModule
36  */
37 
38 import jdk.test.lib.compiler.InMemoryJavaCompiler;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class MismatchedPatchModule {
42     private static String moduleJar;
43 
44     public static void main(String args[]) throws Throwable {
45 
46         // Create a class file in the module java.naming. This class file
47         // will be put in the javanaming.jar file.
48         String source = "package javax.naming.spi; "                +
49                         "public class NamingManager { "             +
50                         "    static { "                             +
51                         "        System.out.println(\"I pass!\"); " +
52                         "    } "                                    +
53                         "}";
54 
55         ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
56              InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
57              System.getProperty("test.classes"));
58 
59         // Build the jar file that will be used for the module "java.naming".
60         JarBuilder.build("javanaming", "javax/naming/spi/NamingManager");
61         moduleJar = TestCommon.getTestJar("javanaming.jar");
62 
63         // Case 1: --patch-module specified for dump time
64         System.out.println("Case 1: --patch-module specified for dump time and run time");
65         OutputAnalyzer output =
66             TestCommon.dump(null,
67                 TestCommon.list("javax/naming/spi/NamingManager"),
68                 "--patch-module=java.naming=" + moduleJar,
69                 "PatchMain", "javax.naming.spi.NamingManager");
70         output.shouldHaveExitValue(1)
71               .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module");
72 
73         // Case 2: --patch-module specified for run time but not for dump time
74         System.out.println("Case 2: --patch-module specified for run time but not for dump time");
75         String appJar = JarBuilder.build("PatchMain-app", "PatchMain");
76         output =
77             TestCommon.dump(appJar,
78                 TestCommon.list("javax/naming/spi/NamingManager"),
79                 "PatchMain", "javax.naming.spi.NamingManager");
80         TestCommon.checkDump(output);
81 
82         // javax.naming.spi.NamingManager is patched at runtime
83         TestCommon.run(
84             "-XX:+UnlockDiagnosticVMOptions",
85             "--patch-module=java.naming=" + moduleJar,
86             "--patch-module=java.naming2=" + moduleJar,
87             "-Xlog:class+path=info",
88             "PatchMain", "javax.naming.spi.NamingManager")
89             .assertSilentlyDisabledCDS(0, "I pass!");
90     }
91 }
