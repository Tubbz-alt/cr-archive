25 /*
26  * @test
27  * @summary At run time, it is OK to append new elements to the classpath that was used at dump time.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/Hello.java
31  * @compile test-classes/HelloMore.java
32  * @run driver AppendClasspath
33  */
34 
35 import java.io.File;
36 import java.nio.file.Files;
37 import java.nio.file.Paths;
38 import java.nio.file.StandardCopyOption;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class AppendClasspath {
42 
43   public static void main(String[] args) throws Exception {
44     String appJar = JarBuilder.getOrCreateHelloJar();
45     String appJar2 = JarBuilder.build("AppendClasspath_HelloMore", "HelloMore");
46 
47     // Dump an archive with a specified JAR file in -classpath
48     TestCommon.testDump(appJar, TestCommon.list("Hello"));
49 
50     // PASS: 1) runtime with classpath containing the one used in dump time
51     TestCommon.run(
52         "-cp", appJar + File.pathSeparator + appJar2,
53         "HelloMore")
54       .assertNormalExit();
55 
56     // PASS: 2) runtime has an non-existing jar in the -cp
57     String classDir = System.getProperty("test.classes");
58     String newFile = "non-exist.jar";
59     String nonExistPath = classDir + File.separator + newFile;
60     String classPath = appJar + File.pathSeparator + nonExistPath;
61     File nonExistJar = new File(classDir, newFile);
62     if (nonExistJar.exists()) {
63         nonExistJar.delete();
64     }
65     TestCommon.run(
66         "-cp", classPath,
67         "-Xlog:class+path=trace",
68         "Hello")
69       .assertNormalExit();
70 
71     final String errorMessage1 = "Unable to use shared archive";
72     final String errorMessage2 = "shared class paths mismatch";
73     // FAIL: 1) runtime with classpath different from the one used in dump time
74     // (runtime has an extra jar file prepended to the class path)
75     TestCommon.run(
76         "-Xlog:cds",
77         "-cp", appJar2 + File.pathSeparator + appJar,
78         "HelloMore")
79         .assertAbnormalExit(errorMessage1, errorMessage2);
80 
81     // FAIL: 2) runtime with classpath part of the one used in dump time
82     TestCommon.testDump(appJar + File.pathSeparator + appJar2,
83                                       TestCommon.list("Hello"));
84     TestCommon.run(
85         "-Xlog:cds",
86         "-cp", appJar2,
87         "Hello")
88         .assertAbnormalExit(errorMessage1, errorMessage2);
89 
90     // FAIL: 3) runtime with same set of jar files in the classpath but
91     // with different order
92     TestCommon.run(
93         "-Xlog:cds",
94         "-cp", appJar2 + File.pathSeparator + appJar,
95         "HelloMore")
96         .assertAbnormalExit(errorMessage1, errorMessage2);
97     }
98 }
