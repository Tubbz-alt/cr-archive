24 import sun.hotspot.WhiteBox;
25 
26 public class CondyHelloApp {
27     public static void main(String[] args) throws Throwable {
28         int n;
29         if ((n = CondyHello.doit()) != 12345) {
30             throw new java.lang.RuntimeException("Expected 12345 but got " + n);
31         } else {
32             System.out.println("n = " + n + " (as expected)");
33         }
34         WhiteBox wb = WhiteBox.getWhiteBox();
35         if (wb.isSharedClass(CondyHelloApp.class)) {
36             if (!wb.isSharedClass(CondyHello.class)) {
37                 throw new java.lang.RuntimeException("CondyHello should be in shared space but is not.");
38             } else {
39                 System.out.println("CondyHello.class is shared (as expected)");
40             }
41         }
42    }
43 }
