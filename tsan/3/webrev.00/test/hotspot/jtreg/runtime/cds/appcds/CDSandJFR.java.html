25 /**
26  * @test
27  * @summary Make sure CDS and JFR work together.
28  * @requires vm.hasJFR & vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes test-classes
30  * @build Hello GetFlightRecorder
31  * @run driver ClassFileInstaller -jar CDSandJFR.jar Hello GetFlightRecorder GetFlightRecorder$TestEvent GetFlightRecorder$SimpleEvent
32  * @run main/othervm/timeout=500 CDSandJFR
33  */
34 
35 import jdk.test.lib.BuildHelper;
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class CDSandJFR {
39     static String[] classes = {
40         "jdk/jfr/Event",
41         "jdk/jfr/events/FileReadEvent",
42         "jdk/jfr/events/FileWriteEvent",
43         "jdk/jfr/events/SocketReadEvent",
44         "jdk/jfr/events/SocketWriteEvent",
45         "jdk/jfr/events/ExceptionThrownEvent",
46         "jdk/jfr/events/ExceptionStatisticsEvent",
47         "jdk/jfr/events/ErrorThrownEvent",
48         "jdk/jfr/events/ActiveSettingEvent",
49         "jdk/jfr/events/ActiveRecordingEvent",
50         "Hello",
51         "GetFlightRecorder",
52         "GetFlightRecorder$TestEvent",
53     };
54 
55     public static void main(String[] args) throws Exception {
56         test(classes);
57     }
58 
59     static void test(String[] classes) throws Exception {
60         String appJar = ClassFileInstaller.getJarPath("CDSandJFR.jar");
61         OutputAnalyzer output;
62         output = TestCommon.testDump(appJar, TestCommon.list(classes));
63         TestCommon.checkDump(output, "Skipping jdk/jfr/Event: JFR event class");
64 
65         output = TestCommon.exec(appJar,
66                                  "-XX:StartFlightRecording=dumponexit=true",
67                                  "Hello");
68         TestCommon.checkExec(output, "Hello World");
69 
70         TestCommon.checkExec(TestCommon.exec(appJar,
71                                              "-XX:FlightRecorderOptions=retransform=true",
72                                              "GetFlightRecorder"));
73         TestCommon.checkExec(TestCommon.exec(appJar,
74                                              "-XX:FlightRecorderOptions=retransform=false",
75                                              "GetFlightRecorder"));
76 
77         // Test dumping with flight recorder enabled.
78         output = TestCommon.testDump(appJar, TestCommon.list(classes),
79                                      "-XX:StartFlightRecording=dumponexit=true");
80         TestCommon.checkDump(output, "warning: JFR will be disabled during CDS dumping");
81     }
82 }
