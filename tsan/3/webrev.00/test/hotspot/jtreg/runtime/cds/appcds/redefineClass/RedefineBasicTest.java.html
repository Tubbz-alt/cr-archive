25 /*
26  * @test
27  * @summary Run /serviceability/jvmti/RedefineClasses/RedefineRunningMethods in AppCDS mode to
28  *          make sure class redefinition works with CDS.
29  * @requires vm.cds
30  * @library /test/lib /test/hotspot/jtreg/serviceability/jvmti/RedefineClasses /test/hotspot/jtreg/runtime/cds/appcds
31  * @run driver RedefineClassHelper
32  * @build sun.hotspot.WhiteBox RedefineBasic
33  * @run driver RedefineBasicTest
34  */
35 
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class RedefineBasicTest {
39     public static String sharedClasses[] = {
40         "RedefineBasic",
41         "RedefineBasic_B",
42         "RedefineBasic$SubclassOfB",
43         "RedefineBasic$Subclass2OfB",
44         "RedefineClassHelper",
45         "jdk/test/lib/compiler/InMemoryJavaCompiler",
46         "jdk/test/lib/compiler/InMemoryJavaCompiler$FileManagerWrapper",
47         "jdk/test/lib/compiler/InMemoryJavaCompiler$FileManagerWrapper$1",
48         "jdk/test/lib/compiler/InMemoryJavaCompiler$MemoryJavaFileObject"
49     };
50 
51     public static void main(String[] args) throws Exception {
52         String wbJar =
53             ClassFileInstaller.writeJar("WhiteBox.jar", "sun.hotspot.WhiteBox");
54         String appJar =
55             ClassFileInstaller.writeJar("RedefineBasic.jar", sharedClasses);
56         String useWb = "-Xbootclasspath/a:" + wbJar;
57 
58         OutputAnalyzer output;
59         TestCommon.testDump(appJar, sharedClasses, useWb);
60 
61         // redefineagent.jar is created by executing "@run driver RedefineClassHelper"
62         // which should be called before executing RedefineBasicTest
63         output = TestCommon.exec(appJar, useWb,
64                                  "-XX:+UnlockDiagnosticVMOptions",
65                                  "-XX:+WhiteBoxAPI",
66                                  "-javaagent:redefineagent.jar",
67                                  "RedefineBasic");
68         TestCommon.checkExec(output);
69     }
70 }
