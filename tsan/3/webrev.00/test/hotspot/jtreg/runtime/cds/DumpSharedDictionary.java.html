24 /*
25  * @test
26  * @bug 8130072
27  * @summary Check that Shared Dictionary is printed out with jcmd
28  * @requires vm.cds
29  * @library /test/lib
30  * @run driver DumpSharedDictionary
31  */
32 
33 import jdk.test.lib.cds.CDSTestUtils;
34 import jdk.test.lib.process.ProcessTools;
35 import jdk.test.lib.process.OutputAnalyzer;
36 import jdk.test.lib.JDKToolFinder;
37 
38 public class DumpSharedDictionary {
39 
40     public static void main(String[] args) throws Exception {
41         if (args.length == 0) {
42             // Start this process
43             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
44                 "-XX:SharedArchiveFile=./DumpSharedDictionary.jsa",
45                 "-Xshare:dump");
46 
47             OutputAnalyzer out = CDSTestUtils.executeAndLog(pb, "dump");
48             out.shouldHaveExitValue(0);
49 
50             String testjdkPath = System.getProperty("test.jdk");
51 
52             pb = ProcessTools.createJavaProcessBuilder(
53                     "-XX:SharedArchiveFile=./DumpSharedDictionary.jsa",
54                     "-Dtest.jdk=" + testjdkPath,
55                     "-Xshare:on", "DumpSharedDictionary", "test");
56 
57             out = CDSTestUtils.executeAndLog(pb, "exec");
58             if (!CDSTestUtils.isUnableToMap(out)) {
59                 out.shouldHaveExitValue(0);
60             }
61         } else {
62             // Grab my own PID
63             String pid = Long.toString(ProcessTools.getProcessId());
64 
65             ProcessBuilder pb = new ProcessBuilder();
66             pb.command(new String[] {JDKToolFinder.getJDKTool("jcmd"), pid, "VM.systemdictionary"});
67             OutputAnalyzer output = CDSTestUtils.executeAndLog(pb, "jcmd-systemdictionary");
68             try {
69                 output.shouldContain("Shared Dictionary statistics:");
70                 output.shouldContain("Number of buckets");
71                 output.shouldContain("Number of entries");
72                 output.shouldContain("Maximum bucket size");
73             } catch (RuntimeException e) {
74                 output.shouldContain("Unknown diagnostic command");
75             }
76 
77             pb.command(new String[] {JDKToolFinder.getJDKTool("jcmd"), pid, "VM.systemdictionary", "-verbose"});
78             output = CDSTestUtils.executeAndLog(pb, "jcmd-systemdictionary-verbose");
79             try {
80                 output.shouldContain("Shared Dictionary");
81                 output.shouldContain("Dictionary for loader data: 0x");
82                 output.shouldContain("^java.lang.String");
83             } catch (RuntimeException e) {
84                 output.shouldContain("Unknown diagnostic command");
85             }
86         }
87     }
88 }
