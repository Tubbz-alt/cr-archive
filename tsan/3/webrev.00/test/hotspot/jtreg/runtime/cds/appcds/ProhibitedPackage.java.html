25 /*
26  * @test
27  * @summary AppCDS handling of prohibited package.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/ProhibitedHelper.java test-classes/Prohibited.jasm
31  * @run driver ProhibitedPackage
32  */
33 
34 import jdk.test.lib.cds.CDSOptions;
35 import jdk.test.lib.Platform;
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class ProhibitedPackage {
39 
40     public static void main(String[] args) throws Exception {
41         JarBuilder.build("prohibited_pkg", "java/lang/Prohibited", "ProhibitedHelper");
42 
43         String appJar = TestCommon.getTestJar("prohibited_pkg.jar");
44 
45         // Test support for customer loaders
46         if (Platform.areCustomLoadersSupportedForCDS() &&
47             !TestCommon.isDynamicArchive()) {
48             String classlist[] = new String[] {
49                 "java/lang/Object id: 1",
50                 "java/lang/Prohibited id: 2 super: 1 source: " + appJar
51             };
52 
53             // Make sure a class in a prohibited package for a custom loader
54             // will be ignored during dumping.
55             TestCommon.dump(appJar,  classlist, "-Xlog:cds")
56                 .shouldContain("Dumping")
57                 .shouldContain("[cds] Prohibited package for non-bootstrap classes: java/lang/Prohibited.class")
58                 .shouldHaveExitValue(0);
59         }
60 
61 
62         // Make sure a class in a prohibited package for a non-custom loader
63         // will be ignored during dumping.
64         TestCommon.dump(appJar,
65                         TestCommon.list("java/lang/Prohibited", "ProhibitedHelper"),
66                         "-Xlog:class+load")
67             .shouldContain("Dumping")
68             .shouldNotContain("[info][class,load] java.lang.Prohibited source: ")
69             .shouldHaveExitValue(0);
70 
71         // Try loading the class in a prohibited package with various -Xshare
72         // modes. The class shouldn't be loaded and appropriate exceptions
73         // are expected.
74 
75         OutputAnalyzer output;
76 
77         // -Xshare:on
78         TestCommon.run(
79             "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
80             "-cp", appJar, "ProhibitedHelper")
81           .assertNormalExit("Prohibited package name: java.lang");
82 
83         // -Xshare:auto
84         output = TestCommon.execAuto(
85             "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
86             "-cp", appJar, "ProhibitedHelper");
87         CDSOptions opts = (new CDSOptions()).setXShareMode("auto");
88         TestCommon.checkExec(output, opts, "Prohibited package name: java.lang");
89 
90         // -Xshare:off
91         output = TestCommon.execOff(
92             "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
93             "-cp", appJar, "ProhibitedHelper");
94         output.shouldContain("Prohibited package name: java.lang");
95     }
96 }
