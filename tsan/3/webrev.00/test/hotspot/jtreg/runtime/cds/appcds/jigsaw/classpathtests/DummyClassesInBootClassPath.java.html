25 /*
26  * @test
27  * @summary Ensure that classes found in jimage takes precedence over classes found in -Xbootclasspath/a.
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  * @compile ../../test-classes/DummyClassHelper.java
31  * @compile ../../test-classes/java/net/HttpCookie.jasm
32  * @compile ../../../javax/annotation/processing/FilerException.jasm
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  * @run driver DummyClassesInBootClassPath
36  */
37 
38 import java.io.File;
39 import java.util.List;
40 import java.util.ArrayList;
41 import jdk.test.lib.process.OutputAnalyzer;
42 
43 public class DummyClassesInBootClassPath {
44     private static final String METHOD_NAME = "thisClassIsDummy()";
45 
46     static void checkOutput(OutputAnalyzer output, String[] classNames) throws Exception {
47         for (int i = 0; i < classNames.length; i++) {
48             String cn = classNames[i].replace('/', '.');
49             TestCommon.checkExec(output,
50                 "java.lang.NoSuchMethodException: " + cn + "." +
51                 METHOD_NAME);
52             output.shouldNotContain(cn + ".class should be in shared space.");
53         }
54     }
55 
56     public static void main(String[] args) throws Exception {
57         String classNames[] = { "java/net/HttpCookie",
58                                 "javax/annotation/processing/FilerException"};
59         JarBuilder.build("dummyClasses", classNames[0], classNames[1]);
60 
61         String appJar = TestCommon.getTestJar("dummyClasses.jar");
62         OutputAnalyzer dumpOutput = TestCommon.dump(
63             appJar, classNames, "-Xbootclasspath/a:" + appJar);
64         TestCommon.checkDump(dumpOutput);
65 
66         List<String> argsList = new ArrayList<String>();
67         for (int i = 0; i < classNames.length; i++) {
68             argsList.add(classNames[i].replace('/', '.'));
69         }
70         String[] arguments = new String[argsList.size()];
71         arguments = argsList.toArray(arguments);
72         TestCommon.run(
73             "-Xbootclasspath/a:" + appJar,
74             "DummyClassHelper", arguments[0], arguments[1])
75           .assertNormalExit(output -> checkOutput(output, classNames));
76 
77         JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
78         String whiteBoxJar = TestCommon.getTestJar("WhiteBox.jar");
79         String bootClassPath = "-Xbootclasspath/a:" + appJar +
80             File.pathSeparator + whiteBoxJar;
81         dumpOutput = TestCommon.dump(
82             appJar, classNames, bootClassPath);
83         argsList.add("testWithWhiteBox");
84         arguments = new String[argsList.size()];
85         arguments = argsList.toArray(arguments);
86         String[] opts = {"-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
87             bootClassPath, "-Xlog:class+path=trace",
88             "DummyClassHelper", arguments[0], arguments[1], arguments[2]};
89         TestCommon.run(opts)
90           .assertNormalExit(output -> checkOutput(output, classNames));
91     }
92 }
