25 /*
26  * @test
27  * @summary Load app classes from CDS archive in parallel threads,
28  * use initial transformation (CFLH)
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  *     /test/hotspot/jtreg/runtime/cds/appcds/test-classes /test/hotspot/jtreg/runtime/cds/appcds/jvmti
31  *     /test/hotspot/jtreg/testlibrary/jvmti
32  * @requires vm.cds
33  * @requires !vm.graal.enabled
34  * @build TransformUtil TransformerAgent ParallelLoad
35  * @run driver ParallelLoadAndTransformTest
36  */
37 import java.util.List;
38 import java.util.stream.Collectors;
39 import java.util.stream.IntStream;
40 
41 public class ParallelLoadAndTransformTest {
42 
43     public static void main(String[] args) throws Exception {
44         String prop = "-Dappcds.parallel.transform.mode=cflh";
45         String appJar = ClassFileInstaller.writeJar("parallel_load.jar",
46                             getClassList(true));
47         String agentJar = prepareAgent();
48 
49         TestCommon.test(appJar, getClassList(false),
50                         "-javaagent:" + agentJar + "=ParallelClassTr.*",
51                         prop, "ParallelLoad");
52     }
53 
54 
55     private static String[] getClassList(boolean includeWatchdog) {
56         List<String> classList =
57             IntStream.range(0, ParallelClassesTransform.NUMBER_OF_CLASSES)
58             .mapToObj(i -> "ParallelClassTr" + i)
59             .collect(Collectors.toList());
60 
61         classList.add("ParallelLoad");
62         classList.add("ParallelLoadThread");
63         if (includeWatchdog)
64             classList.add("ParallelLoadWatchdog");
65 
66         return classList.toArray(new String[0]);
67     }
68 
69 
70     // Agent is the same for all test cases
71     private static String prepareAgent() throws Exception {
72         String agentClasses[] = {
73             "TransformerAgent",
74             "TransformerAgent$SimpleTransformer",
75             "TransformUtil"
76         };
77 
78         String manifest = "../../../../../testlibrary/jvmti/TransformerAgent.mf";
79 
80         return ClassFileInstaller.writeJar("TransformerAgent.jar",
81             ClassFileInstaller.Manifest.fromSourceFile(manifest),
82                                         agentClasses);
83     }
84 
85 }
