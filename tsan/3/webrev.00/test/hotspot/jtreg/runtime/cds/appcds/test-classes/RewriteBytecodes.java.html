25 import java.io.File;
26 import java.lang.invoke.MethodHandles;
27 import sun.hotspot.WhiteBox;
28 
29 public class RewriteBytecodes {
30   public static void main(String args[]) throws Throwable {
31     String from = "___xxx___";
32     String to   = "___yyy___";
33     File clsFile = new File(args[0]);
34     Class superClass = Util.defineModifiedClass(MethodHandles.lookup(), clsFile, from, to);
35 
36     Child child = new Child();
37 
38     if (child.getClass().getSuperclass() != superClass) {
39       throw new RuntimeException("Mismatched super class");
40     }
41     // Even if the Super class is not loaded from the CDS archive, make sure the Child class
42     // can still be loaded successfully, and properly inherits from the rewritten version
43     // of Super.
44     if (!child.toString().equals(to)) {
45       throw new RuntimeException("Wrong output, expected: " + to + ", but got: " + child.toString());
46     }
47 
48     WhiteBox wb = WhiteBox.getWhiteBox();
49     if (wb.isSharedClass(superClass)) {
50       throw new RuntimeException("wb.isSharedClass(superClass) should be false");
51     }
52     if (wb.isSharedClass(child.getClass())) {
53       throw new RuntimeException("wb.isSharedClass(child.getClass()) should be false");
54     }
55   }
56 }
