25 /*
26  * @test
27  * @summary Make sure classes intended for custom loaders cannot be loaded by BOOT/EXT/APP loaders
28  * @requires vm.cds
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
31  * @compile test-classes/UnintendedLoaders.java test-classes/CustomLoadee.java
32  * @build sun.hotspot.WhiteBox
33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
34  * @run driver UnintendedLoadersTest
35  */
36 
37 import jdk.test.lib.process.OutputAnalyzer;
38 import sun.hotspot.WhiteBox;
39 
40 public class UnintendedLoadersTest {
41     public static void main(String[] args) throws Exception {
42         String wbJar = JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
43         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
44 
45         String appJar = JarBuilder.build("UnintendedLoaders_app", "UnintendedLoaders");
46         String customJarPath = JarBuilder.build("UnintendedLoaders_custom", "CustomLoadee");
47 
48         // Dump the archive
49         String classlist[] = new String[] {
50             "UnintendedLoadersTest",
51             "java/lang/Object id: 1",
52 
53             // Without "loader:" keyword.
54             "CustomLoadee id: 2 super: 1 source: " + customJarPath,
55         };
56 
57         OutputAnalyzer output;
58         TestCommon.testDump(appJar, classlist,
59                             // command-line arguments ...
60                             use_whitebox_jar);
61 
62         output = TestCommon.exec(appJar,
63                                  // command-line arguments ...
64                                  use_whitebox_jar,
65                                  "-XX:+UnlockDiagnosticVMOptions",
66                                  "-XX:+WhiteBoxAPI",
67                                  "UnintendedLoaders");
68         TestCommon.checkExec(output);
69     }
70 }
