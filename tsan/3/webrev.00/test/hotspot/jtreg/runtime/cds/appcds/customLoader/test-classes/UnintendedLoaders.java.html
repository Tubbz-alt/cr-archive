25 import sun.hotspot.WhiteBox;
26 
27 public class UnintendedLoaders {
28     public static void main(String[] args) throws Exception {
29         ClassLoader loaders[] = new ClassLoader[2];
30         loaders[0] = UnintendedLoaders.class.getClassLoader(); // app loader
31         loaders[1] = loaders[0].getParent();                   // platform loader
32 
33         String[] names = {
34             "CustomLoadee",
35         };
36 
37 
38         for (int i=0; i<3; i++) {
39             for (String s : names) {
40                 try {
41                     if (i <= 1) {
42                         System.out.println(loaders[i].loadClass(s));
43                     } else {
44                         System.out.println(Class.forName(s));
45                     }
46                 } catch (ClassNotFoundException e) {
47                     System.out.println("Expected exception:" + e);
48                     continue;
49                 }
50                 throw new RuntimeException("The class \"" + s + "\" should not be resolved by the application or platform class loader");
51             }
52         }
53     }
54 }
