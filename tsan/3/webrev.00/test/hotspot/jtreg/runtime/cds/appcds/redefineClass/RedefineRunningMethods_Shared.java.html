25 /*
26  * @test
27  * @summary Run /serviceability/jvmti/RedefineClasses/RedefineRunningMethods in AppCDS mode to
28  *          make sure class redefinition works with CDS.
29  * @requires vm.cds
30  * @library /test/lib /test/hotspot/jtreg/serviceability/jvmti/RedefineClasses /test/hotspot/jtreg/runtime/cds/appcds
31  * @run driver RedefineClassHelper
32  * @build sun.hotspot.WhiteBox RedefineRunningMethods_SharedHelper
33  * @run driver RedefineRunningMethods_Shared
34  */
35 
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class RedefineRunningMethods_Shared {
39     public static String shared_classes[] = {
40         "RedefineRunningMethods_Shared",
41         "RedefineRunningMethods_SharedHelper",
42         "RedefineRunningMethods",
43         "RedefineRunningMethods$1",
44         "RedefineRunningMethods$2",
45         "RedefineRunningMethods$3",
46         "RedefineRunningMethods_B",
47         "RedefineClassHelper",
48         "jdk/test/lib/compiler/InMemoryJavaCompiler",
49         "jdk/test/lib/compiler/InMemoryJavaCompiler$FileManagerWrapper",
50         "jdk/test/lib/compiler/InMemoryJavaCompiler$FileManagerWrapper$1",
51         "jdk/test/lib/compiler/InMemoryJavaCompiler$MemoryJavaFileObject"
52     };
53 
54     public static void main(String[] args) throws Exception {
55         String wbJar = ClassFileInstaller.writeJar("WhiteBox.jar", "sun.hotspot.WhiteBox");
56         String appJar = ClassFileInstaller.writeJar("RedefineRunningMethods_Shared.jar", shared_classes);
57         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
58 
59         OutputAnalyzer output;
60         TestCommon.testDump(appJar, shared_classes,
61                             // command-line arguments ...
62                             use_whitebox_jar);
63 
64         // RedefineRunningMethods.java contained this:
65         // @run main/othervm -javaagent:redefineagent.jar -Xlog:redefine+class+iklass+add=trace,redefine+class+iklass+purge=trace RedefineRunningMethods
66         output = TestCommon.exec(appJar,
67                                  // command-line arguments ...
68                                  use_whitebox_jar,
69                                  "-XX:+UnlockDiagnosticVMOptions",
70                                  "-XX:+WhiteBoxAPI",
71                                  // These arguments are expected by RedefineRunningMethods
72                                  "-javaagent:redefineagent.jar",
73                                  "-Xlog:redefine+class+iklass+add=trace,redefine+class+iklass+purge=trace",
74                                  "RedefineRunningMethods_SharedHelper");
75         TestCommon.checkExec(output);
76     }
77 }
