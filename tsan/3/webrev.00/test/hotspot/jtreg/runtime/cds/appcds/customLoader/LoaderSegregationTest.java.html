 25 /*
 26  * @test
 27  * @summary Check that during dumping, the classes for BOOT/EXT/APP loaders are segregated from the
 28  *          custom loader classes.
 29  * @requires vm.cds
 30  * @requires vm.cds.custom.loaders
 31  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 32  * @compile test-classes/LoaderSegregation.java
 33  *          test-classes/CustomLoadee.java test-classes/CustomLoadee2.java
 34  *          test-classes/CustomInterface2_ia.java test-classes/CustomInterface2_ib.java
 35  *          test-classes/CustomLoadee3.java test-classes/CustomLoadee3Child.java
 36  *          test-classes/OnlyBuiltin.java
 37  *          test-classes/OnlyUnregistered.java
 38  *          ../test-classes/Util.java
 39  * @build sun.hotspot.WhiteBox
 40  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 41  * @run driver LoaderSegregationTest
 42  */
 43 
 44 import jdk.test.lib.process.OutputAnalyzer;
 45 import sun.hotspot.WhiteBox;
 46 
 47 /**
 48  * See "Handling of the classes in the AppCDS archive" at the top of
 49  * systemDicrionatyShared.hpp.
 50  *
 51  * This test ensure that the 2 types of archived classes (BUILTIN and UNREGISTERED)
 52  * are segregated at both dump-time and run time:
 53  *
 54  * [A] An archived BUILTIN class cannot be a subclass of a non-BUILTIN class.
 55  * [B] An archived BUILTIN class cannot implement a non-BUILTIN interface.
 56  * [C] BUILTIN and UNREGISTERED classes can be loaded only by their corresponding
 57  *     type of loaders.
 58  *
 59  */
 60 public class LoaderSegregationTest {
 61     public static void main(String[] args) throws Exception {
 62         String wbJar = JarBuilder.build(true, "WhiteBox", "sun/hotspot/WhiteBox");
 63         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
 64 
 65         String appJar = JarBuilder.build("LoaderSegregation_app", "LoaderSegregation", "LoaderSegregation$1",
 66                                          "CustomLoadee", "CustomLoadee2", "CustomLoadee3Child", "CustomInterface2_ia",
 67                                          "OnlyBuiltin", "Util");
 68 
 69         String app2Jar = JarBuilder.build("LoaderSegregation_app2", "CustomLoadee3", "CustomInterface2_ib");
 70 
 71         String customJarPath = JarBuilder.build("LoaderSegregation_custom", "CustomLoadee",
 72                                                 "CustomLoadee2", "CustomInterface2_ia", "CustomInterface2_ib",
 73                                                 "CustomLoadee3", "CustomLoadee3Child",
 74                                                 "OnlyBuiltin", "OnlyUnregistered");
 75 
 76         // Dump the archive
 77         String classlist[] = new String[] {
 78             "LoaderSegregation",
 79             "java/lang/Object id: 1",
 80 
 81             // These are the UNREGISTERED classes: they have "source:"
 82             // but they don't have "loader:".
 83             "CustomLoadee id: 2 super: 1 source: " + customJarPath,
 84 
 85             "CustomInterface2_ia id: 3 super: 1 source: " + customJarPath,
 86             "CustomInterface2_ib id: 4 super: 1 source: " + customJarPath,
 87             "CustomLoadee2 id: 5 super: 1 interfaces: 3 4 source: " + customJarPath,
 88 
 89             "CustomLoadee3 id: 6 super: 1 source: " + customJarPath,
 90             "CustomLoadee3Child id: 7 super: 6 source: " + customJarPath,
 91 
 92             // At dump time, the following BUILTIN classes are loaded after the UNREGISTERED
 93             // classes from above. However, at dump time, they cannot use the UNREGISTERED classes are their
 94             // super or interface.
 95             "CustomLoadee",          // can be loaded at dump time
 96             "CustomLoadee2",         // cannot be loaded at dump time (interface missing)
 97             "CustomLoadee3Child",    // cannot be loaded at dump time (super missing)
 98 
 99             // Check that BUILTIN and UNREGISTERED classes can be loaded only by their
100             // corresponding type of loaders.
101             "OnlyBuiltin",
102             "OnlyUnregistered id: 9 super: 1 source: " + customJarPath,
103         };
104 
105         OutputAnalyzer output;
106         TestCommon.testDump(appJar, classlist,
107                             // command-line arguments ...
108                             use_whitebox_jar);
109 
110         output = TestCommon.exec(TestCommon.concatPaths(appJar, app2Jar),
111                                  // command-line arguments ...
112                                  use_whitebox_jar,
113                                  "-XX:+UnlockDiagnosticVMOptions",
114                                  "-XX:+WhiteBoxAPI",
115                                  "LoaderSegregation", customJarPath);
116         TestCommon.checkExec(output);
117     }
118 }
