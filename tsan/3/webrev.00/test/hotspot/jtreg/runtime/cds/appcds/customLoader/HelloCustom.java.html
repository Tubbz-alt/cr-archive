25 /*
26  * @test
27  * @summary Hello World test for AppCDS custom loader support
28  * @requires vm.cds
29  * @requires vm.cds.custom.loaders
30  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /runtime/testlibrary
31  * @compile test-classes/HelloUnload.java test-classes/CustomLoadee.java
32  * @build sun.hotspot.WhiteBox ClassUnloadCommon
33  * @run driver ClassFileInstaller -jar hello.jar HelloUnload ClassUnloadCommon ClassUnloadCommon$1 ClassUnloadCommon$TestFailure
34  * @run driver ClassFileInstaller -jar hello_custom.jar CustomLoadee
35  * @run driver ClassFileInstaller -jar WhiteBox.jar sun.hotspot.WhiteBox
36  * @run driver HelloCustom
37  */
38 
39 import jdk.test.lib.process.OutputAnalyzer;
40 import sun.hotspot.WhiteBox;
41 
42 public class HelloCustom {
43     public static void main(String[] args) throws Exception {
44         run();
45     }
46     public static void run(String... extra_runtime_args) throws Exception {
47         String wbJar = ClassFileInstaller.getJarPath("WhiteBox.jar");
48         String use_whitebox_jar = "-Xbootclasspath/a:" + wbJar;
49 
50         String appJar = ClassFileInstaller.getJarPath("hello.jar");
51         String customJarPath = ClassFileInstaller.getJarPath("hello_custom.jar");
52 
53         // Dump the archive
54         String classlist[] = new String[] {
55             "HelloUnload",
56             "java/lang/Object id: 1",
57             "CustomLoadee id: 2 super: 1 source: " + customJarPath
58         };
59 
60         OutputAnalyzer output;
61         TestCommon.testDump(appJar, classlist,
62                             // command-line arguments ...
63                             use_whitebox_jar);
64 
65         output = TestCommon.exec(appJar,
66                                  TestCommon.concat(extra_runtime_args,
67                                      // command-line arguments ...
68                                      use_whitebox_jar,
69                                      "-XX:+UnlockDiagnosticVMOptions",
70                                      "-XX:+WhiteBoxAPI",
71                                      "HelloUnload", customJarPath, "true", "true"));
72         TestCommon.checkExec(output);
73     }
74 }
