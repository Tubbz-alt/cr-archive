25 import sun.hotspot.WhiteBox;
26 
27 public class GCSharedStringsDuringDumpWb {
28     public static void main(String[] args) throws Exception {
29         WhiteBox wb = WhiteBox.getWhiteBox();
30         String s = "shared_test_string_unique_14325";
31         s = s.intern();
32         CheckString(wb, s);
33         for (int i=0; i<100000; i++) {
34             s = "generated_string " + i;
35             s = s.intern();
36             CheckString(wb, s);
37         }
38     }
39 
40     public static void CheckString(WhiteBox wb, String s) {
41         if (!wb.areSharedStringsIgnored() && !wb.isShared(s)) {
42             throw new RuntimeException("String is not shared.");
43         }
44     }
45 }
