25 /*
26  * @test
27  * @summary Test locking on shared strings
28  * @requires vm.cds.archived.java.heap
29  * @library /test/hotspot/jtreg/runtime/cds/appcds /test/lib
30  * @compile LockStringTest.java LockStringValueTest.java
31  * @build sun.hotspot.WhiteBox
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  * @run driver LockSharedStrings
34  */
35 
36 public class LockSharedStrings {
37     public static void main(String[] args) throws Exception {
38         SharedStringsUtils.run(args, LockSharedStrings::test);
39     }
40 
41     public static void test(String[] args) throws Exception {
42         SharedStringsUtils.buildJarAndWhiteBox("LockStringTest", "LockStringValueTest");
43 
44         SharedStringsUtils.dumpWithWhiteBox(
45             TestCommon.list("LockStringTest", "LockStringValueTest"),
46             "ExtraSharedInput.txt", "-Xlog:cds,cds+hashtables");
47 
48         String[] extraMatch = new String[] {"LockStringTest: PASS"};
49         SharedStringsUtils.runWithArchiveAndWhiteBox(extraMatch, "LockStringTest");
50 
51         extraMatch = new String[] {"LockStringValueTest: PASS"};
52         SharedStringsUtils.runWithArchiveAndWhiteBox(extraMatch, "LockStringValueTest",
53             "--add-opens=java.base/java.lang=ALL-UNNAMED");
54     }
55 }
