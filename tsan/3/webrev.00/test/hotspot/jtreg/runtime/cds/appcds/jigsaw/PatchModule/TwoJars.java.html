25 /*
26  * @test
27  * @requires vm.cds
28  * @summary a patched class found in --patch-module should be used at runtime
29  * @library ../..
30  * @library /test/hotspot/jtreg/testlibrary
31  * @library /test/lib
32  * @build PatchMain
33  * @run driver TwoJars
34  */
35 
36 import java.io.File;
37 import jdk.test.lib.compiler.InMemoryJavaCompiler;
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 public class TwoJars {
41     private static String moduleJar;
42     private static String moduleJar2;
43 
44     public static void main(String args[]) throws Throwable {
45 
46         // Create a class file in the module java.naming. This class file
47         // will be put in the javanaming.jar file.
48         String source = "package javax.naming.spi; "                +
49                         "public class NamingManager { "             +
50                         "    static { "                             +
51                         "        System.out.println(\"I pass!\"); " +
52                         "    } "                                    +
53                         "}";
54 
55         // Create a class file in the module java.naming. This class file
56         // will be put in the javanaming2.jar file.
57         String source2 = "package javax.naming.spi; "                +
58                         "public class DirectoryManager { "             +
59                         "    static { "                             +
60                         "        System.out.println(\"I fail!\"); " +
61                         "    } "                                    +
62                         "}";
63 
64         ClassFileInstaller.writeClassToDisk("javax/naming/spi/NamingManager",
65              InMemoryJavaCompiler.compile("javax.naming.spi.NamingManager", source, "--patch-module=java.naming"),
66              System.getProperty("test.classes"));
67 
68         // Build the jar file that will be used for the module "java.naming".
69         JarBuilder.build("javanaming", "javax/naming/spi/NamingManager");
70         moduleJar = TestCommon.getTestJar("javanaming.jar");
71 
72         ClassFileInstaller.writeClassToDisk("javax/naming/spi/DirectoryManager",
73              InMemoryJavaCompiler.compile("javax.naming.spi.DirectoryManager", source2, "--patch-module=java.naming"),
74              System.getProperty("test.classes"));
75 
76         // Build the jar file that will be used for the module "java.naming".
77         JarBuilder.build("javanaming2", "javax/naming/spi/DirectoryManager");
78         moduleJar2 = TestCommon.getTestJar("javanaming2.jar");
79 
80         System.out.println("Test dumping with --patch-module");
81         OutputAnalyzer output =
82             TestCommon.dump(null,
83                 TestCommon.list("javax/naming/spi/NamingManager"),
84                 "--patch-module=java.naming=" + moduleJar2 + File.pathSeparator + moduleJar,
85                 "-Xlog:class+load",
86                 "-Xlog:class+path=info",
87                 "PatchMain", "javax.naming.spi.NamingManager");
88         output.shouldHaveExitValue(1)
89               .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module");
90 
91         TestCommon.run(
92             "-XX:+UnlockDiagnosticVMOptions",
93             "--patch-module=java.naming=" + moduleJar2 + File.pathSeparator + moduleJar,
94             "-Xlog:class+load",
95             "-Xlog:class+path=info",
96             "PatchMain", "javax.naming.spi.NamingManager")
97             .assertSilentlyDisabledCDS(0, "I pass!");
98     }
99 }
