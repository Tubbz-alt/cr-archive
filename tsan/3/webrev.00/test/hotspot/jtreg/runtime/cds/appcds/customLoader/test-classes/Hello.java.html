25 import java.io.*;
26 import java.net.*;
27 import sun.hotspot.WhiteBox;
28 
29 public class Hello {
30     public static void main(String args[]) throws Exception {
31         String path = args[0];
32         URL url = new File(path).toURI().toURL();
33         URL[] urls = new URL[] {url};
34         System.out.println(path);
35         System.out.println(url);
36 
37         URLClassLoader urlClassLoader = new URLClassLoader(urls);
38         Class c = urlClassLoader.loadClass("CustomLoadee");
39         System.out.println(c);
40         System.out.println(c.getClassLoader());
41 
42         // [1] Check that CustomLoadee is defined by the correct loader
43         if (c.getClassLoader() != urlClassLoader) {
44             throw new RuntimeException("c.getClassLoader() == " + c.getClassLoader() +
45                                        ", expected == " + urlClassLoader);
46         }
47 
48         // [2] Check that CustomLoadee is loaded from shared archive.
49         WhiteBox wb = WhiteBox.getWhiteBox();
50         if (wb.isSharedClass(Hello.class)) {
51             if (!wb.isSharedClass(c)) {
52                 throw new RuntimeException("wb.isSharedClass(c) should be true");
53             }
54         }
55     }
56 }
