 25 /*
 26  * @test
 27  * @requires vm.cds
 28  * @summary the class in the -cp is a subclass of the class in --patch-module. The
 29  *          patched class should be used at runtime.
 30  * @library ../..
 31  * @library /test/hotspot/jtreg/testlibrary
 32  * @library /test/lib
 33  * @build PatchMain
 34  * @run driver SubClassOfPatchedClass
 35  */
 36 
 37 import java.io.File;
 38 import jdk.test.lib.compiler.InMemoryJavaCompiler;
 39 import jdk.test.lib.process.OutputAnalyzer;
 40 
 41 public class SubClassOfPatchedClass {
 42     private static String moduleJar;
 43     private static String appJar;
 44 
 45     public static void main(String args[]) throws Throwable {
 46 
 47         // Create a class file in the module java.naming. This class file
 48         // will be put in the javanaming.jar file.
 49         String source = "package javax.naming; "                +
 50                         "public class Reference { "             +
 51                         "    static { "                             +
 52                         "        System.out.println(\"I pass!\"); " +
 53                         "    } "                                    +
 54                         "}";
 55 
 56         String classDir = System.getProperty("test.classes");
 57 
 58         ClassFileInstaller.writeClassToDisk("javax/naming/Reference",
 59              InMemoryJavaCompiler.compile("javax.naming.Reference", source, "--patch-module=java.naming"),
 60              classDir);
 61 
 62         // Build the jar file that will be used for the module "java.naming".
 63         JarBuilder.build("javanaming", "javax/naming/Reference");
 64         moduleJar = TestCommon.getTestJar("javanaming.jar");
 65 
 66         String source2 = "package mypackage; "                +
 67                         "public class MyReference extends javax.naming.Reference { "             +
 68                         "    static { "                             +
 69                         "        System.out.println(\"MyReference!\"); " +
 70                         "    } "                                    +
 71                         "    public MyReference(String mystring) { " +
 72                         "        super(mystring); " +
 73                         "    } " +
 74                         "}";
 75         ClassFileInstaller.writeClassToDisk("mypackage/MyReference",
 76              InMemoryJavaCompiler.compile("mypackage.MyReference", source2),
 77              classDir);
 78 
 79         JarBuilder.build("myjavanaming", "mypackage/MyReference");
 80         appJar = TestCommon.getTestJar("myjavanaming.jar");
 81 
 82         System.out.println("Test dumping with --patch-module");
 83         OutputAnalyzer output =
 84             TestCommon.dump(appJar,
 85                 TestCommon.list("javax/naming/Reference", "mypackage/MyReference"),
 86                 "--patch-module=java.naming=" + moduleJar,
 87                 "-Xlog:class+load",
 88                 "PatchMain", "javax.naming.Reference", "mypackage.MyReference");
 89         output.shouldHaveExitValue(1)
 90               .shouldContain("Cannot use the following option when dumping the shared archive: --patch-module");
 91 
 92         String classPath = appJar + File.pathSeparator + classDir;
 93         System.out.println("classPath: " + classPath);
 94         TestCommon.run(
 95             "-XX:+UnlockDiagnosticVMOptions",
 96             "-cp", classPath,
 97             "--patch-module=java.naming=" + moduleJar,
 98             "-Xlog:class+load",
 99             "PatchMain", "javax.naming.Reference", "mypackage.MyReference")
100             .assertSilentlyDisabledCDS(0, "MyReference source: file:", "I pass!");
101     }
102 }
