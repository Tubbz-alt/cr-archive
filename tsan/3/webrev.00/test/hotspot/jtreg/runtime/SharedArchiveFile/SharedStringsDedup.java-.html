26  * @summary Test -Xshare:auto with shared strings and -XX:+UseStringDeduplication
27  * @requires vm.cds.archived.java.heap
28  * @library /test/lib

34 import jdk.test.lib.cds.CDSTestUtils;
35 import jdk.test.lib.process.OutputAnalyzer;
36 import java.io.File;
37 
38 // The main purpose is to test the interaction between shared strings
39 // and -XX:+UseStringDeduplication. We run in -Xshare:auto mode so
40 // we don't need to worry about CDS archive mapping failure (which
41 // doesn't happen often so it won't impact coverage).
42 public class SharedStringsDedup {
43     public static void main(String[] args) throws Exception {
44         OutputAnalyzer out =
45             CDSTestUtils.createArchive();
46         CDSTestUtils.checkDump(out, "Shared string table stats");
47         CDSTestUtils.runWithArchiveAndCheck("-XX:+UseStringDeduplication");
48     }
49 }
