24 /**
25  * @test
26  * @summary Check to make sure that shared strings in the bootstrap CDS archive
27  *          are actually shared
28  * @requires vm.cds.archived.java.heap
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  * @build SharedStringsWb sun.hotspot.WhiteBox
33  * @run driver ClassFileInstaller -jar whitebox.jar sun.hotspot.WhiteBox
34  * @run driver SharedStrings
35  */
36 
37 import jdk.test.lib.cds.CDSTestUtils;
38 import jdk.test.lib.process.ProcessTools;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class SharedStrings {
42     public static void main(String[] args) throws Exception {
43         // Note: This is a basic sanity test for Shared Strings feature.
44         // This also serves as a reference on how to use this feature,
45         // hence the command lines are spelled out instead of using the
46         // test utils methods.
47         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
48             "-XX:+UnlockDiagnosticVMOptions",
49             "-XX:SharedArchiveFile=./SharedStrings.jsa",
50             "-Xlog:cds,cds+hashtables",
51             // Needed for bootclasspath match, for CDS to work with WhiteBox API
52             "-Xbootclasspath/a:" + ClassFileInstaller.getJarPath("whitebox.jar"),
53             "-Xshare:dump");
54 
55         OutputAnalyzer out = CDSTestUtils.executeAndLog(pb, "dump");
56         CDSTestUtils.checkDump(out, "Shared string table stats");
57 
58 
59         pb = ProcessTools.createJavaProcessBuilder(
60                 "-XX:+UnlockDiagnosticVMOptions",
61                 "-XX:SharedArchiveFile=./SharedStrings.jsa",
62                 // needed for access to white box test API
63                 "-Xbootclasspath/a:" + ClassFileInstaller.getJarPath("whitebox.jar"),
64                 "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
65                 "-Xshare:on", "-showversion", "SharedStringsWb");
66 
67         out = CDSTestUtils.executeAndLog(pb, "exec");
68         CDSTestUtils.checkExec(out);
69     }
70 }
