24 /**
25  * @test Default CDS archive
26  * @summary Sharing should be enabled by default on supported platform/binaries.
27  *          No -Xshare:dump is needed. No -Xshare:auto or -Xshare:on in needed.
28  *          Verify a set of well-known shared classes.
29  * @requires vm.cds
30  * @library /test/lib
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
36  *      -XX:+WhiteBoxAPI CheckSharingWithDefaultArchive -showversion
37  */
38 import jdk.test.lib.Platform;
39 import jtreg.SkippedException;
40 import sun.hotspot.WhiteBox;
41 
42 public class CheckSharingWithDefaultArchive {
43     public static void main(String[] args) throws Exception {
44         if (!Platform.isDefaultCDSArchiveSupported()) {
45             throw new SkippedException("Supported platform");
46         }
47 
48         WhiteBox wb = WhiteBox.getWhiteBox();
49         String classes[] = {"java.lang.Object",
50                             "java.lang.String",
51                             "java.lang.Class"};
52         // If maping fails, sharing is disabled
53         if (wb.isSharingEnabled()) {
54             for (int i = 0; i < classes.length; i++) {
55                 Class c = Class.forName(classes[i]);
56                 if (wb.isSharedClass(c)) {
57                     System.out.println(classes[i] + " is shared.");
58                 } else {
59                     throw new RuntimeException(classes[i] + " is not shared");
60                 }
61             }
62         } else {
63            throw new SkippedException("Sharing is not enabled.");
64         }
65     }
66 }
