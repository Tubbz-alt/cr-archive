24 #include <jni.h>
25 
26 jmethodID mid;
27 jclass cls;
28 static int count = 0;
29 
30 JNIEXPORT jint JNICALL
31 Java_RedefineDeleteJmethod_jniCallDeleteMe(JNIEnv* env, jobject obj) {
32 
33     if (count == 0) {
34       count++;
35       cls = (*env)->FindClass(env, "B");
36       if (NULL == cls) {
37           (*env)->FatalError(env, "could not find class");
38       }
39 
40       mid = (*env)->GetStaticMethodID(env, cls, "deleteMe", "()I");
41       if (NULL == mid) {
42           (*env)->FatalError(env, "could not find method");
43       }
44     }
45 
46     return (*env)->CallStaticIntMethod(env, cls, mid);
47 }
