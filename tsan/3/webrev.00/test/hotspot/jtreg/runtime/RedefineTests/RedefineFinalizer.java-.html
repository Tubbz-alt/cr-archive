37 /*
38  * Regression test for hitting:
39  *
40  * assert(f == k->has_finalizer()) failed: inconsistent has_finalizer
41  *
42  * when redefining finalizer method
43  */
44 
45 
46 // package access top-level class to avoid problem with RedefineClassHelper
47 // and nested types.
48 class RedefineFinalizer_B {
49     protected void finalize() {
50         // should be empty
51     }
52 }
53 
54 public class RedefineFinalizer {
55 
56     public static String newB =
57                 "class RedefineFinalizer_B {" +
58                 "   protected void finalize() { " +
59                 "       System.out.println(\"Finalizer called\");" +
60                 "   }" +
61                 "}";
62 
63     public static void main(String[] args) throws Exception {
64         RedefineClassHelper.redefineClass(RedefineFinalizer_B.class, newB);
65 
66         A a = new A();
67     }
68 
69     static class A extends RedefineFinalizer_B {
70     }
71 }
