24 /*
25  * @test WhiteBox
26  * @bug 8011675
27  * @summary verify that whitebox can be used even if not all functions are declared in java-part
28  * @author igor.ignatyev@oracle.com
29  * @modules java.base/jdk.internal.misc
30  * @library /test/lib
31  * @compile WhiteBox.java
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:-CheckIntrinsics sun.hotspot.WhiteBox
34  */
35 
36 package sun.hotspot;
37 
38 public class WhiteBox {
39     @SuppressWarnings("serial")
40     public static class WhiteBoxPermission extends java.security.BasicPermission {
41         // ClassFileInstaller is hard-coded to copy WhiteBox$WhiteBoxPermission, so let's
42         // make a fake one here as well.
43         public WhiteBoxPermission(String s) {
44             super(s);
45         }
46     }
47 
48     private static native void registerNatives();
49     static { registerNatives(); }
50     public native int notExistedMethod();
51     public native int getHeapOopSize();
52     public static void main(String[] args) {
53         WhiteBox wb = new WhiteBox();
54         if (wb.getHeapOopSize() < 0) {
55             throw new Error("wb.getHeapOopSize() < 0");
56         }
57         boolean catched = false;
58         try {
59             wb.notExistedMethod();
60         } catch (UnsatisfiedLinkError e) {
61             catched = true;
62         }
63         if (!catched) {
64             throw new Error("wb.notExistedMethod() was invoked");
65         }
66     }
67 }
