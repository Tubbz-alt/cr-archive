 24 package sun.hotspot.tools.ctw;
 25 
 26 import java.io.IOException;
 27 import java.io.PrintStream;
 28 import java.lang.management.ManagementFactory;
 29 
 30 import java.nio.file.Files;
 31 import java.nio.file.Paths;
 32 
 33 import java.util.Arrays;
 34 import java.util.List;
 35 import java.util.concurrent.*;
 36 
 37 public class CompileTheWorld {
 38     // in case when a static constructor changes System::out and System::err
 39     // we hold these values of output streams
 40     static PrintStream OUT = System.out;
 41     static final PrintStream ERR = System.err;
 42     /**
 43      * Entry point. Compiles classes in {@code targets}
 44      *
 45      * @param targets each element can be either
 46      *                'modules:&lt;comma separated list of modules to compile&gt'
 47      *                 or path to jimage file
 48      *                 or path to jar/zip, dir contains classes
 49      *                 or path to .lst file contains list of classes to compile
 50      */
 51     public static void main(String[] targets) {
 52         if (targets.length == 0) {
 53             throw new IllegalArgumentException("Expect a compile target.");
 54         }
 55         String logfile = Utils.LOG_FILE;
 56         PrintStream os = null;
 57         if (logfile != null) {
 58             try {
 59                 os = new PrintStream(Files.newOutputStream(Paths.get(logfile)), true);
 60             } catch (IOException io) {
 61             }
 62         }
 63         if (os != null) {
 64             OUT = os;
 65         }
 66 
 67         boolean passed = false;
 68 
 69         try {
 70             try {
 71                 if (ManagementFactory.getCompilationMXBean() == null) {
 72                     throw new RuntimeException(
 73                             "CTW can not work in interpreted mode");
 74                 }
 75             } catch (java.lang.NoClassDefFoundError e) {
 76                 // compact1, compact2 support
 77             }
 78             ExecutorService executor = createExecutor();
 79             long start = System.currentTimeMillis();
 80             try {
 81                 Arrays.stream(targets)
 82                       .map(PathHandler::create)
 83                       .flatMap(List::stream)
 84                       .forEach(p -> {
 85                           try {
 86                               p.process(executor);
 87                           } finally {
 88                               p.close();
 89                           }
 90                         });
 91             } finally {
 92                 await(executor);
 93             }
 94             CompileTheWorld.OUT.println(String.format("Done (%d classes, %d methods, %d ms)",
 95                     PathHandler.getProcessedClassCount(),
 96                     Compiler.getMethodCount(),
 97                     System.currentTimeMillis() - start));
 98             passed = true;
 99         } catch (Throwable t){
100             t.printStackTrace(ERR);
101         } finally {
102             if (OUT != System.out) {
103                 try {
104                     OUT.close();
105                 } catch (Throwable ignore) {
106                 }
107             }
108             // <clinit> might have started new threads
109             System.exit(passed ? 0 : 1);
110         }
111     }
112 
113     private static ExecutorService createExecutor() {
114         final int threadsCount = Math.min(
115                 Runtime.getRuntime().availableProcessors(),
116                 Utils.CI_COMPILER_COUNT);
117         ExecutorService result;
118         if (threadsCount > 1) {
119             result = new ThreadPoolExecutor(threadsCount, threadsCount,
120                     /* keepAliveTime */ 0L, TimeUnit.MILLISECONDS,
121                     new ArrayBlockingQueue<>(threadsCount),
122                     new ThreadPoolExecutor.CallerRunsPolicy());
123         } else {
124             result = new CurrentThreadExecutor();
125         }
126         return result;
127     }
128 
129     private static void await(ExecutorService executor) {
130         executor.shutdown();
131         while (!executor.isTerminated()) {
132             try {
133                 executor.awaitTermination(Long.MAX_VALUE, TimeUnit.NANOSECONDS);
134             } catch (InterruptedException ie) {
135                 Thread.currentThread().interrupt();
136                 break;
137             }
138         }
139     }
140 
141     private static class CurrentThreadExecutor extends AbstractExecutorService {
142         private boolean isShutdown;
143 
144         @Override
145         public void shutdown() {
146             this.isShutdown = true;
147         }
148 
149         @Override
150         public List<Runnable> shutdownNow() {
151             return null;
152         }
153 
154         @Override
155         public boolean isShutdown() {
156             return isShutdown;
157         }
158 
159         @Override
160         public boolean isTerminated() {
161             return isShutdown;
162         }
163 
164         @Override
165         public boolean awaitTermination(long timeout, TimeUnit unit)
166                 throws InterruptedException {
167             return isShutdown;
168         }
169 
170         @Override
171         public void execute(Runnable command) {
172             command.run();
173         }
174     }
175 }
