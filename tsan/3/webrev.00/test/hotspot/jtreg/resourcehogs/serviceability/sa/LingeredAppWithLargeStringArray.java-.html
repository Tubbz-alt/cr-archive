24 import jdk.test.lib.apps.LingeredApp;
25 
26 public class LingeredAppWithLargeStringArray extends LingeredApp {
27     public static void main(String args[]) {
28         String[] hugeArray = new String[Integer.MAX_VALUE/8];
29         String[] smallArray = {"Just", "for", "testing"};
30         for (int i = 0; i < hugeArray.length/16; i++) {
31             hugeArray[i] = new String(smallArray[i%3]);
32         }
33         LingeredApp.main(args);
34     }
35  }
