 25 /*
 26  * @test
 27  * @summary Ensure that certain JFR events return correct results for resource values
 28  *          when run inside Docker container, such as available CPU and memory.
 29  *          Also make sure that PIDs are based on value provided by container,
 30  *          not by the host system.
 31  * @requires (docker.support & os.maxMemory >= 2g)
 32  * @library /test/lib
 33  * @modules java.base/jdk.internal.misc
 34  *          java.management
 35  *          jdk.jartool/sun.tools.jar
 36  * @build JfrReporter
 37  * @run driver TestJFREvents
 38  */
 39 import java.util.List;
 40 import jdk.test.lib.containers.docker.Common;
 41 import jdk.test.lib.containers.docker.DockerRunOptions;
 42 import jdk.test.lib.containers.docker.DockerTestUtils;
 43 import jdk.test.lib.Asserts;
 44 import jdk.test.lib.process.OutputAnalyzer;
 45 import jdk.test.lib.Utils;
 46 
 47 
 48 public class TestJFREvents {
 49     private static final String imageName = Common.imageName("jfr-events");
 50     private static final String TEST_ENV_VARIABLE = "UNIQUE_VARIABLE_ABC592903XYZ";
 51     private static final String TEST_ENV_VALUE = "unique_value_abc592903xyz";
 52     private static final int availableCPUs = Runtime.getRuntime().availableProcessors();
 53 
 54     public static void main(String[] args) throws Exception {
 55         System.out.println("Test Environment: detected availableCPUs = " + availableCPUs);
 56         if (!DockerTestUtils.canTestDocker()) {
 57             return;
 58         }
 59 
 60         DockerTestUtils.buildJdkDockerImage(imageName, "Dockerfile-BasicTest", "jdk-docker");
 61 
 62         try {
 63 
 64             long MB = 1024*1024;
 65             testMemory("200m", "" + 200*MB);
 66             testMemory("500m", "" + 500*MB);
 67             testMemory("1g", "" + 1024*MB);
 68 
 69             testProcessInfo();
 70 
 71             testEnvironmentVariables();
 72         } finally {
 73             DockerTestUtils.removeDockerImage(imageName);
 74         }
 75     }
 76 
 77     // This test case is currently not in use.
 78     // Once new Container events are available, this test case can be used to test
 79     // processor-related configuration such as active processor count (see JDK-8203359).
 80     private static void cpuTestCase() throws Exception {
 81             // leave one CPU for system and tools, otherwise this test may be unstable
 82             int maxNrOfAvailableCpus =  availableCPUs - 1;
 83             for (int i=1; i < maxNrOfAvailableCpus; i = i * 2) {
 84                 testCPUInfo("jdk.ContainerConfiguration", i, i);
 85             }
 86     }
 87 
 88     private static void testCPUInfo(String eventName, int valueToSet, int expectedValue) throws Exception {
 89         Common.logNewTestCase("CPUInfo: --cpus = " + valueToSet);
 90         String fieldName = "activeProcessorCount";
 91         DockerTestUtils.dockerRunJava(
 92                                       commonDockerOpts()
 93                                       .addDockerOpts("--cpus=" + valueToSet)
 94                                       .addClassOptions(eventName))
 95             .shouldHaveExitValue(0)
 96             .shouldContain(fieldName + " = " + expectedValue);
 97     }
 98 
 99 
100     private static void testMemory(String valueToSet, String expectedValue) throws Exception {
101         Common.logNewTestCase("Memory: --memory = " + valueToSet);
102         DockerTestUtils.dockerRunJava(
103                                       commonDockerOpts()
104                                       .addDockerOpts("--memory=" + valueToSet)
105                                       .addClassOptions("jdk.PhysicalMemory"))
106             .shouldHaveExitValue(0)
107             .shouldContain("totalSize = " + expectedValue);
108     }
109 
110 
111     private static void testProcessInfo() throws Exception {
112         Common.logNewTestCase("ProcessInfo");
113         DockerTestUtils.dockerRunJava(
114                                       commonDockerOpts()
115                                       .addClassOptions("jdk.SystemProcess"))
116             .shouldHaveExitValue(0)
117             .shouldContain("pid = 1");
118     }
119 
120 
121     private static DockerRunOptions commonDockerOpts() {
122         return new DockerRunOptions(imageName, "/jdk/bin/java", "JfrReporter")
123             .addDockerOpts("--volume", Utils.TEST_CLASSES + ":/test-classes/")
124             .addJavaOpts("-cp", "/test-classes/");
125     }
126 
127 
128     private static void testEnvironmentVariables() throws Exception {
129         Common.logNewTestCase("EnvironmentVariables");
130 
131         List<String> cmd = DockerTestUtils.buildJavaCommand(
132                                       commonDockerOpts()
133                                       .addClassOptions("jdk.InitialEnvironmentVariable"));
134 
135         ProcessBuilder pb = new ProcessBuilder(cmd);
136         // Container has JAVA_HOME defined via the Dockerfile; make sure
137         // it is reported by JFR event.
138         // Environment variable set in host system should not be visible inside a container,
139         // and should not be reported by JFR.
140         pb.environment().put(TEST_ENV_VARIABLE, TEST_ENV_VALUE);
141 
142         System.out.println("[COMMAND]\n" + Utils.getCommandLine(pb));
143         OutputAnalyzer out = new OutputAnalyzer(pb.start());
144         System.out.println("[STDERR]\n" + out.getStderr());
145         System.out.println("[STDOUT]\n" + out.getStdout());
146 
147         out.shouldHaveExitValue(0)
148             .shouldContain("key = JAVA_HOME")
149             .shouldContain("value = /jdk")
150             .shouldNotContain(TEST_ENV_VARIABLE)
151             .shouldNotContain(TEST_ENV_VALUE);
152     }
153 }
