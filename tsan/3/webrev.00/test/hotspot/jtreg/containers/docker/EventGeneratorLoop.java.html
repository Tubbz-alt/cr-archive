23 import jdk.jfr.Event;
24 import jdk.jfr.Description;
25 import jdk.jfr.Label;
26 
27 
28 // This class generates simple event in a loop for a specified time.
29 public class EventGeneratorLoop {
30     public static final String MAIN_METHOD_STARTED = "MAIN_METHOD_STARTED";
31 
32     @Label("SimpleEvent")
33     @Description("Simple custom event")
34     static class SimpleEvent extends Event {
35         @Label("Message")
36         String msg;
37 
38         @Label("Count")
39         int count;
40     }
41 
42     public static void main(String[] args) throws Exception {
43         if ((args.length < 1) || (args[0] == null)) {
44             throw new IllegalArgumentException("Expecting one argument: time to run (seconds)");
45         }
46         int howLong = Integer.parseInt(args[0]);
47 
48         System.out.println(MAIN_METHOD_STARTED + ", argument is " + howLong);
49 
50         for (int i=0; i < howLong; i++) {
51             SimpleEvent ev = new SimpleEvent();
52             ev.msg = "Hello";
53             ev.count = i;
54             ev.commit();
55 
56             try { Thread.sleep(1000); } catch (InterruptedException e) {}
57             System.out.print(".");
58         }
59 
60         System.out.println("EventGeneratorLoop is coming to an end");
61     }
62 
63 }
