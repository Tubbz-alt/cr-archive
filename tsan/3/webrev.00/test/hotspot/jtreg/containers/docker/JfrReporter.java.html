25 import java.nio.file.Path;
26 import java.nio.file.Paths;
27 import jdk.jfr.Recording;
28 import jdk.jfr.ValueDescriptor;
29 import jdk.jfr.consumer.RecordedEvent;
30 import jdk.jfr.consumer.RecordingFile;
31 
32 // This class is intended to run inside a container
33 public class JfrReporter {
34     public static void main(String[] args) throws Exception {
35         String eventName = args[0];
36         try(Recording r = new Recording()) {
37             r.enable(eventName);
38             r.start();
39             r.stop();
40             Path p = Paths.get("/", "tmp", eventName + ".jfr");
41             r.dump(p);
42             for (RecordedEvent e : RecordingFile.readAllEvents(p)) {
43                 System.out.println("===== EventType: " + e.getEventType().getName());
44                 for (ValueDescriptor v : e.getEventType().getFields()) {
45                     System.out.println(v.getName() + " = " + e.getValue(v.getName()));
46                 }
47             }
48         }
49     }
50 }
