24 package gc.arguments;

26 /*
27  * @test TestSelectDefaultGC
28  * @summary Test selection of GC when no GC option is specified
29  * @bug 8068582
30  * @key gc
31  * @library /test/lib
32  * @requires vm.gc=="null"

33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @run driver gc.arguments.TestSelectDefaultGC
36  */
37 
38 import jdk.test.lib.process.OutputAnalyzer;
39 import jdk.test.lib.process.ProcessTools;
40 
41 public class TestSelectDefaultGC {
42     public static void assertVMOption(OutputAnalyzer output, String option, boolean value) {
43         output.shouldMatch(" " + option + " .*=.* " + value + " ");
44     }
45 
46     public static void testDefaultGC(boolean actAsServer) throws Exception {
47         String[] args = new String[] {
48           "-XX:" + (actAsServer ? "+" : "-") + "AlwaysActAsServerClassMachine",
49           "-XX:" + (actAsServer ? "-" : "+") + "NeverActAsServerClassMachine",
50           "-XX:+PrintFlagsFinal",
51           "-version"
52         };
53 
54         // Start VM without specifying GC
55         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(args);
56         OutputAnalyzer output = new OutputAnalyzer(pb.start());
57         output.shouldHaveExitValue(0);
58 
59         final boolean isServer = actAsServer;
60 
61         // Verify GC selection
62         // G1 is default for server class machines
63         assertVMOption(output, "UseG1GC",            isServer);
64         // Serial is default for non-server class machines
65         assertVMOption(output, "UseSerialGC",        !isServer);
66         // CMS is never default
67         assertVMOption(output, "UseConcMarkSweepGC", false);
68     }
69 
70     public static void main(String[] args) throws Exception {
71         // Test server class machine
72         testDefaultGC(false);
73 
74         // Test non-server class machine
75         testDefaultGC(true);
76     }
77 }
