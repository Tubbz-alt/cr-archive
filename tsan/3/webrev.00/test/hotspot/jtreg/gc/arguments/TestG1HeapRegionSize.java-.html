24 package gc.arguments;
25 
26 /*
27  * @test TestG1HeapRegionSize
28  * @key gc
29  * @bug 8021879
30  * @requires vm.gc.G1
31  * @summary Verify that the flag G1HeapRegionSize is updated properly
32  * @modules java.base/jdk.internal.misc
33  * @modules java.management/sun.management
34  * @library /test/lib
35  * @run main gc.arguments.TestG1HeapRegionSize
36  */
37 
38 import java.util.regex.Matcher;
39 import java.util.regex.Pattern;
40 
41 import java.util.ArrayList;
42 import java.util.Arrays;
43 
44 import jdk.test.lib.process.OutputAnalyzer;
45 import jdk.test.lib.process.ProcessTools;
46 
47 public class TestG1HeapRegionSize {
48 
49   private static void checkG1HeapRegionSize(String[] flags, int expectedValue, int exitValue) throws Exception {
50     ArrayList<String> flagList = new ArrayList<String>();
51     flagList.addAll(Arrays.asList(flags));
52     flagList.add("-XX:+UseG1GC");
53     flagList.add("-XX:+PrintFlagsFinal");
54     flagList.add("-version");
55 
56     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(flagList.toArray(new String[0]));
57     OutputAnalyzer output = new OutputAnalyzer(pb.start());
58     output.shouldHaveExitValue(exitValue);
59 
60     if (exitValue == 0) {
61       String stdout = output.getStdout();
62       int flagValue = getFlagValue("G1HeapRegionSize", stdout);
63       if (flagValue != expectedValue) {
64         throw new RuntimeException("Wrong value for G1HeapRegionSize. Expected " + expectedValue + " but got " + flagValue);
65       }
66     }
67   }
68 
69   private static int getFlagValue(String flag, String where) {
70     Matcher m = Pattern.compile(flag + "\\s+:?=\\s+\\d+").matcher(where);
71     if (!m.find()) {
72       throw new RuntimeException("Could not find value for flag " + flag + " in output string");
73     }
74     String match = m.group();
75     return Integer.parseInt(match.substring(match.lastIndexOf(" ") + 1, match.length()));
76   }
77 
78   public static void main(String args[]) throws Exception {
79     final int M = 1024 * 1024;
80 
81     checkG1HeapRegionSize(new String[] { "-Xmx64m"   /* default is 1m */        },  1*M, 0);
82     checkG1HeapRegionSize(new String[] { "-Xmx64m",  "-XX:G1HeapRegionSize=2m"  },  2*M, 0);
83     checkG1HeapRegionSize(new String[] { "-Xmx64m",  "-XX:G1HeapRegionSize=3m"  },  2*M, 0);
84     checkG1HeapRegionSize(new String[] { "-Xmx256m", "-XX:G1HeapRegionSize=32m" }, 32*M, 0);
85     checkG1HeapRegionSize(new String[] { "-Xmx256m", "-XX:G1HeapRegionSize=64m" }, 32*M, 1);
86   }
87 }
