24 package gc.arguments;
25 
26 /*
27  * @test TestG1PercentageOptions
28  * @key gc
29  * @bug 8068942
30  * @requires vm.gc.G1
31  * @summary Test argument processing of various percentage options
32  * @library /test/lib
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @run driver gc.arguments.TestG1PercentageOptions
36  */
37 
38 import jdk.test.lib.process.OutputAnalyzer;
39 import jdk.test.lib.process.ProcessTools;
40 
41 public class TestG1PercentageOptions {
42 
43     private static final class OptionDescription {
44         public final String name;
45         public final String[] valid;
46         public final String[] invalid;
47 
48         OptionDescription(String name_, String[] valid_, String[] invalid_) {
49             name = name_;
50             valid = valid_;
51             invalid = invalid_;
52         }
53     }
54 
55     private static final String[] defaultValid = new String[] {
56         "0", "1", "50", "95", "100" };
57     private static final String[] defaultInvalid = new String[] {
58         "-10", "110", "bad" };
59 
60     // All of the G1 product arguments that are percentages.
61     private static final OptionDescription[] percentOptions = new OptionDescription[] {
62         new OptionDescription("G1ConfidencePercent", defaultValid, defaultInvalid)
63         // Other percentage options are not yet validated by argument processing.
64     };
65 
66     private static void check(String flag, boolean is_valid) throws Exception {
67         String[] flags = new String[] { "-XX:+UseG1GC", flag, "-version" };
68         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(flags);
69         OutputAnalyzer output = new OutputAnalyzer(pb.start());
70         if (is_valid) {
71             output.shouldHaveExitValue(0);
72         } else {
73             output.shouldHaveExitValue(1);
74         }
75     }
76 
77     private static
78     void check(String name, String value, boolean is_valid) throws Exception {
79         check("-XX:" + name + "=" + value, is_valid);
80     }
81 
82     public static void main(String args[]) throws Exception {
83         for (OptionDescription option : percentOptions) {
84             for (String value : option.valid) {
85                 check(option.name, value, true);
86             }
87             for (String value : option.invalid) {
88                 check(option.name, value, false);
89             }
90             check("-XX:" + option.name, false);
91             check("-XX:+" + option.name, false);
92             check("-XX:-" + option.name, false);
93         }
94     }
95 }
