 24 package gc.arguments;
 25 
 26 import java.lang.management.GarbageCollectorMXBean;
 27 import java.lang.management.ManagementFactory;
 28 import java.util.Arrays;
 29 import java.util.Objects;
 30 
 31 /**
 32  * Helper class with enum representation of GC types.
 33  */
 34 public final class GCTypes {
 35 
 36     private static <T extends GCType> T getCurrentGCType(Class<T> type) {
 37         return ManagementFactory.getGarbageCollectorMXBeans().stream()
 38                 .map(bean -> getGCTypeByName(type, bean.getName()))
 39                 .filter(Objects::nonNull)
 40                 .findFirst()
 41                 .orElse(null);
 42     }
 43 
 44     private static <T extends GCType> T getGCTypeByName(Class<T> type, String name) {
 45         return Arrays.stream(type.getEnumConstants())
 46                 .filter(e -> e.getGCName().equals(name))
 47                 .findFirst()
 48                 .orElse(null);
 49     }
 50 
 51     private static <T extends GCType> GarbageCollectorMXBean getGCBeanByType(Class<T> type) {
 52         return ManagementFactory.getGarbageCollectorMXBeans().stream()
 53                 .filter(bean -> Arrays.stream(type.getEnumConstants())
 54                         .filter(enumName -> enumName.getGCName().equals(bean.getName()))
 55                         .findFirst()
 56                         .isPresent()
 57                 )
 58                 .findFirst()
 59                 .orElse(null);
 60     }
 61 
 62     /**
 63      * Helper interface used by GCTypes static methods
 64      * to get gcTypeName field of *GCType classes.
 65      */
 66     private interface GCType {
 67 
 68         String getGCName();
 69     }
 70 
 71     public static enum YoungGCType implements GCType {
 72         DefNew("Copy"),
 73         ParNew("ParNew"),
 74         PSNew("PS Scavenge"),
 75         G1("G1 Young Generation");
 76 
 77         @Override
 78         public String getGCName() {
 79             return gcTypeName;
 80         }
 81         private final String gcTypeName;
 82 
 83         private YoungGCType(String name) {
 84             gcTypeName = name;
 85         }
 86 
 87         public static YoungGCType getYoungGCType() {
 88             return GCTypes.getCurrentGCType(YoungGCType.class);
 89         }
 90 
 91         public static GarbageCollectorMXBean getYoungGCBean() {
 92             return GCTypes.getGCBeanByType(YoungGCType.class);
 93         }
 94     }
 95 
 96     public static enum OldGCType implements GCType {
 97         Serial("MarkSweepCompact"),
 98         CMS("ConcurrentMarkSweep"),
 99         PSOld("PS MarkSweep"),
100         G1("G1 Old Generation");
101 
102         private final String gcTypeName;
103 
104         private OldGCType(String name) {
105             gcTypeName = name;
106         }
107 
108         public static OldGCType getOldGCType() {
109             return GCTypes.getCurrentGCType(OldGCType.class);
110         }
111 
112         @Override
113         public String getGCName() {
114             return gcTypeName;
115         }
116     }
117 }
