 24 package gc.arguments;
 25 
 26 /*
 27  * @test TestSmallInitialHeapWithLargePageAndNUMA
 28  * @bug 8023905
 29  * @requires os.family == "linux"
 30  * @requires vm.gc.Parallel
 31  * @summary Check large pages and NUMA are working together via the output message.
 32  * @library /test/lib
 33  * @modules java.base/jdk.internal.misc
 34  * @modules java.management/sun.management
 35  * @build TestSmallInitialHeapWithLargePageAndNUMA
 36  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 37  * @run main/othervm -Xbootclasspath/a:. -XX:+UseHugeTLBFS -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.arguments.TestSmallInitialHeapWithLargePageAndNUMA
 38 */
 39 
 40 import jdk.test.lib.process.ProcessTools;
 41 import jdk.test.lib.process.OutputAnalyzer;
 42 import sun.hotspot.WhiteBox;
 43 
 44 public class TestSmallInitialHeapWithLargePageAndNUMA {
 45 
 46   private static final String MSG_EXIT_TOO_SMALL_HEAP = "Failed initializing NUMA with large pages. Too small heap size";
 47   private static final String MSG_GC_TRIGGERED_BEFORE_INIT = "GC triggered before VM initialization completed.";
 48 
 49   public static void main(String[] args) throws Exception {
 50 
 51     WhiteBox wb = WhiteBox.getWhiteBox();
 52     long heapAlignment = wb.getHeapAlignment();
 53 
 54     // When using large pages, Linux does not support freeing parts of reserved and committed memory.
 55     // And current Linux implementation uses page size as a condition to actually freeing memory.
 56     // If we allocate pages less than NUMA node, NUMA will try to use default page size and
 57     // this will free the memory which Linux does not support.
 58     // Assume the minimum NUMA node as 2.
 59     long initHeap = heapAlignment;
 60     long maxHeap = heapAlignment * 2;
 61 
 62     String[] vmArgs = {"-XX:+UseParallelGC",
 63                        "-Xms" + String.valueOf(initHeap),
 64                        "-Xmx" + String.valueOf(maxHeap),
 65                        "-XX:+UseNUMA",
 66                        "-XX:+UseHugeTLBFS",
 67                        "-XX:+PrintFlagsFinal",
 68                        "-version"};
 69 
 70     ProcessBuilder pb_enabled = ProcessTools.createJavaProcessBuilder(vmArgs);
 71     OutputAnalyzer analyzer = new OutputAnalyzer(pb_enabled.start());
 72 
 73     if (largePageOrNumaEnabled(analyzer)) {
 74       // We reach here, if both NUMA and HugeTLB are supported.
 75       // However final flags will not be printed as NUMA initialization will be failed.
 76       checkAnalyzerValues(analyzer, 1, MSG_EXIT_TOO_SMALL_HEAP);
 77     }
 78   }
 79 
 80   // If both NUMA and large pages are enabled, VM will exit during NUMA initialization
 81   // under the small heap configuration. So final flags will not be printed.
 82   private static boolean largePageOrNumaEnabled(OutputAnalyzer analyzer) {
 83     String output = analyzer.getOutput();
 84 
 85     return !output.contains("[Global flags]");
 86   }
 87 
 88   // We need to test with small heap but fastdebug binary fails to initialize because of the small heap.
 89   // So skip that case.
 90   private static void checkAnalyzerValues(OutputAnalyzer analyzer, int expectedExitValue, String expectedMessage) {
 91     String output = analyzer.getOutput();
 92 
 93     // If the VM exits because of the small heap, skip checking the exit value.
 94     if (!output.contains(MSG_GC_TRIGGERED_BEFORE_INIT)) {
 95       analyzer.shouldHaveExitValue(expectedExitValue);
 96     }
 97     if (expectedMessage != null) {
 98       analyzer.shouldContain(expectedMessage);
 99     }
100   }
101 }
