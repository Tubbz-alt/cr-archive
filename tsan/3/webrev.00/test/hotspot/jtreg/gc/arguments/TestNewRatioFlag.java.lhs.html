 26 /*
 27  * @test TestNewRatioFlag
 28  * @key gc
 29  * @bug 8025166
 30  * @summary Verify that heap devided among generations according to NewRatio
 31  * @requires vm.gc != "Z" & vm.gc != "Shenandoah"
 32  * @library /test/lib
 33  * @library /
 34  * @modules java.base/jdk.internal.misc
 35  *          java.management
 36  * @build sun.hotspot.WhiteBox
 37  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 38  * @run driver gc.arguments.TestNewRatioFlag
 39  */
 40 
 41 import java.util.Arrays;
 42 import java.util.Collections;
 43 import java.util.LinkedList;
 44 import jdk.test.lib.process.ProcessTools;
 45 import jdk.test.lib.process.OutputAnalyzer;
 46 import jdk.test.lib.Utils;
 47 import sun.hotspot.WhiteBox;
 48 
 49 public class TestNewRatioFlag {
 50 
 51     public static final long M = 1024 * 1024;
 52     public static final long HEAP_SIZE = 100 * M;
 53 
 54     public static void main(String args[]) throws Exception {
 55         LinkedList<String> options = new LinkedList<>(
 56                 Arrays.asList(Utils.getFilteredTestJavaOpts("(-XX:[^ ]*NewSize=[^ ]+)|(-Xm[ns][^ ]+)"))
 57         );
 58 
 59         testNewRatio(4, options);
 60         testNewRatio(6, options);
 61         testNewRatio(10, options);
 62         testNewRatio(15, options);
 63         testNewRatio(20, options);
 64     }
 65 
 66     /**
 67      * Verify that actual size of young gen conforms specified NewRatio
 68      *
 69      * @param ratio value of NewRatio option
 70      * @param options additional options for VM
 71      */
 72     public static void testNewRatio(int ratio, LinkedList<String> options) throws Exception {
 73         LinkedList<String> vmOptions = new LinkedList<>(options);
 74         Collections.addAll(vmOptions,
 75                 "-Xbootclasspath/a:.",
 76                 "-XX:+UnlockDiagnosticVMOptions",
 77                 "-XX:+WhiteBoxAPI",
 78                 "-XX:GCLockerEdenExpansionPercent=0",
 79                 "-Xmx" + HEAP_SIZE,
 80                 "-Xms" + HEAP_SIZE,
 81                 "-XX:NewRatio=" + ratio,
 82                 "-XX:-UseLargePages",
 83                 NewRatioVerifier.class.getName(),
 84                 Integer.toString(ratio)
 85         );
 86 
 87         ProcessBuilder procBuilder = ProcessTools.createJavaProcessBuilder(vmOptions.toArray(new String[vmOptions.size()]));
 88         OutputAnalyzer analyzer = new OutputAnalyzer(procBuilder.start());
 89         analyzer.shouldHaveExitValue(0);
 90         System.out.println(analyzer.getOutput());
 91     }
 92 
 93     public static class NewRatioVerifier {
 94 
 95         static WhiteBox wb = WhiteBox.getWhiteBox();
 96 
 97         public static void main(String args[]) {
 98             if (args.length != 1) {
 99                 throw new IllegalArgumentException("Expected 1 arg: <expectedRatio>");
100             }
101             int expectedRatio = Integer.valueOf(args[0]);
102             switch (GCTypes.YoungGCType.getYoungGCType()) {
103                 case DefNew:
104                 case ParNew:
105                     verifyDefNewNewRatio(expectedRatio);
106                     break;
107                 case PSNew:
108                     verifyPSNewRatio(expectedRatio);
109                     break;
110                 case G1:
111                     verifyG1NewRatio(expectedRatio);
112                     break;
113                 default:
114                     throw new RuntimeException("Unexpected young GC type");
115             }
116         }
117 
118         /**
119          * Verify NewSize for DefNew and ParNew collectors.
120          *
121          * Compare expected NewSize calculated according to sizing policies used by DefNew
122          * with NewSize value reported by MemoryPoolMXBeans.
123          */
124         public static void verifyDefNewNewRatio(int expectedRatio) {
125             long initEden = HeapRegionUsageTool.getEdenUsage().getInit();
126             long initSurv = HeapRegionUsageTool.getSurvivorUsage().getInit();
127             long initOld = HeapRegionUsageTool.getOldUsage().getInit();
128 
129             long newSize = initEden + 2 * initSurv;
130 
131             long expectedNewSize = HeapRegionUsageTool.alignDown(initOld / expectedRatio,
132                     wb.getHeapSpaceAlignment());
133 
134             if (expectedNewSize != newSize) {
135                 throw new RuntimeException("Expected young gen size is: " + expectedNewSize
136                         + ", but observed new size is: " + newSize);
137             }
138         }
139 
140         /**
141          * Verify NewSize for PS collector.
142          * Expected NewSize calculated according to alignment policies used by PS
143          * and then compared with actual NewSize obtained from MemoryPoolMXBeans.
144          */
145         public static void verifyPSNewRatio(int expectedRatio) {
146             long initEden = HeapRegionUsageTool.getEdenUsage().getInit();
147             long initSurv = HeapRegionUsageTool.getSurvivorUsage().getInit();
148             long initOld = HeapRegionUsageTool.getOldUsage().getInit();
149 
150             long newSize = initEden + 2 * initSurv;
151 
152             long alignedDownNewSize = HeapRegionUsageTool.alignDown(initOld / expectedRatio,
153                     wb.getHeapSpaceAlignment());
154             long expectedNewSize = HeapRegionUsageTool.alignUp(alignedDownNewSize,
155                     wb.psVirtualSpaceAlignment());
156 
157             if (expectedNewSize != newSize) {
158                 throw new RuntimeException("Expected young gen size is: " + expectedNewSize
159                         + ", but observed new size is: " + newSize);
160             }
161         }
162 
163         /**
164          * Verify NewSize for G1 GC.
165          * Amount of young regions calculated according to sizing policies used by G1
166          * and then compared with actual number of young regions derived from
167          * values reported by MemoryPoolMXBeans and region size.
168          */
169         public static void verifyG1NewRatio(int expectedRatio) {
170             long initEden = HeapRegionUsageTool.getEdenUsage().getInit();
171             long initSurv = HeapRegionUsageTool.getSurvivorUsage().getInit();
172             long maxOld = HeapRegionUsageTool.getOldUsage().getMax();
173 
174             int regionSize = wb.g1RegionSize();
175             int youngListLength = (int) ((initEden + initSurv) / regionSize);
176             int maxRegions = (int) (maxOld / regionSize);
177             int expectedYoungListLength = (int) (maxRegions / (double) (expectedRatio + 1));
178 
179             if (youngListLength != expectedYoungListLength) {
180                 throw new RuntimeException("Expected G1 young list length is: " + expectedYoungListLength
181                         + ", but observed young list length is: " + youngListLength);
182             }
183         }
184     }
185 }
