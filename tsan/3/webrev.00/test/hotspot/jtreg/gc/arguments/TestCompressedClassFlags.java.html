24 package gc.arguments;
25 
26 import jdk.test.lib.process.OutputAnalyzer;
27 import jdk.test.lib.process.ProcessTools;
28 import jdk.test.lib.Platform;
29 
30 /*
31  * @test
32  * @bug 8015107
33  * @summary Tests that VM prints a warning when -XX:CompressedClassSpaceSize
34  *          is used together with -XX:-UseCompressedClassPointers
35  * @library /test/lib
36  * @library /
37  * @modules java.base/jdk.internal.misc
38  *          java.management
39  * @run main gc.arguments.TestCompressedClassFlags
40  */
41 public class TestCompressedClassFlags {
42     public static void main(String[] args) throws Exception {
43         if (Platform.is64bit()) {
44             OutputAnalyzer output = runJava("-XX:CompressedClassSpaceSize=1g",
45                                             "-XX:-UseCompressedClassPointers",
46                                             "-version");
47             output.shouldContain("warning");
48             output.shouldNotContain("error");
49             output.shouldHaveExitValue(0);
50         }
51     }
52 
53     private static OutputAnalyzer runJava(String ... args) throws Exception {
54         ProcessBuilder pb = GCArguments.createJavaProcessBuilder(args);
55         return new OutputAnalyzer(pb.start());
56     }
57 }
