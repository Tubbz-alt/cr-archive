26 /*
27  * @test TestDisableDefaultGC
28  * @summary Test that the VM complains when the default GC is disabled and no other GC is specified
29  * @bug 8068579
30  * @key gc
31  * @library /test/lib

32  * @requires vm.gc=="null"
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @run driver gc.arguments.TestDisableDefaultGC
36  */
37 
38 import jdk.test.lib.process.ProcessTools;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class TestDisableDefaultGC {
42     public static void main(String[] args) throws Exception {
43         // Start VM, disabling all possible default GCs
44         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:-UseSerialGC",
45                                                                   "-XX:-UseParallelGC",
46                                                                   "-XX:-UseG1GC",
47                                                                   "-XX:-UseConcMarkSweepGC",
48                                                                   "-XX:+UnlockExperimentalVMOptions",
49                                                                   "-XX:-UseShenandoahGC",
50                                                                   "-XX:-UseZGC",
51                                                                   "-version");
52         OutputAnalyzer output = new OutputAnalyzer(pb.start());
53         output.shouldMatch("Garbage collector not selected");
54         output.shouldHaveExitValue(1);
55     }
56 }
