24 package gc.arguments;
25 
26 /*
27  * @test TestAggressiveHeap
28  * @key gc
29  * @bug 8179084
30  * @requires vm.gc.Parallel
31  * @summary Test argument processing for -XX:+AggressiveHeap.
32  * @library /test/lib

33  * @modules java.base java.management
34  * @run driver gc.arguments.TestAggressiveHeap
35  */
36 
37 import java.lang.management.ManagementFactory;
38 import javax.management.MBeanServer;
39 import javax.management.ObjectName;
40 
41 import jdk.test.lib.process.OutputAnalyzer;
42 import jdk.test.lib.process.ProcessTools;
43 import jtreg.SkippedException;
44 
45 public class TestAggressiveHeap {
46 
47     public static void main(String args[]) throws Exception {
48         if (canUseAggressiveHeapOption()) {
49             testFlag();
50         }
51     }
52 
53     // Note: Not a normal boolean flag; -XX:-AggressiveHeap is invalid.
54     private static final String option = "-XX:+AggressiveHeap";
55 
56     // Option requires at least 256M, else error during option processing.
57     private static final long minMemory = 256 * 1024 * 1024;
58 
59     // Setting the heap to half of the physical memory is not suitable for
60     // a test environment with many tests running concurrently, setting to
61     // half of the required size instead.
62     private static final String heapSizeOption = "-Xmx128M";
63 
64     // bool UseParallelGC = true {product} {command line}
65     private static final String parallelGCPattern =
66         " *bool +UseParallelGC *= *true +\\{product\\} *\\{command line\\}";
67 
68     private static void testFlag() throws Exception {
69         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
70             option, heapSizeOption, "-XX:+PrintFlagsFinal", "-version");
71 
72         OutputAnalyzer output = new OutputAnalyzer(pb.start());
73 
74         output.shouldHaveExitValue(0);
75 
76         String value = output.firstMatch(parallelGCPattern);
77         if (value == null) {
78             throw new RuntimeException(
79                 option + " didn't set UseParallelGC as if from command line");
80         }
81     }
82 
83     private static boolean haveRequiredMemory() throws Exception {
84         MBeanServer server = ManagementFactory.getPlatformMBeanServer();
85         ObjectName os = new ObjectName("java.lang", "type", "OperatingSystem");
86         Object attr = server.getAttribute(os, "TotalPhysicalMemorySize");
87         String value = attr.toString();
88         long memory = Long.parseLong(value);
89         return memory >= minMemory;
90     }
91 
92     private static boolean canUseAggressiveHeapOption() throws Exception {
93         if (!haveRequiredMemory()) {
94             throw new SkippedException("Skipping test of " + option + " : insufficient memory");
95         }
96         return true;
97     }
98 }
