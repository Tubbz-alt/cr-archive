 24 package gc.arguments;
 25 

 26 import jdk.test.lib.process.ExitCode;
 27 import jdk.test.lib.cli.CommandLineOptionTest;
 28 
 29 /**
 30  * @test
 31  * @bug 8031323
 32  * @summary Verify SurvivorAlignmentInBytes option processing.
 33  * @library /test/lib
 34  * @requires vm.opt.SurvivorAlignmentInBytes == null
 35  *           & vm.opt.ObjectAlignmentInBytes == null
 36  *           & vm.opt.UnlockExperimentalVMOptions == null
 37  *           & (vm.opt.IgnoreUnrecognizedVMOptions == null
 38  *              | vm.opt.IgnoreUnrecognizedVMOptions == "false")
 39  * @modules java.base/jdk.internal.misc
 40  *          java.management
 41  * @run main gc.arguments.TestSurvivorAlignmentInBytesOption
 42  */
 43 public class TestSurvivorAlignmentInBytesOption {
 44     public static void main(String args[]) throws Throwable {
 45         String optionName = "SurvivorAlignmentInBytes";
 46         String unlockExperimentalVMOpts = "UnlockExperimentalVMOptions";
 47         String optionIsExperimental
 48                 = CommandLineOptionTest.getExperimentalOptionErrorMessage(
 49                 optionName);
 50         String valueIsTooSmall= ".*SurvivorAlignmentInBytes.*must be greater"
 51                 + " than or equal to ObjectAlignmentInBytes.*";
 52         String mustBePowerOf2 = ".*SurvivorAlignmentInBytes.*must be "
 53                 + "power of 2.*";
 54 
 55         // Verify that without -XX:+UnlockExperimentalVMOptions usage of
 56         // SurvivorAlignmentInBytes option will cause JVM startup failure
 57         // with the warning message saying that that option is experimental.
 58         String shouldFailMessage = String.format("JVM option '%s' is "
 59                 + "experimental.%nJVM startup should fail without "
 60                 + "-XX:+UnlockExperimentalVMOptions option", optionName);
 61         CommandLineOptionTest.verifyJVMStartup(
 62                 new String[]{optionIsExperimental}, null,
 63                 shouldFailMessage, shouldFailMessage,
 64                 ExitCode.FAIL, false,
 65                 "-XX:-UnlockExperimentalVMOptions",
 66                 CommandLineOptionTest.prepareBooleanFlag(
 67                         unlockExperimentalVMOpts, false),
 68                 CommandLineOptionTest.prepareNumericFlag(optionName, 64));
 69 
 70         // Verify that with -XX:+UnlockExperimentalVMOptions passed to JVM
 71         // usage of SurvivorAlignmentInBytes option won't cause JVM startup
 72         // failure.
 73         String shouldPassMessage = String.format("JVM option '%s' is "
 74                 + "experimental.%nJVM startup should pass with "
 75                 + "-XX:+UnlockExperimentalVMOptions option", optionName);
 76         String noWarningMessage = "There should be no warnings when use "
 77                 + "with -XX:+UnlockExperimentalVMOptions option";
 78         CommandLineOptionTest.verifyJVMStartup(
 79                 null, new String[]{optionIsExperimental},
 80                 shouldPassMessage, noWarningMessage,
 81                 ExitCode.OK, false,
 82                 CommandLineOptionTest.prepareBooleanFlag(
 83                         unlockExperimentalVMOpts, true),
 84                 CommandLineOptionTest.prepareNumericFlag(optionName, 64));
 85 
 86         // Verify that if specified SurvivorAlignmentInBytes is lower than
 87         // ObjectAlignmentInBytes, then the JVM startup will fail with
 88         // appropriate error message.
 89         shouldFailMessage = String.format("JVM startup should fail with "
 90                 + "'%s' option value lower than ObjectAlignmentInBytes", optionName);
 91         CommandLineOptionTest.verifyJVMStartup(
 92                 new String[]{valueIsTooSmall}, null,
 93                 shouldFailMessage, shouldFailMessage,
 94                 ExitCode.FAIL, false,
 95                 CommandLineOptionTest.prepareBooleanFlag(
 96                         unlockExperimentalVMOpts, true),
 97                 CommandLineOptionTest.prepareNumericFlag(optionName, 2));



 98 
 99         // Verify that if specified SurvivorAlignmentInBytes value is not
100         // a power of 2 then the JVM startup will fail with appropriate error
101         // message.
102         shouldFailMessage = String.format("JVM startup should fail with "
103                 + "'%s' option value is not a power of 2", optionName);
104         CommandLineOptionTest.verifyJVMStartup(
105                 new String[]{mustBePowerOf2}, null,
106                 shouldFailMessage, shouldFailMessage,
107                 ExitCode.FAIL, false,
108                 CommandLineOptionTest.prepareBooleanFlag(
109                         unlockExperimentalVMOpts, true),
110                 CommandLineOptionTest.prepareNumericFlag(optionName, 127));
111 
112         // Verify that if SurvivorAlignmentInBytes has correct value, then
113         // the JVM will be started without errors.
114         shouldPassMessage = String.format("JVM startup should pass with "
115                 + "correct '%s' option value", optionName);
116         noWarningMessage = String.format("There should be no warnings when use "
117                 + "correct '%s' option value", optionName);
118         CommandLineOptionTest.verifyJVMStartup(
119                 null, new String[]{".*SurvivorAlignmentInBytes.*"},
120                 shouldPassMessage, noWarningMessage,
121                 ExitCode.OK, false,
122                 CommandLineOptionTest.prepareBooleanFlag(
123                         unlockExperimentalVMOpts, true),
124                 CommandLineOptionTest.prepareNumericFlag(optionName, 128));
125 
126         // Verify that we can setup different SurvivorAlignmentInBytes values.
127         for (int alignment = 32; alignment <= 128; alignment *= 2) {
128             shouldPassMessage = String.format("JVM startup should pass with "
129                     + "'%s' = %d", optionName, alignment);
130             CommandLineOptionTest.verifyOptionValue(optionName,
131                     Integer.toString(alignment), shouldPassMessage,
132                     CommandLineOptionTest.prepareBooleanFlag(
133                             unlockExperimentalVMOpts, true),
134                     CommandLineOptionTest.prepareNumericFlag(
135                             optionName, alignment));
136         }
137     }
138 }
