24 package gc.logging;
25 
26 /*
27  * @test TestGCId
28  * @bug 8043607
29  * @summary Ensure that the GCId is logged
30  * @requires vm.gc=="null"
31  * @key gc
32  * @library /test/lib
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @build sun.hotspot.WhiteBox
36  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
37  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.logging.TestGCId
38  */
39 
40 import jdk.test.lib.process.OutputAnalyzer;
41 import jdk.test.lib.process.ProcessTools;

42 import sun.hotspot.gc.GC;
43 
44 public class TestGCId {
45   public static void main(String[] args) throws Exception {
46     testGCId("UseParallelGC");
47     testGCId("UseG1GC");
48     testGCId("UseConcMarkSweepGC");
49     testGCId("UseSerialGC");










50     if (GC.Shenandoah.isSupported()) {
51         testGCId("UseShenandoahGC");





52     }
53   }
54 
55   private static void verifyContainsGCIDs(OutputAnalyzer output) {
56     output.shouldMatch("\\[.*\\]\\[.*\\]\\[.*\\] GC\\(0\\) ");
57     output.shouldMatch("\\[.*\\]\\[.*\\]\\[.*\\] GC\\(1\\) ");
58     output.shouldHaveExitValue(0);
59   }
60 
61   private static void testGCId(String gcFlag) throws Exception {
62     ProcessBuilder pb_default =
63       ProcessTools.createJavaProcessBuilder("-XX:+UnlockExperimentalVMOptions", "-XX:+" + gcFlag, "-Xlog:gc", "-Xmx10M", GCTest.class.getName());
64     verifyContainsGCIDs(new OutputAnalyzer(pb_default.start()));
65   }
66 
67   static class GCTest {
68     private static byte[] garbage;
69     public static void main(String [] args) {
70       System.out.println("Creating garbage");
71       // create 128MB of garbage. This should result in at least one GC
72       for (int i = 0; i < 1024; i++) {
73         garbage = new byte[128 * 1024];
74       }
75       // do a system gc to get one more gc
76       System.gc();
77       System.out.println("Done");
78     }
79   }
80 }
