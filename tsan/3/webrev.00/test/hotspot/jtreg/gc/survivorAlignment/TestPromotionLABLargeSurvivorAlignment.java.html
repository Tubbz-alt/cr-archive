26 /**
27  * @test gc.survivorAlignment.TestPromotionLABLargeSurvivorAlignment
28  * @bug 8060463
29  * @summary Verify that objects promoted from eden space to survivor space
30  *          with large values for SurvivorAlignmentInBytes succeed.
31  * @requires vm.opt.ExplicitGCInvokesConcurrent != true
32  * @run main/othervm -Xmx128m
33  *                   -XX:+UnlockExperimentalVMOptions
34  *                   -XX:SurvivorAlignmentInBytes=8 -XX:SurvivorRatio=1
35  *                   -XX:-ExplicitGCInvokesConcurrent -XX:-ResizePLAB
36  *                   gc.survivorAlignment.TestPromotionLABLargeSurvivorAlignment
37  * @run main/othervm -Xmx128m
38  *                   -XX:+UnlockExperimentalVMOptions
39  *                   -XX:SurvivorAlignmentInBytes=16 -XX:SurvivorRatio=1
40  *                   -XX:-ExplicitGCInvokesConcurrent -XX:-ResizePLAB
41  *                   gc.survivorAlignment.TestPromotionLABLargeSurvivorAlignment
42  * @run main/othervm -Xmx128m
43  *                   -XX:+UnlockExperimentalVMOptions
44  *                   -XX:SurvivorAlignmentInBytes=256 -XX:SurvivorRatio=1
45  *                   -XX:-ExplicitGCInvokesConcurrent -XX:-ResizePLAB
46  *                   gc.survivorAlignment.TestPromotionLABLargeSurvivorAlignment
47  */
48 public class TestPromotionLABLargeSurvivorAlignment {
49     public static void main(String args[]) {
50         Object garbage[] = new Object[1000000];
51         for (int i = 0; i < garbage.length; i++) {
52             garbage[i] = new byte[0];
53         }
54         for (int i = 0; i < 2; i++) {
55             System.gc();
56         }
57     }
58 }
