24 /*
25  * @test
26  * @bug 4098578
27  * @summary Check if the VM properly throws OutOfMemoryError
28  * @author Sheng Liang
29  * @run main/othervm -Xmx128m gc.InfiniteList
30  */
31 
32 package gc;
33 
34 public class InfiniteList {
35     InfiniteList next;
36     long data[] = new long[50000];
37     public static void main(String[] args) throws Exception {
38         InfiniteList p, q;
39         p = new InfiniteList ();
40         p.data[p.data.length -1 ] = 999;
41         try {
42             while (p != null) {
43                 q = new InfiniteList ();
44                 q.next = p;
45                 p = q;
46             }
47             throw new Exception ("OutOfMemoryError not thrown as expected.");
48         } catch (OutOfMemoryError e) {
49             return;
50         }
51     }
52 }
