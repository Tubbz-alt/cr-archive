24 package gc.parallel;
25 
26 /**
27  * @test AdaptiveGCBoundary
28  * @key gc regression
29  * @summary UseAdaptiveGCBoundary is broken
30  * @bug 8014546
31  * @library /test/lib
32  * @modules java.base/jdk.internal.misc
33  *          java.management
34  * @run main/othervm gc.parallel.AdaptiveGCBoundary
35  * @author jon.masamitsu@oracle.com
36  */
37 
38 import jdk.test.lib.process.ProcessTools;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class AdaptiveGCBoundary {
42   public static void main(String args[]) throws Exception {
43 
44     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
45       "-showversion",
46       "-XX:+UseParallelGC",
47       "-XX:+UseAdaptiveGCBoundary",
48       "-XX:+PrintCommandLineFlags",
49       SystemGCCaller.class.getName()
50       );
51 
52     OutputAnalyzer output = new OutputAnalyzer(pb.start());
53 
54     output.shouldContain("+UseAdaptiveGCBoundary");
55 
56     output.shouldNotContain("error");
57 
58     output.shouldHaveExitValue(0);
59   }
60   static class SystemGCCaller {
61     public static void main(String [] args) {
62       System.gc();
63     }
64   }
65 }
