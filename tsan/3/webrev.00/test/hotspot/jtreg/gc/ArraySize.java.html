24 /*
25  * @test
26  * @bug 4063078
27  * @summary Allocating a ridiculously large array should not crash the VM
28  * @run main/othervm -Xmx128m -Xms32m gc.ArraySize
29  */
30 
31 package gc;
32 
33 public class ArraySize {
34 
35     public static void main(String[] args) throws Exception {
36         boolean thrown = false;
37         try {
38             byte[] buf = new byte[Integer.MAX_VALUE - 1];
39             System.out.print(buf[0]);
40         } catch (OutOfMemoryError x) {
41             thrown = true;
42         }
43         if (! thrown) {
44             throw new Exception("Didn't throw expected OutOfMemoryError");
45         }
46     }
47 
48 }
