24 package gc.startup_warnings;
25 
26 /*
27  * @test TestCMS
28  * @key gc
29  * @bug 8006398 8155948 8179013
30  * @summary Test that CMS prints a warning message
31  * @requires !vm.graal.enabled
32  * @library /test/lib
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @run main gc.startup_warnings.TestCMS
36  */
37 
38 import jdk.test.lib.process.ProcessTools;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 public class TestCMS {
42 
43   public static void runTest(String[] args) throws Exception {
44     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(args);
45     OutputAnalyzer output = new OutputAnalyzer(pb.start());
46     output.shouldContain("deprecated");
47     output.shouldNotContain("error");
48     output.shouldHaveExitValue(0);
49   }
50 
51   public static void main(String args[]) throws Exception {
52     runTest(new String[] {"-XX:+UseConcMarkSweepGC", "-version"});
53     runTest(new String[] {"-Xconcgc", "-version"});
54     runTest(new String[] {"-Xnoconcgc", "-version"});
55   }
56 
57 }
