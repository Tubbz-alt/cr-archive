 39 /*
 40  * @test TestNumWorkerOutputCMS
 41  * @bug 8165292
 42  * @key gc
 43  * @comment Graal does not support CMS
 44  * @requires vm.gc=="null" & !vm.graal.enabled
 45  * @modules java.base/jdk.internal.misc
 46  * @library /test/lib
 47  * @build sun.hotspot.WhiteBox
 48  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 49  * @run main/othervm -XX:+UseConcMarkSweepGC gc.TestNumWorkerOutput UseConcMarkSweepGC
 50  */
 51 
 52 import sun.hotspot.WhiteBox;
 53 
 54 import java.util.regex.Matcher;
 55 import java.util.regex.Pattern;
 56 
 57 import jdk.test.lib.process.OutputAnalyzer;
 58 import jdk.test.lib.process.ProcessTools;
 59 
 60 public class TestNumWorkerOutput {
 61 
 62     public static void checkPatternOnce(String pattern, String what) throws Exception {
 63         Pattern r = Pattern.compile(pattern);
 64         Matcher m = r.matcher(what);
 65 
 66         if (!m.find()) {
 67             throw new RuntimeException("Could not find pattern " + pattern + " in output");
 68         }
 69         if (m.find()) {
 70             throw new RuntimeException("Could find pattern " + pattern + " in output more than once");
 71         }
 72     }
 73 
 74     public static void runTest(String gcArg) throws Exception {
 75         final String[] arguments = {
 76             "-Xbootclasspath/a:.",
 77             "-XX:+UnlockExperimentalVMOptions",
 78             "-XX:+UnlockDiagnosticVMOptions",
 79             "-XX:+WhiteBoxAPI",
 80             "-XX:+" + gcArg,
 81             "-Xmx10M",
 82             "-XX:+PrintGCDetails",
 83             GCTest.class.getName()
 84             };
 85 
 86         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(arguments);
 87         OutputAnalyzer output = new OutputAnalyzer(pb.start());
 88 
 89         output.shouldHaveExitValue(0);
 90 
 91         System.out.println(output.getStdout());
 92 
 93         String stdout = output.getStdout();
 94 
 95         checkPatternOnce(".*[info.*].*[gc,task.*].*GC\\(0\\) .*Using \\d+ workers of \\d+ for evacuation.*", stdout);
 96     }
 97 
 98     public static void main(String[] args) throws Exception {
 99         runTest(args[0]);
100     }
101 
102     static class GCTest {
103         private static final WhiteBox WB = WhiteBox.getWhiteBox();
104 
105         public static Object holder;
106 
107         public static void main(String [] args) {
108             holder = new byte[100];
109             WB.youngGC();
110             System.out.println(holder);
111         }
112     }
113 }
