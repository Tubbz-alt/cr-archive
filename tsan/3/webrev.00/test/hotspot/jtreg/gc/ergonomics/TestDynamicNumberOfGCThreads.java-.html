26 /*
27  * @test TestDynamicNumberOfGCThreads
28  * @bug 8017462
29  * @summary Ensure that UseDynamicNumberOfGCThreads runs
30  * @requires vm.gc=="null"
31  * @key gc
32  * @modules java.base/jdk.internal.misc
33  * @library /test/lib
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.ergonomics.TestDynamicNumberOfGCThreads
37  */
38 
39 import jdk.test.lib.process.OutputAnalyzer;
40 import jdk.test.lib.process.ProcessTools;
41 import sun.hotspot.gc.GC;
42 
43 public class TestDynamicNumberOfGCThreads {
44   public static void main(String[] args) throws Exception {
45 
46     testDynamicNumberOfGCThreads("UseConcMarkSweepGC");
47 
48     testDynamicNumberOfGCThreads("UseG1GC");
49 
50     testDynamicNumberOfGCThreads("UseParallelGC");
51 
52     if (GC.Shenandoah.isSupported()) {
53         testDynamicNumberOfGCThreads("UseShenandoahGC");
54     }
55   }
56 
57   private static void verifyDynamicNumberOfGCThreads(OutputAnalyzer output) {
58     output.shouldHaveExitValue(0); // test should run succesfully
59     output.shouldContain("new_active_workers");
60   }
61 
62   private static void testDynamicNumberOfGCThreads(String gcFlag) throws Exception {
63     // UseDynamicNumberOfGCThreads and TraceDynamicGCThreads enabled
64     String[] baseArgs = {"-XX:+UnlockExperimentalVMOptions", "-XX:+" + gcFlag, "-Xmx10M", "-XX:+UseDynamicNumberOfGCThreads", "-Xlog:gc+task=trace", GCTest.class.getName()};
65 
66     // Base test with gc and +UseDynamicNumberOfGCThreads:
67     ProcessBuilder pb_enabled = ProcessTools.createJavaProcessBuilder(baseArgs);
68     verifyDynamicNumberOfGCThreads(new OutputAnalyzer(pb_enabled.start()));
69 
70     // Ensure it also works on uniprocessors or if user specifies -XX:ParallelGCThreads=1:
71     String[] extraArgs = {"-XX:+UnlockDiagnosticVMOptions", "-XX:+ForceDynamicNumberOfGCThreads", "-XX:ParallelGCThreads=1"};
72     String[] finalArgs = new String[baseArgs.length + extraArgs.length];
73     System.arraycopy(extraArgs, 0, finalArgs, 0,                extraArgs.length);
74     System.arraycopy(baseArgs,  0, finalArgs, extraArgs.length, baseArgs.length);
75     pb_enabled = ProcessTools.createJavaProcessBuilder(finalArgs);
76     verifyDynamicNumberOfGCThreads(new OutputAnalyzer(pb_enabled.start()));
77 
78     // Turn on parallel reference processing
79     String[] parRefProcArg = {"-XX:+ParallelRefProcEnabled", "-XX:-ShowMessageBoxOnError"};
80     String[] parRefArgs = new String[baseArgs.length + parRefProcArg.length];
81     System.arraycopy(parRefProcArg, 0, parRefArgs, 0,                parRefProcArg.length);
82     System.arraycopy(baseArgs,  0, parRefArgs, parRefProcArg.length, baseArgs.length);
83     pb_enabled = ProcessTools.createJavaProcessBuilder(parRefArgs);
84     verifyDynamicNumberOfGCThreads(new OutputAnalyzer(pb_enabled.start()));
85   }
86 
87   static class GCTest {
88     private static byte[] garbage;
89     public static void main(String [] args) {
90       System.out.println("Creating garbage");
91       // create 128MB of garbage. This should result in at least one GC
92       for (int i = 0; i < 1024; i++) {
93         garbage = new byte[128 * 1024];
94       }
95       System.out.println("Done");
96     }
97   }
98 }
