26 /*
27  * @test TestInitialGCThreadLogging
28  * @bug 8157240
29  * @summary Check trace logging of initial GC threads.
30  * @requires vm.gc=="null"
31  * @key gc
32  * @modules java.base/jdk.internal.misc
33  * @library /test/lib
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.ergonomics.TestInitialGCThreadLogging
37  */
38 
39 import jdk.test.lib.process.ProcessTools;
40 import jdk.test.lib.process.OutputAnalyzer;

41 import sun.hotspot.gc.GC;
42 
43 public class TestInitialGCThreadLogging {
44   public static void main(String[] args) throws Exception {

45 
46     testInitialGCThreadLogging("UseConcMarkSweepGC", "GC Thread");
47 
48     testInitialGCThreadLogging("UseG1GC", "GC Thread");

49 
50     testInitialGCThreadLogging("UseParallelGC", "ParGC Thread");



51 
52     if (GC.Shenandoah.isSupported()) {
53         testInitialGCThreadLogging("UseShenandoahGC", "Shenandoah GC Thread");





54     }
55   }
56 
57   private static void verifyDynamicNumberOfGCThreads(OutputAnalyzer output, String threadName) {
58     output.shouldHaveExitValue(0); // test should run succesfully
59     output.shouldContain(threadName);
60   }
61 
62   private static void testInitialGCThreadLogging(String gcFlag, String threadName) throws Exception {
63     // UseDynamicNumberOfGCThreads and TraceDynamicGCThreads enabled
64     String[] baseArgs = {"-XX:+UnlockExperimentalVMOptions", "-XX:+" + gcFlag, "-Xmx10M", "-XX:+UseDynamicNumberOfGCThreads", "-Xlog:gc+task=trace", "-version"};
65 
66     // Base test with gc and +UseDynamicNumberOfGCThreads:
67     ProcessBuilder pb_enabled = ProcessTools.createJavaProcessBuilder(baseArgs);
68     verifyDynamicNumberOfGCThreads(new OutputAnalyzer(pb_enabled.start()), threadName);
69   }
70 }
