 25 package gc.epsilon;
 26 
 27 /**
 28  * @test TestMemoryMXBeans
 29  * @key gc
 30  * @requires vm.gc.Epsilon & !vm.graal.enabled
 31  * @summary Test JMX memory beans
 32  * @modules java.base/jdk.internal.misc
 33  *          java.management
 34  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseEpsilonGC          -Xmx1g gc.epsilon.TestMemoryMXBeans   -1 1024
 35  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseEpsilonGC -Xms1g   -Xmx1g gc.epsilon.TestMemoryMXBeans 1024 1024
 36  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseEpsilonGC -Xms128m -Xmx1g gc.epsilon.TestMemoryMXBeans  128 1024
 37  */
 38 
 39 import java.lang.management.*;
 40 
 41 public class TestMemoryMXBeans {
 42 
 43     static volatile Object sink;
 44 
 45     public static void main(String[] args) throws Exception {
 46         if (args.length < 2) {
 47             throw new IllegalStateException("Should provide expected heap sizes");
 48         }
 49 
 50         long initSize = 1L * Integer.parseInt(args[0]) * 1024 * 1024;
 51         long maxSize =  1L * Integer.parseInt(args[1]) * 1024 * 1024;
 52 
 53         testMemoryBean(initSize, maxSize);
 54         testAllocs();
 55     }
 56 
 57     public static void testMemoryBean(long initSize, long maxSize) {
 58         MemoryMXBean memoryMXBean = ManagementFactory.getMemoryMXBean();
 59         long heapInit = memoryMXBean.getHeapMemoryUsage().getInit();
 60         long heapMax = memoryMXBean.getHeapMemoryUsage().getMax();
 61         memoryMXBean.getNonHeapMemoryUsage().getInit(); // value not used
 62         memoryMXBean.getNonHeapMemoryUsage().getMax();  // value not used
 63 
 64         if (initSize > 0 && heapInit != initSize) {
 65             throw new IllegalStateException("Init heap size is wrong: " + heapInit + " vs " + initSize);
 66         }
 67         if (maxSize > 0 && heapMax != maxSize) {
 68             throw new IllegalStateException("Max heap size is wrong: " + heapMax + " vs " + maxSize);
 69         }
 70     }
 71 
 72     public static void testAllocs() throws Exception {
 73         MemoryMXBean memoryMXBean = ManagementFactory.getMemoryMXBean();
 74 
 75         // Try multiple times, to capture either APIs we call allocate lazily, or the background threads allocating
 76         int maxTries = 10;
 77         int tries = 0;
 78 
 79         while (true) {
 80             // Compute how much we waste during the calls themselves:
 81             long heapUsed1 = memoryMXBean.getHeapMemoryUsage().getUsed();
 82             long heapUsed2 = memoryMXBean.getHeapMemoryUsage().getUsed();
 83             long adj = heapUsed2 - heapUsed1;
 84 
 85             heapUsed1 = memoryMXBean.getHeapMemoryUsage().getUsed();
 86             sink = new int[1024*1024];
 87             heapUsed2 = memoryMXBean.getHeapMemoryUsage().getUsed();
 88 
 89             long diff = heapUsed2 - heapUsed1 - adj;
 90             long min =  8 + 4*1024*1024;
 91             long max = 16 + 4*1024*1024;
 92             if ((min <= diff && diff <= max)) {
 93               // Success
 94               return;
 95             }
 96 
 97             if (tries++ > maxTries) {
 98               throw new IllegalStateException("Allocation did not change used space right: " + diff + " should be in [" + min + ", " + max + "]");
 99             }
100 
101             // Wait and try again
102             Thread.sleep(1000);
103         }
104     }
105 
106 }
