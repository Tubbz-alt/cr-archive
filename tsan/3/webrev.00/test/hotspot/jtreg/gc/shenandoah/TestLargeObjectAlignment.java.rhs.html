25 /*
26  * @test TestLargeObjectAlignment
27  * @summary Shenandoah crashes with -XX:ObjectAlignmentInBytes=16
28  * @key gc
29  * @requires vm.gc.Shenandoah & !vm.graal.enabled & (vm.bits == "64")
30  *
31  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -Xint                   TestLargeObjectAlignment
32  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -XX:-TieredCompilation  TestLargeObjectAlignment
33  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -XX:TieredStopAtLevel=1 TestLargeObjectAlignment
34  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -XX:TieredStopAtLevel=4 TestLargeObjectAlignment
35  */
36 
37 import java.util.ArrayList;
38 import java.util.List;
39 import java.util.concurrent.ThreadLocalRandom;
40 
41 public class TestLargeObjectAlignment {
42 
43     static final int SLABS_COUNT = Integer.getInteger("slabs", 10000);
44     static final int NODE_COUNT = Integer.getInteger("nodes", 10000);
45     static final long TIME_NS = 1000L * 1000L * Integer.getInteger("timeMs", 5000);
46 
47     static Object[] objects;
48 
49     public static void main(String[] args) throws Exception {
50         objects = new Object[SLABS_COUNT];
51 
52         long start = System.nanoTime();
53         while (System.nanoTime() - start < TIME_NS) {
54             objects[ThreadLocalRandom.current().nextInt(SLABS_COUNT)] = createSome();
55         }
56     }
57 
58     public static Object createSome() {
59         List<Integer> result = new ArrayList<Integer>();
60         for (int c = 0; c < NODE_COUNT; c++) {
61             result.add(new Integer(c));
62         }
63         return result;
64     }
65 
66 }
