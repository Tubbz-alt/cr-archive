24 #include <jni.h>
25 #include <string.h>
26 
27 jobject global_ref = NULL;
28 jobject weak_global_ref = NULL;
29 
30 JNIEXPORT void JNICALL
31 Java_TestJNIGlobalRefs_makeGlobalRef(JNIEnv *env, jclass unused, jobject o) {
32   global_ref = (*env)->NewGlobalRef(env, o);
33 }
34 
35 JNIEXPORT void JNICALL
36 Java_TestJNIGlobalRefs_makeWeakGlobalRef(JNIEnv *env, jclass unused, jobject o) {
37   weak_global_ref = (*env)->NewWeakGlobalRef(env, o);
38 }
39 
40 JNIEXPORT jobject JNICALL
41 Java_TestJNIGlobalRefs_readGlobalRef(JNIEnv *env, jclass unused) {
42   return global_ref;
43 }
44 
45 JNIEXPORT jobject JNICALL
46 Java_TestJNIGlobalRefs_readWeakGlobalRef(JNIEnv *env, jclass unused) {
47   return weak_global_ref;
48 }
