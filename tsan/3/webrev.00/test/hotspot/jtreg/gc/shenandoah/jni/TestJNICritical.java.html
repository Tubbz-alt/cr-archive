25 /* @test TestJNICritical
26  * @summary test JNI critical arrays support in Shenandoah
27  * @key gc
28  * @requires vm.gc.Shenandoah & !vm.graal.enabled
29  *
30  * @run main/othervm/native -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:+ShenandoahVerify                 TestJNICritical
31  * @run main/othervm/native -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ShenandoahGCHeuristics=aggressive TestJNICritical
32  */
33 
34 import java.util.Arrays;
35 
36 public class TestJNICritical {
37     static {
38         System.loadLibrary("TestJNICritical");
39     }
40 
41     private static final int NUM_RUNS   = 10000;
42     private static final int ARRAY_SIZE = 10000;
43     private static int[] a;
44     private static int[] b;
45 
46     private static native void copyAtoB(int[] a, int[] b);
47 
48     public static void main(String[] args) {
49         a = new int[ARRAY_SIZE];
50         b = new int[ARRAY_SIZE];
51         for (int i = 0; i < NUM_RUNS; i++) {
52             test();
53         }
54     }
55 
56     private static void test() {
57         int[] a1 = new int[ARRAY_SIZE];
58         int[] b1 = new int[ARRAY_SIZE];
59         fillArray(a);
60         copyAtoB(a, b);
61         copyAtoB(a1, b1); // Don't optimize out garbage arrays.
62         if (!Arrays.equals(a, b)) {
63             throw new RuntimeException("arrays not equal");
64         }
65     }
66 
67     private static void fillArray(int[] array) {
68         for (int i = 0; i < ARRAY_SIZE; i++) {
69             int val = (int) (Math.random() * Integer.MAX_VALUE);
70             array[i] = val;
71         }
72     }
73 }
