50 public class TestRetainObjects {
51 
52     static final int COUNT = 10_000_000;
53     static final int WINDOW = 10_000;
54 
55     static final String[] reachable = new String[WINDOW];
56 
57     public static void main(String[] args) throws Exception {
58         int rIdx = 0;
59         for (int c = 0; c < COUNT; c++) {
60             reachable[rIdx] = ("LargeString" + c);
61             rIdx++;
62             if (rIdx >= WINDOW) {
63                 rIdx = 0;
64             }
65         }
66     }
