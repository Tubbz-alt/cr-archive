25 /**
26  * @test TestMemoryPools
27  * @summary Test JMX memory pools
28  * @key gc
29  * @requires vm.gc.Shenandoah & !vm.graal.enabled
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xmx1g -Xms1g TestMemoryPools
33  */
34 
35 import java.lang.management.*;
36 import java.util.*;
37 
38 public class TestMemoryPools {
39 
40     public static void main(String[] args) throws Exception {
41         List<MemoryManagerMXBean> mms = ManagementFactory.getMemoryManagerMXBeans();
42         if (mms == null) {
43             throw new RuntimeException("getMemoryManagerMXBeans is null");
44         }
45         if (mms.isEmpty()) {
46             throw new RuntimeException("getMemoryManagerMXBeans is empty");
47         }
48         for (MemoryManagerMXBean mmBean : mms) {
49             String[] names = mmBean.getMemoryPoolNames();
50             if (names == null) {
51                 throw new RuntimeException("getMemoryPoolNames() is null");
52             }
53             if (names.length == 0) {
54                 throw new RuntimeException("getMemoryPoolNames() is empty");
55             }
56             for (String name : names) {
57                 if (name == null) {
58                     throw new RuntimeException("pool name is null");
59                 }
60                 if (name.length() == 0) {
61                     throw new RuntimeException("pool name is empty");
62                 }
63             }
64         }
65     }
66 }
