24 /*
25  * @test TestWrongArrayMember
26  * @key gc
27  * @requires vm.gc.Shenandoah
28  *
29  * @run main/othervm -Xmx128m -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC                                      TestWrongArrayMember
30  * @run main/othervm -Xmx128m -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ShenandoahGCHeuristics=traversal TestWrongArrayMember
31  */
32 
33 public class TestWrongArrayMember {
34     public static void main(String... args) throws Exception {
35         Object[] src = new Object[3];
36         src[0] = new Integer(0);
37         src[1] = new Object();
38         src[2] = new Object();
39         Object[] dst = new Integer[3];
40         dst[0] = new Integer(1);
41         dst[1] = new Integer(2);
42         dst[2] = new Integer(3);
43         try {
44             System.arraycopy(src, 0, dst, 0, 3);
45             throw new RuntimeException("Expected ArrayStoreException");
46         } catch (ArrayStoreException e) {
47             if (src[0] != dst[0]) {
48                 throw new RuntimeException("First element not copied");
49             } else if (src[1] == dst[1] || src[2] == dst[2]) {
50                 throw new RuntimeException("Second and third elements are affected");
51             } else {
52                 return; // Passed!
53             }
54         }
55     }
56 }
