24 /**
25  * @test TestThreadFailure
26  * @summary Test OOME in separate thread is recoverable
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  * @library /test/lib
30  * @run main TestThreadFailure
31  */
32 
33 import java.util.*;
34 
35 import jdk.test.lib.process.OutputAnalyzer;
36 import jdk.test.lib.process.ProcessTools;
37 
38 public class TestThreadFailure {
39 
40     static final int SIZE = 1024;
41     static final int COUNT = 16;
42 
43     static class NastyThread extends Thread {
44         @Override
45         public void run() {
46             List<Object> root = new ArrayList<Object>();
47             while (true) {
48                 root.add(new Object[SIZE]);
49             }
50         }
51     }
52 
53     public static void main(String[] args) throws Exception {
54         if (args.length > 0) {
55             for (int t = 0; t < COUNT; t++) {
56                 Thread thread = new NastyThread();
57                 thread.start();
58                 thread.join();
59             }
60             System.out.println("All good");
61             return;
62         }
63 
64         {
65             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
66                     "-Xmx16m",
67                     "-XX:+UnlockExperimentalVMOptions",
68                     "-XX:+UseShenandoahGC",
69                     TestThreadFailure.class.getName(),
70                     "test");
71 
72             OutputAnalyzer analyzer = new OutputAnalyzer(pb.start());
73             analyzer.shouldHaveExitValue(0);
74             analyzer.shouldContain("java.lang.OutOfMemoryError");
75             analyzer.shouldContain("All good");
76         }
77 
78         {
79             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
80                     "-Xmx128m",
81                     "-XX:+UnlockExperimentalVMOptions",
82                     "-XX:+UseShenandoahGC",
83                     TestThreadFailure.class.getName(),
84                     "test");
85 
86             OutputAnalyzer analyzer = new OutputAnalyzer(pb.start());
87             analyzer.shouldHaveExitValue(0);
88             analyzer.shouldContain("java.lang.OutOfMemoryError");
89             analyzer.shouldContain("All good");
90         }
91     }
92 }
