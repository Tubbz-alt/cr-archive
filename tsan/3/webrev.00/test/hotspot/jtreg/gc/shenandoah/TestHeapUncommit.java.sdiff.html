 53 import java.util.Random;
 54 
 55 public class TestHeapUncommit {
 56 
 57     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation
 58 
 59     static volatile Object sink;
 60 
 61     public static void main(String[] args) throws Exception {
 62         final int min = 0;
 63         final int max = 384 * 1024;
 64         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
 65 
 66         Random r = new Random();
 67         for (long c = 0; c < count; c++) {
 68             sink = new int[min + r.nextInt(max - min)];
 69         }
 70     }
