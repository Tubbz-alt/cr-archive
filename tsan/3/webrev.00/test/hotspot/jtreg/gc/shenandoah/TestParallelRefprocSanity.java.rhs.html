34  */

36 import java.lang.ref.*;
37 
38 public class TestParallelRefprocSanity {
39 
40     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation
41 
42     static volatile Object sink;
43 
44     public static void main(String[] args) throws Exception {
45         long count = TARGET_MB * 1024 * 1024 / 32;
46         for (long c = 0; c < count; c++) {
47             sink = new WeakReference<Object>(new Object());
48         }
49     }
50 
51 }
