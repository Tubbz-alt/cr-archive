41     public static void main(String[] args) throws Exception {
42         if (args.length < 2) {
43             throw new IllegalStateException("Should provide expected heap sizes");

46         long initSize = 1L * Integer.parseInt(args[0]) * 1024 * 1024;

52     public static void testMemoryBean(long initSize, long maxSize) {
53         MemoryMXBean memoryMXBean = ManagementFactory.getMemoryMXBean();
54         long heapInit = memoryMXBean.getHeapMemoryUsage().getInit();
55         long heapCommitted = memoryMXBean.getHeapMemoryUsage().getCommitted();
56         long heapMax = memoryMXBean.getHeapMemoryUsage().getMax();
57         long nonHeapInit = memoryMXBean.getNonHeapMemoryUsage().getInit();

60 
61         if (initSize > 0 && heapInit != initSize) {
62             throw new IllegalStateException("Init heap size is wrong: " + heapInit + " vs " + initSize);

37  */

39 import java.lang.management.*;
40 import java.util.*;
41 
42 public class TestMemoryMXBeans {
43 
44     public static void main(String[] args) throws Exception {
45         if (args.length < 2) {
46             throw new IllegalStateException("Should provide expected heap sizes");
47         }
48 
49         long initSize = 1L * Integer.parseInt(args[0]) * 1024 * 1024;
50         long maxSize  = 1L * Integer.parseInt(args[1]) * 1024 * 1024;
51 
52         // wait for GC to uncommit
53         Thread.sleep(1000);
54 
55         testMemoryBean(initSize, maxSize);
56     }
57 
58     public static void testMemoryBean(long initSize, long maxSize) {
59         MemoryMXBean memoryMXBean = ManagementFactory.getMemoryMXBean();
60         long heapInit = memoryMXBean.getHeapMemoryUsage().getInit();
61         long heapCommitted = memoryMXBean.getHeapMemoryUsage().getCommitted();
62         long heapMax = memoryMXBean.getHeapMemoryUsage().getMax();
63         long nonHeapInit = memoryMXBean.getNonHeapMemoryUsage().getInit();
64         long nonHeapCommitted = memoryMXBean.getNonHeapMemoryUsage().getCommitted();
65         long nonHeapMax = memoryMXBean.getNonHeapMemoryUsage().getMax();
66 
67         if (initSize > 0 && heapInit != initSize) {
68             throw new IllegalStateException("Init heap size is wrong: " + heapInit + " vs " + initSize);
69         }
70         if (maxSize > 0 && heapMax != maxSize) {
71             throw new IllegalStateException("Max heap size is wrong: " + heapMax + " vs " + maxSize);
72         }
73         if (initSize > 0 && maxSize > 0 && initSize != maxSize && heapCommitted == heapMax) {
74             throw new IllegalStateException("Committed heap size is max: " + heapCommitted +
75                                             " (init: " + initSize + ", max: " + maxSize + ")");
76         }
77         if (initSize > 0 && maxSize > 0 && initSize == maxSize && heapCommitted != heapMax) {
78             throw new IllegalStateException("Committed heap size is not max: " + heapCommitted +
79                                             " (init: " + initSize + ", max: " + maxSize + ")");
80         }
81         if (initSize > 0 && heapCommitted < initSize) {
82             throw new IllegalStateException("Committed heap size is below min: " + heapCommitted +
83                                             " (init: " + initSize + ", max: " + maxSize + ")");
84         }
85     }
86 }
