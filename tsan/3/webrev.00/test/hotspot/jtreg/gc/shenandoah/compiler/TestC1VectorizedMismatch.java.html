32 import java.util.Arrays;
33 
34 public class TestC1VectorizedMismatch {
35 
36     private static final int NUM_RUNS = 10000;
37     private static final int ARRAY_SIZE = 10000;
38     private static int[] a;
39     private static int[] b;
40 
41     public static void main(String[] args) {
42         a = new int[ARRAY_SIZE];
43         b = new int[ARRAY_SIZE];
44         for (int i = 0; i < NUM_RUNS; i++) {
45             test();
46         }
47     }
48 
49     private static void test() {
50         int[] a1 = new int[ARRAY_SIZE];
51         int[] b1 = new int[ARRAY_SIZE];
52         fillArray(a);
53         System.arraycopy(a, 0, b, 0, ARRAY_SIZE);
54         if (!Arrays.equals(a, b)) {
55             throw new RuntimeException("arrays not equal");
56         }
57     }
58 
59     private static void fillArray(int[] array) {
60         for (int i = 0; i < ARRAY_SIZE; i++) {
61             int val = (int) (Math.random() * Integer.MAX_VALUE);
62             array[i] = val;
63         }
64     }
