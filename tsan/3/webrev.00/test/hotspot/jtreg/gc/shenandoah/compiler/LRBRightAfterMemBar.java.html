24 /**
25  * @test
26  * @bug 8237007
27  * @summary Shenandoah: assert(_base == Tuple) failure during C2 compilation
28  * @key gc
29  * @requires vm.flavor == "server"
30  * @requires vm.gc.Shenandoah & !vm.graal.enabled
31  *
32  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:-BackgroundCompilation -XX:+UseShenandoahGC LRBRightAfterMemBar
33  *
34  */
35 
36 public class LRBRightAfterMemBar {
37     private static Object field1;
38     private static Object field2;
39     static volatile int barrier;
40 
41     public static void main(String[] args) {
42         for (int i = 0; i < 20_000; i++) {
43             test(true, true, new Object());
44             test(false, false, new Object());
45         }
46     }
47 
48     private static Object test(boolean flag, boolean flag2, Object o2) {
49         for (int i = 0; i < 10; i++) {
50             barrier = 0x42; // Membar
51             if (o2 == null) { // hoisted out of loop
52             }
53             // The following line is converted to a CMove with an out
54             // of loop control once the null check above is
55             // hoisted. The CMove is pinned right after the membar and
56             // assigned the membar as control.
57             Object o = flag ? field1 : field2;
58             if (flag2) {
59                 return o;
60             }
61         }
62 
63         return null;
64     }
65 }
