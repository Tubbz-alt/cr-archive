24 #include <jni.h>
25 #include <string.h>
26 
27 JNIEXPORT void JNICALL
28 Java_TestJNICritical_copyAtoB(JNIEnv *env, jclass unused, jintArray a, jintArray b) {
29   jint len = (*env)->GetArrayLength(env, a);
30   jint* aa = (*env)->GetPrimitiveArrayCritical(env, a, 0);
31   jint* bb = (*env)->GetPrimitiveArrayCritical(env, b, 0);
32   memcpy(bb, aa, len * sizeof(jint));
33   (*env)->ReleasePrimitiveArrayCritical(env, b, bb, 0);
34   (*env)->ReleasePrimitiveArrayCritical(env, a, aa, 0);
35 }
