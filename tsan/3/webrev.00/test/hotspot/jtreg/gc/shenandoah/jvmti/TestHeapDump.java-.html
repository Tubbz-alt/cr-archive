24 /**
25  * @test TestHeapDump
26  * @summary Tests JVMTI heap dumps
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  * @compile TestHeapDump.java
30  * @run main/othervm/native/timeout=300 -agentlib:TestHeapDump -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xmx128m -XX:ShenandoahGCHeuristics=aggressive -XX:+UseCompressedOops TestHeapDump
31  * @run main/othervm/native/timeout=300 -agentlib:TestHeapDump -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xmx128m -XX:ShenandoahGCHeuristics=aggressive -XX:-UseCompressedOops TestHeapDump
32  */
33 
34 public class TestHeapDump {
35 
36     private static final int NUM_ITER = 10000;
37 
38     private static final int ARRAY_SIZE = 1000;
39 
40     private static final int EXPECTED_OBJECTS =
41             ARRAY_SIZE +   // array reachable from instance field
42                     1 +            // static field root
43                     1;             // local field root
44 
45     static {
46         try {
47             System.loadLibrary("TestHeapDump");
48         } catch (UnsatisfiedLinkError ule) {
49             System.err.println("Could not load TestHeapDump library");
50             System.err.println("java.library.path: "
51                     + System.getProperty("java.library.path"));
52             throw ule;
53         }
54     }
55 
56     native static int heapdump(Class<?> filterClass);
57 
58     public static void main(String args[]) {
59         new TestHeapDump().run();
60     }
61 
62     // This root needs to be discovered
63     static Object root = new TestObject();
64 
65     // This field needs to be discovered
66     TestObject[] array;
67 
68     public void run() {
69         array = new TestObject[ARRAY_SIZE];
70         for (int i = 0; i < ARRAY_SIZE; i++) {
71             array[i] = new TestObject();
72         }
73         TestObject localRoot = new TestObject();
74         for (int i = 0; i < NUM_ITER; i++) {
75             int numObjs = heapdump(TestObject.class);
76             if (numObjs != EXPECTED_OBJECTS) {
77                 throw new RuntimeException("Expected " + EXPECTED_OBJECTS + " objects, but got " + numObjs);
78             }
79         }
80     }
81 
82     // We look for the instances of this class during the heap scan
83     public static class TestObject {}
84 }
