24 /**
25  * @test TestExpandedWBLostNullCheckDep
26  * @summary Logic that moves a null check in the expanded barrier may cause a memory access that doesn't depend on the barrier to bypass the null check
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  * @requires vm.flavor == "server"
30  * @run main/othervm -XX:-BackgroundCompilation -XX:-UseOnStackReplacement -XX:-TieredCompilation
31  *                   -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC
32  *                   -XX:+StressGCM -XX:+StressLCM TestExpandedWBLostNullCheckDep
33  */
34 
35 public class TestExpandedWBLostNullCheckDep {
36 
37     static void test(int i, int[] arr) {
38         // arr.length depends on a null check for arr
39         if (i < 0 || i >= arr.length) {
40         }
41         // The write barrier here also depends on the null check. The
42         // null check is moved in the barrier to enable implicit null
43         // checks. The null check must not be moved arr.length
44         arr[i] = 0x42;
45     }
46 
47     static public void main(String[] args) {
48         int[] int_arr = new int[10];
49         for (int i = 0; i < 20000; i++) {
50             test(0, int_arr);
51         }
52         try {
53             test(0, null);
54         } catch (NullPointerException npe) {}
55     }
56 }
