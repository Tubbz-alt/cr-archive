50 import java.util.concurrent.ThreadLocalRandom;
51 
52 public class TestSieveObjects {
53 
54     static final int COUNT = 100_000_000;
55     static final int WINDOW = 1_000_000;
56     static final int PAYLOAD = 100;
57 
58     static final MyObject[] arr = new MyObject[WINDOW];
59 
60     public static void main(String[] args) throws Exception {
61         int rIdx = 0;
62         for (int c = 0; c < COUNT; c++) {
63             MyObject v = arr[rIdx];
64             if (v != null) {
65                 if (v.x != rIdx) {
66                     throw new IllegalStateException("Illegal value at index " + rIdx + ": " + v.x);
67                 }
68                 if (ThreadLocalRandom.current().nextInt(1000) > 100) {
69                     arr[rIdx] = null;
70                 }
71             } else {
72                 if (ThreadLocalRandom.current().nextInt(1000) > 500) {
73                     arr[rIdx] = new MyObject(rIdx);
74                 }
75             }
76             rIdx++;
77             if (rIdx >= WINDOW) {
78                 rIdx = 0;
79             }
80         }
81     }
82 
83     public static class MyObject {
84         public int x;
85         public byte[] payload;
86 
87         public MyObject(int x) {
88             this.x = x;
89             this.payload = new byte[PAYLOAD];
90         }
91     }
92 
93 }
