48 public class TestStringInternCleanup {
49 
50     static final int COUNT = 1_000_000;
51     static final int WINDOW = 1_000;
52 
53     static final String[] reachable = new String[WINDOW];
54 
55     public static void main(String[] args) throws Exception {
56         int rIdx = 0;
57         for (int c = 0; c < COUNT; c++) {
58             reachable[rIdx] = ("LargeInternedString" + c).intern();
59             rIdx++;
60             if (rIdx >= WINDOW) {
61                 rIdx = 0;
62             }
63         }
64     }
65 
66 }
