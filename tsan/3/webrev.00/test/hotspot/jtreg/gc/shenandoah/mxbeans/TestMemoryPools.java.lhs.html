24 /**
25  * @test TestMemoryPools
26  * @summary Test JMX memory pools
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xmx1g -Xms1g TestMemoryPools
32  */
33 
34 import java.lang.management.*;
35 import java.util.*;
36 
37 public class TestMemoryPools {
38 
39     public static void main(String[] args) throws Exception {
40         List<MemoryManagerMXBean> mms = ManagementFactory.getMemoryManagerMXBeans();
41         if (mms == null) {
42             throw new RuntimeException("getMemoryManagerMXBeans is null");
43         }
44         if (mms.isEmpty()) {
45             throw new RuntimeException("getMemoryManagerMXBeans is empty");
46         }
47         for (MemoryManagerMXBean mmBean : mms) {
48             String[] names = mmBean.getMemoryPoolNames();
49             if (names == null) {
50                 throw new RuntimeException("getMemoryPoolNames() is null");
51             }
52             if (names.length == 0) {
53                 throw new RuntimeException("getMemoryPoolNames() is empty");
54             }
55             for (String name : names) {
56                 if (name == null) {
57                     throw new RuntimeException("pool name is null");
58                 }
59                 if (name.length() == 0) {
60                     throw new RuntimeException("pool name is empty");
61                 }
62             }
63         }
64     }
65 }
