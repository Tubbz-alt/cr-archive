 41 /* @test TestPinnedGarbage
 42  * @summary Test that garbage in the pinned region does not crash VM
 43  * @key gc
 44  * @requires vm.gc.Shenandoah & !vm.graal.enabled
 45  *
 46  * @run main/othervm/native -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -Xmx512m
 47  *      -XX:+UseShenandoahGC -XX:ShenandoahGCHeuristics=aggressive
 48  *      TestPinnedGarbage
 49  *
 50  * @run main/othervm/native -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -Xmx512m
 51  *      -XX:+UseShenandoahGC
 52  *      -XX:+ShenandoahVerify
 53  *      TestPinnedGarbage
 54  */
 55 
 56 import java.util.Arrays;
 57 import java.util.concurrent.*;
 58 
 59 public class TestPinnedGarbage {
 60     static {
 61         System.loadLibrary("TestPinnedGarbage");
 62     }
 63 
 64     private static final int NUM_RUNS      = 1_000;
 65     private static final int OBJS_COUNT    = 1_000;
 66     private static final int GARBAGE_COUNT = 1_000_000;
 67 
 68     private static native void pin(int[] a);
 69     private static native void unpin(int[] a);
 70 
 71     public static void main(String[] args) {
 72         for (int i = 0; i < NUM_RUNS; i++) {
 73             test();
 74         }
 75     }
 76 
 77     private static void test() {
 78         Object[] objs = new Object[OBJS_COUNT];
 79         for (int i = 0; i < OBJS_COUNT; i++) {
 80             objs[i] = new MyClass();
 81         }
 82 
 83         int[] cog = new int[10];
 84         int cogIdx = ThreadLocalRandom.current().nextInt(OBJS_COUNT);
 85         objs[cogIdx] = cog;
 86         pin(cog);
 87 
 88         for (int i = 0; i < GARBAGE_COUNT; i++) {
 89             int rIdx = ThreadLocalRandom.current().nextInt(OBJS_COUNT);
 90             if (rIdx != cogIdx) {
 91                 objs[rIdx] = new MyClass();
 92             }
 93         }
 94 
 95         unpin(cog);
 96     }
 97 
 98     public static class MyClass {
 99         public Object ref = new Object();
100     }
101 
102 }
