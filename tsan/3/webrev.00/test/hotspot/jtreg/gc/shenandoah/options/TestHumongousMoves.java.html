41  *      TestHumongousMoves
42  */
43 
44 import java.util.Random;
45 
46 public class TestHumongousMoves {
47 
48     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation
49 
50     static volatile Object sink;
51 
52     public static void main(String[] args) throws Exception {
53         final int min = 0;
54         final int max = 384 * 1024;
55         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
56 
57         Random r = new Random();
58         for (long c = 0; c < count; c++) {
59             sink = new int[min + r.nextInt(max - min)];
60         }
61     }
62 
63 }
