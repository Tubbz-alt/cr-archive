 25 /*
 26  * @test TestArgumentRanges
 27  * @summary Test that Shenandoah arguments are checked for ranges where applicable
 28  * @key gc
 29  * @requires vm.gc.Shenandoah & !vm.graal.enabled
 30  * @library /test/lib
 31  * @modules java.base/jdk.internal.misc
 32  *          java.management
 33  * @run driver TestArgumentRanges
 34  */
 35 
 36 import jdk.test.lib.process.ProcessTools;
 37 import jdk.test.lib.process.OutputAnalyzer;
 38 
 39 public class TestArgumentRanges {
 40     public static void main(String[] args) throws Exception {
 41         testRange("ShenandoahGarbageThreshold", 0, 100);
 42         testRange("ShenandoahMinFreeThreshold", 0, 100);
 43         testRange("ShenandoahAllocationThreshold", 0, 100);
 44         testHeuristics();
 45     }
 46 
 47     private static void testHeuristics() throws Exception {
 48 
 49         {
 50             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
 51                     "-XX:+UnlockExperimentalVMOptions",
 52                     "-XX:+UseShenandoahGC",
 53                     "-XX:ShenandoahGCHeuristics=aggressive",
 54                     "-version");
 55             OutputAnalyzer output = new OutputAnalyzer(pb.start());
 56             output.shouldHaveExitValue(0);
 57         }
 58         {
 59             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
 60                     "-XX:+UnlockExperimentalVMOptions",
 61                     "-XX:+UseShenandoahGC",
 62                     "-XX:ShenandoahGCHeuristics=static",
 63                     "-version");
 64             OutputAnalyzer output = new OutputAnalyzer(pb.start());
 65             output.shouldHaveExitValue(0);
 66         }
 67         {
 68             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
 69                     "-XX:+UnlockExperimentalVMOptions",
 70                     "-XX:+UseShenandoahGC",
 71                     "-XX:ShenandoahGCHeuristics=fluff",
 72                     "-version");
 73             OutputAnalyzer output = new OutputAnalyzer(pb.start());
 74             output.shouldMatch("Unknown -XX:ShenandoahGCHeuristics option");
 75             output.shouldHaveExitValue(1);
 76         }
 77     }
 78 
 79     private static void testRange(String option, int min, int max) throws Exception {
 80         {
 81             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
 82                     "-XX:+UnlockExperimentalVMOptions",
 83                     "-XX:+UseShenandoahGC",
 84                     "-XX:" + option + "=" + (max + 1),
 85                     "-version");
 86             OutputAnalyzer output = new OutputAnalyzer(pb.start());
 87             output.shouldHaveExitValue(1);
 88         }
 89         {
 90             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
 91                     "-XX:+UnlockExperimentalVMOptions",
 92                     "-XX:+UseShenandoahGC",
 93                     "-XX:" + option + "=" + max,
 94                     "-version");
 95             OutputAnalyzer output = new OutputAnalyzer(pb.start());
 96             output.shouldHaveExitValue(0);
 97         }
 98         {
 99             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
100                     "-XX:+UnlockExperimentalVMOptions",
101                     "-XX:+UseShenandoahGC",
102                     "-XX:" + option + "=" + (min - 1),
103                     "-version");
104             OutputAnalyzer output = new OutputAnalyzer(pb.start());
105             output.shouldHaveExitValue(1);
106         }
107         {
108             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
109                     "-XX:+UnlockExperimentalVMOptions",
110                     "-XX:+UseShenandoahGC",
111                     "-XX:" + option + "=" + min,
112                     "-version");
113             OutputAnalyzer output = new OutputAnalyzer(pb.start());
114             output.shouldHaveExitValue(0);
115         }
116     }
117 }
