31 import java.util.Arrays;
32 
33 public class TestC1VectorizedMismatch {
34 
35     private static final int NUM_RUNS = 10000;
36     private static final int ARRAY_SIZE = 10000;
37     private static int[] a;
38     private static int[] b;
39 
40     public static void main(String[] args) {
41         a = new int[ARRAY_SIZE];
42         b = new int[ARRAY_SIZE];
43         for (int i = 0; i < NUM_RUNS; i++) {
44             test();
45         }
46     }
47 
48     private static void test() {
49         int[] a1 = new int[ARRAY_SIZE];
50         int[] b1 = new int[ARRAY_SIZE];
51         fillArray(a);
52         System.arraycopy(a, 0, b, 0, ARRAY_SIZE);
53         if (!Arrays.equals(a, b)) {
54             throw new RuntimeException("arrays not equal");
55         }
56     }
57 
58     private static void fillArray(int[] array) {
59         for (int i = 0; i < ARRAY_SIZE; i++) {
60             int val = (int) (Math.random() * Integer.MAX_VALUE);
61             array[i] = val;
62         }
63     }
