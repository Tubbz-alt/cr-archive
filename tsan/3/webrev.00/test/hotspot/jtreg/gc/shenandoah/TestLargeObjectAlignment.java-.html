24 /*
25  * @test TestLargeObjectAlignment
26  * @summary Shenandoah crashes with -XX:ObjectAlignmentInBytes=16
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  *
30  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -Xint                   TestLargeObjectAlignment
31  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -XX:-TieredCompilation  TestLargeObjectAlignment
32  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -XX:TieredStopAtLevel=1 TestLargeObjectAlignment
33  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ObjectAlignmentInBytes=16 -XX:TieredStopAtLevel=4 TestLargeObjectAlignment
34  */
35 
36 import java.util.ArrayList;
37 import java.util.List;
38 import java.util.concurrent.ThreadLocalRandom;
39 
40 public class TestLargeObjectAlignment {
41 
42     static final int SLABS_COUNT = Integer.getInteger("slabs", 10000);
43     static final int NODE_COUNT = Integer.getInteger("nodes", 10000);
44     static final long TIME_NS = 1000L * 1000L * Integer.getInteger("timeMs", 5000);
45 
46     static Object[] objects;
47 
48     public static void main(String[] args) throws Exception {
49         objects = new Object[SLABS_COUNT];
50 
51         long start = System.nanoTime();
52         while (System.nanoTime() - start < TIME_NS) {
53             objects[ThreadLocalRandom.current().nextInt(SLABS_COUNT)] = createSome();
54         }
55     }
56 
57     public static Object createSome() {
58         List<Integer> result = new ArrayList<Integer>();
59         for (int c = 0; c < NODE_COUNT; c++) {
60             result.add(new Integer(c));
61         }
62         return result;
63     }
64 
65 }
