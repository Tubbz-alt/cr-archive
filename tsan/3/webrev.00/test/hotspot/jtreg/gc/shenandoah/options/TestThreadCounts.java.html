25 /*
26  * @test TestThreadCounts
27  * @summary Test that Shenandoah GC thread counts are handled well
28  * @key gc
29  * @requires vm.gc.Shenandoah & !vm.graal.enabled
30  * @library /test/lib
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  * @run driver TestThreadCounts
34  */
35 
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class TestThreadCounts {
40     public static void main(String[] args) throws Exception {
41         for (int conc = 0; conc < 16; conc++) {
42             for (int par = 0; par < 16; par++) {
43                 testWith(conc, par);
44             }
45         }
46     }
47 
48     private static void testWith(int conc, int par) throws Exception {
49         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockDiagnosticVMOptions",
50                 "-XX:+UnlockExperimentalVMOptions",
51                 "-XX:+UseShenandoahGC",
52                 "-XX:ConcGCThreads=" + conc,
53                 "-XX:ParallelGCThreads=" + par,
54                 "-version");
55         OutputAnalyzer output = new OutputAnalyzer(pb.start());
56 
57         if (conc == 0) {
58             output.shouldContain("Shenandoah expects ConcGCThreads > 0");
59             output.shouldHaveExitValue(1);
60         } else if (par == 0) {
61             output.shouldContain("Shenandoah expects ParallelGCThreads > 0");
62             output.shouldHaveExitValue(1);
63         } else if (conc > par) {
64             output.shouldContain("Shenandoah expects ConcGCThreads <= ParallelGCThreads");
65             output.shouldHaveExitValue(1);
66         } else {
67             output.shouldNotContain("Shenandoah expects ConcGCThreads <= ParallelGCThreads");
68             output.shouldHaveExitValue(0);
69         }
70     }
71 
72 }
