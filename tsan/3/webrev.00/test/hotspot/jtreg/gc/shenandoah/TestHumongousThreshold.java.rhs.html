108 import java.util.Random;
109 
110 public class TestHumongousThreshold {
111 
112     static final long TARGET_MB = Long.getLong("target", 20_000); // 20 Gb allocation
113 
114     static volatile Object sink;
115 
116     public static void main(String[] args) throws Exception {
117         final int min = 0;
118         final int max = 384 * 1024;
119         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
120 
121         Random r = new Random();
122         for (long c = 0; c < count; c++) {
123             sink = new int[min + r.nextInt(max - min)];
124         }
125     }
126 
127 }
