31 public class TestC1ArrayCopyNPE {
32 
33     private static final int NUM_RUNS = 10000;
34     private static final int ARRAY_SIZE = 10000;
35     private static int[] a;
36     private static int[] b;
37 
38     public static void main(String[] args) {
39         a = null;
40         b = new int[ARRAY_SIZE];
41         for (int i = 0; i < NUM_RUNS; i++) {
42             test();
43         }
44         a = new int[ARRAY_SIZE];
45         b = null;
46         for (int i = 0; i < NUM_RUNS; i++) {
47             test();
48         }
49     }
50 
51     private static void test() {
52         try {
53             System.arraycopy(a, 0, b, 0, ARRAY_SIZE);
54             throw new RuntimeException("test failed");
55         } catch (NullPointerException ex) {
56             // Ok
57         }
58     }
59 }
