40 import java.util.Random;
41 
42 public class TestElasticTLAB {
43 
44     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation
45 
46     static volatile Object sink;
47 
48     public static void main(String[] args) throws Exception {
49         final int min = 0;
50         final int max = 384 * 1024;
51         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
52 
53         Random r = new Random();
54         for (long c = 0; c < count; c++) {
55             sink = new int[min + r.nextInt(max - min)];
56         }
57     }
58 
59 }
