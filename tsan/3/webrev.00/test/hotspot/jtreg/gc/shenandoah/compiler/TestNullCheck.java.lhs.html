24 /**
25  * @test TestNullCheck
26  * @summary implicit null check on brooks pointer must not cause crash
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  * @run main/othervm -XX:-BackgroundCompilation -XX:-UseOnStackReplacement -XX:-TieredCompilation
30  *                   -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC
31  *                   -Xmx4G -XX:HeapBaseMinAddress=0x800000000 TestNullCheck
32  */
33 
34 // HeapBaseMinAddress above forces compressed oops with a base
35 
36 public class TestNullCheck {
37 
38     int f;
39 
40     static int test1(TestNullCheck o) {
41         return o.f;
42     }
43 
44     static TestNullCheck static_obj = new TestNullCheck();
45 
46     static int test2() {
47         return static_obj.f;
48     }
49 
50     static public void main(String[] args) {
51         TestNullCheck o = new TestNullCheck();
52         for (int i = 0; i < 20000; i++) {
53             test1(o);
54             test2();
55         }
56         try {
57             test1(null);
58         } catch (NullPointerException npe) {}
59         static_obj = null;
60         try {
61             test2();
62         } catch (NullPointerException npe) {}
63     }
64 }
