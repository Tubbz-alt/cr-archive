24 /*
25  * @test TestArrayCopyCheckCast
26  * @key gc
27  * @requires vm.gc.Shenandoah
28  *
29  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:TieredStopAtLevel=0 -Xmx16m TestArrayCopyCheckCast
30  */
31 public class TestArrayCopyCheckCast {
32 
33     static class Foo {}
34     static class Bar {}
35 
36     public static void main(String[] args) throws Exception {
37         try {
38             Object[] array1 = new Object[1];
39             array1[0] = new Bar();
40             Foo[] array2 = new Foo[1];
41             System.arraycopy(array1, 0, array2, 0, 1);
42             throw new RuntimeException();
43         } catch (ArrayStoreException ex) {
44             // expected
45         }
46     }
47 
48 }
