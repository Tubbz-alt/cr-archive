48 import java.util.*;
49 import java.util.concurrent.*;
50 
51 public class TestAllocHumongousFragment {
52 
53     static final long TARGET_MB = Long.getLong("target", 30_000); // 30 Gb allocations
54     static final long LIVE_MB   = Long.getLong("occupancy", 700); // 700 Mb alive
55 
56     static volatile Object sink;
57 
58     static List<int[]> objects;
59 
60     public static void main(String[] args) throws Exception {
61         final int min = 128 * 1024;
62         final int max = 16 * 1024 * 1024;
63         final long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
64 
65         objects = new ArrayList<>();
66         long current = 0;
67 
68         Random r = new Random();
69         for (long c = 0; c < count; c++) {
70             while (current > LIVE_MB * 1024 * 1024) {
71                 int idx = ThreadLocalRandom.current().nextInt(objects.size());
72                 int[] remove = objects.remove(idx);
73                 current -= remove.length * 4 + 16;
74             }
75 
76             int[] newObj = new int[min + r.nextInt(max - min)];
77             current += newObj.length * 4 + 16;
78             objects.add(newObj);
79             sink = new Object();
80 
81             System.out.println("Allocated: " + (current / 1024 / 1024) + " Mb");
82         }
83     }
84 
85 }
