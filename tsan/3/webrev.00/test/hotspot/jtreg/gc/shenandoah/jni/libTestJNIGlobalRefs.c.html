25 #include <jni.h>
26 #include <string.h>
27 
28 jobject global_ref = NULL;
29 jobject weak_global_ref = NULL;
30 
31 JNIEXPORT void JNICALL
32 Java_TestJNIGlobalRefs_makeGlobalRef(JNIEnv *env, jclass unused, jobject o) {
33   global_ref = (*env)->NewGlobalRef(env, o);
34 }
35 
36 JNIEXPORT void JNICALL
37 Java_TestJNIGlobalRefs_makeWeakGlobalRef(JNIEnv *env, jclass unused, jobject o) {
38   weak_global_ref = (*env)->NewWeakGlobalRef(env, o);
39 }
40 
41 JNIEXPORT jobject JNICALL
42 Java_TestJNIGlobalRefs_readGlobalRef(JNIEnv *env, jclass unused) {
43   return global_ref;
44 }
45 
46 JNIEXPORT jobject JNICALL
47 Java_TestJNIGlobalRefs_readWeakGlobalRef(JNIEnv *env, jclass unused) {
48   return weak_global_ref;
49 }
