39 import java.util.Random;
40 
41 public class TestElasticTLAB {
42 
43     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation
44 
45     static volatile Object sink;
46 
47     public static void main(String[] args) throws Exception {
48         final int min = 0;
49         final int max = 384 * 1024;
50         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
51 
52         Random r = new Random();
53         for (long c = 0; c < count; c++) {
54             sink = new int[min + r.nextInt(max - min)];
55         }
56     }
57 
58 }
