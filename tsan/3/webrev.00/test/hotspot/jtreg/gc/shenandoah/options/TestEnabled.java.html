24 import java.lang.management.GarbageCollectorMXBean;
25 import java.lang.management.ManagementFactory;
26 
27 /*
28  * @test TestEnabled
29  * @key gc
30  * @requires vm.gc.Shenandoah & vm.gc == "null" & !vm.graal.enabled
31  * @run main/othervm -Dexpected=false -Xmx64m                                                       TestEnabled
32  * @run main/othervm -Dexpected=true  -Xmx64m -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC TestEnabled
33  */
34 
35 /*
36  * @test TestEnabledAlready
37  * @key gc
38  * @requires vm.gc.Shenandoah & vm.gc == "Shenandoah" & !vm.graal.enabled
39  * @run main/othervm -Dexpected=true -Xmx64m                                                        TestEnabled
40  */
41 public class TestEnabled {
42 
43     public static void main(String... args) {
44         boolean expected = Boolean.getBoolean("expected");
45         boolean actual = isEnabled();
46         if (expected != actual) {
47             throw new IllegalStateException("Error: expected = " + expected + ", actual = " + actual);
48         }
49     }
50 
51     public static boolean isEnabled() {
52         for (GarbageCollectorMXBean bean : ManagementFactory.getGarbageCollectorMXBeans()) {
53             if (bean.getName().contains("Shenandoah")) {
54                 return true;
55             }
56         }
57         return false;
58     }
59 
60 }
