24 /**
25  * @test TestWriteBarrierClearControl
26  * @summary Clearing control during final graph reshape causes memory barrier to loose dependency on null check
27  * @key gc
28  * @requires vm.gc.Shenandoah & !vm.graal.enabled
29  * @requires vm.flavor == "server"
30  * @run main/othervm -XX:-BackgroundCompilation -XX:-UseOnStackReplacement -XX:-TieredCompilation
31  *                   -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC
32  *                   -XX:+UnlockDiagnosticVMOptions -XX:+StressLCM -XX:+StressGCM
33  *                   TestWriteBarrierClearControl
34  *
35  */
36 public class TestWriteBarrierClearControl {
37 
38     int f;
39 
40     static void test1(TestWriteBarrierClearControl o) {
41         o.f = 0x42;
42     }
43 
44     static TestWriteBarrierClearControl fo = new TestWriteBarrierClearControl();
45 
46     static void test2() {
47         TestWriteBarrierClearControl o = fo;
48         o.f = 0x42;
49     }
50 
51     static public void main(String[] args) {
52         TestWriteBarrierClearControl o = new TestWriteBarrierClearControl();
53         for (int i = 0; i < 20000; i++) {
54             test1(o);
55             test2();
56         }
57         try {
58             test1(null);
59         } catch (NullPointerException npe) {}
60         fo = null;
61         try {
62             test2();
63         } catch (NullPointerException npe) {}
64     }
65 }
