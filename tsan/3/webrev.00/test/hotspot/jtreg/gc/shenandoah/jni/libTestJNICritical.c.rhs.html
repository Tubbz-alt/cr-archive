25 #include <jni.h>
26 #include <string.h>
27 
28 JNIEXPORT void JNICALL
29 Java_TestJNICritical_copyAtoB(JNIEnv *env, jclass unused, jintArray a, jintArray b) {
30   jint len = (*env)->GetArrayLength(env, a);
31   jint* aa = (*env)->GetPrimitiveArrayCritical(env, a, 0);
32   jint* bb = (*env)->GetPrimitiveArrayCritical(env, b, 0);
33   memcpy(bb, aa, len * sizeof(jint));
34   (*env)->ReleasePrimitiveArrayCritical(env, b, bb, 0);
35   (*env)->ReleasePrimitiveArrayCritical(env, a, aa, 0);
36 }
