25 /* @test TestC1ArrayCopyNPE
26  * @summary test C1 arraycopy intrinsic
27  * @key gc
28  * @requires vm.gc.Shenandoah & !vm.graal.enabled
29  * @run main/othervm -XX:TieredStopAtLevel=1 -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:ShenandoahGCHeuristics=aggressive TestC1ArrayCopyNPE
30  */
31 
32 public class TestC1ArrayCopyNPE {
33 
34     private static final int NUM_RUNS = 10000;
35     private static final int ARRAY_SIZE = 10000;
36     private static int[] a;
37     private static int[] b;
38 
39     public static void main(String[] args) {
40         a = null;
41         b = new int[ARRAY_SIZE];
42         for (int i = 0; i < NUM_RUNS; i++) {
43             test();
44         }
45         a = new int[ARRAY_SIZE];
46         b = null;
47         for (int i = 0; i < NUM_RUNS; i++) {
48             test();
49         }
50     }
51 
52     private static void test() {
53         try {
54             System.arraycopy(a, 0, b, 0, ARRAY_SIZE);
55             throw new RuntimeException("test failed");
56         } catch (NullPointerException ex) {
57             // Ok
58         }
59     }
60 }
