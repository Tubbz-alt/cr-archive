43 public class TestLotsOfCycles {
44 
45     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation, around 1K cycles to handle
46     static final long STRIDE = 100_000;
47 
48     static volatile Object sink;
49 
50     public static void main(String[] args) throws Exception {
51         long count = TARGET_MB * 1024 * 1024 / 16;
52         for (long c = 0; c < count; c += STRIDE) {
53             for (long s = 0; s < STRIDE; s++) {
54                 sink = new Object();
55             }
56             Thread.sleep(1);
57         }
58     }
59 
60 }
