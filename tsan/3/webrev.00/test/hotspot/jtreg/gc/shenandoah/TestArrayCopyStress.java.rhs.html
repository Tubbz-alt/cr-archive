25 import java.util.concurrent.*;
26 
27 /*
28  * @test TestArrayCopyStress
29  * @key gc
30  * @requires vm.gc.Shenandoah & !vm.graal.enabled
31  *
32  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -XX:TieredStopAtLevel=0 -Xmx16m TestArrayCopyStress
33  */
34 public class TestArrayCopyStress {
35 
36     private static final int ARRAY_SIZE = 1000;
37     private static final int ITERATIONS = 10000;
38 
39     static class Foo {
40         int num;
41 
42         Foo(int num) {
43             this.num = num;
44         }
45     }
46 
47     static class Bar {}
48 
49     public static void main(String[] args) throws Exception {
50         for (int i = 0; i < ITERATIONS; i++) {
51             testConjoint();
52         }
53     }
54 
55     private static void testConjoint() {
56         Foo[] array = new Foo[ARRAY_SIZE];
57         for (int i = 0; i < ARRAY_SIZE; i++) {
58             array[i] = new Foo(i);
59         }
60 
61         int src_idx = ThreadLocalRandom.current().nextInt(0, ARRAY_SIZE);
62         int dst_idx = ThreadLocalRandom.current().nextInt(0, ARRAY_SIZE);
63         int len = ThreadLocalRandom.current().nextInt(0, Math.min(ARRAY_SIZE - src_idx, ARRAY_SIZE - dst_idx));
64         System.arraycopy(array, src_idx, array, dst_idx, len);
65 
66         for (int i = 0; i < ARRAY_SIZE; i++) {
67             if (i >= dst_idx && i < dst_idx + len) {
68                 assertEquals(array[i].num, i - (dst_idx - src_idx));
69             } else {
70                 assertEquals(array[i].num, i);
71             }
72         }
73     }
74 
75     private static void assertEquals(int a, int b) {
76         if (a != b) throw new RuntimeException("assert failed");
77     }
78 
79 }
