 38 import java.util.Random;
 39 
 40 public class TestHumongousThreshold {
 41 
 42     static final long TARGET_MB = Long.getLong("target", 20_000); // 20 Gb allocation
 43 
 44     static volatile Object sink;
 45 
 46     public static void main(String[] args) throws Exception {
 47         final int min = 0;
 48         final int max = 384 * 1024;
 49         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
 50 
 51         Random r = new Random();
 52         for (long c = 0; c < count; c++) {
 53             sink = new int[min + r.nextInt(max - min)];
 54         }
 55     }
