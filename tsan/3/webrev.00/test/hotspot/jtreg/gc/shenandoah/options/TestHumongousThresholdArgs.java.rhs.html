25 /*
26  * @test TestHumongousThresholdArgs
27  * @summary Test that Shenandoah humongous threshold args are checked
28  * @key gc
29  * @requires vm.gc.Shenandoah & !vm.graal.enabled
30  * @library /test/lib
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  * @run driver TestHumongousThresholdArgs
34  */
35 
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class TestHumongousThresholdArgs {
40     public static void main(String[] args) throws Exception {
41         {
42             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockExperimentalVMOptions",
43                     "-XX:+UseShenandoahGC",
44                     "-version");
45             OutputAnalyzer output = new OutputAnalyzer(pb.start());
46             output.shouldHaveExitValue(0);
47         }
48 
49         int[] valid = new int[] {1, 10, 50, 90, 100};
50         int[] invalid = new int[] {-100, -1, 0, 101, 1000};
51 
52         for (int v : valid) {
53             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockExperimentalVMOptions",
54                     "-XX:+UseShenandoahGC",
55                     "-XX:ShenandoahHumongousThreshold=" + v,
56                     "-version");
57             OutputAnalyzer output = new OutputAnalyzer(pb.start());
58             output.shouldHaveExitValue(0);
59         }
60 
61         for (int v : invalid) {
62             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UnlockExperimentalVMOptions",
63                     "-XX:+UseShenandoahGC",
64                     "-XX:ShenandoahHumongousThreshold=" + v,
65                     "-version");
66             OutputAnalyzer output = new OutputAnalyzer(pb.start());
67             output.shouldHaveExitValue(1);
68         }
69     }
70 }
