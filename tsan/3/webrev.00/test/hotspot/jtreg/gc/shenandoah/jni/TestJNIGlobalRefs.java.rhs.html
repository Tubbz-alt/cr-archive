 36 /* @test TestJNIGlobalRefs
 37  * @summary Test JNI Global Refs with Shenandoah
 38  * @key gc
 39  * @requires vm.gc.Shenandoah & !vm.graal.enabled
 40  *
 41  * @run main/othervm/native -Xmx1g -Xlog:gc -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions
 42  *      -XX:+UseShenandoahGC -XX:ShenandoahGCHeuristics=aggressive
 43  *      TestJNIGlobalRefs
 44  */
 45 
 46 import java.util.Arrays;
 47 import java.util.Random;
 48 
 49 public class TestJNIGlobalRefs {
 50     static {
 51         System.loadLibrary("TestJNIGlobalRefs");
 52     }
 53 
 54     private static final int TIME_MSEC = 120000;
 55     private static final int ARRAY_SIZE = 10000;
 56 
 57     private static native void makeGlobalRef(Object o);
 58     private static native void makeWeakGlobalRef(Object o);
 59     private static native Object readGlobalRef();
 60     private static native Object readWeakGlobalRef();
 61 
 62     public static void main(String[] args) throws Throwable {
 63         seedGlobalRef();
 64         seedWeakGlobalRef();
 65         long start = System.currentTimeMillis();
 66         long current = start;
 67         while (current - start < TIME_MSEC) {
 68             testGlobal();
 69             testWeakGlobal();
 70             Thread.sleep(1);
 71             current = System.currentTimeMillis();
 72         }
 73     }
 74 
 75     private static void seedGlobalRef() {
 76         int[] a = new int[ARRAY_SIZE];
 77         fillArray(a, 1337);
 78         makeGlobalRef(a);
 79     }
 80 
 81     private static void seedWeakGlobalRef() {
 82         int[] a = new int[ARRAY_SIZE];
 83         fillArray(a, 8080);
 84         makeWeakGlobalRef(a);
 85     }
 86 
 87     private static void testGlobal() {
 88         int[] a = (int[]) readGlobalRef();
 89         checkArray(a, 1337);
 90     }
 91 
 92     private static void testWeakGlobal() {
 93         int[] a = (int[]) readWeakGlobalRef();
 94         if (a != null) {
 95             checkArray(a, 8080);
 96         } else {
 97             // weak reference is cleaned, recreate:
 98             seedWeakGlobalRef();
 99         }
100     }
101 
102     private static void fillArray(int[] array, int seed) {
103         Random r = new Random(seed);
104         for (int i = 0; i < ARRAY_SIZE; i++) {
105             array[i] = r.nextInt();
106         }
107     }
108 
109     private static void checkArray(int[] array, int seed) {
110         Random r = new Random(seed);
111         if (array.length != ARRAY_SIZE) {
112             throw new IllegalStateException("Illegal array length: " + array.length + ", but expected " + ARRAY_SIZE);
113         }
114         for (int i = 0; i < ARRAY_SIZE; i++) {
115             int actual = array[i];
116             int expected = r.nextInt();
117             if (actual != expected) {
118                 throw new IllegalStateException("Incorrect array data: " + actual + ", but expected " + expected);
119             }
120         }
121     }
122 }
