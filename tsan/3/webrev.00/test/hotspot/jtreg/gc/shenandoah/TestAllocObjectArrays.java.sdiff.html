 54 public class TestAllocObjectArrays {
 55 
 56     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation
 57 
 58     static volatile Object sink;
 59 
 60     public static void main(String[] args) throws Exception {
 61         final int min = 0;
 62         final int max = 384 * 1024;
 63         long count = TARGET_MB * 1024 * 1024 / (16 + 4 * (min + (max - min) / 2));
 64 
 65         Random r = new Random();
 66         for (long c = 0; c < count; c++) {
 67             sink = new Object[min + r.nextInt(max - min)];
 68         }
 69     }
