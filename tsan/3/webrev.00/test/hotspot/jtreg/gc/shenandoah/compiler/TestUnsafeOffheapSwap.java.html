 24 /**
 25  * @test TestUnsafeOffheapSwap
 26  * @summary Miscompilation in Unsafe off-heap swap routines
 27  * @key gc
 28  * @requires vm.gc.Shenandoah & !vm.graal.enabled
 29  * @modules java.base/jdk.internal.misc:+open
 30  *
 31  * @run main/othervm -XX:-UseOnStackReplacement -XX:-BackgroundCompilation -XX:-TieredCompilation
 32  *                   -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC
 33  *                   TestUnsafeOffheapSwap
 34  */
 35 
 36 import java.util.*;
 37 import jdk.internal.misc.Unsafe;
 38 
 39 public class TestUnsafeOffheapSwap {
 40 
 41     static final int SIZE = 10000;
 42     static final long SEED = 1;
 43 
 44     static final jdk.internal.misc.Unsafe UNSAFE = Unsafe.getUnsafe();
 45     static final int SCALE = UNSAFE.ARRAY_INT_INDEX_SCALE;
 46 
 47     static Memory mem;
 48     static int[] arr;
 49 
 50     public static void main(String[] args) throws Exception {
 51         // Bug is exposed when memory.addr is not known statically
 52         mem = new Memory(SIZE*SCALE);
 53         arr = new int[SIZE];
 54 
 55         for (int i = 0; i < 10; i++) {
 56             test();
 57         }
 58     }
 59 
 60     static void test() {
 61         Random rnd = new Random(SEED);
 62         for (int i = 0; i < SIZE; i++) {
 63             int value = rnd.nextInt();
 64             mem.setInt(i, value);
 65             arr[i] = value;
 66         }
 67 
 68         for (int i = 0; i < SIZE; i++) {
 69             if (arr[i] != mem.getInt(i)) {
 70                 throw new IllegalStateException("TESTBUG: Values mismatch before swaps");
 71             }
 72         }
 73 
 74         for (int i = 1; i < SIZE; i++) {
 75             mem.swap(i - 1, i);
 76             int tmp = arr[i - 1];
 77             arr[i - 1] = arr[i];
 78             arr[i] = tmp;
 79         }
 80 
 81         for (int i = 0; i < SIZE; i++) {
 82             if (arr[i] != mem.getInt(i)) {
 83                 throw new IllegalStateException("Values mismatch after swaps");
 84             }
 85         }
 86     }
 87 
 88     static class Memory {
 89         private final long addr;
 90 
 91         Memory(int size) {
 92             addr = UNSAFE.allocateMemory(size);
 93         }
 94 
 95         public int getInt(int idx) {
 96             return UNSAFE.getInt(addr + idx*SCALE);
 97         }
 98 
 99         public void setInt(int idx, int val) {
100             UNSAFE.putInt(addr + idx*SCALE, val);
101         }
102 
103         public void swap(int a, int b) {
104             int tmp = getInt(a);
105             setInt(a, getInt(b));
106             setInt(b, tmp);
107         }
108     }
109 }
