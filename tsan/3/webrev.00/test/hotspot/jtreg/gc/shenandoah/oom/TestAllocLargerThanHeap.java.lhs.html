24 /**
25  * @test TestAllocLargerThanHeap
26  * @summary Test that allocation of the object larger than heap fails predictably
27  * @key gc
28  * @requires vm.gc.Shenandoah
29  * @library /test/lib
30  * @run main TestAllocLargerThanHeap
31  */
32 
33 import jdk.test.lib.process.OutputAnalyzer;
34 import jdk.test.lib.process.ProcessTools;
35 
36 public class TestAllocLargerThanHeap {
37 
38     static final int SIZE = 16 * 1024 * 1024;
39 
40     static volatile Object sink;
41 
42     public static void work() throws Exception {
43         sink = new Object[SIZE];
44     }
45 
46     public static void main(String[] args) throws Exception {
47         if (args.length > 0) {
48             work();
49             return;
50         }
51 
52         {
53             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
54                     "-Xmx16m",
55                     "-XX:+UnlockExperimentalVMOptions",
56                     "-XX:+UseShenandoahGC",
57                     TestAllocLargerThanHeap.class.getName(),
58                     "test");
59 
60             OutputAnalyzer analyzer = new OutputAnalyzer(pb.start());
61             analyzer.shouldHaveExitValue(1);
62             analyzer.shouldContain("java.lang.OutOfMemoryError: Java heap space");
63         }
64 
65         {
66             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
67                     "-Xmx1g",
68                     "-XX:+UnlockExperimentalVMOptions",
69                     "-XX:+UseShenandoahGC",
70                     TestAllocLargerThanHeap.class.getName(),
71                     "test");
72 
73             OutputAnalyzer analyzer = new OutputAnalyzer(pb.start());
74             analyzer.shouldHaveExitValue(0);
75             analyzer.shouldNotContain("java.lang.OutOfMemoryError: Java heap space");
76         }
77     }
78 }
