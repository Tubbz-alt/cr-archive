24 #include <jni.h>
25 #include <string.h>
26 
27 static jint* pinned;
28 
29 JNIEXPORT void JNICALL
30 Java_TestPinnedGarbage_pin(JNIEnv *env, jclass unused, jintArray a) {
31   pinned = (*env)->GetPrimitiveArrayCritical(env, a, 0);
32 }
33 
34 JNIEXPORT void JNICALL
35 Java_TestPinnedGarbage_unpin(JNIEnv *env, jclass unused, jintArray a) {
36   (*env)->ReleasePrimitiveArrayCritical(env, a, pinned, 0);
37 }
