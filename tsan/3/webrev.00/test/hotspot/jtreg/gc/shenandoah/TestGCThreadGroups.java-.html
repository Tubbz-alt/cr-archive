46 
47 public class TestGCThreadGroups {
48 
49     static final long TARGET_MB = Long.getLong("target", 10_000); // 10 Gb allocation, around 1K cycles to handle
50     static final long STRIDE = 100_000;
51 
52     static volatile Object sink;
53 
54     public static void main(String[] args) throws Exception {
55         long count = TARGET_MB * 1024 * 1024 / 16;
56         for (long c = 0; c < count; c += STRIDE) {
57             for (long s = 0; s < STRIDE; s++) {
58                 sink = new Object();
59             }
60             Thread.sleep(1);
61         }
62     }
63 
64 }
