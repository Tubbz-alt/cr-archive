 48  * @comment Graal does not support CMS

 53 class GCBeanDescription {
 54     public String name;
 55     public String[] poolNames;
 56 
 57     public GCBeanDescription(String name, String[] poolNames) {
 58         this.name = name;
 59         this.poolNames = poolNames;
 60     }
 61 }
 62 
 63 public class TestMemoryMXBeansAndPoolsPresence {
 64     public static void test(GCBeanDescription... expectedBeans) {
 65         List<MemoryPoolMXBean> memoryPools = ManagementFactory.getMemoryPoolMXBeans();
 66 
 67         List<GarbageCollectorMXBean> gcBeans = ManagementFactory.getGarbageCollectorMXBeans();
 68         assertEQ(expectedBeans.length, gcBeans.size());
 69 
 70         for (GCBeanDescription desc : expectedBeans) {

 73                                                         .collect(Collectors.toList());
 74             assertEQ(beans.size(), 1);
 75 
 76             GarbageCollectorMXBean bean = beans.get(0);
 77             assertEQ(desc.name, bean.getName());
 78 
 79             String[] pools = bean.getMemoryPoolNames();
 80             assertEQ(desc.poolNames.length, pools.length);
 81             for (int i = 0; i < desc.poolNames.length; i++) {
 82                 assertEQ(desc.poolNames[i], pools[i]);
 83             }
 84         }
 85     }
 86 
 87     public static void main(String[] args) {
 88         switch (args[0]) {
 89             case "G1":
 90                 test(new GCBeanDescription("G1 Young Generation", new String[] {"G1 Eden Space", "G1 Survivor Space", "G1 Old Gen"}),
 91                      new GCBeanDescription("G1 Old Generation",   new String[] {"G1 Eden Space", "G1 Survivor Space", "G1 Old Gen"}));
 92                 break;
 93             case "CMS":
 94                 test(new GCBeanDescription("ParNew",              new String[] {"Par Eden Space", "Par Survivor Space"}),
 95                      new GCBeanDescription("ConcurrentMarkSweep", new String[] {"Par Eden Space", "Par Survivor Space", "CMS Old Gen"}));
 96                 break;
 97             case "Parallel":
 98                 test(new GCBeanDescription("PS Scavenge",         new String[] {"PS Eden Space", "PS Survivor Space"}),
 99                      new GCBeanDescription("PS MarkSweep",        new String[] {"PS Eden Space", "PS Survivor Space", "PS Old Gen"}));
100                 break;
101             case "Serial":
102                 test(new GCBeanDescription("Copy",              new String[] {"Eden Space", "Survivor Space"}),
103                      new GCBeanDescription("MarkSweepCompact",  new String[] {"Eden Space", "Survivor Space", "Tenured Gen"}));
104                 break;
105             default:
106                 assertTrue(false);
107                 break;
108 
109         }
110     }
111 }
