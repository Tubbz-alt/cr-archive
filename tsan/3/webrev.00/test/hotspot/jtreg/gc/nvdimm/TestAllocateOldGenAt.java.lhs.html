24 package gc.nvdimm;
25 
26 /* @test TestAllocateOldGenAt.java
27  * @key gc
28  * @summary Test to check allocation of Java Heap with AllocateOldGenAt option
29  * @requires vm.gc=="null" & os.family != "aix"
30  * @requires test.vm.gc.nvdimm
31  * @library /test/lib
32  * @modules java.base/jdk.internal.misc
33  * @run main gc.nvdimm.TestAllocateOldGenAt
34  */
35 
36 import jdk.test.lib.JDKToolFinder;
37 import jdk.test.lib.process.ProcessTools;
38 import jdk.test.lib.process.OutputAnalyzer;
39 import java.util.ArrayList;
40 import java.util.Collections;
41 
42 public class TestAllocateOldGenAt {
43   private static ArrayList<String> commonOpts;
44 
45   public static void main(String args[]) throws Exception {
46     commonOpts = new ArrayList<>();
47 
48     String testVmOptsStr = System.getProperty("test.java.opts");
49     if (!testVmOptsStr.isEmpty()) {
50       String[] testVmOpts = testVmOptsStr.split(" ");
51       Collections.addAll(commonOpts, testVmOpts);
52     }
53     String test_dir = System.getProperty("test.dir", ".");
54     Collections.addAll(commonOpts, new String[] {"-XX:+UnlockExperimentalVMOptions",
55                                                  "-XX:AllocateOldGenAt=" + test_dir,
56                                                  "-Xmx32m",
57                                                  "-Xms32m",
58                                                  "-version"});
59 
60     runTest("-XX:+UseG1GC");
61     runTest("-XX:+UseParallelOldGC -XX:-UseAdaptiveGCBoundary");
62     runTest("-XX:+UseParallelOldGC -XX:+UseAdaptiveGCBoundary");
63   }
64 
65   private static void runTest(String... extraFlags) throws Exception {
66     ArrayList<String> testOpts = new ArrayList<>();
67     Collections.addAll(testOpts, commonOpts.toArray(new String[commonOpts.size()]));
68     Collections.addAll(testOpts, extraFlags);
69 
70     System.out.print("Testing:\n" + JDKToolFinder.getJDKTool("java"));
71     for (int i = 0; i < testOpts.size(); i += 1) {
72       System.out.print(" " + testOpts.get(i));
73     }
74     System.out.println();
75 
76     ProcessBuilder pb =
77       ProcessTools.createJavaProcessBuilder(testOpts.toArray(new String[testOpts.size()]));
78     OutputAnalyzer output = new OutputAnalyzer(pb.start());
79 
80     output.shouldHaveExitValue(0);
81 
82   }
83 }
