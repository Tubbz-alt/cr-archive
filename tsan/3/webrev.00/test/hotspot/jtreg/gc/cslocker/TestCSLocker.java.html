 24 package gc.cslocker;
 25 
 26 import static gc.testlibrary.Allocation.blackHole;
 27 
 28 /*
 29  * @test TestCSLocker
 30  * @key gc
 31  * @bug 6186200
 32  * @library /
 33  * @summary This short test check RFE 6186200 changes. One thread locked
 34  * @summary completely in JNI CS, while other is trying to allocate memory
 35  * @summary provoking GC. OOM means FAIL, deadlock means PASS.
 36  * @run main/native/othervm -Xmx256m gc.cslocker.TestCSLocker
 37  */
 38 
 39 public class TestCSLocker extends Thread
 40 {
 41     static int timeout = 5000;
 42     public static void main(String args[]) throws Exception {
 43         long startTime = System.currentTimeMillis();
 44 
 45         // start garbage producer thread
 46         GarbageProducer garbageProducer = new GarbageProducer(1000000, 10);
 47         garbageProducer.start();
 48 
 49         // start CS locker thread
 50         CSLocker csLocker = new CSLocker();
 51         csLocker.start();
 52 
 53         // check timeout to success deadlocking
 54         while(System.currentTimeMillis() < startTime + timeout) {
 55             System.out.println("sleeping...");
 56             sleep(1000);
 57         }
 58 
 59         csLocker.unlock();
 60         garbageProducer.interrupt();
 61     }
 62 }
 63 
 64 class GarbageProducer extends Thread
 65 {
 66     private int size;
 67     private int sleepTime;
 68 
 69     GarbageProducer(int size, int sleepTime) {
 70         this.size = size;
 71         this.sleepTime = sleepTime;
 72     }
 73 
 74     public void run() {
 75         boolean isRunning = true;
 76 
 77         while (isRunning) {
 78             try {
 79                 blackHole(new int[size]);
 80                 sleep(sleepTime);
 81             } catch (InterruptedException e) {
 82                 isRunning = false;
 83             }
 84         }
 85     }
 86 }
 87 
 88 class CSLocker extends Thread
 89 {
 90     static { System.loadLibrary("TestCSLocker"); }
 91 
 92     public void run() {
 93         int[] a = new int[10];
 94         a[0] = 1;
 95         if (!lock(a)) {
 96             throw new RuntimeException("failed to acquire CSLock");
 97         }
 98     }
 99 
100     native boolean lock(int[] array);
101     native void unlock();
102 }
