 24 package gc.testlibrary.g1;
 25 
 26 import static java.util.stream.IntStream.range;
 27 import static jdk.test.lib.Asserts.assertTrue;
 28 import static sun.hotspot.WhiteBox.getWhiteBox;
 29 
 30 import java.util.ArrayList;
 31 import java.util.List;
 32 
 33 import gc.testlibrary.Helpers;
 34 import jdk.test.lib.Asserts;
 35 
 36 /**
 37  * Utility class to reliably provoke a mixed GC. The class allocates several arrays and
 38  * promotes them to the old generation. All GCs are triggered by Whitebox to ensure that they occur.
 39  * Must use G1.
 40  */
 41 public class MixedGCProvoker {
 42 
 43     /**
 44      * Allocate a few objects that are supposed to end up in the old generation as they are held live by the
 45      * given array. Caller must make sure by running this test with e.g. -XX:+AlwaysTenure that the objects
 46      * to actually be tenured.
 47      * Mixes live and dead objects, allocating about two regions worth of objects.
 48      * @param liveOldObjects Array receiving the live objects after this method.
 49      * @param g1HeapRegionSize Size of a G1 heap region.
 50      * @param arraySize Size of the byte arrays to use.
 51      */
 52     public static void allocateOldObjects(
 53             List<byte[]> liveOldObjects,
 54             int g1HeapRegionSize,
 55             int arraySize) {
 56 
 57         var toUnreachable = new ArrayList<byte[]>();
 58 
 59         // Allocates buffer and promotes it to the old gen. Mix live and dead old objects.
 60         // allocate about two regions of old memory. At least one full old region will guarantee
 61         // mixed collection in the future
 62         range(0, g1HeapRegionSize/arraySize).forEach(n -> {
 63             liveOldObjects.add(new byte[arraySize]);
 64             toUnreachable.add(new byte[arraySize]);
 65         });
 66 
 67         // Do one young collection, AlwaysTenure will force promotion.
 68         getWhiteBox().youngGC();
 69 
 70         // Check it is promoted & keep alive
 71         Asserts.assertTrue(getWhiteBox().isObjectInOldGen(liveOldObjects), "List of the objects is suppose to be in OldGen");
 72         Asserts.assertTrue(getWhiteBox().isObjectInOldGen(toUnreachable), "List of the objects is suppose to be in OldGen");
 73     }
 74 
 75     /**
 76      * Provoke at least one mixed gc by starting a marking cycle, waiting for its end and triggering two GCs.
 77      * @param liveOldObjects The objects supposed to survive this marking cycle.
 78      */
 79     public static void provokeMixedGC(List<byte[]> liveOldObjects) {
 80         Helpers.waitTillCMCFinished(getWhiteBox(), 10);
 81         getWhiteBox().g1StartConcMarkCycle();
 82         Helpers.waitTillCMCFinished(getWhiteBox(), 10);
 83         getWhiteBox().youngGC(); // the "Prepare Mixed" gc
 84         getWhiteBox().youngGC(); // the "Mixed" gc
 85 
 86         // check that liveOldObjects still alive
 87         assertTrue(getWhiteBox().isObjectInOldGen(liveOldObjects), "List of the objects is suppose to be in OldGen");
 88     }
 89 
 90     /**
 91      * Provoke a mixed GC on a G1 heap with the given heap region size.
 92      * @param g1HeapRegionSize The size of your regions in bytes
 93      */
 94     public static void allocateAndProvokeMixedGC(int g1HeapRegionSize) {
 95         final var arraySize = 20_000;
 96         var liveOldObjects = new ArrayList<byte[]>();
 97 
 98         getWhiteBox().fullGC(); // Make sure the heap is in a known state.
 99         allocateOldObjects(liveOldObjects, g1HeapRegionSize, arraySize);
100         provokeMixedGC(liveOldObjects);
101     }
102 }
