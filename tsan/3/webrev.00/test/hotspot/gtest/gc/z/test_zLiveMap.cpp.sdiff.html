28 class ZLiveMapTest : public ::testing::Test {
29 protected:

31     // Large ZPages only have room for one object.
32     ZLiveMap livemap(1);
33 
34     bool inc_live;
35     uintptr_t object = 0u;
36 
37     // Mark the object strong.

39

42
