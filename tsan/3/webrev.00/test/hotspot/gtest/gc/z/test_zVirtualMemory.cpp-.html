24 #include "precompiled.hpp"
25 #include "gc/z/zVirtualMemory.inline.hpp"
26 #include "utilities/debug.hpp"
27 #include "unittest.hpp"
28 
29 TEST(ZVirtualMemory, split) {
30   const size_t PageSize = 2 * M;
31 
32   ZVirtualMemory mem(0, 10 * PageSize);
33 
34   ZVirtualMemory mem_split0 = mem.split(0 * PageSize);
35   EXPECT_EQ(mem_split0.size(),  0 * PageSize);
36   EXPECT_EQ(       mem.size(), 10 * PageSize);
37 
38   ZVirtualMemory mem_split1 = mem.split(5u * PageSize);
39   EXPECT_EQ(mem_split1.size(),  5 * PageSize);
40   EXPECT_EQ(       mem.size(),  5 * PageSize);
41 
42   ZVirtualMemory mem_split2 = mem.split(5u * PageSize);
43   EXPECT_EQ(mem_split2.size(),  5 * PageSize);
44   EXPECT_EQ(       mem.size(),  0 * PageSize);
45 
46   ZVirtualMemory mem_split3 = mem.split(0 * PageSize);
47   EXPECT_EQ(mem_split3.size(),  0 * PageSize);
48 }
