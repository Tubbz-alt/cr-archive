24 #include "precompiled.hpp"
25 #include "gc/z/zVirtualMemory.inline.hpp"
26 #include "unittest.hpp"
27 
28 TEST(ZVirtualMemory, split) {
29   ZVirtualMemory vmem(0, 10);
30 
31   ZVirtualMemory vmem0 = vmem.split(0);
32   EXPECT_EQ(vmem0.size(), 0u);
33   EXPECT_EQ(vmem.size(), 10u);
34 
35   ZVirtualMemory vmem1 = vmem.split(5);
36   EXPECT_EQ(vmem1.size(), 5u);
37   EXPECT_EQ(vmem.size(), 5u);
38 
39   ZVirtualMemory vmem2 = vmem.split(5);
40   EXPECT_EQ(vmem2.size(), 5u);
41   EXPECT_EQ(vmem.size(), 0u);
42 
43   ZVirtualMemory vmem3 = vmem.split(0);
44   EXPECT_EQ(vmem3.size(), 0u);
45 }
