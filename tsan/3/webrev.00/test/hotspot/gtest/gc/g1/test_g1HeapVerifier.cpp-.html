25 #include "precompiled.hpp"
26 #include "gc/g1/g1Arguments.hpp"
27 #include "gc/g1/g1HeapVerifier.hpp"
28 #include "logging/logConfiguration.hpp"
29 #include "logging/logTestFixture.hpp"
30 #include "unittest.hpp"
31 
32 class G1HeapVerifierTest : public LogTestFixture {
33   protected:
34     static void parse_verification_type(const char* type) {
35       G1Arguments::parse_verification_type(type);
36     }
37 };
38 
39 TEST_VM_F(G1HeapVerifierTest, parse) {
40   LogConfiguration::configure_stdout(LogLevel::Off, true, LOG_TAGS(gc, verify));
41 
42   // Default is to verify everything.
43   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyAll));
44   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyYoungNormal));
45   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyConcurrentStart));
46   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyMixed));
47   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyRemark));
48   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyCleanup));
49   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyFull));
50 
51   // Setting one will disable all other.
52   G1HeapVerifierTest::parse_verification_type("full");
53   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyAll));
54   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyYoungNormal));
55   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyConcurrentStart));
56   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyMixed));
57   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyRemark));
58   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyCleanup));
59   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyFull));
60 
61   // Verify case sensitivity.
62   G1HeapVerifierTest::parse_verification_type("YOUNG-NORMAL");
63   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyYoungNormal));
64   G1HeapVerifierTest::parse_verification_type("young-normal");
65   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyYoungNormal));
66 
67   // Verify perfect match
68   G1HeapVerifierTest::parse_verification_type("mixedgc");
69   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyMixed));
70   G1HeapVerifierTest::parse_verification_type("mixe");
71   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyMixed));
72   G1HeapVerifierTest::parse_verification_type("mixed");
73   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyMixed));
74 
75   // Verify the last three
76   G1HeapVerifierTest::parse_verification_type("concurrent-start");
77   G1HeapVerifierTest::parse_verification_type("remark");
78   G1HeapVerifierTest::parse_verification_type("cleanup");
79   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyRemark));
80   ASSERT_TRUE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyCleanup));
81 
82   // Enabling all is not the same as G1VerifyAll
83   ASSERT_FALSE(G1HeapVerifier::should_verify(G1HeapVerifier::G1VerifyAll));
84 }
