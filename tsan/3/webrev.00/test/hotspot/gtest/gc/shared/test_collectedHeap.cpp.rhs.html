24 #include "precompiled.hpp"
25 #include "gc/shared/collectedHeap.hpp"
26 #include "memory/universe.hpp"
27 #include "unittest.hpp"

29 TEST_VM(CollectedHeap, is_in) {
30   CollectedHeap* heap = Universe::heap();
31 
32   uintptr_t epsilon = (uintptr_t) MinObjAlignment;
33   uintptr_t outside_heap = (uintptr_t) &epsilon;

34 
35   // Test that NULL is not in the heap.
36   ASSERT_FALSE(heap->is_in(NULL)) << "NULL is unexpectedly in the heap";
37 
38   // Test that a pointer to outside the heap start is reported as outside the heap.
39   ASSERT_FALSE(heap->is_in((void*)outside_heap)) << "outside_heap: " << outside_heap












40           << " is unexpectedly in the heap";
41 }
