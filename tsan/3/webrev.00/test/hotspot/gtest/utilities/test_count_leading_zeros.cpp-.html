25 #include "precompiled.hpp"
26 #include "utilities/count_leading_zeros.hpp"
27 #include "utilities/globalDefinitions.hpp"
28 #include "unittest.hpp"
29 
30 TEST(count_leading_zeros, one_or_two_set_bits) {
31   unsigned i = 0;                  // Position of a set bit.
32   for (uint32_t ix = 1; ix != 0; ix <<= 1, ++i) {
33     unsigned j = 0;                // Position of a set bit.
34     for (uint32_t jx = 1; jx != 0; jx <<= 1, ++j) {
35       uint32_t value = ix | jx;
36       EXPECT_EQ(31u - MAX2(i, j), count_leading_zeros(value))
37         << "value = " << value;
38     }
39   }
40 }
41 
42 TEST(count_leading_zeros, high_zeros_low_ones) {
43   unsigned i = 0;                  // Number of leading zeros
44   uint32_t value = ~(uint32_t)0;
45   for ( ; value != 0; value >>= 1, ++i) {
46     EXPECT_EQ(i, count_leading_zeros(value))
47       << "value = " << value;
48   }
49 }
50 
51 TEST(count_leading_zeros, high_ones_low_zeros) {
52   uint32_t value = ~(uint32_t)0;
53   for ( ; value != 0; value <<= 1) {
54     EXPECT_EQ(0u, count_leading_zeros(value))
55       << "value = " << value;
56   }
57 }
