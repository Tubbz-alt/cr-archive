30 #define TEST_FLAG(f, type, value)                                \
31   do {                                                           \
32     ASSERT_TRUE(JVMFlag::find_flag(#f)->is_ ## type());          \
33     type original_value = f;                                     \
34     {                                                            \
35       FLAG_GUARD(f);                                             \
36       f = value;                                                 \
37     }                                                            \
38     ASSERT_EQ(original_value, f);                                \
39   } while (0)
