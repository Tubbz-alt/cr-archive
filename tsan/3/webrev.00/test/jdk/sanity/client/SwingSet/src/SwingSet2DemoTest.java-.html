 48 /*
 49  * @test
 50  * @key headful
 51  * @summary Verifies check box menu item, radio button menu item, nested menus
 52  * and themes using SwingSet2 main window. Testing of other swing components
 53  * are covered in SwingSet3 demo tests.
 54  *
 55  * @library /sanity/client/lib/jemmy/src
 56  * @library /sanity/client/lib/Extensions/src
 57  * @library /sanity/client/lib/SwingSet2/src
 58  * @modules java.desktop
 59  *          java.logging
 60  * @build org.jemmy2ext.JemmyExt
 61  * @build SwingSet2
 62  * @run testng/timeout=600 SwingSet2DemoTest
 63  */
 64 @Listeners(GuiTestListener.class)
 65 public class SwingSet2DemoTest {
 66 
 67     private static final String OCEAN_THEME_NAME = "Ocean";
 68     private static final String STEEL_THEME_NAME = "Steel";
 69     private static final int TOOLTIP_DISMISS_DELAY = 60000;
 70 
 71     /**
 72      * Testing check box menu item, radio button menu item, nested menus and
 73      * themes. Testing of all other main swing components are covered in
 74      * SwingSet3 demo tests.
 75      *
 76      * @throws Exception
 77      */
 78     @Test(dataProvider = "availableLookAndFeels", dataProviderClass = TestHelpers.class)
 79     public void test(String lookAndFeel) throws Exception {
 80         UIManager.setLookAndFeel(lookAndFeel);
 81 
 82         new ClassReference(SwingSet2.class.getCanonicalName()).startApplication();
 83         JFrameOperator frameOperator = new JFrameOperator(SwingSet2.FRAME_TITLE);
 84         frameOperator.setComparator(EXACT_STRING_COMPARATOR);
 85 
 86         checkCheckBoxMenuItem(frameOperator);
 87         JMenuOperator themesMenu = new JMenuOperator(frameOperator, SwingSet2.THEMES_MENU_TEXT);
 88         // Themes menu will be enabled only on MetalLookAndFeel
 89         if(themesMenu.isEnabled()) {
 90             checkMenuOnMenuAndRadioButtonMenuItem(frameOperator, themesMenu);
 91             checkThemes(frameOperator, themesMenu);
 92         }
 93     }
 94 
 95     /**
 96      * Verifies the properties of nested menu and radio button menu item
 97      * @param frameOperator
 98      * @param themesMenu
 99      */
100     private void checkMenuOnMenuAndRadioButtonMenuItem(
101             JFrameOperator frameOperator, JMenuOperator themesMenu) {
102 
103         themesMenu.push();
104         themesMenu.waitSelected(true);
105         // Font menu is a nested menu, selecting that and verifying its
106         // sub menu items are launched
107         JMenuOperator fontMenu = new JMenuOperator(frameOperator, SwingSet2.FONT_MENU_TEXT);
108         JRadioButtonMenuItemOperator fontPlainMenuItem = new JRadioButtonMenuItemOperator(
109                 (JRadioButtonMenuItem)fontMenu.showMenuItem(
110                         SwingSet2.FONT_PLAIN_MENU_TEXT, "/").getSource());
111         JRadioButtonMenuItemOperator fontBoldMenuItem = new JRadioButtonMenuItemOperator(
112                 (JRadioButtonMenuItem)fontMenu.showMenuItem(
113                         SwingSet2.FONT_BOLD_MENU_TEXT, "/").getSource());
114         JButtonOperator testComp =
115                 new JButtonOperator(frameOperator, OptionPaneDemo.INPUT_BUTTON);
116         checkRadioButtonMenuItem(fontBoldMenuItem, fontPlainMenuItem, testComp,
117                 component -> component.getFont().isBold());
118         checkRadioButtonMenuItem(fontPlainMenuItem, fontBoldMenuItem, testComp,
119                 component -> component.getFont().isPlain());
120     }
121 
122     /**
123      * Verifies the properties of radio button menu item
124      * @param pressMenuItem
125      * @param unPressMenuItem
126      * @param testComp
127      * @param validator
128      */
129     private void checkRadioButtonMenuItem(JRadioButtonMenuItemOperator pressMenuItem,
130             JRadioButtonMenuItemOperator unPressMenuItem,
131             ComponentOperator testComp, ComponentChooser validator){
132 
133         // Selecting a radio button menu item and verifying the pressed menu
134         // item is selected, other one is unselected and corresponding ui
135         // changes also happened
136         pressMenuItem.push();
137         pressMenuItem.waitSelected(true);
138         unPressMenuItem.waitSelected(false);
139         testComp.waitStateOnQueue(validator);
140     }
141 
142     /**
143      * Verifies the properties of check box menu item
144      * @param frameOperator
145      */
146     private void checkCheckBoxMenuItem(JFrameOperator frameOperator) {
147 
148         ToolTipManager.sharedInstance().setDismissDelay(TOOLTIP_DISMISS_DELAY);
149         JToggleButtonOperator testComp = new JToggleButtonOperator(
150                 frameOperator, new NameComponentChooser(SwingSet2.getString(
151                         OptionPaneDemo.DEMO_NAME + SwingSet2.NAME_PROP_SUFFIX)));
152         JMenuOperator optionsMenu = new JMenuOperator(frameOperator, SwingSet2.OPTIONS_MENU_TEXT);
153         JCheckBoxMenuItemOperator toolTipMenuItem = new JCheckBoxMenuItemOperator(
154                 (JCheckBoxMenuItem)optionsMenu.showMenuItem(
155                         SwingSet2.TOOLTIP_MENU_TEXT, "/").getSource());
156         // Selecting and deselecting tooltip checkbox menu item and verifying
157         // tooltip is showing for demo toggle button
158         toolTipMenuItem.push();
159         toolTipMenuItem.waitSelected(false);
160         boolean isToolTipTimeout = false;
161         try {
162             testComp.showToolTip();
163         } catch (TimeoutExpiredException e) {
164             isToolTipTimeout = true;
165         }
166         assertTrue(isToolTipTimeout, "Tooltip is showing even after unchecking the checkbox menu"
167                 + " item 'Enable Tool Tips'");
168         toolTipMenuItem.push();
169         toolTipMenuItem.waitSelected(true);
170         testComp.showToolTip();
171     }
172 
173     /**
174      * Verifies the different themes by applying different themes
175      * @param frameOperator
176      * @param themesMenu
177      */
178     private void checkThemes(JFrameOperator frameOperator, JMenuOperator themesMenu) {
179         String themeMenuNames [] = {SwingSet2.OCEAN_MENU_TEXT, SwingSet2.AQUA_MENU_TEXT,
180                 SwingSet2.STEEL_MENU_TEXT, SwingSet2.CHARCOAL_MENU_TEXT,
181                 SwingSet2.CONTRAST_MENU_TEXT, SwingSet2.EMERALD_MENU_TEXT, SwingSet2.RUBY_MENU_TEXT};
182         String themeNames [] = {OCEAN_THEME_NAME, AquaTheme.NAME, STEEL_THEME_NAME,
183                 CharcoalTheme.NAME, ContrastTheme.NAME, EmeraldTheme.NAME, RubyTheme.NAME};
184 
185         for (int i = 0; i < themeMenuNames.length; i++) {
186             int themeIndex = i;
187             JRadioButtonMenuItemOperator menuItem = new JRadioButtonMenuItemOperator(
188                     (JRadioButtonMenuItem)themesMenu.showMenuItem(
189                             themeMenuNames[themeIndex], "/").getSource());
190             menuItem.push();
191             menuItem.waitSelected(true);
192             menuItem.waitStateOnQueue(comp -> themeNames[themeIndex].
193                     equals(MetalLookAndFeel.getCurrentTheme().getName()));
194         }
195     }
