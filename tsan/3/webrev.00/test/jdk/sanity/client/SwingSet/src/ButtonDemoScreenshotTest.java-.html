 33 import org.testng.annotations.Test;

 35 import java.awt.Component;
 36 import java.awt.Robot;
 37 import java.awt.image.BufferedImage;
 38 
 39 import javax.swing.UIManager;
 40 import static com.sun.swingset3.demos.button.ButtonDemo.DEMO_TITLE;
 41 import static org.jemmy2ext.JemmyExt.*;
 42 
 43 /*
 44  * @test
 45  * @key headful screenshots intermittent
 46  * @summary Verifies buttons on SwingSet3 ButtonDemo page by clicking each
 47  *          button, taking its screenshots and checking that pressed button
 48  *          image is different from initial button image.
 49  *
 50  * @library /sanity/client/lib/jemmy/src
 51  * @library /sanity/client/lib/Extensions/src
 52  * @library /sanity/client/lib/SwingSet3/src
 53  * @modules java.desktop
 54  *          java.logging
 55  * @build org.jemmy2ext.JemmyExt
 56  * @build com.sun.swingset3.demos.button.ButtonDemo
 57  * @run testng/timeout=600 ButtonDemoScreenshotTest
 58  */
 59 @Listeners(GuiTestListener.class)
 60 public class ButtonDemoScreenshotTest {
 61 
 62     private static final int[] BUTTONS = {0, 1, 2, 3, 4, 5}; // "open browser" buttons (6, 7) open a browser, so ignore
 63     private static StrictImageComparator sComparator = null;
 64 
 65     @BeforeClass
 66     public void init() {
 67         sComparator = new StrictImageComparator();
 68     }
 69 
 70     @Test(dataProvider = "availableLookAndFeels", dataProviderClass = TestHelpers.class)
 71     public void test(String lookAndFeel) throws Exception {
 72         UIManager.setLookAndFeel(lookAndFeel);
 73         Robot rob = new Robot();
 74 
 75         new ClassReference(ButtonDemo.class.getCanonicalName()).startApplication();
 76 
 77         JFrameOperator mainFrame = new JFrameOperator(DEMO_TITLE);
 78         waitImageIsStill(rob, mainFrame);
 79 
 80         // Check all the buttons
 81         for (int i : BUTTONS) {
 82             checkButton(mainFrame, i, rob);
 83         }
 84     }
 85 
 86     private void checkButton(JFrameOperator jfo, int i, Robot rob) {
 87         JButtonOperator button = new JButtonOperator(jfo, i);
 88 
 89         //additional instrumentation for JDK-8198920. To be removed after the bug is fixed
 90         java.util.concurrent.atomic.AtomicBoolean actionListenerCalled = new java.util.concurrent.atomic.AtomicBoolean(false);
 91         button.addActionListener(e -> actionListenerCalled.set(true));
 92         //end of instrumentation for JDK-8198920
 93 
 94         button.moveMouse(button.getCenterX(), button.getCenterY());
 95 
 96         BufferedImage initialButtonImage = capture(rob, button);
 97         assertNotBlack(initialButtonImage);
 98         save(initialButtonImage, "button" + i + ".png");
 99 
100         BufferedImage[] pressedImage = new BufferedImage[1];
101 
102         button.pressMouse();
103         //additional instrumentation for JDK-8198920. To be removed after the bug is fixed
104         button.getOutput().printTrace("JDK-8198920: Button pressed at " + System.currentTimeMillis());
105         //end of instrumentation for JDK-8198920
106         try {
107             waitPressed(button);
108             //additional instrumentation for JDK-8198920. To be removed after the bug is fixed
109             button.getOutput().printTrace("JDK-8198920: Button press confirmed by " + System.currentTimeMillis());
110             //end of instrumentation for JDK-8198920
111             button.waitState(new ComponentChooser() {
112                 public boolean checkComponent(Component c) {
113                     pressedImage[0] = capture(rob, button);
114                     assertNotBlack(pressedImage[0]);
115                     return !sComparator.compare(initialButtonImage, pressedImage[0]);
116                 }
117 
118                 public String getDescription() {
119                     return "Button with new image";
120                 }
121             });
122         } finally {
123             if (pressedImage[0] != null) {
124                 save(pressedImage[0], "button" + i + "_pressed.png");
125             }
126             button.releaseMouse();
127             //additional instrumentation for JDK-8198920. To be removed after the bug is fixed
128             button.getOutput().printTrace("JDK-8198920: Button released at " + System.currentTimeMillis());
129             try {
130                 button.waitState(comp -> actionListenerCalled.get());
131                 button.getOutput().printTrace("JDK-8198920: Action listener was called by " + System.currentTimeMillis());
132             } catch (org.netbeans.jemmy.TimeoutExpiredException e) {
133                 button.getOutput().printTrace("JDK-8198920: Action listener was not called by " + System.currentTimeMillis());
134             }
135             //end of instrumentation for JDK-8198920
136         }
137     }
138 }
