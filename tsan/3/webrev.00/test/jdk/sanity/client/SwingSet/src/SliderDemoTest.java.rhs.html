 44 /*
 45  * @test
 46  * @key headful
 47  * @summary Verifies SwingSet3 SliderDemo by moving the sliders through
 48  *  different means, checking the slider value corresponding to it,
 49  *  checking maximum and minimum values, checking Snap to tick is working
 50  *  and checking the presence of ticks and labels.
 51  *
 52  * @library /sanity/client/lib/jemmy/src
 53  * @library /sanity/client/lib/Extensions/src
 54  * @library /sanity/client/lib/SwingSet3/src
 55  * @modules java.desktop
 56  *          java.logging
 57  * @build org.jemmy2ext.JemmyExt
 58  * @build com.sun.swingset3.demos.slider.SliderDemo
 59  * @run testng/timeout=600 SliderDemoTest
 60  */
 61 @Listeners(GuiTestListener.class)
 62 public class SliderDemoTest {
 63 
 64     private static final int PLAIN_SLIDER_MINIMUM = -10;
 65     private static final int PLAIN_SLIDER_MAXIMUM = 100;
 66     private static final int HORIZONTAL_MINOR_TICKS_SLIDER_MINIMUM = 0;
 67     private static final int HORIZONTAL_MINOR_TICKS_SLIDER_MAXIMUM = 11;
 68     private static final int VERTICAL_MINOR_TICKS_SLIDER_MINIMUM = 0;
 69     private static final int VERTICAL_MINOR_TICKS_SLIDER_MAXIMUM = 100;

 70 
 71     @Test(dataProvider = "availableLookAndFeels", dataProviderClass = TestHelpers.class)
 72     public void test(String lookAndFeel) throws Exception {
 73         UIManager.setLookAndFeel(lookAndFeel);

 74         new ClassReference(SliderDemo.class.getCanonicalName()).startApplication();
 75         JFrameOperator frame = new JFrameOperator(DEMO_TITLE);
 76         plain(frame, HORIZONTAL_PLAIN_SLIDER);
 77         majorTicks(frame, HORIZONTAL_MAJOR_TICKS_SLIDER);
 78         minorTicks(frame, HORIZONTAL_MINOR_TICKS_SLIDER);
 79         disabled(frame, HORIZONTAL_DISABLED_SLIDER);
 80         plain(frame, VERTICAL_PLAIN_SLIDER);
 81         majorTicks(frame, VERTICAL_MAJOR_TICKS_SLIDER);
 82         minorTicks(frame, VERTICAL_MINOR_TICKS_SLIDER);
 83         disabled(frame, VERTICAL_DISABLED_SLIDER);
 84     }
 85 
 86     private void plain(JFrameOperator jfo, String accessibleName) {
 87         JSliderOperator jso = new JSliderOperator(jfo,
 88                 new AccessibleNameChooser(accessibleName));
 89         if (accessibleName.equals(HORIZONTAL_PLAIN_SLIDER)) {
 90             jso.waitHasFocus();
 91             checkKeyboard(jso);
 92             checkMouse(jso);
 93         }
 94         checkMaximum(jso, PLAIN_SLIDER_MAXIMUM);
 95         checkMinimum(jso, PLAIN_SLIDER_MINIMUM);
 96         checkMoveForward(jso, 10);
 97     }
 98 
 99     private void majorTicks(JFrameOperator jfo, String accessibleName) {
100         JSliderOperator jso = new JSliderOperator(jfo,
101                 new AccessibleNameChooser(accessibleName));
102         checkMoveForward(jso, 40);
103         assertTrue(jso.getPaintTicks());
104         assertEquals(100, jso.getMajorTickSpacing());
105     }
106 
107     private void minorTicks(JFrameOperator jfo, String accessibleName) {
108         JSliderOperator jso = new JSliderOperator(jfo,
109                 new AccessibleNameChooser(accessibleName));
110         if (accessibleName.equals(HORIZONTAL_MINOR_TICKS_SLIDER)) {
111             checkMaximum(jso, HORIZONTAL_MINOR_TICKS_SLIDER_MAXIMUM);
112             checkMinimum(jso, HORIZONTAL_MINOR_TICKS_SLIDER_MINIMUM);
113             checkMoveForward(jso, 2);
114             checkSnapToTick(jso, 5, 6);
115             assertEquals(5, jso.getMajorTickSpacing());
116             assertEquals(1, jso.getMinorTickSpacing());
117         } else {
118             checkMaximum(jso, VERTICAL_MINOR_TICKS_SLIDER_MAXIMUM);
119             checkMinimum(jso, VERTICAL_MINOR_TICKS_SLIDER_MINIMUM);
120             checkMoveForward(jso, 10);
121             assertEquals(20, jso.getMajorTickSpacing());
122             assertEquals(5, jso.getMinorTickSpacing());
123         }
124         assertTrue(jso.getPaintTicks());
125         assertTrue(jso.getPaintLabels());
126     }
127 
128     private void disabled(JFrameOperator jfo, String accessibleName)
129             throws InterruptedException {
130         JSliderOperator jso = new JSliderOperator(jfo,
131                 new AccessibleNameChooser(accessibleName));
132         int initialvalue;
133         initialvalue = jso.getValue();
134         jso.clickMouse(jso.getCenterXForClick(), jso.getCenterYForClick(), 10);
135         Thread.sleep(500);
136         assertFalse(jso.hasFocus());
137         assertEquals(initialvalue, jso.getValue());
138     }
139 
140     private void checkMaximum(JSliderOperator jso, int maxValue) {
141         jso.scrollToMaximum();
142         waitSliderValue(jso, jSlider -> jSlider.getValue() == maxValue);
143     }
144 
145     private void checkMinimum(JSliderOperator jso, int minValue) {
146         jso.scrollToMinimum();
147         waitSliderValue(jso, jSlider -> jSlider.getValue() == minValue);
148     }
149 
150     private void checkKeyboard(JSliderOperator jso) {
151         boolean isMotif = LookAndFeel.isMotif();
152         checkKeyPress(jso, KeyEvent.VK_HOME,
153                 jSlider -> jSlider.getValue() == jso.getMinimum());
154 
155         {
156             int expectedValue = jso.getValue() + 1;
157             checkKeyPress(jso, KeyEvent.VK_UP,
158                     jSlider -> jSlider.getValue() >= expectedValue);
159         }
160         {
161             int expectedValue = jso.getValue() + 1;
162             checkKeyPress(jso, KeyEvent.VK_RIGHT,
163                     jSlider -> jSlider.getValue() >= expectedValue);
164         }
165         if (!isMotif) {
166             int expectedValue = jso.getValue() + 11;
167             checkKeyPress(jso, KeyEvent.VK_PAGE_UP,
168                     jSlider -> jSlider.getValue() >= expectedValue);
169         }
170 
171         checkKeyPress(jso, KeyEvent.VK_END,
172                 jSlider -> jSlider.getValue() == jso.getMaximum());
173 
174         {
175             int expectedValue = jso.getValue() - 1;
176             checkKeyPress(jso, KeyEvent.VK_DOWN,
177                     jSlider -> jSlider.getValue() <= expectedValue);
178         }
179         {
180             int expectedValue = jso.getValue() - 1;
181             checkKeyPress(jso, KeyEvent.VK_LEFT,
182                     jSlider -> jSlider.getValue() <= expectedValue);
183         }
184         if (!isMotif) {
185             int expectedValue = jso.getValue() - 11;
186             checkKeyPress(jso, KeyEvent.VK_PAGE_DOWN,
187                     jSlider -> jSlider.getValue() <= expectedValue);
188         }
189     }
190 
191     private void checkKeyPress(JSliderOperator jso, int keyCode,
192             Predicate<JSliderOperator> predicate) {
193         jso.pushKey(keyCode);
194         waitSliderValue(jso, predicate);
195     }
196 
197     private void waitSliderValue(JSliderOperator jso,
198             Predicate<JSliderOperator> predicate) {
199         jso.waitState(new ComponentChooser() {
200             public boolean checkComponent(Component comp) {
201                 return predicate.test(jso);
202             }
203 
204             public String getDescription() {
205                 return "Wait till Slider attains the specified state.";
206             }
207         });
208     }
209 
210     private void checkMoveForward(JSliderOperator jso, int value) {
211         jso.setValue(jso.getMinimum());
212         int finalValue = jso.getValue() + value;
213         jso.scrollToValue(finalValue);
214         waitSliderValue(jso, jSlider -> jSlider.getValue() == finalValue);
215     }
216 
217     private void checkSnapToTick(JSliderOperator jso, int expectedLower,
218             int expectedHigher) {
219         jso.pressMouse(jso.getCenterXForClick(), jso.getCenterYForClick());
220         waitSliderValue(jso, jSlider -> jSlider.getValue() == expectedLower
221                 || jSlider.getValue() == expectedHigher);
222         jso.releaseMouse();
223     }
224 
225     private void checkMouse(JSliderOperator jso) {
226         // Check mouse dragging by pressing on the center of Slider and then
227         // dragging the mouse till the end of the track.
228         // We set the initial value of the slider as 45,
229         // which is the value of the slider at the middle.
230         jso.setValue((jso.getMaximum() + jso.getMinimum()) / 2);
231         jso.pressMouse(jso.getCenterXForClick(), jso.getCenterYForClick());
232         jso.dragMouse(jso.getWidth() + 10, jso.getHeight());
233         waitSliderValue(jso, jSlider -> jSlider.getValue() == jSlider.getMaximum());
234         jso.releaseMouse();
235 
236         // Check mouse click by clicking on the center of the track 2 times
237         // and waiting till the slider value has changed from its previous
238         // value as a result of the clicks.
239         jso.clickMouse(jso.getCenterXForClick(), jso.getCenterYForClick(), 2);
240         waitSliderValue(jso, jSlider -> jSlider.getValue() != jSlider.getMaximum());
241     }
