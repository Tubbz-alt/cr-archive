 59     public static void main(String[] args) {
 60         GraphicsEnvironment ge =
 61                 GraphicsEnvironment.getLocalGraphicsEnvironment();
 62         GraphicsDevice gd = ge.getDefaultScreenDevice();
 63         GraphicsConfiguration gc = gd.getDefaultConfiguration();
 64         testGC(gc);

 67             throw new RuntimeException("Test FAILED. See err output for more");
 68         }
 69

 73     private static void testInvalidType(AccelSurface surface, int type) {
 74         long ret = surface.getNativeResource(type);
 75         System.out.printf("  getNativeResource(%d)=0x%x\n", type, ret);
 76         if (ret != 0l) {
 77             System.err.printf(
 78                     "FAILED: surface.getNativeResource(%d) returned" +
 79                     " 0x%s. It should have have returned 0L\n",
 80                     type, ret);
 81             failed = true;
 82         }
 83     }

 85     private static void testD3DDeviceResourceField(final AccelSurface surface) {
 86         try {
 87             Class d3dc = Class.forName("sun.java2d.d3d.D3DSurfaceData");
 88             if (d3dc.isInstance(surface)) {
 89                 Field f = d3dc.getDeclaredField("D3D_DEVICE_RESOURCE");
 90                 f.setAccessible(true);
 91                 int d3dDR = (Integer)f.get(null);
 92 
 93                 System.out.printf(
 94                         "  getNativeResource(D3D_DEVICE_RESOURCE)=0x%x\n",
 95                         surface.getNativeResource(d3dDR));
 96             }
 97         } catch (ClassNotFoundException e) {}
 98         catch (IllegalAccessException e) {}
 99         catch (NoSuchFieldException e) {
100             System.err.println("Failed: D3DSurfaceData.D3D_DEVICE_RESOURCE" +
101                                " field not found!");
102             failed = true;
103         }
104     }

106     private static void printSurface(Surface s) {
107         if (s instanceof AccelSurface) {
108             final AccelSurface surface = (AccelSurface) s;
109             System.out.println(" Accel Surface: ");
110             System.out.println("  type=" + surface.getType());
111             System.out.println("  bounds=" + surface.getBounds());
112             System.out.println("  nativeBounds=" + surface.getNativeBounds());
113             System.out.println("  isSurfaceLost=" + surface.isSurfaceLost());
114             System.out.println("  isValid=" + surface.isValid());
115             RenderQueue rq = surface.getContext().getRenderQueue();
116             rq.lock();
117             try {
118                 rq.flushAndInvokeNow(new Runnable() {
119                     public void run() {
120                         System.out.printf("  getNativeResource(TEXTURE)=0x%x\n",
121                             surface.getNativeResource(TEXTURE));
122                         System.out.printf("  getNativeResource(RT_TEXTURE)=0x%x\n",
123                             surface.getNativeResource(RT_TEXTURE));
124                         System.out.printf("  getNativeResource(RT_PLAIN)=0x%x\n",
125                             surface.getNativeResource(RT_PLAIN));
126                         System.out.printf(
127                             "  getNativeResource(FLIP_BACKBUFFER)=0x%x\n",
128                             surface.getNativeResource(FLIP_BACKBUFFER));
129 
130                         testD3DDeviceResourceField(surface);
131 
132                         testInvalidType(surface, -1);
133                         testInvalidType(surface, -150);
134                         testInvalidType(surface, 300);
135                         testInvalidType(surface, Integer.MAX_VALUE);
136                         testInvalidType(surface, Integer.MIN_VALUE);
137                     }
138                 });
139             } finally {
140                 rq.unlock();
141             }
142         } else {
143             System.out.println("null accelerated surface");
144         }
145     }

155     private static void testGC(GraphicsConfiguration gc) {
156         if (!(gc instanceof AccelGraphicsConfig)) {
157             System.out.println("Test passed: no hw accelerated configs found.");
158             return;
159         }
160         System.out.println("AccelGraphicsConfig exists, testing.");
161         AccelGraphicsConfig agc = (AccelGraphicsConfig) gc;
162         printAGC(agc);
163 
164         testContext(agc);
165 
166         VolatileImage vi = gc.createCompatibleVolatileImage(10, 10);
167         vi.validate(gc);
168         if (vi instanceof DestSurfaceProvider) {
169             System.out.println("Passed: VI is DestSurfaceProvider");
170             Surface s = ((DestSurfaceProvider) vi).getDestSurface();
171             if (s instanceof AccelSurface) {
172                 System.out.println("Passed: Obtained Accel Surface");
173                 printSurface((AccelSurface) s);
174             }
175             Graphics g = vi.getGraphics();
176             if (g instanceof DestSurfaceProvider) {
177                 System.out.println("Passed: VI graphics is " +
178                                    "DestSurfaceProvider");
179                 printSurface(((DestSurfaceProvider) g).getDestSurface());
180             }
181         } else {
182             System.out.println("VI is not DestSurfaceProvider");
183         }
184         testVICreation(agc, CAPS_RT_TEXTURE_ALPHA, TRANSLUCENT, RT_TEXTURE);
185         testVICreation(agc, CAPS_RT_TEXTURE_OPAQUE, OPAQUE, RT_TEXTURE);
186         testVICreation(agc, CAPS_RT_PLAIN_ALPHA, TRANSLUCENT, RT_PLAIN);
187         testVICreation(agc, agc.getContextCapabilities().getCaps(), OPAQUE,
188                        TEXTURE);
189         testForNPEDuringCreation(agc);
190     }

192     private static void testVICreation(AccelGraphicsConfig agc, int cap,
193                                        int transparency, int type)
194     {
195         int caps = agc.getContextCapabilities().getCaps();
196         int w = 11, h = 17;
197 
198         VolatileImage vi =
199             agc.createCompatibleVolatileImage(w, h, transparency, type);
200         if ((cap & caps) != 0) {
201             if (vi == null) {
202                 System.out.printf("Failed: cap=%d is supported but " +
203                                   "image wasn't created\n", cap);
204                 throw new RuntimeException("Failed: image wasn't created " +
205                                            "for supported cap");
206             } else {
207                 if (!(vi instanceof DestSurfaceProvider)) {
208                     throw new RuntimeException("Failed: created VI is not " +
209                                                "DestSurfaceProvider");
210                 }
211                 Surface s = ((DestSurfaceProvider) vi).getDestSurface();
212                 if (s instanceof AccelSurface) {
213                     AccelSurface as = (AccelSurface) s;
214                     printSurface(as);
215                     if (as.getType() != type) {
216                         throw new RuntimeException("Failed: returned VI is" +
217                                 " of incorrect type: " + as.getType() +
218                                 " requested type=" + type);
219                     } else {
220                         System.out.printf("Passed: VI of type %d was " +
221                                 "created for cap=%d\n", type, cap);
222                     }
223                     if (as.getType() == TEXTURE) {
224                         boolean ex = false;
225                         try {
226                             Graphics g = vi.getGraphics();
227                             g.dispose();
228                         } catch (UnsupportedOperationException e) {
229                             ex = true;
230                         }
231                         if (!ex) {
232                             throw new RuntimeException("Failed: " +
233                                 "texture.getGraphics() didn't throw exception");
234                         } else {
235                             System.out.println("Passed: VI.getGraphics()" +
236                                     " threw exception for texture-based VI");
237                         }
238                     }
239                 } else {
240                     System.out.printf("Passed: VI of type %d was " +
241                             "created for cap=%d but accel surface is null\n",
242                             type, cap);
243                 }
244             }
245         } else {
246             if (vi != null) {
247                 throw new RuntimeException("Failed: created VI for " +
248                                            "unsupported cap=" + cap);
249             }
250         }
251     }

253     private static void testContext(final AccelGraphicsConfig agc) {
254         BufferedContext c = agc.getContext();
255 
256         RenderQueue rq = c.getRenderQueue();
257         rq.lock();
258         try {
259             c.saveState();
260             rq.flushNow();
261             c.restoreState();
262             rq.flushNow();
263             System.out.println("Passed: Save/Restore");
264         } finally {
265             rq.unlock();
266         }
267     }

269     private static void testForNPEDuringCreation(AccelGraphicsConfig agc) {
270         int iterations = 100;
271         HashSet<VolatileImage> vis = new HashSet<VolatileImage>();
272         GraphicsConfiguration gc = (GraphicsConfiguration)agc;
273         Rectangle r = gc.getBounds();
274         long ram = gc.getDevice().getAvailableAcceleratedMemory();
275         if (ram > 0) {
276             // guesstimate the number of iterations needed to exhaust vram
277             int i = 2 *
278                 (int)(ram / (r.width * r.height * gc.getColorModel().getPixelSize()/8));
279             iterations = Math.max(iterations, i);
280             System.err.println("iterations="+iterations);
281         }
282         for (int i = 0; i < iterations; i++) {
283             VolatileImage vi =
284                 agc.createCompatibleVolatileImage(r.width, r.height,
285                                                   Transparency.OPAQUE,
286                                                   AccelSurface.RT_PLAIN);
287             if (vi == null) {
288                 break;
289             }
290             vis.add(vi);
291         }
292         for (VolatileImage vi : vis) {
293             vi.flush();
294         }
295         vis = null;
296 
297         System.out.println("Passed: testing for possible NPEs " +
298                            "during VI creation");
299     }
