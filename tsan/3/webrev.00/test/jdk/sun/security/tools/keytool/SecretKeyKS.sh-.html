45 # set platform-dependent variables

47 case "$OS" in
48   SunOS | Linux | Darwin | AIX )
49     NULL=/dev/null
50     PS=":"
51     FS="/"
52     ;;
53   CYGWIN* )

55     PS=";"
56     FS="/"

58   Windows_* )
59     NULL=NUL
60     PS=";"
61     FS="\\"

63   * )
64     echo "Unrecognized operating system!"
65     exit 1;
66     ;;
67 esac
