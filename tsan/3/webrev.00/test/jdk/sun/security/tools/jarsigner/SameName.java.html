24 /*
25  * @test
26  * @bug 6866479
27  * @summary libzip.so caused JVM to crash when running jarsigner
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.Platform;
32 import jdk.test.lib.SecurityTools;
33 import jdk.test.lib.util.JarUtils;
34 
35 import java.nio.file.Files;
36 import java.nio.file.Path;
37 import java.util.List;
38 
39 public class SameName {
40     public static void main(String[] args) throws Exception {
41 
42         String signedJar = Platform.isWindows() ? "EM.jar" : "em.jar";
43 
44         Files.write(Path.of("A"), List.of("A"));
45         JarUtils.createJarFile(Path.of("em.jar"), Path.of("."), Path.of("A"));
46 
47         SecurityTools.keytool("-storepass changeit -keypass changeit "
48                 + "-keystore ks -keyalg rsa  -alias a -dname CN=a "
49                 + "-keyalg rsa -genkey -validity 300")
50                 .shouldHaveExitValue(0);
51 
52         SecurityTools.jarsigner("-keystore ks -storepass changeit "
53                 + "-signedjar " + signedJar + " em.jar a")
54                 .shouldHaveExitValue(0);
55     }
56 }
