24 # @test
25 # @bug 6543940 6868865
26 # @summary Exception thrown when signing a jarfile in java 1.5
27 #
28 # @run shell oldsig.sh
29 
30 if [ "${TESTSRC}" = "" ] ; then
31   TESTSRC="."
32 fi
33 if [ "${TESTCLASSES}" = "" ] ; then
34   TESTCLASSES="."
35 fi
36 if [ "${TESTJAVA}" = "" ] ; then
37   echo "TESTJAVA not set.  Test cannot execute."
38   echo "FAILED!!!"
39   exit 1
40 fi
41 
42 # set platform-dependent variables
43 OS=`uname -s`
44 case "$OS" in
45   SunOS | Linux | Darwin | AIX )
46     NULL=/dev/null
47     PS=":"
48     FS="/"
49     CP="${FS}bin${FS}cp -f"
50     ;;
51   CYGWIN* )
52     NULL=/dev/null
53     PS=";"
54     FS="/"
55     CP="cp -f"
56     ;;
57   Windows_* )
58     NULL=NUL
59     PS=";"
60     FS="\\"
61     CP="cp -f"
62     ;;
63   * )
64     echo "Unrecognized operating system!"
65     exit 1;
66     ;;
67 esac
68 
69 # copy jar file into writeable location
70 ${CP} ${TESTSRC}${FS}oldsig${FS}A.jar B.jar
71 ${CP} ${TESTSRC}${FS}oldsig${FS}A.class B.class
72 
73 ${TESTJAVA}${FS}bin${FS}jar ${TESTTOOLVMOPTS} uvf B.jar B.class
74 ${TESTJAVA}${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS} \
75     -keystore ${TESTSRC}${FS}JarSigning.keystore \
76     -storepass bbbbbb \
77     -digestalg SHA1 \
78     B.jar c
79 ${TESTJAVA}${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS} -verify B.jar
