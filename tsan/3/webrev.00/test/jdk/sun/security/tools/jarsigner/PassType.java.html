24 /*
25  * @test
26  * @bug 6868579
27  * @summary RFE: jarsigner to support reading password from environment variable
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.process.OutputAnalyzer;
33 import jdk.test.lib.process.ProcessTools;
34 import jdk.test.lib.util.JarUtils;
35 
36 import java.nio.file.Files;
37 import java.nio.file.Path;
38 import java.util.List;
39 
40 public class PassType {
41 
42     static OutputAnalyzer jarsignerWithEnv(String cmd) throws Throwable {
43         ProcessBuilder pb = SecurityTools.getProcessBuilder(
44                 "jarsigner", List.of(cmd.trim().split("\\s+")));
45         pb.environment().put("PASSENV", "test12");
46         return ProcessTools.executeCommand(pb);
47     }
48 
49     static OutputAnalyzer keytoolWithEnv(String cmd) throws Throwable {
50         ProcessBuilder pb = SecurityTools.getProcessBuilder(
51                 "keytool", List.of(cmd.trim().split("\\s+")));
52         pb.environment().put("PASSENV", "test12");
53         return ProcessTools.executeCommand(pb);
54     }
55 
56     public static void main(String[] args) throws Throwable {
57 
58         SecurityTools.keytool("-keystore ks -validity 300 -keyalg rsa "
59                 + "-alias a -dname CN=a -keyalg rsa -genkey "
60                 + "-storepass test12 -keypass test12")
61                 .shouldHaveExitValue(0);
62         keytoolWithEnv("-keystore ks -validity 300 -keyalg rsa "
63                 + "-alias b -dname CN=b -keyalg rsa -genkey "
64                 + "-storepass:env PASSENV -keypass:env PASSENV")
65                 .shouldHaveExitValue(0);
66         Files.write(Path.of("passfile"), List.of("test12"));
67         SecurityTools.keytool("-keystore ks -validity 300 -keyalg rsa "
68                 + "-alias c -dname CN=c -keyalg rsa -genkey "
69                 + "-storepass:file passfile -keypass:file passfile")
70                 .shouldHaveExitValue(0);
71 
72         Files.write(Path.of("A"), List.of("A"));
73         JarUtils.createJarFile(Path.of("a.jar"), Path.of("."), Path.of("A"));
74 
75         // Sign
76         SecurityTools.jarsigner("-keystore ks -storepass test12 a.jar a")
77                 .shouldHaveExitValue(0);
78         jarsignerWithEnv("-keystore ks -storepass:env PASSENV a.jar b")
79                 .shouldHaveExitValue(0);
80         SecurityTools.jarsigner("-keystore ks -storepass:file passfile a.jar c")
81                 .shouldHaveExitValue(0);
82 
83         // Verify
84         SecurityTools.jarsigner("-keystore ks -storepass test12 "
85                 + "-verify -debug -strict a.jar")
86                 .shouldHaveExitValue(0);
87         jarsignerWithEnv("-keystore ks -storepass:env PASSENV "
88                 + "-verify -debug -strict a.jar")
89                 .shouldHaveExitValue(0);
90         SecurityTools.jarsigner("-keystore ks -storepass:file passfile "
91                 + "-verify -debug -strict a.jar")
92                 .shouldHaveExitValue(0);
93     }
94 }
