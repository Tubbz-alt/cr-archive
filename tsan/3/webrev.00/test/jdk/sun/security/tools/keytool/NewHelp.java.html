24 /*
25  * @test
26  * @bug 6324292
27  * @summary keytool -help is unhelpful
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 
33 public class NewHelp {
34     public static void main(String[] args) throws Exception {
35         SecurityTools.keytool("-help")
36                 .shouldHaveExitValue(0)
37                 .shouldContain("Commands:");
38         SecurityTools.keytool("-help -list")
39                 .shouldHaveExitValue(0)
40                 .shouldContain("Options:");
41     }
42 }
