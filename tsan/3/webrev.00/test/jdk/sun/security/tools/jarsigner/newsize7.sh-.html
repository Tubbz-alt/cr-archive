24 # @test
25 # @bug 6561126
26 # @summary keytool should use larger default keysize for keypairs
27 #
28 # @run shell newsize7.sh
29 
30 # set a few environment variables so that the shell-script can run stand-alone
31 # in the source directory
32 if [ "${TESTSRC}" = "" ] ; then
33    TESTSRC="."
34 fi
35 
36 if [ "${TESTJAVA}" = "" ] ; then
37   JAVA_CMD=`which java`
38   TESTJAVA=`dirname $JAVA_CMD`/..
39 fi
40 
41 # set platform-dependent variables
42 OS=`uname -s`
43 case "$OS" in
44   Windows_* )
45     FS="\\"
46     ;;
47   * )
48     FS="/"
49     ;;
50 esac
51 
52 KSFILE=ns7.jks
53 
54 KT="${TESTJAVA}${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -keystore ns7.jks -storepass changeit -keypass changeit -keyalg rsa"
55 JAR="${TESTJAVA}${FS}bin${FS}jar ${TESTTOOLVMOPTS}"
56 JS="${TESTJAVA}${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS} -keystore ns7.jks -storepass changeit"
57 
58 rm ns7.*
59 
60 $KT -genkeypair -alias me -dname CN=Me
61 
62 touch ns7.txt
63 $JAR cvf ns7.jar ns7.txt
64 
65 $JS ns7.jar me
66 $JAR xvf ns7.jar
67 
68 grep SHA-256 META-INF/MANIFEST.MF || exit 1
69 grep SHA-256 META-INF/ME.SF || exit 2
70 
71 #rm -rf META-INF
72 
73 exit 0
