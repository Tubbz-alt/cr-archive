24 import java.io.FileNotFoundException;
25 import java.io.FileOutputStream;
26 import java.io.IOException;
27 
28 /*
29  * A custom output stream that redirects the testing outputs to a file, called
30  * details.out. It also calls another output stream to save some outputs to
31  * other files.
32  */
33 public class DetailsOutputStream extends FileOutputStream {
34 
35     private PhaseOutputStream phaseOutputStream = new PhaseOutputStream();
36 
37     public DetailsOutputStream() throws FileNotFoundException {
38         super("details.out", true);

39     }
40 
41     public void transferPhase() throws IOException {
42         if (phaseOutputStream.isCorePhase()) {
43             phaseOutputStream.write(HtmlHelper.endHtml());
44             phaseOutputStream.write(HtmlHelper.endPre());
45         }
46 
47         phaseOutputStream.transfer();
48 
49         if (phaseOutputStream.isCorePhase()) {
50             phaseOutputStream.write(HtmlHelper.startHtml());
51             phaseOutputStream.write(HtmlHelper.startPre());
52         }
53     }
54 
55     @Override
56     public void write(byte[] b) throws IOException {
57         super.write(b);
58         phaseOutputStream.write(b);
59     }
60 
61     @Override
62     public void write(int b) throws IOException {
63         super.write(b);
64         phaseOutputStream.write(b);
65     }
66 
67     @Override
68     public void write(byte b[], int off, int len) throws IOException {
69         super.write(b, off, len);
70         phaseOutputStream.write(b, off, len);
71     }
72 
73     public void writeAnchorName(String name, String text) throws IOException {
74         super.write((text).getBytes());
75         super.write('\n');
76         phaseOutputStream.write(HtmlHelper.anchorName(name, text));
77         phaseOutputStream.write('\n');
78     }
79 }
