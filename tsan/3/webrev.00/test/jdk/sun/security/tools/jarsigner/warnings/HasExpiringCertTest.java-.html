 24 import jdk.test.lib.process.OutputAnalyzer;
 25 import jdk.test.lib.util.JarUtils;
 26 
 27 /**
 28  * @test
 29  * @bug 8024302 8026037
 30  * @summary Test for hasExpiringCert warning
 31  * @library /test/lib ../
 32  * @build jdk.test.lib.util.JarUtils
 33  * @run main HasExpiringCertTest
 34  */
 35 public class HasExpiringCertTest extends Test {
 36 
 37     static final int SHORT_VALIDITY = 90; // less than 6 month
 38 
 39     /**
 40      * The test signs and verifies a jar that contains entries
 41      * whose signer certificate will expire within six months (hasExpiringCert).
 42      * Warning message is expected.
 43      */
 44     public static void main(String[] args) throws Throwable {
 45         HasExpiringCertTest test = new HasExpiringCertTest();
 46         test.start();
 47     }
 48 
 49     private void start() throws Throwable {
 50         // create a jar file that contains one class file
 51         Utils.createFiles(FIRST_FILE);
 52         JarUtils.createJar(UNSIGNED_JARFILE, FIRST_FILE);
 53 
 54         // create key pair for jar signing
 55         createAlias(CA_KEY_ALIAS);
 56         createAlias(KEY_ALIAS);
 57 
 58         issueCert(
 59                 KEY_ALIAS,
 60                 "-validity", Integer.toString(SHORT_VALIDITY));
 61 
 62         // sign jar
 63         OutputAnalyzer analyzer = jarsigner(
 64                 "-keystore", KEYSTORE,
 65                 "-verbose",
 66                 "-storepass", PASSWORD,
 67                 "-keypass", PASSWORD,
 68                 "-signedjar", SIGNED_JARFILE,
 69                 UNSIGNED_JARFILE,
 70                 KEY_ALIAS);
 71 
 72         checkSigning(analyzer, HAS_EXPIRING_CERT_SIGNING_WARNING);
 73 
 74         // verify signed jar
 75         analyzer = jarsigner(
 76                 "-verify",
 77                 "-verbose",
 78                 "-keystore", KEYSTORE,
 79                 "-storepass", PASSWORD,
 80                 "-keypass", PASSWORD,
 81                 SIGNED_JARFILE,
 82                 KEY_ALIAS);
 83 
 84         checkVerifying(analyzer, 0, HAS_EXPIRING_CERT_VERIFYING_WARNING);
 85 
 86         // verify signed jar in strict mode
 87         analyzer = jarsigner(
 88                 "-verify",
 89                 "-verbose",
 90                 "-strict",
 91                 "-keystore", KEYSTORE,
 92                 "-storepass", PASSWORD,
 93                 "-keypass", PASSWORD,
 94                 SIGNED_JARFILE,
 95                 KEY_ALIAS);
 96 
 97         checkVerifying(analyzer, 0, HAS_EXPIRING_CERT_VERIFYING_WARNING);
 98 
 99         System.out.println("Test passed");
100     }
101 
102 }
