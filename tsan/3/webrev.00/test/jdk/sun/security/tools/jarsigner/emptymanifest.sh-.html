24 # @test
25 # @bug 6712755
26 # @summary jarsigner fails to sign itextasian.jar since 1.5.0_b14, it works with 1.5.0_13
27 #
28 # @run shell emptymanifest.sh
29 #
30 
31 if [ "${TESTJAVA}" = "" ] ; then
32   JAVAC_CMD=`which javac`
33   TESTJAVA=`dirname $JAVAC_CMD`/..
34 fi
35 
36 # set platform-dependent variables
37 OS=`uname -s`
38 case "$OS" in
39   Windows_* )
40     FS="\\"
41     ;;
42   * )
43     FS="/"
44     ;;
45 esac
46 
47 KS=emptymanifest.ks
48 JFILE=em.jar
49 
50 KT="$TESTJAVA${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -storepass changeit -keypass changeit -keystore $KS"
51 JAR="$TESTJAVA${FS}bin${FS}jar ${TESTTOOLVMOPTS}"
52 JAVA="$TESTJAVA${FS}bin${FS}java ${TESTVMOPTS}"
53 JAVAC="$TESTJAVA${FS}bin${FS}javac ${TESTTOOLVMOPTS} ${TESTJAVACOPTS}"
54 JARSIGNER="$TESTJAVA${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS} -keystore $KS -storepass changeit"
55 
56 rm $KS $JFILE
57 echo A > A
58 echo B > B
59 mkdir META-INF
60 cat <<EOF > CrLf.java
61 class CrLf {
62   public static void main(String[] args) throws Exception {
63     System.out.write(new byte[] {'\r', '\n'});
64   }
65 }
66 EOF
67 $JAVAC CrLf.java
68 $JAVA CrLf > META-INF${FS}MANIFEST.MF
69 zip $JFILE META-INF${FS}MANIFEST.MF A B
70 
71 $KT -alias a -dname CN=a -keyalg rsa -genkey -validity 300
72 
73 $JARSIGNER $JFILE a || exit 1
74 $JARSIGNER -verify -debug -strict $JFILE || exit 2
75 
76 exit 0
