 24 /*
 25  * A helper that is used for creating HTML elements.
 26  */
 27 public class HtmlHelper {
 28 
 29     private static final String STYLE
 30             = "style=\"font-family: Courier New; "
 31             + "font-size: 12px; "
 32             + "white-space: pre-wrap\"";
 33 
 34     public static String htmlRow(String... values) {
 35         StringBuilder row = new StringBuilder();
 36         row.append(startTr());
 37         for (String value : values) {
 38             row.append(startTd());
 39             row.append(value);
 40             row.append(endTd());
 41         }
 42         row.append(endTr());
 43         return row.toString();
 44     }
 45 
 46     public static String startHtml() {
 47         return startTag("html");
 48     }
 49 
 50     public static String endHtml() {
 51         return endTag("html");
 52     }
 53 
 54     public static String startPre() {
 55         return startTag("pre " + STYLE);
 56     }
 57 
 58     public static String endPre() {
 59         return endTag("pre");
 60     }
 61 
 62     public static String startTable() {
 63         return startTag("table " + STYLE);
 64     }
 65 
 66     public static String endTable() {
 67         return endTag("table");
 68     }
 69 
 70     public static String startTr() {
 71         return startTag("tr");
 72     }
 73 
 74     public static String endTr() {
 75         return endTag("tr");
 76     }
 77 
 78     public static String startTd() {
 79         return startTag("td");
 80     }
 81 
 82     public static String endTd() {
 83         return endTag("td");
 84     }
 85 
 86     public static String startTag(String tag) {
 87         return "<" + tag + ">";
 88     }
 89 
 90     public static String endTag(String tag) {
 91         return "</" + tag + ">";
 92     }
 93 
 94     public static String anchorName(String name, String text) {
 95         return "<a name=" + name + ">" + text + "</a>";
 96     }
 97 
 98     public static String anchorLink(String file, String anchorName,
 99             String text) {
100         return "<a href=" + file + "#" + anchorName + ">" + text + "</a>";
101     }
102 }
