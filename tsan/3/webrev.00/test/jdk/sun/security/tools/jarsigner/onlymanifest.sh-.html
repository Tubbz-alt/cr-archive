24 # @test
25 # @bug 7004035
26 # @summary signed jar with only META-INF/* inside is not verifiable
27 #
28 
29 if [ "${TESTJAVA}" = "" ] ; then
30   JAVAC_CMD=`which javac`
31   TESTJAVA=`dirname $JAVAC_CMD`/..
32 fi
33 
34 # set platform-dependent variables
35 OS=`uname -s`
36 case "$OS" in
37   Windows_* )
38     FS="\\"
39     ;;
40   * )
41     FS="/"
42     ;;
43 esac
44 
45 KS=onlymanifest.jks
46 JFILE=onlymanifest.jar
47 
48 KT="$TESTJAVA${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -storepass changeit -keypass changeit \
49         -keystore $KS -keyalg rsa"
50 JAR="$TESTJAVA${FS}bin${FS}jar ${TESTTOOLVMOPTS}"
51 JARSIGNER="$TESTJAVA${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS}"
52 
53 rm $KS $JFILE 2> /dev/null
54 
55 # Create an empty jar file with only MANIFEST.MF
56 
57 echo "Key: Value" > manifest
58 $JAR cvfm $JFILE manifest
59 
60 $KT -alias ca -dname CN=ca -genkey -validity 300 || exit 1
61 $KT -alias a -dname CN=a -genkey -validity 300 || exit 2
62 $KT -alias a -certreq | $KT -gencert -alias ca -validity 300 | $KT -import -alias a || exit 3
63 $JARSIGNER -keystore $KS -storepass changeit $JFILE a -debug -strict || exit 4
64 $JARSIGNER -keystore $KS -storepass changeit -verify $JFILE a -debug -strict \
65         > onlymanifest.out || exit 5
66 
67 grep unsigned onlymanifest.out && exit 6
68 
69 exit 0
