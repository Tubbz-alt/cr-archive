34 # set platform-dependent variables
35

37 case "$OS" in
38   Windows* )

43     ;;

45 
46 KS=crl.jks
47 
48 KT="$TESTJAVA${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -storepass changeit -keypass changeit -keystore $KS -keyalg rsa"
