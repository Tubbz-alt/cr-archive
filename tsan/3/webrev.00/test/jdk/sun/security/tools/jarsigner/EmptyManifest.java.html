24 /*
25  * @test
26  * @bug 6712755
27  * @summary jarsigner fails to sign itextasian.jar since 1.5.0_b14,
28  *          it works with 1.5.0_13
29  * @library /test/lib
30  */
31 
32 import jdk.test.lib.SecurityTools;
33 
34 import java.io.FileOutputStream;
35 import java.util.zip.ZipEntry;
36 import java.util.zip.ZipOutputStream;
37 
38 public class EmptyManifest {
39 
40     public static void main(String[] args) throws Exception {
41 
42         try (FileOutputStream fos = new FileOutputStream("em.jar");
43              ZipOutputStream zos = new ZipOutputStream(fos)) {
44             zos.putNextEntry(new ZipEntry("META-INF/MANIFEST.MF"));
45             zos.write(new byte[]{'\r', '\n'});
46             zos.putNextEntry(new ZipEntry("A"));
47             zos.write(new byte[10]);
48             zos.putNextEntry(new ZipEntry("B"));
49             zos.write(new byte[0]);
50         }
51 
52         SecurityTools.keytool("-keystore ks -storepass changeit "
53                 + "-keypass changeit -alias a -dname CN=a -keyalg rsa "
54                 + "-genkey -validity 300");
55 
56         SecurityTools.jarsigner("-keystore ks -storepass changeit em.jar a")
57                 .shouldHaveExitValue(0);
58         SecurityTools.jarsigner("-keystore ks -storepass changeit -verify "
59                 + "-debug -strict em.jar")
60                 .shouldHaveExitValue(0);
61     }
62 }
