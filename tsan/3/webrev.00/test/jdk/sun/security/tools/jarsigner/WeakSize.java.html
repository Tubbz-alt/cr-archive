24 /*
25  * @test
26  * @bug 8044755
27  * @summary Add a test for algorithm constraints check in jarsigner
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.process.OutputAnalyzer;
33 import jdk.test.lib.util.JarUtils;
34 
35 import java.nio.file.Path;
36 
37 public class WeakSize {
38 
39     static OutputAnalyzer kt(String cmd) throws Exception {
40         // The sigalg used is MD2withRSA, which is obsolete.
41         return SecurityTools.keytool("-storepass changeit -keypass changeit "
42                 + "-keystore ks -keyalg rsa -sigalg MD2withRSA " + cmd);
43     }
44 
45     static void gencert(String owner, String cmd) throws Exception {
46         kt("-certreq -alias " + owner + " -file tmp.req");
47         kt("-gencert -infile tmp.req -outfile tmp.cert " + cmd);
48         kt("-import -alias " + owner + " -file tmp.cert");
49     }
50 
51     public static void main(String[] args) throws Exception {
52 
53         kt("-genkeypair -alias ca -dname CN=CA -ext bc");
54         kt("-genkeypair -alias signer -dname CN=Signer");
55         gencert("signer", "-alias ca -ext ku=dS -rfc");
56 
57         JarUtils.createJarFile(Path.of("a.jar"), Path.of("."), Path.of("ks"));
58 
59         // We always trust a TrustedCertificateEntry
60         SecurityTools.jarsigner("-keystore ks -storepass changeit "
61                 + "-strict -debug a.jar ca")
62                 .shouldNotContain("chain is invalid");
63 
64         // An end-entity cert must follow algorithm constraints
65         SecurityTools.jarsigner("-keystore ks -storepass changeit "
66                 + "-strict -debug a.jar signer")
67                 .shouldContain("chain is invalid");
68     }
69 }
