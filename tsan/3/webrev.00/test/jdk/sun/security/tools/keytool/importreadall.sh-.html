24 # @test
25 # @bug 6819272
26 # @summary keytool -importcert should read the whole input
27 #
28 # @run shell importreadall.sh
29 
30 # set a few environment variables so that the shell-script can run stand-alone
31 # in the source directory
32 if [ "${TESTSRC}" = "" ] ; then
33    TESTSRC="."
34 fi
35 
36 if [ "${TESTJAVA}" = "" ] ; then
37   JAVA_CMD=`which java`
38   TESTJAVA=`dirname $JAVA_CMD`/..
39 fi
40 
41 # set platform-dependent variables
42 OS=`uname -s`
43 case "$OS" in
44   Windows_* )
45     FS="\\"
46     ;;
47   * )
48     FS="/"
49     ;;
50 esac
51 
52 KEYTOOL="${TESTJAVA}${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -keystore importreadall.jks -storepass changeit -keypass changeit -keyalg rsa"
53 
54 # In case the test is run twice in the same directory
55 
56 $KEYTOOL -delete -alias a
57 $KEYTOOL -delete -alias ca
58 $KEYTOOL -genkeypair -alias a -dname CN=a || exit 1
59 $KEYTOOL -genkeypair -alias ca -dname CN=ca || exit 2
60 $KEYTOOL -certreq -alias a | $KEYTOOL -gencert -alias ca | $KEYTOOL -importcert -alias a
61 
62 exit $?
