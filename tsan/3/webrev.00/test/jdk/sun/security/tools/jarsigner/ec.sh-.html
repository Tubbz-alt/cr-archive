25 # @bug 6870812
26 # @summary enhance security tools to use ECC algorithm
27 #
28 
29 if [ "${TESTJAVA}" = "" ] ; then
30   JAVAC_CMD=`which javac`
31   TESTJAVA=`dirname $JAVAC_CMD`/..
32 fi
33 
34 # set platform-dependent variables
35 OS=`uname -s`
36 case "$OS" in
37   Windows_* )
38     FS="\\"
39     ;;
40   * )
41     FS="/"
42     ;;
43 esac
44 
45 KS=ec.jks
46 JFILE=ec.jar
47 
48 KT="$TESTJAVA${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -storepass changeit -keypass changeit -keystore $KS"
49 JAR="$TESTJAVA${FS}bin${FS}jar ${TESTTOOLVMOPTS}"
50 JARSIGNER="$TESTJAVA${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS}"
51 
52 rm $KS $JFILE
53 echo A > A
54 $JAR cvf $JFILE A
55 
56 $KT -alias ca -dname CN=ca -keyalg ec -genkey -validity 300 || exit 11
57 
58 $KT -alias a -dname CN=a -keyalg ec -genkey || exit 11
59 $KT -alias a -certreq | $KT -gencert -alias ca -validity 300 | $KT -import -alias a || exit 111
60 
61 $KT -alias b -dname CN=b -keyalg ec -genkey || exit 12
62 $KT -alias b -certreq | $KT -gencert -alias ca -validity 300 | $KT -import -alias b || exit 121
63 
64 # Ensure that key length is sufficient for the intended hash (SHA512withECDSA)
65 $KT -alias c -dname CN=c -keyalg ec -genkey -keysize 521 || exit 13
66 $KT -alias c -certreq | $KT -gencert -alias ca -validity 300 | $KT -import -alias c || exit 131
67 
68 $KT -alias x -dname CN=x -keyalg ec -genkey -validity 300 || exit 14
69 $KT -alias x -certreq | $KT -gencert -alias ca -validity 300 | $KT -import -alias x || exit 141
70 
71 $JARSIGNER -keystore $KS -storepass changeit $JFILE a -debug -strict || exit 21
72 $JARSIGNER -keystore $KS -storepass changeit $JFILE b -debug -strict -sigalg SHA1withECDSA || exit 22
73 $JARSIGNER -keystore $KS -storepass changeit $JFILE c -debug -strict -sigalg SHA512withECDSA || exit 23
74 
75 $JARSIGNER -keystore $KS -storepass changeit -verify $JFILE a -debug -strict || exit 31
76 $JARSIGNER -keystore $KS -storepass changeit -verify $JFILE b -debug -strict || exit 32
77 $JARSIGNER -keystore $KS -storepass changeit -verify $JFILE c -debug -strict || exit 33
78 
79 # Not signed by x, should exit with non-zero
80 $JARSIGNER -keystore $KS -storepass changeit -verify $JFILE x -debug -strict && exit 34
81 
82 exit 0
