24 import jdk.test.lib.SecurityTools;
25 import jdk.test.lib.process.OutputAnalyzer;
26 
27 /**
28  * @test
29  * @bug 8212003 8214024
30  * @summary Deprecating the default keytool -keyalg option
31  * @library /test/lib
32  */
33 
34 public class RemoveKeyAlgDefault {
35 
36     private static final String COMMON = "-keystore ks -storetype jceks "
37             + "-storepass changeit -keypass changeit";
38 
39     public static void main(String[] args) throws Throwable {
40 
41         kt("-genkeypair -keyalg DSA -alias a -dname CN=A")
42                 .shouldHaveExitValue(0)
43                 .shouldContain("Generating")
44                 .shouldNotContain("-keyalg option must be specified");
45 
46         kt("-genkeypair -alias b -dname CN=B")
47                 .shouldHaveExitValue(1)
48                 .shouldContain("-keyalg option must be specified");
49 
50         kt("-genseckey -keyalg DES -alias c")
51                 .shouldHaveExitValue(0)
52                 .shouldContain("Generated")
53                 .shouldNotContain("-keyalg option must be specified");
54 
55         kt("-genseckey -alias d")
56                 .shouldHaveExitValue(1)
57                 .shouldContain("-keyalg option must be specified");
58     }
59 
60     private static OutputAnalyzer kt(String cmd) throws Throwable {
61         return SecurityTools.keytool(COMMON + " " + cmd);
62     }
63 }
