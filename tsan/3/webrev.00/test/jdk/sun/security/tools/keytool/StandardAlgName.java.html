24 /*
25  * @test
26  * @bug 4909889
27  * @summary KeyTool accepts any input that user make as long as we can make some
28  *          sense out of it, but when comes to present the info the user, it
29  *          promotes a standard look.
30  * @author Andrew Fan
31  * @library /test/lib
32  * @run main/timeout=240 StandardAlgName
33  */
34 
35 import jdk.test.lib.SecurityTools;
36 
37 public class StandardAlgName {
38     public static void main(String[] args) throws Exception {
39         // CA
40         SecurityTools.keytool("-genkey", "-v", "-alias", "pkcs12testCA",
41                 "-keyalg", "RsA", "-keysize", "2048",
42                 "-sigalg", "ShA1wItHRSA",
43                 "-dname", "cn=PKCS12 Test CA, ou = Security SQE, o = JavaSoft, c = US",
44                 "-validity", "3650",
45                 "-keypass", "storepass", "-keystore", "keystoreCA.jceks.data",
46                 "-storepass", "storepass", "-storetype", "jceKS")
47                 .shouldHaveExitValue(0)
48                 .shouldNotContain("RsA")
49                 .shouldNotContain("ShA1wItHRSA")
50                 .shouldContain("RSA")
51                 .shouldContain("SHA1withRSA");
52 
53         // Lead
54         SecurityTools.keytool("-genkey", "-v", "-alias", "pkcs12testLead",
55                 "-keyalg", "rSA", "-keysize", "1024",
56                 "-sigalg", "mD5withRSA",
57                 "-dname", "cn=PKCS12 Test Lead, ou=Security SQE, o=JavaSoft, c=US",
58                 "-validity", "3650",
59                 "-keypass", "storepass", "-keystore", "keystoreLead.jceks.data",
60                 "-storepass", "storepass", "-storetype", "jCeks")
61                 .shouldHaveExitValue(0)
62                 .shouldNotContain("rSA")
63                 .shouldNotContain("mD5withRSA")
64                 .shouldContain("RSA")
65                 .shouldContain("MD5withRSA");
66 
67         // End User 1
68         SecurityTools.keytool("-genkey", "-v", "-alias", "pkcs12testEndUser1",
69                 "-keyalg", "RSa", "-keysize", "1024",
70                 "-sigalg", "sHa1wIThRSA",
71                 "-dname", "cn=PKCS12 Test End User 1, ou=Security SQE, o=JavaSoft, c=US",
72                 "-validity", "3650",
73                 "-keypass", "storepass", "-keystore", "keystoreEndUser1.jceks.data",
74                 "-storepass", "storepass", "-storetype", "Jceks")
75                 .shouldHaveExitValue(0)
76                 .shouldNotContain("RSa")
77                 .shouldNotContain("sHa1wIThRSA")
78                 .shouldContain("RSA")
79                 .shouldContain("SHA1withRSA");
80     }
81 }
