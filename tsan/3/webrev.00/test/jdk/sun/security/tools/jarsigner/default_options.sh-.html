24 # @test
25 # @bug 8049834
26 # @summary Two security tools tests do not run with only JRE
27 #
28 
29 if [ "${TESTJAVA}" = "" ] ; then
30   JAVAC_CMD=`which javac`
31   TESTJAVA=`dirname $JAVAC_CMD`/..
32 fi
33 
34 PASS=changeit
35 export PASS
36 
37 TESTTOOLVMOPTS="$TESTTOOLVMOPTS -J-Duser.language=en -J-Duser.country=US"
38 
39 KS=ks
40 KEYTOOL="$TESTJAVA/bin/keytool ${TESTTOOLVMOPTS} -storepass:env PASS -keypass:env PASS -keystore $KS"
41 JAR="$TESTJAVA/bin/jar ${TESTTOOLVMOPTS}"
42 JARSIGNER="$TESTJAVA/bin/jarsigner ${TESTTOOLVMOPTS}"
43 
44 rm $KS 2> /dev/null
45 
46 $KEYTOOL -genkeypair -dname CN=A -alias a -keyalg rsa || exit 1
47 $KEYTOOL -genkeypair -dname CN=CA -alias ca -keyalg rsa || exit 2
48 $KEYTOOL -alias a -certreq |
49     $KEYTOOL -alias ca -gencert |
50     $KEYTOOL -alias a -import || exit 3
51 
52 cat <<EOF > js.conf
53 jarsigner.all = -keystore \${user.dir}/$KS -storepass:env PASS -debug -strict
54 jarsigner.sign = -digestalg SHA1
55 jarsigner.verify = -verbose:summary
56 
57 EOF
58 
59 $JAR cvf a.jar ks js.conf
60 
61 $JARSIGNER -conf js.conf a.jar a || exit 21
62 $JARSIGNER -conf js.conf -verify a.jar > jarsigner.out || exit 22
63 grep "and 1 more" jarsigner.out || exit 23
64 $JAR xvf a.jar META-INF/MANIFEST.MF
65 grep "SHA1-Digest" META-INF/MANIFEST.MF || exit 24
66 
67 echo Done
68 exit 0
