24 /*
25  * @test
26  * @bug 7004035
27  * @summary signed jar with only META-INF/* inside is not verifiable
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.process.OutputAnalyzer;
33 
34 import java.nio.file.Files;
35 import java.nio.file.Path;
36 import java.util.List;
37 
38 public class OnlyManifest {
39     static OutputAnalyzer kt(String cmd) throws Exception {
40         return SecurityTools.keytool("-storepass changeit -keypass changeit "
41                 + "-keystore ks -keyalg rsa " + cmd);
42     }
43 
44     static void gencert(String owner, String cmd) throws Exception {
45         kt("-certreq -alias " + owner + " -file tmp.req");
46         kt("-gencert -infile tmp.req -outfile tmp.cert " + cmd);
47         kt("-import -alias " + owner + " -file tmp.cert");
48     }
49 
50     public static void main(String[] args) throws Exception {
51         // Create an empty jar file with only MANIFEST.MF
52         Files.write(Path.of("manifest"), List.of("Key: Value"));
53         SecurityTools.jar("cvfm a.jar manifest");
54 
55         kt("-alias ca -dname CN=ca -genkey -validity 300 -ext bc:c")
56                 .shouldHaveExitValue(0);
57         kt("-alias a -dname CN=a -genkey -validity 300")
58                 .shouldHaveExitValue(0);
59         gencert("a", "-alias ca -validity 300");
60 
61         SecurityTools.jarsigner("-keystore ks -storepass changeit"
62                 + " a.jar a -debug -strict")
63                 .shouldHaveExitValue(0);
64         SecurityTools.jarsigner("-keystore ks -storepass changeit"
65                 + " -verify a.jar a -debug -strict")
66                 .shouldHaveExitValue(0)
67                 .shouldNotContain("unsigned");
68     }
69 }
