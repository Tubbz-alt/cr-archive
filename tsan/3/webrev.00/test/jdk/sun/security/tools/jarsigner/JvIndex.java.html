24 /*
25  * @test
26  * @bug 8022761
27  * @summary regression: SecurityException is NOT thrown while trying
28  *          to pack a wrongly signed Indexed Jar file
29  * @library /test/lib
30  */
31 
32 import jdk.test.lib.SecurityTools;
33 import jdk.test.lib.util.JarUtils;
34 
35 import java.nio.file.Files;
36 import java.nio.file.Path;
37 import java.util.List;
38 
39 public class JvIndex {
40     public static void main(String[] args) throws Exception {
41 
42         Files.write(Path.of("abcde"), List.of("12345"));
43         JarUtils.createJarFile(Path.of("jvindex.jar"), Path.of("."),
44                 Path.of("abcde"));
45         SecurityTools.keytool("-storepass changeit -keypass changeit "
46                 + "-keystore ks -keyalg rsa -alias a -dname CN=a "
47                 + "-genkey -validity 300")
48                 .shouldHaveExitValue(0);
49         SecurityTools.jarsigner("-keystore ks -storepass changeit jvindex.jar a")
50                 .shouldHaveExitValue(0);
51 
52         SecurityTools.jar("i jvindex.jar");
53 
54         // Make sure the $F line has "sm" (signed and in manifest)
55         SecurityTools.jarsigner("-keystore ks -storepass changeit -verify "
56                 + "-verbose jvindex.jar")
57                 .shouldHaveExitValue(0)
58                 .shouldMatch("sm.*abcde");
59     }
60 }
