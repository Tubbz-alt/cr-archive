24 /*
25  * @test
26  * @bug 6543940 6868865 8217375
27  * @summary Exception thrown when signing a jarfile in java 1.5
28  * @library /test/lib
29  */
30 /*
31  * See also PreserveRawManifestEntryAndDigest.java for tests with arbitrarily
32  * formatted individual sections in addition the the main attributes tested
33  * here.
34  */
35 
36 import jdk.test.lib.SecurityTools;
37 import jdk.test.lib.util.JarUtils;
38 
39 import java.nio.file.Files;
40 import java.nio.file.Path;
41 
42 public class OldSig {
43     public static void main(String[] args) throws Exception {
44         Path src = Path.of(System.getProperty("test.src"));
45         // copy jar file into writeable location
46         Files.copy(src.resolve("oldsig/A.jar"), Path.of("B.jar"));
47         Files.copy(src.resolve("oldsig/A.class"), Path.of("B.class"));
48 
49         JarUtils.updateJarFile(Path.of("B.jar"), Path.of("."),
50                 Path.of("B.class"));
51 
52         String ksArgs = "-keystore " + src.resolve("JarSigning.keystore")
53                 + " -storepass bbbbbb";
54         SecurityTools.jarsigner(ksArgs + " -digestalg SHA1 B.jar c");
55         SecurityTools.jarsigner("-verify B.jar").shouldHaveExitValue(0);
56         SecurityTools.jarsigner("-verify " + ksArgs + " -verbose B.jar c")
57                 .stdoutShouldMatch("^smk .* B[.]class$").shouldHaveExitValue(0);
58     }
59 }
