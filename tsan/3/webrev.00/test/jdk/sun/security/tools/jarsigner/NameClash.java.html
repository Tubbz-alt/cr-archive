24 /*
25  * @test
26  * @bug 6876328
27  * @summary different names for the same digest algorithms breaks jarsigner
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.util.JarUtils;
33 
34 import java.nio.file.Files;
35 import java.nio.file.Path;
36 import java.util.List;
37 
38 public class NameClash {
39     public static void main(String[] args) throws Exception {
40         String common = "-storepass changeit -keypass changeit -keystore ks ";
41 
42         SecurityTools.keytool(common + "-alias a -dname CN=a -keyalg rsa "
43                 + "-genkey -validity 300");
44         SecurityTools.keytool(common + "-alias b -dname CN=b -keyalg rsa "
45                 + "-genkey -validity 300");
46 
47         Files.write(Path.of("A"), List.of("A"));
48         JarUtils.createJarFile(Path.of("a.jar"), Path.of("."), Path.of("A"));
49 
50         SecurityTools.jarsigner(common + "a.jar a -digestalg SHA1")
51                 .shouldHaveExitValue(0);
52         SecurityTools.jarsigner(common + "a.jar b -digestalg SHA-1")
53                 .shouldHaveExitValue(0);
54 
55         SecurityTools.jarsigner(common + "-verify -debug -strict a.jar")
56                 .shouldHaveExitValue(0);
57     }
58 }
