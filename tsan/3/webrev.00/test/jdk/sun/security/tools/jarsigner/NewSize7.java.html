24 /*
25  * @test
26  * @bug 6561126
27  * @summary keytool should use larger default keysize for keypairs
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.Asserts;
32 import jdk.test.lib.SecurityTools;
33 import jdk.test.lib.util.JarUtils;
34 
35 import java.io.InputStream;
36 import java.nio.file.Files;
37 import java.nio.file.Path;
38 import java.util.jar.JarFile;
39 import java.util.jar.Manifest;
40 
41 public class NewSize7 {
42     public static void main(String[] args) throws Exception {
43         String common = "-storepass changeit -keypass changeit -keystore ks ";
44         SecurityTools.keytool(common
45                 + "-keyalg rsa -genkeypair -alias me -dname CN=Me");
46         Files.write(Path.of("ns7.txt"), new byte[0]);
47         JarUtils.createJarFile(Path.of("ns7.jar"), Path.of("."),
48                 Path.of("ns7.txt"));
49         SecurityTools.jarsigner(common + "ns7.jar me");
50 
51         try (JarFile jf = new JarFile("ns7.jar")) {
52             try (InputStream is = jf.getInputStream(
53                     jf.getEntry("META-INF/MANIFEST.MF"))) {
54                 Asserts.assertTrue(new Manifest(is).getAttributes("ns7.txt")
55                         .keySet().stream()
56                         .anyMatch(s -> s.toString().contains("SHA-256")));
57             }
58             try (InputStream is = jf.getInputStream(
59                     jf.getEntry("META-INF/ME.SF"))) {
60                 Asserts.assertTrue(new Manifest(is).getAttributes("ns7.txt")
61                         .keySet().stream()
62                         .anyMatch(s -> s.toString().contains("SHA-256")));
63             }
64         }
65     }
66 }
