24 /*
25  * @test
26  * @bug 8021789
27  * @summary jarsigner parses alias as command line option (depending on locale)
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.util.JarUtils;
33 
34 import java.nio.file.Files;
35 import java.nio.file.Path;
36 import java.util.List;
37 
38 public class Collator {
39     public static void main(String[] args) throws Exception {
40 
41         Files.write(Path.of("collator"), List.of("12345"));
42         JarUtils.createJarFile(
43                 Path.of("collator.jar"), Path.of("."), Path.of("collator"));
44 
45         SecurityTools.keytool("-storepass changeit -keypass changeit "
46                 + "-keyalg rsa -keystore collator.jks -alias debug "
47                 + "-dname CN=debug -genkey -validity 300")
48                 .shouldHaveExitValue(0);
49 
50         // use "debug" as alias name
51         SecurityTools.jarsigner("-keystore collator.jks "
52                 + "-storepass changeit collator.jar debug")
53                 .shouldHaveExitValue(0);
54 
55         // use "" as alias name (although there will be a warning)
56         SecurityTools.jarsigner("-keystore", "collator.jks",
57                 "-storepass", "changeit", "-verify", "collator.jar", "")
58                 .shouldHaveExitValue(0);
59     }
60 }
