30 # set a few environment variables so that the shell-script can run stand-alone
31 # in the source directory
32 if [ "${TESTSRC}" = "" ] ; then
33    TESTSRC="."
34 fi
35 
36 if [ "${TESTJAVA}" = "" ] ; then
37    echo "TESTJAVA not set.  Test cannot execute."
38    echo "FAILED!!!"
39    exit 1
40 fi
41 
42 # set platform-dependent variables
43 OS=`uname -s`
44 case "$OS" in
45   SunOS )
46     FILESEP="/"
47     ;;
48   Linux )
49     FILESEP="/"
50     ;;
51   Darwin )
52     FILESEP="/"
53     ;;
54   AIX )
55     PATHSEP=":"
56     FILESEP="/"
57     ;;
58   CYGWIN* )
59     FILESEP="/"
60     ;;
61   Windows* )
62     FILESEP="\\"
63     ;;
64   * )
65     echo "Unrecognized system!"
66     exit 1;
67     ;;
68 esac
69 
70 ${TESTJAVA}${FILESEP}bin${FILESEP}keytool ${TESTTOOLVMOPTS} \
71         -list -v \
72         -keystore ${TESTSRC}${FILESEP}CloneKeyAskPassword.jks \
73         -storepass test123
74 
75 exit $?
