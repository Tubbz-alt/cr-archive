24 import java.security.NoSuchAlgorithmException;
25 import java.security.Signature;
26 
27 /*
28  * This class is used for returning some specific JDK information.
29  */
30 public class JdkUtils {
31 
32     static final String M_JAVA_RUNTIME_VERSION = "javaRuntimeVersion";
33     static final String M_IS_SUPPORTED_SIGALG = "isSupportedSigalg";
34 
35     // Returns the JDK build version.
36     static String javaRuntimeVersion() {
37         return System.getProperty("java.runtime.version");
38     }
39 
40     // Checks if the specified signature algorithm is supported by the JDK.
41     static boolean isSupportedSigalg(String sigalg) {
42         boolean isSupported = false;
43         try {
44             isSupported = Signature.getInstance(sigalg) != null;
45         } catch (NoSuchAlgorithmException e) { }
46 
47         if (!isSupported) {
48             System.out.println(sigalg + " is not supported yet.");
49         }
50 
51         return isSupported;
52     }
53 
54     public static void main(String[] args) {
55         if (M_JAVA_RUNTIME_VERSION.equals(args[0])) {
56             System.out.print(javaRuntimeVersion());
57         } else if (M_IS_SUPPORTED_SIGALG.equals(args[0])) {
58             System.out.print(isSupportedSigalg(args[1]));
59         }
60     }
61 }
