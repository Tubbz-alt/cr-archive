24 import java.io.FileNotFoundException;
25 import java.io.FileOutputStream;
26 import java.io.IOException;
27 
28 /*
29  * A custom output stream that redirects the testing outputs to a file, called
30  * details.out. It also calls another output stream to save some outputs to
31  * other files.
32  */
33 public class DetailsOutputStream extends FileOutputStream {
34 
35     private PhaseOutputStream phaseOutputStream = new PhaseOutputStream();
36 
37     public DetailsOutputStream(String filename) throws FileNotFoundException {
38         super(filename != null && !filename.isEmpty() ? filename :
39             "details.out", true);
40     }
41 
42     public void transferPhase() throws IOException {
43         if (phaseOutputStream.isCorePhase()) {
44             phaseOutputStream.write(HtmlHelper.endHtml());
45             phaseOutputStream.write(HtmlHelper.endPre());
46         }
47 
48         phaseOutputStream.transfer();
49 
50         if (phaseOutputStream.isCorePhase()) {
51             phaseOutputStream.write(HtmlHelper.startHtml());
52             phaseOutputStream.write(HtmlHelper.startPre());
53         }
54     }
55 
56     @Override
57     public void write(byte[] b) throws IOException {
58         super.write(b);
59         phaseOutputStream.write(b);
60     }
61 
62     @Override
63     public void write(int b) throws IOException {
64         super.write(b);
65         phaseOutputStream.write(b);
66     }
67 
68     @Override
69     public void write(byte b[], int off, int len) throws IOException {
70         super.write(b, off, len);
71         phaseOutputStream.write(b, off, len);
72     }
73 
74     public void writeAnchorName(String name, String text) throws IOException {
75         super.write((text).getBytes());
76         super.write('\n');
77         phaseOutputStream.write(HtmlHelper.anchorName(name, text));
78         phaseOutputStream.write('\n');
79     }
80 }
