24 # @test
25 # @bug 8022761
26 # @summary regression: SecurityException is NOT thrown while trying to pack a wrongly signed Indexed Jar file
27 #
28 
29 if [ "${TESTJAVA}" = "" ] ; then
30   JAVAC_CMD=`which javac`
31   TESTJAVA=`dirname $JAVAC_CMD`/..
32 fi
33 
34 # set platform-dependent variables
35 OS=`uname -s`
36 case "$OS" in
37   Windows_* )
38     FS="\\"
39     ;;
40   * )
41     FS="/"
42     ;;
43 esac
44 
45 F=abcde
46 KS=jvindex.jks
47 JFILE=jvindex.jar
48 
49 KT="$TESTJAVA${FS}bin${FS}keytool ${TESTTOOLVMOPTS} -storepass changeit -keypass changeit \
50         -keystore $KS -keyalg rsa"
51 JAR="$TESTJAVA${FS}bin${FS}jar ${TESTTOOLVMOPTS}"
52 JARSIGNER="$TESTJAVA${FS}bin${FS}jarsigner ${TESTTOOLVMOPTS} -keystore $KS -storepass changeit"
53 
54 rm $F $KS $JFILE 2> /dev/null
55 
56 echo 12345 > $F
57 $JAR cvf $JFILE $F
58 
59 ERR=""
60 
61 $KT -alias a -dname CN=a -genkey -validity 300 || ERR="$ERR 1"
62 
63 $JARSIGNER $JFILE a || ERR="$ERR 2"
64 $JAR i $JFILE
65 
66 # Make sure the $F line has "sm" (signed and in manifest)
67 $JARSIGNER -verify -verbose $JFILE | grep $F | grep sm || ERR="$ERR 3"
68 
69 if [ "$ERR" = "" ]; then
70     exit 0
71 else
72     echo "ERR is $ERR"
73     exit 1
74 fi
