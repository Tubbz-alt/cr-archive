24 /*
25  * @test
26  * @bug 6890876 6950931
27  * @summary jarsigner can add CRL info into signed jar (updated)
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.process.OutputAnalyzer;
33 
34 public class Crl {
35     static OutputAnalyzer kt(String cmd) throws Exception {
36         return SecurityTools.keytool("-storepass changeit -keypass changeit "
37                 + "-keystore ks -keyalg rsa " + cmd);
38     }
39 
40     public static void main(String[] args) throws Exception {
41         kt("-alias a -dname CN=a -keyalg rsa -genkey -validity 300");
42         kt("-alias a -gencrl -id 1:1 -id 2:2 -file crl1")
43                 .shouldHaveExitValue(0);
44         kt("-alias a -gencrl -id 3:3 -id 4:4 -file crl2")
45                 .shouldHaveExitValue(0);
46         kt("-alias a -gencrl -id 5:1 -id 6:2 -file crl3")
47                 .shouldHaveExitValue(0);
48 
49         // Test keytool -printcrl
50 
51         kt("-printcrl -file crl1").shouldHaveExitValue(0);
52         kt("-printcrl -file crl2").shouldHaveExitValue(0);
53         kt("-printcrl -file crl3").shouldHaveExitValue(0);
54 
55         // Test keytool -ext crl
56 
57         kt("-alias b -dname CN=c -keyalg rsa -genkey -validity 300 "
58                 + "-ext crl=uri:http://www.example.com/crl")
59                 .shouldHaveExitValue(0);
60     }
61 }
