 24 import jdk.test.lib.process.OutputAnalyzer;
 25 import jdk.test.lib.util.JarUtils;
 26 
 27 /**
 28  * @test
 29  * @bug 8024302 8026037 8176320
 30  * @summary The test signs and verifies a jar file with -tsacert option
 31  * @library /lib/testlibrary warnings
 32  * @library /test/lib
 33  * @modules java.base/sun.security.pkcs
 34  *          java.base/sun.security.timestamp
 35  *          java.base/sun.security.tools.keytool
 36  *          java.base/sun.security.util
 37  *          java.base/sun.security.x509
 38  *          java.management
 39  * @build jdk.test.lib.util.JarUtils
 40  * @run main TsacertOptionTest
 41  */
 42 public class TsacertOptionTest extends Test {
 43 
 44     private static final String FILENAME = TsacertOptionTest.class.getName()
 45             + ".txt";
 46     private static final String SIGNING_KEY_ALIAS = "sign_alias";
 47     private static final String TSA_KEY_ALIAS = "ts";
 48 
 49     private static final String PASSWORD = "changeit";
 50 
 51     /**
 52      * The test signs and verifies a jar file with -tsacert option,
 53      * and checks that no warning was shown.
 54      * A certificate that is addressed in -tsacert option contains URL to TSA
 55      * in Subject Information Access extension.
 56      */
 57     public static void main(String[] args) throws Throwable {
 58         TsacertOptionTest test = new TsacertOptionTest();
 59         test.start();
 60     }
 61 
 62     void start() throws Throwable {
 63         // create a jar file that contains one file
 64         Utils.createFiles(FILENAME);
 65         JarUtils.createJar(UNSIGNED_JARFILE, FILENAME);
 66 
 67         // create key pair for jar signing
 68         keytool(
 69                 "-genkey",
 70                 "-alias", CA_KEY_ALIAS,
 71                 "-keyalg", KEY_ALG,
 72                 "-keysize", Integer.toString(KEY_SIZE),
 73                 "-keystore", KEYSTORE,
 74                 "-storepass", PASSWORD,
 75                 "-keypass", PASSWORD,
 76                 "-dname", "CN=CA",
 77                 "-ext", "bc:c",
 78                 "-validity", Integer.toString(VALIDITY)).shouldHaveExitValue(0);
 79         keytool(
 80                 "-genkey",
 81                 "-alias", SIGNING_KEY_ALIAS,
 82                 "-keyalg", KEY_ALG,
 83                 "-keysize", Integer.toString(KEY_SIZE),
 84                 "-keystore", KEYSTORE,
 85                 "-storepass", PASSWORD,
 86                 "-keypass", PASSWORD,
 87                 "-dname", "CN=Test").shouldHaveExitValue(0);
 88         keytool(
 89                 "-certreq",
 90                 "-alias", SIGNING_KEY_ALIAS,
 91                 "-keystore", KEYSTORE,
 92                 "-storepass", PASSWORD,
 93                 "-keypass", PASSWORD,
 94                 "-file", "certreq").shouldHaveExitValue(0);
 95         keytool(
 96                 "-gencert",
 97                 "-alias", CA_KEY_ALIAS,
 98                 "-keystore", KEYSTORE,
 99                 "-storepass", PASSWORD,
100                 "-keypass", PASSWORD,
101                 "-validity", Integer.toString(VALIDITY),
102                 "-infile", "certreq",
103                 "-outfile", "cert").shouldHaveExitValue(0);
104         keytool(
105                 "-importcert",
106                 "-alias", SIGNING_KEY_ALIAS,
107                 "-keystore", KEYSTORE,
108                 "-storepass", PASSWORD,
109                 "-keypass", PASSWORD,
110                 "-file", "cert").shouldHaveExitValue(0);
111 
112 
113         try (TimestampCheck.Handler tsa = TimestampCheck.Handler.init(0,
114                 KEYSTORE)) {
115 
116             // look for free network port for TSA service
117             int port = tsa.getPort();
118             String host = "127.0.0.1";
119             String tsaUrl = "http://" + host + ":" + port;
120 
121             // create key pair for TSA service
122             // SubjectInfoAccess extension contains URL to TSA service
123             keytool(
124                     "-genkey",
125                     "-v",
126                     "-alias", TSA_KEY_ALIAS,
127                     "-keyalg", KEY_ALG,
128                     "-keysize", Integer.toString(KEY_SIZE),
129                     "-keystore", KEYSTORE,
130                     "-storepass", PASSWORD,
131                     "-keypass", PASSWORD,
132                     "-dname", "CN=TSA",
133                     "-ext", "ExtendedkeyUsage:critical=timeStamping",
134                     "-ext", "SubjectInfoAccess=timeStamping:URI:" + tsaUrl,
135                     "-validity", Integer.toString(VALIDITY)).shouldHaveExitValue(0);
136 
137             // start TSA
138             tsa.start();
139 
140             // sign jar file
141             // specify -tsadigestalg option because
142             // TSA server uses SHA-1 digest algorithm
143              OutputAnalyzer analyzer = jarsigner(
144                     "-J-Dhttp.proxyHost=",
145                     "-J-Dhttp.proxyPort=",
146                     "-J-Djava.net.useSystemProxies=",
147                     "-verbose",
148                     "-keystore", KEYSTORE,
149                     "-storepass", PASSWORD,
150                     "-keypass", PASSWORD,
151                     "-signedjar", SIGNED_JARFILE,
152                     "-tsacert", TSA_KEY_ALIAS,
153                     "-tsadigestalg", "SHA-1",
154                     UNSIGNED_JARFILE,
155                     SIGNING_KEY_ALIAS);
156 
157             analyzer.shouldHaveExitValue(0);
158             analyzer.stdoutShouldNotContain(WARNING);
159             analyzer.shouldContain(JAR_SIGNED);
160 
161             // verify signed jar
162             analyzer = jarsigner(
163                     "-verbose",
164                     "-verify",
165                     "-keystore", KEYSTORE,
166                     "-storepass", PASSWORD,
167                     SIGNED_JARFILE);
168 
169             analyzer.shouldHaveExitValue(0);
170             analyzer.stdoutShouldNotContain(WARNING);
171             analyzer.shouldContain(JAR_VERIFIED);
172         }
173 
174         System.out.println("Test passed");
175     }
176 
177 }
