 24 import jdk.test.lib.process.OutputAnalyzer;
 25 import jdk.test.lib.util.JarUtils;
 26 
 27 /**
 28  * @test
 29  * @bug 8024302 8026037
 30  * @summary Test for badExtendedKeyUsage warning
 31  * @library /test/lib ../
 32  * @build jdk.test.lib.util.JarUtils
 33  * @run main BadExtendedKeyUsageTest
 34  */
 35 public class BadExtendedKeyUsageTest extends Test {
 36 
 37     /**
 38      * The test signs and verifies a jar that contains entries
 39      * whose signer certificate's ExtendedKeyUsage extension
 40      * doesn't allow code signing (badExtendedKeyUsage).
 41      * Warning message is expected.
 42      */
 43     public static void main(String[] args) throws Throwable {
 44         BadExtendedKeyUsageTest test = new BadExtendedKeyUsageTest();
 45         test.start();
 46     }
 47 
 48     private void start() throws Throwable {
 49         // create a jar file that contains one class file
 50         Utils.createFiles(FIRST_FILE);
 51         JarUtils.createJar(UNSIGNED_JARFILE, FIRST_FILE);
 52 
 53         // create a certificate whose signer certificate's
 54         // ExtendedKeyUsage extension doesn't allow code signing
 55         // create key pair for jar signing
 56         createAlias(CA_KEY_ALIAS);
 57         createAlias(KEY_ALIAS);
 58 
 59         issueCert(
 60                 KEY_ALIAS,
 61                 "-ext", "ExtendedkeyUsage=serverAuth",
 62                 "-validity", Integer.toString(VALIDITY));
 63 
 64         // sign jar
 65         OutputAnalyzer analyzer = jarsigner(
 66                 "-verbose",
 67                 "-keystore", KEYSTORE,
 68                 "-storepass", PASSWORD,
 69                 "-keypass", PASSWORD,
 70                 "-signedjar", SIGNED_JARFILE,
 71                 UNSIGNED_JARFILE,
 72                 KEY_ALIAS);
 73 
 74         checkSigning(analyzer, BAD_EXTENDED_KEY_USAGE_SIGNING_WARNING);
 75 
 76         // verify signed jar
 77         analyzer = jarsigner(
 78                 "-verify",
 79                 "-verbose",
 80                 "-keystore", KEYSTORE,
 81                 "-storepass", PASSWORD,
 82                 "-keypass", PASSWORD,
 83                 SIGNED_JARFILE);
 84 
 85         checkVerifying(analyzer, 0, BAD_EXTENDED_KEY_USAGE_VERIFYING_WARNING);
 86 
 87         // verity signed jar in strict mode
 88         analyzer = jarsigner(
 89                 "-verify",
 90                 "-verbose",
 91                 "-strict",
 92                 "-keystore", KEYSTORE,
 93                 "-storepass", PASSWORD,
 94                 "-keypass", PASSWORD,
 95                 SIGNED_JARFILE);
 96 
 97         checkVerifying(analyzer, BAD_EXTENDED_KEY_USAGE_EXIT_CODE,
 98                 BAD_EXTENDED_KEY_USAGE_VERIFYING_WARNING);
 99 
100         System.out.println("Test passed");
101     }
102 
103 }
