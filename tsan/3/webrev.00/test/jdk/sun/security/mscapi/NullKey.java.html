24 import java.security.InvalidKeyException;
25 import java.security.PublicKey;
26 import java.security.Signature;
27 import java.util.List;
28 
29 /**
30  * @test
31  * @bug 8225180
32  * @requires os.family == "windows"
33  * @summary SunMSCAPI Signature should throw InvalidKeyException when
34  *          initialized with a null key
35  */
36 
37 public class NullKey {
38     public static void main(String[] args) throws Exception {
39         for (String alg : List.of(
40                 "SHA256withRSA", "SHA256withECDSA", "RSASSA-PSS")) {
41             Signature sig = Signature.getInstance(alg, "SunMSCAPI");
42             try {
43                 sig.initSign(null);
44             } catch (InvalidKeyException e) {
45                 // Expected
46             }
47             try {
48                 sig.initVerify((PublicKey)null);
49             } catch (InvalidKeyException e) {
50                 // Expected
51             }
52         }
53     }
54 }
