 24 /*
 25  * @test
 26  * @bug 8223063
 27  * @requires os.family == "windows"
 28  * @library /test/lib
 29  * @summary Support CNG RSA keys
 30  */
 31 
 32 import jdk.test.lib.SecurityTools;
 33 import jdk.test.lib.process.ProcessTools;
 34 
 35 import java.io.File;
 36 import java.security.KeyStore;
 37 import java.security.MessageDigest;
 38 import java.security.PrivateKey;
 39 import java.security.PublicKey;
 40 import java.security.Signature;
 41 import java.security.cert.X509Certificate;
 42 import java.util.List;
 43 import java.util.Random;
 44 
 45 public class VeryLongAlias {
 46 
 47     static String alias = String.format("%0512d", new Random().nextInt(100000));
 48 
 49     public static void main(String[] args) throws Throwable {
 50 
 51         SecurityTools.keytool("-genkeypair -storetype pkcs12 -keystore ks"
 52                 + " -storepass changeit -keyalg RSA -dname CN=A -alias "
 53                 + alias);
 54         String id = ((X509Certificate)KeyStore.getInstance(
 55                     new File("ks"), "changeit".toCharArray())
 56                 .getCertificate(alias)).getSerialNumber().toString(16);
 57         try {
 58             // Importing pkcs12 file. Long alias is only supported by CNG.
 59             ProcessTools.executeCommand("certutil", "-v", "-p", "changeit",
 60                         "-csp", "Microsoft Software Key Storage Provider",
 61                         "-user", "-importpfx", "MY", "ks", "NoRoot,NoExport")
 62                     .shouldHaveExitValue(0);
 63             test();
 64         } finally {
 65             ProcessTools.executeCommand("certutil", "-user", "-delstore", "MY",
 66                     id);
 67         }
 68     }
 69 
 70     static void test() throws Exception {
 71 
 72         char[] pass = "changeit".toCharArray();
 73 
 74         KeyStore k1 = KeyStore.getInstance("Windows-MY");
 75         k1.load(null, null);
 76 
 77         KeyStore k2 = KeyStore.getInstance(new File("ks"), pass);
 78 
 79         PrivateKey p1 = (PrivateKey)k1.getKey(alias, null);
 80         PublicKey u1 = k1.getCertificate(alias).getPublicKey();
 81 
 82         PrivateKey p2 = (PrivateKey)k2.getKey(alias, pass);
 83         PublicKey u2 = k2.getCertificate(alias).getPublicKey();
 84 
 85         System.out.println(p1.toString());
 86         System.out.println(u1.toString());
 87         if (!p1.toString().contains("type=CNG")) {
 88             throw new Exception("Not a CNG key");
 89         }
 90 
 91         testSignature(p1, u1);
 92         testSignature(p1, u2);
 93         testSignature(p2, u1);
 94         testSignature(p2, u2);
 95     }
 96 
 97     static void testSignature(PrivateKey p, PublicKey u) throws Exception {
 98         byte[] data = "hello".getBytes();
 99         for (String alg : List.of(
100                 "NONEwithRSA", "SHA1withRSA",
101                 "SHA256withRSA", "SHA512withRSA")) {
102             if (alg.contains("NONE")) {
103                 data = MessageDigest.getInstance("SHA-256").digest(data);
104             }
105             Signature s1 = Signature.getInstance(alg);
106             Signature s2 = Signature.getInstance(alg);
107             s1.initSign(p);
108             s2.initVerify(u);
109             s1.update(data);
110             s2.update(data);
111             if (!s2.verify(s1.sign())) {
112                 throw new Exception("Error");
113             }
114         }
115     }
116 }
