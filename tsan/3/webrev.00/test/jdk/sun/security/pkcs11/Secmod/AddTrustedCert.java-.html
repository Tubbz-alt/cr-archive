 24 /*
 25  * @test
 26  * @bug 6298106
 27  * @summary make sure we can add a trusted cert to the NSS KeyStore module
 28  * @author Andreas Sterbenz
 29  * @library /test/lib ..
 30  * @modules jdk.crypto.cryptoki
 31  * @run main/othervm AddTrustedCert
 32  * @run main/othervm AddTrustedCert sm policy
 33  */
 34 
 35 import java.io.File;
 36 import java.io.FileInputStream;
 37 import java.io.InputStream;
 38 import java.security.KeyStore;
 39 import java.security.KeyStore.TrustedCertificateEntry;
 40 import java.security.Provider;
 41 import java.security.Security;
 42 import java.security.cert.CertificateFactory;
 43 import java.security.cert.X509Certificate;
 44 import java.util.Collection;
 45 import java.util.Collections;
 46 import java.util.TreeSet;
 47 
 48 public class AddTrustedCert extends SecmodTest {
 49 
 50     public static void main(String[] args) throws Exception {
 51         if (args.length > 1 && "sm".equals(args[0])) {
 52             System.setProperty("java.security.policy",
 53                     BASE + File.separator + args[1]);
 54         }
 55 
 56         if (initSecmod() == false) {
 57             return;
 58         }
 59 
 60         X509Certificate cert;
 61         try (InputStream in = new FileInputStream(BASE + SEP + "anchor.cer")) {
 62             CertificateFactory factory =
 63                     CertificateFactory.getInstance("X.509");
 64             cert = (X509Certificate)factory.generateCertificate(in);
 65         }
 66 
 67         String configName = BASE + SEP + "nss.cfg";
 68         Provider p = getSunPKCS11(configName);
 69 
 70         System.out.println(p);
 71         Security.addProvider(p);
 72 
 73         if (args.length > 1 && "sm".equals(args[0])) {
 74             System.setProperty("java.security.policy",
 75                     BASE + File.separator + args[1]);
 76             System.setSecurityManager(new SecurityManager());
 77         }
 78 
 79         KeyStore ks = KeyStore.getInstance(PKCS11, p);
 80         ks.load(null, password);
 81         Collection<String> aliases = new TreeSet<>(Collections.list(
 82                 ks.aliases()));
 83         System.out.println("entries: " + aliases.size());
 84         System.out.println(aliases);
 85         int size1 = aliases.size();
 86 
 87         String alias = "anchor";
 88         if (ks.containsAlias(alias)) {
 89             throw new Exception("Alias exists: " + alias);
 90         }
 91 
 92         ks.setCertificateEntry(alias, cert);
 93         KeyStore.Entry first = ks.getEntry(alias, null);
 94         System.out.println("first entry = " + first);
 95         if (!ks.entryInstanceOf(alias, TrustedCertificateEntry.class)) {
 96             throw new Exception("Unexpected first entry type: " + first);
 97         }
 98 
 99         ks.setCertificateEntry(alias, cert);
100         KeyStore.Entry second = ks.getEntry(alias, null);
101         System.out.println("second entry = " + second);
102         if (!ks.entryInstanceOf(alias, TrustedCertificateEntry.class)) {
103             throw new Exception("Unexpected second entry type: "
104                     + second);
105         }
106 
107         aliases = new TreeSet<>(Collections.list(ks.aliases()));
108         System.out.println("entries: " + aliases.size());
109         System.out.println(aliases);
110         int size2 = aliases.size();
111 
112         if ((size2 != size1 + 1) || (aliases.contains(alias) == false)) {
113             throw new Exception("Trusted cert not added");
114         }
115         X509Certificate cert2 = (X509Certificate)ks.getCertificate(alias);
116         if (cert.equals(cert2) == false) {
117             throw new Exception("KeyStore returned incorrect certificate");
118         }
119 
120         ks.deleteEntry(alias);
121         if (ks.containsAlias(alias)) {
122             throw new Exception("Alias still exists: " + alias);
123         }
124 
125         System.out.println("OK");
126     }
127 
128 }
