 36 import java.nio.ByteBuffer;
 37 import java.security.MessageDigest;
 38 import java.security.Provider;
 39 import java.util.Arrays;
 40 import java.util.Random;
 41 
 42 public class ByteBuffers extends PKCS11Test {
 43 






 44     public static void main(String[] args) throws Exception {
 45         main(new ByteBuffers(), args);
 46     }
 47 
 48     @Override
 49     public void main(Provider p) throws Exception {
 50         if (p.getService("MessageDigest", "MD5") == null) {
 51             System.out.println("Provider does not support MD5, skipping");
 52             return;
 53         }
 54 
 55         Random random = new Random();
 56         int n = 10 * 1024;
 57         byte[] t = new byte[n];
 58         random.nextBytes(t);
 59 
 60         MessageDigest md = MessageDigest.getInstance("MD5", p);
 61         byte[] d1 = md.digest(t);
















 62 
 63         // test 1: ByteBuffer with an accessible backing array
 64         ByteBuffer b1 = ByteBuffer.allocate(n + 256);
 65         b1.position(random.nextInt(256));
 66         b1.limit(b1.position() + n);
 67         ByteBuffer b2 = b1.slice();
 68         b2.put(t);
 69         b2.clear();
 70         byte[] d2 = digest(md, b2, random);
 71         if (Arrays.equals(d1, d2) == false) {
 72             throw new Exception("Test 1 failed");
 73         }
 74 
 75         // test 2: direct ByteBuffer
 76         ByteBuffer b3 = ByteBuffer.allocateDirect(t.length);
 77         b3.put(t);
 78         b3.clear();
 79         byte[] d3 = digest(md, b3, random);
 80         if (Arrays.equals(d1, d2) == false) {
 81             throw new Exception("Test 2 failed");
 82         }
 83 
 84         // test 3: ByteBuffer without an accessible backing array
 85         b2.clear();
 86         ByteBuffer b4 = b2.asReadOnlyBuffer();
 87         byte[] d4 = digest(md, b4, random);
 88         if (Arrays.equals(d1, d2) == false) {
 89             throw new Exception("Test 3 failed");
 90         }
 91         System.out.println("All tests passed");
 92     }
 93 
 94     private static byte[] digest(MessageDigest md, ByteBuffer b, Random random) throws Exception {

 95         int lim = b.limit();
 96         b.limit(random.nextInt(lim));
 97         md.update(b);
 98         if (b.hasRemaining()) {
 99             throw new Exception("Buffer not consumed");
100         }
101         b.limit(lim);
102         md.update(b);
103         if (b.hasRemaining()) {
104             throw new Exception("Buffer not consumed");
105         }
106         return md.digest();
107     }
108 }

 35 import java.nio.ByteBuffer;
 36 import java.security.*;

 37 import java.util.Arrays;
 38 import java.util.Random;
 39 
 40 public class ByteBuffers extends PKCS11Test {
 41 
 42     static final String[] ALGS = {
 43         "SHA-224", "SHA-256", "SHA-384", "SHA-512", "SHA-512/224", "SHA-512/256"
 44     };
 45 
 46     private static Random random = new Random();
 47 
 48     public static void main(String[] args) throws Exception {
 49         main(new ByteBuffers(), args);
 50     }
 51 
 52     @Override
 53     public void main(Provider p) throws Exception {






 54         int n = 10 * 1024;
 55         byte[] t = new byte[n];
 56         random.nextBytes(t);
 57 
 58         for (String alg : ALGS) {
 59             runTest(p, alg, t);
 60         }
 61     }
 62 
 63     private void runTest(Provider p, String alg, byte[] data) throws Exception {
 64         System.out.println("Test against " + p.getName() + " and " + alg);
 65         MessageDigest md;
 66         try {
 67             md = MessageDigest.getInstance(alg, p);
 68         } catch (NoSuchAlgorithmException e) {
 69             System.out.println("Skip " + alg + " due to no support");
 70             return;
 71         }
 72 
 73         byte[] d1 = md.digest(data);
 74 
 75         int n = data.length;
 76 
 77         // test 1: ByteBuffer with an accessible backing array
 78         ByteBuffer b1 = ByteBuffer.allocate(n + 256);
 79         b1.position(random.nextInt(256));
 80         b1.limit(b1.position() + n);
 81         ByteBuffer b2 = b1.slice();
 82         b2.put(data);
 83         b2.clear();
 84         byte[] d2 = digest(md, b2);
 85         if (Arrays.equals(d1, d2) == false) {
 86             throw new Exception("Test 1 failed");
 87         }
 88 
 89         // test 2: direct ByteBuffer
 90         ByteBuffer b3 = ByteBuffer.allocateDirect(n);
 91         b3.put(data);
 92         b3.clear();
 93         byte[] d3 = digest(md, b3);
 94         if (Arrays.equals(d1, d2) == false) {
 95             throw new Exception("Test 2 failed");
 96         }
 97 
 98         // test 3: ByteBuffer without an accessible backing array
 99         b2.clear();
100         ByteBuffer b4 = b2.asReadOnlyBuffer();
101         byte[] d4 = digest(md, b4);
102         if (Arrays.equals(d1, d2) == false) {
103             throw new Exception("Test 3 failed");
104         }
105         System.out.println("All tests passed");
106     }
107 
108     private static byte[] digest(MessageDigest md, ByteBuffer b)
109             throws Exception {
110         int lim = b.limit();
111         b.limit(random.nextInt(lim));
112         md.update(b);
113         if (b.hasRemaining()) {
114             throw new Exception("Buffer not consumed");
115         }
116         b.limit(lim);
117         md.update(b);
118         if (b.hasRemaining()) {
119             throw new Exception("Buffer not consumed");
120         }
121         return md.digest();
122     }
123 }
