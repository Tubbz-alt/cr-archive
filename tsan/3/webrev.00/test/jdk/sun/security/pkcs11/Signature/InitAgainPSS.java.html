23 import java.security.*;
24 import java.security.spec.*;
25 
26 /**
27  * @test
28  * @bug 8080462
29  * @summary Make sure old state is cleared when init is called again
30  * @library /test/lib ..
31  * @modules jdk.crypto.cryptoki
32  */
33 public class InitAgainPSS extends PKCS11Test {
34 
35     public static void main(String[] args) throws Exception {
36         main(new InitAgainPSS(), args);
37     }
38 
39     @Override
40     public void main(Provider p) throws Exception {
41         Signature s1;
42         try {
43             s1 = Signature.getInstance("RSASSA-PSS", p);
44         } catch (NoSuchAlgorithmException e) {
45             System.out.println("Skip testing RSASSA-PSS" +
46                 " due to no support");
47             return;
48         }
49 
50         byte[] msg = "hello".getBytes();
51 
52         Signature s2 = Signature.getInstance("RSASSA-PSS", p);
53 
54         PSSParameterSpec params = new PSSParameterSpec("SHA-256", "MGF1",
55             new MGF1ParameterSpec("SHA-256"), 32,
56             PSSParameterSpec.TRAILER_FIELD_BC);
57         s1.setParameter(params);
58         s2.setParameter(params);
59 
60         KeyPairGenerator kpg = KeyPairGenerator.getInstance("RSA", p);
61         kpg.initialize(2048);
62         KeyPair kp = kpg.generateKeyPair();
63 
64         s1.initSign(kp.getPrivate());
65 
66         s1.update(msg);
67         s1.initSign(kp.getPrivate());
68         s1.update(msg);
69         // Data digested in s1:
70         // Before this fix, msg | msg
71         // After this fix, msg
72 
73         s2.initVerify(kp.getPublic());
74         s2.update(msg);
75         s2.initVerify(kp.getPublic());
76         s2.update(msg);
77         s2.initVerify(kp.getPublic());
78         s2.update(msg);
79         // Data digested in s2:
80         // Before this fix, msg | msg | msg
81         // After this fix, msg
82 
83         if (!s2.verify(s1.sign())) {
84             throw new Exception();
85         }
86     }
87 }
