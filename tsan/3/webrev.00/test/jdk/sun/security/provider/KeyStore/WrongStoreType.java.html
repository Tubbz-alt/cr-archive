24 /*
25  * @test
26  * @bug 8234744
27  * @summary KeyStore.store can write wrong type of file
28  */
29 
30 import java.io.File;
31 import java.io.FileInputStream;
32 import java.io.FileOutputStream;
33 import java.io.InputStream;
34 import java.io.OutputStream;
35 import java.nio.file.Files;
36 import java.nio.file.Path;
37 import java.security.KeyStore;
38 
39 public class WrongStoreType {
40     public static void main(String ... args) throws Exception {
41 
42         char[] password = "changeit".toCharArray();
43         KeyStore ks = KeyStore.getInstance("PKCS12");
44 
45         ks.load(null, null);
46         System.out.println(ks.getType());
47 
48         Files.createFile(Path.of("emptyfile"));
49         try (InputStream in = new FileInputStream("emptyfile")) {
50             ks.load(in, password);
51         } catch (Exception e) {
52             System.out.println(e);
53         }
54 
55         System.out.println(ks.getType());
56         try (OutputStream out = new FileOutputStream("newfile")) {
57             ks.store(out, password);
58         }
59 
60         ks = KeyStore.getInstance(new File("newfile"), password);
61         String type = ks.getType();
62         if (!type.equalsIgnoreCase("PKCS12")) {
63             throw new Exception("see storetype " + type);
64         }
65     }
66 }
