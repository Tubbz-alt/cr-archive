24 /*
25  * @test
26  * @bug 4503229
27  * @summary default RSA KeyFactory can return broken RSAPrivateCrtKey objects
28  *      This test was taken directly from the bug report, which
29  *      was fixed in the crippled JSAFE provider, and needed
30  *      to be brought forward into JSSE.
31  * @author Brad Wetmore
32  */
33 
34 import java.security.*;
35 import java.security.interfaces.*;
36 import java.security.spec.*;
37 import java.math.BigInteger;
38 
39 public class BrokenRSAPrivateCrtKey {
40     public static void main(String[] args) throws Exception {
41         KeyPairGenerator generator =
42                 KeyPairGenerator.getInstance("RSA", "SunJSSE");
43         generator.initialize(512);
44 
45         KeyPair pair = generator.generateKeyPair();
46 
47         RSAPrivateCrtKey privatekey = (RSAPrivateCrtKey) pair.getPrivate();
48 
49         RSAPrivateCrtKeySpec spec =
50                 new RSAPrivateCrtKeySpec(privatekey.getModulus(),
51                 privatekey.getPublicExponent(),
52                 privatekey.getPrivateExponent(),
53                 privatekey.getPrimeP(), privatekey.getPrimeQ(),
54                 privatekey.getPrimeExponentP(),
55                 privatekey.getPrimeExponentQ(),
56                 privatekey.getCrtCoefficient());
57 
58         KeyFactory factory = KeyFactory.getInstance("RSA", "SunJSSE");
59 
60         PrivateKey privatekey2 = factory.generatePrivate(spec);
61 
62         BigInteger pe =
63                 ((RSAPrivateCrtKey) privatekey2).getPublicExponent();
64 
65         System.out.println("public exponent: " + pe);
66     }
67 }
