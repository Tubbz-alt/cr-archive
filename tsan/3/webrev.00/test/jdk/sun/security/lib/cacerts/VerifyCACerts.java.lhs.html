238     // Exception list to 90 days expiry policy
239     // No error will be reported if certificate in this list expires
240     private static final HashSet<String> EXPIRY_EXC_ENTRIES = new HashSet<>();








241 
242     // Ninety days in milliseconds
243     private static final long NINETY_DAYS = 7776000000L;
244 
245     private static boolean atLeastOneFailed = false;
246 
247     private static MessageDigest md;
248 
249     public static void main(String[] args) throws Exception {
250         System.out.println("cacerts file: " + CACERTS);
251         md = MessageDigest.getInstance("SHA-256");









252         KeyStore ks = KeyStore.getInstance("JKS");
253         ks.load(new FileInputStream(CACERTS), "changeit".toCharArray());
254 
255         // check the count of certs inside
256         if (ks.size() != COUNT) {
257             atLeastOneFailed = true;
258             System.err.println("ERROR: " + ks.size() + " entries, should be "
259                     + COUNT);
260         }
261 
262         // check that all entries in the map are in the keystore
263         for (String alias : FINGERPRINT_MAP.keySet()) {
264             if (!ks.isCertificateEntry(alias)) {
265                 atLeastOneFailed = true;
266                 System.err.println("ERROR: " + alias + " is not in cacerts");
267             }
268         }
269 
270         // pull all the trusted self-signed CA certs out of the cacerts file
271         // and verify their signatures
272         Enumeration<String> aliases = ks.aliases();
273         while (aliases.hasMoreElements()) {
274             String alias = aliases.nextElement();
275             System.out.println("\nVerifying " + alias);
276             if (!ks.isCertificateEntry(alias)) {
277                 atLeastOneFailed = true;
278                 System.err.println("ERROR: " + alias
279                         + " is not a trusted cert entry");
280             }
281             X509Certificate cert = (X509Certificate) ks.getCertificate(alias);
282             if (!checkFingerprint(alias, cert)) {
283                 atLeastOneFailed = true;
284                 System.err.println("ERROR: " + alias + " SHA-256 fingerprint is incorrect");
285             }
286             // Make sure cert can be self-verified
287             try {
288                 cert.verify(cert.getPublicKey());
289             } catch (Exception e) {
290                 atLeastOneFailed = true;
291                 System.err.println("ERROR: cert cannot be verified:"
292                         + e.getMessage());
293             }
294 
295             // Make sure cert is not expired or not yet valid
296             try {
297                 cert.checkValidity();
298             } catch (CertificateExpiredException cee) {
299                 if (!EXPIRY_EXC_ENTRIES.contains(alias)) {
300                     atLeastOneFailed = true;
301                     System.err.println("ERROR: cert is expired");
302                 }
303             } catch (CertificateNotYetValidException cne) {
304                 atLeastOneFailed = true;
305                 System.err.println("ERROR: cert is not yet valid");
306             }
307 
308             // If cert is within 90 days of expiring, mark as failure so
309             // that cert can be scheduled to be removed/renewed.
310             Date notAfter = cert.getNotAfter();
311             if (notAfter.getTime() - System.currentTimeMillis() < NINETY_DAYS) {
312                 if (!EXPIRY_EXC_ENTRIES.contains(alias)) {
313                     atLeastOneFailed = true;
314                     System.err.println("ERROR: cert \"" + alias + "\" expiry \""
315                             + notAfter.toString() + "\" will expire within 90 days");
316                 }
317             }
318         }
319 
320         if (atLeastOneFailed) {
321             throw new Exception("At least one cacert test failed");
322         }
323     }

325     private static boolean checkFingerprint(String alias, Certificate cert)
326             throws Exception {
327         String fingerprint = FINGERPRINT_MAP.get(alias);
328         if (fingerprint == null) {
329             // no entry for alias
330             return true;
331         }
332         System.out.println("Checking fingerprint of " + alias);
333         byte[] digest = md.digest(cert.getEncoded());
334         return fingerprint.equals(toHexString(digest));
335     }

337     private static String toHexString(byte[] block) {
338         StringBuilder buf = new StringBuilder();
339         int len = block.length;
340         for (int i = 0; i < len; i++) {
341             buf.append(String.format("%02X", block[i]));
342             if (i < len - 1) {
343                 buf.append(":");
344             }
345         }
346         return buf.toString();
347     }
