173 public class TestHostnameChecker {
174 
175     private final static String PATH = System.getProperty("test.src", ".");
176 
177     public static void main(String[] args) throws Exception {
178         CertificateFactory cf = CertificateFactory.getInstance("X.509");
179         InputStream in = new FileInputStream(new File(PATH, "cert1.crt"));
180         X509Certificate cert1 = (X509Certificate)cf.generateCertificate(in);
181         in.close();
182         in = new FileInputStream(new File(PATH, "cert2.crt"));
183         X509Certificate cert2 = (X509Certificate)cf.generateCertificate(in);
184         in.close();
185         in = new FileInputStream(new File(PATH, "cert3.crt"));
186         X509Certificate cert3 = (X509Certificate)cf.generateCertificate(in);
187         in.close();
188         in = new FileInputStream(new File(PATH, "cert4.crt"));
189         X509Certificate cert4 = (X509Certificate)cf.generateCertificate(in);
190         in.close();
191         in = new FileInputStream(new File(PATH, "cert5.crt"));
192         X509Certificate cert5 = (X509Certificate)cf.generateCertificate(in);
193         in.close();
194 
195         HostnameChecker checker = HostnameChecker.getInstance(
196                                         HostnameChecker.TYPE_TLS);
197         System.out.println("TLS tests.........");
198         System.out.println("==================");
199         check(checker, "foo1.com", cert1, true);
200         check(checker, "foo2.com", cert1, false);
201         check(checker, "1.2.3.4", cert2, false);
202         check(checker, "foo1.com", cert3, false);
203         check(checker, "foo2.com", cert3, false);
204         check(checker, "altfoo1.com", cert3, true);
205         check(checker, "altfoo2.com", cert3, true);
206         check(checker, "5.6.7.8", cert3, true);
207         check(checker, "foo.bar.com", cert4, true);
208         check(checker, "altfoo.bar.com", cert4, true);
209         check(checker, "2001:db8:3c4d:15::1a2f:1a2b", cert5, true);
210         check(checker, "2001:0db8:3c4d:0015:0000:0000:1a2f:1a2b", cert5, true);
211         check(checker, "2002:db8:3c4d:15::1a2f:1a2b", cert5, false);
212 
213         checker = HostnameChecker.getInstance(
214                                 HostnameChecker.TYPE_LDAP);
215         System.out.println();
216         System.out.println("LDAP tests.........");
217         System.out.println("==================");
218         check(checker, "foo1.com", cert1, true);
219         check(checker, "foo2.com", cert1, false);
220         check(checker, "foo1.com", cert3, false);
221         check(checker, "foo2.com", cert3, false);
222         check(checker, "altfoo1.com", cert3, true);
223         check(checker, "altfoo2.com", cert3, true);
224         check(checker, "5.6.7.8", cert3, true);
225         check(checker, "foo.bar.com", cert4, true);
226         check(checker, "altfoo.bar.com", cert4, false);
227     }
228 
229     private static void check(HostnameChecker checker, String name,
230                  X509Certificate cert, boolean expectedResult)
231                  throws Exception {
232         try {
233             checker.match(name, cert);
234             if (expectedResult == false) {
235                 throw new Exception("Passed invalid test: " + name);
236             }
237         } catch (CertificateException e) {
238             if (expectedResult == true) {
239                 throw e;
240             }
241         }
242         System.out.println("OK: " + name);
243     }
244 }
