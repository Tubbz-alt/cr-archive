24 /*
25  * @test
26  * @bug 8221801
27  * @library /test/lib
28  * @summary Update src/java.base/share/legal/public_suffix.md
29  */
30 
31 import jdk.test.lib.Asserts;
32 
33 import java.io.IOException;
34 import java.nio.file.Files;
35 import java.nio.file.Path;
36 import java.util.Objects;
37 import java.util.regex.Matcher;
38 import java.util.regex.Pattern;
39 
40 public class Versions {
41 
42     public static void main(String[] args) throws Exception {
43 
44         Path root = Path.of(System.getProperty("test.root"));
45         Path jdk = Path.of(System.getProperty("test.jdk"));
46 
47         Path version = root.resolve("../../make/data/publicsuffixlist/VERSION");
48         Path mdSrc = root.resolve("../../src/java.base/share/legal/public_suffix.md");
49         Path mdImage = jdk.resolve("legal/java.base/public_suffix.md");
50 
51         // Files in src should either both exist or not
52         if (!Files.exists(version) && !Files.exists(mdSrc)) {
53             System.out.println("Source not available. Cannot proceed.");
54             return;
55         }
56 
57         String s1 = findURL(version);
58         String s2 = findURL(mdSrc);
59 
60         Asserts.assertEQ(s1, s2);
61 
62         String s3 = findURL(mdImage);
63         Asserts.assertEQ(s2, s3);
64     }
65 
66     static Pattern URL_PATTERN = Pattern.compile(
67             "(https://raw.githubusercontent.com.*?public_suffix_list.dat)");
68 
69     static String findURL(Path p) throws IOException  {
70         return Files.lines(p)
71                 .map(Versions::matchURL)
72                 .filter(Objects::nonNull)
73                 .findFirst()
74                 .orElseThrow();
75     }
76 
77     static String matchURL(String input) {
78         Matcher m = URL_PATTERN.matcher(input);
79         return m.find() ? m.group(1) : null;
80     }
81 }
