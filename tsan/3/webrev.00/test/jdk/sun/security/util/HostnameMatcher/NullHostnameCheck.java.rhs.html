 54     public static void main(String[] args) throws Exception {
 55         String protocol = args[0];
 56         String password = "123456";
 57         SSLContext serverCtx = SSLContextBuilder.builder()
 58                 .keyStore(KeyStoreUtils.loadKeyStoreBase64(
 59                         keystoreB64, password))
 60                 .kmfPassphrase(password)
 61                 .protocol(protocol)
 62                 .build();


 63         SSLEngine serverEngine = serverCtx.createSSLEngine("localhost", -1);
 64         serverEngine.setUseClientMode(false);
 65 
 66         SSLContext clientCtx = SSLContext.getInstance(protocol);
 67         clientCtx.init(null, new TrustManager[] {
 68                 new X509TrustManager() {
 69                     @Override
 70                     public void checkClientTrusted(
 71                         X509Certificate[] x509Certificates, String s) {
 72                     }
 73 
 74                     @Override
 75                     public void checkServerTrusted(
 76                         X509Certificate[] x509Certificates, String s) {
 77                     }
 78 
 79                     @Override
 80                     public X509Certificate[] getAcceptedIssuers() {
 81                         return new X509Certificate[0];
 82                     }
 83                 }
 84         }, null);
 85 
 86         SSLEngine clientEngine = clientCtx.createSSLEngine();
 87         clientEngine.setUseClientMode(true);
 88 
 89         SSLParameters sslParameters = clientEngine.getSSLParameters();
 90         sslParameters.setEndpointIdentificationAlgorithm("HTTPS");
 91         clientEngine.setSSLParameters(sslParameters);
 92         try {
 93             handshake(clientEngine, serverEngine);
 94             throw new Exception("Value was not null.  Unexpected.");
 95         } catch (SSLHandshakeException e) {
 96             if (e.getCause() instanceof CertificateException) {
 97                 System.out.println("Correct Exception class thrown:\n\t" +
 98                         e.getMessage());
 99                 return;
100             }
101             throw e;
102         }
103     }

105     private static void handshake(SSLEngine clientEngine,
106             SSLEngine serverEngine) throws SSLException{
107         ByteBuffer cTOs = ByteBuffer.allocate(
108                 clientEngine.getSession().getPacketBufferSize());
109         ByteBuffer sTOc = ByteBuffer.allocate(
110                 serverEngine.getSession().getPacketBufferSize());
111 
112         ByteBuffer serverAppReadBuffer = ByteBuffer.allocate(
113                 serverEngine.getSession().getApplicationBufferSize());
114         ByteBuffer clientAppReadBuffer = ByteBuffer.allocate(
115                 clientEngine.getSession().getApplicationBufferSize());
116 
117         clientEngine.beginHandshake();
118         serverEngine.beginHandshake();
119 
120         ByteBuffer empty = ByteBuffer.allocate(0);
121 
122         SSLEngineResult clientResult;
123         SSLEngineResult serverResult;
124 
125         boolean clientHandshakeFinished = false;
126         boolean serverHandshakeFinished = false;
127 
128         do {
129             if (!clientHandshakeFinished) {
130                 clientResult = clientEngine.wrap(empty, cTOs);
131                 runDelegatedTasks(clientResult, clientEngine);
132 
133                 if (isHandshakeFinished(clientResult)) {
134                     clientHandshakeFinished = true;
135                 }
136             }
137 
138             if (!serverHandshakeFinished) {
139                 serverResult = serverEngine.wrap(empty, sTOc);
140                 runDelegatedTasks(serverResult, serverEngine);
141 
142                 if (isHandshakeFinished(serverResult)) {
143                     serverHandshakeFinished = true;
144                 }
145             }
146 
147             cTOs.flip();
148             sTOc.flip();
149 
150             if (!clientHandshakeFinished) {
151                 clientResult = clientEngine.unwrap(sTOc, clientAppReadBuffer);
152 
153                 runDelegatedTasks(clientResult, clientEngine);
154 
155                 if (isHandshakeFinished(clientResult)) {
156                     clientHandshakeFinished = true;
157                 }
158             }
159 
160             if (!serverHandshakeFinished) {
161                 serverResult = serverEngine.unwrap(cTOs, serverAppReadBuffer);
162                 runDelegatedTasks(serverResult, serverEngine);
163 
164                 if (isHandshakeFinished(serverResult)) {
165                     serverHandshakeFinished = true;
166                 }
167             }
168 
169             sTOc.compact();
170             cTOs.compact();
171         } while (!clientHandshakeFinished || !serverHandshakeFinished);
172     }
173 
174     private static boolean isHandshakeFinished(SSLEngineResult result) {
175         return result.getHandshakeStatus() ==
176                 SSLEngineResult.HandshakeStatus.FINISHED;
177     }
178 
179     private static void runDelegatedTasks(SSLEngineResult result,
180             SSLEngine engine) {
181         if (result.getHandshakeStatus() ==
182                 SSLEngineResult.HandshakeStatus.NEED_TASK) {
183             for (;;) {
184                 Runnable task = engine.getDelegatedTask();
185                 if (task == null) {
186                     break;
187                 }
188                 task.run();
189             }
190         }
191     }
