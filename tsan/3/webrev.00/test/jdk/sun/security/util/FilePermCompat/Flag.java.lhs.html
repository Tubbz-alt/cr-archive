24 /*
25  * @test
26  * @bug 8164705


27  * @summary check jdk.filepermission.canonicalize
28  * @run main/othervm/policy=flag.policy
29   *     -Djdk.io.permissionsUseCanonicalPath=true Flag true true
30  * @run main/othervm/policy=flag.policy
31   *     -Djdk.io.permissionsUseCanonicalPath=false Flag false true
32  * @run main/othervm/policy=flag.policy Flag false true
33  */
34 
35 import java.io.File;
36 import java.io.FilePermission;
37 import java.lang.*;

38 
39 public class Flag {
40     public static void main(String[] args) throws Exception {
41 
































































42         boolean test1;
43         boolean test2;
44 
45         String here = System.getProperty("user.dir");
46         File abs = new File(here, "x");
47         FilePermission fp1 = new FilePermission("x", "read");
48         FilePermission fp2 = new FilePermission(abs.toString(), "read");
49         test1 = fp1.equals(fp2);
50 
51         try {
52             System.getSecurityManager().checkPermission(fp2);
53             test2 = true;
54         } catch (SecurityException se) {
55             test2 = false;
56         }
57 
58         if (test1 != Boolean.parseBoolean(args[0]) ||
59                 test2 != Boolean.parseBoolean(args[1])) {
60             throw new Exception("Test failed: " + test1 + " " + test2);
61         }
62     }
63 }
