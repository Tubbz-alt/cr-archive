24 /*
25  * @test
26  * @bug 8022444
27  * @summary Test ObjectIdentifier.equals(Object obj)
28  * @modules java.base/sun.security.util
29  */
30 
31 import sun.security.util.ObjectIdentifier;
32 
33 public class OidEquals {
34     public static void main(String[] args) throws Exception {
35         ObjectIdentifier oid1 = new ObjectIdentifier("1.3.6.1.4.1.42.2.17");
36         ObjectIdentifier oid2 =
37                 new ObjectIdentifier(new int[]{1, 3, 6, 1, 4, 1, 42, 2, 17});
38         ObjectIdentifier oid3 = new ObjectIdentifier("1.2.3.4");
39 
40         assertEquals(oid1, oid1);
41         assertEquals(oid1, oid2);
42         assertNotEquals(oid1, oid3);
43         assertNotEquals(oid1, "1.3.6.1.4.1.42.2.17");
44 
45         System.out.println("Tests passed.");
46     }
47 
48     static void assertEquals(ObjectIdentifier oid, Object obj)
49             throws Exception {
50         if (!oid.equals(obj)) {
51             throw new Exception("The ObjectIdentifier " + oid.toString() +
52                     " should be equal to the Object " + obj.toString());
53         }
54     }
55 
56     static void assertNotEquals(ObjectIdentifier oid, Object obj)
57             throws Exception {
58         if (oid.equals(obj)) {
59             throw new Exception("The ObjectIdentifier " + oid.toString() +
60                     " should not be equal to the Object " + obj.toString());
61         }
62     }
63 }
