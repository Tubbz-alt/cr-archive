 24 /*
 25  * @test
 26  * @bug 7167293
 27  * @summary FtpURLConnection doesn't close FTP connection when FileNotFoundException is thrown
 28  * @library ../www/ftptest/
 29  * @build FtpServer FtpCommandHandler FtpAuthHandler FtpFileSystemHandler
 30  * @run main FtpURLConnectionLeak
 31  */
 32 import java.io.FileNotFoundException;
 33 import java.io.InputStream;
 34 import java.io.OutputStream;
 35 import java.net.URL;
 36 
 37 public class FtpURLConnectionLeak {
 38 
 39     public static void main(String[] args) throws Exception {
 40         FtpServer server = new FtpServer(0);
 41         server.setFileSystemHandler(new CustomFileSystemHandler("/"));
 42         server.setAuthHandler(new MyAuthHandler());
 43         int port = server.getLocalPort();
 44         server.start();
 45         URL url = new URL("ftp://localhost:" + port + "/filedoesNotExist.txt");
 46         try (server) {
 47             for (int i = 0; i < 3; i++) {
 48                 try {
 49                     InputStream stream = url.openStream();
 50                 } catch (FileNotFoundException expected) {
 51                     // should always reach this point since the path does not exist
 52                     System.out.println("caught expected " + expected);
 53                     int times = 1;
 54                     do {
 55                         // give some time to close the connection...
 56                         System.out.println("sleeping... " + times);
 57                         Thread.sleep(times * 1000);
 58                     } while (server.activeClientsCount() > 0 && times++ < 5);
 59 
 60                     if (server.activeClientsCount() > 0) {
 61                         server.killClients();
 62                         throw new RuntimeException("URLConnection didn't close the" +
 63                                 " FTP connection on FileNotFoundException");
 64                     }
 65                 }
 66             }
 67         }
 68     }
 69 
 70     static class CustomFileSystemHandler implements FtpFileSystemHandler {
 71 
 72         private String currentDir;
 73 
 74         public CustomFileSystemHandler(String path) {
 75             currentDir = path;
 76         }
 77 
 78         @Override
 79         public boolean cd(String path) {
 80             currentDir = path;
 81             return true;
 82         }
 83 
 84         @Override
 85         public boolean cdUp() {
 86             throw new UnsupportedOperationException("Not supported yet.");
 87         }
 88 
 89         @Override
 90         public String pwd() {
 91             throw new UnsupportedOperationException("Not supported yet.");
 92         }
 93 
 94         @Override
 95         public boolean fileExists(String name) {
 96             throw new UnsupportedOperationException("Not supported yet.");
 97         }
 98 
 99         @Override
100         public InputStream getFile(String name) {
101             return null; //return null so that server will return 550 File not found.
102         }
103 
104         @Override
105         public long getFileSize(String name) {
106             throw new UnsupportedOperationException("Not supported yet.");
107         }
108 
109         @Override
110         public InputStream listCurrentDir() {
111             return null;
112         }
113 
114         @Override
115         public OutputStream putFile(String name) {
116             throw new UnsupportedOperationException("Not supported yet.");
117         }
118 
119         @Override
120         public boolean removeFile(String name) {
121             throw new UnsupportedOperationException("Not supported yet.");
122         }
123 
124         @Override
125         public boolean mkdir(String name) {
126             throw new UnsupportedOperationException("Not supported yet.");
127         }
128 
129         @Override
130         public boolean rename(String from, String to) {
131             throw new UnsupportedOperationException("Not supported yet.");
132         }
133 
134     }
135 
136     static class MyAuthHandler implements FtpAuthHandler {
137 
138         @Override
139         public int authType() {
140             return 0;
141         }
142 
143         @Override
144         public boolean authenticate(String user, String password) {
145             return true;
146         }
147 
148         @Override
149         public boolean authenticate(String user, String password, String account) {
150             return true;
151         }
152     }
