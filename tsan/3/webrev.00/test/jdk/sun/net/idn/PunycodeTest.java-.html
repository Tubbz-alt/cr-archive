 60     static void fail(String msg, String input) {
 61         System.out.println(msg+" input: "+input);
 62         throw new RuntimeException(msg+" input: "+input);
 63     }

 66     public int testCount = 0;
 67 
 68     private int input_length, j;
 69     private int output_length[] = new int[1];
 70     private boolean case_flags[] = new boolean[unicode_max_length];
 71 
 72     public String testEncoding(String inputS) {
 73         char input[] = new char[unicode_max_length];
 74         int codept = 0;
 75         char uplus[] = new char[2];
 76         StringBuffer output;
 77         int c;
 78 
 79         /* Read the input code points: */
 80 
 81         input_length = 0;
 82 
 83         Scanner sc = new Scanner(inputS);
 84 
 85         while (sc.hasNext()) {  // need to stop at end of line
 86             try {
 87                 String next = sc.next();
 88                 uplus[0] = next.charAt(0);
 89                 uplus[1] = next.charAt(1);
 90                 codept = Integer.parseInt(next.substring(2), 16);
 91             } catch (Exception ex) {
 92                 fail(invalid_input, inputS);
 93             }
 94 
 95             if (uplus[1] != '+' || codept > Integer.MAX_VALUE) {
 96                 fail(invalid_input, inputS);
 97             }
 98 
 99             if (input_length == unicode_max_length) fail(too_big, inputS);
100 
101             if (uplus[0] == 'u') case_flags[input_length] = false;
102             else if (uplus[0] == 'U') case_flags[input_length] = true;
103             else fail(invalid_input, inputS);
104 
105             input[input_length++] = (char)codept;
106         }
107 
108         /* Encode: */
109 
110         output_length[0] = ace_max_length;
111         try {
112             output = Punycode.encode((new StringBuffer()).append(input, 0, input_length), case_flags);
113         } catch (Exception e) {
114             fail(invalid_input, inputS);
115             // never reach here, just to make compiler happy
116             return null;
117         }
118 
119         testCount++;
120         return output.toString();
121     }
122 
123     public String testDecoding(String inputS) {
124         char input[] = new char[0];
125         int pp;
126         StringBuffer output;
127 
128         /* Read the Punycode input string and convert to ASCII: */
129 
130         if (inputS.length() <= ace_max_length+2) {
131             input = inputS.toCharArray();
132         } else {
133             fail(invalid_input, inputS);
134         }
135         input_length = input.length;
136 
137         /* Decode: */
138 
139         output_length[0] = unicode_max_length;
140         try {
141             output = Punycode.decode((new StringBuffer()).append(input, 0, input_length), case_flags);
142         } catch (Exception e) {
143             fail(invalid_input, inputS);
144             // never reach here, just to make compiler happy
145             return null;
146         }
147 
148         /* Output the result: */
149         StringBuffer result = new StringBuffer();
150         for (j = 0;  j < output.length();  ++j) {
151             result.append(String.format("%s+%04X ",
152                     case_flags[j] ? "U" : "u",
153                     (int)output.charAt(j) ));
154         }
155 
156         testCount++;
157         return result.substring(0, result.length() - 1);
158     }

251   public static void main(String[] argv) throws Exception {
252       PunycodeTest mytest = new PunycodeTest();
253       for (int i = 0; i < testdata.length; i++) {
254           String encodeResult = mytest.testEncoding(testdata[i][1]);
255           String decodeResult = mytest.testDecoding(testdata[i][2]);
256 
257           checkResult(encodeResult, testdata[i][2]);
258           checkResult(decodeResult, testdata[i][1]);
259       }
260 
261       System.out.println("Test cases: " + mytest.testCount);
262   }
263 
264   public static void checkResult(String actual, String expected) {
265       if (!actual.equals(expected)) {
266           System.out.printf("\n%15s: %s\n", "FAILED", actual);
267           System.out.printf("%15s: %s\n\n", "should be", expected);
268           throw new RuntimeException("Punycode test failed.");
269       } else {
270           System.out.printf("%15s: %s\n", "SUCCEEDED", actual);
271       }
272   }
