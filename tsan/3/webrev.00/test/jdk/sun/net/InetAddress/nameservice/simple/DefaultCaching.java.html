 24 /* @test
 25  * @bug 6442088
 26  * @summary Change default DNS caching behavior for code not running under
 27  *          security manager.
 28  * @run main/othervm/timeout=200 -Dsun.net.inetaddr.ttl=20  DefaultCaching
 29  */
 30 import java.net.InetAddress;
 31 import java.net.UnknownHostException;
 32 import java.security.Security;
 33 import java.io.FileWriter;
 34 import java.io.PrintWriter;
 35 import java.io.BufferedWriter;
 36 import java.nio.file.Files;
 37 import java.nio.file.Path;
 38 import static java.nio.file.StandardCopyOption.REPLACE_EXISTING;
 39 
 40 public class DefaultCaching {
 41 
 42     public static void main(String args[]) throws Exception {
 43 
 44         String hostsFileNameSrc = System.getProperty("test.src", ".") + "/DefaultCachingHosts";
 45         String hostsFileName = System.getProperty("user.dir", ".") + "/DefaultCachingHosts";
 46         if (!hostsFileNameSrc.equals(hostsFileName)) {
 47             Files.copy(Path.of(hostsFileNameSrc), Path.of(hostsFileName), REPLACE_EXISTING);
 48             System.out.println("Host file created: " + hostsFileName);
 49         }
 50         System.setProperty("jdk.net.hosts.file", hostsFileName);
 51         // initial mapping
 52         // name service needs to resolve this.
 53         addMappingToHostsFile("theclub", "129.156.220.219", hostsFileName, false);
 54 
 55         test("theclub", "129.156.220.219", true);      // lk: 1
 56         test("luster", "1.16.20.2", false);            // lk: 2
 57 
 58         // name service now needs to know about luster
 59         addMappingToHostsFile("luster", "10.5.18.21", hostsFileName, true);
 60 
 61         test("luster", "1.16.20.2", false);            // lk: 2
 62         sleep(10+1);
 63         test("luster", "10.5.18.21", true, 3);          // lk: 3
 64         sleep(5);
 65 
 66         // new mapping for theclub and rewrite existing foo and luster mappings
 67         addMappingToHostsFile("theclub", "129.156.220.1", hostsFileName, false);
 68         addMappingToHostsFile("foo", "10.5.18.22", hostsFileName, true);
 69         addMappingToHostsFile("luster", "10.5.18.21", hostsFileName, true);
 70 
 71         test("theclub", "129.156.220.219", true, 3);
 72         test("luster", "10.5.18.21", true, 3);
 73         test("bar", "10.5.18.22", false, 4);
 74         test("foo", "10.5.18.22", true, 5);
 75 
 76         // now delay to see if theclub has expired
 77         sleep(5);
 78 
 79         test("foo", "10.5.18.22", true, 5);
 80         test("theclub", "129.156.220.1", true, 6);
 81 
 82         sleep(11);
 83         // now see if luster has expired
 84         test("luster", "10.5.18.21", true, 7);
 85         test("theclub", "129.156.220.1", true, 7);
 86 
 87         // now delay to see if 3rd has expired
 88         sleep(10+6);
 89 
 90         test("theclub", "129.156.220.1", true, 8);
 91         test("luster", "10.5.18.21", true, 8);
 92         test("foo", "10.5.18.22", true, 9);
 93     }
 94 
 95     /* throws RuntimeException if it fails */
 96 
 97     static void test(String host, String address,
 98                      boolean shouldSucceed, int count) {
 99         test(host, address, shouldSucceed);
100     }
101 
102     static void sleep(int seconds) {
103         try {
104             sleepms(seconds * 1000);
105         } catch (InterruptedException e) {
106             e.printStackTrace();
107         }
108     }
109 
110     static long sleepms(long millis) throws InterruptedException {
111         long start = System.nanoTime();
112         long ms = millis;
113         while (ms > 0) {
114             assert ms < Long.MAX_VALUE/1000_000L;
115             Thread.sleep(ms);
116             long elapsedms = (System.nanoTime() - start)/1000_000L;
117             ms = millis - elapsedms;
118         }
119         return millis - ms;
120     }
121 
122     static void test(String host, String address, boolean shouldSucceed) {
123         InetAddress addr = null;
124         try {
125             addr = InetAddress.getByName(host);
126             if (!shouldSucceed) {
127                 throw new RuntimeException(host+":"+address+": should fail (got "
128                                            + addr + ")");
129             }
130             if (!address.equals(addr.getHostAddress())) {
131                 throw new RuntimeException(host+"/"+address+": compare failed (found "
132                                            + addr + ")");
133             }
134             System.out.println("test: " + host + "/" + address
135                                + " succeeded - got " + addr);
136         } catch (UnknownHostException e) {
137             if (shouldSucceed) {
138                 throw new RuntimeException(host+":"+address+": should succeed");
139             } else {
140                 System.out.println("test: " + host + "/" + address
141                                    + " succeeded - got expected " + e);
142             }
143         }
144     }
145 
146 
147     private static void addMappingToHostsFile(String host,
148                                               String addr,
149                                               String hostsFileName,
150                                               boolean append)
151                                              throws Exception {
152         String mapping = addr + " " + host;
153         try (PrintWriter hfPWriter = new PrintWriter(new BufferedWriter(
154                 new FileWriter(hostsFileName, append)))) {
155             hfPWriter.println(mapping);
156         }
157     }
158 }
