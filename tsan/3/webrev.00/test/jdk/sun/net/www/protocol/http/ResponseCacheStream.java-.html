 34 import java.net.*;
 35 import java.io.*;
 36 import java.util.*;
 37 
 38 public class ResponseCacheStream implements HttpCallback {
 39 
 40     void okReply (HttpTransaction req) throws IOException {
 41         req.setResponseEntityBody ("Hello, This is the response body. Let's make it as long as possible since we need to test the cache mechanism.");
 42         req.sendResponse (200, "Ok");
 43             System.out.println ("Server: sent response");
 44         req.orderlyClose();
 45     }
 46 
 47     public void request (HttpTransaction req) {
 48         try {
 49             okReply (req);
 50         } catch (IOException e) {
 51             e.printStackTrace();
 52         }
 53     }
 54 
 55     static class MyCacheRequest extends CacheRequest {
 56         private OutputStream buf = null;
 57 
 58         public MyCacheRequest(OutputStream out) {
 59             buf = out;
 60         }
 61 
 62         public OutputStream getBody() throws IOException {
 63             return buf;
 64         }
 65 
 66         /**
 67          * Aborts the attempt to cache the response. If an IOException is
 68          * encountered while reading the response or writing to the cache,
 69          * the current cache store operation will be abandoned.
 70          */
 71         public void abort() {
 72         }
 73 
 74     }
 75 
 76     static class MyResponseCache extends ResponseCache {
 77         private ByteArrayOutputStream buf = new ByteArrayOutputStream(1024);
 78 
 79         public MyResponseCache() {
 80         }
 81 
 82         public CacheRequest put(URI uri, URLConnection conn) throws IOException {
 83             return new MyCacheRequest(buf);
 84         }
 85 
 86         public CacheResponse get(URI uri, String rqstMethod, Map<String, List<String>> rqstHeaders) throws IOException {
 87             return null;
 88         }
 89 
 90         public byte[] getBuffer() {
 91             return buf.toByteArray();
 92         }
 93     }
 94 
 95     static TestHttpServer server;
 96 
 97     public static void main(String[] args) throws Exception {
 98         MyResponseCache cache = new MyResponseCache();
 99         try {
100             ResponseCache.setDefault(cache);
101             server = new TestHttpServer (new ResponseCacheStream());
102             System.out.println ("Server: listening on port: " + server.getLocalPort());
103             URL url = new URL ("http://127.0.0.1:"+server.getLocalPort()+"/");
104             System.out.println ("Client: connecting to " + url);
105             HttpURLConnection urlc = (HttpURLConnection)url.openConnection();
106             InputStream is = urlc.getInputStream();
107             System.out.println("is is " + is.getClass() + ". And markSupported: " + is.markSupported());
108             if (is.markSupported()) {
109                 byte[] b = new byte[1024];
110                 byte[] b2 = new byte[32];
111                 int len;
112                 int count;
113                 is.mark(10);
114                 len = is.read(b, 0, 10);
115                 is.reset();
116                 len = 0;
117                 count = 0;
118                 do {
119                     len = is.read(b, count, 40 - count);
120                     if (len > 0)
121                         count += len;
122                 } while (len > 0);
123                 is.mark(20);
124                 len = is.read(b2, 0, 20);
125                 is.reset();
126                 len = is.read(b, count, 10);
127                 count += len;
128                 is.mark(20);
129                 len = is.read(b2, 0, 20);
130                 is.reset();
131                 do {
132                     len = is.read(b, count, 1024 - count);
133                     if (len > 0)
134                         count += len;
135                 } while (len > 0);
136                 is.close();
137                 String s1 = new String(b, 0 , count);
138                 String s2 = new String(cache.getBuffer(), 0 , count);
139                 if (! s1.equals(s2))
140                     throw new RuntimeException("cache got corrupted!");
141             }
142         } catch (Exception e) {
143             if (server != null) {
144                 server.terminate();
145             }
146             throw e;
147         }
148         server.terminate();
149     }
