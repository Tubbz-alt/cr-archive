24 /*
25  * @test
26  * @bug 4251209
27  * @summary Make sure that implicit filenames will be returned as "/"
28  * @modules java.base/sun.net
29  *          java.base/sun.net.www.http
30  * @library /test/lib
31  */
32 
33 import java.net.InetAddress;
34 import java.net.InetSocketAddress;
35 import java.net.URL;
36 import java.net.ServerSocket;
37 import sun.net.www.http.HttpClient;
38 import jdk.test.lib.net.URIBuilder;
39 
40 public class ImplicitFileName {
41 
42     public static void main(String[] args) throws Exception {
43 
44         ServerSocket ss = new ServerSocket();
45         InetAddress loopback = InetAddress.getLoopbackAddress();
46         ss.bind(new InetSocketAddress(loopback, 0));
47 
48         URL url = URIBuilder.newBuilder()
49             .scheme("http")
50             .loopback()
51             .port(ss.getLocalPort())
52             .toURL();
53 
54         HttpClient c = HttpClient.New(url);
55 
56         if (!c.getURLFile().equals("/")) {
57             throw new Exception("Implicit filename in URL " +
58                                 url.toString() + " is not '/'");
59         }
60     }
61 }
