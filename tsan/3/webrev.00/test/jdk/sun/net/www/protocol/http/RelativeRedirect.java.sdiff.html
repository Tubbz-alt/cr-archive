 34 import java.io.*;
 35 import java.net.*;
 36 
 37 public class RelativeRedirect implements HttpCallback {
 38     static int count = 0;
 39     static TestHttpServer server;
 40 
 41     static class MyAuthenticator extends Authenticator {
 42         public MyAuthenticator () {
 43             super ();
 44         }
 45 
 46         public PasswordAuthentication getPasswordAuthentication ()
 47         {
 48             return (new PasswordAuthentication ("user", "Wrongpassword".toCharArray()));
 49         }
 50     }
 51 
 52     void firstReply (HttpTransaction req) throws IOException {
 53         req.addResponseHeader ("Connection", "close");
 54         req.addResponseHeader ("Location", "/redirect/file.html");
 55         req.sendResponse (302, "Moved Permamently");
 56         req.orderlyClose();
 57     }
 58 
 59     void secondReply (HttpTransaction req) throws IOException {
 60         if (req.getRequestURI().toString().equals("/redirect/file.html") &&
 61             req.getRequestHeader("Host").equals("localhost:"+server.getLocalPort())) {
 62             req.setResponseEntityBody ("Hello .");
 63             req.sendResponse (200, "Ok");
 64         } else {
 65             req.setResponseEntityBody (req.getRequestURI().toString());
 66             req.sendResponse (400, "Bad request");
 67         }
 68         req.orderlyClose();
 69 
 70     }
 71     public void request (HttpTransaction req) {
 72         try {
 73             switch (count) {
 74             case 0:
 75                 // server redirect to /redirect/file.html
 76                 firstReply (req);
 77                 break;
 78             case 1:
 79                 // client retry to /redirect/file.html on same server
 80                 secondReply (req);
 81                 break;
 82             }
 83             count ++;
 84         } catch (IOException e) {
 85             e.printStackTrace();
 86         }
 87     }

 89     public static void main (String[] args) throws Exception {

 90         MyAuthenticator auth = new MyAuthenticator ();

 96             System.out.println ("client opening connection to: " + url);
 97             HttpURLConnection urlc = (HttpURLConnection)url.openConnection ();

 99             is.close();
100         } catch (Exception e) {
101             throw new RuntimeException(e);

103             if (server != null) {

 38 public class RelativeRedirect implements HttpCallback {

 53     void firstReply (HttpTransaction req) throws IOException {

 55         req.addResponseHeader ("Location", "/redirect/file.html");
 56         req.sendResponse (302, "Moved Permamently");

 58     }

 60     void secondReply (HttpTransaction req) throws IOException {
 61         if (req.getRequestURI().toString().equals("/redirect/file.html") &&
 62             req.getRequestHeader("Host").equals(authority(server.getLocalPort()))) {
 63             req.setResponseEntityBody ("Hello .");
 64             req.sendResponse (200, "Ok");
 65         } else {
 66             req.setResponseEntityBody (req.getRequestURI().toString());

 82                 break;

 87         }

 90    static String authority(int port) {
 91        InetAddress loopback = InetAddress.getLoopbackAddress();
 92        String hostaddr = loopback.getHostAddress();
 93        if (hostaddr.indexOf(':') > -1) {
 94            hostaddr = "[" + hostaddr + "]";
 95        }
 96        return hostaddr + ":" + port;
 97    }
 98 
 99     public static void main (String[] args) throws Exception {
100         InetAddress loopback = InetAddress.getLoopbackAddress();
101         MyAuthenticator auth = new MyAuthenticator ();
102         Authenticator.setDefault (auth);
103         try {
104             server = new TestHttpServer (new RelativeRedirect(), 1, 10, loopback, 0);
105             System.out.println ("Server: listening on port: " + server.getLocalPort());
106             URL url = new URL("http://" + authority(server.getLocalPort()));
107             System.out.println ("client opening connection to: " + url);
108             HttpURLConnection urlc = (HttpURLConnection)url.openConnection (Proxy.NO_PROXY);
109             InputStream is = urlc.getInputStream ();
110             is.close();
111         } catch (Exception e) {
112             throw new RuntimeException(e);
113         } finally {
