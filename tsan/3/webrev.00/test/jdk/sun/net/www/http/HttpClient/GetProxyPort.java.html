32 import java.net.InetAddress;
33 import java.net.InetSocketAddress;
34 import java.net.ServerSocket;
35 import java.net.URL;
36 import sun.net.www.http.HttpClient;
37 import jdk.test.lib.net.URIBuilder;
38 
39 public class GetProxyPort {
40     public static void main(String[] args) throws Exception {
41         ServerSocket ss = new ServerSocket();
42         InetAddress loopback = InetAddress.getLoopbackAddress();
43         ss.bind(new InetSocketAddress(loopback, 0));
44         URL myURL = URIBuilder.newBuilder()
45             .scheme("http")
46             .loopback()
47             .port(ss.getLocalPort())
48             .toURL();
49         HttpClient httpC = new HttpClient(myURL, null, -1);
50         int port = httpC.getProxyPortUsed();
51     }
52 }
