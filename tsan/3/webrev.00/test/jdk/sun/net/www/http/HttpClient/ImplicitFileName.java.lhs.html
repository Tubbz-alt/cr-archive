24 /*
25  * @test
26  * @bug 4251209
27  * @summary Make sure that implicit filenames will be returned as "/"
28  * @modules java.base/sun.net
29  *          java.base/sun.net.www.http

30  */
31 


32 import java.net.URL;
33 import java.net.ServerSocket;
34 import sun.net.www.http.HttpClient;

35 
36 public class ImplicitFileName {
37 
38     public static void main(String[] args) throws Exception {
39 
40         ServerSocket ss = new ServerSocket(0);


41 
42         URL url = new URL("http://localhost:" + ss.getLocalPort());




43 
44         HttpClient c = HttpClient.New(url);
45 
46         if (!c.getURLFile().equals("/")) {
47             throw new Exception("Implicit filename in URL " +
48                                 url.toString() + " is not '/'");
49         }
50     }
51 }
