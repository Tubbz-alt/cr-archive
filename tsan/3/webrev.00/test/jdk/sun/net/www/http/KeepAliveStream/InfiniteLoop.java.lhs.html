31 import com.sun.net.httpserver.HttpExchange;
32 import com.sun.net.httpserver.HttpHandler;
33 import com.sun.net.httpserver.HttpServer;
34 import java.io.InputStream;
35 import java.io.IOException;
36 import java.io.OutputStream;
37 import java.net.HttpURLConnection;

38 import java.net.InetSocketAddress;

39 import java.net.URL;
40 import java.util.concurrent.Phaser;
41 


42 // Racey test, will not always fail, but if it does then we have a problem.
43 
44 public class InfiniteLoop {
45 
46     public static void main(String[] args) throws Exception {
47         HttpServer server = HttpServer.create(new InetSocketAddress(0), 0);

48         server.createContext("/test/InfiniteLoop", new RespHandler());
49         server.start();
50         try {
51             InetSocketAddress address = server.getAddress();
52             URL url = new URL("http://localhost:" + address.getPort()
53                               + "/test/InfiniteLoop");




54             final Phaser phaser = new Phaser(2);
55             for (int i=0; i<10; i++) {
56                 HttpURLConnection uc = (HttpURLConnection)url.openConnection();

57                 final InputStream is = uc.getInputStream();
58                 final Thread thread = new Thread() {
59                     public void run() {
60                         try {
61                             phaser.arriveAndAwaitAdvance();
62                             while (is.read() != -1)
63                                 Thread.sleep(50);
64                         } catch (Exception x) { x.printStackTrace(); }
65                     }};
66                 thread.start();
67                 phaser.arriveAndAwaitAdvance();
68                 is.close();
69                 System.out.println("returned from close");
70                 thread.join();
71             }
72         } finally {
73             server.stop(0);
74         }
75     }
76 
77     static class RespHandler implements HttpHandler {
78         static final int RESP_LENGTH = 32 * 1024;
79         @Override
80         public void handle(HttpExchange t) throws IOException {
81             InputStream is  = t.getRequestBody();
82             byte[] ba = new byte[8192];
83             while(is.read(ba) != -1);
84 
85             t.sendResponseHeaders(200, RESP_LENGTH);
86             try (OutputStream os = t.getResponseBody()) {
87                 os.write(new byte[RESP_LENGTH]);
88             }
89             t.close();
90         }
91     }
92 }
