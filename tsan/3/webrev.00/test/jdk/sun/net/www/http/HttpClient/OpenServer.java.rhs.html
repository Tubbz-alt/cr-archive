25  * @test
26  * @bug 4154481
27  * @summary Make sure HttpClient has
28  *    doPrivileged() calls at appropriate places.
29  * @modules java.base/sun.net.www.http
30  * @library /test/lib
31  * @run main/othervm/policy=OpenServer.policy OpenServer
32  */
33 
34 import java.net.*;
35 import sun.net.www.http.HttpClient;
36 import jdk.test.lib.net.URIBuilder;
37 
38 public class OpenServer {
39 
40     OpenServer() throws Exception {
41 
42         ServerSocket ss = new ServerSocket();
43         InetAddress loopback = InetAddress.getLoopbackAddress();
44         ss.bind(new InetSocketAddress(loopback, 0));
45 
46         try (ServerSocket toClose = ss) {
47             URL myURL = URIBuilder.newBuilder()
48                 .scheme("http")
49                 .loopback()
50                 .port(ss.getLocalPort())
51                 .toURL();
52 
53             HttpClient httpC = new HttpClient(myURL, null, -1);
54         }
55     }
56 
57     public static void main(String [] args) throws Exception {
58         SecurityManager security = System.getSecurityManager();
59         if (security == null) {
60             security = new SecurityManager();
61             System.setSecurityManager(security);
62         }
63         // Note: we need to have some
64         // permissions in place for this
65         // test.
66         new OpenServer();
67     }
68 }
