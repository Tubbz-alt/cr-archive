24 /**
25  * @test
26  * @bug 4772077
27  * @summary  using defaultReadTimeout appear to retry request upon timeout
28  * @modules java.base/sun.net.www
29  */
30 
31 import java.net.*;
32 import java.io.*;
33 import sun.net.www.*;
34 
35 public class RetryUponTimeout implements Runnable {
36     // run server
37     public void run(){
38         Socket socket = null;
39         try {
40             for (int i = 0; i < 2; i++) {
41                 socket = server.accept();
42                 InputStream is = socket.getInputStream ();
43                 MessageHeader header = new MessageHeader (is);
44                 count++;
45             }
46         } catch (Exception ex) {
47             // ignored
48         } finally {
49             try {
50                 socket.close();
51                 server.close();
52             } catch (IOException ioex) {
53             }
54         }
55 
56     }
57 
58     static ServerSocket server;
59     static int count = 0;
60     public static void main(String[] args) throws Exception {
61         try {
62             server = new ServerSocket (0);
63             int port = server.getLocalPort ();
64             new Thread(new RetryUponTimeout()).start ();
65 
66             URL url = new URL("http://127.0.0.1:"+port);
67             java.net.URLConnection uc = url.openConnection();
68             uc.setReadTimeout(1000);
69             uc.getInputStream();
70         } catch (SocketTimeoutException stex) {
71             // expected exception
72             server.close();
73         }
74 
75         if (count > 1) {
76             throw new RuntimeException("Server received "+count+" requests instead of one.");
77         }
78 
79     }
80 }
