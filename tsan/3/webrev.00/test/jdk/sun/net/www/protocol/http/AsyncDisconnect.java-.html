 24 /*
 25  * @test
 26  * @bug 6358532
 27  * @modules jdk.httpserver
 28  * @run main/othervm AsyncDisconnect
 29  * @summary HttpURLConnection.disconnect doesn't really do the job
 30  */
 31 
 32 import java.net.*;
 33 import java.util.*;
 34 import java.io.*;
 35 import com.sun.net.httpserver.*;
 36 import java.util.concurrent.Executors;
 37 import java.util.concurrent.ExecutorService;
 38 
 39 public class AsyncDisconnect implements Runnable
 40 {
 41     com.sun.net.httpserver.HttpServer httpServer;
 42     MyHandler httpHandler;
 43     ExecutorService executorService;
 44     HttpURLConnection uc;
 45 
 46     public static void main(String[] args) {
 47         new AsyncDisconnect();
 48     }
 49 
 50     public AsyncDisconnect() {
 51         try {
 52             startHttpServer();
 53             doClient();
 54         } catch (IOException ioe) {
 55             System.err.println(ioe);
 56         }
 57     }
 58 
 59     void doClient() {
 60         try {
 61             InetSocketAddress address = httpServer.getAddress();
 62             URL url = new URL("http://" + address.getHostName() + ":" + address.getPort() + "/test/");
 63             uc = (HttpURLConnection)url.openConnection();
 64 
 65             // create a thread that will disconnect the connection
 66             (new Thread(this)).start();
 67 
 68             uc.getInputStream();
 69 
 70             // if we reach here then we have failed
 71             throw new RuntimeException("Failed: We Expect a SocketException to be thrown");
 72 
 73         } catch (SocketException se) {
 74             // this is what we expect to happen and is OK.
 75             //System.out.println(se);
 76         } catch (IOException e) {
 77             e.printStackTrace();
 78         } finally {
 79             httpServer.stop(1);
 80             executorService.shutdown();
 81         }
 82     }
 83 
 84     public void run() {
 85         // wait for the request to be sent to the server before calling disconnect
 86         try { Thread.sleep(2000); }
 87         catch (Exception e) {}
 88 
 89         uc.disconnect();
 90     }
 91 
 92     /**
 93      * Http Server
 94      */
 95     public void startHttpServer() throws IOException {
 96         httpServer = com.sun.net.httpserver.HttpServer.create(new InetSocketAddress(0), 0);
 97         httpHandler = new MyHandler();
 98 
 99         HttpContext ctx = httpServer.createContext("/test/", httpHandler);
100 
101         executorService = Executors.newCachedThreadPool();
102         httpServer.setExecutor(executorService);
103         httpServer.start();
104     }
105 
106     class MyHandler implements HttpHandler {
107         public void handle(HttpExchange t) throws IOException {
108             // give the other thread a chance to close the connection
109             try { Thread.sleep(4000); }
110             catch (Exception e) {}
111 
112             t.sendResponseHeaders(400, -1);
113             t.close();
114         }
115     }
116 
117 }
