24 /**
25  * @test
26  * @bug 4773417 5003746



27  * @summary  HttpURLConnection.getInputStream() produces IOException with
28  *           bad stack trace; HttpURLConnection.getInputStream loses
29  *           exception message, exception class
30  */
31 import java.net.*;
32 import java.io.IOException;

33 
34 public class StackTraceTest {
35     public static void main(String[] args) throws Exception {
36         URL url;
37         try (ServerSocket ss = new ServerSocket(0)) {  // refusing socket
38             url = new URL("http://localhost:" + ss.getLocalPort() + "/");
39         }




40         URLConnection uc = url.openConnection();
41 
42         // Trigger implicit connection by trying to retrieve bogus
43         // response header, and force remembered exception
44         uc.getHeaderFieldKey(20);
45 
46         try {
47             uc.getInputStream();  // expect to throw
48             throw new RuntimeException("Expected getInputStream to throw");
49         } catch (IOException ioe) {
50             if (!(ioe instanceof ConnectException))
51                 throw new RuntimeException("Expect ConnectException, got " + ioe);
52             if (ioe.getMessage() == null)
53                 throw new RuntimeException("Exception message is null");
54             if (ioe.getCause() == null)
55                 throw new RuntimeException("Excepting a chained exception, but got: ",
56                                            ioe);
57         }
58     }
59 }
