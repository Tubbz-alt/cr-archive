 35 import java.net.*;
 36 import java.io.*;
 37 import java.util.*;
 38 import com.sun.net.httpserver.*;
 39 import java.util.concurrent.*;
 40 import javax.net.ssl.*;
 41 import jdk.test.lib.net.SimpleSSLContext;
 42 import jdk.test.lib.net.URIBuilder;
 43 
 44 public class RedirectOnPost {
 45 
 46 
 47     public static void main(String[] args) throws Exception {
 48         ExecutorService e= Executors.newFixedThreadPool(5);
 49         SSLContext ctx = new SimpleSSLContext().get();
 50         HttpServer httpServer = getHttpServer(e);
 51         HttpsServer httpsServer = getHttpsServer(e, ctx);
 52 
 53         try {
 54             // take the keystore from elsewhere in test hierarchy
 55             int port = httpServer.getAddress().getPort();
 56             int sslPort = httpsServer.getAddress().getPort();
 57             httpServer.start();
 58             httpsServer.start();
 59             runTest("http", port, null);
 60             runTest("https", sslPort, ctx);
 61             System.out.println("Main thread waiting");
 62         } finally {
 63             httpServer.stop(0);
 64             httpsServer.stop(0);
 65             e.shutdownNow();
 66         }
 67     }
 68 
 69     public static void runTest(String scheme, int port, SSLContext ctx) throws Exception
 70     {
 71         byte[] buf = "Hello world".getBytes();
 72 
 73         URL url = URIBuilder.newBuilder()
 74             .scheme(scheme)
 75             .loopback()
 76             .port(port)
 77             .path("/test/a")
 78             .toURL();
 79         System.out.println("URL: " + url);
 80         HttpURLConnection con = (HttpURLConnection)url.openConnection();
 81         if (con instanceof HttpsURLConnection) {
 82             HttpsURLConnection ssl = (HttpsURLConnection)con;
 83             ssl.setHostnameVerifier(new HostnameVerifier() {
 84                 public boolean verify(String host, SSLSession sess) {
 85                     return true;
 86                 }
 87             });
 88             ssl.setSSLSocketFactory (ctx.getSocketFactory());
 89         }
 90         con.setDoOutput(true);
 91         con.setDoInput(true);
 92         con.setRequestMethod("POST");
 93         try (OutputStream out = con.getOutputStream()) {
 94             out.write(buf);
 95         }
 96         try (InputStream in = con.getInputStream()) {
 97             byte[] newBuf = readFully(in);
 98         }
 99     }
100 
101     private static byte[] readFully(InputStream istream) throws IOException {
102         ByteArrayOutputStream bout = new ByteArrayOutputStream();
103         byte[] buf = new byte[1024];
104         int num = 0;
105 
106         if (istream != null) {
107             while ((num = istream.read(buf)) != -1) {
108                 bout.write(buf, 0, num);
109             }
110         }
111         byte[] ret = bout.toByteArray();
112         return ret;
113     }

116     static class Handler implements HttpHandler {
117 
118         String scheme;
119         int port;
120 
121         Handler(String scheme, int port) {
122           this.scheme = scheme;
123           this.port = port;
124         }
125 
126         int calls = 0;
127 
128         public void handle(HttpExchange msg) {
129             try {
130                 String method = msg.getRequestMethod();
131                 URL baseURL = URIBuilder.newBuilder()
132                     .scheme(scheme)
133                     .loopback()
134                     .path("/test/b")
135                     .port(port)
136                     .toURLUnchecked();
137                 System.out.println ("Server: " + baseURL);
138                 if (calls++ == 0) {
139                     System.out.println ("Server: redirecting");
140                     InputStream is = msg.getRequestBody();
141                     byte[] buf = readFully(is);
142                     is.close();
143                     Headers h = msg.getResponseHeaders();
144                     h.add("Location", baseURL.toString());
145                     msg.sendResponseHeaders(302, -1);
146                     msg.close();
147                 } else {
148                     System.out.println ("Server: second call");
149                     InputStream is = msg.getRequestBody();
150                     byte[] buf = readFully(is);
151                     is.close();
152                     msg.sendResponseHeaders(200, -1);
153                     msg.close();
154                 }
155             }
156             catch(Exception e) {
157                 e.printStackTrace();
158             }
159             finally {
160                 msg.close();
161             }
162         }
163     }
164 
165     private static HttpServer getHttpServer(ExecutorService execs)
166         throws Exception
167     {
168         InetSocketAddress inetAddress = new InetSocketAddress(
169                 InetAddress.getLoopbackAddress(), 0);
170         HttpServer testServer = HttpServer.create(inetAddress, 15);
171         int port = testServer.getAddress().getPort();
172         testServer.setExecutor(execs);
173 
174         HttpContext context = testServer.createContext("/test");
175         context.setHandler(new Handler("http", port));
176         return testServer;
177     }
178 
179     private static HttpsServer getHttpsServer(
180         ExecutorService execs, SSLContext ctx
181     )
182         throws Exception
183     {
184         InetSocketAddress inetAddress = new InetSocketAddress(
185                 InetAddress.getLoopbackAddress(), 0);
186         HttpsServer testServer = HttpsServer.create(inetAddress, 15);
187         int port = testServer.getAddress().getPort();
188         testServer.setExecutor(execs);
189         testServer.setHttpsConfigurator(new HttpsConfigurator (ctx));
190 
191         HttpContext context = testServer.createContext("/test");
192         context.setHandler(new Handler("https", port));
193         return testServer;
194     }
