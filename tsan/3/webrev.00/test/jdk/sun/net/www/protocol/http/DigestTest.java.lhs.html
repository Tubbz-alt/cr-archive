 89                 MessageHeader header = new MessageHeader (is);
 90                 String raw = header.findValue ("Authorization");
 91                 HeaderParser parser = new HeaderParser (raw);
 92                 String cnonce = parser.findValue ("cnonce");
 93                 String cnstring = parser.findValue ("nc");

107     static char[] passwd = "password".toCharArray();
108     static String username = "user";
109     static String nonce = "abcdefghijklmnopqrstuvwxyz";
110     static String realm = "wallyworld";
111     static String uri = "/foo.html";

113     private String getAuthorization (String uri, String method, String cnonce, String cnstring) {
114         String response;
115 
116         try {
117             response = computeDigest(false, username,passwd,realm,
118                                         method, uri, nonce, cnonce, cnstring);
119         } catch (NoSuchAlgorithmException ex) {
120             return null;
121         }
122 
123         String value = "Digest"
124                         + " qop=auth\""
125                         + "\", cnonce=\"" + cnonce
126                         + "\", rspauth=\"" + response
127                         + "\", nc=\"" + cnstring + "\"";
128         return (value+ "\r\n");
129     }
130 
131     private String computeDigest(
132                         boolean isRequest, String userName, char[] password,
133                         String realm, String connMethod,
134                         String requestURI, String nonceString,
135                         String cnonce, String ncValue
136                     ) throws NoSuchAlgorithmException
137     {
138 
139         String A1, HashA1;
140 
141         MessageDigest md = MessageDigest.getInstance("MD5");
142 
143         {
144             A1 = userName + ":" + realm + ":";
145             HashA1 = encode(A1, password, md);
146         }
147 
148         String A2;
149         if (isRequest) {
150             A2 = connMethod + ":" + requestURI;
151         } else {
152             A2 = ":" + requestURI;
153         }
154         String HashA2 = encode(A2, null, md);
155         String combo, finalHash;
156 
157         { /* RRC2617 when qop=auth */
158             combo = HashA1+ ":" + nonceString + ":" + ncValue + ":" +
159                         cnonce + ":auth:" +HashA2;
160 
161         }
162         finalHash = encode(combo, null, md);
163         return finalHash;
164     }

166     private final static char charArray[] = {
167         '0', '1', '2', '3', '4', '5', '6', '7',
168         '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'
169     };

171     private String encode(String src, char[] passwd, MessageDigest md) {
172         md.update(src.getBytes());
173         if (passwd != null) {
174             byte[] passwdBytes = new byte[passwd.length];
175             for (int i=0; i<passwd.length; i++)
176                 passwdBytes[i] = (byte)passwd[i];
177             md.update(passwdBytes);
178             Arrays.fill(passwdBytes, (byte)0x00);
179         }
180         byte[] digest = md.digest();
181 
182         StringBuffer res = new StringBuffer(digest.length * 2);
183         for (int i = 0; i < digest.length; i++) {
184             int hashchar = ((digest[i] >>> 4) & 0xf);
185             res.append(charArray[hashchar]);
186             hashchar = (digest[i] & 0xf);
187             res.append(charArray[hashchar]);
188         }
189         return res.toString();
190     }
