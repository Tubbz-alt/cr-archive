 36 /* Creates a simple proxy and http server that just return 200 OK.
 37  * Open a URL pointing to the http server and specify that the
 38  * connection should use the proxy. Now make a second connection
 39  * to the same URL, specifying that no proxy is to be used.
 40  * We count the amount of requests being sent to each server. There
 41  * should be only one request sent to each.
 42  */
 43 
 44 public class ProxyFromCache
 45 {
 46     public static void main(String[] args) {
 47         ServerSocket proxySSocket, httpSSocket;
 48         int proxyPort, httpPort;

 49 
 50         try {
 51             proxySSocket = new ServerSocket(0);

 52             proxyPort = proxySSocket.getLocalPort();
 53             httpSSocket = new ServerSocket(0);

 54             httpPort = httpSSocket.getLocalPort();
 55         } catch (Exception e) {
 56             System.out.println ("Exception: " + e);
 57             return;
 58         }
 59 
 60         SimpleServer proxyServer = new SimpleServer(proxySSocket);
 61         proxyServer.start();
 62         SimpleServer httpServer = new SimpleServer(httpSSocket);
 63         httpServer.start();
 64 
 65         InetSocketAddress addr = new InetSocketAddress("localhost", proxyPort);
 66         Proxy proxy = new Proxy(Proxy.Type.HTTP, addr);
 67 
 68         try {
 69             String urlStr = "http://localhost:" + httpPort + "/";
 70             URL url = new URL(urlStr);






 71 
 72             // 1st connection.
 73             HttpURLConnection uc = (HttpURLConnection) url.openConnection(proxy);
 74             InputStream is = uc.getInputStream();
 75 
 76             byte[] ba = new byte[1024];
 77             while(is.read(ba) != -1);
 78             is.close();
 79 
 80             // 2nd connection.
 81             uc = (HttpURLConnection) url.openConnection(Proxy.NO_PROXY);
 82             is = uc.getInputStream();
 83 
 84             while(is.read(ba) != -1);
 85             is.close();
 86 
 87             try {
 88                 proxySSocket.close();
 89                 httpSSocket.close();
 90             } catch (IOException e) {}
 91 
 92             proxyServer.terminate();
 93             httpServer.terminate();
 94 
 95             int httpCount = httpServer.getConnectionCount();
 96             int proxyCount = proxyServer.getConnectionCount();
 97 
 98             if (proxyCount != 1 && httpCount != 1) {
 99                 System.out.println("Proxy = " + proxyCount + ", http = " + httpCount);
100                 throw new RuntimeException("Failed: Proxy being sent " + proxyCount  + " requests");
101             }
102         } catch (IOException e) {
103             throw new RuntimeException(e);
104         }
105     }
106 }

 38 /* Creates a simple proxy and http server that just return 200 OK.
 39  * Open a URL pointing to the http server and specify that the

 46 public class ProxyFromCache
 47 {
 48     public static void main(String[] args) throws Exception {
 49         ServerSocket proxySSocket, httpSSocket;
 50         int proxyPort, httpPort;
 51         InetAddress loopback = InetAddress.getLoopbackAddress();
 52 
 53         try {
 54             proxySSocket = new ServerSocket();
 55             proxySSocket.bind(new InetSocketAddress(loopback, 0));
 56             proxyPort = proxySSocket.getLocalPort();
 57             httpSSocket = new ServerSocket();
 58             httpSSocket.bind(new InetSocketAddress(loopback, 0));
 59             httpPort = httpSSocket.getLocalPort();
 60         } catch (Exception e) {
 61             System.out.println ("Exception: " + e);
 62             throw e;
 63         }
 64 
 65         SimpleServer proxyServer = new SimpleServer(proxySSocket);
 66         proxyServer.start();
 67         SimpleServer httpServer = new SimpleServer(httpSSocket);
 68         httpServer.start();
 69 
 70         InetSocketAddress addr = new InetSocketAddress(loopback, proxyPort);
 71         Proxy proxy = new Proxy(Proxy.Type.HTTP, addr);
 72 
 73         try {
 74             URL url = URIBuilder.newBuilder()
 75                       .scheme("http")
 76                       .loopback()
 77                       .port(httpPort)
 78                       .path("/")
 79                       .toURL();
 80 
 81             String urlStr = url.toString();
 82 
 83             // 1st connection.
 84             HttpURLConnection uc = (HttpURLConnection) url.openConnection(proxy);
 85             InputStream is = uc.getInputStream();
 86 
 87             byte[] ba = new byte[1024];
 88             while(is.read(ba) != -1);
 89             is.close();
 90 
 91             // 2nd connection.
 92             uc = (HttpURLConnection) url.openConnection(Proxy.NO_PROXY);
 93             is = uc.getInputStream();
 94 
 95             while(is.read(ba) != -1);
 96             is.close();
 97
