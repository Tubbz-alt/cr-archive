 24 /*
 25  * @test
 26  * @bug 6614957
 27  * @summary HttpsURLConnection not using the set SSLSocketFactory for creating all its Sockets
 28  * @modules jdk.httpserver
 29  * @run main/othervm HttpsSocketFacTest
 30  *
 31  *     SunJSSE does not support dynamic system properties, no way to re-use
 32  *     system properties in samevm/agentvm mode.
 33  */
 34 
 35 import javax.net.SocketFactory;
 36 import javax.net.ssl.HostnameVerifier;
 37 import javax.net.ssl.HttpsURLConnection;
 38 import javax.net.ssl.SSLContext;
 39 import javax.net.ssl.SSLSession;
 40 import javax.net.ssl.SSLSocketFactory;
 41 import java.security.NoSuchAlgorithmException;
 42 import java.net.InetAddress;
 43 import java.net.InetSocketAddress;
 44 import java.net.Socket;
 45 import java.net.URL;
 46 import java.io.BufferedWriter;
 47 import java.io.InputStream;
 48 import java.io.IOException;
 49 import java.io.OutputStreamWriter;
 50 import com.sun.net.httpserver.HttpExchange;
 51 import com.sun.net.httpserver.HttpHandler;
 52 import com.sun.net.httpserver.HttpsConfigurator;
 53 
 54 /*
 55  * This class tests that the HTTPS protocol handler is using its socket factory for
 56  * creating new Sockets. It does this by wrapping the default SSLSocketFactory with
 57  * its own socket factory, SimpleSSLSocketFactory, and verifying that when a https
 58  * connection is made one of the socket factories createSocket methods, that
 59  * actually creates a Socket, is being invoked by the protocol handler.
 60  */
 61 
 62 public class HttpsSocketFacTest
 63 {
 64     /*
 65      * Where do we find the keystores?
 66      */
 67     static String pathToStores = "../../../../../../javax/net/ssl/etc";
 68     static String keyStoreFile = "keystore";
 69     static String trustStoreFile = "truststore";
 70     static String passwd = "passphrase";
 71 
 72     com.sun.net.httpserver.HttpsServer httpsServer;
 73     MyHandler httpHandler;
 74 
 75     public static void main(String[] args) {
 76         String keyFilename =
 77             System.getProperty("test.src", "./") + "/" + pathToStores +
 78                 "/" + keyStoreFile;
 79         String trustFilename =
 80             System.getProperty("test.src", "./") + "/" + pathToStores +
 81                 "/" + trustStoreFile;
 82 
 83         System.setProperty("javax.net.ssl.keyStore", keyFilename);
 84         System.setProperty("javax.net.ssl.keyStorePassword", passwd);
 85         System.setProperty("javax.net.ssl.trustStore", trustFilename);
 86         System.setProperty("javax.net.ssl.trustStorePassword", passwd);
 87 
 88         new HttpsSocketFacTest();
 89     }
 90 
 91     public HttpsSocketFacTest() {
 92         try {
 93             startHttpsServer();
 94             doClient();
 95         } catch (NoSuchAlgorithmException e) {
 96             e.printStackTrace();
 97         } catch (IOException ioe) {
 98             ioe.printStackTrace();
 99         } finally {
100            httpsServer.stop(1);
101         }
102     }
103 
104     void doClient() throws IOException {
105         InetSocketAddress address = httpsServer.getAddress();
106         URL url = new URL("https://localhost:" + address.getPort() + "/test6614957/");
107         System.out.println("trying to connect to " + url + "...");
108 
109         HttpsURLConnection uc = (HttpsURLConnection) url.openConnection();
110         SimpleSSLSocketFactory sssf = new SimpleSSLSocketFactory();
111         uc.setSSLSocketFactory(sssf);
112         uc.setHostnameVerifier(new AllHostnameVerifier());
113         InputStream is = uc.getInputStream();
114 
115         byte[] ba = new byte[1024];
116         int read = 0;
117         while ((read = is.read(ba)) != -1) {
118             System.out.println(new String(ba, 0, read));
119         }
120 
121         System.out.println("SimpleSSLSocketFactory.socketCreated = " + sssf.socketCreated);
122         System.out.println("SimpleSSLSocketFactory.socketWrapped = " + sssf.socketWrapped);
123 
124         if (!sssf.socketCreated)
125             throw new RuntimeException("Failed: Socket Factory not being called to create Socket");
126     }
127 
128     /**
129      * Https Server
130      */
131     public void startHttpsServer() throws IOException, NoSuchAlgorithmException  {
132         httpsServer = com.sun.net.httpserver.HttpsServer.create(new InetSocketAddress(0), 0);
133         httpsServer.createContext("/test6614957/", new MyHandler());
134         httpsServer.setHttpsConfigurator(new HttpsConfigurator(SSLContext.getDefault()));
135         httpsServer.start();
136     }
137 
138     class MyHandler implements HttpHandler {
139         private String message = "This is a message!";
140 
141         @Override
142         public void handle(HttpExchange t) throws IOException {
143             t.sendResponseHeaders(200, message.length());
144             BufferedWriter writer = new BufferedWriter( new OutputStreamWriter(t.getResponseBody(), "ISO8859-1"));
145             writer.write(message, 0, message.length());
146             writer.close();
147             t.close();
148         }
149     }
150 
151     /**
152      * Simple wrapper on default SSLSocketFactory
153      */
154     class SimpleSSLSocketFactory extends SSLSocketFactory
155     {
156         /*
157          * true if this factory has been used to create a new Socket, i.e.
158          * one of the SocketFactory methods has been called.
159          */
160         boolean socketCreated = false;
161 
162         /*
163          * true if this factory has been used to wrap a Socket, i.e.
164          * the SSLSocketFactory method,
165          * createSocket(Socket, String, int, boolean), has been called.
166          */
167         boolean socketWrapped = false;
168 
169         // methods for SocketFactory
170         @Override
171         public Socket createSocket() throws IOException {
172             socketCreated = true;
173             return SocketFactory.getDefault().createSocket();
174         }
175 
176         @Override
177         public Socket createSocket(InetAddress host, int port) throws IOException {
178             socketCreated = true;
179             return SocketFactory.getDefault().createSocket(host, port);
180         }
181 
182         @Override
183         public Socket createSocket(InetAddress address, int port, InetAddress localAddress,
184                                    int localPort) throws IOException {
185             socketCreated = true;
186             return SocketFactory.getDefault().createSocket(address, port, localAddress, localPort);
187         }
188 
189         @Override
190         public Socket createSocket(String host, int port) throws IOException {
191             socketCreated = true;
192             return SocketFactory.getDefault().createSocket(host, port);
193         }
194 
195         @Override
196         public Socket createSocket(String host, int port, InetAddress localHost,
197                                    int localPort) throws IOException {
198             socketCreated = true;
199             return SocketFactory.getDefault().createSocket(host, port, localHost, localPort);
200         }
201 
202         // methods from SSLSocketFactory
203         @Override
204         public Socket createSocket(Socket s, String host, int port,
205                                    boolean autoClose) throws IOException {
206             socketWrapped = true;
207             return ((SSLSocketFactory) SSLSocketFactory.getDefault()).createSocket
208                                                                (s, host, port, autoClose);
209         }
210 
211         @Override
212         public String[] getDefaultCipherSuites() {
213             return ((SSLSocketFactory) SSLSocketFactory.getDefault()).getDefaultCipherSuites();
214         }
215 
216         @Override
217         public String[] getSupportedCipherSuites()  {
218              return ((SSLSocketFactory) SSLSocketFactory.getDefault()).getSupportedCipherSuites();
219         }
220     }
221 
222     class AllHostnameVerifier implements HostnameVerifier
223     {
224         @Override
225         public boolean verify(String hostname, SSLSession session) {
226             return true;
227         }
228     }
