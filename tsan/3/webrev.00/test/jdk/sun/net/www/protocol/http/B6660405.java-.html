 51         {
 52             Map<String, List<String>> h = new HashMap<String, List<String>>();

106     void doClient() {
107         ResponseCache.setDefault(new MyResponseCache());
108         try {
109             InetSocketAddress address = httpServer.getAddress();
110 
111             // GET Request
112             URL url = new URL("http://localhost:" + address.getPort() + "/test/index.html");
113             HttpURLConnection uc = (HttpURLConnection)url.openConnection();
114             int code = uc.getResponseCode();
115             System.err.println("response code = " + code);
116             int l = uc.getContentLength();
117             System.err.println("content-length = " + l);
118             InputStream in = uc.getInputStream();
119             int i = 0;
120             // Read till end of stream
121             do {
122                 i = in.read();
123             } while (i != -1);
124             in.close();
125         } catch (IOException e) {
126             throw new RuntimeException("Got the wrong InputStream after checking headers");
127         } finally {
128             httpServer.stop(1);
129             executorService.shutdown();
130         }
131     }
132 
133     /**
134      * Http Server
135      */
136     public void startHttpServer() throws IOException {
137         httpServer = com.sun.net.httpserver.HttpServer.create(new InetSocketAddress(0), 0);
138 
139         // create HttpServer context
140         HttpContext ctx = httpServer.createContext("/test/", new MyHandler());
141 
142         executorService = Executors.newCachedThreadPool();
143         httpServer.setExecutor(executorService);
144         httpServer.start();
145     }

152

160             t.sendResponseHeaders(302, -1);
161             t.close();
162         }
