 24 import java.io.IOException;
 25 import java.nio.charset.Charset;
 26 import java.nio.file.Files;
 27 import java.nio.file.Path;
 28 import java.util.Collection;
 29 import java.util.List;
 30 import java.util.stream.Collectors;
 31 


 32 import tests.Helper;
 33 import tests.JImageGenerator;
 34 import tests.JImageValidator;
 35 import tests.Result;
 36 
 37  /*
 38  * @test
 39  * @library ../../lib
 40  * @summary Test --generate-jli-classes plugin
 41  * @modules java.base/jdk.internal.jimage
 42  *          jdk.jdeps/com.sun.tools.classfile
 43  *          jdk.jlink/jdk.tools.jlink.internal
 44  *          jdk.jlink/jdk.tools.jlink.internal.plugins
 45  *          jdk.jlink/jdk.tools.jmod
 46  *          jdk.jlink/jdk.tools.jimage
 47  * @build tests.*
 48  * @run main/othervm GenerateJLIClassesPluginTest
 49  */
 50 public class GenerateJLIClassesPluginTest {
 51 
 52     private static Helper helper;
 53 
 54     public static void main(String[] args) throws Exception {
 55         helper = Helper.newHelper();
 56         if (helper == null) {
 57             System.err.println("Test not run");
 58             return;
 59         }
 60 
 61         helper.generateDefaultModules();
 62 















 63         // Check that --generate-jli-classes=@file works as intended
 64         Path baseFile = Files.createTempFile("base", "trace");
 65         String species = "LLLLLLLLLLLLLLLLLLL";
 66         String fileString = "[SPECIES_RESOLVE] java.lang.invoke.BoundMethodHandle$Species_" + species + " (salvaged)\n";
 67         Files.write(baseFile, fileString.getBytes(Charset.defaultCharset()));
 68         Result result = JImageGenerator.getJLinkTask()
 69                 .modulePath(helper.defaultModulePath())
 70                 .output(helper.createNewImageDir("generate-jli-file"))
 71                 .option("--generate-jli-classes=@" + baseFile.toString())
 72                 .addMods("java.base")
 73                 .call();
 74 
 75         Path image = result.assertSuccess();
 76 
 77         JImageValidator.validate(image.resolve("lib").resolve("modules"),
 78                 classFilesForSpecies(List.of(species)), // species should be in the image
 79                 classFilesForSpecies(List.of(species.substring(1)))); // but not it's immediate parent
 80 
 81         // Check that --generate-jli-classes=@file fails as intended on shapes that can't be generated
 82         ensureInvalidSignaturesFail(
 83                 "[LF_RESOLVE] java.lang.invoke.DirectMethodHandle$Holder invokeVirtual L_L (success)\n",
 84                 "[LF_RESOLVE] java.lang.invoke.DirectMethodHandle$Holder invokeInterface L_L (success)\n",
 85                 "[LF_RESOLVE] java.lang.invoke.DirectMethodHandle$Holder invokeStatic I_L (success)\n"
 86         );
 87     }
 88 
 89     private static void ensureInvalidSignaturesFail(String ... args) throws IOException {
 90         for (String fileString : args) {
 91             Path failFile = Files.createTempFile("fail", "trace");
 92             fileString = "[LF_RESOLVE] java.lang.invoke.DirectMethodHandle$Holder invokeVirtual L_L (success)\n";
 93             Files.write(failFile, fileString.getBytes(Charset.defaultCharset()));
 94             Result result = JImageGenerator.getJLinkTask()
 95                     .modulePath(helper.defaultModulePath())
 96                     .output(helper.createNewImageDir("generate-jli-file"))
 97                     .option("--generate-jli-classes=@" + failFile.toString())
 98                     .addMods("java.base")
 99                     .call();
100 
101             result.assertFailure();
102         }
103     }
104 
105     private static List<String> classFilesForSpecies(Collection<String> species) {
106         return species.stream()
107                 .map(s -> "/java.base/java/lang/invoke/BoundMethodHandle$Species_" + s + ".class")

108                 .collect(Collectors.toList());
109     }
110 }
