24 #include <jni.h>
25 
26 static jlong fib(jint num) {
27     if (num == 0) {
28         return 0;
29     }
30     if (num <= 2) {
31         return 1;
32     }
33     return fib(num - 2) + fib(num -1);
34 }
35 
36 static void callCallback(JNIEnv *env, jclass cls, jobject target, jlong result) {
37     jmethodID mid = (*env)->GetMethodID(env, cls, "callback", "(J)V");
38     if (mid == NULL) {
39         jclass nsme = (jclass) (*env)->NewGlobalRef(env,
40                                                     (*env)->FindClass(env,
41                                                                       "java/lang/NoSuchMethodException"));
42         if (nsme != NULL) {
43             (*env)->ThrowNew(env, nsme, "Can't find method callback()");
44         }
45         return;
46     }
47     (*env)->CallVoidMethod(env, target, mid, result);
48 }
49 
50 static void calculateAndCallCallback(JNIEnv *env, jclass cls, jobject target, jint num) {
51     jlong result = -1;
52     result = fib(num);
53     callCallback(env, cls, target, result);
54 }
55 
56 JNIEXPORT void JNICALL
57 Java_fib_FibJNI_callJNI(JNIEnv *env, jclass cls, jobject target, jint num) {
58     calculateAndCallCallback(env, cls, target, num);
59 }
