23 package fib;
24 public class FibJNI {
25 
26     static {
27         // Native lib used for debug symbols stripping
28         System.loadLibrary("Fib");
29     }
30 
31     private final int num;
32     private final long expected;
33 
34     public FibJNI(int num, long expected) {
35         this.num = num;
36         this.expected = expected;
37     }
38 
39     public void callNative() {
40         callJNI(this, num);
41     }
42 
43     // Called from JNI library libFib
44     private void callback(long val) {
45         System.out.println("Debug: result was: " + val);
46         if (val != expected) {
47             throw new RuntimeException("Expected " + expected + " but got: " +val);
48         }
49     }
50 
51     public static native void callJNI(Object target, int num);
52 
53     public static void main(String[] args) {
54         if (args.length != 2) {
55             System.err.println("Usage: " + FibJNI.class.getName() + " <input> <expectedResult>");
56         }
57         int input = Integer.parseInt(args[0]);
58         long expected = Long.parseLong(args[1]);
59         FibJNI fib = new FibJNI(input, expected);
60         fib.callNative();
61         System.out.println("DEBUG: Sanity check for " + FibJNI.class.getSimpleName() + " passed.");
62     }
63 }
