24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 import jdk.jpackage.test.Annotations.Test;
28 
29 /**
30  * Test --win-menu and --win-menu-group parameters.
31  * Output of the test should be WinMenuGroupTest-1.0.exe installer.
32  * The output installer should provide the
33  * same functionality as the default installer (see description of the default
34  * installer in SimplePackageTest.java) plus
35  * it should create a shortcut for application launcher in Windows Menu in
36  * "C:\ProgramData\Microsoft\Windows\Start Menu\Programs\WinMenuGroupTest_MenuGroup" folder.
37  */
38 
39 /*
40  * @test
41  * @summary jpackage with --win-menu and --win-menu-group
42  * @library ../helpers
43  * @key jpackagePlatformPackage
44  * @build jdk.jpackage.test.*
45  * @requires (os.family == "windows")
46  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
47  * @compile WinMenuGroupTest.java
48  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
49  *  --jpt-run=WinMenuGroupTest
50  */
51 
52 public class WinMenuGroupTest {
53     @Test
54     public static void test() {
55         new PackageTest()
56         .forTypes(PackageType.WINDOWS)
57         .configureHelloApp()
58         .addInitializer(cmd -> cmd.addArguments(
59                 "--win-menu", "--win-menu-group", "WinMenuGroupTest_MenuGroup"))
60         .run();
61     }
62 }
