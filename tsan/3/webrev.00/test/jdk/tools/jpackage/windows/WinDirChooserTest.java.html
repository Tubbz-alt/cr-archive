24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 
28 /**
29  * Test --win-dir-chooser parameter. Output of the test should be
30  * WinDirChooserTest-1.0.exe installer. The output installer should provide the
31  * same functionality as the default installer (see description of the default
32  * installer in SimplePackageTest.java) plus provide an option for user to
33  * change the default installation directory.
34  */
35 
36 /*
37  * @test
38  * @summary jpackage with --win-dir-chooser
39  * @library ../helpers
40  * @key jpackagePlatformPackage
41  * @build jdk.jpackage.test.*
42  * @requires (os.family == "windows")
43  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
44  * @run main/othervm/timeout=360 -Xmx512m WinDirChooserTest
45  */
46 
47 public class WinDirChooserTest {
48     public static void main(String[] args) {
49         TKit.run(args, () -> {
50             new PackageTest()
51             .forTypes(PackageType.WINDOWS)
52             .configureHelloApp()
53             .addInitializer(cmd -> cmd.addArgument("--win-dir-chooser")).run();
54         });
55     }
56 }
