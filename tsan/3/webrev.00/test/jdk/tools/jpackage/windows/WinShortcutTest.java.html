24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 import jdk.jpackage.test.Annotations.Test;
28 
29 /**
30  * Test --win-shortcut parameter. Output of the test should be
31  * WinShortcutTest-1.0.exe installer. The output installer should provide the
32  * same functionality as the default installer (see description of the default
33  * installer in SimplePackageTest.java) plus install application shortcut on the
34  * desktop.
35  */
36 
37 /*
38  * @test
39  * @summary jpackage with --win-shortcut
40  * @library ../helpers
41  * @key jpackagePlatformPackage
42  * @build jdk.jpackage.test.*
43  * @requires (os.family == "windows")
44  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
45  * @compile WinShortcutTest.java
46  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
47  *  --jpt-run=WinShortcutTest
48  */
49 
50 public class WinShortcutTest {
51     @Test
52     public static void test() {
53         new PackageTest()
54         .forTypes(PackageType.WINDOWS)
55         .configureHelloApp()
56         .addInitializer(cmd -> cmd.addArgument("--win-shortcut"))
57         .run();
58     }
59 }
