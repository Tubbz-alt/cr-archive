24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 import jdk.jpackage.test.Annotations.Test;
28 
29 /**
30  * Test --win-menu parameter. Output of the test should be WinMenuTest-1.0.exe
31  * installer. The output installer should provide the same functionality as the
32  * default installer (see description of the default installer in
33  * SimplePackageTest.java), except it should create a menu shortcut.
34  */
35 
36 /*
37  * @test
38  * @summary jpackage with --win-menu
39  * @library ../helpers
40  * @key jpackagePlatformPackage
41  * @build jdk.jpackage.test.*
42  * @requires (os.family == "windows")
43  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
44  * @compile WinMenuTest.java
45  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
46  *  --jpt-run=WinMenuTest
47  */
48 
49 public class WinMenuTest {
50     @Test
51     public static void test() {
52         new PackageTest()
53         .forTypes(PackageType.WINDOWS)
54         .configureHelloApp()
55         .addInitializer(cmd -> cmd.addArgument("--win-menu")).run();
56     }
57 }
