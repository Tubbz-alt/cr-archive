25 package jdk.jpackage.test;
26 
27 
28 import java.nio.file.Path;
29 import java.util.spi.ToolProvider;
30 
31 public enum JavaTool {
32     JAVA, JAVAC, JPACKAGE, JAR, JLINK, JMOD;
33 
34     JavaTool() {
35         this.path = Path.of(System.getProperty("java.home")).resolve(
36                 relativePathInJavaHome()).toAbsolutePath().normalize();
37         if (!path.toFile().exists()) {
38             throw new RuntimeException(String.format(
39                     "Unable to find tool [%s] at path=[%s]", toolName(), path));
40         }
41     }
42 
43     Path getPath() {
44         return path;
45     }
46 
47     public ToolProvider asToolProvider() {
48         return ToolProvider.findFirst(toolName()).orElse(null);
49     }
50 
51     Path relativePathInJavaHome() {
52         Path path = Path.of("bin", toolName());
53         if (TKit.isWindows()) {
54             path = path.getParent().resolve(path.getFileName().toString() + ".exe");
55         }
56         return path;
57     }
58 
59     private String toolName() {
60         return name().toLowerCase();
61     }
62 
63     private Path path;
64 }
