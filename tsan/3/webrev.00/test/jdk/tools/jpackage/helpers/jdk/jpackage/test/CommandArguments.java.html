23 package jdk.jpackage.test;
24 
25 import java.nio.file.Path;
26 import java.util.ArrayList;
27 import java.util.Arrays;
28 import java.util.List;
29 import java.util.stream.Collectors;
30 
31 public class CommandArguments<T> {
32 
33     CommandArguments() {
34         args = new ArrayList<>();
35     }
36 
37     final public T addArgument(String v) {
38         args.add(v);
39         return (T) this;
40     }
41 
42     final public T addArguments(List<String> v) {
43         args.addAll(v);
44         return (T) this;
45     }
46 
47     final public T addArgument(Path v) {
48         return addArgument(v.toString());
49     }
50 
51     final public T addArguments(String... v) {
52         return addArguments(Arrays.asList(v));
53     }
54 
55     final public T addPathArguments(List<Path> v) {
56         return addArguments(v.stream().map((p) -> p.toString()).collect(
57                 Collectors.toList()));
58     }
59 
60     final public List<String> getAllArguments() {
61         return List.copyOf(args);
62     }
63 
64     protected void verifyMutable() {
65         if (!isMutable()) {
66             throw new UnsupportedOperationException(
67                     "Attempt to modify immutable object");
68         }
69     }
70 
71     protected boolean isMutable() {
72         return true;
73     }
74 
75     protected List<String> args;
76 }
