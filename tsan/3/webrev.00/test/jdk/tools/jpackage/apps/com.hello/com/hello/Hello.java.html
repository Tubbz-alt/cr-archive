24 package com.hello;
25 
26 import java.io.BufferedWriter;
27 import java.io.File;
28 import java.io.FileWriter;
29 import java.io.PrintWriter;
30 
31 public class Hello {
32 
33     private static final String MSG = "jpackage test application";
34     private static final int EXPECTED_NUM_OF_PARAMS = 3; // Starts at 1
35 
36     public static void main(String[] args) {
37         String outputFile = "appOutput.txt";
38         File file = new File(outputFile);
39 
40         try (PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(file)))) {
41             System.out.println(MSG);
42             out.println(MSG);
43 
44             System.out.println("args.length: " + args.length);
45             out.println("args.length: " + args.length);
46 
47             for (String arg : args) {
48                 System.out.println(arg);
49                 out.println(arg);
50             }
51 
52             for (int index = 1; index <= EXPECTED_NUM_OF_PARAMS; index++) {
53                 String value = System.getProperty("param" + index);
54                 if (value != null) {
55                     System.out.println("-Dparam" + index + "=" + value);
56                     out.println("-Dparam" + index + "=" + value);
57                 }
58             }
59         } catch (Exception ex) {
60             System.err.println(ex.toString());
61         }
62     }
63 
64 }
