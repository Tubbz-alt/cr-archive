 24 import java.io.BufferedWriter;
 25 import java.io.File;
 26 import java.io.FileWriter;
 27 import java.io.PrintWriter;
 28 import java.awt.Desktop;
 29 import java.awt.desktop.OpenFilesEvent;
 30 import java.awt.desktop.OpenFilesHandler;
 31 import java.util.List;
 32 
 33 public class Hello implements OpenFilesHandler {
 34 
 35     private static final String MSG = "jpackage test application";
 36     private static final int EXPECTED_NUM_OF_PARAMS = 3; // Starts at 1
 37     private static List<File> files;
 38 
 39     public static void main(String[] args) {
 40         if(Desktop.getDesktop().isSupported(Desktop.Action.APP_OPEN_FILE)) {
 41             Desktop.getDesktop().setOpenFileHandler(new Hello());
 42             try {
 43                 Thread.sleep(1000);
 44             } catch (InterruptedException e) {
 45                 e.printStackTrace();
 46             }
 47         }
 48         printToStdout(args);
 49         if (args.length == 1 || (files != null && files.size() == 1)) { // Called via file association
 50             printToFile(args);
 51         }
 52     }
 53 
 54     private static void printToStdout(String[] args) {
 55         System.out.println(MSG);
 56 
 57         System.out.println("args.length: " + (files == null ? args.length : args.length + files.size()));
 58 
 59         for (String arg : args) {
 60             System.out.println(arg);
 61         }
 62 
 63         if (files != null) {
 64             for (File file : files) {
 65                 System.out.println(file.getAbsolutePath());
 66             }
 67         }
 68 
 69         for (int index = 1; index <= EXPECTED_NUM_OF_PARAMS; index++) {
 70             String value = System.getProperty("param" + index);
 71             if (value != null) {
 72                 System.out.println("-Dparam" + index + "=" + value);
 73             }
 74         }
 75     }
 76 
 77     private static void printToFile(String[] args) {
 78         File inputFile = files == null ? new File(args[0]) : files.get(0);
 79         String outputFile = inputFile.getParent() + File.separator + "appOutput.txt";
 80         File file = new File(outputFile);
 81 
 82         try (PrintWriter out
 83                 = new PrintWriter(new BufferedWriter(new FileWriter(file)))) {
 84             out.println(MSG);
 85 
 86             out.println("args.length: " + (files == null ? args.length : args.length + files.size()));
 87 
 88             for (String arg : args) {
 89                 out.println(arg);
 90             }
 91 
 92             if (files != null) {
 93                 for (File f : files) {
 94                     out.println(f.getAbsolutePath());
 95                 }
 96             }
 97 
 98             for (int index = 1; index <= EXPECTED_NUM_OF_PARAMS; index++) {
 99                 String value = System.getProperty("param" + index);
100                 if (value != null) {
101                     out.println("-Dparam" + index + "=" + value);
102                 }
103             }
104         } catch (Exception ex) {
105             System.err.println(ex.getMessage());
106         }
107     }
108 
109     @Override
110     public void openFiles(OpenFilesEvent e) {
111         files = e.getFiles();
112     }
113 }
