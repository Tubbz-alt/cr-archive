24 import java.nio.file.Path;
25 import jdk.jpackage.test.TKit;
26 import jdk.jpackage.test.JPackageCommand;
27 import jdk.jpackage.test.PackageTest;
28 import jdk.jpackage.test.Annotations.Test;
29 
30 /**
31  * Test --app-image parameter. The output installer should provide the same
32  * functionality as the default installer (see description of the default
33  * installer in SimplePackageTest.java)
34  */
35 
36 /*
37  * @test
38  * @summary jpackage with --app-image
39  * @key jpackagePlatformPackage
40  * @library ../helpers
41  * @requires (jpackage.test.SQETest == null)
42  * @build jdk.jpackage.test.*
43  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
44  * @compile AppImagePackageTest.java
45  * @run main/othervm/timeout=540 -Xmx512m jdk.jpackage.test.Main
46  *  --jpt-run=AppImagePackageTest
47  */
48 public class AppImagePackageTest {
49 
50     @Test
51     public static void test() {
52         Path appimageOutput = TKit.workDir().resolve("appimage");
53 
54         JPackageCommand appImageCmd = JPackageCommand.helloAppImage()
55                 .setArgumentValue("--dest", appimageOutput);
56 
57         new PackageTest()
58         .addRunOnceInitializer(() -> appImageCmd.execute())
59         .addInitializer(cmd -> {
60             cmd.addArguments("--app-image", appImageCmd.outputBundle());
61             cmd.removeArgumentWithValue("--input");
62         }).addBundleDesktopIntegrationVerifier(false).run();
63     }
64 }
