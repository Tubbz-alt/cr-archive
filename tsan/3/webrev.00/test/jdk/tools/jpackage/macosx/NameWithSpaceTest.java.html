24 import jdk.jpackage.test.PackageTest;
25 import jdk.jpackage.test.Annotations.Test;
26 
27 /**
28  * Name with space packaging test. Output of the test should be
29  * "Name With Space-*.*" package bundle.
30  *
31  * macOS only:
32  *
33  * Test should generates basic pkg and dmg. Name of packages and application itself
34  * should have name: "Name With Space". Package should be installed into "/Applications"
35  * folder and verified that it can be installed and run.
36  */
37 
38 /*
39  * @test
40  * @summary jpackage test with name containing spaces
41  * @library ../helpers
42  * @build jdk.jpackage.test.*
43  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
44  * @compile NameWithSpaceTest.java
45  * @requires (os.family == "mac")
46  * @key jpackagePlatformPackage
47  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
48  *  --jpt-run=NameWithSpaceTest
49  */
50 public class NameWithSpaceTest {
51 
52     @Test
53     public static void test() {
54         new PackageTest()
55         .configureHelloApp()
56         .addBundleDesktopIntegrationVerifier(false)
57         .addInitializer(cmd -> {
58             cmd.setArgumentValue("--name", "Name With Space");
59         })
60         .run();
61     }
62 }
