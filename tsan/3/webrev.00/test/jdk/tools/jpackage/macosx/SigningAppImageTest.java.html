24 import java.nio.file.Path;
25 import jdk.jpackage.test.JPackageCommand;
26 import jdk.jpackage.test.TKit;
27 
28 /**
29  * Tests generation of app image with --mac-sign and related arguments. Test will
30  * generate app image and verify signature of main launcher and app bundle itself.
31  * This test requires that machine is configured with test certificate for
32  * "Developer ID Application: jpackage.openjdk.java.net" in jpackagerTest keychain with
33  * always allowed access to this keychain for user which runs test.
34  */
35 
36 /*
37  * @test
38  * @summary jpackage with --type app-image --mac-sign
39  * @library ../helpers
40  * @library /test/lib
41  * @library base
42  * @build SigningBase
43  * @build SigningCheck
44  * @build jtreg.SkippedException
45  * @build jdk.jpackage.test.*
46  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
47  * @requires (os.family == "mac")
48  * @run main/othervm -Xmx512m SigningAppImageTest
49  */
50 public class SigningAppImageTest {
51 
52     public static void main(String[] args) throws Exception {
53         TKit.run(args, () -> {
54             SigningCheck.checkCertificates();
55 
56             JPackageCommand cmd = JPackageCommand.helloAppImage();
57             cmd.addArguments("--mac-sign", "--mac-signing-key-user-name",
58                     SigningBase.DEV_NAME, "--mac-signing-keychain",
59                     "jpackagerTest.keychain");
60             cmd.executeAndAssertHelloAppImageCreated();
61 
62             Path launcherPath = cmd.appLauncherPath();
63             SigningBase.verifyCodesign(launcherPath, true);
64 
65             Path appImage = cmd.outputBundle();
66             SigningBase.verifyCodesign(appImage, true);
67             SigningBase.verifySpctl(appImage, "exec");
68         });
69     }
70 }
