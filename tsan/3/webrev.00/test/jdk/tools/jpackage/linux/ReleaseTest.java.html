24 import jdk.jpackage.test.PackageType;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.TKit;
27 
28 
29 /**
30  * Test --linux-app-release parameter. Output of the test should be
31  * releasetest_1.0-Rc3_amd64.deb or releasetest-1.0-Rc3.amd64.rpm package
32  * bundle. The output package should provide the same functionality as the
33  * default package.
34  *
35  * deb:
36  * Version property of the package should end with -Rc3 substring.
37  *
38  * rpm:
39  * Release property of the package should be set to Rc3 value.
40  */
41 
42 /*
43  * @test
44  * @summary jpackage with --linux-app-release
45  * @library ../helpers
46  * @key jpackagePlatformPackage
47  * @build jdk.jpackage.test.*
48  * @requires (os.family == "linux")
49  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
50  * @run main/othervm/timeout=360 -Xmx512m ReleaseTest
51  */
52 public class ReleaseTest {
53 
54     public static void main(String[] args) {
55         final String RELEASE = "Rc3";
56 
57         TKit.run(args, () -> {
58             new PackageTest()
59             .forTypes(PackageType.LINUX)
60             .configureHelloApp()
61             .addInitializer(cmd -> {
62                 cmd.addArguments("--linux-app-release", RELEASE);
63             })
64             .forTypes(PackageType.LINUX_RPM)
65             .addBundlePropertyVerifier("Release", RELEASE)
66             .forTypes(PackageType.LINUX_DEB)
67             .addBundlePropertyVerifier("Version", propValue -> {
68                 return propValue.endsWith("-" + RELEASE);
69             }, "ends with")
70             .run();
71         });
72     }
73 }
