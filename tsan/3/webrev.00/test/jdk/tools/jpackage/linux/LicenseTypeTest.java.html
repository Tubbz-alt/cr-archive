24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 
28 
29 /**
30  * Test --linux-rpm-license-type parameter. Output of the test should be
31  * licensetypetest-1.0-1.amd64.rpm package bundle. The output package
32  * should provide the same functionality as the
33  * default package.
34  * License property of the package should be set to JP_LICENSE_TYPE.
35  */
36 
37 
38 /*
39  * @test
40  * @summary jpackage with --linux-rpm-license-type
41  * @library ../helpers
42  * @key jpackagePlatformPackage
43  * @build jdk.jpackage.test.*
44  * @requires (os.family == "linux")
45  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
46  * @run main/othervm/timeout=360 -Xmx512m LicenseTypeTest
47  */
48 public class LicenseTypeTest {
49 
50     public static void main(String[] args) {
51         final String LICENSE_TYPE = "JP_LICENSE_TYPE";
52 
53         TKit.run(args, () -> {
54             new PackageTest().forTypes(PackageType.LINUX_RPM).configureHelloApp()
55             .addInitializer(cmd -> {
56                 cmd.addArguments("--linux-rpm-license-type", LICENSE_TYPE);
57             })
58             .addBundlePropertyVerifier("License", LICENSE_TYPE)
59             .run();
60         });
61     }
62 }
