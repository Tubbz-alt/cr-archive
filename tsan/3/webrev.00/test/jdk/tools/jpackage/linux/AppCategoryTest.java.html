24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 
28 
29 /**
30  * Test --linux-app-category parameter. Output of the test should be
31  * appcategorytest_1.0-1_amd64.deb or appcategorytest-1.0-1.amd64.rpm package
32  * bundle. The output package should provide the same functionality as the
33  * default package.
34  *
35  * deb:
36  * Section property of the package should be set to Foo value.
37  *
38  * rpm:
39  * Group property of the package should be set to Foo value.
40  */
41 
42 
43 /*
44  * @test
45  * @summary jpackage with --linux-app-category
46  * @library ../helpers
47  * @key jpackagePlatformPackage
48  * @build jdk.jpackage.test.*
49  * @requires (os.family == "linux")
50  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
51  * @run main/othervm/timeout=360 -Xmx512m AppCategoryTest
52  */
53 public class AppCategoryTest {
54 
55     public static void main(String[] args) {
56         final String CATEGORY = "Foo";
57 
58         TKit.run(args, () -> {
59             new PackageTest()
60             .forTypes(PackageType.LINUX)
61             .configureHelloApp()
62             .addInitializer(cmd -> {
63                 cmd.addArguments("--linux-app-category", CATEGORY);
64             })
65             .forTypes(PackageType.LINUX_DEB)
66             .addBundlePropertyVerifier("Section", CATEGORY)
67             .forTypes(PackageType.LINUX_RPM)
68             .addBundlePropertyVerifier("Group", CATEGORY)
69             .run();
70         });
71     }
72 }
