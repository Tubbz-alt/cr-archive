24 /*
25  * @Xtest
26  * @bug 8008077
27  * @summary new type annotation location: lambda expressions
28  * @compile Lambda.java
29  * @author Werner Dietl
30  */
31 
32 import java.lang.annotation.*;
33 
34 public class Lambda {
35 
36     interface LambdaInt {
37         <S, T> void generic(S p1, T p2);
38     }
39 
40     static class LambdaImpl implements LambdaInt {
41         <S, T> LambdaImpl(S p1, T p2) {}
42         public <S, T> void generic(S p1, T p2) {}
43     }
44 
45     LambdaInt getMethodRefTA(LambdaImpl r) {
46         return r::<@TA Object, @TB Object>generic;
47     }
48 
49     LambdaInt getConstructorRefTA() {
50         return LambdaImpl::<@TA Object, @TB Object>new;
51     }
52 
53 }
54 
55 @Target({ElementType.TYPE_USE, ElementType.TYPE_PARAMETER})
56 @interface TA { }
57 
58 @Target({ElementType.TYPE_USE, ElementType.TYPE_PARAMETER})
59 @interface TB { }
