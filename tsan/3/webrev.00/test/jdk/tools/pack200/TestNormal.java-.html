25 /*
26  * @test
27  * @run main/timeout=600 TestNormal
28  * @bug 8020802 8156807
29  * @summary Need an ability to create jar files that are invariant to the pack200 packing/unpacking
30  * @author Alexander Zuev
31  */
32 
33 import java.io.*;
34 
35 public class TestNormal {
36     private static String FS = File.separator;
37 
38     public static void main(String args[]) throws Exception {
39         String testdir = Utils.TEST_CLS_DIR.getAbsolutePath();
40 
41         try {
42             String jarCmd = Utils.getJarCmd();
43             String packCmd = Utils.getPack200Cmd();
44 
45             // create the original jar
46             Utils.runExec(jarCmd, "cf", "original.jar", "-C", testdir, ".");
47 
48             // create the reference jar
49             Utils.runExec(packCmd, "-r", "repacked.jar", "original.jar");
50 
51             // create the normalized jar using jar(1)
52             Utils.runExec(jarCmd, "cnf", "normalized.jar", "-C", testdir, ".");
53 
54             // compare archive contents bit wise, these should be identical!
55             Utils.doCompareBitWise(new File("repacked.jar"),
56                     new File("normalized.jar"));
57         } finally {
58            Utils.cleanup();
59         }
60     }
61 }
