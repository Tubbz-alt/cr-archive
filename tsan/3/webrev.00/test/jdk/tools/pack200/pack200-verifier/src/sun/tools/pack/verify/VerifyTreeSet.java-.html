24 package sun.tools.pack.verify;
25 
26 import java.util.*;
27 /*
28  * @author ksrini
29  */
30 
31 class VerifyTreeSet<K> extends java.util.TreeSet {
32 
33     VerifyTreeSet() {
34         super();
35     }
36 
37     public VerifyTreeSet(Comparator c) {
38         super(c);
39     }
40 
41     public TreeSet<K> diff(TreeSet in) {
42         TreeSet<K> delta = (TreeSet<K>) this.clone();
43         delta.removeAll(in);
44         return delta;
45     }
46 }
