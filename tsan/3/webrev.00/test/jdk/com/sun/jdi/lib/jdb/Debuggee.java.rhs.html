 24 package lib.jdb;
 25 
 26 import jdk.test.lib.Utils;
 27 import jdk.test.lib.process.ProcessTools;
 28 
 29 import java.io.Closeable;
 30 import java.io.IOException;
 31 import java.util.LinkedList;
 32 import java.util.List;
 33 import java.util.concurrent.TimeUnit;
 34 import java.util.concurrent.TimeoutException;
 35 import java.util.regex.Matcher;
 36 import java.util.regex.Pattern;
 37 import java.util.stream.Collectors;
 38 
 39 /**
 40  * Helper class to run java debuggee and parse agent listening transport/address.
 41  * Usage:
 42  *   1)
 43  *      Debugee debugee = Debuggee.launcher("MyClass").setTransport("dt_shmem").launch();
 44  *      try {
 45  *          String transport = debuggee.getTransport();
 46  *          String addr = debuggee.getAddress();
 47  *      } finally {
 48  *          debuggee.shutdown();
 49  *      }
 50  *   2) (using try-with-resource)
 51  *      try (Debugee debugee = Debuggee.launcher("MyClass").launch()) {
 52  *          String transport = debuggee.getTransport();
 53  *          String addr = debuggee.getAddress();
 54  *      }
 55  *   3)
 56  *      ProcessBuilder pb = Debuggee.launcher("MyClass").setSuspended(false).prepare();
 57  *      ProcessTools.executeProcess(pb);
 58  */
 59 public class Debuggee implements Closeable {
 60 
 61     public static Launcher launcher(String mainClass) {
 62         return new Launcher(mainClass);
 63     }
 64 
 65     public static class Launcher {
 66         private final String mainClass;
 67         private final List<String> options = new LinkedList<>();
 68         private String vmOptions = null;
 69         private String transport = "dt_socket";
 70         private String address = null;
 71         private boolean suspended = true;
 72         private boolean addTestVmAndJavaOptions = true;
 73 
 74         private Launcher(String mainClass) {
 75             this.mainClass = mainClass;
 76         }
 77         public Launcher addOption(String option) {
 78             options.add(option);
 79             return this;
 80         }
 81         public Launcher addOptions(List<String> options) {
 82             this.options.addAll(options);
 83             return this;
 84         }
 85         public Launcher addVMOptions(String vmOptions) {
 86             this.vmOptions = vmOptions;
 87             return this;
 88         }
 89         // default is "dt_socket"
 90         public Launcher setTransport(String value) {
 91             transport = value;
 92             return this;
 93         }
 94         // default is "null" (auto-generate)
 95         public Launcher setAddress(String value) {
 96             address = value;
 97             return this;
 98         }
 99         // default is "true"
100         public Launcher setSuspended(boolean value) {
101             suspended = value;
102             return this;
103         }
104         // default is "true"
105         public Launcher addTestVmAndJavaOptions(boolean value) {
106             addTestVmAndJavaOptions = value;
107             return this;
108         }
109 
110         public ProcessBuilder prepare() {
111             List<String> debuggeeArgs = new LinkedList<>();
112             if (vmOptions != null) {
113                 debuggeeArgs.add(vmOptions);
114             }
115             debuggeeArgs.add("-agentlib:jdwp=transport=" + transport
116                     + (address == null ? "" : ",address=" + address)
117                     + ",server=y,suspend=" + (suspended ? "y" : "n"));
118             debuggeeArgs.addAll(options);
119             debuggeeArgs.add(mainClass);
120             return ProcessTools.createJavaProcessBuilder(addTestVmAndJavaOptions,
121                     debuggeeArgs.toArray(new String[0]));
122         }
123 
124         public Debuggee launch(String name) {
125             return new Debuggee(prepare(), name);
126         }
127         public Debuggee launch() {
128             return launch("debuggee");
129         }
130     }
131 
132     // starts the process, waits for "Listening for transport" output and detects transport/address
133     private Debuggee(ProcessBuilder pb, String name) {
134         // debuggeeListen[0] - transport, debuggeeListen[1] - address
135         String[] debuggeeListen = new String[2];
136         Pattern listenRegexp = Pattern.compile("Listening for transport \\b(.+)\\b at address: \\b(.+)\\b");
137         try {
138             p = ProcessTools.startProcess(name, pb,
139                     s -> output.add(s),  // output consumer
140                     s -> {  // warm-up predicate
141                         Matcher m = listenRegexp.matcher(s);
142                         if (!m.matches()) {
143                             return false;
144                         }
145                         debuggeeListen[0] = m.group(1);
146                         debuggeeListen[1] = m.group(2);
147                         return true;
148                     },
149                     30, TimeUnit.SECONDS);
150             transport = debuggeeListen[0];
151             address = debuggeeListen[1];
152         } catch (IOException | InterruptedException | TimeoutException ex) {
153             throw new RuntimeException("failed to launch debuggee", ex);
154         }
155     }
156 
157     private final Process p;
158     private final List<String> output = new LinkedList<>();
159     private final String transport;
160     private final String address;
161 
162     public void shutdown() {
163         try {
164             close();
165         } catch (IOException ex) {
166             // ignore
167         }
168     }
169 
170     // waits until the process shutdown or crash
171     public boolean waitFor(long timeout, TimeUnit unit) {
172         try {
173             return p.waitFor(Utils.adjustTimeout(timeout), unit);
174         } catch (InterruptedException e) {
175             return false;
176         }
177     }
178 
179     // returns the whole debuggee output as a string
180     public String getOutput() {
181         return output.stream().collect(Collectors.joining(Utils.NEW_LINE));
182     }
183 
184     String getTransport() {
185         return transport;
186     }
187 
188     public String getAddress() {
189         return address;
190     }
191 
192     @Override
193     public void close() throws IOException {
194         if (p.isAlive()) {
195             p.destroy();
196         }
197     }
198 
199 }
