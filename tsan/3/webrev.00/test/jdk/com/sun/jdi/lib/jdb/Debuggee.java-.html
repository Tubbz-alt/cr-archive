 24 package lib.jdb;
 25 
 26 import jdk.test.lib.Utils;
 27 import jdk.test.lib.process.ProcessTools;
 28 
 29 import java.io.Closeable;
 30 import java.io.IOException;
 31 import java.util.LinkedList;
 32 import java.util.List;
 33 import java.util.concurrent.TimeUnit;
 34 import java.util.concurrent.TimeoutException;
 35 import java.util.regex.Matcher;
 36 import java.util.regex.Pattern;
 37 import java.util.stream.Collectors;
 38 
 39 /**
 40  * Helper class to run java debuggee and parse agent listening transport/address.
 41  * Usage:
 42  *   1)
 43  *      Debugee debugee = Debuggee.launcher("MyClass").setTransport("dt_shmem").launch();
 44  *      try {
 45  *          String transport = debuggee.getTransport();
 46  *          String addr = debuggee.getAddress();
 47  *      } finally {
 48  *          debuggee.shutdown();
 49  *      }
 50  *   2) (using try-with-resource)
 51  *      try (Debugee debugee = Debuggee.launcher("MyClass").launch()) {
 52  *          String transport = debuggee.getTransport();
 53  *          String addr = debuggee.getAddress();
 54  *      }
 55  *   3)
 56  *      ProcessBuilder pb = Debuggee.launcher("MyClass").setSuspended(false).prepare();
 57  *      ProcessTools.executeProcess(pb);
 58  */
 59 public class Debuggee implements Closeable {
 60 
 61     public static Launcher launcher(String mainClass) {
 62         return new Launcher(mainClass);
 63     }
 64 
 65     public static class Launcher {
 66         private final String mainClass;
 67         private final List<String> options = new LinkedList<>();
 68         private String transport = "dt_socket";
 69         private String address = null;
 70         private boolean suspended = true;
 71         private boolean addTestVmAndJavaOptions = true;
 72 
 73         private Launcher(String mainClass) {
 74             this.mainClass = mainClass;
 75         }
 76         public Launcher addOption(String option) {
 77             options.add(option);
 78             return this;
 79         }
 80         public Launcher addOptions(List<String> options) {
 81             this.options.addAll(options);
 82             return this;
 83         }
 84         // default is "dt_socket"
 85         public Launcher setTransport(String value) {
 86             transport = value;
 87             return this;
 88         }
 89         // default is "null" (auto-generate)
 90         public Launcher setAddress(String value) {
 91             address = value;
 92             return this;
 93         }
 94         // default is "true"
 95         public Launcher setSuspended(boolean value) {
 96             suspended = value;
 97             return this;
 98         }
 99         // default is "true"
100         public Launcher addTestVmAndJavaOptions(boolean value) {
101             addTestVmAndJavaOptions = value;
102             return this;
103         }
104 
105         public ProcessBuilder prepare() {
106             List<String> debuggeeArgs = new LinkedList<>();
107             debuggeeArgs.add("-agentlib:jdwp=transport=" + transport
108                     + (address == null ? "" : ",address=" + address)
109                     + ",server=y,suspend=" + (suspended ? "y" : "n"));
110             debuggeeArgs.addAll(options);
111             debuggeeArgs.add(mainClass);
112             return ProcessTools.createJavaProcessBuilder(addTestVmAndJavaOptions,
113                     debuggeeArgs.toArray(new String[0]));
114         }
115 
116         public Debuggee launch(String name) {
117             return new Debuggee(prepare(), name);
118         }
119         public Debuggee launch() {
120             return launch("debuggee");
121         }
122     }
123 
124     // starts the process, waits for "Listening for transport" output and detects transport/address
125     private Debuggee(ProcessBuilder pb, String name) {
126         // debuggeeListen[0] - transport, debuggeeListen[1] - address
127         String[] debuggeeListen = new String[2];
128         Pattern listenRegexp = Pattern.compile("Listening for transport \\b(.+)\\b at address: \\b(.+)\\b");
129         try {
130             p = ProcessTools.startProcess(name, pb,
131                     s -> output.add(s),  // output consumer
132                     s -> {  // warm-up predicate
133                         Matcher m = listenRegexp.matcher(s);
134                         if (!m.matches()) {
135                             return false;
136                         }
137                         debuggeeListen[0] = m.group(1);
138                         debuggeeListen[1] = m.group(2);
139                         return true;
140                     },
141                     30, TimeUnit.SECONDS);
142             transport = debuggeeListen[0];
143             address = debuggeeListen[1];
144         } catch (IOException | InterruptedException | TimeoutException ex) {
145             throw new RuntimeException("failed to launch debuggee", ex);
146         }
147     }
148 
149     private final Process p;
150     private final List<String> output = new LinkedList<>();
151     private final String transport;
152     private final String address;
153 
154     public void shutdown() {
155         try {
156             close();
157         } catch (IOException ex) {
158             // ignore
159         }
160     }
161 
162     // waits until the process shutdown or crash
163     public boolean waitFor(long timeout, TimeUnit unit) {
164         try {
165             return p.waitFor(Utils.adjustTimeout(timeout), unit);
166         } catch (InterruptedException e) {
167             return false;
168         }
169     }
170 
171     // returns the whole debuggee output as a string
172     public String getOutput() {
173         return output.stream().collect(Collectors.joining(Utils.NEW_LINE));
174     }
175 
176     String getTransport() {
177         return transport;
178     }
179 
180     public String getAddress() {
181         return address;
182     }
183 
184     @Override
185     public void close() throws IOException {
186         if (p.isAlive()) {
187             p.destroy();
188         }
189     }
190 
191 }
