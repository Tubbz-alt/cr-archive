 26  * @bug 4401399

 34 import com.sun.jdi.ReferenceType;
 35 import com.sun.jdi.ThreadReference;
 36 import com.sun.jdi.event.BreakpointEvent;
 37 import com.sun.jdi.event.MonitorContendedEnterEvent;
 38 import com.sun.jdi.event.MonitorContendedEnteredEvent;
 39 import com.sun.jdi.event.MonitorWaitEvent;
 40 import com.sun.jdi.event.MonitorWaitedEvent;
 41 import com.sun.jdi.request.EventRequest;
 42 import com.sun.jdi.request.MonitorContendedEnterRequest;
 43 import com.sun.jdi.request.MonitorContendedEnteredRequest;
 44 import com.sun.jdi.request.MonitorWaitRequest;
 45 import com.sun.jdi.request.MonitorWaitedRequest;
 46 
 47 /********** target program **********/
 48 
 49 class MonitorEventTestTarg {
 50     public static Object endingMonitor;
 51     public static Object startingMonitor;
 52     public static final long timeout = 30 * 6000; // milliseconds
 53 
 54     public static volatile boolean aboutEnterLock;
 55 
 56     static void foo() {
 57         System.out.println("Howdy!");
 58     }
 59 
 60     public static void main(String[] args){
 61         endingMonitor = new Object();
 62         startingMonitor = new Object();
 63 
 64         myThread t1 = new myThread();
 65         foo();
 66         aboutEnterLock = false;
 67 
 68         synchronized(endingMonitor) {
 69 
 70             // run thread
 71             try {
 72                 // start thread
 73                 synchronized (startingMonitor) {
 74                     t1.start();
 75                     startingMonitor.wait(timeout);
 76                 }
 77             } catch (InterruptedException e) {
 78                 System.out.println("Interrupted exception " + e);
 79             }
 80 
 81             Thread.yield();
 82             while(!(aboutEnterLock && t1.getState() == Thread.State.BLOCKED)) {
 83                 try {
 84                     Thread.sleep(1000);
 85                 }catch(Exception x){
 86                     System.out.println(x);
 87                 }
 88             }
 89         }
 90         try {
 91             t1.join(timeout);
 92         } catch (Exception x){
 93             System.out.println("Exception while thread.join :" + x);
 94         }
 95         System.out.println("Test exiting");
 96     }
 97 }
 98 
 99 class myThread extends Thread {
100     public void run() {
101         synchronized(MonitorEventTestTarg.startingMonitor) {
102             MonitorEventTestTarg.startingMonitor.notify();
103         }
104 
105         // contended enter wait until main thread release monitor
106         MonitorEventTestTarg.aboutEnterLock = true;
107         synchronized (MonitorEventTestTarg.endingMonitor) {
108         }
109     }
110 }
111 
112 
113     /********** test program **********/
114 
115 public class MonitorEventTest extends TestScaffold {
116     ReferenceType targetClass;
117     ThreadReference mainThread;

118     MonitorContendedEnterRequest contendedEnterRequest;
119     MonitorWaitedRequest monitorWaitedRequest;
120     MonitorContendedEnteredRequest contendedEnteredRequest;
121     MonitorWaitRequest monitorWaitRequest;
122 
123     static int actualWaitCount = 0;
124     static int actualWaitedCount = 0;
125     static int actualContendedEnterCount = 0;
126     static int actualContendedEnteredCount= 0;
127 
128     MonitorEventTest (String args[]) {
129         super(args);
130     }
131 
132     public static void main(String[] args)      throws Exception {
133         new MonitorEventTest(args).startTests();
134     }
135 
136     /********** event handlers **********/
137 
138     public void monitorContendedEnter(MonitorContendedEnterEvent event) {
139 
140         actualContendedEnterCount++;
141     }
142 
143     public void monitorContendedEntered(MonitorContendedEnteredEvent event) {
144 
145         actualContendedEnteredCount++;
146 
147     }
148 
149     public void monitorWait(MonitorWaitEvent event) {
150 
151         actualWaitCount++;
152 
153     }
154     public void monitorWaited(MonitorWaitedEvent event) {
155 
156         actualWaitedCount++;
157 
158     }
159 
160 
161 
162     /********** test core **********/
163 
164     protected void runTests() throws Exception {
165         /*
166          * Get to the top of main()
167          * to determine targetClass and mainThread
168          */
169         BreakpointEvent bpe = startToMain("MonitorEventTestTarg");
170         targetClass = bpe.location().declaringType();
171         mainThread = bpe.thread();
172 

173         if (vm().canRequestMonitorEvents()) {
174             contendedEnterRequest = eventRequestManager().createMonitorContendedEnterRequest();
175             contendedEnterRequest.setSuspendPolicy(EventRequest.SUSPEND_NONE);
176             contendedEnterRequest.enable();
177             contendedEnteredRequest = eventRequestManager().createMonitorContendedEnteredRequest();
178             contendedEnteredRequest.setSuspendPolicy(EventRequest.SUSPEND_NONE);
179             contendedEnteredRequest.enable();
180             monitorWaitRequest = eventRequestManager().createMonitorWaitRequest();
181             monitorWaitRequest.setSuspendPolicy(EventRequest.SUSPEND_NONE);
182             monitorWaitRequest.enable();
183             monitorWaitedRequest = eventRequestManager().createMonitorWaitedRequest();
184             monitorWaitedRequest.setSuspendPolicy(EventRequest.SUSPEND_NONE);
185             monitorWaitedRequest.enable();
186         } else {
187             System.out.println("request monitor events not supported " );
188         }
189 
190 
191         resumeTo("MonitorEventTestTarg", "foo", "()V");
192 
193         /*
194          * resume until end
195          */
196         listenUntilVMDisconnect();
197 
198         /*
199          * At least one of each type event should have recevied by this test.
200          */
201         if (vm().canRequestMonitorEvents()) {
202             if (actualContendedEnterCount == 0) {
203                 failure("Did not receive any  contended enter event.");
204             }
205             if (actualContendedEnteredCount == 0) {
206                 failure("Did not receive any contended entered event. ");
207             }
208             if (actualWaitCount == 0) {
209                 failure("Did not receive any contended monitor wait event");
210             }
211             if (actualWaitedCount == 0) {
212                 failure("Did not receive any contended monitor waited event");
213             }
214         }
215 
216 
217         /*
218          * deal with results of test
219          * if anything has called failure("foo") testFailed will be true
220          */
221         if (!testFailed) {
222             println("MonitorEventTest: passed");
223         } else {
224             throw new Exception("MonitorEventTest: failed");
225         }
226     }
227 }
