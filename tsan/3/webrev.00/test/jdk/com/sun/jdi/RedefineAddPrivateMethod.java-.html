24 /*
25  * @test
26  * @bug 8149743
27  * @summary crash when adding a breakpoint after redefining to add a private static method
28  * @comment converted from test/jdk/com/sun/jdi/RedefineAddPrivateMethod.sh
29  *
30  * @library /test/lib
31  * @compile -g RedefineAddPrivateMethod.java
32  * @run main/othervm RedefineAddPrivateMethod
33  */
34 
35 import jdk.test.lib.process.OutputAnalyzer;
36 import lib.jdb.JdbCommand;
37 import lib.jdb.JdbTest;
38 
39 class RedefineAddPrivateMethodTarg {
40     static public void main(String[] args) {
41         System.out.println("@1 breakpoint");
42         System.out.println("@2 breakpoint");
43     }
44 
45     // @1 uncomment private static void test() {}
46 }
47 
48 public class RedefineAddPrivateMethod extends JdbTest {
49     public static void main(String argv[]) {
50         new RedefineAddPrivateMethod().run();
51     }
52 
53     private RedefineAddPrivateMethod() {
54         super(DEBUGGEE_CLASS, SOURCE_FILE);
55     }
56 
57     private static final String DEBUGGEE_CLASS = RedefineAddPrivateMethodTarg.class.getName();
58     private static final String SOURCE_FILE = "RedefineAddPrivateMethod.java";
59 
60     @Override
61     protected void runCases() {
62         setBreakpoints(1);
63         jdb.command(JdbCommand.run());
64 
65         redefineClass(1, "-g");
66         // ensure "test()" method has been added successfully
67         execCommand(JdbCommand.eval(DEBUGGEE_CLASS + ".test()"))
68                 .shouldNotContain("ParseException");
69 
70         setBreakpoints(2);
71         jdb.command(JdbCommand.run());
72 
73         jdb.quit();
74 
75         new OutputAnalyzer(getDebuggeeOutput())
76                 .shouldNotContain("Internal exception:");
77     }
78 }
