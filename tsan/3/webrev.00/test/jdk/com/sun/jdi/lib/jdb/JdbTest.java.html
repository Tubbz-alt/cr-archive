 24 package lib.jdb;
 25 
 26 import jdk.test.lib.process.OutputAnalyzer;
 27 
 28 import java.io.IOException;
 29 import java.nio.file.Files;
 30 import java.nio.file.Paths;
 31 import java.util.Arrays;
 32 import java.util.LinkedList;
 33 import java.util.List;
 34 import java.util.concurrent.TimeUnit;
 35 import java.util.stream.Collectors;
 36 
 37 public abstract class JdbTest {
 38 
 39     public static class LaunchOptions {
 40         public final String debuggeeClass;
 41         public final List<String> debuggeeOptions = new LinkedList<>();
 42         public String sourceFilename;
 43         public String vmOptions = null;
 44 
 45         public LaunchOptions(String debuggeeClass) {
 46             this.debuggeeClass = debuggeeClass;
 47         }
 48         public LaunchOptions addDebuggeeOption(String option) {
 49             debuggeeOptions.add(option);
 50             return this;
 51         }
 52         public LaunchOptions addDebuggeeOptions(String[] options) {
 53             debuggeeOptions.addAll(Arrays.asList(options));
 54             return this;
 55         }
 56         public LaunchOptions setSourceFilename(String name) {
 57             sourceFilename = name;
 58             return this;
 59         }
 60         public LaunchOptions addVMOptions(String vmOptions) {
 61             this.vmOptions = vmOptions;
 62             return this;
 63         }
 64     }
 65 
 66     public JdbTest(LaunchOptions launchOptions) {
 67         this.launchOptions = launchOptions;
 68     }
 69     public JdbTest(String debuggeeClass) {
 70         this(new LaunchOptions(debuggeeClass));
 71     }
 72 
 73     // sourceFilename is used by setBreakpoints and redefineClass
 74     public JdbTest(String debuggeeClass, String sourceFilename) {
 75         this(new LaunchOptions(debuggeeClass).setSourceFilename(sourceFilename));
 76     }
 77 
 78     protected Jdb jdb;
 79     protected Debuggee debuggee;
 80     protected LaunchOptions launchOptions;
 81 
 82     // returns the whole jdb output as a string
 83     public String getJdbOutput() {
 84         return jdb == null ? "" : jdb.getJdbOutput();
 85     }
 86 
 87     // returns the whole debuggee output as a string
 88     public String getDebuggeeOutput() {
 89         return debuggee == null ? "" : debuggee.getOutput();
 90     }
 91 
 92     public void run() {
 93         try {
 94             setup();
 95             runCases();
 96         } catch (Throwable e) {
 97             jdb.log("=======================================");
 98             jdb.log("Exception thrown during test execution: " + e.getMessage());
 99             jdb.log("=======================================");
100             throw e;
101         } finally {
102             shutdown();
103         }
104     }
105 
106     protected void setup() {
107         /* run debuggee as:
108             java -agentlib:jdwp=transport=dt_socket,server=n,suspend=y <debuggeeClass>
109         it reports something like : Listening for transport dt_socket at address: 60810
110         after that connect jdb by:
111             jdb -connect com.sun.jdi.SocketAttach:port=60810
112         */
113         // launch debuggee
114         debuggee = Debuggee.launcher(launchOptions.debuggeeClass)
115                 .addOptions(launchOptions.debuggeeOptions)
116                 .addVMOptions(launchOptions.vmOptions)
117                 .launch();
118 
119         // launch jdb
120         try {
121             jdb = new Jdb("-connect", "com.sun.jdi.SocketAttach:port=" + debuggee.getAddress());
122         } catch (Throwable ex) {
123             // terminate debuggee if something went wrong
124             debuggee.shutdown();
125             throw ex;
126         }
127         // wait while jdb is initialized
128         jdb.waitForPrompt(1, false);
129     }
130 
131     protected abstract void runCases();
132 
133     protected void shutdown() {
134         if (jdb != null) {
135             jdb.shutdown();
136         }
137         // shutdown debuggee
138         if (debuggee != null) {
139             if (!debuggee.waitFor(10, TimeUnit.SECONDS)) {
140                 debuggee.shutdown();
141             }
142         }
143     }
144 
145     protected static final String lineSeparator = System.getProperty("line.separator");
146 
147 
148     // Parses the specified source file for "@{id} breakpoint" tags and returns
149     // list of the line numbers containing the tag.
150     // Example:
151     //   System.out.println("BP is here");  // @1 breakpoint
152     public static List<Integer> parseBreakpoints(String filePath, int id) {
153         final String pattern = "@" + id + " breakpoint";
154         int lineNum = 1;
155         List<Integer> result = new LinkedList<>();
156         try {
157             for (String line: Files.readAllLines(Paths.get(filePath))) {
158                 if (line.contains(pattern)) {
159                     result.add(lineNum);
160                 }
161                 lineNum++;
162             }
163         } catch (IOException ex) {
164             throw new RuntimeException("failed to parse " + filePath, ex);
165         }
166         return result;
167     }
168 
169     // sets breakpoints to the lines parsed by {@code parseBreakpoints}
170     // returns number of the breakpoints set.
171     public static int setBreakpoints(Jdb jdb, String debuggeeClass, String sourcePath, int id) {
172         List<Integer> bps = parseBreakpoints(sourcePath, id);
173         for (int bp : bps) {
174             String reply = jdb.command(JdbCommand.stopAt(debuggeeClass, bp)).stream()
175                     .collect(Collectors.joining("\n"));
176             if (reply.contains("Unable to set")) {
177                 throw new RuntimeException("jdb failed to set breakpoint at " + debuggeeClass + ":" + bp);
178             }
179 
180         }
181         return bps.size();
182     }
183 
184     // sets breakpoints to the lines parsed by {@code parseBreakpoints}
185     // from the file from test source directory.
186     // returns number of the breakpoints set.
187     protected int setBreakpointsFromTestSource(String debuggeeFileName, int id) {
188         return setBreakpoints(jdb, launchOptions.debuggeeClass,
189                               getTestSourcePath(debuggeeFileName), id);
190     }
191 
192     // sets breakpoints in the class {@code launchOptions.debuggeeClass}
193     // to the lines parsed by {@code parseBreakpoints}
194     // from the file from test source directory specified by {@code launchOptions.sourceFilename}.
195     // returns number of the breakpoints set.
196     protected int setBreakpoints(int id) {
197         verifySourceFilename();
198         return setBreakpointsFromTestSource(launchOptions.sourceFilename, id);
199     }
200 
201     // transforms class with the specified id (see {@code ClassTransformer})
202     // and executes "redefine" jdb command for {@code launchOptions.debuggeeClass}.
203     // returns reply for the command.
204     protected List<String> redefineClass(int id, String... compilerOptions) {
205         verifySourceFilename();
206         String transformedClassFile = ClassTransformer.fromTestSource(launchOptions.sourceFilename)
207                 .transform(id, launchOptions.debuggeeClass, compilerOptions);
208         return jdb.command(JdbCommand.redefine(launchOptions.debuggeeClass, transformedClassFile));
209     }
210 
211     // gets full test source path for the given test filename
212     public static String getTestSourcePath(String fileName) {
213         return Paths.get(System.getProperty("test.src")).resolve(fileName).toString();
214     }
215 
216     // verifies that sourceFilename is specified in ctor
217     private void verifySourceFilename() {
218         if (launchOptions.sourceFilename == null) {
219             throw new RuntimeException("launchOptions.sourceFilename must be specified.");
220         }
221     }
222 
223     protected OutputAnalyzer execCommand(JdbCommand cmd) {
224         List<String> reply = jdb.command(cmd);
225         return new OutputAnalyzer(reply.stream().collect(Collectors.joining(lineSeparator)));
226     }
227 
228     // helpers for "eval" jdb command.
229     // executes "eval <expr>" and verifies output contains the specified text
230     protected void evalShouldContain(String expr, String expectedString) {
231         execCommand(JdbCommand.eval(expr))
232                 .shouldContain(expectedString);
233     }
234     // executes "eval <expr>" and verifies output does not contain the specified text
235     protected void evalShouldNotContain(String expr, String unexpectedString) {
236         execCommand(JdbCommand.eval(expr))
237                 .shouldNotContain(unexpectedString);
238     }
