24 /* @test
25    @bug 6635110
26    @key headful
27    @summary GTK icons should not throw NPE when called by non-GTK UI
28    @author Peter Zhelezniakov
29    @run main Test6635110
30 */
31 
32 import javax.swing.*;
33 import java.awt.*;
34 import java.awt.image.BufferedImage;
35 import javax.swing.plaf.basic.*;
36 
37 
38 public class Test6635110 implements Runnable {
39 
40     static final int WIDTH = 160;
41     static final int HEIGHT = 80;
42     final BufferedImage IMAGE =
43             new BufferedImage(WIDTH, HEIGHT, BufferedImage.TYPE_INT_ARGB);
44 
45     @Override public void run() {
46         JMenu menu = new JMenu("menu");
47         menu.setUI(new BasicMenuUI());
48         paint(menu);
49 
50         JToolBar tb = new JToolBar();
51         tb.setFloatable(true);
52         tb.setUI(new BasicToolBarUI());
53         paint(tb);
54     }
55 
56     void paint(Component c) {
57         c.setSize(WIDTH, HEIGHT);
58         c.paint(IMAGE.getGraphics());
59     }
60 
61     public static void main(String[] args) throws Exception {
62         try {
63             UIManager.setLookAndFeel("com.sun.java.swing.plaf.gtk.GTKLookAndFeel");
64         } catch (Exception e) {
65             System.out.println("GTKLookAndFeel cannot be set, skipping this test");
66             return;
67         }
68         SwingUtilities.invokeAndWait(new Test6635110());
69     }
70 }
