24 /* @test
25    @bug 6963870

26    @summary Tests that GTKPainter.ListTableFocusBorder.getBorderInsets()
27             doesn't return null
28    @author Peter Zhelezniakov
29    @run main Test6963870
30 */
31 
32 import java.awt.Insets;
33 import javax.swing.SwingUtilities;
34 import javax.swing.UIManager;
35 import javax.swing.border.Border;
36 
37 public class Test6963870 implements Runnable {
38 
39     final static String[] UI_NAMES = {
40         "List.focusCellHighlightBorder",
41         "List.focusSelectedCellHighlightBorder",
42         "List.noFocusBorder",
43         "Table.focusCellHighlightBorder",
44         "Table.focusSelectedCellHighlightBorder",
45     };
46 
47     public void run() {
48         for (String uiName: UI_NAMES) {
49             test(uiName);
50         }
51     }
52 
53     void test(String uiName) {
54         Border b = UIManager.getBorder(uiName);
55         Insets i = b.getBorderInsets(null);
56         if (i == null) {
57             throw new RuntimeException("getBorderInsets() returns null for " + uiName);
58         }
59     }
60 
61     public static void main(String[] args) throws Exception {
62         try {
63             UIManager.setLookAndFeel("com.sun.java.swing.plaf.gtk.GTKLookAndFeel");
64         } catch (Exception e) {
65             System.out.println("GTKLookAndFeel cannot be set, skipping this test");
66             return;
67         }
68 
69         SwingUtilities.invokeAndWait(new Test6963870());
70     }
71 }
