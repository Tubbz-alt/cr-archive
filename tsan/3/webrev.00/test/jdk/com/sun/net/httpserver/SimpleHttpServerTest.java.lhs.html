25  * @test
26  * @bug 8015692

27  * @summary  Test HttpServer instantiation, start, and stop repeated in a loop
28  *           Testing for Bind exception on Windows



29  */
30 
31 import java.net.InetSocketAddress;
32 import java.net.ServerSocket;
33 
34 import com.sun.net.httpserver.HttpServer;
35 
36 
37 public class SimpleHttpServerTest {
38 
39     public static void main(String[] args) throws Exception {
40 
41         System.out.println(System.getProperty("java.version"));
42         InetSocketAddress serverAddr = new InetSocketAddress(0);
43         HttpServer server = HttpServer.create(serverAddr, 0);
44         final int serverPort = server.getAddress().getPort();
45         server.start();
46         server.stop(0);
47         serverAddr = new InetSocketAddress(serverPort);
48         int exceptionCount = 0;

49         System.out.println("Using serverPort == " + serverPort);
50         for (int i = 0; i < 100; i++) {
51             try {
52                 server = HttpServer.create(serverAddr, 0);
53                 server.start();
54                 server.stop(0);
55             } catch (Exception ex) {
56                 ex.printStackTrace();
57                 exceptionCount++;













58             }

59         }
60         if (exceptionCount > 0) {
61            throw new RuntimeException("Test Failed");

62         }
63     }
64 }
