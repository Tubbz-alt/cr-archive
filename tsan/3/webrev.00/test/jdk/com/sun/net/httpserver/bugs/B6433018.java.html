 24 /**
 25  * @test
 26  * @bug 6433018
 27  * @summary  HTTP server sometimes sends bad request for browsers javascript
 28  * @run main B6433018
 29  * @run main/othervm -Djava.net.preferIPv6Addresses=true B6433018
 30  */
 31 
 32 import com.sun.net.httpserver.*;
 33 
 34 import java.util.concurrent.*;
 35 import java.io.*;
 36 import java.net.*;
 37 
 38 public class B6433018 {
 39 
 40     static final String CRLF = "\r\n";
 41 
 42     /* invalid HTTP POST with extra CRLF at end */
 43     /* This checks that the server is able to handle it
 44      * and recognise the second request */
 45 
 46     static final String cmd =
 47         "POST /test/item HTTP/1.1"+CRLF+
 48         "Keep-Alive: 300"+CRLF+
 49         "Proxy-Connection: keep-alive"+CRLF+
 50         "Content-Type: text/xml"+CRLF+
 51         "Content-Length: 22"+CRLF+
 52         "Pragma: no-cache"+CRLF+
 53         "Cache-Control: no-cache"+CRLF+ CRLF+
 54         "<item desc=\"excuse\" />"+CRLF+
 55         "GET /test/items HTTP/1.1"+CRLF+
 56         "Host: araku:9999"+CRLF+
 57         "Accept-Language: en-us,en;q=0.5"+CRLF+
 58         "Accept-Encoding: gzip,deflate"+CRLF+
 59         "Accept-Charset: ISO-8859-1,utf-8;q=0.7,*;q=0.7"+CRLF+
 60         "Keep-Alive: 300"+CRLF+
 61         "Proxy-Connection: keep-alive"+CRLF+
 62         "Pragma: no-cache"+CRLF+
 63         "Cache-Control: no-cache"+CRLF+CRLF;
 64 
 65     public static void main(String[] args) throws Exception {
 66         CountDownLatch finished = new CountDownLatch(2);
 67         Handler handler = new Handler(finished);
 68         InetAddress loopback = InetAddress.getLoopbackAddress();
 69         InetSocketAddress addr = new InetSocketAddress(loopback, 0);
 70         HttpServer server = HttpServer.create(addr, 0);
 71         HttpContext ctx = server.createContext("/test", handler);
 72 
 73         server.start();
 74         int port = server.getAddress().getPort();
 75         try (Socket s = new Socket(loopback, port);
 76              OutputStream os = s.getOutputStream()) {
 77             os.write(cmd.getBytes());
 78             finished.await(30, TimeUnit.SECONDS);
 79         } finally {
 80             server.stop(2);
 81         }
 82 
 83         if (finished.getCount() != 0)
 84             throw new RuntimeException("did not receive the 2 requests");
 85 
 86         System.out.println("OK");
 87     }
 88 
 89     static class Handler implements HttpHandler {
 90         private final CountDownLatch finished;
 91 
 92         Handler(CountDownLatch finished) {
 93             this.finished = finished;
 94         }
 95 
 96         @Override
 97         public void handle(HttpExchange t) throws IOException {
 98             try (InputStream is = t.getRequestBody()) {
 99                 Headers map = t.getRequestHeaders();
100                 Headers rmap = t.getResponseHeaders();
101                 while (is.read() != -1);
102             }
103             t.sendResponseHeaders(200, -1);
104             t.close();
105             finished.countDown();
106         }
107     }
108 }
