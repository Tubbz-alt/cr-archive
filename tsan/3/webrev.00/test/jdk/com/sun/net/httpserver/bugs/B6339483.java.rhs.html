33 import com.sun.net.httpserver.*;
34 
35 import java.util.*;
36 import java.util.concurrent.*;
37 import java.io.*;
38 import java.net.*;
39 import java.security.*;
40 import java.security.cert.*;
41 import javax.net.ssl.*;
42 import jdk.test.lib.net.URIBuilder;
43 
44 public class B6339483 {
45 
46     public static void main (String[] args) throws Exception {
47         InetAddress loopback = InetAddress.getLoopbackAddress();
48         InetSocketAddress addr = new InetSocketAddress (loopback, 0);
49         HttpServer server = HttpServer.create (addr, 0);
50         HttpContext ctx = server.createContext ("/test");
51         ExecutorService executor = Executors.newCachedThreadPool();
52         server.setExecutor (executor);
53         server.start ();
54 
55         URL url = URIBuilder.newBuilder()
56             .scheme("http")
57             .loopback()
58             .port(server.getAddress().getPort())
59             .path("/test/foo.html")
60             .toURL();
61         HttpURLConnection urlc = (HttpURLConnection)url.openConnection(Proxy.NO_PROXY);
62         try {
63             InputStream is = urlc.getInputStream();
64             int c = 0;
65             while (is.read()!= -1) {
66                 c ++;
67             }
68         } catch (IOException e) {
69             server.stop(2);
70             executor.shutdown();
71             System.out.println ("OK");
72         }
73     }
74 }
