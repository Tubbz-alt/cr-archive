26  * @bug 6270015
27  * @summary  Light weight HTTP server
28  */
29 
30 import java.net.*;
31 import java.util.concurrent.*;
32 import java.io.*;
33 import com.sun.net.httpserver.*;
34 
35 public class Test11 {
36     static class Handler implements HttpHandler {
37         public void handle(HttpExchange t) throws IOException {
38             read (t.getRequestBody());
39             String response = "response";
40             t.sendResponseHeaders (200, response.length());
41             OutputStream os = t.getResponseBody();
42             os.write (response.getBytes ("ISO8859_1"));
43             t.close();
44         }
45 
46         void read (InputStream is ) throws IOException {
47             byte[] b = new byte [8096];
48             while (is.read (b) != -1) {}
49         }
50     }
51 
52     public static void main (String[] args) throws Exception {
53         System.out.print ("Test 11: ");
54         HttpServer server = HttpServer.create(new InetSocketAddress(0), 0);
55         ExecutorService s = Executors.newCachedThreadPool();
56         try {
57             HttpContext ctx = server.createContext (
58                 "/foo/bar/", new Handler ()
59             );
60             s =  Executors.newCachedThreadPool();
61             server.start ();
62             URL url = new URL ("http://localhost:" + server.getAddress().getPort()+
63                     "/Foo/bar/test.html");
64             HttpURLConnection urlc = (HttpURLConnection)url.openConnection();
65             int r = urlc.getResponseCode();
66             if (r == 200) {
67                 throw new RuntimeException ("wrong response received");
68             }
69             System.out.println ("OK");
70         } finally {
71             s.shutdown();
72             server.stop(2);
73         }
74     }
75 }
