106     static ClientThread test (boolean fixedLen, String protocol, String root, int port, String f, int size) throws Exception {
107         ClientThread t = new ClientThread (fixedLen, protocol, root, port, f, size);
108         t.start();
109         return t;
110     }

112     static Object fileLock = new Object();
113 
114     static class ClientThread extends Thread {
115 
116         boolean fixedLen;
117         String protocol;
118         String root;
119         int port;
120         String f;
121         int size;
122 
123         ClientThread (boolean fixedLen, String protocol, String root, int port, String f, int size) {
124             this.fixedLen = fixedLen;
125             this.protocol = protocol;
126             this.root = root;
127             this.port = port;
128             this.f =  f;
129             this.size = size;
130         }
131 
132         public void run () {
133             try {
134                 URL url = new URL (protocol+"://localhost:"+port+"/test1/"+f);
135                 HttpURLConnection urlc = (HttpURLConnection) url.openConnection();






136                 if (urlc instanceof HttpsURLConnection) {
137                     HttpsURLConnection urlcs = (HttpsURLConnection) urlc;
138                     urlcs.setHostnameVerifier (new HostnameVerifier () {
139                         public boolean verify (String s, SSLSession s1) {
140                             return true;
141                         }
142                     });
143                     urlcs.setSSLSocketFactory (clientCtx.getSocketFactory());
144                 }
145                 byte [] buf = new byte [4096];
146 
147                 String s = "chunk";
148                 if (fixedLen) {
149                     urlc.setRequestProperty ("XFixed", "yes");
150                     s = "fixed";
151                 }
152                 InputStream is = urlc.getInputStream();
153                 File temp;
154                 synchronized (fileLock) {
155                     temp = File.createTempFile (s, null);
156                     temp.deleteOnExit();
157                 }
158                 OutputStream fout = new BufferedOutputStream (new FileOutputStream(temp));
159                 int c, count = 0;
160                 while ((c=is.read(buf)) != -1) {
161                     count += c;
162                     fout.write (buf, 0, c);
163                 }
164                 is.close();
165                 fout.close();
166 
167                 if (count != size) {
168                     System.out.println ("wrong amount of data returned");
169                     System.out.println ("fixedLen = "+fixedLen);
170                     System.out.println ("protocol = "+protocol);
171                     System.out.println ("root = "+root);
172                     System.out.println ("port = "+port);
173                     System.out.println ("f = "+f);
174                     System.out.println ("size = "+size);
175                     System.out.println ("temp = "+temp);
176                     System.out.println ("count = "+count);
177                     error = true;
178                 }
179                 String orig = root + "/" + f;
180                 compare (new File(orig), temp);
181                 temp.delete();
182             } catch (IOException e) {
183                 e.printStackTrace();
184                 error = true;
185             }
186         }
187     }
188 
189     /* compare the contents of the two files */
190 
191     static void compare (File f1, File f2) throws IOException {
192         InputStream i1 = new BufferedInputStream (new FileInputStream(f1));
193         InputStream i2 = new BufferedInputStream (new FileInputStream(f2));
194 
195         int c1,c2;
196         try {
197             while ((c1=i1.read()) != -1) {
198                 c2 = i2.read();
199                 if (c1 != c2) {
200                     throw new RuntimeException ("file compare failed 1");
201                 }
202             }
203             if (i2.read() != -1) {
204                 throw new RuntimeException ("file compare failed 2");
205             }
206         } finally {
207             i1.close();
208             i2.close();
209         }
210     }
