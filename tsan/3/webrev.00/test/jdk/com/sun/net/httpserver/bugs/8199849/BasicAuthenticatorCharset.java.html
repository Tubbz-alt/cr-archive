 24 /**
 25  * @test
 26  * @bug 8199849
 27  * @library /test/lib
 28  * @run main/othervm/timeout=6000 BasicAuthenticatorCharset
 29  * @summary Check for correct use of character sets with BasicAuthenticator() authentication
 30  */
 31 
 32 import com.sun.net.httpserver.*;
 33 import jdk.test.lib.net.URIBuilder;
 34 
 35 import java.io.IOException;
 36 import java.io.InputStream;
 37 import java.net.*;
 38 
 39 import java.net.http.HttpClient;
 40 import java.net.http.HttpRequest;
 41 import java.net.http.HttpResponse;
 42 import java.util.concurrent.ExecutorService;
 43 import java.util.concurrent.Executors;
 44 import java.nio.charset.Charset;
 45 
 46 import static java.nio.charset.StandardCharsets.UTF_8;
 47 import static java.nio.charset.StandardCharsets.ISO_8859_1;
 48 
 49 /**
 50  * Test authentication
 51  */
 52 
 53 public class BasicAuthenticatorCharset {
 54 
 55     public static volatile int failCount = 0;
 56 
 57     static final String TEST_USER = "test";
 58     static final String UNICODE_PW = "Selam D\u00fcnya. Ho\u015f\u00e7akal D\u00fcnya";
 59 
 60     static Handler testHandler;
 61     static HttpServer testHttpServer;
 62     static java.net.Authenticator clientAuth;
 63     static HttpClient client;
 64 
 65     static class Handler implements HttpHandler {
 66         public void handle(HttpExchange t) throws IOException {
 67             InputStream is = t.getRequestBody();
 68             while (is.read() != -1) ;
 69             is.close();
 70             t.sendResponseHeaders(200, -1);
 71             t.close();
 72         }
 73     }
 74 
 75     static class ClientAuthenticator extends java.net.Authenticator {
 76         public PasswordAuthentication getPasswordAuthentication() {
 77             return new PasswordAuthentication(TEST_USER, UNICODE_PW.toCharArray());
 78         }
 79     }
 80 
 81     static void setAuthenticationPW(String path, String realm, String testPW, Charset charset) {
 82         HttpContext ctx = testHttpServer.createContext(path, testHandler);
 83         BasicAuthenticator auth;
 84         if (charset != null) {
 85             auth = new BasicAuthenticator(realm, charset) {
 86                 public boolean checkCredentials(String username, String pw) {
 87                     return username.equals(TEST_USER) && pw.equals(testPW);
 88                 }
 89             };
 90         } else {
 91             auth = new BasicAuthenticator(realm) {
 92                 public boolean checkCredentials(String username, String pw) {
 93                     return username.equals(TEST_USER) && pw.equals(testPW);
 94                 }
 95             };
 96         }
 97         ctx.setAuthenticator(auth);
 98     }
 99 
100     static void connectAndAuth(String path, int expectedStatus) throws Exception {
101         // path is prepended with /old or /new for old and new http client
102         URL oldurl = URIBuilder.newBuilder()
103                 .scheme("http")
104                 .loopback()
105                 .port(testHttpServer.getAddress().getPort())
106                 .path("/old" + path)
107                 .toURL();
108 
109         URI newuri = URIBuilder.newBuilder()
110             .scheme("http")
111             .loopback()
112             .port(testHttpServer.getAddress().getPort())
113             .path("/new" + path)
114             .build();
115 
116         // check old client
117 
118         HttpURLConnection testConnection = (HttpURLConnection) oldurl.openConnection(Proxy.NO_PROXY);
119 
120         // Check for successful authentication
121         int status = testConnection.getResponseCode();
122         if (status != 401) {
123             InputStream is = testConnection.getInputStream();
124             while (is.read() != -1) ;
125             is.close();
126         }
127         if (status != expectedStatus) {
128             System.err.println("Error (old): " + path);
129             failCount++;
130         }
131 
132         HttpRequest request = HttpRequest.newBuilder()
133             .uri(newuri)
134             .GET()
135             .build();
136 
137         status = -1;
138         try {
139             HttpResponse<Void> response = client.send(request, HttpResponse.BodyHandlers.discarding());
140             status = response.statusCode();
141         } catch (IOException e) {
142             System.out.println("NEW: " + e);
143             status = 401; // limitation in new API.
144         }
145         if (status != expectedStatus) {
146             System.err.println("Error (new): " + path);
147             failCount++;
148         }
149     }
150 
151     public static void main(String[] args) throws Exception {
152         clientAuth = new ClientAuthenticator();
153         client = HttpClient.newBuilder()
154             .authenticator(clientAuth)
155             .build();
156 
157         String defaultCharset = System.getProperty("file.encoding");
158         boolean isUTF8 = defaultCharset.equalsIgnoreCase("UTF-8");
159         testHandler = new Handler();
160         InetSocketAddress addr = new InetSocketAddress(InetAddress.getLoopbackAddress(), 0);
161         testHttpServer = HttpServer.create(addr, 0);
162 
163         // Set the passing credentials OLD client
164         setAuthenticationPW("/old/test1/", "passingCharset@test.realm", UNICODE_PW, UTF_8);
165         setAuthenticationPW("/old/test2/", "failingCharset@test.realm", UNICODE_PW, ISO_8859_1);
166         setAuthenticationPW("/old/test3/", "defaultCharset@test.realm", UNICODE_PW, null);
167 
168         // Set the passing credentials NEW client
169         setAuthenticationPW("/new/test1/", "passingCharset@test.realm", UNICODE_PW, UTF_8);
170         setAuthenticationPW("/new/test2/", "failingCharset@test.realm", UNICODE_PW, ISO_8859_1);
171         setAuthenticationPW("/new/test3/", "defaultCharset@test.realm", UNICODE_PW, null);
172 
173         ExecutorService executor = Executors.newCachedThreadPool();
174         testHttpServer.setExecutor(executor);
175         testHttpServer.start();
176         java.net.Authenticator.setDefault(clientAuth);
177 
178         connectAndAuth("/test1/passingCharset.html", 200);
179         connectAndAuth("/test2/failingCharset.html", 401);
180         if (isUTF8) {
181             connectAndAuth("/test3/defaultCharset.html", 200);
182         }
183 
184         testHttpServer.stop(2);
185         executor.shutdown();
186 
187         // should fail once with UNICODE_PW and unsupporting character set
188         if (failCount > 0)
189             throw new RuntimeException("Fail count : " + failCount);
190     }
191 }
