27  * @summary  NullPointerException when creating a HttpContext with no handler



28  */
29 
30 import com.sun.net.httpserver.*;
31 
32 import java.util.*;
33 import java.util.concurrent.*;
34 import java.io.*;
35 import java.net.*;
36 import java.security.*;
37 import java.security.cert.*;
38 import javax.net.ssl.*;

39 
40 public class B6339483 {
41 
42     public static void main (String[] args) throws Exception {
43         InetSocketAddress addr = new InetSocketAddress (0);

44         HttpServer server = HttpServer.create (addr, 0);
45         HttpContext ctx = server.createContext ("/test");
46         ExecutorService executor = Executors.newCachedThreadPool();
47         server.setExecutor (executor);
48         server.start ();
49 
50         URL url = new URL ("http://localhost:"+server.getAddress().getPort()+"/test/foo.html");
51         HttpURLConnection urlc = (HttpURLConnection)url.openConnection ();





52         try {
53             InputStream is = urlc.getInputStream();
54             int c = 0;
55             while (is.read()!= -1) {
56                 c ++;
57             }
58         } catch (IOException e) {
59             server.stop(2);
60             executor.shutdown();
61             System.out.println ("OK");
62         }
63     }
64 }
