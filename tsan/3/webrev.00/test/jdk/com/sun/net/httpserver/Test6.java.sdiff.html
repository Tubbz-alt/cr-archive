 44 /**
 45  * Test POST large file via chunked encoding (unusually small chunks)
 46  */
 47 
 48 public class Test6 extends Test {
 49 
 50     public static void main (String[] args) throws Exception {
 51         Handler handler = new Handler();
 52         InetAddress loopback = InetAddress.getLoopbackAddress();
 53         InetSocketAddress addr = new InetSocketAddress(loopback, 0);
 54         HttpServer server = HttpServer.create (addr, 0);
 55         HttpContext ctx = server.createContext ("/test", handler);
 56         ExecutorService executor = Executors.newCachedThreadPool();
 57         server.setExecutor (executor);
 58         server.start ();
 59 
 60         URL url = URIBuilder.newBuilder()
 61             .scheme("http")
 62             .loopback()
 63             .port(server.getAddress().getPort())
 64             .path("/test/foo.html")
 65             .toURL();
 66 
 67         System.out.print ("Test6: " );
 68         HttpURLConnection urlc = (HttpURLConnection)url.openConnection(Proxy.NO_PROXY);
 69         urlc.setDoOutput (true);
 70         urlc.setRequestMethod ("POST");
 71         urlc.setChunkedStreamingMode (32); // small chunks
 72         OutputStream os = new BufferedOutputStream (urlc.getOutputStream());
 73         for (int i=0; i<SIZE; i++) {
 74             os.write (i % 100);
 75         }
 76         os.close();
 77         int resp = urlc.getResponseCode();
 78         if (resp != 200) {
 79             throw new RuntimeException ("test failed response code");
 80         }
 81         if (error) {
 82             throw new RuntimeException ("test failed error");
 83         }
 84         delay();
 85         server.stop(2);
 86         executor.shutdown();
 87         System.out.println ("OK");
