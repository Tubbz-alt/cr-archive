140         static HttpURLConnection getHttpURLConnection(URL url, int timeout) throws IOException {
141 
142                 HttpURLConnection httpURLConnection =
143                     (HttpURLConnection) url.openConnection(Proxy.NO_PROXY);
144 
145                 httpURLConnection.setConnectTimeout(40000);
146                 httpURLConnection.setReadTimeout(timeout);
147                 httpURLConnection.setDoOutput(true);
148                 httpURLConnection.setDoInput(true);
149                 httpURLConnection.setUseCaches(false);
150                 httpURLConnection.setAllowUserInteraction(false);
151                 httpURLConnection.setRequestMethod("POST");
152 
153                 // HttpURLConnection httpURLConnection = new MyHttpURLConnection(url);
154 
155                 return httpURLConnection;
156         }
