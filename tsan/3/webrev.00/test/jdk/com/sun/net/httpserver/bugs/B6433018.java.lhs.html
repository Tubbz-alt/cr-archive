 24 /**
 25  * @test
 26  * @bug 6433018
 27  * @summary  HTTP server sometimes sends bad request for browsers javascript


 28  */
 29 
 30 import com.sun.net.httpserver.*;
 31 
 32 import java.util.concurrent.*;
 33 import java.io.*;
 34 import java.net.*;
 35 
 36 public class B6433018 {
 37 
 38     static final String CRLF = "\r\n";
 39 
 40     /* invalid HTTP POST with extra CRLF at end */
 41     /* This checks that the server is able to handle it
 42      * and recognise the second request */
 43 
 44     static final String cmd =
 45         "POST /test/item HTTP/1.1"+CRLF+
 46         "Keep-Alive: 300"+CRLF+
 47         "Proxy-Connection: keep-alive"+CRLF+
 48         "Content-Type: text/xml"+CRLF+
 49         "Content-Length: 22"+CRLF+
 50         "Pragma: no-cache"+CRLF+
 51         "Cache-Control: no-cache"+CRLF+ CRLF+
 52         "<item desc=\"excuse\" />"+CRLF+
 53         "GET /test/items HTTP/1.1"+CRLF+
 54         "Host: araku:9999"+CRLF+
 55         "Accept-Language: en-us,en;q=0.5"+CRLF+
 56         "Accept-Encoding: gzip,deflate"+CRLF+
 57         "Accept-Charset: ISO-8859-1,utf-8;q=0.7,*;q=0.7"+CRLF+
 58         "Keep-Alive: 300"+CRLF+
 59         "Proxy-Connection: keep-alive"+CRLF+
 60         "Pragma: no-cache"+CRLF+
 61         "Cache-Control: no-cache"+CRLF+CRLF;
 62 
 63     public static void main(String[] args) throws Exception {
 64         CountDownLatch finished = new CountDownLatch(2);
 65         Handler handler = new Handler(finished);
 66         InetSocketAddress addr = new InetSocketAddress(0);

 67         HttpServer server = HttpServer.create(addr, 0);
 68         HttpContext ctx = server.createContext("/test", handler);
 69 
 70         server.start();
 71         int port = server.getAddress().getPort();
 72         try (Socket s = new Socket("localhost", port);
 73              OutputStream os = s.getOutputStream()) {
 74             os.write(cmd.getBytes());
 75             finished.await(30, TimeUnit.SECONDS);
 76         } finally {
 77             server.stop(2);
 78         }
 79 
 80         if (finished.getCount() != 0)
 81             throw new RuntimeException("did not receive the 2 requests");
 82 
 83         System.out.println("OK");
 84     }
 85 
 86     static class Handler implements HttpHandler {
 87         private final CountDownLatch finished;
 88 
 89         Handler(CountDownLatch finished) {
 90             this.finished = finished;
 91         }
 92 
 93         @Override
 94         public void handle(HttpExchange t) throws IOException {
 95             try (InputStream is = t.getRequestBody()) {
 96                 Headers map = t.getRequestHeaders();
 97                 Headers rmap = t.getResponseHeaders();
 98                 while (is.read() != -1);
 99             }
100             t.sendResponseHeaders(200, -1);
101             t.close();
102             finished.countDown();
103         }
104     }
105 }
