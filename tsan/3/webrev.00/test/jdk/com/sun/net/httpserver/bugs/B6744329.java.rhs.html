 33 import com.sun.net.httpserver.*;
 34 
 35 import java.util.*;
 36 import java.util.concurrent.*;
 37 import java.io.*;
 38 import java.net.*;
 39 import java.security.*;
 40 import java.security.cert.*;
 41 import javax.net.ssl.*;
 42 import jdk.test.lib.net.URIBuilder;
 43 
 44 public class B6744329 {
 45 
 46     public static void main (String[] args) throws Exception {
 47         Handler handler = new Handler();
 48         InetAddress loopback = InetAddress.getLoopbackAddress();
 49         InetSocketAddress addr = new InetSocketAddress (loopback, 0);
 50         HttpServer server = HttpServer.create (addr, 0);
 51         HttpContext ctx = server.createContext ("/test", handler);
 52         ExecutorService executor = Executors.newCachedThreadPool();
 53         server.setExecutor (executor);
 54         server.start ();
 55 
 56         URL url = URIBuilder.newBuilder()
 57             .scheme("http")
 58             .loopback()
 59             .port(server.getAddress().getPort())
 60             .path("/test/foo.html")
 61             .toURL();
 62         HttpURLConnection urlc = (HttpURLConnection)url.openConnection(Proxy.NO_PROXY);
 63         try {
 64             InputStream is = urlc.getInputStream();
 65             int c = 0;
 66             while (is.read()!= -1) {
 67                 c ++;
 68             }
 69             System.out.println ("OK");
 70         } catch (IOException e) {
 71             System.out.println ("exception");
 72             e.printStackTrace();
 73             error = true;
 74         }
 75         server.stop(2);
 76         executor.shutdown();
 77         if (error) {
 78             throw new RuntimeException ("Test failed");
 79         }
 80     }
 81 
 82     public static boolean error = false;
 83 
 84     /* this must be the same size as in ChunkedOutputStream.java
 85      */
 86     final static int CHUNK_SIZE = 4096;
 87 
 88     static class Handler implements HttpHandler {
 89         int invocation = 1;
 90         public void handle (HttpExchange t)
 91             throws IOException
 92         {
 93             InputStream is = t.getRequestBody();
 94             Headers map = t.getRequestHeaders();
 95             Headers rmap = t.getResponseHeaders();
 96             while (is.read () != -1) ;
 97             is.close();
 98             /* chunked response */
 99             t.sendResponseHeaders (200, 0);
100             OutputStream os = t.getResponseBody();
101             byte[] first = new byte [CHUNK_SIZE * 2];
102             byte[] second = new byte [2];
103             os.write (first);
104             os.write ('x');
105             os.write ('x');
106             /* An index out of bounds exception will be thrown
107              * below, which is caught by server, and connection
108              * will be closed. resulting in IOException to client
109              * - if bug present
110              */
111             os.write ('x');
112             os.write ('x');
113             os.write ('x');
114             t.close();
115         }
116     }
