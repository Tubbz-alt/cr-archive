 41 /**
 42  * Test https POST large file via chunked encoding (unusually small chunks)
 43  */
 44 
 45 public class Test6a extends Test {
 46 
 47     public static void main (String[] args) throws Exception {
 48         Handler handler = new Handler();
 49         InetSocketAddress addr = new InetSocketAddress (0);

 50         HttpsServer server = HttpsServer.create (addr, 0);
 51         HttpContext ctx = server.createContext ("/test", handler);
 52         ExecutorService executor = Executors.newCachedThreadPool();
 53         SSLContext ssl = new SimpleSSLContext().get();
 54         server.setExecutor (executor);
 55         server.setHttpsConfigurator(new HttpsConfigurator (ssl));
 56         server.start ();
 57 
 58         URL url = new URL ("https://localhost:"+server.getAddress().getPort()+"/test/foo.html");
 59         System.out.print ("Test6a: " );
 60         HttpsURLConnection urlc = (HttpsURLConnection)url.openConnection ();







 61         urlc.setDoOutput (true);
 62         urlc.setRequestMethod ("POST");
 63         urlc.setChunkedStreamingMode (32); // small chunks
 64         urlc.setSSLSocketFactory (ssl.getSocketFactory());
 65         urlc.setHostnameVerifier (new DummyVerifier());
 66         OutputStream os = new BufferedOutputStream (urlc.getOutputStream());
 67         for (int i=0; i<SIZE; i++) {
 68             os.write (i % 100);
 69         }
 70         os.close();
 71         int resp = urlc.getResponseCode();
 72         if (resp != 200) {
 73             throw new RuntimeException ("test failed response code");
 74         }
 75         if (error) {
 76             throw new RuntimeException ("test failed error");
 77         }
 78         delay();
 79         server.stop(2);
 80         executor.shutdown();

 74         urlc.setChunkedStreamingMode (32); // small chunks
 75         urlc.setSSLSocketFactory (ssl.getSocketFactory());
 76         urlc.setHostnameVerifier (new DummyVerifier());
