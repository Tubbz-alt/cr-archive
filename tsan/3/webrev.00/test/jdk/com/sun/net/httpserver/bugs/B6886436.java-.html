55         HttpURLConnection urlc = (HttpURLConnection)url.openConnection ();
56         try {
57             InputStream is = urlc.getInputStream();
58             while (is.read()!= -1) ;
59             is.close ();

61             urlc.setReadTimeout (3000);
62             is = urlc.getInputStream();
63             while (is.read()!= -1);
64             is.close ();
65 
66         } catch (IOException e) {
67             server.stop(2);
68             executor.shutdown();
69             throw new RuntimeException ("Test failed");
70         }
71         server.stop(2);
72         executor.shutdown();
73         System.out.println ("OK");
