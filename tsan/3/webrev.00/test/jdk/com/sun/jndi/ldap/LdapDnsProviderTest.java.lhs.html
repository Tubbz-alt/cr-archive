 53 class DNSSecurityManager extends SecurityManager {
 54 
 55 
 56 
 57     /* run main/othervm LdapDnsProviderTest
 58 
 59      * run main/othervm LdapDnsProviderTest nosm
 60      * run main/othervm LdapDnsProviderTest smnodns
 61      * run main/othervm LdapDnsProviderTest smdns
 62      * run main/othervm LdapDnsProviderTest nosmbaddns
 63      */
 64 
 65     private boolean dnsProvider = false;
 66 
 67     public void setAllowDnsProvider(boolean allow) {
 68         dnsProvider = allow;
 69     }
 70 
 71     @Override
 72     public void checkPermission(Permission p) {
 73         if (p.getName().equals("ldapDnsProvider") && !dnsProvider) {
 74             throw new SecurityException(p.getName());
 75         }
 76     }
 77 }

 79 class ProviderTest implements Callable<Boolean> {
 80 
 81     private final String url;
 82     private final String expected;
 83     private final Hashtable<String, String> env = new Hashtable<>(11);
 84 
 85     public ProviderTest(String url, String expected) {
 86         this.url = url;
 87         this.expected = expected;
 88         env.put(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
 89     }
 90 
 91     boolean shutItDown(InitialContext ctx) {
 92         try {
 93             if (ctx != null) ctx.close();
 94             return true;
 95         } catch (NamingException ex) {
 96             return false;
 97         }
 98     }
 99 
100     public Boolean call() {
101         boolean passed;
102         InitialContext ctx = null;
103 
104         if (url != null) {
105             env.put(Context.PROVIDER_URL, url);
106         }
107 
108         try {
109             ctx = new InitialDirContext(env);
110             SearchControls scl = new SearchControls();
111             scl.setSearchScope(SearchControls.SUBTREE_SCOPE);
112             ((InitialDirContext)ctx).search(
113                     "ou=People,o=Test", "(objectClass=*)", scl);
114             throw new RuntimeException("Search should not complete");
115         } catch (NamingException e) {
116             e.printStackTrace();
117             passed = e.toString().contains(expected);
118         } finally {
119             shutItDown(ctx);
120         }
121         return passed;
122     }
123 }

125 public class LdapDnsProviderTest {
126 
127     private static final String TEST_CLASSES =
128             System.getProperty("test.classes", ".");
129 
130     public static void writeFile(String content, File dstFile)
131         throws IOException
132     {
133         try (FileOutputStream dst = new FileOutputStream(dstFile)) {
134             byte[] buf = content.getBytes();
135             dst.write(buf, 0, buf.length);
136         }
137     }
138 
139     public static void installServiceConfigurationFile(String content) {
140         String filename = "javax.naming.ldap.spi.LdapDnsProvider";
141 
142         File dstDir = new File(TEST_CLASSES, "META-INF/services");
143         if (!dstDir.exists()) {
144             if (!dstDir.mkdirs()) {
145                 throw new RuntimeException(
146                     "could not create META-INF/services directory " + dstDir);
147             }
148         }
149         File dstFile = new File(dstDir, filename);
150 
151         try {
152             writeFile(content, dstFile);
153         } catch (IOException e) {
154             throw new RuntimeException("could not install " + dstFile, e);
155         }
156     }
157 
158     public static void main(String[] args) throws Exception {
159         if (args.length > 0 && args[0].equals("nosm")) {
160             // no security manager, serviceloader
161             installServiceConfigurationFile("dnsprovider.TestDnsProvider");
162             runTest("ldap:///dc=example,dc=com", "yupyupyup:389");
163         } else if (args.length > 0 && args[0].equals("smnodns")) {
164             // security manager & serviceloader
165             installServiceConfigurationFile("dnsprovider.TestDnsProvider");
166             // install security manager
167             System.setSecurityManager(new DNSSecurityManager());
168             runTest("ldap:///dc=example,dc=com", "ServiceConfigurationError");
169         } else if (args.length > 0 && args[0].equals("smdns")) {
170             // security manager & serviceloader
171             DNSSecurityManager sm = new DNSSecurityManager();
172             installServiceConfigurationFile("dnsprovider.TestDnsProvider");
173             // install security manager
174             System.setSecurityManager(sm);
175             sm.setAllowDnsProvider(true);
176             runTest("ldap:///dc=example,dc=com", "yupyupyup:389");
177         } else if (args.length > 0 && args[0].equals("nosmbaddns")) {
178             // no security manager, no serviceloader
179             // DefaultLdapDnsProvider
180             installServiceConfigurationFile("dnsprovider.MissingDnsProvider");
181             // no SecurityManager
182             runTest("ldap:///dc=example,dc=com", "not found");
183         } else {
184             // no security manager, no serviceloader
185             // DefaultLdapDnsProvider
186             System.err.println("TEST_CLASSES:");
187             System.err.println(TEST_CLASSES);
188             File f = new File(
189                     TEST_CLASSES, "META-INF/services/javax.naming.ldap.spi.LdapDnsProvider");
190             if (f.exists()) {
191                 f.delete();
192             }
193 
194             // no SecurityManager
195             runTest("ldap:///dc=example,dc=com", "localhost:389");
196             runTest("ldap://localhost/dc=example,dc=com", "localhost:389");
197             runTest("ldap://localhost:111/dc=example,dc=com", "localhost:111");
198             runTest("ldaps://localhost:111/dc=example,dc=com", "localhost:111");
199             runTest("ldaps://localhost/dc=example,dc=com", "localhost:636");
200             runTest(null, "localhost:389");
201             runTest("", "ConfigurationException");
202         }
203     }
204 
205     private static void runTest(String url, String expected) {
206         FutureTask<Boolean> future =
207             new FutureTask<>(
208                     new ProviderTest(url, expected));
209         new Thread(future).start();
210 
211         System.err.println("Testing: " + url + ", " + expected);
212         while (!future.isDone()) {
213             try {
214                 if (!future.get()) {
215                     System.err.println("Test failed");
216                     throw new RuntimeException(
217                             "Test failed, ProviderTest returned false");
218                 }
219             } catch (Exception e) {
220                 if (!e.toString().contains(expected)) {
221                     System.err.println("Test failed");
222                     throw new RuntimeException(
223                             "Test failed, unexpected result");
224                 }
225             }
226         }
227         System.err.println("Test passed");
228     }
229
