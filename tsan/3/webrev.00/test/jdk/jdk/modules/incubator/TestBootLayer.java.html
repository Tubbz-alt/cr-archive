24 import java.util.Set;
25 import java.util.function.Predicate;
26 import java.util.regex.Pattern;
27 import java.util.stream.Collectors;
28 
29 /**
30  * Launched by the test ServiceBinding to test modules in the boot layer.
31  */
32 
33 public class TestBootLayer {
34     public static void main(String[] args) throws Exception {
35         Pattern splitter = Pattern.compile(",");
36 
37         // the names of all modules in the boot layer
38         Set<String> modules = ModuleLayer.boot().modules().stream()
39                 .map(Module::getName)
40                 .collect(Collectors.toSet());
41 
42         // throw exception if an expected module is not in the boot layer
43         splitter.splitAsStream(args[0])
44                 .filter(Predicate.not(String::isEmpty))
45                 .filter(mn -> !modules.contains(mn))
46                 .findAny()
47                 .ifPresent(mn -> {
48                     throw new RuntimeException(mn + " not in boot layer!!!");
49                 });
50 
51         // throw exception if an unexpected module is in the boot layer
52         splitter.splitAsStream(args[1])
53                 .filter(Predicate.not(String::isEmpty))
54                 .filter(mn -> modules.contains(mn))
55                 .findAny()
56                 .ifPresent(mn -> {
57                     throw new RuntimeException(mn + " in boot layer!!!!");
58                 });
59     }
60 }
