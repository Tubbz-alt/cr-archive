25 import java.lang.module.ModuleDescriptor;
26 import java.lang.module.ModuleFinder;
27 import java.lang.module.ModuleReference;
28 
29 /**
30  * Test the set of modules in the boot layer includes all modules that export
31  * an API. Also test that java.se is not resolved.
32  */
33 
34 public class TestRootModules {
35     public static void main(String[] args) {
36         // all modules that export an API should be resolved
37         // For now, this test ignores the ModuleResolution attribute
38         ModuleLayer bootLayer = ModuleLayer.boot();
39         ModuleFinder.ofSystem().findAll().stream()
40             .map(ModuleReference::descriptor)
41             .filter(descriptor -> descriptor.exports()
42                     .stream()
43                     .filter(e -> !e.isQualified())
44                     .findAny()
45                     .isPresent())
46             .map(ModuleDescriptor::name)
47             .forEach(name -> {
48                 if (!bootLayer.findModule(name).isPresent())
49                     throw new RuntimeException(name + " not in boot layer");
50             });
51 
52         // java.se should not be resolved
53         ModuleLayer.boot()
54                 .findModule("java.se")
55                 .map(m -> { throw new RuntimeException("java.se should not be resolved"); });
56     }
57 }
