24  /*
25  * @test
26  * @bug 8145635
27  * @summary Add TCP_QUICKACK socket option
28  * @modules jdk.net
29  * @run main QuickAckTest
30  */
31 import java.io.IOException;
32 import java.net.DatagramSocket;
33 import java.net.InetAddress;
34 import java.net.MulticastSocket;
35 import java.net.ServerSocket;
36 import java.net.Socket;
37 import jdk.net.ExtendedSocketOptions;
38 import jdk.net.Sockets;
39 
40 public class QuickAckTest {
41 
42     public static void main(String args[]) throws IOException {
43 
44         try (ServerSocket ss = new ServerSocket(0);
45                 Socket s = new Socket(InetAddress.getLoopbackAddress(), ss.getLocalPort());
46                 DatagramSocket ds = new DatagramSocket(0);
47                 MulticastSocket mc = new MulticastSocket(0)) {
48 
49             if (ss.supportedOptions().contains(ExtendedSocketOptions.TCP_QUICKACK)) {
50                 Sockets.setOption(ss, ExtendedSocketOptions.TCP_QUICKACK, true);
51                 if (!ss.getOption(ExtendedSocketOptions.TCP_QUICKACK)) {
52                     throw new RuntimeException("Test failed, TCP_QUICKACK should"
53                             + " have been set");
54                 }
55             }
56             if (s.supportedOptions().contains(ExtendedSocketOptions.TCP_QUICKACK)) {
57                 Sockets.setOption(s, ExtendedSocketOptions.TCP_QUICKACK, true);
58                 if (!s.getOption(ExtendedSocketOptions.TCP_QUICKACK)) {
59                     throw new RuntimeException("Test failed, TCP_QUICKACK should"
60                             + " have been set");
61                 }
62             }
63             if (ds.supportedOptions().contains(ExtendedSocketOptions.TCP_QUICKACK)) {
64                 throw new RuntimeException("Test failed, TCP_QUICKACK is applicable"
65                         + " for TCP Sockets only.");
66             }
67             if (mc.supportedOptions().contains(ExtendedSocketOptions.TCP_QUICKACK)) {
68                 throw new RuntimeException("Test failed, TCP_QUICKACK is applicable"
69                         + " for TCP Sockets only");
70             }
71         }
72     }
73 }
