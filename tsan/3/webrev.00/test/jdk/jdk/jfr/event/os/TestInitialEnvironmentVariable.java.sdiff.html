42 /**

49 public class TestInitialEnvironmentVariable {
50     private final static String EVENT_NAME = EventNames.InitialEnvironmentVariable;
51 
52     public static void main(String[] args) throws Exception {
53         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(Test.class.getName());
54         setEnv(pb.environment());
55         (new OutputAnalyzer(pb.start())).shouldHaveExitValue(0);
56     }
57 
58     static void setEnv(Map<String, String> env) {
59         env.put("keytest1", "value1");
60         env.put("keytest2", "value 2");
61     }
62 
63     static class Test {
64         public static void main(String[] args) throws Exception {
65             Map<String, String> env = new HashMap<>();
66             setEnv(env);
67 
68             Recording recording = new Recording();
69             recording.enable(EVENT_NAME);
70             recording.start();
71             recording.stop();
72             List<RecordedEvent> events = Events.fromRecording(recording);
73             Events.hasEvents(events);
74             for (RecordedEvent event : events) {
75                 System.out.println("Event: " + event);
76                 String key = Events.assertField(event, "key").notNull().getValue();
77                 String value = Events.assertField(event, "value").notNull().getValue();
78                 if (env.containsKey(key)) {
79                     assertEquals(value, env.get(key), "Wrong value for key: " + key);
80                     env.remove(key);
81                 }
82             }
83             assertTrue(env.isEmpty(), "Missing env keys " + env.keySet());
84         }

85     }
