26 package jdk.jfr.event.gc.configuration;
27 
28 import jdk.jfr.consumer.RecordedEvent;
29 import jdk.test.lib.jfr.EventVerifier;
30 
31 /* See the shell script wrapper for the flags used when invoking the JVM */
32 public class TestGCHeapConfigurationEventWithHeapBasedOops extends GCHeapConfigurationEventTester {
33     public static void main(String[] args) throws Exception {
34         GCHeapConfigurationEventTester t = new TestGCHeapConfigurationEventWithHeapBasedOops();
35         t.run();
36     }
37 
38     @Override
39     protected EventVerifier createVerifier(RecordedEvent e) {
40         return new HeapBasedOopsVerifier(e);
41     }
42 }
43 
44 class HeapBasedOopsVerifier extends GCHeapConfigurationEventVerifier {
45     public HeapBasedOopsVerifier(RecordedEvent e) {
46         super(e);
47     }
48 
49     @Override
50     public void verify() throws Exception {
51         // Can't verify min and initial heap size due to constraints on
52         // physical memory on tests machines
53         verifyMaxHeapSizeIs(gigabytes(31));
54         verifyUsesCompressedOopsIs(true);
55         verifyObjectAlignmentInBytesIs(8);
56         verifyHeapAddressBitsIs(32);
57         verifyCompressedOopModeContains("Non-zero");
58     }
59 }
