26 package jdk.jfr.event.runtime;
27 
28 import java.time.Duration;
29 import java.util.List;
30 
31 import jdk.jfr.Recording;
32 import jdk.jfr.consumer.RecordedClass;
33 import jdk.jfr.consumer.RecordedClassLoader;
34 import jdk.jfr.consumer.RecordedEvent;
35 import jdk.test.lib.Asserts;
36 import jdk.test.lib.jfr.EventNames;
37 import jdk.test.lib.jfr.Events;
38 import jdk.test.lib.jfr.TestClassLoader;
39 
40 /**
41  * @test
42  * @summary The test verifies that a class unload event is created when class is unloaded
43  * @key jfr
44  * @requires vm.hasJFR
45  * @library /test/lib /test/jdk
46  * @build jdk.jfr.event.runtime.TestClasses
47  * @run main/othervm -Xlog:class+unload -Xlog:gc -Xmx16m jdk.jfr.event.runtime.TestClassUnloadEvent
48  */
49 
50 /**
51  * System.gc() will trigger class unloading if -XX:+ExplicitGCInvokesConcurrent is NOT set.
52  * If this flag is set G1 will never unload classes on System.gc() and
53  * CMS will not guarantee that all semantically dead classes will be unloaded.
54  * As far as the "jfr" key guarantees no VM flags are set from the outside
55  * it should be enough with System.gc().
56  */
57 public final class TestClassUnloadEvent {
58     private final static String TEST_CLASS_NAME = "jdk.jfr.event.runtime.TestClasses";
59     private final static String EVENT_PATH = EventNames.ClassUnload;
60 
61     // Declare unloadableClassLoader as "public static"
62     // to prevent the compiler to optimize away all unread writes
63     public static TestClassLoader unloadableClassLoader;
64 
65     public static void main(String[] args) throws Throwable {
66         Recording recording = new Recording();
67         recording.enable(EVENT_PATH).withThreshold(Duration.ofMillis(0));
68         unloadableClassLoader = new TestClassLoader();
69         recording.start();
70         unloadableClassLoader.loadClass(TEST_CLASS_NAME);
71         unloadableClassLoader = null;
72         System.gc();
73         recording.stop();
74 
75         List<RecordedEvent> events = Events.fromRecording(recording);
76         Events.hasEvents(events);
77         boolean isAnyFound = false;
78         for (RecordedEvent event : events) {
79             System.out.println("Event:" + event);
80             RecordedClass unloadedClass = event.getValue("unloadedClass");
81             if (TEST_CLASS_NAME.equals(unloadedClass.getName())) {
82                 RecordedClassLoader definingClassLoader = unloadedClass.getClassLoader();
83                 Asserts.assertEquals(TestClassLoader.class.getName(), definingClassLoader.getType().getName(),
84                     "Expected " + TestClassLoader.class.getName() + ", got " + definingClassLoader.getType().getName());
85                 Asserts.assertEquals(TestClassLoader.CLASS_LOADER_NAME, definingClassLoader.getName(),
86                     "Expected " + TestClassLoader.CLASS_LOADER_NAME + ", got " + definingClassLoader.getName());
87                 Asserts.assertFalse(isAnyFound, "Found more than 1 event");
88                 isAnyFound = true;
89             }
90         }
91         Asserts.assertTrue(isAnyFound, "No events found");
92     }
93 }
