24 # @key jfr
25 # @requires vm.hasJFR
26 # @requires vm.gc == "Parallel" | vm.gc == null
27 # @library /test/lib /test/jdk
28 # @build jdk.jfr.event.gc.configuration.TestGCHeapConfigurationEventWithZeroBasedOops

31 uses_64_bit_testjava() {
32   ${TESTJAVA}/bin/java ${TESTVMOPTS} -version 2>&1 | grep '64-Bit' > /dev/null
33 }
34 
35 uses_windows_or_linux() {
36     case `uname -s` in
37       Linux | CYGWIN* | Windows* )
38         return 0
39         ;;
40     esac
41     return 1
42 }
43 
44 TEST='jdk.jfr.event.gc.configuration.TestGCHeapConfigurationEventWithZeroBasedOops'
45 
46 OPTIONS='-XX:+UnlockExperimentalVMOptions -XX:-UseFastUnorderedTimeStamps -XX:+UseParallelGC -XX:+UseParallelOldGC -XX:+UseCompressedOops -Xmx4g'
47 
48 if [ -z "${TESTCLASSPATH}" ]; then
49     echo "Using TESTCLASSES"
50     MY_CLASSPATH=${TESTCLASSES}
51 else
52     echo "Using TESTCLASSPATH"
53     MY_CLASSPATH=${TESTCLASSPATH}
54 fi
55 
56 if uses_windows_or_linux && uses_64_bit_testjava; then
57   printenv
58   echo "${TESTJAVA}/bin/java ${TESTVMOPTS} ${OPTIONS} -cp ${MY_CLASSPATH} ${TEST}"
59   ${TESTJAVA}/bin/java ${TESTVMOPTS} ${OPTIONS} -cp ${MY_CLASSPATH} ${TEST}
60 fi
