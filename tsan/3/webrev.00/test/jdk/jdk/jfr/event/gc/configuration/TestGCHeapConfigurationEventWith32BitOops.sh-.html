24 # @key jfr
25 # @requires vm.hasJFR
26 # @requires vm.gc == "Parallel" | vm.gc == null
27 # @library /test/lib /test/jdk
28 # @build jdk.jfr.event.gc.configuration.TestGCHeapConfigurationEventWith32BitOops sun.hotspot.WhiteBox
29 # @run main ClassFileInstaller sun.hotspot.WhiteBox
30 # @run shell TestGCHeapConfigurationEventWith32BitOops.sh
31 
32 uses_64_bit_testjava() {
33   ${TESTJAVA}/bin/java ${TESTVMOPTS} -version 2>&1 | grep '64-Bit' > /dev/null
34 }
35 
36 uses_windows_or_linux() {
37     case `uname -s` in
38       Linux | CYGWIN* | Windows* )
39         return 0
40         ;;
41     esac
42     return 1
43 }
44 
45 TEST='jdk.jfr.event.gc.configuration.TestGCHeapConfigurationEventWith32BitOops'
46 
47 OPTIONS='-XX:+UnlockExperimentalVMOptions -XX:-UseFastUnorderedTimeStamps -XX:+UseParallelGC -XX:+UseParallelOldGC -XX:+UseCompressedOops -Xmx100m -Xms100m -XX:InitialHeapSize=100m -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI'
48 
49 if [ -z "${TESTCLASSPATH}" ]; then
50     echo "Using TESTCLASSES"
51     MY_CLASSPATH=${TESTCLASSES}
52 else
53     echo "Using TESTCLASSPATH"
54     MY_CLASSPATH=${TESTCLASSPATH}
55 fi
56 
57 if uses_windows_or_linux && uses_64_bit_testjava; then
58   printenv
59   echo "${TESTJAVA}/bin/java ${TESTVMOPTS} ${OPTIONS} -cp ${MY_CLASSPATH} ${TEST}"
60   ${TESTJAVA}/bin/java ${TESTVMOPTS} ${OPTIONS} -cp ${MY_CLASSPATH} ${TEST}
61 fi
