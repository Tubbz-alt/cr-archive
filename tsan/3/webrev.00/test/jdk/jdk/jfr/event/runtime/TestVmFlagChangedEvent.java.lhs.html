 26 package jdk.jfr.event.runtime;
 27 
 28 import static jdk.test.lib.Asserts.assertEquals;
 29 import static jdk.test.lib.Asserts.assertTrue;
 30 
 31 import java.lang.management.ManagementFactory;
 32 
 33 import com.sun.management.HotSpotDiagnosticMXBean;
 34 
 35 import jdk.jfr.Recording;
 36 import jdk.jfr.consumer.RecordedEvent;
 37 import jdk.test.lib.jfr.EventNames;
 38 import jdk.test.lib.jfr.Events;
 39 
 40 /**
 41  * @test TestVmFlagChangedEvent
 42  * @key jfr
 43  * @requires vm.hasJFR
 44  * @library /test/lib
 45  * @modules jdk.jfr
 46  *          jdk.management
 47  *
 48  * @run main/othervm jdk.jfr.event.runtime.TestVmFlagChangedEvent
 49  */
 50 public final class TestVmFlagChangedEvent {
 51 
 52     public static void main(String[] args) throws Throwable {
 53         EventFlag[] eventFlags = {
 54             new EventFlag(EventNames.LongFlagChanged, "CMSWaitDuration", "2500"),
 55             new EventFlag(EventNames.StringFlagChanged, "HeapDumpPath", "/a/sample/path"),
 56             new EventFlag(EventNames.BooleanFlagChanged, "HeapDumpOnOutOfMemoryError", "true")
 57         };
 58 
 59         Recording recording = new Recording();
 60         for (EventFlag eventFlag : eventFlags) {
 61             recording.enable(eventFlag.eventName);
 62         }
 63 
 64         recording.start();
 65         HotSpotDiagnosticMXBean mbean = ManagementFactory.getPlatformMXBean(HotSpotDiagnosticMXBean.class);
 66         for (EventFlag eventFlag : eventFlags) {
 67             eventFlag.update(mbean);
 68         }
 69         recording.stop();
 70 
 71         for (RecordedEvent event : Events.fromRecording(recording)) {
 72             final String flagName = Events.assertField(event, "name").getValue();
 73             System.out.println("flag name=" + flagName);
 74             for (EventFlag eventFlag : eventFlags) {
 75                 if (flagName.equals(eventFlag.eventLabel)) {
 76                     System.out.println("Event:" + event);
 77                     Object newValue = Events.assertField(event, "newValue").getValue();
 78                     Object oldValue = Events.assertField(event, "oldValue").getValue();
 79                     System.out.println("newValue:" + asText(newValue));
 80                     System.out.println("oldValue:" + asText(oldValue));
 81                     assertEquals(eventFlag.newValue, asText(newValue), "Wrong new value: expected" + eventFlag.newValue);
 82                     assertEquals(eventFlag.oldValue, asText(oldValue), "Wrong old value: expected" + eventFlag.oldValue);
 83                     Events.assertField(event, "origin").equal("Management");
 84                     eventFlag.isFound = true;
 85                     break;
 86                 }
 87             }
 88         }
 89         for (EventFlag eventFlag : eventFlags) {
 90             assertTrue(eventFlag.isFound, "Missing flag change for: " + eventFlag.eventLabel);
 91         }
 92     }
 93 
 94     private static String asText(Object value) {
 95         if (value == null) {
 96             return ""; // HotSpotDiagnosticMXBean interface return "" for unset values
 97         }
 98         return String.valueOf(value);
 99     }
100 
101     private static class EventFlag {
102         final String eventName;
103         final String eventLabel;
104         String newValue;
105         String oldValue;
106         boolean isFound = false;
107 
108         EventFlag(String eventName, String eventLabel, String newValue) {
109             this.eventName = eventName;
110             this.eventLabel = eventLabel;
111             this.newValue = newValue;
112         }
113 
114         void update(HotSpotDiagnosticMXBean mbean) {
115             this.oldValue = mbean.getVMOption(eventLabel).getValue();
116             mbean.setVMOption(eventLabel, newValue);
117         }
118     }
119 }
