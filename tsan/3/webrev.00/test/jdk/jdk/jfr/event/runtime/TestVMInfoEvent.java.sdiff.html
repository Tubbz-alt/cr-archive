30 import java.lang.management.ManagementFactory;
31 import java.lang.management.RuntimeMXBean;

33 import java.util.stream.Collectors;

35 import jdk.jfr.Recording;

41 /**
42  * The test will verify that JVM Information event values are delivered
43  * and compare them with the RuntimeMXBean's values.




44  */
45 public class TestVMInfoEvent {
46     private final static String EVENT_NAME = EventNames.JVMInformation;
47 
48     public static void main(String[] args) throws Exception {




49         RuntimeMXBean mbean = ManagementFactory.getRuntimeMXBean();
50         Recording recording = new Recording();
51         recording.enable(EVENT_NAME);
52         recording.start();
53         recording.stop();
54 
55         List<RecordedEvent> events = Events.fromRecording(recording);
56         Events.hasEvents(events);
57         for (RecordedEvent event : events) {
58             System.out.println("Event:" + event);
59             Events.assertField(event, "jvmName").equal(mbean.getVmName());
60             String jvmVersion = Events.assertField(event, "jvmVersion").notEmpty().getValue();
61             if (!jvmVersion.contains(mbean.getVmVersion())) {
62                 Asserts.fail(String.format("%s does not contain %s", jvmVersion, mbean.getVmVersion()));
63             }
64 
65             String jvmArgs = Events.assertField(event, "jvmArguments").notNull().getValue();
66             String jvmFlags = Events.assertField(event, "jvmFlags").notNull().getValue();
67             Long pid = Events.assertField(event, "pid").atLeast(0L).getValue();
68             Asserts.assertEquals(pid, ProcessHandle.current().pid());
69             String eventArgs = (jvmFlags.trim() + " " + jvmArgs).trim();
70             String beanArgs = mbean.getInputArguments().stream().collect(Collectors.joining(" "));
71             Asserts.assertEquals(eventArgs, beanArgs, "Wrong inputArgs");
72 
73             final String javaCommand = mbean.getSystemProperties().get("sun.java.command");
74             Events.assertField(event, "javaArguments").equal(javaCommand);
75             Events.assertField(event, "jvmStartTime").equal(mbean.getStartTime());
76         }
77     }
78 



79 }
