32  * @requires vm.hasJFR
33  * @requires (vm.gc == "ConcMarkSweep" | vm.gc == null) & !vm.graal.enabled
34  * @requires vm.opt.ExplicitGCInvokesConcurrent != false
35  * @library /test/lib /test/jdk
36  * @run driver jdk.jfr.event.gc.collection.TestGCCauseWithCMSConcurrent
37  */
38 public class TestGCCauseWithCMSConcurrent {
39     public static void main(String[] args) throws Exception {
40         String testID = "CMSConcurrent";
41         String[] vmFlags = {"-XX:+UseConcMarkSweepGC", "-XX:+ExplicitGCInvokesConcurrent"};
42         String[] gcNames = {GCHelper.gcConcurrentMarkSweep, GCHelper.gcParNew, GCHelper.gcSerialOld};
43         String[] gcCauses = {"CMS Concurrent Mark", "Allocation Failure", "System.gc()"};
44         GCGarbageCollectionUtil.test(testID, vmFlags, gcNames, gcCauses);
45     }
46 }
