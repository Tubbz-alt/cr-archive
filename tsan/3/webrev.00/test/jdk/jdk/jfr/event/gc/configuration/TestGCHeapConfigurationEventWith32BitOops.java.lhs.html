26 package jdk.jfr.event.gc.configuration;
27 
28 import jdk.jfr.consumer.RecordedEvent;
29 import jdk.test.lib.jfr.EventVerifier;
30 import jdk.test.lib.jfr.GCHelper;
31 import sun.hotspot.WhiteBox;
32 














33 /* See the shell script wrapper for the flags used when invoking the JVM */
34 public class TestGCHeapConfigurationEventWith32BitOops extends GCHeapConfigurationEventTester {
35     public static void main(String[] args) throws Exception {
36         GCHeapConfigurationEventTester t = new TestGCHeapConfigurationEventWith32BitOops();
37         t.run();
38     }
39 
40     @Override
41     protected EventVerifier createVerifier(RecordedEvent e) {
42         return new ThirtyTwoBitsVerifier(e);
43     }
44 }
45 
46 class ThirtyTwoBitsVerifier extends GCHeapConfigurationEventVerifier {
47     public ThirtyTwoBitsVerifier(RecordedEvent event) {
48         super(event);
49     }
50 
51     @Override
52     public void verify() throws Exception {
53         WhiteBox wb = WhiteBox.getWhiteBox();
54         long heapAlignment = wb.getHeapAlignment();
55         long alignedHeapSize = GCHelper.alignUp(megabytes(100), heapAlignment);
56         verifyMinHeapSizeIs(megabytes(100));
57         verifyInitialHeapSizeIs(alignedHeapSize);
58         verifyMaxHeapSizeIs(alignedHeapSize);
59         verifyUsesCompressedOopsIs(true);
60         verifyObjectAlignmentInBytesIs(8);
61         verifyHeapAddressBitsIs(32);
62         verifyCompressedOopModeIs("32-bit");
63     }
64 }
