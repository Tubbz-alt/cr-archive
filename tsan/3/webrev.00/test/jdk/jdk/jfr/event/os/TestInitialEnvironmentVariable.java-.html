26 package jdk.jfr.event.os;
27 
28 import static jdk.test.lib.Asserts.assertEquals;
29 import static jdk.test.lib.Asserts.assertTrue;
30 
31 import java.util.HashMap;
32 import java.util.List;
33 import java.util.Map;
34 
35 import jdk.jfr.Recording;
36 import jdk.jfr.consumer.RecordedEvent;
37 import jdk.test.lib.jfr.EventNames;
38 import jdk.test.lib.jfr.Events;
39 
40 public class TestInitialEnvironmentVariable {
41     private final static String EVENT_NAME = EventNames.InitialEnvironmentVariable;
42 
43     public static void main(String[] args) throws Exception {
44         Map<String, String> env = new HashMap<>();
45         env.put("keytest1", "value1");
46         env.put("keytest2", "value 2");
47 
48         Recording recording = new Recording();
49         recording.enable(EVENT_NAME);
50         recording.start();
51         recording.stop();
52         List<RecordedEvent> events = Events.fromRecording(recording);
53         Events.hasEvents(events);
54         for (RecordedEvent event : events) {
55             System.out.println("Event: " + event);
56             String key = Events.assertField(event, "key").notNull().getValue();
57             String value = Events.assertField(event, "value").notNull().getValue();
58             if (env.containsKey(key)) {
59                 assertEquals(value, env.get(key), "Wrong value for key: " + key);
60                 env.remove(key);
61             }
62         }
63         assertTrue(env.isEmpty(), "Missing env keys " + env.keySet());
64     }
65 }
