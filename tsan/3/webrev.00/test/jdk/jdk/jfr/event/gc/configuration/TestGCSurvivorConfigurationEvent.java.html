26 package jdk.jfr.event.gc.configuration;
27 
28 import static jdk.test.lib.Asserts.assertGreaterThanOrEqual;
29 
30 import java.util.List;
31 
32 import jdk.jfr.Recording;
33 import jdk.jfr.consumer.RecordedEvent;
34 import jdk.test.lib.jfr.EventNames;
35 import jdk.test.lib.jfr.EventVerifier;
36 import jdk.test.lib.jfr.Events;
37 
38 /**
39  * @test
40  * @key jfr
41  * @requires vm.hasJFR
42  * @requires vm.gc == "Parallel" | vm.gc == null
43  * @library /test/lib /test/jdk
44  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:-UseFastUnorderedTimeStamps -XX:+UseParallelGC -XX:MaxTenuringThreshold=13 -XX:InitialTenuringThreshold=9 jdk.jfr.event.gc.configuration.TestGCSurvivorConfigurationEvent
45  */
46 public class TestGCSurvivorConfigurationEvent {
47     public static void main(String[] args) throws Exception {
48         Recording recording = new Recording();
49         recording.enable(EventNames.GCSurvivorConfiguration);
50         recording.start();
51         recording.stop();
52         List<RecordedEvent> events = Events.fromRecording(recording);
53         assertGreaterThanOrEqual(events.size(), 1, "Expected at least 1 event");
54         GCSurvivorConfigurationEventVerifier verifier = new GCSurvivorConfigurationEventVerifier(events.get(0));
55         verifier.verify();
56     }
57 }
58 
59 class GCSurvivorConfigurationEventVerifier extends EventVerifier {
60     public GCSurvivorConfigurationEventVerifier(RecordedEvent event) {
61         super(event);
62     }
63 
64     public void verify() throws Exception {
65         verifyMaxTenuringThresholdIs(13);
66         verifyInitialTenuringThresholdIs(9);
67     }
68 
69     private void verifyMaxTenuringThresholdIs(int expected) throws Exception {
70         verifyEquals("maxTenuringThreshold", (byte)expected);
71     }
72 
73     private void verifyInitialTenuringThresholdIs(int expected) throws Exception {
74         verifyEquals("initialTenuringThreshold", (byte)expected);
75     }
76 }
