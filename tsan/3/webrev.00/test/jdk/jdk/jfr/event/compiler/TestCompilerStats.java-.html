26 package jdk.jfr.event.compiler;
27 
28 import java.util.List;
29 
30 import jdk.jfr.Recording;
31 import jdk.jfr.consumer.RecordedEvent;
32 import jdk.test.lib.jfr.EventNames;
33 import jdk.test.lib.jfr.Events;
34 
35 /**
36  * @test
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  * @run main/othervm jdk.jfr.event.compiler.TestCompilerStats
41  */
42 public class TestCompilerStats {
43     private final static String EVENT_NAME = EventNames.CompilerStatistics;
44 
45     public static void main(String[] args) throws Exception {
46         Recording recording = new Recording();
47         recording.enable(EVENT_NAME);
48         recording.start();
49         recording.stop();
50 
51         List<RecordedEvent> events = Events.fromRecording(recording);
52         Events.hasEvents(events);
53         for (RecordedEvent event : events) {
54             System.out.println("Event:" + event);
55             Events.assertField(event, "compileCount").atLeast(0);
56             Events.assertField(event, "bailoutCount").atLeast(0);
57             Events.assertField(event, "invalidatedCount").atLeast(0);
58             Events.assertField(event, "osrCompileCount").atLeast(0);
59             Events.assertField(event, "standardCompileCount").atLeast(0);
60             Events.assertField(event, "osrBytesCompiled").atLeast(0L);
61             Events.assertField(event, "standardBytesCompiled").atLeast(0L);
62             Events.assertField(event, "nmetodsSize").atLeast(0L);
63             Events.assertField(event, "nmetodCodeSize").atLeast(0L);
64             Events.assertField(event, "peakTimeSpent").atLeast(0L);
65             Events.assertField(event, "totalTimeSpent").atLeast(0L);
66         }
67     }
68 }
