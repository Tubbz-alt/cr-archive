24 package jdk.jfr.event.security;
25 
26 import java.security.Security;
27 import java.util.List;
28 
29 import jdk.jfr.Recording;
30 import jdk.jfr.consumer.RecordedEvent;
31 import jdk.test.lib.Asserts;
32 import jdk.test.lib.jfr.EventNames;
33 import jdk.test.lib.jfr.Events;
34 import jdk.test.lib.security.JDKSecurityProperties;
35 
36 /*
37  * @test
38  * @bug 8148188
39  * @summary Enhance the security libraries to record events of interest
40  * @key jfr
41  * @requires vm.hasJFR
42  * @library /test/lib
43  * @run main/othervm jdk.jfr.event.security.TestSecurityPropertyModificationEvent
44  */
45 public class TestSecurityPropertyModificationEvent {
46 
47     static List<String> keys = JDKSecurityProperties.getKeys();
48     static String keyValue = "shouldBecomeAnEvent";
49 
50     public static void main(String[] args) throws Exception {









51         try (Recording recording = new Recording()) {
52             recording.enable(EventNames.SecurityProperty);
53             recording.start();
54             for (String key: keys) {
55                 Security.setProperty(key, keyValue);
56             }
57             recording.stop();
58 
59             List<RecordedEvent> events = Events.fromRecording(recording);
60             Asserts.assertEquals(events.size(), keys.size(),
61                     "Incorrect number of events");
62             assertEvent(events);
63         }
64     }
65 
66     private static void assertEvent(List<RecordedEvent> events) throws Exception {
67         int i = 1;
68         for (RecordedEvent e : events) {
69             if (keys.contains(e.getString("key"))) {
70                 Events.assertField(e, "value").equal(keyValue);
71                 i++;
72             } else {
73                 System.out.println(events);
74                 throw new Exception("Unexpected event at index:" + i);
75             }
76         }
77     }
78 }
