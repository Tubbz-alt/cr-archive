26 package jdk.jfr.event.gc.configuration;
27 
28 import static jdk.test.lib.Asserts.assertGreaterThanOrEqual;
29 
30 import java.util.List;
31 
32 import jdk.jfr.Recording;
33 import jdk.jfr.consumer.RecordedEvent;
34 import jdk.test.lib.jfr.EventNames;
35 import jdk.test.lib.jfr.EventVerifier;
36 import jdk.test.lib.jfr.Events;
37 
38 /**
39  * @test
40  * @key jfr
41  * @requires vm.hasJFR
42  * @requires vm.gc == "Parallel" | vm.gc == null
43  * @library /test/lib /test/jdk
44  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:-UseFastUnorderedTimeStamps -XX:+UseParallelGC -XX:+UseParallelOldGC -XX:+UseTLAB -XX:MinTLABSize=3k -XX:TLABRefillWasteFraction=96 jdk.jfr.event.gc.configuration.TestGCTLABConfigurationEvent
45  */
46 public class TestGCTLABConfigurationEvent {
47     public static void main(String[] args) throws Exception {
48         Recording recording = new Recording();
49         recording.enable(EventNames.GCTLABConfiguration);
50         recording.start();
51         recording.stop();
52         List<RecordedEvent> events = Events.fromRecording(recording);
53         assertGreaterThanOrEqual(events.size(), 1, "Expected at least 1 event");
54         GCTLABConfigurationEventVerifier verifier = new GCTLABConfigurationEventVerifier(events.get(0));
55         verifier.verify();
56     }
57 }
58 
59 class GCTLABConfigurationEventVerifier extends EventVerifier {
60     public GCTLABConfigurationEventVerifier(RecordedEvent event) {
61         super(event);
62     }
63 
64     @Override
65     public void verify() throws Exception {
66         verifyUsesTLABsIs(true);
67         verifyMinTLABSizeIs(kilobyte(3));
68         verifyTLABRefillWasteLimitIs(96);
69     }
70 
71     void verifyUsesTLABsIs(boolean expected) throws Exception {
72         verifyEquals("usesTLABs", expected);
73     }
74 
75     void verifyMinTLABSizeIs(long expected) throws Exception {
76         verifyEquals("minTLABSize", expected);
77     }
78 
79     void verifyTLABRefillWasteLimitIs(long expected) throws Exception {
80         verifyEquals("tlabRefillWasteLimit", expected);
81     }
82 
83     private int kilobyte(int num) {
84         return 1024 * num;
85     }
86 }
