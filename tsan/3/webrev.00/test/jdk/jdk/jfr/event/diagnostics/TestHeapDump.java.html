26 package jdk.jfr.event.diagnostics;
27 
28 import java.lang.management.ManagementFactory;
29 import java.nio.file.Files;
30 import java.nio.file.Path;
31 import java.nio.file.Paths;
32 import java.util.List;
33 
34 import javax.management.MBeanServer;
35 import javax.management.ObjectName;
36 
37 import jdk.jfr.Recording;
38 import jdk.jfr.consumer.RecordedEvent;
39 import jdk.test.lib.jfr.EventNames;
40 import jdk.test.lib.jfr.Events;
41 
42 /**
43  * @test
44  * @key jfr
45  * @requires vm.hasJFR
46  * @library /test/lib
47  * @modules java.management
48  * @run main/othervm jdk.jfr.event.diagnostics.TestHeapDump
49  */
50 public class TestHeapDump {
51     private final static String EVENT_NAME = EventNames.HeapDump;
52 
53     public static void main(String[] args) throws Exception {
54 
55         Path path = Paths.get("dump.hprof").toAbsolutePath();
56         try (Recording r = new Recording()) {
57             r.enable(EVENT_NAME);
58             r.start();
59             heapDump(path);
60             r.stop();
61             List<RecordedEvent> events = Events.fromRecording(r);
62             if (events.size() != 1) {
63                 throw new Exception("Expected one event, got " + events.size());
64             }
65             RecordedEvent e = events.get(0);
66             Events.assertField(e, "destination").equal(path.toString());
67             Events.assertField(e, "gcBeforeDump").equal(true);
68             Events.assertField(e, "onOutOfMemoryError").equals(false);
69             Events.assertField(e, "size").equals(Files.size(path));
70             System.out.println(e);
71         }
72     }
73 
74     private static void heapDump(Path path) throws Exception {
75         ObjectName objectName = new ObjectName("com.sun.management:type=HotSpotDiagnostic");
76         MBeanServer mbeanServer = ManagementFactory.getPlatformMBeanServer();
77         Object[] parameters = new Object[2];
78         parameters[0] = path.toString();
79         parameters[1] = true;
80         String[] signature = new String[] { String.class.getName(), boolean.class.toString() };
81         mbeanServer.invoke(objectName, "dumpHeap", parameters, signature);
82     }
83 }
