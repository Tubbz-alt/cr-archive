26 package jdk.jfr.event.runtime;
27 
28 import java.time.Duration;
29 import java.util.List;
30 
31 import jdk.jfr.Recording;
32 import jdk.jfr.consumer.RecordedClass;
33 import jdk.jfr.consumer.RecordedClassLoader;
34 import jdk.jfr.consumer.RecordedEvent;
35 import jdk.test.lib.Asserts;
36 import jdk.test.lib.jfr.EventNames;
37 import jdk.test.lib.jfr.Events;
38 import jdk.test.lib.jfr.TestClassLoader;
39 
40 /**
41  * @test
42  * @summary The test verifies that a class unload event is created when class is unloaded
43  * @key jfr
44  * @requires vm.hasJFR
45  * @library /test/lib /test/jdk
46  * @build jdk.jfr.event.runtime.TestClasses
47  * @run main/othervm -Xlog:class+unload -Xlog:gc -Xmx16m jdk.jfr.event.runtime.TestClassUnloadEvent
48  */
49 
50 public final class TestClassUnloadEvent {
51     private final static String TEST_CLASS_NAME = "jdk.jfr.event.runtime.TestClasses";
52     private final static String EVENT_PATH = EventNames.ClassUnload;
53 
54     // Declare unloadableClassLoader as "public static"
55     // to prevent the compiler to optimize away all unread writes
56     public static TestClassLoader unloadableClassLoader;
57 
58     public static void main(String[] args) throws Throwable {
59         Recording recording = new Recording();
60         recording.enable(EVENT_PATH).withThreshold(Duration.ofMillis(0));
61         unloadableClassLoader = new TestClassLoader();
62         recording.start();
63         unloadableClassLoader.loadClass(TEST_CLASS_NAME);
64         unloadableClassLoader = null;
65         System.gc();
66         recording.stop();
67 
68         List<RecordedEvent> events = Events.fromRecording(recording);
69         Events.hasEvents(events);
70         boolean isAnyFound = false;
71         for (RecordedEvent event : events) {
72             System.out.println("Event:" + event);
73             RecordedClass unloadedClass = event.getValue("unloadedClass");
74             if (TEST_CLASS_NAME.equals(unloadedClass.getName())) {
75                 RecordedClassLoader definingClassLoader = unloadedClass.getClassLoader();
76                 Asserts.assertEquals(TestClassLoader.class.getName(), definingClassLoader.getType().getName(),
77                     "Expected " + TestClassLoader.class.getName() + ", got " + definingClassLoader.getType().getName());
78                 Asserts.assertEquals(TestClassLoader.CLASS_LOADER_NAME, definingClassLoader.getName(),
79                     "Expected " + TestClassLoader.CLASS_LOADER_NAME + ", got " + definingClassLoader.getName());
80                 Asserts.assertFalse(isAnyFound, "Found more than 1 event");
81                 isAnyFound = true;
82             }
83         }
84         Asserts.assertTrue(isAnyFound, "No events found");
85     }
86 }
