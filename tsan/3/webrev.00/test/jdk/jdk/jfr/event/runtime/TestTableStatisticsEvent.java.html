26 package jdk.jfr.event.runtime;
27 
28 import java.util.List;
29 import java.util.stream.Collectors;
30 
31 import jdk.jfr.Recording;
32 import jdk.jfr.consumer.RecordedClass;
33 import jdk.jfr.consumer.RecordedClassLoader;
34 import jdk.jfr.consumer.RecordedEvent;
35 import jdk.test.lib.Asserts;
36 import jdk.test.lib.jfr.EventNames;
37 import jdk.test.lib.jfr.Events;
38 
39 /**
40  * @test
41  * @key jfr
42  * @requires vm.hasJFR
43  * @library /test/lib /test/jdk
44  * @build jdk.jfr.event.runtime.TestClasses
45  * @run main/othervm jdk.jfr.event.runtime.TestTableStatisticsEvent
46  * @bug 8185525
47  */
48 public final class TestTableStatisticsEvent {
49 
50   public static void main(String[] args) throws Throwable {
51     try (Recording recording = new Recording()) {
52       recording.enable(EventNames.SymbolTableStatistics);
53       recording.enable(EventNames.StringTableStatistics);
54       recording.enable(EventNames.PlaceholderTableStatistics);
55       recording.enable(EventNames.LoaderConstraintsTableStatistics);
56       recording.enable(EventNames.ProtectionDomainCacheTableStatistics);
57       recording.start();
58       recording.stop();
59 
60       List<RecordedEvent> events = Events.fromRecording(recording);
61       verifyTable(events, EventNames.SymbolTableStatistics);
62       verifyTable(events, EventNames.StringTableStatistics);
63       verifyTable(events, EventNames.PlaceholderTableStatistics);
64       verifyTable(events, EventNames.LoaderConstraintsTableStatistics);
65       verifyTable(events, EventNames.ProtectionDomainCacheTableStatistics);
66     }
67   }
68 
69   private static void verifyTable(List<RecordedEvent> allEvents, String eventName) throws Exception {
70     List<RecordedEvent> eventsForTable = allEvents.stream().filter(e -> e.getEventType().getName().equals(eventName)).collect(Collectors.toList());
71     if (eventsForTable.isEmpty()) {
72       throw new Exception("No events for " + eventName);
73     }
74     for (RecordedEvent event : eventsForTable) {
75       Events.assertField(event, "bucketCount").atLeast(0L);
76       long entryCount = Events.assertField(event, "entryCount").atLeast(0L).getValue();
77       Events.assertField(event, "totalFootprint").atLeast(0L);
78       float averageBucketCount = Events.assertField(event, "bucketCountAverage").atLeast(0.0f).getValue();
79       Events.assertField(event, "bucketCountMaximum").atLeast((long)averageBucketCount);
80       Events.assertField(event, "bucketCountVariance").atLeast(0.0f);
81       Events.assertField(event, "bucketCountStandardDeviation").atLeast(0.0f);
82       float insertionRate = Events.assertField(event, "insertionRate").atLeast(0.0f).getValue();
83       float removalRate = Events.assertField(event, "removalRate").atLeast(0.0f).getValue();
84       if ((insertionRate > 0.0f) && (insertionRate > removalRate)) {
85         Asserts.assertGreaterThan(entryCount, 0L, "Entries marked as added, but no entries found for " + eventName);
86       }
87     }
88   }
89 }
