26 package jdk.jfr.event.oldobject;
27 
28 import java.time.Duration;
29 import java.util.ArrayList;
30 import java.util.List;
31 
32 import jdk.jfr.Recording;
33 import jdk.jfr.consumer.RecordedEvent;
34 import jdk.jfr.internal.test.WhiteBox;
35 import jdk.test.lib.jfr.EventNames;
36 import jdk.test.lib.jfr.Events;
37 
38 /**
39  * @test
40  * @key jfr
41  * @requires vm.hasJFR
42  * @requires vm.gc == "null"
43  * @library /test/lib /test/jdk
44  * @modules jdk.jfr/jdk.jfr.internal.test
45  * @run main/othervm -XX:TLABSize=2k jdk.jfr.event.oldobject.TestObjectAge
46  */
47 public class TestObjectAge {
48 
49     public final static List<Object[]> leak = new ArrayList<>(OldObjects.MIN_SIZE);
50 
51     public static void main(String[] args) throws Exception {
52         WhiteBox.setWriteAllObjectSamples(true);
53 
54         try (Recording recording = new Recording()) {
55             recording.enable(EventNames.OldObjectSample).withStackTrace().with("cutoff", "1 h");
56             recording.start();
57 
58             // Allocate array to trigger sampling code path for interpreter / c1
59             for (int i = 0; i < OldObjects.MIN_SIZE; i++) {
60                 leak.add(new Object[0]);
61             }
62             recording.stop();
63 
64             List<RecordedEvent> events = Events.fromRecording(recording);
65             // OK if we sometimes don't get an event
66             for (RecordedEvent event : events) {
67                 Duration objectAge = event.getDuration("objectAge");
68                 // Sanity check
69                 if (objectAge.isNegative() || objectAge.toMinutes() > 60) {
70                     throw new Exception("Object age " + objectAge + " is not reasonable");
71                 }
72             }
73         }
74     }
75 }
