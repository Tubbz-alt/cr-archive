26 package jdk.jfr.startupargs;
27 
28 import java.time.Duration;
29 
30 import jdk.jfr.FlightRecorder;
31 import jdk.jfr.Recording;
32 import jdk.jfr.internal.PlatformRecording;
33 import jdk.jfr.internal.PrivateAccess;
34 
35 /**
36  * @test
37  * @summary Start a recording with a flush interval
38  * @key jfr
39  * @requires vm.hasJFR
40  * @library /test/lib /test/jdk
41  * @modules jdk.jfr/jdk.jfr.internal
42  * @run main/othervm -XX:StartFlightRecording=flush-interval=2s jdk.jfr.startupargs.TestFlushInterval
43  */
44 public class TestFlushInterval {
45 
46     public static void main(String[] args) throws Exception {
47         for (Recording r : FlightRecorder.getFlightRecorder().getRecordings()) {
48             PrivateAccess p = PrivateAccess.getInstance();
49             PlatformRecording pr = p.getPlatformRecording(r);
50             Duration d = pr.getFlushInterval();
51             if (d.equals(Duration.ofSeconds(2))) {
52                 return; //OK
53             } else {
54                 throw new Exception("Unexpected flush-interval " + d);
55             }
56         }
57         throw new Exception("No recording found");
58     }
59 
60 }
