26 package jdk.jfr.startupargs;
27 
28 import jdk.jfr.Event;
29 import jdk.jfr.EventType;
30 import jdk.jfr.FlightRecorder;
31 import jdk.jfr.Name;
32 import jdk.jfr.Recording;
33 
34 /**
35  * @test
36  * @summary Start a FlightRecording without any settings (not even default).
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  * @run main/othervm jdk.jfr.startupargs.TestStartNoSettings
41  *      -XX:StartFlightRecording=settings=none
42  */
43 public class TestStartNoSettings {
44 
45     @Name("UserEvent")
46     static class UserEvent extends Event {
47     }
48 
49     public static void main(String[] a) throws Exception {
50         boolean userEnabled = false;
51         try (Recording r = new Recording()) {
52             r.start();
53             UserEvent e = new UserEvent();
54             e.commit();
55             for (EventType et : FlightRecorder.getFlightRecorder().getEventTypes()) {
56                 if (et.isEnabled()) {
57                     if (!et.getName().equals("UserEvent")) {
58                         throw new Exception("Only 'UserEvent' should be enabled");
59                     }
60                     userEnabled = true;
61                 }
62             }
63         }
64 
65         if (!userEnabled)  {
66             throw new Exception("Expected 'UserEvent' to be enabled with -XX:StartFlightRecording=settings=none");
67         }
68     }
69 }
