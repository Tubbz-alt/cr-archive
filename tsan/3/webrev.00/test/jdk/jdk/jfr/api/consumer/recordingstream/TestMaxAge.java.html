26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.time.Duration;
29 
30 import jdk.jfr.consumer.RecordingStream;
31 import jdk.test.lib.jfr.EventNames;
32 
33 /**
34  * @test
35  * @summary Tests RecordingStream::setMaxAge(...)
36  * @key jfr
37  * @requires vm.hasJFR
38  * @library /test/lib
39  * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestMaxAge
40  */
41 public class TestMaxAge {
42 
43     public static void main(String... args) throws Exception {
44         Duration testDuration = Duration.ofMillis(1234567);
45         try (RecordingStream r = new RecordingStream()) {
46             r.setMaxAge(testDuration);
47             r.enable(EventNames.ActiveRecording);
48             r.onEvent(e -> {
49                 System.out.println(e);
50                 Duration d = e.getDuration("maxAge");
51                 System.out.println(d.toMillis());
52                 if (testDuration.equals(d)) {
53                     r.close();
54                     return;
55                 }
56                 System.out.println("Max age not set, was " + d.toMillis() + " ms , but expected " + testDuration.toMillis() + " ms");
57             });
58             r.start();
59         }
60     }
61 }
