26 package jdk.jfr.api.consumer.streaming;
27 
28 import java.util.concurrent.CountDownLatch;
29 
30 import jdk.jfr.Event;
31 import jdk.jfr.FlightRecorder;
32 import jdk.jfr.Period;
33 import jdk.jfr.consumer.RecordingStream;
34 
35 /**
36  * @test
37  * @summary Verifies that it is possible to stream contents of ongoing
38  *          recordings
39  * @key jfr
40  * @requires vm.hasJFR
41  * @library /test/lib
42  * @run main/othervm -Xlog:jfr+system+streaming=trace
43  *      jdk.jfr.api.consumer.streaming.TestStartSingleChunk
44  */
45 public class TestStartSingleChunk {
46 
47     @Period("500 ms")
48     static class ElkEvent extends Event {
49     }
50 
51     static class FrogEvent extends Event {
52     }
53 
54     static class LionEvent extends Event {
55     }
56 
57     public static void main(String... args) throws Exception {
58         CountDownLatch frogLatch = new CountDownLatch(1);
59         CountDownLatch lionLatch = new CountDownLatch(1);
60         CountDownLatch elkLatch = new CountDownLatch(3);
61 
62         FlightRecorder.addPeriodicEvent(ElkEvent.class, () -> {
63             ElkEvent ee = new ElkEvent();
64             ee.commit();
65         });
66         try (RecordingStream s = new RecordingStream()) {
67             s.onEvent(ElkEvent.class.getName(), e -> {
68                 System.out.println("Found elk!");
69                 elkLatch.countDown();
70             });
71             s.onEvent(LionEvent.class.getName(), e -> {
72                 System.out.println("Found lion!");
73                 lionLatch.countDown();
74             });
75             s.onEvent(FrogEvent.class.getName(), e -> {
76                 System.out.println("Found frog!");
77                 frogLatch.countDown();
78             });
79             s.startAsync();
80             FrogEvent fe = new FrogEvent();
81             fe.commit();
82 
83             LionEvent le = new LionEvent();
84             le.commit();
85 
86             frogLatch.await();
87             lionLatch.await();
88             elkLatch.await();
89         }
90     }
91 }
