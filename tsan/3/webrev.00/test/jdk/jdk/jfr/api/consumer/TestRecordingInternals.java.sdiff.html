43 public class TestRecordingInternals {
44 
45     public static void main(String[] args) throws Throwable {
46         try (Recording continuous = new Recording()) {
47             continuous.start();
48             for (int i = 0; i < 3; i++) {
49                 // Each new recording will create a new chunk
50                 // with a new set of constant pools, i.e.
51                 // constant pools for threads and thread groups
52                 createProfilingRecording(i);
53             }
54             continuous.stop();
55             int i = 0;
56             for (RecordedEvent e : Events.fromRecording(continuous)) {
57                 Integer id = e.getValue("id");
58                 RecordedThread rt = e.getThread();
59                 Asserts.assertEquals(id.toString(), rt.getJavaName(), "Thread name should match id");
60                 Asserts.assertEquals(id.toString(), rt.getThreadGroup().getName(), "Thread group name should match id");
61                 Asserts.assertEquals(id, Integer.valueOf(i), "Chunks out of order");
62                 i++;

63             }
64         }
65     }
66 
67     private static void createProfilingRecording(int id) throws InterruptedException {
68         try (Recording r = new Recording()) {
69             r.start();
70             ThreadGroup tg = new ThreadGroup(String.valueOf(id));
71             Thread t = new Thread(tg, () -> {
72                 SimpleEvent event = new SimpleEvent();
73                 event.id = id;
74                 event.commit();
75             }, String.valueOf(id));
76             t.start();
77             t.join();
78             r.stop();
79         }
80     }
