26 package jdk.jfr.api.consumer.security;
27 
28 import java.io.IOException;
29 
30 import jdk.jfr.consumer.EventStream;
31 import jdk.jfr.consumer.RecordingStream;
32 
33 /**
34  * @test
35  * @summary Tests that streaming doesn't work if
36  *          FlightRecordingPermission("accessFlightRecorder") is missing
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  *
41  * @run main/othervm/secure=java.lang.SecurityManager/java.security.policy=no-permission.policy
42  *      jdk.jfr.api.consumer.security.TestMissingPermission
43  */
44 public class TestMissingPermission {
45 
46     public static void main(String... args) throws Exception {
47         testOpenRepository();
48         testRecordingStream();
49     }
50 
51     private static void testRecordingStream() throws IOException {
52         try {
53             try (EventStream es = EventStream.openRepository()) {
54                 throw new AssertionError("Should not be able to create EventStream without FlightRecorderPermission");
55             }
56         } catch (SecurityException se) {
57             // OK, as expected
58         }
59     }
60 
61     private static void testOpenRepository() throws IOException {
62         try {
63             try (RecordingStream es = new RecordingStream()) {
64                 throw new AssertionError("Should not be able to create RecordingStream without FlightRecorderPermission");
65             }
66         } catch (SecurityException se) {
67             // OK, as expected
68         }
69     }
70 }
