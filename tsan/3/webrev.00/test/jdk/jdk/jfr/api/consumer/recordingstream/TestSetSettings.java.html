26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.util.HashMap;
29 import java.util.Map;
30 import java.util.concurrent.CountDownLatch;
31 
32 import jdk.jfr.Enabled;
33 import jdk.jfr.Event;
34 import jdk.jfr.Name;
35 import jdk.jfr.consumer.RecordingStream;
36 
37 /**
38  * @test
39  * @summary Tests RecordingStream::setSettings
40  * @key jfr
41  * @requires vm.hasJFR
42  * @library /test/lib
43  * @run main/othervm -Xlog:jfr+system+parser jdk.jfr.api.consumer.recordingstream.TestSetSettings
44  */
45 public final class TestSetSettings {
46 
47     @Name("LateBloomer")
48     @Enabled(false)
49     private final static class LateBloomer extends Event {
50     }
51 
52     private static CountDownLatch lateBloomer = new CountDownLatch(1);
53 
54     public static void main(String... args) throws Exception {
55         try (RecordingStream r = new RecordingStream()) {
56             r.startAsync();
57             Map<String, String> settings = new HashMap<String, String>();
58             settings.put("LateBloomer#enabled", "true");
59             r.setSettings(settings);
60             r.onEvent("LateBloomer", e -> {
61                 lateBloomer.countDown();
62             });
63             LateBloomer event = new LateBloomer();
64             event.commit();
65             lateBloomer.await();
66         }
67     }
68 }
