26 package jdk.jfr.api.consumer.streaming;
27 
28 import java.time.Duration;
29 import java.util.concurrent.CountDownLatch;
30 
31 import jdk.jfr.Enabled;
32 import jdk.jfr.Event;
33 import jdk.jfr.FlightRecorder;
34 import jdk.jfr.consumer.RecordingStream;
35 
36 /**
37  * @test
38  * @summary Verifies that it is possible to stream contents from specified event
39  *          settings
40  * @key jfr
41  * @requires vm.hasJFR
42  * @library /test/lib
43  *
44  * @run main/othervm jdk.jfr.api.consumer.streaming.TestEnableEvents
45  */
46 public class TestEnableEvents {
47 
48     @Enabled(false)
49     static class HorseEvent extends Event {
50     }
51 
52     @Enabled(false)
53     static class ElephantEvent extends Event {
54     }
55 
56     @Enabled(false)
57     static class TigerEvent extends Event {
58     }
59 
60     public static void main(String... args) throws Exception {
61         CountDownLatch elephantLatch = new CountDownLatch(1);
62         CountDownLatch tigerLatch = new CountDownLatch(1);
63         CountDownLatch horseLatch = new CountDownLatch(1);
64 
65         FlightRecorder.addPeriodicEvent(ElephantEvent.class, () -> {
66             HorseEvent ze = new HorseEvent();
67             ze.commit();
68         });
69 
70         try (RecordingStream s = new RecordingStream()) {
71             s.enable(HorseEvent.class.getName()).withPeriod(Duration.ofMillis(50));
72             s.enable(TigerEvent.class.getName());
73             s.enable(ElephantEvent.class.getName());
74             s.onEvent(TigerEvent.class.getName(), e -> {
75                 System.out.println("Event: " + e.getEventType().getName());
76                 System.out.println("Found tiger!");
77                 tigerLatch.countDown();
78             });
79             s.onEvent(HorseEvent.class.getName(), e -> {
80                 System.out.println("Found horse!");
81                 horseLatch.countDown();
82             });
83             s.onEvent(ElephantEvent.class.getName(), e -> {
84                 System.out.println("Found elelphant!");
85                 elephantLatch.countDown();
86             });
87             s.startAsync();
88             TigerEvent te = new TigerEvent();
89             te.commit();
90             ElephantEvent ee = new ElephantEvent();
91             ee.commit();
92             elephantLatch.await();
93             horseLatch.await();
94             tigerLatch.await();
95         }
96 
97     }
98 
99 }
