26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.util.concurrent.CountDownLatch;
29 
30 import jdk.jfr.Event;
31 import jdk.jfr.FlightRecorder;
32 import jdk.jfr.consumer.RecordingStream;
33 
34 /**
35  * @test
36  * @summary Tests that a RecordingStream is closed if the underlying Recording
37  *          is stopped.
38  * @key jfr
39  * @requires vm.hasJFR
40  * @library /test/lib
41  * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestStoppedRecording
42  */
43 public class TestStoppedRecording {
44 
45     private static final class StopEvent extends Event {
46     }
47 
48     public static void main(String... args) throws Exception {
49         CountDownLatch latch = new CountDownLatch(1);
50         try (RecordingStream rs = new RecordingStream()) {
51             rs.onEvent(e -> {
52                 FlightRecorder.getFlightRecorder().getRecordings().get(0).stop();
53             });
54             rs.onClose(() -> {
55                 latch.countDown();
56             });
57             rs.startAsync();
58             StopEvent stop = new StopEvent();
59             stop.commit();
60             latch.await();
61         }
62     }
63 }
