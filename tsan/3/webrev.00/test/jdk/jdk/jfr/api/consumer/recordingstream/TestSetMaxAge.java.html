26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.time.Duration;
29 
30 import jdk.jfr.consumer.RecordingStream;
31 import jdk.test.lib.jfr.EventNames;
32 
33 /**
34  * @test
35  * @summary Tests RecordingStrream::setMaxAge
36  * @key jfr
37  * @requires vm.hasJFR
38  * @library /test/lib
39  * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestSetMaxAge
40  */
41 public class TestSetMaxAge {
42 
43     public static void main(String... args) throws Exception {
44         Duration expecteddAge = Duration.ofMillis(123456789);
45         try (RecordingStream r = new RecordingStream()) {
46             r.setMaxAge(expecteddAge);
47             r.enable(EventNames.ActiveRecording);
48             r.onEvent(e -> {
49                 System.out.println(e);
50                 Duration age = e.getDuration("maxAge");
51                 if (expecteddAge.equals(age)) {
52                     r.close();
53                     return;
54                 }
55                 System.out.println("Max age not set, was " + age + ", but expected " + expecteddAge);
56             });
57             r.start();
58         }
59     }
60 }
