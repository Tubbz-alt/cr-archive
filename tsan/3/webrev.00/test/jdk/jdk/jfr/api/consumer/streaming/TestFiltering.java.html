26 package jdk.jfr.api.consumer.streaming;
27 
28 import java.util.concurrent.CountDownLatch;
29 import java.util.concurrent.atomic.AtomicInteger;
30 
31 import jdk.jfr.Event;
32 import jdk.jfr.consumer.RecordingStream;
33 
34 /**
35  * @test
36  * @summary Verifies that it is possible to filter a stream for an event
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  * @run main/othervm jdk.jfr.api.consumer.streaming.TestFiltering
41  */
42 public class TestFiltering {
43 
44     static class SnakeEvent extends Event {
45         int id;
46     }
47 
48     static class EelEvent extends Event {
49         int id;
50     }
51 
52     public static void main(String... args) throws Exception {
53         CountDownLatch l = new CountDownLatch(1);
54         String eventName = SnakeEvent.class.getName();
55         AtomicInteger idCounter = new AtomicInteger(-1);
56         try (RecordingStream e = new RecordingStream()) {
57             e.onEvent(eventName, event -> {
58                 if (!event.getEventType().getName().equals(eventName)) {
59                     throw new InternalError("Unexpected event " + e);
60                 }
61                 if (event.getInt("id") != idCounter.incrementAndGet()) {
62                     throw new InternalError("Incorrect id");
63                 }
64                 if (idCounter.get() == 99) {
65                     l.countDown();
66                 }
67             });
68             e.startAsync();
69             for (int i = 0; i < 100; i++) {
70                 SnakeEvent se = new SnakeEvent();
71                 se.id = i;
72                 se.commit();
73 
74                 EelEvent ee = new EelEvent();
75                 ee.id = i;
76                 ee.commit();
77             }
78             l.await();
79         }
80     }
81 }
