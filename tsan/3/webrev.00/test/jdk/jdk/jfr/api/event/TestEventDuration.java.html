25 package jdk.jfr.api.event;
26 
27 import java.util.List;
28 
29 import jdk.jfr.Recording;
30 import jdk.jfr.consumer.RecordedEvent;
31 import jdk.test.lib.jfr.Events;
32 import jdk.test.lib.jfr.SimpleEvent;
33 
34 /**
35  * @test
36  * @summary Tests that a duration is recorded.
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  * @run main/othervm jdk.jfr.api.event.TestEventDuration
41  */
42 public class TestEventDuration {
43 
44     public static int counter;
45 
46     public static void main(String[] args) throws Exception {
47 
48         try(Recording r = new Recording()) {
49             r.start();
50             SimpleEvent e = new SimpleEvent();
51             e.begin();
52             for (int i = 0; i < 10_000;i++) {
53                 counter+=i;
54             }
55             e.end();
56             e.commit();
57 
58             r.stop();
59             List<RecordedEvent> events = Events.fromRecording(r);
60             if (events.get(0).getDuration().toNanos() < 1) {
61                 throw new AssertionError("Expected a duration");
62             }
63         }
64 
65     }
66 
67 }
