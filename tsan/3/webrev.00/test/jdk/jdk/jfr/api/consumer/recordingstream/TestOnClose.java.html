26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.util.concurrent.CountDownLatch;
29 import java.util.concurrent.atomic.AtomicBoolean;
30 
31 import jdk.jfr.Event;
32 import jdk.jfr.consumer.RecordingStream;
33 
34 /**
35  * @test
36  * @summary Tests RecordingStream::onClose(...)
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestOnClose
41  */
42 public class TestOnClose {
43 
44     private static class CloseEvent extends Event {
45     }
46 
47     public static void main(String... args) throws Exception {
48         testOnCloseNull();
49         testOnClosedUnstarted();
50         testOnClosedStarted();
51     }
52 
53     private static void testOnCloseNull() {
54        try (RecordingStream rs = new RecordingStream()) {
55           try {
56               rs.onClose(null);
57               throw new AssertionError("Expected NullPointerException from onClose(null");
58           } catch (NullPointerException npe) {
59               // OK; as expected
60           }
61        }
62     }
63 
64     private static void testOnClosedStarted() throws InterruptedException {
65         AtomicBoolean onClose = new AtomicBoolean(false);
66         CountDownLatch event = new CountDownLatch(1);
67         try (RecordingStream r = new RecordingStream()) {
68             r.onEvent(e -> {
69                 event.countDown();
70             });
71             r.onClose(() -> {
72                 onClose.set(true);
73             });
74             r.startAsync();
75             CloseEvent c = new CloseEvent();
76             c.commit();
77             event.await();
78         }
79         if (!onClose.get()) {
80             throw new AssertionError("OnClose was not called");
81         }
82     }
83 
84     private static void testOnClosedUnstarted() {
85         AtomicBoolean onClose = new AtomicBoolean(false);
86         try (RecordingStream r = new RecordingStream()) {
87             r.onClose(() -> {
88                 onClose.set(true);
89             });
90         }
91         if (!onClose.get()) {
92             throw new AssertionError("OnClose was not called");
93         }
94     }
95 }
