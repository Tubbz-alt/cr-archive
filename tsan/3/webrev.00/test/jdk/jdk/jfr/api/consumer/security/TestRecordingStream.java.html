26 package jdk.jfr.api.consumer.security;
27 
28 import java.time.Instant;
29 import java.util.concurrent.CountDownLatch;
30 
31 import jdk.jfr.consumer.RecordingStream;
32 import jdk.test.lib.jfr.EventNames;
33 
34 /**
35  * @test
36  * @summary Tests that a RecordingStream works using only
37  *          FlightRecordingPermission("accessFlightRecorder")
38  * @key jfr
39  * @requires vm.hasJFR
40  * @library /test/lib
41  *
42  * @run main/othervm/secure=java.lang.SecurityManager/java.security.policy=local-streaming.policy
43  *      jdk.jfr.api.consumer.security.TestStreamingLocal
44  */
45 public class TestRecordingStream {
46 
47     public static void main(String... args) throws Exception {
48         CountDownLatch latch = new CountDownLatch(1);
49         try (RecordingStream r = new RecordingStream()) {
50             // Enable JVM event, no write permission needed
51             r.enable(EventNames.JVMInformation);
52             r.setStartTime(Instant.EPOCH);
53             r.onEvent(EventNames.JVMInformation, e -> {
54                 latch.countDown();
55             });
56             r.startAsync();
57             latch.await();
58         }
59     }
60 }
