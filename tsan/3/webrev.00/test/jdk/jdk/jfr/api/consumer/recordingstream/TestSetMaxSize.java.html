26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import jdk.jfr.consumer.RecordingStream;
29 import jdk.test.lib.jfr.EventNames;
30 
31 /**
32 * @test
33 * @summary Tests RecordingStrream::setMaxSize
34 * @key jfr
35 * @requires vm.hasJFR
36 * @library /test/lib
37 * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestSetMaxSize
38 */
39 public class TestSetMaxSize {
40 
41    public static void main(String... args) throws Exception {
42        long testSize = 123456789;
43        try (RecordingStream r = new RecordingStream()) {
44            r.setMaxSize(123456789);
45            r.enable(EventNames.ActiveRecording);
46            r.onEvent(e -> {
47                System.out.println(e);
48                long size= e.getLong("maxSize");
49                if (size == testSize) {
50                    r.close();
51                    return;
52                }
53                System.out.println("Max size not set, was " + size + ", but expected " + testSize);
54            });
55            r.start();
56        }
57    }
58 }
