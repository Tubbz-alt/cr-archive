26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.util.concurrent.CountDownLatch;
29 
30 import jdk.jfr.Event;
31 import jdk.jfr.consumer.RecordingStream;
32 
33 /**
34  * @test
35  * @summary Tests RecordingStream::startAsync()
36  * @key jfr
37  * @requires vm.hasJFR
38  * @library /test/lib
39  * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestStartAsync
40  */
41 public class TestStartAsync {
42     static class StartEvent extends Event {
43     }
44 
45     public static void main(String... args) throws Exception {
46         testStart();
47         testStartTwice();
48         testStartClosed();
49     }
50 
51     private static void testStartTwice() throws Exception {
52         try (RecordingStream rs = new RecordingStream()) {
53             rs.startAsync();
54             try {
55                 rs.startAsync();
56                 throw new AssertionError("Expected IllegalStateException if started twice");
57             } catch (IllegalStateException ise) {
58                 // OK, as expected
59             }
60         }
61     }
62 
63     static void testStart() throws Exception {
64         CountDownLatch started = new CountDownLatch(1);
65         try (RecordingStream rs = new RecordingStream()) {
66             rs.onEvent(e -> {
67                 started.countDown();
68             });
69             rs.startAsync();
70             StartEvent e = new StartEvent();
71             e.commit();
72             started.await();
73         }
74     }
75 
76     static void testStartClosed() {
77         RecordingStream rs = new RecordingStream();
78         rs.close();
79         try {
80             rs.startAsync();
81             throw new AssertionError("Expected IllegalStateException");
82         } catch (IllegalStateException ise) {
83             // OK, as expected.
84         }
85     }
86 }
