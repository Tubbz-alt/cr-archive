26 package jdk.jfr.api.consumer;
27 
28 import java.nio.file.Path;
29 import java.nio.file.Paths;
30 import java.util.LinkedList;
31 import java.util.List;
32 
33 import jdk.jfr.Event;
34 import jdk.jfr.Recording;
35 import jdk.jfr.consumer.RecordedEvent;
36 import jdk.jfr.consumer.RecordingFile;
37 import jdk.test.lib.Asserts;
38 import jdk.test.lib.Utils;
39 
40 
41 /**
42  * @test
43  * @summary Reads the recorded file two times and verifies that both reads are the same
44  * @key jfr
45  * @requires vm.hasJFR
46  * @library /test/lib
47  * @run main/othervm jdk.jfr.api.consumer.TestReadTwice
48  */
49 public class TestReadTwice {
50 
51     private static class MyEvent extends Event {
52     }
53 
54     public static void main(String[] args) throws Throwable {
55         Recording r = new Recording();
56         r.enable(MyEvent.class).withoutStackTrace();
57         r.start();
58 
59         // Commit a single event to the recording
60         MyEvent event = new MyEvent();
61         event.commit();
62 
63         r.stop();
64 
65         // Dump the recording to a file
66         Path path = Utils.createTempFile("read-twice", ".jfr");
67         System.out.println("Dumping to " + path);
68         r.dump(path);
69         r.close();
70 
71         // Read all events from the file in one go
72         List<RecordedEvent> events = RecordingFile.readAllEvents(path);
73 
74         // Read again the same events one by one
75         RecordingFile rfile = new RecordingFile(path);
76         List<RecordedEvent> events2 = new LinkedList<>();
77         while (rfile.hasMoreEvents()) {
78             events2.add(rfile.readEvent());
79         }
80 
81         // Compare sizes
82         Asserts.assertEquals(events.size(), events2.size());
83         rfile.close();

84     }
85 }
