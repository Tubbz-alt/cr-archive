26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.time.Duration;
29 import java.util.concurrent.CompletableFuture;
30 import java.util.concurrent.ExecutionException;
31 
32 import jdk.jfr.consumer.RecordingStream;
33 
34 /**
35  * @test
36  * @summary Test RecordingStream::awaitTermination(...)
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib
40  * @run main/othervm jdk.jfr.api.consumer.recordingstream.TestAwaitTermination
41  */
42 public class TestAwaitTermination {
43 
44     public static void main(String... args) throws Exception {
45         testAwaitClose();
46         testAwaitTimeOut();
47     }
48 
49     private static void testAwaitClose() throws InterruptedException, ExecutionException {
50         try (RecordingStream r = new RecordingStream()) {
51             r.startAsync();
52             var c = CompletableFuture.runAsync(() -> {
53                 try {
54                     r.awaitTermination();
55                 } catch (InterruptedException e) {
56                     e.printStackTrace();
57                 }
58             });
59             r.close();
60             c.get();
61         }
62     }
63 
64     private static void testAwaitTimeOut() throws InterruptedException, ExecutionException {
65         try (RecordingStream r = new RecordingStream()) {
66             r.startAsync();
67             var c = CompletableFuture.runAsync(() -> {
68                 try {
69                     r.awaitTermination(Duration.ofMillis(10));
70                 } catch (InterruptedException e) {
71                     e.printStackTrace();
72                 }
73             });
74             c.get();
75             r.close();
76         }
77     }
78 }
