26 package jdk.jfr.api.consumer.streaming;
27 
28 import java.util.concurrent.CountDownLatch;
29 
30 import jdk.jfr.Event;
31 import jdk.jfr.FlightRecorder;
32 import jdk.jfr.Registered;
33 import jdk.jfr.consumer.RecordingStream;
34 
35 /**
36  * @test
37  * @summary Test that it is possible to register new metadata in a chunk
38  * @key jfr
39  * @requires vm.hasJFR
40  * @library /test/lib
41  * @run main/othervm jdk.jfr.api.consumer.streaming.TestEventRegistration
42  */
43 public class TestEventRegistration {
44     @Registered(false)
45     static class StreamEvent1 extends Event {
46     }
47 
48     @Registered(false)
49     static class StreamEvent2 extends Event {
50     }
51 
52     public static void main(String... args) throws Exception {
53 
54         CountDownLatch s1Latch = new CountDownLatch(1);
55         CountDownLatch s2Latch = new CountDownLatch(1);
56         try (RecordingStream es = new RecordingStream()) {
57             es.onEvent(StreamEvent1.class.getName(), e -> {
58                 s1Latch.countDown();
59             });
60             es.onEvent(StreamEvent2.class.getName(), e -> {
61                 s2Latch.countDown();
62             });
63             es.startAsync();
64             System.out.println("Registering " + StreamEvent1.class.getName());
65             FlightRecorder.register(StreamEvent1.class);
66             StreamEvent1 s1 = new StreamEvent1();
67             s1.commit();
68             System.out.println(StreamEvent1.class.getName() + " commited");
69             System.out.println("Awaiting latch for " + StreamEvent1.class.getName());
70             s1Latch.await();
71             System.out.println();
72             System.out.println("Registering " + StreamEvent2.class.getName());
73             FlightRecorder.register(StreamEvent2.class);
74             StreamEvent2 s2 = new StreamEvent2();
75             s2.commit();
76             System.out.println(StreamEvent2.class.getName() + " commited");
77             System.out.println("Awaiting latch for " + StreamEvent2.class.getName());
78             s2Latch.await();
79         }
80     }
81 }
