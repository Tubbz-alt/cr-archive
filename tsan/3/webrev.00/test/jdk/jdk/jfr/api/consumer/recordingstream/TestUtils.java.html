26 package jdk.jfr.api.consumer.recordingstream;
27 
28 import java.util.concurrent.CountDownLatch;
29 
30 public class TestUtils {
31 
32     public static final class TestError extends Error {
33         private static final long serialVersionUID = 1L;
34     }
35 
36     public static final class TestException extends Exception {
37         private static final long serialVersionUID = 1L;
38         private volatile boolean printed;
39 
40         @Override
41         public void printStackTrace() {
42             super.printStackTrace();
43             printed = true;
44         }
45 
46         public boolean isPrinted() {
47             return printed;
48         }
49     }
50 
51     // Can throw checked exception as unchecked.
52     @SuppressWarnings("unchecked")
53     public static <T extends Throwable> void throwUnchecked(Throwable e) throws T {
54         throw (T) e;
55     }
56 
57     public static void installUncaughtException(CountDownLatch receivedError, Throwable expected) {
58         Thread.currentThread().setUncaughtExceptionHandler((thread, throwable) -> {
59             if (throwable == expected) {
60                 System.out.println("Received uncaught exception " + expected.getClass());
61                 receivedError.countDown();
62             }
63         });
64     }
65 }
