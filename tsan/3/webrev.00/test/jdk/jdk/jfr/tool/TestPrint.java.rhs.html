26 package jdk.jfr.tool;
27 
28 import java.io.FileWriter;
29 import java.nio.file.Files;
30 import java.nio.file.Path;
31 
32 import jdk.test.lib.Utils;
33 import jdk.test.lib.process.OutputAnalyzer;
34 
35 /**
36  * @test
37  * @summary Test jfr print
38  * @key jfr
39  * @requires vm.hasJFR
40  * @library /test/lib /test/jdk
41  * @run main/othervm jdk.jfr.tool.TestPrint
42  */
43 public class TestPrint {
44 
45     public static void main(String[] args) throws Throwable {
46 
47         OutputAnalyzer output = ExecuteHelper.jfr("print");
48         output.shouldContain("missing file");
49 
50         output = ExecuteHelper.jfr("print", "missing.jfr");
51         output.shouldContain("could not open file ");
52 
53         Path file = Utils.createTempFile("faked-print-file",  ".jfr");
54         FileWriter fw = new FileWriter(file.toFile());
55         fw.write('d');
56         fw.close();
57         output = ExecuteHelper.jfr("print", "--wrongOption", file.toAbsolutePath().toString());
58         output.shouldContain("unknown option");
59         Files.delete(file);
60     }
61 }
