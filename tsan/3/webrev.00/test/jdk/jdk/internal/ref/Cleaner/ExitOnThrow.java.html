33  *        jdk.test.lib.Platform
34  *        jdk.test.lib.process.*
35  * @run main ExitOnThrow
36  * @summary Ensure that if a cleaner throws an exception then the VM exits
37  */
38 
39 import jdk.internal.ref.Cleaner;
40 
41 import jdk.test.lib.process.ProcessTools;
42 
43 public class ExitOnThrow {
44 
45     public static void main(String[] args) throws Exception {
46         if (args.length == 0) {
47             ProcessTools.executeTestJvm("--add-exports", "java.base/jdk.internal.ref=ALL-UNNAMED",
48                                         "ExitOnThrow",
49                                         "-executeCleaner")
50                         .outputTo(System.out)
51                         .errorTo(System.out)
52                         .shouldHaveExitValue(1)
53                         .shouldContain("java.lang.RuntimeException: Foo!");
54         } else {
55             Cleaner.create(new Object(),
56                            () -> { throw new RuntimeException("Foo!"); } );
57             while (true) {
58                 System.gc();
59                 Thread.sleep(100);
60             }
61         }
62     }
63 
64 }
