24 import com.sun.net.httpserver.*;
25 
26 import java.io.IOException;
27 import java.io.InputStream;
28 import java.io.OutputStream;
29 import java.net.InetAddress;
30 import java.net.InetSocketAddress;
31 import java.nio.file.Files;
32 import java.nio.file.Path;
33 import java.nio.file.Paths;
34 
35 /**
36  * Extremely simple server that only performs one task.  The server listens for
37  * requests on the ephemeral port.  If it sees a request that begins with
38  * "/multi-release.jar", it consumes the request and returns a stream of bytes
39  * representing the jar file multi-release.jar found in "userdir".
40  */
41 class SimpleHttpServer {
42     private static final String userdir = System.getProperty("user.dir", ".");
43     private static final Path multirelease = Paths.get(userdir, "multi-release.jar");
44 
45     private final HttpServer server;
46     private final InetAddress address;
47 
48     public SimpleHttpServer() throws IOException {
49         this(null);
50     }
51 
52     public SimpleHttpServer(InetAddress addr) throws IOException {
53         address = addr;
54         server = HttpServer.create();
55     }
56 
57     public void start() throws IOException {
58         server.bind(new InetSocketAddress(address, 0), 0);
59         server.createContext("/multi-release.jar", t -> {
60             try (InputStream is = t.getRequestBody()) {
61                 is.readAllBytes();  // probably not necessary to consume request
62                 byte[] bytes = Files.readAllBytes(multirelease);
63                 t.sendResponseHeaders(200, bytes.length);
64                 try (OutputStream os = t.getResponseBody()) {
65                     os.write(bytes);
66                 }
67             }
68         });
69         server.setExecutor(null); // creates a default executor
70         server.start();
71     }
72 
73     public void stop() {
74         server.stop(0);
75     }
76 
77     int getPort() {
78         return server.getAddress().getPort();
79     }
80 }
