 24 /*
 25  * @test
 26  * @bug 4727547
 27  * @summary SocksSocketImpl throws NullPointerException
 28  * @build SocksServer
 29  * @run main/othervm SocksV4Test
 30  */
 31 
 32 import java.io.IOException;
 33 import java.net.*;
 34 
 35 public class SocksV4Test {
 36 
 37     // An unresolvable host
 38     static final String HOSTNAME = "doesnot.exist.invalid";
 39     static final String USER = "johndoe";
 40     static final String PASSWORD = "helloworld";
 41 
 42     public static void main(String[] args) throws Exception {
 43         Authenticator.setDefault(new Auth());
 44         UHETest();
 45         getLocalPortTest();
 46     }
 47 
 48     static class Auth extends Authenticator {
 49         protected PasswordAuthentication getPasswordAuthentication() {
 50             return new PasswordAuthentication(USER, PASSWORD.toCharArray());
 51         }
 52     }
 53 
 54     public static void getLocalPortTest() throws Exception {
 55         // We actually use V5 for this test because that is the default
 56         // protocol version used by the client and it doesn't really handle
 57         // down grading very well.
 58         try (SocksServer srvr = new SocksServer(0, false);
 59              ServerSocket ss = new ServerSocket(0)) {
 60 
 61             srvr.addUser(USER, PASSWORD);
 62             int serverPort = ss.getLocalPort();
 63             srvr.start();
 64             int proxyPort = srvr.getPort();
 65             System.out.printf("Server port %d, Proxy port %d\n", serverPort, proxyPort);
 66             Proxy sp = new Proxy(Proxy.Type.SOCKS,
 67                     new InetSocketAddress("localhost", proxyPort));
 68             // Let's create an unresolved address
 69             InetSocketAddress ad = new InetSocketAddress("127.0.0.1", serverPort);
 70             try (Socket s = new Socket(sp)) {
 71                 s.connect(ad, 10000);
 72                 int pp = s.getLocalPort();
 73                 System.out.println("Local port = " + pp);
 74                 if (pp == serverPort || pp == proxyPort)
 75                     throw new RuntimeException("wrong port returned");
 76             } catch (UnknownHostException ex) {
 77                 throw new RuntimeException(ex);
 78             } catch (IOException ioe) {
 79                 throw new RuntimeException(ioe);
 80             }
 81         }
 82     }
 83 
 84     public static void UHETest() throws Exception {
 85         // sanity before running the test
 86         assertUnresolvableHost(HOSTNAME);
 87 
 88         // Create a SOCKS V4 proxy
 89         try (SocksServer srvr = new SocksServer(0, true)) {
 90             srvr.start();
 91             Proxy sp = new Proxy(Proxy.Type.SOCKS,
 92                     new InetSocketAddress("localhost", srvr.getPort()));
 93             // Let's create an unresolved address
 94             InetSocketAddress ad = new InetSocketAddress(HOSTNAME, 1234);
 95             try (Socket s = new Socket(sp)) {
 96                 s.connect(ad, 10000);
 97             } catch (UnknownHostException ex) {
 98                 // OK, that's what we expected
 99             } catch (NullPointerException npe) {
100                 // Not OK, this used to be the bug
101                 throw new RuntimeException("Got a NUllPointerException");
102             }
103         }
104     }
105 
106     static void assertUnresolvableHost(String host) {
107         InetAddress addr = null;
108         try {
109             addr = InetAddress.getByName(host);
110         } catch (UnknownHostException x) {
111             // OK, expected
112         }
113         if (addr != null)
114             throw new RuntimeException("Test cannot run. resolvable address:" + addr);
115     }
116 }
