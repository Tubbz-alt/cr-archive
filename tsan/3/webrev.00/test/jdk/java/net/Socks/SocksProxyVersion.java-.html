 24 /*
 25  * @test
 26  * @bug 6964547 5001942 8129444
 27  * @run main/othervm SocksProxyVersion
 28  * @summary test socksProxyVersion system property
 29  */
 30 
 31 import java.net.InetSocketAddress;
 32 import java.net.ServerSocket;
 33 import java.net.Socket;
 34 import java.net.SocketException;
 35 import java.io.IOException;
 36 import java.net.InetAddress;
 37 import java.net.Proxy;
 38 
 39 public class SocksProxyVersion implements Runnable {
 40     final ServerSocket ss;
 41     volatile boolean failed;
 42     volatile boolean stopped = false;
 43     volatile int expected;
 44 
 45     public static void main(String[] args) throws Exception {
 46         if (InetAddress.getLocalHost().isLoopbackAddress()) {
 47             System.out.println("Test cannot run. getLocalHost returns a loopback address");
 48             return;
 49         }
 50         new SocksProxyVersion();
 51     }
 52 
 53     public SocksProxyVersion() throws Exception {
 54         ss = new ServerSocket(0);
 55         int port = ss.getLocalPort();
 56         Thread serverThread = new Thread(this);
 57         serverThread.start();
 58         try (ServerSocket socket = ss) {
 59             runTest(port);
 60         } finally {
 61             stopped = true;
 62         }
 63 
 64         serverThread.join();
 65         if (failed) {
 66             throw new RuntimeException("socksProxyVersion not being set correctly");
 67         }
 68     }
 69 
 70     final void runTest(int port) throws Exception {
 71         /*
 72          * Retrieving the IP Address of the machine
 73          * since "localhost" is bypassed as a non-proxy host
 74          */
 75         String addr = InetAddress.getLocalHost().getHostAddress();
 76 
 77         System.setProperty("socksProxyHost", addr);
 78         System.setProperty("socksProxyPort", Integer.toString(port));
 79 
 80         Proxy proxy = new Proxy(Proxy.Type.SOCKS,
 81                                 new InetSocketAddress(addr, port));
 82 
 83         // SOCKS V4
 84         System.setProperty("socksProxyVersion", Integer.toString(4));
 85         this.expected = 4;
 86         check(new Socket(), addr, port);
 87         check(new Socket(proxy), addr, port);
 88 
 89         // SOCKS V5
 90         System.setProperty("socksProxyVersion", Integer.toString(5));
 91         this.expected = 5;
 92         check(new Socket(), addr, port);
 93         check(new Socket(proxy), addr, port);
 94     }
 95 
 96     private void check(Socket socket, String addr, int port)
 97         throws IOException
 98     {
 99         try (Socket s = socket) {
100             socket.connect(new InetSocketAddress(addr, port));
101         } catch (SocketException e) {
102             // java.net.SocketException: Malformed reply from SOCKS server
103             // This exception is OK, since the "server" does not implement
104             // the socks protocol. It just verifies the version and closes.
105         }
106     }
107 
108     @Override
109     public void run() {
110         int count = 0;
111         try {
112             while (!stopped) {
113                 try (Socket s = ss.accept()) {
114                     int version = (s.getInputStream()).read();
115                     if (version != expected) {
116                         System.out.printf("Iteration: %d, Got: %d, expected: %d%n",
117                                           count, version, expected);
118                         failed = true;
119                     }
120                 }
121                 count++;
122             }
123         } catch (IOException e) {
124             if (!ss.isClosed()) {
125                 e.printStackTrace();
126             }
127             // ignore, server socket was closed
128         }
129     }
130 }
