494     public SocksServer(InetAddress addr, int port, boolean useV4) throws IOException {
495         this.port = port;
496         this.useV4 = useV4;
497         server = new ServerSocket();
498         if (port == 0 && addr == null) {
499             server.bind(null);
500             this.port = server.getLocalPort();
501         } else if (port == 0 && addr != null) {
502             server.bind(new InetSocketAddress(addr, 0));
503             this.port = server.getLocalPort();
504         } else if (addr == null) {
505             assert port != 0;
506             server.bind(new InetSocketAddress(port));
507         } else {
508             assert port != 0;
509             server.bind(new InetSocketAddress(addr, port));
