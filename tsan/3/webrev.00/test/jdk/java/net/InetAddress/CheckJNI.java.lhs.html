24 /* @test
25  * @bug 4889870 4890033
26  * @summary java -Xcheck:jni failing in net code on Solaris / [Datagram]Socket.getLocalAddress() failure
27  * @library /test/lib
28  * @build jdk.test.lib.NetworkConfiguration
29  *        jdk.test.lib.Platform
30  * @run main/othervm -Xcheck:jni CheckJNI
31  */
32 
33 import java.net.*;
34 import java.util.*;
35 import java.util.stream.Collectors;
36 import jdk.test.lib.NetworkConfiguration;

37 
38 public class CheckJNI {
39     static Socket s;
40     static ServerSocket server;
41     static DatagramSocket dg1, dg2;
42 
43     public static void main (String[] args) throws Exception {
44         /* try to invoke as much java.net native code as possible */
45 
46         System.out.println ("Testing IPv4 Socket/ServerSocket");
47         server = new ServerSocket (0);
48         s = new Socket ("127.0.0.1", server.getLocalPort());
49         s.close();
50         server.close();






51 
52         System.out.println ("Testing IPv4 DatagramSocket");
53         dg1 = new DatagramSocket (0, InetAddress.getByName ("127.0.0.1"));
54         dg2 = new DatagramSocket (0, InetAddress.getByName ("127.0.0.1"));
55         testDatagrams (dg1, dg2);
56 
57         /* Find link local IPv6 addrs to test */
58         List<Inet6Address> addrs = NetworkConfiguration.probe()
59                 .ip6Addresses()
60                 .filter(Inet6Address::isLinkLocalAddress)
61                 .collect(Collectors.toList());
62 
63         server = new ServerSocket(0);
64         for (Inet6Address ia6 : addrs) {
65             System.out.println("Address:" + ia6);
66             System.out.println("Testing IPv6 Socket");
67             s = new Socket(ia6, server.getLocalPort());
68             s.close();
69 
70             System.out.println("Testing IPv6 DatagramSocket");
71             dg1 = new DatagramSocket(0, ia6);
72             dg2 = new DatagramSocket(0, ia6);
73             testDatagrams(dg1, dg2);
74         }







75         server.close();
76         System.out.println ("OK");
77     }
78 
79     static void testDatagrams (DatagramSocket s1, DatagramSocket s2) throws Exception {



80         DatagramPacket p1 = new DatagramPacket (
81                 "hello world".getBytes(),
82                 0, "hello world".length(), s2.getLocalAddress(),
83                 s2.getLocalPort()
84         );
85 
86         DatagramPacket p2 = new DatagramPacket (new byte[128], 128);
87         s1.send (p1);
88         s2.receive (p2);
89         s1.close ();
90         s2.close ();
91     }
92 }
