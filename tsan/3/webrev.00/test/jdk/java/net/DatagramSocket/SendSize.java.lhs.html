 24 /* @test
 25  *
 26  * @bug 4095393
 27  * @test main SendSize
 28  * @summary this tests a regression in 1.2, beta 2 and earlier where
 29  * the DatagramPackets sent the entire buffer, not the buffer length
 30  * of the packet.
 31  *
 32  * @author Benjamin Renaud
 33  */
 34 
 35 import java.io.IOException;
 36 import java.net.DatagramPacket;
 37 import java.net.DatagramSocket;
 38 import java.net.InetAddress;

 39 
 40 public class SendSize {
 41     static final int bufferLength = 512;
 42     static final int packetLength = 256;
 43 
 44     public static void main(String[] args) throws Exception {
 45         DatagramSocket serverSocket = new DatagramSocket();


 46         Thread server = new ServerThread(serverSocket);
 47         server.start();
 48         Thread client = new ClientThread(serverSocket.getLocalPort());
 49         client.start();
 50         server.join();
 51         client.join();
 52     }
 53 
 54     static class ServerThread extends Thread {
 55         DatagramSocket server;
 56 
 57         ServerThread(DatagramSocket server) {
 58             this.server = server;
 59         }
 60 
 61         public void run() {
 62             try {
 63                 System.err.println("started server thread: " + server);
 64                 byte[] buf = new byte[1024];
 65                 for (int i = 0; i < 10; i++) {
 66                     DatagramPacket receivePacket = new DatagramPacket(buf,
 67                             buf.length);
 68                     server.receive(receivePacket);
 69                     int len = receivePacket.getLength();
 70                     switch (len) {
 71                         case packetLength:
 72                             System.out.println("receive length is: " + len);
 73                             break;
 74                         default:
 75                             throw new RuntimeException(
 76                                     "receive length is: " + len +
 77                                             ", send length: " + packetLength +
 78                                             ", buffer length: " + buf.length);
 79                     }
 80                 }
 81                 return;
 82             } catch (Exception e) {
 83                 e.printStackTrace();
 84                 throw new RuntimeException("caught: " + e);
 85             } finally {
 86                 if (server != null) { server.close(); }
 87             }
 88         }
 89     }
 90 
 91     static class ClientThread extends Thread {
 92 
 93         int serverPort;
 94         DatagramSocket client;
 95         InetAddress host;
 96 
 97         ClientThread(int serverPort)throws IOException {
 98             this.serverPort = serverPort;
 99             this.host = InetAddress.getLocalHost();
100             this.client = new DatagramSocket();
101         }
102 
103         public void run() {
104             try {
105                 System.err.println("started client thread: " + client);
106                 byte[] buf = new byte[bufferLength];
107                 DatagramPacket sendPacket =
108                     new DatagramPacket(buf, packetLength,
109                                        host, serverPort);
110                 for (int i = 0; i < 10; i++) {
111                     client.send(sendPacket);
112                 }
113                 System.err.println("sent 10 packets");
114                 return;
115             } catch (Exception e) {
116                 e.printStackTrace();
117                 throw new RuntimeException("caught: " + e);
118             } finally {
119                 if (client != null) { client.close(); }
120             }
121         }
122     }
123 }
