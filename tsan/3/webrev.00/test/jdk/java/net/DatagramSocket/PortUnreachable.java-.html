 24 /**
 25  * @test
 26  * @bug 4361783
 27  * @summary  Test to see if ICMP Port Unreachable on non-connected
 28  *           DatagramSocket causes a SocketException "socket closed"
 29  *           exception on Windows 2000.
 30  */
 31 import java.net.BindException;
 32 import java.net.InetAddress;
 33 import java.net.DatagramSocket;
 34 import java.net.DatagramPacket;
 35 
 36 public class PortUnreachable {
 37 
 38     DatagramSocket clientSock;
 39     int serverPort;
 40     int clientPort;
 41 
 42     public void serverSend() {
 43         try {
 44             InetAddress addr = InetAddress.getLocalHost();
 45             Thread.currentThread().sleep(1000);
 46             // send a delayed packet which should mean a delayed icmp
 47             // port unreachable
 48             byte b[] = "A late msg".getBytes();
 49             DatagramPacket packet = new DatagramPacket(b, b.length, addr,
 50                                                        serverPort);
 51             clientSock.send(packet);
 52 
 53             DatagramSocket sock = recreateServerSocket(serverPort);
 54             b = "Greetings from the server".getBytes();
 55             packet = new DatagramPacket(b, b.length, addr, clientPort);
 56             sock.send(packet);
 57             sock.close();
 58         } catch (Exception e) {
 59             e.printStackTrace();
 60         }
 61     }
 62 
 63     DatagramSocket recreateServerSocket (int serverPort) throws Exception {
 64         DatagramSocket serverSocket = null;
 65         int retryCount = 0;
 66         System.out.println("Attempting to recreate server socket with port: " +
 67                 serverPort);
 68         while (serverSocket == null) {
 69             try {
 70                 serverSocket = new DatagramSocket(serverPort);
 71             } catch (BindException bEx) {
 72                 if (retryCount++ < 5) {
 73                     Thread.sleep(500);
 74                 } else {
 75                     System.out.println("Give up after 5 retries");
 76                     throw bEx;
 77                 }
 78             }
 79         }
 80 
 81         System.out.println("PortUnreachableTest.recreateServerSocket: returning socket == "
 82                 + serverSocket.getLocalAddress() + ":" + serverSocket.getLocalPort());
 83         return serverSocket;
 84     }
 85 
 86     PortUnreachable() throws Exception {
 87 
 88         clientSock = new DatagramSocket();
 89         clientPort = clientSock.getLocalPort();
 90 
 91     }
 92 
 93     void execute () throws Exception{
 94 
 95         // pick a port for the server
 96         DatagramSocket sock2 = new DatagramSocket();
 97         serverPort = sock2.getLocalPort();
 98 
 99         // send a burst of packets to the unbound port - we should get back
100         // icmp port unreachable messages
101         //
102         InetAddress addr = InetAddress.getLocalHost();
103         byte b[] = "Hello me".getBytes();
104         DatagramPacket packet = new DatagramPacket(b, b.length, addr,
105                                                    serverPort);
106         //close just before sending
107         sock2.close();
108         for (int i=0; i<100; i++)
109             clientSock.send(packet);
110 
111         serverSend();
112         // try to receive
113         b = new byte[25];
114         packet = new DatagramPacket(b, b.length, addr, serverPort);
115         clientSock.setSoTimeout(10000);
116         clientSock.receive(packet);
117         System.out.println("client received data packet " + new String(packet.getData()));
118 
119         // done
120         clientSock.close();
121     }
122 
123     public static void main(String[] args) throws Exception {
124         PortUnreachable test = new PortUnreachable();
125         test.execute();
126     }
127 
128 }
