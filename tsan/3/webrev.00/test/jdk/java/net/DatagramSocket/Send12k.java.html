24 /**
25  * @test
26  * @bug 4094894
27  * @summary On W95/W98 it's not possible to send a datagram >12k
28  *          via the loopback address.
29  * @run main Send12k
30  * @run main/othervm -Djava.net.preferIPv6Addresses=true Send12k
31  */
32 
33 import java.net.*;
34 import java.io.*;
35 
36 public class Send12k {
37 
38 
39     public static void main(String args[]) throws Exception {
40 
41          int SEND_SIZE=0;
42 
43          if(System.getProperty("os.name").contains("Mac")) {
44              SEND_SIZE = 16 * 576;
45          } else {
46              SEND_SIZE = 16 * 1024;
47          }
48         InetAddress localHost = InetAddress.getLocalHost();
49         DatagramSocket s1 = new DatagramSocket();
50         DatagramSocket s2 = new DatagramSocket(0, localHost);
51 
52         byte b1[] = new byte[ SEND_SIZE ];
53         DatagramPacket p1 = new DatagramPacket(b1, 0, b1.length,
54                                                localHost,
55                                                s2.getLocalPort());
56         boolean sendOkay = true;
57         try {
58             System.out.println("Sending to: [" + localHost + "]:" + s2.getLocalPort());
59             s1.send(p1);
60         } catch (IOException e) {
61             /*
62              * Prior to merlin a send of > 12k to loopback address
63              * would fail silently.
64              */
65             System.out.println("Sending failed: " + e);
66             sendOkay = false;
67         }
68 
69         if (sendOkay) {
70             byte b2[] = new byte[ SEND_SIZE * 2];
71             DatagramPacket p2 = new DatagramPacket( b2, SEND_SIZE * 2 );
72             s2.setSoTimeout(2000);
73 
74             try {
75                 s2.receive(p1);
76             } catch (InterruptedIOException ioe) {
77                 throw new Exception("Datagram not received within timeout");
78             }
79 
80             if (p1.getLength() != SEND_SIZE) {
81                 throw new Exception("Received datagram incorrect size");
82             }
83         }
84 
85     }
86 
87 }
