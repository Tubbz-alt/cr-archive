24 /**
25  * @test
26  * @bug 4163126 8222829
27  * @summary Test to see if timeout hangs. Also checks that
28  * negative timeout value fails as expected.
29  * @run testng DatagramTimeout
30  */
31 import java.net.DatagramPacket;
32 import java.net.DatagramSocket;
33 import java.net.SocketTimeoutException;
34 import java.nio.channels.DatagramChannel;
35 
36 import org.testng.annotations.Test;
37 
38 import static org.testng.Assert.expectThrows;
39 
40 
41 public class DatagramTimeout {
42     private static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
43     private static final Class<SocketTimeoutException> STE = SocketTimeoutException.class;

44 
45     /**
46      * Test DatagramSocket setSoTimeout with a valid timeout value.
47      */
48     @Test
49     public void testSetTimeout() throws Exception {
50         try (DatagramSocket s = new DatagramSocket()) {
51             byte[] buffer = new byte[50];
52             DatagramPacket p = new DatagramPacket(buffer, buffer.length);
53             s.setSoTimeout(2);
54             expectThrows(STE, () -> s.receive(p));
55         }
56     }
57 
58     /**
59      * Test DatagramSocket setSoTimeout with a negative timeout.
60      */
61     @Test
62     public void testSetNegativeTimeout() throws Exception {
63         try (DatagramSocket s = new DatagramSocket()) {
64             expectThrows(IAE, () -> s.setSoTimeout(-1));
65         }


66     }
67 
68     /**
69      * Test DatagramSocketAdaptor setSoTimeout with a negative timeout.
70      */
71     @Test
72     public void testNegativeTimeout() throws Exception {
73         try (DatagramChannel dc = DatagramChannel.open()) {
74             var s = dc.socket();
75             expectThrows(IAE, () -> s.setSoTimeout(-1));
76         }
77     }
78 }
