24 /**
25  * @test
26  * @bug 4163126
27  * @summary  test to see if timeout hangs
28  * @run main/timeout=15 DatagramTimeout
29  */
30 import java.net.DatagramPacket;
31 import java.net.DatagramSocket;
32 import java.net.SocketTimeoutException;
33 
34 public class DatagramTimeout {
35     public static void main(String[] args) throws Exception {
36         boolean success = false;
37         DatagramSocket sock = new DatagramSocket();
38 
39         try {
40             DatagramPacket p;
41             byte[] buffer = new byte[50];
42             p = new DatagramPacket(buffer, buffer.length);
43             sock.setSoTimeout(2);
44             sock.receive(p);
45         } catch (SocketTimeoutException e) {
46             success = true;
47         } finally {
48             sock.close();
49         }
50         if (!success)
51             throw new RuntimeException("Socket timeout failure.");
52     }
53 
54 }
