 43 /*
 44  * @test
 45  * @bug 8236105
 46  * @summary check that DatagramSocket, MulticastSocket,
 47  *          DatagramSocketAdaptor and DatagramChannel all
 48  *          throw expected Execption when passed a DatagramPacket
 49  *          with invalid details

 53 public class SendCheck {
 54     static final Class<IOException> IOE = IOException.class;
 55     static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
 56 
 57     static final byte[] buf = {0, 1, 2};
 58     static DatagramSocket socket;
 59 
 60     @BeforeTest
 61     public void setUp() {
 62         try {
 63             socket = new DatagramSocket();
 64         } catch (Exception e) {
 65             throw new ExceptionInInitializerError(e);
 66         }
 67     }
 68 
 69     @AfterTest
 70     public void closeDown() {
 71         socket.close();
 72     }

 74     static final class Packet {
 75         private Packet(String description, DatagramPacket packet) {
 76             this.description = description;
 77             this.packet = packet;
 78         }
 79 
 80         final String description;
 81         final DatagramPacket packet;
 82 
 83         public String toString() {
 84             return description;
 85         }
 86 
 87         public static Packet of(DatagramPacket packet) {
 88             InetAddress address = packet.getAddress();
 89             int port = packet.getPort();
 90             String description;
 91             if (address == null) {
 92                 description = "<null>:" + port;
 93             } else if (port < 0) {
 94                 description = packet.getAddress().toString() + ":" + port;
 95             } else {
 96                 description = packet.getSocketAddress().toString();
 97             }
 98             return new Packet(description, packet);
 99         }
100     }
101 
102     @DataProvider(name = "packets")
103     static Object[][] providerIO() throws IOException {
104         var wildcard = new InetSocketAddress(0);
105 
106         /*
107         Commented until JDK-8236852 is fixed
108 
109         // loopback w/port 0 -- DC, DSA, MS, DS throws IO
110         var pkt1 = new DatagramPacket(buf, 0, buf.length);
111         pkt1.setAddress(InetAddress.getLoopbackAddress());
112         pkt1.setPort(0);
113          */
114 
115         /*
116         Commented until JDK-8236852 is fixed
117 
118         // wildcard w/port 0 -- DC, DSA, MS, DS throws IO
119         var pkt2 = new DatagramPacket(buf, 0, buf.length);
120         pkt2.setAddress(wildcard.getAddress());
121         pkt2.setPort(0);
122         */
123 
124         // loopback w/port -1 -- DC, DSA, MS, DS throws IAE
125         var pkt3 = new DatagramPacket(buf, 0, buf.length);
126         pkt3.setAddress(InetAddress.getLoopbackAddress());
127 
128         // wildcard w/port -1 -- DC, DSA, MS, DS throws IAE
129         var pkt4 = new DatagramPacket(buf, 0, buf.length);
130         pkt4.setAddress(wildcard.getAddress());
131 
132         /*
133         Commented until JDK-8236807 is fixed
134 
135         // wildcard addr w/valid port -- DS, MS throws IO ; DC, DSA doesn't throw
136         var pkt5 = new DatagramPacket(buf, 0, buf.length);
137         var addr1 = wildcard.getAddress();
138         pkt5.setAddress(addr1);
139         pkt5.setPort(socket.getLocalPort());
140         */
141 
142         // PKTS 3 & 4: invalid port -1
143         List<Packet> iaePackets = List.of(Packet.of(pkt3), Packet.of(pkt4));
144 
145         List<Sender> senders = List.of(
146                 Sender.of(new DatagramSocket(null)),
147                 Sender.of(new MulticastSocket(null), (byte) 0),
148                 Sender.of(DatagramChannel.open()),
149                 Sender.of(DatagramChannel.open().socket())
150         );
151 
152         List<Object[]> testcases = new ArrayList<>();
153         for (var p : iaePackets) {
154             addTestCaseFor(testcases, senders, p, IAE);
155         }
156 
157         return testcases.toArray(new Object[0][0]);
158     }
159 
160     static void addTestCaseFor(List<Object[]> testcases,
161                                List<Sender> senders, Packet p,
162                                Class<? extends Throwable> exception) {
163         for (var s : senders) {
164             Object[] testcase = new Object[]{s, p, exception};
165             testcases.add(testcase);
166         }
167     }
168 
169     @Test(dataProvider = "packets")
170     public static void channelSendCheck(Sender<IOException> sender,
171                                         Packet packet,
172                                         Class<? extends Throwable> exception) {
173         DatagramPacket pkt = packet.packet;
174         if (exception != null) {
175             Throwable t = expectThrows(exception, () -> sender.send(pkt));
176             System.out.printf("%s got expected exception %s%n", packet.toString(), t);
177         } else {
178             try {
179                 sender.send(pkt);
180             } catch (IOException x) {
181                 throw new AssertionError("Unexpected exception for " + sender + " / " + packet, x);
182             }
183         }
184     }
185 
186     interface Sender<E extends Exception> extends AutoCloseable {
187         void send(DatagramPacket p) throws E;
188 
189         void close() throws E;
190 
191         static Sender<IOException> of(DatagramSocket socket) {
192             return new SenderImpl<>(socket, socket::send, socket::close);
193         }
194 
195         static Sender<IOException> of(MulticastSocket socket, byte ttl) {
196             SenderImpl.Send<IOException> send =
197                     (pkt) -> socket.send(pkt, ttl);
198             return new SenderImpl<>(socket, send, socket::close);
199         }
200 
201         static Sender<IOException> of(DatagramChannel socket) {
202             SenderImpl.Send<IOException> send =
203                     (pkt) -> socket.send(ByteBuffer.wrap(pkt.getData()),
204                             pkt.getSocketAddress());
205             return new SenderImpl<>(socket, send, socket::close);
206         }
207     }
208 
209     static final class SenderImpl<E extends Exception> implements Sender<E> {
210         @FunctionalInterface
211         interface Send<E extends Exception> {
212             void send(DatagramPacket p) throws E;
213         }
214 
215         @FunctionalInterface
216         interface Closer<E extends Exception> {
217             void close() throws E;
218         }
219 
220         private final Send<E> send;
221         private final Closer<E> closer;
222         private final Object socket;
223 
224         public SenderImpl(Object socket, Send<E> send, Closer<E> closer) {
225             this.socket = socket;
226             this.send = send;
227             this.closer = closer;
228         }
229 
230         @Override
231         public void send(DatagramPacket p) throws E {
232             send.send(p);
233         }
234 
235         @Override
236         public void close() throws E {
237             closer.close();
238         }
239 
240         @Override
241         public String toString() {
242             return socket.getClass().getSimpleName();
243         }
244     }
