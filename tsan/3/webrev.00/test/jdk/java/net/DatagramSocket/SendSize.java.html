 24 /* @test
 25  *
 26  * @bug 4095393
 27  * @test main SendSize
 28  * @summary this tests a regression in 1.2, beta 2 and earlier where
 29  * the DatagramPackets sent the entire buffer, not the buffer length
 30  * of the packet.
 31  *
 32  * @author Benjamin Renaud
 33  */
 34 
 35 import java.io.IOException;
 36 import java.net.DatagramPacket;
 37 import java.net.DatagramSocket;
 38 import java.net.InetAddress;
 39 import java.net.InetSocketAddress;
 40 
 41 public class SendSize {
 42     static final int bufferLength = 512;
 43     static final int packetLength = 256;
 44 
 45     public static void main(String[] args) throws Exception {
 46         DatagramSocket serverSocket = new DatagramSocket(
 47                 new InetSocketAddress(InetAddress.getLocalHost(), 0)
 48         );
 49         Thread server = new ServerThread(serverSocket);
 50         server.start();
 51         Thread client = new ClientThread(serverSocket.getLocalPort());
 52         client.start();
 53         server.join();
 54         client.join();
 55     }
 56 
 57     static class ServerThread extends Thread {
 58         DatagramSocket server;
 59 
 60         ServerThread(DatagramSocket server) {
 61             this.server = server;
 62         }
 63 
 64         public void run() {
 65             try {
 66                 System.err.println("started server thread: " + server);
 67                 byte[] buf = new byte[1024];
 68                 for (int i = 0; i < 10; i++) {
 69                     DatagramPacket receivePacket = new DatagramPacket(buf,
 70                             buf.length);
 71                     server.receive(receivePacket);
 72                     int len = receivePacket.getLength();
 73                     switch (len) {
 74                         case packetLength:
 75                             System.out.println("receive length is: " + len);
 76                             break;
 77                         default:
 78                             throw new RuntimeException(
 79                                     "receive length is: " + len +
 80                                             ", send length: " + packetLength +
 81                                             ", buffer length: " + buf.length);
 82                     }
 83                 }
 84                 return;
 85             } catch (Exception e) {
 86                 e.printStackTrace();
 87                 throw new RuntimeException("caught: " + e);
 88             } finally {
 89                 if (server != null) { server.close(); }
 90             }
 91         }
 92     }
 93 
 94     static class ClientThread extends Thread {
 95 
 96         int serverPort;
 97         DatagramSocket client;
 98         InetAddress host;
 99 
100         ClientThread(int serverPort)throws IOException {
101             this.serverPort = serverPort;
102             this.host = InetAddress.getLocalHost();
103             this.client = new DatagramSocket();
104         }
105 
106         public void run() {
107             try {
108                 System.err.println("started client thread: " + client);
109                 byte[] buf = new byte[bufferLength];
110                 DatagramPacket sendPacket =
111                     new DatagramPacket(buf, packetLength,
112                                        host, serverPort);
113                 for (int i = 0; i < 10; i++) {
114                     client.send(sendPacket);
115                 }
116                 System.err.println("sent 10 packets");
117                 return;
118             } catch (Exception e) {
119                 e.printStackTrace();
120                 throw new RuntimeException("caught: " + e);
121             } finally {
122                 if (client != null) { client.close(); }
123             }
124         }
125     }
126 }
