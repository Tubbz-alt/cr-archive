 24 /**
 25  * @test
 26  * @bug 4361783
 27  * @key intermittent
 28  * @summary Test to see if ICMP Port Unreachable on non-connected
 29  *          DatagramSocket causes a SocketException "socket closed"
 30  *          exception on Windows 2000.
 31  * @run main/othervm PortUnreachable
 32  */
 33 
 34 import java.net.BindException;


 35 import java.net.DatagramPacket;
 36 import java.net.DatagramSocket;
 37 import java.net.InetAddress;
 38 import java.net.InetSocketAddress;
 39 
 40 public class PortUnreachable {
 41 
 42     DatagramSocket clientSock;
 43     int serverPort;
 44     int clientPort;
 45 
 46     public void serverSend() {
 47         try {
 48             InetAddress addr = InetAddress.getLocalHost();
 49             Thread.sleep(1000);
 50             // send a delayed packet which should mean a delayed icmp
 51             // port unreachable
 52             byte b[] = "A late msg".getBytes();
 53             DatagramPacket packet = new DatagramPacket(b, b.length, addr,
 54                                                        serverPort);
 55             clientSock.send(packet);
 56 
 57             DatagramSocket sock = recreateServerSocket(serverPort);
 58             b = "Greetings from the server".getBytes();
 59             packet = new DatagramPacket(b, b.length, addr, clientPort);
 60             sock.send(packet);
 61             Thread.sleep(500);  // give time to the kernel to send packet
 62             sock.close();
 63         } catch (Exception e) {
 64             e.printStackTrace();
 65         }
 66     }
 67 
 68     DatagramSocket recreateServerSocket (int serverPort) throws Exception {
 69         DatagramSocket serverSocket = null;
 70         int retryCount = 0;
 71         long sleeptime = 0;
 72         System.out.println("Attempting to recreate server socket with port: " +
 73                 serverPort);
 74         // it's possible that this method intermittently fails, if some other
 75         // process running on the machine grabs the port we want before us,
 76         // and doesn't release it before the 10 * 500 ms are elapsed...
 77         while (serverSocket == null) {
 78             try {
 79                 serverSocket = new DatagramSocket(serverPort, InetAddress.getLocalHost());
 80             } catch (BindException bEx) {
 81                 if (retryCount++ < 10) {
 82                     sleeptime += sleepAtLeast(500);
 83                 } else {
 84                     System.out.println("Give up after 10 retries and " + sleeptime(sleeptime));
 85                     System.out.println("Has some other process grabbed port " + serverPort + "?");
 86                     throw bEx;
 87                 }
 88             }
 89         }
 90 
 91         System.out.println("PortUnreachableTest.recreateServerSocket: returning socket == "
 92                 + serverSocket.getLocalAddress() + ":" + serverSocket.getLocalPort()
 93                 + " obtained at " + attempt(retryCount) + " attempt with " + sleeptime(sleeptime));
 94         return serverSocket;
 95     }
 96 
 97     long sleepAtLeast(long millis) throws Exception {
 98         long start = System.nanoTime();
 99         long ms = millis;
100         while (ms > 0) {
101             assert ms < Long.MAX_VALUE/1000_000L;
102             Thread.sleep(ms);
103             long elapsedms = (System.nanoTime() - start)/1000_000L;
104             ms = millis - elapsedms;
105         }
106         return millis - ms;
107     }
108 
109     String attempt(int retry) {
110         switch (retry) {
111             case 0: return "first";
112             case 1: return "second";
113             case 2: return "third";
114             default: return retry + "th";
115         }
116     }
117 
118     String sleeptime(long millis) {
119         if (millis == 0) return "no sleep";
120         long sec = millis / 1000L;
121         long ms =  millis % 1000L;
122         String sleeptime = "";
123         if (millis > 0) {
124            if (sec > 0) {
125                sleeptime = "" + sec + " s" +
126                    (ms > 0 ? " " : "");
127             }
128             if (ms > 0 ) {
129                 sleeptime += ms + " ms";
130             }
131         } else sleeptime = millis + " ms"; // should not happen
132         return sleeptime + " of sleep time";
133     }
134 
135     PortUnreachable() throws Exception {
136         clientSock = new DatagramSocket(new InetSocketAddress(InetAddress.getLocalHost(), 0));
137         clientPort = clientSock.getLocalPort();
138 
139     }
140 
141     void execute () throws Exception{
142 
143         // pick a port for the server
144         DatagramSocket sock2 = new DatagramSocket(new InetSocketAddress(InetAddress.getLocalHost(), 0));
145         serverPort = sock2.getLocalPort();
146 
147         // send a burst of packets to the unbound port - we should get back
148         // icmp port unreachable messages
149         //
150         InetAddress addr = InetAddress.getLocalHost();
151         byte b[] = "Hello me".getBytes();
152         DatagramPacket packet = new DatagramPacket(b, b.length, addr,
153                                                    serverPort);
154         //close just before sending
155         sock2.close();
156         for (int i=0; i<100; i++)
157             clientSock.send(packet);
158 
159         serverSend();
160 
161         // try to receive
162         b = new byte[25];
163         packet = new DatagramPacket(b, b.length, addr, serverPort);
164         clientSock.setSoTimeout(10000);
165         clientSock.receive(packet);
166         System.out.println("client received data packet " + new String(packet.getData()));
167 
168         // done
169         clientSock.close();
170     }
171 
172     public static void main(String[] args) throws Exception {
173         // A BindException might be thrown intermittently. In that case retry
174         // 3 times before propagating the exception to finish execution.
175         int catchCount = 0;
176 
177         while (true) {
178             try {
179                 PortUnreachable test = new PortUnreachable();
180                 test.execute();
181                 return;
182             } catch (BindException bEx) {
183                 if (++catchCount > 3) {
184                     throw bEx;
185                 }
186             }
187         }
188     }
189 }
