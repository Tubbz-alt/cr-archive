 33 import java.lang.management.ManagementFactory;

 54 public class UnreferencedDatagramSockets {
 55 
 56     /**
 57      * The set of sockets we have to check up on.
 58      */
 59     final static ArrayDeque<NamedWeak> pendingSockets = new ArrayDeque<>(5);
 60 
 61     /**
 62      * Queued objects when they are unreferenced.
 63      */
 64     final static ReferenceQueue<Object> pendingQueue = new ReferenceQueue<>();
 65 
 66     // Server to echo a datagram packet
 67     static class Server implements Runnable {
 68 
 69         DatagramSocket ss;
 70 
 71         Server() throws IOException {
 72             ss = new DatagramSocket(0);
 73             System.out.printf("  DatagramServer addr: %s: %d%n",
 74                     this.getHost(), this.getPort());
 75             pendingSockets.add(new NamedWeak(ss, pendingQueue, "serverDatagramSocket"));
 76             extractRefs(ss, "serverDatagramSocket");
 77         }
 78 
 79         InetAddress getHost() throws UnknownHostException {
 80             InetAddress localhost = InetAddress.getByName("localhost"); //.getLocalHost();
 81             return localhost;
 82         }
 83 
 84         int getPort() {
 85             return ss.getLocalPort();
 86         }
 87 
 88         // Receive a byte and send back a byte
 89         public void run() {
 90             try {
 91                 byte[] buffer = new byte[50];
 92                 DatagramPacket p = new DatagramPacket(buffer, buffer.length);
 93                 ss.receive(p);
 94                 buffer[0] += 1;
 95                 ss.send(p);         // send back +1
 96 
 97                 // do NOT close but 'forget' the datagram socket reference
 98                 ss = null;
 99             } catch (Exception ioe) {
100                 ioe.printStackTrace();
101             }
102         }
103     }
104 
105     public static void main(String args[]) throws Exception {
106 
107         // Create and close a DatagramSocket to warm up the FD count for side effects.
108         try (DatagramSocket s = new DatagramSocket(0)) {
109             // no-op; close immediately
110             s.getLocalPort();   // no-op
111         }
112 
113         long fdCount0 = getFdCount();
114         listProcFD();
115 
116         // start a server
117         Server svr = new Server();
118         Thread thr = new Thread(svr);
119         thr.start();
120 
121         DatagramSocket client = new DatagramSocket(0);
122         client.connect(svr.getHost(), svr.getPort());
123         pendingSockets.add(new NamedWeak(client, pendingQueue, "clientDatagramSocket"));
124         extractRefs(client, "clientDatagramSocket");
125 
126         byte[] msg = new byte[1];
127         msg[0] = 1;
128         DatagramPacket p = new DatagramPacket(msg, msg.length, svr.getHost(), svr.getPort());
129         client.send(p);
130 
131         p = new DatagramPacket(msg, msg.length);
132         client.receive(p);
133 
134         System.out.printf("echo received from: %s%n", p.getSocketAddress());
135         if (msg[0] != 2) {
136             throw new AssertionError("incorrect data received: expected: 2, actual: " + msg[0]);
137         }
138 
139         // Do NOT close the DatagramSocket; forget it
140 
141         Object ref;
142         int loops = 20;
143         while (!pendingSockets.isEmpty() && loops-- > 0) {
144             ref = pendingQueue.remove(1000L);
145             if (ref != null) {
146                 pendingSockets.remove(ref);
147                 System.out.printf("  ref freed: %s, remaining: %d%n", ref, pendingSockets.size());
148             } else {
149                 client = null;
150                 p = null;
151                 msg = null;
152                 System.gc();
153             }
154         }
155 
156         thr.join();
157 
158         // List the open file descriptors
159         long fdCount = getFdCount();
160         System.out.printf("Initial fdCount: %d, final fdCount: %d%n", fdCount0, fdCount);
161         listProcFD();
162 
163         if (loops == 0) {
164             throw new AssertionError("Not all references reclaimed");
165         }
166     }
167 
168     // Get the count of open file descriptors, or -1 if not available
169     private static long getFdCount() {
170         OperatingSystemMXBean mxBean = ManagementFactory.getOperatingSystemMXBean();
171         return (mxBean instanceof UnixOperatingSystemMXBean)
172                 ? ((UnixOperatingSystemMXBean) mxBean).getOpenFileDescriptorCount()
173                 : -1L;
174     }
175 
176     // Reflect to find references in the datagram implementation that will be gc'd
177     private static void extractRefs(DatagramSocket s, String name) {
178         try {
179 
180             Field socketImplField = DatagramSocket.class.getDeclaredField("impl");
181             socketImplField.setAccessible(true);
182             Object socketImpl = socketImplField.get(s);
183 
184             Field fileDescriptorField = DatagramSocketImpl.class.getDeclaredField("fd");
185             fileDescriptorField.setAccessible(true);
186             FileDescriptor fileDescriptor = (FileDescriptor) fileDescriptorField.get(socketImpl);
187             extractRefs(fileDescriptor, name);
188 
189             Class<?> socketImplClass = socketImpl.getClass();
190             System.out.printf("socketImplClass: %s%n", socketImplClass);
191             if (socketImplClass.getName().equals("java.net.TwoStacksPlainDatagramSocketImpl")) {
192                 Field fileDescriptor1Field = socketImplClass.getDeclaredField("fd1");
193                 fileDescriptor1Field.setAccessible(true);
194                 FileDescriptor fileDescriptor1 = (FileDescriptor) fileDescriptor1Field.get(socketImpl);
195                 extractRefs(fileDescriptor1, name + "::twoStacksFd1");
196 
197             } else {
198                 System.out.printf("socketImpl class name not matched: %s != %s%n",
199                         socketImplClass.getName(), "java.net.TwoStacksPlainDatagramSocketImpl");
200             }
201         } catch (NoSuchFieldException | IllegalAccessException ex) {
202             ex.printStackTrace();
203             throw new AssertionError("missing field", ex);
204         }
205     }
206 
207     private static void extractRefs(FileDescriptor fileDescriptor, String name) {
208         Object cleanup = null;
209         int rawfd = -1;
210         try {
211             if (fileDescriptor != null) {
212                 Field fd1Field = FileDescriptor.class.getDeclaredField("fd");
213                 fd1Field.setAccessible(true);
214                 rawfd = fd1Field.getInt(fileDescriptor);
215 
216                 Field cleanupfdField = FileDescriptor.class.getDeclaredField("cleanup");
217                 cleanupfdField.setAccessible(true);
218                 cleanup = cleanupfdField.get(fileDescriptor);
219                 pendingSockets.add(new NamedWeak(fileDescriptor, pendingQueue,
220                         name + "::fileDescriptor: " + rawfd));
221                 pendingSockets.add(new NamedWeak(cleanup, pendingQueue, name + "::fdCleanup: " + rawfd));
222 
223             }
224         } catch (NoSuchFieldException | IllegalAccessException ex) {
225             ex.printStackTrace();
226             throw new AssertionError("missing field", ex);
227         } finally {
228             System.out.print(String.format("  %s:: fd: %s, fd: %d, cleanup: %s%n",
229                     name, fileDescriptor, rawfd, cleanup));
230         }
231     }

233     /**
234      * Method to list the open file descriptors (if supported by the 'lsof' command).
235      */
236     static void listProcFD() {
237         List<String> lsofDirs = List.of("/usr/bin", "/usr/sbin");
238         Optional<Path> lsof = lsofDirs.stream()
239                 .map(s -> Paths.get(s, "lsof"))
240                 .filter(f -> Files.isExecutable(f))
241                 .findFirst();
242         lsof.ifPresent(exe -> {
243             try {
244                 System.out.printf("Open File Descriptors:%n");
245                 long pid = ProcessHandle.current().pid();
246                 ProcessBuilder pb = new ProcessBuilder(exe.toString(), "-p", Integer.toString((int) pid));
247                 pb.inheritIO();
248                 Process p = pb.start();
249                 p.waitFor(10, TimeUnit.SECONDS);
250             } catch (IOException | InterruptedException ie) {
251                 ie.printStackTrace();
252             }
253         });
254     }

256     // Simple class to identify which refs have been queued
257     static class NamedWeak extends WeakReference<Object> {
258         private final String name;
259 
260         NamedWeak(Object o, ReferenceQueue<Object> queue, String name) {
261             super(o, queue);
262             this.name = name;
263         }
264 
265         public String toString() {
266             return name + "; " + super.toString();
267         }
268     }
