 24 /**
 25  * @test
 26  * @bug 4380568 7095949
 27  * @library /test/lib
 28  * @summary  HttpURLConnection does not support 307 redirects
 29  */
 30 import java.io.*;
 31 import java.net.*;
 32 import jdk.test.lib.net.URIBuilder;
 33 import static java.net.Proxy.NO_PROXY;
 34 
 35 class RedirServer extends Thread {
 36 
 37     static final int TIMEOUT = 10 * 1000;
 38 
 39     ServerSocket ss;
 40     int port;
 41 
 42     String reply1Part1 = "HTTP/1.1 307 Temporary Redirect\r\n" +
 43         "Date: Mon, 15 Jan 2001 12:18:21 GMT\r\n" +
 44         "Server: Apache/1.3.14 (Unix)\r\n" +
 45         "Location: http://localhost:";
 46     String reply1Part2 = "/redirected.html\r\n" +
 47         "Connection: close\r\n" +
 48         "Content-Type: text/html; charset=iso-8859-1\r\n\r\n" +
 49         "<html>Hello</html>";
 50 
 51     RedirServer (ServerSocket ss) throws IOException {
 52         this.ss = ss;
 53         this.ss.setSoTimeout(TIMEOUT);
 54         port = this.ss.getLocalPort();
 55     }
 56 
 57     String reply2 = "HTTP/1.1 200 Ok\r\n" +
 58         "Date: Mon, 15 Jan 2001 12:18:21 GMT\r\n" +
 59         "Server: Apache/1.3.14 (Unix)\r\n" +
 60         "Connection: close\r\n" +
 61         "Content-Type: text/html; charset=iso-8859-1\r\n\r\n" +
 62         "World";
 63 
 64     static final byte[] requestEnd = new byte[] {'\r', '\n', '\r', '\n' };
 65 
 66     // Read until the end of a HTTP request
 67     void readOneRequest(InputStream is) throws IOException {
 68         int requestEndCount = 0, r;
 69         while ((r = is.read()) != -1) {
 70             if (r == requestEnd[requestEndCount]) {
 71                 requestEndCount++;
 72                 if (requestEndCount == 4) {
 73                     break;
 74                 }
 75             } else {
 76                 requestEndCount = 0;
 77             }
 78         }
 79     }
 80 
 81     public void run () {
 82         try {
 83             try (Socket s = ss.accept()) {
 84                 s.setSoTimeout(TIMEOUT);
 85                 readOneRequest(s.getInputStream());
 86                 String reply = reply1Part1 + port + reply1Part2;
 87                 s.getOutputStream().write(reply.getBytes());
 88             }
 89 
 90             /* wait for redirected connection */
 91             try (Socket s = ss.accept()) {
 92                 s.setSoTimeout(TIMEOUT);
 93                 readOneRequest(s.getInputStream());
 94                 s.getOutputStream().write(reply2.getBytes());
 95             }
 96         } catch (Exception e) {
 97             e.printStackTrace();
 98         } finally {
 99             try { ss.close(); } catch (IOException unused) {}
100         }
101     }
102 };
103 
104 public class Redirect307Test {
105     public static void main(String[] args) throws Exception {
106         ServerSocket sock = new ServerSocket(0, 0, InetAddress.getLoopbackAddress());
107         int port = sock.getLocalPort();
108         RedirServer server = new RedirServer(sock);
109         server.start();
110 
111         URL url = URIBuilder.newBuilder()
112                 .scheme("http")
113                 .loopback()
114                 .port(port)
115                 .toURL();
116         URLConnection conURL =  url.openConnection(NO_PROXY);
117         conURL.setDoInput(true);
118         conURL.setAllowUserInteraction(false);
119         conURL.setUseCaches(false);
120 
121         try (InputStream in = conURL.getInputStream()) {
122             if ((in.read() != (int)'W') || (in.read()!=(int)'o')) {
123                 throw new RuntimeException ("Unexpected string read");
124             }
125         }
126     }
127 }
