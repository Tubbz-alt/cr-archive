24 /*
25  * @test
26  * @bug 4389976
27  * @library /test/lib
28  * @summary    can't unblock read() of InputStream from URL connection
29  * @run main/timeout=40/othervm -Dsun.net.client.defaultReadTimeout=2000 TimeoutTest
30  */
31 
32 import java.io.*;
33 import java.net.*;
34 import jdk.test.lib.net.URIBuilder;
35 
36 public class TimeoutTest {
37 
38     class Server extends Thread {
39         ServerSocket server;
40         Server (ServerSocket server) {
41             super ();
42             this.server = server;
43         }
44         public void run () {
45             try {
46                 Socket s = server.accept ();
47                 while (!finished ()) {
48                     Thread.sleep (1000);
49                 }
50                 s.close();
51             } catch (Exception e) {
52             }
53         }
54         boolean isFinished = false;
55 
56         synchronized boolean finished () {
57             return (isFinished);
58         }
59         synchronized void done () {
60             isFinished = true;
61         }
62     }
63 
64     public static void main(String[] args) throws Exception {
65         TimeoutTest t = new TimeoutTest ();
66         t.test ();
67     }
68 
69     public void test() throws Exception {
70         ServerSocket ss = new ServerSocket();
71         ss.bind(new InetSocketAddress(InetAddress.getLoopbackAddress(), 0));
72         Server s = new Server (ss);
73         try{
74             URL url = URIBuilder.newBuilder()
75                 .scheme("http")
76                 .loopback()
77                 .port(ss.getLocalPort())
78                 .toURL();
79             System.out.println("URL: " + url);
80             URLConnection urlc = url.openConnection ();
81             InputStream is = urlc.getInputStream ();
82             throw new RuntimeException("Should have received timeout");
83         } catch (SocketTimeoutException e) {
84             return;
85         } finally {
86             s.done();
87             ss.close();
88         }
89     }
90 }
