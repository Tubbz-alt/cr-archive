 51 public class AcceptCauseFileDescriptorLeak {
 52     private static final int REPS = 2048;
 53     private static final int THRESHOLD = 1024;
 54 
 55     public static void main(String[] args) throws Exception {
 56         if (args.length != 0) {
 57             OutputAnalyzer analyzer = execCmd("ulimit -n -H");
 58             String output = analyzer.getOutput();
 59             if (output == null || output.length() == 0) {
 60                 throw new RuntimeException("\"ulimit -n -H\" output nothing"
 61                         + " and its exit code is " + analyzer.getExitValue());
 62             } else {
 63                 output = output.trim();
 64                 // Set max open file descriptors to 1024
 65                 // if it is unlimited or greater than 1024,
 66                 // otherwise just do test directly
 67                 if ("unlimited".equals(output)
 68                         || Integer.valueOf(output) > THRESHOLD) {
 69                     analyzer = execCmd("ulimit -n " + THRESHOLD + "; "
 70                             + composeJavaTestStr());
 71                     System.out.println("Output: ["
 72                             + analyzer.getOutput() + "]");
 73                     int rc = analyzer.getExitValue();
 74                     if (rc != 0) {
 75                         throw new RuntimeException(
 76                                 "Unexpected exit code: " + rc);
 77                     }
 78                     return;
 79                 }
 80             }
 81         }
 82 
 83         final ServerSocket ss = new ServerSocket(0) {
 84             public Socket accept() throws IOException {
 85                 Socket s = new Socket() {
 86                 };
 87                 s.setSoTimeout(10000);
 88                 implAccept(s);
 89                 return s;
 90             }
 91         };

 92         Thread t = new Thread(new Runnable() {
 93             public void run() {

 94                 try {
 95                     for (int i = 0; i < REPS; i++) {
 96                         (new Socket("localhost", ss.getLocalPort())).close();
 97                     }
 98                 } catch (IOException e) {
 99                     e.printStackTrace();


100                 }
101             }
102         });
103         t.start();

104         try {
105             for (int i = 0; i < REPS; i++) {
106                 ss.accept().close();
107             }
108         } finally {

109             ss.close();
110         }
111         t.join();
112     }

123         System.out.println("Executing: " + cmd);
124         ProcessBuilder pb = new ProcessBuilder(cmd);

 57 public class AcceptCauseFileDescriptorLeak {
 58     private static final int REPS = 2048;
 59     private static final int THRESHOLD = 1024;
 60 
 61     public static void main(String[] args) throws Exception {
 62         if (args.length != 0) {
 63             OutputAnalyzer analyzer = execCmd("ulimit -n -H");
 64             String output = analyzer.getOutput();
 65             if (output == null || output.length() == 0) {
 66                 throw new RuntimeException("\"ulimit -n -H\" output nothing"
 67                         + " and its exit code is " + analyzer.getExitValue());
 68             } else {
 69                 output = output.trim();
 70                 // Set max open file descriptors to 1024
 71                 // if it is unlimited or greater than 1024,
 72                 // otherwise just do test directly
 73                 if ("unlimited".equals(output)
 74                         || Integer.valueOf(output) > THRESHOLD) {
 75                     analyzer = execCmd("ulimit -n " + THRESHOLD + "; "
 76                             + composeJavaTestStr());
 77                     System.out.println("Output: ["
 78                             + analyzer.getOutput() + "]");
 79                     int rc = analyzer.getExitValue();
 80                     if (rc != 0) {
 81                         throw new RuntimeException(
 82                                 "Unexpected exit code: " + rc);
 83                     }
 84                     return;
 85                 }
 86             }
 87         }
 88 
 89         final ServerSocket ss = new ServerSocket() {
 90             public Socket accept() throws IOException {
 91                 Socket s = new Socket() {
 92                 };
 93                 s.setSoTimeout(10000);
 94                 implAccept(s);
 95                 return s;
 96             }
 97         };
 98         ss.bind(new InetSocketAddress(InetAddress.getLoopbackAddress(), 0));
 99         Thread t = new Thread(new Runnable() {
100             public void run() {
101                 int repsCompleted = 0;
102                 try {
103                     for (; repsCompleted < REPS; repsCompleted++) {
104                         (new Socket(InetAddress.getLoopbackAddress(), ss.getLocalPort())).close();
105                     }
106                 } catch (IOException e) {
107                     e.printStackTrace();
108                 } finally {
109                     System.out.println("Client iterations completed:" + repsCompleted);
110                 }
111             }
112         });
113         t.start();
114         int repsCompleted = 0;
115         try {
116             for (; repsCompleted < REPS; repsCompleted++) {
117                 ss.accept().close();
118             }
119         } finally {
120             System.out.println("Server iterations completed:" + repsCompleted);
121             ss.close();
122         }
123         t.join();
124     }
