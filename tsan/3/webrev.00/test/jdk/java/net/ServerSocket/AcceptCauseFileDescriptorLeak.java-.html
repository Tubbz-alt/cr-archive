 24 /*
 25  * author Edward Wang
 26  *
 27  * @test
 28  * @bug 6368984
 29  * @summary Configuring unconnected Socket before passing to implAccept
 30  *          can cause fd leak
 31  * @requires (os.family != "windows")
 32  * @library /test/lib
 33  * @build jdk.test.lib.Utils
 34  *        jdk.test.lib.Asserts
 35  *        jdk.test.lib.JDKToolFinder
 36  *        jdk.test.lib.JDKToolLauncher
 37  *        jdk.test.lib.Platform
 38  *        jdk.test.lib.process.*
 39  *        AcceptCauseFileDescriptorLeak
 40  * @run main/othervm AcceptCauseFileDescriptorLeak root
 41  */
 42 
 43 import java.io.IOException;
 44 import java.net.ServerSocket;
 45 import java.net.Socket;
 46 import java.util.List;
 47 
 48 import jdk.test.lib.JDKToolFinder;
 49 import jdk.test.lib.process.OutputAnalyzer;
 50 
 51 public class AcceptCauseFileDescriptorLeak {
 52     private static final int REPS = 2048;
 53     private static final int THRESHOLD = 1024;
 54 
 55     public static void main(String[] args) throws Exception {
 56         if (args.length != 0) {
 57             OutputAnalyzer analyzer = execCmd("ulimit -n -H");
 58             String output = analyzer.getOutput();
 59             if (output == null || output.length() == 0) {
 60                 throw new RuntimeException("\"ulimit -n -H\" output nothing"
 61                         + " and its exit code is " + analyzer.getExitValue());
 62             } else {
 63                 output = output.trim();
 64                 // Set max open file descriptors to 1024
 65                 // if it is unlimited or greater than 1024,
 66                 // otherwise just do test directly
 67                 if ("unlimited".equals(output)
 68                         || Integer.valueOf(output) > THRESHOLD) {
 69                     analyzer = execCmd("ulimit -n " + THRESHOLD + "; "
 70                             + composeJavaTestStr());
 71                     System.out.println("Output: ["
 72                             + analyzer.getOutput() + "]");
 73                     int rc = analyzer.getExitValue();
 74                     if (rc != 0) {
 75                         throw new RuntimeException(
 76                                 "Unexpected exit code: " + rc);
 77                     }
 78                     return;
 79                 }
 80             }
 81         }
 82 
 83         final ServerSocket ss = new ServerSocket(0) {
 84             public Socket accept() throws IOException {
 85                 Socket s = new Socket() {
 86                 };
 87                 s.setSoTimeout(10000);
 88                 implAccept(s);
 89                 return s;
 90             }
 91         };
 92         Thread t = new Thread(new Runnable() {
 93             public void run() {
 94                 try {
 95                     for (int i = 0; i < REPS; i++) {
 96                         (new Socket("localhost", ss.getLocalPort())).close();
 97                     }
 98                 } catch (IOException e) {
 99                     e.printStackTrace();
100                 }
101             }
102         });
103         t.start();
104         try {
105             for (int i = 0; i < REPS; i++) {
106                 ss.accept().close();
107             }
108         } finally {
109             ss.close();
110         }
111         t.join();
112     }
113 
114     /**
115      * Execute command with shell
116      *
117      * @param  command
118      * @return OutputAnalyzer
119      * @throws IOException
120      */
121     static OutputAnalyzer execCmd(String command) throws IOException {
122         List<String> cmd = List.of("sh", "-c", command);
123         System.out.println("Executing: " + cmd);
124         ProcessBuilder pb = new ProcessBuilder(cmd);
125         return new OutputAnalyzer(pb.start());
126     }
127 
128     static String composeJavaTestStr() {
129         return JDKToolFinder.getTestJDKTool("java") + " "
130                 + AcceptCauseFileDescriptorLeak.class.getName();
131     }
132 }
