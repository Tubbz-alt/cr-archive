24 import java.io.IOException;
25 import java.io.UncheckedIOException;
26 import java.net.InetAddress;
27 import java.net.MulticastSocket;
28 import java.net.NetworkInterface;
29 
30 import jdk.test.lib.NetworkConfiguration;

31 
32 /**
33  * @test
34  * @bug 4091811 4148753 4102731
35  * @summary Test java.net.MulticastSocket joinGroup and leaveGroup
36  * @library /test/lib
37  * @build jdk.test.lib.NetworkConfiguration
38  *        jdk.test.lib.Platform
39  * @run main JoinLeave
40  * @run main/othervm -Djava.net.preferIPv4Stack=true JoinLeave
41  */
42 public class JoinLeave {
43 
44     public static void main(String args[]) throws IOException {

45         InetAddress ip4Group = InetAddress.getByName("224.80.80.80");
46         InetAddress ip6Group = InetAddress.getByName("ff02::a");
47 
48         NetworkConfiguration nc = NetworkConfiguration.probe();
49         nc.ip4MulticastInterfaces().forEach(nic -> joinLeave(ip4Group, nic));
50         nc.ip6MulticastInterfaces().forEach(nic -> joinLeave(ip6Group, nic));
51     }
52 
53     static void joinLeave(InetAddress group, NetworkInterface nif) {
54         System.out.println("Joining:" + group + " on " + nif);
55         try (MulticastSocket soc = new MulticastSocket()) {
56             soc.setNetworkInterface(nif);
57             soc.joinGroup(group);
58             soc.leaveGroup(group);
59         } catch (IOException e) {
60             throw new UncheckedIOException(e);
61         }
62     }
63 }
