 60 public class UnreferencedMulticastSockets {
 61 
 62     /**
 63      * The set of sockets we have to check up on.
 64      */
 65     final static ArrayDeque<NamedWeak> pendingSockets = new ArrayDeque<>(5);
 66 
 67     /**
 68      * Queued objects when they are unreferenced.
 69      */
 70     final static ReferenceQueue<Object> pendingQueue = new ReferenceQueue<>();
 71 
 72     // Server to echo a datagram packet
 73     static class Server implements Runnable {
 74 
 75         MulticastSocket ss;
 76         final int port;
 77         final Phaser phaser = new Phaser(2);
 78         Server() throws IOException {
 79             InetAddress loopback = InetAddress.getLoopbackAddress();
 80             InetSocketAddress serverAddress =
 81                 new InetSocketAddress(loopback, 0);
 82             ss = new MulticastSocket(serverAddress);
 83             port = ss.getLocalPort();
 84             System.out.printf("  DatagramServer addr: %s: %d%n",
 85                     this.getHost(), this.getPort());
 86             pendingSockets.add(new NamedWeak(ss, pendingQueue, "serverMulticastSocket"));
 87             extractRefs(ss, "serverMulticastSocket");
 88         }
 89 
 90         InetAddress getHost() throws UnknownHostException {
 91             InetAddress localhost = InetAddress.getLoopbackAddress();
 92             return localhost;
 93         }
 94 
 95         int getPort() {
 96             return port;
 97         }
 98 
 99         // Receive a byte and send back a byte
100         public void run() {
101             try {
102                 byte[] buffer = new byte[50];
103                 DatagramPacket p = new DatagramPacket(buffer, buffer.length);
104                 ss.receive(p);
105                 System.out.printf("Server: ping received from: %s%n", p.getSocketAddress());
106                 phaser.arriveAndAwaitAdvance(); // await the client...
107                 buffer[0] += 1;
108                 System.out.printf("Server: sending echo to: %s%n", p.getSocketAddress());
109                 ss.send(p);         // send back +1
110 
111                 System.out.printf("Server: awaiting client%n");
112                 phaser.arriveAndAwaitAdvance(); // await the client...
113                 // do NOT close but 'forget' the socket reference
114                 System.out.printf("Server: forgetting socket...%n");
115                 ss = null;
116             } catch (Throwable ioe) {
117                 ioe.printStackTrace();
118             }
119         }
120     }

122     public static void main(String args[]) throws Exception {
123         IPSupport.throwSkippedExceptionIfNonOperational();
124 
125         InetSocketAddress clientAddress =
126                 new InetSocketAddress(InetAddress.getLoopbackAddress(), 0);
127 
128         // Create and close a MulticastSocket to warm up the FD count for side effects.
129         try (MulticastSocket s = new MulticastSocket(clientAddress)) {
130             // no-op; close immediately
131             s.getLocalPort();   // no-op
132         }
133 
134         long fdCount0 = getFdCount();
135         listProcFD();
136 
137         // start a server
138         Server svr = new Server();
139         Thread thr = new Thread(svr);
140         thr.start();
141 
142         // It is possible under some circumstances that the client
143         // might get bound to the same port than the server: this
144         // would make the test fail - so if this happen we try to
145         // bind to a specific port by incrementing the server port.
146         MulticastSocket client = null;
147         int serverPort = svr.getPort();
148         int maxtries = 20;
149         for (int i = 0; i < maxtries; i++) {
150             try {
151                 System.out.printf("Trying to bind client to: %s%n", clientAddress);
152                 client = new MulticastSocket(clientAddress);
153                 if (client.getLocalPort() != svr.getPort()) break;
154                 client.close();
155             } catch (IOException x) {
156                 System.out.printf("Couldn't create client after %d attempts: %s%n", i, x);
157                 if (i == maxtries) throw x;
158             }
159             if (i == maxtries) {
160                 String msg = String.format("Couldn't create client after %d attempts", i);
161                 System.out.println(msg);
162                 throw new AssertionError(msg);
163             }
164             clientAddress = new InetSocketAddress(clientAddress.getAddress(), serverPort + i);
165         }
166 
167         System.out.printf("  client bound port: %s:%d%n",
168                 client.getLocalAddress(), client.getLocalPort());
169         client.connect(svr.getHost(), svr.getPort());
170         pendingSockets.add(new NamedWeak(client, pendingQueue, "clientMulticastSocket"));
171         extractRefs(client, "clientMulticastSocket");
172 
173         byte[] msg = new byte[1];
174         msg[0] = 1;
175         DatagramPacket p = new DatagramPacket(msg, msg.length, svr.getHost(), svr.getPort());
176         client.send(p);
177         System.out.printf("  ping sent to: %s:%d%n", svr.getHost(), svr.getPort());
178         svr.phaser.arriveAndAwaitAdvance(); // wait until the server has received its packet
179 
180         p = new DatagramPacket(msg, msg.length);
181         client.receive(p);
182 
183         System.out.printf("  echo received from: %s%n", p.getSocketAddress());
184         if (msg[0] != 2) {
185             throw new AssertionError("incorrect data received: expected: 2, actual: " + msg[0]);
186         }
187         svr.phaser.arriveAndAwaitAdvance(); // let the server null out its socket
188 
189         // Do NOT close the MulticastSocket; forget it
190 
191         Object ref;
192         int loops = 20;
193         while (!pendingSockets.isEmpty() && loops-- > 0) {
194             ref = pendingQueue.remove(1000L);
195             if (ref != null) {
196                 pendingSockets.remove(ref);
197                 System.out.printf("  ref freed: %s, remaining: %d%n", ref, pendingSockets.size());
198             } else {
199                 client = null;
200                 p = null;
201                 msg = null;
202                 System.gc();
203             }
204         }
205 
206         thr.join();
207 
208         // List the open file descriptors
209         long fdCount = getFdCount();
210         System.out.printf("Initial fdCount: %d, final fdCount: %d%n", fdCount0, fdCount);
211         listProcFD();
212 
213         if (loops == 0) {
214             throw new AssertionError("Not all references reclaimed");
215         }
216     }
217 
218     // Get the count of open file descriptors, or -1 if not available
219     private static long getFdCount() {
220         OperatingSystemMXBean mxBean = ManagementFactory.getOperatingSystemMXBean();
221         return (mxBean instanceof UnixOperatingSystemMXBean)
222                 ? ((UnixOperatingSystemMXBean) mxBean).getOpenFileDescriptorCount()
223                 : -1L;
224     }

226     // Reflect to find references in the socket implementation that will be gc'd
227     private static void extractRefs(MulticastSocket s, String name) {
228         try {
229 
230             Field socketImplField = DatagramSocket.class.getDeclaredField("impl");
231             socketImplField.setAccessible(true);
232             Object socketImpl = socketImplField.get(s);
233 
234             Field fileDescriptorField = DatagramSocketImpl.class.getDeclaredField("fd");
235             fileDescriptorField.setAccessible(true);
236             FileDescriptor fileDescriptor = (FileDescriptor) fileDescriptorField.get(socketImpl);
237             extractRefs(fileDescriptor, name);
238 
239             Class<?> socketImplClass = socketImpl.getClass();
240             System.out.printf("socketImplClass: %s%n", socketImplClass);
241             if (socketImplClass.getName().equals("java.net.TwoStacksPlainDatagramSocketImpl")) {
242                 Field fileDescriptor1Field = socketImplClass.getDeclaredField("fd1");
243                 fileDescriptor1Field.setAccessible(true);
244                 FileDescriptor fileDescriptor1 = (FileDescriptor) fileDescriptor1Field.get(socketImpl);
245                 extractRefs(fileDescriptor1, name + "::twoStacksFd1");
246 
247             } else {
248                 System.out.printf("socketImpl class name not matched: %s != %s%n",
249                         socketImplClass.getName(), "java.net.TwoStacksPlainDatagramSocketImpl");
250             }
251         } catch (NoSuchFieldException | IllegalAccessException ex) {
252             ex.printStackTrace();
253             throw new AssertionError("missing field", ex);
254         }
255     }
256 
257     private static void extractRefs(FileDescriptor fileDescriptor, String name) {
258         Object cleanup = null;
259         int rawfd = -1;
260         try {
261             if (fileDescriptor != null) {
262                 Field fd1Field = FileDescriptor.class.getDeclaredField("fd");
263                 fd1Field.setAccessible(true);
264                 rawfd = fd1Field.getInt(fileDescriptor);
265 
266                 Field cleanupfdField = FileDescriptor.class.getDeclaredField("cleanup");
267                 cleanupfdField.setAccessible(true);
268                 cleanup = cleanupfdField.get(fileDescriptor);
269                 pendingSockets.add(new NamedWeak(fileDescriptor, pendingQueue,
270                         name + "::fileDescriptor: " + rawfd));
271                 pendingSockets.add(new NamedWeak(cleanup, pendingQueue, name + "::fdCleanup: " + rawfd));
272 
273             }
274         } catch (NoSuchFieldException | IllegalAccessException ex) {
275             ex.printStackTrace();
276             throw new AssertionError("missing field", ex);
277         } finally {
278             System.out.print(String.format("  %s:: fd: %s, fd: %d, cleanup: %s%n",
279                     name, fileDescriptor, rawfd, cleanup));
280         }
281     }

283     /**
284      * Method to list the open file descriptors (if supported by the 'lsof' command).
285      */
286     static void listProcFD() {
287         List<String> lsofDirs = List.of("/usr/bin", "/usr/sbin");
288         Optional<Path> lsof = lsofDirs.stream()
289                 .map(s -> Paths.get(s, "lsof"))
290                 .filter(f -> Files.isExecutable(f))
291                 .findFirst();
292         lsof.ifPresent(exe -> {
293             try {
294                 System.out.printf("Open File Descriptors:%n");
295                 long pid = ProcessHandle.current().pid();
296                 ProcessBuilder pb = new ProcessBuilder(exe.toString(), "-p", Integer.toString((int) pid));
297                 pb.inheritIO();
298                 Process p = pb.start();
299                 p.waitFor(10, TimeUnit.SECONDS);
300             } catch (IOException | InterruptedException ie) {
301                 ie.printStackTrace();
302             }
303         });
304     }

307     // Simple class to identify which refs have been queued
308     static class NamedWeak extends WeakReference<Object> {
309         private final String name;
310 
311         NamedWeak(Object o, ReferenceQueue<Object> queue, String name) {
312             super(o, queue);
313             this.name = name;
314         }
315 
316         public String toString() {
317             return name + "; " + super.toString();
318         }
319     }
