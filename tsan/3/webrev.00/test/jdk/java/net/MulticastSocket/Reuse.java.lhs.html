24 import java.net.MulticastSocket;
25 import java.net.BindException;
26 





27 public class Reuse {
28     public static void main(String[] args) throws Exception {
29         MulticastSocket s1, s2;
30 
31         try {
32             s1 = new MulticastSocket(4160);
33             s2 = new MulticastSocket(4160);
34             s1.close();
35             s2.close();
36         } catch (BindException e) {
37             throw new RuntimeException("MulticastSocket do no set SO_REUSEADDR");
38         }
39     }
40 }
