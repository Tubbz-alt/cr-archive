 74     /* basic TCP connectivity test using IPv6 only and IPv4/IPv6 together */
 75 
 76     static void test1 () throws Exception {
 77         server = new ServerSocket (0);
 78         int port = server.getLocalPort();
 79         // try Ipv6 only
 80         c1 = new Socket ("::1", port);
 81         s1 = server.accept ();
 82         simpleDataExchange (c1, s1);
 83         s1.close ();
 84         c1.close();
 85         // try with both IPv4 and Ipv6
 86         c1 = new Socket ("127.0.0.1", port);
 87         c2 = new Socket ("::1", port);
 88         s1 = server.accept();
 89         s2 = server.accept();
 90         s1peer = s1.getInetAddress();
 91         s2peer = s2.getInetAddress();
 92         if (s1peer instanceof Inet6Address) {
 93             t_assert ((s2peer instanceof Inet4Address));
 94             simpleDataExchange (c2, s1);
 95             simpleDataExchange (c1, s2);
 96         } else {
 97             t_assert ((s2peer instanceof Inet6Address));
 98             simpleDataExchange (c1, s1);
 99             simpleDataExchange (c2, s2);
100         }
101         c1.close();
102         c2.close();
103         s1.close();
104         s2.close();
105         server.close ();
106         System.out.println ("Test1: OK");
107     }

160     /* Test timeouts on accept(), connect() */
161 
162     static void test3 () throws Exception {
163         server = new ServerSocket (0);
164         server.setSoTimeout (5000);
165         int port = server.getLocalPort();
166         long t1 = System.currentTimeMillis();
167         try {
168             server.accept ();
169             throw new RuntimeException ("accept should not have returned");
170         } catch (SocketTimeoutException e) {}
171         t1 = System.currentTimeMillis() - t1;
172         checkTime (t1, 5000);
173 
174         c1 = new Socket ();
175         c1.connect (new InetSocketAddress (ia4addr, port), 1000);
176         s1 = server.accept ();
177         simpleDataExchange (c1,s1);
178         c2 = new Socket ();
179         c2.connect (new InetSocketAddress (ia6addr, port), 1000);
180         s2 = server.accept ();
181         simpleDataExchange (c2,s2);
182         c3 = new Socket ();
183         c3.connect (new InetSocketAddress (ia6addr, port), 1000);
184         s3 = server.accept ();
185         c2.close (); s2.close();
186         server.close();
187         simpleDataExchange (c3,s3);
188         c1.close (); c2.close();
189         s1.close (); s2.close();
190 
191         System.out.println ("Test3: OK");
192     }
