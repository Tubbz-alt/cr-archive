31 
32 import java.nio.channels.*;
33 import java.net.*;
34 import jdk.test.lib.net.IPSupport;
35 
36 public class MinimumRcvBufferSize {
37 
38     public static void main(String args[]) throws Exception {
39         IPSupport.throwSkippedExceptionIfNonOperational();
40 
41         boolean error = false;
42         ServerSocketChannel channel = ServerSocketChannel.open();
43         int before = channel.getOption(StandardSocketOptions.SO_RCVBUF);
44         channel.setOption(StandardSocketOptions.SO_RCVBUF, Integer.MAX_VALUE);
45         int after = channel.getOption(StandardSocketOptions.SO_RCVBUF);
46         if (before > after) {
47             System.err.println("Test failed: SO_RCVBUF");
48             error = true;
49         }
50 
51         SocketChannel channel1 = SocketChannel.open();
52         before = channel1.getOption(StandardSocketOptions.SO_SNDBUF);
53         channel1.setOption(StandardSocketOptions.SO_SNDBUF, Integer.MAX_VALUE);
54         after = channel1.getOption(StandardSocketOptions.SO_SNDBUF);
55         if (before > after) {
56             System.err.println("Test failed: SO_SNDBUF");
57             error = true;
58         }
59         if (error)
60             throw new RuntimeException("Test failed");
61     }
62 }
