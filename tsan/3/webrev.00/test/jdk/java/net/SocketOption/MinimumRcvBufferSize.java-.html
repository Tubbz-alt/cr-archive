24 /*
25  * @test
26  * @bug 8170920
27  * @run main MinimumRcvBufferSize
28  * @run main/othervm -Djava.net.preferIPv4Stack=true MinimumRcvBufferSize
29  */
30 
31 import java.nio.channels.*;
32 import java.net.*;
33 
34 public class MinimumRcvBufferSize {
35 
36     public static void main(String args[]) throws Exception {
37         boolean error = false;
38         ServerSocketChannel channel = ServerSocketChannel.open();
39         int before = channel.getOption(StandardSocketOptions.SO_RCVBUF);
40         channel.setOption(StandardSocketOptions.SO_RCVBUF, Integer.MAX_VALUE);
41         int after = channel.getOption(StandardSocketOptions.SO_RCVBUF);
42         if (before > after) {
43             System.err.println("Test failed: SO_RCVBUF");
44             error = true;
45         }
46 
47         SocketChannel channel1 = SocketChannel.open();
48         before = channel1.getOption(StandardSocketOptions.SO_SNDBUF);
49         channel1.setOption(StandardSocketOptions.SO_SNDBUF, Integer.MAX_VALUE);
50         after = channel1.getOption(StandardSocketOptions.SO_SNDBUF);
51         if (before > after) {
52             System.err.println("Test failed: SO_SNDBUF");
53             error = true;
54         }
55         if (error)
56             throw new RuntimeException("Test failed");
57     }
58 }
