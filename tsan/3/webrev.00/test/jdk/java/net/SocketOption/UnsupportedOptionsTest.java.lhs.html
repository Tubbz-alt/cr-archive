 24 import java.io.IOException;
 25 import java.lang.reflect.Field;
 26 import java.net.*;
 27 import java.util.ArrayList;
 28 import java.util.List;
 29 


 30 /*
 31  * @test
 32  * @bug 8143554 8044773

 33  * @summary Test checks that UnsupportedOperationException for unsupported
 34  *          SOCKET_OPTIONS is thrown by both getOption() and setOption() methods.
 35  * @requires !vm.graal.enabled
 36  * @run main UnsupportedOptionsTest
 37  * @run main/othervm -Djava.net.preferIPv4Stack=true UnsupportedOptionsTest
 38  * @run main/othervm --limit-modules=java.base UnsupportedOptionsTest
 39  */
 40 
 41 public class UnsupportedOptionsTest {
 42 
 43     private static final List<SocketOption<?>> socketOptions = new ArrayList<>();
 44 
 45     static {
 46         socketOptions.add(StandardSocketOptions.IP_MULTICAST_IF);
 47         socketOptions.add(StandardSocketOptions.IP_MULTICAST_LOOP);
 48         socketOptions.add(StandardSocketOptions.IP_MULTICAST_TTL);
 49         socketOptions.add(StandardSocketOptions.IP_TOS);
 50         socketOptions.add(StandardSocketOptions.SO_BROADCAST);
 51         socketOptions.add(StandardSocketOptions.SO_KEEPALIVE);
 52         socketOptions.add(StandardSocketOptions.SO_LINGER);
 53         socketOptions.add(StandardSocketOptions.SO_RCVBUF);
 54         socketOptions.add(StandardSocketOptions.SO_REUSEADDR);
 55         socketOptions.add(StandardSocketOptions.SO_SNDBUF);
 56         socketOptions.add(StandardSocketOptions.TCP_NODELAY);
 57 
 58         try {
 59             Class<?> c = Class.forName("jdk.net.ExtendedSocketOptions");
 60             Field field = c.getField("SO_FLOW_SLA");
 61             socketOptions.add((SocketOption<?>)field.get(null));
 62             field = c.getField("TCP_QUICKACK");
 63             socketOptions.add((SocketOption<?>)field.get(null));







 64         } catch (ClassNotFoundException e) {
 65             // ignore, jdk.net module not present
 66         } catch (ReflectiveOperationException e) {
 67             throw new AssertionError(e);
 68         }
 69     }
 70 
 71     public static void main(String[] args) throws IOException {


 72         Socket s = new Socket();
 73         ServerSocket ss = new ServerSocket();
 74         DatagramSocket ds = new DatagramSocket();
 75         MulticastSocket ms = new MulticastSocket();
 76 
 77         for (SocketOption option : socketOptions) {
 78             if (!s.supportedOptions().contains(option)) {
 79                 testUnsupportedSocketOption(s, option);
 80             }
 81 
 82             if (!ss.supportedOptions().contains(option)) {
 83                 testUnsupportedSocketOption(ss, option);
 84             }
 85 
 86             if (!ms.supportedOptions().contains(option)) {
 87                 testUnsupportedSocketOption(ms, option);
 88             }
 89 
 90             if (!ds.supportedOptions().contains(option)) {
 91                 testUnsupportedSocketOption(ds, option);
 92             }
 93         }
 94     }
 95 
 96     /*
 97      * Check that UnsupportedOperationException for unsupported option is
 98      * thrown from both getOption() and setOption() methods.
 99      */
100     private static void testUnsupportedSocketOption(Object socket,
101                                                     SocketOption option) {
102         testSet(socket, option);
103         testGet(socket, option);
104     }
105 
106     private static void testSet(Object socket, SocketOption option) {
107         try {
108             setOption(socket, option);
109         } catch (UnsupportedOperationException e) {
110             System.out.println("UnsupportedOperationException was throw " +
111                     "as expected. Socket: " + socket + " Option: " + option);
112             return;
113         } catch (Exception e) {
114             throw new RuntimeException("FAIL. Unexpected exception.", e);
115         }
116         throw new RuntimeException("FAIL. UnsupportedOperationException " +
117                 "hasn't been thrown. Socket: " + socket + " Option: " + option);
118     }
119 
120     private static void testGet(Object socket, SocketOption option) {
121         try {
122             getOption(socket, option);
123         } catch (UnsupportedOperationException e) {
124             System.out.println("UnsupportedOperationException was throw " +
125                     "as expected. Socket: " + socket + " Option: " + option);
126             return;
127         } catch (Exception e) {
128             throw new RuntimeException("FAIL. Unexpected exception.", e);
129         }
130         throw new RuntimeException("FAIL. UnsupportedOperationException " +
131                 "hasn't been thrown. Socket: " + socket + " Option: " + option);
132     }
133 
134     private static void getOption(Object socket,
135                                   SocketOption option) throws IOException {
136         if (socket instanceof Socket) {
137             ((Socket) socket).getOption(option);
138         } else if (socket instanceof ServerSocket) {
139             ((ServerSocket) socket).getOption(option);
140         } else if (socket instanceof DatagramSocket) {
141             ((DatagramSocket) socket).getOption(option);
142         } else {
143             throw new RuntimeException("Unsupported socket type");
144         }
145     }
146 
147     private static void setOption(Object socket,
148                                   SocketOption option) throws IOException {
149         if (socket instanceof Socket) {
150             ((Socket) socket).setOption(option, null);
151         } else if (socket instanceof ServerSocket) {
152             ((ServerSocket) socket).setOption(option, null);
153         } else if (socket instanceof DatagramSocket) {
154             ((DatagramSocket) socket).setOption(option, null);
155         } else {
156             throw new RuntimeException("Unsupported socket type");
157         }
158     }
159 }
