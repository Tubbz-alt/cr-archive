24 /**
25  * @test
26  * @bug 4163126
27  * @summary  test to see if timeout hangs
28  * @run main/timeout=15 Timeout
29  * @run main/othervm/timeout=15 -Djava.net.preferIPv4Stack=true Timeout
30  */
31 import java.net.*;
32 import java.io.*;
33 
34 public class Timeout {
35     public static void main(String[] args) throws Exception {
36         boolean success = false;
37         ServerSocket sock = new ServerSocket(0);
38         try {
39             sock.setSoTimeout(2);
40             sock.accept();
41         } catch (InterruptedIOException e) {
42             success = true;
43         } finally {
44             sock.close();
45         }
46         if (!success)
47             throw new RuntimeException("Socket timeout failure.");
48     }
49 }
