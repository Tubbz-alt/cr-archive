24 import java.util.ArrayList;
25 import java.util.List;
26 import java.util.concurrent.CompletableFuture;

27 import static java.util.concurrent.CompletableFuture.*;
28 
29 /*
30  * @test
31  * @bug 4344135

32  * @summary Check that {Socket,ServerSocket,DatagramSocket}.close will
33  *          cause any thread blocked on the socket to throw a SocketException.
34  * @run main AsyncClose
35  * @run main/othervm -Djava.net.preferIPv4Stack=true AsyncClose

36  */
37 
38 public class AsyncClose {
39 
40     public static void main(String args[]) throws Exception {

41 
42         AsyncCloseTest tests[] = {
43             new Socket_getInputStream_read(),
44             new Socket_getInputStream_read(20000),
45             new Socket_getOutputStream_write(),
46             new DatagramSocket_receive(),
47             new DatagramSocket_receive(20000),
48             new ServerSocket_accept(),
49             new ServerSocket_accept(20000),
50         };
51 
52         int failures = 0;
53 
54         List<CompletableFuture<AsyncCloseTest>> cfs = new ArrayList<>();
55         for (AsyncCloseTest test : tests)
56             cfs.add( supplyAsync(() -> test.go()));
57 
58         for (CompletableFuture<AsyncCloseTest> cf : cfs) {
59             AsyncCloseTest test = cf.get();
60 
61             System.out.println("******************************");
62             System.out.println("Test: " + test.description());
63             if (test.hasPassed()) {
64                 System.out.println("Passed.");
65             } else {
66                 System.out.println("Failed: " + test.failureReason());
67                 failures++;
68             }
69             System.out.println("");
70         }
71 
72         if (failures > 0)
73             throw new Exception(failures + " sub-tests failed - see log.");
74     }
75 }
