24 /*
25  * @bug 4468997
26  * @summary SO_LINGER is ignored on Windows with Winsock 2
27  * @run main RST
28  * @run main/othervm -Djava.net.preferIPv4Stack=true RST
29  */
30 import java.net.*;
31 import java.io.*;
32 
33 public class RST implements Runnable {
34 
35     Socket client;
36 
37     public void run() {
38         try {
39             client.setSoLinger(true, 0);        // hard reset
40             client.close();
41         } catch (Exception e) {
42             e.printStackTrace();
43         }
44     }
45 
46     RST() throws Exception {
47         ServerSocket ss = new ServerSocket(0);
48         client = new Socket("localhost", ss.getLocalPort());
49         Socket server = ss.accept();
50 
51         Thread thr = new Thread(this);
52         thr.start();
53 
54         SocketException exc = null;
55         try {
56             InputStream in = server.getInputStream();
57 
58             /*
59              * This read should throw a SocketException indicating a
60              * connection reset.
61              */
62             int n = in.read();
63         } catch (SocketException se) {
64             exc = se;
65         }
66 
67         server.close();
68         ss.close();
69 
70         if (exc == null) {
71             throw new Exception("Expected SocketException not thrown");
72         }
73         if (exc.getMessage().toLowerCase().indexOf("reset") == -1) {
74             throw new Exception("SocketException thrown but not expected \"connection reset\"");
75         }
76     }
77 
78 
79     public static void main(String args[]) throws Exception {
80         new RST();
81     }
82 }
