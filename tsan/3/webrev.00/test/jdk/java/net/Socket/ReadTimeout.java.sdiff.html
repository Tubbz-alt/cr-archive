32 import java.net.*;
33 import java.io.*;

34 
35 public class ReadTimeout  {
36     public static void main(String args[]) throws Exception {


37     InetAddress  sin = null;
38     Socket       soc = null,soc1 = null;
39     InputStream  is = null;
40     OutputStream os = null;
41     ServerSocket srv = null;
42     int          port = 0;
43     int          tout = 1000;
44 
45     sin = InetAddress.getLocalHost();
46     srv = new ServerSocket(port);

47     port = srv.getLocalPort();
48     soc = new Socket(sin, port);
49     soc1 = srv.accept();
50     soc.setSoTimeout(tout);
51 
52     try {
53       is = soc.getInputStream();
54       os = soc1.getOutputStream();
55       is.read();
56     } catch(InterruptedIOException e) {
57     } finally {
58         soc.close();
59         soc1.close();
60         srv.close();
61     }
62   }
