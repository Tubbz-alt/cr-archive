24 /*
25  * @test
26  * @bug 8201510
27  * @library /test/lib
28  * @summary Make sure IPv6 addresses are rejected when the System option
29  *          java.net.preferIPv4Stack is set to true
30  * @run main/othervm -Djava.net.preferIPv4Stack=true RejectIPv6
31  */
32 
33 import java.net.InetSocketAddress;
34 import java.net.ServerSocket;
35 import java.net.Socket;
36 import java.net.SocketException;
37 import jdk.test.lib.net.IPSupport;
38 
39 public class RejectIPv6 {
40 
41     public static void main(String [] argv) throws Throwable {
42         IPSupport.throwSkippedExceptionIfNonOperational();
43 
44         ServerSocket serverSocket = new ServerSocket(0);
45         serverSocket.setSoTimeout(1000);
46         int serverPort = serverSocket.getLocalPort();
47         Socket clientSocket = new Socket();
48 
49         test("bind", () -> clientSocket.bind(
50                 new InetSocketAddress("::1", 0)));
51 
52         test("connect", () -> clientSocket.connect(
53                 new InetSocketAddress("::1", serverPort), 1000));
54     }
55 
56     static void test(String msg, CodeToTest codeToTest) throws Throwable {
57         Thread client = new Thread(() ->
58             {
59                 try {
60                     codeToTest.run();
61                     throw new RuntimeException(msg +
62                             " failed to reject IPv6 address");
63                 } catch (SocketException ok) {
64                 } catch (Exception exc) {
65                     throw new RuntimeException("unexpected", exc);
66                 }
67             });
68         client.start();
69         client.join();
70     }
71 
72     interface CodeToTest {
73         void run() throws Exception;
74     }
75 }
