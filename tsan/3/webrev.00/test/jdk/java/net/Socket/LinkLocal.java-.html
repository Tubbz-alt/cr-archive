 24 /*
 25  * @test
 26  * @bug 4469866
 27  * @summary Connecting to a link-local IPv6 address should not
 28  *          causes a SocketException to be thrown.
 29  * @library /test/lib
 30  * @build jdk.test.lib.NetworkConfiguration
 31  *        jdk.test.lib.Platform
 32  * @run main LinkLocal
 33  * @run main/othervm -Djava.net.preferIPv4Stack=true LinkLocal
 34  */
 35 
 36 import jdk.test.lib.NetworkConfiguration;
 37 
 38 import java.net.*;
 39 import java.util.List;
 40 import java.util.stream.Collectors;
 41 
 42 public class LinkLocal {
 43 
 44     static int testCount = 0;
 45     static int failed = 0;
 46 
 47     static void TcpTest(InetAddress ia) throws Exception {
 48         System.out.println("**************************************");
 49         testCount++;
 50         System.out.println("Test " + testCount + ": TCP connect to " + ia);
 51 
 52         /*
 53          * Create ServerSocket on wildcard address and then
 54          * try to connect Socket to link-local address.
 55          */
 56         ServerSocket ss = new ServerSocket(0);
 57 
 58         Socket s = new Socket();
 59         try {
 60             s.connect(new InetSocketAddress(ia, ss.getLocalPort()));
 61 
 62             System.out.println("Test passed - connection established.");
 63 
 64             // connection was established so accept it
 65             Socket s2 = ss.accept();
 66             s2.close();
 67         } catch (SocketException e) {
 68             failed++;
 69             System.out.println("Test failed: " + e);
 70         } finally {
 71             s.close();
 72             ss.close();
 73         }
 74     }
 75 
 76     static void UdpTest(InetAddress ia, boolean connected) throws Exception {
 77 
 78         System.out.println("**************************************");
 79         testCount++;
 80 
 81         if (connected) {
 82             System.out.println("Test " + testCount + ": UDP connect to " + ia);
 83         } else {
 84             System.out.println("Test " + testCount + ": UDP send to " + ia);
 85         }
 86 
 87         DatagramSocket ds1 = new DatagramSocket();
 88         DatagramSocket ds2 = new DatagramSocket();
 89 
 90         try {
 91             byte b[] = "Hello".getBytes();
 92             DatagramPacket p = new DatagramPacket(b, b.length);
 93 
 94             if (connected) {
 95                 ds1.connect( new InetSocketAddress(ia, ds2.getLocalPort()) );
 96                 System.out.println("DatagramSocket connected.");
 97             } else {
 98                 p.setAddress(ia);
 99                 p.setPort(ds2.getLocalPort());
100             }
101             ds1.send(p);
102             System.out.println("Packet has been sent.");
103 
104             ds2.setSoTimeout(5000);
105             ds2.receive(p);
106             System.out.println("Test passed - packet received.");
107         } catch (SocketException e) {
108             failed++;
109             System.out.println("Test failed: " + e);
110         } finally {
111             ds1.close();
112             ds2.close();
113         }
114     }
115 
116     static void TestAddress(InetAddress ia) throws Exception {
117         TcpTest(ia);
118         UdpTest(ia, true);      /* unconnected */
119         UdpTest(ia, false);     /* connected */
120     }
121 
122     public static void main(String args[]) throws Exception {
123 
124         /*
125          * If an argument is provided ensure that it's
126          * a link-local IPv6 address.
127          */
128         if (args.length > 0) {
129             InetAddress ia = InetAddress.getByName(args[0]);
130 
131             if ( !(ia instanceof Inet6Address) ||
132                 !ia.isLinkLocalAddress()) {
133                 throw new Exception(ia +
134                         " is not a link-local IPv6 address");
135             }
136 
137             TestAddress(ia);
138         }
139 
140         /*
141          * If no argument is provided then enumerate the
142          * local addresses and run the test on each link-local
143          * IPv6 address.
144          */
145         if (args.length == 0) {
146             List<Inet6Address> addrs = NetworkConfiguration.probe()
147                     .ip6Addresses()
148                     .filter(Inet6Address::isLinkLocalAddress)
149                     .collect(Collectors.toList());
150 
151             for (Inet6Address addr : addrs) {
152                 TestAddress(addr);
153             }
154         }
155 
156         /*
157          * Print results
158          */
159         if (testCount == 0) {
160             System.out.println("No link-local IPv6 addresses - test skipped!");
161         } else {
162             System.out.println("**************************************");
163             System.out.println(testCount + " test(s) executed, " +
164                 failed + " failed.");
165             if (failed > 0) {
166                 throw new Exception( failed + " test(s) failed.");
167             }
168         }
169     }
170 }
