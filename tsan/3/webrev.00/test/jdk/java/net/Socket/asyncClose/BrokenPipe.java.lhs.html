24 /*
25  * @test
26  * @bug 4511404

27  * @summary Check that a broken pipe error doesn't throw an exception
28  *          indicating the socket is closed.
29  * @run main BrokenPipe
30  * @run main/othervm -Djava.net.preferIPv4Stack=true BrokenPipe
31  */
32 import java.io.*;
33 import java.net.*;

34 
35 public class BrokenPipe {
36 
37     private static class Closer implements Runnable {
38         private final Socket s;
39 
40         Closer(Socket s) {
41             this.s = s;
42         }
43 
44         public void run() {
45             try {
46                 /* gives time for 'write' to block */
47                 Thread.sleep(5000);
48                 s.close();
49             } catch (Exception e) {
50                 e.printStackTrace();
51             }
52         }
53     }
54 
55     public static void main(String[] args) throws Exception {
56         ServerSocket ss = new ServerSocket(0);


57         Socket client = new Socket(InetAddress.getLocalHost(),
58                                    ss.getLocalPort());
59         Socket server = ss.accept();
60         ss.close();
61         new Thread(new Closer(server)).start();
62 
63         try {
64             client.getOutputStream().write(new byte[1000000]);
65         } catch (IOException ioe) {
66             /*
67              * Check that the exception text doesn't indicate the
68              * socket is closed. In tiger we should be able to
69              * replace this by catching a more specific exception.
70              */
71             String text = ioe.getMessage();
72             if (text.toLowerCase().indexOf("closed") >= 0) {
73                 throw ioe;
74             }
75         } finally {
76             server.close();
77         }
78     }
79 
80 }
