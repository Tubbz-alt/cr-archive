 24 /*
 25  * @test
 26  * @bug 4091859 8189366

 27  * @summary Test Socket.available()
 28  * @run main CloseAvailable
 29  * @run main/othervm -Djava.net.preferIPv4Stack=true CloseAvailable
 30  */
 31 
 32 import java.net.*;
 33 import java.io.*;

 34 
 35 
 36 public class CloseAvailable {
 37 
 38     public static void main(String[] args) throws Exception {


 39         testClose();
 40 
 41         testEOF(true);
 42         testEOF(false);
 43         testIOEOnClosed(true);
 44         testIOEOnClosed(false);
 45     }
 46 
 47     static void testClose() throws IOException {
 48         boolean error = true;
 49         InetAddress addr = InetAddress.getLocalHost();
 50         ServerSocket ss = new ServerSocket(0);
 51         int port = ss.getLocalPort();
 52 
 53         Thread t = new Thread(new Thread("Close-Available-1") {
 54             public void run() {
 55                 try {
 56                     Socket s = new Socket(addr, port);
 57                     s.close();
 58                 } catch (Exception e) {
 59                     e.printStackTrace();
 60                 }
 61             }
 62         });
 63 
 64         t.start();
 65 
 66         Socket  soc = ss.accept();
 67         ss.close();
 68 
 69         DataInputStream is = new DataInputStream(soc.getInputStream());
 70         is.close();
 71 
 72         try {
 73             is.available();
 74         }
 75         catch (IOException ex) {
 76             error = false;
 77         }
 78         if (error)
 79             throw new RuntimeException("Available() can be called after stream closed.");
 80     }
 81 
 82     // Verifies consistency of `available` behaviour when EOF reached, both
 83     // explicitly and implicitly.
 84     static void testEOF(boolean readUntilEOF) throws IOException {
 85         System.out.println("testEOF, readUntilEOF: " + readUntilEOF);
 86         InetAddress addr = InetAddress.getLoopbackAddress();
 87         ServerSocket ss = new ServerSocket();
 88         ss.bind(new InetSocketAddress(addr, 0), 0);
 89         int port = ss.getLocalPort();
 90 
 91         try (Socket s = new Socket(addr, port)) {
 92             s.getOutputStream().write(0x42);
 93             s.shutdownOutput();
 94 
 95             try (Socket soc = ss.accept()) {
 96                 ss.close();
 97 
 98                 InputStream is = soc.getInputStream();
 99                 int b = is.read();
100                 assert b == 0x42;
101                 assert !s.isClosed();
102                 if (readUntilEOF) {
103                     b = is.read();
104                     assert b == -1;
105                 }
106 
107                 int a;
108                 for (int i = 0; i < 100; i++) {
109                     a = is.available();
110                     System.out.print(a + ", ");
111                     if (a != 0)
112                         throw new RuntimeException("Unexpected non-zero available: " + a);
113                 }
114                 assert !s.isClosed();
115                 assert is.read() == -1;
116             }
117         }
118         System.out.println("\ncomplete");
119     }
120 
121     // Verifies IOException thrown by `available`, on a closed input stream
122     // that may, or may not, have reached EOF prior to closure.
123     static void testIOEOnClosed(boolean readUntilEOF) throws IOException {
124         System.out.println("testIOEOnClosed, readUntilEOF: " + readUntilEOF);
125         InetAddress addr = InetAddress.getLoopbackAddress();
126         ServerSocket ss = new ServerSocket();
127         ss.bind(new InetSocketAddress(addr, 0), 0);
128         int port = ss.getLocalPort();
129 
130         try (Socket s = new Socket(addr, port)) {
131             s.getOutputStream().write(0x43);
132             s.shutdownOutput();
133 
134             try (Socket soc = ss.accept()) {
135                 ss.close();
136 
137                 InputStream is = soc.getInputStream();
138                 int b = is.read();
139                 assert b == 0x43;
140                 assert !s.isClosed();
141                 if (readUntilEOF) {
142                     b = is.read();
143                     assert b == -1;
144                 }
145                 is.close();
146                 try {
147                     b = is.available();
148                     throw new RuntimeException("UNEXPECTED successful read: " + b);
149                 } catch (IOException expected) {
150                     System.out.println("caught expected IOException:" + expected);
151                 }
152             }
153         }
154         System.out.println("\ncomplete");
155     }
156 }
