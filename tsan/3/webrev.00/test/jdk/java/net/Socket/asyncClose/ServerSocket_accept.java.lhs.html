24 /*
25  * Tests that a thread blocked in ServerSocket.accept
26  * throws a SocketException if the socket is asynchronously closed.
27  */
28 import java.io.IOException;
29 import java.net.*;



30 import java.util.concurrent.CountDownLatch;

31 
32 public class ServerSocket_accept extends AsyncCloseTest implements Runnable {
33     private final ServerSocket ss;
34     private final int timeout;
35     private final CountDownLatch latch;

36 
37     public ServerSocket_accept() throws IOException {
38        this(0);
39     }
40 
41     public ServerSocket_accept(int timeout) throws IOException {
42         this.timeout = timeout;
43         latch = new CountDownLatch(1);
44         ss = new ServerSocket(0, 0, InetAddress.getLoopbackAddress());
45     }
46 
47     public String description() {
48         String s = "ServerSocket.accept()";
49         if (timeout > 0) {
50             s += " (with timeout)";
51         }
52         return s;
53     }
54 
55     public void run() {
56         try {
57             latch.countDown();
58             Socket s = ss.accept();
59             failed("ServerSocket.accept() returned unexpectly!!" + " - " + s);






60         } catch (SocketException se) {
61             closed();
62         } catch (Exception e) {
63             failed(e.getMessage());
64         }
65     }
66 
67     public AsyncCloseTest go(){
68         try {
69             Thread thr = new Thread(this);
70             thr.start();
71             latch.await();
72             Thread.sleep(5000); //sleep, so ServerSocket.accept() can block

73             ss.close();
74             thr.join();
75 
76             if (isClosed()) {
77                 return passed();
78             } else {
79                 return failed("ServerSocket.accept() wasn't preempted");
80             }
81         } catch (Exception x) {
82             failed(x.getMessage());
83             throw new RuntimeException(x);
84         }
85     }
86 }
