 24 /*
 25  * @test
 26  * @bug 4508149

 27  * @summary Setting ServerSocket.setSoTimeout shouldn't cause
 28  *          the timeout to be inherited by accepted connections
 29  * @run main InheritTimeout
 30  * @run main/othervm -Djava.net.preferIPv4Stack=true InheritTimeout
 31  */
 32 
 33 import java.net.*;
 34 import java.io.InputStream;

 35 
 36 public class InheritTimeout {
 37 
 38     class Reaper extends Thread {
 39         Socket s;
 40         int timeout;
 41 
 42         Reaper(Socket s, int timeout) {
 43             this.s = s;
 44             this.timeout = timeout;
 45         }
 46 
 47         public void run() {
 48             try {
 49                 Thread.currentThread().sleep(timeout);
 50                 s.close();
 51             } catch (Exception e) {
 52             }
 53         }
 54     }
 55 
 56    InheritTimeout() throws Exception {
 57         ServerSocket ss = new ServerSocket(0);


 58         ss.setSoTimeout(1000);
 59 
 60         InetAddress ia = InetAddress.getLocalHost();
 61         InetSocketAddress isa =
 62             new InetSocketAddress(ia, ss.getLocalPort());
 63 
 64         // client establishes the connection
 65         Socket s1 = new Socket();
 66         s1.connect(isa);
 67 
 68         // receive the connection
 69         Socket s2 = ss.accept();
 70 
 71         // schedule reaper to close the socket in 5 seconds
 72         Reaper r = new Reaper(s2, 5000);
 73         r.start();
 74 
 75         boolean readTimedOut = false;
 76         try {
 77             s2.getInputStream().read();
 78         } catch (SocketTimeoutException te) {
 79             readTimedOut = true;
 80         } catch (SocketException e) {
 81             if (!s2.isClosed()) {
 82                 throw e;
 83             }
 84         }
 85 
 86         s1.close();
 87         ss.close();
 88 
 89         if (readTimedOut) {
 90             throw new Exception("Unexpected SocketTimeoutException throw!");
 91         }
 92    }
 93 
 94    public static void main(String args[]) throws Exception {

 95         new InheritTimeout();
 96    }
 97 }
