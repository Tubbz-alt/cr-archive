24 /**
25  * @test
26  * @bug 6210227
27  * @summary  REGRESSION: Socket.getLocalAddress() returns address of 0.0.0.0 on outbound TCP
28  * @run main B6210227
29  * @run main/othervm -Djava.net.preferIPv4Stack=true B6210227
30  */
31 
32 import java.util.*;
33 import java.net.*;
34 
35 public class B6210227 {
36     public static void main(String[] args) throws Exception
37     {
38         ServerSocket ss = new ServerSocket(0);
39         int port = ss.getLocalPort();
40 
41         byte[] bad = {0,0,0,0};
42         try {
43             InetSocketAddress isa = new InetSocketAddress(InetAddress.getLocalHost(), port);
44             Socket s = new Socket();
45             s.connect( isa, 1000 );
46             InetAddress iaLocal = s.getLocalAddress(); // if this comes back as 0.0. 0.0 this would demonstrate issue
47             String      sLocalHostname = iaLocal.getHostName();
48             if (Arrays.equals (iaLocal.getAddress(), bad)) {
49                 throw new RuntimeException ("0.0.0.0 returned");
50             }
51             System.out.println("local hostname is "+sLocalHostname );
52         } catch(Exception e) {
53             System.out.println("Exception happened");
54             throw e;
55         } finally {
56             ss.close();
57         }
58     }
59 }
