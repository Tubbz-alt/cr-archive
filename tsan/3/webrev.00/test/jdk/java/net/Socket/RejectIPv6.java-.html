24 /*
25  * @test
26  * @bug 8201510
27  * @summary Make sure IPv6 addresses are rejected when the System option
28  *          java.net.preferIPv4Stack is set to true
29  * @run main/othervm -Djava.net.preferIPv4Stack=true RejectIPv6
30  */
31 
32 import java.net.InetSocketAddress;
33 import java.net.ServerSocket;
34 import java.net.Socket;
35 import java.net.SocketException;
36 
37 public class RejectIPv6 {
38 
39     public static void main(String [] argv) throws Throwable {
40         ServerSocket serverSocket = new ServerSocket(0);
41         serverSocket.setSoTimeout(1000);
42         int serverPort = serverSocket.getLocalPort();
43         Socket clientSocket = new Socket();
44 
45         test("bind", () -> clientSocket.bind(
46                 new InetSocketAddress("::1", 0)));
47 
48         test("connect", () -> clientSocket.connect(
49                 new InetSocketAddress("::1", serverPort), 1000));
50     }
51 
52     static void test(String msg, CodeToTest codeToTest) throws Throwable {
53         Thread client = new Thread(() ->
54             {
55                 try {
56                     codeToTest.run();
57                     throw new RuntimeException(msg +
58                             " failed to reject IPv6 address");
59                 } catch (SocketException ok) {
60                 } catch (Exception exc) {
61                     throw new RuntimeException("unexpected", exc);
62                 }
63             });
64         client.start();
65         client.join();
66     }
67 
68     interface CodeToTest {
69         void run() throws Exception;
70     }
71 }
