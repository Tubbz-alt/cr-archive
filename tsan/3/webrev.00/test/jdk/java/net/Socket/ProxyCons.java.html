24 /*
25  * @test
26  * @bug 4097826
27  * @library /test/lib
28  * @summary SOCKS support inadequate
29  * @run main/timeout=40/othervm -DsocksProxyHost=nonexistant ProxyCons
30  * @run main/timeout=40/othervm -DsocksProxyHost=nonexistant -Djava.net.preferIPv4Stack=true ProxyCons
31  */
32 
33 import java.net.*;
34 import jdk.test.lib.net.IPSupport;
35 
36 public class ProxyCons {
37     class Server extends Thread {
38         ServerSocket server;
39         Server (ServerSocket server) {
40             super ();
41             this.server = server;
42         }
43         public void run () {
44             try {
45                 Socket s = server.accept ();
46                 s.close();
47                 while (!finished ()) {
48                     Thread.sleep (500);
49                 }
50             } catch (Exception e) {
51             }
52         }
53         boolean isFinished = false;
54 
55         synchronized boolean finished () {
56             return (isFinished);
57         }
58         synchronized void done () {
59             isFinished = true;
60         }
61     }
62 
63     public ProxyCons() {
64     }
65 
66     void test() throws Exception {
67         InetAddress localHost = InetAddress.getLocalHost();
68         ServerSocket ss = new ServerSocket();
69         ss.bind(new InetSocketAddress(localHost, 0));
70         try {
71             Server s = new Server(ss);
72             s.start();
73             Socket sock = new Socket(Proxy.NO_PROXY);
74             sock.connect(new InetSocketAddress(localHost, ss.getLocalPort()));
75             s.done();
76             sock.close();
77         } catch (java.io.IOException e) {
78             throw new RuntimeException(e);
79         } finally {
80             ss.close();
81         }
82     }
83 
84     public static void main(String[] args) throws Exception {
85         IPSupport.throwSkippedExceptionIfNonOperational();
86 
87         ProxyCons c = new ProxyCons();
88         c.test();
89     }
90 }
