24 /*
25  * @test
26  * @bug 4151834
27  * @summary Test Socket.setSoLinger
28  * @run main SetSoLinger
29  * @run main/othervm -Djava.net.preferIPv4Stack=true SetSoLinger
30  */
31 
32 import java.net.*;
33 
34 public class SetSoLinger {
35     static final int LINGER = 65546;
36 
37     public static void main(String args[]) throws Exception {
38         int value;
39         InetAddress addr = InetAddress.getLocalHost();
40         ServerSocket ss = new ServerSocket(0);
41         int port = ss.getLocalPort();
42 
43         Socket s = new Socket(addr, port);
44         Socket soc = ss.accept();
45         soc.setSoLinger(true, LINGER);
46         value = soc.getSoLinger();
47         soc.close();
48         s.close();
49         ss.close();
50 
51         if(value != 65535)
52             throw new RuntimeException("Failed. Value not properly reduced.");
53     }
54 }
