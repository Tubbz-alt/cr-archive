56         ServerSocket ss = new ServerSocket(0);


57         Socket client = new Socket(InetAddress.getLocalHost(),
58                                    ss.getLocalPort());
59         Socket server = ss.accept();
60         ss.close();
61         new Thread(new Closer(server)).start();
62 
63         try {
64             client.getOutputStream().write(new byte[1000000]);

76             server.close();
