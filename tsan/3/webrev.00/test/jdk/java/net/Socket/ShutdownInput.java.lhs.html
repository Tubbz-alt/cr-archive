24 /*
25  * @test
26  * @bug 7014860

27  * @summary Socket.getInputStream().available() not clear for
28  *          case that connection is shutdown for reading
29  * @run main ShutdownInput
30  * @run main/othervm -Djava.net.preferIPv4Stack=true ShutdownInput
31  */
32 
33 import java.io.InputStream;
34 import java.io.OutputStream;
35 import java.net.InetAddress;
36 import java.net.InetSocketAddress;
37 import java.net.ServerSocket;
38 import java.net.Socket;
39 import java.nio.channels.ServerSocketChannel;
40 import java.nio.channels.SocketChannel;

41 
42 public class ShutdownInput {
43     static boolean failed = false;
44 
45     public static void main(String args[]) throws Exception {
46         InetAddress iaddr = InetAddress.getLocalHost();
47 
48         try ( ServerSocket ss = new ServerSocket(0);


49               Socket s1 = new Socket(iaddr, ss.getLocalPort());
50               Socket s2 = ss.accept() ) {
51 
52             test(s1, s2, "Testing NET");
53         }
54 
55         // check the NIO socket adapter
56         try (ServerSocketChannel sc = ServerSocketChannel.open().bind(null);

57              SocketChannel s1 = SocketChannel.open(
58                      new InetSocketAddress(iaddr, sc.socket().getLocalPort()));
59              SocketChannel s2 = sc.accept() ) {
60 
61             test(s1.socket(), s2.socket(), "Testing NIO");
62         }
63 
64         if (failed) {
65             throw new RuntimeException("Failed: check output");
66         }
67     }
68 
69     public static void test(Socket s1, Socket s2, String mesg) throws Exception {
70         OutputStream os = s1.getOutputStream();
71         os.write("This is a message".getBytes("US-ASCII"));
72 
73         InputStream in = s2.getInputStream();
74         s2.shutdownInput();
75 
76         if (in.available() != 0) {
77             failed = true;
78             System.out.println(mesg + ":" + s2 + " in.available() should be 0, " +
79                                "but returns "+ in.available());
80         }
81 
82         byte[] ba = new byte[2];
83         if (in.read() != -1 ||
84             in.read(ba) != -1 ||
85             in.read(ba, 0, ba.length) != -1) {
86 
87             failed = true;
88             System.out.append(mesg + ":" + s2 + " in.read() should be -1");
89         }
90     }
91 }
