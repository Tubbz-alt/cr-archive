31 import java.io.IOException;

32 import java.net.ServerSocket;
33 import java.net.Socket;
34 
35 public class SocketGrowth {
36 
37     public static void main(String[] args) throws IOException {
38 
39         try (ServerSocket ss = new ServerSocket(0)) {
40             try (Socket s = new Socket("localhost", ss.getLocalPort());
41                     Socket peer = ss.accept()) {
42                 for (int i=0; i<1000000; i++) {
43                     // buggy JDK will run out of memory in this loop
44                     s.getOutputStream();
45                     // test InputStream also while we're here

48                 }
49             }
50         }
