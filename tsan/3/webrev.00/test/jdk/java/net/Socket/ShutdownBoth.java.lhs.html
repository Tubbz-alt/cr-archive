24 /*
25  * @test
26  * @bug 4425485
27  * @summary Check that shutdownInput followed by shutdownOutput
28  *          doesn't throw an exception.
29  */
30 import java.net.*;
31 
32 public class ShutdownBoth {
33 
34     public static void main(String args[]) throws Exception {
35         ServerSocket ss = new ServerSocket(0);

36         Socket s1 = new Socket(ss.getInetAddress(), ss.getLocalPort());
37         Socket s2 = ss.accept();
38 
39         try {
40             s1.shutdownInput();
41             s1.shutdownOutput();            // failed b55
42         } finally {
43             s1.close();
44             s2.close();
45             ss.close();
46         }
47     }
48 
49 }
