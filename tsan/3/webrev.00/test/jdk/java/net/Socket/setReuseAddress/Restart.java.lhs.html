24 /*
25  * @test
26  * @bug 4476378

27  * @summary Check that SO_REUSEADDR allows a server to restart
28  *          after a crash.
29  * @run main Restart
30  * @run main/othervm -Dsun.net.useExclusiveBind Restart
31  * @run main/othervm -Dsun.net.useExclusiveBind=true Restart
32  * @run main/othervm -Djava.net.preferIPv4Stack=true Restart
33  * @run main/othervm -Dsun.net.useExclusiveBind
34  *                   -Djava.net.preferIPv4Stack=true Restart
35  * @run main/othervm -Dsun.net.useExclusiveBind=true
36  *                   -Djava.net.preferIPv4Stack=true Restart
37  */
38 import java.net.*;

39 
40 public class Restart {
41 
42     /*
43      * Test that a server can bind to the same port after
44      * a crash -- ie: while previous connection still in
45      * TIME_WAIT state we should be able to re-bind if
46      * SO_REUSEADDR is enabled.
47      */
48 
49     public static void main(String args[]) throws Exception {
50         ServerSocket ss = new ServerSocket(0);



51         Socket s1 = null, s2 = null;
52         try {
53             int port = ss.getLocalPort();
54 
55             s1 = new Socket(InetAddress.getLocalHost(), port);
56             s2 = ss.accept();
57 
58             // close server socket and the accepted connection
59             ss.close();
60             s2.close();
61 
62             ss = new ServerSocket();
63             ss.bind( new InetSocketAddress(port) );
64             ss.close();
65 
66             // close the client socket
67             s1.close();
68         } catch (BindException be) {
69             if (System.getProperty("sun.net.useExclusiveBind") != null) {
70                 // exclusive bind, expected exception
71             } else {
72                 throw be;
73             }
74         } finally {
75             if (ss != null) ss.close();
76             if (s1 != null) s1.close();
77             if (s2 != null) s2.close();
78         }
79     }
80 }
