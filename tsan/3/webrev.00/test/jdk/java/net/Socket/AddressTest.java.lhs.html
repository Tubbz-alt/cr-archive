 24 /*
 25  * @test
 26  * @bug 4507501

 27  * @summary Test various methods that should throw IAE when passed improper
 28  *          SocketAddress
 29  * @run main AddressTest
 30  * @run main/othervm -Djava.net.preferIPv4Stack=true AddressTest
 31  */
 32 
 33 import java.net.*;

 34 
 35 public class AddressTest {
 36     class MySocketAddress extends SocketAddress {
 37         public MySocketAddress() {
 38         }
 39     }
 40 
 41     public AddressTest() throws Exception {
 42         SocketAddress addr = new MySocketAddress();
 43         Socket soc = new Socket();
 44         ServerSocket serv = new ServerSocket();
 45         DatagramSocket ds = new DatagramSocket((SocketAddress)null);
 46         DatagramPacket pac = new DatagramPacket(new byte[20], 20);
 47         MulticastSocket mul = new MulticastSocket((SocketAddress) null);
 48         boolean ok = false;
 49         try {
 50             soc.bind(addr);
 51         } catch (IllegalArgumentException e) {
 52             ok = true;
 53         } catch (Exception e2) {
 54         }
 55         if (!ok)
 56             throw new RuntimeException("Socket.bind should throw IllegalArgumentException!");
 57 
 58         ok = false;
 59         soc.close();
 60         soc = new Socket();
 61         try {
 62             soc.connect(addr, 100);
 63         } catch (IllegalArgumentException e) {
 64             ok = true;
 65         } catch (Exception e2) {
 66         }
 67         if (!ok)
 68             throw new RuntimeException("Socket.connect should throw IllegalArgumentException!");
 69 
 70         ok = false;
 71         try {
 72             serv.bind(addr);
 73         } catch (IllegalArgumentException e) {
 74             ok = true;
 75         } catch (Exception e2) {
 76         }
 77         if (!ok)
 78             throw new RuntimeException("ServerSocket.bind should throw IllegalArgumentException!");
 79 
 80         ok = false;
 81 
 82         try {
 83             pac.setSocketAddress(addr);
 84         } catch (IllegalArgumentException e) {
 85             ok = true;
 86         } catch (Exception e2) {
 87         }
 88 
 89         if (!ok)
 90             throw new RuntimeException("DatagramPacket.setSocketAddress should throw IllegalArgumentException");
 91 
 92         ok = false;
 93 
 94         try {
 95             ds.bind(addr);
 96         } catch (IllegalArgumentException e) {
 97             ok = true;
 98         } catch (Exception e2) {
 99         }
100 
101         if (!ok)
102             throw new RuntimeException("DatagramSocket.bind should throw IllegalArgumentException");
103 
104         ok = false;
105 
106         try {
107             ds.connect(addr);
108         } catch (IllegalArgumentException e) {
109             ok = true;
110         } catch (Exception e2) {
111         }
112 
113         if (!ok)
114             throw new RuntimeException("DatagramSocket.connect should throw IllegalArgumentException");
115 
116         ok = false;
117 
118         try {
119             mul.bind(addr);
120         } catch (IllegalArgumentException e) {
121             ok = true;
122         } catch (Exception e2) {
123         }
124 
125         if (!ok)
126             throw new RuntimeException("MulticastSocket.bind should throw IllegalArgumentException");
127 
128         ok = false;
129 
130         mul.close();
131         mul = new MulticastSocket(new InetSocketAddress(0));
132         try {
133             mul.joinGroup(addr, null);
134         } catch (IllegalArgumentException e) {
135             ok = true;
136         } catch (Exception e2) {
137         }
138 
139         if (!ok)
140             throw new RuntimeException("MulticastSocket.joinGroup should throw IllegalArgumentException");
141 
142         ok = false;
143         try {
144             mul.leaveGroup(addr, null);
145         } catch (IllegalArgumentException e) {
146             ok = true;
147         } catch (Exception e2) {
148         }
149 
150         if (!ok)
151             throw new RuntimeException("MulticastSocket.leaveGroup should throw IllegalArgumentException");
152 
153     }
154 
155     public static void main(String[] args) throws Exception {

156         new AddressTest();
157     }
158 }
