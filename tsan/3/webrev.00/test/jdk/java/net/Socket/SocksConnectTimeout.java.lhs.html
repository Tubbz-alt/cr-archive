 24 /*
 25  * @test
 26  * @bug 6223635

 27  * @summary Code hangs at connect call even when Timeout is specified
 28  * @run main SocksConnectTimeout
 29  * @run main/othervm -Djava.net.preferIPv4Stack=true SocksConnectTimeout
 30  */
 31 
 32 import java.net.InetAddress;
 33 import java.net.InetSocketAddress;
 34 import java.net.Proxy;
 35 import java.net.Socket;
 36 import java.net.ServerSocket;
 37 import java.net.SocketTimeoutException;
 38 import java.io.IOException;
 39 import java.io.Closeable;
 40 import java.util.concurrent.Phaser;
 41 import java.util.concurrent.TimeUnit;

 42 
 43 public class SocksConnectTimeout {
 44     static ServerSocket serverSocket;
 45     static final boolean debug = true;
 46     static final Phaser startPhaser = new Phaser(2);
 47     static final Phaser finishPhaser = new Phaser(2);
 48     static int failed, passed;
 49 
 50     public static void main(String[] args) {


 51         try {
 52             serverSocket = new ServerSocket(0);


 53 
 54             (new Thread() {
 55                 @Override
 56                 public void run() { serve(); }
 57             }).start();
 58 
 59             Proxy socksProxy = new Proxy(Proxy.Type.SOCKS,
 60                 new InetSocketAddress(InetAddress.getLocalHost(), serverSocket.getLocalPort()));
 61 
 62             test(socksProxy);
 63         } catch (IOException e) {
 64             unexpected(e);
 65         } finally {
 66             close(serverSocket);
 67 
 68             if (failed > 0)
 69                 throw new RuntimeException("Test Failed: passed:" + passed + ", failed:" + failed);
 70         }
 71     }
 72 
 73     static void test(Proxy proxy) {
 74         startPhaser.arriveAndAwaitAdvance();
 75         Socket socket = null;
 76         try {
 77             socket = new Socket(proxy);
 78             connectWithTimeout(socket);
 79             failed("connected successfully!");
 80         } catch (SocketTimeoutException socketTimeout) {
 81             debug("Passed: Received: " + socketTimeout);
 82             passed();
 83         } catch (Exception exception) {
 84             failed("Connect timeout test failed", exception);
 85         } finally {
 86             finishPhaser.arriveAndAwaitAdvance();
 87             close(socket);
 88         }
 89     }
 90 
 91     static void connectWithTimeout(Socket socket) throws IOException {
 92         socket.connect(new InetSocketAddress(InetAddress.getLocalHost(), 1234), 500);
 93     }
 94 
 95     static void serve() {
 96         Socket client = null;
 97         try {
 98             startPhaser.arriveAndAwaitAdvance();
 99             client = serverSocket.accept();
100             finishPhaser.awaitAdvanceInterruptibly(finishPhaser.arrive(), 5, TimeUnit.SECONDS);
101         } catch (Exception e) {
102             unexpected(e);
103         } finally {
104             close(client);
105         }
106     }
107 
108     static void debug(String message) {
109         if (debug)
110             System.out.println(message);
111     }
112 
113     static void unexpected(Exception e ) {
114         System.out.println("Unexcepted Exception: " + e);
115     }
116 
117     static void close(Closeable closeable) {
118         if (closeable != null) try { closeable.close(); } catch (IOException e) {unexpected(e);}
119     }
120 
121     static void failed(String message) {
122         System.out.println(message);
123         failed++;
124     }
125 
126     static void failed(String message, Exception e) {
127         System.out.println(message);
128         System.out.println(e);
129         failed++;
130     }
131 
132     static void passed() { passed++; };
133 
134 }
