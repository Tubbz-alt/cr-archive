24 /*
25  * @test
26  * @bug 4097826

27  * @summary SOCKS support inadequate
28  * @run main/timeout=40/othervm -DsocksProxyHost=nonexistant ProxyCons
29  * @run main/timeout=40/othervm -DsocksProxyHost=nonexistant -Djava.net.preferIPv4Stack=true ProxyCons
30  */
31 
32 import java.net.*;


33 public class ProxyCons {
34     class Server extends Thread {
35         ServerSocket server;
36         Server (ServerSocket server) {
37             super ();
38             this.server = server;
39         }
40         public void run () {
41             try {
42                 Socket s = server.accept ();
43                 s.close();
44                 while (!finished ()) {
45                     Thread.sleep (500);
46                 }
47             } catch (Exception e) {
48             }
49         }
50         boolean isFinished = false;
51 
52         synchronized boolean finished () {
53             return (isFinished);
54         }
55         synchronized void done () {
56             isFinished = true;
57         }
58     }
59 
60     public ProxyCons() {
61     }
62 
63     void test() throws Exception {
64         ServerSocket ss = new ServerSocket(0);


65         try {
66             Server s = new Server(ss);
67             s.start();
68             Socket sock = new Socket(Proxy.NO_PROXY);
69             sock.connect(new InetSocketAddress("localhost", ss.getLocalPort()));
70             s.done();
71             sock.close();
72         } catch (java.io.IOException e) {
73             throw new RuntimeException(e);
74         } finally {
75             ss.close();
76         }
77     }
78 
79     public static void main(String[] args) throws Exception {


80         ProxyCons c = new ProxyCons();
81         c.test();
82     }
83 }
