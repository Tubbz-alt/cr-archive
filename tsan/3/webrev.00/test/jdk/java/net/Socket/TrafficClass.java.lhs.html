24 /*
25  * @test
26  * @bug 4511783

27  * @summary Test that setTrafficClass/getTraffiClass don't
28  *          throw an exception
29  * @run main TrafficClass
30  * @run main/othervm -Djava.net.preferIPv4Stack=true TrafficClass
31  */
32 import java.net.*;
33 import java.nio.*;
34 import java.nio.channels.*;

35 
36 public class TrafficClass {
37 
38     static final int IPTOS_RELIABILITY = 0x4;
39 
40     static int failures = 0;
41 
42     static void testDatagramSocket(DatagramSocket s) {
43         try {
44             s.setTrafficClass( IPTOS_RELIABILITY );
45             int tc = s.getTrafficClass();
46         } catch (Exception e) {
47             failures++;
48             System.err.println("testDatagramSocket failed: " + e);
49         }
50     }
51 
52     static void testSocket(Socket s) {
53         try {
54             s.setTrafficClass(IPTOS_RELIABILITY);
55             int tc = s.getTrafficClass();
56         } catch (Exception e) {
57             failures++;
58             System.err.println("testSocket failed: " + e);
59         }
60 
61     }
62 
63     public static void main(String args[]) throws Exception {

64 
65         DatagramSocket ds = new DatagramSocket();
66         testDatagramSocket(ds);
67 
68         DatagramChannel dc = DatagramChannel.open();
69         testDatagramSocket(dc.socket());
70 
71         Socket s = new Socket();
72         testSocket(s);
73 
74         SocketChannel sc = SocketChannel.open();
75         testSocket(sc.socket());
76 
77         if (failures > 0) {
78             throw new Exception(failures + " sub-test(s) failed - " +
79                 "see log for details.");
80         }
81     }
82 
83 }
