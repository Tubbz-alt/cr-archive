29  * @run main/othervm -Djava.net.preferIPv4Stack=true TestTcpNoDelay
30  */
31 
32 import java.net.*;
33 import java.io.IOException;

34 
35 public class TestTcpNoDelay
36 {
37     public static void main(String[] args) {


38         try {
39             Socket socket = new Socket();
40             boolean on = socket.getTcpNoDelay();
41             System.out.println("Get TCP_NODELAY = " + on);
42 
43             boolean opposite = on ? false: true;
44             System.out.println("Set TCP_NODELAY to " + opposite);
45             socket.setTcpNoDelay(opposite);
46 
47             boolean noDelay = socket.getTcpNoDelay();
48             System.out.println("Get TCP_NODELAY = " + noDelay);
49 
50             if (noDelay != opposite)
51                 throw new RuntimeException("setTcpNoDelay no working as expected");
52 
53         } catch (IOException e){
54             e.printStackTrace();
55         }
56     }
57 
58 }
