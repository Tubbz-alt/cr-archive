 24 /* @test
 25  * @bug  6598160

 26  * @summary Windows IPv6 Socket implementation doesn't set the handle to not inherit
 27  * @author Chris Hegarty
 28  * @run main InheritHandle
 29  * @run main/othervm -Djava.net.preferIPv4Stack=true InheritHandle
 30  */
 31 
 32 import java.net.BindException;
 33 import java.net.ServerSocket;
 34 import java.io.File;
 35 import java.io.IOException;

 36 
 37 /**
 38  * This test is only really applicable to Windows machines that are running IPv6, but
 39  * it should still pass on other platforms so we can just run it.
 40  */
 41 
 42 public class InheritHandle
 43 {
 44     static String java = System.getProperty("java.home") + File.separator +
 45                          "bin" + File.separator + "java";
 46 
 47     public static void main(String[] args) {


 48         if (args.length == 1) {
 49             doWait();
 50         } else {
 51             startTest();
 52         }
 53 
 54     }
 55 
 56     static void startTest() {
 57         ServerSocket ss;
 58         int port;
 59         Process process;
 60 
 61         // create a ServerSocket listening on any port
 62         try {
 63             ss = new ServerSocket(0);
 64             port = ss.getLocalPort();
 65             System.out.println("First ServerSocket listening on port " + port);
 66         } catch (IOException e) {
 67             System.out.println("Cannot create ServerSocket");
 68             e.printStackTrace();
 69             return;
 70         }
 71 
 72         // create another java process that simply waits. If the bug is present this
 73         // process will inherit the native IPv6 handle for ss and cause the second
 74         // ServerSocket constructor to throw a BindException
 75         try {
 76             process = Runtime.getRuntime().exec(java + " InheritHandle -doWait");
 77         } catch (IOException ioe) {
 78             System.out.println("Cannot create process");
 79             ioe.printStackTrace();
 80             try {
 81                 ss.close();
 82             } catch (IOException e) {
 83                 e.printStackTrace();
 84             }
 85             return;
 86         }
 87 
 88         // Allow some time for the process to get started
 89         try {
 90             System.out.println("waiting...");
 91             Thread.sleep(2 * 1000);
 92 
 93             System.out.println("Now close the socket and try to create another" +
 94                                " one listening on the same port");
 95             ss.close();
 96             int retries = 0;
 97             while (retries < 5) {
 98                 try (ServerSocket s = new ServerSocket(port);) {
 99                     System.out.println("Second ServerSocket created successfully");
100                     break;
101                 } catch (BindException e) {
102                     System.out.println("BindException \"" + e.getMessage() + "\", retrying...");
103                     Thread.sleep(100L);
104                     retries ++;
105                     continue;
106                 }
107             }
108 
109         } catch (InterruptedException ie) {
110         } catch (IOException ioe) {
111             throw new RuntimeException("Failed: " + ioe);
112         } finally {
113             process.destroy();
114         }
115 
116         System.out.println("OK");
117     }
118 
119     static void doWait() {
120         try {
121             Thread.sleep(200 * 1000);
122         } catch (InterruptedException ie) {
123         }
124     }
125 }
