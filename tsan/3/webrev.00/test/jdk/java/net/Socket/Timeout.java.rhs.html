24 /**
25  * @test
26  * @bug 4163126
27  * @library /test/lib
28  * @summary  test to see if timeout hangs
29  * @run main/timeout=15 Timeout
30  * @run main/othervm/timeout=15 -Djava.net.preferIPv4Stack=true Timeout
31  */
32 import java.net.*;
33 import java.io.*;
34 import jdk.test.lib.net.IPSupport;
35 
36 public class Timeout {
37     public static void main(String[] args) throws Exception {
38         IPSupport.throwSkippedExceptionIfNonOperational();
39 
40         boolean success = false;
41         ServerSocket sock = new ServerSocket(0);
42         try {
43             sock.setSoTimeout(2);
44             sock.accept();
45         } catch (InterruptedIOException e) {
46             success = true;
47         } finally {
48             sock.close();
49         }
50         if (!success)
51             throw new RuntimeException("Socket timeout failure.");
52     }
53 }
