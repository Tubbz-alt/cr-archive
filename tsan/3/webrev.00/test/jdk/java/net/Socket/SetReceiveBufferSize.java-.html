25  * @test
26  * @bug 4326250
27  * @summary Test Socket.setReceiveBufferSize() throwin IllegalArgumentException
28  *
29  */
30 
31 import java.net.Socket;
32 import java.net.ServerSocket;
33 
34 public class SetReceiveBufferSize {
35     public static void main(String[] args) throws Exception {
36         SetReceiveBufferSize s = new SetReceiveBufferSize();
37     }
38 
39     public SetReceiveBufferSize() throws Exception {
40         ServerSocket ss = new ServerSocket(0);
41         Socket s = new Socket("localhost", ss.getLocalPort());
42         Socket accepted = ss.accept();
43         try {
44             s.setReceiveBufferSize(0);
45         } catch (IllegalArgumentException e) {
46             return;
47         } catch (Exception ex) {
48         } finally {
49             ss.close();
50             s.close();
51             accepted.close();
52         }
53         throw new RuntimeException("IllegalArgumentException not thrown!");
54     }
55 }
