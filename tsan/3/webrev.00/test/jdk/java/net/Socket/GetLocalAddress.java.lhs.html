47         addr = InetAddress.getLocalHost();
48         ss = new ServerSocket(0);

49         port = ss.getLocalPort();

51         Thread t = new Thread(new GetLocalAddress());
52         t.start();
53         Socket soc = ss.accept();
54 
55         if(addr.equals(soc.getLocalAddress())) {
56            error = false;
57            }

63     public void run() {
64         try {
65             Socket s = new Socket(addr, port);

68         }
69     }
70 
71     static void testBindNull() throws Exception {
72         try (Socket soc = new Socket()) {
73             soc.bind(null);
74             if (!soc.isBound())
75                 throw new RuntimeException(
76                     "should be bound after bind(null)");
77             if (soc.getLocalPort() <= 0)
78                 throw new RuntimeException(
79                    "bind(null) failed, local port: " + soc.getLocalPort());
80             if (soc.getLocalAddress() == null)
81                  throw new RuntimeException(
82                    "bind(null) failed, local address is null");
83         }
84     }
