 62     /**
 63      * Tests that a client will be unblocked and will throw an IOException
 64      * if an error occurs while the client is waiting for more data.
 65      * @throws InterruptedException
 66      * @throws ExecutionException
 67      */
 68     @Test
 69     public static void testOnError() throws InterruptedException, ExecutionException {
 70         CountDownLatch latch = new CountDownLatch(1);
 71         BodySubscriber<InputStream> isb = BodySubscribers.ofInputStream();
 72         ErrorTestSubscription s = new ErrorTestSubscription(isb);
 73         CompletionStage<Throwable> cs =
 74                 isb.getBody().thenApplyAsync((is) -> s.accept(latch, is));
 75         latch.await();
 76         isb.onSubscribe(s);
 77         s.t.join();
 78         Throwable result = cs.toCompletableFuture().get();
 79         Throwable t = result;
 80         if (!(t instanceof IOException)) {
 81             throw new RuntimeException("Failed to receive excpected IOException", result);
 82         } else {
 83             System.out.println("Got expected exception: " + t);
 84         }
 85         while (t != null) {
 86             if (t instanceof TestException) break;
 87             t = t.getCause();
 88         }
 89         if (t instanceof TestException) {
 90             System.out.println("Got expected cause: " + t);
 91         } else {
 92             throw new RuntimeException("Failed to receive excpected TestException", result);
 93         }
 94     }

 96     static class ErrorTestSubscription implements Flow.Subscription {
 97         final BodySubscriber<InputStream> isb;
 98         final Thread t = new Thread() {
 99             @Override
100             public void run() {
101                 try {
102                     // Give time to
103                     System.out.println("waiting...");
104                     Thread.sleep(1000);
105                 } catch (InterruptedException e) {
106 
107                 }
108                 System.out.println("Calling onError...");
109                 isb.onError(new TestException());
110             }
111         };
112 
113         ErrorTestSubscription(BodySubscriber<InputStream> isb) {
114             this.isb = isb;
115         }
116 
117         int requested = 0;
118 
119         @Override
120         public void request(long n) {
121             System.out.println("Got request: " + n);
122             if (requested == 0 && n > 0) {
123                 //isb.onNext(List.of(java.nio.ByteBuffer.wrap(new byte[] {0x01})));
124                 requested += n;
125                 t.start();
126             }
127         }
128 
129         @Override
130         public void cancel() {
131         }
132 
133         public Throwable accept(CountDownLatch latch, InputStream is) {
134             System.out.println("got " + is);
135             try {
136                 latch.countDown();
137                 System.out.println("reading all bytes");
138                 is.readAllBytes();
139                 System.out.println("all bytes read");
140             } catch (IOException e) {
141                 return e;
142             } finally {
143                 try {
144                     is.close();
145                 } catch (IOException e) {
146                     return e;
147                 }
148             }
149             return is == null ? new NullPointerException() : null;
150         }
151     }

154         try {
155             is.close();

191     @Test
192     public static void testReadParameters() throws InterruptedException, ExecutionException, IOException {
193         BodySubscriber<InputStream> isb = BodySubscribers.ofInputStream();
194         InputStream is = isb.getBody().toCompletableFuture().get();
195 
196         Throwable ex;
197 
198         // len == 0
199         assertEquals(is.read(new byte[16], 0, 0), 0);
200         assertEquals(is.read(new byte[16], 16, 0), 0);
201 
202         // index == -1
203         ex = expectThrows(OOB, () -> is.read(new byte[16], -1, 10));
204         System.out.println("OutOfBoundsException thrown as expected: " + ex);
205 
206         // large offset
207         ex = expectThrows(OOB, () -> is.read(new byte[16], 17, 10));
208         System.out.println("OutOfBoundsException thrown as expected: " + ex);
209 
210         ex = expectThrows(OOB, () -> is.read(new byte[16], 10, 10));
211         System.out.println("OutOfBoundsException thrown as expected: " + ex);
212 
213         // null value
214         ex = expectThrows(NPE, () -> is.read(null, 0, 10));
215         System.out.println("NullPointerException thrown as expected: " + ex);
216     }

218     @Test
219     public static void testSubscribeAndClose()
220             throws InterruptedException, ExecutionException
221     {
222         BodySubscriber<InputStream> isb = BodySubscribers.ofInputStream();
223         TestCancelOnCloseSubscription s = new TestCancelOnCloseSubscription();
224         InputStream is = isb.getBody().toCompletableFuture().get();
225         isb.onSubscribe(s);
226         if (s.cancelled.get()) {
227             throw new RuntimeException("subscription cancelled");
228         }
229         CompletableFuture<String> cf = CompletableFuture.supplyAsync(
230                 () -> HttpResponseInputStreamTest.readAllBytes(is))
231                 .thenApply(String::new);
232         while (s.request.get() == 0) {
233             Thread.sleep(100);
234         }
235         isb.onNext(List.of(ByteBuffer.wrap("coucou".getBytes())));
236         close(is);
237         System.out.println(s);
238         if (!s.cancelled.get()) {
239             throw new RuntimeException("subscription not cancelled");
240         }
241         if (s.request.get() == 0) {
242             throw new RuntimeException("subscription has no demand");
243         }
244         try {
245             System.out.println("read " + cf.get() + "!");
246             throw new RuntimeException("expected IOException not raised");
247         } catch (ExecutionException | CompletionException x) {
248             if (x.getCause() instanceof IOException) {
249                 System.out.println("Got expected IOException: " + x.getCause());
250             } else {
251                 throw x;
252             }
253         }
254     }

256     static class TestCancelOnCloseSubscription implements Flow.Subscription {
257         final AtomicLong request = new AtomicLong();
258         final AtomicBoolean cancelled = new AtomicBoolean();
259 
260         @Override
261         public void request(long n) {
262             request.addAndGet(n);
263         }
264 
265         @Override
266         public void cancel() {
267             cancelled.set(true);
268         }
269 
270         @Override
271         public String toString() {
272             return String.format("%s(request=%d, cancelled=%s)",
273                     this.getClass().getSimpleName(),
274                     request.get(),
275                     cancelled.get());
276         }
277     }
