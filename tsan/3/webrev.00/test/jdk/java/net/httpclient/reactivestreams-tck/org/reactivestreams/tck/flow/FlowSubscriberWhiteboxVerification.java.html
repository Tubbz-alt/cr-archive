24 package org.reactivestreams.tck.flow;
25 
26 import org.reactivestreams.FlowAdapters;
27 import org.reactivestreams.Subscriber;
28 import org.reactivestreams.tck.SubscriberWhiteboxVerification;
29 import org.reactivestreams.tck.TestEnvironment;
30 import org.reactivestreams.tck.flow.support.SubscriberWhiteboxVerificationRules;
31 
32 import java.util.concurrent.Flow;
33 
34 /**
35  * Provides whitebox style tests for verifying {@link java.util.concurrent.Flow.Subscriber}
36  * and {@link java.util.concurrent.Flow.Subscription} specification rules.
37  *
38  * @see java.util.concurrent.Flow.Subscriber
39  * @see java.util.concurrent.Flow.Subscription
40  */
41 public abstract class FlowSubscriberWhiteboxVerification<T> extends SubscriberWhiteboxVerification<T>
42   implements SubscriberWhiteboxVerificationRules {
43 
44   protected FlowSubscriberWhiteboxVerification(TestEnvironment env) {
45     super(env);
46   }
47 
48   @Override
49   final public Subscriber<T> createSubscriber(WhiteboxSubscriberProbe<T> probe) {
50     return FlowAdapters.toSubscriber(createFlowSubscriber(probe));
51   }
52   /**
53    * This is the main method you must implement in your test incarnation.
54    * It must create a new {@link org.reactivestreams.Subscriber} instance to be subjected to the testing logic.
55    *
56    * In order to be meaningfully testable your Subscriber must inform the given
57    * `WhiteboxSubscriberProbe` of the respective events having been received.
58    */
59   protected abstract Flow.Subscriber<T> createFlowSubscriber(WhiteboxSubscriberProbe<T> probe);
60 }
