24 import org.reactivestreams.tck.TestEnvironment;
25 import org.reactivestreams.tck.flow.FlowPublisherVerification;
26 
27 import java.net.http.HttpRequest.BodyPublishers;
28 import java.nio.ByteBuffer;
29 import java.util.Collections;
30 import java.util.concurrent.Flow.Publisher;
31 
32 /* See TckDriver.java for more information */
33 public class BodyPublishersOfByteArrays
34         extends FlowPublisherVerification<ByteBuffer> {
35 
36     private static final int ELEMENT_SIZE = 16 * 1024;
37 
38     public BodyPublishersOfByteArrays() {
39         super(new TestEnvironment(450L));
40     }
41 
42     @Override
43     public Publisher<ByteBuffer> createFlowPublisher(long nElements) {
44         byte[] bytes = S.arrayOfNRandomBytes(ELEMENT_SIZE);
45         return BodyPublishers.ofByteArrays(
46                 Collections.nCopies((int) nElements, bytes));
47     }
48 
49     @Override
50     public Publisher<ByteBuffer> createFailedFlowPublisher() {
51         return null;
52     }
53 }
