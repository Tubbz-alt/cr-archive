24 import org.reactivestreams.tck.TestEnvironment;
25 import org.reactivestreams.tck.flow.FlowPublisherVerification;
26 
27 import java.io.IOException;
28 import java.io.UncheckedIOException;
29 import java.net.http.HttpRequest.BodyPublishers;
30 import java.nio.ByteBuffer;
31 import java.nio.file.Files;
32 import java.nio.file.Path;
33 import java.util.concurrent.Flow.Publisher;
34 import java.util.concurrent.atomic.AtomicLong;
35 
36 /* See TckDriver.java for more information */
37 public class BodyPublishersOfFile
38         extends FlowPublisherVerification<ByteBuffer> {
39 
40     private static final int ELEMENT_SIZE = 16 * 1024;
41     private static final AtomicLong UNIQUE_NUMBERS = new AtomicLong();
42 
43     public BodyPublishersOfFile() {
44         super(new TestEnvironment(450L));
45     }
46 
47     @Override
48     public Publisher<ByteBuffer> createFlowPublisher(long nElements) {
49         try {
50             Path f = createFile(nElements * ELEMENT_SIZE);
51             return BodyPublishers.ofFile(f);
52         } catch (IOException e) {
53             throw new UncheckedIOException(e);
54         }
55     }
56 
57     private static Path createFile(long nBytes) throws IOException {
58         String name = "f" + UNIQUE_NUMBERS.getAndIncrement();
59         Path f = Files.createFile(Path.of(name));
60         return Files.write(f, S.arrayOfNRandomBytes(nBytes));
61     }
62 
63     @Override
64     public Publisher<ByteBuffer> createFailedFlowPublisher() {
65         return null;
66     }
67 
68     @Override
69     public long maxElementsFromPublisher() {
70         return 21;
71     }
72 }
