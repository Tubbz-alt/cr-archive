 24 package org.reactivestreams.example.unicast;
 25 
 26 import org.reactivestreams.Subscriber;
 27 import org.reactivestreams.Subscription;
 28 
 29 /**
 30  * SyncSubscriber is an implementation of Reactive Streams `Subscriber`,
 31  * it runs synchronously (on the Publisher's thread) and requests one element
 32  * at a time and invokes a user-defined method to process each element.
 33  *
 34  * NOTE: The code below uses a lot of try-catches to show the reader where exceptions can be expected, and where they are forbidden.
 35  */
 36 public abstract class SyncSubscriber<T> implements Subscriber<T> {
 37   private Subscription subscription; // Obeying rule 3.1, we make this private!
 38   private boolean done = false;
 39 
 40   @Override public void onSubscribe(final Subscription s) {
 41     // As per rule 2.13, we need to throw a `java.lang.NullPointerException` if the `Subscription` is `null`
 42     if (s == null) throw null;
 43 
 44     if (subscription != null) { // If someone has made a mistake and added this Subscriber multiple times, let's handle it gracefully
 45       try {
 46         s.cancel(); // Cancel the additional subscription
 47       } catch(final Throwable t) {
 48         //Subscription.cancel is not allowed to throw an exception, according to rule 3.15
 49         (new IllegalStateException(s + " violated the Reactive Streams rule 3.15 by throwing an exception from cancel.", t)).printStackTrace(System.err);
 50       }
 51     } else {
 52       // We have to assign it locally before we use it, if we want to be a synchronous `Subscriber`
 53       // Because according to rule 3.10, the Subscription is allowed to call `onNext` synchronously from within `request`
 54       subscription = s;
 55       try {
 56         // If we want elements, according to rule 2.1 we need to call `request`
 57         // And, according to rule 3.2 we are allowed to call this synchronously from within the `onSubscribe` method
 58         s.request(1); // Our Subscriber is unbuffered and modest, it requests one element at a time
 59       } catch(final Throwable t) {
 60         // Subscription.request is not allowed to throw according to rule 3.16
 61         (new IllegalStateException(s + " violated the Reactive Streams rule 3.16 by throwing an exception from request.", t)).printStackTrace(System.err);
 62       }
 63     }
 64   }
 65 
 66   @Override public void onNext(final T element) {
 67     if (subscription == null) { // Technically this check is not needed, since we are expecting Publishers to conform to the spec
 68       (new IllegalStateException("Publisher violated the Reactive Streams rule 1.09 signalling onNext prior to onSubscribe.")).printStackTrace(System.err);
 69     } else {
 70       // As per rule 2.13, we need to throw a `java.lang.NullPointerException` if the `element` is `null`
 71       if (element == null) throw null;
 72 
 73       if (!done) { // If we aren't already done
 74         try {
 75           if (whenNext(element)) {
 76             try {
 77               subscription.request(1); // Our Subscriber is unbuffered and modest, it requests one element at a time
 78             } catch (final Throwable t) {
 79               // Subscription.request is not allowed to throw according to rule 3.16
 80               (new IllegalStateException(subscription + " violated the Reactive Streams rule 3.16 by throwing an exception from request.", t)).printStackTrace(System.err);
 81             }
 82           } else {
 83             done();
 84           }
 85         } catch (final Throwable t) {
 86           done();
 87           try {
 88             onError(t);
 89           } catch (final Throwable t2) {
 90             //Subscriber.onError is not allowed to throw an exception, according to rule 2.13
 91             (new IllegalStateException(this + " violated the Reactive Streams rule 2.13 by throwing an exception from onError.", t2)).printStackTrace(System.err);
 92           }
 93         }
 94       }
 95     }
 96   }
 97 
 98   // Showcases a convenience method to idempotently marking the Subscriber as "done", so we don't want to process more elements
 99   // herefor we also need to cancel our `Subscription`.
100   private void done() {
101     //On this line we could add a guard against `!done`, but since rule 3.7 says that `Subscription.cancel()` is idempotent, we don't need to.
102     done = true; // If we `whenNext` throws an exception, let's consider ourselves done (not accepting more elements)
103     try {
104       subscription.cancel(); // Cancel the subscription
105     } catch(final Throwable t) {
106       //Subscription.cancel is not allowed to throw an exception, according to rule 3.15
107       (new IllegalStateException(subscription + " violated the Reactive Streams rule 3.15 by throwing an exception from cancel.", t)).printStackTrace(System.err);
108     }
109   }
110 
111   // This method is left as an exercise to the reader/extension point
112   // Returns whether more elements are desired or not, and if no more elements are desired
113   protected abstract boolean whenNext(final T element);
114 
115   @Override public void onError(final Throwable t) {
116     if (subscription == null) { // Technically this check is not needed, since we are expecting Publishers to conform to the spec
117       (new IllegalStateException("Publisher violated the Reactive Streams rule 1.09 signalling onError prior to onSubscribe.")).printStackTrace(System.err);
118     } else {
119       // As per rule 2.13, we need to throw a `java.lang.NullPointerException` if the `Throwable` is `null`
120       if (t == null) throw null;
121       // Here we are not allowed to call any methods on the `Subscription` or the `Publisher`, as per rule 2.3
122       // And anyway, the `Subscription` is considered to be cancelled if this method gets called, as per rule 2.4
123     }
124   }
125 
126   @Override public void onComplete() {
127     if (subscription == null) { // Technically this check is not needed, since we are expecting Publishers to conform to the spec
128       (new IllegalStateException("Publisher violated the Reactive Streams rule 1.09 signalling onComplete prior to onSubscribe.")).printStackTrace(System.err);
129     } else {
130       // Here we are not allowed to call any methods on the `Subscription` or the `Publisher`, as per rule 2.3
131       // And anyway, the `Subscription` is considered to be cancelled if this method gets called, as per rule 2.4
132     }
133   }
134 }
