24 package org.reactivestreams.tck;
25 
26 import org.reactivestreams.Publisher;
27 import org.reactivestreams.tck.flow.support.Function;
28 import org.reactivestreams.tck.flow.support.HelperPublisher;
29 import org.reactivestreams.tck.flow.support.InfiniteHelperPublisher;
30 
31 import java.util.concurrent.ExecutorService;
32 
33 /**
34  * Type which is able to create elements based on a seed {@code id} value.
35  * <p>
36  * Simplest implementations will simply return the incoming id as the element.
37  *
38  * @param <T> type of element to be delivered to the Subscriber
39  */
40 public abstract class WithHelperPublisher<T> {
41 
42   /** ExecutorService to be used by the provided helper {@link org.reactivestreams.Publisher} */
43   public abstract ExecutorService publisherExecutorService();
44 
45   /**
46    * Implement this method to match your expected element type.
47    * In case of implementing a simple Subscriber which is able to consume any kind of element simply return the
48    * incoming {@code element} element.
49    * <p>
50    * Sometimes the Subscriber may be limited in what type of element it is able to consume, this you may have to implement
51    * this method such that the emitted element matches the Subscribers requirements. Simplest implementations would be
52    * to simply pass in the {@code element} as payload of your custom element, such as appending it to a String or other identifier.
53    * <p>
54    * <b>Warning:</b> This method may be called concurrently by the helper publisher, thus it should be implemented in a
55    * thread-safe manner.
56    *
57    * @return element of the matching type {@code T} that will be delivered to the tested Subscriber
58    */
59   public abstract T createElement(int element);
60 
61   /**
62    * Helper method required for creating the Publisher to which the tested Subscriber will be subscribed and tested against.
63    * <p>
64    * By default an <b>asynchronously signalling Publisher</b> is provided, which will use {@link #createElement(int)}
65    * to generate elements type your Subscriber is able to consume.
66    * <p>
67    * Sometimes you may want to implement your own custom custom helper Publisher - to validate behaviour of a Subscriber
68    * when facing a synchronous Publisher for example. If you do, it MUST emit the exact number of elements asked for
69    * (via the {@code elements} parameter) and MUST also must treat the following numbers of elements in these specific ways:
70    * <ul>
71    *   <li>
72    *     If {@code elements} is {@code Long.MAX_VALUE} the produced stream must be infinite.
73    *   </li>
74    *   <li>
75    *     If {@code elements} is {@code 0} the {@code Publisher} should signal {@code onComplete} immediatly.
76    *     In other words, it should represent a "completed stream".
77    *   </li>
78    * </ul>
79    */
80   @SuppressWarnings("unchecked")
81   public Publisher<T> createHelperPublisher(long elements) {
82     final Function<Integer, T> mkElement = new Function<Integer, T>() {
83       @Override public T apply(Integer id) throws Throwable {
84         return createElement(id);
85       }
86     };
87 
88     if (elements > Integer.MAX_VALUE) return new InfiniteHelperPublisher(mkElement, publisherExecutorService());
89     else return new HelperPublisher(0, (int) elements, mkElement, publisherExecutorService());
90   }
91 
92 }
