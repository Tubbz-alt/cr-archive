 24 import com.sun.net.httpserver.*;

 25 import java.io.*;
 26 import java.net.InetAddress;
 27 import java.net.InetSocketAddress;
 28 import java.net.URI;
 29 import java.security.*;
 30 import java.util.*;
 31 import java.util.logging.*;
 32 import java.util.concurrent.Executors;
 33 import java.util.concurrent.ExecutorService;

 36 public class Server {
 37 
 38     HttpsServer server;
 39     final ExecutorService exec;
 40     final int port;
 41 
 42     // certfile: needs to be good or bad, ie. bad contains an otherwise valid
 43     // cert but whose CN contains a different host. good must be correct
 44 
 45     // assuming the TLS handshake succeeds, the server returns a 200 OK
 46     // response with a short text string.
 47     public Server(String certfile) throws Exception {
 48         initLogger();
 49         SSLContext ctx = getContext("TLSv1.2", certfile);
 50         Configurator cfg = new Configurator(ctx);
 51         InetSocketAddress addr = new InetSocketAddress(InetAddress.getLoopbackAddress(),0);

 52         server = HttpsServer.create(addr, 10);
 53         server.setHttpsConfigurator(cfg);
 54         server.createContext("/", new MyHandler());
 55         server.setExecutor((exec=Executors.newCachedThreadPool()));
 56         port = server.getAddress().getPort();
 57         System.out.println ("Listening on port " + port);
 58         server.start();
 59     }
 60 
 61     int getPort() {
 62         return port;
 63     }
 64 
 65     void stop() {
 66         server.stop(1);
 67         exec.shutdownNow();
 68     }
 69 
 70     SSLContext getContext(String protocol, String certfile) throws Exception {
 71         char[] passphrase = "passphrase".toCharArray();
 72         KeyStore ks = KeyStore.getInstance("JKS");
 73         ks.load(new FileInputStream(certfile), passphrase);
 74 
 75         KeyManagerFactory kmf = KeyManagerFactory.getInstance("SunX509");
 76         kmf.init(ks, passphrase);
 77 
 78         TrustManagerFactory tmf = TrustManagerFactory.getInstance("SunX509");
 79         tmf.init(ks);
 80 
 81         SSLContext ssl = SSLContext.getInstance(protocol);
 82         ssl.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
 83         return ssl;
 84     }
 85 
 86     Logger logger;
 87 
 88     void initLogger() {
 89         logger = Logger.getLogger("com.sun.net.httpserver");
 90         Handler h = new ConsoleHandler();
 91         logger.setLevel(Level.ALL);
 92         h.setLevel(Level.ALL);
 93         logger.addHandler(h);
 94     }
 95 
 96     String responseBody = "Greetings from localhost";
 97 
 98     class MyHandler implements HttpHandler {
 99 
100         @Override
101         public void handle(HttpExchange e) throws IOException {
102             System.out.println("Server: received " + e.getRequestURI());
103             InputStream is = e.getRequestBody();
104             byte[] buf = new byte[128];
105             while (is.read(buf) != -1);
106             is.close();
107             e.sendResponseHeaders(200, responseBody.length());
108             OutputStream os = e.getResponseBody();
109             os.write(responseBody.getBytes("ISO8859_1"));
110             os.close();
111         }
112     }
113 
114     class Configurator extends HttpsConfigurator {
115         public Configurator(SSLContext ctx) throws Exception {
116             super(ctx);
117         }
118 
119         public void configure(HttpsParameters params) {
120             SSLParameters p = getSSLContext().getDefaultSSLParameters();
121             for (String cipher : p.getCipherSuites())
122                 System.out.println("Cipher: " + cipher);
123             System.err.println("PArams = " + p);
124             params.setSSLParameters(p);
125         }
126     }
127 }
