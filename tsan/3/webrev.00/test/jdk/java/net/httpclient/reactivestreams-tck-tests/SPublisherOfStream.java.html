24 import org.reactivestreams.tck.TestEnvironment;
25 import org.reactivestreams.tck.flow.FlowPublisherVerification;
26 
27 import java.util.concurrent.Flow.Publisher;
28 import java.util.stream.LongStream;
29 import java.util.stream.Stream;
30 
31 /* See TckDriver.java for more information */
32 public class SPublisherOfStream
33         extends FlowPublisherVerification<Long> {
34 
35     public SPublisherOfStream() {
36         super(new TestEnvironment(450L));
37     }
38 
39     @Override
40     public Publisher<Long> createFlowPublisher(long nElements) {
41         Stream<Long> s = LongStream.range(0, nElements).boxed();
42         return S.publisherOfStream(s);
43     }
44 
45     @Override
46     public Publisher<Long> createFailedFlowPublisher() {
47         return null;
48     }
49 }
