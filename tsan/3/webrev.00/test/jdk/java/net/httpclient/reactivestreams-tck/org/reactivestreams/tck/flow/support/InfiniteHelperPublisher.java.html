24 package org.reactivestreams.tck.flow.support;
25 
26 import org.reactivestreams.example.unicast.AsyncIterablePublisher;
27 
28 import java.util.Iterator;
29 import java.util.concurrent.Executor;
30 
31 public class InfiniteHelperPublisher<T> extends AsyncIterablePublisher<T> {
32 
33     public InfiniteHelperPublisher(final Function<Integer, T> create, final Executor executor) {
34         super(new Iterable<T>() {
35           @Override public Iterator<T> iterator() {
36             return new Iterator<T>() {
37               private int at = 0;
38 
39               @Override public boolean hasNext() { return true; }
40               @Override public T next() {
41                 try {
42                   return create.apply(at++); // Wraps around on overflow
43                 } catch (Throwable t) {
44                   throw new IllegalStateException(
45                     String.format("Failed to create element in %s for id %s!", getClass().getSimpleName(), at - 1), t);
46                 }
47               }
48               @Override public void remove() { throw new UnsupportedOperationException(); }
49             };
50           }
51         }, executor);
52     }
53 }
