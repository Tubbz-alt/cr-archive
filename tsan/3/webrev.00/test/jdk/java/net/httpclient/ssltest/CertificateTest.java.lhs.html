 24 import java.io.File;

 25 import java.io.IOException;
 26 import java.net.URI;
 27 import java.net.http.HttpClient;
 28 import java.net.http.HttpResponse.BodyHandlers;
 29 import java.net.http.HttpRequest;
 30 import java.net.http.HttpResponse;


 31 import javax.net.ssl.SSLContext;
 32 import javax.net.ssl.SSLException;
 33 import javax.net.ssl.SSLParameters;
 34 import static java.net.http.HttpClient.Builder.NO_PROXY;


 35 
 36 /*
 37  * @test

 38  * @build Server CertificateTest
 39  * @run main/othervm CertificateTest good.keystore expectSuccess
 40  * @run main/othervm CertificateTest bad.keystore expectFailure
 41  * @run main/othervm
 42  *      -Djdk.internal.httpclient.disableHostnameVerification
 43  *       CertificateTest bad.keystore expectSuccess
 44  * @run main/othervm
 45  *      -Djdk.internal.httpclient.disableHostnameVerification=true
 46  *       CertificateTest bad.keystore expectSuccess
 47  * @run main/othervm
 48  *      -Djdk.internal.httpclient.disableHostnameVerification=false
 49  *       CertificateTest bad.keystore expectFailure
 50  * @run main/othervm
 51  *      -Djdk.internal.httpclient.disableHostnameVerification=xxyyzz
 52  *       CertificateTest bad.keystore expectFailure
 53  * @run main/othervm CertificateTest loopback.keystore expectSuccess
 54  */
 55 
 56 /**
 57  * The test runs a number of times. In all cases it uses a valid self-signed certificate
 58  * that is installed in the trust store (so is trusted) and the same cert is supplied
 59  * by the server for its own identity. Two servers on two different ports are used
 60  * on the remote end.
 61  *
 62  * For the "good" run the cert contains the correct hostname of the target server
 63  * and therefore should be accepted by the cert checking code in the client.
 64  * For the "bad" run, the cert contains an invalid hostname, and should be rejected.



 65  */
 66 public class CertificateTest {
 67     static SSLContext ctx;
 68     static SSLParameters params;
 69     static boolean expectSuccess;
 70     static String trustStoreProp;
 71     static Server server;
 72     static int port;
 73 
 74     static String TESTSRC = System.getProperty("test.src");
 75     public static void main(String[] args) throws Exception
 76     {
 77         try {
 78             String keystore = args[0];
 79             trustStoreProp = TESTSRC + File.separatorChar + keystore;
 80 
 81             String passOrFail = args[1];
 82 
 83             if (passOrFail.equals("expectSuccess")) {
 84                 expectSuccess = true;
 85             } else {
 86                 expectSuccess = false;
 87             }
 88             server = new Server(trustStoreProp);


 89             port = server.getPort();
 90             System.setProperty("javax.net.ssl.trustStore", trustStoreProp);
 91             System.setProperty("javax.net.ssl.trustStorePassword", "passphrase");
 92             init();
 93             test(args);
 94         } finally {
 95             server.stop();


 96         }
 97     }
 98 
 99     static void init() throws Exception
100     {
101         ctx = SSLContext.getDefault();
102         params = ctx.getDefaultSSLParameters();
103         //params.setProtocols(new String[] { "TLSv1.2" });



104     }
105 
106     static void test(String[] args) throws Exception
107     {
108         String uri_s;
109         if (args[0].equals("loopback.keystore"))
110             uri_s = "https://127.0.0.1:" + Integer.toString(port) + "/foo";
111         else
112             uri_s = "https://localhost:" + Integer.toString(port) + "/foo";
113         String error = null;
114         Exception exception = null;
115         System.out.println("Making request to " + uri_s);


116         HttpClient client = HttpClient.newBuilder()
117                 .proxy(NO_PROXY)
118                 .sslContext(ctx)
119                 .sslParameters(params)
120                 .build();
121 
122         HttpRequest request = HttpRequest.newBuilder(new URI(uri_s))
123                 .version(HttpClient.Version.HTTP_1_1)
124                 .GET()
125                 .build();
126 
127         try {
128             HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
129             System.out.printf("Status code %d received\n", response.statusCode());
130             if (expectSuccess && response.statusCode() != 200)
131                 error = "Test failed: good: status should be 200";
132             else if (!expectSuccess)
133                 error = "Test failed: bad: status should not be 200";
134         } catch (IOException e) {
135             // there must be an SSLException as the exception or cause
136             checkExceptionOrCause(SSLException.class, e);
137             System.err.println("Caught Exception " + e + ". expectSuccess = " + expectSuccess);
138             exception = e;
139             if (expectSuccess)
140                 error = "Test failed: expectSuccess:true, but got unexpected exception";
141         }
142         if (error != null)
143             throw new RuntimeException(error, exception);
144     }
145 
146     static void checkExceptionOrCause(Class<? extends Throwable> clazz, Throwable t) {
147         final Throwable original = t;
148         do {
149             if (clazz.isInstance(t)) {
150                 System.out.println("Found expected exception/cause: " + t);
151                 return; // found
152             }
153         } while ((t = t.getCause()) != null);
154         original.printStackTrace(System.out);
155         throw new RuntimeException("Expected " + clazz + "in " + original);
156     }
157 }
