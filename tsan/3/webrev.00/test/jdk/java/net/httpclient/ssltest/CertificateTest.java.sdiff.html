 61 /**
 62  * The test runs a number of times. In all cases it uses a valid self-signed certificate
 63  * that is installed in the trust store (so is trusted) and the same cert is supplied
 64  * by the server for its own identity. Two servers on two different ports are used
 65  * on the remote end.
 66  *
 67  * The GOOD_CERT cert contains the correct hostname of the target server
 68  * and therefore should be accepted by the cert checking code in the client.
 69  * The BAD_CERT cert contains an invalid hostname, and should be rejected.
 70  * The LOOPBACK_CERT cert contains an invalid hostname, but it also contains a
 71  * subject alternative name for IP address 127.0.0.1, so it should be accepted
 72  * for this address.
 73  */

 81     public static void main(String[] args) throws Exception
 82     {
 83         try {
 84             String certName = args[0];


 85             String passOrFail = args[1];
 86 
 87             if (passOrFail.equals("expectSuccess")) {
 88                 expectSuccess = true;
 89             } else {
 90                 expectSuccess = false;
 91             }

 93             cert = Cert.valueOf(certName);
 94             server = new Server(getSSLContext(cert));
 95             port = server.getPort();

 97         } finally {

 99                 server.stop();

104     private static SSLContext getSSLContext(Cert cert) throws Exception {
105         SSLContextBuilder builder = SSLContextBuilder.builder();
106         builder.trustStore(
107                 KeyStoreUtils.createTrustStore(new String[] { cert.certStr }));
108         builder.keyStore(KeyStoreUtils.createKeyStore(
109                 new KeyEntry[] { new KeyEntry(cert.keyAlgo,
110                         cert.keyStr, new String[] { cert.certStr }) }));
111         return builder.build();
112     }
