 52 public class LargeResponseContent {
 53     final ServerSocket server;
 54     final int port;
 55 
 56     public LargeResponseContent() throws Exception {
 57         server = new ServerSocket(0, 10, InetAddress.getLoopbackAddress());
 58         Thread serverThread = new Thread(this::handleConnection);
 59         serverThread.setDaemon(false);
 60         port = server.getLocalPort();
 61         serverThread.start();
 62     }
 63 
 64     void runClient() throws IOException, InterruptedException {
 65         URI uri = URIBuilder.newBuilder()
 66             .scheme("http")
 67             .loopback()
 68             .port(port)
 69             .path("/foo")
 70             .buildUnchecked();
 71         System.out.println("URI: " + uri);
 72         HttpClient client = HttpClient.newHttpClient();
 73         HttpRequest request = HttpRequest.newBuilder(uri)
 74                 .GET()
 75                 .build();
 76         HttpResponse<Long> response = client.send(request, new ClientHandler());
 77         System.out.println("Response code = " + response.statusCode());
 78         long blen = response.body();
 79         if (blen != CONTENT_LEN)
 80             throw new RuntimeException("wrong content length");
 81     }
 82 
 83     public static void main(String[] args) throws Exception {
 84         System.out.println ("CONTENT_LEN = " + CONTENT_LEN);
 85         System.out.println ("CLEN_STR = " + CLEN_STR);
 86         LargeResponseContent test = new LargeResponseContent();
 87         test.runClient();
 88     }
 89 
 90     static class ClientHandler implements HttpResponse.BodyHandler<Long> {
 91 
 92         @Override
 93         public HttpResponse.BodySubscriber<Long> apply(HttpResponse.ResponseInfo responseInfo) {
 94             HttpHeaders headers = responseInfo.headers();
 95             headers.firstValue("content-length");
 96             long  clen = headers.firstValueAsLong("content-length").orElse(-1);
 97             if (clen != CONTENT_LEN)
 98                 return new Subscriber(new RuntimeException("Wrong content length received"));
 99             return new Subscriber(null);
100         }
101     }
102 
103     static class Subscriber implements HttpResponse.BodySubscriber<Long> {
104         final CompletableFuture<Long> cf = new CompletableFuture<>();
105         volatile Flow.Subscription subscription;
106         volatile long counter = 0;
107 
108         Subscriber(Throwable t) {
109             if (t != null)
110                 cf.completeExceptionally(t);
111         }
112 
113         @Override
114         public CompletionStage<Long> getBody() {
115             return cf;
116         }
117 
118         @Override
119         public void onSubscribe(Flow.Subscription subscription) {
120             this.subscription = subscription;
121             subscription.request(Long.MAX_VALUE);
122         }
123 
124         @Override
125         public void onNext(List<ByteBuffer> item) {
126             long v = 0;
127             for (ByteBuffer b : item)
128                 v+= b.remaining();
129             counter += v;
130         }
131 
132         @Override
133         public void onError(Throwable throwable) {
134             throwable.printStackTrace();
135         }
136 
137         @Override
138         public void onComplete() {
139             cf.complete(counter);
140         }
141     }
142 
143     static final long CONTENT_LEN = Integer.MAX_VALUE + 1000L;
144     static final String CLEN_STR = Long.valueOf(CONTENT_LEN).toString();
145 
146     static String RESPONSE = "HTTP/1.1 200 OK\r\n" +
147             "Content-length: " + CLEN_STR + "\r\n" +
148             "\r\n";
149 
150 
151     void readHeaders(InputStream is) throws IOException {
152         String s = "";
153         byte[] buf = new byte[128];
154         while (!s.endsWith("\r\n\r\n")) {
155             int c = is.read(buf);
156             String f = new String(buf, 0, c, StandardCharsets.ISO_8859_1);
157             s = s + f;
158         }
159     }
160 
161     public void handleConnection() {
162         long remaining = CONTENT_LEN;
163         try {
164             Socket socket = server.accept();
165             InputStream is = socket.getInputStream();
166             readHeaders(is); // read first byte
167             OutputStream os = socket.getOutputStream();
168             os.write(RESPONSE.getBytes());
169             byte[] buf = new byte[64 * 1024];
170             while (remaining > 0) {
171                 int amount = (int)Math.min(remaining, buf.length);
172                 os.write(buf, 0, amount);
173                 remaining -= amount;
174             }
175             System.out.println("Server: finished writing");
176             os.close();
177 
178         } catch (IOException e) {
179             long sent = CONTENT_LEN - remaining;
180             System.out.println("Sent " + sent);
181             e.printStackTrace();
182         }
183     }
