24 package org.reactivestreams.tck.flow;
25 
26 import org.reactivestreams.Publisher;
27 import org.reactivestreams.FlowAdapters;
28 import org.reactivestreams.tck.PublisherVerification;
29 import org.reactivestreams.tck.TestEnvironment;
30 
31 import java.util.concurrent.Flow;
32 
33 /**
34  * Provides tests for verifying a Java 9+ {@link java.util.concurrent.Flow.Publisher} specification rules.
35  *
36  * @see java.util.concurrent.Flow.Publisher
37  */
38 public abstract class FlowPublisherVerification<T> extends PublisherVerification<T> {
39 
40   public FlowPublisherVerification(TestEnvironment env, long publisherReferenceGCTimeoutMillis) {
41     super(env, publisherReferenceGCTimeoutMillis);
42   }
43 
44   public FlowPublisherVerification(TestEnvironment env) {
45     super(env);
46   }
47 
48   @Override
49   final public Publisher<T> createPublisher(long elements) {
50     final Flow.Publisher<T> flowPublisher = createFlowPublisher(elements);
51     return FlowAdapters.toPublisher(flowPublisher);
52   }
53   /**
54    * This is the main method you must implement in your test incarnation.
55    * It must create a Publisher for a stream with exactly the given number of elements.
56    * If `elements` is `Long.MAX_VALUE` the produced stream must be infinite.
57    */
58   public abstract Flow.Publisher<T> createFlowPublisher(long elements);
59 
60   @Override
61   final public Publisher<T> createFailedPublisher() {
62     final Flow.Publisher<T> failed = createFailedFlowPublisher();
63     if (failed == null) return null; // because `null` means "SKIP" in createFailedPublisher
64     else return FlowAdapters.toPublisher(failed);
65   }
66   /**
67    * By implementing this method, additional TCK tests concerning a "failed" publishers will be run.
68    *
69    * The expected behaviour of the {@link Flow.Publisher} returned by this method is hand out a subscription,
70    * followed by signalling {@code onError} on it, as specified by Rule 1.9.
71    *
72    * If you ignore these additional tests, return {@code null} from this method.
73    */
74   public abstract Flow.Publisher<T> createFailedFlowPublisher();
75 }
