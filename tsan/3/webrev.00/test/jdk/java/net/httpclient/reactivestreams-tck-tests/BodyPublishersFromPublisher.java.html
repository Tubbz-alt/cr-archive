24 import org.reactivestreams.tck.TestEnvironment;
25 import org.reactivestreams.tck.flow.FlowPublisherVerification;
26 
27 import java.net.http.HttpRequest.BodyPublishers;
28 import java.nio.ByteBuffer;
29 import java.util.concurrent.Flow.Publisher;
30 import java.util.stream.Stream;
31 
32 /* See TckDriver.java for more information */
33 public class BodyPublishersFromPublisher
34         extends FlowPublisherVerification<ByteBuffer> {
35 
36     public BodyPublishersFromPublisher() {
37         super(new TestEnvironment(450L));
38     }
39 
40     @Override
41     public Publisher<ByteBuffer> createFlowPublisher(long nElements) {
42         Stream<ByteBuffer> buffers =
43                 Stream.generate(() -> S.bufferOfNRandomBytes(1024))
44                       .limit(nElements);
45         Publisher<ByteBuffer> pub = S.publisherOfStream(buffers);
46         return BodyPublishers.fromPublisher(pub);
47     }
48 
49     @Override
50     public Publisher<ByteBuffer> createFailedFlowPublisher() {
51         return BodyPublishers.fromPublisher(S.newErroredPublisher());
52     }
53 }
