24 package org.reactivestreams.example.unicast;
25 
26 import java.util.Collections;
27 import java.util.Iterator;
28 import java.util.concurrent.Executor;
29 import org.reactivestreams.Subscription;
30 import org.reactivestreams.Subscriber;
31 import org.reactivestreams.Publisher;
32 
33 public class NumberIterablePublisher extends AsyncIterablePublisher<Integer> {
34     public NumberIterablePublisher(final int from, final int to, final Executor executor) {
35         super(new Iterable<Integer>() {
36           { if(from > to) throw new IllegalArgumentException("from must be equal or greater than to!"); }
37           @Override public Iterator<Integer> iterator() {
38             return new Iterator<Integer>() {
39               private int at = from;
40               @Override public boolean hasNext() { return at < to; }
41               @Override public Integer next() {
42                 if (!hasNext()) return Collections.<Integer>emptyList().iterator().next();
43                 else return at++;
44               }
45               @Override public void remove() { throw new UnsupportedOperationException(); }
46             };
47           }
48         }, executor);
49     }
50 }
