 24 import com.sun.net.httpserver.*;
 25 
 26 import java.io.*;
 27 import java.net.InetAddress;
 28 import java.net.InetSocketAddress;
 29 import java.util.logging.*;
 30 import java.util.concurrent.Executors;
 31 import java.util.concurrent.ExecutorService;
 32 import javax.net.ssl.*;
 33 
 34 public class Server {
 35 
 36     HttpsServer server;
 37     final ExecutorService exec;
 38     final int port;
 39 
 40     // certfile: needs to be good or bad, ie. bad contains an otherwise valid
 41     // cert but whose CN contains a different host. good must be correct
 42 
 43     // assuming the TLS handshake succeeds, the server returns a 200 OK
 44     // response with a short text string.
 45     public Server(SSLContext ctx) throws Exception {
 46         initLogger();
 47         Configurator cfg = new Configurator(ctx);
 48         InetSocketAddress addr = new InetSocketAddress(
 49                 InetAddress.getLoopbackAddress(), 0);
 50         server = HttpsServer.create(addr, 10);
 51         server.setHttpsConfigurator(cfg);
 52         server.createContext("/", new MyHandler());
 53         server.setExecutor((exec = Executors.newCachedThreadPool()));
 54         port = server.getAddress().getPort();
 55         System.out.println("Listening on port " + port);
 56         server.start();
 57     }
 58 
 59     int getPort() {
 60         return port;
 61     }
 62 
 63     void stop() {
 64         server.stop(1);
 65         exec.shutdownNow();
 66     }
 67 
 68     Logger logger;
 69 
 70     void initLogger() {
 71         logger = Logger.getLogger("com.sun.net.httpserver");
 72         Handler h = new ConsoleHandler();
 73         logger.setLevel(Level.ALL);
 74         h.setLevel(Level.ALL);
 75         logger.addHandler(h);
 76     }
 77 
 78     String responseBody = "Greetings from localhost";
 79 
 80     class MyHandler implements HttpHandler {
 81 
 82         @Override
 83         public void handle(HttpExchange e) throws IOException {
 84             System.out.println("Server: received " + e.getRequestURI());
 85             InputStream is = e.getRequestBody();
 86             byte[] buf = new byte[128];
 87             while (is.read(buf) != -1);
 88             is.close();
 89             e.sendResponseHeaders(200, responseBody.length());
 90             OutputStream os = e.getResponseBody();
 91             os.write(responseBody.getBytes("ISO8859_1"));
 92             os.close();
 93         }
 94     }
 95 
 96     class Configurator extends HttpsConfigurator {
 97         public Configurator(SSLContext ctx) throws Exception {
 98             super(ctx);
 99         }
100 
101         public void configure(HttpsParameters params) {
102             SSLParameters p = getSSLContext().getDefaultSSLParameters();
103             for (String cipher : p.getCipherSuites())
104                 System.out.println("Cipher: " + cipher);
105             System.err.println("Params = " + p);
106             params.setSSLParameters(p);
107         }
108     }
109 }
