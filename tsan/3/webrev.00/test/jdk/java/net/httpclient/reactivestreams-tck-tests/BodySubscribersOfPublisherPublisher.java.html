24 import org.reactivestreams.tck.TestEnvironment;
25 import org.reactivestreams.tck.flow.FlowPublisherVerification;
26 
27 import java.net.http.HttpResponse.BodySubscriber;
28 import java.net.http.HttpResponse.BodySubscribers;
29 import java.nio.ByteBuffer;
30 import java.util.List;
31 import java.util.concurrent.Flow.Publisher;
32 import java.util.stream.Stream;
33 
34 /* See TckDriver.java for more information */
35 public class BodySubscribersOfPublisherPublisher
36         extends FlowPublisherVerification<List<ByteBuffer>> {
37 
38     public BodySubscribersOfPublisherPublisher() {
39         super(new TestEnvironment(450L));
40     }
41 
42     @Override
43     public Publisher<List<ByteBuffer>> createFlowPublisher(long nElements) {
44         BodySubscriber<Publisher<List<ByteBuffer>>> sub =
45                 BodySubscribers.ofPublisher();
46         Stream<List<ByteBuffer>> buffers =
47                 Stream.generate(() -> S.listOfBuffersFromBufferOfNBytes(1024))
48                       .limit(nElements);
49         Publisher<List<ByteBuffer>> pub = S.publisherOfStream(buffers);
50         pub.subscribe(sub);
51         return sub.getBody().toCompletableFuture().join();
52     }
53 
54     @Override
55     public Publisher<List<ByteBuffer>> createFailedFlowPublisher() {
56         BodySubscriber<Publisher<List<ByteBuffer>>> sub =
57                 BodySubscribers.ofPublisher();
58         Publisher<List<ByteBuffer>> pub = S.newErroredPublisher();
59         pub.subscribe(sub);
60         return sub.getBody().toCompletableFuture().join();
61     }
62 
63     @Override
64     public long maxElementsFromPublisher() {
65         return 21;
66     }
67 }
