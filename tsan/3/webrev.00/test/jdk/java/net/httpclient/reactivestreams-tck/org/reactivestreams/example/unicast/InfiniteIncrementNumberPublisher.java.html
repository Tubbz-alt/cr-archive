24 package org.reactivestreams.example.unicast;
25 
26 import java.util.Iterator;
27 import java.util.concurrent.Executor;
28 
29 import org.reactivestreams.Subscription;
30 import org.reactivestreams.Subscriber;
31 import org.reactivestreams.Publisher;
32 
33 public class InfiniteIncrementNumberPublisher extends AsyncIterablePublisher<Integer> {
34     public InfiniteIncrementNumberPublisher(final Executor executor) {
35         super(new Iterable<Integer>() {
36           @Override public Iterator<Integer> iterator() {
37             return new Iterator<Integer>() {
38               private int at = 0;
39               @Override public boolean hasNext() { return true; }
40               @Override public Integer next() { return at++; } // Wraps around on overflow
41               @Override public void remove() { throw new UnsupportedOperationException(); }
42             };
43           }
44         }, executor);
45     }
46 }
