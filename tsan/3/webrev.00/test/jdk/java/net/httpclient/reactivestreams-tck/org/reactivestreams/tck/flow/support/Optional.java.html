24 package org.reactivestreams.tck.flow.support;
25 
26 import java.util.NoSuchElementException;
27 
28 // simplest possible version of Scala's Option type
29 public abstract class Optional<T> {
30 
31   private static final Optional<Object> NONE = new Optional<Object>() {
32     @Override
33     public Object get() {
34       throw new NoSuchElementException(".get call on None!");
35     }
36 
37     @Override
38     public boolean isEmpty() {
39       return true;
40     }
41   };
42 
43   private Optional() {
44   }
45 
46   @SuppressWarnings("unchecked")
47   public static <T> Optional<T> empty() {
48     return (Optional<T>) NONE;
49   }
50 
51   @SuppressWarnings("unchecked")
52   public static <T> Optional<T> of(T it) {
53     if (it == null) return (Optional<T>) Optional.NONE;
54     else return new Some(it);
55   }
56 
57   public abstract T get();
58 
59   public abstract boolean isEmpty();
60 
61   public boolean isDefined() {
62     return !isEmpty();
63   }
64 
65   public static class Some<T> extends Optional<T> {
66     private final T value;
67 
68     Some(T value) {
69       this.value = value;
70     }
71 
72     @Override
73     public T get() {
74       return value;
75     }
76 
77     @Override
78     public boolean isEmpty() {
79       return false;
80     }
81 
82     @Override
83     public String toString() {
84       return String.format("Some(%s)", value);
85     }
86   }
87 
88   @Override
89   public String toString() {
90     return "None";
91   }
92 }
