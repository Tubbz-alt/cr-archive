24 package org.reactivestreams;
25 
26 /**
27  * A {@link Subscription} represents a one-to-one lifecycle of a {@link Subscriber} subscribing to a {@link Publisher}.
28  * <p>
29  * It can only be used once by a single {@link Subscriber}.
30  * <p>
31  * It is used to both signal desire for data and cancel demand (and allow resource cleanup).
32  *
33  */
34 public interface Subscription {
35     /**
36      * No events will be sent by a {@link Publisher} until demand is signaled via this method.
37      * <p>
38      *  It can be called however often and whenever neededâ€”but if the outstanding cumulative demand ever becomes Long.MAX_VALUE or more,
39      *  it may be treated by the {@link Publisher} as "effectively unbounded".
40      * <p>
41      * Whatever has been requested can be sent by the {@link Publisher} so only signal demand for what can be safely handled.
42      * <p>
43      * A {@link Publisher} can send less than is requested if the stream ends but
44      * then must emit either {@link Subscriber#onError(Throwable)} or {@link Subscriber#onComplete()}.
45      *
46      * @param n the strictly positive number of elements to requests to the upstream {@link Publisher}
47      */
48     public void request(long n);
49 
50     /**
51      * Request the {@link Publisher} to stop sending data and clean up resources.
52      * <p>
53      * Data may still be sent to meet previously signalled demand after calling cancel.
54      */
55     public void cancel();
56 }
