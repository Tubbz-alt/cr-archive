24 import org.reactivestreams.tck.TestEnvironment;
25 import org.reactivestreams.tck.flow.FlowSubscriberBlackboxVerification;
26 
27 import java.net.http.HttpResponse.BodySubscribers;
28 import java.nio.ByteBuffer;
29 import java.nio.charset.StandardCharsets;
30 import java.util.List;
31 import java.util.concurrent.Flow.Subscriber;
32 
33 /* See TckDriver.java for more information */
34 public class BodySubscribersOfString
35         extends FlowSubscriberBlackboxVerification<List<ByteBuffer>> {
36 
37     public BodySubscribersOfString() {
38         super(new TestEnvironment(450L));
39     }
40 
41     @Override
42     public Subscriber<List<ByteBuffer>> createFlowSubscriber() {
43         return BodySubscribers.ofString(StandardCharsets.UTF_8);
44     }
45 
46     @Override
47     public List<ByteBuffer> createElement(int element) {
48         return S.scatterBuffer(
49                 S.bufferOfNRandomASCIIBytes(element % 17));
50     }
51 }
