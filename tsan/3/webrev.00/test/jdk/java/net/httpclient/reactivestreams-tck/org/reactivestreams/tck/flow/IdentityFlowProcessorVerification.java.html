24 package org.reactivestreams.tck.flow;
25 
26 import org.reactivestreams.*;
27 import org.reactivestreams.tck.IdentityProcessorVerification;
28 import org.reactivestreams.tck.TestEnvironment;
29 import org.reactivestreams.tck.flow.support.SubscriberWhiteboxVerificationRules;
30 import org.reactivestreams.tck.flow.support.PublisherVerificationRules;
31 
32 import java.util.concurrent.Flow;
33 
34 public abstract class IdentityFlowProcessorVerification<T> extends IdentityProcessorVerification<T>
35   implements SubscriberWhiteboxVerificationRules, PublisherVerificationRules {
36 
37   public IdentityFlowProcessorVerification(TestEnvironment env) {
38     super(env);
39   }
40 
41   public IdentityFlowProcessorVerification(TestEnvironment env, long publisherReferenceGCTimeoutMillis) {
42     super(env, publisherReferenceGCTimeoutMillis);
43   }
44 
45   public IdentityFlowProcessorVerification(TestEnvironment env, long publisherReferenceGCTimeoutMillis, int processorBufferSize) {
46     super(env, publisherReferenceGCTimeoutMillis, processorBufferSize);
47   }
48 
49   /**
50    * By implementing this method, additional TCK tests concerning a "failed" Flow publishers will be run.
51    *
52    * The expected behaviour of the {@link Flow.Publisher} returned by this method is hand out a subscription,
53    * followed by signalling {@code onError} on it, as specified by Rule 1.9.
54    *
55    * If you want to ignore these additional tests, return {@code null} from this method.
56    */
57   protected abstract Flow.Publisher<T> createFailedFlowPublisher();
58 
59   /**
60    * This is the main method you must implement in your test incarnation.
61    * It must create a {@link Flow.Processor}, which simply forwards all stream elements from its upstream
62    * to its downstream. It must be able to internally buffer the given number of elements.
63    *
64    * @param bufferSize number of elements the processor is required to be able to buffer.
65    */
66   protected abstract Flow.Processor<T,T> createIdentityFlowProcessor(int bufferSize);
67 
68   @Override
69   public final Processor<T, T> createIdentityProcessor(int bufferSize) {
70     return FlowAdapters.toProcessor(createIdentityFlowProcessor(bufferSize));
71   }
72 
73   @Override
74   public final Publisher<T> createFailedPublisher() {
75     Flow.Publisher<T> failed = createFailedFlowPublisher();
76     if (failed == null) return null; // because `null` means "SKIP" in createFailedPublisher
77     else return FlowAdapters.toPublisher(failed);
78   }
79 
80 }
