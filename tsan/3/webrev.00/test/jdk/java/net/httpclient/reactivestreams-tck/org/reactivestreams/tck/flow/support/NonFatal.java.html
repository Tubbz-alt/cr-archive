24 package org.reactivestreams.tck.flow.support;
25 
26 
27 /**
28  * Copy of scala.control.util.NonFatal in order to not depend on scala-library
29  */
30 public class NonFatal {
31   private NonFatal() {
32     // no instances, please.
33   }
34 
35   /**
36    * Returns true if the provided `Throwable` is to be considered non-fatal, or false if it is to be considered fatal
37    *
38    * @param t throwable to be matched for fatal-ness
39    * @return true if is a non-fatal throwable, false otherwise
40    */
41   public static boolean isNonFatal(Throwable t) {
42     if (t instanceof StackOverflowError) {
43       // StackOverflowError ok even though it is a VirtualMachineError
44       return true;
45     } else if (t instanceof VirtualMachineError ||
46         t instanceof ThreadDeath ||
47         t instanceof InterruptedException ||
48         t instanceof LinkageError) {
49       // VirtualMachineError includes OutOfMemoryError and other fatal errors
50       return false;
51     } else {
52       return true;
53     }
54   }
55 }
