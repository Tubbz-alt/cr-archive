 24 import static java.net.http.HttpClient.Builder.NO_PROXY;
 25 
 26 import java.io.IOException;
 27 import java.net.URI;
 28 import java.net.http.HttpClient;

 29 import java.net.http.HttpRequest;
 30 import java.net.http.HttpResponse;
 31 import java.net.http.HttpResponse.BodyHandlers;
 32 
 33 import javax.net.ssl.SSLContext;
 34 import javax.net.ssl.SSLException;
 35 
 36 import jdk.test.lib.security.KeyEntry;
 37 import jdk.test.lib.security.KeyStoreUtils;
 38 import jdk.test.lib.security.SSLContextBuilder;

 40 /*
 41  * @test
 42  * @library /test/lib
 43  * @build Server CertificateTest
 44  * @run main/othervm CertificateTest GOOD_CERT expectSuccess
 45  * @run main/othervm CertificateTest BAD_CERT expectFailure
 46  * @run main/othervm
 47  *      -Djdk.internal.httpclient.disableHostnameVerification
 48  *       CertificateTest BAD_CERT expectSuccess
 49  * @run main/othervm
 50  *      -Djdk.internal.httpclient.disableHostnameVerification=true
 51  *       CertificateTest BAD_CERT expectSuccess
 52  * @run main/othervm
 53  *      -Djdk.internal.httpclient.disableHostnameVerification=false
 54  *       CertificateTest BAD_CERT expectFailure
 55  * @run main/othervm
 56  *      -Djdk.internal.httpclient.disableHostnameVerification=xxyyzz
 57  *       CertificateTest BAD_CERT expectFailure
 58  * @run main/othervm CertificateTest LOOPBACK_CERT expectSuccess
 59  */
 60 
 61 /**
 62  * The test runs a number of times. In all cases it uses a valid self-signed certificate
 63  * that is installed in the trust store (so is trusted) and the same cert is supplied
 64  * by the server for its own identity. Two servers on two different ports are used
 65  * on the remote end.
 66  *
 67  * The GOOD_CERT cert contains the correct hostname of the target server
 68  * and therefore should be accepted by the cert checking code in the client.
 69  * The BAD_CERT cert contains an invalid hostname, and should be rejected.
 70  * The LOOPBACK_CERT cert contains an invalid hostname, but it also contains a
 71  * subject alternative name for IP address 127.0.0.1, so it should be accepted
 72  * for this address.
 73  */
 74 public class CertificateTest {
 75 
 76     private static Cert cert;
 77     static boolean expectSuccess;

 78     static Server server;
 79     static int port;
 80 

 81     public static void main(String[] args) throws Exception
 82     {
 83         try {
 84             String certName = args[0];


 85             String passOrFail = args[1];
 86 
 87             if (passOrFail.equals("expectSuccess")) {
 88                 expectSuccess = true;
 89             } else {
 90                 expectSuccess = false;
 91             }
 92 
 93             cert = Cert.valueOf(certName);
 94             server = new Server(getSSLContext(cert));
 95             port = server.getPort();
 96             test(cert);



 97         } finally {
 98             if (server != null) {
 99                 server.stop();
100             }
101         }
102     }
103 
104     private static SSLContext getSSLContext(Cert cert) throws Exception {
105         SSLContextBuilder builder = SSLContextBuilder.builder();
106         builder.trustStore(
107                 KeyStoreUtils.createTrustStore(new String[] { cert.certStr }));
108         builder.keyStore(KeyStoreUtils.createKeyStore(
109                 new KeyEntry[] { new KeyEntry(cert.keyAlgo,
110                         cert.keyStr, new String[] { cert.certStr }) }));
111         return builder.build();
112     }
113 
114     static void test(Cert cert) throws Exception
115     {
116         String uri_s;
117         if (cert == Cert.LOOPBACK_CERT)
118             uri_s = "https://127.0.0.1:" + Integer.toString(port) + "/foo";
119         else
120             uri_s = "https://localhost:" + Integer.toString(port) + "/foo";
121         String error = null;
122         Exception exception = null;
123         System.out.println("Making request to " + uri_s);
124 
125         SSLContext ctx = getSSLContext(cert);
126         HttpClient client = HttpClient.newBuilder()
127                 .proxy(NO_PROXY)
128                 .sslContext(ctx)
129                 .sslParameters(ctx.getDefaultSSLParameters())
130                 .build();
131 
132         HttpRequest request = HttpRequest.newBuilder(new URI(uri_s))
133                 .version(HttpClient.Version.HTTP_1_1)
134                 .GET()
135                 .build();
136 
137         try {
138             HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
139             System.out.printf("Status code %d received\n", response.statusCode());
140             if (expectSuccess && response.statusCode() != 200)
141                 error = "Test failed: good: status should be 200";
142             else if (!expectSuccess)
143                 error = "Test failed: bad: status should not be 200";
144         } catch (IOException e) {
145             // there must be an SSLException as the exception or cause
146             checkExceptionOrCause(SSLException.class, e);
147             System.err.println("Caught Exception " + e + ". expectSuccess = " + expectSuccess);
148             exception = e;
149             if (expectSuccess)
150                 error = "Test failed: expectSuccess:true, but got unexpected exception";
151         }
152         if (error != null)
153             throw new RuntimeException(error, exception);
154     }
155 
156     static void checkExceptionOrCause(Class<? extends Throwable> clazz, Throwable t) {
157         final Throwable original = t;
158         do {
159             if (clazz.isInstance(t)) {
160                 System.out.println("Found expected exception/cause: " + t);
161                 return; // found
162             }
163         } while ((t = t.getCause()) != null);
164         original.printStackTrace(System.out);
165         throw new RuntimeException("Expected " + clazz + "in " + original);
166     }
167 }
