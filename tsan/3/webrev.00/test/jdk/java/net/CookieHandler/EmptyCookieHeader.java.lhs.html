24 /*
25  * @test
26  * @bug 8015799
27  * @modules jdk.httpserver
28  * @summary HttpURLConnection.getHeaderFields() throws IllegalArgumentException



29  */
30 
31 import com.sun.net.httpserver.*;
32 import java.io.IOException;
33 import java.io.OutputStream;
34 import java.net.*;
35 import java.util.*;

36 
37 public class EmptyCookieHeader {
38 
39     public static void main(String[] args) throws Exception {
40         new EmptyCookieHeader().runTest();
41     }
42 
43     public void runTest() throws Exception {
44         final CookieHandler oldHandler = CookieHandler.getDefault();
45         CookieHandler.setDefault(new TestCookieHandler());
46         HttpServer s = HttpServer.create(new InetSocketAddress(0), 0);

47         try {
48             startServer(s);
49             URL url = new URL("http://localhost:" + s.getAddress().getPort() + "/");
50             HttpURLConnection c = (HttpURLConnection)url.openConnection();





51             c.getHeaderFields();
52         } finally {
53             CookieHandler.setDefault(oldHandler);
54             s.stop(0);
55         }
56     }
57 
58     static void startServer(HttpServer server) throws IOException {
59         server.createContext("/", new EmptyCookieHandler());
60         server.start();
61     }
62 
63     static class EmptyCookieHandler implements HttpHandler {
64 
65         @Override
66         public void handle(HttpExchange exchange) throws IOException {
67             String requestMethod = exchange.getRequestMethod();
68             if (requestMethod.equalsIgnoreCase("GET")) {
69                 Headers responseHeaders = exchange.getResponseHeaders();
70                 responseHeaders.set("Content-Type", "text/plain");
71                 responseHeaders.set("Date", "June 13th 2012");
72 
73                 // No domain value set
74                 responseHeaders.set("Set-Cookie2", "name=value");
75                 responseHeaders.set("Set-Cookie2", "");
76                 exchange.sendResponseHeaders(200, 0);
77                 try (OutputStream os = exchange.getResponseBody()) {
78                     String str = "This is what the server sent!";
79                     os.write(str.getBytes());
80                 }
81             }
82         }
83     }
84 
85     class TestCookieHandler extends CookieHandler {
86         @Override
87         public Map<String,List<String>> get(URI uri,
88                                             Map<String,List<String>> respH) {
89             return new HashMap<>();
90         }
91 
92         @Override
93         public void put(URI uri, Map<String,List<String >> respH) { }
94     }
95 }
