24 /**
25  * @test
26  * @bug 6791927 8233886
27  * @summary Wrong Locale in HttpCookie::expiryDate2DeltaSeconds
28  * @run main/othervm B6791927
29  */
30 
31 import java.net.*;
32 import java.util.List;
33 import java.util.Locale;
34 
35 public class B6791927 {
36     public static final void main(String[] aaParamters) throws Exception {
37         Locale reservedLocale = Locale.getDefault();
38         try {
39             // Forces a non US locale
40             Locale.setDefault(Locale.FRANCE);
41             List<HttpCookie> cookies = HttpCookie.parse("set-cookie:" +
42                     " CUSTOMER=WILE_E_COYOTE;" +
43                     " expires=Sat, 09-Nov-2041 23:12:40 GMT");
44             if (cookies == null || cookies.isEmpty()) {
45                 throw new RuntimeException("No cookie found");
46             }
47             for (HttpCookie c : cookies) {
48                 if (c.getMaxAge() == 0) {
49                     throw new RuntimeException(
50                         "Expiration date shouldn't be 0");
51                 }
52             }
53         } finally {
54             // restore the reserved locale
55             Locale.setDefault(reservedLocale);
56         }
57     }
58 }
