24 /**
25  * @test
26  * @bug 6791927
27  * @summary Wrong Locale in HttpCookie::expiryDate2DeltaSeconds

28  */
29 
30 import java.net.*;
31 import java.util.List;
32 import java.util.Locale;
33 
34 public class B6791927 {
35     public static final void main( String[] aaParamters ) throws Exception{
36         Locale reservedLocale = Locale.getDefault();
37         try {
38             // Forces a non US locale
39             Locale.setDefault(Locale.FRANCE);
40             List<HttpCookie> cookies = HttpCookie.parse("set-cookie: CUSTOMER=WILE_E_COYOTE; expires=Sat, 09-Nov-2019 23:12:40 GMT");


41             if (cookies == null || cookies.isEmpty()) {
42                 throw new RuntimeException("No cookie found");
43             }
44             for (HttpCookie c : cookies) {
45                 if (c.getMaxAge() == 0) {
46                     throw new RuntimeException(
47                         "Expiration date shouldn't be 0");
48                 }
49             }
50         } finally {
51             // restore the reserved locale
52             Locale.setDefault(reservedLocale);
53         }
54     }
55 }
