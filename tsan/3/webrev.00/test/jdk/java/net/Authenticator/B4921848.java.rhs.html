 36 import java.io.*;
 37 import java.net.*;
 38 import jdk.test.lib.net.URIBuilder;
 39 
 40 public class B4921848 implements HttpCallback {
 41 
 42     static int count = 0;
 43 
 44     public void request (HttpTransaction req) {
 45         try {
 46             if (count == 0 ) {
 47                 req.addResponseHeader ("Connection", "close");
 48                 req.addResponseHeader ("WWW-Authenticate", "Basic realm=\"foo\"");
 49                 req.addResponseHeader ("WWW-Authenticate", "Digest realm=\"bar\" domain=/biz nonce=\"hereisanonce\"");
 50                 req.sendResponse (401, "Unauthorized");
 51                 req.orderlyClose();
 52             } else {
 53                 String authheader = req.getRequestHeader ("Authorization");
 54                 if (authheader.startsWith ("Basic")) {
 55                     req.setResponseEntityBody ("Hello .");
 56                     req.sendResponse (200, "Ok");
 57                     req.orderlyClose();
 58                 } else {
 59                     req.sendResponse (400, "Bad Request");
 60                     req.orderlyClose();
 61                 }
 62             }
 63             count ++;
 64         } catch (IOException e) {
 65             e.printStackTrace();
 66         }
 67     }
 68 
 69     static void read (InputStream is) throws IOException {
 70         int c;
 71         System.out.println ("reading");
 72         while ((c=is.read()) != -1) {
 73             System.out.write (c);
 74         }
 75         System.out.println ("");
 76         System.out.println ("finished reading");
 77     }
 78 
 79 
 80     static void client (String u) throws Exception {
 81         URL url = new URL (u);
 82         System.out.println ("client opening connection to: " + u);
 83         URLConnection urlc = url.openConnection ();
 84         InputStream is = urlc.getInputStream ();
 85         read (is);
 86         is.close();
 87     }
 88 
 89     static TestHttpServer server;
 90 
 91     public static void main (String[] args) throws Exception {
 92         MyAuthenticator auth = new MyAuthenticator ();
 93         Authenticator.setDefault (auth);
 94         ProxySelector.setDefault(ProxySelector.of(null)); // no proxy
 95         try {
 96             InetAddress loopback = InetAddress.getLoopbackAddress();
 97             server = new TestHttpServer (new B4921848(), 1, 10, loopback, 0);
 98             String serverURL = URIBuilder.newBuilder()
 99                 .scheme("http")
100                 .loopback()
101                 .port(server.getLocalPort())
102                 .path("/")
103                 .build()
104                 .toString();
105             System.out.println("Server: listening at: " + serverURL);
106             client(serverURL + "d1/d2/d3/foo.html");
107         } catch (Exception e) {
108             if (server != null) {
109                 server.terminate();
110             }
111             throw e;
112         }
113         server.terminate();
114     }
115 
116     public static void except (String s) {
117         server.terminate();
118         throw new RuntimeException (s);
119     }
120 
121     static class MyAuthenticator extends Authenticator {
122         MyAuthenticator () {
123             super ();
124         }
125 
126         public PasswordAuthentication getPasswordAuthentication () {
127             return (new PasswordAuthentication ("user", "passwordNotCheckedAnyway".toCharArray()));
128         }
129 
130     }

132 }
