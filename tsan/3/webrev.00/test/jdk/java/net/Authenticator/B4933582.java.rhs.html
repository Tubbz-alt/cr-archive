 46     static int count = 0;
 47     static String authstring;

 49     void errorReply (HttpTransaction req, String reply) throws IOException {
 50         req.addResponseHeader ("Connection", "close");
 51         req.addResponseHeader ("WWW-Authenticate", reply);
 52         req.sendResponse (401, "Unauthorized");
 53         req.orderlyClose();
 54     }
 55 
 56     void okReply (HttpTransaction req) throws IOException {
 57         req.setResponseEntityBody ("Hello .");
 58         req.sendResponse (200, "Ok");
 59         req.orderlyClose();
 60     }
 61 
 62     static volatile boolean firstTime = true;
 63 
 64     public void request (HttpTransaction req) {
 65         try {
 66             authstring = req.getRequestHeader ("Authorization");
 67             if (firstTime) {
 68                 switch (count) {
 69                 case 0:
 70                     errorReply (req, "Basic realm=\"wallyworld\"");
 71                     break;
 72                 case 1:
 73                     /* client stores a username/pw for wallyworld
 74                      */
 75                     save (authstring);
 76                     okReply (req);
 77                     break;
 78                 }
 79             } else {
 80                 /* check the auth string is premptively set from last time */
 81                 String savedauth = retrieve();
 82                 if (savedauth.equals (authstring)) {
 83                     okReply (req);
 84                 } else {
 85                     System.out.println ("savedauth = " + savedauth);
 86                     System.out.println ("authstring = " + authstring);
 87                     errorReply (req, "Basic realm=\"wallyworld\"");
 88                 }
 89             }
 90             count ++;
 91         } catch (IOException e) {
 92             e.printStackTrace();
 93         }
 94     }
 95 
 96     void save (String s) {
 97         try {
 98             FileOutputStream f = new FileOutputStream ("auth.save");
 99             ObjectOutputStream os = new ObjectOutputStream (f);
100             os.writeObject (s);
101         } catch (IOException e) {
102             assert false;
103         }
104     }
105 
106     String retrieve () {
107         String s = null;
108         try {
109             FileInputStream f = new FileInputStream ("auth.save");
110             ObjectInputStream is = new ObjectInputStream (f);
111             s = (String) is.readObject();
112         } catch (Exception e) {
113             assert false;
114         }
115         return s;
116     }

118     static void read (InputStream is) throws IOException {
119         int c;
120         System.out.println ("reading");
121         while ((c=is.read()) != -1) {
122             System.out.write (c);
123         }
124         System.out.println ("");
125         System.out.println ("finished reading");
126     }

128     static void client (String u) throws Exception {
129         URL url = new URL (u);
130         System.out.println ("client opening connection to: " + u);
131         URLConnection urlc = url.openConnection ();
132         try(InputStream is = urlc.getInputStream ()) {
133             read (is);
134         }
135     }

235     static class CacheImpl extends AuthCacheImpl {
236         HashMap<String,LinkedList<AuthCacheValue>> map;
237         int port; // need to store the port number the server is using
238 
239         CacheImpl () throws IOException {
240             this (-1);
241         }
242 
243         CacheImpl (int port) throws IOException {
244             super();
245             this.port = port;
246             File src = new File ("cache.ser");
247             if (src.exists()) {
248                 try (ObjectInputStream is = new ObjectInputStream(
249                         new FileInputStream(src))) {
250                     map = (HashMap<String,LinkedList<AuthCacheValue>>)is
251                               .readObject();
252                     this.port = (Integer)is.readObject ();
253                     System.out.println ("read port from file " + port);
254                 } catch (ClassNotFoundException e) {
255                     assert false;
256                 }
257                 System.out.println ("setMap from cache.ser");
258             } else {
259                 map = new HashMap<>();
260             }
261             setMap (map);
262         }
263 
264         int getPort () {
265             return port;
266         }
267 
268         private void writeMap () {
269             File dst = new File("cache.ser");
270             try {
271                 dst.delete();
272                 if (!dst.createNewFile()) {
273                     return;
274                 }
275             } catch (IOException e) {
276             }
277 
278             try (ObjectOutputStream os = new ObjectOutputStream(
279                     new FileOutputStream(dst))) {
280                 os.writeObject(map);
281                 os.writeObject(port);
282                 System.out.println("wrote port " + port);
283             } catch (IOException e) {
284             }
285         }

301             System.out.println ("remove: " + pkey + " " + value);
302             super.remove (pkey, value);

304         }
