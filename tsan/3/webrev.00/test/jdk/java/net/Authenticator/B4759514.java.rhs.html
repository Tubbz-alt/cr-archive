 24 /**
 25  * @test
 26  * @bug 4759514
 27  * @modules java.base/sun.net.www
 28  * @library ../../../sun/net/www/httptest/ /test/lib
 29  * @build HttpCallback TestHttpServer ClosedChannelList HttpTransaction
 30  * @run main/othervm B4759514
 31  * @run main/othervm -Djava.net.preferIPv6Addresses=true B4759514
 32  * @summary Digest Authentication is erroniously quoting the nc value, contrary to RFC 2617
 33  */
 34 
 35 import java.io.*;
 36 import java.net.*;
 37 import jdk.test.lib.net.URIBuilder;
 38 
 39 public class B4759514 implements HttpCallback {
 40 
 41     static int count = 0;
 42     static String authstring;
 43 
 44     void errorReply (HttpTransaction req, String reply) throws IOException {
 45         req.addResponseHeader ("Connection", "close");
 46         req.addResponseHeader ("WWW-Authenticate", reply);
 47         req.sendResponse (401, "Unauthorized");
 48         req.orderlyClose();
 49     }
 50 
 51     void okReply (HttpTransaction req) throws IOException {
 52         req.setResponseEntityBody ("Hello .");
 53         req.sendResponse (200, "Ok");
 54         req.orderlyClose();
 55     }
 56 
 57     public void request (HttpTransaction req) {
 58         try {
 59             authstring = req.getRequestHeader ("Authorization");
 60             switch (count) {
 61             case 0:
 62                 errorReply (req, "Digest realm=\"wallyworld\", nonce=\"1234\", domain=\"/\"");
 63                 break;
 64             case 1:
 65                 int n = authstring.indexOf ("nc=");
 66                 if (n != -1) {
 67                     if (authstring.charAt (n+3) == '\"') {
 68                         req.sendResponse (400, "Bad Request");
 69                         break;
 70                     }
 71                 }
 72                 okReply (req);
 73                 break;
 74             }
 75             count ++;
 76         } catch (IOException e) {
 77             e.printStackTrace();
 78         }
 79     }
 80 
 81     static void read (InputStream is) throws IOException {
 82         int c;
 83         while ((c=is.read()) != -1) {
 84             System.out.write (c);
 85         }
 86     }
 87 
 88     static void client (String u) throws Exception {
 89         URL url = new URL (u);
 90         System.out.println ("client opening connection to: " + u);
 91         URLConnection urlc = url.openConnection ();
 92         InputStream is = urlc.getInputStream ();
 93         read (is);
 94         is.close();
 95     }
 96 
 97     static TestHttpServer server;
 98 
 99     public static void main (String[] args) throws Exception {
100         MyAuthenticator auth = new MyAuthenticator ();
101         Authenticator.setDefault (auth);
102         ProxySelector.setDefault(ProxySelector.of(null)); // no proxy
103         try {
104             InetAddress loopback = InetAddress.getLoopbackAddress();
105             server = new TestHttpServer (new B4759514(), 1, 10, loopback, 0);
106             String serverURL = URIBuilder.newBuilder()
107                 .scheme("http")
108                 .loopback()
109                 .port(server.getLocalPort())
110                 .path("/")
111                 .build()
112                 .toString();
113             System.out.println("Server: listening at: " + serverURL);
114             client(serverURL + "d1/foo.html");
115         } catch (Exception e) {
116             if (server != null) {
117                 server.terminate();
118             }
119             throw e;
120         }
121         int f = auth.getCount();
122         if (f != 1) {
123             except ("Authenticator was called "+f+" times. Should be 1");
124         }
125         server.terminate();
126     }
127 
128     public static void except (String s) {
129         server.terminate();
130         throw new RuntimeException (s);
131     }
132 
133     static class MyAuthenticator extends Authenticator {
134         MyAuthenticator () {
135             super ();
136         }
137 
138         int count = 0;
139 
140         public PasswordAuthentication getPasswordAuthentication () {
141             PasswordAuthentication pw;
142             pw = new PasswordAuthentication ("user", "pass1".toCharArray());
143             count ++;
144             return pw;
145         }
146 
147         public int getCount () {
148             return (count);
149         }
150     }
151 }
