 24 /**
 25  * @test
 26  * @bug 6648001
 27  * @modules jdk.httpserver
 28  * @run main/othervm/timeout=20 -ea:sun.net.www.protocol.http.AuthenticationInfo -Dhttp.auth.serializeRequests=true Deadlock
 29  * @summary  cancelling HTTP authentication causes deadlock
 30  */
 31 
 32 import java.util.concurrent.Executors;
 33 import java.util.concurrent.ExecutorService;
 34 import java.io.InputStream;
 35 import java.io.IOException;
 36 import java.net.HttpURLConnection;
 37 import java.net.InetSocketAddress;
 38 import java.net.PasswordAuthentication;
 39 import java.net.URL;
 40 import com.sun.net.httpserver.BasicAuthenticator;
 41 import com.sun.net.httpserver.Headers;
 42 import com.sun.net.httpserver.HttpContext;
 43 import com.sun.net.httpserver.HttpExchange;
 44 import com.sun.net.httpserver.HttpHandler;
 45 import com.sun.net.httpserver.HttpPrincipal;
 46 import com.sun.net.httpserver.HttpServer;
 47 
 48 public class Deadlock {
 49 
 50     public static void main (String[] args) throws Exception {
 51         Handler handler = new Handler();
 52         InetSocketAddress addr = new InetSocketAddress (0);
 53         HttpServer server = HttpServer.create(addr, 0);
 54         HttpContext ctx = server.createContext("/test", handler);
 55         BasicAuthenticator a = new BasicAuthenticator("foobar@test.realm") {
 56             @Override
 57             public boolean checkCredentials (String username, String pw) {
 58                 return "fred".equals(username) && pw.charAt(0) == 'x';
 59             }
 60         };
 61 
 62         ctx.setAuthenticator(a);
 63         ExecutorService executor = Executors.newCachedThreadPool();
 64         server.setExecutor(executor);
 65         server.start ();
 66         java.net.Authenticator.setDefault(new MyAuthenticator());
 67 
 68         System.out.print("Deadlock: " );
 69         for (int i=0; i<2; i++) {
 70             Runner t = new Runner(server, i);
 71             t.start();
 72             t.join();
 73         }
 74         server.stop(2);
 75         executor.shutdown();
 76         if (error) {
 77             throw new RuntimeException("test failed error");
 78         }
 79 
 80         if (count != 2) {
 81             throw new RuntimeException("test failed count = " + count);
 82         }
 83         System.out.println("OK");
 84 
 85     }
 86 
 87     static class Runner extends Thread {
 88         HttpServer server;
 89         int i;
 90         Runner(HttpServer s, int i) {
 91             server = s;
 92             this.i = i;
 93         }
 94 
 95         @Override
 96         public void run() {
 97             URL url;
 98             HttpURLConnection urlc;
 99             try {
100                 url = new URL("http://localhost:"+server.getAddress().getPort()+"/test/foo.html");
101                 urlc = (HttpURLConnection)url.openConnection ();
102             } catch (IOException e) {
103                 error = true;
104                 return;
105             }
106             InputStream is = null;
107             try {
108                 is = urlc.getInputStream();
109                 while (is.read()!= -1) {}
110             } catch (IOException e) {
111                 if (i == 1) error = true;
112             } finally {
113                 if (is != null) try { is.close(); } catch (IOException e) {}
114             }
115         }
116     }
117 
118     public static boolean error = false;
119     public static int count = 0;
120 
121     static class MyAuthenticator extends java.net.Authenticator {
122         @Override
123         public PasswordAuthentication getPasswordAuthentication() {
124             PasswordAuthentication pw;
125             if (!getRequestingPrompt().equals("foobar@test.realm")) {
126                 Deadlock.error = true;
127             }
128             if (count == 0) {
129                 pw = null;
130             } else {
131                 pw = new PasswordAuthentication("fred", "xyz".toCharArray());
132             }
133             count++;
134             return pw;
135         }
136     }
137 
138     static class Handler implements HttpHandler {
139         int invocation = 1;
140 
141         @Override
142         public void handle (HttpExchange t)
143             throws IOException
144         {
145             InputStream is = t.getRequestBody();
146             Headers map = t.getRequestHeaders();
147             Headers rmap = t.getResponseHeaders();
148             while (is.read() != -1);
149             is.close();
150             t.sendResponseHeaders(200, -1);
151             HttpPrincipal p = t.getPrincipal();
152             if (!p.getUsername().equals("fred")) {
153                 error = true;
154             }
155             if (!p.getRealm().equals("foobar@test.realm")) {
156                 error = true;
157             }
158             t.close();
159         }
160     }
