27  * @summary java.net.PlainSocketImpl {set,get}Option throws SocketException
28  *          when socket is closed.
29  */
30 
31 import java.io.*;
32 import java.net.*;
33 
34 public class SetOption {
35 
36     public static void main(String args[]) throws Exception {
37 
38         ServerSocket ss = new ServerSocket(0);
39         Socket s1 = new Socket("localhost", ss.getLocalPort());


40         Socket s2 = ss.accept();
41 
42         s1.close();
43         boolean exc_thrown = false;
44         try {
45             s1.setSoTimeout(1000);
46         } catch (SocketException e) {
47             exc_thrown = true;
48         }
49 
50         if (!exc_thrown) {
51             throw new Exception("SocketException not thrown on closed socket");
52         }
53     }
54 }
