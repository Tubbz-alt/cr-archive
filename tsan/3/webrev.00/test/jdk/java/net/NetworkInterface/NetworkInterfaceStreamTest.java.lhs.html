 24 /* @test
 25  * @bug 8081678 8131155
 26  * @summary Tests for stream returning methods
 27  * @library /lib/testlibrary/bootlib
 28  * @build java.base/java.util.stream.OpTestCase

 30  * @run testng/othervm -Djava.net.preferIPv4Stack=true NetworkInterfaceStreamTest

 33 import org.testng.annotations.Test;
 34 
 35 import java.net.InetAddress;
 36 import java.net.NetworkInterface;
 37 import java.net.SocketException;
 38 import java.util.ArrayList;
 39 import java.util.Collection;
 40 import java.util.Collections;
 41 import java.util.function.Supplier;
 42 import java.util.stream.OpTestCase;
 43 import java.util.stream.Stream;
 44 import java.util.stream.TestData;
 45 


 46 public class NetworkInterfaceStreamTest extends OpTestCase {
 47 
 48     private final static boolean IS_WINDOWS = System.getProperty("os.name").startsWith("Windows");
 49 





 50     @Test
 51     public void testNetworkInterfaces() throws SocketException {
 52         Supplier<Stream<NetworkInterface>> ss = () -> {
 53             try {
 54                 return NetworkInterface.networkInterfaces()
 55                         .filter(ni -> isIncluded(ni));
 56             }
 57             catch (SocketException e) {
 58                 throw new RuntimeException(e);
 59             }
 60         };
 61 
 62         Collection<NetworkInterface> enums = Collections.list(NetworkInterface.getNetworkInterfaces());
 63         Collection<NetworkInterface> expected = new ArrayList<>();
 64         enums.forEach(ni -> {
 65             if (isIncluded(ni)) {
 66                 expected.add(ni);
 67             }
 68         });
 69         withData(TestData.Factory.ofSupplier("Top-level network interfaces", ss))
 70                 .stream(s -> s)
 71                 .expectedResult(expected)
 72                 .exercise();
 73     }
 74 
 75     private Collection<NetworkInterface> getAllNetworkInterfaces() throws SocketException {
 76         Collection<NetworkInterface> anis = new ArrayList<>();
 77         for (NetworkInterface ni : Collections.list(NetworkInterface.getNetworkInterfaces())) {
 78             getAllSubNetworkInterfaces(ni, anis);
 79         }
 80         return anis;
 81     }
 82 
 83     private void getAllSubNetworkInterfaces(NetworkInterface ni, Collection<NetworkInterface> result) {
 84         if (isIncluded(ni)) {
 85             result.add(ni);
 86         }
 87 
 88         for (NetworkInterface sni : Collections.list(ni.getSubInterfaces())) {
 89             getAllSubNetworkInterfaces(sni, result);
 90         }
 91     }
 92 
 93     private Stream<NetworkInterface> allNetworkInterfaces() throws SocketException {
 94         return NetworkInterface.networkInterfaces()
 95                 .filter(ni -> isIncluded(ni))
 96                 .flatMap(this::allSubNetworkInterfaces);
 97     }
 98 
 99     private Stream<NetworkInterface> allSubNetworkInterfaces(NetworkInterface ni) {
100         return Stream.concat(
101                 Stream.of(ni),
102                 ni.subInterfaces().filter(sni -> isIncluded(sni)).flatMap(this::allSubNetworkInterfaces));
103     }
104 
105     @Test
106     public void testSubNetworkInterfaces() throws SocketException {
107         Supplier<Stream<NetworkInterface>> ss = () -> {
108             try {
109                 return allNetworkInterfaces();
110             }
111             catch (SocketException e) {
112                 throw new RuntimeException(e);
113             }
114         };
115 
116         Collection<NetworkInterface> expected = getAllNetworkInterfaces();
117         withData(TestData.Factory.ofSupplier("All network interfaces", ss))
118                 .stream(s -> s)
119                 .expectedResult(expected)
120                 .exercise();
121     }
122 
123 
124     @Test
125     public void testInetAddresses() throws SocketException {
126         Supplier<Stream<InetAddress>> ss = () -> {
127             try {
128                 return NetworkInterface.networkInterfaces()
129                         .filter(ni -> isIncluded(ni))
130                         .flatMap(NetworkInterface::inetAddresses);
131             }
132             catch (SocketException e) {
133                 throw new RuntimeException(e);
134             }
135         };
136 
137         Collection<NetworkInterface> nis = Collections.list(NetworkInterface.getNetworkInterfaces());
138         Collection<InetAddress> expected = new ArrayList<>();
139         for (NetworkInterface ni : nis) {
140             if (isIncluded(ni)) {
141                 expected.addAll(Collections.list(ni.getInetAddresses()));
142             }
143         }
144         withData(TestData.Factory.ofSupplier("All inet addresses", ss))
145                 .stream(s -> s)
146                 .expectedResult(expected)
147                 .exercise();
148     }
149 
150     /**
151      * Check if the input network interface should be included in the test. It is necessary to exclude
152      * "Teredo Tunneling Pseudo-Interface" whose configuration can be variable during a test run.
153      *
154      * @param ni a network interace
155      * @return false if it is a "Teredo Tunneling Pseudo-Interface", otherwise true.
156      */
157     private boolean isIncluded(NetworkInterface ni) {
158         if (!IS_WINDOWS) {
159             return true;
160         }
161 
162         String dName = ni.getDisplayName();
163         return dName == null || !dName.contains("Teredo");
164     }
165 
166 }
