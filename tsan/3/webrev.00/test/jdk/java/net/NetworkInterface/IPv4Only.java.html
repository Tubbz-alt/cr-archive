24 /* @test
25  * @bug   6964714
26  * @library /test/lib
27  * @run main/othervm -Djava.net.preferIPv4Stack=true IPv4Only
28  * @summary Test the networkinterface listing with java.net.preferIPv4Stack=true.
29  */
30 
31 
32 import java.net.*;
33 import java.util.*;
34 import jdk.test.lib.net.IPSupport;
35 
36 public class IPv4Only {
37     public static void main(String[] args) throws Exception {
38         if (IPSupport.hasIPv4()) {
39             System.out.println("Testing IPv4");
40             Enumeration<NetworkInterface> nifs = NetworkInterface.getNetworkInterfaces();
41             while (nifs.hasMoreElements()) {
42                 NetworkInterface nif = nifs.nextElement();
43                 Enumeration<InetAddress> addrs = nif.getInetAddresses();
44                 while (addrs.hasMoreElements()) {
45                    InetAddress hostAddr = addrs.nextElement();
46                    if ( hostAddr instanceof Inet6Address ){
47                         throw new RuntimeException( "NetworkInterfaceV6List failed - found v6 address " + hostAddr.getHostAddress() );
48                    }
49                 }
50             }
51         } else {
52             try {
53                 NetworkInterface.getNetworkInterfaces();
54                 throw new RuntimeException("NetworkInterface.getNetworkInterfaces() should throw SocketException");
55             } catch (SocketException expected) {
56                 System.out.println("caught expected exception: " + expected);
57             }
58 
59             try {
60                 NetworkInterface.networkInterfaces();
61                 throw new RuntimeException("NetworkInterface.networkInterfaces() should throw SocketException");
62             } catch (SocketException expected) {
63                 System.out.println("caught expected exception: " + expected);
64             }
65         }
66     }
67 }
