 24 /* @test
 25  *
 26  * @bug 4091803
 27  *
 28  * @summary this tests that the constructor of DatagramPacket rejects
 29  * bogus arguments properly.
 30  *
 31  * @author Benjamin Renaud
 32  */
 33 import java.io.*;
 34 import java.net.*;
 35 import java.util.*;






 36 
 37 public class Constructor {
 38 
 39     public static void main(String[] args) throws Exception {
 40         testNullPacket();
 41         testNegativeBufferLength();
 42         testPacketLengthTooLarge();
 43         testNegativePortValue();
 44         testPortValueTooLarge();
 45         testSimpleConstructor();
 46         testFullConstructor();
 47         System.err.println("all passed!");

 48     }








 49 
 50     static void testNullPacket() throws Exception {
 51         boolean error = true;
 52         try {
 53             new DatagramPacket(null, 100);
 54         } catch (NullPointerException e) {
 55             /* correct exception */
 56             error = false;
 57         }
 58         if (error) {
 59             throw new RuntimeException("test 1 failed.");
 60         }
 61     }
 62 
 63     static void testNegativeBufferLength() throws Exception {
 64         boolean error = true;
 65         byte[] buf = new byte[128];
 66         try {
 67             /* length lesser than buffer length */
 68             new DatagramPacket(buf, -128);
 69         } catch (IllegalArgumentException e) {
 70             /* correct exception */
 71             error = false;
 72         }
 73         if (error) {
 74             throw new RuntimeException("test 2 failed.");
 75         }
 76     }
 77 
 78     static void testPacketLengthTooLarge() throws Exception {
 79         boolean error = true;
 80         byte[] buf = new byte[128];
 81         try {
 82             /* length greater than buffer length */
 83             new DatagramPacket(buf, 256);
 84         } catch (IllegalArgumentException e) {
 85             /* correct exception */
 86             error = false;
 87         }
 88         if (error) {
 89             throw new RuntimeException("test 3 failed.");
 90         }
 91     }
 92 
 93     static void testNegativePortValue() throws Exception {
 94         boolean error = true;
 95         byte[] buf = new byte[128];
 96         InetAddress host = InetAddress.getLocalHost();
 97         try {
 98             /* negative port */
 99             new DatagramPacket(buf, 100, host, -1);
100         } catch (IllegalArgumentException e) {
101             /* correct exception */
102             error = false;
103         }
104         if (error) {
105             throw new RuntimeException("test 5 failed.");
106         }
107     }
108 
109     static void testPortValueTooLarge() throws Exception {
110         boolean error = true;
111         byte[] buf = new byte[256];
112         InetAddress address = InetAddress.getLocalHost();
113         try {
114             /* invalid port value */
115             new DatagramPacket(buf, 256, address, Integer.MAX_VALUE);
116         } catch (IllegalArgumentException e) {
117             /* correct exception */
118             error = false;
119         }
120         if (error) {
121             throw new RuntimeException("test 6 failed.");
122         }
123     }
124 
125     static void testSimpleConstructor() {
126         byte[] buf = new byte[128];
127         int offset = 10;
128         int length = 50;
129         DatagramPacket packet = new DatagramPacket(buf, offset, length);
130         if (packet.getData() != buf || packet.getOffset() != offset ||
131                packet.getLength() != length) {
132             throw new RuntimeException("simple constructor failed");
133         }
134     }
135 
136     static void testFullConstructor() throws Exception {
137         byte[] buf = new byte[128];
138         int offset = 10;
139         int length = 50;
140         InetAddress address = InetAddress.getLocalHost();
141         int port = 8080;
142         DatagramPacket packet = new DatagramPacket(buf, offset, length,
143                                                    address, port);
144         if (packet.getData() != buf || packet.getOffset() != offset ||
145             packet.getLength() != length ||
146             packet.getAddress() != address ||
147             packet.getPort() != port) {
148             throw new RuntimeException("full constructor failed");
149         }
150     }
151 }
