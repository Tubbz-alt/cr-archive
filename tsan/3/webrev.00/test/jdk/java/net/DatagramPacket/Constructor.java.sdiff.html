 50     static void testNullPacket() throws Exception {
 51         boolean error = true;
 52         try {
 53             new DatagramPacket(null, 100);

 55             /* correct exception */
 56             error = false;

 60         }
 61     }

 63     static void testNegativeBufferLength() throws Exception {
 64         boolean error = true;

 67             /* length lesser than buffer length */

 24 /* @test
 25  * @bug 4091803 7021373


 26  * @summary this tests that the constructor of DatagramPacket rejects
 27  *          bogus arguments properly.
 28  * @run testng Constructor

 29  */
 30 
 31 import java.net.DatagramPacket;
 32 import java.net.InetAddress;
 33 import java.net.InetSocketAddress;
 34 
 35 import org.testng.annotations.Test;
 36 
 37 import static org.testng.Assert.assertFalse;
 38 import static org.testng.Assert.expectThrows;
 39 
 40 public class Constructor {
 41 
 42     private static final byte[] buf = new byte[128];
 43 
 44     private static final InetAddress LOOPBACK = InetAddress.getLoopbackAddress();
 45     private static final Class<NullPointerException> NPE = NullPointerException.class;
 46     private static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
 47 
 48     @Test
 49     public void testNullPacket() {
 50         expectThrows(NPE,
 51                 () -> new DatagramPacket(null, 100));
 52     }
 53     @Test
 54     public void testNull() throws Exception {
 55         expectThrows(NPE, () -> new DatagramPacket(null, 100));
 56         expectThrows(NPE, () -> new DatagramPacket(null, 0, 10));
 57         expectThrows(NPE, () -> new DatagramPacket(null, 0, 10, LOOPBACK, 80));
 58         expectThrows(NPE, () -> new DatagramPacket(null, 10, LOOPBACK, 80));
 59         expectThrows(NPE, () -> new DatagramPacket(null, 0, 10, new InetSocketAddress(80)));
 60         expectThrows(NPE, () -> new DatagramPacket(null, 10, new InetSocketAddress(80)));
 61 
 62         // no Exception expected for null addresses
 63         new DatagramPacket(buf, 10, null, 0);
 64         new DatagramPacket(buf, 10, 10, null, 0);








 65     }
 66 
 67     @Test
 68     public void testNegativeBufferLength() {
 69         /* length lesser than buffer length */
 70         expectThrows(IAE,
 71                 () -> new DatagramPacket(buf, -128));








 72     }
 73 
 74     @Test
 75     public void testPacketLengthTooLarge() {
 76         /* length greater than buffer length */
 77         expectThrows(IAE,
 78                 () -> new DatagramPacket(buf, 256));








 79     }
 80 
 81     @Test
 82     public void testNegativePortValue() throws Exception {
 83         /* negative port */
 84         InetAddress addr = InetAddress.getLocalHost();
 85 
 86         expectThrows(IAE,
 87                 () -> new DatagramPacket(buf, 100, addr, -1));







 88     }
 89 
 90     @Test
 91     public void testPortValueTooLarge() {
 92         /* invalid port value */
 93         expectThrows(IAE,
 94                 () -> new DatagramPacket(buf, 128, LOOPBACK, Integer.MAX_VALUE));









 95     }
 96 
 97     @Test
 98     public void testSimpleConstructor() {
 99         int offset = 10;
100         int length = 50;
101         DatagramPacket pkt = new DatagramPacket(buf, offset, length);
102 
103         assertFalse((pkt.getData() != buf || pkt.getOffset() != offset ||
104                 pkt.getLength() != length), "simple constructor failed");

105     }
106 
107     @Test
108     public void testFullConstructor() {
109         int offset = 10;
110         int length = 50;

111         int port = 8080;
112         DatagramPacket packet = new DatagramPacket(buf, offset, length, LOOPBACK, port);
113 
114         assertFalse((packet.getData() != buf || packet.getOffset() != offset ||
115                 packet.getLength() != length ||
116                 packet.getAddress() != LOOPBACK ||
117                 packet.getPort() != port), "full constructor failed");


118     }
