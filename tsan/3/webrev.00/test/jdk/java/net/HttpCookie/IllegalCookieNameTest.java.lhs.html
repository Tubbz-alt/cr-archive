28 import java.net.*;
29 import java.util.*;
30 import java.io.*;
31 import com.sun.net.httpserver.*;

32 
33 public class IllegalCookieNameTest {
34     public static void main(String[] args) throws IOException {
35         HttpServer s = null;
36         try {
37             InetSocketAddress addr = new InetSocketAddress(0);

38             s = HttpServer.create(addr, 10);
39             s.createContext("/", new HHandler());
40             s.start();
41             String u = "http://127.0.0.1:" + s.getAddress().getPort() + "/";





42             CookieHandler.setDefault(new TestCookieHandler());
43             URL url = new URL(u);
44             HttpURLConnection c = (HttpURLConnection) url.openConnection();
45             c.getHeaderFields();
46             System.out.println ("OK");
47         } finally {
48             s.stop(1);
49         }
50     }
51 }
52 
53 class TestCookieHandler extends CookieHandler {
54     @Override
55     public Map<String, List<String>> get(URI uri, Map<String, List<String>> requestHeaders) {
56         return new HashMap<String, List<String>>();
57     }
58 
59     @Override
60     public void put(URI uri, Map<String, List<String>> responseHeaders) {
61     }
62 }
63 
64 class HHandler implements  HttpHandler {
65     public void handle (HttpExchange e) {
66         try {
67             Headers h = e.getResponseHeaders();
68             h.set ("Set-Cookie", "domain=; expires=Mon, 01-Jan-1990 00:00:00 GMT; path=/; domain=.foo.com");
69             e.sendResponseHeaders(200, -1);
70             e.close();
71         } catch (Exception ex) {
72             System.out.println (ex);
73         }
74     }
75 }
