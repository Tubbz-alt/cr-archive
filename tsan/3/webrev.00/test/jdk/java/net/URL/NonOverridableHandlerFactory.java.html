24 /*
25  * @test
26  * @bug 8217364
27  * @summary Custom URLStreamHandler for jrt or file protocol can override default handler
28  * @modules java.base/java.net:open
29  * @run main/othervm NonOverridableHandlerFactory
30  */
31 
32 import java.io.*;
33 import java.net.*;
34 
35 public class NonOverridableHandlerFactory implements URLStreamHandlerFactory {
36     @Override
37     public URLStreamHandler createURLStreamHandler(String protocol) {
38         return new NonOverridableHandler();
39     }
40 
41     private static class NonOverridableHandler extends URLStreamHandler {
42         @Override
43         protected URLConnection openConnection(URL u) throws IOException {
44             throw new RuntimeException("CustomURLStreamHandler used for " + u);
45         }
46     }
47 
48     public static void main(String[] args) {
49         URL.setURLStreamHandlerFactory(new NonOverridableHandlerFactory());
50         for (String protocol : new String[] {"file", "jrt"}) {
51             String urlString = protocol + ":///test/somefile";
52             try {
53                 new URL(urlString).openConnection();
54             } catch(RuntimeException r) {
55                 throw r;
56             } catch(Throwable t) {
57                 // possible jrt initialization issue on exploded binary image etc.
58                 System.out.println("encountered: " + t +
59                     "(testing: " + urlString + ")");
60             }
61             System.out.println("Default URLStreamHandler used for " + urlString);
62         }
63     }
64 }
