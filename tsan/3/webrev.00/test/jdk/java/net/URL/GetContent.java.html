 24 /**
 25  * @test
 26  * @bug 4145315
 27  * @library /test/lib
 28  * @summary Test a read from nonexistant URL
 29  * @run main GetContent
 30  * @run main/othervm -Djava.net.preferIPv6Addresses GetContent
 31  */
 32 
 33 import java.net.*;
 34 import java.io.*;
 35 import jdk.test.lib.net.URIBuilder;
 36 
 37 public class GetContent implements Runnable {
 38 
 39      ServerSocket ss;
 40 
 41      public void run() {
 42         try (Socket s = ss.accept()) {
 43             s.setTcpNoDelay(true);
 44 
 45             PrintStream out = new PrintStream(
 46                                  new BufferedOutputStream(
 47                                     s.getOutputStream() ));
 48 
 49             out.print("HTTP/1.1 404 Not Found\r\n");
 50             out.print("Connection: close\r\n");
 51             out.print("Content-Type: text/html; charset=iso-8859-1\r\n");
 52             out.print("\r\n");
 53             out.flush();
 54             out.print("<HTML><BODY>Sorry, page not found</BODY></HTML>");
 55             out.flush();
 56 
 57             // wait for client to read response - otherwise http
 58             // client get error and re-establish connection
 59             Thread.sleep(2000);
 60 
 61         } catch (Exception e) {
 62             e.printStackTrace();
 63         } finally {
 64             try { ss.close(); } catch (IOException unused) {}
 65         }
 66      }
 67 
 68      GetContent() throws Exception {
 69          InetAddress loopback = InetAddress.getLoopbackAddress();
 70          ss = new ServerSocket();
 71          ss.bind(new InetSocketAddress(loopback, 0));
 72 
 73          Thread thr = new Thread(this);
 74          thr.start();
 75 
 76          boolean error = true;
 77          try {
 78              java.net.URL url = URIBuilder.newBuilder()
 79                  .scheme("http")
 80                  .host(ss.getInetAddress())
 81                  .port(ss.getLocalPort())
 82                  .path("/no-such-name")
 83                  .toURL();
 84              Object obj = url.openConnection(Proxy.NO_PROXY)
 85                           .getContent();
 86              InputStream in = (InputStream) obj;
 87              byte buff[] = new byte[200];
 88              int len = in.read(buff);
 89          } catch (IOException ex) {
 90              error = false;
 91          }
 92 
 93          if (error)
 94              throw new RuntimeException("No IOException generated.");
 95      }
 96 
 97      public static void main(String args[]) throws Exception {
 98         new GetContent();
 99      }
100 }
