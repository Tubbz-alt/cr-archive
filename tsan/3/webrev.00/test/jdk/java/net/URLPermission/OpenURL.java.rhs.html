31 import java.net.*;
32 import java.io.*;
33 import jdk.test.lib.net.URIBuilder;
34 import static java.net.Proxy.NO_PROXY;
35 
36 public class OpenURL {
37 
38     public static void main (String[] args) throws Exception {
39 
40         System.setSecurityManager(new SecurityManager());
41 
42         try {
43             URL url = URIBuilder.newBuilder()
44                 .scheme("http")
45                 .userInfo("joe")
46                 .loopback()
47                 .path("/a/b")
48                 .toURL();
49             System.out.println("URL: " + url);
50             HttpURLConnection urlc = (HttpURLConnection)url.openConnection(NO_PROXY);
51             InputStream is = urlc.getInputStream();
52             // error will throw exception other than SecurityException
53         } catch (SecurityException e) {
54             System.out.println("OK");
55         }
56     }
57 }
