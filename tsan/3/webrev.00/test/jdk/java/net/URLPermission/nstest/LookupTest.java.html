 24 /**
 25  * @test
 26  * @summary A simple smoke test of the HttpURLPermission mechanism, which checks
 27  *          for either IOException (due to unknown host) or SecurityException
 28  *          due to lack of permission to connect
 29  * @run main/othervm LookupTest
 30  */
 31 
 32 import java.io.BufferedWriter;
 33 import java.io.FilePermission;
 34 import java.io.FileWriter;
 35 import java.io.IOException;
 36 import java.io.InputStream;

 48 import java.net.URLPermission;

 57 public class LookupTest {
 58 
 59     static final Policy DEFAULT_POLICY = Policy.getPolicy();
 60     static int port;
 61     static volatile ServerSocket serverSocket;
 62 
 63     static void test(String url,
 64                      boolean throwsSecException,
 65                      boolean throwsIOException) {
 66         ProxySelector.setDefault(null);
 67         URL u;
 68         InputStream is = null;
 69         try {
 70             u = new URL(url);
 71             System.err.println("Connecting to " + u);
 72             URLConnection urlc = u.openConnection();
 73             is = urlc.getInputStream();
 74         } catch (SecurityException e) {
 75             if (!throwsSecException) {
 76                 throw new RuntimeException("Unexpected SecurityException:", e);
 77             }
 78             return;
 79         } catch (IOException e) {
 80             if (!throwsIOException) {
 81                 System.err.println("Unexpected IOException:" + e.getMessage());
 82                 throw new RuntimeException(e);
 83             }
 84             return;
 85         } finally {
 86             if (is != null) {
 87                 try {
 88                     is.close();
 89                 } catch (IOException e) {
 90                     System.err.println("Unexpected IOException:" + e.getMessage());
 91                     throw new RuntimeException(e);
 92                 }
 93             }
 94         }
 95 
 96         if (throwsSecException || throwsIOException) {
 97             System.err.printf("was expecting a %s\n", throwsSecException
 98                     ? "security exception" : "IOException");
 99             throw new RuntimeException("was expecting an exception");
100         }
101     }
102 
103     static final String CWD = System.getProperty("user.dir", ".");
104 
105     public static void main(String args[]) throws Exception {
106         String hostsFileName = CWD + "/LookupTestHosts";
107         System.setProperty("jdk.net.hosts.file", hostsFileName);
108         addMappingToHostsFile("allowedAndFound.com",
109                               InetAddress.getLoopbackAddress().getHostAddress(),
110                               hostsFileName,
111                               false);
112         addMappingToHostsFile("notAllowedButFound.com",
113                               "99.99.99.99",
114                               hostsFileName,
115                               true);
116         // name "notAllowedAndNotFound.com" is not in map
117         // name "allowedButNotfound.com" is not in map
118         Server server = new Server();
119         try {
120             Policy.setPolicy(new LookupTestPolicy());
121             System.setSecurityManager(new SecurityManager());
122             server.start();
123             test("http://allowedAndFound.com:"       + port + "/foo", false, false);
124             test("http://notAllowedButFound.com:"    + port + "/foo", true, false);
125             test("http://allowedButNotfound.com:"    + port + "/foo", false, true);
126             test("http://notAllowedAndNotFound.com:" + port + "/foo", true, false);
127         } finally {
128             server.terminate();
129         }
130     }
131 
132     static class Server extends Thread {
133         private volatile boolean done;
134 
135         public Server() throws IOException {
136             InetAddress loopback = InetAddress.getLoopbackAddress();
137             serverSocket = new ServerSocket();
138             serverSocket.bind(new InetSocketAddress(loopback, 0));
139             port = serverSocket.getLocalPort();
140         }
141 
142         public void run() {
143             try {
144                 while (!done) {
145                     try (Socket s = serverSocket.accept()) {
146                         readOneRequest(s.getInputStream());
147                         OutputStream o = s.getOutputStream();
148                         String rsp = "HTTP/1.1 200 Ok\r\n" +
149                                      "Connection: close\r\n" +
150                                      "Content-length: 0\r\n\r\n";
151                         o.write(rsp.getBytes(US_ASCII));
152                     }
153                 }
154             } catch (IOException e) {
155                 if (!done)
156                     e.printStackTrace();
157             }
158         }
159 
160         void terminate() {
161             done = true;
162             try { serverSocket.close(); }
163             catch (IOException unexpected) { unexpected.printStackTrace(); }
164         }
165 
166         static final byte[] requestEnd = new byte[] {'\r', '\n', '\r', '\n' };
167 
168         // Read until the end of a HTTP request
169         void readOneRequest(InputStream is) throws IOException {
170             int requestEndCount = 0, r;
171             while ((r = is.read()) != -1) {
172                 if (r == requestEnd[requestEndCount]) {
173                     requestEndCount++;
174                     if (requestEndCount == 4) {
175                         break;
176                     }
177                 } else {
178                     requestEndCount = 0;
179                 }
180             }
181         }
182     }
183 
184     private static void addMappingToHostsFile(String host,
185                                               String addr,
186                                               String hostsFileName,
187                                               boolean append)
188         throws IOException
189     {
190         String mapping = addr + " " + host;
191         try (FileWriter fr = new FileWriter(hostsFileName, append);
192              PrintWriter hfPWriter = new PrintWriter(new BufferedWriter(fr))) {
193             hfPWriter.println(mapping);
194         }
195     }
196 
197     static class LookupTestPolicy extends Policy {
198         final PermissionCollection perms = new Permissions();
199 
200         LookupTestPolicy() throws Exception {
201             perms.add(new NetPermission("setProxySelector"));
202             perms.add(new SocketPermission("localhost:1024-", "resolve,accept"));
203             perms.add(new URLPermission("http://allowedAndFound.com:" + port + "/-", "*:*"));
204             perms.add(new URLPermission("http://allowedButNotfound.com:" + port + "/-", "*:*"));
205             perms.add(new FilePermission("<<ALL FILES>>", "read,write,delete"));
206             //perms.add(new PropertyPermission("java.io.tmpdir", "read"));
207         }
208 
209         public PermissionCollection getPermissions(ProtectionDomain domain) {
210             return perms;
211         }
212 
213         public PermissionCollection getPermissions(CodeSource codesource) {
214             return perms;
215         }
216 
217         public boolean implies(ProtectionDomain domain, Permission perm) {
218             return perms.implies(perm) || DEFAULT_POLICY.implies(domain, perm);
219         }
220     }
