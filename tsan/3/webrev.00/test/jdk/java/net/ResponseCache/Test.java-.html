24 /* @test
25  * @summary Fixed a potential NullPointerException when setting a ResponseCache that returns a null CacheRequest
26  * @bug 4837267
27  * @modules jdk.httpserver
28  * @author Michael McMahon
29  */
30 
31 import com.sun.net.httpserver.*;
32 import java.net.*;
33 import java.io.*;
34 import java.util.*;
35 
36 public class Test
37 {
38 
39     static class MyHandler implements HttpHandler {
40         public void handle(HttpExchange t) throws IOException {
41             byte[] b = new byte[1024];
42             int r = 0;
43             InputStream is = t.getRequestBody();
44             while (is.read(b) != -1) ;
45             String response = "This is the response";
46             t.sendResponseHeaders(200, response.length());
47             OutputStream os = t.getResponseBody();
48             os.write(response.getBytes());
49             os.close();
50         }
51     }
52 
53     public static void main(String args[])  throws Exception {
54         HttpServer server = HttpServer.create(new InetSocketAddress(0), 0);
55         server.createContext("/", new MyHandler());
56         server.start();
57         ResponseCache bak = ResponseCache.getDefault();
58 
59         try {
60             ResponseCache.setDefault (new ResponseCache() {
61                 public CacheResponse get (URI uri, String rqstMethod, Map<String,List<String>> rqstHeaders)
62                     throws IOException {
63                     return null;
64                 }
65                 public CacheRequest put(URI uri, URLConnection conn)  throws IOException
66                 {
67                     return null;
68                 }
69             });
70 
71             URL url = new URL ("http://localhost:" + server.getAddress().getPort() + "/");
72             URLConnection urlc = url.openConnection ();
73             InputStream is = urlc.getInputStream();
74             while (is.read() != -1) ;
75             is.close();
76         } finally {
77             ResponseCache.setDefault(bak);
78             server.stop(0);
79         }
80     }
81 }
