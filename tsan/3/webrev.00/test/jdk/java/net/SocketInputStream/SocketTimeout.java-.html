 24 /**
 25  * @test
 26  * @bug 4158021
 27  * @summary cannot distinguish Thread.interrupt and Socket.setSoTimeout exceptions
 28  * @run main SocketTimeout
 29  * @run main/othervm -Djava.net.preferIPv4Stack=true SocketTimeout
 30  */
 31 
 32 import java.net.*;
 33 import java.io.*;
 34 
 35 public class SocketTimeout  {
 36     static final int TIMEOUT = 1000;
 37 
 38     public static void main(String args[]) throws Exception {
 39     InetAddress  sin = InetAddress.getLocalHost();
 40     Socket       soc = null,soc1 = null;
 41     InputStream  is = null;
 42     ServerSocket srv = null;
 43     int          port = 0;
 44 
 45     srv = new ServerSocket(0);
 46     port = srv.getLocalPort();
 47     soc = new Socket(sin, port);
 48     soc1 = srv.accept();
 49     soc.setSoTimeout(TIMEOUT);
 50     srv.setSoTimeout(TIMEOUT);
 51 
 52     try {
 53       is = soc.getInputStream();
 54       is.read();
 55     } catch(InterruptedIOException e) {
 56         try {
 57             if (! (e instanceof java.net.SocketTimeoutException))
 58                 throw new Exception ("Wrong exception class thrown");
 59         } catch(NoClassDefFoundError e1) {
 60             throw new Exception ("SocketTimeoutException: not found");
 61         }
 62     } finally {
 63         soc.close();
 64         soc1.close();
 65     }
 66 
 67     // now check accept
 68 
 69     try {
 70       srv.accept ();
 71     } catch(InterruptedIOException e) {
 72         try {
 73             if (! (e instanceof java.net.SocketTimeoutException))
 74                 throw new Exception ("Wrong exception class thrown");
 75         } catch(NoClassDefFoundError e1) {
 76             throw new Exception ("SocketTimeoutException: not found");
 77         }
 78     } finally {
 79         srv.close();
 80     }
 81 
 82     // Now check DatagramSocket.receive()
 83 
 84     DatagramSocket dg = new DatagramSocket ();
 85     dg.setSoTimeout (TIMEOUT);
 86 
 87     try {
 88       dg.receive (new DatagramPacket (new byte [64], 64));
 89     } catch(InterruptedIOException e) {
 90         try {
 91             if (! (e instanceof java.net.SocketTimeoutException))
 92                 throw new Exception ("Wrong exception class thrown");
 93         } catch(NoClassDefFoundError e1) {
 94             throw new Exception ("SocketTimeoutException: not found");
 95         }
 96     } finally {
 97         dg.close();
 98     }
 99   }
100 }
