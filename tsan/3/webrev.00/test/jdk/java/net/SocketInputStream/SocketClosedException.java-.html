24 /*
25  * @test
26  * @bug 4681556
27  * @summary Wrong text if a read is performed on a socket after it
28  *      has been closed
29  * @run main SocketClosedException
30  * @run main/othervm -Djava.net.preferIPv4Stack=true SocketClosedException
31  */
32 
33 import java.io.*;
34 import java.net.*;
35 
36 public class SocketClosedException {
37     static void doServerSide() throws Exception {
38         try {
39             Socket socket = serverSocket.accept();
40 
41             OutputStream os = socket.getOutputStream();
42 
43             os.write(85);
44             os.flush();
45             socket.close();
46         } finally {
47             serverSocket.close();
48         }
49     }
50 
51     static void doClientSide(int port) throws Exception {
52         Socket socket = new Socket("localhost", port);
53         InputStream is = socket.getInputStream();
54 
55         is.read();
56         socket.close();
57         is.read();
58     }
59 
60     static ServerSocket serverSocket;
61     static Exception serverException = null;
62 
63     public static void main(String[] args) throws Exception {
64         serverSocket = new ServerSocket(0);
65         startServer();
66         try {
67             doClientSide(serverSocket.getLocalPort());
68         } catch (SocketException e) {
69             if (!e.getMessage().equalsIgnoreCase("Socket closed")) {
70                 throw new Exception("Received a wrong exception message: " +
71                                         e.getMessage());
72             }
73             System.out.println("PASSED: received the right exception message: "
74                                         + e.getMessage());
75         }
76         if (serverException != null) {
77             throw serverException;
78         }
79     }
80 
81     static void startServer() {
82         (new Thread() {
83             public void run() {
84                 try {
85                     doServerSide();
86                 } catch (Exception e) {
87                     e.printStackTrace();
88                 }
89             }
90         }).start();
91     }
92 }
