24 /**
25  * @test
26  * @bug 4473092
27  * @library /test/lib
28  * @summary Method throws IOException when object should be returned
29  * @run main HttpResponseCode
30  * @run main/othervm -Djava.net.preferIPv6Addresses=true HttpResponseCode
31  */
32 import java.net.*;
33 import java.io.*;
34 import jdk.test.lib.net.URIBuilder;
35 
36 public class HttpResponseCode implements Runnable {
37     ServerSocket ss;
38     /*
39      * Our "http" server
40      */
41     public void run() {
42         try {
43             Socket s = ss.accept();
44 
45             BufferedReader in = new BufferedReader(
46                 new InputStreamReader(s.getInputStream()) );
47             String req = in.readLine();
48 
49             PrintStream out = new PrintStream(
50                                  new BufferedOutputStream(
51                                     s.getOutputStream() ));
52 
53 
54             /* send the header */
55             out.print("HTTP/1.1 403 Forbidden\r\n");
56             out.print("\r\n");
57             out.flush();
58 
59             s.close();
60             ss.close();
61         } catch (Exception e) {
62             e.printStackTrace();
63         }
64     }
65 
66     HttpResponseCode() throws Exception {
67         /* start the server */
68         InetAddress loopback = InetAddress.getLoopbackAddress();
69         ss = new ServerSocket();
70         ss.bind(new InetSocketAddress(loopback, 0));
71         (new Thread(this)).start();
72 
73         /* establish http connection to server */
74         URL url = URIBuilder.newBuilder()
75             .scheme("http")
76             .loopback()
77             .port(ss.getLocalPort())
78             .path("/missing.nothtml")
79             .toURL();
80         URLConnection uc = url.openConnection(Proxy.NO_PROXY);
81         int respCode1 = ((HttpURLConnection)uc).getResponseCode();
82         ((HttpURLConnection)uc).disconnect();
83         int respCode2 = ((HttpURLConnection)uc).getResponseCode();
84         if (respCode1 != 403 || respCode2 != 403) {
85             throw new RuntimeException("Testing Http response code failed");
86         }
87     }
88 
89     public static void main(String args[]) throws Exception {
90         new HttpResponseCode();
91     }
92 }
