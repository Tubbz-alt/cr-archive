24 import java.net.*;
25 import java.io.*;
26 
27 /*
28  * Regression test for URLClassLoader getURLs() and addURL() methods.
29  * See RFE 4102580: Need URLClassLoader.getURLs() method
30  */
31 class GetURLsTest {
32     public static void main(String[] args) throws Exception {
33         MyURLClassLoader ucl =
34             new MyURLClassLoader(new URL[] { new File(".").toURL() });
35         p("initial urls = ", ucl.getURLs());
36         URL u = ucl.getResource("GetURLsTest.java");
37         if (u != null) {
38             p("found resource = " + u);
39         }
40         ucl.addURL(new File("jars", "class_path_test.jar").toURL());
41         p("new urls = ", ucl.getURLs());
42         Class c = ucl.loadClass("Foo");
43         p("found class = " + c);
44     }
45 
46     static class MyURLClassLoader extends URLClassLoader {
47         public MyURLClassLoader(URL[] urls) {
48             super(urls);
49         }
50         public void addURL(URL url) {
51             super.addURL(url);
52         }
53     }
54 
55     static void p(String s, URL[] urls) {
56         System.out.print(s);
57         if (urls.length > 0) {
58             for (int i = 0; i < urls.length - 1; i++) {
59                 System.out.print(urls[i] + " ");
60             }
61         }
62         System.out.println(urls[urls.length - 1]);
63     }
64 
65     static void p(String s) {
66         System.out.println(s);
67     }
68 }
