24 import java.net.*;
25 import java.io.*;
26 
27 /*
28  * @test
29  * @summary Regression test for URLClassLoader getURLs() and addURL() methods.
30  * See RFE 4102580: Need URLClassLoader.getURLs() method
31  */
32 public class GetURLsTest {
33     static final String TEST_DIR = System.getProperty("test.src", ".");
34 
35     public static void main(String[] args) throws Exception {
36         File testJars = new File(TEST_DIR, "jars");
37 
38         MyURLClassLoader ucl =
39             new MyURLClassLoader(new URL[] { new File(".").toURL() });
40         p("initial urls = ", ucl.getURLs());
41         URL u = ucl.getResource("GetURLsTest.java");
42         if (u != null) {
43             p("found resource = " + u);
44         }
45         ucl.addURL(new File(testJars, "class_path_test.jar").toURL());
46         p("new urls = ", ucl.getURLs());
47         Class c = ucl.loadClass("Foo");
48         p("found class = " + c);
49     }
50 
51     static class MyURLClassLoader extends URLClassLoader {
52         public MyURLClassLoader(URL[] urls) {
53             super(urls);
54         }
55         public void addURL(URL url) {
56             super.addURL(url);
57         }
58     }
59 
60     static void p(String s, URL[] urls) {
61         System.out.print(s);
62         if (urls.length > 0) {
63             for (int i = 0; i < urls.length - 1; i++) {
64                 System.out.print(urls[i] + " ");
65             }
66         }
67         System.out.println(urls[urls.length - 1]);
68     }
69 
70     static void p(String s) {
71         System.out.println(s);
72     }
73 }
