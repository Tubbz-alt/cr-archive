 72     private static class TestX509Certificate extends X509Certificate {
 73         private final X509Certificate cert;
 74         TestX509Certificate(X509Certificate cert) {
 75             this.cert = cert;
 76         }
 77         public Set<String> getCriticalExtensionOIDs() {
 78            return cert.getCriticalExtensionOIDs();
 79         }
 80         public byte[] getExtensionValue(String oid) {
 81             return cert.getExtensionValue(oid);
 82         }
 83         public Set<String> getNonCriticalExtensionOIDs() {
 84             return cert.getNonCriticalExtensionOIDs();
 85         }
 86         public boolean hasUnsupportedCriticalExtension() {
 87             return cert.hasUnsupportedCriticalExtension();
 88         }
 89         public void checkValidity() throws CertificateExpiredException,
 90             CertificateNotYetValidException {
 91             cert.checkValidity();
 92         }
 93         public void checkValidity(Date date) throws CertificateExpiredException,
 94             CertificateNotYetValidException {
 95             cert.checkValidity(date);
 96         }
 97         public int getVersion() { return cert.getVersion(); }
 98         public BigInteger getSerialNumber() { return cert.getSerialNumber(); }
 99         public Principal getIssuerDN() { return cert.getIssuerDN(); }
100         public Principal getSubjectDN() { return cert.getSubjectDN(); }
101         public Date getNotBefore() { return cert.getNotBefore(); }
102         public Date getNotAfter() { return cert.getNotAfter(); }
103         public byte[] getTBSCertificate() throws CertificateEncodingException {
104             return cert.getTBSCertificate();
105         }
106         public byte[] getSignature() { return cert.getSignature(); }
107         public String getSigAlgName() { return cert.getSigAlgName(); }
108         public String getSigAlgOID() { return cert.getSigAlgOID(); }
109         public byte[] getSigAlgParams() { return cert.getSigAlgParams(); }
110         public boolean[] getIssuerUniqueID() {
111             return cert.getIssuerUniqueID();
112         }
113         public boolean[] getSubjectUniqueID() {
114             return cert.getSubjectUniqueID();
115         }
116         public boolean[] getKeyUsage() { return cert.getKeyUsage(); }
117         public int getBasicConstraints() { return cert.getBasicConstraints(); }
118         public byte[] getEncoded() throws CertificateEncodingException {
119             return cert.getEncoded();
120         }
121         public void verify(PublicKey key) throws CertificateException,
122             InvalidKeyException, NoSuchAlgorithmException,
123             NoSuchProviderException, SignatureException {
124             cert.verify(key);
125         }
126         public void verify(PublicKey key, String sigProvider) throws
127             CertificateException, InvalidKeyException, NoSuchAlgorithmException,
128             NoSuchProviderException, SignatureException {
129             cert.verify(key, sigProvider);
130         }
131         public PublicKey getPublicKey() { return cert.getPublicKey(); }
132         public String toString() { return cert.toString(); }
133     }
134 
135     public static void main(String[] args) throws Exception {
136         X509Certificate cert = CertUtils.getCertFromString(TEST_CERT);
137         new TestX509Certificate(cert).verify(cert.getPublicKey(),
138                                              (Provider)null);
139     }
