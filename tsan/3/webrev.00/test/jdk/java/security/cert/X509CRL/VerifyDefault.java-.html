 75     private static class TestX509CRL extends X509CRL {
 76         private final X509CRL crl;
 77         TestX509CRL(X509CRL crl) {
 78             this.crl = crl;
 79         }
 80         public Set<String> getCriticalExtensionOIDs() {
 81            return crl.getCriticalExtensionOIDs();
 82         }
 83         public byte[] getExtensionValue(String oid) {
 84             return crl.getExtensionValue(oid);
 85         }
 86         public Set<String> getNonCriticalExtensionOIDs() {
 87             return crl.getNonCriticalExtensionOIDs();
 88         }
 89         public boolean hasUnsupportedCriticalExtension() {
 90             return crl.hasUnsupportedCriticalExtension();
 91         }
 92         public Set<? extends X509CRLEntry> getRevokedCertificates() {
 93             return crl.getRevokedCertificates();
 94         }
 95         public X509CRLEntry getRevokedCertificate(BigInteger serialNumber) {
 96             return crl.getRevokedCertificate(serialNumber);
 97         }
 98         public boolean isRevoked(Certificate cert) {
 99             return crl.isRevoked(cert);
100         }
101         public Date getNextUpdate() { return crl.getNextUpdate(); }
102         public Date getThisUpdate() { return crl.getThisUpdate(); }
103         public int getVersion() { return crl.getVersion(); }
104         public Principal getIssuerDN() { return crl.getIssuerDN(); }
105         public byte[] getTBSCertList() throws CRLException {
106             return crl.getTBSCertList();
107         }
108         public byte[] getSignature() { return crl.getSignature(); }
109         public String getSigAlgName() { return crl.getSigAlgName(); }
110         public String getSigAlgOID() { return crl.getSigAlgOID(); }
111         public byte[] getSigAlgParams() { return crl.getSigAlgParams(); }
112         public byte[] getEncoded() throws CRLException {
113             return crl.getEncoded();
114         }
115         public void verify(PublicKey key) throws CRLException,
116             InvalidKeyException, NoSuchAlgorithmException,
117             NoSuchProviderException, SignatureException {
118             crl.verify(key);
119         }
120         public void verify(PublicKey key, String sigProvider) throws
121             CRLException, InvalidKeyException, NoSuchAlgorithmException,
122             NoSuchProviderException, SignatureException {
123             crl.verify(key, sigProvider);
124         }
125         public String toString() { return crl.toString(); }
126     }
127 
128     public static void main(String[] args) throws Exception {
129         X509Certificate cert = CertUtils.getCertFromString(TEST_CERT);
130         X509CRL crl = CertUtils.getCRLFromString(TEST_CRL);
131         new TestX509CRL(crl).verify(cert.getPublicKey(), (Provider)null);
132     }
