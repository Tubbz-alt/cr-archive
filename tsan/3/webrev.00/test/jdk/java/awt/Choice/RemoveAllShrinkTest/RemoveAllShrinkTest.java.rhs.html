24 /*
25   @test
26   @key headful
27   @bug 4851798 8041896 8225126
28   @summary Tests Choice List shrinks after removeAll
29   @run main RemoveAllShrinkTest
30 */
31 
32 import java.awt.BorderLayout;
33 import java.awt.Choice;
34 import java.awt.Color;
35 import java.awt.Frame;
36 import java.awt.Panel;
37 import java.awt.Point;
38 import java.awt.Robot;
39 import java.awt.event.InputEvent;
40 
41 public class RemoveAllShrinkTest {
42 
43     public static void main(String[] args) {
44         Frame f = new Frame();
45         try {
46             Choice choice = new Choice();
47 
48             for (int i = 0; i < 10; ++i) {
49                 choice.addItem("Item " + i);
50             }
51 
52             f.add(choice, BorderLayout.NORTH);
53             Panel panel = new Panel();
54             panel.setBackground(Color.RED);
55             f.add(panel);
56 
57             f.setSize(200, 200);
58             f.setLocationRelativeTo(null);
59             f.setVisible(true);
60             f.toFront();
61 
62             choice.removeAll();
63 
64             try {
65                 Robot robot = new Robot();
66                 robot.setAutoWaitForIdle(true);
67                 robot.setAutoDelay(50);
68 
69                 robot.waitForIdle();
70                 Thread.sleep(200);
71 
72                 Point pt = choice.getLocationOnScreen();
73                 robot.mouseMove(pt.x + choice.getWidth() - choice.getHeight() / 2,
74                         pt.y + choice.getHeight() / 2);
75                 robot.mousePress(InputEvent.BUTTON1_MASK);
76                 robot.mouseRelease(InputEvent.BUTTON1_MASK);
77 
78                 Thread.sleep(400);
79 
80                 Point pt1 = panel.getLocationOnScreen();
81 
82                 Color color = robot.getPixelColor(pt1.x + panel.getWidth() / 2,
83                         pt1.y + panel.getHeight() / 2);
84 
85                 if (!color.equals(Color.RED)) {
86                     throw new RuntimeException("RemoveAllShrinkTest failed. " + color);
87                 }
88             } catch (Exception e) {
89                 throw new RuntimeException("The test was not completed.\n\n" + e);
90             }
91         } finally {
92             f.dispose();
93         }
94     }
95 }
