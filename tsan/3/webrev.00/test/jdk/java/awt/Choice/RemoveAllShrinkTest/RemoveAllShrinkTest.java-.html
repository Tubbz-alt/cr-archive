24 /*
25   @test
26   @key headful
27   @bug 4851798 8041896
28   @summary Tests Choice List shrinks after removeAll
29   @run main RemoveAllShrinkTest
30 */
31 
32 import java.awt.*;
33 import java.awt.event.*;
34 
35 
36 public class RemoveAllShrinkTest {
37 
38     public static void main(String[] args) {
39         Frame f = new Frame();
40         Choice choice = new Choice();
41 
42         for (int i = 0; i < 10; ++i) {
43             choice.addItem("Item " + i);
44         }
45 
46         f.add(choice, BorderLayout.NORTH);
47         Panel panel = new Panel();
48         panel.setBackground(Color.RED);
49         f.add(panel);
50 
51         f.setSize(200, 200);
52         f.setVisible(true);
53         f.toFront();
54 
55         choice.removeAll();
56 
57         try {
58             Robot robot = new Robot();
59             robot.setAutoWaitForIdle(true);
60             robot.setAutoDelay(50);
61 
62             robot.waitForIdle();
63             Thread.sleep(200);
64 
65             Point pt = choice.getLocationOnScreen();
66             robot.mouseMove(pt.x + choice.getWidth() - choice.getHeight() / 2,
67                     pt.y + choice.getHeight() / 2);
68             robot.mousePress(InputEvent.BUTTON1_MASK);
69             robot.mouseRelease(InputEvent.BUTTON1_MASK);
70 
71             Thread.sleep(400);
72 
73             Point pt1 = panel.getLocationOnScreen();
74 
75             Color color = robot.getPixelColor(pt1.x + panel.getWidth() / 2,
76                     pt1.y + panel.getHeight() / 2);
77 
78             if (!color.equals(Color.RED)) {
79                 throw new RuntimeException("RemoveAllShrinkTest failed. " + color);
80             }
81         } catch (Exception e) {
82             throw new RuntimeException("The test was not completed.\n\n" + e);
83         }
84     }
85 }
