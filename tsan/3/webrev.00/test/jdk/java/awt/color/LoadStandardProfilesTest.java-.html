24 /*
25  * @test
26  * @bug 8039271
27  * @summary test all standard profiles load correctly.
28  */
29 
30 import java.awt.color.ICC_Profile;
31 
32 public class LoadStandardProfilesTest {
33 
34     public static void main(String args[]) {
35         try {
36             ICC_Profile sRGB      = ICC_Profile.getInstance("sRGB.pf");
37             ICC_Profile gray      = ICC_Profile.getInstance("GRAY.pf");
38             ICC_Profile pycc      = ICC_Profile.getInstance("PYCC.pf");
39             ICC_Profile ciexyz    = ICC_Profile.getInstance("CIEXYZ.pf");
40             ICC_Profile linearRGB = ICC_Profile.getInstance("LINEAR_RGB.pf");
41 
42             if (sRGB == null ||
43                 gray == null ||
44                 pycc == null ||
45                 ciexyz == null ||
46                 linearRGB == null)
47             {
48                 throw new RuntimeException("null profile.");
49             }
50         } catch (Exception e) {
51            throw new RuntimeException(e);
52         }
53     }
54 }
