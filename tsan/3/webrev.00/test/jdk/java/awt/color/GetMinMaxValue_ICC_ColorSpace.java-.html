23 import java.awt.color.ColorSpace;
24 import java.awt.color.ICC_ColorSpace;
25 import java.awt.color.ICC_Profile;
26 import java.io.IOException;
27 
28 /**
29   * @test
30   * @bug 8072678
31   * @author Prasanta Sadhukhan
32   */
33 
34 public class GetMinMaxValue_ICC_ColorSpace {
35 
36     public static void main(String[] a) throws Exception {
37         ICC_Profile cmyk_profile = ICC_Profile.getInstance(ColorSpace.CS_sRGB);
38         ICC_ColorSpace colorSpace = new ICC_ColorSpace(cmyk_profile);
39         String minstr = null;
40         String maxstr = null;
41 
42         colorSpace.fromRGB(new float[]{4.3f,3.1f,2.2f});
43         try {
44             System.out.println("minvalue " + colorSpace.getMinValue(3));
45         } catch (IllegalArgumentException iae) {
46             minstr = iae.toString();
47         }
48         try {
49             System.out.println("maxvalue " + colorSpace.getMaxValue(3));
50         } catch (IllegalArgumentException iae) {
51             maxstr = iae.toString();
52         }
53 
54         if (minstr.endsWith("+ component") || maxstr.endsWith("+ component")) {
55             System.out.println("Test failed");
56             throw new RuntimeException("IllegalArgumentException contains incorrect text message");
57         } else {
58             System.out.println("Test passed");
59         }
60     }
61 }
