24 /*
25  * @test
26  * @key headful
27  * @bug      8176795
28  * @summary  Test verifies that we get proper color when we draw translucent
29  *           color over an opaque color using X Render extension in Linux.
30  * @requires (os.family == "linux")
31  * @run      main XRenderTranslucentColorDrawTest -Dsun.java2d.xrender=true
32  */
33 
34 import java.awt.Color;
35 import java.awt.Graphics2D;
36 import java.awt.GraphicsConfiguration;
37 import java.awt.GraphicsDevice;
38 import java.awt.GraphicsEnvironment;
39 import java.awt.image.BufferedImage;
40 import java.awt.image.VolatileImage;
41 
42 public class XRenderTranslucentColorDrawTest {
43 
44     public static void main(String[] args) throws Exception {
45         GraphicsEnvironment env = GraphicsEnvironment.
46                 getLocalGraphicsEnvironment();
47         GraphicsConfiguration translucentGC = null;
48         SCREENS: for (GraphicsDevice screen : env.getScreenDevices()) {
49             for (GraphicsConfiguration gc : screen.getConfigurations()) {
50                 if (gc.isTranslucencyCapable()) {
51                     translucentGC = gc;
52                     break SCREENS;
53                 }
54             }
55         }
56         if (translucentGC == null) {
57             throw new RuntimeException("No suitable gc found.");
58         }
59         int width = 10;
60         int height = 10;
61         VolatileImage image = translucentGC.
62                 createCompatibleVolatileImage(width, height);
63         Graphics2D g = image.createGraphics();
64         // draw opaque black color
65         g.setColor(new Color(0xff000000, true));
66         g.fillRect(0, 0, width, height);
67         // draw translucent white color over opaque black color
68         g.setColor(new Color(0x80ffffff, true));
69         g.fillRect(0, 0, width, height);
70         g.dispose();
71         // Get snapshot of VolatileImage to pick color and verify the same
72         BufferedImage snapshot = image.getSnapshot();
73         int argb = snapshot.getRGB(width / 2, height / 2);
74         // we expect the resultant rgb hex value to be ff808080
75         if (!(Integer.toHexString(argb).equals("ff808080"))) {
76             throw new RuntimeException("Using X Render extension for drawing"
77                     + " translucent color is not giving expected results.");
78         }
79     }
80 }
