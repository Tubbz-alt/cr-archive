24 /*
25   @test
26   @bug 6783910
27   @summary java.awt.Color.brighter()/darker() methods make color opaque
28   @author Andrei Dmitriev: area=awt-color
29   @run main OpacityChange
30 */
31 
32 import java.awt.*;
33 
34 public class OpacityChange {
35     private final static int INITIAL_ALPHA = 125;
36 
37     public static void main(String argv[]) {
38         Color color = new Color(20, 20, 20, INITIAL_ALPHA);
39         System.out.println("Initial alpha: " + color.getAlpha());
40         Color colorBrighter = color.brighter();
41         System.out.println("New alpha (after brighter): " + colorBrighter.getAlpha());
42 
43         Color colorDarker = color.darker();
44         System.out.println("New alpha (after darker): " + colorDarker.getAlpha());
45 
46 
47         if (INITIAL_ALPHA != colorBrighter.getAlpha()) {
48             throw new RuntimeException("Brighter color alpha has changed from : " +INITIAL_ALPHA + " to " + colorBrighter.getAlpha());
49         }
50         if (INITIAL_ALPHA != colorDarker.getAlpha()) {
51             throw new RuntimeException("Darker color alpha has changed from : " +INITIAL_ALPHA + " to " + colorDarker.getAlpha());
52         }
53     }
54 }
