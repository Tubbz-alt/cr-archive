24 /*
25  * @test
26  * @bug 8210058
27  * @summary Algorithmic Italic font leans opposite angle in Printing
28  */
29 
30 import java.awt.Color;
31 import java.awt.Font;
32 import java.awt.Graphics2D;
33 import java.awt.image.BufferedImage;
34 import java.io.File;
35 
36 public class RotatedItalicsTest {
37     public static void main(String[] args) throws Exception {
38         File fontFile = new File(System.getProperty("test.src", "."), "A.ttf");
39         Font baseFont = Font.createFont(Font.TRUETYPE_FONT, fontFile);
40         Font font = baseFont.deriveFont(Font.ITALIC, 120);
41 
42         BufferedImage image = new BufferedImage(100, 100,
43                                                 BufferedImage.TYPE_INT_RGB);
44 
45         Graphics2D g = image.createGraphics();
46         g.rotate(Math.PI / 2);
47         g.setFont(font);
48         g.drawString("A", 10, -10);
49         g.dispose();
50 
51         if (image.getRGB(50, 76) != Color.white.getRGB()) {
52             throw new RuntimeException("Wrong glyph rendering");
53         }
54     }
55 }
