31 import java.awt.Font;
32 import java.awt.GraphicsEnvironment;
33 import java.util.Arrays;
34 
35 import sun.font.Font2D;
36 import sun.font.SunFontManager;
37 
38 public class DefaultFontTest {
39     public static void main(String[] args) throws Exception {
40         // this triggers font initialization
41         String[] fontFamilyNames = GraphicsEnvironment.getLocalGraphicsEnvironment().getAvailableFontFamilyNames();
42         if (fontFamilyNames != null) {
43             System.out.println("FontFamilies:");
44             Arrays.asList(fontFamilyNames).stream().forEach((fontname)->System.out.println(fontname));
45         }
46         SunFontManager sfm = SunFontManager.getInstance();
47         Font[] createdFonts = sfm.getCreatedFonts();
48         if (createdFonts != null) {
49             System.out.println("\nCreated Fonts:");
50             Arrays.asList(createdFonts).stream().forEach((font)->System.out.println(font));
51         }
52         Font2D[] registeredFonts = sfm.getRegisteredFonts();
53         if (registeredFonts != null) {
54             System.out.println("\nRegistered Fonts:");
55             Arrays.asList(registeredFonts).stream().forEach((font)->System.out.println(font));
56         }
57         System.out.println("\nDefault physical font: " + sfm.getDefaultPhysicalFont());
58     }
59 }
