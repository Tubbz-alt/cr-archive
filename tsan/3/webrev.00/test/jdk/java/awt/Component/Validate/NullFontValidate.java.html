24 import java.awt.Frame;
25 
26 /**
27  * @test
28  * @bug 8233657
29  * @key headful
30  * @summary tests that no NPE is thrown when calling Component.validate()
31  *          with null font
32  */
33 public final class NullFontValidate {
34 
35     public static void main(final String[] args) {
36         final Frame frame = new Frame();
37         try {
38             frame.pack();
39             frame.setFont(null);
40             frame.validate();
41         } finally {
42             frame.dispose();
43         }
44     }
45 }
