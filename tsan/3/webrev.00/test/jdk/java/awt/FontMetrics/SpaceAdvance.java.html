29 import java.awt.Font;
30 import java.awt.FontMetrics ;
31 import java.awt.Graphics2D;
32 import java.awt.image.BufferedImage;
33 
34 public class SpaceAdvance {
35     public static void main(String[] args) throws Exception {
36 
37         BufferedImage bi = new BufferedImage(1,1,1);
38         Graphics2D g2d = bi.createGraphics();
39         Font font = new Font(Font.DIALOG, Font.PLAIN, 12);
40         if (!font.canDisplay(' ')) {
41             return;
42         }
43         g2d.setFont(font);
44         FontMetrics fm = g2d.getFontMetrics();
45         if (fm.charWidth(' ') == 0) {
46             throw new RuntimeException("Space has char width of 0");
47         }
48     }
49 }
