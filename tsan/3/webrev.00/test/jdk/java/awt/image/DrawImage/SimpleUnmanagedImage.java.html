 61 /**
 62  * @test
 63  * @key headful
 64  * @bug 8029253 6207877
 65  * @summary Tests the case when unmanaged image is drawn to VI.
 66  *          Results of the blit to compatibleImage are used for comparison.
 67  * @author Sergey Bylokhov
 68  * @run main/othervm SimpleUnmanagedImage
 69  * @run main/othervm -Dsun.java2d.uiScale=1 SimpleUnmanagedImage
 70  * @run main/othervm -Dsun.java2d.uiScale=2 SimpleUnmanagedImage
 71  */
 72 public final class SimpleUnmanagedImage {
 73 
 74     // See the same test for managed images: SimpleManagedImage
 75 
 76     private static final int[] TYPES = {TYPE_INT_RGB, TYPE_INT_ARGB,
 77                                         TYPE_INT_ARGB_PRE, TYPE_INT_BGR,
 78                                         TYPE_3BYTE_BGR, TYPE_4BYTE_ABGR,
 79                                         TYPE_4BYTE_ABGR_PRE,
 80                                         /*TYPE_USHORT_565_RGB,
 81                                         TYPE_USHORT_555_RGB, TYPE_BYTE_GRAY,
 82                                         TYPE_USHORT_GRAY,*/ TYPE_BYTE_BINARY,
 83                                         TYPE_BYTE_INDEXED, TYPE_CUSTOM};
 84     private static final int[] TRANSPARENCIES = {OPAQUE, BITMASK, TRANSLUCENT};
 85 
 86     public static void main(final String[] args) throws IOException {
 87         for (final int viType : TRANSPARENCIES) {
 88             for (final int biType : TYPES) {
 89                 BufferedImage bi = makeUnmanagedBI(biType);
 90                 fill(bi);
 91                 test(bi, viType);
 92             }
 93         }
 94     }
 95 
 96     private static void test(BufferedImage bi, int type)
 97             throws IOException {
 98         GraphicsEnvironment ge = GraphicsEnvironment
 99                 .getLocalGraphicsEnvironment();
100         GraphicsConfiguration gc = ge.getDefaultScreenDevice()
101                                      .getDefaultConfiguration();
102         VolatileImage vi = gc.createCompatibleVolatileImage(1000, 1000, type);
103         BufferedImage gold = gc.createCompatibleImage(1000, 1000, type);
104         // draw to compatible Image
105         init(gold);
106         Graphics2D big = gold.createGraphics();
107         big.drawImage(bi, 7, 11, null);
108         big.dispose();
109         // draw to volatile image
110         BufferedImage snapshot;
111         while (true) {
112             vi.validate(gc);
113             if (vi.validate(gc) != VolatileImage.IMAGE_OK) {
114                 try {
115                     Thread.sleep(100);
116                 } catch (final InterruptedException ignored) {
117                 }
118                 continue;
119             }
120             init(vi);
121             Graphics2D vig = vi.createGraphics();
122             vig.drawImage(bi, 7, 11, null);
123             vig.dispose();
124             snapshot = vi.getSnapshot();
125             if (vi.contentsLost()) {
126                 try {
127                     Thread.sleep(100);
128                 } catch (final InterruptedException ignored) {
129                 }
130                 continue;
131             }
132             break;
133         }
134         // validate images
135         for (int x = 0; x < 1000; ++x) {
136             for (int y = 0; y < 1000; ++y) {
137                 if (gold.getRGB(x, y) != snapshot.getRGB(x, y)) {
138                     ImageIO.write(gold, "png", new File("gold.png"));
139                     ImageIO.write(snapshot, "png", new File("bi.png"));
140                     throw new RuntimeException("Test failed.");
141                 }
142             }
143         }
144     }
145 
146     private static BufferedImage makeUnmanagedBI(final int type) {
147         final BufferedImage bi;
148         if (type == TYPE_CUSTOM) {
149             bi = makeCustomUnmanagedBI();
150         } else {
151             bi = new BufferedImage(511, 255, type);
152         }
153         final DataBuffer db = bi.getRaster().getDataBuffer();
154         if (db instanceof DataBufferInt) {
155             ((DataBufferInt) db).getData();
156         } else if (db instanceof DataBufferShort) {
157             ((DataBufferShort) db).getData();
158         } else if (db instanceof DataBufferByte) {
159             ((DataBufferByte) db).getData();
160         }
161         bi.setAccelerationPriority(0.0f);
162         return bi;
163     }
164 
165     /**
166      * Returns the custom buffered image, which mostly identical to
167      * BufferedImage.(w,h,TYPE_3BYTE_BGR), but uses the bigger scanlineStride.
168      * This means that the raster will have gaps, between the rows.
169      */
170     private static BufferedImage makeCustomUnmanagedBI() {
171         int w = 511, h = 255;
172         ColorSpace cs = ColorSpace.getInstance(ColorSpace.CS_sRGB);
173         int[] nBits = {8, 8, 8};
174         int[] bOffs = {2, 1, 0};
175         ColorModel colorModel = new ComponentColorModel(cs, nBits, false, false,
176                                                         Transparency.OPAQUE,
177                                                         DataBuffer.TYPE_BYTE);
178         WritableRaster raster =
179                 Raster.createInterleavedRaster(DataBuffer.TYPE_BYTE, w, h,
180                                                w * 3 + 2, 3, bOffs, null);
181         return new BufferedImage(colorModel, raster, true, null);
182     }
183 
184     private static void init(final Image image) {
185         final Graphics2D graphics = (Graphics2D) image.getGraphics();
186         graphics.setComposite(AlphaComposite.Src);
187         graphics.setColor(new Color(0, 0, 0, 0));
188         graphics.fillRect(0, 0, image.getWidth(null), image.getHeight(null));
189         graphics.dispose();
190     }
191 
192     private static void fill(final Image image) {
193         final Graphics2D graphics = (Graphics2D) image.getGraphics();
194         graphics.setComposite(AlphaComposite.Src);
195         for (int i = 0; i < image.getHeight(null); ++i) {
196             graphics.setColor(new Color(i, 0, 0));
197             graphics.fillRect(0, i, image.getWidth(null), 1);
198         }
199         graphics.dispose();
200     }
