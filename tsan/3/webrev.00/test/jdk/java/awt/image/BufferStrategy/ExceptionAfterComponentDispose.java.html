 24 import java.awt.Canvas;
 25 import java.awt.EventQueue;
 26 import java.awt.Frame;
 27 import java.awt.image.BufferStrategy;
 28 import java.util.List;
 29 
 30 import javax.swing.JFrame;
 31 
 32 /**
 33  * @test
 34  * @bug 6933331
 35  * @key headful
 36  * @summary Tests that BufferStrategy works after the peer is recreated
 37  */
 38 public final class ExceptionAfterComponentDispose {
 39 
 40     public static void main(final String[] args) throws Exception {
 41         // Will check Canvas on the main thread
 42         for (int size = -3; size < 3; ++size) {
 43             for (int numBuffers = 1; numBuffers < 3; ++numBuffers) {
 44                 testCanvas(size, numBuffers);
 45             }
 46         }
 47         // Will check JFrame on EDT
 48         EventQueue.invokeAndWait(() -> {
 49             for (int size = -3; size < 3; ++size) {
 50                 for (int numBuffers = 1; numBuffers < 3; ++numBuffers) {
 51                     testJFrame(size, numBuffers);
 52                 }
 53             }
 54         });
 55     }
 56 
 57     private static void testCanvas(final int size, final int numBuffers) {
 58         final Frame frame = new Frame();
 59         try {
 60             final Canvas canvas = new Canvas();
 61             frame.setLayout(null); // we would like to control the size
 62             frame.add(canvas);
 63             frame.pack(); // creates a peer
 64             canvas.setSize(size, size);
 65 
 66             // BufferStrategy works when the peer just created
 67             canvas.createBufferStrategy(numBuffers);
 68             final BufferStrategy bs = canvas.getBufferStrategy();
 69             if (bs.getDrawGraphics() == null) {
 70                 throw new RuntimeException("DrawGraphics is null");
 71             }
 72 
 73             // BufferStrategy should work when the peer is recreated
 74             frame.dispose();
 75             frame.pack();
 76             if (canvas.getBufferStrategy() != bs) {
 77                 throw new RuntimeException("BS was changed");
 78             }
 79             if (bs.getDrawGraphics() == null) {
 80                 throw new RuntimeException("DrawGraphics is null");
 81             }
 82         } finally {
 83             frame.dispose();
 84         }
 85     }
 86 
 87     private static void testJFrame(final int size, final int numBuffers) {
 88         for (final boolean undecorated : List.of(true, false)) {
 89             final JFrame frame = new JFrame();
 90             try {
 91                 frame.setUndecorated(undecorated);
 92                 frame.pack(); // creates a peer
 93                 frame.setSize(size, size);
 94 
 95                 // BufferStrategy works when the peer just created
 96                 frame.createBufferStrategy(numBuffers);
 97                 final BufferStrategy bs = frame.getBufferStrategy();
 98                 if (bs.getDrawGraphics() == null) {
 99                     throw new RuntimeException("DrawGraphics is null");
100                 }
101 
102                 // BufferStrategy should work when the peer is recreated
103                 frame.dispose();
104                 frame.pack();
105                 if (frame.getBufferStrategy() != bs) {
106                     throw new RuntimeException("BS was changed");
107                 }
108                 if (bs.getDrawGraphics() == null) {
109                     throw new RuntimeException("DrawGraphics is null");
110                 }
111             } finally {
112                 frame.dispose();
113             }
114         }
115     }
