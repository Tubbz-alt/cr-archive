24 /*
25   @test
26   @key headful
27   @bug 6448069
28   @summary namefilter is not called for file dialog on windows
29   @library ../../regtesthelpers
30   @build Util
31   @run main FilenameFilterTest
32 */
33 







34 import java.awt.*;
35 
36 import java.io.File;
37 import java.io.FilenameFilter;
38 
39 import test.java.awt.regtesthelpers.Util;
40 
41 public class FilenameFilterTest {









42 
43     static volatile boolean filter_was_called = false;
44     static FileDialog fd;








45 
46     public static void main(final String[] args) {
47         EventQueue.invokeLater(new Runnable() {
48                 public void run() {
49                     fd = new FileDialog(new Frame(""), "hello world", FileDialog.LOAD);
50                     fd.setFilenameFilter(new FilenameFilter() {
51                             public boolean accept(File dir, String name) {
52                                 filter_was_called = true;
53                                 System.out.println(Thread.currentThread() + " name = " + name );
54                                 return true;
55                             }
56                         });
57                     fd.setDirectory(System.getProperty("test.src"));
58                     fd.setVisible(true);
59                 }
60             });
61         Util.waitForIdle(null);
62         if (fd == null) {
63             throw new RuntimeException("fd is null (very unexpected thing :(");
64         }
65         //Wait a little; some native dialog implementations may take a while
66         //to initialize and call the filter. See 6959787 for an example.
67         try {
68             Thread.sleep(5000);
69         } catch (Exception ex) {
70         }
71         fd.dispose();
72         if (!filter_was_called) {
73             throw new RuntimeException("Filter was not called");
74         }
75     }

76 }// class FilenameFilterTest
