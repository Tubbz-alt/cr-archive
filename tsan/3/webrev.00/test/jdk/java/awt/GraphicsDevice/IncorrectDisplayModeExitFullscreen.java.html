 25 import java.awt.Color;
 26 import java.awt.DisplayMode;
 27 import java.awt.Frame;
 28 import java.awt.GraphicsDevice;
 29 import java.awt.GraphicsEnvironment;
 30 
 31 /**
 32  * @test
 33  * @key headful
 34  * @bug 8019587
 35  * @author Sergey Bylokhov
 36  * @library /lib/client/
 37  * @build ExtendedRobot
 38  * @run main IncorrectDisplayModeExitFullscreen
 39  */
 40 public class IncorrectDisplayModeExitFullscreen {
 41     static ExtendedRobot robot;
 42 
 43     public static void main(final String[] args) {
 44 
 45         final GraphicsDevice[] devices =
 46                 GraphicsEnvironment.getLocalGraphicsEnvironment()
 47                                    .getScreenDevices();
 48         if (devices.length < 2 || devices[0].getDisplayModes().length < 2
 49                 || !devices[0].isDisplayChangeSupported()
 50                 || !devices[0].isFullScreenSupported()
 51                 || !devices[1].isFullScreenSupported()) {
 52             System.err.println("Testcase is not applicable");
 53             return;
 54         }
 55         final DisplayMode defaultDM = devices[0].getDisplayMode();
 56         final DisplayMode[] dms = devices[0].getDisplayModes();
 57         DisplayMode nonDefaultDM = null;
 58 
 59         for (final DisplayMode dm : dms) {
 60             if (!dm.equals(defaultDM)) {
 61                 nonDefaultDM = dm;
 62                 break;
 63             }
 64         }
 65         if (nonDefaultDM == null) {
 66             System.err.println("Testcase is not applicable");
 67             return;
 68         }
 69 
 70         try {
 71             robot = new ExtendedRobot();
 72         }catch(Exception ex) {
 73             ex.printStackTrace();
 74             throw new RuntimeException("Unexpected failure");
 75         }
 76 
 77         final Frame frame = new Frame();
 78         frame.setBackground(Color.GREEN);
 79         frame.setUndecorated(true);
 80         try {
 81             devices[0].setFullScreenWindow(frame);
 82             sleep();
 83             devices[0].setDisplayMode(nonDefaultDM);
 84             sleep();
 85             devices[1].setFullScreenWindow(frame);
 86             sleep();
 87             if (!defaultDM.equals(devices[0].getDisplayMode())) {
 88                 throw new RuntimeException("DisplayMode is not restored");
 89             }
 90         } finally {
 91             // cleaning up
 92             devices[0].setFullScreenWindow(null);
 93             devices[1].setFullScreenWindow(null);
 94             frame.dispose();
 95         }
 96     }
 97     private static void sleep() {
 98         robot.waitForIdle(1500);
 99     }
100 }
