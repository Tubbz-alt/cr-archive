103     int getFileSize(String fileName) {
104         try {
105             URL url = new URL(getCodeBase(), fileName);



106             InputStream inStream = url.openStream();
107             BufferedInputStream fontStream = new BufferedInputStream(inStream);
108             int size = 0;
