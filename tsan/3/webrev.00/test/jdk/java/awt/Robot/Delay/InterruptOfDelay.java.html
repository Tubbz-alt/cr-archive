24 import java.awt.Robot;
25 import java.io.ByteArrayOutputStream;
26 import java.io.PrintStream;
27 
28 /**
29  * @test
30  * @key headful
31  * @bug 8210231
32  */
33 public final class InterruptOfDelay {
34 
35     static final class DormantThread extends Thread {
36 
37         public void run() {
38             final PrintStream old = System.err;
39             final ByteArrayOutputStream err = new ByteArrayOutputStream();
40             System.setErr(new PrintStream(err));
41             try {
42                 new Robot().delay(10000);
43             } catch (final Exception e) {
44                 throw new RuntimeException(e);
45             } finally {
46                 System.setErr(old);
47             }
48             if (!err.toString().isBlank()) {
49                 throw new RuntimeException("Error stream is not empty: " + err);
50             }
51             if (!Thread.currentThread().isInterrupted()) {
52                 throw new RuntimeException("Thread was not interrupted");
53             }
54         }
55     }
56 
57     public static void main(final String args[]) throws Exception {
58         final Thread t = new DormantThread();
59         t.start();
60         Thread.sleep(1000);
61         t.interrupt();
62         t.join();
63     }
64 }
