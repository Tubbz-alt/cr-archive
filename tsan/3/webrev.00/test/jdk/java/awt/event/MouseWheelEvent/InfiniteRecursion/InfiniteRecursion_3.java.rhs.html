 34 /**
 35  * InfiniteRecursion_3.java
 36  *
 37  * summary: put a JButton into Applet.
 38  * Add MouseWheelListener to Applet.
 39  * Rotating a wheel over the JButton would result in stack overflow.
 40  */
 41 
 42 import java.awt.*;
 43 import java.awt.event.*;
 44 import javax.swing.*;
 45 import test.java.awt.regtesthelpers.Util;
 46 import test.java.awt.regtesthelpers.AbstractTest;


 47 
 48 public class InfiniteRecursion_3 extends Frame {
 49     final static Robot robot = Util.createRobot();
 50     final static int MOVE_COUNT = 5;
 51     //*2 for both rotation directions,
 52     //*2 as Java sends the wheel event to every for nested component in hierarchy under cursor
 53     final static int EXPECTED_COUNT = MOVE_COUNT * 2 * 2;
 54     static int actualEvents = 0;
 55 
 56     public static void main(final String[] args) {
 57         InfiniteRecursion_3 app = new InfiniteRecursion_3();
 58         app.init();
 59         app.start();
 60     }
 61 
 62     public void init()
 63     {
 64         setLayout (new BorderLayout ());
 65     }//End  init()
 66 
 67     public void start ()
 68     {
 69         JButton jButton = new JButton();
 70 
 71         this.setSize(200, 200);
 72         this.addMouseWheelListener(new MouseWheelListener() {
 73                 public void mouseWheelMoved(MouseWheelEvent e)
 74                 {
 75                     System.out.println("Wheel moved on APPLET : "+e);
 76                     actualEvents++;
 77                 }
 78             });
 79 
 80         this.add(jButton);
 81         this.setLocationRelativeTo(null);
 82         this.setVisible(true);
 83         this.validate();
 84 
 85         Util.waitForIdle(robot);
 86 
 87         Util.pointOnComp(jButton, robot);
 88         Util.waitForIdle(robot);
 89 
 90         for (int i = 0; i < MOVE_COUNT; i++){
 91             robot.mouseWheel(1);
 92             robot.delay(10);
 93         }
 94 
 95         for (int i = 0; i < MOVE_COUNT; i++){
 96             robot.mouseWheel(-1);
 97             robot.delay(10);
 98         }
 99 
100         Util.waitForIdle(robot);
101         //Not fair to check for multiplier 4 as it's not specified actual number of WheelEvents
102         //result in a single wheel rotation.
103         if (actualEvents != EXPECTED_COUNT) {
104             AbstractTest.fail("Expected events count: "+ EXPECTED_COUNT+" Actual events count: "+ actualEvents);
105         }
106     }// start()
107 }
