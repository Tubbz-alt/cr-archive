24 /*
25  * @test
26  * @key headful
27  * @bug        4752312
28  * @summary    Tests that after moving non-focusable window it ungrabs mouse pointer

29  * @library    ../../regtesthelpers
30  * @build      Util
31  * @run        main FrameJumpingToMouse
32  */
33 




34 import java.awt.Point;
35 import java.awt.Robot;


36 import java.awt.event.InputEvent;
37 import javax.swing.JFrame;
38 import test.java.awt.regtesthelpers.Util;
39 
40 public class FrameJumpingToMouse {
41 
42     JFrame frame = new JFrame("Test jumping frame");
43     Robot robot = Util.createRobot();
44 
45     public static void main(String[] args) {
46         FrameJumpingToMouse test = new FrameJumpingToMouse();
47         test.init();
48         test.start();
49     }
50 
51     public void init() {
52         frame.setFocusableWindowState(false);
53         frame.setBounds(100, 100, 100, 100);
54     }
55 
56     public void start() {
57         frame.setVisible(true);
58         Util.waitTillShown(frame);
59 
60         Point loc = frame.getLocationOnScreen();
61         robot.mouseMove(loc.x + frame.getWidth() / 4, loc.y + frame.getInsets().top / 2);
62         robot.delay(50);
63         robot.mousePress(InputEvent.BUTTON1_MASK);
64         robot.delay(50);
65         robot.mouseMove(loc.x + 100, loc.y + 50);
66         robot.delay(50);
67         robot.mouseRelease(InputEvent.BUTTON1_MASK);
68 
69         Util.waitForIdle(robot);
70 
71         loc = frame.getLocation();
72         robot.mouseMove(loc.x + frame.getWidth() / 2, loc.y + frame.getHeight() / 2);
73         Util.waitForIdle(robot);
74 
75         if (!(frame.getLocation().equals(loc))) {
76             throw new RuntimeException("Test failed: frame is moving to mouse with grab!");
77         }
78         System.out.println("Test passed.");
79     }
80 }
