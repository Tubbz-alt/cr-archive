 24 import java.awt.Button;
 25 import java.awt.Choice;
 26 import java.awt.FlowLayout;
 27 import java.awt.Frame;
 28 import java.awt.Point;
 29 import java.awt.Robot;
 30 import java.awt.Window;
 31 import java.awt.event.FocusAdapter;
 32 import java.awt.event.FocusEvent;
 33 import java.awt.event.InputEvent;
 34 import java.awt.event.KeyEvent;
 35 import java.util.concurrent.CountDownLatch;
 36 import java.util.concurrent.TimeUnit;
 37 
 38 /**
 39  * @test
 40  * @bug 4478780
 41  * @key headful
 42  * @summary Tests that Choice can be accessed and controlled by keyboard.
 43  */
 44 public class AccessibleChoiceTest {
 45     //Declare things used in the test, like buttons and labels here
 46     Frame frame = new Frame("window owner");
 47     Window win = new Window(frame);
 48     Choice choice = new Choice();
 49     Button def = new Button("default owner");
 50     CountDownLatch go = new CountDownLatch(1);
 51 
 52     public static void main(final String[] args) {
 53         AccessibleChoiceTest app = new AccessibleChoiceTest();
 54         app.test();
 55     }
 56 
 57     private void test() {
 58         try {
 59             init();
 60             start();
 61         } finally {
 62             if (frame != null) frame.dispose();
 63             if (win != null) win.dispose();
 64         }
 65     }
 66 
 67     public void init() {
 68         win.setLayout (new FlowLayout ());
 69         win.add(def);
 70         def.addFocusListener(new FocusAdapter() {
 71                 public void focusGained(FocusEvent e) {
 72                     go.countDown();
 73                 }
 74             });
 75         choice.add("One");
 76         choice.add("Two");
 77         win.add(choice);
 78     }
 79 
 80     public void start () {
 81         frame.setVisible(true);
 82         win.pack();
 83         win.setLocation(100, 200);
 84         win.setVisible(true);
 85 
 86         Robot robot = null;
 87         try {
 88             robot = new Robot();
 89         } catch (Exception ex) {
 90             throw new RuntimeException("Can't create robot");
 91         }
 92         robot.delay(2000);
 93         robot.setAutoDelay(150);
 94         robot.setAutoWaitForIdle(true);
 95 
 96         // Focus default button and wait till it gets focus
 97         Point loc = def.getLocationOnScreen();
 98         robot.mouseMove(loc.x+2, loc.y+2);
 99         robot.mousePress(InputEvent.BUTTON1_MASK);
100         robot.mouseRelease(InputEvent.BUTTON1_MASK);
101 
102         try {
103             go.await(1, TimeUnit.SECONDS);
104         } catch (InterruptedException ie) {
105             throw new RuntimeException("Interrupted !!!");
106         }
107 
108         if (!def.isFocusOwner()) {
109             throw new RuntimeException("Button doesn't have focus");
110         }
111 
112         // Press Tab key to move focus to Choice
113         robot.keyPress(KeyEvent.VK_TAB);
114         robot.keyRelease(KeyEvent.VK_TAB);
115 
116         robot.delay(500);
117 
118         // Press Down key to select next item in the choice(Motif 2.1)
119         // If bug exists we won't be able to do so
120         robot.keyPress(KeyEvent.VK_DOWN);
121         robot.keyRelease(KeyEvent.VK_DOWN);
122 
123         String osName = System.getProperty("os.name").toLowerCase();
124         if (osName.startsWith("mac")) {
125             robot.keyPress(KeyEvent.VK_DOWN);
126             robot.keyRelease(KeyEvent.VK_DOWN);
127             robot.keyPress(KeyEvent.VK_ENTER);
128             robot.keyRelease(KeyEvent.VK_ENTER);
129         }
130 
131         robot.delay(1000);
132 
133         // On success second item should be selected
134         if (choice.getSelectedItem() != choice.getItem(1)) {
135             throw new RuntimeException("Choice can't be controlled by keyboard");
136         }
137     }
138 }
