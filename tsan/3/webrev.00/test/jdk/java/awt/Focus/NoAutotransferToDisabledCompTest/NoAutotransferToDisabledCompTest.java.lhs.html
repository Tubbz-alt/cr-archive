 24 /*
 25   @test
 26   @key headful
 27   @bug       4685768
 28   @summary   Tests that auto-transfering focus doesn't stuck on a disabled component.
 29   @author    Anton Tarasov: area=awt.focus
 30   @library   ../../regtesthelpers
 31   @build     Util
 32   @run       main NoAutotransferToDisabledCompTest
 33 */
 34 
 35 import java.awt.Robot;
 36 import javax.swing.*;
 37 import java.awt.*;
 38 import java.awt.event.*;
 39 import java.applet.Applet;
 40 import test.java.awt.regtesthelpers.Util;
 41 
 42 public class NoAutotransferToDisabledCompTest extends Applet {
 43     Robot robot;
 44     JFrame frame = new JFrame("Frame");
 45     JButton b0 = new JButton("b0");
 46     JButton b1 = new JButton("b1");
 47     JButton b2 = new JButton("b2");
 48 
 49     public static void main(String[] args) {
 50         NoAutotransferToDisabledCompTest app = new NoAutotransferToDisabledCompTest();
 51         app.init();
 52         app.start();
 53     }
 54 
 55     public void init() {
 56         robot = Util.createRobot();
 57         frame.add(b0);
 58         frame.add(b1);
 59         frame.add(b2);
 60         frame.setLayout(new FlowLayout());
 61         frame.pack();
 62 
 63         b1.addActionListener(new ActionListener() {
 64             public void actionPerformed(ActionEvent e) {
 65                 b1.setEnabled(false);
 66                 b2.setEnabled(false);
 67             }
 68         });
 69     }
 70 
 71     public void start() {
 72         Util.showWindowWait(frame);
 73 
 74         // Request focus on b1.
 75         if (!Util.focusComponent(b1, 2000)) {
 76             throw new TestErrorException("couldn't focus " + b1);
 77         }
 78 
 79         // Activate b1.
 80         robot.keyPress(KeyEvent.VK_SPACE);
 81         robot.delay(50);
 82         robot.keyRelease(KeyEvent.VK_SPACE);
 83         Util.waitForIdle(robot);

 84 
 85         // Check that focus has been transfered to b0.
 86         if (!b0.hasFocus()) {
 87             throw new TestFailedException("focus wasn't auto-transfered properly!");
 88         }
 89         System.out.println("Test passed.");
 90     }
 91 }
 92

 94  * Thrown when the behavior being verified is found wrong.
 95  */

 97     TestFailedException(String msg) {
 98         super("Test failed: " + msg);
 99     }
100 }

103  * Thrown when an error not related to the behavior being verified is encountered.
104  */
105 class TestErrorException extends RuntimeException {
106     TestErrorException(String msg) {
107         super("Unexpected error: " + msg);
108     }
109 }
