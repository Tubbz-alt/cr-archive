 24 /*
 25   test
 26   @bug       6480534
 27   @summary   A Frame changing its state from ICONIFIED to NORMAL should regain focus.
 28   @author    anton.tarasov@...: area=awt.focus
 29   @run       applet DeiconifiedFrameLoosesFocus.html
 30 */
 31 
 32 import java.awt.*;
 33 import java.applet.Applet;
 34 import test.java.awt.regtesthelpers.Util;
 35 
 36 public class DeiconifiedFrameLoosesFocus extends Applet {
 37     Robot robot;
 38     static final Frame frame = new Frame("Frame");
 39 
 40     public static void main(String[] args) {
 41         DeiconifiedFrameLoosesFocus app = new DeiconifiedFrameLoosesFocus();
 42         app.init();
 43         app.start();
 44     }
 45 
 46     public void init() {
 47         robot = Util.createRobot();
 48 
 49         // Create instructions for the user here, as well as set up
 50         // the environment -- set the layout manager, add buttons,
 51         // etc.
 52         this.setLayout (new BorderLayout ());
 53     }
 54 
 55     public void start() {
 56         if (!Toolkit.getDefaultToolkit().isFrameStateSupported(Frame.ICONIFIED) ||
 57             !Toolkit.getDefaultToolkit().isFrameStateSupported(Frame.NORMAL))
 58         {
 59             System.out.println("Frame.ICONIFIED or Frame.NORMAL state is unsupported.");
 60             return;
 61         }
 62 
 63         frame.setSize(100, 100);
 64 
 65         frame.setVisible(true);
 66 
 67         Util.waitForIdle(robot);
 68 
 69         if (!frame.isFocused()) {
 70             Util.clickOnTitle(frame, robot);
 71             Util.waitForIdle(robot);
 72         }
 73 
 74         if (!frame.isFocused()) {
 75             throw new Error("Test error: couldn't focus the Frame.");
 76         }
 77 
 78         test();
 79         System.out.println("Test passed.");
 80     }
 81 
 82     void test() {
 83         frame.setExtendedState(Frame.ICONIFIED);
 84 
 85         Util.waitForIdle(robot);
 86 
 87         frame.setExtendedState(Frame.NORMAL);
 88 
 89         Util.waitForIdle(robot);
 90 
 91         if (!frame.isFocused()) {
 92             throw new TestFailedException("the Frame didn't regain focus after restoring!");
 93         }
 94     }
 95 }
 96 
 97 class TestFailedException extends RuntimeException {
 98     TestFailedException(String msg) {
 99         super("Test failed: " + msg);
100     }
101 }
