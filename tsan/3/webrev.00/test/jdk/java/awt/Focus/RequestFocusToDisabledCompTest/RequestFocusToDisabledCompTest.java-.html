24 /*
25   @test
26   @key headful
27   @bug       4685768
28   @summary   Tests that it's possible to manually request focus on a disabled component.
29   @author    Anton Tarasov: area=awt.focus
30   @library   ../../regtesthelpers
31   @build     Util
32   @run       main RequestFocusToDisabledCompTest
33 */
34 
35 import java.awt.Robot;
36 import javax.swing.*;
37 import java.awt.*;
38 import java.awt.event.*;
39 import java.applet.Applet;
40 import test.java.awt.regtesthelpers.Util;
41 
42 public class RequestFocusToDisabledCompTest extends Applet {
43     Robot robot;
44     JFrame frame = new JFrame("Frame");
45     JButton b0 = new JButton("b0");
46     JButton b1 = new JButton("b1");
47 
48     public static void main(String[] args) {
49         RequestFocusToDisabledCompTest app = new RequestFocusToDisabledCompTest();
50         app.init();
51         app.start();
52     }
53 
54     public void init() {
55         robot = Util.createRobot();
56         frame.add(b0);
57         frame.add(b1);
58         frame.setLayout(new FlowLayout());
59         frame.pack();
60 
61         b1.setEnabled(false);
62     }
63 
64     public void start() {
65         Util.showWindowWait(frame);
66 
67         if (!b0.hasFocus()) {
68             // Request focus on b0.
69             if (!Util.focusComponent(b0, 2000)) {
70                 throw new TestErrorException("couldn't focus " + b0);
71             }
72         }
73 
74         // Try to request focus on b1.
75         if (!Util.focusComponent(b1, 2000)) {
76             throw new TestFailedException("focus wasn't requested on disabled " + b1);
77         }
78         System.out.println("Test passed.");
79     }
80 }
81 
82 /**
83  * Thrown when the behavior being verified is found wrong.
84  */
85 class TestFailedException extends RuntimeException {
86     TestFailedException(String msg) {
87         super("Test failed: " + msg);
88     }
89 }
90 
91 /**
92  * Thrown when an error not related to the behavior being verified is encountered.
93  */
94 class TestErrorException extends RuntimeException {
95     TestErrorException(String msg) {
96         super("Unexpected error: " + msg);
97     }
98 }
