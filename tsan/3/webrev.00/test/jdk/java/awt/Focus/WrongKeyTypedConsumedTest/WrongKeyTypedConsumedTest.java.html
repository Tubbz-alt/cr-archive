 24 /*
 25   @test
 26   @key headful
 27   @bug      4782886
 28   @summary  FocusManager consumes wrong KEY_TYPED events
 29   @library  ../../regtesthelpers
 30   @build    Util
 31   @run      main WrongKeyTypedConsumedTest
 32 */
 33 
 34 import java.awt.AWTKeyStroke;
 35 import java.awt.BorderLayout;
 36 import java.awt.KeyboardFocusManager;
 37 import java.awt.Robot;
 38 
 39 import java.awt.event.KeyEvent;
 40 
 41 import java.util.HashSet;
 42 import java.util.Set;
 43 
 44 import javax.swing.JCheckBox;
 45 import javax.swing.JFrame;
 46 import javax.swing.JTextArea;
 47 
 48 import test.java.awt.regtesthelpers.Util;
 49 
 50 public class WrongKeyTypedConsumedTest
 51 {
 52     Robot robot = Util.createRobot();
 53 
 54     public static void main(String[] args) {
 55         WrongKeyTypedConsumedTest test = new WrongKeyTypedConsumedTest();
 56         test.start();
 57     }
 58 
 59     public void start ()
 60     {
 61         JFrame frame = new JFrame("The Frame");
 62         Set ftk = new HashSet();
 63         ftk.add(AWTKeyStroke.getAWTKeyStroke(KeyEvent.VK_DOWN, 0));
 64         frame.getContentPane().
 65             setFocusTraversalKeys(KeyboardFocusManager.FORWARD_TRAVERSAL_KEYS,
 66                                   ftk);
 67 
 68         JCheckBox checkbox = new JCheckBox("test");
 69         frame.getContentPane().add(checkbox, BorderLayout.NORTH);
 70 
 71         JTextArea textarea = new JTextArea(40, 10);
 72         frame.getContentPane().add(textarea);
 73 
 74         frame.pack();
 75         frame.setLocationRelativeTo(null);
 76         frame.setVisible(true);
 77         Util.waitForIdle(robot);
 78 
 79         if (!frame.isActive()) {
 80             throw new RuntimeException("Test Fialed: frame isn't active");
 81         }
 82 
 83         // verify if checkbox has focus
 84         if (!checkbox.isFocusOwner()) {
 85             checkbox.requestFocusInWindow();
 86             Util.waitForIdle(robot);
 87             if (!checkbox.isFocusOwner()) {
 88                 throw new RuntimeException("Test Failed: checkbox doesn't have focus");
 89             }
 90         }
 91 
 92         // press VK_DOWN
 93         robot.keyPress(KeyEvent.VK_DOWN);
 94         robot.delay(50);
 95         robot.keyRelease(KeyEvent.VK_DOWN);
 96         robot.delay(50);
 97 
 98         Util.waitForIdle(robot);
 99 
100         // verify if text area has focus
101         if (!textarea.isFocusOwner()) {
102             throw new RuntimeException("Test Failed: focus wasn't transfered to text area");
103         }
104         // press '1'
105         robot.keyPress(KeyEvent.VK_1);
106         robot.delay(50);
107         robot.keyRelease(KeyEvent.VK_1);
108         robot.delay(50);
109 
110         Util.waitForIdle(robot);
111 
112         // verify if KEY_TYPED arrived
113         if (!"1".equals(textarea.getText())) {
114             throw new RuntimeException("Test Failed: text area text is \"" + textarea.getText() + "\", not \"1\"");
115         }
116         System.out.println("Test Passed");
117     }
118 }
