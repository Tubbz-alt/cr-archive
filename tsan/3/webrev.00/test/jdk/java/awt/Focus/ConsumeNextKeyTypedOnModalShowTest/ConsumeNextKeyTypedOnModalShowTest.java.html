 24 /*
 25   @test
 26   @key headful
 27   @bug       6637607
 28   @summary   Showing a modal dlg on TAB KEY_PRESS shouldn't consume inappropriate KEY_TYPED.
 29   @library   ../../regtesthelpers
 30   @build     Util
 31   @run       main ConsumeNextKeyTypedOnModalShowTest
 32 */
 33 
 34 import java.awt.*;
 35 import java.awt.event.*;
 36 import test.java.awt.regtesthelpers.Util;
 37 
 38 public class ConsumeNextKeyTypedOnModalShowTest {
 39     Robot robot;
 40     Frame frame = new Frame("Frame");
 41     Dialog dialog = new Dialog(frame, "Dialog", true);
 42     TextField tf0 = new TextField();
 43     TextField tf1 = new TextField();
 44     Button button = new Button("Button");
 45 
 46     public static void main(String[] args) {
 47         ConsumeNextKeyTypedOnModalShowTest app = new ConsumeNextKeyTypedOnModalShowTest();
 48         app.init();
 49         app.start();
 50     }
 51 
 52     public void init() {
 53         robot = Util.createRobot();
 54 
 55         tf0.setPreferredSize(new Dimension(50, 30));
 56         tf1.setPreferredSize(new Dimension(50, 30));
 57         frame.setLayout(new FlowLayout());
 58         frame.add(tf0);
 59         frame.add(tf1);
 60         frame.pack();
 61 
 62         dialog.add(button);
 63         dialog.pack();
 64 
 65         Toolkit.getDefaultToolkit().addAWTEventListener(new AWTEventListener() {
 66             public void eventDispatched(AWTEvent e) {
 67                 if (e.getID() == KeyEvent.KEY_PRESSED && e.getSource() == tf0) {
 68                     dialog.setVisible(true);
 69                 }
 70             }
 71         }, KeyEvent.KEY_EVENT_MASK);
 72     }
 73 
 74     public void start() {
 75         frame.setLocationRelativeTo(null);
 76         frame.setVisible(true);
 77         Util.waitTillShown(frame);
 78 
 79         // Show the dialog.
 80         robot.keyPress(KeyEvent.VK_TAB);
 81         robot.delay(50);
 82         robot.keyRelease(KeyEvent.VK_TAB);
 83 
 84         Util.waitForIdle(robot);
 85 
 86         // Dispose the dialog.
 87         Runnable action = new Runnable() {
 88             public void run() {
 89                 dialog.dispose();
 90             }
 91         };
 92         if (!Util.trackFocusGained(tf1, action, 2000, false)) {
 93             throw new RuntimeException("Test failed: TAB was processed incorrectly!");
 94         }
 95 
 96         // Check for type-ability.
 97         robot.keyPress(KeyEvent.VK_A);
 98         robot.delay(50);
 99         robot.keyRelease(KeyEvent.VK_A);
100 
101         Util.waitForIdle(robot);
102 
103         if (tf1.getText().equals("")) {
104             throw new RuntimeException("Test failed: couldn't type a char!");
105         }
106         System.out.println("Test passed.");
107     }
108 }
