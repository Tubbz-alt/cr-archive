24 /*
25   @test
26   @key headful
27   @bug        6480534
28   @summary    A Frame changing its state from ICONIFIED to NORMAL should regain focus.
29   @library    ../../regtesthelpers
30   @build      Util
31   @run        main DeiconifiedFrameLoosesFocus
32 */
33 
34 import java.awt.*;
35 import test.java.awt.regtesthelpers.Util;
36 
37 public class DeiconifiedFrameLoosesFocus {
38     Robot robot;
39     static final Frame frame = new Frame("Frame");
40 
41     public static void main(String[] args) {
42         DeiconifiedFrameLoosesFocus app = new DeiconifiedFrameLoosesFocus();
43         app.init();
44         app.start();
45     }
46 
47     public void init() {
48         robot = Util.createRobot();
49     }
50 
51     public void start() {
52         if (!Toolkit.getDefaultToolkit().isFrameStateSupported(Frame.ICONIFIED) ||
53             !Toolkit.getDefaultToolkit().isFrameStateSupported(Frame.NORMAL))
54         {
55             System.out.println("Frame.ICONIFIED or Frame.NORMAL state is unsupported.");
56             return;
57         }
58 
59         frame.setSize(100, 100);
60 
61         frame.setVisible(true);
62 
63         Util.waitForIdle(robot);
64 
65         if (!frame.isFocused()) {
66             Util.clickOnTitle(frame, robot);
67             Util.waitForIdle(robot);
68         }
69 
70         if (!frame.isFocused()) {
71             throw new Error("Test error: couldn't focus the Frame.");
72         }
73 
74         test();
75         System.out.println("Test passed.");
76     }
77 
78     void test() {
79         frame.setExtendedState(Frame.ICONIFIED);
80 
81         Util.waitForIdle(robot);
82 
83         frame.setExtendedState(Frame.NORMAL);
84 
85         Util.waitForIdle(robot);
86 
87         if (!frame.isFocused()) {
88             throw new TestFailedException("the Frame didn't regain focus after restoring!");
89         }
90     }
91 }
92 
93 class TestFailedException extends RuntimeException {
94     TestFailedException(String msg) {
95         super("Test failed: " + msg);
96     }
97 }
