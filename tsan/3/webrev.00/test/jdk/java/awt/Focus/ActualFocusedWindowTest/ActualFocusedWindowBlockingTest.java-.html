 24 /*
 25   @test
 26   @key headful
 27   @bug       6314575
 28   @summary   Tests that previosly focused owned window doesn't steal focus when an owner's component requests focus.
 29   @author    Anton.Tarasov: area=awt.focus
 30   @library   ../../regtesthelpers
 31   @build     Util
 32   @run       main ActualFocusedWindowBlockingTest
 33 */
 34 
 35 import java.awt.*;
 36 import java.awt.event.*;
 37 import java.applet.Applet;
 38 import java.util.concurrent.atomic.AtomicBoolean;
 39 import java.lang.reflect.InvocationTargetException;
 40 import test.java.awt.regtesthelpers.Util;
 41 
 42 public class ActualFocusedWindowBlockingTest extends Applet {
 43     Robot robot = Util.createRobot();
 44     Frame owner = new Frame("Owner Frame");
 45     Window win = new Window(owner);
 46     Frame frame = new Frame("Auxiliary Frame");
 47     Button fButton = new Button("frame button") {public String toString() {return "Frame_Button";}};
 48     Button wButton = new Button("window button") {public String toString() {return "Window_Button";}};
 49     Button aButton = new Button("auxiliary button") {public String toString() {return "Auxiliary_Button";}};
 50 
 51     public static void main(String[] args) {
 52         ActualFocusedWindowBlockingTest app = new ActualFocusedWindowBlockingTest();
 53         app.init();
 54         app.start();
 55     }
 56 
 57     public void init() {
 58         Toolkit.getDefaultToolkit().addAWTEventListener(new AWTEventListener() {
 59                 public void eventDispatched(AWTEvent e) {
 60                     System.out.println("--> " + e);
 61                 }
 62             }, FocusEvent.FOCUS_EVENT_MASK | WindowEvent.WINDOW_FOCUS_EVENT_MASK);
 63 
 64         owner.add(fButton);
 65         win.add(wButton);
 66         frame.add(aButton);
 67 
 68         owner.setName("OWNER_FRAME");
 69         win.setName("OWNED_WINDOW");
 70         frame.setName("AUX_FRAME");
 71 
 72         tuneAndShowWindows(new Window[] {owner, win, frame});
 73     }
 74 
 75     public void start() {
 76         if ("sun.awt.motif.MToolkit".equals(Toolkit.getDefaultToolkit().getClass().getName())) {
 77             System.out.println("No testing on Motif. Test passed.");
 78             return;
 79         }
 80 
 81         System.out.println("\nTest started:\n");
 82 
 83         // Test 1.
 84 
 85         clickOnCheckFocus(wButton);
 86         clickOnCheckFocus(aButton);
 87 
 88         Util.clickOnComp(fButton, robot);
 89         if (!testFocused(fButton)) {
 90             throw new TestFailedException("The owner's component [" + fButton + "] couldn't be focused by click");
 91         }
 92 
 93         // Test 2.
 94 
 95         clickOnCheckFocus(wButton);
 96         clickOnCheckFocus(aButton);
 97 
 98         fButton.requestFocus();
 99         Util.waitForIdle(robot);
100         if (!testFocused(fButton)) {
101             throw new TestFailedException("The owner's component [" + fButton + "] couldn't be focused by request");
102         }
103 
104         // Test 3.
105 
106         clickOnCheckFocus(wButton);
107         clickOnCheckFocus(aButton);
108         clickOnCheckFocus(fButton);
109         clickOnCheckFocus(aButton);
110 
111         Util.clickOnTitle(owner, robot);
112         if (!testFocused(fButton)) {
113             throw new TestFailedException("The owner's component [" + fButton + "] couldn't be focused as the most recent focus owner");
114         }
115 
116         System.out.println("Test passed.");
117     }
118 
119     void tuneAndShowWindows(Window[] arr) {
120         int y = 0;
121         for (Window w: arr) {
122             w.setLayout(new FlowLayout());
123             w.setBounds(100, y, 400, 150);
124             w.setBackground(Color.blue);
125             w.setVisible(true);
126             y += 200;
127             Util.waitForIdle(robot);
128         }
129     }
130 
131     void clickOnCheckFocus(Component c) {
132         if (c instanceof Frame) {
133             Util.clickOnTitle((Frame)c, robot);
134         } else {
135             Util.clickOnComp(c, robot);
136         }
137         if (!testFocused(c)) {
138             throw new TestErrorException(c + "couldn't get focus by click.");
139         }
140     }
141 
142     boolean testFocused(Component c) {
143         for (int i=0; i<10; i++) {
144             if (KeyboardFocusManager.getCurrentKeyboardFocusManager().getFocusOwner() == c) {
145                 return true;
146             }
147             Util.waitForIdle(robot);
148         }
149         return false;
150     }
151 
152     // Thrown when the behavior being verified is found wrong.
153     class TestFailedException extends RuntimeException {
154         TestFailedException(String msg) {
155             super("Test failed: " + msg);
156         }
157     }
158 
159     // Thrown when an error not related to the behavior being verified is encountered.
160     class TestErrorException extends RuntimeException {
161         TestErrorException(String msg) {
162             super("Unexpected error: " + msg);
163         }
164     }
165 }
