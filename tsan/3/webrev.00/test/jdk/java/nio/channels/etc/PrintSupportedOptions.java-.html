24 /**
25  * @test
26  * @requires (os.family == "linux" | os.family == "mac" | os.family == "windows")
27  * @bug 8209152
28  * @run main PrintSupportedOptions
29  * @run main/othervm -Djava.net.preferIPv4Stack=true PrintSupportedOptions
30  */
31 
32 import java.io.IOException;
33 import java.net.SocketOption;
34 import java.nio.channels.*;
35 
36 public class PrintSupportedOptions {
37 
38     @FunctionalInterface
39     interface NetworkChannelSupplier<T extends NetworkChannel> {
40         T get() throws IOException;
41     }
42 
43     public static void main(String[] args) throws IOException {
44         test(() -> SocketChannel.open());
45         test(() -> ServerSocketChannel.open());
46         test(() -> DatagramChannel.open());
47 
48         test(() -> AsynchronousSocketChannel.open());
49         test(() -> AsynchronousServerSocketChannel.open());
50     }
51 
52     @SuppressWarnings("unchecked")
53     static <T extends NetworkChannel>
54     void test(NetworkChannelSupplier<T> supplier) throws IOException {
55         try (T ch = supplier.get()) {
56             System.out.println(ch);
57             for (SocketOption<?> opt : ch.supportedOptions()) {
58                 Object value = ch.getOption(opt);
59                 System.out.format(" %s -> %s%n", opt.name(), value);
60                 if (value != null) {
61                     ch.setOption((SocketOption<Object>) opt, value);
62                 }
63             }
64         }
65     }
66 }
