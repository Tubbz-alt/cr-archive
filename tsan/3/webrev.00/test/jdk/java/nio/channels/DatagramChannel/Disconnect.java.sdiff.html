41         IPSupport.throwSkippedExceptionIfNonOperational();
42 
43         // test with default protocol family
44         try (DatagramChannel dc = DatagramChannel.open()) {
45             test(dc);
46             test(dc);
47         }
48 
49         if (IPSupport.hasIPv4()) {
