 25  * @bug 8236184
 26  * @library /test/lib
 27  * @build jdk.test.lib.NetworkConfiguration
 28  *        jdk.test.lib.net.IPSupport
 29  * @requires (os.family == "linux") | (os.family == "windows")
 30  * @run main Loopback
 31  * @run main/othervm -Djava.net.preferIPv4Stack=true Loopback
 32  * @summary Test the IP_MULTICAST_LOOP option
 33  */
 34 
 35 import java.io.IOException;
 36 import java.net.InetAddress;
 37 import java.net.InetSocketAddress;
 38 import java.net.NetworkInterface;
 39 import java.net.ProtocolFamily;
 40 import java.net.SocketAddress;
 41 import java.net.StandardSocketOptions;
 42 import java.nio.ByteBuffer;
 43 import java.nio.channels.DatagramChannel;
 44 import java.nio.channels.SelectionKey;
 45 import java.nio.channels.Selector;
 46 import java.util.List;
 47 import java.util.stream.Collectors;
 48 import static java.net.StandardProtocolFamily.INET;
 49 import static java.net.StandardProtocolFamily.INET6;
 50 import static java.net.StandardSocketOptions.IP_MULTICAST_IF;
 51 import static java.net.StandardSocketOptions.IP_MULTICAST_LOOP;
 52 
 53 import jdk.test.lib.NetworkConfiguration;
 54 import jdk.test.lib.net.IPSupport;
 55 
 56 public class Loopback {
 57 
 58     static final ProtocolFamily UNSPEC = () -> "UNSPEC";
 59 
 60     public static void main(String[] args) throws Exception {
 61         IPSupport.throwSkippedExceptionIfNonOperational();
 62 
 63         // IPv4 and IPv6 interfaces that support multicasting
 64         NetworkConfiguration config = NetworkConfiguration.probe();
 65         List<NetworkInterface> ip4MulticastInterfaces = config.ip4MulticastInterfaces()
 66                 .collect(Collectors.toList());
 67         List<NetworkInterface> ip6MulticastInterfaces = config.ip6MulticastInterfaces()
 68                 .collect(Collectors.toList());
 69 
 70         // IPv4 multicast group
 71         InetAddress ip4Group = InetAddress.getByName("225.4.5.6");
 72         for (NetworkInterface ni : ip4MulticastInterfaces) {
 73             test(UNSPEC, ip4Group, ni);
 74             test(INET, ip4Group, ni);
 75             if (IPSupport.hasIPv6()) {
 76                 test(INET6, ip4Group, ni);
 77             }
 78         }
 79 
 80         // IPv6 multicast group
 81         InetAddress ip6Group = InetAddress.getByName("ff02::a");
 82         for (NetworkInterface ni : ip6MulticastInterfaces) {
 83             test(UNSPEC, ip6Group, ni);
 84             test(INET6, ip6Group, ni);
 85         }
 86     }
 87 
 88     /**
 89      * Joins a multicast group and send datagrams to that group with both
 90      * IP_MULTICAST_LOOP enabled and disabled.
 91      */
 92     static void test(ProtocolFamily family, InetAddress group, NetworkInterface ni)
 93         throws IOException
 94     {
 95         System.out.format("\n%s socket\n", family.name());
 96         DatagramChannel dc;
 97         if (family == UNSPEC) {
 98             dc = DatagramChannel.open();
 99         } else {
100             dc = DatagramChannel.open(family);
101         }
102         try (dc) {
103             dc.setOption(StandardSocketOptions.SO_REUSEADDR, true);
104             dc.bind(new InetSocketAddress(0));
105             int localPort = dc.socket().getLocalPort();
106             SocketAddress target = new InetSocketAddress(group, localPort);
107 
108             System.out.format("join %s @ %s%n", group.getHostAddress(), ni.getName());
109             dc.join(group, ni);
110 
111             System.out.format("set outgoing multicast interface to %s%n", ni.getName());
112             dc.setOption(IP_MULTICAST_IF, ni);
113 
114             // -- IP_MULTICAST_LOOP enabled --
115 
116             assertTrue(dc.getOption(IP_MULTICAST_LOOP), "IP_MULTICAST_LOOP not enabled");
117             System.out.println("IP_MULTICAST_LOOP enabled");
118 
119             // send datagram to multicast group
120             System.out.format("send %s -> %s%n", dc.getLocalAddress(), target);
121             ByteBuffer src = ByteBuffer.wrap("hello".getBytes("UTF-8"));
122             dc.send(src, target);
123 
124             // receive datagram sent to multicast group
125             ByteBuffer dst = ByteBuffer.allocate(100);
126             int senderPort;
127             do {
128                 dst.clear();
129                 SocketAddress sender = dc.receive(dst);
130                 System.out.format("received %s from %s%n", dst, sender);
131                 senderPort = ((InetSocketAddress) sender).getPort();
132             } while (senderPort != localPort);
133             dst.flip();
134             assertTrue(dst.remaining() == src.capacity(), "Unexpected message size");
135 
136             // -- IP_MULTICAST_LOOP disabled --
137 
138             dc.setOption(StandardSocketOptions.IP_MULTICAST_LOOP, false);
139             System.out.println("IP_MULTICAST_LOOP disabled");
140 
141             // send datagram to multicast group
142             System.out.format("send %s -> %s%n", dc.getLocalAddress(), target);
143             src.clear();
144             dc.send(src, target);
145 
146             // test that we don't receive the datagram sent to multicast group
147             dc.configureBlocking(false);
148             try (Selector sel = Selector.open()) {
149                 dc.register(sel, SelectionKey.OP_READ);
150                 boolean done = false;
151                 while (!done) {
152                     int n = sel.select(3000);
153                     System.out.format("selected %d%n", n);
154                     if (n == 0) {
155                         // timeout, no datagram received
156                         done = true;
157                     } else {
158                         sel.selectedKeys().clear();
159                         SocketAddress sender = dc.receive(dst);
160                         if (sender != null) {
161                             System.out.format("received %s from %s%n", dst, sender);
162                             senderPort = ((InetSocketAddress) sender).getPort();
163                             assertTrue(senderPort != localPort, "Unexpected message");
164                         }
165                     }
166                 }
167             }
168         }
169     }
170 
171     static void assertTrue(boolean e, String msg) {
172         if (!e) throw new RuntimeException(msg);
173     }
174 }
