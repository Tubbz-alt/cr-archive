24 /**
25  * @test
26  * @bug 8216472
27  * @summary native call in WindowsSelectorImpl.SubSelector.poll can use
28  *     more stack space than available in a shadow zone, this can cause
29  *     a crash if selector is called from a deep recursive java call
30  * @requires (os.family == "windows")
31  */
32 
33 import java.nio.channels.Selector;
34 
35 public class StackOverflowTest {
36 
37     public static void main(String[] args) throws Exception {
38         try (var sel = Selector.open()) {
39             recursiveSelect(sel);
40         } catch (StackOverflowError e) {
41             // ignore SOE from java calls
42         }
43     }
44 
45     static void recursiveSelect(Selector sel) throws Exception {
46         sel.selectNow();
47         recursiveSelect(sel);
48     }
49 }
