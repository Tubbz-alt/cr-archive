24 /**
25  * A simplified Unix domain socket which can read and write bytes at a time
26  * used for simulating external launchers which use UNIX sockets to talk
27  * the VM.
28  */
29 
30 import java.io.IOException;
31 
32 public class UnixDomainSocket {
33 
34     static {
35         System.loadLibrary("InheritedChannel");
36         init();
37     }
38 
39     private final int fd;
40 
41     public UnixDomainSocket(int fd) {
42         this.fd = fd;
43     }
44 
45     public int read() throws IOException {
46         return read0(fd);
47     }
48 
49     public void write(int w) throws IOException {
50         write0(fd, w);
51     }
52 
53     public void close() {
54         close0(fd);
55     }
56 
57     public int fd() {
58         return fd;
59     }
60 
61     public String toString() {
62         return "UnixDomainSocket: fd=" + Integer.toString(fd);
63     }
64 
65     /* read and write bytes with UNIX domain sockets */
66 
67     private static native int read0(int fd) throws IOException;
68     private static native void write0(int fd, int w) throws IOException;
69     private static native void close0(int fd);
70     private static native void init();
71     public static native UnixDomainSocket[] socketpair();
72 }
