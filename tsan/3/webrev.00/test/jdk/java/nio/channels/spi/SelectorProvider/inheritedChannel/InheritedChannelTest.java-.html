 41 import java.nio.file.Files;
 42 import java.nio.file.Path;
 43 import java.nio.file.Paths;
 44 import java.util.ArrayList;
 45 import java.util.List;
 46 import java.util.Map;
 47 
 48 import jdk.test.lib.JDKToolFinder;
 49 import jdk.test.lib.Utils;
 50 import jdk.test.lib.process.ProcessTools;
 51 import jdk.test.lib.Platform;
 52 
 53 import org.testng.annotations.DataProvider;
 54 import org.testng.annotations.Test;
 55 
 56 import static java.util.Arrays.asList;
 57 
 58 public class InheritedChannelTest {
 59 
 60     private static final String TEST_SRC = System.getProperty("test.src");
 61     private static final String TEST_CLASSES = System.getProperty("test.class.path");
 62     private static final Path POLICY_PASS = Paths.get(TEST_SRC, "java.policy.pass");
 63     private static final Path POLICY_FAIL = Paths.get(TEST_SRC, "java.policy.fail");
 64 
 65     private static final String OS = System.getProperty("os.name").toLowerCase();
 66     private static final String OS_NAME = OS.startsWith("sunos") ? "solaris" : OS;
 67 
 68     private static final String ARCH = System.getProperty("os.arch");
 69     private static final String OS_ARCH = ARCH.equals("i386") ? "i586" : ARCH;
 70 
 71     private static final Path libraryPath
 72             = Paths.get(System.getProperty("java.library.path"));
 73 
 74     @DataProvider
 75     public Object[][] testCases() {
 76         return new Object[][]{
 77             { "UnixSocketTest", List.of(UnixSocketTest.class.getName())},
 78             { "StateTest", List.of(StateTest.class.getName()) },
 79             { "EchoTest",  List.of(EchoTest.class.getName())  },
 80             { "CloseTest", List.of(CloseTest.class.getName()) },
 81 
 82             // run StateTest with a SecurityManager set
 83             // Note that the system properties are arguments to StateTest and not options.
 84             // These system properties are passed to the launched service as options:
 85             // java [-options] class [args...]
 86             { "StateTest run with " + POLICY_PASS, List.of(StateTest.class.getName(),
 87                                                            "-Djava.security.manager",
 88                                                            "-Djava.security.policy="
 89                                                            + POLICY_PASS)
 90             },
 91             { "StateTest run with " + POLICY_FAIL, List.of(StateTest.class.getName(),
 92                                                            "-expectFail",
 93                                                            "-Djava.security.manager",
 94                                                            "-Djava.security.policy="
 95                                                            + POLICY_FAIL)
 96             }
 97         };
 98     }
 99 
100     @Test(dataProvider = "testCases")
101     public void test(String desc, List<String> opts) throws Throwable {
102         String pathVar = Platform.sharedLibraryPathVariableName();
103         System.out.println(pathVar + "=" + libraryPath);
104 
105         List<String> args = new ArrayList<>();
106         args.add(JDKToolFinder.getJDKTool("java"));
107         args.addAll(asList(Utils.getTestJavaOpts()));
108         args.addAll(List.of("--add-opens", "java.base/java.io=ALL-UNNAMED",
109                             "--add-opens", "java.base/sun.nio.ch=ALL-UNNAMED"));
110         args.addAll(opts);
111 
112         ProcessBuilder pb = new ProcessBuilder(args);
113 
114         Map<String, String> env = pb.environment();
115         env.put("CLASSPATH", TEST_CLASSES);
116         env.put(pathVar, libraryPath.toString());
117 
118         ProcessTools.executeCommand(pb).shouldHaveExitValue(0);
119     }
120 }
