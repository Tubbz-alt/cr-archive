24 /* @test
25  * @bug 7161881

26  * @run main/othervm -Djava.net.preferIPv6Addresses=true BindNull
27  * @summary Make sure the bind method uses an ipv4 address for the null case
28  *          when the DatagramChannel is connected to an IPv4 socket and
29  *          -Djava.net.preferIPv6Addresses=true.
30  */
31 
32 import java.io.*;
33 import java.net.*;
34 import java.nio.channels.*;

35 
36 public class BindNull {
37     public static void main(String[] args) throws IOException {
38         try (DatagramChannel dc = DatagramChannel.open()) {
39             dc.bind(null);
40         }
41         try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET)) {
42             dc.bind(null);


43         }
44         try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET6)) {
45             dc.bind(null);
46         } catch (UnsupportedOperationException uoe) {
47             // IPv6 not available
48         }
49     }
50 }
