24 /* @test
25  * @bug 6435300
26  * @summary Check using IPv6 address does not crash the VM
27  * @run main/othervm UseDGWithIPv6
28  * @run main/othervm -Djava.net.preferIPv4Stack=true UseDGWithIPv6
29  */
30 
31 import java.io.IOException;
32 import java.net.InetSocketAddress;
33 import java.net.SocketAddress;
34 import java.nio.ByteBuffer;
35 import java.nio.channels.DatagramChannel;
36 import java.nio.channels.UnsupportedAddressTypeException;
37 
38 public class UseDGWithIPv6 {
39     static String[] targets = {
40         "3ffe:e00:811:b::21:5",
41         "15.70.186.80"
42     };
43     static int BUFFER_LEN = 10;
44     static int port = 12345;
45 
46     public static void main(String[] args) throws IOException
47     {
48         ByteBuffer data = ByteBuffer.wrap("TESTING DATA".getBytes());
49         DatagramChannel dgChannel = DatagramChannel.open();
50 
51         for(int i = 0; i < targets.length; i++){
52             data.rewind();
53             SocketAddress sa = new InetSocketAddress(targets[i], port);
54             System.out.println("-------------\nDG_Sending data:" +
55                                "\n    remaining:" + data.remaining() +
56                                "\n     position:" + data.position() +
57                                "\n        limit:" + data.limit() +
58                                "\n     capacity:" + data.capacity() +
59                                " bytes on DG channel to " + sa);
60             try {
61                 int n = dgChannel.send(data, sa);
62                 System.out.println("DG_Sent " + n + " bytes");
63             } catch (UnsupportedAddressTypeException e) {
64                 System.out.println("Ignoring unsupported address type");
65             } catch (IOException e) {
66                 //This regression test is to check vm crash only, so ioe is OK.
67                 e.printStackTrace();
68             }
69         }
70         dgChannel.close();
71     }
72 }
