 28 import java.net.InetAddress;
 29 import java.net.InetSocketAddress;
 30 import java.nio.channels.SelectionKey;
 31 import java.nio.channels.Selector;
 32 import java.nio.channels.ServerSocketChannel;
 33 import java.nio.channels.SocketChannel;
 34 
 35 /*
 36  * @test
 37  * @bug 6429204 8203766 8205641
 38  * @summary SelectionKey.interestOps does not update interest set on Windows.
 39  * @author Frank Ding
 40  * @run main/timeout=1200 RacyDeregister
 41  */
 42 public class RacyDeregister {
 43 
 44     // 90% of 1200 second timeout as milliseconds
 45     static final int TIMEOUT_THRESHOLD_MILLIS = 1200*900;
 46 
 47     // Time at start of main().
 48     static long t0;
 49 
 50     static boolean notified;
 51     static final Object selectorLock = new Object();
 52     static final Object notifyLock = new Object();
 53     /**
 54      * null: not terminated
 55      * true: passed
 56      * false: failed
 57      */
 58     static volatile Boolean succTermination = null;
 59 
 60     public static void main(String[] args) throws Exception {
 61         t0 = System.currentTimeMillis();
 62 
 63         InetAddress addr = InetAddress.getByName(null);
 64         ServerSocketChannel sc = ServerSocketChannel.open();
 65         sc.socket().bind(new InetSocketAddress(addr, 0));
 66 
 67         SocketChannel.open(new InetSocketAddress(addr,
 68                 sc.socket().getLocalPort()));
 69 
 70         SocketChannel accepted = sc.accept();
 71         accepted.configureBlocking(false);
 72 
 73         SocketChannel.open(new InetSocketAddress(addr,
 74                 sc.socket().getLocalPort()));
 75         SocketChannel accepted2 = sc.accept();
 76         accepted2.configureBlocking(false);
 77 
 78         final Selector sel = Selector.open();
 79         SelectionKey key2 = accepted2.register(sel, SelectionKey.OP_READ);
 80         final SelectionKey[] key = new SelectionKey[]{
 81             accepted.register(sel, SelectionKey.OP_READ)};
 82 
 83         // thread that will be changing key[0].interestOps to OP_READ | OP_WRITE
 84         new Thread() {
 85 
 86             public void run() {
 87                 try {
 88                     for (int k = 0; k < 15; k++) {
 89                         System.out.format("outer loop %3d at %7d ms%n", k,
 90                             System.currentTimeMillis() - t0);
 91                         System.out.flush();
 92                         for (int i = 0; i < 10000; i++) {
 93                             synchronized (notifyLock) {
 94                                 synchronized (selectorLock) {
 95                                     sel.wakeup();
 96                                     key[0].interestOps(SelectionKey.OP_READ
 97                                             | SelectionKey.OP_WRITE);
 98                                 }
 99                                 notified = false;
100                                 long beginTime = System.currentTimeMillis();
101                                 while (true) {
102                                     notifyLock.wait(5000);
103                                     if (notified) {
104                                         break;
105                                     }
106                                     long endTime = System.currentTimeMillis();
107                                     if (endTime - beginTime > 5000) {
108                                         for (int j = 0; j < 60; j++) {
109                                             Thread.sleep(1000);
110                                             if (notified) {
111                                                 long t =
112                                                     System.currentTimeMillis();
113                                                 System.err.printf
114                                                     ("Notified after %d ms%n",
115                                                      t - beginTime);
116                                                 System.err.flush();
117                                                 break;
118                                             }
119                                         }
120                                         succTermination = false;
121                                         // wake up main thread doing select()
122                                         sel.wakeup();
123                                         return;
124                                     }
125                                 }
126                             }
127                         }
128                         long t = System.currentTimeMillis();
129                         if (t - t0 > TIMEOUT_THRESHOLD_MILLIS) {
130                             System.err.format
131                                 ("Timeout after %d outer loop iterations%n", k);
132                             System.err.flush();
133                             succTermination = false;
134                             // wake up main thread doing select()
135                             sel.wakeup();
136                             return;
137                         }
138                     }
139                     succTermination = true;
140                     // wake up main thread doing select()
141                     sel.wakeup();
142                 } catch (Exception e) {
143                     System.out.println(e);
144                     System.out.flush();
145                     succTermination = true;
146                     // wake up main thread doing select()
147                     sel.wakeup();
148                 }
149             }
150         }.start();
151 
152         // main thread will be doing registering/deregistering with the sel
153         while (true) {
154             sel.select();
155             if (Boolean.TRUE.equals(succTermination)) {
156                 System.out.println("Test passed");
157                 System.out.flush();
158                 sel.close();
159                 sc.close();
160                 break;
161             } else if (Boolean.FALSE.equals(succTermination)) {
162                 System.err.println("Failed to pass the test");
163                 System.err.flush();
164                 sel.close();
165                 sc.close();
166                 throw new RuntimeException("Failed to pass the test");
167             }
168             synchronized (selectorLock) {
169             }
170             if (sel.selectedKeys().contains(key[0]) && key[0].isWritable()) {
171                 synchronized (notifyLock) {
172                     notified = true;
173                     notifyLock.notify();
174                     key[0].cancel();
175                     sel.selectNow();
176                     key2 = accepted2.register(sel, SelectionKey.OP_READ);
177                     key[0] = accepted.register(sel, SelectionKey.OP_READ);
178                 }
179             }
180             key2.cancel();
181             sel.selectedKeys().clear();
182         }
183     }
184 }
