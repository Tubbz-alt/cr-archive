24 /* @test
25  * @bug 7132924
26  * @library /test/lib
27  * @key intermittent
28  * @summary Test DatagramChannel.disconnect when DatagramChannel is connected to an IPv4 socket
29  * @run main Disconnect
30  * @run main/othervm -Djava.net.preferIPv4Stack=true Disconnect
31  */
32 
33 import java.net.*;
34 import java.nio.*;
35 import java.nio.channels.*;
36 import java.io.IOException;
37 import jdk.test.lib.net.IPSupport;
38 
39 public class Disconnect {
40     public static void main(String[] args) throws IOException {
41         IPSupport.throwSkippedExceptionIfNonOperational();
42 
43         // test with default protocol family
44         try (DatagramChannel dc = DatagramChannel.open()) {
45             test(dc);
46             test(dc);
47         }
48 
49         if (IPSupport.hasIPv4()) {
50             // test with IPv4 only
51             try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET)) {
52                 test(dc);
53                 test(dc);
54             }
55         }
56 
57         if (IPSupport.hasIPv6()) {
58             // test with IPv6 only
59             try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET6)) {
60                 test(dc);
61                 test(dc);
62             }
63         }
64     }
65 
66     /**
67      * Connect DatagramChannel to a server, write a datagram and disconnect. Invoke
68      * a second or subsequent time with the same DatagramChannel instance to check
69      * that disconnect works as expected.
70      */
71     static void test(DatagramChannel dc) throws IOException {
72         try (DatagramChannel server = DatagramChannel.open()) {
73             server.bind(new InetSocketAddress(0));
74 
75             InetAddress lh = InetAddress.getLocalHost();
76             dc.connect(new InetSocketAddress(lh, server.socket().getLocalPort()));
77 
78             dc.write(ByteBuffer.wrap("hello".getBytes()));
79 
80             ByteBuffer bb = ByteBuffer.allocate(100);
81             server.receive(bb);
82 
83             dc.disconnect();
84 
85             try {
86                 dc.write(ByteBuffer.wrap("another message".getBytes()));
87                 throw new RuntimeException("write should fail, not connected");
88             } catch (NotYetConnectedException expected) {
89             }
90         }
91     }
92 }
