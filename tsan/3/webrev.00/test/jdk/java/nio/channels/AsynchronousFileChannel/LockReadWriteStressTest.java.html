24 /* @test
25  * @bug 8184157
26  * @summary Ensure that correct PendingFuture is used in Iocp completion status event handler
27  * @requires (os.family == "windows")
28  */
29 
30 import java.nio.ByteBuffer;
31 import java.nio.channels.AsynchronousFileChannel;
32 import java.nio.channels.FileLock;
33 import java.nio.file.Files;
34 import java.nio.file.Path;
35 import java.nio.file.Paths;
36 
37 import static java.nio.file.StandardOpenOption.*;
38 
39 public class LockReadWriteStressTest {
40     public static void main(String[] args) throws Exception {
41         Path path = Path.of("blah");
42         ByteBuffer buf = ByteBuffer.allocate(16);
43         for (int i=0; i < 1000; i++) {
44             try (AsynchronousFileChannel ch = AsynchronousFileChannel.open(path,READ, WRITE, CREATE)) {
45                 FileLock lock = ch.lock().get();
46                 ch.read(buf, 0).get();
47                 buf.rewind();
48                 ch.write(buf, 0).get();
49                 lock.release();
50                 buf.clear();
51             }
52         }
53     }
54 }
