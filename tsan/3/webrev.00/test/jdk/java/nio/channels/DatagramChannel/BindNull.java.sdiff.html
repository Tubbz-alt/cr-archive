24 /* @test
25  * @bug 7161881
26  * @library /test/lib
27  * @run main/othervm -Djava.net.preferIPv6Addresses=true BindNull
28  * @summary Make sure the bind method uses an ipv4 address for the null case
29  *          when the DatagramChannel is connected to an IPv4 socket and
30  *          -Djava.net.preferIPv6Addresses=true.
31  */
32 
33 import java.io.*;
34 import java.net.*;
35 import java.nio.channels.*;
36 import jdk.test.lib.net.IPSupport;
37 
38 public class BindNull {
39     public static void main(String[] args) throws IOException {
40         try (DatagramChannel dc = DatagramChannel.open()) {
41             dc.bind(null);
42         }
43         if (IPSupport.hasIPv4()) {
44             try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET)) {
45                 dc.bind(null);
46             }
47         }
48         if (IPSupport.hasIPv6()) {
49             try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET6)) {
50                 dc.bind(null);
51             }
52         }
53     }
54 }
