24 /* @test
25  * @bug 7132924
26  * @key intermittent
27  * @summary Test DatagramChannel.disconnect when DatagramChannel is connected to an IPv4 socket
28  * @run main Disconnect
29  * @run main/othervm -Djava.net.preferIPv4Stack=true Disconnect
30  */
31 
32 import java.net.*;
33 import java.nio.*;
34 import java.nio.channels.*;
35 import java.io.IOException;
36 
37 public class Disconnect {
38     public static void main(String[] args) throws IOException {
39         // test with default protocol family
40         try (DatagramChannel dc = DatagramChannel.open()) {
41             test(dc);
42             test(dc);
43         }
44 
45         // test with IPv4 only
46         try (DatagramChannel dc = DatagramChannel.open(StandardProtocolFamily.INET)) {
47             test(dc);
48             test(dc);
49         }
50     }
51 
52     /**
53      * Connect DatagramChannel to a server, write a datagram and disconnect. Invoke
54      * a second or subsequent time with the same DatagramChannel instance to check
55      * that disconnect works as expected.
56      */
57     static void test(DatagramChannel dc) throws IOException {
58         try (DatagramChannel server = DatagramChannel.open()) {
59             server.bind(new InetSocketAddress(0));
60 
61             InetAddress lh = InetAddress.getLocalHost();
62             dc.connect(new InetSocketAddress(lh, server.socket().getLocalPort()));
63 
64             dc.write(ByteBuffer.wrap("hello".getBytes()));
65 
66             ByteBuffer bb = ByteBuffer.allocate(100);
67             server.receive(bb);
68 
69             dc.disconnect();
70 
71             try {
72                 dc.write(ByteBuffer.wrap("another message".getBytes()));
73                 throw new RuntimeException("write should fail, not connected");
74             } catch (NotYetConnectedException expected) {
75             }
76         }
77     }
78 }
