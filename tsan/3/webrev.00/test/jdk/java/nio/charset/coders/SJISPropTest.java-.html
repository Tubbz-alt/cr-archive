24 /*
25  * Regression test class run by SJISMappingPropTest.java to verify
26  * that sun.nio.cs.map property is correctly interpreted in
27  * multibyte Japanese locales
28  */
29 
30 public class SJISPropTest {
31     public static void main(String[] args) throws Exception {
32         boolean sjisIsMS932 = false;
33 
34         if (args[0].equals("MS932"))
35                 sjisIsMS932 = true;
36         byte[] testBytes = { (byte)0x81, (byte)0x60 };
37 
38         // JIS X based Shift_JIS and Windows-31J differ
39         // in a number of mappings including this one.
40 
41         String expectedMS932 = new String("\uFF5E");
42         String expectedSJIS = new String("\u301C");
43 
44         // Alias "shift_jis" will map to Windows-31J
45         // if the sun.nio.cs.map system property is defined as
46         // "Windows-31J/Shift_JIS". This should work in all
47         // multibyte (especially Japanese) locales.
48 
49         String s = new String(testBytes, "shift_jis");
50 
51         if (sjisIsMS932 && !s.equals(expectedMS932))
52             throw new Exception("not MS932");
53         else if (!sjisIsMS932 && !s.equals(expectedSJIS))
54             throw new Exception("not SJIS");
55     }
56 }
