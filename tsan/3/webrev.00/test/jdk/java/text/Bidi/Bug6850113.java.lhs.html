23 /*
24  * @test
25  * @bug 6850113
26  * @summary Verify the return value of digit() for some digits.
27  * @modules java.base/sun.text.normalizer
28  * @compile -XDignore.symbol.file=true Bug6850113.java
29  * @run main Bug6850113
30  */
31 
32 import sun.text.normalizer.UCharacter;
33 
34 public class Bug6850113 {
35 
36     public static void main(String[] args) {
37         boolean err = false;
38 
39         // Fullwidth Latin capital letters
40         for (int i = 0xff21, j = 10; i <= 0xff3a; i++, j++) {
41             if (UCharacter.digit(i, 36) != j) {
42                 err = true;
43                 System.out.println("Error: UCharacter.digit(0x" +
44                     Integer.toHexString(i) + ", 36) returned " +
45                     UCharacter.digit(i, 36) + ", expected=" + j);
46             }
47         }
48 
49         // Fullwidth Latin small letters
50         for (int i = 0xff41, j = 10; i <= 0xff5a; i++, j++) {
51             if (UCharacter.digit(i, 36) != j) {
52                 err = true;
53                 System.out.println("Error: UCharacter.digit(0x" +
54                     Integer.toHexString(i) + ", 36) returned " +
55                     UCharacter.digit(i, 36) + ", expected=" + j);
56             }
57         }
58 
59         if (err) {
60             throw new RuntimeException("UCharacter.digit():  Wrong return value");
61         }
62    }
63 }
