 33 import java.io.File;
 34 import java.io.FileInputStream;
 35 import java.util.Currency;
 36 import java.util.Locale;
 37 import java.util.Properties;
 38 import java.util.StringTokenizer;
 39 import java.util.TimeZone;
 40 import java.text.DecimalFormatSymbols;
 41 import java.text.NumberFormat;
 42 import java.text.SimpleDateFormat;
 43 
 44 public class CurrencyFormat {
 45 
 46     private static boolean isCompat;
 47 
 48     public static void main(String[] args) throws Exception {
 49         isCompat = "COMPAT".equals(args[0]);
 50         testFormatting();
 51         testSymbols();
 52     }
 53 
 54     static void testFormatting() {
 55         boolean failed = false;
 56         Locale[] locales = {
 57             Locale.US,
 58             Locale.JAPAN,
 59             Locale.GERMANY,
 60             Locale.ITALY,
 61             new Locale("it", "IT", "EURO"),
 62             Locale.forLanguageTag("de-AT"),
 63             Locale.forLanguageTag("fr-CH"),
 64         };
 65         Currency[] currencies = {
 66             null,
 67             Currency.getInstance("USD"),
 68             Currency.getInstance("JPY"),
 69             Currency.getInstance("DEM"),
 70             Currency.getInstance("EUR"),
 71         };
 72         String[][] expecteds = {
 73             {"$1,234.56", "$1,234.56", "JPY1,235", "DEM1,234.56", "EUR1,234.56"},
 74             {"\uFFE51,235", "USD1,234.56", "\uFFE51,235", "DEM1,234.56", "EUR1,234.56"},
 75             {"1.234,56 \u20AC", "1.234,56 USD", "1.235 JPY", "1.234,56 DM", "1.234,56 \u20AC"},
 76             {"\u20AC 1.234,56", "USD 1.234,56", "JPY 1.235", "DEM 1.234,56", "\u20AC 1.234,56"},
 77             {"\u20AC 1.234,56", "USD 1.234,56", "JPY 1.235", "DEM 1.234,56", "\u20AC 1.234,56"},
 78             {"\u20AC 1.234,56", "USD 1.234,56", "JPY 1.235", "DEM 1.234,56", "\u20AC 1.234,56"},
 79             {"SFr. 1'234.56", "USD 1'234.56", "JPY 1'235", "DEM 1'234.56", "EUR 1'234.56"},
 80         };
 81         String[][] expecteds_cldr = {
 82             {"$1,234.56", "$1,234.56", "\u00a51,235", "DEM1,234.56", "\u20ac1,234.56"},
 83             {"\uFFE51,235", "$1,234.56", "\uFFE51,235", "DEM1,234.56", "\u20ac1,234.56"},
 84             {"1.234,56\u00a0\u20ac", "1.234,56\u00a0$", "1.235\u00a0\u00a5", "1.234,56\u00a0DM", "1.234,56\u00a0\u20ac"},
 85             {"1.234,56\u00a0\u20ac", "1.234,56\u00a0USD", "1.235\u00a0JPY", "1.234,56\u00a0DEM", "1.234,56\u00a0\u20ac"},
 86             {"1.234,56\u00a0\u20ac", "1.234,56\u00a0USD", "1.235\u00a0JPY", "1.234,56\u00a0DEM", "1.234,56\u00a0\u20ac"},
 87             {"\u20ac\u00a01.234,56", "$\u00a01.234,56", "\u00a5\u00a01.235", "DM\u00a01.234,56", "\u20ac\u00a01.234,56"},
 88             {"1\u202f234.56\u00a0CHF", "1\u202f234.56\u00a0$US", "1\u202f235\u00a0JPY", "1\u202f234.56\u00a0DEM", "1\u202f234.56\u00a0\u20ac"},
 89         };
 90 
 91         for (int i = 0; i < locales.length; i++) {
 92             Locale locale = locales[i];
 93             NumberFormat format = NumberFormat.getCurrencyInstance(locale);
 94             for (int j = 0; j < currencies.length; j++) {
 95                 Currency currency = currencies[j];
 96                 String expected = isCompat ? expecteds[i][j] : expecteds_cldr[i][j];
 97                 if (currency != null) {
 98                     format.setCurrency(currency);
 99                     int digits = currency.getDefaultFractionDigits();
100                     format.setMinimumFractionDigits(digits);
101                     format.setMaximumFractionDigits(digits);
102                 }
103                 String result = format.format(1234.56);
104                 if (!result.equals(expected)) {
105                     failed = true;
106                     System.out.println("FAIL: Locale " + locale
107                         + (currency == null ? ", default currency" : (", currency: " + currency))
108                         + ", expected: " + expected
109                         + ", actual: " + result);
110                 }
111             }
112         }
113 
114         if (failed) {
115             throw new RuntimeException();
116         }
117     }
118 
119     static void testSymbols() throws Exception {
120         if (!isCompat) {
121             // For COMPAT only.
122             return;
123         }
124 
125         FileInputStream stream = new FileInputStream(new File(System.getProperty("test.src", "."), "CurrencySymbols.properties"));
126         Properties props = new Properties();
127         props.load(stream);
128         SimpleDateFormat format = null;
129 
130         Locale[] locales = NumberFormat.getAvailableLocales();
131         for (int i = 0; i < locales.length; i++) {
132             Locale locale = locales[i];
133             DecimalFormatSymbols symbols = DecimalFormatSymbols.getInstance(locale);
134             String result = symbols.getCurrencySymbol();
135             String expected = (String) props.get(locale.toString());
136 
137             if (expected == null) {
138                 System.out.println("Warning: No expected currency symbol defined for locale " + locale);
139             } else {
140                     if (expected.contains(";")) {
141                         StringTokenizer tokens = new StringTokenizer(expected, ";");
142                         int tokensCount = tokens.countTokens();
143 
144                         if (tokensCount == 3) {
145                             expected = tokens.nextToken();
146                             if (format == null) {
147                                 format = new SimpleDateFormat("yyyy-MM-dd-HH-mm-ss", Locale.US);
148                                 format.setTimeZone(TimeZone.getTimeZone("GMT"));
149                                 format.setLenient(false);
150                             }
151 
152                             if (format.parse(tokens.nextToken()).getTime() < System.currentTimeMillis()) {
153                                 expected = tokens.nextToken();
154                             }
155                         }
156                     }
157 
158                     if (!expected.equals(result)) {
159                         throw new RuntimeException("Wrong currency symbol for locale " +
160                             locale + ", expected: " + expected + ", got: " + result);
161                     }
162             }
163         }
164     }
