 24 import java.io.IOException;
 25 import java.nio.file.Files;
 26 import java.nio.file.Path;
 27 import java.nio.file.Paths;
 28 import java.util.List;
 29 import java.util.stream.Stream;
 30 
 31 import jdk.test.lib.JDKToolLauncher;
 32 import jdk.test.lib.compiler.CompilerUtils;
 33 import jdk.test.lib.process.ProcessTools;
 34 
 35 import static java.nio.file.StandardCopyOption.REPLACE_EXISTING;
 36 
 37 public class ModuleTestUtil {
 38 
 39     private ModuleTestUtil() {
 40         // Private constructor to prevent class instantiation
 41     }
 42 
 43     /**
 44      * Compile all the java sources and copy the resource files in the module.
 45      *
 46      * @param src path to the source directory
 47      * @param dest path to the destination directory
 48      * @param mn module name
 49      * @param resFormat resource format
 50      */
 51     public static void prepareModule(Path src, Path dest, String mn,
 52             String resFormat) {
 53         compileModule(src, dest, mn);
 54         copyResFiles(src, dest, mn, resFormat);
 55     }
 56 
 57     /**
 58      * Compile all the java sources in the module.
 59      *
 60      * @param src path to the source directory
 61      * @param dest path to the destination directory
 62      * @param mn module name
 63      */
 64     public static void compileModule(Path src, Path dest, String mn) {
 65         try {
 66             boolean compiled = CompilerUtils.compile(src.resolve(mn), dest,
 67                     "--module-source-path", src.toString());
 68             if (!compiled) {
 69                 throw new RuntimeException("Compile module " + mn + " failed.");
 70             }
 71         } catch (IOException e) {
 72             throw new RuntimeException("Compile module " + mn + " failed.");
 73         }
 74     }
 75 
 76     /**
 77      * Compile all the java sources in the unnamed package.
 78      *
 79      * @param src path to the source directory
 80      * @param dest path to the destination directory
 81      * @param pn package name
 82      */
 83     public static void compilePkg(Path src, Path dest, String pn) {
 84         try {
 85             boolean compiled = CompilerUtils.compile(src.resolve(pn),
 86                     dest.resolve(pn));
 87             if (!compiled) {
 88                 throw new RuntimeException("Compile package " + pn + " failed.");
 89             }
 90         } catch (IOException e) {
 91             throw new RuntimeException("Compile package " + pn + " failed.");
 92         }
 93     }
 94 
 95     /**
 96      * Copy all the resource files.
 97      *
 98      * @param src path to the source directory
 99      * @param dest path to the destination directory
100      * @param mn module name
101      * @param resFormat resource format
102      */
103     public static void copyResFiles(Path src, Path dest, String mn,
104             String resFormat) {
105         try (Stream<Path> stream = Files.walk(src.resolve(mn))
106                 .filter(path -> path.toString().endsWith(resFormat))) {
107             stream.forEach(f -> {
108                 String resName = f.toString();
109                 String relativePath = resName.substring(src.toString().length());
110                 Path destFile = Paths.get(dest.toString() + relativePath);
111                 try {
112                     Path destParentDir = destFile.getParent();
113                     if (Files.notExists(destParentDir)) {
114                         Files.createDirectories(destParentDir);
115                     }
116                     Files.copy(f, destFile, REPLACE_EXISTING);
117                 } catch (IOException e) {
118                     throw new RuntimeException("Copy " + f.toString() + " to "
119                             + destFile.toString() + " failed.");
120                 }
121             });
122         } catch (IOException e) {
123             throw new RuntimeException("Copy resource files failed.");
124         }
125     }
126 
127     /**
128      * Run the module test.
129      *
130      * @param mp module path
131      * @param mn module name
132      * @param localeList locale list
133      */
134     public static void runModule(String mp, String mn, List<String> localeList)
135             throws Throwable {
136         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("java");
137         launcher.addToolArg("-ea")
138                 .addToolArg("-esa")
139                 .addToolArg("-p")
140                 .addToolArg(mp)
141                 .addToolArg("-m")
142                 .addToolArg(mn);
143         localeList.forEach(launcher::addToolArg);
144 
145         int exitCode = ProcessTools.executeCommand(launcher.getCommand())
146                                    .getExitValue();
147         if (exitCode != 0) {
148             throw new RuntimeException("Execution of the test failed. "
149                     + "Unexpected exit code: " + exitCode);
150         }
151     }
152 
153     /**
154      * Run the module test with a jar file specified by the classpath.
155      *
156      * @param cp classpath
157      * @param mp module path
158      * @param mn module name
159      * @param localeList locale list
160      * @param expected expected execution status
161      */
162     public static void runModuleWithCp(String cp, String mp, String mn,
163             List<String> localeList, boolean expected) throws Throwable {
164         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("java");
165         launcher.addToolArg("-ea")
166                 .addToolArg("-esa")
167                 .addToolArg("-cp")
168                 .addToolArg(cp)
169                 .addToolArg("-p")
170                 .addToolArg(mp)
171                 .addToolArg("-m")
172                 .addToolArg(mn);
173         localeList.forEach(launcher::addToolArg);
174 
175         int exitCode = ProcessTools.executeCommand(launcher.getCommand())
176                                    .getExitValue();
177         if (expected) {
178             if (exitCode != 0) {
179                 throw new RuntimeException("Execution of the test loads bundles "
180                         + "from the jar file specified by the class-path failed. "
181                         + "Unexpected exit code: " + exitCode);
182             }
183         } else {
184             if (exitCode == 0) {
185                 throw new RuntimeException("Execution of the test not loads bundles "
186                         + "from the jar file specified by the class-path failed. "
187                         + "Unexpected exit code: " + exitCode);
188             }
189         }
190     }
191 }
