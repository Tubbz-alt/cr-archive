128             Permissions permissions = new Permissions();
129             for (Permission perm : perms) {
130                 permissions.add(perm);
131             }
132             return permissions;
133         }

135         SimplePolicy() {
136             this(permissions(new LoggingPermission("control", null),
137                     new RuntimePermission("accessClassInPackage.sun.util.logging")
138             ));

142             this.perms = perms;
143         }
144 
145         @Override

147             return perms.implies(permission) || DEFAULT_POLICY.implies(domain, permission);
