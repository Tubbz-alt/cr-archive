 34 /*
 35  * @test
 36  * @run main DoneTimedGetLoops 300
 37  * @summary isDone returning true guarantees that subsequent timed get
 38  * will never throw TimeoutException.
 39  */
 40 
 41 import java.util.concurrent.ExecutionException;
 42 import java.util.concurrent.FutureTask;
 43 import java.util.concurrent.TimeUnit;
 44 import java.util.concurrent.atomic.AtomicReference;
 45 
 46 @SuppressWarnings({"unchecked", "rawtypes", "deprecation"})
 47 public class DoneTimedGetLoops {
 48     final long testDurationMillisDefault = 10L * 1000L;
 49     final long testDurationMillis;
 50 
 51     static class PublicFutureTask extends FutureTask<Boolean> {
 52         static final Runnable noop = new Runnable() { public void run() {} };
 53         PublicFutureTask() { super(noop, null); }
 54         public void set(Boolean v) { super.set(v); }
 55         public void setException(Throwable t) { super.setException(t); }
 56     }
 57 
 58     DoneTimedGetLoops(String[] args) {
 59         testDurationMillis = (args.length > 0) ?
 60             Long.valueOf(args[0]) : testDurationMillisDefault;
 61     }
 62 
 63     void test(String[] args) throws Throwable {
 64         final long testDurationNanos = testDurationMillis * 1000L * 1000L;
 65         final long quittingTimeNanos = System.nanoTime() + testDurationNanos;
 66         final long timeoutMillis = 10L * 1000L;
 67 
 68         final AtomicReference<PublicFutureTask> normalRef
 69             = new AtomicReference<>();
 70         final AtomicReference<PublicFutureTask> abnormalRef
 71             = new AtomicReference<>();
 72 
 73         final Throwable throwable = new Throwable();
 74 
 75         abstract class CheckedThread extends Thread {
 76             CheckedThread(String name) {
 77                 super(name);
 78                 setDaemon(true);
 79                 start();
 80             }
 81             /** Polls for quitting time. */
 82             protected boolean quittingTime() {
 83                 return System.nanoTime() - quittingTimeNanos > 0;
 84             }
 85             /** Polls occasionally for quitting time. */
 86             protected boolean quittingTime(long i) {
 87                 return (i % 1024) == 0 && quittingTime();
 88             }
 89             protected abstract void realRun() throws Exception;
 90             public void run() {
 91                 try { realRun(); } catch (Throwable t) { unexpected(t); }
 92             }
 93         }
 94 
 95         Thread setter = new CheckedThread("setter") {
 96             protected void realRun() {
 97                 while (! quittingTime()) {
 98                     PublicFutureTask future = new PublicFutureTask();
 99                     normalRef.set(future);
100                     future.set(Boolean.TRUE);
101                 }}};
102 
103         Thread setterException = new CheckedThread("setterException") {
104             protected void realRun() {
105                 while (! quittingTime()) {
106                     PublicFutureTask future = new PublicFutureTask();
107                     abnormalRef.set(future);
108                     future.setException(throwable);
109                 }}};
110 
111         Thread doneTimedGetNormal = new CheckedThread("doneTimedGetNormal") {
112             protected void realRun() throws Exception {
113                 while (! quittingTime()) {
114                     PublicFutureTask future = normalRef.get();
115                     if (future != null) {
116                         while (!future.isDone())
117                             ;
118                         check(future.get(0L, TimeUnit.HOURS) == Boolean.TRUE);
119                     }}}};
120 
121         Thread doneTimedGetAbnormal = new CheckedThread("doneTimedGetAbnormal") {
122             protected void realRun() throws Exception {
123                 while (! quittingTime()) {
124                     PublicFutureTask future = abnormalRef.get();
125                     if (future != null) {
126                         while (!future.isDone())
127                             ;
128                         try { future.get(0L, TimeUnit.HOURS); fail(); }
129                         catch (ExecutionException t) {
130                             check(t.getCause() == throwable);
131                         }
132                     }}}};
133 
134         for (Thread thread : new Thread[] {
135                  setter,
136                  setterException,
137                  doneTimedGetNormal,
138                  doneTimedGetAbnormal }) {
139             thread.join(timeoutMillis + testDurationMillis);
140             if (thread.isAlive()) {
141                 System.err.printf("Hung thread: %s%n", thread.getName());
142                 failed++;
143                 for (StackTraceElement e : thread.getStackTrace())
144                     System.err.println(e);
145                 thread.join(timeoutMillis);
146             }
147         }
148     }
149 
150     //--------------------- Infrastructure ---------------------------
151     volatile int passed = 0, failed = 0;
152     void pass() {passed++;}
153     void fail() {failed++; Thread.dumpStack();}
154     void fail(String msg) {System.err.println(msg); fail();}
155     void unexpected(Throwable t) {failed++; t.printStackTrace();}
156     void check(boolean cond) {if (cond) pass(); else fail();}
157     void equal(Object x, Object y) {
158         if (x == null ? y == null : x.equals(y)) pass();
159         else fail(x + " not equal to " + y);}
160     public static void main(String[] args) throws Throwable {
161         new DoneTimedGetLoops(args).instanceMain(args);}
162     public void instanceMain(String[] args) throws Throwable {
163         try {test(args);} catch (Throwable t) {unexpected(t);}
164         System.out.printf("%nPassed = %d, failed = %d%n%n", passed, failed);
165         if (failed > 0) throw new AssertionError("Some tests failed");}
166 }
