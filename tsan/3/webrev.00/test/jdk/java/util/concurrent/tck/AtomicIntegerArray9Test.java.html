 48     /**
 49      * get and set for out of bound indices throw IndexOutOfBoundsException
 50      */
 51     public void testIndexing() {
 52         AtomicIntegerArray aa = new AtomicIntegerArray(SIZE);
 53         for (int index : new int[] { -1, SIZE }) {
 54             final int j = index;
 55 
 56             assertThrows(
 57                 IndexOutOfBoundsException.class,
 58                 () -> aa.getPlain(j),
 59                 () -> aa.getOpaque(j),
 60                 () -> aa.getAcquire(j),
 61                 () -> aa.setPlain(j, 1),
 62                 () -> aa.setOpaque(j, 1),
 63                 () -> aa.setRelease(j, 1),
 64                 () -> aa.compareAndExchange(j, 1, 2),
 65                 () -> aa.compareAndExchangeAcquire(j, 1, 2),
 66                 () -> aa.compareAndExchangeRelease(j, 1, 2),
 67                 () -> aa.weakCompareAndSetPlain(j, 1, 2),
 68                 () -> aa.weakCompareAndSetVolatile(j, 1, 2),
 69                 () -> aa.weakCompareAndSetAcquire(j, 1, 2),
 70                 () -> aa.weakCompareAndSetRelease(j, 1, 2));
 71         }
 72     }

 74     /**
 75      * getPlain returns the last value set
 76      */
 77     public void testGetPlainSet() {
 78         AtomicIntegerArray aa = new AtomicIntegerArray(SIZE);
 79         for (int i = 0; i < SIZE; i++) {
 80             aa.set(i, 1);
 81             assertEquals(1, aa.getPlain(i));
 82             aa.set(i, 2);
 83             assertEquals(2, aa.getPlain(i));
 84             aa.set(i, -3);
 85             assertEquals(-3, aa.getPlain(i));
 86         }
 87     }

164     /**
165      * compareAndExchange succeeds in changing value if equal to
166      * expected else fails
167      */
168     public void testCompareAndExchange() {
169         AtomicIntegerArray aa = new AtomicIntegerArray(SIZE);
170         for (int i = 0; i < SIZE; i++) {
171             aa.set(i, 1);
172             assertEquals(1, aa.compareAndExchange(i, 1, 2));
173             assertEquals(2, aa.compareAndExchange(i, 2, -4));
174             assertEquals(-4, aa.get(i));
175             assertEquals(-4, aa.compareAndExchange(i,-5, 7));
176             assertEquals(-4, aa.get(i));
177             assertEquals(-4, aa.compareAndExchange(i, -4, 7));
178             assertEquals(7, aa.get(i));
179         }
180     }

218     /**
219      * repeated weakCompareAndSetPlain succeeds in changing value when equal
220      * to expected
221      */
222     public void testWeakCompareAndSetPlain() {
223         AtomicIntegerArray aa = new AtomicIntegerArray(SIZE);
224         for (int i = 0; i < SIZE; i++) {
225             aa.set(i, 1);
226             do {} while (!aa.weakCompareAndSetPlain(i, 1, 2));
227             do {} while (!aa.weakCompareAndSetPlain(i, 2, -4));
228             assertEquals(-4, aa.get(i));
229             do {} while (!aa.weakCompareAndSetPlain(i, -4, 7));
230             assertEquals(7, aa.get(i));
231         }
232     }
