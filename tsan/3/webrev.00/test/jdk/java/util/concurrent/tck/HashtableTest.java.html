34 import java.util.Map;
35 import java.util.Hashtable;
36 
37 import junit.framework.Test;
38 
39 public class HashtableTest extends JSR166TestCase {
40     public static void main(String[] args) {
41         main(suite(), args);
42     }
43     public static Test suite() {
44         class Implementation implements MapImplementation {
45             public Class<?> klazz() { return Hashtable.class; }
46             public Map emptyMap() { return new Hashtable(); }
47             public boolean isConcurrent() { return true; }
48             public boolean permitsNullKeys() { return false; }
49             public boolean permitsNullValues() { return false; }
50             public boolean supportsSetValue() { return true; }
51         }
52         return newTestSuite(MapTest.testSuite(new Implementation()));
53     }
54 }
