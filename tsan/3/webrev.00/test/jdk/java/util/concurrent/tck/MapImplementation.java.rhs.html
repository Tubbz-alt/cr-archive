35 import java.util.Map;
36 
37 /** Allows tests to work with different Map implementations. */
38 public interface MapImplementation {
39     /** Returns the Map implementation class. */
40     public Class<?> klazz();
41     /** Returns an empty map. */
42     public Map emptyMap();
43 
44     // General purpose implementations can use Integers for key and value
45     default Object makeKey(int i) { return i; }
46     default Object makeValue(int i) { return i; }
47     default int keyToInt(Object key) { return (Integer) key; }
48     default int valueToInt(Object value) { return (Integer) value; }
49 
50     public boolean isConcurrent();
51     default boolean remappingFunctionCalledAtMostOnce() { return true; };
52     public boolean permitsNullKeys();
53     public boolean permitsNullValues();
54     public boolean supportsSetValue();
55 }
