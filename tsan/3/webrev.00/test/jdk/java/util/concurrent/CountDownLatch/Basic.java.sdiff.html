 37 public class Basic {
 38     static final long LONG_DELAY_MS = Utils.adjustTimeout(10_000);

 40     interface AwaiterFactory {
 41         Awaiter getAwaiter();
 42     }
 43 
 44     abstract static class Awaiter extends Thread {
 45         private volatile Throwable result = null;
 46         protected void result(Throwable result) { this.result = result; }
 47         public Throwable result() { return this.result; }
 48     }
 49 
 50     private void toTheStartingGate(CountDownLatch gate) {
 51         try {
 52             gate.await();
 53         }
 54         catch (Throwable t) { fail(t); }
 55     }
 56 
 57     private Awaiter awaiter(final CountDownLatch latch,
 58                             final CountDownLatch gate) {
 59         return new Awaiter() { public void run() {
 60             System.out.println("without millis: " + latch.toString());
 61             gate.countDown();
 62 
 63             try {
 64                 latch.await();
 65                 System.out.println("without millis - ComingOut");
 66             }
 67             catch (Throwable result) { result(result); }}};
 68     }
 69 
 70     private Awaiter awaiter(final CountDownLatch latch,
 71                             final CountDownLatch gate,
 72                             final long millis) {
 73         return new Awaiter() { public void run() {
 74             System.out.println("with millis: "+latch.toString());
 75             gate.countDown();
 76 
 77             try {
 78                 latch.await(millis, TimeUnit.MILLISECONDS);
 79                 System.out.println("with millis - ComingOut");
 80             }
 81             catch (Throwable result) { result(result); }}};
 82     }
 83 
 84     AwaiterFactory awaiterFactory(CountDownLatch latch, CountDownLatch gate) {
 85         return () -> awaiter(latch, gate);






 86     }
 87 
 88     AwaiterFactory timedAwaiterFactory(CountDownLatch latch, CountDownLatch gate) {
 89         return () -> awaiter(latch, gate, LONG_DELAY_MS);



 90     }

 92     //----------------------------------------------------------------
 93     // Normal use
 94     //----------------------------------------------------------------
 95     public static void normalUse() throws Throwable {
 96         int count = 0;
 97         Basic test = new Basic();
 98         CountDownLatch latch = new CountDownLatch(3);
 99         Awaiter[] a = new Awaiter[12];
100 
101         for (int i = 0; i < 3; i++) {
102             CountDownLatch gate = new CountDownLatch(4);
103             AwaiterFactory factory1 = test.awaiterFactory(latch, gate);
104             AwaiterFactory factory2 = test.timedAwaiterFactory(latch, gate);
105             a[count] = factory1.getAwaiter(); a[count++].start();
106             a[count] = factory1.getAwaiter(); a[count++].start();
107             a[count] = factory2.getAwaiter(); a[count++].start();
108             a[count] = factory2.getAwaiter(); a[count++].start();
109             test.toTheStartingGate(gate);
110             System.out.println("Main Thread: " + latch.toString());
111             latch.countDown();
112             checkCount(latch, 2-i);
113         }
114         for (int i = 0; i < 12; i++)
115             a[i].join();
116 
117         for (int i = 0; i < 12; i++)
118             checkResult(a[i], null);
119     }

208     private static void checkResult(Awaiter a, Class c) {
209         Throwable t = a.result();
210         if (! ((t == null && c == null) || c.isInstance(t))) {
211             System.out.println("Mismatch: " + t + ", " + c.getName());
212             failures.getAndIncrement();
213         }

 32 import java.util.concurrent.CountDownLatch;
 33 import java.util.concurrent.ThreadLocalRandom;
 34 import java.util.concurrent.TimeUnit;
 35 import java.util.concurrent.atomic.AtomicInteger;
 36 import java.util.function.Supplier;
 37 import jdk.test.lib.Utils;

 39 public class Basic {
 40     static final long LONG_DELAY_MS = Utils.adjustTimeout(10_000);
 41 




 42     abstract static class Awaiter extends Thread {
 43         volatile Throwable exception;
 44         volatile boolean interrupted;
 45         abstract void realRun() throws Exception;
 46         public final void run() {
 47             try { realRun(); }
 48             catch (Throwable ex) { exception = ex; }
 49             interrupted = Thread.interrupted();
 50         };


 51     }
 52 
 53     static Awaiter awaiter(CountDownLatch latch,
 54                            CountDownLatch gate) {
 55         return new Awaiter() {
 56             public void realRun() throws InterruptedException {
 57                 gate.countDown();


 58                 latch.await();
 59             }};
















 60     }
 61 
 62     static Awaiter awaiter(CountDownLatch latch,
 63                            CountDownLatch gate,
 64                            long timeoutMillis) {
 65         return new Awaiter() {
 66             public void realRun() throws InterruptedException {
 67                 gate.countDown();
 68                 latch.await(timeoutMillis, TimeUnit.MILLISECONDS);
 69             }};
 70     }
 71 
 72     static Supplier<Awaiter> randomAwaiterSupplier(
 73             CountDownLatch latch, CountDownLatch gate) {
 74         return () -> (ThreadLocalRandom.current().nextBoolean())
 75             ? awaiter(latch, gate)
 76             : awaiter(latch, gate, LONG_DELAY_MS);
 77     }
 78 
 79     //----------------------------------------------------------------
 80     // Normal use
 81     //----------------------------------------------------------------
 82     public static void normalUse() throws Throwable {
 83         int count = 0;

 84         CountDownLatch latch = new CountDownLatch(3);
 85         Awaiter[] a = new Awaiter[12];
 86 
 87         for (int i = 0; i < 3; i++) {
 88             CountDownLatch gate = new CountDownLatch(4);
 89             Supplier<Awaiter> s = randomAwaiterSupplier(latch, gate);
 90             a[count] = s.get(); a[count++].start();
 91             a[count] = s.get(); a[count++].start();
 92             a[count] = s.get(); a[count++].start();
 93             a[count] = s.get(); a[count++].start();
 94             gate.await();


 95             latch.countDown();
 96             checkCount(latch, 2-i);
 97         }
 98         for (Awaiter awaiter : a)
 99             awaiter.join();
100         for (Awaiter awaiter : a)
101             checkException(awaiter, null);

102     }

104     //----------------------------------------------------------------
105     // One thread interrupted
106     //----------------------------------------------------------------
107     public static void threadInterrupted() throws Throwable {
108         int count = 0;

109         CountDownLatch latch = new CountDownLatch(3);
110         Awaiter[] a = new Awaiter[12];
111 
112         for (int i = 0; i < 3; i++) {
113             CountDownLatch gate = new CountDownLatch(4);
114             Supplier<Awaiter> s = randomAwaiterSupplier(latch, gate);
115             a[count] = s.get(); a[count++].start();
116             a[count] = s.get(); a[count++].start();
117             a[count] = s.get(); a[count++].start();
118             a[count] = s.get(); a[count++].start();
119             gate.await();
120             a[count-1].interrupt();


121             latch.countDown();
122             checkCount(latch, 2-i);
123         }
124         for (Awaiter awaiter : a)
125             awaiter.join();
126         for (int i = 0; i < a.length; i++) {
127             Awaiter awaiter = a[i];
128             Throwable ex = awaiter.exception;
129             if ((i % 4) == 3 && !awaiter.interrupted)
130                 checkException(awaiter, InterruptedException.class);
131             else
132                 checkException(awaiter, null);
133         }
134     }

136     //----------------------------------------------------------------
137     // One thread timed out
138     //----------------------------------------------------------------
139     public static void timeOut() throws Throwable {
140         int count = 0;

141         CountDownLatch latch = new CountDownLatch(3);
142         Awaiter[] a = new Awaiter[12];
143 
144         long[] timeout = { 0L, 5L, 10L };
145 
146         for (int i = 0; i < 3; i++) {
147             CountDownLatch gate = new CountDownLatch(4);
148             Supplier<Awaiter> s = randomAwaiterSupplier(latch, gate);
149             a[count] = awaiter(latch, gate, timeout[i]); a[count++].start();
150             a[count] = s.get(); a[count++].start();
151             a[count] = s.get(); a[count++].start();
152             a[count] = s.get(); a[count++].start();
153             gate.await();


154             latch.countDown();
155             checkCount(latch, 2-i);
156         }
157         for (Awaiter awaiter : a)
158             awaiter.join();
159         for (Awaiter awaiter : a)
160             checkException(awaiter, null);

161     }

197         if (! ((ex == null && c == null) || c.isInstance(ex)))
198             fail("Expected: " + c + ", got: " + ex);
