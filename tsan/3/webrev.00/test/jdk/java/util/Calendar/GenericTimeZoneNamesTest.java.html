24 /*
25  * @test
26  * @bug 8003267
27  * @summary Unit test for generic time zone names support. This test is locale
28  *          data-dependent and assumes that both JRE and CLDR have the same
29  *          geneic time zone names in English.
30  * @modules java.base/sun.util.locale.provider
31  * @comment Locale providers: default
32  * @run main GenericTimeZoneNamesTest en-US
33  * @comment Locale providers: CLDR
34  * @run main/othervm -Djava.locale.providers=CLDR GenericTimeZoneNamesTest en-US
35 */
36 
37 import java.util.Locale;
38 import java.util.TimeZone;
39 
40 import sun.util.locale.provider.TimeZoneNameUtility;
41 
42 public class GenericTimeZoneNamesTest {
43     private static final String[] PT = {
44         "America/Los_Angeles", "US/Pacific", "PST"
45     };
46 
47     private static int errors = 0;
48 
49     public static void main(String[] args) {
50         for (String tag : args) {
51             Locale locale = Locale.forLanguageTag(tag);
52             for (String tzid : PT) {
53                 test(tzid, TimeZone.LONG, locale, "Pacific Time");
54                 test(tzid, TimeZone.SHORT, locale, "PT");
55             }
56         }
57 
58         if (errors != 0) {
59             throw new RuntimeException("test failed");
60         }
61     }
62 
63     private static void test(String tzid, int style, Locale locale, String expected) {
64         // No public API to get generic time zone names (JDK 8)
65         String got = TimeZoneNameUtility.retrieveGenericDisplayName(tzid, style, locale);
66         if (!expected.equals(got)) {
67             System.err.printf("test: tzid=%s, locale=%s, style=%d, got=\"%s\", expected=\"%s\"%n",
68                               tzid, locale, style, got, expected);
69             errors++;
70         }
71     }
72 }
