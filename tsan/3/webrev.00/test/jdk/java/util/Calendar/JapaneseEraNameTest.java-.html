24 /*
25  * @test
26  * @bug 8202088 8207152 8217609 8219890
27  * @summary Test the localized Japanese new era name (May 1st. 2019-)
28  *      is retrieved no matter CLDR provider contains the name or not.
29  * @modules jdk.localedata
30  * @run testng/othervm JapaneseEraNameTest
31  * @run testng/othervm -Djava.locale.providers=CLDR JapaneseEraNameTest
32  */
33 
34 import static java.util.Calendar.*;
35 import static java.util.Locale.*;
36 import java.util.Calendar;
37 import java.util.Locale;
38 
39 import org.testng.annotations.DataProvider;
40 import org.testng.annotations.Test;
41 import static org.testng.Assert.assertEquals;
42 
43 @Test
44 public class JapaneseEraNameTest {
45     static final Calendar c = new Calendar.Builder()
46             .setCalendarType("japanese")
47             .setFields(ERA, 5, YEAR, 1, MONTH, MAY, DAY_OF_MONTH, 1)
48             .build();
49 
50     @DataProvider(name="names")
51     Object[][] names() {
52         return new Object[][] {
53             // type,    locale,  name
54             { LONG,     JAPAN,   "\u5143\u53f7" }, // NewEra
55             { LONG,     US,      "NewEra" },
56             { LONG,     CHINA,   "NewEra" },
57             { SHORT,    JAPAN,   "\u5143\u53f7" },
58             { SHORT,    US,      "NewEra" },
59             { SHORT,    CHINA,   "N" },
60         };
61     }
62 
63     @Test(dataProvider="names")
64     public void testJapaneseNewEraName(int type, Locale locale, String expected) {
65         assertEquals(c.getDisplayName(ERA, type, locale), expected);
66     }
67 }
