24 /**
25  * @test
26  * @bug 4842702 8211765
27  * @summary Check that constructors throw specified exceptions
28  * @author Martin Buchholz
29  */
30 
31 import java.util.jar.JarFile;
32 import java.io.File;
33 import java.io.IOException;
34 
35 public class Constructor {
36     private static void Unreached (Object o)
37         throws Exception
38     {
39         // Should never get here
40         throw new Exception ("Expected exception was not thrown");
41     }
42 
43     public static void main(String[] args)
44         throws Exception
45     {
46         try { Unreached (new JarFile ((File) null, true, JarFile.OPEN_READ)); }
47         catch (NullPointerException e) {}
48 
49         try { Unreached (new JarFile ((File) null, true)); }
50         catch (NullPointerException e) {}
51 
52         try { Unreached (new JarFile ((File) null)); }
53         catch (NullPointerException e) {}
54 
55         try { Unreached (new JarFile ((String) null, true)); }
56         catch (NullPointerException e) {}
57 
58         try { Unreached (new JarFile ((String) null)); }
59         catch (NullPointerException e) {}
60 
61         try { Unreached (new JarFile ("NoSuchJar.jar")); }
62         catch (IOException e) {}
63 
64         try { Unreached (new JarFile (new File ("NoSuchJar.jar"))); }
65         catch (IOException e) {}
66 
67         // Test that an IOExcception is thrown when an invalid charater
68         // is part of the path on Windows and Unix
69         final String invalidOSPath = System.getProperty("os.name")
70                 .startsWith("Windows") ? "C:\\*" : "foo\u0000bar";
71 
72         try { Unreached (new JarFile (invalidOSPath)); }
73         catch (IOException e) {}
74     }
75 }
