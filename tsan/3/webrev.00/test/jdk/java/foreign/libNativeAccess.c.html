 29 JNIEXPORT jbyte JNICALL
 30 Java_TestNative_getByteRaw(JNIEnv *env, jclass cls, jlong addr, jint index) {
 31     jbyte *arr = (jbyte*)(uintptr_t)addr;
 32     return arr[index];
 33 }
 34 
 35 JNIEXPORT jbyte JNICALL
 36 Java_TestNative_getByteBuffer(JNIEnv *env, jclass cls, jobject buf, jint index) {
 37     jlong addr = (jlong)(uintptr_t)(*env)->GetDirectBufferAddress(env, buf);
 38     return Java_TestNative_getByteRaw(env, cls, addr, index);
 39 }
