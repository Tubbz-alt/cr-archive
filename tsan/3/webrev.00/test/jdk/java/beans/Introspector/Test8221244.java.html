24 /*
25  * @test
26  * @bug 8221244
27  * @summary Unexpected behavior of PropertyDescription.getReadMethod for boolean properties
28  */
29 
30 import java.beans.PropertyDescriptor;
31 
32 public class Test8221244 {
33 
34     public static void main(String[] args) {
35         test("bv", "isBv"); // boolean value
36         test("bo", "getBo"); // Boolean object
37         test("io", "getIo"); // Integer object
38     }
39 
40     private static void test(String propertyName, String expectedGetterName) {
41         PropertyDescriptor pd = BeanUtils.getPropertyDescriptor(TestBean.class, propertyName);
42         String getterName = pd.getReadMethod().getName();
43         if (!getterName.equals(expectedGetterName)) {
44             throw new Error("unexpected getter: " + getterName);
45         }
46     }
47 }
48 
49 /*
50  * Bean class with multiple properties (each property has "is"/"get" getters)
51  *
52  * For boolean properties, the "is" getter should be used
53  */
54 class TestBean {
55 
56     // boolean value
57     private boolean bv;
58 
59     public boolean isBv() {
60         return bv;
61     }
62 
63     public boolean getBv() {
64         return bv;
65     }
66 
67     // Boolean object
68     private Boolean bo;
69     public Boolean isBo() {
70         return bo;
71     }
72     public Boolean getBo() {
73         return bo;
74     }
75 
76     // Integer object
77     private Integer io;
78     public Integer isIo() {
79         return io;
80     }
81     public Integer getIo() {
82         return io;
83     }
84 }
