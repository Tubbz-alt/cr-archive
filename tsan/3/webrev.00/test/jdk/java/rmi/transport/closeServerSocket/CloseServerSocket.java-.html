 95     private static void verifyPortFree(int port) throws IOException {
 96         ServerSocket ss = new ServerSocket(port);
 97         ss.close();
 98         System.err.println("- port " + port + " is free");
 99     }
100 
101     private static void verifyPortInUse(int port) throws IOException {
102         try {
103             verifyPortFree(port);
104         } catch (BindException e) {
105             System.err.println("- port " + port + " is in use");
106             return;
107         }
108     }
109 
110     private static class SSF implements RMIServerSocketFactory {
111         boolean serverSocketClosed = false;
112         SSF() { };
113 
114         public ServerSocket createServerSocket(int port) throws IOException {
115             return new SS(port);
116         }
117 
118         private class SS extends ServerSocket {
119             SS(int port) throws IOException {
120                 super(port);
121                 System.err.println("- created server socket: " + this);
122             };
123 
124             public void close() throws IOException {
125                 synchronized (SSF.this) {
126                     serverSocketClosed = true;
127                     SSF.this.notifyAll();
128                 }
129                 System.err.println("- closing server socket: " + this);
130                 super.close();
131             }
132         }
133     }
