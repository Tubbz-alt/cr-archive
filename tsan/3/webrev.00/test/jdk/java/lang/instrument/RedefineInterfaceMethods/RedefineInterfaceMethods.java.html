24 /*
25  * @test
26  * @bug 8225325
27  * @summary Add tests for redefining a class' private method during resolution of the bootstrap specifier
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  * @modules java.compiler
31  *          java.instrument
32  * @compile ../NamedBuffer.java
33  * @compile redef/Xost.java
34  * @run main RedefineClassHelper
35  * @run main/othervm -javaagent:redefineagent.jar -Xlog:redefine+class*=trace RedefineInterfaceMethods
36  */
37 
38 class Host {
39     static void log(String msg) { System.out.println(msg); }
40 
41     static interface B {
42         int ORIGINAL_RETURN = 1;
43         int NEW_RETURN = 2;
44 
45         private int privateMethod() {
46             Runnable race1 = () -> log("Hello from inside privateMethod");
47             race1.run();
48             return ORIGINAL_RETURN;
49         }
50 
51         public default int defaultMethod(String p) {
52             log(p + "from interface B's defaultMethod");
53             return privateMethod();
54         }
55     }
56 
57     static class Impl implements B {
58     }
59 }
60 
61 public class RedefineInterfaceMethods {
62     private static byte[] bytesForHostClass(char replace) throws Throwable {
63         return NamedBuffer.bytesForHostClass(replace, "Host$B");
64     }
65 
66     public static void main(String[] args) throws Throwable {
67         Host.Impl impl = new Host.Impl();
68 
69         int res = impl.defaultMethod("Hello ");
70         if (res != Host.B.ORIGINAL_RETURN)
71             throw new Error("defaultMethod returned " + res +
72                             " expected " + Host.B.ORIGINAL_RETURN);
73 
74         byte[] buf = bytesForHostClass('X');
75         RedefineClassHelper.redefineClass(Host.B.class, buf);
76 
77         res = impl.defaultMethod("Goodbye ");
78         if (res != Host.B.NEW_RETURN)
79             throw new Error("defaultMethod returned " + res +
80                             " expected " + Host.B.NEW_RETURN);
81     }
82 }
