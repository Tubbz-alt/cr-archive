24 /*
25  * @test
26  * @bug 8225325
27  * @summary Add tests for redefining a class' private method during resolution of the bootstrap specifier
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  * @modules java.compiler
31  *          java.instrument
32  * @compile ../NamedBuffer.java
33  * @compile redef/Xost.java
34  * @run main RedefineClassHelper
35  * @run main/othervm -javaagent:redefineagent.jar -Xlog:redefine+class*=trace RedefineMathTesterMethods
36  */
37 
38 interface MathOperation {
39     public int operation(int a, int b);
40 }
41 
42 class Host {
43     static class B {
44         public static int operate(int a, int b, MathOperation mathOperation) {
45             return mathOperation.operation(a, b);
46         }
47 
48         static int test_math() {
49             MathOperation addition = (int a, int b) -> a + b;
50             return operate(10, 5, addition);
51         }
52     }
53 }
54 
55 public class RedefineMathTesterMethods {
56     private static byte[] bytesForHostClass(char replace) throws Throwable {
57         return NamedBuffer.bytesForHostClass(replace, "Host$B");
58     }
59 
60     public static void main(String[] args) throws Throwable {
61         int res = Host.B.test_math();
62         System.out.println("Result = " + res);
63         if (res != 15) {
64             throw new Error("test_math returned " + res + " expected " + 15);
65         }
66 
67         byte[] buf = bytesForHostClass('X');
68         RedefineClassHelper.redefineClass(Host.B.class, buf);
69 
70         res = Host.B.test_math();
71         if (res != 5)
72             throw new Error("test_math returned " + res + " expected " + 5);
73     }
74 }
