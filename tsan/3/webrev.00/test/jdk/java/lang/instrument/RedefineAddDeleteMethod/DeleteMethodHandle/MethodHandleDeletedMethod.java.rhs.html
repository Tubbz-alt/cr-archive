27  * @summary Break ResolvedMethodTable with redefined nest class.

32  * @compile ../../NamedBuffer.java
33  * @compile redef/Xost.java
34  * @run main RedefineClassHelper
35  * @run main/othervm -XX:+AllowRedefinitionToAddDeleteMethods -javaagent:redefineagent.jar -Xlog:redefine+class+update*=debug,membername+table=debug MethodHandleDeletedMethod
36  */
37 
38 import java.io.File;
39 import java.io.FileInputStream;
40 import java.lang.invoke.*;
41 
42 class Host {
43     static MethodHandle fooMH;
44 
45     static class A {
46         private static void foo() { System.out.println("OLD foo called"); }
47     }
48     static void bar() throws NoSuchMethodError {
49         A.foo();
50     }
51     static void barMH() throws Throwable {
52         fooMH.invokeExact();
53     }
54 
55     public static void reresolve() throws Throwable {
56         fooMH = MethodHandles.lookup().findStatic(A.class, "foo", MethodType.methodType(void.class));
57     }
58 
59     static {
60         try {
61           fooMH = MethodHandles.lookup().findStatic(A.class, "foo", MethodType.methodType(void.class));
62         } catch (ReflectiveOperationException ex) {
63         }
64     }
65 }
66 
67 public class MethodHandleDeletedMethod {
68 
69     static final String DEST = System.getProperty("test.classes");
70     static final boolean VERBOSE = false;
71 
72     private static byte[] bytesForHostClass(char replace) throws Throwable {
73         return NamedBuffer.bytesForHostClass(replace, "Host$A");
74     }
75 
76     public static void main(java.lang.String[] unused) throws Throwable {
77         Host h = new Host();
78         h.bar();
79         h.barMH();
80         byte[] buf = bytesForHostClass('X');
81         RedefineClassHelper.redefineClass(Host.A.class, buf);
82         try {
83             h.bar();    // call deleted Method directly
84             throw new RuntimeException("Failed, expected NSME");
85         } catch (NoSuchMethodError nsme) {
86             System.out.println("Received expected NSME");
87         }
88         try {
89             h.barMH();  // call through MethodHandle for deleted Method
90             throw new RuntimeException("Failed, expected NSME");
91         } catch (NoSuchMethodError nsme) {
92             System.out.println("Received expected NSME");
93         }
94         System.out.println("Passed.");
95     }
96 }
