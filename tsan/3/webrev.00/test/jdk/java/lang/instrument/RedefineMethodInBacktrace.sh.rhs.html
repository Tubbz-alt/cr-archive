37 then

41

43 then
44   COMPILEJAVA="${TESTJAVA}"
45 fi
46 echo "COMPILEJAVA=${COMPILEJAVA}"
47 
48 if [ "${TESTSRC}" = "" ]
49 then
50   echo "TESTSRC not set.  Test cannot execute.  Failed."

66

72     -XX:+AllowRedefinitionToAddDeleteMethods \
73     -classpath "${TESTCLASSES}" RedefineMethodInBacktraceApp > output.log 2>&1
