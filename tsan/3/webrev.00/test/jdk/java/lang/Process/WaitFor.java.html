24 /*
25  * @test
26  * @bug 8220684
27  * @summary Process.waitFor(long, TimeUnit) can return false for a process
28  *          that exited within the timeout
29  * @run main/othervm WaitFor
30  */
31 
32 import java.io.InputStream;
33 import java.io.OutputStream;
34 import java.util.concurrent.TimeUnit;
35 
36 public class WaitFor {
37     public static void main(String[] args) throws Throwable {
38         int failCnt = 0;
39         for (int i = 0; i < 30; ++i) {
40             Process proc = new MyProcess(new ProcessBuilder("true").start());
41             boolean exited = proc.waitFor(100, TimeUnit.MILLISECONDS);
42             if (!exited && !proc.isAlive()) failCnt++;
43         }
44         if (failCnt > 10) {
45             throw new RuntimeException(failCnt + " processes were still alive"
46                 + " after timeout");
47         }
48     }
49 }
50 
51 /**
52  * This class uses the default implementation of java.lang.Process#waitFor(long,
53  * TimeUnit), and delegates all other calls to the actual implementation of
54  * Process.
55  */
56 class MyProcess extends Process {
57     Process impl;
58     public MyProcess(Process impl) { this.impl = impl; }
59     public OutputStream getOutputStream() { return impl.getOutputStream(); }
60     public InputStream getInputStream() { return impl.getInputStream(); }
61     public InputStream getErrorStream() { return impl.getErrorStream(); }
62     public int waitFor() throws InterruptedException { return impl.waitFor(); }
63     public int exitValue() { return impl.exitValue(); }
64     public void destroy() { impl.destroy(); }
65     public ProcessHandle toHandle() { return impl.toHandle(); }
66 }
