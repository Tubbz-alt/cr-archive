 24 /*
 25  * @test
 26  * @bug 8229785
 27  * @library /test/lib
 28  * @build jdk.unsupported/*
 29  * @summary MethodTypeDesc::resolveConstantDesc with security manager
 30  * @run main/othervm/policy=test.policy ResolveConstantDesc
 31  */
 32 
 33 import java.lang.constant.MethodTypeDesc;
 34 import java.lang.invoke.MethodHandles;
 35 import java.lang.invoke.MethodHandles.Lookup;
 36 import java.lang.invoke.MethodType;
 37 import java.security.AccessControlException;;
 38 import java.security.Permission;;
 39 
 40 import static jdk.test.lib.Asserts.*;
 41 
 42 /*
 43  * MethodTypeDesc::resolveConstantDec may get security exception depending
 44  * on the access of Lookup object
 45  */
 46 public class ResolveConstantDesc {
 47     private static final String DESCRIPTOR = "()Ljdk/internal/misc/VM;";
 48 
 49     public static void main(String... args) throws Exception {
 50         // private Lookup object has access to classes exported from another module
 51         Lookup lookup = sun.misc.Test.LOOKUP;
 52         Module m = lookup.lookupClass().getModule();
 53 
 54         MethodType mtype = MethodType.fromMethodDescriptorString(DESCRIPTOR, ClassLoader.getPlatformClassLoader());
 55         Class<?> target = mtype.returnType();
 56         Module javaBase = target.getModule();
 57         assertTrue(javaBase.isExported(target.getPackageName(), m));
 58 
 59         // MethodType that references java.base internal class
 60         MethodTypeDesc mtd = MethodTypeDesc.ofDescriptor(DESCRIPTOR);
 61         testInaccessibleClass(mtd);
 62 
 63         // Lookup has no access to JDK internal API; IAE
 64         throwIAE(MethodHandles.lookup(), mtd);
 65 
 66         // resolve successfully if Lookup has access to sun.misc and security permission
 67         MethodTypeDesc.ofDescriptor("()Lsun/misc/Unsafe;")
 68                       .resolveConstantDesc(MethodHandles.lookup());
 69     }
 70 
 71     /*
 72      * Test Lookup with different access
 73      */
 74     private static void testInaccessibleClass(MethodTypeDesc mtd) throws Exception {
 75         Lookup lookup = sun.misc.Test.LOOKUP;
 76         // full power lookup can resolve MethodTypeDesc of java.base internal types
 77         mtd.resolveConstantDesc(lookup);
 78 
 79         // drop PRIVATE access; fail package access check
 80         throwACC(lookup.dropLookupMode(Lookup.PRIVATE), mtd);
 81 
 82         // jdk.internal.access is not accessible by jdk.unsupported
 83         MethodTypeDesc mtd1 = MethodTypeDesc.ofDescriptor("()Ljdk/internal/access/SharedSecrets;");
 84         throwIAE(lookup, mtd1);
 85     }
 86 
 87     // IAE thrown when resolving MethodType using the given Lookup object
 88     private static void throwIAE(Lookup lookup, MethodTypeDesc mtd) throws Exception {
 89         try {
 90             MethodType mtype = (MethodType)mtd.resolveConstantDesc(lookup);
 91             throw new RuntimeException("unexpected IAE not thrown");
 92         } catch (IllegalAccessException e) { }
 93     }
 94 
 95     private static void throwACC(Lookup lookup, MethodTypeDesc mtd) throws Exception {
 96         try {
 97             MethodType mtype = (MethodType)mtd.resolveConstantDesc(lookup);
 98             throw new RuntimeException("unexpected IAE not thrown");
 99         } catch (AccessControlException e) {
100             Permission perm = e.getPermission();
101             if (!(perm instanceof RuntimePermission &&
102                   "accessClassInPackage.jdk.internal.misc".equals(perm.getName()))) {
103                 throw e;
104             }
105         }
106     }
107 }
