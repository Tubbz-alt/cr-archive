24 /*
25  * @bug      6829503
26  * @summary  1) Test Console and DeleteOnExitHook can be initialized
27  *              while shutdown is in progress
28  *           2) Test if files that are added by the application shutdown
29  *              hook are deleted on exit during shutdown
30  */
31 import java.io.*;
32 public class ShutdownHooks {
33     private static File file;
34     public static void main(String[] args) throws Exception {
35         if (args.length != 2) {
36             throw new IllegalArgumentException("Usage: ShutdownHooks <dir> <filename>");
37         }
38 
39         // Add a shutdown hook
40         Runtime.getRuntime().addShutdownHook(new Cleaner());
41 
42         File dir = new File(args[0]);
43         file = new File(dir, args[1]);
44         // write to file
45         System.out.println("writing to "+ file);
46         try (PrintWriter pw = new PrintWriter(file)) {
47             pw.println("Shutdown begins");
48         }
49     }
50 
51     public static class Cleaner extends Thread {
52         public void run() {
53             // register the Console's shutdown hook while the application
54             // shutdown hook is running
55             Console cons = System.console();
56             // register the DeleteOnExitHook while the application
57             // shutdown hook is running
58             file.deleteOnExit();
59             try (PrintWriter pw = new PrintWriter(file)) {
60                 pw.println("file is being deleted");
61             } catch (FileNotFoundException e) {
62                 throw new RuntimeException(e);
63             }
64         }
65     }
66 
67 }
