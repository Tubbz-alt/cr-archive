24 /**
25  * @test
26  * @bug 4637504 4653814
27  * @summary Destroy should close stderr, stdout and stdin
28  * @author kladko
29  */
30 
31 public class LotsOfDestroys {
32     static final int RUNS = 400;
33 
34     public static void main(String[] args) throws Exception {
35         if (! UnixCommands.isUnix) {
36             System.out.println("For UNIX only");
37             return;
38         }
39         UnixCommands.ensureCommandsAvailable("echo");
40 
41         for (int i = 0; i <= RUNS; i++) {
42             Process process = Runtime.getRuntime().exec(
43                     UnixCommands.echo() + " x");
44             process.destroy();
45         }
46     }
47 }
