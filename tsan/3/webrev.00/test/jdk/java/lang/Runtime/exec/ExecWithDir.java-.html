24 /* @test
25  * @bug 4750978
26  * @run main/othervm/timeout=300 ExecWithDir
27  * @summary Ensure that we can fork-and-exec repeatedly when a new working
28  *          directory is specified
29  */
30 
31 import java.io.File;
32 
33 public class ExecWithDir {
34 
35     private static final int N = 500;
36 
37     public static void main(String args[]) throws Exception {
38         if (! UnixCommands.isUnix) {
39             System.out.println("For UNIX only");
40             return;
41         }
42         UnixCommands.ensureCommandsAvailable("true");
43 
44         final String trueCmd = UnixCommands.findCommand("true");
45         File dir = new File(".");
46         for (int i = 1; i <= N; i++) {
47             System.out.print(i);
48             System.out.print(" e");
49             Process p = Runtime.getRuntime().exec(trueCmd, null, dir);
50             System.out.print('w');
51             int s = p.waitFor();
52             System.out.println("x " + s);
53             if (s != 0) throw new Error("Unexpected return code " + s);
54 
55             // Avoid "Too many open files"
56             p.getInputStream().close();
57             p.getOutputStream().close();
58             p.getErrorStream().close();
59         }
60     }
61 }
