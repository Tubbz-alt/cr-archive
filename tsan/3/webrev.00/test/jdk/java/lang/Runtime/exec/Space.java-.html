24 /* @test 1.4 00/10/18
25    @bug 4231349
26    @summary test runtime.exec on windows for extra space in cmd
27  */
28 import java.io.*;
29 
30 public class Space {
31     public static void main(String[] args) throws Exception {
32         if (File.separatorChar == '\\') {
33             try {
34             Process p = Runtime.getRuntime().exec( "cmd /c echo hello" );
35             BufferedReader reader = new BufferedReader(
36                                     new InputStreamReader(p.getInputStream()));
37             p.waitFor();
38             String echo = reader.readLine();
39             if (echo.length() == 6)
40                 throw new RuntimeException("Extra space in command.");
41             } catch (IOException e) {
42             // not Win NT - cmd doesnt exist
43             return;
44             }
45         }
46     }
47 }
