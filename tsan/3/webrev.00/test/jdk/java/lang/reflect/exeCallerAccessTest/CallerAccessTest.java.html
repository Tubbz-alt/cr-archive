25 /**
26  * @test
27  * @bug 8221530
28  * @summary Test uses custom launcher that starts VM using JNI that verifies
29  *          reflection API with null caller class
30  * @library /test/lib
31  * @requires os.family != "aix"
32  * @run main/native CallerAccessTest
33  */
34 
35 // Test disabled on AIX since we cannot invoke the JVM on the primordial thread.
36 
37 import java.io.File;
38 import java.util.Map;
39 import jdk.test.lib.Platform;
40 import jdk.test.lib.Utils;
41 import jdk.test.lib.process.OutputAnalyzer;
42 
43 import java.io.IOException;
44 import java.nio.file.Path;
45 import java.nio.file.Paths;
46 
47 public class CallerAccessTest {
48     public static void main(String[] args) throws IOException {
49         Path launcher = Paths.get(System.getProperty("test.nativepath"), "CallerAccessTest");
50         ProcessBuilder pb = new ProcessBuilder(launcher.toString());
51         Map<String, String> env = pb.environment();
52 
53         String libName = Platform.isWindows() ? "bin" : "lib";
54         Path libPath = Paths.get(Utils.TEST_JDK).resolve(libName);
55         String libDir = libPath.toAbsolutePath().toString();
56         String serverDir = libPath.resolve("server").toAbsolutePath().toString();
57 
58         // set up shared library path
59         String sharedLibraryPathEnvName = Platform.sharedLibraryPathVariableName();
60         env.compute(sharedLibraryPathEnvName,
61                     (k, v) -> (v == null) ? libDir : v + File.pathSeparator + libDir);
62         env.compute(sharedLibraryPathEnvName,
63                     (k, v) -> (v == null) ? serverDir : v + File.pathSeparator + serverDir);
64 
65         System.out.println("Launching: " + launcher + " shared library path: " +
66                            env.get(sharedLibraryPathEnvName));
67         new OutputAnalyzer(pb.start()).shouldHaveExitValue(0);
68     }
69 }
