 24 /**
 25  * @test
 26  * @build TrySetAccessibleTest
 27  * @modules java.base/java.lang:open
 28  *          java.base/jdk.internal.module
 29  *          java.base/jdk.internal.perf
 30  *          java.base/jdk.internal.misc:+open
 31  * @run testng/othervm --illegal-access=deny TrySetAccessibleTest
 32  * @summary Test AccessibleObject::trySetAccessible method
 33  */
 34 
 35 import java.lang.reflect.AccessibleObject;
 36 import java.lang.reflect.Constructor;
 37 import java.lang.reflect.Field;
 38 import java.lang.reflect.InvocationTargetException;
 39 import java.lang.reflect.Method;
 40 
 41 import jdk.internal.misc.Unsafe;
 42 import jdk.internal.module.ModulePath;
 43 import jdk.internal.perf.Perf;
 44 
 45 import org.testng.annotations.Test;
 46 import static org.testng.Assert.*;
 47 
 48 @Test
 49 public class TrySetAccessibleTest {
 50     /**
 51      * Invoke a private constructor on a public class in an exported package
 52      */
 53     public void testPrivateConstructorInExportedPackage() throws Exception {
 54         Constructor<?> ctor = Perf.class.getDeclaredConstructor();
 55 
 56         try {
 57             ctor.newInstance();
 58             assertTrue(false);
 59         } catch (IllegalAccessException expected) { }
 60 
 61         assertFalse(ctor.trySetAccessible());
 62         assertFalse(ctor.canAccess(null));
 63     }
 64 
 65     /**
 66      * Invoke a private constructor on a public class in an open package
 67      */
 68     public void testPrivateConstructorInOpenedPackage() throws Exception {
 69         Constructor<?> ctor = Unsafe.class.getDeclaredConstructor();
 70 
 71         try {
 72             ctor.newInstance();
 73             assertTrue(false);
 74         } catch (IllegalAccessException expected) { }
 75 
 76         assertTrue(ctor.trySetAccessible());
 77         assertTrue(ctor.canAccess(null));
 78         Unsafe unsafe = (Unsafe) ctor.newInstance();
 79     }
 80 
 81     /**
 82      * Invoke a private method on a public class in an exported package
 83      */
 84     public void testPrivateMethodInExportedPackage() throws Exception {
 85         Method m = ModulePath.class.getDeclaredMethod("packageName", String.class);
 86         try {
 87             m.invoke(null);
 88             assertTrue(false);
 89         } catch (IllegalAccessException expected) { }
 90 
 91         assertFalse(m.trySetAccessible());
 92         assertFalse(m.canAccess(null));
 93     }
 94 
 95 
 96     /**
 97      * Invoke a private method on a public class in an open package
 98      */
 99     public void testPrivateMethodInOpenedPackage() throws Exception {
100         Method m = Unsafe.class.getDeclaredMethod("throwIllegalAccessError");
101         assertFalse(m.canAccess(null));
102 
103         try {
104             m.invoke(null);
105             assertTrue(false);
106         } catch (IllegalAccessException expected) { }
107 
108         assertTrue(m.trySetAccessible());
109         assertTrue(m.canAccess(null));
110         try {
111             m.invoke(null);
112             assertTrue(false);
113         } catch (InvocationTargetException e) {
114             // thrown by throwIllegalAccessError
115             assertTrue(e.getCause() instanceof IllegalAccessError);
116         }
117     }
118 
119     /**
120      * Invoke a private method on a public class in an exported package
121      */
122     public void testPrivateFieldInExportedPackage() throws Exception {
123         Field f = Perf.class.getDeclaredField("instance");
124         try {
125             f.get(null);
126             assertTrue(false);
127         } catch (IllegalAccessException expected) { }
128 
129         assertFalse(f.trySetAccessible());
130         assertFalse(f.canAccess(null));
131         try {
132             f.get(null);
133             assertTrue(false);
134         } catch (IllegalAccessException expected) {}
135     }
136 
137     /**
138      * Access a private field in a public class that is an exported package
139      */
140     public void testPrivateFieldInOpenedPackage() throws Exception {
141         Field f = Unsafe.class.getDeclaredField("theUnsafe");
142 
143         try {
144             f.get(null);
145             assertTrue(false);
146         } catch (IllegalAccessException expected) { }
147 
148         assertTrue(f.trySetAccessible());
149         assertTrue(f.canAccess(null));
150         Unsafe unsafe = (Unsafe) f.get(null);
151     }
152 
153 
154     /**
155      * Invoke a public constructor on a public class in a non-exported package
156      */
157     public void testPublicConstructorInNonExportedPackage() throws Exception {
158         Class<?> clazz = Class.forName("sun.security.x509.X500Name");
159         Constructor<?> ctor = clazz.getConstructor(String.class);
160 
161         try {
162             ctor.newInstance("cn=duke");
163             assertTrue(false);
164         } catch (IllegalAccessException expected) { }
165 
166         assertFalse(ctor.trySetAccessible());
167         assertFalse(ctor.canAccess(null));
168         assertTrue(ctor.trySetAccessible() == ctor.isAccessible());
169     }
170 
171 
172     /**
173      * Access a public field in a public class that in a non-exported package
174      */
175     public void testPublicFieldInNonExportedPackage() throws Exception {
176         Class<?> clazz = Class.forName("sun.security.x509.X500Name");
177         Field f = clazz.getField("SERIALNUMBER_OID");
178 
179         try {
180             f.get(null);
181             assertTrue(false);
182         } catch (IllegalAccessException expected) { }
183 
184         assertFalse(f.trySetAccessible());
185         assertFalse(f.canAccess(null));
186     }
187 
188 
189     /**
190      * Test that the Class constructor cannot be make accessible.
191      */
192     public void testJavaLangClass() throws Exception {
193 
194         // non-public constructor
195         Constructor<?> ctor
196             = Class.class.getDeclaredConstructor(ClassLoader.class, Class.class);
197         AccessibleObject[] ctors = { ctor };
198 
199         assertFalse(ctor.trySetAccessible());
200         assertFalse(ctor.canAccess(null));
201     }
202 
203 }
