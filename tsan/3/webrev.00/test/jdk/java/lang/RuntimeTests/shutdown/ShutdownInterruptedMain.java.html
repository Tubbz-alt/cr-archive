24 /*
25  * @test
26  * @bug 8154017
27  * @library /test/lib
28  * @summary Shutdown hooks are racing against shutdown sequence,
29             if System.exit()-calling thread is interrupted
30  * @run main ShutdownInterruptedMain exec
31  */
32 
33 import jdk.test.lib.process.OutputAnalyzer;
34 import static jdk.test.lib.process.ProcessTools.createJavaProcessBuilder;
35 import static jdk.test.lib.process.ProcessTools.executeProcess;
36 
37 public class ShutdownInterruptedMain {
38 
39     public static void main(String[] args) throws Exception {
40         if (args.length > 0) {
41             ProcessBuilder pb = createJavaProcessBuilder(true, "ShutdownInterruptedMain");
42             OutputAnalyzer output = executeProcess(pb);
43             output.shouldContain("Shutdown Hook");
44             output.shouldHaveExitValue(0);
45             return;
46         }
47 
48         Runtime.getRuntime().addShutdownHook(new Thread() {
49             @Override
50             public void run() {
51                 // Wait for the race to unfold:
52                 try {
53                     Thread.sleep(5_000);
54                 } catch (InterruptedException e) {}
55                 System.out.println("Shutdown Hook");
56                 System.out.flush();
57             }
58         });
59         Thread.currentThread().interrupt();
60         System.exit(0);
61     }
62 }
