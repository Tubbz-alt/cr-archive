24 /**
25  * @test
26  * @bug 4369826 8078582 8190884
27  * @run main/othervm LotsOfOutput
28  * @summary Process with lots of output should not crash VM
29  * @author kladko
30  */
31 
32 public class LotsOfOutput {
33     static final Runtime runtime = Runtime.getRuntime();
34 
35     // Allow memory to grow by up to 1Mb total
36     static final int THRESHOLD = 1048576;
37 
38     // Compute used memory
39     static long usedMemory() {
40         return runtime.totalMemory() - runtime.freeMemory();
41     }
42 
43     public static void main(String[] args) throws Exception {
44         if (! UnixCommands.isUnix) {
45             System.out.println("For UNIX only");
46             return;
47         }
48         UnixCommands.ensureCommandsAvailable("cat");
49 
50         Process p = runtime.exec(UnixCommands.cat() + " /dev/zero");
51         long prev = usedMemory();
52         int growing = 0;
53         for (int i = 1; i < 10; i++) {
54             Thread.sleep(100);
55             long used = usedMemory();
56             if (used != prev) {
57                 System.out.printf("consuming memory: i: %d, prev: %d, used: %d, delta: %d%n",
58                         i, prev, used, used - prev);
59             }
60             if (used > prev + THRESHOLD)
61                 growing += 1;
62             prev = used;
63         }
64         if (growing > 2)
65             throw new Exception("Process consumes memory: growing " + growing);
66 
67     }
68 }
