24 /* @test
25    @bug 4925610
26    @summary Check for the appropriate exceptions when a bad envp is passed.
27    @author Martin Buchholz
28 */
29 
30 public class BadEnvp {
31 
32     public static void main(String[] args) throws Exception {
33         Runtime r = Runtime.getRuntime();
34         java.io.File dir = new java.io.File(".");
35 
36         String[] envpWithNull = {"FOO=BAR",null};
37         try {
38             r.exec("echo", envpWithNull);
39             throw new Exception("Expected NullPointerException not thrown");
40         } catch (NullPointerException e) {} // OK
41 
42         try {
43             r.exec("echo", envpWithNull, dir);
44             throw new Exception("Expected NullPointerException not thrown");
45         } catch (NullPointerException e) {} // OK
46 
47         try {
48             r.exec(new String[]{"echo"}, envpWithNull);
49             throw new Exception("Expected NullPointerException not thrown");
50         } catch (NullPointerException e) {} // OK
51 
52         try {
53             r.exec(new String[]{"echo"}, envpWithNull, dir);
54             throw new Exception("Expected NullPointerException not thrown");
55         } catch (NullPointerException e) {} // OK
56     }
57 }
