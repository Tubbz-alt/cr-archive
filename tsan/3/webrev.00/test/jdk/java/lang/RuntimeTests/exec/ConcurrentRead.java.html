24 /**
25  * @test
26  * @bug 4619744
27  * @summary Test that Process input/out can be concurrently read/written
28  * @author kladko
29  */
30 
31 import java.io.InputStream;
32 import java.io.OutputStream;
33 
34 public class ConcurrentRead {
35 
36     static volatile Exception savedException;
37 
38     public static void main(String[] args) throws Exception {
39         if (! UnixCommands.isUnix) {
40             System.out.println("For UNIX only");
41             return;
42         }
43         UnixCommands.ensureCommandsAvailable("tee");
44 
45         Process p = Runtime.getRuntime().exec(UnixCommands.tee());
46         OutputStream out = p.getOutputStream();
47         InputStream in = p.getInputStream();
48         Thread t1 = new WriterThread(out, in);
49         t1.start();
50         Thread t2 = new WriterThread(out, in);
51         t2.start();
52         t1.join();
53         t2.join();
54         if (savedException != null)
55             throw savedException;
56     }
57 
58     static class WriterThread extends Thread {
59         OutputStream out;
60         InputStream in;
61         WriterThread(OutputStream out, InputStream in) {
62             this.out = out;
63             this.in = in;
64         }
65         public void run(){
66             try {
67                 out.write('a');
68                 out.flush();
69                 if (in.read() == -1) // got end-of-stream
70                     throw new Exception("End of stream in writer thread");
71             } catch (Exception e) {
72                 savedException = e;
73             }
74         }
75     }
76 }
