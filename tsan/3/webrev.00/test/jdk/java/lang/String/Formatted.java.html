24 /*
25  * @test
26  * bug 8203444
27  * @summary Unit tests for instance versions of String#format
28  * @compile --enable-preview -source ${jdk.version} Formatted.java
29  * @run main/othervm --enable-preview Formatted
30  */
31 
32 import java.util.Locale;
33 
34 public class Formatted {
35     public static void main(String[] args) {
36         test1();
37     }
38 
39     /*
40      * Test String#formatted(Object... args) functionality.
41      */
42     static void test1() {
43         check("formatted(Object... args)",
44                 "Test this %s".formatted("and that"),
45                 String.format("Test this %s", "and that"));
46     }
47 
48     static void check(String test, String output, String expected) {
49         if (output != expected && (output == null || !output.equals(expected))) {
50             System.err.println("Testing " + test + ": unexpected result");
51             System.err.println("Output: " + output);
52             System.err.println("Expected: " + expected);
53             throw new RuntimeException();
54         }
55     }
56 }
