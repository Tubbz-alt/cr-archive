26 /* @test
27  * @bug 8229785
28  * @summary Test MethodType.fromMethodDescriptorString with security manager
29  * @run main/othervm test.java.lang.invoke.MethodTypeSecurityManager
30  * @run main/othervm/policy=getclassloader.policy test.java.lang.invoke.MethodTypeSecurityManager access
31  */
32 
33 package test.java.lang.invoke;
34 
35 import java.lang.invoke.MethodType;
36 import java.security.AccessControlException;
37 import java.security.Permission;
38 
39 public class MethodTypeSecurityManager {
40     private static boolean hasClassLoaderAccess;
41     public static void main(String... args) throws Throwable {
42         ClassLoader platformLoader = ClassLoader.getPlatformClassLoader();
43         ClassLoader appLoader = ClassLoader.getSystemClassLoader();
44         hasClassLoaderAccess = args.length == 1 && "access".equals(args[0]);
45 
46         assert hasClassLoaderAccess || System.getSecurityManager() == null;
47         if (!hasClassLoaderAccess) {
48             System.setSecurityManager(new SecurityManager());
49         }
50 
51         // require getClassLoader permission
52         throwACC("()Ljdk/internal/misc/VM;", null);
53         // package access check when app class loader loads the class
54         throwACC("()Ljdk/internal/misc/VM;", appLoader);
55 
56         // if using the platform class loader, no package access check
57         MethodType.fromMethodDescriptorString("()Ljdk/internal/misc/VM;", platformLoader);
58     }
59 
60     private static void throwACC(String desc, ClassLoader loader) {
61         try {
62             MethodType.fromMethodDescriptorString(desc, loader);
63             throw new RuntimeException("should never leak JDK internal class");
64         } catch (AccessControlException e) {
65             System.out.println(e.getMessage());
66             Permission perm = e.getPermission();
67             if (!(perm instanceof RuntimePermission)) throw e;
68             // ACC thrown either no "getClassLoader" permission or no package access
69             switch (perm.getName()) {
70                 case "getClassLoader":
71                     if (!hasClassLoaderAccess) break;
72                 case "accessClassInPackage.jdk.internal.misc":
73                     break;
74                 default:
75                     throw e;
76             }
77         }
78     }
79 }
