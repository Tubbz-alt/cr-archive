31 package test.java.lang.invoke;
32 
33 import java.lang.invoke.MethodHandles;
34 
35 public class FindClassSecurityManager {
36     public static void main(String[] args) throws Throwable {
37         assert null != System.getSecurityManager();
38         Class<?> thisClass = FindClassSecurityManager.class;
39         MethodHandles.Lookup lookup = MethodHandles.lookup();
40         Class<?> lookedUp = lookup.findClass(thisClass.getName());
41         assert thisClass == lookedUp;
42         Class<?> accessed = lookup.accessClass(thisClass);
43         assert thisClass == accessed;
44     }
45 }
