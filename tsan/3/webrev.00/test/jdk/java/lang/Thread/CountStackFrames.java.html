24 /* @test
25  * @bug 8205132
26  * @summary Test Thread.countStackFrames()
27  * @run testng CountStackFrames
28  */
29 
30 import org.testng.annotations.Test;
31 
32 public class CountStackFrames {
33 
34     // current thread
35     @Test(expectedExceptions = UnsupportedOperationException.class)
36     public void testCurrentThread() {
37         Thread.currentThread().countStackFrames();
38     }
39 
40     // unstarted thread
41     @Test(expectedExceptions = UnsupportedOperationException.class)
42     public void testUnstartedThread() {
43         Thread thread = new Thread(() -> { });
44         thread.countStackFrames();
45     }
46 
47     // terminated thread
48     @Test(expectedExceptions = UnsupportedOperationException.class)
49     public void testTerminatedThread() throws Exception {
50         Thread thread = new Thread(() -> { });
51         thread.start();
52         thread.join();
53         thread.countStackFrames();
54     }
55 
56 }
