24 /*
25  * @test
26  * @bug 8067951
27  * @summary Unit test for internal ClassLoader#initializePath().
28  *          Quoted entries should get unquoted on Windows.
29  *          Empty entries should be replaced with dot.
30  * @library /test/lib
31  * @modules java.base/java.lang:open
32  * @build jdk.test.lib.Platform
33  * @run main LibraryPathProperty
34  */
35 
36 import java.lang.reflect.Method;
37 import java.io.File;
38 import java.util.Arrays;
39 import jdk.test.lib.Platform;
40 
41 public class LibraryPathProperty {
42 
43     static final String propName = "test.property.name";
44     static final String SP = File.pathSeparator;
45     static Method method;
46 
47     public static void main(String[] args) throws Throwable {
48         method = ClassLoader.class
49                 .getDeclaredMethod("initializePath",
50                                    String.class);
51         method.setAccessible(true);
52 
53         test("", ".");
54         test(SP, ".", ".");
55         test("a" + SP, "a", ".");
56         test(SP + "b", ".", "b");
57         test("a" + SP + SP + "b", "a", ".", "b");
58 
59         if (Platform.isWindows()) {
60             // on Windows parts of paths may be quoted
61             test("\"\"", ".");
62             test("\"\"" + SP, ".", ".");
63             test(SP + "\"\"", ".", ".");
64             test("a" + SP + "\"b\"" + SP, "a", "b", ".");
65             test(SP + "\"a\"" + SP + SP + "b", ".", "a", ".", "b");
66             test("\"a\"" + SP + "\"b\"", "a", "b");
67             test("\"/a/\"b" + SP + "c", "/a/b", "c");
68             test("\"/a;b\"" + SP + "c", "/a;b", "c");
69             test("\"/a:b\"" + SP + "c", "/a:b", "c");
70             test("\"/a" + SP + "b\"" + SP + "c", "/a" + SP + "b", "c");
71             test("/\"a\"\";\"\"b\"" + SP + "\"c\"", "/a;b", "c");
72             test("/\"a:\"b" + SP + "c", "/a:b", "c");
73         }
74     }
75 
76     static void test(String s, String... expected) throws Throwable {
77         System.setProperty(propName, s);
78         String[] res = (String[])method.invoke(null, propName);
79         if (!Arrays.asList(res).equals(Arrays.asList(expected))) {
80             throw new RuntimeException("Parsing [" + s + "] " +
81                     " result " + Arrays.asList(res) +
82                     " doesn't match " + Arrays.asList(expected));
83         }
84     }
85 }
