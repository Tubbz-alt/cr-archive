24 /* @test
25  * @bug 4405949
26  * @summary Verify that back references are used when writing multiple type
27  *          strings that are equal() to one another.
28  */
29 
30 import java.io.*;
31 
32 public class TypeStringBackRef implements Serializable {
33     private static final long serialVersionUID = 1L;
34 
35     String a, b, c, d, e, f, g;
36 
37     public static void main(String[] args) throws Exception {
38         ByteArrayOutputStream bout = new ByteArrayOutputStream();
39         ObjectOutputStream oout = new ObjectOutputStream(bout);
40         oout.writeObject(ObjectStreamClass.lookup(TypeStringBackRef.class));
41         oout.close();
42         if (bout.size() != 116) {
43             throw new Error("Wrong data length: " + bout.size());
44         }
45     }
46 }
