24 /* @test
25  * @bug 4502808
26  * @summary Verify that if the custom serialization method (i.e., readExternal,
27  *          writeExternal, readObject or writeObject) of an object closes the
28  *          stream it is passed, (de)serialization of that object can still
29  *          complete.
30  */
31 
32 import java.io.*;
33 
34 class A implements Externalizable {

35 
36     public A() {}
37 
38     public void writeExternal(ObjectOutput out) throws IOException {
39         out.writeInt(0);
40         out.close();
41     }
42 
43     public void readExternal(ObjectInput in)
44         throws IOException, ClassNotFoundException
45     {
46         in.readInt();
47         in.close();
48     }
49 }
50 
51 class B implements Serializable {

52 
53     private void writeObject(ObjectOutputStream out) throws IOException {
54         out.defaultWriteObject();
55         out.close();
56     }
57 
58     private void readObject(ObjectInputStream in)
59         throws IOException, ClassNotFoundException
60     {
61         in.defaultReadObject();
62         in.close();
63     }
64 }
65 
66 public class SurvivePrematureClose {
67 
68     public static void main(String[] args) throws Exception {
69         writeRead(new A());
70         writeRead(new B());
71     }
72 
73     static void writeRead(Object obj) throws Exception {
74         ByteArrayOutputStream bout = new ByteArrayOutputStream();
75         ObjectOutputStream oout = new ObjectOutputStream(bout);
76         oout.writeObject(obj);
77         oout.close();
78         ObjectInputStream oin = new ObjectInputStream(
79             new ByteArrayInputStream(bout.toByteArray()));
80         oin.readObject();
81     }
82 }
