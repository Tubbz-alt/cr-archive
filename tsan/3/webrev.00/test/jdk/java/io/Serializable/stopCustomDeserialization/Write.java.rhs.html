24 /* @test
25  * @bug 4663191
26  * @summary Verify that readObject and readObjectNoData methods will not be
27  *          called on an object being deserialized if that object is already
28  *          tagged with a ClassNotFoundException.
29  *
30  * @clean Write Read A B C X
31  * @compile Write.java
32  * @run main Write
33  * @clean Write Read A B C X
34  * @compile Read.java
35  * @run main Read
36  */
37 
38 import java.io.*;
39 
40 class A implements Serializable {
41     private static final long serialVersionUID = 0L;
42     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
43     Object x = new X();
44 }
45 
46 class C extends A {
47     private static final long serialVersionUID = 0L;
48 }
49 
50 class X implements Serializable {
51     private static final long serialVersionUID = 0L;
52 }
53 
54 public class Write {
55     public static void main(String[] args) throws Exception {
56         ObjectOutputStream oout =
57             new ObjectOutputStream(new FileOutputStream("tmp.ser"));
58         oout.writeObject(new C());
59         oout.writeObject("after");
60         oout.close();
61     }
62 }
