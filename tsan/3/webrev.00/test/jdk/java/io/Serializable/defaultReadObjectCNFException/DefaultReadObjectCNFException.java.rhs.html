24 /* @test
25  * @bug 4662327
26  * @summary Verify that ObjectInputStream.defaultReadObject() throws a
27  *          ClassNotFoundException if any of the non-primitive field values it
28  *          reads in are tagged with ClassNotFoundExceptions.
29  */
30 
31 import java.io.*;
32 
33 class Foo implements Serializable {
34     private static final long serialVersionUID = 1L;
35 
36     @SuppressWarnings("serial") /* Incorrect use is being tested */
37     Object obj = new Bar();
38 
39     private void readObject(ObjectInputStream in)
40         throws IOException, ClassNotFoundException
41     {
42         in.defaultReadObject();
43         if (obj == null) {
44             throw new Error(
45                 "ClassNotFoundException masked by defaultReadObject()");
46         }
47     }
48 }
49 
50 class Bar implements Serializable {
51     private static final long serialVersionUID = 1L;
52 }
53 
54 class TestObjectInputStream extends ObjectInputStream {
55     TestObjectInputStream(InputStream in) throws IOException { super(in); }
56 
57     protected Class<?> resolveClass(ObjectStreamClass desc)
58         throws IOException, ClassNotFoundException
59     {
60         if (desc.getName().equals(Bar.class.getName())) {
61             throw new ClassNotFoundException();
62         }
63         return super.resolveClass(desc);
64     }
65 }
66 
67 public class DefaultReadObjectCNFException {
68     public static void main(String[] args) throws Exception {
69         ByteArrayOutputStream bout = new ByteArrayOutputStream();
70         ObjectOutputStream oout = new ObjectOutputStream(bout);
71         oout.writeObject(new Foo());
72         oout.writeObject("after");
73         oout.close();
74         ObjectInputStream oin = new TestObjectInputStream(
75             new ByteArrayInputStream(bout.toByteArray()));
76         try {
77             oin.readObject();
78         } catch (ClassNotFoundException e) {
79             // expected
80         }
81         if (!oin.readObject().equals("after")) {
82             throw new Error("subsequent object corrupted");
83         }
84     }
85 }
