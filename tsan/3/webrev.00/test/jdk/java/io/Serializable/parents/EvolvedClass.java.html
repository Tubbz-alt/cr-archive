28 import java.io.*;
29 
30 public class EvolvedClass {
31     public static void main(String args[]) throws Exception{
32         ASubClass corg = new ASubClass(1, "SerializedByEvolvedClass");
33         ASubClass cnew = null;
34 
35         // Deserialize in to new class object
36         FileInputStream fi = new FileInputStream("parents.ser");
37         try {
38             ObjectInputStream si = new ObjectInputStream(fi);
39             cnew = (ASubClass) si.readObject();
40         } finally {
41             fi.close();
42         }
43 
44         System.out.println("Printing the deserialized class: ");
45         System.out.println();
46         System.out.println(cnew);
47     }
48 }

51 /* During deserialization, the no-arg constructor of a serializable base class
52  * must not be invoked.
53  */
54 class ASuperClass implements Serializable {
55     private static final long serialVersionUID = 1L;
56 
57     String name;
58 
59     ASuperClass() {
60         /*
61          * This method is not to be executed during deserialization for this
62          * example. Must call no-arg constructor of class Object which is the
63          * base class for ASuperClass.
64          */
65         throw new Error("ASuperClass: Wrong no-arg constructor invoked");
66     }
67 
68     ASuperClass(String name) {
69         this.name = new String(name);
70     }
71 
72     public String toString() {
73         return("Name:  " + name);
74     }
75 }
76 
77 class ASubClass extends ASuperClass implements Serializable {
78     int num;
79 
80     private static final long serialVersionUID =6341246181948372513L;
81     ASubClass(int num, String name) {
82         super(name);
83         this.num = num;
84     }
85 
86     public String toString() {
87         return (super.toString() + "\nNum:  " + num);
88     }
89 }
