24 /* @test
25  * @bug 4313167
26  *
27  * @clean Write Read A B C
28  * @compile Write.java
29  * @run main Write
30  * @clean Write Read A B C
31  * @compile Read.java
32  * @run main Read
33  *
34  * @summary Verify that ClassNotFoundExceptions caused by values referenced
35  *          (perhaps transitively) by "skipped" fields will not cause
36  *          deserialization failure.
37  */
38 
39 import java.io.*;
40 
41 class A implements Serializable {
42     private static final long serialVersionUID = 0L;
43     // all three following fields not present on reading side
44     B b = new B();
45     C c = new C();
46     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
47     Object ca = new Object[] { new C() };
48 }
49 
50 class B implements Serializable {
51     private static final long serialVersionUID = 0L;
52     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
53     Object c = new C();
54 }
55 
56 class C implements Serializable {       // class not present on reading side
57     private static final long serialVersionUID = 0L;
58 }
59 
60 public class Write {
61     public static void main(String[] args) throws Exception {
62         ObjectOutputStream oout =
63             new ObjectOutputStream(new FileOutputStream("tmp.ser"));
64         oout.writeObject(new A());
65         oout.close();
66     }
67 }
