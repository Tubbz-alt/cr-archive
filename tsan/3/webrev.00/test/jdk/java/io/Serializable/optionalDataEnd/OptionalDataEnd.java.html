24 /* @test
25  * @bug 4402870
26  * @summary Verify that an OptionalDataException with eof == true is thrown
27  *          when a call to ObjectInputStream.readObject() attempts to read past
28  *          the end of custom data.
29  */
30 
31 import java.io.*;
32 
33 class Foo implements Serializable {
34     private static final long serialVersionUID = 1L;
35 
36     int reps;
37 
38     Foo(int reps) {
39         this.reps = reps;
40     }
41 
42     private void writeObject(ObjectOutputStream out) throws IOException {
43         out.defaultWriteObject();
44         for (int i = 0; i < reps; i++) {
45             out.writeObject(i);
46         }
47     }
48 
49     private void readObject(ObjectInputStream in)
50         throws IOException, ClassNotFoundException
51     {
52         in.defaultReadObject();
53         for (int i = 0; i < reps; i++) {
54             in.readObject();
55         }
56         try {
57             in.readObject();
58             throw new Error();
59         } catch (OptionalDataException ex) {
60             if (! (ex.eof && (ex.length == 0))) {
61                 throw new Error();
62             }
63         }
64     }
65 }
66 
67 public class OptionalDataEnd {
68     public static void main(String[] args) throws Exception {
69         ByteArrayOutputStream bout = new ByteArrayOutputStream();
70         ObjectOutputStream oout = new ObjectOutputStream(bout);
71         oout.writeObject(new Foo(5));
72         oout.writeObject(new Foo(0));
73         oout.close();
74 
75         ObjectInputStream oin = new ObjectInputStream(
76             new ByteArrayInputStream(bout.toByteArray()));
77         oin.readObject();
78         oin.readObject();
79         oin.close();
80     }
81 }
