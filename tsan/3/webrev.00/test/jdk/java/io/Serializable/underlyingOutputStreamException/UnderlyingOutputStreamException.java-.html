24 /* @test
25  * @summary Verify that if ObjectOutputStream's underlying OutputStream throws
26  *          an IOException, the original IOException (not a
27  *          StreamCorruptedException) will be thrown to the writing thread.
28  */
29 
30 import java.io.*;
31 
32 class OriginalIOException extends IOException {
33 }
34 
35 class BrokenOutputStream extends OutputStream {
36     boolean broken = false;
37 
38     public void write(int b) throws IOException {
39         if (broken) {
40             throw new OriginalIOException();
41         }
42     }
43 }
44 
45 public class UnderlyingOutputStreamException {
46     public static void main(String[] args) throws Exception {
47         BrokenOutputStream bout = new BrokenOutputStream();
48         ObjectOutputStream oout = new ObjectOutputStream(bout);
49         bout.broken = true;
50         try {
51             oout.writeObject("foo");
52             throw new Error();
53         } catch (OriginalIOException ex) {
54         }
55     }
56 }
