24 /* @test
25  * @bug 4387368
26  * @summary Verify that object whose class declared a serial persistent field
27  *          that does not match any actual field cannot be serialized via
28  *          default serialization.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     private static final ObjectStreamField[] serialPersistentFields = {
35         new ObjectStreamField("nonexistent", int.class)
36     };
37 }
38 
39 class B implements Serializable {
40     private static final ObjectStreamField[] serialPersistentFields = {
41         new ObjectStreamField("mismatched", int.class)
42     };
43     private float mismatched;
44 }
45 
46 class C implements Serializable {
47     private static final ObjectStreamField[] serialPersistentFields = {
48         new ObjectStreamField("existent", int.class)
49     };
50     private int existent;
51 }
52 
53 public class BadSerialPersistentField {
54     public static void main(String[] args) throws Exception {
55         ByteArrayOutputStream bout = new ByteArrayOutputStream();
56         ObjectOutputStream oout;
57 
58         oout = new ObjectOutputStream(bout);
59         try {
60             oout.writeObject(new A());
61             throw new Error();
62         } catch (InvalidClassException ex) {
63         }
64 
65         oout = new ObjectOutputStream(bout);
66         try {
67             oout.writeObject(new B());
68             throw new Error();
69         } catch (InvalidClassException ex) {
70         }
71 
72         oout = new ObjectOutputStream(bout);
73         oout.writeObject(new C());
74     }
75 }
