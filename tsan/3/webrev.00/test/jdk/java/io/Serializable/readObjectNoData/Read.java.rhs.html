 24 /*
 25  * @bug 4363844
 26  * @summary Verify that a custom readObjectNoData method, if defined properly
 27  *          by a serializable superclass, gets invoked during deserialization
 28  *          of a subclass instance whose serialized form omits a class
 29  *          descriptor corresponding to the superclass.
 30  */
 31 
 32 import java.io.*;
 33 
 34 /* Non-serializable superclass which defines readObjectNoData:
 35  * readObjectNoData should not get called.
 36  */
 37 class A {
 38     private static final long serialVersionUID = 0L;
 39     boolean aCalled = false;
 40     private void readObjectNoData() throws ObjectStreamException {
 41         aCalled = true;
 42     }
 43 }
 44 
 45 /* Serializable superclass which defines readObjectNoData with wrong signature:
 46  * readObjectNoData should not get called.
 47  */
 48 class B extends A implements Serializable {
 49     private static final long serialVersionUID = 0L;
 50     boolean bCalled = false;
 51     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
 52     private void readObjectNoData(int wrong) throws ObjectStreamException {
 53         bCalled = true;
 54     }
 55 }
 56 
 57 /* Serializable superclass which defines readObjectNoData correctly, and is not
 58  * listed in stream: readObjectNoData should get called.
 59  */
 60 class C extends B {
 61     private static final long serialVersionUID = 0L;
 62     boolean cCalled = false;
 63     private void readObjectNoData() throws ObjectStreamException {
 64         cCalled = true;
 65     }
 66 }
 67 
 68 /* Serializable superclass which defines readObjectNoData correctly, but whose
 69  * corresponding class descriptor is listed in stream: readObjectNoData should
 70  * not get called.
 71  */
 72 class D extends C {
 73     private static final long serialVersionUID = 0L;
 74     boolean dCalled = false;
 75     private void readObjectNoData() throws ObjectStreamException {
 76         dCalled = true;
 77     }
 78 }
 79 
 80 /* Serializable superclass which defines readObjectNoData with wrong access:
 81  * readObjectNoData should not get called.
 82  */
 83 class E extends D {
 84     private static final long serialVersionUID = 0L;
 85     boolean eCalled = false;
 86     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
 87     void readObjectNoData() throws ObjectStreamException {
 88         eCalled = true;
 89     }
 90 }
 91 
 92 /* Instance class.
 93  */
 94 class F extends E {
 95     private static final long serialVersionUID = 0L;
 96 }
 97 
 98 public class Read {
 99     public static void main(String[] args) throws Exception {
100         FileInputStream in = new FileInputStream("tmp.ser");
101         try {
102             ObjectInputStream oin = new ObjectInputStream(in);
103             F f = (F) oin.readObject();
104             if (f.aCalled || f.bCalled || f.dCalled || f.eCalled) {
105                 throw new Error("readObjectNoData invoked erroneously");
106             }
107             if (! f.cCalled) {
108                 throw new Error("readObjectNoData not invoked");
109             }
110         } finally {
111             in.close();
112         }
113     }
114 }
