24 /* @test
25  * @bug 4387368
26  * @summary Verify that object whose class declared a serial persistent field
27  *          that does not match any actual field cannot be serialized via
28  *          default serialization.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     private static final long serialVersionUID = 1L;
35 
36     private static final ObjectStreamField[] serialPersistentFields = {
37         new ObjectStreamField("nonexistent", int.class)
38     };
39 }
40 
41 class B implements Serializable {
42     private static final long serialVersionUID = 1L;
43 
44     private static final ObjectStreamField[] serialPersistentFields = {
45         new ObjectStreamField("mismatched", int.class)
46     };
47     private float mismatched;
48 }
49 
50 class C implements Serializable {
51     private static final long serialVersionUID = 1L;
52 
53     private static final ObjectStreamField[] serialPersistentFields = {
54         new ObjectStreamField("existent", int.class)
55     };
56     private int existent;
57 }
58 
59 public class BadSerialPersistentField {
60     public static void main(String[] args) throws Exception {
61         ByteArrayOutputStream bout = new ByteArrayOutputStream();
62         ObjectOutputStream oout;
63 
64         oout = new ObjectOutputStream(bout);
65         try {
66             oout.writeObject(new A());
67             throw new Error();
68         } catch (InvalidClassException ex) {
69         }
70 
71         oout = new ObjectOutputStream(bout);
72         try {
73             oout.writeObject(new B());
74             throw new Error();
75         } catch (InvalidClassException ex) {
76         }
77 
78         oout = new ObjectOutputStream(bout);
79         oout.writeObject(new C());
80     }
81 }
