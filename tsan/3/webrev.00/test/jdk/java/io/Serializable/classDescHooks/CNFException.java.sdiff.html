33 class CNFInputStream extends ObjectInputStream {
34 
35     CNFInputStream(InputStream in) throws IOException {
36         super(in);
37     }
38 
39     protected ObjectStreamClass readClassDescriptor()
40         throws IOException, ClassNotFoundException
41     {
42         throw new ClassNotFoundException("foobar");
43     }
44 }
45 
46 public class CNFException {
47     public static void main(String[] args) throws Exception {
48         ByteArrayOutputStream bout = new ByteArrayOutputStream();
49         ObjectOutputStream oout = new ObjectOutputStream(bout);
50         oout.writeObject(new Integer(5));
51         oout.close();
52         ObjectInputStream oin =
53             new CNFInputStream(new ByteArrayInputStream(bout.toByteArray()));
54         try {
55             oin.readObject();
56             throw new Error("expected InvalidClassException");
57         } catch (InvalidClassException e) {
58             Throwable cause = e.getCause();
59             if (!(cause instanceof ClassNotFoundException)) {
60                 throw new Error(
61                     "expected ClassNotFoundException as cause, not " + cause);
62             }
63         }
64     }
65 }
