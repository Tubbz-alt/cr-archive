24 /* @test
25  * @bug 4337857
26  *
27  * @clean Write Read A B
28  * @compile Write.java
29  * @run main Write
30  * @clean Write Read A B
31  * @compile Read.java
32  * @run main Read
33  *
34  * @summary Verify that custom serialization methods declared with incorrect
35  *          return types are not invoked.
36  */
37 
38 import java.io.*;
39 
40 class B implements Serializable {
41     private static final long serialVersionUID = 0L;
42 
43     static boolean writeObjectCalled;
44     static boolean writeReplaceCalled;
45 
46     @SuppressWarnings("serial") /* Incorrect use is being tested */
47     private Object writeObject(ObjectOutputStream out) throws IOException {
48         writeObjectCalled = true;
49         out.defaultWriteObject();
50         return null;
51     }
52 
53     @SuppressWarnings("serial") /* Incorrect use is being tested */
54     private B writeReplace() throws ObjectStreamException {
55         writeReplaceCalled = true;
56         return this;
57     }
58 }
59 
60 public class Write {
61     public static void main(String[] args) throws Exception {
62         ObjectOutputStream oout =
63             new ObjectOutputStream(new FileOutputStream("tmp.ser"));
64         oout.writeObject(new B());
65         oout.close();
66 
67         if (B.writeObjectCalled) {
68             throw new Error("writeObject with wrong return type called");
69         } else if (B.writeReplaceCalled) {
70             throw new Error("writeReplace with wrong return type called");
71         }
72     }
73 }
