24 /* @test
25  * @summary Test proper functioning of class-defined writeReplace methods.
26  */
27 
28 import java.io.*;
29 
30 public class WriteReplace {
31 
32     static class ReplaceMe implements Serializable {
33         private Object obj;
34         private boolean writeReplaceCalled = false;
35 
36         public ReplaceMe(Object obj) {
37             this.obj = obj;
38         }
39 
40         private Object writeReplace() throws ObjectStreamException {
41             if (writeReplaceCalled) {
42                 throw new Error("multiple calls to writeReplace");
43             }
44             writeReplaceCalled = true;
45             return obj;
46         }
47     }
48 
49     public static void main(String[] args) throws Exception {
50         final int nobjs = 10;
51         final int nrounds = 10;
52 
53         Object common = "foo";
54         ReplaceMe[] objs = new ReplaceMe[nobjs];
55         for (int i = 0; i < nobjs; i++) {
56             objs[i] = new ReplaceMe(common);
57         }
58 
59         ByteArrayOutputStream bout = new ByteArrayOutputStream();
60         ObjectOutputStream oout = new ObjectOutputStream(bout);
61         for (int i = 0; i < nrounds; i++) {
62             for (int j = 0; j < nobjs; j++) {
63                 oout.writeObject(objs[j]);
64             }
65         }
66 
67         oout.flush();
68         ByteArrayInputStream bin =
69             new ByteArrayInputStream(bout.toByteArray());
70         ObjectInputStream oin = new ObjectInputStream(bin);
71         common = null;
72 
73         for (int i = 0; i < nrounds; i++) {
74             for (int j = 0; j < nobjs; j++) {
75                 if (common == null) {
76                     common = oin.readObject();
77                 } else {
78                     if (oin.readObject() != common) {
79                         throw new Error("incorrect replacement object");
80                     }
81                 }
82             }
83         }
84     }
85 }
