36 import java.io.*;
37 
38 class NewExternFieldClass implements Externalizable {
39     byte l;
40 
41     public NewExternFieldClass() {
42         l = 0;
43     }
44 
45     public NewExternFieldClass(byte value) {
46         l = value;
47     }
48 
49     public void readExternal(ObjectInput s)
50         throws IOException, ClassNotFoundException
51     {
52         l = s.readByte();
53         System.out.println("readExternal read " + l);
54     }
55 
56     public void writeExternal(ObjectOutput s) throws IOException
57     {
58         s.writeByte(l);
59     }
60 }
61 
62 class D implements Serializable {
63     public int x;
64     D(int y) {
65         x = y;
66     }
67 }
68 
69 class A implements Serializable  {
70     // Version 1.1 of class A.  Added superclass NewSerializableSuper.
71     private static final long serialVersionUID = 1L;
72     NewExternFieldClass foo;
73     D zoo;
74 
75     int bar;
76     A() {
77         bar = 4;
78         foo = new NewExternFieldClass((byte)66);
79         zoo = new D(22);
80     }
81 }
82 
83 public class WriteAddedField {
84     public static void main(String args[]) throws IOException {
85         A a = new A();
86         File f = new File("tmp.ser");
87         ObjectOutput out =
88             new ObjectOutputStream(new FileOutputStream(f));
89         out.writeObject(a);
90         out.writeObject(new A());
91         out.close();
92     }
93 }
