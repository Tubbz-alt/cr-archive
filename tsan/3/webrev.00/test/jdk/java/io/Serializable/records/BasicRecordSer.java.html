 58     // a mix of a few record and non-record classes
 59 
 60     record Empty () implements Serializable { }
 61 
 62     record Foo (int i) implements Serializable { }
 63 
 64     static class Bar implements Serializable {
 65         final Foo foo;
 66         final long along;
 67         Bar(Foo foo, long along) { this.foo = foo; this.along = along; }
 68         @Override
 69         public boolean equals(Object obj) {
 70             if (!(obj instanceof Bar))
 71                 return false;
 72             Bar other = (Bar)obj;
 73             if (this.foo.equals(other.foo) && this.along == other.along)
 74                 return true;
 75             return false;
 76         }
 77         @Override
 78         public String toString() {
 79             return format("Bar[foo=%s, along=%d]", foo, along);
 80         }
 81     }
 82 
 83     record Baz (Bar bar, float afloat, Foo foo) implements Serializable { }
 84 
 85     record Bat (Empty e1, Foo foo1, Bar bar1, float afloat, Foo foo2, Empty e2, Bar bar2)
 86         implements Serializable { }
 87 
 88     record Cheese<A, B>(A a, B b) implements Serializable { }

 90     interface ThrowingExternalizable extends Externalizable {
 91         default void writeExternal(ObjectOutput out) {
 92             fail("should not reach here");
 93         }
 94         default void readExternal(ObjectInput in) {
 95             fail("should not reach here");
 96         }
 97     }

 99     record Wibble () implements ThrowingExternalizable { }
100 
101     record Wobble (Foo foo) implements ThrowingExternalizable { }
102 
103     record Wubble (Wobble wobble, Wibble wibble, String s) implements ThrowingExternalizable { }

105     @DataProvider(name = "serializable")
106     public Object[][] serializable() {
107         Foo foo = new Foo(23);
108         return new Object[][] {
109             new Object[] { new Empty()                                                       },
110             new Object[] { new Foo(22)                                                       },
111             new Object[] { new Foo[] { new Foo(24), new Foo(25) }                            },

124     /** Tests serializing and deserializing a number of records. */
125     @Test(dataProvider = "serializable")
126     public void testSerializable(Object objToSerialize) throws Exception {
127         out.println("\n---");
128         out.println("serializing : " + objToSerialize);
129         var objDeserialized = serializeDeserialize(objToSerialize);
130         out.println("deserialized: " + objDeserialized);
131         assertEquals(objToSerialize, objDeserialized);
132         assertEquals(objDeserialized, objToSerialize);
133     }

135     /** Tests back references of Serializable record objects in the stream. */
136     @Test
137     public void testSerializableBackRefs() throws Exception {
138         out.println("\n---");
139         Foo foo = new Foo(32);
140         Foo[] objToSerialize = new Foo[] { foo, foo, foo, foo, foo };
141         out.println("serializing : " + objToSerialize);
142         Foo[] objDeserialized = (Foo[])serializeDeserialize(objToSerialize);
143         out.println("deserialized: " + objDeserialized);
144         assertEquals(objToSerialize, objDeserialized);
145         assertEquals(objDeserialized, objToSerialize);
146 
147         for (Foo f : objDeserialized)
148             assertTrue(objDeserialized[0] == f);
149     }

151     /** Tests back references of Externalizable record objects in the stream. */
152     @Test
153     public void testExternalizableBackRefs() throws Exception {
154         out.println("\n---");
155         Foo foo = new Foo(33);
156         Wobble wobble = new Wobble(foo);
157         Wobble[] objToSerialize = new Wobble[] { wobble, wobble, wobble, wobble };
158         out.println("serializing : " + objToSerialize);
159         Wobble[] objDeserialized = (Wobble[])serializeDeserialize(objToSerialize);
160         out.println("deserialized: " + objDeserialized);
161         assertEquals(objToSerialize, objDeserialized);
162         assertEquals(objDeserialized, objToSerialize);
163 
164         for (Wobble w : objDeserialized) {
165             assertTrue(objDeserialized[0] == w);
166             assertTrue(objDeserialized[0].foo() == w.foo());
167         }
168     }

196     static final Class<NotSerializableException> NSE = NotSerializableException.class;

198     /** Tests that non-Serializable record objects throw NotSerializableException. */

200     public void testNotSerializable(Object objToSerialize) throws Exception {
201         out.println("\n---");
202         out.println("serializing : " + objToSerialize);
203         NotSerializableException expected = expectThrows(NSE, () -> serialize(objToSerialize));
204         out.println("caught expected NSE:" + expected);

206     }

208     // --- constructor invocation counting
209 
210     static volatile int e_ctrInvocationCount;
211 
212     record E () implements Serializable {
213         public E() {  e_ctrInvocationCount++; }
214     }
215 
216     /** Tests that the record's constructor is invoke exactly once per deserialization. */
217     @Test
218     public void testCtrCalledOnlyOnce() throws Exception {
219         out.println("\n---");
220         var objToSerialize = new E();
221         e_ctrInvocationCount = 0;  // reset
222         out.println("serializing : " + objToSerialize);
223         var objDeserialized = serializeDeserialize(objToSerialize);
224         out.println("deserialized: " + objDeserialized);
225         assertEquals(objToSerialize, objDeserialized);
226         assertEquals(objDeserialized, objToSerialize);
227         assertEquals(e_ctrInvocationCount, 1);
228     }

232     static volatile int g_ctrInvocationCount;
233 
234     record F (int x){
235         public F(int x) {  this.x = x; g_ctrInvocationCount++; }
236     }
237     static class G implements Serializable {
238         F f = new F(89);
239     }
240 
241     /** Tests that the record's constructor is NOT invoke during failed deserialization. */
242     @Test
243     public void testCtrNotCalled() {
244         out.println("\n---");
245         var objToSerialize = new G();
246         g_ctrInvocationCount = 0;  // reset
247         out.println("serializing : " + objToSerialize);
248         NotSerializableException expected = expectThrows(NSE, () -> serialize(objToSerialize));
249         out.println("caught expected NSE:" + expected);
250         assertEquals(g_ctrInvocationCount, 0);
251     }

255     static <T> byte[] serialize(T obj) throws IOException {
256         ByteArrayOutputStream baos = new ByteArrayOutputStream();
257         ObjectOutputStream oos = new ObjectOutputStream(baos);
258         oos.writeObject(obj);
259         oos.close();
260         return baos.toByteArray();
261     }
262 
263     @SuppressWarnings("unchecked")
264     static <T> T deserialize(byte[] streamBytes)
265         throws IOException, ClassNotFoundException
266     {
267         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
268         ObjectInputStream ois  = new ObjectInputStream(bais);
269         return (T) ois.readObject();
270     }
271 
272     static <T> T serializeDeserialize(T obj)
273         throws IOException, ClassNotFoundException
274     {
275         return deserialize(serialize(obj));
276     }
