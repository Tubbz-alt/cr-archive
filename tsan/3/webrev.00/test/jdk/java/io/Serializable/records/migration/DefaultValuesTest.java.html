 24 /*
 25  * @test
 26  * @summary Checks that the appropriate default value is given to the canonical ctr
 27  * @library /test/lib
 28  * @modules jdk.compiler
 29  * @compile --enable-preview -source ${jdk.version} AssignableFrom.java Point.java
 30  *          DefaultValues.java SuperStreamFields.java DefaultValuesTest.java
 31  * @run testng/othervm --enable-preview DefaultValuesTest
 32  */
 33 
 34 import java.io.ByteArrayOutputStream;
 35 import java.io.DataOutputStream;
 36 import org.testng.annotations.Test;
 37 import static java.io.ObjectStreamConstants.*;
 38 import static java.lang.System.out;
 39 import static org.testng.Assert.*;
 40 
 41 /**
 42  * Basic test to check that default primitive / reference values are
 43  * presented to the record's canonical constructor, for fields not in
 44  * the stream.
 45  */
 46 public class DefaultValuesTest extends AbstractTest {
 47 
 48     /**
 49      * Returns a stream of bytes for the given class, uid, flags. The
 50      * stream will have no stream field values.
 51      */
 52     static byte[] byteStreamFor(String className, long uid, byte flags)
 53         throws Exception
 54     {
 55         ByteArrayOutputStream baos = new ByteArrayOutputStream();
 56         DataOutputStream dos = new DataOutputStream(baos);
 57         dos.writeShort(STREAM_MAGIC);
 58         dos.writeShort(STREAM_VERSION);
 59         dos.writeByte(TC_OBJECT);
 60         dos.writeByte(TC_CLASSDESC);
 61         dos.writeUTF(className);
 62         dos.writeLong(uid);
 63         dos.writeByte(flags);
 64         dos.writeShort(0);             // number of fields
 65         dos.writeByte(TC_ENDBLOCKDATA);   // no annotations
 66         dos.writeByte(TC_NULL);           // no superclasses
 67         dos.close();
 68         return baos.toByteArray();
 69     }
 70 
 71     /** Basic test with a simple Point. */
 72     @Test
 73     public void testPoint() throws Exception {
 74         out.println("--- testPoint ");
 75 
 76         byte[] bytes = byteStreamFor("PointImpl", 5L, (byte)SC_SERIALIZABLE);
 77 
 78         Point point = deserializeAsPlain(bytes);
 79         out.println("deserialized: " + point);
 80         assertEquals(point.x(), 0);
 81         assertEquals(point.y(), 0);
 82 
 83         point = deserializeAsRecord(bytes);
 84         out.println("deserialized: " + point);
 85         assertEquals(point.x(), 0);
 86         assertEquals(point.y(), 0);
 87     }
 88 
 89     // ---
 90 
 91     static class Defaults {  // default values
 92         static boolean bool;
 93         static byte by;
 94         static char ch;
 95         static short sh;
 96         static int i;
 97         static long l;
 98         static float f;
 99         static double d;
100         static String str;
101         static Object obj;
102         static int[] ia;
103         static Object[] oa;
104     }
105 
106     /** A more comprehensive test with all primitives, reference, and array. */
107     @Test
108     public void testAllDefaults() throws Exception {
109         out.println("--- testAllDefaults ");
110 
111         Point point = newPlainPoint(43, 98);
112         DefaultValues o = newPlainDefaultValues(point);
113         out.println("serialize   :" + o);
114         byte[] bytes = serialize(o);
115         DefaultValues o1 = deserializeAsRecord(bytes);
116         out.println("deserialized: " + o1);
117 
118         assertEquals(o1.point().x(), point.x());   // sanity
119         assertEquals(o1.point().y(), point.y());   // sanity
120         assertTrue(o1.bool() == Defaults.bool);
121         assertTrue(o1.by() == Defaults.by);
122         assertTrue(o1.ch() == Defaults.ch);
123         assertTrue(o1.sh() == Defaults.sh);
124         assertTrue(o1.i() == Defaults.i);
125         assertTrue(o1.l() == Defaults.l);
126         assertTrue(o1.f() == Defaults.f);
127         assertTrue(o1.d() == Defaults.d);
128         assertTrue(o1.str() == Defaults.str);
129         assertTrue(o1.obj() == Defaults.obj);
130         assertTrue(o1.ia() == Defaults.ia);
131         assertTrue(o1.oa() == Defaults.oa);
132     }
133 }
