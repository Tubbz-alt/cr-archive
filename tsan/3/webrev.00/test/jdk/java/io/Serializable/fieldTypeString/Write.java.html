24 /* @test
25  * @bug 4404696
26  *
27  * @clean Write Read Foo Bar
28  * @compile Write.java
29  * @run main Write
30  * @clean Write Read Foo Bar
31  * @compile Read.java
32  * @run main Read
33  *
34  * @summary Verify that serialization does not require matching type strings
35  *          for non-primitive fields.
36  *
37  * NOTE: This test should be removed if it is determined that serialization
38  * *should* consider type strings when matching non-primitive fields.
39  */
40 
41 import java.io.*;
42 
43 class Foo implements Serializable {
44     private static final long serialVersionUID = 0L;
45     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
46     Object obj;
47 
48     Foo(Object obj) {
49         this.obj = obj;
50     }
51 }
52 
53 class Bar implements Serializable {
54     private static final long serialVersionUID = 0L;
55     int q;
56 }
57 
58 public class Write {
59     public static void main(String[] args) throws Exception {
60         ObjectOutputStream oout =
61             new ObjectOutputStream(new FileOutputStream("foo.ser"));
62         oout.writeObject(new Foo("foo"));
63         oout.writeObject(new Foo(0));
64         oout.close();
65 
66         oout = new ObjectOutputStream(new FileOutputStream("bar.ser"));
67         oout.writeObject(new Bar());
68         oout.close();
69     }
70 }
