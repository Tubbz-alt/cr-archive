 24 /*
 25  * @test
 26  * @summary Ensures basic behavior of cycles from record components
 27  * @compile --enable-preview -source ${jdk.version} CycleTest.java
 28  * @run testng/othervm --enable-preview CycleTest
 29  * @run testng/othervm/java.security.policy=empty_security.policy --enable-preview CycleTest
 30  */

 32 import java.io.ByteArrayInputStream;
 33 import java.io.ByteArrayOutputStream;
 34 import java.io.IOException;
 35 import java.io.ObjectInputStream;
 36 import java.io.ObjectOutputStream;
 37 import java.io.Serializable;
 38 import org.testng.annotations.Test;
 39 import static java.lang.System.out;
 40 import static org.testng.Assert.assertEquals;
 41 import static org.testng.Assert.assertTrue;
 42 
 43 public class CycleTest {
 44 
 45     record R (int x, int y, C c) implements Serializable { }
 46 
 47     static class C implements Serializable {
 48         Object obj;  // mutable
 49     }
 50 
 51     /**
 52      * Deserialization of a record object, r, does not support references to r,
 53      * from any of r's components. All references will be null.
 54      */
 55     @Test
 56     public void testCycle1() throws Exception  {
 57         out.println("\n---");
 58         C c = new C();
 59         R r = new R(1, 2, c);
 60         c.obj = r;  // cycle, targeting record r
 61 
 62         out.println("serializing : " + r);
 63         R deserializedObj = serializeDeserialize(r);
 64         out.println("deserialized: " + deserializedObj);
 65         assertEquals(deserializedObj.x(), 1);           // sanity
 66         assertEquals(deserializedObj.y(), 2);           // sanity
 67         assertTrue(deserializedObj.c() instanceof C);   // sanity
 68         assertEquals(deserializedObj.c().obj, null);    // cycle, expect null
 69     }
 70 
 71     /**
 72      * An object, c, reconstructed before the record, r, can be referenced from
 73      * a record component. It's a cycle, from within the record, but the record,
 74      * r, is not the target.
 75      */
 76     @Test
 77     public void testCycle2() throws Exception  {
 78         out.println("\n---");
 79         C c = new C();
 80         R r = new R(3, 4, c);
 81         c.obj = r;  // cycle, serializing c first should preserve the cycle
 82 
 83         out.println("serializing : " + c);
 84         C deserializedObj = serializeDeserialize(c);
 85         out.println("deserialized: " + deserializedObj);
 86         assertTrue(deserializedObj instanceof C);         // sanity
 87         assertTrue(deserializedObj.obj != null);          // expect non-null, r
 88         assertEquals(((R)deserializedObj.obj).x(), 3);    // sanity
 89         assertEquals(((R)deserializedObj.obj).y(), 4);    // sanity
 90     }
 91 
 92     record R2 (int x, int y, C c1, C c2) implements Serializable { }
 93 
 94     /**
 95      * Cycles, of non-record objects, within record components should be fine.
 96      */
 97     @Test
 98     public void testCycle3() throws Exception {
 99         out.println("\n---");
100         C c1 = new C();
101         C c2 = new C();
102         c1.obj = c2;  // --\-- cycle
103         c2.obj = c1;  //    \- cycle
104         R2 r = new R2(5, 6, c1, c2);
105 
106         out.println("serializing : " + r);
107         R2 deserializedObj = serializeDeserialize(r);
108         out.println("deserialized: " + deserializedObj);
109         assertEquals(deserializedObj.x(), 5);         // sanity
110         assertEquals(deserializedObj.y(), 6);         // sanity
111 
112         c1 = deserializedObj.c1();
113         c2 = deserializedObj.c2();
114         assertTrue(c1.obj == c2);
115         assertTrue(c2.obj == c1);
116     }
117 
118     record R3 (long l, R r) implements Serializable { }
119 
120     /**
121      * Deserialization of a record object, r, ( with a cycle ), should still
122      * deserialize from within a record, r3.
123      */
124     @Test
125     public void testCycle4() throws Exception  {
126         out.println("\n---");
127         C c = new C();
128         R r = new R(7, 8, c);
129         c.obj = r;  // cycle, targeting record r
130         R3 r3 = new R3(9, r); // record within a record
131 
132         out.println("serializing : " + r3);
133         R3 deserializedObj = serializeDeserialize(r3);
134         out.println("deserialized: " + deserializedObj);
135         assertTrue(deserializedObj.r() != null);
136         assertEquals(deserializedObj.l(), 9);              // sanity
137         assertEquals(deserializedObj.r().x(), 7);          // sanity
138         assertEquals(deserializedObj.r().y(), 8);          // sanity
139         assertTrue(deserializedObj.r().c() instanceof C);  // sanity
140         assertEquals(deserializedObj.r().c().obj, null);   // cycle, expect null
141     }
142 
143     // --- infra
144 
145     static <T> byte[] serialize(T obj) throws IOException {
146         ByteArrayOutputStream baos = new ByteArrayOutputStream();
147         ObjectOutputStream oos = new ObjectOutputStream(baos);
148         oos.writeObject(obj);
149         oos.close();
150         return baos.toByteArray();
151     }
152 
153     @SuppressWarnings("unchecked")
154     static <T> T deserialize(byte[] streamBytes)
155         throws IOException, ClassNotFoundException
156     {
157         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
158         ObjectInputStream ois  = new ObjectInputStream(bais);
159         return (T) ois.readObject();
160     }
161 
162     static <T> T serializeDeserialize(T obj)
163         throws IOException, ClassNotFoundException
164     {
165         return deserialize(serialize(obj));
166     }
167 }
