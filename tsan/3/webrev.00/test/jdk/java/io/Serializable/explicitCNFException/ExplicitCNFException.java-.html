24 /* @test
25  * @bug 4407956
26  * @summary Verify that ClassNotFoundExceptions explicitly constructed and
27  *          thrown from with custom readObject/readExternal methods are
28  *          propagated properly.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     private void readObject(ObjectInputStream in)
35         throws ClassNotFoundException, IOException
36     {
37         throw new ClassNotFoundException();
38     }
39 }
40 
41 class B implements Externalizable {
42     public B() {}
43 
44     public void writeExternal(ObjectOutput out) throws IOException {}
45 
46     public void readExternal(ObjectInput in)
47         throws ClassNotFoundException, IOException
48     {
49         throw new ClassNotFoundException();
50     }
51 }
52 
53 public class ExplicitCNFException {
54     public static void main(String[] args) throws Exception {
55         test(new A());
56         test(new Object[]{ new A() });
57         test(new B());
58         test(new Object[]{ new B() });
59     }
60 
61     static void test(Object obj) throws Exception {
62         ByteArrayOutputStream bout = new ByteArrayOutputStream();
63         ObjectOutputStream oout = new ObjectOutputStream(bout);
64         oout.useProtocolVersion(ObjectStreamConstants.PROTOCOL_VERSION_2);
65         oout.writeObject(obj);
66         oout.close();
67         ObjectInputStream oin = new ObjectInputStream(
68             new ByteArrayInputStream(bout.toByteArray()));
69         try {
70             oin.readObject();
71             throw new Error();  // should not succeed
72         } catch (ClassNotFoundException ex) {
73         }
74     }
75 }
