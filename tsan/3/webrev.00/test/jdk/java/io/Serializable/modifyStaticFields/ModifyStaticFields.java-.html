24 /* @test
25  * @bug 4334623
26  * @summary Verify that serialPersistentFields cannot be used to cause
27  *          deserialization to set the value of a static field.
28  */
29 
30 import java.io.*;
31 
32 public class ModifyStaticFields implements Serializable {
33     private static final ObjectStreamField[] serialPersistentFields =
34         new ObjectStreamField[] { new ObjectStreamField("str", String.class) };
35 
36     static String str = "foo";
37 
38     public static void main(String[] args) throws Exception {
39         ByteArrayOutputStream bout = new ByteArrayOutputStream();
40         ObjectOutputStream oout = new ObjectOutputStream(bout);
41         oout.writeObject(new ModifyStaticFields());
42         oout.close();
43 
44         String origStr = str;
45         ObjectInputStream oin = new ObjectInputStream(
46             new ByteArrayInputStream(bout.toByteArray()));
47         oin.readObject();
48         if (str != origStr) {
49             throw new Error("deserialization modified static field");
50         }
51     }
52 
53     private void writeObject(ObjectOutputStream out) throws IOException {
54         ObjectOutputStream.PutField pf = out.putFields();
55         pf.put("str", "bar");
56         out.writeFields();
57     }
58 }
