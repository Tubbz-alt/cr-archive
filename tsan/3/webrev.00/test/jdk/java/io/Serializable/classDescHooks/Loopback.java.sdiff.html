24 /* @test
25  * @bug 4461299
26  * @summary Verify that serialization functions properly if
27  *          ObjectInputStream.readClassDescriptor() returns a local class
28  *          descriptor for which the serialVersionUID has not yet been
29  *          calculated.
30  */
31 
32 import java.io.*;
33 import java.util.*;
34 
35 class LoopbackOutputStream extends ObjectOutputStream {
36     LinkedList descs;
37 
38     LoopbackOutputStream(OutputStream out, LinkedList descs)
39         throws IOException
40     {
41         super(out);
42         this.descs = descs;
43     }
44 
45     protected void writeClassDescriptor(ObjectStreamClass desc)
46         throws IOException
47     {
48         descs.add(desc);
49     }
50 }
51 
52 class LoopbackInputStream extends ObjectInputStream {
53     LinkedList descs;
54 
55     LoopbackInputStream(InputStream in, LinkedList descs) throws IOException {
56         super(in);
57         this.descs = descs;
58     }
59 
60     protected ObjectStreamClass readClassDescriptor()
61         throws IOException, ClassNotFoundException
62     {
63         return (ObjectStreamClass) descs.removeFirst();
64     }
65 }
66 
67 public class Loopback implements Serializable {


68     String str;
69 
70     Loopback(String str) {
71         this.str = str;
72     }
73 
74     public static void main(String[] args) throws Exception {
75         Loopback lb = new Loopback("foo");
76         LinkedList descs = new LinkedList();
77         ByteArrayOutputStream bout = new ByteArrayOutputStream();
78         LoopbackOutputStream lout = new LoopbackOutputStream(bout, descs);
79         lout.writeObject(lb);
80         lout.close();
81 
82         LoopbackInputStream lin = new LoopbackInputStream(
83             new ByteArrayInputStream(bout.toByteArray()), descs);
84         Loopback lbcopy = (Loopback) lin.readObject();

86             throw new Error();
87         }
88     }
