36 import java.net.URLClassLoader;
37 import java.net.URL;
38 import java.nio.file.Path;
39 import java.nio.file.Paths;
40 
41 import jdk.test.lib.util.JarUtils;
42 
43 public class DeserializeButtonTest {
44     public static void main(String[] args) throws Exception {
45         setup();
46 
47         try (URLClassLoader ldr =
48             new URLClassLoader(new URL[]{ new URL("file:cb.jar") })) {
49             Runnable r = (Runnable) Class.forName("Foo", true, ldr).newInstance();

50             r.run();
51         }
52     }
53 
54     private static void setup() throws Exception {
55         Path classes = Paths.get(System.getProperty("test.classes", ""));
56         JarUtils.createJarFile(Paths.get("cb.jar"),
57                                classes,
58                                classes.resolve("Foo.class"),
59                                classes.resolve("Foo$TestElement.class"));
60     }
