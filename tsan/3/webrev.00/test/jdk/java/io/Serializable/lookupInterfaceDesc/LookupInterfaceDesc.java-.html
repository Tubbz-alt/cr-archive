24 /* @test
25  * @bug 4402227
26  * @summary Verify that ObjectStreamClass.lookup() functions properly
27  *          for interfaces.
28  */
29 
30 import java.io.*;
31 
32 interface Foo extends Serializable {
33     static final long serialVersionUID = 0xCAFE;
34 }
35 
36 interface Bar extends Externalizable {
37     static final long serialVersionUID = 0xBABE;
38 }
39 
40 interface Gub {
41 }
42 
43 public class LookupInterfaceDesc {
44     public static void main(String[] args) throws Exception {
45         ObjectStreamClass desc = ObjectStreamClass.lookup(Foo.class);
46         if ((desc.getSerialVersionUID() != Foo.serialVersionUID) ||
47             (desc.getFields().length != 0))
48         {
49             throw new Error();
50         }
51 
52         desc = ObjectStreamClass.lookup(Bar.class);
53         if ((desc.getSerialVersionUID() != Bar.serialVersionUID) ||
54             (desc.getFields().length != 0))
55         {
56             throw new Error();
57         }
58 
59         if (ObjectStreamClass.lookup(Gub.class) != null) {
60             throw new Error();
61         }
62     }
63 }
