24 /*
25  * @bug 4663191
26  * @summary Verify that readObject and readObjectNoData methods will not be
27  *          called on an object being deserialized if that object is already
28  *          tagged with a ClassNotFoundException.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     private static final long serialVersionUID = 0L;
35     @SuppressWarnings("serial") /* Incorrect declarations are being tested */
36     Object x;
37 }
38 
39 class B extends A {
40     private static final long serialVersionUID = 0L;
41     private void readObjectNoData() throws ObjectStreamException {
42         throw new Error("readObjectNoData called");
43     }
44 }
45 
46 class C extends B {
47     private static final long serialVersionUID = 0L;
48     private void readObject(ObjectInputStream in)
49         throws IOException, ClassNotFoundException
50     {
51         throw new Error("readObject called");
52     }
53 }
54 
55 public class Read {
56     public static void main(String[] args) throws Exception {
57         FileInputStream in = new FileInputStream("tmp.ser");
58         try {
59             ObjectInputStream oin = new ObjectInputStream(in);
60             try {
61                 oin.readObject();
62                 throw new Error("readObject should not succeed");
63             } catch (ClassNotFoundException e) {
64                 // expected
65             }
66             if (!oin.readObject().equals("after")) {
67                 throw new Error("subsequent object corrupted");
68             }
69         } finally {
70             in.close();
71         }
72     }
73 }
