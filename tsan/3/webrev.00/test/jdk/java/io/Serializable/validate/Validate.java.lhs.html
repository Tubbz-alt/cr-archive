24 /*
25 @test
26 @bug 4094892
27 @summary Verify that an object is not validated more than once during deserialization.
28 
29 */
30 
31 import java.io.*;
32 import java.util.Date;
33 
34 public class Validate {
35     public static void main(String[] args) throws Exception {
36         try {
37             // Write class out
38             ByteArrayOutputStream baos = new ByteArrayOutputStream();
39             ObjectOutput out = new ObjectOutputStream(baos);
40 
41             Class1 c1 = new Class1(11, 22);
42             out.writeObject(c1);
43             out.writeObject(new Class1(22,33));
44             out.writeObject(new Date());
45             out.writeObject(new Date());
46             out.writeObject(new Date());
47             out.writeObject(new Date());
48             out.flush();
49             out.close();
50 
51             // Read it back
52             ByteArrayInputStream bais =
53                 new ByteArrayInputStream(baos.toByteArray());
54             ObjectInputStream in = new ObjectInputStream(bais);
55             Class1 cc1 = (Class1) in.readObject();
56             Class1 cc2 = (Class1) in.readObject();
57             System.out.println("date: " + in.readObject());
58             System.out.println("date: " + in.readObject());
59             System.out.println("date: " + in.readObject());
60             System.out.println("date: " + in.readObject());
61             in.close();
62 
63             System.out.println(cc1.a + " " + cc1.b);
64             System.out.println(cc2.a + " " + cc2.b);
65         } catch (IOException e) {
66             e.printStackTrace();
67             throw e;
68         } catch (ClassNotFoundException e) {
69             e.printStackTrace();
70             throw e;
71         }
72     }
73 }
74 
75 class Class1 implements Serializable, ObjectInputValidation {


76     int a, b;
77     transient int validates;
78 
79     public Class1(int aa, int bb) {
80         a = aa;
81         b = bb;
82     }
83     public void validateObject() throws InvalidObjectException {
84         if (validates > 0)
85             throw new Error("Implementation error: Re-validating object " + this.toString());
86         validates++;
87 
88         System.out.println("Validating " + this.toString());
89         if (a > b) {
90             throw new InvalidObjectException("Fields cannot be negative");
91         }
92     }
93     private void readObject(ObjectInputStream in)
94         throws IOException, ClassNotFoundException {
95             in.registerValidation(this, 1);
96             in.defaultReadObject();
97     }
98 }
