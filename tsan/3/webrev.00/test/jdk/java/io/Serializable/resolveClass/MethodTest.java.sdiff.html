24 /* @test
25  * @bug 4413434
26  * @summary Verify that generated java.lang.reflect implementation classes do
27  *          not interfere with serialization's class resolution mechanism.
28  */
29 
30 import java.io.*;
31 import java.lang.reflect.*;
32 
33 public class MethodTest implements Serializable {


34     public static void main(String[] args) throws Exception {
35         Method readObject = ObjectInputStream.class.getDeclaredMethod(
36             "readObject", new Class[0]);
37         ByteArrayOutputStream bout = new ByteArrayOutputStream();
38         ObjectOutputStream oout = new ObjectOutputStream(bout);
39         oout.writeObject(new MethodTest());
40         oout.close();
41 
42         for (int i = 0; i < 100; i++) {
43             ObjectInputStream oin = new ObjectInputStream(
44                 new ByteArrayInputStream(bout.toByteArray()));
45             readObject.invoke(oin, new Object[0]);
46         }
47     }
48 }
