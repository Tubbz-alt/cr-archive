 24 /* @test
 25  * @bug 4140729
 26  * @summary Verify that writeReplace & readResolve are called by serialization.
 27  *          readResolve is used to maintain the invariant that the enums
 28  *          of TypeSafeEnum are singletons.
 29  */
 30 import java.io.*;
 31 
 32 public class TypeSafeEnum implements Serializable, ObjectInputValidation {
 33     private static final long serialVersionUID = 1L;
 34 
 35     private static int numWriteObject = 0;
 36     private static int numReadObject = 0;
 37 
 38 
 39     private String value;
 40     private TypeSafeEnum(String value) {
 41         this.value = value;
 42     }
 43 
 44     public static final TypeSafeEnum FIRST = new TypeSafeEnum("First");
 45     public static final TypeSafeEnum SECOND = new TypeSafeEnum("Second");
 46     public static final TypeSafeEnum THIRD = new TypeSafeEnum("Third");
 47     static int numReadResolve = 0;
 48     static int numWriteReplace = 0;
 49     static boolean verbose = false;
 50 
 51 
 52     private Object writeReplace() {
 53         numWriteReplace++;
 54         if (verbose) {
 55             System.out.println("TypeSafeEnum.writeReplace() " +
 56                                this.toString());
 57         }
 58         return this;
 59     }
 60 
 61     private Object readResolve() {
 62         numReadResolve++;
 63         if (verbose) {
 64             System.out.println("readResolve called on " + this.toString());
 65         }
 66         if (value.equals(FIRST.value)) {
 67             return FIRST;
 68         } else if (value.equals(SECOND.value)) {
 69             return SECOND;
 70         } else if (value.equals(THIRD.value)) {
 71             return THIRD;
 72         } else {
 73             //unknown type safe enum
 74             return this;
 75         }
 76     }
 77 
 78     private void readObject(ObjectInputStream in)
 79         throws IOException, ClassNotFoundException
 80     {
 81         numReadObject++;
 82         in.defaultReadObject();
 83         if (verbose) {
 84             System.out.println("TypeSafeEnum.readObject() " + this.toString());
 85         }
 86         if (value == null) {
 87             in.registerValidation(this, 0);
 88         }
 89     }
 90 
 91     public void validateObject() throws InvalidObjectException {
 92         // only top level case has null for value, validate.
 93         if (numWriteObject != 4) {
 94             throw new Error("Expected 4 calls to writeObject, only " +
 95                             numWriteObject + " made");
 96         }
 97         if (numReadObject != 4) {
 98             throw new Error("Expected 4 calls to readObject, only " +
 99                             numReadObject + " made");
100         }
101         if (numWriteReplace != 4) {
102             throw new Error("Expected 4 calls to writeReplace, only " +
103                             numWriteReplace + " made");
104         }
105         if (numReadResolve != 4) {
106             throw new Error("Expected 4 calls to readResolve, only " +
107                             numReadResolve + " made");
108         }
109     }
110 
111     private void writeObject(ObjectOutputStream out) throws IOException
112     {
113         numWriteObject++;
114         out.defaultWriteObject();
115         if (verbose) {
116             System.out.println("TypeSafeEnum.writeObject() " +
117                                this.toString());
118         }
119     }
120 
121     public String toString() {
122         return super.toString() + " value=" + value;
123     }
124 
125     public static void main(String args[])
126         throws IOException, ClassNotFoundException
127     {
128         if (args.length > 0 && args[0].equals("-verbose"))
129             verbose = true;
130 
131         TypeSafeEnum[] writeArray = new TypeSafeEnum[7];
132         writeArray[0] = FIRST;
133         writeArray[1] = SECOND;
134         writeArray[2] = THIRD;
135         writeArray[3] = FIRST;
136         writeArray[4] = SECOND;
137         writeArray[5] = THIRD;
138         writeArray[6] = new TypeSafeEnum("Third");
139 
140         ByteArrayOutputStream baos = new ByteArrayOutputStream();
141         ObjectOutputStream os = new ObjectOutputStream(baos);
142         os.writeObject(writeArray);
143         os.close();
144 
145         TypeSafeEnum[] readArray;
146         ObjectInputStream in =
147            new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
148         readArray = (TypeSafeEnum[])in.readObject();
149         in.close();
150 
151         for (int i= 0; i < writeArray.length - 1 ; i++) {
152             if (writeArray[i] != readArray[i]) {
153                 throw new Error("Serializa/deserialize did not preserve " +
154                                 "singleton for " +
155                                 readArray[i].toString() +
156                                 " and " + writeArray[i].toString());
157             }
158         }
159    }
160 };
