 24 import java.io.ByteArrayInputStream;
 25 import java.io.ByteArrayOutputStream;
 26 import java.io.IOException;
 27 import java.io.InputStream;
 28 import java.io.ObjectInputStream;
 29 import java.io.ObjectOutputStream;
 30 import java.io.ObjectStreamClass;
 31 import java.net.MalformedURLException;
 32 import java.net.URI;
 33 import java.net.URL;
 34 import java.net.URLClassLoader;
 35 import java.nio.file.Path;
 36 import jdk.test.lib.compiler.CompilerUtils;
 37 import org.testng.annotations.BeforeTest;
 38 import static org.testng.Assert.*;
 39 
 40 /**
 41  * An abstract superclass for tests that require to serialize and deserialize
 42  * record-like and record classes. Can be used for determining migration and
 43  * interop between record-like and record classes.
 44  */
 45 public class AbstractTest {
 46 
 47     private static final String VERSION = Integer.toString(Runtime.version().feature());
 48 
 49     static final String TEST_SRC = System.getProperty("test.src", ".");
 50     static final String TEST_CLASSES = System.getProperty("test.classes", ".");
 51     static final Path TEST_CLASSES_DIR = Path.of(TEST_CLASSES);
 52 
 53     static final Path PLAIN_SRC_DIR = Path.of(TEST_SRC, "plain");
 54     static final Path PLAIN_DEST_DIR = Path.of("plain");
 55 
 56     static final Path RECORD_SRC_DIR = Path.of(TEST_SRC, "record");
 57     static final Path RECORD_DEST_DIR = Path.of("record");
 58 
 59     @BeforeTest
 60     public void setup() throws IOException {
 61         assertTrue(CompilerUtils.compile(PLAIN_SRC_DIR, PLAIN_DEST_DIR,
 62                    "--enable-preview", "-source", VERSION,
 63                    "--class-path", TEST_CLASSES_DIR.toString()));
 64 
 65         assertTrue(CompilerUtils.compile(RECORD_SRC_DIR, RECORD_DEST_DIR,
 66                    "--enable-preview", "-source", VERSION,
 67                    "--class-path", TEST_CLASSES_DIR.toString()));
 68     }
 69 
 70     static <T> byte[] serialize(T obj) throws IOException {
 71         ByteArrayOutputStream baos = new ByteArrayOutputStream();
 72         ObjectOutputStream oos = new ObjectOutputStream(baos);
 73         oos.writeObject(obj);
 74         oos.close();
 75         return baos.toByteArray();
 76     }
 77 
 78     @SuppressWarnings("unchecked")
 79     static <T> T deserializeAsPlain(byte[] streamBytes)
 80         throws IOException, ClassNotFoundException
 81     {
 82         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
 83         ObjectInputStream ois  = new PlainObjectInputStream(bais);
 84         return (T) ois.readObject();
 85     }
 86 
 87     @SuppressWarnings("unchecked")
 88     static <T> T deserializeAsRecord(byte[] streamBytes)
 89         throws IOException, ClassNotFoundException
 90     {
 91         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
 92         ObjectInputStream ois  = new RecordObjectInputStream(bais);
 93         return (T) ois.readObject();
 94     }
 95 
 96     static Point newPlainPoint(int x, int y) throws Exception {
 97         Class<?> c = Class.forName("PointImpl", true, plainLoader);
 98         var point = (Point)c.getConstructor(int.class, int.class).newInstance(x, y);
 99         assert !point.getClass().isRecord();
100         return point;
101     }
102 
103     static DefaultValues newPlainDefaultValues(Point point) throws Exception {
104         Class<?> c = Class.forName("DefaultValuesImpl", true, plainLoader);
105         var defaultValues =  (DefaultValues)c.getConstructor(Point.class).newInstance(point);
106         assert !defaultValues.getClass().isRecord();
107         return defaultValues;
108     }
109 
110     static final ClassLoader plainLoader = plainLoader();
111 
112     static final ClassLoader plainLoader() {
113         try {
114             return new URLClassLoader("plain-loader",
115                                       new URL[] { URI.create(PLAIN_DEST_DIR.toUri() + "/").toURL() },
116                                       AbstractTest.class.getClassLoader());
117         } catch(MalformedURLException e) {
118             throw new AssertionError(e);
119         }
120     }
121 
122     static final ClassLoader recordLoader = recordLoader();
123 
124     static final ClassLoader recordLoader() {
125         try {
126             return new URLClassLoader("record-loader",
127                                       new URL[] { URI.create(RECORD_DEST_DIR.toUri() + "/").toURL() },
128                                       AbstractTest.class.getClassLoader());
129         } catch(MalformedURLException e) {
130             throw new AssertionError(e);
131         }
132     }
133 
134     static class PlainObjectInputStream extends ObjectInputStream {
135 
136         PlainObjectInputStream(InputStream is) throws IOException {
137             super(is);
138         }
139 
140         @Override
141         protected Class<?> resolveClass(ObjectStreamClass desc)
142             throws ClassNotFoundException
143         {
144             return Class.forName(desc.getName(), false, plainLoader);
145         }
146     }
147 
148     /**
149      * An OIS implementation that delegates its class lookup to a loader
150      * that attempts to locate the class equivalent of the specified stream
151      * class description first in the test's record implementations.
152      */
153     static class RecordObjectInputStream extends ObjectInputStream {
154 
155         RecordObjectInputStream(InputStream is) throws IOException {
156             super(is);
157         }
158 
159         @Override
160         protected Class<?> resolveClass(ObjectStreamClass desc)
161             throws ClassNotFoundException
162         {
163             return Class.forName(desc.getName(), false, recordLoader);
164         }
165     }
166 }
