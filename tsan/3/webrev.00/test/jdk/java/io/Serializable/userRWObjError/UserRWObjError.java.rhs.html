24 /* @test
25  * @bug 4082734
26  * @summary Ensure that Error exception is propogated from Serializable class'
27  *          readObject & writeObject method.
28  */
29 import java.io.*;
30 
31 /*
32  * Failure output:
33  * joef/scarry>java UserRWObjError
34  * Test FAILED:
35  *  java.lang.ClassCastException: java.lang.OutOfMemoryError
36  *       at java.io.ObjectOutputStream.invokeObjectWriter(ObjectOutputStream.java:1379)
37  *       at java.io.ObjectOutputStream.outputObject(ObjectOutputStream.java:755)
38  *       at java.io.ObjectOutputStream.writeObject(Objec
39  */
40 
41 public class UserRWObjError implements java.io.Serializable {
42     private static final long serialVersionUID = 1L;
43 
44     public static void main(String[] args) throws Exception {
45         try {
46             UserRWObjError obj = new UserRWObjError();
47             ObjectOutputStream out =
48                 new ObjectOutputStream(new ByteArrayOutputStream());
49             out.writeObject(obj);
50         } catch (ClassCastException e) {
51             throw e;
52         } catch (OutOfMemoryError e) {
53             System.err.println("Test PASSED:");
54             e.printStackTrace();
55         } catch (Exception e) {
56             System.err.println("An Unexpected exception occurred:");
57             throw e;
58         }
59     }
60 
61     private void writeObject(ObjectOutputStream out) throws IOException {
62         throw new OutOfMemoryError();
63     }
64 }
