24 /* @test
25  * @summary Verify that ClassCastException is thrown when deserializing
26  *          an object and one of its object fields is  incompatibly replaced
27  *          by either replaceObject/resolveObject.
28  *
29  */
30 import java.io.*;
31 
32 class A implements Serializable {
33     private static final long serialVersionUID = 1L;
34 }
35 
36 class B implements Serializable {
37     private static final long serialVersionUID = 1L;
38 }
39 
40 class Container implements Serializable {
41     private static final long serialVersionUID = 1L;
42     A a = new A();
43 }
44 
45 class ReplacerObjectOutputStream extends ObjectOutputStream {
46     static B b = new B();
47   public ReplacerObjectOutputStream(OutputStream out) throws IOException {
48     super(out);
49     enableReplaceObject(true);
50   }
51 
52   protected Object replaceObject(Object obj) throws IOException {
53       if(obj instanceof A) {
54           System.err.println("replaceObject(" + obj.toString() + ") with " +
55                              b.toString());
56           return b;
57       } else return obj;
58   }
59 }
60 
61 public class BadSubstByReplace {
62     public static void main(String args[]) throws IOException, ClassNotFoundException {
63         Container c = new Container();
64         ByteArrayOutputStream baos = new ByteArrayOutputStream();
65         ReplacerObjectOutputStream out =   new ReplacerObjectOutputStream(baos);
66         out.writeObject(c);
67         out.close();
68         ObjectInputStream in =
69             new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
70         try {
71             c = (Container)in.readObject(); // throws IllegalArgumentException.
72             throw new Error("Should have thrown ClassCastException");
73         } catch ( ClassCastException e) {
74             System.err.println("Caught expected exception " + e.toString());
75             e.printStackTrace();
76         } finally {
77             in.close();
78         }
79     }
80 }
