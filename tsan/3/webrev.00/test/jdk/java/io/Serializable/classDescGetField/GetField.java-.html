24 /* @test
25  * @bug 4180735
26  * @clean Foo GetField
27  * @build GetField
28  * @run main GetField
29  *
30  * @summary Make sure that getField() of ObjectStreamClass works correctly for
31  *          object types.
32  */
33 import java.io.*;
34 
35 public class GetField implements Serializable{
36     String str;
37     int i;
38 
39     public static void main(String[] args) throws Exception {
40         ObjectStreamClass cl = ObjectStreamClass.lookup(GetField.class);
41         if (cl == null)
42             throw new RuntimeException("Cannot resolve class : GetField");
43 
44         if (cl.getField("str") == null)
45             throw new RuntimeException(
46                 "ObjectStreamClass.getField() failed for object type");
47 
48         if (cl.getField("i") == null)
49             throw new RuntimeException(
50                 "ObjectStreamClass.getField() failed for primitive type");
51     }
52 }
