24 import java.io.Serializable;
25 import java.net.InetAddress;
26 import java.util.Objects;
27 
28 public class PointImpl implements Point, Serializable {
29 
30     private static final long serialVersionUID = 5L;
31 
32     final InetAddress moreCruft = InetAddress.getLoopbackAddress();  // some stream cruft
33     final boolean a = true;
34     final int x;
35     final int cruft = 9;
36     final int y;
37     final InetAddress addr;
38 
39     public PointImpl(int x, int y) {
40         this(x, y, null);
41     }
42 
43     public PointImpl(int x, int y, InetAddress addr) {
44         this.x = x;
45         this.y = y;
46         this.addr = addr;
47     }
48 
49     public int x() {
50         return x;
51     }
52 
53     public int y() {
54         return y;
55     }
56 
57     @Override
58     public String toString() {
59         return String.format("Point[x=%d, y=%d, addr=%s]", x, y, addr);
60     }
61 
62     @Override
63     public boolean equals(Object obj) {
64         if (obj instanceof Point) {
65             Point other = (Point) obj;
66             if (this.x == other.x() && this.y == other.y())
67                 return true;
68         }
69         return false;
70     }
71 
72     @Override
73     public int hashCode() {
74         return Objects.hash(x, y);
75     }
76 }
