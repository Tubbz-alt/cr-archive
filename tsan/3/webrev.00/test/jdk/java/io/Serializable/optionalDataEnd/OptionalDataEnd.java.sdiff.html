27  *          when a call to ObjectInputStream.readObject() attempts to read past
28  *          the end of custom data.
29  */
30 
31 import java.io.*;
32 
33 class Foo implements Serializable {


34     int reps;
35 
36     Foo(int reps) {
37         this.reps = reps;
38     }
39 
40     private void writeObject(ObjectOutputStream out) throws IOException {
41         out.defaultWriteObject();
42         for (int i = 0; i < reps; i++) {
43             out.writeObject(new Integer(i));
44         }
45     }
46 
47     private void readObject(ObjectInputStream in)
48         throws IOException, ClassNotFoundException
49     {
50         in.defaultReadObject();
51         for (int i = 0; i < reps; i++) {
52             in.readObject();
53         }
