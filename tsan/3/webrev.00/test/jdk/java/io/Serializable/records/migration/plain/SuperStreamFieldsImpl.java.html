24 /** A class, that when serialized, will have superclass state in the stream. */
25 public class SuperStreamFieldsImpl extends Foo
26     implements SuperStreamFields, java.io.Serializable
27 {
28     private static final long serialVersionUID = 0L;
29 
30     final String str;
31     final int[] x;
32     final int y;
33 
34     public SuperStreamFieldsImpl(String str, int[] x, int y) {
35         this.str = str;
36         this.x = x;
37         this.y = y;
38     }
39 
40     @Override
41     public String str() {
42         return str;
43     }
44 
45     @Override
46     public int[] x() {
47         return x;
48     }
49 
50     @Override
51     public int y() {
52         return y;
53     }
54 
55     @Override
56     public String toString() {
57         return String.format("SuperStreamFieldsImpl[str=%s, x=%s, y=%d]", str, x, y);
58     }
59 }
60 
61 class Foo implements java.io.Serializable {
62     private final String str ;
63     private final int x;
64 
65     public Foo() {
66         str = "chegar";
67         x = 10;
68     }
69 }
