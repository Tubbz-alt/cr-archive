 36     float f;

 39         this.i = i;
 40         this.f = f;
 41     }
 42 
 43     private void writeObject(ObjectOutputStream out) throws IOException {
 44         out.writeInt(i);
 45         out.writeFloat(f);
 46         out.writeUTF("skip me");
 47     }
