24 /* @test
25  * @summary Verify that if ObjectOutputStream's underlying OutputStream throws
26  *          an IOException, the original IOException (not a
27  *          StreamCorruptedException) will be thrown to the writing thread.
28  */
29 
30 import java.io.*;
31 
32 class OriginalIOException extends IOException {
33     private static final long serialVersionUID = 1L;
34 }
35 
36 class BrokenOutputStream extends OutputStream {
37     boolean broken = false;
38 
39     public void write(int b) throws IOException {
40         if (broken) {
41             throw new OriginalIOException();
42         }
43     }
44 }
45 
46 public class UnderlyingOutputStreamException {
47     public static void main(String[] args) throws Exception {
48         BrokenOutputStream bout = new BrokenOutputStream();
49         ObjectOutputStream oout = new ObjectOutputStream(bout);
50         bout.broken = true;
51         try {
52             oout.writeObject("foo");
53             throw new Error();
54         } catch (OriginalIOException ex) {
55         }
56     }
57 }
