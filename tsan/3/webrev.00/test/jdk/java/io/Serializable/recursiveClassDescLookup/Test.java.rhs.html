24 /* @test
25  * @bug 4803747
26  * @run main/timeout=20 Test
27  * @summary Verify that a nested call to ObjectStreamClass.lookup from within
28  *          the static initializer of a serializable class will not cause
29  *          deadlock.
30  */
31 
32 import java.io.*;
33 
34 class Foo implements Serializable {
35     private static final long serialVersionUID = 1L;
36     static {
37         ObjectStreamClass.lookup(Foo.class);
38     }
39 }
40 
41 public class Test {
42     public static void main(String[] args) throws Exception {
43         Class<?> fooCl = Class.forName("Foo", false, Test.class.getClassLoader());
44         ObjectStreamClass.lookup(fooCl);
45         System.out.println("done.");
46     }
47 }
