 56 @Test
 57 public class GlobalFilterTest {
 58     private static final String serialPropName = "jdk.serialFilter";
 59     private static final String badSerialFilter = "java.lang.StringBuffer;!*";
 60     private static final String origSerialFilterProperty =
 61             System.setProperty(serialPropName, badSerialFilter);

 63     /**
 64      * DataProvider of patterns and objects derived from the configured process-wide filter.
 65      * @return Array of arrays of pattern, object, allowed boolean, and API factory
 66      */
 67     @DataProvider(name="globalPatternElements")
 68     Object[][] globalPatternElements() {
 69         String globalFilter =
 70                 System.getProperty("expected-" + serialPropName,
 71                         Security.getProperty(serialPropName));
 72         if (globalFilter == null) {
 73             return new Object[0][];
 74         }
 75 
 76         String[] patterns = globalFilter.split(";");
 77         Object[][] objects = new Object[patterns.length][];
 78 
 79         for (int i = 0; i < patterns.length; i++) {
 80             Object o;
 81             boolean allowed;
 82             String pattern = patterns[i].trim();
 83             if (pattern.contains("=")) {
 84                 allowed = false;
 85                 o = SerialFilterTest.genTestObject(pattern, false);
 86             } else {
 87                 allowed = !pattern.startsWith("!");
 88                 o = (allowed)
 89                     ? SerialFilterTest.genTestObject(pattern, true)
 90                     : SerialFilterTest.genTestObject(pattern.substring(1), false);
 91 
 92                 Assert.assertNotNull(o, "fail generation failed");
 93             }
 94             objects[i] = new Object[3];
 95             objects[i][0] = pattern;
 96             objects[i][1] = allowed;
 97             objects[i][2] = o;
 98         }
 99         return objects;
100     }

102     /**
103      * Test that the process-wide filter is set when the properties are set
104      * and has the toString matching the configured pattern.
105      */
106     @Test()
107     static void globalFilter() {



108         ObjectInputFilter filter = ObjectInputFilter.Config.getSerialFilter();
109 
110         // Check that the System.setProperty(jdk.serialFilter) DOES NOT affect the filter.
111         String asSetSystemProp = System.getProperty(serialPropName,
112                 Security.getProperty(serialPropName));
113         Assert.assertNotEquals(Objects.toString(filter, null), asSetSystemProp,
114                 "System.setProperty(\"jdk.serialfilter\", ...) should not change filter: " +
115                 asSetSystemProp);
116 
117         String pattern =
118                 System.getProperty("expected-" + serialPropName,
119                         Security.getProperty(serialPropName));
120         System.out.printf("global pattern: %s, filter: %s%n", pattern, filter);
121         Assert.assertEquals(Objects.toString(filter, null), pattern,
122                 "process-wide filter pattern does not match");
123     }

125     /**
126      * If the Global filter is already set, it should always refuse to be
127      * set again.
128      * If there is a security manager, setting the serialFilter should fail
129      * without the appropriate permission.
130      * If there is no security manager then setting it should work.
131      */
132     @Test()
133     static void setGlobalFilter() {
134         SecurityManager sm = System.getSecurityManager();
135         ObjectInputFilter filter = new SerialFilterTest.Validator();
136         ObjectInputFilter global = ObjectInputFilter.Config.getSerialFilter();
137         if (global != null) {
138             // once set, can never be re-set
139             try {
140                 ObjectInputFilter.Config.setSerialFilter(filter);
141                 Assert.fail("set only once process-wide filter");
142             } catch (IllegalStateException ise) {
143                 // Normal, once set can never be re-set even if no security manager
144             } catch (SecurityException se) {
145                 if (sm == null) {
146                     Assert.fail("wrong exception when security manager is not set", se);
147                 }
148             }
149         } else {
150             if (sm == null) {
151                 // no security manager
152                 try {
153                     ObjectInputFilter.Config.setSerialFilter(filter);
154                     // Note once set, it can not be reset; so other tests
155                     System.out.printf("Global Filter set to Validator%n");
156                 } catch (SecurityException se) {
157                     Assert.fail("setGlobalFilter should not get SecurityException", se);
158                 }
159                 try {
160                     // Try to set it again, expecting it to throw
161                     ObjectInputFilter.Config.setSerialFilter(filter);
162                     Assert.fail("set only once process-wide filter");
163                 } catch (IllegalStateException ise) {
164                     // Normal case
165                 }
166             } else {
167                 // Security manager
168                 SecurityException expectSE = null;
169                 try {
170                     sm.checkPermission(new SerializablePermission("serialFilter"));
171                 } catch (SecurityException se1) {
172                     expectSE = se1;
173                 }
174                 SecurityException actualSE = null;
175                 try {
176                     ObjectInputFilter.Config.setSerialFilter(filter);
177                 } catch (SecurityException se2) {
178                     actualSE = se2;
179                 }
180                 if (expectSE == null | actualSE == null) {
181                     Assert.assertEquals(expectSE, actualSE, "SecurityException");
182                 } else {
183                     Assert.assertEquals(expectSE.getClass(), actualSE.getClass(),
184                             "SecurityException class");
185                 }
186             }
187         }
188     }

190     /**
191      * For each pattern in the process-wide filter test a generated object
192      * against the default process-wide filter.
193      *
194      * @param pattern a pattern extracted from the configured global pattern
195      */
196     @Test(dataProvider = "globalPatternElements")
197     static void globalFilterElements(String pattern, boolean allowed,Object obj) {
198         testGlobalPattern(pattern, obj, allowed);
199     }
200 
201     /**
202      * Serialize and deserialize an object using the default process-wide filter
203      * and check allowed or reject.
204      *
205      * @param pattern the pattern
206      * @param object the test object
207      * @param allowed the expected result from ObjectInputStream (exception or not)
208      */
209     static void testGlobalPattern(String pattern, Object object, boolean allowed) {
210         try {
211 //            System.out.printf("global %s pattern: %s, obj: %s%n", (allowed ? "allowed" : "not allowed"), pattern, object);
212             byte[] bytes = SerialFilterTest.writeObjects(object);
213             try (ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
214                  ObjectInputStream ois = new ObjectInputStream(bais)) {
215                 Object o = ois.readObject();
216             } catch (EOFException eof) {
217                 // normal completion
218             } catch (ClassNotFoundException cnf) {
219                 Assert.fail("Deserializing", cnf);
220             }
221             Assert.assertTrue(allowed, "filter should have thrown an exception");
222         } catch (IllegalArgumentException iae) {
223             Assert.fail("bad format pattern", iae);
224         } catch (InvalidClassException ice) {
225             Assert.assertFalse(allowed, "filter should not have thrown an exception: " + ice);
226         } catch (IOException ioe) {
227             Assert.fail("Unexpected IOException", ioe);
228         }
229     }
