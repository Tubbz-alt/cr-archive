24 /* @test
25  * @summary Test proper functioning of class-defined writeReplace methods.
26  */
27 
28 import java.io.*;
29 
30 public class WriteReplace {
31 
32     static class ReplaceMe implements Serializable {
33         private static final long serialVersionUID = 1L;
34 
35         @SuppressWarnings("serial") /* Incorrect use is being tested */
36         private Object obj;
37         private boolean writeReplaceCalled = false;
38 
39         public ReplaceMe(Object obj) {
40             this.obj = obj;
41         }
42 
43         private Object writeReplace() throws ObjectStreamException {
44             if (writeReplaceCalled) {
45                 throw new Error("multiple calls to writeReplace");
46             }
47             writeReplaceCalled = true;
48             return obj;
49         }
50     }
51 
52     public static void main(String[] args) throws Exception {
53         final int nobjs = 10;
54         final int nrounds = 10;
55 
56         Object common = "foo";
57         ReplaceMe[] objs = new ReplaceMe[nobjs];
58         for (int i = 0; i < nobjs; i++) {
59             objs[i] = new ReplaceMe(common);
60         }
61 
62         ByteArrayOutputStream bout = new ByteArrayOutputStream();
63         ObjectOutputStream oout = new ObjectOutputStream(bout);
64         for (int i = 0; i < nrounds; i++) {
65             for (int j = 0; j < nobjs; j++) {
66                 oout.writeObject(objs[j]);
67             }
68         }
69 
70         oout.flush();
71         ByteArrayInputStream bin =
72             new ByteArrayInputStream(bout.toByteArray());
73         ObjectInputStream oin = new ObjectInputStream(bin);
74         common = null;
75 
76         for (int i = 0; i < nrounds; i++) {
77             for (int j = 0; j < nobjs; j++) {
78                 if (common == null) {
79                     common = oin.readObject();
80                 } else {
81                     if (oin.readObject() != common) {
82                         throw new Error("incorrect replacement object");
83                     }
84                 }
85             }
86         }
87     }
88 }
