36 import java.io.*;
37 
38 class NewExternFieldClass implements Externalizable {
39     private static final long serialVersionUID = 1L;
40 
41     byte l;
42 
43     public NewExternFieldClass() {
44         l = 0;
45     }
46 
47     public NewExternFieldClass(byte value) {
48         l = value;
49     }
50 
51     public void readExternal(ObjectInput s)
52         throws IOException, ClassNotFoundException
53     {
54         l = s.readByte();
55         System.out.println("readExternal read " + l);
56     }
57 
58     public void writeExternal(ObjectOutput s) throws IOException
59     {
60         s.writeByte(l);
61     }
62 }
63 
64 class D implements Serializable {
65     private static final long serialVersionUID = 1L;
66 
67     public int x;
68     D(int y) {
69         x = y;
70     }
71 }
72 
73 class A implements Serializable  {
74     // Version 1.1 of class A.  Added superclass NewSerializableSuper.
75     private static final long serialVersionUID = 1L;
76     NewExternFieldClass foo;
77     D zoo;
78 
79     int bar;
80     A() {
81         bar = 4;
82         foo = new NewExternFieldClass((byte)66);
83         zoo = new D(22);
84     }
85 }
86 
87 public class WriteAddedField {
88     public static void main(String args[]) throws IOException {
89         A a = new A();
90         File f = new File("tmp.ser");
91         ObjectOutput out =
92             new ObjectOutputStream(new FileOutputStream(f));
93         out.writeObject(a);
94         out.writeObject(new A());
95         out.close();
96     }
97 }
