25  * @bug 4511532
26  * @summary Verify that the message string of a ClassCastException thrown by
27  *          ObjectInputStream when attempting to assign a value to a field of
28  *          an incompatible type contains the names of the value's class, the
29  *          field's declaring class, the field's type, and the field itself.
30  */
31 
32 import java.io.*;
33 
34 class Foo implements Serializable {
35     private static final long serialVersionUID = 0L;
36     private Float bar;
37 }
38 
39 class Gub extends Foo {}


40 
41 public class Read {
42     public static void main(String[] args) throws Exception {
43         FileInputStream in = new FileInputStream("tmp.ser");
44         try {
45             ObjectInputStream oin = new ObjectInputStream(in);
46             oin.readObject();
47             throw new Error("readObject should not succeed");
48         } catch (ClassCastException e) {
49             String s = e.getMessage();
50             System.out.println("ClassCastException message: " + s);
51             if (s == null ||
52                 s.indexOf(Foo.class.getName()) == -1 ||
53                 s.indexOf(Integer.class.getName()) == -1 ||
54                 s.indexOf(Float.class.getName()) == -1 ||
55                 s.indexOf(Gub.class.getName()) == -1 ||
56                 s.indexOf("bar") == -1)
57             {
58                 throw new Error("ClassNotFoundException message incomplete");
59             }
