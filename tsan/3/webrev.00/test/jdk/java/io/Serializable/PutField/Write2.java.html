24 /* @test
25  * @bug 4453723
26  *
27  * @clean Write2 Read2 Foo
28  * @compile Write2.java
29  * @run main Write2
30  * @clean Write2 Read2 Foo
31  * @compile Read2.java
32  * @run main Read2
33  *
34  * @summary Verify that ObjectOutputStream.PutField.write() works for objects
35  *          that do not define primitive serializable fields.
36  */
37 
38 import java.io.*;
39 
40 class Foo implements Serializable {
41     private static final long serialVersionUID = 0L;
42     private static final ObjectStreamField[] serialPersistentFields = {
43         new ObjectStreamField("s1", String.class),
44         new ObjectStreamField("s2", String.class)
45     };
46 
47     @SuppressWarnings("deprecation")
48     private void writeObject(ObjectOutputStream out) throws IOException {
49         ObjectOutputStream.PutField fields = out.putFields();
50         fields.put("s1", "qwerty");
51         fields.put("s2", "asdfg");
52         fields.write(out);
53     }
54 }
55 
56 public class Write2 {
57     public static void main(String[] args) throws Exception {
58         ObjectOutputStream oout =
59             new ObjectOutputStream(new FileOutputStream("tmp.ser"));
60         oout.writeObject(new Foo());
61         oout.close();
62     }
63 }
