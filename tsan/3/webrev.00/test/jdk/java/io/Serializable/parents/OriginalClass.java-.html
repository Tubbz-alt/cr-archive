24 /* @test
25  * @bug 4186885
26  * @clean OriginalClass EvolvedClass
27  * @build OriginalClass
28  * @run main OriginalClass
29  * @build EvolvedClass
30  * @run main EvolvedClass
31  * @summary To ensure that during deserializing classes, only the highest
32  *          non-serializable class in the hierarchy has its no-arg constructor
33  *          invoked.
34  *
35  */
36 
37 import java.io.*;
38 public class OriginalClass {
39 
40     public static void main(String args[]) throws Exception{
41         ASubClass corg = new ASubClass(1);
42         ASubClass cnew = null;
43 
44         // Serialize the subclass
45         FileOutputStream fo = new FileOutputStream("parents.ser");
46         try {
47             ObjectOutputStream so = new ObjectOutputStream(fo);
48             so.writeObject(corg);
49             so.flush();
50         } finally {
51             fo.close();
52         }
53 
54         System.out.println("Printing the serialized class: ");
55         System.out.println();
56         System.out.println(corg);
57     }
58 }
59 
60 
61 class ASubClass implements Serializable {
62     int num;
63 
64     ASubClass(int num) {
65         this.num = num;
66     }
67 
68     public String toString() {
69         return ("\nNum:  " + num);
70     }
71 }
