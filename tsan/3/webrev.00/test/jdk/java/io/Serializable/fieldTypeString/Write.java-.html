24 /* @test
25  * @bug 4404696
26  *
27  * @clean Write Read Foo Bar
28  * @compile Write.java
29  * @run main Write
30  * @clean Write Read Foo Bar
31  * @compile Read.java
32  * @run main Read
33  *
34  * @summary Verify that serialization does not require matching type strings
35  *          for non-primitive fields.
36  *
37  * NOTE: This test should be removed if it is determined that serialization
38  * *should* consider type strings when matching non-primitive fields.
39  */
40 
41 import java.io.*;
42 
43 class Foo implements Serializable {
44     private static final long serialVersionUID = 0L;
45     Object obj;
46 
47     Foo(Object obj) {
48         this.obj = obj;
49     }
50 }
51 
52 class Bar implements Serializable {
53     private static final long serialVersionUID = 0L;
54     int q;
55 }
56 
57 public class Write {
58     public static void main(String[] args) throws Exception {
59         ObjectOutputStream oout =
60             new ObjectOutputStream(new FileOutputStream("foo.ser"));
61         oout.writeObject(new Foo("foo"));
62         oout.writeObject(new Foo(new Integer(0)));
63         oout.close();
64 
65         oout = new ObjectOutputStream(new FileOutputStream("bar.ser"));
66         oout.writeObject(new Bar());
67         oout.close();
68     }
69 }
