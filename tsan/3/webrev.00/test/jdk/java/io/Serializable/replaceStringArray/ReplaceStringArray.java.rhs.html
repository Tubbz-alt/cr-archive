 24 /* @test
 25  * @bug 4099013
 26  * @summary Enable substitution of String and Array by ObjectStreams.
 27  */

 29 import java.io.*;
 30 import java.lang.reflect.Array;
 31 
 32 class A implements Serializable {
 33     private static final long serialVersionUID = 1L;
 34 
 35     String stringA;
 36     String stringB;
 37     String stringC;
 38     String[] arrayOfString;
 39 
 40     A() {
 41         stringA = "hello";
 42         stringB = "goodbye";
 43         stringC = stringA;
 44         arrayOfString = new String[2];
 45         for (int i = 0; i < arrayOfString.length; i++)
 46             arrayOfString[i] = new String("array element " + i);
 47     }
 48 
 49     void report() {
 50             System.out.println("stringA = " + stringA);
 51             System.out.println("stringB = " + stringB);
 52             System.out.println("stringC = " + stringC);
 53             System.out.println("length of arrayOfString = " +
 54                                arrayOfString.length);
 55             for (int i = 0; i < arrayOfString.length; i++)
 56                 System.out.println("arrayOfString[" + i + "]= " +
 57                                    arrayOfString[i]);
 58     }
 59 }

 61 class SubstituteObjectOutputStream extends ObjectOutputStream {
 62     public int numStringsReplaced = 0;
 63     public int numArraysCounted = 0;
 64 
 65     public SubstituteObjectOutputStream(OutputStream out) throws IOException {
 66         super(out);
 67         enableReplaceObject(true);
 68     }
 69 
 70     protected Object replaceObject(Object obj) throws IOException {
 71         if (obj instanceof String) {
 72             numStringsReplaced++;
 73             return obj + "_WriteReplaced";
 74         }
 75         if (obj.getClass().isArray()) {
 76             Object[] array = (Object[]) obj;
 77             /* Double the array.
 78              * Initialize new array elements with original array. */
 79             Class<?> arrayComponentType = array.getClass().getComponentType();
 80             Object[] newarray =
 81                 (Object[])Array.newInstance(arrayComponentType,
 82                                             array.length * 2);
 83             for (int i = 0; i < array.length; i++)
 84                 newarray[i] = array[i];
 85             for (int ni = array.length; ni < 2* array.length; ni++)
 86                 newarray[ni] = array[ni - array.length];
 87             numArraysCounted++;
 88             obj = newarray;
 89         }
 90         return obj;
 91     }
 92 }

 94 class SubstituteObjectInputStream extends ObjectInputStream {
 95     public int numStringsReplaced = 0;
 96     public int numArraysCounted = 0;
 97 
 98     public SubstituteObjectInputStream(InputStream in) throws IOException {
 99         super(in);
100         enableResolveObject(true);
101     }
102 
103     protected Object resolveObject(Object obj) throws IOException {
104         if (obj instanceof String) {
105             numStringsReplaced++;
106             return obj + "_ReadResolved";
107         }
108         if (obj.getClass().isArray()) {
109             Object[] array = (Object[])obj;
110 
111             /* Double the array.
112              * Initialize new array elements with original array. */
113             Class<?> arrayComponentType = array.getClass().getComponentType();
114             Object[] newarray =
115                 (Object[])Array.newInstance(arrayComponentType,
116                                             array.length * 2);
117             for (int i = 0; i < array.length; i++)
118                 newarray[i] = array[i];
119             for (int ni = array.length; ni < 2* array.length; ni++)
120                 newarray[ni] = array[ni - array.length];
121             numArraysCounted++;
122             obj = newarray;
123         }
124         return obj;
125     }
126 }

128 public class ReplaceStringArray {
129     public static void main(String args[]) throws IOException, ClassNotFoundException {
130         boolean verbose = false;
131         if (args.length >= 1 && args[0].compareTo("verbose") == 0)
132             verbose = true;
133 
134 
135         A a = new A();
136         if (verbose) {
137             System.out.println("Value of Class A");
138             a.report();
139             System.out.println("");
140         }
141 
142 
143         /* Serialize object a to bytestream. */
144         if (verbose) {
145             System.out.println("Serialize A to SubstituteObjectOutputStream");
146             System.out.println("");
147         }
148         ByteArrayOutputStream baos = new ByteArrayOutputStream();
149         SubstituteObjectOutputStream out =   new SubstituteObjectOutputStream(baos);
150         out.writeObject(a);
151         out.close();
152         a = null;
153 
154         /* Validate that writeReplace called by SubstituteObjectOutputStream.*/
155         boolean expectedResult = (out.numStringsReplaced == 4);
156         if (!expectedResult)
157             throw new Error("Expected " + 4 + " strings to be replaced during serialization;" +
158                             " only " + out.numStringsReplaced + " strings were replaced.");
159         if (out.numArraysCounted != 1)
160             throw new Error("Expected 1 array during serialization; only " +
161                             out.numArraysCounted + " arrays");
162 
163         if (verbose) {
164             System.out.println("DeSerialize A from SubstituteObjectInputStream");
165             System.out.println("");
166         }
167         SubstituteObjectInputStream in =
168             new SubstituteObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
169         a = (A)in.readObject();
170         in.close();
171 
172         /* Validate that readResolve called by SubstituteObjectInputStream.*/
173         if (in.numStringsReplaced != 4)
174             throw new Error("Expected 4 strings to be resolved during deserialization;" +
175                             " only " + in.numStringsReplaced + " strings were resolved.");
176         if (in.numArraysCounted != 1)
177             throw new Error("Expected 1 array during deserialization; only " +
178                             out.numArraysCounted + " arrays");
179         if (a.arrayOfString.length != 8)
180             throw new Error("Expected a.arrayOfString.length to be 8, observed " +
181                             a.arrayOfString.length);
182         if (verbose) {
183             System.out.println("Value of Class A after serialize/deserialize with writeReplace/readResolve");
184             a.report();
185         }
186     }
187 }
