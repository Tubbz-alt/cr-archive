24 /* @test
25  * @bug 4373844
26  * @summary Verify that partially initialized ObjectStreamClass instances
27  *          cannot be obtained from ObjectStreamClass.lookup() in the event
28  *          that the target class is uninitializable.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     // existence of SVUID forces class initialization during classdesc init
35     private static final long serialVersionUID = 0L;
36 
37     static {
38         if ("foo".equals("foo")) {      // force class initialization failure
39             throw new Error();
40         }
41     }
42 }
43 
44 public class PartialClassDesc {
45     public static void main(String[] args) throws Exception {
46         Class<?> cl = Class.forName(
47             "A", false, PartialClassDesc.class.getClassLoader());
48         ObjectStreamClass desc = null;
49         try {
50             desc = ObjectStreamClass.lookup(cl);
51         } catch (Throwable th) {
52         }
53         try {
54             desc = ObjectStreamClass.lookup(cl);
55         } catch (Throwable th) {
56         }
57         if (desc != null) {
58             throw new Error("should not be able to obtain class descriptor");
59         }
60     }
61 }
