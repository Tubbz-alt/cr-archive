24 /* @test
25  * @bug 4662327
26  * @summary Verify that ObjectInputStream.defaultReadObject() throws a
27  *          ClassNotFoundException if any of the non-primitive field values it
28  *          reads in are tagged with ClassNotFoundExceptions.
29  */
30 
31 import java.io.*;
32 
33 class Foo implements Serializable {
34     Object obj = new Bar();
35 
36     private void readObject(ObjectInputStream in)
37         throws IOException, ClassNotFoundException
38     {
39         in.defaultReadObject();
40         if (obj == null) {
41             throw new Error(
42                 "ClassNotFoundException masked by defaultReadObject()");
43         }
44     }
45 }
46 
47 class Bar implements Serializable {}
48 
49 class TestObjectInputStream extends ObjectInputStream {
50     TestObjectInputStream(InputStream in) throws IOException { super(in); }
51 
52     protected Class resolveClass(ObjectStreamClass desc)
53         throws IOException, ClassNotFoundException
54     {
55         if (desc.getName().equals(Bar.class.getName())) {
56             throw new ClassNotFoundException();
57         }
58         return super.resolveClass(desc);
59     }
60 }
61 
62 public class DefaultReadObjectCNFException {
63     public static void main(String[] args) throws Exception {
64         ByteArrayOutputStream bout = new ByteArrayOutputStream();
65         ObjectOutputStream oout = new ObjectOutputStream(bout);
66         oout.writeObject(new Foo());
67         oout.writeObject("after");
68         oout.close();
69         ObjectInputStream oin = new TestObjectInputStream(
70             new ByteArrayInputStream(bout.toByteArray()));
71         try {
72             oin.readObject();
73         } catch (ClassNotFoundException e) {
74             // expected
75         }
76         if (!oin.readObject().equals("after")) {
77             throw new Error("subsequent object corrupted");
78         }
79     }
80 }
