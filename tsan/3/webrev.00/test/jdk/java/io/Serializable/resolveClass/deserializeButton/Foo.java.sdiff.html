27  *          correctly resolved during deserialization when read in by custom
28  *          readObject() method of a bootstrap class (in this case,
29  *          java.util.Vector).
30  */

32 import java.io.*;
33 import java.util.Vector;
34 
35 public class Foo implements Runnable {
36 
37     static class TestElement extends Object implements Serializable {}


38 
39     public void run() {
40         try {
41             Vector<TestElement> container = new Vector<TestElement>();
42             container.add(new TestElement());
43 
44             // iterate to trigger java.lang.reflect code generation
45             for (int i = 0; i < 100; i++) {
46                 ByteArrayOutputStream bout = new ByteArrayOutputStream();
47                 ObjectOutputStream oout = new ObjectOutputStream(bout);
48                 oout.writeObject(container);
49                 oout.close();
50                 ObjectInputStream oin = new ObjectInputStream(
51                     new ByteArrayInputStream(bout.toByteArray()));
52                 oin.readObject();
53             }
54         } catch (Exception ex) {
55             throw new Error(
56                 "Error occured while (de)serializing container: ", ex);
57         }
