24 /* @test
25  * @bug 4334623
26  * @summary Verify that serialPersistentFields cannot be used to cause
27  *          deserialization to set the value of a static field.
28  */
29 
30 import java.io.*;
31 
32 public class ModifyStaticFields implements Serializable {
33     private static final long serialVersionUID = 1L;
34 
35     private static final ObjectStreamField[] serialPersistentFields =
36         new ObjectStreamField[] { new ObjectStreamField("str", String.class) };
37 
38     static String str = "foo";
39 
40     public static void main(String[] args) throws Exception {
41         ByteArrayOutputStream bout = new ByteArrayOutputStream();
42         ObjectOutputStream oout = new ObjectOutputStream(bout);
43         oout.writeObject(new ModifyStaticFields());
44         oout.close();
45 
46         String origStr = str;
47         ObjectInputStream oin = new ObjectInputStream(
48             new ByteArrayInputStream(bout.toByteArray()));
49         oin.readObject();
50         if (str != origStr) {
51             throw new Error("deserialization modified static field");
52         }
53     }
54 
55     private void writeObject(ObjectOutputStream out) throws IOException {
56         ObjectOutputStream.PutField pf = out.putFields();
57         pf.put("str", "bar");
58         out.writeFields();
59     }
60 }
