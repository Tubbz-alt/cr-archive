 32 import java.io.ByteArrayInputStream;
 33 import java.io.ByteArrayOutputStream;
 34 import java.io.DataInputStream;
 35 import java.io.DataOutputStream;
 36 import java.io.IOException;
 37 import java.io.ObjectInputStream;
 38 import java.io.ObjectOutputStream;
 39 import java.io.Serializable;
 40 import java.util.ArrayList;
 41 import java.util.List;
 42 import java.util.stream.LongStream;
 43 import org.testng.annotations.DataProvider;
 44 import org.testng.annotations.Test;
 45 import static java.io.ObjectStreamConstants.*;
 46 import static java.lang.System.out;
 47 import static org.testng.Assert.assertEquals;
 48 import static org.testng.Assert.assertTrue;
 49 
 50 public class SerialVersionUIDTest {
 51 
 52     record R1 () implements Serializable {
 53         private static final long serialVersionUID = 1L;
 54     }
 55 
 56     record R2 (int x, int y) implements Serializable {
 57         private static final long serialVersionUID = 0L;
 58     }
 59 
 60     record R3 () implements Serializable { }
 61 
 62     record R4 (String s) implements Serializable { }
 63 
 64     record R5 (long l) implements Serializable {
 65         private static final long serialVersionUID = 5678L;
 66     }
 67 
 68     @DataProvider(name = "recordObjects")
 69     public Object[][] recordObjects() {
 70         return new Object[][] {
 71             new Object[] { new R1(),        1L    },
 72             new Object[] { new R2(1, 2),    0L    },
 73             new Object[] { new R3(),        0L    },
 74             new Object[] { new R4("s"),     0L    },
 75             new Object[] { new R5(7L),      5678L },
 76         };
 77     }
 78 
 79     /**
 80      * Tests that a declared SUID for a record class is inserted into the stream.
 81      */
 82     @Test(dataProvider = "recordObjects")
 83     public void testSerialize(Object objectToSerialize, long expectedUID)
 84         throws Exception
 85     {
 86         out.println("\n---");
 87         out.println("serializing : " + objectToSerialize);
 88         byte[] bytes = serialize(objectToSerialize);
 89 
 90         ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
 91         DataInputStream dis = new DataInputStream(bais);
 92 
 93         // sanity
 94         assertEquals(dis.readShort(), STREAM_MAGIC);
 95         assertEquals(dis.readShort(), STREAM_VERSION);
 96         assertEquals(dis.readByte(), TC_OBJECT);
 97         assertEquals(dis.readByte(), TC_CLASSDESC);
 98         assertEquals(dis.readUTF(), objectToSerialize.getClass().getName());
 99 
100         // verify that the UID is as expected
101         assertEquals(dis.readLong(), expectedUID);
102     }
103 
104     @DataProvider(name = "recordClasses")
105     public Object[][] recordClasses() {
106         List<Object[]> list = new ArrayList<>();
107         List<Class<?>> recordClasses = List.of(R1.class, R2.class, R3.class, R4.class, R5.class);
108         LongStream.of(0L, 1L, 100L, 10_000L, 1_000_000L).forEach(suid ->
109                 recordClasses.stream()
110                              .map(cl -> new Object[] {cl, suid})
111                              .forEach(list::add));
112         return list.stream().toArray(Object[][]::new);
113     }
114 
115     /**
116      * Tests that matching of the serialVersionUID values ( stream value
117      * and runtime class value ) is waived for record classes.
118      */
119     @Test(dataProvider = "recordClasses")
120     public void testSerializeFromClass(Class<? extends Record> cl, long suid)
121         throws Exception
122     {
123         out.println("\n---");
124         byte[] bytes = byteStreamFor(cl.getName(), suid);
125         Object obj = deserialize(bytes);
126         assertEquals(obj.getClass(), cl);
127         assertTrue(obj.getClass().isRecord());
128     }
129 
130     // --- infra

132     /**
133      * Returns a stream of bytes for the given class and uid. The
134      * stream will have no stream field values.
135      */
136     static byte[] byteStreamFor(String className, long uid)
137         throws Exception
138     {
139         ByteArrayOutputStream baos = new ByteArrayOutputStream();
140         DataOutputStream dos = new DataOutputStream(baos);
141         dos.writeShort(STREAM_MAGIC);
142         dos.writeShort(STREAM_VERSION);
143         dos.writeByte(TC_OBJECT);
144         dos.writeByte(TC_CLASSDESC);
145         dos.writeUTF(className);
146         dos.writeLong(uid);
147         dos.writeByte(SC_SERIALIZABLE);
148         dos.writeShort(0);                // number of fields
149         dos.writeByte(TC_ENDBLOCKDATA);   // no annotations
150         dos.writeByte(TC_NULL);           // no superclasses
151         dos.close();
152         return baos.toByteArray();
153     }
154 
155     static <T> byte[] serialize(T obj) throws IOException {
156         ByteArrayOutputStream baos = new ByteArrayOutputStream();
157         ObjectOutputStream oos = new ObjectOutputStream(baos);
158         oos.writeObject(obj);
159         oos.close();
160         return baos.toByteArray();
161     }
162 
163     static <T> T deserialize(byte[] streamBytes)
164         throws IOException, ClassNotFoundException
165     {
166         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
167         ObjectInputStream ois  = new ObjectInputStream(bais);
168         return (T) ois.readObject();
169     }
170 }
