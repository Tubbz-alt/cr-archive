 24 /*
 25  * @test
 26  * @summary Ensures that the serialization implementation can *always* access
 27  *          the record constructor
 28  * @compile --enable-preview -source ${jdk.version} ConstructorAccessTest.java
 29  * @run testng/othervm --enable-preview ConstructorAccessTest
 30  * @run testng/othervm/java.security.policy=empty_security.policy --enable-preview ConstructorAccessTest
 31  */

 33 import java.io.ByteArrayInputStream;
 34 import java.io.ByteArrayOutputStream;
 35 import java.io.IOException;
 36 import java.io.ObjectInput;
 37 import java.io.ObjectInputStream;
 38 import java.io.ObjectOutput;
 39 import java.io.ObjectOutputStream;
 40 import java.io.Externalizable;
 41 import java.io.Serializable;
 42 import org.testng.annotations.DataProvider;
 43 import org.testng.annotations.Test;
 44 import static java.lang.System.out;
 45 import static org.testng.Assert.assertEquals;
 46 import static org.testng.Assert.fail;
 47 
 48 /*implicit*/ record Aux1 (int x) implements Serializable { }
 49 
 50 /*implicit*/ record Aux2 (int x) implements Serializable { }
 51 
 52 public class ConstructorAccessTest {
 53 
 54     public record A (int x) implements Serializable { }
 55 
 56     protected static record B (long l) implements Serializable { }
 57 
 58     /*implicit*/ static record C (float f) implements Serializable { }
 59 
 60     private record D (double d) implements Serializable { }
 61 
 62     interface ThrowingExternalizable extends Externalizable {
 63         default void writeExternal(ObjectOutput out) {
 64             fail("should not reach here");
 65         }
 66         default void readExternal(ObjectInput in) {
 67             fail("should not reach here");
 68         }
 69     }
 70 
 71     public record E (short s) implements ThrowingExternalizable { }
 72 
 73     protected static record F (long l) implements ThrowingExternalizable { }
 74 
 75     /*implicit*/ static record G (double d) implements ThrowingExternalizable { }
 76 
 77     private record H (double d) implements ThrowingExternalizable { }
 78 
 79     @DataProvider(name = "recordInstances")
 80     public Object[][] recordInstances() {
 81         return new Object[][] {
 82             new Object[] { new A(34)        },
 83             new Object[] { new B(44L)       },
 84             new Object[] { new C(4.5f)      },
 85             new Object[] { new D(440d)      },
 86             new Object[] { new E((short)12) },
 87             new Object[] { new F(45L)       },
 88             new Object[] { new G(0.4d)      },
 89             new Object[] { new H(440d)      },
 90             new Object[] { new Aux1(63)     },
 91             new Object[] { new Aux2(64)     },
 92         };
 93     }
 94 
 95     @Test(dataProvider = "recordInstances")
 96     public void roundTrip(Object objToSerialize) throws Exception {
 97         out.println("\n---");
 98         out.println("serializing : " + objToSerialize);
 99         var objDeserialized = serializeDeserialize(objToSerialize);
100         out.println("deserialized: " + objDeserialized);
101         assertEquals(objToSerialize, objDeserialized);
102         assertEquals(objDeserialized, objToSerialize);
103     }
104 
105     // ---
106 
107     static <T> byte[] serialize(T obj) throws IOException {
108         ByteArrayOutputStream baos = new ByteArrayOutputStream();
109         ObjectOutputStream oos = new ObjectOutputStream(baos);
110         oos.writeObject(obj);
111         oos.close();
112         return baos.toByteArray();
113     }
114 
115     @SuppressWarnings("unchecked")
116     static <T> T deserialize(byte[] streamBytes)
117         throws IOException, ClassNotFoundException
118     {
119         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
120         ObjectInputStream ois  = new ObjectInputStream(bais);
121         return (T) ois.readObject();
122     }
123 
124     static <T> T serializeDeserialize(T obj)
125         throws IOException, ClassNotFoundException
126     {
127         return deserialize(serialize(obj));
128     }
129 }
