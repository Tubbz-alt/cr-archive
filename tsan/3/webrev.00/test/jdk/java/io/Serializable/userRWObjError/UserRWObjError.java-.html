24 /* @test
25  * @bug 4082734
26  * @summary Ensure that Error exception is propogated from Serializable class'
27  *          readObject & writeObject method.
28  */
29 import java.io.*;
30 
31 /*
32  * Failure output:
33  * joef/scarry>java UserRWObjError
34  * Test FAILED:
35  *  java.lang.ClassCastException: java.lang.OutOfMemoryError
36  *       at java.io.ObjectOutputStream.invokeObjectWriter(ObjectOutputStream.java:1379)
37  *       at java.io.ObjectOutputStream.outputObject(ObjectOutputStream.java:755)
38  *       at java.io.ObjectOutputStream.writeObject(Objec
39  */
40 
41 public class UserRWObjError implements java.io.Serializable {
42 
43     public static void main(String[] args) throws Exception {
44         try {
45             UserRWObjError obj = new UserRWObjError();
46             ObjectOutputStream out =
47                 new ObjectOutputStream(new ByteArrayOutputStream());
48             out.writeObject(obj);
49         } catch (ClassCastException e) {
50             throw e;
51         } catch (OutOfMemoryError e) {
52             System.err.println("Test PASSED:");
53             e.printStackTrace();
54         } catch (Exception e) {
55             System.err.println("An Unexpected exception occurred:");
56             throw e;
57         }
58     }
59 
60     private void writeObject(ObjectOutputStream out) throws IOException {
61         throw new OutOfMemoryError();
62     }
63 }
