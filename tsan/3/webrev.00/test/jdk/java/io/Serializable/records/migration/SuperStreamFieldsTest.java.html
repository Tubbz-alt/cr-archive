 24 /*
 25  * @test
 26  * @summary superclass fields in the stream should be discarded
 27  * @library /test/lib
 28  * @modules jdk.compiler
 29  * @compile --enable-preview -source ${jdk.version} AssignableFrom.java Point.java
 30  *          DefaultValues.java SuperStreamFields.java SuperStreamFieldsTest.java
 31  * @run testng/othervm --enable-preview SuperStreamFieldsTest
 32  */
 33 
 34 import org.testng.annotations.DataProvider;
 35 import org.testng.annotations.Test;
 36 import static java.lang.System.out;
 37 import static org.testng.Assert.assertEquals;
 38 
 39 /**
 40  *  Tests that superclass fields in the stream are discarded.
 41  */
 42 public class SuperStreamFieldsTest extends AbstractTest {
 43 
 44     @DataProvider(name = "plainInstances")
 45     public Object[][] plainInstances() {
 46         return new Object[][] {
 47             new Object[] { newPlainSuperStreamFields("cat", new int[] { 1 },    1)   },
 48             new Object[] { newPlainSuperStreamFields("dog", new int[] { 2, 3 }, 4)   },
 49             new Object[] { newPlainSuperStreamFields("rat", new int[] { 5, 6 }, 7)   },
 50         };
 51     }
 52 
 53     /** Serializes non-record (plain) instance, deserializes as a record. */
 54     @Test(dataProvider = "plainInstances")
 55     public void testPlainToRecord(SuperStreamFields objToSerialize) throws Exception {
 56         assert !objToSerialize.getClass().isRecord();
 57         out.println("serialize   : " + objToSerialize);
 58         byte[] bytes = serialize(objToSerialize);
 59         SuperStreamFields objDeserialized = deserializeAsRecord(bytes);
 60         assert objDeserialized.getClass().isRecord();
 61         out.println("deserialized: " + objDeserialized);
 62 
 63         assertEquals(objToSerialize.str(), objDeserialized.str());
 64         assertEquals(objToSerialize.x(),   objDeserialized.x());
 65         assertEquals(objToSerialize.y(),   objDeserialized.y());
 66     }
 67 
 68     @DataProvider(name = "recordInstances")
 69     public Object[][] recordInstances() {
 70         return new Object[][] {
 71             new Object[] { newRecordSuperStreamFields("goat",   new int[] { 56 },     66)   },
 72             new Object[] { newRecordSuperStreamFields("rabbit", new int[] { 22, 75 }, 77)   },
 73             new Object[] { newRecordSuperStreamFields("mouse",  new int[] { 12, 18 }, 88)   },
 74         };
 75     }
 76 
 77     /** Serializes record instance, deserializes as non-record (plain). */
 78     @Test(dataProvider = "recordInstances")
 79     public void testRecordToPlain(SuperStreamFields objToSerialize) throws Exception {
 80         assert objToSerialize.getClass().isRecord();
 81         out.println("serialize   : " + objToSerialize);
 82         byte[] bytes = serialize(objToSerialize);
 83         SuperStreamFields objDeserialized = deserializeAsPlain(bytes);
 84         assert !objDeserialized.getClass().isRecord();
 85         out.println("deserialized: " + objDeserialized);
 86 
 87         assertEquals(objToSerialize.str(), objDeserialized.str());
 88         assertEquals(objToSerialize.x(),   objDeserialized.x());
 89         assertEquals(objToSerialize.y(),   objDeserialized.y());
 90 
 91     }
 92 
 93     static SuperStreamFields newPlainSuperStreamFields(String str, int[] x, int y) {
 94         try {
 95             Class<?> c = Class.forName("SuperStreamFieldsImpl", true, plainLoader);
 96             var obj = (SuperStreamFields) c.getConstructor(String.class, int[].class, int.class)
 97                                            .newInstance(str, x, y);
 98             assert !obj.getClass().isRecord();
 99             return obj;
100         } catch (ReflectiveOperationException e) {
101             throw new AssertionError(e);
102         }
103     }
104 
105     static SuperStreamFields newRecordSuperStreamFields(String str, int[] x, int y) {
106         try {
107             Class<?> c = Class.forName("SuperStreamFieldsImpl", true, recordLoader);
108             var obj = (SuperStreamFields) c.getConstructor(String.class, int[].class, int.class)
109                                            .newInstance(str, x, y);
110             assert obj.getClass().isRecord();
111             return obj;
112         } catch (ReflectiveOperationException e) {
113             throw new AssertionError(e);
114         }
115     }
116 }
