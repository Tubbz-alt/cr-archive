24 /*
25  * @bug 4663191
26  * @summary Verify that readObject and readObjectNoData methods will not be
27  *          called on an object being deserialized if that object is already
28  *          tagged with a ClassNotFoundException.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     private static final long serialVersionUID = 0L;

35     Object x;
36 }
37 
38 class B extends A {
39     private static final long serialVersionUID = 0L;
40     private void readObjectNoData() throws ObjectStreamException {
41         throw new Error("readObjectNoData called");
42     }
43 }
44 
45 class C extends B {
46     private static final long serialVersionUID = 0L;
47     private void readObject(ObjectInputStream in)
48         throws IOException, ClassNotFoundException
49     {
50         throw new Error("readObject called");
51     }
52 }
53 
54 public class Read {
55     public static void main(String[] args) throws Exception {
56         FileInputStream in = new FileInputStream("tmp.ser");
57         try {
58             ObjectInputStream oin = new ObjectInputStream(in);
59             try {
60                 oin.readObject();
61                 throw new Error("readObject should not succeed");
62             } catch (ClassNotFoundException e) {
63                 // expected
64             }
65             if (!oin.readObject().equals("after")) {
66                 throw new Error("subsequent object corrupted");
67             }
68         } finally {
69             in.close();
70         }
71     }
72 }
