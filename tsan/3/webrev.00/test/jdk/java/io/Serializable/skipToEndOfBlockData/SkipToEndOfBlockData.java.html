 24 /* @test
 25  * @bug 4228592
 26  * @summary Ensure that ObjectInputStream properly skips over block data when a
 27  *          class that defines readObject() or readExternal() fails to read all
 28  *          of the data written by the corresponding writeObject() or
 29  *          writeExternal() method.
 30  */
 31 
 32 import java.io.*;
 33 
 34 class MismatchedRead implements Serializable {
 35     private static final long serialVersionUID = 1L;
 36 
 37     int i;
 38     float f;
 39 
 40     MismatchedRead(int i, float f) {
 41         this.i = i;
 42         this.f = f;
 43     }
 44 
 45     private void writeObject(ObjectOutputStream out) throws IOException {
 46         out.writeInt(i);
 47         out.writeFloat(f);
 48         out.writeUTF("skip me");
 49     }
 50 
 51     private void readObject(ObjectInputStream in)
 52         throws IOException, ClassNotFoundException
 53     {
 54         i = in.readInt();
 55         f = in.readFloat();
 56     }
 57 
 58     public boolean equals(Object obj) {
 59         if (! (obj instanceof MismatchedRead))
 60             return false;
 61         MismatchedRead other = (MismatchedRead) obj;
 62         return (i == other.i && f == other.f);
 63     }
 64 
 65     public int hashCode() {
 66         return i;
 67     }
 68 }
 69 
 70 class MismatchedReadExternal implements Externalizable {
 71     private static final long serialVersionUID = 1L;
 72 
 73     int i;
 74     float f;
 75 
 76     public MismatchedReadExternal() {
 77         this(0, (float) 0.0);
 78     }
 79 
 80     MismatchedReadExternal(int i, float f) {
 81         this.i = i;
 82         this.f = f;
 83     }
 84 
 85     public void writeExternal(ObjectOutput out) throws IOException {
 86         out.writeInt(i);
 87         out.writeFloat(f);
 88         out.writeUTF("skip another");
 89     }
 90 
 91     public void readExternal(ObjectInput in) throws IOException {
 92         i = in.readInt();
 93         f = in.readFloat();
 94     }
 95 
 96     public boolean equals(Object obj) {
 97         if (! (obj instanceof MismatchedReadExternal))
 98             return false;
 99         MismatchedReadExternal other = (MismatchedReadExternal) obj;
100         return (i == other.i && f == other.f);
101     }
102 
103     public int hashCode() {
104         return i;
105     }
106 }
107 
108 class InnocentBystander implements Serializable {
109     private static final long serialVersionUID = 1L;
110 
111     String s;
112 
113     InnocentBystander(String s) {
114         this.s = s;
115     }
116 
117     public boolean equals(Object obj) {
118         if (! (obj instanceof InnocentBystander))
119             return false;
120         InnocentBystander other = (InnocentBystander) obj;
121         if (s != null)
122             return s.equals(other.s);
123         return (s == other.s);
124     }
125 
126     public int hashCode() {
127         return s.hashCode();
128     }
129 }
130 
131 public class SkipToEndOfBlockData {
132     public static void main(String[] args) throws Exception {
133         ObjectOutputStream oout;
134         ObjectInputStream oin;
135         ByteArrayOutputStream bout;
136         ByteArrayInputStream bin;
137         MismatchedRead mr, mrcopy;
138         MismatchedReadExternal mre, mrecopy;
139         InnocentBystander ib, ibcopy;
140 
141         bout = new ByteArrayOutputStream();
142         oout = new ObjectOutputStream(bout);
143         mr = new MismatchedRead(1, (float) 2.34);
144         mre = new MismatchedReadExternal(5, (float) 6.78);
145         ib = new InnocentBystander("foo");
146 
147         oout.writeObject(mr);
148         oout.writeObject(mre);
149         oout.writeObject(ib);
150         oout.flush();
151 
152         bin = new ByteArrayInputStream(bout.toByteArray());
153         oin = new ObjectInputStream(bin);
154         mrcopy = (MismatchedRead) oin.readObject();
155         mrecopy = (MismatchedReadExternal) oin.readObject();
156         ibcopy = (InnocentBystander) oin.readObject();
157 
158         if (! (mr.equals(mrcopy) && mre.equals(mrecopy) && ib.equals(ibcopy)))
159             throw new Error("copy not equal to original");
160     }
161 }
