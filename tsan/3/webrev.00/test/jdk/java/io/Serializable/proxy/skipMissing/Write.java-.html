24 /* @test
25  *
26  * @clean Write Read A B I Handler
27  * @compile Write.java Handler.java
28  * @run main Write
29  * @clean Write Read A B I Handler
30  * @compile Read.java Handler.java
31  * @run main Read
32  *
33  * @summary Verify that ObjectInputStream can skip over unresolvable serialized
34  *          proxy instances.
35  */
36 
37 import java.io.*;
38 import java.lang.reflect.*;
39 
40 interface I {}          // interface present only on writing side
41 
42 class A implements Serializable {
43     private static final long serialVersionUID = 0L;
44     String a = "a";
45     Object proxy;
46     String z = "z";
47 
48     A(Object proxy) { this.proxy = proxy; }
49 }
50 
51 class B implements Serializable {
52     private static final long serialVersionUID = 0L;
53     String s = "s";
54     transient Object proxy;
55 
56     B(Object proxy) { this.proxy = proxy; }
57 
58     private void writeObject(ObjectOutputStream out) throws IOException {
59         out.defaultWriteObject();
60         out.writeObject(proxy);
61     }
62 }
63 
64 public class Write {
65     public static void main(String[] args) throws Exception {
66         Object proxy = Proxy.newProxyInstance(
67             Write.class.getClassLoader(),
68             new Class[] { I.class }, new Handler());
69         ObjectOutputStream oout = new ObjectOutputStream(
70             new FileOutputStream("tmp.ser"));
71         oout.writeObject(new A(proxy));
72         oout.reset();
73         oout.writeObject(new B(proxy));
74         oout.writeObject(proxy);
75         oout.close();
76     }
77 }
