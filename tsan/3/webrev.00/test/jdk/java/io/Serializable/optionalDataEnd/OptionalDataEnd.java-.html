24 /* @test
25  * @bug 4402870
26  * @summary Verify that an OptionalDataException with eof == true is thrown
27  *          when a call to ObjectInputStream.readObject() attempts to read past
28  *          the end of custom data.
29  */
30 
31 import java.io.*;
32 
33 class Foo implements Serializable {
34     int reps;
35 
36     Foo(int reps) {
37         this.reps = reps;
38     }
39 
40     private void writeObject(ObjectOutputStream out) throws IOException {
41         out.defaultWriteObject();
42         for (int i = 0; i < reps; i++) {
43             out.writeObject(new Integer(i));
44         }
45     }
46 
47     private void readObject(ObjectInputStream in)
48         throws IOException, ClassNotFoundException
49     {
50         in.defaultReadObject();
51         for (int i = 0; i < reps; i++) {
52             in.readObject();
53         }
54         try {
55             in.readObject();
56             throw new Error();
57         } catch (OptionalDataException ex) {
58             if (! (ex.eof && (ex.length == 0))) {
59                 throw new Error();
60             }
61         }
62     }
63 }
64 
65 public class OptionalDataEnd {
66     public static void main(String[] args) throws Exception {
67         ByteArrayOutputStream bout = new ByteArrayOutputStream();
68         ObjectOutputStream oout = new ObjectOutputStream(bout);
69         oout.writeObject(new Foo(5));
70         oout.writeObject(new Foo(0));
71         oout.close();
72 
73         ObjectInputStream oin = new ObjectInputStream(
74             new ByteArrayInputStream(bout.toByteArray()));
75         oin.readObject();
76         oin.readObject();
77         oin.close();
78     }
79 }
