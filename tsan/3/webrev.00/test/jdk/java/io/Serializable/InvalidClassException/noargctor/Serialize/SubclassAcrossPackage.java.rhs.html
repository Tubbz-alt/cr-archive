176 public class SubclassAcrossPackage {
177     public static void main(String args[])
178         throws IOException, ClassNotFoundException
179         {
180             ByteArrayOutputStream baos = new ByteArrayOutputStream();
181             ObjectOutputStream out =   new ObjectOutputStream(baos);
182             out.writeObject(new PublicSerializable());
183             out.writeObject(new ProtectedSerializable());
184             out.writeObject(new SamePackageSerializable());
185             out.writeObject(new SamePackageProtectedSerializable());
186             out.writeObject(new DifferentPackageSerializable());
187 
188             InputStream is = new ByteArrayInputStream(baos.toByteArray());
189             ObjectInputStream in = new ObjectInputStream(is);
190             /* (PublicSerializable)*/ in.readObject();
191             /* (ProtectedSerializable) */ in.readObject();
192             /* (SamePackageSerializable) */ in.readObject();
193             /* (SamePackageProtectedSerializable) */ in.readObject();
194             try {
195             /* (DifferentPackageSerializable) */ in.readObject();
196             } catch (InvalidClassException e) {
197             }
198             in.close();
199 
200             baos.reset();
201             out = new ObjectOutputStream(baos);
202             out.writeObject(new PrivateSerializable());
203             in = new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
204             try {
205                 /* (PrivateSerializable) */ in.readObject();
206                 throw new Error("Expected InvalidClassException reading PrivateSerialziable");
207             } catch (InvalidClassException e) {
208             }
209             in.close();
210 
211             baos.reset();
212             out = new ObjectOutputStream(baos);
213             out.writeObject(new SamePackagePrivateSerializable());
214             in = new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
215             try {
216                 /* (SamePackagePrivateSerializable) */ in.readObject();
217                 throw new Error("Expected InvalidClassException reading PrivateSerialziable");
218             } catch (InvalidClassException e) {
219             }
220             in.close();
221 
222             baos.reset();
223             out = new ObjectOutputStream(baos);
224             out.writeObject(new ExternalizablePublicCtor());
225 
226             in = new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
227             /* (ExternalizablePublicCtor) */ in.readObject();
