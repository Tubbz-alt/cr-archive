31  * @compile Read.java
32  * @run main Read
33  *
34  * @summary Verify that the message string of a ClassCastException thrown by
35  *          ObjectInputStream when attempting to assign a value to a field of
36  *          an incompatible type contains the names of the value's class, the
37  *          field's declaring class, the field's type, and the field itself.
38  */

40 import java.io.*;
41 
42 class Foo implements Serializable {
43     private static final long serialVersionUID = 0L;
44     private Integer bar = 0;
45 }
46 
47 class Gub extends Foo {
48     private static final long serialVersionUID = 1L;
49 }
50 
51 public class Write {
52     public static void main(String[] args) throws Exception {
53         ObjectOutputStream oout =
54             new ObjectOutputStream(new FileOutputStream("tmp.ser"));
55         oout.writeObject(new Gub());
56         oout.close();
57     }
58 }
