 27  * @library /test/lib

 33 
 34 import java.math.BigDecimal;
 35 import java.math.BigInteger;
 36 import org.testng.annotations.DataProvider;
 37 import org.testng.annotations.Test;
 38 import static java.lang.System.out;
 39 import static org.testng.Assert.assertEquals;
 40 
 41 /**
 42  * Basic test to check that stream field values that are not the exact
 43  * declared param/field type, but assignable to a declared supertype,
 44  * are bound/assigned correctly.
 45  */
 46 public class AssignableFromTest extends AbstractTest {
 47 
 48     @DataProvider(name = "plainInstances")
 49     public Object[][] plainInstances() {
 50         return new Object[][] {
 51             new Object[] { newPlainAssignableFrom(Byte.valueOf((byte)11))   },
 52             new Object[] { newPlainAssignableFrom(Short.valueOf((short)22)) },
 53             new Object[] { newPlainAssignableFrom(Integer.valueOf(33))      },
 54             new Object[] { newPlainAssignableFrom(Long.valueOf(44))         },
 55             new Object[] { newPlainAssignableFrom(BigDecimal.valueOf(55))   },
 56             new Object[] { newPlainAssignableFrom(BigInteger.valueOf(66))   },
 57         };
 58     }
 59 
 60     /** Serialize non-record (plain) instances, deserialize as a record. */
 61     @Test(dataProvider = "plainInstances")
 62     public void testPlainToRecord(AssignableFrom objToSerialize) throws Exception {
 63         assert !objToSerialize.getClass().isRecord();
 64         out.println("serialize   : " + objToSerialize);
 65         byte[] bytes = serialize(objToSerialize);
 66         AssignableFrom objDeserialized = deserializeAsRecord(bytes);
 67         assert objDeserialized.getClass().isRecord();
 68         out.println("deserialized: " + objDeserialized);
 69 
 70         assertEquals(objToSerialize.number(), objDeserialized.number());
 71         assertEquals(objDeserialized.number(), objToSerialize.number());
 72         assertEquals(objDeserialized.number().getClass(), objDeserialized.number().getClass());
 73 
 74     }
 75 
 76     @DataProvider(name = "recordInstances")
 77     public Object[][] recordInstances() {
 78         return new Object[][] {
 79             new Object[] { newRecordAssignableFrom(Byte.valueOf((byte)21))   },
 80             new Object[] { newRecordAssignableFrom(Short.valueOf((short)32)) },
 81             new Object[] { newRecordAssignableFrom(Integer.valueOf(43))      },
 82             new Object[] { newRecordAssignableFrom(Long.valueOf(54))         },
 83             new Object[] { newRecordAssignableFrom(BigDecimal.valueOf(65))   },
 84             new Object[] { newRecordAssignableFrom(BigInteger.valueOf(76))   },
 85         };
 86     }
 87 
 88     /** Serialize record instances, deserialize as non-record (plain). */
 89     @Test(dataProvider = "recordInstances")
 90     public void testRecordToPlain(AssignableFrom objToSerialize) throws Exception {
 91         assert objToSerialize.getClass().isRecord();
 92         out.println("serialize   : " + objToSerialize);
 93         byte[] bytes = serialize(objToSerialize);
 94         AssignableFrom objDeserialized = deserializeAsPlain(bytes);
 95         assert !objDeserialized.getClass().isRecord();
 96         out.println("deserialized: " + objDeserialized);
 97 
 98         assertEquals(objToSerialize.number(), objDeserialized.number());
 99         assertEquals(objDeserialized.number(), objToSerialize.number());
100         assertEquals(objDeserialized.number().getClass(), objDeserialized.number().getClass());
101     }
102 
103     static AssignableFrom newPlainAssignableFrom(Number number) {
104         try {
105             Class<?> c = Class.forName("AssignableFromImpl", true, plainLoader);
106             var obj = (AssignableFrom) c.getConstructor(Number.class).newInstance(number);
107             assert !obj.getClass().isRecord();
108             return obj;
109         } catch (ReflectiveOperationException e) {
110             throw new AssertionError(e);
111         }
112     }
113 
114     static AssignableFrom newRecordAssignableFrom(Number number) {
115         try {
116             Class<?> c = Class.forName("AssignableFromImpl", true, recordLoader);
117             var obj = (AssignableFrom) c.getConstructor(Number.class).newInstance(number);
118             assert obj.getClass().isRecord();
119             return obj;
120         } catch (ReflectiveOperationException e) {
121             throw new AssertionError(e);
122         }
123     }
124 }
