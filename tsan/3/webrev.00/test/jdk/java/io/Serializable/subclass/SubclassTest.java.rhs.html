 24 /*
 25  * @test
 26  * @bug 4100915
 27  * @summary Verify that [write/read]ObjectOverride methods get called.
 28  *          Test verifies that ALL methods to write an object can
 29  *          be overridden. However, the testing for reading an object
 30  *          is incomplete. Only test that readObjectOverride is called.
 31  *          An entire protocol would need to be implemented and written
 32  *          out before being able to test the input side of the API.
 33  *
 34  *          Also, would be appropriate that this program verify
 35  *          that if SerializablePermission "enableSubclassImplementation"
 36  *          is not in the security policy and security is enabled, that
 37  *          a security exception is thrown when constructing the
 38  *          ObjectOutputStream subclass.
 39  *
 40  *
 41  * @compile AbstractObjectInputStream.java AbstractObjectOutputStream.java
 42  * @compile XObjectInputStream.java XObjectOutputStream.java
 43  * @compile SubclassTest.java
 44  * @run main SubclassTest
 45  * @run main/othervm/policy=Allow.policy SubclassTest -expectSecurityException
 46  */
 47 
 48 import java.io.ByteArrayInputStream;
 49 import java.io.ByteArrayOutputStream;
 50 import java.io.IOException;
 51 import java.io.ObjectInputStream;
 52 import java.io.ObjectOutputStream;
 53 import java.io.ObjectStreamField;
 54 import java.io.Serializable;
 55 
 56 /**
 57  * Test if customized readObject and writeObject are called.
 58  */
 59 class B implements Serializable {
 60     private static final long serialVersionUID = 1L;
 61 
 62     public int publicIntField;
 63     public static int numWriteObjectCalled = 0;
 64     B(int v) {
 65         publicIntField = v;
 66     }
 67     private void writeObject(ObjectOutputStream os) throws IOException {
 68         numWriteObjectCalled++;
 69         os.defaultWriteObject();
 70     }
 71 
 72     private void readObject(ObjectInputStream is)
 73         throws IOException, ClassNotFoundException
 74     {
 75         is.defaultReadObject();
 76     }
 77 
 78 };

 80 /**
 81  * Test PutFields interface.
 82  */
 83 
 84 class C implements Serializable {
 85     private static final long serialVersionUID = 1L;
 86 
 87     public int xx1;
 88     public int xx2;
 89     private static final ObjectStreamField[] serialPersistentFields = {
 90         new ObjectStreamField("x1", Integer.TYPE),
 91         new ObjectStreamField("x2", Integer.TYPE),
 92         new ObjectStreamField("x3", Integer.TYPE),
 93         new ObjectStreamField("x4", Integer.TYPE)
 94     };
 95     C() {
 96         xx1 = 300;
 97         xx2 = 400;
 98     }
 99 
100     private void writeObject(ObjectOutputStream os) throws IOException {
101         ObjectOutputStream.PutField putFields = os.putFields();
102         putFields.put("x1", xx1);
103         putFields.put("x2", xx2);
104         putFields.put("x3", xx1 * 2);
105         putFields.put("x4", xx2 * 2);
106         os.writeFields();
107     }
108 
109 };

112 class A implements Serializable {
113     private static final long serialVersionUID = 1L;
114 
115     public int  publicIntField;
116     public long publicLongField;
117     public B    publicBField;
118     public B[]  publicBArray = { new B(4), new B(6)};
119     public C    publicCField;
120 
121     public A() {
122         publicIntField = 3;
123         publicLongField = 10L;
124         publicBField = new B(5);
125         publicCField = new C();
126     }
127 };
128 
129 public class SubclassTest {
130     public static void main(String argv[])
131         throws IOException, ClassNotFoundException
132     {
133         boolean expectSecurityException = false;
134 
135         if (argv.length > 0 &&
136             argv[0].compareTo("-expectSecurityException") == 0)
137             expectSecurityException = true;
138 
139         ByteArrayOutputStream baos = new ByteArrayOutputStream(20);
140         XObjectOutputStream os = null;
141         try {
142             os = new XObjectOutputStream(baos);
143             if (expectSecurityException)
144                 throw new Error("Assertion failure. " +
145                                 "Expected a security exception on previous line.");
146         } catch (SecurityException e) {
147             if (expectSecurityException) {
148                 System.err.println("Caught expected security exception.");
149                 return;
150             }
151             throw e;
152         }
153         os.writeObject(new A());
154         os.close();
155         if (B.numWriteObjectCalled != 3)
156             throw new Error("Expected B.writeObject() to be called 3 times;" +
157                             " observed only " + B.numWriteObjectCalled + " times");
158 
159         XObjectInputStream is =
160             new XObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
161         try {
162             A a = (A)is.readObject();
163             throw new Error("Expected readObjectOverride() to be called and throw IOException(not implemented)");
164         } catch (IOException e) {
165         }
166         is.close();
167     }
168 };
