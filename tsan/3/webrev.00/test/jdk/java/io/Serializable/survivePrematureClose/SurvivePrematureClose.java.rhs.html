24 /* @test
25  * @bug 4502808
26  * @summary Verify that if the custom serialization method (i.e., readExternal,
27  *          writeExternal, readObject or writeObject) of an object closes the
28  *          stream it is passed, (de)serialization of that object can still
29  *          complete.
30  */
31 
32 import java.io.*;
33 
34 class A implements Externalizable {
35     private static final long serialVersionUID = 1L;
36 
37     public A() {}
38 
39     public void writeExternal(ObjectOutput out) throws IOException {
40         out.writeInt(0);
41         out.close();
42     }
43 
44     public void readExternal(ObjectInput in)
45         throws IOException, ClassNotFoundException
46     {
47         in.readInt();
48         in.close();
49     }
50 }
51 
52 class B implements Serializable {
53     private static final long serialVersionUID = 1L;
54 
55     private void writeObject(ObjectOutputStream out) throws IOException {
56         out.defaultWriteObject();
57         out.close();
58     }
59 
60     private void readObject(ObjectInputStream in)
61         throws IOException, ClassNotFoundException
62     {
63         in.defaultReadObject();
64         in.close();
65     }
66 }
67 
68 public class SurvivePrematureClose {
69 
70     public static void main(String[] args) throws Exception {
71         writeRead(new A());
72         writeRead(new B());
73     }
74 
75     static void writeRead(Object obj) throws Exception {
76         ByteArrayOutputStream bout = new ByteArrayOutputStream();
77         ObjectOutputStream oout = new ObjectOutputStream(bout);
78         oout.writeObject(obj);
79         oout.close();
80         ObjectInputStream oin = new ObjectInputStream(
81             new ByteArrayInputStream(bout.toByteArray()));
82         oin.readObject();
83     }
84 }
