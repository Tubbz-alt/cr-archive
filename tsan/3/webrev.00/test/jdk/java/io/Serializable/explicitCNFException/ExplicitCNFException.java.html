24 /* @test
25  * @bug 4407956
26  * @summary Verify that ClassNotFoundExceptions explicitly constructed and
27  *          thrown from with custom readObject/readExternal methods are
28  *          propagated properly.
29  */
30 
31 import java.io.*;
32 
33 class A implements Serializable {
34     private static final long serialVersionUID = 1L;
35 
36     private void readObject(ObjectInputStream in)
37         throws ClassNotFoundException, IOException
38     {
39         throw new ClassNotFoundException();
40     }
41 }
42 
43 class B implements Externalizable {
44     private static final long serialVersionUID = 1L;
45 
46     public B() {}
47 
48     public void writeExternal(ObjectOutput out) throws IOException {}
49 
50     public void readExternal(ObjectInput in)
51         throws ClassNotFoundException, IOException
52     {
53         throw new ClassNotFoundException();
54     }
55 }
56 
57 public class ExplicitCNFException {
58     public static void main(String[] args) throws Exception {
59         test(new A());
60         test(new Object[]{ new A() });
61         test(new B());
62         test(new Object[]{ new B() });
63     }
64 
65     static void test(Object obj) throws Exception {
66         ByteArrayOutputStream bout = new ByteArrayOutputStream();
67         ObjectOutputStream oout = new ObjectOutputStream(bout);
68         oout.useProtocolVersion(ObjectStreamConstants.PROTOCOL_VERSION_2);
69         oout.writeObject(obj);
70         oout.close();
71         ObjectInputStream oin = new ObjectInputStream(
72             new ByteArrayInputStream(bout.toByteArray()));
73         try {
74             oin.readObject();
75             throw new Error();  // should not succeed
76         } catch (ClassNotFoundException ex) {
77         }
78     }
79 }
