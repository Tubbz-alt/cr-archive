24 /* @test
25  * @summary Verify that ClassCastException is thrown when deserializing
26  *          an object and one of its object fields is  incompatibly replaced
27  *          by either replaceObject/resolveObject.
28  *
29  */
30 import java.io.*;
31 
32 class A implements Serializable {}
33 
34 class B implements Serializable {}
35 
36 class Container implements Serializable {
37     A a = new A();
38 }
39 
40 class ReplacerObjectOutputStream extends ObjectOutputStream {
41     static B b = new B();
42   public ReplacerObjectOutputStream(OutputStream out) throws IOException {
43     super(out);
44     enableReplaceObject(true);
45   }
46 
47   protected Object replaceObject(Object obj) throws IOException {
48       if(obj instanceof A) {
49           System.err.println("replaceObject(" + obj.toString() + ") with " +
50                              b.toString());
51           return b;
52       } else return obj;
53   }
54 }
55 
56 public class BadSubstByReplace {
57     public static void main(String args[]) throws IOException, ClassNotFoundException {
58         Container c = new Container();
59         ByteArrayOutputStream baos = new ByteArrayOutputStream();
60         ReplacerObjectOutputStream out =   new ReplacerObjectOutputStream(baos);
61         out.writeObject(c);
62         out.close();
63         ObjectInputStream in =
64             new ObjectInputStream(new ByteArrayInputStream(baos.toByteArray()));
65         try {
66             c = (Container)in.readObject(); // throws IllegalArgumentException.
67             throw new Error("Should have thrown ClassCastException");
68         } catch ( ClassCastException e) {
69             System.err.println("Caught expected exception " + e.toString());
70             e.printStackTrace();
71         } finally {
72             in.close();
73         }
74     }
75 }
