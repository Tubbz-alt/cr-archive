24 /*
25  * @test
26  * @bug 4348213
27  * @build UnnamedPackageSwitchTest pkg.A
28  * @run main UnnamedPackageSwitchTest
29  * @summary Verify that deserialization allows an incoming class descriptor
30  *          representing a class in the unnamed package to be resolved to a
31  *          local class with the same name in a named package, and vice-versa.
32  */
33 
34 import java.io.ByteArrayInputStream;
35 import java.io.ByteArrayOutputStream;
36 import java.io.InputStream;
37 import java.io.IOException;
38 import java.io.ObjectOutputStream;
39 import java.io.ObjectInputStream;
40 import java.io.ObjectStreamClass;
41 import java.io.Serializable;
42 
43 class A implements Serializable {
44     private static final long serialVersionUID = 0L;
45 }
46 
47 class TestObjectInputStream extends ObjectInputStream {
48     TestObjectInputStream(InputStream in) throws IOException { super(in); }
49     protected Class resolveClass(ObjectStreamClass desc)
50         throws IOException, ClassNotFoundException
51     {
52         String name = desc.getName();
53         if (name.equals("A")) {
54             return pkg.A.class;
55         } else if (name.equals("pkg.A")) {
56             return A.class;
57         } else {
58             return super.resolveClass(desc);
59         }
60     }
61 }
62 
63 public class UnnamedPackageSwitchTest {
64     public static void main(String[] args) throws Exception {
65         ByteArrayOutputStream bout = new ByteArrayOutputStream();
66         ObjectOutputStream oout = new ObjectOutputStream(bout);
67         oout.writeObject(new A());
68         oout.writeObject(new pkg.A());
69         oout.close();
70 
71         ObjectInputStream oin = new TestObjectInputStream(
72             new ByteArrayInputStream(bout.toByteArray()));
73         oin.readObject();
74         oin.readObject();
75     }
76 }
