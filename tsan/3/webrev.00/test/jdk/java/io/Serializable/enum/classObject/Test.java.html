24 /* @test
25  * @bug 4838379
26  * @summary Verify that serialization of Class objects for enum types works
27  *          properly.
28  */
29 
30 import java.io.*;
31 
32 enum Foo { foo, bar { int i = 0; }, baz { double d = 3.0; } }
33 
34 public class Test {
35     public static void main(String[] args) throws Exception {
36         ByteArrayOutputStream bout = new ByteArrayOutputStream();
37         ObjectOutputStream oout = new ObjectOutputStream(bout);
38         Class<?>[] classes = { Enum.class, Foo.foo.getClass(),
39                             Foo.bar.getClass(), Foo.baz.getClass() };
40         for (int i = 0; i < classes.length; i++) {
41             oout.writeObject(classes[i]);
42         }
43         oout.close();
44 
45         ObjectInputStream oin = new ObjectInputStream(
46             new ByteArrayInputStream(bout.toByteArray()));
47         for (int i = 0; i < classes.length; i++) {
48             Object obj = oin.readObject();
49             if (obj != classes[i]) {
50                 throw new Error("expected " + classes[i] + ", got " + obj);
51             }
52         }
53     }
54 }
