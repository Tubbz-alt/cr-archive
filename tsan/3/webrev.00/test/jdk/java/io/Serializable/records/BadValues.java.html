 24 /*
 25  * @test
 26  * @summary Basic test for ClassNotFoundException
 27  * @run testng BadValues
 28  */
 29 
 30 import java.io.ByteArrayInputStream;
 31 import java.io.ByteArrayOutputStream;
 32 import java.io.DataOutputStream;
 33 import java.io.IOException;
 34 import java.io.ObjectInputStream;
 35 import org.testng.annotations.Test;
 36 import static java.io.ObjectStreamConstants.*;
 37 import static java.lang.System.out;
 38 import static org.testng.Assert.*;
 39 
 40 /**
 41  * Not directly related to records but provokes surrounding code, and ensures
 42  * that the record changes don't break anything.
 43  */
 44 public class BadValues {
 45 
 46     /**
 47      * Returns a stream of bytes for the given class, uid, and flags. The
 48      * stream will have no stream field values.
 49      */
 50     static byte[] byteStreamFor(String className, long uid, byte flags)
 51         throws Exception
 52     {
 53         ByteArrayOutputStream baos = new ByteArrayOutputStream();
 54         DataOutputStream dos = new DataOutputStream(baos);
 55         dos.writeShort(STREAM_MAGIC);
 56         dos.writeShort(STREAM_VERSION);
 57         dos.writeByte(TC_OBJECT);
 58         dos.writeByte(TC_CLASSDESC);
 59         dos.writeUTF(className);
 60         dos.writeLong(uid);
 61         dos.writeByte(flags);
 62         dos.writeShort(0);                // number of fields
 63         dos.writeByte(TC_ENDBLOCKDATA);   // no annotations
 64         dos.writeByte(TC_NULL);           // no superclasses
 65         dos.write(TC_ENDBLOCKDATA);       // end block - for SC_WRITE_METHOD
 66         dos.close();
 67         return baos.toByteArray();
 68     }
 69 
 70     static final Class<ClassNotFoundException> CNFE = ClassNotFoundException.class;
 71 
 72     @Test
 73     public void testNotFoundSer() throws Exception {
 74         out.println("\n---");
 75         byte[] bytes = byteStreamFor("XxYyZz", 0L, (byte)SC_SERIALIZABLE);
 76         Throwable t = expectThrows(CNFE, () -> deserialize(bytes));
 77         out.println("caught expected CNFE: " + t);
 78     }
 79 
 80     @Test
 81     public void testNotFoundSerWr() throws Exception {
 82         out.println("\n---");
 83         byte[] bytes = byteStreamFor("XxYyZz", 0L, (byte)(SC_SERIALIZABLE | SC_WRITE_METHOD));
 84         Throwable t = expectThrows(CNFE, () -> deserialize(bytes));
 85         out.println("caught expected CNFE: " + t);
 86     }
 87 
 88     @Test
 89     public void testNotFoundExt() throws Exception {
 90         out.println("\n---");
 91         byte[] bytes = byteStreamFor("AaBbCc", 0L, (byte)SC_EXTERNALIZABLE);
 92         Throwable t = expectThrows(CNFE, () -> deserialize(bytes));
 93         out.println("caught expected CNFE: " + t);
 94     }
 95 
 96     @Test
 97     public void testNotFoundExtWr() throws Exception {
 98         out.println("\n---");
 99         byte[] bytes = byteStreamFor("AaBbCc", 0L, (byte)(SC_SERIALIZABLE | SC_WRITE_METHOD));
100         Throwable t = expectThrows(CNFE, () -> deserialize(bytes));
101         out.println("caught expected CNFE: " + t);
102     }
103 
104     @SuppressWarnings("unchecked")
105     static <T> T deserialize(byte[] streamBytes)
106         throws IOException, ClassNotFoundException
107     {
108         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
109         ObjectInputStream ois  = new ObjectInputStream(bais);
110         return (T) ois.readObject();
111     }
112 }
