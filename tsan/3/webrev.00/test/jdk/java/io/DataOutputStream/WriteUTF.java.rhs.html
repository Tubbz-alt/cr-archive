24 /* @test
25  * @bug 4260284 8219196 8223254
26  * @summary Test if DataOutputStream will overcount written field.
27  * @requires (sun.arch.data.model == "64" & os.maxMemory >= 4g)
28  * @run testng/othervm -Xmx4g WriteUTF
29  */
30 
31 import java.io.ByteArrayOutputStream;
32 import java.io.DataOutputStream;
33 import java.io.IOException;
34 import java.io.UTFDataFormatException;
35 
36 import org.testng.annotations.Test;
37 
38 public class WriteUTF {
39     @Test
40     public static void overcountWrittenField() throws IOException {
41         ByteArrayOutputStream baos = new ByteArrayOutputStream();
42         DataOutputStream dos = new DataOutputStream(baos);
43         dos.writeUTF("Hello, World!");  // 15
44         dos.flush();
45         if  (baos.size() != dos.size())
46             throw new RuntimeException("Miscounted bytes in DataOutputStream.");
47     }
48 
49     private static void writeUTF(int size) throws IOException {
50         // this character gives 3 bytes when encoded using UTF-8
51         String s = "\u0800".repeat(size);
52         ByteArrayOutputStream bos = new ByteArrayOutputStream();
53         DataOutputStream dos = new DataOutputStream(bos);
54         dos.writeUTF(s);
55     }
56 
57     @Test(expectedExceptions = UTFDataFormatException.class)
58     public void utfDataFormatException() throws IOException {
59         writeUTF(1 << 16);
60     }
61 
62     // Without 8219196 fix, throws ArrayIndexOutOfBoundsException instead of
63     // expected UTFDataFormatException. Requires 4GB of heap (-Xmx4g) to run
64     // without throwing an OutOfMemoryError.
65     @Test(expectedExceptions = UTFDataFormatException.class)
66     public void arrayIndexOutOfBoundsException() throws IOException {
67         writeUTF(Integer.MAX_VALUE / 3 + 1);
68     }
69 }
