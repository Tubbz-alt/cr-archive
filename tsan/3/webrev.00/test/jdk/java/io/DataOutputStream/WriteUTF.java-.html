24 /* @test
25    @bug 4260284
26    @summary Test if DataOutputStream will overcount written field.
27 */
28 
29 import java.io.*;
30 
31 public class WriteUTF {
32     public static void main(String[] args) throws Exception {
33         ByteArrayOutputStream baos = new ByteArrayOutputStream();
34         DataOutputStream dos = new DataOutputStream(baos);
35         dos.writeUTF("Hello, World!");  // 15
36         dos.flush();
37         if  (baos.size() != dos.size())
38             throw new RuntimeException("Miscounted bytes in DataOutputStream.");
39     }
40 }
