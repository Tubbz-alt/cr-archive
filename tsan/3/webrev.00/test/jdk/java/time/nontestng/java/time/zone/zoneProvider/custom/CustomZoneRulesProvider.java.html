24 package custom;
25 
26 import java.time.ZoneId;
27 import java.time.zone.ZoneRules;
28 import java.time.zone.ZoneRulesProvider;
29 import java.util.Set;
30 import java.util.NavigableMap;
31 import java.util.TreeMap;
32 
33 public class CustomZoneRulesProvider extends ZoneRulesProvider {
34     @Override
35     protected Set<String> provideZoneIds() {
36         return Set.of("Custom/Timezone");
37     }
38 
39     @Override
40     protected ZoneRules provideRules(String zoneId, boolean forCaching) {
41         return ZoneId.of("UTC").getRules();
42     }
43 
44     @Override
45     protected NavigableMap<String, ZoneRules> provideVersions(String zoneId) {
46         var map = new TreeMap<String, ZoneRules>();
47         map.put("bogusVersion", getRules(zoneId, false));
48         return map;
49     }
50 }
