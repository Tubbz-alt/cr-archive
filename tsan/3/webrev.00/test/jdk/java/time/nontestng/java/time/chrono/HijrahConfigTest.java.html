24 import java.nio.file.Files;
25 import java.nio.file.Path;
26 
27 import tests.Helper;
28 import tests.JImageGenerator;
29 
30 /*
31  * @test
32  * @summary Tests whether a custom Hijrah configuration properties file works correctly
33  * @bug 8187987
34  * @requires (vm.compMode != "Xcomp" & os.maxMemory >= 2g)
35  * @library /tools/lib
36  * @modules java.base/jdk.internal.jimage
37  *          jdk.jdeps/com.sun.tools.classfile
38  *          jdk.jlink/jdk.tools.jimage
39  *          jdk.compiler
40  * @build HijrahConfigCheck tests.*
41  * @run main/othervm -Xmx1g HijrahConfigTest
42  */
43 public class HijrahConfigTest {
44 
45     private static final String TEST_CONFIG = "hijrah-config-Hijrah-test_islamic-test.properties";
46 
47     public static void main(String[] args) throws Exception {
48         Helper helper = Helper.newHelper();
49         if (helper == null) {
50             System.err.println("Test not run");
51             return;
52         }
53 
54         // Create the test JDK image
55         Path outputPath = helper.createNewImageDir("HijrahConfigTest");
56         JImageGenerator.getJLinkTask()
57                 .output(outputPath)
58                 .addMods("java.base")
59                 .call().assertSuccess();
60 
61         // Install the test hijrah configuration properties
62         Path confPath = outputPath.resolve("conf").resolve("chronology");
63         Files.createDirectory(confPath);
64         Files.copy(Path.of(System.getProperty("test.src"), TEST_CONFIG),
65                 confPath.resolve(TEST_CONFIG));
66 
67         // Run tests
68         Path launcher = outputPath.resolve("bin").resolve("java");
69         ProcessBuilder builder = new ProcessBuilder(
70                 launcher.toAbsolutePath().toString(), "-ea", "-esa", "HijrahConfigCheck");
71         Process p = builder.inheritIO().start();
72         p.waitFor();
73         int exitValue = p.exitValue();
74         if (exitValue != 0) {
75             throw new RuntimeException("HijrahConfigTest failed. Exit value: " + exitValue);
76         }
77     }
78 }
