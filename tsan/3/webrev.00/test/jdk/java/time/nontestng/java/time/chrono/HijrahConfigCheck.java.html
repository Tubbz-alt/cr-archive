24 import java.time.LocalDate;
25 import java.time.LocalDateTime;
26 import java.time.LocalTime;
27 import java.time.ZoneOffset;
28 import java.time.chrono.ChronoLocalDateTime;
29 import java.time.chrono.Chronology;
30 import java.util.Locale;
31 
32 public class HijrahConfigCheck {
33     private static final String CALTYPE = "islamic-test";
34 
35     public static void main(String... args) {
36         // Availability test
37         if (Chronology.getAvailableChronologies().stream()
38                 .filter(c -> c.getCalendarType().equals(CALTYPE))
39                 .count() != 1) {
40             throw new RuntimeException(CALTYPE + " chronology was not found, or " +
41                     "appeared more than once in Chronology.getAvailableChronologies()");
42         }
43 
44         // Instantiation tests
45         Chronology c1 = Chronology.of(CALTYPE);
46         Chronology c2 = Chronology.ofLocale(Locale.forLanguageTag("und-u-ca-" + CALTYPE ));
47         if (!c1.equals(c2)) {
48             throw new RuntimeException(CALTYPE + " chronologies differ. c1: " + c1 +
49                                         ", c2: " + c2);
50         }
51 
52         // Date test
53         // 2020-01-10 is AH 1000-01-10 in islamic-test config
54         LocalDateTime iso = LocalDateTime.of(LocalDate.of(2020, 1, 10), LocalTime.MIN);
55         ChronoLocalDateTime hijrah = c1.date(1000, 1, 10).atTime(LocalTime.MIN);
56         if (!iso.toInstant(ZoneOffset.UTC).equals(hijrah.toInstant(ZoneOffset.UTC))) {
57             throw new RuntimeException("test Hijrah date is incorrect. LocalDate: " +
58                     iso + ", test date: " + hijrah);
59         }
60     }
61 }
