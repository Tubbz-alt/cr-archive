24 package custom;
25 
26 import java.util.Locale;
27 import java.util.TimeZone;
28 import java.util.spi.TimeZoneNameProvider;
29 
30 public class CustomTimeZoneNameProvider extends TimeZoneNameProvider {
31 
32     public static final String ZONE_ID = "Custom/Timezone";
33 
34     @Override
35     public String getDisplayName(String ID, boolean daylight, int style, Locale locale) {
36         if (ZONE_ID.equals(ID)) {
37             switch (style) {
38                 case TimeZone.SHORT:
39                     if (daylight) {
40                         return "CUST_ST";
41                     } else {
42                         return "CUST_WT";
43                     }
44                 case TimeZone.LONG:
45                     if (daylight) {
46                         return "Custom Summer Time";
47                     } else {
48                         return "Custom Winter Time";
49                     }
50             }
51         }
52         return null;
53     }
54 
55     @Override
56     public String getGenericDisplayName(String ID, int style, Locale locale) {
57         if (ZONE_ID.equals(ID)) {
58             switch (style) {
59                 case TimeZone.SHORT:
60                     return "Custom Time";
61                 case TimeZone.LONG:
62                     return "Custom Timezone Time";
63             }
64         }
65         return null;
66     }
67 
68     @Override
69     public boolean isSupportedLocale(Locale locale) {
70         return true;
71     }
72 
73     @Override
74     public Locale[] getAvailableLocales() {
75         return new Locale[]{
76             Locale.getDefault()
77         };
78     }
79 }
