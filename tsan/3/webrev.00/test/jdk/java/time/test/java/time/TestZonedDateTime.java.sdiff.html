60 package test.java.time;
61 
62 import static org.testng.Assert.assertEquals;
63 
64 import java.time.Duration;
65 import java.time.LocalDateTime;
66 import java.time.ZonedDateTime;
67 import java.time.ZoneId;
68 
69 import org.testng.annotations.Test;
70 
71 /**
72  * Test ZonedDateTime.
73  *
74  * @bug 8211990
75  */
76 @Test
77 public class TestZonedDateTime extends AbstractTest {
78 
79     @Test
80     public void test_immutable() {
81         assertImmutable(ZonedDateTime.class);
82     }
83 
84     @Test
85     public void test_duration() {
86         ZoneId tokyo = ZoneId.of("Asia/Tokyo");
87         ZoneId sanJose = ZoneId.of("America/Los_Angeles");
88 
89         ZonedDateTime end = ZonedDateTime.of(LocalDateTime.MAX, sanJose);
90         ZonedDateTime start = end.withZoneSameLocal(tokyo);
91 
92         assertEquals(Duration.between(start, end), Duration.ofHours(17));
93     }
94 }
