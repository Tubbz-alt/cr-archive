763     public void test_localizedBy(Locale locale, Chronology chrono, ZoneId zone,
764                                 Chronology chronoExpected, ZoneId zoneExpected,
765                                 String formatExpected) {
766         DateTimeFormatter dtf =
767             DateTimeFormatter.ofLocalizedDateTime(FormatStyle.FULL, FormatStyle.FULL)
768                 .withChronology(chrono).withZone(zone).localizedBy(locale);
769         assertEquals(dtf.getChronology(), chronoExpected);
770         assertEquals(dtf.getZone(), zoneExpected);
771         String formatted = dtf.format(ZDT);
772         assertEquals(formatted, formatExpected);
773         assertEquals(dtf.parse(formatted, ZonedDateTime::from),
774             zoneExpected != null ? ZDT.withZoneSameInstant(zoneExpected) : ZDT);
775     }
