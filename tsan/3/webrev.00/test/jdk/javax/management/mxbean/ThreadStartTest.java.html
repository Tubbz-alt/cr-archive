24 /*
25  * @test
26  * @bug 8231666
27  * @summary Test checks that new threads could be successfully started when the thread
28  * table introduced in 8185005 is growing. The test enables the thread table by calling
29  * ThreadMXBean.getThreadInfo() and then creates a number of threads to force the thread
30  * table to grow.
31  *
32  * @run main ThreadStartTest
33  */
34 
35 import java.lang.management.ManagementFactory;
36 import java.lang.management.ThreadMXBean;
37 
38 public class ThreadStartTest {
39     public static void main(String[] args) {
40 
41         ThreadMXBean mbean = ManagementFactory.getThreadMXBean();
42         // Enable thread table
43         mbean.getThreadInfo(Thread.currentThread().getId());
44 
45         // Create a large number of threads to make the thread table grow
46         for (int i = 0; i < 1000; i++) {
47             Thread t = new Thread(() -> {
48                 try {
49                     Thread.sleep(1000);
50                 } catch (InterruptedException ex) {
51                 }
52             });
53             t.start();
54         }
55     }
56 }
