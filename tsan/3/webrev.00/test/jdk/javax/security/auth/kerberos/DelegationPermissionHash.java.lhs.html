26 import javax.security.auth.kerberos.DelegationPermission;
27 
28 /*
29  * @test
30  * @bug 8129575
31  * @summary Checks if DelegationPermission.hashCode() works fine
32  */
33 public class DelegationPermissionHash {
34 
35     static final String princ1 = "backup/bar.example.com@EXAMPLE.COM";
36     static final String princ2 = "backup/foo.example.com@EXAMPLE.COM";
37     static final String ONE_SPACE = " ";
38     static final String TWO_SPACES = "  ";
39     static final String QUOTE = "\"";
40 
41     public static void main(String[] args) {
42         DelegationPermission one = new DelegationPermission(
43                 QUOTE + princ1 + QUOTE + ONE_SPACE + QUOTE + princ2 + QUOTE);
44         DelegationPermission two = new DelegationPermission(
45                 QUOTE + princ1 + QUOTE + TWO_SPACES + QUOTE + princ2 + QUOTE);
46 
47         System.out.println("one.getName() = " + one.getName());
48         System.out.println("two.getName() = " + two.getName());
49 
50         if (!one.implies(two) || !two.implies(one)) {
51             throw new RuntimeException("Test failed: "
52                     + "one and two don't imply each other");
53         }
54 
55         if (!one.equals(two)) {
56             throw new RuntimeException("Test failed: one is not equal to two");
57         }
58 
59         System.out.println("one.hashCode() = " + one.hashCode());
60         System.out.println("two.hashCode() = " + two.hashCode());
61         if (one.hashCode() != two.hashCode()) {
62             throw new RuntimeException("Test failed: hash codes are not equal");
63         }
64 
65         System.out.println("Test passed");
66     }
67 }
