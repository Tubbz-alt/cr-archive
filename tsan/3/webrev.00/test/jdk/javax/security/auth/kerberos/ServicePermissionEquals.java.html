24 /*
25  * @test
26  * @bug 8231365
27  * @library /test/lib
28  * @summary ServicePermission::equals doesn't comply to the spec
29  */
30 
31 import jdk.test.lib.Asserts;
32 
33 import javax.security.auth.kerberos.ServicePermission;
34 
35 public class ServicePermissionEquals {
36     public static void main(String[] args) throws Exception {
37         ServicePermission p1 = new ServicePermission("user@REALM", "initiate");
38         ServicePermission p2 = new ServicePermission("user@REALM", "accept");
39         ServicePermission p3 = new ServicePermission("user@REALM", "initiate,accept");
40 
41         Asserts.assertNotEquals(p1.hashCode(), p2.hashCode());
42         Asserts.assertNotEquals(p1.hashCode(), p3.hashCode());
43         Asserts.assertNotEquals(p3.hashCode(), p2.hashCode());
44 
45         Asserts.assertFalse(p1.equals(p2));
46         Asserts.assertFalse(p1.equals(p3));
47         Asserts.assertFalse(p3.equals(p2));
48     }
49 }
