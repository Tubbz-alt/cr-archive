24 import javax.security.auth.kerberos.DelegationPermission;
25 
26 /*
27  * @test
28  * @bug 8129575
29  * @summary Checks if DelegationPermission.hashCode() works fine
30  */
31 public class DelegationPermissionHash {
32 
33     static final String princ1 = "backup/bar.example.com@EXAMPLE.COM";
34     static final String princ2 = "backup/foo.example.com@EXAMPLE.COM";
35     static final String ONE_SPACE = " ";
36     static final String TWO_SPACES = "  ";
37     static final String QUOTE = "\"";
38 
39     public static void main(String[] args) {
40         DelegationPermission one = new DelegationPermission(
41                 QUOTE + princ1 + QUOTE + ONE_SPACE + QUOTE + princ2 + QUOTE);
42         DelegationPermission two = new DelegationPermission(
43                 QUOTE + princ1 + QUOTE + TWO_SPACES + QUOTE + princ2 + QUOTE);
44 
45         System.out.println("one.getName() = " + one.getName());
46         System.out.println("two.getName() = " + two.getName());
47 
48         if (!one.implies(two) || !two.implies(one)) {
49             throw new RuntimeException("Test failed: "
50                     + "one and two don't imply each other");
51         }
52 
53         if (!one.equals(two)) {
54             throw new RuntimeException("Test failed: one is not equal to two");
55         }
56 
57         System.out.println("one.hashCode() = " + one.hashCode());
58         System.out.println("two.hashCode() = " + two.hashCode());
59         if (one.hashCode() != two.hashCode()) {
60             throw new RuntimeException("Test failed: hash codes are not equal");
61         }
62 
63         System.out.println("Test passed");
64     }
65 }
