52 import javax.net.ssl.SSLEngine;
53 import java.security.Security;
54 
55 /**
56  * Test common DTLS cipher suites.
57  */
58 public class CipherSuite extends DTLSOverDatagram {
59 
60     // use the specific cipher suite
61     volatile static String cipherSuite;
62 
63     public static void main(String[] args) throws Exception {
64         if (args.length > 1 && "re-enable".equals(args[1])) {
65             Security.setProperty("jdk.tls.disabledAlgorithms", "");
66         }
67 
68         cipherSuite = args[0];
69 
70         CipherSuite testCase = new CipherSuite();
71         testCase.runTest(testCase);
72     }
73 
74     @Override
75     SSLEngine createSSLEngine(boolean isClient) throws Exception {
76         SSLEngine engine = super.createSSLEngine(isClient);
77 
78         if (isClient) {
79             engine.setEnabledCipherSuites(new String[]{cipherSuite});
80         }
81 
82         return engine;
83     }
84 }
