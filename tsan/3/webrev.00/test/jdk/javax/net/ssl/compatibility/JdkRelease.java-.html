24 /*
25  * JDK major versions.
26  */
27 public enum JdkRelease {
28 
29     JDK7(7, "1.7"),
30     JDK8(8, "1.8"),
31     JDK9(9, "9"),
32     JDK10(10, "10"),
33     JDK11(11, "11"),
34     JDK12(12, "12");
35 
36     public final int sequence;
37     public final String release;
38 
39     private JdkRelease(int sequence, String release) {
40         this.sequence = sequence;
41         this.release = release;
42     }
43 
44     public static JdkRelease getRelease(String jdkVersion) {
45         if (jdkVersion.startsWith(JDK7.release)) {
46             return JDK7;
47         } else if (jdkVersion.startsWith(JDK8.release)) {
48             return JDK8;
49         } else if (jdkVersion.startsWith(JDK9.release)) {
50             return JDK9;
51         } else if (jdkVersion.startsWith(JDK10.release)) {
52             return JDK10;
53         } else if (jdkVersion.startsWith(JDK11.release)) {
54             return JDK11;
55         } else if (jdkVersion.startsWith(JDK12.release)) {
56             return JDK12;
57         }
58 
59         return null;
60     }
61 }
