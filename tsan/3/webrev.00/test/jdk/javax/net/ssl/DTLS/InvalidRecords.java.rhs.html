37 import java.net.DatagramPacket;
38 import java.net.SocketAddress;
39 
40 /**
41  * Test that if handshake messages are crasged, the handshake would fail
42  * because of handshaking hash verification.
43  */
44 public class InvalidRecords extends DTLSOverDatagram {
45     boolean needInvalidRecords = true;
46 
47     public static void main(String[] args) throws Exception {
48         InvalidRecords testCase = new InvalidRecords();
49         testCase.runTest(testCase);
50     }
51 
52     @Override
53     public boolean isGoodJob() {
54         return false;
55     }
56 
57     @Override
58     DatagramPacket createHandshakePacket(byte[] ba, SocketAddress socketAddr) {
59         if ((ba.length >= 60) &&
60                 (ba[0x00] == (byte)0x16) && (ba[0x0D] == (byte)0x01) &&
61                 (ba[0x3B] == (byte)0x00) && (ba[0x3C] > 0)) {
62 
63             // ba[0x00]: record type
64             // ba[0x0D]: handshake type
65             // ba[0x3B]: length of session ID
66             // ba[0x3C]: length of cookie
67 
68             if (!needInvalidRecords) {
69                 // The 2nd ClientHello with cookie.  The 1st one should be
70                 // rejected as expected.
71                 //
72                 // This may happen if the last few bytes of the packet are
73                 // for supported_version extension.
74                 throw new RuntimeException(
75                     "the crashed handshake message was rejected as expected");
76             }
77 
78             // ClientHello with cookie
79             needInvalidRecords = false;
80             System.out.println("invalidate ClientHello message");
81             if (ba[ba.length - 1] == (byte)0xFF) {
82                 ba[ba.length - 1] = (byte)0xFE;
83             } else {
84                 ba[ba.length - 1] = (byte)0xFF;
85             }
86         }
87 
88         return super.createHandshakePacket(ba, socketAddr);
89     }
90 }
