24 // SunJSSE does not support dynamic system properties, no way to re-use
25 // system properties in samevm/agentvm mode.

27 /*
28  * @test
29  * @bug 8043758
30  * @summary Datagram Transport Layer Security (DTLS)
31  * @modules java.base/sun.security.util
32  * @build DTLSOverDatagram
33  * @run main/othervm -Djdk.tls.client.enableStatusRequestExtension=false
34  *      NoMacInitialClientHello
35  */
36 
37 import java.net.DatagramPacket;
38 import java.net.SocketAddress;
39 
40 /**
41  * Test that a server is able to discard invalid initial ClientHello silently.
42  */
43 public class NoMacInitialClientHello extends DTLSOverDatagram {
44     boolean needInvalidRecords = true;
45 
46     public static void main(String[] args) throws Exception {
47         System.setProperty("jdk.tls.useExtendedMasterSecret", "false");
48         NoMacInitialClientHello testCase = new NoMacInitialClientHello();
49         testCase.runTest(testCase);
50     }
51 
52     @Override
53     DatagramPacket createHandshakePacket(byte[] ba, SocketAddress socketAddr) {
54         if (needInvalidRecords && (ba.length >= 60) &&
55             (ba[0] == (byte)0x16) && (ba[13] == (byte)0x01)) {  // ClientHello
56 
57             needInvalidRecords = false;
58             System.out.println("invalidate ClientHello message");
59             if (ba[ba.length - 1] == (byte)0xFF) {
60                 ba[ba.length - 1] = (byte)0xFE;
61             } else {
62                 ba[ba.length - 1] = (byte)0xFF;
63             }
64         }
65 
66         return super.createHandshakePacket(ba, socketAddr);
67     }
68 }
