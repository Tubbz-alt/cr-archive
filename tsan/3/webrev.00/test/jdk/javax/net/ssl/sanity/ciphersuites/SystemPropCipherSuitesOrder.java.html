 23 import java.util.Arrays;
 24 import javax.net.ssl.SSLServerSocket;
 25 import javax.net.ssl.SSLSocket;
 26 
 27 /*
 28  * @test
 29  * @bug 8234728
 30  * @library /javax/net/ssl/templates
 31  *          /javax/net/ssl/TLSCommon
 32  * @summary Test TLS ciphersuites order set through System properties
 33  * @run main/othervm
 34  *      -Djdk.tls.client.cipherSuites=TLS_CHACHA20_POLY1305_SHA256,TLS_AES_128_GCM_SHA256,TLS_AES_256_GCM_SHA384
 35  *      -Djdk.tls.server.cipherSuites=TLS_AES_256_GCM_SHA384,TLS_CHACHA20_POLY1305_SHA256,TLS_AES_128_GCM_SHA256
 36  *      SystemPropCipherSuitesOrder TLSv1.3
 37  * @run main/othervm
 38  *      -Djdk.tls.client.cipherSuites=TLS_CHACHA20_POLY1305_SHA256,TLS_AES_128_GCM_SHA256,TLS_AES_256_GCM_SHA384
 39  *      SystemPropCipherSuitesOrder TLSv1.3
 40  * @run main/othervm
 41  *      -Djdk.tls.server.cipherSuites=TLS_AES_128_GCM_SHA256,TLS_CHACHA20_POLY1305_SHA256,TLS_AES_256_GCM_SHA384
 42  *      SystemPropCipherSuitesOrder TLSv1.3
 43  * @run main/othervm
 44  *      -Djdk.tls.client.cipherSuites=TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256,TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384
 45  *      -Djdk.tls.server.cipherSuites=TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256,TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384
 46  *      SystemPropCipherSuitesOrder TLSv1.2
 47  * @run main/othervm
 48  *      -Djdk.tls.client.cipherSuites=TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256,TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384
 49  *      SystemPropCipherSuitesOrder TLSv1.2
 50  * @run main/othervm
 51  *      -Djdk.tls.server.cipherSuites=TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256,TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384
 52  *      SystemPropCipherSuitesOrder TLSv1.2
 53  * @run main/othervm
 54  *      -Djdk.tls.client.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 55  *      -Djdk.tls.server.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 56  *      SystemPropCipherSuitesOrder TLSv1.1
 57  * @run main/othervm
 58  *      -Djdk.tls.client.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 59  *      SystemPropCipherSuitesOrder TLSv1.1
 60  * @run main/othervm
 61  *      -Djdk.tls.server.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 62  *      SystemPropCipherSuitesOrder TLSv1.1
 63  * @run main/othervm
 64  *      -Djdk.tls.client.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 65  *      -Djdk.tls.server.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 66  *      SystemPropCipherSuitesOrder TLSv1
 67  * @run main/othervm
 68  *      -Djdk.tls.client.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 69  *      SystemPropCipherSuitesOrder TLSv1
 70  * @run main/othervm
 71  *      -Djdk.tls.server.cipherSuites=TLS_RSA_WITH_AES_128_CBC_SHA,TLS_RSA_WITH_AES_256_CBC_SHA
 72  *      SystemPropCipherSuitesOrder TLSv1
 73  */
 74 public class SystemPropCipherSuitesOrder extends SSLSocketTemplate {
 75 
 76     private final String protocol;
 77     private static String[] servercipherSuites;
 78     private static String[] clientcipherSuites;
 79 
 80     public static void main(String[] args) {
 81         servercipherSuites
 82                 = toArray(System.getProperty("jdk.tls.server.cipherSuites"));
 83         clientcipherSuites
 84                 = toArray(System.getProperty("jdk.tls.client.cipherSuites"));
 85         System.out.printf("SYSTEM PROPERTIES: ServerProp:%s - ClientProp:%s%n",
 86                 Arrays.deepToString(servercipherSuites),
 87                 Arrays.deepToString(clientcipherSuites));
 88 
 89         try {
 90             new SystemPropCipherSuitesOrder(args[0]).run();
 91         } catch (Exception e) {
 92             throw new RuntimeException(e);
 93         }
 94     }
 95 
 96     private SystemPropCipherSuitesOrder(String protocol) {
 97         this.protocol = protocol;
 98     }
 99 
100     // Servers are configured before clients, increment test case after.
101     @Override
102     protected void configureClientSocket(SSLSocket socket) {
103         socket.setEnabledProtocols(new String[]{protocol});
104     }
105 
106     @Override
107     protected void configureServerSocket(SSLServerSocket serverSocket) {
108         serverSocket.setEnabledProtocols(new String[]{protocol});
109     }
110 
111     protected void runServerApplication(SSLSocket socket) throws Exception {
112         if (servercipherSuites != null) {
113             System.out.printf("SERVER: SystemProperty:%s - "
114                     + "getEnabledCipherSuites:%s%n",
115                     Arrays.deepToString(servercipherSuites),
116                     Arrays.deepToString(socket.getEnabledCipherSuites()));
117         }
118         if (servercipherSuites != null && !Arrays.equals(
119                 servercipherSuites, socket.getEnabledCipherSuites())) {
120             throw new RuntimeException("Unmatched server side CipherSuite order");
121         }
122         super.runServerApplication(socket);
123     }
124 
125     protected void runClientApplication(SSLSocket socket) throws Exception {
126         if (clientcipherSuites != null) {
127             System.out.printf("CLIENT: SystemProperty:%s - "
128                     + "getEnabledCipherSuites:%s%n",
129                     Arrays.deepToString(clientcipherSuites),
130                     Arrays.deepToString(socket.getEnabledCipherSuites()));
131         }
132         if (clientcipherSuites != null && !Arrays.equals(clientcipherSuites,
133                 socket.getEnabledCipherSuites())) {
134             throw new RuntimeException("Unmatched client side CipherSuite order");
135         }
136         super.runClientApplication(socket);
137     }
138 
139     private static String[] toArray(String prop) {
140         return (prop != null) ? prop.split(",") : null;
141     }
142 }
