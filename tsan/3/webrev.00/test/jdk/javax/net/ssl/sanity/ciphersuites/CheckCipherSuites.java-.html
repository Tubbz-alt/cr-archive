202     private static void showSuites(String[] suites) {
203         if ((suites == null) || (suites.length == 0)) {
204             System.out.println("<none>");
205         }
206         for (int i = 0; i < suites.length; i++) {
207             System.out.println("  " + suites[i]);
208         }
209     }

211     public static void main(String[] args) throws Exception {
212         long start = System.currentTimeMillis();
213 
214         if (args.length != 1) {
215             throw new Exception("One arg required");
216         }
217 
218         String[] ENABLED;
219         String[] SUPPORTED;
220         if (args[0].equals("default")) {
221             ENABLED = ENABLED_DEFAULT;
222             SUPPORTED = SUPPORTED_DEFAULT;
223         } else if (args[0].equals("limited")) {
224             Security.setProperty("crypto.policy", "limited");
225             ENABLED = ENABLED_LIMITED;
226             SUPPORTED = SUPPORTED_LIMITED;
227         } else {
228             throw new Exception("Illegal argument");
229         }
230 
231         SSLSocketFactory factory = (SSLSocketFactory)SSLSocketFactory.getDefault();
232         SSLSocket socket = (SSLSocket)factory.createSocket();
233         String[] enabled = socket.getEnabledCipherSuites();
234 
235         System.out.println("Default enabled ciphersuites:");
236         showSuites(enabled);
237 
238         if (Arrays.equals(ENABLED, enabled) == false) {
239             System.out.println("*** MISMATCH, should be ***");
240             showSuites(ENABLED);
241             throw new Exception("Enabled ciphersuite mismatch");
242         }
243         System.out.println("OK");
244         System.out.println();
245 
246         String[] supported = socket.getSupportedCipherSuites();
247         System.out.println("Supported ciphersuites:");
248         showSuites(supported);
249 
250         if (Arrays.equals(SUPPORTED, supported) == false) {
251             System.out.println("*** MISMATCH, should be ***");
252             showSuites(SUPPORTED);
253             throw new Exception("Supported ciphersuite mismatch");
254         }
255         System.out.println("OK");
256 
257         long end = System.currentTimeMillis();
258         System.out.println("Done (" + (end - start) + " ms).");
259     }
