37 import java.util.List;
38 import java.util.ArrayList;
39 import java.net.DatagramPacket;
40 import java.net.SocketAddress;
41 import javax.net.ssl.SSLEngine;
42 
43 /**
44  * Test that DTLS implementation is able to do retransmission internally
45  * automatically if packet get lost.
46  */
47 public class Retransmission extends DTLSOverDatagram {
48     boolean needPacketLoss = true;
49 
50     public static void main(String[] args) throws Exception {
51         Retransmission testCase = new Retransmission();
52         testCase.runTest(testCase);
53     }
54 
55     @Override
56     boolean produceHandshakePackets(SSLEngine engine, SocketAddress socketAddr,
57             String side, List<DatagramPacket> packets) throws Exception {
58 
59         boolean finished = super.produceHandshakePackets(
60                 engine, socketAddr, side, packets);
61 
62         if (!needPacketLoss || (!engine.getUseClientMode())) {
63             return finished;
64         }
65 
66         List<DatagramPacket> parts = new ArrayList<>();
67         int lostSeq = 2;
68         for (DatagramPacket packet : packets) {
69             lostSeq--;
70             if (lostSeq == 0) {
71                 needPacketLoss = false;
72                 // loss this packet
73                 System.out.println("Loss a packet");
74                 continue;
75             }
76 
77             parts.add(packet);
78         }
79 
80         packets.clear();
81         packets.addAll(parts);
82 
83         return finished;
84     }
85 }
