23 /**
24   * @test
25   * @bug 8197785
26   * @summary Tests if AccessibleBundle will reload the ResourceBundle for every
27   * call to toDisplayString.
28   * @run main AccessibilityBundleMemoryLeakTest
29   */
30 import java.lang.reflect.Field;
31 import java.util.Hashtable;
32 import java.util.Locale;
33 
34 import javax.accessibility.AccessibleRole;
35 import javax.accessibility.AccessibleBundle;
36 
37 public class AccessibilityBundleMemoryLeakTest extends AccessibleRole {
38     public AccessibilityBundleMemoryLeakTest() {
39         super("");
40     }
41 
42     public static void main(String... args) throws Exception {
43         AccessibilityBundleMemoryLeakTest role = new AccessibilityBundleMemoryLeakTest();
44 
45         Field field = AccessibleBundle.class.getDeclaredField("table");
46         field.setAccessible(true);
47 
48         final Hashtable table = (Hashtable)field.get(role);
49         Locale locale = Locale.getDefault();
50 
51         role.toDisplayString();
52         Object obj = table.get(locale);
53 
54         role.toDisplayString();
55         Object obj1 = table.get(locale);
56 
57         if (obj != obj1) {
58             throw new RuntimeException("Test case failed: AccessibleBundle allocates new value for existing key!");
59         }
60 
61         System.out.println("Test passed.");
62 
63     }
64 }
