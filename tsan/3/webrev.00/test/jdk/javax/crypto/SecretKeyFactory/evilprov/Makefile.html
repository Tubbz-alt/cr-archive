24 # Java paths
25 #JAVA_BASE=PATH_TO_JAVA_IMG_DIR
26 JAVABIN=$(JAVA_BASE)/bin
27 JAVAC=$(JAVABIN)/javac
28 JAVA=$(JAVABIN)/java
29 JAR=$(JAVABIN)/jar 
30 JARSIGNER=$(JAVABIN)/jarsigner
31 
32 # Compile-time flags and paths
33 JFLAGS=-Xlint:all
34 SRCPATH=com/evilprovider
35 CLASSDST=classes
36 
37 PROVJAR=evilprov.jar
38 KSTORE=PATH_TO_KEYSTORE
39 KALIAS=PLACE_SIGNING_ALIAS_HERE
40 MODVER=1.0
41 
42 all: $(PROVJAR)
43 
44 %.class: %.java
45 	mkdir -p $(CLASSDST)
46 	$(JAVAC) -d $(CLASSDST) $(JFLAGS) $<
47 
48 $(PROVJAR): $(SRCPATH)/EvilHmacSHA1.class $(SRCPATH)/EvilProvider.class module-info.class
49 	$(JAR) --create --file $(PROVJAR) --module-version $(MODVER) -C $(CLASSDST) .
50 
51 signed: $(PROVJAR)
52 	jarsigner -keystore $(KSTORE) $(PROVJAR).jar $(KALIAS)
53 
54 clean:
55 	rm -rf $(CLASSDST) $(PROVJAR)
