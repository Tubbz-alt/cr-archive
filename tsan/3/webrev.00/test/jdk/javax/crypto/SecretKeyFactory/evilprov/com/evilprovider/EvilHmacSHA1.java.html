26 package com.evilprovider;
27 
28 import java.security.*;
29 import java.security.spec.*;
30 import java.nio.ByteBuffer;
31 
32 import javax.crypto.*;
33 
34 public final class EvilHmacSHA1 extends MacSpi {
35     private final Mac internalMac;
36 
37     public EvilHmacSHA1() throws GeneralSecurityException {
38         internalMac = Mac.getInstance("HmacSHA1", "SunJCE");
39     }
40 
41     @Override
42     protected byte[] engineDoFinal() {
43         return internalMac.doFinal();
44     }
45 
46     @Override
47     protected int engineGetMacLength() {
48         return internalMac.getMacLength();
49     }
50 
51     @Override
52     protected void engineInit(Key key, AlgorithmParameterSpec spec)
53             throws InvalidKeyException, InvalidAlgorithmParameterException {
54         SecretKey sKey;
55         if (key instanceof SecretKey) {
56             sKey = (SecretKey)key;
57         } else {
58             throw new IllegalArgumentException("Key must be a SecretKey");
59         }
60 
61         byte[] sKeyEnc = sKey.getEncoded();
62         int keyBits = sKeyEnc.length * 8;
63         if (keyBits < 160) {
64             throw new IllegalArgumentException("Key must be at least 160 bits");
65         }
66 
67         // Pass through to init
68         internalMac.init(key, spec);
69     }
70 
71     @Override
72     protected void engineReset() {
73         internalMac.reset();
74     }
75 
76     @Override
77     protected void engineUpdate(byte input) {
78         internalMac.update(input);
79     }
80 
81     @Override
82     protected void engineUpdate(byte[] input, int offset, int len) {
83         internalMac.update(input, offset, len);
84     }
85 
86     @Override
87     protected void engineUpdate(ByteBuffer input) {
88         internalMac.update(input);
89     }
90 }
