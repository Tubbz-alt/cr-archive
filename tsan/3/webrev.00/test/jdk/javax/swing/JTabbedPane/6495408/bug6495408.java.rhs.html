24 import javax.swing.*;
25 import java.awt.*;
26 /*
27  * @test
28  * @key headful
29  * @bug 6495408
30  * @summary REGRESSION: JTabbedPane throws ArrayIndexOutOfBoundsException
31  * @author Alexander Potochkin
32  * @run main bug6495408
33  */
34 
35 public class bug6495408 {
36 
37     static JTabbedPane tabbedPane;
38     static JFrame frame;
39 
40     public static void main(String[] args) throws Exception {
41         try {
42             final Robot robot = new Robot();
43             robot.setAutoDelay(50);
44 
45             SwingUtilities.invokeAndWait(new Runnable() {
46 
47                 public void run() {
48                     frame = new JFrame();
49                     frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
50                     tabbedPane = new JTabbedPane();
51                     tabbedPane.setTabPlacement(JTabbedPane.LEFT);
52                     tabbedPane.addTab("Hello", null);
53                     frame.add(tabbedPane);
54                     frame.setSize(400, 400);
55                     frame.setLocationRelativeTo(null);
56                     frame.setVisible(true);
57                 }
58             });
59 
60             robot.waitForIdle();
61 
62             final Rectangle d = new Rectangle();
63             final Point p = new Point();
64 
65             for (int i = 0; i < 7; i++) {
66                 SwingUtilities.invokeLater(new Runnable() {
67 
68                     public void run() {
69                         int tab = tabbedPane.getTabCount() - 1;
70                         Rectangle bounds = tabbedPane.getBoundsAt(tab);
71                         if (bounds != null) {
72                             d.setBounds(bounds);
73                             p.setLocation(d.x + d.width / 2, d.y + d.height / 2);
74                             SwingUtilities.convertPointToScreen(p, tabbedPane);
75                             robot.mouseMove(p.x, p.y + d.height);
76                             tabbedPane.addTab("Hello", null);
77                         }
78                     }
79                 });
80             }
81         } finally {
82             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
83         }
84     }
85 }
