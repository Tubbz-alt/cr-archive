24 import java.awt.*;
25 import java.awt.event.*;
26 import javax.swing.*;
27 import javax.swing.event.*;
28 
29 /*
30  * @test
31  * @key headful
32  * @bug 4361477
33  * @summary JTabbedPane throws ArrayOutOfBoundsException
34  * @author Oleg Mokhovikov
35  * @run main bug4361477
36  */
37 public class bug4361477 {
38 
39     static JTabbedPane tabbedPane;
40     volatile static boolean bStateChanged = false;
41     volatile static Rectangle bounds;
42 
43     public static void main(String args[]) throws Exception {
44 
45         Robot robot = new Robot();
46         robot.setAutoDelay(50);
47 
48         SwingUtilities.invokeAndWait(new Runnable() {
49 
50             @Override
51             public void run() {
52                 createAndShowUI();
53             }
54         });
55 
56         robot.waitForIdle();
57 
58         SwingUtilities.invokeAndWait(new Runnable() {
59 
60             @Override
61             public void run() {
62                 bounds = tabbedPane.getUI().getTabBounds(tabbedPane, 0);
63             }
64         });
65 
66         Point location = bounds.getLocation();
67         SwingUtilities.convertPointToScreen(location, tabbedPane);
68         robot.mouseMove(location.x + 1, location.y + 1);
69         robot.mousePress(InputEvent.BUTTON1_MASK);
70         robot.mouseRelease(InputEvent.BUTTON1_MASK);
71 
72         if (!bStateChanged) {
73             throw new RuntimeException("Tabbed pane state is not changed");
74         }
75     }
76 
77     static void createAndShowUI() {
78 
79         final JFrame frame = new JFrame();
80         tabbedPane = new JTabbedPane();
81         tabbedPane.add("Tab0", new JPanel());
82         tabbedPane.add("Tab1", new JPanel());
83         tabbedPane.add("Tab2", new JPanel());
84         tabbedPane.setSelectedIndex(2);
85         tabbedPane.addChangeListener(new ChangeListener() {
86 
87             public void stateChanged(final ChangeEvent pick) {
88                 bStateChanged = true;
89                 if (tabbedPane.getTabCount() == 3) {
90                     tabbedPane.remove(2);
91                 }
92             }
93         });
94 
95         frame.getContentPane().add(tabbedPane);
96         frame.setSize(300, 200);
97         frame.setVisible(true);
98     }
99 }
