 24 import java.awt.*;
 25 import java.awt.event.*;
 26 import javax.swing.*;
 27 import javax.swing.event.*;
 28 
 29 /*
 30  * @test
 31  * @key headful
 32  * @bug 4361477
 33  * @summary JTabbedPane throws ArrayOutOfBoundsException
 34  * @author Oleg Mokhovikov
 35  * @run main bug4361477
 36  */
 37 public class bug4361477 {
 38 
 39     static JTabbedPane tabbedPane;
 40     volatile static boolean bStateChanged = false;
 41     volatile static Rectangle bounds;
 42     static JFrame frame;
 43 
 44     public static void main(String args[]) throws Exception {
 45         try {
 46 
 47             Robot robot = new Robot();
 48             robot.setAutoDelay(50);
 49 
 50             SwingUtilities.invokeAndWait(new Runnable() {
 51 
 52                 @Override
 53                 public void run() {
 54                     createAndShowUI();
 55                 }
 56             });
 57 
 58             robot.waitForIdle();
 59 
 60             SwingUtilities.invokeAndWait(new Runnable() {
 61 
 62                 @Override
 63                 public void run() {
 64                     bounds = tabbedPane.getUI().getTabBounds(tabbedPane, 0);
 65                 }
 66             });
 67 
 68             Point location = bounds.getLocation();
 69             SwingUtilities.convertPointToScreen(location, tabbedPane);
 70             robot.mouseMove(location.x + 1, location.y + 1);
 71             robot.mousePress(InputEvent.BUTTON1_MASK);
 72             robot.mouseRelease(InputEvent.BUTTON1_MASK);
 73 
 74             if (!bStateChanged) {
 75                 throw new RuntimeException("Tabbed pane state is not changed");
 76             }
 77         } finally {
 78             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
 79         }
 80     }
 81 
 82     static void createAndShowUI() {
 83 
 84         frame = new JFrame();
 85         tabbedPane = new JTabbedPane();
 86         tabbedPane.add("Tab0", new JPanel());
 87         tabbedPane.add("Tab1", new JPanel());
 88         tabbedPane.add("Tab2", new JPanel());
 89         tabbedPane.setSelectedIndex(2);
 90         tabbedPane.addChangeListener(new ChangeListener() {
 91 
 92             public void stateChanged(final ChangeEvent pick) {
 93                 bStateChanged = true;
 94                 if (tabbedPane.getTabCount() == 3) {
 95                     tabbedPane.remove(2);
 96                 }
 97             }
 98         });
 99 
100         frame.getContentPane().add(tabbedPane);
101         frame.setSize(300, 200);
102         frame.setVisible(true);
103     }
104 }
