24 import javax.swing.*;
25 import java.awt.*;
26 /*
27  * @test
28  * @key headful
29  * @bug 6495408
30  * @summary REGRESSION: JTabbedPane throws ArrayIndexOutOfBoundsException
31  * @author Alexander Potochkin
32  * @run main bug6495408
33  */
34 
35 public class bug6495408 {
36 
37     static JTabbedPane tabbedPane;
38 
39     public static void main(String[] args) throws Exception {
40         final Robot robot = new Robot();
41         robot.setAutoDelay(50);
42 
43         SwingUtilities.invokeAndWait(new Runnable() {
44 
45             public void run() {
46                 final JFrame frame = new JFrame();
47                 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
48                 tabbedPane = new JTabbedPane();
49                 tabbedPane.setTabPlacement(JTabbedPane.LEFT);
50                 tabbedPane.addTab("Hello", null);
51                 frame.add(tabbedPane);
52                 frame.setSize(400, 400);
53                 frame.setLocationRelativeTo(null);
54                 frame.setVisible(true);
55             }
56         });
57 
58         robot.waitForIdle();
59 
60         final Rectangle d = new Rectangle();
61         final Point p = new Point();
62 
63         for (int i = 0; i < 7; i++) {
64             SwingUtilities.invokeLater(new Runnable() {
65 
66                 public void run() {
67                     int tab = tabbedPane.getTabCount() - 1;
68                     Rectangle bounds = tabbedPane.getBoundsAt(tab);
69                     if (bounds != null) {
70                         d.setBounds(bounds);
71                         p.setLocation(d.x + d.width / 2, d.y + d.height / 2);
72                         SwingUtilities.convertPointToScreen(p, tabbedPane);
73                         robot.mouseMove(p.x, p.y + d.height);
74                         tabbedPane.addTab("Hello", null);
75                     }
76                 }
77             });
78         }
79     }
80 }
