 35 import javax.swing.*;
 36 import java.awt.event.*;
 37 import java.awt.*;
 38 
 39 public class bug4750590 {
 40 
 41     public static PassedListener pass = new PassedListener();
 42     public static volatile boolean passed = false;
 43     private static JFrame mainFrame;
 44 
 45     public static void main(String args[]) throws Throwable {
 46         try {
 47             SwingUtilities.invokeAndWait(new Runnable() {
 48                 @Override
 49                 public void run() {
 50                     createAndShowGUI();
 51                 }
 52             });
 53 
 54             Robot robo = new Robot();
 55             robo.setAutoDelay(500);
 56             robo.waitForIdle();
 57 
 58             Util.hitMnemonics(robo, KeyEvent.VK_F);
 59             robo.keyPress(KeyEvent.VK_M);
 60             robo.keyRelease(KeyEvent.VK_M);
 61 
 62             robo.waitForIdle();
 63 
 64             if (passed) {
 65                 System.out.println("Test passed!");
 66             } else {
 67                 throw new RuntimeException("Test FAILED!");
 68             }
 69         } finally {
 70             if (mainFrame != null) SwingUtilities.invokeAndWait(() -> mainFrame.dispose());
 71         }
 72     }
 73 
 74     private static void createAndShowGUI() {
 75         mainFrame = new JFrame("Bug 4750590");
 76         JMenuBar mbar = new JMenuBar();
 77         JMenu menu = new JMenu("File");
 78         menu.setMnemonic('F');
 79         JMenu submenu = new JMenu("Submenu");
 80         submenu.add(new JMenuItem("SubMenu Item 1")).setMnemonic('S');
 81         submenu.add(new JMenuItem("SubMenu Item 2"));
 82         menu.add(submenu);
 83 
 84         menu.add(new JMenuItem("Menu Item 1"));
 85         JMenuItem menuItem = menu.add(new JMenuItem("Menu Item 2"));
 86         menuItem.setMnemonic('M');
 87         menuItem.addActionListener(pass);
 88         mbar.add(menu);
 89         mainFrame.setJMenuBar(mbar);
 90 
 91         mainFrame.setSize(200, 200);
 92         mainFrame.setLocation(200, 200);
 93         mainFrame.setVisible(true);
 94         mainFrame.toFront();
 95     }
 96 
 97     public static class PassedListener implements ActionListener {
 98         public void actionPerformed(ActionEvent ev) {
 99             passed = true;
100         }
101     }
102 
103 }
