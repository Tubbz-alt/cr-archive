23 /*
24  * @test
25  * @key headful
26  * @bug 4665214
27  * @summary Makes sure that RepaintManager doesn't attempt to repaint
28  *          a frame when it is iconified.
29  * @author Scott Violet
30  * @run main IconifyTest
31  */
32 import java.awt.*;
33 import java.awt.event.*;
34 import javax.swing.*;
35 
36 public class IconifyTest {
37     private static volatile boolean windowIconifiedIsCalled = false;
38     private static volatile boolean frameIsRepainted = false;
39     static JFrame frame;
40     static JButton button;
41 
42     public static void main(String[] args) throws Throwable {
43         Robot robot = new Robot();
44         SwingUtilities.invokeAndWait(new Runnable() {
45             public void run() {
46                 frame = new JFrame();
47                 button = new JButton("HI");
48                 frame.getContentPane().add(button);
49                 frame.addWindowListener(new WindowAdapter() {
50                     public void windowIconified(WindowEvent e) {
51                         windowIconifiedIsCalled = true;
52                         RepaintManager rm = RepaintManager.currentManager(null);
53                         rm.paintDirtyRegions();
54                         button.repaint();
55                         if (!rm.getDirtyRegion(button).isEmpty()) {
56                             frameIsRepainted = true;
57                         }
58                     }
59                 });
60                 frame.pack();
61                 frame.setVisible(true);
62             }
63         });
64         robot.waitForIdle();
65 
66         SwingUtilities.invokeAndWait(new Runnable() {
67             public void run() {
68                 frame.setExtendedState(Frame.ICONIFIED);
69             }
70         });
71         robot.waitForIdle();
72 
73         if (!windowIconifiedIsCalled) {
74             throw new Exception("Test failed: window was not iconified.");
75         }
76         if (frameIsRepainted) {
77             throw new Exception("Test failed: frame was repainted when window was iconified.");
78         }
79     }
80 }
