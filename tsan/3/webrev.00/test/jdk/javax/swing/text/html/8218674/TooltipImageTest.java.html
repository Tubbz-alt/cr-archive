24 /**
25  * @test
26  * @key headful
27  * @bug 8218674
28  * @summary Tests if Images are rendered and scaled correctly in JToolTip.
29  * @run main TooltipImageTest
30  */
31 import java.awt.Dimension;
32 import java.awt.Insets;
33 import javax.swing.JToolTip;
34 import javax.swing.SwingUtilities;
35 
36 public class TooltipImageTest {
37 
38     private static void checkSize(JToolTip tip, int width, int height) {
39         Dimension d = tip.getPreferredSize();
40         Insets insets = tip.getInsets();
41         //6 seems to be the extra width being allocated for some reason
42         //for a tooltip window.
43         if (!((d.width - insets.right - insets.left - 6) == width) &&
44             !((d.height - insets.top - insets.bottom) == height)) {
45             throw new RuntimeException("Test case fails: Expected width, height is " + width + ", " + height +
46                     " whereas actual width, height are " + (d.width - insets.right - insets.left - 6) + " " +
47                     (d.height - insets.top - insets.bottom));
48         }
49      }
50 
51     public static void main(String[] args) throws Exception {
52         String PATH = TooltipImageTest.class.getResource("circle.png").getPath();
53         SwingUtilities.invokeAndWait(() -> {
54             JToolTip tip = new JToolTip();
55             tip.setTipText("<html><img width=\"100\" src=\"file:///" + PATH + "\"></html>");
56             checkSize(tip, 100, 100);
57 
58             tip.setTipText("<html><img height=\"100\" src=\"file:///" + PATH + "\"></html>");
59             checkSize(tip, 100, 100);
60 
61             tip.setTipText("<html><img src=\"file:///" + PATH + "\"></html>");
62             checkSize(tip, 200, 200);
63 
64             tip.setTipText("<html><img width=\"50\" src=\"file:///" + PATH + "\"></html>");
65             checkSize(tip, 50, 50);
66 
67             tip.setTipText("<html><img height=\"50\" src=\"file:///" + PATH + "\"></html>");
68             checkSize(tip, 50, 50);
69 
70             tip.setTipText("<html><img width=\"100\" height=\"50\" src=\"file:///" + PATH + "\"></html>");
71             checkSize(tip, 100, 50);
72         });
73 
74         System.out.println("Test case passed.");
75     }
76 }
