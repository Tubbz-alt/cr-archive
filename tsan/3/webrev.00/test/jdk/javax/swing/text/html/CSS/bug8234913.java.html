 25 import jdk.test.lib.process.OutputAnalyzer;
 26 import jdk.test.lib.process.ProcessTools;
 27 
 28 import javax.swing.text.MutableAttributeSet;
 29 import javax.swing.text.SimpleAttributeSet;
 30 import javax.swing.text.html.CSS;
 31 import javax.swing.text.html.StyleSheet;
 32 import java.util.ArrayList;
 33 import java.util.Collections;
 34 import java.util.List;
 35 
 36 /*
 37  * @test
 38  * @bug 8234913
 39  * @library /test/lib
 40  * @summary The test project is launched with the option -verbose:class. If the exception is thrown, its class will be loaded.
 41  *          Since the exception is thrown, but then caught, it is not possible to catch it in the test case with try-catch.
 42  *          The class loader logs have to be monitored.
 43  * @run driver bug8234913
 44  */
 45 public class bug8234913 {
 46     // Output analysis taken from test/hotspot/jtreg/runtime/logging/ClassLoadUnloadTest.java
 47     private static ProcessBuilder pb;
 48 
 49     static ProcessBuilder exec(String... args) throws Exception {
 50         List<String> argsList = new ArrayList<>();
 51         Collections.addAll(argsList, args);
 52         Collections.addAll(argsList, "-Xmn8m");
 53         Collections.addAll(argsList, "-Dtest.class.path=" + System.getProperty("test.class.path", "."));
 54         Collections.addAll(argsList, FontSizePercentTest.class.getName());
 55         return ProcessTools.createJavaProcessBuilder(argsList.toArray(new String[argsList.size()]));
 56     }
 57 
 58     static void checkFor(String... outputStrings) throws Exception {
 59         OutputAnalyzer out = new OutputAnalyzer(pb.start());
 60         for (String s : outputStrings) {
 61             out.shouldContain(s);
 62         }
 63         out.shouldHaveExitValue(0);
 64     }
 65 
 66     static void checkAbsent(String... outputStrings) throws Exception {
 67         OutputAnalyzer out = new OutputAnalyzer(pb.start());
 68         for (String s : outputStrings) {
 69             out.shouldNotContain(s);
 70         }
 71         out.shouldHaveExitValue(0);
 72     }
 73 
 74     public static void main(String[] args) throws Exception {
 75         pb = exec("-verbose:class");
 76         checkFor("[class,load] javax.swing.text.html.CSS$LengthUnit"); // the class that parses %
 77         checkAbsent("[class,load] java.lang.NumberFormatException");
 78     }
 79 
 80     static class FontSizePercentTest {
 81         public static void main(String... args) throws Exception {
 82             StyleSheet ss = new StyleSheet();
 83             MutableAttributeSet attr = new SimpleAttributeSet();
 84             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "100%");
 85 
 86             attr.removeAttribute(CSS.Attribute.FONT_SIZE);
 87             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "10pt");
 88 
 89             attr.removeAttribute(CSS.Attribute.FONT_SIZE);
 90             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "10px");
 91 
 92             attr.removeAttribute(CSS.Attribute.FONT_SIZE);
 93             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "10mm");
 94 
 95             attr.removeAttribute(CSS.Attribute.FONT_SIZE);
 96             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "10cm");
 97 
 98             attr.removeAttribute(CSS.Attribute.FONT_SIZE);
 99             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "10pc");
100 
101             attr.removeAttribute(CSS.Attribute.FONT_SIZE);
102             ss.addCSSAttribute(attr, CSS.Attribute.FONT_SIZE, "10in");
103         }
104     }
