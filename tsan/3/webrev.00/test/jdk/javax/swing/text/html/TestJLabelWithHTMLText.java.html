24 /**
25  * @test
26  * @bug 8230235 8235744
27  * @summary Tests if JLabel with HTML text having empty img tag and
28  *      documentBaseKey set renders properly without NPE
29  * @run main TestJLabelWithHTMLText
30  */
31 import java.net.MalformedURLException;
32 import java.net.URL;
33 import javax.swing.JLabel;
34 import javax.swing.SwingUtilities;
35 import javax.swing.plaf.basic.BasicHTML;
36 
37 public class TestJLabelWithHTMLText {
38 
39     public static void main(String[] args) throws Exception {
40         SwingUtilities.invokeAndWait(() -> {
41             JLabel label = new JLabel();
42             try {
43                 label.putClientProperty(BasicHTML.documentBaseKey,
44                         new URL("http://localhost"));
45             } catch (MalformedURLException e) {
46                 e.printStackTrace();
47             }
48             label.setText("<html><img src=''></html>");
49         });
50     }
51 }
