 23 /*
 24    @test
 25    @key headful
 26    @bug 4973721
 27    @summary Up and Down Arrow key buttons are not working for the JSpinner in Synth LAF
 28    @library ../../regtesthelpers
 29    @build Util
 30    @author Oleg Mokhovikov
 31    @run main bug4973721
 32  */
 33 
 34 import java.awt.Robot;
 35 import javax.swing.event.ChangeListener;
 36 import javax.swing.event.ChangeEvent;
 37 import java.awt.event.KeyEvent;
 38 import java.awt.event.FocusListener;
 39 import java.awt.event.FocusEvent;
 40 import javax.swing.*;
 41 
 42 public class bug4973721 implements ChangeListener, FocusListener {
 43     static volatile boolean bStateChanged = false;
 44     static volatile boolean bFocusGained = false;
 45     static JSpinner spinner;
 46     static final Object listener = new bug4973721();
 47 
 48     public void focusLost(FocusEvent e) {}
 49 
 50     public synchronized void focusGained(FocusEvent e) {
 51         System.out.println("focusGained");
 52         bFocusGained = true;
 53         notifyAll();
 54     }
 55 
 56     public synchronized void stateChanged(ChangeEvent e) {
 57         System.out.println("stateChanged");
 58         bStateChanged = true;
 59         notifyAll();
 60     }
 61 
 62     public static void main(String[] args) throws Exception {
 63         UIManager.setLookAndFeel("javax.swing.plaf.synth.SynthLookAndFeel");
 64 
 65         SwingUtilities.invokeAndWait(new Runnable() {
 66             public void run() {
 67                 final JFrame frame = new JFrame();
 68                 spinner = new JSpinner();
 69                 frame.getContentPane().add(spinner);
 70                 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 71 
 72                 frame.pack();
 73                 frame.setVisible(true);
 74                 spinner.addChangeListener((ChangeListener)listener);
 75                 spinner.addFocusListener((FocusListener)listener);
 76                 spinner.requestFocus();
 77 
 78             }
 79         });
 80 
 81         synchronized(listener) {
 82             if (!bFocusGained) {
 83                 System.out.println("waiting focusGained...");
 84                 try { listener.wait(5000); } catch (InterruptedException e) {}
 85             }
 86         }
 87 
 88         boolean hasFocus = Util.invokeOnEDT(new java.util.concurrent.Callable<Boolean>() {
 89             @Override
 90             public Boolean call() throws Exception {
 91                 return spinner.hasFocus();
 92             }
 93         });
 94 
 95         if (!bFocusGained && !hasFocus) {
 96             throw new RuntimeException("Couldn't request focus for spinner");
 97         }
 98         Robot robot = new Robot();
 99         robot.setAutoDelay(50);
100 
101         Util.hitKeys(robot, KeyEvent.VK_UP);
102         robot.waitForIdle();
103         Thread.sleep(1000);
104 
105         if (!bStateChanged) {
106             throw new RuntimeException("Up arrow key button doesn't work for a spinner in Synth L&F");
107         }
108 
109         bStateChanged = false;
110 
111         Util.hitKeys(robot, KeyEvent.VK_DOWN);
112         robot.waitForIdle();
113         Thread.sleep(1000);
114 
115         if (!bStateChanged) {
116             throw new RuntimeException("Down arrow key button doesn't work for a spinner in Synth L&F");
117         }
118     }
119 }
