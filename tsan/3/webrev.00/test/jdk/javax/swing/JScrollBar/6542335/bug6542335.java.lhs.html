 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 6542335
 28  * @summary different behavior on knob of scroll bar between 1.4.2 and 5.0
 29  * @author  Alexander Potochkin
 30  * @run main bug6542335
 31  */
 32 
 33 import javax.swing.*;
 34 import javax.swing.plaf.basic.BasicScrollBarUI;
 35 import java.awt.*;
 36 import java.awt.event.InputEvent;
 37 
 38 public class bug6542335 {
 39     private static JScrollBar sb;
 40     private static MyScrollBarUI ui;

 41 
 42     public static void main(String[] args) throws Exception {
 43         final Robot robot = new Robot();
 44         robot.setAutoDelay(10);

 45 
 46         final Rectangle[] thumbBounds = new Rectangle[1];
 47 
 48         SwingUtilities.invokeAndWait(new Runnable() {
 49             public void run() {
 50                 final JFrame frame = new JFrame("bug6542335");
 51                 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 52 
 53                 sb = new JScrollBar(0, 0, 1, 0, 1);
 54 
 55                 ui = new MyScrollBarUI();
 56                 sb.setUI(ui);
 57 
 58                 sb.setPreferredSize(new Dimension(200, 17));
 59                 DefaultBoundedRangeModel rangeModel = new DefaultBoundedRangeModel();
 60                 rangeModel.setMaximum(100);
 61                 rangeModel.setMinimum(0);
 62                 rangeModel.setExtent(50);
 63                 rangeModel.setValue(50);
 64 
 65                 sb.setModel(rangeModel);
 66                 frame.add(sb, BorderLayout.NORTH);
 67 
 68                 frame.setSize(200, 100);
 69                 frame.setVisible(true);
 70             }
 71         });
 72 
 73         robot.waitForIdle();
 74 
 75         SwingUtilities.invokeAndWait(new Runnable() {
 76             public void run() {
 77                 thumbBounds[0] = new Rectangle(ui.getThumbBounds());
 78 
 79                 Point l = sb.getLocationOnScreen();
 80 
 81                 robot.mouseMove(l.x + (int) (0.75 * sb.getWidth()), l.y + sb.getHeight() / 2);
 82                 robot.mousePress(InputEvent.BUTTON1_MASK);
 83                 robot.mouseRelease(InputEvent.BUTTON1_MASK);
 84             }
 85         });
 86 
 87         robot.waitForIdle();
 88 
 89         SwingUtilities.invokeAndWait(new Runnable() {
 90             public void run() {
 91                 Rectangle newThumbBounds = ui.getThumbBounds();
 92 
 93                 if (!thumbBounds[0].equals(newThumbBounds)) {
 94                     throw new RuntimeException("Test failed.\nOld bounds: " + thumbBounds[0] +
 95                     "\nNew bounds: " + newThumbBounds);

 96                 }
 97             }
 98         });


 99     }
100 
101     static class MyScrollBarUI extends BasicScrollBarUI {
102         public Rectangle getThumbBounds() {
103             return super.getThumbBounds();
104         }
105     }
106 }
