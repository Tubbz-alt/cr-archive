 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 6542335
 28  * @summary different behavior on knob of scroll bar between 1.4.2 and 5.0
 29  * @author  Alexander Potochkin
 30  * @run main bug6542335
 31  */
 32 
 33 import javax.swing.*;
 34 import javax.swing.plaf.basic.BasicScrollBarUI;
 35 import java.awt.*;
 36 import java.awt.event.InputEvent;
 37 
 38 public class bug6542335 {
 39     private static JScrollBar sb;
 40     private static MyScrollBarUI ui;
 41     private static JFrame frame;
 42 
 43     public static void main(String[] args) throws Exception {
 44         try {
 45             final Robot robot = new Robot();
 46             robot.setAutoDelay(10);
 47 
 48             final Rectangle[] thumbBounds = new Rectangle[1];
 49 
 50             SwingUtilities.invokeAndWait(new Runnable() {
 51                 public void run() {
 52                     frame = new JFrame("bug6542335");
 53                     frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 54 
 55                     sb = new JScrollBar(0, 0, 1, 0, 1);
 56 
 57                     ui = new MyScrollBarUI();
 58                     sb.setUI(ui);
 59 
 60                     sb.setPreferredSize(new Dimension(200, 17));
 61                     DefaultBoundedRangeModel rangeModel = new DefaultBoundedRangeModel();
 62                     rangeModel.setMaximum(100);
 63                     rangeModel.setMinimum(0);
 64                     rangeModel.setExtent(50);
 65                     rangeModel.setValue(50);
 66 
 67                     sb.setModel(rangeModel);
 68                     frame.add(sb, BorderLayout.NORTH);
 69 
 70                     frame.setSize(200, 100);
 71                     frame.setVisible(true);
 72                 }
 73             });
 74 
 75             robot.waitForIdle();
 76 
 77             SwingUtilities.invokeAndWait(new Runnable() {
 78                 public void run() {
 79                     thumbBounds[0] = new Rectangle(ui.getThumbBounds());
 80 
 81                     Point l = sb.getLocationOnScreen();
 82 
 83                     robot.mouseMove(l.x + (int) (0.75 * sb.getWidth()), l.y + sb.getHeight() / 2);
 84                     robot.mousePress(InputEvent.BUTTON1_MASK);
 85                     robot.mouseRelease(InputEvent.BUTTON1_MASK);
 86                 }
 87             });
 88 
 89             robot.waitForIdle();
 90 
 91             SwingUtilities.invokeAndWait(new Runnable() {
 92                 public void run() {
 93                     Rectangle newThumbBounds = ui.getThumbBounds();
 94 
 95                     if (!thumbBounds[0].equals(newThumbBounds)) {
 96                         throw new RuntimeException("Test failed.\nOld bounds: " + thumbBounds[0] +
 97                         "\nNew bounds: " + newThumbBounds);
 98                     }
 99                 }
100             });
101         } finally {
102                 if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
103         }
104     }
105 
106     static class MyScrollBarUI extends BasicScrollBarUI {
107         public Rectangle getThumbBounds() {
108             return super.getThumbBounds();
109         }
110     }
111 }
