24 /*
25  * @test
26  * @key headful
27  * @bug 6987844
28  * @summary Incorrect width of JComboBox drop down
29  * @author Alexander Potochkin
30  * @run main bug6987844
31  */
32 
33 import javax.swing.*;
34 import java.awt.*;
35 import java.awt.event.InputEvent;
36 
37 public class bug6987844 {
38     static JMenu menu1;
39     static JMenu menu2;
40     static JFrame frame;
41 
42     public static void main(String... args) throws Exception {
43         try {
44             Robot robot = new Robot();
45             robot.setAutoDelay(200);
46 
47             SwingUtilities.invokeAndWait(new Runnable() {
48                 public void run() {
49                     frame = new JFrame();
50                     frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
51 
52                     JMenuBar bar = new JMenuBar();
53                     menu1 = new JMenu("Menu1");
54                     menu1.add(new JMenuItem("item"));
55                     bar.add(menu1);
56                     menu2 = new JMenu("Menu2");
57                     menu2.add(new JMenuItem("item"));
58                     menu2.add(new JMenuItem("item"));
59                     bar.add(menu2);
60 
61                     frame.setJMenuBar(bar);
62                     frame.pack();
63 
64                     frame.setVisible(true);
65                 }
66             });
67             robot.waitForIdle();
68             Point point1 = menu1.getLocationOnScreen();
69             Point point2 = menu2.getLocationOnScreen();
70 
71             robot.mouseMove(point1.x + 1, point1.y + 1);
72             robot.mousePress(InputEvent.BUTTON1_MASK);
73             robot.mouseRelease(InputEvent.BUTTON1_MASK);
74 
75             robot.mouseMove(point2.x + 1, point2.y + 1);
76             robot.mousePress(InputEvent.BUTTON1_MASK);
77             robot.mouseRelease(InputEvent.BUTTON1_MASK);
78             robot.mousePress(InputEvent.BUTTON1_MASK);
79             robot.mouseRelease(InputEvent.BUTTON1_MASK);
80 
81             robot.mouseMove(point1.x + 1, point1.y + 1);
82             robot.waitForIdle();
83 
84             SwingUtilities.invokeAndWait(new Runnable() {
85                 public void run() {
86                     Dimension popupSize1 = menu1.getPopupMenu().getSize();
87                     Dimension popupSize2 = menu2.getPopupMenu().getSize();
88                     if (popupSize1.equals(popupSize2)) {
89                         throw new RuntimeException("First popup unexpedetly changed its size");
90                     }
91                 }
92             });
93         } finally {
94             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
95         }
96     }
97 }
