25  * @test
26  * @key headful
27  * @bug 6987844
28  * @summary Incorrect width of JComboBox drop down
29  * @author Alexander Potochkin
30  * @run main bug6987844
31  */
32 
33 import javax.swing.*;
34 import java.awt.*;
35 import java.awt.event.InputEvent;
36 
37 public class bug6987844 {
38     static JMenu menu1;
39     static JMenu menu2;
40 
41     public static void main(String... args) throws Exception {
42         Robot robot = new Robot();
43         robot.setAutoDelay(200);
44 
45         SwingUtilities.invokeAndWait(new Runnable() {
46             public void run() {
47                 JFrame frame = new JFrame();
48                 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
49 
50                 JMenuBar bar = new JMenuBar();
51                 menu1 = new JMenu("Menu1");
52                 menu1.add(new JMenuItem("item"));
53                 bar.add(menu1);
54                 menu2 = new JMenu("Menu2");
55                 menu2.add(new JMenuItem("item"));
56                 menu2.add(new JMenuItem("item"));
57                 bar.add(menu2);
58 
59                 frame.setJMenuBar(bar);
60                 frame.pack();
61 
62                 frame.setVisible(true);
63             }
64         });
65         robot.waitForIdle();
66         Point point1 = menu1.getLocationOnScreen();
67         Point point2 = menu2.getLocationOnScreen();
68 
69         robot.mouseMove(point1.x + 1, point1.y + 1);
70         robot.mousePress(InputEvent.BUTTON1_MASK);
71         robot.mouseRelease(InputEvent.BUTTON1_MASK);
72 
73         robot.mouseMove(point2.x + 1, point2.y + 1);
74         robot.mousePress(InputEvent.BUTTON1_MASK);
75         robot.mouseRelease(InputEvent.BUTTON1_MASK);
76         robot.mousePress(InputEvent.BUTTON1_MASK);
77         robot.mouseRelease(InputEvent.BUTTON1_MASK);
78 
79         robot.mouseMove(point1.x + 1, point1.y + 1);
80         robot.waitForIdle();
81 
82         SwingUtilities.invokeAndWait(new Runnable() {
83             public void run() {
84                 Dimension popupSize1 = menu1.getPopupMenu().getSize();
85                 Dimension popupSize2 = menu2.getPopupMenu().getSize();
86                 if (popupSize1.equals(popupSize2)) {
87                     throw new RuntimeException("First popup unexpedetly changed its size");
88                 }
89             }
90         });
91     }
92 }
