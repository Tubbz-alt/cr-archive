 38 import jdk.test.lib.Platform;
 39 import java.awt.*;
 40 import java.awt.event.KeyEvent;
 41 import javax.swing.*;
 42 
 43 public class bug6827786 {
 44 
 45     private static JMenu menu;
 46     private static Component focusable;
 47     private static JFrame frame;
 48 
 49     public static void main(String[] args) throws Exception {
 50         try {
 51             Robot robot = new Robot();
 52             robot.setAutoDelay(50);
 53             // move mouse outside menu to prevent auto selection
 54            robot.mouseMove(100,100);
 55 
 56             SwingUtilities.invokeAndWait(new Runnable() {
 57 
 58                 public void run() {
 59                     createAndShowGUI();
 60                 }
 61             });
 62 
 63             robot.waitForIdle();
 64 
 65             SwingUtilities.invokeAndWait(new Runnable() {
 66 
 67                 public void run() {
 68                     focusable.requestFocus();
 69                 }
 70             });
 71 
 72             robot.waitForIdle();
 73             checkfocus();
 74 
 75             // select menu
 76             if (Platform.isOSX()) {
 77                 Util.hitKeys(robot, KeyEvent.VK_CONTROL, KeyEvent.VK_ALT, KeyEvent.VK_F);
 78             } else {
 79                 Util.hitKeys(robot, KeyEvent.VK_ALT, KeyEvent.VK_F);
 80             }
 81             // select submenu
 82             Util.hitKeys(robot, KeyEvent.VK_S);
 83             robot.waitForIdle();
 84             // verify submenu is selected
 85             verify(1);
 86 
 87             Util.hitKeys(robot, KeyEvent.VK_S);
 88             robot.waitForIdle();
 89             // verify last item is selected
 90             verify(2);
 91 
 92             Util.hitKeys(robot, KeyEvent.VK_S);
 93             robot.waitForIdle();
 94             // selection should wrap to first item
 95             verify(0);
 96 
 97             System.out.println("PASSED");
 98         } finally {
 99             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
100         }
101     }
102 
103     private static void checkfocus() throws Exception {
104         SwingUtilities.invokeAndWait(new Runnable() {
105 
106             public void run() {
107                 if (!focusable.isFocusOwner()) {
108                     throw new RuntimeException("Button is not the focus owner.");
109                 }
110             }
111         });
112     }
113 
114     private static void verify(final int index) throws Exception {
115         SwingUtilities.invokeAndWait(new Runnable() {
116 
117             @Override
118             public void run() {
119                 MenuElement[] path =
120                         MenuSelectionManager.defaultManager().getSelectedPath();
121                 MenuElement item = path[3];
122                 if (item != menu.getMenuComponent(index)) {
123                     System.err.println("Selected: " + item);
124                     System.err.println("Should be: "
125                             + menu.getMenuComponent(index));
126                     throw new RuntimeException("Test Failed");
127                 }
128             }
129         });
130     }
131 
132     private static JMenuBar createMenuBar() {
133         menu = new JMenu("File");
134         menu.setMnemonic('F');
135 
136         menu.add(new JMenuItem("Save", 'S'));
137 
138         JMenu sub = new JMenu("Submenu");
139         sub.setMnemonic('S');
140         sub.add(new JMenuItem("Sub Item"));
141         menu.add(sub);
142 
143         menu.add(new JMenuItem("Special", 'S'));
144 
145         JMenuBar bar = new JMenuBar();
146         bar.add(menu);
147         return bar;
148     }
149 
150     private static void createAndShowGUI() {
151         frame = new JFrame("bug6827786");
152         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
153         frame.setJMenuBar(createMenuBar());
154         focusable = new JButton("Set Focus Here");
155         frame.add(focusable);
156         frame.pack();
157         frame.setLocationRelativeTo(null);
158         frame.setVisible(true);
159     }
160 }
