26  * @key headful
27  * @summary Checks that JLayer inside JViewport works is correctly laid out
28  * @author Alexander Potochkin

30  * @build ExtendedRobot
31  * @run main bug6824395
32  */
33 
34 
35 
36 import javax.swing.*;
37 import javax.swing.plaf.LayerUI;
38 import java.awt.*;
39 
40 public class bug6824395 {
41 
42     static JScrollPane scrollPane;
43 
44     public static void main(String[] args) throws Exception {
45         SwingUtilities.invokeAndWait(new Runnable() {
46             public void run() {
47                 JFrame frame = new JFrame("testing");
48                 frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
49 
50                 JEditorPane editorPane = new JEditorPane();
51                 String str = "hello\n";
52                 for(int i = 0; i<5; i++) {
53                     str += str;
54                 }
55 
56                 editorPane.setText(str);
57 
58                 JLayer<JEditorPane> editorPaneLayer = new JLayer<JEditorPane>(editorPane);
59                 LayerUI<JComponent> layerUI = new LayerUI<JComponent>();
60                 editorPaneLayer.setUI(layerUI);
61 
62                 scrollPane = new JScrollPane(editorPaneLayer);
63                 scrollPane.setViewportBorder(null);
64                 scrollPane.setPreferredSize(new Dimension(200, 250));
65                 frame.add(scrollPane);
66 
67                 frame.setSize(200, 200);
68                 frame.pack();
69                 frame.setVisible(true);
70             }
71         });
72         try {
73              ExtendedRobot robot = new ExtendedRobot();
74              robot.waitForIdle(300);
75          }catch(Exception ex) {
76              ex.printStackTrace();
77              throw new Error("Unexpected Failure");
78          }
79         SwingUtilities.invokeAndWait(new Runnable() {
80             public void run() {
81                 if (scrollPane.getViewportBorderBounds().width != scrollPane.getViewport().getView().getWidth()) {
82                     throw new RuntimeException("Wrong component's width!");
83                 }
84             }
85         });
86     }
87 }
