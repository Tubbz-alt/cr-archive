25  * @test
26  * @key headful
27  * @summary Checks that JLayer inside JViewport works is correctly laid out
28  * @author Alexander Potochkin
29  * @library /lib/client/
30  * @build ExtendedRobot
31  * @run main bug6824395
32  */
33 
34 
35 
36 import javax.swing.*;
37 import javax.swing.plaf.LayerUI;
38 import java.awt.*;
39 
40 public class bug6824395 {
41 
42     static JScrollPane scrollPane;
43     static JFrame frame;
44 
45     public static void main(String[] args) throws Exception {
46         try {
47             SwingUtilities.invokeAndWait(new Runnable() {
48                 public void run() {
49                     frame = new JFrame("testing");
50                     frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
51 
52                     JEditorPane editorPane = new JEditorPane();
53                     String str = "hello\n";
54                     for(int i = 0; i<5; i++) {
55                         str += str;
56                     }
57 
58                     editorPane.setText(str);
59 
60                     JLayer<JEditorPane> editorPaneLayer = new JLayer<JEditorPane>(editorPane);
61                     LayerUI<JComponent> layerUI = new LayerUI<JComponent>();
62                     editorPaneLayer.setUI(layerUI);
63 
64                     scrollPane = new JScrollPane(editorPaneLayer);
65                     scrollPane.setViewportBorder(null);
66                     scrollPane.setPreferredSize(new Dimension(200, 250));
67                     frame.add(scrollPane);
68 
69                     frame.setSize(200, 200);
70                     frame.pack();
71                     frame.setVisible(true);
72                 }
73             });
74             try {
75                  ExtendedRobot robot = new ExtendedRobot();
76                  robot.waitForIdle(300);
77              }catch(Exception ex) {
78                  ex.printStackTrace();
79                  throw new Error("Unexpected Failure");
80              }
81             SwingUtilities.invokeAndWait(new Runnable() {
82                 public void run() {
83                     if (scrollPane.getViewportBorderBounds().width != scrollPane.getViewport().getView().getWidth()) {
84                         throw new RuntimeException("Wrong component's width!");
85                     }
86                 }
87             });
88         } finally {
89             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
90         }
91     }
92 }
