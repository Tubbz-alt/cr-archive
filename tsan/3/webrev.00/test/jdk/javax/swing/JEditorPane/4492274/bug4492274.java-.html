 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 4492274
 28  * @summary  Tests if JEditorPane.getPage() correctly returns anchor reference.
 29  * @author Denis Sharypov
 30  * @run main bug4492274
 31  */
 32 
 33 import javax.swing.*;
 34 import javax.swing.text.html.HTMLEditorKit;
 35 import java.awt.*;
 36 import java.io.File;
 37 import java.net.URL;
 38 
 39 public class bug4492274 {
 40 
 41     private static URL page;
 42 
 43     private static JEditorPane jep;
 44 
 45     public static void main(String args[]) throws Exception {
 46 
 47         Robot robot = new Robot();
 48         SwingUtilities.invokeAndWait(new Runnable() {
 49             @Override
 50             public void run() {
 51                 createAndShowGUI();
 52             }
 53         });
 54 
 55         robot.waitForIdle();
 56 
 57         SwingUtilities.invokeAndWait(new Runnable() {
 58             @Override
 59             public void run() {
 60                 try {
 61                     page = new URL(page, "#linkname");
 62                     jep.setPage(page);
 63                 } catch (Exception e) {
 64                     throw new RuntimeException(e);
 65                 }
 66             }
 67         });
 68 
 69         robot.waitForIdle();
 70 
 71         if (getPageAnchor() == null) {
 72             throw new RuntimeException("JEditorPane.getPage() returns null anchor reference");
 73         }
 74 
 75     }
 76 
 77     private static String getPageAnchor() throws Exception {
 78         final String[] result = new String[1];
 79 
 80         SwingUtilities.invokeAndWait(new Runnable() {
 81             @Override
 82             public void run() {
 83                 result[0] = jep.getPage().getRef();
 84             }
 85         });
 86 
 87         return result[0];
 88     }
 89 
 90     private static void createAndShowGUI() {
 91         try {
 92             File file = new File(System.getProperty("test.src", "."), "test.html");
 93             page = file.toURI().toURL();
 94 
 95             JFrame f = new JFrame();
 96 
 97             jep = new JEditorPane();
 98             jep.setEditorKit(new HTMLEditorKit());
 99             jep.setEditable(false);
100             jep.setPage(page);
101 
102             JScrollPane sp = new JScrollPane(jep);
103 
104             f.getContentPane().add(sp);
105             f.setSize(500, 500);
106             f.setVisible(true);
107 
108         } catch (Exception e) {
109             throw new RuntimeException(e);
110         }
111     }
112 }
