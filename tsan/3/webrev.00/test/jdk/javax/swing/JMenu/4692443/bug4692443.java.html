 24 /*
 25  * @test
 26  * @key headful
 27  * @library ../../regtesthelpers
 28  * @build Util
 29  * @bug 4692443 7105030
 30  * @summary JMenu: MenuListener.menuSelected() event fired too late when using mnemonics
 31  * @author Alexander Zuev
 32  * @run main bug4692443
 33  */
 34 
 35 import javax.swing.*;
 36 import javax.swing.event.*;
 37 import java.awt.event.*;
 38 import java.awt.*;
 39 
 40 public class bug4692443 {
 41 
 42     public static PassedListener pass;
 43     public static FailedListener fail;
 44     public static volatile Boolean passed;
 45     public static JFrame mainFrame;
 46 
 47     public static void main(String args[]) throws Throwable {
 48         try {
 49             fail = new FailedListener();
 50             pass = new PassedListener();
 51             passed = false;
 52             Robot robo = new Robot();
 53 
 54             SwingUtilities.invokeAndWait(new Runnable() {
 55                 public void run() {
 56                     createAndShowGUI();
 57                 }
 58             });
 59 
 60             robo.waitForIdle();
 61 
 62             int altKey = java.awt.event.KeyEvent.VK_ALT;
 63             robo.setAutoDelay(100);
 64             Util.hitMnemonics(robo, KeyEvent.VK_F); // Enter File menu
 65             robo.keyPress(KeyEvent.VK_S);  // Enter submenu
 66             robo.keyRelease(KeyEvent.VK_S);
 67             robo.keyPress(KeyEvent.VK_O); // Launch "One" action
 68             robo.keyRelease(KeyEvent.VK_O);
 69             robo.keyPress(KeyEvent.VK_M); // Launch "One" action
 70             robo.keyRelease(KeyEvent.VK_M);
 71 
 72             robo.waitForIdle();
 73 
 74             if (!passed) {
 75                 throw new RuntimeException("Test failed.");
 76             }
 77         } finally {
 78              if (mainFrame != null) SwingUtilities.invokeAndWait(() -> mainFrame.dispose());
 79         }
 80     }
 81 
 82     private static void createAndShowGUI() {
 83         mainFrame = new JFrame("Bug 4692443");
 84         JMenuBar mbar = new JMenuBar();
 85         JMenu menu = new JMenu("File");
 86         menu.setMnemonic('F');
 87         menu.add(new JMenuItem("Menu Item 1")).setMnemonic('I');
 88         final JMenu submenu = new JMenu("Submenu");
 89         submenu.setMnemonic('S');
 90         submenu.addMenuListener(new MenuListener() {
 91             public void menuSelected(MenuEvent e) {
 92                 JMenuItem item = submenu.add(new JMenuItem("One", 'O'));
 93                 item.addActionListener(pass);
 94                 submenu.add(new JMenuItem("Two", 'w'));
 95                 submenu.add(new JMenuItem("Three", 'r'));
 96             }
 97             public void menuDeselected(MenuEvent e) {
 98                 submenu.removeAll();
 99             }
100             public void menuCanceled(MenuEvent e) {
101                 submenu.removeAll();
102             }
103         });
104         menu.add(submenu);
105         JMenuItem menuItem = menu.add(new JMenuItem("Menu Item 2"));
106         menuItem.setMnemonic('M');
107         menuItem.addActionListener(fail);
108         mbar.add(menu);
109         mainFrame.setJMenuBar(mbar);
110 
111         mainFrame.setSize(200, 200);
112         mainFrame.setLocation(200, 200);
113         mainFrame.setVisible(true);
114         mainFrame.toFront();
115     }
116 
117     public static class FailedListener implements ActionListener {
118         public void actionPerformed(ActionEvent ev) {
119             throw new RuntimeException("Test failed.");
120         }
121     }
122 
123     public static class PassedListener implements ActionListener {
124         public void actionPerformed(ActionEvent ev) {
125             passed = true;
126         }
127     }
128 
129 }
