 24 /**
 25  * @test
 26  * @key headful
 27  * @bug 6276087
 28  * @author Romain Guy
 29  * @summary Tests opacity of a popup menu.
 30  */
 31 import java.awt.*;
 32 import java.awt.event.*;
 33 
 34 import javax.swing.*;



 35 
 36 public class NonOpaquePopupMenuTest extends JFrame {
 37 
 38     private static JMenu fileMenu;

 39 
 40     public NonOpaquePopupMenuTest() {
 41         getContentPane().setBackground(java.awt.Color.RED);
 42         JMenuBar menuBar = new JMenuBar();
 43         fileMenu = new JMenu("File");
 44         JMenuItem menuItem = new JMenuItem("New");
 45         menuBar.add(fileMenu);
 46         setJMenuBar(menuBar);
 47 
 48         fileMenu.add(menuItem);
 49         fileMenu.getPopupMenu().setOpaque(false);
 50 
 51         setSize(new Dimension(640, 480));
 52         setVisible(true);
 53     }
 54 
 55     public static void main(String[] args) throws Throwable {
 56         Robot robot = new Robot();
 57         robot.setAutoDelay(250);










 58 
 59         SwingUtilities.invokeAndWait(new Runnable() {
 60 
 61             @Override
 62             public void run() {
 63                 new NonOpaquePopupMenuTest();
 64             }
 65         });
 66 
 67         robot.waitForIdle();
 68 
 69         Point p = getMenuClickPoint();
 70         robot.mouseMove(p.x, p.y);
 71         robot.mousePress(InputEvent.BUTTON1_MASK);
 72         robot.mouseRelease(InputEvent.BUTTON1_MASK);
 73 
 74         robot.waitForIdle();
 75 
 76         if (isParentOpaque()) {
 77             throw new RuntimeException("Popup menu parent is opaque");

 78         }
 79 
 80     }
 81 
 82     private static boolean isParentOpaque() throws Exception {
 83         final boolean result[] = new boolean[1];
 84 
 85         SwingUtilities.invokeAndWait(new Runnable() {
 86 
 87             @Override
 88             public void run() {
 89                 result[0] = fileMenu.getPopupMenu().getParent().isOpaque();
 90             }
 91         });
 92 
 93         return result[0];
 94     }
 95 
 96     private static Point getMenuClickPoint() throws Exception {
 97         final Point[] result = new Point[1];
 98 
 99         SwingUtilities.invokeAndWait(new Runnable() {
100 
101             @Override
102             public void run() {
103                 Point p = fileMenu.getLocationOnScreen();
104                 Dimension size = fileMenu.getSize();
105 
106                 result[0] = new Point(p.x + size.width / 2,
107                         p.y + size.height / 2);
108             }
109         });
110 
111         return result[0];
112 
113     }
114 }
