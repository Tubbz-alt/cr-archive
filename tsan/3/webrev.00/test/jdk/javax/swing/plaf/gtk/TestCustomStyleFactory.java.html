24 /**
25  * @test
26  * @key headful
27  * @bug 8222759
28  * @requires (os.family == "linux")
29  * @summary Tests if a custom style factory can be used in GTKL&F after
30  *  removing unnecessary casts
31  * @run main TestCustomStyleFactory
32  */
33 
34 import javax.swing.JComponent;
35 import javax.swing.JLabel;
36 import javax.swing.UIManager;
37 import javax.swing.UnsupportedLookAndFeelException;
38 import javax.swing.plaf.synth.Region;
39 import javax.swing.plaf.synth.SynthLookAndFeel;
40 import javax.swing.plaf.synth.SynthStyleFactory;
41 import javax.swing.plaf.synth.SynthStyle;
42 
43 public class TestCustomStyleFactory {
44     private static final String GTK_LAF_CLASS = "GTKLookAndFeel";
45 
46     public static void main(String args[]) throws Exception {
47         if (!System.getProperty("os.name").startsWith("Linux")) {
48             System.out.println("This test is meant for Linux platform only");
49             return;
50         }
51 
52         for (UIManager.LookAndFeelInfo lookAndFeelInfo :
53                 UIManager.getInstalledLookAndFeels()) {
54             if (lookAndFeelInfo.getClassName().contains(GTK_LAF_CLASS)) {
55                 try {
56                     UIManager.setLookAndFeel(lookAndFeelInfo.getClassName());
57                 } catch (final UnsupportedLookAndFeelException ignored) {
58                     System.out.println("GTK L&F could not be set, so this " +
59                             "test can not be run in this scenario ");
60                     return;
61                 }
62             }
63         }
64         final SynthStyleFactory original = SynthLookAndFeel.getStyleFactory();
65         SynthLookAndFeel.setStyleFactory(new SynthStyleFactory() {
66             @Override
67             public SynthStyle getStyle(JComponent c, Region id) {
68                 return original.getStyle(c, id);
69             }
70         });
71         new JLabel("test");
72     }
73 }
