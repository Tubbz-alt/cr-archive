 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 6711682
 28  * @summary  JCheckBox in JTable: checkbox doesn't alaways respond to the first mouse click
 29  * @author Alexander Potochkin
 30  * @run main bug6711682
 31  */
 32 
 33 import javax.swing.*;
 34 import javax.swing.event.CellEditorListener;
 35 import javax.swing.table.TableCellEditor;
 36 import javax.swing.table.TableCellRenderer;
 37 import java.awt.*;
 38 import java.awt.event.InputEvent;
 39 import java.awt.event.KeyEvent;
 40 import java.util.EventObject;
 41 
 42 public class bug6711682 {
 43     private static JCheckBox editorCb;
 44     private static JCheckBox rendererCb;
 45     private static JTable table;
 46     private static JFrame f;
 47 
 48     public static void main(String[] args) throws Exception {
 49         try {
 50             Robot robot = new Robot();
 51             robot.setAutoDelay(50);
 52             SwingUtilities.invokeAndWait(new Runnable() {
 53                 public void run() {
 54                     createAndShowGUI();
 55                 }
 56             });
 57             robot.waitForIdle();
 58             Point l = table.getLocationOnScreen();
 59             int h = table.getRowHeight();
 60             for (int i = 0; i < 3; i++) {
 61                 robot.mouseMove(l.x + 5, l.y + 5 + i * h);
 62                 robot.mousePress(InputEvent.BUTTON1_MASK);
 63                 robot.mouseRelease(InputEvent.BUTTON1_MASK);
 64             }
 65             // Without pressing F2 the last table's cell
 66             // reported <code>false</code> value
 67             // note that I can't press it inside the for loop
 68             // because it doesn't reproduce the bug
 69             robot.keyPress(KeyEvent.VK_F2);
 70             robot.keyRelease(KeyEvent.VK_F2);









 71 
 72             for (int i = 0; i < 3; i++) {
 73                 if (!Boolean.TRUE.equals(table.getValueAt(i, 0))) {
 74                     throw new RuntimeException("Row #" + i + " checkbox is not selected");
 75                 }
 76             }
 77             for (int i = 2; i >= 0; i--) {
 78                 robot.mouseMove(l.x + 5, l.y + 5 + i * h);
 79                 robot.mousePress(InputEvent.BUTTON1_MASK);
 80                 robot.mouseRelease(InputEvent.BUTTON1_MASK);
 81             }
 82             robot.keyPress(KeyEvent.VK_F2);
 83             robot.keyRelease(KeyEvent.VK_F2);
 84             for (int i = 0; i < 3; i++) {
 85                 if (Boolean.TRUE.equals(table.getValueAt(i, 0))) {
 86                     throw new RuntimeException("Row #" + i + " checkbox is selected");
 87                 }
 88             }
 89         } finally {
 90             if (f != null) SwingUtilities.invokeAndWait(() -> f.dispose());
 91         }
 92     }
 93 
 94     private static void createAndShowGUI() {
 95         editorCb = new JCheckBox();
 96         rendererCb = new JCheckBox();
 97         f = new JFrame("Table with CheckBox");
 98         Container p = f.getContentPane();
 99         p.setLayout(new BorderLayout());
100         table = new JTable(new Object[][]{{false}, {false}, {false}}, new Object[]{"CheckBox"});
101         TableCellEditor editor = new TableCellEditor() {
102             int editedRow;
103 
104             public Component getTableCellEditorComponent(JTable table,
105                                                          Object value, boolean isSelected, int row, int column) {
106                 this.editedRow = row;
107                 editorCb.setSelected(Boolean.TRUE.equals(value));
108                 editorCb.setBackground(UIManager.getColor("Table.selectionBackground"));
109                 return editorCb;
110             }
111 
112             public void addCellEditorListener(CellEditorListener l) {
113             }
114 
115             public void cancelCellEditing() {
116             }
117 
118             public Object getCellEditorValue() {
119                 return editorCb.isSelected();
120             }
121 
122             public boolean isCellEditable(EventObject anEvent) {
123                 return true;
124             }
125 
126             public void removeCellEditorListener(CellEditorListener l) {
127             }
128 
129             public boolean shouldSelectCell(EventObject anEvent) {
130                 return true;
131             }
132 
133             public boolean stopCellEditing() {
134                 table.getModel().setValueAt(editorCb.isSelected(), editedRow, 0);
135                 return true;
136             }
137         };
138         table.getColumnModel().getColumn(0).setCellEditor(editor);
139 
140         TableCellRenderer renderer = new TableCellRenderer() {
141             public Component getTableCellRendererComponent(JTable table,
142                                                            Object value, boolean isSelected, boolean hasFocus,
143                                                            int row, int column) {
144                 rendererCb.setSelected(Boolean.TRUE.equals(value));
145                 return rendererCb;
146             }
147         };
148         table.getColumnModel().getColumn(0).setCellRenderer(renderer);
149 
150         p.add(table, BorderLayout.CENTER);
151 
152         f.pack();
153         f.setVisible(true);
154     }
155 }
