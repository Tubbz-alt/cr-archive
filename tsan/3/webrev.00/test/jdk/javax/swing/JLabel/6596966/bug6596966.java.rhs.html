 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 6596966
 28  * @summary Some JFileChooser mnemonics do not work with sticky keys
 29  * @library ../../regtesthelpers
 30  * @library /test/lib
 31  * @build Util jdk.test.lib.Platform
 32  * @run main bug6596966
 33  * @author Pavel Porvatov
 34  */
 35 
 36 import java.awt.*;
 37 import java.awt.event.KeyEvent;
 38 import java.awt.event.InputEvent;
 39 import java.util.ArrayList;
 40 import javax.swing.*;
 41 
 42 import jdk.test.lib.Platform;
 43 
 44 public class bug6596966 {
 45     private static JFrame frame;
 46 
 47     private static JLabel label;
 48     private static JButton button;
 49     private static JComboBox comboBox;
 50 
 51     public static void main(String[] args) throws Exception {
 52         try {
 53             Robot robot = new Robot();
 54 
 55             SwingUtilities.invokeAndWait(new Runnable() {
 56                 public void run() {
 57                     button = new JButton("Button");
 58                     comboBox = new JComboBox();
 59 
 60                     label = new JLabel("Label");
 61                     label.setDisplayedMnemonic('L');
 62                     label.setLabelFor(comboBox);
 63 
 64                     JPanel pnContent = new JPanel();
 65 
 66                     pnContent.add(button);
 67                     pnContent.add(label);
 68                     pnContent.add(comboBox);
 69 
 70                     frame = new JFrame();
 71 
 72                     frame.add(pnContent);
 73                     frame.pack();
 74                     frame.setVisible(true);
 75                 }
 76             });
 77 
 78             robot.waitForIdle();
 79 
 80 
 81             int keyMask = InputEvent.ALT_MASK;
 82             if (Platform.isOSX()) {
 83                 keyMask = InputEvent.CTRL_MASK | InputEvent.ALT_MASK;
 84             }
 85             ArrayList<Integer> keys = Util.getKeyCodesFromKeyMask(keyMask);
 86             for (int i = 0; i < keys.size(); ++i) {
 87                 robot.keyPress(keys.get(i));
 88             }
 89 
 90             robot.keyPress(KeyEvent.VK_L);
 91 
 92             robot.waitForIdle();
 93             Toolkit.getDefaultToolkit().getSystemEventQueue().postEvent(new KeyEvent(label, KeyEvent.KEY_RELEASED,
 94                     EventQueue.getMostRecentEventTime(), 0, KeyEvent.VK_L, 'L'));
 95 
 96             robot.waitForIdle();
 97 
 98             try {
 99                 SwingUtilities.invokeAndWait(new Runnable() {
100                     public void run() {
101                         if (!comboBox.isFocusOwner()) {
102                             throw new RuntimeException("comboBox isn't focus owner");
103                         }
104                     }
105                 });
106             } finally {
107                 robot.keyRelease(KeyEvent.VK_L);
108                 for (int i = 0; i < keys.size(); ++i) {
109                     robot.keyRelease(keys.get(i));
110                 }
111                 robot.waitForIdle();




112             }
113         } finally {
114             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
115         }
116     }
117 }
