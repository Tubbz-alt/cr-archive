 39 import jdk.test.lib.Platform;

 45 public class bug4796987 {
 46 
 47     private static JButton button1;
 48     private static JButton button2;

 50     public static void main(String[] args) throws Exception {
 51         if (Platform.isWindows()

 58     private static void testButtonBorder() throws Exception {
 59         Robot robot = new Robot();
 60         robot.setAutoDelay(50);
 61 
 62         SwingUtilities.invokeAndWait(new Runnable() {
 63 
 64             public void run() {
 65                 createAndShowGUI();
 66             }
 67         });
 68 
 69         robot.waitForIdle();
 70         Thread.sleep(500);
 71 
 72         Point p1 = Util.getCenterPoint(button1);
 73         Point p2 = Util.getCenterPoint(button2);
 74 
 75         Color color = robot.getPixelColor(p1.x, p2.x);
 76         for (int dx = p1.x; dx < p2.x - p1.x; dx++) {
 77             robot.mouseMove(p1.x + dx, p1.y);
 78             if (!color.equals(robot.getPixelColor(p1.x + dx, p1.y))) {
 79                 throw new RuntimeException("Button has border and background!");
 80             }
 81         }
 82     }
 83

 86         button.setBorderPainted(false);
 87         button.setFocusable(false);
 88         return button;
 89     }

 92         JFrame frame = new JFrame("Test");
 93         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 94         frame.setSize(200, 200);
 95 
 96         JButton button = new JButton();

 98 
 99         JPanel panel = new JPanel(new BorderLayout(50, 50));
100         panel.add(getButton(), BorderLayout.CENTER);

102         panel.add(button2 = getButton(), BorderLayout.EAST);
