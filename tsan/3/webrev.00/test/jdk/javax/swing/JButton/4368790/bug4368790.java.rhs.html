24 /*
25  * @test
26  * @key headful
27  *    @bug 4368790
28  *    @summary JButton stays pressed when focus stolen
29  *    @author Alexander Potochkin
30  *    @run main bug4368790
31  */
32 
33 import javax.swing.*;
34 import java.awt.*;
35 import java.awt.event.KeyEvent;
36 
37 public class bug4368790 {
38     private static JButton b1;
39     private static JFrame frame;
40 
41     private static void createGui() {
42         frame = new JFrame();
43         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
44         frame.setLayout(new FlowLayout());
45 
46         b1 = new JButton("Button1");
47         frame.add(b1);
48         frame.add(new JButton("Button2"));
49 
50         frame.setSize(200, 200);
51         frame.setLocationRelativeTo(null);
52         frame.setVisible(true);
53         b1.requestFocus();
54     }
55 
56     public static void main(String[] args) throws Exception {
57         try {
58             Robot robot = new Robot();
59             robot.setAutoDelay(50);
60             SwingUtilities.invokeAndWait(new Runnable() {
61                 public void run() {
62                     bug4368790.createGui();
63                 }
64             });
65             robot.waitForIdle();
66             robot.keyPress(KeyEvent.VK_SPACE);
67             robot.keyPress(KeyEvent.VK_TAB);
68             robot.keyRelease(KeyEvent.VK_TAB);
69             robot.keyRelease(KeyEvent.VK_SPACE);
70             robot.waitForIdle();
71             if (b1.getModel().isPressed()) {
72                 throw new RuntimeException("The button is unexpectedly pressed");
73             }
74         } finally {
75             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());







76         }
77     }
78 }
