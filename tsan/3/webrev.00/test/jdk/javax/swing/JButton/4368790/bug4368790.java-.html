24 /*
25  * @test
26  * @key headful
27  *    @bug 4368790
28  *    @summary JButton stays pressed when focus stolen
29  *    @author Alexander Potochkin
30  *    @run main bug4368790
31  */
32 
33 import javax.swing.*;
34 import java.awt.*;
35 import java.awt.event.KeyEvent;
36 
37 public class bug4368790 {
38     private static JButton b1;
39 
40     private static void createGui() {
41         final JFrame frame = new JFrame();
42         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
43         frame.setLayout(new FlowLayout());
44 
45         b1 = new JButton("Button1");
46         frame.add(b1);
47         frame.add(new JButton("Button2"));
48 
49         frame.setSize(200, 200);
50         frame.setLocationRelativeTo(null);
51         frame.setVisible(true);
52         b1.requestFocus();
53     }
54 
55     public static void main(String[] args) throws Exception {
56         Robot robot = new Robot();
57         robot.setAutoDelay(50);
58         SwingUtilities.invokeAndWait(new Runnable() {
59             public void run() {
60                 bug4368790.createGui();
61             }
62         });
63         robot.waitForIdle();
64         robot.keyPress(KeyEvent.VK_SPACE);
65         robot.keyPress(KeyEvent.VK_TAB);
66         robot.keyRelease(KeyEvent.VK_TAB);
67         robot.keyRelease(KeyEvent.VK_SPACE);
68         robot.waitForIdle();
69         if (b1.getModel().isPressed()) {
70             throw new RuntimeException("The button is unexpectedly pressed");
71         }
72     }
73 }
