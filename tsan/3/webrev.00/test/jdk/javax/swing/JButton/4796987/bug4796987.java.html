 24 /*
 25  * @test
 26  * @bug 4796987
 27  * @key headful
 28  * @requires (os.family == "windows")
 29  * @summary XP Only: JButton.setBorderPainted() does not work with XP L&F
 30  * @author Alexander Scherbatiy
 31  * @library ../../regtesthelpers
 32  * @library /test/lib
 33  * @modules java.desktop/com.sun.java.swing.plaf.windows
 34  *          java.desktop/sun.awt
 35  * @build jdk.test.lib.OSVersion jdk.test.lib.Platform
 36  * @build Util
 37  * @run main bug4796987
 38  */
 39 
 40 import jdk.test.lib.Platform;
 41 import jdk.test.lib.OSVersion;
 42 import java.awt.*;
 43 import javax.swing.*;
 44 import com.sun.java.swing.plaf.windows.WindowsLookAndFeel;
 45 
 46 public class bug4796987 {
 47 
 48     private static JButton button1;
 49     private static JButton button2;
 50     private static JFrame frame;
 51 
 52     public static void main(String[] args) throws Exception {
 53         try {
 54             if (Platform.isWindows()
 55                 && OSVersion.current().equals(OSVersion.WINDOWS_XP)) {
 56                 UIManager.setLookAndFeel(new WindowsLookAndFeel());
 57                 testButtonBorder();
 58             }
 59         } finally {
 60             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
 61         }
 62     }
 63 
 64     private static void testButtonBorder() throws Exception {
 65         Robot robot = new Robot();
 66         robot.setAutoDelay(50);
 67 
 68         SwingUtilities.invokeAndWait(new Runnable() {
 69 
 70             public void run() {
 71                 createAndShowGUI();
 72             }
 73         });
 74 
 75         robot.waitForIdle();
 76         Thread.sleep(500);
 77 
 78         Point p1 = Util.getCenterPoint(button1);
 79         Point p2 = Util.getCenterPoint(button2);
 80 
 81         Color color = robot.getPixelColor(p1.x, p2.x);
 82         for (int dx = p1.x; dx < p2.x - p1.x; dx++) {
 83             robot.mouseMove(p1.x + dx, p1.y);
 84             if (!color.equals(robot.getPixelColor(p1.x + dx, p1.y))) {
 85                 throw new RuntimeException("Button has border and background!");
 86             }
 87         }
 88     }
 89 
 90     private static JButton getButton() {
 91         JButton button = new JButton();
 92         button.setBorderPainted(false);
 93         button.setFocusable(false);
 94         return button;
 95     }
 96 
 97     private static void createAndShowGUI() {
 98         frame = new JFrame("Test");
 99         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
100         frame.setSize(200, 200);
101 
102         JButton button = new JButton();
103         button.setBorder(null);
104 
105         JPanel panel = new JPanel(new BorderLayout(50, 50));
106         panel.add(getButton(), BorderLayout.CENTER);
107         panel.add(button1 = getButton(), BorderLayout.WEST);
108         panel.add(button2 = getButton(), BorderLayout.EAST);
109         frame.getContentPane().add(panel);
110         frame.setVisible(true);
111     }
112 }
