 24 import java.awt.BorderLayout;
 25 import java.awt.Color;
 26 import java.awt.Point;
 27 import java.awt.Rectangle;
 28 import java.awt.Robot;
 29 import java.awt.event.InputEvent;
 30 import java.awt.image.BufferedImage;
 31 import javax.swing.JDesktopPane;
 32 import javax.swing.JFrame;
 33 import javax.swing.JInternalFrame;
 34 import javax.swing.SwingUtilities;
 35 
 36 /**
 37  * @test
 38  * @key headful
 39  * @bug 8160248 8160332 8186513
 40  * @summary Dragged internal frame leaves artifacts for floating point ui scale
 41  * @run main/othervm -Dsun.java2d.uiScale=1.2 JInternalFrameDraggingTest
 42  * @run main/othervm -Dsun.java2d.uiScale=1.5 JInternalFrameDraggingTest
 43  * @run main/othervm -Dsun.java2d.uiScale=1 JInternalFrameDraggingTest
 44  * @run main/othervm -Dsun.java2d.uiScale=2.5 JInternalFrameDraggingTest
 45  */
 46 
 47 public class JInternalFrameDraggingTest {
 48 
 49     private static JFrame frame;
 50     private static JDesktopPane desktopPane;
 51     private static JInternalFrame internalFrame;
 52     private static int FRAME_SIZE = 500;
 53     private static Color BACKGROUND_COLOR = Color.ORANGE;
 54 
 55     public static void main(String[] args) throws Exception {
 56 
 57         Robot robot = new Robot();
 58         robot.setAutoDelay(20);
 59         SwingUtilities.invokeAndWait(JInternalFrameDraggingTest::createAndShowGUI);
 60         robot.waitForIdle();
 61 
 62         final int translate = FRAME_SIZE / 4;
 63         moveFrame(robot, translate, translate / 2, translate / 2);
 64         robot.waitForIdle();
 65 
 66         Point p = getDesktopPaneLocation();
 67         int size = translate / 2;
 68         Rectangle rect = new Rectangle(p.x, p.y, size, size);
 69         BufferedImage img = robot.createScreenCapture(rect);
 70 
 71         int testRGB = BACKGROUND_COLOR.getRGB();
 72         for (int i = 1; i < size; i++) {
 73             int rgbCW = img.getRGB(i, size / 2);
 74             int rgbCH = img.getRGB(size / 2, i);
 75             if (rgbCW != testRGB || rgbCH != testRGB) {
 76                 System.out.println("i " + i + " rgbCW " +
 77                                    Integer.toHexString(rgbCW) +
 78                                    " testRGB " + Integer.toHexString(testRGB) +
 79                                    " rgbCH " + Integer.toHexString(rgbCH));
 80                 throw new RuntimeException("Background color is wrong!");

 81             }


 82         }
 83     }
 84 
 85     private static void createAndShowGUI() {
 86 
 87         frame = new JFrame();
 88         frame.setUndecorated(true);
 89         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 90         frame.setLayout(new BorderLayout());
 91 
 92         desktopPane = new JDesktopPane();
 93         desktopPane.setBackground(BACKGROUND_COLOR);
 94 
 95         frame.add(desktopPane, BorderLayout.CENTER);
 96         frame.setSize(FRAME_SIZE, FRAME_SIZE);
 97         frame.setVisible(true);
 98 
 99         internalFrame = new JInternalFrame("Test");
100         internalFrame.setSize(FRAME_SIZE / 2, FRAME_SIZE / 2);
101         desktopPane.add(internalFrame);
102         internalFrame.setVisible(true);
103         internalFrame.setResizable(true);
104 
105         frame.setVisible(true);
106     }
107 
108     private static void moveFrame(Robot robot, int w, int h, int N) throws Exception {
109         Point p = getInternalFrameLocation();
110         int xs = p.x + 100;
111         int ys = p.y + 15;
112         robot.mouseMove(xs, ys);
113         try {
114             robot.mousePress(InputEvent.BUTTON1_MASK);
115 
116             int dx = w / N;
117             int dy = h / N;
118 
119             int y = ys;
120             for (int x = xs; x < xs + w; x += dx, y += dy) {
121                 robot.mouseMove(x, y);
122             }
123         } finally {
124             robot.mouseRelease(InputEvent.BUTTON1_MASK);
125         }
126     }
127 
128     private static Point getInternalFrameLocation() throws Exception {
129         final Point[] points = new Point[1];
130         SwingUtilities.invokeAndWait(() -> {
131             points[0] = internalFrame.getLocationOnScreen();
132         });
133         return points[0];
134     }
135 
136     private static Point getDesktopPaneLocation() throws Exception {
137         final Point[] points = new Point[1];
138         SwingUtilities.invokeAndWait(() -> {
139             points[0] = desktopPane.getLocationOnScreen();
140         });
141         return points[0];
142     }
143 }
