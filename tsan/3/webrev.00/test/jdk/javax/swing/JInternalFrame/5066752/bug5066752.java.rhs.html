23 /*
24   @test
25   @key headful
26   @bug 5066752
27   @summary  Transparent JDesktopPane impossible because isOpaque() returns true
28   @author mb50250: area=JDesktopPane
29   @library ../../regtesthelpers
30   @build Util
31   @run main bug5066752
32 */
33 
34 import java.awt.*;
35 import javax.swing.*;
36 
37 public class bug5066752
38 {
39     private static JFrame frame;
40 
41     public static void main(String[] args) throws Exception {
42         try {
43             Robot r = new Robot();
44             SwingUtilities.invokeAndWait(new Runnable() {
45                 public void run() {
46                     createAndShowGUI();
47                 }
48             });
49             r.waitForIdle();
50 
51             r.delay(600);
52 
53             Point p = Util.getCenterPoint(frame);
54             Color color =  r.getPixelColor((int) p.getX(), (int) p.getY());
55             if (!color.equals(Color.RED)) {
56                 throw new Exception("Test failed: JDesktopPane isn't transparent. Expected color is (red color): " + Color.RED + ", actual color is: " + color);
57             }
58         } finally {
59             if (frame != null) SwingUtilities.invokeAndWait(() -> frame.dispose());
60         }
61     }
62 
63     private static void createAndShowGUI() {
64         frame = new JFrame();
65 
66         frame.setLayout(new BorderLayout());
67         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
68 
69         JPanel panel = new JPanel();
70         panel.setLayout(new BorderLayout());
71         panel.setBackground(Color.RED);
72         frame.add(panel, BorderLayout.CENTER);
73 
74         JDesktopPane dp = new JDesktopPane();
75         dp.setOpaque(false);
76         panel.add(dp, BorderLayout.CENTER);
77 
78         frame.setBounds(200, 200, 300, 200);
79         frame.setVisible(true);
80     }
81 }
