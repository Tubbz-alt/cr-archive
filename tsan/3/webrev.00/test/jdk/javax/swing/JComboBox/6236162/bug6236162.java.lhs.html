 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 6236162
 28  * @summary Checks that there is no an inconsistence in combo box
 29  *          behavior when user points an item in combo popup
 30  *          by mouse and then uses UP/DOWN keys.
 31  * @library ../../regtesthelpers
 32  * @build Util
 33  * @author Mikhail Lapshin
 34  * @run main bug6236162
 35  */
 36 
 37 import javax.swing.*;
 38 import javax.swing.plaf.basic.*;
 39 import javax.swing.plaf.metal.MetalComboBoxUI;
 40 import java.awt.*;
 41 import java.awt.event.KeyEvent;
 42 
 43 public class bug6236162 {
 44     private static JFrame frame;
 45     private static JComboBox combo;
 46     private static MyComboUI comboUI;
 47 
 48     public static void main(String[] args) throws Exception {
 49         UIManager.setLookAndFeel(UIManager.getCrossPlatformLookAndFeelClassName());
 50         SwingUtilities.invokeAndWait(new Runnable() {
 51             public void run() {
 52                 createAndShowGUI();
 53             }
 54         });
 55         test();
 56         System.out.println("Test passed");




 57     }
 58 
 59     private static void createAndShowGUI() {
 60         frame = new JFrame("bug6236162");
 61 
 62         combo = new JComboBox(new String[]{"one", "two", "three", "four", "five"});
 63         combo.setEditable(true);
 64         comboUI = new MyComboUI();
 65         combo.setUI(comboUI);
 66         combo.setSelectedIndex(3);
 67         frame.getContentPane().add(combo);
 68 
 69         frame.pack();
 70         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 71         frame.setLocationRelativeTo(null);
 72         frame.setVisible(true);
 73     }
 74 
 75     private static void test() throws AWTException {
 76         Robot robot = new Robot();
 77         robot.setAutoDelay(50);
 78 
 79         // Open popup menu
 80         robot.waitForIdle();
 81         Util.hitKeys(robot, KeyEvent.VK_DOWN);
 82 
 83         // Move mouse to the first popup menu item
 84         robot.waitForIdle();
 85         Point p = combo.getLocationOnScreen();
 86         Dimension size = combo.getSize();
 87         p.x += size.width / 2;
 88         p.y += size.height;
 89         float dy = 1;
 90         robot.mouseMove(p.x, p.y - 5);
 91         for (int i=1; i <= 10; i++) {
 92             robot.mouseMove((int)(p.x), (int)(p.y - 5 + dy*i));
 93         }
 94 
 95         // Select the second popup menu item
 96         robot.waitForIdle();
 97         Util.hitKeys(robot, KeyEvent.VK_DOWN);
 98 
 99         robot.waitForIdle();
100         JList list = comboUI.getComboPopup().getList();
101         if (list.getSelectedIndex() != 1) {
102             throw new RuntimeException("There is an inconsistence in combo box " +
103                     "behavior when user points an item in combo popup " +
104                     "by mouse and then uses UP/DOWN keys.");
105         }
106     }
107 
108 
109     // Gives access to BasicComboBoxUI.popup field
110     private static class MyComboUI extends MetalComboBoxUI {
111         public ComboPopup getComboPopup() {
112             return popup;
113         }
114     }
115 }
