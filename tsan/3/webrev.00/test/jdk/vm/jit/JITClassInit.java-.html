32 public class JITClassInit {

39 
40 class TokenTable {
41     public TokenTable() {
42         new TokenTypeIterator(this);
43     }
44 
45     public void for_token_type(Token t) {
46         t.keyword_character_class();
47     }
48 }
49 
50 class Token {
51     public Object keyword_character_class() {
52         return new Object();
53     }
54 }
55 
56 class NameOrKeywordToken extends Token {
57     static TokenTable kt = new TokenTable();
58     public Object keyword_character_class() {
59         return new Object();
60     }
61 }
62 
63 class CapKeywordToken extends NameOrKeywordToken {
64     public Object keyword_character_class() {
65         return new Object();
66     }
67 };
68 
69 
70 class TokenTypeIterator {
71     public TokenTypeIterator(TokenTable c) {
72         c.for_token_type(new CapKeywordToken());
73         c.for_token_type(new NameOrKeywordToken());
74     }
75 }
