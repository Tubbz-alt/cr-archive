24 /*
25  * @test
26  * @bug 4098578
27  * @summary Check if the VM properly throws OutOfMemoryError
28  * @author Sheng Liang
29  * @run main/othervm -Xmx25M InfiniteList
30  */
31 public class InfiniteList {
32     InfiniteList next;
33     long data[] = new long[50000];
34     public static void main(String[] args) throws Exception {
35         InfiniteList p, q;
36         p = new InfiniteList ();
37         p.data[p.data.length -1 ] = 999;
38         try {
39             while (p != null) {
40                 q = new InfiniteList ();
41                 q.next = p;
42                 p = q;
43             }
44             throw new Exception ("OutOfMemoryError not thrown as expected.");
45         } catch (OutOfMemoryError e) {
46             return;
47         }
48     }
49 }
