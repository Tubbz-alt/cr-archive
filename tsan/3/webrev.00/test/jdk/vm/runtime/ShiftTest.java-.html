24 /*
25  * @test
26  * @bug 4093292
27  * @summary Test for correct code generation by the JIT
28  */
29 public class ShiftTest {
30     static final int w = 32;
31 
32     private static void doTest(long ct) throws Exception {
33         int S22 = 0xc46cf7c2;
34         int S23 = 0xcfda9162;
35         int S24 = 0xd029aa4c;
36         int S25 = 0x17cf1801;
37         int A = (int)(ct & 0xffffffffL);
38         int B = (int)(ct >>> 32);
39         int x, y;
40         x = B - S25;
41         y = A & (w-1);
42         B = ((x >>> y) | (x << (w-y))) ^ A;
43         x = A - S24;
44         y = B & (w-1);
45         A = ((x >>> y) | (x << (w-y))) ^ B;
46         x = B - S23;
47         y = A & (w-1);
48         B = ((x >>> y) | (x << (w-y))) ^ A;
49         x = A - S22;
50         y = B & (w-1);
51         A = ((x >>> y) | (x << (w-y))) ^ B;
52         String astr = Integer.toHexString(A);
53         String bstr = Integer.toHexString(B);
54         System.err.println("A = " + astr + " B = " + bstr);
55         if ((!astr.equals("dcb38144")) ||
56             (!bstr.equals("1916de73"))) {
57             throw new RuntimeException("Unexpected shift results!");
58         }
59         System.err.println("Test passed");
60     }
61 
62     public static void main(String[] args) throws Exception {
63         doTest(0x496def29b74be041L);
64     }
65 }
