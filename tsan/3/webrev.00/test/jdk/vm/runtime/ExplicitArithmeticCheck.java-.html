24 /*
25  * @test
26  * @bug 4221448
27  * @summary Use explicit check for integer arithmetic exception on win32.
28  */
29 
30 public class ExplicitArithmeticCheck {
31     public static void main(String argv[]) throws Exception {
32         for (int i = 0; i < 64; i++) {
33           boolean result = false;
34           int n;
35           try {
36               n = 0 / 0;
37           } catch (ArithmeticException e) {
38               result = true;
39           }
40           if (result == false) {
41             throw new Error("Failed to throw correct exception!");
42           }
43           result = false;
44           try {
45               n = 0 % 0;
46           } catch (ArithmeticException e) {
47               result = true;
48           }
49           if (result == false) {
50             throw new Error("Failed to throw correct exception!");
51           }
52           try {
53               n = 0x80000000 / -1;
54           } catch (Throwable t) {
55             throw new Error("Failed to throw correct exception!");
56           }
57           if (n != 0x80000000) {
58             throw new Error("Incorrect integer arithmetic ! ");
59           }
60           try {
61               n = 0x80000000 % -1;
62           } catch (Throwable t) {
63             throw new Error("Failed to throw correct exception!");
64           }
65           if (n != 0) {
66             throw new Error("Incorrect integer arithmetic!");
67           }
68         }
69     }
70 }
