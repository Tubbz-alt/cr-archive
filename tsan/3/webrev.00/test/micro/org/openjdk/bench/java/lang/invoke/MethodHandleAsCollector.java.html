 23 package org.openjdk.bench.java.lang.invoke;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Mode;
 28 import org.openjdk.jmh.annotations.OutputTimeUnit;
 29 import org.openjdk.jmh.annotations.Scope;
 30 import org.openjdk.jmh.annotations.Setup;
 31 import org.openjdk.jmh.annotations.State;
 32 
 33 import java.lang.invoke.MethodHandle;
 34 import java.lang.invoke.MethodHandles;
 35 import java.lang.invoke.MethodType;
 36 import java.util.concurrent.TimeUnit;
 37 
 38 /**
 39  * Microbenchmark assesses MethodHandle.asCollector() performance
 40  */
 41 @BenchmarkMode(Mode.AverageTime)
 42 @OutputTimeUnit(TimeUnit.NANOSECONDS)
 43 @State(Scope.Thread)
 44 public class MethodHandleAsCollector {
 45 
 46     /*
 47     * Implementation notes:
 48     *   - simple array-parameter method is being called
 49     *   - baselineRaw calls method directly with dynamically instantiating the array
 50     *   - baselineCached calls method directly with pre-cached array
 51     *   - additional testCreate() test harnesses the collector acquisition performance
 52     *   - testCollector() can be faster than both baselines: it can wrapping array at all
 53     */
 54 
 55     public int i;
 56     private static MethodHandle mh;
 57     private static MethodHandle collectorMH;
 58     private static int[] cachedArgs;
 59 
 60     @Setup
 61     public void setup() throws IllegalAccessException, NoSuchMethodException {
 62         mh = MethodHandles.lookup().findVirtual(MethodHandleAsCollector.class, "doWork", MethodType.methodType(void.class, int[].class));
 63         collectorMH = mh.asCollector(int[].class, 5);
 64         cachedArgs = new int[]{1, 2, 3, 4, 5};
 65     }
 66 
 67     @Benchmark
 68     public void baselineMH() throws Throwable {
 69         mh.invokeExact(this, new int[] { 1, 2, 3, 4, 5 });
 70     }
 71 
 72     @Benchmark
 73     public void baselineMHCached() throws Throwable {
 74         mh.invokeExact(this, cachedArgs);
 75     }
 76 
 77     @Benchmark
 78     public void baselineRaw() throws Throwable {
 79         doWork(new int[] { 1, 2, 3, 4, 5});
 80     }
 81 
 82     @Benchmark
 83     public void baselineRawCached() throws Throwable {
 84         doWork(cachedArgs);
 85     }
 86 
 87     @Benchmark
 88     public MethodHandle testCreate() {
 89         return mh.asCollector(int[].class, 5);
 90     }
 91 
 92     @Benchmark
 93     public void testCollector() throws Throwable {
 94         collectorMH.invokeExact(this, 1, 2, 3, 4, 5);
 95     }
 96 
 97     public void doWork(int[] args) {
 98         for (int a : args) {
 99             i += a;
100         }
101     }
102 
103 }
