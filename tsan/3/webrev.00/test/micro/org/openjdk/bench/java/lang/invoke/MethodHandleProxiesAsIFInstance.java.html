 23 package org.openjdk.bench.java.lang.invoke;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Mode;
 28 import org.openjdk.jmh.annotations.OutputTimeUnit;
 29 import org.openjdk.jmh.annotations.Scope;
 30 import org.openjdk.jmh.annotations.Setup;
 31 import org.openjdk.jmh.annotations.State;
 32 
 33 import java.lang.invoke.MethodHandle;
 34 import java.lang.invoke.MethodHandleProxies;
 35 import java.lang.invoke.MethodHandles;
 36 import java.lang.invoke.MethodType;
 37 import java.util.concurrent.TimeUnit;
 38 
 39 /**
 40  * Benchmark evaluates the performance of MethodHandleProxies.*
 41  */
 42 @BenchmarkMode(Mode.AverageTime)
 43 @OutputTimeUnit(TimeUnit.NANOSECONDS)
 44 @State(Scope.Thread)
 45 public class MethodHandleProxiesAsIFInstance {
 46 
 47     /**
 48      * Implementation notes:
 49      *   - asInterfaceInstance() can only target static MethodHandle (adapters needed to call instance method?)
 50      *   - baselineCompute will quickly degrade to GC test, if escape analysis is unable to spare the allocation
 51      *   - testCreate* will always be slower if allocation is not eliminated; baselineAllocCompute makes sure allocation is present
 52      */
 53 
 54     public int i;
 55 
 56     private MethodHandle target;
 57     private Doable precreated;
 58 
 59     @Setup
 60     public void setup() throws Throwable {
 61         target = MethodHandles.lookup().findStatic(MethodHandleProxiesAsIFInstance.class, "doWork", MethodType.methodType(int.class, int.class));
 62         precreated = MethodHandleProxies.asInterfaceInstance(Doable.class, target);
 63     }
 64 
 65     @Benchmark
 66     public Doable testCreate() {
 67         Doable doable = MethodHandleProxies.asInterfaceInstance(Doable.class, target);
 68         return doable;              // make sure allocation happens
 69     }
 70 
 71     @Benchmark
 72     public Doable testCreateCall() {
 73         Doable doable = MethodHandleProxies.asInterfaceInstance(Doable.class, target);
 74         i = doable.doWork(i);       // make sure computation happens
 75         return null;                // let allocation be eliminated
 76     }
 77 
 78     @Benchmark
 79     public Doable testCall() {
 80         i = precreated.doWork(i);   // make sure computation happens
 81         return precreated;
 82     }
 83 
 84     @Benchmark
 85     public Doable baselineCompute() {
 86         Doable doable = new Doable() {
 87             @Override
 88             public int doWork(int i) {
 89                 return MethodHandleProxiesAsIFInstance.doWork(i);
 90             }
 91         };
 92 
 93         i = doable.doWork(i);       // make sure computation happens
 94         return null;                // let allocation be eliminated
 95     }
 96 
 97     @Benchmark
 98     public Doable baselineAllocCompute() {
 99         Doable doable = new Doable() {
100             @Override
101             public int doWork(int i) {
102                 return MethodHandleProxiesAsIFInstance.doWork(i);
103             }
104         };
105 
106         i = doable.doWork(i);       // make sure computation happens
107         return doable;              // make sure allocation happens
108     }
109 
110     public static int doWork(int i) {
111         return i + 1;
112     }
113 
114     public interface Doable {
115         int doWork(int i);
116     }
117 
118 }
