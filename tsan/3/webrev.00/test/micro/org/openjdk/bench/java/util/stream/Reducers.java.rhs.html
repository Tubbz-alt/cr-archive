23 package org.openjdk.bench.java.util.stream;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Param;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.concurrent.TimeUnit;
35 import java.util.function.LongBinaryOperator;
36 import java.util.stream.LongStream;
37 
38 /**
39  * Benchmark for checking different reduce schemes.
40  */
41 @BenchmarkMode(Mode.Throughput)
42 @OutputTimeUnit(TimeUnit.SECONDS)
43 @State(Scope.Thread)
44 public class Reducers {
45 
46     /**
47      * Implementation notes:
48      *   - parallel version requires thread-safe sink, we use the same for sequential version for better comparison
49      *   - operations are explicit inner classes to untangle unwanted lambda effects
50      *   - the result of applying consecutive operations is the same, in order to have the same number of elements in sink
51      */
52 
53     @Param("100000")
54     private int size;
55 
56     private LongBinaryOperator op;
57 
58     @Setup
59     public void setup() {
60         op = new LongBinaryOperator() {
61             @Override
62             public long applyAsLong(long l, long r) {
63                 return (l > r) ? l : r;
64             }
65         };
66     }
67 
68     @Benchmark
69     public long seq_reduce() {
70         return LongStream.range(0, size).reduce(op).getAsLong();
71     }
72 
73     @Benchmark
74     public long par_reduce() {
75         return LongStream.range(0, size).parallel().reduce(op).getAsLong();
76     }
77 
78     @Benchmark
79     public long seq_reduce_base() {
80         return LongStream.range(0, size).reduce(0L, op);
81     }
82 
83     @Benchmark
84     public long par_reduce_base() {
85         return LongStream.range(0, size).parallel().reduce(0L, op);
86     }
87 
88 
89 }
