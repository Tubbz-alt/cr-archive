 23 package org.openjdk.bench.java.util.stream.ops.value;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Mode;
 28 import org.openjdk.jmh.annotations.OutputTimeUnit;
 29 import org.openjdk.jmh.annotations.Param;
 30 import org.openjdk.jmh.annotations.Scope;
 31 import org.openjdk.jmh.annotations.Setup;
 32 import org.openjdk.jmh.annotations.State;
 33 
 34 import java.util.concurrent.TimeUnit;
 35 import java.util.function.LongPredicate;
 36 import java.util.stream.LongStream;
 37 
 38 /**
 39  * Benchmark for anyMatch() operation.
 40  */
 41 @BenchmarkMode(Mode.Throughput)
 42 @OutputTimeUnit(TimeUnit.SECONDS)
 43 @State(Scope.Thread)
 44 public class AnyMatch {
 45 
 46     /**
 47      * Implementation notes:
 48      *   - operations are explicit inner classes to untangle unwanted lambda effects
 49      *   - the predicates are always false to avert shortcurcuiting
 50      */
 51     @Param("100000")
 52     private int size;
 53 
 54     private LongPredicate p1, p2, p3;
 55 
 56     @Setup
 57     public void setup() {
 58         p1 = new LongPredicate() {
 59             @Override
 60             public boolean test(long v) {
 61                 return false;
 62             }
 63         };
 64         p2 = new LongPredicate() {
 65             @Override
 66             public boolean test(long v) {
 67                 return false;
 68             }
 69         };
 70         p3 = new LongPredicate() {
 71             @Override
 72             public boolean test(long v) {
 73                 return false;
 74             }
 75         };
 76     }
 77 
 78     @Benchmark
 79     public boolean seq_invoke() {
 80         return LongStream.range(0, size).anyMatch(p1);
 81     }
 82 
 83     @Benchmark
 84     public int seq_chain111() {
 85         int s = 0;
 86         s += (LongStream.range(0, size).anyMatch(p1)) ? 1 : 0;
 87         s += (LongStream.range(0, size).anyMatch(p1)) ? 1 : 0;
 88         s += (LongStream.range(0, size).anyMatch(p1)) ? 1 : 0;
 89         return s;
 90     }
 91 
 92     @Benchmark
 93     public int seq_chain123() {
 94         int s = 0;
 95         s += (LongStream.range(0, size).anyMatch(p1)) ? 1 : 0;
 96         s += (LongStream.range(0, size).anyMatch(p2)) ? 1 : 0;
 97         s += (LongStream.range(0, size).anyMatch(p3)) ? 1 : 0;
 98         return s;
 99     }
100 
101     @Benchmark
102     public boolean par_invoke() {
103         return LongStream.range(0, size).parallel().anyMatch(p1);
104     }
105 
106     @Benchmark
107     public int par_chain111() {
108         int s = 0;
109         s += (LongStream.range(0, size).parallel().anyMatch(p1)) ? 1 : 0;
110         s += (LongStream.range(0, size).parallel().anyMatch(p1)) ? 1 : 0;
111         s += (LongStream.range(0, size).parallel().anyMatch(p1)) ? 1 : 0;
112         return s;
113     }
114 
115     @Benchmark
116     public int par_chain123() {
117         int s = 0;
118         s += (LongStream.range(0, size).parallel().anyMatch(p1)) ? 1 : 0;
119         s += (LongStream.range(0, size).parallel().anyMatch(p2)) ? 1 : 0;
120         s += (LongStream.range(0, size).parallel().anyMatch(p3)) ? 1 : 0;
121         return s;
122     }
123 
124 }
