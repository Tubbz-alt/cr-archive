23 package org.openjdk.bench.javax.tools;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Scope;
30 import org.openjdk.jmh.annotations.Setup;
31 import org.openjdk.jmh.annotations.State;
32 
33 import javax.tools.JavaCompiler;
34 import javax.tools.SimpleJavaFileObject;
35 import javax.tools.StandardJavaFileManager;
36 import javax.tools.ToolProvider;
37 import java.net.URI;
38 import java.util.ArrayList;
39 import java.util.List;
40 import java.util.concurrent.TimeUnit;
41 
42 @BenchmarkMode(Mode.AverageTime)
43 @OutputTimeUnit(TimeUnit.MILLISECONDS)
44 @State(Scope.Thread)
45 public class Javac {
46     private List<JavaSourceFromString> compilationUnits;
47     private JavaCompiler compiler;
48     private StandardJavaFileManager fileManager;
49 
50     @Setup
51     public void prepare() {
52         String helloWorld = "class Apan {   \n" + "  public static void main(String args[]) {\n"
53                 + "     System.out.println(\"hej apa\");\n" + "  }\n" + "}\n";
54 
55         compiler = ToolProvider.getSystemJavaCompiler();
56         fileManager = compiler.getStandardFileManager(null, null, null);
57         compilationUnits = new ArrayList<>();
58         compilationUnits.add(new JavaSourceFromString("Apan", helloWorld));
59     }
60 
61     @Benchmark
62     public Boolean testCompile() throws Exception {
63         JavaCompiler.CompilationTask task = compiler.getTask(null, fileManager, null, null, null, compilationUnits);
64         return task.call();
65     }
66 
67     private static class JavaSourceFromString extends SimpleJavaFileObject {
68         final String code;
69 
70         JavaSourceFromString(String name, String code) {
71             super(URI.create("string:///" + name.replace('.', '/') + Kind.SOURCE.extension),
72                     Kind.SOURCE);
73             this.code = code;
74         }
75 
76         @Override
77         public CharSequence getCharContent(boolean ignoreEncodingErrors) {
78             return code;
79         }
80     }
81 }
