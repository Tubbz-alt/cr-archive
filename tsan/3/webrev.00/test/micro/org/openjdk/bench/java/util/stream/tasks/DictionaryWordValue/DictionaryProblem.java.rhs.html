 23 package org.openjdk.bench.java.util.stream.tasks.DictionaryWordValue;
 24 
 25 import org.openjdk.bench.java.util.stream.tasks.DataProviders;
 26 
 27 import java.util.Collections;
 28 import java.util.Comparator;
 29 import java.util.List;
 30 import java.util.stream.Collectors;
 31 import java.util.stream.Stream;
 32 
 33 public class DictionaryProblem {
 34 
 35     private static final int DICTIONARY_REPEAT_RATE = 40;
 36 
 37     private static final String[] dict;
 38 
 39     static {
 40         int size;
 41         int idx = 0;
 42         List<String> d = Collections.emptyList();
 43         try (Stream<String> s = DataProviders.dictionary()) {
 44             d = s.collect(Collectors.<String>toList());
 45         } catch (Exception e) {
 46             // ignore
 47         }
 48         size = d.size() * DICTIONARY_REPEAT_RATE;
 49         dict = new String[size];
 50         for (int i = 0; i < DICTIONARY_REPEAT_RATE; i++) {
 51             d.sort(new IdxComparator(i));
 52             for (String s : d) {
 53                 // copy the whole string
 54                 dict[idx++] = new String(s.toCharArray());
 55             }
 56         }
 57         assert (idx == dict.length);
 58     }
 59 
 60     public static String[] get() {
 61         return dict;
 62     }
 63 
 64     /**
 65      * A word value is the sum of alphabet value of each characters in a word.
 66      *
 67      * @param word The word
 68      * @return The word value
 69      */
 70     public static int wordValue(String word) {
 71         char[] ar = word.toLowerCase().toCharArray();
 72         int value = 0;
 73         for (char c: ar) {
 74             int v = c - 'a' + 1;
 75             if (v < 1 || v > 26) {
 76                 // skip non-alphabet
 77                 continue;
 78             }
 79             value += (c - 'a' + 1);
 80         }
 81         return value;
 82     }
 83 
 84     static class IdxComparator implements Comparator<String> {
 85         private final int index;
 86 
 87         public IdxComparator(int i) {
 88             index = i;
 89         }
 90 
 91         @Override
 92         public int compare(String a, String b) {
 93             if (a.length() > index && b.length() > index) {
 94                 return (a.charAt(index) - b.charAt(index));
 95             } else {
 96                 return (a.length() - b.length());
 97             }
 98         }
 99     }
100 
101 }
