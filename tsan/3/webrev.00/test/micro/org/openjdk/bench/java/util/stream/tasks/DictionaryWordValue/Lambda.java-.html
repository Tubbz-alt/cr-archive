23 package org.openjdk.bench.java.util.stream.tasks.DictionaryWordValue;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Level;
28 import org.openjdk.jmh.annotations.Mode;
29 import org.openjdk.jmh.annotations.OutputTimeUnit;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.Arrays;
35 import java.util.concurrent.TimeUnit;
36 
37 /**
38  * Bulk scenario: searching max "wordValue" through the dictionary (array of strings).
39  */
40 @BenchmarkMode(Mode.Throughput)
41 @OutputTimeUnit(TimeUnit.SECONDS)
42 @State(Scope.Benchmark)
43 public class Lambda {
44 
45     @Setup(Level.Trial)
46     public void loadData() {
47         // cause classload and problem initialization
48         DictionaryProblem.class.getName();
49     }
50 
51     public static Integer maxInt(Integer l, Integer r) {
52         return l > r ? l : r;
53     }
54 
55     @Benchmark
56     public int bulk_seq_lambda() {
57         return Arrays.stream(DictionaryProblem.get())
58                 .map(s -> DictionaryProblem.wordValue(s))
59                 .reduce(0, (l, r) -> l > r ? l : r);
60     }
61 
62     @Benchmark
63     public int bulk_seq_mref() {
64         return Arrays.stream(DictionaryProblem.get())
65                 .map(DictionaryProblem::wordValue)
66                 .reduce(0, Lambda::maxInt);
67     }
68 
69     @Benchmark
70     public int bulk_par_lambda() {
71         return Arrays.stream(DictionaryProblem.get()).parallel()
72                 .map(s -> DictionaryProblem.wordValue(s))
73                 .reduce(0, (l, r) -> l > r ? l : r);
74     }
75 
76     @Benchmark
77     public int bulk_par_mref() {
78         return Arrays.stream(DictionaryProblem.get()).parallel()
79                 .map(DictionaryProblem::wordValue)
80                 .reduce(0, Lambda::maxInt);
81     }
82 
83 }
