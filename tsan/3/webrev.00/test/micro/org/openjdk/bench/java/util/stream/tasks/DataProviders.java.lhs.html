24 package org.openjdk.bench.java.util.stream.tasks;
25 
26 import java.io.BufferedReader;
27 import java.io.IOException;
28 import java.io.InputStreamReader;
29 import java.util.stream.Stream;
30 
31 public class DataProviders {
32     public static Stream<String> dictionary() throws IOException {
33         BufferedReader r = new BufferedReader(new InputStreamReader(DataProviders.class.getResourceAsStream("cmudict-0.7b.txt")));
34         // Strip out the copyright notice and special chars
35         return r.lines().filter(w -> w.charAt(0) >= 'A' && w.charAt(0) <= 'Z').map(w -> w.substring(0, w.indexOf(" "))).onClose(() -> {
36             try {
37                 r.close();
38             } catch (IOException e) {
39                 // swallow
40             }
41         });
42     }
43 }
