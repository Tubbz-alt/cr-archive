 23 package org.openjdk.bench.java.lang.invoke;
 24 
 25 
 26 import org.openjdk.jmh.annotations.Benchmark;
 27 import org.openjdk.jmh.annotations.BenchmarkMode;
 28 import org.openjdk.jmh.annotations.Mode;
 29 import org.openjdk.jmh.annotations.OutputTimeUnit;
 30 import org.openjdk.jmh.annotations.Scope;
 31 import org.openjdk.jmh.annotations.State;
 32 
 33 import java.lang.invoke.MethodHandle;
 34 import java.lang.invoke.MethodHandles;
 35 import java.lang.invoke.MethodType;
 36 import java.util.concurrent.TimeUnit;
 37 
 38 /**
 39  * Benchmark assesses MethodHandle.publicLookup() performance
 40  */
 41 @BenchmarkMode(Mode.AverageTime)
 42 @OutputTimeUnit(TimeUnit.NANOSECONDS)
 43 @State(Scope.Thread)
 44 public class LookupPublicFind {
 45 
 46     /*
 47         Implementation notes:
 48             - findSpecial is not tested, unable to do this with public lookup object
 49      */
 50 
 51     @Benchmark
 52     public MethodHandle testConstructor() throws Exception {
 53         return MethodHandles.publicLookup().findConstructor(Victim.class, MethodType.methodType(void.class));
 54     }
 55 
 56     @Benchmark
 57     public MethodHandle testGetter() throws Exception {
 58         return MethodHandles.publicLookup().findGetter(Victim.class, "field", int.class);
 59     }
 60 
 61     @Benchmark
 62     public MethodHandle testSetter() throws Exception {
 63         return MethodHandles.publicLookup().findSetter(Victim.class, "field", int.class);
 64     }
 65 
 66     @Benchmark
 67     public MethodHandle testStatic() throws Exception {
 68         return MethodHandles.publicLookup().findStatic(Victim.class, "staticWork", MethodType.methodType(int.class));
 69     }
 70 
 71     @Benchmark
 72     public MethodHandle testStaticGetter() throws Exception {
 73         return MethodHandles.publicLookup().findStaticGetter(Victim.class, "staticField", int.class);
 74     }
 75 
 76     @Benchmark
 77     public MethodHandle testStaticSetter() throws Exception {
 78         return MethodHandles.publicLookup().findStaticSetter(Victim.class, "staticField", int.class);
 79     }
 80 
 81     @Benchmark
 82     public MethodHandle testVirtual() throws Exception {
 83         return MethodHandles.publicLookup().findVirtual(Victim.class, "virtualWork", MethodType.methodType(int.class));
 84     }
 85 
 86     public static class Victim {
 87 
 88         public static int staticField;
 89         public int field;
 90 
 91         public Victim() {
 92             // do nothing
 93         }
 94 
 95         public int virtualWork() {
 96             return 1;
 97         }
 98 
 99         public static int staticWork() {
100             return 1;
101         }
102 
103     }
104 
105 
106 }
