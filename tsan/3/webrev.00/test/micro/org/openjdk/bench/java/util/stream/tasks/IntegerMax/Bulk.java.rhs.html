 23 package org.openjdk.bench.java.util.stream.tasks.IntegerMax;

 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Level;
 28 import org.openjdk.jmh.annotations.Mode;
 29 import org.openjdk.jmh.annotations.OutputTimeUnit;
 30 import org.openjdk.jmh.annotations.Scope;
 31 import org.openjdk.jmh.annotations.Setup;
 32 import org.openjdk.jmh.annotations.State;

 34 import java.util.Arrays;
 35 import java.util.concurrent.RecursiveTask;
 36 import java.util.concurrent.TimeUnit;
 37 import java.util.function.BinaryOperator;
 38 
 39 /**
 40  * Bulk scenario: search for max value in array.
 41  *
 42  * This test covers bulk infrastructure only. Refer to other tests for lambda-specific cases.
 43  */
 44 @BenchmarkMode(Mode.Throughput)
 45 @OutputTimeUnit(TimeUnit.SECONDS)
 46 @State(Scope.Benchmark)
 47 public class Bulk {
 48 
 49     private IntegerMaxProblem problem;
 50 
 51     @Setup(Level.Trial)
 52     public void populateData(){
 53         problem = new IntegerMaxProblem();
 54     }
 55 
 56     @Benchmark
 57     public int hm_seq() {
 58         int max = Integer.MIN_VALUE;
 59         for (int v: problem.get()) {
 60             if (v > max) {
 61                 max = v;
 62             }
 63         }
 64         return max;
 65     }
 66 
 67     @Benchmark
 68     public int hm_par() {
 69         return new MaxIntTask(problem.get(), 0, problem.get().length).invoke();
 70     }
 71 
 72     @Benchmark
 73     public int bulk_seq_inner() {
 74         return Arrays.stream(problem.get()).reduce(Integer.MIN_VALUE, new BinaryOperator<Integer>() {
 75             @Override
 76             public Integer apply(Integer l, Integer r) {
 77                 return l > r ? l : r;
 78             }
 79         });
 80     }
 81 
 82     @Benchmark
 83     public int bulk_par_inner() {
 84         return Arrays.stream(problem.get()).parallel().reduce(Integer.MIN_VALUE, new BinaryOperator<Integer>() {
 85             @Override
 86             public Integer apply(Integer l, Integer r) {
 87                 return l > r ? l : r;
 88             }
 89         });
 90     }
 91 
 92     static class MaxIntTask extends RecursiveTask<Integer> {
 93         private static final int FORK_LIMIT = 1000;
 94         final Integer[] data;
 95         final int start, end;
 96 
 97         MaxIntTask(Integer[] data, int start, int end) {
 98             this.data = data;
 99             this.start = start;
100             this.end = end;
101         }
102 
103         @Override
104         protected Integer compute() {
105             int size = end - start;
106             if (size > FORK_LIMIT) {
107                 int mid = start + size / 2;
108                 MaxIntTask t1 = new MaxIntTask(data, start, mid);
109                 MaxIntTask t2 = new MaxIntTask(data, mid, end);
110                 t1.fork();
111                 Integer v2 = t2.invoke();
112                 Integer v1 = t1.join();
113                 return Math.max(v1, v2);
114             } else {
115                 int max = Integer.MIN_VALUE;
116                 for (int i = start; i < end; i++) {
117                     if (data[i] > max) {
118                         max = data[i];
119                     }
120                 }
121                 return max;
122             }
123         }
124     }
125 
126 }
