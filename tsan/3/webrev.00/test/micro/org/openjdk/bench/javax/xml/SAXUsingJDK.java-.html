23 package org.openjdk.bench.javax.xml;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.xml.sax.InputSource;
27 import org.xml.sax.XMLReader;
28 
29 import javax.xml.parsers.SAXParser;
30 import javax.xml.parsers.SAXParserFactory;
31 import java.io.ByteArrayInputStream;
32 
33 /**
34  * Micro testing SAXParser performance using the JDK classes
35  */
36 public class SAXUsingJDK extends AbstractXMLMicro {
37 
38     @Benchmark
39     public XMLReader testParse() throws Exception {
40         SAXParserFactory spf = SAXParserFactory.newInstance();
41         byte[] bytes = getFileBytesFromResource(doc);
42         spf.setValidating(false);
43         SAXParser parser = spf.newSAXParser();
44         XMLReader reader = parser.getXMLReader();
45         InputSource source = new InputSource();
46         ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
47 
48         source.setByteStream(bais);
49         reader.parse(source);
50         return reader;
51     }
52 
53 }
