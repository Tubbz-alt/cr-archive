 23 package org.openjdk.bench.java.util.stream.ops.value;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Mode;
 28 import org.openjdk.jmh.annotations.OutputTimeUnit;
 29 import org.openjdk.jmh.annotations.Param;
 30 import org.openjdk.jmh.annotations.Scope;
 31 import org.openjdk.jmh.annotations.Setup;
 32 import org.openjdk.jmh.annotations.State;
 33 
 34 import java.util.concurrent.TimeUnit;
 35 import java.util.function.LongBinaryOperator;
 36 import java.util.stream.LongStream;
 37 
 38 /**
 39  * Benchmark for reduce() operation.
 40  */
 41 @BenchmarkMode(Mode.Throughput)
 42 @OutputTimeUnit(TimeUnit.SECONDS)
 43 @State(Scope.Thread)
 44 public class Reduce {
 45 
 46     /**
 47      * Implementation notes:
 48      *   - parallel version requires thread-safe sink, we use the same for sequential version for better comparison
 49      *   - operations are explicit inner classes to untangle unwanted lambda effects
 50      */
 51     @Param("100000")
 52     private int size;
 53 
 54     private LongBinaryOperator op1, op2, op3;
 55 
 56     @Setup
 57     public void setup() {
 58         op1 = new LongBinaryOperator() {
 59             @Override
 60             public long applyAsLong(long l, long r) {
 61                 return (l > r) ? l : r;
 62             }
 63         };
 64         op2 = new LongBinaryOperator() {
 65             @Override
 66             public long applyAsLong(long l, long r) {
 67                 return (l > r) ? l : r;
 68             }
 69         };
 70         op3 = new LongBinaryOperator() {
 71             @Override
 72             public long applyAsLong(long l, long r) {
 73                 return (l > r) ? l : r;
 74             }
 75         };
 76     }
 77 
 78     @Benchmark
 79     public long seq_invoke() {
 80         return LongStream.range(0, size).reduce(op1).getAsLong();
 81     }
 82 
 83     @Benchmark
 84     public long seq_chain_111() {
 85         long l = 0;
 86         l += LongStream.range(0, size).reduce(op1).getAsLong();
 87         l += LongStream.range(0, size).reduce(op1).getAsLong();
 88         l += LongStream.range(0, size).reduce(op1).getAsLong();
 89         return l;
 90     }
 91 
 92     @Benchmark
 93     public long seq_chain_123() {
 94         long l = 0;
 95         l += LongStream.range(0, size).reduce(op1).getAsLong();
 96         l += LongStream.range(0, size).reduce(op2).getAsLong();
 97         l += LongStream.range(0, size).reduce(op3).getAsLong();
 98         return l;
 99     }
100 
101     @Benchmark
102     public long par_invoke() {
103         return LongStream.range(0, size).parallel().reduce(op1).getAsLong();
104     }
105 
106     @Benchmark
107     public long par_chain_111() {
108         long l = 0;
109         l += LongStream.range(0, size).parallel().reduce(op1).getAsLong();
110         l += LongStream.range(0, size).parallel().reduce(op1).getAsLong();
111         l += LongStream.range(0, size).parallel().reduce(op1).getAsLong();
112         return l;
113     }
114 
115     @Benchmark
116     public long par_chain_123() {
117         long l = 0;
118         l += LongStream.range(0, size).parallel().reduce(op1).getAsLong();
119         l += LongStream.range(0, size).parallel().reduce(op2).getAsLong();
120         l += LongStream.range(0, size).parallel().reduce(op3).getAsLong();
121         return l;
122     }
123 
124 }
