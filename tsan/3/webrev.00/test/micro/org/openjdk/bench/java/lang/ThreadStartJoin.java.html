23 package org.openjdk.bench.java.lang;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Scope;
30 import org.openjdk.jmh.annotations.State;
31 import org.openjdk.jmh.annotations.TearDown;
32 
33 import java.util.concurrent.TimeUnit;
34 
35 /**
36  * Test to launch and gather threads. Measure time for different parts.
37  */
38 @BenchmarkMode(Mode.AverageTime)
39 @OutputTimeUnit(TimeUnit.MICROSECONDS)
40 @State(Scope.Benchmark)
41 public class ThreadStartJoin {
42 
43     private static final int THREADS = Runtime.getRuntime().availableProcessors();
44 
45     private Thread[] ts;
46 
47     @Benchmark
48     public void test() throws InterruptedException {
49         ts = new TestThread[THREADS];
50 
51         for (int i = 0; i < THREADS; i++) {
52             ts[i] = new TestThread();
53         }
54 
55         for (int i = 0; i < THREADS; i++) {
56             ts[i].start();
57         }
58 
59         for (Thread mythread : ts) {
60             mythread.join();
61 
62             if (mythread.isAlive()) {
63                 throw new IllegalStateException("Couldn't join in time in LPHhello.");
64             }
65         }
66     }
67 
68     @TearDown
69     public final void cleanup() {
70         if (ts != null) {
71             for (Thread t : ts) {
72                 if (t != null) {
73                     t.interrupt();
74                 }
75             }
76             ts = null;
77         }
78     }
79 
80     static final class TestThread extends Thread {
81         private static int num = 0;
82 
83         public TestThread() {
84             super(TestThread.name());
85         }
86 
87         private synchronized static String name() {
88             return "TestThread-" + num++;
89         }
90 
91         public void run() {
92             // do nothing
93         }
94 
95     }
96 
97 }
