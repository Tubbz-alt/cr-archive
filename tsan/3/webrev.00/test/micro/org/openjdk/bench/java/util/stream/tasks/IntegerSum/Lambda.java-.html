23 package org.openjdk.bench.java.util.stream.tasks.IntegerSum;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Level;
28 import org.openjdk.jmh.annotations.Mode;
29 import org.openjdk.jmh.annotations.OutputTimeUnit;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.Arrays;
35 import java.util.concurrent.TimeUnit;
36 
37 /**
38  * Bulk scenario: compute the sum of Integer array.
39  *
40  * This test covers lambda-specific solutions for the problem.
41  */
42 @BenchmarkMode(Mode.Throughput)
43 @OutputTimeUnit(TimeUnit.SECONDS)
44 @State(Scope.Benchmark)
45 public class Lambda {
46 
47     private IntegerSumProblem problem;
48 
49     @Setup(Level.Trial)
50     public void populateData(){
51         problem = new IntegerSumProblem();
52     }
53 
54     public static Integer sum(Integer a, Integer b) {
55         return a + b;
56     }
57 
58     @Benchmark
59     public int bulk_seq_lambda() {
60         return Arrays.stream(problem.get()).reduce(0, (l, r) -> l + r);
61     }
62 
63     @Benchmark
64     public int bulk_par_lambda() {
65         return Arrays.stream(problem.get()).parallel().reduce(0, (l, r) -> l + r);
66     }
67 
68     @Benchmark
69     public int bulk_seq_mref() {
70         return Arrays.stream(problem.get()).reduce(0, Lambda::sum);
71     }
72 
73     @Benchmark
74     public int bulk_par_mref() {
75         return Arrays.stream(problem.get()).parallel().reduce(0, Lambda::sum);
76     }
77 
78     @Benchmark
79     public int bulk_seq_Integer_mref() {
80         return Arrays.stream(problem.get()).reduce(0, Integer::sum);
81     }
82 
83     @Benchmark
84     public int bulk_par_Integer_mref() {
85         return Arrays.stream(problem.get()).parallel().reduce(0, Integer::sum);
86     }
87 }
