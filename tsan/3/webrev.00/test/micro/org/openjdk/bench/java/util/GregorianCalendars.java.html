23 package org.openjdk.bench.java.util;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Scope;
30 import org.openjdk.jmh.annotations.Setup;
31 import org.openjdk.jmh.annotations.State;
32 
33 import java.util.Calendar;
34 import java.util.GregorianCalendar;
35 import java.util.concurrent.TimeUnit;
36 
37 @BenchmarkMode(Mode.AverageTime)
38 @OutputTimeUnit(TimeUnit.NANOSECONDS)
39 @State(Scope.Thread)
40 public class GregorianCalendars {
41 
42     private Calendar calendar;
43     private long millis;
44 
45     @Setup
46     public void setup() {
47         calendar = new GregorianCalendar();
48         millis = 4711L;
49     }
50 
51     @Benchmark
52     public GregorianCalendar testConstructor() {
53         return new GregorianCalendar();
54     }
55 
56     @Benchmark
57     public void testUpdate() {
58         calendar.setTimeInMillis(millis);
59         millis += 1L;
60     }
61 }
