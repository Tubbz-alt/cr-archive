23 package org.openjdk.bench.java.util.stream;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Param;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.concurrent.TimeUnit;
35 import java.util.function.LongPredicate;
36 import java.util.stream.LongStream;
37 
38 /**
39  * Benchmark for checking different "allMatch" schemes.
40  */
41 @BenchmarkMode(Mode.Throughput)
42 @OutputTimeUnit(TimeUnit.SECONDS)
43 @State(Scope.Thread)
44 public class AllMatcher {
45 
46     /**
47      * Implementation notes:
48      *   - operations are explicit inner classes to untangle unwanted lambda effects
49      *   - all operations have similar semantics
50      *   - these use the qualifier duality: all(P(x)) === !(exists(!P(x))
51      */
52 
53     @Param("100000")
54     private int size;
55 
56     private LongPredicate op;
57 
58     @Setup
59     public void setup() {
60         op = new LongPredicate() {
61             @Override
62             public boolean test(long x) {
63                 return true;
64             }
65         };
66     }
67 
68     @Benchmark
69     public boolean seq_anyMatch() {
70         return LongStream.range(0, size).allMatch(op);
71     }
72 
73     @Benchmark
74     public boolean seq_filter_findFirst() {
75         return !(LongStream.range(0, size).filter(op.negate()).findFirst().isPresent());
76     }
77 
78     @Benchmark
79     public boolean seq_filter_findAny() {
80         return !(LongStream.range(0, size).filter(op.negate()).findAny().isPresent());
81     }
82 
83     @Benchmark
84     public boolean par_anyMatch() {
85         return LongStream.range(0, size).parallel().allMatch(op);
86     }
87 
88     @Benchmark
89     public boolean par_filter_findFirst() {
90         return !(LongStream.range(0, size).parallel().filter(op.negate()).findFirst().isPresent());
91     }
92 
93     @Benchmark
94     public boolean par_filter_findAny() {
95         return !(LongStream.range(0, size).parallel().filter(op.negate()).findAny().isPresent());
96     }
97 
98 }
