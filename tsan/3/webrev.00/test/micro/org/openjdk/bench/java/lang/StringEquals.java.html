23 package org.openjdk.bench.java.lang;
24 
25 import org.openjdk.jmh.annotations.*;
26 import java.util.concurrent.TimeUnit;
27 
28 /*
29  * This benchmark naively explores String::equals performance
30  */
31 @BenchmarkMode(Mode.AverageTime)
32 @OutputTimeUnit(TimeUnit.NANOSECONDS)
33 @State(Scope.Benchmark)
34 public class StringEquals {
35 
36     public String test = new String("0123456789");
37     public String test2 = new String("tgntogjnrognagronagroangroarngorngaorng");
38     public String test3 = new String(test); // equal to test, but not same
39     public String test4 = new String("0123\u01FF");
40     public String test5 = new String(test4); // equal to test4, but not same
41     public String test6 = new String("0123456780");
42     public String test7 = new String("0123\u01FE");
43 
44     @Benchmark
45     public boolean different() {
46         return test.equals(test2);
47     }
48 
49     @Benchmark
50     public boolean equal() {
51         return test.equals(test3);
52     }
53 
54     @Benchmark
55     public boolean almostEqual() {
56         return test.equals(test6);
57     }
58 
59     @Benchmark
60     public boolean almostEqualUTF16() {
61         return test4.equals(test7);
62     }
63 
64     @Benchmark
65     public boolean differentCoders() {
66         return test.equals(test4);
67     }
68 
69     @Benchmark
70     public boolean equalsUTF16() {
71         return test5.equals(test4);
72     }
73 }
