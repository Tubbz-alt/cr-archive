23 package org.openjdk.bench.java.lang.invoke;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Scope;
30 import org.openjdk.jmh.annotations.Setup;
31 import org.openjdk.jmh.annotations.State;
32 
33 import java.lang.invoke.MethodType;
34 import java.util.concurrent.TimeUnit;
35 
36 /**
37  * Assesses general MethodType performance.
38  */
39 @BenchmarkMode(Mode.AverageTime)
40 @OutputTimeUnit(TimeUnit.NANOSECONDS)
41 @State(Scope.Benchmark)
42 public class MethodTypeAcquire {
43 
44     private MethodType pTypes;
45 
46     @Setup
47     public void setup() {
48         pTypes = MethodType.methodType(A.class, B.class);
49     }
50 
51     @Benchmark
52     public MethodType baselineRaw() {
53         return pTypes;
54     }
55 
56     @Benchmark
57     public MethodType testReturnVoid() {
58         return MethodType.methodType(void.class);
59     }
60 
61     @Benchmark
62     public MethodType testReturnInt() {
63         return MethodType.methodType(int.class);
64     }
65 
66     @Benchmark
67     public MethodType testReturnObject() {
68         return MethodType.methodType(Object.class);
69     }
70 
71     @Benchmark
72     public MethodType testSinglePType() {
73         return MethodType.methodType(void.class, int.class);
74     }
75 
76     @Benchmark
77     public MethodType testMultiPType() {
78         return MethodType.methodType(void.class, A.class, B.class);
79     }
80 
81     @Benchmark
82     public MethodType testMultiPType_Arg() {
83         return MethodType.methodType(void.class, pTypes);
84     }
85 
86     public static class A {}
87     public static class B {}
88 
89 }
