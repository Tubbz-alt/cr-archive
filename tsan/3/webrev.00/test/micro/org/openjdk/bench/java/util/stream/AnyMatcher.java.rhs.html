23 package org.openjdk.bench.java.util.stream;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Param;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.concurrent.TimeUnit;
35 import java.util.function.LongPredicate;
36 import java.util.stream.LongStream;
37 
38 /**
39  * Benchmark for checking different "anyMatch" schemes.
40  */
41 @BenchmarkMode(Mode.Throughput)
42 @OutputTimeUnit(TimeUnit.SECONDS)
43 @State(Scope.Thread)
44 public class AnyMatcher {
45 
46     /**
47      * Implementation notes:
48      *   - operations are explicit inner classes to untangle unwanted lambda effects
49      *   - all operations have similar semantics
50      */
51 
52     @Param("100000")
53     private int size;
54 
55     private LongPredicate op;
56 
57     @Setup
58     public void setup() {
59         op = new LongPredicate() {
60             @Override
61             public boolean test(long x) {
62                 return false;
63             }
64         };
65     }
66 
67     @Benchmark
68     public boolean seq_anyMatch() {
69         return LongStream.range(0, size).anyMatch(op);
70     }
71 
72     @Benchmark
73     public boolean seq_filter_findFirst() {
74         return LongStream.range(0, size).filter(op).findFirst().isPresent();
75     }
76 
77     @Benchmark
78     public boolean seq_filter_findAny() {
79         return LongStream.range(0, size).filter(op).findAny().isPresent();
80     }
81 
82     @Benchmark
83     public boolean par_anyMatch() {
84         return LongStream.range(0, size).parallel().anyMatch(op);
85     }
86 
87     @Benchmark
88     public boolean par_filter_findFirst() {
89         return LongStream.range(0, size).parallel().filter(op).findFirst().isPresent();
90     }
91 
92     @Benchmark
93     public boolean par_filter_findAny() {
94         return LongStream.range(0, size).parallel().filter(op).findAny().isPresent();
95     }
96 
97 }
