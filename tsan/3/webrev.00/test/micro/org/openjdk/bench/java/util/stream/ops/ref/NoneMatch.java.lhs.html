 23 package org.openjdk.bench.java.util.stream.ops.ref;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Mode;
 28 import org.openjdk.jmh.annotations.OutputTimeUnit;
 29 import org.openjdk.jmh.annotations.Param;
 30 import org.openjdk.jmh.annotations.Scope;
 31 import org.openjdk.jmh.annotations.Setup;
 32 import org.openjdk.jmh.annotations.State;
 33 
 34 import java.util.concurrent.TimeUnit;
 35 import java.util.function.Predicate;
 36 import java.util.stream.LongStream;
 37 
 38 /**
 39  * Benchmark for noneMatch() operation.
 40  */
 41 @BenchmarkMode(Mode.Throughput)
 42 @OutputTimeUnit(TimeUnit.SECONDS)
 43 @State(Scope.Thread)
 44 public class NoneMatch {
 45 
 46     /**
 47      * Implementation notes:
 48      *   - operations are explicit inner classes to untangle unwanted lambda effects
 49      *   - the predicates are always false to avert shortcurcuiting
 50      */
 51 
 52     @Param("100000")
 53     private int size;
 54 
 55     private Predicate<Long> p1, p2, p3;
 56 
 57     @Setup
 58     public void setup() {
 59         p1 = new Predicate<Long>() {
 60             @Override
 61             public boolean test(Long v) {
 62                 return false;
 63             }
 64         };
 65         p2 = new Predicate<Long>() {
 66             @Override
 67             public boolean test(Long v) {
 68                 return false;
 69             }
 70         };
 71         p3 = new Predicate<Long>() {
 72             @Override
 73             public boolean test(Long v) {
 74                 return false;
 75             }
 76         };
 77     }
 78 
 79     @Benchmark
 80     public boolean seq_invoke() {
 81         return LongStream.range(0, size).boxed().noneMatch(p1);
 82     }
 83 
 84     @Benchmark
 85     public int seq_chain111() {
 86         int s = 0;
 87         s += (LongStream.range(0, size).boxed().noneMatch(p1)) ? 1 : 0;
 88         s += (LongStream.range(0, size).boxed().noneMatch(p1)) ? 1 : 0;
 89         s += (LongStream.range(0, size).boxed().noneMatch(p1)) ? 1 : 0;
 90         return s;
 91     }
 92 
 93     @Benchmark
 94     public int seq_chain123() {
 95         int s = 0;
 96         s += (LongStream.range(0, size).boxed().noneMatch(p1)) ? 1 : 0;
 97         s += (LongStream.range(0, size).boxed().noneMatch(p2)) ? 1 : 0;
 98         s += (LongStream.range(0, size).boxed().noneMatch(p3)) ? 1 : 0;
 99         return s;
100     }
101 
102     @Benchmark
103     public boolean par_invoke() {
104         return LongStream.range(0, size).parallel().boxed().noneMatch(p1);
105     }
106 
107     @Benchmark
108     public int par_chain111() {
109         int s = 0;
110         s += (LongStream.range(0, size).parallel().boxed().noneMatch(p1)) ? 1 : 0;
111         s += (LongStream.range(0, size).parallel().boxed().noneMatch(p1)) ? 1 : 0;
112         s += (LongStream.range(0, size).parallel().boxed().noneMatch(p1)) ? 1 : 0;
113         return s;
114     }
115 
116     @Benchmark
117     public int par_chain123() {
118         int s = 0;
119         s += (LongStream.range(0, size).parallel().boxed().noneMatch(p1)) ? 1 : 0;
120         s += (LongStream.range(0, size).parallel().boxed().noneMatch(p2)) ? 1 : 0;
121         s += (LongStream.range(0, size).parallel().boxed().noneMatch(p3)) ? 1 : 0;
122         return s;
123     }
124 
125 }
