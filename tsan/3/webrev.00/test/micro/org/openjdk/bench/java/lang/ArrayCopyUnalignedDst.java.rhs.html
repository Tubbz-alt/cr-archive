23 package org.openjdk.bench.java.lang;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Param;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.concurrent.TimeUnit;
35 
36 /**
37  * Benchmark measuring Unaligned System.arraycopy.
38  */
39 @BenchmarkMode(Mode.AverageTime)
40 @OutputTimeUnit(TimeUnit.NANOSECONDS)
41 @State(Scope.Thread)
42 public class ArrayCopyUnalignedDst {
43 
44     @Param({"1", "3", "5", "10", "20", "70", "150", "300", "600", "1200"})
45     int length;
46 
47     int fromPos, toPos;
48     byte[] fromByteArr, toByteArr;
49     char[] fromCharArr, toCharArr;
50     int[] fromIntArr, toIntArr;
51     long[] fromLongArr, toLongArr;
52 
53     @Setup
54     public void setup() {
55         // Dst position Unaligned
56         fromPos = 8;
57         toPos = 9;
58 
59         fromByteArr = new byte[1210];
60         toByteArr = new byte[1210];
61         fromCharArr = new char[1210];
62         toCharArr = new char[1210];
63         fromIntArr = new int[1210];
64         toIntArr = new int[1210];
65         fromLongArr = new long[1210];
66         toLongArr = new long[1210];
67     }
68 
69     @Benchmark
70     public void testByte() {
71         System.arraycopy(fromByteArr, fromPos, toByteArr, toPos, length);
72     }
73 
74     @Benchmark
75     public void testChar() {
76         System.arraycopy(fromCharArr, fromPos, toCharArr, toPos, length);
77     }
78 
79     @Benchmark
80     public void testInt() {
81         System.arraycopy(fromIntArr, fromPos, toIntArr, toPos, length);
82     }
83 
84     @Benchmark
85     public void testLong() {
86         System.arraycopy(fromLongArr, fromPos, toLongArr, toPos, length);
87     }
88 }
