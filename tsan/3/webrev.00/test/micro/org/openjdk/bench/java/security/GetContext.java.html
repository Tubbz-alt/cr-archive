25 package org.openjdk.bench.java.security;
26 
27 import org.openjdk.jmh.annotations.Benchmark;
28 import org.openjdk.jmh.annotations.BenchmarkMode;
29 import org.openjdk.jmh.annotations.Mode;
30 import org.openjdk.jmh.annotations.OutputTimeUnit;
31 import org.openjdk.jmh.annotations.Param;
32 import org.openjdk.jmh.annotations.Scope;
33 import org.openjdk.jmh.annotations.State;
34 
35 import java.security.AccessControlContext;
36 import java.security.AccessController;
37 import java.security.PrivilegedAction;
38 import java.util.concurrent.TimeUnit;
39 
40 /**
41  * Benchmark measuring AccessController.getContext
42  */
43 @BenchmarkMode(Mode.AverageTime)
44 @OutputTimeUnit(TimeUnit.NANOSECONDS)
45 @State(Scope.Thread)
46 public abstract class GetContext {
47 
48     public static class Top extends GetContext {
49 
50         @Benchmark
51         public AccessControlContext testNonPriv() {
52             return AccessController.getContext();
53         }
54 
55         @Benchmark
56         public AccessControlContext testPriv() {
57             PrivilegedAction<AccessControlContext> pa = () -> AccessController.getContext();
58             return AccessController.doPrivileged(pa);
59         }
60     }
61 
62     public static class Deep extends GetContext {
63 
64         @Param({"2", "50"})
65         int depth;
66 
67         private AccessControlContext recurse(int depth) {
68             if (depth > 0) {
69                 return recurse(depth - 1);
70             } else {
71                 return AccessController.getContext();
72             }
73         }
74 
75         @Benchmark
76         public AccessControlContext testNonPrivRecurse() {
77             return recurse(depth);
78         }
79 
80         @Benchmark
81         public AccessControlContext testPrivInline() {
82             PrivilegedAction<AccessControlContext> pa = () -> recurse(depth);
83             return AccessController.doPrivileged(pa);
84         }
85     }
86 }
