23 package org.openjdk.bench.java.util.logging;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Scope;
30 import org.openjdk.jmh.annotations.Setup;
31 import org.openjdk.jmh.annotations.State;
32 import org.openjdk.jmh.infra.Blackhole;
33 
34 import java.util.concurrent.TimeUnit;
35 import java.util.logging.Handler;
36 import java.util.logging.Level;
37 import java.util.logging.Logger;
38 
39 @BenchmarkMode(Mode.AverageTime)
40 @OutputTimeUnit(TimeUnit.NANOSECONDS)
41 @State(Scope.Thread)
42 public class LogRecord {
43     private Logger logger;
44 
45     @Setup
46     public void setup(final Blackhole bh) {
47         try {
48             logger = Logger.getLogger("logger");
49             logger.addHandler(new Handler() {
50                 @Override
51                 public void publish(java.util.logging.LogRecord record) {
52                     bh.consume(record);
53                 }
54 
55                 @Override
56                 public void flush() {
57                     // do nothing
58                 }
59 
60                 @Override
61                 public void close() throws SecurityException {
62                     // do nothing
63                 }
64             });
65             logger.setLevel(Level.FINE);
66             logger.setUseParentHandlers(false);
67         } catch (Exception e) {
68             e.printStackTrace();
69         }
70 
71     }
72 
73     @Benchmark
74     public void testFine() {
75        logger.log(Level.FINE, "test message");
76     }
77 
78 }
