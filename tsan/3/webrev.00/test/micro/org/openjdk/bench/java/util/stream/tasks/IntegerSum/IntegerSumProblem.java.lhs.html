23 package org.openjdk.bench.java.util.stream.tasks.IntegerSum;
24 
25 import java.util.Random;
26 
27 public class IntegerSumProblem {
28 
29     private static final int DATA_SIZE = Integer.getInteger("bench.problemSize", 10*1024*1024);
30 
31     private final Integer[] data = new Integer[DATA_SIZE];
32 
33     public IntegerSumProblem() {
34         // use fixed seed to reduce run-to-run variance
35         Random rand = new Random(0x30052012);
36 
37         for (int i = 0; i < data.length; i++) {
38             data[i] = new Integer(rand.nextInt());
39         }
40     }
41 
42     public Integer[] get() {
43         return data;
44     }
45 }
