23 package org.openjdk.bench.java.lang;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.CompilerControl;
28 import org.openjdk.jmh.annotations.Mode;
29 import org.openjdk.jmh.annotations.OutputTimeUnit;
30 
31 import java.util.concurrent.TimeUnit;
32 
33 /**
34  * Tests Thread.getStackTrace()
35  */
36 @BenchmarkMode(Mode.AverageTime)
37 @OutputTimeUnit(TimeUnit.NANOSECONDS)
38 public class GetStackTrace {
39 
40     @Benchmark
41     @CompilerControl(CompilerControl.Mode.DONT_INLINE)
42     public StackTraceElement[] depth0() {
43         return Thread.currentThread().getStackTrace();
44     }
45 
46     @Benchmark
47     @CompilerControl(CompilerControl.Mode.DONT_INLINE)
48     public StackTraceElement[] depth1() {
49         return depth0();
50     }
51 
52     @Benchmark
53     @CompilerControl(CompilerControl.Mode.DONT_INLINE)
54     public StackTraceElement[] depth2() {
55         return depth1();
56     }
57 }
