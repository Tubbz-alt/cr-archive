23 package org.openjdk.bench.java.util.stream;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.BenchmarkMode;
27 import org.openjdk.jmh.annotations.Mode;
28 import org.openjdk.jmh.annotations.OutputTimeUnit;
29 import org.openjdk.jmh.annotations.Param;
30 import org.openjdk.jmh.annotations.Scope;
31 import org.openjdk.jmh.annotations.Setup;
32 import org.openjdk.jmh.annotations.State;
33 
34 import java.util.concurrent.TimeUnit;
35 import java.util.function.LongPredicate;
36 import java.util.stream.LongStream;
37 
38 /**
39  * Benchmark for checking different "noneMatch" schemes.
40  *
41  * @author Aleksey Shipilev, aleksey.shipilev@oracle.com
42  */
43 @BenchmarkMode(Mode.Throughput)
44 @OutputTimeUnit(TimeUnit.SECONDS)
45 @State(Scope.Thread)
46 public class NoneMatcher {
47 
48     /**
49      * Implementation notes:
50      *   - operations are explicit inner classes to untangle unwanted lambda effects
51      *   - all operations have similar semantics
52      *   - these use the qualifier duality:
53      *         exists(P(x)) === !all(!P(x)) =>
54      *         !exists(P(x)) === all(!P(x))
55      */
56 
57     @Param("100000")
58     private int size;
59 
60     private LongPredicate op;
61 
62     @Setup
63     public void setup() {
64         op = new LongPredicate() {
65             @Override
66             public boolean test(long v) {
67                 return false;
68             }
69         };
70     }
71 
72     @Benchmark
73     public boolean seq_anyMatch() {
74         return LongStream.range(0, size).noneMatch(op);
75     }
76 
77     @Benchmark
78     public boolean par_anyMatch() {
79         return LongStream.range(0, size).parallel().noneMatch(op);
80     }
81 
82 }
