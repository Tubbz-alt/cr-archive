65 // Macro for referencing flags.  This is public as we want the user to
66 // use this syntax to reference Google Mock flags.
67 #define GMOCK_FLAG(name) FLAGS_gmock_##name
68 
69 #if !defined(GMOCK_DECLARE_bool_)
70 
71 // Macros for declaring flags.
72 # define GMOCK_DECLARE_bool_(name) extern GTEST_API_ bool GMOCK_FLAG(name)
73 # define GMOCK_DECLARE_int32_(name) \
74     extern GTEST_API_ ::testing::internal::Int32 GMOCK_FLAG(name)
75 # define GMOCK_DECLARE_string_(name) \
76     extern GTEST_API_ ::std::string GMOCK_FLAG(name)
77 
78 // Macros for defining flags.
79 # define GMOCK_DEFINE_bool_(name, default_val, doc) \
80     GTEST_API_ bool GMOCK_FLAG(name) = (default_val)
81 # define GMOCK_DEFINE_int32_(name, default_val, doc) \
82     GTEST_API_ ::testing::internal::Int32 GMOCK_FLAG(name) = (default_val)
83 # define GMOCK_DEFINE_string_(name, default_val, doc) \
84     GTEST_API_ ::std::string GMOCK_FLAG(name) = (default_val)
85 
86 #endif  // !defined(GMOCK_DECLARE_bool_)
87 
88 #endif  // GMOCK_INCLUDE_GMOCK_INTERNAL_GMOCK_PORT_H_
