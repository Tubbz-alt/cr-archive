40 // This file implements the following syntax:
41 //
42 //   ON_CALL(mock_object.Method(...))
43 //     .With(...) ?
44 //     .WillByDefault(...);
45 //
46 // where With() is optional and WillByDefault() must appear exactly
47 // once.
48 //
49 //   EXPECT_CALL(mock_object.Method(...))
50 //     .With(...) ?
51 //     .Times(...) ?
52 //     .InSequence(...) *
53 //     .WillOnce(...) *
54 //     .WillRepeatedly(...) ?
55 //     .RetiresOnSaturation() ? ;
56 //
57 // where all clauses are optional and WillOnce() can be repeated.
58

77 // Initializes Google Mock.  This must be called before running the
78 // tests.  In particular, it parses the command line for the flags
79 // that Google Mock recognizes.  Whenever a Google Mock flag is seen,
80 // it is removed from argv, and *argc is decremented.
81 //
82 // No value is returned.  Instead, the Google Mock flag variables are
83 // updated.
84 //
85 // Since Google Test is needed for Google Mock to work, this function
86 // also initializes Google Test and parses its flags, if that hasn't
87 // been done.
88 GTEST_API_ void InitGoogleMock(int* argc, char** argv);
89 
90 // This overloaded version can be used in Windows programs compiled in
91 // UNICODE mode.
92 GTEST_API_ void InitGoogleMock(int* argc, wchar_t** argv);
