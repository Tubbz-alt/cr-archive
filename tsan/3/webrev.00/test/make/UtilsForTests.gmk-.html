28 # Assert two strings are equal
29 # 1 - Tested value
30 # 2 - Exepected value
31 # 3 - Error message
32 define assert-equals
33   ifneq ($$(strip $1),$$(strip $2))
34     $$(error $3 - Expected >$$(strip $2)< - Got >$$(strip $1)<)
35   endif
36 endef
