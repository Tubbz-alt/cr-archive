24 /*
25  * @test
26  * @bug 8041793
27  * @summary Verify that javap prints newline character after AnnotationDefault value
28  * @library /tools/lib
29  * @modules jdk.compiler/com.sun.tools.javac.api
30  *          jdk.compiler/com.sun.tools.javac.main
31  *          jdk.jdeps/com.sun.tools.javap
32  * @build toolbox.ToolBox toolbox.JavacTask toolbox.JavapTask toolbox.Assert
33  * @run main AnnotationDefaultNewlineTest
34  */
35 
36 import java.util.Collections;
37 import java.util.List;
38 
39 import toolbox.Assert;
40 import toolbox.JavacTask;
41 import toolbox.JavapTask;
42 import toolbox.Task;
43 import toolbox.ToolBox;
44 
45 public class AnnotationDefaultNewlineTest {
46 
47     private static final String TestSrc =
48             "public @interface AnnotationDefault { \n" +
49                 "public abstract int value() default 1; \n" +
50             "}";
51 
52     private static final String ExpectedSubstring =
53             "    AnnotationDefault:\n" +
54             "      default_value: I#9\n";
55 
56     public static void main(String[] args) throws Exception {
57         ToolBox tb = new ToolBox();
58         new JavacTask(tb).sources(TestSrc).run();
59 
60         List<String> res = new JavapTask(tb)
61                 .options("-v")
62                 .classes("AnnotationDefault.class")
63                 .run()
64                 .getOutputLines(Task.OutputKind.DIRECT);
65 
66         List<String> goldenList = tb.split(ExpectedSubstring, "\n");
67         Boolean found = Collections.indexOfSubList(res, goldenList) > -1;
68 
69         Assert.check(found, "expected output not found: " + ExpectedSubstring);
70     }
71 }
