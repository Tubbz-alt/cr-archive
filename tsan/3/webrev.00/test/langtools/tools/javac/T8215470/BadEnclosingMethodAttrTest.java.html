24 /*
25  * @test
26  * @bug 8215470
27  * @summary Bad EnclosingMethod attribute on classes declared in lambdas
28  * @modules jdk.compiler/com.sun.tools.javac.util
29  */
30 
31 import java.lang.reflect.ParameterizedType;
32 import java.lang.reflect.Type;
33 import com.sun.tools.javac.util.Assert;
34 
35 public class BadEnclosingMethodAttrTest<T> {
36     protected BadEnclosingMethodAttrTest() {
37         Assert.check(getClass().getEnclosingMethod().toString().equals("static void BadEnclosingMethodAttrTest.lambdaScope(java.lang.Object)"));
38         Type typeFromEnclosingMethod = getClass().getEnclosingMethod().getGenericParameterTypes()[0];
39         ParameterizedType paramType = (ParameterizedType) getClass().getGenericSuperclass();
40         Type typeFromGenericClass = paramType.getActualTypeArguments()[0];
41         Assert.check(typeFromEnclosingMethod.equals(typeFromGenericClass));
42     }
43 
44     static <X> void lambdaScope(X x) {
45         Runnable r = () -> {
46             new BadEnclosingMethodAttrTest<X>() {};
47         };
48         r.run();
49     }
50 
51     public static void main(final String[] args) {
52         lambdaScope("");
53     }
54 }
