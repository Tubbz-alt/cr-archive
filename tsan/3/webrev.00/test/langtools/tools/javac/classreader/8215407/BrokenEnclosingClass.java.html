24 /*
25  * @test
26  * @bug 8215407
27  * @summary Verify broken EnclosingMethod attribute does not break ClassReader.
28  * @library /tools/javac/lib
29  * @modules java.compiler
30  * @build JavacTestingAbstractProcessor
31  * @compile BrokenEnclosingClass.java UnrelatedClass.jcod Enclosing$1.jcod
32  * @compile -processor BrokenEnclosingClass BrokenEnclosingClass.java
33  */
34 
35 import java.util.Set;
36 
37 import javax.annotation.processing.RoundEnvironment;
38 import javax.lang.model.element.TypeElement;
39 
40 public class BrokenEnclosingClass extends JavacTestingAbstractProcessor {
41 
42     @Override
43     public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
44         if (processingEnv.getElementUtils().getTypeElement("UnrelatedClass") == null) {
45             throw new AssertionError("Cannot find UnrelatedClass.");
46         }
47         if (processingEnv.getElementUtils().getTypeElement("Enclosing$1") != null) {
48             throw new AssertionError("Enclosing$1 was found.");
49         }
50         return false;
51     }
52 
53 }
