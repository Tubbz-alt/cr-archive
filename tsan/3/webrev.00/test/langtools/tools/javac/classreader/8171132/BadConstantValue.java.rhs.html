 38 import com.sun.tools.classfile.ClassFile;
 39 import com.sun.tools.classfile.ClassWriter;
 40 import com.sun.tools.classfile.Field;
 41 import com.sun.tools.javac.api.JavacTaskImpl;
 42 import com.sun.tools.javac.code.ClassFinder.BadClassFile;
 43 import com.sun.tools.javac.code.Symtab;
 44 import com.sun.tools.javac.jvm.Target;
 45 import com.sun.tools.javac.util.Assert;
 46 import com.sun.tools.javac.util.JCDiagnostic;
 47 import com.sun.tools.javac.util.Names;
 48 import java.io.File;
 49 import java.io.FileWriter;
 50 import java.io.IOException;
 51 import java.io.PrintWriter;
 52 import java.io.StringWriter;
 53 import java.util.Arrays;
 54 import java.util.Objects;
 55 import javax.tools.JavaCompiler;
 56 import javax.tools.ToolProvider;
 57 
 58 public class BadConstantValue {
 59 
 60     static final File classesdir = new File("badconstants");
 61 
 62     public static void main(String[] args) throws Exception {
 63         // report errors for ConstantValues of the wrong type
 64         testInvalidConstantType("int");
 65         testInvalidConstantType("short");
 66         testInvalidConstantType("byte");
 67         testInvalidConstantType("char");
 68         testInvalidConstantType("boolean");
 69 
 70         // report errors for ConstantValues outside the expected range
 71         testValidConstRange("int", Integer.MAX_VALUE);
 72         testValidConstRange("int", Integer.MIN_VALUE);
 73 
 74         testValidConstRange("short", Short.MAX_VALUE);
 75         testValidConstRange("short", Short.MIN_VALUE);
 76         testInvalidConstRange("short", Short.MAX_VALUE + 1);
 77         testInvalidConstRange("short", Short.MIN_VALUE - 1);
 78 
 79         testValidConstRange("byte", Byte.MAX_VALUE);
 80         testValidConstRange("byte", Byte.MIN_VALUE);
 81         testInvalidConstRange("byte", Byte.MAX_VALUE + 1);
 82         testInvalidConstRange("byte", Byte.MIN_VALUE - 1);
 83 
 84         testValidConstRange("char", Character.MAX_VALUE);
 85         testValidConstRange("char", Character.MIN_VALUE);
 86         testInvalidConstRange("char", Character.MAX_VALUE + 1);
 87         testInvalidConstRange("char", Character.MIN_VALUE - 1);
 88 
 89         testValidConstRange("boolean", 0);
 90         testValidConstRange("boolean", 1);
 91         testInvalidConstRange("boolean", 2);
 92         testInvalidConstRange("boolean", Integer.MIN_VALUE);
 93         testInvalidConstRange("boolean", Integer.MAX_VALUE);
 94     }
 95 
 96     /**
 97      * Tests that a constant value of the given {@code type} and initialized with an out-of-range
 98      * {@code value} is rejected.
 99      */
100     private static void testInvalidConstRange(String type, int value) throws Exception {
101         createConstantWithValue(type, value);
102         BadClassFile badClassFile = loadBadClass("Lib");
103         if (badClassFile == null) {
104             throw new AssertionError("did not see expected error");
105         }
106         JCDiagnostic diagnostic = (JCDiagnostic) badClassFile.getDiagnostic().getArgs()[1];
107         assertEquals("compiler.misc.bad.constant.range", diagnostic.getCode());
108         assertEquals(3, diagnostic.getArgs().length);
109         assertEquals(value, diagnostic.getArgs()[0]);
110         assertEquals("B", diagnostic.getArgs()[1].toString());
111         assertEquals(type, String.valueOf(diagnostic.getArgs()[2]));
112     }
113 
114     /**
115      * Tests that a constant value of the given {@code type} and initialized with {@code value} is
116      * accepted.
117      */
118     private static void testValidConstRange(String type, int value) throws Exception {
119         createConstantWithValue(type, value);
120         BadClassFile badClassFile = loadBadClass("Lib");
121         if (badClassFile != null) {
122           throw new AssertionError("saw unexpected error", badClassFile);
123         }
124     }
125 
126     /**
127      * Creates a class file containing a constant field with the given type and value, which may be
128      * outside the expected range.
129      */
130     private static void createConstantWithValue(String type, int value) throws Exception {
131         // Create a class with two constants, A and B. A is of type int and has value "actual";
132         // B is of type "type" and is initialized to that type's default value.
133         File lib = writeFile(classesdir, "Lib.java", String.format(
134                 "class Lib { static final int A = %s; static final %s B = %s; }",
135                 value, type, (type.equals("boolean") ? "false" : "0")));
136         compile("-d", classesdir.getPath(), lib.getPath());
137         // Lib.class may possibly not get a newer timestamp. Make sure .java file won't get used.
138         lib.delete();
139         File libClass = new File(classesdir, "Lib.class");
140         // Rewrite the class to only have field B of type "type" and with "value" (potentially
141         // out of range).
142         swapConstantValues(libClass);
143     }
144 
145     /** Tests that a field of the given integral type with a constant string value is rejected. */
146     private static void testInvalidConstantType(String type) throws Exception {
147         // create a class file with field that has an invalid CONSTANT_String ConstantValue
148         File lib = writeFile(classesdir, "Lib.java", String.format(
149                 "class Lib { static final String A = \"hello\"; static final %s CONST = %s; }",
150                 type, type.equals("boolean") ? "false" : "0"));
151         compile("-d", classesdir.getPath(), lib.getPath());
152         // Lib.class may possibly not get a newer timestamp. Make sure .java file won't get used.
153         lib.delete();
154         File libClass = new File(classesdir, "Lib.class");
155         swapConstantValues(libClass);
156 
157         BadClassFile badClassFile = loadBadClass("Lib");
158 
159         JCDiagnostic diagnostic = (JCDiagnostic) badClassFile.getDiagnostic().getArgs()[1];
160         assertEquals("compiler.misc.bad.constant.value", diagnostic.getCode());
161         assertEquals(3, diagnostic.getArgs().length);
162         assertEquals("hello", diagnostic.getArgs()[0]);
163         assertEquals("CONST", diagnostic.getArgs()[1].toString());
164         assertEquals("Integer", diagnostic.getArgs()[2]);
165     }
166 
167     private static BadClassFile loadBadClass(String className) {
168         // load the class, and save the thrown BadClassFile exception
169         JavaCompiler c = ToolProvider.getSystemJavaCompiler();
170         JavacTaskImpl task = (JavacTaskImpl) c.getTask(null, null, null,
171                 Arrays.asList("-classpath", classesdir.getPath()), null, null);
172         Names names = Names.instance(task.getContext());
173         Symtab syms = Symtab.instance(task.getContext());
174         task.ensureEntered();
175         try {
176             syms.enterClass(syms.unnamedModule, names.fromString(className)).complete();
177         } catch (BadClassFile e) {
178             return e;
179         }
180         return null;
181     }
182 
183     /**
184      * Given a class file with two constant fields A and B, replaces both with a single field with
185      * B's type and A's ConstantValue attribute.
186      */
187     private static void swapConstantValues(File file) throws Exception {
188         ClassFile cf = ClassFile.read(file);
189         Field a = cf.fields[0];
190         Field b = cf.fields[1];
191         Field[] fields = {
192             new Field(b.access_flags, b.name_index, b.descriptor, a.attributes),
193         };
194         cf = new ClassFile(cf.magic, Target.JDK1_7.minorVersion, Target.JDK1_7.majorVersion,
195                 cf.constant_pool, cf.access_flags, cf.this_class, cf.super_class, cf.interfaces,
196                 fields, cf.methods, cf.attributes);
197         new ClassWriter().write(cf, file);
198     }
199 
200     static String compile(String... args) throws Exception {
201         System.err.println("compile: " + Arrays.asList(args));
202         StringWriter sw = new StringWriter();
203         PrintWriter pw = new PrintWriter(sw);
204         int rc = com.sun.tools.javac.Main.compile(args, pw);
205         pw.close();
206         String out = sw.toString();
207         if (out.length() > 0) {
208             System.err.println(out);
209         }
210         if (rc != 0) {
211             throw new AssertionError("compilation failed, rc=" + rc);
212         }
213         return out;
214     }
215 
216     static File writeFile(File dir, String path, String body) throws IOException {
217         File f = new File(dir, path);
218         f.getParentFile().mkdirs();
219         FileWriter out = new FileWriter(f);
220         out.write(body);
221         out.close();
222         return f;
223     }
224 
225     static void assertEquals(Object expected, Object actual) {
226         Assert.check(Objects.equals(expected, actual),
227                 String.format("expected: %s, but was: %s", expected, actual));
228     }
