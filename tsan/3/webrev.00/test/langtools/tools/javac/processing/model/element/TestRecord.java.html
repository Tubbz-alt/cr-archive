121             """
122             import java.io.Serializable;

133     @Test
134     public void testAnnoProcessing(Path base) throws Exception {
135         Path src = base.resolve("src");
136         Path r = src.resolve("Records");
137 
138         Path classes = base.resolve("classes");
139 
140         Files.createDirectories(classes);
141 
142         tb.writeJavaFiles(r, SOURCE);
143 
144         for (Mode mode : new Mode[] {Mode.API}) {
145             new JavacTask(tb, mode)
146                     .options("-nowarn",
147                         "-processor", Processor.class.getName(),
148                         "--enable-preview",
149                         "-source", Integer.toString(Runtime.version().feature()))
150                     .files(findJavaFiles(src))
151                     .outdir(classes)
152                     .run()
153                     .writeAll()
154                     .getOutputLines(Task.OutputKind.DIRECT);
155         }
156     }

158     public static final class Processor extends JavacTestingAbstractProcessor {
159         int recordCount = 0;
160 
161         public boolean process(Set<? extends TypeElement> annotations,
162                                RoundEnvironment roundEnv) {
163             if (!roundEnv.processingOver()) {
164                 ElementScanner scanner = new RecordScanner();
165                 for(Element rootElement : roundEnv.getRootElements()) {
166                     scanner.visit(rootElement);
167                 }
168 
169                 if (recordCount != 1)
170                     throw new RuntimeException("Bad record count " +
171                             recordCount);
172             }
173             return true;
174         }
175 
176         /**
177          * Verify that a record modeled as an element behaves as expected
178          * under 6 and latest specific visitors.
179          */
180         private static void testRecord(Element element, Elements elements) {
181             ElementVisitor visitor6 = new ElementKindVisitor6<Void, Void>() {};
182 
183             try {
184                 visitor6.visit(element);
185                 throw new RuntimeException("Expected UnknownElementException not thrown.");
186             } catch (UnknownElementException uee) {
187                 ; // Expected.
188             }
189 
190             ElementKindVisitor visitorLatest =
191                     new ElementKindVisitor<Object, Void>() {
192                         @Override
193                         public Object visitTypeAsRecord(TypeElement e,
194                                                         Void p) {
195                             System.out.println("printing record " + e);
196                             List<? extends Element> enclosedElements = e.getEnclosedElements();
197                             for (Element elem : enclosedElements) {
198                                 if (elem.getKind() == ElementKind.RECORD_COMPONENT)
199                                     continue; // "marathonTime" as a record component is Origin.EXPLICIT
200                                 System.out.println("name " + elem.getSimpleName());
201                                 System.out.println("origin " + elements.getOrigin(elem));
202                                 String simpleName = elem.getSimpleName().toString();
203                                 switch (simpleName) {
204                                     case "<init>":
205                                     case "readResolve":
206                                         if (elements.getOrigin(elem) != Elements.Origin.MANDATED) {
207                                             throw new RuntimeException("MANDATED origin expected for " + simpleName);
208                                         }
209                                         break;
210                                     default:
211                                         break;
212                                 }
213                             }
214                             return e; // a non-null value
215                         }
216                     };
217 
218             if (visitorLatest.visit(element) == null) {
219                 throw new RuntimeException("Null result of record visitation.");
220             }
221         }
222 
223         class RecordScanner extends ElementScanner<Void, Void> {
224 
225             public RecordScanner() {
226                 super();
227             }
228 
229             @Override
230             public Void visitType(TypeElement element, Void p) {
231                 System.out.println("Name: " + element.getSimpleName() +
232                         "\tKind: " + element.getKind());
233                 if (element.getKind() == ElementKind.RECORD) {
234                     testRecord(element, elements);
235                     recordCount++;
236                 }
237                 return super.visitType(element, p);
238             }
239         }
