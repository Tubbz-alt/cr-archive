 24 /*
 25  * @test
 26  * @bug 8224687
 27  * @summary Test Element.asType and its overrides
 28  * @library /tools/javac/lib
 29  * @build   JavacTestingAbstractProcessor TestElementAsType
 30  * @compile -processor TestElementAsType -proc:only TestElementAsType.java
 31  */
 32 
 33 import java.util.*;
 34 import javax.annotation.processing.*;
 35 import javax.lang.model.element.*;
 36 import javax.lang.model.type.*;
 37 import javax.lang.model.util.*;
 38 import static javax.lang.model.util.ElementFilter.*;
 39 
 40 /*
 41  * The leaf subinterfaces of javax.lang.model.Element of interest here
 42  * are:
 43  *
 44  * ExecutableElement
 45  * ModuleElement
 46  * PackageElement
 47  * TypeElement
 48  * TypeParameterElement
 49  * VariableElement
 50  *
 51  * For each of these categories of elements, at least one mapping of
 52  * the result of its asType() method should be tested for its
 53  * TypeMirror and TypeKind. Some elements will have more than one
 54  * possible TypeMirror and/or TypeKind in its image under
 55  * asType(). For example, a VariableElement representing a
 56  * ElementKind.FIELD could be a DeclaredType, ArrayType,
 57  * PrimitiveType, etc. In contrast, a ModuleElement should always map
 58  * to the pseudo-type of a NoType with a TypeKind of MODULE.
 59  */
 60 public class TestElementAsType extends JavacTestingAbstractProcessor {
 61     public boolean process(Set<? extends TypeElement> annotations,
 62                            RoundEnvironment roundEnv) {
 63         if (!roundEnv.processingOver()) {
 64             List<Element> elts = new ArrayList<>();
 65             elts.add(Objects.requireNonNull(eltUtils.getModuleElement("java.base")));
 66             elts.add(Objects.requireNonNull(eltUtils.getPackageElement("java.lang")));
 67             elts.add(Objects.requireNonNull(eltUtils.getTypeElement("java.lang.String")));
 68 
 69             // Get elements representing a class, field, method,
 70             // constructor, and type variable from the nested class.
 71             TypeElement tmp =
 72                 Objects.requireNonNull(eltUtils.getTypeElement("TestElementAsType.NestedClass"));
 73             elts.add(tmp);
 74             elts.add(ElementFilter.fieldsIn(tmp.getEnclosedElements()).get(0));
 75             elts.add(ElementFilter.methodsIn(tmp.getEnclosedElements()).get(0));
 76             elts.add(ElementFilter.constructorsIn(tmp.getEnclosedElements()).get(0));
 77             elts.add(tmp.getTypeParameters().get(0));
 78 
 79             // For a variety of different kinds of elements, check that
 80             // the TypeKind and TypeMirror subinterface is as expected.
 81             for(Element elt : elts) {
 82                 ElementKind eltKind = elt.getKind();
 83                 Class<?> expectedTypeClass = elementKindToTypeClass.get(eltKind);
 84                 TypeKind expectedTypeKind = elementKindToTypeKind.get(eltKind);
 85 
 86                 TypeMirror typeMirror = elt.asType();
 87                 TypeKind typeKind = typeMirror.getKind();
 88 
 89                 System.out.printf("%s\t%s\t%s%n",
 90                                   typeMirror,
 91                                   typeMirror.getClass(),
 92                                   typeKind);
 93 
 94                 if (expectedTypeKind != typeKind) {
 95                     System.out.printf("TypeKind mismatch on ''%s'';%n\texpected %s but got %s%n",
 96                                       typeMirror, expectedTypeKind, typeKind);
 97                     throw new RuntimeException();
 98                 }
 99 
100                 Class<?> typeImplClass = typeMirror.getClass();
101                 if (!expectedTypeClass.isAssignableFrom(typeImplClass)) {
102                     System.out.printf("Unexpected assignability failure on ''%s'';%n" +
103                                       "expected to be able to assign%n\t''%s'' to%n\t''%s''%n",
104                                       typeMirror, typeImplClass, expectedTypeClass);
105                     throw new RuntimeException();
106                 }
107             }
108         }
109         return true;
110     }
111 
112     /*
113      * For both of the maps below, a ElementKind value is mapped to
114      * one value related to an element's asType image. In some cases,
115      * the ElementKind -> (TypeMirror type, TypeKind) mapping is
116      * always the same, such as ElementKind.PACKAGE mapping to
117      * (NoType.class, TypeKind.PACKAGE). In other cases, such as for a
118      * field, there are many possible mappings and they are not
119      * attempted to be examined exhaustively by this test.
120      */
121     private static final Map<ElementKind, Class<?>> elementKindToTypeClass =
122         Map.of(ElementKind.CLASS,          DeclaredType.class,
123                ElementKind.CONSTRUCTOR,    ExecutableType.class,
124                ElementKind.METHOD,         ExecutableType.class,
125                ElementKind.PACKAGE,        NoType.class,
126                ElementKind.MODULE,         NoType.class,
127                ElementKind.TYPE_PARAMETER, TypeVariable.class,
128                // For the field NestedClass.name that is tested, a
129                // declared type is used.
130                ElementKind.FIELD,          DeclaredType.class);
131 
132     private static final Map<ElementKind, TypeKind> elementKindToTypeKind =
133         Map.of(ElementKind.CLASS,          TypeKind.DECLARED,
134                ElementKind.CONSTRUCTOR,    TypeKind.EXECUTABLE,
135                ElementKind.METHOD,         TypeKind.EXECUTABLE,
136                ElementKind.PACKAGE,        TypeKind.PACKAGE,
137                ElementKind.MODULE,         TypeKind.MODULE,
138                ElementKind.TYPE_PARAMETER, TypeKind.TYPEVAR,
139                // For the field NestedClass.name that is tested, a
140                // declared type is used.
141                ElementKind.FIELD,          TypeKind.DECLARED);
142 
143     static class NestedClass<N>  {
144         public NestedClass() {super();}
145 
146         String name() {
147             return name;
148         }
149 
150         private static String name = "NestedClass";
151     }
152 }
