 24 /*
 25  * @test
 26  * @bug 8222369
 27  * @summary Test behavior of ExecutableElement.getReceiverType
 28  * @library /tools/javac/lib
 29  * @build   JavacTestingAbstractProcessor TestExecutableReceiverType
 30  * @compile -processor TestExecutableReceiverType -proc:only TestExecutableReceiverType.java
 31  */
 32 
 33 import java.util.Set;
 34 import java.lang.annotation.*;
 35 import javax.annotation.processing.*;
 36 import javax.lang.model.element.*;
 37 import javax.lang.model.type.*;
 38 import javax.lang.model.util.*;
 39 import static javax.tools.Diagnostic.Kind.*;
 40 
 41 /**
 42  * Verify that proper type objects are returned from ExecutableElement.getReceiverType
 43  */
 44 public class TestExecutableReceiverType extends JavacTestingAbstractProcessor {
 45     public boolean process(Set<? extends TypeElement> annotations,
 46                            RoundEnvironment roundEnv) {
 47         if (!roundEnv.processingOver()) {
 48             int count = 0;
 49             count += testType(elements.getTypeElement("MethodHost"));
 50             count += testType(elements.getTypeElement("MethodHost.Nested"));
 51 
 52             if (count == 0) {
 53                 messager.printMessage(ERROR, "No executables visited.");
 54             }
 55         }
 56         return true;
 57     }
 58 
 59     int testType(TypeElement typeElement) {
 60         int count = 0;
 61         for (ExecutableElement executable :
 62                  ElementFilter.constructorsIn(typeElement.getEnclosedElements())) {
 63             count += testExecutable(executable);
 64         }
 65 
 66         for (ExecutableElement executable :
 67                  ElementFilter.methodsIn(typeElement.getEnclosedElements())) {
 68             count += testExecutable(executable);
 69         }
 70         return count;
 71     }
 72 
 73     int testExecutable(ExecutableElement executable) {
 74         TypeKind expectedKind = executable.getAnnotation(ReceiverTypeKind.class).value();
 75         TypeKind actualKind = executable.getReceiverType().getKind();
 76 
 77         if (actualKind != expectedKind) {
 78             messager.printMessage(ERROR,
 79                                   String.format("Unexpected TypeKind on receiver of %s:" +
 80                                                 " expected %s\t got %s%n",
 81                                                 executable, expectedKind, actualKind));
 82         }
 83         return 1;
 84     }
 85 }
 86 
 87 @Retention(RetentionPolicy.RUNTIME)
 88 @interface ReceiverTypeKind {
 89     TypeKind value();
 90 }
 91 
 92 /**
 93  * Class to host various methods, etc.
 94  */
 95 class MethodHost {
 96     @ReceiverTypeKind(TypeKind.NONE)
 97     public MethodHost() {}
 98 
 99     @ReceiverTypeKind(TypeKind.NONE)
100     public static void foo() {return;}
101 
102     @ReceiverTypeKind(TypeKind.NONE)
103     public void bar() {return;}
104 
105     @ReceiverTypeKind(TypeKind.DECLARED)
106     public void quux(MethodHost this) {return;}
107 
108     private class Nested {
109         @ReceiverTypeKind(TypeKind.DECLARED)
110         public Nested(MethodHost MethodHost.this) {}
111     }
112 }
