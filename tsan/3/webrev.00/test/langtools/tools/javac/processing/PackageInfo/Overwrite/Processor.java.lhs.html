20 import static java.nio.charset.StandardCharsets.UTF_8;
21 
22 import java.io.IOError;
23 import java.io.IOException;
24 import java.io.OutputStream;
25 import java.util.Set;
26 import javax.annotation.processing.RoundEnvironment;
27 import javax.annotation.processing.SupportedAnnotationTypes;
28 import javax.lang.model.element.PackageElement;
29 import javax.lang.model.element.TypeElement;
30 
31 @SupportedAnnotationTypes("*")
32 public class Processor extends JavacTestingAbstractProcessor {
33 
34     boolean first = true;
35 
36     @Override
37     public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
38         if (first) {
39             // Annotations are present on the initial package-info loaded from the classpath.
40             PackageElement p = processingEnv.getElementUtils().getPackageElement("p");
41             if (p.getAnnotationMirrors().isEmpty()) {
42                 throw new AssertionError(
43                         "expected package annotations: " + p.getAnnotationMirrors());
44             }
45             // Overwrite the package-info with a new unannotated package-info.
46             try (OutputStream os =
47                     processingEnv
48                             .getFiler()
49                             .createSourceFile("p.package-info")
50                             .openOutputStream()) {
51                 os.write("package p;".getBytes(UTF_8));
52             } catch (IOException e) {
53                 throw new IOError(e);
54             }
55             first = false;
56         }
57         // The package-info's symbol should be reset between rounds, and when annotation
58         // processing is over the package-info should be unannotated.
59         PackageElement p = processingEnv.getElementUtils().getPackageElement("p");
60         if (roundEnv.processingOver()) {
61             if (!p.getAnnotationMirrors().isEmpty()) {
62                 throw new AssertionError(
63                         "expected no package annotations: " + p.getAnnotationMirrors());
64             }
65         }
66         return false;
67     }
68 }
