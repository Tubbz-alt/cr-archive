20 import java.util.Set;
21 import javax.annotation.processing.RoundEnvironment;
22 import javax.annotation.processing.SupportedAnnotationTypes;
23 import javax.lang.model.element.PackageElement;
24 import javax.lang.model.element.TypeElement;
25 
26 @SupportedAnnotationTypes("*")
27 public class Processor extends JavacTestingAbstractProcessor {
28 
29     @Override
30     public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
31         PackageElement p = processingEnv.getElementUtils().getPackageElement("p");
32         // Ensure that the annotations on the package-info for p are present during all annotation
33         // processing rounds.
34         if (p.getAnnotationMirrors().isEmpty()) {
35             throw new AssertionError("expected package annotations");
36         }
37         return false;
