24 package processor;
25 
26 import javax.annotation.processing.AbstractProcessor;
27 import javax.annotation.processing.SupportedAnnotationTypes;
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import javax.lang.model.element.TypeElement;
31 import java.util.Set;
32 import javax.annotation.processing.RoundEnvironment;
33 import javax.tools.Diagnostic.Kind;
34 import javax.lang.model.element.AnnotationMirror;
35 
36 @SupportedAnnotationTypes("annotation.ModuleWarn")
37 public class ModuleWarnProcessor extends AbstractProcessor {
38 
39     @Override
40     public SourceVersion getSupportedSourceVersion() {
41         return this.processingEnv.getSourceVersion().compareTo(SourceVersion.RELEASE_9) < 0 ? SourceVersion.RELEASE_9 : this.processingEnv.getSourceVersion();
42     }
43 
44     public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
45         annotations.stream().flatMap(annotation -> roundEnv.getElementsAnnotatedWith(annotation).stream()).forEach(element -> {
46             for(AnnotationMirror annotationMirror : element.getAnnotationMirrors()) {
47                 this.processingEnv.getMessager().printMessage(Kind.MANDATORY_WARNING, "Module warning", element, annotationMirror);
48             }
49         });
50         return false;
51     }
52 }
