 24 /*
 25  * @test
 26  * @bug  8235541
 27  * @summary Test that the binding variable kind is appropriately set
 28  * @library /tools/javac/lib
 29  * @modules jdk.compiler
 30  * @build   JavacTestingAbstractProcessor
 31  * @compile TestBindingVariable.java
 32  * @compile --enable-preview -source ${jdk.version} -processor TestBindingVariable -proc:only TestBindingVariableData.java
 33  */
 34 
 35 import javax.annotation.processing.*;
 36 import javax.lang.model.element.*;
 37 import javax.lang.model.util.*;
 38 import java.util.*;
 39 import com.sun.source.tree.*;
 40 import com.sun.source.util.*;
 41 
 42 /**
 43  * Using the tree API, retrieve element representations of the
 44  * binding variables of the type test pattern, and verify their kind
 45  * tags are set appropriately.
 46  */
 47 public class TestBindingVariable extends JavacTestingAbstractProcessor implements AutoCloseable {
 48     int bindingVariableCount = 0;
 49 
 50     public boolean process(Set<? extends TypeElement> annotations,
 51                           RoundEnvironment roundEnv) {
 52        if (!roundEnv.processingOver()) {
 53            Trees trees = Trees.instance(processingEnv);
 54 
 55            for(Element rootElement : roundEnv.getRootElements()) {
 56                TreePath treePath = trees.getPath(rootElement);
 57 
 58                (new BindingVariableScanner(trees)).
 59                    scan(treePath.getCompilationUnit(), null);
 60            }
 61            if (bindingVariableCount != 2)
 62                throw new RuntimeException("Bad binding variable count " +
 63                                           bindingVariableCount);
 64        }
 65        return true;
 66     }
 67 
 68     @Override
 69     public void close() {}
 70 
 71     /**
 72      * Verify that a binding variable modeled as an element behaves
 73      * as expected under 6 and latest specific visitors.
 74      */
 75     private static void testBindingVariable(Element element) {
 76         ElementVisitor visitor6 = new ElementKindVisitor6<Void, Void>() {};
 77 
 78         try {
 79             visitor6.visit(element);
 80             throw new RuntimeException("Expected UnknownElementException not thrown.");
 81         } catch (UnknownElementException uee) {
 82             ; // Expected.
 83         }
 84 
 85         ElementKindVisitor visitorLatest =
 86             new ElementKindVisitor<Object, Void>() {
 87             @Override
 88             public Object visitVariableAsBindingVariable(VariableElement e,
 89                                                          Void p) {
 90                 return e; // a non-null value
 91             }
 92         };
 93 
 94         if (visitorLatest.visit(element) == null) {
 95             throw new RuntimeException("Null result of resource variable visitation.");
 96         }
 97     }
 98 
 99     class BindingVariableScanner extends TreePathScanner<Void, Void> {
100         private Trees trees;
101 
102         public BindingVariableScanner(Trees trees) {
103             super();
104             this.trees = trees;
105         }
106         @Override
107         public Void visitBindingPattern(BindingPatternTree node, Void p) {
108             handleCurrentTreeAsBindingVar();
109             return super.visitBindingPattern(node, p);
110         }
111 
112         @Override
113         public Void visitIdentifier(IdentifierTree node, Void p) {
114             if (node.getName().contentEquals("bindingVar")) {
115                 handleCurrentTreeAsBindingVar();
116             }
117             return super.visitIdentifier(node, p);
118         }
119 
120         private void handleCurrentTreeAsBindingVar() {
121            Element element = trees.getElement(getCurrentPath());
122 
123            System.out.println("Name: " + element.getSimpleName() +
124                               "\tKind: " + element.getKind());
125            if (element.getKind() != ElementKind.BINDING_VARIABLE) {
126                throw new RuntimeException("Expected a binding variable, but got: " +
127                                           element.getKind());
128            }
129            testBindingVariable(element);
130            bindingVariableCount++;
131         }
132     }
133 }
