36 import java.util.*;
37 import javax.annotation.processing.*;
38 import javax.lang.model.SourceVersion;
39 import static javax.lang.model.SourceVersion.*;
40 import javax.lang.model.element.*;
41 import javax.lang.model.util.*;
42 import static javax.lang.model.util.ElementFilter.*;
43 import static javax.tools.Diagnostic.Kind.*;
44 import static javax.tools.StandardLocation.*;
45 
46 /**
47  * Test basic workings of Elements.getPackageOf
48  */
49 public class TestGetPackageOf extends JavacTestingAbstractProcessor {
50     /**
51      * Check expected behavior on classes and packages and other elements.
52      */
53     public boolean process(Set<? extends TypeElement> annotations,
54                            RoundEnvironment roundEnv) {
55         if (!roundEnv.processingOver()) {
56             TypeElement    charElt     = eltUtils.getTypeElement("java.lang.Character");
57             PackageElement javaLangPkg = eltUtils.getPackageElement("java.lang");
58             PackageElement unnamedPkg  = eltUtils.getPackageElement("");
59 
60             Map<Element, PackageElement> testCases =
61                 Map.of(javaLangPkg, javaLangPkg,
62                        charElt,     javaLangPkg,
63                        unnamedPkg,  unnamedPkg);
64 
65             for (var testCase : testCases.entrySet()) {
66                 checkPkg(testCase.getKey(), testCase.getValue());
67             }
68 
69             // The package of fields and methods and nested types of
70             // java.lang.Character is java.lang.
71             for (Element e : charElt.getEnclosedElements()) {
72                 checkPkg(e, javaLangPkg);
73             }
74 
75             // A module has a null package.
76             checkPkg(eltUtils.getModuleElement("java.base"), null);
77         }
78         return true;
79     }
80 
81     private void checkPkg(Element e, PackageElement expectedPkg) {
82         PackageElement actualPkg = eltUtils.getPackageOf(e);
83         if (!Objects.equals(actualPkg, expectedPkg)) {
84             throw new RuntimeException(String.format("Unexpected package ``%s''' for %s %s, expected ``%s''%n",
85                                                      actualPkg, e.getKind(), e.toString(), expectedPkg));
86         }
87     }
88 }
