165     @Test
166     public void testAnnoProcessing(Path base) throws Exception {
167         Path src = base.resolve("src");
168         Path r = src.resolve("Records");
169 
170         Path classes = base.resolve("classes");
171 
172         Files.createDirectories(classes);
173 
174         tb.writeJavaFiles(r, SOURCE);
175 
176         for (Mode mode : new Mode[] {Mode.API}) {
177             new JavacTask(tb, mode)
178                     .options("-nowarn",
179                             "-processor", Processor.class.getName(),
180                             "--enable-preview",
181                             "-source", Integer.toString(Runtime.version().feature()))
182                     .files(findJavaFiles(src))
183                     .outdir(classes)
184                     .run()
185                     .writeAll()
186                     .getOutputLines(Task.OutputKind.DIRECT);
187         }
188     }

190     @SupportedAnnotationTypes("*")
191     public static final class Processor extends JavacTestingAbstractProcessor {
192         public boolean process(Set<? extends TypeElement> tes, RoundEnvironment renv) {
193             for (TypeElement te : tes) {
194                 switch (te.toString()) {
195                     case "ParameterContainer" :
196                         checkElements(te, renv, 1, Set.of(ElementKind.PARAMETER), "Parameter");
197                         break;
198                     case "MethodContainer":
199                         checkElements(te, renv, 1, Set.of(ElementKind.METHOD), "Method");
200                         break;
201                     case "FieldContainer":
202                         checkElements(te, renv, 1, Set.of(ElementKind.FIELD), "Field");
203                         break;
204                     case "AllContainer":
205                         checkElements(te, renv, 4,
206                                 Set.of(ElementKind.FIELD,
207                                         ElementKind.METHOD,
208                                         ElementKind.PARAMETER,
209                                         ElementKind.RECORD_COMPONENT), "All");
210                         break;
211                     case "RecComponentContainer":
212                         checkElements(te, renv, 1, Set.of(ElementKind.RECORD_COMPONENT), "RecComponent");
213                         break;
214                     case "RecComponentAndFieldContainer":
215                         checkElements(te, renv, 2, Set.of(ElementKind.RECORD_COMPONENT, ElementKind.FIELD),
216                                 "RecComponentAndField");
217                         break;
218                     default:
219                         // ignore, just another annotation like Target, we don't care about
220                 }
221             }
222             return true;
223         }
224 
225         void checkElements(TypeElement te,
226                            RoundEnvironment renv,
227                            int numberOfElementsAppliedTo,
228                            Set<ElementKind> kinds,
229                            String nameOfRepeatableAnno) {
230             Set<? extends Element> annotatedElements = renv.getElementsAnnotatedWith(te);
231             Assert.check(annotatedElements.size() == numberOfElementsAppliedTo, "for type element " + te + " expected = " + numberOfElementsAppliedTo + " found = " + annotatedElements.size());
232             for (Element e : annotatedElements) {
233                 Symbol s = (Symbol) e;
234                 Assert.check(kinds.contains(s.getKind()));
235                 java.util.List<? extends AnnotationMirror> annoMirrors = e.getAnnotationMirrors();
236                 Assert.check(annoMirrors.size() == 1, "there must be only one annotation container");
237                 AnnotationMirror annotationMirror = annoMirrors.get(0);
238                 Map<? extends ExecutableElement, ? extends AnnotationValue> map = annotationMirror.getElementValues();
239                 // as we are dealing with a container that contains two annotations, its value is a list of
240                 // contained annotations
241                 List<? extends AnnotationMirror> containedAnnotations = (List<? extends AnnotationMirror>) map.values().iterator().next().getValue();
242                 Assert.check(containedAnnotations.size() == 2, "there can only be two repeated annotations");
243                 for (AnnotationMirror annoMirror : containedAnnotations) {
244                     Assert.check(annoMirror.getAnnotationType().toString().equals(nameOfRepeatableAnno),
245                             "incorrect declared type for contained annotation");
246                 }
247             }
248         }
249     }
