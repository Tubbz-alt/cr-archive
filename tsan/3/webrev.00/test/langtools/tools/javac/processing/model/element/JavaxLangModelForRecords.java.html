 81         runTests(m -> new Object[] { Paths.get(m.getName()) });

 84     Path[] findJavaFiles(Path... paths) throws IOException {
 85         return tb.findJavaFiles(paths);
 86     }
 87 
 88     void checkOutputContains(String log, String... expect) throws Exception {
 89         for (String e : expect) {
 90             if (!log.contains(e)) {
 91                 throw new Exception("expected output not found: " + e);
 92             }
 93         }
 94     }
 95 
 96     @Test
 97     public void testQualifiedClassForProcessing(Path base) throws Exception {
 98         Path src = base.resolve("src");
 99         Path r = src.resolve("R");
100 
101         Path classes = base.resolve("classes");
102 
103         Files.createDirectories(classes);
104 
105         tb.writeJavaFiles(r,
106                 "record R(int i) {}");
107 
108         List<String> expected = List.of("Note: field: i",
109                 "Note: record component: i",
110                 "Note: testQualifiedClassForProcessing" + File.separator + "src" + File.separator
111                      + "R" + File.separator + "R.java uses preview language features.",
112                 "Note: Recompile with -Xlint:preview for details.");
113 
114         for (Mode mode : new Mode[] {Mode.API}) {
115             List<String> log = new JavacTask(tb, mode)
116                     .options("-processor", QualifiedClassForProcessing.class.getName(),
117                             "--enable-preview",
118                             "-source", Integer.toString(Runtime.version().feature()))
119                     .files(findJavaFiles(src))
120                     .outdir(classes)
121                     .run()
122                     .writeAll()
123                     .getOutputLines(Task.OutputKind.DIRECT);
124 
125             if (!expected.equals(log))
126                 throw new AssertionError("Unexpected output: " + log);
127         }
128     }
129 
130     @SupportedAnnotationTypes("*")
131     public static final class QualifiedClassForProcessing extends JavacTestingAbstractProcessor {
132 
133         @Override
134         public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
135             if (processingEnv.getElementUtils().getTypeElement("R") == null) {
136                 throw new AssertionError("No R record found.");
137             }
138 
139             Messager messager = processingEnv.getMessager();
140 
141             for (TypeElement clazz : ElementFilter.typesIn(roundEnv.getRootElements())) {
142                 for (VariableElement field : ElementFilter.fieldsIn(clazz.getEnclosedElements())) {
143                     messager.printMessage(Kind.NOTE, "field: " + field.getSimpleName());
144                 }
145                 for (RecordComponentElement rc : ElementFilter.recordComponentsIn(clazz.getEnclosedElements())) {
146                     messager.printMessage(Kind.NOTE, "record component: " + rc.getSimpleName());
147                 }
148             }
149 
150             return false;
151         }
152 
153         @Override
154         public SourceVersion getSupportedSourceVersion() {
155             return SourceVersion.latest();
156         }
157     }
