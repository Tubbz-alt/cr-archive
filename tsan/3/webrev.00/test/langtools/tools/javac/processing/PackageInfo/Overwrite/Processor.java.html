24 import static java.nio.charset.StandardCharsets.UTF_8;
25 
26 import java.io.IOError;
27 import java.io.IOException;
28 import java.io.OutputStream;
29 import java.util.Set;
30 import javax.annotation.processing.RoundEnvironment;
31 import javax.annotation.processing.SupportedAnnotationTypes;
32 import javax.lang.model.element.PackageElement;
33 import javax.lang.model.element.TypeElement;
34 
35 @SupportedAnnotationTypes("*")
36 public class Processor extends JavacTestingAbstractProcessor {
37 
38     boolean first = true;
39 
40     @Override
41     public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
42         if (first) {
43             // Annotations are present on the initial package-info loaded from the classpath.
44             PackageElement p = processingEnv.getElementUtils().getPackageElement("p");
45             if (p.getAnnotationMirrors().isEmpty()) {
46                 throw new AssertionError(
47                         "expected package annotations: " + p.getAnnotationMirrors());
48             }
49             // Overwrite the package-info with a new unannotated package-info.
50             try (OutputStream os =
51                     processingEnv
52                             .getFiler()
53                             .createSourceFile("p.package-info")
54                             .openOutputStream()) {
55                 os.write("package p;".getBytes(UTF_8));
56             } catch (IOException e) {
57                 throw new IOError(e);
58             }
59             first = false;
60         }
61         // The package-info's symbol should be reset between rounds, and when annotation
62         // processing is over the package-info should be unannotated.
63         PackageElement p = processingEnv.getElementUtils().getPackageElement("p");
64         if (roundEnv.processingOver()) {
65             if (!p.getAnnotationMirrors().isEmpty()) {
66                 throw new AssertionError(
67                         "expected no package annotations: " + p.getAnnotationMirrors());
68             }
69         }
70         return false;
71     }
72 }
