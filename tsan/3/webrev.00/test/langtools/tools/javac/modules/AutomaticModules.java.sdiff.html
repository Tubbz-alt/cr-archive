821     @Test
822     public void testAutomaticModulePatchingAndAllModulePath(Path base) throws Exception {
823         Path modulePath = base.resolve("module-path");
824 
825         Files.createDirectories(modulePath);
826 
827         Path libaSrc = base.resolve("libaSrc");
828         tb.writeJavaFiles(libaSrc,
829                           "module liba { exports api1; }",
830                           "package api1; public class Api1 {}");
831         Path libaClasses = modulePath.resolve("liba");
832         tb.createDirectories(libaClasses);
833 
834         new JavacTask(tb)
835             .outdir(libaClasses)
836             .files(findJavaFiles(libaSrc))
837             .run()
838             .writeAll();
839 
840         Path libbSrc = base.resolve("libbSrc");
841         tb.writeJavaFiles(libbSrc,
842                           "module libb { exports api2; }",
843                           "package api2; public class Api2 {}");
844         Path libbClasses = modulePath.resolve("libb");
845         tb.createDirectories(libbClasses);
846 
847         new JavacTask(tb)
848             .outdir(libbClasses)
849             .files(findJavaFiles(libbSrc))
850             .run()
851             .writeAll();
852 
853         Path automaticSrc = base.resolve("automaticSrc");
854         tb.writeJavaFiles(automaticSrc, "package aut; public class Aut1 { api1.Api1 a1; }");
855         Path automaticClasses = base.resolve("automaticClasses");
856         tb.createDirectories(automaticClasses);
857 
858         new JavacTask(tb)
859             .outdir(automaticClasses)
860             .options("--add-modules", "liba",
861                      "--module-path", modulePath.toString())
862             .files(findJavaFiles(automaticSrc))
863             .run()
864             .writeAll()
865             .getOutput(Task.OutputKind.DIRECT);
866 
867         Path automaticJar = modulePath.resolve("automatic-1.0.jar");
868 
869         try (ZipOutputStream out = new ZipOutputStream(Files.newOutputStream(automaticJar))) {
870             out.putNextEntry(new ZipEntry("aut/Aut1.class"));
871             Files.copy(automaticClasses.resolve("aut").resolve("Aut1.class"), out);
872         }
873 
874         Path src = base.resolve("src");
875 
876         tb.writeJavaFiles(src,
877                           "package aut; public class Aut2 { api2.Api2 a2; aut.Aut1 aut1;}");
878 
879         Path classes = base.resolve("classes");
880 
881         Files.createDirectories(classes);
882 
883         new JavacTask(tb)
884                 .options("--module-path", modulePath.toString(),
885                          "--patch-module", "automatic=" + src.toString(),
886                          "--add-modules", "ALL-MODULE-PATH")
887                 .outdir(classes)
888                 .files(findJavaFiles(src))
889                 .run(Task.Expect.SUCCESS)
890                 .writeAll()
891                 .getOutputLines(Task.OutputKind.DIRECT);
892 
893         new JavacTask(tb)
894                 .options("--module-path", modulePath.toString(),
895                          "--patch-module", "automatic=" + src.toString(),
896                          "--module-source-path", "dummy",
897                          "--add-modules", "ALL-MODULE-PATH")
898                 .outdir(classes)
899                 .files(findJavaFiles(src))
900                 .run(Task.Expect.SUCCESS)
901                 .writeAll()
902                 .getOutputLines(Task.OutputKind.DIRECT);
903     }
