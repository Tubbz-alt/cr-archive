14         (o instanceof String ? o.hashCode() : o.wait()).toString();

16         System.out.println(switch (s) {case "" -> o.hashCode(); default -> o.wait();});

18         (switch (s) {case "" -> o.hashCode(); default -> o.wait();}).toString();
