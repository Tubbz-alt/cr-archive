24 /*
25  * @test
26  * @bug 8206986
27  * @summary Check switch expressions
28  * @compile --enable-preview -source ${jdk.version} ExpressionSwitchCodeFromJLS.java
29  * @run main/othervm --enable-preview ExpressionSwitchCodeFromJLS
30  */
31 
32 public class ExpressionSwitchCodeFromJLS {
33     static void howMany(int k) {
34         switch (k) {
35             case 1: System.out.print("one ");
36             case 2: System.out.print("too ");
37             case 3: System.out.println("many");
38         }
39     }
40     static void howManyRule(int k) {
41         switch (k) {
42             case 1 -> System.out.println("one");
43             case 2 -> System.out.println("two");
44             case 3 -> System.out.println("many");
45         }
46     }
47     static void howManyGroup(int k) {
48         switch (k) {
49             case 1: System.out.println("one");
50                 break;  // exit the switch
51             case 2: System.out.println("two");
52                 break;  // exit the switch
53             case 3: System.out.println("many");
54                 break;  // not needed, but good style
55         }
56     }
57     public static void main(String[] args) {
58         howMany(3);
59         howMany(2);
60         howMany(1);
61         howManyRule(1);
62         howManyRule(2);
63         howManyRule(3);
64         howManyGroup(1);
65         howManyGroup(2);
66         howManyGroup(3);
67     }
68 }
