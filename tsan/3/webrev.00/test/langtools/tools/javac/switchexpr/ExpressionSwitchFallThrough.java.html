24 /*
25  * @test
26  * @bug 8206986
27  * @summary Check fall through in switch expressions.
28  * @compile ExpressionSwitchFallThrough.java
29  * @run main ExpressionSwitchFallThrough
30  */
31 
32 import java.util.Objects;
33 import java.util.function.Function;
34 
35 public class ExpressionSwitchFallThrough {
36     public static void main(String... args) {
37         new ExpressionSwitchFallThrough().run();
38     }
39 
40     private void run() {
41         runTest(this::expression1);
42         runTest(this::expression2);
43     }
44 
45     private void runTest(Function<T, String> print) {
46         check(T.A,  print, "ab");
47         check(T.B,  print, "b");
48         check(T.C,  print, "");
49     }
50 
51     private String expression1(T t) {
52         String help = "";
53         return switch (t) {
54             case A: help = "a";
55             case B: help += "b";
56             default: yield help;
57         };
58     }
59 
60     private String expression2(T t) {
61         String help = "";
62         return switch (t) {
63             case A: help = "a";
64             case B: help += "b";
65             default: yield help;
66         };
67     }
68 
69     private void check(T t, Function<T, String> print, String expected) {
70         String result = print.apply(t);
71         if (!Objects.equals(result, expected)) {
72             throw new AssertionError("Unexpected result: " + result);
73         }
74     }
75 
76     enum T {
77         A, B, C;
78     }
79 }
