 38                         case 1:
 39                             try {
 40                                 new TryCatch().throwException();
 41                                 break -1;
 42                             } catch(Throwable ex) {
 43                                 break val;
 44                             }
 45                         default: break -1;
 46                     };

 53                                         break -1;
 54                                     } catch(Throwable ex) {

 56                                     }

 58                             });
