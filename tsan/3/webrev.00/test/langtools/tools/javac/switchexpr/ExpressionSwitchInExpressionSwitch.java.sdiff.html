26  * @bug 8206986
27  * @summary Check switch expressions embedded in switch expressions.
28  * @compile --enable-preview -source ${jdk.version} ExpressionSwitchInExpressionSwitch.java
29  * @run main/othervm --enable-preview ExpressionSwitchInExpressionSwitch
30  */
31 
32 public class ExpressionSwitchInExpressionSwitch {
33     public static void main(String[] args) {
34 
35         int j = 42;
36         int i = switch (j) {
37             default -> (switch (j) { default -> 0; } )+1;
38         };
39         if (i!=1) {
40             throw new AssertionError("Unexpected result: " + i);
41         }
42         i = switch (j) {
43             default -> {
44                 int k = switch (j) {
45                     default -> {
46                         break 42;
47                     }
48                 };
49                 System.out.println("didn't break to the top level");
50                 break 43;
51             }
52         };
53         if (i!=43) {
54             throw new AssertionError("Unexpected result: " + i);
55         }
56     }
57 }
