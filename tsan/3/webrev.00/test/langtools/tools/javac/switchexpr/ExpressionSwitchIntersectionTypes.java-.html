24 /*
25  * @test
26  * @bug 8206986
27  * @summary Verify behavior when an intersection type is inferred for switch expression.
28  * @compile --enable-preview -source ${jdk.version} ExpressionSwitchIntersectionTypes.java
29  * @run main/othervm --enable-preview ExpressionSwitchIntersectionTypes
30  */
31 
32 public class ExpressionSwitchIntersectionTypes<X  extends java.io.Serializable & Runnable> {
33 
34     void test1(int i, X x) {
35         Runnable r1 = switch (i) {
36             default -> x;
37         };
38         r1.run();
39     }
40 
41     void test2(int i, X x) {
42         (switch (i) {
43             default -> x;
44         }).run();
45     }
46 
47     public static void main(String[] args) {
48         ExpressionSwitchIntersectionTypes t = new ExpressionSwitchIntersectionTypes();
49         try {
50             t.test1(0, "");
51             throw new AssertionError("Expected exception didn't occur.");
52         } catch (ClassCastException ex) {
53             //good
54         }
55         try {
56             t.test2(0, "");
57             throw new AssertionError("Expected exception didn't occur.");
58         } catch (ClassCastException ex) {
59             //good
60         }
61     }
62 
63 }
