 56     private String print1(int i, int j) {
 57         switch (i) {
 58             case 0:
 59                 return switch (j) {
 60                     case 0:
 61                         if (true) break "0-0";
 62                     case 1:
 63                         break "0-1";
 64                     default:
 65                         break "0-X";
 66                 };
 67             default: return "X";
 68         }

 71     private String print2(int i, int j, int k) {
 72         return switch (i) {
 73             case 0:
 74                 String r;
 75                 OUTER: switch (j) {
 76                     case 0:
 77                         String res;
 78                         INNER: switch (k) {
 79                             case 0: res = "0-0-0"; break;
 80                             case 1: res = "0-0-1"; break;
 81                             case 2: res = "0-0-2"; break INNER;
 82                             default: r = "0-0-X"; break OUTER;
 83                         }
 84                         r = res;
 85                         break;
 86                     case 1:
 87                         r = "0-1";
 88                         break;
 89                     default:
 90                         r = "0-X";
 91                         break;
 92                 }
 93                 break r;
 94             case 1:
 95                 break "1";
 96             case 2:
 97                 LOP: while (j-- > 0) {
 98                     if (k == 5) {
 99                         k--;
100                         continue;
101                     }
102                     break LOP;
103                 }
104                 Supplier<String> getter = () -> { return "2-X-5"; };
105                 break getter.get();
106             default:
107                 break "X";
108         };
109     }
