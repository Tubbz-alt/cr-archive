24 /*
25  * @test
26  * @bug 6942649 8225748
27  * @summary add hidden option to identify location and version of javac classes
28  * @modules jdk.compiler
29  */
30 
31 import java.io.*;
32 
33 public class T6942649 {
34     public static void main(String... args) throws Exception {
35         new T6942649().run();
36     }
37 
38     void run() throws Exception {
39         test("-XDshowClass", "com.sun.tools.javac.Main", false);
40         test("-XDshowClass=com.sun.tools.javac.util.Log", "com.sun.tools.javac.util.Log", false);
41     }
42 
43     void test(String opt, String clazz, boolean checkLocation) throws Exception {
44         System.err.println("test " + opt);
45         StringWriter sw = new StringWriter();
46         PrintWriter pw = new PrintWriter(sw);
47         int rc = com.sun.tools.javac.Main.compile(new String[] { opt }, pw);
48         pw.close();
49         String out = sw.toString();
50         System.err.println("javac rc=" + rc + "\n" + out);
51 
52         if (!out.contains(clazz))
53             throw new Exception("class name not found in output");
54 
55         // (JDK 9) We can no longer guarantee to be able to show a URL for the
56         // location of the class, especially when the class is in the system image
57         if (checkLocation) {
58             int lastDot = clazz.lastIndexOf(".");
59             if (!out.contains(clazz.substring(lastDot + 1) + ".class"))
60                 throw new Exception("location of class not found in output");
61         }
62 
63         if (!out.contains("SHA-256 checksum: "))
64             throw new Exception("checksum not found in output");
65     }
66 }
