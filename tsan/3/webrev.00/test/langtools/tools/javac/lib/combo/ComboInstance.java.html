 24 package combo;
 25 
 26 import javax.tools.StandardJavaFileManager;
 27 import java.util.Optional;
 28 
 29 import com.sun.tools.javac.file.BaseFileManager;
 30 
 31 /**
 32  * This class is the common superclass of all combo test instances. It defines few helper methods
 33  * to build compilation tasks using the shared context object, as well as entry points for
 34  * signalling test failures.
 35  */
 36 public abstract class ComboInstance<X extends ComboInstance<X>> {
 37 
 38     /** The test instance result status. */
 39     private ResultStatus resultStatus = ResultStatus.PASSED;
 40 
 41     /** The test instance execution environment. */
 42     private ComboTestHelper<X>.Env env;
 43 
 44     /**
 45      * Entry point for executing a combo test instance; first, the test environment is saved
 46      * in the corresponding field, then the instance is run (see {@link ComboInstance#doWork()}.
 47      * During execution, the result status will be updated to match the test outcome.
 48      */
 49     final void run(ComboTestHelper<X>.Env env) {
 50         try {
 51             this.env = env;
 52             doWork();
 53             if (resultStatus.isSuccess()) {
 54                 env.info().passCount++;
 55             }
 56         } catch (Throwable ex) {
 57             resultStatus = ResultStatus.ERROR;
 58             env.info().errCount++;
 59             env.info().lastError = Optional.of(ex);
 60         } finally {
 61             this.env = null;
 62             ((BaseFileManager) env.fileManager()).clear();
 63         }
 64     }
 65 
 66     /**
 67      * Retrieve a unique ID associated with this test instance.
 68      */
 69     public int id() {
 70         return env.info().comboCount;
 71     }
 72 
 73     /**
 74      * Retrieve shared file manager.
 75      */
 76     public StandardJavaFileManager fileManager() {
 77         return env.fileManager();
 78     }
 79 
 80     /**
 81      * Create a new compilation task using shared compilation context.
 82      */
 83     protected ComboTask newCompilationTask() {
 84         return new ComboTask(env);
 85     }
 86 
 87     /**
 88      * Main test execution entry point; subclasses must implement this method to define the test
 89      * logic.
 90      */
 91     protected abstract void doWork() throws Throwable;
 92 
 93     /**
 94      * Report a test failure.
 95      */
 96     protected void fail() {
 97         //dump some default info (such as dimension bindings)
 98         fail("Combo instance failed; " + env.bindings);
 99     }
100 
101     /**
102      * Report a test failure with corresponding failure message.
103      */
104     protected void fail(String msg) {
105         resultStatus = ResultStatus.FAILED;
106         env.info().failCount++;
107         env.info().lastFailure = Optional.of(msg);
108     }
109 
110     /**
111      * The status associated with this test instance execution.
112      */
113     enum ResultStatus {
114         /** Test passed. */
115         PASSED(true),
116         /** Test failed. */
117         FAILED(false),
118         /** Test thrown unexpected error/exception. */
119         ERROR(false);
120 
121         boolean success;
122 
123         ResultStatus(boolean success) {
124             this.success = success;
125         }
126 
127         boolean isSuccess() {
128             return success;
129         }
130     }
131 }
