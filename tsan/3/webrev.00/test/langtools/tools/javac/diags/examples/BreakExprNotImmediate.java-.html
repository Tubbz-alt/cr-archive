39 class BreakExprNotImmediate {
40     int t(int i) {
41         return switch (i) {
42             case 0:
43                 for (; ;) {
44                     break 1 + 1;
45                 }
46             case 1:
47                 for (String s : new String[0]) {
48                     break 1 + 1;
49                 }
50             case 2:
51                 while (true) {
52                     break 1 + 1;
53                 }
54             case 3:
55                 do {
56                     break 1 + 1;
57                 } while (true);
58             case 4:
59                 switch (i) {
60                     default: break 1 + 1;
61                 }
62                 break 0;
63         };
64     }
65 }
