24 /*
25  * @test
26  * @bug 8227010
27  * @summary Verify error recovery after variable redeclaration
28  * @compile/fail/ref=LocalVarHiding.out -XDrawDiagnostics LocalVarHiding.java
29  */
30 
31 public class LocalVarHiding {
32     public void test(Decl l1) {
33         System.err.println(l1.originalMethod());
34 
35         OverrideVar l1 = new OverrideVar();
36 
37         System.err.println(l1.overrideMethod());
38 
39         Decl l2 = new Decl();
40 
41         System.err.println(l2.originalMethod());
42 
43         OverrideVar l2 = new OverrideVar();
44 
45         System.err.println(l2.overrideMethod());
46 
47         Decl l3 = new Decl();
48 
49         System.err.println(l3.originalMethod());
50 
51         {
52             OverrideVar l3 = new OverrideVar();
53 
54             System.err.println(l3.overrideMethod());
55         }
56 
57         Decl l4 = new Decl();
58 
59         System.err.println(l4.originalMethod());
60 
61         try {
62             throw new OverrideVar();
63         } catch (OverrideVar l4) {
64             System.err.println(l4.overrideMethod());
65         }
66 
67         Decl l5 = new Decl();
68 
69         System.err.println(l5.originalMethod());
70 
71         try (OverrideVar l5 = null) {
72             System.err.println(l5.overrideMethod());
73         }
74 
75         Decl l6 = new Decl();
76 
77         System.err.println(l6.originalMethod());
78 
79         I i1 = l6 -> {
80             System.err.println(l6.overrideMethod());
81         };
82         I i2 = (OverrideVar l6) -> {
83             System.err.println(l6.overrideMethod());
84         };
85     }
86 
87     public class Decl {
88         public int originalMethod() {}
89     }
90 
91     public class OverrideVar extends Exception implements AutoCloseable {
92         @Override public void close() throws Exception {}
93         public int overrideMethod() {}
94     }
95 
96     public interface I {
97         public void run(OverrideVar ov);
98     }
99 }
